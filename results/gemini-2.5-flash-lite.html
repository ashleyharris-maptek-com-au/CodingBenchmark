            return self.w * self.h

                    best_free_rect_idx = i

import numpy as np

def rotate_point(point, angle, origin=(0, 0)):
    """Rotates a point around an origin."""
    ox, oy = origin
    px, py = point

    qx = ox + math.cos(angle) * (px - ox) - math.sin(angle) * (py - oy)
    qy = oy + math.sin(angle) * (py - oy) + math.cos(angle) * (px - ox)
    return qx, qy

def translate_point(point, translation):
    """Translates a point."""
    return point[0] + translation[0], point[1] + translation[1]

def transform_polygon(polygon, translation, rotation):
    """Applies translation and rotation to a polygon's vertices."""
    rotated_polygon = [rotate_point(p, rotation) for p in polygon]
    transformed_polygon = [translate_point(p, translation) for p in rotated_polygon]
    return transformed_polygon

def is_point_in_polygon(point, polygon):
    """Checks if a point is inside a polygon using the ray casting algorithm."""
    x, y = point
    n = len(polygon)
    inside = False

    p1x, p1y = polygon[0]
    for i in range(n + 1):
        p2x, p2y = polygon[i % n]
        if y > min(p1y, p2y):
            if y <= max(p1y, p2y):
                if x <= max(p1x, p2x):
                    if p1y != p2y:
                        xinters = (y - p1y) * (p2x - p1x) / (p2y - p1y) + p1x
                    if p1x == p2x or x <= xinters:
                        inside = not inside
        p1x, p1y = p2x, p2y

    return inside

def do_segments_intersect(p1, q1, p2, q2):
    """Checks if two line segments intersect."""
    def orientation(p, q, r):
        val = (q[1] - p[1]) * (r[0] - q[0]) - (q[0] - p[0]) * (r[1] - q[1])
        if val == 0: return 0  # Collinear
        return 1 if val > 0 else 2 # Clockwise or Counterclockwise

    def on_segment(p, q, r):
        return (q[0] <= max(p[0], r[0]) and q[0] >= min(p[0], r[0]) and
                q[1] <= max(p[1], r[1]) and q[1] >= min(p[1], r[1]))

    o1 = orientation(p1, q1, p2)
    o2 = orientation(p1, q1, q2)
    o3 = orientation(p2, q2, p1)
    o4 = orientation(p2, q2, q1)

    if o1 != o2 and o3 != o4:
        return True

    if o1 == 0 and on_segment(p1, p2, q1): return True
    if o2 == 0 and on_segment(p1, q2, q1): return True
    if o3 == 0 and on_segment(p2, p1, q2): return True
    if o4 == 0 and on_segment(p2, q1, q2): return True

    return False

def polygon_intersects(poly1, poly2):
    """Checks if two polygons intersect."""
    # Check if any vertex of poly1 is inside poly2
    for p in poly1:
        if is_point_in_polygon(p, poly2):
            return True
    # Check if any vertex of poly2 is inside poly1
    for p in poly2:
        if is_point_in_polygon(p, poly1):
            return True

    # Check for edge intersections
    n1, n2 = len(poly1), len(poly2)

            p2 = poly2[j]
            q2 = poly2[(j + 1) % n2]
            if do_segments_intersect(p1, q1, p2, q2):
                return True
    return False

def get_polygon_bounding_box(polygon):
    """Calculates the bounding box of a polygon."""
    min_x = min(p[0] for p in polygon)
    max_x = max(p[0] for p in polygon)
    min_y = min(p[1] for p in polygon)
    max_y = max(p[1] for p in polygon)
    return min_x, min_y, max_x, max_y

def bounding_boxes_overlap(bb1, bb2):

    return not (bb1[2] < bb2[0] or bb1[0] > bb2[2] or bb1[3] < bb2[1] or bb1[1] > bb2[3])

def rotate_point(point, angle, origin=(0, 0)):
    """Rotates a point around an origin."""
    ox, oy = origin
    px, py = point

    qx = ox + math.cos(angle) * (px - ox) - math.sin(angle) * (py - oy)
    qy = oy + math.sin(angle) * (py - oy) + math.cos(angle) * (px - ox)
    return qx, qy

def translate_point(point, translation):
    """Translates a point."""
    return point[0] + translation[0], point[1] + translation[1]

def transform_polygon(polygon, translation, rotation):
    """Applies translation and rotation to a polygon's vertices."""
    rotated_polygon = [rotate_point(p, rotation) for p in polygon]
    transformed_polygon = [translate_point(p, translation) for p in rotated_polygon]
    return transformed_polygon

def is_point_in_polygon(point, polygon):
    """Checks if a point is inside a polygon using the ray casting algorithm."""
    x, y = point
    n = len(polygon)
    inside = False

    p1x, p1y = polygon[0]
    for i in range(n + 1):
        p2x, p2y = polygon[i % n]
        if y > min(p1y, p2y):
            if y <= max(p1y, p2y):
                if x <= max(p1x, p2x):
                    if p1y != p2y:
                        xinters = (y - p1y) * (p2x - p1x) / (p2y - p1y) + p1x
                    if p1x == p2x or x <= xinters:
                        inside = not inside
        p1x, p1y = p2x, p2y

    return inside

def do_segments_intersect(p1, q1, p2, q2):
    """Checks if two line segments intersect."""
    def orientation(p, q, r):
        val = (q[1] - p[1]) * (r[0] - q[0]) - (q[0] - p[0]) * (r[1] - q[1])
        if val == 0: return 0  # Collinear
        return 1 if val > 0 else 2 # Clockwise or Counterclockwise

    def on_segment(p, q, r):
        return (q[0] <= max(p[0], r[0]) and q[0] >= min(p[0], r[0]) and
                q[1] <= max(p[1], r[1]) and q[1] >= min(p[1], r[1]))

    o1 = orientation(p1, q1, p2)
    o2 = orientation(p1, q1, q2)
    o3 = orientation(p2, q2, p1)
    o4 = orientation(p2, q2, q1)

    if o1 != o2 and o3 != o4:
        return True

    if o1 == 0 and on_segment(p1, p2, q1): return True
    if o2 == 0 and on_segment(p1, q2, q1): return True
    if o3 == 0 and on_segment(p2, p1, q2): return True
    if o4 == 0 and on_segment(p2, q1, q2): return True

    return False

def polygon_intersects(poly1, poly2):
    """Checks if two polygons intersect."""
    # Check if any vertex of poly1 is inside poly2
    for p in poly1:
        if is_point_in_polygon(p, poly2):
            return True
    # Check if any vertex of poly2 is inside poly1
    for p in poly2:
        if is_point_in_polygon(p, poly1):
            return True

    # Check for edge intersections
    n1, n2 = len(poly1), len(poly2)
    for i in range(n1):
        p1 = poly1[i]
        q1 = poly1[(i + 1) % n1]
        for j in range(n2):
            p2 = poly2[j]
            q2 = poly2[(j + 1) % n2]
            if do_segments_intersect(p1, q1, p2, q2):
                return True
    return False

def is_polygon_contained(inner_poly, outer_poly):
    """Checks if inner_poly is completely inside outer_poly."""
    # All vertices of inner_poly must be inside outer_poly
    for p in inner_poly:
        if not is_point_in_polygon(p, outer_poly):
            return False
    # Additionally, check for edge intersections to handle cases where inner polygon might 'poke out'
    # due to concavities of outer polygon not captured by vertex check alone.
    # However, a simpler check is often sufficient for packing if we ensure no overlaps.
    # For strict containment, a more robust approach involving checking if any edge of inner_poly
    # intersects an edge of outer_poly *externally* or if an inner vertex is outside would be needed.
    # For this problem, given we check for overlaps, vertex containment is a strong indicator.
    return True

def get_polygon_bounding_box(polygon):
    """Calculates the bounding box of a polygon."""
    min_x = min(p[0] for p in polygon)
    max_x = max(p[0] for p in polygon)
    min_y = min(p[1] for p in polygon)
    max_y = max(p[1] for p in polygon)
    return min_x, min_y, max_x, max_y
