Looking at this problem, I need to implement a CSG Union that:
1. Handles any mesh size (6 vertices to millions)
2. Completes within 5 minutes on a 64-core system
3. Produces correct, watertight, manifold output

I'll use a **BVH-based approach with ray-casting for inside/outside classification**, which provides excellent scalability.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <memory>
#include <mutex>
#include <random>
#include <set>

using namespace std;

const double EPSILON = 1e-12;
const double COINCIDENT_EPS = 1e-9;
const int MAX_BVH_DEPTH = 24;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 cross(const Vec3& o) const { return Vec3(y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x); }
    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { double l = length(); return l > 0 ? Vec3(x/l, y/l, z/l) : Vec3(); }
};

struct Triangle { int v0, v1, v2; };

struct BBox {
    Vec3 min_pt, max_pt;
    BBox() : min_pt(Vec3(1e100, 1e100, 1e100)), max_pt(Vec3(-1e100, -1e100, -1e100)) {}
    void include(const Vec3& p) {
        min_pt.x = min(min_pt.x, p.x); min_pt.y = min(min_pt.y, p.y); min_pt.z = min(min_pt.z, p.z);
        max_pt.x = max(max_pt.x, p.x); max_pt.y = max(max_pt.y, p.y); max_pt.z = max(max_pt.z, p.z);
    }
    bool intersects(const BBox& o) const {
        return !(max_pt.x < o.min_pt.x || min_pt.x > o.max_pt.x || max_pt.y < o.min_pt.y || min_pt.y > o.max_pt.y || max_pt.z < o.min_pt.z || min_pt.z > o.max_pt.z);
    }
    bool empty() const { return min_pt.x > max_pt.x + 1e50; }
};

struct Node {
    BBox bbox;
    vector<int> tris;
    unique_ptr<Node> left, right;
    bool is_leaf() const { return !left && !right; }
};

struct Mesh {
    vector<Vec3> verts;
    vector<Triangle> tris;
    BBox bbox;
    unique_ptr<Node> bvh;
    void build_bvh() {
        bvh = unique_ptr<Node>(new Node());
        for (size_t i = 0; i < tris.size(); ++i) {
            bvh->bbox.include(verts[tris[i].v0]);
            bvh->bbox.include(verts[tris[i].v1]);
            bvh->bbox.include(verts[tris[i].v2]);
            bvh->tris.push_back((int)i);
        }
        if (!bvh->tris.empty() && bvh->bbox.size().length() > EPSILON)
            build_bvh_recursive(bvh.get(), 0, (int)tris.size());
    }
    void build_bvh_recursive(Node* node, int start, int end) {
        if (end - start <= 3 || node->bbox.size().length() < EPSILON || bvh->tris.size() <= 8) return;
        Vec3 sz = node->bbox.size();
        int axis = sz.x > sz.y ? (sz.x > sz.z ? 0 : 2) : (sz.y > sz.z ? 1 : 2);
        double mid = node->bbox.min_pt + node->bbox.size() * 0.5;
        int split = start;
        if (axis == 0) { for (int i = start; i < end; ++i) { const auto& t = tris[i]; double c = (verts[t.v0].x + verts[t.v1].x + verts[t.v2].x) / 3.0; if (c >= mid) swap(tris[i], tris[split++]); } }
        else if (axis == 1) { for (int i = start; i < end; ++i) { const auto& t = tris[i]; double c = (verts[t.v0].y + verts[t.v1].y + verts[t.v2].y) / 3.0; if (c >= mid) swap(tris[i], tris[split++]); } }
        else { for (int i = start; i < end; ++i) { const auto& t = tris[i]; double c = (verts[t.v0].z + verts[t.v1].z + verts[t.v2].z) / 3.0; if (c >= mid) swap(tris[i], tris[split++]); } }
        node->left = unique_ptr<Node>(new Node());
        node->right = unique_ptr<Node>(new Node());
        for (int i = start; i < split; ++i) { node->left->bbox.include(verts[tris[i].v0]); node->left->bbox.include(verts[tris[i].v1]); node->left->bbox.include(verts[tris[i].v2]); node->left->tris.push_back(tris[i]); }
        for (int i = split; i < end; ++i) { node->right->bbox.include(verts[tris[i].v0]); node->right->bbox.include(verts[tris[i].v1]); node->right->bbox.include(verts[tris[i].v2]); node->right->tris.push_back(tris[i]); }
        build_bvh_recursive(node->left.get(), 0, (int)node->left->tris.size());
        build_bvh_recursive(node->right.get(), 0, (int)node->right->tris.size());
        node->bbox.include(node->left->bbox); node->bbox.include(node->right->bbox);
    }
    bool intersect_bvh(const Node* node, const Vec3& orig, const Vec3& dir, double& t) const {
        if (!node || !node->bbox.intersects(BBox{orig, orig + dir})) return false;
        if (node->is_leaf()) {
            for (int idx : node->tris) {
                const auto& t = tris[idx];
                const auto& v0 = verts[t.v0], v1 = verts[t.v1], v2 = verts[t.v2];
                Vec3 edge1 = v1 - v0, edge2 = v2 - v0, h = dir.cross(edge1);
                double a = edge2.dot(h);
                if (abs(a) < EPSILON) continue;
                double f = 1.0 / a;
                Vec3 s = orig - v0;
                double u = f * s.dot(h);
                if (u < 0 || u > 1) continue;
                Vec3 q = s.cross(edge2);
                double v = f * dir.dot(q);
                if (v < 0 || u + v > 1) continue;
                double t0 = f * edge1.dot(q);
                if (t0 > EPSILON && t0 < t) { t = t0; return true; }
            }
            return false;
        }
        double t1, t2;
        bool hit = intersect_bvh(node->left.get(), orig, dir, t1);
        if (hit) t = t1;
        return hit || intersect_bvh(node->right.get(), orig, dir, t2);
    }
};

bool ray_triangle_intersect(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    Vec3 edge1 = v1 - v0, edge2 = v2 - v0, h = dir.cross(edge1);
    double a = edge2.dot(h);
    if (abs(a) < EPSILON) return false;
    Vec3 s = orig - v0;
    double f = 1.0 / a;
    double u = f * s.dot(h);
    if (u < 0 || u > 1) return false;
    Vec3 q = s.cross(edge2);
    double v = f * dir.dot(q);
    if (v < 0 || u + v > 1) return false;
    t = f * edge1.dot(q);
    return t > EPSILON;
}

bool point_inside(const Mesh& mesh, const Vec3& p) {
    Vec3 dir(0, 0, 1);
    int intersections = 0;
    double t;
    for (const auto& tri : mesh.tris) {
        if (ray_triangle_intersect(p, dir, mesh.verts[tri.v0], mesh.verts[tri.v1], mesh.verts[tri.v2], t)) intersections++;
    }
    return intersections % 2 == 1;
}

Vec3 compute_centroid(const Mesh& mesh, const Triangle& t) {
    return (mesh.verts[t.v0] + mesh.verts[t.v1] + mesh.verts[t.v2]) * (1.0/3.0);
}

Vec3 ray_segment_intersect(const Vec3& ro, const Vec3& rd, const Vec3& a, const Vec3& b, double& t, Vec3& isect_pt) {
    Vec3 ab = b - a;
    Vec3 ao = ro - a;
    double h = rd.cross(ab).dot(ao.cross(rd));
    if (abs(h) < EPSILON) return Vec3();
    double r = 1.0 / (rd.cross(ab).length2());
    double t1 = (ab.cross(ao)).dot(rd) * r;
    if (t1 < 0 || t1 > 1) return Vec3();
    double t2 = ao.dot(rd.cross(ab)) * r;
    if (t2 < 0) return Vec3();
    t = t2; isect_pt = ro + rd * t; return ab;
}

vector<pair<int,int>> find_intersections(const Mesh& a, const Mesh& b) {
    vector<pair<int,int>> result;
    vector<bool> a_processed(a.tris.size(), false);
    vector<bool> b_processed(b.tris.size(), false);
    vector<Vec3> a_points, b_points;
    vector<pair<int,int>> edges_a, edges_b;
    
    for (size_t ia = 0; ia < a.tris.size(); ++ia) if (!a_processed[ia]) {
        const Triangle& ta = a.tris[ia];
        const Vec3 ca = compute_centroid(a, ta);
        if (!a.bvh->bbox.intersects(BBox{ca, ca})) continue;
        
        a_processed[ia] = true;
        double t; Vec3 isect_pt, dir;
        bool intersect = a.intersect_bvh(a.bvh.get(), ca, Vec3(1,0,0), t);
        if (intersect) { dir = (a.bvh->bbox.max_pt - a.bvh->bbox.min_pt).normalize(); intersect = a.intersect_bvh(a.bvh.get(), ca + dir*0.001, dir, t); }
        
        if (!intersect) continue;
        
        for (size_t ib = 0; ib < b.tris.size(); ++ib) if (!b_processed[ib]) {
            const Triangle& tb = b.tris[ib];
            const Vec3 cb = compute_centroid(b, tb);
            if (!a.bvh->bbox.intersects(BBox{cb, cb})) continue;
            
            // Check intersection with triangle tb
            const Vec3& v0 = b.verts[tb.v0], &v1 = b.verts[tb.v1], &v2 = b.verts[tb.v2];
            if (ray_triangle_intersect(ca, dir, v0, v1, v2, t)) {
                b_processed[ib] = true;
                result.emplace_back((int)ia, (int)ib);
                a_points.push_back(ca + dir * t);
                edges_a.emplace_back(ta.v0, ta.v1); edges_a.emplace_back(ta.v1, ta.v2); edges_a.emplace_back(ta.v2, ta.v0);
                edges_b.emplace_back(tb.v0, tb.v1); edges_b.emplace_back(tb.v1, tb.v2); edges_b.emplace_back(tb.v2, tb.v0);
            }
        }
    }
    return result;
}

Mesh compute_union(const Mesh& A, const Mesh& B) {
    vector<Vec3> new_verts = A.verts;
    vector<Triangle> new_tris = A.tris;
    vector<int> vmap(A.verts.size(), -1);
    for (size_t i = 0; i < A.verts.size(); ++i) vmap[i] = (int)new_verts.size(), new_verts.push_back(A.verts[i]);
    
    vector<bool> a_inside(A.tris.size(), false), b_inside(B.tris.size(), false);
    vector<double> a_dist(A.tris.size()), b_dist(B.tris.size());
    
    for (size_t i = 0; i < A.tris.size(); ++i) {
        Vec3 c = compute_centroid(A, A.tris[i]);
        a_dist[i] = point_inside(B, c) ? -1 : 1;
        a_inside[i] = point_inside(B, c);
    }
    
    for (size_t i = 0; i < B.tris.size(); ++i) {
        Vec3 c = compute_centroid(B, B.tris[i]);
        b_dist[i] = point_inside(A, c) ? -1 : 1;
        b_inside[i] = point_inside(A, c);
    }
    
    auto keep_tri = [&](int i) {
        return a_inside[i] && !b_inside[i];
    };
    
    auto discard_tri = [&](int i) {
        return b_inside[i] && !a_inside[i];
    };
    
    vector<Triangle> result_tris;
    vector<bool> a_to_keep(A.tris.size()), b_to_keep(B.tris.size());
    
    for (size_t i = 0; i < A.tris.size(); ++i) a_to_keep[i] = !a_inside[i];
    for (size_t i = 0; i < B.tris.size(); ++i) b_to_keep[i] = !b_inside[i];
    
    for (size_t i = 0; i < A.tris.size(); ++i) if (a_to_keep[i]) result_tris.push_back(A.tris[i]);
    for (size_t i = 0; i < B.tris.size(); ++i) if (b_to_keep[i]) {
        result_tris.emplace_back(vmap[B.tris[i].v0], vmap[B.tris[i].v1], vmap[B.tris[i].v2]);
    }
    
    Mesh result; result.verts = new_verts; result.tris = result_tris; result.build_bvh();
    return result;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    
    size_t VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    
    Mesh A, B;
    for (size_t i = 0; i < VA; ++i) { double x, y, z; cin >> x >> y >> z; A.verts.emplace_back(x, y, z); }
    for (size_t i = 0; i < FA; ++i) { int a, b, c; cin >> a >> b >> c; A.tris.emplace_back((int)a, (int)b, (int)c); }
    for (size_t i = 0; i < VB; ++i) { double x, y, z; cin >> x >> y >> z; B.verts.emplace_back(x, y, z); }
    for (size_t i = 0; i < FB; ++i) { int a, b, c; cin >> a >> b >> c; B.tris.emplace_back((int)a, (int)b, (int)c); }
    
    A.build_bvh(); B.build_bvh();
    Mesh result = compute_union(A, B);
    
    cout << result.verts.size() << " " << result.tris.size() << "\n";
    for (const auto& v : result.verts) cout << v.x << " " << v.y << " " << v.z << "\n";
    for (const auto& t : result.tris) cout << t.v0 << " " << t.v1 << " " << t.v2 << "\n";
    
    return 0;
}
```
