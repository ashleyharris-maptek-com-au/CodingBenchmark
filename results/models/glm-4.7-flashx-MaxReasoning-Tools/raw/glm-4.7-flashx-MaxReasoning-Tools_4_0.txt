// Using vertices: (0.5,0.5,0.5), (0.5,-0.5,-0.5), (-0.5,0.5,-0.5), (-0.5,-0.5,0.5)
static const double VERTICES[4][3] = {
    {0.5, 0.5, 0.5},
    {0.5, -0.5, -0.5},
    {-0.5, 0.5, -0.5},
    {-0.5, -0.5, 0.5}

// Rotation matrices for Euler angles (rx, ry, rz in radians)
void rotatePoint(double x, double y, double z, double rx, double ry, double rz,
                 double& outx, double& outy, double& outz) {
    // Apply rotation around Z axis
    double x1 = x * cos(rz) - y * sin(rz);
    double y1 = x * sin(rz) + y * cos(rz);
    double z1 = z;
    
    // Apply rotation around Y axis
    double x2 = x1 * cos(ry) + z1 * sin(ry);
    double y2 = y1;
    double z2 = -x1 * sin(ry) + z1 * cos(ry);
    
    // Apply rotation around X axis
    outx = x2;
    outy = y2 * cos(rx) - z2 * sin(rx);
    outz = y2 * sin(rx) + z2 * cos(rx);
}

// Axis-Aligned Bounding Box
struct AABB {
    double minx, maxx, miny, maxy, minz, maxz;
    
    AABB() : minx(1e100), maxx(-1e100), miny(1e100), maxy(-1e100), minz(1e100), maxz(-1e100) {}
    
    void addPoint(double x, double y, double z) {
        minx = std::min(minx, x);
        maxx = std::max(maxx, x);
        miny = std::min(miny, y);
        maxy = std::max(maxy, y);
        minz = std::min(minz, z);
        maxz = std::max(maxz, z);
    }
    
    bool intersects(const AABB& other) const {
        return !(maxx < other.minx || minx > other.maxx ||
                 maxy < other.miny || miny > other.maxy ||
                 maxz < other.minz || minz > other.maxz);
    }
};

// Spatial hash for fast collision detection
class SpatialHash {
private:
    double cellSize;
    std::vector<std::vector<size_t>> grid;
    std::vector<AABB> aabbs;
    
public:
    SpatialHash(double cs, size_t gridWidth, size_t gridHeight, size_t gridDepth)
        : cellSize(cs) {
        grid.resize(gridWidth * gridHeight * gridDepth);
    }
    
    uint64_t getCellKey(double x, double y, double z) const {
        int ix = static_cast<int>(std::floor(x / cellSize));
        int iy = static_cast<int>(std::floor(y / cellSize));
        int iz = static_cast<int>(std::floor(z / cellSize));
        return static_cast<uint64_t>(ix) | 
               (static_cast<uint64_t>(iy) << 32) | 
               (static_cast<uint64_t>(iz) << 48);
    }
    
    void insert(const AABB& aabb) {
        size_t index = aabbs.size();
        aabbs.push_back(aabb);
        
        double mincx = std::floor(aabb.minx / cellSize);
        double maxcx = std::floor(aabb.maxx / cellSize);
        double mincy = std::floor(aabb.miny / cellSize);
        double maxcy = std::floor(aabb.maxy / cellSize);
        double mincz = std::floor(aabb.minz / cellSize);
        double maxcz = std::floor(aabb.maxz / cellSize);
        
        for (int ix = static_cast<int>(mincx); ix <= static_cast<int>(maxcx); ++ix) {
            for (int iy = static_cast<int>(mincy); iy <= static_cast<int>(maxcy); ++iy) {
                for (int iz = static_cast<int>(mincz); iz <= static_cast<int>(maxcz); ++iz) {
                    size_t key = static_cast<size_t>(ix) + 
                                 static_cast<size_t>(iy) * 1024 + 
                                 static_cast<size_t>(iz) * 1024 * 1024;
                    grid[key].push_back(index);
                }
            }
        }
    }
    
    bool overlaps(const AABB& aabb) const {
        double mincx = std::floor(aabb.minx / cellSize);
        double maxcx = std::floor(aabb.maxx / cellSize);
        double mincy = std::floor(aabb.miny / cellSize);
        double maxcy = std::floor(aabb.maxy / cellSize);
        double mincz = std::floor(aabb.minz / cellSize);
        double maxcz = std::floor(aabb.maxz / cellSize);
        
        for (int ix = static_cast<int>(mincx); ix <= static_cast<int>(maxcx); ++ix) {
            for (int iy = static_cast<int>(mincy); iy <= static_cast<int>(maxcy); ++iy) {
                for (int iz = static_cast<int>(mincz); iz <= static_cast<int>(maxcz); ++iz) {
                    size_t key = static_cast<size_t>(ix) + 
                                 static_cast<size_t>(iy) * 1024 + 
                                 static_cast<size_t>(iz) * 1024 * 1024;
                    
                    for (size_t idx : grid[key]) {
                        if (aabbs[idx].intersects(aabb)) {
                            return true;
                        }
                    }
                }
            }
        }
        return false;
    }
};

// Tetrahedron
struct Tetrahedron {
    double cx, cy, cz;
    double rx, ry, rz;
    AABB aabb;
    
    void computeAABB() {
        aabb = AABB();
        for (int i = 0; i < 4; ++i) {
            double x, y, z;
            rotatePoint(VERTICES[i][0], VERTICES[i][1], VERTICES[i][2],
                       rx, ry, rz, x, y, z);
            aabb.addPoint(cx + x, cy + y, cz + z);
        }
    }
};

// Project tetrahedron to its bounding box dimensions
void projectToAABB(double cx, double cy, double cz, double rx, double ry, double rz,
                   double& halfW, double& halfH, double& halfD) {
    double minX = 1e100, maxX = -1e100, minY = 1e100, maxY = -1e100, 
           minZ = 1e100, maxZ = -1e100;
    
    for (int i = 0; i < 4; ++i) {
        double x, y, z;
        rotatePoint(VERTICES[i][0], VERTICES[i][1], VERTICES[i][2],
                   rx, ry, rz, x, y, z);
        double px = cx + x, py = cy + y, pz = cz + z;
        minX = std::min(minX, px);
        maxX = std::max(maxX, px);
        minY = std::min(minY, py);
        maxY = std::max(maxY, py);
        minZ = std::min(minZ, pz);
        maxZ = std::max(maxZ, pz);
    }
    
    halfW = (maxX - minX) / 2.0;
    halfH = (maxY - minY) / 2.0;
    halfD = (maxZ - minZ) / 2.0;
}
