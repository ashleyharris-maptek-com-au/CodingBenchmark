```glsl
#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Light direction from fragment to light
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Diffuse term (Lambertian)
    float diffuse = max(dot(N, L), 0.0);
    
    // Checkerboard pattern at 8x8 frequency
    vec2 uv8 = uv * 8.0;
    vec2 cell = floor(uv8);
    float checker = mod(cell.x + cell.y, 2.0);
    
    // Checker colors alternating 0.2 and 0.9 by parity
    vec3 checkerColor = mix(vec3(0.2), vec3(0.9), checker);
    
    // Apply Lambertian shading factor: (0.5 + 0.5 * diffuse)
    vec3 finalColor = checkerColor * (0.5 + 0.5 * diffuse);
    
    // Output with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
