Looking at this TSP challenge, I need an adaptive approach that:
1. Uses exact algorithms for tiny instances (N ≤ 10)
2. Uses heuristics for medium instances (N ≤ 1000)
3. Uses optimized heuristics for large instances (N > 1000)

Let me design and implement a complete solution:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <limits>
#include <random>

using namespace std;

typedef long long ll;
typedef double ld;

struct City {
    ld x, y;
    City(ld x = 0, ld y = 0) : x(x), y(y) {}
};

static inline ld dist2(const City& a, const City& b) {
    ld dx = a.x - b.x;
    ld dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline vector<int> nearest_neighbor(const vector<City>& cities, int start = 0) {
    int n = cities.size();
    vector<int> order(n);
    vector<char> visited(n, 0);
    
    order[0] = start;
    visited[start] = 1;
    int current = start;
    
    for (int i = 1; i < n; ++i) {
        int best = -1;
        ld bestd2 = numeric_limits<ld>::infinity();
        
        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                ld d = dist2(cities[current], cities[j]);
                if (d < bestd2) {
                    bestd2 = d;
                    best = j;
                }
            }
        }
        visited[best] = 1;
        order[i] = best;
        current = best;
    }
    
    return order;
}

static inline void two_opt_random(vector<int>& order, const vector<City>& cities, int iters) {
    int n = order.size();
    if (n < 3) return;
    
    random_device rd;
    mt19937 gen(rd());
    uniform_int_distribution<int> rng(0, n - 1);
    
    for (int t = 0; t < iters; ++t) {
        int i = rng(gen);
        int j = rng(gen);
        
        if (i > j) swap(i, j);
        if (i + 1 >= j) continue;
        
        ld oldd = dist2(cities[order[i]], cities[order[i + 1]]) +
                 dist2(cities[order[j]], cities[order[(j + 1) % n]]);
        ld newd = dist2(cities[order[i]], cities[order[j]]) +
                 dist2(cities[order[i + 1]], cities[order[(j + 1) % n]]);
        
        if (newd < oldd) {
            int l = i + 1, r = j;
            while (l < r) swap(order[l++], order[r--]);
        }
    }
}

static inline vector<int> exact_brute(const vector<City>& cities) {
    int n = cities.size();
    if (n <= 2) return {0, 1};
    
    vector<int> best(n), perm(n - 1);
    for (int i = 0; i < n - 1; ++i) perm[i] = i + 1;
    
    ld bestd = numeric_limits<ld>::infinity();
    do {
        for (int i = 0; i < n - 1; ++i) best[i + 1] = perm[i];
        ld d = 0;
        for (int i = 0; i < n; ++i)
            d += dist2(cities[best[i]], cities[best[(i + 1) % n]]);
        if (d < bestd) bestd = d;
    } while (next_permutation(perm.begin(), perm.end()));
    
    best[0] = 0;
    for (int i = 0; i < n - 1; ++i) best[i + 1] = perm[i];
    return best;
}

static inline vector<int> solve_tsp(const vector<City>& cities) {
    int n = cities.size();
    if (n <= 1) return {0};
    if (n == 2) return {0, 1};
    
    if (n <= 10) return exact_brute(cities);
    
    auto tour = nearest_neighbor(cities);
    
    if (n <= 100) {
        two_opt_random(tour, cities, 5000);
    } else if (n <= 1000) {
        two_opt_random(tour, cities, 500);
    } else if (n <= 10000) {
        two_opt_random(tour, cities, 100);
    } else {
        two_opt_random(tour, cities, 10);
    }
    
    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        ld x, y;
        cin >> x >> y;
        cities[i] = City(x, y);
    }
    
    auto tour = solve_tsp(cities);
    
    for (size_t i = 0; i < tour.size(); ++i) {
        cout << tour[i];
        if (i + 1 < tour.size()) cout << " ";
    }
    cout << "\n";
    
    return 0;
}
```

**Adaptive Strategy Summary:**

| Scale | N Range | Algorithm | Why |
|-------|---------|-----------|-----|
| Trivial | 10 | Exact brute force | 9! = 362,880 permutations feasible |
| Small | 10-100 | Nearest neighbor + 5000 2-opt iterations | Good quality heuristic |
| Medium | 100-1000 | NN + 500 2-opt iterations | Handles 10⁵-10⁶ distance checks |
| Large | 1000-10000 | NN + 100 2-opt iterations | Efficient optimization |
| Extreme | 10000+ | NN + 10 2-opt iterations | Fast enough for millions |

**Key optimizations:**
- **Squared distances**: Avoids expensive `sqrt()` calls
- **Single pass 2-opt**: Random pair selection (probabilistic improvement) instead of exhaustive search
- **Memory efficient**: Uses `vector<char>` for visited flags (1 byte vs potentially padded `bool`)
- **Adaptive iterations**: Reduces computation burden for large N while maintaining solution quality
