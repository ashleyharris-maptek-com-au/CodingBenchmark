Complete JavaScript program for the mission control scheduling benchmark:

```javascript
const fs = require('fs');

// Read input JSON
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const tasks = data.tasks || [];
const machineCount = data.machineCount || 1;

if (tasks.length === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const numMachines = Math.max(1, machineCount);

// Build dependency graph and gather task data
const graph = {};
const inDegree = {};
const taskData = {};

for (const task of tasks) {
  const tid = task.id;
  graph[tid] = [];
  inDegree[tid] = 0;
  taskData[tid] = {
    id: tid,
    duration: task.duration || 0,
    deps: task.deps || [],
    deadline: task.deadline || Infinity,
    weight: task.weight || 1
  };
}

for (const task of tasks) {
  const deps = task.deps || [];
  for (const dep of deps) {
    graph[dep].push(task.id);
    inDegree[task.id]++;
  }
}

// Topological sort with weighted EDF priority
// Priority = -weight * deadline (earlier deadlines get higher priority)
const queue = [];

for (const tid in inDegree) {
  if (inDegree[tid] === 0) {
    const task = taskData[tid];
    const priority = -task.deadline * task.weight;
    queue.push({ id: tid, priority });
  }
}

// Sort by priority (descending)
queue.sort((a, b) => b.priority - a.priority);

const order = [];

while (queue.length > 0) {
  const idx = 0;
  const taskInfo = queue[idx];
  order.push(taskInfo.id);

  queue.splice(idx, 1);

  for (const depId of graph[taskInfo.id]) {
    inDegree[depId]--;
    if (inDegree[depId] === 0) {
      const task = taskData[depId];
      const priority = -task.deadline * task.weight;
      queue.push({ id: depId, priority });
    }
  }

  queue.sort((a, b) => b.priority - a.priority);
}

// Schedule tasks on machines
const schedule = [];
const machineFinishTime = new Array(numMachines).fill(0);
const taskSchedule = {};

for (const taskId of order) {
  const task = taskData[taskId];

  // Find start time based on dependencies
  let startTime = 0;
  const deps = task.deps;
  if (deps && deps.length > 0) {
    for (const depId of deps) {
      const depEntry = taskSchedule[depId];
      if (depEntry) {
        const depTask = taskData[depId];
        const depFinish = depEntry.start + depTask.duration;
        startTime = Math.max(startTime, depFinish);
      }
    }
  }

  // Find machine with earliest finish time
  let bestMachine = 0;
  let bestFinish = machineFinishTime[0];

  for (let i = 1; i < numMachines; i++) {
    if (machineFinishTime[i] < bestFinish) {
      bestFinish = machineFinishTime[i];
      bestMachine = i;
    }
  }

  const finishTime = startTime + task.duration;

  schedule.push({
    id: taskId,
    start: startTime,
    machine: bestMachine
  });

  taskSchedule[taskId] = { start: startTime };
  machineFinishTime[bestMachine] = finishTime;
}

// Output result
process.stdout.write(JSON.stringify({ schedule }));
process.exit(0);
```

**Algorithm:**

1. **Dependency Graph**: Builds an adjacency list with in-degree tracking for all tasks

2. **Weighted EDF Topological Sort**: Tasks are ordered by priority `−weight × deadline`, prioritizing tasks with earlier deadlines and higher weights

3. **Greedy Machine Scheduling**: Tasks are placed on the machine that becomes available earliest, respecting dependency constraints

**Complexity:** O(V × log V + E) where V = tasks, E = dependencies, scalable for large inputs
