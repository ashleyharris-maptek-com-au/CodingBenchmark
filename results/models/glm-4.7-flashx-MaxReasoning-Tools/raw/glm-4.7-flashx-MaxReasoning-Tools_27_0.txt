Looking at this Graph Coloring problem, I need to handle graphs ranging from 50 nodes to millions of nodes. I'll implement a multi-strategy approach:

1. **Small graphs (≤100 nodes)**: Exact backtracking with MRV (Minimum Remaining Values) and LCO (Least Constraining Value) heuristics
2. **Medium graphs (≤500 nodes, ≤30 colors)**: DSATUR algorithm (Degree of Saturation) - proven to work well in practice
3. **Large graphs**: Fast greedy degree coloring
4. **Complete graphs**: Trivial case

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

class GraphColoring {
private:
    int n, k;
    vector<vector<int>> adj;
    vector<int> color;
    vector<int> degrees;

public:
    GraphColoring(int n, int k) : n(n), k(k), adj(n), color(n), degrees(n) {
        fill(color.begin(), color.end(), -1);
    }

    void add_edge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
        degrees[u]++;
        degrees[v]++;
    }

    bool is_complete() const {
        for (int i = 0; i < n; i++) {
            if (adj[i].size() != n - 1) return false;
        }
        return true;
    }

    bool greedy_coloring() {
        fill(color.begin(), color.end(), -1);

        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            return degrees[a] > degrees[b];
        });

        for (int v : order) {
            vector<bool> used(k, false);
            for (int neighbor : adj[v]) {
                if (color[neighbor] != -1) {
                    used[color[neighbor]] = true;
                }
            }
            int c = 0;
            for (; c < k; c++) {
                if (!used[c]) break;
            }
            if (c == k) return false;
            color[v] = c;
        }
        return true;
    }

    bool dsatur() {
        fill(color.begin(), color.end(), -1);

        vector<int> saturation(n, 0);
        vector<bool> colored(n, false);

        while (true) {
            int best = -1, max_sat = -1;
            for (int i = 0; i < n; i++) {
                if (!colored[i] && saturation[i] > max_sat) {
                    max_sat = saturation[i];
                    best = i;
                }
            }
            if (best == -1) break;

            vector<bool> used(k, false);
            for (int neighbor : adj[best]) {
                if (colored[neighbor]) {
                    used[color[neighbor]] = true;
                }
            }
            int c = 0;
            for (; c < k; c++) {
                if (!used[c]) break;
            }
            if (c == k) return false;

            color[best] = c;
            colored[best] = true;

            for (int neighbor : adj[best]) {
                if (!colored[neighbor]) {
                    bool would_conflict = false;
                    for (int nn : adj[neighbor]) {
                        if (colored[nn] && color[nn] == c) {
                            would_conflict = true;
                            break;
                        }
                    }
                    if (would_conflict) saturation[neighbor]++;
                }
            }
        }
        return true;
    }

    bool backtrack(int idx) {
        if (idx == n) return true;

        int min_idx = -1, min_options = k + 1;

        for (int i = 0; i < n; i++) {
            if (color[i] == -1) {
                int options = 0;
                for (int c = 0; c < k; c++) {
                    bool valid = true;
                    for (int nn : adj[i]) {
                        if (color[nn] != -1 && color[nn] == c) {
                            valid = false;
                            break;
                        }
                    }
                    if (valid) options++;
                }
                if (options == 0) return false;
                if (options < min_options) {
                    min_options = options;
                    min_idx = i;
                }
            }
        }

        vector<bool> used(k, false);
        for (int neighbor : adj[min_idx]) {
            if (color[neighbor] != -1) {
                used[color[neighbor]] = true;
            }
        }

        for (int c = 0; c < k; c++) {
            if (!used[c]) {
                color[min_idx] = c;
                if (backtrack(idx + 1)) return true;
                color[min_idx] = -1;
            }
        }
        return false;
    }

    bool backtracking_coloring() {
        fill(color.begin(), color.end(), -1);
        return backtrack(0);
    }

    vector<int> solve() {
        if (is_complete()) {
            if (n <= k) {
                fill(color.begin(), color.end(), -1);
                for (int i = 0; i < n; i++) {
                    color[i] = i;
                }
                return color;
            }
            return {};
        }

        if (n <= 100) {
            if (backtracking_coloring()) return color;
            return {};
        }

        if (n <= 500 && k <= 30) {
            if (dsatur()) return color;
        }

        if (greedy_coloring()) return color;

        if (n <= 200) {
            fill(color.begin(), color.end(), -1);
            if (backtracking_coloring()) return color;
        }

        return {};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    GraphColoring gc(n, k);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        gc.add_edge(u, v);
    }

    vector<int> result = gc.solve();

    if (result.empty()) {
        cout << -1 << "\n";
    } else {
        for (int i = 0; i < n; i++) {
            cout << result[i];
            if (i < n - 1) cout << " ";
        }
        cout << "\n";
    }

    return 0;
}
```
