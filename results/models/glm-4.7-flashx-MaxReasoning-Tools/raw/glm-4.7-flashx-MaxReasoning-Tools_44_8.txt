```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Normalized world-space normal
    vec3 N = normalize(normal);

    // Light direction (assumed directional)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse term
    float d = max(dot(N, L), 0.0);

    // Quantize diffuse term into bands
    float band;
    if (d >= 0.75)
        band = 1.0;
    else if (d >= 0.5)
        band = 0.7;
    else if (d >= 0.25)
        band = 0.4;
    else
        band = 0.2;

    // Multiply vertex color by the band and output with full opacity
    outColor = vec4(color * band, 1.0);
}
```
