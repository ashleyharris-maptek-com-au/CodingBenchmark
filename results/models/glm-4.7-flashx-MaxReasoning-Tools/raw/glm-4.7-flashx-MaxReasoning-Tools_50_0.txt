# 3D Boid Flocking Compute Shader

```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

const float EPSILON = 0.0001;

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against gl_GlobalInvocationID.x >= N
    if (id >= N) {
        return;
    }
    
    uint inIdx = id * 2;
    uint outIdx = id * 2;
    
    vec4 inPos = inData[inIdx];
    vec4 inVel = inData[inIdx + 1];
    
    vec3 pos = inPos.xyz;
    vec3 vel = inVel.xyz;
    
    // Compute boid parameters from uniforms
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    vec3 sep = vec3(0.0);
    vec3 align = vec3(0.0);
    vec3 coh = vec3(0.0);
    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;
    
    // Iterate over all boids
    for (uint j = 0; j < N; j++) {
        if (id == j) {
            continue;  // Skip self
        }
        
        uint jIdx = j * 2;
        vec4 otherPos = inData[jIdx];
        vec4 otherVel = inData[jIdx + 1];
        
        vec3 diff = pos - otherPos.xyz;
        float distSq = dot(diff, diff);
        
        // Skip zero-distance neighbors
        if (distSq < EPSILON) {
            continue;
        }
        
        // Separation: steer away from nearby boids
        if (distSq < sepRadius * sepRadius) {
            float dist = sqrt(distSq);
            // Weight by inverse distance (stronger when closer)
            float weight = sepRadius / dist;
            sep += diff * weight;
            sepCount++;
        }
        
        // Alignment: average velocity of nearby boids
        if (distSq < alignRadius * alignRadius) {
            align += otherVel.xyz;
            alignCount++;
        }
        
        // Cohesion: average position of nearby boids
        if (distSq < cohRadius * cohRadius) {
            coh += otherPos.xyz;
            cohCount++;
        }
    }
    
    // Average alignment force
    if (alignCount > 0) {
        align /= float(alignCount);
    }
    
    // Average cohesion force (desired direction to centroid)
    if (cohCount > 0) {
        coh /= float(cohCount);
        coh -= pos;  // Direction toward centroid
        // Normalize cohesion direction
        float cohLen = length(coh);
        if (cohLen > EPSILON) {
            coh /= cohLen;
        }
    }
    
    // Average separation force
    if (sepCount > 0) {
        sep /= float(sepCount);
        // Normalize separation direction
        float sepLen = length(sep);
        if (sepLen > EPSILON) {
            sep /= sepLen;
        }
    }
    
    // Accumulate weighted flocking forces
    vec3 accel = vec3(0.0);
    accel += sep * sepWeight;
    accel += align * alignWeight;
    accel += coh * cohWeight;
    
    // Boundary force: push toward center (boundSize * 0.5)
    float boundaryLimit = boundSize * 0.5;
    vec3 boundaryForce = vec3(0.0);
    
    if (pos.x > boundaryLimit) {
        boundaryForce.x = -boundForce * (pos.x - boundaryLimit);
    } else if (pos.x < -boundaryLimit) {
        boundaryForce.x = boundForce * (boundaryLimit - pos.x);
    }
    
    if (pos.y > boundaryLimit) {
        boundaryForce.y = -boundForce * (pos.y - boundaryLimit);
    } else if (pos.y < -boundaryLimit) {
        boundaryForce.y = boundForce * (boundaryLimit - pos.y);
    }
    
    if (pos.z > boundaryLimit) {
        boundaryForce.z = -boundForce * (pos.z - boundaryLimit);
    } else if (pos.z < -boundaryLimit) {
        boundaryForce.z = boundForce * (boundaryLimit - pos.z);
    }
    
    accel += boundaryForce;
    
    // Update velocity by accel * dt
    vel += accel * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }
    
    // Update position by vel * dt
    pos += vel * dt;
    
    // Write output (pos with w=0, vel with w=0)
    outData[outIdx] = vec4(pos, 0.0);
    outData[outIdx + 1] = vec4(vel, 0.0);
}
```

## Key Features

| Feature | Implementation |
|---------|----------------|
| **Bounds check** | `if (id >= N) return;` at start |
| **Self exclusion** | `if (id == j) continue;` in loop |
| **Zero-distance guard** | `if (distSq < EPSILON) continue;` |
| **Force averaging** | Divides accumulated sums by `count` (handles zero division) |
| **Speed clamping** | Normalizes velocity if `length(vel) > maxSpeed` |
| **Boundary force** | Applies `-boundForce * (pos - boundLimit)` when outside center region |
| **Separation weighting** | Inverse distance weighting (`sepRadius / dist`) for stronger repulsion when closer |
