; SPIR-V
; Version: 1.0
; Extensions: 
; Addressing Model: Logical
; Memory Model: GLSL450
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%mat4 = OpTypeMatrix %v4float 4
%_struct_10 = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_10 = OpTypePointer Uniform %_struct_10
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Input_v3float_3 = OpTypePointer Input %v3float
%_ptr_Input_v3float_4 = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %bool = OpTypeBool

; Constants
%const_0 = OpConstant %float 0
%const_1 = OpConstant %float 1
%const_2 = OpConstant %float 2
%const_3 = OpConstant %float 3
%const_4 = OpConstant %float 4
%const_20 = OpConstant %float 20
%const_6 = OpConstant %float 6
%const_1_5 = OpConstant %float 1.5
%const_0_5 = OpConstant %float 0.5
%const_true = OpConstantTrue %bool
%const_false = OpConstantFalse %bool

; Uniform buffer variable
%uniformBuffer = OpVariable %_ptr_Uniform_10 Uniform

; Input variables with Location decorations
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%color = OpVariable %_ptr_Input_v3float_4 Input
%outColor = OpVariable %_ptr_Output_v4float Output

; Decoration for output
OpDecorate %outColor Location 0

; Function
%main = OpFunction %void None %void
%main_label = OpLabel

; Get worldPos, normal, uv, color from inputs
%worldPos_load = OpLoad %v3float %worldPos
%normal_load = OpLoad %v3float %normal
%uv_load = OpLoad %v2float %uv
%color_load = OpLoad %v3float %color

; Get uniform buffer components
%model = OpAccessChain %_ptr_Uniform_mat4 %uniformBuffer %const_0
%view = OpAccessChain %_ptr_Uniform_mat4 %uniformBuffer %const_1
%projection = OpAccessChain %_ptr_Uniform_mat4 %uniformBuffer %const_2
%lightPos = OpAccessChain %_ptr_Uniform_v4float %uniformBuffer %const_3
%cameraPos = OpAccessChain %_ptr_Uniform_v4float %uniformBuffer %const_4
%params = OpAccessChain %_ptr_Uniform_v4float %uniformBuffer %const_5

; Extract scalar values
%lightPos_x = OpCompositeExtract %float %lightPos 0
%cameraPos_x = OpCompositeExtract %float %cameraPos 0
%params_x = OpCompositeExtract %float %params 0

; c = (uv.x * 3 - 2, uv.y * 2 - 1)
%uv_x = OpCompositeExtract %float %uv_load 0
%uv_y = OpCompositeExtract %float %uv_load 1
%c_real = OpFMul %float %uv_x %const_3
%c_real = OpFSub %float %c_real %const_2
%c_imag = OpFMul %float %uv_y %const_2
%c_imag = OpFSub %float %c_imag %const_1

; Initialize Z = (0, 0)
%z_real = OpConstant %float 0
%z_imag = OpConstant %float 0

; Initialize iterations = 0
%iterations = OpConstant %float 0

; Main iteration loop header
%loop_header = OpLabel
%current_iterations = OpLoad %float %iterations

; Check if iterations >= 20 (maximum)
%max_iterations_reached = OpFOrdGreaterThanEqual %bool %current_iterations %const_20

; Check if |Z|^2 > 4
%z_real_sq = OpFMul %float %z_real %z_real
%z_imag_sq = OpFMul %float %z_imag %z_imag
%z_real_imag_sq = OpFAdd %float %z_real_sq %z_imag_sq
%escape = OpFOrdGreaterThan %bool %z_real_imag_sq %const_4

; Determine if we should continue or break
%should_continue = OpLogicalAnd %bool %max_iterations_reached %escape

OpSelectionMerge %loop_merge None
OpBranchConditional %should_continue %loop_merge %loop_continue

%loop_continue = OpLabel
; Update iterations
%new_iterations = OpFAdd %float %current_iterations %const_1

; Z = Z^2 + c
%z_real_2 = OpFMul %float %z_real %z_real
%z_imag_2 = OpFMul %float %z_imag %z_imag
%z_real_new = OpFSub %float %z_real_2 %z_imag_2
%z_real_new = OpFAdd %float %z_real_new %c_real
%z_real_3 = OpFMul %float %z_real %const_2
%z_imag_4 = OpFMul %float %z_real %z_imag
%z_imag_5 = OpFMul %float %z_imag_4 %const_2
%z_imag_new = OpFAdd %float %z_imag_5 %c_imag

; Update Z
OpStore %iterations %new_iterations
%z_real = OpCopyObject %float %z_real_new
%z_imag = OpCopyObject %float %z_imag_new

OpBranch %loop_header

%loop_merge = OpLabel
%current_iterations = OpLoad %float %iterations

; Calculate t = iterations / 20
%t = OpFMul %float %current_iterations %const_1_5
%color_t = OpFMul %float %t %const_0_5
%color_1_t = OpFSub %float %const_1 %t

; Color = (t, t*0.5, 1.0-t), alpha = 1.0
%outColor_r = OpFMul %float %t %const_1
%outColor_g = OpFMul %float %color_t %const_1
%outColor_b = OpFMul %float %color_1_t %const_1
%outColor_a = OpFMul %float %const_1 %const_1

; Create vec4
%outColor_vec = OpCompositeConstruct %v4float %outColor_r %outColor_g %outColor_b %outColor_a

; Store output
OpStore %outColor %outColor_vec

OpReturn
OpFunctionEnd
