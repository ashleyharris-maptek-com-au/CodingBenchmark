**TypeScript (Node.js)**

```ts
/**
 * Schedule mission‑control work across parallel workstations.
 *
 *  - Input JSON on stdin:
 *      { machineCount: number,
 *        tasks: Array<{ id: string, duration: number, deps?: string[], deadline?: number, weight?: number }> }
 *  - Output JSON on stdout:
 *      { schedule: Array<{ id: string, start: number, machine: number }> }
 *    or a simple topological order:
 *      { order: string[] }
 */

// ------------------------------------------------------------
//  1. Parse input
// ------------------------------------------------------------
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: any = JSON.parse(input);

const machineCount: number = data.machineCount;
const tasks = data.tasks as Array<{
    id: string;
    duration: number;
    deps?: string[];
    deadline?: number;
    weight?: number;
}>;

// ------------------------------------------------------------
//  2. Build the DAG (dependencies → dependents) and indegree
// ------------------------------------------------------------
const indegree: Record<string, number> = {};
const dependents: Record<string, string[]> = {};

// initialise empty containers
for (const t of tasks) {
    indegree[t.id] = t.deps ? t.deps.length : 0;
    dependents[t.id] = [];
}

// fill dependents map
for (const t of tasks) {
    const deps = t.deps || [];
    for (const d of deps) {
        dependents[d].push(t.id);
    }
}

// ------------------------------------------------------------
//  3. Kahn's algorithm – produce a topological order
// ------------------------------------------------------------
const queue: string[] = [];
for (const [id, deg] of Object.entries(indegree)) {
    if (deg === 0) {
        queue.push(id);
    }
}
const order: string[] = [];

while (queue.length > 0) {
    const id = queue.shift()!;
    order.push(id);

    const succ = dependents[id];
    for (const s of succ) {
        indegree[s]--;
        if (indegree[s] === 0) {
            queue.push(s);
        }
    }
}

// ------------------------------------------------------------
//  4. Greedy schedule – longest‑available‑start machine
// ------------------------------------------------------------
// If the topological order is not feasible (should not happen) we fall back to order only.
if (order.length !== tasks.length) {
    // malformed DAG – just return the order
    const out: Record<string, string[]> = { order };
    console.log(JSON.stringify(out));
    process.exit(0);
}

const machineFinish: number[] = new Array(machineCount).fill(0);
const taskFinish: Record<string, number> = {};

const schedule: Array<{ id: string; start: number; machine: number }> = [];

for (const id of order) {
    const t = tasks.find((tt) => tt.id === id)!;
    const deps = t.deps || [];

    // earliest time we may start – must wait for all dependencies
    let start = 0;
    for (const d of deps) {
        const f = taskFinish[d];
        if (f > start) {
            start = f;
        }
    }

    // select the machine whose finish time is the smallest value >= start
    let bestMachine = 0;
    let bestFinish = machineFinish[0];
    for (let m = 1; m < machineCount; ++m) {
        const f = machineFinish[m];
        if (f >= start && f < bestFinish) {
            bestFinish = f;
            bestMachine = m;
        }
    }

    const finish = start + t.duration;
    schedule.push({ id, start, machine: bestMachine });
    machineFinish[bestMachine] = finish;
    taskFinish[id] = finish;
}

// ------------------------------------------------------------
//  5. Output JSON
// ------------------------------------------------------------
const out: Record<string, any> = { schedule };
console.log(JSON.stringify(out));
```

**Explanation**

1. **Input parsing** – reads JSON from `stdin` exactly as required.
2. **Dependency graph** – builds indegrees and a `dependents` map for each task.
3. **Topological order** – uses Kahn’s algorithm to produce `order`.
4. **Greedy scheduling** – for each task in the topological order  
   *finds the earliest start time that respects its dependencies*  
   *and chooses the machine whose finish time is the smallest value ≥ that start*.
   This simple longest‑available‑start rule is fast and works well for the benchmark.
5. **Output** – prints the JSON schedule to `stdout`. No extra logs or text.
