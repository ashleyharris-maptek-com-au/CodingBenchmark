```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Compute light direction (from surface to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Compute view direction (from surface to camera)
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Compute reflection vector
    vec3 R = reflect(-L, N);
    
    // Diffuse term
    float diffuse = max(dot(N, L), 0.0);
    
    // Specular term
    float specular = pow(max(dot(R, V), 0.0), 32.0);
    
    // Grayscale intensity: 0.2 + 0.5*diffuse + 0.8*specular
    float intensity = 0.2 + 0.5 * diffuse + 0.8 * specular;
    
    // Output color (grayscale with alpha 1.0)
    outColor = vec4(intensity, intensity, intensity, 1.0);
}
```
