```glsl
#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction (assuming point light at lightPos)
    vec3 L = ubo.lightPos.xyz - worldPos;
    
    // Lambertian diffuse shading
    float diffuse = max(dot(N, normalize(L)), 0.0);
    
    // Apply Lambertian shading factor: (0.3 + 0.7 * diffuse)
    float shadingFactor = 0.3 + 0.7 * diffuse;
    
    // 6x6 UV polka dot pattern
    vec2 cellUV = floor(uv * 6.0);
    vec2 cellCenter = (cellUV + 0.5) / 6.0;
    
    // Distance from UV coordinate to cell center
    float dist = length(uv - cellCenter);
    
    // Dot radius in UV space: 0.3 cells / 6 cells per row = 0.05
    float radius = 0.3 / 6.0;
    
    // Circular dot mask with smooth edges
    float dotMask = 1.0 - smoothstep(radius, radius + 0.01, dist);
    
    // Base colors: dark blue background and yellow dots
    vec3 bgColor = vec3(0.1, 0.1, 0.3);
    vec3 dotColor = vec3(1.0, 0.8, 0.0);
    
    // Mix colors based on dot mask
    vec3 baseColor = mix(bgColor, dotColor, dotMask);
    
    // Apply shading factor to base color
    outColor = vec4(baseColor * shadingFactor, 1.0);
}
```
