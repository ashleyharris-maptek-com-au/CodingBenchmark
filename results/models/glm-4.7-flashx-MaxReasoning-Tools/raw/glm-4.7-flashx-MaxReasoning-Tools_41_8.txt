; SPIR-V
OpVersion 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%bool = OpTypeBool
%float_ptr = OpTypePointer Uniform %float
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr = OpTypePointer Uniform %ubo

%worldPos_in = OpVariable %vec3_ptr Input
%normal_in = OpVariable %vec3_ptr Input
%uv_in = OpVariable %vec2_ptr Input
%tangent_in = OpVariable %vec3_ptr Input
%color_in = OpVariable %vec3_ptr Input
%outColor = OpVariable %vec4_ptr Output

OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %uv_in Location 2
OpDecorate %tangent_in Location 3
OpDecorate %color_in Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Block
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

OpEntryPoint Fragment %main "main" %worldPos_in %normal_in %uv_in %tangent_in %color_in %outColor
OpExecutionMode %main OriginUpperLeft

%float_1 = OpConstant %float 1.0
%float_0 = OpConstant %float 0.0
%float_0_75 = OpConstant %float 0.75
%float_0_5 = OpConstant %float 0.5
%float_0_25 = OpConstant %float 0.25
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_0_2 = OpConstant %float 0.2

%main = OpFunction %void None %void_fn
%main_label = OpLabel

%worldPos = OpLoad %vec3 %worldPos_in
%normal = OpLoad %vec3 %normal_in
%uv = OpLoad %vec2 %uv_in
%tangent = OpLoad %vec3 %tangent_in
%color = OpLoad %vec3 %color_in

%ubo_val = OpLoad %ubo %ubo_var
%model_mat = OpCompositeExtract %mat4 %ubo_val 0
%view_mat = OpCompositeExtract %mat4 %ubo_val 1
%proj_mat = OpCompositeExtract %mat4 %ubo_val 2
%lightPos = OpCompositeExtract %vec4 %ubo_val 3
%cameraPos = OpCompositeExtract %vec4 %ubo_val 4
%params = OpCompositeExtract %vec4 %ubo_val 5

%model_col0 = OpCompositeExtract %vec4 %model_mat 0
%model_col1 = OpCompositeExtract %vec4 %model_mat 1
%model_col2 = OpCompositeExtract %vec4 %model_mat 2
%model_col3 = OpCompositeExtract %vec4 %model_mat 3
%view_col0 = OpCompositeExtract %vec4 %view_mat 0
%view_col1 = OpCompositeExtract %vec4 %view_mat 1
%view_col2 = OpCompositeExtract %vec4 %view_mat 2
%view_col3 = OpCompositeExtract %vec4 %view_mat 3
%proj_col0 = OpCompositeExtract %vec4 %proj_mat 0
%proj_col1 = OpCompositeExtract %vec4 %proj_mat 1
%proj_col2 = OpCompositeExtract %vec4 %proj_mat 2
%proj_col3 = OpCompositeExtract %vec4 %proj_mat 3

%model_mat = OpCompositeConstruct %mat4 %model_col0 %model_col1 %model_col2 %model_col3
%view_mat = OpCompositeConstruct %mat4 %view_col0 %view_col1 %view_col2 %view_col3
%proj_mat = OpCompositeConstruct %mat4 %proj_col0 %proj_col1 %proj_col2 %proj_col3

%lightPosXYZ = OpVectorShuffle %vec3 %lightPos 0 1 2
%L = OpFSub %vec3 %lightPosXYZ %worldPos
%Llen = OpDot %float %L %L
%invLlen = OpFDiv %float %float_1 %Llen
%L = OpVectorTimesScalar %vec3 %L %invLlen

%N = OpLoad %vec3 %normal_in
%Nlen = OpDot %float %N %N
%invNlen = OpFDiv %float %float_1 %Nlen
%N = OpVectorTimesScalar %vec3 %N %invNlen

%diff = OpDot %float %N %L
%diff = OpFSub %float %diff %float_0
%diff = OpFMax %float %diff %float_0

%cond1 = OpFOrdGreaterThanEqual %bool %diff %float_0_75
%bandValue = OpSelect %float %cond1 %float_1 %float_0_2
%cond2 = OpFOrdGreaterThanEqual %bool %diff %float_0_5
%bandValue = OpSelect %float %cond2 %float_0_7 %bandValue
%cond3 = OpFOrdGreaterThanEqual %bool %diff %float_0_25
%bandValue = OpSelect %float %cond3 %float_0_4 %bandValue

%colScaled = OpVectorTimesScalar %vec3 %color %bandValue
%outVec4 = OpCompositeConstruct %vec4 %colScaled %float_1

OpStore %outColor %outVec4
OpReturn
OpFunctionEnd
