/// Type alias for bitset representation
type Bitset = u64;

/// Represents a subset of indices (up to 64 elements for a single bitset)
#[derive(Clone, Debug)]
struct Subset {
    bits: Bitset,
    sum: i64,
}

impl Subset {
    fn new() -> Self {
        Subset { bits: 0, sum: 0 }
    }

    fn from_value(idx: usize, value: i64) -> Self {
        Subset {
            bits: 1u64 << idx,
            sum: value,
        }
    }

    fn empty() -> Self {
        Subset { bits: 0, sum: 0 }
    }
}

/// Returns true if this subset contains element at index idx
fn contains(idx: usize, bits: Bitset) -> bool {
    (bits >> idx) & 1 == 1
}

/// Checks if this is an empty subset
fn is_empty(bits: Bitset) -> bool {
    bits == 0
}

/// Bitset-based dynamic programming for subset sum
/// Time: O(n * target / word_size), Space: O(target / word_size)
fn dp_bitset(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if target < 0 {
        return None;
    }
    if target == 0 {
        return Some((0, Vec::new()));
    }

    let n = nums.len();
    let bits_per_word = 64;
    let mut dp = vec![Bitset::MAX; (target as usize / bits_per_word + 1) as usize];
    dp[0] = 1u64; // 0 can be formed with empty set

    for (idx, &num) in nums.iter().enumerate() {
        if num < 0 {
            continue; // Skip negative numbers for this simple DP
        }
        if num == 0 {
            continue; // Zero doesn't change sum
        }
        if num > target as i64 {
            continue;
        }

        let num = num as usize;
        let word_idx_start = (target - num as i64) as usize / bits_per_word;
        let mut word_idx = target as usize / bits_per_word;

        while word_idx >= word_idx_start {
            let shifted = dp[word_idx] << num;
            dp[word_idx] |= shifted;
            word_idx -= 1;
        }
    }

    // Check if target is reachable
    let target_word = target as usize / bits_per_word;
    let target_bit = target as usize % bits_per_word;
    if (dp[target_word] >> target_bit) & 1 == 0 {
        return None;
    }

    // Reconstruct subset
    let mut subset: Vec<usize> = Vec::new();
    let mut remaining = target as usize;
    let mut bits = dp[target_word];

    for idx in (0..n).rev() {
        if nums[idx] < 0 {
            continue;
        }
        let num = nums[idx] as usize;
        if num == 0 {
            continue;
        }
        if num > remaining {
            continue;
        }
        if (bits >> idx) & 1 != 0 {
            subset.push(idx);
            remaining -= num;
        }
    }

    subset.reverse();
    Some((0, subset))
}

/// Sparse DP for moderate targets
fn dp_sparse(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if target < 0 {
        return None;
    }
    if target == 0 {
        return Some((0, Vec::new()));
    }

    let n = nums.len();
    let mut sums: HashSet<i64> = HashSet::new();
    sums.insert(0);

    let mut best_subset: Option<Vec<usize>> = None;

    for idx in 0..n {
        let num = nums[idx];
        if num == 0 {
            continue;
        }
        let mut new_sums: HashSet<i64> = HashSet::new();
        for &sum_val in &sums {
            let new_sum = sum_val + num;
            if new_sum == target {
                let mut subset = best_subset.clone().unwrap_or_default();
                subset.push(idx);
                return Some((0, subset));
            }
            if new_sum < target && new_sums.insert(new_sum) {
                // New sum found, track subset
                let mut new_subset = best_subset.clone().unwrap_or_default();
                new_subset.push(idx);
                // For now, just store the sum without full subset tracking
                // We'll rebuild later
            }
        }
        // Simplified: track all reachable sums
        for &sum_val in &sums {
            new_sums.insert(sum_val);
            let new_sum = sum_val + num;
            if new_sum <= target {
                new_sums.insert(new_sum);
            }
        }
        sums = new_sums;
    }

    None
}

/// Meet-in-the-middle algorithm
/// Time: O(2^(n/2)), Space: O(2^(n/2))
fn meet_in_middle(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    let n = nums.len();
    if n == 0 {
        if target == 0 {
            return Some((0, Vec::new()));
        }
        return None;
    }

    let mid = n / 2;
    let left: Vec<i64> = nums[..mid].to_vec();
    let right: Vec<i64> = nums[mid..].to_vec();

    // Generate all subsets of left half
    let left_count = 1 << left.len();
    let mut left_subsets: Vec<(Bitset, i64)> = Vec::with_capacity(left_count);
    for mask in 0..left_count {
        let mut bits = 0u64;
        let mut sum: i64 = 0;
        for i in 0..left.len() {
            if (mask >> i) & 1 == 1 {
                bits |= 1u64 << i;
                sum += left[i];
            }
        }
        left_subsets.push((bits, sum));
    }

    // Generate all subsets of right half
    let right_count = 1 << right.len();
    let mut right_subsets: Vec<(Bitset, i64)> = Vec::with_capacity(right_count);
    for mask in 0..right_count {
        let mut bits = 0u64;
        let mut sum: i64 = 0;
        for i in 0..right.len() {
            if (mask >> i) & 1 == 1 {
                bits |= 1u64 << i;
                sum += right[i];
            }
        }
        right_subsets.push((bits, sum));
    }

    // Sort by sum for binary search
    right_subsets.sort_by_key(|&(bits, sum)| sum);

    // For each left subset, check if we can find complement in right
    for (left_bits, left_sum) in left_subsets {
        if left_sum == target {
            let mut subset: Vec<usize> = Vec::new();
            // Extract indices from left bits
            let mut temp_bits = left_bits;
            let mut idx = 0;
            while temp_bits != 0 {
                if temp_bits & 1 == 1 {
                    subset.push(idx);
                }
                temp_bits >>= 1;
                idx += 1;
            }
            return Some((0, subset));
        }

        let complement = target - left_sum;
        if complement < 0 {
            continue;
        }

        // Binary search for complement in right_subsets
        let right_len = right_subsets.len();
        let mut lo = 0;
        let mut hi = right_len;

        while lo < hi {
            let mid = lo + (hi - lo) / 2;
            if right_subsets[mid].1 < complement {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }

        if lo < right_len && right_subsets[lo].1 == complement {
            // Found match! Combine bits
            let right_bits = right_subsets[lo].0;
            let mut subset: Vec<usize> = Vec::new();

            // Extract indices from left bits
            let mut temp_bits = left_bits;
            let mut idx = 0;
            while temp_bits != 0 {
                if temp_bits & 1 == 1 {
                    subset.push(idx);
                }
                temp_bits >>= 1;
                idx += 1;
            }

            // Extract indices from right bits (offset by mid)
            let mut temp_bits = right_bits;
            let mut idx = mid;
            while temp_bits != 0 {
                if temp_bits & 1 == 1 {
                    subset.push(idx);
                }
                temp_bits >>= 1;
                idx += 1;
            }

            subset.sort();
            return Some((0, subset));
        }
    }

    None
}

/// Pisinger's algorithm for subset sum
/// Time: O(n * min(W, R)), Space: O(R)
fn pisinger(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if target == 0 {
        return Some((0, Vec::new()));
    }
    if nums.is_empty() {
        return None;
    }

    let n = nums.len();
    let mut a = nums.to_vec();
    let mut b = 0i64;
    let mut x = 0usize;
    let mut S = 0usize;

    // Partition into positive and negative numbers
    let mut pos: Vec<i64> = Vec::new();
    let mut neg: Vec<i64> = Vec::new();
    let mut sum_pos: i64 = 0;
    let mut sum_neg: i64 = 0;

    for &num in &a {
        if num >= 0 {
            pos.push(num);
            sum_pos += num;
        } else {
            neg.push(num);
            sum_neg += num;
        }
    }

    // If sum of all positives is less than target and sum of all negatives is greater than target
    if sum_pos < target && sum_neg > target {
        return None;
    }

    // If sum of all positives is exactly target
    if sum_pos == target {
        let mut subset: Vec<usize> = pos.iter()
            .enumerate()
            .filter(|(_, &num)| num > 0)
            .map(|(i, _)| i)
            .collect();
        subset.sort();
        return Some((0, subset));
    }

    // If sum of all negatives is exactly target
    if sum_neg == target {
        let mut subset: Vec<usize> = neg.iter()
            .enumerate()
            .filter(|(_, &num)| num < 0)
            .map(|(i, _)| i)
            .collect();
        subset.sort();
        return Some((0, subset));
    }

    // Find subset of negatives with sum close to target
    // Using meet-in-the-middle for negatives
    if !neg.is_empty() {
        let neg_count = neg.len();
        if neg_count <= 30 {
            // Brute force for small number of negatives
            let neg_len = neg.len();
            let max_mask = 1 << neg_len;
            let mut best_sum: Option<i64> = None;
            let mut best_bits: Option<Bitset> = None;

            for mask in 0..max_mask {
                let mut sum: i64 = 0;
                for i in 0..neg_len {
                    if (mask >> i) & 1 == 1 {
                        sum += neg[i];
                    }
                }
                if (target - sum) >= 0 && sum <= 0 {
                    if let Some(best) = best_sum {
                        if sum > best {
                            best_sum = Some(sum);
                            best_bits = Some(mask as Bitset);
                        }
                    } else {
                        best_sum = Some(sum);
                        best_bits = Some(mask as Bitset);
                    }
                }
            }

            if let (Some(sum_neg_subset), bits) = (best_sum, best_bits) {
                // Find subset of positives that makes up the difference
                let remaining = target - sum_neg_subset;
                if remaining >= 0 {
                    if remaining == 0 {
                        // Only negative subset
                        let mut subset: Vec<usize> = Vec::new();
                        let mut temp_bits = bits;
                        let mut idx = 0;
                        while temp_bits != 0 {
                            if temp_bits & 1 == 1 {
                                subset.push(idx);
                            }
                            temp_bits >>= 1;
                            idx += 1;
                        }
                        subset.sort();
                        return Some((0, subset));
                    }

                    // Use DP for positives
                    let pos_dp: Vec<bool> = vec![false; (remaining as usize + 1) as usize];
                    pos_dp[0] = true;
                    let mut pos_idx: Vec<Option<usize>> = vec![None; (remaining as usize + 1) as usize];
                    pos_idx[0] = Some(0);

                    for &num in &pos {
                        if num <= remaining {
                            for sum in (0..=(remaining - num as i64) as usize).rev() {
                                if pos_dp[sum] && !pos_dp[sum + num as usize] {
                                    pos_dp[sum + num as usize] = true;
                                    pos_idx[sum + num as usize] = Some(pos.iter().position(|&x| x == num).unwrap());
                                }
                            }
                        }
                    }

                    if pos_dp[remaining as usize] {
                        // Reconstruct positive subset
                        let mut pos_subset: Vec<usize> = Vec::new();
                        let mut current = remaining as usize;
                        while current > 0 {
                            if let Some(idx) = pos_idx[current] {
                                pos_subset.push(idx);
                                current -= pos[idx] as usize;
                            }
                        }

                        // Combine negative and positive subsets
                        let mut subset: Vec<usize> = Vec::new();

                        // Add positive indices
                        for &idx in &pos_subset {
                            subset.push(idx);
                        }

                        // Add negative indices
                        let mut temp_bits = bits;
                        let mut idx = 0;
                        while temp_bits != 0 {
                            if temp_bits & 1 == 1 {
                                subset.push(idx + pos.len());
                            }
                            temp_bits >>= 1;
                            idx += 1;
                        }

                        subset.sort();
                        return Some((0, subset));
                    }
                }
            }
        }
    }

    // Fallback to meet-in-the-middle for remaining problem
    meet_in_middle(nums, target)
}

/// Partition-based subset sum using Karmarkar-Karp heuristic for large instances
fn partition_kk(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if target == 0 {
        return Some((0, Vec::new()));
    }

    // Use simple greedy approach with Karmarkar-Karp refinement for large instances
    let n = nums.len();
    if n > 10000 {
        // For very large instances, use greedy approximation
        let mut selected: Vec<usize> = Vec::new();
        let mut remaining: Vec<i64> = nums.to_vec();
        let mut current_sum: i64 = 0;

        while !remaining.is_empty() {
            // Find element closest to current sum
            let mut best_idx = 0;
            let mut best_diff = i64::MAX;
            for (idx, &num) in remaining.iter().enumerate() {
                let diff = (target - current_sum).abs() - num.abs();
                if diff < best_diff {
                    best_diff = diff;
                    best_idx = idx;
                }
            }

            selected.push(best_idx);
            current_sum += nums[best_idx];
            remaining.remove(best_idx);
        }

        if current_sum == target {
            selected.sort();
            return Some((0, selected));
        }
        None
    } else {
        meet_in_middle(nums, target)
    }
}

/// Main solving function
fn solve_subset_sum(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if nums.is_empty() {
        if target == 0 {
            return Some((0, Vec::new()));
        }
        return None;
    }

    let n = nums.len();
    
    // Check for trivial cases
    if target == 0 {
        return Some((0, Vec::new()));
    }
    
    // Check if any single element equals target
    for (idx, &num) in nums.iter().enumerate() {
        if num == target {
            return Some((0, vec![idx]));
        }
    }
    
    // Check if all elements sum to target
    let total: i64 = nums.iter().sum();
    if total == target {
        let indices: Vec<usize> = (0..n).collect();
        return Some((0, indices));
    }

    let algorithm = select_algorithm(n, target, 1000);

    match algorithm {
        Algorithm::MeetInMiddle => meet_in_middle(nums, target),
        Algorithm::DPBitset => dp_bitset(nums, target),
        Algorithm::DPSparse => dp_sparse(nums, target),
        Algorithm::Pisinger => pisinger(nums, target),
    }
}

/// Reads all numbers from stdin
fn read_numbers<R: BufRead>(reader: R) -> (usize, i64, Vec<i64>) {
    let mut lines = reader.lines();
    
    // Read first line: num_numbers target
    let first_line = lines.next().unwrap_or(Ok(String::new())).unwrap_or_default();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    
    let num_numbers: usize = parts[0].parse().unwrap_or(0);
    let target: i64 = parts.get(1).and_then(|s| s.parse().ok()).unwrap_or(0);
    
    // Read numbers
    let mut nums = Vec::with_capacity(num_numbers);
    for line in lines {
        let line = line.unwrap_or_default();
        for token in line.split_whitespace() {
            if let Ok(num) = token.parse::<i64>() {
                nums.push(num);
                if nums.len() >= num_numbers {
                    break;
                }
            }
        }
        if nums.len() >= num_numbers {
            break;
        }
    }
    
    (num_numbers, target, nums)
}

fn dp_bitset(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {

    }
    if target == 0 {

            dp_pos[word_idx] |= shifted;

    let target_bit = pos_target % bits_per_word;

        let num = pos[idx];
        if num <= 0 || num > remaining {

        let num_usize = num as usize;

            remaining -= num;

    }

    subset.sort();
    Some((0, subset))

/// Meet-in-the-middle algorithm
fn meet_in_middle(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    let n = nums.len();
    if n == 0 {
        if target == 0 {
            return Some((0, Vec::new()));
        }
        return None;
    }

    let mid = n / 2;
    let left: Vec<i64> = nums[..mid].to_vec();
    let right: Vec<i64> = nums[mid..].to_vec();

    // Generate all subsets of left half with their sums
    let left_subsets: Vec<(u64, i64)> = (0..1u64 << left.len())
        .map(|mask| {
            let mut bits = 0u64;
            let mut sum: i64 = 0;
            for i in 0..left.len() {
                if (mask >> i) & 1 == 1 {
                    bits |= 1u64 << i;
                    sum += left[i];
                }
            }
            (bits, sum)
        })
        .collect();

    // Generate and sort all subsets of right half by sum
    let right_subsets: Vec<(u64, i64)> = (0..1u64 << right.len())
        .map(|mask| {
            let mut bits = 0u64;
            let mut sum: i64 = 0;
            for i in 0..right.len() {
                if (mask >> i) & 1 == 1 {
                    bits |= 1u64 << i;
                    sum += right[i];
                }
            }
            (bits, sum)
        })
        .collect();

    right_subsets.sort_by_key(|&(_, sum)| sum);

    // Find matching pair
    for &(left_bits, left_sum) in &left_subsets {
        let complement = target - left_sum;
        if complement < 0 {
            continue;
        }

        // Binary search for complement
        let right_len = right_subsets.len();

            let mid = lo + (hi - lo) / 2;
            if right_subsets[mid].1 < complement {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }

        if lo < right_len && right_subsets[lo].1 == complement {
            // Combine subsets
            let (right_bits, _) = right_subsets[lo];
            let mut subset: Vec<usize> = Vec::new();

            // Left half indices
            let mut temp_bits = left_bits;
            let mut idx = 0;
            while temp_bits != 0 {
                if temp_bits & 1 == 1 {
                    subset.push(idx);

                temp_bits >>= 1;
                idx += 1;

            let mut idx = mid;

                }
                temp_bits >>= 1;
                idx += 1;
            }

            subset.sort();
            return Some((0, subset));

/// Bitset-based dynamic programming for subset sum
/// Time: O(n * target / word_size), Space: O(target / word_size)
fn dp_bitset(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if target == 0 {
        return Some((0, Vec::new()));
    }
    if nums.is_empty() {
        return None;
    }

    let n = nums.len();
    if n == 0 {
        return None;
    }

    // Separate positive and negative numbers for separate DP passes
    let mut pos: Vec<i64> = Vec::with_capacity(n);
    let mut neg: Vec<i64> = Vec::with_capacity(n);
    
    for &num in nums {
        if num > 0 {
            pos.push(num);
        } else {
            neg.push(num);
        }
    }

    let bits_per_word = size_of::<usize>() * 8;
    
    // DP for positive numbers
    let pos_target = target as usize;
    let mut dp_pos = vec![0u64; (pos_target / bits_per_word + 1) as usize];
    dp_pos[0] = 1u64;
    
    for &num in &pos {
        let num = num as usize;
        if num == 0 || num > pos_target {
            continue;
        }
        for word_idx in (0..=pos_target / bits_per_word).rev() {
            let shifted = dp_pos[word_idx] << num;
            dp_pos[word_idx] |= shifted;
        }
    }

    let target_word = pos_target / bits_per_word;
    let target_bit = pos_target % bits_per_word;

    // Check if target is reachable using only positive numbers
    if (dp_pos[target_word] >> target_bit) & 1 != 0 {
        // Reconstruct subset
        let mut subset: Vec<usize> = Vec::new();
        let mut remaining = pos_target;
        let mut bits = dp_pos[target_word];

        for idx in (0..pos.len()).rev() {
            let num = pos[idx];
            if num <= 0 || num > remaining {
                continue;
            }
            let num_usize = num as usize;
            if (bits >> idx) & 1 != 0 {
                subset.push(idx);
                remaining -= num_usize;
            }
        }

        subset.sort();
        return Some((0, subset));
    }

    // Try DP for negative numbers (target is negative or we need negative numbers)
    let neg_target = (-target) as usize;
    let mut dp_neg = vec![0u64; (neg_target / bits_per_word + 1) as usize];
    dp_neg[0] = 1u64;
    
    for &num in &neg {
        let num = (-num) as usize;
        if num == 0 || num > neg_target {
            continue;
        }
        for word_idx in (0..=neg_target / bits_per_word).rev() {
            let shifted = dp_neg[word_idx] << num;
            dp_neg[word_idx] |= shifted;
        }
    }

    let target_word = neg_target / bits_per_word;
    let target_bit = neg_target % bits_per_word;
    
    if (dp_neg[target_word] >> target_bit) & 1 == 0 {
        return None;
    }

    // Reconstruct subset with negative numbers
    let mut subset: Vec<usize> = Vec::new();
    let mut remaining = neg_target;
    let mut bits = dp_neg[target_word];

    for idx in (0..neg.len()).rev() {
        let num = (-neg[idx]) as usize;
        if num == 0 || num > remaining {
            continue;
        }
        if (bits >> idx) & 1 != 0 {
            subset.push(idx + pos.len());
            remaining -= num;
        }
    }

    subset.sort();
    Some((0, subset))
}

/// Meet-in-the-middle algorithm
/// Time: O(2^(n/2)), Space: O(2^(n/2))
fn meet_in_middle(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    let n = nums.len();
    if n == 0 {
        if target == 0 {
            return Some((0, Vec::new()));
        }
        return None;
    }

    let mid = n / 2;
    let left: Vec<i64> = nums[..mid].to_vec();
    let right: Vec<i64> = nums[mid..].to_vec();

    let left_count = 1usize << left.len();
    let right_count = 1usize << right.len();

    // Generate all subsets of left half with their sums
    let left_subsets: Vec<(u64, i64)> = (0..left_count)
        .map(|mask| {
            let mut bits = 0u64;
            let mut sum: i64 = 0;
            for i in 0..left.len() {
                if (mask >> i) & 1 == 1 {
                    bits |= 1u64 << i;
                    sum += left[i];
                }
            }
            (bits, sum)
        })
        .collect();

    // Generate all subsets of right half and sort by sum
    let mut right_subsets: Vec<(u64, i64)> = (0..right_count)
        .map(|mask| {
            let mut bits = 0u64;
            let mut sum: i64 = 0;
            for i in 0..right.len() {
                if (mask >> i) & 1 == 1 {
                    bits |= 1u64 << i;
                    sum += right[i];
                }
            }
            (bits, sum)
        })
        .collect();

    right_subsets.sort_by_key(|&(_, sum)| sum);

    // Find matching pair (left_sum + right_sum = target)
    for &(left_bits, left_sum) in &left_subsets {
        let complement = target - left_sum;
        if complement < 0 {
            continue;
        }

        // Binary search for complement
        let right_len = right_subsets.len();
        let mut lo = 0;
        let mut hi = right_len;
        while lo < hi {
            let mid = lo + (hi - lo) / 2;
            if right_subsets[mid].1 < complement {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }

        if lo < right_len && right_subsets[lo].1 == complement {
            // Combine subsets
            let (right_bits, _) = right_subsets[lo];
            let mut subset: Vec<usize> = Vec::new();

            // Extract indices from left half
            let mut temp_bits = left_bits;
            let mut idx = 0;
            while temp_bits != 0 {
                if temp_bits & 1 == 1 {
                    subset.push(idx);
                }
                temp_bits >>= 1;
                idx += 1;
            }

            // Extract indices from right half (offset by mid)
            let mut temp_bits = right_bits;
            let mut idx = mid;
            while temp_bits != 0 {
                if temp_bits & 1 == 1 {
                    subset.push(idx);
                }
                temp_bits >>= 1;
                idx += 1;
            }

            subset.sort();
            return Some((0, subset));
        }
    }

    None
}

/// Sparse DP for cases where bitset would be too large
fn dp_sparse(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if target == 0 {
        return Some((0, Vec::new()));
    }
    if nums.is_empty() {
        return None;
    }

    let n = nums.len();
    let mut sums: HashSet<i64> = HashSet::new();
    sums.insert(0);

    for idx in 0..n {
        let num = nums[idx];
        let mut new_sums: HashSet<i64> = HashSet::new();
        for &sum_val in &sums {
            let new_sum = sum_val + num;
            if new_sum == target {
                return Some((0, vec![idx]));
            }
            if new_sum != target && new_sums.insert(new_sum) {
                continue;
            }
        }
        sums = new_sums;
    }

    None
}

/// Main solving function with adaptive algorithm selection
fn solve_subset_sum(nums: &[i64], target: i64) -> Option<(usize, Vec<usize>)> {
    if nums.is_empty() {
        return if target == 0 {
            Some((0, Vec::new()))
        } else {
            None
        };
    }

    // Check for trivial cases
    if target == 0 {
        return Some((0, Vec::new()));
    }

    let n = nums.len();

    // Check if any single element equals target
    for (idx, &num) in nums.iter().enumerate() {
        if num == target {
            return Some((0, vec![idx]));
        }
    }

    // Check if all elements sum to target
    let total: i64 = nums.iter().sum();
    if total == target {
        let indices: Vec<usize> = (0..n).collect();
        return Some((0, indices));
    }

    // For small n, use meet-in-the-middle (up to 60 elements)
    if n <= 60 {
        return meet_in_middle(nums, target);
    }

    // For moderate n with manageable target, use DP bitset
    if target > 0 && target <= 10_000_000 {
        return dp_bitset(nums, target);
    }

    // For large n with large target, use meet-in-the-middle
    if n > 60 {
        return meet_in_middle(nums, target);
    }

    // Fallback to sparse DP
    dp_sparse(nums, target)
}
