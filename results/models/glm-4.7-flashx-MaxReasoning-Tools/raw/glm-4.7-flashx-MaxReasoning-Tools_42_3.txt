```spirv
OpCapability Geometry
OpExtension "SPV_KHR_shader_ballot"
OpExtension "SPV_KHR_storage_class_storage_buffer"
OpMemoryModel Logical GLSL450

%glsl450 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%int = OpTypeInt 32 0
%uint = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%struct_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%worldPos_in = OpTypeArray %v3float %uint_2
%ptr_worldPos_in = OpTypePointer Input %worldPos_in
%worldPos = OpVariable %ptr_worldPos_in Input

%color_in = OpTypeArray %v3float %uint_2
%ptr_color_in = OpTypePointer Input %color_in
%color = OpVariable %ptr_color_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

%ptr_struct_ubo = OpTypePointer Uniform %struct_ubo
%ubo = OpVariable %ptr_struct_ubo Uniform
%ptr_mat4 = OpTypePointer Uniform %mat4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%uint_0 = OpConstant %uint 0
%float_one = OpConstant %float 1.0
%float_zero = OpConstant %float 0.0
%float_thickness = OpConstant %float 0.03

%funcType = OpTypeFunction %void
%main = OpFunction %void None %funcType
%label0 = OpLabel

;--- read input ---------------------------------------------------------
%worldPos0 = OpLoad %v3float (OpAccessChain %ptr_worldPos_in %worldPos %int_0)
%worldPos1 = OpLoad %v3float (OpAccessChain %ptr_worldPos_in %worldPos %int_1)

%color0 = OpLoad %v3float (OpAccessChain %ptr_color_in %color %int_0)
%color1 = OpLoad %v3float (OpAccessChain %ptr_color_in %color %int_1)

;--- compute direction ---------------------------------------------------
%worldPos0_x = OpCompositeExtract %float %worldPos0 0
%worldPos1_x = OpCompositeExtract %float %worldPos1 0
%dir_x = OpFSub %float %worldPos1_x %worldPos0_x

%worldPos0_y = OpCompositeExtract %float %worldPos0 1
%worldPos1_y = OpCompositeExtract %float %worldPos1 1
%dir_y = OpFSub %float %worldPos1_y %worldPos0_y

%worldPos0_z = OpCompositeExtract %float %worldPos0 2
%worldPos1_z = OpCompositeExtract %float %worldPos1 2
%dir_z = OpFSub %float %worldPos1_z %worldPos0_z

%dir = OpCompositeConstruct %v3float %dir_x %dir_y %dir_z

;--- perpendicular in the XY plane ---------------------------------------
%perp_x = OpFNegate %float %dir_y
%perp_y = OpFAdd %float %dir_x %float_zero
%perp_z = OpFAdd %float %float_zero %float_zero

%perp_raw = OpCompositeConstruct %v3float %perp_x %perp_y %perp_z
%perp_len = OpExtInst %float %glsl450 VectorLength %perp_raw
%perp_len_recip = OpFDiv %float %float_one %perp_len
%perp = OpFMul %v3float %perp_raw %perp_len_recip
%perp_thick = OpFMul %v3float %perp %float_thickness

;--- generate the four world positions ----------------------------------
%p0_plus = OpFAdd %v3float %worldPos0 %perp_thick
%p0_minus = OpFSub %v3float %worldPos0 %perp_thick
%p1_plus = OpFAdd %v3float %worldPos1 %perp_thick
%p1_minus = OpFSub %v3float %worldPos1 %perp_thick

;--- project to clip space -----------------------------------------------
%proj_mat_val = OpAccessChain %ptr_mat4 %ubo %int_0
%proj = OpLoad %mat4 %proj_mat_val

%p0_plus_x = OpCompositeExtract %float %p0_plus 0
%p0_plus_y = OpCompositeExtract %float %p0_plus 1
%p0_plus_z = OpCompositeExtract %float %p0_plus 2
%p0_plus_vec4 = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0_plus_z %float_one

%p0_minus_x = OpCompositeExtract %float %p0_minus 0
%p0_minus_y = OpCompositeExtract %float %p0_minus 1
%p0_minus_z = OpCompositeExtract %float %p0_minus 2
%p0_minus_vec4 = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0_minus_z %float_one

%p1_plus_x = OpCompositeExtract %float %p1_plus 0
%p1_plus_y = OpCompositeExtract %float %p1_plus 1
%p1_plus_z = OpCompositeExtract %float %p1_plus 2
%p1_plus_vec4 = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %p1_plus_z %float_one

%p1_minus_x = OpCompositeExtract %float %p1_minus 0
%p1_minus_y = OpCompositeExtract %float %p1_minus 1
%p1_minus_z = OpCompositeExtract %float %p1_minus 2
%p1_minus_vec4 = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %p1_minus_z %float_one

%clip0 = OpExtInst %v4float %glsl450 MatrixTimesVector %proj %p0_plus_vec4
%clip1 = OpExtInst %v4float %glsl450 MatrixTimesVector %proj %p0_minus_vec4
%clip2 = OpExtInst %v4float %glsl450 MatrixTimesVector %proj %p1_plus_vec4
%clip3 = OpExtInst %v4float %glsl450 MatrixTimesVector %proj %p1_minus_vec4

;--- assemble output ------------------------------------------------------
%ptr_out
