Looking at this problem, I need to find the longest common substring among N strings. Given the extreme scale requirements, I'll use a hash-based approach with binary search on the answer length, which adapts well to all problem sizes.

**Key insights:**
1. Binary search on substring length reduces the complexity from O(N·L³) to O(N·M·log M) where M is string length
2. Rolling hash allows O(1) substring hash computation after building the initial hash array
3. Using 64-bit arithmetic (modulo 2⁶⁴) provides extremely low collision probability while avoiding overflow issues

**Strategy:**
- For each candidate length L, compute rolling hashes for all substrings in all strings
- Track common hashes across all strings using hash sets
- Use binary search to find the maximum valid length

**Adaptive features:**
- Uses heap-allocated structures to handle very large strings without stack overflow
- Early termination when a string is too short for the candidate length
- Efficient hash operations using simple arithmetic that compiles well with MSVC

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>
#include <cstdint>
#include <limits>

using namespace std;

const uint64_t BASE = 127;
const uint64_t MOD = (1ULL << 64);

struct HashContext {
    vector<uint64_t> power;
    
    void init(int max_len) {
        power.resize(max_len + 1);
        power[0] = 1;
        for (int i = 1; i <= max_len; i++) {
            power[i] = power[i - 1] * BASE;
        }
    }
    
    vector<uint64_t> build(const string& s) {
        int n = s.length();
        vector<uint64_t> result(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            result[i] = result[i - 1] * BASE + (uint64_t)s[i - 1];
        }
        return result;
    }
    
    uint64_t get(const vector<uint64_t>& h, int l, int r) {
        return h[r] - h[l] * power[r - l];
    }
};

bool has_common_substring(const vector<string>& strings, int L, string& result, HashContext& ctx) {
    if (L == 0) return false;
    int n = strings.size();
    if (n == 0) return false;
    
    auto hash1 = ctx.build(strings[0]);
    unordered_set<uint64_t> common_hashes;
    for (int i = 0; i <= strings[0].length() - L; i++) {
        uint64_t h = ctx.get(hash1, i, i + L);
        common_hashes.insert(h);
    }
    
    for (int i = 1; i < n; i++) {
        if (L > strings[i].length()) return false;
        
        auto hash = ctx.build(strings[i]);
        unordered_set<uint64_t> cur_hashes;
        
        for (int j = 0; j <= strings[i].length() - L; j++) {
            uint64_t h = ctx.get(hash, j, j + L);
            if (common_hashes.count(h)) {
                cur_hashes.insert(h);
            }
        }
        
        common_hashes = move(cur_hashes);
        if (common_hashes.empty()) return false;
    }
    
    for (int i = 0; i <= strings[0].length() - L; i++) {
        uint64_t h = ctx.get(hash1, i, i + L);
        if (common_hashes.count(h)) {
            result = strings[0].substr(i, L);
            return true;
        }
    }
    
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    if (!(cin >> n)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    
    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }
    
    if (n == 0) {
        cout << "\n";
        return 0;
    }
    
    auto min_len_it = min_element(strings.begin(), strings.end(), 
        [](const string& a, const string& b) { return a.length() < b.length(); });
    int min_len = min_len_it->length();
    
    if (min_len == 0) {
        cout << "\n";
        return 0;
    }
    
    HashContext ctx;
    ctx.init(min_len);
    
    int low = 1, high = min_len, best_len = 0;
    string best_substring;
    
    while (low <= high) {
        int mid = (low + high) / 2;
        string cur_substring;
        if (has_common_substring(strings, mid, cur_substring, ctx)) {
            best_len = mid;
            best_substring = cur_substring;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    cout << best_substring << "\n";
    return 0;
}
```
