using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;
using System.Threading.Tasks;
using System.Diagnostics;

public static class BinPackingSolver
{
    // 3D Vector structure
    public struct Vector3
    {
        public float x, y, z;
        public Vector3(float x, float y, float z) { this.x = x; this.y = y; this.z = z; }
        
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.x + b.x, a.y + b.y, a.z + b.z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.x - b.x, a.y - b.y, a.z - b.z);
        public static Vector3 operator *(Vector3 a, float s) => new Vector3(a.x * s, a.y * s, a.z * s);
        
        public float Dot(Vector3 v) => x * v.x + y * v.y + z * v.z;
        public Vector3 Cross(Vector3 v) => new Vector3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
        public float LengthSq() => x * x + y * y + z * z;
        public float Length() => (float)Math.Sqrt(LengthSq());
        public Vector3 Normalize() 
        {
            float len = Length();
            return len > 1e-6f ? *this * (1 / len) : new Vector3(1, 0, 0);
        }
    }

    // Quaternion for rotation (w, x, y, z order)
    public struct Quaternion
    {
        public float w, x, y, z;
        public Quaternion(float w, float x, float y, float z) { this.w = w; this.x = x; this.y = y; this.z = z; }
        
        public static Quaternion Identity => new Quaternion(1, 0, 0, 0);
        
        // Create quaternion from axis-angle
        public static Quaternion FromAxisAngle(Vector3 axis, float angle)
        {
            float half = angle * 0.5f;
            float sin = (float)Math.Sin(half);
            return new Quaternion((float)Math.Cos(half), axis.x * sin, axis.y * sin, axis.z * sin);
        }
        
        // Quaternion multiplication
        public static Quaternion operator *(Quaternion a, Quaternion b) => new Quaternion(
            a.w * b.w - a.x * b.x - a.y * b.y - a.z * b.z,
            a.w * b.x + a.x * b.w + a.y * b.z - a.z * b.y,
            a.w * b.y - a.x * b.z + a.y * b.w + a.z * b.x,
            a.w * b.z + a.x * b.y - a.y * b.x + a.z * b.w);
        
        // Rotate a 3D vector using quaternion
        public Vector3 Rotate(Vector3 v)
        {
            Quaternion q = this, qInv = new Quaternion(q.w, -q.x, -q.y, -q.z);
            Quaternion qp = new Quaternion(q.w * 0 - q.x * v.x - q.y * v.y - q.z * v.z,
                q.w * v.x + q.x * 0 + q.y * v.z - q.z * v.y,
                q.w * v.y - q.x * v.z + q.y * 0 + q.z * v.x,
                q.w * v.z + q.x * v.y - q.y * v.x + q.z * 0);
            float t0 = qp.w * qInv.w - qp.x * qInv.x - qp.y * qInv.y - qp.z * qInv.z;
            float t1 = qp.w * qInv.x + qp.x * qInv.w + qp.y * qInv.z - qp.z * qInv.y;
            float t2 = qp.w * qInv.y - qp.x * qInv.z + qp.y * qInv.w + qp.z * qInv.x;
            float t3 = qp.w * qInv.z + qp.x * qInv.y - qp.y * qInv.x + qp.z * qInv.w;
            return new Vector3(t1, t2, t3);
        }
    }

    // Polyhedron definition
    public class Polyhedron
    {
        public List<Vector3> Vertices = new List<Vector3>();
        public List<int[]> Faces = new List<int[]>();
        public List<Vector3> Normals = new List<Vector3>();
        public Vector3 Min, Max;
        
        public void Compute() 
        {
            Normals.Clear();
            foreach (var face in Faces) 
            {
                var v0 = Vertices[face[0]], v1 = Vertices[face[1]], v2 = Vertices[face[2]];
                var n = new Vector3(v1 - v0).Cross(new Vector3(v2 - v0)).Normalize();
                Normals.Add(n);
            }
            if (Vertices.Count == 0) { Min = Max = new Vector3(0, 0, 0); return; }
            Min = Vertices[0]; Max = Vertices[0];
            foreach (var v in Vertices) { Min = new Vector3(Math.Min(Min.x, v.x), Math.Min(Min.y, v.y), Math.Min(Min.z, v.z)); Max = new Vector3(Math.Max(Max.x, v.x), Math.Max(Max.y, v.y), Math.Max(Max.z, v.z)); }
        }
    }

    // Generates all 24 unique rotations (6 faces × 4 rotations per face)
    public static class Rotations
    {
        private static readonly Quaternion[] Rot = new Quaternion[24];
        static Rotations()
        {
            int i = 0;
            for (int f = 0; f < 6; f++)
            {
                Vector3 axis = f == 0 ? new Vector3(0, 1, 0) :      // +Y up
                               f == 1 ? new Vector3(0, -1, 0) :    // -Y down
                               f == 2 ? new Vector3(1, 0, 0) :     // +X right
                               f == 3 ? new Vector3(-1, 0, 0) :    // -X left
                               f == 4 ? new Vector3(0, 0, 1) :     // +Z front
                               new Vector3(0, 0, -1);              // -Z back
                for (int r = 0; r < 4; r++)
                    Rot[i++] = Quaternion.FromAxisAngle(axis, r * 90f * (float)Math.PI / 180f);
            }
        }
        public static Quaternion[] All => Rot;
        public static Quaternion Random() => Rot[Random.Shared.Next(Rot.Length)];
    }

    // Collision detection using Separating Axis Theorem
    public static class Collision
    {
        public static bool Collide(Polyhedron a, Polyhedron b, Vector3 pa, Quaternion qa, Vector3 pb, Quaternion qb)
        {
            // Quick bounding box rejection test
            var aa = GetAABB(a, pa, qa), ba = GetAABB(b, pb, qb);
            if (aa[1].x < ba[0].x || ba[1].x < aa[0].x ||
                aa[1].y < ba[0].y || ba[1].y < aa[0].y ||
                aa[1].z < ba[0].z || ba[1].z < aa[0].z)
                return false;

            // Collect all unique edge directions from both polyhedra
            var edges = new HashSet<Vector3>();
            foreach (var n in b.Normals) { edges.Add(n); edges.Add(new Vector3(-n.x, -n.y, -n.z)); }
            foreach (var n in b.Normals) 
                for (int i = 0; i < 3; i++) 
                    for (int j = i + 1; j < 3; j++) 
                    {
                        var e1 = b.Vertices[i + 1] - b.Vertices[i];
                        var e2 = b.Vertices[j + 1] - b.Vertices[j];
                        var c = new Vector3(e1).Cross(e2);
                        if (c.LengthSq() > 1e-8f) 
                        {
                            edges.Add(c.Normalize());
                            edges.Add(new Vector3(-c.x, -c.y, -c.z));
                        }
                    }

            // SAT test for each edge direction
            foreach (var dir in edges)
            {
                float minA = float.MaxValue, maxA = float.MinValue;
                float minB = float.MaxValue, maxB = float.MinValue;
                
                foreach (var v in a.Vertices)
                {
                    var rv = qa.Rotate(v);
                    var proj = (rv + pa).Dot(dir);
                    minA = Math.Min(minA, proj);
                    maxA = Math.Max(maxA, proj);
                }
                
                foreach (var v in b.Vertices)
                {
                    var rv = qb.Rotate(v);
                    var proj = (rv + pb).Dot(dir);
                    minB = Math.Min(minB, proj);
                    maxB = Math.Max(maxB, proj);
                }
                
                // Found separating axis → no collision
                if (maxA < minB || maxB < minA)
                    return false;
            }
            
            return true;
        }
        
        // Get axis-aligned bounding box of rotated polyhedron
        public static Vector3[] GetAABB(Polyhedron p, Vector3 pos, Quaternion rot)
        {
            var min = p.Vertices[0], max = p.Vertices[0];
            foreach (var v in p.Vertices)
            {
                var rv = rot.Rotate(v);
                var w = rv + pos;
                min = new Vector3(Math.Min(min.x, w.x), Math.Min(min.y, w.y), Math.Min(min.z, w.z));
                max = new Vector3(Math.Max(max.x, w.x), Math.Max(max.y, w.y), Math.Max(max.z, w.z));
            }
            return new[] { min, max };
        }
    }

    // Represents a placed polyhedron with its position and rotation
    public class Placement 
    { 
        public int Index; 
        public Vector3 Pos; 
        public Quaternion Rot; 
        public Placement(int i, Vector3 p, Quaternion r) { Index = i; Pos = p; Rot = r; } 
    }

    // Strategy for trivial cases (small containers, few polyhedra)
    public static class StrategyTrivial
    {
        public static List<Placement> Solve(Polyhedron[] polys, Vector3 container)
        {
            var results = new List<Placement>();
            var grid = new HashSet<long>();
            int maxAttempts = 5000;
            
            for (int t = 0; t < maxAttempts && results.Count < polys.Length * 2; t++)
            {
                int idx = Random.Shared.Next(polys.Length);
                var rot = Rotations.Random();
                var aabb = Collision.GetAABB(polys[idx], Vector3.Zero, rot);
                
                // Check if fits in container
                if (aabb[1].x <= container.x && aabb[1].y <= container.y && aabb[1].z <= container.z)
                {
                    // Spatial hash key for grid placement
                    long key = ((long)(aabb[0].x * 1000) << 44) | ((long)(aabb[0].y * 1000) << 24) | (long)(aabb[0].z * 1000);
                    
                    if (!grid.Contains(key))
                    {
                        bool ok = true;
                        foreach (var r in results)
                        {
                            if (Collision.Collide(polys[idx], polys[r.Index], r.Pos, r.Rot, Vector3.Zero, rot))
                            { ok = false; break; }
                        }
                        if (ok) 
                        {
                            results.Add(new Placement(idx, aabb[0], rot));
                            grid.Add(key);
                        }
                    }
                }
            }
            return results;
        }
    }

    // Strategy for medium cases (intermediate size)
    public static class StrategyMedium
    {
        public static List<Placement> Solve(Polyhedron[] polys, Vector3 container)
        {
            var results = new List<Placement>();
            var remaining = new List<Polyhedron>(polys);
            
            while (remaining.Count > 0)
            {
                // Greedy: place poly with best fit first
                int bestIdx = 0;
                float bestFit = float.MaxValue;
                for (int i = 0; i < remaining.Count; i++)
                {
                    float fit = remaining[i].Max.Length() / container.Length();
                    if (fit < bestFit) { bestFit = fit; bestIdx = i; }
                }
                
                var p = remaining[bestIdx]; 
                remaining.RemoveAt(bestIdx);
                
                var aabb = Collision.GetAABB(p, Vector3.Zero, Quaternion.Identity);
                // Check if it fits in container bounds
                if (aabb[1].x > container.x || aabb[1].y > container.y || aabb[1].z > container.z)
                    continue;
                
                // Try all rotations
                foreach (var rot in Rotations.All)
                {
                    aabb = Collision.GetAABB(p, Vector3.Zero, rot);
                    if (aabb[1].x > container.x || aabb[1].y > container.y || aabb[1].z > container.z)
                        continue;
                    
                    bool ok = true;
                    foreach (var r in results)
                        if (Collision.Collide(p, polys[r.Index], Vector3.Zero, rot, r.Pos, r.Rot))
                        { ok = false; break; }
                    
                    if (ok) 
                    {
                        results.Add(new Placement(bestIdx, aabb[0], rot));
                        break;
                    }
                }
            }
            
            return results.Count > 0 ? results : StrategyTrivial.Solve(polys, container);
        }
    }

    // Strategy for large cases (high-dimensional search space)
    public static class StrategyLarge
    {
        public static List<Placement> Solve(Polyhedron[] polys, Vector3 container)
        {
            var results = new List<Placement>();
            var remaining = new List<Polyhedron>(polys);
            
            // Limit to reasonable number of placements
            int max = Math.Min(polys.Length, (int)(container.Length() / 5));
            
            // Sort by size (largest first for better packing)
            remaining.Sort((a, b) => b.Max.Length().CompareTo(a.Max.Length()));
            
            // Multiple optimization passes
            for (int iter = 0; iter < 3; iter++)
            {
                var placed = new List<Placement>();
                var rem = new List<Polyhedron>(remaining);
                
                // Randomized search within constraints
                for (int t = 0; t < max * 20 && rem.Count > 0; t++)
                {
                    int idx = Random.Shared.Next(rem.Count);
                    var p = rem[idx]; 
                    rem.RemoveAt(idx);
                    
                    foreach (var rot in Rotations.All)
                    {
                        var aabb = Collision.GetAABB(p, Vector3.Zero, rot);
                        if (aabb[1].x > container.x || aabb[1].y > container.y || aabb[1].z > container.z)
                            continue;
                        
                        bool ok = true;
                        foreach (var r in placed)
                            if (Collision.Collide(p, polys[r.Index], Vector3.Zero, rot, r.Pos, r.Rot))
                            { ok = false; break; }
                        
                        if (ok) 
                        {
                            placed.Add(new Placement(idx, aabb[0], rot));
                            break;
                        }
                    }
                    
                    if (placed.Count >= max) break;
                }
                
                if (placed.Count > results.Count) 
                    results = placed;
            }
            
            return results.Count > 0 ? results : StrategyMedium.Solve(polys, container);
        }
    }

    // Strategy for extreme cases (massive scale with parallel search)
    public static class StrategyExtreme
    {
        public static List<Placement> Solve(Polyhedron[] polys, Vector3 container)
        {
            var best = new List<Placement>();
            
            // Cap at 24 orientations × polyhedra combination limit
            int max = Math.Min(polys.Length, (int)(container.Length() / 2));
            
            // Run multiple trials in parallel
            var tasks = new Task<List<Placement>>[8];
            for (int i = 0; i < 8; i++)
            {
                int seed = i * 12345;
                tasks[i] = Task.Run(() => RunTrial(polys, container, max, seed));
            }
            
            Task.WaitAll(tasks);
            
            foreach (var t in tasks) 
                if (t.Result.Count > best.Count) 
                    best = t.Result;
            
            return best.Count > 0 ? best : StrategyLarge.Solve(polys, container);
        }
        
        private static List<Placement> RunTrial(Polyhedron[] polys, Vector3 container, int max, int seed)
        {
            Random rng = new Random(seed);
            var results = new List<Placement>();
            var remaining = new List<Polyhedron>(polys);
            
            // Sort by size
            remaining.Sort((a, b) => b.Max.Length().CompareTo(a.Max.Length()));
            
            for (int t = 0; t < max * 100 && remaining.Count > 0; t++)
            {
                int idx = Random.Shared.Next(remaining.Count);
                var p = remaining[idx]; 
                remaining.RemoveAt(idx);
                
                foreach (var rot in Rotations.All)
                {
                    var aabb = Collision.GetAABB(p, Vector3.Zero, rot);
                    if (aabb[1].x > container.x || aabb[1].y > container.y || aabb[1].z > container.z)
                        continue;
                    
                    bool ok = true;
                    foreach (var r in results)
                        if (Collision.Collide(p, polys[r.Index], Vector3.Zero, rot, r.Pos, r.Rot))
                        { ok = false; break; }
                    
                    if (ok) 
                    {
                        results.Add(new Placement(idx, aabb[0], rot));
                        break;
                    }
                }
            }
            return results;
        }
    }

    // Main entry point
    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        
        // Parse V and F from first line
        var header = Console.ReadLine()!.Split();
        int v = int.Parse(header[0]);
        int f = int.Parse(header[1]);
        
        // Parse faces first (before vertices)
        var polys = new List<Polyhedron>(f);
        for (int i = 0; i < f; i++)
        {
            polys.Add(new Polyhedron());
            var parts = Console.ReadLine()!.Split();
            polys[i].Faces.Add(new[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) });
        }
        
        // Parse vertices
        for (int i = 0; i < f; i++)
        {
            for (int j = 0; j < v; j++)
            {
                var parts = Console.ReadLine()!.Split();
                polys[i].Vertices.Add(new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2])));
            }
        }
        
        // Parse container size
        var cc = Console.ReadLine()!.Split();
        Vector3 container = new Vector3(float.Parse(cc[0]), float.Parse(cc[1]), float.Parse(cc[2]));
        
        // Precompute normals and bounding boxes
        foreach (var p in polys) p.Compute();
        
        // Choose adaptive strategy
        List<Placement> result;
        if (polys.Length <= 5)
            result = StrategyTrivial.Solve(polys.ToArray(), container);
        else if (polys.Length <= 20 && container.Length() < 100)
            result = StrategyMedium.Solve(polys.ToArray(), container);
        else if (polys.Length <= 100 && container.Length() < 500)
            result = StrategyLarge.Solve(polys.ToArray(), container);
        else
            result = StrategyExtreme.Solve(polys.ToArray(), container);
        
        // Output results
        Console.WriteLine(result.Count);
        foreach (var r in result)
            Console.WriteLine($"{r.Pos.x:F2} {r.Pos.y:F2} {r.Pos.z:F2} {r.Rot.w:F6} {r.Rot.x:F6} {r.Rot.y:F6} {r.Rot.z:F6}");
        
        var elapsed = sw.ElapsedMilliseconds;
        Console.Error.WriteLine($"Solved in {elapsed}ms with {result.Count} polyhedra");
    }
}
