```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Brick pattern parameters
    float brickScale = 8.0;
    float rowScale = 16.0;
    float mortarX = 0.05;
    float mortarY = 0.1;
    float4 mortarColor = float4(0.7, 0.7, 0.7, 1.0);
    float4 brickColor = float4(0.7, 0.2, 0.1, 1.0);
    
    // Scale UVs
    float u = input.uv.x * brickScale;
    float v = input.uv.y * rowScale;
    
    // Staggered rows: offset every other row by half brick width
    v = floor(v) + (step(0.5, frac(v)) * 0.5);
    
    // Create brick grid coordinates
    float brickX = floor(u);
    float brickY = floor(v);
    float fx = frac(u);
    float fy = frac(v);
    
    // Calculate mortar mask
    float mortarMask = max(
        step(1.0 - mortarX, fx),
        step(1.0 - mortarY, fy)
    );
    
    // Apply multiplier: 1.0 for brick, mortarX for mortar width
    float brickMultiplier = (1.0 - mortarMask) + mortarMask * mortarX;
    
    // Get base color (mix brick and mortar colors)
    float4 baseColor = brickColor * brickMultiplier + mortarColor * mortarMask;
    
    // Lambertian shading
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(normalize(input.normal), lightDir), 0.0);
    float lambertian = 0.4 + 0.6 * diffuse;
    
    // Final color with shading
    float4 finalColor = baseColor * lambertian;
    finalColor.a = 1.0;
    
    return finalColor;
}
```
