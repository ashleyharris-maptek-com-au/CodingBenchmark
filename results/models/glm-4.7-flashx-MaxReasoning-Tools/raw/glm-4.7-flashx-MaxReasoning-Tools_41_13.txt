; SPIR-V 1.0

OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo

%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%mat4v4float = OpTypeMatrix %v4float 4

%_ptr_Input_v2float = OpTypePointer Input %v2float

%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_mat4v4float = OpTypePointer Uniform %mat4v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%float_0_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_1_0 = OpConstant %float 1

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input

%tangent = OpVariable %_ptr_Input_v3float_1 Input
%color = OpVariable %_ptr_Input_v3float_2 Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_mat4v4float Uniform

%main = OpFunction %void None %void

%main_label = OpLabel
%worldPos_vec4 = OpCompositeConstruct %v4float %worldPos_0 %worldPos_1 %worldPos_2 %float_1_0
%model = OpLoad %mat4v4float %ubo
%model_vec4_0 = OpCompositeExtract %v4float %model 0
%model_vec4_1 = OpCompositeExtract %v4float %model 1
%model_vec4_2 = OpCompositeExtract %v4float %model 2
%model_vec4_3 = OpCompositeExtract %v4float %model 3
%mul = OpFMul %v4float %worldPos_vec4 %model_vec4_0
%add = OpFAdd %v4float %mul %model_vec4_1
%mul_1 = OpFMul %v4float %add %model_vec4_2
%mul_2 = OpFMul %v4float %mul_1 %model_vec4_3
%view = OpLoad %mat4v4float %ubo
%view_vec4_0 = OpCompositeExtract %v4float %view 0
%view_vec4_1 = OpCompositeExtract %v4float %view 1
%view_vec4_2 = OpCompositeExtract %v4float %view 2
%view_vec4_3 = OpCompositeExtract %v4float %view 3
%mul_3 = OpFMul %v4float %mul_2 %view_vec4_0
%add_1 = OpFAdd %v4float %mul_3 %view_vec4_1
%mul_4 = OpFMul %v4float %add_1 %view_vec4_2
%mul_5 = OpFMul %v4float %mul_4 %view_vec4_3
%projection = OpLoad %mat4v4float %ubo
%projection_vec4_0 = OpCompositeExtract %v4float %projection 0
%projection_vec4_1 = OpCompositeExtract %v4float %projection 1
%projection_vec4_2 = OpCompositeExtract %v4float %projection 2
%projection_vec4_3 = OpCompositeExtract %v4float %projection 3
%mul_6 = OpFMul %v4float %mul_5 %projection_vec4_0
%add_2 = OpFAdd %v4float %mul_6 %projection_vec4_1
%mul_7 = OpFMul %v4float %add_2 %projection_vec4_2
%mul_8 = OpFMul %v4float %mul_7 %projection_vec4_3
%mul_9 = OpFMul %v4float %mul_8 %float_2
%mul_10 = OpFMul %v4float %mul_9 %float_2
%add_3 = OpFAdd %v4float %mul_10 %float_2
%mul_11 = OpFMul %v4float %add_3 %float_2
%div = OpFDiv %v4float %mul_11 %float_2
%div_1 = OpFDiv %v4float %div %float_2
%worldPos_x = OpCompositeExtract %float %div_1 0
%worldPos_y = OpCompositeExtract %float %div_1 1
%mul_12 = OpFMul %float %worldPos_y %float_0_5
%add_4 = OpFAdd %float %mul_12 %float_0_5
%t = OpExtInst %float %None 30 %add_4
%clamp = OpExtInst %float %None 34 %t %float_0 %float_1
%color_bottom = OpCompositeConstruct %v4float %float_0_1 %float_0_4 %float_0_1 %float_1
%color_top = OpCompositeConstruct %v4float %float_1 %float_1 %float_1 %float_1
%mixed = OpExtInst %v4float %None 44 %color_bottom %color_top %clamp
%outColor_vec4 = OpCompositeConstruct %v4float %mixed_0 %mixed_1 %mixed_2 %mixed_3
OpStore %outColor %outColor_vec4
