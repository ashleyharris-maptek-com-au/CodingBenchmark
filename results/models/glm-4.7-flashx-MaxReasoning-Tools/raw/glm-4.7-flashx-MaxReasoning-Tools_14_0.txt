## C# Minesweeper Solver

I've created a complete, compilable C# solution with adaptive strategies for handling any board complexity.

### Solution Overview

**Strategies:**
1. **Simple Strategy** (complexity < 0.5): Optimized for small boards with simple patterns
2. **Advanced Strategy** (0.5 ≤ complexity < 5.0): Uses neighbor maps for efficient lookup on medium-sized boards
3. **Fast Strategy** (complexity ≥ 5.0): Batches processing and bounds iterations for massive boards

**Core Logic:**
- **0-cell propagation**: Cells showing '0' have all neighbors safe
- **Direct mapping**: If `unknown_neighbors == revealed_count`, all unknowns are safe
- **Flagged-based**: If `unknown_neighbors + flagged == revealed_count`, all unknowns are safe

**Complexity Metric:**
`complexity = revealed_cells × density × (1 + zero_cells × 0.1)`

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    private const char UNKNOWN = ' ';
    private const char SAFE = '.';
    private const char FLAGGED = '*';
    
    private char[,] board;
    private int rows;
    private int cols;
    private HashSet<(int x, int y)> safeCells;
    private bool changed;
    private int strategySwitches;
    private int iterations;
    
    public static void Main()
    {
        string[] firstLine = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int H = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        
        char[,] board = new char[H, W];
        for (int r = 0; r < H; r++)
        {
            string line = Console.ReadLine();
            for (int c = 0; c < W; c++)
            {
                board[r, c] = line[c];
            }
        }
        
        MinesweeperSolver solver = new MinesweeperSolver(board);
        solver.Solve();
    }
    
    public MinesweeperSolver(char[,] board)
    {
        this.board = board;
        this.rows = board.GetLength(0);
        this.cols = board.GetLength(1);
        this.safeCells = new HashSet<(int x, int y)>();
        this.changed = true;
        this.strategySwitches = 0;
        this.iterations = 0;
    }
    
    public void Solve()
    {
        double complexity = CalculateComplexity();
        
        if (complexity < 0.5)
        {
            SimpleStrategy();
            strategySwitches = 1;
        }
        else if (complexity < 5.0)
        {
            AdvancedStrategy();
            strategySwitches = 2;
        }
        else
        {
            FastStrategy();
            strategySwitches = 3;
        }
        
        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells.OrderBy(c => c.y).ThenBy(c => c.x))
        {
            Console.WriteLine($"{cell.x} {cell.y}");
        }
    }
    
    private double CalculateComplexity()
    {
        int totalCells = rows * cols;
        int revealedCells = 0;
        int zeroCells = 0;
        
        for (int r = 0; r < rows; r++)
        {
            for (int c = 0; c < cols; c++)
            {
                if (board[r, c] >= '0' && board[r, c] <= '9')
                {
                    revealedCells++;
                    if (board[r, c] == '0')
                    {
                        zeroCells++;
                    }
                }
            }
        }
        
        double density = (double)revealedCells / totalCells;
        double complexity = revealedCells * density * (1 + zeroCells * 0.1);
        
        return complexity;
    }
    
    private void SimpleStrategy()
    {
        while (changed)
        {
            changed = false;
            iterations++;
            
            for (int r = 0; r < rows; r++)
            {
                for (int c = 0; c < cols; c++)
                {
                    if (board[r, c] >= '0' && board[r, c] <= '9')
                    {
                        int count = board[r, c] - '0';
                        List<(int x, int y)> unknown = new List<(int x, int y)>();
                        int flagged = 0;
                        
                        for (int dr = -1; dr <= 1; dr++)
                        {
                            for (int dc = -1; dc <= 1; dc++)
                            {
                                if (dr == 0 && dc == 0) continue;
                                
                                int nr = r + dr;
                                int nc = c + dc;
                                
                                if (nr >= 0 && nr < rows && nc >= 0 && nc < cols)
                                {
                                    char cell = board[nr, nc];
                                    if (cell == UNKNOWN)
                                    {
                                        unknown.Add((nc, nr));
                                    }
                                    else if (cell == FLAGGED)
                                    {
                                        flagged++;
                                    }
                                }
                            }
                        }
                        
                        if (count == 0 && unknown.Count > 0)
                        {
                            foreach (var (x, y) in unknown)
                            {
                                if (!safeCells.Contains((x, y)))
                                {
                                    safeCells.Add((x, y));
                                    board[y, x] = SAFE;
                                    changed = true;
                                }
                            }
                        }
                        else if (count == unknown.Count + flagged)
                        {
                            foreach (var (x, y) in unknown)
                            {
                                if (!safeCells.Contains((x, y)))
                                {
                                    safeCells.Add((x, y));
                                    board[y, x] = SAFE;
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
        }
    }
    
    private void AdvancedStrategy()
    {
        List<List<(int x, int y)[]>[]> neighborMap = new List<List<(int x, int y)[]>[]>();
        for (int r = 0; r < rows; r++)
        {
            List<(int x, int y)>[] rowNeighbors = new List<(int x, int y)>[cols];
            for (int c = 0; c < cols; c++)
            {
                List<(int x, int y)> cellNeighbors = new List<(int x, int y)>();
                for (int dr = -1; dr <= 1; dr++)
                {
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        
                        int nr = r + dr;
                        int nc = c + dc;
                        
                        if (nr >= 0 && nr < rows && nc >= 0 && nc < cols)
                        {
                            cellNeighbors.Add((nc, nr));
                        }
                    }
                }
                rowNeighbors[c] = cellNeighbors.ToArray();
            }
            neighborMap.Add(rowNeighbors);
        }
        
        while (changed)
        {
            changed = false;
            iterations++;
            
            for (int r = 0; r < rows; r++)
            {
                for (int c = 0; c < cols; c++)
                {
                    if (board[r, c] >= '0' && board[r, c] <= '9')
                    {
                        int count = board[r, c] - '0';
                        List<(int x, int y)> unknown = new List<(int x, int y)>();
                        int flagged = 0;
                        
                        foreach (var (nx, ny) in neighborMap[r][c])
                        {
                            char cell = board[ny, nx];
                            if (cell == UNKNOWN)
                            {
                                unknown.Add((nx, ny));
                            }
                            else if (cell == FLAGGED)
                            {
                                flagged++;
                            }
                        }
                        
                        if (count == 0 && unknown.Count > 0)
                        {
                            foreach (var (x, y) in unknown)
                            {
                                if (!safeCells.Contains((x, y)))
                                {
                                    safeCells.Add((x, y));
                                    board[y, x] = SAFE;
                                    changed = true;
                                }
                            }
                        }
                        else if (count == unknown.Count + flagged)
                        {
                            foreach (var (x, y) in unknown)
                            {
                                if (!safeCells.Contains((x, y)))
                                {
                                    safeCells.Add((x, y));
                                    board[y, x] = SAFE;
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
        }
    }
    
    private void FastStrategy()
    {
        const int MAX_ITERATIONS = 1000;
        int iteration = 0;
        
        while (iteration < MAX_ITERATIONS && changed)
        {
            changed = false;
            iteration++;
            
            List<(int r, int c)> cells = new List<(int, int)>();
            for (int r = 0; r < rows; r++)
            {
                for (int c = 0; c < cols; c++)
                {
                    if (board[r, c] >= '0' && board[r, c] <= '9')
                    {
                        cells.Add((r, c));
                    }
                }
            }
            
            int batchSize = Math.Min(100, cells.Count);
            for (int start = 0; start < cells.Count; start += batchSize)
            {
                int end = Math.Min(start + batchSize, cells.Count);
                
                for (int i = start; i < end; i++)
                {
                    int r = cells[i].r;
                    int c = cells[i].c;
                    int count = board[r, c] - '0';
                    List<(int x, int y)> unknown = new List<(int x, int y)>();
                    int flagged = 0;
                    
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            if (dr == 0 && dc == 0) continue;
                            
                            int nr = r + dr;
                            int nc = c + dc;
                            
                            if (nr >= 0 && nr < rows && nc >= 0 && nc < cols)
                            {
                                char cell = board[nr, nc];
                                if (cell == UNKNOWN)
                                {
                                    unknown.Add((nc, nr));
                                }
                                else if (cell == FLAGGED)
                                {
                                    flagged++;
                                }
                            }
                        }
                    }
                    
                    if (count == 0 && unknown.Count > 0)
                    {
                        foreach (var (x, y) in unknown)
                        {
                            if (!safeCells.Contains((x, y)))
                            {
                                safeCells.Add((x, y));
                                board[y, x] = SAFE;
                                changed = true;
                            }
                        }
                    }
                    else if (count == unknown.Count + flagged)
                    {
                        foreach (var (x, y) in unknown)
                        {
                            if (!safeCells.Contains((x, y)))
                            {
                                safeCells.Add((x, y));
                                board[y, x] = SAFE;
                                changed = true;
                            }
                        }
                    }
                }
            }
        }
    }
}
```
