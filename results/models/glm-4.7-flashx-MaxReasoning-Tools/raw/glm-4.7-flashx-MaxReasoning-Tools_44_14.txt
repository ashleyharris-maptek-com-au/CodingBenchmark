```glsl
#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 viewDir = normalize(ubo.cameraPos.xyz - worldPos);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    
    mat3 model3x3 = mat3(ubo.model);
    vec3 worldNormal = normalize(model3x3 * normal);
    
    float diffuse = max(0.0, dot(worldNormal, lightDir));
    vec3 diffuseColor = color * diffuse;
    
    vec3 baseColor = vec3(0.8, 0.2, 0.2);
    vec3 sceneColor = mix(baseColor, diffuseColor, 0.5);
    
    float dist = length(worldPos - ubo.cameraPos.xyz);
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);
    vec3 fogColor = vec3(0.7, 0.7, 0.8);
    
    vec3 finalColor = mix(sceneColor, fogColor, fogFactor);
    outColor = vec4(finalColor, 1.0);
}
```
