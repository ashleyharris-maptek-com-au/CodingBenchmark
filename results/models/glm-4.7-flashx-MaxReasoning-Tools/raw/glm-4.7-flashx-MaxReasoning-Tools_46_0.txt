```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main()
{
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against gl_GlobalInvocationID.x >= N
    if (idx >= N)
    {
        return;
    }

    // Read current position and mass
    vec4 posMass = inData[idx * 2u];
    vec3 position = posMass.xyz;
    float mass = posMass.w;

    // Read current velocity
    vec4 velData = inData[idx * 2u + 1u];
    vec3 velocity = velData.xyz;

    // Initialize acceleration
    vec3 acceleration = vec3(0.0);

    // Physics parameters
    float G = physics.z;
    float softening = physics.y;
    float dt = physics.x;

    // Compute acceleration from all other bodies
    for (uint j = 0u; j < N; j++)
    {
        // Skip self-interaction
        if (j == idx)
        {
            continue;
        }

        // Get position and mass of body j
        vec4 jPosMass = inData[j * 2u];
        vec3 pos_j = jPosMass.xyz;
        float mass_j = jPosMass.w;

        // Displacement vector: r_j - r_i
        vec3 r_vec = pos_j - position;

        // Squared distance with softening
        float r2 = dot(r_vec, r_vec) + softening * softening;
        float r = sqrt(r2);

        // Add gravitational acceleration:
        //   acc += G * m_j * (r_j - r_i) / (|r|^2 + softening^2)^(3/2)
        acceleration += G * mass_j * r_vec / (r * r2);
    }

    // Update velocity: v = v + a * dt
    velocity += acceleration * dt;

    // Update position: r = r + v * dt
    position += velocity * dt;

    // Write updated position+mass
    outData[idx * 2u] = vec4(position, mass);

    // Write updated velocity
    outData[idx * 2u + 1u] = vec4(velocity, 0.0);
}
```
