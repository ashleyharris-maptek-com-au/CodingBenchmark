Looking at this problem, I need to implement a Minimum Vertex Cover solver for very large graphs (2M vertices, 15M edges). Since this is NP-hard, I'll use the classic **2-approximation greedy algorithm**:

1. Iterate through edges
2. For each uncovered edge, add both endpoints to the cover
3. Mark all incident edges as covered
4. This guarantees a 2-approximation (cover size ≤ 2× optimal)

For efficiency with 15M edges:
- Use compact edge representation and adjacency lists
- Use a u64 bitset for fast edge coverage tracking
- Use `BufWriter` for efficient output

```rust
use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let num_vertices: usize = parts.next().unwrap().parse().unwrap();
    let num_edges: usize = parts.next().unwrap().parse().unwrap();
    
    let mut edges: Vec<[usize; 2]> = Vec::with_capacity(num_edges);
    let mut adj: Vec<Vec<usize>> = Vec::with_capacity(num_vertices);
    adj.resize_with(num_vertices, || Vec::new());
    
    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        edges.push([u, v]);
        adj[u].push(edges.len() - 1);
        adj[v].push(edges.len() - 1);
    }
    
    let num_words = (num_edges + 63) / 64;
    let mut edge_covered: Vec<u64> = vec![0u64; num_words];
    
    let mut selected = vec![false; num_vertices];
    let mut cover: Vec<usize> = Vec::new();
    
    let mut edge_idx: usize = 0;
    while edge_idx < num_edges {
        let word_idx = edge_idx / 64;
        let bit_idx = edge_idx % 64;
        
        if edge_covered[word_idx] & (1u64 << bit_idx) != 0 {
            edge_idx += 1;
            continue;
        }
        
        let [u, v] = edges[edge_idx];
        selected[u] = true;
        selected[v] = true;
        cover.push(u);
        cover.push(v);
        
        for &e2 in &adj[u] {
            let w2 = e2 / 64;
            let b2 = e2 % 64;
            edge_covered[w2] |= 1u64 << b2;
        }
        for &e2 in &adj[v] {
            let w2 = e2 / 64;
            let b2 = e2 % 64;
            edge_covered[w2] |= 1u64 << b2;
        }
    }
    
    cover.sort_unstable();
    cover.dedup();
    
    writeln!(out, "{}", cover.len()).unwrap();
    for v in cover {
        write!(out, "{} ", v).unwrap();
    }
    writeln!(out).unwrap();
}
```
