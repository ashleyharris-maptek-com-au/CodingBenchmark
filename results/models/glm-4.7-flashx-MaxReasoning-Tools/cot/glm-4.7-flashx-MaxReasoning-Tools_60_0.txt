    const { machineCount, tasks } = input;

    const graph = {};
    const inDegree = {};
    const taskData = {};
    
    for (const task of tasks) {
        graph[task.id] = [];
        inDegree[task.id] = 0;
        taskData[task.id] = task;
    }

                graph[dep].push(task.id);
                inDegree[task.id]++;
            }
        }

    const priorityQueue = [];
    
    for (const [id, deg] of Object.entries(inDegree)) {

            priorityQueue.push({ id, remaining: taskData[id].duration });
        }
    }
    
    // Sort by remaining processing time (MRPT)
    priorityQueue.sort((a, b) => a.remaining - b.remaining);
    
    const order = [];
    const remaining = {};
    
    for (const task of tasks) {
        remaining[task.id] = task.duration;
    }
    
    while (priorityQueue.length > 0) {

        const idx = 0;

        remaining[id]--;

            remaining[depId]--;

                // Add to priority queue
                priorityQueue.push({ id: depId, remaining: taskData[depId].duration });
            }
        }

        priorityQueue.sort((a, b) => a.remaining - b.remaining);

            start: startTime,
            machine: bestMachine
