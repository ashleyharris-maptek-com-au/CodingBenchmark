        double epsilon = 1e-10;
        double chiMin = 0;
        double chiMax = 2 * Math.PI;

        double y = 0;
        
        while (chiMax - chiMin > epsilon)
        {
            double chi = (chiMin + chiMax) / 2;
            double sinChi = Math.Sin(chi);
            double cosChi = Math.Cos(chi);
            
            y = r1 + r2 + A * chi * sinChi / (1 - cosChi);
            
            if (y < 0)
            {
                yMin = y;
                chiMin = chi;
            }
            else
            {
                yMax = y;
                chiMax = chi;
            }

        double g = A * Math.Sqrt(y / mu);
        double gDot = 1 - y / r2;

        Vector3 v2 = (gDot * r2Vec - r1Vec) / g;

    public static LambertResult SolveLambert(Vector3 r1, Vector3 v1, Vector3 r2, Vector3 v2, double dtDays)
    {
        const double mu = 1.32712440018e20;
        double dt = dtDays * 86400.0;
        
        double r1_mag = r1.Length();
        double r2_mag = r2.Length();
        
        // Cross product (z-component) for direction
        double crossZ = r1.X * r2.Y - r1.Y * r2.X;
        
        // Calculate transfer angle cosine
        double cos_dnu = Vector3.Dot(r1, r2) / (r1_mag * r2_mag);
        cos_dnu = Math.Max(-1, Math.Min(1, cos_dnu));
        
        // Calculate A
        double A = Math.Sqrt(r1_mag * r2_mag * (1 - cos_dnu) * (1 + cos_dnu));
        if (crossZ < 0) A = -A;
        
        // Universal variable iteration using Newton's method
        double chi = 0.5;
        double y;
        int maxIter = 200;
        double eps = 1e-14;
        
        for (int iter = 0; iter < maxIter; iter++)
        {
            double sinChi = Math.Sin(chi);
            double cosChi = Math.Cos(chi);
            
            double alpha = (chi - sinChi) / (1 - cosChi);
            y = r1_mag + r2_mag + A * sinChi / (1 - cosChi);
            
            if (y <= 0)
            {
                // Adjust chi for next iteration
                chi *= 0.5;
                continue;
            }
            
            double n = Math.Sqrt(mu / y);
            double t_chi = (chi * sinChi + A * alpha) / n;
            
            double dt_error = t_chi - dt;
            
            if (Math.Abs(dt_error) < eps) break;
            
            // Derivative
            double f1 = Math.Sqrt(r1_mag) / Math.Sqrt(r2_mag) * (chi * sinChi + (1 - cosChi) * A * sinChi / (chi * sinChi)) / (1 - cosChi);
            double f2 = chi + A * (chi * sinChi + (1 - cosChi) * sinChi) / (chi * (1 - cosChi));
            double df_dchi = (1 - cosChi) * f1 + (chi - sinChi) * f2 / (1 - cosChi);
            
            chi -= dt_error / df_dchi;
            
            // Clamp chi
            if (chi < 0.001) chi = 0.001;
        }
        
        // Compute f and g functions
        double sinChi = Math.Sin(chi);
        double cosChi = Math.Cos(chi);
        y = r1_mag + r2_mag + A * sinChi / (1 - cosChi);
        
        double f = 1 - y / r1_mag;
        double g = A * Math.Sqrt(y / mu);
        double g_dot = 1 - y / r2_mag;
        
        Vector3 v1 = (r2 - f * r1) / g;
        Vector3 v2 = (g_dot * r2 - r1) / g;
        
        return new LambertResult { V_launch = v1, V_arrival = v2 };
    }
