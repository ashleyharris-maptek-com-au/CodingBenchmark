        # conflict: backtrack
        # pop assignments until assign_index
        while len(assign_stack) > assign_index:
            var, is_true = assign_stack.pop()
            if is_true:
                assignment &= ~(1 << var)
            else:
                assignment |= (1 << var)
            unassigned_mask |= (1 << var)
        # pop decision stack entry
        decision_stack.pop()
