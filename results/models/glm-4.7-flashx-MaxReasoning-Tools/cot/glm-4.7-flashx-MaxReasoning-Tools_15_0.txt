fn adaptive_scale(nv: usize, vertices: &[[f64; 3]]) -> f64 {
    if nv == 0 { return 1.0; }
    
    // Calculate polygon area
    let area = polygon_area(vertices);
    
    // Adjust scale based on polygon size
    // Smaller polygons can use smaller scale, larger polygons need more scale
    let avg_radius = polygon_radius(vertices);
    let scale = 5.0 / (avg_radius + 1e-6);
    scale.min(10.0).max(0.5)

fn adaptive_strategy(nv: usize, scale: f64) -> Strategy {

    Fan,
    Hybrid,
    Grid,

fn polygon_area(vertices: &[[f64; 3]]) -> f64 {
    let mut area = 0.0;
    let n = vertices.len();
    for i in 0..n {
        let j = (i + 1) % n;
        area += vertices[i][0] * vertices[j][1] - vertices[j][0] * vertices[i][1];
    }
    area.abs() / 2.0
}

fn polygon_radius(vertices: &[[f64; 3]]) -> f64 {
    let center = polygon_center(vertices);
    let mut max_dist = 0.0;
    for v in vertices {
        let dist = ((v[0] - center[0]).powi(2) + (v[1] - center[1]).powi(2)).sqrt();
        max_dist = max_dist.max(dist);
    }
    max_dist
}

fn polygon_center(vertices: &[[f64; 3]]) -> [f64; 2] {
    let mut x = 0.0, y = 0.0;
    for v in vertices {
        x += v[0];
        y += v[1];
    }
    [x / vertices.len() as f64, y / vertices.len() as f64]
}

fn generate_tetrahedrons(
    vertices: &[[f64; 3]], 
    sun: &[f64; 3], 
    strategy: Strategy,
    scale: f64,

    match strategy {
        Strategy::Fan => generate_fan(vertices, sun, scale),
        Strategy::Hybrid => generate_hybrid(vertices, sun, scale),
        Strategy::Grid => generate_grid(vertices, sun, scale),
    }
}

fn generate_fan(vertices: &[[f64; 3]], sun: &[f64; 3], scale: f64) -> Vec<Tetrahedron> {
    let mut result = Vec::new();
    let center = polygon_center(vertices);
    
    // Fan triangulation from the center
    let n = vertices.len();
    for i in 1..n-1 {
        let p0 = vertices[i];
        let p1 = vertices[(i + 1) % n];
        
        let tri = [
            [center[0], center[1], 0.0],
            [p0[0], p0[1], 0.0],
            [p1[0], p1[1], 0.0],
        ];
        
        let tetra = create_tetra_from_triangle(&tri, sun, scale);
        result.push(tetra);
    }
    
    result
}

fn generate_hybrid(vertices: &[[f64; 3]], sun: &[f64; 3], scale: f64) -> Vec<Tetrahedron> {
    let mut result = Vec::new();

    }

fn generate_grid(vertices: &[[f64; 3]], sun: &[f64; 3], scale: f64) -> Vec<Tetrahedron> {
    let (min_x, max_x, min_y, max_y) = bounding_box(vertices);
    let width = max_x - min_x;
    let height = max_y - min_y;
    
    // Divide into a grid of triangles
    let cell_size = (width + height) / 4.0; // 4x4 grid
    let nx = ((max_x - min_x) / cell_size + 1.0) as usize;
    let ny = ((max_y - min_y) / cell_size + 1.0) as usize;
    
    let mut result = Vec::new();
    
    for i in 0..nx {
        for j in 0..ny {
            let cx = min_x + i as f64 * cell_size;
            let cy = min_y + j as f64 * cell_size;
            
            // Check if cell is inside polygon
            if point_in_polygon([cx, cy], vertices) {
                // Create two triangles for the cell
                let tri1 = [
                    [cx, cy, 0.0],
                    [cx + cell_size, cy, 0.0],
                    [cx, cy + cell_size, 0.0],
                ];
                
                let tri2 = [
                    [cx + cell_size, cy, 0.0],
                    [cx + cell_size, cy + cell_size, 0.0],
                    [cx, cy + cell_size, 0.0],
                ];
                
                result.push(create_tetra_from_triangle(&tri1, sun, scale));
                result.push(create_tetra_from_triangle(&tri2, sun, scale));
            }
        }
    }
    
    result
}

fn bounding_box(vertices: &[[f64; 3]]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    
    for v in vertices {
        min_x = min_x.min(v[0]);
        max_x = max_x.max(v[0]);
        min_y = min_y.min(v[1]);
        max_y = max_y.max(v[1]);
    }
    
    (min_x, max_x, min_y, max_y)
}

fn point_in_polygon(point: [f64; 2], vertices: &[[f64; 3]]) -> bool {
    let (x, y) = (point[0], point[1]);
    let mut inside = false;
    
    for i in 0..vertices.len() {
        let j = (i + 1) % vertices.len();
        let (x1, y1) = (vertices[i][0], vertices[i][1]);
        let (x2, y2) = (vertices[j][0], vertices[j][1]);
        
        if ((y1 > y) != (y2 > y)) &&
           (x < (x2 - x1) * (y - y1) / (y2 - y1) + x1) {
            inside = !inside;
        }
    }
    
    inside
}

fn triangle_area(tri: &[[f64; 3]]) -> f64 {
    let (x1, y1, _z1) = (tri[0][0], tri[0][1], tri[0][2]);
    let (x2, y2, _z2) = (tri[1][0], tri[1][1], tri[1][2]);
    let (x3, y3, _z3) = (tri[2][0], tri[2][1], tri[2][2]);
    
    ((x2 - x1) * (y3 - y1) - (x3 - x1) * (y2 - y1)).abs() / 2.0

fn calculate_rotation(ref: [[f64; 3]; 3], target: [[f64; 3]; 3]) -> [[f64; 3]; 3] {
    // Calculate normals
    let n1 = normalize(cross_product(&ref[1] - &ref[0], &ref[2] - &ref[0]));
    let n2 = normalize(cross_product(&target[1] - &target[0], &target[2] - &target[0]));
    
    // Calculate edge vectors
    let e1 = ref[1] - ref[0];
    let e2 = ref[2] - ref[0];
    let t1 = target[1] - target[0];
    let t2 = target[2] - target[0];
    
    // First rotation: align edge e1 to t1
    let c = clamp(dot_product(&e1, &t1) / (norm(&e1) * norm(&t1)), -1.0, 1.0);
    let s = (1.0 - c * c).sqrt();
    let cross = cross_product(&e1, &t1);
    
    let kx = cross[0];
    let ky = cross[1];
    let kz = cross[2];
    
    // Rotation matrix R_k
    let r_k = [
        [c + kx * kx * (1.0 - c), kx * ky * (1.0 - c) - kz * s, kx * kz * (1.0 - c) + ky * s],
        [ky * kx * (1.0 - c) + kz * s, c + ky * ky * (1.0 - c), ky * kz * (1.0 - c) - kx * s],
        [kz * kx * (1.0 - c) - ky * s, kz * ky * (1.0 - c) + kx * s, c + kz * kz * (1.0 - c)],
    ];
    
    // Apply R_k to e2, get e2'
    let e2_prime = [
        r_k[0][0] * e2[0] + r_k[0][1] * e2[1] + r_k[0][2] * e2[2],
        r_k[1][0] * e2[0] + r_k[1][1] * e2[1] + r_k[1][2] * e2[2],
        r_k[2][0] * e2[0] + r_k[2][1] * e2[1] + r_k[2][2] * e2[2],
    ];
    
    // Align e2' to t2
    let c2 = clamp(dot_product(&e2_prime, &t2) / (norm(&e2_prime) * norm(&t2)), -1.0, 1.0);
    let s2 = (1.0 - c2 * c2).sqrt();
    let cross2 = cross_product(&e2_prime, &t2);
    
    let kx2 = cross2[0];
    let ky2 = cross2[1];
    let kz2 = cross2[2];
    
    let r_k2 = [
        [c2 + kx2 * kx2 * (1.0 - c2), kx2 * ky2 * (1.0 - c2) - kz2 * s2, kx2 * kz2 * (1.0 - c2) + ky2 * s2],
        [ky2 * kx2 * (1.0 - c2) + kz2 * s2, c2 + ky2 * ky2 * (1.0 - c2), ky2 * kz2 * (1.0 - c2) - kx2 * s2],
        [kz2 * kx2 * (1.0 - c2) - ky2 * s2, kz2 * ky2 * (1.0 - c2) + kx2 * s2, c2 + kz2 * kz2 * (1.0 - c2)],
    ];
    
    // Total rotation: R = R_k2 * R_k
    let mut rot = [[0.0; 3]; 3];
    for i in 0..3 {
        for j in 0..3 {
            rot[i][j] = r_k2[i][0] * r_k[i][j] + r_k2[i][1] * r_k[i][j] + r_k2[i][2] * r_k[i][j];
        }
    }
    
    rot
}

fn matrix_to_quaternion(m: [[f64; 3]; 3]) -> [f64; 4] {
    let (w, x, y, z) = if m[0][0] >= m[1][1] && m[0][0] >= m[2][2] {
        let t = 1.0 + m[0][0] - m[1][1] - m[2][2];
        (t.sqrt() / 2.0, (m[2][1] - m[1][2]) / (4.0 * t.sqrt()),
         (m[0][1] + m[1][0]) / (4.0 * t.sqrt()),
         (m[0][2] + m[2][0]) / (4.0 * t.sqrt()))
    } else if m[1][1] >= m[2][2] {
        let t = 1.0 + m[1][1] - m[0][0] - m[2][2];
        ((m[0][2] - m[2][0]) / (4.0 * t.sqrt()), t.sqrt() / 2.0,
         (m[0][1] + m[1][0]) / (4.0 * t.sqrt()),
         (m[1][2] + m[2][1]) / (4.0 * t.sqrt()))
    } else {
        let t = 1.0 + m[2][2] - m[0][0] - m[1][1];
        ((m[1][0] - m[0][1]) / (4.0 * t.sqrt()),
         (m[0][2] + m[2][0]) / (4.0 * t.sqrt()),
         (m[1][2] + m[2][1]) / (4.0 * t.sqrt()),
         t.sqrt() / 2.0)
    };
    
    [w, x, y, z]
}
