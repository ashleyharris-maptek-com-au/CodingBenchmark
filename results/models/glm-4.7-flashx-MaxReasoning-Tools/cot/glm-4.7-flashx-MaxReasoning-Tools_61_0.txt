function distance(p1: [number, number], p2: [number, number]): number {
  const dx = p1[0] - p2[0];
  const dy = p1[1] - p2[1];
  return Math.sqrt(dx * dx + dy * dy);

// Create routes
function solve(): {routes: Array<Array<string | number>>} {
  const numPackages = packages.length;
  
  // If only 1 truck, visit all packages
  if (truckCount <= 1) {
    // Nearest neighbor algorithm
    const route: Array<string | number> = [];
    let remaining = [...packages];
    
    while (remaining.length > 0) {
      let currentIndex = 0;
      let minDistance = Infinity;
      
      for (let i = 0; i < remaining.length; i++) {
        const d = distance(depot, [remaining[i].x, remaining[i].y]);
        if (d < minDistance) {
          minDistance = d;
          currentIndex = i;
        }
      }
      
      route.push(remaining[currentIndex].id);
      remaining.splice(currentIndex, 1);
    }
    
    // Add depot to start and end
    return {routes: [route]};
  }
  
  // For multiple trucks, use clustering approach
  // Sort packages by distance from depot
  const sortedByDistance = [...packages].sort((a, b) => {
    const da = distance(depot, [a.x, a.y]);
    const db = distance(depot, [b.x, b.y]);
    return da - db;
  });
  
  // Create routes by round-robin assignment
  const routes: Array<Array<string | number>> = Array.from({length: truckCount}, () => []);
  
  for (let i = 0; i < sortedByDistance.length; i++) {
    const truckIndex = i % truckCount;
    routes[truckIndex].push(sortedByDistance[i].id);
  }
  
  return {routes};
}

const data = JSON.parse(input);

const depot = data.depot as [number, number];
const truckCount = data.truckCount as number;
const packages = data.packages as Array<{id: string | number, x: number, y}>;

function distance(p1: [number, number], p2: [number, number]): number {
  const dx = p1[0] - p2[0];
  const dy = p1[1] - p2[1];
  return Math.sqrt(dx * dx + dy * dy);
}

// K-means clustering
function kMeansClustering(packages: typeof packages, k: number): Array<Array<{id: string | number, x: number, y}>> {
  if (packages.length <= k) {
    return packages.map(p => [p]);
  }
  
  // Initialize centroids (first k packages)
  const centroids = packages.slice(0, k).map(p => [p.x, p.y]);
  const packagePoints = packages.map(p => [p.x, p.y]);
  const packageIds = packages.map(p => p.id);
  
  let changed = true;
  while (changed) {
    changed = false;
    
    // Assign packages to nearest centroid
    const assignments = new Array(k).fill(0).map(() => [] as number[]);
    for (let i = 0; i < packagePoints.length; i++) {
      let nearest = 0;
      let minDist = Infinity;
      for (let j = 0; j < centroids.length; j++) {
        const d = distance(packagePoints[i], centroids[j]);
        if (d < minDist) {
          minDist = d;
          nearest = j;
        }
      }
      assignments[nearest].push(i);
    }
    
    // Update centroids
    const newCentroids = new Array(k).fill(null).map(() => [0, 0] as [number, number]);
    const counts = new Array(k).fill(0);
    
    for (let j = 0; j < k; j++) {
      for (const i of assignments[j]) {
        newCentroids[j][0] += packagePoints[i][0];
        newCentroids[j][1] += packagePoints[i][1];
        counts[j]++;
      }
    }
    
    for (let j = 0; j < k; j++) {
      if (counts[j] > 0) {
        newCentroids[j][0] /= counts[j];
        newCentroids[j][1] /= counts[j];
      }
    }
    
    // Check if centroids changed
    for (let j = 0; j < k; j++) {
      if (distance(newCentroids[j], centroids[j]) > 1e-6) {
        changed = true;
        break;
      }
    }
    
    centroids.splice(0, k, ...newCentroids);
  }
  
  // Group packages by assignments
  const result: Array<Array<{id: string | number, x: number, y}>> = new Array(k).fill(null).map(() => []);
  for (let j = 0; j < k; j++) {
    for (const i of assignments[j]) {
      result[j].push({id: packageIds[i], x: packagePoints[i][0], y: packagePoints[i][1]});
    }
  }
  
  return result;
}

// Nearest neighbor TSP solver
function nearestNeighbor(routePoints: [number, number][], routeIds: (string | number)[], start: [number, number]): (string | number)[] {
  const result = [...routeIds];
  const remaining = new Set(routePoints.map((_, i) => i));
  
  let currentIndex = start;
  let resultIndex = 0;
  
  while (remaining.size > 0) {
    let nearest = -1;
    let minDist = Infinity;
    
    for (const i of remaining) {
      const d = distance(currentIndex, routePoints[i]);
      if (d < minDist) {
        minDist = d;
        nearest = i;
      }
    }
    
    if (nearest >= 0) {
      result[resultIndex++] = routeIds[nearest];
      remaining.delete(nearest);
      currentIndex = routePoints[nearest];
    } else {
      break;
    }
  }
  
  return result;
}

function solve(): {routes: Array<Array<string | number>>} {
  const numPackages = packages.length;
  
  // K-means clustering
  const clusters = kMeansClustering(packages, truckCount);
  
  // Build routes
  const routes: Array<Array<string | number>> = [];
  for (const cluster of clusters) {
    if (cluster.length === 0) continue;
    
    const clusterIds = cluster.map(p => p.id);
    const clusterPoints = cluster.map(p => [p.x, p.y]);
    
    // Find starting point (nearest to depot)
    let minDist = Infinity;
    let startIdx = 0;
    for (let i = 0; i < clusterPoints.length; i++) {
      const d = distance(depot, clusterPoints[i]);
      if (d < minDist) {
        minDist = d;
        startIdx = i;
      }
    }
    
    // Run nearest neighbor
    const route = nearestNeighbor(clusterPoints, clusterIds, depot);
    routes.push(route);
  }
  
  return {routes};
}

  }

}

function solve(): {routes: Array<Array<string | number>>} {
  const numPackages = packages.length;
  
  // Edge case: no packages
  if (numPackages === 0) {
    return {routes: []};
  }
  
  // Edge case: more trucks than packages
  if (truckCount >= numPackages) {
    return {routes: packages.map(p => [p.id])};
  }
  
  // Sort packages by distance from depot
  const sortedByDistance = [...packages].sort((a, b) => 
    distanceToDepot(a.x, a.y) - distanceToDepot(b.x, b.y)
  );
  
  // Create routes by round-robin assignment
  const routes: Array<Array<string | number>> = Array.from({length: truckCount}, () => []);
  
  for (let i = 0; i < sortedByDistance.length; i++) {
    const truckIndex = i % truckCount;
    routes[truckIndex].push(sortedByDistance[i].id);
  }
  
  // Optimize each route using nearest neighbor
  for (let t = 0; t < routes.length; t++) {
    const route = routes[t];
    if (route.length <= 1) continue;
    
    const pointIds = route.map(id => {
      const p = packages.find(pkg => pkg.id === id)!;
      return {id, x: p.x, y: p.y};
    });
    
    // Find starting point (nearest to depot)
    let minDist = Infinity;
    let startIdx = 0;
    for (let i = 0; i < pointIds.length; i++) {
      const d = distanceToDepot(pointIds[i].x, pointIds[i].y);
      if (d < minDist) {
        minDist = d;
        startIdx = i;
      }
    }
    
    // Nearest neighbor TSP
    const visited = new Set<number>();
    const newRoute: Array<string | number> = [];
    let currentIndex = depot;
    let currentIdx = startIdx;
    
    while (visited.size < pointIds.length) {
      visited.add(currentIdx);
      newRoute.push(pointIds[currentIdx].id);
      
      let nearestIdx = -1;
      let minDistToCurrent = Infinity;
      
      for (let i = 0; i < pointIds.length; i++) {
        if (!visited.has(i)) {
          const d = distance(currentIndex, [pointIds[i].x, pointIds[i].y]);
          if (d < minDistToCurrent) {
            minDistToCurrent = d;
            nearestIdx = i;
          }
        }
      }
      
      if (nearestIdx >= 0) {
        currentIndex = [pointIds[nearestIdx].x, pointIds[nearestIdx].y];
        currentIdx = nearestIdx;
      } else {
        break;
      }
    }
    
    routes[t] = newRoute;
  }
  
  return {routes};
}
