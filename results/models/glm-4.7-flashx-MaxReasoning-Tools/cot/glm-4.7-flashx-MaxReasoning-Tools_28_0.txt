        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }
    
    // Create adjacency lists
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); num_vertices];
    for (i, &(u, v)) in edges.iter().enumerate() {
        adj[u].push(i);
        adj[v].push(i);

        if edge_covered[current_edge] {
            current_edge += 1;
            continue;
        }

        selected[u] = true;
        selected[v] = true;

        for &edge_idx in &adj[v] {
            edge_covered[edge_idx] = true;
        }
        
        current_edge += 1;

    // Output
    writeln!(out, "{}", cover.len()).unwrap();
