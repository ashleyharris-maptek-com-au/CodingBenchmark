```hlsl
[[vk::location(0)]] struct PSInput {
    float3 worldPos : TEXCOORD0;
    float3 normal   : TEXCOORD1;
    float2 uv       : TEXCOORD2;
    float3 tangent  : TEXCOORD3;
    float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]] cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    // Light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    // Lambertian diffuse term
    float d = max(dot(N, L), 0.0);

    // Quantize into toon bands
    float b1 = step(0.75, d);
    float b2 = step(0.50, d) - b1;
    float b3 = step(0.25, d) - b2 - b1;
    float b0 = 1.0 - b1 - b2 - b3;

    // Band intensities
    float intensity = b1 * 1.0 + b2 * 0.7 + b3 * 0.4 + b0 * 0.2;

    // Apply to vertex color
    float4 finalColor = float4(input.color * intensity, 1.0);

    return finalColor;
}
```
