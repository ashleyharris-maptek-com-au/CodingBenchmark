typedef long long ll;
typedef double ld;

struct City {
    ld x, y;
    City(ld x = 0, ld y = 0) : x(x), y(y) {}
};

static inline ld dist2(const City& a, const City& b) {
    ld dx = a.x - b.x;
    ld dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline ld total_dist(const vector<int>& tour, const vector<City>& cities) {
    ld sum = 0;
    int n = tour.size();
    for (int i = 0; i < n; ++i) {
        sum += dist2(cities[tour[i]], cities[tour[(i + 1) % n]]);
    }
    return sum;
}

static inline vector<int> nearest_neighbor(const vector<City>& cities, int start = 0) {
    int n = cities.size();
    vector<int> order(n);
    vector<char> visited(n, 0);
    
    order[0] = start;
    visited[start] = 1;
    int current = start;
    
    for (int i = 1; i < n; ++i) {
        int best = -1;
        ld bestd2 = numeric_limits<ld>::infinity();
        
        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                ld d = dist2(cities[current], cities[j]);
                if (d < bestd2) {
                    bestd2 = d;
                    best = j;
                }
            }
        }
        visited[best] = 1;
        order[i] = best;
        current = best;
    }
    
    return order;
}

static inline void two_opt(vector<int>& order, const vector<City>& cities) {
    int n = order.size();
    if (n < 3) return;
    
    bool improved = true;
    while (improved) {
        improved = false;
        
        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 2; j < n; ++j) {
                ld oldd = dist2(cities[order[i]], cities[order[i + 1]]) + 
                         dist2(cities[order[j]], cities[order[(j + 1) % n]]);
                ld newd = dist2(cities[order[i]], cities[order[j]]) + 
                         dist2(cities[order[i + 1]], cities[order[(j + 1) % n]]);
                
                if (newd < oldd) {
                    int l = i + 1, r = j;
                    while (l < r) swap(order[l++], order[r--]);
                    improved = true;
                }
            }
        }
    }
}

static inline void two_opt_random(vector<int>& order, const vector<City>& cities, int iters = 1000) {
    int n = order.size();
    if (n < 3) return;
    
    random_device rd;
    mt19937 gen(rd());
    uniform_int_distribution<int> rng(0, n - 1);
    
    for (int t = 0; t < iters * n; ++t) {
        int i = rng(gen);
        int j = rng(gen);
        
        if (i > j) swap(i, j);
        if (i + 1 >= j) continue;
        
        ld oldd = dist2(cities[order[i]], cities[order[i + 1]]) + 
                 dist2(cities[order[j]], cities[order[(j + 1) % n]]);
        ld newd = dist2(cities[order[i]], cities[order[j]]) + 
                 dist2(cities[order[i + 1]], cities[order[(j + 1) % n]]);
        
        if (newd < oldd) {
            int l = i + 1, r = j;
            while (l < r) swap(order[l++], order[r--]);
        }
    }
}

static inline vector<int> exact_brute(const vector<City>& cities) {
    int n = cities.size();
    if (n == 1) return {0};
    if (n == 2) return {0, 1};
    
    vector<int> best(n), perm(n - 1);
    for (int i = 0; i < n - 1; ++i) perm[i] = i + 1;
    
    ld bestd = numeric_limits<ld>::infinity();
    do {
        for (int i = 0; i < n - 1; ++i) best[i + 1] = perm[i];
        ld d = total_dist(best, cities);
        if (d < bestd) {
            bestd = d;
            for (int i = 0; i < n; ++i) best[i] = perm[i - 1 < 0 ? n - 1 : i - 1];
        }
    } while (next_permutation(perm.begin(), perm.end()));
    
    return best;
}

    int n = cities.size();
    
    if (n <= 1) return {0};

    if (n <= 10) return exact_brute(cities);
