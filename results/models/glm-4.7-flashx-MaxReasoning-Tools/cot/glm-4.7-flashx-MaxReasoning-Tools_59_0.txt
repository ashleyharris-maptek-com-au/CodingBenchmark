```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob = data.probability; // 2D array
const drones = data.drones; // array of {id, x, y}
const maxStepsPerDrone = data.maxStepsPerDrone || new Array(drones.length).fill(Infinity);
// if maxStepsPerDrone is number, treat as uniform

let paths = [];

const rows = prob.length;
if (rows === 0) {
  console.log(JSON.stringify({paths: []}));
  process.exit(0);
}
const cols = prob[0].length;

// Directions: up, down, left, right (order for tie-breaking)
const dirs = [[-1, 0], [1, 0], [0, -1], [0, 1]];

for (let i = 0; i < drones.length; i++) {
  let startX = drones[i].x;
  let startY = drones[i].y;
  // Ensure within bounds
  if (startX < 0 || startX >= rows || startY < 0 || startY >= cols) {
    paths.push([]);
    continue;
  }
  // If start cell is obstacle, maybe treat as invalid
  if (prob[startX][startY] === -1) {
    paths.push([]);
    continue;
  }

  let stepsLeft = typeof maxStepsPerDrone === 'number' ? maxStepsPerDrone : (maxStepsPerDrone[i] || 0);
  let x = startX;
  let y = startY;
  let path = [[x, y]];

  while (stepsLeft > 0) {
    let bestVal = -Infinity;
    let bestIdx = -1;

    for (let d = 0; d < dirs.length; d++) {
      let nx = x + dirs[d][0];
      let ny = y + dirs[d][1];
      if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && prob[nx][ny] !== -1) {
        let val = prob[nx][ny];
        if (val > bestVal || (val === bestVal && d < bestIdx)) {
          bestVal = val;
          bestIdx = d;
        }
      }
    }

    if (bestIdx !== -1) {
      let dx = dirs[bestIdx][0];
      let dy = dirs[bestIdx][1];
      x += dx;
      y += dy;
      path.push([x, y]);
      stepsLeft--;
    } else {
      // No valid neighbor: stay in place; break
      break;
    }
  }

  paths.push(path);
}

console.log(JSON.stringify({paths}));
```
