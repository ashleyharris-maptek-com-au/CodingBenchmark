    static List<(int x1, int y1, int x2, int y2)> FindPathToOrigin(int startX, int startY, int startZ, 
        int xCount, int yCount, int zCount)
    {
        // A* search with slope constraint
        // State: (x, y, z, direction)
        // Direction to avoid backtracking: 0=up, 1=right, 2=down, 3=left
        
        var openSet = new PriorityQueue<(int x, int y, int z, int dir), int>();
        var cameFrom = new Dictionary<(int x, int y, int z, int dir), (int x, int y, int z, int dir)>();
        var gScore = new Dictionary<(int x, int y, int z, int dir), int>();
        
        // 4 directions (excluding direction 0 to prevent immediate reversal)
        // We use: 0=up in Z, 1=right in X, 2=down in Z, 3=left in X
        // Actually, let's use standard 4-connectivity in XY + Z moves
        
        // Initial state
        openSet.Enqueue((startX, startY, startZ, -1), 0);
        gScore[(startX, startY, startZ, -1)] = 0;
        
        var visited = new HashSet<(int x, int y, int z, int dir)>();
        
        while (openSet.Count > 0)
        {
            var current = openSet.Dequeue();
            
            if (current.x == 0 && current.y == 0)
            {
                // Reconstruct path
                var path = new List<(int x1, int y1, int x2, int y2)>();
                var curr = current;
                while (cameFrom.ContainsKey(curr))
                {
                    var prev = cameFrom[curr];
                    path.Add((prev.x, prev.y, curr.x, curr.y));
                    curr = prev;
                }
                path.Reverse();
                return path;
            }
            
            visited.Add(current);
            
            // Generate neighbors
            var neighbors = new List<(int x, int y, int z, int dir, int cost)>();
            
            // XY moves (4-connected)
            // Up in Y (positive Y)
            if (current.y < yCount - 1 && current.dir != 3)
            {
                int nx = current.x;
                int ny = current.y + 1;
                int dz = 0;
                int cost = 1; // Flat movement
                neighbors.Add((nx, ny, current.z, 1, cost));
            }
            
            // Down in Y (negative Y)
            if (current.y > 0 && current.dir != 1)
            {
                int nx = current.x;
                int ny = current.y - 1;
                int dz = 0;
                int cost = 1; // Flat movement
                neighbors.Add((nx, ny, current.z, 3, cost));
            }
            
            // Up in X (positive X)
            if (current.x < xCount - 1 && current.dir != 3)
            {
                int nx = current.x + 1;
                int ny = current.y;
                int dz = 0;
                int cost = 1; // Flat movement
                neighbors.Add((nx, ny, current.z, 2, cost));
            }
            
            // Down in X (negative X)
            if (current.x > 0 && current.dir != 2)
            {
                int nx = current.x - 1;
                int ny = current.y;
                int dz = 0;
                int cost = 1; // Flat movement
                neighbors.Add((nx, ny, current.z, 0, cost));
            }
            
            // Z moves (up and down, slope constrained)
            // Only allow Z change if it doesn't violate slope constraint
            // |dz| <= |dx| + |dy| = 1 for XY steps
            
            // Up in Z
            if (current.z < zCount - 1)
            {
                int cost = 5; // Uphill movement
                neighbors.Add((current.x, current.y, current.z + 1, 0, cost));
            }
            
            // Down in Z
            if (current.z > 0)
            {
                int cost = 0; // Downhill movement
                neighbors.Add((current.x, current.y, current.z - 1, 1, cost));
            }
            
            foreach (var (nx, ny, nz, ndir, ncost) in neighbors)
            {
                if (visited.Contains((nx, ny, nz, ndir))) continue;
                
                int newG = gScore[current] + ncost;
                
                if (!gScore.ContainsKey((nx, ny, nz, ndir)) || newG < gScore[(nx, ny, nz, ndir)])
                {
                    gScore[(nx, ny, nz, ndir)] = newG;
                    int h = Math.Abs(nx - 0) + Math.Abs(ny - 0) + Math.Abs(nz - 0); // Manhattan heuristic
                    int f = newG + h;
                    openSet.Enqueue((nx, ny, nz, ndir), f);
                    cameFrom[(nx, ny, nz, ndir)] = current;
                }
            }
        }
        
        // No path found
        return null;
    }

// Priority queue for A*
class PriorityQueue<T, TPriority> where TPriority : IComparable<TPriority>
{
    private readonly List<(T Item, TPriority Priority)> _items = new List<(T, TPriority)>();
    
    public void Enqueue(T item, TPriority priority)
    {
        _items.Add((item, priority));
    }
    
    public T Dequeue()
    {
        var bestIndex = 0;
        for (int i = 1; i < _items.Count; i++)
        {
            if (_items[i].Priority.CompareTo(_items[bestIndex].Priority) < 0)
            {
                bestIndex = i;
            }
        }
        var best = _items[bestIndex];
        _items.RemoveAt(bestIndex);
        return best.Item;
    }
    
    public int Count => _items.Count;
}
