   function solve() {
       const fs = require('fs');
       const input = fs.readFileSync(0, 'utf8');
       const data = JSON.parse(input);

       const width = data.width;
       const height = data.height;
       const grid = data.grid; // array of strings
       const ignitions = data.ignitions; // array of [x,y]
       const towns = data.towns; // array of {id,x,y,value}
       const cutBudget = data.cutBudget;

       const N = width * height;
       const cut = new Uint8Array(N); // 0 uncut, 1 cut
       const type = new Uint8Array(N); // 0 '.' forest, 1 '~' water/rock

       // Build town map: key "x,y" -> value
       const townMap = new Map();
       for (const t of towns) {
           townMap.set(`${t.x},${t.y}`, t.value);
       }

       // Build ignition set
       const ignitionSet = new Set();
       for (const [x, y] of ignitions) {
           const idx = y * width + x;
           ignitionSet.add(idx);
       }

       // adjacency list for forest cells
       const adj = [];
       for (let y = 0; y < height; y++) {
           for (let x = 0; x < width; x++) {
               const idx = y * width + x;
               if (grid[y][x] === '.') {
                   type[idx] = 0;
                   adj[idx] = [];
                   // neighbors: N, S, E, W
                   if (x > 0) adj[idx].push((y) * width + (x - 1));
                   if (x < width - 1) adj[idx].push((y) * width + (x + 1));
                   if (y > 0) adj[idx].push((y - 1) * width + x);
                   if (y < height - 1) adj[idx].push((y + 1) * width + x);
               } else {
                   type[idx] = 1; // water/rock
                   adj[idx] = [];
               }
           }
       }

       // for each component, collect node indices
       const visitedComp = new Uint8Array(N); // 0 unvisited, 1 visited

       const firebreaks = [];

       // helper to get neighbors for given node based on current cut status
       function getNeighbors(idx) {
           const list = [];
           const x = idx % width;
           const y = Math.floor(idx / width);
           if (x > 0 && !cut[idx - 1]) list.push(idx - 1);
           if (x < width - 1 && !cut[idx + 1]) list.push(idx + 1);
           if (y > 0 && !cut[idx - width]) list.push(idx - width);
           if (y < height - 1 && !cut[idx + width]) list.push(idx + width);
           return list;
       }

       // recursion with increased stack size
       const v8 = require('v8');
       v8.setFlagsFromString('--max-call-stack-size=200000');

       // For each iteration, compute components reachable from ignitions.
       while (cutBudget > 0) {
           // Reset visited for this iteration
           visitedComp.fill(0);
           const queue = [];
           // BFS from all ignitions
           for (const idx of ignitionSet) {
               if (!cut[idx]) {
                   if (visitedComp[idx] === 0) {
                       visitedComp[idx] = 1;
                       queue.push(idx);
                   }
               }
           }
           // BFS
           while (queue.length > 0) {
               const u = queue.shift();
               for (const v of getNeighbors(u)) {
                   if (visitedComp[v] === 0) {
                       visitedComp[v] = 1;
                       queue.push(v);
                   }
               }
           }

           // collect components that have both ignition and towns (i.e., reachable from ignition and contain towns)
           const components = [];
           for (let idx = 0; idx < N; idx++) {
               if (visitedComp[idx] === 1 && type[idx] === 0 && !cut[idx]) {
                   // start a new component
                   const comp = [];
                   const stack = [idx];
                   visitedComp[idx] = 2; // mark as in comp list
                   while (stack.length > 0) {
                       const u = stack.pop();
                       comp.push(u);
                       for (const v of getNeighbors(u)) {
                           if (visitedComp[v] === 1 && type[v] === 0 && !cut[v]) {
                               visitedComp[v] = 2;
                               stack.push(v);
                           }
                       }
                   }
                   components.push(comp);
               }
           }

           let bestGain = -1;
           let bestIdx = -1;

           // Process each component
           for (const comp of components) {
               if (comp.length === 0) continue;

               // Build adjacency for component (neighbors within component, not cut)
               // Since we already have adjacency from earlier (global) but we must respect cut.
               // We'll build component adjacency on the fly.
               // For each node, find neighbors among comp.
               // We'll use a map node->neighbors list (indices)
               const nodeAdj = new Array(comp.length);
               for (let i = 0; i < comp.length; i++) {
                   const u = comp[i];
                   const nb = [];
                   for (const v of adj[u]) {
                       // v is neighbor of u (global adjacency list). Need to check if v is within comp and not cut.
                       // Since we only have adjacency list from original '.' cells, v could be cut now (blocked). So skip if cut.
                       if (!cut[v]) {
                           // find v index in comp
                           const vi = comp.indexOf(v); // O(n) per neighbor -> O(E) overall. This could be expensive but okay for moderate components.
                           if (vi !== -1) {
                               nb.push(vi);
                           }
                       }
                   }
                   nodeAdj[i] = nb;
               }

               // Tarjan recursion
               const disc = new Int32Array(comp.length);
               const low = new Int32Array(comp.length);
               const parent = new Int32Array(comp.length).fill(-1);
               const subtreeVal = new Int32Array(comp.length);
               const hasIgn = new Uint8Array(comp.length);
               const artic = new Uint8Array(comp.length);
               let timer = 0;

               // town value per node
               const townVal = new Int32Array(comp.length);
               for (let i = 0; i < comp.length; i++) {
                   const idx = comp[i];
                   const key = `${idx % width},${Math.floor(idx / width)}`;
                   townVal[i] = townMap.get(key) || 0;
               }

               // ignition presence per node
               const isIgn = new Uint8Array(comp.length);
               for (let i = 0; i < comp.length; i++) {
                   const idx = comp[i];
                   isIgn[i] = ignitionSet.has(idx) ? 1 : 0;
               }

               // Recursion function
               function dfs(u, p) {
                   disc[u] = low[u] = timer++;
                   subtreeVal[u] = townVal[u];
                   hasIgn[u] = isIgn[u];
                   let children = 0;
                   for (const v of nodeAdj[u]) {
                       if (v === p) continue;
                       if (disc[v] === -1) {
                           dfs(v, u);
                           low[u] = Math.min(low[u], low[v]);
                           subtreeVal[u] += subtreeVal[v];
                           hasIgn[u] = hasIgn[u] || hasIgn[v];
                           children++;
                           if (p === -1 && children > 1) artic[u] = 1;
                           if (p !== -1 && low[v] >= disc[u]) artic[u] = 1;
                       } else if (disc[v] < disc[u]) {
                           low[u] = Math.min(low[u], disc[v]);
                       }
                   }
               }

               // Run DFS from the first node (comp[0]) as root
               dfs(0, -1);

               // Compute gains for articulation points
               for (let u = 0; u < comp.length; u++) {
                   if (artic[u] === 0) continue;
                   // Determine which child subtrees are separated
                   // For each neighbor v that is not parent:
                   // For root, parent = -1, so consider all neighbors not parent.
                   // For non-root, consider neighbors where (parent != -1 && low[v] >= disc[u]) or (p == -1).
                   // Need to identify children nodes (v where parent is u) not just any neighbor.
                   // But we need to know children list: those v where parent[v] == u.
                   // We can get child list using nodeAdj and parent array.
                   for (const v of nodeAdj[u]) {
                       if (v === parent[u]) continue; // skip parent
                       if (parent[u] === -1) {
                           // root: consider all children (v where parent[v] == u)
                           if (parent[v] === u) {
                               if (!hasIgn[v]) {
                                   gain = subtreeVal[v];
                                   if (gain > bestGain) {
                                       bestGain = gain;
                                       bestIdx = comp[u]; // original index
                                   }
                               }
                           }
                       } else {
                           if (low[v] >= disc[u] && parent[v] === u) {
                               if (!hasIgn[v]) {
                                   gain = subtreeVal[v];
                                   if (gain > bestGain) {
                                       bestGain = gain;
                                       bestIdx = comp[u];
                                   }
                               }
                           }
                       }
                   }
               }
           }

           if (bestGain <= 0) {
               break; // no positive gain, stop
           }
           // Cut the cell
           firebreaks.push([bestIdx % width, Math.floor(bestIdx / width)]);
           cut[bestIdx] = 1;
           cutBudget--;
       }

       // Output JSON
       console.log(JSON.stringify({ firebreaks }));
   }
