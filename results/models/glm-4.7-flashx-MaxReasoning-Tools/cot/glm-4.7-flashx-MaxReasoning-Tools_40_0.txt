class BIT {
public:
    vector<int64_t> tree;
    int n;
    BIT(int n) : n(n), tree(n+1, 0) {}
    void add(int idx, int64_t val) {
        for (++idx; idx <= n; idx += idx & -idx) tree[idx] += val;
    }
    int64_t sum(int idx) { // [0..idx] inclusive
        int64_t res = 0;
        for (++idx; idx > 0; idx -= idx & -idx) res += tree[idx];
        return res;
    }
    int64_t range(int l, int r) { return sum(r) - (l>0 ? sum(l-1) : 0); }
};
