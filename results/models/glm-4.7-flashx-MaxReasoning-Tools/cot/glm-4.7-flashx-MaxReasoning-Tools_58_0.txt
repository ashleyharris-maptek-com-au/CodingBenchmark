function computeDistances(grid, jobs, robots) {
   const rows = grid.length;
   const cols = grid[0].length;
   const jobMap = new Map();
   for (let i=0; i<jobs.length; i++) {
       const job = jobs[i];
       jobMap.set(`${job.x},${job.y}`, i);
   }
   const R = robots.length;
   const J = jobs.length;
   const dist = new Array(R).fill().map(() => new Array(J).fill(-1)); // -1 means not reached

   for (let r = 0; r < R; r++) {
       const robot = robots[r];
       const visited = new Int8Array(rows * cols);
       const queue = [];
       let head = 0;
       const sx = robot.x, sy = robot.y;
       const startIdx = sy*cols + sx;
       visited[startIdx] = 1;
       queue.push(startIdx, 0); // store distance later

       let found = 0;
       while (head < queue.length) {
           const idx = queue[head++];
           const d = queue[head++];
           // compute x,y from idx
           const x = idx % cols;
           const y = Math.floor(idx / cols);
           // check if job at this cell
           const jobKey = `${x},${y}`;
           const jobIdx = jobMap.get(jobKey);
           if (jobIdx !== undefined && dist[r][jobIdx] === -1) {
               dist[r][jobIdx] = d;
               found++;
               if (found === J) break; // all jobs assigned
           }

           // neighbors
           const nx = x+1; if (nx < cols && grid[y][nx] !== '#') { const nIdx = y*cols+nx; if (visited[nIdx] === 0) { visited[nIdx]=1; queue.push(nIdx, d+1); } }

       }
   }

class MinHeap {
    constructor() {
        this.heap = [];
    }
    push(val) { this.heap.push(val); this._siftUp(this.heap.length - 1); }
    pop() {
        if (this.heap.length === 0) return null;
        const top = this.heap[0];
        const last = this.heap.pop();
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this._siftDown(0);
        }
        return top;
    }
    _siftUp(i) {
        while (i > 0) {
            const parent = Math.floor((i - 1) / 2);
            if (this.heap[parent][0] <= this.heap[i][0]) break;
            [this.heap[parent], this.heap[i]] = [this.heap[i], this.heap[parent]];
            i = parent;
        }
    }
    _siftDown(i) {
        const n = this.heap.length;
        while (true) {
            let left = 2*i + 1;
            let right = 2*i + 2;
            let smallest = i;
            if (left < n && this.heap[left][0] < this.heap[smallest][0]) smallest = left;
            if (right < n && this.heap[right][0] < this.heap[smallest][0]) smallest = right;
            if (smallest === i) break;
            [this.heap[i], this.heap[smallest]] = [this.heap[smallest], this.heap[i]];
            i = smallest;
        }
    }
    size() { return this.heap.length; }
    peek() { return this.heap[0]; }
}

   - `jobs`: array of objects with `id`, `x`, `y`, `priority`

2. Output: JSON with `{"assignments": [[jobId, jobId, ...], ...]}` where array index = robot id

3. Each job must appear exactly once

4. The grader uses shortest grid walking distance through open cells

5. The reference solution is a deterministic greedy algorithm

Let me think about the approach:

For this problem, we need to assign each job to a robot. The cost of assigning a job to a robot is the shortest grid distance from the robot to the job.

A simple approach that works well for this type of problem is to:
1. Compute the distance from each robot to each job
2. Use a greedy assignment based on some criterion

For the greedy reference, a common approach is:
- Sort jobs by priority (descending)
- For each job, assign it to the nearest available robot
- If no robot is available (all are busy), wait for a robot

const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

// Distance from robot to job
function computeDistance(robot, job) {
    let dx = Math.abs(robot.x - job.x);
    let dy = Math.abs(robot.y - job.y);
    // Manhattan distance through open cells, accounting for obstacles
    // This is a simplified approach - in reality, we'd need A* or BFS
    
    // BFS for actual shortest path
    const rows = grid.length;
    const cols = grid[0].length;
    
    const start = { x: robot.x, y: robot.y };
    const target = { x: job.x, y: job.y };
    
    // BFS
    const queue = [{ x: start.x, y: start.y, dist: 0 }];
    const visited = new Set();
    visited.add(`${start.x},${start.y}`);
    
    while (queue.length > 0) {
        const { x, y, dist } = queue.shift();
        
        if (x === target.x && y === target.y) {
            return dist;
        }
        
        const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        
        for (const [dx, dy] of directions) {
            const nx = x + dx;
            const ny = y + dy;
            
            if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.' && !visited.has(`${nx},${ny}`)) {
                visited.add(`${nx},${ny}`);
                queue.push({ x: nx, y: ny, dist: dist + 1 });
            }
        }
    }
    
    return -1; // No path found
}

// Greedy assignment
function assignJobs(robots, jobs) {
    const numRobots = robots.length;
    const assignments = Array.from({ length: numRobots }, () => []);
    const jobStatus = new Array(jobs.length).fill(false);
    
    // Sort jobs by priority descending
    const sortedJobs = [...jobs].sort((a, b) => b.priority - a.priority);
    
    for (const job of sortedJobs) {
        let bestRobot = -1;
        let bestDistance = Infinity;
        
        for (let i = 0; i < numRobots; i++) {
            if (assignments[i].length > 0) {
                // Robot is busy - compute distance from job to next unassigned job
                const nextJob = assignments[i][assignments[i].length - 1];
                const dist = computeDistance(jobs[nextJob], job);
                if (dist < bestDistance) {
                    bestDistance = dist;
                    bestRobot = i;
                }
            } else {
                // Robot is idle
                const dist = computeDistance(robots[i], job);
                if (dist < bestDistance) {
                    bestDistance = dist;
                    bestRobot = i;
                }
            }
        }
        
        if (bestRobot !== -1) {
            assignments[bestRobot].push(job.id);
        }
    }
    
    return assignments;
}

const result = { assignments: assignJobs(robots, jobs) };
