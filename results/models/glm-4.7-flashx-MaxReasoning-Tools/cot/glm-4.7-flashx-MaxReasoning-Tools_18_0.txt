// Karger's randomized algorithm for larger graphs
pair<int, vector<pair<int, int>>> karger(int n, const vector<vector<int>>& adj) {
    random_device rd;
    mt19937 gen(rd());
    
    vector<int> parent(n);
    for (int i = 0; i < n; i++) parent[i] = i;
    
    function<int(int)> find = [&](int x) {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    };
    
    vector<pair<int, int>> edges;
    for (int u = 0; u < n; u++) {
        for (int v : adj[u]) {
            if (u < v) {
                edges.emplace_back(u, v);
            }
        }
    }
    
    while (n > 2) {
        if (edges.empty()) break;
        
        uniform_int_distribution<> dis(0, edges.size() - 1);
        auto [u, v] = edges[dis(gen)];
        
        int root_u = find(u);
        int root_v = find(v);
        if (root_u == root_v) continue;
        
        parent[root_v] = root_u;
        
        // Remove edges that are within the same component
        auto it = remove_if(edges.begin(), edges.end(), [&](const auto& e) {
            return find(e.first) == find(e.second);
        });
        edges.erase(it, edges.end());
        
        n--;
    }
    
    vector<pair<int, int>> cut_edges;
    for (const auto& [u, v] : edges) {
        if (find(u) != find(v)) {
            cut_edges.emplace_back(u, v);
        }
    }
    
    return {static_cast<int>(cut_edges.size()), cut_edges};
}

vector<pair<int, int>> find_bridges(int n, const vector<vector<int>>& adj) {
    vector<int> disc(n, -1);
    vector<int> low(n, -1);
    vector<bool> visited(n, false);
    vector<pair<int, int>> bridges;
    int time = 0;
    
    function<void(int, int)> dfs = [&](int u, int parent) {
        disc[u] = low[u] = ++time;
        visited[u] = true;
        
        for (int v : adj[u]) {
            if (v == parent) continue;
            if (disc[v] == -1) {
                dfs(v, u);
                low[u] = min(low[u], low[v]);
                if (low[v] > disc[u]) {
                    bridges.emplace_back(u, v);
                }
            } else {
                low[u] = min(low[u], disc[v]);
            }
        }

            dfs(i, -1);
        }
    }

Actually, I realize the issue. When we merge vertices, we're not actually "removing" the second_last vertex. We're just updating the vertex_index array. The in_A array is reset at the beginning of each phase, so it doesn't matter that we're using the same vertex indices.

The key insight is: in each phase, we add vertices to A one by one, starting from any vertex. When we reach the end, we have two vertices (s and t) that are connected by many edges to A. These edges form the cut. After recording the cut, we merge s and t.

When we merge s and t, we update the weights. The in_A array is reset for the next phase. The algorithm continues with the contracted graph.

At the end, we've found the minimum cut of the original graph. The edges we recorded are from the original graph because we're using the original vertex indices.

pair<int, vector<pair<int, int>>> karger(int n, const vector<vector<int>>& adj) {
    random_device rd;
    mt19937 gen(rd());
    
    vector<int> parent(n);
    for (int i = 0; i < n; i++) parent[i] = i;
    
    function<int(int)> find = [&](int x) {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    };
    
    vector<pair<int, int>> edges;
    for (int u = 0; u < n; u++) {
        for (int v : adj[u]) {
            if (u < v) {
                edges.emplace_back(u, v);
            }
        }
    }
    
    while (n > 2) {
        if (edges.empty()) break;
        
        uniform_int_distribution<> dis(0, edges.size() - 1);
        auto [u, v] = edges[dis(gen)];
        
        int root_u = find(u);
        int root_v = find(v);
        if (root_u == root_v) continue;
        
        parent[root_v] = root_u;
        
        edges.erase(remove_if(edges.begin(), edges.end(), [&](const auto& e) {
            return find(e.first) == find(e.second);
        }), edges.end());
        
        n--;
    }
    
    vector<pair<int, int>> cut_edges;
    for (const auto& [u, v] : edges) {
        if (find(u) != find(v)) {
            cut_edges.emplace_back(u, v);
        }
    }
    
    return {static_cast<int>(cut_edges.size()), cut_edges};
}

vector<pair<int, int>> find_bridges(int n, const vector<vector<int>>& adj) {
    vector<int> disc(n, -1);
    vector<int> low(n, -1);
    vector<bool> visited(n, false);
    vector<pair<int, int>> bridges;
    int time = 0;
    
    function<void(int, int)> dfs = [&](int u, int parent) {
        disc[u] = low[u] = ++time;
        visited[u] = true;
        
        for (int v : adj[u]) {
            if (v == parent) continue;
            if (disc[v] == -1) {
                dfs(v, u);
                low[u] = min(low[u], low[v]);
                if (low[v] > disc[u]) {
                    bridges.emplace_back(u, v);
                }
            } else {
                low[u] = min(low[u], disc[v]);
            }
        }
    };
    
    for (int i = 0; i < n; i++) {
        if (disc[i] == -1) {
            dfs(i, -1);
        }
    }
    
    return bridges;
}
