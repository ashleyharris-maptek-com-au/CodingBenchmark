use std::collections::VecDeque;

fn solve(maze: &[Vec<u8>], h: usize, w: usize) -> Option<Vec<(usize, usize)>> {
    // Find start (A) and end (B)
    let start = find_cell(maze, b'A')?;
    let end = find_cell(maze, b'B')?;
    
    // BFS
    let mut visited = vec![0u8; h * w];
    let mut distance = vec![0i32; h * w];
    
    let mut queue = VecDeque::new();
    queue.push_back(start);
    visited[start.1 * w + start.0] = 1;
    distance[start.1 * w + start.0] = 0;
    
    let dirs = [(0, 1), (0, -1), (1, 0), (-1, 0)];
    
    while let Some((x, y)) = queue.pop_front() {
        let d = distance[y * w + x];
        
        if x == end.0 && y == end.1 {
            break;
        }
        
        for (dx, dy) in dirs.iter() {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let nidx = (ny * w as i32) as usize + nx as usize;
                if maze[ny as usize][nx as usize] == b' ' && visited[nidx] == 0 {
                    visited[nidx] = 1;
                    distance[nidx] = d + 1;
                    queue.push_back((nx as usize, ny as usize));
                }
            }
        }
    }
    
    // Reconstruct path
    if visited[end.1 * w + end.0] == 0 {
        return None;
    }
    
    let mut path = Vec::new();
    let mut x = end.0;
    let mut y = end.1;
    
    while x != start.0 || y != start.1 {
        path.push((x, y));
        let d = distance[y * w + x];
