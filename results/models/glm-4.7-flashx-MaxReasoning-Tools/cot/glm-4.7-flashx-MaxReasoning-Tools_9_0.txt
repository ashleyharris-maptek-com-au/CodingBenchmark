Let me think about the 1D cutting stock problem. Given a stock length and a list of cut lengths, I need to pack them into the minimum number of stock pieces.

This is essentially a bin packing problem, which is NP-hard. However, for large inputs, I need an efficient algorithm.

Let me think about approaches:

1. **First Fit Decreasing (FFD)**: Sort cuts in decreasing order, then for each cut, place it in the first stock that can accommodate it. This gives a good approximation (≤ 11/9 OPT + 6/9).

2. **Best Fit Decreasing (BFD)**: Sort cuts in decreasing order, then for each cut, place it in the stock that will have the most remaining space (or least remaining space that can fit it). This also gives a good approximation.

3. **Branch and Bound**: For smaller problems, we can use exact methods.

2. Read the N cut lengths
3. Sort cuts in descending order
4. For each cut, find the first stock that can fit it
5. If no stock can fit it, create a new stock

```csharp
using System;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var sw = new Stopwatch();
        sw.Start();
        
        // Use a stream reader for efficient reading
        using var reader = new StreamReader(Console.OpenStandardInput());
        var firstLine = reader.ReadLine();
        if (firstLine == null) return;
        
        var parts = firstLine.Split();
        int n = int.Parse(parts[0]);
        int stockLength = int.Parse(parts[1]);
        
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(reader.ReadLine()!);
        }
        
        // Sort cuts in descending order
        Array.Sort(cuts);
        Array.Reverse(cuts);
        
        // Use two arrays for remaining capacities and assigned cuts
        // stockRemaining[i] = remaining capacity of stock i
        // stockCuts[i] = bitmask or list of cuts in stock i
        int[] stockRemaining = new int[n];
        int[] stockCuts = new int[n]; // will store indices of cuts in each stock
        
        int numStocks = 0;
        
        for (int i = 0; i < n; i++)
        {
            int cut = cuts[i];
            int bestIdx = -1;
            int bestDiff = int.MaxValue;
            
            // Find the stock with most remaining capacity that can fit the cut
            for (int j = 0; j < numStocks; j++)
            {
                if (stockRemaining[j] >= cut)
                {
                    int diff = stockRemaining[j] - cut;
                    if (diff < bestDiff)
                    {
                        bestDiff = diff;
                        bestIdx = j;
                    }
                }
            }
            
            if (bestIdx != -1)
            {
                stockRemaining[bestIdx] -= cut;
                stockCuts[bestIdx] |= (1 << i);
            }
            else
            {
                stockRemaining[numStocks] = stockLength - cut;
                stockCuts[numStocks] = (1 << i);
                numStocks++;
            }
        }
        
        sw.Stop();
        Console.WriteLine(sw.Elapsed.TotalMilliseconds);
        Console.WriteLine(numStocks);
        for (int i = 0; i < numStocks; i++)
        {
            int mask = stockCuts[i];
            var indices = new System.Collections.Generic.List<int>();
            for (int j = 0; j < n; j++)
            {
                if ((mask & (1 << j)) != 0)
                {
                    indices.Add(j);
                }
            }
            Console.WriteLine(string.Join(" ", indices));
        }
    }
}
```

```csharp
using System;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var sw = new Stopwatch();
        sw.Start();
        
        using var reader = new StreamReader(Console.OpenStandardInput());
        var firstLine = reader.ReadLine();
        if (firstLine == null) return;
        
        var parts = firstLine.Split();
        int n = int.Parse(parts[0]);
        int stockLength = int.Parse(parts[1]);
        
        int[] cuts = new int[n];
        var cutStrs = reader.ReadLine()!.Split();
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(cutStrs[i]);
        }
        
        // Sort cuts in descending order
        Array.Sort(cuts);
        Array.Reverse(cuts);
        
        // Find pairs that fit together
        int[] stockOfCut = new int[n];
        int numStocks = 0;
        
        int left = 0;
        int right = n - 1;
        
        while (left <= right)
        {
            if (cuts[left] + cuts[right] <= stockLength)
            {
                // Pair these two cuts
                stockOfCut[left] = numStocks;
                stockOfCut[right] = numStocks;
                left++;
                right--;
                numStocks++;
            }
            else
            {
                // Can't pair, put right cut alone
                stockOfCut[right] = numStocks;
                right--;
                numStocks++;
            }
        }
        
        // Handle remaining single cuts
        while (left <= right)
        {
            stockOfCut[left] = numStocks;
            left++;
            numStocks++;
        }
        
        // Count cuts per stock
        var stockCounts = new int[numStocks];
        for (int i = 0; i < n; i++)
        {
            stockCounts[stockOfCut[i]]++;
        }
        
        // Build output
        var outputs = new string[numStocks][];
        var outputPositions = new int[numStocks];
        for (int i = 0; i < numStocks; i++)
        {
            outputs[i] = new string[stockCounts[i]];
            outputPositions[i] = 0;
        }
        
        for (int i = 0; i < n; i++)
        {
            int s = stockOfCut[i];
            outputs[s][outputPositions[s]++] = i.ToString();
        }
        
        Console.WriteLine(numStocks);
        for (int i = 0; i < numStocks; i++)
        {
            Console.WriteLine(string.Join(" ", outputs[i]));
        }
        
        sw.Stop();
        // Console.WriteLine(sw.Elapsed.TotalMilliseconds);
    }
}
```

Let me implement a Binary Indexed Tree (Fenwick Tree) to maintain stocks by remaining capacity:

```csharp
using System;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var sw = new Stopwatch();
        sw.Start();
        
        using var reader = new StreamReader(Console.OpenStandardInput());
        var firstLine = reader.ReadLine();
        if (firstLine == null) return;
        
        var parts = firstLine.Split();
        int n = int.Parse(parts[0]);
        int stockLength = int.Parse(parts[1]);
        
        int[] cuts = new int[n];
        var cutStrs = reader.ReadLine()!.Split();
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(cutStrs[i]);
        }
        
        // First Fit with binary search for efficiency
        int[] stockOfCut = new int[n];
        int[] remainingInStock = new int[n];
        int numStocks = 0;
        
        // Sorted list of (remaining, index)
        var remainingIndices = new List<(int remaining, int index)>();
        
        for (int i = 0; i < n; i++)
        {
            int cut = cuts[i];
            
            // Binary search for stock with most remaining capacity that can fit
            int left = 0, right = remainingIndices.Count;
            int bestIdx = -1;
            while (left < right)
            {
                int mid = (left + right) / 2;
                if (remainingIndices[mid].remaining >= cut)
                {
                    right = mid;
                }
                else
                {
                    left = mid + 1;
                }
            }
            
            if (left < remainingIndices.Count)
            {
                bestIdx = remainingIndices[left].index;
                remainingIndices.RemoveAt(left);
                remainingInStock[bestIdx] -= cut;
                stockOfCut[i] = bestIdx;
                
                // Reinsert into sorted list
                int insertPos = remainingIndices.Count;
                while (insertPos > 0 && remainingIndices[insertPos - 1].remaining < remainingInStock[bestIdx])
                {
                    insertPos--;
                }
                remainingIndices.Insert(insertPos, (remainingInStock[bestIdx], bestIdx));
            }
            else
            {
                remainingInStock[numStocks] = stockLength - cut;
                remainingIndices.Add((remainingInStock[numStocks], numStocks));
                stockOfCut[i] = numStocks;
                numStocks++;
            }
        }
        
        // Count cuts per stock
        var stockCounts = new int[numStocks];
        for (int i = 0; i < n; i++)
        {
            stockCounts[stockOfCut[i]]++;
        }
        
        // Build output strings
        var outputs = new string[numStocks][];
        var outputPositions = new int[numStocks];
        for (int i = 0; i < numStocks; i++)
        {
            outputs[i] = new string[stockCounts[i]];
            outputPositions[i] = 0;
        }
        
        for (int i = 0; i < n; i++)
        {
            int s = stockOfCut[i];
            outputs[s][outputPositions[s]++] = i.ToString();
        }
        
        Console.WriteLine(numStocks);
        for (int i = 0; i < numStocks; i++)
        {
            Console.WriteLine(string.Join(" ", outputs[i]));
        }
    }
```
