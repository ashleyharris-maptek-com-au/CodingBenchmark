function parallelKMeans(points, K, numWorkers) {
  // Initialize cluster centers: choose K random points (or farthest point first)
  const centroids = initializeCentroids(points, K);

  let assignments = new Array(points.length).fill(0);
  let changed = true;
  const maxIter = 10;

  for (let iter = 0; iter < maxIter && changed; iter++) {
    // Assign each point to nearest centroid
    changed = assignPoints(points, centroids, assignments, numWorkers);
    // Compute new centroids
    centroids = computeCentroids(points, assignments, K);
  }

  // Group points by cluster index
  const clusters = new Array(K).fill(null).map(() => []);
  for (let i = 0; i < points.length; i++) {
    clusters[assignments[i]].push(points[i]);
  }

  return { clusters, centroids };
}

function initializeCentroids(points, K) {
  // Use farthest first: start with a random point, then repeatedly pick farthest from current centroids
  const centroids = [];
  let indices = new Set();
  // pick first random point
  let firstIdx = Math.floor(Math.random() * points.length);
  centroids.push({ x: points[firstIdx].x, y: points[firstIdx].y });
  indices.add(firstIdx);

  // pick farthest remaining points K-1 times
  for (let k = 1; k < K; k++) {
    let maxDist = -1;
    let farthestIdx = -1;
    for (let i = 0; i < points.length; i++) {
      if (indices.has(i)) continue;
      let d = 0;
      for (let c of centroids) {
        const dx = points[i].x - c.x;
        const dy = points[i].y - c.y;
        d += dx * dx + dy * dy;
      }
      if (d > maxDist) {
        maxDist = d;
        farthestIdx = i;
      }
    }
    if (farthestIdx !== -1) {
      centroids.push({ x: points[farthestIdx].x, y: points[farthestIdx].y });
      indices.add(farthestIdx);
    }
  }
  return centroids;
}

function assignPoints(points, centroids, assignments, numWorkers) {
  const chunkSize = Math.ceil(points.length / numWorkers);
  const promises = [];

  for (let w = 0; w < numWorkers; w++) {
    const start = w * chunkSize;
    const end = Math.min((w + 1) * chunkSize, points.length);
    promises.push(new Promise((resolve) => {
      const worker = new Worker(kmeansWorkerCode, {
        workerData: { points: points.slice(start, end), centroids }
      });
      worker.on('message', (chunkAssignments) => resolve({ workerIdx: w, assignments: chunkAssignments }));
      worker.on('error', (err) => {
        console.error(err);
        resolve({ workerIdx: w, assignments: new Array(end - start).fill(0) });
      });
      worker.on('exit', (code) => {
        if (code !== 0) resolve({ workerIdx: w, assignments: new Array(end - start).fill(0) });
      });
    }));
  }

  return Promise.all(promises).then(results => {
    let changed = false;
    for (let i = 0; i < results.length; i++) {
      const { assignments: chunkAssignments } = results[i];
      for (let j = 0; j < chunkAssignments.length; j++) {
        if (assignments[start + j] !== chunkAssignments[j]) changed = true;
        assignments[start + j] = chunkAssignments[j];
      }
    }
    return changed;
  });
}

function computeCentroids(points, assignments, K) {
  const sums = new Array(K).fill(0).map(() => ({ x: 0, y: 0, count: 0 }));
  for (let i = 0; i < points.length; i++) {
    const c = assignments[i];
    sums[c].x += points[i].x;
    sums[c].y += points[i].y;
    sums[c].count++;
  }
  const centroids = [];
  for (let k = 0; k < K; k++) {
    if (sums[k].count === 0) {
      // If empty cluster, choose a random point
      const randomPoint = points[Math.floor(Math.random() * points.length)];
      centroids.push({ x: randomPoint.x, y: randomPoint.y });
    } else {
      centroids.push({ x: sums[k].x / sums[k].count, y: sums[k].y / sums[k].count });
    }
  }
  return centroids;
}

function buildRoute(clusterPoints, depot) {
  if (clusterPoints.length === 0) return [];
  // nearest neighbor
  const route = [];
  const visited = new Set();
  // start from depot
  let current = { x: depot[0], y: depot[1] };
  while (visited.size < clusterPoints.length) {
    let bestIdx = -1;
    let bestDist = Infinity;
    for (let i = 0; i < clusterPoints.length; i++) {
      const p = clusterPoints[i];
      if (visited.has(p.id)) continue;
      const d = distance(current.x, current.y, p.x, p.y);
      if (d < bestDist) {
        bestDist = d;
        bestIdx = i;
      }
    }
    if (bestIdx !== -1) {
      const p = clusterPoints[bestIdx];
      visited.add(p.id);
      route.push(p.id);
      current = { x: p.x, y: p.y };
    } else break; // should not happen
  }
  // optional 2-opt improvement
  const improved = twoOpt(route, clusterPoints, depot);
  return improved;
}

function twoOpt(route, points, depot) {
  let improved = true;
  const n = route.length;
  while (improved) {
    improved = false;
    // Compute total distance
    let totalDist = distance(depot[0], depot[1], points[route[0]].x, points[route[0]].y);
    for (let i = 0; i < n - 1; i++) {
      totalDist += distance(points[route[i]].x, points[route[i]].y, points[route[i+1]].x, points[route[i+1]].y);
    }
    totalDist += distance(points[route[n-1]].x, points[route[n-1]].y, depot[0], depot[1]);

    for (let i = 0; i < n - 1; i++) {
      for (let j = i + 1; j < n; j++) {
        // Consider swapping edges (i,i+1) and (j,j+1)
        // i+1 may equal j, skip.
        if (i + 1 === j) continue;
        // Compute delta
        const a = points[route[i]];
        const b = points[route[i+1]];
        const c = points[route[j]];
        const d = points[route[(j+1) % n]];

        // distances in original tour
        const d1 = distance(a.x, a.y, b.x, b.y);
        const d2 = distance(c.x, c.y, d.x, d.y);
        // distances after swap
        const d1p = distance(a.x, a.y, c.x, c.y);
        const d2p = distance(b.x, b.y, d.x, d.y);

        const delta = d1 + d2 - (d1p + d2p);
        if (delta < -1e-9) { // improvement
          // perform 2-opt swap
          const newRoute = route.slice(0, i+1).concat(route.slice(i+1, j+1).reverse()).concat(route.slice(j+1));
          route.length = 0;
          for (const idx of newRoute) route.push(idx);
          improved = true;
          // break inner loops to restart
          break;
        }
      }
      if (improved) break;
    }
  }
  return route;
}

function distance(x1, y1, x2, y2) {
    return Math.hypot(x1 - x2, y1 - y2);
}

function squaredDistance(x1, y1, x2, y2) {
    const dx = x1 - x2;
    const dy = y1 - y2;
    return dx * dx + dy * dy;
}

// Deterministic k-means (farthest-first initialization)
function deterministicKMeans(points, K) {
    if (K === 1) {
        // single cluster
        return [points];
    }
    
    // Sort points by id ascending for deterministic tie-breaking
    points.sort((a, b) => a.id - b.id);
    // We'll assign centroids deterministically:
    const centroids = [];
    // Initialize first centroid as first point
    centroids.push({ x: points[0].x, y: points[0].y, idx: points[0].idx });
    
    // For remaining centroids, select farthest point from existing centroids
    for (let c = 1; c < K; c++) {
        let maxDist = -1;
        let bestPointIdx = -1;
        for (let i = 0; i < points.length; i++) {
            // sum of squared distances to all centroids
            let totalSqDist = 0;
            for (const centroid of centroids) {
                totalSqDist += squaredDistance(points[i].x, points[i].y, centroid.x, centroid.y);
            }
            // If tie, choose smallest id (which corresponds to smallest idx because sorted by id)
            if (totalSqDist > maxDist || (totalSqDist === maxDist && points[i].idx < points[bestPointIdx].idx)) {
                maxDist = totalSqDist;
                bestPointIdx = i;
            }
        }
        if (bestPointIdx !== -1) {
            centroids.push({ x: points[bestPointIdx].x, y: points[bestPointIdx].y, idx: points[bestPointIdx].idx });
        }
    }

    // K-means iterations
    const maxIter = 5;
    const assignments = new Array(points.length).fill(0);
    let converged = false;

    for (let iter = 0; iter < maxIter && !converged; iter++) {
        // Assign points to nearest centroid
        for (let i = 0; i < points.length; i++) {
            let bestDist = Infinity;
            let bestCentroid = 0;
            for (let c = 0; c < centroids.length; c++) {
                const d = squaredDistance(points[i].x, points[i].y, centroids[c].x, centroids[c].y);
                if (d < bestDist || (d === bestDist && c < bestCentroid)) {
                    bestDist = d;
                    bestCentroid = c;
                }
            }
            assignments[i] = bestCentroid;
        }
        // Update centroids
        const sumX = new Array(K).fill(0);
        const sumY = new Array(K).fill(0);
        const count = new Array(K).fill(0);
        for (let i = 0; i < points.length; i++) {
            const c = assignments[i];
            sumX[c] += points[i].x;
            sumY[c] += points[i].y;
            count[c]++;
        }
        let newCentroids = [];
        for (let c = 0; c < K; c++) {
            if (count[c] === 0) {
                // empty cluster: choose point with smallest id
                // we need to find the smallest id among points
                // but since points are sorted by id, we can pick first point (or with smallest id)
                const best = points[0]; // smallest id
                newCentroids.push({ x: best.x, y: best.y, idx: best.idx });
            } else {
                newCentroids.push({ x: sumX[c] / count[c], y: sumY[c] / count[c], idx: centroids[c].idx });
            }
        }
        // Check convergence: if no centroid moved significantly?
        // For simplicity, just break if no empty clusters (empty clusters already handled) or maxIter reached
        // We'll consider convergence when assignments stable; but for deterministic simplicity, we break after maxIter.
        converged = true;
        for (let c = 0; c < K; c++) {
            if (Math.abs(newCentroids[c].x - centroids[c].x) > 1e-9 ||
                Math.abs(newCentroids[c].y - centroids[c].y) > 1e-9) {
                converged = false;
                break;
            }
        }
        centroids.length = 0;
        for (const cent of newCentroids) centroids.push(cent);
    }

    // Group points by cluster
    const clusters = new Array(K).fill(null).map(() => []);
    for (let i = 0; i < points.length; i++) {
        const c = assignments[i];
        clusters[c].push(points[i]);
    }
    return clusters;
}

function buildRoute(points, depot) {
    const n = points.length;
    if (n === 0) return [];

    // Determine if using NN (small cluster) or angular order (large cluster)
    const LARGE_THRESHOLD = 5000;
    if (n <= LARGE_THRESHOLD) {
        return buildRouteNN(points, depot);
    } else {
        return buildRouteAngular(points, depot);
    }
}

function buildRouteNN(points, depot) {
    const depotX = depot[0];
    const depotY = depot[1];
    const visited = new Set();
    const route = [];
    let currentX = depotX;
    let currentY = depotY;

    while (visited.size < points.length) {
        let bestIdx = -1;
        let bestDist = Infinity;
        for (let i = 0; i < points.length; i++) {
            const p = points[i];
            if (visited.has(p.id)) continue;
            const d = distance(currentX, currentY, p.x, p.y);
            if (d < bestDist || (d === bestDist && p.id < points[bestIdx].id)) {
                bestDist = d;
                bestIdx = i;
            }
        }
        if (bestIdx === -1) break; // should not happen
        const p = points[bestIdx];
        visited.add(p.id);
        route.push(p.id);
        currentX = p.x;
        currentY = p.y;
    }

    // simple 2-opt pass
    return simple2Opt(route, points, depot);
}

function buildRouteAngular(points, depot) {
    const depotX = depot[0];
    const depotY = depot[1];
    const n = points.length;
    // compute angle for each point
    const angledPoints = points.map(p => ({
        id: p.id,
        x: p.x,
        y: p.y,
        angle: Math.atan2(p.y - depotY, p.x - depotX)
    }));
    // sort by angle
    angledPoints.sort((a, b) => a.angle - b.angle);
    // initial route
    const route = angledPoints.map(p => p.id);
    // optional limited 2-opt
    const improved = simple2Opt(route, angledPoints.map(p => ({id: p.id, x: p.x, y: p.y})), depot);
    return improved;
}

function simple2Opt(route, points, depot) {
    const n = route.length;
    if (n <= 1) return route;

    const depotX = depot[0];
    const depotY = depot[1];

    // compute distances needed
    // We'll precompute distances between points for O(1) look-up
    const distCache = new Map();
    function dist(iIdx, jIdx) {
        const key = [iIdx, jIdx].sort().join(',');
        if (distCache.has(key)) return distCache.get(key);
        const d = distance(points[iIdx].x, points[iIdx].y, points[jIdx].x, points[jIdx].y);
        distCache.set(key, d);
        return d;
    }

    // compute initial total distance (including depot edges)
    function totalDist(route) {
        let d = distance(depotX, depotY, points[route[0]].x, points[route[0]].y);
        for (let i = 0; i < route.length - 1; i++) {
            d += dist(route[i], route[i+1]);
        }
        d += distance(points[route[route.length-1]].x, points[route[route.length-1]].y, depotX, depotY);
        return d;
    }

    let improved = true;
    while (improved) {
        improved = false;
        for (let i = 0; i < n - 1; i++) {
            for (let j = i + 2; j < n; j++) {
                // edges: i->i+1 and j->j+1 (with wrap if j+1 == n)
                // if j+1 == n, we treat as wrap to depot; but our delta formula uses distances between points, ignoring depot edges.
                const a = points[route[i]];
                const b = points[route[i+1]];
                const c = points[route[j]];
                const d = points[route[(j+1) % n]]; // wrap

                // compute delta: d(a,c) + d(b,d) - d(a,b) - d(c,d)
                const d_ac = dist(i, j);
                const d_bd = dist(i+1, (j+1) % n);
                const d_ab = dist(i, i+1);
                const d_cd = dist(j, (j+1) % n);

                const delta = d_ac + d_bd - d_ab - d_cd;
                if (delta < 0) {
                    // perform swap: reverse segment between i+1 and j
                    const newRoute = route.slice(0, i+1).concat(route.slice(i+1, j+1).reverse()).concat(route.slice(j+1));
                    route.length = 0;
                    for (const r of newRoute) route.push(r);
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
    }
    return route;
}

                const delta = d_ac + d_bd - d_ab - d_cd;
                if (delta < 0) {
                    // reverse segment between i+1 and j
                    const newRoute = route.slice(0, i+1).concat(route.slice(i+1, j+1).reverse()).concat(route.slice(j+1));
                    route.length = 0;
                    for (const r of newRoute) route.push(r);
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
    }
    return route;
}
