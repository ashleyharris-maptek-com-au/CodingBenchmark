    {

        List<int> circuit = new List<int>();

            foreach (var (to, w) in adj[i])
                circuitEdges[i].Add((to, w));

        stack.Push(0);

            {
                var (v, w) = circuitEdges[u].Last();
                circuitEdges[u].RemoveAt(circuitEdges[u].Count - 1);
                stack.Push(v);
            }
            else
            {
                circuit.Add(u);
                stack.Pop();
            }

        // Reverse to get the circuit
        circuit.Reverse();

        List<int> oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
            if (degree[i] % 2 == 1)
                oddVertices.Add(i);
        
        int k = oddVertices.Count;
        
        if (k == 0)
        {
            // Eulerian graph - find Eulerian circuit
            var circuit = Hierholzer(adj);
            PrintCircuit(circuit, output);
            Console.Write(output);
            return;
        }
        
        // Compute all-pairs shortest paths using Floyd-Warshall
        int[,] dist = FloydWarshall(adj, n);
        int[,] next = ComputeNextMatrix(dist, n);
        
        // Find minimum-weight perfect matching
        var matching = FindMinimumMatching(oddVertices, dist);
        
        // Build graph with additional paths
        var adjWithPaths = BuildGraphWithPaths(adj, oddVertices, matching, next);
        
        // Find Eulerian circuit
        var circuit = Hierholzer(adjWithPaths);
        PrintCircuit(circuit, output);
        Console.Write(output);
    }
    
    static int[,] FloydWarshall(List<(int to, int weight)>[] adj, int n)
    {
        int[,] dist = new int[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                dist[i, j] = (i == j) ? 0 : int.MaxValue / 2;
            }
            foreach (var (to, w) in adj[i])
                dist[i, to] = Math.Min(dist[i, to], w);
        }
        
        for (int kk = 0; kk < n; kk++)
        {
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if (dist[i, kk] + dist[kk, j] < dist[i, j])
                        dist[i, j] = dist[i, kk] + dist[kk, j];
                }
            }
        }
        
        return dist;
    }
    
    static int[,] ComputeNextMatrix(int[,] dist, int n)
    {
        int[,] next = new int[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i != j && dist[i, j] != int.MaxValue / 2)
                {
                    // Find the first vertex in the path from i to j
                    foreach (var (to, w) in adj[i])
                    {
                        if (dist[i, to] + dist[to, j] == dist[i, j])
                        {
                            next[i, j] = to;
                            break;
                        }
                    }
                }
            }
        }
        return next;
    }
    
    static List<(int, int)> FindMinimumMatching(List<int> oddVertices, int[,] dist)
    {
        int k = oddVertices.Count;
        
        // Use brute force for small k
        if (k <= 8)
        {
            int bestCost = int.MaxValue;
            List<(int, int)> bestMatching = null;
            
            // Generate all perfect matchings
            var indices = Enumerable.Range(0, k / 2).ToArray();
            foreach (var perm in GenerateMatchings(indices))
            {
                int cost = 0;
                for (int i = 0; i < perm.Length; i++)
                {
                    cost += dist[oddVertices[2 * i], oddVertices[2 * i + 1]];
                }
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestMatching = perm.ToList();
                }
            }
            
            return bestMatching;
        }
        else
        {
            // Use DP with bitmask for larger k
            return FindMatchingDP(oddVertices, dist);
        }
    }
    
    static IEnumerable<List<(int, int)>> GenerateMatchings<T>(T[] array)
    {
        if (array.Length == 0)
        {
            yield return new List<(int, int)>();
        }
        else
        {
            for (int i = 0; i < array.Length; i++)
            {
                var remaining = array.Skip(i + 1).ToArray();
                foreach (var matching in GenerateMatchings(remaining))
                {
                    yield return new List<(int, int)> { (array[i], remaining[0]) }.Concat(matching).ToList();
                }
            }
        }
    }
    
    static List<(int, int)> FindMatchingDP(List<int> oddVertices, int[,] dist)
    {
        int k = oddVertices.Count;
        int half = k / 2;
        
        // dp[mask][i] = minimum cost to match first i pairs where mask indicates which vertices are matched
        int[,] dp = new int[1 << k, half + 1];
        int[,] parent = new int[1 << k, half];
        
        const int INF = int.MaxValue / 2;
        for (int mask = 0; mask < (1 << k); mask++)
        {
            for (int i = 0; i <= half; i++)
                dp[mask, i] = INF;
        }
        dp[0, 0] = 0;
        
        for (int mask = 0; mask < (1 << k); mask++)
        {
            int pairs = CountSetBits(mask) / 2;
            if (pairs >= half) continue;
            
            // Find first unmatched vertex
            int first = -1;
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }
            
            if (first == -1) continue;
            
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0 && i != first)
                {
                    int newMask = mask | (1 << first) | (1 << i);
                    int cost = dist[oddVertices[first], oddVertices[i]];
                    if (dp[mask, pairs] + cost < dp[newMask, pairs + 1])
                    {
                        dp[newMask, pairs + 1] = dp[mask, pairs] + cost;
                        parent[newMask, pairs] = i;
                    }
                }
            }
        }
        
        // Reconstruct matching
        int fullMask = (1 << k) - 1;
        var matching = new List<(int, int)>();
        int remainingPairs = half;
        
        while (remainingPairs > 0)
        {
            int i = GetFirstSetBit(fullMask);
            int matchedTo = parent[fullMask, remainingPairs - 1];
            matching.Add((i, matchedTo));
            fullMask &= ~((1 << i) | (1 << matchedTo));
            remainingPairs--;
        }
        
        return matching;
    }
    
    static int CountSetBits(int x)
    {
        int count = 0;
        while (x > 0)
        {
            count += x & 1;
            x >>= 1;
        }
        return count;
    }
    
    static int GetFirstSetBit(int x)
    {
        int pos = 0;
        while ((x & 1) == 0)
        {
            x >>= 1;
            pos++;
        }
        return pos;
    }
    
    static List<(int to, int weight)>[] BuildGraphWithPaths(List<(int to, int weight)>[] adj, List<int> oddVertices, List<(int, int)> matching, int[,] next)
    {
        int n = adj.Length;
        List<(int to, int weight)>[] adjWithPaths = new List<(int to, int weight)>[n];
        for (int i = 0; i < n; i++)
            adjWithPaths[i] = new List<(int to, int weight)>();
        
        // Copy original edges
        for (int i = 0; i < n; i++)
        {
            foreach (var (to, w) in adj[i])
                adjWithPaths[i].Add((to, w));
        }
        
        // Add paths between matched odd vertices
        foreach (var (u, v) in matching)
        {
            u = oddVertices[u];
            v = oddVertices[v];
            
            int curr = u;
            while (curr != v)
            {
                int nextNode = next[curr, v];
                if (nextNode == -1) break; // Shouldn't happen
                adjWithPaths[curr].Add((nextNode, 1)); // Weight doesn't matter for Eulerian circuit
                adjWithPaths[nextNode].Add((curr, 1));
                curr = nextNode;
            }
        }
        
        return adjWithPaths;
    }
    
    static List<int> Hierholzer(List<(int to, int weight)>[] adj)
    {
        int n = adj.Length;
        List<int> circuit = new List<int>();
        List<(int to, int weight)>[] edgeList = new List<(int to, int weight)>[n];
        
        for (int i = 0; i < n; i++)
            edgeList[i] = new List<(int to, int weight)>(adj[i]);
        
        Stack<int> stack = new Stack<int>();
        stack.Push(0);
        
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (edgeList[u].Count > 0)
            {
                var (v, _) = edgeList[u].Last();
                edgeList[u].RemoveAt(edgeList[u].Count - 1);
                stack.Push(v);
            }
            else
            {
                circuit.Add(u);
                stack.Pop();
            }
        }
        
        circuit.Reverse();
        return circuit;
    }
    
    static void PrintCircuit(List<int> circuit, StringBuilder output)
    {
        foreach (int v in circuit)
            output.Append(v + " ");
        output.AppendLine();
    }
