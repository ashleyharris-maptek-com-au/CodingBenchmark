struct TerrainData {
    size_t width;
    size_t depth;
    vector<uint16_t> ground;      // top of ground rock
    vector<uint16_t> float_bottom; // bottom of floating rock
    vector<uint16_t> float_top;   // top of floating rock
    bool loaded = false;
    std::mutex mtx;
    std::condition_variable cv;
};

bool loadTerrain(TerrainData& data, const string& path) {
    ifstream f(path, ios::binary | ios::ate);
    if (!f) return false;
    auto fsize = f.tellg();
    f.seekg(0);
    vector<char> buf(fsize);
    f.read(buf.data(), fsize);
    // parse
    const size_t cells = data.width * data.depth;
    data.ground.resize(cells);
    data.float_bottom.resize(cells);
    data.float_top.resize(cells);
    for (size_t i = 0; i < cells; ++i) {
        const char* p = buf.data() + i * 6;
        uint16_t a = *reinterpret_cast<const uint16_t*>(p);
        uint16_t b = *reinterpret_cast<const uint16_t*>(p + 2);
        uint16_t c = *reinterpret_cast<const uint16_t*>(p + 4);
        data.ground[i] = a;
        data.float_bottom[i] = b;
        data.float_top[i] = c;
    }
    data.loaded = true;
    data.cv.notify_one();
    return true;
}

void terrainLoadingThread(TerrainData& data, const string& path) {
    loadTerrain(data, path);
}

double clamp(double v, double lo, double hi) {
    if (v < lo) return lo;
    if (v > hi) return hi;
    return v;
}
