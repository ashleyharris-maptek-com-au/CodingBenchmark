// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

    vec4 outData[];  // length = N * 2
};

    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against gl_GlobalInvocationID.x >= N
    if (id >= N) {
        return;
    }
    
    uint inIdx = id * 2;
    uint outIdx = id * 2;
    
    vec4 inPos = inData[inIdx];
    vec4 inVel = inData[inIdx + 1];
    
    vec3 pos = inPos.xyz;
    vec3 vel = inVel.xyz;
    
    // Compute boid parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    vec3 sep = vec3(0.0);
    vec3 align = vec3(0.0);
    vec3 coh = vec3(0.0);
    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;
    
    // Iterate over all boids
    for (uint j = 0; j < N; j++) {
        if (id == j) {
            continue;  // Skip self
        }
        
        uint jIdx = j * 2;
        vec4 otherPos = inData[jIdx];
        vec4 otherVel = inData[jIdx + 1];
        
        vec3 diff = pos - otherPos.xyz;
        float distSq = dot(diff, diff);
        
        // Skip zero-distance neighbors
        if (distSq == 0.0) {
            continue;
        }
        
        // Separation
        if (distSq < sepRadius * sepRadius) {
            float dist = sqrt(distSq);
            // Weight by inverse distance
            float weight = sepRadius / dist;
            sep += diff * weight;
            sepCount++;
        }
        
        // Alignment
        if (distSq < alignRadius * alignRadius) {
            align += otherVel.xyz;
            alignCount++;
        }
        
        // Cohesion
        if (distSq < cohRadius * cohRadius) {
            coh += otherPos.xyz;
            cohCount++;
        }
    }

    if (alignCount > 0) {
        align /= float(alignCount);
    }

        coh /= float(cohCount);

        if (cohLen > 0.0001) {
            coh /= cohLen;

    // Average separation
    if (sepCount > 0) {
        sep /= float(sepCount);
        // Normalize separation
        float sepLen = length(sep);
        if (sepLen > 0.0001) {
            sep /= sepLen;
        }
    }

    vec3 accel = vec3(0.0);
    accel += sep * sepWeight;
    accel += align * alignWeight;
    accel += coh * cohWeight;

        boundaryForce.x = boundForce * (boundLimit - pos.x);

        boundaryForce.z = boundForce * (boundLimit - pos.z);
    }
    
    // Push toward center if at boundary
    if (pos.x > boundSize * 0.5) {
        boundaryForce.x = -boundForce * (pos.x - boundSize * 0.5);
    } else if (pos.x < -boundSize * 0.5) {
        boundaryForce.x = boundForce * (boundSize * 0.5 - pos.x);

        boundaryForce.z = boundForce * (boundSize * 0.5 - pos.z);

    // Update velocity
    vel += accel * dt;
    
    // Clamp speed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }
    
    // Update position
    pos += vel * dt;
    
    // Write output
    outData[outIdx] = vec4(pos, 0.0);
    outData[outIdx + 1] = vec4(vel, 0.0);
