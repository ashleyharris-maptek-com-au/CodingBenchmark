const topologicalOrder: string[] = [];

  const id = queue.shift();

  let taskStart = 0;
  for (const dep of deps) {

    if (depFinish !== undefined && depFinish > taskStart) {

interface Task {
  id: string;
  duration: number;
  deps?: string[];
  deadline?: number;
  weight?: number;
}

function solve() {
  const require: any = require;
  const fs = require('fs');
  const input: string = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);
  const machineCount = data.machineCount;
  const tasks: Record<string, Task> = data.tasks;
  // Build indegree, dependents
  const indegree: Record<string, number> = {};
  const dependents: Record<string, string[]> = {};
  for (const [id, task] of Object.entries(tasks)) {
    indegree[id] = task.deps?.length ?? 0;
    dependents[id] = [];
  }
  for (const [id, task] of Object.entries(tasks)) {
    for (const dep of task.deps ?? []) {
      dependents[dep].push(id);
    }
  }
  const queue: string[] = Object.keys(indegree).filter(id => indegree[id] === 0);
  const order: string[] = [];
  while (queue.length > 0) {
    const id = queue.shift()!;
    order.push(id);
    for (const dep of dependents[id]) {
      indegree[dep]--;
      if (indegree[dep] === 0) {
        queue.push(dep);
      }
    }
  }
  // schedule
  const machineFinish = new Array<number>(machineCount).fill(0);
  const taskFinish: Record<string, number> = {};
  const schedule: { id: string; start: number; machine: number }[] = [];
  for (const id of order) {
    const task = tasks[id];
    let start = 0;
    for (const dep of task.deps ?? []) {
      const f = taskFinish[dep];
      if (f > start) start = f;
    }
    // find first machine where machineFinish >= start
    let bestMachine = 0;
    let bestFinish = machineFinish[0];
    for (let i = 1; i < machineCount; i++) {
      const f = machineFinish[i];
      if (f >= start && f < bestFinish) {
        bestFinish = f;
        bestMachine = i;
      }
    }
    const finish = start + task.duration;
    schedule.push({ id, start, machine: bestMachine });
    machineFinish[bestMachine] = finish;
    taskFinish[id] = finish;
  }
  // output schedule
  console.log(JSON.stringify({ schedule }));
}

solve();
