  probability: number[][];
  drones: Array<{id: number, x: number, y: number}>;
  maxStepsPerDrone: number;
}

interface OutputData {
  paths: Array<Array<[number, number]>>;
}

interface Node {
  x: number;
  y: number;
  g: number;
  h: number;
  f: number;
  parent?: Node;
}

function heuristic(a: {x: number, y: number}, b: {x: number, y: number}): number {
  return Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
}

function getNeighbors(node: {x: number, y: number}, rows: number, cols: number, obstacles: Set<string>): {x: number, y: number}[] {
  const dirs = [[0, -1], [0, 1], [-1, 0], [1, 0]]; // left, right, up, down
  const neighbors: {x: number, y: number}[] = [];
  
  for (const [dx, dy] of dirs) {
    const nx = node.x + dx;
    const ny = node.y + dy;
    if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && !obstacles.has(`${nx},${ny}`)) {
      neighbors.push({x: nx, y: ny});
    }
  }
  
  return neighbors;
}

function astar(start: {x: number, y: number}, goal: {x: number, y: number}, rows: number, cols: number, obstacles: Set<string>): {x: number, y}[] | null {
  const openSet: Node[] = [];
  const cameFrom = new Map<string, Node>();
  const gScore = new Map<string, number>();
  const fScore = new Map<string, number>();
  
  const startKey = `${start.x},${start.y}`;
  const goalKey = `${goal.x},${goal.y}`;
  
  gScore.set(startKey, 0);
  fScore.set(startKey, heuristic(start, goal));
  
  openSet.push({x: start.x, y: start.y, g: 0, h: heuristic(start, goal), f: heuristic(start, goal)});
  
  while (openSet.length > 0) {
    // Find node with lowest f
    let minIndex = 0;
    for (let i = 1; i < openSet.length; i++) {
      if (openSet[i].f < openSet[minIndex].f) {
        minIndex = i;
      }
    }
    
    const current = openSet.splice(minIndex, 1)[0];
    const currentKey = `${current.x},${current.y}`;
    
    if (current.x === goal.x && current.y === goal.y) {
      // Reconstruct path
      const path: {x: number, y: number}[] = [];
      let node = current;
      while (node) {
        path.push({x: node.x, y: node.y});
        const key = `${node.x},${node.y}`;
        node = cameFrom.get(key);
      }
      return path.reverse();
    }
    
    for (const neighbor of getNeighbors(current, rows, cols, obstacles)) {
      const neighborKey = `${neighbor.x},${neighbor.y}`;
      const tentativeG = current.g + 1;
      
      if (tentativeG < (gScore.get(neighborKey) ?? Infinity)) {
        cameFrom.set(neighborKey, current);
        gScore.set(neighborKey, tentativeG);
        fScore.set(neighborKey, tentativeG + heuristic(neighbor, goal));
        
        // Add to open set if not present
        const existingIndex = openSet.findIndex(n => n.x === neighbor.x && n.y === neighbor.y);
        if (existingIndex === -1) {
          openSet.push({
            x: neighbor.x,
            y: neighbor.y,
            g: tentativeG,
            h: heuristic(neighbor, goal),
            f: tentativeG + heuristic(neighbor, goal)
          });
        }
      }
    }
  }
  
  return null; // No path found
}

function getNeighbors(node: {x: number, y: number}, rows: number, cols: number, obstacles: Set<string>): {x: number, y: number}[] {
