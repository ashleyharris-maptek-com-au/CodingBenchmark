#[derive(Copy, Clone, Default)]
struct Vec3 {
    x: f32,
    y: f32,
    z: f32,
}

impl Vec3 {
    fn new(x: f32, y: f32, z: f32) -> Self {
        Vec3 { x, y, z }
    }
    
    fn distance_sq(&self, other: &Vec3) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    
    // Read header
    let mut header = String::new();
    stdin.lock().read_line(&mut header).expect("Failed to read header");
    let mut parts = header.trim().split_whitespace();
    let num_points: u64 = parts.next().unwrap().parse().expect("Invalid num_points");
    let num_clusters: u64 = parts.next().unwrap().parse().expect("Invalid num_clusters");
    
    let num_points = num_points as usize;
    let num_clusters = num_clusters as usize;

fn read_points<R: Read>(reader: &R, num_points: usize) -> Vec<Vec3> {
    let mut points = Vec::with_capacity(num_points);
    let mut buf = [0u8; 12];
    
    for _ in 0..num_points {
        reader.read_exact(&mut buf).expect("Failed to read point");
        let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);

fn standard_kmeans(points: &[Vec3], num_clusters: usize) -> Vec<u32> {
    let mut centroids: Vec<Vec3> = (0..num_clusters)
        .map(|i| points[i])
        .collect();
    
    let mut cluster_counts = vec![1usize; num_clusters];
    let mut cluster_sums: Vec<Vec3> = centroids.clone();
    
    let mut assignments = vec![0u32; points.len()];
    
    for _ in 0..50 {
        // Assign points to nearest centroid
        assignments = points.iter()
            .enumerate()
            .map(|(i, p)| {
                centroids.iter()
                    .enumerate()
                    .map(|(j, c)| (c.distance_sq(p), j))
                    .min_by_key(|&(d, _)| d)
                    .unwrap().1 as u32
            })
            .collect();
        
        // Recalculate centroids
        centroids = vec![Vec3::default(); num_clusters];
        cluster_counts = vec![0usize; num_clusters];
        cluster_sums = vec![Vec3::default(); num_clusters];
        
        for (i, p) in points.iter().enumerate() {
            let c = assignments[i] as usize;
            cluster_sums[c].x += p.x;
            cluster_sums[c].y += p.y;
            cluster_sums[c].z += p.z;
            cluster_counts[c] += 1;
        }
        
        for c in 0..num_clusters {
            if cluster_counts[c] > 0 {
                centroids[c].x = cluster_sums[c].x / cluster_counts[c] as f32;
                centroids[c].y = cluster_sums[c].y / cluster_counts[c] as f32;
                centroids[c].z = cluster_sums[c].z / cluster_counts[c] as f32;
            }
        }
    }
    
    assignments
}
