fn polygon_edges_intersect(p1: &Vec<Point>, p2: &Vec<Point>) -> bool {
    // Check if any edge of p1 intersects any edge of p2
    let n1 = p1.len();
    let n2 = p2.len();

        let (x1, y1) = (p1[i].x, p1[i].y);

            let (x4, y4) = (p2[(j + 1) % n2].x, p2[(j + 1) % n2].y);
            
            // Edge intersection
            let denom = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
            if denom == 0.0 {
                // Parallel edges
                continue;

            let (x3, y3) = (p2.points[j].x, p2.points[j].y);
            let (x4, y4) = (p2.points[(j + 1) % p2.points.len()].x, p2.points[(j + 1) % p2.points.len()].y);
            
            let denom = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
            if denom == 0.0 {
                continue;
            }
            
            let ua = ((x4 - x3) * (y1 - y3) - (y4 - y3) * (x1 - x3)) / denom;
            let ub = ((x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3)) / denom;
            
            if (0.0 <= ua && ua <= 1.0) && (0.0 <= ub && ub <= 1.0) {
                return true;
            }
        }
    }
    
    false
}

fn rotate_point(p: Point, cx: f64, cy: f64, angle: f64) -> Point {
    let cos = angle.cos();
    let sin = angle.sin();
    let dx = p.x - cx;
    let dy = p.y - cy;
    Point {
        x: cx + dx * cos - dy * sin,
        y: cy + dx * sin + dy * cos,
    }

fn area(poly: &Polygon) -> f64 {

    let n = poly.points.len();
    for i in 0..n {
        let j = (i + 1) % n;
        sum += poly.points[i].x * poly.points[j].y;
        sum -= poly.points[j].x * poly.points[i].y;
    }
    0.5 * sum.abs()
}

fn find_placement(
    piece: &Polygon,
    stock: &Polygon,
    placed: &Vec<Polygon>,
    angle_step: f64,
    num_angles: usize,
) -> Option<(f64, f64, f64)> {
    let stock_bb = bounding_box(stock);
    let (sxmin, symin, sxmax, symax) = stock_bb;
    
    for rot in 0..num_angles {
        let angle = rot as f64 * angle_step;
        let rotated = translate_polygon(
            &Polygon { points: piece.points.iter().map(|p| rotate_point(*p, 0.0, 0.0, angle)).collect() },
            0.0, 0.0,
        );
        
        let piece_bb = bounding_box(&rotated);
        let (pxmin, pymin, pxmax, pymax) = piece_bb;
        
        // Search positions
        let grid_step = 1.0;
        let x_range = sxmax - sxmin - (pxmax - pxmin);
        let y_range = symax - symin - (pymax - pymin);
        
        if x_range <= 0 || y_range <= 0 {
            continue;
        }
        
        let mut best_pos = None;
        
        // Try grid positions
        for gx in 0..=(x_range / grid_step) as i32 {
            let px = sxmin + gx as f64 + pxmin;
            if px + (pxmax - pxmin) > sxmax + 0.5 {
                break;
            }
            
            for gy in 0..=(y_range / grid_step) as i32 {
                let py = symin + gy as f64 + pymin;
                if py + (pymax - pymin) > symax + 0.5 {
                    break;
                }
                
                let placed_piece = translate_polygon(&rotated, px, py);
                
                // Check if fits in stock
                if !polygon_contains(&placed_piece, stock) {
                    continue;
                }
                
                // Check overlap with existing pieces
                let mut can_place = true;
                for existing in placed {
                    if polygon_overlap(&placed_piece, existing) {
                        can_place = false;
                        break;
                    }
                }
                
                if can_place {
                    best_pos = Some((px, py, angle));
                    break;
                }
            }
            
            if best_pos.is_some() {
                break;
            }
        }
        
        if let Some(pos) = best_pos {
            return Some(pos);
        }
    }
    
    None
}

fn rotate_point(p: Point, cx: f64, cy: f64, angle: f64) -> Point {
    let cos = angle.cos();
    let sin = angle.sin();
    let dx = p.x - cx;
    let dy = p.y - cy;
    Point { x: cx + dx * cos - dy * sin, y: cy + dx * sin + dy * cos }
}

fn translate_polygon(poly: &Polygon, dx: f64, dy: f64) -> Polygon {
    Polygon { points: poly.points.iter().map(|p| Point { x: p.x + dx, y: p.y + dy }).collect() }
}

fn area(poly: &Polygon) -> f64 {
    let mut sum = 0.0;
    let n = poly.points.len();
    for i in 0..n {
        sum += poly.points[i].x * poly.points[(i + 1) % n].y - poly.points[(i + 1) % n].x * poly.points[i].y;
    }
    0.5 * sum.abs()
}

fn bounding_box(poly: &Polygon) -> (f64, f64, f64, f64) {
    let mut xmin = f64::INFINITY, ymin = f64::INFINITY, xmax = f64::NEG_INFINITY, ymax = f64::NEG_INFINITY;
    for p in &poly.points {
        xmin = xmin.min(p.x); ymin = ymin.min(p.y);
        xmax = xmax.max(p.x); ymax = ymax.max(p.y);
    }
    (xmin, ymin, xmax, ymax)
}

fn polygon_from_points(points: Vec<Point>) -> Polygon {
    Polygon { points }
}

fn read_polygon() -> Polygon {
    let stdin = io::stdin();
    let line = stdin.lock().lines().next().unwrap().unwrap();
    let pv: usize = line.trim().parse().unwrap();
    let mut points = Vec::with_capacity(pv);
    for _ in 0..pv {
        let line = stdin.lock().lines().next().unwrap().unwrap();
        let coords: Vec<f64> = line.split_whitespace().map(|x| x.parse().unwrap()).collect();
        points.push(Point { x: coords[0], y: coords[1] });
    }
    polygon_from_points(points)
}

fn read_input() -> (Vec<Polygon>, Vec<Polygon>) {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first: Vec<usize> = lines.next().unwrap().unwrap().split_whitespace().map(|x| x.parse().unwrap()).collect();
    let sv = first[0]; let np = first[1];
    let mut stocks = Vec::with_capacity(sv);
    for _ in 0..sv { stocks.push(read_polygon()); }
    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np { pieces.push(read_polygon()); }
    (stocks, pieces)
}

fn find_best_placement(piece: &Polygon, stock: &Polygon, placed: &[Polygon], angle_step: f64, num_angles: usize) -> Option<(f64, f64, f64)> {
    let (sxmin, symin, sxmax, symax) = bounding_box(stock);
    let stock_area = (sxmax - sxmin) * (symax - symin);
    let piece_bb = bounding_box(piece);
    if (piece_bb.2 - piece_bb.0) * (piece_bb.3 - piece_bb.1) >= stock_area { return None; }
    
    for rot in 0..num_angles {
        let angle = rot as f64 * angle_step;
        let rotated = translate_polygon(&Polygon { points: piece.points.iter().map(|p| rotate_point(*p, 0.0, 0.0, angle)).collect() }, 0.0, 0.0);
        let (pxmin, pymin, pxmax, pymax) = bounding_box(&rotated);
        if (pxmax - pxmin) > (sxmax - sxmin) || (pymax - pymin) > (symax - symin) { continue; }
        
        let mut best_pos = None;
        for (grid_scale, step) in [(1.0, 5.0), (0.5, 2.5)].iter() {
            let grid_step = *step;
            let max_x = (sxmax - pxmin) / grid_step;
            let max_y = (symax - pymin) / grid_step;
            for gx in 0..=(max_x as i32) {
                let px = sxmin + gx as f64 * grid_step + pxmin;
                if px + (pxmax - pxmin) > sxmax + 0.5 { break; }
                for gy in 0..=(max_y as i32) {
                    let py = symin + gy as f64 * grid_step + pymin;
                    if py + (pymax - pymin) > symax + 0.5 { break; }
                    let placed_piece = translate_polygon(&rotated, px, py);
                    if polygon_contains(&placed_piece, stock) && !placed.iter().any(|p| polygon_overlap(&placed_piece, p)) {
                        best_pos = Some((px, py, angle));
                        break;
                    }
                }
                if best_pos.is_some() { break; }
            }
            if best_pos.is_some() { break; }
        }
        if let Some(pos) = best_pos { return Some(pos); }
    }
    None
}

from math import cos, sin, pi

def transform(poly, x, y, angle):
    cx = sum(p[0] for p in poly) / len(poly)
    cy = sum(p[1] for p in poly) / len(poly)
    
    centered = []
    for p in poly:
        dx = p[0] - cx
        dy = p[1] - cy
        rx = dx * cos(angle) - dy * sin(angle)
        ry = dx * sin(angle) + dy * cos(angle)
        centered.append((rx, ry))
    
    final = []
    for p in centered:
        final.append((p[0] + x, p[1] + y))
    
    return final

def polygon_area(poly):
    if len(poly) < 3:
        return 0.0
    area = 0.0
    n = len(poly)
    for i in range(n):
        j = (i + 1) % n
        area += poly[i][0] * poly[j][1] - poly[j][0] * poly[i][1]
    return abs(area / 2.0)

def point_in_polygon(p, poly):
    if not poly:
        return False
    inside = False
    n = len(poly)
    for i in range(n):
        x1, y1 = poly[i]
        x2, y2 = poly[(i+1)%n]
        if (y1 > p[1]) != (y2 > p[1]):
            x_intersect = x1 + (p[1] - y1) * (x2 - x1) / (y2 - y1)
            if p[0] < x_intersect:
                inside = not inside
    return inside

def edges_intersect(e1, e2):
    x1, y1, x2, y2 = e1
    x3, y3, x4, y4 = e2
    denom = (y4-y3)*(x2-x1) - (x4-x3)*(y2-y1)
    if abs(denom) < 1e-10:
        return False
    ua = ((x4-x3)*(y1-y3) - (y4-y3)*(x1-x3)) / denom
    ub = ((x2-x1)*(y1-y3) - (y2-y1)*(x1-x3)) / denom
    return (ua > 1e-10 and ua < 1.0 - 1e-10) and (ub > 1e-10 and ub < 1.0 - 1e-10)

def polygons_overlap(poly1, poly2):
