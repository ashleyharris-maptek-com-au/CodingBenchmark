Schedule greedy_schedule(const vector<Job>& jobs, int num_machines) {
    int num_jobs = jobs.size();
    int num_tasks_total = 0;
    for (const auto& job : jobs) {
        num_tasks_total += job.tasks.size();
    }
    
    Schedule schedule(num_jobs, num_machines);

        int current_time = 0;

            const Task& task = jobs[j].tasks[t];
