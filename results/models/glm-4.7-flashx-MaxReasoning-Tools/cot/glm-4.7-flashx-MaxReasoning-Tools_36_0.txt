```
struct Bitset {
    vector<uint64_t> data;
    int n;
    Bitset(int sz) : n(sz), data((sz+63)/64, 0) {}
    void set(int i) {
        data[i/64] |= (1ULL << (i & 63));
    }
    void reset(int i) {
        data[i/64] &= ~(1ULL << (i & 63));
    }
    bool test(int i) const {
        return (data[i/64] >> (i & 63)) & 1;
    }
    int count() const {
        int cnt = 0;
        for (auto &w : data) {
            cnt += __builtin_popcountll(w);
        }
        return cnt;
    }
    Bitset operator&(const Bitset &o) const {
        Bitset res(n);
        for (size_t i=0;i<data.size();++i)
            res.data[i] = data[i] & o.data[i];
        return res;
    }
    Bitset operator~() const {
        Bitset res(n);
        for (size_t i=0;i<data.size();++i)
            res.data[i] = ~data[i] & ((1ULL << (n%64)) - 1); // Not needed if using padding bits to zero.
        return res;
    }
};
```

### Implementation for n <= 64 (bitmask MIS algorithm)

We'll implement:

```
struct Node {
    int idx;
    int degree;
};
vector<Node> nodes(n);
for (i=0; i<n; ++i) {
    nodes[i].idx = i;
    nodes[i].degree = adj[i].size();
}
sort(nodes.begin(), nodes.end(), [](const Node& a, const Node& b){ return a.degree > b.degree; });
vector<int> order;
order.reserve(n);
for (auto& n: nodes) order.push_back(n.idx);

int adj_mask[n];
for (int i=0; i<n; ++i) {
    int mask = 0;
    for (int nb: adj[i]) mask |= 1 << nb;
    adj_mask[i] = mask;
}

int best_size = 0;
vector<int> best_set;
vector<int> cur_set;

function<void(int idx, int excluded_mask, int cur_size)> dfs = [&](int idx, int excluded_mask, int cur_size){
    if (cur_size + __builtin_popcount(((1ULL<<n)-1) ^ excluded_mask) <= best_size) return;
    // prune based on degree bound: find max degree among remaining
    // compute remaining mask
    int remaining = ((1ULL<<n)-1) ^ excluded_mask;
    int max_deg = 0;
    // naive scanning over vertices not excluded
    for (int v=0; v<n; ++v) {
        if (!(excluded_mask>>v &1)) {
            max_deg = max(max_deg, nodes[v].degree);
        }
    }
    int ub = cur_size + __builtin_popcount(remaining) - max_deg;
    if (ub <= best_size) return;

    while (idx < n) {
        int v = order[idx];
        if (!(excluded_mask>>v &1)) {
            cur_set.push_back(v);
            dfs(idx+1, excluded_mask | adj_mask[v] | (1<<v), cur_size+1);
            cur_set.pop_back();
        }
        idx++;
    }
    if (cur_size > best_size) {
        best_size = cur_size;
        best_set = cur_set;
    }
};

dfs(0,0,0);
```

Note: The recursion enumerates all subsets based on order, which avoids duplicates (excluded_mask includes already excluded vertices). This ensures each subset considered once.

Time complexity: worst-case exponential but for n <= 64 it's okay for typical tests.

### Implementation for n > 64 (Tomita algorithm on complement)

We'll use `std::vector<uint64_t>` for adjacency bitsets.

Steps:

- Use `using AdjMask = std::vector<uint64_t>;`
- For each vertex v, compute `adj_mask[v]` (original adjacency). We can store `orig_adj_mask[v]`.

- Build `orig_adj_mask[v]`:

```
vector<AdjMask> orig_adj(n);
for (int i=0; i<n; ++i) orig_adj[i].assign((n+63)/64, 0);
for each edge (u,v):
    int ui=u/64, vi=v/64, ubit=u&63, vbit=v&63;
    orig_adj[u][ui] |= (1ULL << ubit);
    orig_adj[v][vi] |= (1ULL << vbit);
```

- Build complement adjacency: `comp_adj[v]` = all bits not in `orig_adj[v]` and not self-bit.

We can compute `comp_adj[v] = full_without_self ^ orig_adj[v]`, where `full_without_self` is vector of chunks of all bits except v.

Implementation of `full_without_self` for each vertex v:

```
vector<AdjMask> full_without_self(n);
for (int v=0; v<n; ++v) {
    int chunks = (n+63)/64;
    full_without_self[v].assign(chunks, (1ULL<<64)-1);
    int vi = v/64, vbit = v&63;
    full_without_self[v][vi] &= ~(1ULL << vbit);
}
```

Then `comp_adj[v] = full_without_self[v] ^ orig_adj[v];` (XOR flips bits in orig_adj). This yields complement adjacency.

Alternatively, we could compute per chunk: `comp_adj[v][k] = (~orig_adj[v][k]) & full_bit`, with appropriate self-bit handling.

Note: Since we are using XOR, for each chunk, we need to ensure that bits are flipped for orig_adj entries, leaving all other bits as 1 (except self-bit). Using XOR with full_without_self works because full_without_self is all ones except self-bit; XOR with orig_adj will invert orig_adj bits.

Now we have complement adjacency bitsets.

Compute degree array `deg_comp[v]` = popcount(comp_adj[v]).

Now implement Tomita algorithm using `AdjMask` for P_mask and X_mask. Since they are bitsets across chunks, we need functions for operations:

- `lowest_bit_set(const AdjMask& mask)`: find the first bit set (lowest index). Use loop over chunks: for k in [0..C-1], if mask[k] != 0, then bit index = k*64 + __builtin_ctzll(mask[k]).

- `mask_and(const AdjMask& a, const AdjMask& b)`: result each chunk = a[k] & b[k].

- `mask_andnot(const AdjMask& a, const AdjMask& b)`: result each chunk = a[k] & ~b[k].

```
AdjMask P_mask(C), X_mask(C);
mask_fill(P_mask, n);

function<void(const AdjMask&, const AdjMask&)> expand = [&](const AdjMask& P_mask, const AdjMask& X_mask) {
    if (popcount(P_mask) == 0 && popcount(X_mask) == 0) {
        if (R.size() > best_size) {
            best_size = R.size();
            best_set = R;
        }
        return;
    }
    // simple bound
    if (R.size() + popcount(P_mask) <= best_size) return;
    // choose pivot u from P ∪ X with max degree in P (i.e., max popcount(comp_adj[u] & P_mask))
    // We'll just pick vertex with max deg_comp among P ∪ X (simple)
    // Actually Tomita chooses pivot with max |N(u) ∩ P|
    // We'll compute that.
    // We'll iterate over vertices in P|X, compute popcount(comp_adj[v] & P_mask)
    // This scanning can be heavy but okay.
    int u = -1;
    int max_adj_to_P = -1;
    AdjMask combined = P_mask; // Actually need P_mask | X_mask
    combined = mask_or(combined, X_mask);
    for (int v = 0; v < n; ++v) {
        if (test_bit(combined, v)) {
            int adj = popcount(comp_adj[v] & P_mask);
            if (adj > max_adj_to_P) {
                max_adj_to_P = adj;
                u = v;
            }
        }
    }
    // iterate over vertices v in P that are neighbors of u in complement (i.e., comp_adj[u] & P_mask)
    AdjMask neighbor_u = comp_adj[u] & P_mask;
    while (neighbor_u) {
        int v = lowest_bit_set(neighbor_u);
        neighbor_u = mask_andnot(neighbor_u, AdjMask(1ULL << v)); // Actually we need to remove the bit we process
        // branch include v
        R.push_back(v);
        AdjMask new_P = P_mask & comp_adj[v];
        AdjMask new_X = X_mask & comp_adj[v];
        expand(new_P, new_X);
        R.pop_back();
        // remove v from P, add to X
        P_mask = mask_andnot(P_mask, AdjMask(1ULL << v));
        X_mask = mask_or(X_mask, AdjMask(1ULL << v));
    }
};
```

```
#include <bits/stdc++.h>
using namespace std;

typedef unsigned long long ULL;

struct AdjMask {
    vector<ULL> bits;
    AdjMask() = default;
    AdjMask(int c) : bits(c, 0) {}
    void fill(int n) {
        int c = bits.size();
        for (int k = 0; k < c - 1; ++k) bits[k] = (ULL)(-1);
        int last = n & 63;
        bits[c-1] = last ? ((1ULL << last) - 1) : 0ULL;
    }
    int size() const { return bits.size(); }
    bool operator[](int i) const {
        int k = i >> 6;
        int b = i & 63;
        return (bits[k] >> b) & 1ULL;
    }
    void set(int v, bool val = true) {
        int k = v >> 6;
        int b = v & 63;
        if (val) bits[k] |= (1ULL << b);
        else bits[k] &= ~(1ULL << b);
    }
    void setAll(bool val = true) {
        for (auto& x : bits) x = val ? (ULL)(-1) : 0ULL;
    }
    void clear() { for (auto& x : bits) x = 0ULL; }
    void flip() {
        for (auto& x : bits) x = ~x;
    }
};

int popcount(const AdjMask& m) {
    int cnt = 0;
    for (auto x : m.bits) cnt += __builtin_popcountll(x);
    return cnt;
}

int lowest_bit(const AdjMask& m) {
    for (size_t k = 0; k < m.bits.size(); ++k) {
        if (m.bits[k]) {
            int b = __builtin_ctzll(m.bits[k]);
            return k * 64 + b;
        }
    }
    return -1;
}

AdjMask setbit(int v) {
    int c = (v + 63) / 64;
    AdjMask res(c);
    int k = v >> 6, b = v & 63;
    res.bits[k] |= (1ULL << b);
    return res;
}

AdjMask mask_or(const AdjMask& a, const AdjMask& b) {
    AdjMask res(a);
    for (size_t k = 0; k < res.bits.size(); ++k)
        res.bits[k] |= b.bits[k];
    return res;
}

AdjMask mask_and(const AdjMask& a, const AdjMask& b) {
    AdjMask res(a);
    for (size_t k = 0; k < res.bits.size(); ++k)
        res.bits[k] &= b.bits[k];
    return res;
}

AdjMask mask_andnot(const AdjMask& a, const AdjMask& b) {
    AdjMask res(a);
    for (size_t k = 0; k < res.bits.size(); ++k)
        res.bits[k] &= ~b.bits[k];
    return res;
}

AdjMask mask_not(const AdjMask& a) {
    AdjMask res(a);
    for (auto& x : res.bits) x = ~x;
    return res;
}

AdjMask mask_fill(int n) {
    int c = (n + 63) / 64;
    AdjMask res(c);
    res.fill(n);
    return res;
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    if (n == 0) {
        cout << "0\n";

    vector<set<int>> adj_set(n);
    for (int i = 0; i < m; ++i) {

    // For small n <= 64, use direct bitmask algorithm on original graph

        for (int i = 0; i < n; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            int da = 0, db = 0;
            for (int nb : adj_set[a]) ++da;
            for (int nb : adj_set[b]) ++db;
            return da > db;

        for (int i = 0; i < n; ++i) deg[i] = adj_set[i].size();

        for (int i = 0; i < n; ++i) {
            uint64_t mask = 0;
            for (int nb : adj_set[i]) mask |= 1ULL << nb;
            adj_mask[i] = mask;
        }
        vector<int> best_set;
        int best_size = 0;
        vector<int> cur_set;
        auto dfs = [&](int idx, uint64_t excluded, int cur_sz) {
            // bound: size + remaining <= best_size
            if (cur_sz + __builtin_popcountll(((1ULL << n) - 1) ^ excluded) <= best_size) return;
            // compute max degree among remaining vertices
            int max_deg = 0;
            for (int v = 0; v < n; ++v) {
                if (!(excluded & (1ULL << v))) {
                    max_deg = max(max_deg, deg[v]);
                }
            }
            int ub = cur_sz + __builtin_popcountll(((1ULL << n) - 1) ^ excluded) - max_deg;
            if (ub <= best_size) return;
            for (int i = idx; i < n; ++i) {
                int v = order[i];

                    cur_set.push_back(v);
                    dfs(i + 1, excluded | adj_mask[v] | (1ULL << v), cur_sz + 1);
                    cur_set.pop_back();
                }

            if (cur_sz > best_size) {
                best_size = cur_sz;
                best_set = cur_set;

            cout << best_set[i] << (i + 1 == best_set.size() ? "\n" : " ");

    // For larger n, use complement graph + Tomita algorithm
    // Build complement adjacency bitsets
    vector<AdjMask> comp_adj(n, AdjMask(C));
    for (int v = 0; v < n; ++v) {

        for (int i = 0; i < C; ++i) {

            else comp_adj[v].bits[i] = (1ULL << 64) - 1;

    vector<int> order(n);
    for (int i = 0; i < n; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) { return deg_comp[a] > deg_comp[b]; });

    AdjMask P_mask = mask_fill(n);
    AdjMask X_mask; X_mask.clear();
    vector<int> R; // current clique
    vector<int> best_set;
    int best_size = 0;
    auto expand = [&](const AdjMask& P, const AdjMask& X) {
        if (popcount(P) == 0 && popcount(X) == 0) {
            if (R.size() > best_size) {
                best_size = R.size();
                best_set = R;
            }
            return;
        }
        // simple bound
        if (R.size() + popcount(P) <= best_size) return;
        // choose pivot u from P ∪ X with max degree in P
        int u = -1;
        int max_adj_to_P = -1;
        AdjMask combined = mask_or(P, X);
        for (int v = 0; v < n; ++v) {
            if (combined[v]) {
                int adj = popcount(mask_and(comp_adj[v], P));
                if (adj > max_adj_to_P) {
                    max_adj_to_P = adj;
                    u = v;
                }

        // iterate over vertices v in P that are neighbors of u in complement
        AdjMask neighbor_u = mask_and(comp_adj[u], P);
        while (neighbor_u) {
            int v = lowest_bit(neighbor_u);
            neighbor_u = mask_andnot(neighbor_u, setbit(v));
            // include v
            R.push_back(v);
            AdjMask new_P = mask_and(P, comp_adj[v]);
            AdjMask new_X = mask_and(X, comp_adj[v]);
            expand(new_P, new_X);
            R.pop_back();

            P = mask_andnot(P, setbit(v));
            X = mask_or(X, setbit(v));
        }

    // output best_set
    cout << best_size << "\n";
    for (size_t i = 0; i < best_set.size(); ++i) {

    }
}
```
