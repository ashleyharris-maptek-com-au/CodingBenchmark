## Key Concepts

**ILS signals:**
- `glideslope_N_dots`: deviation from 3° glideslope (+above/-below)
- `localizer_N_dots`: deviation from runway centerline (+right/-left)

**Controls:**
- `elevator`: -1.0 (nose down) to +1.0 (nose up) relative to trim
- `aileron`: -1.0 (roll left) to +1.0 (roll right)
- `rudder`: -1.0 (yaw left) to +1.0 (yaw right)
- `throttle_1/2`: 0.0 to 1.0
- `elevator_trim`: -1.0 to +1.0
- `flaps`: degrees (0-40)
- `gear`: 0.0 (up) or 1.0 (down)
- `speedbrake`: 0.0 to 1.0
- `spoiler`: 0.0 or 1.0
- `autobrake`: 0.0 to 1.0
- `reverse_1/2`: 0.0 to 1.0

**Stabilization criteria:**
- Speed within 10kt of target
- Glide slope within 1/3 dot (±3.5°)
- Localizer within 1/3 dot (±1.75°)
- Vertical speed -700 to -800 fpm
- On glide path (1000ft-200ft)

**Flare:**
- Start when radio altitude < 50ft
- Target touchdown speed = target_vref_kt
- Level out with 0° pitch

**Touchdown:**

    """Get valid value from dicts, use average of valid readings"""
    values = []
    for d in dicts:
        if key in d and key + "_valid" in d and d[key + "_valid"]:

### Control Laws

#### Phase 1: Intercept (>1000ft)
- Set flaps (typically 30° by 1000ft)
- Lower gear (typically 1,000 ft AGL)
- Track glideslope and localizer

Glide slope correction:
- If above glideslope (+dots), pitch down
- If below glideslope (-dots), pitch up

Localizer correction:
- If right of centerline (+dots), roll right
- If left of centerline (-dots), roll left

Speed control:
- Use throttle to maintain target IAS
- Use elevator for pitch

#### Phase 2: Approach (1000ft-200ft)
- Stabilize
- Use flaps to 40°
- Trim for level flight on approach

#### Phase 3: Flare (<50ft)
- Reduce descent rate to 0
- Target touchdown speed (vref)
- Pitch up to level out

#### Phase 4: Touchdown & Rollout
- Deploy spoilers
- Deploy reverse thrust
- Deploy autobrake
- Hold runway heading

### Go-Around Logic
- If unstable or unsafe, abort landing
- TOGA thrust

    target_alt_ft = sensors.get('target_alt_ft', 0)
    
    # Flap and gear configuration
    flaps = 15
    gear = 1.0
    
    # Get sensor values
    gs_dots = get_redundant_value(sensors, 'glideslope_1_dots', 'glideslope_1_valid')
    gs_valid = any(sensors.get(f'glideslope_{i}_valid', False) for i in range(1, 4))
    loc_dots = get_redundant_value(sensors, 'localizer_1_dots', 'localizer_1_valid')
    loc_valid = any(sensors.get(f'localizer_{i}_valid', False) for i in range(1, 4))
    ias_1 = sensors.get('ias_1_kt', 0)
    ias_valid = sensors.get('ias_1_valid', False)
    vs_fpm = get_irs_average()['vs_fpm']
    
    # Calculate control commands
    elevator_cmd = 0
    aileron_cmd = 0
    rudder_cmd = 0
    throttle_1 = 0.5
    throttle_2 = 0.5
    elevator_trim = 0
    
    # Glide slope control (correct pitch)
    if gs_valid:
        gs_error = gs_dots - target_gs_dots  # Positive = above, negative = below
        pitch_correction = gs_error * 0.3
        elevator_cmd += pitch_correction
    
    # Localizer control (correct roll)
    if loc_valid:
        loc_error = loc_dots - target_loc_dots  # Positive = right, negative = left
        roll_correction = loc_error * 0.2
        aileron_cmd -= roll_correction
    
    # Speed control
    if ias_valid:
        speed_error = ias_1 - target_ias_kt
        throttle_cmd = 0.5 + speed_error * 0.01
        throttle_1 = throttle_cmd
        throttle_2 = throttle_cmd
    else:

        throttle_1 = s['last_throttle_1']
        throttle_2 = s['last_throttle_2']

    elevator_trim = -vs_fpm * 0.0005

    elevator_cmd = min(max(elevator_cmd, -0.5), 0.5)

    throttle_1 = min(max(throttle_1, 0), 1)
    throttle_2 = min(max(throttle_2, 0), 1)
    elevator_trim = min(max(elevator_trim, -0.3), 0.3)
    
    return {
        'elevator': elevator_cmd,
        'aileron': aileron_cmd,
        'rudder': rudder_cmd,
        'throttle_1': throttle_1,
        'throttle_2': throttle_2,
        'elevator_trim': elevator_trim,
        'flaps': flaps,
        'gear': gear,
        'speedbrake': 0,
        'spoiler': 0,

def approach_phase(sensors, dt, s):
    """Phase 2: Stabilize on approach at 1000ft-200ft"""
    target_ias_kt = sensors.get('target_ias_kt', 137)
    target_vref_kt = sensors.get('target_vref_kt', 137)
    
    # Full flap configuration
    flaps = 40
    gear = 1.0
    
    # Get sensor values
    gs_dots = get_redundant_value(sensors, 'glideslope_1_dots', 'glideslope_1_valid')
    gs_valid = any(sensors.get(f'glideslope_{i}_valid', False) for i in range(1, 4))
    loc_dots = get_redundant_value(sensors, 'localizer_1_dots', 'localizer_1_valid')
    loc_valid = any(sensors.get(f'localizer_{i}_valid', False) for i in range(1, 4))
    ias_1 = sensors.get('ias_1_kt', 0)
    ias_valid = sensors.get('ias_1_valid', False)
    vs_fpm = get_irs_average()['vs_fpm']
    
    # Calculate control commands
    elevator_cmd = 0
    aileron_cmd = 0
    rudder_cmd = 0
    throttle_1 = 0.5
    throttle_2 = 0.5
    elevator_trim = 0
    
    # Glide slope control (tighter control in approach)
    if gs_valid:
        gs_error = gs_dots - target_gs_dots
        pitch_correction = gs_error * 0.5
        elevator_cmd += pitch_correction
    
    # Localizer control
    if loc_valid:
        loc_error = loc_dots - target_loc_dots
        roll_correction = loc_error * 0.3
        aileron_cmd -= roll_correction
    
    # Speed control - aim for vref
    target_throttle = 0.5
    if ias_valid:
        speed_error = ias_1 - target_vref_kt
        target_throttle = 0.5 - speed_error * 0.01
    else:
        target_throttle = 0.5
    
    # Smooth throttle adjustment
    throttle_1 = s['last_throttle_1'] + (target_throttle - s['last_throttle_1']) * dt
    throttle_2 = s['last_throttle_2'] + (target_throttle - s['last_throttle_2']) * dt
    
    # Trim elevator for level approach
    elevator_trim = -vs_fpm * 0.0008
    
    # Rudder for yaw stability
    yaw_correction = s['throttle_diff'] * 0.1
    rudder_cmd = yaw_correction
    
    # Limit commands
    elevator_cmd = min(max(elevator_cmd, -0.4), 0.4)
    aileron_cmd = min(max(aileron_cmd, -0.2), 0.2)
    rudder_cmd = min(max(rudder_cmd, -0.2), 0.2)
    throttle_1 = min(max(throttle_1, 0), 1)
    throttle_2 = min(max(throttle_2, 0), 1)
    elevator_trim = min(max(elevator_trim, -0.2), 0.2)
    
    return {
        'elevator': elevator_cmd,
        'aileron': aileron_cmd,
        'rudder': rudder_cmd,
        'throttle_1': throttle_1,
        'throttle_2': throttle_2,
        'elevator_trim': elevator_trim,
        'flaps': flaps,
        'gear': gear,
        'speedbrake': 0,
        'spoiler': 0,
        'autobrake': 0,
        'reverse_1': 0,
        'reverse_2': 0,
    }

def flare_phase(sensors, dt, s):
    """Phase 3: Flare at < 50ft radio altitude"""
    target_vref_kt = sensors.get('target_vref_kt', 137)
    
    flaps = 40
    gear = 1.0
    
    # Get sensor values
    vs_fpm = get_irs_average()['vs_fpm']
    ias_1 = sensors.get('ias_1_kt', 0)
    ias_valid = sensors.get('ias_1_valid', False)
    radio_alt_ft = sensors.get('radio_alt_ft', 0)
    weight_on_wheels = sensors.get('weight_on_wheels', False)
    
    # Calculate control commands
    elevator_cmd = 0
    aileron_cmd = 0
    rudder_cmd = 0
    throttle_1 = 0.0
    throttle_2 = 0.0
    elevator_trim = 0
    
    # Flare: reduce descent rate
    if vs_fpm < -200:
        target_vs = -150
    elif vs_fpm < 0:
        target_vs = -50
    else:
        target_vs = 0
    
    pitch_error = -vs_fpm - target_vs
    elevator_cmd = pitch_error * 0.002
    
    # Maintain target speed
    target_throttle = 0.0
    if ias_valid:
        speed_error = ias_1 - target_vref_kt
        if abs(speed_error) < 20:
            target_throttle = 0.0  # Idle
        else:
            target_throttle = 0.3  # Some thrust if too slow
    
    throttle_1 = target_throttle
    throttle_2 = target_throttle
    
    # Increase elevator for flare if needed
    if vs_fpm < -500 and radio_alt_ft < 50:
        elevator_cmd += 0.1
    
    # Trim elevator for level flight
    elevator_trim = elevator_cmd * 0.7
    
    # Rudder for yaw stability
    yaw_correction = s['throttle_diff'] * 0.1
    rudder_cmd = yaw_correction
    
    # Limit commands
    elevator_cmd = min(max(elevator_cmd, -0.3), 0.5)
    aileron_cmd = min(max(aileron_cmd, -0.2), 0.2)
    rudder_cmd = min(max(rudder_cmd, -0.2), 0.2)
    elevator_trim = min(max(elevator_trim, -0.1), 0.1)
    
    return {
        'elevator': elevator_cmd,
        'aileron': aileron_cmd,
        'rudder': rudder_cmd,
        'throttle_1': throttle_1,
        'throttle_2': throttle_2,
        'elevator_trim': elevator_trim,
        'flaps': flaps,
        'gear': gear,
        'speedbrake': 0,
        'spoiler': 0,
        'autobrake': 0,
        'reverse_1': 0,
        'reverse_2': 0,
    }

def rollout_phase(sensors, dt, s):
    """Phase 4: Touchdown and rollout"""
    groundspeed_kt = sensors.get('groundspeed_kt', 0)
    weight_on_wheels = sensors.get('weight_on_wheels', False)
    radio_alt_ft = sensors.get('radio_alt_ft', 0)
    
    # Full flaps and gear
    flaps = 40
    gear = 1.0
    
    # Deploy spoilers on touchdown
    spoiler = 1.0 if weight_on_wheels else 0
    
    # Autobrake based on groundspeed
    if groundspeed_kt > 100:
        autobrake = 0.5
    elif groundspeed_kt > 50:
        autobrake = 0.8
    else:
        autobrake = 1.0
    
    # Idle thrust on ground
    throttle_1 = 0.0
    throttle_2 = 0.0
    
    # Reverse thrust if speed is high
    reverse_1 = 0.0
    reverse_2 = 0.0
    if groundspeed_kt > 20 and weight_on_wheels:
        reverse_1 = 1.0
        reverse_2 = 1.0
    
    # Keep aligned with runway (control yaw with rudder)
    rudder_cmd = 0
    
    # Limit commands
    aileron_cmd = min(max(aileron_cmd, -0.1), 0.1)
    rudder_cmd = min(max(rudder_cmd, -0.2), 0.2)
    
    return {
        'elevator': 0,
        'aileron': aileron_cmd,
        'rudder': rudder_cmd,
        'throttle_1': throttle_1,
        'throttle_2': throttle_2,
        'elevator_trim': 0,
        'flaps': flaps,
        'gear': gear,
        'speedbrake': 0,
        'spoiler': spoiler,
        'autobrake': autobrake,
        'reverse_1': reverse_1,
        'reverse_2': reverse_2,
    }
