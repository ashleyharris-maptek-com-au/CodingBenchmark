```python
def autopilot_step(state, dt):
    # globals
    if not hasattr(autopilot_step, "state"):
        autopilot_step.state = AutopilotState()
    A = autopilot_step.state

    # Extract data
    def get_float(key, default=None):
        v = state.get(key)
        if isinstance(v, (int, float)):
            return float(v)
        return default

    # Wheel speeds
    ws = []
    for name in ["wheel_speed_fl", "wheel_speed_fr", "wheel_speed_rl", "wheel_speed_rr"]:
        speed = get_float(name + "_kmh")
        valid = get_float(name + "_valid")
        if valid and speed is not None:
            if 0 <= speed <= 250:
                ws.append(speed)
    wheel_speed = statistics.median(ws) if ws else None

    # GPS speeds
    gps_speeds = []
    for i in range(1,3):
        valid = get_float(f"gps_{i}_valid")
        speed = get_float(f"gps_{i}_speed_kmh")
        if valid and speed is not None and 0 <= speed <= 250:
            gps_speeds.append(speed)
    gps_speed = statistics.median(gps_speeds) if gps_speeds else None

    # Combine speeds
    current_speed = wheel_speed if wheel_speed is not None else (gps_speed if gps_speed is not None else A.prev_speed)

    # Yaw rate
    gyros = []
    for i in range(1,4):
        valid = get_float(f"gyro_{i}_valid")
        yaw = get_float(f"gyro_{i}_yaw_rate_dps")
        if valid and yaw is not None:
            if -15 <= yaw <= 15:
                gyros.append(yaw)
    yaw_rate = statistics.median(gyros) if gyros else None

    # Lane offset
    lane_offsets = []
    for i in range(1,4):
        valid = get_float(f"vision_{i}_lane_offset_valid")
        offset = get_float(f"vision_{i}_lane_offset_m")
        if valid and offset is not None:
            if -5 <= offset <= 5:
                lane_offsets.append(offset)
    lane_offset = statistics.median(lane_offsets) if lane_offsets else None

    # Left/right line distances
    left_distances = []
    right_distances = []
    for i in range(1,4):
        valid_l = get_float(f"vision_{i}_left_line_valid")
        left = get_float(f"vision_{i}_left_line_dist_m")
        if valid_l and left is not None and left >= 0:
            left_distances.append(left)
        valid_r = get_float(f"vision_{i}_right_line_valid")
        right = get_float(f"vision_{i}_right_line_dist_m")
        if valid_r and right is not None and right >= 0:
            right_distances.append(right)
    left_line_dist = statistics.median(left_distances) if left_distances else None
    right_line_dist = statistics.median(right_distances) if right_distances else None

    # Lane types
    left_line_types = []
    for i in range(1,4):
        valid = get_float(f"vision_{i}_left_line_valid")
        typ = state.get(f"vision_{i}_left_line_type")
        if valid and typ:
            left_line_types.append(typ)
    # etc.

    # Speed limits
    speed_limits = []
    for i in range(1,4):
        valid = get_float(f"vision_{i}_speed_limit_valid")
        limit = get_float(f"vision_{i}_speed_limit_kmh")
        if valid and limit is not None:
            if 0 <= limit <= 200:
                speed_limits.append(limit)
    speed_limit = statistics.median(speed_limits) if speed_limits else None

    # Obstacles
    obstacles = []
    for i in range(1,4):
        valid = get_float(f"vision_{i}_obstacle_valid")
        detected = state.get(f"vision_{i}_obstacle_detected")
        if valid and detected:
            dist = get_float(f"vision_{i}_obstacle_dist_m")
            rel_speed = get_float(f"vision_{i}_obstacle_rel_speed_kmh")
            if dist is not None and rel_speed is not None:
                obstacles.append((dist, rel_speed))
    # find closest
    closest_obstacle = min(obstacles, key=lambda x: x[0]) if obstacles else None

    # Traction
    mu = get_float("estimated_mu")
    rain = get_float("rain_sensor")
    temp = get_float("temperature_c")
    # Use mu for traction factor
    mu_factor = 1.0
    if mu is not None:
        if mu < 0.3:
            mu_factor = 0.4
        elif mu < 0.5:
            mu_factor = 0.7

    # Steering angle sensors
    steering_angles = []
    for i in range(1,3):
        valid = get_float(f"steering_angle_{i}_valid")
        angle = get_float(f"steering_angle_{i}_deg")
        if valid and angle is not None:
            if -40 <= angle <= 40:
                steering_angles.append(angle)
    steering_feedback = statistics.median(steering_angles) if steering_angles else None

    # Determine target speed and target lane
    target_lane = get_float("target_lane")  # expects int 0-2
    target_speed = get_float("target_speed_kmh")
    # use speed limit if lower
    if speed_limit is not None and target_speed > speed_limit:
        target_speed = speed_limit

    # Apply mu factor
    target_speed = target_speed * mu_factor
    # Ensure non-negative
    target_speed = max(0.0, target_speed)

    # Compute current lane index (0-2)
    lane_width = 3.5
    if lane_offset is None:
        # estimate from line distances
        if left_line_dist is not None and right_line_dist is not None:
            lane_offset_est = (left_line_dist - right_line_dist) / 2.0
            lane_offset = lane_offset_est
        else:
            lane_offset = 0.0
    current_lane = int(round(lane_offset / lane_width))
    # Clip to 0-2
    current_lane = max(0, min(2, current_lane))

    # Lateral control error
    desired_lane_offset = 0.0
    if target_lane != current_lane:
        desired_lane_offset = (target_lane - current_lane) * lane_width
    lat_error = desired_lane_offset - lane_offset

    # Obstacle avoidance
    safe_distance = 30.0 + current_speed * 0.5  # meters, approximate
    obstacle_brake = False
    lane_change_needed = False
    if closest_obstacle:
        dist, rel_speed = closest_obstacle
        # relative_speed positive means closing in (our speed slower)
        if rel_speed > 0 and dist < safe_distance:
            obstacle_brake = True
        # also consider front obstacles that are far but lane change possible
        if rel_speed > 0 and dist < 50.0 and target_lane != current_lane:
            lane_change_needed = True
    # total brake if obstacle brake true and speed > 0
    brake_cmd = 0.0
    throttle_cmd = 0.0
    steering_cmd = 0.0

    # Traction: limit throttle
    throttle_max = 1.0 * mu_factor

    if obstacle_brake:
        brake_cmd = 1.0  # full stop
        # if very close, disable throttle and steering
        throttle_cmd = 0.0
    elif lane_change_needed:
        # decelerate for lane change
        throttle_cmd = 0.0
        brake_cmd = 0.0
        steering_cmd = 1.0 if target_lane > current_lane else -1.0
    else:
        # Normal speed control
        # PID for speed
        speed_error = target_speed - current_speed
        A.speed_integral += speed_error * dt
        # clamp integral
        A.speed_integral = max(-5.0, min(5.0, A.speed_integral))
        speed_out = 5.0 * speed_error + 2.0 * (speed_error - A.speed_error_prev) / dt
        A.speed_error_prev = speed_error
        throttle_cmd = max(0.0, min(throttle_max, speed_out))

        # Lateral control PID
        A.lat_integral += lat_error * dt
        # clamp integral
        A.lat_integral = max(-2.0, min(2.0, A.lat_integral))
        lat_out = 2.0 * lat_error + 3.0 * (lat_error - A.lat_error_prev) / dt + 0.5 * A.lat_integral
        A.lat_error_prev = lat_error
        # add yaw rate damping
        if yaw_rate is not None:
            lat_out += 0.5 * yaw_rate  # yaw_rate in deg/s, scaling needed
        max_steer_deg = 30.0  # deg
        steering_angle = max(-max_steer_deg, min(max_steer_deg, lat_out))
        steering_cmd = steering_angle / max_steer_deg

        # Apply lane change guidance override: if target lane different, adjust steering cmd
        if target_lane != current_lane:
            # shift steering towards target lane offset
            steering_cmd = max(-1.0, min(1.0, steering_cmd + 0.2 if target_lane > current_lane else steering_cmd - 0.2))

    # Add some smoothing / median filter for steering
    A.steering_queue.append(steering_cmd)
    if len(A.steering_queue) >= 5:
        steering_cmd = statistics.median(A.steering_queue)
    A.prev_steering = steering_cmd
    A.prev_throttle = throttle_cmd
    A.prev_brake = brake_cmd

    result = {}
    if steering_cmd != A.prev_steering or not (steering_cmd is None):
        result["steering"] = steering_cmd
    if throttle_cmd != A.prev_throttle:
        result["throttle"] = throttle_cmd
    if brake_cmd != A.prev_brake:
        result["brake"] = brake_cmd
    return result
```
