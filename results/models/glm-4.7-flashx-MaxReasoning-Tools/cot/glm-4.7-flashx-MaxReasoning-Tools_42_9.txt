```glsl
void main() {
    // Access gl_Position for all 4 vertices
    vec4 p0 = gl_in[0].gl_Position;
    vec4 p1 = gl_in[1].gl_Position;
    vec4 p2 = gl_in[2].gl_Position;
    vec4 p3 = gl_in[3].gl_Position;
    
    // Access worldPos and color
    vec3 worldPos1 = worldPos[1];
    vec3 worldPos2 = worldPos[2];
    vec3 color1 = color[1];
    vec3 color2 = color[2];
    
    // Compute tangents (in clip space)
    vec2 t0 = normalize(vec2(p2.x - p1.x, p2.y - p1.y));
    vec2 t1 = normalize(vec2(p3.x - p2.x, p3.y - p2.y));
    
    // Compute perpendiculars
    vec2 perp0 = vec2(-t0.y, t0.x);
    vec2 perp1 = vec2(-t1.y, t1.x);
    
    float thickness = 0.025;
    
    // Emit vertices as triangle strip
    // Start side 1 (positive)
    vec4 pos0 = p1;
    pos0.xy += perp0 * thickness;
    gl_out[0].gl_Position = pos0;
    outColor = color1;
    OpEmitVertex;
    
    // Start side 2 (negative)
    vec4 pos1 = p1;
    pos1.xy -= perp0 * thickness;
    gl_out[1].gl_Position = pos1;
    outColor = color1;
    OpEmitVertex;
    
    // End side 1 (positive)
    vec4 pos2 = p2;
    pos2.xy += perp1 * thickness;
    gl_out[2].gl_Position = pos2;
    outColor = color2;
    OpEmitVertex;
    
    // End side 2 (negative)
    vec4 pos3 = p2;
    pos3.xy -= perp1 * thickness;
    gl_out[3].gl_Position = pos3;
    outColor = color2;
    OpEmitVertex;
    
    OpEndPrimitive;
}
```
