use std::cmp::Ordering;
use std::collections::{BinaryHeap, VecDeque};
use std::io::{self, Read};

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin()
            .read_to_end(&mut data)
            .expect("failed to read stdin");
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            value = value
                .saturating_mul(10)
                .saturating_add((self.data[self.pos] - b'0') as usize);
            self.pos += 1;
        }
        value
    }
}

struct Graph {
    n: usize,
    offsets: Vec<usize>,
    adj: Vec<usize>,
    forced: Vec<bool>,
}

impl Graph {
    fn neighbors(&self, v: usize) -> &[usize] {
        &self.adj[self.offsets[v]..self.offsets[v + 1]]
    }
}

fn build_graph(n: usize, m: usize, input: &mut FastInput) -> Graph {
    let mut deg = vec![0usize; n];
    let mut forced = vec![false; n];
    let mut edges = Vec::with_capacity(m);

    for _ in 0..m {
        let u = input.next_usize();
        let v = input.next_usize();
        if u == v {
            if u < n {
                forced[u] = true;
            }
            continue;
        }
        if u < n && v < n {
            deg[u] += 1;
            deg[v] += 1;
            edges.push((u, v));
        }
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i];
    }

    let mut next = offsets[..n].to_vec();
    let mut adj = vec![0usize; offsets[n]];
    for (u, v) in edges {
        adj[next[u]] = v;
        next[u] += 1;
        adj[next[v]] = u;
        next[v] += 1;
    }

    Graph {
        n,
        offsets,
        adj,
        forced,
    }
}

struct SolverState<'a> {
    graph: &'a Graph,
    active: Vec<bool>,
    in_cover: Vec<bool>,
    deg: Vec<u32>,
    active_edges: usize,
}

impl<'a> SolverState<'a> {
    fn new(graph: &'a Graph) -> Self {
        let mut deg = vec![0u32; graph.n];
        for (v, d) in deg.iter_mut().enumerate() {
            *d = (graph.offsets[v + 1] - graph.offsets[v]) as u32;
        }
        Self {
            graph,
            active: vec![true; graph.n],
            in_cover: vec![false; graph.n],
            deg,
            active_edges: graph.adj.len() / 2,
        }
    }

    fn remove_vertex(&mut self, v: usize, take_into_cover: bool) {
        if !self.active[v] {
            return;
        }
        self.active[v] = false;
        if take_into_cover {
            self.in_cover[v] = true;
        }
        for &u in self.graph.neighbors(v) {
            if self.active[u] {
                self.active_edges -= 1;
                self.deg[u] -= 1;
            }
        }
        self.deg[v] = 0;
    }

    fn single_active_neighbor(&self, v: usize) -> Option<usize> {
        self.graph
            .neighbors(v)
            .iter()
            .copied()
            .find(|&u| self.active[u])
    }

    fn apply_initial_forced(&mut self) {
        for v in 0..self.graph.n {
            if self.graph.forced[v] {
                self.remove_vertex(v, true);
            }
        }
    }

    fn apply_reductions(&mut self) {
        let mut queue = VecDeque::new();
        for v in 0..self.graph.n {
            if self.active[v] && self.deg[v] <= 1 {
                queue.push_back(v);
            }
        }

        while let Some(v) = queue.pop_front() {
            if !self.active[v] {
                continue;
            }
            match self.deg[v] {
                0 => {
                    self.remove_vertex(v, false);
                }
                1 => {
                    if let Some(u) = self.single_active_neighbor(v) {
                        self.remove_vertex(u, true);
                        for &w in self.graph.neighbors(u) {
                            if self.active[w] && self.deg[w] <= 1 {
                                queue.push_back(w);
                            }
                        }
                        if self.active[v] && self.deg[v] <= 1 {
                            queue.push_back(v);
                        }
                    } else {
                        self.remove_vertex(v, false);
                    }
                }
                _ => {}
            }
        }
    }

    fn remaining_vertices(&self) -> Vec<usize> {
        let mut rem = Vec::new();
        rem.reserve(self.graph.n.min(self.active_edges.saturating_mul(2)));
        for v in 0..self.graph.n {
            if self.active[v] && self.deg[v] > 0 {
                rem.push(v);
            }
        }
        rem
    }

    fn greedy_finish(&mut self) {
        #[derive(Copy, Clone, Eq, PartialEq)]
        struct Entry {
            deg: u32,
            v: usize,
        }

        impl Ord for Entry {
            fn cmp(&self, other: &Self) -> Ordering {
                self.deg.cmp(&other.deg).then(self.v.cmp(&other.v))
            }
        }

        impl PartialOrd for Entry {
            fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
                Some(self.cmp(other))
            }
        }

        let mut heap = BinaryHeap::new();
        for v in 0..self.graph.n {
            if self.active[v] && self.deg[v] > 0 {
                heap.push(Entry {
                    deg: self.deg[v],
                    v,
                });
            }
        }

        while self.active_edges > 0 {
            let Some(top) = heap.pop() else {
                break;
            };
            if !self.active[top.v] || self.deg[top.v] != top.deg || top.deg == 0 {
                continue;
            }
            self.remove_vertex(top.v, true);
            for &u in self.graph.neighbors(top.v) {
                if self.active[u] && self.deg[u] > 0 {
                    heap.push(Entry {
                        deg: self.deg[u],
                        v: u,
                    });
                }
            }
        }
    }
}

fn greedy_local_cover(adj: &[u64], mut active: u64) -> (u32, u64) {
    let mut chosen = 0u64;
    let mut count = 0u32;
    while active != 0 {
        let mut best_v = None;
        let mut best_deg = 0u32;
        let mut scan = active;
        while scan != 0 {
            let bit = scan & (!scan + 1);
            let v = bit.trailing_zeros() as usize;
            let deg = (adj[v] & active).count_ones();
            if deg > best_deg {
                best_deg = deg;
                best_v = Some(v);
            }
            scan ^= bit;
        }
        if best_deg == 0 {
            break;
        }
        let v = best_v.expect("best vertex must exist");
        chosen |= 1u64 << v;
        count += 1;
        active &= !(1u64 << v);
    }
    (count, chosen)
}

fn local_matching_lower_bound(adj: &[u64], active: u64) -> u32 {
    let mut temp = active;
    let mut bound = 0u32;
    while temp != 0 {
        let u = temp.trailing_zeros() as usize;
        let nbrs = adj[u] & temp;
        if nbrs != 0 {
            let v = nbrs.trailing_zeros() as usize;
            temp &= !(1u64 << u);
            temp &= !(1u64 << v);
            bound += 1;
        } else {
            temp &= !(1u64 << u);
        }
    }
    bound
}

fn reduce_local_state(adj: &[u64], mut active: u64, mut chosen: u64, mut count: u32) -> (u64, u64, u32) {
    loop {
        let mut changed = false;
        let mut scan = active;
        while scan != 0 {
            let bit = scan & (!scan + 1);
            let u = bit.trailing_zeros() as usize;
            let nbrs = adj[u] & active;
            let deg = nbrs.count_ones();
            if deg == 0 {
                active &= !bit;
                changed = true;
                break;
            }
            if deg == 1 {
                let v = nbrs.trailing_zeros() as usize;
                let v_bit = 1u64 << v;
                if active & v_bit != 0 {
                    chosen |= v_bit;
                    count += 1;
                    active &= !v_bit;
                    changed = true;
                    break;
                }
            }
            scan ^= bit;
        }
        if !changed {
            break;
        }
    }
    (active, chosen, count)
}

fn exact_small_cover(adj: &[u64]) -> u64 {
    fn dfs(adj: &[u64], active: u64, chosen: u64, count: u32, best_count: &mut u32, best_mask: &mut u64) {
        let (active, chosen, count) = reduce_local_state(adj, active, chosen, count);
        if count >= *best_count {
            return;
        }

        let lower_bound = local_matching_lower_bound(adj, active);
        if count + lower_bound >= *best_count {
            return;
        }

        let mut pivot = None;
        let mut pivot_deg = 0u32;
        let mut scan = active;
        while scan != 0 {
            let bit = scan & (!scan + 1);
            let u = bit.trailing_zeros() as usize;
            let deg = (adj[u] & active).count_ones();
            if deg > pivot_deg {
                pivot_deg = deg;
                pivot = Some(u);
            }
            scan ^= bit;
        }

        if pivot_deg == 0 {
            *best_count = count;
            *best_mask = chosen;
            return;
        }

        let u = pivot.expect("pivot must exist");
        let nbrs = adj[u] & active;
        let v = nbrs.trailing_zeros() as usize;

        let deg_u = (adj[u] & active).count_ones();
        let deg_v = (adj[v] & active).count_ones();
        let (first, second) = if deg_u >= deg_v { (u, v) } else { (v, u) };

        let first_bit = 1u64 << first;
        dfs(
            adj,
            active & !first_bit,
            chosen | first_bit,
            count + 1,
            best_count,
            best_mask,
        );

        let second_bit = 1u64 << second;
        dfs(
            adj,
            active & !second_bit,
            chosen | second_bit,
            count + 1,
            best_count,
            best_mask,
        );
    }

    let n = adj.len();
    let active = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };
    let (mut best_count, mut best_mask) = greedy_local_cover(adj, active);
    dfs(adj, active, 0, 0, &mut best_count, &mut best_mask);
    best_mask
}

fn try_exact_on_remainder(state: &mut SolverState<'_>) -> bool {
    let remaining = state.remaining_vertices();
    if remaining.is_empty() {
        return true;
    }
    if remaining.len() > 62 {
        return false;
    }

    let mut map = vec![usize::MAX; state.graph.n];
    for (i, &v) in remaining.iter().enumerate() {
        map[v] = i;
    }

    let mut adj = vec![0u64; remaining.len()];
    for (i, &v) in remaining.iter().enumerate() {
        let mut bits = 0u64;
        for &u in state.graph.neighbors(v) {
            let idx = map[u];
            if idx != usize::MAX {
                bits |= 1u64 << idx;
            }
        }
        adj[i] = bits;
    }

    let chosen = exact_small_cover(&adj);
    for (i, &v) in remaining.iter().enumerate() {
        if (chosen >> i) & 1 == 1 {
            state.in_cover[v] = true;
        }
    }
    true
}

fn prune_cover(graph: &Graph, in_cover: &mut [bool]) {
    let mut queue = VecDeque::new();
    for (v, &selected) in in_cover.iter().enumerate() {
        if selected && !graph.forced[v] {
            queue.push_back(v);
        }
    }

    while let Some(v) = queue.pop_front() {
        if !in_cover[v] || graph.forced[v] {
            continue;
        }
        let mut removable = true;
        for &u in graph.neighbors(v) {
            if !in_cover[u] {
                removable = false;
                break;
            }
        }
        if removable {
            in_cover[v] = false;
        }
    }
}

fn repair_cover(graph: &Graph, in_cover: &mut [bool]) {
    for v in 0..graph.n {
        for &u in graph.neighbors(v) {
            if u > v && !in_cover[u] && !in_cover[v] {
                let pick = if graph.neighbors(u).len() > graph.neighbors(v).len() {
                    u
                } else {
                    v
                };
                in_cover[pick] = true;
            }
        }
    }
    for (v, forced) in graph.forced.iter().copied().enumerate() {
        if forced {
            in_cover[v] = true;
        }
    }
}

fn main() {
    let mut input = FastInput::new();
    if input.data.is_empty() {
        println!("0");
        println!();
        return;
    }

    let n = input.next_usize();
    let m = input.next_usize();
    let graph = build_graph(n, m, &mut input);

    let mut state = SolverState::new(&graph);
    state.apply_initial_forced();
    state.apply_reductions();

    if state.active_edges > 0 && !try_exact_on_remainder(&mut state) {
        state.greedy_finish();
    }

    repair_cover(&graph, &mut state.in_cover);
    prune_cover(&graph, &mut state.in_cover);
    repair_cover(&graph, &mut state.in_cover);

    let mut result = Vec::new();
    for (v, &selected) in state.in_cover.iter().enumerate() {
        if selected {
            result.push(v);
        }
    }

    println!("{}", result.len());
    if result.is_empty() {
        println!();
        return;
    }
    for (i, v) in result.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
