declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = Array.isArray(data.grid) ? data.grid : [];
const robots: Robot[] = Array.isArray(data.robots) ? data.robots : [];
const jobs: Job[] = Array.isArray(data.jobs) ? data.jobs : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;

const cellIndex = new Int32Array(width * Math.max(height, 1));
cellIndex.fill(-1);
const cellsX: number[] = [];
const cellsY: number[] = [];

for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) !== 35) {
      cellIndex[y * width + x] = cellsX.length;
      cellsX.push(x);
      cellsY.push(y);
    }
  }
}

const openCount = cellsX.length;
const neighbors: Int32Array[] = new Array(openCount);
for (let i = 0; i < openCount; i++) {
  const x = cellsX[i];
  const y = cellsY[i];
  const n: number[] = [];
  if (x > 0) {
    const v = cellIndex[y * width + (x - 1)];
    if (v >= 0) n.push(v);
  }
  if (x + 1 < width) {
    const v = cellIndex[y * width + (x + 1)];
    if (v >= 0) n.push(v);
  }
  if (y > 0) {
    const v = cellIndex[(y - 1) * width + x];
    if (v >= 0) n.push(v);
  }
  if (y + 1 < height) {
    const v = cellIndex[(y + 1) * width + x];
    if (v >= 0) n.push(v);
  }
  neighbors[i] = Int32Array.from(n);
}

function toOpenIndex(x: number, y: number): number {
  if (x < 0 || y < 0 || x >= width || y >= height) return -1;
  return cellIndex[y * width + x];
}

function manhattan(ax: number, ay: number, bx: number, by: number): number {
  return Math.abs(ax - bx) + Math.abs(ay - by);
}

const distanceCache = new Map<number, Int32Array>();

function buildDistances(source: number): Int32Array {
  const cached = distanceCache.get(source);
  if (cached) return cached;

  const dist = new Int32Array(openCount);
  dist.fill(-1);
  const queue = new Int32Array(openCount);
  let head = 0;
  let tail = 0;
  dist[source] = 0;
  queue[tail++] = source;

  while (head < tail) {
    const cur = queue[head++];
    const nextDist = dist[cur] + 1;
    const adj = neighbors[cur];
    for (let i = 0; i < adj.length; i++) {
      const nb = adj[i];
      if (dist[nb] !== -1) continue;
      dist[nb] = nextDist;
      queue[tail++] = nb;
    }
  }

  distanceCache.set(source, dist);
  return dist;
}

type RobotState = {
  id: number;
  cell: number;
  x: number;
  y: number;
  travel: number;
};

type JobState = Job & {
  cell: number;
  order: number;
};

const robotStates: RobotState[] = robots.map((robot) => ({
  id: robot.id,
  cell: toOpenIndex(robot.x, robot.y),
  x: robot.x,
  y: robot.y,
  travel: 0,
}));

const sortedJobs: JobState[] = jobs
  .map((job, index) => ({
    ...job,
    cell: toOpenIndex(job.x, job.y),
    order: index,
  }))
  .sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    if (a.id !== b.id) return a.id - b.id;
    return a.order - b.order;
  });

let maxRobotId = -1;
for (let i = 0; i < robots.length; i++) {
  if (robots[i].id > maxRobotId) maxRobotId = robots[i].id;
}
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

function chooseRobot(job: JobState): RobotState {
  let best: RobotState | null = null;
  let bestReachable = false;
  let bestDist = Number.POSITIVE_INFINITY;
  let bestFallback = Number.POSITIVE_INFINITY;

  for (let i = 0; i < robotStates.length; i++) {
    const robot = robotStates[i];
    let reachable = false;
    let dist = Number.POSITIVE_INFINITY;

    if (robot.cell >= 0 && job.cell >= 0) {
      const d = buildDistances(robot.cell)[job.cell];
      if (d >= 0) {
        reachable = true;
        dist = d;
      }
    }

    const fallback = manhattan(robot.x, robot.y, job.x, job.y);

    if (best === null) {
      best = robot;
      bestReachable = reachable;
      bestDist = dist;
      bestFallback = fallback;
      continue;
    }

    if (reachable !== bestReachable) {
      if (reachable) {
        best = robot;
        bestReachable = true;
        bestDist = dist;
        bestFallback = fallback;
      }
      continue;
    }

    if (reachable) {
      if (dist !== bestDist) {
        if (dist < bestDist) {
          best = robot;
          bestDist = dist;
          bestFallback = fallback;
        }
        continue;
      }
    } else if (fallback !== bestFallback) {
      if (fallback < bestFallback) {
        best = robot;
        bestFallback = fallback;
      }
      continue;
    }

    if (robot.travel !== best.travel) {
      if (robot.travel < best.travel) {
        best = robot;
        bestDist = dist;
        bestFallback = fallback;
      }
      continue;
    }

    if (robot.id < best.id) {
      best = robot;
      bestDist = dist;
      bestFallback = fallback;
    }
  }

  return best!;
}

for (let i = 0; i < sortedJobs.length; i++) {
  const job = sortedJobs[i];
  const robot = chooseRobot(job);
  assignments[robot.id].push(job.id);

  if (robot.cell >= 0 && job.cell >= 0) {
    const d = buildDistances(robot.cell)[job.cell];
    if (d >= 0) robot.travel += d;
  } else {
    robot.travel += manhattan(robot.x, robot.y, job.x, job.y);
  }

  robot.cell = job.cell;
  robot.x = job.x;
  robot.y = job.y;
}

process.stdout.write(JSON.stringify({ assignments }));
