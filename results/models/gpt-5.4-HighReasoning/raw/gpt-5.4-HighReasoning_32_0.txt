#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u;
    int v;
    long long w;
};

struct AdjEdge {
    int to;
    int id;
    long long w;
};

struct DSU {
    vector<int> p;
    vector<int> r;

    explicit DSU(int n) : p(n), r(n, 0) {
        iota(p.begin(), p.end(), 0);
    }

    int find(int x) {
        if (p[x] != x) p[x] = find(p[x]);
        return p[x];
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static const long long INF = numeric_limits<long long>::max() / 4;

struct ShortestPathData {
    vector<vector<long long>> dist;
    vector<vector<int>> prev_v;
    vector<vector<int>> prev_e;
};

ShortestPathData run_terminal_dijkstras(
    int n,
    const vector<int>& terminals,
    const vector<vector<AdjEdge>>& graph
) {
    int k = static_cast<int>(terminals.size());
    ShortestPathData data;
    data.dist.assign(k, vector<long long>(n, INF));
    data.prev_v.assign(k, vector<int>(n, -1));
    data.prev_e.assign(k, vector<int>(n, -1));

    using Node = pair<long long, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;

    for (int t = 0; t < k; ++t) {
        auto& dist = data.dist[t];
        auto& prev_v = data.prev_v[t];
        auto& prev_e = data.prev_e[t];
        dist[terminals[t]] = 0;
        pq.push({0, terminals[t]});

        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dist[v]) continue;
            for (const auto& e : graph[v]) {
                long long nd = d + e.w;
                if (nd < dist[e.to]) {
                    dist[e.to] = nd;
                    prev_v[e.to] = v;
                    prev_e[e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    return data;
}

void add_terminal_path(
    int term_index,
    int target,
    const vector<int>& terminals,
    const ShortestPathData& sp,
    vector<char>& used_edge
) {
    int v = target;
    while (v != terminals[term_index] && v != -1) {
        int eid = sp.prev_e[term_index][v];
        int pv = sp.prev_v[term_index][v];
        if (eid == -1 || pv == -1) break;
        used_edge[eid] = 1;
        v = pv;
    }
}

vector<int> build_tree_from_selected_edges(
    int n,
    const vector<int>& terminals,
    const vector<Edge>& edges,
    const vector<char>& selected_edge
) {
    vector<int> candidate_edges;
    candidate_edges.reserve(edges.size());
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        if (selected_edge[i]) candidate_edges.push_back(i);
    }

    if (terminals.size() <= 1 || candidate_edges.empty()) return {};

    sort(candidate_edges.begin(), candidate_edges.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });

    DSU dsu(n);
    vector<int> tree_edges;
    tree_edges.reserve(candidate_edges.size());
    for (int eid : candidate_edges) {
        const auto& e = edges[eid];
        if (dsu.unite(e.u, e.v)) tree_edges.push_back(eid);
    }

    vector<vector<pair<int, int>>> tree_adj(n);
    vector<int> deg(n, 0);
    vector<char> in_tree(edges.size(), 0);
    for (int eid : tree_edges) {
        in_tree[eid] = 1;
        const auto& e = edges[eid];
        tree_adj[e.u].push_back({e.v, eid});
        tree_adj[e.v].push_back({e.u, eid});
        ++deg[e.u];
        ++deg[e.v];
    }

    vector<char> is_terminal(n, 0);
    for (int t : terminals) is_terminal[t] = 1;

    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!is_terminal[v] && deg[v] == 1) q.push(v);
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (is_terminal[v] || deg[v] != 1) continue;

        for (auto [to, eid] : tree_adj[v]) {
            if (!in_tree[eid]) continue;
            in_tree[eid] = 0;
            --deg[v];
            --deg[to];
            if (!is_terminal[to] && deg[to] == 1) q.push(to);
            break;
        }
    }

    vector<int> result;
    for (int eid : tree_edges) {
        if (in_tree[eid]) result.push_back(eid);
    }
    return result;
}

vector<int> solve_heuristic(
    int n,
    const vector<int>& terminals,
    const vector<Edge>& edges,
    const vector<vector<AdjEdge>>& graph,
    const ShortestPathData& sp
) {
    int k = static_cast<int>(terminals.size());
    if (k <= 1) return {};

    struct TermEdge {
        int a;
        int b;
        long long w;
    };

    vector<TermEdge> metric_edges;
    metric_edges.reserve(static_cast<size_t>(k) * (k - 1) / 2);
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            metric_edges.push_back({i, j, sp.dist[i][terminals[j]]});
        }
    }

    sort(metric_edges.begin(), metric_edges.end(), [](const TermEdge& x, const TermEdge& y) {
        if (x.w != y.w) return x.w < y.w;
        if (x.a != y.a) return x.a < y.a;
        return x.b < y.b;
    });

    DSU dsu(k);
    vector<char> used_edge(edges.size(), 0);
    for (const auto& e : metric_edges) {
        if (!dsu.unite(e.a, e.b)) continue;
        add_terminal_path(e.a, terminals[e.b], terminals, sp, used_edge);
    }

    return build_tree_from_selected_edges(n, terminals, edges, used_edge);
}

vector<int> solve_exact(
    int n,
    const vector<int>& terminals,
    const vector<Edge>& edges,
    const vector<vector<AdjEdge>>& graph,
    const ShortestPathData& sp
) {
    int k = static_cast<int>(terminals.size());
    if (k <= 1) return {};

    int subset_count = 1 << k;
    size_t total_states = static_cast<size_t>(subset_count) * static_cast<size_t>(n);

    vector<long long> dp(total_states, INF);
    vector<int> split_mask(total_states, -1);
    vector<int> relax_prev_v(total_states, -1);
    vector<int> relax_prev_e(total_states, -1);

    auto index_of = [n](int mask, int v) -> size_t {
        return static_cast<size_t>(mask) * static_cast<size_t>(n) + static_cast<size_t>(v);
    };

    for (int i = 0; i < k; ++i) {
        int mask = 1 << i;
        for (int v = 0; v < n; ++v) {
            dp[index_of(mask, v)] = sp.dist[i][v];
        }
    }

    using Node = pair<long long, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;

    for (int mask = 1; mask < subset_count; ++mask) {
        if ((mask & (mask - 1)) != 0) {
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (sub > other) continue;
                for (int v = 0; v < n; ++v) {
                    size_t idx = index_of(mask, v);
                    long long cand = dp[index_of(sub, v)] + dp[index_of(other, v)];
                    if (cand < dp[idx]) {
                        dp[idx] = cand;
                        split_mask[idx] = sub;
                        relax_prev_v[idx] = -1;
                        relax_prev_e[idx] = -1;
                    }
                }
            }
        }

        while (!pq.empty()) pq.pop();
        for (int v = 0; v < n; ++v) {
            long long d = dp[index_of(mask, v)];
            if (d < INF) pq.push({d, v});
        }

        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            size_t idx_v = index_of(mask, v);
            if (d != dp[idx_v]) continue;
            for (const auto& e : graph[v]) {
                size_t idx_to = index_of(mask, e.to);
                long long nd = d + e.w;
                if (nd < dp[idx_to]) {
                    dp[idx_to] = nd;
                    split_mask[idx_to] = -1;
                    relax_prev_v[idx_to] = v;
                    relax_prev_e[idx_to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int full_mask = subset_count - 1;
    int best_root = -1;
    long long best_cost = INF;
    for (int v = 0; v < n; ++v) {
        long long cur = dp[index_of(full_mask, v)];
        if (cur < best_cost) {
            best_cost = cur;
            best_root = v;
        }
    }

    vector<char> used_edge(edges.size(), 0);
    vector<char> seen(total_states, 0);

    function<void(int, int)> build = [&](int mask, int v) {
        size_t idx = index_of(mask, v);
        if (seen[idx]) return;
        seen[idx] = 1;

        if ((mask & (mask - 1)) == 0) {
            int bit = 0;
            while (((mask >> bit) & 1) == 0) ++bit;
            add_terminal_path(bit, v, terminals, sp, used_edge);
            return;
        }

        if (relax_prev_v[idx] != -1) {
            used_edge[relax_prev_e[idx]] = 1;
            build(mask, relax_prev_v[idx]);
            return;
        }

        int sub = split_mask[idx];
        if (sub != -1) {
            build(sub, v);
            build(mask ^ sub, v);
        }
    };

    build(full_mask, best_root);
    return build_tree_from_selected_edges(n, terminals, edges, used_edge);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<Edge> edges;
    edges.reserve(m);
    vector<vector<AdjEdge>> graph(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        graph[u].push_back({v, i, w});
        graph[v].push_back({u, i, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k <= 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    ShortestPathData sp = run_terminal_dijkstras(n, terminals, graph);

    bool disconnected = false;
    for (int i = 0; i < k; ++i) {
        for (int j = 0; j < k; ++j) {
            if (sp.dist[i][terminals[j]] >= INF / 2) disconnected = true;
        }
    }

    if (disconnected) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    size_t exact_states = (k >= 31) ? numeric_limits<size_t>::max() : (static_cast<size_t>(1) << k) * static_cast<size_t>(n);
    bool use_exact = (k <= 15 && exact_states <= 6000000ULL);

    vector<int> answer_edges = use_exact
        ? solve_exact(n, terminals, edges, graph, sp)
        : solve_heuristic(n, terminals, edges, graph, sp);

    long long total_weight = 0;
    for (int eid : answer_edges) total_weight += edges[eid].w;

    cout << total_weight << ' ' << answer_edges.size() << '\n';
    for (int eid : answer_edges) {
        cout << edges[eid].u << ' ' << edges[eid].v << '\n';
    }

    return 0;
}
