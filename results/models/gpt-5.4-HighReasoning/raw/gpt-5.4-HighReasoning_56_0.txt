const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const rawPackages = Array.isArray(data.packages) ? data.packages : [];
const n = rawPackages.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const maxRoutes = Math.max(1, Math.min(n, (data.truckCount | 0) || 1));
const depotX = Number(depot[0]) || 0;
const depotY = Number(depot[1]) || 0;

const pts = rawPackages.map((p, idx) => {
  const x = Number(p.x) || 0;
  const y = Number(p.y) || 0;
  const dx = x - depotX;
  const dy = y - depotY;
  return {
    idx,
    id: p.id,
    x,
    y,
    dx,
    dy,
    angle: Math.atan2(dy, dx),
    r2: dx * dx + dy * dy
  };
});

const distDepot = new Array(n);
const distMatrix = new Map();

function pairKey(a, b) {
  return a < b ? a * n + b : b * n + a;
}

function distanceIdx(a, b) {
  if (a === b) return 0;
  const key = pairKey(a, b);
  const cached = distMatrix.get(key);
  if (cached !== undefined) return cached;
  const pa = pts[a];
  const pb = pts[b];
  const d = Math.hypot(pa.x - pb.x, pa.y - pb.y);
  distMatrix.set(key, d);
  return d;
}

for (let i = 0; i < n; i++) {
  distDepot[i] = Math.hypot(pts[i].dx, pts[i].dy);
}

function sortIndices(compareFn) {
  const arr = new Array(n);
  for (let i = 0; i < n; i++) arr[i] = i;
  arr.sort(compareFn);
  return arr;
}

function rotateArray(arr, start) {
  if (start <= 0 || start >= arr.length) return arr.slice();
  return arr.slice(start).concat(arr.slice(0, start));
}

function reverseCopy(arr) {
  const out = arr.slice();
  out.reverse();
  return out;
}

function angleOrder() {
  const arr = sortIndices((a, b) => {
    const pa = pts[a];
    const pb = pts[b];
    if (pa.angle !== pb.angle) return pa.angle - pb.angle;
    if (pa.r2 !== pb.r2) return pa.r2 - pb.r2;
    return a - b;
  });
  if (arr.length <= 1) return arr;

  let bestStart = 0;
  let bestGap = -1;
  for (let i = 0; i < arr.length; i++) {
    const a1 = pts[arr[i]].angle;
    const a2 = pts[arr[(i + 1) % arr.length]].angle;
    let gap = a2 - a1;
    if (i === arr.length - 1) gap += Math.PI * 2;
    if (gap > bestGap) {
      bestGap = gap;
      bestStart = (i + 1) % arr.length;
    }
  }
  return rotateArray(arr, bestStart);
}

function snakeOrder(primary) {
  const chunk = Math.max(8, Math.ceil(Math.sqrt(n)));
  const arr = sortIndices((a, b) => {
    const pa = pts[a];
    const pb = pts[b];
    const av = primary === 'x' ? pa.x : pa.y;
    const bv = primary === 'x' ? pb.x : pb.y;
    if (av !== bv) return av - bv;
    const as = primary === 'x' ? pa.y : pa.x;
    const bs = primary === 'x' ? pb.y : pb.x;
    if (as !== bs) return as - bs;
    return a - b;
  });

  const out = [];
  for (let i = 0; i < arr.length; i += chunk) {
    const part = arr.slice(i, i + chunk);
    part.sort((a, b) => {
      const pa = pts[a];
      const pb = pts[b];
      const av = primary === 'x' ? pa.y : pa.x;
      const bv = primary === 'x' ? pb.y : pb.x;
      if (av !== bv) return av - bv;
      const as = primary === 'x' ? pa.x : pa.y;
      const bs = primary === 'x' ? pb.x : pb.y;
      if (as !== bs) return as - bs;
      return a - b;
    });
    if (((i / chunk) & 1) === 1) part.reverse();
    for (let j = 0; j < part.length; j++) out.push(part[j]);
  }
  return out;
}

function part1By1(v) {
  v &= 0xffff;
  v = (v | (v << 8)) & 0x00FF00FF;
  v = (v | (v << 4)) & 0x0F0F0F0F;
  v = (v | (v << 2)) & 0x33333333;
  v = (v | (v << 1)) & 0x55555555;
  return v >>> 0;
}

function mortonCode(x, y) {
  return ((part1By1(y) << 1) | part1By1(x)) >>> 0;
}

function mortonOrder() {
  let minX = Infinity;
  let minY = Infinity;
  let maxX = -Infinity;
  let maxY = -Infinity;
  for (let i = 0; i < n; i++) {
    const p = pts[i];
    if (p.x < minX) minX = p.x;
    if (p.y < minY) minY = p.y;
    if (p.x > maxX) maxX = p.x;
    if (p.y > maxY) maxY = p.y;
  }
  const spanX = maxX - minX || 1;
  const spanY = maxY - minY || 1;
  const codes = new Array(n);
  for (let i = 0; i < n; i++) {
    const p = pts[i];
    const sx = Math.max(0, Math.min(65535, Math.floor(((p.x - minX) / spanX) * 65535)));
    const sy = Math.max(0, Math.min(65535, Math.floor(((p.y - minY) / spanY) * 65535)));
    codes[i] = mortonCode(sx, sy);
  }
  return sortIndices((a, b) => {
    const d = codes[a] - codes[b];
    return d !== 0 ? d : a - b;
  });
}

function evaluateOrder(order) {
  if (order.length === 0) return { cost: 0, routes: [] };
  if (order.length === 1) {
    return { cost: 2 * distDepot[order[0]], routes: [order.slice()] };
  }

  let base = distDepot[order[0]] + distDepot[order[order.length - 1]];
  const deltas = [];
  for (let i = 0; i < order.length - 1; i++) {
    const a = order[i];
    const b = order[i + 1];
    const edge = distanceIdx(a, b);
    base += edge;
    const delta = distDepot[a] + distDepot[b] - edge;
    if (delta < -1e-9) {
      deltas.push([delta, i]);
    }
  }

  deltas.sort((u, v) => u[0] - v[0] || u[1] - v[1]);
  const take = Math.min(maxRoutes - 1, deltas.length);
  const splitSet = new Set();
  let total = base;
  for (let i = 0; i < take; i++) {
    total += deltas[i][0];
    splitSet.add(deltas[i][1]);
  }

  const routes = [];
  let start = 0;
  for (let i = 0; i < order.length - 1; i++) {
    if (splitSet.has(i)) {
      routes.push(order.slice(start, i + 1));
      start = i + 1;
    }
  }
  routes.push(order.slice(start));
  return { cost: total, routes };
}

function routeCost(route) {
  if (route.length === 0) return 0;
  let total = distDepot[route[0]] + distDepot[route[route.length - 1]];
  for (let i = 0; i < route.length - 1; i++) {
    total += distanceIdx(route[i], route[i + 1]);
  }
  return total;
}

function reverseSegment(arr, i, j) {
  while (i < j) {
    const tmp = arr[i];
    arr[i] = arr[j];
    arr[j] = tmp;
    i++;
    j--;
  }
}

function improveRoute(route) {
  const m = route.length;
  if (m < 3) return route;

  const maxWindow = m <= 80 ? m : 40;
  let changed = true;
  let passes = 0;

  while (changed && passes < 3) {
    changed = false;
    passes++;

    for (let i = 0; i < m - 1; i++) {
      const a = i === 0 ? -1 : route[i - 1];
      const b = route[i];
      const ab = a === -1 ? distDepot[b] : distanceIdx(a, b);
      const maxJ = Math.min(m - 1, i + maxWindow);
      for (let j = i + 1; j <= maxJ; j++) {
        const c = route[j];
        const d = j === m - 1 ? -1 : route[j + 1];
        const cd = d === -1 ? distDepot[c] : distanceIdx(c, d);
        const ac = a === -1 ? distDepot[c] : distanceIdx(a, c);
        const bd = d === -1 ? distDepot[b] : distanceIdx(b, d);
        if (ac + bd + 1e-9 < ab + cd) {
          reverseSegment(route, i, j);
          changed = true;
        }
      }
    }

    for (let i = 0; i < m; i++) {
      const node = route[i];
      const prev = i === 0 ? -1 : route[i - 1];
      const next = i === m - 1 ? -1 : route[i + 1];
      const removeCost =
        (prev === -1 ? distDepot[node] : distanceIdx(prev, node)) +
        (next === -1 ? distDepot[node] : distanceIdx(node, next)) -
        (prev === -1
          ? (next === -1 ? 0 : distDepot[next])
          : next === -1
            ? distDepot[prev]
            : distanceIdx(prev, next));

      const from = Math.max(0, i - maxWindow);
      const to = Math.min(m - 1, i + maxWindow);
      let bestPos = -1;
      let bestGain = 0;

      for (let pos = from; pos <= to + 1; pos++) {
        if (pos === i || pos === i + 1) continue;

        const leftIndex = pos - 1;
        const rightIndex = pos >= m ? -1 : pos;
        const left = leftIndex < 0 ? -1 : route[leftIndex];
        const right = rightIndex === -1 ? -1 : route[rightIndex];
        const addCost =
          (left === -1 ? distDepot[node] : distanceIdx(left, node)) +
          (right === -1 ? distDepot[node] : distanceIdx(node, right)) -
          (left === -1
            ? (right === -1 ? 0 : distDepot[right])
            : right === -1
              ? distDepot[left]
              : distanceIdx(left, right));

        const gain = removeCost - addCost;
        if (gain > bestGain + 1e-9) {
          bestGain = gain;
          bestPos = pos;
        }
      }

      if (bestPos !== -1) {
        route.splice(i, 1);
        let insertPos = bestPos;
        if (bestPos > i) insertPos--;
        route.splice(insertPos, 0, node);
        changed = true;
      }
    }
  }

  return route;
}

const candidates = [];
const ang = angleOrder();
candidates.push(ang);
candidates.push(reverseCopy(ang));

const snakeX = snakeOrder('x');
candidates.push(snakeX);
candidates.push(reverseCopy(snakeX));

const snakeY = snakeOrder('y');
candidates.push(snakeY);
candidates.push(reverseCopy(snakeY));

const morton = mortonOrder();
candidates.push(morton);
candidates.push(reverseCopy(morton));

let best = null;
for (const order of candidates) {
  const evaluated = evaluateOrder(order);
  if (!best || evaluated.cost < best.cost - 1e-9) {
    best = evaluated;
  }
}

for (let i = 0; i < best.routes.length; i++) {
  improveRoute(best.routes[i]);
}

let improved = 0;
for (let i = 0; i < best.routes.length; i++) improved += routeCost(best.routes[i]);

// If local search somehow made things worse, fall back to the unsafely-improved-free best candidate.
if (!Number.isFinite(improved)) {
  process.stdout.write(JSON.stringify({ routes: best.routes.map(route => route.map(idx => pts[idx].id)) }));
  process.exit(0);
}

const result = {
  routes: best.routes.map(route => route.map(idx => pts[idx].id))
};

process.stdout.write(JSON.stringify(result));
