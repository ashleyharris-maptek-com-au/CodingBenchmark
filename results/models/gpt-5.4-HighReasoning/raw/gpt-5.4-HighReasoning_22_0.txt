using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private static readonly Dictionary<char, PieceRotation[]> Pieces = BuildPieces();
    private static readonly char[] PieceOrder = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };

    public static void Main()
    {
        var scanner = new FastScanner(Console.In);
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var ai = new TetrisAi(width, height);

        for (int turn = 0; turn < numPieces; turn++)
        {
            string token = scanner.Next();
            if (string.IsNullOrEmpty(token))
            {
                break;
            }

            char piece = char.ToUpperInvariant(token[0]);
            Move best = ai.ChooseMove(piece, numPieces - turn);

            Console.WriteLine($"{best.Rotation} {best.Column}");
            Console.Out.Flush();

            ai.ApplyChosenMove(best);

            string result = scanner.Next();
            if (string.IsNullOrEmpty(result))
            {
                break;
            }

            if (result.StartsWith("cleared_", StringComparison.OrdinalIgnoreCase))
            {
                if (int.TryParse(result.Substring(8), NumberStyles.Integer, CultureInfo.InvariantCulture, out int cleared))
                {
                    ai.AcceptJudgeResult(cleared, false);
                }
            }
            else if (result.Equals("ok", StringComparison.OrdinalIgnoreCase))
            {
                ai.AcceptJudgeResult(0, false);
            }
            else if (result.Equals("gameover", StringComparison.OrdinalIgnoreCase))
            {
                ai.AcceptJudgeResult(0, true);
                break;
            }
        }
    }

    private static Dictionary<char, PieceRotation[]> BuildPieces()
    {
        var baseShapes = new Dictionary<char, (int X, int Y)[]>
        {
            ['I'] = new[] { (0, 0), (1, 0), (2, 0), (3, 0) },
            ['O'] = new[] { (0, 0), (1, 0), (0, 1), (1, 1) },
            ['T'] = new[] { (0, 0), (1, 0), (2, 0), (1, 1) },
            ['S'] = new[] { (1, 0), (2, 0), (0, 1), (1, 1) },
            ['Z'] = new[] { (0, 0), (1, 0), (1, 1), (2, 1) },
            ['J'] = new[] { (0, 0), (0, 1), (1, 1), (2, 1) },
            ['L'] = new[] { (2, 0), (0, 1), (1, 1), (2, 1) },
        };

        var result = new Dictionary<char, PieceRotation[]>();

        foreach (var entry in baseShapes)
        {
            var list = new List<PieceRotation>();
            var seen = new HashSet<string>(StringComparer.Ordinal);
            var current = entry.Value;

            for (int rotation = 0; rotation < 4; rotation++)
            {
                var normalized = Normalize(current);
                string key = string.Join(";", normalized.Select(c => $"{c.X},{c.Y}"));
                if (seen.Add(key))
                {
                    list.Add(new PieceRotation(rotation, normalized));
                }

                current = RotateClockwise(current);
            }

            result[entry.Key] = list.ToArray();
        }

        return result;
    }

    private static (int X, int Y)[] RotateClockwise((int X, int Y)[] cells)
    {
        int maxY = cells.Max(c => c.Y);
        var rotated = new (int X, int Y)[cells.Length];
        for (int i = 0; i < cells.Length; i++)
        {
            rotated[i] = (maxY - cells[i].Y, cells[i].X);
        }

        return rotated;
    }

    private static (int X, int Y)[] Normalize((int X, int Y)[] cells)
    {
        int minX = cells.Min(c => c.X);
        int minY = cells.Min(c => c.Y);
        var normalized = new (int X, int Y)[cells.Length];
        for (int i = 0; i < cells.Length; i++)
        {
            normalized[i] = (cells[i].X - minX, cells[i].Y - minY);
        }

        Array.Sort(normalized, (a, b) =>
        {
            int cmp = a.Y.CompareTo(b.Y);
            return cmp != 0 ? cmp : a.X.CompareTo(b.X);
        });
        return normalized;
    }

    private readonly struct PieceRotation
    {
        public readonly int Rotation;
        public readonly (int X, int Y)[] Cells;
        public readonly int Width;
        public readonly int Height;

        public PieceRotation(int rotation, (int X, int Y)[] cells)
        {
            Rotation = rotation;
            Cells = cells;
            Width = cells.Max(c => c.X) + 1;
            Height = cells.Max(c => c.Y) + 1;
        }
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int Column;
        public readonly BoardState State;
        public readonly double Score;

        public Move(int rotation, int column, BoardState state, double score)
        {
            Rotation = rotation;
            Column = column;
            State = state;
            Score = score;
        }
    }

    private sealed class TetrisAi
    {
        private readonly int width;
        private readonly int height;
        private BoardState board;
        private BoardState pendingBoard;

        public TetrisAi(int width, int height)
        {
            this.width = width;
            this.height = height;
            board = new BoardState(width, height);
            pendingBoard = board;
        }

        public Move ChooseMove(char piece, int piecesRemaining)
        {
            var placements = EnumerateMoves(board, piece);
            if (placements.Count == 0)
            {
                var fallback = new BoardState(width, height);
                return new Move(0, 0, fallback, double.NegativeInfinity);
            }

            bool danger = board.MaxHeight >= height - 4 || board.Holes > Math.Max(6, width / 2);
            bool smallBoard = width <= 12 && height <= 24;
            bool useLookahead = smallBoard && !danger && piecesRemaining > 8;

            Move best = placements[0];
            double bestScore = double.NegativeInfinity;

            foreach (var move in placements)
            {
                double score = move.Score;
                if (useLookahead)
                {
                    score += EvaluateLookahead(move.State, piecesRemaining);
                }
                else if (danger)
                {
                    score += EvaluateEmergency(move.State);
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    best = new Move(move.Rotation, move.Column, move.State, score);
                }
            }

            pendingBoard = best.State;
            return best;
        }

        public void ApplyChosenMove(Move move)
        {
            pendingBoard = move.State;
        }

        public void AcceptJudgeResult(int cleared, bool gameOver)
        {
            if (!gameOver)
            {
                board = pendingBoard;
            }
        }

        private double EvaluateLookahead(BoardState state, int piecesRemaining)
        {
            int sampleCount = piecesRemaining > 30 ? 7 : 4;
            double total = 0.0;

            for (int i = 0; i < sampleCount; i++)
            {
                char nextPiece = PieceOrder[i];
                var nextMoves = EnumerateMoves(state, nextPiece);
                if (nextMoves.Count == 0)
                {
                    total -= 50000.0;
                    continue;
                }

                double localBest = double.NegativeInfinity;
                for (int j = 0; j < nextMoves.Count; j++)
                {
                    localBest = Math.Max(localBest, nextMoves[j].Score);
                }

                total += localBest;
            }

            return total / sampleCount * 0.18;
        }

        private double EvaluateEmergency(BoardState state)
        {
            return -20.0 * state.MaxHeight - 18.0 * state.Holes - 12.0 * state.Bumpiness + 30.0 * state.WellDepth;
        }

        private List<Move> EnumerateMoves(BoardState source, char piece)
        {
            var rotations = Pieces[piece];
            var moves = new List<Move>(32);

            foreach (var rotation in rotations)
            {
                int maxColumn = width - rotation.Width;
                for (int column = 0; column <= maxColumn; column++)
                {
                    if (!TryPlace(source, rotation, column, out var next))
                    {
                        continue;
                    }

                    double score = EvaluateBoard(next, source);
                    moves.Add(new Move(rotation.Rotation, column, next, score));
                }
            }

            moves.Sort((a, b) =>
            {
                int cmp = b.Score.CompareTo(a.Score);
                if (cmp != 0) return cmp;
                cmp = a.Rotation.CompareTo(b.Rotation);
                return cmp != 0 ? cmp : a.Column.CompareTo(b.Column);
            });

            if (moves.Count > 18)
            {
                moves.RemoveRange(18, moves.Count - 18);
            }

            return moves;
        }

        private bool TryPlace(BoardState source, PieceRotation rotation, int column, out BoardState next)
        {
            int y = 0;
            if (Collides(source, rotation, column, y))
            {
                next = default;
                return false;
            }

            while (!Collides(source, rotation, column, y + 1))
            {
                y++;
            }

            next = source.Clone();
            foreach (var cell in rotation.Cells)
            {
                int boardX = column + cell.X;
                int boardY = y + cell.Y;
                if (boardY < 0 || boardY >= height)
                {
                    return false;
                }

                next.Rows[boardY] |= 1 << boardX;
            }

            next.ClearFullLines();
            next.RecomputeMetrics();
            return true;
        }

        private bool Collides(BoardState source, PieceRotation rotation, int column, int y)
        {
            foreach (var cell in rotation.Cells)
            {
                int boardX = column + cell.X;
                int boardY = y + cell.Y;
                if (boardX < 0 || boardX >= width || boardY >= height)
                {
                    return true;
                }

                if (boardY >= 0 && ((source.Rows[boardY] >> boardX) & 1) != 0)
                {
                    return true;
                }
            }

            return false;
        }

        private double EvaluateBoard(BoardState state, BoardState previous)
        {
            double dangerRatio = (double)state.MaxHeight / height;
            double score =
                state.LinesCleared * 1200.0
                - state.AggregateHeight * (dangerRatio > 0.7 ? 5.8 : 3.9)
                - state.Holes * (dangerRatio > 0.7 ? 18.0 : 11.5)
                - state.Bumpiness * 5.0
                - state.Blockades * 7.0
                - state.RowTransitions * 2.0
                - state.ColumnTransitions * 2.5
                - state.MaxHeight * 7.5
                + state.WellDepth * 1.8;

            if (state.LinesCleared == 4)
            {
                score += 350.0;
            }

            if (state.Holes < previous.Holes)
            {
                score += 70.0 * (previous.Holes - state.Holes);
            }

            if (state.MaxHeight >= height - 2)
            {
                score -= 100000.0;
            }

            return score;
        }
    }

    private struct BoardState
    {
        public readonly int Width;
        public readonly int Height;
        public int[] Rows;
        public int AggregateHeight;
        public int MaxHeight;
        public int Holes;
        public int Bumpiness;
        public int Blockades;
        public int RowTransitions;
        public int ColumnTransitions;
        public int WellDepth;
        public int LinesCleared;

        public BoardState(int width, int height)
        {
            Width = width;
            Height = height;
            Rows = new int[height];
            AggregateHeight = 0;
            MaxHeight = 0;
            Holes = 0;
            Bumpiness = 0;
            Blockades = 0;
            RowTransitions = 0;
            ColumnTransitions = 0;
            WellDepth = 0;
            LinesCleared = 0;
        }

        public BoardState Clone()
        {
            var clone = this;
            clone.Rows = (int[])Rows.Clone();
            clone.LinesCleared = 0;
            return clone;
        }

        public void ClearFullLines()
        {
            int fullMask = (1 << Width) - 1;
            int write = Height - 1;
            int cleared = 0;

            for (int read = Height - 1; read >= 0; read--)
            {
                if ((Rows[read] & fullMask) == fullMask)
                {
                    cleared++;
                    continue;
                }

                Rows[write--] = Rows[read];
            }

            while (write >= 0)
            {
                Rows[write--] = 0;
            }

            LinesCleared = cleared;
        }

        public void RecomputeMetrics()
        {
            AggregateHeight = 0;
            MaxHeight = 0;
            Holes = 0;
            Bumpiness = 0;
            Blockades = 0;
            RowTransitions = 0;
            ColumnTransitions = 0;
            WellDepth = 0;

            var heights = new int[Width];

            for (int x = 0; x < Width; x++)
            {
                bool seenBlock = false;
                int columnHeight = 0;
                int holesBelow = 0;
                int filledAboveHoles = 0;

                for (int y = 0; y < Height; y++)
                {
                    bool filled = ((Rows[y] >> x) & 1) != 0;
                    if (!seenBlock && filled)
                    {
                        seenBlock = true;
                        columnHeight = Height - y;
                    }

                    if (seenBlock)
                    {
                        if (!filled)
                        {
                            holesBelow++;
                        }
                        else if (holesBelow > 0)
                        {
                            filledAboveHoles++;
                        }
                    }
                }

                heights[x] = columnHeight;
                AggregateHeight += columnHeight;
                MaxHeight = Math.Max(MaxHeight, columnHeight);
                Holes += holesBelow;
                Blockades += filledAboveHoles;
            }

            for (int x = 0; x < Width - 1; x++)
            {
                Bumpiness += Math.Abs(heights[x] - heights[x + 1]);
            }

            for (int y = 0; y < Height; y++)
            {
                int transitions = 0;
                bool previousFilled = true;
                for (int x = 0; x < Width; x++)
                {
                    bool filled = ((Rows[y] >> x) & 1) != 0;
                    if (filled != previousFilled)
                    {
                        transitions++;
                    }
                    previousFilled = filled;
                }

                if (!previousFilled)
                {
                    transitions++;
                }
                RowTransitions += transitions;
            }

            for (int x = 0; x < Width; x++)
            {
                int transitions = 0;
                bool previousFilled = true;
                for (int y = 0; y < Height; y++)
                {
                    bool filled = ((Rows[y] >> x) & 1) != 0;
                    if (filled != previousFilled)
                    {
                        transitions++;
                    }
                    previousFilled = filled;
                }

                if (!previousFilled)
                {
                    transitions++;
                }
                ColumnTransitions += transitions;
            }

            for (int x = 0; x < Width; x++)
            {
                int left = x == 0 ? Height : heights[x - 1];
                int right = x == Width - 1 ? Height : heights[x + 1];
                int current = heights[x];
                int well = Math.Max(0, Math.Min(left, right) - current);
                WellDepth += well * (well + 1) / 2;
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly TextReader reader;
        private readonly char[] buffer = new char[1 << 12];
        private int index;
        private int size;

        public FastScanner(TextReader reader)
        {
            this.reader = reader;
        }

        public string Next()
        {
            SkipWhiteSpace();
            if (Peek() < 0)
            {
                return string.Empty;
            }

            var chars = new List<char>(8);
            while (true)
            {
                int ch = Peek();
                if (ch < 0 || char.IsWhiteSpace((char)ch))
                {
                    break;
                }

                chars.Add((char)Read());
            }

            return new string(chars.ToArray());
        }

        public int NextInt()
        {
            string token = Next();
            return int.Parse(token, CultureInfo.InvariantCulture);
        }

        private void SkipWhiteSpace()
        {
            while (true)
            {
                int ch = Peek();
                if (ch < 0 || !char.IsWhiteSpace((char)ch))
                {
                    return;
                }

                Read();
            }
        }

        private int Peek()
        {
            if (index >= size)
            {
                size = reader.Read(buffer, 0, buffer.Length);
                index = 0;
                if (size == 0)
                {
                    return -1;
                }
            }

            return buffer[index];
        }

        private int Read()
        {
            int ch = Peek();
            if (ch >= 0)
            {
                index++;
            }
            return ch;
        }
    }
}
