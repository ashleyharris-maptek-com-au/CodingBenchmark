import math
import statistics


_AP = {
    "initialized": False,
    "last_loc": 0.0,
    "last_gs": 0.0,
    "last_ias": 140.0,
    "last_agl": 1500.0,
    "last_pitch": 0.0,
    "last_roll": 0.0,
    "loc_int": 0.0,
    "gs_int": 0.0,
    "spd_int": 0.0,
    "trim_int": 0.0,
    "go_around": False,
    "baro_ground_ref": None,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _num(x, default=None):
    try:
        v = float(x)
    except (TypeError, ValueError):
        return default
    if not math.isfinite(v):
        return default
    return v


def _median(values, default=0.0):
    vals = [v for v in values if v is not None and math.isfinite(v)]
    return statistics.median(vals) if vals else default


def _vote(state, stem, suffix, last_key, spread_limit=2.0, smooth=0.35):
    vals = []
    for i in (1, 2, 3):
        valid = state.get(f"{stem}_{i}_valid", True)
        val = _num(state.get(f"{stem}_{i}_{suffix}"))
        if valid and val is not None:
            vals.append(val)

    last = _AP[last_key]
    if vals:
        med = statistics.median(vals)
        filtered = [v for v in vals if abs(v - med) <= spread_limit]
        if not filtered:
            filtered = vals
        blended = sum(filtered) / len(filtered)
        out = last * (1.0 - smooth) + blended * smooth
    else:
        out = last
    _AP[last_key] = out
    return out, bool(vals)


def _engine_throttle(base, eng_num):
    running = bool(_AP_state.get(f"engine_{eng_num}_running", True))
    fire = bool(_AP_state.get(f"engine_{eng_num}_fire", False))
    if fire:
        return 0.0
    return _clamp(base, 0.0, 1.0) if running else 0.0


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP_state
    _AP_state = state
    dt = _clamp(_num(dt, 0.05), 0.02, 0.2)

    if not _AP["initialized"]:
        _AP["initialized"] = True
        baros = []
        for i in (1, 2, 3):
            if state.get(f"baro_alt_{i}_valid", True):
                v = _num(state.get(f"baro_alt_{i}_ft"))
                if v is not None:
                    baros.append(v)
        if baros:
            _AP["baro_ground_ref"] = min(baros) - 1500.0

    prev_ias = _AP["last_ias"]
    loc, loc_valid = _vote(state, "localizer", "dots", "last_loc", spread_limit=1.2)
    gs, gs_valid = _vote(state, "glideslope", "dots", "last_gs", spread_limit=1.2)
    ias, ias_valid = _vote(state, "ias", "kt", "last_ias", spread_limit=12.0, smooth=0.45)
    pitch, _ = _vote(state, "irs", "pitch_deg", "last_pitch", spread_limit=6.0, smooth=0.5)
    roll, _ = _vote(state, "irs", "roll_deg", "last_roll", spread_limit=8.0, smooth=0.5)

    vs_vals = []
    hdg_vals = []
    track_vals = []
    gs_vals = []
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", True):
            v = _num(state.get(f"irs_{i}_vs_fpm"))
            if v is not None:
                vs_vals.append(v)
            v = _num(state.get(f"irs_{i}_heading_deg"))
            if v is not None:
                hdg_vals.append(v)
            v = _num(state.get(f"irs_{i}_track_deg"))
            if v is not None:
                track_vals.append(v)
            v = _num(state.get(f"irs_{i}_gs_kt"))
            if v is not None:
                gs_vals.append(v)
    vs_fpm = _median(vs_vals, default=_num(state.get("target_vs_fpm"), -700.0) or -700.0)
    heading = _median(hdg_vals, default=_num(state.get("target_hdg_deg"), 0.0) or 0.0)
    track = _median(track_vals, default=heading)
    inertial_gs = _median(gs_vals, default=_num(state.get("groundspeed_kt"), ias) or ias)

    dme_vals = [_num(state.get("dme_1_nm")), _num(state.get("dme_2_nm"))]
    dme_vals = [v for v in dme_vals if v is not None and v >= 0.0]
    dme = min(dme_vals) if dme_vals else None

    radio_alt = _num(state.get("radio_alt_ft"))
    radio_valid = radio_alt is not None and -10.0 <= radio_alt <= 5000.0

    baro_vals = []
    for i in (1, 2, 3):
        if state.get(f"baro_alt_{i}_valid", True):
            v = _num(state.get(f"baro_alt_{i}_ft"))
            if v is not None:
                baro_vals.append(v)
    baro_alt = _median(baro_vals, default=None)

    if bool(state.get("on_ground")) or bool(state.get("weight_on_wheels")):
        agl = 0.0
        if baro_alt is not None:
            _AP["baro_ground_ref"] = baro_alt
    elif radio_valid:
        agl = max(0.0, radio_alt)
    else:
        dme_agl = dme * 318.0 if dme is not None else None
        baro_agl = None
        if baro_alt is not None and _AP["baro_ground_ref"] is not None:
            baro_agl = max(0.0, baro_alt - _AP["baro_ground_ref"])
        agl = min([v for v in (dme_agl, baro_agl) if v is not None], default=_AP["last_agl"])
    _AP["last_agl"] = agl

    target_vref = _num(state.get("target_vref_kt"), 137.0) or 137.0
    target_ias = _num(state.get("target_ias_kt"), target_vref + 5.0) or (target_vref + 5.0)
    target_loc = _num(state.get("target_loc_dots"), 0.0) or 0.0
    target_gs = _num(state.get("target_gs_dots"), 0.0) or 0.0
    target_hdg = _num(state.get("target_hdg_deg"), heading) or heading

    running_engines = int(bool(state.get("engine_1_running", True))) + int(bool(state.get("engine_2_running", True)))
    dual_engine_out = running_engines == 0

    speed_add = 0.0
    if agl > 1000.0:
        speed_add = 10.0
    elif agl > 500.0:
        speed_add = 5.0
    elif agl < 50.0:
        speed_add = -2.0
    cmd_speed = max(target_vref + 2.0, target_ias + speed_add)
    if dual_engine_out:
        cmd_speed = max(cmd_speed, target_vref + 8.0)

    loc_err = loc - target_loc
    gs_err = gs - target_gs
    speed_err = cmd_speed - ias
    crab_err = ((track - heading + 540.0) % 360.0) - 180.0
    hdg_err = ((target_hdg - heading + 540.0) % 360.0) - 180.0

    if agl > 1000.0:
        loc_limit = 1.6
        gs_limit = 1.4
        spd_limit = 30.0
    elif agl > 500.0:
        loc_limit = 1.2
        gs_limit = 1.0
        spd_limit = 24.0
    elif agl > 200.0:
        loc_limit = 0.9
        gs_limit = 0.9
        spd_limit = 18.0
    else:
        loc_limit = 0.7
        gs_limit = 0.7
        spd_limit = 14.0

    ias_rate = (ias - prev_ias) / dt
    severe_sink = agl < 400.0 and vs_fpm < -1400.0
    windshear_like = agl < 400.0 and speed_err > 12.0 and ias_rate < -8.0
    unstable = (
        abs(loc_err) > loc_limit
        or abs(gs_err) > gs_limit
        or abs(speed_err) > spd_limit
        or abs(roll) > 25.0
        or severe_sink
    )
    decision_height = 200.0
    guidance_bad = (not loc_valid) or (not gs_valid and agl < 1000.0)

    if not _AP["go_around"] and not dual_engine_out:
        if agl > 35.0 and agl < 700.0 and (unstable or windshear_like):
            _AP["go_around"] = True
        if agl < decision_height and (guidance_bad or abs(loc_err) > 0.8 or abs(gs_err) > 0.8):
            _AP["go_around"] = True
        if bool(state.get("below_decision_height")) and (guidance_bad or unstable):
            _AP["go_around"] = True

    if _AP["go_around"]:
        if agl > 1500.0:
            _AP["go_around"] = False
        else:
            tgt_pitch = 10.0 if running_engines >= 2 else 7.0
            pitch_err = tgt_pitch - pitch
            _AP["trim_int"] = _clamp(_AP["trim_int"] + pitch_err * dt * 0.015, -0.7, 0.7)
            return {
                "elevator": _clamp(0.08 * pitch_err - 0.00025 * vs_fpm, -1.0, 1.0),
                "aileron": _clamp(-0.25 * roll - 0.06 * hdg_err - 0.18 * loc_err, -1.0, 1.0),
                "rudder": _clamp(-0.05 * crab_err - 0.04 * hdg_err, -1.0, 1.0),
                "throttle_1": _engine_throttle(1.0, 1),
                "throttle_2": _engine_throttle(1.0, 2),
                "elevator_trim": _AP["trim_int"],
                "flaps": 15 if agl > 400.0 else 20,
                "gear": 1.0,
                "speedbrake": 0.0,
                "spoiler": 0.0,
                "autobrake": 0.0,
                "reverse_1": 0.0,
                "reverse_2": 0.0,
            }

    on_ground = bool(state.get("on_ground")) or bool(state.get("weight_on_wheels"))
    groundspeed = _num(state.get("groundspeed_kt"), inertial_gs) or inertial_gs
    if on_ground:
        reverse = 1.0 if groundspeed > 80.0 else 0.45 if groundspeed > 40.0 else 0.0
        return {
            "elevator": _clamp(-0.05 * pitch, -0.3, 0.3),
            "aileron": _clamp(-0.18 * roll, -0.5, 0.5),
            "rudder": _clamp(-0.06 * crab_err, -1.0, 1.0),
            "throttle_1": 0.0,
            "throttle_2": 0.0,
            "elevator_trim": _AP["trim_int"],
            "flaps": 30,
            "gear": 1.0,
            "speedbrake": 1.0,
            "spoiler": 1.0,
            "autobrake": 1.0 if groundspeed > 50.0 else 0.6,
            "reverse_1": reverse if bool(state.get("engine_1_running", True)) else 0.0,
            "reverse_2": reverse if bool(state.get("engine_2_running", True)) else 0.0,
        }

    if agl > 2000.0:
        flap_cmd = 5
        gear_cmd = 0.0
    elif agl > 1200.0:
        flap_cmd = 10
        gear_cmd = 1.0
    elif agl > 700.0:
        flap_cmd = 15
        gear_cmd = 1.0
    elif agl > 300.0:
        flap_cmd = 25
        gear_cmd = 1.0
    else:
        flap_cmd = 30
        gear_cmd = 1.0

    if dual_engine_out and agl > 100.0:
        flap_cmd = min(flap_cmd, 15)

    flare = agl < 50.0
    deep_flare = agl < 20.0

    if flare:
        desired_pitch = 2.5 if agl > 20.0 else 4.0
        if dual_engine_out:
            desired_pitch = 3.0
        pitch_err = desired_pitch - pitch
        elevator = 0.10 * pitch_err - 0.00018 * vs_fpm
        throttle_base = 0.12 if agl > 20.0 and not dual_engine_out else 0.0
    else:
        desired_vs = -720.0
        if abs(gs_err) > 0.15:
            desired_vs += -320.0 * gs_err
        if agl > 1000.0:
            desired_vs = _clamp(desired_vs, -1100.0, -500.0)
        elif agl > 300.0:
            desired_vs = _clamp(desired_vs, -950.0, -450.0)
        else:
            desired_vs = _clamp(desired_vs, -850.0, -350.0)
        vs_err = desired_vs - vs_fpm
        _AP["gs_int"] = _clamp(_AP["gs_int"] + gs_err * dt, -1.5, 1.5)
        desired_pitch = _clamp(2.0 - 1.8 * gs_err + 0.003 * vs_err, -3.0, 8.0)
        pitch_err = desired_pitch - pitch
        elevator = 0.075 * pitch_err + 0.11 * (-gs_err) + 0.00012 * vs_err + 0.04 * _AP["gs_int"]

        _AP["spd_int"] = _clamp(_AP["spd_int"] + speed_err * dt * 0.08, -0.35, 0.45)
        thrust_bias = 0.56 if agl > 1000.0 else 0.51 if agl > 300.0 else 0.47
        if dual_engine_out:
            thrust_bias = 0.0
        throttle_base = thrust_bias + 0.020 * speed_err + 0.00020 * (desired_vs - vs_fpm) + _AP["spd_int"]

    _AP["loc_int"] = _clamp(_AP["loc_int"] + loc_err * dt, -2.0, 2.0)
    desired_bank = _clamp(-10.0 * loc_err - 1.4 * crab_err - 0.25 * hdg_err - 1.2 * _AP["loc_int"], -18.0, 18.0)
    if agl < 200.0:
        desired_bank = _clamp(desired_bank, -8.0, 8.0)
    bank_err = desired_bank - roll
    aileron = 0.060 * bank_err - 0.04 * crab_err

    thrust_asym = (_num(state.get("engine_2_thrust_pct"), 0.0) or 0.0) - (_num(state.get("engine_1_thrust_pct"), 0.0) or 0.0)
    rudder = -0.09 * crab_err - 0.06 * loc_err - 0.003 * thrust_asym
    if deep_flare:
        rudder *= 0.7

    _AP["trim_int"] = _clamp(_AP["trim_int"] + pitch_err * dt * 0.01, -0.8, 0.8)

    if dual_engine_out:
        throttle_1 = 0.0
        throttle_2 = 0.0
    else:
        throttle_1 = _engine_throttle(throttle_base, 1)
        throttle_2 = _engine_throttle(throttle_base, 2)

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": throttle_1,
        "throttle_2": throttle_2,
        "elevator_trim": _AP["trim_int"],
        "flaps": flap_cmd,
        "gear": gear_cmd,
        "speedbrake": 0.0,
        "spoiler": 0.0,
        "autobrake": 0.0,
        "reverse_1": 0.0,
        "reverse_2": 0.0,
    }
