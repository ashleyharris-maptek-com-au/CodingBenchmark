#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <numeric>
#include <queue>
#include <vector>

using namespace std;

struct MaxWeightCliqueSolver {
    int n;
    vector<long long> weight;
    vector<vector<uint64_t>> adj;
    vector<int> degree;

    long long best_weight = 0;
    vector<int> best_clique;
    vector<int> current_clique;

    static int popcount64(uint64_t x) {
        int cnt = 0;
        while (x) {
            x &= x - 1;
            ++cnt;
        }
        return cnt;
    }

    MaxWeightCliqueSolver(const vector<long long>& w, const vector<vector<uint64_t>>& a)
        : n(static_cast<int>(w.size())), weight(w), adj(a), degree(n, 0) {
        for (int v = 0; v < n; ++v) {
            int d = 0;
            for (uint64_t word : adj[v]) {
                d += popcount64(word);
            }
            degree[v] = d;
        }
    }

    bool connected(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void color_sort(const vector<int>& cand, vector<int>& order, vector<long long>& bound) const {
        vector<int> rem = cand;
        sort(rem.begin(), rem.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            if (weight[a] != weight[b]) return weight[a] > weight[b];
            return a < b;
        });

        order.clear();
        bound.clear();
        long long cumulative = 0;

        while (!rem.empty()) {
            vector<int> color_class;
            vector<int> next_rem;
            long long class_max = 0;

            for (int v : rem) {
                bool ok = true;
                for (int u : color_class) {
                    if (connected(u, v)) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    color_class.push_back(v);
                    class_max = max(class_max, weight[v]);
                } else {
                    next_rem.push_back(v);
                }
            }

            cumulative += class_max;
            for (int v : color_class) {
                order.push_back(v);
                bound.push_back(cumulative);
            }
            rem.swap(next_rem);
        }
    }

    void expand(const vector<int>& cand, long long current_weight) {
        if (cand.empty()) {
            if (current_weight > best_weight) {
                best_weight = current_weight;
                best_clique = current_clique;
            }
            return;
        }

        vector<int> order;
        vector<long long> bound;
        color_sort(cand, order, bound);

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (current_weight + bound[i] <= best_weight) {
                return;
            }

            int v = order[i];
            current_clique.push_back(v);

            vector<int> next_cand;
            next_cand.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (connected(v, u)) {
                    next_cand.push_back(u);
                }
            }

            long long next_weight = current_weight + weight[v];
            if (next_cand.empty()) {
                if (next_weight > best_weight) {
                    best_weight = next_weight;
                    best_clique = current_clique;
                }
            } else {
                expand(next_cand, next_weight);
            }

            current_clique.pop_back();
        }
    }

    pair<long long, vector<int>> solve() {
        vector<int> cand(n);
        iota(cand.begin(), cand.end(), 0);
        expand(cand, 0);
        return {best_weight, best_clique};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<vector<int>> graph(n);
    vector<char> banned(n, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) {
            banned[u] = 1;
            continue;
        }
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    vector<int> answer(n, 0);
    long long objective = 0;
    vector<char> used(n, 0);

    for (int start = 0; start < n; ++start) {
        if (used[start] || banned[start] || c[start] <= 0) continue;

        vector<int> component;
        queue<int> q;
        q.push(start);
        used[start] = 1;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            if (banned[v] || c[v] <= 0) continue;
            component.push_back(v);
            for (int to : graph[v]) {
                if (!used[to] && !banned[to] && c[to] > 0) {
                    used[to] = 1;
                    q.push(to);
                }
            }
        }

        if (component.empty()) continue;
        if (component.size() == 1) {
            int v = component[0];
            answer[v] = 1;
            objective += c[v];
            continue;
        }

        int k = static_cast<int>(component.size());
        vector<int> id(n, -1);
        for (int i = 0; i < k; ++i) id[component[i]] = i;

        vector<vector<char>> original_adj(k, vector<char>(k, 0));
        for (int i = 0; i < k; ++i) {
            int v = component[i];
            for (int to : graph[v]) {
                int j = id[to];
                if (j != -1) {
                    original_adj[i][j] = 1;
                }
            }
        }

        int words = (k + 63) >> 6;
        vector<vector<uint64_t>> comp_adj(k, vector<uint64_t>(words, 0));
        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) {
                if (i == j) continue;
                if (!original_adj[i][j]) {
                    comp_adj[i][j >> 6] |= 1ULL << (j & 63);
                }
            }
        }

        vector<long long> w(k);
        for (int i = 0; i < k; ++i) w[i] = c[component[i]];

        MaxWeightCliqueSolver solver(w, comp_adj);
        auto [best_weight, clique] = solver.solve();
        objective += best_weight;
        for (int local : clique) {
            answer[component[local]] = 1;
        }
    }

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
