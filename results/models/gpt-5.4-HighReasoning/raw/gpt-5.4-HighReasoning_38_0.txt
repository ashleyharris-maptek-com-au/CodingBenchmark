import sys


def main() -> None:
    lines = [line.rstrip("\n") for line in sys.stdin]
    if len(lines) < 2:
        print("NO SOLUTION")
        return

    first = lines[0].split()
    if len(first) != 2:
        print("NO SOLUTION")
        return

    try:
        u, s = map(int, first)
    except ValueError:
        print("NO SOLUTION")
        return

    try:
        universe_line = lines[1].split()
        universe = set(map(int, universe_line))
    except ValueError:
        print("NO SOLUTION")
        return

    expected_universe = set(range(u))
    if universe != expected_universe:
        print("NO SOLUTION")
        return

    if len(lines) < 2 + s:
        print("NO SOLUTION")
        return

    sets = []
    element_to_sets = {x: [] for x in range(u)}

    for idx in range(s):
        raw = lines[2 + idx].split()
        try:
            elems = set(map(int, raw))
        except ValueError:
            print("NO SOLUTION")
            return

        if any(x < 0 or x >= u for x in elems):
            print("NO SOLUTION")
            return

        sets.append(elems)
        for x in elems:
            element_to_sets[x].append(idx)

    for x in range(u):
        if not element_to_sets[x]:
            print("NO SOLUTION")
            return

    covered = [False] * u
    chosen = []

    def search() -> bool:
        if all(covered):
            return True

        target = None
        candidates = None
        best_count = None

        for x in range(u):
            if covered[x]:
                continue

            valid = []
            for set_idx in element_to_sets[x]:
                subset = sets[set_idx]
                if all(not covered[y] for y in subset):
                    valid.append(set_idx)

            if not valid:
                return False

            count = len(valid)
            if best_count is None or count < best_count or (count == best_count and x < target):
                best_count = count
                target = x
                candidates = valid

        for set_idx in sorted(candidates):
            subset = sets[set_idx]
            for y in subset:
                covered[y] = True
            chosen.append(set_idx)

            if search():
                return True

            chosen.pop()
            for y in subset:
                covered[y] = False

        return False

    if search():
        chosen.sort()
        print("SOLUTION")
        print(" ".join(map(str, chosen)))
    else:
        print("NO SOLUTION")


if __name__ == "__main__":
    main()
