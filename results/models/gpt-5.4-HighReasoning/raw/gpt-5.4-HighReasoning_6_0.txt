using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double Infinity = 1e100;
    private const double Epsilon = 1e-9;

    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        double mu = scanner.NextDouble();

        var start = OrbitState.Read(scanner);
        var stations = new OrbitState[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = OrbitState.Read(scanner);
        }

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        var features = new OrbitFeatures[n + 1];
        features[0] = OrbitFeatures.FromState(start, mu);
        for (int i = 0; i < n; i++)
        {
            features[i + 1] = OrbitFeatures.FromState(stations[i], mu);
        }

        var cost = BuildCostMatrix(features, mu);
        int[] route = n <= 13
            ? SolveExactly(cost, n)
            : SolveHeuristic(cost, n);

        Console.WriteLine(string.Join(" ", route));
    }

    private static double[,] BuildCostMatrix(OrbitFeatures[] features, double mu)
    {
        int count = features.Length;
        var cost = new double[count, count];
        for (int i = 0; i < count; i++)
        {
            for (int j = 0; j < count; j++)
            {
                if (i == j)
                {
                    cost[i, j] = 0.0;
                    continue;
                }

                cost[i, j] = EstimateTransferCost(features[i], features[j], mu);
            }
        }

        return cost;
    }

    private static double EstimateTransferCost(OrbitFeatures a, OrbitFeatures b, double mu)
    {
        double r1 = Math.Max(6671.0, a.CharacterRadius);
        double r2 = Math.Max(6671.0, b.CharacterRadius);
        double sum = r1 + r2;

        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);

        double hohmann1 = Math.Abs(v1 * (Math.Sqrt(Math.Max(0.0, 2.0 * r2 / sum)) - 1.0));
        double hohmann2 = Math.Abs(v2 * (1.0 - Math.Sqrt(Math.Max(0.0, 2.0 * r1 / sum))));
        double hohmann = hohmann1 + hohmann2;

        double planeAngle = SafeAcos(Vec3.Dot(a.PlaneNormal, b.PlaneNormal));
        double planeChange = 2.0 * Math.Min(v1, v2) * Math.Sin(planeAngle * 0.5);

        double eccGap = (a.EccentricityVector - b.EccentricityVector).Length();
        double eccPenalty = 0.5 * Math.Sqrt(mu / ((r1 + r2) * 0.5)) * Math.Min(2.0, eccGap);

        double radiusGapPenalty = 0.02 * Math.Abs(a.Position.Length() - b.Position.Length());
        double speedGapPenalty = 0.15 * Math.Abs(a.Speed - b.Speed);

        double phaseAngle = SafeAcos(Vec3.Dot(a.Position.Unit(), b.Position.Unit()));
        double phasePenalty = 0.02 * ((v1 + v2) * 0.5) * phaseAngle;

        return hohmann
            + 0.7 * planeChange
            + 0.35 * eccPenalty
            + radiusGapPenalty
            + speedGapPenalty
            + phasePenalty;
    }

    private static int[] SolveExactly(double[,] cost, int n)
    {
        int states = 1 << n;
        var dp = new double[states, n];
        var parent = new int[states, n];

        for (int mask = 0; mask < states; mask++)
        {
            for (int i = 0; i < n; i++)
            {
                dp[mask, i] = Infinity;
                parent[mask, i] = -1;
            }
        }

        for (int i = 0; i < n; i++)
        {
            dp[1 << i, i] = cost[0, i + 1];
        }

        for (int mask = 1; mask < states; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0)
                {
                    continue;
                }

                double current = dp[mask, last];
                if (current >= Infinity * 0.5)
                {
                    continue;
                }

                int remaining = ((states - 1) ^ mask);
                while (remaining != 0)
                {
                    int bit = remaining & -remaining;
                    int next = BitIndex(bit);
                    int nextMask = mask | bit;
                    double candidate = current + cost[last + 1, next + 1];
                    if (candidate + Epsilon < dp[nextMask, next])
                    {
                        dp[nextMask, next] = candidate;
                        parent[nextMask, next] = last;
                    }

                    remaining ^= bit;
                }
            }
        }

        int fullMask = states - 1;
        double bestCost = Infinity;
        int bestLast = 0;
        for (int last = 0; last < n; last++)
        {
            if (dp[fullMask, last] < bestCost)
            {
                bestCost = dp[fullMask, last];
                bestLast = last;
            }
        }

        var route = new int[n];
        int maskCursor = fullMask;
        int cursor = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = cursor;
            int prev = parent[maskCursor, cursor];
            maskCursor ^= 1 << cursor;
            cursor = prev;
        }

        return route;
    }

    private static int[] SolveHeuristic(double[,] cost, int n)
    {
        var rng = new Random(123456789);
        var candidates = new List<int[]>();

        candidates.Add(BuildNearestNeighbor(cost, n, randomized: false, topK: 1, rng));
        candidates.Add(BuildCheapestInsertion(cost, n));

        int randomizedStarts = n <= 25 ? 4 : n <= 60 ? 8 : 12;
        int topK = n <= 25 ? 3 : n <= 60 ? 5 : 7;
        for (int i = 0; i < randomizedStarts; i++)
        {
            candidates.Add(BuildNearestNeighbor(cost, n, randomized: true, topK, rng));
        }

        int[] best = candidates[0];
        double bestCost = RouteCost(best, cost);
        for (int i = 1; i < candidates.Count; i++)
        {
            double candidateCost = RouteCost(candidates[i], cost);
            if (candidateCost < bestCost)
            {
                best = candidates[i];
                bestCost = candidateCost;
            }
        }

        best = (int[])best.Clone();
        ImproveRoute(best, cost, n <= 40 ? 10 : 6);
        return best;
    }

    private static int[] BuildNearestNeighbor(double[,] cost, int n, bool randomized, int topK, Random rng)
    {
        var route = new int[n];
        var used = new bool[n];

        int currentNode = 0;
        for (int step = 0; step < n; step++)
        {
            int choice;
            if (!randomized)
            {
                choice = FindBestUnused(currentNode, used, cost);
            }
            else
            {
                choice = FindRandomizedUnused(currentNode, used, cost, topK, rng);
            }

            route[step] = choice;
            used[choice] = true;
            currentNode = choice + 1;
        }

        return route;
    }

    private static int[] BuildCheapestInsertion(double[,] cost, int n)
    {
        var used = new bool[n];
        var route = new List<int>(n);

        int first = FindBestUnused(0, used, cost);
        route.Add(first);
        used[first] = true;

        while (route.Count < n)
        {
            double bestDelta = Infinity;
            int bestNode = -1;
            int bestPos = 0;

            for (int node = 0; node < n; node++)
            {
                if (used[node])
                {
                    continue;
                }

                for (int pos = 0; pos <= route.Count; pos++)
                {
                    double delta = InsertionDelta(route, pos, node, cost);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestNode = node;
                        bestPos = pos;
                    }
                }
            }

            route.Insert(bestPos, bestNode);
            used[bestNode] = true;
        }

        return route.ToArray();
    }

    private static int FindBestUnused(int fromNode, bool[] used, double[,] cost)
    {
        int best = -1;
        double bestCost = Infinity;
        for (int i = 0; i < used.Length; i++)
        {
            if (used[i])
            {
                continue;
            }

            double candidate = cost[fromNode, i + 1];
            if (candidate < bestCost)
            {
                bestCost = candidate;
                best = i;
            }
        }

        return best;
    }

    private static int FindRandomizedUnused(int fromNode, bool[] used, double[,] cost, int topK, Random rng)
    {
        var pool = new List<(double c, int idx)>();
        for (int i = 0; i < used.Length; i++)
        {
            if (!used[i])
            {
                pool.Add((cost[fromNode, i + 1], i));
            }
        }

        pool.Sort((a, b) => a.c.CompareTo(b.c));
        int pickCount = Math.Min(topK, pool.Count);
        int selected = rng.Next(pickCount);
        return pool[selected].idx;
    }

    private static double InsertionDelta(List<int> route, int pos, int node, double[,] cost)
    {
        int insertNode = node + 1;
        if (route.Count == 0)
        {
            return cost[0, insertNode];
        }

        if (pos == 0)
        {
            return cost[0, insertNode] + cost[insertNode, route[0] + 1] - cost[0, route[0] + 1];
        }

        if (pos == route.Count)
        {
            return cost[route[pos - 1] + 1, insertNode];
        }

        int left = route[pos - 1] + 1;
        int right = route[pos] + 1;
        return cost[left, insertNode] + cost[insertNode, right] - cost[left, right];
    }

    private static void ImproveRoute(int[] route, double[,] cost, int maxPasses)
    {
        double current = RouteCost(route, cost);
        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            if (TryImproveByReverse(route, cost, ref current))
            {
                improved = true;
            }

            if (TryImproveByRelocate(route, cost, ref current))
            {
                improved = true;
            }

            if (TryImproveBySwap(route, cost, ref current))
            {
                improved = true;
            }

            if (!improved)
            {
                break;
            }
        }
    }

    private static bool TryImproveByReverse(int[] route, double[,] cost, ref double current)
    {
        bool improvedAny = false;
        bool improved;
        do
        {
            improved = false;
            for (int i = 0; i < route.Length - 1 && !improved; i++)
            {
                for (int j = i + 1; j < route.Length; j++)
                {
                    Array.Reverse(route, i, j - i + 1);
                    double candidate = RouteCost(route, cost);
                    if (candidate + Epsilon < current)
                    {
                        current = candidate;
                        improved = true;
                        improvedAny = true;
                        break;
                    }

                    Array.Reverse(route, i, j - i + 1);
                }
            }
        } while (improved);

        return improvedAny;
    }

    private static bool TryImproveByRelocate(int[] route, double[,] cost, ref double current)
    {
        bool improvedAny = false;
        bool improved;
        do
        {
            improved = false;
            for (int i = 0; i < route.Length && !improved; i++)
            {
                for (int j = 0; j < route.Length; j++)
                {
                    if (i == j)
                    {
                        continue;
                    }

                    Move(route, i, j);
                    double candidate = RouteCost(route, cost);
                    if (candidate + Epsilon < current)
                    {
                        current = candidate;
                        improved = true;
                        improvedAny = true;
                        break;
                    }

                    Move(route, j, i);
                }
            }
        } while (improved);

        return improvedAny;
    }

    private static bool TryImproveBySwap(int[] route, double[,] cost, ref double current)
    {
        bool improvedAny = false;
        bool improved;
        do
        {
            improved = false;
            for (int i = 0; i < route.Length - 1 && !improved; i++)
            {
                for (int j = i + 1; j < route.Length; j++)
                {
                    Swap(route, i, j);
                    double candidate = RouteCost(route, cost);
                    if (candidate + Epsilon < current)
                    {
                        current = candidate;
                        improved = true;
                        improvedAny = true;
                        break;
                    }

                    Swap(route, i, j);
                }
            }
        } while (improved);

        return improvedAny;
    }

    private static double RouteCost(int[] route, double[,] cost)
    {
        if (route.Length == 0)
        {
            return 0.0;
        }

        double total = cost[0, route[0] + 1];
        for (int i = 1; i < route.Length; i++)
        {
            total += cost[route[i - 1] + 1, route[i] + 1];
        }

        return total;
    }

    private static void Move(int[] route, int from, int to)
    {
        if (from == to)
        {
            return;
        }

        int value = route[from];
        if (from < to)
        {
            Array.Copy(route, from + 1, route, from, to - from);
            route[to] = value;
        }
        else
        {
            Array.Copy(route, to, route, to + 1, from - to);
            route[to] = value;
        }
    }

    private static void Swap(int[] route, int a, int b)
    {
        int temp = route[a];
        route[a] = route[b];
        route[b] = temp;
    }

    private static int BitIndex(int bit)
    {
        int index = 0;
        while ((bit >>= 1) != 0)
        {
            index++;
        }

        return index;
    }

    private static double SafeAcos(double x)
    {
        if (x < -1.0)
        {
            x = -1.0;
        }
        else if (x > 1.0)
        {
            x = 1.0;
        }

        return Math.Acos(x);
    }

    private readonly struct OrbitState
    {
        public OrbitState(Vec3 position, Vec3 velocity)
        {
            Position = position;
            Velocity = velocity;
        }

        public Vec3 Position { get; }
        public Vec3 Velocity { get; }

        public static OrbitState Read(FastScanner scanner)
        {
            var position = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
            var velocity = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
            return new OrbitState(position, velocity);
        }
    }

    private readonly struct OrbitFeatures
    {
        public OrbitFeatures(Vec3 position, Vec3 planeNormal, Vec3 eccentricityVector, double eccentricity, double semiMajorAxis, double speed, double characterRadius)
        {
            Position = position;
            PlaneNormal = planeNormal;
            EccentricityVector = eccentricityVector;
            Eccentricity = eccentricity;
            SemiMajorAxis = semiMajorAxis;
            Speed = speed;
            CharacterRadius = characterRadius;
        }

        public Vec3 Position { get; }
        public Vec3 PlaneNormal { get; }
        public Vec3 EccentricityVector { get; }
        public double Eccentricity { get; }
        public double SemiMajorAxis { get; }
        public double Speed { get; }
        public double CharacterRadius { get; }

        public static OrbitFeatures FromState(OrbitState state, double mu)
        {
            Vec3 r = state.Position;
            Vec3 v = state.Velocity;
            double rMag = r.Length();
            double vMag = v.Length();

            Vec3 h = Vec3.Cross(r, v);
            double hMag = Math.Max(Epsilon, h.Length());
            Vec3 normal = h / hMag;

            Vec3 eVec = (Vec3.Cross(v, h) / mu) - (r / Math.Max(Epsilon, rMag));
            double eccentricity = eVec.Length();

            double specificEnergy = 0.5 * vMag * vMag - mu / Math.Max(Epsilon, rMag);
            double semiMajorAxis = specificEnergy < -Epsilon
                ? -mu / (2.0 * specificEnergy)
                : rMag;

            double characterRadius = semiMajorAxis > 0.0
                ? semiMajorAxis * (1.0 + 0.5 * eccentricity * eccentricity)
                : rMag;

            characterRadius = Math.Max(6671.0, Math.Min(120000.0, characterRadius));

            return new OrbitFeatures(r, normal, eVec, eccentricity, semiMajorAxis, vMag, characterRadius);
        }
    }

    private readonly struct Vec3
    {
        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double X { get; }
        public double Y { get; }
        public double Z { get; }

        public double Length()
        {
            return Math.Sqrt((X * X) + (Y * Y) + (Z * Z));
        }

        public Vec3 Unit()
        {
            double len = Length();
            return len > Epsilon ? this / len : new Vec3(1.0, 0.0, 0.0);
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);

        public static double Dot(Vec3 a, Vec3 b)
        {
            return (a.X * b.X) + (a.Y * b.Y) + (a.Z * b.Z);
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                (a.Y * b.Z) - (a.Z * b.Y),
                (a.Z * b.X) - (a.X * b.Z),
                (a.X * b.Y) - (a.Y * b.X));
        }
    }

    private sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _index;

        public FastScanner()
        {
            string input = Console.In.ReadToEnd();
            _tokens = input.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        }

        public int NextInt()
        {
            return int.Parse(_tokens[_index++], CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(_tokens[_index++], CultureInfo.InvariantCulture);
        }
    }
}
