using System;
using System.Diagnostics;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        int nn = checked(n * n);

        var flow = new int[nn];
        var dist = new int[nn];

        for (int i = 0; i < nn; i++) flow[i] = scanner.NextInt();
        for (int i = 0; i < nn; i++) dist[i] = scanner.NextInt();

        int[] bestPerm;
        long bestCost;

        if (n <= 10)
        {
            SolveExactly(n, flow, dist, out bestPerm, out bestCost);
        }
        else
        {
            SolveHeuristically(n, flow, dist, out bestPerm, out bestCost);
        }

        bestCost = ComputeCost(n, flow, dist, bestPerm);

        Console.WriteLine(bestCost);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(bestPerm[i]);
        }
        Console.WriteLine();
    }

    private static void SolveExactly(int n, int[] flow, int[] dist, out int[] bestPerm, out long bestCost)
    {
        var flowScore = ComputeFlowScores(n, flow);
        var facilityOrder = CreateIndexArray(n);
        Array.Sort(facilityOrder, (a, b) => flowScore[b].CompareTo(flowScore[a]));

        var perm = new int[n];
        Array.Fill(perm, -1);
        var used = new bool[n];
        bestPerm = new int[n];
        bestCost = long.MaxValue;

        void Dfs(int depth, long partialCost)
        {
            if (partialCost >= bestCost) return;

            if (depth == n)
            {
                bestCost = partialCost;
                Array.Copy(perm, bestPerm, n);
                return;
            }

            int facility = facilityOrder[depth];
            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long delta = (long)flow[facility * n + facility] * dist[loc * n + loc];
                for (int d = 0; d < depth; d++)
                {
                    int other = facilityOrder[d];
                    int otherLoc = perm[other];
                    delta += (long)flow[facility * n + other] * dist[loc * n + otherLoc];
                    delta += (long)flow[other * n + facility] * dist[otherLoc * n + loc];
                }

                perm[facility] = loc;
                used[loc] = true;
                Dfs(depth + 1, partialCost + delta);
                used[loc] = false;
                perm[facility] = -1;
            }
        }

        Dfs(0, 0L);
    }

    private static void SolveHeuristically(int n, int[] flow, int[] dist, out int[] bestPerm, out long bestCost)
    {
        var timer = Stopwatch.StartNew();
        const long totalBudgetMs = 290000;

        var flowScore = ComputeFlowScores(n, flow);
        var distScore = ComputeDistanceScores(n, dist);
        var facilityOrder = CreateIndexArray(n);
        var locationOrder = CreateIndexArray(n);

        Array.Sort(facilityOrder, (a, b) => flowScore[b].CompareTo(flowScore[a]));
        Array.Sort(locationOrder, (a, b) => distScore[a].CompareTo(distScore[b]));

        int candidateCount = n <= 80 ? Math.Min(n - 1, 24) : n <= 400 ? 16 : 12;
        var nearest = BuildNearestLocations(n, dist, candidateCount);

        int starts = n <= 200 ? 6 : n <= 800 ? 4 : 3;
        var rng = new Random(123456789);
        bestPerm = new int[n];
        bestCost = long.MaxValue;

        for (int start = 0; start < starts && timer.ElapsedMilliseconds < totalBudgetMs - 1000; start++)
        {
            var perm = CreateInitialPermutation(n, facilityOrder, locationOrder, start, rng);
            var locToFacility = InvertPermutation(perm);
            long cost = ComputeCost(n, flow, dist, perm);

            long remaining = totalBudgetMs - timer.ElapsedMilliseconds;
            long slice = Math.Max(1000, remaining / Math.Max(1, starts - start));
            long target = Math.Min(totalBudgetMs, timer.ElapsedMilliseconds + slice);

            ImproveBySwaps(n, flow, dist, perm, locToFacility, nearest, facilityOrder, rng, timer, target, ref cost);

            if (cost < bestCost)
            {
                bestCost = cost;
                bestPerm = (int[])perm.Clone();
            }
        }

        while (timer.ElapsedMilliseconds < totalBudgetMs - 500)
        {
            var perm = (int[])bestPerm.Clone();
            var locToFacility = InvertPermutation(perm);
            Perturb(perm, locToFacility, rng, Math.Min(6, Math.Max(2, n / 400)));
            long cost = ComputeCost(n, flow, dist, perm);
            ImproveBySwaps(n, flow, dist, perm, locToFacility, nearest, facilityOrder, rng, timer, totalBudgetMs, ref cost);

            if (cost < bestCost)
            {
                bestCost = cost;
                bestPerm = perm;
            }
            else
            {
                break;
            }
        }
    }

    private static long[] ComputeFlowScores(int n, int[] flow)
    {
        var score = new long[n];
        for (int i = 0; i < n; i++)
        {
            int rowBase = i * n;
            for (int j = 0; j < n; j++)
            {
                score[i] += flow[rowBase + j];
                score[i] += flow[j * n + i];
            }
        }
        return score;
    }

    private static long[] ComputeDistanceScores(int n, int[] dist)
    {
        var score = new long[n];
        for (int i = 0; i < n; i++)
        {
            int rowBase = i * n;
            for (int j = 0; j < n; j++)
            {
                score[i] += dist[rowBase + j];
                score[i] += dist[j * n + i];
            }
        }
        return score;
    }

    private static int[] BuildNearestLocations(int n, int[] dist, int k)
    {
        var nearest = new int[n][];
        for (int i = 0; i < n; i++)
        {
            var bestLoc = new int[k];
            var bestVal = new int[k];
            for (int t = 0; t < k; t++)
            {
                bestLoc[t] = -1;
                bestVal[t] = int.MaxValue;
            }

            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                int value = dist[i * n + j] + dist[j * n + i];
                int pos = -1;
                for (int t = 0; t < k; t++)
                {
                    if (value < bestVal[t])
                    {
                        pos = t;
                        break;
                    }
                }

                if (pos < 0) continue;
                for (int t = k - 1; t > pos; t--)
                {
                    bestVal[t] = bestVal[t - 1];
                    bestLoc[t] = bestLoc[t - 1];
                }
                bestVal[pos] = value;
                bestLoc[pos] = j;
            }

            nearest[i] = bestLoc;
        }

        return nearest;
    }

    private static int[] CreateInitialPermutation(int n, int[] facilityOrder, int[] locationOrder, int variant, Random rng)
    {
        var facilities = (int[])facilityOrder.Clone();
        var locations = (int[])locationOrder.Clone();
        int block = variant == 0 ? 1 : Math.Min(8, Math.Max(2, n / 50 + 2));

        if (variant > 0)
        {
            ShuffleWithinBlocks(facilities, block, rng);
            ShuffleWithinBlocks(locations, block, rng);
        }

        var perm = new int[n];
        for (int i = 0; i < n; i++)
        {
            perm[facilities[i]] = locations[i];
        }
        return perm;
    }

    private static void ShuffleWithinBlocks(int[] values, int block, Random rng)
    {
        for (int start = 0; start < values.Length; start += block)
        {
            int end = Math.Min(values.Length, start + block);
            for (int i = end - 1; i > start; i--)
            {
                int j = start + rng.Next(i - start + 1);
                (values[i], values[j]) = (values[j], values[i]);
            }
        }
    }

    private static int[] InvertPermutation(int[] perm)
    {
        int n = perm.Length;
        var inv = new int[n];
        for (int facility = 0; facility < n; facility++)
        {
            inv[perm[facility]] = facility;
        }
        return inv;
    }

    private static void Perturb(int[] perm, int[] locToFacility, Random rng, int swaps)
    {
        int n = perm.Length;
        for (int s = 0; s < swaps; s++)
        {
            int a = rng.Next(n);
            int b = rng.Next(n - 1);
            if (b >= a) b++;
            ApplySwap(perm, locToFacility, a, b);
        }
    }

    private static void ImproveBySwaps(
        int n,
        int[] flow,
        int[] dist,
        int[] perm,
        int[] locToFacility,
        int[][] nearest,
        int[] facilityOrder,
        Random rng,
        Stopwatch timer,
        long deadlineMs,
        ref long currentCost)
    {
        int randomTrials = n <= 200 ? 8 : n <= 800 ? 6 : 4;
        int stagnantPasses = 0;

        while (timer.ElapsedMilliseconds < deadlineMs)
        {
            bool improvedPass = false;

            for (int idx = 0; idx < n; idx++)
            {
                if (timer.ElapsedMilliseconds >= deadlineMs) return;

                int a = facilityOrder[idx];
                int pa = perm[a];
                long bestDelta = 0;
                int bestB = -1;

                int[] candidateLocs = nearest[pa];
                for (int c = 0; c < candidateLocs.Length; c++)
                {
                    int loc = candidateLocs[c];
                    if (loc < 0) continue;
                    int b = locToFacility[loc];
                    if (b == a) continue;

                    long delta = ComputeSwapDelta(n, flow, dist, perm, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestB = b;
                    }
                }

                for (int t = 0; t < randomTrials; t++)
                {
                    int b = rng.Next(n - 1);
                    if (b >= a) b++;

                    long delta = ComputeSwapDelta(n, flow, dist, perm, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestB = b;
                    }
                }

                if (bestB >= 0)
                {
                    ApplySwap(perm, locToFacility, a, bestB);
                    currentCost += bestDelta;
                    improvedPass = true;
                }
            }

            if (!improvedPass)
            {
                stagnantPasses++;
                if (stagnantPasses >= 2) break;
            }
            else
            {
                stagnantPasses = 0;
            }
        }
    }

    private static long ComputeSwapDelta(int n, int[] flow, int[] dist, int[] perm, int a, int b)
    {
        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            delta += (long)flow[a * n + k] * (dist[pb * n + pk] - dist[pa * n + pk]);
            delta += (long)flow[b * n + k] * (dist[pa * n + pk] - dist[pb * n + pk]);
            delta += (long)flow[k * n + a] * (dist[pk * n + pb] - dist[pk * n + pa]);
            delta += (long)flow[k * n + b] * (dist[pk * n + pa] - dist[pk * n + pb]);
        }

        delta += (long)flow[a * n + a] * (dist[pb * n + pb] - dist[pa * n + pa]);
        delta += (long)flow[b * n + b] * (dist[pa * n + pa] - dist[pb * n + pb]);
        delta += (long)flow[a * n + b] * (dist[pb * n + pa] - dist[pa * n + pb]);
        delta += (long)flow[b * n + a] * (dist[pa * n + pb] - dist[pb * n + pa]);

        return delta;
    }

    private static void ApplySwap(int[] perm, int[] locToFacility, int a, int b)
    {
        int pa = perm[a];
        int pb = perm[b];
        perm[a] = pb;
        perm[b] = pa;
        locToFacility[pa] = b;
        locToFacility[pb] = a;
    }

    private static long ComputeCost(int n, int[] flow, int[] dist, int[] perm)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int flowBase = i * n;
            int distBase = pi * n;
            for (int j = 0; j < n; j++)
            {
                total += (long)flow[flowBase + j] * dist[distBase + perm[j]];
            }
        }
        return total;
    }

    private static int[] CreateIndexArray(int n)
    {
        var arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = i;
        return arr;
    }
}
