#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Coord = std::vector<int>;

struct VectorHash {
    static std::uint64_t mix(std::uint64_t x) {
        x += 0x9e3779b97f4a7c15ULL;
        x = (x ^ (x >> 30U)) * 0xbf58476d1ce4e5b9ULL;
        x = (x ^ (x >> 27U)) * 0x94d049bb133111ebULL;
        return x ^ (x >> 31U);
    }

    std::size_t operator()(const Coord& v) const noexcept {
        std::uint64_t h = 0x1234567890abcdefULL;
        for (std::size_t i = 0; i < v.size(); ++i) {
            const std::uint64_t part =
                static_cast<std::uint64_t>(static_cast<std::int64_t>(v[i]) + 0x100000000ULL);
            h ^= mix(part + 0x9e3779b97f4a7c15ULL * (i + 1));
            h = mix(h);
        }
        return static_cast<std::size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

struct Node {
    Coord pos;
    int g = 0;
    int h = 0;
    int parent = -1;
    Move move;
};

struct QueueItem {
    int f = 0;
    int g = 0;
    int index = -1;

    bool operator<(const QueueItem& other) const {
        if (f != other.f) return f > other.f;
        return g < other.g;
    }
};

class SnakeSolver {
public:
    void run() {
        readInput();
        removeFoodAtHeadIfPresent();

        std::vector<Move> answer;
        answer.reserve(static_cast<std::size_t>(maxTurns));

        for (int turn = 0; turn < maxTurns && !foods.empty(); ++turn) {
            const Move move = chooseMove();
            if (!applyMove(move)) {
                break;
            }
            answer.push_back(move);
        }

        for (const Move& move : answer) {
            std::cout << move.axis << ' ' << move.dir << '\n';
        }
    }

private:
    int dims = 0;
    int maxTurns = 0;
    Coord bounds;
    std::deque<Coord> body;
    std::unordered_map<Coord, int, VectorHash> occupied;
    std::unordered_set<Coord, VectorHash> obstacles;
    std::unordered_map<Coord, int, VectorHash> foods;

    void readInput() {
        std::cin >> dims >> maxTurns;
        bounds.resize(dims);
        for (int i = 0; i < dims; ++i) {
            std::cin >> bounds[i];
        }

        int snakeLength = 0;
        Coord head(dims);
        std::cin >> snakeLength;
        for (int i = 0; i < dims; ++i) {
            std::cin >> head[i];
        }

        if (snakeLength <= 0) snakeLength = 1;
        for (int i = 0; i < snakeLength; ++i) {
            body.push_back(head);
        }
        occupied[head] = snakeLength;

        int foodCount = 0;
        std::cin >> foodCount;
        for (int i = 0; i < foodCount; ++i) {
            Coord p(dims);
            for (int d = 0; d < dims; ++d) {
                std::cin >> p[d];
            }
            foods[p] = 1;
        }

        int obstacleCount = 0;
        std::cin >> obstacleCount;
        for (int i = 0; i < obstacleCount; ++i) {
            Coord p(dims);
            for (int d = 0; d < dims; ++d) {
                std::cin >> p[d];
            }
            obstacles.insert(std::move(p));
        }
    }

    void removeFoodAtHeadIfPresent() {
        auto it = foods.find(body.front());
        if (it == foods.end()) return;
        foods.erase(it);
    }

    static int manhattan(const Coord& a, const Coord& b) {
        int total = 0;
        for (std::size_t i = 0; i < a.size(); ++i) {
            total += std::abs(a[i] - b[i]);
        }
        return total;
    }

    bool inBounds(const Coord& p) const {
        for (int i = 0; i < dims; ++i) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    }

    bool cellBlocked(const Coord& p, bool tailMoves, const Coord& tail) const {
        if (!inBounds(p)) return true;
        if (obstacles.find(p) != obstacles.end()) return true;

        auto it = occupied.find(p);
        if (it == occupied.end()) return false;

        int count = it->second;
        if (tailMoves && p == tail) {
            --count;
        }
        return count > 0;
    }

    bool applyMove(const Move& move) {
        Coord next = body.front();
        next[move.axis] += move.dir;

        const bool grow = foods.find(next) != foods.end();
        const Coord tail = body.back();
        const bool tailMoves = !grow;

        if (cellBlocked(next, tailMoves, tail)) {
            return false;
        }

        body.push_front(next);
        ++occupied[next];

        if (grow) {
            foods.erase(next);
        } else {
            auto it = occupied.find(tail);
            if (it != occupied.end()) {
                --it->second;
                if (it->second == 0) occupied.erase(it);
            }
            body.pop_back();
        }
        return true;
    }

    std::vector<std::pair<Coord, int>> nearestFoods(int limit) const {
        std::vector<std::pair<Coord, int>> items;
        items.reserve(foods.size());
        const Coord& head = body.front();
        for (const auto& [food, _] : foods) {
            items.push_back({food, manhattan(head, food)});
        }
        std::sort(items.begin(), items.end(), [](const auto& lhs, const auto& rhs) {
            return lhs.second < rhs.second;
        });
        if (static_cast<int>(items.size()) > limit) items.resize(limit);
        return items;
    }

    std::vector<Move> reconstructPath(const std::vector<Node>& nodes, int idx) const {
        std::vector<Move> path;
        while (idx >= 0 && nodes[idx].parent >= 0) {
            path.push_back(nodes[idx].move);
            idx = nodes[idx].parent;
        }
        std::reverse(path.begin(), path.end());
        return path;
    }

    std::vector<Move> findPathTo(const Coord& goal, int budget) const {
        const Coord& start = body.front();
        const Coord& tail = body.back();

        if (start == goal) return {};

        std::priority_queue<QueueItem> open;
        std::vector<Node> nodes;
        nodes.reserve(static_cast<std::size_t>(budget / 2 + 8));
        std::unordered_map<Coord, int, VectorHash> bestG;
        bestG.reserve(static_cast<std::size_t>(budget * 2 + 8));

        nodes.push_back(Node{start, 0, manhattan(start, goal), -1, Move{}});
        open.push(QueueItem{nodes[0].g + nodes[0].h, nodes[0].g, 0});
        bestG[start] = 0;

        int expanded = 0;
        while (!open.empty() && expanded < budget) {
            const QueueItem top = open.top();
            open.pop();
            const Node& cur = nodes[top.index];

            auto bestIt = bestG.find(cur.pos);
            if (bestIt == bestG.end() || bestIt->second != cur.g) {
                continue;
            }

            if (cur.pos == goal) {
                return reconstructPath(nodes, top.index);
            }

            ++expanded;
            for (int axis = 0; axis < dims; ++axis) {
                for (int dir : {-1, 1}) {
                    Coord next = cur.pos;
                    next[axis] += dir;

                    const bool grow = (next == goal) && (foods.find(goal) != foods.end());
                    const bool tailMoves = !grow;
                    if (cellBlocked(next, tailMoves, tail)) {
                        continue;
                    }

                    const int ng = cur.g + 1;
                    auto it = bestG.find(next);
                    if (it != bestG.end() && it->second <= ng) {
                        continue;
                    }

                    const int nh = manhattan(next, goal);
                    const int idx = static_cast<int>(nodes.size());
                    nodes.push_back(Node{next, ng, nh, top.index, Move{axis, dir}});
                    bestG[next] = ng;
                    open.push(QueueItem{ng + nh, ng, idx});
                }
            }
        }

        return {};
    }

    Move safeFallbackMove() const {
        const Coord& head = body.front();
        const Coord& tail = body.back();
        const Coord target = foods.empty() ? tail : nearestFoods(1).front().first;

        int bestScore = std::numeric_limits<int>::min();
        Move bestMove{0, 1};

        for (int axis = 0; axis < dims; ++axis) {
            for (int dir : {-1, 1}) {
                Coord next = head;
                next[axis] += dir;
                const bool grow = foods.find(next) != foods.end();
                const bool tailMoves = !grow;
                if (cellBlocked(next, tailMoves, tail)) continue;

                int freeDegree = 0;
                for (int nextAxis = 0; nextAxis < dims; ++nextAxis) {
                    for (int nextDir : {-1, 1}) {
                        Coord probe = next;
                        probe[nextAxis] += nextDir;
                        if (!cellBlocked(probe, true, tail)) {
                            ++freeDegree;
                        }
                    }
                }

                const int progress = 1000 - manhattan(next, target);
                const int eatBonus = grow ? 500 : 0;
                const int score = freeDegree * 10000 + progress + eatBonus;
                if (score > bestScore) {
                    bestScore = score;
                    bestMove = Move{axis, dir};
                }
            }
        }

        return bestMove;
    }

    int searchBudget() const {
        if (dims <= 2) return 250000;
        if (dims <= 4) return 150000;
        if (dims <= 8) return 60000;
        if (dims <= 12) return 25000;
        return 12000;
    }

    Move chooseMove() const {
        const int budget = searchBudget();
        const auto candidates = nearestFoods(std::min<int>(6, static_cast<int>(foods.size())));

        std::vector<Move> bestPath;
        for (std::size_t i = 0; i < candidates.size(); ++i) {
            const int perTargetBudget = std::max(2000, budget / static_cast<int>(candidates.size()));
            std::vector<Move> path = findPathTo(candidates[i].first, perTargetBudget);
            if (!path.empty() && (bestPath.empty() || path.size() < bestPath.size())) {
                bestPath = std::move(path);
                if (bestPath.size() == 1) break;
            }
        }

        if (!bestPath.empty()) {
            return bestPath.front();
        }

        if (body.size() > 1) {
            std::vector<Move> tailPath = findPathTo(body.back(), budget / 2);
            if (!tailPath.empty()) {
                return tailPath.front();
            }
        }

        return safeFallbackMove();
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    SnakeSolver solver;
    solver.run();
    return 0;
}
