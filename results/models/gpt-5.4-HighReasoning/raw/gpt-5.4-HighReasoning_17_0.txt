using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

internal static class Program
{
    private struct Box
    {
        public int Index;
        public int W;
        public int H;
        public int D;
        public int Volume;
        public int MinDim;
        public int MaxDim;
    }

    private struct Orientation
    {
        public int W;
        public int H;
        public int D;
    }

    private struct Placement
    {
        public int BoxIndex;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private sealed class Shelf
    {
        public int Y;
        public int Height;
        public int UsedWidth;
        public readonly List<Placement> Placements = new List<Placement>();
    }

    private sealed class Layer
    {
        public int Z;
        public int Depth;
        public int UsedHeight;
        public readonly List<Shelf> Shelves = new List<Shelf>();
    }

    private sealed class HeuristicResult
    {
        public readonly List<Placement> Placements = new List<Placement>();
        public int Count => Placements.Count;
        public long UsedVolume;
    }

    private sealed class SmallResult
    {
        public readonly List<Placement> Placements = new List<Placement>();
        public int Count => Placements.Count;
        public long UsedVolume;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private static int _containerW;
    private static int _containerH;
    private static int _containerD;
    private static Box[] _boxes = Array.Empty<Box>();
    private static Orientation[][] _orientations = Array.Empty<Orientation[]>();
    private static SmallResult _bestSmall = new SmallResult();

    public static void Main()
    {
        var first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        var head = first.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(head[0]);
        _containerW = int.Parse(head[1]);
        _containerH = int.Parse(head[2]);
        _containerD = int.Parse(head[3]);

        _boxes = new Box[n];
        _orientations = new Orientation[n][];
        for (int i = 0; i < n; i++)
        {
            var parts = Console.ReadLine()!.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            int w = int.Parse(parts[0]);
            int h = int.Parse(parts[1]);
            int d = int.Parse(parts[2]);
            _boxes[i] = new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Volume = w * h * d,
                MinDim = Math.Min(w, Math.Min(h, d)),
                MaxDim = Math.Max(w, Math.Max(h, d))
            };
            _orientations[i] = BuildOrientations(w, h, d);
        }

        List<Placement> bestPlacements;
        if (ShouldUseSmallSearch(n))
        {
            bestPlacements = SolveSmall();
        }
        else
        {
            bestPlacements = SolveHeuristic();
        }

        Console.WriteLine(bestPlacements.Count);
        foreach (var p in bestPlacements)
        {
            Console.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
    }

    private static bool ShouldUseSmallSearch(int n)
    {
        long volume = 1L * _containerW * _containerH * _containerD;
        return n <= 18 && volume <= 2_000_000L;
    }

    private static Orientation[] BuildOrientations(int a, int b, int c)
    {
        var seen = new HashSet<long>();
        var list = new List<Orientation>(6);
        AddOrientation(a, b, c, seen, list);
        AddOrientation(a, c, b, seen, list);
        AddOrientation(b, a, c, seen, list);
        AddOrientation(b, c, a, seen, list);
        AddOrientation(c, a, b, seen, list);
        AddOrientation(c, b, a, seen, list);
        return list.ToArray();
    }

    private static void AddOrientation(int w, int h, int d, HashSet<long> seen, List<Orientation> list)
    {
        long key = ((long)w << 40) ^ ((long)h << 20) ^ (uint)d;
        if (seen.Add(key))
        {
            list.Add(new Orientation { W = w, H = h, D = d });
        }
    }

    private static List<Placement> SolveSmall()
    {
        _bestSmall = new SmallResult();
        var spaces = new List<Space>
        {
            new Space { X = 0, Y = 0, Z = 0, W = _containerW, H = _containerH, D = _containerD }
        };

        var candidates = Enumerable.Range(0, _boxes.Length)
            .Where(CanFitContainer)
            .OrderByDescending(i => _boxes[i].Volume)
            .ThenByDescending(i => _boxes[i].MaxDim)
            .ToArray();

        var used = new bool[_boxes.Length];
        var current = new List<Placement>();
        SearchSmall(spaces, candidates, used, current, 0L, 0);
        return _bestSmall.Placements;
    }

    private static void SearchSmall(
        List<Space> spaces,
        int[] candidates,
        bool[] used,
        List<Placement> current,
        long usedVolume,
        int placedCount)
    {
        if (placedCount > _bestSmall.Count || (placedCount == _bestSmall.Count && usedVolume > _bestSmall.UsedVolume))
        {
            _bestSmall = new SmallResult { UsedVolume = usedVolume };
            _bestSmall.Placements.AddRange(current);
        }

        int remainingFitCount = 0;
        for (int i = 0; i < candidates.Length; i++)
        {
            if (!used[candidates[i]])
            {
                remainingFitCount++;
            }
        }

        if (placedCount + remainingFitCount <= _bestSmall.Count)
        {
            return;
        }

        int spaceIndex = SelectSpace(spaces);
        if (spaceIndex < 0)
        {
            return;
        }

        var space = spaces[spaceIndex];
        bool placedAny = false;

        foreach (int boxId in candidates)
        {
            if (used[boxId])
            {
                continue;
            }

            var box = _boxes[boxId];
            foreach (var o in _orientations[boxId])
            {
                if (o.W > space.W || o.H > space.H || o.D > space.D)
                {
                    continue;
                }

                placedAny = true;
                used[boxId] = true;

                var placement = new Placement
                {
                    BoxIndex = box.Index,
                    X = space.X,
                    Y = space.Y,
                    Z = space.Z,
                    W = o.W,
                    H = o.H,
                    D = o.D
                };
                current.Add(placement);

                var nextSpaces = CloneSpacesWithout(spaces, spaceIndex);
                AddSplitSpaces(nextSpaces, space, o.W, o.H, o.D);
                NormalizeSpaces(nextSpaces);
                SearchSmall(nextSpaces, candidates, used, current, usedVolume + box.Volume, placedCount + 1);

                current.RemoveAt(current.Count - 1);
                used[boxId] = false;
            }
        }

        if (!placedAny)
        {
            var nextSpaces = CloneSpacesWithout(spaces, spaceIndex);
            SearchSmall(nextSpaces, candidates, used, current, usedVolume, placedCount);
        }
    }

    private static int SelectSpace(List<Space> spaces)
    {
        int best = -1;
        long bestVolume = -1;
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            if (s.W <= 0 || s.H <= 0 || s.D <= 0)
            {
                continue;
            }

            long volume = 1L * s.W * s.H * s.D;
            if (volume > bestVolume)
            {
                bestVolume = volume;
                best = i;
            }
        }
        return best;
    }

    private static List<Space> CloneSpacesWithout(List<Space> spaces, int skipIndex)
    {
        var next = new List<Space>(spaces.Count + 2);
        for (int i = 0; i < spaces.Count; i++)
        {
            if (i != skipIndex)
            {
                next.Add(spaces[i]);
            }
        }
        return next;
    }

    private static void AddSplitSpaces(List<Space> spaces, Space s, int w, int h, int d)
    {
        if (s.W - w > 0)
        {
            spaces.Add(new Space { X = s.X + w, Y = s.Y, Z = s.Z, W = s.W - w, H = s.H, D = s.D });
        }
        if (s.H - h > 0)
        {
            spaces.Add(new Space { X = s.X, Y = s.Y + h, Z = s.Z, W = w, H = s.H - h, D = s.D });
        }
        if (s.D - d > 0)
        {
            spaces.Add(new Space { X = s.X, Y = s.Y, Z = s.Z + d, W = w, H = h, D = s.D - d });
        }
    }

    private static void NormalizeSpaces(List<Space> spaces)
    {
        spaces.RemoveAll(s => s.W <= 0 || s.H <= 0 || s.D <= 0);
        spaces.Sort((a, b) =>
        {
            long vb = 1L * b.W * b.H * b.D;
            long va = 1L * a.W * a.H * a.D;
            int cmp = vb.CompareTo(va);
            if (cmp != 0)
            {
                return cmp;
            }
            cmp = a.Z.CompareTo(b.Z);
            if (cmp != 0)
            {
                return cmp;
            }
            cmp = a.Y.CompareTo(b.Y);
            if (cmp != 0)
            {
                return cmp;
            }
            return a.X.CompareTo(b.X);
        });

        for (int i = 0; i < spaces.Count; i++)
        {
            if (spaces[i].W <= 0)
            {
                continue;
            }
            for (int j = spaces.Count - 1; j > i; j--)
            {
                if (Contains(spaces[i], spaces[j]))
                {
                    spaces.RemoveAt(j);
                }
            }
        }
    }

    private static bool Contains(Space outer, Space inner)
    {
        return inner.X >= outer.X &&
               inner.Y >= outer.Y &&
               inner.Z >= outer.Z &&
               inner.X + inner.W <= outer.X + outer.W &&
               inner.Y + inner.H <= outer.Y + outer.H &&
               inner.Z + inner.D <= outer.Z + outer.D;
    }

    private static bool CanFitContainer(int boxId)
    {
        foreach (var o in _orientations[boxId])
        {
            if (o.W <= _containerW && o.H <= _containerH && o.D <= _containerD)
            {
                return true;
            }
        }
        return false;
    }

    private static List<Placement> SolveHeuristic()
    {
        var candidates = BuildHeuristicOrders();
        HeuristicResult best = null;
        foreach (var order in candidates)
        {
            var result = RunShelfHeuristic(order);
            if (best == null ||
                result.Count > best.Count ||
                (result.Count == best.Count && result.UsedVolume > best.UsedVolume))
            {
                best = result;
            }
        }

        return best?.Placements ?? new List<Placement>();
    }

    private static List<int[]> BuildHeuristicOrders()
    {
        var valid = Enumerable.Range(0, _boxes.Length).Where(CanFitContainer).ToArray();

        int[] volAsc = valid
            .OrderBy(i => _boxes[i].Volume)
            .ThenBy(i => _boxes[i].MaxDim)
            .ThenBy(i => _boxes[i].MinDim)
            .ToArray();

        int[] minDimAsc = valid
            .OrderBy(i => _boxes[i].MinDim)
            .ThenBy(i => _boxes[i].Volume)
            .ThenBy(i => _boxes[i].MaxDim)
            .ToArray();

        int[] footprintAsc = valid
            .OrderBy(i => BestFootprint(i))
            .ThenBy(i => BestDepth(i))
            .ThenBy(i => _boxes[i].Volume)
            .ToArray();

        int[] volDesc = valid
            .OrderByDescending(i => _boxes[i].Volume)
            .ThenBy(i => _boxes[i].MinDim)
            .ToArray();

        int[] maxDimAsc = valid
            .OrderBy(i => _boxes[i].MaxDim)
            .ThenBy(i => _boxes[i].Volume)
            .ToArray();

        return new List<int[]> { volAsc, minDimAsc, footprintAsc, volDesc, maxDimAsc };
    }

    private static int BestFootprint(int boxId)
    {
        int best = int.MaxValue;
        foreach (var o in _orientations[boxId])
        {
            if (o.W <= _containerW && o.H <= _containerH && o.D <= _containerD)
            {
                int fp = o.W * o.H;
                if (fp < best)
                {
                    best = fp;
                }
            }
        }
        return best;
    }

    private static int BestDepth(int boxId)
    {
        int best = int.MaxValue;
        foreach (var o in _orientations[boxId])
        {
            if (o.W <= _containerW && o.H <= _containerH && o.D <= _containerD && o.D < best)
            {
                best = o.D;
            }
        }
        return best;
    }

    private static HeuristicResult RunShelfHeuristic(int[] order)
    {
        var result = new HeuristicResult();
        var layers = new List<Layer>();
        int usedDepth = 0;

        foreach (int boxId in order)
        {
            if (TryPlaceInExisting(layers, boxId, result))
            {
                result.UsedVolume += _boxes[boxId].Volume;
                continue;
            }

            if (TryCreateShelf(layers, boxId, result))
            {
                result.UsedVolume += _boxes[boxId].Volume;
                continue;
            }

            if (TryCreateLayer(layers, ref usedDepth, boxId, result))
            {
                result.UsedVolume += _boxes[boxId].Volume;
            }
        }

        return result;
    }

    private static bool TryPlaceInExisting(List<Layer> layers, int boxId, HeuristicResult result)
    {
        int bestLayer = -1;
        int bestShelf = -1;
        Orientation bestOrientation = default;
        long bestScore = long.MaxValue;

        for (int li = 0; li < layers.Count; li++)
        {
            var layer = layers[li];
            for (int si = 0; si < layer.Shelves.Count; si++)
            {
                var shelf = layer.Shelves[si];
                foreach (var o in _orientations[boxId])
                {
                    if (o.D > layer.Depth || o.H > shelf.Height || shelf.UsedWidth + o.W > _containerW)
                    {
                        continue;
                    }

                    long score = 1_000_000L * (layer.Depth - o.D)
                               + 1_000L * (shelf.Height - o.H)
                               + (_containerW - (shelf.UsedWidth + o.W));
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestLayer = li;
                        bestShelf = si;
                        bestOrientation = o;
                    }
                }
            }
        }

        if (bestLayer < 0)
        {
            return false;
        }

        var chosenLayer = layers[bestLayer];
        var chosenShelf = chosenLayer.Shelves[bestShelf];
        var placement = new Placement
        {
            BoxIndex = _boxes[boxId].Index,
            X = chosenShelf.UsedWidth,
            Y = chosenShelf.Y,
            Z = chosenLayer.Z,
            W = bestOrientation.W,
            H = bestOrientation.H,
            D = bestOrientation.D
        };
        chosenShelf.UsedWidth += bestOrientation.W;
        chosenShelf.Placements.Add(placement);
        result.Placements.Add(placement);
        return true;
    }

    private static bool TryCreateShelf(List<Layer> layers, int boxId, HeuristicResult result)
    {
        int bestLayer = -1;
        Orientation bestOrientation = default;
        long bestScore = long.MaxValue;

        for (int li = 0; li < layers.Count; li++)
        {
            var layer = layers[li];
            foreach (var o in _orientations[boxId])
            {
                if (o.D > layer.Depth || o.W > _containerW || layer.UsedHeight + o.H > _containerH)
                {
                    continue;
                }

                long score = 1_000_000L * (layer.Depth - o.D)
                           + 1_000L * (_containerH - (layer.UsedHeight + o.H))
                           + (_containerW - o.W);
                if (score < bestScore)
                {
                    bestScore = score;
                    bestLayer = li;
                    bestOrientation = o;
                }
            }
        }

        if (bestLayer < 0)
        {
            return false;
        }

        var chosenLayer = layers[bestLayer];
        var shelf = new Shelf
        {
            Y = chosenLayer.UsedHeight,
            Height = bestOrientation.H,
            UsedWidth = bestOrientation.W
        };
        chosenLayer.UsedHeight += bestOrientation.H;
        chosenLayer.Shelves.Add(shelf);

        var placement = new Placement
        {
            BoxIndex = _boxes[boxId].Index,
            X = 0,
            Y = shelf.Y,
            Z = chosenLayer.Z,
            W = bestOrientation.W,
            H = bestOrientation.H,
            D = bestOrientation.D
        };
        shelf.Placements.Add(placement);
        result.Placements.Add(placement);
        return true;
    }

    private static bool TryCreateLayer(List<Layer> layers, ref int usedDepth, int boxId, HeuristicResult result)
    {
        Orientation bestOrientation = default;
        long bestScore = long.MaxValue;
        bool found = false;

        foreach (var o in _orientations[boxId])
        {
            if (o.W > _containerW || o.H > _containerH || usedDepth + o.D > _containerD)
            {
                continue;
            }

            long score = 1_000_000L * (_containerD - (usedDepth + o.D))
                       + 1_000L * (_containerH - o.H)
                       + (_containerW - o.W);
            if (score < bestScore)
            {
                bestScore = score;
                bestOrientation = o;
                found = true;
            }
        }

        if (!found)
        {
            return false;
        }

        var layer = new Layer
        {
            Z = usedDepth,
            Depth = bestOrientation.D,
            UsedHeight = bestOrientation.H
        };
        var shelf = new Shelf
        {
            Y = 0,
            Height = bestOrientation.H,
            UsedWidth = bestOrientation.W
        };
        layer.Shelves.Add(shelf);
        layers.Add(layer);

        var placement = new Placement
        {
            BoxIndex = _boxes[boxId].Index,
            X = 0,
            Y = 0,
            Z = usedDepth,
            W = bestOrientation.W,
            H = bestOrientation.H,
            D = bestOrientation.D
        };
        shelf.Placements.Add(placement);
        result.Placements.Add(placement);
        usedDepth += bestOrientation.D;
        return true;
    }
}
