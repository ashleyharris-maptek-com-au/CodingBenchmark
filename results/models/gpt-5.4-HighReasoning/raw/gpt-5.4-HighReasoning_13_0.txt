#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Solver {
    vector<string> strings;
    vector<int> order;

    static constexpr uint64_t BASE1 = 1469598103934665603ULL;
    static constexpr uint64_t BASE2 = 1099511628211ULL;

    struct HashKey {
        uint64_t a;
        uint64_t b;

        bool operator==(const HashKey& other) const noexcept {
            return a == other.a && b == other.b;
        }
    };

    struct HashKeyHasher {
        size_t operator()(const HashKey& key) const noexcept {
            uint64_t x = key.a ^ (key.b + 0x9e3779b97f4a7c15ULL + (key.a << 6) + (key.a >> 2));
            return static_cast<size_t>(x ^ (x >> 32));
        }
    };

    struct RollingHashWitness {
        bool exists = false;
        vector<int> positions;
    };

    struct SAMSolver {
        struct Edge {
            int to = -1;
            int next = -1;
            unsigned char ch = 0;
        };

        struct State {
            int link = -1;
            int len = 0;
            int first_pos = -1;
            int head = -1;
        };

        vector<State> st;
        vector<Edge> edges;
        vector<int> best;
        vector<int> topo;

        explicit SAMSolver(size_t reserve_len = 0) {
            st.reserve(reserve_len * 2 + 1);
            edges.reserve(reserve_len * 4 + 1);
            st.push_back(State{});
        }

        int find_transition(int v, unsigned char ch) const {
            for (int e = st[v].head; e != -1; e = edges[e].next) {
                if (edges[e].ch == ch) {
                    return edges[e].to;
                }
            }
            return -1;
        }

        void set_transition(int v, unsigned char ch, int to) {
            for (int e = st[v].head; e != -1; e = edges[e].next) {
                if (edges[e].ch == ch) {
                    edges[e].to = to;
                    return;
                }
            }
            edges.push_back(Edge{to, st[v].head, ch});
            st[v].head = static_cast<int>(edges.size()) - 1;
        }

        int clone_state(int src) {
            int id = static_cast<int>(st.size());
            st.push_back(st[src]);
            vector<pair<unsigned char, int>> copied;
            copied.reserve(8);
            for (int e = st[src].head; e != -1; e = edges[e].next) {
                copied.push_back({edges[e].ch, edges[e].to});
            }
            st[id].head = -1;
            for (auto it = copied.rbegin(); it != copied.rend(); ++it) {
                edges.push_back(Edge{it->second, st[id].head, it->first});
                st[id].head = static_cast<int>(edges.size()) - 1;
            }
            return id;
        }

        void build(const string& s) {
            int last = 0;
            for (int i = 0; i < static_cast<int>(s.size()); ++i) {
                unsigned char ch = static_cast<unsigned char>(s[i]);
                int cur = static_cast<int>(st.size());
                st.push_back(State{0, st[last].len + 1, i, -1});
                int p = last;
                while (p != -1 && find_transition(p, ch) == -1) {
                    set_transition(p, ch, cur);
                    p = st[p].link;
                }
                if (p == -1) {
                    st[cur].link = 0;
                } else {
                    int q = find_transition(p, ch);
                    if (st[p].len + 1 == st[q].len) {
                        st[cur].link = q;
                    } else {
                        int clone = clone_state(q);
                        st[clone].len = st[p].len + 1;
                        while (p != -1 && find_transition(p, ch) == q) {
                            set_transition(p, ch, clone);
                            p = st[p].link;
                        }
                        st[q].link = clone;
                        st[cur].link = clone;
                    }
                }
                last = cur;
            }

            best.resize(st.size());
            for (size_t i = 0; i < st.size(); ++i) {
                best[i] = st[i].len;
            }

            topo.resize(st.size());
            iota(topo.begin(), topo.end(), 0);
            sort(topo.begin(), topo.end(), [&](int a, int b) {
                return st[a].len < st[b].len;
            });
        }

        void intersect_with(const string& s) {
            vector<int> mx(st.size(), 0);
            int v = 0;
            int l = 0;
            for (char raw : s) {
                unsigned char ch = static_cast<unsigned char>(raw);
                while (v != 0 && find_transition(v, ch) == -1) {
                    v = st[v].link;
                    l = st[v].len;
                }
                int to = find_transition(v, ch);
                if (to != -1) {
                    v = to;
                    ++l;
                } else {
                    v = 0;
                    l = 0;
                }
                if (mx[v] < l) {
                    mx[v] = l;
                }
            }

            for (int i = static_cast<int>(topo.size()) - 1; i > 0; --i) {
                int state = topo[i];
                int parent = st[state].link;
                if (parent >= 0) {
                    int propagated = min(mx[state], st[parent].len);
                    if (mx[parent] < propagated) {
                        mx[parent] = propagated;
                    }
                }
            }

            for (size_t i = 0; i < st.size(); ++i) {
                best[i] = min(best[i], mx[i]);
            }
        }

        string answer_from(const string& source) const {
            int best_len = 0;
            int state_id = 0;
            for (size_t i = 0; i < st.size(); ++i) {
                if (best[i] > best_len) {
                    best_len = best[i];
                    state_id = static_cast<int>(i);
                }
            }
            if (best_len == 0) {
                return "";
            }
            int end_pos = st[state_id].first_pos;
            return source.substr(end_pos - best_len + 1, best_len);
        }
    };

    static pair<uint64_t, uint64_t> advance_hash(pair<uint64_t, uint64_t> h, unsigned char ch) {
        h.first = h.first * BASE1 + static_cast<uint64_t>(ch) + 1;
        h.second = h.second * BASE2 + static_cast<uint64_t>(ch) + 1;
        return h;
    }

    static bool contains_substring(const string& haystack, const string& needle) {
        return haystack.find(needle) != string::npos;
    }

    pair<int, size_t> shortest_index_and_length() const {
        int idx = 0;
        size_t len = strings[0].size();
        for (int i = 1; i < static_cast<int>(strings.size()); ++i) {
            if (strings[i].size() < len) {
                len = strings[i].size();
                idx = i;
            }
        }
        return {idx, len};
    }

    bool should_use_sam(size_t min_len, size_t total_chars) const {
        if (strings.size() <= 1) {
            return true;
        }
        if (min_len > 1'000'000) {
            return false;
        }
        if (total_chars > 4'000'000) {
            return false;
        }
        return true;
    }

    vector<uint64_t> build_powers(uint64_t base, size_t max_len) const {
        vector<uint64_t> p(max_len + 1, 1);
        for (size_t i = 1; i <= max_len; ++i) {
            p[i] = p[i - 1] * base;
        }
        return p;
    }

    RollingHashWitness exists_common_len(
        int len,
        const vector<uint64_t>& pow1,
        const vector<uint64_t>& pow2,
        bool capture_positions
    ) const {
        RollingHashWitness out;
        if (len == 0) {
            out.exists = true;
            out.positions = {0};
            return out;
        }

        int base_idx = shortest_index_and_length().first;
        const string& base = strings[base_idx];
        if (len > static_cast<int>(base.size())) {
            return out;
        }

        unordered_map<HashKey, int, HashKeyHasher> candidates;
        candidates.reserve(base.size() - len + 1);

        pair<uint64_t, uint64_t> h = {0, 0};
        for (int i = 0; i < len; ++i) {
            h = advance_hash(h, static_cast<unsigned char>(base[i]));
        }
        candidates.emplace(HashKey{h.first, h.second}, 0);

        for (int i = len; i < static_cast<int>(base.size()); ++i) {
            unsigned char out_ch = static_cast<unsigned char>(base[i - len]);
            unsigned char in_ch = static_cast<unsigned char>(base[i]);
            h.first = h.first * BASE1 + static_cast<uint64_t>(in_ch) + 1 - pow1[len] * (static_cast<uint64_t>(out_ch) + 1);
            h.second = h.second * BASE2 + static_cast<uint64_t>(in_ch) + 1 - pow2[len] * (static_cast<uint64_t>(out_ch) + 1);
            candidates.emplace(HashKey{h.first, h.second}, i - len + 1);
        }

        for (int sidx = 0; sidx < static_cast<int>(strings.size()); ++sidx) {
            if (sidx == base_idx) {
                continue;
            }
            const string& s = strings[sidx];
            if (len > static_cast<int>(s.size())) {
                return out;
            }

            unordered_set<HashKey, HashKeyHasher> seen;
            seen.reserve(candidates.size());

            pair<uint64_t, uint64_t> cur = {0, 0};
            for (int i = 0; i < len; ++i) {
                cur = advance_hash(cur, static_cast<unsigned char>(s[i]));
            }
            HashKey key{cur.first, cur.second};
            if (candidates.find(key) != candidates.end()) {
                seen.insert(key);
            }

            for (int i = len; i < static_cast<int>(s.size()); ++i) {
                unsigned char out_ch = static_cast<unsigned char>(s[i - len]);
                unsigned char in_ch = static_cast<unsigned char>(s[i]);
                cur.first = cur.first * BASE1 + static_cast<uint64_t>(in_ch) + 1 - pow1[len] * (static_cast<uint64_t>(out_ch) + 1);
                cur.second = cur.second * BASE2 + static_cast<uint64_t>(in_ch) + 1 - pow2[len] * (static_cast<uint64_t>(out_ch) + 1);
                key = HashKey{cur.first, cur.second};
                if (candidates.find(key) != candidates.end()) {
                    seen.insert(key);
                }
            }

            if (seen.empty()) {
                return out;
            }

            for (auto it = candidates.begin(); it != candidates.end(); ) {
                if (seen.find(it->first) == seen.end()) {
                    it = candidates.erase(it);
                } else {
                    ++it;
                }
            }
            if (candidates.empty()) {
                return out;
            }
        }

        out.exists = !candidates.empty();
        if (capture_positions && out.exists) {
            out.positions.reserve(candidates.size());
            for (const auto& kv : candidates) {
                out.positions.push_back(kv.second);
            }
        }
        return out;
    }

    string solve_with_sam() const {
        auto [base_idx, _] = shortest_index_and_length();
        SAMSolver sam(strings[base_idx].size());
        sam.build(strings[base_idx]);
        for (int i = 0; i < static_cast<int>(strings.size()); ++i) {
            if (i == base_idx) {
                continue;
            }
            sam.intersect_with(strings[i]);
        }
        return sam.answer_from(strings[base_idx]);
    }

    string solve_with_hash() const {
        auto [base_idx, min_len] = shortest_index_and_length();
        vector<uint64_t> pow1 = build_powers(BASE1, min_len);
        vector<uint64_t> pow2 = build_powers(BASE2, min_len);

        int lo = 0;
        int hi = static_cast<int>(min_len);
        while (lo < hi) {
            int mid = lo + (hi - lo + 1) / 2;
            if (exists_common_len(mid, pow1, pow2, false).exists) {
                lo = mid;
            } else {
                hi = mid - 1;
            }
        }

        if (lo == 0) {
            return "";
        }

        RollingHashWitness witness = exists_common_len(lo, pow1, pow2, true);
        const string& base = strings[base_idx];
        for (int pos : witness.positions) {
            string candidate = base.substr(pos, lo);
            bool ok = true;
            for (const string& s : strings) {
                if (!contains_substring(s, candidate)) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return candidate;
            }
        }

        return "";
    }

    string solve() const {
        if (strings.empty()) {
            return "";
        }
        if (strings.size() == 1) {
            return strings[0];
        }

        size_t total_chars = 0;
        for (const string& s : strings) {
            total_chars += s.size();
        }
        auto [_, min_len] = shortest_index_and_length();

        if (should_use_sam(min_len, total_chars)) {
            return solve_with_sam();
        }
        return solve_with_hash();
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    if (!(cin >> n)) {
        return 0;
    }
    cin.ignore(numeric_limits<streamsize>::max(), '\n');

    Solver solver;
    solver.strings.resize(max(n, 0));
    for (int i = 0; i < n; ++i) {
        getline(cin, solver.strings[i]);
    }

    cout << solver.solve() << '\n';
    return 0;
}
