use std::io::{self, BufRead, Write};

const UP: u8 = 0;
const DOWN: u8 = 1;
const LEFT: u8 = 2;
const RIGHT: u8 = 3;

fn bit_get(bits: &[u64], idx: u32) -> bool {
    let i = idx as usize;
    ((bits[i >> 6] >> (i & 63)) & 1) != 0
}

fn bit_set(bits: &mut [u64], idx: u32) {
    let i = idx as usize;
    bits[i >> 6] |= 1u64 << (i & 63);
}

fn dir_set(dirs: &mut [u64], idx: u32, dir: u8) {
    let i = idx as usize;
    let word = i >> 5;
    let shift = ((i & 31) << 1) as u32;
    let mask = 0b11u64 << shift;
    dirs[word] = (dirs[word] & !mask) | ((dir as u64) << shift);
}

fn dir_get(dirs: &[u64], idx: u32) -> u8 {
    let i = idx as usize;
    let word = i >> 5;
    let shift = ((i & 31) << 1) as u32;
    ((dirs[word] >> shift) & 0b11) as u8
}

fn step(idx: u32, width: u32, dir: u8) -> u32 {
    match dir {
        UP => idx - width,
        DOWN => idx + width,
        LEFT => idx - 1,
        RIGHT => idx + 1,
        _ => unreachable!(),
    }
}

fn expand_frontier(
    current: &mut Vec<u32>,
    next: &mut Vec<u32>,
    visited_this: &mut [u64],
    parent_this: &mut [u64],
    visited_other: &[u64],
    walls: &[u64],
    width: u32,
    total: u32,
) -> Option<u32> {
    next.clear();

    for &idx in current.iter() {
        let col = idx % width;

        if idx >= width {
            let nbr = idx - width;
            if !bit_get(walls, nbr) && !bit_get(visited_this, nbr) {
                bit_set(visited_this, nbr);
                dir_set(parent_this, nbr, DOWN);
                if bit_get(visited_other, nbr) {
                    return Some(nbr);
                }
                next.push(nbr);
            }
        }

        if idx + width < total {
            let nbr = idx + width;
            if !bit_get(walls, nbr) && !bit_get(visited_this, nbr) {
                bit_set(visited_this, nbr);
                dir_set(parent_this, nbr, UP);
                if bit_get(visited_other, nbr) {
                    return Some(nbr);
                }
                next.push(nbr);
            }
        }

        if col > 0 {
            let nbr = idx - 1;
            if !bit_get(walls, nbr) && !bit_get(visited_this, nbr) {
                bit_set(visited_this, nbr);
                dir_set(parent_this, nbr, RIGHT);
                if bit_get(visited_other, nbr) {
                    return Some(nbr);
                }
                next.push(nbr);
            }
        }

        if col + 1 < width {
            let nbr = idx + 1;
            if !bit_get(walls, nbr) && !bit_get(visited_this, nbr) {
                bit_set(visited_this, nbr);
                dir_set(parent_this, nbr, LEFT);
                if bit_get(visited_other, nbr) {
                    return Some(nbr);
                }
                next.push(nbr);
            }
        }
    }

    std::mem::swap(current, next);
    None
}

fn build_path(
    meet: u32,
    start: u32,
    goal: u32,
    parent_from_start: &[u64],
    parent_from_goal: &[u64],
    width: u32,
) -> Vec<u32> {
    let mut left = Vec::new();
    let mut cur = meet;
    left.push(cur);
    while cur != start {
        cur = step(cur, width, dir_get(parent_from_start, cur));
        left.push(cur);
    }
    left.reverse();

    let mut path = left;
    cur = meet;
    while cur != goal {
        cur = step(cur, width, dir_get(parent_from_goal, cur));
        path.push(cur);
    }
    path
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = io::BufReader::new(stdin.lock());

    let mut first_line = String::new();
    if reader.read_line(&mut first_line)? == 0 {
        return Ok(());
    }

    let mut dims = first_line.split_whitespace();
    let height: u32 = dims.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let width: u32 = dims.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    if height == 0 || width == 0 {
        println!("0");
        return Ok(());
    }

    let total_u64 = height as u64 * width as u64;
    if total_u64 > u32::MAX as u64 {
        println!("0");
        return Ok(());
    }
    let total = total_u64 as u32;

    let bit_words = (total as usize + 63) >> 6;
    let dir_words = (total as usize + 31) >> 5;

    let mut walls = vec![0u64; bit_words];
    let mut start = None;
    let mut goal = None;

    let mut line = String::new();
    for r in 0..height {
        line.clear();
        reader.read_line(&mut line)?;
        while line.ends_with('\n') || line.ends_with('\r') {
            line.pop();
        }

        let bytes = line.as_bytes();
        for c in 0..width as usize {
            let ch = bytes.get(c).copied().unwrap_or(b' ');
            let idx = r * width + c as u32;
            match ch {
                b'#' => bit_set(&mut walls, idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    let Some(start) = start else {
        println!("0");
        return Ok(());
    };
    let Some(goal) = goal else {
        println!("0");
        return Ok(());
    };

    if start == goal {
        let x = start % width;
        let y = start / width;
        println!("1");
        println!("{x} {y}");
        return Ok(());
    }

    let mut visited_start = vec![0u64; bit_words];
    let mut visited_goal = vec![0u64; bit_words];
    let mut parent_start = vec![0u64; dir_words];
    let mut parent_goal = vec![0u64; dir_words];

    bit_set(&mut visited_start, start);
    bit_set(&mut visited_goal, goal);

    let mut start_frontier = vec![start];
    let mut goal_frontier = vec![goal];
    let mut start_next = Vec::new();
    let mut goal_next = Vec::new();

    let meet = loop {
        if start_frontier.is_empty() || goal_frontier.is_empty() {
            println!("0");
            return Ok(());
        }

        let found = if start_frontier.len() <= goal_frontier.len() {
            expand_frontier(
                &mut start_frontier,
                &mut start_next,
                &mut visited_start,
                &mut parent_start,
                &visited_goal,
                &walls,
                width,
                total,
            )
        } else {
            expand_frontier(
                &mut goal_frontier,
                &mut goal_next,
                &mut visited_goal,
                &mut parent_goal,
                &visited_start,
                &walls,
                width,
                total,
            )
        };

        if let Some(cell) = found {
            break cell;
        }
    };

    let path = build_path(meet, start, goal, &parent_start, &parent_goal, width);

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    writeln!(out, "{}", path.len())?;
    for idx in path {
        let x = idx % width;
        let y = idx / width;
        writeln!(out, "{x} {y}")?;
    }
    Ok(())
}
