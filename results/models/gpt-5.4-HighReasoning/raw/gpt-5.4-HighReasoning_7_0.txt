struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& o) {
        x += o.x; y += o.y; z += o.z; return *this;
    }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double length2(const Vec3& v) {
    return dot(v, v);
}

struct Tri {
    int a = 0;
    int b = 0;
    int c = 0;
};

struct AABB {
    Vec3 lo{
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity()
    };
    Vec3 hi{
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity()
    };

    void expand(const Vec3& p) {
        lo.x = std::min(lo.x, p.x);
        lo.y = std::min(lo.y, p.y);
        lo.z = std::min(lo.z, p.z);
        hi.x = std::max(hi.x, p.x);
        hi.y = std::max(hi.y, p.y);
        hi.z = std::max(hi.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.lo);
        expand(b.hi);
    }

    Vec3 extent() const { return hi - lo; }
};

static double pointAABBDistance2(const Vec3& p, const AABB& b) {
    auto axis = [](double v, double lo, double hi) {
        if (v < lo) return lo - v;
        if (v > hi) return v - hi;
        return 0.0;
    };
    const double dx = axis(p.x, b.lo.x, b.hi.x);
    const double dy = axis(p.y, b.lo.y, b.hi.y);
    const double dz = axis(p.z, b.lo.z, b.hi.z);
    return dx * dx + dy * dy + dz * dz;
}

static double pointTriangleDistance2(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    const Vec3 ab = b - a;
    const Vec3 ac = c - a;
    const Vec3 ap = p - a;
    const double d1 = dot(ab, ap);
    const double d2 = dot(ac, ap);
    if (d1 <= 0.0 && d2 <= 0.0) return length2(ap);

    const Vec3 bp = p - b;
    const double d3 = dot(ab, bp);
    const double d4 = dot(ac, bp);
    if (d3 >= 0.0 && d4 <= d3) return length2(bp);

    const double vc = d1 * d4 - d3 * d2;
    if (vc <= 0.0 && d1 >= 0.0 && d3 <= 0.0) {
        const double v = d1 / (d1 - d3);
        const Vec3 proj = a + ab * v;
        return length2(p - proj);
    }

    const Vec3 cp = p - c;
    const double d5 = dot(ab, cp);
    const double d6 = dot(ac, cp);
    if (d6 >= 0.0 && d5 <= d6) return length2(cp);

    const double vb = d5 * d2 - d1 * d6;
    if (vb <= 0.0 && d2 >= 0.0 && d6 <= 0.0) {
        const double w = d2 / (d2 - d6);
        const Vec3 proj = a + ac * w;
        return length2(p - proj);
    }

    const double va = d3 * d6 - d5 * d4;
    if (va <= 0.0 && (d4 - d3) >= 0.0 && (d5 - d6) >= 0.0) {
        const Vec3 bc = c - b;
        const double w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
        const Vec3 proj = b + bc * w;
        return length2(p - proj);
    }

    const Vec3 n = cross(ab, ac);
    const double inv = 1.0 / std::sqrt(length2(n));
    const double dist = std::abs(dot(p - a, n)) * inv;
    return dist * dist;
}

struct MeshBVH {
    struct Node {
        AABB box;
        int left = -1;
        int right = -1;
        int start = 0;
        int count = 0;
        bool leaf = false;
    };

    const std::vector<Vec3>* vertices = nullptr;
    const std::vector<Tri>* tris = nullptr;
    std::vector<int> order;
    std::vector<AABB> triBoxes;
    std::vector<Vec3> triCenters;
    std::vector<Node> nodes;

    explicit MeshBVH(const std::vector<Vec3>& v, const std::vector<Tri>& t) : vertices(&v), tris(&t) {
        const int n = static_cast<int>(t.size());
        order.resize(n);
        std::iota(order.begin(), order.end(), 0);
        triBoxes.resize(n);
        triCenters.resize(n);
        for (int i = 0; i < n; ++i) {
            const Tri& tr = (*tris)[i];
            AABB b;
            b.expand((*vertices)[tr.a]);
            b.expand((*vertices)[tr.b]);
            b.expand((*vertices)[tr.c]);
            triBoxes[i] = b;
            triCenters[i] = (b.lo + b.hi) * 0.5;
        }
        nodes.reserve(std::max(1, n * 2));
        if (n > 0) {
            build(0, n);
        }
    }

    int build(int start, int count) {
        Node node;
        node.start = start;
        node.count = count;
        for (int i = 0; i < count; ++i) {
            node.box.expand(triBoxes[order[start + i]]);
        }
        const int idx = static_cast<int>(nodes.size());
        nodes.push_back(node);
        if (count <= 8) {
            nodes[idx].leaf = true;
            return idx;
        }
        AABB cb;
        for (int i = 0; i < count; ++i) {
            cb.expand(triCenters[order[start + i]]);
        }
        const Vec3 ext = cb.extent();
        int axis = 0;
        if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;

        const int mid = start + count / 2;
        auto coord = [&](int triIndex) {
            const Vec3& c = triCenters[triIndex];
            return axis == 0 ? c.x : (axis == 1 ? c.y : c.z);
        };
        std::nth_element(order.begin() + start, order.begin() + mid, order.begin() + start + count,
            [&](int lhs, int rhs) { return coord(lhs) < coord(rhs); });

        nodes[idx].left = build(start, count / 2);
        nodes[idx].right = build(mid, count - count / 2);
        return idx;
    }

    double nearestDistance2(const Vec3& p) const {
        if (nodes.empty()) return std::numeric_limits<double>::infinity();
        double best = std::numeric_limits<double>::infinity();
        std::vector<int> stack;
        stack.push_back(0);
        while (!stack.empty()) {
            const int ni = stack.back();
            stack.pop_back();
            const Node& node = nodes[ni];
            if (pointAABBDistance2(p, node.box) >= best) continue;
            if (node.leaf) {
                for (int i = 0; i < node.count; ++i) {
                    const Tri& tr = (*tris)[order[node.start + i]];
                    best = std::min(best, pointTriangleDistance2(p, (*vertices)[tr.a], (*vertices)[tr.b], (*vertices)[tr.c]));
                }
            } else {
                const double dl = pointAABBDistance2(p, nodes[node.left].box);
                const double dr = pointAABBDistance2(p, nodes[node.right].box);
                if (dl < dr) {
                    if (node.right >= 0) stack.push_back(node.right);
                    if (node.left >= 0) stack.push_back(node.left);
                } else {
                    if (node.left >= 0) stack.push_back(node.left);
                    if (node.right >= 0) stack.push_back(node.right);
                }
            }
        }
        return best;
    }

    static bool rayIntersectsAABB(const Vec3& o, const AABB& b) {
        const double inf = std::numeric_limits<double>::infinity();
        double tmin = -inf;
        double tmax = inf;

        auto slab = [&](double origin, double dir, double mn, double mx) -> bool {
            if (std::abs(dir) < 1e-15) {
                return origin >= mn && origin <= mx;
            }
            const double inv = 1.0 / dir;
            double t1 = (mn - origin) * inv;
            double t2 = (mx - origin) * inv;
            if (t1 > t2) std::swap(t1, t2);
            tmin = std::max(tmin, t1);
            tmax = std::min(tmax, t2);
            return tmax >= tmin;
        };

        if (!slab(o.x, 1.0, b.lo.x, b.hi.x)) return false;
        if (!slab(o.y, 0.0, b.lo.y, b.hi.y)) return false;
        if (!slab(o.z, 0.0, b.lo.z, b.hi.z)) return false;
        return tmax >= std::max(0.0, tmin);
    }

    static bool rayHitsTrianglePositiveX(const Vec3& o, const Vec3& a, const Vec3& b, const Vec3& c) {
        const Vec3 dir{1.0, 0.0, 0.0};
        const Vec3 e1 = b - a;
        const Vec3 e2 = c - a;
        const Vec3 p = cross(dir, e2);
        const double det = dot(e1, p);
        if (std::abs(det) < 1e-14) return false;
        const double invDet = 1.0 / det;
        const Vec3 t = o - a;
        const double u = dot(t, p) * invDet;
        if (u < 0.0 || u > 1.0) return false;
        const Vec3 q = cross(t, e1);
        const double v = dot(dir, q) * invDet;
        if (v < 0.0 || u + v > 1.0) return false;
        const double tt = dot(e2, q) * invDet;
        return tt > 1e-12;
    }

    bool contains(const Vec3& p) const {
        if (nodes.empty()) return false;
        Vec3 o = p;
        const AABB& root = nodes[0].box;
        const Vec3 ext = root.extent();
        const double jitter = 1e-9 * (std::max({ext.x, ext.y, ext.z, 1.0}));
        o.y += jitter * 0.6180339887498948;
        o.z += jitter * 0.3819660112501052;

        int hits = 0;
        std::vector<int> stack;
        stack.push_back(0);
        while (!stack.empty()) {
            const int ni = stack.back();
            stack.pop_back();
            const Node& node = nodes[ni];
            if (!rayIntersectsAABB(o, node.box)) continue;
            if (node.leaf) {
                for (int i = 0; i < node.count; ++i) {
                    const Tri& tr = (*tris)[order[node.start + i]];
                    if (rayHitsTrianglePositiveX(o, (*vertices)[tr.a], (*vertices)[tr.b], (*vertices)[tr.c])) {
                        ++hits;
                    }
                }
            } else {
                if (node.left >= 0) stack.push_back(node.left);
                if (node.right >= 0) stack.push_back(node.right);
            }
        }
        return (hits & 1) != 0;
    }

    double signedDistance(const Vec3& p) const {
        const double d = std::sqrt(nearestDistance2(p));
        return contains(p) ? -d : d;
    }
};

struct EdgeKey {
    std::uint64_t a = 0;
    std::uint64_t b = 0;

    bool operator==(const EdgeKey& o) const {
        return a == o.a && b == o.b;
    }
};

struct EdgeKeyHash {
    std::size_t operator()(const EdgeKey& k) const noexcept {
        const std::uint64_t x = k.a * 0x9E3779B185EBCA87ull;
        const std::uint64_t y = k.b * 0xC2B2AE3D27D4EB4Full;
        return static_cast<std::size_t>(x ^ (y >> 1));
    }
};

struct GridCorner {
    Vec3 p;
    double v = 0.0;
    std::uint64_t gid = 0;
};

struct UnionField {
    const MeshBVH& a;
    const MeshBVH& b;

    double operator()(const Vec3& p) const {
        return std::min(a.signedDistance(p), b.signedDistance(p));
    }
};

static int chooseResolution(std::size_t totalFaces, const Vec3& extent) {
    const double longest = std::max({extent.x, extent.y, extent.z, 1e-9});
    const double shapeBias = (extent.x + extent.y + extent.z) / longest;
    double base = 28.0 + std::pow(static_cast<double>(std::max<std::size_t>(1, totalFaces)), 1.0 / 3.0) * 2.4;
    base *= std::clamp(shapeBias / 2.0, 0.8, 1.3);
    int r = static_cast<int>(std::round(base));
    r = std::clamp(r, 24, 96);
    return r;
}

static std::uint64_t gridId(int x, int y, int z, int nx, int ny) {
    return static_cast<std::uint64_t>((static_cast<std::uint64_t>(z) * ny + y) * nx + x);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return 0;
    }

    std::vector<Vec3> vertsA(VA), vertsB(VB);
    std::vector<Tri> trisA(FA), trisB(FB);
    for (int i = 0; i < VA; ++i) std::cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    for (int i = 0; i < FA; ++i) std::cin >> trisA[i].a >> trisA[i].b >> trisA[i].c;
    for (int i = 0; i < VB; ++i) std::cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    for (int i = 0; i < FB; ++i) std::cin >> trisB[i].a >> trisB[i].b >> trisB[i].c;

    AABB box;
    for (const auto& v : vertsA) box.expand(v);
    for (const auto& v : vertsB) box.expand(v);
    Vec3 ext = box.extent();
    const double longest = std::max({ext.x, ext.y, ext.z, 1.0});
    const double padding = longest * 0.05 + 1e-6;
    box.lo = box.lo - Vec3{padding, padding, padding};
    box.hi = box.hi + Vec3{padding, padding, padding};
    ext = box.extent();

    MeshBVH bvhA(vertsA, trisA);
    MeshBVH bvhB(vertsB, trisB);
    UnionField field{bvhA, bvhB};

    const int baseRes = chooseResolution(static_cast<std::size_t>(FA) + static_cast<std::size_t>(FB), ext);
    const double longestAxis = std::max({ext.x, ext.y, ext.z});
    const int nx = std::max(2, static_cast<int>(std::round(baseRes * ext.x / longestAxis))) + 1;
    const int ny = std::max(2, static_cast<int>(std::round(baseRes * ext.y / longestAxis))) + 1;
    const int nz = std::max(2, static_cast<int>(std::round(baseRes * ext.z / longestAxis))) + 1;

    const Vec3 step{
        ext.x / static_cast<double>(nx - 1),
        ext.y / static_cast<double>(ny - 1),
        ext.z / static_cast<double>(nz - 1)
    };

    const std::size_t sampleCount = static_cast<std::size_t>(nx) * ny * nz;
    std::vector<double> samples(sampleCount);

    auto sampleAt = [&](int x, int y, int z) -> double& {
        return samples[(static_cast<std::size_t>(z) * ny + y) * nx + x];
    };

    for (int z = 0; z < nz; ++z) {
        for (int y = 0; y < ny; ++y) {
            for (int x = 0; x < nx; ++x) {
                const Vec3 p{
                    box.lo.x + step.x * x,
                    box.lo.y + step.y * y,
                    box.lo.z + step.z * z
                };
                sampleAt(x, y, z) = field(p);
            }
        }
    }

    std::vector<Vec3> outVerts;
    std::vector<std::array<int, 3>> outFaces;
    outVerts.reserve(sampleCount / 4);
    outFaces.reserve(sampleCount / 2);
    std::unordered_map<EdgeKey, int, EdgeKeyHash> edgeVertex;
    edgeVertex.reserve(sampleCount);

    auto getCorner = [&](int x, int y, int z) {
        GridCorner c;
        c.p = {
            box.lo.x + step.x * x,
            box.lo.y + step.y * y,
            box.lo.z + step.z * z
        };
        c.v = sampleAt(x, y, z);
        c.gid = gridId(x, y, z, nx, ny);
        return c;
    };

    auto getEdgeVertex = [&](const GridCorner& a, const GridCorner& b) -> int {
        EdgeKey key{std::min(a.gid, b.gid), std::max(a.gid, b.gid)};
        auto it = edgeVertex.find(key);
        if (it != edgeVertex.end()) return it->second;

        const double denom = (a.v - b.v);
        double t = std::abs(denom) < 1e-15 ? 0.5 : a.v / (a.v - b.v);
        t = std::clamp(t, 0.0, 1.0);
        const Vec3 p = a.p + (b.p - a.p) * t;
        const int idx = static_cast<int>(outVerts.size());
        outVerts.push_back(p);
        edgeVertex.emplace(key, idx);
        return idx;
    };

    auto emitTri = [&](int i0, int i1, int i2) {
        if (i0 == i1 || i1 == i2 || i0 == i2) return;
        const Vec3& a = outVerts[i0];
        const Vec3& b = outVerts[i1];
        const Vec3& c = outVerts[i2];
        if (length2(cross(b - a, c - a)) < 1e-24) return;
        outFaces.push_back({i0, i1, i2});
    };

    auto polygoniseTet = [&](const std::array<GridCorner, 4>& tv) {
        int inside[4];
        int outside[4];
        int ic = 0;
        int oc = 0;
        for (int i = 0; i < 4; ++i) {
            if (tv[i].v <= 0.0) inside[ic++] = i;
            else outside[oc++] = i;
        }
        if (ic == 0 || ic == 4) return;

        if (ic == 1) {
            const int i = inside[0];
            const int a = getEdgeVertex(tv[i], tv[outside[0]]);
            const int b = getEdgeVertex(tv[i], tv[outside[1]]);
            const int c = getEdgeVertex(tv[i], tv[outside[2]]);
            emitTri(a, b, c);
            return;
        }
        if (ic == 3) {
            const int o = outside[0];
            const int a = getEdgeVertex(tv[o], tv[inside[0]]);
            const int b = getEdgeVertex(tv[o], tv[inside[1]]);
            const int c = getEdgeVertex(tv[o], tv[inside[2]]);
            emitTri(a, c, b);
            return;
        }

        const int i0 = inside[0];
        const int i1 = inside[1];
        const int o0 = outside[0];
        const int o1 = outside[1];
        const int a = getEdgeVertex(tv[i0], tv[o0]);
        const int b = getEdgeVertex(tv[i0], tv[o1]);
        const int c = getEdgeVertex(tv[i1], tv[o0]);
        const int d = getEdgeVertex(tv[i1], tv[o1]);
        emitTri(a, b, c);
        emitTri(b, d, c);
    };

    const int tetIdx[6][4] = {
        {0, 5, 1, 6},
        {0, 1, 2, 6},
        {0, 2, 3, 6},
        {0, 3, 7, 6},
        {0, 7, 4, 6},
        {0, 4, 5, 6}
    };

    for (int z = 0; z < nz - 1; ++z) {
        for (int y = 0; y < ny - 1; ++y) {
            for (int x = 0; x < nx - 1; ++x) {
                std::array<GridCorner, 8> c = {
                    getCorner(x, y, z),
                    getCorner(x + 1, y, z),
                    getCorner(x + 1, y + 1, z),
                    getCorner(x, y + 1, z),
                    getCorner(x, y, z + 1),
                    getCorner(x + 1, y, z + 1),
                    getCorner(x + 1, y + 1, z + 1),
                    getCorner(x, y + 1, z + 1)
                };

                bool anyInside = false;
                bool anyOutside = false;
                for (const auto& cc : c) {
                    anyInside = anyInside || (cc.v <= 0.0);
                    anyOutside = anyOutside || (cc.v > 0.0);
                }
                if (!(anyInside && anyOutside)) continue;

                for (const auto& ids : tetIdx) {
                    std::array<GridCorner, 4> tet = {c[ids[0]], c[ids[1]], c[ids[2]], c[ids[3]]};
                    polygoniseTet(tet);
                }
            }
        }
    }

    std::cout << outVerts.size() << ' ' << outFaces.size() << '\n';
    std::cout << std::setprecision(17);
    for (const auto& v : outVerts) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& f : outFaces) {
        std::cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }
    return 0;
}
