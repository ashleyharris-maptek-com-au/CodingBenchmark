#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <cstdlib>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <utility>
#include <vector>

using namespace std;

class GraphColorer {
public:
    GraphColorer(int n_, int m_, int k_)
        : n(n_), m(m_), k(k_), adj(n_), degree(n_, 0), rng(123456789ULL) {}

    void add_edge(int u, int v) {
        if (u == v) return;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> solve() {
        build_graph_metadata();
        if (n == 0) return {};
        if (k <= 0) return fail_coloring();

        if (contains_clique_lower_bound() > k) {
            return fail_coloring();
        }

        const bool use_exact =
            n <= 220 &&
            m <= 25000 &&
            static_cast<long long>(n) * static_cast<long long>(k) <= 30000;

        if (use_exact) {
            vector<int> exact = solve_exact_dsatur();
            if (!exact.empty()) return exact;
        }

        vector<int> best = solve_heuristic();
        if (is_valid(best)) return best;

        if (n <= 5000) {
            vector<int> fallback = solve_exact_limited();
            if (!fallback.empty()) return fallback;
        }

        return best;
    }

private:
    int n, m, k;
    vector<vector<int>> adj;
    vector<int> degree;
    mt19937_64 rng;

    void build_graph_metadata() {
        for (int i = 0; i < n; ++i) {
            auto &nbrs = adj[i];
            sort(nbrs.begin(), nbrs.end());
            nbrs.erase(unique(nbrs.begin(), nbrs.end()), nbrs.end());
            degree[i] = static_cast<int>(nbrs.size());
        }
    }

    vector<int> fail_coloring() const {
        return vector<int>(n, 0);
    }

    bool is_valid(const vector<int> &color) const {
        if (static_cast<int>(color.size()) != n) return false;
        for (int u = 0; u < n; ++u) {
            if (color[u] < 0 || color[u] >= k) return false;
            for (int v : adj[u]) {
                if (u < v && color[u] == color[v]) return false;
            }
        }
        return true;
    }

    int contains_clique_lower_bound() const {
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            return a < b;
        });

        vector<int> greedy_color(n, -1);
        int used = 0;
        vector<int> mark(n + 1, -1);
        for (int u : order) {
            for (int v : adj[u]) {
                if (greedy_color[v] != -1) mark[greedy_color[v]] = u;
            }
            int c = 0;
            while (mark[c] == u) ++c;
            greedy_color[u] = c;
            used = max(used, c + 1);
        }
        return used;
    }

    vector<int> build_order_static() const {
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            return a < b;
        });
        return order;
    }

    vector<int> greedy_coloring(const vector<int> &order) const {
        vector<int> color(n, -1);
        vector<int> used(k, -1);
        for (int u : order) {
            for (int v : adj[u]) {
                int c = color[v];
                if (c >= 0 && c < k) used[c] = u;
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (used[c] != u) {
                    chosen = c;
                    break;
                }
            }
            if (chosen == -1) {
                int best_c = 0;
                int best_conflicts = numeric_limits<int>::max();
                for (int c = 0; c < k; ++c) {
                    int conflicts = 0;
                    for (int v : adj[u]) {
                        if (color[v] == c) ++conflicts;
                    }
                    if (conflicts < best_conflicts) {
                        best_conflicts = conflicts;
                        best_c = c;
                    }
                }
                chosen = best_c;
            }
            color[u] = chosen;
        }
        return color;
    }

    vector<int> solve_heuristic() {
        vector<int> best;
        long long best_conflicts = numeric_limits<long long>::max();

        vector<int> base_order = build_order_static();
        vector<int> color = greedy_coloring(base_order);
        best = color;
        best_conflicts = count_conflicts(best);
        if (best_conflicts == 0) return best;

        vector<int> order = base_order;
        const int restarts = n < 20000 ? 24 : 10;
        const int repair_rounds = n < 20000 ? 6000 : 2500;

        for (int attempt = 0; attempt < restarts; ++attempt) {
            if (attempt > 0) perturb_order(order, attempt);
            color = greedy_coloring(order);
            repair_coloring(color, repair_rounds);
            long long conflicts = count_conflicts(color);
            if (conflicts < best_conflicts) {
                best_conflicts = conflicts;
                best = color;
                if (best_conflicts == 0) return best;
            }
        }

        return best;
    }

    void perturb_order(vector<int> &order, int attempt) {
        if (attempt % 3 == 1) {
            stable_sort(order.begin(), order.end(), [&](int a, int b) {
                long long sa = static_cast<long long>(degree[a]) * 1024 + static_cast<long long>(rng() % 1024);
                long long sb = static_cast<long long>(degree[b]) * 1024 + static_cast<long long>(rng() % 1024);
                if (sa != sb) return sa > sb;
                return a < b;
            });
        } else if (attempt % 3 == 2) {
            shuffle(order.begin(), order.end(), rng);
            stable_sort(order.begin(), order.end(), [&](int a, int b) {
                if ((degree[a] / 4) != (degree[b] / 4)) return degree[a] > degree[b];
                return a < b;
            });
        } else {
            shuffle(order.begin(), order.end(), rng);
        }
    }

    long long count_conflicts(const vector<int> &color) const {
        long long conflicts = 0;
        for (int u = 0; u < n; ++u) {
            for (int v : adj[u]) {
                if (u < v && color[u] == color[v]) ++conflicts;
            }
        }
        return conflicts;
    }

    void repair_coloring(vector<int> &color, int rounds) {
        vector<int> conflict_vertices;
        conflict_vertices.reserve(n);
        vector<int> color_count(k, 0);

        for (int round = 0; round < rounds; ++round) {
            conflict_vertices.clear();
            for (int u = 0; u < n; ++u) {
                bool bad = false;
                for (int v : adj[u]) {
                    if (color[u] == color[v]) {
                        bad = true;
                        break;
                    }
                }
                if (bad) conflict_vertices.push_back(u);
            }

            if (conflict_vertices.empty()) return;

            int u = conflict_vertices[rng() % conflict_vertices.size()];
            fill(color_count.begin(), color_count.end(), 0);
            for (int v : adj[u]) {
                int c = color[v];
                if (c >= 0 && c < k) ++color_count[c];
            }

            int current = color[u];
            int best_color = current;
            int best_penalty = color_count[current];

            for (int c = 0; c < k; ++c) {
                int penalty = color_count[c];
                if (penalty < best_penalty) {
                    best_penalty = penalty;
                    best_color = c;
                }
            }

            if (best_penalty == 0) {
                color[u] = best_color;
                continue;
            }

            if (!try_bfs_recolor(u, color)) {
                color[u] = best_color;
            }
        }
    }

    bool try_bfs_recolor(int start, vector<int> &color) {
        vector<int> forbidden(k, 0);
        for (int v : adj[start]) forbidden[color[v]] = 1;
        for (int c = 0; c < k; ++c) {
            if (!forbidden[c]) {
                color[start] = c;
                return true;
            }
        }

        vector<int> candidates;
        candidates.reserve(k);
        for (int c = 0; c < k; ++c) candidates.push_back(c);
        shuffle(candidates.begin(), candidates.end(), rng);

        const int max_trials = min(k, 6);
        for (int idx = 0; idx < max_trials; ++idx) {
            int target_color = candidates[idx];
            vector<int> blockers;
            for (int v : adj[start]) {
                if (color[v] == target_color) blockers.push_back(v);
            }

            bool movable = true;
            for (int v : blockers) {
                int alt = find_free_color_except(v, color, target_color, start);
                if (alt == -1) {
                    movable = false;
                    break;
                }
            }

            if (!movable) continue;

            for (int v : blockers) {
                int alt = find_free_color_except(v, color, target_color, start);
                color[v] = alt;
            }
            color[start] = target_color;
            return true;
        }
        return false;
    }

    int find_free_color_except(int u, const vector<int> &color, int banned_color, int ignored_neighbor) const {
        vector<int> used(k, 0);
        for (int v : adj[u]) {
            if (v == ignored_neighbor) continue;
            used[color[v]] = 1;
        }
        for (int c = 0; c < k; ++c) {
            if (c != banned_color && !used[c]) return c;
        }
        return -1;
    }

    vector<int> solve_exact_dsatur() {
        vector<int> color(n, -1);
        vector<int> sat_deg(n, 0);
        vector<vector<char>> seen(n, vector<char>(k, 0));
        int colored = 0;

        function<bool()> dfs = [&]() -> bool {
            if (colored == n) return true;

            int u = -1;
            int best_sat = -1;
            int best_deg = -1;
            for (int i = 0; i < n; ++i) {
                if (color[i] != -1) continue;
                if (sat_deg[i] > best_sat || (sat_deg[i] == best_sat && degree[i] > best_deg)) {
                    best_sat = sat_deg[i];
                    best_deg = degree[i];
                    u = i;
                }
            }

            vector<int> options;
            options.reserve(k);
            for (int c = 0; c < k; ++c) {
                bool ok = true;
                for (int v : adj[u]) {
                    if (color[v] == c) {
                        ok = false;
                        break;
                    }
                }
                if (ok) options.push_back(c);
            }
            if (options.empty()) return false;

            sort(options.begin(), options.end(), [&](int a, int b) {
                int impact_a = 0;
                int impact_b = 0;
                for (int v : adj[u]) {
                    if (color[v] == -1) {
                        impact_a += !seen[v][a];
                        impact_b += !seen[v][b];
                    }
                }
                return impact_a < impact_b;
            });

            for (int c : options) {
                color[u] = c;
                ++colored;
                vector<int> changed;
                changed.reserve(adj[u].size());
                for (int v : adj[u]) {
                    if (color[v] == -1 && !seen[v][c]) {
                        seen[v][c] = 1;
                        ++sat_deg[v];
                        changed.push_back(v);
                    }
                }

                if (dfs()) return true;

                for (int v : changed) {
                    seen[v][c] = 0;
                    --sat_deg[v];
                }
                --colored;
                color[u] = -1;
            }
            return false;
        };

        if (dfs()) return color;
        return {};
    }

    vector<int> solve_exact_limited() {
        const auto start = chrono::steady_clock::now();
        const auto time_limit = chrono::milliseconds(1500);

        vector<int> color(n, -1);
        vector<int> sat_deg(n, 0);
        vector<vector<char>> seen(n, vector<char>(k, 0));
        int colored = 0;

        function<bool()> dfs = [&]() -> bool {
            if (chrono::steady_clock::now() - start > time_limit) return false;
            if (colored == n) return true;

            int u = -1;
            int best_sat = -1;
            int best_deg = -1;
            for (int i = 0; i < n; ++i) {
                if (color[i] != -1) continue;
                if (sat_deg[i] > best_sat || (sat_deg[i] == best_sat && degree[i] > best_deg)) {
                    best_sat = sat_deg[i];
                    best_deg = degree[i];
                    u = i;
                }
            }

            vector<int> options;
            for (int c = 0; c < k; ++c) {
                bool ok = true;
                for (int v : adj[u]) {
                    if (color[v] == c) {
                        ok = false;
                        break;
                    }
                }
                if (ok) options.push_back(c);
            }
            shuffle(options.begin(), options.end(), rng);

            for (int c : options) {
                color[u] = c;
                ++colored;
                vector<int> changed;
                for (int v : adj[u]) {
                    if (color[v] == -1 && !seen[v][c]) {
                        seen[v][c] = 1;
                        ++sat_deg[v];
                        changed.push_back(v);
                    }
                }
                if (dfs()) return true;
                for (int v : changed) {
                    seen[v][c] = 0;
                    --sat_deg[v];
                }
                --colored;
                color[u] = -1;
            }
            return false;
        };

        if (dfs()) return color;
        return {};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    GraphColorer solver(n, m, k);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            solver.add_edge(u, v);
        }
    }

    vector<int> answer = solver.solve();
    if (static_cast<int>(answer.size()) != n) {
        answer.assign(n, 0);
    }

    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        int c = answer[i];
        if (c < 0) c = 0;
        if (k > 0) c %= k;
        cout << c;
    }
    cout << '\n';
    return 0;
}
