using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var outTemp = new List<int>[n];
        var inTemp = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outTemp[i] = new List<int>();
            inTemp[i] = new List<int>();
        }

        bool[] selfLoop = new bool[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            outTemp[u].Add(v);
            inTemp[v].Add(u);
            if (u == v) selfLoop[u] = true;
        }

        int[][] adj = new int[n][];
        int[][] rev = new int[n][];
        for (int i = 0; i < n; i++)
        {
            adj[i] = outTemp[i].ToArray();
            rev[i] = inTemp[i].ToArray();
        }

        var solver = new DirectedFeedbackVertexSetSolver(n, adj, rev, selfLoop);
        var answer = solver.SolveAll();
        answer.Sort();

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(answer.Count);
        if (answer.Count > 0)
        {
            writer.WriteLine(string.Join(" ", answer));
        }
        else
        {
            writer.WriteLine();
        }
    }
}

internal sealed class DirectedFeedbackVertexSetSolver
{
    private readonly int n;
    private readonly int[][] adj;
    private readonly int[][] rev;
    private readonly bool[] selfLoop;

    private readonly int[] subsetStamp;
    private readonly int[] subsetPos;
    private int currentSubsetStamp;

    private readonly int[] visitStamp;
    private int currentVisitStamp;

    private readonly int[] compId;
    private readonly int[] iterIndex;
    private readonly int[] bfsParent;
    private readonly int[] bfsStamp;
    private int currentBfsStamp;

    public DirectedFeedbackVertexSetSolver(int n, int[][] adj, int[][] rev, bool[] selfLoop)
    {
        this.n = n;
        this.adj = adj;
        this.rev = rev;
        this.selfLoop = selfLoop;

        subsetStamp = new int[n];
        subsetPos = new int[n];
        visitStamp = new int[n];
        compId = new int[n];
        iterIndex = new int[n];
        bfsParent = new int[n];
        bfsStamp = new int[n];
    }

    public List<int> SolveAll()
    {
        var all = new List<int>(n);
        for (int i = 0; i < n; i++) all.Add(i);
        return SolveComponent(all);
    }

    private List<int> SolveComponent(List<int> vertices)
    {
        var forced = new List<int>();
        var reduced = Reduce(vertices, forced);
        if (reduced.Count == 0) return forced;

        var sccs = GetNonTrivialSccs(reduced);
        if (sccs.Count == 0) return forced;
        if (sccs.Count > 1)
        {
            foreach (var scc in sccs)
            {
                forced.AddRange(SolveComponent(scc));
            }
            return forced;
        }

        var upper = GreedyUpperBound(sccs[0]);
        var best = new List<int>(forced.Count + upper.Count);
        best.AddRange(forced);
        best.AddRange(upper);
        int bestSize = best.Count;

        var current = new List<int>(forced);
        Search(sccs[0], current, ref best, ref bestSize);
        return best;
    }

    private void Search(List<int> vertices, List<int> current, ref List<int> best, ref int bestSize)
    {
        if (current.Count >= bestSize) return;

        var forced = new List<int>();
        var reduced = Reduce(vertices, forced);
        int baseCount = current.Count + forced.Count;
        if (baseCount >= bestSize) return;

        if (forced.Count > 0) current.AddRange(forced);

        var sccs = GetNonTrivialSccs(reduced);
        if (sccs.Count == 0)
        {
            if (current.Count < bestSize)
            {
                bestSize = current.Count;
                best = new List<int>(current);
            }
            if (forced.Count > 0) current.RemoveRange(current.Count - forced.Count, forced.Count);
            return;
        }

        if (sccs.Count > 1)
        {
            var merged = new List<int>(current);
            int total = merged.Count;
            foreach (var component in sccs)
            {
                var part = SolveComponent(component);
                total += part.Count;
                if (total >= bestSize) break;
                merged.AddRange(part);
            }

            if (total < bestSize)
            {
                bestSize = total;
                best = merged;
            }

            if (forced.Count > 0) current.RemoveRange(current.Count - forced.Count, forced.Count);
            return;
        }

        var scc = sccs[0];
        int lowerBound = baseCount + GreedyCyclePackingLowerBound(scc);
        if (lowerBound >= bestSize)
        {
            if (forced.Count > 0) current.RemoveRange(current.Count - forced.Count, forced.Count);
            return;
        }

        var cycle = FindCycleThroughPivot(scc);
        if (cycle.Count == 0)
        {
            if (current.Count < bestSize)
            {
                bestSize = current.Count;
                best = new List<int>(current);
            }
            if (forced.Count > 0) current.RemoveRange(current.Count - forced.Count, forced.Count);
            return;
        }

        MarkSubset(scc);
        cycle.Sort((a, b) => ScoreVertexInMarkedSubset(b).CompareTo(ScoreVertexInMarkedSubset(a)));

        foreach (int v in cycle)
        {
            if (current.Count + 1 >= bestSize) break;

            var next = new List<int>(scc.Count - 1);
            foreach (int u in scc)
            {
                if (u != v) next.Add(u);
            }

            current.Add(v);
            Search(next, current, ref best, ref bestSize);
            current.RemoveAt(current.Count - 1);
        }

        if (forced.Count > 0) current.RemoveRange(current.Count - forced.Count, forced.Count);
    }

    private List<int> GreedyUpperBound(List<int> vertices)
    {
        var answer = new List<int>();
        var work = new List<int>(vertices);

        while (true)
        {
            var forced = new List<int>();
            work = Reduce(work, forced);
            if (forced.Count > 0) answer.AddRange(forced);
            if (work.Count == 0) break;

            var sccs = GetNonTrivialSccs(work);
            if (sccs.Count == 0) break;
            if (sccs.Count > 1)
            {
                foreach (var component in sccs)
                {
                    answer.AddRange(GreedyUpperBound(component));
                }
                break;
            }

            var scc = sccs[0];
            MarkSubset(scc);
            int bestVertex = scc[0];
            int bestScore = ScoreVertexInMarkedSubset(bestVertex);
            for (int i = 1; i < scc.Count; i++)
            {
                int v = scc[i];
                int score = ScoreVertexInMarkedSubset(v);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestVertex = v;
                }
            }

            answer.Add(bestVertex);
            var next = new List<int>(scc.Count - 1);
            foreach (int v in scc)
            {
                if (v != bestVertex) next.Add(v);
            }
            work = next;
        }

        return answer;
    }

    private int GreedyCyclePackingLowerBound(List<int> vertices)
    {
        var work = new List<int>(vertices);
        int lb = 0;

        while (true)
        {
            var forced = new List<int>();
            work = Reduce(work, forced);
            if (work.Count == 0) return lb;

            var sccs = GetNonTrivialSccs(work);
            if (sccs.Count == 0) return lb;

            var picked = sccs[0];
            var cycle = FindCycleThroughPivot(picked);
            if (cycle.Count == 0) return lb;

            lb++;
            var remove = new HashSet<int>(cycle);
            var next = new List<int>(work.Count - cycle.Count);
            foreach (int v in work)
            {
                if (!remove.Contains(v)) next.Add(v);
            }
            work = next;
        }
    }

    private List<int> Reduce(List<int> vertices, List<int> forced)
    {
        int k = vertices.Count;
        if (k == 0) return vertices;

        int stamp = ++currentSubsetStamp;
        for (int i = 0; i < k; i++)
        {
            int v = vertices[i];
            subsetStamp[v] = stamp;
            subsetPos[v] = i;
        }

        bool[] alive = new bool[k];
        int[] indeg = new int[k];
        int[] outdeg = new int[k];
        var queue = new Queue<int>();

        for (int i = 0; i < k; i++)
        {
            int v = vertices[i];
            alive[i] = true;

            int inCount = 0;
            foreach (int u in rev[v])
            {
                if (subsetStamp[u] == stamp) inCount++;
            }
            indeg[i] = inCount;

            int outCount = 0;
            foreach (int to in adj[v])
            {
                if (subsetStamp[to] == stamp) outCount++;
            }
            outdeg[i] = outCount;

            if (selfLoop[v] || inCount == 0 || outCount == 0)
            {
                queue.Enqueue(i);
            }
        }

        while (queue.Count > 0)
        {
            int idx = queue.Dequeue();
            if (!alive[idx]) continue;

            int v = vertices[idx];
            alive[idx] = false;

            if (selfLoop[v])
            {
                forced.Add(v);
            }

            foreach (int to in adj[v])
            {
                if (subsetStamp[to] != stamp) continue;
                int j = subsetPos[to];
                if (!alive[j]) continue;
                indeg[j]--;
                if (indeg[j] == 0) queue.Enqueue(j);
            }

            foreach (int from in rev[v])
            {
                if (subsetStamp[from] != stamp) continue;
                int j = subsetPos[from];
                if (!alive[j]) continue;
                outdeg[j]--;
                if (outdeg[j] == 0) queue.Enqueue(j);
            }
        }

        var remaining = new List<int>();
        for (int i = 0; i < k; i++)
        {
            if (alive[i]) remaining.Add(vertices[i]);
        }
        return remaining;
    }

    private void MarkSubset(List<int> vertices)
    {
        int stamp = ++currentSubsetStamp;
        for (int i = 0; i < vertices.Count; i++)
        {
            int v = vertices[i];
            subsetStamp[v] = stamp;
            subsetPos[v] = i;
        }
    }

    private int ScoreVertexInMarkedSubset(int v)
    {
        int stamp = currentSubsetStamp;
        int inCount = 0;
        foreach (int u in rev[v])
        {
            if (subsetStamp[u] == stamp) inCount++;
        }

        int outCount = 0;
        foreach (int to in adj[v])
        {
            if (subsetStamp[to] == stamp) outCount++;
        }

        return inCount * outCount + inCount + outCount;
    }

    private List<List<int>> GetNonTrivialSccs(List<int> vertices)
    {
        if (vertices.Count == 0) return new List<List<int>>();

        int stamp = ++currentSubsetStamp;
        for (int i = 0; i < vertices.Count; i++)
        {
            int v = vertices[i];
            subsetStamp[v] = stamp;
            subsetPos[v] = i;
            compId[v] = -1;
        }

        var order = new List<int>(vertices.Count);
        int seenStamp = ++currentVisitStamp;

        foreach (int start in vertices)
        {
            if (visitStamp[start] == seenStamp) continue;

            var stack = new Stack<int>();
            stack.Push(start);
            visitStamp[start] = seenStamp;
            iterIndex[start] = 0;

            while (stack.Count > 0)
            {
                int v = stack.Peek();
                int idx = iterIndex[v];
                var edges = adj[v];

                while (idx < edges.Length)
                {
                    int to = edges[idx++];
                    if (subsetStamp[to] != stamp || visitStamp[to] == seenStamp) continue;
                    iterIndex[v] = idx;
                    stack.Push(to);
                    visitStamp[to] = seenStamp;
                    iterIndex[to] = 0;
                    goto ContinueOuter;
                }

                stack.Pop();
                order.Add(v);
                continue;

            ContinueOuter:
                continue;
            }
        }

        var result = new List<List<int>>();
        int compCounter = 0;

        for (int oi = order.Count - 1; oi >= 0; oi--)
        {
            int start = order[oi];
            if (compId[start] != -1) continue;

            var comp = new List<int>();
            var stack = new Stack<int>();
            stack.Push(start);
            compId[start] = compCounter;

            while (stack.Count > 0)
            {
                int v = stack.Pop();
                comp.Add(v);
                foreach (int from in rev[v])
                {
                    if (subsetStamp[from] != stamp || compId[from] != -1) continue;
                    compId[from] = compCounter;
                    stack.Push(from);
                }
            }

            if (comp.Count > 1)
            {
                result.Add(comp);
            }
            else if (selfLoop[comp[0]])
            {
                result.Add(comp);
            }

            compCounter++;
        }

        return result;
    }

    private List<int> FindCycleThroughPivot(List<int> vertices)
    {
        MarkSubset(vertices);
        int stamp = currentSubsetStamp;

        int pivot = vertices[0];
        int bestScore = ScoreVertexInMarkedSubset(pivot);
        for (int i = 1; i < vertices.Count; i++)
        {
            int v = vertices[i];
            int score = ScoreVertexInMarkedSubset(v);
            if (score > bestScore)
            {
                bestScore = score;
                pivot = v;
            }
        }

        int bfsStampValue = ++currentBfsStamp;
        var queue = new Queue<int>();

        foreach (int to in adj[pivot])
        {
            if (to == pivot || subsetStamp[to] != stamp) continue;
            if (bfsStamp[to] == bfsStampValue) continue;
            bfsStamp[to] = bfsStampValue;
            bfsParent[to] = pivot;
            queue.Enqueue(to);
        }

        int end = -1;
        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            foreach (int to in adj[v])
            {
                if (to == pivot)
                {
                    end = v;
                    queue.Clear();
                    break;
                }
                if (subsetStamp[to] != stamp || bfsStamp[to] == bfsStampValue) continue;
                bfsStamp[to] = bfsStampValue;
                bfsParent[to] = v;
                queue.Enqueue(to);
            }
        }

        if (end == -1)
        {
            return FindAnyCycle(vertices, stamp);
        }

        var reversed = new List<int>();
        int cur = end;
        while (cur != pivot)
        {
            reversed.Add(cur);
            cur = bfsParent[cur];
        }
        reversed.Add(pivot);
        reversed.Reverse();
        return reversed;
    }

    private List<int> FindAnyCycle(List<int> vertices, int stamp)
    {
        int colorStamp = ++currentVisitStamp;
        int activeStamp = ++currentVisitStamp;
        Array.Fill(iterIndex, 0);

        foreach (int start in vertices)
        {
            if (visitStamp[start] == colorStamp) continue;

            var stack = new Stack<int>();
            var parent = new Dictionary<int, int>();
            stack.Push(start);
            visitStamp[start] = colorStamp;
            bfsStamp[start] = activeStamp;
            parent[start] = -1;

            while (stack.Count > 0)
            {
                int v = stack.Peek();
                var edges = adj[v];
                bool advanced = false;

                while (iterIndex[v] < edges.Length)
                {
                    int to = edges[iterIndex[v]++];
                    if (subsetStamp[to] != stamp) continue;

                    if (bfsStamp[to] == activeStamp)
                    {
                        var cycle = new List<int> { to };
                        int cur = v;
                        while (cur != to && cur != -1)
                        {
                            cycle.Add(cur);
                            cur = parent[cur];
                        }
                        cycle.Reverse();
                        return cycle;
                    }

                    if (visitStamp[to] == colorStamp) continue;

                    visitStamp[to] = colorStamp;
                    bfsStamp[to] = activeStamp;
                    parent[to] = v;
                    stack.Push(to);
                    advanced = true;
                    break;
                }

                if (advanced) continue;
                bfsStamp[v] = 0;
                stack.Pop();
            }
        }

        return new List<int>();
    }
}

internal sealed class FastScanner
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int length;
    private int ptr;

    public FastScanner(Stream stream)
    {
        this.stream = stream;
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= 32 && c >= 0);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int value = 0;
        while (c > 32)
        {
            value = value * 10 + (c - '0');
            c = Read();
        }

        return value * sign;
    }

    private int Read()
    {
        if (ptr >= length)
        {
            length = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (length <= 0) return -1;
        }
        return buffer[ptr++];
    }
}
