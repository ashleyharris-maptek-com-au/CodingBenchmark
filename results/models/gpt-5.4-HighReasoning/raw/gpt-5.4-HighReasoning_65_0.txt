type TaskInput = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
};

type Task = {
  id: string;
  duration: number;
  deadline: number;
  weight: number;
  deps: number[];
  succs: number[];
};

type ScheduleEntry = {
  id: string;
  start: number;
  machine: number;
};

declare const require: any;
declare const process: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

class BinaryHeap<T> {
  private readonly items: T[] = [];
  private readonly less: (a: T, b: T) => boolean;

  constructor(less: (a: T, b: T) => boolean) {
    this.less = less;
  }

  size(): number {
    return this.items.length;
  }

  peek(): T | undefined {
    return this.items[0];
  }

  push(value: T): void {
    const a = this.items;
    a.push(value);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }

  pop(): T | undefined {
    const a = this.items;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        let best = i;
        if (l < a.length && this.less(a[l], a[best])) best = l;
        if (r < a.length && this.less(a[r], a[best])) best = r;
        if (best === i) break;
        const tmp = a[i];
        a[i] = a[best];
        a[best] = tmp;
        i = best;
      }
    }
    return top;
  }
}

const rawTasks: TaskInput[] = Array.isArray(data.tasks) ? data.tasks : [];
const machineCount: number = Math.max(1, Math.floor(Number(data.machineCount) || 1));
const n = rawTasks.length;

const idToIndex = new Map<string, number>();
for (let i = 0; i < n; i++) {
  idToIndex.set(String(rawTasks[i].id), i);
}

const tasks: Task[] = new Array(n);
for (let i = 0; i < n; i++) {
  const t = rawTasks[i];
  const depsRaw = Array.isArray(t.deps) ? t.deps : [];
  const deps: number[] = [];
  for (const depId of depsRaw) {
    const idx = idToIndex.get(String(depId));
    if (idx !== undefined && idx !== i) deps.push(idx);
  }
  tasks[i] = {
    id: String(t.id),
    duration: Math.max(0, Math.floor(Number(t.duration) || 0)),
    deadline: Number.isFinite(Number(t.deadline)) ? Math.floor(Number(t.deadline)) : Number.POSITIVE_INFINITY,
    weight: Math.max(0, Number(t.weight) || 0),
    deps,
    succs: [],
  };
}

for (let i = 0; i < n; i++) {
  for (const d of tasks[i].deps) {
    tasks[d].succs.push(i);
  }
}

const indegree = new Array<number>(n);
for (let i = 0; i < n; i++) indegree[i] = tasks[i].deps.length;

const topo: number[] = [];
const topoQueue: number[] = [];
for (let i = 0; i < n; i++) {
  if (indegree[i] === 0) topoQueue.push(i);
}

for (let head = 0; head < topoQueue.length; head++) {
  const u = topoQueue[head];
  topo.push(u);
  for (const v of tasks[u].succs) {
    indegree[v]--;
    if (indegree[v] === 0) topoQueue.push(v);
  }
}

if (topo.length !== n) {
  const remaining: string[] = [];
  const seen = new Array<boolean>(n).fill(false);
  for (const idx of topo) seen[idx] = true;
  for (let i = 0; i < n; i++) {
    if (!seen[i]) remaining.push(tasks[i].id);
  }
  const order = topo.map((i) => tasks[i].id).concat(remaining);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

const earliestStart = new Array<number>(n).fill(0);
for (const u of topo) {
  const finish = earliestStart[u] + tasks[u].duration;
  for (const v of tasks[u].succs) {
    if (finish > earliestStart[v]) earliestStart[v] = finish;
  }
}

const bottomLevel = new Array<number>(n).fill(0);
const effectiveDeadline = new Array<number>(n).fill(Number.POSITIVE_INFINITY);
const fanoutWork = new Array<number>(n).fill(0);

for (let k = n - 1; k >= 0; k--) {
  const u = topo[k];
  let bestTail = 0;
  let bestDeadline = tasks[u].deadline;
  let succWork = 0;
  for (const v of tasks[u].succs) {
    if (bottomLevel[v] > bestTail) bestTail = bottomLevel[v];
    const propagated = effectiveDeadline[v] - tasks[v].duration;
    if (propagated < bestDeadline) bestDeadline = propagated;
    succWork += tasks[v].duration + 0.25 * fanoutWork[v];
  }
  bottomLevel[u] = tasks[u].duration + bestTail;
  effectiveDeadline[u] = bestDeadline;
  fanoutWork[u] = succWork;
}

const chainSlack = new Array<number>(n);
for (let i = 0; i < n; i++) {
  chainSlack[i] = effectiveDeadline[i] - earliestStart[i] - bottomLevel[i];
}

function betterTask(a: number, b: number): boolean {
  const sa = chainSlack[a];
  const sb = chainSlack[b];
  if (sa !== sb) return sa < sb;

  const wa = tasks[a].weight / Math.max(1, tasks[a].duration);
  const wb = tasks[b].weight / Math.max(1, tasks[b].duration);
  if (wa !== wb) return wa > wb;

  if (bottomLevel[a] !== bottomLevel[b]) return bottomLevel[a] > bottomLevel[b];
  if (effectiveDeadline[a] !== effectiveDeadline[b]) return effectiveDeadline[a] < effectiveDeadline[b];
  if (tasks[a].weight !== tasks[b].weight) return tasks[a].weight > tasks[b].weight;
  if (fanoutWork[a] !== fanoutWork[b]) return fanoutWork[a] > fanoutWork[b];
  if (tasks[a].duration !== tasks[b].duration) return tasks[a].duration > tasks[b].duration;
  return tasks[a].id < tasks[b].id;
}

type Completion = { time: number; machine: number; task: number };

const remainingDeps = new Array<number>(n);
for (let i = 0; i < n; i++) remainingDeps[i] = tasks[i].deps.length;

const ready = new BinaryHeap<number>(betterTask);
for (let i = 0; i < n; i++) {
  if (remainingDeps[i] === 0) ready.push(i);
}

const freeMachines = new BinaryHeap<number>((a, b) => a < b);
for (let i = 0; i < machineCount; i++) freeMachines.push(i);

const running = new BinaryHeap<Completion>((a, b) => {
  if (a.time !== b.time) return a.time < b.time;
  if (a.machine !== b.machine) return a.machine < b.machine;
  return a.task < b.task;
});

const schedule: ScheduleEntry[] = [];
let scheduledCount = 0;
let currentTime = 0;

while (scheduledCount < n) {
  while (freeMachines.size() > 0 && ready.size() > 0) {
    const machine = freeMachines.pop()!;
    const taskIndex = ready.pop()!;
    schedule.push({
      id: tasks[taskIndex].id,
      start: currentTime,
      machine,
    });
    running.push({
      time: currentTime + tasks[taskIndex].duration,
      machine,
      task: taskIndex,
    });
    scheduledCount++;
  }

  if (scheduledCount >= n) break;

  const next = running.pop();
  if (!next) break;
  currentTime = next.time;

  const finished: Completion[] = [next];
  while (running.size() > 0 && running.peek()!.time === currentTime) {
    finished.push(running.pop()!);
  }

  for (const item of finished) {
    freeMachines.push(item.machine);
  }

  for (const item of finished) {
    for (const v of tasks[item.task].succs) {
      remainingDeps[v]--;
      if (remainingDeps[v] === 0) ready.push(v);
    }
  }
}

schedule.sort((a, b) => {
  if (a.start !== b.start) return a.start - b.start;
  if (a.machine !== b.machine) return a.machine - b.machine;
  return a.id < b.id ? -1 : a.id > b.id ? 1 : 0;
});

process.stdout.write(JSON.stringify({ schedule }));
