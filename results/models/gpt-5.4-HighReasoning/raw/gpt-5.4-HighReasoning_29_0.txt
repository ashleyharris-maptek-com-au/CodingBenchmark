using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Threading.Tasks;

public static class Program
{
    static readonly Stopwatch Timer = Stopwatch.StartNew();
    const double TotalTimeLimitSeconds = 295.0;
    const double ExactSearchDeadlineSeconds = 285.0;
    const int ExactKernelLimit = 3500;
    const int HeuristicSeedLimit = 96;

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var graph = Graph.Read(fs, n, m);

        if (graph.VertexCount == 1)
        {
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        int[] core = ComputeCoreNumbers(graph, out int degeneracy);

        var best = new BestClique();
        RunHeuristics(graph, core, best);

        if (best.Size == 0)
        {
            best.Update(new[] { 0 }, 1);
        }

        if (Timer.Elapsed.TotalSeconds < ExactSearchDeadlineSeconds)
        {
            int[] kernel = BuildExactKernel(graph, core, best.Size);
            if (kernel.Length > 0 && kernel.Length <= ExactKernelLimit)
            {
                RunExactSearch(graph, kernel, best);
            }
        }

        Array.Sort(best.Vertices, 0, best.Size);
        Console.WriteLine(best.Size);
        for (int i = 0; i < best.Size; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(best.Vertices[i]);
        }
        Console.WriteLine();
    }

    static void RunHeuristics(Graph g, int[] core, BestClique best)
    {
        int n = g.VertexCount;
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;

        Array.Sort(order, (a, b) =>
        {
            int c = core[b].CompareTo(core[a]);
            if (c != 0) return c;
            c = g.Degree(b).CompareTo(g.Degree(a));
            if (c != 0) return c;
            return a.CompareTo(b);
        });

        int seedCount = Math.Min(n, HeuristicSeedLimit);
        var seeds = new int[seedCount];
        Array.Copy(order, seeds, seedCount);

        if (n > seedCount)
        {
            int step = Math.Max(1, n / Math.Max(1, HeuristicSeedLimit / 2));
            var expanded = new List<int>(seedCount + HeuristicSeedLimit / 2);
            for (int i = 0; i < seedCount; i++) expanded.Add(seeds[i]);
            for (int i = step / 2; i < n && expanded.Count < HeuristicSeedLimit + HeuristicSeedLimit / 2; i += step)
            {
                expanded.Add(order[i]);
            }
            seeds = expanded.ToArray();
        }

        Parallel.ForEach(seeds, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, seed =>
        {
            if (OutOfTime()) return;
            GreedyCliqueFromSeed(g, seed, best);
        });

        for (int i = 0; i < Math.Min(16, order.Length); i++)
        {
            if (OutOfTime()) break;
            GreedyNeighborhoodClique(g, order[i], best);
        }
    }

    static void GreedyCliqueFromSeed(Graph g, int seed, BestClique best)
    {
        int bestSnapshot = best.Size;
        int deg = g.Degree(seed);
        if (deg + 1 <= bestSnapshot) return;

        int[] clique = new int[Math.Max(4, bestSnapshot + 8)];
        int cliqueSize = 0;
        clique[cliqueSize++] = seed;

        int[] cand = new int[deg];
        int candCount = 0;
        int start = g.Offsets[seed];
        int end = g.Offsets[seed + 1];
        for (int i = start; i < end; i++)
        {
            int v = g.Adjacency[i];
            if (g.Degree(v) + 1 > bestSnapshot)
            {
                cand[candCount++] = v;
            }
        }

        while (candCount > 0 && !OutOfTime())
        {
            if (cliqueSize + candCount <= best.Size) break;

            int chosen = ChooseCandidate(g, cand, candCount);
            if (cliqueSize == clique.Length) Array.Resize(ref clique, clique.Length * 2);
            clique[cliqueSize++] = chosen;

            int[] next = new int[candCount];
            int nextCount = IntersectWithNeighborhoodExcluding(g, cand, candCount, chosen, next);
            cand = next;
            candCount = nextCount;
        }

        best.Update(clique, cliqueSize);
    }

    static void GreedyNeighborhoodClique(Graph g, int center, BestClique best)
    {
        int start = g.Offsets[center];
        int end = g.Offsets[center + 1];
        int deg = end - start;
        if (deg == 0) return;

        int[] nbrs = new int[deg];
        Array.Copy(g.Adjacency, start, nbrs, 0, deg);
        Array.Sort(nbrs, (a, b) =>
        {
            int c = g.Degree(b).CompareTo(g.Degree(a));
            if (c != 0) return c;
            return a.CompareTo(b);
        });

        int[] clique = new int[Math.Min(deg + 1, best.Size + 32)];
        int size = 0;
        clique[size++] = center;

        for (int i = 0; i < nbrs.Length && !OutOfTime(); i++)
        {
            int v = nbrs[i];
            bool ok = true;
            for (int j = 1; j < size; j++)
            {
                if (!g.AreAdjacent(v, clique[j]))
                {
                    ok = false;
                    break;
                }
            }
            if (ok)
            {
                if (size == clique.Length) Array.Resize(ref clique, clique.Length * 2);
                clique[size++] = v;
            }
        }

        best.Update(clique, size);
    }

    static int ChooseCandidate(Graph g, int[] cand, int count)
    {
        int best = cand[0];
        int bestDeg = g.Degree(best);
        for (int i = 1; i < count; i++)
        {
            int v = cand[i];
            int d = g.Degree(v);
            if (d > bestDeg || (d == bestDeg && v < best))
            {
                best = v;
                bestDeg = d;
            }
        }
        return best;
    }

    static int IntersectWithNeighborhoodExcluding(Graph g, int[] cand, int candCount, int vertex, int[] output)
    {
        int i = 0;
        int start = g.Offsets[vertex];
        int end = g.Offsets[vertex + 1];
        int p = start;
        int outCount = 0;

        while (i < candCount && p < end)
        {
            int a = cand[i];
            int b = g.Adjacency[p];
            if (a == vertex)
            {
                i++;
            }
            else if (a == b)
            {
                output[outCount++] = a;
                i++;
                p++;
            }
            else if (a < b)
            {
                i++;
            }
            else
            {
                p++;
            }
        }

        return outCount;
    }

    static int[] BuildExactKernel(Graph g, int[] core, int lowerBound)
    {
        if (lowerBound <= 1 && g.VertexCount > ExactKernelLimit) return Array.Empty<int>();

        int n = g.VertexCount;
        int[] candidates = new int[n];
        int candidateCount = 0;
        int minCore = Math.Max(0, lowerBound - 1);
        for (int v = 0; v < n; v++)
        {
            if (core[v] >= minCore && g.Degree(v) >= lowerBound - 1)
            {
                candidates[candidateCount++] = v;
            }
        }

        if (candidateCount == 0) return Array.Empty<int>();
        if (candidateCount > ExactKernelLimit * 8) return Array.Empty<int>();

        int[] map = new int[n];
        Array.Fill(map, -1);
        for (int i = 0; i < candidateCount; i++) map[candidates[i]] = i;

        int[] degInCand = new int[candidateCount];
        for (int i = 0; i < candidateCount; i++)
        {
            int v = candidates[i];
            int count = 0;
            for (int p = g.Offsets[v]; p < g.Offsets[v + 1]; p++)
            {
                if (map[g.Adjacency[p]] >= 0) count++;
            }
            degInCand[i] = count;
        }

        bool[] removed = new bool[candidateCount];
        int[] queue = new int[candidateCount];
        int qh = 0, qt = 0;
        int minDegree = lowerBound - 1;
        for (int i = 0; i < candidateCount; i++)
        {
            if (degInCand[i] < minDegree) queue[qt++] = i;
        }

        while (qh < qt)
        {
            int idx = queue[qh++];
            if (removed[idx]) continue;
            removed[idx] = true;
            int v = candidates[idx];
            for (int p = g.Offsets[v]; p < g.Offsets[v + 1]; p++)
            {
                int uIdx = map[g.Adjacency[p]];
                if (uIdx >= 0 && !removed[uIdx])
                {
                    degInCand[uIdx]--;
                    if (degInCand[uIdx] < minDegree)
                    {
                        queue[qt++] = uIdx;
                    }
                }
            }
        }

        int remain = 0;
        for (int i = 0; i < candidateCount; i++) if (!removed[i]) remain++;
        var kernel = new int[remain];
        int at = 0;
        for (int i = 0; i < candidateCount; i++)
        {
            if (!removed[i]) kernel[at++] = candidates[i];
        }
        return kernel;
    }

    static void RunExactSearch(Graph g, int[] kernel, BestClique best)
    {
        int n = kernel.Length;
        int[] map = new int[g.VertexCount];
        Array.Fill(map, -1);
        for (int i = 0; i < n; i++) map[kernel[i]] = i;

        int words = (n + 63) >> 6;
        ulong[][] adjBits = new ulong[n][];
        int[] subDegrees = new int[n];
        for (int i = 0; i < n; i++)
        {
            adjBits[i] = new ulong[words];
        }

        for (int i = 0; i < n; i++)
        {
            int v = kernel[i];
            int d = 0;
            for (int p = g.Offsets[v]; p < g.Offsets[v + 1]; p++)
            {
                int j = map[g.Adjacency[p]];
                if (j >= 0)
                {
                    adjBits[i][j >> 6] |= 1UL << (j & 63);
                    d++;
                }
            }
            subDegrees[i] = d;
        }

        int[] initial = new int[n];
        for (int i = 0; i < n; i++) initial[i] = i;
        Array.Sort(initial, (a, b) =>
        {
            int c = subDegrees[b].CompareTo(subDegrees[a]);
            if (c != 0) return c;
            return a.CompareTo(b);
        });

        var solver = new ExactCliqueSolver(kernel, adjBits, subDegrees, best);
        solver.Solve(initial, n);
    }

    static int[] ComputeCoreNumbers(Graph g, out int maxCore)
    {
        int n = g.VertexCount;
        int[] degree = new int[n];
        int maxDegree = 0;
        for (int v = 0; v < n; v++)
        {
            int d = g.Degree(v);
            degree[v] = d;
            if (d > maxDegree) maxDegree = d;
        }

        int[] bin = new int[maxDegree + 1];
        for (int v = 0; v < n; v++) bin[degree[v]]++;

        int start = 0;
        for (int d = 0; d <= maxDegree; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        int[] pos = new int[n];
        int[] vert = new int[n];
        for (int v = 0; v < n; v++)
        {
            pos[v] = bin[degree[v]];
            vert[pos[v]] = v;
            bin[degree[v]]++;
        }

        for (int d = maxDegree; d >= 1; d--) bin[d] = bin[d - 1];
        bin[0] = 0;

        int[] core = new int[n];
        maxCore = 0;
        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            int dv = degree[v];
            core[v] = dv;
            if (dv > maxCore) maxCore = dv;
            for (int p = g.Offsets[v]; p < g.Offsets[v + 1]; p++)
            {
                int u = g.Adjacency[p];
                if (degree[u] > dv)
                {
                    int du = degree[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];
                    if (u != w)
                    {
                        pos[u] = pw;
                        pos[w] = pu;
                        vert[pu] = w;
                        vert[pw] = u;
                    }
                    bin[du]++;
                    degree[u]--;
                }
            }
        }

        return core;
    }

    static bool OutOfTime()
    {
        return Timer.Elapsed.TotalSeconds >= TotalTimeLimitSeconds;
    }

    sealed class ExactCliqueSolver
    {
        readonly int[] originalVertex;
        readonly ulong[][] adjBits;
        readonly int[] degrees;
        readonly BestClique best;
        readonly int[] currentClique;
        readonly int[][] levelBuffers;
        readonly int[][] levelOrdered;
        readonly int[][] levelColors;

        public ExactCliqueSolver(int[] originalVertex, ulong[][] adjBits, int[] degrees, BestClique best)
        {
            this.originalVertex = originalVertex;
            this.adjBits = adjBits;
            this.degrees = degrees;
            this.best = best;
            currentClique = new int[originalVertex.Length];
            levelBuffers = new int[originalVertex.Length + 1][];
            levelOrdered = new int[originalVertex.Length + 1][];
            levelColors = new int[originalVertex.Length + 1][];
        }

        public void Solve(int[] initial, int count)
        {
            Expand(initial, count, 0);
        }

        void Expand(int[] vertices, int count, int depth)
        {
            if (count == 0)
            {
                Commit(depth);
                return;
            }

            if (OutOfTime()) return;
            if (depth + count <= best.Size) return;

            int[] ordered = GetBuffer(levelOrdered, depth, count);
            int[] colors = GetBuffer(levelColors, depth, count);
            ColorSort(vertices, count, ordered, colors);

            for (int i = count - 1; i >= 0; i--)
            {
                if (depth + colors[i] <= best.Size) return;
                if (OutOfTime()) return;

                int v = ordered[i];
                currentClique[depth] = v;

                int[] next = GetBuffer(levelBuffers, depth, i);
                int nextCount = 0;
                for (int j = 0; j < i; j++)
                {
                    int u = ordered[j];
                    if (IsAdjacent(v, u))
                    {
                        next[nextCount++] = u;
                    }
                }

                if (nextCount == 0)
                {
                    Commit(depth + 1);
                }
                else
                {
                    Expand(next, nextCount, depth + 1);
                }
            }
        }

        void Commit(int size)
        {
            int[] clique = new int[size];
            for (int i = 0; i < size; i++)
            {
                clique[i] = originalVertex[currentClique[i]];
            }
            best.Update(clique, size);
        }

        void ColorSort(int[] vertices, int count, int[] ordered, int[] colors)
        {
            int[] work = new int[count];
            Array.Copy(vertices, work, count);
            Array.Sort(work, 0, count, Comparer<int>.Create((a, b) =>
            {
                int c = degrees[b].CompareTo(degrees[a]);
                if (c != 0) return c;
                return a.CompareTo(b);
            }));

            int outCount = 0;
            int color = 0;
            int[] remaining = work;
            int remainingCount = count;

            while (remainingCount > 0)
            {
                color++;
                int classCount = 0;
                for (int i = 0; i < remainingCount; i++)
                {
                    int v = remaining[i];
                    bool ok = true;
                    for (int j = outCount - classCount; j < outCount; j++)
                    {
                        if (IsAdjacent(v, ordered[j]))
                        {
                            ok = false;
                            break;
                        }
                    }
                    if (ok)
                    {
                        ordered[outCount] = v;
                        colors[outCount] = color;
                        outCount++;
                        classCount++;
                        remaining[i] = -1;
                    }
                }

                if (outCount == count) break;

                int write = 0;
                for (int i = 0; i < remainingCount; i++)
                {
                    int v = remaining[i];
                    if (v >= 0) remaining[write++] = v;
                }
                remainingCount = write;
            }
        }

        bool IsAdjacent(int a, int b)
        {
            return (adjBits[a][b >> 6] & (1UL << (b & 63))) != 0;
        }

        static int[] GetBuffer(int[][] store, int level, int minSize)
        {
            var arr = store[level];
            if (arr == null || arr.Length < minSize)
            {
                arr = new int[Math.Max(4, minSize)];
                store[level] = arr;
            }
            return arr;
        }
    }

    sealed class BestClique
    {
        readonly object sync = new object();
        public int Size;
        public int[] Vertices = Array.Empty<int>();

        public void Update(int[] candidate, int size)
        {
            if (size <= 0) return;
            if (size < Size) return;

            lock (sync)
            {
                if (size < Size) return;
                if (Vertices.Length < size) Vertices = new int[size];
                Array.Copy(candidate, Vertices, size);
                Size = size;
            }
        }
    }

    sealed class Graph
    {
        public readonly int VertexCount;
        public readonly int[] Offsets;
        public readonly int[] Adjacency;

        Graph(int vertexCount, int[] offsets, int[] adjacency)
        {
            VertexCount = vertexCount;
            Offsets = offsets;
            Adjacency = adjacency;
        }

        public static Graph Read(FastScanner fs, int n, int m)
        {
            int[] u = new int[m];
            int[] v = new int[m];
            int[] deg = new int[n];
            int actual = 0;

            for (int i = 0; i < m; i++)
            {
                int a = fs.NextInt();
                int b = fs.NextInt();
                if (a < 0 || b < 0 || a >= n || b >= n || a == b) continue;
                u[actual] = a;
                v[actual] = b;
                deg[a]++;
                deg[b]++;
                actual++;
            }

            int[] offsets = new int[n + 1];
            for (int i = 0; i < n; i++) offsets[i + 1] = offsets[i] + deg[i];
            int[] cursor = new int[n];
            Array.Copy(offsets, cursor, n);
            int[] adj = new int[offsets[n]];

            for (int i = 0; i < actual; i++)
            {
                int a = u[i];
                int b = v[i];
                adj[cursor[a]++] = b;
                adj[cursor[b]++] = a;
            }

            for (int i = 0; i < n; i++)
            {
                int len = offsets[i + 1] - offsets[i];
                if (len > 1) Array.Sort(adj, offsets[i], len);
            }

            return new Graph(n, offsets, adj);
        }

        public int Degree(int v) => Offsets[v + 1] - Offsets[v];

        public bool AreAdjacent(int a, int b)
        {
            int lo = Offsets[a];
            int hi = Offsets[a + 1] - 1;
            while (lo <= hi)
            {
                int mid = lo + ((hi - lo) >> 1);
                int x = Adjacency[mid];
                if (x == b) return true;
                if (x < b) lo = mid + 1;
                else hi = mid - 1;
            }
            return false;
        }
    }

    sealed class FastScanner
    {
        readonly System.IO.Stream input = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 16];
        int len, ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }
    }
}
