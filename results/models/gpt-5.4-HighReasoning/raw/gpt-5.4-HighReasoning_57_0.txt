const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid || [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
const cutBudget = Math.max(0, data.cutBudget | 0);
const n = width * height;

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[y * width + x] = 1; // '.'
  }
}

const townValueAt = new Float64Array(n);
const townCell = new Int32Array(towns.length);
const isTownCell = new Uint8Array(n);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const idx = t.y * width + t.x;
  townCell[i] = idx;
  if (idx >= 0 && idx < n) {
    townValueAt[idx] += Number(t.value) || 0;
    isTownCell[idx] = 1;
  }
}

const ignitionIndices = [];
const isIgnitionCell = new Uint8Array(n);
for (const p of ignitions) {
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const idx = y * width + x;
  if (!burnable[idx]) continue;
  ignitionIndices.push(idx);
  isIgnitionCell[idx] = 1;
}

const blocked = new Uint8Array(n);
const queue = new Int32Array(Math.max(1, n * 2 + 5));
const seen = new Int32Array(Math.max(1, n));
let seenStamp = 1;

function neighbors(idx, fn) {
  const x = idx % width;
  const y = (idx / width) | 0;
  if (x > 0) fn(idx - 1);
  if (x + 1 < width) fn(idx + 1);
  if (y > 0) fn(idx - width);
  if (y + 1 < height) fn(idx + width);
}

function bumpSeenStamp() {
  seenStamp++;
  if (seenStamp === 0x7fffffff) {
    seen.fill(0);
    seenStamp = 1;
  }
}

function computeBurnedValue() {
  bumpSeenStamp();
  let head = 0;
  let tail = 0;
  let burnedValue = 0;
  for (let i = 0; i < ignitionIndices.length; i++) {
    const s = ignitionIndices[i];
    if (blocked[s] || seen[s] === seenStamp) continue;
    seen[s] = seenStamp;
    queue[tail++] = s;
  }
  while (head < tail) {
    const v = queue[head++];
    burnedValue += townValueAt[v];
    neighbors(v, (u) => {
      if (!burnable[u] || blocked[u] || seen[u] === seenStamp) return;
      seen[u] = seenStamp;
      queue[tail++] = u;
    });
  }
  return burnedValue;
}

function simulateDetailed() {
  const dist = new Int32Array(n);
  const parent = new Int32Array(n);
  dist.fill(-1);
  parent.fill(-1);
  const order = [];
  let head = 0;
  let tail = 0;
  let burnedValue = 0;

  for (let i = 0; i < ignitionIndices.length; i++) {
    const s = ignitionIndices[i];
    if (blocked[s] || dist[s] !== -1) continue;
    dist[s] = 0;
    queue[tail++] = s;
  }

  while (head < tail) {
    const v = queue[head++];
    order.push(v);
    burnedValue += townValueAt[v];
    const dv = dist[v] + 1;
    neighbors(v, (u) => {
      if (!burnable[u] || blocked[u] || dist[u] !== -1) return;
      dist[u] = dv;
      parent[u] = v;
      queue[tail++] = u;
    });
  }

  return { dist, parent, order, burnedValue };
}

class Dinic {
  constructor(n) {
    this.n = n;
    this.g = Array.from({ length: n }, () => []);
    this.level = new Int32Array(n);
    this.it = new Int32Array(n);
  }

  addEdge(fr, to, cap) {
    const fwd = { to, rev: this.g[to].length, cap };
    const rev = { to: fr, rev: this.g[fr].length, cap: 0 };
    this.g[fr].push(fwd);
    this.g[to].push(rev);
  }

  bfs(s, t) {
    this.level.fill(-1);
    let head = 0;
    let tail = 0;
    this.level[s] = 0;
    queue[tail++] = s;
    while (head < tail) {
      const v = queue[head++];
      for (const e of this.g[v]) {
        if (e.cap <= 0 || this.level[e.to] !== -1) continue;
        this.level[e.to] = this.level[v] + 1;
        if (e.to === t) return true;
        queue[tail++] = e.to;
      }
    }
    return this.level[t] !== -1;
  }

  dfs(v, t, f) {
    if (v === t) return f;
    for (let i = this.it[v]; i < this.g[v].length; i++, this.it[v]++) {
      const e = this.g[v][i];
      if (e.cap <= 0 || this.level[v] + 1 !== this.level[e.to]) continue;
      const d = this.dfs(e.to, t, Math.min(f, e.cap));
      if (d > 0) {
        e.cap -= d;
        this.g[e.to][e.rev].cap += d;
        return d;
      }
    }
    return 0;
  }

  maxFlow(s, t) {
    let flow = 0;
    const INF = 1e15;
    while (this.bfs(s, t)) {
      this.it.fill(0);
      while (true) {
        const f = this.dfs(s, t, INF);
        if (f <= 0) break;
        flow += f;
      }
    }
    return flow;
  }

  minCutReachable(s) {
    const reach = new Uint8Array(this.n);
    let head = 0;
    let tail = 0;
    reach[s] = 1;
    queue[tail++] = s;
    while (head < tail) {
      const v = queue[head++];
      for (const e of this.g[v]) {
        if (e.cap <= 0 || reach[e.to]) continue;
        reach[e.to] = 1;
        queue[tail++] = e.to;
      }
    }
    return reach;
  }
}

function localMinCutForTown(townIdx, distFire, remainingBudget, radius) {
  const origin = townCell[townIdx];
  if (origin < 0 || origin >= n || !burnable[origin] || blocked[origin]) return null;

  const regionMark = new Int32Array(n);
  let mark = 1;
  const localQueue = new Int32Array(Math.max(16, Math.min(n, (radius + 3) * (radius + 3) * 2)));
  const region = [];
  const distTown = new Map();

  let head = 0;
  let tail = 0;
  regionMark[origin] = mark;
  distTown.set(origin, 0);
  localQueue[tail++] = origin;
  region.push(origin);

  while (head < tail) {
    const v = localQueue[head++];
    const dv = distTown.get(v);
    if (dv >= radius) continue;
    neighbors(v, (u) => {
      if (!burnable[u] || blocked[u] || regionMark[u] === mark) return;
      regionMark[u] = mark;
      distTown.set(u, dv + 1);
      localQueue[tail++] = u;
      region.push(u);
    });
  }

  if (region.length <= 1) return null;

  const localId = new Map();
  for (let i = 0; i < region.length; i++) localId.set(region[i], i);

  const protectedTownCells = new Set([origin]);
  for (let i = 0; i < towns.length; i++) {
    const c = townCell[i];
    const d = distTown.get(c);
    if (d !== undefined && d <= 2) protectedTownCells.add(c);
  }

  const sourceCells = [];
  for (let i = 0; i < region.length; i++) {
    const v = region[i];
    const dv = distTown.get(v);
    let boundary = dv === radius;
    if (!boundary) {
      neighbors(v, (u) => {
        if (boundary) return;
        if (!burnable[u] || blocked[u]) return;
        if (!localId.has(u)) boundary = true;
      });
    }
    if ((boundary && distFire[v] !== -1) || isIgnitionCell[v]) sourceCells.push(v);
  }

  if (sourceCells.length === 0) return null;

  const nodeCount = region.length * 2 + 2;
  const S = nodeCount - 2;
  const T = nodeCount - 1;
  const dinic = new Dinic(nodeCount);
  const INF = 1e9;

  for (let i = 0; i < region.length; i++) {
    const v = region[i];
    const vin = i * 2;
    const vout = vin + 1;
    const cutCap = (protectedTownCells.has(v) || isIgnitionCell[v]) ? INF : 1;
    dinic.addEdge(vin, vout, cutCap);
  }

  for (let i = 0; i < region.length; i++) {
    const v = region[i];
    const vout = i * 2 + 1;
    neighbors(v, (u) => {
      const j = localId.get(u);
      if (j === undefined) return;
      const uin = j * 2;
      dinic.addEdge(vout, uin, INF);
    });
  }

  for (const v of sourceCells) {
    const i = localId.get(v);
    dinic.addEdge(S, i * 2, INF);
  }
  for (const v of protectedTownCells) {
    const i = localId.get(v);
    if (i !== undefined) dinic.addEdge(i * 2 + 1, T, INF);
  }

  const flow = dinic.maxFlow(S, T);
  if (!(flow > 0) || flow > remainingBudget) return null;

  const reach = dinic.minCutReachable(S);
  const cut = [];
  for (let i = 0; i < region.length; i++) {
    const vin = i * 2;
    const vout = vin + 1;
    if (reach[vin] && !reach[vout]) {
      const v = region[i];
      if (!protectedTownCells.has(v) && !isIgnitionCell[v]) cut.push(v);
    }
  }

  if (cut.length === 0 || cut.length > remainingBudget) return null;
  return cut;
}

function chooseCandidates(state, remainingBudget) {
  const { dist, parent, order } = state;
  const candidates = [];
  const seenKeys = new Set();

  function addCandidate(cells) {
    if (!cells || cells.length === 0 || cells.length > remainingBudget) return;
    const unique = [];
    const localSeen = new Set();
    for (const c of cells) {
      if (blocked[c] || localSeen.has(c)) continue;
      localSeen.add(c);
      unique.push(c);
    }
    if (unique.length === 0 || unique.length > remainingBudget) return;
    unique.sort((a, b) => a - b);
    const key = unique.join(',');
    if (seenKeys.has(key)) return;
    seenKeys.add(key);
    candidates.push(unique);
  }

  const subtree = new Float64Array(n);
  const reachableDegree = new Uint8Array(n);
  for (let i = 0; i < towns.length; i++) {
    const idx = townCell[i];
    if (idx >= 0 && idx < n && dist[idx] !== -1) subtree[idx] += Number(towns[i].value) || 0;
  }

  for (let oi = order.length - 1; oi >= 0; oi--) {
    const v = order[oi];
    let deg = 0;
    neighbors(v, (u) => {
      if (dist[u] !== -1) deg++;
    });
    reachableDegree[v] = deg;
    const p = parent[v];
    if (p !== -1) subtree[p] += subtree[v];
  }

  const scoredSingles = [];
  for (let oi = 0; oi < order.length; oi++) {
    const v = order[oi];
    if (subtree[v] <= 0 || isTownCell[v] || isIgnitionCell[v]) continue;
    const depth = dist[v];
    const deg = reachableDegree[v] || 1;
    const score = subtree[v] / Math.pow(depth + 1, 0.35) / Math.max(1, deg - 0.25);
    scoredSingles.push([score, v]);
  }
  scoredSingles.sort((a, b) => b[0] - a[0]);
  for (let i = 0; i < Math.min(36, scoredSingles.length); i++) addCandidate([scoredSingles[i][1]]);

  const atRiskTownOrder = [];
  for (let i = 0; i < towns.length; i++) {
    const idx = townCell[i];
    if (idx < 0 || idx >= n || dist[idx] === -1) continue;
    const value = Number(towns[i].value) || 0;
    const score = value * (1 + Math.min(12, dist[idx]) * 0.05);
    atRiskTownOrder.push([score, i]);
  }
  atRiskTownOrder.sort((a, b) => b[0] - a[0]);

  for (let k = 0; k < Math.min(10, atRiskTownOrder.length); k++) {
    const townIdx = atRiskTownOrder[k][1];
    let cur = townCell[townIdx];
    for (let step = 0; step < 6; step++) {
      cur = parent[cur];
      if (cur === -1 || isTownCell[cur] || isIgnitionCell[cur]) break;
      addCandidate([cur]);
    }
  }

  const smallRadius = Math.max(3, Math.min(7, ((remainingBudget / 2) | 0) + 2));
  const largeRadius = Math.max(smallRadius, Math.min(12, smallRadius + 3));
  const radii = largeRadius === smallRadius ? [smallRadius] : [smallRadius, largeRadius];
  for (let k = 0; k < Math.min(6, atRiskTownOrder.length); k++) {
    const townIdx = atRiskTownOrder[k][1];
    for (const radius of radii) {
      const cut = localMinCutForTown(townIdx, dist, remainingBudget, radius);
      addCandidate(cut);
    }
  }

  return candidates;
}

function evaluateCandidate(cells) {
  for (const c of cells) blocked[c] = 1;
  const burned = computeBurnedValue();
  for (const c of cells) blocked[c] = 0;
  return burned;
}

const selected = [];
let currentState = simulateDetailed();
let currentBurned = currentState.burnedValue;

for (let remaining = cutBudget; remaining > 0; ) {
  if (currentBurned <= 0) break;
  const candidates = chooseCandidates(currentState, remaining);
  if (candidates.length === 0) break;

  let bestCells = null;
  let bestBurned = currentBurned;
  let bestScore = 0;
  let bestSaved = 0;

  for (const cells of candidates) {
    const burned = evaluateCandidate(cells);
    const saved = currentBurned - burned;
    if (!(saved > 0)) continue;
    const score = saved / Math.pow(cells.length, 1.15);
    if (
      bestCells === null ||
      score > bestScore + 1e-9 ||
      (Math.abs(score - bestScore) <= 1e-9 && (saved > bestSaved || (saved === bestSaved && cells.length < bestCells.length)))
    ) {
      bestCells = cells;
      bestBurned = burned;
      bestScore = score;
      bestSaved = saved;
    }
  }

  if (bestCells === null) break;

  for (const c of bestCells) {
    blocked[c] = 1;
    selected.push([c % width, (c / width) | 0]);
  }
  remaining -= bestCells.length;
  currentState = simulateDetailed();
  currentBurned = currentState.burnedValue;
  if (currentBurned !== bestBurned) currentBurned = bestBurned;
}

console.log(JSON.stringify({ firebreaks: selected }));
