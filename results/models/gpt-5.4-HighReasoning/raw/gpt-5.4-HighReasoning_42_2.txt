%main = OpFunction %void None %func
%entry = OpLabel

%center_pos = OpLoad %v4float %in_pos_ptr
%in_color_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%center_color = OpLoad %v3float %in_color_ptr
%edge_color = OpVectorTimesScalar %v3float %center_color %float_0_8

%cx = OpCompositeExtract %float %center_pos 0
%cy = OpCompositeExtract %float %center_pos 1
%cz = OpCompositeExtract %float %center_pos 2
%cw = OpCompositeExtract %float %center_pos 3

%e0x = OpFAdd %float %cx %float_0_04
%edge0 = OpCompositeConstruct %v4float %e0x %cy %cz %cw

%e1x = OpFAdd %float %cx %float_0_02
%e1y = OpFAdd %float %cy %float_0_034641016
%edge1 = OpCompositeConstruct %v4float %e1x %e1y %cz %cw

%e2x = OpFAdd %float %cx %float_n0_02
%e2y = OpFAdd %float %cy %float_0_034641016
%edge2 = OpCompositeConstruct %v4float %e2x %e2y %cz %cw

%e3x = OpFAdd %float %cx %float_n0_04
%edge3 = OpCompositeConstruct %v4float %e3x %cy %cz %cw

%e4x = OpFAdd %float %cx %float_n0_02

%e5x = OpFAdd %float %cx %float_0_02
%e5y = OpFAdd %float %cy %float_n0_034641016
%edge5 = OpCompositeConstruct %v4float %e5x %e5y %cz %cw

%out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
