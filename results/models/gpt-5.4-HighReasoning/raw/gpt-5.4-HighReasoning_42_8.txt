OpCapability Shader
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_7 = OpConstant %float 0.69999999
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_3 Input
%color = OpVariable %ptr_Input_arr_v3float_3 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %fn
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum012 %float_0_33333334
%d0 = OpFSub %v4float %p0 %centroid
%sd0 = OpVectorTimesScalar %v4float %d0 %float_0_7
%new0 = OpFAdd %v4float %centroid %sd0
%out_pos0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos0 %new0
OpStore %outColor %c0
OpEmitVertex
%d1 = OpFSub %v4float %p1 %centroid
%sd1 = OpVectorTimesScalar %v4float %d1 %float_0_7
%new1 = OpFAdd %v4float %centroid %sd1
%out_pos1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos1 %new1
OpStore %outColor %c1
OpEmitVertex
%d2 = OpFSub %v4float %p2 %centroid
%sd2 = OpVectorTimesScalar %v4float %d2 %float_0_7
%new2 = OpFAdd %v4float %centroid %sd2
%out_pos2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos2 %new2
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
