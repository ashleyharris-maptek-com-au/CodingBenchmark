use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq)]
struct HeapEntry {
    gain: usize,
    degree: usize,
    version: u32,
    vertex: usize,
}

impl Ord for HeapEntry {
    fn cmp(&self, other: &Self) -> Ordering {
        self.gain
            .cmp(&other.gain)
            .then_with(|| self.degree.cmp(&other.degree))
            .then_with(|| other.vertex.cmp(&self.vertex))
            .then_with(|| self.version.cmp(&other.version))
    }
}

impl PartialOrd for HeapEntry {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn add_vertex(
    v: usize,
    solution: &mut [bool],
    dom_count: &mut [u32],
    neighbors: &[Vec<usize>],
    uncovered: &mut usize,
) -> bool {
    if solution[v] {
        return false;
    }
    solution[v] = true;
    for &u in &neighbors[v] {
        if dom_count[u] == 0 {
            *uncovered -= 1;
        }
        dom_count[u] += 1;
    }
    true
}

fn remove_vertex(
    v: usize,
    solution: &mut [bool],
    dom_count: &mut [u32],
    neighbors: &[Vec<usize>],
) {
    solution[v] = false;
    for &u in &neighbors[v] {
        dom_count[u] -= 1;
    }
}

fn compute_gain(v: usize, dom_count: &[u32], neighbors: &[Vec<usize>]) -> usize {
    neighbors[v]
        .iter()
        .filter(|&&u| dom_count[u] == 0)
        .count()
}

fn greedy_dominating_set(neighbors: &[Vec<usize>]) -> Vec<bool> {
    let n = neighbors.len();
    let mut solution = vec![false; n];
    let mut dom_count = vec![0_u32; n];
    let mut uncovered = n;

    for v in 0..n {
        if neighbors[v].len() == 1 {
            add_vertex(v, &mut solution, &mut dom_count, neighbors, &mut uncovered);
        }
    }

    let mut versions = vec![0_u32; n];
    let mut heap = BinaryHeap::new();
    for v in 0..n {
        heap.push(HeapEntry {
            gain: neighbors[v].len(),
            degree: neighbors[v].len(),
            version: versions[v],
            vertex: v,
        });
    }

    while uncovered > 0 {
        let top = loop {
            let mut entry = heap.pop().expect("heap should not be empty");
            if solution[entry.vertex] {
                continue;
            }
            let actual_gain = compute_gain(entry.vertex, &dom_count, neighbors);
            if actual_gain != entry.gain || entry.version != versions[entry.vertex] {
                versions[entry.vertex] += 1;
                entry.gain = actual_gain;
                entry.version = versions[entry.vertex];
                heap.push(entry);
                continue;
            }
            break entry;
        };

        add_vertex(
            top.vertex,
            &mut solution,
            &mut dom_count,
            neighbors,
            &mut uncovered,
        );
    }

    solution
}

fn prune_solution(solution: &mut [bool], neighbors: &[Vec<usize>]) -> Vec<u32> {
    let n = neighbors.len();
    let mut dom_count = vec![0_u32; n];
    for v in 0..n {
        if solution[v] {
            for &u in &neighbors[v] {
                dom_count[u] += 1;
            }
        }
    }

    let mut order: Vec<usize> = (0..n).filter(|&v| solution[v]).collect();
    order.sort_by_key(|&v| neighbors[v].len());

    for v in order {
        if !solution[v] {
            continue;
        }
        let removable = neighbors[v].iter().all(|&u| dom_count[u] >= 2);
        if removable {
            remove_vertex(v, solution, &mut dom_count, neighbors);
        }
    }

    dom_count
}

fn improve_by_single_replacement(solution: &mut [bool], neighbors: &[Vec<usize>], dom_count: &mut [u32]) {
    let n = neighbors.len();
    let mut candidate_hits = vec![0_u32; n];
    let mut seen = Vec::new();

    loop {
        let mut changed = false;
        let selected: Vec<usize> = (0..n).filter(|&v| solution[v]).collect();

        'outer: for s in selected {
            let private_vertices: Vec<usize> = neighbors[s]
                .iter()
                .copied()
                .filter(|&u| dom_count[u] == 1)
                .collect();
            if private_vertices.is_empty() {
                continue;
            }

            seen.clear();
            for &pv in &private_vertices {
                for &cand in &neighbors[pv] {
                    if !solution[cand] && candidate_hits[cand] == 0 {
                        seen.push(cand);
                    }
                    if !solution[cand] {
                        candidate_hits[cand] += 1;
                    }
                }
            }

            let current_degree = neighbors[s].len();
            let mut best = None;
            let mut best_degree = current_degree;
            for &cand in &seen {
                if candidate_hits[cand] as usize == private_vertices.len() {
                    let degree = neighbors[cand].len();
                    if degree > best_degree {
                        best = Some(cand);
                        best_degree = degree;
                    }
                }
            }

            for &cand in &seen {
                candidate_hits[cand] = 0;
            }

            if let Some(cand) = best {
                solution[cand] = true;
                for &u in &neighbors[cand] {
                    dom_count[u] += 1;
                }
                remove_vertex(s, solution, dom_count, neighbors);
                changed = true;
                break 'outer;
            }
        }

        if !changed {
            break;
        }
    }
}

fn build_closed_neighborhoods(n: usize, edges: &[(usize, usize)]) -> Vec<Vec<usize>> {
    let mut adj = vec![Vec::<usize>::new(); n];
    for &(u, v) in edges {
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for list in &mut adj {
        list.sort_unstable();
        list.dedup();
    }

    let mut closed = Vec::with_capacity(n);
    for (v, list) in adj.into_iter().enumerate() {
        let mut full = Vec::with_capacity(list.len() + 1);
        full.push(v);
        full.extend(list);
        closed.push(full);
    }
    closed
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            println!("0");
            println!();
            return;
        }
    };
    let m: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            println!("0");
            println!();
            return;
        }
    };

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        if u < n && v < n {
            edges.push((u, v));
        }
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let neighbors = build_closed_neighborhoods(n, &edges);
    let mut solution = greedy_dominating_set(&neighbors);
    let mut dom_count = prune_solution(&mut solution, &neighbors);
    improve_by_single_replacement(&mut solution, &neighbors, &mut dom_count);
    dom_count = prune_solution(&mut solution, &neighbors);

    debug_assert!(dom_count.iter().all(|&c| c > 0));

    let result: Vec<usize> = solution
        .iter()
        .enumerate()
        .filter_map(|(v, &chosen)| if chosen { Some(v) } else { None })
        .collect();

    println!("{}", result.len());
    if result.is_empty() {
        println!();
    } else {
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{v}");
        }
        println!();
    }
}
