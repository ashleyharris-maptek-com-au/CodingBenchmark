using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private static readonly int[] Dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] Dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        var scanner = new FastInput(Console.In);
        int height = scanner.NextInt();
        int width = scanner.NextInt();

        var board = new char[height][];
        for (int y = 0; y < height; y++)
        {
            string line = scanner.NextRawLine(width);
            if (line.Length < width) line = line.PadRight(width, ' ');
            board[y] = line.ToCharArray();
        }

        var solver = new MinesweeperSolver(board, width, height);
        List<Cell> safeCells = solver.Solve();

        var output = new StringWriter();
        output.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            output.Write(cell.X);
            output.Write(' ');
            output.WriteLine(cell.Y);
        }

        Console.Write(output.ToString());
    }

    private sealed class MinesweeperSolver
    {
        private readonly char[][] _board;
        private readonly int _width;
        private readonly int _height;

        private readonly bool[] _safe;
        private readonly bool[] _mine;
        private readonly bool[] _unknown;

        public MinesweeperSolver(char[][] board, int width, int height)
        {
            _board = board;
            _width = width;
            _height = height;
            int n = width * height;
            _safe = new bool[n];
            _mine = new bool[n];
            _unknown = new bool[n];

            for (int y = 0; y < height; y++)
            {
                for (int x = 0; x < width; x++)
                {
                    char c = board[y][x];
                    int id = Id(x, y);
                    if (c == ' ')
                    {
                        _unknown[id] = true;
                    }
                    else if (c == '*')
                    {
                        _mine[id] = true;
                    }
                }
            }
        }

        public List<Cell> Solve()
        {
            var equations = BuildEquations();
            ApplyBasicDeductions(equations);

            equations = BuildEquations();
            if (equations.Count > 0)
            {
                ApplySubsetDeductions(equations);
                equations = BuildEquations();
            }

            if (equations.Count > 0)
            {
                SolveComponents(equations);
            }

            var result = new List<Cell>();
            for (int y = 0; y < _height; y++)
            {
                for (int x = 0; x < _width; x++)
                {
                    int id = Id(x, y);
                    if (_unknown[id] && _safe[id] && !_mine[id])
                    {
                        result.Add(new Cell(x, y));
                    }
                }
            }

            return result;
        }

        private List<Equation> BuildEquations()
        {
            var equations = new List<Equation>();

            for (int y = 0; y < _height; y++)
            {
                for (int x = 0; x < _width; x++)
                {
                    char c = _board[y][x];
                    if (c < '0' || c > '8') continue;

                    int target = c - '0';
                    int knownMines = 0;
                    var vars = new List<int>(8);

                    ForEachNeighbor(x, y, (nx, ny) =>
                    {
                        int nid = Id(nx, ny);
                        if (_mine[nid] || _board[ny][nx] == '*')
                        {
                            knownMines++;
                        }
                        else if (_unknown[nid] && !_safe[nid])
                        {
                            vars.Add(nid);
                        }
                    });

                    int remaining = target - knownMines;
                    if (remaining < 0) continue;
                    if (vars.Count == 0) continue;

                    vars.Sort();
                    vars = Unique(vars);
                    if (vars.Count == 0) continue;

                    equations.Add(new Equation(vars, remaining));
                }
            }

            return equations;
        }

        private void ApplyBasicDeductions(List<Equation> equations)
        {
            bool changed;
            do
            {
                changed = false;
                foreach (var eq in equations)
                {
                    var reduced = Reduce(eq);
                    if (reduced.Vars.Count == 0) continue;

                    if (reduced.MinesNeeded == 0)
                    {
                        foreach (int id in reduced.Vars)
                        {
                            if (MarkSafe(id)) changed = true;
                        }
                    }
                    else if (reduced.MinesNeeded == reduced.Vars.Count)
                    {
                        foreach (int id in reduced.Vars)
                        {
                            if (MarkMine(id)) changed = true;
                        }
                    }
                }

                if (changed)
                {
                    equations = BuildEquations();
                }
            } while (changed);
        }

        private void ApplySubsetDeductions(List<Equation> equations)
        {
            bool changed;
            int passes = equations.Count <= 5000 ? 3 : 1;

            do
            {
                changed = false;
                var grouped = new Dictionary<int, List<Equation>>();
                foreach (var eq in equations)
                {
                    var reduced = Reduce(eq);
                    if (reduced.Vars.Count == 0) continue;
                    if (!grouped.TryGetValue(reduced.Vars.Count, out var list))
                    {
                        list = new List<Equation>();
                        grouped[reduced.Vars.Count] = list;
                    }
                    list.Add(reduced);
                }

                var all = grouped.Values.SelectMany(v => v).ToList();
                int limit = all.Count > 1200 ? 1200 : all.Count;

                for (int i = 0; i < limit; i++)
                {
                    var a = all[i];
                    for (int j = 0; j < limit; j++)
                    {
                        if (i == j) continue;
                        var b = all[j];
                        if (a.Vars.Count >= b.Vars.Count) continue;
                        if (!IsSubset(a.Vars, b.Vars)) continue;

                        var diff = Difference(b.Vars, a.Vars);
                        int diffMines = b.MinesNeeded - a.MinesNeeded;
                        if (diffMines < 0 || diffMines > diff.Count) continue;

                        if (diff.Count == 0) continue;
                        if (diffMines == 0)
                        {
                            foreach (int id in diff)
                            {
                                if (MarkSafe(id)) changed = true;
                            }
                        }
                        else if (diffMines == diff.Count)
                        {
                            foreach (int id in diff)
                            {
                                if (MarkMine(id)) changed = true;
                            }
                        }
                    }
                }

                if (changed)
                {
                    ApplyBasicDeductions(BuildEquations());
                    equations = BuildEquations();
                }
            } while (changed && --passes > 0);
        }

        private void SolveComponents(List<Equation> equations)
        {
            var reduced = new List<Equation>(equations.Count);
            foreach (var eq in equations)
            {
                var r = Reduce(eq);
                if (r.Vars.Count > 0 && r.MinesNeeded >= 0 && r.MinesNeeded <= r.Vars.Count)
                {
                    reduced.Add(r);
                }
            }

            if (reduced.Count == 0) return;

            var adjacency = new Dictionary<int, List<int>>();
            for (int i = 0; i < reduced.Count; i++)
            {
                foreach (int varId in reduced[i].Vars)
                {
                    if (!adjacency.TryGetValue(varId, out var list))
                    {
                        list = new List<int>();
                        adjacency[varId] = list;
                    }
                    list.Add(i);
                }
            }

            var visitedEq = new bool[reduced.Count];
            foreach (int startVar in adjacency.Keys)
            {
                var seedEqs = adjacency[startVar];
                if (seedEqs.Count == 0 || visitedEq[seedEqs[0]]) continue;

                var compEqIdx = new List<int>();
                var compVars = new HashSet<int>();
                var eqQueue = new Queue<int>();

                foreach (int eqIdx in seedEqs)
                {
                    if (!visitedEq[eqIdx])
                    {
                        visitedEq[eqIdx] = true;
                        eqQueue.Enqueue(eqIdx);
                    }
                }

                while (eqQueue.Count > 0)
                {
                    int eqIdx = eqQueue.Dequeue();
                    compEqIdx.Add(eqIdx);
                    foreach (int varId in reduced[eqIdx].Vars)
                    {
                        if (!compVars.Add(varId)) continue;
                        if (!adjacency.TryGetValue(varId, out var linked)) continue;
                        foreach (int nextEq in linked)
                        {
                            if (!visitedEq[nextEq])
                            {
                                visitedEq[nextEq] = true;
                                eqQueue.Enqueue(nextEq);
                            }
                        }
                    }
                }

                AnalyzeComponent(compVars.ToList(), compEqIdx.Select(i => reduced[i]).ToList());
            }
        }

        private void AnalyzeComponent(List<int> vars, List<Equation> equations)
        {
            if (vars.Count == 0) return;

            vars.Sort();
            int variableCount = vars.Count;
            int equationCount = equations.Count;

            if (variableCount > 28 || equationCount > 64)
            {
                AnalyzeByLayeredPropagation(vars, equations);
                return;
            }

            var indexOf = new Dictionary<int, int>(variableCount);
            for (int i = 0; i < variableCount; i++)
            {
                indexOf[vars[i]] = i;
            }

            var localEqs = new LocalEquation[equationCount];
            var varToEqs = new List<int>[variableCount];
            for (int i = 0; i < variableCount; i++) varToEqs[i] = new List<int>();

            for (int i = 0; i < equationCount; i++)
            {
                int[] localVars = new int[equations[i].Vars.Count];
                for (int j = 0; j < equations[i].Vars.Count; j++)
                {
                    int local = indexOf[equations[i].Vars[j]];
                    localVars[j] = local;
                    varToEqs[local].Add(i);
                }

                localEqs[i] = new LocalEquation(localVars, equations[i].MinesNeeded);
            }

            var domains = new sbyte[variableCount];
            for (int i = 0; i < variableCount; i++) domains[i] = -1;

            var mineSeen = new bool[variableCount];
            var safeSeen = new bool[variableCount];
            int solutions = 0;
            const int solutionCap = 200000;

            bool Dfs()
            {
                if (solutions >= solutionCap) return false;

                int pick = SelectVariable(domains, localEqs);
                if (pick == -1)
                {
                    solutions++;
                    for (int i = 0; i < variableCount; i++)
                    {
                        if (domains[i] == 1) mineSeen[i] = true;
                        else safeSeen[i] = true;
                    }
                    return true;
                }

                foreach (sbyte value in EnumerateValues(pick, localEqs))
                {
                    var changes = new List<(int varIdx, sbyte oldValue)>();
                    var eqChanges = new List<(int eqIdx, int minesLeft, int unknownLeft)>();

                    if (!Assign(pick, value, domains, localEqs, varToEqs, changes, eqChanges))
                    {
                        Rollback(domains, localEqs, changes, eqChanges);
                        continue;
                    }

                    if (!Propagate(domains, localEqs, varToEqs, changes, eqChanges))
                    {
                        Rollback(domains, localEqs, changes, eqChanges);
                        continue;
                    }

                    if (!Dfs())
                    {
                        Rollback(domains, localEqs, changes, eqChanges);
                        return false;
                    }

                    Rollback(domains, localEqs, changes, eqChanges);
                }

                return true;
            }

            Dfs();

            if (solutions == 0) return;

            for (int i = 0; i < variableCount; i++)
            {
                if (safeSeen[i] && !mineSeen[i]) MarkSafe(vars[i]);
                if (mineSeen[i] && !safeSeen[i]) MarkMine(vars[i]);
            }
        }

        private void AnalyzeByLayeredPropagation(List<int> vars, List<Equation> equations)
        {
            foreach (int candidate in vars)
            {
                bool minePossible = IsAssignmentFeasible(candidate, true, vars, equations);
                bool safePossible = IsAssignmentFeasible(candidate, false, vars, equations);

                if (safePossible && !minePossible) MarkSafe(candidate);
                else if (minePossible && !safePossible) MarkMine(candidate);
            }
        }

        private bool IsAssignmentFeasible(int forcedVar, bool forcedMine, List<int> vars, List<Equation> equations)
        {
            var state = new Dictionary<int, sbyte>(vars.Count) { [forcedVar] = forcedMine ? (sbyte)1 : (sbyte)0 };
            bool changed;
            int guard = 0;

            do
            {
                changed = false;
                foreach (var eq in equations)
                {
                    int minesLeft = eq.MinesNeeded;
                    var undecided = new List<int>(eq.Vars.Count);

                    foreach (int varId in eq.Vars)
                    {
                        if (state.TryGetValue(varId, out sbyte val))
                        {
                            if (val == 1) minesLeft--;
                        }
                        else
                        {
                            undecided.Add(varId);
                        }
                    }

                    if (minesLeft < 0 || minesLeft > undecided.Count) return false;
                    if (undecided.Count == 0) continue;

                    if (minesLeft == 0)
                    {
                        foreach (int id in undecided)
                        {
                            if (!SetState(state, id, 0)) return false;
                            changed = true;
                        }
                    }
                    else if (minesLeft == undecided.Count)
                    {
                        foreach (int id in undecided)
                        {
                            if (!SetState(state, id, 1)) return false;
                            changed = true;
                        }
                    }
                }
            } while (changed && ++guard < 512);

            return true;
        }

        private static bool SetState(Dictionary<int, sbyte> state, int id, sbyte value)
        {
            if (state.TryGetValue(id, out sbyte existing))
            {
                return existing == value;
            }

            state[id] = value;
            return true;
        }

        private static IEnumerable<sbyte> EnumerateValues(int varIdx, LocalEquation[] equations)
        {
            int pressureMine = 0;
            int pressureSafe = 0;

            foreach (var eq in equations)
            {
                bool contains = false;
                for (int i = 0; i < eq.Vars.Length; i++)
                {
                    if (eq.Vars[i] == varIdx)
                    {
                        contains = true;
                        break;
                    }
                }

                if (!contains) continue;
                if (eq.MinesLeft > 0) pressureMine++;
                if (eq.UnknownLeft - eq.MinesLeft > 0) pressureSafe++;
            }

            if (pressureMine > pressureSafe)
            {
                yield return 1;
                yield return 0;
            }
            else
            {
                yield return 0;
                yield return 1;
            }
        }

        private static int SelectVariable(sbyte[] domains, LocalEquation[] equations)
        {
            int bestVar = -1;
            int bestScore = -1;

            for (int i = 0; i < domains.Length; i++)
            {
                if (domains[i] != -1) continue;

                int score = 0;
                foreach (var eq in equations)
                {
                    bool contains = false;
                    for (int j = 0; j < eq.Vars.Length; j++)
                    {
                        if (eq.Vars[j] == i)
                        {
                            contains = true;
                            break;
                        }
                    }

                    if (!contains) continue;
                    score += 16 - Math.Min(15, eq.UnknownLeft);
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    bestVar = i;
                }
            }

            return bestVar;
        }

        private static bool Assign(
            int varIdx,
            sbyte value,
            sbyte[] domains,
            LocalEquation[] equations,
            List<int>[] varToEqs,
            List<(int varIdx, sbyte oldValue)> changes,
            List<(int eqIdx, int minesLeft, int unknownLeft)> eqChanges)
        {
            if (domains[varIdx] != -1) return domains[varIdx] == value;

            changes.Add((varIdx, domains[varIdx]));
            domains[varIdx] = value;

            foreach (int eqIdx in varToEqs[varIdx])
            {
                eqChanges.Add((eqIdx, equations[eqIdx].MinesLeft, equations[eqIdx].UnknownLeft));
                equations[eqIdx].UnknownLeft--;
                if (value == 1) equations[eqIdx].MinesLeft--;
                if (equations[eqIdx].MinesLeft < 0 || equations[eqIdx].MinesLeft > equations[eqIdx].UnknownLeft)
                {
                    return false;
                }
            }

            return true;
        }

        private static bool Propagate(
            sbyte[] domains,
            LocalEquation[] equations,
            List<int>[] varToEqs,
            List<(int varIdx, sbyte oldValue)> changes,
            List<(int eqIdx, int minesLeft, int unknownLeft)> eqChanges)
        {
            bool changed;
            int guard = 0;

            do
            {
                changed = false;
                for (int eqIdx = 0; eqIdx < equations.Length; eqIdx++)
                {
                    var eq = equations[eqIdx];
                    if (eq.UnknownLeft == 0)
                    {
                        if (eq.MinesLeft != 0) return false;
                        continue;
                    }

                    if (eq.MinesLeft == 0 || eq.MinesLeft == eq.UnknownLeft)
                    {
                        sbyte forced = (sbyte)(eq.MinesLeft == 0 ? 0 : 1);
                        for (int i = 0; i < eq.Vars.Length; i++)
                        {
                            int varIdx = eq.Vars[i];
                            if (domains[varIdx] != -1) continue;
                            if (!Assign(varIdx, forced, domains, equations, varToEqs, changes, eqChanges))
                            {
                                return false;
                            }
                            changed = true;
                        }
                    }
                }
            } while (changed && ++guard < 1024);

            return true;
        }

        private static void Rollback(
            sbyte[] domains,
            LocalEquation[] equations,
            List<(int varIdx, sbyte oldValue)> changes,
            List<(int eqIdx, int minesLeft, int unknownLeft)> eqChanges)
        {
            for (int i = eqChanges.Count - 1; i >= 0; i--)
            {
                var change = eqChanges[i];
                equations[change.eqIdx].MinesLeft = change.minesLeft;
                equations[change.eqIdx].UnknownLeft = change.unknownLeft;
            }

            for (int i = changes.Count - 1; i >= 0; i--)
            {
                var change = changes[i];
                domains[change.varIdx] = change.oldValue;
            }
        }

        private Equation Reduce(Equation eq)
        {
            int minesNeeded = eq.MinesNeeded;
            var vars = new List<int>(eq.Vars.Count);

            foreach (int id in eq.Vars)
            {
                if (_mine[id]) minesNeeded--;
                else if (_unknown[id] && !_safe[id]) vars.Add(id);
            }

            return new Equation(vars, minesNeeded);
        }

        private bool MarkSafe(int id)
        {
            if (_mine[id] || _safe[id] || !_unknown[id]) return false;
            _safe[id] = true;
            return true;
        }

        private bool MarkMine(int id)
        {
            if (_mine[id] || !_unknown[id]) return false;
            _mine[id] = true;
            _safe[id] = false;
            return true;
        }

        private void ForEachNeighbor(int x, int y, Action<int, int> action)
        {
            for (int k = 0; k < 8; k++)
            {
                int nx = x + Dx[k];
                int ny = y + Dy[k];
                if ((uint)nx < (uint)_width && (uint)ny < (uint)_height)
                {
                    action(nx, ny);
                }
            }
        }

        private int Id(int x, int y) => y * _width + x;

        private static List<int> Unique(List<int> sorted)
        {
            if (sorted.Count <= 1) return sorted;
            var result = new List<int>(sorted.Count) { sorted[0] };
            for (int i = 1; i < sorted.Count; i++)
            {
                if (sorted[i] != sorted[i - 1]) result.Add(sorted[i]);
            }
            return result;
        }

        private static bool IsSubset(List<int> a, List<int> b)
        {
            int i = 0, j = 0;
            while (i < a.Count && j < b.Count)
            {
                if (a[i] == b[j]) { i++; j++; }
                else if (a[i] > b[j]) j++;
                else return false;
            }
            return i == a.Count;
        }

        private static List<int> Difference(List<int> superset, List<int> subset)
        {
            var result = new List<int>(superset.Count);
            int i = 0, j = 0;
            while (i < superset.Count)
            {
                if (j < subset.Count && superset[i] == subset[j])
                {
                    i++;
                    j++;
                }
                else
                {
                    result.Add(superset[i]);
                    i++;
                }
            }
            return result;
        }
    }

    private readonly struct Cell
    {
        public readonly int X;
        public readonly int Y;

        public Cell(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private readonly struct Equation
    {
        public readonly List<int> Vars;
        public readonly int MinesNeeded;

        public Equation(List<int> vars, int minesNeeded)
        {
            Vars = vars;
            MinesNeeded = minesNeeded;
        }
    }

    private sealed class LocalEquation
    {
        public readonly int[] Vars;
        public int MinesLeft;
        public int UnknownLeft;

        public LocalEquation(int[] vars, int minesLeft)
        {
            Vars = vars;
            MinesLeft = minesLeft;
            UnknownLeft = vars.Length;
        }
    }

    private sealed class FastInput
    {
        private readonly TextReader _reader;
        private readonly Queue<string> _tokens = new Queue<string>();

        public FastInput(TextReader reader)
        {
            _reader = reader;
        }

        public int NextInt()
        {
            while (_tokens.Count == 0)
            {
                string? line = _reader.ReadLine();
                if (line == null) throw new EndOfStreamException();
                foreach (string part in line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries))
                {
                    _tokens.Enqueue(part);
                }
            }

            return int.Parse(_tokens.Dequeue());
        }

        public string NextRawLine(int expectedWidth)
        {
            if (_tokens.Count > 0)
            {
                string first = _tokens.Dequeue();
                if (first.Length == expectedWidth) return first;
                return first;
            }

            string? line = _reader.ReadLine();
            return line ?? string.Empty;
        }
    }
}
