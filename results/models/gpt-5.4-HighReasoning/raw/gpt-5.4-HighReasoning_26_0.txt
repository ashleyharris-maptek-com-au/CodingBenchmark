use std::env;
use std::fs::{remove_file, File, OpenOptions};
use std::io::{self, BufRead, BufReader, BufWriter, Read, Seek, SeekFrom, Write};
use std::path::{Path, PathBuf};
use std::time::{SystemTime, UNIX_EPOCH};

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn dist2(self, other: Point) -> f64 {
        let dx = (self.x - other.x) as f64;
        let dy = (self.y - other.y) as f64;
        let dz = (self.z - other.z) as f64;
        dx * dx + dy * dy + dz * dz
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Accum {
    sx: f64,
    sy: f64,
    sz: f64,
    count: u64,
}

impl Accum {
    fn add(&mut self, p: Point) {
        self.sx += p.x as f64;
        self.sy += p.y as f64;
        self.sz += p.z as f64;
        self.count += 1;
    }
}

struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        let init = if seed == 0 { 0x9E3779B97F4A7C15 } else { seed };
        Self { state: init }
    }

    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E3779B97F4A7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }

    fn gen_range(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            (self.next_u64() % upper as u64) as usize
        }
    }

    fn gen_f64(&mut self) -> f64 {
        const SCALE: f64 = 1.0 / ((1u64 << 53) as f64);
        ((self.next_u64() >> 11) as f64) * SCALE
    }
}

fn read_header<R: BufRead>(reader: &mut R) -> io::Result<(u64, usize)> {
    let mut line = Vec::new();
    let n = reader.read_until(b'\n', &mut line)?;
    if n == 0 {
        return Err(io::Error::new(io::ErrorKind::UnexpectedEof, "missing header"));
    }
    let header = std::str::from_utf8(&line)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "header is not valid UTF-8"))?;
    let mut parts = header.split_whitespace();
    let num_points = parts
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing num_points"))?
        .parse::<u64>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_points"))?;
    let num_clusters = parts
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing num_clusters"))?
        .parse::<usize>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_clusters"))?;
    Ok((num_points, num_clusters))
}

fn temp_path(prefix: &str) -> PathBuf {
    let mut path = env::temp_dir();
    let ts = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_nanos();
    path.push(format!("{}_{}_{}.bin", prefix, std::process::id(), ts));
    path
}

fn decode_point(bytes: &[u8; 12]) -> Point {
    Point {
        x: f32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]),
        y: f32::from_le_bytes([bytes[4], bytes[5], bytes[6], bytes[7]]),
        z: f32::from_le_bytes([bytes[8], bytes[9], bytes[10], bytes[11]]),
    }
}

fn nearest_center(point: Point, centers: &[Point]) -> usize {
    let mut best = 0usize;
    let mut best_dist = point.dist2(centers[0]);
    for (i, center) in centers.iter().enumerate().skip(1) {
        let d = point.dist2(*center);
        if d < best_dist {
            best_dist = d;
            best = i;
        }
    }
    best
}

fn reservoir_add(sample: &mut Vec<Point>, cap: usize, seen: u64, point: Point, rng: &mut Rng) {
    if cap == 0 {
        return;
    }
    if sample.len() < cap {
        sample.push(point);
    } else {
        let idx = rng.next_u64() % seen;
        if idx < cap as u64 {
            sample[idx as usize] = point;
        }
    }
}

fn choose_seed_pool(sample: &[Point], seed_pool_size: usize) -> Vec<Point> {
    if sample.len() <= seed_pool_size {
        return sample.to_vec();
    }
    let mut pool = Vec::with_capacity(seed_pool_size);
    let last = sample.len() - 1;
    for i in 0..seed_pool_size {
        let idx = i * last / (seed_pool_size - 1);
        pool.push(sample[idx]);
    }
    pool
}

fn init_centers(sample: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    if k == 0 {
        return Vec::new();
    }
    if sample.is_empty() {
        return vec![Point::default(); k];
    }

    let target_pool = (k.saturating_mul(8)).clamp(16_384, 200_000);
    let pool = choose_seed_pool(sample, target_pool.max(k).min(sample.len().max(1)));

    let mut centers = Vec::with_capacity(k);
    centers.push(pool[rng.gen_range(pool.len())]);

    let mut min_d2 = vec![f64::INFINITY; pool.len()];
    while centers.len() < k {
        let last = *centers.last().unwrap();
        let mut sum = 0.0f64;
        for (i, p) in pool.iter().enumerate() {
            let d = p.dist2(last);
            if d < min_d2[i] {
                min_d2[i] = d;
            }
            sum += min_d2[i];
        }
        if sum <= 0.0 {
            centers.push(pool[rng.gen_range(pool.len())]);
            continue;
        }
        let mut ticket = rng.gen_f64() * sum;
        let mut chosen = pool[pool.len() - 1];
        for (i, p) in pool.iter().enumerate() {
            ticket -= min_d2[i];
            if ticket <= 0.0 {
                chosen = *p;
                break;
            }
        }
        centers.push(chosen);
    }
    centers
}

fn update_centers_from_accums(centers: &mut [Point], accums: &[Accum], sample: &[Point], rng: &mut Rng) -> f64 {
    let mut total_shift = 0.0f64;
    for i in 0..centers.len() {
        let old = centers[i];
        let new_center = if accums[i].count > 0 {
            Point {
                x: (accums[i].sx / accums[i].count as f64) as f32,
                y: (accums[i].sy / accums[i].count as f64) as f32,
                z: (accums[i].sz / accums[i].count as f64) as f32,
            }
        } else if !sample.is_empty() {
            sample[rng.gen_range(sample.len())]
        } else {
            old
        };
        total_shift += old.dist2(new_center);
        centers[i] = new_center;
    }
    total_shift
}

fn full_kmeans(points: &[Point], centers: &mut [Point], sample: &[Point], rng: &mut Rng, max_iters: usize) {
    if points.is_empty() || centers.is_empty() {
        return;
    }
    for _ in 0..max_iters {
        let mut accums = vec![Accum::default(); centers.len()];
        for &p in points {
            let idx = nearest_center(p, centers);
            accums[idx].add(p);
        }
        let shift = update_centers_from_accums(centers, &accums, sample, rng);
        if shift / centers.len() as f64 <= 1e-8 {
            break;
        }
    }
}

fn streaming_lloyd(path: &Path, n: u64, centers: &mut [Point], sample: &[Point], rng: &mut Rng, passes: usize) -> io::Result<()> {
    if n == 0 || centers.is_empty() {
        return Ok(());
    }
    let mut raw = [0u8; 12];
    for _ in 0..passes {
        let mut accums = vec![Accum::default(); centers.len()];
        let mut reader = BufReader::with_capacity(8 << 20, File::open(path)?);
        for _ in 0..n {
            reader.read_exact(&mut raw)?;
            let p = decode_point(&raw);
            let idx = nearest_center(p, centers);
            accums[idx].add(p);
        }
        let shift = update_centers_from_accums(centers, &accums, sample, rng);
        if shift / centers.len() as f64 <= 1e-8 {
            break;
        }
    }
    Ok(())
}

fn streaming_online(path: &Path, n: u64, centers: &mut [Point], passes: usize) -> io::Result<()> {
    if n == 0 || centers.is_empty() {
        return Ok(());
    }
    let mut visits = vec![0u64; centers.len()];
    let mut raw = [0u8; 12];
    for _ in 0..passes {
        let mut reader = BufReader::with_capacity(8 << 20, File::open(path)?);
        for _ in 0..n {
            reader.read_exact(&mut raw)?;
            let p = decode_point(&raw);
            let idx = nearest_center(p, centers);
            visits[idx] += 1;
            let eta = 1.0f32 / (visits[idx] as f32).sqrt().max(1.0);
            centers[idx].x += (p.x - centers[idx].x) * eta;
            centers[idx].y += (p.y - centers[idx].y) * eta;
            centers[idx].z += (p.z - centers[idx].z) * eta;
        }
    }
    Ok(())
}

fn assign_and_write_labels(
    points_path: &Path,
    labels_path: &Path,
    n: u64,
    centers: &[Point],
) -> io::Result<Vec<u64>> {
    let mut counts = vec![0u64; centers.len()];
    let mut donor_positions: Vec<Vec<u64>> = vec![Vec::new(); centers.len()];
    let mut reader = BufReader::with_capacity(8 << 20, File::open(points_path)?);
    let mut writer = BufWriter::with_capacity(8 << 20, File::create(labels_path)?);
    let mut raw = [0u8; 12];

    for idx in 0..n {
        reader.read_exact(&mut raw)?;
        let point = decode_point(&raw);
        let cluster = nearest_center(point, centers);
        counts[cluster] += 1;
        if donor_positions[cluster].len() < 8 {
            donor_positions[cluster].push(idx);
        }
        writer.write_all(&(cluster as u32).to_le_bytes())?;
    }
    writer.flush()?;

    if n >= centers.len() as u64 {
        let mut patch_file = OpenOptions::new().read(true).write(true).open(labels_path)?;
        for missing in 0..centers.len() {
            if counts[missing] != 0 {
                continue;
            }
            let mut donor = None;
            let mut donor_count = 0u64;
            for i in 0..counts.len() {
                if counts[i] > donor_count && counts[i] > 1 && !donor_positions[i].is_empty() {
                    donor = Some(i);
                    donor_count = counts[i];
                }
            }
            if let Some(src) = donor {
                let pos = donor_positions[src]
                    .pop()
                    .expect("donor_positions[src] checked as non-empty");
                patch_file.seek(SeekFrom::Start(pos * 4))?;
                patch_file.write_all(&(missing as u32).to_le_bytes())?;
                counts[src] -= 1;
                counts[missing] = 1;
            }
        }
        patch_file.flush()?;
    }

    Ok(counts)
}

fn stream_file_to_stdout(path: &Path) -> io::Result<()> {
    let mut reader = BufReader::with_capacity(8 << 20, File::open(path)?);
    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(8 << 20, stdout.lock());
    io::copy(&mut reader, &mut writer)?;
    writer.flush()
}

fn solve() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = BufReader::with_capacity(8 << 20, stdin.lock());
    let (num_points, num_clusters) = read_header(&mut input)?;

    if num_clusters == 0 {
        return Ok(());
    }

    let mut rng = Rng::new(
        num_points
            ^ ((num_clusters as u64) << 32)
            ^ 0xA5A5_5A5A_DEAD_BEEF,
    );

    let points_path = temp_path("rust_kmeans_points");
    let labels_path = temp_path("rust_kmeans_labels");

    let small_in_memory = num_points <= 5_000_000;
    let sample_cap = if num_points <= 10_000_000 {
        (num_clusters.saturating_mul(128)).clamp(32_768, 2_000_000)
    } else if num_points <= 1_000_000_000 {
        (num_clusters.saturating_mul(64)).clamp(65_536, 1_000_000)
    } else {
        (num_clusters.saturating_mul(32)).clamp(65_536, 500_000)
    };

    let mut sample = Vec::with_capacity(sample_cap.min(num_points as usize));
    let mut maybe_points = if small_in_memory {
        Some(Vec::with_capacity(num_points as usize))
    } else {
        None
    };

    {
        let mut spill = BufWriter::with_capacity(8 << 20, File::create(&points_path)?);
        let mut raw = [0u8; 12];
        for seen in 1..=num_points {
            input.read_exact(&mut raw)?;
            spill.write_all(&raw)?;
            let p = decode_point(&raw);
            reservoir_add(&mut sample, sample_cap, seen, p, &mut rng);
            if let Some(points) = maybe_points.as_mut() {
                points.push(p);
            }
        }
        spill.flush()?;
    }

    let mut centers = init_centers(&sample, num_clusters, &mut rng);
    if centers.is_empty() {
        return Ok(());
    }

    if let Some(points) = maybe_points.as_ref() {
        let iters = if num_points <= 1_000_000 { 20 } else { 12 };
        full_kmeans(points, &mut centers, &sample, &mut rng, iters);
    } else if num_points <= 100_000_000 && num_clusters <= 2_048 {
        streaming_lloyd(&points_path, num_points, &mut centers, &sample, &mut rng, 8)?;
    } else if num_points <= 1_000_000_000 {
        streaming_lloyd(&points_path, num_points, &mut centers, &sample, &mut rng, 4)?;
    } else {
        streaming_online(&points_path, num_points, &mut centers, 2)?;
    }

    let _counts = assign_and_write_labels(&points_path, &labels_path, num_points, &centers)?;
    stream_file_to_stdout(&labels_path)?;

    let _ = remove_file(&points_path);
    let _ = remove_file(&labels_path);
    Ok(())
}

fn main() {
    if let Err(err) = solve() {
        let _ = writeln!(io::stderr().lock(), "error: {err}");
        std::process::exit(1);
    }
}
