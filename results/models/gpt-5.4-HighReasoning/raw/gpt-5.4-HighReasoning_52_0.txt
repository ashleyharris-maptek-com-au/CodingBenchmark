import math
from statistics import median


_AUTO_STATE = {
    "speed_ms": 0.0,
    "yaw_rate_dps": 0.0,
    "heading_deg": None,
    "lane_offset_m": 0.0,
    "lane_width_m": 3.6,
    "curvature_1pm": 0.0,
    "speed_limit_kmh": 130.0,
    "front_dist_m": 200.0,
    "front_rel_speed_kmh": 0.0,
    "steer_i": 0.0,
    "speed_i": 0.0,
    "last_steering": 0.0,
    "last_throttle": 0.0,
    "last_brake": 0.0,
    "brake_weak_counter": 0,
    "throttle_stuck_counter": 0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _is_num(x):
    return isinstance(x, (int, float)) and math.isfinite(x)


def _median_or_none(values):
    vals = [v for v in values if _is_num(v)]
    return median(vals) if vals else None


def _lowpass(prev, new, alpha):
    if new is None:
        return prev
    if prev is None:
        return new
    return prev + alpha * (new - prev)


def _wrap_angle_deg(a):
    while a > 180.0:
        a -= 360.0
    while a < -180.0:
        a += 360.0
    return a


def _plausible(v, lo, hi):
    return _is_num(v) and lo <= v <= hi


def autopilot_step(state: dict, dt: float) -> dict:
    global _AUTO_STATE

    dt_in = state.get("dt", dt)
    dt = dt_in if _plausible(dt_in, 0.01, 0.5) else dt
    dt = dt if _plausible(dt, 0.01, 0.5) else 0.05

    mu = state.get("estimated_mu", 0.9)
    mu = mu if _plausible(mu, 0.05, 1.2) else 0.9
    rain = bool(state.get("rain_sensor", False))

    speed_candidates = []
    for corner in ("fl", "fr", "rl", "rr"):
        v = state.get(f"wheel_speed_{corner}_kmh")
        valid = bool(state.get(f"wheel_speed_{corner}_valid", True))
        if valid and _plausible(v, 0.0, 260.0):
            speed_candidates.append(v / 3.6)
    for name in ("vehicle_speed_1_kmh", "vehicle_speed_2_kmh"):
        v = state.get(name)
        if _plausible(v, 0.0, 260.0):
            speed_candidates.append(v / 3.6)
    for idx in (1, 2):
        if bool(state.get(f"gps_{idx}_valid", False)):
            v = state.get(f"gps_{idx}_speed_kmh")
            if _plausible(v, 0.0, 260.0):
                speed_candidates.append(v / 3.6)

    fused_speed = _median_or_none(speed_candidates)
    if fused_speed is None:
        fused_speed = _AUTO_STATE["speed_ms"]
    speed_jump = abs(fused_speed - _AUTO_STATE["speed_ms"])
    if speed_jump > 12.0:
        fused_speed = _AUTO_STATE["speed_ms"] + math.copysign(12.0, fused_speed - _AUTO_STATE["speed_ms"])
    speed_ms = _lowpass(_AUTO_STATE["speed_ms"], fused_speed, 0.35)

    yaw_candidates = []
    for idx in (1, 2, 3):
        y = state.get(f"gyro_{idx}_yaw_rate_dps")
        valid = bool(state.get(f"gyro_{idx}_valid", True))
        if valid and _plausible(y, -90.0, 90.0):
            yaw_candidates.append(y)

    gps_heading_rates = []
    gps_headings = []
    for idx in (1, 2):
        if bool(state.get(f"gps_{idx}_valid", False)):
            hdg = state.get(f"gps_{idx}_heading_deg")
            if _plausible(hdg, -360.0, 360.0):
                gps_headings.append(hdg)
    gps_heading = _median_or_none(gps_headings)
    if gps_heading is not None and _AUTO_STATE["heading_deg"] is not None:
        d = _wrap_angle_deg(gps_heading - _AUTO_STATE["heading_deg"])
        gps_heading_rates.append(d / dt)
    if gps_heading_rates:
        yaw_candidates.append(_median_or_none(gps_heading_rates))

    fused_yaw = _median_or_none(yaw_candidates)
    if fused_yaw is None:
        fused_yaw = _AUTO_STATE["yaw_rate_dps"]
    yaw_rate_dps = _lowpass(_AUTO_STATE["yaw_rate_dps"], fused_yaw, 0.4)

    accel_lat = []
    accel_lon = []
    for idx in (1, 2, 3):
        valid = bool(state.get(f"accel_{idx}_valid", True))
        alat = state.get(f"accel_{idx}_lateral_ms2")
        alon = state.get(f"accel_{idx}_longitudinal_ms2")
        if valid and _plausible(alat, -15.0, 15.0):
            accel_lat.append(alat)
        if valid and _plausible(alon, -15.0, 15.0):
            accel_lon.append(alon)
    lat_acc_ms2 = _median_or_none(accel_lat) or 0.0
    lon_acc_ms2 = _median_or_none(accel_lon) or 0.0

    lane_offsets = []
    lane_widths = []
    curvatures = []
    speed_limits = []
    obstacle_dists = []
    obstacle_rel_speeds = []
    vision_good = 0

    for idx in (1, 2, 3):
        left_ok = bool(state.get(f"vision_{idx}_left_line_valid", False))
        right_ok = bool(state.get(f"vision_{idx}_right_line_valid", False))
        left = state.get(f"vision_{idx}_left_line_dist_m")
        right = state.get(f"vision_{idx}_right_line_dist_m")
        lane_offset = state.get(f"vision_{idx}_lane_offset_m")
        curvature = state.get(f"vision_{idx}_road_curvature_1pm")
        speed_limit = state.get(f"vision_{idx}_speed_limit_kmh")
        speed_limit_valid = bool(state.get(f"vision_{idx}_speed_limit_valid", False))
        obstacle_valid = bool(state.get(f"vision_{idx}_obstacle_valid", False))
        obstacle_detected = bool(state.get(f"vision_{idx}_obstacle_detected", False))
        obstacle_dist = state.get(f"vision_{idx}_obstacle_dist_m")
        obstacle_rel = state.get(f"vision_{idx}_obstacle_rel_speed_kmh")

        if left_ok and right_ok and _plausible(left, 0.1, 8.0) and _plausible(right, 0.1, 8.0):
            width = left + right
            if 2.8 <= width <= 4.8:
                lane_widths.append(width)
                lane_offsets.append(0.5 * (right - left))
                vision_good += 1
        if _plausible(lane_offset, -2.5, 2.5):
            lane_offsets.append(lane_offset)
            vision_good += 1
        if _plausible(curvature, -0.02, 0.02):
            curvatures.append(curvature)
        if speed_limit_valid and _plausible(speed_limit, 20.0, 140.0):
            speed_limits.append(speed_limit)
        if obstacle_valid and obstacle_detected and _plausible(obstacle_dist, 0.0, 250.0):
            obstacle_dists.append(obstacle_dist)
            if _plausible(obstacle_rel, -180.0, 180.0):
                obstacle_rel_speeds.append(obstacle_rel)

    lane_width_m = _median_or_none(lane_widths)
    lane_width_m = _lowpass(_AUTO_STATE["lane_width_m"], lane_width_m, 0.25)
    lane_offset_m = _median_or_none(lane_offsets)
    lane_offset_m = _lowpass(_AUTO_STATE["lane_offset_m"], lane_offset_m, 0.35)
    curvature_1pm = _median_or_none(curvatures)
    curvature_1pm = _lowpass(_AUTO_STATE["curvature_1pm"], curvature_1pm, 0.3)
    speed_limit_kmh = _median_or_none(speed_limits)
    speed_limit_kmh = _lowpass(_AUTO_STATE["speed_limit_kmh"], speed_limit_kmh, 0.25)

    front_lidar = []
    left_lidar = []
    right_lidar = []
    for ang in ("-090", "-060", "-045", "-030", "-015", "-005", "+005", "+015", "+030", "+045", "+060", "+090"):
        r = state.get(f"lidar_{ang}_deg_range_m")
        valid = bool(state.get(f"lidar_{ang}_deg_valid", True))
        if not (valid and _plausible(r, 0.0, 300.0)):
            continue
        if ang in ("-015", "-005", "+005", "+015"):
            front_lidar.append(r)
        if ang in ("+030", "+045", "+060", "+090"):
            left_lidar.append(r)
        if ang in ("-030", "-045", "-060", "-090"):
            right_lidar.append(r)

    front_dist_from_lidar = min(front_lidar) if front_lidar else None
    front_dist_from_vision = _median_or_none(obstacle_dists)
    front_dist_m = _median_or_none([front_dist_from_lidar, front_dist_from_vision])
    front_dist_m = _lowpass(_AUTO_STATE["front_dist_m"], front_dist_m, 0.45)
    front_rel_speed_kmh = _median_or_none(obstacle_rel_speeds)
    front_rel_speed_kmh = _lowpass(_AUTO_STATE["front_rel_speed_kmh"], front_rel_speed_kmh, 0.35)

    target_speed_kmh = state.get("target_speed_kmh", 100.0)
    target_speed_kmh = target_speed_kmh if _plausible(target_speed_kmh, 0.0, 180.0) else 100.0
    if _plausible(speed_limit_kmh, 20.0, 140.0):
        target_speed_kmh = min(target_speed_kmh, speed_limit_kmh)

    mu_speed_scale = _clamp(math.sqrt(max(0.08, mu) / 0.9), 0.35, 1.0)
    if rain:
        mu_speed_scale *= 0.9
    target_speed_kmh *= mu_speed_scale

    curv_abs = abs(curvature_1pm)
    if curv_abs > 1e-4:
        lat_acc_limit = min(2.8, 0.33 * mu * 9.81 + 0.8)
        curve_speed_ms = math.sqrt(max(0.0, lat_acc_limit / curv_abs))
        target_speed_kmh = min(target_speed_kmh, curve_speed_ms * 3.6)

    current_lane = int(state.get("current_lane", 1)) if _is_num(state.get("current_lane", 1)) else 1
    target_lane = int(state.get("target_lane", current_lane)) if _is_num(state.get("target_lane", current_lane)) else current_lane
    current_lane = int(_clamp(current_lane, 0, 2))
    target_lane = int(_clamp(target_lane, 0, 2))

    front_rel_speed_ms = (front_rel_speed_kmh or 0.0) / 3.6
    obstacle_speed_ms = max(0.0, speed_ms + front_rel_speed_ms)
    follow_gap_m = 8.0 + speed_ms * (1.4 + (0.8 if mu < 0.45 else 0.0))
    ttc = 999.0
    if front_dist_m is not None and front_rel_speed_ms < -0.1:
        ttc = front_dist_m / max(0.1, -front_rel_speed_ms)

    emergency = False
    caution = False
    if front_dist_m is not None:
        if front_dist_m < max(6.0, speed_ms * 0.55) or ttc < (1.6 if mu > 0.45 else 2.4):
            emergency = True
        elif front_dist_m < follow_gap_m or ttc < (3.2 if mu > 0.45 else 4.2):
            caution = True
            target_speed_kmh = min(target_speed_kmh, obstacle_speed_ms * 3.6)

    left_clear = (min(left_lidar) if left_lidar else 999.0) > max(10.0, speed_ms * 0.6)
    right_clear = (min(right_lidar) if right_lidar else 999.0) > max(10.0, speed_ms * 0.6)

    lane_change_allowed = vision_good >= 1 and mu > 0.28 and speed_ms < 42.0
    desired_lane = target_lane
    if emergency and lane_change_allowed:
        if target_lane > current_lane and left_clear:
            desired_lane = current_lane + 1
            emergency = False
        elif target_lane < current_lane and right_clear:
            desired_lane = current_lane - 1
            emergency = False
        elif left_clear and current_lane < 2:
            desired_lane = current_lane + 1
            emergency = False
        elif right_clear and current_lane > 0:
            desired_lane = current_lane - 1
            emergency = False

    lane_delta = desired_lane - current_lane
    desired_offset_m = _clamp(lane_delta * lane_width_m, -1.2 * lane_width_m, 1.2 * lane_width_m)
    lane_error_m = desired_offset_m - lane_offset_m

    crosswind_term = _clamp(lat_acc_ms2 * 0.06, -0.15, 0.15) if abs(yaw_rate_dps) < 4.0 else 0.0
    steer_limit = 0.85 if mu > 0.45 else 0.55 if mu > 0.28 else 0.38
    kp_lane = 0.34 if vision_good else 0.18
    kd_yaw = 0.024 if mu > 0.35 else 0.016
    ff_curve = _clamp(curvature_1pm * (22.0 + 0.25 * speed_ms * speed_ms), -0.4, 0.4)
    _AUTO_STATE["steer_i"] = _clamp(_AUTO_STATE["steer_i"] + lane_error_m * dt * 0.18, -0.25, 0.25)
    steering = kp_lane * lane_error_m - kd_yaw * yaw_rate_dps + ff_curve + _AUTO_STATE["steer_i"] + crosswind_term
    steering = _clamp(steering, -steer_limit, steer_limit)

    target_speed_ms = max(0.0, target_speed_kmh / 3.6)
    speed_error = target_speed_ms - speed_ms
    accel_cmd = 0.55 * speed_error + 0.03 * (0.0 - lon_acc_ms2)
    if caution and front_dist_m is not None:
        gap_error = front_dist_m - follow_gap_m
        accel_cmd = min(accel_cmd, 0.20 * gap_error + 0.35 * front_rel_speed_ms)
    if emergency:
        accel_cmd = min(accel_cmd, -6.5 if mu > 0.4 else -4.0)

    accel_up = 1.6 if mu > 0.45 else 0.9 if mu > 0.28 else 0.5
    accel_down = -3.2 if mu > 0.45 else -2.0 if mu > 0.28 else -1.2
    accel_cmd = _clamp(accel_cmd, accel_down, accel_up)

    throttle = 0.0
    brake = 0.0
    if accel_cmd >= 0.0:
        throttle = _clamp(accel_cmd / max(0.4, accel_up), 0.0, 1.0)
        if abs(steering) > 0.55:
            throttle *= 0.65
    else:
        brake = _clamp((-accel_cmd) / max(1.0, -accel_down), 0.0, 1.0)

    if mu < 0.28:
        throttle = min(throttle, 0.28)
        brake = min(brake, 0.55)
    elif mu < 0.45:
        throttle = min(throttle, 0.45)
        brake = min(brake, 0.75)

    brake_pressures = []
    for idx in (1, 2):
        valid = bool(state.get(f"brake_pressure_{idx}_valid", True))
        p = state.get(f"brake_pressure_{idx}_pct")
        if valid and _plausible(p, 0.0, 100.0):
            brake_pressures.append(p)
    brake_pressure_pct = _median_or_none(brake_pressures)
    if brake > 0.45 and (brake_pressure_pct is None or brake_pressure_pct < 8.0):
        _AUTO_STATE["brake_weak_counter"] += 1
    else:
        _AUTO_STATE["brake_weak_counter"] = max(0, _AUTO_STATE["brake_weak_counter"] - 1)
    if _AUTO_STATE["brake_weak_counter"] > 4:
        throttle = 0.0
        brake = 1.0
        if emergency and lane_change_allowed:
            if left_clear and current_lane < 2:
                steering = max(steering, 0.35)
            elif right_clear and current_lane > 0:
                steering = min(steering, -0.35)

    if _AUTO_STATE["last_throttle"] < 0.1 and speed_error < -1.0 and speed_ms > target_speed_ms + 1.5:
        _AUTO_STATE["throttle_stuck_counter"] += 1
    else:
        _AUTO_STATE["throttle_stuck_counter"] = max(0, _AUTO_STATE["throttle_stuck_counter"] - 1)
    if _AUTO_STATE["throttle_stuck_counter"] > 6:
        throttle = 0.0
        brake = max(brake, 0.35 if mu > 0.35 else 0.2)

    steer_angles = []
    for idx in (1, 2):
        valid = bool(state.get(f"steering_angle_{idx}_valid", True))
        a = state.get(f"steering_angle_{idx}_deg")
        if valid and _plausible(a, -720.0, 720.0):
            steer_angles.append(a)
    if len(steer_angles) == 2 and abs(steer_angles[0] - steer_angles[1]) > 10.0:
        steering *= 0.7

    _AUTO_STATE["speed_ms"] = speed_ms
    _AUTO_STATE["yaw_rate_dps"] = yaw_rate_dps
    _AUTO_STATE["heading_deg"] = gps_heading if gps_heading is not None else _AUTO_STATE["heading_deg"]
    _AUTO_STATE["lane_offset_m"] = lane_offset_m
    _AUTO_STATE["lane_width_m"] = lane_width_m
    _AUTO_STATE["curvature_1pm"] = curvature_1pm
    _AUTO_STATE["speed_limit_kmh"] = speed_limit_kmh
    _AUTO_STATE["front_dist_m"] = front_dist_m
    _AUTO_STATE["front_rel_speed_kmh"] = front_rel_speed_kmh
    _AUTO_STATE["last_steering"] = steering
    _AUTO_STATE["last_throttle"] = throttle
    _AUTO_STATE["last_brake"] = brake

    return {
        "steering": float(_clamp(steering, -1.0, 1.0)),
        "throttle": float(_clamp(throttle, 0.0, 1.0)),
        "brake": float(_clamp(brake, 0.0, 1.0)),
    }
