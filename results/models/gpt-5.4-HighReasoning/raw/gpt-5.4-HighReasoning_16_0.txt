#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <random>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    int duration = 0;
};

struct Instance {
    int jobs_count = 0;
    int machines_count = 0;
    int total_tasks = 0;
    vector<vector<Task>> jobs;
    vector<vector<long long>> remaining_work;
    vector<long long> machine_total_load;
};

struct Schedule {
    vector<vector<long long>> starts;
    long long makespan = numeric_limits<long long>::max();
};

enum class Rule {
    Hybrid,
    MWKR,
    SPT,
    LPT,
    ECT,
    MOPNR,
    Bottleneck,
    RandomizedHybrid
};

struct Candidate {
    int job = -1;
    int index = -1;
    int machine = -1;
    int duration = 0;
    int remaining_ops = 0;
    long long est = 0;
    long long ect = 0;
    long long remaining_work = 0;
    long long machine_load = 0;
};

static bool better_candidate(const Candidate& a, const Candidate& b, Rule rule, uint64_t seed_mix = 0) {
    auto finish_compare = [&]() {
        if (a.ect != b.ect) return a.ect < b.ect;
        if (a.est != b.est) return a.est < b.est;
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        return a.job < b.job;
    };

    switch (rule) {
    case Rule::Hybrid:
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        if (a.machine_load != b.machine_load) return a.machine_load > b.machine_load;
        if (a.est != b.est) return a.est < b.est;
        if (a.duration != b.duration) return a.duration > b.duration;
        return a.job < b.job;
    case Rule::MWKR:
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        if (a.duration != b.duration) return a.duration > b.duration;
        if (a.est != b.est) return a.est < b.est;
        return a.job < b.job;
    case Rule::SPT:
        if (a.duration != b.duration) return a.duration < b.duration;
        if (a.est != b.est) return a.est < b.est;
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        return a.job < b.job;
    case Rule::LPT:
        if (a.duration != b.duration) return a.duration > b.duration;
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        if (a.est != b.est) return a.est < b.est;
        return a.job < b.job;
    case Rule::ECT:
        return finish_compare();
    case Rule::MOPNR:
        if (a.remaining_ops != b.remaining_ops) return a.remaining_ops > b.remaining_ops;
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        if (a.est != b.est) return a.est < b.est;
        return a.job < b.job;
    case Rule::Bottleneck:
        if (a.machine_load != b.machine_load) return a.machine_load > b.machine_load;
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        if (a.est != b.est) return a.est < b.est;
        return a.job < b.job;
    case Rule::RandomizedHybrid: {
        if (a.remaining_work != b.remaining_work) return a.remaining_work > b.remaining_work;
        if (a.machine_load != b.machine_load) return a.machine_load > b.machine_load;
        uint64_t ra = seed_mix ^ (uint64_t(a.job) * 0x9E3779B97F4A7C15ULL + uint64_t(a.index) * 0xBF58476D1CE4E5B9ULL);
        uint64_t rb = seed_mix ^ (uint64_t(b.job) * 0x9E3779B97F4A7C15ULL + uint64_t(b.index) * 0xBF58476D1CE4E5B9ULL);
        if ((ra & 1023ULL) != (rb & 1023ULL)) return (ra & 1023ULL) < (rb & 1023ULL);
        if (a.est != b.est) return a.est < b.est;
        if (a.duration != b.duration) return a.duration > b.duration;
        return a.job < b.job;
    }
    }
    return a.job < b.job;
}

static Instance read_instance() {
    Instance inst;
    cin >> inst.jobs_count >> inst.machines_count;
    inst.jobs.resize(inst.jobs_count);
    inst.remaining_work.resize(inst.jobs_count);
    inst.machine_total_load.assign(inst.machines_count, 0);

    for (int j = 0; j < inst.jobs_count; ++j) {
        int t;
        cin >> t;
        inst.jobs[j].resize(t);
        for (int i = 0; i < t; ++i) {
            cin >> inst.jobs[j][i].machine >> inst.jobs[j][i].duration;
            inst.machine_total_load[inst.jobs[j][i].machine] += inst.jobs[j][i].duration;
            ++inst.total_tasks;
        }
    }

    for (int j = 0; j < inst.jobs_count; ++j) {
        int t = static_cast<int>(inst.jobs[j].size());
        inst.remaining_work[j].assign(t + 1, 0);
        for (int i = t - 1; i >= 0; --i) {
            inst.remaining_work[j][i] = inst.remaining_work[j][i + 1] + inst.jobs[j][i].duration;
        }
    }

    return inst;
}

static long long compute_makespan(const vector<long long>& job_ready) {
    long long result = 0;
    for (long long v : job_ready) result = max(result, v);
    return result;
}

static Schedule build_schedule_gt(const Instance& inst, Rule rule, uint64_t seed = 0) {
    vector<int> next_index(inst.jobs_count, 0);
    vector<long long> job_ready(inst.jobs_count, 0);
    vector<long long> machine_ready(inst.machines_count, 0);

    Schedule sched;
    sched.starts.resize(inst.jobs_count);
    for (int j = 0; j < inst.jobs_count; ++j) {
        sched.starts[j].assign(inst.jobs[j].size(), 0);
    }

    vector<Candidate> candidates;
    candidates.reserve(inst.jobs_count);

    for (int placed = 0; placed < inst.total_tasks; ++placed) {
        candidates.clear();
        Candidate pivot;
        bool has_pivot = false;

        for (int j = 0; j < inst.jobs_count; ++j) {
            int idx = next_index[j];
            if (idx >= static_cast<int>(inst.jobs[j].size())) continue;

            const Task& task = inst.jobs[j][idx];
            Candidate c;
            c.job = j;
            c.index = idx;
            c.machine = task.machine;
            c.duration = task.duration;
            c.remaining_ops = static_cast<int>(inst.jobs[j].size()) - idx;
            c.remaining_work = inst.remaining_work[j][idx];
            c.machine_load = inst.machine_total_load[c.machine];
            c.est = max(job_ready[j], machine_ready[c.machine]);
            c.ect = c.est + c.duration;
            candidates.push_back(c);

            if (!has_pivot || c.ect < pivot.ect || (c.ect == pivot.ect && c.est < pivot.est) ||
                (c.ect == pivot.ect && c.est == pivot.est && c.job < pivot.job)) {
                pivot = c;
                has_pivot = true;
            }
        }

        Candidate chosen;
        bool has_chosen = false;
        for (const Candidate& c : candidates) {
            if (c.machine != pivot.machine) continue;
            if (c.est >= pivot.ect) continue;
            if (!has_chosen || better_candidate(c, chosen, rule, seed + static_cast<uint64_t>(placed) * 1315423911ULL)) {
                chosen = c;
                has_chosen = true;
            }
        }
        if (!has_chosen) chosen = pivot;

        long long start = max(job_ready[chosen.job], machine_ready[chosen.machine]);
        long long finish = start + chosen.duration;
        sched.starts[chosen.job][chosen.index] = start;
        job_ready[chosen.job] = finish;
        machine_ready[chosen.machine] = finish;
        ++next_index[chosen.job];
    }

    sched.makespan = compute_makespan(job_ready);
    return sched;
}

struct ExactSolver {
    const Instance& inst;
    vector<int> next_index;
    vector<long long> job_ready;
    vector<long long> machine_ready;
    vector<long long> machine_remaining_load;
    vector<vector<long long>> current_starts;
    Schedule best;

    explicit ExactSolver(const Instance& instance, Schedule initial_best)
        : inst(instance),
          next_index(instance.jobs_count, 0),
          job_ready(instance.jobs_count, 0),
          machine_ready(instance.machines_count, 0),
          machine_remaining_load(instance.machine_total_load),
          current_starts(instance.jobs_count),
          best(std::move(initial_best)) {
        for (int j = 0; j < inst.jobs_count; ++j) {
            current_starts[j].assign(inst.jobs[j].size(), 0);
        }
    }

    long long lower_bound() const {
        long long lb = 0;
        for (int j = 0; j < inst.jobs_count; ++j) {
            lb = max(lb, job_ready[j] + inst.remaining_work[j][next_index[j]]);
        }
        for (int m = 0; m < inst.machines_count; ++m) {
            lb = max(lb, machine_ready[m] + machine_remaining_load[m]);
        }
        return lb;
    }

    void dfs(int scheduled) {
        if (scheduled == inst.total_tasks) {
            long long mk = compute_makespan(job_ready);
            if (mk < best.makespan) {
                best.makespan = mk;
                best.starts = current_starts;
            }
            return;
        }

        if (lower_bound() >= best.makespan) return;

        vector<Candidate> branch;
        branch.reserve(inst.jobs_count);
        for (int j = 0; j < inst.jobs_count; ++j) {
            int idx = next_index[j];
            if (idx >= static_cast<int>(inst.jobs[j].size())) continue;
            const Task& task = inst.jobs[j][idx];
            Candidate c;
            c.job = j;
            c.index = idx;
            c.machine = task.machine;
            c.duration = task.duration;
            c.remaining_ops = static_cast<int>(inst.jobs[j].size()) - idx;
            c.remaining_work = inst.remaining_work[j][idx];
            c.machine_load = inst.machine_total_load[c.machine];
            c.est = max(job_ready[j], machine_ready[c.machine]);
            c.ect = c.est + c.duration;
            branch.push_back(c);
        }

        sort(branch.begin(), branch.end(), [&](const Candidate& a, const Candidate& b) {
            if (a.ect != b.ect) return a.ect < b.ect;
            return better_candidate(a, b, Rule::Hybrid);
        });

        for (const Candidate& c : branch) {
            long long start = max(job_ready[c.job], machine_ready[c.machine]);
            long long finish = start + c.duration;
            if (finish >= best.makespan) continue;

            int saved_next = next_index[c.job];
            long long saved_job_ready = job_ready[c.job];
            long long saved_machine_ready = machine_ready[c.machine];

            current_starts[c.job][c.index] = start;
            ++next_index[c.job];
            job_ready[c.job] = finish;
            machine_ready[c.machine] = finish;
            machine_remaining_load[c.machine] -= c.duration;

            dfs(scheduled + 1);

            machine_remaining_load[c.machine] += c.duration;
            machine_ready[c.machine] = saved_machine_ready;
            job_ready[c.job] = saved_job_ready;
            next_index[c.job] = saved_next;
        }
    }
};

static Schedule solve_adaptive(const Instance& inst) {
    vector<pair<Rule, uint64_t>> runs;

    if (inst.total_tasks <= 40) {
        runs = {
            {Rule::Hybrid, 1}, {Rule::MWKR, 2}, {Rule::ECT, 3}, {Rule::SPT, 4},
            {Rule::Bottleneck, 5}, {Rule::LPT, 6}, {Rule::MOPNR, 7},
            {Rule::RandomizedHybrid, 8}, {Rule::RandomizedHybrid, 9},
            {Rule::RandomizedHybrid, 10}, {Rule::RandomizedHybrid, 11}, {Rule::RandomizedHybrid, 12}
        };
    } else if (inst.total_tasks <= 1500) {
        runs = {
            {Rule::Hybrid, 1}, {Rule::MWKR, 2}, {Rule::ECT, 3}, {Rule::SPT, 4},
            {Rule::Bottleneck, 5}, {Rule::RandomizedHybrid, 6},
            {Rule::RandomizedHybrid, 7}, {Rule::RandomizedHybrid, 8}
        };
    } else if (inst.total_tasks <= 12000) {
        runs = {
            {Rule::Hybrid, 1}, {Rule::MWKR, 2}, {Rule::ECT, 3},
            {Rule::Bottleneck, 4}, {Rule::RandomizedHybrid, 5}
        };
    } else {
        runs = {
            {Rule::Hybrid, 1}, {Rule::MWKR, 2}, {Rule::Bottleneck, 3}
        };
    }

    Schedule best;
    bool has_best = false;
    for (const auto& [rule, seed] : runs) {
        Schedule cand = build_schedule_gt(inst, rule, seed);
        if (!has_best || cand.makespan < best.makespan) {
            best = std::move(cand);
            has_best = true;
        }
    }

    if (inst.total_tasks <= 16) {
        ExactSolver exact(inst, best);
        exact.dfs(0);
        best = std::move(exact.best);
    }

    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance inst = read_instance();
    Schedule best = solve_adaptive(inst);

    for (int j = 0; j < inst.jobs_count; ++j) {
        for (int i = 0; i < static_cast<int>(inst.jobs[j].size()); ++i) {
            if (i) cout << ' ';
            cout << best.starts[j][i] << ' ' << inst.jobs[j][i].machine;
        }
        cout << '\n';
    }

    return 0;
}
