using System;
using System.Buffers;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const byte NoDir = 255;

    private static readonly int[] Dx = { -1, 1, 0, 0 };
    private static readonly int[] Dy = { 0, 0, -1, 1 };

    public static void Main()
    {
        var reader = new BufferedByteReader(Console.OpenStandardInput());
        string header = reader.ReadHeaderLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine(0);
            return;
        }

        string[] parts = header.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6)
        {
            Console.WriteLine(0);
            return;
        }

        int xCount = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int yCount = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int zCount = int.Parse(parts[2], CultureInfo.InvariantCulture);

        long areaLong = (long)xCount * yCount;
        if (areaLong <= 0 || areaLong > int.MaxValue)
        {
            Console.WriteLine(0);
            return;
        }

        int area = (int)areaLong;
        var topHeight = new int[area];
        var topByte = new byte[area];
        Array.Fill(topHeight, -1);

        for (int x = 0; x < xCount; x++)
        {
            int baseIndex = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                int idx = baseIndex + y;
                int highest = -1;
                byte highestByte = 0;
                for (int z = 0; z < zCount; z++)
                {
                    byte d = reader.ReadByteValue();
                    int mass = d & 15;
                    if (mass > 0)
                    {
                        highest = z;
                        highestByte = d;
                    }
                }

                topHeight[idx] = highest;
                topByte[idx] = highestByte;
            }
        }

        PlannerMode mode = ChooseMode(area, zCount);
        var candidates = BuildCandidates(topHeight, topByte, xCount, yCount, mode);
        var currentHeight = (int[])topHeight.Clone();

        byte[] parentDir = Array.Empty<byte>();
        if (mode != PlannerMode.LargeMonotone)
        {
            parentDir = BuildReverseTree(topHeight, xCount, yCount);
        }

        var operations = new List<Operation>(candidates.Count);

        foreach (var candidate in candidates)
        {
            int idx = candidate.Index;
            if (currentHeight[idx] != candidate.Z)
            {
                continue;
            }

            if (candidate.X == 0 && candidate.Y == 0)
            {
                operations.Add(new Operation(candidate.X, candidate.Y, candidate.Z, Array.Empty<int>()));
                currentHeight[idx]--;
                continue;
            }

            PathResult result;
            if (mode == PlannerMode.LargeMonotone)
            {
                result = TryMonotoneRoutes(candidate, currentHeight, xCount, yCount);
            }
            else
            {
                result = TryTreeRoute(candidate, currentHeight, yCount, parentDir);
                if (!result.Success)
                {
                    result = TryAStar(candidate, currentHeight, xCount, yCount);
                }
            }

            if (!result.Success || candidate.GrossRevenue <= result.FuelCost)
            {
                continue;
            }

            operations.Add(new Operation(candidate.X, candidate.Y, candidate.Z, result.Route));
            currentHeight[idx]--;
        }

        WriteOutput(operations, yCount);
    }

    private static PlannerMode ChooseMode(int area, int zCount)
    {
        if (area <= 250_000 && zCount <= 2_000)
        {
            return PlannerMode.SmallExact;
        }

        if (area <= 2_000_000)
        {
            return PlannerMode.MediumExact;
        }

        return PlannerMode.LargeMonotone;
    }

    private static List<Candidate> BuildCandidates(int[] topHeight, byte[] topByte, int xCount, int yCount, PlannerMode mode)
    {
        int maxCandidates = mode switch
        {
            PlannerMode.SmallExact => 12000,
            PlannerMode.MediumExact => 4000,
            _ => 1200
        };

        int minValue = mode == PlannerMode.LargeMonotone ? 4 : 1;

        var list = new List<Candidate>();
        int area = topHeight.Length;
        for (int idx = 0; idx < area; idx++)
        {
            int z = topHeight[idx];
            if (z < 0)
            {
                continue;
            }

            byte d = topByte[idx];
            int mass = d & 15;
            int value = d >> 4;
            if (mass <= 0 || value < minValue)
            {
                continue;
            }

            int x = idx / yCount;
            int y = idx - (x * yCount);
            long gross = (long)mass * value * 1000L;
            int manhattan = x + y;
            long roughProfit = gross - manhattan;
            if (x != 0 || y != 0)
            {
                if (roughProfit <= 0)
                {
                    continue;
                }
            }

            list.Add(new Candidate(idx, x, y, z, mass, value, gross, roughProfit));
        }

        list.Sort(static (a, b) =>
        {
            int byProfit = b.RoughProfit.CompareTo(a.RoughProfit);
            if (byProfit != 0) return byProfit;
            int byValue = b.Value.CompareTo(a.Value);
            if (byValue != 0) return byValue;
            int byMass = b.Mass.CompareTo(a.Mass);
            if (byMass != 0) return byMass;
            return a.Index.CompareTo(b.Index);
        });

        if (list.Count > maxCandidates)
        {
            list.RemoveRange(maxCandidates, list.Count - maxCandidates);
        }

        return list;
    }

    private static byte[] BuildReverseTree(int[] height, int xCount, int yCount)
    {
        int area = height.Length;
        var dist = new int[area];
        var parentDir = new byte[area];
        Array.Fill(dist, int.MaxValue);
        Array.Fill(parentDir, NoDir);

        var pq = new PriorityQueue<int, int>();
        dist[0] = 0;
        pq.Enqueue(0, 0);

        while (pq.TryDequeue(out int current, out int currentDist))
        {
            if (currentDist != dist[current])
            {
                continue;
            }

            int cx = current / yCount;
            int cy = current - (cx * yCount);
            int ch = height[current];

            for (int dir = 0; dir < 4; dir++)
            {
                int nx = cx + Dx[dir];
                int ny = cy + Dy[dir];
                if ((uint)nx >= (uint)xCount || (uint)ny >= (uint)yCount)
                {
                    continue;
                }

                int next = nx * yCount + ny;
                int nh = height[next];
                if (next != 0 && nh < 0)
                {
                    continue;
                }

                if (ch - nh > 1)
                {
                    continue;
                }

                int stepCost = ForwardCost(nh, ch);
                int nd = currentDist + stepCost;
                if (nd < dist[next])
                {
                    dist[next] = nd;
                    parentDir[next] = EncodeDir(-Dx[dir], -Dy[dir]);
                    pq.Enqueue(next, nd);
                }
            }
        }

        return parentDir;
    }

    private static PathResult TryTreeRoute(Candidate candidate, int[] currentHeight, int yCount, byte[] parentDir)
    {
        int source = candidate.Index;
        if (source < 0 || source >= parentDir.Length || parentDir[source] == NoDir)
        {
            return PathResult.Failure;
        }

        int current = source;
        int currentZ = candidate.Z;
        int fuel = 0;
        var route = new List<int>();
        var seen = new HashSet<int>();

        while (current != 0)
        {
            if (!seen.Add(current))
            {
                return PathResult.Failure;
            }

            byte dir = parentDir[current];
            if (dir == NoDir)
            {
                return PathResult.Failure;
            }

            int x = current / yCount;
            int y = current - (x * yCount);
            int nx = x + DirDx(dir);
            int ny = y + DirDy(dir);
            int next = nx * yCount + ny;
            int nextZ = currentHeight[next];
            if (next != 0 && nextZ < 0)
            {
                return PathResult.Failure;
            }
            if (nextZ - currentZ > 1)
            {
                return PathResult.Failure;
            }

            fuel += ForwardCost(currentZ, nextZ);
            route.Add(next);
            current = next;
            currentZ = nextZ;
        }

        return new PathResult(true, fuel, route.ToArray());
    }

    private static PathResult TryMonotoneRoutes(Candidate candidate, int[] currentHeight, int xCount, int yCount)
    {
        PathResult best = PathResult.Failure;
        bool found = false;

        PathResult a = TryMonotone(candidate, currentHeight, yCount, xFirst: true);
        if (a.Success)
        {
            best = a;
            found = true;
        }

        PathResult b = TryMonotone(candidate, currentHeight, yCount, xFirst: false);
        if (b.Success && (!found || b.FuelCost < best.FuelCost))
        {
            best = b;
            found = true;
        }

        return found ? best : PathResult.Failure;
    }

    private static PathResult TryMonotone(Candidate candidate, int[] currentHeight, int yCount, bool xFirst)
    {
        int x = candidate.X;
        int y = candidate.Y;
        int currentZ = candidate.Z;
        int fuel = 0;
        var route = new List<int>(x + y);

        if (xFirst)
        {
            while (x > 0)
            {
                int nx = x - 1;
                int next = nx * yCount + y;
                int nextZ = currentHeight[next];
                if (next != 0 && nextZ < 0)
                {
                    return PathResult.Failure;
                }
                if (nextZ - currentZ > 1)
                {
                    return PathResult.Failure;
                }

                fuel += ForwardCost(currentZ, nextZ);
                route.Add(next);
                x = nx;
                currentZ = nextZ;
            }

            while (y > 0)
            {
                int ny = y - 1;
                int next = yCount == 0 ? 0 : x * yCount + ny;
                int nextZ = currentHeight[next];
                if (next != 0 && nextZ < 0)
                {
                    return PathResult.Failure;
                }
                if (nextZ - currentZ > 1)
                {
                    return PathResult.Failure;
                }

                fuel += ForwardCost(currentZ, nextZ);
                route.Add(next);
                y = ny;
                currentZ = nextZ;
            }
        }
        else
        {
            while (y > 0)
            {
                int ny = y - 1;
                int next = x * yCount + ny;
                int nextZ = currentHeight[next];
                if (next != 0 && nextZ < 0)
                {
                    return PathResult.Failure;
                }
                if (nextZ - currentZ > 1)
                {
                    return PathResult.Failure;
                }

                fuel += ForwardCost(currentZ, nextZ);
                route.Add(next);
                y = ny;
                currentZ = nextZ;
            }

            while (x > 0)
            {
                int nx = x - 1;
                int next = nx * yCount + y;
                int nextZ = currentHeight[next];
                if (next != 0 && nextZ < 0)
                {
                    return PathResult.Failure;
                }
                if (nextZ - currentZ > 1)
                {
                    return PathResult.Failure;
                }

                fuel += ForwardCost(currentZ, nextZ);
                route.Add(next);
                x = nx;
                currentZ = nextZ;
            }
        }

        return new PathResult(true, fuel, route.ToArray());
    }

    private static PathResult TryAStar(Candidate candidate, int[] currentHeight, int xCount, int yCount)
    {
        int source = candidate.Index;
        if (source == 0)
        {
            return new PathResult(true, 0, Array.Empty<int>());
        }

        var open = new PriorityQueue<int, int>();
        var best = new Dictionary<int, int>(1024);
        var previous = new Dictionary<int, int>(1024);
        int startHeuristic = candidate.X + candidate.Y;
        best[source] = 0;
        open.Enqueue(source, startHeuristic);

        while (open.TryDequeue(out int current, out _))
        {
            int currentG = best[current];
            if (current == 0)
            {
                return ReconstructAStar(previous, currentG, source);
            }

            int cx = current / yCount;
            int cy = current - (cx * yCount);
            int currentZ = current == source ? candidate.Z : currentHeight[current];

            for (int dir = 0; dir < 4; dir++)
            {
                int nx = cx + Dx[dir];
                int ny = cy + Dy[dir];
                if ((uint)nx >= (uint)xCount || (uint)ny >= (uint)yCount)
                {
                    continue;
                }

                int next = nx * yCount + ny;
                int nextZ = currentHeight[next];
                if (next != 0 && nextZ < 0)
                {
                    continue;
                }
                if (nextZ - currentZ > 1)
                {
                    continue;
                }

                int tentative = currentG + ForwardCost(currentZ, nextZ);
                if (best.TryGetValue(next, out int old) && old <= tentative)
                {
                    continue;
                }

                best[next] = tentative;
                previous[next] = current;
                int priority = tentative + nx + ny;
                open.Enqueue(next, priority);
            }
        }

        return PathResult.Failure;
    }

    private static PathResult ReconstructAStar(Dictionary<int, int> previous, int fuelCost, int source)
    {
        var reversed = new List<int>();
        int current = 0;
        while (current != source)
        {
            reversed.Add(current);
            if (!previous.TryGetValue(current, out current))
            {
                return PathResult.Failure;
            }
        }

        reversed.Reverse();
        return new PathResult(true, fuelCost, reversed.ToArray());
    }

    private static int ForwardCost(int fromZ, int toZ)
    {
        if (toZ > fromZ) return 5;
        if (toZ == fromZ) return 1;
        return 0;
    }

    private static byte EncodeDir(int dx, int dy)
    {
        if (dx == -1) return 0;
        if (dx == 1) return 1;
        if (dy == -1) return 2;
        return 3;
    }

    private static int DirDx(byte dir) => dir switch
    {
        0 => -1,
        1 => 1,
        _ => 0
    };

    private static int DirDy(byte dir) => dir switch
    {
        2 => -1,
        3 => 1,
        _ => 0
    };

    private static void WriteOutput(List<Operation> operations, int yCount)
    {
        var sb = new StringBuilder();
        sb.AppendLine(operations.Count.ToString(CultureInfo.InvariantCulture));

        foreach (var op in operations)
        {
            int routeCount = op.Route.Length;
            int intermediateCount = routeCount == 0 ? 0 : routeCount - 1;
            sb.Append(intermediateCount.ToString(CultureInfo.InvariantCulture));
            sb.Append(' ');
            sb.Append(op.X.ToString(CultureInfo.InvariantCulture));
            sb.Append(' ');
            sb.Append(op.Y.ToString(CultureInfo.InvariantCulture));
            sb.Append(' ');
            sb.Append(op.Z.ToString(CultureInfo.InvariantCulture));

            for (int i = 0; i < intermediateCount; i++)
            {
                int cell = op.Route[i];
                sb.Append(' ');
                sb.Append((cell / yCount).ToString(CultureInfo.InvariantCulture));
                sb.Append(' ');
                sb.Append((cell % yCount).ToString(CultureInfo.InvariantCulture));
            }

            int dest = routeCount == 0 ? 0 : op.Route[routeCount - 1];
            sb.Append(' ');
            sb.Append((dest / yCount).ToString(CultureInfo.InvariantCulture));
            sb.Append(' ');
            sb.Append((dest % yCount).ToString(CultureInfo.InvariantCulture));
            sb.AppendLine();
        }

        Console.Write(sb.ToString());
    }

    private enum PlannerMode
    {
        SmallExact,
        MediumExact,
        LargeMonotone
    }

    private readonly record struct Candidate(
        int Index,
        int X,
        int Y,
        int Z,
        int Mass,
        int Value,
        long GrossRevenue,
        long RoughProfit);

    private readonly record struct PathResult(bool Success, int FuelCost, int[] Route)
    {
        public static PathResult Failure => new(false, 0, Array.Empty<int>());
    }

    private readonly record struct Operation(int X, int Y, int Z, int[] Route);

    private sealed class BufferedByteReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _offset;
        private int _length;

        public BufferedByteReader(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = ArrayPool<byte>.Shared.Rent(bufferSize);
        }

        public string ReadHeaderLine()
        {
            var sb = new StringBuilder();
            while (true)
            {
                int b = ReadByteInternal();
                if (b < 0)
                {
                    break;
                }

                if (b == '\r')
                {
                    continue;
                }

                if (b == '\n')
                {
                    break;
                }

                sb.Append((char)b);
            }

            return sb.ToString();
        }

        public byte ReadByteValue()
        {
            int b = ReadByteInternal();
            return b < 0 ? (byte)0 : (byte)b;
        }

        private int ReadByteInternal()
        {
            if (_offset >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _offset = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_offset++];
        }
    }
}
