const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robots = Array.isArray(data.robots) ? data.robots.slice() : [];
const jobs = Array.isArray(data.jobs) ? data.jobs.slice() : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

const open = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row[x] !== '#') {
      open[y * width + x] = 1;
    }
  }
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function toIndex(x, y) {
  return y * width + x;
}

function bfsFrom(sourceIndex) {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);

  if (sourceIndex < 0 || sourceIndex >= cellCount || open[sourceIndex] === 0) {
    return dist;
  }

  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;
  dist[sourceIndex] = 0;
  queue[tail++] = sourceIndex;

  while (head < tail) {
    const idx = queue[head++];
    const base = dist[idx] + 1;
    const x = idx % width;
    const y = (idx / width) | 0;

    if (x > 0) {
      const next = idx - 1;
      if (open[next] && dist[next] === -1) {
        dist[next] = base;
        queue[tail++] = next;
      }
    }
    if (x + 1 < width) {
      const next = idx + 1;
      if (open[next] && dist[next] === -1) {
        dist[next] = base;
        queue[tail++] = next;
      }
    }
    if (y > 0) {
      const next = idx - width;
      if (open[next] && dist[next] === -1) {
        dist[next] = base;
        queue[tail++] = next;
      }
    }
    if (y + 1 < height) {
      const next = idx + width;
      if (open[next] && dist[next] === -1) {
        dist[next] = base;
        queue[tail++] = next;
      }
    }
  }

  return dist;
}

const distanceCache = new Map();

function getDistances(sourceIndex) {
  let cached = distanceCache.get(sourceIndex);
  if (!cached) {
    cached = bfsFrom(sourceIndex);
    distanceCache.set(sourceIndex, cached);
  }
  return cached;
}

const maxRobotId = robots.reduce((acc, robot) => Math.max(acc, robot.id), -1);
const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);
const currentPositions = new Map();
const travelTotals = new Map();

for (const robot of robots) {
  currentPositions.set(robot.id, { x: robot.x, y: robot.y });
  travelTotals.set(robot.id, 0);
}

jobs.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

for (const job of jobs) {
  const targetReachable = inBounds(job.x, job.y) ? toIndex(job.x, job.y) : -1;
  let bestRobotId = -1;
  let bestReachable = -1;
  let bestDistance = 0;
  let bestTravel = 0;

  for (const robot of robots) {
    const pos = currentPositions.get(robot.id);
    const sourceIndex = inBounds(pos.x, pos.y) ? toIndex(pos.x, pos.y) : -1;

    let reachable = 0;
    let distance = Number.MAX_SAFE_INTEGER;
    if (sourceIndex !== -1 && targetReachable !== -1) {
      const dist = getDistances(sourceIndex)[targetReachable];
      if (dist !== -1) {
        reachable = 1;
        distance = dist;
      }
    }

    const travel = travelTotals.get(robot.id) || 0;
    const isBetter =
      bestRobotId === -1 ||
      reachable > bestReachable ||
      (reachable === bestReachable && distance < bestDistance) ||
      (reachable === bestReachable && distance === bestDistance && travel < bestTravel) ||
      (reachable === bestReachable && distance === bestDistance && travel === bestTravel && robot.id < bestRobotId);

    if (isBetter) {
      bestRobotId = robot.id;
      bestReachable = reachable;
      bestDistance = distance;
      bestTravel = travel;
    }
  }

  if (bestRobotId === -1) {
    continue;
  }

  assignments[bestRobotId].push(job.id);
  currentPositions.set(bestRobotId, { x: job.x, y: job.y });
  if (bestReachable === 1 && bestDistance !== Number.MAX_SAFE_INTEGER) {
    travelTotals.set(bestRobotId, bestTravel + bestDistance);
  }
}

process.stdout.write(JSON.stringify({ assignments }));
