OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%3 = OpTypeFunction %void

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in_2 = OpTypePointer Input %arr_gl_PerVertex_in_2
%ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float_2 = OpTypePointer Input %arr_v3float_2
%ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_2 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_2 Input
%normal = OpVariable %ptr_Input_arr_v3float_2 Input
%color = OpVariable %ptr_Input_arr_v3float_2 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %3
%5 = OpLabel
%wp0_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_0
%wp1_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_1
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%p0 = OpLoad %v3float %wp0_ptr
%p1 = OpLoad %v3float %wp1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir2 = OpFSub %v2float %p1xy %p0xy
%dirx = OpCompositeExtract %float %dir2 0
%diry = OpCompositeExtract %float %dir2 1
%neg_diry = OpFNegate %float %diry
%perp_raw = OpCompositeConstruct %v2float %neg_diry %dirx
%perp = OpExtInst %v2float %1 Normalize %perp_raw
%shaft_off = OpVectorTimesScalar %v2float %perp %float_0_015
%arrow_off = OpVectorTimesScalar %v2float %perp %float_0_05
%delta3 = OpFSub %v3float %p1 %p0
%delta3_75 = OpVectorTimesScalar %v3float %delta3 %float_0_75
%mid3 = OpFAdd %v3float %p0 %delta3_75
%midxy = OpVectorShuffle %v2float %mid3 %mid3 0 1
%p0z = OpCompositeExtract %float %p0 2
%midz = OpCompositeExtract %float %mid3 2
%p1z = OpCompositeExtract %float %p1 2
%p0_plus = OpFAdd %v2float %p0xy %shaft_off
%p0_minus = OpFSub %v2float %p0xy %shaft_off
%mid_plus = OpFAdd %v2float %midxy %shaft_off
%mid_minus = OpFSub %v2float %midxy %shaft_off
%arrow_plus = OpFAdd %v2float %midxy %arrow_off
%arrow_minus = OpFSub %v2float %midxy %arrow_off
%gl_out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0
%p0_plus_x = OpCompositeExtract %float %p0_plus 0
%p0_plus_y = OpCompositeExtract %float %p0_plus 1
%p0_plus_pos = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0z %float_1
OpStore %gl_out_pos %p0_plus_pos
OpStore %outColor %c0
OpEmitVertex
%p0_minus_x = OpCompositeExtract %float %p0_minus 0
%p0_minus_y = OpCompositeExtract %float %p0_minus 1
%p0_minus_pos = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0z %float_1
OpStore %gl_out_pos %p0_minus_pos
OpStore %outColor %c0
OpEmitVertex
%mid_plus_x = OpCompositeExtract %float %mid_plus 0
%mid_plus_y = OpCompositeExtract %float %mid_plus 1
%mid_plus_pos = OpCompositeConstruct %v4float %mid_plus_x %mid_plus_y %midz %float_1
OpStore %gl_out_pos %mid_plus_pos
OpStore %outColor %c0
OpEmitVertex
%mid_minus_x = OpCompositeExtract %float %mid_minus 0
%mid_minus_y = OpCompositeExtract %float %mid_minus 1
%mid_minus_pos = OpCompositeConstruct %v4float %mid_minus_x %mid_minus_y %midz %float_1
OpStore %gl_out_pos %mid_minus_pos
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive
%arrow_plus_x = OpCompositeExtract %float %arrow_plus 0
%arrow_plus_y = OpCompositeExtract %float %arrow_plus 1
%arrow_plus_pos = OpCompositeConstruct %v4float %arrow_plus_x %arrow_plus_y %midz %float_1
OpStore %gl_out_pos %arrow_plus_pos
OpStore %outColor %c1
OpEmitVertex
%arrow_minus_x = OpCompositeExtract %float %arrow_minus 0
%arrow_minus_y = OpCompositeExtract %float %arrow_minus 1
%arrow_minus_pos = OpCompositeConstruct %v4float %arrow_minus_x %arrow_minus_y %midz %float_1
OpStore %gl_out_pos %arrow_minus_pos
OpStore %outColor %c1
OpEmitVertex
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%tip_pos = OpCompositeConstruct %v4float %p1x %p1y %p1z %float_1
OpStore %gl_out_pos %tip_pos
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
