using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double DaySeconds = 86400.0;
    private const double EarthRadius = 6_371_000.0;
    private const double RequiredEarthClearance = EarthRadius + 1_000_000.0;
    private const double InterceptTolerance = 1_000_000.0;

    public static void Main()
    {
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            Console.WriteLine("2");
            Console.WriteLine("0 0 0 0");
            Console.WriteLine("1 0 0 0");
            return;
        }

        var parser = new TokenParser(input);
        int numBodies = parser.NextInt();
        double warningDays = parser.NextDouble();
        double impactorMass = parser.NextDouble();
        double deltaVBudget = parser.NextDouble();
        double asteroidMass = parser.NextDouble();
        double t0UnixUtc = parser.NextDouble();
        _ = t0UnixUtc;

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            string name = parser.NextString();
            double mass = parser.NextDouble();
            var pos = new Vec3(parser.NextDouble(), parser.NextDouble(), parser.NextDouble());
            var vel = new Vec3(parser.NextDouble(), parser.NextDouble(), parser.NextDouble());
            bodies.Add(new Body(name, mass, pos, vel));
        }

        string bspPath = parser.HasMore ? parser.NextString() : string.Empty;
        _ = bspPath;

        int earthIndex = FindBodyIndex(bodies, "Earth");
        int asteroidIndex = FindBodyIndex(bodies, "Asteroid");
        int sunIndex = FindBodyIndex(bodies, "Sun");

        if (earthIndex < 0 || asteroidIndex < 0)
        {
            Console.WriteLine("2");
            Console.WriteLine("0 0 0 0");
            Console.WriteLine("1 0 0 0");
            return;
        }

        if (sunIndex < 0)
        {
            sunIndex = FindMostMassiveBodyIndex(bodies);
        }

        double horizonDays = Math.Max(warningDays + 30.0, 120.0);
        double ephemerisStepDays = Clamp(horizonDays / 4000.0, 0.25, 3.0);
        var ephemeris = BuildEphemeris(bodies, horizonDays, ephemerisStepDays);

        var closestApproach = FindClosestApproach(ephemeris, earthIndex, asteroidIndex, 0.0, horizonDays);
        double earthImpactTime = closestApproach.TimeDays;
        double nominalMissDistance = closestApproach.Distance;

        var plan = SearchMission(
            bodies,
            ephemeris,
            earthIndex,
            asteroidIndex,
            sunIndex,
            warningDays,
            earthImpactTime,
            nominalMissDistance,
            impactorMass,
            asteroidMass,
            deltaVBudget);

        EmitPlan(plan);
    }

    private static MissionPlan SearchMission(
        List<Body> bodies,
        Ephemeris ephemeris,
        int earthIndex,
        int asteroidIndex,
        int sunIndex,
        double warningDays,
        double earthImpactTime,
        double nominalMissDistance,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget)
    {
        double muSun = G * Math.Max(bodies[sunIndex].Mass, 1.0);
        double latestIntercept = Math.Min(earthImpactTime - 2.0, warningDays);
        if (latestIntercept < 5.0)
        {
            latestIntercept = Math.Max(1.5, warningDays * 0.9);
        }

        double earliestIntercept = Math.Min(2.0, latestIntercept * 0.4);
        earliestIntercept = Math.Max(0.5, earliestIntercept);

        MissionPlan best = MissionPlan.Failure();
        int interceptSamples = 40;
        int launchSamples = 26;

        for (int i = 0; i < interceptSamples; i++)
        {
            double a = interceptSamples == 1 ? 0.0 : (double)i / (interceptSamples - 1);
            a = Math.Pow(a, 0.8);
            double interceptTime = earliestIntercept + (latestIntercept - earliestIntercept) * a;
            double latestLaunch = Math.Min(warningDays, interceptTime - 0.75);
            if (latestLaunch <= 0.0)
            {
                continue;
            }

            for (int j = 0; j < launchSamples; j++)
            {
                double b = launchSamples == 1 ? 0.0 : (double)j / (launchSamples - 1);
                b = Math.Pow(b, 1.2);
                double launchTime = latestLaunch * b;
                double tofDays = interceptTime - launchTime;
                if (tofDays < 0.5)
                {
                    continue;
                }

                var candidate = BuildCandidate(
                    bodies,
                    ephemeris,
                    earthIndex,
                    asteroidIndex,
                    sunIndex,
                    muSun,
                    launchTime,
                    interceptTime,
                    earthImpactTime,
                    nominalMissDistance,
                    impactorMass,
                    asteroidMass,
                    deltaVBudget);

                if (candidate.Score > best.Score)
                {
                    best = candidate;
                }
            }
        }

        if (!best.IsValid)
        {
            return BuildFallbackPlan(ephemeris, earthIndex, asteroidIndex, warningDays);
        }

        return best;
    }

    private static MissionPlan BuildCandidate(
        List<Body> bodies,
        Ephemeris ephemeris,
        int earthIndex,
        int asteroidIndex,
        int sunIndex,
        double muSun,
        double launchTime,
        double interceptTime,
        double earthImpactTime,
        double nominalMissDistance,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget)
    {
        State earthLaunch = ephemeris.GetBodyState(earthIndex, launchTime);
        State asteroidIntercept = ephemeris.GetBodyState(asteroidIndex, interceptTime);
        State sunLaunch = ephemeris.GetBodyState(sunIndex, launchTime);
        State sunIntercept = ephemeris.GetBodyState(sunIndex, interceptTime);

        Vec3 r1Helio = earthLaunch.Position - sunLaunch.Position;
        Vec3 r2Helio = asteroidIntercept.Position - sunIntercept.Position;

        if (!LambertSolver.TrySolve(r1Helio, r2Helio, (interceptTime - launchTime) * DaySeconds, muSun, out var transfer))
        {
            return MissionPlan.Failure();
        }

        Vec3 launchVelocity = transfer.V1 + sunLaunch.Velocity;
        Vec3 dvLaunch = launchVelocity - earthLaunch.Velocity;
        double dv1 = dvLaunch.Length();
        if (dv1 > deltaVBudget)
        {
            return MissionPlan.Failure();
        }

        var sc0 = new State(earthLaunch.Position, earthLaunch.Velocity + dvLaunch);
        double midTime = launchTime + 0.5 * (interceptTime - launchTime);
        State scMid = PropagateSpacecraft(ephemeris, launchTime, sc0, midTime);
        State asteroidMid = ephemeris.GetBodyState(asteroidIndex, midTime);
        State sunMid = ephemeris.GetBodyState(sunIndex, midTime);

        if (!LambertSolver.TrySolve(
                scMid.Position - sunMid.Position,
                asteroidIntercept.Position - sunIntercept.Position,
                (interceptTime - midTime) * DaySeconds,
                muSun,
                out var secondLeg))
        {
            return MissionPlan.Failure();
        }

        Vec3 desiredMidVelocity = secondLeg.V1 + sunMid.Velocity;
        Vec3 dvMid = desiredMidVelocity - scMid.Velocity;
        double dv2 = dvMid.Length();

        if (dv1 + dv2 > deltaVBudget)
        {
            return MissionPlan.Failure();
        }

        State scAfterMid = new State(scMid.Position, scMid.Velocity + dvMid);

        double finalBurnTime = midTime + 0.85 * (interceptTime - midTime);
        finalBurnTime = Math.Min(interceptTime - 0.02, finalBurnTime);
        finalBurnTime = Math.Max(midTime + 0.02, finalBurnTime);

        State scPreFinal = PropagateSpacecraft(ephemeris, midTime, scAfterMid, finalBurnTime);
        State asteroidFinal = ephemeris.GetBodyState(asteroidIndex, interceptTime);
        double finalLegSeconds = Math.Max((interceptTime - finalBurnTime) * DaySeconds, 60.0);

        Vec3 lineOfSightVelocity = (asteroidFinal.Position - scPreFinal.Position) / finalLegSeconds;
        Vec3 dvFinal = lineOfSightVelocity - scPreFinal.Velocity;
        double dv3 = dvFinal.Length();

        double totalDv = dv1 + dv2 + dv3;
        if (totalDv > deltaVBudget)
        {
            return MissionPlan.Failure();
        }

        State scAfterFinal = new State(scPreFinal.Position, scPreFinal.Velocity + dvFinal);
        State scImpact = PropagateSpacecraft(ephemeris, finalBurnTime, scAfterFinal, interceptTime);

        Vec3 interceptError = scImpact.Position - asteroidIntercept.Position;
        double missDistance = interceptError.Length();
        if (missDistance > InterceptTolerance)
        {
            return MissionPlan.Failure();
        }

        Vec3 relativeVelocity = scImpact.Velocity - asteroidIntercept.Velocity;
        double relativeSpeed = relativeVelocity.Length();
        if (relativeSpeed < 1.0)
        {
            return MissionPlan.Failure();
        }

        double asteroidDeltaV = relativeSpeed * Math.Max(impactorMass, 1.0) / Math.Max(asteroidMass, 1.0);
        double timeToEarthSeconds = Math.Max(0.0, (earthImpactTime - interceptTime) * DaySeconds);
        double projectedDeflection = asteroidDeltaV * timeToEarthSeconds;
        double requiredDeflection = Math.Max(1.0, RequiredEarthClearance - nominalMissDistance);
        double score = projectedDeflection - requiredDeflection;

        if (score <= 0.0)
        {
            score -= 0.05 * totalDv;
        }
        else
        {
            score += 0.1 * relativeSpeed - 0.02 * totalDv - 0.001 * missDistance;
        }

        return new MissionPlan(
            true,
            score,
            new[]
            {
                new Burn(launchTime, dvLaunch),
                new Burn(midTime, dvMid),
                new Burn(finalBurnTime, dvFinal),
                new Burn(interceptTime, Vec3.Zero)
            });
    }

    private static MissionPlan BuildFallbackPlan(Ephemeris ephemeris, int earthIndex, int asteroidIndex, double warningDays)
    {
        State earth = ephemeris.GetBodyState(earthIndex, 0.0);
        State asteroid = ephemeris.GetBodyState(asteroidIndex, Math.Max(1.0, Math.Min(warningDays, 5.0)));
        Vec3 chase = asteroid.Position - earth.Position;
        double dt = Math.Max(1.0, Math.Min(warningDays, 5.0)) * DaySeconds;
        Vec3 dv = chase / dt + asteroid.Velocity - earth.Velocity;
        return new MissionPlan(
            true,
            -1e30,
            new[]
            {
                new Burn(0.0, dv),
                new Burn(Math.Max(1.0, Math.Min(warningDays, 5.0)), Vec3.Zero)
            });
    }

    private static ClosestApproach FindClosestApproach(Ephemeris ephemeris, int bodyA, int bodyB, double tMin, double tMax)
    {
        int samples = Math.Max(128, (int)Math.Ceiling((tMax - tMin) / ephemeris.StepDays));
        double bestT = tMin;
        double bestD = double.PositiveInfinity;

        for (int i = 0; i <= samples; i++)
        {
            double t = tMin + (tMax - tMin) * i / samples;
            double d = (ephemeris.GetBodyState(bodyA, t).Position - ephemeris.GetBodyState(bodyB, t).Position).Length();
            if (d < bestD)
            {
                bestD = d;
                bestT = t;
            }
        }

        double left = Math.Max(tMin, bestT - ephemeris.StepDays * 2.0);
        double right = Math.Min(tMax, bestT + ephemeris.StepDays * 2.0);

        for (int iter = 0; iter < 40; iter++)
        {
            double m1 = left + (right - left) / 3.0;
            double m2 = right - (right - left) / 3.0;
            double d1 = (ephemeris.GetBodyState(bodyA, m1).Position - ephemeris.GetBodyState(bodyB, m1).Position).LengthSquared();
            double d2 = (ephemeris.GetBodyState(bodyA, m2).Position - ephemeris.GetBodyState(bodyB, m2).Position).LengthSquared();
            if (d1 < d2)
            {
                right = m2;
            }
            else
            {
                left = m1;
            }
        }

        double time = 0.5 * (left + right);
        double dist = (ephemeris.GetBodyState(bodyA, time).Position - ephemeris.GetBodyState(bodyB, time).Position).Length();
        return new ClosestApproach(time, dist);
    }

    private static Ephemeris BuildEphemeris(List<Body> bodies, double horizonDays, double stepDays)
    {
        int n = bodies.Count;
        int steps = (int)Math.Ceiling(horizonDays / stepDays) + 1;
        var positions = new Vec3[n][];
        var velocities = new Vec3[n][];
        for (int i = 0; i < n; i++)
        {
            positions[i] = new Vec3[steps];
            velocities[i] = new Vec3[steps];
        }

        var pos = bodies.Select(b => b.Position).ToArray();
        var vel = bodies.Select(b => b.Velocity).ToArray();

        for (int s = 0; s < steps; s++)
        {
            for (int i = 0; i < n; i++)
            {
                positions[i][s] = pos[i];
                velocities[i][s] = vel[i];
            }

            if (s + 1 == steps)
            {
                break;
            }

            double dt = stepDays * DaySeconds;
            Rk4BodiesStep(bodies, pos, vel, dt);
        }

        return new Ephemeris(stepDays, horizonDays, bodies, positions, velocities);
    }

    private static void Rk4BodiesStep(List<Body> bodies, Vec3[] pos, Vec3[] vel, double dt)
    {
        int n = bodies.Count;
        var a1 = ComputeBodyAccelerations(bodies, pos);
        var p2 = new Vec3[n];
        var v2 = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            p2[i] = pos[i] + vel[i] * (0.5 * dt);
            v2[i] = vel[i] + a1[i] * (0.5 * dt);
        }

        var a2 = ComputeBodyAccelerations(bodies, p2);
        var p3 = new Vec3[n];
        var v3 = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            p3[i] = pos[i] + v2[i] * (0.5 * dt);
            v3[i] = vel[i] + a2[i] * (0.5 * dt);
        }

        var a3 = ComputeBodyAccelerations(bodies, p3);
        var p4 = new Vec3[n];
        var v4 = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            p4[i] = pos[i] + v3[i] * dt;
            v4[i] = vel[i] + a3[i] * dt;
        }

        var a4 = ComputeBodyAccelerations(bodies, p4);
        for (int i = 0; i < n; i++)
        {
            Vec3 dp = (vel[i] + v2[i] * 2.0 + v3[i] * 2.0 + v4[i]) * (dt / 6.0);
            Vec3 dv = (a1[i] + a2[i] * 2.0 + a3[i] * 2.0 + a4[i]) * (dt / 6.0);
            pos[i] += dp;
            vel[i] += dv;
        }
    }

    private static Vec3[] ComputeBodyAccelerations(List<Body> bodies, Vec3[] pos)
    {
        int n = bodies.Count;
        var acc = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            Vec3 ai = Vec3.Zero;
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                {
                    continue;
                }

                Vec3 d = pos[j] - pos[i];
                double r2 = d.LengthSquared() + 1.0;
                double invR = 1.0 / Math.Sqrt(r2);
                double invR3 = invR * invR * invR;
                ai += d * (G * bodies[j].Mass * invR3);
            }

            acc[i] = ai;
        }

        return acc;
    }

    private static State PropagateSpacecraft(Ephemeris ephemeris, double t0Days, State initial, double t1Days)
    {
        if (t1Days <= t0Days)
        {
            return initial;
        }

        double durationDays = t1Days - t0Days;
        int steps = Math.Max(8, (int)Math.Ceiling(durationDays / Math.Min(0.5, ephemeris.StepDays)));
        double dtDays = durationDays / steps;
        double dt = dtDays * DaySeconds;

        Vec3 p = initial.Position;
        Vec3 v = initial.Velocity;
        double t = t0Days;

        for (int i = 0; i < steps; i++)
        {
            var k1 = SpacecraftDerivative(ephemeris, t, p, v);
            var k2 = SpacecraftDerivative(ephemeris, t + 0.5 * dtDays, p + k1.Velocity * (0.5 * dt), v + k1.Acceleration * (0.5 * dt));
            var k3 = SpacecraftDerivative(ephemeris, t + 0.5 * dtDays, p + k2.Velocity * (0.5 * dt), v + k2.Acceleration * (0.5 * dt));
            var k4 = SpacecraftDerivative(ephemeris, t + dtDays, p + k3.Velocity * dt, v + k3.Acceleration * dt);

            p += (k1.Velocity + k2.Velocity * 2.0 + k3.Velocity * 2.0 + k4.Velocity) * (dt / 6.0);
            v += (k1.Acceleration + k2.Acceleration * 2.0 + k3.Acceleration * 2.0 + k4.Acceleration) * (dt / 6.0);
            t += dtDays;
        }

        return new State(p, v);
    }

    private static Derivative SpacecraftDerivative(Ephemeris ephemeris, double timeDays, Vec3 position, Vec3 velocity)
    {
        Vec3 acc = Vec3.Zero;
        for (int i = 0; i < ephemeris.BodyCount; i++)
        {
            State body = ephemeris.GetBodyState(i, timeDays);
            Vec3 d = body.Position - position;
            double r2 = d.LengthSquared() + 1.0;
            double invR = 1.0 / Math.Sqrt(r2);
            double invR3 = invR * invR * invR;
            acc += d * (G * ephemeris.GetBodyMass(i) * invR3);
        }

        return new Derivative(velocity, acc);
    }

    private static void EmitPlan(MissionPlan plan)
    {
        Console.WriteLine(plan.Burns.Length.ToString(CultureInfo.InvariantCulture));
        foreach (var burn in plan.Burns)
        {
            Console.WriteLine(string.Format(
                CultureInfo.InvariantCulture,
                "{0:R} {1:R} {2:R} {3:R}",
                burn.TimeDays,
                burn.DeltaV.X,
                burn.DeltaV.Y,
                burn.DeltaV.Z));
        }
    }

    private static int FindBodyIndex(List<Body> bodies, string name)
    {
        for (int i = 0; i < bodies.Count; i++)
        {
            if (string.Equals(bodies[i].Name, name, StringComparison.OrdinalIgnoreCase))
            {
                return i;
            }
        }

        return -1;
    }

    private static int FindMostMassiveBodyIndex(List<Body> bodies)
    {
        int best = 0;
        double bestMass = double.NegativeInfinity;
        for (int i = 0; i < bodies.Count; i++)
        {
            if (bodies[i].Mass > bestMass)
            {
                bestMass = bodies[i].Mass;
                best = i;
            }
        }

        return best;
    }

    private static double Clamp(double value, double min, double max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }

    private readonly record struct ClosestApproach(double TimeDays, double Distance);
    private readonly record struct Body(string Name, double Mass, Vec3 Position, Vec3 Velocity);
    private readonly record struct State(Vec3 Position, Vec3 Velocity);
    private readonly record struct Derivative(Vec3 Velocity, Vec3 Acceleration);
    private readonly record struct Burn(double TimeDays, Vec3 DeltaV);

    private readonly record struct MissionPlan(bool IsValid, double Score, Burn[] Burns)
    {
        public static MissionPlan Failure() => new MissionPlan(false, double.NegativeInfinity, Array.Empty<Burn>());
    }

    private sealed class Ephemeris
    {
        private readonly Vec3[][] _positions;
        private readonly Vec3[][] _velocities;
        private readonly double[] _masses;

        public Ephemeris(double stepDays, double horizonDays, List<Body> bodies, Vec3[][] positions, Vec3[][] velocities)
        {
            StepDays = stepDays;
            HorizonDays = horizonDays;
            _positions = positions;
            _velocities = velocities;
            BodyCount = positions.Length;
            _masses = new double[BodyCount];
            for (int i = 0; i < BodyCount; i++)
            {
                _masses[i] = bodies[i].Mass;
            }
        }

        public double StepDays { get; }
        public double HorizonDays { get; }
        public int BodyCount { get; }

        public double GetBodyMass(int index) => _masses[index];

        public State GetBodyState(int bodyIndex, double timeDays)
        {
            double t = Program.Clamp(timeDays, 0.0, HorizonDays);
            double f = t / StepDays;
            int i0 = (int)Math.Floor(f);
            int i1 = Math.Min(i0 + 1, _positions[bodyIndex].Length - 1);
            double a = f - i0;

            if (i0 >= _positions[bodyIndex].Length - 1)
            {
                i0 = _positions[bodyIndex].Length - 2;
                i1 = _positions[bodyIndex].Length - 1;
                a = 1.0;
            }

            Vec3 p = Vec3.Lerp(_positions[bodyIndex][i0], _positions[bodyIndex][i1], a);
            Vec3 v = Vec3.Lerp(_velocities[bodyIndex][i0], _velocities[bodyIndex][i1], a);
            return new State(p, v);
        }
    }

    private readonly record struct LambertResult(Vec3 V1, Vec3 V2);

    private static class LambertSolver
    {
        public static bool TrySolve(Vec3 r1, Vec3 r2, double dt, double mu, out LambertResult result)
        {
            result = default;
            if (dt <= 0.0 || mu <= 0.0)
            {
                return false;
            }

            bool okShort = TrySolveBranch(r1, r2, dt, mu, 1.0, out var shortWay);
            bool okLong = TrySolveBranch(r1, r2, dt, mu, -1.0, out var longWay);

            if (okShort && okLong)
            {
                double scoreShort = shortWay.V1.Length() + shortWay.V2.Length();
                double scoreLong = longWay.V1.Length() + longWay.V2.Length();
                result = scoreShort <= scoreLong ? shortWay : longWay;
                return true;
            }

            if (okShort)
            {
                result = shortWay;
                return true;
            }

            if (okLong)
            {
                result = longWay;
                return true;
            }

            return false;
        }

        private static bool TrySolveBranch(Vec3 r1, Vec3 r2, double dt, double mu, double sinSign, out LambertResult result)
        {
            result = default;

            double r1Mag = r1.Length();
            double r2Mag = r2.Length();
            if (r1Mag < 1.0 || r2Mag < 1.0)
            {
                return false;
            }

            double cosTheta = Clamp(r1.Dot(r2) / (r1Mag * r2Mag), -1.0, 1.0);
            double sinTheta = sinSign * Math.Sqrt(Math.Max(0.0, 1.0 - cosTheta * cosTheta));
            double oneMinusCos = 1.0 - cosTheta;
            if (Math.Abs(oneMinusCos) < 1e-12)
            {
                return false;
            }

            double a = sinTheta * Math.Sqrt(r1Mag * r2Mag / oneMinusCos);
            if (Math.Abs(a) < 1e-12)
            {
                return false;
            }

            double zMin = -16.0 * Math.PI * Math.PI;
            double zMax = 16.0 * Math.PI * Math.PI;
            double z = 0.0;

            for (int iter = 0; iter < 100; iter++)
            {
                z = 0.5 * (zMin + zMax);
                double c = StumpffC(z);
                double s = StumpffS(z);
                double y = r1Mag + r2Mag + a * ((z * s - 1.0) / Math.Sqrt(c));
                if (y <= 0.0)
                {
                    zMin = z;
                    continue;
                }

                double x = Math.Sqrt(y / c);
                double dtZ = (x * x * x * s + a * Math.Sqrt(y)) / Math.Sqrt(mu);
                if (Math.Abs(dtZ - dt) < 1e-6 * Math.Max(1.0, dt))
                {
                    break;
                }

                if (dtZ <= dt)
                {
                    zMin = z;
                }
                else
                {
                    zMax = z;
                }
            }

            double cFinal = StumpffC(z);
            double sFinal = StumpffS(z);
            double yFinal = r1Mag + r2Mag + a * ((z * sFinal - 1.0) / Math.Sqrt(cFinal));
            if (yFinal <= 0.0)
            {
                return false;
            }

            double f = 1.0 - yFinal / r1Mag;
            double g = a * Math.Sqrt(yFinal / mu);
            double gDot = 1.0 - yFinal / r2Mag;
            if (Math.Abs(g) < 1e-12)
            {
                return false;
            }

            Vec3 v1 = (r2 - r1 * f) / g;
            Vec3 v2 = (r2 * gDot - r1) / g;
            result = new LambertResult(v1, v2);
            return true;
        }

        private static double StumpffC(double z)
        {
            if (z > 1e-8)
            {
                double sz = Math.Sqrt(z);
                return (1.0 - Math.Cos(sz)) / z;
            }

            if (z < -1e-8)
            {
                double sz = Math.Sqrt(-z);
                return (1.0 - Math.Cosh(sz)) / z;
            }

            return 0.5 - z / 24.0 + z * z / 720.0;
        }

        private static double StumpffS(double z)
        {
            if (z > 1e-8)
            {
                double sz = Math.Sqrt(z);
                return (sz - Math.Sin(sz)) / (sz * sz * sz);
            }

            if (z < -1e-8)
            {
                double sz = Math.Sqrt(-z);
                return (Math.Sinh(sz) - sz) / (sz * sz * sz);
            }

            return 1.0 / 6.0 - z / 120.0 + z * z / 5040.0;
        }
    }

    private sealed class TokenParser
    {
        private readonly string[] _tokens;
        private int _index;

        public TokenParser(string input)
        {
            _tokens = input.Split((string[]?)null, StringSplitOptions.RemoveEmptyEntries);
            _index = 0;
        }

        public bool HasMore => _index < _tokens.Length;

        public string NextString()
        {
            return _tokens[_index++];
        }

        public int NextInt()
        {
            return int.Parse(NextString(), CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(NextString(), CultureInfo.InvariantCulture);
        }
    }

    public readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 Zero => new Vec3(0.0, 0.0, 0.0);

        public double Length() => Math.Sqrt(LengthSquared());
        public double LengthSquared() => X * X + Y * Y + Z * Z;
        public double Dot(Vec3 other) => X * other.X + Y * other.Y + Z * other.Z;

        public static Vec3 Lerp(Vec3 a, Vec3 b, double t)
        {
            return a + (b - a) * t;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
    }
