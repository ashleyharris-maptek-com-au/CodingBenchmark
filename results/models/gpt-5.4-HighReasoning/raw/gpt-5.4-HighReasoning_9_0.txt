using System;
using System.IO;

public static class Program
{
    private struct Cut
    {
        public int Length;
        public int Index;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }

    private sealed class FastWriter : IDisposable
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr;

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[11];
            int len = 0;
            while (value > 0)
            {
                tmp[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }

            for (int i = len - 1; i >= 0; --i)
            {
                WriteByte(tmp[i]);
            }
        }

        public void WriteSpace() => WriteByte((byte)' ');
        public void WriteLine() => WriteByte((byte)'\n');

        private void WriteByte(byte b)
        {
            if (_ptr == _buffer.Length)
            {
                Flush();
            }

            _buffer[_ptr++] = b;
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void Dispose() => Flush();
    }

    private sealed class Treap
    {
        private readonly long[] _key;
        private readonly int[] _left;
        private readonly int[] _right;
        private readonly uint[] _priority;
        private readonly uint[] _state;
        private int _root;
        private int _count;

        public Treap(int capacity)
        {
            _key = new long[capacity + 1];
            _left = new int[capacity + 1];
            _right = new int[capacity + 1];
            _priority = new uint[capacity + 1];
            _state = new uint[capacity + 1];
        }

        public int CreateNode(long key)
        {
            int id = ++_count;
            _key[id] = key;
            _left[id] = 0;
            _right[id] = 0;
            _priority[id] = NextRandom();
            return id;
        }

        public void Insert(int node)
        {
            _left[node] = 0;
            _right[node] = 0;
            _root = Insert(_root, node);
        }

        public void UpdateKey(int node, long oldKey, long newKey)
        {
            int removed = 0;
            _root = Erase(_root, oldKey, ref removed);
            _key[node] = newKey;
            _left[node] = 0;
            _right[node] = 0;
            _root = Insert(_root, node);
        }

        public long GetKey(int node) => _key[node];

        public int LowerBound(long target)
        {
            int cur = _root;
            int best = 0;
            while (cur != 0)
            {
                if (_key[cur] >= target)
                {
                    best = cur;
                    cur = _left[cur];
                }
                else
                {
                    cur = _right[cur];
                }
            }

            return best;
        }

        private int Insert(int root, int node)
        {
            if (root == 0) return node;

            if (_priority[node] > _priority[root])
            {
                Split(root, _key[node], out int leftTree, out int rightTree);
                _left[node] = leftTree;
                _right[node] = rightTree;
                return node;
            }

            if (_key[node] < _key[root])
            {
                _left[root] = Insert(_left[root], node);
            }
            else
            {
                _right[root] = Insert(_right[root], node);
            }

            return root;
        }

        private void Split(int root, long key, out int leftTree, out int rightTree)
        {
            if (root == 0)
            {
                leftTree = 0;
                rightTree = 0;
                return;
            }

            if (_key[root] < key)
            {
                Split(_right[root], key, out int splitLeft, out int splitRight);
                _right[root] = splitLeft;
                leftTree = root;
                rightTree = splitRight;
            }
            else
            {
                Split(_left[root], key, out int splitLeft, out int splitRight);
                _left[root] = splitRight;
                leftTree = splitLeft;
                rightTree = root;
            }
        }

        private int Erase(int root, long key, ref int removed)
        {
            if (root == 0) return 0;

            if (_key[root] == key)
            {
                removed = root;
                return Merge(_left[root], _right[root]);
            }

            if (key < _key[root])
            {
                _left[root] = Erase(_left[root], key, ref removed);
            }
            else
            {
                _right[root] = Erase(_right[root], key, ref removed);
            }

            return root;
        }

        private int Merge(int leftTree, int rightTree)
        {
            if (leftTree == 0) return rightTree;
            if (rightTree == 0) return leftTree;

            if (_priority[leftTree] > _priority[rightTree])
            {
                _right[leftTree] = Merge(_right[leftTree], rightTree);
                return leftTree;
            }

            _left[rightTree] = Merge(leftTree, _left[rightTree]);
            return rightTree;
        }

        private uint NextRandom()
        {
            uint x = _state[0] + 0x9E3779B9u;
            _state[0] = x;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            return x;
        }
    }

    private static long MakeKey(int remaining, int stockId)
    {
        return ((long)remaining << 32) | (uint)stockId;
    }

    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        int stockLength = scanner.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int length = scanner.NextInt();
            if (length > stockLength)
            {
                throw new InvalidOperationException("Cut length exceeds stock length.");
            }

            cuts[i] = new Cut { Length = length, Index = i };
        }

        Array.Sort(cuts, static (a, b) => b.Length.CompareTo(a.Length));

        var nextCut = new int[n];
        Array.Fill(nextCut, -1);

        var stockHead = new int[n];
        var stockTail = new int[n];
        var stockRemaining = new int[n];
        var stockNode = new int[n];
        Array.Fill(stockHead, -1);
        Array.Fill(stockTail, -1);

        var treap = new Treap(n);
        int stockCount = 0;

        for (int i = 0; i < n; i++)
        {
            int cutLength = cuts[i].Length;
            int cutIndex = cuts[i].Index;

            int node = treap.LowerBound((long)cutLength << 32);
            int stockId;

            if (node == 0)
            {
                stockId = stockCount++;
                int remaining = stockLength - cutLength;
                stockRemaining[stockId] = remaining;
                stockHead[stockId] = cutIndex;
                stockTail[stockId] = cutIndex;
                stockNode[stockId] = treap.CreateNode(MakeKey(remaining, stockId));
                treap.Insert(stockNode[stockId]);
            }
            else
            {
                long key = treap.GetKey(node);
                stockId = (int)(uint)key;

                int tail = stockTail[stockId];
                nextCut[tail] = cutIndex;
                stockTail[stockId] = cutIndex;

                int oldRemaining = stockRemaining[stockId];
                int newRemaining = oldRemaining - cutLength;
                stockRemaining[stockId] = newRemaining;
                treap.UpdateKey(stockNode[stockId], MakeKey(oldRemaining, stockId), MakeKey(newRemaining, stockId));
            }
        }

        using var writer = new FastWriter();
        writer.WriteInt(stockCount);
        writer.WriteLine();

        for (int stockId = 0; stockId < stockCount; stockId++)
        {
            int cut = stockHead[stockId];
            bool first = true;
            while (cut != -1)
            {
                if (!first) writer.WriteSpace();
                writer.WriteInt(cut);
                first = false;
                cut = nextCut[cut];
            }

            writer.WriteLine();
        }
    }
}
