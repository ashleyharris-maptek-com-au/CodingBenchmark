#include <algorithm>
#include <array>
#include <chrono>
#include <cctype>
#include <cmath>
#include <iostream>
#include <limits>
#include <optional>
#include <sstream>
#include <string>
#include <vector>

using namespace std;

struct Move {
    int from = -1;
    int to = -1;
    char promotion = 0;
    bool isCapture = false;
    bool isEnPassant = false;
    bool isCastleKingSide = false;
    bool isCastleQueenSide = false;
};

struct Undo {
    Move move;
    char movedPiece = '.';
    char capturedPiece = '.';
    int castlingRights = 0;
    int epSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
};

struct Position {
    array<char, 64> board{};
    bool whiteToMove = true;
    int castlingRights = 0;
    int epSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
};

static constexpr int CASTLE_WK = 1;
static constexpr int CASTLE_WQ = 2;
static constexpr int CASTLE_BK = 4;
static constexpr int CASTLE_BQ = 8;
static constexpr int INF = 1000000000;
static constexpr int MATE_SCORE = 1000000;

static bool isWhitePiece(char p) { return p >= 'A' && p <= 'Z'; }
static bool isBlackPiece(char p) { return p >= 'a' && p <= 'z'; }
static bool isEmpty(char p) { return p == '.'; }
static int fileOf(int sq) { return sq & 7; }
static int rankOf(int sq) { return sq >> 3; }
static bool onBoard(int sq) { return sq >= 0 && sq < 64; }

static string squareToString(int sq) {
    string s;
    s += char('a' + fileOf(sq));
    s += char('1' + rankOf(sq));
    return s;
}

static int parseSquare(const string& s) {
    if (s.size() != 2) return -1;
    if (s[0] < 'a' || s[0] > 'h' || s[1] < '1' || s[1] > '8') return -1;
    return (s[1] - '1') * 8 + (s[0] - 'a');
}

static int pieceValue(char p) {
    switch (toupper(static_cast<unsigned char>(p))) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 0;
        default: return 0;
    }
}

static int pieceSquareValue(char p, int sq, int phase) {
    int f = fileOf(sq);
    int r = rankOf(sq);
    bool white = isWhitePiece(p);
    int rr = white ? r : 7 - r;
    int centerDist = abs(f - 3) + abs(rr - 3);
    switch (toupper(static_cast<unsigned char>(p))) {
        case 'P':
            return rr * 12 - abs(f - 3) * 3 + ((f >= 2 && f <= 5) ? 4 : 0);
        case 'N':
            return 34 - centerDist * 7 + (rr >= 2 && rr <= 5 ? 4 : 0);
        case 'B':
            return 24 - centerDist * 4 + (rr >= 2 && rr <= 5 ? 4 : 0);
        case 'R':
            return rr * 2 + ((f > 0 && f < 7) ? 5 : 0);
        case 'Q':
            return 10 - centerDist * 2;
        case 'K':
            if (phase <= 1300) {
                return 28 - centerDist * 7;
            }
            return -centerDist * 8 - rr * 2 + ((f == 6 || f == 2) ? 10 : 0);
        default:
            return 0;
    }
}

static bool parseFEN(const string& fen, Position& pos) {
    fill(pos.board.begin(), pos.board.end(), '.');
    istringstream iss(fen);
    string boardPart, sidePart, castlePart, epPart;
    if (!(iss >> boardPart >> sidePart >> castlePart >> epPart >> pos.halfmoveClock >> pos.fullmoveNumber)) {
        return false;
    }

    int rank = 7;
    int file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            if (file != 8 || rank == 0) return false;
            --rank;
            file = 0;
            continue;
        }
        if (isdigit(static_cast<unsigned char>(c))) {
            file += c - '0';
            if (file > 8) return false;
            continue;
        }
        if (string("PNBRQKpnbrqk").find(c) == string::npos) return false;
        if (file >= 8 || rank < 0) return false;
        pos.board[rank * 8 + file] = c;
        ++file;
    }
    if (rank != 0 || file != 8) return false;

    if (sidePart == "w") pos.whiteToMove = true;
    else if (sidePart == "b") pos.whiteToMove = false;
    else return false;

    pos.castlingRights = 0;
    if (castlePart != "-") {
        for (char c : castlePart) {
            if (c == 'K') pos.castlingRights |= CASTLE_WK;
            else if (c == 'Q') pos.castlingRights |= CASTLE_WQ;
            else if (c == 'k') pos.castlingRights |= CASTLE_BK;
            else if (c == 'q') pos.castlingRights |= CASTLE_BQ;
            else return false;
        }
    }

    pos.epSquare = (epPart == "-") ? -1 : parseSquare(epPart);
    if (epPart != "-" && pos.epSquare == -1) return false;
    return true;
}

static int findKing(const Position& pos, bool white) {
    char target = white ? 'K' : 'k';
    for (int i = 0; i < 64; ++i) {
        if (pos.board[i] == target) return i;
    }
    return -1;
}

static bool squareAttacked(const Position& pos, int sq, bool byWhite) {
    int f = fileOf(sq);
    int r = rankOf(sq);

    if (byWhite) {
        if (r > 0 && f > 0 && pos.board[sq - 9] == 'P') return true;
        if (r > 0 && f < 7 && pos.board[sq - 7] == 'P') return true;
    } else {
        if (r < 7 && f > 0 && pos.board[sq + 7] == 'p') return true;
        if (r < 7 && f < 7 && pos.board[sq + 9] == 'p') return true;
    }

    static const int knightDf[8] = {1, 2, 2, 1, -1, -2, -2, -1};
    static const int knightDr[8] = {2, 1, -1, -2, -2, -1, 1, 2};
    for (int i = 0; i < 8; ++i) {
        int nf = f + knightDf[i];
        int nr = r + knightDr[i];
        if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
        char p = pos.board[nr * 8 + nf];
        if (byWhite && p == 'N') return true;
        if (!byWhite && p == 'n') return true;
    }

    static const int bishopDf[4] = {1, 1, -1, -1};
    static const int bishopDr[4] = {1, -1, 1, -1};
    for (int d = 0; d < 4; ++d) {
        int nf = f + bishopDf[d];
        int nr = r + bishopDr[d];
        while (nf >= 0 && nf <= 7 && nr >= 0 && nr <= 7) {
            char p = pos.board[nr * 8 + nf];
            if (!isEmpty(p)) {
                if (byWhite && (p == 'B' || p == 'Q')) return true;
                if (!byWhite && (p == 'b' || p == 'q')) return true;
                break;
            }
            nf += bishopDf[d];
            nr += bishopDr[d];
        }
    }

    static const int rookDf[4] = {1, -1, 0, 0};
    static const int rookDr[4] = {0, 0, 1, -1};
    for (int d = 0; d < 4; ++d) {
        int nf = f + rookDf[d];
        int nr = r + rookDr[d];
        while (nf >= 0 && nf <= 7 && nr >= 0 && nr <= 7) {
            char p = pos.board[nr * 8 + nf];
            if (!isEmpty(p)) {
                if (byWhite && (p == 'R' || p == 'Q')) return true;
                if (!byWhite && (p == 'r' || p == 'q')) return true;
                break;
            }
            nf += rookDf[d];
            nr += rookDr[d];
        }
    }

    for (int df = -1; df <= 1; ++df) {
        for (int dr = -1; dr <= 1; ++dr) {
            if (df == 0 && dr == 0) continue;
            int nf = f + df;
            int nr = r + dr;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
            char p = pos.board[nr * 8 + nf];
            if (byWhite && p == 'K') return true;
            if (!byWhite && p == 'k') return true;
        }
    }

    return false;
}

static bool inCheck(const Position& pos, bool white) {
    int kingSq = findKing(pos, white);
    if (kingSq == -1) return false;
    return squareAttacked(pos, kingSq, !white);
}

static Undo makeMove(Position& pos, const Move& mv) {
    Undo u;
    u.move = mv;
    u.castlingRights = pos.castlingRights;
    u.epSquare = pos.epSquare;
    u.halfmoveClock = pos.halfmoveClock;
    u.fullmoveNumber = pos.fullmoveNumber;
    u.movedPiece = pos.board[mv.from];
    u.capturedPiece = mv.isEnPassant ? (pos.whiteToMove ? 'p' : 'P') : pos.board[mv.to];

    char movingPiece = pos.board[mv.from];
    pos.board[mv.from] = '.';

    if (mv.isCastleKingSide) {
        if (pos.whiteToMove) {
            pos.board[6] = 'K';
            pos.board[5] = 'R';
            pos.board[7] = '.';
        } else {
            pos.board[62] = 'k';
            pos.board[61] = 'r';
            pos.board[63] = '.';
        }
    } else if (mv.isCastleQueenSide) {
        if (pos.whiteToMove) {
            pos.board[2] = 'K';
            pos.board[3] = 'R';
            pos.board[0] = '.';
        } else {
            pos.board[58] = 'k';
            pos.board[59] = 'r';
            pos.board[56] = '.';
        }
    } else {
        if (mv.isEnPassant) {
            int capSq = pos.whiteToMove ? mv.to - 8 : mv.to + 8;
            pos.board[capSq] = '.';
        }
        char placed = movingPiece;
        if (mv.promotion) {
            placed = pos.whiteToMove ? mv.promotion : static_cast<char>(tolower(static_cast<unsigned char>(mv.promotion)));
        }
        pos.board[mv.to] = placed;
    }

    pos.epSquare = -1;
    if (toupper(static_cast<unsigned char>(movingPiece)) == 'P' && abs(mv.to - mv.from) == 16) {
        pos.epSquare = (mv.from + mv.to) / 2;
    }

    if (movingPiece == 'K') pos.castlingRights &= ~(CASTLE_WK | CASTLE_WQ);
    if (movingPiece == 'k') pos.castlingRights &= ~(CASTLE_BK | CASTLE_BQ);
    if (movingPiece == 'R') {
        if (mv.from == 0) pos.castlingRights &= ~CASTLE_WQ;
        if (mv.from == 7) pos.castlingRights &= ~CASTLE_WK;
    }
    if (movingPiece == 'r') {
        if (mv.from == 56) pos.castlingRights &= ~CASTLE_BQ;
        if (mv.from == 63) pos.castlingRights &= ~CASTLE_BK;
    }
    if (u.capturedPiece == 'R') {
        if (mv.to == 0) pos.castlingRights &= ~CASTLE_WQ;
        if (mv.to == 7) pos.castlingRights &= ~CASTLE_WK;
    }
    if (u.capturedPiece == 'r') {
        if (mv.to == 56) pos.castlingRights &= ~CASTLE_BQ;
        if (mv.to == 63) pos.castlingRights &= ~CASTLE_BK;
    }

    if (toupper(static_cast<unsigned char>(movingPiece)) == 'P' || mv.isCapture) pos.halfmoveClock = 0;
    else ++pos.halfmoveClock;

    if (!pos.whiteToMove) ++pos.fullmoveNumber;
    pos.whiteToMove = !pos.whiteToMove;
    return u;
}

static void undoMove(Position& pos, const Undo& u) {
    pos.whiteToMove = !pos.whiteToMove;
    pos.castlingRights = u.castlingRights;
    pos.epSquare = u.epSquare;
    pos.halfmoveClock = u.halfmoveClock;
    pos.fullmoveNumber = u.fullmoveNumber;

    const Move& mv = u.move;
    pos.board[mv.from] = u.movedPiece;

    if (mv.isCastleKingSide) {
        if (pos.whiteToMove) {
            pos.board[6] = '.';
            pos.board[5] = '.';
            pos.board[7] = 'R';
        } else {
            pos.board[62] = '.';
            pos.board[61] = '.';
            pos.board[63] = 'r';
        }
        pos.board[mv.to] = '.';
    } else if (mv.isCastleQueenSide) {
        if (pos.whiteToMove) {
            pos.board[2] = '.';
            pos.board[3] = '.';
            pos.board[0] = 'R';
        } else {
            pos.board[58] = '.';
            pos.board[59] = '.';
            pos.board[56] = 'r';
        }
        pos.board[mv.to] = '.';
    } else {
        pos.board[mv.to] = u.capturedPiece;
        if (mv.isEnPassant) {
            pos.board[mv.to] = '.';
            int capSq = pos.whiteToMove ? mv.to - 8 : mv.to + 8;
            pos.board[capSq] = u.capturedPiece;
        }
    }
}

static void addPromotionMoves(vector<Move>& moves, int from, int to, bool capture, bool ep = false) {
    for (char promo : {'Q', 'R', 'B', 'N'}) {
        Move mv;
        mv.from = from;
        mv.to = to;
        mv.promotion = promo;
        mv.isCapture = capture;
        mv.isEnPassant = ep;
        moves.push_back(mv);
    }
}

static void generatePseudoMoves(const Position& pos, vector<Move>& moves) {
    moves.clear();
    bool white = pos.whiteToMove;

    for (int sq = 0; sq < 64; ++sq) {
        char p = pos.board[sq];
        if (isEmpty(p)) continue;
        if (white && !isWhitePiece(p)) continue;
        if (!white && !isBlackPiece(p)) continue;

        int f = fileOf(sq);
        int r = rankOf(sq);
        char up = static_cast<char>(toupper(static_cast<unsigned char>(p)));

        if (up == 'P') {
            int dir = white ? 8 : -8;
            int startRank = white ? 1 : 6;
            int promoRank = white ? 6 : 1;
            int one = sq + dir;
            if (onBoard(one) && isEmpty(pos.board[one])) {
                if (r == promoRank) addPromotionMoves(moves, sq, one, false);
                else {
                    moves.push_back(Move{sq, one, 0, false, false, false, false});
                    int two = sq + dir * 2;
                    if (r == startRank && isEmpty(pos.board[two])) {
                        moves.push_back(Move{sq, two, 0, false, false, false, false});
                    }
                }
            }
            for (int df : {-1, 1}) {
                int nf = f + df;
                if (nf < 0 || nf > 7) continue;
                int to = one + df;
                if (!onBoard(to)) continue;
                char target = pos.board[to];
                bool enemy = !isEmpty(target) && (white ? isBlackPiece(target) : isWhitePiece(target));
                if (enemy) {
                    if (r == promoRank) addPromotionMoves(moves, sq, to, true);
                    else moves.push_back(Move{sq, to, 0, true, false, false, false});
                } else if (to == pos.epSquare) {
                    Move mv;
                    mv.from = sq;
                    mv.to = to;
                    mv.isCapture = true;
                    mv.isEnPassant = true;
                    moves.push_back(mv);
                }
            }
            continue;
        }

        if (up == 'N') {
            static const int kDf[8] = {1, 2, 2, 1, -1, -2, -2, -1};
            static const int kDr[8] = {2, 1, -1, -2, -2, -1, 1, 2};
            for (int i = 0; i < 8; ++i) {
                int nf = f + kDf[i];
                int nr = r + kDr[i];
                if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                int to = nr * 8 + nf;
                char target = pos.board[to];
                if (isEmpty(target) || (white ? isBlackPiece(target) : isWhitePiece(target))) {
                    moves.push_back(Move{sq, to, 0, !isEmpty(target), false, false, false});
                }
            }
            continue;
        }

        if (up == 'B' || up == 'R' || up == 'Q') {
            static const int df[8] = {1, 1, -1, -1, 1, -1, 0, 0};
            static const int dr[8] = {1, -1, 1, -1, 0, 0, 1, -1};
            int start = (up == 'B') ? 0 : (up == 'R' ? 4 : 0);
            int end = (up == 'B') ? 4 : (up == 'R' ? 8 : 8);
            for (int d = start; d < end; ++d) {
                int nf = f + df[d];
                int nr = r + dr[d];
                while (nf >= 0 && nf <= 7 && nr >= 0 && nr <= 7) {
                    int to = nr * 8 + nf;
                    char target = pos.board[to];
                    if (isEmpty(target)) {
                        moves.push_back(Move{sq, to, 0, false, false, false, false});
                    } else {
                        if (white ? isBlackPiece(target) : isWhitePiece(target)) {
                            moves.push_back(Move{sq, to, 0, true, false, false, false});
                        }
                        break;
                    }
                    nf += df[d];
                    nr += dr[d];
                }
            }
            continue;
        }

        if (up == 'K') {
            for (int df = -1; df <= 1; ++df) {
                for (int dr = -1; dr <= 1; ++dr) {
                    if (df == 0 && dr == 0) continue;
                    int nf = f + df;
                    int nr = r + dr;
                    if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                    int to = nr * 8 + nf;
                    char target = pos.board[to];
                    if (isEmpty(target) || (white ? isBlackPiece(target) : isWhitePiece(target))) {
                        moves.push_back(Move{sq, to, 0, !isEmpty(target), false, false, false});
                    }
                }
            }

            if (white) {
                if ((pos.castlingRights & CASTLE_WK) &&
                    pos.board[5] == '.' && pos.board[6] == '.' &&
                    pos.board[4] == 'K' && pos.board[7] == 'R' &&
                    !squareAttacked(pos, 4, false) &&
                    !squareAttacked(pos, 5, false) &&
                    !squareAttacked(pos, 6, false)) {
                    Move mv;
                    mv.from = 4;
                    mv.to = 6;
                    mv.isCastleKingSide = true;
                    moves.push_back(mv);
                }
                if ((pos.castlingRights & CASTLE_WQ) &&
                    pos.board[1] == '.' && pos.board[2] == '.' && pos.board[3] == '.' &&
                    pos.board[4] == 'K' && pos.board[0] == 'R' &&
                    !squareAttacked(pos, 4, false) &&
                    !squareAttacked(pos, 3, false) &&
                    !squareAttacked(pos, 2, false)) {
                    Move mv;
                    mv.from = 4;
                    mv.to = 2;
                    mv.isCastleQueenSide = true;
                    moves.push_back(mv);
                }
            } else {
                if ((pos.castlingRights & CASTLE_BK) &&
                    pos.board[61] == '.' && pos.board[62] == '.' &&
                    pos.board[60] == 'k' && pos.board[63] == 'r' &&
                    !squareAttacked(pos, 60, true) &&
                    !squareAttacked(pos, 61, true) &&
                    !squareAttacked(pos, 62, true)) {
                    Move mv;
                    mv.from = 60;
                    mv.to = 62;
                    mv.isCastleKingSide = true;
                    moves.push_back(mv);
                }
                if ((pos.castlingRights & CASTLE_BQ) &&
                    pos.board[57] == '.' && pos.board[58] == '.' && pos.board[59] == '.' &&
                    pos.board[60] == 'k' && pos.board[56] == 'r' &&
                    !squareAttacked(pos, 60, true) &&
                    !squareAttacked(pos, 59, true) &&
                    !squareAttacked(pos, 58, true)) {
                    Move mv;
                    mv.from = 60;
                    mv.to = 58;
                    mv.isCastleQueenSide = true;
                    moves.push_back(mv);
                }
            }
        }
    }
}

static vector<Move> generateLegalMoves(Position& pos) {
    vector<Move> pseudo;
    vector<Move> legal;
    generatePseudoMoves(pos, pseudo);
    legal.reserve(pseudo.size());
    for (const Move& mv : pseudo) {
        Undo u = makeMove(pos, mv);
        bool illegal = inCheck(pos, !pos.whiteToMove);
        undoMove(pos, u);
        if (!illegal) legal.push_back(mv);
    }
    return legal;
}

static int evaluate(const Position& pos) {
    int score = 0;
    int phase = 0;
    for (char p : pos.board) {
        char up = static_cast<char>(toupper(static_cast<unsigned char>(p)));
        if (up == 'N' || up == 'B') phase += 320;
        else if (up == 'R') phase += 500;
        else if (up == 'Q') phase += 900;
    }
    for (int sq = 0; sq < 64; ++sq) {
        char p = pos.board[sq];
        if (isEmpty(p)) continue;
        int val = pieceValue(p) + pieceSquareValue(p, sq, phase);
        score += isWhitePiece(p) ? val : -val;
    }
    return pos.whiteToMove ? score : -score;
}

struct Searcher {
    Position& rootPos;
    chrono::steady_clock::time_point deadline;
    bool timedOut = false;
    Move bestMove;
    int nodes = 0;

    bool outOfTime() {
        if (timedOut) return true;
        if ((nodes & 2047) == 0 && chrono::steady_clock::now() >= deadline) {
            timedOut = true;
        }
        return timedOut;
    }

    int moveScore(const Position& pos, const Move& mv) {
        if (mv.isCastleKingSide || mv.isCastleQueenSide) return 50;
        char mover = pos.board[mv.from];
        char victim = mv.isEnPassant ? (pos.whiteToMove ? 'p' : 'P') : pos.board[mv.to];
        int score = 0;
        if (mv.promotion) score += 800 + pieceValue(mv.promotion);
        if (mv.isCapture) score += 10000 + pieceValue(victim) - pieceValue(mover) / 10;
        return score;
    }

    int quiescence(Position& pos, int alpha, int beta) {
        ++nodes;
        if (outOfTime()) return evaluate(pos);

        int standPat = evaluate(pos);
        if (standPat >= beta) return beta;
        if (standPat > alpha) alpha = standPat;

        vector<Move> moves = generateLegalMoves(pos);
        stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveScore(pos, a) > moveScore(pos, b);
        });

        for (const Move& mv : moves) {
            if (!mv.isCapture && !mv.promotion) continue;
            Undo u = makeMove(pos, mv);
            int score = -quiescence(pos, -beta, -alpha);
            undoMove(pos, u);
            if (timedOut) return alpha;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int negamax(Position& pos, int depth, int alpha, int beta, int ply) {
        ++nodes;
        if (outOfTime()) return evaluate(pos);

        bool side = pos.whiteToMove;
        bool checked = inCheck(pos, side);
        if (depth <= 0 && !checked) {
            return quiescence(pos, alpha, beta);
        }
        if (depth <= 0) depth = 1;

        vector<Move> moves = generateLegalMoves(pos);
        if (moves.empty()) {
            return checked ? (-MATE_SCORE + ply) : 0;
        }

        stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveScore(pos, a) > moveScore(pos, b);
        });

        int best = -INF;
        for (const Move& mv : moves) {
            Undo u = makeMove(pos, mv);
            int score = -negamax(pos, depth - 1, -beta, -alpha, ply + 1);
            undoMove(pos, u);
            if (timedOut) return alpha;

            if (score > best) best = score;
            if (score > alpha) alpha = score;
            if (alpha >= beta) break;
        }
        return best;
    }

    optional<Move> search(double timeLimitSeconds) {
        vector<Move> rootMoves = generateLegalMoves(rootPos);
        if (rootMoves.empty()) return nullopt;

        bestMove = rootMoves.front();
        auto now = chrono::steady_clock::now();
        double safety = max(0.05, min(0.25, timeLimitSeconds * 0.05));
        deadline = now + chrono::duration_cast<chrono::steady_clock::duration>(
            chrono::duration<double>(max(0.01, timeLimitSeconds - safety)));

        for (int depth = 1; depth <= 64; ++depth) {
            if (chrono::steady_clock::now() >= deadline) break;

            stable_sort(rootMoves.begin(), rootMoves.end(), [&](const Move& a, const Move& b) {
                return moveScore(rootPos, a) > moveScore(rootPos, b);
            });

            Move iterationBest = rootMoves.front();
            int iterationBestScore = -INF;
            int alpha = -INF;
            int beta = INF;

            for (const Move& mv : rootMoves) {
                Undo u = makeMove(rootPos, mv);
                int score = -negamax(rootPos, depth - 1, -beta, -alpha, 1);
                undoMove(rootPos, u);
                if (timedOut) break;
                if (score > iterationBestScore) {
                    iterationBestScore = score;
                    iterationBest = mv;
                }
                alpha = max(alpha, score);
            }

            if (timedOut) break;
            bestMove = iterationBest;

            auto it = find_if(rootMoves.begin(), rootMoves.end(), [&](const Move& mv) {
                return mv.from == bestMove.from && mv.to == bestMove.to &&
                       mv.promotion == bestMove.promotion &&
                       mv.isCastleKingSide == bestMove.isCastleKingSide &&
                       mv.isCastleQueenSide == bestMove.isCastleQueenSide &&
                       mv.isEnPassant == bestMove.isEnPassant;
            });
            if (it != rootMoves.end()) {
                rotate(rootMoves.begin(), it, it + 1);
            }
        }

        return bestMove;
    }
};

static string moveToString(Position pos, const Move& mv) {
    string result;
    char moving = pos.board[mv.from];

    if (mv.isCastleKingSide) result = "O-O";
    else if (mv.isCastleQueenSide) result = "O-O-O";
    else {
        char pieceChar = static_cast<char>(toupper(static_cast<unsigned char>(moving)));
        if (pieceChar == 'P') result += 'p';
        else result += pieceChar;
        result += squareToString(mv.from);
        if (mv.isCapture) result += 'x';
        result += squareToString(mv.to);
        if (mv.isEnPassant) result += "ep";
        if (mv.promotion) {
            result += '/';
            result += static_cast<char>(toupper(static_cast<unsigned char>(mv.promotion)));
        }
    }

    Undo u = makeMove(pos, mv);
    bool oppInCheck = inCheck(pos, pos.whiteToMove);
    if (oppInCheck) {
        vector<Move> replies = generateLegalMoves(pos);
        result += replies.empty() ? '#' : '+';
    }
    undoMove(pos, u);
    return result;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string timeoutLine;
    string fen;
    if (!getline(cin, timeoutLine)) return 0;
    if (!getline(cin, fen)) return 0;

    double timeoutSeconds = 5.0;
    try {
        timeoutSeconds = stod(timeoutLine);
    } catch (...) {
        timeoutSeconds = 5.0;
    }

    Position pos;
    if (!parseFEN(fen, pos)) {
        cout << "0000\n";
        return 0;
    }

    Searcher searcher{pos};
    optional<Move> best = searcher.search(timeoutSeconds);
    if (!best.has_value()) {
        cout << "0000\n";
        return 0;
    }

    cout << moveToString(pos, *best) << '\n';
    return 0;
}
