import math
from statistics import median


_MEM = {
    "initialized": False,
    "pos": [0.0, 0.0, 0.0],
    "vel": [0.0, 0.0, 0.0],
    "bias": [0.0, 0.0, 0.0],
    "prev_cmd": [0.0, 0.0, 0.0],
}


def _clamp(v, lo, hi):
    return lo if v < lo else hi if v > hi else v


def _norm(v):
    return math.sqrt(sum(x * x for x in v))


def _median_or_none(values):
    vals = [v for v in values if v is not None]
    return median(vals) if vals else None


def _fuse_scalar_candidates(candidates, fallback=None, jump_limit=None):
    vals = [v for v in candidates if v is not None and math.isfinite(v)]
    if not vals:
        return fallback
    if jump_limit is not None and fallback is not None:
        vals = [v for v in vals if abs(v - fallback) <= jump_limit] or vals
    return float(median(vals))


def _sensor_component(state, prefix, axis, kind="pos"):
    if prefix == "radar":
        value_key = f"radar_rel_{kind}_{axis}_{'m' if kind == 'pos' else 'ms'}"
        valid_key = f"radar_rel{'v_' if kind == 'vel' else '_'}{axis}_valid"
    else:
        value_key = f"lidar_rel_{kind}_{axis}_{'m' if kind == 'pos' else 'ms'}"
        valid_key = f"lidar_rel{'v_' if kind == 'vel' else '_'}{axis}_valid"
    if state.get(valid_key):
        v = state.get(value_key)
        if isinstance(v, (int, float)) and math.isfinite(v):
            return float(v)
    return None


def _fuse_vector(state, kind, predicted, range_est):
    out = []
    for i, axis in enumerate("xyz"):
        radar_v = _sensor_component(state, "radar", axis, kind)
        lidar_v = _sensor_component(state, "lidar", axis, kind)
        candidates = []
        if range_est is not None and range_est < 120.0:
            candidates.extend([lidar_v, lidar_v, radar_v])
        else:
            candidates.extend([radar_v, lidar_v])
        jump = 80.0 if kind == "pos" else 4.0
        out.append(_fuse_scalar_candidates(candidates, predicted[i], jump))
    return out


def _fuse_range(state, pos_guess):
    samples = []
    for i in (1, 2, 3):
        if state.get(f"range_{i}_valid"):
            v = state.get(f"range_{i}_m")
            if isinstance(v, (int, float)) and math.isfinite(v) and v >= 0.0:
                samples.append(float(v))
    if state.get("dock_camera_active"):
        v = state.get("dock_camera_range_m")
        if isinstance(v, (int, float)) and math.isfinite(v) and v >= 0.0:
            samples.append(float(v))
    geom = _norm(pos_guess) if pos_guess is not None else None
    return _fuse_scalar_candidates(samples + [geom], geom, jump_limit=150.0)


def _fuse_accel(state):
    out = []
    for axis in "xyz":
        vals = []
        for i in (1, 2, 3):
            if state.get(f"accel_{i}_{axis}_valid"):
                v = state.get(f"accel_{i}_{axis}_ms2")
                if isinstance(v, (int, float)) and math.isfinite(v):
                    vals.append(float(v))
        out.append(_median_or_none(vals) or 0.0)
    return out


def _fuse_gyro(state):
    out = []
    for axis in ("roll", "pitch", "yaw"):
        vals = []
        for i in (1, 2, 3):
            if state.get(f"gyro_{i}_{axis}_valid"):
                v = state.get(f"gyro_{i}_{axis}_dps")
                if isinstance(v, (int, float)) and math.isfinite(v):
                    vals.append(math.radians(float(v)))
        out.append(_median_or_none(vals) or 0.0)
    return out


def _fuse_star_tracker(state):
    out = []
    for axis in ("roll", "pitch", "yaw"):
        vals = []
        for i in (1, 2):
            if state.get(f"star_tracker_{i}_{axis}_valid"):
                v = state.get(f"star_tracker_{i}_{axis}_deg")
                if isinstance(v, (int, float)) and math.isfinite(v):
                    vals.append(math.radians(float(v)))
        out.append(_median_or_none(vals) or 0.0)
    return out


def _approach_speed_limit(r):
    if r > 500.0:
        return 3.0
    if r > 200.0:
        return 1.5
    if r > 50.0:
        return 0.6
    if r > 10.0:
        return 0.2
    return 0.08


def _target_velocity(pos, rng):
    x, y, z = pos
    vtx = 0.0
    vtz = 0.0

    if rng > 200.0:
        vty = _clamp(-0.010 * y, -_approach_speed_limit(rng), _approach_speed_limit(rng))
        # Far away, exploit orbital mechanics: if behind (y<0), bias retrograde.
        orbital_bias = -0.45 if y < -20.0 else 0.45 if y > 20.0 else 0.0
        vty += orbital_bias
        # Keep radial/cross-track small while avoiding direct hard radial chase.
        vtx = _clamp(-0.003 * x, -0.25, 0.25)
        vtz = _clamp(-0.003 * z, -0.25, 0.25)
    elif rng > 50.0:
        vty = _clamp(-0.012 * y, -_approach_speed_limit(rng), _approach_speed_limit(rng))
        vtx = _clamp(-0.010 * x, -0.20, 0.20)
        vtz = _clamp(-0.010 * z, -0.20, 0.20)
    elif rng > 10.0:
        vty = _clamp(-0.015 * y, -_approach_speed_limit(rng), _approach_speed_limit(rng))
        vtx = _clamp(-0.020 * x, -0.12, 0.12)
        vtz = _clamp(-0.020 * z, -0.12, 0.12)
    else:
        vty = _clamp(-0.020 * y, -_approach_speed_limit(rng), _approach_speed_limit(rng))
        vtx = _clamp(-0.030 * x, -0.08, 0.08)
        vtz = _clamp(-0.030 * z, -0.08, 0.08)

    return [vtx, vty, vtz]


def autopilot_step(state: dict, dt: float) -> dict:
    global _MEM

    dt = max(0.02, float(dt))

    fuel_pct = float(state.get("fuel_remaining_pct", 100.0) or 100.0)
    fuel_kg = float(state.get("fuel_remaining_kg", 500.0) or 500.0)
    low_fuel = fuel_pct < 15.0 or fuel_kg < 75.0

    accel = _fuse_accel(state)
    gyro = _fuse_gyro(state)
    attitude = _fuse_star_tracker(state)

    pred_pos = _MEM["pos"][:] if _MEM["initialized"] else [0.0, 0.0, 0.0]
    pred_vel = _MEM["vel"][:] if _MEM["initialized"] else [0.0, 0.0, 0.0]
    if _MEM["initialized"]:
        for i in range(3):
            pred_pos[i] += pred_vel[i] * dt + 0.5 * accel[i] * dt * dt
            pred_vel[i] += accel[i] * dt

    coarse_range = _fuse_range(state, pred_pos if _MEM["initialized"] else [0.0, 0.0, 0.0])
    pos = _fuse_vector(state, "pos", pred_pos, coarse_range)
    rng = _fuse_range(state, pos)
    vel = _fuse_vector(state, "vel", pred_vel, rng)

    if not _MEM["initialized"]:
        _MEM["initialized"] = True

    # Adapt a small persistent trim against unknown constant disturbances.
    for i in range(3):
        pred_acc = 0.25 * _MEM["prev_cmd"][i]
        resid = _clamp(accel[i] - pred_acc, -0.05, 0.05)
        _MEM["bias"][i] = 0.995 * _MEM["bias"][i] + 0.005 * resid

    target_vel = _target_velocity(pos, rng)

    kp = 0.018 if rng > 200.0 else 0.032 if rng > 50.0 else 0.060 if rng > 10.0 else 0.085
    kd = 0.32 if rng > 200.0 else 0.42 if rng > 50.0 else 0.60 if rng > 10.0 else 0.85

    # CW-inspired feed-forward: along-track error should mainly drive tangential thrust.
    x, y, z = pos
    vx, vy, vz = vel
    ex = x
    ey = y
    ez = z
    evx = vx - target_vel[0]
    evy = vy - target_vel[1]
    evz = vz - target_vel[2]

    cmd_x = -kp * ex - kd * evx
    cmd_y = -kp * 0.55 * ey - kd * evy
    cmd_z = -kp * ez - kd * evz

    if rng > 200.0:
        if y < -20.0:
            cmd_y -= 0.20
        elif y > 20.0:
            cmd_y += 0.20
        cmd_x *= 0.55
        cmd_z *= 0.75

    if rng < 20.0:
        cmd_x *= 0.65
        cmd_y *= 0.55
        cmd_z *= 0.65

    # Range and closing-rate protection near docking.
    closing_speed = 0.0
    if rng > 1e-6:
        closing_speed = -(x * vx + y * vy + z * vz) / rng
    speed_limit = _approach_speed_limit(rng)
    if closing_speed > speed_limit:
        brake = _clamp((closing_speed - speed_limit) * (1.6 if rng < 50.0 else 1.0), 0.0, 0.8)
        if rng > 1e-6:
            cmd_x += brake * x / rng
            cmd_y += brake * y / rng
            cmd_z += brake * z / rng

    # Near-field direct centering is acceptable.
    if rng < 50.0:
        cmd_x += _clamp(-0.015 * x, -0.25, 0.25)
        cmd_z += _clamp(-0.015 * z, -0.25, 0.25)
    if rng < 10.0:
        cmd_y += _clamp(-0.020 * y, -0.20, 0.20)

    # Disturbance trim and fuel-aware authority limiting.
    cmd_x -= 6.0 * _MEM["bias"][0]
    cmd_y -= 6.0 * _MEM["bias"][1]
    cmd_z -= 6.0 * _MEM["bias"][2]

    thrust_cap = 0.40 if low_fuel and rng > 50.0 else 0.28 if low_fuel else 1.0
    if low_fuel and rng > 100.0:
        cmd_x *= 0.75
        cmd_z *= 0.75

    thrust_x = _clamp(cmd_x, -thrust_cap, thrust_cap)
    thrust_y = _clamp(cmd_y, -thrust_cap, thrust_cap)
    thrust_z = _clamp(cmd_z, -thrust_cap, thrust_cap)

    cam_yaw = 0.0
    cam_pitch = 0.0
    if state.get("dock_camera_active"):
        bx = state.get("dock_camera_bearing_x") if state.get("dock_camera_bearing_x_valid") else None
        by = state.get("dock_camera_bearing_y") if state.get("dock_camera_bearing_y_valid") else None
        bz = state.get("dock_camera_bearing_z") if state.get("dock_camera_bearing_z_valid") else None
        if all(isinstance(v, (int, float)) and math.isfinite(v) for v in (bx, by, bz)):
            mag = max(1e-9, math.sqrt(bx * bx + by * by + bz * bz))
            bx, by, bz = bx / mag, by / mag, bz / mag
            cam_yaw = math.atan2(bx, by)
            cam_pitch = -math.atan2(bz, by)

    roll_err, pitch_err, yaw_err = attitude
    att_rate_roll = _clamp(-0.8 * roll_err - 0.35 * gyro[0], -0.035, 0.035)
    att_rate_pitch = _clamp(0.7 * cam_pitch - 0.4 * pitch_err - 0.35 * gyro[1], -0.035, 0.035)
    att_rate_yaw = _clamp(0.7 * cam_yaw - 0.4 * yaw_err - 0.35 * gyro[2], -0.035, 0.035)

    _MEM["pos"] = pos
    _MEM["vel"] = vel
    _MEM["prev_cmd"] = [thrust_x, thrust_y, thrust_z]

    return {
        "thrust_x": thrust_x,
        "thrust_y": thrust_y,
        "thrust_z": thrust_z,
        "att_rate_roll": att_rate_roll,
        "att_rate_pitch": att_rate_pitch,
        "att_rate_yaw": att_rate_yaw,
    }
