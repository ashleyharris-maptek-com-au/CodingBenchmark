const fs = require('fs');

class Heap {
  constructor(compare) {
    this.data = [];
    this.compare = compare;
  }

  size() {
    return this.data.length;
  }

  peek() {
    return this.data[0];
  }

  push(value) {
    const data = this.data;
    data.push(value);
    let i = data.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(data[i], data[p]) >= 0) break;
      [data[i], data[p]] = [data[p], data[i]];
      i = p;
    }
  }

  pop() {
    const data = this.data;
    if (data.length === 0) return undefined;
    const root = data[0];
    const last = data.pop();
    if (data.length > 0) {
      data[0] = last;
      let i = 0;
      while (true) {
        let best = i;
        const l = i * 2 + 1;
        const r = l + 1;
        if (l < data.length && this.compare(data[l], data[best]) < 0) best = l;
        if (r < data.length && this.compare(data[r], data[best]) < 0) best = r;
        if (best === i) break;
        [data[i], data[best]] = [data[best], data[i]];
        i = best;
      }
    }
    return root;
  }
}

function toNumber(value, fallback) {
  const num = Number(value);
  return Number.isFinite(num) ? num : fallback;
}

function main() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);

  const rawTasks = Array.isArray(data.tasks) ? data.tasks : [];
  const n = rawTasks.length;
  const machineCount = Math.max(1, Math.floor(toNumber(data.machineCount, 1)));

  if (n === 0) {
    process.stdout.write(JSON.stringify({ schedule: [] }));
    return;
  }

  const idToIndex = new Map();
  for (let i = 0; i < n; i++) {
    idToIndex.set(rawTasks[i].id, i);
  }

  const tasks = new Array(n);
  const dependents = Array.from({ length: n }, () => []);
  const indegree = new Array(n).fill(0);

  for (let i = 0; i < n; i++) {
    const task = rawTasks[i] || {};
    const depsRaw = Array.isArray(task.deps) ? task.deps : [];
    const deps = [];
    const seen = new Set();
    for (let j = 0; j < depsRaw.length; j++) {
      const depIndex = idToIndex.get(depsRaw[j]);
      if (depIndex === undefined || depIndex === i || seen.has(depIndex)) continue;
      seen.add(depIndex);
      deps.push(depIndex);
      dependents[depIndex].push(i);
    }
    indegree[i] = deps.length;
    tasks[i] = {
      id: task.id,
      duration: Math.max(0, Math.floor(toNumber(task.duration, 0))),
      deadline: Number.isFinite(Number(task.deadline)) ? Number(task.deadline) : Number.POSITIVE_INFINITY,
      weight: Math.max(0, toNumber(task.weight, 0)),
      deps,
    };
  }

  const topo = [];
  const queue = [];
  let qHead = 0;
  const indegreeWork = indegree.slice();
  for (let i = 0; i < n; i++) {
    if (indegreeWork[i] === 0) queue.push(i);
  }
  while (qHead < queue.length) {
    const u = queue[qHead++];
    topo.push(u);
    const outs = dependents[u];
    for (let i = 0; i < outs.length; i++) {
      const v = outs[i];
      indegreeWork[v]--;
      if (indegreeWork[v] === 0) queue.push(v);
    }
  }

  if (topo.length !== n) {
    process.stdout.write(JSON.stringify({ order: rawTasks.map((t) => t.id) }));
    return;
  }

  const bottomLevel = new Array(n).fill(0);
  const descendantPressure = new Array(n).fill(0);
  const minDescDeadline = new Array(n).fill(Number.POSITIVE_INFINITY);

  for (let pos = n - 1; pos >= 0; pos--) {
    const u = topo[pos];
    const outs = dependents[u];
    let bestChildBL = 0;
    let pressure = tasks[u].weight;
    let bestDeadline = tasks[u].deadline;
    for (let i = 0; i < outs.length; i++) {
      const v = outs[i];
      if (bottomLevel[v] > bestChildBL) bestChildBL = bottomLevel[v];
      pressure += descendantPressure[v] * 0.35;
      if (minDescDeadline[v] < bestDeadline) bestDeadline = minDescDeadline[v];
    }
    bottomLevel[u] = tasks[u].duration + bestChildBL;
    descendantPressure[u] = pressure;
    minDescDeadline[u] = bestDeadline;
  }

  const releaseTime = new Array(n).fill(0);
  const remaining = indegree.slice();
  const scheduled = new Array(n).fill(false);
  const schedule = [];

  const readyHeap = new Heap((a, b) => {
    if (a.baseScore !== b.baseScore) return b.baseScore - a.baseScore;
    if (a.tieDeadline !== b.tieDeadline) return a.tieDeadline - b.tieDeadline;
    if (a.tieWeight !== b.tieWeight) return b.tieWeight - a.tieWeight;
    return a.index - b.index;
  });

  const futureHeap = new Heap((a, b) => {
    if (a.release !== b.release) return a.release - b.release;
    return a.index - b.index;
  });

  const machineHeap = new Heap((a, b) => {
    if (a.time !== b.time) return a.time - b.time;
    return a.machine - b.machine;
  });

  function makeReadyNode(index) {
    const task = tasks[index];
    const finiteDeadline = Number.isFinite(task.deadline);
    const deadlineBias = finiteDeadline ? Math.max(-1e6, Math.min(1e6, 100000 - task.deadline)) : -1e5;
    return {
      index,
      baseScore:
        bottomLevel[index] * 10000 +
        descendantPressure[index] * 400 +
        task.weight * 200 +
        dependents[index].length * 25 +
        deadlineBias,
      tieDeadline: finiteDeadline ? task.deadline : Number.MAX_SAFE_INTEGER,
      tieWeight: task.weight,
    };
  }

  for (let machine = 0; machine < machineCount; machine++) {
    machineHeap.push({ time: 0, machine });
  }

  for (let i = 0; i < n; i++) {
    if (remaining[i] === 0) readyHeap.push(makeReadyNode(i));
  }

  function dynamicScore(index, currentTime) {
    const task = tasks[index];
    const deadline = task.deadline;
    const finishIfNow = currentTime + task.duration;
    const ownLate = Number.isFinite(deadline) ? Math.max(0, finishIfNow - deadline) : 0;
    const chainDeadline = minDescDeadline[index];
    const chainSlack = Number.isFinite(chainDeadline) ? chainDeadline - currentTime - bottomLevel[index] : 1e9;
    const slackRisk = chainSlack < 0 ? -chainSlack : 1 / (chainSlack + 1);
    return (
      ownLate * (task.weight + 1) * 1e8 +
      slackRisk * (descendantPressure[index] + 1) * 5e5 +
      bottomLevel[index] * 1e3 +
      descendantPressure[index] * 50 +
      task.weight * 20 +
      dependents[index].length
    );
  }

  function pickTask(currentTime) {
    const sample = [];
    const limit = Math.min(32, readyHeap.size());
    for (let i = 0; i < limit; i++) {
      const node = readyHeap.pop();
      if (!node) break;
      sample.push(node);
    }
    let bestPos = 0;
    let bestScore = -Infinity;
    for (let i = 0; i < sample.length; i++) {
      const score = dynamicScore(sample[i].index, currentTime);
      if (score > bestScore) {
        bestScore = score;
        bestPos = i;
      }
    }
    const chosen = sample[bestPos];
    for (let i = 0; i < sample.length; i++) {
      if (i !== bestPos) readyHeap.push(sample[i]);
    }
    return chosen.index;
  }

  let scheduledCount = 0;
  while (scheduledCount < n) {
    const machine = machineHeap.pop();
    let currentTime = machine.time;

    while (futureHeap.size() && futureHeap.peek().release <= currentTime) {
      readyHeap.push(makeReadyNode(futureHeap.pop().index));
    }

    if (readyHeap.size() === 0) {
      if (futureHeap.size() === 0) {
        break;
      }
      currentTime = Math.max(currentTime, futureHeap.peek().release);
      while (futureHeap.size() && futureHeap.peek().release <= currentTime) {
        readyHeap.push(makeReadyNode(futureHeap.pop().index));
      }
    }

    const taskIndex = pickTask(currentTime);
    const task = tasks[taskIndex];
    const start = currentTime;
    const finish = start + task.duration;

    scheduled[taskIndex] = true;
    scheduledCount++;
    schedule.push({ id: task.id, start, machine: machine.machine });
    machineHeap.push({ time: finish, machine: machine.machine });

    const outs = dependents[taskIndex];
    for (let i = 0; i < outs.length; i++) {
      const v = outs[i];
      if (finish > releaseTime[v]) releaseTime[v] = finish;
      remaining[v]--;
      if (remaining[v] === 0 && !scheduled[v]) {
        futureHeap.push({ release: releaseTime[v], index: v });
      }
    }
  }

  if (scheduledCount !== n) {
    process.stdout.write(JSON.stringify({ order: topo.map((i) => tasks[i].id) }));
    return;
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
