               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %uv %outColor
               OpExecutionMode %main OriginUpperLeft
               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Function_float = OpTypePointer Function %float
%_ptr_Function_int = OpTypePointer Function %int
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_20 = OpConstant %float 20
%float_0_5 = OpConstant %float 0.5
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%main = OpFunction %void None %func
%entry = OpLabel
%zx = OpVariable %_ptr_Function_float Function
%zy = OpVariable %_ptr_Function_float Function
%cx = OpVariable %_ptr_Function_float Function
%cy = OpVariable %_ptr_Function_float Function
%iter = OpVariable %_ptr_Function_int Function
%uvVal = OpLoad %v2float %uv
%u = OpCompositeExtract %float %uvVal 0
%v = OpCompositeExtract %float %uvVal 1
%u3 = OpFMul %float %u %float_3
%cxVal = OpFSub %float %u3 %float_2
%v2 = OpFMul %float %v %float_2
%cyVal = OpFSub %float %v2 %float_1
               OpStore %cx %cxVal
               OpStore %cy %cyVal
               OpStore %zx %float_0
               OpStore %zy %float_0
               OpStore %iter %int_0
               OpBranch %loop_header
%loop_header = OpLabel
               OpLoopMerge %loop_merge %loop_continue None
               OpBranch %loop_check
%loop_check = OpLabel
%i0 = OpLoad %int %iter
%lt20 = OpSLessThan %bool %i0 %int_20
%zx0 = OpLoad %float %zx
%zy0 = OpLoad %float %zy
%zx2 = OpFMul %float %zx0 %zx0
%zy2 = OpFMul %float %zy0 %zy0
%r2 = OpFAdd %float %zx2 %zy2
%inside = OpFOrdLessThanEqual %bool %r2 %float_4
%keepGoing = OpLogicalAnd %bool %lt20 %inside
               OpSelectionMerge %loop_merge None
               OpBranchConditional %keepGoing %loop_body %loop_merge
%loop_body = OpLabel
%xy = OpFMul %float %zx0 %zy0
%two_xy = OpFMul %float %xy %float_2
%cx0 = OpLoad %float %cx
%cy0 = OpLoad %float %cy
%x_minus_y = OpFSub %float %zx2 %zy2
%new_zx = OpFAdd %float %x_minus_y %cx0
%new_zy = OpFAdd %float %two_xy %cy0
               OpStore %zx %new_zx
               OpStore %zy %new_zy
               OpBranch %loop_continue
%loop_continue = OpLabel
%i1 = OpLoad %int %iter
%i2 = OpIAdd %int %i1 %int_1
               OpStore %iter %i2
               OpBranch %loop_header
%loop_merge = OpLabel
%i_final = OpLoad %int %iter
%i_float = OpConvertSToF %float %i_final
%t = OpFDiv %float %i_float %float_20
%g = OpFMul %float %t %float_0_5
%b = OpFSub %float %float_1 %t
%out = OpCompositeConstruct %v4float %t %g %b %float_1
               OpStore %outColor %out
               OpReturn
               OpFunctionEnd
