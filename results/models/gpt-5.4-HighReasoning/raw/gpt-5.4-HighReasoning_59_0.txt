const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone = Math.max(0, data.maxStepsPerDrone | 0);

const height = grid.length;
const width = height > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const cellCount = width * height;

function output(paths) {
  process.stdout.write(JSON.stringify({ paths }));
}

if (width === 0 || height === 0 || drones.length === 0) {
  output(drones.map(() => [[0, 0]]));
  process.exit(0);
}

function idx(x, y) {
  return y * width + x;
}

function xOf(i) {
  return i % width;
}

function yOf(i) {
  return (i / width) | 0;
}

const value = new Float64Array(cellCount);
const passable = new Uint8Array(cellCount);
let openCells = 0;
let totalPositive = 0;

for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    const i = idx(x, y);
    const v = Number(row[x]) || 0;
    value[i] = v > 0 ? v : 0;
    if (row[x] !== -1) {
      passable[i] = 1;
      openCells++;
      if (v > 0) totalPositive++;
    }
  }
}

const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

const componentId = new Int32Array(cellCount);
componentId.fill(-1);
const components = [];
const queue = new Int32Array(cellCount);

for (let start = 0; start < cellCount; start++) {
  if (!passable[start] || componentId[start] !== -1) continue;
  let head = 0;
  let tail = 0;
  queue[tail++] = start;
  componentId[start] = components.length;
  const cells = [];
  const valuable = [];

  while (head < tail) {
    const cur = queue[head++];
    cells.push(cur);
    if (value[cur] > 0) valuable.push(cur);
    const cx = xOf(cur);
    const cy = yOf(cur);
    for (let d = 0; d < 4; d++) {
      const nx = cx + dirs[d][0];
      const ny = cy + dirs[d][1];
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const ni = idx(nx, ny);
      if (!passable[ni] || componentId[ni] !== -1) continue;
      componentId[ni] = components.length;
      queue[tail++] = ni;
    }
  }

  valuable.sort((a, b) => value[b] - value[a]);
  components.push({ cells, valuable });
}

const potential = new Float64Array(cellCount);
const radius = openCells <= 2500 ? 6 : openCells <= 15000 ? 5 : 4;
const offsets = [];
for (let dy = -radius; dy <= radius; dy++) {
  for (let dx = -radius; dx <= radius; dx++) {
    const md = Math.abs(dx) + Math.abs(dy);
    if (md <= radius) {
      offsets.push([dx, dy, 1 / (md + 1)]);
    }
  }
}
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const here = idx(x, y);
    if (!passable[here]) continue;
    let sum = 0;
    for (let k = 0; k < offsets.length; k++) {
      const ox = x + offsets[k][0];
      const oy = y + offsets[k][1];
      if (ox < 0 || oy < 0 || ox >= width || oy >= height) continue;
      const oi = idx(ox, oy);
      if (value[oi] > 0) sum += value[oi] * offsets[k][2];
    }
    potential[here] = sum;
  }
}

const covered = new Uint8Array(cellCount);
const visitCount = new Uint16Array(cellCount);

function clampStart(drone) {
  let x = Math.max(0, Math.min(width - 1, drone.x | 0));
  let y = Math.max(0, Math.min(height - 1, drone.y | 0));
  if (passable[idx(x, y)]) return idx(x, y);

  let best = -1;
  let bestDist = Infinity;
  for (let i = 0; i < cellCount; i++) {
    if (!passable[i]) continue;
    const d = Math.abs(xOf(i) - x) + Math.abs(yOf(i) - y);
    if (d < bestDist) {
      bestDist = d;
      best = i;
    }
  }
  return best === -1 ? 0 : best;
}

const states = drones.map((drone) => {
  const start = clampStart(drone);
  covered[start] = 1;
  visitCount[start]++;
  return {
    pos: start,
    remaining: maxStepsPerDrone,
    path: [[xOf(start), yOf(start)]],
  };
});

const dist = new Int32Array(cellCount);
const parent = new Int32Array(cellCount);
const gain = new Float64Array(cellCount);
const mark = new Int32Array(cellCount);
let bfsToken = 1;
const bfsQueue = new Int32Array(cellCount);
const seenList = new Int32Array(cellCount);

function chooseMove(state) {
  const remaining = state.remaining;
  if (remaining <= 0) return [state.pos];

  const horizonBase = openCells <= 2500 ? 28 : openCells <= 20000 ? 22 : 16;
  const horizon = Math.min(
    remaining,
    Math.max(6, Math.min(horizonBase, Math.floor(Math.sqrt(openCells / Math.max(1, drones.length))) + 8))
  );

  bfsToken++;
  let head = 0;
  let tail = 0;
  let seenCount = 0;
  const start = state.pos;

  mark[start] = bfsToken;
  dist[start] = 0;
  parent[start] = -1;
  gain[start] = 0;
  bfsQueue[tail++] = start;
  seenList[seenCount++] = start;

  while (head < tail) {
    const cur = bfsQueue[head++];
    const curDist = dist[cur];
    if (curDist >= horizon) continue;
    const cx = xOf(cur);
    const cy = yOf(cur);
    const curGain = gain[cur];

    for (let d = 0; d < 4; d++) {
      const nx = cx + dirs[d][0];
      const ny = cy + dirs[d][1];
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const ni = idx(nx, ny);
      if (!passable[ni]) continue;

      const nd = curDist + 1;
      const add = covered[ni] ? 0 : value[ni];
      const newGain = curGain + add;

      if (mark[ni] !== bfsToken) {
        mark[ni] = bfsToken;
        dist[ni] = nd;
        parent[ni] = cur;
        gain[ni] = newGain;
        bfsQueue[tail++] = ni;
        seenList[seenCount++] = ni;
      } else if (nd < dist[ni] || (nd === dist[ni] && newGain > gain[ni])) {
        dist[ni] = nd;
        parent[ni] = cur;
        gain[ni] = newGain;
        bfsQueue[tail++] = ni;
      }
    }
  }

  let best = start;
  let bestScore = -1e100;

  for (let s = 0; s < seenCount; s++) {
    const cell = seenList[s];
    const d = dist[cell];
    if (d === 0) continue;

    const g = gain[cell];
    const crowdPenalty = visitCount[cell] * 0.15;
    const future = potential[cell] * 0.22;
    const score = (g * 1.35 + future) / Math.pow(d + 1, 0.82) - crowdPenalty;

    if (score > bestScore) {
      bestScore = score;
      best = cell;
    }
  }

  if (best === start || bestScore <= 0.000001) {
    let bestNeighbor = start;
    let bestNeighborScore = -1e100;
    const cx = xOf(start);
    const cy = yOf(start);
    const stayScore = potential[start] * 0.05 - visitCount[start] * 0.25;
    bestNeighborScore = stayScore;

    for (let d = 0; d < 4; d++) {
      const nx = cx + dirs[d][0];
      const ny = cy + dirs[d][1];
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const ni = idx(nx, ny);
      if (!passable[ni]) continue;
      const immediate = covered[ni] ? 0 : value[ni] * 3.0;
      const score = immediate + potential[ni] * 0.18 - visitCount[ni] * 0.2;
      if (score > bestNeighborScore) {
        bestNeighborScore = score;
        bestNeighbor = ni;
      }
    }

    if (bestNeighbor === start) return [start];
    return [start, bestNeighbor];
  }

  const rev = [];
  let cur = best;
  while (cur !== -1) {
    rev.push(cur);
    cur = parent[cur];
  }
  rev.reverse();
  return rev;
}

let active = true;
while (active) {
  active = false;
  for (let i = 0; i < states.length; i++) {
    const state = states[i];
    if (state.remaining <= 0) continue;
    active = true;

    const segment = chooseMove(state);
    for (let p = 1; p < segment.length && state.remaining > 0; p++) {
      const cell = segment[p];
      state.pos = cell;
      state.remaining--;
      covered[cell] = 1;
      visitCount[cell]++;
      state.path.push([xOf(cell), yOf(cell)]);
    }

    if (segment.length === 1 && state.remaining > 0) {
      state.remaining--;
      visitCount[state.pos]++;
      state.path.push([xOf(state.pos), yOf(state.pos)]);
    }
  }
}

for (const state of states) {
  while (state.path.length < maxStepsPerDrone + 1) {
    state.path.push([xOf(state.pos), yOf(state.pos)]);
  }
}

output(states.map((s) => s.path));
