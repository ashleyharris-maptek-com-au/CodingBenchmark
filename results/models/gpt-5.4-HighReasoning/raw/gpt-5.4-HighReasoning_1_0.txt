#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

struct Point {
    float x = 0.0f;
    float y = 0.0f;
};

static inline double dist_xy(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return std::sqrt(dx * dx + dy * dy);
}

static double route_length(const std::vector<int>& route, const std::vector<Point>& pts) {
    const int n = static_cast<int>(route.size());
    if (n <= 1) {
        return 0.0;
    }
    double total = 0.0;
    for (int i = 0; i < n; ++i) {
        total += dist_xy(pts[route[i]], pts[route[(i + 1) % n]]);
    }
    return total;
}

static void rotate_to_zero_front(std::vector<int>& route) {
    const auto it = std::find(route.begin(), route.end(), 0);
    if (it != route.end()) {
        std::rotate(route.begin(), it, route.end());
    }
}

static uint64_t spread_bits(uint32_t v) {
    uint64_t x = v;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2)) & 0x3333333333333333ULL;
    x = (x | (x << 1)) & 0x5555555555555555ULL;
    return x;
}

static uint64_t morton_key(float x, float y, float min_x, float min_y, double scale_x, double scale_y) {
    const uint32_t qx = (scale_x == 0.0)
        ? 0u
        : static_cast<uint32_t>((static_cast<double>(x) - static_cast<double>(min_x)) * scale_x + 0.5);
    const uint32_t qy = (scale_y == 0.0)
        ? 0u
        : static_cast<uint32_t>((static_cast<double>(y) - static_cast<double>(min_y)) * scale_y + 0.5);
    return spread_bits(qx) | (spread_bits(qy) << 1);
}

static std::vector<int> build_morton_order(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    if (n <= 1) {
        return order;
    }

    float min_x = pts[0].x;
    float max_x = pts[0].x;
    float min_y = pts[0].y;
    float max_y = pts[0].y;
    for (const Point& p : pts) {
        min_x = std::min(min_x, p.x);
        max_x = std::max(max_x, p.x);
        min_y = std::min(min_y, p.y);
        max_y = std::max(max_y, p.y);
    }

    const double dx = static_cast<double>(max_x) - static_cast<double>(min_x);
    const double dy = static_cast<double>(max_y) - static_cast<double>(min_y);
    const double scale_x = (dx > 0.0) ? (4294967295.0 / dx) : 0.0;
    const double scale_y = (dy > 0.0) ? (4294967295.0 / dy) : 0.0;

    struct KeyIndex {
        uint64_t key;
        int idx;
    };

    std::vector<KeyIndex> keyed;
    keyed.reserve(n);
    for (int i = 0; i < n; ++i) {
        keyed.push_back(KeyIndex{morton_key(pts[i].x, pts[i].y, min_x, min_y, scale_x, scale_y), i});
    }

    std::sort(keyed.begin(), keyed.end(), [](const KeyIndex& a, const KeyIndex& b) {
        if (a.key != b.key) {
            return a.key < b.key;
        }
        return a.idx < b.idx;
    });

    for (int i = 0; i < n; ++i) {
        order[i] = keyed[i].idx;
    }
    rotate_to_zero_front(order);
    return order;
}

static std::vector<int> build_snake_order(const std::vector<Point>& pts, bool primary_x) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    if (n <= 1) {
        return order;
    }

    std::sort(order.begin(), order.end(), [&](int a, int b) {
        const Point& pa = pts[a];
        const Point& pb = pts[b];
        if (primary_x) {
            if (pa.x != pb.x) {
                return pa.x < pb.x;
            }
            if (pa.y != pb.y) {
                return pa.y < pb.y;
            }
        } else {
            if (pa.y != pb.y) {
                return pa.y < pb.y;
            }
            if (pa.x != pb.x) {
                return pa.x < pb.x;
            }
        }
        return a < b;
    });

    int block = static_cast<int>(std::sqrt(static_cast<double>(n)));
    if (block < 16) {
        block = 16;
    }
    if (block > n) {
        block = n;
    }

    for (int start = 0, stripe = 0; start < n; start += block, ++stripe) {
        const int end = std::min(n, start + block);
        std::sort(order.begin() + start, order.begin() + end, [&](int a, int b) {
            const Point& pa = pts[a];
            const Point& pb = pts[b];
            if (primary_x) {
                if (pa.y != pb.y) {
                    return (stripe % 2 == 0) ? (pa.y < pb.y) : (pa.y > pb.y);
                }
                if (pa.x != pb.x) {
                    return pa.x < pb.x;
                }
            } else {
                if (pa.x != pb.x) {
                    return (stripe % 2 == 0) ? (pa.x < pb.x) : (pa.x > pb.x);
                }
                if (pa.y != pb.y) {
                    return pa.y < pb.y;
                }
            }
            return a < b;
        });
    }

    rotate_to_zero_front(order);
    return order;
}

static void two_opt(std::vector<int>& route, const std::vector<Point>& pts, int max_window, int max_passes) {
    const int n = static_cast<int>(route.size());
    if (n < 4) {
        rotate_to_zero_front(route);
        return;
    }

    if (max_window <= 0 || max_window > n) {
        max_window = n;
    }

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 2 < n; ++i) {
            int local_end = std::min(n - 1, i + max_window);
            for (int j = i + 2; j <= local_end; ++j) {
                if (i == 0 && j == n - 1) {
                    continue;
                }
                const int a = route[i];
                const int b = route[i + 1];
                const int c = route[j];
                const int d = route[(j + 1) % n];

                const double before = dist_xy(pts[a], pts[b]) + dist_xy(pts[c], pts[d]);
                const double after = dist_xy(pts[a], pts[c]) + dist_xy(pts[b], pts[d]);
                if (after + 1e-12 < before) {
                    std::reverse(route.begin() + i + 1, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) {
            break;
        }
    }
    rotate_to_zero_front(route);
}

static std::vector<int> solve_exact_held_karp(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 1) {
        return std::vector<int>{0};
    }

    const size_t states = size_t(1) << n;
    const double inf = std::numeric_limits<double>::infinity();
    std::vector<double> dp(states * n, inf);
    std::vector<int> parent(states * n, -1);

    auto at = [n](size_t mask, int j) -> size_t {
        return mask * static_cast<size_t>(n) + static_cast<size_t>(j);
    };

    dp[at(1u, 0)] = 0.0;
    for (size_t mask = 1; mask < states; ++mask) {
        if ((mask & 1u) == 0u) {
            continue;
        }
        for (int j = 0; j < n; ++j) {
            if ((mask & (size_t(1) << j)) == 0u) {
                continue;
            }
            const double cur = dp[at(mask, j)];
            if (!std::isfinite(cur)) {
                continue;
            }
            for (int k = 1; k < n; ++k) {
                if ((mask & (size_t(1) << k)) != 0u) {
                    continue;
                }
                const size_t next_mask = mask | (size_t(1) << k);
                const double cand = cur + dist_xy(pts[j], pts[k]);
                const size_t pos = at(next_mask, k);
                if (cand < dp[pos]) {
                    dp[pos] = cand;
                    parent[pos] = j;
                }
            }
        }
    }

    const size_t full = states - 1;
    double best = inf;
    int best_end = 0;
    for (int j = 1; j < n; ++j) {
        const double cand = dp[at(full, j)] + dist_xy(pts[j], pts[0]);
        if (cand < best) {
            best = cand;
            best_end = j;
        }
    }

    std::vector<int> rev;
    rev.reserve(n);
    size_t mask = full;
    int cur = best_end;
    while (cur != -1) {
        rev.push_back(cur);
        const int prev = parent[at(mask, cur)];
        mask &= ~(size_t(1) << cur);
        cur = prev;
    }
    std::reverse(rev.begin(), rev.end());
    rotate_to_zero_front(rev);
    return rev;
}

static std::vector<int> solve_nearest_insertion(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 1) {
        return std::vector<int>{0};
    }
    if (n == 2) {
        return std::vector<int>{0, 1};
    }

    int farthest = 1;
    double farthest_dist = dist_xy(pts[0], pts[1]);
    for (int i = 2; i < n; ++i) {
        const double d = dist_xy(pts[0], pts[i]);
        if (d > farthest_dist) {
            farthest_dist = d;
            farthest = i;
        }
    }

    std::vector<int> tour;
    tour.reserve(n);
    tour.push_back(0);
    tour.push_back(farthest);

    std::vector<char> in_tour(n, 0);
    in_tour[0] = 1;
    in_tour[farthest] = 1;

    const double inf = std::numeric_limits<double>::infinity();
    std::vector<double> best_to_tour(n, inf);
    for (int i = 0; i < n; ++i) {
        if (!in_tour[i]) {
            best_to_tour[i] = std::min(dist_xy(pts[i], pts[0]), dist_xy(pts[i], pts[farthest]));
        }
    }

    while (static_cast<int>(tour.size()) < n) {
        int pick = -1;
        double best_attach = inf;
        for (int i = 0; i < n; ++i) {
            if (!in_tour[i] && best_to_tour[i] < best_attach) {
                best_attach = best_to_tour[i];
                pick = i;
            }
        }

        int best_pos = 0;
        double best_delta = inf;
        const int m = static_cast<int>(tour.size());
        for (int i = 0; i < m; ++i) {
            const int a = tour[i];
            const int b = tour[(i + 1) % m];
            const double delta = dist_xy(pts[a], pts[pick]) + dist_xy(pts[pick], pts[b]) - dist_xy(pts[a], pts[b]);
            if (delta < best_delta) {
                best_delta = delta;
                best_pos = i + 1;
            }
        }

        tour.insert(tour.begin() + best_pos, pick);
        in_tour[pick] = 1;
        best_to_tour[pick] = inf;

        for (int i = 0; i < n; ++i) {
            if (!in_tour[i]) {
                const double d = dist_xy(pts[i], pts[pick]);
                if (d < best_to_tour[i]) {
                    best_to_tour[i] = d;
                }
            }
        }
    }

    rotate_to_zero_front(tour);
    return tour;
}

static std::vector<int> choose_best_candidate(const std::vector<Point>& pts, bool include_snakes) {
    std::vector<int> best = build_morton_order(pts);
    double best_len = route_length(best, pts);

    if (include_snakes) {
        std::vector<int> cand_x = build_snake_order(pts, true);
        const double len_x = route_length(cand_x, pts);
        if (len_x < best_len) {
            best_len = len_x;
            best.swap(cand_x);
        }

        std::vector<int> cand_y = build_snake_order(pts, false);
        const double len_y = route_length(cand_y, pts);
        if (len_y < best_len) {
            best = std::move(cand_y);
        }
    }

    rotate_to_zero_front(best);
    return best;
}

static std::vector<int> solve_tsp(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 0) {
        return {};
    }
    if (n == 1) {
        return std::vector<int>{0};
    }
    if (n <= 18) {
        return solve_exact_held_karp(pts);
    }
    if (n <= 2500) {
        std::vector<int> route = solve_nearest_insertion(pts);
        two_opt(route, pts, n, 3);
        return route;
    }
    if (n <= 200000) {
        std::vector<int> route = choose_best_candidate(pts, true);
        const int window = (n <= 20000) ? 48 : 24;
        const int passes = (n <= 50000) ? 2 : 1;
        two_opt(route, pts, window, passes);
        return route;
    }

    std::vector<int> route = build_morton_order(pts);
    if (n <= 1000000) {
        two_opt(route, pts, 12, 1);
    }
    return route;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::vector<Point> pts;
    pts.resize(std::max(n, 0));
    for (int i = 0; i < n; ++i) {
        std::cin >> pts[i].x >> pts[i].y;
    }

    std::vector<int> route = solve_tsp(pts);
    for (int i = 0; i < static_cast<int>(route.size()); ++i) {
        if (i) {
            std::cout << ' ';
        }
        std::cout << route[i];
    }
    std::cout << '\n';
    return 0;
}
