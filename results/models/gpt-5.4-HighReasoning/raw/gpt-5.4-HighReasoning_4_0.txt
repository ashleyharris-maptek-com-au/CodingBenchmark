using namespace std;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return sqrt(norm2(v)); }

static inline Vec3 normalize(const Vec3& v) {
    double n = norm(v);
    if (n == 0.0) return {};
    return v / n;
}

struct Plane {
    Vec3 n;
    double d = 0.0;
};

struct Mat3 {
    double a[3][3]{};
};

static Vec3 mul(const array<Vec3, 3>& basis, const Vec3& local) {
    return basis[0] * local.x + basis[1] * local.y + basis[2] * local.z;
}

static Vec3 toLocal(const array<Vec3, 3>& basis, const Vec3& world) {
    return {dot(world, basis[0]), dot(world, basis[1]), dot(world, basis[2])};
}

static array<Vec3, 3> orthonormalBasisFromCovariance(const vector<Vec3>& pts) {
    array<Vec3, 3> basis = {Vec3{1, 0, 0}, Vec3{0, 1, 0}, Vec3{0, 0, 1}};
    if (pts.size() < 3) return basis;

    Vec3 mean;
    for (const auto& p : pts) mean += p;
    mean = mean / static_cast<double>(pts.size());

    double a[3][3]{};
    for (const auto& p : pts) {
        Vec3 q = p - mean;
        a[0][0] += q.x * q.x; a[0][1] += q.x * q.y; a[0][2] += q.x * q.z;
        a[1][0] += q.y * q.x; a[1][1] += q.y * q.y; a[1][2] += q.y * q.z;
        a[2][0] += q.z * q.x; a[2][1] += q.z * q.y; a[2][2] += q.z * q.z;
    }

    double v[3][3]{{1, 0, 0}, {0, 1, 0}, {0, 0, 1}};
    for (int it = 0; it < 24; ++it) {
        int p = 0, q = 1;
        double best = fabs(a[0][1]);
        if (fabs(a[0][2]) > best) { best = fabs(a[0][2]); p = 0; q = 2; }
        if (fabs(a[1][2]) > best) { best = fabs(a[1][2]); p = 1; q = 2; }
        if (best < 1e-12) break;

        double app = a[p][p], aqq = a[q][q], apq = a[p][q];
        double phi = 0.5 * atan2(2.0 * apq, aqq - app);
        double c = cos(phi), s = sin(phi);

        for (int k = 0; k < 3; ++k) {
            double aik = a[p][k], aqk = a[q][k];
            a[p][k] = c * aik - s * aqk;
            a[q][k] = s * aik + c * aqk;
        }
        for (int k = 0; k < 3; ++k) {
            double akp = a[k][p], akq = a[k][q];
            a[k][p] = c * akp - s * akq;
            a[k][q] = s * akp + c * akq;
        }
        for (int k = 0; k < 3; ++k) {
            double vkp = v[k][p], vkq = v[k][q];
            v[k][p] = c * vkp - s * vkq;
            v[k][q] = s * vkp + c * vkq;
        }
    }

    vector<pair<double, Vec3>> eigs;
    for (int i = 0; i < 3; ++i) {
        eigs.push_back({a[i][i], normalize(Vec3{v[0][i], v[1][i], v[2][i]})});
    }
    sort(eigs.begin(), eigs.end(), [](const auto& lhs, const auto& rhs) {
        return lhs.first > rhs.first;
    });

    basis[0] = eigs[0].second;
    basis[1] = normalize(eigs[1].second - basis[0] * dot(eigs[1].second, basis[0]));
    if (norm2(basis[1]) < 1e-12) basis[1] = Vec3{0, 1, 0};
    basis[2] = normalize(cross(basis[0], basis[1]));
    if (norm2(basis[2]) < 1e-12) basis = {Vec3{1, 0, 0}, Vec3{0, 1, 0}, Vec3{0, 0, 1}};
    return basis;
}

static vector<Vec3> buildSupportDirections() {
    vector<Vec3> dirs;
    for (int x = -1; x <= 1; ++x) {
        for (int y = -1; y <= 1; ++y) {
            for (int z = -1; z <= 1; ++z) {
                if (x == 0 && y == 0 && z == 0) continue;
                dirs.push_back(normalize(Vec3{static_cast<double>(x), static_cast<double>(y), static_cast<double>(z)}));
            }
        }
    }
    const double g = 1.6180339887498948482;
    vector<Vec3> extra = {
        {0, 1, g}, {0, -1, g}, {0, 1, -g}, {0, -1, -g},
        {1, g, 0}, {-1, g, 0}, {1, -g, 0}, {-1, -g, 0},
        {g, 0, 1}, {g, 0, -1}, {-g, 0, 1}, {-g, 0, -1}
    };
    for (auto d : extra) dirs.push_back(normalize(d));
    return dirs;
}

static vector<Vec3> reduceVertices(const vector<Vec3>& pts) {
    if (pts.size() <= 96) return pts;

    vector<Vec3> dirs = buildSupportDirections();
    set<int> picked;
    for (const auto& d : dirs) {
        double bestMax = -numeric_limits<double>::infinity();
        double bestMin = numeric_limits<double>::infinity();
        int imax = 0, imin = 0;
        for (int i = 0; i < static_cast<int>(pts.size()); ++i) {
            double proj = dot(pts[i], d);
            if (proj > bestMax) { bestMax = proj; imax = i; }
            if (proj < bestMin) { bestMin = proj; imin = i; }
        }
        picked.insert(imax);
        picked.insert(imin);
    }

    vector<Vec3> reduced;
    reduced.reserve(picked.size());
    for (int idx : picked) reduced.push_back(pts[idx]);

    if (reduced.size() >= 4) return reduced;
    return pts;
}

static vector<Plane> buildConvexHullPlanes(const vector<Vec3>& pts) {
    vector<Plane> planes;
    if (pts.size() < 4) return planes;

    Vec3 centroid;
    for (const auto& p : pts) centroid += p;
    centroid = centroid / static_cast<double>(pts.size());

    const double scaleRef = max(1.0, sqrt(norm2(centroid)) + 1.0);
    const double eps = 1e-9 * scaleRef;
    const int n = static_cast<int>(pts.size());

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 normal = cross(pts[j] - pts[i], pts[k] - pts[i]);
                double len = norm(normal);
                if (len < 1e-12) continue;
                normal = normal / len;
                double d = -dot(normal, pts[i]);

                bool pos = false, neg = false;
                for (int t = 0; t < n; ++t) {
                    double s = dot(normal, pts[t]) + d;
                    if (s > eps) pos = true;
                    if (s < -eps) neg = true;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                if (dot(normal, centroid) + d > 0.0) {
                    normal = normal * -1.0;
                    d = -d;
                }

                bool duplicate = false;
                for (const auto& pl : planes) {
                    if (fabs(dot(pl.n, normal) - 1.0) < 1e-7 && fabs(pl.d - d) < 1e-7) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) planes.push_back({normal, d});
            }
        }
    }
    return planes;
}

static bool insideHull(const vector<Plane>& planes, const Vec3& p, double eps) {
    for (const auto& plane : planes) {
        if (dot(plane.n, p) + plane.d > eps) return false;
    }
    return true;
}

static array<Vec3, 4> baseTetra(double edge) {
    double s = edge / (2.0 * sqrt(2.0));
    return {
        Vec3{ s,  s,  s},
        Vec3{-s, -s,  s},
        Vec3{-s,  s, -s},
        Vec3{ s, -s, -s}
    };
}

static bool tetraFits(const vector<Plane>& planes, const array<Vec3, 4>& verts, const Vec3& c, double eps) {
    for (const auto& v : verts) {
        if (!insideHull(planes, c + v, eps)) return false;
    }
    return true;
}

static vector<Vec3> packFCC(
    const vector<Plane>& planes,
    const vector<Vec3>& hullPts,
    const array<Vec3, 3>& basis,
    const array<Vec3, 4>& tetraVerts,
    double edge,
    int offsetMode
) {
    vector<Vec3> placed;
    if (planes.empty()) return placed;

    Vec3 mn{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    Vec3 mx{-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity()};
    for (const auto& p : hullPts) {
        Vec3 q = toLocal(basis, p);
        mn.x = min(mn.x, q.x); mn.y = min(mn.y, q.y); mn.z = min(mn.z, q.z);
        mx.x = max(mx.x, q.x); mx.y = max(mx.y, q.y); mx.z = max(mx.z, q.z);
    }

    const double R = edge * sqrt(3.0 / 8.0);
    const double a = edge * sqrt(3.0);
    const double eps = 1e-8 * max(1.0, edge);
    const array<Vec3, 4> fccBasis = {
        Vec3{0.0, 0.0, 0.0},
        Vec3{0.0, 0.5, 0.5},
        Vec3{0.5, 0.0, 0.5},
        Vec3{0.5, 0.5, 0.0}
    };

    vector<Vec3> offsets = {
        {0.00, 0.00, 0.00},
        {0.25, 0.25, 0.25},
        {0.50, 0.00, 0.00},
        {0.00, 0.50, 0.00},
        {0.00, 0.00, 0.50},
        {0.50, 0.50, 0.50}
    };
    Vec3 offset = offsets[min(offsetMode, static_cast<int>(offsets.size()) - 1)];

    int ix0 = static_cast<int>(floor((mn.x - R) / a)) - 1;
    int iy0 = static_cast<int>(floor((mn.y - R) / a)) - 1;
    int iz0 = static_cast<int>(floor((mn.z - R) / a)) - 1;
    int ix1 = static_cast<int>(ceil((mx.x + R) / a)) + 1;
    int iy1 = static_cast<int>(ceil((mx.y + R) / a)) + 1;
    int iz1 = static_cast<int>(ceil((mx.z + R) / a)) + 1;

    placed.reserve(static_cast<size_t>(max(0, (ix1 - ix0) * (iy1 - iy0) * (iz1 - iz0))));

    for (int i = ix0; i <= ix1; ++i) {
        for (int j = iy0; j <= iy1; ++j) {
            for (int k = iz0; k <= iz1; ++k) {
                for (const auto& b : fccBasis) {
                    Vec3 q{
                        (static_cast<double>(i) + b.x + offset.x) * a,
                        (static_cast<double>(j) + b.y + offset.y) * a,
                        (static_cast<double>(k) + b.z + offset.z) * a
                    };
                    Vec3 c = mul(basis, q);
                    if (tetraFits(planes, tetraVerts, c, eps)) placed.push_back(c);
                }
            }
        }
    }
    return placed;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge)) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Vec3> pts;
    pts.reserve(max(0, V));
    for (int i = 0; i < V; ++i) {
        Vec3 p;
        cin >> p.x >> p.y >> p.z;
        pts.push_back(p);
    }

    if (pts.size() < 4 || edge <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Vec3> reduced = reduceVertices(pts);
    vector<Plane> planes = buildConvexHullPlanes(reduced);
    if (planes.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    array<Vec3, 3> basis = orthonormalBasisFromCovariance(reduced);
    array<Vec3, 4> tetraVerts = baseTetra(edge);

    Vec3 mn{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    Vec3 mx{-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity()};
    for (const auto& p : reduced) {
        Vec3 q = toLocal(basis, p);
        mn.x = min(mn.x, q.x); mn.y = min(mn.y, q.y); mn.z = min(mn.z, q.z);
        mx.x = max(mx.x, q.x); mx.y = max(mx.y, q.y); mx.z = max(mx.z, q.z);
    }

    const double a = edge * sqrt(3.0);
    double bboxVol = max(0.0, (mx.x - mn.x) * (mx.y - mn.y) * (mx.z - mn.z));
    double estFCC = (a > 0.0) ? (bboxVol * 4.0 / (a * a * a)) : 0.0;

    int offsetsToTry = 1;
    if (estFCC < 3000.0) offsetsToTry = 6;
    else if (estFCC < 30000.0) offsetsToTry = 3;
    else if (estFCC < 200000.0) offsetsToTry = 2;

    vector<Vec3> best;
    for (int mode = 0; mode < offsetsToTry; ++mode) {
        vector<Vec3> cur = packFCC(planes, reduced, basis, tetraVerts, edge, mode);
        if (cur.size() > best.size()) best.swap(cur);
    }

    cout << best.size() << '\n';
    cout << fixed << setprecision(10);
    for (const auto& c : best) {
        cout << c.x << ' ' << c.y << ' ' << c.z << " 0 0 0\n";
    }
    return 0;
}
