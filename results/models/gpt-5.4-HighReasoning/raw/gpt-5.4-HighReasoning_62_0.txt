declare const require: any;
const fs = require('fs');

type Point = [number, number];

interface TownInput {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: Point[];
  towns: TownInput[];
  cutBudget: number;
}

interface BfsResult {
  dist: Int32Array;
  parent: Int32Array;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const cellCount = width * height;

const burnable = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = data.grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) {
      burnable[y * width + x] = 1;
    }
  }
}

const ignitionCell = new Uint8Array(cellCount);
const ignitionIndices: number[] = [];
for (const [x, y] of data.ignitions) {
  if (x >= 0 && x < width && y >= 0 && y < height) {
    const idx = y * width + x;
    if (burnable[idx] && !ignitionCell[idx]) {
      ignitionCell[idx] = 1;
      ignitionIndices.push(idx);
    }
  }
}

const townValue = new Float64Array(cellCount);
const townCell = new Uint8Array(cellCount);
for (const town of data.towns) {
  if (town.x >= 0 && town.x < width && town.y >= 0 && town.y < height) {
    const idx = town.y * width + town.x;
    townValue[idx] += Number(town.value) || 0;
    townCell[idx] = 1;
  }
}

function bfs(cut: Uint8Array): BfsResult {
  const dist = new Int32Array(cellCount);
  const parent = new Int32Array(cellCount);
  dist.fill(-1);
  parent.fill(-1);

  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  for (const idx of ignitionIndices) {
    if (!cut[idx] && burnable[idx] && dist[idx] === -1) {
      dist[idx] = 0;
      queue[tail++] = idx;
    }
  }

  while (head < tail) {
    const u = queue[head++];
    const ux = u % width;
    const uy = (u / width) | 0;
    const nd = dist[u] + 1;

    if (ux > 0) {
      const v = u - 1;
      if (!cut[v] && burnable[v] && dist[v] === -1) {
        dist[v] = nd;
        parent[v] = u;
        queue[tail++] = v;
      }
    }
    if (ux + 1 < width) {
      const v = u + 1;
      if (!cut[v] && burnable[v] && dist[v] === -1) {
        dist[v] = nd;
        parent[v] = u;
        queue[tail++] = v;
      }
    }
    if (uy > 0) {
      const v = u - width;
      if (!cut[v] && burnable[v] && dist[v] === -1) {
        dist[v] = nd;
        parent[v] = u;
        queue[tail++] = v;
      }
    }
    if (uy + 1 < height) {
      const v = u + width;
      if (!cut[v] && burnable[v] && dist[v] === -1) {
        dist[v] = nd;
        parent[v] = u;
        queue[tail++] = v;
      }
    }
  }

  return { dist, parent };
}

function computeArticulationBestCut(cut: Uint8Array, dist: Int32Array): number {
  const disc = new Int32Array(cellCount);
  const low = new Int32Array(cellCount);
  const parent = new Int32Array(cellCount);
  const nextDir = new Int8Array(cellCount);
  const childCount = new Int32Array(cellCount);
  disc.fill(-1);
  parent.fill(-1);

  const subtreeTown = new Float64Array(cellCount);
  const subtreeIgnitions = new Int32Array(cellCount);
  const gain = new Float64Array(cellCount);

  const stack = new Int32Array(cellCount);
  let time = 0;
  let bestIdx = -1;
  let bestGain = 0;
  let bestDegree = 5;

  for (const root of ignitionIndices) {
    if (cut[root] || !burnable[root] || dist[root] === -1 || disc[root] !== -1) {
      continue;
    }

    let top = 0;
    stack[top++] = root;
    parent[root] = -1;

    while (top > 0) {
      const u = stack[top - 1];
      if (disc[u] === -1) {
        disc[u] = low[u] = time++;
        subtreeTown[u] = townValue[u];
        subtreeIgnitions[u] = ignitionCell[u] ? 1 : 0;
        nextDir[u] = 0;
      }

      if (nextDir[u] < 4) {
        const dir = nextDir[u]++;
        let v = -1;
        const ux = u % width;
        const uy = (u / width) | 0;

        if (dir === 0) {
          if (ux > 0) v = u - 1;
        } else if (dir === 1) {
          if (ux + 1 < width) v = u + 1;
        } else if (dir === 2) {
          if (uy > 0) v = u - width;
        } else {
          if (uy + 1 < height) v = u + width;
        }

        if (v === -1 || cut[v] || !burnable[v] || dist[v] === -1) {
          continue;
        }

        if (disc[v] === -1) {
          parent[v] = u;
          childCount[u]++;
          stack[top++] = v;
          continue;
        }

        if (v !== parent[u] && disc[v] < low[u]) {
          low[u] = disc[v];
        }
        continue;
      }

      top--;
      const p = parent[u];
      if (p !== -1) {
        subtreeTown[p] += subtreeTown[u];
        subtreeIgnitions[p] += subtreeIgnitions[u];
        if (low[u] < low[p]) {
          low[p] = low[u];
        }
        if (low[u] >= disc[p] && subtreeIgnitions[u] === 0) {
          gain[p] += subtreeTown[u];
        }
      }
    }
  }

  for (let idx = 0; idx < cellCount; idx++) {
    if (gain[idx] <= 0 || cut[idx] || !burnable[idx] || ignitionCell[idx] || townCell[idx] || dist[idx] === -1) {
      continue;
    }

    let degree = 0;
    const x = idx % width;
    const y = (idx / width) | 0;
    if (x > 0) {
      const v = idx - 1;
      if (!cut[v] && burnable[v] && dist[v] !== -1) degree++;
    }
    if (x + 1 < width) {
      const v = idx + 1;
      if (!cut[v] && burnable[v] && dist[v] !== -1) degree++;
    }
    if (y > 0) {
      const v = idx - width;
      if (!cut[v] && burnable[v] && dist[v] !== -1) degree++;
    }
    if (y + 1 < height) {
      const v = idx + width;
      if (!cut[v] && burnable[v] && dist[v] !== -1) degree++;
    }

    if (gain[idx] > bestGain || (gain[idx] === bestGain && degree < bestDegree)) {
      bestGain = gain[idx];
      bestIdx = idx;
      bestDegree = degree;
    }
  }

  return bestIdx;
}

function computeFallbackBestCut(cut: Uint8Array, dist: Int32Array, parent: Int32Array): number {
  const score = new Float64Array(cellCount);

  for (let idx = 0; idx < cellCount; idx++) {
    if (townValue[idx] <= 0 || dist[idx] === -1) {
      continue;
    }

    const value = townValue[idx];
    let cur = idx;
    let steps = 0;
    while (cur !== -1 && !ignitionCell[cur]) {
      const p = parent[cur];
      if (p === -1) {
        break;
      }
      cur = p;
      steps++;
      if (cur === -1 || ignitionCell[cur] || townCell[cur] || cut[cur]) {
        continue;
      }
      score[cur] += value / (1 + steps * 0.35);
    }
  }

  let bestIdx = -1;
  let bestScore = 0;
  let bestDist = -1;

  for (let idx = 0; idx < cellCount; idx++) {
    if (score[idx] <= 0 || cut[idx] || !burnable[idx] || ignitionCell[idx] || townCell[idx] || dist[idx] === -1) {
      continue;
    }

    if (score[idx] > bestScore || (score[idx] === bestScore && dist[idx] > bestDist)) {
      bestScore = score[idx];
      bestIdx = idx;
      bestDist = dist[idx];
    }
  }

  return bestIdx;
}

const cutBudget = Math.max(0, data.cutBudget | 0);
const cut = new Uint8Array(cellCount);
const chosen: number[] = [];
const maxIterations = Math.min(cutBudget, 64);

for (let iter = 0; iter < maxIterations && chosen.length < cutBudget; iter++) {
  const state = bfs(cut);
  let threatenedTownValue = 0;
  for (let idx = 0; idx < cellCount; idx++) {
    if (townValue[idx] > 0 && state.dist[idx] !== -1) {
      threatenedTownValue += townValue[idx];
    }
  }
  if (threatenedTownValue <= 0) {
    break;
  }

  let nextCut = computeArticulationBestCut(cut, state.dist);
  if (nextCut === -1) {
    nextCut = computeFallbackBestCut(cut, state.dist, state.parent);
  }
  if (nextCut === -1) {
    break;
  }

  cut[nextCut] = 1;
  chosen.push(nextCut);
}

const firebreaks: Point[] = chosen.map((idx) => [idx % width, (idx / width) | 0]);
fs.writeSync(1, JSON.stringify({ firebreaks }));
