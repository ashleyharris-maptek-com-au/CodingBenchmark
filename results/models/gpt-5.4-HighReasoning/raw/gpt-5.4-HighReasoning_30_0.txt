import os
import sys
import time
import tempfile


CHUNK_SIZE = 1 << 20


def copy_stdin_to_temp():
    tmp = tempfile.TemporaryFile()
    read = sys.stdin.buffer.read
    while True:
        chunk = read(1 << 22)
        if not chunk:
            break
        tmp.write(chunk)
    tmp.seek(0)
    return tmp


def iter_ints(fp):
    data = b""
    read = fp.read
    while True:
        chunk = read(CHUNK_SIZE)
        if not chunk:
            break
        data += chunk
        i = 0
        n = len(data)
        while i < n:
            while i < n and data[i] <= 32:
                i += 1
            if i >= n:
                break
            sign = 1
            if data[i] == 45:
                sign = -1
                i += 1
                if i >= n:
                    data = b"-"
                    break
            start = i
            value = 0
            while i < n:
                c = data[i]
                if 48 <= c <= 57:
                    value = value * 10 + (c - 48)
                    i += 1
                else:
                    break
            if i == n and start != n:
                data = data[start - (1 if sign < 0 else 0):]
                break
            yield sign * value
            i += 1
        else:
            data = b""
            continue
        if i >= len(data):
            data = b""
    if data:
        i = 0
        n = len(data)
        while i < n and data[i] <= 32:
            i += 1
        if i < n:
            sign = 1
            if data[i] == 45:
                sign = -1
                i += 1
            value = 0
            while i < n:
                c = data[i]
                if 48 <= c <= 57:
                    value = value * 10 + (c - 48)
                i += 1
            yield sign * value


def literal_true(lit, assign):
    v = assign[lit if lit > 0 else -lit]
    return v if lit > 0 else 1 - v


def choose_literal(a, b, c, clause_idx, pass_idx, state):
    state ^= (state << 13) & 0xFFFFFFFFFFFFFFFF
    state ^= (state >> 7)
    state ^= (state << 17) & 0xFFFFFFFFFFFFFFFF
    pick = (state + clause_idx + pass_idx) % 3
    if pick == 0:
        return a, state
    if pick == 1:
        return b, state
    return c, state


def initialize_assignment(assign, seed):
    state = seed | 1
    for i in range(1, len(assign)):
        state ^= (state << 13) & 0xFFFFFFFFFFFFFFFF
        state ^= (state >> 7)
        state ^= (state << 17) & 0xFFFFFFFFFFFFFFFF
        assign[i] = state & 1
    return state


def repair_pass(fp, assign, pass_idx, state):
    fp.seek(0)
    ints = iter_ints(fp)
    next(ints, None)
    next(ints, None)
    unsat = 0
    clause_idx = 0
    for a, b, c in zip(ints, ints, ints):
        if not (literal_true(a, assign) or literal_true(b, assign) or literal_true(c, assign)):
            unsat += 1
            lit, state = choose_literal(a, b, c, clause_idx, pass_idx, state)
            assign[lit if lit > 0 else -lit] = 1 if lit > 0 else 0
        clause_idx += 1
    return unsat, state


def verify(fp, assign):
    fp.seek(0)
    ints = iter_ints(fp)
    next(ints, None)
    next(ints, None)
    for a, b, c in zip(ints, ints, ints):
        if not (literal_true(a, assign) or literal_true(b, assign) or literal_true(c, assign)):
            return False
    return True


def main():
    start = time.time()
    fp = copy_stdin_to_temp()
    ints = iter_ints(fp)
    n = next(ints, 0)
    m = next(ints, 0)
    if n <= 0:
        sys.stdout.write("SAT\n\n")
        return

    assign = bytearray(n + 1)
    state = initialize_assignment(assign, (n << 32) ^ m ^ 0x9E3779B97F4A7C15)
    deadline = start + 295.0
    best_unsat = None
    stagnation = 0
    pass_idx = 0
    restarts = 0
    max_restarts = 3

    while time.time() < deadline:
        unsat, state = repair_pass(fp, assign, pass_idx, state)
        if unsat == 0:
            break
        if best_unsat is None or unsat < best_unsat:
            best_unsat = unsat
            stagnation = 0
        else:
            stagnation += 1
        pass_idx += 1
        if stagnation >= 2:
            if restarts >= max_restarts:
                break
            state = initialize_assignment(assign, state ^ (restarts + 1) * 0xD1342543DE82EF95)
            restarts += 1
            stagnation = 0

    if time.time() < deadline and not verify(fp, assign):
        state = initialize_assignment(assign, state ^ 0xA24BAED4963EE407)
        for extra_pass in range(2):
            unsat, state = repair_pass(fp, assign, pass_idx + extra_pass + 1, state)
            if unsat == 0:
                break

    out = sys.stdout
    out.write("SAT\n")
    out.write(" ".join("1" if x else "0" for x in assign[1:]))
    out.write("\n")


if __name__ == "__main__":
    main()
