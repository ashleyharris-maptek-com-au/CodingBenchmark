struct MaxCliqueSolver {
    int n;
    vector<vector<uint64_t>> adj;
    vector<int> best;
    vector<int> current;
    int words;

    explicit MaxCliqueSolver(const vector<vector<int>>& g) {
        n = (int)g.size();
        words = (n + 63) >> 6;
        adj.assign(n, vector<uint64_t>(words, 0));
        for (int v = 0; v < n; ++v) {
            for (int u : g[v]) {
                adj[v][u >> 6] |= 1ULL << (u & 63);
            }
        }
    }

    bool connected(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    vector<int> intersect_with_neighbors(const vector<int>& verts, int v) const {
        vector<int> out;
        out.reserve(verts.size());
        for (int u : verts) {
            if (connected(v, u)) out.push_back(u);
        }
        return out;
    }

    void color_sort(const vector<int>& verts, vector<int>& order, vector<int>& bounds) const {
        order.clear();
        bounds.clear();
        vector<vector<int>> color_classes;
        color_classes.reserve(verts.size());

        for (int v : verts) {
            int c = 0;
            for (; c < (int)color_classes.size(); ++c) {
                bool ok = true;
                for (int u : color_classes[c]) {
                    if (connected(u, v)) {
                        ok = false;
                        break;
                    }
                }
                if (ok) break;
            }
            if (c == (int)color_classes.size()) color_classes.push_back({});
            color_classes[c].push_back(v);
        }

        for (int c = 0; c < (int)color_classes.size(); ++c) {
            for (int v : color_classes[c]) {
                order.push_back(v);
                bounds.push_back(c + 1);
            }
        }
    }

    void expand(vector<int>& verts) {
        if (verts.empty()) {
            if (current.size() > best.size()) best = current;
            return;
        }

        vector<int> order, bounds;
        color_sort(verts, order, bounds);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            if ((int)current.size() + bounds[i] <= (int)best.size()) return;
            int v = order[i];
            current.push_back(v);
            vector<int> next;
            next.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (connected(v, u)) next.push_back(u);
            }
            if (next.empty()) {
                if (current.size() > best.size()) best = current;
            } else {
                expand(next);
            }
            current.pop_back();
        }
    }

    vector<int> solve() {
        vector<int> verts(n);
        iota(verts.begin(), verts.end(), 0);
        expand(verts);
        return best;
    }
};

struct VertexCoverSolver {
    int n;
    vector<vector<int>> adj;
    vector<int> best_cover;
    vector<int> mark;
    int stamp = 1;

    explicit VertexCoverSolver(vector<vector<int>> g) : n((int)g.size()), adj(std::move(g)), mark(n, 0) {}

    bool in_set(int v) const {
        return mark[v] == stamp;
    }

    void set_marks(const vector<int>& verts) {
        ++stamp;
        if (stamp == numeric_limits<int>::max()) {
            fill(mark.begin(), mark.end(), 0);
            stamp = 1;
        }
        for (int v : verts) mark[v] = stamp;
    }

    int greedy_matching_lb(const vector<int>& verts) {
        set_marks(verts);
        vector<char> used(n, 0);
        int lb = 0;
        for (int v : verts) {
            if (used[v]) continue;
            for (int u : adj[v]) {
                if (in_set(u) && !used[u]) {
                    used[v] = used[u] = 1;
                    ++lb;
                    break;
                }
            }
        }
        return lb;
    }

    vector<int> greedy_upper_bound(const vector<int>& verts) {
        set_marks(verts);
        vector<int> degree(n, 0);
        int edges = 0;
        for (int v : verts) {
            int d = 0;
            for (int u : adj[v]) {
                if (in_set(u)) ++d;
            }
            degree[v] = d;
            edges += d;
        }
        edges >>= 1;

        vector<char> alive(n, 0);
        for (int v : verts) alive[v] = 1;
        vector<int> cover;
        cover.reserve(verts.size());

        while (edges > 0) {
            int best_v = -1;
            int best_d = -1;
            for (int v : verts) {
                if (alive[v] && degree[v] > best_d) {
                    best_d = degree[v];
                    best_v = v;
                }
            }
            if (best_v == -1 || best_d <= 0) break;
            cover.push_back(best_v);
            alive[best_v] = 0;
            edges -= degree[best_v];
            for (int u : adj[best_v]) {
                if (!alive[u]) continue;
                --degree[u];
            }
            degree[best_v] = 0;
        }
        return cover;
    }

    void reduce(vector<int>& verts, vector<int>& forced_cover) {
        while (true) {
            set_marks(verts);
            vector<int> degree(n, 0), one_neighbor(n, -1);
            int edges = 0;
            for (int v : verts) {
                int d = 0;
                int last = -1;
                for (int u : adj[v]) {
                    if (in_set(u)) {
                        ++d;
                        last = u;
                    }
                }
                degree[v] = d;
                one_neighbor[v] = last;
                edges += d;
            }
            edges >>= 1;
            if (edges == 0) {
                verts.clear();
                return;
            }

            vector<int> next;
            next.reserve(verts.size());
            bool changed = false;

            vector<char> removed(n, 0);
            for (int v : verts) {
                if (degree[v] == 0) {
                    removed[v] = 1;
                    changed = true;
                }
            }
            if (changed) {
                for (int v : verts) {
                    if (!removed[v]) next.push_back(v);
                }
                verts.swap(next);
                continue;
            }

            for (int v : verts) {
                if (degree[v] == 1) {
                    int u = one_neighbor[v];
                    if (u != -1 && !removed[u]) {
                        removed[u] = 1;
                        forced_cover.push_back(u);
                        changed = true;
                    }
                }
            }
            if (changed) {
                next.clear();
                next.reserve(verts.size());
                for (int v : verts) {
                    if (!removed[v]) next.push_back(v);
                }
                verts.swap(next);
                continue;
            }
            return;
        }
    }

    vector<vector<int>> split_components(const vector<int>& verts) {
        set_marks(verts);
        vector<char> seen(n, 0);
        vector<vector<int>> comps;
        for (int s : verts) {
            if (seen[s]) continue;
            queue<int> q;
            q.push(s);
            seen[s] = 1;
            vector<int> comp;
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                comp.push_back(v);
                for (int u : adj[v]) {
                    if (in_set(u) && !seen[u]) {
                        seen[u] = 1;
                        q.push(u);
                    }
                }
            }
            comps.push_back(std::move(comp));
        }
        return comps;
    }

    void dfs(vector<int> verts, vector<int> current_cover) {
        reduce(verts, current_cover);
        if (current_cover.size() >= best_cover.size()) return;
        if (verts.empty()) {
            best_cover = std::move(current_cover);
            return;
        }

        int lb = greedy_matching_lb(verts);
        if (current_cover.size() + lb >= best_cover.size()) return;

        auto comps = split_components(verts);
        if (comps.size() > 1) {
            vector<int> combined = current_cover;
            for (auto& comp : comps) {
                VertexCoverSolver sub(build_induced(comp));
                vector<int> part = sub.solve_cover();
                if (combined.size() + part.size() >= best_cover.size()) return;
                for (int idx : part) combined.push_back(comp[idx]);
            }
            if (combined.size() < best_cover.size()) best_cover = std::move(combined);
            return;
        }

        set_marks(verts);
        int branch_v = -1;
        int branch_deg = -1;
        vector<int> neighbors;
        for (int v : verts) {
            int d = 0;
            for (int u : adj[v]) {
                if (in_set(u)) ++d;
            }
            if (d > branch_deg) {
                branch_deg = d;
                branch_v = v;
            }
        }
        if (branch_v == -1) {
            if (current_cover.size() < best_cover.size()) best_cover = std::move(current_cover);
            return;
        }
        for (int u : adj[branch_v]) {
            if (in_set(u)) neighbors.push_back(u);
        }

        {
            vector<int> next_cover = current_cover;
            next_cover.push_back(branch_v);
            vector<int> next_verts;
            next_verts.reserve(verts.size() - 1);
            for (int v : verts) {
                if (v != branch_v) next_verts.push_back(v);
            }
            dfs(std::move(next_verts), std::move(next_cover));
        }

        {
            vector<char> take_neighbor(n, 0);
            for (int u : neighbors) take_neighbor[u] = 1;
            vector<int> next_cover = current_cover;
            next_cover.insert(next_cover.end(), neighbors.begin(), neighbors.end());
            if (next_cover.size() >= best_cover.size()) return;
            vector<int> next_verts;
            next_verts.reserve(verts.size());
            for (int v : verts) {
                if (v != branch_v && !take_neighbor[v]) next_verts.push_back(v);
            }
            dfs(std::move(next_verts), std::move(next_cover));
        }
    }

    vector<vector<int>> build_induced(const vector<int>& verts) {
        int sz = (int)verts.size();
        vector<int> pos(n, -1);
        for (int i = 0; i < sz; ++i) pos[verts[i]] = i;
        vector<vector<int>> g(sz);
        for (int i = 0; i < sz; ++i) {
            int v = verts[i];
            for (int u : adj[v]) {
                int j = pos[u];
                if (j != -1) g[i].push_back(j);
            }
        }
        return g;
    }

    vector<int> solve_cover() {
        vector<int> verts(n);
        iota(verts.begin(), verts.end(), 0);
        best_cover = greedy_upper_bound(verts);
        dfs(verts, {});
        sort(best_cover.begin(), best_cover.end());
        best_cover.erase(unique(best_cover.begin(), best_cover.end()), best_cover.end());
        return best_cover;
    }
};

static vector<vector<int>> complement_graph(const vector<vector<int>>& g) {
    int n = (int)g.size();
    vector<vector<int>> comp(n);
    vector<char> mark(n, 0);
    for (int v = 0; v < n; ++v) {
        for (int u : g[v]) mark[u] = 1;
        mark[v] = 1;
        for (int u = 0; u < n; ++u) {
            if (!mark[u]) comp[v].push_back(u);
        }
        for (int u : g[v]) mark[u] = 0;
        mark[v] = 0;
    }
    return comp;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for (int v = 0; v < n; ++v) {
        sort(g[v].begin(), g[v].end());
        g[v].erase(unique(g[v].begin(), g[v].end()), g[v].end());
    }

    vector<int> comp_id(n, -1);
    vector<vector<int>> comps;
    for (int s = 0; s < n; ++s) {
        if (comp_id[s] != -1) continue;
        queue<int> q;
        q.push(s);
        comp_id[s] = (int)comps.size();
        comps.push_back({});
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            comps.back().push_back(v);
            for (int u : g[v]) {
                if (comp_id[u] == -1) {
                    comp_id[u] = comp_id[s];
                    q.push(u);
                }
            }
        }
    }

    vector<int> answer;
    answer.reserve(n);

    for (const auto& comp_vertices : comps) {
        if (comp_vertices.empty()) continue;
        if (comp_vertices.size() == 1) {
            answer.push_back(comp_vertices[0]);
            continue;
        }

        int sz = (int)comp_vertices.size();
        vector<int> pos(n, -1);
        for (int i = 0; i < sz; ++i) pos[comp_vertices[i]] = i;

        vector<vector<int>> subg(sz);
        long long edges = 0;
        for (int i = 0; i < sz; ++i) {
            int v = comp_vertices[i];
            for (int u : g[v]) {
                int j = pos[u];
                if (j != -1) subg[i].push_back(j);
            }
            edges += (int)subg[i].size();
        }
        edges >>= 1;

        vector<int> local_is;
        double density = sz <= 1 ? 0.0 : (2.0 * edges) / (1.0 * sz * (sz - 1));
        if (sz <= 120 || density >= 0.55) {
            auto compg = complement_graph(subg);
            MaxCliqueSolver solver(compg);
            local_is = solver.solve();
        } else {
            VertexCoverSolver solver(subg);
            vector<int> cover = solver.solve_cover();
            vector<char> in_cover(sz, 0);
            for (int v : cover) in_cover[v] = 1;
            for (int v = 0; v < sz; ++v) {
                if (!in_cover[v]) local_is.push_back(v);
            }
        }

        for (int v : local_is) answer.push_back(comp_vertices[v]);
    }

    sort(answer.begin(), answer.end());
    cout << answer.size() << '\n';
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
