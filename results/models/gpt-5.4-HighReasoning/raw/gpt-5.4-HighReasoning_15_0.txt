use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1.0e-9;
const MAX_SCALE: f64 = 10.0;
const HEIGHT_STEP: f64 = 30.0;

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn cross(self, other: Self) -> f64 {
        self.x * other.y - self.y * other.x
    }
}

impl std::ops::Add for Vec2 {
    type Output = Self;
    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Vec2 {
    type Output = Self;
    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl std::ops::Mul<f64> for Vec2 {
    type Output = Self;
    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs)
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
}

impl std::ops::Add for Vec3 {
    type Output = Self;
    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y, self.z + rhs.z)
    }
}

impl std::ops::Sub for Vec3 {
    type Output = Self;
    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y, self.z - rhs.z)
    }
}

impl std::ops::Mul<f64> for Vec3 {
    type Output = Self;
    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs, self.z * rhs)
    }
}

#[derive(Clone, Copy, Debug)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quat {
    fn identity() -> Self {
        Self {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }

    fn normalized(self) -> Self {
        let n = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if n <= EPS {
            return Self::identity();
        }
        Self {
            w: self.w / n,
            x: self.x / n,
            y: self.y / n,
            z: self.z / n,
        }
    }

    fn from_euler(rx: f64, ry: f64, rz: f64) -> Self {
        let (sx, cx) = (0.5 * rx).sin_cos();
        let (sy, cy) = (0.5 * ry).sin_cos();
        let (sz, cz) = (0.5 * rz).sin_cos();
        Self {
            w: cx * cy * cz + sx * sy * sz,
            x: sx * cy * cz - cx * sy * sz,
            y: cx * sy * cz + sx * cy * sz,
            z: cx * cy * sz - sx * sy * cz,
        }
        .normalized()
    }

    fn rotate(self, v: Vec3) -> Vec3 {
        let qv = Vec3::new(self.x, self.y, self.z);
        let uv = cross3(qv, v);
        let uuv = cross3(qv, uv);
        v + uv * (2.0 * self.w) + uuv * 2.0
    }
}

fn cross3(a: Vec3, b: Vec3) -> Vec3 {
    Vec3::new(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x,
    )
}

#[derive(Clone)]
struct Candidate {
    quat: Quat,
    anchor: Vec2,
    half_w: f64,
    half_h: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut poly = Vec::with_capacity(nv);
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for _ in 0..nv {
        let x = next_f64(&mut it);
        let y = next_f64(&mut it);
        poly.push(Vec2::new(x, y));
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    let sx = next_f64(&mut it);
    let sy = next_f64(&mut it);
    let mut sz = next_f64(&mut it);
    if sz.abs() < 1.0e-6 {
        sz = -1.0e-6;
    }
    if sz > 0.0 {
        sz = -sz;
    }

    let sun = Vec3::new(sx, sy, sz);
    let candidate = choose_candidate(sun);
    let scale = MAX_SCALE;
    let cell_w = (2.0 * candidate.half_w * scale).max(1.0e-3);
    let cell_h = (2.0 * candidate.half_h * scale).max(1.0e-3);

    let width = (max_x - min_x).max(0.0);
    let height = (max_y - min_y).max(0.0);
    let nx = ((width / cell_w).ceil() as usize).max(1);
    let ny = ((height / cell_h).ceil() as usize).max(1);
    let step_x = if nx == 0 { cell_w } else { width / nx as f64 };
    let step_y = if ny == 0 { cell_h } else { height / ny as f64 };

    let mut centers = Vec::new();
    for iy in 0..ny {
        let gy = min_y + (iy as f64 + 0.5) * step_y;
        for ix in 0..nx {
            let gx = min_x + (ix as f64 + 0.5) * step_x;
            if cell_intersects_polygon(gx, gy, candidate.half_w * scale, candidate.half_h * scale, &poly) {
                centers.push(Vec2::new(gx, gy));
            }
        }
    }

    if centers.is_empty() {
        let gx = 0.5 * (min_x + max_x);
        let gy = 0.5 * (min_y + max_y);
        centers.push(Vec2::new(gx, gy));
    }

    let ux = sun.x / sun.z;
    let uy = sun.y / sun.z;

    println!("{}", centers.len());
    for (i, g) in centers.iter().enumerate() {
        let h = HEIGHT_STEP * i as f64;
        let dx = g.x - scale * candidate.anchor.x;
        let dy = g.y - scale * candidate.anchor.y;
        let px = dx + h * ux;
        let py = dy + h * uy;
        let pz = h;
        println!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
            px,
            py,
            pz,
            candidate.quat.w,
            candidate.quat.x,
            candidate.quat.y,
            candidate.quat.z,
            scale
        );
    }
}

fn next_f64<'a, I>(it: &mut I) -> f64
where
    I: Iterator<Item = &'a str>,
{
    it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0)
}

fn choose_candidate(sun: Vec3) -> Candidate {
    let reference = [
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];

    let mut best: Option<Candidate> = None;
    let pitch_values = [
        -PI / 3.0,
        -PI / 6.0,
        -PI / 12.0,
        0.0,
        PI / 12.0,
        PI / 6.0,
        PI / 3.0,
    ];
    let yaw_step = PI / 4.0;
    let roll_values = [0.0, PI / 4.0, PI / 2.0, 3.0 * PI / 4.0];

    for &pitch in &pitch_values {
        for yaw_idx in 0..8 {
            let yaw = yaw_idx as f64 * yaw_step;
            for &roll in &roll_values {
                let quat = Quat::from_euler(pitch, yaw, roll);
                if let Some(candidate) = build_candidate(quat, sun, &reference) {
                    let score = candidate.half_w * candidate.half_h;
                    let replace = best
                        .as_ref()
                        .map(|old| score > old.half_w * old.half_h)
                        .unwrap_or(true);
                    if replace {
                        best = Some(candidate);
                    }
                }
            }
        }
    }

    best.unwrap_or_else(|| build_candidate(Quat::identity(), sun, &reference).unwrap())
}

fn build_candidate(quat: Quat, sun: Vec3, reference: &[Vec3; 4]) -> Option<Candidate> {
    let mut pts = Vec::with_capacity(reference.len());
    for &v in reference {
        let rv = quat.rotate(v);
        pts.push(project(rv, sun));
    }

    let hull = convex_hull(&pts);
    if hull.len() < 3 {
        return None;
    }

    let mut anchor = Vec2::default();
    for p in &hull {
        anchor = anchor + *p;
    }
    anchor = anchor * (1.0 / hull.len() as f64);

    let (left, right) = horizontal_span(&hull, anchor.y)?;
    let (bottom, top) = vertical_span(&hull, anchor.x)?;
    let half_w = (anchor.x - left).min(right - anchor.x);
    let half_h = (anchor.y - bottom).min(top - anchor.y);

    if half_w <= 1.0e-4 || half_h <= 1.0e-4 {
        return None;
    }

    Some(Candidate {
        quat,
        anchor,
        half_w,
        half_h,
    })
}

fn project(v: Vec3, sun: Vec3) -> Vec2 {
    let t = v.z / sun.z;
    Vec2::new(v.x - t * sun.x, v.y - t * sun.y)
}

fn convex_hull(points: &[Vec2]) -> Vec<Vec2> {
    let mut pts = points.to_vec();
    pts.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal))
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() <= EPS && (a.y - b.y).abs() <= EPS);
    if pts.len() <= 1 {
        return pts;
    }

    let mut lower: Vec<Vec2> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let n = lower.len();
            if (lower[n - 1] - lower[n - 2]).cross(p - lower[n - 1]) <= EPS {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }

    let mut upper: Vec<Vec2> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let n = upper.len();
            if (upper[n - 1] - upper[n - 2]).cross(p - upper[n - 1]) <= EPS {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn horizontal_span(poly: &[Vec2], y: f64) -> Option<(f64, f64)> {
    let mut xs = Vec::new();
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if (a.y - b.y).abs() <= EPS {
            if (y - a.y).abs() <= EPS {
                xs.push(a.x.min(b.x));
                xs.push(a.x.max(b.x));
            }
            continue;
        }
        let ymin = a.y.min(b.y);
        let ymax = a.y.max(b.y);
        if y + EPS < ymin || y - EPS > ymax {
            continue;
        }
        let t = (y - a.y) / (b.y - a.y);
        if (-EPS..=1.0 + EPS).contains(&t) {
            xs.push(a.x + t * (b.x - a.x));
        }
    }
    normalize_span(xs)
}

fn vertical_span(poly: &[Vec2], x: f64) -> Option<(f64, f64)> {
    let mut ys = Vec::new();
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if (a.x - b.x).abs() <= EPS {
            if (x - a.x).abs() <= EPS {
                ys.push(a.y.min(b.y));
                ys.push(a.y.max(b.y));
            }
            continue;
        }
        let xmin = a.x.min(b.x);
        let xmax = a.x.max(b.x);
        if x + EPS < xmin || x - EPS > xmax {
            continue;
        }
        let t = (x - a.x) / (b.x - a.x);
        if (-EPS..=1.0 + EPS).contains(&t) {
            ys.push(a.y + t * (b.y - a.y));
        }
    }
    normalize_span(ys)
}

fn normalize_span(mut values: Vec<f64>) -> Option<(f64, f64)> {
    if values.is_empty() {
        return None;
    }
    values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mut compact: Vec<f64> = Vec::new();
    for v in values {
        if compact.last().map(|u| (u - v).abs() > 1.0e-7).unwrap_or(true) {
            compact.push(v);
        }
    }
    if compact.len() == 1 {
        Some((compact[0], compact[0]))
    } else {
        Some((*compact.first().unwrap(), *compact.last().unwrap()))
    }
}

fn cell_intersects_polygon(cx: f64, cy: f64, half_w: f64, half_h: f64, poly: &[Vec2]) -> bool {
    let rx0 = cx - half_w;
    let rx1 = cx + half_w;
    let ry0 = cy - half_h;
    let ry1 = cy + half_h;

    if point_in_polygon(Vec2::new(cx, cy), poly) {
        return true;
    }

    for &p in poly {
        if p.x >= rx0 - EPS && p.x <= rx1 + EPS && p.y >= ry0 - EPS && p.y <= ry1 + EPS {
            return true;
        }
    }

    let rect = [
        Vec2::new(rx0, ry0),
        Vec2::new(rx1, ry0),
        Vec2::new(rx1, ry1),
        Vec2::new(rx0, ry1),
    ];
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if segment_intersects_rect(a, b, &rect) {
            return true;
        }
    }

    false
}

fn point_in_polygon(p: Vec2, poly: &[Vec2]) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let crosses = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x);
        if crosses {
            inside = !inside;
        }
    }
    inside
}

fn segment_intersects_rect(a: Vec2, b: Vec2, rect: &[Vec2; 4]) -> bool {
    if point_in_rect(a, rect) || point_in_rect(b, rect) {
        return true;
    }
    for i in 0..4 {
        let c = rect[i];
        let d = rect[(i + 1) % 4];
        if segments_intersect(a, b, c, d) {
            return true;
        }
    }
    false
}

fn point_in_rect(p: Vec2, rect: &[Vec2; 4]) -> bool {
    p.x >= rect[0].x - EPS
        && p.x <= rect[2].x + EPS
        && p.y >= rect[0].y - EPS
        && p.y <= rect[2].y + EPS
}

fn segments_intersect(a: Vec2, b: Vec2, c: Vec2, d: Vec2) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if o1.abs() <= EPS && on_segment(a, b, c) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a, b, d) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(c, d, a) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(c, d, b) {
        return true;
    }

    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn orient(a: Vec2, b: Vec2, c: Vec2) -> f64 {
    (b - a).cross(c - a)
}

fn on_segment(a: Vec2, b: Vec2, p: Vec2) -> bool {
    p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}
