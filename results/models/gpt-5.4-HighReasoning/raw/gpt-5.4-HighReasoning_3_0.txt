use std::cmp::Ordering;
use std::collections::{HashMap, VecDeque};
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

struct Scanner {
    data: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            data: input.into_bytes(),
            idx: 0,
        }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut val = 0usize;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            val = val * 10 + (self.data[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        val
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    let components = connected_components(&adj);
    let mut coords = vec![Point::default(); n];

    for comp in &components {
        layout_component(comp, &adj, &mut coords);
    }

    pack_components(&components, &mut coords);

    let mut out = String::with_capacity(n * 32);
    for p in coords {
        out.push_str(&format!("{:.6} {:.6}\n", p.x, p.y));
    }
    print!("{out}");
}

fn connected_components(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut seen = vec![false; n];
    let mut comps = Vec::new();

    for start in 0..n {
        if seen[start] {
            continue;
        }
        seen[start] = true;
        let mut q = VecDeque::new();
        q.push_back(start);
        let mut comp = Vec::new();

        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }

        comps.push(comp);
    }

    comps
}

fn layout_component(nodes: &[usize], adj: &[Vec<usize>], coords: &mut [Point]) {
    let n = nodes.len();
    if n == 0 {
        return;
    }
    if n == 1 {
        coords[nodes[0]] = Point { x: 0.0, y: 0.0 };
        return;
    }

    let degree_sum: usize = nodes.iter().map(|&u| adj[u].len()).sum();
    let edges = degree_sum / 2;
    let density = edges as f64 / n.max(1) as f64;

    if n <= 32 {
        layout_small_force(nodes, adj, coords);
    } else if n <= 2500 && density <= 12.0 {
        layout_layered(nodes, adj, coords);
        let iterations = if n <= 300 { 40 } else if n <= 1000 { 24 } else { 14 };
        force_refine_sparse(nodes, adj, coords, iterations, 3.0);
    } else {
        layout_landmark(nodes, adj, coords);
        let rounds = if n <= 20000 { 6 } else { 3 };
        smooth_component(nodes, adj, coords, rounds, 0.45);
    }

    normalize_component(nodes, coords, 18.0);
}

fn layout_small_force(nodes: &[usize], adj: &[Vec<usize>], coords: &mut [Point]) {
    let n = nodes.len();
    let radius = (n as f64).max(2.0);
    for (i, &u) in nodes.iter().enumerate() {
        let ang = 2.0 * PI * (i as f64) / (n as f64);
        coords[u] = Point {
            x: radius * ang.cos(),
            y: radius * ang.sin(),
        };
    }

    let ideal = 3.0;
    let mut temp = 1.5;
    for _ in 0..120 {
        let mut delta = vec![Point::default(); n];

        for i in 0..n {
            let u = nodes[i];
            for j in (i + 1)..n {
                let v = nodes[j];
                let dx = coords[u].x - coords[v].x;
                let dy = coords[u].y - coords[v].y;
                let dist2 = dx * dx + dy * dy + 1e-6;
                let dist = dist2.sqrt();
                let force = (ideal * ideal) / dist2;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                delta[i].x += fx;
                delta[i].y += fy;
                delta[j].x -= fx;
                delta[j].y -= fy;
            }
        }

        for (i, &u) in nodes.iter().enumerate() {
            for &v in &adj[u] {
                if u < v {
                    let dx = coords[v].x - coords[u].x;
                    let dy = coords[v].y - coords[u].y;
                    let dist = (dx * dx + dy * dy + 1e-6).sqrt();
                    let force = (dist - ideal) * 0.12;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    delta[i].x += fx;
                    delta[i].y += fy;
                    if let Some(j) = nodes.iter().position(|&x| x == v) {
                        delta[j].x -= fx;
                        delta[j].y -= fy;
                    }
                }
            }
        }

        for (i, &u) in nodes.iter().enumerate() {
            let dx = delta[i].x;
            let dy = delta[i].y;
            let len = (dx * dx + dy * dy).sqrt();
            let scale = if len > temp { temp / len } else { 1.0 };
            coords[u].x += dx * scale;
            coords[u].y += dy * scale;
            coords[u].x *= 0.98;
            coords[u].y *= 0.98;
        }
        temp *= 0.985;
    }
}

fn layout_layered(nodes: &[usize], adj: &[Vec<usize>], coords: &mut [Point]) {
    let start = farthest_node(nodes[0], adj).0;
    let root = farthest_node(start, adj).0;
    let dist = bfs_dist(root, adj);
    let pivot = farthest_from_dist(nodes, &dist);
    let dist2 = bfs_dist(pivot, adj);

    let max_layer = nodes.iter().map(|&u| dist[u]).max().unwrap_or(0) as usize;
    let mut layers = vec![Vec::<usize>::new(); max_layer + 1];
    for &u in nodes {
        let d = dist[u].max(0) as usize;
        layers[d].push(u);
    }

    let mut pos = vec![0usize; adj.len()];
    for layer in &mut layers {
        layer.sort_by(|&a, &b| {
            dist2[b]
                .cmp(&dist2[a])
                .then_with(|| adj[b].len().cmp(&adj[a].len()))
                .then_with(|| a.cmp(&b))
        });
        for (i, &u) in layer.iter().enumerate() {
            pos[u] = i;
        }
    }

    for _ in 0..6 {
        for l in 1..layers.len() {
            let (left, right) = layers.split_at_mut(l);
            let current = &mut right[0];
            let prev = &left[l - 1];
            sort_layer_by_barycenter(current, prev, &dist, pos.as_mut_slice(), adj, l as i32 - 1);
        }
        for l in (0..layers.len().saturating_sub(1)).rev() {
            let (left, right) = layers.split_at_mut(l + 1);
            let current = &mut left[l];
            let next = &right[0];
            sort_layer_by_barycenter(current, next, &dist, pos.as_mut_slice(), adj, l as i32 + 1);
        }
    }

    let x_spacing = 2.4;
    let y_spacing = 3.6;
    for (layer_idx, layer) in layers.iter().enumerate() {
        let shift = (layer.len() as f64 - 1.0) * 0.5;
        for (i, &u) in layer.iter().enumerate() {
            coords[u] = Point {
                x: (i as f64 - shift) * x_spacing,
                y: -(layer_idx as f64) * y_spacing,
            };
        }
    }
}

fn sort_layer_by_barycenter(
    layer: &mut [usize],
    _reference_layer: &[usize],
    dist: &[i32],
    pos: &mut [usize],
    adj: &[Vec<usize>],
    target_layer: i32,
) {
    layer.sort_by(|&a, &b| {
        let ba = barycenter(a, dist, pos, adj, target_layer);
        let bb = barycenter(b, dist, pos, adj, target_layer);
        ba.partial_cmp(&bb)
            .unwrap_or(Ordering::Equal)
            .then_with(|| adj[b].len().cmp(&adj[a].len()))
            .then_with(|| a.cmp(&b))
    });
    for (i, &u) in layer.iter().enumerate() {
        pos[u] = i;
    }
}

fn barycenter(u: usize, dist: &[i32], pos: &[usize], adj: &[Vec<usize>], target_layer: i32) -> f64 {
    let mut sum = 0.0;
    let mut count = 0usize;
    for &v in &adj[u] {
        if dist[v] == target_layer {
            sum += pos[v] as f64;
            count += 1;
        }
    }
    if count == 0 {
        pos[u] as f64
    } else {
        sum / count as f64
    }
}

fn layout_landmark(nodes: &[usize], adj: &[Vec<usize>], coords: &mut [Point]) {
    let a = farthest_node(nodes[0], adj).0;
    let da = bfs_dist(a, adj);
    let b = farthest_from_dist(nodes, &da);
    let db = bfs_dist(b, adj);
    let c = farthest_from_dist(nodes, &db);
    let dc = bfs_dist(c, adj);
    let d = farthest_from_dist(nodes, &dc);
    let dd = bfs_dist(d, adj);

    for (i, &u) in nodes.iter().enumerate() {
        let x = (da[u] - db[u]) as f64;
        let y = (dc[u] - dd[u]) as f64;
        coords[u] = Point {
            x: x + (i % 7) as f64 * 0.01,
            y: y + (i % 11) as f64 * 0.01,
        };
    }
}

fn force_refine_sparse(nodes: &[usize], adj: &[Vec<usize>], coords: &mut [Point], iterations: usize, ideal: f64) {
    let mut temp = ideal;
    for _ in 0..iterations {
        let cell = ideal * 2.0;
        let mut grid: HashMap<(i32, i32), Vec<usize>> = HashMap::new();
        for &u in nodes {
            let key = (
                (coords[u].x / cell).floor() as i32,
                (coords[u].y / cell).floor() as i32,
            );
            grid.entry(key).or_default().push(u);
        }

        let mut delta: HashMap<usize, Point> = HashMap::with_capacity(nodes.len());
        for &u in nodes {
            let mut acc = Point::default();
            let cx = (coords[u].x / cell).floor() as i32;
            let cy = (coords[u].y / cell).floor() as i32;

            for dx in -1..=1 {
                for dy in -1..=1 {
                    if let Some(bucket) = grid.get(&(cx + dx, cy + dy)) {
                        for &v in bucket {
                            if u == v {
                                continue;
                            }
                            let rx = coords[u].x - coords[v].x;
                            let ry = coords[u].y - coords[v].y;
                            let dist2 = rx * rx + ry * ry + 1e-6;
                            let dist = dist2.sqrt();
                            let force = (ideal * ideal * 0.35) / dist2;
                            acc.x += rx / dist * force;
                            acc.y += ry / dist * force;
                        }
                    }
                }
            }

            for &v in &adj[u] {
                let rx = coords[v].x - coords[u].x;
                let ry = coords[v].y - coords[u].y;
                let dist = (rx * rx + ry * ry + 1e-6).sqrt();
                let force = (dist - ideal) * 0.09;
                acc.x += rx / dist * force;
                acc.y += ry / dist * force;
            }

            acc.x -= coords[u].x * 0.015;
            acc.y -= coords[u].y * 0.015;
            delta.insert(u, acc);
        }

        for &u in nodes {
            let d = delta[&u];
            let len = (d.x * d.x + d.y * d.y).sqrt();
            let scale = if len > temp { temp / len } else { 1.0 };
            coords[u].x += d.x * scale;
            coords[u].y += d.y * scale;
        }

        temp *= 0.92;
    }
}

fn smooth_component(nodes: &[usize], adj: &[Vec<usize>], coords: &mut [Point], rounds: usize, mix: f64) {
    for _ in 0..rounds {
        let mut next = Vec::with_capacity(nodes.len());
        for &u in nodes {
            if adj[u].is_empty() {
                next.push(coords[u]);
                continue;
            }
            let mut sx = 0.0;
            let mut sy = 0.0;
            for &v in &adj[u] {
                sx += coords[v].x;
                sy += coords[v].y;
            }
            let inv = 1.0 / adj[u].len() as f64;
            let avg = Point {
                x: sx * inv,
                y: sy * inv,
            };
            next.push(Point {
                x: coords[u].x * (1.0 - mix) + avg.x * mix,
                y: coords[u].y * (1.0 - mix) + avg.y * mix,
            });
        }
        for (i, &u) in nodes.iter().enumerate() {
            coords[u] = next[i];
        }
    }
}

fn normalize_component(nodes: &[usize], coords: &mut [Point], target_span: f64) {
    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut sum_x = 0.0;
    let mut sum_y = 0.0;

    for &u in nodes {
        let p = coords[u];
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
        sum_x += p.x;
        sum_y += p.y;
    }

    let cx = sum_x / nodes.len() as f64;
    let cy = sum_y / nodes.len() as f64;
    let span = (max_x - min_x).max(max_y - min_y);
    let scale = if span < 1e-9 { 1.0 } else { target_span / span };

    for (i, &u) in nodes.iter().enumerate() {
        coords[u].x = (coords[u].x - cx) * scale;
        coords[u].y = (coords[u].y - cy) * scale;
        if span < 1e-9 {
            coords[u].x += i as f64;
        }
    }
}

fn pack_components(components: &[Vec<usize>], coords: &mut [Point]) {
    if components.is_empty() {
        return;
    }

    let mut info = Vec::with_capacity(components.len());
    let mut total_area = 0.0;
    for (idx, comp) in components.iter().enumerate() {
        let (min_x, min_y, max_x, max_y) = bbox(comp, coords);
        let w = (max_x - min_x).max(1.0);
        let h = (max_y - min_y).max(1.0);
        total_area += w * h;
        info.push((idx, min_x, min_y, max_x, max_y, w, h));
    }

    info.sort_by(|a, b| {
        let area_a = a.5 * a.6;
        let area_b = b.5 * b.6;
        area_b.partial_cmp(&area_a).unwrap_or(Ordering::Equal)
    });

    let target_width = total_area.sqrt().max(20.0) * 1.35;
    let gap = 8.0;
    let mut cur_x = 0.0;
    let mut cur_y = 0.0;
    let mut row_h = 0.0;
    let mut offsets = vec![(0.0, 0.0); components.len()];

    for (idx, min_x, min_y, _max_x, _max_y, w, h) in info {
        if cur_x > 0.0 && cur_x + w + gap > target_width {
            cur_x = 0.0;
            cur_y += row_h + gap;
            row_h = 0.0;
        }
        offsets[idx] = (cur_x - min_x, cur_y - min_y);
        cur_x += w + gap;
        row_h = row_h.max(h);
    }

    for (idx, comp) in components.iter().enumerate() {
        let (ox, oy) = offsets[idx];
        for &u in comp {
            coords[u].x += ox;
            coords[u].y += oy;
        }
    }
}

fn bbox(nodes: &[usize], coords: &[Point]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &u in nodes {
        let p = coords[u];
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    (min_x, min_y, max_x, max_y)
}

fn bfs_dist(start: usize, adj: &[Vec<usize>]) -> Vec<i32> {
    let n = adj.len();
    let mut dist = vec![-1; n];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        let du = dist[u];
        for &v in &adj[u] {
            if dist[v] == -1 {
                dist[v] = du + 1;
                q.push_back(v);
            }
        }
    }
    dist
}

fn farthest_node(start: usize, adj: &[Vec<usize>]) -> (usize, i32) {
    let dist = bfs_dist(start, adj);
    let mut best = start;
    let mut best_d = dist[start];
    for (u, &d) in dist.iter().enumerate() {
        if d > best_d {
            best_d = d;
            best = u;
        }
    }
    (best, best_d)
}

fn farthest_from_dist(nodes: &[usize], dist: &[i32]) -> usize {
    let mut best = nodes[0];
    let mut best_d = dist[best];
    for &u in nodes {
        if dist[u] > best_d {
            best_d = dist[u];
            best = u;
        }
    }
    best
}
