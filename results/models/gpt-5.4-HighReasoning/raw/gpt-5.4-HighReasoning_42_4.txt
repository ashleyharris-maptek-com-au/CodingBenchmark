OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_0_04 = OpConstant %float 0.0399999991
%float_0_005 = OpConstant %float 0.00499999989
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%_ptr_Input_arr_gl_PerVertex_in_2 = OpTypePointer Input %arr_gl_PerVertex_in_2
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_float = OpTypePointer Input %float
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%_ptr_Input_arr_v3float_2 = OpTypePointer Input %arr_v3float_2
%_ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_2 Input
%worldPos = OpVariable %_ptr_Input_arr_v3float_2 Input
%normal = OpVariable %_ptr_Input_arr_v3float_2 Input
%color = OpVariable %_ptr_Input_arr_v3float_2 Input
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output
%main = OpFunction %void None %3
%5 = OpLabel
%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p0_w = OpCompositeExtract %float %p0 3
%p1_w = OpCompositeExtract %float %p1 3
%p0_ww = OpCompositeConstruct %v2float %p0_w %p0_w
%p1_ww = OpCompositeConstruct %v2float %p1_w %p1_w
%ndc0 = OpFDiv %v2float %p0_xy %p0_ww
%ndc1 = OpFDiv %v2float %p1_xy %p1_ww
%dir = OpFSub %v2float %ndc1 %ndc0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %1 Normalize %perp_raw
%thick0 = OpFMul %float %float_0_04 %p0_w
%thick1 = OpFMul %float %float_0_005 %p1_w
%thick0_v = OpCompositeConstruct %v2float %thick0 %thick0
%thick1_v = OpCompositeConstruct %v2float %thick1 %thick1
%off0 = OpFMul %v2float %perp %thick0_v
%off1 = OpFMul %v2float %perp %thick1_v
%off0_x = OpCompositeExtract %float %off0 0
%off0_y = OpCompositeExtract %float %off0 1
%off1_x = OpCompositeExtract %float %off1 0
%off1_y = OpCompositeExtract %float %off1 1
%off0_4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0
%off1_4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0
%v0 = OpFAdd %v4float %p0 %off0_4
%v1 = OpFSub %v4float %p0 %off0_4
%v2 = OpFAdd %v4float %p1 %off1_4
%v3 = OpFSub %v4float %p1 %off1_4
%gl_out_pos = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos %v0
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out_pos %v1
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out_pos %v2
OpStore %outColor %c1
OpEmitVertex
OpStore %gl_out_pos %v3
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
