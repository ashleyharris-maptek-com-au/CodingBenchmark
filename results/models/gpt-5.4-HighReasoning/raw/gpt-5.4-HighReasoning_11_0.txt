use std::cmp::Ordering;
use std::collections::{HashMap, HashSet};
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn add(self, other: Pt) -> Pt {
        Pt::new(self.x + other.x, self.y + other.y)
    }

    fn sub(self, other: Pt) -> Pt {
        Pt::new(self.x - other.x, self.y - other.y)
    }

    fn scale(self, k: f64) -> Pt {
        Pt::new(self.x * k, self.y * k)
    }

    fn dot(self, other: Pt) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Pt) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn rotate(self, ang: f64) -> Pt {
        let (s, c) = ang.sin_cos();
        Pt::new(self.x * c - self.y * s, self.x * s + self.y * c)
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

impl BBox {
    fn empty() -> Self {
        Self {
            min_x: f64::INFINITY,
            min_y: f64::INFINITY,
            max_x: f64::NEG_INFINITY,
            max_y: f64::NEG_INFINITY,
        }
    }

    fn from_points(points: &[Pt]) -> Self {
        let mut b = Self::empty();
        for &p in points {
            b.min_x = b.min_x.min(p.x);
            b.min_y = b.min_y.min(p.y);
            b.max_x = b.max_x.max(p.x);
            b.max_y = b.max_y.max(p.y);
        }
        b
    }

    fn width(&self) -> f64 {
        self.max_x - self.min_x
    }

    fn height(&self) -> f64 {
        self.max_y - self.min_y
    }

    fn intersects(&self, other: &Self) -> bool {
        self.min_x < other.max_x - EPS
            && self.max_x > other.min_x + EPS
            && self.min_y < other.max_y - EPS
            && self.max_y > other.min_y + EPS
    }

    fn inside(&self, outer: &Self) -> bool {
        self.min_x >= outer.min_x - EPS
            && self.max_x <= outer.max_x + EPS
            && self.min_y >= outer.min_y - EPS
            && self.max_y <= outer.max_y + EPS
    }
}

#[derive(Clone)]
struct Poly {
    verts: Vec<Pt>,
    area: f64,
    centroid: Pt,
    bbox: BBox,
}

impl Poly {
    fn new(mut verts: Vec<Pt>) -> Self {
        if signed_area(&verts) < 0.0 {
            verts.reverse();
        }
        let area2 = signed_area(&verts);
        let area = area2.abs() * 0.5;
        let centroid = polygon_centroid(&verts, area2);
        let bbox = BBox::from_points(&verts);
        Self {
            verts,
            area,
            centroid,
            bbox,
        }
    }
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    poly: Poly,
    centered: Vec<Pt>,
    radius: f64,
    complexity: usize,
}

#[derive(Clone)]
struct RotGeom {
    angle: f64,
    verts: Vec<Pt>,
    bbox: BBox,
    features: Vec<Pt>,
}

#[derive(Clone)]
struct PlacedPoly {
    bbox: BBox,
    verts: Vec<Pt>,
}

struct StockState {
    placed: Vec<PlacedPoly>,
    anchors: Vec<Pt>,
    grid: HashMap<(i32, i32), Vec<usize>>,
    cell: f64,
}

#[derive(Clone, Copy)]
struct Placement {
    stock_idx: usize,
    tx: f64,
    ty: f64,
    angle: f64,
}

#[derive(Clone, Copy)]
struct Params {
    rot_count: usize,
    max_anchors: usize,
    empty_scan: usize,
    feature_count: usize,
    compact: bool,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = next_parse(&mut it);
    let np: usize = next_parse(&mut it);

    let mut stock_verts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = next_parse(&mut it);
        let y: f64 = next_parse(&mut it);
        stock_verts.push(Pt::new(x, y));
    }
    let stock = Poly::new(stock_verts);

    let mut pieces = Vec::with_capacity(np);
    let mut total_vertices = sv;
    for idx in 0..np {
        let pv: usize = next_parse(&mut it);
        total_vertices += pv;
        let mut verts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = next_parse(&mut it);
            let y: f64 = next_parse(&mut it);
            verts.push(Pt::new(x, y));
        }
        let poly = Poly::new(verts);
        let centered: Vec<Pt> = poly.verts.iter().map(|&p| p.sub(poly.centroid)).collect();
        let mut radius: f64 = 0.0;
        for &p in &centered {
            radius = radius.max(p.dot(p).sqrt());
        }
        pieces.push(Piece {
            idx,
            radius,
            complexity: pv,
            centered,
            poly,
        });
    }

    let params = choose_params(np, total_vertices);
    let stock_bbox = stock.bbox;
    let cell = ((stock_bbox.width().max(stock_bbox.height())) / 12.0).max(1.0);

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let pa = &pieces[a];
        let pb = &pieces[b];
        pb.poly
            .area
            .partial_cmp(&pa.poly.area)
            .unwrap_or(Ordering::Equal)
            .then(pb.radius.partial_cmp(&pa.radius).unwrap_or(Ordering::Equal))
            .then(pb.complexity.cmp(&pa.complexity))
    });

    let mut placements: Vec<Option<Placement>> = vec![None; np];
    let mut stocks: Vec<StockState> = Vec::new();

    for &piece_idx in &order {
        let piece = &pieces[piece_idx];
        let rots = build_rotations(piece, &stock, params.rot_count, params.feature_count);
        let mut best: Option<(Placement, Vec<Pt>, BBox)> = None;

        for stock_idx in 0..=stocks.len() {
            let candidate = if stock_idx == stocks.len() {
                try_new_stock(piece, &rots, &stock, &stocks, stock_idx, params)
            } else {
                try_existing_stock(piece, &rots, &stock, &stocks[stock_idx], stock_idx, params)
            };

            if let Some(c) = candidate {
                let score = placement_score(c.0, c.2);
                let better = match best {
                    None => true,
                    Some((bp, _, bb)) => score < placement_score(bp, bb),
                };
                if better {
                    best = Some(c);
                    if stock_idx < stocks.len() {
                        break;
                    }
                }
            }
        }

        let (placement, verts, bbox) = best.expect("Failed to place piece");
        if placement.stock_idx == stocks.len() {
            stocks.push(new_stock_state(&stock, cell));
        }
        commit_placement(&mut stocks[placement.stock_idx], verts, bbox);
        placements[piece.idx] = Some(placement);
    }

    println!("{}", stocks.len());
    for p in placements {
        let p = p.unwrap();
        println!("{} {:.9} {:.9} {:.9}", p.stock_idx, p.tx, p.ty, p.angle);
    }
}

fn next_parse<T: std::str::FromStr>(it: &mut std::str::SplitWhitespace<'_>) -> T {
    it.next().unwrap().parse().ok().unwrap()
}

fn choose_params(np: usize, total_vertices: usize) -> Params {
    if np <= 80 && total_vertices <= 8000 {
        Params {
            rot_count: 28,
            max_anchors: 220,
            empty_scan: 18,
            feature_count: 8,
            compact: true,
        }
    } else if np <= 800 && total_vertices <= 60000 {
        Params {
            rot_count: 16,
            max_anchors: 96,
            empty_scan: 12,
            feature_count: 6,
            compact: true,
        }
    } else {
        Params {
            rot_count: 8,
            max_anchors: 28,
            empty_scan: 8,
            feature_count: 4,
            compact: false,
        }
    }
}

fn build_rotations(piece: &Piece, stock: &Poly, rot_count: usize, feature_count: usize) -> Vec<RotGeom> {
    let mut angles = Vec::new();
    angles.push(0.0);

    let mut stock_dirs = Vec::new();
    for i in 0..stock.verts.len() {
        let a = stock.verts[i];
        let b = stock.verts[(i + 1) % stock.verts.len()];
        let d = b.sub(a);
        let len = d.dot(d);
        if len > EPS {
            stock_dirs.push(d.y.atan2(d.x));
        }
    }

    let mut piece_dirs = Vec::new();
    for i in 0..piece.centered.len() {
        let a = piece.centered[i];
        let b = piece.centered[(i + 1) % piece.centered.len()];
        let d = b.sub(a);
        let len = d.dot(d);
        if len > EPS {
            piece_dirs.push(d.y.atan2(d.x));
        }
    }

    for &sa in stock_dirs.iter().take(8) {
        for &pa in piece_dirs.iter().take(8) {
            angles.push(normalize_angle(sa - pa));
        }
    }

    for i in 0..rot_count {
        angles.push((2.0 * PI * i as f64) / rot_count as f64);
    }

    let mut seen = HashSet::new();
    let mut out = Vec::new();
    for ang in angles {
        let key = (normalize_angle(ang) * 1_000_000.0).round() as i64;
        if !seen.insert(key) {
            continue;
        }
        let verts: Vec<Pt> = piece.centered.iter().map(|&p| p.rotate(ang)).collect();
        let bbox = BBox::from_points(&verts);
        let features = select_features(&verts, bbox, feature_count);
        out.push(RotGeom {
            angle: normalize_angle(ang),
            verts,
            bbox,
            features,
        });
    }
    out.sort_by(|a, b| {
        let aa = a.bbox.width() * a.bbox.height();
        let bb = b.bbox.width() * b.bbox.height();
        aa.partial_cmp(&bb).unwrap_or(Ordering::Equal)
    });
    out
}

fn select_features(verts: &[Pt], bbox: BBox, count: usize) -> Vec<Pt> {
    let mut feats = vec![
        Pt::new(bbox.min_x, bbox.min_y),
        Pt::new(bbox.min_x, bbox.max_y),
        Pt::new(bbox.max_x, bbox.min_y),
        Pt::new(bbox.max_x, bbox.max_y),
    ];

    let mut min_sum = verts[0];
    let mut max_sum = verts[0];
    let mut min_x = verts[0];
    let mut min_y = verts[0];
    for &p in verts {
        if p.x + p.y < min_sum.x + min_sum.y {
            min_sum = p;
        }
        if p.x + p.y > max_sum.x + max_sum.y {
            max_sum = p;
        }
        if p.x < min_x.x {
            min_x = p;
        }
        if p.y < min_y.y {
            min_y = p;
        }
    }
    feats.push(min_sum);
    feats.push(max_sum);
    feats.push(min_x);
    feats.push(min_y);
    feats.truncate(count.max(1));
    dedup_points(&mut feats);
    feats
}

fn dedup_points(points: &mut Vec<Pt>) {
    let mut out: Vec<Pt> = Vec::with_capacity(points.len());
    'outer: for &p in points.iter() {
        for &q in &out {
            if (p.x - q.x).abs() < 1e-7 && (p.y - q.y).abs() < 1e-7 {
                continue 'outer;
            }
        }
        out.push(p);
    }
    *points = out;
}

fn try_existing_stock(
    _piece: &Piece,
    rots: &[RotGeom],
    stock: &Poly,
    state: &StockState,
    stock_idx: usize,
    params: Params,
) -> Option<(Placement, Vec<Pt>, BBox)> {
    let mut anchors = state.anchors.clone();
    anchors.sort_by(|a, b| a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal).then(a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal)));
    if anchors.len() > params.max_anchors {
        anchors.truncate(params.max_anchors);
    }

    for rot in rots {
        for &anchor in &anchors {
            for &feat in &rot.features {
                let trans = anchor.sub(feat);
                if let Some((verts, bbox)) = validate_candidate(stock, state, &rot.verts, trans) {
                    let (tx, ty, verts, bbox) = if params.compact {
                        compact_candidate(stock, state, &rot.verts, trans, verts, bbox)
                    } else {
                        (trans.x, trans.y, verts, bbox)
                    };
                    return Some((
                        Placement {
                            stock_idx,
                            tx,
                            ty,
                            angle: rot.angle,
                        },
                        verts,
                        bbox,
                    ));
                }
            }
        }
    }
    None
}

fn try_new_stock(
    _piece: &Piece,
    rots: &[RotGeom],
    stock: &Poly,
    existing: &[StockState],
    stock_idx: usize,
    params: Params,
) -> Option<(Placement, Vec<Pt>, BBox)> {
    let scratch = new_stock_state(stock, ((stock.bbox.width().max(stock.bbox.height())) / 12.0).max(1.0));
    if let Some(v) = try_existing_stock(_piece, rots, stock, &scratch, stock_idx, params) {
        return Some(v);
    }

    let probe = if existing.is_empty() { params.empty_scan.max(10) } else { params.empty_scan };
    for rot in rots {
        if let Some((verts, bbox, tx, ty)) = scan_empty_stock(stock, rot, probe) {
            return Some((
                Placement {
                    stock_idx,
                    tx,
                    ty,
                    angle: rot.angle,
                },
                verts,
                bbox,
            ));
        }
    }
    None
}

fn new_stock_state(stock: &Poly, cell: f64) -> StockState {
    let mut anchors = stock.verts.clone();
    anchors.push(stock.centroid);
    anchors.push(Pt::new(stock.bbox.min_x, stock.bbox.min_y));
    anchors.push(Pt::new(stock.bbox.min_x, stock.bbox.max_y));
    anchors.push(Pt::new(stock.bbox.max_x, stock.bbox.min_y));
    dedup_points(&mut anchors);
    StockState {
        placed: Vec::new(),
        anchors,
        grid: HashMap::new(),
        cell,
    }
}

fn scan_empty_stock(stock: &Poly, rot: &RotGeom, steps: usize) -> Option<(Vec<Pt>, BBox, f64, f64)> {
    let bbox = stock.bbox;
    let step_x = (bbox.width() / steps as f64).max(1.0);
    let step_y = (bbox.height() / steps as f64).max(1.0);
    let mut points = Vec::new();
    points.push(stock.centroid);
    for iy in 0..=steps {
        for ix in 0..=steps {
            let p = Pt::new(bbox.min_x + ix as f64 * step_x, bbox.min_y + iy as f64 * step_y);
            if point_in_poly_inclusive(p, &stock.verts) {
                points.push(p);
            }
        }
    }
    for anchor in points {
        for &feat in &rot.features {
            let trans = anchor.sub(feat);
            let verts: Vec<Pt> = rot.verts.iter().map(|&p| p.add(trans)).collect();
            let bb = BBox::from_points(&verts);
            if piece_inside_stock(&verts, bb, stock) {
                return Some((verts, bb, trans.x, trans.y));
            }
        }
    }
    None
}

fn validate_candidate(stock: &Poly, state: &StockState, base: &[Pt], trans: Pt) -> Option<(Vec<Pt>, BBox)> {
    let verts: Vec<Pt> = base.iter().map(|&p| p.add(trans)).collect();
    let bbox = BBox::from_points(&verts);
    if !piece_inside_stock(&verts, bbox, stock) {
        return None;
    }
    if overlaps_any(state, &verts, bbox) {
        return None;
    }
    Some((verts, bbox))
}

fn compact_candidate(
    stock: &Poly,
    state: &StockState,
    base: &[Pt],
    trans: Pt,
    mut best_verts: Vec<Pt>,
    mut best_bbox: BBox,
) -> (f64, f64, Vec<Pt>, BBox) {
    let mut best = trans;
    let mut step = stock.bbox.width().max(stock.bbox.height()) / 8.0;
    while step > 1e-3 {
        let probes = [
            Pt::new(best.x - step, best.y),
            Pt::new(best.x, best.y - step),
            Pt::new(best.x - step, best.y - step),
        ];
        let mut improved = false;
        for cand in probes {
            if let Some((verts, bbox)) = validate_candidate(stock, state, base, cand) {
                if bbox.min_y < best_bbox.min_y - EPS
                    || ((bbox.min_y - best_bbox.min_y).abs() < EPS && bbox.min_x < best_bbox.min_x - EPS)
                {
                    best = cand;
                    best_verts = verts;
                    best_bbox = bbox;
                    improved = true;
                }
            }
        }
        if !improved {
            step *= 0.5;
        }
    }
    (best.x, best.y, best_verts, best_bbox)
}

fn piece_inside_stock(verts: &[Pt], bbox: BBox, stock: &Poly) -> bool {
    if !bbox.inside(&stock.bbox) {
        return false;
    }
    for &p in verts {
        if !point_in_poly_inclusive(p, &stock.verts) {
            return false;
        }
    }
    for i in 0..verts.len() {
        let a = verts[i];
        let b = verts[(i + 1) % verts.len()];
        for j in 0..stock.verts.len() {
            let c = stock.verts[j];
            let d = stock.verts[(j + 1) % stock.verts.len()];
            if segments_properly_intersect(a, b, c, d) {
                return false;
            }
        }
    }
    true
}

fn overlaps_any(state: &StockState, verts: &[Pt], bbox: BBox) -> bool {
    let candidates = query_grid(state, bbox);
    for idx in candidates {
        let other = &state.placed[idx];
        if !bbox.intersects(&other.bbox) {
            continue;
        }
        if polys_overlap(verts, &other.verts) {
            return true;
        }
    }
    false
}

fn polys_overlap(a: &[Pt], b: &[Pt]) -> bool {
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if segments_properly_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    if point_in_poly_strict(a[0], b) {
        return true;
    }
    if point_in_poly_strict(b[0], a) {
        return true;
    }
    false
}

fn query_grid(state: &StockState, bbox: BBox) -> Vec<usize> {
    let mut found = HashSet::new();
    let x0 = (bbox.min_x / state.cell).floor() as i32;
    let x1 = (bbox.max_x / state.cell).floor() as i32;
    let y0 = (bbox.min_y / state.cell).floor() as i32;
    let y1 = (bbox.max_y / state.cell).floor() as i32;
    for gx in x0..=x1 {
        for gy in y0..=y1 {
            if let Some(v) = state.grid.get(&(gx, gy)) {
                for &idx in v {
                    found.insert(idx);
                }
            }
        }
    }
    found.into_iter().collect()
}

fn commit_placement(state: &mut StockState, verts: Vec<Pt>, bbox: BBox) {
    let idx = state.placed.len();
    state.placed.push(PlacedPoly {
        bbox,
        verts: verts.clone(),
    });
    index_bbox(&mut state.grid, state.cell, bbox, idx);

    state.anchors.push(Pt::new(bbox.max_x, bbox.min_y));
    state.anchors.push(Pt::new(bbox.min_x, bbox.max_y));
    state.anchors.push(Pt::new(bbox.max_x, bbox.max_y));
    for &p in verts.iter().take(8) {
        state.anchors.push(p);
    }
    dedup_points(&mut state.anchors);
}

fn index_bbox(grid: &mut HashMap<(i32, i32), Vec<usize>>, cell: f64, bbox: BBox, idx: usize) {
    let x0 = (bbox.min_x / cell).floor() as i32;
    let x1 = (bbox.max_x / cell).floor() as i32;
    let y0 = (bbox.min_y / cell).floor() as i32;
    let y1 = (bbox.max_y / cell).floor() as i32;
    for gx in x0..=x1 {
        for gy in y0..=y1 {
            grid.entry((gx, gy)).or_default().push(idx);
        }
    }
}

fn placement_score(p: Placement, bbox: BBox) -> (usize, i64, i64) {
    (
        p.stock_idx,
        (bbox.min_y * 1_000_000.0) as i64,
        (bbox.min_x * 1_000_000.0) as i64,
    )
}

fn normalize_angle(mut a: f64) -> f64 {
    while a < 0.0 {
        a += 2.0 * PI;
    }
    while a >= 2.0 * PI {
        a -= 2.0 * PI;
    }
    a
}

fn signed_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        s += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    s
}

fn polygon_centroid(poly: &[Pt], area2: f64) -> Pt {
    if area2.abs() < EPS {
        let mut acc = Pt::new(0.0, 0.0);
        for &p in poly {
            acc = acc.add(p);
        }
        return acc.scale(1.0 / poly.len() as f64);
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        let c = p.cross(q);
        cx += (p.x + q.x) * c;
        cy += (p.y + q.y) * c;
    }
    Pt::new(cx / (3.0 * area2), cy / (3.0 * area2))
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    orient(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn segments_properly_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if o1.abs() <= EPS && on_segment(a, b, c) {
        return false;
    }
    if o2.abs() <= EPS && on_segment(a, b, d) {
        return false;
    }
    if o3.abs() <= EPS && on_segment(c, d, a) {
        return false;
    }
    if o4.abs() <= EPS && on_segment(c, d, b) {
        return false;
    }
    (o1 > EPS && o2 < -EPS || o1 < -EPS && o2 > EPS)
        && (o3 > EPS && o4 < -EPS || o3 < -EPS && o4 > EPS)
}

fn point_in_poly_inclusive(p: Pt, poly: &[Pt]) -> bool {
    for i in 0..poly.len() {
        if on_segment(poly[i], poly[(i + 1) % poly.len()], p) {
            return true;
        }
    }
    point_in_poly_strict(p, poly)
}

fn point_in_poly_strict(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let x = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x > p.x + EPS {
                inside = !inside;
            }
        }
    }
    inside
}
