OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_h = OpConstant %float 0.0500000007
%float_nh = OpConstant %float -0.0500000007
%v4_top_off = OpConstantComposite %v4float %float_0 %float_h %float_0 %float_0
%v4_right_off = OpConstantComposite %v4float %float_h %float_0 %float_0 %float_0
%v4_left_off = OpConstantComposite %v4float %float_nh %float_0 %float_0 %float_0
%v4_bottom_off = OpConstantComposite %v4float %float_0 %float_nh %float_0 %float_0
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%_ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%_ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%_ptr_Input_v3float = OpTypePointer Input %v3float
%gl_in = OpVariable %_ptr_Input_arr_pv_in Input
%worldPos = OpVariable %_ptr_Input_arr_v3float_1 Input
%normal = OpVariable %_ptr_Input_arr_v3float_1 Input
%color = OpVariable %_ptr_Input_arr_v3float_1 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output
%main = OpFunction %void None %func
%entry = OpLabel
%pos_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%inColor = OpLoad %v3float %color_ptr
%gl_out_pos = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
%top_pos = OpFAdd %v4float %pos %v4_top_off
OpStore %gl_out_pos %top_pos
OpStore %outColor %inColor
OpEmitVertex
%right_pos = OpFAdd %v4float %pos %v4_right_off
OpStore %gl_out_pos %right_pos
OpStore %outColor %inColor
OpEmitVertex
%left_pos = OpFAdd %v4float %pos %v4_left_off
OpStore %gl_out_pos %left_pos
OpStore %outColor %inColor
OpEmitVertex
%bottom_pos = OpFAdd %v4float %pos %v4_bottom_off
OpStore %gl_out_pos %bottom_pos
OpStore %outColor %inColor
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
