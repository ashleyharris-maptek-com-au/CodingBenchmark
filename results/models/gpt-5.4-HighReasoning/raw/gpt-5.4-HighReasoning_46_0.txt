#version 450

layout(local_size_x = 256) in;

// Each body occupies two consecutive vec4 values:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = number of bodies
    vec4 physics;   // x = dt, y = softening, z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint n = counts.x;

    if (i >= n) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softening2 = softening * softening;

    uint baseIndex = i * 2u;
    vec4 posMass = inData[baseIndex];
    vec3 pos = posMass.xyz;
    float mass = posMass.w;
    vec3 vel = inData[baseIndex + 1u].xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < n; ++j) {
        if (j == i) {
            continue;
        }

        uint otherBase = j * 2u;
        vec4 otherPosMass = inData[otherBase];
        vec3 delta = otherPosMass.xyz - pos;
        float dist2 = dot(delta, delta) + softening2;
        float invDist = inversesqrt(dist2);
        float invDist3 = invDist * invDist * invDist;

        acc += G * otherPosMass.w * delta * invDist3;
    }

    vec3 newVel = vel + acc * dt;
    vec3 newPos = pos + newVel * dt;

    outData[baseIndex] = vec4(newPos, mass);
    outData[baseIndex + 1u] = vec4(newVel, 0.0);
}
