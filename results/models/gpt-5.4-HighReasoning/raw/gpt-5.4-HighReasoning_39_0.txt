use std::cmp::Reverse;
use std::io::{self, Read};

struct Solver {
    n: usize,
    half: usize,
    adj: Vec<Vec<usize>>,
    side: Vec<i8>, // -1 = unassigned, 0 = B, 1 = A
    count_to_a: Vec<usize>,
    count_to_b: Vec<usize>,
    unassigned_deg: Vec<usize>,
    assigned_a: usize,
    assigned_b: usize,
    current_cut: usize,
    best_cut: usize,
    best_side: Vec<i8>,
}

impl Solver {
    fn new(adj: Vec<Vec<usize>>) -> Self {
        let n = adj.len();
        let half = n / 2;
        let unassigned_deg = adj.iter().map(|nei| nei.len()).collect::<Vec<_>>();
        Self {
            n,
            half,
            adj,
            side: vec![-1; n],
            count_to_a: vec![0; n],
            count_to_b: vec![0; n],
            unassigned_deg,
            assigned_a: 0,
            assigned_b: 0,
            current_cut: 0,
            best_cut: usize::MAX,
            best_side: vec![0; n],
        }
    }

    fn total_cut_for_a_set(&self, in_a: &[bool]) -> usize {
        let mut cut = 0usize;
        for u in 0..self.n {
            for &v in &self.adj[u] {
                if u < v && in_a[u] != in_a[v] {
                    cut += 1;
                }
            }
        }
        cut
    }

    fn initial_upper_bound(&mut self) {
        let mut order: Vec<usize> = (0..self.n).collect();
        order.sort_by_key(|&u| Reverse(self.adj[u].len()));

        let mut in_a = vec![false; self.n];
        let mut in_b = vec![false; self.n];
        let mut a_vertices: Vec<usize> = Vec::with_capacity(self.half);
        let mut b_vertices: Vec<usize> = Vec::with_capacity(self.half);

        for &u in &order {
            if a_vertices.len() == self.half {
                in_b[u] = true;
                b_vertices.push(u);
                continue;
            }
            if b_vertices.len() == self.half {
                in_a[u] = true;
                a_vertices.push(u);
                continue;
            }

            let add_if_a = self.adj[u].iter().filter(|&&v| in_b[v]).count();
            let add_if_b = self.adj[u].iter().filter(|&&v| in_a[v]).count();
            if add_if_a <= add_if_b {
                in_a[u] = true;
                a_vertices.push(u);
            } else {
                in_b[u] = true;
                b_vertices.push(u);
            }
        }

        let mut improved = true;
        while improved {
            improved = false;
            let mut best_delta = 0isize;
            let mut best_pair = None;

            for &a in &a_vertices {
                for &b in &b_vertices {
                    let mut delta = 0isize;
                    for &x in &self.adj[a] {
                        if x != b {
                            delta += if in_a[x] { 1 } else { -1 };
                        }
                    }
                    for &x in &self.adj[b] {
                        if x != a {
                            delta += if in_b[x] { 1 } else { -1 };
                        }
                    }
                    if self.adj[a].contains(&b) {
                        delta += 0;
                    }
                    if delta < best_delta {
                        best_delta = delta;
                        best_pair = Some((a, b));
                    }
                }
            }

            if let Some((a, b)) = best_pair {
                improved = true;
                in_a[a] = false;
                in_b[a] = true;
                in_b[b] = false;
                in_a[b] = true;
                for x in &mut a_vertices {
                    if *x == a {
                        *x = b;
                        break;
                    }
                }
                for x in &mut b_vertices {
                    if *x == b {
                        *x = a;
                        break;
                    }
                }
            }
        }

        self.best_cut = self.total_cut_for_a_set(&in_a);
        for (i, &flag) in in_a.iter().enumerate() {
            self.best_side[i] = if flag { 1 } else { 0 };
        }
    }

    fn choose_vertex(&self) -> Option<usize> {
        let mut best = None;
        let mut best_key = (0usize, 0usize);
        for u in 0..self.n {
            if self.side[u] != -1 {
                continue;
            }
            let key = (
                self.count_to_a[u] + self.count_to_b[u],
                self.adj[u].len(),
            );
            if key > best_key {
                best_key = key;
                best = Some(u);
            }
        }
        best
    }

    fn lower_bound(&self) -> usize {
        let rem_a = self.half - self.assigned_a;
        let rem_b = self.half - self.assigned_b;
        let remaining = rem_a + rem_b;
        if remaining == 0 {
            return self.current_cut;
        }

        let mut deltas_assigned = Vec::with_capacity(remaining);
        let mut base_assigned = 0usize;

        let mut deltas_unassigned = Vec::with_capacity(remaining);
        let mut base_unassigned = 0usize;

        for u in 0..self.n {
            if self.side[u] != -1 {
                continue;
            }

            let cost_b = self.count_to_a[u];
            let cost_a = self.count_to_b[u];
            base_assigned += cost_b;
            deltas_assigned.push(cost_a as isize - cost_b as isize);

            let d = self.unassigned_deg[u] as isize;
            let cost_ua = if rem_a == 0 {
                isize::MAX / 4
            } else {
                let keep = (rem_a - 1) as isize;
                (d - keep).max(0)
            };
            let cost_ub = if rem_b == 0 {
                isize::MAX / 4
            } else {
                let keep = (rem_b - 1) as isize;
                (d - keep).max(0)
            };
            base_unassigned += cost_ub as usize;
            deltas_unassigned.push(cost_ua - cost_ub);
        }

        deltas_assigned.sort_unstable();
        let mut lb_assigned = base_assigned;
        for &delta in deltas_assigned.iter().take(rem_a) {
            lb_assigned = (lb_assigned as isize + delta) as usize;
        }

        deltas_unassigned.sort_unstable();
        let mut endpoint_lb = base_unassigned as isize;
        for &delta in deltas_unassigned.iter().take(rem_a) {
            endpoint_lb += delta;
        }
        let lb_unassigned = ((endpoint_lb as usize) + 1) / 2;

        self.current_cut + lb_assigned + lb_unassigned
    }

    fn apply_assign(&mut self, v: usize, to_a: bool) -> (usize, Vec<usize>) {
        let prev_cut = self.current_cut;
        self.side[v] = if to_a { 1 } else { 0 };
        if to_a {
            self.assigned_a += 1;
            self.current_cut += self.count_to_b[v];
        } else {
            self.assigned_b += 1;
            self.current_cut += self.count_to_a[v];
        }

        let mut changed = Vec::with_capacity(self.adj[v].len());
        for &u in &self.adj[v] {
            if self.side[u] == -1 {
                self.unassigned_deg[u] -= 1;
                if to_a {
                    self.count_to_a[u] += 1;
                } else {
                    self.count_to_b[u] += 1;
                }
                changed.push(u);
            }
        }
        (prev_cut, changed)
    }

    fn undo_assign(
        &mut self,
        v: usize,
        to_a: bool,
        prev_cut: usize,
        changed: Vec<usize>,
    ) {
        for u in changed {
            self.unassigned_deg[u] += 1;
            if to_a {
                self.count_to_a[u] -= 1;
            } else {
                self.count_to_b[u] -= 1;
            }
        }

        if to_a {
            self.assigned_a -= 1;
        } else {
            self.assigned_b -= 1;
        }
        self.side[v] = -1;
        self.current_cut = prev_cut;
    }

    fn dfs(&mut self) {
        if self.assigned_a > self.half || self.assigned_b > self.half {
            return;
        }

        let lb = self.lower_bound();
        if lb >= self.best_cut {
            return;
        }

        if self.assigned_a == self.half && self.assigned_b == self.half {
            self.best_cut = self.current_cut;
            self.best_side.copy_from_slice(&self.side);
            return;
        }

        let v = match self.choose_vertex() {
            Some(v) => v,
            None => return,
        };

        let rem_a = self.half - self.assigned_a;
        let rem_b = self.half - self.assigned_b;

        let mut branches = Vec::with_capacity(2);
        if rem_a > 0 {
            branches.push((true, self.count_to_b[v]));
        }
        if rem_b > 0 {
            branches.push((false, self.count_to_a[v]));
        }
        branches.sort_by_key(|&(_, added_cut)| added_cut);

        for (to_a, _) in branches {
            let (prev_cut, changed) = self.apply_assign(v, to_a);
            self.dfs();
            self.undo_assign(v, to_a, prev_cut, changed);
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for nei in &mut adj {
        nei.sort_unstable();
        nei.dedup();
    }

    let mut solver = Solver::new(adj);
    solver.initial_upper_bound();

    // Break symmetry: vertex 0 can be fixed in A.
    let (prev_cut, changed) = solver.apply_assign(0, true);
    solver.dfs();
    solver.undo_assign(0, true, prev_cut, changed);

    let mut part_a = Vec::with_capacity(solver.half);
    for i in 0..solver.n {
        if solver.best_side[i] == 1 {
            part_a.push(i);
        }
    }
    part_a.sort_unstable();

    println!("{}", solver.best_cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
