use std::cmp::Reverse;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
}

struct Solver {
    w: usize,
    h: usize,
    require_cycle: bool,
    blocked: Vec<bool>,
    coords: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    start: usize,
    n: usize,
    deadline: Instant,
    timed_out: bool,
}

impl Solver {
    fn new(
        w: usize,
        h: usize,
        blocked_cells: &[(usize, usize)],
        require_cycle: bool,
        time_budget: Duration,
    ) -> Option<Self> {
        let mut blocked = vec![false; w * h];
        for &(x, y) in blocked_cells {
            if x >= w || y >= h {
                return None;
            }
            blocked[y * w + x] = true;
        }
        if w == 0 || h == 0 || blocked[0] {
            return None;
        }

        let mut vertices = Vec::new();
        let mut idx_of_cell = vec![None; w * h];
        let mut coords = Vec::new();
        for y in 0..h {
            for x in 0..w {
                let cell = y * w + x;
                if !blocked[cell] {
                    let idx = vertices.len();
                    vertices.push(cell);
                    idx_of_cell[cell] = Some(idx);
                    coords.push((x, y));
                }
            }
        }
        let n = vertices.len();
        if n == 0 {
            return None;
        }
        let start = idx_of_cell[0]?;
        let mut adj = vec![Vec::new(); n];
        for i in 0..n {
            let cell = vertices[i];
            let x = cell % w;
            let y = cell / w;
            if x > 0 {
                let nc = cell - 1;
                if let Some(j) = idx_of_cell[nc] {
                    adj[i].push(j);
                }
            }
            if x + 1 < w {
                let nc = cell + 1;
                if let Some(j) = idx_of_cell[nc] {
                    adj[i].push(j);
                }
            }
            if y > 0 {
                let nc = cell - w;
                if let Some(j) = idx_of_cell[nc] {
                    adj[i].push(j);
                }
            }
            if y + 1 < h {
                let nc = cell + w;
                if let Some(j) = idx_of_cell[nc] {
                    adj[i].push(j);
                }
            }
        }

        let start_time = Instant::now();
        Some(Self {
            w,
            h,
            require_cycle,
            blocked,
            coords,
            adj,
            start,
            n,
            deadline: start_time + time_budget,
            timed_out: false,
        })
    }

    fn solve(&mut self) -> Option<Vec<(usize, usize)>> {
        if let Some(path) = self.solve_open_rectangle() {
            return Some(path);
        }
        if !self.basic_feasible() {
            return None;
        }

        if self.n <= 84 {
            let mut visited = vec![false; self.n];
            let mut path = Vec::with_capacity(self.n);
            visited[self.start] = true;
            path.push(self.start);
            if self.dfs_exact(self.start, &mut visited, &mut path) {
                return Some(self.materialize(&path));
            }
            return None;
        }

        if let Some(path) = self.greedy_large() {
            return Some(path);
        }

        None
    }

    fn solve_open_rectangle(&self) -> Option<Vec<(usize, usize)>> {
        if self.blocked.iter().any(|&b| b) {
            return None;
        }
        if self.require_cycle {
            self.construct_open_cycle()
        } else {
            Some(self.construct_open_path())
        }
    }

    fn construct_open_path(&self) -> Vec<(usize, usize)> {
        let mut out = Vec::with_capacity(self.w * self.h);
        if self.h == 1 {
            for x in 0..self.w {
                out.push((x, 0));
            }
            return out;
        }
        if self.w == 1 {
            for y in 0..self.h {
                out.push((0, y));
            }
            return out;
        }
        if self.h % 2 == 0 || self.w % 2 == 1 {
            for y in 0..self.h {
                if y % 2 == 0 {
                    for x in 0..self.w {
                        out.push((x, y));
                    }
                } else {
                    for x in (0..self.w).rev() {
                        out.push((x, y));
                    }
                }
            }
        } else {
            for x in 0..self.w {
                if x % 2 == 0 {
                    for y in 0..self.h {
                        out.push((x, y));
                    }
                } else {
                    for y in (0..self.h).rev() {
                        out.push((x, y));
                    }
                }
            }
        }
        out
    }

    fn construct_open_cycle(&self) -> Option<Vec<(usize, usize)>> {
        if self.w == 1 || self.h == 1 {
            return None;
        }
        if self.w % 2 == 1 && self.h % 2 == 1 {
            return None;
        }
        let mut out = Vec::with_capacity(self.w * self.h);
        if self.h % 2 == 0 {
            for x in 0..self.w {
                out.push((x, 0));
            }
            for y in 1..self.h {
                if y % 2 == 1 {
                    for x in (1..self.w).rev() {
                        out.push((x, y));
                    }
                } else {
                    for x in 1..self.w {
                        out.push((x, y));
                    }
                }
            }
            for y in (1..self.h).rev() {
                out.push((0, y));
            }
        } else {
            for y in 0..self.h {
                out.push((0, y));
            }
            for x in 1..self.w {
                if x % 2 == 1 {
                    for y in (1..self.h).rev() {
                        out.push((x, y));
                    }
                } else {
                    for y in 1..self.h {
                        out.push((x, y));
                    }
                }
            }
            for x in (1..self.w).rev() {
                out.push((x, 0));
            }
        }
        Some(out)
    }

    fn basic_feasible(&self) -> bool {
        if self.n == 1 {
            return !self.require_cycle;
        }
        if self.require_cycle && self.n % 2 == 1 {
            return false;
        }

        let mut q = VecDeque::new();
        let mut seen = vec![false; self.n];
        q.push_back(self.start);
        seen[self.start] = true;
        let mut reached = 0usize;
        while let Some(u) = q.pop_front() {
            reached += 1;
            for &v in &self.adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
        if reached != self.n {
            return false;
        }

        let start_color = (self.coords[self.start].0 + self.coords[self.start].1) & 1;
        let mut part = [0usize; 2];
        let mut deg1 = 0usize;
        for i in 0..self.n {
            let (x, y) = self.coords[i];
            part[(x + y) & 1] += 1;
            let d = self.adj[i].len();
            if self.require_cycle {
                if d < 2 {
                    return false;
                }
            } else {
                if d == 0 {
                    return false;
                }
                if d == 1 {
                    deg1 += 1;
                }
            }
        }
        if self.require_cycle {
            if part[0] != part[1] {
                return false;
            }
        } else {
            let diff = part[0].abs_diff(part[1]);
            if diff > 1 {
                return false;
            }
            if self.n % 2 == 1 && part[start_color] != part[1 - start_color] + 1 {
                return false;
            }
            if self.n % 2 == 0 && part[0] != part[1] {
                return false;
            }
            if deg1 > 2 {
                return false;
            }
            if deg1 == 2 && self.adj[self.start].len() != 1 {
                return false;
            }
        }
        true
    }

    fn dfs_exact(&mut self, current: usize, visited: &mut [bool], path: &mut Vec<usize>) -> bool {
        if Instant::now() >= self.deadline {
            self.timed_out = true;
            return false;
        }

        if path.len() == self.n {
            return !self.require_cycle || self.adj[current].contains(&self.start);
        }

        let mut moves = Vec::new();
        for &nxt in &self.adj[current] {
            if visited[nxt] {
                continue;
            }
            visited[nxt] = true;
            if self.residual_feasible(nxt, visited, path.len() + 1, true) {
                let onward = self
                    .adj[nxt]
                    .iter()
                    .filter(|&&u| !visited[u])
                    .count();
                let degree = self.adj[nxt].len();
                moves.push((onward, degree, nxt));
            }
            visited[nxt] = false;
        }

        moves.sort_unstable_by_key(|&(onward, degree, nxt)| (onward, degree, nxt));

        for (_, _, nxt) in moves {
            visited[nxt] = true;
            path.push(nxt);
            if self.dfs_exact(nxt, visited, path) {
                return true;
            }
            path.pop();
            visited[nxt] = false;
            if self.timed_out {
                return false;
            }
        }
        false
    }

    fn residual_feasible(
        &self,
        head: usize,
        visited: &[bool],
        path_len: usize,
        strong_checks: bool,
    ) -> bool {
        let remaining = self.n - path_len;
        if remaining == 0 {
            return !self.require_cycle || self.adj[head].contains(&self.start);
        }

        let mut q = VecDeque::new();
        let mut seen = vec![false; self.n];
        q.push_back(head);
        seen[head] = true;
        let mut reached = 1usize;
        while let Some(u) = q.pop_front() {
            for &v in &self.adj[u] {
                if seen[v] || visited[v] {
                    continue;
                }
                seen[v] = true;
                reached += 1;
                q.push_back(v);
            }
        }
        if reached != remaining + 1 {
            return false;
        }

        let mut leaves = 0usize;
        for v in 0..self.n {
            if visited[v] {
                continue;
            }
            let mut deg = 0usize;
            for &u in &self.adj[v] {
                if !visited[u] || u == head {
                    deg += 1;
                }
            }
            if deg == 0 {
                return false;
            }
            if deg == 1 {
                leaves += 1;
                if leaves > 1 {
                    return false;
                }
            }
        }

        if strong_checks {
            let mut degree1_or_0 = 0usize;
            for v in 0..self.n {
                if visited[v] {
                    continue;
                }
                let deg = self.adj[v]
                    .iter()
                    .filter(|&&u| !visited[u] || u == head)
                    .count();
                if deg <= 1 {
                    degree1_or_0 += 1;
                    if degree1_or_0 > 2 {
                        return false;
                    }
                }
            }
        }

        true
    }

    fn greedy_large(&mut self) -> Option<Vec<(usize, usize)>> {
        let seed = (self.n as u64)
            ^ ((self.w as u64) << 17)
            ^ ((self.h as u64) << 33)
            ^ if self.require_cycle { 0x9E37_79B9_7F4A_7C15 } else { 0 };
        let mut rng = XorShift64::new(seed);
        let max_attempts = 96usize;

        for _ in 0..max_attempts {
            if Instant::now() >= self.deadline {
                self.timed_out = true;
                return None;
            }

            let mut visited = vec![false; self.n];
            let mut path = Vec::with_capacity(self.n);
            let mut current = self.start;
            visited[current] = true;
            path.push(current);

            while path.len() < self.n {
                let step = path.len();
                let mut candidates = Vec::new();
                for &nxt in &self.adj[current] {
                    if visited[nxt] {
                        continue;
                    }
                    visited[nxt] = true;
                    let strong = self.n <= 300 || step % 11 == 0;
                    let ok = self.residual_feasible(nxt, &visited, step + 1, strong);
                    visited[nxt] = false;
                    if !ok {
                        continue;
                    }
                    let onward = self
                        .adj[nxt]
                        .iter()
                        .filter(|&&u| !visited[u])
                        .count();
                    let bias = (rng.next_u64() & 15) as usize;
                    candidates.push((onward, self.adj[nxt].len(), bias, nxt));
                }

                if candidates.is_empty() {
                    break;
                }

                candidates.sort_unstable_by_key(|&(onward, degree, bias, nxt)| {
                    (onward, degree, bias, Reverse(nxt))
                });

                let nxt = candidates[0].3;
                visited[nxt] = true;
                path.push(nxt);
                current = nxt;
            }

            if path.len() == self.n && (!self.require_cycle || self.adj[current].contains(&self.start))
            {
                return Some(self.materialize(&path));
            }
        }
        None
    }

    fn materialize(&self, path: &[usize]) -> Vec<(usize, usize)> {
        path.iter().map(|&i| self.coords[i]).collect()
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("0");
            return;
        }
    };
    let h: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("0");
            return;
        }
    };
    let obstacle_count: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle = match it.next() {
        Some(v) => v != "0",
        None => {
            println!("0");
            return;
        }
    };

    let mut blocked = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let x: usize = match it.next() {
            Some(v) => v.parse().unwrap(),
            None => {
                println!("0");
                return;
            }
        };
        let y: usize = match it.next() {
            Some(v) => v.parse().unwrap(),
            None => {
                println!("0");
                return;
            }
        };
        blocked.push((x, y));
    }

    let time_budget = Duration::from_secs(295);
    let mut solver = match Solver::new(w, h, &blocked, require_cycle, time_budget) {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };

    match solver.solve() {
        Some(path) => {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
        }
        None => {
            println!("0");
        }
    }
}
