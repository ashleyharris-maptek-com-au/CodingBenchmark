declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Package = {
  id: string | number;
  x: number;
  y: number;
};

type InputData = {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
};

const parsed = data as InputData;
const depotX = Number(parsed.depot[0] ?? 0);
const depotY = Number(parsed.depot[1] ?? 0);
const packages: Package[] = Array.isArray(parsed.packages) ? parsed.packages : [];
const n = packages.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const xs = new Array<number>(n);
const ys = new Array<number>(n);
const ids = new Array<string | number>(n);
const radii = new Array<number>(n);
const angles = new Array<number>(n);

let minX = Infinity;
let maxX = -Infinity;
let minY = Infinity;
let maxY = -Infinity;

for (let i = 0; i < n; i++) {
  const p = packages[i];
  const x = Number(p.x);
  const y = Number(p.y);
  xs[i] = x;
  ys[i] = y;
  ids[i] = p.id;
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
  const dx = x - depotX;
  const dy = y - depotY;
  radii[i] = Math.hypot(dx, dy);
  angles[i] = Math.atan2(dy, dx);
}

function distIdx(a: number, b: number): number {
  return Math.hypot(xs[a] - xs[b], ys[a] - ys[b]);
}

function distDepot(a: number): number {
  return Math.hypot(xs[a] - depotX, ys[a] - depotY);
}

function routeLength(order: number[]): number {
  const m = order.length;
  if (m === 0) return 0;
  let total = distDepot(order[0]) + distDepot(order[m - 1]);
  for (let i = 1; i < m; i++) {
    total += distIdx(order[i - 1], order[i]);
  }
  return total;
}

function reverseRange(arr: number[], left: number, right: number): void {
  while (left < right) {
    const tmp = arr[left];
    arr[left] = arr[right];
    arr[right] = tmp;
    left++;
    right--;
  }
}

function twoOpt(order: number[], passes: number, window: number): number[] {
  const m = order.length;
  if (m < 3) return order.slice();

  const arr = order.slice();
  const eps = 1e-9;

  for (let pass = 0; pass < passes; pass++) {
    let improved = false;

    for (let i = 0; i < m - 1; i++) {
      const prevIsDepot = i === 0;
      const a = arr[i];
      const prev = prevIsDepot ? -1 : arr[i - 1];

      const maxJ = Math.min(m - 1, i + window);
      for (let j = i + 1; j <= maxJ; j++) {
        if (j === m - 1 && i === 0) continue;

        const b = arr[j];
        const nextIsDepot = j === m - 1;
        const next = nextIsDepot ? -1 : arr[j + 1];

        const oldCost =
          (prevIsDepot ? distDepot(a) : distIdx(prev, a)) +
          (nextIsDepot ? distDepot(b) : distIdx(b, next));
        const newCost =
          (prevIsDepot ? distDepot(b) : distIdx(prev, b)) +
          (nextIsDepot ? distDepot(a) : distIdx(a, next));

        if (newCost + eps < oldCost) {
          reverseRange(arr, i, j);
          improved = true;
        }
      }
    }

    if (!improved) break;
  }

  return arr;
}

function orOpt(order: number[], passes: number, window: number): number[] {
  const m = order.length;
  if (m < 4) return order.slice();

  const arr = order.slice();
  const eps = 1e-9;

  for (let pass = 0; pass < passes; pass++) {
    let improved = false;

    for (let i = 0; i < m; i++) {
      const node = arr[i];
      const prevI = i === 0 ? -1 : arr[i - 1];
      const nextI = i === m - 1 ? -1 : arr[i + 1];

      const removeCost =
        (prevI === -1 ? distDepot(node) : distIdx(prevI, node)) +
        (nextI === -1 ? distDepot(node) : distIdx(node, nextI)) -
        (prevI === -1
          ? nextI === -1
            ? 0
            : distDepot(nextI)
          : nextI === -1
            ? distDepot(prevI)
            : distIdx(prevI, nextI));

      const start = Math.max(0, i - window);
      const end = Math.min(m - 1, i + window);

      for (let pos = start; pos <= end; pos++) {
        if (pos === i || pos === i - 1) continue;

        const left = arr[pos];
        const right = pos === m - 1 ? -1 : arr[pos + 1];

        const insertCost =
          (left === -1 ? distDepot(node) : distIdx(left, node)) +
          (right === -1 ? distDepot(node) : distIdx(node, right)) -
          (right === -1 ? distDepot(left) : distIdx(left, right));

        if (insertCost + eps < removeCost) {
          const [moved] = arr.splice(i, 1);
          const insertAt = pos < i ? pos + 1 : pos;
          arr.splice(insertAt, 0, moved);
          improved = true;
          break;
        }
      }

      if (improved) break;
    }

    if (!improved) break;
  }

  return arr;
}

function improve(order: number[]): number[] {
  let best = order.slice();
  const w1 = Math.max(12, Math.min(64, Math.floor(Math.sqrt(n) * 3)));
  const w2 = Math.max(8, Math.min(32, Math.floor(Math.sqrt(n) * 2)));
  best = twoOpt(best, 3, w1);
  best = orOpt(best, 2, w2);
  best = twoOpt(best, 2, w2);
  return best;
}

function makeRangeOrder(): number[] {
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  return order;
}

function angleOrder(): number[] {
  return makeRangeOrder().sort((a, b) => {
    const da = angles[a] - angles[b];
    if (da !== 0) return da;
    const dr = radii[a] - radii[b];
    if (dr !== 0) return dr;
    return a - b;
  });
}

function scaleTo16Bit(value: number, low: number, high: number): number {
  if (!(high > low)) return 0;
  const t = (value - low) / (high - low);
  const clamped = t < 0 ? 0 : t > 1 ? 1 : t;
  return Math.max(0, Math.min(65535, Math.floor(clamped * 65535)));
}

function part1By1(v: number): number {
  let x = v & 0x0000ffff;
  x = (x | (x << 8)) & 0x00ff00ff;
  x = (x | (x << 4)) & 0x0f0f0f0f;
  x = (x | (x << 2)) & 0x33333333;
  x = (x | (x << 1)) & 0x55555555;
  return x >>> 0;
}

function mortonCode(x: number, y: number): number {
  return ((part1By1(y) << 1) | part1By1(x)) >>> 0;
}

function mortonOrder(): number[] {
  const keyed = new Array<{ idx: number; key: number }>(n);
  for (let i = 0; i < n; i++) {
    const qx = scaleTo16Bit(xs[i], minX, maxX);
    const qy = scaleTo16Bit(ys[i], minY, maxY);
    keyed[i] = { idx: i, key: mortonCode(qx, qy) };
  }
  keyed.sort((a, b) => (a.key - b.key) || (a.idx - b.idx));
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = keyed[i].idx;
  return order;
}

function snakeOrder(primary: 'x' | 'y'): number[] {
  const idxs = makeRangeOrder().sort((a, b) => {
    const pa = primary === 'x' ? xs[a] : ys[a];
    const pb = primary === 'x' ? xs[b] : ys[b];
    if (pa !== pb) return pa - pb;
    const sa = primary === 'x' ? ys[a] : xs[a];
    const sb = primary === 'x' ? ys[b] : xs[b];
    if (sa !== sb) return sa - sb;
    return a - b;
  });

  const chunks = Math.max(1, Math.round(Math.sqrt(n)));
  const chunkSize = Math.max(1, Math.ceil(n / chunks));
  const order: number[] = [];

  for (let start = 0, block = 0; start < n; start += chunkSize, block++) {
    const slice = idxs.slice(start, Math.min(n, start + chunkSize));
    slice.sort((a, b) => {
      const sa = primary === 'x' ? ys[a] : xs[a];
      const sb = primary === 'x' ? ys[b] : xs[b];
      if (sa !== sb) return block % 2 === 0 ? sa - sb : sb - sa;
      return a - b;
    });
    for (let i = 0; i < slice.length; i++) order.push(slice[i]);
  }

  return order;
}

function chooseBest(candidates: number[][]): number[] {
  let bestOrder = candidates[0];
  let bestLength = Infinity;

  for (const candidate of candidates) {
    const improved = improve(candidate);
    const reversed = improve(improved.slice().reverse());

    const len1 = routeLength(improved);
    if (len1 < bestLength) {
      bestLength = len1;
      bestOrder = improved;
    }

    const len2 = routeLength(reversed);
    if (len2 < bestLength) {
      bestLength = len2;
      bestOrder = reversed;
    }
  }

  return bestOrder;
}

const initialCandidates: number[][] = [
  angleOrder(),
  mortonOrder(),
  snakeOrder('x'),
  snakeOrder('y'),
];

const bestOrder = chooseBest(initialCandidates);
const route = bestOrder.map((idx) => ids[idx]);

process.stdout.write(JSON.stringify({ routes: [route] }));
