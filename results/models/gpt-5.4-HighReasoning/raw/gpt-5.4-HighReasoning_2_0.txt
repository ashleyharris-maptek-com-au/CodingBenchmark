using System;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        if (m == 0)
        {
            Console.WriteLine(0);
            return;
        }

        int[] u = new int[m];
        int[] v = new int[m];
        int[] degree = new int[n];
        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[m * 2];
        int[] next = new int[m * 2];
        int[] edgeId = new int[m * 2];
        int edgePtr = 0;

        for (int i = 0; i < m; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            fs.NextInt(); // Weight is irrelevant for correctness; any closed walk covering all edges is valid.

            u[i] = a;
            v[i] = b;
            degree[a]++;
            degree[b]++;

            to[edgePtr] = b;
            edgeId[edgePtr] = i;
            next[edgePtr] = head[a];
            head[a] = edgePtr++;

            to[edgePtr] = a;
            edgeId[edgePtr] = i;
            next[edgePtr] = head[b];
            head[b] = edgePtr++;
        }

        bool[] seen = new bool[n];
        int[] queue = new int[n];
        int[] order = new int[n];
        int[] parent = new int[n];
        int[] parentEdge = new int[n];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        int qh = 0, qt = 0, orderLen = 0;
        seen[0] = true;
        parent[0] = 0;
        queue[qt++] = 0;

        while (qh < qt)
        {
            int cur = queue[qh++];
            order[orderLen++] = cur;

            for (int e = head[cur]; e != -1; e = next[e])
            {
                int nx = to[e];
                if (seen[nx]) continue;
                seen[nx] = true;
                parent[nx] = cur;
                parentEdge[nx] = edgeId[e];
                queue[qt++] = nx;
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (degree[i] > 0 && !seen[i])
            {
                Console.WriteLine(0);
                return;
            }
        }

        int[] extraCopies = new int[m];
        int[] parity = new int[n];
        for (int i = 0; i < n; i++) parity[i] = degree[i] & 1;

        long totalCopies = m;
        for (int i = orderLen - 1; i >= 1; i--)
        {
            int node = order[i];
            if ((parity[node] & 1) != 0)
            {
                int pe = parentEdge[node];
                extraCopies[pe]++;
                totalCopies++;
                parity[parent[node]] ^= 1;
            }
        }

        int[] eHead = new int[n];
        Array.Fill(eHead, -1);

        int edgeCopies = checked((int)totalCopies);
        int[] eTo = new int[edgeCopies * 2];
        int[] eNext = new int[edgeCopies * 2];
        int[] eCopyId = new int[edgeCopies * 2];
        int ePtr = 0;
        int copyId = 0;

        for (int i = 0; i < m; i++)
        {
            int copies = 1 + extraCopies[i];
            for (int c = 0; c < copies; c++)
            {
                int id = copyId++;

                eTo[ePtr] = v[i];
                eCopyId[ePtr] = id;
                eNext[ePtr] = eHead[u[i]];
                eHead[u[i]] = ePtr++;

                eTo[ePtr] = u[i];
                eCopyId[ePtr] = id;
                eNext[ePtr] = eHead[v[i]];
                eHead[v[i]] = ePtr++;
            }
        }

        int[] iter = new int[n];
        Array.Copy(eHead, iter, n);
        bool[] usedCopy = new bool[edgeCopies];
        int[] stack = new int[edgeCopies + 1];
        int[] route = new int[edgeCopies + 1];
        int top = 0;
        int routeLen = 0;
        stack[0] = 0;

        while (top >= 0)
        {
            int cur = stack[top];
            int e = iter[cur];
            while (e != -1 && usedCopy[eCopyId[e]]) e = eNext[e];
            iter[cur] = e;

            if (e == -1)
            {
                route[routeLen++] = cur;
                top--;
            }
            else
            {
                int id = eCopyId[e];
                usedCopy[id] = true;
                iter[cur] = eNext[e];
                stack[++top] = eTo[e];
            }
        }

        var output = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        for (int i = routeLen - 1; i >= 0; i--)
        {
            if (i != routeLen - 1) output.Write(' ');
            output.Write(route[i]);
        }
        output.WriteLine();
        output.Flush();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }
}
