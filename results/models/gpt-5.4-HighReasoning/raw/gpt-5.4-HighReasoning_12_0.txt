using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Numerics;

public static class Program
{
    private const double Eps = 1e-9;

    public static void Main()
    {
        var solver = new Solver();
        solver.Run(Console.In, Console.Out);
    }

    private sealed class Solver
    {
        public void Run(TextReader reader, TextWriter writer)
        {
            var scanner = new FastScanner(reader);
            int vertexCount = scanner.NextInt();
            int faceCount = scanner.NextInt();

            var vertices = new Vector3[vertexCount];
            for (int i = 0; i < vertexCount; i++)
            {
                vertices[i] = new Vector3(
                    (float)scanner.NextDouble(),
                    (float)scanner.NextDouble(),
                    (float)scanner.NextDouble());
            }

            for (int i = 0; i < faceCount; i++)
            {
                scanner.NextInt();
                scanner.NextInt();
                scanner.NextInt();
            }

            double containerW = scanner.NextDouble();
            double containerH = scanner.NextDouble();
            double containerD = scanner.NextDouble();

            var shape = AnalyzeShape(vertices);
            var scale = ClassifyScale(vertexCount, faceCount, containerW, containerH, containerD);
            var candidateOrientations = BuildOrientations(shape, scale);
            var pack = BuildBestPacking(shape, candidateOrientations, containerW, containerH, containerD, scale);

            writer.WriteLine(pack.Placements.Count.ToString(CultureInfo.InvariantCulture));
            foreach (var placement in pack.Placements)
            {
                writer.WriteLine(string.Create(
                    CultureInfo.InvariantCulture,
                    $"{placement.Tx:F6} {placement.Ty:F6} {placement.Tz:F6} {placement.Qw:F6} {placement.Qx:F6} {placement.Qy:F6} {placement.Qz:F6}"));
            }
        }

        private static ShapeInfo AnalyzeShape(Vector3[] vertices)
        {
            if (vertices.Length == 0)
            {
                return new ShapeInfo(vertices, Vector3.Zero, Vector3.Zero, Vector3.Zero, 0.0);
            }

            Vector3 min = vertices[0];
            Vector3 max = vertices[0];
            Vector3 sum = Vector3.Zero;
            foreach (var v in vertices)
            {
                min = Vector3.Min(min, v);
                max = Vector3.Max(max, v);
                sum += v;
            }

            Vector3 centroid = sum / vertices.Length;
            double radius = 0.0;
            foreach (var v in vertices)
            {
                radius = Math.Max(radius, Vector3.Distance(centroid, v));
            }

            return new ShapeInfo(vertices, min, max, centroid, radius);
        }

        private static ProblemScale ClassifyScale(int vertexCount, int faceCount, double w, double h, double d)
        {
            double maxSide = Math.Max(w, Math.Max(h, d));
            long complexity = (long)vertexCount + faceCount;

            if (maxSide <= 8.0 && complexity <= 200)
            {
                return ProblemScale.Trivial;
            }

            if (maxSide <= 30.0 && complexity <= 10_000)
            {
                return ProblemScale.Medium;
            }

            if (maxSide <= 120.0 && complexity <= 100_000)
            {
                return ProblemScale.Large;
            }

            return ProblemScale.Extreme;
        }

        private static List<OrientationCandidate> BuildOrientations(ShapeInfo shape, ProblemScale scale)
        {
            var candidates = new List<Quaternion>();
            AddAxisAlignedOrientations(candidates);
            AddPrincipalAxisOrientations(shape, candidates);

            int targetSamples = scale switch
            {
                ProblemScale.Trivial => 384,
                ProblemScale.Medium => 192,
                ProblemScale.Large => 80,
                _ => 32
            };

            AddSampledOrientations(candidates, targetSamples);

            var unique = DeduplicateQuaternions(candidates);
            var result = new List<OrientationCandidate>(unique.Count);
            foreach (var q in unique)
            {
                result.Add(EvaluateOrientation(shape, q));
            }

            result.Sort((a, b) =>
            {
                int cmp = a.BoundingVolume.CompareTo(b.BoundingVolume);
                if (cmp != 0) return cmp;
                cmp = a.MaxExtent.CompareTo(b.MaxExtent);
                if (cmp != 0) return cmp;
                return a.SurfaceFootprint.CompareTo(b.SurfaceFootprint);
            });

            int keep = scale switch
            {
                ProblemScale.Trivial => Math.Min(result.Count, 96),
                ProblemScale.Medium => Math.Min(result.Count, 64),
                ProblemScale.Large => Math.Min(result.Count, 28),
                _ => Math.Min(result.Count, 12)
            };

            return result.Take(keep).ToList();
        }

        private static void AddAxisAlignedOrientations(List<Quaternion> quats)
        {
            var axes = new[]
            {
                Vector3.UnitX, -Vector3.UnitX,
                Vector3.UnitY, -Vector3.UnitY,
                Vector3.UnitZ, -Vector3.UnitZ
            };

            foreach (var forward in axes)
            {
                foreach (var up in axes)
                {
                    if (Math.Abs(Vector3.Dot(forward, up)) > 0.001f)
                    {
                        continue;
                    }

                    var m = Matrix4x4.CreateWorld(Vector3.Zero, forward, up);
                    Quaternion q = Quaternion.CreateFromRotationMatrix(Matrix4x4.Transpose(m));
                    quats.Add(Quaternion.Normalize(q));
                }
            }
        }

        private static void AddPrincipalAxisOrientations(ShapeInfo shape, List<Quaternion> quats)
        {
            if (shape.Vertices.Length == 0)
            {
                quats.Add(Quaternion.Identity);
                return;
            }

            var basis = EstimatePrincipalBasis(shape.Vertices, shape.Centroid);
            var signs = new[] { -1f, 1f };
            foreach (float sx in signs)
            {
                foreach (float sy in signs)
                {
                    Vector3 x = basis.X * sx;
                    Vector3 y = basis.Y * sy;
                    Vector3 z = Vector3.Normalize(Vector3.Cross(x, y));
                    if (z.LengthSquared() < 0.5f)
                    {
                        continue;
                    }

                    y = Vector3.Normalize(Vector3.Cross(z, x));
                    var m = new Matrix4x4(
                        x.X, x.Y, x.Z, 0,
                        y.X, y.Y, y.Z, 0,
                        z.X, z.Y, z.Z, 0,
                        0, 0, 0, 1);
                    Quaternion q = Quaternion.CreateFromRotationMatrix(m);
                    quats.Add(Quaternion.Normalize(q));
                }
            }
        }

        private static Basis EstimatePrincipalBasis(Vector3[] vertices, Vector3 centroid)
        {
            double xx = 0, xy = 0, xz = 0, yy = 0, yz = 0, zz = 0;
            foreach (var v in vertices)
            {
                Vector3 d = v - centroid;
                xx += d.X * d.X;
                xy += d.X * d.Y;
                xz += d.X * d.Z;
                yy += d.Y * d.Y;
                yz += d.Y * d.Z;
                zz += d.Z * d.Z;
            }

            var matrix = new double[,]
            {
                { xx, xy, xz },
                { xy, yy, yz },
                { xz, yz, zz }
            };

            Vector3 xAxis = PowerIteration(matrix, new Vector3(1, 1, 0.25f));
            double lambda = RayleighQuotient(matrix, xAxis);
            var deflated = (double[,])matrix.Clone();
            Deflate(deflated, xAxis, lambda);

            Vector3 yAxis = PowerIteration(deflated, new Vector3(-0.5f, 1, 0.75f));
            if (Vector3.Cross(xAxis, yAxis).LengthSquared() < 1e-6f)
            {
                yAxis = Vector3.Normalize(AnyPerpendicular(xAxis));
            }

            Vector3 zAxis = Vector3.Normalize(Vector3.Cross(xAxis, yAxis));
            yAxis = Vector3.Normalize(Vector3.Cross(zAxis, xAxis));
            return new Basis(xAxis, yAxis, zAxis);
        }

        private static Vector3 PowerIteration(double[,] matrix, Vector3 seed)
        {
            Vector3 v = Vector3.Normalize(seed.LengthSquared() < 1e-8f ? Vector3.UnitX : seed);
            for (int i = 0; i < 16; i++)
            {
                var next = new Vector3(
                    (float)(matrix[0, 0] * v.X + matrix[0, 1] * v.Y + matrix[0, 2] * v.Z),
                    (float)(matrix[1, 0] * v.X + matrix[1, 1] * v.Y + matrix[1, 2] * v.Z),
                    (float)(matrix[2, 0] * v.X + matrix[2, 1] * v.Y + matrix[2, 2] * v.Z));
                if (next.LengthSquared() < 1e-12f)
                {
                    break;
                }

                v = Vector3.Normalize(next);
            }

            return v;
        }

        private static double RayleighQuotient(double[,] matrix, Vector3 v)
        {
            Vector3 mv = new Vector3(
                (float)(matrix[0, 0] * v.X + matrix[0, 1] * v.Y + matrix[0, 2] * v.Z),
                (float)(matrix[1, 0] * v.X + matrix[1, 1] * v.Y + matrix[1, 2] * v.Z),
                (float)(matrix[2, 0] * v.X + matrix[2, 1] * v.Y + matrix[2, 2] * v.Z));
            return Vector3.Dot(v, mv);
        }

        private static void Deflate(double[,] matrix, Vector3 v, double lambda)
        {
            double[] a = { v.X, v.Y, v.Z };
            for (int r = 0; r < 3; r++)
            {
                for (int c = 0; c < 3; c++)
                {
                    matrix[r, c] -= lambda * a[r] * a[c];
                }
            }
        }

        private static Vector3 AnyPerpendicular(Vector3 axis)
        {
            Vector3 helper = Math.Abs(axis.X) < 0.7f ? Vector3.UnitX : Vector3.UnitY;
            return Vector3.Cross(axis, helper);
        }

        private static void AddSampledOrientations(List<Quaternion> quats, int targetSamples)
        {
            int axisSamples = Math.Max(8, (int)Math.Sqrt(targetSamples));
            int spinSamples = Math.Max(4, targetSamples / axisSamples);
            int idx = 0;

            foreach (var axis in FibonacciDirections(axisSamples))
            {
                Vector3 unitAxis = Vector3.Normalize(axis);
                for (int s = 0; s < spinSamples; s++)
                {
                    double angle = (2.0 * Math.PI * s) / spinSamples;
                    Quaternion baseRot = RotationFromTo(Vector3.UnitZ, unitAxis);
                    Quaternion spin = Quaternion.CreateFromAxisAngle(unitAxis, (float)angle);
                    Quaternion q = Quaternion.Normalize(spin * baseRot);
                    quats.Add(q);
                    idx++;
                    if (idx >= targetSamples)
                    {
                        return;
                    }
                }
            }
        }

        private static IEnumerable<Vector3> FibonacciDirections(int count)
        {
            double goldenAngle = Math.PI * (3.0 - Math.Sqrt(5.0));
            for (int i = 0; i < count; i++)
            {
                double y = 1.0 - (2.0 * i + 1.0) / count;
                double radius = Math.Sqrt(Math.Max(0.0, 1.0 - y * y));
                double theta = goldenAngle * i;
                yield return new Vector3(
                    (float)(Math.Cos(theta) * radius),
                    (float)y,
                    (float)(Math.Sin(theta) * radius));
            }
        }

        private static Quaternion RotationFromTo(Vector3 from, Vector3 to)
        {
            from = Vector3.Normalize(from);
            to = Vector3.Normalize(to);
            float dot = Vector3.Dot(from, to);
            if (dot > 0.999999f)
            {
                return Quaternion.Identity;
            }

            if (dot < -0.999999f)
            {
                return Quaternion.CreateFromAxisAngle(Vector3.Normalize(AnyPerpendicular(from)), MathF.PI);
            }

            Vector3 axis = Vector3.Normalize(Vector3.Cross(from, to));
            float angle = MathF.Acos(Math.Clamp(dot, -1f, 1f));
            return Quaternion.CreateFromAxisAngle(axis, angle);
        }

        private static List<Quaternion> DeduplicateQuaternions(List<Quaternion> quats)
        {
            var seen = new HashSet<string>(StringComparer.Ordinal);
            var result = new List<Quaternion>();
            foreach (var q0 in quats)
            {
                Quaternion q = Quaternion.Normalize(q0);
                if (q.W < 0)
                {
                    q = new Quaternion(-q.X, -q.Y, -q.Z, -q.W);
                }

                string key = $"{Math.Round(q.W, 5):F5}|{Math.Round(q.X, 5):F5}|{Math.Round(q.Y, 5):F5}|{Math.Round(q.Z, 5):F5}";
                if (seen.Add(key))
                {
                    result.Add(q);
                }
            }

            return result;
        }

        private static OrientationCandidate EvaluateOrientation(ShapeInfo shape, Quaternion q)
        {
            if (shape.Vertices.Length == 0)
            {
                return new OrientationCandidate(q, Vector3.Zero, Vector3.Zero, Vector3.Zero);
            }

            Vector3 min = Rotate(shape.Vertices[0], q);
            Vector3 max = min;
            foreach (var vertex in shape.Vertices)
            {
                Vector3 r = Rotate(vertex, q);
                min = Vector3.Min(min, r);
                max = Vector3.Max(max, r);
            }

            Vector3 dims = max - min;
            return new OrientationCandidate(q, min, max, dims);
        }

        private static PackPlan BuildBestPacking(
            ShapeInfo shape,
            List<OrientationCandidate> orientations,
            double containerW,
            double containerH,
            double containerD,
            ProblemScale scale)
        {
            var boxes = new List<Box>();
            boxes.Add(new Box(0, 0, 0, containerW, containerH, containerD));
            var plan = new PackPlan();

            int maxPlacementsToMaterialize = scale switch
            {
                ProblemScale.Trivial => int.MaxValue,
                ProblemScale.Medium => 1_000_000,
                ProblemScale.Large => 400_000,
                _ => 250_000
            };

            int orientationBudget = scale switch
            {
                ProblemScale.Trivial => orientations.Count,
                ProblemScale.Medium => Math.Min(orientations.Count, 24),
                ProblemScale.Large => Math.Min(orientations.Count, 12),
                _ => Math.Min(orientations.Count, 6)
            };

            int boxBudget = scale switch
            {
                ProblemScale.Trivial => 128,
                ProblemScale.Medium => 96,
                ProblemScale.Large => 48,
                _ => 16
            };

            int depthLimit = scale switch
            {
                ProblemScale.Trivial => 4,
                ProblemScale.Medium => 3,
                ProblemScale.Large => 2,
                _ => 1
            };

            boxes.Sort(BoxComparer.Instance);
            SearchBest(
                boxes,
                orientations.Take(orientationBudget).ToList(),
                0,
                depthLimit,
                boxBudget,
                maxPlacementsToMaterialize,
                ref plan,
                new List<PlacementBlock>());

            if (plan.Placements.Count == 0)
            {
                plan = FallbackSingleBest(containerW, containerH, containerD, orientations, maxPlacementsToMaterialize);
            }

            return plan;
        }

        private static void SearchBest(
            List<Box> boxes,
            List<OrientationCandidate> orientations,
            int depth,
            int depthLimit,
            int boxBudget,
            int placementBudget,
            ref PackPlan best,
            List<PlacementBlock> placedBlocks)
        {
            long currentCount = placedBlocks.Sum(x => x.TotalCount);
            if (currentCount > best.TotalCount)
            {
                best = MaterializePlan(placedBlocks, placementBudget);
            }

            if (depth >= depthLimit || boxes.Count == 0)
            {
                return;
            }

            int usableBoxes = Math.Min(boxes.Count, boxBudget);
            for (int bi = 0; bi < usableBoxes; bi++)
            {
                Box box = boxes[bi];
                foreach (var orientation in orientations)
                {
                    var fill = EvaluateFill(box, orientation);
                    if (fill.Count <= 0)
                    {
                        continue;
                    }

                    long optimistic = currentCount + fill.Count;
                    for (int k = 0; k < boxes.Count; k++)
                    {
                        if (k == bi)
                        {
                            continue;
                        }

                        optimistic += UpperBound(boxes[k], orientations);
                    }

                    if (optimistic <= best.TotalCount)
                    {
                        continue;
                    }

                    var nextBlocks = new List<PlacementBlock>(placedBlocks)
                    {
                        fill.Block
                    };
                    var nextBoxes = SplitBoxes(boxes, bi, fill.Block);
                    SearchBest(nextBoxes, orientations, depth + 1, depthLimit, boxBudget, placementBudget, ref best, nextBlocks);
                }
            }
        }

        private static FillResult EvaluateFill(Box box, OrientationCandidate orientation)
        {
            double dx = orientation.Dimensions.X;
            double dy = orientation.Dimensions.Y;
            double dz = orientation.Dimensions.Z;
            if (dx <= Eps || dy <= Eps || dz <= Eps)
            {
                return FillResult.Empty;
            }

            long nx = (long)Math.Floor((box.W + Eps) / dx);
            long ny = (long)Math.Floor((box.H + Eps) / dy);
            long nz = (long)Math.Floor((box.D + Eps) / dz);
            if (nx <= 0 || ny <= 0 || nz <= 0)
            {
                return FillResult.Empty;
            }

            long count = SafeMultiply(nx, ny, nz);
            if (count <= 0)
            {
                return FillResult.Empty;
            }

            var block = new PlacementBlock(box.X, box.Y, box.Z, nx, ny, nz, orientation);
            return new FillResult(count, block);
        }

        private static long UpperBound(Box box, List<OrientationCandidate> orientations)
        {
            long best = 0;
            foreach (var orientation in orientations)
            {
                double dx = orientation.Dimensions.X;
                double dy = orientation.Dimensions.Y;
                double dz = orientation.Dimensions.Z;
                if (dx <= Eps || dy <= Eps || dz <= Eps)
                {
                    continue;
                }

                long nx = (long)Math.Floor((box.W + Eps) / dx);
                long ny = (long)Math.Floor((box.H + Eps) / dy);
                long nz = (long)Math.Floor((box.D + Eps) / dz);
                if (nx <= 0 || ny <= 0 || nz <= 0)
                {
                    continue;
                }

                best = Math.Max(best, SafeMultiply(nx, ny, nz));
            }

            return best;
        }

        private static List<Box> SplitBoxes(List<Box> boxes, int usedBoxIndex, PlacementBlock block)
        {
            var result = new List<Box>(boxes.Count + 4);
            for (int i = 0; i < boxes.Count; i++)
            {
                if (i != usedBoxIndex)
                {
                    result.Add(boxes[i]);
                }
            }

            Box source = boxes[usedBoxIndex];
            double usedW = block.Nx * block.Orientation.Dimensions.X;
            double usedH = block.Ny * block.Orientation.Dimensions.Y;
            double usedD = block.Nz * block.Orientation.Dimensions.Z;

            AddBoxIfPositive(result, new Box(source.X + usedW, source.Y, source.Z, source.W - usedW, usedH, usedD));
            AddBoxIfPositive(result, new Box(source.X, source.Y + usedH, source.Z, source.W, source.H - usedH, usedD));
            AddBoxIfPositive(result, new Box(source.X, source.Y, source.Z + usedD, source.W, source.H, source.D - usedD));
            AddBoxIfPositive(result, new Box(source.X + usedW, source.Y + usedH, source.Z, source.W - usedW, source.H - usedH, usedD));
            AddBoxIfPositive(result, new Box(source.X + usedW, source.Y, source.Z + usedD, source.W - usedW, usedH, source.D - usedD));
            AddBoxIfPositive(result, new Box(source.X, source.Y + usedH, source.Z + usedD, source.W, source.H - usedH, source.D - usedD));

            result.Sort(BoxComparer.Instance);
            if (result.Count > 128)
            {
                result.RemoveRange(128, result.Count - 128);
            }

            return result;
        }

        private static void AddBoxIfPositive(List<Box> boxes, Box box)
        {
            if (box.W > Eps && box.H > Eps && box.D > Eps)
            {
                boxes.Add(box);
            }
        }

        private static PackPlan MaterializePlan(List<PlacementBlock> blocks, int placementBudget)
        {
            var plan = new PackPlan();
            foreach (var block in blocks)
            {
                plan.TotalCount += block.TotalCount;
            }

            if (plan.TotalCount > placementBudget)
            {
                return plan;
            }

            plan.Placements = new List<Placement>((int)plan.TotalCount);
            foreach (var block in blocks)
            {
                Quaternion q = block.Orientation.Quaternion;
                double dx = block.Orientation.Dimensions.X;
                double dy = block.Orientation.Dimensions.Y;
                double dz = block.Orientation.Dimensions.Z;
                Vector3 min = block.Orientation.Min;

                for (long iz = 0; iz < block.Nz; iz++)
                {
                    double baseZ = block.Z + iz * dz;
                    for (long iy = 0; iy < block.Ny; iy++)
                    {
                        double baseY = block.Y + iy * dy;
                        for (long ix = 0; ix < block.Nx; ix++)
                        {
                            double baseX = block.X + ix * dx;
                            plan.Placements.Add(new Placement(
                                baseX - min.X,
                                baseY - min.Y,
                                baseZ - min.Z,
                                q.W, q.X, q.Y, q.Z));
                        }
                    }
                }
            }

            return plan;
        }

        private static PackPlan FallbackSingleBest(
            double containerW,
            double containerH,
            double containerD,
            List<OrientationCandidate> orientations,
            int placementBudget)
        {
            FillResult bestFill = FillResult.Empty;
            foreach (var orientation in orientations)
            {
                var fill = EvaluateFill(new Box(0, 0, 0, containerW, containerH, containerD), orientation);
                if (fill.Count > bestFill.Count)
                {
                    bestFill = fill;
                }
            }

            if (bestFill.Count <= 0)
            {
                return new PackPlan();
            }

            return MaterializePlan(new List<PlacementBlock> { bestFill.Block }, placementBudget);
        }

        private static long SafeMultiply(long a, long b, long c)
        {
            try
            {
                checked
                {
                    return a * b * c;
                }
            }
            catch
            {
                return long.MaxValue;
            }
        }

        private static Vector3 Rotate(Vector3 v, Quaternion q)
        {
            return Vector3.Transform(v, q);
        }
    }

    private sealed class FastScanner
    {
        private readonly TextReader _reader;
        private readonly char[] _buffer = new char[1 << 16];
        private int _length;
        private int _index;

        public FastScanner(TextReader reader)
        {
            _reader = reader;
        }

        public int NextInt() => (int)NextDouble();

        public double NextDouble()
        {
            string token = NextToken();
            return double.Parse(token, CultureInfo.InvariantCulture);
        }

        private string NextToken()
        {
            SkipWhiteSpace();
            var chars = new List<char>(16);
            while (true)
            {
                int ch = Peek();
                if (ch < 0 || char.IsWhiteSpace((char)ch))
                {
                    break;
                }

                chars.Add((char)Read());
            }

            return new string(chars.ToArray());
        }

        private void SkipWhiteSpace()
        {
            while (true)
            {
                int ch = Peek();
                if (ch < 0 || !char.IsWhiteSpace((char)ch))
                {
                    return;
                }

                Read();
            }
        }

        private int Peek()
        {
            if (_index >= _length)
            {
                _length = _reader.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_index];
        }

        private int Read()
        {
            int ch = Peek();
            if (ch >= 0)
            {
                _index++;
            }

            return ch;
        }
    }

    private enum ProblemScale
    {
        Trivial,
        Medium,
        Large,
        Extreme
    }

    private readonly struct ShapeInfo
    {
        public ShapeInfo(Vector3[] vertices, Vector3 min, Vector3 max, Vector3 centroid, double radius)
        {
            Vertices = vertices;
            Min = min;
            Max = max;
            Centroid = centroid;
            Radius = radius;
        }

        public Vector3[] Vertices { get; }
        public Vector3 Min { get; }
        public Vector3 Max { get; }
        public Vector3 Centroid { get; }
        public double Radius { get; }
    }

    private readonly struct Basis
    {
        public Basis(Vector3 x, Vector3 y, Vector3 z)
        {
            X = Vector3.Normalize(x);
            Y = Vector3.Normalize(y);
            Z = Vector3.Normalize(z);
        }

        public Vector3 X { get; }
        public Vector3 Y { get; }
        public Vector3 Z { get; }
    }

    private readonly struct OrientationCandidate
    {
        public OrientationCandidate(Quaternion quaternion, Vector3 min, Vector3 max, Vector3 dimensions)
        {
            Quaternion = quaternion;
            Min = min;
            Max = max;
            Dimensions = new Vector3(
                Math.Max(dimensions.X, 0f),
                Math.Max(dimensions.Y, 0f),
                Math.Max(dimensions.Z, 0f));
        }

        public Quaternion Quaternion { get; }
        public Vector3 Min { get; }
        public Vector3 Max { get; }
        public Vector3 Dimensions { get; }
        public double BoundingVolume => Dimensions.X * Dimensions.Y * Dimensions.Z;
        public double MaxExtent => Math.Max(Dimensions.X, Math.Max(Dimensions.Y, Dimensions.Z));
        public double SurfaceFootprint => Dimensions.X * Dimensions.Y;
    }

    private readonly struct Box
    {
        public Box(double x, double y, double z, double w, double h, double d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }

        public double X { get; }
        public double Y { get; }
        public double Z { get; }
        public double W { get; }
        public double H { get; }
        public double D { get; }
        public double Volume => W * H * D;
    }

    private sealed class BoxComparer : IComparer<Box>
    {
        public static readonly BoxComparer Instance = new BoxComparer();

        public int Compare(Box a, Box b)
        {
            int cmp = b.Volume.CompareTo(a.Volume);
            if (cmp != 0) return cmp;
            cmp = b.W.CompareTo(a.W);
            if (cmp != 0) return cmp;
            cmp = b.H.CompareTo(a.H);
            if (cmp != 0) return cmp;
            return b.D.CompareTo(a.D);
        }
    }

    private readonly struct PlacementBlock
    {
        public PlacementBlock(double x, double y, double z, long nx, long ny, long nz, OrientationCandidate orientation)
        {
            X = x;
            Y = y;
            Z = z;
            Nx = nx;
            Ny = ny;
            Nz = nz;
            Orientation = orientation;
        }

        public double Z { get; }
        public long Nx { get; }
        public long Ny { get; }
        public long Nz { get; }
        public OrientationCandidate Orientation { get; }
        public long TotalCount => Nx * Ny * Nz;
    }
