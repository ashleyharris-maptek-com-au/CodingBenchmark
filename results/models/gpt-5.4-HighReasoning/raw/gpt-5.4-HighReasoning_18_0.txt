using namespace std;

struct WeightedEdge {
    int u;
    int v;
    long long w;
};

static uint64_t pack_edge(uint32_t u, uint32_t v) {
    return (uint64_t(u) << 32) | uint64_t(v);
}

struct TempEdgeFile {
    filesystem::path path;
    ofstream out;

    TempEdgeFile() {
        auto now = chrono::high_resolution_clock::now().time_since_epoch().count();
        path = filesystem::temp_directory_path() /
               ("mincut_edges_" + to_string(now) + ".bin");
        out.open(path, ios::binary);
        if (!out) {
            throw runtime_error("failed to create temporary edge file");
        }
    }

    void write_edge(uint32_t u, uint32_t v) {
        uint32_t buf[2] = {u, v};
        out.write(reinterpret_cast<const char*>(buf), sizeof(buf));
        if (!out) {
            throw runtime_error("failed to write temporary edge data");
        }
    }

    void close_writer() {
        out.close();
    }

    ~TempEdgeFile() {
        std::error_code ec;
        if (out.is_open()) out.close();
        filesystem::remove(path, ec);
    }
};

class Dinic {
public:
    struct Edge {
        int to;
        int rev;
        long long cap;
    };

    explicit Dinic(int n) : g_(n), level_(n), it_(n) {}

    void add_directed(int u, int v, long long cap) {
        Edge a{v, static_cast<int>(g_[v].size()), cap};
        Edge b{u, static_cast<int>(g_[u].size()), 0};
        g_[u].push_back(a);
        g_[v].push_back(b);
    }

    void add_undirected(int u, int v, long long cap) {
        add_directed(u, v, cap);
        add_directed(v, u, cap);
    }

    long long maxflow(int s, int t) {
        long long flow = 0;
        const long long INF = numeric_limits<long long>::max() / 4;
        while (bfs(s, t)) {
            fill(it_.begin(), it_.end(), 0);
            while (long long pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<char> reachable_from(int s) const {
        vector<char> seen(g_.size(), 0);
        queue<int> q;
        q.push(s);
        seen[s] = 1;
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto& e : g_[v]) {
                if (e.cap > 0 && !seen[e.to]) {
                    seen[e.to] = 1;
                    q.push(e.to);
                }
            }
        }
        return seen;
    }

private:
    bool bfs(int s, int t) {
        fill(level_.begin(), level_.end(), -1);
        queue<int> q;
        level_[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto& e : g_[v]) {
                if (e.cap > 0 && level_[e.to] < 0) {
                    level_[e.to] = level_[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level_[t] >= 0;
    }

    long long dfs(int v, int t, long long pushed) {
        if (v == t) return pushed;
        for (int& i = it_[v]; i < static_cast<int>(g_[v].size()); ++i) {
            Edge& e = g_[v][i];
            if (e.cap <= 0 || level_[e.to] != level_[v] + 1) continue;
            long long got = dfs(e.to, t, min(pushed, e.cap));
            if (got == 0) continue;
            e.cap -= got;
            g_[e.to][e.rev].cap += got;
            return got;
        }
        return 0;
    }

    vector<vector<Edge>> g_;
    vector<int> level_;
    vector<int> it_;
};

static pair<long long, vector<char>> maxflow_mincut_partition(
    int n,
    const vector<WeightedEdge>& edges,
    int s,
    int t) {
    Dinic dinic(n);
    for (const auto& e : edges) {
        dinic.add_undirected(e.u, e.v, e.w);
    }
    long long flow = dinic.maxflow(s, t);
    return {flow, dinic.reachable_from(s)};
}

static pair<long long, vector<char>> gomory_hu_global_mincut(
    int n,
    const vector<WeightedEdge>& edges) {
    if (n <= 1) return {0, vector<char>(n, 0)};

    vector<int> parent(n, 0);
    vector<long long> cut_value(n, 0);
    vector<vector<char>> side_of(n);

    for (int s = 1; s < n; ++s) {
        int t = parent[s];
        auto [flow, side] = maxflow_mincut_partition(n, edges, s, t);
        cut_value[s] = flow;
        side_of[s] = std::move(side);

        for (int v = s + 1; v < n; ++v) {
            if (parent[v] == t && side_of[s][v]) {
                parent[v] = s;
            }
        }

        if (side_of[s][parent[t]]) {
            swap(cut_value[s], cut_value[t]);
            swap(side_of[s], side_of[t]);
            parent[s] = parent[t];
            parent[t] = s;
            for (int v = s + 1; v < n; ++v) {
                if (parent[v] == t && side_of[s][v]) {
                    parent[v] = s;
                }
            }
        }
    }

    long long best = cut_value[1];
    int best_vertex = 1;
    for (int i = 2; i < n; ++i) {
        if (cut_value[i] < best) {
            best = cut_value[i];
            best_vertex = i;
        }
    }
    return {best, side_of[best_vertex]};
}

static pair<long long, vector<char>> stoer_wagner_global_mincut(
    int n,
    const vector<WeightedEdge>& edges) {
    if (n <= 1) return {0, vector<char>(n, 0)};

    vector<unordered_map<int, long long>> g(n);
    g.reserve(n);
    for (const auto& e : edges) {
        g[e.u][e.v] += e.w;
        g[e.v][e.u] += e.w;
    }

    vector<char> alive(n, 1);
    vector<vector<int>> members(n);
    for (int i = 0; i < n; ++i) members[i].push_back(i);

    long long best = numeric_limits<long long>::max();
    vector<char> best_side(n, 0);
    int alive_count = n;

    while (alive_count > 1) {
        vector<long long> w(n, 0);
        vector<char> used(n, 0);
        priority_queue<pair<long long, int>> pq;

        int start = -1;
        for (int i = 0; i < n; ++i) {
            if (alive[i]) {
                start = i;
                break;
            }
        }

        pq.push({0, start});
        int prev = -1;
        int last = -1;

        for (int iter = 0; iter < alive_count; ++iter) {
            int v = -1;
            while (!pq.empty()) {
                auto [score, node] = pq.top();
                pq.pop();
                if (!alive[node] || used[node] || score != w[node]) continue;
                v = node;
                break;
            }
            if (v == -1) {
                for (int i = 0; i < n; ++i) {
                    if (alive[i] && !used[i]) {
                        v = i;
                        break;
                    }
                }
            }

            used[v] = 1;
            prev = last;
            last = v;

            if (iter == alive_count - 1) {
                if (w[v] < best) {
                    best = w[v];
                    fill(best_side.begin(), best_side.end(), 0);
                    for (int x : members[v]) best_side[x] = 1;
                }

                if (prev != -1) {
                    for (const auto& [to, cap] : g[v]) {
                        if (to == prev || !alive[to]) continue;
                        g[prev][to] += cap;
                        g[to][prev] += cap;
                        g[to].erase(v);
                    }
                    g[prev].erase(v);
                    members[prev].insert(members[prev].end(), members[v].begin(), members[v].end());
                    members[v].clear();
                    g[v].clear();
                    alive[v] = 0;
                    --alive_count;
                }
                break;
            }

            for (const auto& [to, cap] : g[v]) {
                if (alive[to] && !used[to]) {
                    w[to] += cap;
                    pq.push({w[to], to});
                }
            }
        }
    }

    return {best, best_side};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m_input;
    if (!(cin >> n >> m_input)) {
        return 0;
    }

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    TempEdgeFile temp_edges;
    vector<uint64_t> normalized_edges;
    if (m_input > 0) {
        normalized_edges.reserve(static_cast<size_t>(m_input));
    }

    for (long long i = 0; i < m_input; ++i) {
        uint32_t u, v;
        cin >> u >> v;
        if (u == v) continue;
        temp_edges.write_edge(u, v);
        if (u > v) swap(u, v);
        normalized_edges.push_back(pack_edge(u, v));
    }
    temp_edges.close_writer();

    sort(normalized_edges.begin(), normalized_edges.end());

    vector<WeightedEdge> edges;
    edges.reserve(normalized_edges.size());
    for (size_t i = 0; i < normalized_edges.size();) {
        size_t j = i + 1;
        while (j < normalized_edges.size() && normalized_edges[j] == normalized_edges[i]) ++j;
        uint64_t key = normalized_edges[i];
        int u = static_cast<int>(key >> 32);
        int v = static_cast<int>(key & 0xffffffffu);
        edges.push_back({u, v, static_cast<long long>(j - i)});
        i = j;
    }
    normalized_edges.clear();
    normalized_edges.shrink_to_fit();

    pair<long long, vector<char>> answer;

    const long long unique_m = static_cast<long long>(edges.size());
    const bool use_stoer_wagner =
        (n <= 3500) || (unique_m > 8LL * n) || (unique_m <= 250000 && n <= 12000);

    if (use_stoer_wagner) {
        answer = stoer_wagner_global_mincut(n, edges);
    } else {
        answer = gomory_hu_global_mincut(n, edges);
    }

    const vector<char>& side = answer.second;

    ifstream in(temp_edges.path, ios::binary);
    if (!in) {
        throw runtime_error("failed to reopen temporary edge file");
    }

    long long cut_edges = 0;
    uint32_t buf[2];
    while (in.read(reinterpret_cast<char*>(buf), sizeof(buf))) {
        if (side[buf[0]] != side[buf[1]]) {
            ++cut_edges;
        }
    }

    cout << cut_edges << '\n';

    in.clear();
    in.seekg(0);
    while (in.read(reinterpret_cast<char*>(buf), sizeof(buf))) {
        if (side[buf[0]] != side[buf[1]]) {
            cout << buf[0] << ' ' << buf[1] << '\n';
        }
    }

    return 0;
}
