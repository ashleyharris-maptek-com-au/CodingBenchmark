OpCapability Shader
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_0_025 = OpConstant %float 0.025
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_gl_PerVertex_in_4 = OpTypePointer Input %arr_gl_PerVertex_in_4
%ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_4 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_4 Input
%normal = OpVariable %ptr_Input_arr_v3float_4 Input
%color = OpVariable %ptr_Input_arr_v3float_4 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %3
%5 = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%t0_delta = OpFSub %v2float %p2_xy %p0_xy
%t1_delta = OpFSub %v2float %p3_xy %p1_xy
%t0 = OpExtInst %v2float %1 Normalize %t0_delta
%t1 = OpExtInst %v2float %1 Normalize %t1_delta
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t0_y = OpFNegate %float %t0_y
%neg_t1_y = OpFNegate %float %t1_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x
%off0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%off1 = OpVectorTimesScalar %v2float %perp1 %float_0_025
%start_plus_xy = OpFAdd %v2float %p1_xy %off0
%start_minus_xy = OpFSub %v2float %p1_xy %off0
%end_plus_xy = OpFAdd %v2float %p2_xy %off1
%end_minus_xy = OpFSub %v2float %p2_xy %off1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3
%sp_x = OpCompositeExtract %float %start_plus_xy 0
%sp_y = OpCompositeExtract %float %start_plus_xy 1
%sm_x = OpCompositeExtract %float %start_minus_xy 0
%sm_y = OpCompositeExtract %float %start_minus_xy 1
%ep_x = OpCompositeExtract %float %end_plus_xy 0
%ep_y = OpCompositeExtract %float %end_plus_xy 1
%em_x = OpCompositeExtract %float %end_minus_xy 0
%em_y = OpCompositeExtract %float %end_minus_xy 1
%start_plus = OpCompositeConstruct %v4float %sp_x %sp_y %p1_z %p1_w
%start_minus = OpCompositeConstruct %v4float %sm_x %sm_y %p1_z %p1_w
%end_plus = OpCompositeConstruct %v4float %ep_x %ep_y %p2_z %p2_w
%end_minus = OpCompositeConstruct %v4float %em_x %em_y %p2_z %p2_w
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%gl_out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos %start_plus
OpStore %outColor %c1
OpEmitVertex
OpStore %gl_out_pos %start_minus
OpStore %outColor %c1
OpEmitVertex
OpStore %gl_out_pos %end_plus
OpStore %outColor %c2
OpEmitVertex
OpStore %gl_out_pos %end_minus
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
