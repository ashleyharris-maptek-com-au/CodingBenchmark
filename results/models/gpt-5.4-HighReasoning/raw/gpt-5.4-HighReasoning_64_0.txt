declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type DroneInput = { id: string; x: number; y: number };

const probability: number[][] = Array.isArray(data.probability) ? data.probability : [];
const drones: DroneInput[] = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone: number = Math.max(0, (data.maxStepsPerDrone | 0) || 0);

const height = probability.length;
const width = height > 0 && Array.isArray(probability[0]) ? probability[0].length : 0;
const size = width * height;

function idx(x: number, y: number): number {
  return y * width + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

if (width === 0 || height === 0 || size === 0) {
  const emptyPaths = drones.map((d) => [[d.x | 0, d.y | 0]]);
  process.stdout.write(JSON.stringify({ paths: emptyPaths }));
  process.exit(0);
}

const values = new Float64Array(size);
const walkable = new Uint8Array(size);

for (let y = 0; y < height; y++) {
  const row = probability[y];
  for (let x = 0; x < width; x++) {
    const cell = Number(row[x]);
    const i = idx(x, y);
    if (cell >= 0) {
      walkable[i] = 1;
      values[i] = cell;
    } else {
      walkable[i] = 0;
      values[i] = -1;
    }
  }
}

const component = new Int32Array(size);
component.fill(-1);
let componentCount = 0;
const queue = new Int32Array(size);

for (let start = 0; start < size; start++) {
  if (!walkable[start] || component[start] !== -1) continue;
  let head = 0;
  let tail = 0;
  queue[tail++] = start;
  component[start] = componentCount;
  while (head < tail) {
    const cur = queue[head++];
    const x = cur % width;
    const y = (cur / width) | 0;
    if (x > 0) {
      const n = cur - 1;
      if (walkable[n] && component[n] === -1) {
        component[n] = componentCount;
        queue[tail++] = n;
      }
    }
    if (x + 1 < width) {
      const n = cur + 1;
      if (walkable[n] && component[n] === -1) {
        component[n] = componentCount;
        queue[tail++] = n;
      }
    }
    if (y > 0) {
      const n = cur - width;
      if (walkable[n] && component[n] === -1) {
        component[n] = componentCount;
        queue[tail++] = n;
      }
    }
    if (y + 1 < height) {
      const n = cur + width;
      if (walkable[n] && component[n] === -1) {
        component[n] = componentCount;
        queue[tail++] = n;
      }
    }
  }
  componentCount++;
}

const neighborhoodOffsets: Array<[number, number, number]> = [];
for (let dy = -2; dy <= 2; dy++) {
  for (let dx = -2; dx <= 2; dx++) {
    const dist = Math.abs(dx) + Math.abs(dy);
    if (dist === 0 || dist > 2) continue;
    const weight = dist === 1 ? 0.6 : 0.28;
    neighborhoodOffsets.push([dx, dy, weight]);
  }
}

const staticDensity = new Float64Array(size);
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const i = idx(x, y);
    if (!walkable[i]) continue;
    let score = values[i] > 0 ? values[i] : 0;
    for (let k = 0; k < neighborhoodOffsets.length; k++) {
      const [dx, dy, weight] = neighborhoodOffsets[k];
      const nx = x + dx;
      const ny = y + dy;
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (walkable[ni] && values[ni] > 0) score += values[ni] * weight;
    }
    staticDensity[i] = score;
  }
}

const seen = new Int32Array(size);
const dist = new Int32Array(size);
const parent = new Int32Array(size);
seen.fill(0);
parent.fill(-1);
let bfsStamp = 1;

const covered = new Uint8Array(size);
const visitCount = new Uint16Array(size);

type DroneState = {
  x: number;
  y: number;
  pos: number;
  remaining: number;
  path: number[][];
  active: boolean;
};

const states: DroneState[] = drones.map((d) => {
  const x = d.x | 0;
  const y = d.y | 0;
  const ok = inBounds(x, y) && walkable[idx(x, y)] === 1;
  const safeX = ok ? x : 0;
  const safeY = ok ? y : 0;
  const pos = idx(safeX, safeY);
  const path = [[safeX, safeY]];
  if (values[pos] > 0) covered[pos] = 1;
  visitCount[pos]++;
  return {
    x: safeX,
    y: safeY,
    pos,
    remaining: maxStepsPerDrone,
    path,
    active: ok,
  };
});

function dynamicNeighborhoodGain(cell: number): number {
  const x = cell % width;
  const y = (cell / width) | 0;
  let gain = covered[cell] || values[cell] <= 0 ? 0 : values[cell];
  for (let k = 0; k < neighborhoodOffsets.length; k++) {
    const [dx, dy, weight] = neighborhoodOffsets[k];
    const nx = x + dx;
    const ny = y + dy;
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!walkable[ni] || covered[ni] || values[ni] <= 0) continue;
    gain += values[ni] * weight;
  }
  return gain;
}

function evaluateCell(cell: number, distance: number, remaining: number): number {
  const immediate = !covered[cell] && values[cell] > 0 ? values[cell] : 0;
  const localGain = dynamicNeighborhoodGain(cell);
  const futureSlack = remaining - distance;
  const slackBonus = futureSlack > 0 ? Math.min(6, futureSlack) * 0.06 : 0;
  const revisitPenalty = visitCount[cell] * 0.35;
  return immediate * 3.2 + localGain * 1.15 - distance * 0.42 - revisitPenalty + slackBonus;
}

function reconstructPath(start: number, target: number): number[] {
  if (start === target) return [start];
  const rev: number[] = [];
  let cur = target;
  while (cur !== -1 && cur !== start) {
    rev.push(cur);
    cur = parent[cur];
  }
  rev.push(start);
  rev.reverse();
  return rev;
}

function bfsBestTarget(start: number, remaining: number): number[] {
  const limit = remaining <= 24 ? remaining : 24;
  const myStamp = bfsStamp++;
  let head = 0;
  let tail = 0;
  queue[tail++] = start;
  seen[start] = myStamp;
  dist[start] = 0;
  parent[start] = -1;

  let bestCell = start;
  let bestScore = evaluateCell(start, 0, remaining);

  while (head < tail) {
    const cur = queue[head++];
    const curDist = dist[cur];
    const score = evaluateCell(cur, curDist, remaining);
    if (
      score > bestScore + 1e-9 ||
      (Math.abs(score - bestScore) <= 1e-9 && staticDensity[cur] > staticDensity[bestCell])
    ) {
      bestScore = score;
      bestCell = cur;
    }
    if (curDist >= limit) continue;

    const x = cur % width;
    const y = (cur / width) | 0;
    if (x > 0) {
      const n = cur - 1;
      if (walkable[n] && seen[n] !== myStamp) {
        seen[n] = myStamp;
        dist[n] = curDist + 1;
        parent[n] = cur;
        queue[tail++] = n;
      }
    }
    if (x + 1 < width) {
      const n = cur + 1;
      if (walkable[n] && seen[n] !== myStamp) {
        seen[n] = myStamp;
        dist[n] = curDist + 1;
        parent[n] = cur;
        queue[tail++] = n;
      }
    }
    if (y > 0) {
      const n = cur - width;
      if (walkable[n] && seen[n] !== myStamp) {
        seen[n] = myStamp;
        dist[n] = curDist + 1;
        parent[n] = cur;
        queue[tail++] = n;
      }
    }
    if (y + 1 < height) {
      const n = cur + width;
      if (walkable[n] && seen[n] !== myStamp) {
        seen[n] = myStamp;
        dist[n] = curDist + 1;
        parent[n] = cur;
        queue[tail++] = n;
      }
    }
  }

  return reconstructPath(start, bestCell);
}

function chooseNeighbor(state: DroneState): number {
  const cur = state.pos;
  const x = state.x;
  const y = state.y;
  let best = cur;
  let bestScore = evaluateCell(cur, 0, state.remaining) - 0.5;

  const candidates: number[] = [cur];
  if (x > 0) candidates.push(cur - 1);
  if (x + 1 < width) candidates.push(cur + 1);
  if (y > 0) candidates.push(cur - width);
  if (y + 1 < height) candidates.push(cur + width);

  for (let i = 0; i < candidates.length; i++) {
    const n = candidates[i];
    if (!walkable[n]) continue;
    const score = evaluateCell(n, 1, state.remaining) - (n === cur ? 0.2 : 0);
    if (
      score > bestScore + 1e-9 ||
      (Math.abs(score - bestScore) <= 1e-9 && staticDensity[n] > staticDensity[best])
    ) {
      bestScore = score;
      best = n;
    }
  }
  return best;
}

function stepDrone(state: DroneState): void {
  if (!state.active || state.remaining <= 0) return;

  const currentComponent = component[state.pos];
  if (currentComponent === -1) {
    while (state.remaining > 0) {
      state.path.push([state.x, state.y]);
      state.remaining--;
    }
    return;
  }

  const route = bfsBestTarget(state.pos, state.remaining);
  const pathMoves = route.length - 1;
  if (pathMoves <= 0) {
    const next = chooseNeighbor(state);
    const nx = next % width;
    const ny = (next / width) | 0;
    state.x = nx;
    state.y = ny;
    state.pos = next;
    state.path.push([nx, ny]);
    state.remaining--;
    visitCount[next]++;
    if (values[next] > 0) covered[next] = 1;
    return;
  }

  let advance = pathMoves;
  if (state.remaining > 8 && advance > 6) advance = 6;
  if (state.remaining > 20 && advance > 8) advance = 8;
  if (advance > state.remaining) advance = state.remaining;

  for (let step = 1; step <= advance; step++) {
    const next = route[step];
    const nx = next % width;
    const ny = (next / width) | 0;
    state.x = nx;
    state.y = ny;
    state.pos = next;
    state.path.push([nx, ny]);
    state.remaining--;
    visitCount[next]++;
    if (values[next] > 0) covered[next] = 1;
    if (state.remaining <= 0) break;
  }
}

let progress = true;
while (progress) {
  progress = false;
  for (let i = 0; i < states.length; i++) {
    if (states[i].remaining > 0) {
      const before = states[i].remaining;
      stepDrone(states[i]);
      if (states[i].remaining < before) progress = true;
    }
  }
}

for (let i = 0; i < states.length; i++) {
  const state = states[i];
  while (state.remaining > 0) {
    state.path.push([state.x, state.y]);
    state.remaining--;
  }
}

const result = { paths: states.map((s) => s.path) };
process.stdout.write(JSON.stringify(result));
