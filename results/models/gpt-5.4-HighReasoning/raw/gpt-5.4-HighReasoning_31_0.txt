use std::cmp::Reverse;
use std::collections::{HashMap, HashSet};
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Node {
    sum: i64,
    parent: u32,
    picked_index: i32,
}

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, pos: 0 })
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }
        let mut value = 0i64;
        let mut found_digit = false;
        while self.pos < self.data.len() {
            let b = self.data[self.pos];
            if !b.is_ascii_digit() {
                break;
            }
            found_digit = true;
            value = value.saturating_mul(10).saturating_add((b - b'0') as i64);
            self.pos += 1;
        }
        if found_digit {
            Some(value.saturating_mul(sign))
        } else {
            None
        }
    }
}

fn gcd_u64(mut a: u64, mut b: u64) -> u64 {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn reconstruct(nodes: &[Node], mut node_idx: usize) -> Vec<usize> {
    let mut out = Vec::new();
    while node_idx != 0 {
        let node = nodes[node_idx];
        if node.picked_index >= 0 {
            out.push(node.picked_index as usize);
        }
        node_idx = node.parent as usize;
    }
    out.reverse();
    out
}

fn normalize_solution(values: &[i64], target: i64, mut indices: Vec<usize>) -> Option<Vec<usize>> {
    let mut seen = HashSet::with_capacity(indices.len());
    let mut sum = 0i64;
    for &idx in &indices {
        if idx >= values.len() || !seen.insert(idx) {
            return None;
        }
        sum = sum.saturating_add(values[idx]);
    }
    if sum != target {
        return None;
    }
    indices.sort_unstable();
    Some(indices)
}

fn meet_in_middle(values: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    let left_n = n / 2;
    let right_n = n - left_n;
    if left_n > 20 || right_n > 20 {
        return None;
    }

    let mut right_sums = Vec::with_capacity(1usize << right_n);
    for mask in 0u64..(1u64 << right_n) {
        let mut sum = 0i64;
        for bit in 0..right_n {
            if ((mask >> bit) & 1) != 0 {
                sum = sum.saturating_add(values[left_n + bit].0);
            }
        }
        right_sums.push((sum, mask));
    }
    right_sums.sort_unstable_by_key(|&(sum, _)| sum);

    for left_mask in 0u64..(1u64 << left_n) {
        let mut left_sum = 0i64;
        for bit in 0..left_n {
            if ((left_mask >> bit) & 1) != 0 {
                left_sum = left_sum.saturating_add(values[bit].0);
            }
        }
        let want = target.saturating_sub(left_sum);
        if let Ok(pos) = right_sums.binary_search_by_key(&want, |&(sum, _)| sum) {
            let right_mask = right_sums[pos].1;
            let mut ans = Vec::new();
            for bit in 0..left_n {
                if ((left_mask >> bit) & 1) != 0 {
                    ans.push(values[bit].1);
                }
            }
            for bit in 0..right_n {
                if ((right_mask >> bit) & 1) != 0 {
                    ans.push(values[left_n + bit].1);
                }
            }
            return Some(ans);
        }
    }
    None
}

fn exact_positive_dp(values: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let t = usize::try_from(target).ok()?;
    let mut reachable = vec![false; t + 1];
    let mut parent_sum = vec![usize::MAX; t + 1];
    let mut parent_idx = vec![usize::MAX; t + 1];
    reachable[0] = true;

    for &(value, original_idx) in values {
        if value > target {
            continue;
        }
        let v = value as usize;
        for s in (v..=t).rev() {
            if !reachable[s] && reachable[s - v] {
                reachable[s] = true;
                parent_sum[s] = s - v;
                parent_idx[s] = original_idx;
            }
        }
        if reachable[t] {
            break;
        }
    }

    if !reachable[t] {
        return None;
    }

    let mut ans = Vec::new();
    let mut cur = t;
    while cur != 0 {
        ans.push(parent_idx[cur]);
        cur = parent_sum[cur];
    }
    Some(ans)
}

fn positive_branch_and_bound(values: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    fn dfs(
        i: usize,
        remaining: u64,
        values: &[(u64, usize)],
        suffix: &[u64],
        chosen: &mut Vec<usize>,
    ) -> bool {
        if remaining == 0 {
            return true;
        }
        if i == values.len() || remaining > suffix[i] {
            return false;
        }

        let (value, idx) = values[i];
        if value <= remaining {
            chosen.push(idx);
            if dfs(i + 1, remaining - value, values, suffix, chosen) {
                return true;
            }
            chosen.pop();
        }
        dfs(i + 1, remaining, values, suffix, chosen)
    }

    let mut sorted = values.to_vec();
    sorted.sort_unstable_by_key(|&(v, _)| Reverse(v));
    let mut suffix = vec![0u64; sorted.len() + 1];
    for i in (0..sorted.len()).rev() {
        suffix[i] = suffix[i + 1].saturating_add(sorted[i].0);
    }
    let mut chosen = Vec::new();
    if dfs(0, target, &sorted, &suffix, &mut chosen) {
        Some(chosen)
    } else {
        None
    }
}

fn sparse_exact(values: &[(i64, usize)], target: i64, max_states: usize) -> Option<Vec<usize>> {
    let mut nodes = Vec::with_capacity(max_states.min(1024));
    nodes.push(Node {
        sum: 0,
        parent: u32::MAX,
        picked_index: -1,
    });

    let mut frontier: HashMap<i64, usize> = HashMap::new();
    frontier.insert(0, 0);

    for &(value, original_idx) in values {
        let mut additions = Vec::new();
        let mut local_seen = HashSet::new();
        for (&sum, &node_idx) in &frontier {
            let next_sum = sum.saturating_add(value);
            if frontier.contains_key(&next_sum) || !local_seen.insert(next_sum) {
                continue;
            }
            additions.push((next_sum, node_idx, original_idx));
        }

        if frontier.len().saturating_add(additions.len()) > max_states {
            return None;
        }

        for (sum, parent_idx, original_idx) in additions {
            let new_idx = nodes.len();
            nodes.push(Node {
                sum,
                parent: parent_idx as u32,
                picked_index: original_idx as i32,
            });
            frontier.insert(sum, new_idx);
        }

        if let Some(&node_idx) = frontier.get(&target) {
            return Some(reconstruct(&nodes, node_idx));
        }
    }
    None
}

fn greedy_positive(values: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let mut sorted = values.to_vec();
    sorted.sort_unstable_by_key(|&(v, _)| Reverse(v));
    let mut remaining = target;
    let mut ans = Vec::new();
    for &(value, idx) in &sorted {
        if value <= remaining {
            remaining -= value;
            ans.push(idx);
            if remaining == 0 {
                return Some(ans);
            }
        }
    }
    None
}

fn select_positive_candidates(values: &[(u64, usize)], target: u64, limit: usize) -> Vec<(i64, usize)> {
    if values.len() <= limit {
        return values.iter().map(|&(v, idx)| (v as i64, idx)).collect();
    }

    let small_take = limit / 4;
    let large_take = limit / 4;
    let close_take = limit.saturating_sub(small_take + large_take);

    let mut picked = HashSet::with_capacity(limit * 2);
    let mut out = Vec::with_capacity(limit);

    let mut by_value = values.to_vec();
    by_value.sort_unstable_by_key(|&(v, _)| v);

    for &(value, idx) in by_value.iter().take(small_take) {
        if picked.insert(idx) {
            out.push((value as i64, idx));
        }
    }
    for &(value, idx) in by_value.iter().rev().take(large_take) {
        if picked.insert(idx) {
            out.push((value as i64, idx));
        }
    }

    let mut by_closeness = values.to_vec();
    by_closeness.sort_unstable_by_key(|&(v, _)| target.abs_diff(v));
    for &(value, idx) in by_closeness.iter().take(close_take) {
        if picked.insert(idx) {
            out.push((value as i64, idx));
        }
    }
    out
}

fn select_mixed_candidates(values: &[(i64, usize)], target: i64, limit: usize) -> Vec<(i64, usize)> {
    if values.len() <= limit {
        return values.to_vec();
    }

    let close_take = limit / 2;
    let abs_take = limit.saturating_sub(close_take);
    let mut picked = HashSet::with_capacity(limit * 2);
    let mut out = Vec::with_capacity(limit);

    let mut by_closeness = values.to_vec();
    by_closeness.sort_unstable_by_key(|&(v, _)| target.abs_diff(v));
    for &(value, idx) in by_closeness.iter().take(close_take) {
        if picked.insert(idx) {
            out.push((value, idx));
        }
    }

    let mut by_abs = values.to_vec();
    by_abs.sort_unstable_by_key(|&(v, _)| Reverse(v.unsigned_abs()));
    for &(value, idx) in by_abs.iter().take(abs_take) {
        if picked.insert(idx) {
            out.push((value, idx));
        }
    }
    out
}

fn beam_search(values: &[(i64, usize)], target: i64, width: usize, mode: usize) -> Option<Vec<usize>> {
    let mut ordered = values.to_vec();
    match mode {
        0 => ordered.sort_unstable_by_key(|&(v, _)| Reverse(v.unsigned_abs())),
        1 => ordered.sort_unstable_by_key(|&(v, _)| target.abs_diff(v)),
        _ => ordered.sort_unstable_by_key(|&(v, _)| Reverse((target - v).unsigned_abs())),
    }

    let mut nodes = Vec::with_capacity(width.saturating_mul(4).max(16));
    nodes.push(Node {
        sum: 0,
        parent: u32::MAX,
        picked_index: -1,
    });
    let mut beam = vec![0usize];

    for &(value, original_idx) in &ordered {
        let mut best_by_sum: HashMap<i64, usize> = HashMap::with_capacity(width.saturating_mul(2));

        for &node_idx in &beam {
            let keep_sum = nodes[node_idx].sum;
            best_by_sum.entry(keep_sum).or_insert(node_idx);

            let next_sum = keep_sum.saturating_add(value);
            best_by_sum.entry(next_sum).or_insert_with(|| {
                let new_idx = nodes.len();
                nodes.push(Node {
                    sum: next_sum,
                    parent: node_idx as u32,
                    picked_index: original_idx as i32,
                });
                new_idx
            });
        }

        if let Some(&node_idx) = best_by_sum.get(&target) {
            return Some(reconstruct(&nodes, node_idx));
        }

        let mut next_beam: Vec<usize> = best_by_sum.into_values().collect();
        next_beam.sort_unstable_by_key(|&idx| {
            let node = nodes[idx];
            (target.abs_diff(node.sum), node.sum.unsigned_abs())
        });
        if next_beam.len() > width {
            next_beam.truncate(width);
        }
        beam = next_beam;
    }
    None
}

fn solve(values: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    for (idx, &value) in values.iter().enumerate() {
        if value == target {
            return Some(vec![idx]);
        }
    }

    let mut gcd = 0u64;
    let mut min_sum = 0i64;
    let mut max_sum = 0i64;
    let mut all_non_negative = true;
    let mut all_non_positive = true;

    for &value in values {
        gcd = gcd_u64(gcd, value.unsigned_abs());
        if value > 0 {
            max_sum = max_sum.saturating_add(value);
            all_non_positive = false;
        } else if value < 0 {
            min_sum = min_sum.saturating_add(value);
            all_non_negative = false;
        }
    }

    if gcd != 0 && target % (gcd as i64) != 0 {
        return None;
    }
    if target < min_sum || target > max_sum {
        return None;
    }

    let indexed: Vec<(i64, usize)> = values.iter().copied().enumerate().map(|(i, v)| (v, i)).collect();

    if values.len() <= 40 {
        if let Some(ans) = meet_in_middle(&indexed, target) {
            return normalize_solution(values, target, ans);
        }
    }

    if all_non_negative {
        if target < 0 {
            return None;
        }
        let positives: Vec<(u64, usize)> = values
            .iter()
            .copied()
            .enumerate()
            .filter_map(|(idx, value)| {
                if value > 0 && value <= target {
                    Some((value as u64, idx))
                } else {
                    None
                }
            })
            .collect();

        let total: u128 = positives.iter().map(|&(v, _)| v as u128).sum();
        if total < target as u128 {
            return None;
        }

        if target <= 2_000_000 && positives.len() <= 200_000 {
            if let Some(ans) = exact_positive_dp(&positives, target as u64) {
                return normalize_solution(values, target, ans);
            }
        }

        if positives.len() <= 72 {
            if let Some(ans) = positive_branch_and_bound(&positives, target as u64) {
                return normalize_solution(values, target, ans);
            }
        }

        if positives.len() <= 120 {
            let exact_input: Vec<(i64, usize)> = positives.iter().map(|&(v, idx)| (v as i64, idx)).collect();
            if let Some(ans) = sparse_exact(&exact_input, target, 1_500_000) {
                return normalize_solution(values, target, ans);
            }
        }

        if let Some(ans) = greedy_positive(&positives, target as u64) {
            return normalize_solution(values, target, ans);
        }

        let candidates = select_positive_candidates(&positives, target as u64, 20_000);
        let width = if candidates.len() <= 4_000 { 4_096 } else { 2_048 };
        for mode in 0..3 {
            if let Some(ans) = beam_search(&candidates, target, width, mode) {
                return normalize_solution(values, target, ans);
            }
        }
        return None;
    }

    if all_non_positive {
        let flipped: Vec<i64> = values.iter().map(|&v| v.saturating_neg()).collect();
        let flipped_target = target.saturating_neg();
        return solve(&flipped, flipped_target);
    }

    if values.len() <= 64 {
        if let Some(ans) = sparse_exact(&indexed, target, 2_000_000) {
            return normalize_solution(values, target, ans);
        }
    } else if values.len() <= 120 {
        if let Some(ans) = sparse_exact(&indexed, target, 1_000_000) {
            return normalize_solution(values, target, ans);
        }
    }

    let candidates = select_mixed_candidates(&indexed, target, 25_000);
    let width = if candidates.len() <= 4_000 { 4_096 } else { 2_048 };
    for mode in 0..3 {
        if let Some(ans) = beam_search(&candidates, target, width, mode) {
            return normalize_solution(values, target, ans);
        }
    }
    None
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let declared_n = input.next_i64().unwrap_or(0).max(0) as usize;
    let target = input.next_i64().unwrap_or(0);

    let mut values = Vec::with_capacity(declared_n);
    while values.len() < declared_n {
        match input.next_i64() {
            Some(v) => values.push(v),
            None => break,
        }
    }

    let mut output = String::new();
    match solve(&values, target) {
        Some(indices) => {
            output.push_str("YES\n");
            if !indices.is_empty() {
                for (i, idx) in indices.iter().enumerate() {
                    if i > 0 {
                        output.push(' ');
                    }
                    output.push_str(&idx.to_string());
                }
            }
            output.push('\n');
        }
        None => {
            output.push_str("NO\n");
        }
    }

    print!("{output}");
    Ok(())
}
