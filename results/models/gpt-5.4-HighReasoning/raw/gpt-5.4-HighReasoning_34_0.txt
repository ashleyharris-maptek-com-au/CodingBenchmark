import math
import sys


def euclidean(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_cost(route, points, depot):
    if not route:
        return 0.0
    total = euclidean(depot, points[route[0]])
    for i in range(len(route) - 1):
        total += euclidean(points[route[i]], points[route[i + 1]])
    total += euclidean(points[route[-1]], depot)
    return total


def nearest_neighbor_route(nodes, points, depot):
    if len(nodes) <= 2:
        return nodes[:]

    unused = set(nodes)
    start = min(nodes, key=lambda idx: euclidean(depot, points[idx]))
    route = [start]
    unused.remove(start)
    current = start

    while unused:
        nxt = min(unused, key=lambda idx: euclidean(points[current], points[idx]))
        route.append(nxt)
        unused.remove(nxt)
        current = nxt

    return route


def two_opt(route, points, depot):
    if len(route) < 4:
        return route

    improved = True
    while improved:
        improved = False
        for i in range(len(route) - 2):
            for j in range(i + 2, len(route)):
                a = depot if i == 0 else points[route[i - 1]]
                b = points[route[i]]
                c = points[route[j]]
                d = depot if j == len(route) - 1 else points[route[j + 1]]

                current = euclidean(a, b) + euclidean(c, d)
                candidate = euclidean(a, c) + euclidean(b, d)
                if candidate + 1e-12 < current:
                    route[i : j + 1] = reversed(route[i : j + 1])
                    improved = True
        # Keep looping until a full pass produces no improvement.
    return route


def improve_route(nodes, points, depot):
    route = nearest_neighbor_route(nodes, points, depot)
    return two_opt(route, points, depot)


def build_segment_costs(order, demands, points, depot, capacity):
    n = len(order)
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + demands[order[i]]

    seg_cost = [[math.inf] * (n + 1) for _ in range(n)]
    for i in range(n):
        total_demand = 0
        path_cost = 0.0
        prev_point = None
        for j in range(i, n):
            node = order[j]
            total_demand += demands[node]
            if total_demand > capacity:
                break
            if prev_point is not None:
                path_cost += euclidean(prev_point, points[node])
            prev_point = points[node]
            seg_cost[i][j + 1] = (
                euclidean(depot, points[order[i]])
                + path_cost
                + euclidean(points[node], depot)
            )
    return prefix, seg_cost


def partition_by_sweep(order, demands, points, depot, capacity, max_routes):
    n = len(order)
    prefix, seg_cost = build_segment_costs(order, demands, points, depot, capacity)

    dp = [[math.inf] * (n + 1) for _ in range(max_routes + 1)]
    parent = [[-1] * (n + 1) for _ in range(max_routes + 1)]
    dp[0][0] = 0.0

    for r in range(1, max_routes + 1):
        for end in range(1, n + 1):
            start = end - 1
            while start >= 0 and prefix[end] - prefix[start] <= capacity:
                cost = seg_cost[start][end]
                if cost < math.inf and dp[r - 1][start] + cost < dp[r][end]:
                    dp[r][end] = dp[r - 1][start] + cost
                    parent[r][end] = start
                start -= 1

    best_r = min(range(1, max_routes + 1), key=lambda r: dp[r][n])
    if not math.isfinite(dp[best_r][n]):
        return None

    segments = []
    end = n
    r = best_r
    while end > 0:
        start = parent[r][end]
        if start < 0:
            return None
        segments.append(order[start:end])
        end = start
        r -= 1
    segments.reverse()
    return segments


def candidate_orders(points, depot):
    angles = []
    for idx, (x, y) in enumerate(points):
        angle = math.atan2(y - depot[1], x - depot[0])
        angles.append((angle, idx))
    angles.sort()

    n = len(angles)
    if n == 0:
        return [[]]

    starts = []
    if n <= 32:
        starts = list(range(n))
    else:
        samples = min(16, n)
        starts = sorted({(i * n) // samples for i in range(samples)})

    orders = []
    for start in starts:
        rotated = angles[start:] + angles[:start]
        orders.append([idx for _, idx in rotated])
    return orders


def first_fit_fallback(points, demands, depot, capacity, max_routes):
    customers = list(range(len(points)))
    customers.sort(
        key=lambda idx: (
            -demands[idx],
            -euclidean(depot, points[idx]),
        )
    )

    routes = [[] for _ in range(max_routes)]
    loads = [0] * max_routes

    for customer in customers:
        chosen = -1
        best_slack = math.inf
        for r in range(max_routes):
            if loads[r] + demands[customer] <= capacity:
                slack = capacity - (loads[r] + demands[customer])
                if slack < best_slack:
                    best_slack = slack
                    chosen = r
        if chosen < 0:
            return None
        routes[chosen].append(customer)
        loads[chosen] += demands[customer]

    routes = [route for route in routes if route]
    return [improve_route(route, points, depot) for route in routes]


def solve(points, demands, depot, k, capacity):
    n = len(points)
    if n == 0:
        return []

    best_routes = None
    best_cost = math.inf

    for order in candidate_orders(points, depot):
        segments = partition_by_sweep(order, demands, points, depot, capacity, k)
        if segments is None:
            continue

        routes = [improve_route(segment, points, depot) for segment in segments]
        total = sum(route_cost(route, points, depot) for route in routes)
        if total < best_cost:
            best_cost = total
            best_routes = routes

    if best_routes is None:
        best_routes = first_fit_fallback(points, demands, depot, capacity, k)
        if best_routes is None:
            return None

    return best_routes


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))
    depot = (float(next(it)), float(next(it)))

    points = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        demand = int(next(it))
        points.append((x, y))
        demands.append(demand)

    if any(d > capacity for d in demands) or sum(demands) > k * capacity:
        print("IMPOSSIBLE")
        return

    routes = solve(points, demands, depot, k, capacity)
    if routes is None:
        print("IMPOSSIBLE")
        return

    total = sum(route_cost(route, points, depot) for route in routes)

    print(f"{total:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))


if __name__ == "__main__":
    main()
