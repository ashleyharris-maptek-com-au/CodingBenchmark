#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <stdexcept>
#include <tuple>
#include <utility>
#include <vector>

struct Rect {
    int id = -1;
    int w = 0;
    int h = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
};

struct Candidate {
    bool ok = false;
    int freeIndex = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    int score1 = std::numeric_limits<int>::max();
    int score2 = std::numeric_limits<int>::max();
};

class MaxRectsBin {
public:
    MaxRectsBin(int width, int height) : boardW(width), boardH(height) {
        freeRects.push_back({0, 0, boardW, boardH});
    }

    Candidate findBest(int rw, int rh) const {
        Candidate best;
        for (int i = 0; i < static_cast<int>(freeRects.size()); ++i) {
            const FreeRect& fr = freeRects[i];
            tryCandidate(i, fr, rw, rh, 0, best);
            if (rw != rh) {
                tryCandidate(i, fr, rh, rw, 1, best);
            }
        }
        return best;
    }

    Placement place(const Candidate& c, int boardIndex) {
        Placement p{boardIndex, c.x, c.y, c.rotated};
        FreeRect used{c.x, c.y, c.w, c.h};
        splitFreeRects(used);
        pruneFreeRects();
        return p;
    }

private:
    int boardW = 0;
    int boardH = 0;
    std::vector<FreeRect> freeRects;

    static bool contains(const FreeRect& a, const FreeRect& b) {
        return b.x >= a.x && b.y >= a.y &&
               b.x + b.w <= a.x + a.w &&
               b.y + b.h <= a.y + a.h;
    }

    static bool overlaps(const FreeRect& a, const FreeRect& b) {
        return a.x < b.x + b.w && a.x + a.w > b.x &&
               a.y < b.y + b.h && a.y + a.h > b.y;
    }

    static void considerBest(Candidate& best, const Candidate& cur) {
        if (!cur.ok) {
            return;
        }
        if (!best.ok || cur.score1 < best.score1 ||
            (cur.score1 == best.score1 && cur.score2 < best.score2)) {
            best = cur;
        }
    }

    static void tryCandidate(int index, const FreeRect& fr, int rw, int rh, int rotated, Candidate& best) {
        if (rw > fr.w || rh > fr.h) {
            return;
        }
        const int leftoverH = fr.h - rh;
        const int leftoverW = fr.w - rw;
        const int shortSide = std::min(leftoverW, leftoverH);
        const int longSide = std::max(leftoverW, leftoverH);
        Candidate cur;
        cur.ok = true;
        cur.freeIndex = index;
        cur.x = fr.x;
        cur.y = fr.y;
        cur.w = rw;
        cur.h = rh;
        cur.rotated = rotated;
        cur.score1 = shortSide;
        cur.score2 = longSide;
        considerBest(best, cur);
    }

    void splitFreeRects(const FreeRect& used) {
        std::vector<FreeRect> next;
        next.reserve(freeRects.size() + 4);
        for (const FreeRect& fr : freeRects) {
            if (!overlaps(fr, used)) {
                next.push_back(fr);
                continue;
            }
            if (used.x > fr.x) {
                next.push_back({fr.x, fr.y, used.x - fr.x, fr.h});
            }
            if (used.x + used.w < fr.x + fr.w) {
                next.push_back({used.x + used.w, fr.y, fr.x + fr.w - (used.x + used.w), fr.h});
            }
            if (used.y > fr.y) {
                next.push_back({fr.x, fr.y, fr.w, used.y - fr.y});
            }
            if (used.y + used.h < fr.y + fr.h) {
                next.push_back({fr.x, used.y + used.h, fr.w, fr.y + fr.h - (used.y + used.h)});
            }
        }
        freeRects.swap(next);
    }

    void pruneFreeRects() {
        std::vector<FreeRect> pruned;
        pruned.reserve(freeRects.size());
        for (int i = 0; i < static_cast<int>(freeRects.size()); ++i) {
            if (freeRects[i].w <= 0 || freeRects[i].h <= 0) {
                continue;
            }
            bool redundant = false;
            for (int j = 0; j < static_cast<int>(freeRects.size()); ++j) {
                if (i == j) {
                    continue;
                }
                if (contains(freeRects[j], freeRects[i])) {
                    redundant = true;
                    break;
                }
            }
            if (!redundant) {
                pruned.push_back(freeRects[i]);
            }
        }
        freeRects.swap(pruned);
    }
};

struct Shelf {
    int y = 0;
    int height = 0;
    int usedWidth = 0;
};

struct ShelfCandidate {
    bool ok = false;
    int shelfIndex = -1;
    bool createNewShelf = false;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    int score1 = std::numeric_limits<int>::max();
    int score2 = std::numeric_limits<int>::max();
};

struct ShelfBoard {
    int usedHeight = 0;
    std::vector<Shelf> shelves;
};

static bool betterCandidate(const Candidate& a, const Candidate& b) {
    if (!a.ok) {
        return false;
    }
    if (!b.ok) {
        return true;
    }
    return std::tie(a.score1, a.score2) < std::tie(b.score1, b.score2);
}

static bool betterShelf(const ShelfCandidate& a, const ShelfCandidate& b) {
    if (!a.ok) {
        return false;
    }
    if (!b.ok) {
        return true;
    }
    return std::tie(a.score1, a.score2) < std::tie(b.score1, b.score2);
}

static std::vector<int> buildOrder(const std::vector<Rect>& rects) {
    std::vector<int> order(rects.size());
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        const Rect& ra = rects[a];
        const Rect& rb = rects[b];
        const long long areaA = 1LL * ra.w * ra.h;
        const long long areaB = 1LL * rb.w * rb.h;
        if (areaA != areaB) {
            return areaA > areaB;
        }
        const int maxA = std::max(ra.w, ra.h);
        const int maxB = std::max(rb.w, rb.h);
        if (maxA != maxB) {
            return maxA > maxB;
        }
        const int minA = std::min(ra.w, ra.h);
        const int minB = std::min(rb.w, rb.h);
        if (minA != minB) {
            return minA > minB;
        }
        return ra.id < rb.id;
    });
    return order;
}

static std::vector<Placement> solveSmall(const std::vector<Rect>& rects, int boardW, int boardH) {
    const std::vector<int> order = buildOrder(rects);
    std::vector<Placement> out(rects.size());
    std::vector<MaxRectsBin> bins;
    bins.reserve(rects.size());

    for (int idx : order) {
        const Rect& r = rects[idx];
        Candidate best;
        int bestBin = -1;

        for (int b = 0; b < static_cast<int>(bins.size()); ++b) {
            Candidate c = bins[b].findBest(r.w, r.h);
            if (betterCandidate(c, best)) {
                best = c;
                bestBin = b;
            }
        }

        if (!best.ok) {
            bins.emplace_back(boardW, boardH);
            bestBin = static_cast<int>(bins.size()) - 1;
            best = bins.back().findBest(r.w, r.h);
            if (!best.ok) {
                throw std::runtime_error("Rectangle does not fit in board.");
            }
        }

        out[r.id] = bins[bestBin].place(best, bestBin);
    }
    return out;
}

static void tryShelfPlacement(const ShelfBoard& board, int boardW, int boardH, int rw, int rh, int rotated, ShelfCandidate& best) {
    for (int i = 0; i < static_cast<int>(board.shelves.size()); ++i) {
        const Shelf& s = board.shelves[i];
        if (rh > s.height || s.usedWidth + rw > boardW) {
            continue;
        }
        ShelfCandidate cur;
        cur.ok = true;
        cur.shelfIndex = i;
        cur.createNewShelf = false;
        cur.x = s.usedWidth;
        cur.y = s.y;
        cur.w = rw;
        cur.h = rh;
        cur.rotated = rotated;
        cur.score1 = boardW - (s.usedWidth + rw);
        cur.score2 = s.height - rh;
        if (betterShelf(cur, best)) {
            best = cur;
        }
    }

    if (board.usedHeight + rh <= boardH && rw <= boardW) {
        ShelfCandidate cur;
        cur.ok = true;
        cur.shelfIndex = static_cast<int>(board.shelves.size());
        cur.createNewShelf = true;
        cur.x = 0;
        cur.y = board.usedHeight;
        cur.w = rw;
        cur.h = rh;
        cur.rotated = rotated;
        cur.score1 = boardH - (board.usedHeight + rh);
        cur.score2 = boardW - rw;
        if (betterShelf(cur, best)) {
            best = cur;
        }
    }
}

static ShelfCandidate findBestShelf(const ShelfBoard& board, int boardW, int boardH, int rw, int rh) {
    ShelfCandidate best;
    tryShelfPlacement(board, boardW, boardH, rw, rh, 0, best);
    if (rw != rh) {
        tryShelfPlacement(board, boardW, boardH, rh, rw, 1, best);
    }
    return best;
}

static Placement applyShelf(ShelfBoard& board, const ShelfCandidate& c, int boardIndex) {
    if (c.createNewShelf) {
        board.shelves.push_back({c.y, c.h, c.w});
        board.usedHeight += c.h;
    } else {
        board.shelves[c.shelfIndex].usedWidth += c.w;
    }
    return Placement{boardIndex, c.x, c.y, c.rotated};
}

static std::vector<Placement> solveLarge(const std::vector<Rect>& rects, int boardW, int boardH) {
    const std::vector<int> order = buildOrder(rects);
    std::vector<Placement> out(rects.size());
    std::vector<ShelfBoard> boards;
    boards.reserve(rects.size() / 4 + 1);

    constexpr int scanLimit = 64;

    for (int idx : order) {
        const Rect& r = rects[idx];
        ShelfCandidate best;
        int bestBoard = -1;

        const int start = std::max(0, static_cast<int>(boards.size()) - scanLimit);
        for (int b = start; b < static_cast<int>(boards.size()); ++b) {
            ShelfCandidate c = findBestShelf(boards[b], boardW, boardH, r.w, r.h);
            if (betterShelf(c, best)) {
                best = c;
                bestBoard = b;
            }
        }

        if (!best.ok) {
            boards.push_back(ShelfBoard{});
            bestBoard = static_cast<int>(boards.size()) - 1;
            best = findBestShelf(boards.back(), boardW, boardH, r.w, r.h);
            if (!best.ok) {
                throw std::runtime_error("Rectangle does not fit in board.");
            }
        }

        out[r.id] = applyShelf(boards[bestBoard], best, bestBoard);
    }

    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    int boardW = 0;
    int boardH = 0;
    if (!(std::cin >> n >> boardW >> boardH)) {
        return 0;
    }

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
        const bool fits = (rects[i].w <= boardW && rects[i].h <= boardH) ||
                          (rects[i].h <= boardW && rects[i].w <= boardH);
        if (!fits) {
            std::cout << 0 << '\n';
            return 0;
        }
    }

    std::vector<Placement> ans;
    if (n <= 2000) {
        ans = solveSmall(rects, boardW, boardH);
    } else {
        ans = solveLarge(rects, boardW, boardH);
    }

    int numBoards = 0;
    for (const Placement& p : ans) {
        numBoards = std::max(numBoards, p.board + 1);
    }

    std::cout << numBoards << '\n';
    for (const Placement& p : ans) {
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }
    return 0;
}
