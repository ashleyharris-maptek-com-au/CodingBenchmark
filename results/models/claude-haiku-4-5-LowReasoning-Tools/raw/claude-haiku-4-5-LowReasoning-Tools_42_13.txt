OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_0_008 = OpConstant %float 0.008

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input

%arr_v3float_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3float = OpTypePointer Input %arr_v3float_in
%worldPos_in = OpVariable %ptr_in_arr_v3float Input
%normal_in = OpVariable %ptr_in_arr_v3float Input
%color_in = OpVariable %ptr_in_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%func_void = OpTypeFunction %void
%func_void_int_int = OpTypeFunction %void %int %int

%emit_edge_quad = OpFunction %void None %func_void_int_int
%idx_a = OpFunctionParameter %int
%idx_b = OpFunctionParameter %int

%emit_label = OpLabel

%ptr_a = OpAccessChain %ptr_in_v4float %gl_in %idx_a %int_0
%pos_a = OpLoad %v4float %ptr_a
%ptr_b = OpAccessChain %ptr_in_v4float %gl_in %idx_b %int_0
%pos_b = OpLoad %v4float %ptr_b

%pos_a_v3 = OpVectorShuffle %v3float %pos_a %pos_a 0 1 2
%pos_b_v3 = OpVectorShuffle %v3float %pos_b %pos_b 0 1 2

%ptr_ca = OpAccessChain %ptr_in_v3float %color_in %idx_a
%color_a = OpLoad %v3float %ptr_ca
%ptr_cb = OpAccessChain %ptr_in_v3float %color_in %idx_b
%color_b = OpLoad %v3float %ptr_cb

%color_sum = OpFAdd %v3float %color_a %color_b
%color_avg = OpVectorTimesScalar %v3float %color_sum %float_0_5

%edge_dir = OpFSub %v3float %pos_b_v3 %pos_a_v3

%edge_x = OpCompositeExtract %float %edge_dir 0
%edge_y = OpCompositeExtract %float %edge_dir 1

%perp_x = OpFNegate %float %edge_y
%perp_y = OpCopy %float %edge_x
%perp_z = OpConstant %float 0.0

%perp = OpCompositeConstruct %v3float %perp_x %perp_y %perp_z
%perp_scaled = OpVectorTimesScalar %v3float %perp %float_0_008

%v0_pos = OpFSub %v3float %pos_a_v3 %perp_scaled
%v0_x = OpCompositeExtract %float %v0_pos 0
%v0_y = OpCompositeExtract %float %v0_pos 1
%v0_z = OpCompositeExtract %float %v0_pos 2
%v0_pos4 = OpCompositeConstruct %v4float %v0_x %v0_y %v0_z %float_1
%ptr_v0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %ptr_v0 %v0_pos4
OpStore %outColor %color_avg
OpEmitVertex

%v1_pos = OpFAdd %v3float %pos_a_v3 %perp_scaled
%v1_x = OpCompositeExtract %float %v1_pos 0
%v1_y = OpCompositeExtract %float %v1_pos 1
%v1_z = OpCompositeExtract %float %v1_pos 2
%v1_pos4 = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1
OpStore %ptr_v0 %v1_pos4
OpStore %outColor %color_avg
OpEmitVertex

%v2_pos = OpFAdd %v3float %pos_b_v3 %perp_scaled
%v2_x = OpCompositeExtract %float %v2_pos 0
%v2_y = OpCompositeExtract %float %v2_pos 1
%v2_z = OpCompositeExtract %float %v2_pos 2
%v2_pos4 = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1
OpStore %ptr_v0 %v2_pos4
OpStore %outColor %color_avg
OpEmitVertex

%v3_pos = OpFSub %v3float %pos_b_v3 %perp_scaled
%v3_x = OpCompositeExtract %float %v3_pos 0
%v3_y = OpCompositeExtract %float %v3_pos 1
%v3_z = OpCompositeExtract %float %v3_pos 2
%v3_pos4 = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1
OpStore %ptr_v0 %v3_pos4
OpStore %outColor %color_avg
OpEmitVertex

OpEndPrimitive
OpReturn
%emit_end = OpFunctionEnd

%main = OpFunction %void None %func_void
%main_label = OpLabel

%call_edge1 = OpFunctionCall %void %emit_edge_quad %int_0 %int_2
%call_edge2 = OpFunctionCall %void %emit_edge_quad %int_2 %int_4
%call_edge3 = OpFunctionCall %void %emit_edge_quad %int_4 %int_0

OpReturn
%main_end = OpFunctionEnd
