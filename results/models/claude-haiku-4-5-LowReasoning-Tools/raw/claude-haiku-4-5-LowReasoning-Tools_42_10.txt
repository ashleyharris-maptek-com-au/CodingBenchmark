OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Flat

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_4_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_pv_in = OpTypePointer Input %arr_4_pv_in
%gl_in = OpVariable %ptr_input_arr_pv_in Input

%arr_4_v3 = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3 = OpTypePointer Input %arr_4_v3
%color_in = OpVariable %ptr_input_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_gl_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_gl_pv Output

%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output

%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_input_v3 = OpTypePointer Input %v3float

%main_func_type = OpTypeFunction %void
%main = OpFunction %void None %main_func_type
%main_entry = OpLabel

; Load 4 input positions
%pos0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p0_vec = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p1_vec = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%p2_vec = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_3 %int_0
%p3_vec = OpLoad %v4float %pos3_ptr

; Load colors
%col1_ptr = OpAccessChain %ptr_input_v3 %color_in %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_input_v3 %color_in %int_2
%col2 = OpLoad %v3float %col2_ptr

; Extract XYZ from positions
%p0_xyz = OpVectorShuffle %v3float %p0_vec %p0_vec 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1_vec %p1_vec 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2_vec %p2_vec 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3_vec %p3_vec 0 1 2

; Compute tangent direction: P2 - P1
%tangent = OpFSub %v3float %p2_xyz %p1_xyz
%tangent_norm = OpNormalize %v3float %tangent

; Compute perpendicular: cross(tangent, z-axis)
%z_axis = OpCompositeConstruct %v3float %float_0 %float_0 %float_1
%perp_dir = OpCross %v3float %tangent_norm %z_axis
%perp_norm = OpNormalize %v3float %perp_dir

; Scale perpendicular by thickness
%perp_thick = OpVectorTimesScalar %v3float %perp_norm %float_0_02

; Emit vertex for t=0.0
%pos_t0 = OpVectorTimesScalar %v3float %p1_xyz %float_1
%v_t0_a = OpFAdd %v3float %pos_t0 %perp_thick
%v_t0_a_4d = OpCompositeConstruct %v4float %v_t0_a %float_1
%gl_pos_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %gl_pos_ptr %v_t0_a_4d
%col_t0 = OpVectorTimesScalar %v3float %col1 %float_1
OpStore %outColor %col_t0
OpEmitVertex

%v_t0_b = OpFSub %v3float %pos_t0 %perp_thick
%v_t0_b_4d = OpCompositeConstruct %v4float %v_t0_b %float_1
OpStore %gl_pos_ptr %v_t0_b_4d
OpStore %outColor %col_t0
OpEmitVertex

; Emit vertex for t=0.25
%inv_t1a = OpConstant %float 0.75
%t1b = OpConstant %float 0.25
%p1_scaled_1 = OpVectorTimesScalar %v3float %p1_xyz %inv_t1a
%p2_scaled_1 = OpVectorTimesScalar %v3float %p2_xyz %t1b
%pos_t1 = OpFAdd %v3float %p1_scaled_1 %p2_scaled_1

%v_t1_a = OpFAdd %v3float %pos_t1 %perp_thick
%v_t1_a_4d = OpCompositeConstruct %v4float %v_t1_a %float_1
OpStore %gl_pos_ptr %v_t1_a_4d
%col_scale_1a = OpConstant %float 0.75
%col_scale_1b = OpConstant %float 0.25
%col_1a = OpVectorTimesScalar %v3float %col1 %col_scale_1a
%col_1b = OpVectorTimesScalar %v3float %col2 %col_scale_1b
%col_t1 = OpFAdd %v3float %col_1a %col_1b
OpStore %outColor %col_t1
OpEmitVertex

%v_t1_b = OpFSub %v3float %pos_t1 %perp_thick
%v_t1_b_4d = OpCompositeConstruct %v4float %v_t1_b %float_1
OpStore %gl_pos_ptr %v_t1_b_4d
OpStore %outColor %col_t1
OpEmitVertex

; Emit vertex for t=0.5
%inv_t2a = OpConstant %float 0.5
%t2b = OpConstant %float 0.5
%p1_scaled_2 = OpVectorTimesScalar %v3float %p1_xyz %inv_t2a
%p2_scaled_2 = OpVectorTimesScalar %v3float %p2_xyz %t2b
%pos_t2 = OpFAdd %v3float %p1_scaled_2 %p2_scaled_2

%v_t2_a = OpFAdd %v3float %pos_t2 %perp_thick
%v_t2_a_4d = OpCompositeConstruct %v4float %v_t2_a %float_1
OpStore %gl_pos_ptr %v_t2_a_4d
%col_t2 = OpFAdd %v3float %col_1a %col_1b
OpStore %outColor %col_t2
OpEmitVertex

%v_t2_b = OpFSub %v3float %pos_t2 %perp_thick
%v_t2_b_4d = OpCompositeConstruct %v4float %v_t2_b %float_1
OpStore %gl_pos_ptr %v_t2_b_4d
OpStore %outColor %col_t2
OpEmitVertex

; Emit vertex for t=0.75
%inv_t3a = OpConstant %float 0.25
%t3b = OpConstant %float 0.75
%p1_scaled_3 = OpVectorTimesScalar %v3float %p1_xyz %inv_t3a
%p2_scaled_3 = OpVectorTimesScalar %v3float %p2_xyz %t3b
%pos_t3 = OpFAdd %v3float %p1_scaled_3 %p2_scaled_3

%v_t3_a = OpFAdd %v3float %pos_t3 %perp_thick
%v_t3_a_4d = OpCompositeConstruct %v4float %v_t3_a %float_1
OpStore %gl_pos_ptr %v_t3_a_4d
%col_scale_3a = OpConstant %float 0.25
%col_scale_3b = OpConstant %float 0.75
%col_3a = OpVectorTimesScalar %v3float %col1 %col_scale_3a
%col_3b = OpVectorTimesScalar %v3float %col2 %col_scale_3b
%col_t3 = OpFAdd %v3float %col_3a %col_3b
OpStore %outColor %col_t3
OpEmitVertex

%v_t3_b = OpFSub %v3float %pos_t3 %perp_thick
%v_t3_b_4d = OpCompositeConstruct %v4float %v_t3_b %float_1
OpStore %gl_pos_ptr %v_t3_b_4d
OpStore %outColor %col_t3
OpEmitVertex

; Emit vertex for t=1.0
%pos_t4 = OpVectorTimesScalar %v3float %p2_xyz %float_1

%v_t4_a = OpFAdd %v3float %pos_t4 %perp_thick
%v_t4_a_4d = OpCompositeConstruct %v4float %v_t4_a %float_1
OpStore %gl_pos_ptr %v_t4_a_4d
%col_t4 = OpVectorTimesScalar %v3float %col2 %float_1
OpStore %outColor %col_t4
OpEmitVertex

%v_t4_b = OpFSub %v3float %pos_t4 %perp_thick
%v_t4_b_4d = OpCompositeConstruct %v4float %v_t4_b %float_1
OpStore %gl_pos_ptr %v_t4_b_4d
OpStore %outColor %col_t4
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
