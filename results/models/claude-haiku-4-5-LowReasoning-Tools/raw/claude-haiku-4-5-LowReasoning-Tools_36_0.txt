#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<int> adj[100];
int best_size = 0;
vector<int> best_set;

int upper_bound_greedy(vector<bool>& used) {
    vector<bool> forbidden(n, false);
    int count = 0;

    for (int i = 0; i < n; i++) {
        if (used[i]) {
            forbidden[i] = true;
            for (int u : adj[i]) {
                forbidden[u] = true;
            }
            count++;
        }
    }

    for (int i = 0; i < n; i++) {
        if (!forbidden[i]) {
            count++;
            forbidden[i] = true;
            for (int u : adj[i]) {
                forbidden[u] = true;
            }
        }
    }

    return count;
}

void backtrack(vector<int>& current, int start, vector<bool>& used) {
    if (current.size() > best_size) {
        best_size = current.size();
        best_set = current;
    }

    int upper = upper_bound_greedy(used);
    if (upper <= best_size) {
        return;
    }

    for (int i = start; i < n; i++) {
        if (!used[i]) {
            bool can_add = true;
            for (int u : current) {
                for (int neighbor : adj[u]) {
                    if (neighbor == i) {
                        can_add = false;
                        break;
                    }
                }
                if (!can_add) break;
            }

            if (can_add) {
                used[i] = true;
                current.push_back(i);
                backtrack(current, i + 1, used);
                current.pop_back();
                used[i] = false;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> current;
    vector<bool> used(n, false);
    backtrack(current, 0, used);

    cout << best_size << "\n";
    for (int i = 0; i < (int)best_set.size(); i++) {
        if (i > 0) cout << " ";
        cout << best_set[i];
    }
    cout << "\n";

    return 0;
}
