OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

OpDecorate %UBO Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

OpMemberDecorate %UBO 0 Offset 0 ColMajor MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64 ColMajor MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128 ColMajor MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Function_vec3 = OpTypePointer Function %vec3
%ptr_Function_vec4 = OpTypePointer Function %vec4
%ptr_Function_float = OpTypePointer Function %float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%fn_void = OpTypeFunction %void

%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

%const_0 = OpConstant %int 0
%const_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_8 = OpConstant %float 8.0

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal_raw = OpLoad %vec3 %in_normal
%uv = OpLoad %vec2 %in_uv

; Normalize normal
%normal = OpExtInst %vec3 %GLSL_std_450 OpNormalize %normal_raw

; Load light position
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %const_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

; Calculate UV checkerboard (8x8)
%uv_scaled = OpVectorTimesScalar %vec2 %uv %float_8
%u_scaled = OpCompositeExtract %float %uv_scaled 0
%v_scaled = OpCompositeExtract %float %uv_scaled 1

%u_floor = OpExtInst %float %GLSL_std_450 OpFloor %u_scaled
%v_floor = OpExtInst %float %GLSL_std_450 OpFloor %v_scaled

%u_int = OpConvertFToS %int %u_floor
%v_int = OpConvertFToS %int %v_floor

%sum = OpIAdd %int %u_int %v_int
%parity = OpSMod %int %sum %const_0
%is_even = OpIEqual %bool %parity %const_0

; Select checkerboard color
%color_bright = OpConstantComposite %vec4 %float_0_9 %float_0_9 %float_0_9 %float_1
%color_dark = OpConstantComposite %vec4 %float_0_2 %float_0_2 %float_0_2 %float_1
%checker = OpSelect %vec4 %is_even %color_bright %color_dark

; Calculate Lambertian shading
%light_dir_raw = OpFSub %vec3 %lightPos %worldPos
%light_dir = OpExtInst %vec3 %GLSL_std_450 OpNormalize %light_dir_raw

%diffuse_dot = OpDot %float %normal %light_dir
%diffuse = OpExtInst %float %GLSL_std_450 OpFMax %diffuse_dot %float_0

; Shading factor: 0.5 + 0.5 * diffuse
%shading_half_diff = OpFMul %float %float_0_5 %diffuse
%shading_factor = OpFAdd %float %float_0_5 %shading_half_diff

; Apply shading to checker color
%shaded = OpVectorTimesScalar %vec4 %checker %shading_factor

OpStore %out_color %shaded
OpReturn

OpFunctionEnd
