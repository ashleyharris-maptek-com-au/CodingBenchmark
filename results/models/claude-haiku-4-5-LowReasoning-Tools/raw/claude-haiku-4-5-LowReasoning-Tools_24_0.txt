#include <iostream>
#include <fstream>
#include <thread>
#include <mutex>
#include <chrono>
#include <cmath>
#include <queue>
#include <atomic>
#include <vector>
#include <iomanip>

using namespace std;

struct TerrainCell {
    uint16_t ground_top;
    uint16_t float_bottom;
    uint16_t float_top;
};

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel_remaining;
};

struct Command {
    double thrust, pitch_cmd, yaw_cmd, roll_cmd;
};

class LunarLander {
public:
    int width, depth, height;
    double gravity, max_thrust;
    double target_x, target_y, target_z;
    double start_x, start_y, start_z;

    vector<vector<TerrainCell>> terrain;

    mutex state_mutex, command_mutex;
    State current_state;
    Command last_command = {0, 0, 0, 0};

    bool terrain_loaded = false;
    atomic<bool> running = true;

    void load_terrain(const string& path) {
        ifstream file(path, ios::binary);
        terrain.resize(width, vector<TerrainCell>(depth));

        for (int x = 0; x < width; x++) {
            for (int y = 0; y < depth; y++) {
                uint16_t a, b, c;
                file.read((char*)&a, 2);
                file.read((char*)&b, 2);
                file.read((char*)&c, 2);
                terrain[x][y] = {a, b, c};
            }
        }
        terrain_loaded = true;
    }

    bool is_collision(double x, double y, double z) {
        if (x < 0 || x >= width || y < 0 || y >= depth || z < 0 || z >= height) {
            return true;
        }

        int ix = (int)x;
        int iy = (int)y;

        if (ix >= width) ix = width - 1;
        if (iy >= depth) iy = depth - 1;

        auto& cell = terrain[ix][iy];

        if (z < cell.ground_top) return true;
        if (z >= cell.float_bottom && z < cell.float_top) return true;

        return false;
    }

    double get_terrain_height(double x, double y) {
        int ix = (int)x;
        int iy = (int)y;

        if (ix >= width) ix = width - 1;
        if (iy >= depth) iy = depth - 1;
        if (ix < 0) ix = 0;
        if (iy < 0) iy = 0;

        return terrain[ix][iy].ground_top;
    }

    Command compute_control(const State& state) {
        Command cmd = {0, 0, 0, 0};

        double landing_zone = max(5.0, min(width, depth) * 0.02);

        double dx = target_x - state.x;
        double dy = target_y - state.y;
        double dz = target_z - state.z;

        double horiz_dist = sqrt(dx*dx + dy*dy);
        double speed = sqrt(state.vx*state.vx + state.vy*state.vy + state.vz*state.vz);

        double vertical_vel = state.vz;
        double desired_descent = -0.5;
        double vertical_error = desired_descent - vertical_vel;

        double horiz_vel_mag = sqrt(state.vx*state.vx + state.vy*state.vy);

        double thrust_frac = 0.4;

        if (dz < 50 && horiz_dist < landing_zone * 3) {
            thrust_frac = 0.35 + 0.15 * (vertical_error / 5.0);
            thrust_frac = max(0.1, min(1.0, thrust_frac));
        } else if (dz > 100) {
            thrust_frac = 0.3;
        } else {
            thrust_frac = 0.4;
        }

        if (horiz_vel_mag > 0.1 || horiz_dist > 5) {
            double pitch_target = 0;
            if (horiz_dist > 10) {
                pitch_target = (state.vy > 0 ? 1.0 : -1.0) * 0.3;
                if (dx != 0) {
                    pitch_target = atan2(dx, 50) * 0.5;
                }
            }
            cmd.pitch_cmd = (pitch_target - state.pitch) * 0.5;
            cmd.pitch_cmd = max(-1.0, min(1.0, cmd.pitch_cmd));
        }

        double target_yaw = atan2(dx, dy);
        double yaw_diff = target_yaw - state.yaw;

        while (yaw_diff > M_PI) yaw_diff -= 2*M_PI;
        while (yaw_diff < -M_PI) yaw_diff += 2*M_PI;

        cmd.yaw_cmd = yaw_diff * 0.3;
        cmd.yaw_cmd = max(-1.0, min(1.0, cmd.yaw_cmd));

        cmd.thrust = thrust_frac;
        cmd.roll_cmd = -state.roll * 0.2;

        return cmd;
    }

    void input_thread_fn() {
        string line;
        getline(cin, line);
        sscanf(line.c_str(), "%d %d %d %lf %lf", &width, &depth, &height, &gravity, &max_thrust);

        getline(cin, line);
        sscanf(line.c_str(), "%lf %lf %lf %lf %lf %lf", &start_x, &start_y, &start_z, &target_x, &target_y, &target_z);

        string terrain_path;
        getline(cin, terrain_path);

        load_terrain(terrain_path);

        getline(cin, line);

        while (running) {
            if (getline(cin, line)) {
                State s;
                if (sscanf(line.c_str(), "%lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
                    &s.x, &s.y, &s.z, &s.vx, &s.vy, &s.vz, &s.pitch, &s.yaw, &s.roll,
                    &s.pitch_rate, &s.yaw_rate, &s.roll_rate, &s.fuel_remaining) == 13) {
                    lock_guard<mutex> lock(state_mutex);
                    current_state = s;
                }
            }
        }
    }

    void run() {
        thread input_thread(&LunarLander::input_thread_fn, this);

        auto start_time = chrono::high_resolution_clock::now();

        while (running) {
            if (!terrain_loaded) {
                this_thread::sleep_for(chrono::milliseconds(10));
                continue;
            }

            State state_copy;
            {
                lock_guard<mutex> lock(state_mutex);
                state_copy = current_state;
            }

            if (state_copy.fuel_remaining <= 0) {
                running = false;
                break;
            }

            Command cmd = compute_control(state_copy);

            {
                lock_guard<mutex> lock(command_mutex);
                last_command = cmd;
            }

            cout << fixed << setprecision(6) << cmd.thrust << " "
                 << cmd.pitch_cmd << " " << cmd.yaw_cmd << " " << cmd.roll_cmd << "\n";
            cout.flush();

            this_thread::sleep_for(chrono::milliseconds(100));
        }

        input_thread.join();
    }
};

int main() {
    LunarLander lander;
    lander.run();
    return 0;
}
