const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

function bfs(grid, start, end) {
  const rows = grid.length;
  const cols = grid[0].length;

  if (start.x === end.x && start.y === end.y) return 0;

  const queue = [[start.x, start.y, 0]];
  const visited = new Set();
  visited.add(`${start.x},${start.y}`);

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift();

    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]];
    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.') {
        const key = `${nx},${ny}`;
        if (!visited.has(key)) {
          if (nx === end.x && ny === end.y) {
            return dist + 1;
          }
          visited.add(key);
          queue.push([nx, ny, dist + 1]);
        }
      }
    }
  }

  return Infinity;
}

const { grid, robots, jobs } = data;

// Calculate distances from each robot to each job
const distances = Array(robots.length).fill(null).map(() => Array(jobs.length).fill(Infinity));
for (let i = 0; i < robots.length; i++) {
  for (let j = 0; j < jobs.length; j++) {
    distances[i][j] = bfs(grid, robots[i], jobs[j]);
  }
}

// Greedy assignment: repeatedly assign closest available (robot, job) pair
const assignments = Array(robots.length).fill(null).map(() => []);
const assigned = new Set();

const pairs = [];
for (let i = 0; i < robots.length; i++) {
  for (let j = 0; j < jobs.length; j++) {
    pairs.push({ robot: i, job: j, dist: distances[i][j] });
  }
}

pairs.sort((a, b) => a.dist - b.dist);

for (const pair of pairs) {
  if (!assigned.has(pair.job)) {
    assignments[pair.robot].push(jobs[pair.job].id);
    assigned.add(pair.job);
  }
}
