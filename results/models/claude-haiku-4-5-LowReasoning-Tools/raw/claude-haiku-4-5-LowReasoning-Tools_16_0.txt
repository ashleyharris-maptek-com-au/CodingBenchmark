#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> jobs(J);

    // Read input
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        for (int t = 0; t < T; t++) {
            int machine, duration;
            cin >> machine >> duration;
            jobs[j].push_back({machine, duration});
        }
    }

    // Store schedule: start_time for each task
    vector<vector<int>> schedule(J);
    for (int j = 0; j < J; j++) {
        schedule[j].resize(jobs[j].size());
    }

    // Track when each machine becomes free
    vector<int> machine_free_time(M, 0);

    // Track when each job will be ready (after current task completes)
    vector<int> job_free_time(J, 0);

    // Greedy scheduling: process each job's tasks in order
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < jobs[j].size(); t++) {
            int machine = jobs[j][t].first;
            int duration = jobs[j][t].second;

            // Task starts when both machine is free AND previous task in job is done
            int start_time = max(machine_free_time[machine], job_free_time[j]);

            schedule[j][t] = start_time;

            // Update availability
            machine_free_time[machine] = start_time + duration;
            job_free_time[j] = start_time + duration;
        }
    }

    // Output: for each job, print start_time machine pairs
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < jobs[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << schedule[j][t] << " " << jobs[j][t].first;
        }
        cout << "\n";
    }

    return 0;
}
