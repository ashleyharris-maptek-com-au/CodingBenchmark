#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <algorithm>
#include <set>
#include <map>
#include <cstring>

using namespace std;

const long long INF = 1e18;

int n, m, k;
vector<pair<int, long long>> adj[505];
int terminals[55];
long long dist[505][505];
int next_hop[505][505];

void floyd_warshall() {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (i == j) {
                dist[i][j] = 0;
                next_hop[i][j] = j;
            } else {
                dist[i][j] = INF;
                next_hop[i][j] = -1;
            }
        }
    }

    for (int u = 0; u < n; u++) {
        for (auto [v, w] : adj[u]) {
            if (w < dist[u][v]) {
                dist[u][v] = w;
                next_hop[u][v] = v;
            }
        }
    }

    for (int mid = 0; mid < n; mid++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (dist[i][mid] < INF && dist[mid][j] < INF) {
                    long long new_dist = dist[i][mid] + dist[mid][j];
                    if (new_dist < dist[i][j]) {
                        dist[i][j] = new_dist;
                        next_hop[i][j] = next_hop[i][mid];
                    }
                }
            }
        }
    }
}

long long dp[33][505];
int parent_subset[33][505];
int parent_vertex[33][505];

set<pair<int, int>> edges_in_tree;

void reconstruct(int mask, int v) {
    if (__builtin_popcount(mask) == 1) {
        return;
    }

    if (parent_vertex[mask][v] == -1) {
        int best_sub = -1;
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int comp = mask ^ sub;
            if (dp[sub][v] + dp[comp][v] == dp[mask][v]) {
                best_sub = sub;
                break;
            }
        }
        if (best_sub != -1) {
            reconstruct(best_sub, v);
            reconstruct(mask ^ best_sub, v);
        }
    } else {
        int u = parent_vertex[mask][v];
        int prev = v;
        while (u != prev) {
            edges_in_tree.insert({min(u, prev), max(u, prev)});
            int next = next_hop[u][v];
            prev = u;
            u = next;
        }
        reconstruct(parent_subset[mask][v], u);
    }
}

long long solve() {
    if (k == 1) {
        return 0;
    }

    floyd_warshall();

    // Initialize DP
    for (int mask = 0; mask < (1 << k); mask++) {
        for (int v = 0; v < n; v++) {
            dp[mask][v] = INF;
            parent_subset[mask][v] = -1;
            parent_vertex[mask][v] = -1;
        }
    }

    // Base cases: single terminals
    for (int i = 0; i < k; i++) {
        int mask = (1 << i);
        dp[mask][terminals[i]] = 0;
    }

    // DP loop
    for (int mask = 1; mask < (1 << k); mask++) {
        // Combine subsets
        for (int v = 0; v < n; v++) {
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                int comp = mask ^ sub;
                long long new_cost = dp[sub][v] + dp[comp][v];
                if (new_cost < dp[mask][v]) {
                    dp[mask][v] = new_cost;
                    parent_subset[mask][v] = sub;
                    parent_vertex[mask][v] = -1;
                }
            }
        }

        // Extend via shortest paths
        for (int v = 0; v < n; v++) {
            for (int u = 0; u < n; u++) {
                if (dp[mask][u] < INF && dist[u][v] < INF) {
                    long long new_cost = dp[mask][u] + dist[u][v];
                    if (new_cost < dp[mask][v]) {
                        dp[mask][v] = new_cost;
                        parent_subset[mask][v] = mask;
                        parent_vertex[mask][v] = u;
                    }
                }
            }
        }
    }

    long long ans = INF;
    int best_v = -1;
    int full_mask = (1 << k) - 1;
    for (int v = 0; v < n; v++) {
        if (dp[full_mask][v] < ans) {
            ans = dp[full_mask][v];
            best_v = v;
        }
    }

    if (ans < INF) {
        edges_in_tree.clear();
        reconstruct(full_mask, best_v);
    }

    return ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;

    for (int i = 0; i < m; i++) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    long long ans = solve();

    cout << ans << " " << edges_in_tree.size() << "\n";
    for (auto [u, v] : edges_in_tree) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
