OpCapability Geometry
OpCapability ClipDistance
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO_Block BufferBlock
OpMemberDecorate %UBO_Block 0 Offset 0
OpMemberDecorate %UBO_Block 1 Offset 64
OpMemberDecorate %UBO_Block 2 Offset 128
OpMemberDecorate %UBO_Block 3 Offset 192
OpMemberDecorate %UBO_Block 4 Offset 208
OpMemberDecorate %UBO_Block 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%uint_4 = OpConstant %int 4

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input arrays for per-vertex attributes
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output per-vertex color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO
%UBO_Block = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO_Block
%ubo = OpVariable %ptr_uniform_UBO Uniform

; Type pointers for indexing
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

; Declare extended instruction set for math
%glsl_ext_inst = OpExtInstImport "GLSL.std.450"

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_thickness = OpConstant %float 0.025

; Function signature
%fn_void = OpTypeFunction %void

; Main function
%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load positions from gl_in array
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr

%pos1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr

%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr

%pos3_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr

; Load colors
%col1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%col1 = OpLoad %v3float %col1_ptr

%col2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr

; Extract xyz components
%pos0_xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%pos1_xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%pos2_xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%pos3_xyz = OpVectorShuffle %v3float %pos3 %pos3 0 1 2

; Compute tangent at start: t0 = normalize(pos2 - pos0)
%delta0 = OpFSub %v3float %pos2_xyz %pos0_xyz
%t0 = OpExtInst %v3float %glsl_ext_inst Normalize %delta0

; Compute tangent at end: t1 = normalize(pos3 - pos1)
%delta1 = OpFSub %v3float %pos3_xyz %pos1_xyz
%t1 = OpExtInst %v3float %glsl_ext_inst Normalize %delta1

; Compute perpendicular at start: perp0 = vec2(-t0.y, t0.x)
%t0_x = OpVectorShuffle %float %t0 %t0 0
%t0_y = OpVectorShuffle %float %t0 %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0_z = OpConstant %float 0.0
%perp0 = OpCompositeConstruct %v3float %neg_t0_y %t0_x %perp0_z

; Compute perpendicular at end: perp1 = vec2(-t1.y, t1.x)
%t1_x = OpVectorShuffle %float %t1 %t1 0
%t1_y = OpVectorShuffle %float %t1 %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v3float %neg_t1_y %t1_x %perp0_z

; Scale perpendiculars by thickness
%perp0_scaled = OpVectorTimesScalar %v3float %perp0 %float_thickness
%perp1_scaled = OpVectorTimesScalar %v3float %perp1 %float_thickness

; Emit 4 vertices as triangle strip
; Vertex 0: start + perp0*thickness
%v0_pos = OpFAdd %v3float %pos1_xyz %perp0_scaled
%v0_pos_4 = OpCompositeConstruct %v4float %v0_pos %float_1
%out_pos_ptr_0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_ptr_0 %v0_pos_4
OpStore %outColor %col1
OpEmitVertex

; Vertex 1: start - perp0*thickness
%neg_perp0_scaled = OpVectorNegate %v3float %perp0_scaled
%v1_pos = OpFAdd %v3float %pos1_xyz %neg_perp0_scaled
%v1_pos_4 = OpCompositeConstruct %v4float %v1_pos %float_1
OpStore %out_pos_ptr_0 %v1_pos_4
OpStore %outColor %col1
OpEmitVertex

; Vertex 2: end + perp1*thickness
%v2_pos = OpFAdd %v3float %pos2_xyz %perp1_scaled
%v2_pos_4 = OpCompositeConstruct %v4float %v2_pos %float_1
OpStore %out_pos_ptr_0 %v2_pos_4
OpStore %outColor %col2
OpEmitVertex

; Vertex 3: end - perp1*thickness
%neg_perp1_scaled = OpVectorNegate %v3float %perp1_scaled
%v3_pos = OpFAdd %v3float %pos2_xyz %neg_perp1_scaled
%v3_pos_4 = OpCompositeConstruct %v4float %v3_pos %float_1
OpStore %out_pos_ptr_0 %v3_pos_4
OpStore %outColor %col2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
