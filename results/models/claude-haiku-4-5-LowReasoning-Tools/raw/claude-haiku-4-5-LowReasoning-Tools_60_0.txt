const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Index tasks by ID and build dependency relationships
const taskMap = {};
const dependencies = {}; // task -> list of dependency task IDs
const dependents = {}; // task -> list of tasks that depend on it

tasks.forEach(task => {
  taskMap[task.id] = task;
  dependencies[task.id] = task.deps || [];
  dependents[task.id] = [];
});

// Build reverse dependency graph
tasks.forEach(task => {
  (task.deps || []).forEach(depId => {
    if (!dependents[depId]) {
      dependents[depId] = [];
    }
    dependents[depId].push(task.id);
  });
});

// Priority heuristic for task scheduling
// Higher weight and earlier deadline get higher priority
// Longer duration also gets slight boost (helps minimize makespan)
function getPriority(taskId) {
  const task = taskMap[taskId];
  const weight = task.weight || 1;
  const deadline = task.deadline !== undefined ? task.deadline : Infinity;

  // Weight is primary factor (weighted lateness objective)
  // Deadline is secondary (avoid lateness penalties)
  // Duration helps with makespan (longer tasks earlier)
  let priority = weight * 10000;
  if (deadline !== Infinity) {
    priority += (10000 - deadline);
  }
  priority += task.duration;

  return priority;
}

// Get tasks that are ready to schedule (all dependencies satisfied)
function getReadyTasks(completed) {
  const ready = [];
  for (const taskId of Object.keys(dependencies)) {
    if (!completed.has(taskId)) {
      const deps = dependencies[taskId];
      if (deps.every(d => completed.has(d))) {
        ready.push(taskId);
      }
    }
  }
  return ready;
}

// Greedy list scheduling algorithm
const machineFinishTime = new Array(machineCount).fill(0);
const taskFinishTime = {};
const completed = new Set();
const schedule = [];

// Schedule all tasks
while (completed.size < tasks.length) {
  const readyTasks = getReadyTasks(completed);

  // Sort by priority (higher priority first)
  readyTasks.sort((a, b) => getPriority(b) - getPriority(a));

  // Schedule the highest priority ready task
  const taskId = readyTasks[0];
  const task = taskMap[taskId];

  // Find earliest time this task can start
  // (respecting when all dependencies complete)
  let earliestStart = 0;
  dependencies[taskId].forEach(depId => {
    earliestStart = Math.max(earliestStart, taskFinishTime[depId] || 0);
  });

  // Find machine that becomes available earliest
  let bestMachine = 0;
  let bestStartTime = Math.max(earliestStart, machineFinishTime[0]);

  for (let m = 1; m < machineCount; m++) {
    const potentialStart = Math.max(earliestStart, machineFinishTime[m]);
    if (potentialStart < bestStartTime) {
      bestStartTime = potentialStart;
      bestMachine = m;
    }
  }

  const endTime = bestStartTime + task.duration;

  // Record the scheduled task
  schedule.push({
    id: taskId,
    start: bestStartTime,
    machine: bestMachine
  });

  // Update state
  taskFinishTime[taskId] = endTime;
  machineFinishTime[bestMachine] = endTime;
  completed.add(taskId);
}

// Output the schedule
console.log(JSON.stringify({ schedule }));
