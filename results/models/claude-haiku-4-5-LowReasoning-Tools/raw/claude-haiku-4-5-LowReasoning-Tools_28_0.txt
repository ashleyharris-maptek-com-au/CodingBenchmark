use std::io::{self, BufRead};
use std::collections::{HashMap, HashSet};

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap()?;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_edges: usize = parts[1].parse().unwrap();

    let mut adj: HashMap<usize, HashSet<usize>> = HashMap::new();

    for _ in 0..num_edges {
        let line = lines.next().unwrap()?;
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();

        adj.entry(u).or_insert_with(HashSet::new).insert(v);
        adj.entry(v).or_insert_with(HashSet::new).insert(u);
    }

    let mut cover: HashSet<usize> = HashSet::new();

    while !adj.is_empty() {
        let max_vertex = *adj.keys().max_by_key(|&&v| adj[&v].len()).unwrap();
        let neighbors: Vec<usize> = adj[&max_vertex].iter().cloned().collect();

        cover.insert(max_vertex);
        adj.remove(&max_vertex);

        for neighbor in neighbors {
            if let Some(neighbor_set) = adj.get_mut(&neighbor) {
                neighbor_set.remove(&max_vertex);
            }
        }
    }

    println!("{}", cover.len());
    let mut vertices: Vec<usize> = cover.into_iter().collect();
    vertices.sort_unstable();
    println!("{}", vertices.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" "));

    Ok(())
}
