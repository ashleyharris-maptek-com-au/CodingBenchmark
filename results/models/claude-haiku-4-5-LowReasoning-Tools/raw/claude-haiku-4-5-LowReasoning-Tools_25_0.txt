using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class AsteroidInterceptor
{
    struct Body
    {
        public string Name;
        public double Mass;
        public double[] Pos;
        public double[] Vel;
    }

    struct Burn
    {
        public double Time;
        public double[] DeltaV;
    }

    const double G = 6.67430e-11;
    const double AU = 1.495978707e11;
    const double DAY_SECONDS = 86400.0;

    static void Main()
    {
        try
        {
            var input = Console.In.ReadToEnd().Trim().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);

            var headerParts = input[0].Split();
            int numBodies = int.Parse(headerParts[0]);
            double warningDays = double.Parse(headerParts[1]);
            double impactorMass = double.Parse(headerParts[2]);
            double deltaVBudget = double.Parse(headerParts[3]);
            double asteroidMass = double.Parse(headerParts[4]);
            long t0UnixUtc = long.Parse(headerParts[5]);

            var bodies = new Body[numBodies];
            for (int i = 0; i < numBodies; i++)
            {
                var parts = input[i + 1].Split();
                bodies[i].Name = parts[0];
                bodies[i].Mass = double.Parse(parts[1]);
                bodies[i].Pos = new[] { double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4]) };
                bodies[i].Vel = new[] { double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]) };
            }

            string bspPath = input[numBodies + 1].Trim();

            var burns = SolveInterception(bodies, warningDays, impactorMass, deltaVBudget, asteroidMass);

            Console.WriteLine(burns.Count);
            foreach (var burn in burns)
            {
                Console.WriteLine($"{burn.Time} {burn.DeltaV[0]} {burn.DeltaV[1]} {burn.DeltaV[2]}");
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine("0");
        }
    }

    static List<Burn> SolveInterception(Body[] bodies, double warningDays, double impactorMass, double deltaVBudget, double asteroidMass)
    {
        var burns = new List<Burn>();

        var earth = Array.Find(bodies, b => b.Name == "Earth");
        var asteroid = Array.Find(bodies, b => b.Name == "Asteroid");

        if (earth.Name == null || asteroid.Name == null)
        {
            return burns;
        }

        double launchTime = CalculateLaunchTime(asteroid, earth, warningDays);

        if (launchTime < 0 || launchTime > warningDays)
        {
            launchTime = 0;
        }

        var sunIdx = Array.FindIndex(bodies, b => b.Name == "Sun");
        Body sun = sunIdx >= 0 ? bodies[sunIdx] : new Body { Mass = 1.98892e30, Pos = new double[3], Vel = new double[3] };

        var spacecraftPos = (double[])earth.Pos.Clone();
        var spacecraftVel = (double[])earth.Vel.Clone();

        var launchDv = CalculateLaunchBurn(spacecraftPos, spacecraftVel, asteroid.Pos, asteroid.Vel, sun.Pos, sun.Mass);
        burns.Add(new Burn { Time = launchTime, DeltaV = launchDv });

        PropagateOrbit(spacecraftPos, spacecraftVel, sun.Pos, sun.Mass, launchTime, ref asteroid);

        double midCourseTime = launchTime + (asteroid.Pos[0] - spacecraftPos[0]) / Math.Max(1, spacecraftVel[0]);
        if (midCourseTime > launchTime && midCourseTime < warningDays * 0.8)
        {
            var correctionDv = CalculateCorrection(spacecraftPos, spacecraftVel, asteroid.Pos, asteroid.Vel, sun.Pos, sun.Mass, deltaVBudget);
            if (Magnitude(correctionDv) > 0.01)
            {
                burns.Add(new Burn { Time = midCourseTime, DeltaV = correctionDv });
                for (int i = 0; i < 3; i++) spacecraftVel[i] += correctionDv[i];
            }
        }

        double impactTime = CalculateImpactTime(spacecraftPos, spacecraftVel, asteroid.Pos, asteroid.Vel, sun.Pos, sun.Mass, warningDays);

        if (impactTime > 0 && impactTime <= warningDays)
        {
            var impactDv = new double[3];
            burns.Add(new Burn { Time = impactTime, DeltaV = impactDv });
        }

        return burns;
    }

    static double CalculateLaunchTime(Body asteroid, Body earth, double warningDays)
    {
        double relDist = Magnitude(Subtract(asteroid.Pos, earth.Pos));
        double launchTime = Math.Max(0, Math.Min(warningDays * 0.1, warningDays));
        return launchTime;
    }

    static double[] CalculateLaunchBurn(double[] scPos, double[] scVel, double[] astPos, double[] astVel, double[] sunPos, double sunMass)
    {
        var targetVel = Subtract(astVel, scVel);
        var dv = new double[3];

        for (int i = 0; i < 3; i++)
        {
            dv[i] = targetVel[i] * 0.1;
        }

        return dv;
    }

    static double[] CalculateCorrection(double[] scPos, double[] scVel, double[] astPos, double[] astVel, double[] sunPos, double sunMass, double budget)
    {
        var relPos = Subtract(astPos, scPos);
        var relVel = Subtract(astVel, scVel);

        var dv = new double[3];
        double relDist = Magnitude(relPos);

        if (relDist > 0)
        {
            for (int i = 0; i < 3; i++)
            {
                dv[i] = relVel[i] * 0.05;
            }
        }

        if (Magnitude(dv) > budget * 0.3)
        {
            for (int i = 0; i < 3; i++)
            {
                dv[i] *= (budget * 0.3) / Magnitude(dv);
            }
        }

        return dv;
    }

    static double CalculateImpactTime(double[] scPos, double[] scVel, double[] astPos, double[] astVel, double[] sunPos, double sunMass, double warningDays)
    {
        var relPos = Subtract(astPos, scPos);
        var relVel = Subtract(astVel, scVel);

        double a = Dot(relVel, relVel);
        double b = 2 * Dot(relPos, relVel);
        double c = Dot(relPos, relPos) - 1e12;

        double disc = b * b - 4 * a * c;
        if (disc < 0 || a < 1e-6)
        {
            return warningDays * 0.9;
        }

        double t1 = (-b + Math.Sqrt(disc)) / (2 * a);
        return Math.Min(t1, warningDays * 0.95);
    }

    static void PropagateOrbit(double[] pos, double[] vel, double[] sunPos, double sunMass, double time, ref Body body)
    {
        double dt = (time * DAY_SECONDS) / 100;
        for (int step = 0; step < 100; step++)
        {
            var acc = GravAccel(pos, sunPos, sunMass);
            for (int i = 0; i < 3; i++)
            {
                vel[i] += acc[i] * dt;
                pos[i] += vel[i] * dt;
            }
        }
        body.Pos = pos;
        body.Vel = vel;
    }

    static double[] GravAccel(double[] pos, double[] sunPos, double sunMass)
    {
        var r = Subtract(sunPos, pos);
        double dist = Magnitude(r);
        double dist3 = dist * dist * dist;

        var acc = new double[3];
        if (dist3 > 1e-10)
        {
            double coeff = G * sunMass / dist3;
            for (int i = 0; i < 3; i++)
            {
                acc[i] = coeff * r[i];
            }
        }
        return acc;
    }

    static double[] Subtract(double[] a, double[] b)
    {
        return new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    }

    static double[] Add(double[] a, double[] b)
    {
        return new[] { a[0] + b[0], a[1] + b[1], a[2] + b[2] };
    }

    static double Dot(double[] a, double[] b)
    {
        return a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
    }

    static double Magnitude(double[] v)
    {
        return Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
    }
}
