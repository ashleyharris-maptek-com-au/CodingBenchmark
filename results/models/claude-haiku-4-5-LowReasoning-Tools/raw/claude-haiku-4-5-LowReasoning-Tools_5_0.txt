use std::io::{self, BufRead};
use std::collections::{HashSet, VecDeque};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<u32> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let width = parts[0] as i32;
    let height = parts[1] as i32;
    let obstacle_count = parts[2] as usize;
    let require_cycle = parts[3] != 0;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<i32> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.insert((coords[0], coords[1]));
    }

    // Total cells to visit
    let total_cells = (width * height) as usize - obstacle_count;

    // Quick feasibility checks
    if !is_feasible(width, height, &obstacles, total_cells) {
        println!("0");
        return;
    }

    // Try to find a Hamiltonian path
    let mut visited = HashSet::new();
    let mut path = Vec::new();

    visited.insert((0, 0));
    path.push((0, 0));

    if dfs(&mut visited, &mut path, width, height, &obstacles, total_cells, require_cycle) {
        // Found a path
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        // No path exists
        println!("0");
    }
}

fn is_feasible(width: i32, height: i32, obstacles: &HashSet<(i32, i32)>, total_cells: usize) -> bool {
    // Check if (0, 0) is not an obstacle
    if obstacles.contains(&(0, 0)) {
        return false;
    }

    // Check bipartite coloring: in a Hamiltonian path on a grid, colors must differ by at most 1
    let mut black_count = 0;
    let mut white_count = 0;

    for x in 0..width {
        for y in 0..height {
            if !obstacles.contains(&(x, y)) {
                if (x + y) % 2 == 0 {
                    black_count += 1;
                } else {
                    white_count += 1;
                }
            }
        }
    }

    let diff = (black_count - white_count).abs();
    if diff > 1 {
        return false;
    }

    // Check if grid is connected
    let mut visited = HashSet::new();
    let mut queue = VecDeque::new();
    queue.push_back((0, 0));
    visited.insert((0, 0));

    while let Some((x, y)) = queue.pop_front() {
        let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];
        for (dx, dy) in &directions {
            let nx = x + dx;
            let ny = y + dy;

            if nx >= 0 && nx < width && ny >= 0 && ny < height &&
               !obstacles.contains(&(nx, ny)) && !visited.contains(&(nx, ny)) {
                visited.insert((nx, ny));
                queue.push_back((nx, ny));
            }
        }
    }

    visited.len() == total_cells
}

fn dfs(
    visited: &mut HashSet<(i32, i32)>,
    path: &mut Vec<(i32, i32)>,
    width: i32,
    height: i32,
    obstacles: &HashSet<(i32, i32)>,
    total_cells: usize,
    require_cycle: bool,
) -> bool {
    if visited.len() == total_cells {
        if require_cycle {
            let (last_x, last_y) = path[path.len() - 1];
            let (start_x, start_y) = path[0];
            return (last_x - start_x).abs() + (last_y - start_y).abs() == 1;
        } else {
            return true;
        }
    }

    let (x, y) = path[path.len() - 1];
    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    // Find unvisited neighbors and sort by degree (most constrained first)
    let mut neighbors = Vec::new();

    for (dx, dy) in &directions {
        let nx = x + dx;
        let ny = y + dy;

        if nx >= 0 && nx < width && ny >= 0 && ny < height &&
           !obstacles.contains(&(nx, ny)) && !visited.contains(&(nx, ny)) {
            neighbors.push((nx, ny));
        }
    }

    if neighbors.is_empty() {
        return false;
    }

    // Sort neighbors by their future degree (most constrained first)
    neighbors.sort_by_key(|&(nx, ny)| {
        let mut count = 0;
        for (dx, dy) in &directions {
            let nnx = nx + dx;
            let nny = ny + dy;

            if nnx >= 0 && nnx < width && nny >= 0 && nny < height &&
               !obstacles.contains(&(nnx, nny)) && !visited.contains(&(nnx, nny)) {
                count += 1;
            }
        }
        count
    });

    for (nx, ny) in neighbors {
        visited.insert((nx, ny));
        path.push((nx, ny));

        if dfs(visited, path, width, height, obstacles, total_cells, require_cycle) {
            return true;
        }

        visited.remove(&(nx, ny));
        path.pop();
    }

    false
}
