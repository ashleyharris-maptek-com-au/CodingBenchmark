using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int H = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);

        char[][] board = new char[H][];
        for (int i = 0; i < H; i++)
        {
            string line = Console.ReadLine();
            board[i] = new char[W];
            for (int j = 0; j < W; j++)
            {
                board[i][j] = j < line.Length ? line[j] : ' ';
            }
        }

        var safeCells = FindSafeCells(board, H, W);

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells.OrderBy(c => c.Item1).ThenBy(c => c.Item2))
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    static List<(int, int)> FindSafeCells(char[][] board, int H, int W)
    {
        var safe = new HashSet<(int, int)>();
        var state = new char[H][];
        for (int i = 0; i < H; i++)
        {
            state[i] = (char[])board[i].Clone();
        }

        bool changed = true;
        while (changed)
        {
            changed = false;

            // Basic constraint propagation
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    if (char.IsDigit(state[y][x]))
                    {
                        int mineCount = state[y][x] - '0';
                        var neighbors = GetNeighbors(x, y, H, W);

                        int flaggedCount = 0;
                        var unknownNeighbors = new List<(int, int)>();

                        foreach (var (nx, ny) in neighbors)
                        {
                            if (state[ny][nx] == '*')
                                flaggedCount++;
                            else if (state[ny][nx] == ' ')
                                unknownNeighbors.Add((nx, ny));
                        }

                        // If all mines found, remaining unknowns are safe
                        if (flaggedCount == mineCount)
                        {
                            foreach (var (nx, ny) in unknownNeighbors)
                            {
                                if (state[ny][nx] == ' ')
                                {
                                    state[ny][nx] = '.';
                                    safe.Add((nx, ny));
                                    changed = true;
                                }
                            }
                        }

                        // If unknowns + flagged = required mines, all unknowns are mines
                        else if (flaggedCount + unknownNeighbors.Count == mineCount)
                        {
                            foreach (var (nx, ny) in unknownNeighbors)
                            {
                                if (state[ny][nx] == ' ')
                                {
                                    state[ny][nx] = 'M';
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
        }

        return safe.ToList();
    }

    static List<(int, int)> GetNeighbors(int x, int y, int H, int W)
    {
        var neighbors = new List<(int, int)>();
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                int ny = y + dy;
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    neighbors.Add((nx, ny));
            }
        }
        return neighbors;
    }
}
