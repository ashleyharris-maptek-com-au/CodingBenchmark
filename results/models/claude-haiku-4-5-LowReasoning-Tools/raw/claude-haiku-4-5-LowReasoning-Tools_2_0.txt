using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostmanSolver
{
    static void Main()
    {
        string line = Console.ReadLine();
        var parts = line.Split();
        int n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        var graph = new List<(int neighbor, int weight)>[n];
        for (int i = 0; i < n; i++)
            graph[i] = new List<(int, int)>();

        for (int i = 0; i < m; i++)
        {
            line = Console.ReadLine();
            parts = line.Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            graph[u].Add((v, w));
            graph[v].Add((u, w));
        }

        var route = SolveChinesePostman(n, graph);
        Console.WriteLine(string.Join(" ", route));
    }

    static int[] SolveChinesePostman(int n, List<(int, int)>[] graph)
    {
        // Find odd-degree vertices
        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
            if (graph[i].Count % 2 == 1)
                oddVertices.Add(i);

        // If all vertices have even degree, graph is Eulerian
        if (oddVertices.Count == 0)
            return FindEulerianCircuit(n, graph);

        // Compute shortest paths between all pairs
        var dist = ComputeShortestPaths(n, graph);

        // Find minimum weight perfect matching on odd vertices
        var matching = FindMinWeightMatching(oddVertices, dist);

        // Create augmented graph with virtual edges from matching
        var augmentedGraph = new List<(int, int)>[n];
        for (int i = 0; i < n; i++)
            augmentedGraph[i] = new List<(int, int)>(graph[i]);

        foreach (var (u, v) in matching)
        {
            augmentedGraph[u].Add((v, 0));
            augmentedGraph[v].Add((u, 0));
        }

        return FindEulerianCircuit(n, augmentedGraph);
    }

    static int[][] ComputeShortestPaths(int n, List<(int, int)>[] graph)
    {
        const int INF = 1000000;
        var dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++)
                dist[i][j] = INF;
            dist[i][i] = 0;
        }

        // Initialize with direct edges
        for (int u = 0; u < n; u++)
            foreach (var (v, w) in graph[u])
                dist[u][v] = Math.Min(dist[u][v], w);

        // Floyd-Warshall
        for (int k = 0; k < n; k++)
            for (int i = 0; i < n; i++)
                for (int j = 0; j < n; j++)
                    if (dist[i][k] < INF && dist[k][j] < INF)
                        dist[i][j] = Math.Min(dist[i][j], dist[i][k] + dist[k][j]);

        return dist;
    }

    static List<(int, int)> FindMinWeightMatching(List<int> oddVertices, int[][] dist)
    {
        int k = oddVertices.Count;
        int fullMask = (1 << k) - 1;

        var dp = new int[1 << k];
        var parent = new int[1 << k];
        for (int i = 0; i < (1 << k); i++)
        {
            dp[i] = int.MaxValue;
            parent[i] = -1;
        }
        dp[0] = 0;

        // DP: process each state
        for (int mask = 0; mask <= fullMask; mask++)
        {
            if (dp[mask] == int.MaxValue) continue;

            // Find first unmatched vertex
            int i = -1;
            for (int j = 0; j < k; j++)
                if ((mask & (1 << j)) == 0) { i = j; break; }

            if (i == -1) continue;

            // Match with another unmatched vertex
            for (int j = i + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;

                int newMask = mask | (1 << i) | (1 << j);
                int cost = dist[oddVertices[i]][oddVertices[j]];
                if (dp[mask] + cost < dp[newMask])
                {
                    dp[newMask] = dp[mask] + cost;
                    parent[newMask] = mask;
                }
            }
        }

        // Reconstruct matching
        var matching = new List<(int, int)>();
        int currentMask = fullMask;
        while (currentMask != 0)
        {
            int prevMask = parent[currentMask];
            int diff = currentMask ^ prevMask;

            var indices = new List<int>();
            for (int i = 0; i < k; i++)
                if ((diff & (1 << i)) != 0)
                    indices.Add(i);

            matching.Add((oddVertices[indices[0]], oddVertices[indices[1]]));
            currentMask = prevMask;
        }

        return matching;
    }

    static int[] FindEulerianCircuit(int n, List<(int, int)>[] graph)
    {
        var adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int u = 0; u < n; u++)
            foreach (var (v, _) in graph[u])
                adj[u].Add(v);

        var circuit = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (adj[u].Count > 0)
            {
                int v = adj[u][adj[u].Count - 1];
                adj[u].RemoveAt(adj[u].Count - 1);

                // Remove reverse edge
                for (int i = 0; i < adj[v].Count; i++)
                {
                    if (adj[v][i] == u)
                    {
                        adj[v].RemoveAt(i);
                        break;
                    }
                }

                stack.Push(v);
            }
            else
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit.ToArray();
    }
}
