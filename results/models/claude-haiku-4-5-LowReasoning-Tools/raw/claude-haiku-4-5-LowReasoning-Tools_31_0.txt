use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    let mut line = String::new();
    reader.read_line(&mut line).unwrap();

    let parts: Vec<&str> = line.split_whitespace().collect();
    let target: i64 = parts[1].parse().unwrap();

    line.clear();
    reader.read_line(&mut line).unwrap();

    let nums: Vec<i64> = line
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();

    if let Some(indices) = solve_subset_sum(&nums, target) {
        println!("YES");
        print!("[");
        for (i, &idx) in indices.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", idx);
        }
        println!("]");
    } else {
        println!("NO");
    }
}

fn solve_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();

    if n == 0 {
        return if target == 0 { Some(vec![]) } else { None };
    }

    if target == 0 {
        return Some(vec![]);
    }

    if n <= 20 {
        brute_force(nums, target)
    } else if n <= 1000 && all_positive(nums) {
        dp_with_reconstruction(nums, target)
    } else {
        dp_with_hash(nums, target)
    }
}

fn all_positive(nums: &[i64]) -> bool {
    nums.iter().all(|&x| x > 0)
}

fn brute_force(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    for mask in 0..(1u32 << n) {
        let mut sum = 0i64;
        let mut indices = vec![];
        for i in 0..n {
            if (mask & (1 << i)) != 0 {
                sum += nums[i];
                indices.push(i);
            }
        }
        if sum == target {
            return Some(indices);
        }
    }
    None
}

fn dp_with_reconstruction(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();

    if target < 0 {
        return None;
    }

    let target_usize = target as usize;
    let mut dp = vec![false; target_usize + 1];
    let mut parent = vec![None; target_usize + 1];

    dp[0] = true;

    for i in 0..n {
        let num = nums[i];
        if num > 0 && (num as usize) <= target_usize {
            for j in (num as usize..=target_usize).rev() {
                if dp[j - (num as usize)] {
                    dp[j] = true;
                    parent[j] = Some((j - (num as usize), i));
                }
            }
        }
    }

    if !dp[target_usize] {
        return None;
    }

    let mut indices = vec![];
    let mut current = target_usize;
    while let Some((prev, idx)) = parent[current] {
        indices.push(idx);
        current = prev;
    }

    indices.reverse();
    Some(indices)
}

fn dp_with_hash(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut current_states: HashSet<(i64, Vec<usize>)> = HashSet::new();
    current_states.insert((0, vec![]));

    for (i, &num) in nums.iter().enumerate() {
        let mut next_states = current_states.clone();
        for (sum, mut indices) in current_states {
            let new_sum = sum + num;
            if new_sum == target {
                indices.push(i);
                return Some(indices);
            }
            if new_sum < target {
                let mut new_indices = indices.clone();
                new_indices.push(i);
                next_states.insert((new_sum, new_indices));
            }
        }
        current_states = next_states;

        if current_states.len() > 100000 {
            break;
        }
    }

    for (sum, indices) in current_states {
        if sum == target {
            return Some(indices);
        }
    }

    None
}
