using System;
using System.Collections.Generic;
using System.Linq;

class QAPSolver
{
    static void Main()
    {
        int n = int.Parse(Console.ReadLine());

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }

        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }

        int[] bestPerm = SolveQAP(flow, dist, n);
        long bestCost = CalculateCost(flow, dist, bestPerm);

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static long CalculateCost(int[][] flow, int[][] dist, int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < perm.Length; i++)
        {
            for (int j = 0; j < perm.Length; j++)
            {
                cost += (long)flow[i][j] * dist[perm[i]][perm[j]];
            }
        }
        return cost;
    }

    static long CalculateSwapCostDelta(int[][] flow, int[][] dist, int[] perm, int i, int j)
    {
        int n = perm.Length;
        long delta = 0;

        int locI = perm[i];
        int locJ = perm[j];

        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;

            int locK = perm[k];

            delta -= (long)flow[i][k] * dist[locI][locK];
            delta += (long)flow[i][k] * dist[locJ][locK];

            delta -= (long)flow[k][i] * dist[locK][locI];
            delta += (long)flow[k][i] * dist[locK][locJ];

            delta -= (long)flow[j][k] * dist[locJ][locK];
            delta += (long)flow[j][k] * dist[locI][locK];

            delta -= (long)flow[k][j] * dist[locK][locJ];
            delta += (long)flow[k][j] * dist[locK][locI];
        }

        delta -= (long)flow[i][j] * dist[locI][locJ];
        delta += (long)flow[i][j] * dist[locJ][locI];

        delta -= (long)flow[j][i] * dist[locJ][locI];
        delta += (long)flow[j][i] * dist[locI][locJ];

        return delta;
    }

    static int[] SolveQAP(int[][] flow, int[][] dist, int n)
    {
        if (n <= 10)
        {
            return SolveSmall(flow, dist, n);
        }
        else if (n <= 500)
        {
            return SolveMedium(flow, dist, n);
        }
        else
        {
            return SolveLarge(flow, dist, n);
        }
    }

    static int[] SolveSmall(int[][] flow, int[][] dist, int n)
    {
        int[] bestPerm = Enumerable.Range(0, n).ToArray();
        long bestCost = CalculateCost(flow, dist, bestPerm);

        var rand = new Random(42);
        for (int trial = 0; trial < 20; trial++)
        {
            int[] perm = Enumerable.Range(0, n).ToArray();
            for (int i = n - 1; i > 0; i--)
            {
                int j = rand.Next(i + 1);
                int temp = perm[i];
                perm[i] = perm[j];
                perm[j] = temp;
            }

            perm = LocalSearch2Opt(flow, dist, perm);
            long cost = CalculateCost(flow, dist, perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestPerm = perm;
            }
        }

        return bestPerm;
    }

    static int[] SolveMedium(int[][] flow, int[][] dist, int n)
    {
        int[] perm = NearestNeighbor(flow, dist, n);
        perm = LocalSearch2Opt(flow, dist, perm);
        return perm;
    }

    static int[] SolveLarge(int[][] flow, int[][] dist, int n)
    {
        int[] perm = NearestNeighbor(flow, dist, n);
        perm = LocalSearch2OptLimited(flow, dist, perm, 10000);
        return perm;
    }

    static int[] NearestNeighbor(int[][] flow, int[][] dist, int n)
    {
        int[] perm = new int[n];
        bool[] used = new bool[n];

        perm[0] = 0;
        used[0] = true;

        for (int i = 1; i < n; i++)
        {
            int bestLoc = -1;
            long bestDelta = long.MaxValue;

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long delta = 0;
                for (int j = 0; j < i; j++)
                {
                    delta += (long)flow[i][j] * dist[loc][perm[j]];
                    delta += (long)flow[j][i] * dist[perm[j]][loc];
                }

                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestLoc = loc;
                }
            }

            perm[i] = bestLoc;
            used[bestLoc] = true;
        }

        return perm;
    }

    static int[] LocalSearch2Opt(int[][] flow, int[][] dist, int[] perm)
    {
        int n = perm.Length;
        bool improved = true;

        while (improved)
        {
            improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long delta = CalculateSwapCostDelta(flow, dist, perm, i, j);
                    if (delta < 0)
                    {
                        int temp = perm[i];
                        perm[i] = perm[j];
                        perm[j] = temp;
                        improved = true;
                    }
                }
            }
        }

        return perm;
    }

    static int[] LocalSearch2OptLimited(int[][] flow, int[][] dist, int[] perm, int maxIterations)
    {
        int n = perm.Length;
        int iterations = 0;
        bool improved = true;

        while (improved && iterations < maxIterations)
        {
            improved = false;

            for (int i = 0; i < n - 1 && iterations < maxIterations; i++)
            {
                for (int j = i + 1; j < n && iterations < maxIterations; j++)
                {
                    iterations++;

                    long delta = CalculateSwapCostDelta(flow, dist, perm, i, j);
                    if (delta < 0)
                    {
                        int temp = perm[i];
                        perm[i] = perm[j];
                        perm[j] = temp;
                        improved = true;
                    }
                }
            }
        }

        return perm;
    }
}
