OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Decorations for input/output
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Type declarations
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; UBO structure definition (std140 layout)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer types
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

; Constant definitions
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_25 = OpConstant %float 0.25
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_75 = OpConstant %float 0.75
%int_3 = OpConstant %int 3

; Global variables
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v4float Input
%out_color = OpVariable %ptr_Output_v4float Output

; Main function definition
%main_fn_type = OpTypeFunction %void
%main = OpFunction %void None %main_fn_type
%entry_block = OpLabel

; Load input values
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal
%color = OpLoad %v4float %in_color

; Normalize the interpolated normal
%normal_normalized = OpNormalize %v3float %normal

; Load light position from UBO (member 3)
%light_pos_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_3
%light_pos_v4 = OpLoad %v4float %light_pos_ptr

; Extract xyz components from light position (vec4 -> vec3)
%light_pos = OpVectorShuffle %v3float %light_pos_v4 %light_pos_v4 0 1 2

; Compute light direction from fragment to light
%light_dir_unnormalized = OpFSub %v3float %light_pos %worldPos
%light_direction = OpNormalize %v3float %light_dir_unnormalized

; Compute Lambertian diffuse: dot(N, L)
%dot_product = OpDot %float %normal_normalized %light_direction

; Clamp to [0, 1]: max(dot(N, L), 0)
%diffuse_term = OpFMax %float %dot_product %float_0

; Quantize diffuse into bands: thresholds 0.75/0.5/0.25 with values 1.0/0.7/0.4/0.2
; Nested select approach:
; if (d >= 0.75) return 1.0
; else if (d >= 0.5) return 0.7
; else if (d >= 0.25) return 0.4
; else return 0.2

%cmp_0_75 = OpFOrdGreaterThanEqual %bool %diffuse_term %float_0_75
%band_level_1 = OpSelect %float %cmp_0_75 %float_1 %float_0_7

%cmp_0_5 = OpFOrdGreaterThanEqual %bool %diffuse_term %float_0_5
%band_level_2 = OpSelect %float %cmp_0_5 %band_level_1 %float_0_4

%cmp_0_25 = OpFOrdGreaterThanEqual %bool %diffuse_term %float_0_25
%band_value = OpSelect %float %cmp_0_25 %band_level_2 %float_0_2

; Extract RGB from vertex color
%color_rgb = OpVectorShuffle %v3float %color %color 0 1 2

; Replicate band value across RGB channels
%band_rgb = OpCompositeConstruct %v3float %band_value %band_value %band_value

; Multiply vertex color by band value
%final_rgb = OpFMul %v3float %color_rgb %band_rgb

; Construct final output color with alpha = 1.0
%final_color = OpCompositeConstruct %v4float %final_rgb %float_1

; Store output color
OpStore %out_color %final_color

; End main function
OpReturn
OpFunctionEnd
