use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Clone, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Vec2 { x, y }
    }

    fn add(&self, other: &Vec2) -> Vec2 {
        Vec2 { x: self.x + other.x, y: self.y + other.y }
    }

    fn sub(&self, other: &Vec2) -> Vec2 {
        Vec2 { x: self.x - other.x, y: self.y - other.y }
    }

    fn mul(&self, scalar: f64) -> Vec2 {
        Vec2 { x: self.x * scalar, y: self.y * scalar }
    }

    fn len(&self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }

    fn normalized(&self) -> Vec2 {
        let len = self.len();
        if len < 1e-10 {
            Vec2::new(0.0, 0.0)
        } else {
            self.mul(1.0 / len)
        }
    }
}

struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(n: usize, edges: Vec<(usize, usize)>) -> Self {
        let mut adj = vec![Vec::new(); n];
        for &(u, v) in &edges {
            adj[u].push(v);
            adj[v].push(u);
        }
        Graph { n, edges, adj }
    }

    fn density(&self) -> f64 {
        if self.n <= 1 {
            return 0.0;
        }
        let max_edges = self.n * (self.n - 1) / 2;
        self.edges.len() as f64 / max_edges as f64
    }
}

fn fruchterman_reingold(graph: &Graph, max_iters: usize, temp: f64) -> Vec<Vec2> {
    let mut pos: Vec<Vec2> = (0..graph.n)
        .map(|_| Vec2::new(rand_f64() * 100.0, rand_f64() * 100.0))
        .collect();

    let k = (100.0 / (graph.n as f64).sqrt()).max(0.5);
    let mut temperature = temp;

    for _iter in 0..max_iters {
        let mut forces: Vec<Vec2> = vec![Vec2::new(0.0, 0.0); graph.n];

        // Repulsive forces
        for i in 0..graph.n {
            for j in (i + 1)..graph.n {
                let delta = pos[i].sub(&pos[j]);
                let dist = delta.len().max(0.1);
                let repel = delta.mul((k * k) / (dist * dist * dist));
                forces[i] = forces[i].add(&repel);
                forces[j] = forces[j].sub(&repel);
            }
        }

        // Attractive forces
        for &(u, v) in &graph.edges {
            let delta = pos[v].sub(&pos[u]);
            let dist = delta.len().max(0.1);
            let attract = delta.mul((dist * dist) / (k * 2.0));
            forces[u] = forces[u].add(&attract);
            forces[v] = forces[v].sub(&attract);
        }

        // Update positions
        for i in 0..graph.n {
            let disp_len = forces[i].len();
            let disp = forces[i].normalized().mul(disp_len.min(temperature));
            pos[i] = pos[i].add(&disp);
        }

        temperature *= 0.99;
    }

    pos
}

fn fast_layout(graph: &Graph) -> Vec<Vec2> {
    // For very large graphs, use a simpler grid-based layout
    let grid_size = (graph.n as f64).sqrt().ceil() as usize;
    let mut pos = vec![Vec2::new(0.0, 0.0); graph.n];

    for i in 0..graph.n {
        let x = (i % grid_size) as f64 * 100.0;
        let y = (i / grid_size) as f64 * 100.0;
        pos[i] = Vec2::new(x, y);
    }

    // Quick force-directed refinement with fewer iterations
    let k = (100.0 / (graph.n as f64).sqrt()).max(0.5);
    let iters = (100.0 / (graph.n as f64).log10().max(1.0)) as usize;

    for _ in 0..iters {
        let mut forces: Vec<Vec2> = vec![Vec2::new(0.0, 0.0); graph.n];

        // Only repel nearby nodes
        for i in 0..graph.n {
            for j in (i + 1)..graph.n.min(i + 50) {
                let delta = pos[i].sub(&pos[j]);
                let dist = delta.len().max(0.1);
                if dist < 200.0 {
                    let repel = delta.mul((k * k) / (dist * dist * dist));
                    forces[i] = forces[i].add(&repel);
                    forces[j] = forces[j].sub(&repel);
                }
            }
        }

        // Attract connected nodes
        for &(u, v) in &graph.edges {
            let delta = pos[v].sub(&pos[u]);
            let dist = delta.len().max(0.1);
            let attract = delta.mul((dist * dist) / (k * 2.0));
            forces[u] = forces[u].add(&attract);
            forces[v] = forces[v].sub(&attract);
        }

        for i in 0..graph.n {
            let disp = forces[i].mul(0.01);
            pos[i] = pos[i].add(&disp);
        }
    }

    pos
}

fn solve_layout(graph: &Graph) -> Vec<Vec2> {
    match graph.n {
        0..=100 => {
            fruchterman_reingold(graph, 300, 50.0)
        }
        101..=1000 => {
            fruchterman_reingold(graph, 150, 30.0)
        }
        _ => {
            fast_layout(graph)
        }
    }
}

// Simple PRNG for reproducibility
static mut RNG_STATE: u64 = 12345;

fn rand_f64() -> f64 {
    unsafe {
        RNG_STATE = RNG_STATE.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        ((RNG_STATE >> 32) as f64) / 4294967296.0
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    let mut edges = Vec::new();
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        if u != v {
            edges.push((u, v));
        }
    }

    let graph = Graph::new(n, edges);
    let positions = solve_layout(&graph);

    println!("{}", n);
    for pos in positions {
        println!("{} {}", pos.x, pos.y);
    }
}
