#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <set>
#include <map>
#include <array>
#include <unordered_map>

using namespace std;

constexpr double EPS = 1e-10;

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x); }
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const {
        double len = length();
        return len > EPS ? Vec3(x / len, y / len, z / len) : Vec3(0, 0, 0);
    }
    double distance(const Vec3& v) const { return (*this - v).length(); }
};

struct Mesh {
    vector<Vec3> vertices;
    vector<array<int, 3>> faces;
};

Mesh readMesh(int numVerts, int numFaces) {
    Mesh mesh;
    mesh.vertices.resize(numVerts);
    mesh.faces.resize(numFaces);

    for (int i = 0; i < numVerts; i++) {
        cin >> mesh.vertices[i].x >> mesh.vertices[i].y >> mesh.vertices[i].z;
    }

    for (int i = 0; i < numFaces; i++) {
        cin >> mesh.faces[i][0] >> mesh.faces[i][1] >> mesh.faces[i][2];
    }

    return mesh;
}

bool rayTriangleIntersect(const Vec3& rayOrigin, const Vec3& rayDir,
                          const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    const double EPSILON = 1e-8;
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 h = rayDir.cross(edge2);
    double a = edge1.dot(h);

    if (fabs(a) < EPSILON) return false;

    double f = 1.0 / a;
    Vec3 s = rayOrigin - v0;
    double u = f * s.dot(h);

    if (u < -EPSILON || u > 1.0 + EPSILON) return false;

    Vec3 q = s.cross(edge1);
    double v = f * rayDir.dot(q);

    if (v < -EPSILON || u + v > 1.0 + EPSILON) return false;

    t = f * edge2.dot(q);
    return t > EPSILON;
}

bool isPointInMesh(const Vec3& point, const Mesh& mesh) {
    Vec3 rayDir(1.0, 0.0, 0.0);
    int crossings = 0;

    for (const auto& face : mesh.faces) {
        double t;
        if (rayTriangleIntersect(point, rayDir,
                                mesh.vertices[face[0]],
                                mesh.vertices[face[1]],
                                mesh.vertices[face[2]], t)) {
            crossings++;
        }
    }

    return (crossings % 2) == 1;
}

Vec3 getTriangleCenter(const Mesh& mesh, const array<int, 3>& face) {
    Vec3 center = (mesh.vertices[face[0]] + mesh.vertices[face[1]] + mesh.vertices[face[2]]) * (1.0 / 3.0);
    return center;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA = readMesh(VA, FA);
    Mesh meshB = readMesh(VB, FB);

    vector<Vec3> resultVertices;
    vector<array<int, 3>> resultFaces;
    map<pair<pair<double, double>, double>, int> vertexMap;

    auto addVertex = [&](const Vec3& v) -> int {
        auto key = make_pair(make_pair(round(v.x * 1e6) / 1e6, round(v.y * 1e6) / 1e6),
                            round(v.z * 1e6) / 1e6);
        if (vertexMap.find(key) == vertexMap.end()) {
            vertexMap[key] = resultVertices.size();
            resultVertices.push_back(v);
        }
        return vertexMap[key];
    };

    // Add vertices from A and B
    map<int, int> mapA, mapB;
    for (int i = 0; i < VA; i++) {
        mapA[i] = addVertex(meshA.vertices[i]);
    }
    for (int i = 0; i < VB; i++) {
        mapB[i] = addVertex(meshB.vertices[i]);
    }

    // Include faces from A that are outside B
    for (const auto& face : meshA.faces) {
        Vec3 center = getTriangleCenter(meshA, face);
        if (!isPointInMesh(center, meshB)) {
            resultFaces.push_back({
                mapA[face[0]], mapA[face[1]], mapA[face[2]]
            });
        }
    }

    // Include faces from B that are outside A
    for (const auto& face : meshB.faces) {
        Vec3 center = getTriangleCenter(meshB, face);
        if (!isPointInMesh(center, meshA)) {
            resultFaces.push_back({
                mapB[face[0]], mapB[face[1]], mapB[face[2]]
            });
        }
    }

    // Output result
    cout << resultVertices.size() << " " << resultFaces.size() << "\n";
    for (const auto& v : resultVertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : resultFaces) {
        cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }

    return 0;
}
