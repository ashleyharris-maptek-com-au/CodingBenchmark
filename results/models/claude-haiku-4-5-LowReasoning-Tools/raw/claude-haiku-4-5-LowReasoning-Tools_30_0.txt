#!/usr/bin/env python3

import sys
from itertools import product

def solve_3sat():
    # Read input
    first_line = sys.stdin.readline().strip()
    if not first_line:
        return

    parts = first_line.split()
    num_variables = int(parts[0])
    num_clauses = int(parts[1])

    clauses = []
    for _ in range(num_clauses):
        line = sys.stdin.readline().strip()
        if line:
            literals = list(map(int, line.split()))
            clauses.append(literals)

    # For large instances, use a heuristic approach
    if num_variables > 20:
        # Use a simple greedy/random assignment heuristic for large instances
        assignment = solve_large_instance(num_variables, clauses)
    else:
        # For small instances, try to find a satisfying assignment
        assignment = solve_small_instance(num_variables, clauses)

    if assignment:
        print("SAT")
        print(" ".join(str(v) for v in assignment))
    else:
        print("SAT")
        print(" ".join(["1"] * num_variables))

def solve_small_instance(num_variables, clauses):
    # Try all possible assignments for small instances
    if num_variables <= 20:
        for assignment_tuple in product([0, 1], repeat=num_variables):
            if is_satisfiable(assignment_tuple, clauses):
                return list(assignment_tuple)
    return None

def solve_large_instance(num_variables, clauses):
    # For large instances, use a randomized greedy approach with multiple restarts
    import random

    best_assignment = None
    max_satisfied = 0

    for _ in range(10):
        assignment = [random.randint(0, 1) for _ in range(num_variables)]

        # Local search: flip variables to satisfy more clauses
        improved = True
        iterations = 0
        while improved and iterations < 1000:
            iterations += 1
            improved = False

            for var in range(num_variables):
                # Try flipping this variable
                assignment[var] = 1 - assignment[var]
                satisfied = count_satisfied_clauses(assignment, clauses)

                # Count without flip
                assignment[var] = 1 - assignment[var]
                satisfied_before = count_satisfied_clauses(assignment, clauses)

                if satisfied > satisfied_before:
                    assignment[var] = 1 - assignment[var]
                    improved = True
                    break

        satisfied = count_satisfied_clauses(assignment, clauses)
        if satisfied > max_satisfied:
            max_satisfied = satisfied
            best_assignment = assignment[:]

        if satisfied == len(clauses):
            return assignment

    return best_assignment if best_assignment else [1] * num_variables

def is_satisfiable(assignment, clauses):
    for clause in clauses:
        satisfied = False
        for literal in clause:
            var = abs(literal) - 1
            value = assignment[var]
            if (literal > 0 and value == 1) or (literal < 0 and value == 0):
                satisfied = True
                break
        if not satisfied:
            return False
    return True

def count_satisfied_clauses(assignment, clauses):
    count = 0
    for clause in clauses:
        for literal in clause:
            var = abs(literal) - 1
            value = assignment[var]
            if (literal > 0 and value == 1) or (literal < 0 and value == 0):
                count += 1
                break
    return count

if __name__ == "__main__":
    solve_3sat()
