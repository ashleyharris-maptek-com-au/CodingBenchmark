const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

// Convert grid strings to 2D array for easier access
const gridMap = grid.map(row => row.split(''));

// Simulate fire spread with given firebreaks
function simulateFire(firebreaks) {
  const burned = new Set();
  const firebreakSet = new Set(firebreaks.map(([x, y]) => `${x},${y}`));

  // BFS to spread fire
  const queue = [...ignitions];
  for (const [x, y] of ignitions) {
    burned.add(`${x},${y}`);
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];

    // Try spreading in 4 directions
    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;
      const key = `${nx},${ny}`;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
          !burned.has(key) && !firebreakSet.has(key) &&
          gridMap[ny][nx] === '.') {
        burned.add(key);
        queue.push([nx, ny]);
      }
    }
  }

  return burned;
}

// Calculate value lost if given cells are burned
function calculateDamage(burned) {
  let damage = 0;
  for (const town of towns) {
    if (burned.has(`${town.x},${town.y}`)) {
      damage += town.value;
    }
  }
  return damage;
}

// Get all burnable cells with priority score based on proximity to towns
function getCellScores(burned) {
  const scores = new Map();

  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      if (gridMap[y][x] === '.' && !burned.has(`${x},${y}`)) {
        const key = `${x},${y}`;
        let score = 0;

        // Score based on distance to high-value towns
        // A cell is valuable if blocking it prevents fire from reaching towns
        for (const town of towns) {
          const dist = Math.abs(x - town.x) + Math.abs(y - town.y);
          if (dist <= 20) { // Consider nearby towns
            score += town.value / (dist + 1);
          }
        }

        if (score > 0) {
          scores.set(key, score);
        }
      }
    }
  }

  return scores;
}

// Greedy approach: repeatedly pick the cell that saves the most value
function findOptimalFirebreaks() {
  let firebreaks = [];
  let baseDamage = calculateDamage(simulateFire(firebreaks));

  for (let i = 0; i < cutBudget; i++) {
    let bestCell = null;
    let bestSaving = 0;
    let bestDamage = baseDamage;

    const burned = simulateFire(firebreaks);
    const candidates = getCellScores(burned);

    // Try each candidate cell
    for (const [key, score] of candidates) {
      if (score > 0) {
        const [x, y] = key.split(',').map(Number);
        const testBreaks = [...firebreaks, [x, y]];
        const newDamage = calculateDamage(simulateFire(testBreaks));
        const saving = baseDamage - newDamage;

        if (saving > bestSaving) {
          bestSaving = saving;
          bestCell = [x, y];
          bestDamage = newDamage;
        }
      }
    }

    if (bestCell && bestSaving > 0) {
      firebreaks.push(bestCell);
      baseDamage = bestDamage;
    } else {
      // No more beneficial cuts, stop
      break;
    }
  }

  return firebreaks;
}

const firebreaks = findOptimalFirebreaks();
