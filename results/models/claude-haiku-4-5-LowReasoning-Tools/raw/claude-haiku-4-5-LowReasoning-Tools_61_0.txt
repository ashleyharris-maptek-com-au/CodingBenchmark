declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: string;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

const inputData: Input = data;

function distance(x1: number, y1: number, x2: number, y2: number): number {
  return Math.sqrt((x1 - x2) ** 2 + (y1 - y2) ** 2);
}

function nearestNeighbor(depot: [number, number], packages: Package[]): Package[] {
  if (packages.length === 0) return [];

  const route: Package[] = [];
  const visited = new Set<string>();
  let x = depot[0];
  let y = depot[1];

  while (visited.size < packages.length) {
    let nearest: Package | null = null;
    let nearestDist = Infinity;

    for (const pkg of packages) {
      if (!visited.has(pkg.id)) {
        const dist = distance(x, y, pkg.x, pkg.y);
        if (dist < nearestDist) {
          nearestDist = dist;
          nearest = pkg;
        }
      }
    }

    if (!nearest) break;
    route.push(nearest);
    visited.add(nearest.id);
    x = nearest.x;
    y = nearest.y;
  }

  return route;
}

function solveVRP(depot: [number, number], packages: Package[], truckCount: number): (string | number)[][] {
  if (packages.length === 0) return [];

  const numTrucks = Math.min(truckCount, packages.length);

  // Group packages by angle from depot for spatial coherence
  const packagesWithAngle = packages.map(p => ({
    ...p,
    angle: Math.atan2(p.y - depot[1], p.x - depot[0])
  }));

  packagesWithAngle.sort((a, b) => a.angle - b.angle);

  const routes: (string | number)[][] = [];
  const groupSize = Math.ceil(packages.length / numTrucks);

  for (let i = 0; i < packages.length; i += groupSize) {
    const group = packagesWithAngle.slice(i, Math.min(i + groupSize, packages.length));
    if (group.length > 0) {
      const route = nearestNeighbor(depot, group);
      routes.push(route.map(p => p.id));
    }
  }

  return routes;
}

const routes = solveVRP(inputData.depot, inputData.packages, inputData.truckCount);
