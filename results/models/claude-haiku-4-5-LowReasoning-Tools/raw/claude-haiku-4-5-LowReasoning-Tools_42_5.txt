OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpName %main "main"

OpDecorate %gl_in BuiltIn gl_in
OpDecorate %inWorldPos Location 0
OpDecorate %inColor Location 2
OpDecorate %gl_out Block
OpMemberDecorate %gl_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_neg_0_015 = OpConstant %float -0.015
%float_neg_0_05 = OpConstant %float -0.05
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

%dir = OpFSub %v3float %p1_xyz %p0_xyz
%dir_xy = OpVectorShuffle %v2float %dir %dir 0 1
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y

%perp_2d = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%dot_self = OpDot %float %perp_2d %perp_2d
%perp_len = OpExtInst %float %1 OpSqrt %dot_self
%inv_len = OpFDiv %float %float_1 %perp_len
%perp_2d_norm = OpVectorTimesScalar %v2float %perp_2d %inv_len
%perp = OpCompositeConstruct %v3float %perp_2d_norm %float_0

%diff = OpFSub %v3float %p1_xyz %p0_xyz
%scaled = OpVectorTimesScalar %v3float %diff %float_0_75
%mid = OpFAdd %v3float %p0_xyz %scaled

%perp_thick = OpVectorTimesScalar %v3float %perp %float_0_015
%perp_thick_neg = OpVectorTimesScalar %v3float %perp %float_neg_0_015

%out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%shaft_1_pos = OpFAdd %v3float %p0_xyz %perp_thick
%shaft_1 = OpCompositeConstruct %v4float %shaft_1_pos %float_1
OpStore %out_ptr %shaft_1
OpStore %outColor %c0
OpEmitVertex

%shaft_2_pos = OpFAdd %v3float %p0_xyz %perp_thick_neg
%shaft_2 = OpCompositeConstruct %v4float %shaft_2_pos %float_1
OpStore %out_ptr %shaft_2
OpStore %outColor %c0
OpEmitVertex

%shaft_3_pos = OpFAdd %v3float %mid %perp_thick
%shaft_3 = OpCompositeConstruct %v4float %shaft_3_pos %float_1
OpStore %out_ptr %shaft_3
OpStore %outColor %c0
OpEmitVertex

%shaft_4_pos = OpFAdd %v3float %mid %perp_thick_neg
%shaft_4 = OpCompositeConstruct %v4float %shaft_4_pos %float_1
OpStore %out_ptr %shaft_4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

%perp_arrow = OpVectorTimesScalar %v3float %perp %float_0_05
%perp_arrow_neg = OpVectorTimesScalar %v3float %perp %float_neg_0_05

%arrow_1_pos = OpFAdd %v3float %mid %perp_arrow
%arrow_1 = OpCompositeConstruct %v4float %arrow_1_pos %float_1
OpStore %out_ptr %arrow_1
OpStore %outColor %c1
OpEmitVertex

%arrow_2_pos = OpFAdd %v3float %mid %perp_arrow_neg
%arrow_2 = OpCompositeConstruct %v4float %arrow_2_pos %float_1
OpStore %out_ptr %arrow_2
OpStore %outColor %c1
OpEmitVertex

%arrow_3 = OpCompositeConstruct %v4float %p1_xyz %float_1
OpStore %out_ptr %arrow_3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
