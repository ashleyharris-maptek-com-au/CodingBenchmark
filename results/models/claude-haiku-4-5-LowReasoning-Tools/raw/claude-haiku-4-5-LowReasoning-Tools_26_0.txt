use std::io::{self, Read, Write, BufReader, BufWriter};
use std::cmp;

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();

    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    // Read num_points and num_clusters
    let mut header = [0u8; 8];
    reader.read_exact(&mut header)?;

    let num_points = u32::from_le_bytes([header[0], header[1], header[2], header[3]]) as usize;
    let num_clusters = u32::from_le_bytes([header[4], header[5], header[6], header[7]]) as usize;

    let num_clusters = cmp::min(num_clusters, num_points);

    // Choose algorithm based on dataset size
    let cluster_assignments = if num_points <= 10_000_000 {
        kmeans_full(num_points, num_clusters, &mut reader)?
    } else {
        kmeans_streaming(num_points, num_clusters, &mut reader)?
    };

    // Write output
    for &cluster in &cluster_assignments {
        let bytes = (cluster as u32).to_le_bytes();
        writer.write_all(&bytes)?;
    }

    writer.flush()?;
    Ok(())
}

// Full k-means for small datasets that fit in memory
fn kmeans_full(num_points: usize, num_clusters: usize, reader: &mut BufReader<io::StdinLock>) -> io::Result<Vec<u32>> {
    let mut buffer = vec![0u8; num_points * 12];
    reader.read_exact(&mut buffer)?;

    let mut points = Vec::with_capacity(num_points);
    for chunk in buffer.chunks(12) {
        let x = f32::from_le_bytes([chunk[0], chunk[1], chunk[2], chunk[3]]);
        let y = f32::from_le_bytes([chunk[4], chunk[5], chunk[6], chunk[7]]);
        let z = f32::from_le_bytes([chunk[8], chunk[9], chunk[10], chunk[11]]);
        points.push([x, y, z]);
    }

    kmeans_algorithm(&points, num_clusters, 100)
}

// Streaming k-means with limited memory for large datasets
fn kmeans_streaming(num_points: usize, num_clusters: usize, reader: &mut BufReader<io::StdinLock>) -> io::Result<Vec<u32>> {
    let batch_size = cmp::min(500_000, cmp::max(10_000, num_points / 50));

    let mut centroids = vec![[0.0f32; 3]; num_clusters];
    let mut assignments = vec![0u32; num_points];
    let mut cluster_counts = vec![0usize; num_clusters];
    let mut cluster_sums = vec![[0.0f32; 3]; num_clusters];

    // Initialize centroids sparsely
    let step = cmp::max(1, num_points / num_clusters);
    let mut buffer = vec![0u8; batch_size * 12];
    let mut point_idx = 0;
    let mut init_idx = 0;

    // First pass: initialize centroids
    while point_idx < num_points && init_idx < num_clusters {
        let remaining = num_points - point_idx;
        let current_batch_size = cmp::min(batch_size, remaining);
        let bytes_to_read = current_batch_size * 12;

        match reader.read_exact(&mut buffer[..bytes_to_read]) {
            Ok(_) => {},
            Err(ref e) if e.kind() == io::ErrorKind::UnexpectedEof => break,
            Err(e) => return Err(e),
        }

        for (i, chunk) in buffer[..bytes_to_read].chunks(12).enumerate() {
            if (point_idx + i) % step == 0 && init_idx < num_clusters {
                let x = f32::from_le_bytes([chunk[0], chunk[1], chunk[2], chunk[3]]);
                let y = f32::from_le_bytes([chunk[4], chunk[5], chunk[6], chunk[7]]);
                let z = f32::from_le_bytes([chunk[8], chunk[9], chunk[10], chunk[11]]);
                centroids[init_idx] = [x, y, z];
                init_idx += 1;
            }
        }

        point_idx += current_batch_size;
    }

    // Second pass: assign all points (single pass streaming)
    point_idx = 0;

    while point_idx < num_points {
        let remaining = num_points - point_idx;
        let current_batch_size = cmp::min(batch_size, remaining);
        let bytes_to_read = current_batch_size * 12;

        match reader.read_exact(&mut buffer[..bytes_to_read]) {
            Ok(_) => {},
            Err(ref e) if e.kind() == io::ErrorKind::UnexpectedEof => break,
            Err(e) => return Err(e),
        }

        for (i, chunk) in buffer[..bytes_to_read].chunks(12).enumerate() {
            let x = f32::from_le_bytes([chunk[0], chunk[1], chunk[2], chunk[3]]);
            let y = f32::from_le_bytes([chunk[4], chunk[5], chunk[6], chunk[7]]);
            let z = f32::from_le_bytes([chunk[8], chunk[9], chunk[10], chunk[11]]);

            let cluster = find_nearest_centroid(&centroids, [x, y, z]);
            assignments[point_idx + i] = cluster as u32;

            // Update centroid incrementally
            cluster_sums[cluster][0] += x;
            cluster_sums[cluster][1] += y;
            cluster_sums[cluster][2] += z;
            cluster_counts[cluster] += 1;
        }

        point_idx += current_batch_size;

        // Periodically update centroids to track data distribution
        if point_idx % (batch_size * 10) == 0 {
            for c in 0..num_clusters {
                if cluster_counts[c] > 0 {
                    let count = cluster_counts[c] as f32;
                    centroids[c][0] = cluster_sums[c][0] / count;
                    centroids[c][1] = cluster_sums[c][1] / count;
                    centroids[c][2] = cluster_sums[c][2] / count;
                }
            }
            cluster_counts.fill(0);
            for c in 0..num_clusters {
                cluster_sums[c] = [0.0, 0.0, 0.0];
            }
        }
    }

    Ok(assignments)
}

fn kmeans_algorithm(points: &[[f32; 3]], num_clusters: usize, max_iters: usize) -> io::Result<Vec<u32>> {
    let num_points = points.len();

    if num_points == 0 || num_clusters == 0 {
        return Ok(Vec::new());
    }

    let num_clusters = cmp::min(num_clusters, num_points);

    // Initialize centroids: k-means++ style (pick distant points)
    let mut centroids = vec![[0.0f32; 3]; num_clusters];
    let mut used = vec![false; num_points];

    // Pick first centroid randomly (just use first point for determinism)
    centroids[0] = points[0];
    used[0] = true;

    // Pick remaining centroids that are far from existing ones
    for c in 1..num_clusters {
        let mut max_min_dist = 0.0f32;
        let mut best_idx = 0;

        for (i, &point) in points.iter().enumerate() {
            if used[i] {
                continue;
            }

            let mut min_dist = f32::INFINITY;
            for j in 0..c {
                let dx = point[0] - centroids[j][0];
                let dy = point[1] - centroids[j][1];
                let dz = point[2] - centroids[j][2];
                let dist = dx * dx + dy * dy + dz * dz;
                min_dist = min_dist.min(dist);
            }

            if min_dist > max_min_dist {
                max_min_dist = min_dist;
                best_idx = i;
            }
        }

        centroids[c] = points[best_idx];
        used[best_idx] = true;
    }

    let mut assignments = vec![0u32; num_points];

    // K-means iterations
    for _iter in 0..max_iters {
        // Assign points to nearest centroid
        for (i, &point) in points.iter().enumerate() {
            assignments[i] = find_nearest_centroid(&centroids, point) as u32;
        }

        // Update centroids
        let mut new_centroids = vec![[0.0f32; 3]; num_clusters];
        let mut counts = vec![0u32; num_clusters];

        for (i, &point) in points.iter().enumerate() {
            let cluster = assignments[i] as usize;
            new_centroids[cluster][0] += point[0];
            new_centroids[cluster][1] += point[1];
            new_centroids[cluster][2] += point[2];
            counts[cluster] += 1;
        }

        // Normalize and handle empty clusters
        for c in 0..num_clusters {
            if counts[c] > 0 {
                let count = counts[c] as f32;
                new_centroids[c][0] /= count;
                new_centroids[c][1] /= count;
                new_centroids[c][2] /= count;
            } else {
                new_centroids[c] = centroids[c];
            }
        }

        // Check convergence
        let mut converged = true;
        for c in 0..num_clusters {
            let dx = new_centroids[c][0] - centroids[c][0];
            let dy = new_centroids[c][1] - centroids[c][1];
            let dz = new_centroids[c][2] - centroids[c][2];
            if dx * dx + dy * dy + dz * dz > 1e-6 {
                converged = false;
                break;
            }
        }

        centroids = new_centroids;

        if converged {
            break;
        }
    }

    // Final assignment
    for (i, &point) in points.iter().enumerate() {
        assignments[i] = find_nearest_centroid(&centroids, point) as u32;
    }

    // Ensure all clusters are used
    let mut used_clusters = vec![false; num_clusters];
    for &cluster in &assignments {
        used_clusters[cluster as usize] = true;
    }

    let mut unused_clusters: Vec<usize> = (0..num_clusters)
        .filter(|c| !used_clusters[*c])
        .collect();

    if !unused_clusters.is_empty() {
        // Assign furthest points from their clusters to unused clusters
        for (i, &point) in points.iter().enumerate() {
            if unused_clusters.is_empty() {
                break;
            }

            let current_cluster = assignments[i] as usize;
            let dist_to_centroid = {
                let dx = point[0] - centroids[current_cluster][0];
                let dy = point[1] - centroids[current_cluster][1];
                let dz = point[2] - centroids[current_cluster][2];
                dx * dx + dy * dy + dz * dz
            };

            if dist_to_centroid > 1.0 {
                let new_cluster = unused_clusters.pop().unwrap();
                assignments[i] = new_cluster as u32;
                centroids[new_cluster] = point;
            }
        }
    }

    Ok(assignments)
}

fn find_nearest_centroid(centroids: &[[f32; 3]], point: [f32; 3]) -> usize {
    let mut min_dist = f32::INFINITY;
    let mut best = 0;

    for (c, &centroid) in centroids.iter().enumerate() {
        let dx = point[0] - centroid[0];
        let dy = point[1] - centroid[1];
        let dz = point[2] - centroid[2];
        let dist = dx * dx + dy * dy + dz * dz;

        if dist < min_dist {
            min_dist = dist;
            best = c;
        }
    }

    best
}
