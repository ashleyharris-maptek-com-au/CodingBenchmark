use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Debug, Clone)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    position: [f64; 3],
    values: Vec<f64>,
}

#[derive(Debug, Clone)]
struct Anomaly {
    hole_id: usize,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    // Read header
    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    let _num_holes: usize = parts[0].parse().unwrap();
    let _num_properties: usize = parts[1].parse().unwrap();

    // Read property names
    line.clear();
    reader.read_line(&mut line).unwrap();
    let property_names: Vec<String> = line.trim().split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // Parse all samples
    let mut samples = Vec::new();

    loop {
        line.clear();
        if reader.read_line(&mut line).unwrap() == 0 { break; }

        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        if parts.is_empty() { break; }

        let hole_id: usize = parts[0].parse().unwrap();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let _length: f64 = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        // Normalize direction vector
        let dir_len = (dir_x*dir_x + dir_y*dir_y + dir_z*dir_z).sqrt();
        let dir = if dir_len > 0.0 {
            [dir_x/dir_len, dir_y/dir_len, dir_z/dir_len]
        } else {
            [0.0, 0.0, 1.0]
        };

        // Read samples for this hole
        for sample_idx in 0..num_samples {
            line.clear();
            if reader.read_line(&mut line).unwrap() == 0 { break; }

            let parts: Vec<&str> = line.trim().split_whitespace().collect();
            if parts.is_empty() { break; }

            let depth: f64 = parts[0].parse().unwrap();

            // Calculate 3D position along the drillhole
            let pos = [
                start_x + dir[0] * depth,
                start_y + dir[1] * depth,
                start_z + dir[2] * depth,
            ];

            let values: Vec<f64> = parts[1..].iter()
                .map(|s| s.parse().unwrap_or(0.0))
                .collect();

            samples.push(Sample {
                hole_id,
                sample_idx,
                position: pos,
                values,
            });
        }
    }

    // Detect anomalies using adaptive strategy
    let mut anomalies = Vec::new();

    if samples.len() > 5000 {
        detect_anomalies_indexed(&samples, &property_names, &mut anomalies);
    } else {
        detect_anomalies_brute_force(&samples, &property_names, &mut anomalies);
    }

    // Sort by confidence descending
    anomalies.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    // Output
    println!("{}", anomalies.len());
    for anomaly in anomalies.iter().take(10000) {
        println!("{} {} {} {:.2}", anomaly.hole_id, anomaly.sample_idx, anomaly.property_name, anomaly.confidence);
    }
}

fn detect_anomalies_brute_force(
    samples: &[Sample],
    property_names: &[String],
    anomalies: &mut Vec<Anomaly>,
) {
    for (i, sample) in samples.iter().enumerate() {
        let mut neighbors = Vec::new();

        for (j, other) in samples.iter().enumerate() {
            if i == j { continue; }

            let dist_sq = distance_squared(sample.position, other.position);
            if dist_sq < 625.0 {
                neighbors.push(other);
            }
        }

        check_anomalies(sample, &neighbors, property_names, anomalies);
    }
}

fn detect_anomalies_indexed(
    samples: &[Sample],
    property_names: &[String],
    anomalies: &mut Vec<Anomaly>,
) {
    let grid_size = 50.0;
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();

    for (idx, sample) in samples.iter().enumerate() {
        let gx = (sample.position[0] / grid_size).floor() as i32;
        let gy = (sample.position[1] / grid_size).floor() as i32;
        let gz = (sample.position[2] / grid_size).floor() as i32;
        grid.entry((gx, gy, gz)).or_insert_with(Vec::new).push(idx);
    }

    for (i, sample) in samples.iter().enumerate() {
        let gx = (sample.position[0] / grid_size).floor() as i32;
        let gy = (sample.position[1] / grid_size).floor() as i32;
        let gz = (sample.position[2] / grid_size).floor() as i32;

        let mut neighbors = Vec::new();

        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    if let Some(cell_samples) = grid.get(&(gx + dx, gy + dy, gz + dz)) {
                        for &j in cell_samples {
                            if i == j { continue; }
                            let dist_sq = distance_squared(sample.position, samples[j].position);
                            if dist_sq < 625.0 {
                                neighbors.push(&samples[j]);
                            }
                        }
                    }
                }
            }
        }

        check_anomalies(sample, &neighbors, property_names, anomalies);
    }
}

fn distance_squared(a: [f64; 3], b: [f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx*dx + dy*dy + dz*dz
}

fn check_anomalies(
    sample: &Sample,
    neighbors: &[&Sample],
    property_names: &[String],
    anomalies: &mut Vec<Anomaly>,
) {
    if neighbors.is_empty() { return; }

    for (prop_idx, &value) in sample.values.iter().enumerate() {
        let mut neighbor_values = Vec::new();
        for neighbor in neighbors {
            if prop_idx < neighbor.values.len() {
                neighbor_values.push(neighbor.values[prop_idx]);
            }
        }

        if neighbor_values.is_empty() { continue; }

        let mean = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
        let variance = neighbor_values.iter()
            .map(|v| (v - mean).powi(2))
            .sum::<f64>() / neighbor_values.len() as f64;
        let std_dev = variance.sqrt();

        if std_dev > 1e-10 {
            let z_score = (value - mean).abs() / std_dev;
            if z_score > 2.5 {
                let confidence = (z_score - 2.5).min(10.0);
                anomalies.push(Anomaly {
                    hole_id: sample.hole_id,
                    sample_idx: sample.sample_idx,
                    property_name: property_names[prop_idx].clone(),
                    confidence,
                });
            }
        }
    }
}
