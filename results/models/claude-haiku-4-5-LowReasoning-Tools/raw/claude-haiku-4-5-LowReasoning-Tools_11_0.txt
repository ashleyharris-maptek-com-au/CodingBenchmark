use std::io::{self, BufRead};

#[derive(Clone, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Vec2 { x, y }
    }

    fn add(&self, other: &Vec2) -> Vec2 {
        Vec2 {
            x: self.x + other.x,
            y: self.y + other.y,
        }
    }

    fn sub(&self, other: &Vec2) -> Vec2 {
        Vec2 {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }

    fn rotate(&self, angle: f64) -> Vec2 {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Vec2 {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn cross(&self, other: &Vec2) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn dot(&self, other: &Vec2) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn dist_sq(&self, other: &Vec2) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        dx * dx + dy * dy
    }
}

struct Polygon {
    vertices: Vec<Vec2>,
}

impl Polygon {
    fn new(vertices: Vec<Vec2>) -> Self {
        Polygon { vertices }
    }

    fn translate(&self, offset: &Vec2) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|v| v.add(offset)).collect(),
        }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|v| v.rotate(angle)).collect(),
        }
    }

    fn transform(&self, offset: &Vec2, angle: f64) -> Polygon {
        self.rotate(angle).translate(offset)
    }

    fn point_in_polygon(&self, point: &Vec2) -> bool {
        let n = self.vertices.len();
        let mut crosses = 0;

        for i in 0..n {
            let a = &self.vertices[i];
            let b = &self.vertices[(i + 1) % n];

            if (a.y <= point.y && point.y < b.y) || (b.y <= point.y && point.y < a.y) {
                let x_intersect = a.x + (point.y - a.y) * (b.x - a.x) / (b.y - a.y);
                if point.x < x_intersect {
                    crosses += 1;
                }
            }
        }

        crosses % 2 == 1
    }

    fn point_in_polygon_or_boundary(&self, point: &Vec2) -> bool {
        if self.point_in_polygon(point) {
            return true;
        }

        for i in 0..self.vertices.len() {
            let a = &self.vertices[i];
            let b = &self.vertices[(i + 1) % self.vertices.len()];
            let ab = b.sub(a);
            let ap = point.sub(a);
            let t = ap.dot(&ab) / ab.dot(&ab);
            if t >= 0.0 && t <= 1.0 {
                let closest = a.add(&Vec2::new(ab.x * t, ab.y * t));
                if (closest.x - point.x).abs() < 1e-9 && (closest.y - point.y).abs() < 1e-9 {
                    return true;
                }
            }
        }
        false
    }

    fn fits_in(&self, container: &Polygon) -> bool {
        for v in &self.vertices {
            if !container.point_in_polygon_or_boundary(v) {
                return false;
            }
        }
        true
    }

    fn seg_intersect(p1: &Vec2, p2: &Vec2, p3: &Vec2, p4: &Vec2) -> bool {
        let d1 = p2.sub(p1);
        let d2 = p4.sub(p3);
        let d3 = p3.sub(p1);

        let cross1 = d1.cross(&d2);
        if cross1.abs() < 1e-10 {
            return false;
        }

        let t1 = d3.cross(&d2) / cross1;
        let t2 = d3.cross(&d1) / cross1;

        t1 > 1e-10 && t1 < 1.0 - 1e-10 && t2 > 1e-10 && t2 < 1.0 - 1e-10
    }

    fn overlaps(&self, other: &Polygon) -> bool {
        for i in 0..self.vertices.len() {
            for j in 0..other.vertices.len() {
                if Polygon::seg_intersect(
                    &self.vertices[i],
                    &self.vertices[(i + 1) % self.vertices.len()],
                    &other.vertices[j],
                    &other.vertices[(j + 1) % other.vertices.len()],
                ) {
                    return true;
                }
            }
        }

        for v in &self.vertices {
            if other.point_in_polygon(v) {
                return true;
            }
        }

        for v in &other.vertices {
            if self.point_in_polygon(v) {
                return true;
            }
        }

        false
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        stock_vertices.push(Vec2::new(x, y));
    }

    let stock_poly = Polygon::new(stock_vertices);

    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap().unwrap();
        let pv: usize = line.parse().unwrap();
        let mut verts = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let parts: Vec<&str> = line.split_whitespace().collect();
            let x: f64 = parts[0].parse().unwrap();
            let y: f64 = parts[1].parse().unwrap();
            verts.push(Vec2::new(x, y));
        }
        pieces.push(Polygon::new(verts));
    }

    let mut stocks: Vec<Vec<(usize, Vec2, f64)>> = vec![Vec::new()];
    let mut placed = vec![false; np];

    for piece_idx in 0..np {
        let mut placed_flag = false;

        for stock_idx in 0..stocks.len() {
            if placed_flag {
                break;
            }

            let piece = &pieces[piece_idx];
            let rotations = if np > 500 { 8 } else { 16 };

            for rot_idx in 0..rotations {
                if placed_flag {
                    break;
                }

                let angle = 2.0 * std::f64::consts::PI * (rot_idx as f64) / (rotations as f64);
                let rotated = piece.rotate(angle);

                let bbox_min_x = rotated.vertices.iter().map(|v| v.x).fold(f64::INFINITY, f64::min);
                let bbox_max_x = rotated.vertices.iter().map(|v| v.x).fold(f64::NEG_INFINITY, f64::max);
                let bbox_min_y = rotated.vertices.iter().map(|v| v.y).fold(f64::INFINITY, f64::min);
                let bbox_max_y = rotated.vertices.iter().map(|v| v.y).fold(f64::NEG_INFINITY, f64::max);

                let stock_bbox_min_x =
                    stock_poly.vertices.iter().map(|v| v.x).fold(f64::INFINITY, f64::min);
                let stock_bbox_max_x =
                    stock_poly.vertices.iter().map(|v| v.x).fold(f64::NEG_INFINITY, f64::max);
                let stock_bbox_min_y =
                    stock_poly.vertices.iter().map(|v| v.y).fold(f64::INFINITY, f64::min);
                let stock_bbox_max_y =
                    stock_poly.vertices.iter().map(|v| v.y).fold(f64::NEG_INFINITY, f64::max);

                let width = bbox_max_x - bbox_min_x;
                let height = bbox_max_y - bbox_min_y;

                for tx_offset in [0, 10, -10, 20, -20] {
                    for ty_offset in [0, 10, -10, 20, -20] {
                        let tx = stock_bbox_min_x - bbox_min_x + (tx_offset as f64);
                        let ty = stock_bbox_min_y - bbox_min_y + (ty_offset as f64);

                        let offset = Vec2::new(tx, ty);
                        let transformed = rotated.translate(&offset);

                        if !transformed.fits_in(&stock_poly) {
                            continue;
                        }

                        let mut valid = true;
                        for (other_idx, other_offset, other_angle) in &stocks[stock_idx] {
                            let other_transformed =
                                pieces[*other_idx].transform(other_offset, *other_angle);
                            if transformed.overlaps(&other_transformed) {
                                valid = false;
                                break;
                            }
                        }

                        if valid {
                            stocks[stock_idx].push((piece_idx, offset.clone(), angle));
                            placed[piece_idx] = true;
                            placed_flag = true;
                            break;
                        }
                    }
                    if placed_flag {
                        break;
                    }
                }
            }
        }

        if !placed[piece_idx] {
            stocks.push(vec![(piece_idx, Vec2::new(0.0, 0.0), 0.0)]);
            placed[piece_idx] = true;
        }
    }

    println!("{}", stocks.len());
    for stock_idx in 0..stocks.len() {
        for (piece_idx, offset, angle) in &stocks[stock_idx] {
            println!("{} {} {} {}", stock_idx, offset.x, offset.y, angle);
        }
    }
}
