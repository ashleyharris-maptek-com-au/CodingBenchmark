import math

# Module-level state for persistence across calls
_state_history = {
    'rel_pos': [0.0, 0.0, 0.0],
    'rel_vel': [0.0, 0.0, 0.0],
    'fuel_low': False,
    'approach_phase': 'far',
    'sensor_health': {}
}

def autopilot_step(state: dict, dt: float) -> dict:
    global _state_history

    # Sensor fusion: select best available sensors
    rel_pos = fuse_position(state)
    rel_vel = fuse_velocity(state)
    distance = math.sqrt(rel_pos[0]**2 + rel_pos[1]**2 + rel_pos[2]**2)
    speed = math.sqrt(rel_vel[0]**2 + rel_vel[1]**2 + rel_vel[2]**2)

    # Check fuel status
    fuel_pct = state.get('fuel_remaining_pct', 100)
    _state_history['fuel_low'] = fuel_pct < 15

    # Determine approach phase
    if distance < 2 and speed < 0.3:
        approach_phase = 'docked'
    elif distance < 10:
        approach_phase = 'final'
    elif distance < 50:
        approach_phase = 'close'
    elif distance < 200:
        approach_phase = 'medium'
    else:
        approach_phase = 'far'

    _state_history['approach_phase'] = approach_phase
    _state_history['rel_pos'] = rel_pos
    _state_history['rel_vel'] = rel_vel

    # Compute thrust command
    thrust_x, thrust_y, thrust_z = compute_thrust(rel_pos, rel_vel, distance, approach_phase, state)

    # Clamp thrusts to [-1, 1]
    thrust_x = max(-1.0, min(1.0, thrust_x))
    thrust_y = max(-1.0, min(1.0, thrust_y))
    thrust_z = max(-1.0, min(1.0, thrust_z))

    # Attitude rates: minimal unless needed
    att_rate_roll = 0.0
    att_rate_pitch = 0.0
    att_rate_yaw = 0.0

    # Minor attitude adjustment if needed for cross-track
    if abs(rel_pos[2]) > 20:
        att_rate_yaw = 0.01 * (rel_pos[2] / abs(rel_pos[2]))

    return {
        'thrust_x': thrust_x,
        'thrust_y': thrust_y,
        'thrust_z': thrust_z,
        'att_rate_roll': att_rate_roll,
        'att_rate_pitch': att_rate_pitch,
        'att_rate_yaw': att_rate_yaw
    }

def fuse_position(state):
    """Fuse position from multiple sensors."""
    sources = []

    # Lidar (preferred, most accurate)
    if (state.get('lidar_rel_x_valid') and state.get('lidar_rel_y_valid') and
        state.get('lidar_rel_z_valid')):
        sources.append({
            'pos': [state['lidar_rel_pos_x_m'], state['lidar_rel_pos_y_m'], state['lidar_rel_pos_z_m']],
            'weight': 1.0
        })

    # Radar (secondary)
    if (state.get('radar_rel_x_valid') and state.get('radar_rel_y_valid') and
        state.get('radar_rel_z_valid')):
        sources.append({
            'pos': [state['radar_rel_pos_x_m'], state['radar_rel_pos_y_m'], state['radar_rel_pos_z_m']],
            'weight': 0.8
        })

    # Docking camera (high accuracy when available)
    if state.get('dock_camera_active') and state.get('dock_camera_bearing_x_valid'):
        camera_range = state.get('dock_camera_range_m', 50)
        bx = state.get('dock_camera_bearing_x', 0)
        by = state.get('dock_camera_bearing_y', 0)
        bz = state.get('dock_camera_bearing_z', 0)
        norm = math.sqrt(bx**2 + by**2 + bz**2)
        if norm > 0:
            sources.append({
                'pos': [bx/norm * camera_range, by/norm * camera_range, bz/norm * camera_range],
                'weight': 0.9
            })

    # GPS (use if available)
    if (state.get('gps_1_x_valid') and state.get('gps_1_y_valid') and
        state.get('gps_1_z_valid')):
        sources.append({
            'pos': [state['gps_1_pos_x_m'], state['gps_1_pos_y_m'], state['gps_1_pos_z_m']],
            'weight': 0.7
        })

    if not sources:
        return [0.0, 0.0, 0.0]

    # Weighted average
    total_weight = sum(s['weight'] for s in sources)
    fused = [0.0, 0.0, 0.0]
    for source in sources:
        w = source['weight'] / total_weight
        for i in range(3):
            fused[i] += w * source['pos'][i]

    return fused

def fuse_velocity(state):
    """Fuse velocity from multiple sensors."""
    sources = []

    # Lidar
    if (state.get('lidar_relv_x_valid') and state.get('lidar_relv_y_valid') and
        state.get('lidar_relv_z_valid')):
        sources.append({
            'vel': [state['lidar_rel_vel_x_ms'], state['lidar_rel_vel_y_ms'], state['lidar_rel_vel_z_ms']],
            'weight': 1.0
        })

    # Radar
    if (state.get('radar_relv_x_valid') and state.get('radar_relv_y_valid') and
        state.get('radar_relv_z_valid')):
        sources.append({
            'vel': [state['radar_rel_vel_x_ms'], state['radar_rel_vel_y_ms'], state['radar_rel_vel_z_ms']],
            'weight': 0.8
        })

    if not sources:
        return [0.0, 0.0, 0.0]

    total_weight = sum(s['weight'] for s in sources)
    fused = [0.0, 0.0, 0.0]
    for source in sources:
        w = source['weight'] / total_weight
        for i in range(3):
            fused[i] += w * source['vel'][i]

    return fused

def compute_thrust(rel_pos, rel_vel, distance, approach_phase, state):
    """Compute thrust command based on relative state and approach phase."""

    x, y, z = rel_pos
    vx, vy, vz = rel_vel

    fuel_pct = state.get('fuel_remaining_pct', 100)
    fuel_mult = 1.0 if fuel_pct > 30 else 0.5

    if approach_phase == 'docked':
        # Hold position gently
        return 0.1 * fuel_mult * (-vy), 0.1 * fuel_mult * (-vx), 0.1 * fuel_mult * (-vz)

    elif approach_phase == 'final':
        # Very gentle approach, avoid overshooting
        kp_pos = 0.02
        kd_vel = 0.08

        thrust_x = fuel_mult * (kp_pos * x - kd_vel * vx)
        thrust_y = fuel_mult * (kp_pos * y - kd_vel * vy)
        thrust_z = fuel_mult * (kp_pos * z - kd_vel * vz)

        return thrust_x, thrust_y, thrust_z

    elif approach_phase == 'close':
        # Linear approximation valid here
        kp_pos = 0.05
        kd_vel = 0.1

        thrust_x = fuel_mult * (kp_pos * x - kd_vel * vx)
        thrust_y = fuel_mult * (kp_pos * y - kd_vel * vy)
        thrust_z = fuel_mult * (kp_pos * z - kd_vel * vz)

        return thrust_x, thrust_y, thrust_z

    elif approach_phase == 'medium':
        # V-bar approach preferred
        kp_pos = 0.08
        kd_vel = 0.12

        thrust_x = fuel_mult * (kp_pos * x - kd_vel * vx)
        thrust_y = fuel_mult * (kp_pos * y - kd_vel * vy)
        thrust_z = fuel_mult * (kp_pos * z - kd_vel * vz)

        return thrust_x, thrust_y, thrust_z

    else:  # far approach
        # Use orbital mechanics: retrograde if behind (y < 0) to lower orbit and go faster
        if y < -100:  # Definitely behind
            # Retrograde thrust to lower orbit
            thrust_y = -0.5 * fuel_mult
            thrust_x = -0.1 * fuel_mult * (x / (distance + 0.1))
            thrust_z = -0.1 * fuel_mult * (z / (distance + 0.1))
        else:
            # Gentle approach
            kp_pos = 0.03
            kd_vel = 0.08

            thrust_x = fuel_mult * (kp_pos * x - kd_vel * vx)
            thrust_y = fuel_mult * (kp_pos * y - kd_vel * vy)
            thrust_z = fuel_mult * (kp_pos * z - kd_vel * vz)

        return thrust_x, thrust_y, thrust_z
