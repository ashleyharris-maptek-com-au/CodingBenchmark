import sys
import math
from typing import List, Tuple, Dict

def read_input():
    lines = sys.stdin.read().strip().split('\n')
    n, k, capacity = map(int, lines[0].split())
    depot_x, depot_y = map(int, lines[1].split())

    customers = []
    for i in range(2, 2 + n):
        x, y, demand = map(int, lines[i].split())
        customers.append((x, y, demand))

    return n, k, capacity, (depot_x, depot_y), customers

def distance(p1: Tuple[int, int], p2: Tuple[int, int]) -> float:
    return math.sqrt((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2)

def route_distance(depot: Tuple[int, int], route: List[int], customers: List[Tuple[int, int, int]]) -> float:
    if not route:
        return 0
    total = distance(depot, (customers[route[0]][0], customers[route[0]][1]))
    for i in range(len(route) - 1):
        total += distance((customers[route[i]][0], customers[route[i]][1]),
                         (customers[route[i+1]][0], customers[route[i+1]][1]))
    total += distance((customers[route[-1]][0], customers[route[-1]][1]), depot)
    return total

def solve_cvrp(n: int, k: int, capacity: int, depot: Tuple[int, int],
               customers: List[Tuple[int, int, int]]) -> Tuple[float, List[List[int]]]:

    unrouted = set(range(n))
    routes = []

    while unrouted:
        route = []
        current_load = 0
        current_pos = depot

        candidates = sorted(unrouted,
                          key=lambda c: distance(current_pos, (customers[c][0], customers[c][1])))

        for customer in candidates:
            demand = customers[customer][2]
            cust_pos = (customers[customer][0], customers[customer][1])

            if current_load + demand <= capacity:
                route.append(customer)
                current_load += demand
                current_pos = cust_pos
                unrouted.remove(customer)

        if route:
            routes.append(route)
        else:
            if unrouted:
                customer = unrouted.pop()
                routes.append([customer])

    for route in routes:
        improved = True
        while improved:
            improved = False
            for i in range(len(route)):
                for j in range(i + 2, len(route)):
                    new_route = route[:i+1] + route[i+1:j+1][::-1] + route[j+1:]
                    if route_distance(depot, new_route, customers) < route_distance(depot, route, customers):
                        route[:] = new_route
                        improved = True
                        break
                if improved:
                    break

    total_distance = sum(route_distance(depot, route, customers) for route in routes)

    return total_distance, routes

def main():
    n, k, capacity, depot, customers = read_input()

    if n == 0:
        print(0)
        print(0)
        return

    total_dist, routes = solve_cvrp(n, k, capacity, depot, customers)

    print(f"{total_dist:.1f}")
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
