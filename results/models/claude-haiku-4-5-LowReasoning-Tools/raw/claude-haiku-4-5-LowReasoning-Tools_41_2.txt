; SPIR-V
; Version: 1.0
; Generator: Claude
; Bound: 60
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Decorations for UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

; UBO struct
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

; Pointer types
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float

%func_type = OpTypeFunction %void

; Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%out_color = OpVariable %ptr_Output_v4float Output
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load UV coordinates
%uv_loaded = OpLoad %v2float %in_uv

; Extract UV components
%uv_x = OpCompositeExtract %float %uv_loaded 0
%uv_y = OpCompositeExtract %float %uv_loaded 1

; Construct output color (uv.x, uv.y, 0.0, 1.0)
%color_out = OpCompositeConstruct %v4float %uv_x %uv_y %float_0 %float_1

; Store to output
OpStore %out_color %color_out

OpReturn
OpFunctionEnd
