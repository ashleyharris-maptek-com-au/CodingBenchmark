use std::io::{self, Read};
use std::collections::{VecDeque, HashMap};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();

    let lines: Vec<&str> = input.lines().collect();
    let first_line: Vec<&str> = lines[0].split_whitespace().collect();
    let h: usize = first_line[0].parse().unwrap();
    let w: usize = first_line[1].parse().unwrap();

    // Read maze, padding rows with spaces if needed
    let mut maze: Vec<Vec<char>> = Vec::new();
    for i in 1..=h {
        if i < lines.len() {
            let mut row: Vec<char> = lines[i].chars().collect();
            while row.len() < w {
                row.push(' ');
            }
            maze.push(row);
        } else {
            maze.push(vec![' '; w]);
        }
    }

    // Find start (A) and end (B) positions
    let mut start = (0, 0);
    let mut end = (0, 0);
    let mut found_start = false;
    let mut found_end = false;

    for i in 0..h {
        for j in 0..w {
            if j < maze[i].len() {
                if maze[i][j] == 'A' && !found_start {
                    start = (j, i);
                    found_start = true;
                }
                if maze[i][j] == 'B' && !found_end {
                    end = (j, i);
                    found_end = true;
                }
            }
        }
    }

    // BFS to find shortest path
    let mut queue = VecDeque::new();
    let mut visited = HashMap::new();
    let mut parent: HashMap<(usize, usize), (usize, usize)> = HashMap::new();

    queue.push_back(start);
    visited.insert(start, true);

    while let Some((x, y)) = queue.pop_front() {
        if (x, y) == end {
            break;
        }

        // Explore 4 directions: up, down, right, left
        let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];

        for (dx, dy) in &directions {
            let nx = (x as i32 + dx) as usize;
            let ny = (y as i32 + dy) as usize;

            if nx < w && ny < h && !visited.contains_key(&(nx, ny)) {
                let cell = maze[ny][nx];
                if cell != '#' {
                    visited.insert((nx, ny), true);
                    parent.insert((nx, ny), (x, y));
                    queue.push_back((nx, ny));
                }
            }
        }
    }

    // Reconstruct path from start to end
    if visited.contains_key(&end) {
        let mut path = Vec::new();
        let mut current = end;

        while current != start {
            path.push(current);
            current = parent[&current];
        }
        path.push(start);
        path.reverse();

        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
