#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string findLCS(const vector<string>& strings) {
    if (strings.empty()) return "";

    const string& first = strings[0];
    int n = first.length();

    // Try all substrings of the first string, from longest to shortest
    for (int len = n; len >= 1; --len) {
        for (int i = 0; i <= n - len; ++i) {
            string candidate = first.substr(i, len);

            // Check if this substring exists in all other strings
            bool foundInAll = true;
            for (size_t j = 1; j < strings.size(); ++j) {
                if (strings[j].find(candidate) == string::npos) {
                    foundInAll = false;
                    break;
                }
            }

            if (foundInAll) {
                return candidate;
            }
        }
    }

    return "";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; ++i) {
        getline(cin, strings[i]);
    }

    cout << findLCS(strings) << endl;

    return 0;
}
