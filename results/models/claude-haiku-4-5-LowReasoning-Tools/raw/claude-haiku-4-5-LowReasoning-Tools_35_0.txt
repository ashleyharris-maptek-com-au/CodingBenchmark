use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: num_vertices num_edges
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_vertices: usize = parts[0].parse().unwrap();
    let num_edges: usize = parts[1].parse().unwrap();

    // Build adjacency list
    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];

    for _ in 0..num_edges {
        let edge_line = lines.next().unwrap().unwrap();
        let edge_parts: Vec<&str> = edge_line.split_whitespace().collect();
        let u: usize = edge_parts[0].parse().unwrap();
        let v: usize = edge_parts[1].parse().unwrap();
        adj[u].insert(v);
        adj[v].insert(u);
    }

    // Greedy algorithm for minimum dominating set
    let mut dominating_set = Vec::new();
    let mut uncovered: HashSet<usize> = (0..num_vertices).collect();

    while !uncovered.is_empty() {
        // Find the vertex that covers the most uncovered vertices
        let mut best_vertex = *uncovered.iter().next().unwrap();
        let mut best_count = 0;

        for &v in &uncovered {
            let mut count = 1; // The vertex itself
            for &neighbor in &adj[v] {
                if uncovered.contains(&neighbor) {
                    count += 1;
                }
            }
            if count > best_count {
                best_count = count;
                best_vertex = v;
            }
        }

        // Add best_vertex to dominating set
        dominating_set.push(best_vertex);
        uncovered.remove(&best_vertex);

        // Remove neighbors from uncovered
        for &neighbor in &adj[best_vertex].clone() {
            uncovered.remove(&neighbor);
        }
    }

    // Output
    println!("{}", dominating_set.len());
    if !dominating_set.is_empty() {
        let output = dominating_set.iter()
            .map(|x| x.to_string())
            .collect::<Vec<_>>()
            .join(" ");
        println!("{}", output);
    }
}
