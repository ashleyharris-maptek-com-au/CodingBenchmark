[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the normal
    float3 norm = normalize(input.normal);

    // Create 8x8 checkerboard pattern
    float2 checkUV = input.uv * 8.0;
    float checker = mod(floor(checkUV.x) + floor(checkUV.y), 2.0);
    float checkerColor = lerp(0.2, 0.9, checker);

    // Compute light direction
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);

    // Lambertian shading: diffuse = dot(normal, lightDir)
    float diffuse = max(dot(norm, lightDir), 0.0);
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Apply checkerboard pattern with Lambertian shading
    float3 finalColor = float3(checkerColor, checkerColor, checkerColor) * shadingFactor;

    return float4(finalColor, 1.0);
}
