; SPIR-V

; Bound: 300

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

; Decoration for output
OpDecorate %out_color Location 0

; UBO decorations
OpDecorate %UBO Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpMemberDecorate %UBO 0 Offset 0 ColMajor MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64 ColMajor MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128 ColMajor MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Import GLSL450 extended instruction set
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointers
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

; Variables
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3.0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%base_color = OpConstantComposite %vec3 %float_0_2 %float_0_3 %float_0_2
%rim_tint = OpConstantComposite %vec3 %float_1 %float_0_8 %float_0_5

; Function types
%func_type = OpTypeFunction %void

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%tangent = OpLoad %vec3 %in_tangent
%vertexColor = OpLoad %vec3 %in_color

; Normalize normal
%norm_glsl = OpExtInst %vec3 %GLSL450 Normalize %normal

; Load light position from UBO (member offset 3)
%light_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %int_3
%lightPos = OpLoad %vec4 %light_ptr

; Load camera position from UBO (member offset 4)
%cam_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %int_4
%cameraPos = OpLoad %vec4 %cam_ptr

; Extract xyz from light and camera positions
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

; Calculate light direction (light - world)
%light_dir = OpFSub %vec3 %lightPos_xyz %worldPos
%light_dir_norm = OpExtInst %vec3 %GLSL450 Normalize %light_dir

; Calculate view direction (camera - world)
%view_dir = OpFSub %vec3 %cameraPos_xyz %worldPos
%view_dir_norm = OpExtInst %vec3 %GLSL450 Normalize %view_dir

; Calculate diffuse: max(dot(N, L), 0)
%dotNL = OpDot %float %norm_glsl %light_dir_norm
%dotNL_clamped = OpExtInst %float %GLSL450 FMax %dotNL %float_0

; Calculate Fresnel rim: pow(1 - max(dot(N, V), 0), 3)
%dotNV = OpDot %float %norm_glsl %view_dir_norm
%dotNV_clamped = OpExtInst %float %GLSL450 FMax %dotNV %float_0
%one_minus_dotNV = OpFSub %float %float_1 %dotNV_clamped
%rim = OpExtInst %float %GLSL450 Pow %one_minus_dotNV %float_3

; Calculate diffuse modulation factor: 0.3 + 0.5 * diffuse
%half_diffuse = OpFMul %float %float_0_5 %dotNL_clamped
%diffuse_mod = OpFAdd %float %float_0_3 %half_diffuse

; Calculate base color modulated by diffuse factor
%base_modulated = OpVectorTimesScalar %vec3 %base_color %diffuse_mod

; Calculate rim lighting contribution: rim_tint * rim
%rim_contribution = OpVectorTimesScalar %vec3 %rim_tint %rim

; Combine: base color + rim lighting
%final_color = OpFAdd %vec3 %base_modulated %rim_contribution

; Create output vec4 with alpha = 1.0
%final_output = OpCompositeConstruct %vec4 %final_color %float_1

; Store output
OpStore %out_color %final_output

OpReturn
OpFunctionEnd
