using System;
using System.Collections.Generic;
using System.Linq;

class BinPacker3D
{
    struct Vector3
    {
        public float X, Y, Z;
        public Vector3(float x, float y, float z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, float s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
        public static float Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public float Length => (float)Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vector3 Normalized => this * (1f / Length);
    }

    struct Quaternion
    {
        public float W, X, Y, Z;
        public Quaternion(float w, float x, float y, float z) { W = w; X = x; Y = y; Z = z; }

        public Vector3 RotatePoint(Vector3 p)
        {
            float num = X * 2f;
            float num2 = Y * 2f;
            float num3 = Z * 2f;
            float num4 = X * num;
            float num5 = Y * num2;
            float num6 = Z * num3;
            float num7 = X * num2;
            float num8 = X * num3;
            float num9 = Y * num3;
            float num10 = W * num;
            float num11 = W * num2;
            float num12 = W * num3;

            Vector3 result;
            result.X = (1f - (num5 + num6)) * p.X + (num7 - num12) * p.Y + (num8 + num11) * p.Z;
            result.Y = (num7 + num12) * p.X + (1f - (num4 + num6)) * p.Y + (num9 - num10) * p.Z;
            result.Z = (num8 - num11) * p.X + (num9 + num10) * p.Y + (1f - (num4 + num5)) * p.Z;
            return result;
        }
    }

    struct AABB
    {
        public Vector3 Min, Max;
        public AABB(Vector3 min, Vector3 max) { Min = min; Max = max; }

        public bool Intersects(AABB other)
        {
            return Min.X <= other.Max.X && Max.X >= other.Min.X &&
                   Min.Y <= other.Max.Y && Max.Y >= other.Min.Y &&
                   Min.Z <= other.Max.Z && Max.Z >= other.Min.Z;
        }

        public bool Contains(Vector3 p)
        {
            return p.X >= Min.X && p.X <= Max.X &&
                   p.Y >= Min.Y && p.Y <= Max.Y &&
                   p.Z >= Min.Z && p.Z <= Max.Z;
        }
    }

    struct Placement
    {
        public Vector3 Translation;
        public Quaternion Rotation;
        public AABB Bounds;
    }

    static void Main()
    {
        string line;
        var parts = Console.ReadLine().Split();
        int V = int.Parse(parts[0]);
        int F = int.Parse(parts[1]);

        var vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            parts = Console.ReadLine().Split();
            vertices[i] = new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2]));
        }

        var faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            parts = Console.ReadLine().Split();
            faces[i] = new int[3] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) };
        }

        parts = Console.ReadLine().Split();
        float W = float.Parse(parts[0]);
        float H = float.Parse(parts[1]);
        float D = float.Parse(parts[2]);

        var placements = PackItems(vertices, faces, W, H, D);

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X:F3} {p.Translation.Y:F3} {p.Translation.Z:F3} {p.Rotation.W:F3} {p.Rotation.X:F3} {p.Rotation.Y:F3} {p.Rotation.Z:F3}");
        }
    }

    static List<Placement> PackItems(Vector3[] vertices, int[][] faces, float W, float H, float D)
    {
        var placements = new List<Placement>();
        var container = new AABB(new Vector3(0, 0, 0), new Vector3(W, H, D));

        float scale = Math.Max(Math.Max(W, H), D);
        int maxAttempts = scale > 100 ? 50 : (scale > 50 ? 100 : 200);
        int rotationCount = scale > 100 ? 4 : (scale > 50 ? 8 : 12);

        var rotations = GenerateRotations(rotationCount);

        for (int attempt = 0; attempt < maxAttempts; attempt++)
        {
            foreach (var rot in rotations)
            {
                var bounds = ComputeBounds(vertices, rot);

                for (float x = 0; x < W; x += Math.Max(1f, (W - bounds.Max.X + bounds.Min.X) / 10f))
                {
                    for (float y = 0; y < H; y += Math.Max(1f, (H - bounds.Max.Y + bounds.Min.Y) / 10f))
                    {
                        for (float z = 0; z < D; z += Math.Max(1f, (D - bounds.Max.Z + bounds.Min.Z) / 10f))
                        {
                            var translation = new Vector3(x, y, z);
                            var rotatedBounds = new AABB(bounds.Min + translation, bounds.Max + translation);

                            if (!container.Contains(rotatedBounds.Min) || !container.Contains(rotatedBounds.Max))
                                continue;

                            bool collides = false;
                            foreach (var p in placements)
                            {
                                if (rotatedBounds.Intersects(p.Bounds))
                                {
                                    collides = true;
                                    break;
                                }
                            }

                            if (!collides)
                            {
                                placements.Add(new Placement { Translation = translation, Rotation = rot, Bounds = rotatedBounds });
                                goto NextAttempt;
                            }
                        }
                    }
                }
            }
            NextAttempt:;
        }

        return placements;
    }

    static List<Quaternion> GenerateRotations(int count)
    {
        var rotations = new List<Quaternion> { new Quaternion(1, 0, 0, 0) };

        for (int i = 0; i < count; i++)
        {
            float angle = (float)(2 * Math.PI * i / count);
            rotations.Add(new Quaternion((float)Math.Cos(angle / 2), (float)Math.Sin(angle / 2), 0, 0));
            rotations.Add(new Quaternion((float)Math.Cos(angle / 2), 0, (float)Math.Sin(angle / 2), 0));
            rotations.Add(new Quaternion((float)Math.Cos(angle / 2), 0, 0, (float)Math.Sin(angle / 2)));
        }

        return rotations;
    }

    static AABB ComputeBounds(Vector3[] vertices, Quaternion rot)
    {
        float minX = float.MaxValue, minY = float.MaxValue, minZ = float.MaxValue;
        float maxX = float.MinValue, maxY = float.MinValue, maxZ = float.MinValue;

        foreach (var v in vertices)
        {
            var rotated = rot.RotatePoint(v);
            minX = Math.Min(minX, rotated.X);
            minY = Math.Min(minY, rotated.Y);
            minZ = Math.Min(minZ, rotated.Z);
            maxX = Math.Max(maxX, rotated.X);
            maxY = Math.Max(maxY, rotated.Y);
            maxZ = Math.Max(maxZ, rotated.Z);
        }

        return new AABB(new Vector3(minX, minY, minZ), new Vector3(maxX, maxY, maxZ));
    }
}
