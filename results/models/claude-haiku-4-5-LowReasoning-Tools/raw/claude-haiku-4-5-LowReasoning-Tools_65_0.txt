declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface ScheduleEntry {
  id: string;
  start: number;
  machine: number;
}

interface Output {
  schedule: ScheduleEntry[];
}

const { machineCount, tasks } = data;

const taskMap = new Map<string, Task>();
tasks.forEach((task: Task) => taskMap.set(task.id, task));

const earliestStart = new Map<string, number>();

function calculateEarliestStart(taskId: string): number {
  if (earliestStart.has(taskId)) {
    return earliestStart.get(taskId)!;
  }

  const task = taskMap.get(taskId);
  if (!task) return 0;

  let maxDependencyEnd = 0;
  for (const depId of task.deps) {
    const depTask = taskMap.get(depId);
    if (depTask) {
      const depEarliestStart = calculateEarliestStart(depId);
      const depEnd = depEarliestStart + depTask.duration;
      maxDependencyEnd = Math.max(maxDependencyEnd, depEnd);
    }
  }

  earliestStart.set(taskId, maxDependencyEnd);
  return maxDependencyEnd;
}

tasks.forEach((task: Task) => calculateEarliestStart(task.id));

const machineEndTime: number[] = Array(machineCount).fill(0);
const schedule: ScheduleEntry[] = [];

const sortedTasks = [...tasks].sort((a: Task, b: Task) => {
  const priorityA = a.deadline / Math.max(a.weight, 0.1);
  const priorityB = b.deadline / Math.max(b.weight, 0.1);

  if (priorityA !== priorityB) {
    return priorityA - priorityB;
  }

  const earliestA = earliestStart.get(a.id) || 0;
  const earliestB = earliestStart.get(b.id) || 0;

  return earliestA - earliestB;
});

for (const task of sortedTasks) {
  const earliestStartTime = earliestStart.get(task.id) || 0;

  let bestMachine = 0;
  let bestStartTime = Math.max(machineEndTime[0], earliestStartTime);

  for (let machine = 1; machine < machineCount; machine++) {
    const startTime = Math.max(machineEndTime[machine], earliestStartTime);
    if (startTime < bestStartTime) {
      bestStartTime = startTime;
      bestMachine = machine;
    }
  }

  schedule.push({
    id: task.id,
    start: bestStartTime,
    machine: bestMachine
  });

  machineEndTime[bestMachine] = bestStartTime + task.duration;
}

const output: Output = { schedule };
console.log(JSON.stringify(output));
