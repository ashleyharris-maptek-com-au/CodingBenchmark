OpCapability Geometry
               OpCapability GLSL450
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Patch
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %gl_out Block
               OpMemberDecorate %gl_out 0 BuiltIn Position
               OpDecorate %outColor Location 0

               %void = OpTypeVoid
               %bool = OpTypeBool
               %int = OpTypeInt 32 1
               %uint = OpTypeInt 32 0
               %float = OpTypeFloat 32
               %v3float = OpTypeVector %float 3
               %v4float = OpTypeVector %float 4

               %uint_3 = OpConstant %uint 3
               %int_0 = OpConstant %int 0
               %int_1 = OpConstant %int 1
               %int_2 = OpConstant %int 2
               %float_0p08 = OpConstant %float 0.08
               %float_1 = OpConstant %float 1.0

               %gl_PerVertex_in = OpTypeStruct %v4float
               %gl_PerVertex_out = OpTypeStruct %v4float
               %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
               %arr_v3f = OpTypeArray %v3float %uint_3

               %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
               %ptr_in_arr_v3f = OpTypePointer Input %arr_v3f
               %ptr_in_v4f = OpTypePointer Input %v4float
               %ptr_in_v3f = OpTypePointer Input %v3float
               %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
               %ptr_out_v4f = OpTypePointer Output %v4float
               %ptr_out_v3f = OpTypePointer Output %v3float

               %func_void = OpTypeFunction %void

               %gl_in = OpVariable %ptr_in_arr_pv Input
               %worldPos_in = OpVariable %ptr_in_arr_v3f Input
               %normal_in = OpVariable %ptr_in_arr_v3f Input
               %color_in = OpVariable %ptr_in_arr_v3f Input
               %gl_out = OpVariable %ptr_out_pv Output
               %outColor = OpVariable %ptr_out_v3f Output

               %main = OpFunction %void None %func_void
               %entry = OpLabel

               %pos0_ptr = OpAccessChain %ptr_in_v4f %gl_in %int_0 %int_0
               %pos0 = OpLoad %v4float %pos0_ptr
               %pos0_v3 = OpVectorShuffle %v3float %pos0 %pos0 0 1 2

               %pos1_ptr = OpAccessChain %ptr_in_v4f %gl_in %int_1 %int_0
               %pos1 = OpLoad %v4float %pos1_ptr
               %pos1_v3 = OpVectorShuffle %v3float %pos1 %pos1 0 1 2

               %pos2_ptr = OpAccessChain %ptr_in_v4f %gl_in %int_2 %int_0
               %pos2 = OpLoad %v4float %pos2_ptr
               %pos2_v3 = OpVectorShuffle %v3float %pos2 %pos2 0 1 2

               %e1 = OpFSub %v3float %pos1_v3 %pos0_v3
               %e2 = OpFSub %v3float %pos2_v3 %pos0_v3

               %normal_raw = OpExtInst %v3float %1 Cross %e1 %e2
               %normal = OpExtInst %v3float %1 Normalize %normal_raw

               %offset = OpVectorTimesScalar %v3float %normal %float_0p08

               %color0_ptr = OpAccessChain %ptr_in_v3f %color_in %int_0
               %color0 = OpLoad %v3float %color0_ptr
               %new_pos0_v3 = OpFAdd %v3float %pos0_v3 %offset
               %new_pos0 = OpCompositeConstruct %v4float %new_pos0_v3 %float_1
               %out_pos_ptr = OpAccessChain %ptr_out_v4f %gl_out %int_0
               OpStore %out_pos_ptr %new_pos0
               OpStore %outColor %color0
               OpEmitVertex

               %color1_ptr = OpAccessChain %ptr_in_v3f %color_in %int_1
               %color1 = OpLoad %v3float %color1_ptr
               %new_pos1_v3 = OpFAdd %v3float %pos1_v3 %offset
               %new_pos1 = OpCompositeConstruct %v4float %new_pos1_v3 %float_1
               OpStore %out_pos_ptr %new_pos1
               OpStore %outColor %color1
               OpEmitVertex

               %color2_ptr = OpAccessChain %ptr_in_v3f %color_in %int_2
               %color2 = OpLoad %v3float %color2_ptr
               %new_pos2_v3 = OpFAdd %v3float %pos2_v3 %offset
               %new_pos2 = OpCompositeConstruct %v4float %new_pos2_v3 %float_1
               OpStore %out_pos_ptr %new_pos2
               OpStore %outColor %color2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
