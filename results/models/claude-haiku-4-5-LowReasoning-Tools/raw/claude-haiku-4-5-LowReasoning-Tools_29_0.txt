using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

class MaximumCliqueSolver
{
    private static Stopwatch _sw;
    private const int TIMEOUT_MS = 280000; // 280 seconds, leaving buffer

    static void Main()
    {
        _sw = Stopwatch.StartNew();

        var input = Console.ReadLine().Split();
        int numVertices = int.Parse(input[0]);
        int numEdges = int.Parse(input[1]);

        var adj = new List<HashSet<int>>(numVertices);
        for (int i = 0; i < numVertices; i++)
            adj.Add(new HashSet<int>());

        for (int i = 0; i < numEdges; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        var bestClique = FindMaximumClique(adj, numVertices);

        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
        }
    }

    static List<int> FindMaximumClique(List<HashSet<int>> adj, int numVertices)
    {
        var bestClique = new List<int>();

        // For very large graphs, use greedy heuristic
        if (numVertices > 5000)
        {
            return GreedyClique(adj, numVertices);
        }

        // Try exact algorithm for smaller graphs
        var candidates = Enumerable.Range(0, numVertices).ToHashSet();
        BacktrackClique(adj, new List<int>(), candidates, bestClique);

        // If exact approach times out or fails, fallback to greedy
        if (bestClique.Count == 0 && _sw.ElapsedMilliseconds > TIMEOUT_MS / 2)
        {
            return GreedyClique(adj, numVertices);
        }

        return bestClique;
    }

    static void BacktrackClique(List<HashSet<int>> adj, List<int> current,
        HashSet<int> candidates, List<int> bestClique)
    {
        // Timeout check
        if (_sw.ElapsedMilliseconds > TIMEOUT_MS)
            return;

        // Pruning: if current + all remaining can't beat best, skip
        if (current.Count + candidates.Count <= bestClique.Count)
            return;

        if (candidates.Count == 0)
        {
            if (current.Count > bestClique.Count)
            {
                bestClique.Clear();
                bestClique.AddRange(current);
            }
            return;
        }

        // Order candidates by degree in remaining subgraph for better branching
        var candidateList = candidates
            .OrderByDescending(v => candidates.Count(x => x != v && adj[v].Contains(x)))
            .ToList();

        foreach (var v in candidateList)
        {
            if (_sw.ElapsedMilliseconds > TIMEOUT_MS)
                return;

            current.Add(v);

            // Find intersection of candidates with neighbors of v
            var newCandidates = new HashSet<int>();
            foreach (var c in candidates)
            {
                if (c > v && adj[v].Contains(c))
                    newCandidates.Add(c);
            }

            BacktrackClique(adj, current, newCandidates, bestClique);
            current.RemoveAt(current.Count - 1);
        }
    }

    static List<int> GreedyClique(List<HashSet<int>> adj, int numVertices)
    {
        var best = new List<int>();

        // Try starting from multiple high-degree vertices for better coverage
        int sampleSize = Math.Min(Math.Max(10, numVertices / 100), 500);
        var orderedVertices = Enumerable.Range(0, numVertices)
            .OrderByDescending(v => adj[v].Count)
            .Take(sampleSize)
            .ToList();

        foreach (var start in orderedVertices)
        {
            if (_sw.ElapsedMilliseconds > TIMEOUT_MS)
                break;

            var clique = BuildCliqueFrom(adj, start);
            if (clique.Count > best.Count)
                best = clique;
        }

        return best;
    }

    static List<int> BuildCliqueFrom(List<HashSet<int>> adj, int start)
    {
        var clique = new List<int> { start };
        var candidates = new HashSet<int>(adj[start]);

        while (candidates.Count > 0)
        {
            // Choose vertex with most connections to remaining candidates
            var next = candidates
                .OrderByDescending(v => candidates.Count(x => x != v && adj[v].Contains(x)))
                .First();

            clique.Add(next);

            // Retain only vertices connected to all in current clique
            var newCandidates = new HashSet<int>();
            foreach (var c in candidates)
            {
                if (c != next && adj[next].Contains(c))
                    newCandidates.Add(c);
            }
            candidates = newCandidates;
        }

        return clique;
    }
}
