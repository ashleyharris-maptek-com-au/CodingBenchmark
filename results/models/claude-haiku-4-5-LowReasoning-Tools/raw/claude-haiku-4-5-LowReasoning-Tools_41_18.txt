OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_outColor Location 0

OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

%ubo = OpVariable %ptr_Uniform_UBO Uniform

%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_outColor = OpVariable %ptr_Output_vec4 Output

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_0_5 = OpConstant %float 0.5
%int_4 = OpConstant %int 4
%ext_glsl_std_450 = OpExtInstImport "GLSL.std.450"

%fn_type = OpTypeFunction %void

%main = OpFunction %void None %fn_type
%entry = OpLabel

; Load inputs
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos

; Get camera position from UBO
%cameraPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_4
%cameraPos_vec4 = OpLoad %vec4 %cameraPos_ptr
%cameraPos = OpVectorShuffle %vec3 %cameraPos_vec4 %cameraPos_vec4 0 1 2

; Calculate V = normalize(cameraPos - worldPos)
%viewDir = OpFSub %vec3 %cameraPos %worldPos
%viewDir_norm = OpExtInst %vec3 %ext_glsl_std_450 Normalize %viewDir

; Normalize normal
%normal_norm = OpExtInst %vec3 %ext_glsl_std_450 Normalize %normal

; Calculate NdotV = max(dot(N, V), 0)
%dot_result = OpDot %float %normal_norm %viewDir_norm
%ndotv = OpExtInst %float %ext_glsl_std_450 FMax %dot_result %float_0

; Check if NdotV < 0.5
%cmp = OpFOrdLessThan %bool %ndotv %float_0_5

; Branch for color calculation
OpSelectionMerge %merge_color None
OpBranchConditional %cmp %branch_low %branch_high

%branch_low = OpLabel
; If NdotV < 0.5: r=1, g=2*NdotV, b=0
%g_low = OpFMul %float %float_2 %ndotv
%color_low = OpCompositeConstruct %vec4 %float_1 %g_low %float_0 %float_1
OpBranch %merge_color

%branch_high = OpLabel
; If NdotV >= 0.5: r=g=1-(NdotV-0.5)*2, b=(NdotV-0.5)*2
%ndotv_offset = OpFSub %float %ndotv %float_0_5
%factor = OpFMul %float %ndotv_offset %float_2
%rg_high = OpFSub %float %float_1 %factor
%b_high = %factor
%color_high = OpCompositeConstruct %vec4 %rg_high %rg_high %b_high %float_1
OpBranch %merge_color

%merge_color = OpLabel
%final_color = OpPhi %vec4 %color_low %branch_low %color_high %branch_high

; Store output
OpStore %out_outColor %final_color

OpReturn
OpFunctionEnd
