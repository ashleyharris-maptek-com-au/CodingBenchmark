               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0
        %void = OpTypeVoid
       %float = OpTypeFloat 32
      %v2float = OpTypeVector %float 2
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4
        %uint = OpTypeInt 32 0
      %uint_2 = OpConstant %uint 2
         %int = OpTypeInt 32 1
       %int_0 = OpConstant %int 0
       %int_1 = OpConstant %int 1
      %float_0 = OpConstant %float 0
      %float_1 = OpConstant %float 1
     %float_03 = OpConstant %float 0.03
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_2
      %arr_c_2 = OpTypeArray %v3float %uint_2
  %ptr_in_arr_c = OpTypePointer Input %arr_c_2
%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
 %ptr_out_color = OpTypePointer Output %v3float
 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_out_v4 = OpTypePointer Output %v4float
 %func_type = OpTypeFunction %void
       %gl_in = OpVariable %ptr_in_arr_pv Input
      %color = OpVariable %ptr_in_arr_c Input
      %gl_out = OpVariable %ptr_out_pv Output
    %outColor = OpVariable %ptr_out_color Output
       %main = OpFunction %void None %func_type
      %entry = OpLabel
   %p0_addr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_addr
   %p1_addr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_addr
   %c0_addr = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %c0_addr
   %c1_addr = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %c1_addr
        %dir = OpFSub %v4float %p1 %p0
    %dir_xy = OpVectorShuffle %v2float %dir %dir 0 1
    %dir_x = OpCompositeExtract %float %dir_xy 0
    %dir_y = OpCompositeExtract %float %dir_xy 1
 %neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v2float %neg_dir_y %dir_x
   %dot_xx = OpFMul %float %dir_x %dir_x
   %dot_yy = OpFMul %float %dir_y %dir_y
  %dot_sum = OpFAdd %float %dot_xx %dot_yy
  %perp_len = OpExtInst %float %1 Sqrt %dot_sum
 %perp_norm_x = OpFDiv %float %neg_dir_y %perp_len
 %perp_norm_y = OpFDiv %float %dir_x %perp_len
   %perp_x = OpFMul %float %perp_norm_x %float_03
   %perp_y = OpFMul %float %perp_norm_y %float_03
   %p0_x = OpCompositeExtract %float %p0 0
   %p0_y = OpCompositeExtract %float %p0 1
   %p0_z = OpCompositeExtract %float %p0 2
   %p0_w = OpCompositeExtract %float %p0 3
   %p1_x = OpCompositeExtract %float %p1 0
   %p1_y = OpCompositeExtract %float %p1 1
   %p1_z = OpCompositeExtract %float %p1 2
   %p1_w = OpCompositeExtract %float %p1 3
   %v0_x = OpFAdd %float %p0_x %perp_x
   %v0_y = OpFAdd %float %p0_y %perp_y
   %v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
   %v1_x = OpFSub %float %p0_x %perp_x
   %v1_y = OpFSub %float %p0_y %perp_y
   %v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
   %v2_x = OpFAdd %float %p1_x %perp_x
   %v2_y = OpFAdd %float %p1_y %perp_y
   %v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
   %v3_x = OpFSub %float %p1_x %perp_x
   %v3_y = OpFSub %float %p1_y %perp_y
   %v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
   %out_pos_addr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_addr %v0_pos
               OpStore %outColor %c0
               OpEmitVertex
               OpStore %out_pos_addr %v1_pos
               OpStore %outColor %c0
               OpEmitVertex
               OpStore %out_pos_addr %v2_pos
               OpStore %outColor %c1
               OpEmitVertex
               OpStore %out_pos_addr %v3_pos
               OpStore %outColor %c1
               OpEmitVertex
               OpEndPrimitive
