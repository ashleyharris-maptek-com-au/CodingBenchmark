import math
from collections import deque

_state = {
    'speed_filter': deque(maxlen=5),
    'lane_offset_filter': deque(maxlen=5),
    'yaw_rate_filter': deque(maxlen=5),
    'accel_filter': deque(maxlen=5),
    'steering_pid': {'integral': 0, 'prev_error': 0},
    'speed_pid': {'integral': 0, 'prev_error': 0},
    'lane_history': deque(maxlen=10),
    'obstacle_frames': 0,
    'last_valid_heading': 0,
}

def autopilot_step(state: dict, dt: float) -> dict:
    global _state

    # Sensor fusion with voting/median
    def fuse_valid_sensors(values, valid_flags, sensor_names):
        valid_values = [v for v, valid in zip(values, valid_flags) if valid]
        if not valid_values:
            return None
        if len(valid_values) >= 2:
            valid_values.sort()
            return valid_values[len(valid_values) // 2]
        return valid_values[0] if valid_values else None

    # Wheel speeds (km/h to m/s)
    wheel_speeds_kmh = [
        state.get('wheel_speed_fl_kmh', 0),
        state.get('wheel_speed_fr_kmh', 0),
        state.get('wheel_speed_rl_kmh', 0),
        state.get('wheel_speed_rr_kmh', 0),
    ]
    wheel_valid = [
        state.get('wheel_speed_fl_valid', False),
        state.get('wheel_speed_fr_valid', False),
        state.get('wheel_speed_rl_valid', False),
        state.get('wheel_speed_rr_valid', False),
    ]
    fused_wheel_speed = fuse_valid_sensors(wheel_speeds_kmh, wheel_valid, ['FL', 'FR', 'RL', 'RR'])
    fused_wheel_speed = (fused_wheel_speed / 3.6) if fused_wheel_speed else 0

    # Vehicle speed from dedicated sensors
    vehicle_speed_1 = state.get('vehicle_speed_1_kmh', 0) / 3.6
    vehicle_speed_2 = state.get('vehicle_speed_2_kmh', 0) / 3.6
    avg_vehicle_speed = (vehicle_speed_1 + vehicle_speed_2) / 2 if (vehicle_speed_1 + vehicle_speed_2) > 0 else fused_wheel_speed

    _state['speed_filter'].append(avg_vehicle_speed)
    current_speed = sum(_state['speed_filter']) / len(_state['speed_filter']) if _state['speed_filter'] else avg_vehicle_speed

    # Gyroscope fusion - yaw rate
    gyro_yaw_rates = [
        state.get('gyro_1_yaw_rate_dps', 0),
        state.get('gyro_2_yaw_rate_dps', 0),
        state.get('gyro_3_yaw_rate_dps', 0),
    ]
    gyro_valid = [
        state.get('gyro_1_valid', False),
        state.get('gyro_2_valid', False),
        state.get('gyro_3_valid', False),
    ]
    fused_yaw_rate = fuse_valid_sensors(gyro_yaw_rates, gyro_valid, ['G1', 'G2', 'G3'])
    fused_yaw_rate = fused_yaw_rate if fused_yaw_rate else 0

    _state['yaw_rate_filter'].append(fused_yaw_rate)
    fused_yaw_rate = sum(_state['yaw_rate_filter']) / len(_state['yaw_rate_filter']) if _state['yaw_rate_filter'] else fused_yaw_rate

    # Accelerometer fusion
    accel_lateral = [
        state.get('accel_1_lateral_ms2', 0),
        state.get('accel_2_lateral_ms2', 0),
        state.get('accel_3_lateral_ms2', 0),
    ]
    accel_valid = [
        state.get('accel_1_valid', False),
        state.get('accel_2_valid', False),
        state.get('accel_3_valid', False),
    ]
    fused_accel_lat = fuse_valid_sensors(accel_lateral, accel_valid, ['A1', 'A2', 'A3'])
    fused_accel_lat = fused_accel_lat if fused_accel_lat else 0

    # GPS position and heading
    gps_1_valid = state.get('gps_1_valid', False)
    gps_2_valid = state.get('gps_2_valid', False)
    gps_heading_1 = state.get('gps_1_heading_deg', _state['last_valid_heading'])
    gps_heading_2 = state.get('gps_2_heading_deg', _state['last_valid_heading'])

    if gps_1_valid or gps_2_valid:
        headings = []
        if gps_1_valid:
            headings.append(gps_heading_1)
            _state['last_valid_heading'] = gps_heading_1
        if gps_2_valid:
            headings.append(gps_heading_2)
            _state['last_valid_heading'] = gps_heading_2
        current_heading = sum(headings) / len(headings) if headings else _state['last_valid_heading']
    else:
        current_heading = _state['last_valid_heading']

    # Vision system fusion - lane offset
    vision_lane_offsets = [
        state.get('vision_1_lane_offset_m', 0),
        state.get('vision_2_lane_offset_m', 0),
        state.get('vision_3_lane_offset_m', 0),
    ]
    vision_valid = [
        state.get('vision_1_left_line_valid', False) and state.get('vision_1_right_line_valid', False),
        state.get('vision_2_left_line_valid', False) and state.get('vision_2_right_line_valid', False),
        state.get('vision_3_left_line_valid', False) and state.get('vision_3_right_line_valid', False),
    ]
    lane_offset = fuse_valid_sensors(vision_lane_offsets, vision_valid, ['V1', 'V2', 'V3'])
    lane_offset = lane_offset if lane_offset else 0

    _state['lane_offset_filter'].append(lane_offset)
    lane_offset = sum(_state['lane_offset_filter']) / len(_state['lane_offset_filter']) if _state['lane_offset_filter'] else lane_offset

    # Lidar obstacle detection
    lidar_angles = [-90, -60, -45, -30, -15, -5, 5, 15, 30, 45, 60, 90]
    lidar_valid_count = 0
    lidar_avg_range = 0
    for angle in lidar_angles:
        angle_str = f"lidar_{angle:+03d}_deg_range_m"
        valid_str = f"lidar_{angle:+03d}_deg_valid"
        if state.get(valid_str, False):
            lidar_avg_range += state.get(angle_str, 0)
            lidar_valid_count += 1

    lidar_avg_range = lidar_avg_range / lidar_valid_count if lidar_valid_count > 0 else 100

    # Vision obstacle detection
    vision_obstacles = [
        state.get('vision_1_obstacle_detected', False) and state.get('vision_1_obstacle_valid', False),
        state.get('vision_2_obstacle_detected', False) and state.get('vision_2_obstacle_valid', False),
        state.get('vision_3_obstacle_detected', False) and state.get('vision_3_obstacle_valid', False),
    ]
    vision_obstacle_dists = [
        state.get('vision_1_obstacle_dist_m', 100) if vision_obstacles[0] else 100,
        state.get('vision_2_obstacle_dist_m', 100) if vision_obstacles[1] else 100,
        state.get('vision_3_obstacle_dist_m', 100) if vision_obstacles[2] else 100,
    ]
    obstacle_detected = any(vision_obstacles)
    closest_obstacle = min(vision_obstacle_dists) if obstacle_detected else 100

    # Speed limit
    speed_limits = [
        state.get('vision_1_speed_limit_kmh', 130) if state.get('vision_1_speed_limit_valid', False) else 130,
        state.get('vision_2_speed_limit_kmh', 130) if state.get('vision_2_speed_limit_valid', False) else 130,
        state.get('vision_3_speed_limit_kmh', 130) if state.get('vision_3_speed_limit_valid', False) else 130,
    ]
    legal_speed_limit = min(speed_limits) if speed_limits else 130

    # Target speed
    target_speed_kmh = state.get('target_speed_kmh', 100)
    target_speed = min(target_speed_kmh, legal_speed_limit) / 3.6

    # Friction and surface conditions
    mu = state.get('estimated_mu', 1.0)
    rain = state.get('rain_sensor', 0)
    temp = state.get('temperature_c', 20)

    # Adjust max lateral accel based on friction
    max_lat_accel = 8 * mu

    # Steering control using PID for lane keeping
    lane_offset_error = -lane_offset  # Negative because we want to center (offset = 0)
    max_steering = 1.0

    if abs(lane_offset_error) > 0.3:
        kp_lane = 0.3 if mu > 0.5 else 0.15
        ki_lane = 0.05 if mu > 0.5 else 0.02
        kd_lane = 0.1 if mu > 0.5 else 0.05

        _state['steering_pid']['integral'] += lane_offset_error * dt
        _state['steering_pid']['integral'] = max(-0.5, min(0.5, _state['steering_pid']['integral']))

        derivative = (lane_offset_error - _state['steering_pid']['prev_error']) / dt if dt > 0 else 0

        steering = kp_lane * lane_offset_error + ki_lane * _state['steering_pid']['integral'] + kd_lane * derivative
        steering = max(-max_steering, min(max_steering, steering))

        _state['steering_pid']['prev_error'] = lane_offset_error
    else:
        steering = lane_offset * 0.1
        _state['steering_pid']['integral'] *= 0.9
        _state['steering_pid']['prev_error'] = lane_offset_error

    # Obstacle avoidance
    if obstacle_detected and closest_obstacle < 20:
        _state['obstacle_frames'] = 10

    if _state['obstacle_frames'] > 0:
        target_speed *= 0.7
        _state['obstacle_frames'] -= 1

    # Speed control using PID
    speed_error = target_speed - current_speed

    kp_speed = 2.0
    ki_speed = 0.3
    kd_speed = 0.5

    _state['speed_pid']['integral'] += speed_error * dt
    _state['speed_pid']['integral'] = max(-1.0, min(1.0, _state['speed_pid']['integral']))

    derivative = (speed_error - _state['speed_pid']['prev_error']) / dt if dt > 0 else 0

    accel_cmd = kp_speed * speed_error + ki_speed * _state['speed_pid']['integral'] + kd_speed * derivative
    _state['speed_pid']['prev_error'] = speed_error

    # Limit acceleration based on friction
    max_accel = 4.0 * mu
    accel_cmd = max(-max_accel, min(max_accel, accel_cmd))

    # Convert to throttle/brake
    if accel_cmd > 0.1:
        throttle = min(1.0, accel_cmd / max_accel)
        brake = 0.0
    elif accel_cmd < -0.5:
        throttle = 0.0
        brake = min(1.0, -accel_cmd / max_accel)
    else:
        throttle = 0.0
        brake = 0.0

    # Emergency braking
    if closest_obstacle < 10:
        brake = 1.0
        throttle = 0.0

    # Reduce aggressiveness on ice
    if mu < 0.3:
        throttle *= 0.5
        steering *= 0.5

    return {
        'steering': max(-1.0, min(1.0, steering)),
        'throttle': max(0.0, min(1.0, throttle)),
        'brake': max(0.0, min(1.0, brake)),
    }
