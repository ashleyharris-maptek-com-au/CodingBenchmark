using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
        public long Volume;
    }

    struct Placement
    {
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        string[] line = Console.ReadLine().Split();
        int N = int.Parse(line[0]);
        int W = int.Parse(line[1]);
        int H = int.Parse(line[2]);
        int D = int.Parse(line[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            string[] dims = Console.ReadLine().Split();
            int w = int.Parse(dims[0]);
            int h = int.Parse(dims[1]);
            int d = int.Parse(dims[2]);
            boxes.Add(new Box { Index = i, W = w, H = h, D = d, Volume = (long)w * h * d });
        }

        // Sort by volume descending for better packing
        var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ToList();

        var placements = new List<Placement>();
        var packed = new List<(int index, int x, int y, int z)>();

        foreach (var box in sortedBoxes)
        {
            var placement = FindBestPlacement(box, placements, W, H, D);
            if (placement.HasValue)
            {
                var p = placement.Value;
                placements.Add(p);
                packed.Add((box.Index, p.X, p.Y, p.Z));
            }
        }

        Console.WriteLine(packed.Count);
        foreach (var (index, x, y, z) in packed.OrderBy(p => p.index))
        {
            Console.WriteLine($"{index} {x} {y} {z}");
        }
    }

    static Placement? FindBestPlacement(Box box, List<Placement> placements, int W, int H, int D)
    {
        var candidates = new List<(int x, int y, int z, long score)>();

        // Try multiple orientations
        var orientations = GetOrientations(box);

        foreach (var (w, h, d) in orientations)
        {
            // Try positions along existing placements + origin
            var positions = GetCandidatePositions(placements, W, H, D);

            foreach (var (x, y, z) in positions)
            {
                if (x + w <= W && y + h <= H && z + d <= D && CanPlace(x, y, z, w, h, d, placements))
                {
                    long score = -(long)x * H * D - (long)y * W * D - (long)z * W * H;
                    candidates.Add((x, y, z, score));
                }
            }
        }

        if (candidates.Count == 0)
            return null;

        var best = candidates.OrderBy(c => c.score).First();
        return FindPlacementWithOrientation(box, best.x, best.y, best.z, placements, W, H, D);
    }

    static List<(int w, int h, int d)> GetOrientations(Box box)
    {
        var dims = new[] { box.W, box.H, box.D }.OrderByDescending(x => x).ToArray();
        var orientations = new HashSet<(int, int, int)>();

        foreach (var a in dims)
            foreach (var b in dims.Where(x => x != a || dims.Count(y => y == x) > 1))
            {
                int c = dims.FirstOrDefault(x => x != a && x != b);
                if (c == 0) c = (dims.Sum() - a - b);
                orientations.Add((a, b, c));
            }

        return orientations.ToList();
    }

    static HashSet<(int, int, int)> GetCandidatePositions(List<Placement> placements, int W, int H, int D)
    {
        var positions = new HashSet<(int, int, int)> { (0, 0, 0) };

        foreach (var p in placements)
        {
            positions.Add((p.X + p.W, p.Y, p.Z));
            positions.Add((p.X, p.Y + p.H, p.Z));
            positions.Add((p.X, p.Y, p.Z + p.D));
        }

        return positions;
    }

    static bool CanPlace(int x, int y, int z, int w, int h, int d, List<Placement> placements)
    {
        foreach (var p in placements)
        {
            if (!(x + w <= p.X || x >= p.X + p.W ||
                  y + h <= p.Y || y >= p.Y + p.H ||
                  z + d <= p.Z || z >= p.Z + p.D))
            {
                return false;
            }
        }
        return true;
    }

    static Placement? FindPlacementWithOrientation(Box box, int x, int y, int z, List<Placement> placements, int W, int H, int D)
    {
        var orientations = GetOrientations(box);
        foreach (var (w, h, d) in orientations)
        {
            if (x + w <= W && y + h <= H && z + d <= D && CanPlace(x, y, z, w, h, d, placements))
            {
                return new Placement { X = x, Y = y, Z = z, W = w, H = h, D = d };
            }
        }
        return null;
    }
}
