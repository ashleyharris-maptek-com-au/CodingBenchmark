OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Input/Output decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Type declarations
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v2 = OpTypePointer Input %v2float
%ptr_out_v4 = OpTypePointer Output %v4float

%main_func_type = OpTypeFunction %void

; Constants
%c_i_0 = OpConstant %int 0
%c_i_3 = OpConstant %int 3
%c_i_4 = OpConstant %int 4

%c_f_0 = OpConstant %float 0.0
%c_f_0p1 = OpConstant %float 0.1
%c_f_0p6 = OpConstant %float 0.6
%c_f_0p8 = OpConstant %float 0.8
%c_f_1 = OpConstant %float 1.0
%c_f_64 = OpConstant %float 64.0

; Global variables
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_uv = OpVariable %ptr_in_v2 Input
%in_tangent = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input
%out_color = OpVariable %ptr_out_v4 Output
%ubo_var = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void None %main_func_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal

; Normalize normal
%N = OpNormalize %v3float %normal

; Load light position from UBO (member 3)
%light_v4_ptr = OpAccessChain %ptr_uniform_v4 %ubo_var %c_i_3
%lightPos_v4 = OpLoad %v4float %light_v4_ptr
%lightPos = OpVectorShuffle %v3float %lightPos_v4 %lightPos_v4 0 1 2

; Load camera position from UBO (member 4)
%camera_v4_ptr = OpAccessChain %ptr_uniform_v4 %ubo_var %c_i_4
%cameraPos_v4 = OpLoad %v4float %camera_v4_ptr
%cameraPos = OpVectorShuffle %v3float %cameraPos_v4 %cameraPos_v4 0 1 2

; Calculate L = normalize(lightPos - worldPos)
%L_unnorm = OpFSub %v3float %lightPos %worldPos
%L = OpNormalize %v3float %L_unnorm

; Calculate V = normalize(cameraPos - worldPos)
%V_unnorm = OpFSub %v3float %cameraPos %worldPos
%V = OpNormalize %v3float %V_unnorm

; Calculate H = normalize(L + V)
%LplusV = OpFAdd %v3float %L %V
%H = OpNormalize %v3float %LplusV

; Calculate diffuse = max(dot(N, L), 0)
%dot_NL = OpDot %float %N %L
%diffuse = OpFMax %float %dot_NL %c_f_0

; Calculate specular = pow(max(dot(N, H), 0), 64)
%dot_NH = OpDot %float %N %H
%dot_NH_clamped = OpFMax %float %dot_NH %c_f_0
%specular = OpPow %float %dot_NH_clamped %c_f_64

; Calculate intensity = 0.1 + 0.6*diffuse + 0.8*specular
%diffuse_term = OpFMul %float %c_f_0p6 %diffuse
%with_diffuse = OpFAdd %float %c_f_0p1 %diffuse_term
%specular_term = OpFMul %float %c_f_0p8 %specular
%intensity = OpFAdd %float %with_diffuse %specular_term

; Clamp intensity to [0, 1]
%intensity_clamped = OpFMin %float %intensity %c_f_1
%intensity_final = OpFMax %float %intensity_clamped %c_f_0

; Create output color (R, G, B, A)
%out_rgba = OpCompositeConstruct %v4float %intensity_final %intensity_final %intensity_final %c_f_1
OpStore %out_color %out_rgba

OpReturn
OpFunctionEnd
