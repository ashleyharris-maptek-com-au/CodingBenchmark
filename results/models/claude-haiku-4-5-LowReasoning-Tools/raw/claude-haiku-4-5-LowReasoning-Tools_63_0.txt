declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Point {
  x: number;
  y: number;
}

function bfs(grid: string[], start: Point, end: Point): number {
  const rows = grid.length;
  const cols = grid[0].length;

  if (start.x === end.x && start.y === end.y) {
    return 0;
  }

  const visited = new Set<string>();
  const queue: Array<[number, number, number]> = [[start.x, start.y, 0]];
  visited.add(`${start.x},${start.y}`);

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift()!;

    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]];

    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.') {
        const key = `${nx},${ny}`;

        if (!visited.has(key)) {
          visited.add(key);

          if (nx === end.x && ny === end.y) {
            return dist + 1;
          }

          queue.push([nx, ny, dist + 1]);
        }
      }
    }
  }

  return Infinity;
}

function solve(data: any) {
  const { grid, robots, jobs } = data;

  const assignmentMap = new Map<number, number[]>();
  for (const robot of robots) {
    assignmentMap.set(robot.id, []);
  }

  // Precompute distances from robots to jobs
  const distances = new Map<string, number>();
  for (const robot of robots) {
    for (const job of jobs) {
      const distance = bfs(grid, { x: robot.x, y: robot.y }, { x: job.x, y: job.y });
      distances.set(`${robot.id},${job.id}`, distance);
    }
  }

  // Sort jobs by priority (descending) and then by id (ascending)
  const sortedJobs = [...jobs].sort((a, b) => b.priority - a.priority || a.id - b.id);

  // Assign each job to the closest robot
  for (const job of sortedJobs) {
    let bestRobot = -1;
    let bestDistance = Infinity;

    for (const robot of robots) {
      const distance = distances.get(`${robot.id},${job.id}`) || Infinity;
      if (distance < bestDistance) {
        bestDistance = distance;
        bestRobot = robot.id;
      }
    }

    if (bestRobot !== -1) {
      assignmentMap.get(bestRobot)!.push(job.id);
    }
  }

  // Convert map to array
  const maxRobotId = Math.max(...robots.map(r => r.id));
  const assignments: number[][] = [];
  for (let i = 0; i <= maxRobotId; i++) {
    assignments.push(assignmentMap.get(i) || []);
  }

  return { assignments };
}

const result = solve(data);
