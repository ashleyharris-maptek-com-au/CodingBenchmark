; SPIR-V Assembly: Flat-shaded geometry shader with simple lighting
; Input: InputTrianglesAdjacency (6 vertices)
; Output: OutputTriangleStrip (3 vertices)
; Uses main triangle vertices at indices 0, 2, 4

OpCapability Geometry
OpCapability GLSL450
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Location 0
OpDecorate %inColor Location 0
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%int_6 = OpConstant %int 6
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3.0

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr6_pv = OpTypeArray %gl_PerVertex_in %int_6
%arr6_v3 = OpTypeArray %v3float %int_6

%ptr_in_arr_pv = OpTypePointer Input %arr6_pv
%ptr_in_arr_v3 = OpTypePointer Input %arr6_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%func_type = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load positions at indices 0, 2, 4
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; Extract xyz components
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p4_xyz = OpVectorShuffle %v3float %p4 %p4 0 1 2

; Compute edge vectors
%e1 = OpFSub %v3float %p2_xyz %p0_xyz
%e2 = OpFSub %v3float %p4_xyz %p0_xyz

; Extract components for cross product
%e1_x = OpVectorExtractDynamic %float %e1 %int_0
%e1_y = OpVectorExtractDynamic %float %e1 %int_1
%e1_z = OpVectorExtractDynamic %float %e1 %int_2
%e2_x = OpVectorExtractDynamic %float %e2 %int_0
%e2_y = OpVectorExtractDynamic %float %e2 %int_1
%e2_z = OpVectorExtractDynamic %float %e2 %int_2

; Compute cross product manually: (a×b) = (a.y*b.z-a.z*b.y, a.z*b.x-a.x*b.z, a.x*b.y-a.y*b.x)
%e1y_e2z = OpFMul %float %e1_y %e2_z
%e1z_e2y = OpFMul %float %e1_z %e2_y
%cx = OpFSub %float %e1y_e2z %e1z_e2y

%e1z_e2x = OpFMul %float %e1_z %e2_x
%e1x_e2z = OpFMul %float %e1_x %e2_z
%cy = OpFSub %float %e1z_e2x %e1x_e2z

%e1x_e2y = OpFMul %float %e1_x %e2_y
%e1y_e2x = OpFMul %float %e1_y %e2_x
%cz = OpFSub %float %e1x_e2y %e1y_e2x

%cross = OpCompositeConstruct %v3float %cx %cy %cz

; Normalize face normal (Normalize = 69)
%normal = OpExtInst %v3float %glsl 69 %cross

; Load colors at indices 0, 2, 4
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr

%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr

%c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%c4 = OpLoad %v3float %c4_ptr

; Average colors
%sum1 = OpFAdd %v3float %c0 %c2
%sum2 = OpFAdd %v3float %sum1 %c4
%base_color = OpFDiv %v3float %sum2 %float_3

; Light direction: normalize(vec3(1, 1, 1))
%light_unnorm = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%light_dir = OpExtInst %v3float %glsl 69 %light_unnorm

; Diffuse: max(dot(normal, light), 0.0) (FMax = 40)
%dot_val = OpDot %float %normal %light_dir
%diffuse = OpExtInst %float %glsl 40 %float_0 %dot_val

; Final color: base * (0.2 + 0.8 * diffuse)
%light_mult = OpFMul %float %float_0_8 %diffuse
%light_factor = OpFAdd %float %float_0_2 %light_mult
%final_color = OpVectorTimesScalar %v3float %base_color %light_factor

; Emit vertex 0
%pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr %p0
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 2
OpStore %pos_ptr %p2
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 4
OpStore %pos_ptr %p4
OpStore %outColor %final_color
OpEmitVertex

; End primitive
OpEndPrimitive
OpReturn
OpFunctionEnd
