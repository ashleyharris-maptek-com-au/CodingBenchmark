#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> color(n, -1);

    // Sort vertices by degree (descending) - Welsh-Powell algorithm
    vector<int> order(n);
    for (int i = 0; i < n; i++) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        return adj[a].size() > adj[b].size();
    });

    // Greedy coloring: assign smallest available color to each vertex
    for (int vertex : order) {
        vector<bool> used(k, false);
        for (int neighbor : adj[vertex]) {
            if (color[neighbor] != -1) {
                used[color[neighbor]] = true;
            }
        }

        // Assign smallest available color
        for (int c = 0; c < k; c++) {
            if (!used[c]) {
                color[vertex] = c;
                break;
            }
        }
    }

    // Output colors for each vertex
    for (int i = 0; i < n; i++) {
        cout << color[i];
        if (i < n - 1) cout << " ";
    }
    cout << "\n";

    return 0;
}
