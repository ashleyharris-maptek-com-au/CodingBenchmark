@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

fn custom_hash(data: array<u32, 16>, nonce: u32) -> vec2<u32> {
    var state: array<u32, 16>;

    // Initialize state and XOR with nonce
    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = data[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing
    for (var round: u32 = 0u; round < 8u; round = round + 1u) {
        // Group 0: state[0..3]
        var a = state[0u];
        var b = state[1u];
        var c = state[2u];
        var d = state[3u];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[0u] = a;
        state[1u] = b;
        state[2u] = c;
        state[3u] = d;

        // Group 1: state[4..7]
        a = state[4u];
        b = state[5u];
        c = state[6u];
        d = state[7u];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[4u] = a;
        state[5u] = b;
        state[6u] = c;
        state[7u] = d;

        // Group 2: state[8..11]
        a = state[8u];
        b = state[9u];
        c = state[10u];
        d = state[11u];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[8u] = a;
        state[9u] = b;
        state[10u] = c;
        state[11u] = d;

        // Group 3: state[12..15]
        a = state[12u];
        b = state[13u];
        c = state[14u];
        d = state[15u];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[12u] = a;
        state[13u] = b;
        state[14u] = c;
        state[15u] = d;
    }

    // Fold 16 words to 2
    var h0 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^ state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    var h1 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^ state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

    return vec2<u32>(h0, h1);
}

fn count_trailing_zeros(h0: u32, h1: u32) -> u32 {
    if (h0 == 0u) {
        if (h1 == 0u) {
            return 64u;
        }
        return 32u + countTrailingZeros(h1);
    }
    return countTrailingZeros(h0);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    // Guard: if out of range, return
    if (gid.x >= params.z) {
        return;
    }

    let my_nonce = params.y + gid.x;
    let hash_result = custom_hash(baseData, my_nonce);
    let h0 = hash_result.x;
    let h1 = hash_result.y;

    let trailing_zeros = count_trailing_zeros(h0, h1);

    if (trailing_zeros >= params.x) {
        result[0u] = 1u;
        result[1u] = my_nonce;
        result[2u] = h0;
        result[3u] = h1;
    }
}
