use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n = parts[0].parse::<usize>().unwrap();
    let m = parts[1].parse::<usize>().unwrap();

    let mut edges = Vec::new();
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u = parts[0].parse::<usize>().unwrap();
        let v = parts[1].parse::<usize>().unwrap();
        edges.push((u, v));
        adj[u].push(v);
        adj[v].push(u);
    }

    let target_size = n / 2;
    let mut best_partition = vec![false; n];
    let mut best_cut = m + 1;

    let iterations = std::cmp::min(100, std::cmp::max(10, 1000 / n));

    for trial in 0..iterations {
        let mut partition = vec![false; n];

        if trial == 0 {
            for i in 0..target_size {
                partition[i] = true;
            }
        } else {
            let seed = trial as u64;
            let mut rng = seed;
            for _ in 0..target_size {
                rng = rng.wrapping_mul(1103515245).wrapping_add(12345);
                let idx = (rng as usize) % n;
                if !partition[idx] {
                    partition[idx] = true;
                } else {
                    for j in 0..n {
                        if !partition[j] {
                            partition[j] = true;
                            break;
                        }
                    }
                }
            }
        }

        partition = local_search(&edges, &adj, partition, target_size, n);

        let cut = calculate_cut(&edges, &partition);
        if cut < best_cut {
            best_cut = cut;
            best_partition = partition;
        }
    }

    let mut partition_a = Vec::new();
    for i in 0..n {
        if best_partition[i] {
            partition_a.push(i);
        }
    }

    println!("{}", best_cut);
    for (i, &v) in partition_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}

fn calculate_cut(edges: &[(usize, usize)], partition: &[bool]) -> usize {
    edges.iter().filter(|(u, v)| partition[*u] != partition[*v]).count()
}

fn local_search(
    edges: &[(usize, usize)],
    adj: &[Vec<usize>],
    mut partition: Vec<bool>,
    target_size: usize,
    n: usize,
) -> Vec<bool> {
    let mut improved = true;
    let mut iterations = 0;
    let max_iterations = 100;

    while improved && iterations < max_iterations {
        improved = false;
        iterations += 1;

        for i in 0..n {
            let in_a = partition[i];
            let a_count = partition.iter().filter(|&&x| x).count();

            let can_swap = (in_a && a_count > target_size) || (!in_a && a_count < target_size);

            if !can_swap {
                continue;
            }

            partition[i] = !partition[i];
            let new_cut = calculate_cut(edges, &partition);
            partition[i] = !partition[i];
            let old_cut = calculate_cut(edges, &partition);

            if new_cut < old_cut {
                partition[i] = !partition[i];
                improved = true;
            }
        }
    }

    partition
}
