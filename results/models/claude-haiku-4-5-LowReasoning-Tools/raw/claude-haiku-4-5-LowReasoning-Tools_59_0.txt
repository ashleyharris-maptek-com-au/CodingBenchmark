const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { probability, drones, maxStepsPerDrone } = data;

const rows = probability.length;
const cols = probability[0].length;

// Track which cells have been covered
const covered = new Set();

// Direction vectors: up, down, left, right, stay
const directions = [
  [-1, 0], // up
  [1, 0],  // down
  [0, -1], // left
  [0, 1],  // right
  [0, 0]   // stay
];

function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && probability[x][y] !== -1;
}

function planPath(startX, startY, maxSteps) {
  const path = [[startX, startY]];
  let currentX = startX;
  let currentY = startY;

  // Mark starting position as covered
  const cellKey = `${currentX},${currentY}`;
  if (probability[currentX][currentY] > 0) {
    covered.add(cellKey);
  }

  for (let step = 0; step < maxSteps; step++) {
    let bestDir = null;
    let bestScore = -Infinity;

    // Evaluate all possible moves
    for (const [dx, dy] of directions) {
      const nx = currentX + dx;
      const ny = currentY + dy;

      if (isValid(nx, ny)) {
        const cellKey = `${nx},${ny}`;
        const cellValue = probability[nx][ny];

        // Score: uncovered cells with positive value are prioritized
        let score = 0;
        if (!covered.has(cellKey) && cellValue > 0) {
          score = cellValue;
        }

        // Tie-breaking: prefer movement over staying in place
        if (dx !== 0 || dy !== 0) {
          score += 0.00001;
        }

        if (score > bestScore) {
          bestScore = score;
          bestDir = [dx, dy];
        }
      }
    }

    if (bestDir !== null) {
      currentX += bestDir[0];
      currentY += bestDir[1];
      path.push([currentX, currentY]);

      const cellKey = `${currentX},${currentY}`;
      if (probability[currentX][currentY] > 0) {
        covered.add(cellKey);
      }
    }
  }

  return path;
}

// Plan paths for all drones
const paths = drones.map(drone =>
  planPath(drone.x, drone.y, maxStepsPerDrone)
);

// Output the result
console.log(JSON.stringify({ paths }));
