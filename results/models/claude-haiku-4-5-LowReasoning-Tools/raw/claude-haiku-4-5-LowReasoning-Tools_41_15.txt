OpCapability Shader
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Input decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

; Output decoration
OpDecorate %out_color Location 0

; UBO decorations
OpDecorate %ubo_type Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; UBO member decorations
OpMemberDecorate %ubo_type 0 Offset 0
OpMemberDecorate %ubo_type 1 Offset 64
OpMemberDecorate %ubo_type 2 Offset 128
OpMemberDecorate %ubo_type 3 Offset 192
OpMemberDecorate %ubo_type 4 Offset 208
OpMemberDecorate %ubo_type 5 Offset 224
OpMemberDecorate %ubo_type 0 ColMajor
OpMemberDecorate %ubo_type 0 MatrixStride 16
OpMemberDecorate %ubo_type 1 ColMajor
OpMemberDecorate %ubo_type 1 MatrixStride 16
OpMemberDecorate %ubo_type 2 ColMajor
OpMemberDecorate %ubo_type 2 MatrixStride 16

; Type definitions
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO struct type
%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointer types
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_ubo_type = OpTypePointer Uniform %ubo_type
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Function type
%func_type = OpTypeFunction %void

; Constants
%f0 = OpConstant %float 0.0
%f0p2 = OpConstant %float 0.2
%f0p4 = OpConstant %float 0.4
%f0p5 = OpConstant %float 0.5
%f0p7 = OpConstant %float 0.7
%f0p75 = OpConstant %float 0.75
%f1 = OpConstant %float 1.0
%i0 = OpConstant %int 0
%i3 = OpConstant %int 3

; Variables
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output
%ubo_var = OpVariable %ptr_Uniform_ubo_type Uniform

; Extended instruction set
%glsl = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal

; Load light position from UBO (member 3)
%lightPtr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %i3
%lightVec4 = OpLoad %vec4 %lightPtr

; Extract xyz from light position
%lightPos = OpVectorShuffle %vec3 %lightVec4 %lightVec4 0 1 2

; Calculate light direction L = normalize(lightPos - worldPos)
%L_unnorm = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %glsl Normalize %L_unnorm

; Normalize normal
%N = OpExtInst %vec3 %glsl Normalize %normal

; Calculate t = dot(N, L) * 0.5 + 0.5
%dotNL = OpDot %float %N %L
%t_scaled = OpFMul %float %dotNL %f0p5
%t = OpFAdd %float %t_scaled %f0p5

; Clamp t to [0, 1]
%t_clamped = OpExtInst %float %glsl FClamp %t %f0 %f1

; Create cool and warm colors
%cool = OpCompositeConstruct %vec3 %f0p2 %f0p2 %f0p75
%warm = OpCompositeConstruct %vec3 %f0p7 %f0p7 %f0p4

; Interpolate: result = cool * (1 - t) + warm * t
%one_minus_t = OpFSub %float %f1 %t_clamped
%cool_part = OpVectorTimesScalar %vec3 %cool %one_minus_t
%warm_part = OpVectorTimesScalar %vec3 %warm %t_clamped
%result_color = OpFAdd %vec3 %cool_part %warm_part

; Create final output with alpha = 1.0
%output = OpCompositeConstruct %vec4 %result_color %f1

; Store result
OpStore %out_color %output

OpReturn
OpFunctionEnd
