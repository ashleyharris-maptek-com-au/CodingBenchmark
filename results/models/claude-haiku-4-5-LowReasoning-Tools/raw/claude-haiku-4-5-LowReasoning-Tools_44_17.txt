#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Scale UV coordinates by 8x16
    vec2 scaledUV = uv * vec2(8.0, 16.0);

    // Stagger rows: offset odd rows by half a brick width
    float row = floor(scaledUV.y);
    if (mod(row, 2.0) > 0.5) {
        scaledUV.x += 0.5;
    }

    // Get the fractional part within each brick unit
    vec2 brickFrac = fract(scaledUV);

    // Mortar thickness: 0.05 in X, 0.1 in Y
    const float mortarX = 0.05;
    const float mortarY = 0.1;

    // Determine if we're in mortar or brick
    bool inMortarX = brickFrac.x < mortarX || brickFrac.x > (1.0 - mortarX);
    bool inMortarY = brickFrac.y < mortarY || brickFrac.y > (1.0 - mortarY);
    bool isMortar = inMortarX || inMortarY;

    // Choose color: mortar is (0.7, 0.7, 0.7), brick is (0.7, 0.2, 0.1)
    vec3 baseColor = isMortar ? vec3(0.7, 0.7, 0.7) : vec3(0.7, 0.2, 0.1);

    // Apply Lambertian shading
    vec3 n = normalize(normal);
    vec3 l = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(n, l), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Final color with shading and alpha 1.0
    vec3 finalColor = baseColor * shadingFactor;
    outColor = vec4(finalColor, 1.0);
}
