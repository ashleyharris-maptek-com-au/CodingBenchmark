using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    private int width;
    private int height;
    private bool[,] board;

    // Piece definitions: each piece type maps to its rotations
    // Each rotation is a list of (row, col) offsets from anchor point
    private static readonly Dictionary<char, List<int[][]>> PieceDefinitions = new()
    {
        {'I', new List<int[][]>
        {
            new[] {new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {0,3}},
            new[] {new[] {0,0}, new[] {1,0}, new[] {2,0}, new[] {3,0}}
        }},
        {'O', new List<int[][]>
        {
            new[] {new[] {0,0}, new[] {0,1}, new[] {1,0}, new[] {1,1}}
        }},
        {'T', new List<int[][]>
        {
            new[] {new[] {0,1}, new[] {1,0}, new[] {1,1}, new[] {1,2}},
            new[] {new[] {0,0}, new[] {1,0}, new[] {1,1}, new[] {2,0}},
            new[] {new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {1,1}},
            new[] {new[] {0,1}, new[] {1,0}, new[] {1,1}, new[] {2,1}}
        }},
        {'S', new List<int[][]>
        {
            new[] {new[] {0,1}, new[] {0,2}, new[] {1,0}, new[] {1,1}},
            new[] {new[] {0,0}, new[] {1,0}, new[] {1,1}, new[] {2,1}}
        }},
        {'Z', new List<int[][]>
        {
            new[] {new[] {0,0}, new[] {0,1}, new[] {1,1}, new[] {1,2}},
            new[] {new[] {0,1}, new[] {1,0}, new[] {1,1}, new[] {2,0}}
        }},
        {'J', new List<int[][]>
        {
            new[] {new[] {0,0}, new[] {1,0}, new[] {1,1}, new[] {1,2}},
            new[] {new[] {0,0}, new[] {0,1}, new[] {1,0}, new[] {2,0}},
            new[] {new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {1,2}},
            new[] {new[] {0,0}, new[] {1,0}, new[] {2,0}, new[] {2,1}}
        }},
        {'L', new List<int[][]>
        {
            new[] {new[] {0,2}, new[] {1,0}, new[] {1,1}, new[] {1,2}},
            new[] {new[] {0,0}, new[] {1,0}, new[] {2,0}, new[] {2,1}},
            new[] {new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {1,0}},
            new[] {new[] {0,0}, new[] {0,1}, new[] {1,1}, new[] {2,1}}
        }}
    };

    static void Main()
    {
        string[] dims = Console.ReadLine().Split();
        int w = int.Parse(dims[0]);
        int h = int.Parse(dims[1]);
        int numPieces = int.Parse(dims[2]);

        var player = new TetrisPlayer(w, h);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine().Trim();
            var (rotation, column) = player.FindBestPlacement(pieceType);

            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();

            if (result == "gameover")
                break;

            player.PlacePiece(pieceType, rotation, column);
        }
    }

    public TetrisPlayer(int width, int height)
    {
        this.width = width;
        this.height = height;
        this.board = new bool[height, width];
    }

    public (int rotation, int column) FindBestPlacement(char pieceType)
    {
        var rotations = PieceDefinitions[pieceType];
        int bestRotation = 0;
        int bestColumn = 0;
        double bestScore = double.MinValue;

        for (int rot = 0; rot < rotations.Count; rot++)
        {
            for (int col = 0; col < width; col++)
            {
                int dropRow = GetDropRow(rotations[rot], col);
                if (dropRow >= 0)
                {
                    double score = EvaluatePlacement(rotations[rot], dropRow, col);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRotation = rot;
                        bestColumn = col;
                    }
                }
            }
        }

        return (bestRotation, bestColumn);
    }

    private int GetDropRow(int[][] shape, int column)
    {
        for (int dropRow = height - 1; dropRow >= 0; dropRow--)
        {
            bool canPlace = true;

            foreach (var cell in shape)
            {
                int row = dropRow - cell[0];
                int col = column + cell[1];

                if (col < 0 || col >= width || row < 0)
                {
                    canPlace = false;
                    break;
                }

                if (row >= height || board[row, col])
                {
                    canPlace = false;
                    break;
                }
            }

            if (canPlace)
                return dropRow;
        }

        return -1;
    }

    private double EvaluatePlacement(int[][] shape, int dropRow, int column)
    {
        var testBoard = (bool[,])board.Clone();

        // Place piece on test board
        foreach (var cell in shape)
        {
            int row = dropRow - cell[0];
            int col = column + cell[1];
            if (row >= 0 && row < height && col >= 0 && col < width)
                testBoard[row, col] = true;
        }

        // Clear complete lines
        int clearedLines = 0;
        for (int row = 0; row < height; row++)
        {
            bool complete = true;
            for (int col = 0; col < width; col++)
            {
                if (!testBoard[row, col])
                {
                    complete = false;
                    break;
                }
            }
            if (complete)
                clearedLines++;
        }

        int maxHeight = GetMaxHeight(testBoard);
        int holes = CountHoles(testBoard);
        double roughness = GetRoughness(testBoard);

        // Adaptive scoring based on board state
        double score = clearedLines * 500 - maxHeight * 5 - holes * 100 - roughness * 2;

        // Early game: prioritize height control
        if (maxHeight < height / 3)
            score -= maxHeight * 10;

        // Late game: prioritize clearing lines
        if (maxHeight > height * 2 / 3 && clearedLines > 0)
            score += clearedLines * 1000;

        return score;
    }

    private int GetMaxHeight(bool[,] testBoard)
    {
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                if (testBoard[row, col])
                    return height - row;
            }
        }
        return 0;
    }

    private int CountHoles(bool[,] testBoard)
    {
        int holes = 0;
        for (int col = 0; col < width; col++)
        {
            bool blockFound = false;
            for (int row = 0; row < height; row++)
            {
                if (testBoard[row, col])
                {
                    blockFound = true;
                }
                else if (blockFound)
                {
                    holes++;
                }
            }
        }
        return holes;
    }

    private double GetRoughness(bool[,] testBoard)
    {
        var heights = new int[width];

        for (int col = 0; col < width; col++)
        {
            for (int row = 0; row < height; row++)
            {
                if (testBoard[row, col])
                {
                    heights[col] = height - row;
                    break;
                }
            }
        }

        double roughness = 0;
        for (int col = 0; col < width - 1; col++)
        {
            roughness += Math.Abs(heights[col] - heights[col + 1]);
        }

        return roughness;
    }

    public void PlacePiece(char pieceType, int rotation, int column)
    {
        var shape = PieceDefinitions[pieceType][rotation % PieceDefinitions[pieceType].Count];
        int dropRow = GetDropRow(shape, column);

        if (dropRow >= 0)
        {
            foreach (var cell in shape)
            {
                int row = dropRow - cell[0];
                int col = column + cell[1];
                if (row >= 0 && row < height && col >= 0 && col < width)
                    board[row, col] = true;
            }

            ClearLines();
        }
    }

    private void ClearLines()
    {
        var newBoard = new bool[height, width];
        int writeRow = height - 1;

        for (int readRow = height - 1; readRow >= 0; readRow--)
        {
            bool complete = true;
            for (int col = 0; col < width; col++)
            {
                if (!board[readRow, col])
                {
                    complete = false;
                    break;
                }
            }

            if (!complete)
            {
                for (int col = 0; col < width; col++)
                {
                    newBoard[writeRow, col] = board[readRow, col];
                }
                writeRow--;
            }
        }

        board = newBoard;
    }
}
