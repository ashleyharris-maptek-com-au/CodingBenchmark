def autopilot_step(state: dict, dt: float) -> dict:
    import math

    # Module-level state persistence
    if not hasattr(autopilot_step, 'initialized'):
        autopilot_step.initialized = True
        autopilot_step.phase = 'intercept'
        autopilot_step.go_around_count = 0
        autopilot_step.flare_started = False
        autopilot_step.touchdown_detected = False
        autopilot_step.descent_rate_history = []
        autopilot_step.glideslope_deviation_history = []
        autopilot_step.last_radio_alt = None

    # Reset on ground after touchdown
    if state.get('on_ground') and state.get('weight_on_wheels'):
        if not autopilot_step.touchdown_detected:
            autopilot_step.touchdown_detected = True
            autopilot_step.phase = 'touchdown'

    # Sensor fusion - get best available readings
    def get_valid_glideslope_dots():
        readings = []
        for i in [1, 2, 3]:
            key_dots = f'glideslope_{i}_dots'
            key_valid = f'glideslope_{i}_valid'
            if state.get(key_valid) and key_dots in state:
                readings.append(state[key_dots])
        return readings

    def get_valid_localizer_dots():
        readings = []
        for i in [1, 2, 3]:
            key_dots = f'localizer_{i}_dots'
            key_valid = f'localizer_{i}_valid'
            if state.get(key_valid) and key_dots in state:
                readings.append(state[key_dots])
        return readings

    def get_valid_airspeed():
        readings = []
        for i in [1, 2, 3]:
            key = f'ias_{i}_kt'
            key_valid = f'ias_{i}_valid'
            if state.get(key_valid) and key in state:
                readings.append(state[key])
        return readings

    def get_valid_altitude():
        readings = []
        for i in [1, 2, 3]:
            key = f'baro_alt_{i}_ft'
            key_valid = f'baro_alt_{i}_valid'
            if state.get(key_valid) and key in state:
                readings.append(state[key])
        return readings

    def median(lst):
        if not lst:
            return 0
        sorted_lst = sorted(lst)
        n = len(sorted_lst)
        if n % 2 == 1:
            return sorted_lst[n // 2]
        return (sorted_lst[n // 2 - 1] + sorted_lst[n // 2]) / 2

    def mean(lst):
        return sum(lst) / len(lst) if lst else 0

    # Get best readings using sensor fusion
    gs_dots_readings = get_valid_glideslope_dots()
    loc_dots_readings = get_valid_localizer_dots()
    airspeed_readings = get_valid_airspeed()
    altitude_readings = get_valid_altitude()

    gs_dots = median(gs_dots_readings) if gs_dots_readings else 0
    loc_dots = median(loc_dots_readings) if loc_dots_readings else 0
    ias_kt = median(airspeed_readings) if airspeed_readings else 100
    baro_alt_ft = median(altitude_readings) if altitude_readings else 10000

    radio_alt_ft = state.get('radio_alt_ft', 1000)
    target_ias_kt = state.get('target_ias_kt', 137)
    target_vref_kt = state.get('target_vref_kt', 137)
    below_decision_height = state.get('below_decision_height', False)

    # Calculate descent rate
    if autopilot_step.last_radio_alt is not None:
        descent_rate = (autopilot_step.last_radio_alt - radio_alt_ft) / dt if dt > 0 else 0
        autopilot_step.descent_rate_history.append(descent_rate)
        if len(autopilot_step.descent_rate_history) > 20:
            autopilot_step.descent_rate_history.pop(0)
    autopilot_step.last_radio_alt = radio_alt_ft

    avg_descent_rate = mean(autopilot_step.descent_rate_history) if autopilot_step.descent_rate_history else 0

    # Phase determination
    if radio_alt_ft > 1000:
        autopilot_step.phase = 'intercept'
    elif radio_alt_ft > 200:
        autopilot_step.phase = 'approach'
    elif radio_alt_ft > 50:
        autopilot_step.phase = 'flare'
        autopilot_step.flare_started = True
    elif radio_alt_ft > 0 and (state.get('weight_on_wheels') or state.get('on_ground')):
        autopilot_step.phase = 'touchdown'

    # Go-around detection: unstable approach
    is_unstable = False
    if autopilot_step.phase in ['intercept', 'approach']:
        if abs(gs_dots) > 1.5:
            is_unstable = True
        elif abs(loc_dots) > 1.5:
            is_unstable = True
        elif abs(ias_kt - target_ias_kt) > 15:
            is_unstable = True
        elif below_decision_height and abs(gs_dots) > 0.5:
            is_unstable = True

    if is_unstable and autopilot_step.go_around_count < 10:
        autopilot_step.go_around_count += 1
        return {
            'throttle_1': 1.0,
            'throttle_2': 1.0,
            'elevator': 0.3,
            'speedbrake': 0.0,
            'flaps': 0
        }

    output = {}

    # Elevator control - primary glideslope tracking
    if autopilot_step.phase == 'intercept' or autopilot_step.phase == 'approach':
        gs_error = gs_dots
        elevator_cmd = -gs_error * 0.15

        if autopilot_step.glideslope_deviation_history:
            gs_rate = (gs_error - autopilot_step.glideslope_deviation_history[-1]) / dt if dt > 0 else 0
            elevator_cmd -= gs_rate * 0.05

        autopilot_step.glideslope_deviation_history.append(gs_error)
        if len(autopilot_step.glideslope_deviation_history) > 20:
            autopilot_step.glideslope_deviation_history.pop(0)

        elevator_cmd = max(-0.8, min(0.8, elevator_cmd))
        output['elevator'] = elevator_cmd

    elif autopilot_step.phase == 'flare':
        target_descent = 300
        descent_error = abs(avg_descent_rate) - target_descent
        elevator_cmd = descent_error * 0.001
        elevator_cmd = max(-0.3, min(0.5, elevator_cmd))
        output['elevator'] = elevator_cmd
        output['throttle_1'] = 0.1
        output['throttle_2'] = 0.1

    elif autopilot_step.phase == 'touchdown':
        output['elevator'] = 0.0
        output['throttle_1'] = 0.0
        output['throttle_2'] = 0.0
        output['spoiler'] = 1.0
        output['reverse_1'] = 0.8
        output['reverse_2'] = 0.8
        output['autobrake'] = 1.0

    else:
        output['elevator'] = 0.0

    # Aileron control - localizer tracking
    if autopilot_step.phase in ['intercept', 'approach', 'flare']:
        loc_error = loc_dots
        aileron_cmd = loc_error * 0.12
        aileron_cmd = max(-0.7, min(0.7, aileron_cmd))
        output['aileron'] = aileron_cmd
    else:
        output['aileron'] = 0.0

    # Speed control via throttle
    if autopilot_step.phase == 'intercept':
        speed_error = ias_kt - target_ias_kt
        throttle_cmd = 0.4 - speed_error * 0.003
        throttle_cmd = max(0.15, min(0.8, throttle_cmd))
        output['throttle_1'] = throttle_cmd
        output['throttle_2'] = throttle_cmd
        output['flaps'] = 15
        output['gear'] = 1.0

    elif autopilot_step.phase == 'approach':
        speed_error = ias_kt - target_ias_kt
        throttle_cmd = 0.25 - speed_error * 0.004
        throttle_cmd = max(0.05, min(0.5, throttle_cmd))
        output['throttle_1'] = throttle_cmd
        output['throttle_2'] = throttle_cmd
        output['flaps'] = 30
        output['gear'] = 1.0
        output['speedbrake'] = 0.0

    elif autopilot_step.phase == 'flare':
        output['throttle_1'] = 0.05
        output['throttle_2'] = 0.05
        output['flaps'] = 40
        output['gear'] = 1.0

    # Rudder trim for crosswind
    if autopilot_step.phase in ['intercept', 'approach']:
        output['rudder'] = 0.0

    return output
