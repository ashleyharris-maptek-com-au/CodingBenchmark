OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inWorldPos Location 0
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               OpTypeVoid %void
               OpTypeFloat %float 32
               OpTypeInt %int 32 1
               OpTypeInt %uint 32 0
               OpTypeVector %v3float %float 3
               OpTypeVector %v4float %float 4

        %uint_0 = OpConstant %uint 0
        %uint_1 = OpConstant %uint 1
         %int_0 = OpConstant %int 0
        %float_0 = OpConstant %float 0.0
        %float_1 = OpConstant %float 1.0
     %float_0_05 = OpConstant %float 0.05

       %gl_PerVertex_in = OpTypeStruct %v4float
    %arr_v4_1 = OpTypeArray %gl_PerVertex_in %uint_1
 %ptr_Input_arr_v4_1 = OpTypePointer Input %arr_v4_1
        %gl_in = OpVariable %ptr_Input_arr_v4_1 Input

       %gl_PerVertex_out = OpTypeStruct %v4float
    %ptr_Output_v4 = OpTypePointer Output %v4float
       %gl_out = OpVariable %ptr_Output_v4 Output

       %arr_v3_1 = OpTypeArray %v3float %uint_1
 %ptr_Input_arr_v3_1 = OpTypePointer Input %arr_v3_1
    %inWorldPos = OpVariable %ptr_Input_arr_v3_1 Input
       %inColor = OpVariable %ptr_Input_arr_v3_1 Input

    %ptr_Output_v3 = OpTypePointer Output %v3float
     %outColor = OpVariable %ptr_Output_v3 Output

     %ptr_Input_v4 = OpTypePointer Input %v4float
     %ptr_Input_v3 = OpTypePointer Input %v3float

    %func_type = OpTypeFunction %void
         %main = OpFunction %void None %func_type
        %entry = OpLabel

        %in_pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %uint_0 %uint_0
          %in_pos = OpLoad %v4float %in_pos_ptr

       %in_color_ptr = OpAccessChain %ptr_Input_v3 %inColor %uint_0
        %in_color = OpLoad %v3float %in_color_ptr

              %x = OpCompositeExtract %float %in_pos 0
              %y = OpCompositeExtract %float %in_pos 1
              %h = %float_0_05

              %y_top = OpFAdd %float %y %h
             %top_pos = OpCompositeConstruct %v4float %x %y_top %float_0 %float_1
                        OpStore %gl_out %top_pos
                        OpStore %outColor %in_color
                        OpEmitVertex

             %x_right = OpFAdd %float %x %h
            %right_pos = OpCompositeConstruct %v4float %x_right %y %float_0 %float_1
                        OpStore %gl_out %right_pos
                        OpStore %outColor %in_color
                        OpEmitVertex

              %x_left = OpFSub %float %x %h
             %left_pos = OpCompositeConstruct %v4float %x_left %y %float_0 %float_1
                        OpStore %gl_out %left_pos
                        OpStore %outColor %in_color
                        OpEmitVertex

            %y_bottom = OpFSub %float %y %h
           %bottom_pos = OpCompositeConstruct %v4float %x %y_bottom %float_0 %float_1
                        OpStore %gl_out %bottom_pos
                        OpStore %outColor %in_color
                        OpEmitVertex

                        OpEndPrimitive
                        OpReturn
                        OpFunctionEnd
