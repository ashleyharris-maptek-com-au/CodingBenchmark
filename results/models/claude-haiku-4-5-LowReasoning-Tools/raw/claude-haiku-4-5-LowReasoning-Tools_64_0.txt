declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Drone {
  id: number;
  x: number;
  y: number;
}

const probability: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone;

const height = probability.length;
const width = height > 0 ? probability[0].length : 0;

// Track which cells have been visited for coverage scoring
const visited = new Set<string>();

function posToKey(x: number, y: number): string {
  return `${x},${y}`;
}

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height && probability[y][x] !== -1;
}

function getNeighbors(x: number, y: number): Array<[number, number]> {
  const neighbors: Array<[number, number]> = [];
  // Stay in place
  neighbors.push([x, y]);
  // 4-directional movement
  for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
    const nx = x + dx;
    const ny = y + dy;
    if (isValid(nx, ny)) {
      neighbors.push([nx, ny]);
    }
  }
  return neighbors;
}

// Initialize paths
const paths: Array<Array<[number, number]>> = drones.map(d => [[d.x, d.y]]);
const positions = drones.map(d => [d.x, d.y]);

// Mark starting positions as visited
for (const drone of drones) {
  visited.add(posToKey(drone.x, drone.y));
}

// Simulate steps - each drone greedily selects the best move
for (let step = 0; step < maxStepsPerDrone; step++) {
  for (let droneIdx = 0; droneIdx < drones.length; droneIdx++) {
    const [x, y] = positions[droneIdx];
    const neighbors = getNeighbors(x, y);

    // Score each neighbor: prioritize uncovered cells with high probability value
    let bestNeighbor: [number, number] = [x, y];
    let bestScore = -Infinity;

    for (const [nx, ny] of neighbors) {
      const key = posToKey(nx, ny);
      const value = probability[ny][nx];
      const isNew = !visited.has(key);

      // Heavy weight on uncovered cells to maximize unique coverage
      const score = value + (isNew ? 100000 : 0);

      if (score > bestScore) {
        bestScore = score;
        bestNeighbor = [nx, ny];
      }
    }

    // Move drone to best neighbor
    positions[droneIdx] = bestNeighbor;
    paths[droneIdx].push(bestNeighbor);
    visited.add(posToKey(bestNeighbor[0], bestNeighbor[1]));
  }
}

const result = {
  paths: paths
};
