using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

class VoxelMining
{
    struct Voxel
    {
        public int mass;
        public int value;
        public bool exists;
    }

    static void Main()
    {
        try
        {
            var reader = Console.In;
            var headerLine = reader.ReadLine();
            var parts = headerLine.Split(' ');

            int xCount = int.Parse(parts[0]);
            int yCount = int.Parse(parts[1]);
            int zCount = int.Parse(parts[2]);
            double xBlockSize = double.Parse(parts[3]);
            double yBlockSize = double.Parse(parts[4]);
            double zBlockSize = double.Parse(parts[5]);

            var voxels = new Voxel[xCount, yCount, zCount];

            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    for (int z = 0; z < zCount; z++)
                    {
                        int b = reader.Read();
                        if (b == -1) b = 0;

                        voxels[x, y, z] = new Voxel
                        {
                            mass = b % 16,
                            value = b / 16,
                            exists = b > 0
                        };
                    }
                }
            }

            var operations = new List<string>();
            var valuable = new List<(int x, int y, int z, int value, int mass)>();

            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    for (int z = 0; z < zCount; z++)
                    {
                        if (voxels[x, y, z].exists && voxels[x, y, z].value > 0)
                        {
                            valuable.Add((x, y, z, voxels[x, y, z].value, voxels[x, y, z].mass));
                        }
                    }
                }
            }

            valuable = valuable.OrderByDescending(v => v.value * v.mass).ToList();

            foreach (var (x, y, z, value, mass) in valuable)
            {
                if (!voxels[x, y, z].exists) continue;
                if (!IsExposed(voxels, x, y, z, xCount, yCount, zCount)) continue;

                var path = FindPath(x, y, z, 0, 0, voxels, xCount, yCount, zCount);
                if (path != null && path.Count > 0)
                {
                    string op = $"{path.Count} {x} {y} {z}";
                    foreach (var (sx, sy) in path)
                    {
                        op += $" {sx} {sy}";
                    }
                    op += $" 0 0";
                    operations.Add(op);
                    voxels[x, y, z].exists = false;
                }
            }

            Console.WriteLine(operations.Count);
            foreach (var op in operations)
            {
                Console.WriteLine(op);
            }
        }
        catch (Exception e)
        {
            Console.Error.WriteLine($"Error: {e}");
            Console.WriteLine(0);
        }
    }

    static bool IsExposed(Voxel[,,] voxels, int x, int y, int z, int xCount, int yCount, int zCount)
    {
        if (!voxels[x, y, z].exists) return false;

        if (z + 1 >= zCount || !voxels[x, y, z + 1].exists) return true;
        if (x > 0 && !voxels[x - 1, y, z].exists) return true;
        if (x < xCount - 1 && !voxels[x + 1, y, z].exists) return true;
        if (y > 0 && !voxels[x, y - 1, z].exists) return true;
        if (y < yCount - 1 && !voxels[x, y + 1, z].exists) return true;

        return false;
    }

    static List<(int, int)> FindPath(int sx, int sy, int sz, int dx, int dy, Voxel[,,] voxels,
        int xCount, int yCount, int zCount)
    {
        if (sx == dx && sy == dy) return new List<(int, int)>();

        var queue = new Queue<(int x, int y, List<(int, int)> path)>();
        var visited = new HashSet<(int, int)>();

        queue.Enqueue((sx, sy, new List<(int, int)>()));
        visited.Add((sx, sy));

        while (queue.Count > 0)
        {
            var (x, y, path) = queue.Dequeue();

            var directions = new[] { (1, 0), (-1, 0), (0, 1), (0, -1) };

            var orderedDirs = directions
                .OrderBy(d => Math.Abs(x + d.Item1 - dx) + Math.Abs(y + d.Item2 - dy))
                .ToList();

            foreach (var (dx2, dy2) in orderedDirs)
            {
                int nx = x + dx2;
                int ny = y + dy2;

                if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount)
                    continue;

                if (visited.Contains((nx, ny)))
                    continue;

                visited.Add((nx, ny));

                if (nx == dx && ny == dy)
                {
                    var finalPath = new List<(int, int)>(path);
                    finalPath.Add((nx, ny));
                    return finalPath;
                }

                var newPath = new List<(int, int)>(path);
                newPath.Add((nx, ny));
                queue.Enqueue((nx, ny, newPath));
            }
        }

        return null;
    }
}
