def solve_exact_cover():
    line = input().split()
    U, S = int(line[0]), int(line[1])

    sets = []
    for _ in range(S):
        line = input().split()
        if line:
            elements = set(map(int, line))
        else:
            elements = set()
        sets.append(elements)

    def backtrack(covered, used):
        # If all elements are covered, we found a solution
        if covered == set(range(U)):
            return sorted(used)

        # Find the first uncovered element
        for elem in range(U):
            if elem not in covered:
                # Try each set that contains this element
                for set_idx in range(S):
                    if set_idx not in used and elem in sets[set_idx]:
                        # Check if this set doesn't overlap with already covered elements
                        if not (sets[set_idx] & covered):
                            result = backtrack(
                                covered | sets[set_idx],
                                used | {set_idx}
                            )
                            if result is not None:
                                return result
                # No valid set found to cover elem
                return None

        return None

    solution = backtrack(set(), set())

    if solution is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(map(str, solution)))

solve_exact_cover()
