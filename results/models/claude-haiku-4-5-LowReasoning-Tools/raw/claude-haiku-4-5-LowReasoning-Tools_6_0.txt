using System;
using System.Collections.Generic;
using System.Linq;

class OrbitTSP
{
    const double MU = 398600.4418;
    const double EARTH_RADIUS = 6371.0;

    struct Orbit
    {
        public double X, Y, Z, VX, VY, VZ;
    }

    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        double mu = double.Parse(input[1]);

        var orbits = new Orbit[n + 1];
        for (int i = 0; i <= n; i++)
        {
            var line = Console.ReadLine().Split();
            orbits[i] = new Orbit
            {
                X = double.Parse(line[0]),
                Y = double.Parse(line[1]),
                Z = double.Parse(line[2]),
                VX = double.Parse(line[3]),
                VY = double.Parse(line[4]),
                VZ = double.Parse(line[5])
            };
        }

        // Build cost matrix
        var costs = new double[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j) costs[i, j] = 0;
                else costs[i, j] = EstimateDeltaV(orbits[0], orbits[i + 1], orbits[j + 1], mu);
            }
        }

        // Solve TSP based on problem size
        int[] solution;
        if (n <= 5)
        {
            solution = BruteForceTSP(costs, n);
        }
        else if (n <= 12)
        {
            solution = NearestNeighborWithOpt(costs, n);
        }
        else
        {
            solution = GreedyHeuristic(costs, n);
        }

        Console.WriteLine(string.Join(" ", solution.Select(x => x.ToString())));
    }

    static double EstimateDeltaV(Orbit start, Orbit from, Orbit to, double mu)
    {
        double dist_from = Math.Sqrt((from.X - start.X) * (from.X - start.X) +
                                     (from.Y - start.Y) * (from.Y - start.Y) +
                                     (from.Z - start.Z) * (from.Z - start.Z));
        double dist_to = Math.Sqrt((to.X - start.X) * (to.X - start.X) +
                                   (to.Y - start.Y) * (to.Y - start.Y) +
                                   (to.Z - start.Z) * (to.Z - start.Z));
        double dist_between = Math.Sqrt((to.X - from.X) * (to.X - from.X) +
                                        (to.Y - from.Y) * (to.Y - from.Y) +
                                        (to.Z - from.Z) * (to.Z - from.Z));

        if (dist_from < EARTH_RADIUS || dist_to < EARTH_RADIUS) return double.MaxValue / 2;

        // Simple delta-v: combining distance between stations with orbital mechanics
        double vfrom = Math.Sqrt(from.VX * from.VX + from.VY * from.VY + from.VZ * from.VZ);
        double vto = Math.Sqrt(to.VX * to.VX + to.VY * to.VY + to.VZ * to.VZ);

        double dv = Math.Abs(vto - vfrom) + dist_between * 0.001;
        return dv;
    }

    static int[] BruteForceTSP(double[,] costs, int n)
    {
        var perm = Enumerable.Range(0, n).ToArray();
        var best = perm.ToArray();
        double bestCost = CalculateCost(best, costs);

        while (NextPermutation(perm))
        {
            double cost = CalculateCost(perm, costs);
            if (cost < bestCost)
            {
                bestCost = cost;
                best = perm.ToArray();
            }
        }

        return best;
    }

    static bool NextPermutation(int[] arr)
    {
        int i = arr.Length - 2;
        while (i >= 0 && arr[i] >= arr[i + 1]) i--;
        if (i < 0) return false;

        int j = arr.Length - 1;
        while (arr[j] <= arr[i]) j--;
        (arr[i], arr[j]) = (arr[j], arr[i]);
        Array.Reverse(arr, i + 1, arr.Length - i - 1);
        return true;
    }

    static int[] NearestNeighborWithOpt(double[,] costs, int n)
    {
        var solution = NearestNeighbor(costs, n);
        TwoOpt(solution, costs);
        return solution;
    }

    static int[] NearestNeighbor(double[,] costs, int n)
    {
        var visited = new bool[n];
        var tour = new int[n];
        tour[0] = 0;
        visited[0] = true;

        for (int i = 1; i < n; i++)
        {
            int nearest = -1;
            double minCost = double.MaxValue;
            for (int j = 0; j < n; j++)
            {
                if (!visited[j] && costs[tour[i - 1], j] < minCost)
                {
                    minCost = costs[tour[i - 1], j];
                    nearest = j;
                }
            }
            tour[i] = nearest;
            visited[nearest] = true;
        }

        return tour;
    }

    static void TwoOpt(int[] tour, double[,] costs)
    {
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < tour.Length - 1; i++)
            {
                for (int j = i + 2; j < tour.Length; j++)
                {
                    int a = tour[i], b = tour[i + 1];
                    int c = tour[j], d = tour[(j + 1) % tour.Length];
                    double oldCost = costs[a, b] + costs[c, d];
                    double newCost = costs[a, c] + costs[b, d];
                    if (newCost < oldCost)
                    {
                        Array.Reverse(tour, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
    }

    static int[] GreedyHeuristic(double[,] costs, int n)
    {
        // Use nearest neighbor with limited 2-opt for large problems
        var solution = NearestNeighbor(costs, n);

        // Light 2-opt: only check nearby swaps
        for (int iter = 0; iter < 3; iter++)
        {
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = Math.Min(i + 10, n - 1); j > i + 1; j--)
                {
                    int a = solution[i], b = solution[i + 1];
                    int c = solution[j], d = solution[(j + 1) % n];
                    double oldCost = costs[a, b] + costs[c, d];
                    double newCost = costs[a, c] + costs[b, d];
                    if (newCost < oldCost)
                    {
                        Array.Reverse(solution, i + 1, j - i);
                        break;
                    }
                }
            }
        }

        return solution;
    }

    static double CalculateCost(int[] tour, double[,] costs)
    {
        double total = 0;
        for (int i = 0; i < tour.Length; i++)
        {
            total += costs[tour[i], tour[(i + 1) % tour.Length]];
        }
        return total;
    }
}
