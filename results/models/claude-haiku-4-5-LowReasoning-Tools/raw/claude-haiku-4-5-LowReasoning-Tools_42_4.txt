OpCapability Geometry
OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations for user-defined inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Decorations for user-defined outputs
OpDecorate %outColor Location 0

; =========== Type Definitions ===========
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; =========== Constants ===========
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0p04 = OpConstant %float 0.04
%float_0p005 = OpConstant %float 0.005

; =========== Input Array Types ===========
; gl_PerVertex input (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%array_2_pv_in = OpTypeArray %gl_PerVertex_in %int_2
%ptr_Input_array_pv_in = OpTypePointer Input %array_2_pv_in
%ptr_Input_v4float = OpTypePointer Input %v4float

; User-defined input arrays (vec3, size 2)
%array_2_v3 = OpTypeArray %v3float %int_2
%ptr_Input_array_v3 = OpTypePointer Input %array_2_v3
%ptr_Input_v3float = OpTypePointer Input %v3float

; =========== Output Types ===========
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

; =========== Variable Declarations ===========
%gl_in = OpVariable %ptr_Input_array_pv_in Input
%worldPos = OpVariable %ptr_Input_array_v3 Input
%normal = OpVariable %ptr_Input_array_v3 Input
%color = OpVariable %ptr_Input_array_v3 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%outColor = OpVariable %ptr_Output_v3float Output

; =========== Function Definition ===========
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input positions (gl_in[0].gl_Position and gl_in[1].gl_Position)
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load input colors
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Extract XYZ components from positions
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

; Compute line direction vector
%dir = OpFSub %v3float %p1_xyz %p0_xyz

; Compute perpendicular vector (rotate direction in XY plane by 90 degrees)
; perp = vec3(-dir.y, dir.x, 0)
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0

; Normalize perpendicular vector using Sqrt extended instruction
%perp_len_sq = OpDot %float %perp_unnorm %perp_unnorm
%ext_glsl = OpExtInstImport "GLSL.std.450"
%perp_len = OpExtInst %float %ext_glsl Sqrt %perp_len_sq
%perp_len_recip = OpFDiv %float %float_1 %perp_len
%perp = OpVectorTimesScalar %v3float %perp_unnorm %perp_len_recip

; Compute thickness vectors
%thick_start = OpVectorTimesScalar %v3float %perp %float_0p04
%thick_end = OpVectorTimesScalar %v3float %perp %float_0p005

; Compute four vertex positions for triangle strip
; Vertex 0: p0 + thick_start (top-left at start)
%v0 = OpFAdd %v3float %p0_xyz %thick_start
%v0_4 = OpCompositeConstruct %v4float %v0 %float_1

; Vertex 1: p0 - thick_start (bottom-left at start)
%v1 = OpFSub %v3float %p0_xyz %thick_start
%v1_4 = OpCompositeConstruct %v4float %v1 %float_1

; Vertex 2: p1 + thick_end (top-right at end)
%v2 = OpFAdd %v3float %p1_xyz %thick_end
%v2_4 = OpCompositeConstruct %v4float %v2 %float_1

; Vertex 3: p1 - thick_end (bottom-right at end)
%v3 = OpFSub %v3float %p1_xyz %thick_end
%v3_4 = OpCompositeConstruct %v4float %v3 %float_1

; Get pointer to output position
%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Emit Vertex 0: (p0 + perp) with color c0
OpStore %gl_out_pos_ptr %v0_4
OpStore %outColor %c0
OpEmitVertex

; Emit Vertex 1: (p0 - perp) with color c0
OpStore %gl_out_pos_ptr %v1_4
OpStore %outColor %c0
OpEmitVertex

; Emit Vertex 2: (p1 + perp) with color c1
OpStore %gl_out_pos_ptr %v2_4
OpStore %outColor %c1
OpEmitVertex

; Emit Vertex 3: (p1 - perp) with color c1
OpStore %gl_out_pos_ptr %v3_4
OpStore %outColor %c1
OpEmitVertex

; End the primitive (complete the triangle strip)
OpEndPrimitive

; Return from function
OpReturn
OpFunctionEnd
