#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <map>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

typedef vector<int> Position;

int D, max_turns;
vector<int> bounds;
set<Position> obstacles;

int manhattan_distance(const Position& a, const Position& b) {
    int dist = 0;
    for (int i = 0; i < D; i++) {
        dist += abs(a[i] - b[i]);
    }
    return dist;
}

bool is_valid(const Position& pos, const set<Position>& snake_body_except_tail) {
    for (int i = 0; i < D; i++) {
        if (pos[i] < 0 || pos[i] >= bounds[i]) {
            return false;
        }
    }
    if (obstacles.count(pos)) {
        return false;
    }
    if (snake_body_except_tail.count(pos)) {
        return false;
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int max_turns_input;
    cin >> D >> max_turns_input;
    max_turns = max_turns_input;

    bounds.resize(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    int snake_length;
    Position head(D);
    cin >> snake_length;
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }

    // Initialize snake body
    vector<Position> snake;
    snake.push_back(head);
    Position curr = head;
    for (int i = 1; i < snake_length; i++) {
        Position prev = curr;
        if (curr[0] > 0) {
            curr[0]--;
        } else if (curr[1] > 0) {
            curr[1]--;
        } else {
            // Default backward extension
            curr[0]--;
        }
        snake.push_back(curr);
    }

    int F;
    cin >> F;
    vector<Position> food(F);
    set<Position> food_set;
    for (int i = 0; i < F; i++) {
        food[i].resize(D);
        for (int j = 0; j < D; j++) {
            cin >> food[i][j];
        }
        food_set.insert(food[i]);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        Position obs(D);
        for (int j = 0; j < D; j++) {
            cin >> obs[j];
        }
        obstacles.insert(obs);
    }

    head = snake[0];

    for (int turn = 0; turn < max_turns && !food_set.empty(); turn++) {
        // Find nearest food
        Position target = food[0];
        int min_dist = INT_MAX;

        for (const auto& f : food_set) {
            int dist = manhattan_distance(head, f);
            if (dist < min_dist) {
                min_dist = dist;
                target = f;
            }
        }

        // Find best dimension to move along
        int best_dim = 0;
        int max_diff = 0;

        for (int d = 0; d < D; d++) {
            int diff = abs(head[d] - target[d]);
            if (diff > max_diff) {
                max_diff = diff;
                best_dim = d;
            }
        }

        int direction = (head[best_dim] < target[best_dim]) ? 1 : -1;

        // Try the best dimension first
        Position new_head = head;
        new_head[best_dim] += direction;

        set<Position> snake_body_no_tail(snake.begin(), snake.end() - 1);

        if (!is_valid(new_head, snake_body_no_tail)) {
            // Try all other moves
            bool found = false;
            for (int d = 0; d < D && !found; d++) {
                for (int dir : {1, -1}) {
                    Position test_head = head;
                    test_head[d] += dir;

                    if (is_valid(test_head, snake_body_no_tail)) {
                        best_dim = d;
                        direction = dir;
                        new_head = test_head;
                        found = true;
                        break;
                    }
                }
            }

            if (!found) {
                break;
            }
        }

        cout << best_dim << " " << direction << "\n";
        cout.flush();

        head = new_head;

        if (food_set.count(head)) {
            food_set.erase(head);
        } else {
            snake.pop_back();
        }

        snake.insert(snake.begin(), head);
    }

    return 0;
}
