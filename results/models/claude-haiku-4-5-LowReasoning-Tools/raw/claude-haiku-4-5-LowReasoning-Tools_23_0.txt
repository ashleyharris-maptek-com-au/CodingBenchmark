use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

fn normalize_angle(mut a: f64) -> f64 {
    while a > std::f64::consts::PI {
        a -= 2.0 * std::f64::consts::PI;
    }
    while a < -std::f64::consts::PI {
        a += 2.0 * std::f64::consts::PI;
    }
    a
}

fn main() {
    let stdin = io::stdin();
    let mut stdout = io::stdout();
    let mut lines = stdin.lock().lines();

    // Parse initial configuration
    let config: Vec<f64> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let target_info: Vec<f64> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let gravity = config[2];
    let max_thrust = config[3];
    let fuel_init = config[4];
    let start_x = target_info[0];
    let start_y = target_info[1];
    let target_x = target_info[2];
    let target_y = target_info[3];

    // Skip map data
    for _ in 0..(config[1] as usize) {
        let _ = lines.next();
    }

    let state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel_remaining: fuel_init,
    }));

    let state_clone = Arc::clone(&state);

    // Input reading thread - handles stdin blocking separately
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let parts: Vec<f64> = line.split_whitespace()
                    .map(|s| s.parse().unwrap_or(0.0))
                    .collect();
                if parts.len() >= 7 {
                    let mut s = state_clone.lock().unwrap();
                    s.x = parts[0];
                    s.y = parts[1];
                    s.vx = parts[2];
                    s.vy = parts[3];
                    s.angle = parts[4];
                    s.angular_vel = parts[5];
                    s.fuel_remaining = parts[6];
                }
            }
        }
    });

    // Control loop at 20Hz
    loop {
        let current = state.lock().unwrap();

        // Calculate vector to target
        let dx = target_x - current.x;
        let dy = target_y - current.y;
        let dist = (dx * dx + dy * dy).sqrt();

        // Desired angle: point toward target with consideration for proximity
        let desired_angle = if dist < 5.0 {
            // Very close to target, descend vertically
            std::f64::consts::PI / 2.0
        } else {
            // Calculate angle to target
            let angle_to_target = dy.atan2(dx);
            angle_to_target + std::f64::consts::PI / 2.0
        };

        // Calculate turn rate to reach desired angle
        let angle_diff = normalize_angle(desired_angle - current.angle);
        let turn = (angle_diff / std::f64::consts::PI).max(-1.0).min(1.0);

        // Calculate thrust based on descent rate and position
        let thrust = if current.fuel_remaining <= 0.0 {
            0.0
        } else if dist < 3.0 {
            // Very close to target - control descent speed
            if current.vy > 2.0 {
                1.0
            } else if current.vy > 0.5 {
                0.6
            } else if current.vy > -0.5 {
                0.3
            } else {
                0.1
            }
        } else if dist < 15.0 {
            // Approaching target
            if current.vy > 2.5 {
                1.0
            } else if current.vy > 1.0 {
                0.7
            } else if current.vy > -0.5 {
                0.4
            } else {
                0.1
            }
        } else {
            // Far from target - manage altitude and descent
            if current.vy > 3.0 {
                1.0
            } else if current.vy > 1.5 {
                0.6
            } else if current.vy > 0.0 {
                0.4
            } else {
                0.2
            }
        };

        drop(current);

        // Output control command
        let _ = writeln!(stdout, "{} {}", thrust, turn);
        let _ = stdout.flush();

        // Sleep to maintain 20Hz output rate (50ms per iteration)
        thread::sleep(Duration::from_millis(50));
    }
}
