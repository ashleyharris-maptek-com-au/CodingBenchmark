#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <cstring>
#include <random>

using namespace std;

struct City {
    float x, y;
    int idx;
};

// Morton code (Z-order curve) for space-filling curve approach
uint64_t morton_encode(float x, float y, float min_x, float min_y, float scale) {
    uint32_t ix = (uint32_t)((x - min_x) * scale);
    uint32_t iy = (uint32_t)((y - min_y) * scale);

    uint64_t code = 0;
    for (int i = 0; i < 32; ++i) {
        code |= ((uint64_t)(ix & (1U << i)) << i) | ((uint64_t)(iy & (1U << i)) << (i + 1));
    }
    return code;
}

// Nearest neighbor heuristic starting from city 0
vector<int> nearest_neighbor(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    int current = 0;
    for (int i = 1; i < n; ++i) {
        float min_dist = numeric_limits<float>::max();
        int nearest = -1;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                float dx = cities[current].x - cities[j].x;
                float dy = cities[current].y - cities[j].y;
                float dist = dx*dx + dy*dy;
                if (dist < min_dist) {
                    min_dist = dist;
                    nearest = j;
                }
            }
        }

        current = nearest;
        visited[current] = true;
        tour.push_back(current);
    }

    return tour;
}

// 2-opt improvement with limited iterations
void two_opt_improve(vector<int>& tour, const vector<City>& cities, int max_iterations) {
    int n = tour.size();
    bool improved = true;
    int iterations = 0;

    while (improved && iterations < max_iterations) {
        improved = false;
        iterations++;

        for (int i = 0; i < n - 2; ++i) {
            for (int j = i + 2; j < n; ++j) {
                int a = tour[i];
                int b = tour[i + 1];
                int c = tour[j];
                int d = tour[(j + 1) % n];

                float dist_before = (cities[a].x - cities[b].x) * (cities[a].x - cities[b].x) +
                                   (cities[a].y - cities[b].y) * (cities[a].y - cities[b].y) +
                                   (cities[c].x - cities[d].x) * (cities[c].x - cities[d].x) +
                                   (cities[c].y - cities[d].y) * (cities[c].y - cities[d].y);

                float dist_after = (cities[a].x - cities[c].x) * (cities[a].x - cities[c].x) +
                                  (cities[a].y - cities[c].y) * (cities[a].y - cities[c].y) +
                                  (cities[b].x - cities[d].x) * (cities[b].x - cities[d].x) +
                                  (cities[b].y - cities[d].y) * (cities[b].y - cities[d].y);

                if (dist_after < dist_before) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Space-filling curve approach (Z-order)
vector<int> space_filling_curve_solve(const vector<City>& cities) {
    int n = cities.size();

    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;

    for (const auto& city : cities) {
        min_x = min(min_x, city.x);
        max_x = max(max_x, city.x);
        min_y = min(min_y, city.y);
        max_y = max(max_y, city.y);
    }

    float range = max(max_x - min_x, max_y - min_y);
    if (range < 1e-6) range = 1.0f;
    float scale = 4294967295.0f / range;

    vector<pair<uint64_t, int>> codes;
    codes.reserve(n);

    for (int i = 0; i < n; ++i) {
        uint64_t code = morton_encode(cities[i].x, cities[i].y, min_x, min_y, scale);
        codes.push_back({code, i});
    }

    sort(codes.begin(), codes.end());

    vector<int> tour;
    tour.reserve(n);
    for (const auto& p : codes) {
        tour.push_back(p.second);
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].idx = i;
    }

    vector<int> tour;

    if (n <= 40) {
        // Small: nearest neighbor + aggressive 2-opt
        tour = nearest_neighbor(cities);
        two_opt_improve(tour, cities, n * 10);
    } else if (n <= 1000) {
        // Medium: nearest neighbor + moderate 2-opt
        tour = nearest_neighbor(cities);
        two_opt_improve(tour, cities, 100);
    } else if (n <= 100000) {
        // Large: nearest neighbor + light 2-opt
        tour = nearest_neighbor(cities);
        two_opt_improve(tour, cities, 5);
    } else {
        // Very large: space-filling curve
        tour = space_filling_curve_solve(cities);
    }

    // Output tour
    cout << tour[0];
    for (int i = 1; i < n; ++i) {
        cout << " " << tour[i];
    }
    cout << "\n";

    return 0;
}
