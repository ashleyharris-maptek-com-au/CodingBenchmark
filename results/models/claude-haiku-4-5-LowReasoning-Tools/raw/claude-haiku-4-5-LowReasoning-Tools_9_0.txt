using System;
using System.Collections.Generic;
using System.Linq;

class CuttingStock
{
    static void Main()
    {
        // Read input
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stock_length = int.Parse(line1[1]);

        var cuts = Console.ReadLine().Split().Select(int.Parse).ToArray();

        // First-fit decreasing algorithm
        var stocks = new List<List<int>>();
        var stockUsage = new List<int>();

        // Create array of (cut_length, original_index) and sort descending
        var indexed_cuts = cuts.Select((len, idx) => (len, idx)).ToArray();
        Array.Sort(indexed_cuts, (a, b) => b.len.CompareTo(a.len));

        // Pack each cut
        foreach (var (cut_len, idx) in indexed_cuts)
        {
            // Try to fit in existing stock
            bool placed = false;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockUsage[i] + cut_len <= stock_length)
                {
                    stocks[i].Add(idx);
                    stockUsage[i] += cut_len;
                    placed = true;
                    break;
                }
            }

            // If not placed, create new stock
            if (!placed)
            {
                stocks.Add(new List<int> { idx });
                stockUsage.Add(cut_len);
            }
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
