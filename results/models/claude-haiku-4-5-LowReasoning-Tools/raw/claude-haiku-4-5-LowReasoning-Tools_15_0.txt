use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Vec3 { x, y, z }
    }

    fn normalize(&self) -> Vec3 {
        let len = (self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if len > 1e-10 {
            Vec3::new(self.x / len, self.y / len, self.z / len)
        } else {
            *self
        }
    }

    fn dot(&self, other: Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(&self, other: Vec3) -> Vec3 {
        Vec3::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn identity() -> Self {
        Quaternion { w: 1.0, x: 0.0, y: 0.0, z: 0.0 }
    }

    fn from_axis_angle(axis: Vec3, angle: f64) -> Self {
        let axis = axis.normalize();
        let half_angle = angle / 2.0;
        let sin_half = half_angle.sin();
        Quaternion {
            w: half_angle.cos(),
            x: axis.x * sin_half,
            y: axis.y * sin_half,
            z: axis.z * sin_half,
        }
    }
}

fn read_input() -> (Vec<(f64, f64)>, Vec3) {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut polygon = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        polygon.push((parts[0], parts[1]));
    }

    let line = lines.next().unwrap().unwrap();
    let parts: Vec<f64> = line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sun_dir = Vec3::new(parts[0], parts[1], parts[2]).normalize();

    (polygon, sun_dir)
}

fn point_in_polygon(x: f64, y: f64, polygon: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = polygon.len() - 1;

    for i in 0..polygon.len() {
        let (xi, yi) = polygon[i];
        let (xj, yj) = polygon[j];

        if ((yi > y) != (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }

    inside
}

fn polygon_bounds(polygon: &[(f64, f64)]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for &(x, y) in polygon {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    (min_x, max_x, min_y, max_y)
}

fn tetrahedron_shadow_covers_point(
    tet_center: Vec3,
    tet_scale: f64,
    sun_dir: Vec3,
    point: (f64, f64),
) -> bool {
    let ref_vertices = [
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];

    if sun_dir.z.abs() < 1e-10 {
        return false;
    }

    let mut shadow_points = Vec::new();
    for vertex in &ref_vertices {
        let scaled = Vec3::new(
            vertex.x * tet_scale,
            vertex.y * tet_scale,
            vertex.z * tet_scale,
        );
        let world_pos = Vec3::new(
            tet_center.x + scaled.x,
            tet_center.y + scaled.y,
            tet_center.z + scaled.z,
        );

        let t = -world_pos.z / sun_dir.z;
        let shadow_x = world_pos.x + t * sun_dir.x;
        let shadow_y = world_pos.y + t * sun_dir.y;
        shadow_points.push((shadow_x, shadow_y));
    }

    point_in_polygon(point.0, point.1, &shadow_points)
}

fn solve_coverage(polygon: &[(f64, f64)], sun_dir: Vec3) -> Vec<(Vec3, Quaternion, f64)> {
    let (min_x, max_x, min_y, max_y) = polygon_bounds(polygon);

    let width = max_x - min_x;
    let height = max_y - min_y;
    let area = width * height;

    let mut tetrahedrons = Vec::new();

    let grid_size = ((area / (std::f64::consts::PI * 5.0)).sqrt()).max(0.5);

    let mut x = min_x;
    while x <= max_x {
        let mut y = min_y;
        while y <= max_y {
            let center = (x + grid_size / 2.0, y + grid_size / 2.0);

            if point_in_polygon(center.0, center.1, polygon) {
                let mut covered = false;
                for (tet_center, _, tet_scale) in &tetrahedrons {
                    if tetrahedron_shadow_covers_point(*tet_center, *tet_scale, sun_dir, center) {
                        covered = true;
                        break;
                    }
                }

                if !covered {
                    let scale = (grid_size / 3.0).min(10.0).max(0.2);
                    let tet_center = Vec3::new(center.0, center.1, 2.0 + scale);
                    let rotation = Quaternion::identity();
                    tetrahedrons.push((tet_center, rotation, scale));
                }
            }

            y += grid_size;
        }
        x += grid_size;
    }

    tetrahedrons
}

fn main() {
    let (polygon, sun_dir) = read_input();
    let tetrahedrons = solve_coverage(&polygon, sun_dir);

    println!("{}", tetrahedrons.len());
    for (center, rotation, scale) in tetrahedrons {
        println!("{} {} {} {} {} {} {} {}",
            center.x, center.y, center.z,
            rotation.w, rotation.x, rotation.y, rotation.z,
            scale);
    }
}
