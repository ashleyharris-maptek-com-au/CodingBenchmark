  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4

        %ubo = OpTypeStruct %mat4float %mat4float %mat4float %v3float %v3float %v3float
 %ptr_ubo = OpTypePointer Uniform %ubo
%ptr_func_v4 = OpTypePointer Function %v4float

      %uint_0 = OpConstant %uint 0

        %gl_in = OpVariable %ptr_in_arr Input

       %main = OpFunction %void None %void_fn
      %entry = OpLabel
   %p1_coord = OpVariable %ptr_func_v4 Function
   %p2_coord = OpVariable %ptr_func_v4 Function
   %p0_coord = OpVariable %ptr_func_v4 Function
   %p3_coord = OpVariable %ptr_func_v4 Function
   %tangent1 = OpVariable %ptr_func_v3 Function
   %tangent2 = OpVariable %ptr_func_v3 Function
      %delta = OpVariable %ptr_func_v3 Function
     %normal_vec = OpVariable %ptr_func_v3 Function
    %thickness = OpVariable %ptr_func_f Function
      %color1 = OpVariable %ptr_func_v3 Function
      %color2 = OpVariable %ptr_func_v3 Function
      %mix_t = OpVariable %ptr_func_f Function
      %part = OpVariable %ptr_func_i Function
      %quad_v = OpVariable %ptr_func_i Function
      %start_frac = OpVariable %ptr_func_f Function
      %end_frac = OpVariable %ptr_func_f Function
      %current_interp = OpVariable %ptr_func_f Function
        %pos_3d = OpVariable %ptr_func_v3 Function
       %pos_4d = OpVariable %ptr_func_v4 Function
    %lerp_col = OpVariable %ptr_func_v3 Function

    %p1_ac = OpAccessChain %ptr_in_arr Input %int_1 %int_0
      %p1 = OpLoad %v4float %p1_ac
               OpStore %p1_coord %p1

    %p2_ac = OpAccessChain %ptr_in_arr Input %int_2 %int_0
      %p2 = OpLoad %v4float %p2_ac
               OpStore %p2_coord %p2

    %p0_ac = OpAccessChain %ptr_in_arr Input %int_0 %int_0
      %p0 = OpLoad %v4float %p0_ac
               OpStore %p0_coord %p0

    %p3_ac = OpAccessChain %ptr_in_arr Input %int_3 %int_0
      %p3 = OpLoad %v4float %p3_ac
               OpStore %p3_coord %p3

   %col1_ptr = OpAccessChain %ptr_in_v3_arr Input %int_1
      %col1 = OpLoad %v3float %col1_ptr
               OpStore %color1 %col1

   %col2_ptr = OpAccessChain %ptr_in_v3_arr Input %int_2
      %col2 = OpLoad %v3float %col2_ptr
               OpStore %color2 %col2

               OpStore %thickness %float_0p02

   %p1_3d = OpVectorShuffle %v3float %p1 %p1 0 1 2
   %p2_3d = OpVectorShuffle %v3float %p2 %p2 0 1 2
   %seg_delta = OpFSub %v3float %p2_3d %p1_3d
               OpStore %delta %seg_delta

   %seg_len2 = OpDot %float %seg_delta %seg_delta
   %seg_len = OpSqrt %float %seg_len2
   %perp_z = OpCompositeConstruct %v3float %float_0 %float_0 %float_1
   %cross = OpExtInst %v3float %glsl_ext_inst Cross %seg_delta %perp_z
   %cross_len2 = OpDot %float %cross %cross
   %cross_len = OpSqrt %float %cross_len2
   %cross_safe = OpFAdd %float %cross_len %float_0p5
   %cross_norm = OpFDiv %float %float_1 %cross_safe
   %cross_normalized = OpVectorTimesScalar %v3float %cross %cross_norm
               OpStore %normal_vec %cross_normalized

               OpStore %part %int_0

               OpBranch %loop_part
       %loop_part = OpLabel
    %part_val = OpLoad %int %part
    %part_cond = OpSLessThan %bool %part_val %int_2
               OpLoopMerge %loop_end %loop_continue None
               OpBranchConditional %part_cond %loop_body %loop_end

      %loop_body = OpLabel
    %part_v = OpLoad %int %part
    %is_first = OpIEqual %bool %part_v %int_0
    %part_offset = OpSelect %int %is_first %int_0 %int_2

    %start_t = OpConvertSToF %float %part_offset
    %start_div = OpFDiv %float %start_t %float_3
               OpStore %start_frac %start_div

    %part_of1 = OpIAdd %int %part_offset %int_1
    %end_t = OpConvertSToF %float %part_of1
    %end_div = OpFDiv %float %end_t %float_3
               OpStore %end_frac %end_div

               OpStore %quad_v %int_0
               OpBranch %quad_loop

        %quad_loop = OpLabel
     %quad_idx = OpLoad %int %quad_v
     %quad_cond = OpSLessThan %bool %quad_idx %int_4
                OpLoopMerge %quad_end %quad_cont None
                OpBranchConditional %quad_cond %quad_body %quad_end

       %quad_body = OpLabel
     %quad_i = OpLoad %int %quad_v
     %quad_f = OpConvertSToF %float %quad_i
     %quad_mix = OpFDiv %float %quad_f %float_2
     %start_f = OpLoad %float %start_frac
     %end_f = OpLoad %float %end_frac
     %seg_range = OpFSub %float %end_f %start_f
     %quad_pos = OpFMul %float %quad_mix %seg_range
     %quad_t = OpFAdd %float %start_f %quad_pos
                OpStore %current_interp %quad_t

     %p1_v4 = OpLoad %v4float %p1_coord
     %p2_v4 = OpLoad %v4float %p2_coord
     %p1_3d_v = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
     %p2_3d_v = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
     %del = OpFSub %v3float %p2_3d_v %p1_3d_v
     %seg_len2_v = OpDot %float %del %del
     %seg_len_v = OpSqrt %float %seg_len2_v

     %cur_t = OpLoad %float %current_interp
     %pos_along = OpVectorTimesScalar %v3float %del %cur_t
     %pos_base = OpFAdd %v3float %p1_3d_v %pos_along
                OpStore %pos_3d %pos_base

     %norm_v = OpLoad %v3float %normal_vec
     %thick = OpLoad %float %thickness
     %odd_quad = OpLoad %int %quad_v
     %is_odd_i = OpUMod %int %odd_quad %int_2
     %is_odd_bool = OpINotEqual %bool %is_odd_i %int_0

     %sign_mult = OpSelect %float %is_odd_bool %float_neg1 %float_1
     %offset_mag = OpFMul %float %thick %sign_mult
     %offset_vec = OpVectorTimesScalar %v3float %norm_v %offset_mag
     %final_pos = OpFAdd %v3float %pos_base %offset_vec

     %col1_v = OpLoad %v3float %color1
     %col2_v = OpLoad %v3float %color2
     %col_delta = OpFSub %v3float %col2_v %col1_v
     %col_interp = OpVectorTimesScalar %v3float %col_delta %cur_t
     %final_col = OpFAdd %v3float %col1_v %col_interp

     %z_comp = OpCompositeExtract %float %p1_v4 3
     %final_4d = OpCompositeConstruct %v4float %final_pos %z_comp

     %out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out_pos_ptr %final_4d

                OpStore %outColor %final_col

                OpEmitVertex

       %quad_cont = OpLabel
     %next_quad = OpLoad %int %quad_v
     %inc_quad = OpIAdd %int %next_quad %int_1
                OpStore %quad_v %inc_quad
                OpBranch %quad_loop

        %quad_end = OpLabel
                OpEndPrimitive

       %loop_continue = OpLabel
     %next_part = OpLoad %int %part
     %inc_part = OpIAdd %int %next_part %int_1
                OpStore %part %inc_part
                OpBranch %loop_part

        %loop_end = OpLabel
                OpReturn
                OpFunctionEnd
