#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <cmath>

struct Rect {
    int w, h, idx;
};

struct Placed {
    int board, x, y, w, h;
    bool rotated;
};

struct Block {
    int x, y, w, h;
};

bool intersects(int x1, int y1, int w1, int h1, int x2, int y2, int w2, int h2) {
    return !(x1 + w1 <= x2 || x2 + w2 <= x1 || y1 + h1 <= y2 || y2 + h2 <= y1);
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, board_width, board_height;
    std::cin >> n >> board_width >> board_height;

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    // Sort by area descending (First Fit Decreasing)
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        long long area_a = (long long)a.w * a.h;
        long long area_b = (long long)b.w * b.h;
        return area_a > area_b;
    });

    std::vector<Placed> placed(n);
    std::vector<std::vector<Block>> boards;

    for (const auto& rect : rects) {
        bool found = false;
        int best_board = -1, best_x = -1, best_y = -1, best_w = rect.w, best_h = rect.h;
        long long best_score = LLONG_MAX;

        // Try existing boards
        for (int b = 0; b < boards.size(); b++) {
            auto& blocks = boards[b];

            // Generate candidate positions from existing blocks
            std::set<std::pair<int, int>> candidates;
            candidates.insert({0, 0});
            for (const auto& block : blocks) {
                if (block.x + block.w < board_width) {
                    candidates.insert({block.x + block.w, block.y});
                }
                if (block.y + block.h < board_height) {
                    candidates.insert({block.x, block.y + block.h});
                }
            }

            for (auto [x, y] : candidates) {
                // Try without rotation
                if (x + rect.w <= board_width && y + rect.h <= board_height) {
                    bool can_place = true;
                    for (const auto& block : blocks) {
                        if (intersects(x, y, rect.w, rect.h, block.x, block.y, block.w, block.h)) {
                            can_place = false;
                            break;
                        }
                    }
                    if (can_place) {
                        long long score = (long long)x * board_height + y;
                        if (score < best_score) {
                            best_board = b;
                            best_x = x;
                            best_y = y;
                            best_w = rect.w;
                            best_h = rect.h;
                            best_score = score;
                            found = true;
                        }
                    }
                }

                // Try with rotation
                if (rect.w != rect.h && x + rect.h <= board_width && y + rect.w <= board_height) {
                    bool can_place = true;
                    for (const auto& block : blocks) {
                        if (intersects(x, y, rect.h, rect.w, block.x, block.y, block.w, block.h)) {
                            can_place = false;
                            break;
                        }
                    }
                    if (can_place) {
                        long long score = (long long)x * board_height + y;
                        if (score < best_score) {
                            best_board = b;
                            best_x = x;
                            best_y = y;
                            best_w = rect.h;
                            best_h = rect.w;
                            best_score = score;
                            found = true;
                        }
                    }
                }
            }
        }

        if (found) {
            auto& blocks = boards[best_board];
            blocks.push_back({best_x, best_y, best_w, best_h});
            placed[rect.idx] = {best_board, best_x, best_y, best_w, best_h, best_w != rect.w};
        } else {
            // Create new board
            boards.push_back({});

            // Try without rotation first
            if (rect.w <= board_width && rect.h <= board_height) {
                boards.back().push_back({0, 0, rect.w, rect.h});
                placed[rect.idx] = {(int)boards.size() - 1, 0, 0, rect.w, rect.h, false};
            } else if (rect.h <= board_width && rect.w <= board_height) {
                boards.back().push_back({0, 0, rect.h, rect.w});
                placed[rect.idx] = {(int)boards.size() - 1, 0, 0, rect.h, rect.w, true};
            }
        }
    }

    std::cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        std::cout << placed[i].board << " " << placed[i].x << " " << placed[i].y << " " << (placed[i].rotated ? 1 : 0) << "\n";
    }

    return 0;
}
