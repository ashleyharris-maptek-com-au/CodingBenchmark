using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int n, m;
    static List<int>[] adj;
    static List<int> bestFVS;
    static HashSet<int> currentFVS;

    static void Main()
    {
        string[] input = Console.ReadLine().Split();
        n = int.Parse(input[0]);
        m = int.Parse(input[1]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string[] edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
        }

        bestFVS = new List<int>();
        currentFVS = new HashSet<int>();
        Solve();

        Console.WriteLine(bestFVS.Count);
        if (bestFVS.Count > 0)
            Console.WriteLine(string.Join(" ", bestFVS));
    }

    static bool IsAcyclic()
    {
        int[] color = new int[n];

        for (int i = 0; i < n; i++)
        {
            if (color[i] == 0 && !currentFVS.Contains(i))
            {
                if (HasCycleDFS(i, color))
                    return false;
            }
        }
        return true;
    }

    static bool HasCycleDFS(int u, int[] color)
    {
        color[u] = 1;

        foreach (int v in adj[u])
        {
            if (currentFVS.Contains(v)) continue;

            if (color[v] == 1) return true;
            if (color[v] == 0 && HasCycleDFS(v, color))
                return true;
        }

        color[u] = 2;
        return false;
    }

    static List<int> FindCycle()
    {
        int[] color = new int[n];

        for (int i = 0; i < n; i++)
        {
            if (color[i] == 0 && !currentFVS.Contains(i))
            {
                var cycle = FindCycleDFS(i, color, new List<int>());
                if (cycle.Count > 0) return cycle;
            }
        }
        return new List<int>();
    }

    static List<int> FindCycleDFS(int u, int[] color, List<int> path)
    {
        color[u] = 1;
        path.Add(u);

        foreach (int v in adj[u])
        {
            if (currentFVS.Contains(v)) continue;

            if (color[v] == 1)
            {
                int idx = path.IndexOf(v);
                var cycle = path.GetRange(idx, path.Count - idx);
                path.RemoveAt(path.Count - 1);
                color[u] = 2;
                return cycle;
            }

            if (color[v] == 0)
            {
                var cycle = FindCycleDFS(v, color, path);
                if (cycle.Count > 0)
                {
                    path.RemoveAt(path.Count - 1);
                    color[u] = 2;
                    return cycle;
                }
            }
        }

        path.RemoveAt(path.Count - 1);
        color[u] = 2;
        return new List<int>();
    }

    static void Solve()
    {
        if (IsAcyclic())
        {
            if (currentFVS.Count < bestFVS.Count || bestFVS.Count == 0)
                bestFVS = currentFVS.ToList();
            return;
        }

        if (bestFVS.Count > 0 && currentFVS.Count >= bestFVS.Count)
            return;

        var cycle = FindCycle();

        if (cycle.Count == 0)
            return;

        foreach (int v in cycle)
        {
            if (currentFVS.Contains(v)) continue;

            currentFVS.Add(v);
            Solve();
            currentFVS.Remove(v);
        }
    }
}
