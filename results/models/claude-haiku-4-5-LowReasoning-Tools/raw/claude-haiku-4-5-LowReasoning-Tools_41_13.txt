OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for input/output locations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Decorations for UBO
OpDecorate %UBO Block
OpDecorate %ubo_binding DescriptorSet 0
OpDecorate %ubo_binding Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4

; UBO struct type
%UBO = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float

; Pointer types
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float

; Function type
%func_type = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%v3_bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%v3_top = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Variables
%ubo_binding = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v4float Input
%out_color = OpVariable %ptr_Output_v4float Output

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load worldPos
%worldPos = OpLoad %v3float %in_worldPos

; Extract Y component (index 1)
%worldPos_y = OpCompositeExtract %float %worldPos 1

; Calculate t = worldPos.y * 0.5 + 0.5
%t_scaled = OpFMul %float %worldPos_y %float_0_5
%t_raw = OpFAdd %float %t_scaled %float_0_5

; Clamp t to [0, 1]
%t_min = OpFMin %float %t_raw %float_1
%t = OpFMax %float %t_min %float_0

; Mix colors: result = bottom + (top - bottom) * t
%color_diff = OpFSub %v3float %v3_top %v3_bottom
%scaled_diff = OpVectorTimesScalar %v3float %color_diff %t
%mixed_color = OpFAdd %v3float %v3_bottom %scaled_diff

; Construct output vec4 with alpha = 1.0
%output_color = OpCompositeConstruct %v4float %mixed_color %float_1

; Store output
OpStore %out_color %output_color

OpReturn
OpFunctionEnd
