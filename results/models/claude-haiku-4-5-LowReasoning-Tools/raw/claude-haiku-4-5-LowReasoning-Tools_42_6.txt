OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %arr_pv_in ArrayStride 16
               OpDecorate %gl_in Binding 0
               OpDecorate %gl_in DescriptorSet 0
               OpDecorate %color Location 2
               OpDecorate %gl_out Block
               OpMemberDecorate %gl_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
       %func = OpTypeFunction %void
     %float = OpTypeFloat 32
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0
     %v4float = OpTypeVector %float 4
     %v3float = OpTypeVector %float 3
       %bool = OpTypeBool
   %uint_3 = OpConstant %uint 3
   %int_0 = OpConstant %int 0
   %int_1 = OpConstant %int 1
   %int_2 = OpConstant %int 2
   %int_3 = OpConstant %int 3
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr Input
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_Function_int = OpTypePointer Function %int
       %main = OpFunction %void None %func
      %label = OpLabel
       %i_var = OpVariable %ptr_Function_int Function
               OpStore %i_var %int_0
               OpBranch %loop_check
  %loop_check = OpLabel
      %i = OpLoad %int %i_var
      %cond = OpSLessThan %bool %i %int_3
               OpLoopMerge %loop_end %loop_continue None
               OpBranchConditional %cond %loop_body %loop_end
  %loop_body = OpLabel
      %pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %i %int_0
      %pos = OpLoad %v4float %pos_ptr
      %out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr %pos
      %color_ptr = OpAccessChain %ptr_in_v3 %color %i
      %color_val = OpLoad %v3float %color_ptr
               OpStore %outColor %color_val
               OpEmitVertex
               OpBranch %loop_continue
%loop_continue = OpLabel
      %i_next = OpIAdd %int %i %int_1
               OpStore %i_var %i_next
               OpBranch %loop_check
  %loop_end = OpLabel
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
