#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> c(n);
    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    vector<pair<int, int>> constraints(m);
    for (int i = 0; i < m; i++) {
        cin >> constraints[i].first >> constraints[i].second;
    }

    long long best_objective = -1;
    vector<int> best_solution(n, 0);

    // Enumerate all 2^n possible assignments
    for (long long mask = 0; mask < (1LL << n); mask++) {
        // Check if this assignment satisfies all constraints
        bool valid = true;
        for (const auto& [u, v] : constraints) {
            int x_u = (mask >> u) & 1;
            int x_v = (mask >> v) & 1;
            if (x_u + x_v > 1) {
                valid = false;
                break;
            }
        }

        if (valid) {
            // Calculate objective value for this assignment
            long long objective = 0;
            for (int i = 0; i < n; i++) {
                if ((mask >> i) & 1) {
                    objective += c[i];
                }
            }

            // Update best solution if this is better
            if (objective > best_objective) {
                best_objective = objective;
                for (int i = 0; i < n; i++) {
                    best_solution[i] = (mask >> i) & 1;
                }
            }
        }
    }

    // Output the result
    cout << best_objective << "\n";
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << best_solution[i];
    }
    cout << "\n";

    return 0;
}
