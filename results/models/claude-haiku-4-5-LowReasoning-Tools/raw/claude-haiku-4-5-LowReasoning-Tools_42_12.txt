OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 15
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Flatten
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeUInt 32
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4
    %ptr_float = OpTypePointer Function %float
    %ptr_v3float = OpTypePointer Function %v3float
    %ptr_v4float = OpTypePointer Function %v4float

%gl_PerVertex_in = OpTypeStruct %v4float
   %uint_6 = OpConstant %uint 6
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float
     %gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
 %normal_in = OpVariable %ptr_in_arr_v3 Input
  %color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
   %ptr_out_v4float = OpTypePointer Output %v4float
   %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3float = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3float Output

%func_type = OpTypeFunction %void

    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_2 = OpConstant %int 2
    %int_3 = OpConstant %int 3
    %int_4 = OpConstant %int 4
    %int_5 = OpConstant %int 5

   %float_0 = OpConstant %float 0
   %float_1 = OpConstant %float 1
 %float_0p3 = OpConstant %float 0.3

 %vec_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

       %main = OpFunction %void None %func_type
      %entry = OpLabel

; Load the 6 input vertices (0,2,4 are triangle, 1,3,5 are adjacent)
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr
%pos4_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%pos5_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_5 %int_0
%pos5 = OpLoad %v4float %pos5_ptr

; Load colors
%col0 = OpLoad %v3float %color_in %int_0
%col2 = OpLoad %v3float %color_in %int_2
%col4 = OpLoad %v3float %color_in %int_4

; Emit the main triangle with dimmed colors (0.3x)
; Triangle vertex 0
%col0_dimmed = OpVectorTimesScalar %v3float %col0 %float_0p3
%gl_out_pos0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gl_out_pos0 %pos0
OpStore %outColor %col0_dimmed
OpEmitVertex

; Triangle vertex 2
%col2_dimmed = OpVectorTimesScalar %v3float %col2 %float_0p3
OpStore %gl_out_pos0 %pos2
OpStore %outColor %col2_dimmed
OpEmitVertex

; Triangle vertex 4
%col4_dimmed = OpVectorTimesScalar %v3float %col4 %float_0p3
OpStore %gl_out_pos0 %pos4
OpStore %outColor %col4_dimmed
OpEmitVertex

OpEndPrimitive

; Check edge 0-2: adjacent vertex is pos1
; If pos1 == pos0 or pos1 == pos2, it's a boundary
%pos1_xy0 = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%pos0_xy = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%pos2_xy = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%eq_1_0 = OpFOrdEqual %bool %pos1_xy0 %pos0_xy
%eq_1_2 = OpFOrdEqual %bool %pos1_xy0 %pos2_xy
%edge_0_2_boundary = OpLogicalOr %bool %eq_1_0 %eq_1_2

OpSelectionMerge %merge_edge_0_2 None
OpBranchConditional %edge_0_2_boundary %emit_edge_0_2 %merge_edge_0_2

%emit_edge_0_2 = OpLabel
; Emit edge 0-2 as white line (simplified: just 2 verts)
OpStore %outColor %vec_white
OpStore %gl_out_pos0 %pos0
OpEmitVertex
OpStore %outColor %vec_white
OpStore %gl_out_pos0 %pos2
OpEmitVertex
OpEndPrimitive
OpBranch %merge_edge_0_2

%merge_edge_0_2 = OpLabel

; Check edge 2-4: adjacent vertex is pos3
%pos3_xy = OpVectorShuffle %v3float %pos3 %pos3 0 1 2
%eq_3_2 = OpFOrdEqual %bool %pos3_xy %pos2_xy
%pos4_xy = OpVectorShuffle %v3float %pos4 %pos4 0 1 2
%eq_3_4 = OpFOrdEqual %bool %pos3_xy %pos4_xy
%edge_2_4_boundary = OpLogicalOr %bool %eq_3_2 %eq_3_4

OpSelectionMerge %merge_edge_2_4 None
OpBranchConditional %edge_2_4_boundary %emit_edge_2_4 %merge_edge_2_4

%emit_edge_2_4 = OpLabel
OpStore %outColor %vec_white
OpStore %gl_out_pos0 %pos2
OpEmitVertex
OpStore %outColor %vec_white
OpStore %gl_out_pos0 %pos4
OpEmitVertex
OpEndPrimitive
OpBranch %merge_edge_2_4

%merge_edge_2_4 = OpLabel

; Check edge 4-0: adjacent vertex is pos5
%pos5_xy = OpVectorShuffle %v3float %pos5 %pos5 0 1 2
%pos0_xy_again = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%eq_5_4 = OpFOrdEqual %bool %pos5_xy %pos4_xy
%eq_5_0 = OpFOrdEqual %bool %pos5_xy %pos0_xy_again
%edge_4_0_boundary = OpLogicalOr %bool %eq_5_4 %eq_5_0

OpSelectionMerge %merge_edge_4_0 None
OpBranchConditional %edge_4_0_boundary %emit_edge_4_0 %merge_edge_4_0

%emit_edge_4_0 = OpLabel
OpStore %outColor %vec_white
OpStore %gl_out_pos0 %pos4
OpEmitVertex
OpStore %outColor %vec_white
OpStore %gl_out_pos0 %pos0
OpEmitVertex
OpEndPrimitive
OpBranch %merge_edge_4_0

%merge_edge_4_0 = OpLabel

       OpReturn
       OpFunctionEnd
