using System;
using System.IO;
using System.Threading;
using System.Collections.Generic;
using System.Diagnostics;
using System.Runtime.CompilerServices;

class Program {
    static volatile bool timeout = false;
    static Stopwatch sw = Stopwatch.StartNew();
    
    static int N, M;
    static int[][] adj;
    
    static volatile int best_clique_size = 0;
    static int[] best_clique = new int[0];
    static object best_clique_lock = new object();

    static class FastIO {
        static byte[] buffer = new byte[1024 * 1024];
        static int pos = 0;
        static int len = 0;
        static Stream stream;

        public static void Init(Stream s) {
            stream = s;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        static void ReadMore() {
            if (pos >= len) {
                len = stream.Read(buffer, 0, buffer.Length);
                pos = 0;
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public static int NextInt() {
            ReadMore();
            if (len == 0) return -1;
            while (buffer[pos] < '0' || buffer[pos] > '9') {
                pos++;
                ReadMore();
                if (len == 0) return -1;
            }
            int res = 0;
            while (pos < len && buffer[pos] >= '0' && buffer[pos] <= '9') {
                res = res * 10 + (buffer[pos] - '0');
                pos++;
                if (pos >= len) ReadMore();
            }
            return res;
        }
    }

    class ThreadState {
        public int[] gen;
        public int current_gen;
        public int[] color_mark;
        public int color_mark_gen;
        public int[] color_count;
        public int[] color_start;
        public int[] stack_P;
        public int[] stack_colors;
        public int[] stack_P_sorted;
        public int[] stack_colors_sorted;

        public ThreadState(int N) {
            gen = new int[N];
            color_mark = new int[N + 2];
            color_count = new int[N + 2];
            color_start = new int[N + 2];
            stack_P = new int[1024];
            stack_colors = new int[1024];
            stack_P_sorted = new int[1024];
            stack_colors_sorted = new int[1024];
        }

        public void EnsureCapacity(int required) {
            if (stack_P.Length < required) {
                int newSize = Math.Max(required, stack_P.Length * 2);
                Array.Resize(ref stack_P, newSize);
                Array.Resize(ref stack_colors, newSize);
                Array.Resize(ref stack_P_sorted, newSize);
                Array.Resize(ref stack_colors_sorted, newSize);
            }
        }
    }

    static void Search(int[] C, int c_len, int p_offset, int p_len, ThreadState state, int current_offset) {
        if (timeout) return;
        if (c_len > best_clique_size) {
            lock (best_clique_lock) {
                if (c_len > best_clique_size) {
                    best_clique_size = c_len;
                    best_clique = new int[c_len];
                    Array.Copy(C, best_clique, c_len);
                }
            }
        }
        if (p_len == 0) return;

        state.EnsureCapacity(current_offset + p_len);

        int max_color = 0;
        
        for (int i = 0; i < p_len; i++) {
            int v = state.stack_P[p_offset + i];
            
            if (state.current_gen >= 2000000000) {
                Array.Clear(state.gen, 0, state.gen.Length);
                state.current_gen = 0;
            }
            state.current_gen++;
            foreach (int u in adj[v]) {
                state.gen[u] = state.current_gen;
            }

            if (state.color_mark_gen >= 2000000000) {
                Array.Clear(state.color_mark, 0, state.color_mark.Length);
                state.color_mark_gen = 0;
            }
            state.color_mark_gen++;
            for (int j = 0; j < i; j++) {
                int u = state.stack_P[p_offset + j];
                if (state.gen[u] == state.current_gen) {
                    state.color_mark[state.stack_colors[current_offset + j]] = state.color_mark_gen;
                }
            }
            
            int c = 1;
            while (state.color_mark[c] == state.color_mark_gen) c++;
            state.stack_colors[current_offset + i] = c;
            if (c > max_color) max_color = c;
        }

        Array.Clear(state.color_count, 1, max_color);
        for (int i = 0; i < p_len; i++) {
            state.color_count[state.stack_colors[current_offset + i]]++;
        }
        state.color_start[1] = 0;
        for (int c = 2; c <= max_color; c++) {
            state.color_start[c] = state.color_start[c - 1] + state.color_count[c - 1];
        }
        for (int i = 0; i < p_len; i++) {
            int v = state.stack_P[p_offset + i];
            int c = state.stack_colors[current_offset + i];
            int pos = current_offset + state.color_start[c]++;
            state.stack_P_sorted[pos] = v;
            state.stack_colors_sorted[pos] = c;
        }

        for (int i = p_len - 1; i >= 0; i--) {
            if (timeout) return;
            int v = state.stack_P_sorted[current_offset + i];
            int c = state.stack_colors_sorted[current_offset + i];

            if (c_len + c <= best_clique_size) {
                break;
            }

            if (state.current_gen >= 2000000000) {
                Array.Clear(state.gen, 0, state.gen.Length);
                state.current_gen = 0;
            }
            state.current_gen++;
            foreach (int u in adj[v]) {
                state.gen[u] = state.current_gen;
            }
            
            int p_next_len = 0;
            int next_p_offset = current_offset + p_len;
            state.EnsureCapacity(next_p_offset + i);
            
            for (int j = 0; j < i; j++) {
                int u = state.stack_P_sorted[current_offset + j];
                if (state.gen[u] == state.current_gen) {
                    state.stack_P[next_p_offset + p_next_len++] = u;
                }
            }

            C[c_len] = v;
            Search(C, c_len + 1, next_p_offset, p_next_len, state, next_p_offset + p_next_len);
        }
    }

    static void Main(string[] args) {
        Thread timerThread = new Thread(() => {
            while (sw.ElapsedMilliseconds < 290000) {
                Thread.Sleep(100);
            }
            timeout = true;
        });
        timerThread.IsBackground = true;
        timerThread.Start();

        FastIO.Init(Console.OpenStandardInput());
        N = FastIO.NextInt();
        if (N == -1) return;
        M = FastIO.NextInt();

        if (N == 0) {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[] U = new int[M];
        int[] V = new int[M];
        int[] deg = new int[N];
        for (int i = 0; i < M; i++) {
            U[i] = FastIO.NextInt();
            V[i] = FastIO.NextInt();
            deg[U[i]]++;
            deg[V[i]]++;
        }

        adj = new int[N][];
        for (int i = 0; i < N; i++) adj[i] = new int[deg[i]];
        int[] count = new int[N];
        for (int i = 0; i < M; i++) {
            int u = U[i], v = V[i];
            adj[u][count[u]++] = v;
            adj[v][count[v]++] = u;
        }
        U = null; V = null; GC.Collect();

        for (int i = 0; i < N; i++) {
            if (adj[i].Length > 0) {
                Array.Sort(adj[i]);
                int unique = 1;
                for (int j = 1; j < adj[i].Length; j++) {
                    if (adj[i][j] != adj[i][j - 1]) adj[i][unique++] = adj[i][j];
                }
                int final_len = 0;
                for (int j = 0; j < unique; j++) {
                    if (adj[i][j] != i) adj[i][final_len++] = adj[i][j];
                }
                if (final_len != adj[i].Length) Array.Resize(ref adj[i], final_len);
                deg[i] = adj[i].Length;
            }
        }

        int[] head = new int[N];
        int[] next = new int[N];
        int[] prev = new int[N];
        for (int i = 0; i < N; i++) head[i] = -1;

        for (int i = 0; i < N; i++) {
            int d = deg[i];
            next[i] = head[d];
            if (head[d] != -1) prev[head[d]] = i;
            prev[i] = -1;
            head[d] = i;
        }

        int min_deg = 0;
        int[] degeneracy_order = new int[N];
        int order_idx = 0;
        bool[] removed = new bool[N];
        int[] new_deg = new int[N];
        Array.Copy(deg, new_deg, N);

        for (int step = 0; step < N; step++) {
            while (min_deg < N && head[min_deg] == -1) min_deg++;
            if (min_deg == N) break;
            int v = head[min_deg];
            
            head[min_deg] = next[v];
            if (head[min_deg] != -1) prev[head[min_deg]] = -1;
            
            degeneracy_order[order_idx++] = v;
            removed[v] = true;
            
            foreach (int u in adj[v]) {
                if (!removed[u]) {
                    int d = new_deg[u];
                    if (prev[u] != -1) next[prev[u]] = next[u];
                    else head[d] = next[u];
                    if (next[u] != -1) prev[next[u]] = prev[u];
                    
                    new_deg[u]--;
                    d--;
                    
                    next[u] = head[d];
                    if (head[d] != -1) prev[head[d]] = u;
                    prev[u] = -1;
                    head[d] = u;
                    
                    if (d < min_deg) min_deg = d;
                }
            }
        }

        int[] oldToNew = new int[N];
        for (int i = 0; i < N; i++) oldToNew[degeneracy_order[i]] = i;

        int[][] new_adj = new int[N][];
        for (int i = 0; i < N; i++) {
            int u = degeneracy_order[i]; 
            int[] neighbors = adj[u];
            new_adj[i] = new int[neighbors.Length];
            for (int j = 0; j < neighbors.Length; j++) {
                new_adj[i][j] = oldToNew[neighbors[j]];
            }
            Array.Sort(new_adj[i]);
        }
        adj = new_adj;

        int h_count = Math.Min(N, 1000);
        int[] h_gen = new int[N];
        int h_current_gen = 0;
        for (int start = N - 1; start >= N - h_count; start--) {
            int[] C = new int[N];
            int c_len = 0;
            C[c_len++] = start;
            
            h_current_gen++;
            foreach (int u in adj[start]) h_gen[u] = h_current_gen;

            for (int i = N - 1; i >= 0; i--) {
                if (i == start) continue;
                int v = i;
                if (h_gen[v] == h_current_gen) {
                    bool ok = true;
                    for (int j = 1; j < c_len; j++) {
                        if (Array.BinarySearch(adj[v], C[j]) < 0) {
                            ok = false;
                            break;
                        }
                    }
                    if (ok) C[c_len++] = v;
                }
            }
            if (c_len > best_clique_size) {
                best_clique_size = c_len;
                best_clique = new int[c_len];
                Array.Copy(C, best_clique, c_len);
            }
        }

        int[] P_top = new int[N];
        for (int i = 0; i < N; i++) P_top[i] = i;

        ThreadState topState = new ThreadState(N);
        int max_color_top = 0;
        for (int i = 0; i < N; i++) {
            int v = P_top[i];
            topState.current_gen++;
            foreach (int u in adj[v]) topState.gen[u] = topState.current_gen;

            topState.color_mark_gen++;
            for (int j = 0; j < i; j++) {
                int u = P_top[j];
                if (topState.gen[u] == topState.current_gen) {
                    topState.color_mark[topState.stack_colors[j]] = topState.color_mark_gen;
                }
            }
            int c = 1;
            while (topState.color_mark[c] == topState.color_mark_gen) c++;
            topState.stack_colors[i] = c;
            if (c > max_color_top) max_color_top = c;
        }

        Array.Clear(topState.color_count, 1, max_color_top);
        for (int i = 0; i < N; i++) topState.color_count[topState.stack_colors[i]]++;
        topState.color_start[1] = 0;
        for (int c = 2; c <= max_color_top; c++) topState.color_start[c] = topState.color_start[c - 1] + topState.color_count[c - 1];
        for (int i = 0; i < N; i++) {
            int v = P_top[i];
            int c = topState.stack_colors[i];
            int pos = topState.color_start[c]++;
            topState.stack_P_sorted[pos] = v;
            topState.stack_colors_sorted[pos] = c;
        }

        int active_threads = Environment.ProcessorCount;
        if (active_threads < 1) active_threads = 1;
        Thread[] threads = new Thread[active_threads];
        int top_idx = N;

        for (int t = 0; t < active_threads; t++) {
            threads[t] = new Thread(() => {
                ThreadState state = new ThreadState(N);
                int[] C = new int[N]; 

                while (!timeout) {
                    int i = Interlocked.Decrement(ref top_idx);
                    if (i < 0) break;

                    int v = topState.stack_P_sorted[i];
                    int c = topState.stack_colors_sorted[i];

                    if (1 + c <= best_clique_size) {
                        Interlocked.Exchange(ref top_idx, -1);
                        break;
                    }

                    if (state.current_gen >= 2000000000) {
                        Array.Clear(state.gen, 0, state.gen.Length);
                        state.current_gen = 0;
                    }
                    state.current_gen++;
                    foreach (int u in adj[v]) state.gen[u] = state.current_gen;

                    int p_next_len = 0;
                    state.EnsureCapacity(i);
                    
                    for (int j = 0; j < i; j++) {
                        int u = topState.stack_P_sorted[j];
                        if (state.gen[u] == state.current_gen) {
                            state.stack_P[p_next_len++] = u;
                        }
                    }

                    C[0] = v;
                    Search(C, 1, 0, p_next_len, state, p_next_len);
                }
            }, 1024 * 1024 * 64);
            threads[t].Start();
        }

        foreach (var t in threads) t.Join();

        int[] final_clique = new int[best_clique_size];
        for (int i = 0; i < best_clique_size; i++) {
            final_clique[i] = degeneracy_order[best_clique[i]];
        }
        Array.Sort(final_clique);
        Console.WriteLine(best_clique_size);
        Console.WriteLine(string.Join(" ", final_clique));
    }
}
