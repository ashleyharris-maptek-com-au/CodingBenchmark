; SPIR-V
; Version: 1.0

OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%bool = OpTypeBool
%int = OpTypeInt 32 1

%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%float_8 = OpConstant %float 8.0
%vec2_8 = OpConstantComposite %vec2 %float_8 %float_8
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_0_2 = OpConstant %float 0.2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

%int_3 = OpConstant %int 3
%int_1 = OpConstant %int 1
%int_0 = OpConstant %int 0

%main = OpFunction %void None %func
%label = OpLabel

%uv_val = OpLoad %vec2 %uv
%scaled = OpFMul %vec2 %uv_val %vec2_8
%floored = OpExtInst %vec2 %1 Floor %scaled

%fx = OpCompositeExtract %float %floored 0
%fy = OpCompositeExtract %float %floored 1
%ix = OpConvertFToS %int %fx
%iy = OpConvertFToS %int %fy
%isum = OpIAdd %int %ix %iy
%ibit = OpBitwiseAnd %int %isum %int_1
%icmp = OpIEqual %bool %ibit %int_0
%baseColor = OpSelect %float %icmp %float_0_9 %float_0_2

%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

%wp = OpLoad %vec3 %worldPos
%L_dir = OpFSub %vec3 %lightPos %wp
%L = OpExtInst %vec3 %1 Normalize %L_dir

%n_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %1 Normalize %n_val

%dot = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %dot %float_0

%diff_05 = OpFMul %float %float_0_5 %diffuse
%shade = OpFAdd %float %float_0_5 %diff_05

%final_color = OpFMul %float %baseColor %shade
%out_vec4 = OpCompositeConstruct %vec4 %final_color %final_color %final_color %float_1
OpStore %outColor %out_vec4

OpReturn
OpFunctionEnd
