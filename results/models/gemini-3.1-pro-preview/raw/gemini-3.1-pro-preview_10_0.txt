#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;

const long long INF = 1e18;

chrono::time_point<chrono::steady_clock> start_time;

double elapsed() {
    auto now = chrono::steady_clock::now();
    return chrono::duration<double>(now - start_time).count();
}

struct FRect {
    int x, y, w, h;
};

struct Board {
    vector<FRect> free_rects;
    int max_dim1, max_dim2;

    Board(int w, int h) {
        free_rects.push_back({0, 0, w, h});
        update_stats();
    }

    void update_stats() {
        max_dim1 = 0;
        max_dim2 = 0;
        for (const auto& f : free_rects) {
            int fw = f.w;
            int fh = f.h;
            if (fw < fh) {
                if (fw > max_dim1) max_dim1 = fw;
                if (fh > max_dim2) max_dim2 = fh;
            } else {
                if (fh > max_dim1) max_dim1 = fh;
                if (fw > max_dim2) max_dim2 = fw;
            }
        }
    }
};

struct Placement {
    int board_index;
    int x, y;
    int rotated;
};

struct RectInput {
    int id;
    int w, h;
};

struct Score {
    long long score1;
    long long score2;
    bool operator<(const Score& o) const {
        if (score1 != o.score1) return score1 < o.score1;
        return score2 < o.score2;
    }
};

enum FitAlgo { BSSF, BAF, BLSF };

Score get_bssf(const FRect& F, int w, int h) {
    return { min(F.w - w, F.h - h), max(F.w - w, F.h - h) };
}

Score get_baf(const FRect& F, int w, int h) {
    return { 1LL * F.w * F.h - 1LL * w * h, (long long)min(F.w - w, F.h - h) };
}

Score get_blsf(const FRect& F, int w, int h) {
    return { max(F.w - w, F.h - h), min(F.w - w, F.h - h) };
}

bool split_rect(const FRect& F, int px, int py, int pw, int ph, vector<FRect>& new_rects) {
    if (px >= F.x + F.w || px + pw <= F.x || py >= F.y + F.h || py + ph <= F.y) {
        return false;
    }
    if (py > F.y) new_rects.push_back({F.x, F.y, F.w, py - F.y});
    if (py + ph < F.y + F.h) new_rects.push_back({F.x, py + ph, F.w, F.y + F.h - (py + ph)});
    if (px > F.x) new_rects.push_back({F.x, F.y, px - F.x, F.h});
    if (px + pw < F.x + F.w) new_rects.push_back({px + pw, F.y, F.x + F.w - (px + pw), F.h});
    return true;
}

void prune(vector<FRect>& rects, int max_free) {
    for (int i = 0; i < (int)rects.size(); ++i) {
        for (int j = 0; j < (int)rects.size(); ++j) {
            if (i == j) continue;
            if (rects[i].x >= rects[j].x && rects[i].y >= rects[j].y &&
                rects[i].x + rects[i].w <= rects[j].x + rects[j].w &&
                rects[i].y + rects[i].h <= rects[j].y + rects[j].h) {
                rects[i] = rects.back();
                rects.pop_back();
                --i;
                break;
            }
        }
    }
    if ((int)rects.size() > max_free) {
        sort(rects.begin(), rects.end(), [](const FRect& a, const FRect& b) {
            return 1LL * a.w * a.h > 1LL * b.w * b.h;
        });
        rects.resize(max_free);
    }
}

void place_rect(Board& b, int px, int py, int pw, int ph, int max_free) {
    int n = b.free_rects.size();
    vector<FRect> new_rects;
    new_rects.reserve(n * 4);
    for (int i = 0; i < n; ++i) {
        if (split_rect(b.free_rects[i], px, py, pw, ph, new_rects)) {
            b.free_rects[i] = b.free_rects.back();
            b.free_rects.pop_back();
            --i;
            --n;
        }
    }
    b.free_rects.insert(b.free_rects.end(), new_rects.begin(), new_rects.end());
    prune(b.free_rects, max_free);
    b.update_stats();
}

vector<Placement> solve_heuristic(vector<RectInput> rects, int BOARD_W, int BOARD_H, int sort_algo, FitAlgo fit_algo, int max_free, bool is_first) {
    if (sort_algo == 0) {
        sort(rects.begin(), rects.end(), [](const RectInput& a, const RectInput& b) {
            long long a_area = 1LL * a.w * a.h;
            long long b_area = 1LL * b.w * b.h;
            if (a_area != b_area) return a_area > b_area;
            return max(a.w, a.h) > max(b.w, b.h);
        });
    } else if (sort_algo == 1) {
        sort(rects.begin(), rects.end(), [](const RectInput& a, const RectInput& b) {
            int a_max = max(a.w, a.h);
            int b_max = max(b.w, b.h);
            if (a_max != b_max) return a_max > b_max;
            return 1LL * a.w * a.h > 1LL * b.w * b.h;
        });
    } else if (sort_algo == 2) {
        sort(rects.begin(), rects.end(), [](const RectInput& a, const RectInput& b) {
            int a_per = a.w + a.h;
            int b_per = b.w + b.h;
            if (a_per != b_per) return a_per > b_per;
            return 1LL * a.w * a.h > 1LL * b.w * b.h;
        });
    }

    vector<Board> boards;
    boards.reserve(rects.size());
    vector<Placement> ans(rects.size());

    for (const auto& r : rects) {
        if (!is_first && elapsed() > 28.0) {
            return {}; 
        }

        int rw = min(r.w, r.h);
        int rh = max(r.w, r.h);

        int best_board = -1;
        int best_frect = -1;
        bool best_rot = false;
        Score best_score = {INF, INF};

        for (int b = 0; b < (int)boards.size(); ++b) {
            if (rw > boards[b].max_dim1 || rh > boards[b].max_dim2) continue;

            for (int f = 0; f < (int)boards[b].free_rects.size(); ++f) {
                const auto& F = boards[b].free_rects[f];
                
                if (r.w <= F.w && r.h <= F.h) {
                    Score s;
                    if (fit_algo == BSSF) s = get_bssf(F, r.w, r.h);
                    else if (fit_algo == BAF) s = get_baf(F, r.w, r.h);
                    else s = get_blsf(F, r.w, r.h);
                    
                    if (s < best_score) {
                        best_score = s; best_board = b; best_frect = f; best_rot = false;
                    }
                }
                
                if (r.h <= F.w && r.w <= F.h) {
                    Score s;
                    if (fit_algo == BSSF) s = get_bssf(F, r.h, r.w);
                    else if (fit_algo == BAF) s = get_baf(F, r.h, r.w);
                    else s = get_blsf(F, r.h, r.w);
                    
                    if (s < best_score) {
                        best_score = s; best_board = b; best_frect = f; best_rot = true;
                    }
                }
            }
        }

        if (best_board == -1) {
            boards.emplace_back(BOARD_W, BOARD_H);
            best_board = boards.size() - 1;
            best_frect = 0;
            
            Score s1 = {INF, INF}, s2 = {INF, INF};
            if (r.w <= BOARD_W && r.h <= BOARD_H) {
                if (fit_algo == BSSF) s1 = get_bssf(boards.back().free_rects[0], r.w, r.h);
                else if (fit_algo == BAF) s1 = get_baf(boards.back().free_rects[0], r.w, r.h);
                else s1 = get_blsf(boards.back().free_rects[0], r.w, r.h);
                best_rot = false;
            }
            if (r.h <= BOARD_W && r.w <= BOARD_H) {
                if (fit_algo == BSSF) s2 = get_bssf(boards.back().free_rects[0], r.h, r.w);
                else if (fit_algo == BAF) s2 = get_baf(boards.back().free_rects[0], r.h, r.w);
                else s2 = get_blsf(boards.back().free_rects[0], r.h, r.w);
                if (s2 < s1) best_rot = true;
            }
        }

        const auto& F = boards[best_board].free_rects[best_frect];
        int px = F.x;
        int py = F.y;
        int pw = best_rot ? r.h : r.w;
        int ph = best_rot ? r.w : r.h;

        ans[r.id] = {best_board, px, py, best_rot ? 1 : 0};
        
        place_rect(boards[best_board], px, py, pw, ph, max_free);
    }
    
    return ans;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, BOARD_W, BOARD_H;
    if (!(cin >> N >> BOARD_W >> BOARD_H)) return 0;

    vector<RectInput> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        cin >> rects[i].w >> rects[i].h;
    }

    start_time = chrono::steady_clock::now();

    int max_free = 400;
    if (N > 2000) max_free = 200;
    if (N > 10000) max_free = 100;
    if (N > 30000) max_free = 50;

    int best_total_boards = 2e9;
    vector<Placement> best_ans;

    vector<pair<int, FitAlgo>> heuristics = {
        {0, BAF}, {0, BSSF}, {0, BLSF},
        {1, BAF}, {1, BSSF}, {1, BLSF},
        {2, BAF}, {2, BSSF}, {2, BLSF}
    };

    bool is_first = true;
    for (const auto& heur : heuristics) {
        if (!is_first && elapsed() > 28.0) break;
        
        auto ans = solve_heuristic(rects, BOARD_W, BOARD_H, heur.first, heur.second, max_free, is_first);
        if (ans.empty()) continue; 

        int max_board = 0;
        for (const auto& p : ans) {
            if (p.board_index > max_board) max_board = p.board_index;
        }
        int total_boards = max_board + 1;

        if (total_boards < best_total_boards) {
            best_total_boards = total_boards;
            best_ans = ans;
        }
        is_first = false;
    }

    cout << best_total_boards << "\n";
    for (int i = 0; i < N; ++i) {
        cout << best_ans[i].board_index << " " 
             << best_ans[i].x << " " 
             << best_ans[i].y << " " 
             << best_ans[i].rotated << "\n";
    }

    return 0;
}
