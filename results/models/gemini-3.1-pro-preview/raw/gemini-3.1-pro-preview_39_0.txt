use std::io::{self, Read};
use std::time::Instant;

struct XorShift {
    state: u64,
}

impl XorShift {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    
    fn next(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    
    fn next_usize(&mut self, max: usize) -> usize {
        (self.next() % (max as u64)) as usize
    }
}

fn shuffle(arr: &mut [usize], rng: &mut XorShift) {
    let n = arr.len();
    if n <= 1 { return; }
    for i in (1..n).rev() {
        let j = rng.next_usize(i + 1);
        arr.swap(i, j);
    }
}

struct SolverState {
    best_cut: u32,
    best_part: Vec<u8>,
    start_time: Instant,
    call_count: u32,
    timeout: bool,
}

fn kernighan_lin(
    graph: &Vec<Vec<usize>>,
    adj: &Vec<u16>,
    use_adj: bool,
    n: usize,
    iterations: usize,
) -> (u32, Vec<u8>) {
    if n == 0 { return (0, vec![]); }
    
    let mut best_overall_cut = u32::MAX;
    let mut best_overall_part = vec![];
    let mut rng = XorShift::new(123456789);

    for _ in 0..iterations {
        let mut part = vec![0u8; n];
        let mut nodes: Vec<usize> = (0..n).collect();
        shuffle(&mut nodes, &mut rng);
        for i in 0..n/2 {
            part[nodes[i]] = 1;
        }

        let mut current_cut = 0;
        for u in 0..n {
            for &v in &graph[u] {
                if u < v && part[u] != part[v] {
                    current_cut += 1;
                }
            }
        }

        let mut improved = true;
        while improved {
            improved = false;
            
            let mut d = vec![0i32; n];
            for u in 0..n {
                for &v in &graph[u] {
                    if part[u] != part[v] {
                        d[u] += 1;
                    } else {
                        d[u] -= 1;
                    }
                }
            }

            let mut unvisited = vec![true; n];
            let mut max_gains = Vec::with_capacity(n / 2);
            let mut swaps = Vec::with_capacity(n / 2);

            for _ in 0..n/2 {
                let mut valid_u = Vec::with_capacity(n/2);
                let mut valid_v = Vec::with_capacity(n/2);
                for i in 0..n {
                    if unvisited[i] {
                        if part[i] == 0 {
                            valid_u.push(i);
                        } else {
                            valid_v.push(i);
                        }
                    }
                }

                if valid_u.is_empty() || valid_v.is_empty() {
                    break;
                }

                let mut best_gain = i32::MIN;
                let mut best_pair = (n, n);

                for &u in &valid_u {
                    let du = d[u];
                    for &v in &valid_v {
                        let is_edge = if use_adj {
                            adj[u * n + v] as i32
                        } else {
                            let mut cnt = 0;
                            for &neighbor in &graph[u] {
                                if neighbor == v { cnt += 1; }
                            }
                            cnt
                        };
                        
                        let gain = du + d[v] - 2 * is_edge;
                        if gain > best_gain {
                            best_gain = gain;
                            best_pair = (u, v);
                        }
                    }
                }

                if best_pair.0 == n { break; }
                
                let (u, v) = best_pair;
                unvisited[u] = false;
                unvisited[v] = false;
                max_gains.push(best_gain);
                swaps.push((u, v));
                
                for &w in &graph[u] {
                    if unvisited[w] {
                        if part[w] == part[u] {
                            d[w] += 2;
                        } else {
                            d[w] -= 2;
                        }
                    }
                }
                for &w in &graph[v] {
                    if unvisited[w] {
                        if part[w] == part[v] {
                            d[w] += 2;
                        } else {
                            d[w] -= 2;
                        }
                    }
                }
            }

            let mut max_prefix_sum = 0i32;
            let mut current_sum = 0i32;
            let mut best_k = 0;
            
            for k in 0..max_gains.len() {
                current_sum += max_gains[k];
                if current_sum > max_prefix_sum {
                    max_prefix_sum = current_sum;
                    best_k = k + 1;
                }
            }

            if max_prefix_sum > 0 {
                improved = true;
                for k in 0..best_k {
                    let (u, v) = swaps[k];
                    part[u] = 1;
                    part[v] = 0;
                }
                current_cut = (current_cut as i32 - max_prefix_sum) as u32;
            }
        }

        if current_cut < best_overall_cut {
            best_overall_cut = current_cut;
            best_overall_part = part;
        }
    }

    (best_overall_cut, best_overall_part)
}

fn get_mas_order(graph: &Vec<Vec<usize>>, n: usize) -> Vec<usize> {
    if n == 0 { return vec![]; }
    let mut order = Vec::with_capacity(n);
    let mut added = vec![false; n];
    let mut degrees = vec![0; n];
    for u in 0..n {
        degrees[u] = graph[u].len();
    }
    
    let mut first = 0;
    let mut max_deg = 0;
    for u in 0..n {
        if degrees[u] > max_deg {
            max_deg = degrees[u];
            first = u;
        }
    }
    
    order.push(first);
    added[first] = true;
    
    let mut connected = vec![0; n];
    for &v in &graph[first] {
        connected[v] += 1;
    }
    
    for _ in 1..n {
        let mut best_node = 0;
        let mut best_conn = -1;
        let mut best_deg = -1;
        
        for u in 0..n {
            if !added[u] {
                if connected[u] as i32 > best_conn || (connected[u] as i32 == best_conn && degrees[u] as i32 > best_deg) {
                    best_conn = connected[u] as i32;
                    best_deg = degrees[u] as i32;
                    best_node = u;
                }
            }
        }
        
        order.push(best_node);
        added[best_node] = true;
        for &v in &graph[best_node] {
            connected[v] += 1;
        }
    }
    
    order
}

fn dfs(
    idx: usize,
    count0: usize,
    count1: usize,
    current_cut: u32,
    lb_extra: u32,
    edges_to_0: &mut [u32],
    edges_to_1: &mut [u32],
    part: &mut [u8],
    state: &mut SolverState,
    graph: &Vec<Vec<usize>>,
    order: &[usize],
    n: usize,
) {
    if state.timeout {
        return;
    }
    
    state.call_count += 1;
    if state.call_count & 4095 == 0 {
        if state.start_time.elapsed().as_millis() > 1900 {
            state.timeout = true;
            return;
        }
    }

    if count0 == n / 2 {
        let mut final_cut = current_cut;
        for i in idx..n {
            let u = order[i];
            final_cut += edges_to_0[u];
        }
        if final_cut < state.best_cut {
            state.best_cut = final_cut;
            state.best_part = part.to_vec();
            for i in idx..n {
                let u = order[i];
                state.best_part[u] = 1;
            }
        }
        return;
    }
    
    if count1 == n / 2 {
        let mut final_cut = current_cut;
        for i in idx..n {
            let u = order[i];
            final_cut += edges_to_1[u];
        }
        if final_cut < state.best_cut {
            state.best_cut = final_cut;
            state.best_part = part.to_vec();
            for i in idx..n {
                let u = order[i];
                state.best_part[u] = 0;
            }
        }
        return;
    }

    if current_cut + lb_extra >= state.best_cut {
        return;
    }

    let u = order[idx];
    let cost0 = edges_to_1[u];
    let cost1 = edges_to_0[u];
    
    let try_0_first = cost0 <= cost1;

    macro_rules! assign_0 {
        () => {
            let added_cut = edges_to_1[u];
            let mut new_lb_extra = lb_extra - std::cmp::min(edges_to_0[u], edges_to_1[u]);
            
            for &w in &graph[u] {
                if part[w] == 2 {
                    let old_min = std::cmp::min(edges_to_0[w], edges_to_1[w]);
                    edges_to_0[w] += 1;
                    let new_min = std::cmp::min(edges_to_0[w], edges_to_1[w]);
                    new_lb_extra += new_min - old_min;
                }
            }
            
            part[u] = 0;
            dfs(idx + 1, count0 + 1, count1, current_cut + added_cut, new_lb_extra, edges_to_0, edges_to_1, part, state, graph, order, n);
            part[u] = 2;
            
            for &w in &graph[u] {
                if part[w] == 2 {
                    edges_to_0[w] -= 1;
                }
            }
        }
    }

    macro_rules! assign_1 {
        () => {
            let added_cut = edges_to_0[u];
            let mut new_lb_extra = lb_extra - std::cmp::min(edges_to_0[u], edges_to_1[u]);
            
            for &w in &graph[u] {
                if part[w] == 2 {
                    let old_min = std::cmp::min(edges_to_0[w], edges_to_1[w]);
                    edges_to_1[w] += 1;
                    let new_min = std::cmp::min(edges_to_0[w], edges_to_1[w]);
                    new_lb_extra += new_min - old_min;
                }
            }
            
            part[u] = 1;
            dfs(idx + 1, count0, count1 + 1, current_cut + added_cut, new_lb_extra, edges_to_0, edges_to_1, part, state, graph, order, n);
            part[u] = 2;
            
            for &w in &graph[u] {
                if part[w] == 2 {
                    edges_to_1[w] -= 1;
                }
            }
        }
    }

    if try_0_first {
        assign_0!();
        assign_1!();
    } else {
        assign_1!();
        assign_0!();
    }
}

fn solve(graph: &Vec<Vec<usize>>, adj: &Vec<u16>, use_adj: bool, n: usize) -> (u32, Vec<u8>) {
    if n == 0 { return (0, vec![]); }
    
    let kl_iters = if n <= 50 {
        2000
    } else if n <= 100 {
        1000
    } else if n <= 200 {
        400
    } else if n <= 500 {
        100
    } else {
        20
    };
    
    let (best_cut, best_part) = kernighan_lin(graph, adj, use_adj, n, kl_iters);
    
    let order = get_mas_order(graph, n);
    
    let mut state = SolverState {
        best_cut,
        best_part: best_part.clone(),
        start_time: Instant::now(),
        call_count: 0,
        timeout: false,
    };
    
    let mut edges_to_0 = vec![0; n];
    let mut edges_to_1 = vec![0; n];
    let mut part = vec![2u8; n];
    
    let u = order[0];
    part[u] = 0;
    for &w in &graph[u] {
        edges_to_0[w] += 1;
    }
    
    let mut initial_lb_extra = 0;
    for w in 0..n {
        if part[w] == 2 {
            initial_lb_extra += std::cmp::min(edges_to_0[w], edges_to_1[w]);
        }
    }
    
    dfs(1, 1, 0, 0, initial_lb_extra, &mut edges_to_0, &mut edges_to_1, &mut part, &mut state, graph, &order, n);
    
    (state.best_cut, state.best_part)
}

fn main() {
    let mut input = String::new();
    if let Err(_) = io::stdin().read_to_string(&mut input) {
        return;
    }
    let mut tokens = input.split_whitespace();

    let n_str = tokens.next();
    if n_str.is_none() { return; }
    let n: usize = n_str.unwrap().parse().unwrap();
    let m: usize = tokens.next().unwrap().parse().unwrap();

    let mut graph = vec![vec![]; n];
    let use_adj = n <= 5000;
    let mut adj = if use_adj { vec![0u16; n * n] } else { vec![] };

    for _ in 0..m {
        let u: usize = tokens.next().unwrap().parse().unwrap();
        let v: usize = tokens.next().unwrap().parse().unwrap();
        if u != v {
            graph[u].push(v);
            graph[v].push(u);
            if use_adj {
                adj[u * n + v] = adj[u * n + v].saturating_add(1);
                adj[v * n + u] = adj[v * n + u].saturating_add(1);
            }
        }
    }

    let (best_cut, best_part) = solve(&graph, &adj, use_adj, n);

    println!("{}", best_cut);
    let mut part_a = vec![];
    for i in 0..n {
        if best_part[i] == 0 {
            part_a.push(i);
        }
    }
    
    for i in 0..part_a.len() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", part_a[i]);
    }
    println!();
}
