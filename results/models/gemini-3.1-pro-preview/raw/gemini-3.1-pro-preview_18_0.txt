class FastIO {
    static const int BUF_SIZE = 1 << 16;
    char buf[BUF_SIZE];
    int pos, len;
public:
    FastIO() : pos(0), len(0) {}
    inline int nextInt() {
        int x = 0;
        char c;
        do {
            c = getChar();
            if (c == EOF) return -1;
        } while (c <= ' ');
        while (c > ' ') {
            x = x * 10 + (c - '0');
            c = getChar();
        }
        return x;
    }
    inline uint64_t nextUInt64() {
        uint64_t x = 0;
        char c;
        do {
            c = getChar();
            if (c == EOF) return (uint64_t)-1;
        } while (c <= ' ');
        while (c > ' ') {
            x = x * 10 + (c - '0');
            c = getChar();
        }
        return x;
    }
private:
    inline char getChar() {
        if (pos == len) {
            pos = 0;
            len = fread(buf, 1, BUF_SIZE, stdin);
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }
};

struct Edge {
    int to;
    int cap;
    int flow;
    int rev;
    int id; 
};

int N;
uint64_t M;
vector<pair<int, int>> original_edges;
vector<vector<Edge>> adj;

int best_cut;
vector<int> best_cut_edges_id;

void print_cut() {
    cout << best_cut_edges_id.size() << "\n";
    for (int id : best_cut_edges_id) {
        cout << original_edges[id].first << " " << original_edges[id].second << "\n";
    }
}

void run_stoer_wagner() {
    vector<vector<int>> adj_sw(N, vector<int>(N, 0));
    for (auto& e : original_edges) {
        adj_sw[e.first][e.second] += 1;
        adj_sw[e.second][e.first] += 1;
    }
    
    vector<vector<int>> nodes(N);
    for (int i = 0; i < N; ++i) nodes[i].push_back(i);
    
    int best_cut_sw = INT_MAX;
    vector<int> best_partition;
    
    vector<int> active;
    for (int i = 0; i < N; ++i) active.push_back(i);
    
    for (int phase = 1; phase < N; ++phase) {
        int n_active = active.size();
        vector<int> w(N, 0);
        vector<bool> added(N, false);
        int prev = -1, last = -1;
        
        for (int i = 0; i < n_active; ++i) {
            int sel = -1;
            for (int j_idx = 0; j_idx < n_active; ++j_idx) {
                int j = active[j_idx];
                if (!added[j] && (sel == -1 || w[j] > w[sel])) {
                    sel = j;
                }
            }
            if (i == n_active - 2) prev = sel;
            if (i == n_active - 1) last = sel;
            added[sel] = true;
            for (int j_idx = 0; j_idx < n_active; ++j_idx) {
                int j = active[j_idx];
                if (!added[j]) {
                    w[j] += adj_sw[sel][j];
                }
            }
        }
        
        if (w[last] < best_cut_sw) {
            best_cut_sw = w[last];
            best_partition = nodes[last];
        }
        
        for (int j_idx = 0; j_idx < n_active; ++j_idx) {
            int j = active[j_idx];
            if (j != prev && j != last) {
                adj_sw[prev][j] += adj_sw[last][j];
                adj_sw[j][prev] += adj_sw[j][last];
            }
        }
        nodes[prev].insert(nodes[prev].end(), nodes[last].begin(), nodes[last].end());
        
        for (int j_idx = 0; j_idx < n_active; ++j_idx) {
            if (active[j_idx] == last) {
                active[j_idx] = active.back();
                active.pop_back();
                break;
            }
        }
    }
    
    vector<bool> in_partition(N, false);
    for (int v : best_partition) in_partition[v] = true;
    
    vector<pair<int, int>> cut_edges;
    for (auto& e : original_edges) {
        if (in_partition[e.first] != in_partition[e.second]) {
            cut_edges.push_back(e);
        }
    }
    
    cout << cut_edges.size() << "\n";
    for (auto& e : cut_edges) cout << e.first << " " << e.second << "\n";
}

int timer_dfs = 1;
vector<int> tin, low;
int found_bridge_id = -1;

void run_bridge_finding() {
    tin.assign(N, 0);
    low.assign(N, 0);
    timer_dfs = 1;
    found_bridge_id = -1;
    
    struct Frame {
        int v;
        int edge_id_to_parent;
        int idx;
    };
    vector<Frame> st;
    
    for (int start_node = 0; start_node < N; ++start_node) {
        if (tin[start_node]) continue;
        
        st.push_back({start_node, -1, 0});
        tin[start_node] = low[start_node] = timer_dfs++;
        
        while (!st.empty()) {
            auto& f = st.back();
            int v = f.v;
            if (f.idx < (int)adj[v].size()) {
                auto& edge = adj[v][f.idx++];
                int to = edge.to;
                int id = edge.id;
                if (id == f.edge_id_to_parent) continue;
                
                if (tin[to]) {
                    low[v] = min(low[v], tin[to]);
                } else {
                    tin[to] = low[to] = timer_dfs++;
                    st.push_back({to, id, 0});
                }
            } else {
                st.pop_back();
                if (!st.empty() && st.back().v != -1) { 
                    int p = st.back().v;
                    low[p] = min(low[p], low[v]);
                    if (low[v] > tin[p]) {
                        found_bridge_id = f.edge_id_to_parent;
                        return;
                    }
                }
            }
        }
    }
}

int edmonds_karp(int s, int t, int limit, vector<int>& cut_edges_out) {
    for (int i = 0; i < N; ++i) {
        for (auto& edge : adj[i]) {
            edge.flow = 0;
        }
    }
    
    int flow = 0;
    vector<pair<int, int>> parent_edge(N, {-1, -1});
    while (flow < limit) {
        fill(parent_edge.begin(), parent_edge.end(), pair<int,int>{-1, -1});
        parent_edge[s] = {-2, -2};
        queue<int> q;
        q.push(s);
        bool reached = false;
        while (!q.empty() && !reached) {
            int v = q.front(); q.pop();
            for (int i = 0; i < (int)adj[v].size(); ++i) {
                auto& edge = adj[v][i];
                if (edge.cap - edge.flow > 0 && parent_edge[edge.to].first == -1) {
                    parent_edge[edge.to] = {v, i};
                    if (edge.to == t) {
                        reached = true;
                        break;
                    }
                    q.push(edge.to);
                }
            }
        }
        if (!reached) break;
        
        int curr = t;
        while (curr != s) {
            int p = parent_edge[curr].first;
            int idx = parent_edge[curr].second;
            adj[p][idx].flow += 1;
            int rev_idx = adj[p][idx].rev;
            adj[curr][rev_idx].flow -= 1;
            curr = p;
        }
        flow += 1;
    }
    
    if (flow < limit) {
        vector<bool> visited(N, false);
        queue<int> q;
        q.push(s);
        visited[s] = true;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& edge : adj[v]) {
                if (edge.cap - edge.flow > 0 && !visited[edge.to]) {
                    visited[edge.to] = true;
                    q.push(edge.to);
                }
            }
        }
        
        cut_edges_out.clear();
        for (int i = 0; i < N; ++i) {
            if (visited[i]) {
                for (auto& edge : adj[i]) {
                    if (!visited[edge.to] && edge.cap == 1) {
                        cut_edges_out.push_back(edge.id);
                    }
                }
            }
        }
    }
    return flow;
}

void build_adj() {
    adj.assign(N, vector<Edge>());
    for (int i = 0; i < (int)original_edges.size(); ++i) {
        int u = original_edges[i].first;
        int v = original_edges[i].second;
        adj[u].push_back({v, 1, 0, (int)adj[v].size(), i});
        adj[v].push_back({u, 1, 0, (int)adj[u].size() - 1, i});
    }
}

void run_edmonds_karp_randomized(int s) {
    auto start_time = chrono::steady_clock::now();
    vector<int> targets;
    for (int i = 0; i < N; ++i) {
        if (i != s) targets.push_back(i);
    }
    mt19937 rng(42);
    shuffle(targets.begin(), targets.end(), rng);
    
    for (int t : targets) {
        auto now = chrono::steady_clock::now();
        double elapsed = chrono::duration<double>(now - start_time).count();
        if (elapsed > 4.0) {
            break;
        }
        
        vector<int> current_cut;
        int flow = edmonds_karp(s, t, best_cut, current_cut);
        if (flow < best_cut) {
            best_cut = flow;
            best_cut_edges_id = current_cut;
            if (best_cut <= 2) {
                break;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    FastIO fio;
    N = fio.nextInt();
    if (N == -1) return 0;
    M = fio.nextUInt64();
    
    if (M > 200000000ULL) {
        vector<int> degree(N, 0);
        FILE* tmp = tmpfile();
        for (uint64_t i = 0; i < M; ++i) {
            int u = fio.nextInt();
            int v = fio.nextInt();
            if (u == -1 || v == -1) break;
            degree[u]++;
            degree[v]++;
            uint32_t edge[2] = {(uint32_t)u, (uint32_t)v};
            fwrite(edge, sizeof(uint32_t), 2, tmp);
        }
        int v_min = 0;
        for (int i = 1; i < N; ++i) {
            if (degree[i] < degree[v_min]) {
                v_min = i;
            }
        }
        rewind(tmp);
        vector<pair<int, int>> best_edges;
        for (uint64_t i = 0; i < M; ++i) {
            uint32_t edge[2];
            if (fread(edge, sizeof(uint32_t), 2, tmp) != 2) break;
            if (edge[0] == (uint32_t)v_min || edge[1] == (uint32_t)v_min) {
                best_edges.push_back({(int)edge[0], (int)edge[1]});
            }
        }
        cout << best_edges.size() << "\n";
        for (auto& e : best_edges) {
            cout << e.first << " " << e.second << "\n";
        }
        return 0;
    }
    
    original_edges.reserve(M);
    vector<int> degree(N, 0);
    for (uint64_t i = 0; i < M; ++i) {
        int u = fio.nextInt();
        int v = fio.nextInt();
        if (u == -1 || v == -1) break;
        original_edges.push_back({u, v});
        degree[u]++;
        degree[v]++;
    }
    
    if (N <= 1500) {
        run_stoer_wagner();
        return 0;
    }
    
    int v_min = 0, v_max = 0;
    for (int i = 1; i < N; ++i) {
        if (degree[i] < degree[v_min]) v_min = i;
        if (degree[i] > degree[v_max]) v_max = i;
    }
    
    best_cut = degree[v_min];
    for (int i = 0; i < (int)original_edges.size(); ++i) {
        if (original_edges[i].first == v_min || original_edges[i].second == v_min) {
            best_cut_edges_id.push_back(i);
        }
    }
    
    if (best_cut == 1) {
        print_cut();
        return 0;
    }
    
    build_adj();
    
    run_bridge_finding();
    if (found_bridge_id != -1) {
        best_cut = 1;
        best_cut_edges_id = {found_bridge_id};
        print_cut();
        return 0;
    }
    
    if (best_cut > 2) {
        run_edmonds_karp_randomized(v_max);
    }
    
    print_cut();
    return 0;
}
