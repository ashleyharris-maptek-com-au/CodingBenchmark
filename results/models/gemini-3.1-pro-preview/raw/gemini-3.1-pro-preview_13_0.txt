#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <future>
#include <cstring>

using namespace std;

struct HashPos {
    uint64_t h1, h2;
    int pos;
};

int main() {
    // Optimize standard I/O operations for performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;
    string line;
    getline(cin, line); // consume newline

    vector<string> strings;
    strings.reserve(n);
    for(int i = 0; i < n; ++i){
        string s;
        getline(cin, s);
        // Safely handle carriage return for Windows inputs
        if(!s.empty() && s.back() == '\r') s.pop_back();
        strings.push_back(move(s));
    }

    if(n == 0){
        cout << "\n";
        return 0;
    }

    if(n == 1){
        cout << strings[0] << "\n";
        return 0;
    }

    // Sort strings by length, shortest first to minimize search space
    sort(strings.begin(), strings.end(), [](const string& a, const string& b){
        return a.size() < b.size();
    });

    int min_len = (int)strings[0].size();
    if(min_len == 0){
        cout << "\n";
        return 0;
    }

    // Two different large primes for base to compute a 128-bit rolling hash
    uint64_t B1 = 1000000007ULL;
    uint64_t B2 = 1000000009ULL;

    vector<uint64_t> power1(min_len + 1);
    vector<uint64_t> power2(min_len + 1);
    power1[0] = 1; power2[0] = 1;
    for(int i = 1; i <= min_len; ++i){
        power1[i] = power1[i-1] * B1;
        power2[i] = power2[i-1] * B2;
    }

    int M_max = min_len;
    vector<int> head;
    vector<int> next_node;
    vector<HashPos> valid;
    vector<vector<uint8_t>> present_arrays(n);

    // Preallocate presence tracking arrays for thread safety and speed
    for(int k = 1; k < n; ++k) {
        present_arrays[k].resize(M_max + 1);
    }

    // Function to check if a common substring of length L exists
    auto check = [&](int L) -> vector<int> {
        if(L == 0) return {0};
        int M = min_len - L + 1;
        int table_size = 1;
        while(table_size < M * 3) table_size *= 2; // Optimal load factor

        head.resize(table_size);
        memset(head.data(), 0xff, table_size * sizeof(int));
        next_node.resize(M + 1);
        valid.clear();
        valid.reserve(M + 1);

        const string& S0 = strings[0];
        uint64_t h1 = 0, h2 = 0;
        for(int i = 0; i < L; ++i){
            h1 = h1 * B1 + (uint8_t)S0[i];
            h2 = h2 * B2 + (uint8_t)S0[i];
        }

        uint64_t BL1 = power1[L];
        uint64_t BL2 = power2[L];
        int mask = table_size - 1;

        auto add_hash = [&](uint64_t ch1, uint64_t ch2, int pos) {
            uint64_t comb = ch1 ^ (ch2 * 0x9e3779b97f4a7c15ULL);
            int bucket = comb & mask;
            int idx = head[bucket];
            bool found = false;
            while(idx != -1){
                if(valid[idx].h1 == ch1 && valid[idx].h2 == ch2){
                    found = true;
                    break;
                }
                idx = next_node[idx];
            }
            if(!found){
                int new_idx = (int)valid.size();
                valid.push_back({ch1, ch2, pos});
                next_node[new_idx] = head[bucket];
                head[bucket] = new_idx;
            }
        };

        add_hash(h1, h2, 0);
        for(int i = L; i < (int)S0.size(); ++i){
            h1 = h1 * B1 + (uint8_t)S0[i] - (uint8_t)S0[i-L] * BL1;
            h2 = h2 * B2 + (uint8_t)S0[i] - (uint8_t)S0[i-L] * BL2;
            add_hash(h1, h2, i - L + 1);
        }

        // Reset presence array for actual checked length
        for(int k = 1; k < n; ++k){
            memset(present_arrays[k].data(), 0, valid.size());
        }

        vector<future<void>> futures;
        for(int k = 1; k < n; ++k){
            // Process other strings concurrently
            futures.push_back(async(launch::async | launch::deferred, [&, k]() {
                const string& S = strings[k];
                if((int)S.size() < L) return;
                uint8_t* present = present_arrays[k].data();
                uint64_t lh1 = 0, lh2 = 0;
                for(int i = 0; i < L; ++i){
                    lh1 = lh1 * B1 + (uint8_t)S[i];
                    lh2 = lh2 * B2 + (uint8_t)S[i];
                }
                
                auto check_hash = [&](uint64_t ch1, uint64_t ch2) {
                    uint64_t comb = ch1 ^ (ch2 * 0x9e3779b97f4a7c15ULL);
                    int bucket = comb & mask;
                    int idx = head[bucket];
                    while(idx != -1){
                        if(valid[idx].h1 == ch1 && valid[idx].h2 == ch2){
                            present[idx] = 1;
                            break;
                        }
                        idx = next_node[idx];
                    }
                };

                check_hash(lh1, lh2);
                for(int i = L; i < (int)S.size(); ++i){
                    lh1 = lh1 * B1 + (uint8_t)S[i] - (uint8_t)S[i-L] * BL1;
                    lh2 = lh2 * B2 + (uint8_t)S[i] - (uint8_t)S[i-L] * BL2;
                    check_hash(lh1, lh2);
                }
            }));
        }

        for(auto& f : futures) f.get();

        vector<int> res;
        for(int i = 0; i < (int)valid.size(); ++i){
            bool ok = true;
            for(int k = 1; k < n; ++k){
                if(!present_arrays[k][i]){
                    ok = false;
                    break;
                }
            }
            if(ok) res.push_back(valid[i].pos);
        }
        return res;
    };

    // Binary search to find the maximum valid length
    int low = 1, high = min_len;
    int best_L = 0;
    int best_pos = -1;

    while(low <= high){
        int mid = low + (high - low) / 2;
        vector<int> res = check(mid);
        if(!res.empty()){
            best_L = mid;
            best_pos = res[0];
            low = mid + 1; // Try finding a longer common substring
        } else {
            high = mid - 1; // Try a shorter common substring
        }
    }

    if(best_L > 0 && best_pos != -1){
        cout << strings[0].substr(best_pos, best_L) << "\n";
    } else {
        cout << "\n";
    }

    return 0;
}
