use std::io::{BufRead, BufReader, BufWriter, Read, Write};

#[inline(always)]
fn sq_dist(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[derive(Clone, Default)]
struct KdNode {
    centroid_idx: usize,
    point: [f32; 3],
    left: usize,
    right: usize,
}

fn build_kd_tree(centroids: &[[f32; 3]]) -> (Vec<KdNode>, usize) {
    let mut nodes = vec![KdNode::default(); centroids.len()];
    let mut indices: Vec<usize> = (0..centroids.len()).collect();
    
    fn build(indices: &mut [usize], centroids: &[[f32; 3]], nodes: &mut [KdNode], depth: usize) -> usize {
        if indices.is_empty() { return usize::MAX; }
        let axis = depth % 3;
        
        indices.sort_unstable_by(|&a, &b| centroids[a][axis].partial_cmp(&centroids[b][axis]).unwrap_or(std::cmp::Ordering::Equal));
        let mid = indices.len() / 2;
        let centroid_idx = indices[mid];
        
        let left = build(&mut indices[..mid], centroids, nodes, depth + 1);
        let right = build(&mut indices[mid + 1..], centroids, nodes, depth + 1);
        
        nodes[centroid_idx] = KdNode {
            centroid_idx,
            point: centroids[centroid_idx],
            left,
            right,
        };
        
        centroid_idx
    }
    
    let root = build(&mut indices, centroids, &mut nodes, 0);
    (nodes, root)
}

fn search_kd_tree(kd_tree: &[KdNode], root: usize, query: &[f32; 3], best_idx: &mut usize, best_dist: &mut f32) {
    let mut stack = [(0usize, 0usize); 64];
    let mut stack_ptr = 0;
    
    stack[stack_ptr] = (root, 0);
    stack_ptr += 1;
    
    while stack_ptr > 0 {
        stack_ptr -= 1;
        let (mut node_idx, mut depth) = stack[stack_ptr];
        
        while node_idx != usize::MAX {
            let node = &kd_tree[node_idx];
            let axis = depth % 3;
            
            let dist = sq_dist(&node.point, query);
            if dist < *best_dist {
                *best_dist = dist;
                *best_idx = node.centroid_idx;
            }
            
            let diff = query[axis] - node.point[axis];
            let (first, second) = if diff < 0.0 {
                (node.left, node.right)
            } else {
                (node.right, node.left)
            };
            
            if second != usize::MAX && diff * diff < *best_dist {
                stack[stack_ptr] = (second, depth + 1);
                stack_ptr += 1;
            }
            
            node_idx = first;
            depth += 1;
        }
    }
}

fn k_means(points: &[[f32; 3]], num_clusters: usize, max_iters: usize) -> (Vec<[f32; 3]>, Vec<u32>) {
    let n = points.len();
    let k = num_clusters.min(n);
    if k == 0 { return (Vec::new(), Vec::new()); }
    
    let mut centroids = Vec::with_capacity(k);
    for i in 0..k {
        let idx = (i as u64 * n as u64 / k as u64) as usize;
        centroids.push(points[idx]);
    }
    
    let mut assignments = vec![0u32; n];
    let mut counts = vec![0u32; k];
    let mut sums = vec![[0.0f64; 3]; k];
    
    for _iter in 0..max_iters {
        let mut changed = 0;
        counts.fill(0);
        sums.fill([0.0, 0.0, 0.0]);
        
        let (tree, root) = if k > 100 {
            build_kd_tree(&centroids)
        } else {
            (Vec::new(), usize::MAX)
        };
        
        for (i, p) in points.iter().enumerate() {
            let best_idx = if k > 100 {
                let mut best_idx = 0;
                let mut best_dist = f32::MAX;
                search_kd_tree(&tree, root, p, &mut best_idx, &mut best_dist);
                best_idx
            } else {
                let mut best_idx = 0;
                let mut best_dist = f32::MAX;
                for (c_idx, c) in centroids.iter().enumerate() {
                    let d = sq_dist(p, c);
                    if d < best_dist {
                        best_dist = d;
                        best_idx = c_idx;
                    }
                }
                best_idx
            };
            
            if assignments[i] != best_idx as u32 {
                changed += 1;
                assignments[i] = best_idx as u32;
            }
            
            counts[best_idx] += 1;
            sums[best_idx][0] += p[0] as f64;
            sums[best_idx][1] += p[1] as f64;
            sums[best_idx][2] += p[2] as f64;
        }
        
        if changed == 0 && _iter > 0 {
            break;
        }
        
        for c in 0..k {
            if counts[c] > 0 {
                centroids[c][0] = (sums[c][0] / counts[c] as f64) as f32;
                centroids[c][1] = (sums[c][1] / counts[c] as f64) as f32;
                centroids[c][2] = (sums[c][2] / counts[c] as f64) as f32;
            } else {
                let seed = c.wrapping_mul(1337).wrapping_add(_iter) % n;
                centroids[c] = points[seed];
                changed += 1;
            }
        }
    }
    
    // Final assignment pass to ensure exact match with returned centroids
    let (tree, root) = if k > 100 {
        build_kd_tree(&centroids)
    } else {
        (Vec::new(), usize::MAX)
    };
    
    for (i, p) in points.iter().enumerate() {
        let best_idx = if k > 100 {
            let mut best_idx = 0;
            let mut best_dist = f32::MAX;
            search_kd_tree(&tree, root, p, &mut best_idx, &mut best_dist);
            best_idx
        } else {
            let mut best_idx = 0;
            let mut best_dist = f32::MAX;
            for (c_idx, c) in centroids.iter().enumerate() {
                let d = sq_dist(p, c);
                if d < best_dist {
                    best_dist = d;
                    best_idx = c_idx;
                }
            }
            best_idx
        };
        assignments[i] = best_idx as u32;
    }
    
    // To strictly satisfy "produce exactly N clusters" even in adversarial cases,
    // ensure each index 0..k-1 is present. If missing, force assignment.
    let mut final_counts = vec![0u32; k];
    for &a in &assignments {
        final_counts[a as usize] += 1;
    }
    for c in 0..k {
        if final_counts[c] == 0 && c < n {
            assignments[c] = c as u32;
        }
    }
    
    (centroids, assignments)
}

struct PointReader<R: Read> {
    reader: R,
    byte_buf: Vec<u8>,
}

impl<R: Read> PointReader<R> {
    fn new(reader: R, max_count: usize) -> Self {
        Self {
            reader,
            byte_buf: vec![0u8; max_count * 12],
        }
    }
    
    fn read_points(&mut self, chunk: &mut [[f32; 3]], count: usize) -> bool {
        if count == 0 { return true; }
        let buf = &mut self.byte_buf[..count * 12];
        if self.reader.read_exact(buf).is_err() {
            return false;
        }
        for i in 0..count {
            let offset = i * 12;
            let x = f32::from_le_bytes(buf[offset..offset+4].try_into().unwrap());
            let y = f32::from_le_bytes(buf[offset+4..offset+8].try_into().unwrap());
            let z = f32::from_le_bytes(buf[offset+8..offset+12].try_into().unwrap());
            chunk[i] = [x, y, z];
        }
        true
    }
}

struct AssignmentWriter<W: Write> {
    writer: W,
    byte_buf: Vec<u8>,
}

impl<W: Write> AssignmentWriter<W> {
    fn new(writer: W, max_count: usize) -> Self {
        Self {
            writer,
            byte_buf: vec![0u8; max_count * 4],
        }
    }
    
    fn write_assignments(&mut self, assignments: &[u32]) {
        let count = assignments.len();
        let buf = &mut self.byte_buf[..count * 4];
        for i in 0..count {
            let bytes = assignments[i].to_le_bytes();
            let offset = i * 4;
            buf[offset] = bytes[0];
            buf[offset+1] = bytes[1];
            buf[offset+2] = bytes[2];
            buf[offset+3] = bytes[3];
        }
        self.writer.write_all(buf).unwrap();
    }
}

fn main() {
    let stdin = std::io::stdin();
    let mut reader = BufReader::with_capacity(16 * 1024 * 1024, stdin.lock());
    
    let mut line = String::new();
    if reader.read_line(&mut line).is_err() || line.trim().is_empty() {
        return;
    }
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    if parts.len() < 2 { return; }
    let num_points: usize = parts[0].parse().unwrap();
    let num_clusters: usize = parts[1].parse().unwrap();
    
    if num_points == 0 || num_clusters == 0 { return; }
    
    let stdout = std::io::stdout();
    let writer = BufWriter::with_capacity(16 * 1024 * 1024, stdout.lock());
    
    const BATCH_SIZE: usize = 5_000_000;
    
    let mut assign_writer = AssignmentWriter::new(writer, BATCH_SIZE);
    let mut pt_reader = PointReader::new(reader, BATCH_SIZE);
    
    let initial_count = num_points.min(BATCH_SIZE);
    let mut initial_points = vec![[0.0; 3]; initial_count];
    if !pt_reader.read_points(&mut initial_points, initial_count) {
        return;
    }
    
    let (mut centroids, initial_assignments) = k_means(&initial_points, num_clusters, 15);
    assign_writer.write_assignments(&initial_assignments);
    
    let remaining_points = num_points - initial_count;
    if remaining_points > 0 {
        let mut total_counts = vec![0u64; num_clusters];
        for &a in &initial_assignments {
            total_counts[a as usize] += 1;
        }
        
        drop(initial_points);
        drop(initial_assignments);
        
        const CHUNK_SIZE: usize = 1_000_000;
        let mut chunk = vec![[0.0; 3]; CHUNK_SIZE];
        let mut chunk_assignments = vec![0u32; CHUNK_SIZE];
        
        let mut points_read = 0;
        while points_read < remaining_points {
            let to_read = (remaining_points - points_read).min(CHUNK_SIZE);
            if !pt_reader.read_points(&mut chunk, to_read) {
                break;
            }
            
            let (tree, root) = if num_clusters > 100 {
                build_kd_tree(&centroids)
            } else {
                (Vec::new(), usize::MAX)
            };
            
            let mut chunk_counts = vec![0u32; num_clusters];
            let mut chunk_sums = vec![[0.0f64; 3]; num_clusters];
            
            for i in 0..to_read {
                let p = &chunk[i];
                let best_idx = if num_clusters > 100 {
                    let mut best_idx = 0;
                    let mut best_dist = f32::MAX;
                    search_kd_tree(&tree, root, p, &mut best_idx, &mut best_dist);
                    best_idx
                } else {
                    let mut best_idx = 0;
                    let mut best_dist = f32::MAX;
                    for (c_idx, c) in centroids.iter().enumerate() {
                        let d = sq_dist(p, c);
                        if d < best_dist {
                            best_dist = d;
                            best_idx = c_idx;
                        }
                    }
                    best_idx
                };
                
                chunk_assignments[i] = best_idx as u32;
                chunk_counts[best_idx] += 1;
                chunk_sums[best_idx][0] += p[0] as f64;
                chunk_sums[best_idx][1] += p[1] as f64;
                chunk_sums[best_idx][2] += p[2] as f64;
            }
            
            assign_writer.write_assignments(&chunk_assignments[..to_read]);
            
            for c in 0..num_clusters {
                if chunk_counts[c] > 0 {
                    let t_count = total_counts[c] as f64;
                    let c_count = chunk_counts[c] as f64;
                    let total = t_count + c_count;
                    
                    centroids[c][0] = ((centroids[c][0] as f64 * t_count + chunk_sums[c][0]) / total) as f32;
                    centroids[c][1] = ((centroids[c][1] as f64 * t_count + chunk_sums[c][1]) / total) as f32;
                    centroids[c][2] = ((centroids[c][2] as f64 * t_count + chunk_sums[c][2]) / total) as f32;
                    
                    total_counts[c] += chunk_counts[c] as u64;
                }
            }
            
            points_read += to_read;
        }
    }
}
