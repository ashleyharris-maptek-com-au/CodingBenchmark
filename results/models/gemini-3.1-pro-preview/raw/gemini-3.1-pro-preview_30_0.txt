import sys

def solve():
    def get_ints():
        for line in sys.stdin:
            for token in line.split():
                yield int(token)
    
    tokens = get_ints()
    try:
        num_vars = next(tokens)
        num_clauses = next(tokens)
    except StopIteration:
        return

    try:
        clauses = memoryview(bytearray(num_clauses * 3 * 4)).cast('i')
    except MemoryError:
        print("UNSAT")
        return
        
    idx = 0
    for _ in range(num_clauses):
        clauses[idx] = next(tokens)
        clauses[idx+1] = next(tokens)
        clauses[idx+2] = next(tokens)
        idx += 3

    watch_head = memoryview(bytearray((2 * num_vars + 1) * 4)).cast('i')
    for i in range(2 * num_vars + 1):
        watch_head[i] = -1
        
    watch_next = memoryview(bytearray(num_clauses * 2 * 4)).cast('i')
    watch_clause = memoryview(bytearray(num_clauses * 2 * 4)).cast('i')
    
    watch_alloc = 0
    
    for i in range(num_clauses):
        l0 = clauses[3*i]
        idx0 = l0 if l0 > 0 else num_vars - l0
        watch_clause[watch_alloc] = i
        watch_next[watch_alloc] = watch_head[idx0]
        watch_head[idx0] = watch_alloc
        watch_alloc += 1
        
        l1 = clauses[3*i + 1]
        idx1 = l1 if l1 > 0 else num_vars - l1
        watch_clause[watch_alloc] = i
        watch_next[watch_alloc] = watch_head[idx1]
        watch_head[idx1] = watch_alloc
        watch_alloc += 1

    assignment = bytearray(num_vars + 1)
    
    trail = memoryview(bytearray((num_vars + 1) * 4)).cast('i')
    trail_len = 0
    
    trail_lim = memoryview(bytearray((num_vars + 1) * 4)).cast('i')
    trail_lim_len = 0
    
    q = memoryview(bytearray((num_vars + 1) * 4)).cast('i')
    q_len = 0
    q_head = 0

    def add_assignment(lit):
        nonlocal trail_len, q_len
        var = abs(lit)
        val = 1 if lit > 0 else 2
        
        curr = assignment[var]
        if curr == val:
            return True
        if curr != 0:
            return False
            
        assignment[var] = val
        trail[trail_len] = lit
        trail_len += 1
        
        q[q_len] = lit
        q_len += 1
        return True

    def propagate():
        nonlocal q_head
        while q_head < q_len:
            p = q[q_head]
            q_head += 1
            
            false_lit = -p
            idx_false = false_lit if false_lit > 0 else num_vars - false_lit
            
            curr_watch = watch_head[idx_false]
            watch_head[idx_false] = -1
            
            conflict = False
            
            while curr_watch != -1:
                c_idx = watch_clause[curr_watch]
                next_w = watch_next[curr_watch]
                
                if conflict:
                    watch_next[curr_watch] = watch_head[idx_false]
                    watch_head[idx_false] = curr_watch
                    curr_watch = next_w
                    continue

                l0 = clauses[3*c_idx]
                l1 = clauses[3*c_idx + 1]
                l2 = clauses[3*c_idx + 2]
                
                if l0 == false_lit:
                    l0, l1 = l1, l0
                    clauses[3*c_idx] = l0
                    clauses[3*c_idx + 1] = l1
                    
                v0 = abs(l0)
                if assignment[v0] == (1 if l0 > 0 else 2):
                    watch_next[curr_watch] = watch_head[idx_false]
                    watch_head[idx_false] = curr_watch
                    curr_watch = next_w
                    continue
                    
                v2 = abs(l2)
                if assignment[v2] != (2 if l2 > 0 else 1):
                    clauses[3*c_idx + 1] = l2
                    clauses[3*c_idx + 2] = l1
                    
                    idx2 = l2 if l2 > 0 else num_vars - l2
                    watch_next[curr_watch] = watch_head[idx2]
                    watch_head[idx2] = curr_watch
                    curr_watch = next_w
                    continue
                    
                watch_next[curr_watch] = watch_head[idx_false]
                watch_head[idx_false] = curr_watch
                
                if assignment[v0] == 0:
                    add_assignment(l0)
                else:
                    conflict = True
                
                curr_watch = next_w
                
            if conflict:
                return False
        return True

    decision_var = 1
    is_sat = False
    
    while True:
        if not propagate():
            can_backtrack = False
            while trail_lim_len > 0:
                trail_lim_len -= 1
                limit = trail_lim[trail_lim_len]
                decision_lit = trail[limit]
                
                while trail_len > limit:
                    trail_len -= 1
                    lit = trail[trail_len]
                    assignment[abs(lit)] = 0
                    
                q_len = 0
                q_head = 0
                
                flipped_lit = -decision_lit
                add_assignment(flipped_lit)
                can_backtrack = True
                break
                
            if not can_backtrack:
                is_sat = False
                break
        else:
            while decision_var <= num_vars and assignment[decision_var] != 0:
                decision_var += 1
                
            if decision_var > num_vars:
                is_sat = True
                break
                
            trail_lim[trail_lim_len] = trail_len
            trail_lim_len += 1
            
            add_assignment(decision_var)
            
    if is_sat:
        print("SAT")
        chunk = []
        for i in range(1, num_vars + 1):
            chunk.append("1" if assignment[i] == 1 else "0")
            if len(chunk) >= 50000:
                sys.stdout.write(" ".join(chunk) + " ")
                chunk.clear()
        if chunk:
            sys.stdout.write(" ".join(chunk))
        sys.stdout.write("\n")
    else:
        print("UNSAT")

if __name__ == '__main__':
    solve()
