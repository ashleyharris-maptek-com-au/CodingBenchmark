use std::io::{self, Read};
use std::collections::{VecDeque, HashMap};
use std::time::Instant;
use std::thread;

struct QuadTree {
    nodes: Vec<QuadNode>,
}

#[derive(Clone, Copy)]
struct QuadNode {
    mass: f32,
    cm_x: f32,
    cm_y: f32,
    x: f32,
    y: f32,
    half_w: f32,
    first_child: u32,
    node_id: u32,
}

impl QuadNode {
    fn new_empty(x: f32, y: f32, half_w: f32) -> Self {
        Self {
            mass: 0.0,
            cm_x: 0.0,
            cm_y: 0.0,
            x,
            y,
            half_w,
            first_child: !0,
            node_id: !0,
        }
    }
}

fn get_quad(px: f32, py: f32, cx: f32, cy: f32) -> u32 {
    let mut q = 0;
    if px >= cx { q |= 1; }
    if py >= cy { q |= 2; }
    q
}

impl QuadTree {
    fn new(min_x: f32, min_y: f32, max_x: f32, max_y: f32, n: usize) -> Self {
        let center_x = (min_x + max_x) / 2.0;
        let center_y = (min_y + max_y) / 2.0;
        let mut half_w = (max_x - min_x).max(max_y - min_y) / 2.0;
        if half_w < 1e-4 { half_w = 1.0; }
        
        let mut nodes = Vec::with_capacity(n * 4 + 16);
        nodes.push(QuadNode::new_empty(center_x, center_y, half_w));
        Self { nodes }
    }

    fn insert(&mut self, pt_idx: usize, pos_x: &[f32], pos_y: &[f32]) {
        let p_x = pos_x[pt_idx];
        let p_y = pos_y[pt_idx];
        let mut curr = 0;
        
        loop {
            {
                let node = &mut self.nodes[curr as usize];
                let new_mass = node.mass + 1.0;
                node.cm_x = (node.cm_x * node.mass + p_x) / new_mass;
                node.cm_y = (node.cm_y * node.mass + p_y) / new_mass;
                node.mass = new_mass;
            }

            let first_child = self.nodes[curr as usize].first_child;
            let node_id = self.nodes[curr as usize].node_id;
            let node_x = self.nodes[curr as usize].x;
            let node_y = self.nodes[curr as usize].y;
            let node_hw = self.nodes[curr as usize].half_w;

            if first_child != !0 {
                let quad = get_quad(p_x, p_y, node_x, node_y);
                curr = first_child + quad;
            } else if node_id != !0 {
                if node_hw < 1e-4 {
                    break;
                }
                
                let exist_idx = node_id as usize;
                let ex_x = pos_x[exist_idx];
                let ex_y = pos_y[exist_idx];
                
                self.nodes[curr as usize].node_id = !0;
                let new_first_child = self.nodes.len() as u32;
                self.nodes[curr as usize].first_child = new_first_child;
                
                let nh = node_hw / 2.0;
                
                self.nodes.push(QuadNode::new_empty(node_x - nh, node_y - nh, nh));
                self.nodes.push(QuadNode::new_empty(node_x + nh, node_y - nh, nh));
                self.nodes.push(QuadNode::new_empty(node_x - nh, node_y + nh, nh));
                self.nodes.push(QuadNode::new_empty(node_x + nh, node_y + nh, nh));
                
                let ex_quad = get_quad(ex_x, ex_y, node_x, node_y);
                self.insert_existing(new_first_child + ex_quad, exist_idx, ex_x, ex_y, pos_x, pos_y);
                
                let p_quad = get_quad(p_x, p_y, node_x, node_y);
                curr = new_first_child + p_quad;
            } else {
                self.nodes[curr as usize].node_id = pt_idx as u32;
                break;
            }
        }
    }

    fn insert_existing(&mut self, mut curr: u32, pt_idx: usize, p_x: f32, p_y: f32, pos_x: &[f32], pos_y: &[f32]) {
        loop {
            {
                let node = &mut self.nodes[curr as usize];
                let new_mass = node.mass + 1.0;
                node.cm_x = (node.cm_x * node.mass + p_x) / new_mass;
                node.cm_y = (node.cm_y * node.mass + p_y) / new_mass;
                node.mass = new_mass;
            }
            
            let first_child = self.nodes[curr as usize].first_child;
            let node_id = self.nodes[curr as usize].node_id;
            let node_x = self.nodes[curr as usize].x;
            let node_y = self.nodes[curr as usize].y;
            let node_hw = self.nodes[curr as usize].half_w;
            
            if first_child != !0 {
                let quad = get_quad(p_x, p_y, node_x, node_y);
                curr = first_child + quad;
            } else if node_id != !0 {
                if node_hw < 1e-4 { break; }
                let exist_idx = node_id as usize;
                let ex_x = pos_x[exist_idx];
                let ex_y = pos_y[exist_idx];
                
                self.nodes[curr as usize].node_id = !0;
                let new_first_child = self.nodes.len() as u32;
                self.nodes[curr as usize].first_child = new_first_child;
                
                let nh = node_hw / 2.0;
                
                self.nodes.push(QuadNode::new_empty(node_x - nh, node_y - nh, nh));
                self.nodes.push(QuadNode::new_empty(node_x + nh, node_y - nh, nh));
                self.nodes.push(QuadNode::new_empty(node_x - nh, node_y + nh, nh));
                self.nodes.push(QuadNode::new_empty(node_x + nh, node_y + nh, nh));
                
                let ex_quad = get_quad(ex_x, ex_y, node_x, node_y);
                self.insert_existing(new_first_child + ex_quad, exist_idx, ex_x, ex_y, pos_x, pos_y);
                
                let p_quad = get_quad(p_x, p_y, node_x, node_y);
                curr = new_first_child + p_quad;
            } else {
                self.nodes[curr as usize].node_id = pt_idx as u32;
                break;
            }
        }
    }
}

fn compute_repulsion(tree: &QuadTree, p_idx: usize, p_x: f32, p_y: f32, k2: f32, theta: f32) -> (f32, f32) {
    let mut stack = Vec::with_capacity(128);
    stack.push(0u32);
    
    let mut fx = 0.0;
    let mut fy = 0.0;
    
    while let Some(curr) = stack.pop() {
        let node = &tree.nodes[curr as usize];
        if node.mass == 0.0 { continue; }
        
        let width = node.half_w * 2.0;
        
        if node.first_child == !0 {
            if node.node_id == p_idx as u32 {
                continue;
            }
        }
        
        let mut dx = node.cm_x - p_x;
        let mut dy = node.cm_y - p_y;
        
        if dx.abs() < 1e-5 && dy.abs() < 1e-5 {
            dx = if p_idx % 2 == 0 { 1e-4 } else { -1e-4 };
            dy = if p_idx % 3 == 0 { 1e-4 } else { -1e-4 };
        }
        
        let mut dist_sq = dx * dx + dy * dy;
        if dist_sq < 1e-4 { dist_sq = 1e-4; }
        let dist = dist_sq.sqrt();
        
        if node.first_child == !0 || width / dist < theta {
            let f = (k2 / dist) * node.mass;
            fx -= f * (dx / dist);
            fy -= f * (dy / dist);
        } else {
            stack.push(node.first_child);
            stack.push(node.first_child + 1);
            stack.push(node.first_child + 2);
            stack.push(node.first_child + 3);
        }
    }
    (fx, fy)
}

fn xorshift(seed: &mut u64) -> f32 {
    *seed ^= *seed << 13;
    *seed ^= *seed >> 7;
    *seed ^= *seed << 17;
    ((*seed as u32) as f32) / (std::u32::MAX as f32)
}

fn main() {
    let mut input = Vec::new();
    if io::stdin().read_to_end(&mut input).is_err() {
        return;
    }
    let input_str = String::from_utf8_lossy(&input);
    let mut tokens = input_str.split_whitespace();
    
    let n: usize = match tokens.next() {
        Some(s) => match s.parse() {
            Ok(val) => val,
            Err(_) => return,
        },
        None => return,
    };
    
    let m: usize = match tokens.next() {
        Some(s) => match s.parse() {
            Ok(val) => val,
            Err(_) => return,
        },
        None => return,
    };
    
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = match tokens.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        let v: usize = match tokens.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        edges.push((u, v));
    }
    
    if n == 0 { return; }
    if n == 1 {
        println!("0.0 0.0");
        return;
    }

    let mut pos_x = vec![0.0; n];
    let mut pos_y = vec![0.0; n];
    
    let mut adj = vec![vec![]; n];
    for &(u, v) in &edges {
        if u < n && v < n {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    
    let mut visited = vec![false; n];
    let mut q = VecDeque::new();
    let mut rng_seed = 123456789u64;

    for i in 0..n {
        if !visited[i] {
            let cx = (xorshift(&mut rng_seed) - 0.5) * (n as f32).sqrt() * 2.0;
            let cy = (xorshift(&mut rng_seed) - 0.5) * (n as f32).sqrt() * 2.0;
            
            q.push_back((i, 0));
            visited[i] = true;
            
            let mut level_counts = HashMap::new();
            let mut node_levels = Vec::new();
            
            while let Some((u, d)) = q.pop_front() {
                node_levels.push((u, d));
                *level_counts.entry(d).or_insert(0) += 1;
                
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        q.push_back((v, d + 1));
                    }
                }
            }
            
            let mut level_idx = HashMap::new();
            for (u, d) in node_levels {
                let idx = level_idx.entry(d).or_insert(0);
                let count = level_counts[&d];
                
                if d == 0 {
                    pos_x[u] = cx;
                    pos_y[u] = cy;
                } else {
                    let radius = (d as f32).max(1.0) * 2.0 + (count as f32).sqrt() * 1.0;
                    let angle = 2.0 * std::f32::consts::PI * (*idx as f32) / (count as f32) + (d as f32 * 0.1);
                    pos_x[u] = cx + angle.cos() * radius + (xorshift(&mut rng_seed) - 0.5) * 0.1;
                    pos_y[u] = cy + angle.sin() * radius + (xorshift(&mut rng_seed) - 0.5) * 0.1;
                }
                *idx += 1;
            }
        }
    }
    
    let start_time = Instant::now();
    let time_limit = 28.0;
    let max_iters = 1500;
    let mut iter = 0;
    
    let mut disp_x = vec![0.0; n];
    let mut disp_y = vec![0.0; n];
    
    let num_threads = std::thread::available_parallelism().map(|p| p.get()).unwrap_or(1);
    let chunk_size = (n + num_threads - 1) / num_threads;

    while iter < max_iters {
        let elapsed = start_time.elapsed().as_secs_f32();
        if elapsed > time_limit {
            break;
        }
        
        let progress = (elapsed / time_limit).max(iter as f32 / max_iters as f32).min(1.0);
        let t0 = 5.0 + (n as f32).sqrt() * 0.05;
        let temp = t0 * (1.0 - progress) + 0.05;
        
        let mut min_x = pos_x[0]; let mut max_x = pos_x[0];
        let mut min_y = pos_y[0]; let mut max_y = pos_y[0];
        for i in 1..n {
            if pos_x[i] < min_x { min_x = pos_x[i]; }
            if pos_x[i] > max_x { max_x = pos_x[i]; }
            if pos_y[i] < min_y { min_y = pos_y[i]; }
            if pos_y[i] > max_y { max_y = pos_y[i]; }
        }
        if max_x - min_x < 1e-4 { max_x += 1.0; min_x -= 1.0; }
        if max_y - min_y < 1e-4 { max_y += 1.0; min_y -= 1.0; }
        
        let mut tree = QuadTree::new(min_x, min_y, max_x, max_y, n);
        for i in 0..n {
            tree.insert(i, &pos_x, &pos_y);
        }
        
        thread::scope(|s| {
            let mut dx_chunks = disp_x.chunks_mut(chunk_size);
            let mut dy_chunks = disp_y.chunks_mut(chunk_size);
            
            for (i, (dx_chunk, dy_chunk)) in dx_chunks.zip(dy_chunks).enumerate() {
                let start_idx = i * chunk_size;
                let tree_ref = &tree;
                let px_ref = &pos_x;
                let py_ref = &pos_y;
                
                s.spawn(move || {
                    for j in 0..dx_chunk.len() {
                        let node_idx = start_idx + j;
                        let (rx, ry) = compute_repulsion(tree_ref, node_idx, px_ref[node_idx], py_ref[node_idx], 1.0, 0.8);
                        dx_chunk[j] = rx;
                        dy_chunk[j] = ry;
                    }
                });
            }
        });
        
        for &(u, v) in &edges {
            if u >= n || v >= n { continue; }
            let dx = pos_x[v] - pos_x[u];
            let dy = pos_y[v] - pos_y[u];
            let mut dist_sq = dx * dx + dy * dy;
            if dist_sq < 1e-4 { dist_sq = 1e-4; }
            let dist = dist_sq.sqrt();
            
            let f = dist_sq; 
            let fx = f * (dx / dist);
            let fy = f * (dy / dist);
            
            disp_x[u] += fx;
            disp_y[u] += fy;
            disp_x[v] -= fx;
            disp_y[v] -= fy;
        }
        
        for i in 0..n {
            let mut dx = disp_x[i];
            let mut dy = disp_y[i];
            
            dx -= pos_x[i] * 0.01;
            dy -= pos_y[i] * 0.01;
            
            let disp_len = (dx * dx + dy * dy).sqrt();
            if disp_len > 0.0 {
                let scale = disp_len.min(temp) / disp_len;
                pos_x[i] += dx * scale;
                pos_y[i] += dy * scale;
            }
        }
        
        iter += 1;
    }
    
    for i in 0..n {
        println!("{:.6} {:.6}", pos_x[i], pos_y[i]);
    }
}
