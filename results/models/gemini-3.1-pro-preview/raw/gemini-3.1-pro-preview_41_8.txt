; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_75 = OpConstant %float 0.75
%float_0_5 = OpConstant %float 0.5
%float_0_25 = OpConstant %float 0.25
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_0_2 = OpConstant %float 0.2
%bool = OpTypeBool

%main = OpFunction %void None %func
%label = OpLabel

%pos_val = OpLoad %v3float %worldPos
%norm_val = OpLoad %v3float %normal
%color_val = OpLoad %v3float %color

%N = OpExtInst %v3float %1 Normalize %norm_val

%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos_v4 = OpLoad %v4float %lightPos_ptr
%lightPos_v3 = OpVectorShuffle %v3float %lightPos_v4 %lightPos_v4 0 1 2

%lightDir_unorm = OpFSub %v3float %lightPos_v3 %pos_val
%L = OpExtInst %v3float %1 Normalize %lightDir_unorm

%NdotL = OpDot %float %N %L
%d = OpExtInst %float %1 FMax %NdotL %float_0

%cond_75 = OpFOrdGreaterThanEqual %bool %d %float_0_75
%cond_50 = OpFOrdGreaterThanEqual %bool %d %float_0_5
%cond_25 = OpFOrdGreaterThanEqual %bool %d %float_0_25

%band_25 = OpSelect %float %cond_25 %float_0_4 %float_0_2
%band_50 = OpSelect %float %cond_50 %float_0_7 %band_25
%band = OpSelect %float %cond_75 %float_1 %band_50

%color_scaled = OpVectorTimesScalar %v3float %color_val %band

%out_val = OpCompositeConstruct %v4float %color_scaled %float_1
OpStore %outColor %out_val

OpReturn
OpFunctionEnd
