using System;
using System.Collections.Generic;
using System.IO;

public class ChinesePostman
{
    class Scanner
    {
        Stream stream;
        byte[] buffer = new byte[1024 * 64];
        int head, tail;

        public Scanner(Stream s)
        {
            stream = s;
        }

        public int ReadInt()
        {
            int c = Read();
            while (c <= 32)
            {
                if (c == -1) return -1;
                c = Read();
            }
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int res = 0;
            while (c > 32)
            {
                if (c >= '0' && c <= '9')
                    res = res * 10 + (c - '0');
                c = Read();
            }
            return res * sign;
        }

        int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }

    class Edge
    {
        public int to;
        public int id;
        public int weight;
        public Edge(int to, int id, int weight)
        {
            this.to = to;
            this.id = id;
            this.weight = weight;
        }
    }

    static List<Edge>[] adj;
    static List<bool> edgeUsed;
    static int edgeCount = 0;

    static void AddDuplicateEdge(int u, int v, int w)
    {
        adj[u].Add(new Edge(v, edgeCount, w));
        adj[v].Add(new Edge(u, edgeCount, w));
        edgeUsed.Add(false);
        edgeCount++;
    }

    public static void Main()
    {
        Scanner scanner = new Scanner(Console.OpenStandardInput());
        int N = scanner.ReadInt();
        if (N == -1) return;
        int M = scanner.ReadInt();

        adj = new List<Edge>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<Edge>();
        edgeUsed = new List<bool>(M + 10000);

        for (int i = 0; i < M; i++)
        {
            int u = scanner.ReadInt();
            int v = scanner.ReadInt();
            int w = scanner.ReadInt();
            adj[u].Add(new Edge(v, edgeCount, w));
            adj[v].Add(new Edge(u, edgeCount, w));
            edgeUsed.Add(false);
            edgeCount++;
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (adj[i].Count % 2 != 0)
            {
                oddNodes.Add(i);
            }
        }

        int K = oddNodes.Count;

        if (K > 0)
        {
            if (K <= 20)
            {
                SolveExactDP(N, oddNodes);
            }
            else if (K <= 400)
            {
                SolveGreedy2Opt(N, oddNodes);
            }
            else
            {
                SolveMSDP(N, oddNodes);
            }
        }

        List<int> circuit = new List<int>();
        Stack<int> currPath = new Stack<int>();
        int[] edgeIndex = new int[N];

        currPath.Push(0);
        while (currPath.Count > 0)
        {
            int u = currPath.Peek();
            bool found = false;
            while (edgeIndex[u] < adj[u].Count)
            {
                Edge e = adj[u][edgeIndex[u]];
                edgeIndex[u]++;
                if (!edgeUsed[e.id])
                {
                    edgeUsed[e.id] = true;
                    currPath.Push(e.to);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                circuit.Add(currPath.Pop());
            }
        }

        circuit.Reverse();

        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536))
        {
            writer.Write(circuit.Count);
            for (int i = 0; i < circuit.Count; i++)
            {
                writer.Write(" ");
                writer.Write(circuit[i]);
            }
            writer.WriteLine();
        }
    }

    static void SolveExactDP(int N, List<int> oddNodes)
    {
        int K = oddNodes.Count;
        long[][] distMat = new long[K][];
        int[][] parentMat = new int[K][];

        for (int i = 0; i < K; i++)
        {
            distMat[i] = new long[K];
            parentMat[i] = new int[N];
            long[] d = new long[N];
            Array.Fill(d, long.MaxValue);
            d[oddNodes[i]] = 0;
            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            pq.Enqueue(oddNodes[i], 0);

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int curr, out long curD);
                if (curD > d[curr]) continue;

                foreach (var e in adj[curr])
                {
                    long nd = curD + e.weight;
                    if (nd < d[e.to])
                    {
                        d[e.to] = nd;
                        parentMat[i][e.to] = curr;
                        pq.Enqueue(e.to, nd);
                    }
                }
            }

            for (int j = 0; j < K; j++)
            {
                distMat[i][j] = d[oddNodes[j]];
            }
        }

        long[] dp = new long[1 << K];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;
        int[] bestU = new int[1 << K];
        int[] bestV = new int[1 << K];
        int[] prevMask = new int[1 << K];

        for (int mask = 0; mask < (1 << K); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int u = -1;
            for (int i = 0; i < K; i++)
            {
                if ((mask & (1 << i)) == 0) { u = i; break; }
            }
            if (u == -1) continue;

            for (int v = u + 1; v < K; v++)
            {
                if ((mask & (1 << v)) == 0)
                {
                    int nmask = mask | (1 << u) | (1 << v);
                    long ndist = dp[mask] + distMat[u][v];
                    if (ndist < dp[nmask])
                    {
                        dp[nmask] = ndist;
                        prevMask[nmask] = mask;
                        bestU[nmask] = u;
                        bestV[nmask] = v;
                    }
                }
            }
        }

        int currMask = (1 << K) - 1;
        while (currMask > 0)
        {
            int u = bestU[currMask];
            int v = bestV[currMask];

            int currNode = oddNodes[v];
            while (currNode != oddNodes[u])
            {
                int p = parentMat[u][currNode];
                AddDuplicateEdge(p, currNode, 0);
                currNode = p;
            }

            currMask = prevMask[currMask];
        }
    }

    static void SolveGreedy2Opt(int N, List<int> oddNodes)
    {
        int K = oddNodes.Count;
        long[][] distMat = new long[K][];
        int[][] parentMat = new int[K][];

        for (int i = 0; i < K; i++)
        {
            distMat[i] = new long[K];
            parentMat[i] = new int[N];
            long[] d = new long[N];
            Array.Fill(d, long.MaxValue);
            d[oddNodes[i]] = 0;
            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            pq.Enqueue(oddNodes[i], 0);

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int curr, out long curD);
                if (curD > d[curr]) continue;

                foreach (var e in adj[curr])
                {
                    long nd = curD + e.weight;
                    if (nd < d[e.to])
                    {
                        d[e.to] = nd;
                        parentMat[i][e.to] = curr;
                        pq.Enqueue(e.to, nd);
                    }
                }
            }

            for (int j = 0; j < K; j++)
            {
                distMat[i][j] = d[oddNodes[j]];
            }
        }

        List<Tuple<long, int, int>> allPairs = new List<Tuple<long, int, int>>();
        for (int i = 0; i < K; i++)
        {
            for (int j = i + 1; j < K; j++)
            {
                allPairs.Add(new Tuple<long, int, int>(distMat[i][j], i, j));
            }
        }
        allPairs.Sort();

        bool[] paired = new bool[K];
        List<Tuple<int, int>> pairs = new List<Tuple<int, int>>();

        foreach (var p in allPairs)
        {
            int u = p.Item2;
            int v = p.Item3;
            if (!paired[u] && !paired[v])
            {
                paired[u] = true;
                paired[v] = true;
                pairs.Add(new Tuple<int, int>(u, v));
            }
        }

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < pairs.Count; i++)
            {
                for (int j = i + 1; j < pairs.Count; j++)
                {
                    int A = pairs[i].Item1, B = pairs[i].Item2;
                    int C = pairs[j].Item1, D = pairs[j].Item2;

                    long currentDist = distMat[A][B] + distMat[C][D];
                    long d1 = distMat[A][C] + distMat[B][D];
                    long d2 = distMat[A][D] + distMat[B][C];

                    if (d1 < currentDist && d1 <= d2)
                    {
                        pairs[i] = new Tuple<int, int>(A, C);
                        pairs[j] = new Tuple<int, int>(B, D);
                        improved = true;
                    }
                    else if (d2 < currentDist && d2 < d1)
                    {
                        pairs[i] = new Tuple<int, int>(A, D);
                        pairs[j] = new Tuple<int, int>(B, C);
                        improved = true;
                    }
                }
            }
        }

        foreach (var p in pairs)
        {
            int u = p.Item1;
            int v = p.Item2;
            int currNode = oddNodes[v];
            while (currNode != oddNodes[u])
            {
                int pr = parentMat[u][currNode];
                AddDuplicateEdge(pr, currNode, 0);
                currNode = pr;
            }
        }
    }

    static void SolveMSDP(int N, List<int> oddNodes)
    {
        long[] dist = new long[N];
        int[] owner = new int[N];
        int[] parent = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(owner, -1);
        Array.Fill(parent, -1);

        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        bool[] isPaired = new bool[N];

        foreach (int u in oddNodes)
        {
            dist[u] = 0;
            owner[u] = u;
            pq.Enqueue(u, 0);
        }

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int curr, out long curD);

            if (curD > dist[curr]) continue;

            int O1 = owner[curr];
            if (isPaired[O1]) continue;

            foreach (var e in adj[curr])
            {
                int O2 = owner[e.to];

                if (O2 != -1 && O2 != O1 && !isPaired[O2])
                {
                    isPaired[O1] = true;
                    isPaired[O2] = true;

                    int temp = curr;
                    while (temp != O1)
                    {
                        int p = parent[temp];
                        AddDuplicateEdge(p, temp, 0);
                        temp = p;
                    }

                    temp = e.to;
                    while (temp != O2)
                    {
                        int p = parent[temp];
                        AddDuplicateEdge(p, temp, 0);
                        temp = p;
                    }

                    AddDuplicateEdge(curr, e.to, 0);
                    break;
                }

                long nd = curD + e.weight;
                if (O2 == -1 || isPaired[O2] || nd < dist[e.to])
                {
                    dist[e.to] = nd;
                    owner[e.to] = O1;
                    parent[e.to] = curr;
                    pq.Enqueue(e.to, nd);
                }
            }
        }
    }
}
