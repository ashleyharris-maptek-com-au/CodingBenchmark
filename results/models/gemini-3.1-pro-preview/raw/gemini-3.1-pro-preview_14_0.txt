using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

namespace MinesweeperSolver
{
    class Program
    {
        static void Main(string[] args)
        {
            using var reader = new StreamReader(Console.OpenStandardInput(), bufferSize: 1048576);
            string line = reader.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) return;
            
            string[] dims = line.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int H = int.Parse(dims[0]);
            int W = int.Parse(dims[1]);

            byte[] board = new byte[H * W];
            for (int r = 0; r < H; r++)
            {
                line = reader.ReadLine();
                if (line == null) break;
                int len = Math.Min(W, line.Length);
                int rowOffset = r * W;
                for (int c = 0; c < len; c++)
                {
                    board[rowOffset + c] = (byte)line[c];
                }
            }

            Solver solver = new Solver(H, W, board);
            solver.Solve();

            int safe_count = 0;
            for (int r = 0; r < H; r++)
            {
                int rowOffset = r * W;
                for (int c = 0; c < W; c++)
                {
                    if (board[rowOffset + c] == 'S') safe_count++;
                }
            }

            using var writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 1048576);
            writer.WriteLine(safe_count);
            for (int r = 0; r < H; r++)
            {
                int rowOffset = r * W;
                for (int c = 0; c < W; c++)
                {
                    if (board[rowOffset + c] == 'S')
                    {
                        writer.Write(c);
                        writer.Write(' ');
                        writer.WriteLine(r);
                    }
                }
            }
        }
    }

    class Solver
    {
        int H, W;
        byte[] board;
        Queue<(int r, int c)> active = new Queue<(int r, int c)>();
        HashSet<(int r, int c)> inActive = new HashSet<(int r, int c)>();
        HashSet<(int r, int c)> frontiers = new HashSet<(int r, int c)>();

        public Solver(int h, int w, byte[] b)
        {
            H = h;
            W = w;
            board = b;

            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    byte val = board[r * W + c];
                    if (val >= '0' && val <= '8')
                    {
                        bool hasUnk = false;
                        for (int rr = r - 1; rr <= r + 1; rr++)
                        {
                            for (int cc = c - 1; cc <= c + 1; cc++)
                            {
                                if (rr >= 0 && rr < H && cc >= 0 && cc < W)
                                {
                                    if (board[rr * W + cc] == ' ') { hasUnk = true; break; }
                                }
                            }
                            if (hasUnk) break;
                        }
                        if (hasUnk)
                        {
                            active.Enqueue((r, c));
                            inActive.Add((r, c));
                            frontiers.Add((r, c));
                        }
                    }
                }
            }
        }

        List<(int r, int c)> GetUnknowns((int r, int c) cell)
        {
            var list = new List<(int r, int c)>();
            for (int r = cell.r - 1; r <= cell.r + 1; r++)
            {
                for (int c = cell.c - 1; c <= cell.c + 1; c++)
                {
                    if (r >= 0 && r < H && c >= 0 && c < W)
                    {
                        if (board[r * W + c] == ' ') list.Add((r, c));
                    }
                }
            }
            return list;
        }

        int GetRem((int r, int c) cell)
        {
            int mines = 0;
            for (int r = cell.r - 1; r <= cell.r + 1; r++)
            {
                for (int c = cell.c - 1; c <= cell.c + 1; c++)
                {
                    if (r >= 0 && r < H && c >= 0 && c < W)
                    {
                        byte val = board[r * W + c];
                        if (val == '*' || val == 'M') mines++;
                    }
                }
            }
            return (board[cell.r * W + cell.c] - '0') - mines;
        }

        List<(int r, int c)> GetNumberedNeighbors((int r, int c) cell)
        {
            var list = new List<(int r, int c)>();
            for (int r = cell.r - 1; r <= cell.r + 1; r++)
            {
                for (int c = cell.c - 1; c <= cell.c + 1; c++)
                {
                    if (r >= 0 && r < H && c >= 0 && c < W)
                    {
                        byte val = board[r * W + c];
                        if (val >= '0' && val <= '8') list.Add((r, c));
                    }
                }
            }
            return list;
        }

        void AddNeighborsToActive(int r, int c)
        {
            for (int rr = r - 1; rr <= r + 1; rr++)
            {
                for (int cc = c - 1; cc <= c + 1; cc++)
                {
                    if (rr >= 0 && rr < H && cc >= 0 && cc < W)
                    {
                        byte val = board[rr * W + cc];
                        if (val >= '0' && val <= '8')
                        {
                            if (!inActive.Contains((rr, cc)))
                            {
                                active.Enqueue((rr, cc));
                                inActive.Add((rr, cc));
                            }
                            if (frontiers.Contains((rr, cc)))
                            {
                                if (GetUnknowns((rr, cc)).Count == 0)
                                {
                                    frontiers.Remove((rr, cc));
                                }
                            }
                        }
                    }
                }
            }
        }

        public void Solve()
        {
            while (true)
            {
                bool progress = false;

                // Phase 1: Local reasoning
                while (active.Count > 0)
                {
                    var curr = active.Dequeue();
                    inActive.Remove(curr);

                    var unks = GetUnknowns(curr);
                    if (unks.Count == 0) continue;

                    int rem = GetRem(curr);

                    if (rem == 0)
                    {
                        foreach (var u in unks)
                        {
                            if (board[u.r * W + u.c] == ' ')
                            {
                                board[u.r * W + u.c] = (byte)'S';
                                AddNeighborsToActive(u.r, u.c);
                                progress = true;
                            }
                        }
                    }
                    else if (rem == unks.Count)
                    {
                        foreach (var u in unks)
                        {
                            if (board[u.r * W + u.c] == ' ')
                            {
                                board[u.r * W + u.c] = (byte)'M';
                                AddNeighborsToActive(u.r, u.c);
                                progress = true;
                            }
                        }
                    }
                }

                if (progress) continue;

                // Phase 2: Overlap constraints
                var frontierList = frontiers.ToList();
                bool phase2_progress = false;

                foreach (var A in frontierList)
                {
                    var U_A = GetUnknowns(A);
                    if (U_A.Count == 0) continue;
                    int rem_A = GetRem(A);

                    HashSet<(int, int)> b_cands = new HashSet<(int, int)>();
                    foreach (var u in U_A)
                    {
                        foreach (var n in GetNumberedNeighbors(u))
                        {
                            if (n != A) b_cands.Add(n);
                        }
                    }

                    foreach (var B in b_cands)
                    {
                        var U_B = GetUnknowns(B);
                        if (U_B.Count == 0) continue;
                        int rem_B = GetRem(B);

                        var only_A = U_A.Except(U_B).ToList();
                        var only_B = U_B.Except(U_A).ToList();

                        if (rem_A - rem_B == only_A.Count)
                        {
                            if (only_A.Count > 0 || only_B.Count > 0)
                            {
                                foreach (var u in only_A)
                                {
                                    if (board[u.r * W + u.c] == ' ')
                                    {
                                        board[u.r * W + u.c] = (byte)'M';
                                        AddNeighborsToActive(u.r, u.c);
                                        phase2_progress = true;
                                    }
                                }
                                foreach (var u in only_B)
                                {
                                    if (board[u.r * W + u.c] == ' ')
                                    {
                                        board[u.r * W + u.c] = (byte)'S';
                                        AddNeighborsToActive(u.r, u.c);
                                        phase2_progress = true;
                                    }
                                }
                            }
                        }
                        else if (rem_B - rem_A == only_B.Count)
                        {
                            if (only_A.Count > 0 || only_B.Count > 0)
                            {
                                foreach (var u in only_B)
                                {
                                    if (board[u.r * W + u.c] == ' ')
                                    {
                                        board[u.r * W + u.c] = (byte)'M';
                                        AddNeighborsToActive(u.r, u.c);
                                        phase2_progress = true;
                                    }
                                }
                                foreach (var u in only_A)
                                {
                                    if (board[u.r * W + u.c] == ' ')
                                    {
                                        board[u.r * W + u.c] = (byte)'S';
                                        AddNeighborsToActive(u.r, u.c);
                                        phase2_progress = true;
                                    }
                                }
                            }
                        }

                        if (phase2_progress) break;
                    }
                    if (phase2_progress) break;
                }

                if (phase2_progress) continue;

                // Phase 3: CSP Search
                var comps = GetComponents(frontierList);
                bool phase3_progress = false;

                foreach (var comp in comps)
                {
                    if (SolveCSP(comp))
                    {
                        phase3_progress = true;
                        break;
                    }
                }

                if (phase3_progress) continue;

                break;
            }
        }

        List<List<(int r, int c)>> GetComponents(List<(int r, int c)> frontierList)
        {
            HashSet<(int r, int c)> all_unks = new HashSet<(int r, int c)>();
            foreach (var f in frontierList)
            {
                foreach (var u in GetUnknowns(f)) all_unks.Add(u);
            }

            var comps = new List<List<(int r, int c)>>();
            var visited = new HashSet<(int r, int c)>();

            foreach (var u in all_unks)
            {
                if (visited.Contains(u)) continue;
                var comp = new List<(int r, int c)>();
                var q = new Queue<(int r, int c)>();
                q.Enqueue(u);
                visited.Add(u);

                while (q.Count > 0)
                {
                    var curr = q.Dequeue();
                    comp.Add(curr);

                    foreach (var num in GetNumberedNeighbors(curr))
                    {
                        foreach (var nu in GetUnknowns(num))
                        {
                            if (!visited.Contains(nu))
                            {
                                visited.Add(nu);
                                q.Enqueue(nu);
                            }
                        }
                    }
                }
                comps.Add(comp);
            }
            return comps;
        }

        bool SolveCSP(List<(int r, int c)> comp)
        {
            HashSet<(int r, int c)> comp_constraints = new HashSet<(int r, int c)>();
            foreach (var u in comp)
            {
                foreach (var c in GetNumberedNeighbors(u))
                {
                    comp_constraints.Add(c);
                }
            }

            var vars = comp.ToArray();
            int N = vars.Length;
            int[] mine_counts = new int[N];
            int solutions = 0;

            var constraints = comp_constraints.ToArray();
            int M = constraints.Length;
            int[] c_rem = new int[M];
            int[] c_unk = new int[M];

            for (int i = 0; i < M; i++)
            {
                c_rem[i] = GetRem(constraints[i]);
                c_unk[i] = GetUnknowns(constraints[i]).Count;
            }

            List<int>[] var_to_constraints = new List<int>[N];
            for (int i = 0; i < N; i++)
            {
                var_to_constraints[i] = new List<int>();
                for (int j = 0; j < M; j++)
                {
                    if (Math.Abs(vars[i].r - constraints[j].r) <= 1 &&
                        Math.Abs(vars[i].c - constraints[j].c) <= 1)
                    {
                        var_to_constraints[i].Add(j);
                    }
                }
            }

            Array.Sort(var_to_constraints, vars, (a, b) => b.Count.CompareTo(a.Count));

            const int MAX_SOLUTIONS = 10000;
            bool abort = false;
            bool[] current_assignment = new bool[N];

            long startTicks = System.Diagnostics.Stopwatch.GetTimestamp();
            long maxTicks = System.Diagnostics.Stopwatch.Frequency / 10;
            int calls = 0;

            void Backtrack(int idx)
            {
                if (abort) return;

                if ((calls++ & 0xFFF) == 0)
                {
                    if (System.Diagnostics.Stopwatch.GetTimestamp() - startTicks > maxTicks)
                    {
                        abort = true;
                        return;
                    }
                }

                if (idx == N)
                {
                    for (int i = 0; i < M; i++)
                    {
                        if (c_rem[i] != 0) return;
                    }

                    solutions++;
                    for (int i = 0; i < N; i++)
                    {
                        if (current_assignment[i]) mine_counts[i]++;
                    }

                    if (solutions >= MAX_SOLUTIONS) abort = true;
                    return;
                }

                // Try Safe
                bool can_be_safe = true;
                foreach (int c_idx in var_to_constraints[idx])
                {
                    c_unk[c_idx]--;
                    if (c_rem[c_idx] > c_unk[c_idx]) can_be_safe = false;
                }

                if (can_be_safe)
                {
                    current_assignment[idx] = false;
                    Backtrack(idx + 1);
                }

                foreach (int c_idx in var_to_constraints[idx])
                {
                    c_unk[c_idx]++;
                }

                if (abort) return;

                // Try Mine
                bool can_be_mine = true;
                foreach (int c_idx in var_to_constraints[idx])
                {
                    c_rem[c_idx]--;
                    c_unk[c_idx]--;
                    if (c_rem[c_idx] < 0) can_be_mine = false;
                }

                if (can_be_mine)
                {
                    current_assignment[idx] = true;
                    Backtrack(idx + 1);
                }

                foreach (int c_idx in var_to_constraints[idx])
                {
                    c_rem[c_idx]++;
                    c_unk[c_idx]++;
                }
            }

            Backtrack(0);

            if (abort || solutions == 0) return false;

            bool deduced = false;
            for (int i = 0; i < N; i++)
            {
                if (mine_counts[i] == solutions)
                {
                    if (board[vars[i].r * W + vars[i].c] == ' ')
                    {
                        board[vars[i].r * W + vars[i].c] = (byte)'M';
                        AddNeighborsToActive(vars[i].r, vars[i].c);
                        deduced = true;
                    }
                }
                else if (mine_counts[i] == 0)
                {
                    if (board[vars[i].r * W + vars[i].c] == ' ')
                    {
                        board[vars[i].r * W + vars[i].c] = (byte)'S';
                        AddNeighborsToActive(vars[i].r, vars[i].c);
                        deduced = true;
                    }
                }
            }

            return deduced;
        }
    }
}
