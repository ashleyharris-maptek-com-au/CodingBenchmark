use std::io::{self, BufRead};
use std::time::{Instant, Duration};

const TIME_LIMIT_SECS: f64 = 285.0; // 4 minutes 45 seconds to be safe

#[derive(Clone, Copy, Debug)]
struct Point { x: f64, y: f64 }

#[derive(Clone, Debug)]
struct BoundingBox {
    min_x: f64, max_x: f64,
    min_y: f64, max_y: f64,
}

impl BoundingBox {
    fn from_points(points: &[Point]) -> Self {
        let mut min_x = f64::MAX; let mut max_x = f64::MIN;
        let mut min_y = f64::MAX; let mut max_y = f64::MIN;
        for p in points {
            if p.x < min_x { min_x = p.x; }
            if p.x > max_x { max_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.y > max_y { max_y = p.y; }
        }
        Self { min_x, max_x, min_y, max_y }
    }
    
    fn intersects(&self, other: &BoundingBox) -> bool {
        self.min_x <= other.max_x + 1e-6 &&
        self.max_x >= other.min_x - 1e-6 &&
        self.min_y <= other.max_y + 1e-6 &&
        self.max_y >= other.min_y - 1e-6
    }
    
    fn is_inside(&self, other: &BoundingBox) -> bool {
        self.min_x >= other.min_x - 1e-6 &&
        self.max_x <= other.max_x + 1e-6 &&
        self.min_y >= other.min_y - 1e-6 &&
        self.max_y <= other.max_y + 1e-6
    }
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
    bb: BoundingBox,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let bb = BoundingBox::from_points(&vertices);
        Self { vertices, bb }
    }
    
    fn update_bounding_box(&mut self) {
        self.bb = BoundingBox::from_points(&self.vertices);
    }

    fn translate(&self, tx: f64, ty: f64) -> Self {
        let vertices = self.vertices.iter().map(|p| Point { x: p.x + tx, y: p.y + ty }).collect();
        let mut bb = self.bb.clone();
        bb.min_x += tx; bb.max_x += tx;
        bb.min_y += ty; bb.max_y += ty;
        Self { vertices, bb }
    }
}

fn calculate_area(vertices: &[Point]) -> f64 {
    let mut area = 0.0;
    let n = vertices.len();
    for i in 0..n {
        let j = (i + 1) % n;
        area += vertices[i].x * vertices[j].y;
        area -= vertices[j].x * vertices[i].y;
    }
    (area / 2.0).abs()
}

fn polygon_centroid(vertices: &[Point]) -> Point {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut area = 0.0;
    let n = vertices.len();
    for i in 0..n {
        let j = (i + 1) % n;
        let cross = vertices[i].x * vertices[j].y - vertices[j].x * vertices[i].y;
        cx += (vertices[i].x + vertices[j].x) * cross;
        cy += (vertices[i].y + vertices[j].y) * cross;
        area += cross;
    }
    area *= 0.5;
    if area.abs() < 1e-9 {
        let mut sum_x = 0.0;
        let mut sum_y = 0.0;
        for p in vertices {
            sum_x += p.x;
            sum_y += p.y;
        }
        return Point { x: sum_x / n as f64, y: sum_y / n as f64 };
    }
    cx /= 6.0 * area;
    cy /= 6.0 * area;
    Point { x: cx, y: cy }
}

fn shrink_polygon(poly: &mut Polygon, factor: f64) {
    let c = polygon_centroid(&poly.vertices);
    for p in &mut poly.vertices {
        p.x = c.x + (p.x - c.x) * factor;
        p.y = c.y + (p.y - c.y) * factor;
    }
    poly.update_bounding_box();
}

fn rotate_polygon(points: &[Point], angle: f64) -> Vec<Point> {
    let cos_a = angle.cos();
    let sin_a = angle.sin();
    points.iter().map(|p| Point {
        x: p.x * cos_a - p.y * sin_a,
        y: p.x * sin_a + p.y * cos_a,
    }).collect()
}

fn cross_product(p1: Point, p2: Point, p3: Point) -> f64 {
    (p2.x - p1.x) * (p3.y - p1.y) - (p2.y - p1.y) * (p3.x - p1.x)
}

fn on_segment(p: Point, q: Point, r: Point) -> bool {
    q.x <= p.x.max(r.x) + 1e-9 && q.x >= p.x.min(r.x) - 1e-9 &&
    q.y <= p.y.max(r.y) + 1e-9 && q.y >= p.y.min(r.y) - 1e-9
}

fn segments_intersect(p1: Point, q1: Point, p2: Point, q2: Point) -> bool {
    let d1 = cross_product(p2, q2, p1);
    let d2 = cross_product(p2, q2, q1);
    let d3 = cross_product(p1, q1, p2);
    let d4 = cross_product(p1, q1, q2);

    if ((d1 > 1e-9 && d2 < -1e-9) || (d1 < -1e-9 && d2 > 1e-9)) &&
       ((d3 > 1e-9 && d4 < -1e-9) || (d3 < -1e-9 && d4 > 1e-9)) {
        return true;
    }

    if d1.abs() <= 1e-9 && on_segment(p2, p1, q2) { return true; }
    if d2.abs() <= 1e-9 && on_segment(p2, q1, q2) { return true; }
    if d3.abs() <= 1e-9 && on_segment(p1, p2, q1) { return true; }
    if d4.abs() <= 1e-9 && on_segment(p1, q2, q1) { return true; }

    false
}

fn is_point_inside_polygon(p: Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let mut j = poly.vertices.len() - 1;
    for i in 0..poly.vertices.len() {
        let vi = poly.vertices[i];
        let vj = poly.vertices[j];
        
        let intersect = ((vi.y > p.y) != (vj.y > p.y)) &&
                        (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn is_polygon_inside(piece: &Polygon, stock: &Polygon) -> bool {
    if !piece.bb.is_inside(&stock.bb) {
        return false;
    }
    
    for i in 0..piece.vertices.len() {
        let e1_1 = piece.vertices[i];
        let e1_2 = piece.vertices[(i+1)%piece.vertices.len()];
        for j in 0..stock.vertices.len() {
            let e2_1 = stock.vertices[j];
            let e2_2 = stock.vertices[(j+1)%stock.vertices.len()];
            if segments_intersect(e1_1, e1_2, e2_1, e2_2) {
                return false;
            }
        }
    }
    
    if !is_point_inside_polygon(piece.vertices[0], stock) {
        return false;
    }
    true
}

fn do_polygons_intersect(p1: &Polygon, p2: &Polygon) -> bool {
    if !p1.bb.intersects(&p2.bb) {
        return false;
    }
    for i in 0..p1.vertices.len() {
        let e1_1 = p1.vertices[i];
        let e1_2 = p1.vertices[(i+1)%p1.vertices.len()];
        for j in 0..p2.vertices.len() {
            let e2_1 = p2.vertices[j];
            let e2_2 = p2.vertices[(j+1)%p2.vertices.len()];
            if segments_intersect(e1_1, e1_2, e2_1, e2_2) {
                return true;
            }
        }
    }
    if is_point_inside_polygon(p1.vertices[0], p2) { return true; }
    if is_point_inside_polygon(p2.vertices[0], p1) { return true; }
    false
}

struct SpatialGrid {
    min_x: f64, min_y: f64,
    cell_w: f64, cell_h: f64,
    cols: usize, rows: usize,
    cells: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(bb: &BoundingBox, cols: usize, rows: usize) -> Self {
        let mut cell_w = (bb.max_x - bb.min_x) / cols as f64;
        let mut cell_h = (bb.max_y - bb.min_y) / rows as f64;
        if cell_w < 1e-6 { cell_w = 1.0; }
        if cell_h < 1e-6 { cell_h = 1.0; }
        Self {
            min_x: bb.min_x,
            min_y: bb.min_y,
            cell_w,
            cell_h,
            cols,
            rows,
            cells: vec![vec![]; cols * rows],
        }
    }
    
    fn insert(&mut self, piece_idx: usize, bb: &BoundingBox) {
        let start_col = ((bb.min_x - self.min_x) / self.cell_w).floor() as isize;
        let end_col = ((bb.max_x - self.min_x) / self.cell_w).ceil() as isize;
        let start_row = ((bb.min_y - self.min_y) / self.cell_h).floor() as isize;
        let end_row = ((bb.max_y - self.min_y) / self.cell_h).ceil() as isize;

        let start_col = start_col.clamp(0, self.cols as isize - 1) as usize;
        let end_col = end_col.clamp(0, self.cols as isize - 1) as usize;
        let start_row = start_row.clamp(0, self.rows as isize - 1) as usize;
        let end_row = end_row.clamp(0, self.rows as isize - 1) as usize;

        for r in start_row..=end_row {
            for c in start_col..=end_col {
                self.cells[r * self.cols + c].push(piece_idx);
            }
        }
    }
    
    fn get_candidates(&self, bb: &BoundingBox, res: &mut Vec<usize>) {
        res.clear();
        let start_col = ((bb.min_x - self.min_x) / self.cell_w).floor() as isize;
        let end_col = ((bb.max_x - self.min_x) / self.cell_w).ceil() as isize;
        let start_row = ((bb.min_y - self.min_y) / self.cell_h).floor() as isize;
        let end_row = ((bb.max_y - self.min_y) / self.cell_h).ceil() as isize;

        let start_col = start_col.clamp(0, self.cols as isize - 1) as usize;
        let end_col = end_col.clamp(0, self.cols as isize - 1) as usize;
        let start_row = start_row.clamp(0, self.rows as isize - 1) as usize;
        let end_row = end_row.clamp(0, self.rows as isize - 1) as usize;

        for r in start_row..=end_row {
            for c in start_col..=end_col {
                res.extend(&self.cells[r * self.cols + c]);
            }
        }
        res.sort_unstable();
        res.dedup();
    }
}

struct PlacedPiece {
    id: usize,
    poly: Polygon,
    tx: f64, ty: f64, rot: f64,
}

struct StockState {
    placed_pieces: Vec<PlacedPiece>,
    grid: SpatialGrid,
    stock_bb: BoundingBox,
    placed_area: f64,
}

impl StockState {
    fn new(stock_bb: BoundingBox) -> Self {
        Self {
            placed_pieces: Vec::new(),
            grid: SpatialGrid::new(&stock_bb, 20, 20),
            stock_bb,
            placed_area: 0.0,
        }
    }
    
    fn add_piece(&mut self, id: usize, poly: Polygon, tx: f64, ty: f64, rot: f64, area: f64) {
        self.grid.insert(self.placed_pieces.len(), &poly.bb);
        self.placed_pieces.push(PlacedPiece { id, poly, tx, ty, rot });
        self.placed_area += area;
    }
    
    fn is_valid_placement(&self, piece: &Polygon, stock_poly: &Polygon, candidates_buf: &mut Vec<usize>) -> bool {
        if !is_polygon_inside(piece, stock_poly) {
            return false;
        }
        
        self.grid.get_candidates(&piece.bb, candidates_buf);
        for &idx in candidates_buf.iter() {
            if do_polygons_intersect(piece, &self.placed_pieces[idx].poly) {
                return false;
            }
        }
        true
    }
}

struct Piece {
    id: usize,
    vertices: Vec<Point>,
    area: f64,
}

struct RotatedPiece {
    rot: f64,
    shrunk_poly: Polygon,
    original_poly: Polygon,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());
    
    let first_line = lines.next().unwrap();
    let mut parts = first_line.split_whitespace();
    let sv: usize = parts.next().unwrap().parse().unwrap();
    let np: usize = parts.next().unwrap().parse().unwrap();
    
    let mut stock_vertices = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let mut parts = line.split_whitespace();
        let x: f64 = parts.next().unwrap().parse().unwrap();
        let y: f64 = parts.next().unwrap().parse().unwrap();
        stock_vertices.push(Point { x, y });
    }
    
    let base_stock_poly = Polygon::new(stock_vertices);
    
    let mut pieces = Vec::with_capacity(np);
    for i in 0..np {
        let line = lines.next().unwrap();
        let pv: usize = line.trim().parse().unwrap();
        let mut vertices = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let mut parts = line.split_whitespace();
            let x: f64 = parts.next().unwrap().parse().unwrap();
            let y: f64 = parts.next().unwrap().parse().unwrap();
            vertices.push(Point { x, y });
        }
        let area = calculate_area(&vertices);
        pieces.push(Piece { id: i, vertices, area });
    }
    
    pieces.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap());
    
    let mut stocks: Vec<StockState> = Vec::new();
    let start_time = Instant::now();
    let rotations = [0.0, std::f64::consts::PI / 2.0, std::f64::consts::PI, 3.0 * std::f64::consts::PI / 2.0];
    
    let mut candidates_buf = Vec::new();
    
    for (piece_idx, piece) in pieces.iter().enumerate() {
        let elapsed = start_time.elapsed().as_secs_f64();
        let remaining_time = (TIME_LIMIT_SECS - elapsed).max(0.01);
        let remaining_pieces = (np - piece_idx) as f64;
        let time_for_this_piece = remaining_time / remaining_pieces;
        
        let piece_start = Instant::now();
        
        let mut rotated_pieces = Vec::new();
        for rot in rotations.iter() {
            let original_poly = Polygon::new(rotate_polygon(&piece.vertices, *rot));
            let mut shrunk_poly = original_poly.clone();
            shrink_polygon(&mut shrunk_poly, 0.99999);
            rotated_pieces.push(RotatedPiece {
                rot: *rot,
                shrunk_poly,
                original_poly,
            });
        }
        
        let mut stock_order: Vec<usize> = (0..stocks.len()).collect();
        stock_order.sort_by(|&a, &b| stocks[b].placed_area.partial_cmp(&stocks[a].placed_area).unwrap());
        stock_order.push(stocks.len()); // Option for new stock
        
        let mut best_global_placement = None;
        let mut best_global_score = f64::MAX;
        
        let grid_size = 40;
        
        'stock_loop: for &stock_idx in &stock_order {
            let is_new_stock = stock_idx == stocks.len();
            let penalty = if is_new_stock { 1000000.0 } else { 0.0 };
            let stock_bb = if is_new_stock { &base_stock_poly.bb } else { &stocks[stock_idx].stock_bb };
            
            for rp in &rotated_pieces {
                let min_tx = stock_bb.min_x - rp.original_poly.bb.min_x;
                let max_tx = stock_bb.max_x - rp.original_poly.bb.max_x;
                let min_ty = stock_bb.min_y - rp.original_poly.bb.min_y;
                let max_ty = stock_bb.max_y - rp.original_poly.bb.max_y;
                
                if min_tx > max_tx || min_ty > max_ty {
                    continue;
                }
                
                let mut candidates = Vec::with_capacity((grid_size + 1) * (grid_size + 1));
                for y_step in 0..=grid_size {
                    let ty = min_ty + (max_ty - min_ty) * (y_step as f64 / grid_size as f64);
                    for x_step in 0..=grid_size {
                        let tx = min_tx + (max_tx - min_tx) * (x_step as f64 / grid_size as f64);
                        let score = ty + tx * 0.1;
                        candidates.push((score, tx, ty));
                    }
                }
                candidates.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
                
                let mut loops = 0;
                for (score, tx, ty) in candidates {
                    loops += 1;
                    if loops % 100 == 0 {
                        if piece_start.elapsed().as_secs_f64() > time_for_this_piece {
                            break;
                        }
                    }
                    
                    let total_score = score + penalty;
                    if total_score >= best_global_score {
                        break; 
                    }
                    
                    let test_poly = rp.shrunk_poly.translate(tx, ty);
                    let valid = if is_new_stock {
                        is_polygon_inside(&test_poly, &base_stock_poly)
                    } else {
                        stocks[stock_idx].is_valid_placement(&test_poly, &base_stock_poly, &mut candidates_buf)
                    };
                    
                    if valid {
                        best_global_score = total_score;
                        best_global_placement = Some((stock_idx, tx, ty, rp.rot, rp.original_poly.translate(tx, ty)));
                        break; 
                    }
                }
            }
            
            if is_new_stock && best_global_placement.is_some() {
                break;
            }
        }
        
        if best_global_placement.is_none() {
            let mut rng_seed = 12345 + piece.id as u32;
            fn next_rand(seed: &mut u32) -> f64 {
                *seed = seed.wrapping_mul(1664525).wrapping_add(1013904223);
                (*seed as f64) / (u32::MAX as f64)
            }
        
            let mut attempts = 0;
            'fallback: loop {
                attempts += 1;
                let stock_idx = if next_rand(&mut rng_seed) < 0.2 && !stocks.is_empty() {
                    (next_rand(&mut rng_seed) * stocks.len() as f64) as usize
                } else {
                    stocks.len()
                };
                
                let is_new_stock = stock_idx == stocks.len();
                let stock_bb = if is_new_stock { &base_stock_poly.bb } else { &stocks[stock_idx].stock_bb };
                
                let rp_idx = (next_rand(&mut rng_seed) * rotated_pieces.len() as f64) as usize;
                let rp = &rotated_pieces[rp_idx];
                
                let min_tx = stock_bb.min_x - rp.original_poly.bb.min_x;
                let max_tx = stock_bb.max_x - rp.original_poly.bb.max_x;
                let min_ty = stock_bb.min_y - rp.original_poly.bb.min_y;
                let max_ty = stock_bb.max_y - rp.original_poly.bb.max_y;
                
                if min_tx > max_tx || min_ty > max_ty {
                    if attempts > 10000 { break; }
                    continue;
                }
                
                let tx = min_tx + next_rand(&mut rng_seed) * (max_tx - min_tx);
                let ty = min_ty + next_rand(&mut rng_seed) * (max_ty - min_ty);
                
                let test_poly = rp.shrunk_poly.translate(tx, ty);
                
                let valid = if is_new_stock {
                    is_polygon_inside(&test_poly, &base_stock_poly)
                } else {
                    stocks[stock_idx].is_valid_placement(&test_poly, &base_stock_poly, &mut candidates_buf)
                };
                
                if valid {
                    best_global_placement = Some((stock_idx, tx, ty, rp.rot, rp.original_poly.translate(tx, ty)));
                    break 'fallback;
                }
                
                if attempts > 50000 {
                    best_global_placement = Some((stocks.len(), 0.0, 0.0, rp.rot, rp.original_poly.translate(0.0, 0.0)));
                    break 'fallback;
                }
            }
        }
        
        if let Some((stock_idx, tx, ty, rot, final_poly)) = best_global_placement {
            if stock_idx == stocks.len() {
                let mut new_stock = StockState::new(base_stock_poly.bb.clone());
                new_stock.add_piece(piece.id, final_poly, tx, ty, rot, piece.area);
                stocks.push(new_stock);
            } else {
                stocks[stock_idx].add_piece(piece.id, final_poly, tx, ty, rot, piece.area);
            }
        }
    }
    
    println!("{}", stocks.len());
    let mut output = vec![None; np];
    for (stock_idx, stock) in stocks.iter().enumerate() {
        for placed in &stock.placed_pieces {
            output[placed.id] = Some((stock_idx, placed.tx, placed.ty, placed.rot));
        }
    }
    
    for i in 0..np {
        if let Some((s, x, y, r)) = output[i] {
            println!("{} {} {} {}", s, x, y, r);
        } else {
            println!("0 0.0 0.0 0.0");
        }
    }
}
