#![allow(warnings)]

use std::io::Read;
use std::time::{Instant, Duration};
use std::hash::{BuildHasherDefault, Hasher};

struct FastRng {
    state: u64,
}

impl FastRng {
    fn new(seed: u64) -> Self {
        FastRng { state: seed }
    }
    fn next(&mut self) -> u32 {
        self.state ^= self.state << 13;
        self.state ^= self.state >> 7;
        self.state ^= self.state << 17;
        self.state as u32
    }
}

#[derive(Default)]
struct FastHasher {
    hash: u64,
}

impl Hasher for FastHasher {
    fn write(&mut self, bytes: &[u8]) {
        for &b in bytes {
            self.hash = self.hash.rotate_left(5) ^ (b as u64);
        }
    }
    fn write_i64(&mut self, i: i64) {
        let mut x = i as u64;
        x ^= x >> 30;
        x = x.wrapping_mul(0xbf58476d1ce4e5b9);
        x ^= x >> 27;
        x = x.wrapping_mul(0x94d049bb133111eb);
        x ^= x >> 31;
        self.hash = x;
    }
    fn finish(&self) -> u64 { self.hash }
}

type FastBuildHasher = BuildHasherDefault<FastHasher>;
type FastHashMap<K, V> = std::collections::HashMap<K, V, FastBuildHasher>;

fn read_all_fast() -> (usize, i64, Vec<i64>) {
    let mut buffer = Vec::new();
    std::io::stdin().read_to_end(&mut buffer).unwrap();
    
    let mut pos = 0;
    
    let next_i64 = |buf: &[u8], pos: &mut usize| -> Option<i64> {
        while *pos < buf.len() && buf[*pos] <= b' ' {
            *pos += 1;
        }
        if *pos >= buf.len() { return None; }
        let mut sign = 1;
        if buf[*pos] == b'-' {
            sign = -1;
            *pos += 1;
        } else if buf[*pos] == b'+' {
            *pos += 1;
        }
        let mut res: i64 = 0;
        while *pos < buf.len() && buf[*pos] > b' ' {
            res = res * 10 + (buf[*pos] - b'0') as i64;
            *pos += 1;
        }
        Some(res * sign)
    };
    
    let n = next_i64(&buffer, &mut pos).unwrap_or(0) as usize;
    let target = next_i64(&buffer, &mut pos).unwrap_or(0);
    
    let mut nums = Vec::with_capacity(n);
    while let Some(x) = next_i64(&buffer, &mut pos) {
        nums.push(x);
    }
    
    (n, target, nums)
}

fn print_ans(ans: &[usize], zeroes: &[usize]) {
    println!("YES");
    let mut final_ans = ans.to_vec();
    final_ans.extend_from_slice(zeroes);
    final_ans.sort_unstable();
    
    print!("[");
    for (i, &idx) in final_ans.iter().enumerate() {
        if i > 0 { print!(" "); }
        print!("{}", idx);
    }
    println!("]");
}

fn print_subset<F>(items: &[(i64, usize)], zeroes: &[usize], filter: F) 
where F: Fn(i64) -> bool 
{
    let mut ans = Vec::new();
    for &(v, idx) in items {
        if filter(v) { ans.push(idx); }
    }
    print_ans(&ans, zeroes);
}

fn meet_in_the_middle(items: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let mid = items.len() / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let mut left_sums = Vec::with_capacity(1 << left.len());
    for mask in 0..(1u32 << left.len()) {
        let mut s = 0;
        for i in 0..left.len() {
            if (mask & (1 << i)) != 0 {
                s += left[i].0;
            }
        }
        left_sums.push((s, mask));
    }
    left_sums.sort_unstable_by_key(|k| k.0);

    for mask in 0..(1u32 << right.len()) {
        let mut s = 0;
        for i in 0..right.len() {
            if (mask & (1 << i)) != 0 {
                s += right[i].0;
            }
        }
        let rem = target - s;
        if let Ok(idx) = left_sums.binary_search_by_key(&rem, |k| k.0) {
            let left_mask = left_sums[idx].1;
            let mut ans = Vec::new();
            for i in 0..left.len() {
                if (left_mask & (1 << i)) != 0 {
                    ans.push(left[i].1);
                }
            }
            for i in 0..right.len() {
                if (mask & (1 << i)) != 0 {
                    ans.push(right[i].1);
                }
            }
            return Some(ans);
        }
    }
    None
}

enum DpResult {
    Found(Vec<usize>),
    NoSolution,
    Aborted,
}

fn hashmap_dp(items: &[(i64, usize)], target: i64, max_states: usize, start_time: Instant, time_limit: Duration) -> DpResult {
    let mut first_achieved: FastHashMap<i64, usize> = FastHashMap::default();
    let mut states = Vec::new();
    states.push(0i64);
    
    let mut rem_pos = vec![0; items.len()];
    let mut rem_neg = vec![0; items.len()];
    let mut p_sum = 0;
    let mut n_sum = 0;
    for i in (0..items.len()).rev() {
        if items[i].0 > 0 { p_sum += items[i].0; }
        else { n_sum += items[i].0; }
        rem_pos[i] = p_sum;
        rem_neg[i] = n_sum;
    }

    for (i, &(v, orig_idx)) in items.iter().enumerate() {
        if start_time.elapsed() > time_limit { return DpResult::Aborted; }
        
        let mut new_states = Vec::new();
        for &s in &states {
            let nxt = s + v;
            if nxt == target {
                let mut path = vec![orig_idx];
                let mut curr = s;
                while curr != 0 {
                    let prev_i = first_achieved[&curr];
                    path.push(items[prev_i].1);
                    curr -= items[prev_i].0;
                }
                return DpResult::Found(path);
            }
            if nxt + rem_pos[i] < target { continue; }
            if nxt + rem_neg[i] > target { continue; }
            
            if nxt != 0 && !first_achieved.contains_key(&nxt) {
                new_states.push(nxt);
            }
        }
        
        for &ns in &new_states {
            first_achieved.insert(ns, i);
            states.push(ns);
        }
        
        if states.len() > max_states {
            return DpResult::Aborted;
        }
    }
    DpResult::NoSolution
}

fn solve_bb_exact(
    items: &[(i64, usize)],
    target: i64,
    node_limit: usize,
    start_time: Instant,
    time_limit: Duration
) -> DpResult {
    let mut rem_pos = vec![0; items.len()];
    let mut rem_neg = vec![0; items.len()];
    let mut p_sum = 0;
    let mut n_sum = 0;
    for i in (0..items.len()).rev() {
        if items[i].0 > 0 { p_sum += items[i].0; }
        else { n_sum += items[i].0; }
        rem_pos[i] = p_sum;
        rem_neg[i] = n_sum;
    }

    let mut path = Vec::new();
    let mut ans = None;
    let mut nodes = 0;
    
    fn dfs(
        idx: usize,
        current_sum: i64,
        target: i64,
        items: &[(i64, usize)],
        rem_pos: &[i64],
        rem_neg: &[i64],
        path: &mut Vec<usize>,
        ans: &mut Option<Vec<usize>>,
        nodes: &mut usize,
        max_nodes: usize,
        start_time: Instant,
        time_limit: Duration
    ) -> bool {
        *nodes += 1;
        if *nodes > max_nodes || start_time.elapsed() > time_limit {
            return true;
        }
        if current_sum == target {
            *ans = Some(path.clone());
            return true;
        }
        if idx == items.len() { return false; }
        
        if current_sum + rem_pos[idx] < target { return false; }
        if current_sum + rem_neg[idx] > target { return false; }

        let try_include_first = if items[idx].0 > 0 { current_sum < target } else { current_sum > target };

        if try_include_first {
            path.push(items[idx].1);
            if dfs(idx + 1, current_sum + items[idx].0, target, items, rem_pos, rem_neg, path, ans, nodes, max_nodes, start_time, time_limit) {
                return true;
            }
            path.pop();
            
            if ans.is_some() { return true; }

            if dfs(idx + 1, current_sum, target, items, rem_pos, rem_neg, path, ans, nodes, max_nodes, start_time, time_limit) {
                return true;
            }
        } else {
            if dfs(idx + 1, current_sum, target, items, rem_pos, rem_neg, path, ans, nodes, max_nodes, start_time, time_limit) {
                return true;
            }
            if ans.is_some() { return true; }

            path.push(items[idx].1);
            if dfs(idx + 1, current_sum + items[idx].0, target, items, rem_pos, rem_neg, path, ans, nodes, max_nodes, start_time, time_limit) {
                return true;
            }
            path.pop();
        }
        false
    }

    let aborted = dfs(0, 0, target, items, &rem_pos, &rem_neg, &mut path, &mut ans, &mut nodes, node_limit, start_time, time_limit);
    
    if let Some(p) = ans {
        DpResult::Found(p)
    } else if aborted {
        DpResult::Aborted
    } else {
        DpResult::NoSolution
    }
}

fn bb_search_closest(
    idx: usize,
    current_pool_sum: i64,
    sub_target: i64,
    items: &[(i64, usize)],
    rem_pos: &[i64],
    rem_neg: &[i64],
    path: &mut Vec<usize>,
    best_path: &mut Vec<usize>,
    best_diff: &mut i64,
    nodes: &mut usize,
    max_nodes: usize
) {
    *nodes += 1;
    let diff = (current_pool_sum - sub_target).abs();
    if diff < *best_diff {
        *best_diff = diff;
        *best_path = path.clone();
        if diff == 0 { return; }
    }
    if *nodes > max_nodes { return; }
    if idx == items.len() { return; }
    
    let min_possible = current_pool_sum + rem_neg[idx];
    let max_possible = current_pool_sum + rem_pos[idx];
    
    if min_possible > sub_target {
        let d = min_possible - sub_target;
        if d >= *best_diff { return; }
    }
    if max_possible < sub_target {
        let d = sub_target - max_possible;
        if d >= *best_diff { return; }
    }

    let try_include_first = if items[idx].0 > 0 { current_pool_sum < sub_target } else { current_pool_sum > sub_target };

    if try_include_first {
        path.push(idx);
        bb_search_closest(idx + 1, current_pool_sum + items[idx].0, sub_target, items, rem_pos, rem_neg, path, best_path, best_diff, nodes, max_nodes);
        path.pop();
        if *best_diff == 0 { return; }

        bb_search_closest(idx + 1, current_pool_sum, sub_target, items, rem_pos, rem_neg, path, best_path, best_diff, nodes, max_nodes);
    } else {
        bb_search_closest(idx + 1, current_pool_sum, sub_target, items, rem_pos, rem_neg, path, best_path, best_diff, nodes, max_nodes);
        if *best_diff == 0 { return; }

        path.push(idx);
        bb_search_closest(idx + 1, current_pool_sum + items[idx].0, sub_target, items, rem_pos, rem_neg, path, best_path, best_diff, nodes, max_nodes);
        path.pop();
    }
}

fn local_search(items: &[(i64, usize)], target: i64, time_limit: Duration, global_start: Instant) -> Option<Vec<usize>> {
    let mut rng = FastRng::new(42);
    let mut current_sum = 0;
    let mut in_subset = vec![false; items.len()];
    
    for (i, &(v, _)) in items.iter().enumerate() {
        if (target > current_sum && v > 0) || (target < current_sum && v < 0) {
            in_subset[i] = true;
            current_sum += v;
        }
    }
    
    if current_sum == target {
        let mut ans = Vec::new();
        for i in 0..items.len() {
            if in_subset[i] { ans.push(items[i].1); }
        }
        return Some(ans);
    }
    
    let mut best_diff_overall = (current_sum - target).abs();
    let mut stuck_count = 0;
    
    while global_start.elapsed() < time_limit {
        let diff = target - current_sum;
        
        let pool_size = 40.min(items.len());
        let mut pool = Vec::with_capacity(pool_size);
        
        let num_biased = pool_size / 2;
        for _ in 0..num_biased {
            let mut best_idx = rng.next() as usize % items.len();
            for _ in 0..5 {
                let idx = rng.next() as usize % items.len();
                let benefit = if in_subset[idx] { -items[idx].0 } else { items[idx].0 };
                let best_benefit = if in_subset[best_idx] { -items[best_idx].0 } else { items[best_idx].0 };
                
                if (diff > 0 && benefit > best_benefit) || (diff < 0 && benefit < best_benefit) {
                    best_idx = idx;
                }
            }
            pool.push(best_idx);
        }
        while pool.len() < pool_size {
            pool.push(rng.next() as usize % items.len());
        }
        pool.sort_unstable();
        pool.dedup();
        
        let mut pool_items = Vec::new();
        let mut base_sum = current_sum;
        for &idx in &pool {
            if in_subset[idx] {
                base_sum -= items[idx].0;
            }
            pool_items.push((items[idx].0, idx));
        }
        
        let sub_target = target - base_sum;
        
        let mut rem_pos = vec![0; pool_items.len()];
        let mut rem_neg = vec![0; pool_items.len()];
        let mut p_sum = 0;
        let mut n_sum = 0;
        for i in (0..pool_items.len()).rev() {
            if pool_items[i].0 > 0 { p_sum += pool_items[i].0; }
            else { n_sum += pool_items[i].0; }
            rem_pos[i] = p_sum;
            rem_neg[i] = n_sum;
        }
        
        let mut path = Vec::new();
        let mut best_path = Vec::new();
        let mut best_diff = i64::MAX;
        let mut nodes = 0;
        
        bb_search_closest(0, 0, sub_target, &pool_items, &rem_pos, &rem_neg, &mut path, &mut best_path, &mut best_diff, &mut nodes, 2000);
        
        for &idx in &pool {
            in_subset[idx] = false;
        }
        let mut new_pool_sum = 0;
        for &idx_in_pool in &best_path {
            let orig_idx = pool_items[idx_in_pool].1;
            in_subset[orig_idx] = true;
            new_pool_sum += pool_items[idx_in_pool].0;
        }
        current_sum = base_sum + new_pool_sum;
        
        if current_sum == target {
            let mut ans = Vec::new();
            for i in 0..items.len() {
                if in_subset[i] { ans.push(items[i].1); }
            }
            return Some(ans);
        }
        
        let current_diff = (current_sum - target).abs();
        if current_diff == best_diff_overall {
            stuck_count += 1;
        } else if current_diff < best_diff_overall {
            best_diff_overall = current_diff;
            stuck_count = 0;
        } else {
            stuck_count += 1;
        }
        
        if stuck_count > 50 {
            let perturb_size = 10.min(items.len());
            for _ in 0..perturb_size {
                let idx = rng.next() as usize % items.len();
                if in_subset[idx] {
                    in_subset[idx] = false;
                    current_sum -= items[idx].0;
                } else {
                    in_subset[idx] = true;
                    current_sum += items[idx].0;
                }
            }
            stuck_count = 0;
            best_diff_overall = (current_sum - target).abs();
        }
    }
    None
}

fn main() {
    let start_time = Instant::now();
    let (_n, target, raw_nums) = read_all_fast();
    
    let mut zeroes = Vec::new();
    let mut items = Vec::new();
    let mut total_pos = 0i64;
    let mut total_neg = 0i64;
    
    for (i, &v) in raw_nums.iter().enumerate() {
        if v == 0 {
            zeroes.push(i);
        } else {
            items.push((v, i));
            if v > 0 { total_pos += v; }
            else { total_neg += v; }
        }
    }
    
    if target == 0 {
        print_ans(&[], &zeroes);
        return;
    }
    if target > total_pos || target < total_neg {
        println!("NO");
        return;
    }
    if target == total_pos {
        print_subset(&items, &zeroes, |v| v > 0);
        return;
    }
    if target == total_neg {
        print_subset(&items, &zeroes, |v| v < 0);
        return;
    }
    
    items.sort_unstable_by_key(|&(v, _)| std::cmp::Reverse(v.abs()));
    
    let n_items = items.len();
    if n_items <= 45 {
        if let Some(ans) = meet_in_the_middle(&items, target) {
            print_ans(&ans, &zeroes);
        } else {
            println!("NO");
        }
        return;
    }
    
    match hashmap_dp(&items, target, 2_000_000, start_time, Duration::from_secs(5)) {
        DpResult::Found(ans) => { print_ans(&ans, &zeroes); return; }
        DpResult::NoSolution => { println!("NO"); return; }
        DpResult::Aborted => {}
    }

    match solve_bb_exact(&items, target, 10_000_000, start_time, Duration::from_secs(10)) {
        DpResult::Found(ans) => { print_ans(&ans, &zeroes); return; }
        DpResult::NoSolution => { println!("NO"); return; }
        DpResult::Aborted => {}
    }

    if let Some(ans) = local_search(&items, target, Duration::from_secs(280), start_time) {
        print_ans(&ans, &zeroes);
    } else {
        println!("NO");
    }
}
