#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <set>
#include <map>
#include <random>
#include <chrono>
#include <iomanip>

using namespace std;

struct Point {
    double x, y, z;
};

Point operator-(const Point& a, const Point& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
Point cross(const Point& a, const Point& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
double dot(const Point& a, const Point& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}
double length(const Point& a) { return sqrt(dot(a, a)); }
Point normalize(Point a) {
    double l = length(a);
    if (l < 1e-18) return {0, 0, 0};
    return {a.x / l, a.y / l, a.z / l};
}

struct Face {
    int v[3];
    Point normal;
    double offset;
    bool active;
    Face(int a, int b, int c, const vector<Point>& pts) {
        v[0] = a; v[1] = b; v[2] = c;
        active = true;
        Point p0 = pts[a], p1 = pts[b], p2 = pts[c];
        normal = normalize(cross(p1 - p0, p2 - p0));
        offset = dot(normal, p0);
    }
    bool canSee(const Point& p, double threshold) const {
        return dot(normal, p) - offset > threshold;
    }
};

struct Tetra {
    double cx, cy, cz, rx, ry, rz;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    auto start_time = chrono::steady_clock::now();

    int V;
    double edge_length;
    if (!(cin >> V >> edge_length)) return 0;
    if (V < 4 || edge_length <= 0) {
        cout << "0\n";
        return 0;
    }

    vector<Point> pts(V);
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;

    for (int i = 0; i < V; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
        min_x = min(min_x, pts[i].x); max_x = max(max_x, pts[i].x);
        min_y = min(min_y, pts[i].y); max_y = max(max_y, pts[i].y);
        min_z = min(min_z, pts[i].z); max_z = max(max_z, pts[i].z);
    }

    double span = max({max_x - min_x, max_y - min_y, max_z - min_z});
    if (span < 1e-9) {
        cout << "0\n";
        return 0;
    }

    mt19937 rng(42);
    shuffle(pts.begin(), pts.end(), rng);

    double noise_scale = span * 1e-9;
    uniform_real_distribution<double> dist(-noise_scale, noise_scale);
    for (auto& p : pts) {
        p.x += dist(rng);
        p.y += dist(rng);
        p.z += dist(rng);
    }

    int ex[6] = {0, 0, 0, 0, 0, 0};
    for (int i = 1; i < V; ++i) {
        if (pts[i].x < pts[ex[0]].x) ex[0] = i;
        if (pts[i].x > pts[ex[1]].x) ex[1] = i;
        if (pts[i].y < pts[ex[2]].y) ex[2] = i;
        if (pts[i].y > pts[ex[3]].y) ex[3] = i;
        if (pts[i].z < pts[ex[4]].z) ex[4] = i;
        if (pts[i].z > pts[ex[5]].z) ex[5] = i;
    }

    set<int> ex_set(ex, ex + 6);
    int idx = 0;
    for (int x : ex_set) {
        swap(pts[idx++], pts[x]);
    }

    double span2 = span * span;
    double span3 = span * span * span;

    int p2 = -1, p3 = -1;
    for (int i = 2; i < V; ++i) {
        if (length(cross(pts[i] - pts[0], pts[1] - pts[0])) > span2 * 1e-9) {
            p2 = i; break;
        }
    }
    if (p2 == -1) { cout << "0\n"; return 0; }
    swap(pts[2], pts[p2]);

    for (int i = 3; i < V; ++i) {
        double vol = dot(cross(pts[1] - pts[0], pts[2] - pts[0]), pts[i] - pts[0]);
        if (abs(vol) > span3 * 1e-9) {
            p3 = i; break;
        }
    }
    if (p3 == -1) { cout << "0\n"; return 0; }
    swap(pts[3], pts[p3]);

    vector<Face> faces;
    int init_v[4] = {0, 1, 2, 3};
    for (int i = 0; i < 4; ++i) {
        int a = init_v[(i + 1) % 4];
        int b = init_v[(i + 2) % 4];
        int c = init_v[(i + 3) % 4];
        Face f(a, b, c, pts);
        if (dot(f.normal, pts[init_v[i]]) - f.offset > 0) {
            f = Face(a, c, b, pts);
        }
        faces.push_back(f);
    }

    vector<int> active_points;
    for (int i = 4; i < V; ++i) active_points.push_back(i);

    double vis_threshold = span * 1e-9;

    for (int i : active_points) {
        vector<int> visible;
        for (size_t j = 0; j < faces.size(); ++j) {
            if (faces[j].active && faces[j].canSee(pts[i], vis_threshold)) {
                visible.push_back(j);
            }
        }
        if (visible.empty()) continue;

        map<pair<int, int>, int> edge_counts;
        for (int f_idx : visible) {
            for (int j = 0; j < 3; ++j) {
                int a = faces[f_idx].v[j];
                int b = faces[f_idx].v[(j + 1) % 3];
                edge_counts[{a, b}]++;
            }
            faces[f_idx].active = false;
        }

        for (auto& kv : edge_counts) {
            if (kv.second == 1) {
                faces.push_back(Face(kv.first.second, kv.first.first, i, pts));
            }
        }
    }

    double s = edge_length / sqrt(2.0);
    double margin = edge_length * 1e-4;

    long long start_y = floor(min_y / s) - 1;
    long long end_y = ceil(max_y / s) + 1;
    long long start_z = floor(min_z / s) - 1;
    long long end_z = ceil(max_z / s) + 1;
    long long start_x_global = floor(min_x / s) - 1;
    long long end_x_global = ceil(max_x / s) + 1;

    double rx = atan2(sqrt(2.0), 1.0);
    double ry = 0.0;
    double rz = acos(-1.0) / 4.0;

    vector<Tetra> placed;
    placed.reserve(100000);
    bool time_out = false;

    for (long long y = start_y; y <= end_y; ++y) {
        for (long long z = start_z; z <= end_z; ++z) {
            double max_min_x = -1e18;
            double min_max_x = 1e18;
            bool possible = true;

            for (const auto& f : faces) {
                if (!f.active) continue;

                for (int i = 0; i < 4; ++i) {
                    double vx_coeff = s * f.normal.x;
                    double y_val = (i == 1 || i == 3) ? (y + 1) * s : y * s;
                    double z_val = (i == 2 || i == 3) ? (z + 1) * s : z * s;
                    double const_term = f.normal.y * y_val + f.normal.z * z_val - f.offset + margin;

                    if (i == 1 || i == 2) const_term += s * f.normal.x;

                    if (vx_coeff > 1e-9 * s) {
                        double bound = -const_term / vx_coeff;
                        if (bound < min_max_x) min_max_x = bound;
                    } else if (vx_coeff < -1e-9 * s) {
                        double bound = -const_term / vx_coeff;
                        if (bound > max_min_x) max_min_x = bound;
                    } else {
                        if (const_term >= 0) {
                            possible = false; break;
                        }
                    }
                }
                if (!possible || max_min_x > min_max_x) {
                    possible = false; break;
                }
            }

            if (!possible) continue;

            long long start_x = max((long long)ceil(max_min_x), start_x_global);
            long long end_x = min((long long)floor(min_max_x), end_x_global);

            for (long long x = start_x; x <= end_x; ++x) {
                placed.push_back({(x + 0.5) * s, (y + 0.5) * s, (z + 0.5) * s, rx, ry, rz});

                if ((placed.size() & 8191) == 0) {
                    auto now = chrono::steady_clock::now();
                    if (chrono::duration_cast<chrono::duration<double>>(now - start_time).count() > 28.0) {
                        time_out = true; break;
                    }
                }
            }
            if (time_out) break;
        }
        if (time_out) break;
    }

    cout << placed.size() << "\n";
    cout << fixed << setprecision(6);
    for (const auto& t : placed) {
        cout << t.cx << " " << t.cy << " " << t.cz << " "
             << t.rx << " " << t.ry << " " << t.rz << "\n";
    }

    return 0;
}
