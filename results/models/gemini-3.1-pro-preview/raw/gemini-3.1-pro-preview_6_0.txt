using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

class OrbitalTSP
{
    struct Orbit
    {
        public int id;
        public double[] r;
        public double[] v;
        public double[] h;
        public double[] e_vec;
        public double a;
    }

    static double[,] dist;
    static int N;
    static int START_NODE;

    static double[] GetNumbers(string line)
    {
        var parts = line.Split(new[] { ' ', '\t', ',', ':', '=' }, StringSplitOptions.RemoveEmptyEntries);
        List<double> nums = new List<double>();
        foreach (var p in parts)
        {
            if (double.TryParse(p, System.Globalization.NumberStyles.Any, System.Globalization.CultureInfo.InvariantCulture, out double val))
            {
                nums.Add(val);
            }
        }
        return nums.Skip(Math.Max(0, nums.Count - 6)).ToArray();
    }

    static Orbit ParseOrbit(string line, double mu, int id = -1)
    {
        double[] nums = GetNumbers(line);
        if (nums.Length < 6) throw new Exception("Invalid orbit line: " + line);
        double[] r = new double[] { nums[0], nums[1], nums[2] };
        double[] v = new double[] { nums[3], nums[4], nums[5] };
        
        double[] h = new double[] {
            r[1]*v[2] - r[2]*v[1],
            r[2]*v[0] - r[0]*v[2],
            r[0]*v[1] - r[1]*v[0]
        };
        
        double r_mag = Math.Sqrt(r[0]*r[0] + r[1]*r[1] + r[2]*r[2]);
        double v_mag2 = v[0]*v[0] + v[1]*v[1] + v[2]*v[2];
        double E = v_mag2 / 2 - mu / r_mag;
        double a = -mu / (2 * E);
        
        double[] v_x_h = new double[] {
            v[1]*h[2] - v[2]*h[1],
            v[2]*h[0] - v[0]*h[2],
            v[0]*h[1] - v[1]*h[0]
        };
        
        double[] e_vec = new double[] {
            v_x_h[0]/mu - r[0]/r_mag,
            v_x_h[1]/mu - r[1]/r_mag,
            v_x_h[2]/mu - r[2]/r_mag
        };
        
        return new Orbit { id = id, r = r, v = v, h = h, e_vec = e_vec, a = a };
    }

    static double GetDistance(Orbit o1, Orbit o2, double mu)
    {
        if (o1.a <= 0 || o2.a <= 0)
        {
            double dr = Math.Sqrt(Math.Pow(o1.r[0]-o2.r[0],2) + Math.Pow(o1.r[1]-o2.r[1],2) + Math.Pow(o1.r[2]-o2.r[2],2));
            double dv = Math.Sqrt(Math.Pow(o1.v[0]-o2.v[0],2) + Math.Pow(o1.v[1]-o2.v[1],2) + Math.Pow(o1.v[2]-o2.v[2],2));
            return dr * 0.001 + dv; 
        }
        
        double a1 = o1.a;
        double a2 = o2.a;
        double v1 = Math.Sqrt(mu / a1);
        double v2 = Math.Sqrt(mu / a2);
        double v_transfer1 = Math.Sqrt(mu * Math.Max(0, 2.0 / a1 - 2.0 / (a1 + a2)));
        double v_transfer2 = Math.Sqrt(mu * Math.Max(0, 2.0 / a2 - 2.0 / (a1 + a2)));
        double dv_hohmann = Math.Abs(v_transfer1 - v1) + Math.Abs(v_transfer2 - v2);
        
        double dot_h = o1.h[0] * o2.h[0] + o1.h[1] * o2.h[1] + o1.h[2] * o2.h[2];
        double mag_h1 = Math.Sqrt(o1.h[0] * o1.h[0] + o1.h[1] * o1.h[1] + o1.h[2] * o1.h[2]);
        double mag_h2 = Math.Sqrt(o2.h[0] * o2.h[0] + o2.h[1] * o2.h[1] + o2.h[2] * o2.h[2]);
        
        double cos_alpha = 1;
        if (mag_h1 > 0 && mag_h2 > 0)
        {
            cos_alpha = dot_h / (mag_h1 * mag_h2);
            if (cos_alpha > 1) cos_alpha = 1;
            if (cos_alpha < -1) cos_alpha = -1;
        }
        double alpha = Math.Acos(cos_alpha);
        
        double v_apo = Math.Sqrt(mu / Math.Max(a1, a2));
        double dv_plane = 2 * v_apo * Math.Sin(alpha / 2);
        
        double de_x = o1.e_vec[0] - o2.e_vec[0];
        double de_y = o1.e_vec[1] - o2.e_vec[1];
        double de_z = o1.e_vec[2] - o2.e_vec[2];
        double de_mag = Math.Sqrt(de_x * de_x + de_y * de_y + de_z * de_z);
        
        double v_avg = Math.Sqrt(mu / ((a1 + a2) / 2));
        double dv_ecc = de_mag * v_avg * 0.5;
        
        return dv_hohmann + dv_plane + dv_ecc;
    }

    static void Main(string[] args)
    {
        List<string> lines = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            if (string.IsNullOrWhiteSpace(line)) continue;
            lines.Add(line.Trim());
        }

        if (lines.Count < 3) return;

        var firstLine = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        N = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1], System.Globalization.CultureInfo.InvariantCulture);

        Orbit startOrbit = ParseOrbit(lines[1], mu, N);
        Orbit[] stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            stations[i] = ParseOrbit(lines[i + 2], mu, i);
        }

        START_NODE = N;
        dist = new double[N + 1, N + 1];
        for (int i = 0; i < N; i++)
        {
            dist[START_NODE, i] = GetDistance(startOrbit, stations[i], mu);
            dist[i, START_NODE] = dist[START_NODE, i];
            for (int j = i + 1; j < N; j++)
            {
                dist[i, j] = GetDistance(stations[i], stations[j], mu);
                dist[j, i] = dist[i, j];
            }
        }

        int[] nnPath = new int[N + 1];
        nnPath[0] = START_NODE;
        bool[] visited = new bool[N];
        int curr = START_NODE;
        for (int i = 1; i <= N; i++)
        {
            double minDist = double.MaxValue;
            int next = -1;
            for (int j = 0; j < N; j++)
            {
                if (!visited[j] && dist[curr, j] < minDist)
                {
                    minDist = dist[curr, j];
                    next = j;
                }
            }
            nnPath[i] = next;
            visited[next] = true;
            curr = next;
        }

        int[] bestPath = (int[])nnPath.Clone();
        double bestCost = CalculateCost(bestPath);
        bestCost = Optimize(bestPath, bestCost);

        Random rng = new Random(42);
        Stopwatch sw = Stopwatch.StartNew();
        
        while (sw.ElapsedMilliseconds < 5000) // 5 seconds
        {
            int[] path = new int[N + 1];
            path[0] = START_NODE;
            for (int i = 0; i < N; i++) path[i + 1] = i;
            for (int i = N; i > 1; i--)
            {
                int r = rng.Next(1, i + 1);
                int temp = path[i];
                path[i] = path[r];
                path[r] = temp;
            }
            
            double cost = CalculateCost(path);
            cost = Optimize(path, cost);
            
            if (cost < bestCost)
            {
                bestCost = cost;
                Array.Copy(path, bestPath, N + 1);
            }
        }

        Console.WriteLine(string.Join(" ", bestPath.Skip(1)));
    }

    static double Optimize(int[] path, double currentCost)
    {
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 1; i <= N - 1; i++)
            {
                for (int j = i + 1; j <= N; j++)
                {
                    double delta = 0;
                    delta -= dist[path[i - 1], path[i]];
                    if (j < N) delta -= dist[path[j], path[j + 1]];
                    
                    delta += dist[path[i - 1], path[j]];
                    if (j < N) delta += dist[path[i], path[j + 1]];
                    
                    if (delta < -1e-7)
                    {
                        Reverse(path, i, j);
                        currentCost += delta;
                        improved = true;
                    }
                }
            }
        }
        return currentCost;
    }

    static void Reverse(int[] path, int i, int j)
    {
        while (i < j)
        {
            int temp = path[i];
            path[i] = path[j];
            path[j] = temp;
            i++;
            j--;
        }
    }

    static double CalculateCost(int[] path)
    {
        double cost = 0;
        for (int i = 0; i < N; i++)
        {
            cost += dist[path[i], path[i + 1]];
        }
        return cost;
    }
}
