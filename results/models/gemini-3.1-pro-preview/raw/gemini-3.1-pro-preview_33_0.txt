using System;
using System.IO;
using System.Runtime.CompilerServices;
using System.Diagnostics;

class Program
{
    static void Main()
    {
        Stopwatch sw = Stopwatch.StartNew();
        
        using (Stream s = Console.OpenStandardInput())
        {
            Scanner sc = new Scanner(s);
            int n = sc.NextInt();
            if (n <= 0) return;
            if (n == 1) 
            {
                int f0 = sc.NextInt();
                int d0 = sc.NextInt();
                Console.WriteLine((long)f0 * d0);
                Console.WriteLine("0");
                return;
            }
            
            int[][] flow = new int[n][];
            int[][] flow_t = new int[n][];
            for (int i = 0; i < n; i++)
            {
                flow[i] = new int[n];
                flow_t[i] = new int[n];
            }
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    flow[i][j] = sc.NextInt();
                }
            }
            
            int[][] dist = new int[n][];
            int[][] dist_t = new int[n][];
            for (int i = 0; i < n; i++)
            {
                dist[i] = new int[n];
                dist_t[i] = new int[n];
            }
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    dist[i][j] = sc.NextInt();
                }
            }
            
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    flow_t[j][i] = flow[i][j];
                    dist_t[j][i] = dist[i][j];
                }
            }
            
            int[] p = GenerateInitialSolution(n, flow, dist);
            long currentCost = CalculateTotalCost(p, flow, dist);
            long bestCost = currentCost;
            int[] bestP = (int[])p.Clone();
            
            int L = Math.Max(100, Math.Min(100000, n * n / 2));
            long[] f = new long[L];
            for (int i = 0; i < L; i++) f[i] = currentCost;
            
            int k = 0;
            FastRandom rnd = new FastRandom(42);
            long timeLimit = 290000;
            
            while (true)
            {
                if ((k & 16383) == 0) 
                {
                    if (sw.ElapsedMilliseconds >= timeLimit) break;
                }

                int u = rnd.Next(n);
                int v = rnd.Next(n);
                while (u == v)
                {
                    v = rnd.Next(n);
                }
                
                long delta = ComputeDelta(u, v, p, flow, flow_t, dist, dist_t);
                long newCost = currentCost + delta;
                
                int v_idx = k % L;
                if (newCost <= f[v_idx] || newCost <= currentCost)
                {
                    int temp = p[u];
                    p[u] = p[v];
                    p[v] = temp;
                    currentCost = newCost;
                    
                    if (currentCost < bestCost)
                    {
                        bestCost = currentCost;
                        Array.Copy(p, bestP, n);
                    }
                }
                f[v_idx] = currentCost;
                k++;
            }
            
            Console.WriteLine(bestCost);
            Console.WriteLine(string.Join(" ", bestP));
        }
    }
    
    static int[] GenerateInitialSolution(int n, int[][] flow, int[][] dist)
    {
        long[] flowSum = new long[n];
        long[] distSum = new long[n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                flowSum[i] += flow[i][j] + flow[j][i];
                distSum[i] += dist[i][j] + dist[j][i];
            }
        }

        int[] f_idx = new int[n];
        int[] d_idx = new int[n];
        for(int i=0; i<n; i++) { f_idx[i] = i; d_idx[i] = i; }

        Array.Sort(f_idx, (a, b) => flowSum[b].CompareTo(flowSum[a]));
        Array.Sort(d_idx, (a, b) => distSum[a].CompareTo(distSum[b]));

        int[] p = new int[n];
        for (int i = 0; i < n; i++)
        {
            p[f_idx[i]] = d_idx[i];
        }
        return p;
    }
    
    static long CalculateTotalCost(int[] p, int[][] flow, int[][] dist)
    {
        long cost = 0;
        int n = p.Length;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[i][j] * dist[p[i]][p[j]];
            }
        }
        return cost;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static long ComputeDelta(int u, int v, int[] p, int[][] flow, int[][] flow_t, int[][] dist, int[][] dist_t)
    {
        long delta = 0;
        int pu = p[u], pv = p[v];
        int n = p.Length;
        int[] flow_u = flow[u];
        int[] flow_v = flow[v];
        int[] flow_t_u = flow_t[u];
        int[] flow_t_v = flow_t[v];
        
        int[] dist_pu = dist[pu];
        int[] dist_pv = dist[pv];
        int[] dist_t_pu = dist_t[pu];
        int[] dist_t_pv = dist_t[pv];

        int min_uv = u < v ? u : v;
        int max_uv = u > v ? u : v;

        for (int i = 0; i < min_uv; i++)
        {
            int pi = p[i];
            int A = dist_pv[pi] - dist_pu[pi];
            int B = dist_t_pv[pi] - dist_t_pu[pi];
            delta += ((long)flow_u[i] - flow_v[i]) * A
                   + ((long)flow_t_u[i] - flow_t_v[i]) * B;
        }

        for (int i = min_uv + 1; i < max_uv; i++)
        {
            int pi = p[i];
            int A = dist_pv[pi] - dist_pu[pi];
            int B = dist_t_pv[pi] - dist_t_pu[pi];
            delta += ((long)flow_u[i] - flow_v[i]) * A
                   + ((long)flow_t_u[i] - flow_t_v[i]) * B;
        }

        for (int i = max_uv + 1; i < n; i++)
        {
            int pi = p[i];
            int A = dist_pv[pi] - dist_pu[pi];
            int B = dist_t_pv[pi] - dist_t_pu[pi];
            delta += ((long)flow_u[i] - flow_v[i]) * A
                   + ((long)flow_t_u[i] - flow_t_v[i]) * B;
        }
        
        int C = dist_pv[pv] - dist_pu[pu];
        int D = dist_pv[pu] - dist_pu[pv];
        delta += ((long)flow_u[u] - flow_v[v]) * C
               + ((long)flow_u[v] - flow_v[u]) * D;
               
        return delta;
    }
}

public class FastRandom
{
    private uint x, y, z, w;

    public FastRandom(uint seed)
    {
        x = seed;
        y = 362436069;
        z = 521288629;
        w = 88675123;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public int Next(int max)
    {
        uint t = x ^ (x << 11);
        x = y; y = z; z = w;
        w = (w ^ (w >> 19)) ^ (t ^ (t >> 8));
        return (int)((w * (ulong)max) >> 32);
    }
}

public class Scanner
{
    private Stream s;
    private byte[] buf = new byte[8192];
    private int head = 0;
    private int tail = 0;

    public Scanner(Stream s)
    {
        this.s = s;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public int NextInt()
    {
        int c = Read();
        while (c < '-') 
        {
            if (c == -1) return 0;
            c = Read();
        }
        bool sign = false;
        if (c == '-')
        {
            sign = true;
            c = Read();
        }
        int res = 0;
        while (c >= '0' && c <= '9')
        {
            res = res * 10 + c - '0';
            c = Read();
        }
        return sign ? -res : res;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private int Read()
    {
        if (head >= tail)
        {
            head = 0;
            tail = s.Read(buf, 0, buf.Length);
            if (tail <= 0) return -1;
        }
        return buf[head++];
    }
}
