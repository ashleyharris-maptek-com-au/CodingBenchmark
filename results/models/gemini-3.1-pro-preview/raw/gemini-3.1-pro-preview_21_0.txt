#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <tuple>
#include <cmath>
#include <algorithm>

int D;

struct Point {
    int coords[20];
    bool operator==(const Point& o) const {
        for(int i = 0; i < D; ++i) if(coords[i] != o.coords[i]) return false;
        return true;
    }
    bool operator!=(const Point& o) const { return !(*this == o); }
};

struct PointHash {
    size_t operator()(const Point& p) const {
        size_t res = 0;
        for(int i = 0; i < D; ++i) {
            res ^= std::hash<int>()(p.coords[i]) + 0x9e3779b9 + (res << 6) + (res >> 2);
        }
        return res;
    }
};

int max_turns;
std::vector<int> bounds;
int snake_length;
Point start_pos;
int F;
std::unordered_set<Point, PointHash> foods;
int O;
std::unordered_set<Point, PointHash> obstacles;

int growth_counter;
std::deque<Point> snake_body;

int get_h(const Point& p) {
    if (foods.empty()) return 0;
    int min_h = 1e9;
    for (const auto& f : foods) {
        int h = 0;
        for (int i = 0; i < D; ++i) {
            h += std::abs(p.coords[i] - f.coords[i]);
        }
        if (h < min_h) min_h = h;
    }
    return min_h;
}

struct AStarNode {
    Point p;
    int g;
    int f;
    bool operator>(const AStarNode& o) const {
        return f > o.f;
    }
};

std::vector<std::pair<int, int>> find_path() {
    std::unordered_map<Point, int, PointHash> old_body_block;
    for (int i = 0; i < (int)snake_body.size(); ++i) {
        int block_until = (int)snake_body.size() - 1 - i + growth_counter;
        if (old_body_block.find(snake_body[i]) == old_body_block.end()) {
            old_body_block[snake_body[i]] = block_until;
        } else {
            old_body_block[snake_body[i]] = std::max(old_body_block[snake_body[i]], block_until);
        }
    }

    std::priority_queue<AStarNode, std::vector<AStarNode>, std::greater<AStarNode>> pq;
    std::unordered_map<Point, int, PointHash> g_map;
    std::unordered_map<Point, std::tuple<Point, int, int>, PointHash> parent_map;

    Point start = snake_body.front();
    pq.push({start, 0, get_h(start)});
    g_map[start] = 0;

    Point target_found;
    bool found = false;

    while (!pq.empty()) {
        auto [curr, g, f] = pq.top();
        pq.pop();

        if (g_map[curr] < g) continue;

        if (foods.count(curr)) {
            target_found = curr;
            found = true;
            break;
        }

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Point next = curr;
                next.coords[axis] += dir;

                if (next.coords[axis] < 0 || next.coords[axis] >= bounds[axis]) continue;
                if (obstacles.count(next)) continue;
                if (g == 0 && snake_body.size() > 1 && next == snake_body[1]) continue;

                int next_g = g + 1;
                if (old_body_block.count(next) && next_g <= old_body_block[next]) continue;

                if (g_map.find(next) == g_map.end() || next_g < g_map[next]) {
                    g_map[next] = next_g;
                    parent_map[next] = {curr, axis, dir};
                    pq.push({next, next_g, next_g + get_h(next)});
                }
            }
        }
    }

    std::vector<std::pair<int, int>> path;
    if (found) {
        Point curr = target_found;
        while (curr != start) {
            auto [p, axis, dir] = parent_map[curr];
            path.push_back({axis, dir});
            curr = p;
        }
        std::reverse(path.begin(), path.end());
    }
    return path;
}

int evaluate_survival_move(Point start_p, const std::unordered_map<Point, int, PointHash>& old_body_block) {
    std::unordered_set<Point, PointHash> visited;
    std::queue<std::pair<Point, int>> q;
    q.push({start_p, 1});
    visited.insert(start_p);
    
    int score = 0;
    while (!q.empty() && score < 1000) {
        auto [curr, dist] = q.front();
        q.pop();
        score++;
        
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Point next = curr;
                next.coords[axis] += dir;

                if (next.coords[axis] < 0 || next.coords[axis] >= bounds[axis]) continue;
                if (obstacles.count(next)) continue;
                if (old_body_block.count(next) && dist + 1 <= old_body_block.at(next)) continue;

                if (visited.find(next) == visited.end()) {
                    visited.insert(next);
                    q.push({next, dist + 1});
                }
            }
        }
    }
    return score;
}

std::pair<int, int> find_survival_move() {
    std::unordered_map<Point, int, PointHash> old_body_block;
    for (int i = 0; i < (int)snake_body.size(); ++i) {
        int block_until = (int)snake_body.size() - 1 - i + growth_counter;
        if (old_body_block.find(snake_body[i]) == old_body_block.end()) {
            old_body_block[snake_body[i]] = block_until;
        } else {
            old_body_block[snake_body[i]] = std::max(old_body_block[snake_body[i]], block_until);
        }
    }

    std::pair<int, int> best_move = {0, 1};
    int max_score = -1;

    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Point next = snake_body.front();
            next.coords[axis] += dir;

            if (next.coords[axis] < 0 || next.coords[axis] >= bounds[axis]) continue;
            if (obstacles.count(next)) continue;
            if (snake_body.size() > 1 && next == snake_body[1]) continue;

            if (old_body_block.count(next) && 1 <= old_body_block[next]) continue;

            int score = evaluate_survival_move(next, old_body_block);
            if (score > max_score) {
                max_score = score;
                best_move = {axis, dir};
            }
        }
    }
    return best_move;
}

void update_game_state(std::pair<int, int> move) {
    int axis = move.first;
    int dir = move.second;
    
    Point next_head = snake_body.front();
    next_head.coords[axis] += dir;
    
    snake_body.push_front(next_head);
    
    if (foods.count(next_head)) {
        foods.erase(next_head);
        growth_counter++;
    }
    
    if (growth_counter > 0) {
        growth_counter--;
    } else {
        snake_body.pop_back();
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    if (!(std::cin >> D >> max_turns)) return 0;

    bounds.resize(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }

    std::cin >> snake_length;
    for (int i = 0; i < D; ++i) {
        std::cin >> start_pos.coords[i];
    }

    std::cin >> F;
    for (int i = 0; i < F; ++i) {
        Point p;
        for (int j = 0; j < D; ++j) {
            std::cin >> p.coords[j];
        }
        foods.insert(p);
    }

    std::cin >> O;
    for (int i = 0; i < O; ++i) {
        Point p;
        for (int j = 0; j < D; ++j) {
            std::cin >> p.coords[j];
        }
        obstacles.insert(p);
    }

    growth_counter = snake_length - 1;
    snake_body.push_back(start_pos);

    std::vector<std::pair<int, int>> current_plan;

    int turn = 0;
    while (turn < max_turns && !foods.empty()) {
        if (current_plan.empty()) {
            current_plan = find_path();
            if (current_plan.empty()) {
                current_plan.push_back(find_survival_move());
            }
        }
        
        auto move = current_plan.front();
        current_plan.erase(current_plan.begin());
        
        std::cout << move.first << " " << move.second << "\n";
        
        update_game_state(move);
        turn++;
    }

    return 0;
}
