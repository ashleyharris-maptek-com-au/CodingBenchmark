; SPIR-V
; Version: 1.0
OpCapability Shader
%ext_inst = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo_var = OpVariable %ptr_ubo Uniform
%ptr_in_vec3 = OpTypePointer Input %vec3
%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_normal = OpVariable %ptr_in_vec3 Input
%ptr_out_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_out_vec4 Output
%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%ptr_ubo_vec4 = OpTypePointer Uniform %vec4

%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_75 = OpConstant %float 0.75
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4

%cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4

%main = OpFunction %void None %func
%label = OpLabel

%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal

%lightPos_ptr = OpAccessChain %ptr_ubo_vec4 %ubo_var %int_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

%lightDir_unnorm = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %ext_inst Normalize %lightDir_unnorm
%N = OpExtInst %vec3 %ext_inst Normalize %normal

%dot_NL = OpDot %float %N %L
%t_mul = OpFMul %float %dot_NL %float_0_5
%t = OpFAdd %float %t_mul %float_0_5
%t_vec3 = OpCompositeConstruct %vec3 %t %t %t

%mixed = OpExtInst %vec3 %ext_inst FMix %cool %warm %t_vec3

%mixed_x = OpCompositeExtract %float %mixed 0
%mixed_y = OpCompositeExtract %float %mixed 1
%mixed_z = OpCompositeExtract %float %mixed 2
%out_color_val = OpCompositeConstruct %vec4 %mixed_x %mixed_y %mixed_z %float_1

OpStore %outColor %out_color_val

OpReturn
OpFunctionEnd
