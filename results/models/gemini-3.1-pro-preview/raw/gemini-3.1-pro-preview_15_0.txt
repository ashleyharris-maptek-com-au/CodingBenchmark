use std::collections::{HashSet, VecDeque};
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, BufRead};

#[derive(Default)]
pub struct FxHasher {
    hash: u64,
}

impl Hasher for FxHasher {
    fn finish(&self) -> u64 {
        self.hash
    }

    fn write(&mut self, bytes: &[u8]) {
        for &byte in bytes {
            self.hash = self.hash.rotate_left(5).bitxor(byte as u64).wrapping_mul(0x517cc1b727220a95);
        }
    }
}

type FastHashSet<T> = HashSet<T, BuildHasherDefault<FxHasher>>;

fn get_rotation(s_dir: (f64, f64, f64)) -> (f64, f64, f64, f64) {
    let z = (0.0, 0.0, 1.0);
    let dot = z.0 * s_dir.0 + z.1 * s_dir.1 + z.2 * s_dir.2;
    if dot > 0.999999 {
        return (1.0, 0.0, 0.0, 0.0);
    } else if dot < -0.999999 {
        return (0.0, 1.0, 0.0, 0.0);
    }
    let cross = (
        z.1 * s_dir.2 - z.2 * s_dir.1,
        z.2 * s_dir.0 - z.0 * s_dir.2,
        z.0 * s_dir.1 - z.1 * s_dir.0,
    );
    let qw = 1.0 + dot;
    let qx = cross.0;
    let qy = cross.1;
    let qz = cross.2;
    let len = (qw * qw + qx * qx + qy * qy + qz * qz).sqrt();
    (qw / len, qx / len, qy / len, qz / len)
}

fn rotate_vector(q: (f64, f64, f64, f64), v: (f64, f64, f64)) -> (f64, f64, f64) {
    let (qw, qx, qy, qz) = q;
    let (vx, vy, vz) = v;
    let u = (qx, qy, qz);
    let cross1 = (
        u.1 * vz - u.2 * vy,
        u.2 * vx - u.0 * vz,
        u.0 * vy - u.1 * vx,
    );
    let cross2 = (
        u.1 * cross1.2 - u.2 * cross1.1,
        u.2 * cross1.0 - u.0 * cross1.2,
        u.0 * cross1.1 - u.1 * cross1.0,
    );
    (
        vx + 2.0 * qw * cross1.0 + 2.0 * cross2.0,
        vy + 2.0 * qw * cross1.1 + 2.0 * cross2.1,
        vz + 2.0 * qw * cross1.2 + 2.0 * cross2.2,
    )
}

fn project(p: (f64, f64, f64), s: (f64, f64, f64)) -> (f64, f64) {
    let t = -p.2 / s.2;
    (p.0 + t * s.0, p.1 + t * s.1)
}

fn world_to_grid(x: f64, y: f64, t1: (f64, f64), t2: (f64, f64), det: f64) -> (f64, f64) {
    let i = (x * t2.1 - y * t2.0) / det;
    let j = (t1.0 * y - t1.1 * x) / det;
    (i, j)
}

fn point_in_polygon(px: f64, py: f64, poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        
        let intersect = ((yi > py) != (yj > py))
            && (px < (xj - xi) * (py - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn ccw(ax: f64, ay: f64, bx: f64, by: f64, cx: f64, cy: f64) -> f64 {
    (cy - ay) * (bx - ax) - (by - ay) * (cx - ax)
}

fn segments_intersect(ax: f64, ay: f64, bx: f64, by: f64, cx: f64, cy: f64, dx: f64, dy: f64) -> bool {
    let cp1 = ccw(ax, ay, cx, cy, dx, dy);
    let cp2 = ccw(bx, by, cx, cy, dx, dy);
    let cp3 = ccw(cx, cy, ax, ay, bx, by);
    let cp4 = ccw(dx, dy, ax, ay, bx, by);

    if cp1 * cp2 < 0.0 && cp3 * cp4 < 0.0 {
        return true;
    }
    
    fn on_segment(px: f64, py: f64, ax: f64, ay: f64, bx: f64, by: f64) -> bool {
        let min_x = ax.min(bx) - 1e-6;
        let max_x = ax.max(bx) + 1e-6;
        let min_y = ay.min(by) - 1e-6;
        let max_y = ay.max(by) + 1e-6;
        px >= min_x && px <= max_x && py >= min_y && py <= max_y
    }
    
    if cp1.abs() < 1e-6 && on_segment(ax, ay, cx, cy, dx, dy) { return true; }
    if cp2.abs() < 1e-6 && on_segment(bx, by, cx, cy, dx, dy) { return true; }
    if cp3.abs() < 1e-6 && on_segment(cx, cy, ax, ay, bx, by) { return true; }
    if cp4.abs() < 1e-6 && on_segment(dx, dy, ax, ay, bx, by) { return true; }

    false
}

fn cell_intersects_polygon(cx: i64, cy: i64, poly: &[(f64, f64)]) -> bool {
    let eps = 1e-5;
    let min_x = cx as f64 - 0.5 - eps;
    let max_x = cx as f64 + 0.5 + eps;
    let min_y = cy as f64 - 0.5 - eps;
    let max_y = cy as f64 + 0.5 + eps;
    
    let corners = [
        (min_x, min_y),
        (max_x, min_y),
        (max_x, max_y),
        (min_x, max_y),
    ];
    
    for &(px, py) in &corners {
        if point_in_polygon(px, py, poly) {
            return true;
        }
    }
    
    for &(px, py) in poly {
        if px >= min_x && px <= max_x && py >= min_y && py <= max_y {
            return true;
        }
    }
    
    let cell_edges = [
        (corners[0], corners[1]),
        (corners[1], corners[2]),
        (corners[2], corners[3]),
        (corners[3], corners[0]),
    ];
    
    for i in 0..poly.len() {
        let p1 = poly[i];
        let p2 = poly[(i + 1) % poly.len()];
        for &(c1, c2) in &cell_edges {
            if segments_intersect(p1.0, p1.1, p2.0, p2.1, c1.0, c1.1, c2.0, c2.1) {
                return true;
            }
        }
    }
    
    false
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(|l| l.ok()).filter(|l| !l.trim().is_empty());
    
    let nv_line = match lines.next() {
        Some(l) => l,
        None => return,
    };
    let nv: usize = nv_line.trim().parse().unwrap();
    
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().unwrap();
        let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        poly.push((parts[0], parts[1]));
    }
    
    let sun_line = lines.next().unwrap();
    let sun_parts: Vec<f64> = sun_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let mut s_dir = (sun_parts[0], sun_parts[1], sun_parts[2]);
    let len = (s_dir.0 * s_dir.0 + s_dir.1 * s_dir.1 + s_dir.2 * s_dir.2).sqrt();
    s_dir = (s_dir.0 / len, s_dir.1 / len, s_dir.2 / len);
    
    let q = get_rotation(s_dir);
    let scale = 10.0;
    
    let u = rotate_vector(q, (scale, 0.0, 0.0));
    let v = rotate_vector(q, (0.0, scale, 0.0));
    
    let u_proj = project(u, s_dir);
    let v_proj = project(v, s_dir);
    
    let t1 = (u_proj.0 + v_proj.0, u_proj.1 + v_proj.1);
    let t2 = (u_proj.0 - v_proj.0, u_proj.1 - v_proj.1);
    let det = t1.0 * t2.1 - t1.1 * t2.0;
    
    let mut grid_poly = Vec::with_capacity(nv);
    for &(x, y) in &poly {
        grid_poly.push(world_to_grid(x, y, t1, t2, det));
    }
    
    let mut queue = VecDeque::new();
    let mut visited = FastHashSet::default();
    let mut result_cells = Vec::new();
    
    for &(px, py) in &grid_poly {
        let start_cell = (px.round() as i64, py.round() as i64);
        if !visited.contains(&start_cell) {
            queue.push_back(start_cell);
            visited.insert(start_cell);
        }
    }
    
    while let Some(cell) = queue.pop_front() {
        if cell_intersects_polygon(cell.0, cell.1, &grid_poly) {
            result_cells.push(cell);
            for di in -1..=1 {
                for dj in -1..=1 {
                    if di == 0 && dj == 0 { continue; }
                    let neighbor = (cell.0 + di, cell.1 + dj);
                    if !visited.contains(&neighbor) {
                        visited.insert(neighbor);
                        queue.push_back(neighbor);
                    }
                }
            }
        }
    }
    
    println!("{}", result_cells.len());
    let d_shift = 40.0 / s_dir.2.abs().max(1e-6);
    
    for (idx, &(i, j)) in result_cells.iter().enumerate() {
        let x_base = i as f64 * t1.0 + j as f64 * t2.0;
        let y_base = i as f64 * t1.1 + j as f64 * t2.1;
        
        let shift = idx as f64 * d_shift;
        let px = x_base - shift * s_dir.0;
        let py = y_base - shift * s_dir.1;
        let pz = 0.0 - shift * s_dir.2;
        
        println!("{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.1}", px, py, pz, q.0, q.1, q.2, q.3, scale);
    }
}
