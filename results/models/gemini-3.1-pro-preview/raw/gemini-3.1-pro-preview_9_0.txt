using System;
using System.IO;
using System.Collections.Generic;
using System.Runtime.CompilerServices;

class Program
{
    static void Main()
    {
        using var stream = Console.OpenStandardInput();
        var scanner = new FastScanner(stream);
        
        int n = scanner.ReadInt();
        if (n == -1) return;
        if (n == 0)
        {
            Console.WriteLine(0);
            return;
        }
        long stockLength = scanner.ReadLong();
        
        long[] lengths = new long[n];
        int[] indices = new int[n];
        
        for (int i = 0; i < n; i++)
        {
            lengths[i] = scanner.ReadLong();
            indices[i] = i;
        }
        
        Array.Sort(lengths, indices, new DescendingComparer());
        
        long[] tree = new long[4 * n];
        Array.Fill(tree, stockLength);
        
        int[] assignedBin = new int[n];
        int maxBin = -1;
        
        for (int i = 0; i < n; i++)
        {
            if (tree[1] < lengths[i])
            {
                throw new Exception("Item exceeds stock length or no bins available.");
            }
            int binIndex = QueryAndUpdate(1, 0, n - 1, lengths[i], tree);
            assignedBin[indices[i]] = binIndex;
            if (binIndex > maxBin)
            {
                maxBin = binIndex;
            }
        }
        
        int numStocks = maxBin + 1;
        
        int[] binCounts = new int[numStocks];
        for (int i = 0; i < n; i++)
        {
            binCounts[assignedBin[i]]++;
        }
        
        int[][] bins = new int[numStocks][];
        for (int i = 0; i < numStocks; i++)
        {
            bins[i] = new int[binCounts[i]];
        }
        
        int[] binPointers = new int[numStocks];
        for (int i = 0; i < n; i++)
        {
            int b = assignedBin[i];
            bins[b][binPointers[b]++] = i;
        }
        
        using var outStream = Console.OpenStandardOutput();
        using var writer = new FastWriter(outStream);
        
        writer.Write(numStocks);
        writer.WriteLine();
        for (int i = 0; i < numStocks; i++)
        {
            var bin = bins[i];
            for (int j = 0; j < bin.Length; j++)
            {
                writer.Write(bin[j]);
                if (j < bin.Length - 1)
                {
                    writer.WriteSpace();
                }
            }
            writer.WriteLine();
        }
    }
    
    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static int QueryAndUpdate(int node, int l, int r, long val, long[] tree)
    {
        if (l == r)
        {
            tree[node] -= val;
            return l;
        }
        int mid = l + (r - l) / 2;
        int res;
        if (tree[2 * node] >= val)
        {
            res = QueryAndUpdate(2 * node, l, mid, val, tree);
        }
        else
        {
            res = QueryAndUpdate(2 * node + 1, mid + 1, r, val, tree);
        }
        tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
        return res;
    }
}

readonly struct DescendingComparer : IComparer<long>
{
    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public int Compare(long x, long y)
    {
        return y.CompareTo(x);
    }
}

class FastScanner
{
    private Stream _stream;
    private byte[] _buffer = new byte[1024 * 1024];
    private int _head;
    private int _tail;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public int ReadInt()
    {
        int c = Read();
        while (c <= 32)
        {
            if (c == -1) return -1;
            c = Read();
        }
        int res = 0;
        while (c > 32)
        {
            res = res * 10 + c - '0';
            c = Read();
        }
        return res;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public long ReadLong()
    {
        int c = Read();
        while (c <= 32)
        {
            if (c == -1) return -1;
            c = Read();
        }
        long res = 0;
        while (c > 32)
        {
            res = res * 10 + c - '0';
            c = Read();
        }
        return res;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private int Read()
    {
        if (_head >= _tail)
        {
            _head = 0;
            _tail = _stream.Read(_buffer, 0, _buffer.Length);
            if (_tail == 0) return -1;
        }
        return _buffer[_head++];
    }
}

class FastWriter : IDisposable
{
    private Stream _stream;
    private byte[] _buffer;
    private int _pos;

    public FastWriter(Stream stream, int bufferSize = 1024 * 1024)
    {
        _stream = stream;
        _buffer = new byte[bufferSize];
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void Write(int value)
    {
        if (value == 0)
        {
            WriteChar('0');
            return;
        }

        if (value < 0)
        {
            WriteChar('-');
            value = -value;
        }

        int temp = value;
        int numDigits = 0;
        while (temp > 0)
        {
            temp /= 10;
            numDigits++;
        }

        if (_pos + numDigits > _buffer.Length)
        {
            Flush();
        }

        int p = _pos + numDigits - 1;
        while (value > 0)
        {
            _buffer[p--] = (byte)('0' + (value % 10));
            value /= 10;
        }
        _pos += numDigits;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteChar(char c)
    {
        if (_pos == _buffer.Length) Flush();
        _buffer[_pos++] = (byte)c;
    }
    
    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteLine() => WriteChar('\n');
    
    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteSpace() => WriteChar(' ');

    public void Flush()
    {
        if (_pos > 0)
        {
            _stream.Write(_buffer, 0, _pos);
            _pos = 0;
        }
    }

    public void Dispose()
    {
        Flush();
    }
}
