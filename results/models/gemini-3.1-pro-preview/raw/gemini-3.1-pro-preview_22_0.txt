Here is the complete, compilable C# code for the optimal Tetris player. 

It implements an adaptive strategy based on Pierre Dellacherie's algorithm. For standard heights, it uses a slightly tweaked "Aggressive" strategy that attempts to leave the rightmost column open to score 4-line Tetrises. When the stack reaches above 50% of the board height, it switches to a "Marathon" strategy with standard Dellacherie weights, which prioritizes indefinite survival and minimizing holes/transitions to weather long games.

```csharp
using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    public struct Point
    {
        public int x, y;
        public Point(int x, int y) { this.x = x; this.y = y; }
    }

    public class Piece
    {
        public string Name;
        public Point[][] Rotations;
        public int[][] Bottoms;
        public int[] Widths;
    }

    static Dictionary<string, Piece> Pieces = new Dictionary<string, Piece>();

    static Piece MakePiece(string name, Point[][] rots)
    {
        Piece p = new Piece { Name = name, Rotations = rots };
        p.Widths = new int[4];
        p.Bottoms = new int[4][];
        for (int i = 0; i < 4; i++)
        {
            int max_x = 0;
            foreach (var pt in rots[i]) if (pt.x > max_x) max_x = pt.x;
            p.Widths[i] = max_x + 1;
            p.Bottoms[i] = new int[p.Widths[i]];
            for (int x = 0; x < p.Widths[i]; x++)
            {
                int min_y = 999;
                foreach (var pt in rots[i])
                {
                    if (pt.x == x && pt.y < min_y) min_y = pt.y;
                }
                p.Bottoms[i][x] = min_y;
            }
        }
        return p;
    }

    static void InitPieces()
    {
        Pieces["I"] = MakePiece("I", new Point[][] {
            new Point[] { new (0,0), new (1,0), new (2,0), new (3,0) },
            new Point[] { new (0,0), new (0,1), new (0,2), new (0,3) },
            new Point[] { new (0,0), new (1,0), new (2,0), new (3,0) },
            new Point[] { new (0,0), new (0,1), new (0,2), new (0,3) }
        });
        Pieces["O"] = MakePiece("O", new Point[][] {
            new Point[] { new (0,0), new (1,0), new (0,1), new (1,1) },
            new Point[] { new (0,0), new (1,0), new (0,1), new (1,1) },
            new Point[] { new (0,0), new (1,0), new (0,1), new (1,1) },
            new Point[] { new (0,0), new (1,0), new (0,1), new (1,1) }
        });
        Pieces["T"] = MakePiece("T", new Point[][] {
            new Point[] { new (0,0), new (1,0), new (2,0), new (1,1) },
            new Point[] { new (0,0), new (0,1), new (0,2), new (1,1) },
            new Point[] { new (1,0), new (0,1), new (1,1), new (2,1) },
            new Point[] { new (1,0), new (0,1), new (1,1), new (1,2) }
        });
        Pieces["S"] = MakePiece("S", new Point[][] {
            new Point[] { new (0,0), new (1,0), new (1,1), new (2,1) },
            new Point[] { new (1,0), new (0,1), new (1,1), new (0,2) },
            new Point[] { new (0,0), new (1,0), new (1,1), new (2,1) },
            new Point[] { new (1,0), new (0,1), new (1,1), new (0,2) }
        });
        Pieces["Z"] = MakePiece("Z", new Point[][] {
            new Point[] { new (1,0), new (2,0), new (0,1), new (1,1) },
            new Point[] { new (0,0), new (0,1), new (1,1), new (1,2) },
            new Point[] { new (1,0), new (2,0), new (0,1), new (1,1) },
            new Point[] { new (0,0), new (0,1), new (1,1), new (1,2) }
        });
        Pieces["J"] = MakePiece("J", new Point[][] {
            new Point[] { new (0,0), new (1,0), new (2,0), new (0,1) },
            new Point[] { new (0,0), new (0,1), new (0,2), new (1,2) },
            new Point[] { new (2,0), new (0,1), new (1,1), new (2,1) },
            new Point[] { new (0,0), new (1,0), new (1,1), new (1,2) }
        });
        Pieces["L"] = MakePiece("L", new Point[][] {
            new Point[] { new (0,0), new (1,0), new (2,0), new (2,1) },
            new Point[] { new (0,0), new (1,0), new (0,1), new (0,2) },
            new Point[] { new (0,0), new (0,1), new (1,1), new (2,1) },
            new Point[] { new (1,0), new (1,1), new (0,2), new (1,2) }
        });
    }

    public class BoardState
    {
        public bool[,] grid;
        public int width, height;
        public int[] columnHeights;

        public BoardState(int w, int h)
        {
            width = w;
            height = h;
            grid = new bool[w, h];
            columnHeights = new int[w];
        }
    }

    struct MoveResult
    {
        public BoardState board;
        public double score;
        public bool valid;
    }

    static MoveResult EvaluateMove(BoardState board, Piece p, int rot, int col, bool aggressive)
    {
        int dropY = 0;
        int[] bottoms = p.Bottoms[rot];

        for (int dx = 0; dx < p.Widths[rot]; dx++)
        {
            int x = col + dx;
            int y = board.columnHeights[x] - bottoms[dx];
            if (y > dropY) dropY = y;
        }

        Point[] blocks = p.Rotations[rot];
        bool[,] newGrid = new bool[board.width, board.height];
        Array.Copy(board.grid, newGrid, board.grid.Length);

        bool gameover = false;
        foreach (var pt in blocks)
        {
            int x = col + pt.x;
            int y = dropY + pt.y;
            if (y >= board.height)
            {
                gameover = true;
            }
            else
            {
                newGrid[x, y] = true;
            }
        }

        if (gameover) return new MoveResult { valid = false, score = -1e9 };

        int cleared = 0;
        int pieceCellsEliminated = 0;
        bool[] lineCleared = new bool[board.height];

        for (int y = dropY; y < dropY + 4 && y < board.height; y++)
        {
            bool full = true;
            for (int x = 0; x < board.width; x++)
            {
                if (!newGrid[x, y])
                {
                    full = false;
                    break;
                }
            }
            if (full)
            {
                cleared++;
                lineCleared[y] = true;
            }
        }

        bool[,] finalGrid;
        int[] finalHeights = new int[board.width];

        if (cleared > 0)
        {
            foreach (var pt in blocks)
            {
                int y = dropY + pt.y;
                if (y < board.height && lineCleared[y])
                {
                    pieceCellsEliminated++;
                }
            }

            finalGrid = new bool[board.width, board.height];
            for (int x = 0; x < board.width; x++)
            {
                int shift = 0;
                int highest = 0;
                for (int y = 0; y < board.height; y++)
                {
                    if (lineCleared[y])
                    {
                        shift++;
                    }
                    else
                    {
                        if (newGrid[x, y])
                        {
                            finalGrid[x, y - shift] = true;
                            highest = y - shift + 1;
                        }
                    }
                }
                finalHeights[x] = highest;
            }
        }
        else
        {
            finalGrid = newGrid;
            for (int x = 0; x < board.width; x++)
            {
                int highest = board.columnHeights[x];
                for (int y = board.height - 1; y >= 0; y--)
                {
                    if (finalGrid[x, y])
                    {
                        highest = y + 1;
                        break;
                    }
                }
                finalHeights[x] = highest;
            }
        }

        int rowTrans = 0, colTrans = 0, holes = 0, wells = 0;
        int maxHeight = 0;
        for (int x = 0; x < board.width; x++)
        {
            if (finalHeights[x] > maxHeight) maxHeight = finalHeights[x];
        }

        for (int y = 0; y < maxHeight; y++)
        {
            bool last = true;
            for (int x = 0; x < board.width; x++)
            {
                bool current = finalGrid[x, y];
                if (current != last) rowTrans++;
                last = current;
            }
            if (!last) rowTrans++;
        }

        for (int x = 0; x < board.width; x++)
        {
            bool last = true;
            for (int y = 0; y <= finalHeights[x]; y++)
            {
                bool current = y < board.height ? finalGrid[x, y] : false;
                if (y == finalHeights[x]) current = false;
                if (current != last) colTrans++;
                last = current;
            }

            for (int y = 0; y < finalHeights[x]; y++)
            {
                if (!finalGrid[x, y]) holes++;
            }
        }

        for (int x = 0; x < board.width; x++)
        {
            if (aggressive && x == board.width - 1) continue;
            int wellDepth = 0;
            for (int y = maxHeight; y >= 0; y--)
            {
                if (y >= board.height) continue;
                bool left = (x == 0) || finalGrid[x - 1, y];
                bool right = (x == board.width - 1) || finalGrid[x + 1, y];
                bool current = finalGrid[x, y];

                if (!current && left && right)
                {
                    wellDepth++;
                    wells += wellDepth;
                }
                else
                {
                    wellDepth = 0;
                }
            }
        }

        double score = 0;
        if (aggressive)
        {
            score += -4.5 * dropY;
            score += (cleared * 3.4) + (cleared == 4 ? 10.0 : 0.0);
            score += -3.2 * rowTrans;
            score += -9.3 * colTrans;
            score += -7.9 * holes;
            score += -3.4 * wells;
        }
        else
        {
            // Standard Dellacherie Weights
            score += -4.500158 * dropY;
            score += 3.418126 * (cleared * pieceCellsEliminated);
            score += -3.217888 * rowTrans;
            score += -9.348695 * colTrans;
            score += -7.899265 * holes;
            score += -3.385597 * wells;
        }

        BoardState nextBoard = new BoardState(board.width, board.height);
        nextBoard.grid = finalGrid;
        nextBoard.columnHeights = finalHeights;
        return new MoveResult { valid = true, score = score, board = nextBoard };
    }

    static string ReadNextToken()
    {
        int c;
        while ((c = Console.Read()) != -1 && char.IsWhiteSpace((char)c)) ;
        if (c == -1) return null;
        StringBuilder sb = new StringBuilder();
        sb.Append((char)c);
        while ((c = Console.Read()) != -1 && !char.IsWhiteSpace((char)c))
        {
            sb.Append((char)c);
        }
        return sb.ToString();
    }

    static void Main(string[] args)
    {
        InitPieces();

        string w_str = ReadNextToken();
        if (w_str == null) return;
        string h_str = ReadNextToken();
        string n_str = ReadNextToken();

        int width = int.Parse(w_str);
        int height = int.Parse(h_str);
        int num_pieces = int.Parse(n_str);

        BoardState board = new BoardState(width, height);

        for (int i = 0; i < num_pieces; i++)
        {
            string pName = null;
            while (true)
            {
                string token = ReadNextToken();
                if (token == null) return;
                if (Pieces.ContainsKey(token))
                {
                    pName = token;
                    break;
                }
            }

            bool aggressive = false;
            int maxHeight = 0;
            for (int x = 0; x < board.width; x++)
            {
                if (board.columnHeights[x] > maxHeight) maxHeight = board.columnHeights[x];
            }
            if (maxHeight < board.height * 0.5)
            {
                aggressive = true;
            }

            Piece p = Pieces[pName];
            double bestScore = -double.MaxValue;
            int bestRot = 0;
            int bestCol = 0;
            BoardState bestBoard = null;

            for (int rot = 0; rot < 4; rot++)
            {
                int pieceWidth = p.Widths[rot];
                for (int col = 0; col <= board.width - pieceWidth; col++)
                {
                    var result = EvaluateMove(board, p, rot, col, aggressive);
                    if (result.valid)
                    {
                        double tieBreaker = (col * 0.0001) - (rot * 0.00001);
                        double finalScore = result.score + tieBreaker;

                        if (finalScore > bestScore)
                        {
                            bestScore = finalScore;
                            bestRot = rot;
                            bestCol = col;
                            bestBoard = result.board;
                        }
                    }
                }
            }

            if (bestBoard == null)
            {
                Console.WriteLine("0 0");
                Console.Out.Flush();
            }
            else
            {
                Console.WriteLine($"{bestRot} {bestCol}");
                Console.Out.Flush();
                board = bestBoard;
            }

            string res = null;
            while (true)
            {
                string token = ReadNextToken();
                if (token == null) return;
                if (token == "ok" || token.StartsWith("cleared_") || token == "gameover")
                {
                    res = token;
                    break;
                }
            }
            if (res == "gameover") break;
        }
    }
}
```
