using System;
using System.Collections.Generic;
using System.IO;

class DirectedFeedbackVertexSet {
    static void Main() {
        var scanner = new Scanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == -1) return;
        int m = scanner.NextInt();

        var G = new Graph();
        for (int i = 0; i < n; i++) G.AddVertex(i);

        for (int i = 0; i < m; i++) {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            G.AddEdge(u, v);
        }

        var fvs = Solve(G, n);
        
        if (fvs == null) {
            Console.WriteLine(0);
            return;
        }

        Console.WriteLine(fvs.Count);
        Console.WriteLine(string.Join(" ", fvs));
    }

    static List<int> Solve(Graph G, int limit) {
        List<int> currentFvs = new List<int>();

        ReduceInPlace(G, currentFvs, limit);

        if (currentFvs.Count > limit) return null;

        if (G.Vertices.Count == 0) return currentFvs;

        var sccs = G.GetSCCs();
        if (sccs.Count > 1) {
            foreach (var scc in sccs) {
                if (scc.Count == 1) continue;
                var subG = G.GetSubgraph(scc);
                var subLimit = limit - currentFvs.Count;
                var subFvs = Solve(subG, subLimit);
                if (subFvs == null) return null;
                currentFvs.AddRange(subFvs);
            }
            return currentFvs;
        }

        int branchV = -1;
        long maxScore = -1;
        foreach(var v in G.Vertices) {
            long score = (long)G.InEdges[v].Count * G.OutEdges[v].Count;
            if (score > maxScore) {
                maxScore = score;
                branchV = v;
            }
        }

        List<int> bestRemaining = null;

        if (limit >= currentFvs.Count + 1) {
            var G1 = G.Clone();
            G1.RemoveVertex(branchV);
            var fvs1 = Solve(G1, limit - currentFvs.Count - 1);
            
            if (fvs1 != null) {
                bestRemaining = fvs1;
                bestRemaining.Add(branchV);
                limit = currentFvs.Count + bestRemaining.Count - 1; 
            }
        }

        if (limit >= currentFvs.Count) {
            var G2 = G.Clone();
            G2.Bypass(branchV);
            var fvs2 = Solve(G2, limit - currentFvs.Count);
            
            if (fvs2 != null) {
                bestRemaining = fvs2;
            }
        }

        if (bestRemaining == null) return null;

        currentFvs.AddRange(bestRemaining);
        return currentFvs;
    }

    static bool ReduceInPlace(Graph G, List<int> fvs, int limit) {
        var qLoop = new Queue<int>();
        var q0 = new Queue<int>();
        var q1 = new Queue<int>();

        foreach (var v in G.Vertices) {
            if (G.OutEdges[v].Contains(v)) qLoop.Enqueue(v);
            else if (G.InEdges[v].Count == 0 || G.OutEdges[v].Count == 0) q0.Enqueue(v);
            else if (G.InEdges[v].Count == 1 || G.OutEdges[v].Count == 1) q1.Enqueue(v);
        }

        bool reducedAny = false;

        while (qLoop.Count > 0 || q0.Count > 0 || q1.Count > 0) {
            if (qLoop.Count > 0) {
                int v = qLoop.Dequeue();
                if (!G.Vertices.Contains(v)) continue;
                if (!G.OutEdges[v].Contains(v)) continue;
                
                fvs.Add(v);
                if (fvs.Count > limit) return true;
                
                var affected = G.GetNeighbors(v);
                G.RemoveVertex(v);
                reducedAny = true;
                
                foreach (var n in affected) CheckVertex(G, n, qLoop, q0, q1);
                continue;
            }

            if (q0.Count > 0) {
                int v = q0.Dequeue();
                if (!G.Vertices.Contains(v)) continue;
                if (G.InEdges[v].Count > 0 && G.OutEdges[v].Count > 0) continue;

                var affected = G.GetNeighbors(v);
                G.RemoveVertex(v);
                reducedAny = true;
                
                foreach (var n in affected) CheckVertex(G, n, qLoop, q0, q1);
                continue;
            }

            if (q1.Count > 0) {
                int v = q1.Dequeue();
                if (!G.Vertices.Contains(v)) continue;
                if (G.InEdges[v].Count != 1 && G.OutEdges[v].Count != 1) continue;
                if (G.InEdges[v].Count == 0 || G.OutEdges[v].Count == 0) {
                    q0.Enqueue(v);
                    continue;
                }
                if (G.OutEdges[v].Contains(v)) {
                    qLoop.Enqueue(v);
                    continue;
                }

                var affected = G.GetNeighbors(v);
                G.Bypass(v);
                reducedAny = true;
                
                foreach (var n in affected) CheckVertex(G, n, qLoop, q0, q1);
                continue;
            }
        }
        return reducedAny;
    }

    static void CheckVertex(Graph G, int v, Queue<int> qLoop, Queue<int> q0, Queue<int> q1) {
        if (!G.Vertices.Contains(v)) return;
        if (G.OutEdges[v].Contains(v)) { qLoop.Enqueue(v); return; }
        int inDeg = G.InEdges[v].Count;
        int outDeg = G.OutEdges[v].Count;
        if (inDeg == 0 || outDeg == 0) q0.Enqueue(v);
        else if (inDeg == 1 || outDeg == 1) q1.Enqueue(v);
    }
}

class Graph {
    public HashSet<int> Vertices = new HashSet<int>();
    public Dictionary<int, HashSet<int>> OutEdges = new Dictionary<int, HashSet<int>>();
    public Dictionary<int, HashSet<int>> InEdges = new Dictionary<int, HashSet<int>>();

    public void AddVertex(int v) {
        if (Vertices.Add(v)) {
            OutEdges[v] = new HashSet<int>();
            InEdges[v] = new HashSet<int>();
        }
    }

    public void AddEdge(int u, int v) {
        AddVertex(u);
        AddVertex(v);
        OutEdges[u].Add(v);
        InEdges[v].Add(u);
    }

    public void RemoveVertex(int v) {
        if (!Vertices.Contains(v)) return;
        
        foreach (var u in InEdges[v]) {
            OutEdges[u].Remove(v);
        }
        foreach (var w in OutEdges[v]) {
            InEdges[w].Remove(v);
        }
        InEdges.Remove(v);
        OutEdges.Remove(v);
        Vertices.Remove(v);
    }

    public void Bypass(int v) {
        var inN = new List<int>(InEdges[v]);
        var outN = new List<int>(OutEdges[v]);
        
        foreach (var u in inN) {
            foreach (var w in outN) {
                if (u != v && w != v) {
                    OutEdges[u].Add(w);
                    InEdges[w].Add(u);
                }
            }
        }
        RemoveVertex(v);
    }

    public HashSet<int> GetNeighbors(int v) {
        var set = new HashSet<int>();
        foreach(var u in InEdges[v]) set.Add(u);
        foreach(var w in OutEdges[v]) set.Add(w);
        set.Remove(v);
        return set;
    }

    public Graph Clone() {
        var g = new Graph();
        g.Vertices = new HashSet<int>(this.Vertices);
        g.OutEdges = new Dictionary<int, HashSet<int>>(this.Vertices.Count);
        g.InEdges = new Dictionary<int, HashSet<int>>(this.Vertices.Count);
        foreach (var v in Vertices) {
            g.OutEdges[v] = new HashSet<int>(this.OutEdges[v]);
            g.InEdges[v] = new HashSet<int>(this.InEdges[v]);
        }
        return g;
    }

    public Graph GetSubgraph(List<int> scc) {
        var g = new Graph();
        var set = new HashSet<int>(scc);
        foreach (var v in scc) {
            g.Vertices.Add(v);
            g.OutEdges[v] = new HashSet<int>();
            g.InEdges[v] = new HashSet<int>();
        }
        foreach (var v in scc) {
            foreach (var w in OutEdges[v]) {
                if (set.Contains(w)) {
                    g.OutEdges[v].Add(w);
                    g.InEdges[w].Add(v);
                }
            }
        }
        return g;
    }

    public List<List<int>> GetSCCs() {
        var order = new List<int>();
        var visited = new HashSet<int>();
        
        foreach (var startNode in Vertices) {
            if (!visited.Contains(startNode)) {
                var stack = new Stack<(int node, IEnumerator<int> enumerator)>();
                visited.Add(startNode);
                stack.Push((startNode, OutEdges[startNode].GetEnumerator()));
                
                while (stack.Count > 0) {
                    var top = stack.Pop();
                    int u = top.node;
                    var enumerator = top.enumerator;
                    
                    bool pushed = false;
                    while (enumerator.MoveNext()) {
                        int v = enumerator.Current;
                        if (!visited.Contains(v)) {
                            visited.Add(v);
                            stack.Push((u, enumerator));
                            stack.Push((v, OutEdges[v].GetEnumerator()));
                            pushed = true;
                            break;
                        }
                    }
                    
                    if (!pushed) {
                        order.Add(u);
                        enumerator.Dispose();
                    }
                }
            }
        }

        order.Reverse();
        visited.Clear();
        var sccs = new List<List<int>>();

        foreach (var startNode in order) {
            if (!visited.Contains(startNode)) {
                var scc = new List<int>();
                var sccStack = new Stack<int>();
                
                visited.Add(startNode);
                sccStack.Push(startNode);
                
                while (sccStack.Count > 0) {
                    int u = sccStack.Pop();
                    scc.Add(u);
                    
                    foreach (var v in InEdges[u]) {
                        if (!visited.Contains(v)) {
                            visited.Add(v);
                            sccStack.Push(v);
                        }
                    }
                }
                sccs.Add(scc);
            }
        }

        return sccs;
    }
}

class Scanner {
    Stream stream;
    byte[] buffer = new byte[1024 * 16];
    int head = 0;
    int tail = 0;

    public Scanner(Stream s) { stream = s; }

    public int NextInt() {
        int c = Read();
        while (c < '0' || c > '9') {
            if (c == -1) return -1;
            c = Read();
        }
        int res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }

    int Read() {
        if (head >= tail) {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}
