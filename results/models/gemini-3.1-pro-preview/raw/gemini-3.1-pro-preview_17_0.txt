using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;

namespace BinPacking3D {
    class Program {
        struct Box {
            public int id;
            public int w, h, d;
            public long Volume => (long)w * h * d;
        }

        struct Space {
            public int minX, minY, minZ;
            public int maxX, maxY, maxZ;
            public int W => maxX - minX;
            public int H => maxY - minY;
            public int D => maxZ - minZ;
            public long Volume => (long)W * H * D;
        }

        struct Vector3 {
            public int X, Y, Z;
            public Vector3(int x, int y, int z) { X = x; Y = y; Z = z; }
        }

        struct PlacedBox {
            public int id;
            public int x, y, z;
            public int w, h, d;
        }

        class Scanner {
            private StreamReader reader;
            public Scanner() {
                reader = new StreamReader(Console.OpenStandardInput(), System.Text.Encoding.ASCII, false, 65536);
            }
            public int NextInt() {
                int c = reader.Read();
                while (c != -1 && (c < '0' || c > '9')) {
                    c = reader.Read();
                }
                if (c == -1) return -1;
                int res = 0;
                while (c >= '0' && c <= '9') {
                    res = res * 10 + (c - '0');
                    c = reader.Read();
                }
                return res;
            }
        }

        static Vector3[] GetOrientations(int w, int h, int d) {
            Vector3[] arr = new Vector3[6];
            arr[0] = new Vector3(w, h, d);
            arr[1] = new Vector3(w, d, h);
            arr[2] = new Vector3(h, w, d);
            arr[3] = new Vector3(h, d, w);
            arr[4] = new Vector3(d, w, h);
            arr[5] = new Vector3(d, h, w);
            
            int count = 0;
            for (int i = 0; i < 6; i++) {
                bool duplicate = false;
                for (int j = 0; j < count; j++) {
                    if (arr[i].X == arr[j].X && arr[i].Y == arr[j].Y && arr[i].Z == arr[j].Z) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) {
                    arr[count++] = arr[i];
                }
            }
            Vector3[] result = new Vector3[count];
            Array.Copy(arr, result, count);
            return result;
        }

        static bool Overlaps(Space S, PlacedBox B) {
            return S.minX < B.x + B.w && S.maxX > B.x &&
                   S.minY < B.y + B.h && S.maxY > B.y &&
                   S.minZ < B.z + B.d && S.maxZ > B.z;
        }

        static void SplitSpace(Space S, PlacedBox B, List<Space> new_spaces) {
            int B_minX = B.x, B_maxX = B.x + B.w;
            int B_minY = B.y, B_maxY = B.y + B.h;
            int B_minZ = B.z, B_maxZ = B.z + B.d;

            if (B_minX > S.minX) new_spaces.Add(new Space { minX = S.minX, maxX = B_minX, minY = S.minY, maxY = S.maxY, minZ = S.minZ, maxZ = S.maxZ });
            if (B_maxX < S.maxX) new_spaces.Add(new Space { minX = B_maxX, maxX = S.maxX, minY = S.minY, maxY = S.maxY, minZ = S.minZ, maxZ = S.maxZ });
            if (B_minY > S.minY) new_spaces.Add(new Space { minX = S.minX, maxX = S.maxX, minY = S.minY, maxY = B_minY, minZ = S.minZ, maxZ = S.maxZ });
            if (B_maxY < S.maxY) new_spaces.Add(new Space { minX = S.minX, maxX = S.maxX, minY = B_maxY, maxY = S.maxY, minZ = S.minZ, maxZ = S.maxZ });
            if (B_minZ > S.minZ) new_spaces.Add(new Space { minX = S.minX, maxX = S.maxX, minY = S.minY, maxY = S.maxY, minZ = S.minZ, maxZ = B_minZ });
            if (B_maxZ < S.maxZ) new_spaces.Add(new Space { minX = S.minX, maxX = S.maxX, minY = S.minY, maxY = S.maxY, minZ = B_maxZ, maxZ = S.maxZ });
        }

        static void Main() {
            Stopwatch sw = Stopwatch.StartNew();
            Scanner scanner = new Scanner();
            
            int N = scanner.NextInt();
            if (N == -1) return;
            int W_container = scanner.NextInt();
            int H_container = scanner.NextInt();
            int D_container = scanner.NextInt();

            Box[] boxes = new Box[N];
            for (int i = 0; i < N; i++) {
                boxes[i] = new Box {
                    id = i,
                    w = scanner.NextInt(),
                    h = scanner.NextInt(),
                    d = scanner.NextInt()
                };
            }

            Array.Sort(boxes, (a, b) => a.Volume.CompareTo(b.Volume));

            List<Space> spaces = new List<Space>();
            spaces.Add(new Space { minX = 0, minY = 0, minZ = 0, maxX = W_container, maxY = H_container, maxZ = D_container });

            List<PlacedBox> placed = new List<PlacedBox>(N);

            int MAX_SPACES = 2000;

            for (int i = 0; i < N; i++) {
                if (sw.ElapsedMilliseconds > 28000) break;

                Box box = boxes[i];
                Vector3[] orientations = GetOrientations(box.w, box.h, box.d);

                Space? bestSpace = null;
                Vector3? bestOri = null;

                long best_minZ = long.MaxValue;
                long best_minY = long.MaxValue;
                long best_minX = long.MaxValue;
                long best_oriZ = long.MaxValue;
                long best_oriY = long.MaxValue;
                long best_oriX = long.MaxValue;
                long best_vol = long.MaxValue;

                for (int sIdx = 0; sIdx < spaces.Count; sIdx++) {
                    Space space = spaces[sIdx];
                    if (space.Volume < box.Volume) continue;
                    if ((space.W < box.w && space.W < box.h && space.W < box.d) ||
                        (space.H < box.w && space.H < box.h && space.H < box.d) ||
                        (space.D < box.w && space.D < box.h && space.D < box.d)) 
                        continue;

                    for (int oIdx = 0; oIdx < orientations.Length; oIdx++) {
                        Vector3 ori = orientations[oIdx];
                        if (ori.X <= space.W && ori.Y <= space.H && ori.Z <= space.D) {
                            bool update = false;
                            if (bestSpace == null) {
                                update = true;
                            } else {
                                if (space.minZ < best_minZ) update = true;
                                else if (space.minZ == best_minZ) {
                                    if (space.minY < best_minY) update = true;
                                    else if (space.minY == best_minY) {
                                        if (space.minX < best_minX) update = true;
                                        else if (space.minX == best_minX) {
                                            if (ori.Z < best_oriZ) update = true;
                                            else if (ori.Z == best_oriZ) {
                                                if (ori.Y < best_oriY) update = true;
                                                else if (ori.Y == best_oriY) {
                                                    if (ori.X < best_oriX) update = true;
                                                    else if (ori.X == best_oriX) {
                                                        if (space.Volume < best_vol) update = true;
                                                    }
                                                }
                                            }
                                        }
                                    }
                                }
                            }

                            if (update) {
                                bestSpace = space;
                                bestOri = ori;
                                best_minZ = space.minZ;
                                best_minY = space.minY;
                                best_minX = space.minX;
                                best_oriZ = ori.Z;
                                best_oriY = ori.Y;
                                best_oriX = ori.X;
                                best_vol = space.Volume;
                            }
                        }
                    }
                }

                if (bestSpace != null) {
                    PlacedBox pb = new PlacedBox {
                        id = box.id,
                        x = bestSpace.Value.minX,
                        y = bestSpace.Value.minY,
                        z = bestSpace.Value.minZ,
                        w = bestOri.Value.X,
                        h = bestOri.Value.Y,
                        d = bestOri.Value.Z
                    };
                    placed.Add(pb);

                    List<Space> survivors = new List<Space>();
                    List<Space> intersectors = new List<Space>();

                    for (int sIdx = 0; sIdx < spaces.Count; sIdx++) {
                        Space space = spaces[sIdx];
                        if (Overlaps(space, pb)) {
                            intersectors.Add(space);
                        } else {
                            survivors.Add(space);
                        }
                    }

                    List<Space> new_spaces = new List<Space>();
                    for (int sIdx = 0; sIdx < intersectors.Count; sIdx++) {
                        SplitSpace(intersectors[sIdx], pb, new_spaces);
                    }

                    new_spaces.Sort((a, b) => b.Volume.CompareTo(a.Volume));
                    List<Space> filtered_new = new List<Space>();
                    for (int j = 0; j < new_spaces.Count; j++) {
                        Space N_space = new_spaces[j];
                        bool subsumed = false;
                        for (int k = 0; k < filtered_new.Count; k++) {
                            Space O_space = filtered_new[k];
                            if (N_space.minX >= O_space.minX && N_space.maxX <= O_space.maxX &&
                                N_space.minY >= O_space.minY && N_space.maxY <= O_space.maxY &&
                                N_space.minZ >= O_space.minZ && N_space.maxZ <= O_space.maxZ) {
                                subsumed = true;
                                break;
                            }
                        }
                        if (!subsumed) {
                            filtered_new.Add(N_space);
                        }
                    }

                    List<Space> final_spaces = new List<Space>(survivors.Count + filtered_new.Count);
                    final_spaces.AddRange(survivors);
                    
                    for (int j = 0; j < filtered_new.Count; j++) {
                        Space N_space = filtered_new[j];
                        bool subsumed = false;
                        for (int k = 0; k < survivors.Count; k++) {
                            Space O_space = survivors[k];
                            if (N_space.minX >= O_space.minX && N_space.maxX <= O_space.maxX &&
                                N_space.minY >= O_space.minY && N_space.maxY <= O_space.maxY &&
                                N_space.minZ >= O_space.minZ && N_space.maxZ <= O_space.maxZ) {
                                subsumed = true;
                                break;
                            }
                        }
                        if (!subsumed) {
                            final_spaces.Add(N_space);
                        }
                    }

                    long nextVol = (i + 1 < N) ? boxes[i+1].Volume : 0;
                    spaces.Clear();
                    for (int j = 0; j < final_spaces.Count; j++) {
                        if (final_spaces[j].Volume >= nextVol) {
                            spaces.Add(final_spaces[j]);
                        }
                    }

                    if (spaces.Count > MAX_SPACES) {
                        spaces.Sort((a, b) => b.Volume.CompareTo(a.Volume));
                        spaces.RemoveRange(MAX_SPACES, spaces.Count - MAX_SPACES);
                    }
                }
            }

            using (var writer = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 65536)) {
                writer.WriteLine(placed.Count);
                for (int i = 0; i < placed.Count; i++) {
                    writer.WriteLine($"{placed[i].id} {placed[i].x} {placed[i].y} {placed[i].z}");
                }
            }
        }
    }
}
