import sys

def solve(X, Y, solution=None):
    if solution is None:
        solution = []
    if not X:
        yield list(solution)
    else:
        # Choose the column with the minimum number of covering rows
        # Tie-break with the column ID to ensure determinism
        c = min(X.keys(), key=lambda c: (len(X[c]), c))
        
        # Iterate over sorted rows for determinism
        for r in sorted(list(X[c])):
            solution.append(r)
            cols = select(X, Y, r)
            for s in solve(X, Y, solution):
                yield s
            deselect(X, Y, r, cols)
            solution.pop()

def select(X, Y, r):
    cols = []
    for j in Y[r]:
        for i in X[j]:
            for k in Y[i]:
                if k != j:
                    X[k].remove(i)
        cols.append(X.pop(j))
    return cols

def deselect(X, Y, r, cols):
    for j in reversed(Y[r]):
        X[j] = cols.pop()
        for i in X[j]:
            for k in Y[i]:
                if k != j:
                    X[k].add(i)

def main():
    input_data = sys.stdin.read().splitlines()
    if not input_data:
        return
        
    input_data = [line.strip() for line in input_data if line.strip()]
    if not input_data:
        return
        
    first_line = input_data[0].split()
    U = int(first_line[0])
    S = int(first_line[1])
    
    X = {j: set() for j in range(U)}
    Y = {i: [] for i in range(S)}
    
    for i in range(S):
        if i + 1 < len(input_data):
            # Parse elements, ensure uniqueness, and sort
            elements = sorted(list(set(int(x) for x in input_data[i+1].split())))
            for e in elements:
                if 0 <= e < U:
                    X[e].add(i)
                    Y[i].append(e)
                
    try:
        sol = sorted(next(solve(X, Y)))
        print("SOLUTION")
        print(" ".join(map(str, sol)))
    except StopIteration:
        print("NO SOLUTION")

if __name__ == '__main__':
    main()
