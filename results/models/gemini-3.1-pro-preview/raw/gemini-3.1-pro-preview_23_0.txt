use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;
use std::f64::consts::PI;

#[derive(Clone, Copy, Debug, Default)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
}

fn next_word(stdin: &mut impl BufRead) -> Option<String> {
    let mut word = String::new();
    let mut in_word = false;
    loop {
        let buf = stdin.fill_buf().ok()?;
        if buf.is_empty() {
            break;
        }
        let mut consumed = 0;
        for &b in buf {
            if b.is_ascii_whitespace() {
                if in_word {
                    break;
                }
            } else {
                in_word = true;
                word.push(b as char);
            }
            consumed += 1;
        }
        stdin.consume(consumed);
        if in_word && consumed < buf.len() {
            break;
        }
    }
    if word.is_empty() { None } else { Some(word) }
}

fn main() {
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();

    let width: usize = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let height: usize = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let gravity: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let max_thrust: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let _fuel: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let _start_x: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let _start_y: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let target_x: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();
    let target_y: f64 = next_word(&mut stdin_lock).unwrap().parse().unwrap();

    // 0 means no obstacle. We store `height - y` so higher values = higher altitude.
    // This avoids `memset` delay for gigantic maps because vec![0; N] uses calloc.
    let mut highest_obstacle_alt = vec![0usize; width];

    let mut map_idx = 0;
    let total_cells = width * height;
    
    // Parse the map quickly
    loop {
        let buf = stdin_lock.fill_buf().unwrap();
        if buf.is_empty() || map_idx == total_cells {
            break;
        }
        let mut consumed = 0;
        for &b in buf {
            if b == b'.' || b == b'#' || b == b'S' || b == b'T' || b == b'L' {
                if b == b'#' || b == b'T' || b == b'L' {
                    let x = map_idx % width;
                    let y = map_idx / width;
                    let alt = height - y;
                    if alt > highest_obstacle_alt[x] {
                        highest_obstacle_alt[x] = alt;
                    }
                }
                map_idx += 1;
            }
            consumed += 1;
            if map_idx == total_cells {
                break;
            }
        }
        stdin_lock.consume(consumed);
        if map_idx == total_cells {
            break; // ensure we break out
        }
    }

    let shared_state = Arc::new(Mutex::new(None));
    let col_clone = Arc::new(highest_obstacle_alt);
    let state_clone = Arc::clone(&shared_state);

    thread::spawn(move || {
        let out = io::stdout();
        let mut out_lock = out.lock();
        let mut landing_phase = false;

        loop {
            thread::sleep(Duration::from_millis(25)); // 40Hz output

            let state_opt = {
                let lock = state_clone.lock().unwrap();
                *lock
            };

            if let Some(state) = state_opt {
                let padding = 8;
                let min_x_idx = (state.x.min(target_x) as isize - padding).max(0) as usize;
                let max_x_idx = (state.x.max(target_x) as isize + padding).min(width as isize - 1) as usize;

                let mut max_alt = 0;
                for i in min_x_idx..=max_x_idx {
                    if col_clone[i] > max_alt {
                        max_alt = col_clone[i];
                    }
                }

                let highest_obstacle_y = if max_alt == 0 { height as f64 } else { (height - max_alt) as f64 };
                let safe_y = (highest_obstacle_y - 20.0).max(5.0);

                let dist_x = (target_x - state.x).abs();
                if dist_x < 2.0 && state.y <= safe_y + 10.0 {
                    landing_phase = true;
                }
                if dist_x > 8.0 {
                    landing_phase = false;
                }

                let mut base_safe_y = safe_y.min(target_y - 30.0);
                if base_safe_y < 5.0 {
                    base_safe_y = 5.0;
                }

                let wp_x = target_x;
                let wp_y = if landing_phase {
                    target_y
                } else {
                    base_safe_y
                };

                let dx = wp_x - state.x;
                let dy = wp_y - state.y;
                let dist = (dx.powi(2) + dy.powi(2)).sqrt();

                let max_speed = if landing_phase {
                    if dist < 8.0 {
                        1.5
                    } else {
                        (dist * 0.2).clamp(2.5, 10.0)
                    }
                } else {
                    (dist * 0.3).clamp(2.0, 35.0)
                };

                let speed_factor = if dist > 0.1 { max_speed / dist } else { 0.0 };
                let vx_target = dx * speed_factor;
                let vy_target = dy * speed_factor;

                let e_vx = vx_target - state.vx;
                let e_vy = vy_target - state.vy;

                let k_p_v = 1.5;
                let a_x_req = e_vx * k_p_v;
                let a_y_req = e_vy * k_p_v;

                let t_req_x = a_x_req;
                let t_req_y = a_y_req - gravity;

                let theta_req = t_req_x.atan2(-t_req_y).clamp(-1.5, 1.5);

                let mut e_theta = (theta_req - state.angle) % (2.0 * PI);
                if e_theta > PI { e_theta -= 2.0 * PI; }
                if e_theta < -PI { e_theta += 2.0 * PI; }

                let turn_p = 4.0;
                let turn_d = 1.5;
                let turn_cmd = (e_theta * turn_p - state.angular_vel * turn_d).clamp(-1.0, 1.0);

                let heading_x = state.angle.sin();
                let heading_y = -state.angle.cos();
                let dot = t_req_x * heading_x + t_req_y * heading_y;

                let mut thrust_cmd = 0.0;
                if dot > 0.0 {
                    thrust_cmd = (dot / max_thrust).clamp(0.0, 1.0);
                }

                // Emergency override: prevent crashing into ground underneath us
                let my_col = state.x.clamp(0.0, width as f64 - 1.0) as usize;
                let alt_below = col_clone[my_col];
                let ground_y = if alt_below == 0 { height as f64 } else { (height - alt_below) as f64 };
                let alt = ground_y - state.y;
                if alt < 15.0 && state.vy > 5.0 {
                    thrust_cmd = 1.0;
                }

                if let Err(_) = writeln!(out_lock, "{:.4} {:.4}", thrust_cmd, turn_cmd) {
                    break;
                }
                let _ = out_lock.flush();
            } else {
                let hover = (gravity / max_thrust).clamp(0.0, 1.0);
                if let Err(_) = writeln!(out_lock, "{:.4} 0.0", hover) {
                    break;
                }
                let _ = out_lock.flush();
            }
        }
    });

    // Read continuous state updates
    loop {
        let x_str = match next_word(&mut stdin_lock) { Some(s) => s, None => break };
        let x: f64 = x_str.parse().unwrap_or(0.0);
        let y: f64 = match next_word(&mut stdin_lock) { Some(s) => s.parse().unwrap_or(0.0), None => break };
        let vx: f64 = match next_word(&mut stdin_lock) { Some(s) => s.parse().unwrap_or(0.0), None => break };
        let vy: f64 = match next_word(&mut stdin_lock) { Some(s) => s.parse().unwrap_or(0.0), None => break };
        let angle: f64 = match next_word(&mut stdin_lock) { Some(s) => s.parse().unwrap_or(0.0), None => break };
        let angular_vel: f64 = match next_word(&mut stdin_lock) { Some(s) => s.parse().unwrap_or(0.0), None => break };
        let _fuel: f64 = match next_word(&mut stdin_lock) { Some(s) => s.parse().unwrap_or(0.0), None => break };

        let mut lock = shared_state.lock().unwrap();
        *lock = Some(LanderState { x, y, vx, vy, angle, angular_vel });
    }
}
