use std::io::{self, BufWriter, Read, Write};

struct Scanner<R: Read> {
    reader: R,
    buffer: Vec<u8>,
    pos: usize,
    len: usize,
}

impl<R: Read> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buffer: vec![0; 1 << 16],
            pos: 0,
            len: 0,
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let mut res = 0;
        let mut found = false;
        loop {
            if self.pos >= self.len {
                self.len = self.reader.read(&mut self.buffer).unwrap_or(0);
                self.pos = 0;
                if self.len == 0 {
                    return if found { Some(res) } else { None };
                }
            }
            let b = self.buffer[self.pos];
            self.pos += 1;
            if b >= b'0' && b <= b'9' {
                res = res * 10 + (b - b'0') as usize;
                found = true;
            } else if found {
                return Some(res);
            }
        }
    }
}

fn add_to_bucket(v: usize, d: usize, bucket_head: &mut [i32], prev: &mut [i32], next: &mut [i32]) {
    let head_val = bucket_head[d];
    next[v] = head_val;
    prev[v] = -1;
    if head_val != -1 {
        prev[head_val as usize] = v as i32;
    }
    bucket_head[d] = v as i32;
}

fn remove_from_bucket(v: usize, d: usize, bucket_head: &mut [i32], prev: &mut [i32], next: &mut [i32]) {
    let p = prev[v];
    let n = next[v];
    if p != -1 {
        next[p as usize] = n;
    } else {
        bucket_head[d] = n;
    }
    if n != -1 {
        prev[n as usize] = p;
    }
}

fn are_adjacent(u: usize, w: usize, head: &[usize], edges: &[usize]) -> bool {
    let len_u = head[u+1] - head[u];
    let len_w = head[w+1] - head[w];
    let (search_in, target) = if len_u < len_w {
        (&edges[head[u]..head[u+1]], w)
    } else {
        (&edges[head[w]..head[w+1]], u)
    };
    search_in.binary_search(&target).is_ok()
}

fn main() {
    let stdin = io::stdin();
    let mut scanner = Scanner::new(stdin.lock());

    let num_vertices = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let num_edges = scanner.next_usize().unwrap_or(0);

    let mut must_include = Vec::new();
    let mut input_edges = Vec::with_capacity(num_edges);
    for _ in 0..num_edges {
        let u = scanner.next_usize().unwrap();
        let v = scanner.next_usize().unwrap();
        if u == v {
            must_include.push(u);
        } else {
            input_edges.push((u, v));
        }
    }

    let mut head = vec![0; num_vertices + 1];
    for &(u, v) in &input_edges {
        head[u + 1] += 1;
        head[v + 1] += 1;
    }
    for i in 1..=num_vertices {
        head[i] += head[i - 1];
    }
    let mut edges = vec![0; head[num_vertices]];
    let mut cur_head = head.clone();
    for &(u, v) in &input_edges {
        edges[cur_head[u]] = v;
        cur_head[u] += 1;
        edges[cur_head[v]] = u;
        cur_head[v] += 1;
    }

    let mut new_edges = Vec::with_capacity(edges.len());
    let mut new_head = vec![0; num_vertices + 1];
    for i in 0..num_vertices {
        let start = head[i];
        let end = head[i + 1];
        edges[start..end].sort_unstable();

        let mut edges_added = 0;
        if start < end {
            new_edges.push(edges[start]);
            edges_added += 1;
            for j in start + 1..end {
                if edges[j] != edges[j - 1] {
                    new_edges.push(edges[j]);
                    edges_added += 1;
                }
            }
        }
        new_head[i + 1] = new_head[i] + edges_added;
    }
    let head = new_head;
    let edges = new_edges;

    let mut bucket_head = vec![-1; num_vertices + 1];
    let mut prev = vec![-1; num_vertices];
    let mut next = vec![-1; num_vertices];
    let mut degree = vec![0; num_vertices];
    let mut in_cover = vec![false; num_vertices];

    let mut max_degree = 0;

    for i in 0..num_vertices {
        let d = head[i + 1] - head[i];
        degree[i] = d;
        if d > max_degree {
            max_degree = d;
        }
        add_to_bucket(i, d, &mut bucket_head, &mut prev, &mut next);
    }

    let mut cover_order = Vec::new();
    let mut degree_1_queue = Vec::new();
    let mut degree_2_queue = Vec::new();

    macro_rules! add_to_cover {
        ($u:expr) => {
            {
                let u = $u;
                in_cover[u] = true;
                cover_order.push(u);
                remove_from_bucket(u, degree[u], &mut bucket_head, &mut prev, &mut next);
                for i in head[u]..head[u+1] {
                    let y = edges[i];
                    if !in_cover[y] {
                        let d = degree[y];
                        remove_from_bucket(y, d, &mut bucket_head, &mut prev, &mut next);
                        degree[y] -= 1;
                        add_to_bucket(y, d - 1, &mut bucket_head, &mut prev, &mut next);
                        if degree[y] == 1 {
                            degree_1_queue.push(y);
                        } else if degree[y] == 2 {
                            degree_2_queue.push(y);
                        }
                    }
                }
            }
        }
    }

    let mut is_must_include = vec![false; num_vertices];
    for &u in &must_include {
        is_must_include[u] = true;
        if !in_cover[u] {
            add_to_cover!(u);
        }
    }

    for i in 0..num_vertices {
        let d = degree[i];
        if !in_cover[i] {
            if d == 1 {
                degree_1_queue.push(i);
            } else if d == 2 {
                degree_2_queue.push(i);
            }
        }
    }

    loop {
        let mut reduced = false;

        while let Some(v) = degree_1_queue.pop() {
            if !in_cover[v] && degree[v] == 1 {
                let mut w = usize::MAX;
                for i in head[v]..head[v+1] {
                    let neighbor = edges[i];
                    if !in_cover[neighbor] {
                        w = neighbor;
                        break;
                    }
                }
                if w != usize::MAX {
                    add_to_cover!(w);
                    reduced = true;
                }
            }
        }

        if reduced { continue; }

        let mut applied_deg2 = false;
        while let Some(v) = degree_2_queue.pop() {
            if !in_cover[v] && degree[v] == 2 {
                let mut u = usize::MAX;
                let mut w = usize::MAX;
                for i in head[v]..head[v+1] {
                    let neighbor = edges[i];
                    if !in_cover[neighbor] {
                        if u == usize::MAX {
                            u = neighbor;
                        } else {
                            w = neighbor;
                            break;
                        }
                    }
                }
                if u != usize::MAX && w != usize::MAX {
                    if are_adjacent(u, w, &head, &edges) {
                        add_to_cover!(u);
                        add_to_cover!(w);
                        applied_deg2 = true;
                        break;
                    }
                }
            }
        }

        if applied_deg2 {
            continue;
        }

        while max_degree > 0 && bucket_head[max_degree] == -1 {
            max_degree -= 1;
        }

        if max_degree == 0 {
            break;
        }

        let u = bucket_head[max_degree] as usize;
        add_to_cover!(u);
    }

    for &u in cover_order.iter().rev() {
        if in_cover[u] {
            let mut redundant = true;
            for i in head[u]..head[u+1] {
                let v = edges[i];
                if !in_cover[v] {
                    redundant = false;
                    break;
                }
            }
            if redundant && !is_must_include[u] {
                in_cover[u] = false;
            }
        }
    }

    let mut final_cover = Vec::new();
    for i in 0..num_vertices {
        if in_cover[i] {
            final_cover.push(i);
        }
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    writeln!(writer, "{}", final_cover.len()).unwrap();
    for (i, &v) in final_cover.iter().enumerate() {
        if i > 0 {
            write!(writer, " ").unwrap();
        }
        write!(writer, "{}", v).unwrap();
    }
    writeln!(writer).unwrap();
}
