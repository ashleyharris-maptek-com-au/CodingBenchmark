using System;
using System.IO;
using System.Collections.Generic;

class Program {
    static void SkipFast(long count, Stream stream, byte[] buffer, ref int bufLen, ref int bufPos) {
        while (count > 0) {
            int avail = bufLen - bufPos;
            if (avail == 0) {
                bufLen = stream.Read(buffer, 0, buffer.Length);
                bufPos = 0;
                if (bufLen == 0) return;
                avail = bufLen;
            }
            long toSkip = Math.Min(count, avail);
            bufPos += (int)toSkip;
            count -= toSkip;
        }
    }

    static void ReadExactly(byte[] dest, long offset, int count, Stream stream, byte[] buffer, ref int bufLen, ref int bufPos) {
        while (count > 0) {
            int avail = bufLen - bufPos;
            if (avail == 0) {
                bufLen = stream.Read(buffer, 0, buffer.Length);
                bufPos = 0;
                if (bufLen == 0) return;
                avail = bufLen;
            }
            int toRead = Math.Min(count, avail);
            Array.Copy(buffer, bufPos, dest, (int)offset, toRead);
            bufPos += toRead;
            offset += toRead;
            count -= toRead;
        }
    }

    static void Main() {
        var stopwatch = System.Diagnostics.Stopwatch.StartNew();
        
        Stream stdin = Console.OpenStandardInput();
        string header = "";
        int b;
        while ((b = stdin.ReadByte()) != -1) {
            char c = (char)b;
            if (c == '\n') break;
            header += c;
        }
        string[] parts = header.Trim().Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6) return;
        
        int XVoxelCount = int.Parse(parts[0]);
        int YVoxelCount = int.Parse(parts[1]);
        int ZVoxelCount = int.Parse(parts[2]);
        
        long xyArea = (long)XVoxelCount * YVoxelCount;
        int storeDepth = (int)Math.Min((long)ZVoxelCount, 2_000_000_000L / xyArea);
        if (storeDepth < 1) storeDepth = 1;
        
        byte[] topLayers = new byte[xyArea * storeDepth];
        
        byte[] buffer = new byte[1024 * 1024];
        int bufLen = 0;
        int bufPos = 0;
        
        for (int x = 0; x < XVoxelCount; x++) {
            for (int y = 0; y < YVoxelCount; y++) {
                long toSkip = ZVoxelCount - storeDepth;
                if (toSkip > 0) {
                    SkipFast(toSkip, stdin, buffer, ref bufLen, ref bufPos);
                }
                long offset = ((long)x * YVoxelCount + y) * storeDepth;
                ReadExactly(topLayers, offset, storeDepth, stdin, buffer, ref bufLen, ref bufPos);
            }
        }
        
        int[,] height = new int[XVoxelCount, YVoxelCount];
        for (int x = 0; x < XVoxelCount; x++) {
            for (int y = 0; y < YVoxelCount; y++) {
                height[x, y] = ZVoxelCount - 1;
            }
        }
        
        string tempFile = Path.GetTempFileName();
        int opCount = 0;
        
        using (var sw = new StreamWriter(new BufferedStream(new FileStream(tempFile, FileMode.Create), 65536))) {
            bool changed = true;
            // Leave 20 seconds for output flush
            while (changed && stopwatch.ElapsedMilliseconds < 280_000) {
                changed = false;
                for (int x = 0; x < XVoxelCount; x++) {
                    for (int y = 0; y < YVoxelCount; y++) {
                        int h = height[x, y];
                        if (h < 0) continue;
                        
                        int z_local = h - (ZVoxelCount - storeDepth);
                        if (z_local < 0) continue;
                        
                        if (x > 0 && height[x-1, y] > h) continue;
                        if (x < XVoxelCount - 1 && height[x+1, y] > h) continue;
                        if (y > 0 && height[x, y-1] > h) continue;
                        if (y < YVoxelCount - 1 && height[x, y+1] > h) continue;
                        
                        long offset = ((long)x * YVoxelCount + y) * storeDepth + z_local;
                        byte voxel = topLayers[(int)offset];
                        int mass = voxel % 16;
                        int value = voxel / 16;
                        
                        if (mass == 0 || value == 0) continue;
                        
                        List<int> path1 = new List<int>((x + y) * 2);
                        long cost1 = 0;
                        int currX = x; int currY = y; int currentH = h - 1;
                        
                        while(currX > 0) {
                            int nextX = currX - 1; int nextY = currY;
                            int nextH = height[nextX, nextY];
                            if (nextH > currentH) cost1 += 5 * mass;
                            else if (nextH == currentH) cost1 += 1 * mass;
                            path1.Add(nextX); path1.Add(nextY);
                            currX = nextX; currentH = nextH;
                        }
                        while(currY > 0) {
                            int nextX = currX; int nextY = currY - 1;
                            int nextH = height[nextX, nextY];
                            if (nextH > currentH) cost1 += 5 * mass;
                            else if (nextH == currentH) cost1 += 1 * mass;
                            path1.Add(nextX); path1.Add(nextY);
                            currX = nextX; currentH = nextH;
                        }
                        
                        List<int> path2 = new List<int>((x + y) * 2);
                        long cost2 = 0;
                        currX = x; currY = y; currentH = h - 1;
                        
                        while(currY > 0) {
                            int nextX = currX; int nextY = currY - 1;
                            int nextH = height[nextX, nextY];
                            if (nextH > currentH) cost2 += 5 * mass;
                            else if (nextH == currentH) cost2 += 1 * mass;
                            path2.Add(nextX); path2.Add(nextY);
                            currX = nextX; currentH = nextH;
                        }
                        while(currX > 0) {
                            int nextX = currX - 1; int nextY = currY;
                            int nextH = height[nextX, nextY];
                            if (nextH > currentH) cost2 += 5 * mass;
                            else if (nextH == currentH) cost2 += 1 * mass;
                            path2.Add(nextX); path2.Add(nextY);
                            currX = nextX; currentH = nextH;
                        }
                        
                        long revenue = 1000L * mass * value;
                        long bestCost = cost1 <= cost2 ? cost1 : cost2;
                        List<int> bestPath = cost1 <= cost2 ? path1 : path2;
                        
                        if (revenue > bestCost) {
                            height[x, y]--;
                            changed = true;
                            opCount++;
                            
                            sw.Write($"{bestPath.Count / 2} {x} {y} {h}");
                            for (int i = 0; i < bestPath.Count; i++) {
                                sw.Write($" {bestPath[i]}");
                            }
                            sw.Write(" 0 0\n");
                        }
                    }
                }
            }
        }
        
        Console.WriteLine(opCount);
        using (var sr = new StreamReader(new BufferedStream(new FileStream(tempFile, FileMode.Open), 65536))) {
            char[] buf = new char[65536];
            int read;
            while ((read = sr.Read(buf, 0, buf.Length)) > 0) {
                Console.Out.Write(buf, 0, read);
            }
        }
        
        try { File.Delete(tempFile); } catch {}
    }
}
