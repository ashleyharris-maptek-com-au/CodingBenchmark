#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <queue>
#include <chrono>
#include <random>

using namespace std;

int greedy_LDF(int V, const vector<vector<int>>& adj, vector<int>& out_color) {
    vector<int> order(V);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return adj[a].size() > adj[b].size();
    });
    
    vector<int> color(V, -1);
    vector<int> used(V, -1);
    int max_c = 0;
    for(int u : order) {
        for(int v : adj[u]) {
            if (color[v] != -1) used[color[v]] = u;
        }
        int c = 0;
        while(c < V && used[c] == u) c++;
        color[u] = c;
        max_c = max(max_c, c);
    }
    out_color = color;
    return max_c + 1;
}

int greedy_ID(int V, const vector<vector<int>>& adj, vector<int>& out_color) {
    vector<int> colored_deg(V, 0);
    vector<int> color(V, -1);
    
    vector<vector<int>> buckets(V);
    vector<int> order(V);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return adj[a].size() < adj[b].size(); 
    });
    buckets[0] = order;
    int max_b = 0;
    
    vector<int> used(V, -1);
    int max_c = 0;
    int colored_count = 0;
    
    while(colored_count < V) {
        int u = -1;
        while(max_b >= 0) {
            while(!buckets[max_b].empty()) {
                int v = buckets[max_b].back();
                buckets[max_b].pop_back();
                if (color[v] == -1 && colored_deg[v] == max_b) {
                    u = v;
                    break;
                }
            }
            if (u != -1) break;
            max_b--;
        }
        if (u == -1) break;
        
        for(int v : adj[u]) {
            if (color[v] != -1) used[color[v]] = u;
        }
        int c = 0;
        while(c < V && used[c] == u) c++;
        color[u] = c;
        max_c = max(max_c, c);
        colored_count++;
        
        for(int v : adj[u]) {
            if (color[v] == -1) {
                colored_deg[v]++;
                buckets[colored_deg[v]].push_back(v);
                if (colored_deg[v] > max_b) max_b = colored_deg[v];
            }
        }
    }
    out_color = color;
    return max_c + 1;
}

bool tabucol(int V, int K, const vector<vector<int>>& adj, vector<int>& color, auto global_start_time, long long time_limit_ms) {
    if (K == 1) {
        return false;
    }

    for(int u=0; u<V; ++u) {
        if (color[u] >= K) {
            vector<int> c_count(K, 0);
            for(int v : adj[u]) {
                if (color[v] < K && color[v] != -1) {
                    c_count[color[v]]++;
                }
            }
            int best_c = 0;
            int min_c = V + 1;
            for(int c=0; c<K; ++c) {
                if (c_count[c] < min_c) {
                    min_c = c_count[c];
                    best_c = c;
                }
            }
            color[u] = best_c;
        }
    }
    
    vector<int> conflicts_count(V, 0);
    int total_conflicts = 0;
    for(int u=0; u<V; ++u) {
        for(int v : adj[u]) {
            if (color[v] == color[u]) {
                conflicts_count[u]++;
                total_conflicts++;
            }
        }
    }
    total_conflicts /= 2;
    
    if (total_conflicts == 0) return true;
    
    bool use_matrix = ( (long long)V * K <= 50000000LL );
    vector<int> adj_color_count;
    vector<int> tabu_until;
    vector<int> tabu_color_fallback;
    vector<int> tabu_until_fallback;
    
    if (use_matrix) {
        adj_color_count.assign(V * K, 0);
        tabu_until.assign(V * K, 0);
        for(int u=0; u<V; ++u) {
            for(int v : adj[u]) {
                adj_color_count[u * K + color[v]]++;
            }
        }
    } else {
        tabu_color_fallback.assign(V, -1);
        tabu_until_fallback.assign(V, 0);
    }
    
    int iter = 0;
    mt19937 rng(1337);
    
    int best_total_conflicts = total_conflicts;
    vector<int> best_color = color;
    
    while(total_conflicts > 0) {
        iter++;
        if (iter % 1000 == 0) {
            auto current_time = chrono::high_resolution_clock::now();
            auto elapsed = chrono::duration_cast<chrono::milliseconds>(current_time - global_start_time).count();
            if (elapsed > time_limit_ms) {
                break;
            }
        }
        
        vector<int> conf_verts;
        for(int u=0; u<V; ++u) {
            if (conflicts_count[u] > 0) {
                conf_verts.push_back(u);
            }
        }
        
        if (conf_verts.empty()) {
            break; 
        }
        
        int best_u = -1;
        int best_c = -1;
        int min_delta = 1e9;
        
        int num_eval = conf_verts.size();
        if (num_eval > 50) num_eval = 50;
        
        for(int i=0; i<num_eval; ++i) {
            int idx = i + rng() % (conf_verts.size() - i);
            swap(conf_verts[i], conf_verts[idx]);
            int u = conf_verts[i];
            
            int old_c = color[u];
            if (use_matrix) {
                int current_conf = adj_color_count[u * K + old_c];
                for(int c=0; c<K; ++c) {
                    if (c == old_c) continue;
                    int new_conf = adj_color_count[u * K + c];
                    int delta = new_conf - current_conf;
                    
                    bool is_tabu = tabu_until[u * K + c] >= iter;
                    
                    if (is_tabu && (total_conflicts + delta >= best_total_conflicts)) {
                        continue;
                    }
                    
                    if (delta < min_delta) {
                        min_delta = delta;
                        best_u = u;
                        best_c = c;
                    } else if (delta == min_delta) {
                        if (rng() % 2 == 0) {
                            best_u = u;
                            best_c = c;
                        }
                    }
                }
            } else {
                vector<int> c_count(K, 0);
                for(int v : adj[u]) {
                    c_count[color[v]]++;
                }
                int current_conf = c_count[old_c];
                for(int c=0; c<K; ++c) {
                    if (c == old_c) continue;
                    int new_conf = c_count[c];
                    int delta = new_conf - current_conf;
                    
                    bool is_tabu = (tabu_color_fallback[u] == c && tabu_until_fallback[u] >= iter);
                    
                    if (is_tabu && (total_conflicts + delta >= best_total_conflicts)) {
                        continue;
                    }
                    
                    if (delta < min_delta) {
                        min_delta = delta;
                        best_u = u;
                        best_c = c;
                    } else if (delta == min_delta) {
                        if (rng() % 2 == 0) {
                            best_u = u;
                            best_c = c;
                        }
                    }
                }
            }
        }
        
        if (best_u == -1) {
            best_u = conf_verts[rng() % conf_verts.size()];
            best_c = rng() % K;
            while (best_c == color[best_u]) {
                best_c = rng() % K;
            }
            if (use_matrix) {
                min_delta = adj_color_count[best_u * K + best_c] - adj_color_count[best_u * K + color[best_u]];
            } else {
                vector<int> c_count(K, 0);
                for(int v : adj[best_u]) c_count[color[v]]++;
                min_delta = c_count[best_c] - c_count[color[best_u]];
            }
        }
        
        int old_c = color[best_u];
        color[best_u] = best_c;
        total_conflicts += min_delta;
        
        if (total_conflicts < best_total_conflicts) {
            best_total_conflicts = total_conflicts;
            best_color = color;
        }
        
        for(int v : adj[best_u]) {
            if (color[v] == old_c) {
                conflicts_count[v]--;
            } else if (color[v] == best_c) {
                conflicts_count[v]++;
            }
            if (use_matrix) {
                adj_color_count[v * K + old_c]--;
                adj_color_count[v * K + best_c]++;
            }
        }
        conflicts_count[best_u] += min_delta;
        
        int tenure = (total_conflicts * 0.6) + (rng() % 10) + 10;
        if (use_matrix) {
            tabu_until[best_u * K + old_c] = iter + tenure;
        } else {
            tabu_color_fallback[best_u] = old_c;
            tabu_until_fallback[best_u] = iter + tenure;
        }
    }
    
    color = best_color;
    return best_total_conflicts == 0;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    auto global_start_time = chrono::high_resolution_clock::now();
    
    int V, E, K;
    if (!(cin >> V >> E >> K)) return 0;
    
    vector<vector<int>> adj(V);
    for(int i=0; i<E; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    vector<int> color_ldf;
    int k_ldf = greedy_LDF(V, adj, color_ldf);
    if (k_ldf <= K) {
        for(int i=0; i<V; ++i) cout << color_ldf[i] << (i == V-1 ? "" : " ");
        cout << "\n";
        return 0;
    }
    
    vector<int> color_id;
    int k_id = greedy_ID(V, adj, color_id);
    if (k_id <= K) {
        for(int i=0; i<V; ++i) cout << color_id[i] << (i == V-1 ? "" : " ");
        cout << "\n";
        return 0;
    }
    
    vector<int> init_color = (k_ldf < k_id) ? color_ldf : color_id;
    
    long long time_limit_ms = 295000;
    
    tabucol(V, K, adj, init_color, global_start_time, time_limit_ms);
    
    for(int i=0; i<V; ++i) cout << init_color[i] << (i == V-1 ? "" : " ");
    cout << "\n";
    
    return 0;
}
