use std::collections::BinaryHeap;
use std::io::{self, BufRead, Write};

#[derive(Clone, Copy)]
struct Point {
    coords: [f64; 3],
    original_idx: u32,
}

#[derive(Clone, Copy)]
struct PointMeta {
    hole_idx: u32,
    sample_idx: u32,
}

#[derive(PartialEq)]
struct HeapElement {
    dist_sq: f64,
    original_idx: u32,
}

impl Eq for HeapElement {}

impl PartialOrd for HeapElement {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        self.dist_sq.partial_cmp(&other.dist_sq)
    }
}

impl Ord for HeapElement {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        self.partial_cmp(other).unwrap()
    }
}

struct Outlier {
    hole_idx: u32,
    sample_idx: u32,
    property_idx: usize,
    confidence: f32,
}

fn build_kdtree_mt(points: &mut [Point], depth: usize, threads_left: usize) {
    if points.len() <= 1 {
        return;
    }
    let axis = depth % 3;
    let mid = points.len() / 2;
    points.select_nth_unstable_by(mid, |a, b| {
        a.coords[axis]
            .partial_cmp(&b.coords[axis])
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let (left_and_mid, right) = points.split_at_mut(mid + 1);
    let left = &mut left_and_mid[0..mid];

    if threads_left > 1 {
        let t1 = threads_left / 2;
        let t2 = threads_left - t1;
        std::thread::scope(|s| {
            s.spawn(|| build_kdtree_mt(left, depth + 1, t1));
            build_kdtree_mt(right, depth + 1, t2);
        });
    } else {
        build_kdtree_mt(left, depth + 1, 1);
        build_kdtree_mt(right, depth + 1, 1);
    }
}

fn knn(
    points: &[Point],
    target: [f64; 3],
    target_original_idx: u32,
    depth: usize,
    k: usize,
    heap: &mut BinaryHeap<HeapElement>,
) {
    if points.is_empty() {
        return;
    }
    let axis = depth % 3;
    let mid = points.len() / 2;
    let node = &points[mid];

    if node.original_idx != target_original_idx {
        let dx = target[0] - node.coords[0];
        let dy = target[1] - node.coords[1];
        let dz = target[2] - node.coords[2];
        let dist_sq = dx * dx + dy * dy + dz * dz;

        if heap.len() < k {
            heap.push(HeapElement {
                dist_sq,
                original_idx: node.original_idx,
            });
        } else if dist_sq < heap.peek().unwrap().dist_sq {
            heap.pop();
            heap.push(HeapElement {
                dist_sq,
                original_idx: node.original_idx,
            });
        }
    }

    let diff = target[axis] - node.coords[axis];
    let (first, second) = if diff < 0.0 {
        (&points[0..mid], &points[mid + 1..])
    } else {
        (&points[mid + 1..], &points[0..mid])
    };

    knn(first, target, target_original_idx, depth + 1, k, heap);

    if heap.len() < k || diff * diff < heap.peek().unwrap().dist_sq {
        knn(second, target, target_original_idx, depth + 1, k, heap);
    }
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut buffer = String::new();

    // Read header
    if reader.read_line(&mut buffer).unwrap_or(0) == 0 {
        return;
    }
    let parts: Vec<&str> = buffer.split_whitespace().collect();
    if parts.len() < 2 {
        return;
    }
    let num_holes: usize = parts[0].parse().unwrap_or(0);
    let num_properties: usize = parts[1].parse().unwrap_or(0);

    if num_holes == 0 || num_properties == 0 {
        println!("0");
        return;
    }

    buffer.clear();
    if reader.read_line(&mut buffer).unwrap_or(0) == 0 {
        return;
    }
    let property_names: Vec<String> = buffer
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    let mut hole_ids = Vec::with_capacity(num_holes);
    let mut points = Vec::new();
    let mut meta = Vec::new();
    let mut flat_properties = Vec::new();

    for hole_idx in 0..num_holes {
        buffer.clear();
        while reader.read_line(&mut buffer).unwrap_or(0) > 0 {
            if !buffer.trim().is_empty() {
                break;
            }
            buffer.clear();
        }

        let parts: Vec<&str> = buffer.split_whitespace().collect();
        if parts.len() < 9 {
            continue;
        }

        hole_ids.push(parts[0].to_string());
        let start_x: f64 = parts[1].parse().unwrap_or(0.0);
        let start_y: f64 = parts[2].parse().unwrap_or(0.0);
        let start_z: f64 = parts[3].parse().unwrap_or(0.0);
        let dir_x: f64 = parts[4].parse().unwrap_or(0.0);
        let dir_y: f64 = parts[5].parse().unwrap_or(0.0);
        let dir_z: f64 = parts[6].parse().unwrap_or(0.0);
        let length: f64 = parts[7].parse().unwrap_or(0.0);
        let num_samples: usize = parts[8].parse().unwrap_or(0);
        let _ = length; // Unused for position, derived from depth

        let dir_len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (nx, ny, nz) = if dir_len > 0.0 {
            (dir_x / dir_len, dir_y / dir_len, dir_z / dir_len)
        } else {
            (0.0, 0.0, 0.0)
        };

        for sample_idx in 0..num_samples {
            buffer.clear();
            while reader.read_line(&mut buffer).unwrap_or(0) > 0 {
                if !buffer.trim().is_empty() {
                    break;
                }
                buffer.clear();
            }

            let mut parts = buffer.split_whitespace();
            let depth: f64 = parts
                .next()
                .and_then(|s| s.parse().ok())
                .unwrap_or(0.0);

            let cx = start_x + nx * depth;
            let cy = start_y + ny * depth;
            let cz = start_z + nz * depth;

            let original_idx = points.len() as u32;
            points.push(Point {
                coords: [cx, cy, cz],
                original_idx,
            });
            meta.push(PointMeta {
                hole_idx: hole_idx as u32,
                sample_idx: sample_idx as u32,
            });

            for _ in 0..num_properties {
                let val = parts
                    .next()
                    .and_then(|s| s.parse::<f32>().ok())
                    .unwrap_or(f32::NAN);
                flat_properties.push(val);
            }
        }
    }

    if points.is_empty() {
        println!("0");
        return;
    }

    let num_threads = std::thread::available_parallelism()
        .map(|n| n.get())
        .unwrap_or(8);

    // Build KD-Tree multi-threaded
    build_kdtree_mt(&mut points, 0, num_threads);

    // Adaptive logic: scale neighborhood search size based on dataset scale
    let k = if points.len() < 10_000 {
        50.min(points.len().saturating_sub(1))
    } else if points.len() < 100_000 {
        30
    } else {
        20
    };

    if k < 5 {
        // Cannot compute robust statistics with fewer than 5 neighbors
        println!("0");
        return;
    }

    let chunk_size = (points.len() + num_threads - 1) / num_threads;
    let chunk_size = chunk_size.max(1000); // Prevent spawning threads for tiny chunks

    let mut all_outliers = std::thread::scope(|s| {
        let mut handles = Vec::new();
        for chunk in points.chunks(chunk_size) {
            let points_ref = &points;
            let meta_ref = &meta;
            let flat_props_ref = &flat_properties;

            let handle = s.spawn(move || {
                let mut local_outliers = Vec::new();
                let mut heap = BinaryHeap::with_capacity(k + 1);
                for p in chunk {
                    heap.clear();
                    knn(points_ref, p.coords, p.original_idx, 0, k, &mut heap);

                    let neighbors: Vec<u32> = heap.into_iter().map(|e| e.original_idx).collect();
                    if neighbors.len() < 5 {
                        continue;
                    }

                    for prop_idx in 0..num_properties {
                        let target_val =
                            flat_props_ref[p.original_idx as usize * num_properties + prop_idx];
                        if target_val.is_nan() {
                            continue;
                        }

                        let mut sum = 0.0;
                        let mut count = 0;
                        for &n_idx in &neighbors {
                            let n_val =
                                flat_props_ref[n_idx as usize * num_properties + prop_idx];
                            if !n_val.is_nan() {
                                sum += n_val;
                                count += 1;
                            }
                        }

                        if count >= 5 {
                            let mean = sum / count as f32;
                            let mut var_sum = 0.0;
                            for &n_idx in &neighbors {
                                let n_val =
                                    flat_props_ref[n_idx as usize * num_properties + prop_idx];
                                if !n_val.is_nan() {
                                    let diff = n_val - mean;
                                    var_sum += diff * diff;
                                }
                            }
                            let variance = var_sum / count as f32;
                            let std_dev = variance.sqrt();

                            let epsilon = 1e-5;
                            let z_score = (target_val - mean).abs() / (std_dev + epsilon);

                            let absolute_diff = (target_val - mean).abs();
                            if z_score > 3.5 && absolute_diff > (mean.abs() * 0.01 + 1e-4) {
                                let m = &meta_ref[p.original_idx as usize];
                                local_outliers.push(Outlier {
                                    hole_idx: m.hole_idx,
                                    sample_idx: m.sample_idx,
                                    property_idx: prop_idx,
                                    confidence: z_score,
                                });
                            }
                        }
                    }
                }
                local_outliers
            });
            handles.push(handle);
        }

        let mut all = Vec::new();
        for handle in handles {
            all.extend(handle.join().unwrap());
        }
        all
    });

    all_outliers.sort_unstable_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let max_output = 100_000;
    all_outliers.truncate(max_output);

    let stdout = io::stdout();
    let mut writer = io::BufWriter::new(stdout.lock());

    writeln!(writer, "{}", all_outliers.len()).unwrap();
    for out in all_outliers {
        let hole_id = &hole_ids[out.hole_idx as usize];
        let prop_name = &property_names[out.property_idx];
        writeln!(
            writer,
            "{} {} {} {:.4}",
            hole_id, out.sample_idx, prop_name, out.confidence
        )
        .unwrap();
    }
}
