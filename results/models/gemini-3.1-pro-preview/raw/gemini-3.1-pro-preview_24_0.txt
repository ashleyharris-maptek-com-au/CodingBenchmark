struct Point3D {
    float x, y, z;
};

struct State {
    float x, y, z;
    float vx, vy, vz;
    float pitch, yaw, roll;
    float pitch_rate, yaw_rate, roll_rate;
    float fuel;
};

std::atomic<bool> path_ready{false};
std::mutex path_mutex;
std::vector<Point3D> shared_waypoints;

bool is_safe(float x, float y, float z, const uint16_t* terrain_data, float width, float depth, float height) {
    if (!terrain_data) return true;
    if (x < 0 || x >= width || y < 0 || y >= depth || z < 0 || z >= height) return false;
    int ix = (int)x;
    int iy = (int)y;
    if (ix < 0) ix = 0; if (ix >= width) ix = width - 1;
    if (iy < 0) iy = 0; if (iy >= depth) iy = depth - 1;

    size_t idx = ((size_t)ix * (size_t)depth + (size_t)iy) * 3;
    uint16_t a = terrain_data[idx];
    uint16_t b = terrain_data[idx+1];
    uint16_t c = terrain_data[idx+2];

    if (z < a) return false;
    if (z >= a && z < b) return true;
    if (z >= b && z < c) return false;
    if (z >= c) return true;
    return false;
}

bool is_safe_margin(float x, float y, float z, const uint16_t* terrain_data, float width, float depth, float height) {
    float margin = 2.0f;
    for (float dx = -margin; dx <= margin + 0.01f; dx += margin) {
        for (float dy = -margin; dy <= margin + 0.01f; dy += margin) {
            for (float dz = -margin; dz <= margin + 0.01f; dz += margin) {
                if (!is_safe(x + dx, y + dy, z + dz, terrain_data, width, depth, height)) return false;
            }
        }
    }
    return true;
}

bool check_line(float x1, float y1, float z1, float x2, float y2, float z2, const uint16_t* terrain_data, float width, float depth, float height) {
    float dx = x2 - x1;
    float dy = y2 - y1;
    float dz = z2 - z1;
    float dist = sqrt(dx*dx + dy*dy + dz*dz);
    int steps = std::ceil(dist / 2.0f);
    if (steps == 0) return true;
    for (int i = 0; i <= steps; ++i) {
        float t = (float)i / steps;
        if (!is_safe_margin(x1 + t*dx, y1 + t*dy, z1 + t*dz, terrain_data, width, depth, height)) return false;
    }
    return true;
}

struct RRTNode {
    float x, y, z;
    int parent;
};

std::vector<Point3D> rrt_connect(Point3D start, Point3D target, const uint16_t* terrain, float width, float depth, float height) {
    float step_size = std::max(20.0f, sqrt(pow(start.x-target.x, 2) + pow(start.y-target.y, 2))/100.0f);
    
    std::vector<RRTNode> treeA, treeB;
    treeA.push_back({start.x, start.y, start.z, -1});
    treeB.push_back({target.x, target.y, target.z, -1});
    bool a_is_start = true;
    
    float min_x = std::max(0.0f, std::min(start.x, target.x) - 1000.0f);
    float max_x = std::min(width, std::max(start.x, target.x) + 1000.0f);
    float min_y = std::max(0.0f, std::min(start.y, target.y) - 1000.0f);
    float max_y = std::min(depth, std::max(start.y, target.y) + 1000.0f);
    
    for (int iter = 0; iter < 50000; ++iter) {
        float qx, qy, qz;
        if (rand() % 100 < 20) {
            qx = treeB.back().x; qy = treeB.back().y; qz = treeB.back().z;
        } else {
            qx = min_x + (max_x - min_x) * ((float)rand() / RAND_MAX);
            qy = min_y + (max_y - min_y) * ((float)rand() / RAND_MAX);
            qz = height * ((float)rand() / RAND_MAX);
        }
        
        float min_d = 1e9f; int nearestA = -1;
        for (size_t i=0; i<treeA.size(); ++i) {
            float dx = treeA[i].x - qx;
            float dy = treeA[i].y - qy;
            float dz = treeA[i].z - qz;
            float d = dx*dx + dy*dy + dz*dz;
            if (d < min_d) { min_d = d; nearestA = i; }
        }
        
        float nx = treeA[nearestA].x, ny = treeA[nearestA].y, nz = treeA[nearestA].z;
        float dx = qx - nx, dy = qy - ny, dz = qz - nz;
        float d = sqrt(dx*dx + dy*dy + dz*dz);
        if (d > step_size) { dx = dx/d * step_size; dy = dy/d * step_size; dz = dz/d * step_size; }
        
        float new_x = nx + dx, new_y = ny + dy, new_z = nz + dz;
        if (check_line(nx, ny, nz, new_x, new_y, new_z, terrain, width, depth, height)) {
            treeA.push_back({new_x, new_y, new_z, nearestA});
            int addedA = treeA.size() - 1;
            
            float min_dB = 1e9f; int nearestB = -1;
            for (size_t i=0; i<treeB.size(); ++i) {
                float bdx = treeB[i].x - new_x;
                float bdy = treeB[i].y - new_y;
                float bdz = treeB[i].z - new_z;
                float dist_sq = bdx*bdx + bdy*bdy + bdz*bdz;
                if (dist_sq < min_dB) { min_dB = dist_sq; nearestB = i; }
            }
            
            float bx = treeB[nearestB].x, by = treeB[nearestB].y, bz = treeB[nearestB].z;
            int currentB = nearestB;
            bool connected = false;
            
            while (true) {
                float bdx = new_x - bx, bdy = new_y - by, bdz = new_z - bz;
                float bd = sqrt(bdx*bdx + bdy*bdy + bdz*bdz);
                if (bd < 0.1f) { connected = true; break; }
                
                float bstep = std::min(step_size, bd);
                float step_x = bx + bdx/bd * bstep;
                float step_y = by + bdy/bd * bstep;
                float step_z = bz + bdz/bd * bstep;
                
                if (check_line(bx, by, bz, step_x, step_y, step_z, terrain, width, depth, height)) {
                    treeB.push_back({step_x, step_y, step_z, currentB});
                    currentB = treeB.size() - 1;
                    bx = step_x; by = step_y; bz = step_z;
                } else {
                    break;
                }
            }
            
            if (connected) {
                std::vector<Point3D> pathA, pathB;
                int c = addedA; while (c != -1) { pathA.push_back({treeA[c].x, treeA[c].y, treeA[c].z}); c = treeA[c].parent; }
                c = currentB; while (c != -1) { pathB.push_back({treeB[c].x, treeB[c].y, treeB[c].z}); c = treeB[c].parent; }
                
                std::vector<Point3D> final_path;
                if (a_is_start) {
                    for (int i = pathA.size()-1; i >= 0; --i) final_path.push_back(pathA[i]);
                    for (size_t i = 0; i < pathB.size(); ++i) final_path.push_back(pathB[i]);
                } else {
                    for (int i = pathB.size()-1; i >= 0; --i) final_path.push_back(pathB[i]);
                    for (size_t i = 0; i < pathA.size(); ++i) final_path.push_back(pathA[i]);
                }
                return final_path;
            }
        }
        std::swap(treeA, treeB);
        a_is_start = !a_is_start;
    }

    return {start, target};
}

std::vector<Point3D> find_path(Point3D start, Point3D target, const uint16_t* terrain, float width, float depth, float height) {
    if (check_line(start.x, start.y, start.z, target.x, target.y, target.z, terrain, width, depth, height)) {
        return {start, target};
    }
    
    for (float fly_z = height - 10.0f; fly_z >= std::max(start.z, target.z) + 10.0f; fly_z -= 50.0f) {
        if (check_line(start.x, start.y, start.z, start.x, start.y, fly_z, terrain, width, depth, height) &&
            check_line(start.x, start.y, fly_z, target.x, target.y, fly_z, terrain, width, depth, height) &&
            check_line(target.x, target.y, fly_z, target.x, target.y, target.z, terrain, width, depth, height)) {
            return {start, {start.x, start.y, fly_z}, {target.x, target.y, fly_z}, target};
        }
    }

    float fly_z = height - 10.0f;
    Point3D s_up = {start.x, start.y, fly_z};
    Point3D t_up = {target.x, target.y, fly_z};
    bool s_escape = false, t_escape = false;

    for (float r = 0; r <= 500; r += 50) {
        for (float ang = 0; ang < 2 * M_PI; ang += M_PI / 4) {
            if (!s_escape) {
                Point3D cand = {start.x + r * cos(ang), start.y + r * sin(ang), fly_z};
                if (check_line(start.x, start.y, start.z, cand.x, cand.y, cand.z, terrain, width, depth, height)) {
                    s_up = cand; s_escape = true;
                }
            }
            if (!t_escape) {
                Point3D cand = {target.x + r * cos(ang), target.y + r * sin(ang), fly_z};
                if (check_line(target.x, target.y, target.z, cand.x, cand.y, cand.z, terrain, width, depth, height)) {
                    t_up = cand; t_escape = true;
                }
            }
        }
        if (s_escape && t_escape) break;
    }

    if (s_escape && t_escape && check_line(s_up.x, s_up.y, s_up.z, t_up.x, t_up.y, t_up.z, terrain, width, depth, height)) {
        return {start, s_up, t_up, target};
    }

    return rrt_connect(start, target, terrain, width, depth, height);
}

std::vector<Point3D> smooth_path(const std::vector<Point3D>& path, const uint16_t* terrain, float width, float depth, float height) {
    if (path.size() <= 2) return path;
    std::vector<Point3D> smoothed;
    smoothed.push_back(path[0]);
    int curr = 0;
    while (curr < (int)path.size() - 1) {
        int furthest = curr + 1;
        for (int i = path.size() - 1; i > curr + 1; --i) {
            if (check_line(path[curr].x, path[curr].y, path[curr].z, path[i].x, path[i].y, path[i].z, terrain, width, depth, height)) {
                furthest = i;
                break;
            }
        }
        smoothed.push_back(path[furthest]);
        curr = furthest;
    }
    return smoothed;
}

void compute_control(const State& st, Point3D target_pos, Point3D target_vel, float max_v, float gravity, float max_thrust, float& thrust_cmd, float& pitch_cmd, float& yaw_cmd, float& roll_cmd) {
    float Kp_pos = 0.2f, Kd_pos = 1.0f;
    float desired_vx = Kp_pos * (target_pos.x - st.x);
    float desired_vy = Kp_pos * (target_pos.y - st.y);
    float desired_vz = Kp_pos * (target_pos.z - st.z);
    
    float v_mag = sqrt(desired_vx*desired_vx + desired_vy*desired_vy + desired_vz*desired_vz);
    if (v_mag > max_v) {
        desired_vx = desired_vx / v_mag * max_v;
        desired_vy = desired_vy / v_mag * max_v;
        desired_vz = desired_vz / v_mag * max_v;
    }
    
    desired_vx += target_vel.x;
    desired_vy += target_vel.y;
    desired_vz += target_vel.z;

    float ax = Kd_pos * (desired_vx - st.vx);
    float ay = Kd_pos * (desired_vy - st.vy);
    float az = Kd_pos * (desired_vz - st.vz) + gravity;

    if (az < 0.1f) az = 0.1f;

    if (az > max_thrust) {
        az = max_thrust;
        ax = 0; ay = 0;
    } else {
        float max_xy_acc = sqrt(max_thrust*max_thrust - az*az);
        float xy_acc = sqrt(ax*ax + ay*ay);
        if (xy_acc > max_xy_acc) {
            ax = ax / xy_acc * max_xy_acc;
            ay = ay / xy_acc * max_xy_acc;
        }
    }

    float acc_len = sqrt(ax*ax + ay*ay + az*az);
    thrust_cmd = acc_len / max_thrust;

    float Dx = ax, Dy = ay, Dz = az;
    if (acc_len > 0.001f) {
        Dx /= acc_len; Dy /= acc_len; Dz /= acc_len;
    } else {
        Dx = 0; Dy = 0; Dz = 1;
    }

    float p1 = atan2(-sqrt(Dx*Dx + Dy*Dy), Dz);
    float y1 = atan2(Dy, Dx);

    float p2 = -p1;
    float y2 = y1 + M_PI;
    if (y2 > M_PI) y2 -= 2 * M_PI;

    auto get_err = [](float tgt, float curr) {
        float e = tgt - curr;
        while (e > M_PI) e -= 2 * M_PI;
        while (e < -M_PI) e += 2 * M_PI;
        return e;
    };

    float dy1 = get_err(y1, st.yaw);
    float dp1 = p1 - st.pitch;
    float dist1 = dy1*dy1 + dp1*dp1;

    float dy2 = get_err(y2, st.yaw);
    float dp2 = p2 - st.pitch;
    float dist2 = dy2*dy2 + dp2*dp2;

    float target_pitch = (dist1 < dist2) ? p1 : p2;
    float target_yaw   = (dist1 < dist2) ? y1 : y2;

    float Kp_ang = 5.0f, Kd_ang = 3.0f;
    pitch_cmd = Kp_ang * get_err(target_pitch, st.pitch) - Kd_ang * st.pitch_rate;
    yaw_cmd = Kp_ang * get_err(target_yaw, st.yaw) - Kd_ang * st.yaw_rate;
    roll_cmd = Kp_ang * get_err(0.0f, st.roll) - Kd_ang * st.roll_rate;

    auto clamp = [](float v) { return v > 1.0f ? 1.0f : (v < -1.0f ? -1.0f : v); };
    pitch_cmd = clamp(pitch_cmd);
    yaw_cmd = clamp(yaw_cmd);
    roll_cmd = clamp(roll_cmd);
    thrust_cmd = thrust_cmd > 1.0f ? 1.0f : (thrust_cmd < 0.0f ? 0.0f : thrust_cmd);
}

int main() {
    std::srand((unsigned)time(NULL));
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    float width, depth, height, gravity, max_thrust, fuel;
    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel)) return 0;

    Point3D start, target;
    std::cin >> start.x >> start.y >> start.z >> target.x >> target.y >> target.z;

    std::string dummy;
    std::getline(std::cin, dummy);
    
    std::string terrain_file;
    std::getline(std::cin, terrain_file);
    size_t first_non_space = terrain_file.find_first_not_of(" \t\r\n");
    if (first_non_space != std::string::npos) {
        terrain_file = terrain_file.substr(first_non_space);
    } else {
        terrain_file = "";
    }
    while (!terrain_file.empty() && (terrain_file.back() == '\r' || terrain_file.back() == '\n' || terrain_file.back() == ' ' || terrain_file.back() == '\t')) {
        terrain_file.pop_back();
    }
    
    HANDLE hFile = CreateFileA(terrain_file.c_str(), GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
    const uint16_t* terrain_data = nullptr;
    if (hFile != INVALID_HANDLE_VALUE) {
        HANDLE hMap = CreateFileMappingA(hFile, NULL, PAGE_READONLY, 0, 0, NULL);
        if (hMap) {
            terrain_data = (const uint16_t*)MapViewOfFile(hMap, FILE_MAP_READ, 0, 0, 0);
        }
    }

    std::thread path_thread([&]() {
        auto path = find_path(start, target, terrain_data, width, depth, height);
        path = smooth_path(path, terrain_data, width, depth, height);
        std::lock_guard<std::mutex> lock(path_mutex);
        shared_waypoints = path;
        path_ready = true;
    });

    size_t current_wp = 1;
    std::string line;
    while (std::getline(std::cin, line)) {
        if (line.empty()) continue;
        State st;
        if (sscanf(line.c_str(), "%f %f %f %f %f %f %f %f %f %f %f %f %f",
            &st.x, &st.y, &st.z, &st.vx, &st.vy, &st.vz,
            &st.pitch, &st.yaw, &st.roll,
            &st.pitch_rate, &st.yaw_rate, &st.roll_rate, &st.fuel) != 13) {
            continue;
        }

        std::vector<Point3D> waypoints;
        bool has_path = false;
        {
            std::lock_guard<std::mutex> lock(path_mutex);
            if (path_ready) {
                waypoints = shared_waypoints;
                has_path = true;
            }
        }

        float thrust_cmd = 0, pitch_cmd = 0, yaw_cmd = 0, roll_cmd = 0;

        if (has_path && waypoints.size() > 1) {
            if (current_wp >= waypoints.size()) current_wp = waypoints.size() - 1;
            Point3D target_pos = waypoints[current_wp];
            
            float dist_to_wp = sqrt(pow(st.x - target_pos.x, 2) + pow(st.y - target_pos.y, 2) + pow(st.z - target_pos.z, 2));
            if (dist_to_wp < 15.0f && current_wp < waypoints.size() - 1) {
                current_wp++;
                target_pos = waypoints[current_wp];
            }

            Point3D target_vel = {0, 0, 0};
            if (current_wp < waypoints.size() - 1) {
                Point3D next_wp = waypoints[current_wp + 1];
                float dx = next_wp.x - target_pos.x;
                float dy = next_wp.y - target_pos.y;
                float dz = next_wp.z - target_pos.z;
                float len = sqrt(dx*dx + dy*dy + dz*dz);
                if (len > 0.001f) {
                    float speed = 15.0f;
                    target_vel = {dx/len * speed, dy/len * speed, dz/len * speed};
                }
            }

            float max_v = 20.0f;
            if (current_wp == waypoints.size() - 1) {
                float dist_to_final = sqrt(pow(target.x - st.x, 2) + pow(target.y - st.y, 2) + pow(target.z - st.z, 2));
                max_v = std::min(20.0f, std::max(1.5f, sqrt(dist_to_final) * 0.8f));
                
                float dist_xy = sqrt(pow(target.x - st.x, 2) + pow(target.y - st.y, 2));
                if (dist_xy < 5.0f && st.z - target.z < 5.0f) {
                    target_pos.z = target.z - 5.0f; // target slightly into ground for firm touchdown
                    max_v = 2.0f;
                }
            }

            compute_control(st, target_pos, target_vel, max_v, gravity, max_thrust, thrust_cmd, pitch_cmd, yaw_cmd, roll_cmd);
        } else {
            compute_control(st, start, {0,0,0}, 2.0f, gravity, max_thrust, thrust_cmd, pitch_cmd, yaw_cmd, roll_cmd);
        }

        std::cout << thrust_cmd << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();
    }

    path_thread.join();
    return 0;
}
