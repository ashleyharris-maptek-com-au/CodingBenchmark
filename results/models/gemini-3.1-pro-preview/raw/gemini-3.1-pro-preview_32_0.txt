#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <algorithm>
#include <numeric>

using namespace std;

const long long INF = 1e18;

struct Edge {
    int to;
    long long weight;
    int id;
};

struct OriginalEdge {
    int u, v;
    long long weight;
    int id;
};

struct DSU {
    vector<int> parent;
    DSU(int n) {
        parent.resize(n);
        iota(parent.begin(), parent.end(), 0);
    }
    int find(int i) {
        if (parent[i] == i) return i;
        return parent[i] = find(parent[i]);
    }
    bool unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
            return true;
        }
        return false;
    }
};

// global for tracing exact DP
vector<int> prev_mask_arr;
vector<int> prev_u_arr;
vector<int> prev_edge_arr;
int V_global;
set<int> candidate_edges_set;

void trace(int mask, int u) {
    int offset = mask * V_global + u;
    if (prev_mask_arr[offset] == -1) return;

    if (prev_mask_arr[offset] == mask) {
        candidate_edges_set.insert(prev_edge_arr[offset]);
        trace(mask, prev_u_arr[offset]);
    } else {
        trace(prev_mask_arr[offset], u);
        trace(mask ^ prev_mask_arr[offset], u);
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int V, E, raw_K;
    if (!(cin >> V >> E >> raw_K)) return 0;
    
    V_global = V;

    vector<OriginalEdge> edge_list(E);
    vector<vector<Edge>> adj(V);

    for (int i = 0; i < E; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edge_list[i] = {u, v, w, i};
        adj[u].push_back({v, w, i});
        adj[v].push_back({u, w, i});
    }

    vector<int> temp_terminals(raw_K);
    for (int i = 0; i < raw_K; ++i) {
        cin >> temp_terminals[i];
    }
    sort(temp_terminals.begin(), temp_terminals.end());
    temp_terminals.erase(unique(temp_terminals.begin(), temp_terminals.end()), temp_terminals.end());
    
    int K = temp_terminals.size();
    vector<int> terminals = temp_terminals;
    vector<bool> is_terminal(V, false);
    for(int t : terminals) is_terminal[t] = true;

    if (K == 0 || K == 1) {
        cout << "0 0\n";
        return 0;
    }

    bool use_dp = false;
    if (K <= 15) {
        long long states = (long long)(1 << K) * V;
        long long memory = states * 20LL; // approx 20 bytes per state
        long long ops = 1;
        for (int i = 0; i < K; ++i) ops *= 3;
        ops /= 2;
        if (memory <= 2048LL * 1024 * 1024 && ops * V <= 500000000LL) {
            use_dp = true;
        }
    }

    if (use_dp) {
        int states_count = (1 << K) * V;
        vector<long long> dp(states_count, INF);
        prev_mask_arr.assign(states_count, -1);
        prev_u_arr.assign(states_count, -1);
        prev_edge_arr.assign(states_count, -1);

        for (int i = 0; i < K; ++i) {
            dp[(1 << i) * V + terminals[i]] = 0;
        }

        for (int mask = 1; mask < (1 << K); ++mask) {
            if ((mask & (mask - 1)) != 0) {
                int lsb = mask & -mask;
                int rem_mask = mask ^ lsb;
                for (int sub = (rem_mask - 1) & rem_mask; sub > 0; sub = (sub - 1) & rem_mask) {
                    int actual_sub = sub | lsb;
                    int actual_rem = mask ^ actual_sub;

                    int sub_offset = actual_sub * V;
                    int rem_offset = actual_rem * V;
                    int mask_offset = mask * V;

                    for (int i = 0; i < V; ++i) {
                        long long val = dp[sub_offset + i] + dp[rem_offset + i];
                        if (val < dp[mask_offset + i]) {
                            dp[mask_offset + i] = val;
                            prev_mask_arr[mask_offset + i] = actual_sub;
                            prev_u_arr[mask_offset + i] = i;
                            prev_edge_arr[mask_offset + i] = -1;
                        }
                    }
                }
                
                int actual_sub = lsb;
                int actual_rem = rem_mask;
                int sub_offset = actual_sub * V;
                int rem_offset = actual_rem * V;
                int mask_offset = mask * V;

                for (int i = 0; i < V; ++i) {
                    long long val = dp[sub_offset + i] + dp[rem_offset + i];
                    if (val < dp[mask_offset + i]) {
                        dp[mask_offset + i] = val;
                        prev_mask_arr[mask_offset + i] = actual_sub;
                        prev_u_arr[mask_offset + i] = i;
                        prev_edge_arr[mask_offset + i] = -1;
                    }
                }
            }

            int mask_offset = mask * V;
            priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
            for (int i = 0; i < V; ++i) {
                if (dp[mask_offset + i] != INF) {
                    pq.push({dp[mask_offset + i], i});
                }
            }

            while (!pq.empty()) {
                auto [d, u] = pq.top();
                pq.pop();

                if (d > dp[mask_offset + u]) continue;

                for (auto& edge : adj[u]) {
                    int v = edge.to;
                    long long w = edge.weight;
                    if (d + w < dp[mask_offset + v]) {
                        dp[mask_offset + v] = d + w;
                        prev_mask_arr[mask_offset + v] = mask;
                        prev_u_arr[mask_offset + v] = u;
                        prev_edge_arr[mask_offset + v] = edge.id;
                        pq.push({dp[mask_offset + v], v});
                    }
                }
            }
        }

        long long min_cost = INF;
        int best_u = -1;
        int final_mask = (1 << K) - 1;
        int final_offset = final_mask * V;
        for (int i = 0; i < V; ++i) {
            if (dp[final_offset + i] < min_cost) {
                min_cost = dp[final_offset + i];
                best_u = i;
            }
        }

        if (min_cost < INF) {
            trace(final_mask, best_u);
        }

    } else {
        vector<long long> dist(V, INF);
        vector<int> source(V, -1);
        vector<int> parent_edge(V, -1);
        vector<int> parent_node(V, -1);

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        for (int t : terminals) {
            dist[t] = 0;
            source[t] = t;
            pq.push({0, t});
        }

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d > dist[u]) continue;

            for (auto& e : adj[u]) {
                int v = e.to;
                if (d + e.weight < dist[v]) {
                    dist[v] = d + e.weight;
                    source[v] = source[u];
                    parent_edge[v] = e.id;
                    parent_node[v] = u;
                    pq.push({dist[v], v});
                }
            }
        }

        struct Bridge {
            int u, v;
            long long weight;
            int original_u, original_v;
            int edge_id;
        };
        vector<Bridge> bridges;
        for (int i = 0; i < E; ++i) {
            int u = edge_list[i].u;
            int v = edge_list[i].v;
            if (source[u] != -1 && source[v] != -1 && source[u] != source[v]) {
                bridges.push_back({source[u], source[v], dist[u] + edge_list[i].weight + dist[v], u, v, i});
            }
        }

        sort(bridges.begin(), bridges.end(), [](const Bridge& a, const Bridge& b) {
            return a.weight < b.weight;
        });

        DSU dsu_terminals(V);
        vector<Bridge> mst_bridges;
        for (auto& b : bridges) {
            if (dsu_terminals.unite(b.u, b.v)) {
                mst_bridges.push_back(b);
            }
        }

        for (auto& b : mst_bridges) {
            candidate_edges_set.insert(b.edge_id);
            int curr = b.original_u;
            while (curr != b.u) {
                candidate_edges_set.insert(parent_edge[curr]);
                curr = parent_node[curr];
            }
            curr = b.original_v;
            while (curr != b.v) {
                candidate_edges_set.insert(parent_edge[curr]);
                curr = parent_node[curr];
            }
        }
    }

    vector<OriginalEdge> candidate_edges;
    for (int id : candidate_edges_set) {
        candidate_edges.push_back(edge_list[id]);
    }

    sort(candidate_edges.begin(), candidate_edges.end(), [](const OriginalEdge& a, const OriginalEdge& b) {
        return a.weight < b.weight;
    });

    DSU dsu_final(V);
    vector<int> tree_edges;
    for (auto& e : candidate_edges) {
        if (dsu_final.unite(e.u, e.v)) {
            tree_edges.push_back(e.id);
        }
    }

    vector<int> degree(V, 0);
    vector<vector<pair<int, int>>> tree_adj(V);
    for (int id : tree_edges) {
        int u = edge_list[id].u;
        int v = edge_list[id].v;
        tree_adj[u].push_back({v, id});
        tree_adj[v].push_back({u, id});
        degree[u]++;
        degree[v]++;
    }

    queue<int> leaves;
    for (int i = 0; i < V; ++i) {
        if (degree[i] == 1 && !is_terminal[i]) {
            leaves.push(i);
        }
    }

    vector<bool> keep_edge(E, false);
    for (int id : tree_edges) keep_edge[id] = true;

    while (!leaves.empty()) {
        int u = leaves.front();
        leaves.pop();
        for (auto& edge : tree_adj[u]) {
            if (keep_edge[edge.second]) {
                keep_edge[edge.second] = false;
                degree[edge.first]--;
                if (degree[edge.first] == 1 && !is_terminal[edge.first]) {
                    leaves.push(edge.first);
                }
            }
        }
    }

    long long final_weight = 0;
    vector<int> final_edges;
    for (int i = 0; i < E; ++i) {
        if (keep_edge[i]) {
            final_weight += edge_list[i].weight;
            final_edges.push_back(i);
        }
    }

    cout << final_weight << " " << final_edges.size() << "\n";
    for (int id : final_edges) {
        cout << edge_list[id].u << " " << edge_list[id].v << "\n";
    }

    return 0;
}
