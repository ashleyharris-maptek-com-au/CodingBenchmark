OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%int = OpTypeInt 32 1

%_ptr_Input_vec3 = OpTypePointer Input %vec3
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal = OpVariable %_ptr_Input_vec3 Input

%_ptr_Output_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %_ptr_Output_vec4 Output

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo_var = OpVariable %_ptr_Uniform_UBO Uniform

%int_3 = OpConstant %int 3
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%float_0_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_0_3 = OpConstant %float 0.3
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_1_0 = OpConstant %float 1.0

%groundColor = OpConstantComposite %vec3 %float_0_3 %float_0_15 %float_0_05
%skyColor = OpConstantComposite %vec3 %float_0_4 %float_0_6 %float_1_0

%main = OpFunction %void None %func
%main_label = OpLabel

%in_normal_val = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %glsl Normalize %in_normal_val

%N_y = OpCompositeExtract %float %N 1
%t0 = OpFMul %float %N_y %float_0_5
%t = OpFAdd %float %t0 %float_0_5
%t_vec3 = OpCompositeConstruct %vec3 %t %t %t

%ambient = OpExtInst %vec3 %glsl FMix %groundColor %skyColor %t_vec3

%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%worldPos = OpLoad %vec3 %in_worldPos

%lightDir = OpFSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %glsl Normalize %lightDir

%N_dot_L = OpDot %float %N %L
%d = OpExtInst %float %glsl FMax %N_dot_L %float_0_0

%diffuse = OpFMul %float %float_0_3 %d
%diffuse_vec3 = OpCompositeConstruct %vec3 %diffuse %diffuse %diffuse

%final_rgb = OpFAdd %vec3 %ambient %diffuse_vec3

%r = OpCompositeExtract %float %final_rgb 0
%g = OpCompositeExtract %float %final_rgb 1
%b = OpCompositeExtract %float %final_rgb 2
%finalColor = OpCompositeConstruct %vec4 %r %g %b %float_1_0

OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
