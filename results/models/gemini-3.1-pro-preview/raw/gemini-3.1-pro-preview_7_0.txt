struct vec3 {
    double x, y, z;
    vec3 operator-(const vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    vec3 operator+(const vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    vec3& operator-=(const vec3& o) { x -= o.x; y -= o.y; z -= o.z; return *this; }
    vec3& operator+=(const vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
};

inline double dot(const vec3& a, const vec3& b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}

inline vec3 cross(const vec3& a, const vec3& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}

inline double length2(const vec3& a) {
    return dot(a, a);
}

inline double length(const vec3& a) {
    return sqrt(length2(a));
}

inline vec3 normalize(const vec3& a) {
    double l = length(a);
    return l > 1e-16 ? a * (1.0 / l) : vec3{1,0,0};
}

struct Triangle {
    int v[3];
};

struct AABB {
    vec3 bmin = {1e30, 1e30, 1e30};
    vec3 bmax = {-1e30, -1e30, -1e30};
    void expand(const vec3& p) {
        bmin.x = min(bmin.x, p.x); bmin.y = min(bmin.y, p.y); bmin.z = min(bmin.z, p.z);
        bmax.x = max(bmax.x, p.x); bmax.y = max(bmax.y, p.y); bmax.z = max(bmax.z, p.z);
    }
    bool intersects(const AABB& o) const {
        return (bmax.x >= o.bmin.x && bmin.x <= o.bmax.x &&
                bmax.y >= o.bmin.y && bmin.y <= o.bmax.y &&
                bmax.z >= o.bmin.z && bmin.z <= o.bmax.z);
    }
};

struct BVHNode {
    AABB box;
    int left, right;
    int tri_start, tri_count;
};

struct Mesh {
    vector<vec3> vertices;
    vector<Triangle> faces;
    vector<vec3> normals;
    
    vector<BVHNode> nodes;
    vector<int> bvh_tri_indices;
    int bvh_root = -1;
};

inline void compute_normals(Mesh& mesh) {
    mesh.normals.resize(mesh.faces.size());
    for (size_t i = 0; i < mesh.faces.size(); ++i) {
        vec3 v0 = mesh.vertices[mesh.faces[i].v[0]];
        vec3 v1 = mesh.vertices[mesh.faces[i].v[1]];
        vec3 v2 = mesh.vertices[mesh.faces[i].v[2]];
        mesh.normals[i] = normalize(cross(v1 - v0, v2 - v0));
    }
}

int build_bvh(vector<int>& tri_indices, int start, int count, Mesh& mesh) {
    AABB box;
    for (int i = 0; i < count; ++i) {
        int idx = tri_indices[start + i];
        box.expand(mesh.vertices[mesh.faces[idx].v[0]]);
        box.expand(mesh.vertices[mesh.faces[idx].v[1]]);
        box.expand(mesh.vertices[mesh.faces[idx].v[2]]);
    }
    
    if (count <= 4) {
        mesh.nodes.push_back({box, -1, -1, start, count});
        return mesh.nodes.size() - 1;
    }
    
    vec3 extent = box.bmax - box.bmin;
    int axis = 0;
    if (extent.y > extent.x) axis = 1;
    if (extent.z > extent.y && extent.z > extent.x) axis = 2;
    
    std::sort(tri_indices.begin() + start, tri_indices.begin() + start + count, [&](int a, int b) {
        vec3 ca = (mesh.vertices[mesh.faces[a].v[0]] + mesh.vertices[mesh.faces[a].v[1]] + mesh.vertices[mesh.faces[a].v[2]]) * (1.0/3.0);
        vec3 cb = (mesh.vertices[mesh.faces[b].v[0]] + mesh.vertices[mesh.faces[b].v[1]] + mesh.vertices[mesh.faces[b].v[2]]) * (1.0/3.0);
        if (axis == 0) return ca.x < cb.x;
        if (axis == 1) return ca.y < cb.y;
        return ca.z < cb.z;
    });
    
    int mid = count / 2;
    int node_idx = mesh.nodes.size();
    mesh.nodes.push_back({box, -1, -1, -1, -1}); 
    
    int left = build_bvh(tri_indices, start, mid, mesh);
    int right = build_bvh(tri_indices, start + mid, count - mid, mesh);
    
    mesh.nodes[node_idx].left = left;
    mesh.nodes[node_idx].right = right;
    
    return node_idx;
}

void query_bvh(int bvh_root, const AABB& box, const Mesh& mesh, vector<int>& result) {
    if (bvh_root == -1) return;
    vector<int> stack;
    stack.push_back(bvh_root);
    while (!stack.empty()) {
        int node_idx = stack.back();
        stack.pop_back();
        if (!mesh.nodes[node_idx].box.intersects(box)) continue;
        
        if (mesh.nodes[node_idx].left == -1) {
            for (int i = 0; i < mesh.nodes[node_idx].tri_count; ++i) {
                result.push_back(mesh.bvh_tri_indices[mesh.nodes[node_idx].tri_start + i]);
            }
        } else {
            stack.push_back(mesh.nodes[node_idx].left);
            stack.push_back(mesh.nodes[node_idx].right);
        }
    }
}

inline bool tri_tri_intersect_fast(const vec3& V0, const vec3& V1, const vec3& V2,
                                   const vec3& U0, const vec3& U1, const vec3& U2, double eps) {
    vec3 N1 = cross(V1 - V0, V2 - V0);
    double len1 = length(N1);
    if (len1 < 1e-16) return false;
    N1 = N1 * (1.0 / len1);
    double d1 = dot(N1, V0);
    
    double du0 = dot(N1, U0) - d1;
    double du1 = dot(N1, U1) - d1;
    double du2 = dot(N1, U2) - d1;
    if ((du0 > eps && du1 > eps && du2 > eps) ||
        (du0 < -eps && du1 < -eps && du2 < -eps)) return false;
        
    vec3 N2 = cross(U1 - U0, U2 - U0);
    double len2 = length(N2);
    if (len2 < 1e-16) return false;
    N2 = N2 * (1.0 / len2);
    double d2 = dot(N2, U0);
    
    double dv0 = dot(N2, V0) - d2;
    double dv1 = dot(N2, V1) - d2;
    double dv2 = dot(N2, V2) - d2;
    if ((dv0 > eps && dv1 > eps && dv2 > eps) ||
        (dv0 < -eps && dv1 < -eps && dv2 < -eps)) return false;
        
    return true;
}

void split_polygon(const vector<vec3>& P, const vec3& N, double d, double eps, vector<vec3>& P_front, vector<vec3>& P_back) {
    int n = P.size();
    if (n == 0) return;
    vector<double> dists(n);
    int pos = 0, neg = 0;
    for (int i=0; i<n; ++i) {
        dists[i] = dot(N, P[i]) - d;
        if (dists[i] > eps) pos++;
        else if (dists[i] < -eps) neg++;
        else dists[i] = 0; 
    }
    if (pos == 0 && neg == 0) {
        P_front = P; return;
    }
    if (pos == 0) { P_back = P; return; }
    if (neg == 0) { P_front = P; return; }
    
    for (int i=0; i<n; ++i) {
        int j = (i + 1) % n;
        double d1 = dists[i], d2 = dists[j];
        vec3 p1 = P[i], p2 = P[j];
        
        if (d1 >= 0) P_front.push_back(p1);
        if (d1 <= 0) P_back.push_back(p1);
        
        if ((d1 > 0 && d2 < 0) || (d1 < 0 && d2 > 0)) {
            double t = d1 / (d1 - d2);
            vec3 p_intersect = p1 + (p2 - p1) * t;
            P_front.push_back(p_intersect);
            P_back.push_back(p_intersect);
        }
    }
}

void clean_polygon(vector<vec3>& P, double eps) {
    if (P.empty()) return;
    vector<vec3> res;
    res.push_back(P[0]);
    for (size_t i = 1; i < P.size(); ++i) {
        if (length2(P[i] - res.back()) > eps*eps) {
            res.push_back(P[i]);
        }
    }
    while (res.size() > 1 && length2(res.back() - res[0]) <= eps*eps) {
        res.pop_back();
    }
    if (res.size() < 3) P.clear();
    else P = res;
}

inline double point_triangle_sq_dist(const vec3& p, const vec3& a, const vec3& b, const vec3& c) {
    vec3 ab = b - a;
    vec3 ac = c - a;
    vec3 ap = p - a;
    double d1 = dot(ab, ap);
    double d2 = dot(ac, ap);
    if (d1 <= 0.0 && d2 <= 0.0) return length2(ap);
    vec3 bp = p - b;
    double d3 = dot(ab, bp);
    double d4 = dot(ac, bp);
    if (d3 >= 0.0 && d4 <= d3) return length2(bp);
    double vc = d1*d4 - d3*d2;
    if (vc <= 0.0 && d1 >= 0.0 && d3 <= 0.0) {
        double v = d1 / (d1 - d3);
        return length2(a + ab * v - p);
    }
    vec3 cp = p - c;
    double d5 = dot(ab, cp);
    double d6 = dot(ac, cp);
    if (d6 >= 0.0 && d5 <= d6) return length2(cp);
    double vb = d5*d2 - d1*d6;
    if (vb <= 0.0 && d2 >= 0.0 && d6 <= 0.0) {
        double w = d2 / (d2 - d6);
        return length2(a + ac * w - p);
    }
    double va = d3*d6 - d5*d4;
    if (va <= 0.0 && (d4 - d3) >= 0.0 && (d5 - d6) >= 0.0) {
        double w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
        return length2(b + (c - b) * w - p);
    }
    double denom = 1.0 / (va + vb + vc);
    double v = vb * denom;
    double w = vc * denom;
    return length2(a + ab * v + ac * w - p);
}

inline bool ray_aabb_intersect(const vec3& orig, const vec3& inv_dir, const AABB& box) {
    double tx1 = (box.bmin.x - orig.x) * inv_dir.x;
    double tx2 = (box.bmax.x - orig.x) * inv_dir.x;
    double tmin = min(tx1, tx2);
    double tmax = max(tx1, tx2);
    
    double ty1 = (box.bmin.y - orig.y) * inv_dir.y;
    double ty2 = (box.bmax.y - orig.y) * inv_dir.y;
    tmin = max(tmin, min(ty1, ty2));
    tmax = min(tmax, max(ty1, ty2));
    
    double tz1 = (box.bmin.z - orig.z) * inv_dir.z;
    double tz2 = (box.bmax.z - orig.z) * inv_dir.z;
    tmin = max(tmin, min(tz1, tz2));
    tmax = min(tmax, max(tz1, tz2));
    
    return tmax >= max(0.0, tmin);
}

inline bool ray_triangle_intersect(const vec3& orig, const vec3& dir,
                                   const vec3& v0, const vec3& v1, const vec3& v2,
                                   double& t, double& u, double& v) {
    vec3 edge1 = v1 - v0;
    vec3 edge2 = v2 - v0;
    vec3 pvec = cross(dir, edge2);
    double det = dot(edge1, pvec);
    if (abs(det) < 1e-12) return false;
    double inv_det = 1.0 / det;
    vec3 tvec = orig - v0;
    u = dot(tvec, pvec) * inv_det;
    if (u < 0.0 || u > 1.0) return false;
    vec3 qvec = cross(tvec, edge1);
    v = dot(dir, qvec) * inv_det;
    if (v < 0.0 || u + v > 1.0) return false;
    t = dot(edge2, qvec) * inv_det;
    return true;
}

vec3 random_dir() {
    static mt19937 gen(1337);
    static uniform_real_distribution<double> dist(-1.0, 1.0);
    while (true) {
        vec3 v = {dist(gen), dist(gen), dist(gen)};
        double l2 = length2(v);
        if (l2 > 0.01 && l2 <= 1.0) {
            return v * (1.0 / sqrt(l2));
        }
    }
}

enum class RayHit { INSIDE, OUTSIDE, ON_BOUNDARY };

struct ClassifyResult {
    RayHit type;
    int closest_tri_idx;
};

ClassifyResult classify_point(const vec3& p, int bvh_root, const Mesh& mesh, double eps) {
    AABB p_box;
    p_box.bmin = p - vec3{eps, eps, eps};
    p_box.bmax = p + vec3{eps, eps, eps};
    vector<int> close_tris;
    query_bvh(bvh_root, p_box, mesh, close_tris);
    
    double min_dist = eps * eps;
    int closest_idx = -1;
    
    for (int t_idx : close_tris) {
        const auto& tri = mesh.faces[t_idx];
        vec3 v0 = mesh.vertices[tri.v[0]];
        vec3 v1 = mesh.vertices[tri.v[1]];
        vec3 v2 = mesh.vertices[tri.v[2]];
        double d2 = point_triangle_sq_dist(p, v0, v1, v2);
        if (d2 < min_dist) {
            min_dist = d2;
            closest_idx = t_idx;
        }
    }
    
    if (closest_idx != -1) {
        return {RayHit::ON_BOUNDARY, closest_idx};
    }
    
    for (int attempt = 0; attempt < 10; ++attempt) {
        vec3 dir = random_dir();
        vec3 inv_dir = {1.0/dir.x, 1.0/dir.y, 1.0/dir.z};
        int hit_count = 0;
        bool retry = false;
        
        vector<int> stack;
        stack.push_back(bvh_root);
        
        while (!stack.empty()) {
            int node_idx = stack.back();
            stack.pop_back();
            
            if (!ray_aabb_intersect(p, inv_dir, mesh.nodes[node_idx].box)) continue;
            
            if (mesh.nodes[node_idx].left == -1) {
                for (int i = 0; i < mesh.nodes[node_idx].tri_count; ++i) {
                    int t_idx = mesh.bvh_tri_indices[mesh.nodes[node_idx].tri_start + i];
                    const auto& tri = mesh.faces[t_idx];
                    vec3 v0 = mesh.vertices[tri.v[0]];
                    vec3 v1 = mesh.vertices[tri.v[1]];
                    vec3 v2 = mesh.vertices[tri.v[2]];
                    
                    double t, u, v;
                    if (ray_triangle_intersect(p, dir, v0, v1, v2, t, u, v)) {
                        if (t > eps) {
                            if (u < 1e-4 || v < 1e-4 || (1.0 - u - v) < 1e-4) {
                                retry = true; break;
                            }
                            hit_count++;
                        }
                    }
                }
            } else {
                stack.push_back(mesh.nodes[node_idx].left);
                stack.push_back(mesh.nodes[node_idx].right);
            }
            if (retry) break;
        }
        
        if (!retry) {
            return { (hit_count % 2 == 1) ? RayHit::INSIDE : RayHit::OUTSIDE, -1 };
        }
    }
    return {RayHit::OUTSIDE, -1};
}

struct Grid {
    double cell_size;
    unordered_map<int64_t, vector<int>> cells;
    
    Grid(double s) : cell_size(s) {}
    
    int64_t hash(int64_t x, int64_t y, int64_t z) const {
        return (x * 73856093) ^ (y * 19349663) ^ (z * 83492791);
    }
    
    void insert(const vec3& p, int idx) {
        int64_t x = std::floor(p.x / cell_size);
        int64_t y = std::floor(p.y / cell_size);
        int64_t z = std::floor(p.z / cell_size);
        cells[hash(x, y, z)].push_back(idx);
    }
    
    vector<int> query(const vec3& p, double radius) const {
        vector<int> res;
        int64_t x0 = std::floor((p.x - radius) / cell_size);
        int64_t x1 = std::floor((p.x + radius) / cell_size);
        int64_t y0 = std::floor((p.y - radius) / cell_size);
        int64_t y1 = std::floor((p.y + radius) / cell_size);
        int64_t z0 = std::floor((p.z - radius) / cell_size);
        int64_t z1 = std::floor((p.z + radius) / cell_size);
        
        for (int64_t x = x0; x <= x1; ++x) {
            for (int64_t y = y0; y <= y1; ++y) {
                for (int64_t z = z0; z <= z1; ++z) {
                    auto it = cells.find(hash(x, y, z));
                    if (it != cells.end()) {
                        res.insert(res.end(), it->second.begin(), it->second.end());
                    }
                }
            }
        }
        return res;
    }
    
    vector<int> query(const AABB& box, const vector<vec3>& vertices) const {
        vector<int> res;
        int64_t x0 = std::floor(box.bmin.x / cell_size);
        int64_t x1 = std::floor(box.bmax.x / cell_size);
        int64_t y0 = std::floor(box.bmin.y / cell_size);
        int64_t y1 = std::floor(box.bmax.y / cell_size);
        int64_t z0 = std::floor(box.bmin.z / cell_size);
        int64_t z1 = std::floor(box.bmax.z / cell_size);
        
        int64_t num_cells = (x1 - x0 + 1) * (y1 - y0 + 1) * (z1 - z0 + 1);
        if (num_cells > 8000) {
            for (const auto& pair : cells) {
                for (int idx : pair.second) {
                    const vec3& p = vertices[idx];
                    if (p.x >= box.bmin.x && p.x <= box.bmax.x &&
                        p.y >= box.bmin.y && p.y <= box.bmax.y &&
                        p.z >= box.bmin.z && p.z <= box.bmax.z) {
                        res.push_back(idx);
                    }
                }
            }
            return res;
        }
        
        for (int64_t x = x0; x <= x1; ++x) {
            for (int64_t y = y0; y <= y1; ++y) {
                for (int64_t z = z0; z <= z1; ++z) {
                    auto it = cells.find(hash(x, y, z));
                    if (it != cells.end()) {
                        for (int idx : it->second) {
                            const vec3& p = vertices[idx];
                            if (p.x >= box.bmin.x && p.x <= box.bmax.x &&
                                p.y >= box.bmin.y && p.y <= box.bmax.y &&
                                p.z >= box.bmin.z && p.z <= box.bmax.z) {
                                res.push_back(idx);
                            }
                        }
                    }
                }
            }
        }
        return res;
    }
};

double dist_point_segment(const vec3& p, const vec3& a, const vec3& b) {
    vec3 ab = b - a;
    double l2 = length2(ab);
    if (l2 < 1e-16) return length(p - a);
    double t = dot(p - a, ab) / l2;
    if (t <= 0.0) return length(p - a);
    if (t >= 1.0) return length(p - b);
    return length(p - (a + ab * t));
}

            for (auto& P : P_list) {
                for (size_t k = 1; k + 1 < P.size(); ++k) {
                    vec3 pv0 = P[0], pv1 = P[k], pv2 = P[k+1];
                    vec3 C = (pv0 + pv1 + pv2) * (1.0/3.0);
                    vec3 N_sub = cross(pv1 - pv0, pv2 - pv0);
                    if (length2(N_sub) < 1e-16) continue;
                    N_sub = normalize(N_sub);
                    
                    ClassifyResult res = classify_point(C, mB.bvh_root, mB, eps);
                    bool keep = false;
                    
                    if (res.type == RayHit::ON_BOUNDARY) {
                        vec3 N_closest = mB.normals[res.closest_tri_idx];
                        if (dot(N_sub, N_closest) > 0.0) {
                            keep = is_A;
                        }
                    } else if (res.type == RayHit::OUTSIDE) {
                        keep = true;
                    }
                    
                    if (keep) {
                        soup.push_back({{pv0, pv1, pv2}});
                    }
                }
            }
        }
    };
    
    process_mesh(mesh_A, mesh_B, true);
    process_mesh(mesh_B, mesh_A, false);
    
    vector<vec3> out_v;
    vector<Triangle> out_tris;
    Grid weld_grid(eps * 5.0);
    
    for (const auto& tri : soup) {
        Triangle t_idx;
        for (int i = 0; i < 3; ++i) {
            vec3 p = tri.v[i];
            vector<int> neighbors = weld_grid.query(p, eps);
            int match = -1;
            for (int n : neighbors) {
                if (length2(out_v[n] - p) < eps*eps) {
                    match = n; break;
                }
            }
            if (match != -1) {
                t_idx.v[i] = match;
            } else {
                t_idx.v[i] = out_v.size();
                weld_grid.insert(p, out_v.size());
                out_v.push_back(p);
            }
        }
        if (t_idx.v[0] != t_idx.v[1] && t_idx.v[1] != t_idx.v[2] && t_idx.v[2] != t_idx.v[0]) {
            out_tris.push_back(t_idx);
        }
    }
    
    double avg_len = 0;
    int count = 0;
    for (const auto& tri : out_tris) {
        vec3 v0 = out_v[tri.v[0]];
        vec3 v1 = out_v[tri.v[1]];
        vec3 v2 = out_v[tri.v[2]];
        avg_len += length(v1 - v0) + length(v2 - v1) + length(v0 - v2);
        count += 3;
    }
    if (count > 0) avg_len /= count;
    double t_cell_size = max(avg_len, eps * 10.0);
    
    Grid t_grid(t_cell_size);
    for (size_t i = 0; i < out_v.size(); ++i) {
        t_grid.insert(out_v[i], i);
    }
    
    vector<Triangle> final_tris;
    vector<Triangle> pending = out_tris;
    
    while (!pending.empty()) {
        Triangle tri = pending.back();
        pending.pop_back();
        
        vec3 v0 = out_v[tri.v[0]];
        vec3 v1 = out_v[tri.v[1]];
        vec3 v2 = out_v[tri.v[2]];
        AABB box;
        box.expand(v0); box.expand(v1); box.expand(v2);
        box.bmin -= vec3{eps, eps, eps};
        box.bmax += vec3{eps, eps, eps};
        
        vector<int> candidates = t_grid.query(box, out_v);
        bool split_tri = false;
        
        for (int w_idx : candidates) {
            if (w_idx == tri.v[0] || w_idx == tri.v[1] || w_idx == tri.v[2]) continue;
            vec3 w = out_v[w_idx];
            
            if (dist_point_segment(w, v0, v1) < eps) {
                pending.push_back({tri.v[0], w_idx, tri.v[2]});
                pending.push_back({w_idx, tri.v[1], tri.v[2]});
                split_tri = true; break;
            }
            if (dist_point_segment(w, v1, v2) < eps) {
                pending.push_back({tri.v[1], w_idx, tri.v[0]});
                pending.push_back({w_idx, tri.v[2], tri.v[0]});
                split_tri = true; break;
            }
            if (dist_point_segment(w, v2, v0) < eps) {
                pending.push_back({tri.v[2], w_idx, tri.v[1]});
                pending.push_back({w_idx, tri.v[0], tri.v[1]});
                split_tri = true; break;
            }
        }
        
        if (!split_tri) {
            final_tris.push_back(tri);
        }
    }
    
    cout << out_v.size() << " " << final_tris.size() << "\n";
    for (const auto& v : out_v) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& t : final_tris) {
        cout << t.v[0] << " " << t.v[1] << " " << t.v[2] << "\n";
    }
    
    return 0;
}
