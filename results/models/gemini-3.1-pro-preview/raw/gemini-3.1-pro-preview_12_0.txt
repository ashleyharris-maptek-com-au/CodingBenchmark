class Program
{
    struct Vector2 {
        public double X, Y;
        public Vector2(double x, double y) { X = x; Y = y; }
    }

    struct Vector3 {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public double this[int i] {
            get { return i == 0 ? X : i == 1 ? Y : Z; }
        }
    }

    struct Quat {
        public double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        public static Quat Multiply(Quat q1, Quat q2) {
            return new Quat(
                q1.W * q2.W - q1.X * q2.X - q1.Y * q2.Y - q1.Z * q2.Z,
                q1.W * q2.X + q1.X * q2.W + q1.Y * q2.Z - q1.Z * q2.Y,
                q1.W * q2.Y - q1.X * q2.Z + q1.Y * q2.W + q1.Z * q2.X,
                q1.W * q2.Z + q1.X * q2.Y - q1.Y * q2.X + q1.Z * q2.W
            );
        }
    }

    class FastScanner {
        Stream stream;
        byte[] buffer = new byte[65536];
        int head, tail;
        public FastScanner(Stream s) { stream = s; }
        public int Read() {
            if (head >= tail) {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
        public string NextToken() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return null;
                c = Read();
            }
            StringBuilder sb = new StringBuilder();
            while (c > 32) {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }
        public double NextDouble() {
            string t = NextToken();
            if (t == null) return 0;
            return double.Parse(t, CultureInfo.InvariantCulture);
        }
        public int NextInt() {
            string t = NextToken();
            if (t == null) return 0;
            return int.Parse(t);
        }
    }

    class RotationScore : IComparable<RotationScore> {
        public Quat Q;
        public long EstCount;
        public double W, H, D;
        public int CompareTo(RotationScore other) => other.EstCount.CompareTo(EstCount);
    }

    static Vector3 Rotate(Vector3 v, Quat q) {
        double tx = 2 * (q.Y * v.Z - q.Z * v.Y);
        double ty = 2 * (q.Z * v.X - q.X * v.Z);
        double tz = 2 * (q.X * v.Y - q.Y * v.X);
        return new Vector3(
            v.X + q.W * tx + q.Y * tz - q.Z * ty,
            v.Y + q.W * ty + q.Z * tx - q.X * tz,
            v.Z + q.W * tz + q.X * ty - q.Y * tx
        );
    }

    static List<Quat> GenerateOrtho() {
        List<Quat> ortho = new List<Quat>();
        ortho.Add(new Quat(1, 0, 0, 0));
        Quat qx = new Quat(0.7071067811865476, 0.7071067811865476, 0, 0);
        Quat qy = new Quat(0.7071067811865476, 0, 0.7071067811865476, 0);
        Quat qz = new Quat(0.7071067811865476, 0, 0, 0.7071067811865476);
        Queue<Quat> q = new Queue<Quat>();
        q.Enqueue(ortho[0]);
        while(q.Count > 0) {
            Quat curr = q.Dequeue();
            Quat[] nexts = new Quat[] { Quat.Multiply(curr, qx), Quat.Multiply(curr, qy), Quat.Multiply(curr, qz) };
            foreach(var n in nexts) {
                bool exists = false;
                foreach(var o in ortho) {
                    double dot = n.W * o.W + n.X * o.X + n.Y * o.Y + n.Z * o.Z;
                    if (Math.Abs(dot) > 0.999) { exists = true; break; }
                }
                if (!exists) {
                    ortho.Add(n);
                    q.Enqueue(n);
                }
            }
        }
        return ortho;
    }

    static bool Intersect2D(Vector2 p1, Vector2 p2, Vector2 p3, Vector2 p4, out Vector2 intersection) {
        intersection = new Vector2(0, 0);
        double d = (p1.X - p2.X) * (p3.Y - p4.Y) - (p1.Y - p2.Y) * (p3.X - p4.X);
        if (Math.Abs(d) < 1e-9) return false;
        double t = ((p1.X - p3.X) * (p3.Y - p4.Y) - (p1.Y - p3.Y) * (p3.X - p4.X)) / d;
        double u = ((p1.X - p3.X) * (p1.Y - p2.Y) - (p1.Y - p3.Y) * (p1.X - p2.X)) / d;
        if (t >= 0 && t <= 1 && u >= 0 && u <= 1) {
            intersection.X = p1.X + t * (p2.X - p1.X);
            intersection.Y = p1.Y + t * (p2.Y - p1.Y);
            return true;
        }
        return false;
    }

    static double FastS(Vector3[] verts, int[] faces, int axis, int res, double minU, double maxU, double minV, double maxV) {
        int iX = axis, iY = (axis+1)%3, iZ = (axis+2)%3;
        double maxT = 0;
        double stepU = (maxU - minU) / res;
        double stepV = (maxV - minV) / res;
        if (stepU < 1e-6 || stepV < 1e-6) return 0;
        
        double[] tMinU = new double[faces.Length/3];
        double[] tMaxU = new double[faces.Length/3];
        double[] tMinV = new double[faces.Length/3];
        double[] tMaxV = new double[faces.Length/3];
        for(int i=0; i<faces.Length; i+=3) {
            int idx = i/3;
            double u0 = verts[faces[i]][iY], v0 = verts[faces[i]][iZ];
            double u1 = verts[faces[i+1]][iY], v1 = verts[faces[i+1]][iZ];
            double u2 = verts[faces[i+2]][iY], v2 = verts[faces[i+2]][iZ];
            tMinU[idx] = Math.Min(u0, Math.Min(u1, u2));
            tMaxU[idx] = Math.Max(u0, Math.Max(u1, u2));
            tMinV[idx] = Math.Min(v0, Math.Min(v1, v2));
            tMaxV[idx] = Math.Max(v0, Math.Max(v1, v2));
        }
        
        for(int iu=0; iu<=res; iu++) {
            double u = minU + iu * stepU;
            for(int iv=0; iv<=res; iv++) {
                double v = minV + iv * stepV;
                double minH = double.MaxValue, maxH = double.MinValue;
                bool hit = false;
                
                for(int i=0; i<faces.Length; i+=3) {
                    int idx = i/3;
                    if (u < tMinU[idx] || u > tMaxU[idx] || v < tMinV[idx] || v > tMaxV[idx]) continue;
                    
                    Vector3 p0 = verts[faces[i]], p1 = verts[faces[i+1]], p2 = verts[faces[i+2]];
                    double p0u = p0[iY], p0v = p0[iZ], p0w = p0[iX];
                    double p1u = p1[iY], p1v = p1[iZ], p1w = p1[iX];
                    double p2u = p2[iY], p2v = p2[iZ], p2w = p2[iX];
                    
                    double e0 = (p1u - p0u) * (v - p0v) - (p1v - p0v) * (u - p0u);
                    double e1 = (p2u - p1u) * (v - p1v) - (p2v - p1v) * (u - p1u);
                    double e2 = (p0u - p2u) * (v - p2v) - (p0v - p2v) * (u - p2u);
                    
                    bool hasNeg = (e0 < -1e-7) || (e1 < -1e-7) || (e2 < -1e-7);
                    bool hasPos = (e0 > 1e-7) || (e1 > 1e-7) || (e2 > 1e-7);
                    
                    if (!(hasNeg && hasPos)) {
                        double nu = (p1v - p0v) * (p2w - p0w) - (p1w - p0w) * (p2v - p0v);
                        double nv = (p1w - p0w) * (p2u - p0u) - (p1u - p0u) * (p2w - p0w);
                        double nw = (p1u - p0u) * (p2v - p0v) - (p1v - p0v) * (p2u - p0u);
                        
                        if (Math.Abs(nw) > 1e-7) {
                            double hitW = p0w - (nu * (u - p0u) + nv * (v - p0v)) / nw;
                            if (hitW < minH) minH = hitW;
                            if (hitW > maxH) maxH = hitW;
                            hit = true;
                        }
                    }
                }
                if (hit) {
                    double th = maxH - minH;
                    if (th > maxT) maxT = th;
                }
            }
        }
        return maxT;
    }

    static double CalculateExactS(Vector3[] verts, int[] faces, int axis, double aabbW) {
        int iX = axis, iY = (axis + 1) % 3, iZ = (axis + 2) % 3;
        
        HashSet<long> edgeSet = new HashSet<long>();
        for(int i=0; i<faces.Length; i+=3) {
            int v0 = faces[i], v1 = faces[i+1], v2 = faces[i+2];
            if (v0 > v1) { int t=v0; v0=v1; v1=t; }
            edgeSet.Add(((long)v0 << 32) | (uint)v1);
            
            v0 = faces[i+1]; v1 = faces[i+2];
            if (v0 > v1) { int t=v0; v0=v1; v1=t; }
            edgeSet.Add(((long)v0 << 32) | (uint)v1);
            
            v0 = faces[i+2]; v1 = faces[i];
            if (v0 > v1) { int t=v0; v0=v1; v1=t; }
            edgeSet.Add(((long)v0 << 32) | (uint)v1);
        }
        
        List<Vector2> eStart = new List<Vector2>();
        List<Vector2> eEnd = new List<Vector2>();
        foreach(long e in edgeSet) {
            int v0 = (int)(e >> 32);
            int v1 = (int)(e & 0xFFFFFFFF);
            eStart.Add(new Vector2(verts[v0][iY], verts[v0][iZ]));
            eEnd.Add(new Vector2(verts[v1][iY], verts[v1][iZ]));
        }
        
        List<Vector2> pts = new List<Vector2>();
        foreach(var v in verts) pts.Add(new Vector2(v[iY], v[iZ]));
        
        long intersectionPairs = (long)eStart.Count * (eStart.Count - 1) / 2;
        if (intersectionPairs * (faces.Length/3) > 100_000_000) {
            return aabbW; 
        }
        
        for(int i=0; i<eStart.Count; i++) {
            for(int j=i+1; j<eStart.Count; j++) {
                if (Intersect2D(eStart[i], eEnd[i], eStart[j], eEnd[j], out Vector2 p)) {
                    pts.Add(p);
                }
            }
        }
        
        if ((long)pts.Count * (faces.Length/3) > 200_000_000) {
            return aabbW; 
        }
        
        double maxT = 0;
        double[] tMinY = new double[faces.Length/3];
        double[] tMaxY = new double[faces.Length/3];
        double[] tMinZ = new double[faces.Length/3];
        double[] tMaxZ = new double[faces.Length/3];
        for(int i=0; i<faces.Length; i+=3) {
            int idx = i/3;
            double y0 = verts[faces[i]][iY], z0 = verts[faces[i]][iZ];
            double y1 = verts[faces[i+1]][iY], z1 = verts[faces[i+1]][iZ];
            double y2 = verts[faces[i+2]][iY], z2 = verts[faces[i+2]][iZ];
            tMinY[idx] = Math.Min(y0, Math.Min(y1, y2));
            tMaxY[idx] = Math.Max(y0, Math.Max(y1, y2));
            tMinZ[idx] = Math.Min(z0, Math.Min(z1, z2));
            tMaxZ[idx] = Math.Max(z0, Math.Max(z1, z2));
        }
        
        foreach(var pt in pts) {
            double u = pt.X, v = pt.Y;
            double minH = double.MaxValue, maxH = double.MinValue;
            bool hit = false;
            
            for(int i=0; i<faces.Length; i+=3) {
                int idx = i/3;
                if (u < tMinY[idx]-1e-6 || u > tMaxY[idx]+1e-6 || v < tMinZ[idx]-1e-6 || v > tMaxZ[idx]+1e-6) continue;
                
                Vector3 p0 = verts[faces[i]], p1 = verts[faces[i+1]], p2 = verts[faces[i+2]];
                double p0u = p0[iY], p0v = p0[iZ], p0w = p0[iX];
                double p1u = p1[iY], p1v = p1[iZ], p1w = p1[iX];
                double p2u = p2[iY], p2v = p2[iZ], p2w = p2[iX];
                
                double e0 = (p1u - p0u) * (v - p0v) - (p1v - p0v) * (u - p0u);
                double e1 = (p2u - p1u) * (v - p1v) - (p2v - p1v) * (u - p1u);
                double e2 = (p0u - p2u) * (v - p2v) - (p0v - p2v) * (u - p2u);
                
                bool hasNeg = (e0 < -1e-6) || (e1 < -1e-6) || (e2 < -1e-6);
                bool hasPos = (e0 > 1e-6) || (e1 > 1e-6) || (e2 > 1e-6);
                
                if (!(hasNeg && hasPos)) {
                    double nu = (p1v - p0v) * (p2w - p0w) - (p1w - p0w) * (p2v - p0v);
                    double nv = (p1w - p0w) * (p2u - p0u) - (p1u - p0u) * (p2w - p0w);
                    double nw = (p1u - p0u) * (p2v - p0v) - (p1v - p0v) * (p2u - p0u);
                    
                    if (Math.Abs(nw) > 1e-9) {
                        double hitW = p0w - (nu * (u - p0u) + nv * (v - p0v)) / nw;
                        if (hitW < minH) minH = hitW;
                        if (hitW > maxH) maxH = hitW;
                        hit = true;
                    }
                }
            }
            if (hit) {
                double th = maxH - minH;
                if (th > maxT) maxT = th;
            }
        }
        return Math.Min(maxT + 1e-6, aabbW);
    }

    static void Main() {
        Thread.CurrentThread.CurrentCulture = CultureInfo.InvariantCulture;
        FastScanner sc = new FastScanner(Console.OpenStandardInput());
        
        string vToken = sc.NextToken();
        if (vToken == null) return;
        int V = int.Parse(vToken);
        int F = sc.NextInt();
        
        Vector3[] verts = new Vector3[V];
        for(int i=0; i<V; i++) {
            verts[i] = new Vector3(sc.NextDouble(), sc.NextDouble(), sc.NextDouble());
        }
        
        int[] faces = new int[F * 3];
        for(int i=0; i<F; i++) {
            faces[i*3] = sc.NextInt();
            faces[i*3+1] = sc.NextInt();
            faces[i*3+2] = sc.NextInt();
        }
        
        double W = sc.NextDouble();
        double H = sc.NextDouble();
        double D = sc.NextDouble();

        if (W <= 0 || H <= 0 || D <= 0) {
            Console.WriteLine(0);
            return;
        }

        Stopwatch sw = Stopwatch.StartNew();
        List<Quat> ortho = GenerateOrtho();
        List<RotationScore> scores = new List<RotationScore>();

        Random rnd = new Random(42);
        int rotIdx = 0;

        while(sw.ElapsedMilliseconds < 200000) { 
            Quat q;
            if (rotIdx < ortho.Count) q = ortho[rotIdx];
            else {
                double u1 = rnd.NextDouble();
                double u2 = rnd.NextDouble();
                double u3 = rnd.NextDouble();
                double sq1 = Math.Sqrt(1 - u1);
                double sq2 = Math.Sqrt(u1);
                double theta1 = 2 * Math.PI * u2;
                double theta2 = 2 * Math.PI * u3;
                q = new Quat(sq2 * Math.Cos(theta2), sq1 * Math.Sin(theta1), sq1 * Math.Cos(theta1), sq2 * Math.Sin(theta2));
            }
            
            Vector3[] rotVerts = new Vector3[V];
            for(int i=0; i<V; i++) rotVerts[i] = Rotate(verts[i], q);
            
            double minX = double.MaxValue, maxX = double.MinValue;
            double minY = double.MaxValue, maxY = double.MinValue;
            double minZ = double.MaxValue, maxZ = double.MinValue;
            for(int i=0; i<V; i++) {
                if (rotVerts[i].X < minX) minX = rotVerts[i].X;
                if (rotVerts[i].X > maxX) maxX = rotVerts[i].X;
                if (rotVerts[i].Y < minY) minY = rotVerts[i].Y;
                if (rotVerts[i].Y > maxY) maxY = rotVerts[i].Y;
                if (rotVerts[i].Z < minZ) minZ = rotVerts[i].Z;
                if (rotVerts[i].Z > maxZ) maxZ = rotVerts[i].Z;
            }
            double w = maxX - minX, h = maxY - minY, d = maxZ - minZ;
            
            if (w <= W && h <= H && d <= D) {
                long estCount = (long)(W/w) * (long)(H/h) * (long)(D/d);
                if (V < 2000) {
                    double sx = FastS(rotVerts, faces, 0, 10, minY, maxY, minZ, maxZ);
                    double sy = FastS(rotVerts, faces, 1, 10, minZ, maxZ, minX, maxX);
                    double sz = FastS(rotVerts, faces, 2, 10, minX, maxX, minY, maxY);
                    if (sx < 1e-6) sx = w;
                    if (sy < 1e-6) sy = h;
                    if (sz < 1e-6) sz = d;
                    long cx = (long)((W - w) / sx + 1) * (long)(H/h) * (long)(D/d);
                    long cy = (long)(W/w) * (long)((H - h) / sy + 1) * (long)(D/d);
                    long cz = (long)(W/w) * (long)(H/h) * (long)((D - d) / sz + 1);
                    estCount = Math.Max(estCount, Math.Max(cx, Math.Max(cy, cz)));
                }
                scores.Add(new RotationScore { Q = q, EstCount = estCount, W=w, H=h, D=d });
            }
            
            rotIdx++;
            if (rotIdx % 50 == 0 && sw.ElapsedMilliseconds > 200000) break;
        }

        scores.Sort();
        int topK = Math.Min(10, scores.Count);

        long bestFinalCount = 0;
        Quat bestQ = new Quat(1,0,0,0);
        int bestStrategy = 0; 
        double bestSx = 0, bestSy = 0, bestSz = 0;
        Vector3 bestMin = new Vector3();
        double bestW=0, bestH=0, bestD=0;

        for(int i=0; i<topK; i++) {
            Quat q = scores[i].Q;
            Vector3[] rotVerts = new Vector3[V];
            for(int j=0; j<V; j++) rotVerts[j] = Rotate(verts[j], q);
            
            double minX = double.MaxValue, maxX = double.MinValue;
            double minY = double.MaxValue, maxY = double.MinValue;
            double minZ = double.MaxValue, maxZ = double.MinValue;
            for(int j=0; j<V; j++) {
                if (rotVerts[j].X < minX) minX = rotVerts[j].X;
                if (rotVerts[j].X > maxX) maxX = rotVerts[j].X;
                if (rotVerts[j].Y < minY) minY = rotVerts[j].Y;
                if (rotVerts[j].Y > maxY) maxY = rotVerts[j].Y;
                if (rotVerts[j].Z < minZ) minZ = rotVerts[j].Z;
                if (rotVerts[j].Z > maxZ) maxZ = rotVerts[j].Z;
            }
            double w = maxX - minX, h = maxY - minY, d = maxZ - minZ;
            
            long c0 = (long)(W/w) * (long)(H/h) * (long)(D/d);
            if (c0 > bestFinalCount) {
                bestFinalCount = c0; bestQ = q; bestStrategy = 0;
                bestMin = new Vector3(minX, minY, minZ);
                bestW = w; bestH = h; bestD = d;
            }
            
            double sx = CalculateExactS(rotVerts, faces, 0, w);
            if (sx < 1e-6) sx = w;
            long c1 = (long)((W - w) / sx + 1) * (long)(H/h) * (long)(D/d);
            if (c1 > bestFinalCount) {
                bestFinalCount = c1; bestQ = q; bestStrategy = 1; bestSx = sx;
                bestMin = new Vector3(minX, minY, minZ);
                bestW = w; bestH = h; bestD = d;
            }
            
            double sy = CalculateExactS(rotVerts, faces, 1, h);
            if (sy < 1e-6) sy = h;
            long c2 = (long)(W/w) * (long)((H - h) / sy + 1) * (long)(D/d);
            if (c2 > bestFinalCount) {
                bestFinalCount = c2; bestQ = q; bestStrategy = 2; bestSy = sy;
                bestMin = new Vector3(minX, minY, minZ);
                bestW = w; bestH = h; bestD = d;
            }
            
            double sz = CalculateExactS(rotVerts, faces, 2, d);
            if (sz < 1e-6) sz = d;
            long c3 = (long)(W/w) * (long)(H/h) * (long)((D - d) / sz + 1);
            if (c3 > bestFinalCount) {
                bestFinalCount = c3; bestQ = q; bestStrategy = 3; bestSz = sz;
                bestMin = new Vector3(minX, minY, minZ);
                bestW = w; bestH = h; bestD = d;
            }
        }

        using (StreamWriter swOut = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1048576)) {
            swOut.WriteLine(bestFinalCount);
            if (bestFinalCount == 0) return;

            string qStr = $"{bestQ.W:0.00000} {bestQ.X:0.00000} {bestQ.Y:0.00000} {bestQ.Z:0.00000}";
            double rx = -bestMin.X;
            double ry = -bestMin.Y;
            double rz = -bestMin.Z;

            if (bestStrategy == 0) {
                long nx = (long)(W / bestW);
                long ny = (long)(H / bestH);
                long nz = (long)(D / bestD);
                for(long i=0; i<nx; i++) {
                    for(long j=0; j<ny; j++) {
                        for(long k=0; k<nz; k++) {
                            double tx = rx + i * bestW;
                            double ty = ry + j * bestH;
                            double tz = rz + k * bestD;
                            swOut.WriteLine($"{tx:0.00000} {ty:0.00000} {tz:0.00000} {qStr}");
                        }
                    }
                }
            }
            else if (bestStrategy == 1) {
                long nx = (long)((W - bestW) / bestSx) + 1;
                long ny = (long)(H / bestH);
                long nz = (long)(D / bestD);
                for(long i=0; i<nx; i++) {
                    for(long j=0; j<ny; j++) {
                        for(long k=0; k<nz; k++) {
                            double tx = rx + i * bestSx;
                            double ty = ry + j * bestH;
                            double tz = rz + k * bestD;
                            swOut.WriteLine($"{tx:0.00000} {ty:0.00000} {tz:0.00000} {qStr}");
                        }
                    }
                }
            }
            else if (bestStrategy == 2) {
                long nx = (long)(W / bestW);
                long ny = (long)((H - bestH) / bestSy) + 1;
                long nz = (long)(D / bestD);
                for(long i=0; i<nx; i++) {
                    for(long j=0; j<ny; j++) {
                        for(long k=0; k<nz; k++) {
                            double tx = rx + i * bestW;
                            double ty = ry + j * bestSy;
                            double tz = rz + k * bestD;
                            swOut.WriteLine($"{tx:0.00000} {ty:0.00000} {tz:0.00000} {qStr}");
                        }
                    }
                }
            }
            else if (bestStrategy == 3) {
                long nx = (long)(W / bestW);
                long ny = (long)(H / bestH);
                long nz = (long)((D - bestD) / bestSz) + 1;
                for(long i=0; i<nx; i++) {
                    for(long j=0; j<ny; j++) {
                        for(long k=0; k<nz; k++) {
                            double tx = rx + i * bestW;
                            double ty = ry + j * bestH;
                            double tz = rz + k * bestSz;
                            swOut.WriteLine($"{tx:0.00000} {ty:0.00000} {tz:0.00000} {qStr}");
                        }
                    }
                }
            }
        }
    }
