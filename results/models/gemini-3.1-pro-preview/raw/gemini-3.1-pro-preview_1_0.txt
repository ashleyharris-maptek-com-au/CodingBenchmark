#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstdint>
#include <numeric>

using namespace std;

#pragma optimize("t", on)

struct Point {
    double x, y;
};

inline double dist_sq(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx*dx + dy*dy;
}

inline double dist(const Point& a, const Point& b) {
    return std::sqrt(dist_sq(a, b));
}

double compute_tour_length(const vector<int>& tour, const vector<Point>& pts) {
    double d = 0;
    int n = tour.size();
    for (int i = 0; i < n - 1; i++) {
        d += dist(pts[tour[i]], pts[tour[i+1]]);
    }
    d += dist(pts[tour[n-1]], pts[tour[0]]);
    return d;
}

void print_tour(const vector<int>& tour, int N) {
    int start_idx = 0;
    for (int i = 0; i < N; i++) {
        if (tour[i] == 0) {
            start_idx = i;
            break;
        }
    }
    for (int i = 0; i < N; i++) {
        cout << tour[(start_idx + i) % N] << (i == N - 1 ? "" : " ");
    }
    cout << "\n";
}

void solve_trivial(int N, const vector<Point>& pts) {
    vector<int> p(N);
    iota(p.begin(), p.end(), 0);
    double best_dist = 1e38;
    vector<int> best_p = p;
    
    do {
        double d = 0;
        for (int i = 0; i < N - 1; i++) d += dist(pts[p[i]], pts[p[i+1]]);
        d += dist(pts[p[N-1]], pts[p[0]]);
        if (d < best_dist) {
            best_dist = d;
            best_p = p;
        }
    } while (next_permutation(p.begin() + 1, p.end()));
    
    print_tour(best_p, N);
}

void apply_2opt(vector<int>& tour, int N, const vector<Point>& pts, int max_passes = 20) {
    bool improved = true;
    int passes = 0;
    
    vector<double> edge_len(N);
    for (int i = 0; i < N; i++) {
        edge_len[i] = dist(pts[tour[i]], pts[tour[(i+1)%N]]);
    }
    
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < N - 1; i++) {
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;
                
                double d1 = edge_len[i] + edge_len[j];
                int p1 = tour[i], p2 = tour[i+1];
                int p3 = tour[j], p4 = tour[(j+1)%N];
                
                double a = dist_sq(pts[p1], pts[p3]);
                double b = dist_sq(pts[p2], pts[p4]);
                
                if (a + b < d1 * d1 + 1e-7) {
                    double sqa = std::sqrt(a);
                    double sqb = std::sqrt(b);
                    double d2 = sqa + sqb;
                    if (d2 < d1 - 1e-9) {
                        int left = i + 1, right = j;
                        while (left < right) {
                            swap(tour[left], tour[right]);
                            left++; right--;
                        }
                        
                        edge_len[i] = sqa;
                        edge_len[j] = sqb;
                        
                        int left_e = i + 1, right_e = j - 1;
                        while (left_e < right_e) {
                            swap(edge_len[left_e], edge_len[right_e]);
                            left_e++; right_e--;
                        }
                        
                        improved = true;
                    }
                }
            }
        }
    }
}

vector<int> build_nn(int start_node, int N, const vector<Point>& pts) {
    vector<int> tour;
    tour.reserve(N);
    vector<bool> visited(N, false);
    tour.push_back(start_node);
    visited[start_node] = true;
    int curr = start_node;
    for (int step = 1; step < N; step++) {
        double min_d2 = 1e38;
        int next_node = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d2 = dist_sq(pts[curr], pts[i]);
                if (d2 < min_d2) {
                    min_d2 = d2;
                    next_node = i;
                }
            }
        }
        tour.push_back(next_node);
        visited[next_node] = true;
        curr = next_node;
    }
    return tour;
}

void solve_small(int N, const vector<Point>& pts) {
    int K = min(N, 100);
    double best_dist = 1e38;
    vector<int> best_tour;
    
    for (int i = 0; i < K; i++) {
        vector<int> tour = build_nn(i, N, pts);
        apply_2opt(tour, N, pts, 20);
        double d = compute_tour_length(tour, pts);
        if (d < best_dist) {
            best_dist = d;
            best_tour = std::move(tour);
        }
    }
    print_tour(best_tour, N);
}

void solve_medium(int N, const vector<Point>& pts) {
    vector<int> tour = build_nn(0, N, pts);
    int max_passes = (N > 5000) ? 5 : 20;
    apply_2opt(tour, N, pts, max_passes);
    print_tour(tour, N);
}

uint64_t hilbert_index(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int i = 31; i >= 0; i--) {
        uint32_t rx = (x >> i) & 1;
        uint32_t ry = (y >> i) & 1;
        d = (d << 2) | ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

void solve_large(int N, const vector<Point>& pts) {
    double min_x = 1e38, max_x = -1e38;
    double min_y = 1e38, max_y = -1e38;
    for (int i = 0; i < N; i++) {
        if (pts[i].x < min_x) min_x = pts[i].x;
        if (pts[i].x > max_x) max_x = pts[i].x;
        if (pts[i].y < min_y) min_y = pts[i].y;
        if (pts[i].y > max_y) max_y = pts[i].y;
    }
    
    double range_x = max_x - min_x;
    double range_y = max_y - min_y;
    double max_range = max(range_x, range_y);
    if (max_range == 0.0) max_range = 1.0;
    
    double scale = 4294967295.0 / max_range;
    
    vector<pair<uint64_t, int>> hilbert(N);
    for (int i = 0; i < N; i++) {
        uint32_t nx = min((uint64_t)4294967295ULL, (uint64_t)((pts[i].x - min_x) * scale));
        uint32_t ny = min((uint64_t)4294967295ULL, (uint64_t)((pts[i].y - min_y) * scale));
        hilbert[i] = {hilbert_index(nx, ny), i};
    }
    
    sort(hilbert.begin(), hilbert.end());
    
    vector<int> tour(N);
    for (int i = 0; i < N; i++) {
        tour[i] = hilbert[i].second;
    }
    
    int window = 50;
    if (N > 1000000) window = 20;
    if (N > 5000000) window = 10;
    
    vector<double> edge_len(N);
    for (int i = 0; i < N; i++) {
        edge_len[i] = dist(pts[tour[i]], pts[tour[(i+1)%N]]);
    }
    
    bool improved = true;
    int passes = 0;
    int max_passes = 10;
    
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < N - 1; i++) {
            int max_j = min(N, i + window);
            for (int j = i + 2; j < max_j; j++) {
                if (i == 0 && j == N - 1) continue;
                
                double d1 = edge_len[i] + edge_len[j];
                int p1 = tour[i], p2 = tour[i+1];
                int p3 = tour[j], p4 = tour[(j+1)%N];
                
                double a = dist_sq(pts[p1], pts[p3]);
                double b = dist_sq(pts[p2], pts[p4]);
                
                if (a + b < d1 * d1 + 1e-7) {
                    double sqa = std::sqrt(a);
                    double sqb = std::sqrt(b);
                    double d2 = sqa + sqb;
                    if (d2 < d1 - 1e-9) {
                        int left = i + 1, right = j;
                        while (left < right) {
                            swap(tour[left], tour[right]);
                            left++; right--;
                        }
                        
                        edge_len[i] = sqa;
                        edge_len[j] = sqb;
                        
                        int left_e = i + 1, right_e = j - 1;
                        while (left_e < right_e) {
                            swap(edge_len[left_e], edge_len[right_e]);
                            left_e++; right_e--;
                        }
                        
                        improved = true;
                    }
                }
            }
        }
    }
    
    print_tour(tour, N);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N;
    if (!(cin >> N)) return 0;
    
    vector<Point> pts(N);
    for (int i = 0; i < N; i++) {
        cin >> pts[i].x >> pts[i].y;
    }
    
    if (N <= 11) {
        solve_trivial(N, pts);
    } else if (N <= 1000) {
        solve_small(N, pts);
    } else if (N <= 30000) {
        solve_medium(N, pts);
    } else {
        solve_large(N, pts);
    }
    
    return 0;
}
