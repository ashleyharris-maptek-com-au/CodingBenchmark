Here's a complete, self-contained C++20 program that implements a chess bot meeting the specification. It:

- Parses FEN input.
- Generates legal moves (including castling, en-passant, promotions).
- Uses iterative deepening alpha-beta with quiescence and a simple evaluation function.
- Uses Zobrist hashing + transposition table and basic move ordering (captures / TT move).
- Respects the timeout and returns the best move found so far.
- Outputs moves in the requested disambiguated algebraic format:
  - piece letter (p for pawn, N/B/R/Q/K for others) + source square + (x if capture) + destination
  - promotions as /Q (or /R /B /N)
  - en-passant moves append ep
  - castling printed as O-O or O-O-O
  - appends + for check and # for checkmate when applicable

Compile with MSVC (cl) using /std:c++20. The program reads two lines from stdin:
timeout_in_seconds
fen_string

Example usage:
Input:
5
rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1

Output (for example):
pe2e4

Code:

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <array>
#include <chrono>
#include <random>
#include <unordered_map>
#include <algorithm>
#include <cstdint>
#include <cassert>

using namespace std;
using u64 = uint64_t;
using Clock = chrono::steady_clock;

static const int EMPTY = 0;
static const int WP = 1, WN = 2, WB = 3, WR = 4, WQ = 5, WK = 6;
static const int BP = -1, BN = -2, BB = -3, BR = -4, BQ = -5, BK = -6;

// Castling flags
static const int WKC = 1, WQC = 2, BKC = 4, BQC = 8;

inline int absPiece(int p) { return p < 0 ? -p : p; }
inline bool isWhite(int p) { return p > 0; }
inline bool isBlack(int p) { return p < 0; }

inline int fileOf(int sq) { return sq & 7; }
inline int rankOf(int sq) { return sq >> 3; }
inline string sqToStr(int sq) {
    string s; s.push_back('a' + fileOf(sq)); s.push_back('1' + rankOf(sq));
    return s;
}

char pieceLetterForOutput(int p) {
    int t = absPiece(p);
    if (t == 1) return 'p';
    if (t == 2) return 'N';
    if (t == 3) return 'B';
    if (t == 4) return 'R';
    if (t == 5) return 'Q';
    if (t == 6) return 'K';
    return '?';
}

// Move struct
struct Move {
    int from, to;
    int piece;     // moving piece (signed)
    int captured;  // captured piece (signed) or 0
    int promotion; // promotion piece type (abs: 2..5), 0 if none (use 2=N,3=B,4=R,5=Q mapping)
    bool isEnPassant;
    bool isCastle;
    Move() : from(0), to(0), piece(0), captured(0), promotion(0), isEnPassant(false), isCastle(false) {}
};

// Zobrist
u64 Z_piece[12][64];
u64 Z_side;
u64 Z_castle[16];
u64 Z_ep[64];

u64 rnd_u64() {
    static std::mt19937_64 rng(0x9E3779B97F4A7C15ULL);
    return rng();
}

void init_zobrist() {
    for (int i = 0; i < 12; ++i)
        for (int j = 0; j < 64; ++j)
            Z_piece[i][j] = rnd_u64();
    Z_side = rnd_u64();
    for (int i = 0; i < 16; ++i) Z_castle[i] = rnd_u64();
    for (int i = 0; i < 64; ++i) Z_ep[i] = rnd_u64();
}

// Map piece signed to index 0..11 (WP..WK, BP..BK)
int pieceToIndex(int p) {
    if (p > 0) return p - 1;
    return 6 + ((-p) - 1);
}

struct TTEntry {
    u64 key;
    int depth;
    int score;
    unsigned char flag; // 0: exact, 1: lowerbound, 2: upperbound
    Move best;
};

struct Board {
    array<int,64> b;
    bool whiteToMove;
    int castling; // bitmask WKC|WQC|BKC|BQC
    int enPassant; // square index or -1
    int halfmoveClock;
    int fullmoveNumber;
    u64 zobristKey;

    Board() {
        b.fill(EMPTY);
        whiteToMove = true;
        castling = 0;
        enPassant = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        zobristKey = 0;
    }

    void setPiece(int sq, int piece) {
        b[sq] = piece;
    }

    // Parse FEN
    bool loadFEN(const string &fen) {
        // Clear
        b.fill(EMPTY);
        enPassant = -1;
        castling = 0;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        zobristKey = 0;

        // Split by spaces
        vector<string> parts;
        string cur;
        for (char c : fen) {
            if (c == ' ') {
                if (!cur.empty()) { parts.push_back(cur); cur.clear(); }
            } else cur.push_back(c);
        }
        if (!cur.empty()) parts.push_back(cur);
        if (parts.size() < 4) return false;

        // Board
        string piecePlacement = parts[0];
        int idx = 0;
        int rank = 7, file = 0;
        for (size_t i = 0; i < piecePlacement.size(); ++i) {
            char c = piecePlacement[i];
            if (c == '/') { rank--; file = 0; continue; }
            if (isdigit((unsigned char)c)) {
                file += c - '0';
                continue;
            }
            int sq = rank * 8 + file;
            int piece = 0;
            if (c == 'P') piece = WP;
            else if (c == 'N') piece = WN;
            else if (c == 'B') piece = WB;
            else if (c == 'R') piece = WR;
            else if (c == 'Q') piece = WQ;
            else if (c == 'K') piece = WK;
            else if (c == 'p') piece = BP;
            else if (c == 'n') piece = BN;
            else if (c == 'b') piece = BB;
            else if (c == 'r') piece = BR;
            else if (c == 'q') piece = BQ;
            else if (c == 'k') piece = BK;
            else return false;
            if (file >= 8 || rank < 0) return false;
            b[sq] = piece;
            file++;
        }

        // Side to move
        whiteToMove = (parts[1] == "w");

        // Castling
        string cr = parts[2];
        if (cr.find('K') != string::npos) castling |= WKC;
        if (cr.find('Q') != string::npos) castling |= WQC;
        if (cr.find('k') != string::npos) castling |= BKC;
        if (cr.find('q') != string::npos) castling |= BQC;

        // En passant
        string ep = parts[3];
        if (ep != "-") {
            if (ep.size() == 2 && ep[0] >= 'a' && ep[0] <= 'h' && ep[1] >= '1' && ep[1] <= '8') {
                int file = ep[0] - 'a';
                int rank = ep[1] - '1';
                enPassant = rank * 8 + file;
            } else {
                enPassant = -1;
            }
        } else enPassant = -1;

        if (parts.size() >= 5) halfmoveClock = stoi(parts[4]);
        if (parts.size() >= 6) fullmoveNumber = stoi(parts[5]);

        // Compute Zobrist
        computeZobrist();
        return true;
    }

    void computeZobrist() {
        zobristKey = 0;
        for (int sq = 0; sq < 64; ++sq) {
            int p = b[sq];
            if (p != EMPTY) {
                int idx = pieceToIndex(p);
                zobristKey ^= Z_piece[idx][sq];
            }
        }
        if (!whiteToMove) zobristKey ^= Z_side;
        zobristKey ^= Z_castle[castling & 15];
        if (enPassant != -1) zobristKey ^= Z_ep[enPassant & 63];
    }

    // Helper to find king square for color
    int kingSquare(bool white) const {
        int target = white ? WK : BK;
        for (int i = 0; i < 64; ++i) if (b[i] == target) return i;
        return -1;
    }
};

// Global helper arrays for simple move generation (knight/king/pawn attack offsets)
vector<int> knightOffsets = {17, 15, 10, 6, -6, -10, -15, -17};
vector<int> kingOffsets = {8, -8, 1, -1, 9, -9, 7, -7};

inline bool on_board(int sq) { return sq >= 0 && sq < 64; }

// Attack detection: is a square attacked by given color?
bool isSquareAttacked(const Board &bd, int sq, bool byWhite) {
    int r = rankOf(sq), f = fileOf(sq);
    // Pawn attacks
    if (byWhite) {
        if (r - 1 >= 0) {
            if (f - 1 >= 0) {
                int s = (r - 1) * 8 + (f - 1);
                if (bd.b[s] == WP) return true;
            }
            if (f + 1 <= 7) {
                int s = (r - 1) * 8 + (f + 1);
                if (bd.b[s] == WP) return true;
            }
        }
    } else {
        if (r + 1 <= 7) {
            if (f - 1 >= 0) {
                int s = (r + 1) * 8 + (f - 1);
                if (bd.b[s] == BP) return true;
            }
            if (f + 1 <= 7) {
                int s = (r + 1) * 8 + (f + 1);
                if (bd.b[s] == BP) return true;
            }
        }
    }

    // Knights
    for (int d : knightOffsets) {
        int t = sq + d;
        if (!on_board(t)) continue;
        // verify file differences to avoid wrap
        int df = fileOf(t) - fileOf(sq);
        int dr = rankOf(t) - rankOf(sq);
        if (abs(df) + abs(dr) != 3 || max(abs(df), abs(dr)) != 2) continue;
        int pc = bd.b[t];
        if (pc != EMPTY && ((byWhite && pc == WN) || (!byWhite && pc == BN))) return true;
    }

    // Sliding: rook/queen and bishop/queen
    // Directions and boundary checks
    // Rook-like directions: N(+8), S(-8), E(+1), W(-1)
    // Bishop-like: NE(+9), NW(+7), SE(-7), SW(-9)
    // N
    for (int t = sq + 8; t < 64; t += 8) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WR || pc == WQ) return true;
                break;
            } else {
                if (pc == BR || pc == BQ) return true;
                break;
            }
        }
    }
    // S
    for (int t = sq - 8; t >= 0; t -= 8) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WR || pc == WQ) return true;
                break;
            } else {
                if (pc == BR || pc == BQ) return true;
                break;
            }
        }
    }
    // E
    for (int t = sq + 1; t < 64 && fileOf(t) > fileOf(t - 1); ++t) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WR || pc == WQ) return true;
                break;
            } else {
                if (pc == BR || pc == BQ) return true;
                break;
            }
        }
    }
    // W
    for (int t = sq - 1; t >= 0 && fileOf(t) < fileOf(t + 1); --t) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WR || pc == WQ) return true;
                break;
            } else {
                if (pc == BR || pc == BQ) return true;
                break;
            }
        }
    }
    // NE (+9)
    for (int t = sq + 9; t < 64 && fileOf(t) > fileOf(t - 9); t += 9) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WB || pc == WQ) return true;
                break;
            } else {
                if (pc == BB || pc == BQ) return true;
                break;
            }
        }
    }
    // NW (+7)
    for (int t = sq + 7; t < 64 && fileOf(t) < fileOf(t - 7); t += 7) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WB || pc == WQ) return true;
                break;
            } else {
                if (pc == BB || pc == BQ) return true;
                break;
            }
        }
    }
    // SE (-7)
    for (int t = sq - 7; t >= 0 && fileOf(t) > fileOf(t + 7); t -= 7) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WB || pc == WQ) return true;
                break;
            } else {
                if (pc == BB || pc == BQ) return true;
                break;
            }
        }
    }
    // SW (-9)
    for (int t = sq - 9; t >= 0 && fileOf(t) < fileOf(t + 9); t -= 9) {
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite) {
                if (pc == WB || pc == WQ) return true;
                break;
            } else {
                if (pc == BB || pc == BQ) return true;
                break;
            }
        }
    }

    // King adjacency
    for (int d : kingOffsets) {
        int t = sq + d;
        if (!on_board(t)) continue;
        int df = abs(fileOf(t) - fileOf(sq));
        int dr = abs(rankOf(t) - rankOf(sq));
        if (max(df, dr) != 1) continue;
        int pc = bd.b[t];
        if (pc != EMPTY) {
            if (byWhite && pc == WK) return true;
            if (!byWhite && pc == BK) return true;
        }
    }
    return false;
}

// Generate pseudo-legal moves for side (not filtering moves that expose king to check)
void generatePseudoLegal(const Board &bd, vector<Move> &out) {
    out.clear();
    bool white = bd.whiteToMove;
    for (int from = 0; from < 64; ++from) {
        int pc = bd.b[from];
        if (pc == EMPTY) continue;
        if (white != (pc > 0)) continue;
        int absPc = absPiece(pc);
        if (absPc == 1) {
            // Pawn
            int dir = white ? 8 : -8;
            int startRank = white ? 1 : 6;
            int promoRank = white ? 7 : 0;
            int r = rankOf(from);
            int one = from + dir;
            if (on_board(one) && bd.b[one] == EMPTY) {
                Move m; m.from = from; m.to = one; m.piece = pc;
                if (rankOf(one) == promoRank) {
                    // promotions to Q,R,B,N (we store promotion as piece abs value)
                    int promos[4] = {5,4,3,2}; // Q R B N
                    for (int k = 0; k < 4; ++k) { m.promotion = promos[k]; out.push_back(m); }
                    m.promotion = 0;
                } else out.push_back(m);
                // two squares
                int two = from + dir*2;
                if (r == startRank && on_board(two) && bd.b[two] == EMPTY) {
                    Move m2; m2.from = from; m2.to = two; m2.piece = pc; m2.promotion = 0;
                    out.push_back(m2);
                }
            }
            // captures
            int caps[2] = {dir + 1, dir - 1};
            for (int d : caps) {
                int to = from + d;
                if (!on_board(to)) continue;
                int ff = fileOf(to), ffrom = fileOf(from);
                if (abs(ff - ffrom) != 1) continue;
                // Normal capture
                if (bd.b[to] != EMPTY && (white != (bd.b[to] > 0))) {
                    Move m; m.from = from; m.to = to; m.piece = pc; m.captured = bd.b[to];
                    if (rankOf(to) == promoRank) {
                        int promos[4] = {5,4,3,2};
                        for (int k = 0; k < 4; ++k) { m.promotion = promos[k]; out.push_back(m); }
                        m.promotion = 0;
                    } else out.push_back(m);
                }
                // En passant
                if (bd.enPassant != -1 && to == bd.enPassant) {
                    Move m; m.from = from; m.to = to; m.piece = pc; m.isEnPassant = true;
                    // captured pawn is behind target
                    int capSq = white ? (to - 8) : (to + 8);
                    m.captured = bd.b[capSq];
                    out.push_back(m);
                }
            }
        } else if (absPc == 2) {
            // Knight
            for (int d : knightOffsets) {
                int to = from + d;
                if (!on_board(to)) continue;
                int df = abs(fileOf(to) - fileOf(from));
                int dr = abs(rankOf(to) - rankOf(from));
                if ((df == 1 && dr == 2) || (df == 2 && dr == 1)) {
                    if (bd.b[to] == EMPTY || (bd.b[to] > 0) != (pc > 0)) {
                        Move m; m.from = from; m.to = to; m.piece = pc;
                        if (bd.b[to] != EMPTY) m.captured = bd.b[to];
                        out.push_back(m);
                    }
                }
            }
        } else if (absPc == 3 || absPc == 5) {
            // Bishop or Queen (diagonals)
            int dirs[4] = {9, 7, -7, -9};
            for (int dir : dirs) {
                int t = from;
                while (true) {
                    // boundary checks before stepping
                    if ((dir == 9 && (fileOf(t) == 7)) || (dir == 7 && (fileOf(t) == 0)) ||
                        (dir == -7 && (fileOf(t) == 7)) || (dir == -9 && (fileOf(t) == 0))) break;
                    t += dir;
                    if (!on_board(t)) break;
                    if (bd.b[t] == EMPTY) {
                        Move m; m.from = from; m.to = t; m.piece = pc; out.push_back(m);
                    } else {
                        if ((bd.b[t] > 0) != (pc > 0)) { Move m; m.from = from; m.to = t; m.piece = pc; m.captured = bd.b[t]; out.push_back(m); }
                        break;
                    }
                }
            }
            // If queen, rook like moves as well
            if (absPc == 5) {
                int rdirs[4] = {8, -8, 1, -1};
                for (int dir : rdirs) {
                    int t = from;
                    while (true) {
                        if ((dir == 1 && fileOf(t) == 7) || (dir == -1 && fileOf(t) == 0)) break;
                        t += dir;
                        if (!on_board(t)) break;
                        if (bd.b[t] == EMPTY) {
                            Move m; m.from = from; m.to = t; m.piece = pc; out.push_back(m);
                        } else {
                            if ((bd.b[t] > 0) != (pc > 0)) { Move m; m.from = from; m.to = t; m.piece = pc; m.captured = bd.b[t]; out.push_back(m); }
                            break;
                        }
                    }
                }
            }
        } else if (absPc == 4) {
            // Rook
            int dirs[4] = {8, -8, 1, -1};
            for (int dir : dirs) {
                int t = from;
                while (true) {
                    if ((dir == 1 && fileOf(t) == 7) || (dir == -1 && fileOf(t) == 0)) break;
                    t += dir;
                    if (!on_board(t)) break;
                    if (bd.b[t] == EMPTY) {
                        Move m; m.from = from; m.to = t; m.piece = pc; out.push_back(m);
                    } else {
                        if ((bd.b[t] > 0) != (pc > 0)) { Move m; m.from = from; m.to = t; m.piece = pc; m.captured = bd.b[t]; out.push_back(m); }
                        break;
                    }
                }
            }
        } else if (absPc == 6) {
            // King
            for (int d : kingOffsets) {
                int to = from + d;
                if (!on_board(to)) continue;
                int df = abs(fileOf(to) - fileOf(from)), dr = abs(rankOf(to) - rankOf(from));
                if (max(df, dr) != 1) continue;
                if (bd.b[to] == EMPTY || (bd.b[to] > 0) != (pc > 0)) {
                    Move m; m.from = from; m.to = to; m.piece = pc;
                    if (bd.b[to] != EMPTY) m.captured = bd.b[to];
                    out.push_back(m);
                }
            }
            // Castling
            if (pc == WK) {
                // White king on e1?
                if ((bd.castling & WKC) && bd.b[4] == WK && bd.b[5] == EMPTY && bd.b[6] == EMPTY) {
                    // Check not in check and squares f1/g1 not attacked
                    if (!isSquareAttacked(bd, 4, false) && !isSquareAttacked(bd, 5, false) && !isSquareAttacked(bd, 6, false)) {
                        Move m; m.from = 4; m.to = 6; m.piece = pc; m.isCastle = true; out.push_back(m);
                    }
                }
                if ((bd.castling & WQC) && bd.b[4] == WK && bd.b[3] == EMPTY && bd.b[2] == EMPTY && bd.b[1] == EMPTY) {
                    if (!isSquareAttacked(bd, 4, false) && !isSquareAttacked(bd, 3, false) && !isSquareAttacked(bd, 2, false)) {
                        Move m; m.from = 4; m.to = 2; m.piece = pc; m.isCastle = true; out.push_back(m);
                    }
                }
            } else if (pc == BK) {
                if ((bd.castling & BKC) && bd.b[60] == BK && bd.b[61] == EMPTY && bd.b[62] == EMPTY) {
                    if (!isSquareAttacked(bd, 60, true) && !isSquareAttacked(bd, 61, true) && !isSquareAttacked(bd, 62, true)) {
                        Move m; m.from = 60; m.to = 62; m.piece = pc; m.isCastle = true; out.push_back(m);
                    }
                }
                if ((bd.castling & BQC) && bd.b[60] == BK && bd.b[59] == EMPTY && bd.b[58] == EMPTY && bd.b[57] == EMPTY) {
                    if (!isSquareAttacked(bd, 60, true) && !isSquareAttacked(bd, 59, true) && !isSquareAttacked(bd, 58, true)) {
                        Move m; m.from = 60; m.to = 58; m.piece = pc; m.isCastle = true; out.push_back(m);
                    }
                }
            }
        }
    }
}

// Make/unmake moves with an "undo" struct
struct Undo {
    int captured;
    int from, to;
    int prevEP;
    int prevCastling;
    int prevHalf;
    u64 prevHash;
    bool movedPieceWasKing;
    bool movedPieceWasRookFromOriginal;
    bool capturedRookFromOriginal;
};

u64 pieceZobristXor(const Board &bd, int piece, int sq) {
    if (piece == EMPTY) return 0;
    int idx = pieceToIndex(piece);
    return Z_piece[idx][sq];
}

void makeMove(Board &bd, const Move &m, Undo &undo) {
    undo.from = m.from; undo.to = m.to;
    undo.captured = m.captured;
    undo.prevEP = bd.enPassant;
    undo.prevCastling = bd.castling;
    undo.prevHalf = bd.halfmoveClock;
    undo.prevHash = bd.zobristKey;
    undo.movedPieceWasKing = (absPiece(m.piece) == 6);
    undo.movedPieceWasRookFromOriginal = false;
    undo.capturedRookFromOriginal = false;

    // Update zobrist: remove moving piece from from
    bd.zobristKey ^= pieceZobristXor(bd, bd.b[m.from], m.from);
    // remove captured piece if any
    if (m.isEnPassant) {
        int capSq = bd.whiteToMove ? (m.to - 8) : (m.to + 8);
        int capP = bd.b[capSq];
        if (capP != EMPTY) {
            bd.zobristKey ^= pieceZobristXor(bd, capP, capSq);
            bd.b[capSq] = EMPTY;
            // if captured rook from original square, update flags
            if (capSq == 0) undo.capturedRookFromOriginal = true; // a1
            if (capSq == 7) undo.capturedRookFromOriginal = true; // h1
            if (capSq == 56) undo.capturedRookFromOriginal = true; // a8
            if (capSq == 63) undo.capturedRookFromOriginal = true; // h8
        }
    } else {
        if (bd.b[m.to] != EMPTY) {
            bd.zobristKey ^= pieceZobristXor(bd, bd.b[m.to], m.to);
            // track if captured rook on original squares
            if (m.to == 0 || m.to == 7 || m.to == 56 || m.to == 63) undo.capturedRookFromOriginal = true;
        }
    }

    // Move piece
    int moved = bd.b[m.from];
    bd.b[m.from] = EMPTY;
    // Castling
    if (m.isCastle) {
        if (m.from == 4 && m.to == 6) { // white O-O
            bd.b[6] = WK; bd.b[5] = WR; bd.b[4] = EMPTY; bd.b[7] = EMPTY;
            // adjust zobrist: add moved king and rook new squares
            bd.zobristKey ^= pieceZobristXor(bd, WK, 6);
            bd.zobristKey ^= pieceZobristXor(bd, WR, 5);
            // remove rook from h1 and king from e1 done above by pieceZobrist xor for from & captured (we already xor removed piece from from)
        } else if (m.from == 4 && m.to == 2) { // white O-O-O
            bd.b[2] = WK; bd.b[3] = WR; bd.b[4] = EMPTY; bd.b[0] = EMPTY;
            bd.zobristKey ^= pieceZobristXor(bd, WK, 2);
            bd.zobristKey ^= pieceZobristXor(bd, WR, 3);
        } else if (m.from == 60 && m.to == 62) { // black O-O
            bd.b[62] = BK; bd.b[61] = BR; bd.b[60] = EMPTY; bd.b[63] = EMPTY;
            bd.zobristKey ^= pieceZobristXor(bd, BK, 62);
            bd.zobristKey ^= pieceZobristXor(bd, BR, 61);
        } else if (m.from == 60 && m.to == 58) { // black O-O-O
            bd.b[58] = BK; bd.b[59] = BR; bd.b[60] = EMPTY; bd.b[56] = EMPTY;
            bd.zobristKey ^= pieceZobristXor(bd, BK, 58);
            bd.zobristKey ^= pieceZobristXor(bd, BR, 59);
        } else {
            // fallback, should not happen
            bd.b[m.to] = moved;
            bd.zobristKey ^= pieceZobristXor(bd, moved, m.to);
        }
        // moving rook from original square modifies castling rights below
    } else {
        // Promotion?
        if (m.promotion) {
            int promoSigned = (m.piece > 0) ? m.promotion : -m.promotion;
            bd.b[m.to] = promoSigned;
            bd.zobristKey ^= pieceZobristXor(bd, promoSigned, m.to);
        } else {
            bd.b[m.to] = moved;
            bd.zobristKey ^= pieceZobristXor(bd, moved, m.to);
        }
    }

    // If moved piece was rook and from original squares, mark for castling rights removal
    if (absPiece(m.piece) == 4) {
        if (m.from == 0) undo.movedPieceWasRookFromOriginal = true;
        if (m.from == 7) undo.movedPieceWasRookFromOriginal = true;
        if (m.from == 56) undo.movedPieceWasRookFromOriginal = true;
        if (m.from == 63) undo.movedPieceWasRookFromOriginal = true;
    }
    // Update castling rights
    int prevCast = bd.castling;
    if (absPiece(m.piece) == 6) {
        // king moved: remove both castling for that color
        if (m.piece > 0) bd.castling &= ~(WKC | WQC);
        else bd.castling &= ~(BKC | BQC);
    }
    if (absPiece(m.piece) == 4) {
        // rook moved from original square
        if (m.from == 0) bd.castling &= ~WQC;
        if (m.from == 7) bd.castling &= ~WKC;
        if (m.from == 56) bd.castling &= ~BQC;
        if (m.from == 63) bd.castling &= ~BKC;
    }
    // If a rook was captured on its original square
    if (undo.capturedRookFromOriginal) {
        if (m.to == 0) bd.castling &= ~WQC;
        if (m.to == 7) bd.castling &= ~WKC;
        if (m.to == 56) bd.castling &= ~BQC;
        if (m.to == 63) bd.castling &= ~BKC;
    }
    // Update zobrist for castling change
    bd.zobristKey ^= Z_castle[prevCast];
    bd.zobristKey ^= Z_castle[bd.castling];

    // Update en-passant
    if (bd.enPassant != -1) bd.zobristKey ^= Z_ep[bd.enPassant];
    bd.enPassant = -1;
    // If pawn moved two squares, set en passant
    if (absPiece(m.piece) == 1 && abs(m.to - m.from) == 16) {
        int ep = (m.from + m.to) / 2;
        bd.enPassant = ep;
        bd.zobristKey ^= Z_ep[bd.enPassant];
    }

    // Update halfmove clock
    if (absPiece(m.piece) == 1 || m.captured) bd.halfmoveClock = 0;
    else bd.halfmoveClock++;

    // Side to move
    bd.whiteToMove = !bd.whiteToMove;
    bd.zobristKey ^= Z_side;
}

void unmakeMove(Board &bd, const Move &m, const Undo &undo) {
    // Restore side to move
    bd.zobristKey = undo.prevHash;
    bd.whiteToMove = !bd.whiteToMove;
    bd.enPassant = undo.prevEP;
    bd.castling = undo.prevCastling;
    bd.halfmoveClock = undo.prevHalf;

    // restore pieces
    // Clear destination
    // For castling need to move rook back properly
    if (m.isCastle) {
        if (m.from == 4 && m.to == 6) { // white O-O
            bd.b[4] = WK; bd.b[7] = WR; bd.b[5] = EMPTY; bd.b[6] = EMPTY;
        } else if (m.from == 4 && m.to == 2) { // white O-O-O
            bd.b[4] = WK; bd.b[0] = WR; bd.b[3] = EMPTY; bd.b[2] = EMPTY;
        } else if (m.from == 60 && m.to == 62) { // black O-O
            bd.b[60] = BK; bd.b[63] = BR; bd.b[61] = EMPTY; bd.b[62] = EMPTY;
        } else if (m.from == 60 && m.to == 58) { // black O-O-O
            bd.b[60] = BK; bd.b[56] = BR; bd.b[59] = EMPTY; bd.b[58] = EMPTY;
        } else {
            // fallback
            bd.b[m.from] = m.piece;
            bd.b[m.to] = undo.captured;
        }
    } else {
        // Normal move
        bd.b[m.from] = m.piece;
        if (m.promotion) {
            // pawn was promoted to promotion piece, so restore pawn
            bd.b[m.from] = (m.piece > 0) ? WP : BP;
        }
        // Clear destination then restore captured (if any)
        bd.b[m.to] = undo.captured;
        if (m.isEnPassant) {
            int capSq = bd.whiteToMove ? (m.to - 8) : (m.to + 8);
            // put captured pawn back
            bd.b[capSq] = undo.captured;
            bd.b[m.to] = EMPTY;
        }
    }
}

// Generate all legal moves (filtering out those leaving king in check)
void generateLegalMoves(Board &bd, vector<Move> &out) {
    out.clear();
    vector<Move> pseudo;
    generatePseudoLegal(bd, pseudo);
    out.reserve(pseudo.size());
    for (const Move &m : pseudo) {
        Undo u;
        makeMove(const_cast<Board&>(bd), m, u); // temporarily apply
        bool inCheck = false;
        int ksq = bd.kingSquare(!bd.whiteToMove); // king of side who just moved? Wait: bd.whiteToMove toggled in makeMove
        // After makeMove, bd.whiteToMove is toggled, so we need to check the side that just moved's king? We need to ensure king of side to move is not in check.
        // We want to check that the player who moved did not leave their king in check. After makeMove, side to move is opponent.
        // So king of side that moved is opposite of bd.whiteToMove
        ksq = bd.kingSquare(!bd.whiteToMove);
        if (ksq == -1) inCheck = true;
        else {
            if (isSquareAttacked(bd, ksq, bd.whiteToMove)) inCheck = true; // attacked by opponent (current side to move)
        }
        unmakeMove(const_cast<Board&>(bd), m, u);
        if (!inCheck) out.push_back(m);
    }
}

// Simple evaluation: material + small piece-square-ish values
int pieceValue(int absP) {
    switch (absP) {
        case 1: return 100;
        case 2: return 320;
        case 3: return 330;
        case 4: return 500;
        case 5: return 900;
        case 6: return 20000;
    }
    return 0;
}

int evaluate(const Board &bd) {
    int score = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int p = bd.b[sq];
        if (p == EMPTY) continue;
        int val = pieceValue(absPiece(p));
        score += (p > 0) ? val : -val;
    }
    // Side to move advantage small
    if (bd.whiteToMove) score += 5; else score -= 5;
    return score;
}

// Transposition table
unordered_map<u64, TTEntry> tt;

// Search globals
u64 nodes = 0;
Clock::time_point endTime;
bool timeUp = false;
int MAX_DEPTH = 64;
const int INF = 30000;

// Quiescence
int quiesce(Board &bd, int alpha, int beta) {
    if ((nodes & 2047ULL) == 0) {
        if (Clock::now() >= endTime) throw 0;
    }
    nodes++;
    int stand = evaluate(bd);
    if (stand >= beta) return beta;
    if (alpha < stand) alpha = stand;

    // generate captures only
    vector<Move> moves;
    generatePseudoLegal(bd, moves);
    // filter to captures & promotions only
    vector<Move> caps;
    for (auto &m : moves) {
        if (m.captured != 0 || m.promotion != 0 || m.isEnPassant) caps.push_back(m);
    }
    // order captures by MVV-LVA naive (capture piece value - attacker value)
    sort(caps.begin(), caps.end(), [](const Move &a, const Move &b) {
        int va = a.captured ? pieceValue(absPiece(a.captured)) : 0;
        int vb = b.captured ? pieceValue(absPiece(b.captured)) : 0;
        return va > vb;
    });

    for (auto &m : caps) {
        Undo u; makeMove(bd, m, u);
        int score = -quiesce(bd, -beta, -alpha);
        unmakeMove(bd, m, u);
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

// Negamax with alpha-beta, iterative deepening will call with increasing depth
int negamax(Board &bd, int depth, int alpha, int beta) {
    if ((nodes & 2047ULL) == 0) {
        if (Clock::now() >= endTime) throw 0;
    }
    nodes++;
    if (depth <= 0) {
        return quiesce(bd, alpha, beta);
    }

    // Transposition table lookup
    auto it = tt.find(bd.zobristKey);
    if (it != tt.end()) {
        TTEntry &e = it->second;
        if (e.depth >= depth) {
            if (e.flag == 0) return e.score;
            if (e.flag == 1) alpha = max(alpha, e.score);
            else if (e.flag == 2) beta = min(beta, e.score);
            if (alpha >= beta) return e.score;
        }
    }

    vector<Move> moves;
    generateLegalMoves(bd, moves);
    if (moves.empty()) {
        // checkmate or stalemate
        int ksq = bd.kingSquare(bd.whiteToMove);
        bool inCheck = (ksq == -1) ? true : isSquareAttacked(bd, ksq, !bd.whiteToMove);
        if (inCheck) return -INF + depth; // checkmate (bad for side to move)
        return 0; // stalemate
    }

    // Order moves: try TT best move first if present
    if (it != tt.end()) {
        Move ttmove = it->second.best;
        for (int i = 0; i < (int)moves.size(); ++i) {
            if (moves[i].from == ttmove.from && moves[i].to == ttmove.to &&
                moves[i].promotion == ttmove.promotion) {
                swap(moves[0], moves[i]);
                break;
            }
        }
    }
    // simple ordering: captures first
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b) {
        int va = a.captured ? pieceValue(absPiece(a.captured)) : 0;
        int vb = b.captured ? pieceValue(absPiece(b.captured)) : 0;
        if (va != vb) return va > vb;
        return a.promotion > b.promotion; // prefer promotions
    });

    int bestScore = -INF;
    Move bestMove;
    for (const Move &m : moves) {
        Undo u; makeMove(bd, m, u);
        int score = -negamax(bd, depth - 1, -beta, -alpha);
        unmakeMove(bd, m, u);
        if (score > bestScore) { bestScore = score; bestMove = m; }
        alpha = max(alpha, score);
        if (alpha >= beta) {
            // store in TT as lowerbound
            TTEntry e; e.key = bd.zobristKey; e.depth = depth; e.score = bestScore; e.flag = 1; e.best = bestMove;
            tt[bd.zobristKey] = e;
            return bestScore;
        }
    }
    // store exact score
    TTEntry e; e.key = bd.zobristKey; e.depth = depth; e.score = bestScore; e.flag = 0; e.best = bestMove;
    tt[bd.zobristKey] = e;
    return bestScore;
}

// Iterative deepening driver: returns best move found (or a fallback)
Move iterativeDeepening(Board &bd, int maxSeconds) {
    Move bestSoFar;
    vector<Move> legal;
    generateLegalMoves(bd, legal);
    if (legal.empty()) return bestSoFar; // none

    // Choose a quick initial best (capture highest or e2e4 style)
    bestSoFar = legal[0];

    nodes = 0;
    tt.clear();
    int depth = 1;
    try {
        for (; depth <= 64; ++depth) {
            int alpha = -INF, beta = INF;
            // try search
            int score = negamax(bd, depth, alpha, beta);
            // If TT has entry for root store the best move
            auto it = tt.find(bd.zobristKey);
            if (it != tt.end()) {
                bestSoFar = it->second.best;
            } else {
                // fallback: pick highest scored move by making each
                int bestScore = -INF;
                for (auto &m : legal) {
                    Undo u; makeMove(bd, m, u);
                    int sc = -quiesce(bd, -INF, INF);
                    unmakeMove(bd, m, u);
                    if (sc > bestScore) { bestScore = sc; bestSoFar = m; }
                }
            }
            // continue deeper until time
        }
    } catch (...) {
        // time up or other; return bestSoFar
    }
    return bestSoFar;
}

// Convert move to required output string
string moveToOutputString(const Board &orig, const Move &mv) {
    if (mv.from == mv.to && mv.piece == 0) return "0000";
    // Castling
    if (mv.isCastle) {
        // After castling, we should detect check/checkmate by simulating
        Board tmp = orig;
        Undo u; makeMove(tmp, mv, u);
        // detect check/mate
        bool oppHasMoves;
        vector<Move> opp;
        generateLegalMoves(tmp, opp);
        oppHasMoves = !opp.empty();
        int ksq = tmp.kingSquare(tmp.whiteToMove);
        bool inCheck = (ksq == -1) ? true : isSquareAttacked(tmp, ksq, !tmp.whiteToMove);
        string s = (mv.to > mv.from) ? "O-O" : "O-O-O";
        if (!oppHasMoves && inCheck) s.push_back('#');
        else if (inCheck) s.push_back('+');
        return s;
    }
    string out;
    char pieceChar = pieceLetterForOutput(mv.piece);
    out.push_back(pieceChar);
    out += sqToStr(mv.from);
    bool isCapture = (mv.captured != 0) || mv.isEnPassant;
    if (isCapture) out.push_back('x');
    out += sqToStr(mv.to);
    if (mv.promotion) {
        out.push_back('/');
        char pch = 'Q';
        if (mv.promotion == 5) pch = 'Q';
        else if (mv.promotion == 4) pch = 'R';
        else if (mv.promotion == 3) pch = 'B';
        else if (mv.promotion == 2) pch = 'N';
        out.push_back(pch);
    }
    if (mv.isEnPassant) out += "ep";

    // check / mate detection
    Board tmp = orig;
    Undo u; makeMove(tmp, mv, u);
    vector<Move> opp;
    generateLegalMoves(tmp, opp);
    int ksq = tmp.kingSquare(tmp.whiteToMove);
    bool inCheck = (ksq == -1) ? true : isSquareAttacked(tmp, ksq, !tmp.whiteToMove);
    if (!opp.empty()) {
        if (inCheck) out.push_back('+');
    } else {
        if (inCheck) out.push_back('#');
        // else stalemate -> no suffix
    }
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    init_zobrist();

    string line1, fen;
    if (!getline(cin, line1)) return 0;
    if (!getline(cin, fen)) return 0;

    double margin = 0.05;
    endTime = Clock::now() + chrono::duration<double>(max(0.01, timeoutSeconds - margin));

    Board bd;
    if (!bd.loadFEN(fen)) {
        cout << "0000\n";

    // If only one legal move, output it immediately
    vector<Move> legal;

    if (legal.size() == 1) {

        return 0;

    Move best;
    try {
        best = iterativeDeepening(bd, (int)timeoutSeconds);

        // fallback best from legal[0]
        best = legal[0];

    string out = moveToOutputString(bd, best);
    cout << out << "\n";
    return 0;
}
```

Notes / limitations:
- This engine aims for correctness and adherence to the required move format. It's a modest-strength engine: it uses alpha-beta with quiescence and simple evaluation (material). It should return a legal move quickly and improve with available time.
- The engine uses iterative deepening and will return the best move found before the timeout. It tries to stop slightly before the timeout (small margin).
- It includes castling, en-passant, and promotions handling; outputs check/mate symbols when detected.
- If a position has no legal moves, it returns "0000" (no move).
