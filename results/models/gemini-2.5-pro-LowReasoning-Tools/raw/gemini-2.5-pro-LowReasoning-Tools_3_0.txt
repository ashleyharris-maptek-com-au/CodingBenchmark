use std::io::{self, BufRead};
use rand::Rng;

const MAX_NODES_FOR_FR: usize = 1000;
const FR_ITERATIONS: usize = 50;
const FAST_LAYOUT_ITERATIONS: usize = 10;

struct Node {
    x: f64,
    y: f64,
    dx: f64,
    dy: f64,
}

struct Edge {
    u: usize,
    v: usize,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        edges.push(Edge { u, v });
    }

    let mut nodes = Vec::with_capacity(n);
    let mut rng = rand::thread_rng();
    for _ in 0..n {
        nodes.push(Node {
            x: rng.gen_range(-1.0..1.0),
            y: rng.gen_range(-1.0..1.0),
            dx: 0.0,
            dy: 0.0,
        });
    }

    if n <= MAX_NODES_FOR_FR {
        layout_fruchterman_reingold(&mut nodes, &edges, FR_ITERATIONS);
    } else {
        layout_fast(&mut nodes, &edges, FAST_LAYOUT_ITERATIONS);
    }

    println!("{}", n);
    for node in nodes {
        println!("{} {}", node.x, node.y);
    }
}

fn layout_fruchterman_reingold(nodes: &mut [Node], edges: &[Edge], iterations: usize) {
    let n = nodes.len();
    if n == 0 {
        return;
    }
    let area = (n as f64).powf(2.0);
    let k = (area / n as f64).sqrt();
    let mut temp = 1.0;

    for _ in 0..iterations {
        // Repulsive forces
        for i in 0..n {
            nodes[i].dx = 0.0;
            nodes[i].dy = 0.0;
            for j in 0..n {
                if i == j {
                    continue;
                }
                let dx = nodes[i].x - nodes[j].x;
                let dy = nodes[i].y - nodes[j].y;
                let dist_sq = dx * dx + dy * dy;
                if dist_sq > 0.0 {
                    let dist = dist_sq.sqrt();
                    let repulsive_force = k * k / dist;
                    nodes[i].dx += dx / dist * repulsive_force;
                    nodes[i].dy += dy / dist * repulsive_force;
                }
            }
        }

        // Attractive forces
        for edge in edges {
            let dx = nodes[edge.v].x - nodes[edge.u].x;
            let dy = nodes[edge.v].y - nodes[edge.u].y;
            let dist_sq = dx * dx + dy * dy;
            if dist_sq > 0.0 {
                let dist = dist_sq.sqrt();
                let attractive_force = dist * dist / k;
                let force_x = dx / dist * attractive_force;
                let force_y = dy / dist * attractive_force;
                nodes[edge.u].dx += force_x;
                nodes[edge.u].dy += force_y;
                nodes[edge.v].dx -= force_x;
                nodes[edge.v].dy -= force_y;
            }
        }

        // Update positions
        for i in 0..n {
            let disp_sq = nodes[i].dx * nodes[i].dx + nodes[i].dy * nodes[i].dy;
            if disp_sq > 0.0 {
                let disp = disp_sq.sqrt();
                let limited_disp = disp.min(temp);
                nodes[i].x += nodes[i].dx / disp * limited_disp;
                nodes[i].y += nodes[i].dy / disp * limited_disp;
            }
        }

        // Cool temperature
        temp *= 1.0 - (1.0 / iterations as f64);
    }
}

fn layout_fast(nodes: &mut [Node], edges: &[Edge], iterations: usize) {
    let n = nodes.len();
    if n == 0 {
        return;
    }
    let k = 1.0 / (n as f64).sqrt();

    for _ in 0..iterations {
        for edge in edges {
            let dx = nodes[edge.v].x - nodes[edge.u].x;
            let dy = nodes[edge.v].y - nodes[edge.u].y;
            let dist = (dx*dx + dy*dy).sqrt();
            if dist > k {
                let force = 0.1 * (dist - k);
                let fx = force * dx / dist;
                let fy = force * dy / dist;
                nodes[edge.u].x += fx;
                nodes[edge.u].y += fy;
                nodes[edge.v].x -= fx;
                nodes[edge.v].y -= fy;
            }
        }
    }
}
