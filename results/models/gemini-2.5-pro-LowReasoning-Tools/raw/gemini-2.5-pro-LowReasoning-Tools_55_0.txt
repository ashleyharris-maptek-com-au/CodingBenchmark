#include <iostream>
#include <string>
#include <vector>
#include <chrono>
#include <cctype>
#include <random>
#include <limits>
#include <algorithm>

enum Piece { EMPTY, W_PAWN, W_KNIGHT, W_BISHOP, W_ROOK, W_QUEEN, W_KING, B_PAWN, B_KNIGHT, B_BISHOP, B_ROOK, B_QUEEN, B_KING, OFF_BOARD };

const int piece_values[] = { 0, 100, 320, 330, 500, 900, 20000, -100, -320, -330, -500, -900, -20000 };

struct Move {
    int from;
    int to;
    int promotion = EMPTY;
    int captured_piece = EMPTY;

    bool operator==(const Move& other) const {
        return from == other.from && to == other.to && promotion == other.promotion;
    }

    std::string to_string(const Board& board) const {
        std::string move_str;

        char piece_letter;
        int piece = board.board[from];
        switch(piece) 
        {
            case W_PAWN: case B_PAWN: piece_letter = 'p'; break;
            case W_KNIGHT: case B_KNIGHT: piece_letter = 'N'; break;
            case W_BISHOP: case B_BISHOP: piece_letter = 'B'; break;
            case W_ROOK: case B_ROOK: piece_letter = 'R'; break;
            case W_QUEEN: case B_QUEEN: piece_letter = 'Q'; break;
            case W_KING: case B_KING: piece_letter = 'K'; break;
        }
    
        std::string from_square;
        from_square += (from % 12 - 3) + 'a';
        from_square += (from / 12 - 2) + '1';

        std::string to_square;
        to_square += (to % 12 - 3) + 'a';
        to_square += (to / 12 - 2) + '1';

        move_str = piece_letter + from_square;

        if (captured_piece != EMPTY) {
            move_str += 'x';
        }

        move_str += to_square;

        if (promotion != EMPTY) {
            move_str += '/';
            switch (promotion) {
                case W_QUEEN: case B_QUEEN: move_str += 'Q'; break;
                case W_ROOK: case B_ROOK: move_str += 'R'; break;
                case W_BISHOP: case B_BISHOP: move_str += 'B'; break;
                case W_KNIGHT: case B_KNIGHT: move_str += 'N'; break;
                default: break;
            }
        }
        return move_str;
    }
};

class Board {
public:
    int board[144];
    bool white_to_move;
    bool white_king_castle;
    bool white_queen_castle;
    bool black_king_castle;
    bool black_queen_castle;
    int en_passant_square;
    int halfmove_clock;
    int fullmove_number;

    Board() {
        for (int i = 0; i < 144; ++i) {
            board[i] = OFF_BOARD;
        }
        for (int rank = 0; rank < 8; ++rank) {
            for (int file = 0; file < 8; ++file) {
                board[24 + rank * 12 + file] = EMPTY;
            }
        }
    }

    void parse_fen(const std::string& fen_string) {
        int fen_index = 0;
        int rank = 7;
        int file = 0;

        // Piece placement
        while (fen_string[fen_index] != ' ') {
            if (fen_string[fen_index] == '/') {
                rank--;
                file = 0;
            } else if (isdigit(fen_string[fen_index])) {
                file += fen_string[fen_index] - '0';
            } else {
                int piece = EMPTY;
                switch (fen_string[fen_index]) {
                    case 'P': piece = W_PAWN; break;
                    case 'N': piece = W_KNIGHT; break;
                    case 'B': piece = W_BISHOP; break;
                    case 'R': piece = W_ROOK; break;
                    case 'Q': piece = W_QUEEN; break;
                    case 'K': piece = W_KING; break;
                    case 'p': piece = B_PAWN; break;
                    case 'n': piece = B_KNIGHT; break;
                    case 'b': piece = B_BISHOP; break;
                    case 'r': piece = B_ROOK; break;
                    case 'q': piece = B_QUEEN; break;
                    case 'k': piece = B_KING; break;
                }
                board[24 + rank * 12 + file] = piece;
                file++;
            }
            fen_index++;
        }

        fen_index++; // Skip the space

        // Active color
        white_to_move = (fen_string[fen_index] == 'w');
        fen_index++;
        if(fen_index < fen_string.length() && fen_string[fen_index] == ' ')
        {
            fen_index++;
        }

        // Castling availability
        white_king_castle = false;
        white_queen_castle = false;
        black_king_castle = false;
        black_queen_castle = false;
        while (fen_index < fen_string.length() && fen_string[fen_index] != ' ') {
            if (fen_string[fen_index] == 'K') white_king_castle = true;
            if (fen_string[fen_index] == 'Q') white_queen_castle = true;
            if (fen_string[fen_index] == 'k') black_king_castle = true;
            if (fen_string[fen_index] == 'q') black_queen_castle = true;
            fen_index++;
        }

       if(fen_index < fen_string.length() && fen_string[fen_index] == ' ')
        {
            fen_index++;
        }

        // En passant
        if (fen_index < fen_string.length() && fen_string[fen_index] != '-') {
            int ep_file = fen_string[fen_index] - 'a';
            int ep_rank = fen_string[fen_index+1] - '1';
            en_passant_square = 24 + ep_rank * 12 + ep_file;
        } else {
            en_passant_square = -1;
        }

        while (fen_index < fen_string.length() && fen_string[fen_index] != ' ') {
            fen_index++;
        }
        fen_index++;
        
        if(fen_index < fen_string.length())
        {
            std::string halfmove_str;
            while (fen_index < fen_string.length() && fen_string[fen_index] != ' ') {
                halfmove_str += fen_string[fen_index];
                fen_index++;
            }
            halfmove_clock = std::stoi(halfmove_str);
        }

        fen_index++;

        if(fen_index < fen_string.length())
        {
            std::string fullmove_str;
            while (fen_index < fen_string.length() && fen_string[fen_index] != ' ') {
                fullmove_str += fen_string[fen_index];
                fen_index++;
            }
            fullmove_number = std::stoi(fullmove_str);
        }
    }

    std::vector<Move> generate_moves();
    void generate_pawn_moves(int from, std::vector<Move>& moves);
    void generate_knight_moves(int from, std::vector<Move>& moves);
    void generate_sliding_moves(int from, std::vector<Move>& moves, const std::vector<int>& directions);
    void generate_king_moves(int from, std::vector<Move>& moves);

    bool make_move(Move& move);
    void unmake_move(const Move& move);

    int evaluate() {
        int score = 0;
        for (int i = 24; i < 120; ++i) {
            if (board[i] != OFF_BOARD) {
                score += piece_values[board[i]];
            }
        }
        return score;
    }
};

void Board::generate_pawn_moves(int from, std::vector<Move>& moves) {
    int direction = white_to_move ? 12 : -12;
    int start_rank = white_to_move ? 3 : 8;
    int promotion_rank = white_to_move ? 8 : 3;

    // Single push
    if (board[from + direction] == EMPTY) {
        if ((from / 12) == promotion_rank) {
            moves.push_back({from, from + direction, white_to_move ? W_QUEEN : B_QUEEN});
            moves.push_back({from, from + direction, white_to_move ? W_ROOK : B_ROOK});
            moves.push_back({from, from + direction, white_to_move ? W_BISHOP : B_BISHOP});
            moves.push_back({from, from + direction, white_to_move ? W_KNIGHT : B_KNIGHT});
        } else {
            moves.push_back({from, from + direction});
        }
    }

    // Double push
    if ((from / 12) == start_rank && board[from + direction] == EMPTY && board[from + 2 * direction] == EMPTY) {
        moves.push_back({from, from + 2 * direction});
    }

    // Captures
    int capture_offsets[] = {11, 13};
    for (int offset : capture_offsets) {
        int to = from + direction - 1 + (offset == 13 ? 2 : 0);
        if (board[to] != OFF_BOARD &&
            ((white_to_move && (board[to] >= B_PAWN && board[to] <= B_KING)) ||
            (!white_to_move && (board[to] >= W_PAWN && board[to] <= W_KING)))) {
             if ((from / 12) == promotion_rank) {
                moves.push_back({from, to, white_to_move ? W_QUEEN : B_QUEEN});
                moves.push_back({from, to, white_to_move ? W_ROOK : B_ROOK});
                moves.push_back({from, to, white_to_move ? W_BISHOP : B_BISHOP});
                moves.push_back({from, to, white_to_move ? W_KNIGHT : B_KNIGHT});
            } else {
                moves.push_back({from, to});
            }
        }
    }
}

void Board::generate_knight_moves(int from, std::vector<Move>& moves) {
    int knight_offsets[] = {-25, -23, -14, -10, 10, 14, 23, 25};
    for (int offset : knight_offsets) {
        int to = from + offset;
        if (board[to] != OFF_BOARD) {
             if (board[to] == EMPTY ||
                (white_to_move && (board[to] >= B_PAWN && board[to] <= B_KING)) ||
                (!white_to_move && (board[to] >= W_PAWN && board[to] <= W_KING))) {
                moves.push_back({from, to});
            }
        }
    }
}

void Board::generate_sliding_moves(int from, std::vector<Move>& moves, const std::vector<int>& directions) {
    for (int direction : directions) {
        int to = from + direction;
        while (board[to] != OFF_BOARD) {
            if (board[to] == EMPTY) {
                moves.push_back({from, to});
            } else {
                if ((white_to_move && (board[to] >= B_PAWN && board[to] <= B_KING)) ||
                    (!white_to_move && (board[to] >= W_PAWN && board[to] <= W_KING))) {
                    moves.push_back({from, to});
                }
                break;
            }
            to += direction;
        }
    }
}

void Board::generate_king_moves(int from, std::vector<Move>& moves) {
    int king_offsets[] = {-13, -12, -11, -1, 1, 11, 12, 13};
    for (int offset : king_offsets) {
        int to = from + offset;
        if (board[to] != OFF_BOARD) {
             if (board[to] == EMPTY ||
                (white_to_move && (board[to] >= B_PAWN && board[to] <= B_KING)) ||
                (!white_to_move && (board[to] >= W_PAWN && board[to] <= W_KING))) {
                moves.push_back({from, to});
            }
        }
    }
}

std::vector<Move> Board::generate_moves() {
    std::vector<Move> moves;
    for (int i = 24; i < 120; ++i) {
        if (board[i] == OFF_BOARD) continue;
        int piece = board[i];
        if ((white_to_move && (piece >= W_PAWN && piece <= W_KING)) ||
            (!white_to_move && (piece >= B_PAWN && piece <= B_KING))) {
            switch (piece) {
                case W_PAWN: case B_PAWN: generate_pawn_moves(i, moves); break;
                case W_KNIGHT: case B_KNIGHT: generate_knight_moves(i, moves); break;
                case W_BISHOP: case B_BISHOP: generate_sliding_moves(i, moves, { -13, -11, 11, 13 }); break;
                case W_ROOK: case B_ROOK: generate_sliding_moves(i, moves, { -12, -1, 1, 12 }); break;
                case W_QUEEN: case B_QUEEN: generate_sliding_moves(i, moves, { -13, -11, 11, 13, -12, -1, 1, 12 }); break;
                case W_KING: case B_KING: generate_king_moves(i, moves); break;
            }
        }
    }
    return moves;
}

bool Board::make_move(Move& move) {
    move.captured_piece = board[move.to];
    board[move.to] = board[move.from];
    board[move.from] = EMPTY;
    if (move.promotion != EMPTY) {
        board[move.to] = move.promotion;
    }
    white_to_move = !white_to_move;
    return true;
}

void Board::unmake_move(const Move& move) {
    white_to_move = !white_to_move;
    board[move.from] = board[move.to];
    board[move.to] = move.captured_piece;
    if (move.promotion != EMPTY) {
        board[move.from] = white_to_move ? W_PAWN : B_PAWN;
    }
}

int negamax(Board& board, int depth, int alpha, int beta, std::chrono::time_point<std::chrono::steady_clock> start_time, int timeout_in_seconds) {
    if (std::chrono::steady_clock::now() - start_time > std::chrono::seconds(timeout_in_seconds)) {
        return board.evaluate();
    }
    if (depth == 0) {
        return board.evaluate();
    }
    int max = -std::numeric_limits<int>::max();
    auto moves = board.generate_moves();
    if (moves.empty()) {
        return board.evaluate();
    }
    for (auto& move : moves) {
        board.make_move(move);
        int score = -negamax(board, depth - 1, -beta, -alpha, start_time, timeout_in_seconds);
        board.unmake_move(move);
        if (score > max) {
            max = score;
        }
        if (score > alpha) {
            alpha = score;
        }
        if (alpha >= beta) {
            break;
        }
    }
    return max;
}

Move search(Board& board, int timeout_in_seconds) {
    auto start_time = std::chrono::steady_clock::now();
    Move best_move = {-1, -1};
    int best_score = -std::numeric_limits<int>::max();
    
    for (int depth = 1; depth < 100; ++depth) {
        auto moves = board.generate_moves();
        if (moves.empty()) {
            return {-1,-1};
        }
        Move current_best_move_for_depth = moves[0];
        int alpha = -std::numeric_limits<int>::max();
        int beta = std::numeric_limits<int>::max();

        for (auto& move : moves) {
            board.make_move(move);
            int score = -negamax(board, depth - 1, -beta, -alpha, start_time, timeout_in_seconds);
            board.unmake_move(move);
            if (std::chrono::steady_clock::now() - start_time > std::chrono::seconds(timeout_in_seconds)) {
                return best_move;
            }
            if (score > best_score) {
                best_score = score;
                current_best_move_for_depth = move;
            }
            if (score > alpha) {
                alpha = score;
            }
        }
        best_move = current_best_move_for_depth;
    }
    return best_move;
}

int main() {
    // Read input
    int timeout_in_seconds;
    std::string fen_string;

    std::cin >> timeout_in_seconds;
    std::getline(std::cin >> std::ws, fen_string);

    Board board;
    board.parse_fen(fen_string);
    
    Move best_move = search(board, timeout_in_seconds);
    
    if(best_move.from != -1)
    {
        best_move.captured_piece = board.board[best_move.to];
        std::cout << best_move.to_string(board) << std::endl;
    } else 
    {
        // No move found
    }

    return 0;
}
