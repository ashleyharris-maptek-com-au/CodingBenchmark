using System;
using System.Collections.Generic;
using System.Linq;

public class MinesweeperSolver
{
    public static void Main(string[] args)
    {
        var line = Console.ReadLine().Split(' ');
        var height = int.Parse(line[0]);
        var width = int.Parse(line[1]);

        var board = new char[height][];
        for (int i = 0; i < height; i++)
        {
            board[i] = Console.ReadLine().ToCharArray();
        }

        var safeCells = Solve(board, height, width);

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.Item2} {cell.Item1}");
        }
    }

    public static List<Tuple<int, int>> Solve(char[][] board, int height, int width)
    {
        var safeCells = new List<Tuple<int, int>>();
        bool changed;

        do
        {
            changed = false;
            var newlyFoundSafeCells = new List<Tuple<int, int>>();

            for (int r = 0; r < height; r++)
            {
                for (int c = 0; c < width; c++)
                {
                    if (char.IsDigit(board[r][c]))
                    {
                        int num = board[r][c] - '0';
                        var neighbors = GetNeighbors(r, c, height, width);
                        int flaggedNeighbors = 0;
                        var unknownNeighbors = new List<Tuple<int, int>>();

                        foreach (var n in neighbors)
                        {
                            if (board[n.Item1][n.Item2] == '*')
                            {
                                flaggedNeighbors++;
                            }
                            else if (board[n.Item1][n.Item2] == ' ')
                            {
                                unknownNeighbors.Add(n);
                            }
                        }

                        if (unknownNeighbors.Any())
                        {
                            if (num == flaggedNeighbors)
                            {
                                foreach (var unknown in unknownNeighbors)
                                {
                                    if (!safeCells.Contains(unknown) && !newlyFoundSafeCells.Contains(unknown))
                                    {
                                        newlyFoundSafeCells.Add(unknown);
                                    }
                                }
                            }
                            else if (num == flaggedNeighbors + unknownNeighbors.Count)
                            {
                                foreach (var unknown in unknownNeighbors)
                                {
                                    board[unknown.Item1][unknown.Item2] = '*';
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
             if (newlyFoundSafeCells.Any())
            {
                foreach(var cell in newlyFoundSafeCells)
                {
                    safeCells.Add(cell);
                    board[cell.Item1][cell.Item2] = 'S'; // Mark as found safe to avoid re-adding
                }
                changed = true;
            }

        } while (changed);

        // Advanced Logic for complex cases
        if (safeCells.Count == 0) {
            var allUnknowns = new List<Tuple<int,int>>();
            for(int r = 0; r < height; r++) for(int c = 0; c < width; c++) if(board[r][c] == ' ') allUnknowns.Add(Tuple.Create(r,c));

            var constraintProvinces = FindConstraintProvinces(board, height, width);
            foreach(var province in constraintProvinces) {
                if (province.Count < 2) continue; // Not a real constraint set

                var border = new List<Tuple<int,int>>();
                var unknowns = new List<Tuple<int,int>>();
                var constraints = new Dictionary<Tuple<int, int>, int>();

                foreach(var cell in province) {
                    if(char.IsDigit(board[cell.Item1][cell.Item2])) {
                       var neighbors = GetNeighbors(cell.Item1, cell.Item2, height, width);
                       int mineCount = board[cell.Item1][cell.Item2] - '0';
                       int flaggedCount = 0;
                       foreach(var n in neighbors) if(board[n.Item1][n.Item2] == '*') flaggedCount++;
                       
                       constraints[cell] = mineCount - flaggedCount;
                       border.Add(cell);
                    } else if (board[cell.Item1][cell.Item2] == ' ') {
                       unknowns.Add(cell);
                    }
                }

                if(unknowns.Count > 0 && border.Count > 0)
                {
                    // This is where a more advanced solver, like a SAT solver, would be used.
                    // For this problem, we'll use a simplified brute-force for small sets of unknowns.
                    // If unknowns > ~16 this is intractable.
                    if (unknowns.Count < 10)
                    {
                        TryBruteForce(board, height, width, unknowns, border, constraints, safeCells);
                    }
                }
            }
        }


        // Reset 'S' back to ' ' for the final output, if needed for validation
        for(int r=0; r<height; r++) {
            for(int c=0; c<width; c++) {
                if(board[r][c] == 'S') board[r][c] = ' ';
            }
        }

        return safeCells.Distinct().ToList(); // Ensure uniqueness
    }

    private static void TryBruteForce(char[][] board, int height, int width, List<Tuple<int, int>> unknowns, List<Tuple<int, int>> border, Dictionary<Tuple<int, int>, int> constraints, List<Tuple<int, int>> safeCells)
    {
        long limit = 1L << unknowns.Count;
        var validSolutions = new List<long>();

        for (long i = 0; i < limit; i++)
        {
            bool possible = true;
            foreach (var constraint in constraints)
            {
                int mines = 0;
                var neighbors = GetNeighbors(constraint.Key.Item1, constraint.Key.Item2, height, width);

                for (int j = 0; j < unknowns.Count; j++)
                {
                    if ((i & (1L << j)) != 0) // Is this unknown a mine in this permutation?
                    {
                        if (neighbors.Contains(unknowns[j]))
                        {
                            mines++;
                        }
                    }
                }
                if (mines != constraint.Value)
                {
                    possible = false;
                    break;
                }
            }

            if (possible)
            {
                validSolutions.Add(i);
            }
        }

        if (validSolutions.Any())
        {
            // Check for cells that are safe in ALL valid solutions
            long allSafe = -1; // all bits set
            foreach (var sol in validSolutions)
            {
                allSafe &= ~sol; // a cell is safe if it's never a mine
            }

            for (int j = 0; j < unknowns.Count; j++)
            {
                if ((allSafe & (1L << j)) != 0)
                {
                    if (!safeCells.Contains(unknowns[j]))
                    {
                        safeCells.Add(unknowns[j]);
                    }
                }
            }
        }
    }


    private static List<List<Tuple<int,int>>> FindConstraintProvinces(char[][] board, int height, int width) {
        var provinces = new List<List<Tuple<int,int>>>();
        var visited = new bool[height, width];

        for(int r=0; r < height; r++) {
            for(int c=0; c < width; c++) {
                if(!visited[r,c] && (board[r][c] == ' ' || char.IsDigit(board[r][c]))) {
                    var province = new List<Tuple<int,int>>();
                    var q = new Queue<Tuple<int,int>>();

                    q.Enqueue(Tuple.Create(r,c));
                    visited[r,c] = true;
                    
                    while(q.Count > 0) {
                        var current = q.Dequeue();
                        province.Add(current);

                        var neighbors = GetNeighbors(current.Item1, current.Item2, height, width);
                        foreach(var n in neighbors) {
                             if(!visited[n.Item1, n.Item2] && (board[n.Item1][n.Item2] == ' ' || char.IsDigit(board[n.Item1][n.Item2]))) {
                                visited[n.Item1,n.Item2] = true;
                                q.Enqueue(n);
                            }
                        }
                    }
                    provinces.Add(province);
                }
            }
        }
        return provinces;
    }


    public static List<Tuple<int, int>> GetNeighbors(int r, int c, int height, int width)
    {
        var neighbors = new List<Tuple<int, int>>();
        for (int i = -1; i <= 1; i++)
        {
            for (int j = -1; j <= 1; j++)
            {
                if (i == 0 && j == 0) continue;
                int newR = r + i;
                int newC = c + j;
                if (newR >= 0 && newR < height && newC >= 0 && newC < width)
                {
                    neighbors.Add(Tuple.Create(newR, newC));
                }
            }
        }
        return neighbors;
    }
}
