use nalgebra::{Point3, Vector3, UnitQuaternion, Point2};
use std::io::{self, Read};

// The vertices of a standard tetrahedron.
const TETRAHEDRON_VERTICES: [Point3<f64>; 4] = [
    Point3::new(1.0, 0.0, -0.707),
    Point3::new(-1.0, 0.0, -0.707),
    Point3::new(0.0, 1.0, 0.707),
    Point3::new(0.0, -1.0, 0.707),
];

// Struct to hold the transformation of a tetrahedron.
struct TetrahedronTransform {
    position: Point3<f64>,
    rotation: UnitQuaternion<f64>,
    scale: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut lines = input.lines();

    // Read the number of vertices.
    let nv: usize = lines.next().unwrap().parse().unwrap();

    // Read the polygon vertices.
    let mut polygon_vertices = Vec::with_capacity(nv);
    for _ in 0..nv {
        let mut coords = lines.next().unwrap().split_whitespace();
        let x: f64 = coords.next().unwrap().parse().unwrap();
        let y: f64 = coords.next().unwrap().parse().unwrap();
        polygon_vertices.push(Point2::new(x, y));
    }

    // Read the sun direction vector.
    let mut sun_coords = lines.next().unwrap().split_whitespace();
    let sx: f64 = sun_coords.next().unwrap().parse().unwrap();
    let sy: f64 = sun_coords.next().unwrap().parse().unwrap();
    let sz: f64 = sun_coords.next().unwrap().parse().unwrap();
    let _sun_direction = Vector3::new(sx, sy, sz).normalize();

    // Triangulate the polygon.
    let mut flat_vertices = Vec::with_capacity(nv * 2);
    for p in &polygon_vertices {
        flat_vertices.push(p.x);
        flat_vertices.push(p.y);
    }
    let hole_indices = Vec::new();
    let triangle_indices = earcutr::earcut(&flat_vertices, &hole_indices, 2);

    let mut tetrahedrons = Vec::new();

    // For each triangle, create a tetrahedron to cover its shadow.
    for triangle in triangle_indices.chunks(3) {
        let v0 = polygon_vertices[triangle[0]];
        let v1 = polygon_vertices[triangle[1]];
        let v2 = polygon_vertices[triangle[2]];
        
        // Calculate the 2D centroid of the triangle
        let centroid2d = Point2::new((v0.x + v1.x + v2.x) / 3.0, (v0.y + v1.y + v2.y) / 3.0);
        
        // The position of the tetrahedron will be high above the centroid.
        // The height will be chosen to avoid intersections, for now a constant.
        let position = Point3::new(centroid2d.x, centroid2d.y, 10.0);

        // We need to find the scale. For simplicity, let's find the max distance from centroid to a vertex.
        let d0: f64 = nalgebra::distance(&centroid2d, &v0);
        let d1: f64 = nalgebra::distance(&centroid2d, &v1);
        let d2: f64 = nalgebra::distance(&centroid2d, &v2);
        let max_dist = d0.max(d1).max(d2);

        // This is a very rough approximation for the scale.
        // A more accurate calculation would involve projecting the tetrahedron and fitting the shadow.
        let scale = max_dist * 1.5;

        // For the rotation, we'll align one of the tetrahedron's axes with the sun direction.
        // This is a simple strategy. A better one would be to orient the tetrahedron
        // to cast the most "circular" shadow.
        let rotation = UnitQuaternion::from_axis_angle(&Vector3::x_axis(), 0.0);

        tetrahedrons.push(TetrahedronTransform {
            position,
            rotation,
            scale,
        });
    }

    // Output the results.
    println!("{}", tetrahedrons.len());
    for t in tetrahedrons {
        println!(
            "{} {} {} {} {} {} {} {}",
            t.position.x, t.position.y, t.position.z,
            t.rotation.w, t.rotation.i, t.rotation.j, t.rotation.k,
            t.scale
        );
    }
}
