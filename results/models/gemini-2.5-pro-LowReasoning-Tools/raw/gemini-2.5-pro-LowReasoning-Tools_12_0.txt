using System;
using System.Collections.Generic;
using System.Numerics;
using System.Linq;

public class BinPackingSolver
{
    public struct Polyhedron
    {
        public Vector3[] Vertices;
        public int[][] Faces;
    }

    public static void Main(string[] args)
    {
        // --- INPUT PARSING ---
        string[] line = Console.ReadLine().Split();
        int vCount = int.Parse(line[0]);
        int fCount = int.Parse(line[1]);

        var vertices = new Vector3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            line = Console.ReadLine().Split();
            vertices[i] = new Vector3(
                float.Parse(line[0]),
                float.Parse(line[1]),
                float.Parse(line[2])
            );
        }

        var faces = new int[fCount][];
        for (int i = 0; i < fCount; i++)
        {
            line = Console.ReadLine().Split();
            faces[i] = new int[] {
                int.Parse(line[0]),
                int.Parse(line[1]),
                int.Parse(line[2])
            };
        }

        var polyhedron = new Polyhedron { Vertices = vertices, Faces = faces };

        line = Console.ReadLine().Split();
        var container = new Vector3(
            float.Parse(line[0]),
            float.Parse(line[1]),
            float.Parse(line[2])
        );

        // --- ADAPTIVE STRATEGY SELECTION ---
        // Based on the container volume, we can select a different strategy.
        double containerVolume = container.X * container.Y * container.Z;
        
        // Trivial/Small: Brute-force with multiple rotations.
        // Medium: Greedy search with a few pre-selected rotations.
        // Large: Grid-based placement, possibly with some randomization.
        // Extreme: Voxel-based approach or Octree for collision detection.
        // For this example, we use a simple grid-based approach for all sizes.

        // --- PACKING LOGIC ---
        var placements = new List<Tuple<Vector3, Quaternion>>();

        // Calculate the bounding box of the polyhedron
        Vector3 min = new Vector3(float.MaxValue);
        Vector3 max = new Vector3(float.MinValue);
        foreach (var v in polyhedron.Vertices)
        {
            min = Vector3.Min(min, v);
            max = Vector3.Max(max, v);
        }
        Vector3 size = max - min;

        // Simple grid-based packing
        // For simplicity, no rotations are applied (identity quaternion)
        var identityRotation = new Quaternion(0, 0, 0, 1);

        for (float x = 0; x <= container.X - size.X; x += size.X)
        {
            for (float y = 0; y <= container.Y - size.Y; y += size.Y)
            {
                for (float z = 0; z <= container.Z - size.Z; z += size.Z)
                {
                    var translation = new Vector3(x, y, z) - min;
                    placements.Add(new Tuple<Vector3, Quaternion>(translation, identityRotation));
                }
            }
        }


        // --- OUTPUT ---
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Item1.X} {p.Item1.Y} {p.Item1.Z} {p.Item2.W} {p.Item2.X} {p.Item2.Y} {p.Item2.Z}");
        }
    }
}
