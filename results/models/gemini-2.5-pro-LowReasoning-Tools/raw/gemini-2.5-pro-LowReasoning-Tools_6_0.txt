using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Threading;

public class OrbitalSolver
{
    public const double Mu = 398600.4418; // Earth gravitational parameter km^3/s^2

    public struct Vector3
    {
        public double X, Y, Z;

        public Vector3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public double Magnitude => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    public class Orbit
    {
        public Vector3 Position;
        public Vector3 Velocity;

        public Orbit(Vector3 position, Vector3 velocity)
        {
            Position = position;
            Velocity = velocity;
        }
    }

    // Simplified Delta-V calculation using velocity difference
    public static double CalculateDeltaV(Orbit start, Orbit end)
    {
        // This is a major simplification. In a real scenario, we'd use Lambert's problem
        // to find the optimal transfer trajectory and its required delta-V.
        // For a TSP heuristic, the magnitude of the velocity difference can be a
        // reasonable proxy for the "cost" of moving between orbital states.
        return (start.Velocity - end.Velocity).Magnitude;
    }

    public static void Main(string[] args)
    {
        // Set culture to invariant to ensure consistent parsing of doubles
        Thread.CurrentThread.CurrentCulture = CultureInfo.InvariantCulture;

        var firstLine = Console.ReadLine()!.Split(' ');
        int n = int.Parse(firstLine[0]);
        // Mu is provided but we use the constant for consistency
        // double mu = double.Parse(firstLine[1]);

        var startOrbitLine = Console.ReadLine()!.Substring("start orbit: ".Length).Split(' ');
        var startPosition = new Vector3(double.Parse(startOrbitLine[0]), double.Parse(startOrbitLine[1]), double.Parse(startOrbitLine[2]));
        var startVelocity = new Vector3(double.Parse(startOrbitLine[3]), double.Parse(startOrbitLine[4]), double.Parse(startOrbitLine[5]));
        var startOrbit = new Orbit(startPosition, startVelocity);

        var stations = new List<Orbit>();
        for (int i = 0; i < n; i++)
        {
            var stationLine = Console.ReadLine()!.Substring("station orbits: ".Length).Split(' ');
            var position = new Vector3(double.Parse(stationLine[0]), double.Parse(stationLine[1]), double.Parse(stationLine[2]));
            var velocity = new Vector3(double.Parse(stationLine[3]), double.Parse(stationLine[4]), double.Parse(stationLine[5]));
            stations.Add(new Orbit(position, velocity));
        }

        // The problem is to find the order of visiting stations, starting from the 'startOrbit'
        // We can model this as a TSP where the "home" city is the startOrbit.
        // Let's create a combined list of all locations to visit, with the start orbit at index 0.
        var allOrbits = new List<Orbit> { startOrbit };
        allOrbits.AddRange(stations);

        int numLocations = allOrbits.Count;
        var tour = new List<int>();
        var visited = new bool[numLocations];
        
        // Start the tour at the startOrbit (index 0)
        int currentCity = 0;
        tour.Add(currentCity);
        visited[currentCity] = true;

        for (int i = 0; i < n; i++) // We need to visit N stations
        {
            double minDistance = double.MaxValue;
            int nextCity = -1;

            for (int j = 1; j < numLocations; j++) // Station indices are 1 to N in allOrbits
            {
                if (!visited[j])
                {
                    double distance = CalculateDeltaV(allOrbits[currentCity], allOrbits[j]);
                    if (distance < minDistance)
                    {
                        minDistance = distance;
                        nextCity = j;
                    }
                }
            }
            
            if(nextCity != -1)
            {
                visited[nextCity] = true;
                tour.Add(nextCity);
                currentCity = nextCity;
            }
        }
        
        // The tour contains indices from the allOrbits list. 
        // We need to output the original station indices (0 to N-1).
        // The tour starts with 0 (startOrbit), so we skip it.
        // The indices in the tour are 1-based relative to the original station list.
        var resultOrder = tour.Skip(1).Select(stationIndex => stationIndex - 1).ToList();

        Console.WriteLine(string.Join(" ", resultOrder));
    }
}
