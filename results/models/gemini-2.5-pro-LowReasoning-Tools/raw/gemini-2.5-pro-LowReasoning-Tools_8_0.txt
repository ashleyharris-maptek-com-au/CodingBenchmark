use std::collections::{HashMap, VecDeque};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let w: usize = parts.next().unwrap().parse().unwrap();

    let mut maze = Vec::with_capacity(h * w);
    let mut start = (0, 0);
    let mut end = (0, 0);

    for r in 0..h {
        let line = lines.next().unwrap().unwrap();
        for (c, char) in line.chars().enumerate() {
            maze.push(char);
            if char == 'A' {
                start = (r, c);
            } else if char == 'B' {
                end = (r, c);
            }
        }
    }

    let mut queue = VecDeque::new();
    queue.push_back(start);

    let mut visited = vec![false; h * w];
    visited[start.0 * w + start.1] = true;

    let mut predecessors = HashMap::new();

    let mut path_found = false;

    while let Some(current) = queue.pop_front() {
        if current == end {
            path_found = true;
            break;
        }

        let (r, c) = current;

        // Neighbors: up, down, left, right
        let neighbors = [
            (r.wrapping_sub(1), c),
            (r + 1, c),
            (r, c.wrapping_sub(1)),
            (r, c + 1),
        ];

        for (nr, nc) in neighbors.iter().cloned() {
            if nr < h && nc < w {
                let index = nr * w + nc;
                if !visited[index] && maze[index] != '#' {
                    visited[index] = true;
                    predecessors.insert((nr, nc), current);
                    queue.push_back((nr, nc));
                }
            }
        }
    }

    if path_found {
        let mut path = Vec::new();
        let mut current = end;
        while current != start {
            path.push(current);
            current = predecessors[&current];
        }
        path.push(start);
        path.reverse();

        println!("{}", path.len());
        for (r, c) in path {
            println!("{} {}", c, r);
        }
    } else {
        println!("0");
    }
}
