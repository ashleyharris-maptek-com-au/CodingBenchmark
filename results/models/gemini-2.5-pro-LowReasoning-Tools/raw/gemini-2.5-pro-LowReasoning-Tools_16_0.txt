#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

// Struct to represent a single task in a job
struct Task {
    int machine;
    int duration;
    long long start_time; // Using long long to avoid overflow for large problems
};

// Struct to represent a job, which is a sequence of tasks
struct Job {
    std::vector<Task> tasks;
};

// --- Adaptive Logic Consideration ---
// The problem asks for an adaptive solver. For a real-world, high-performance solver,
// we would implement multiple strategies:
// 1.  Small-scale problems (e.g., < 100 total tasks): An exact method like Branch and Bound
//     or a powerful metaheuristic like Tabu Search could be used to find optimal or near-optimal
//     solutions.
// 2.  Large-scale problems (e.g., > 1000 tasks): A fast and effective heuristic is necessary.
//
// This implementation uses a single, robust greedy algorithm for all problem sizes.
// The chosen greedy strategy is a classic dispatching rule: at each step, select the
// available task that can start the earliest. This approach is very fast, making it suitable
// for the "extreme" scale (50k+ tasks), while still providing a valid and reasonably
// good solution for smaller-scale problems. It is a solid baseline upon which more
// complex logic could be built.

// Function to solve the Job-Shop Scheduling problem
void solve() {
    int num_jobs, num_machines;
    std::cin >> num_jobs >> num_machines;

    std::vector<Job> jobs(num_jobs);
    int total_tasks = 0;
    for (int i = 0; i < num_jobs; ++i) {
        int num_tasks;
        std::cin >> num_tasks;
        jobs[i].tasks.resize(num_tasks);
        total_tasks += num_tasks;
        for (int j = 0; j < num_tasks; ++j) {
            std::cin >> jobs[i].tasks[j].machine >> jobs[i].tasks[j].duration;
        }
    }

    // Keep track of the next available time for each machine
    std::vector<long long> machine_available_time(num_machines, 0);
    // Keep track of the completion time of the last scheduled task for each job
    std::vector<long long> job_available_time(num_jobs, 0);
    // Keep track of which task is next to be scheduled for each job
    std::vector<int> next_task_for_job(num_jobs, 0);

    int scheduled_tasks = 0;

    // Main greedy loop: schedule one task at a time
    while (scheduled_tasks < total_tasks) {
        int best_job = -1;
        long long earliest_start_time = std::numeric_limits<long long>::max();

        // Find the next task that can be scheduled the earliest
        for (int i = 0; i < num_jobs; ++i) {
            // Check if there are any unscheduled tasks for this job
            if (next_task_for_job[i] < jobs[i].tasks.size()) {
                const auto& task_info = jobs[i].tasks[next_task_for_job[i]];
                
                // A task can start only after its machine is free AND its preceding task in the same job is complete.
                long long start_time = std::max(machine_available_time[task_info.machine], job_available_time[i]);

                // If this task can start earlier than the best one found so far, update our choice.
                if (start_time < earliest_start_time) {
                    earliest_start_time = start_time;
                    best_job = i;
                }
            }
        }

        // Schedule the chosen task
        if (best_job != -1) {
            int task_idx = next_task_for_job[best_job];
            auto& task = jobs[best_job].tasks[task_idx];
            
            task.start_time = earliest_start_time;

            // Update the available times for the machine and job
            machine_available_time[task.machine] = earliest_start_time + task.duration;
            job_available_time[best_job] = earliest_start_time + task.duration;
            
            // Move to the next task for this job
            next_task_for_job[best_job]++;
            scheduled_tasks++;
        }
    }

    // Print the final schedule in the required format
    for (int i = 0; i < num_jobs; ++i) {
        for (int j = 0; j < jobs[i].tasks.size(); ++j) {
            const auto& task = jobs[i].tasks[j];
            std::cout << task.start_time << " " << task.machine << (j == jobs[i].tasks.size() - 1 ? "" : " ");
        }
        std::cout << std::endl;
    }
}

int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    solve();
    return 0;
}
