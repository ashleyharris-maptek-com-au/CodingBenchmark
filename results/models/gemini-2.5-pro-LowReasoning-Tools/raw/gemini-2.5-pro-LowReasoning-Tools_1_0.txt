#include <iostream>
#include <vector>
#include <cmath>
#include <numeric>
#include <algorithm>
#include <limits>
#include <chrono>
#include <map>

// Using double for precision
using coord_t = double;

struct Point {
    coord_t x, y;
};

// Calculate squared Euclidean distance to avoid sqrt
inline double dist_sq(const Point& p1, const Point& p2) {
    double dx = p1.x - p2.x;
    double dy = p1.y - p2.y;
    return dx * dx + dy * dy;
}

std::vector<int> solve_nearest_neighbor(const std::vector<Point>& cities) {
    int n = cities.size();
    if (n == 0) return {};

    std::vector<int> tour;
    tour.reserve(n);
    std::vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current_city = 0;

    for (int i = 1; i < n; ++i) {
        int next_city = -1;
        double min_dist = std::numeric_limits<double>::max();

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d = dist_sq(cities[current_city], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next_city = j;
                }
            }
        }
        if (next_city != -1) {
            tour.push_back(next_city);
            visited[next_city] = true;
            current_city = next_city;
        } else {
            // Fallback for disconnected components (should not happen in TSP)
            for(int j=0; j<n; ++j) {
                if(!visited[j]) {
                    tour.push_back(j);
                    visited[j] = true;
                    current_city = j;
                    break;
                }
            }
        }
    }
    return tour;
}

void two_opt(std::vector<int>& tour, const std::vector<Point>& cities, double time_limit_seconds) {
    int n = tour.size();
    if (n < 4) return;

    auto start_time = std::chrono::high_resolution_clock::now();
    bool improved = true;
    while (improved) {
        auto current_time = std::chrono::high_resolution_clock::now();
        if (std::chrono::duration<double>(current_time - start_time).count() > time_limit_seconds) {
            break;
        }

        improved = false;
        for (int i = 1; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int j_next = (j + 1) % n;

                double current_dist = dist_sq(cities[tour[i-1]], cities[tour[i]]) + dist_sq(cities[tour[j]], cities[tour[j_next]]);
                double new_dist = dist_sq(cities[tour[i-1]], cities[tour[j]]) + dist_sq(cities[tour[i]], cities[tour[j_next]]);

                if (new_dist < current_dist) {
                    std::reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

std::vector<int> solve_grid_nn(const std::vector<Point>& cities) {
    int n = cities.size();
    if (n == 0) return {};

    coord_t min_x = cities[0].x, max_x = cities[0].x;
    coord_t min_y = cities[0].y, max_y = cities[0].y;
    for(int i = 1; i < n; ++i) {
        min_x = std::min(min_x, cities[i].x);
        max_x = std::max(max_x, cities[i].x);
        min_y = std::min(min_y, cities[i].y);
        max_y = std::max(max_y, cities[i].y);
    }

    int grid_dim = static_cast<int>(sqrt(n) / 2.0);
    if (grid_dim < 1) grid_dim = 1;

    coord_t cell_width = (max_x - min_x) / grid_dim;
    coord_t cell_height = (max_y - min_y) / grid_dim;
    if (cell_width <= 0) cell_width = 1;
    if (cell_height <= 0) cell_height = 1;

    std::vector<std::vector<int>> grid(grid_dim * grid_dim);
    for (int i = 0; i < n; ++i) {
        int gx = static_cast<int>((cities[i].x - min_x) / cell_width);
        int gy = static_cast<int>((cities[i].y - min_y) / cell_height);
        if (gx >= grid_dim) gx = grid_dim - 1;
        if (gy >= grid_dim) gy = grid_dim - 1;
        if (gx < 0) gx = 0;
        if (gy < 0) gy = 0;
        grid[gy * grid_dim + gx].push_back(i);
    }

    std::vector<int> tour;
    tour.reserve(n);
    std::vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current_city = 0;
    
    for (int k = 1; k < n; ++k) {
        int best_next = -1;
        double min_dist_sq = std::numeric_limits<double>::max();

        int gx = static_cast<int>((cities[current_city].x - min_x) / cell_width);
        int gy = static_cast<int>((cities[current_city].y - min_y) / cell_height);
        if (gx >= grid_dim) gx = grid_dim - 1;
        if (gy >= grid_dim) gy = grid_dim - 1;
        if (gx < 0) gx = 0;
        if (gy < 0) gy = 0;

        for (int i = std::max(0, gx - 1); i <= std::min(grid_dim - 1, gx + 1); ++i) {
            for (int j = std::max(0, gy - 1); j <= std::min(grid_dim - 1, gy + 1); ++j) {
                int grid_idx = j * grid_dim + i;
                for (int city_idx : grid[grid_idx]) {
                    if (!visited[city_idx]) {
                        double d_sq = dist_sq(cities[current_city], cities[city_idx]);
                        if (d_sq < min_dist_sq) {
                            min_dist_sq = d_sq;
                            best_next = city_idx;
                        }
                    }
                }
            }
        }
        
        if (best_next == -1) {
            for(int i=0; i<n; ++i) {
                if(!visited[i]) {
                    double d_sq = dist_sq(cities[current_city], cities[i]);
                    if (d_sq < min_dist_sq) {
                        min_dist_sq = d_sq;
                        best_next = i;
                    }
                }
            }
        }
        
        if (best_next != -1) {
            current_city = best_next;
            tour.push_back(current_city);
            visited[current_city] = true;
        } else {
             for(int i=0; i<n; ++i) {
                 if(!visited[i]) {
                     tour.push_back(i);
                     visited[i] = true;
                     current_city = i;
                     break;
                 }
             }
        }
    }

    return tour;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int n;
    std::cin >> n;

    if (n == 0) {
        return 0;
    }

    std::vector<Point> cities(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }
    
    if (n == 1){
        std::cout << 0 << std::endl;
        return 0;
    }

    std::vector<int> tour;

    if (n <= 5000) {
        tour = solve_nearest_neighbor(cities);
        two_opt(tour, cities, 5.0); // 5 seconds time limit for 2-opt
    } else {
        tour = solve_grid_nn(cities);
    }

    auto it_zero = std::find(tour.begin(), tour.end(), 0);
    if (it_zero != tour.end()) {
        std::rotate(tour.begin(), it_zero, tour.end());
    }

    for (int i = 0; i < n; ++i) {
        std::cout << tour[i] << (i == n - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
