
use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    // A simplified area calculation using the shoelace formula.
    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let p1 = self.vertices[i];
            let p2 = self.vertices[(i + 1) % n];
            area += p1.x * p2.y - p2.x * p1.y;
        }
        (area / 2.0).abs()
    }

    // A simplified rotation around the origin. A real implementation would rotate around the centroid.
    fn rotated(&self, angle: f64) -> Polygon {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let rotated_vertices = self.vertices.iter().map(|p| {
            Point {
                x: p.x * cos_a - p.y * sin_a,
                y: p.x * sin_a + p.y * cos_a,
            }
        }).collect();
        Polygon { vertices: rotated_vertices }
    }

    fn translated(&self, dx: f64, dy: f64) -> Polygon {
        let translated_vertices = self.vertices.iter().map(|p| {
            Point { x: p.x + dx, y: p.y + dy }
        }).collect();
        Polygon { vertices: translated_vertices }
    }
}

#[derive(Debug, Clone)]
struct Piece {
    id: usize,
    polygon: Polygon,
}

struct PlacedPiece {
    piece_id: usize,
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
    polygon: Polygon,
}

// A very simplified check. This only checks if the bounding boxes intersect.
// A real solution needs a Separating Axis Theorem (SAT) implementation for convex polygons,
// or a more complex algorithm for concave polygons.
fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    let (min1, max1) = bounding_box(poly1);
    let (min2, max2) = bounding_box(poly2);

    if max1.x < min2.x || max2.x < min1.x {
        return false;
    }
    if max1.y < min2.y || max2.y < min1.y {
        return false;
    }
    // This is not a full overlap check, just a bounding box check.
    // A full implementation would require more complex geometry calculations.
    true
}

// A very simplified check. This only checks if all points of the inner polygon are inside the
// bounding box of the outer polygon. This is not a correct check for complex polygons.
fn polygon_is_inside(inner: &Polygon, outer: &Polygon) -> bool {
    let (min_outer, max_outer) = bounding_box(outer);
    for p in &inner.vertices {
        if p.x < min_outer.x || p.x > max_outer.x || p.y < min_outer.y || p.y > max_outer.y {
            return false;
        }
    }
    true
}

fn bounding_box(poly: &Polygon) -> (Point, Point) {
    let mut min_x = f64::MAX;
    let mut min_y = f64::MAX;
    let mut max_x = f64::MIN;
    let mut max_y = f64::MIN;

    for p in &poly.vertices {
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    (Point { x: min_x, y: min_y }, Point { x: max_x, y: max_y })
}

fn solve() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    let first_line = lines.next().unwrap();
    let mut parts = first_line.split_whitespace();
    let sv: usize = parts.next().unwrap().parse().unwrap();
    let np: usize = parts.next().unwrap().parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let mut parts = line.split_whitespace();
        let x: f64 = parts.next().unwrap().parse().unwrap();
        let y: f64 = parts.next().unwrap().parse().unwrap();
        stock_vertices.push(Point { x, y });
    }
    let stock_polygon = Polygon { vertices: stock_vertices };

    let mut pieces = Vec::new();
    for i in 0..np {
        let pv: usize = lines.next().unwrap().parse().unwrap();
        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let mut parts = line.split_whitespace();
            let x: f64 = parts.next().unwrap().parse().unwrap();
            let y: f64 = parts.next().unwrap().parse().unwrap();
            piece_vertices.push(Point { x, y });
        }
        pieces.push(Piece { id: i, polygon: Polygon { vertices: piece_vertices } });
    }

    // A simple greedy strategy: sort pieces by area, descending.
    pieces.sort_by(|a, b| b.polygon.area().partial_cmp(&a.polygon.area()).unwrap());

    let mut placements = Vec::new();
    let mut placed_on_stocks: Vec<Vec<PlacedPiece>> = vec![vec![]];

    for piece in pieces {
        let mut placed = false;
        for stock_index in 0..placed_on_stocks.len() {
            // Simplified placement: try a few rotations and a grid of positions.
            // A better approach would use a more sophisticated placement search.
            for &angle in &[0.0, 1.5708, 3.14159, 4.71239] { // 0, 90, 180, 270 degrees
                let rotated_piece = piece.polygon.rotated(angle);
                // Try to place on a grid
                for i in 0..10 {
                    for j in 0..10 {
                        let dx = i as f64 * 10.0; // These are arbitrary step sizes
                        let dy = j as f64 * 10.0;
                        let translated_piece = rotated_piece.translated(dx, dy);

                        if polygon_is_inside(&translated_piece, &stock_polygon) {
                            let mut overlaps = false;
                            for placed_piece in &placed_on_stocks[stock_index] {
                                if polygons_overlap(&translated_piece, &placed_piece.polygon) {
                                    overlaps = true;
                                    break;
                                }
                            }

                            if !overlaps {
                                let placement = PlacedPiece {
                                    piece_id: piece.id,
                                    stock_index,
                                    x: dx,
                                    y: dy,
                                    rotation: angle,
                                    polygon: translated_piece,
                                };
                                placed_on_stocks[stock_index].push(placement);
                                placed = true;
                                break;
                            }
                        }
                    }
                    if placed { break; }
                }
                if placed { break; }
            }
            if placed { break; }
        }

        if !placed {
            // If it doesn't fit in any existing stock, create a new one.
            placed_on_stocks.push(vec![]);
            let stock_index = placed_on_stocks.len() - 1;
            // Place it at (0,0) of the new stock. This is a gross simplification.
            let placement = PlacedPiece {
                piece_id: piece.id,
                stock_index,
                x: 0.0,
                y: 0.0,
                rotation: 0.0,
                polygon: piece.polygon.clone(),
            };
            placed_on_stocks[stock_index].push(placement);
        }
    }

    let mut final_placements = Vec::with_capacity(np);
    for stock_placements in placed_on_stocks.iter() {
        for p in stock_placements {
            final_placements.push(p);
        }
    }
    final_placements.sort_by_key(|p| p.piece_id);


    println!("{}", placed_on_stocks.len());
    for p in &final_placements {
        println!("{} {} {} {}", p.stock_index, p.x, p.y, p.rotation);
    }
}

fn main() {
    // This is a simplified solution to a very complex problem (NP-hard).
    // The provided solution uses a greedy algorithm and makes many simplifications,
    // especially in collision detection and placement strategy.
    //
    // A robust solution would require:
    // 1. A proper polygon clipping library (like `geo-types` with `clipper`).
    // 2. A robust collision detection algorithm (like Separating Axis Theorem for convex polygons).
    // 3. A more sophisticated placement heuristic (e.g., nesting algorithms, genetic algorithms).
    // 4. A strategy to handle concave polygons if they can appear.
    //
    // This code is a starting point and will not produce optimal results,
    // especially for complex shapes.
    solve();
}
