use std::io::{self, BufRead};
use std::vec::Vec;

struct Solver {
    width: i32,
    height: i32,
    obstacles: Vec<Vec<bool>>,
    require_cycle: bool,
    path: Vec<(i32, i32)>,
    visited: Vec<Vec<bool>>,
    total_cells_to_visit: usize,
}

impl Solver {
    fn new(width: i32, height: i32, obstacle_coords: &[(i32, i32)], require_cycle: bool) -> Self {
        let mut obstacles = vec![vec![false; height as usize]; width as usize];
        let mut obstacle_count = 0;
        for &(x, y) in obstacle_coords {
            if x >= 0 && x < width && y >= 0 && y < height {
                if !obstacles[x as usize][y as usize] {
                    obstacles[x as usize][y as usize] = true;
                    obstacle_count += 1;
                }
            }
        }

        let total_cells_to_visit = (width * height) as usize - obstacle_count;
        let visited = vec![vec![false; height as usize]; width as usize];

        Solver {
            width,
            height,
            obstacles,
            require_cycle,
            path: Vec::with_capacity(total_cells_to_visit),
            visited,
            total_cells_to_visit,
        }
    }

    fn solve(&mut self) -> bool {
        if self.width <= 0 || self.height <= 0 {
            return self.total_cells_to_visit == 0;
        }
        if self.obstacles[0][0] {
            return false;
        }

        if !self.check_color() {
            return false;
        }

        self.path.push((0, 0));
        self.visited[0][0] = true;

        self.backtrack()
    }

    fn check_color(&self) -> bool {
        if self.total_cells_to_visit == 0 {
            return true;
        }

        let mut white_count = 0;
        let mut black_count = 0;
        for x in 0..self.width {
            for y in 0..self.height {
                if !self.obstacles[x as usize][y as usize] {
                    if (x + y) % 2 == 0 {
                        white_count += 1;
                    } else {
                        black_count += 1;
                    }
                }
            }
        }

        if self.require_cycle {
            if self.total_cells_to_visit % 2 != 0 { return false; }
            return white_count == black_count;
        } else {
            let start_is_white = (0 + 0) % 2 == 0;
            if start_is_white {
                return white_count == black_count || white_count == black_count + 1;
            } else { // start is black
                return white_count == black_count || black_count == white_count + 1;
            }
        }
    }

    fn is_valid(&self, x: i32, y: i32) -> bool {
        x >= 0 && x < self.width && y >= 0 && y < self.height && !self.obstacles[x as usize][y as usize]
    }

    fn get_neighbors(&self, x: i32, y: i32) -> Vec<(i32, i32)> {
        let mut neighbors = Vec::with_capacity(4);
        let moves = [(0, 1), (0, -1), (1, 0), (-1, 0)];
        for &(dx, dy) in &moves {
            let nx = x + dx;
            let ny = y + dy;
            if self.is_valid(nx, ny) {
                neighbors.push((nx, ny));
            }
        }
        neighbors
    }

    fn backtrack(&mut self) -> bool {
        if self.path.len() == self.total_cells_to_visit {
            if self.require_cycle {
                let last = self.path.last().unwrap();
                let (lx, ly) = (last.0, last.1);
                // Check if (0,0) is a neighbor
                return (lx == 0 && ly == 1) || (lx == 1 && ly == 0);
            }
            return true;
        }

        let (x, y) = *self.path.last().unwrap();
        
        let mut neighbors = self.get_neighbors(x, y);

        // Warnsdorff's rule adaptation: sort neighbors by their number of unvisited neighbors
        neighbors.sort_by_key(|&(nx, ny)| {
            if self.visited[nx as usize][ny as usize] {
                return 100;
            }
            self.get_neighbors(nx, ny)
                .iter()
                .filter(|&&(nnx, nny)| !self.visited[nnx as usize][nny as usize])
                .count()
        });

        for &(nx, ny) in &neighbors {
            if self.visited[nx as usize][ny as usize] {
                continue;
            }

            // Pruning 1: Don't close a cycle prematurely
            if self.path.len() < self.total_cells_to_visit - 1 {
                let mut has_other_visited_neighbor = false;
                for &(nnx, nny) in &self.get_neighbors(nx, ny) {
                    if (nnx, nny) != (x, y) && self.visited[nnx as usize][nny as usize] {
                        has_other_visited_neighbor = true;
                        break;
                    }
                }
                if has_other_visited_neighbor {
                    continue;
                }
            }
            
            self.path.push((nx, ny));
            self.visited[nx as usize][ny as usize] = true;

            // Pruning 2: Detect fragmentation
            let mut disconnected = false;
            for &(vx, vy) in &self.get_neighbors(x, y) {
                if !self.visited[vx as usize][vy as usize] {
                    let unvisited_neighbors_count = self.get_neighbors(vx, vy)
                        .iter()
                        .filter(|&&(nnx, nny)| !self.visited[nnx as usize][nny as usize])
                        .count();
                    
                    if unvisited_neighbors_count == 0 && self.path.len() < self.total_cells_to_visit {
                        disconnected = true;
                        break;
                    }
                }
            }

            if !disconnected {
                if self.backtrack() {
                    return true;
                }
            }

            self.path.pop();
            self.visited[nx as usize][ny as usize] = false;
        }

        false
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let width: i32 = parts.next().unwrap().parse().unwrap();
    let height: i32 = parts.next().unwrap().parse().unwrap();
    let obstacle_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle_val: i32 = parts.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle_val == 1;

    let mut obstacle_coords = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let mut coords = line.split_whitespace();
        let x: i32 = coords.next().unwrap().parse().unwrap();
        let y: i32 = coords.next().unwrap().parse().unwrap();
        obstacle_coords.push((x, y));
    }

    let mut solver = Solver::new(width, height, &obstacle_coords, require_cycle);

    if solver.solve() {
        println!("{}", solver.path.len());
        for (x, y) in solver.path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
