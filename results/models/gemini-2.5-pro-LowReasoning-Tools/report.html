struct Point {

// Calculate squared Euclidean distance to avoid sqrt
inline double dist_sq(const Point& p1, const Point& p2) {
    double dx = p1.x - p2.x;
    double dy = p1.y - p2.y;
    return dx * dx + dy * dy;
}

std::vector<int> solve_nearest_neighbor(const std::vector<Point>& cities) {
    int n = cities.size();
    if (n == 0) return {};

    std::vector<int> tour;
    tour.reserve(n);
    std::vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current_city = 0;

    for (int i = 1; i < n; ++i) {
        int next_city = -1;
        double min_dist = std::numeric_limits<double>::max();

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d = dist_sq(cities[current_city], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next_city = j;
                }
            }
        }
        if (next_city != -1) {
            tour.push_back(next_city);
            visited[next_city] = true;
            current_city = next_city;
        } else {
            // Fallback for disconnected components (should not happen in TSP)
            for(int j=0; j<n; ++j) {
                if(!visited[j]) {
                    tour.push_back(j);
                    visited[j] = true;
                    current_city = j;
                    break;
                }
            }
        }
    }
    return tour;
}

void two_opt(std::vector<int>& tour, const std::vector<Point>& cities, double time_limit_seconds) {
    int n = tour.size();
    if (n < 4) return;

    auto start_time = std::chrono::high_resolution_clock::now();
    bool improved = true;
    while (improved) {
        auto current_time = std::chrono::high_resolution_clock::now();
        if (std::chrono::duration<double>(current_time - start_time).count() > time_limit_seconds) {
            break;
        }

        improved = false;
        for (int i = 1; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int j_next = (j + 1) % n;

                double current_dist = dist_sq(cities[tour[i-1]], cities[tour[i]]) + dist_sq(cities[tour[j]], cities[tour[j_next]]);
                double new_dist = dist_sq(cities[tour[i-1]], cities[tour[j]]) + dist_sq(cities[tour[i]], cities[tour[j_next]]);

                if (new_dist < current_dist) {
                    std::reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

    let n = nodes.len();

        return;

    let area = (n as f64).powf(2.0);
    let k = (area / n as f64).sqrt();
    let mut temp = 1.0;

        // Repulsive forces
        for i in 0..n {
            nodes[i].dx = 0.0;
            nodes[i].dy = 0.0;

                    continue;

                        disconnected = true;
                        break;
                    }
                }
            }

                if self.backtrack() {
                    return true;
                }
            }

            self.path.pop();

        for i in 0..n {
            let p1 = self.vertices[i];
            let p2 = self.vertices[(i + 1) % n];
            area += p1.x * p2.y - p2.x * p1.y;

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    // A simplified area calculation using the shoelace formula.
    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let p1 = self.vertices[i];
            let p2 = self.vertices[(i + 1) % n];
            area += p1.x * p2.y - p2.x * p1.y;
        }
        (area / 2.0).abs()
    }

    // A simplified rotation around the origin. A real implementation would rotate around the centroid.
    fn rotated(&self, angle: f64) -> Polygon {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let rotated_vertices = self.vertices.iter().map(|p| {
            Point {
                x: p.x * cos_a - p.y * sin_a,
                y: p.x * sin_a + p.y * cos_a,
            }
        }).collect();
        Polygon { vertices: rotated_vertices }
    }

    fn translated(&self, dx: f64, dy: f64) -> Polygon {
        let translated_vertices = self.vertices.iter().map(|p| {
            Point { x: p.x + dx, y: p.y + dy }
        }).collect();
        Polygon { vertices: translated_vertices }
    }
}

// A very simplified check. This only checks if the bounding boxes intersect.
// A real solution needs a Separating Axis Theorem (SAT) implementation for convex polygons,
// or a more complex algorithm for concave polygons.
fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    let (min1, max1) = bounding_box(poly1);
    let (min2, max2) = bounding_box(poly2);

    if max1.x < min2.x || max2.x < min1.x {
        return false;
    }
    if max1.y < min2.y || max2.y < min1.y {
        return false;
    }
    // This is not a full overlap check, just a bounding box check.
    // A full implementation would require more complex geometry calculations.
    true
}

// A very simplified check. This only checks if all points of the inner polygon are inside the
// bounding box of the outer polygon. This is not a correct check for complex polygons.
fn polygon_is_inside(inner: &Polygon, outer: &Polygon) -> bool {
    let (min_outer, max_outer) = bounding_box(outer);
    for p in &inner.vertices {
        if p.x < min_outer.x || p.x > max_outer.x || p.y < min_outer.y || p.y > max_outer.y {
            return false;
        }
    }
    true
}

fn bounding_box(poly: &Polygon) -> (Point, Point) {
    let mut min_x = f64::MAX;
    let mut min_y = f64::MAX;
    let mut max_x = f64::MIN;
    let mut max_y = f64::MIN;

    for p in &poly.vertices {
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    (Point { x: min_x, y: min_y }, Point { x: max_x, y: max_y })
}
