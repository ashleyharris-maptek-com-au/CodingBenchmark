import sys

def solve():
    input_data = sys.stdin.read().splitlines()
    if not input_data:
        return
    
    u, s = map(int, input_data[0].split())
    
    sets = []
    for i in range(1, s + 1):
        sets.append(list(map(int, input_data[i].split())))
        
    # matrix[element] = set of row indices that contain this element
    element_to_sets = {e: set() for e in range(u)}
    for row_idx, elements in enumerate(sets):
        for e in elements:
            if e in element_to_sets:
                element_to_sets[e].add(row_idx)
    
    # set_to_elements[row_idx] = set of elements in this set
    set_to_elements = {row_idx: set(elements) for row_idx, elements in enumerate(sets)}
    
    solution = []
    
    def backtrack(remaining_elements, used_rows):
        if not remaining_elements:
            return used_rows
        
        # Heuristic: choose element with fewest sets
        min_element = min(remaining_elements, key=lambda e: len(element_to_sets[e]))
        
        for row_idx in list(element_to_sets[min_element]):
            row_elements = set_to_elements[row_idx]
            
            # Check if this row can be added (all its elements are in remaining_elements)
            if row_elements.issubset(remaining_elements):
                # Choose this row
                new_used_rows = used_rows + [row_idx]
                new_remaining = remaining_elements - row_elements
                
                # Further refine the search space
                # A row can only be used if it doesn't conflict with chosen row
                # We do this by removing all rows that share an element with row_elements
                
                result = backtrack(new_remaining, new_used_rows)
                if result:
                    return result
                    
        return None

    result = backtrack(set(range(u)), [])
    
    if result is not None:
        print("SOLUTION")
        print(*(result))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
