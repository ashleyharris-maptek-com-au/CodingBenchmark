using System;
using System.Collections.Generic;

namespace TetrisAI
{
    public class Program
    {
        public static void Main(string[] args)
        {
            string? line = Console.ReadLine();
            if (line == null) return;
            string[] input = line.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int width = int.Parse(input[0]);
            int height = int.Parse(input[1]);
            int numPieces = int.Parse(input[2]);

            Board board = new Board(width, height);
            
            for (int i = 0; i < numPieces; i++)
            {
                string? pieceType = Console.ReadLine();
                if (pieceType == null) break;
                Move bestMove = board.FindBestMove(pieceType.Trim());
                Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");
                Console.Out.Flush();
                
                string? result = Console.ReadLine();
                if (result == null || result.Trim() == "gameover") break;
                board.ApplyMove(pieceType.Trim(), bestMove);
            }
        }
    }

    public class Move
    {
        public int Rotation { get; set; }
        public int Column { get; set; }
        public double Score { get; set; }
    }

    public class Board
    {
        private int width;
        private int height;
        private int[,] grid;

        private static readonly Dictionary<string, int[][][]> Tetrominoes = new Dictionary<string, int[][][]>
        {
            { "I", new int[][][] { new int[][] { new[] { 1, 1, 1, 1 } }, new int[][] { new[] { 1 }, new[] { 1 }, new[] { 1 }, new[] { 1 } } } },
            { "O", new int[][][] { new int[][] { new[] { 1, 1 }, new[] { 1, 1 } } } },
            { "T", new int[][][] { new int[][] { new[] { 0, 1, 0 }, new[] { 1, 1, 1 } }, new int[][] { new[] { 1, 0 }, new[] { 1, 1 }, new[] { 1, 0 } }, new int[][] { new[] { 1, 1, 1 }, new[] { 0, 1, 0 } }, new int[][] { new[] { 0, 1 }, new[] { 1, 1 }, new[] { 0, 1 } } } },
            { "S", new int[][][] { new int[][] { new[] { 0, 1, 1 }, new[] { 1, 1, 0 } }, new int[][] { new[] { 1, 0 }, new[] { 1, 1 }, new[] { 0, 1 } } } },
            { "Z", new int[][][] { new int[][] { new[] { 1, 1, 0 }, new[] { 0, 1, 1 } }, new int[][] { new[] { 0, 1 }, new[] { 1, 1 }, new[] { 1, 0 } } } },
            { "J", new int[][][] { new int[][] { new[] { 1, 0, 0 }, new[] { 1, 1, 1 } }, new int[][] { new[] { 1, 1 }, new[] { 1, 0 }, new[] { 1, 0 } }, new int[][] { new[] { 1, 1, 1 }, new[] { 0, 0, 1 } }, new int[][] { new[] { 0, 1 }, new[] { 0, 1 }, new[] { 1, 1 } } } },
            { "L", new int[][][] { new int[][] { new[] { 0, 0, 1 }, new[] { 1, 1, 1 } }, new int[][] { new[] { 1, 0 }, new[] { 1, 0 }, new[] { 1, 1 } }, new int[][] { new[] { 1, 1, 1 }, new[] { 1, 0, 0 } }, new int[][] { new[] { 1, 1 }, new[] { 0, 1 }, new[] { 0, 1 } } } }
        };

        public Board(int width, int height)
        {
            this.width = width;
            this.height = height;
            this.grid = new int[height, width];
        }

        public Move FindBestMove(string pieceType)
        {
            Move bestMove = new Move { Score = double.MinValue };
            var rotations = Tetrominoes[pieceType];

            for (int r = 0; r < rotations.Length; r++)
            {
                int pieceW = rotations[r][0].Length;
                for (int c = 0; c <= width - pieceW; c++)
                {
                    int row = GetLandingRow(rotations[r], c);
                    if (row == -1) continue;

                    double score = EvaluateBoard(rotations[r], r, c, row);
                    if (score > bestMove.Score)
                    {
                        bestMove = new Move { Rotation = r, Column = c, Score = score };
                    }
                }
            }
            return bestMove;
        }

        private int GetLandingRow(int[][] shape, int col)
        {
            for (int r = 0; r <= height - shape.Length; r++)
            {
                if (IsCollision(shape, col, r)) return r - 1;
            }
            return height - shape.Length;
        }

        private bool IsCollision(int[][] shape, int col, int row)
        {
            for (int r = 0; r < shape.Length; r++)
            {
                for (int c = 0; c < shape[r].Length; c++)
                {
                    if (shape[r][c] == 1 && (row + r >= height || grid[row + r, col + c] == 1)) return true;
                }
            }
            return false;
        }

        private double EvaluateBoard(int[][] shape, int rot, int col, int row)
        {
            // Simple heuristic for now: minimize height and holes
            int heightScore = 0;
            int holeScore = 0;
            for (int r = 0; r < height; r++)
            {
                for (int c = 0; c < width; c++)
                {
                    if (grid[r, c] == 1 || (r >= row && r < row + shape.Length && c >= col && c < col + shape[r - row].Length && shape[r - row][c - col] == 1))
                    {
                        heightScore += (height - r);
                    }
                    else if (r > 0 && grid[r - 1, c] == 1)
                    {
                        holeScore++;
                    }
                }
            }
            return -1.0 * heightScore - 5.0 * holeScore;
        }

        public void ApplyMove(string pieceType, Move move)
        {
            int[][] shape = Tetrominoes[pieceType][move.Rotation];
            int row = GetLandingRow(shape, move.Column);
            for (int r = 0; r < shape.Length; r++)
            {
                for (int c = 0; c < shape[r].Length; c++)
                {
                    if (shape[r][c] == 1) grid[row + r, move.Column + c] = 1;
                }
            }
            ClearLines();
        }

        private void ClearLines()
        {
            for (int r = height - 1; r >= 0; r--)
            {
                bool full = true;
                for (int c = 0; c < width; c++) if (grid[r, c] == 0) full = false;
                if (full)
                {
                    for (int r2 = r; r2 > 0; r2--)
                        for (int c = 0; c < width; c++) grid[r2, c] = grid[r2 - 1, c];
                    for (int c = 0; c < width; c++) grid[0, c] = 0;
                    r++;
                }
            }
        }
    }
}
