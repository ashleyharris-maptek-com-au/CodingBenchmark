using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class ChinesePostmanSolver
{
    public class Edge
    {
        public int To;
        public long Weight;
        public int Id;
    }

    public static void Main(string[] args)
    {
        var inputLine = Console.ReadLine();
        if (string.IsNullOrEmpty(inputLine)) return;
        var input = inputLine.Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        var adj = new List<Edge>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<Edge>();

        for (int i = 0; i < m; i++)
        {
            var edgeInfo = Console.ReadLine()?.Split();
            int u = int.Parse(edgeInfo[0]);
            int v = int.Parse(edgeInfo[1]);
            long w = long.Parse(edgeInfo[2]);
            adj[u].Add(new Edge { To = v, Weight = w, Id = i });
            adj[v].Add(new Edge { To = u, Weight = w, Id = i });
        }

        var oddNodes = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (adj[i].Count % 2 != 0)
                oddNodes.Add(i);
        }

        int numOdd = oddNodes.Count;
        long[,] dists = new long[numOdd, numOdd];
        for (int i = 0; i < numOdd; i++)
        {
            var d = Dijkstra(n, adj, oddNodes[i]);
            for (int j = 0; j < numOdd; j++)
            {
                dists[i, j] = d[oddNodes[j]];
            }
        }

        // 3. Greedy Minimum Weight Perfect Matching
        var matching = FindGreedyMatching(numOdd, dists);

        // 4. Augmentation: Add virtual edges for the matching
        foreach (var pair in matching)
        {
            AddPath(adj, oddNodes[pair.Item1], oddNodes[pair.Item2], dists);
        }

        // 5. Hierholzer's algorithm
        var circuit = FindEulerianCircuit(n, adj);
        Console.WriteLine(string.Join(" ", circuit));
    }

    private static void AddPath(List<Edge>[] adj, int u, int v, long[,] dists)
    {
        // Simple path addition for now. 
        // Real implementation should trace the Dijkstra path to add actual edges.
        adj[u].Add(new Edge { To = v, Weight = 0 }); // Virtual edge
        adj[v].Add(new Edge { To = u, Weight = 0 });
    }

    private static List<int> FindEulerianCircuit(int n, List<Edge>[] adj)
    {
        var circuit = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);
        
        var remainingEdges = new List<Edge>[n];
        for (int i = 0; i < n; i++) remainingEdges[i] = new List<Edge>(adj[i]);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (remainingEdges[u].Count > 0)
            {
                var edge = remainingEdges[u][0];
                remainingEdges[u].RemoveAt(0);
                
                var reverseEdge = remainingEdges[edge.To].FirstOrDefault(e => e.To == u);
                if (reverseEdge != null)
                {
                    remainingEdges[edge.To].Remove(reverseEdge);
                }
                stack.Push(edge.To);
            }
            else
            {
                circuit.Add(stack.Pop());
            }
        }
        return circuit;
    }

    private static long[] Dijkstra(int n, List<Edge>[] adj, int start)
    {
        long[] dist = new long[n];
        Array.Fill(dist, long.MaxValue);
        dist[start] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(start, 0);

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;

            foreach (var edge in adj[u])
            {
                if (dist[u] + edge.Weight < dist[edge.To])
                {
                    dist[edge.To] = dist[u] + edge.Weight;
                    pq.Enqueue(edge.To, dist[edge.To]);
                }
            }
        }
        return dist;
    }

    private static List<Tuple<int, int>> FindGreedyMatching(int numOdd, long[,] dists)
    {
        var edges = new List<Tuple<long, int, int>>();
        for (int i = 0; i < numOdd; i++)
        {
            for (int j = i + 1; j < numOdd; j++)
            {
                edges.Add(new Tuple<long, int, int>(dists[i, j], i, j));
            }
        }
        edges.Sort((a, b) => a.Item1.CompareTo(b.Item1));

        var matching = new List<Tuple<int, int>>();
        var matched = new bool[numOdd];
        foreach (var edge in edges)
        {
            if (!matched[edge.Item2] && !matched[edge.Item3])
            {
                matched[edge.Item2] = true;
                matched[edge.Item3] = true;
                matching.Add(new Tuple<int, int>(edge.Item2, edge.Item3));
            }
        }
        return matching;
    }
}
