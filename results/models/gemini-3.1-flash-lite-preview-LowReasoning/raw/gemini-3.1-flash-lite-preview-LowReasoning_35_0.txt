use std::collections::{HashSet, VecDeque};

/// Represents a graph using an adjacency list.
struct Graph {
    nodes: usize,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(nodes: usize) -> Self {
        Graph {
            nodes,
            adj: vec![vec![]; nodes],
        }
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        self.adj[u].push(v);
        self.adj[v].push(u);
    }
}

/// Solves the Minimum Dominating Set problem using a simple backtracking approach.
/// This is an NP-hard problem, so this is exponential in the worst case.
fn solve_min_dominating_set(graph: &Graph) -> HashSet<usize> {
    let mut min_set: HashSet<usize> = (0..graph.nodes).collect();
    
    // Helper function for backtracking
    fn backtrack(
        graph: &Graph,
        current_set: &mut HashSet<usize>,
        covered: &mut Vec<bool>,
        covered_count: usize,
        node_idx: usize,
        min_set: &mut HashSet<usize>,
    ) {
        if covered_count == graph.nodes {
            if current_set.len() < min_set.len() {
                *min_set = current_set.clone();
            }
            return;
        }

        if node_idx == graph.nodes {
            return;
        }

        // Pruning: if current set is already larger than min_set, stop
        if current_set.len() >= min_set.len() {
            return;
        }

        // Option 1: Include node_idx in the dominating set
        current_set.insert(node_idx);
        let mut newly_covered = Vec::new();
        
        // Mark node itself as covered
        if !covered[node_idx] {
            covered[node_idx] = true;
            newly_covered.push(node_idx);
        }
        
        // Mark neighbors as covered
        for &neighbor in &graph.adj[node_idx] {
            if !covered[neighbor] {
                covered[neighbor] = true;
                newly_covered.push(neighbor);
            }
        }

        backtrack(
            graph,
            current_set,
            covered,
            covered_count + newly_covered.len(),
            node_idx + 1,
            min_set,
        );

        // Backtrack
        for &node in &newly_covered {
            covered[node] = false;
        }
        current_set.remove(&node_idx);

        // Option 2: Exclude node_idx from the dominating set
        backtrack(
            graph,
            current_set,
            covered,
            covered_count,
            node_idx + 1,
            min_set,
        );
    }

    let mut current_set = HashSet::new();
    let mut covered = vec![false; graph.nodes];
    backtrack(graph, &mut current_set, &mut covered, 0, 0, &mut min_set);
    
    min_set
}

fn main() {
    // Example usage
    let mut graph = Graph::new(5);
    graph.add_edge(0, 1);
    graph.add_edge(0, 2);
    graph.add_edge(1, 3);
    graph.add_edge(2, 4);

    let result = solve_min_dominating_set(&graph);
    println!("Minimum Dominating Set: {:?}", result);
}
