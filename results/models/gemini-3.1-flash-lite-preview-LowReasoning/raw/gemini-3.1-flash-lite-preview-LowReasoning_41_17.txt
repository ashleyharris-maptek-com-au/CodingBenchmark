; SPIR-V Assembly Fragment Shader
; Simple fragment shader that outputs red (1.0, 0.0, 0.0, 1.0)

               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; --- Decorations ---
               OpDecorate %outColor Location 0

; --- Types and Constants ---
%void        = OpTypeVoid
%float       = OpTypeFloat 32
%vec4        = OpTypeVector %float 4
%ptr_output  = OpTypePointer Output %vec4

; --- Variables ---
%outColor    = OpVariable %ptr_output Output

; --- Main Function ---
%void_f      = OpTypeFunction %void
%main        = OpFunction %void None %void_f
%label       = OpLabel

; Define red color: (1.0, 0.0, 0.0, 1.0)
%f1          = OpConstant %float 1.0
%f0          = OpConstant %float 0.0
%red         = OpConstantComposite %vec4 %f1 %f0 %f0 %f1

; Store red into output variable
               OpStore %outColor %red
               OpReturn
               OpFunctionEnd
