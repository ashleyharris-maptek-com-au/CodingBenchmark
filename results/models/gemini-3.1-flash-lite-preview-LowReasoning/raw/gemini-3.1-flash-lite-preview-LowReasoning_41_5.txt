; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_32 = OpConstant %float 32.0
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %func_void
%label = OpLabel
; Get data
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %1 Normalize %n_raw
%pos = OpLoad %vec3 %worldPos
%ubo_ptr_l = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%l_raw = OpLoad %vec4 %ubo_ptr_l
%l_vec3 = OpVectorShuffle %vec3 %l_raw %l_raw 0 1 2
%l = OpExtInst %vec3 %1 Normalize %l_vec3
%ubo_ptr_c = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%c_raw = OpLoad %vec4 %ubo_ptr_c
%c = OpVectorShuffle %vec3 %c_raw %c_raw 0 1 2
%v_raw = OpFSub %vec3 %c %pos
%v = OpExtInst %vec3 %1 Normalize %v_raw

; Diffuse
%dot_nl = OpDot %float %n %l
%diffuse = OpExtInst %float %1 FMax %dot_nl %float_0

; Specular
%neg_l = OpFNegate %vec3 %l
%r = OpExtInst %vec3 %1 Reflect %neg_l %n
%dot_rv = OpDot %float %r %v
%rv_clamped = OpExtInst %float %1 FMax %dot_rv %float_0
%specular = OpExtInst %float %1 Pow %rv_clamped %float_32

; Intensity = 0.2 + 0.5*diffuse + 0.8*specular
%term1 = OpFMul %float %float_0_5 %diffuse
%term2 = OpFMul %float %float_0_8 %specular
%intensity_tmp = OpFAdd %float %float_0_2 %term1
%intensity = OpFAdd %float %intensity_tmp %term2

%out = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
