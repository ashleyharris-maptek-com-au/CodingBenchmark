; SPIR-V Fragment Shader
; Simple fragment shader that outputs solid red (1.0, 0.0, 0.0, 1.0)

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpDecorate %outColor Location 0

; Types/Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%main_t = OpTypeFunction %void
%float_1 = OpConstant %float 1.0
%float_0 = OpConstant %float 0.0
%red = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

; Outputs
%outColor = OpVariable %Output %v4float

; Main function
%main = OpFunction %void None %main_t
%entry = OpLabel
               OpStore %outColor %red
               OpReturn
               OpFunctionEnd
