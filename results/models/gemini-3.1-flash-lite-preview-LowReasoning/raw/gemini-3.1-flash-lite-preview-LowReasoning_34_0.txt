import sys
import math

def distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve():
    # Read input
    try:
        input_data = sys.stdin.read().split()
        if not input_data:
            return
        
        N = int(input_data[0])
        K = int(input_data[1])
        CAPACITY = int(input_data[2])
        
        depot = (float(input_data[3]), float(input_data[4]))
        
        customers = []
        idx = 5
        for i in range(N):
            x = float(input_data[idx])
            y = float(input_data[idx+1])
            demand = int(input_data[idx+2])
            customers.append({'id': i, 'pos': (x, y), 'demand': demand})
            idx += 3
            
    except (EOFError, IndexError, ValueError):
        return

    # Clarke-Wright Savings Algorithm
    # 1. Initial routes: each customer is a separate route (depot -> customer -> depot)
    routes = [[c] for c in customers]
    
    # 2. Calculate savings: S(i, j) = d(depot, i) + d(depot, j) - d(i, j)
    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            s = distance(depot, customers[i]['pos']) + \
                distance(depot, customers[j]['pos']) - \
                distance(customers[i]['pos'], customers[j]['pos'])
            savings.append((s, i, j))
            
    savings.sort(key=lambda x: x[0], reverse=True)
    
    # 3. Merge routes
    # Helper to find route containing customer
    def find_route(c_id):
        for r in routes:
            if c_id in [c['id'] for c in r]:
                return r
        return None

    for s, i, j in savings:
        r1 = find_route(i)
        r2 = find_route(j)
        
        if r1 != r2 and r1 is not None and r2 is not None:
            # Check if i and j are at the ends of their routes
            # For simplicity, we only merge if they are at the ends
            # r1: [..., i] or [i, ...]
            # r2: [..., j] or [j, ...]
            
            # Check capacity
            total_demand = sum(c['demand'] for c in r1) + sum(c['demand'] for c in r2)
            if total_demand <= CAPACITY:
                # Check if i is end of r1 and j is start of r2 (or vice versa)
                # This is a simplified version of CW
                if r1[-1]['id'] == i and r2[0]['id'] == j:
                    routes.remove(r1)
                    routes.remove(r2)
                    routes.append(r1 + r2)
                elif r1[0]['id'] == i and r2[-1]['id'] == j:
                    routes.remove(r1)
                    routes.remove(r2)
                    routes.append(r2 + r1)
                elif r1[-1]['id'] == i and r2[-1]['id'] == j:
                    routes.remove(r1)
                    routes.remove(r2)
                    routes.append(r1 + r2[::-1])
                elif r1[0]['id'] == i and r2[0]['id'] == j:
                    routes.remove(r1)
                    routes.remove(r2)
                    routes.append(r1[::-1] + r2)

    # Output
    total_dist = 0
    for r in routes:
        # Depot to first
        total_dist += distance(depot, r[0]['pos'])
        # Between customers
        for k in range(len(r) - 1):
            total_dist += distance(r[k]['pos'], r[k+1]['pos'])
        # Last to depot
        total_dist += distance(r[-1]['pos'], depot)
        
    print(f"{total_dist:.4f}")
    print(len(routes))
    for r in routes:
        print(" ".join(str(c['id']) for c in r))

if __name__ == "__main__":
    solve()
