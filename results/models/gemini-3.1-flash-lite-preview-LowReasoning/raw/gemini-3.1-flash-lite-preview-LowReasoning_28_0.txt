use std::io::{self, BufRead};

fn solve() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };

    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    if parts.len() < 2 {
        return;
    }

    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Using an adjacency list approach or edge list approach.
    // For greedy, we can process edges one by one.
    // Vertex Cover greedy approximation: Pick both endpoints of an uncovered edge.
    
    let mut vertex_in_cover = vec![false; num_vertices];
    let mut cover = Vec::new();

    for line in lines {
        if let Ok(l) = line {
            let edge: Vec<usize> = l
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            
            if edge.len() < 2 {
                continue;
            }

            let u = edge[0];
            let v = edge[1];

            if !vertex_in_cover[u] && !vertex_in_cover[v] {
                vertex_in_cover[u] = true;
                vertex_in_cover[v] = true;
                cover.push(u);
                cover.push(v);
            }
        }
    }

    println!("{}", cover.len());
    for (i, v) in cover.iter().enumerate() {
        print!("{}{}", v, if i == cover.len() - 1 { "" } else { " " });
    }
    println!();
}

fn main() {
    solve();
}
