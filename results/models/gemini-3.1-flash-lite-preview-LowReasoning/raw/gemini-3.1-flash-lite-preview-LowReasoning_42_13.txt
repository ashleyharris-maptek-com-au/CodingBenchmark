To write a SPIR-V geometry shader, you need to define the entry point, execution mode (input and output primitives, vertex counts), and the shader logic.

A simple pass-through geometry shader typically takes points, lines, or triangles, and emits them unchanged.

Assuming you are using `spirv-as` to assemble the code, here is a foundational structure for a triangle pass-through geometry shader:

```llvm
; SPIR-V
; Version: 1.0
; Generator: LLVM translator; 0
; Bound: 100
; Schema: 0
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %in_pos %out_pos
               OpExecutionMode %main InputPoints ; Or InputLines / InputTriangles
               OpExecutionMode %main OutputPoints ; Or OutputLineStrip / OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %in_pos Location 0
               OpDecorate %out_pos Location 0

; Types
%void       = OpTypeVoid
%float      = OpTypeFloat 32
%v4float    = OpTypeVector %float 4
%ptr_in     = OpTypePointer Input %v4float
%ptr_out    = OpTypeOutput %v4float
%void_f     = OpTypeFunction %void

; Inputs/Outputs
%in_pos     = OpVariable %ptr_in Input
%out_pos    = OpVariable %ptr_out Output

; Main Function
%main       = OpFunction %void None %void_f
%label      = OpLabel

; Simple pass-through logic for each vertex
               OpLoopMerge %loop_merge %loop_continue None
               OpBranch %loop_start

%loop_start = OpLabel
               ; ... load input, assign to output, emit vertex, end primitive ...
               OpEmitVertex
               OpEndPrimitive
               OpBranch %loop_continue

%loop_continue = OpLabel
               OpBranch %loop_start

%loop_merge = OpLabel
               OpReturn
               OpFunctionEnd
```
