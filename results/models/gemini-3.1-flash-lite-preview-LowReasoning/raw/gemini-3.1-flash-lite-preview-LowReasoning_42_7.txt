; SPIR-V Assembly Geometry Shader: Explode Triangles
OpCapability Geometry
OpCapability Shader
%std = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_arr_v3 = OpTypePointer Input %arr_v3_3
%color_in = OpVariable %ptr_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output ; output per vertex in strip, should be out vec3. The prompt said "layout(location = 0) out vec3 outColor;". In geometry shader, it's just out vec3.

%func_void = OpTypeFunction %void

%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08
%float_3 = OpConstant %float 3.0

%main = OpFunction %void None %func_void
%entry = OpLabel

; 1. Get positions
%ptr_in_0 = OpAccessChain %ptr_arr_pv_in %gl_in %int_0
%ptr_pos_0 = OpAccessChain %ptr_out_pv %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_pos_0
%ptr_pos_1 = OpAccessChain %ptr_out_pv %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_pos_1
%ptr_pos_2 = OpAccessChain %ptr_out_pv %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_pos_2

; 2. e1 = p1-p0, e2 = p2-p0
%v3_p0 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%v3_p1 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%v3_p2 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1 = OpFSub %v3float %v3_p1 %v3_p0
%e2 = OpFSub %v3float %v3_p2 %v3_p0

; 3. N = cross(e1, e2)
%N = OpExtInst %v3float %std Cross %e1 %e2
%N_norm = OpExtInst %v3float %std Normalize %N
%offset = OpVectorTimesScalar %v3float %N_norm %float_0_08

; Process vertices
%i = OpVariable %ptr_arr_v3 %int_0 ; Wait, how to iterate 3 times? Just unroll.

; Vertex 0
%p0_new = OpFAdd %v3float %v3_p0 %offset
%p0_final = OpCompositeConstruct %v4float %p0_new %float_0 ; This is wrong, should be w=1.0
; Correction:
%float_1 = OpConstant %float 1.0
%p0_final_v4 = OpCompositeConstruct %v4float %p0_new %float_1
%ptr_out_pos = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %ptr_out_pos %p0_final_v4
; Output color (index 0)
%ptr_color_0 = OpAccessChain %ptr_arr_v3 %color_in %int_0
%color_0 = OpLoad %v3float %ptr_color_0
OpStore %outColor %color_0
OpEmitVertex

; Vertex 1
%p1_new = OpFAdd %v3float %v3_p1 %offset
%p1_final_v4 = OpCompositeConstruct %v4float %p1_new %float_1
OpStore %ptr_out_pos %p1_final_v4
%ptr_color_1 = OpAccessChain %ptr_arr_v3 %color_in %int_1
%color_1 = OpLoad %v3float %ptr_color_1
OpStore %outColor %color_1
OpEmitVertex

; Vertex 2
%p2_new = OpFAdd %v3float %v3_p2 %offset
%p2_final_v4 = OpCompositeConstruct %v4float %p2_new %float_1
OpStore %ptr_out_pos %p2_final_v4
%ptr_color_2 = OpAccessChain %ptr_arr_v3 %color_in %int_2
%color_2 = OpLoad %v3float %ptr_color_2
OpStore %outColor %color_2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
