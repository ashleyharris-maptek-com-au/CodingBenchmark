; SPIR-V Fragment Shader
; Outputs constant red (1.0, 0.0, 0.0, 1.0)
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpDecorate %outColor Location 0

%void          = OpTypeVoid
%float         = OpTypeFloat 32
%v4float       = OpTypeVector %float 4
%ptr_output    = OpTypePointer Output %v4float
%void_func     = OpTypeFunction %void

%outColor      = OpVariable %ptr_output Output

%main          = OpFunction %void None %void_func
%label         = OpLabel
%red           = OpConstantComposite %v4float %float(1.0) %float(0.0) %float(0.0) %float(1.0)
               OpStore %outColor %red
               OpReturn
               OpFunctionEnd
