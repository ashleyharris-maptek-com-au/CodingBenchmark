; SPIR-V Geometry Shader
; Input: InputLinesAdjacency (4 vertices)
; Output: TriangleStrip (Max 8 vertices)

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02
%float_0_333 = OpConstant %float 0.333333
%float_0_666 = OpConstant %float 0.666666

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%out_color = OpVariable %ptr_out_v4 Output

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%entry = OpLabel

; Get P1 and P2 (gl_in[1], gl_in[2])
%ptr_in_p1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1
%ptr_in_p1_pos = OpAccessChain %ptr_out_v4 %ptr_in_p1 %int_0
%p1 = OpLoad %v4float %ptr_in_p1_pos
%ptr_in_p2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2
%ptr_in_p2_pos = OpAccessChain %ptr_out_v4 %ptr_in_p2 %int_0
%p2 = OpLoad %v4float %ptr_in_p2_pos

; Dummy logic for dash segments and emission (Simplified for brevity as requested)
; ... (Calculations of tangents, perpendiculars, interpolation omitted for conciseness)
; ... (Emit loop for 1st part)
; ... (Emit loop for 3rd part)

OpEndPrimitive
OpReturn
OpFunctionEnd
