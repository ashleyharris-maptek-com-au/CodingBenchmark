; Geometry Shader (Input: InputTrianglesAdjacency, Output: OutputTriangleStrip)
; Flat-shading: Compute normal from face (v0, v2, v4), light, and output to color

               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %inColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %gl_out Output
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
      %float_0 = OpConstant %float 0.0
      %float_0_2 = OpConstant %float 0.2
      %float_0_8 = OpConstant %float 0.8
      %float_1 = OpConstant %float 1.0
      %float_third = OpConstant %float 0.333333333
      %float_1_v3 = OpConstantComposite %v3float %float_1 %float_1 %float_1

      %gl_PerVertex_in = OpTypeStruct %v4float

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out_ptr_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

%arr_in_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_in_v3
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output
%ptr_out_v3 = OpTypePointer Output %v3float

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Main
       %main = OpFunction %void None %func
         %lb = OpLabel

; Get positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2_v4 = OpLoad %v4float %p2_ptr
%p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4_v4 = OpLoad %v4float %p4_ptr
%p4 = OpVectorShuffle %v3float %p4_v4 %p4_v4 0 1 2

; Face normal N = cross(p2-p0, p4-p0)
%v20 = OpFSub %v3float %p2 %p0
%v40 = OpFSub %v3float %p4 %p0
%N_un = OpExtInst %v3float %1 Cross %v20 %v40
%N = OpExtInst %v3float %1 Normalize %N_un

; Diffuse lighting
%L = OpCopyObject %v3float %float_1_v3
%dot = OpExtInst %float %1 Dot %N %L
%clamped_dot = OpExtInst %float %1 FMax %dot %float_0
%diffuse = OpFMul %float %clamped_dot %float_0_8
%ambient = OpFAdd %float %diffuse %float_0_2

; Color average
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%c02 = OpFAdd %v3float %c0 %c2
%c024 = OpFAdd %v3float %c02 %c4
%base_color = OpVectorTimesScalar %v3float %c024 0.333333333

; Final color
%final_color = OpVectorTimesScalar %v3float %base_color %ambient

; Emit 3 vertices
               OpStore %outColor %final_color
               OpStore %gl_out_ptr_pos %p0_v4
               OpEmitVertex
               OpStore %gl_out_ptr_pos %p2_v4
               OpEmitVertex
               OpStore %gl_out_ptr_pos %p4_v4
               OpEmitVertex
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
