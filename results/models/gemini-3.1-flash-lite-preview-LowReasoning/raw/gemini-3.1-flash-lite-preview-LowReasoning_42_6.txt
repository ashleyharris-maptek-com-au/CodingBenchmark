; SPIR-V Geometry Shader: Identity Passthrough
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %color Location 2
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO Type (simplified structure for passthrough)
%ubo_block = OpTypeStruct %v4float ; Placeholder, not used for logic
%ptr_uniform = OpTypePointer Uniform %ubo_block
%ubo = OpVariable %ptr_uniform Uniform

%fn_main = OpTypeFunction %void

; Main
%main = OpFunction %void None %fn_main
%label_entry = OpLabel

; i = 0
%ptr_in_0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
%pos_0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%val_pos_0 = OpLoad %v4float %pos_0
%out_pos_0 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos_0 %val_pos_0
%ptr_color_0 = OpAccessChain %ptr_in_arr_v3 %color %int_0
%val_color_0 = OpLoad %v3float %ptr_color_0
OpStore %outColor %val_color_0
OpEmitVertex

; i = 1
%pos_1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1 %int_0
%val_pos_1 = OpLoad %v4float %pos_1
%out_pos_1 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos_1 %val_pos_1
%ptr_color_1 = OpAccessChain %ptr_in_arr_v3 %color %int_1
%val_color_1 = OpLoad %v3float %ptr_color_1
OpStore %outColor %val_color_1
OpEmitVertex

; i = 2
%pos_2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
%val_pos_2 = OpLoad %v4float %pos_2
%out_pos_2 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos_2 %val_pos_2
%ptr_color_2 = OpAccessChain %ptr_in_arr_v3 %color %int_2
%val_color_2 = OpLoad %v3float %ptr_color_2
OpStore %outColor %val_color_2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
