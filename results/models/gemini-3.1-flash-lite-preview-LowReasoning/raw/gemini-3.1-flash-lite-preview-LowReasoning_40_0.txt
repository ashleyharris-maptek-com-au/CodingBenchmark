#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

// Maximum Weight Independent Set solver
// x_u + x_v <= 1 means u and v cannot both be in the set.
// This is exactly the Maximum Weight Independent Set problem on a graph where edges are the given constraints.

struct Solver {
    int n, m;
    std::vector<long long> weights;
    std::vector<std::vector<int>> adj;
    
    long long max_weight = -1;
    std::vector<int> best_assignment;
    
    std::vector<int> current_assignment;
    std::vector<int> current_set; // nodes selected so far
    
    Solver(int n, int m) : n(n), m(m), weights(n), adj(n), best_assignment(n), current_assignment(n, 0) {}

    // Simple backtracking with basic pruning
    void solve(int node, long long current_weight) {
        if (node == n) {
            if (current_weight > max_weight) {
                max_weight = current_weight;
                best_assignment = current_assignment;
            }
            return;
        }

        // Branch 1: Exclude node
        current_assignment[node] = 0;
        solve(node + 1, current_weight);

        // Branch 2: Include node (if allowed)
        bool can_include = true;
        for (int neighbor : adj[node]) {
            if (neighbor < node && current_assignment[neighbor] == 1) {
                can_include = false;
                break;
            }
        }

        if (can_include) {
            current_assignment[node] = 1;
            solve(node + 1, current_weight + weights[node]);
            current_assignment[node] = 0;
        }
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int n, m;
    if (!(std::cin >> n >> m)) return 0;

    Solver solver(n, m);
    for (int i = 0; i < n; ++i) {
        std::cin >> solver.weights[i];
    }
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        // Assuming 0-indexed u, v based on n variables
        solver.adj[u].push_back(v);
        solver.adj[v].push_back(u);
    }

    solver.solve(0, 0);

    std::cout << solver.max_weight << "\n";
    for (int i = 0; i < n; ++i) {
        std::cout << solver.best_assignment[i] << (i == n - 1 ? "" : " ");
    }
    std::cout << "\n";

    return 0;
}
