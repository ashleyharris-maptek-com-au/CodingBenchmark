struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Procedural Brick Pattern
    float2 uv = input.uv * float2(8.0, 16.0);
    
    // Staggered rows: offset every other row
    float row = floor(uv.y);
    if (fmod(row, 2.0) >= 1.0) {
        uv.x += 0.5;
    }
    
    float2 brickUV = frac(uv);
    float2 brickBorder = float2(0.05, 0.1);
    
    // Check if within mortar
    bool isMortar = (brickUV.x < brickBorder.x) || (brickUV.y < brickBorder.y);
    
    float3 baseColor = isMortar ? float3(0.7, 0.7, 0.7) : float3(0.7, 0.2, 0.1);
    
    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diff = max(dot(N, L), 0.0);
    float3 lighting = (0.4 + 0.6 * diff) * baseColor;
    
    return float4(lighting, 1.0);
}
