#include <iostream>
#include <vector>
#include <string>
#include <thread>
#include <mutex>
#include <queue>
#include <atomic>
#include <fstream>
#include <cmath>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Command {
    double thrust;
    double pitch_cmd;
    double yaw_cmd;
    double roll_cmd;
};

// Global synchronization
std::mutex state_mutex;
State current_state;
std::atomic<bool> running{true};

void read_state() {
    while (running) {
        double x, y, z, vx, vy, vz, p, yaw, r, pr, yr, rr, fuel;
        if (std::cin >> x >> y >> z >> vx >> vy >> vz >> p >> yaw >> r >> pr >> yr >> rr >> fuel) {
            std::lock_guard<std::mutex> lock(state_mutex);
            current_state = {x, y, z, vx, vy, vz, p, yaw, r, pr, yr, rr, fuel};
        } else {
            break;
        }
    }
}

int main() {
    // Initial header reading
    double width, depth, height, gravity, max_thrust, fuel_total;
    std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel_total;
    double tx, ty, tz;
    std::cin >> tx >> ty >> tz;
    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    // Start state reader thread
    std::thread reader(read_state);
    reader.detach();

    // Main control loop
    while (running) {
        State local_state;
        {
            std::lock_guard<std::mutex> lock(state_mutex);
            local_state = current_state;
        }

        // Simple control logic (proportional to target)
        double dz = local_state.z - tz;
        double dvz = local_state.vz;
        
        double thrust = 0.0;
        if (dz > 0) {
            if (dvz < -1.0) thrust = 1.0;
            else if (dz < 10.0) thrust = 0.2;
            else thrust = 0.5;
        }

        std::cout << thrust << " 0.0 0.0 0.0" << std::endl;

        std::this_thread::sleep_for(std::chrono::milliseconds(100));
    }

    return 0;
}
