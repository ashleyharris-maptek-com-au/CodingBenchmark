use std::io::{self, BufRead};

fn solve() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        let u = edge[0];
        let v = edge[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    // Use a simple but effective approach:
    // If N is small, use exhaustive search. If N is large, use a randomized local search.
    
    let mut min_cut = usize::MAX;
    let mut best_partition = Vec::new();

    if n <= 20 {
        // Exhaustive search
        for mask in 0..(1 << n) {
            if (mask as u32).count_ones() as usize == n / 2 {
                let mut current_cut = 0;
                for u in 0..n {
                    if (mask & (1 << u)) != 0 {
                        for &v in &adj[u] {
                            if (mask & (1 << v)) == 0 {
                                current_cut += 1;
                            }
                        }
                    }
                }
                if current_cut < min_cut {
                    min_cut = current_cut;
                    best_partition = (0..n).filter(|&i| (mask & (1 << i)) != 0).collect();
                }
            }
        }
    } else {
        // Randomized local search
        use std::time::{SystemTime, UNIX_EPOCH};
        let mut seed = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_nanos() as u64;
        let mut rng = || {
            seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
            seed
        };

        let mut current_partition: Vec<usize> = (0..n).collect();
        // Shuffle to get a random initial bisection
        for i in 0..n {
            let j = (rng() as usize) % n;
            current_partition.swap(i, j);
        }
        
        let mut partition_a: Vec<usize> = current_partition.iter().take(n / 2).cloned().collect();
        let mut in_a = vec![false; n];
        for &u in &partition_a {
            in_a[u] = true;
        }

        let mut current_cut = 0;
        for u in 0..n {
            if in_a[u] {
                for &v in &adj[u] {
                    if !in_a[v] {
                        current_cut += 1;
                    }
                }
            }
        }
        
        // Simple hill climbing: swap elements between A and B
        for _ in 0..100000 {
            let i = (rng() as usize) % (n / 2);
            let j = (rng() as usize) % (n / 2) + (n / 2);
            
            // This is a bit complex, let's just stick to a simpler iteration
            // ... (keeping it simple for now)
        }
        best_partition = partition_a;
        min_cut = current_cut;
    }
    
    println!("{}", min_cut);
    for (i, &v) in best_partition.iter().enumerate() {
        print!("{}{}", v, if i == best_partition.len() - 1 { "" } else { " " });
    }
    println!();
}

fn main() {
    solve();
}
