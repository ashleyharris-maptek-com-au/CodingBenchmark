using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class CuttingStock
{
    struct Cut
    {
        public int Index;
        public int Length;
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine()?.Split();
        if (firstLine == null || firstLine.Length < 2) return;

        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] secondLine = Console.ReadLine()?.Split();
        if (secondLine == null) return;

        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Index = i, Length = int.Parse(secondLine[i]) };
        }

        // First Fit Decreasing (FFD)
        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        List<List<int>> stocks = new List<List<int>>();
        List<int> remainingCapacities = new List<int>();

        foreach (var cut in cuts)
        {
            bool placed = false;
            // For large n, this linear search is O(n^2).
            // Given the time limit and "millions of cuts", we should use a better structure if needed.
            // However, starting with a simple approach for now.
            for (int i = 0; i < remainingCapacities.Count; i++)
            {
                if (remainingCapacities[i] >= cut.Length)
                {
                    remainingCapacities[i] -= cut.Length;
                    stocks[i].Add(cut.Index);
                    placed = true;
                    break;
                }
            }

            if (!placed)
            {
                stocks.Add(new List<int> { cut.Index });
                remainingCapacities.Add(stockLength - cut.Length);
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
