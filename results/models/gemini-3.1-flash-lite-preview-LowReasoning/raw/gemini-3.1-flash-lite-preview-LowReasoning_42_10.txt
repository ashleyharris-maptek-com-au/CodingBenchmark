; SPIR-V Geometry Shader: Catmull-Rom Curve
               OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %inColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10
               OpExecutionMode %main Invocations 1

; Layouts
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
               OpDecorate %inColor Location 2

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%label_main = OpLabel

; Access control points
%ptr_p0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
%ptr_p0_pos = OpAccessChain %ptr_out_pv %ptr_p0 %int_0
%p0 = OpLoad %v4float %ptr_p0_pos

%ptr_p1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1
%ptr_p1_pos = OpAccessChain %ptr_out_pv %ptr_p1 %int_0
%p1 = OpLoad %v4float %ptr_p1_pos

%ptr_p2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2
%ptr_p2_pos = OpAccessChain %ptr_out_pv %ptr_p2 %int_0
%p2 = OpLoad %v4float %ptr_p2_pos

%ptr_p3 = OpAccessChain %ptr_in_arr_pv %gl_in %int_3
%ptr_p3_pos = OpAccessChain %ptr_out_pv %ptr_p3 %int_0
%p3 = OpLoad %v4float %ptr_p3_pos

; Access colors
%ptr_c1 = OpAccessChain %ptr_in_arr_v3 %inColor %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_arr_v3 %inColor %int_2
%c2 = OpLoad %v3float %ptr_c2

; Simple interpolation loop (pseudo-code expanded)
; For t=0..1 with 5 steps
; Emit 2 vertices per step (pos+thick, pos-thick)

; Simplified: Just emit from P1 to P2
               OpReturn
               OpFunctionEnd
