; SPIR-V Assembly Geometry Shader: Tapered Ribbon from Lines

               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor_out
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; --- Decorations ---
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor_out Location 0
               OpDecorate %in_color Location 2

; --- Types ---
      %void = OpTypeVoid
      %func = OpTypeFunction %void
     %float = OpTypeFloat 32
     %v4float = OpTypeVector %float 4
     %v3float = OpTypeVector %float 3
      %uint = OpTypeInt 32 0
      %int = OpTypeInt 32 1
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
   %float_0 = OpConstant %float 0.0
   %float_1 = OpConstant %float 1.0
   %float_04 = OpConstant %float 0.04
  %float_005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_input_arr = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_input_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_output_pv Output

%ptr_outColor = OpTypePointer Output %v3float
%outColor_out = OpVariable %ptr_outColor Output

%ptr_input_v3 = OpTypePointer Input %v3float
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_2
%in_color = OpVariable %ptr_input_arr_v3 Input

; --- Main ---
%main = OpFunction %void None %func
%label = OpLabel

; Read positions
%ptr_p0 = OpAccessChain %ptr_input_arr %gl_in %int_0 %int_0
%p0_raw = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_input_arr %gl_in %int_1 %int_0
%p1_raw = OpLoad %v4float %ptr_p1

%p0 = OpVectorShuffle %v3float %p0_raw %p0_raw 0 1 2
%p1 = OpVectorShuffle %v3float %p1_raw %p1_raw 0 1 2

; Direction
%dir = OpFSub %v3float %p1 %p0
%dir_norm = OpExtInst %v3float %1 Normalize %dir

; Perp (Cross product with Z-axis for 2D expansion)
%z_axis = OpConstantComposite %v3float %float_0 %float_0 1.0
%perp = OpExtInst %v3float %1 Cross %dir_norm %z_axis

; Colors
%ptr_c0 = OpAccessChain %ptr_input_v3 %in_color %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_input_v3 %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1

; --- Vertex 0: p0 + perp * 0.04 ---
%offset0 = OpVectorTimesScalar %v3float %perp %float_04
%v0_pos = OpFAdd %v3float %p0 %offset0
%v0_pos4 = OpCompositeConstruct %v4float %v0_pos %float_1
%ptr_out_pos = OpAccessChain %ptr_output_pv %gl_out %int_0
OpStore %ptr_out_pos %v0_pos4
OpStore %outColor_out %c0
OpEmitVertex

; --- Vertex 1: p0 - perp * 0.04 ---
%v1_pos = OpFSub %v3float %p0 %offset0
%v1_pos4 = OpCompositeConstruct %v4float %v1_pos %float_1
OpStore %ptr_out_pos %v1_pos4
OpStore %outColor_out %c0
OpEmitVertex

; --- Vertex 2: p1 + perp * 0.005 ---
%offset1 = OpVectorTimesScalar %v3float %perp %float_005
%v2_pos = OpFAdd %v3float %p1 %offset1
%v2_pos4 = OpCompositeConstruct %v4float %v2_pos %float_1
OpStore %ptr_out_pos %v2_pos4
OpStore %outColor_out %c1
OpEmitVertex

; --- Vertex 3: p1 - perp * 0.005 ---
%v3_pos = OpFSub %v3float %p1 %offset1
%v3_pos4 = OpCompositeConstruct %v4float %v3_pos %float_1
OpStore %ptr_out_pos %v3_pos4
OpStore %outColor_out %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
