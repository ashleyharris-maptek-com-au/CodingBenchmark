; SPIR-V Fragment Shader
; Output: Red color (vec4(1.0, 0.0, 0.0, 1.0))

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               
               ; Decorations
               OpDecorate %outColor Location 0
               
               ; Types
          %void = OpTypeVoid
        %fn_void = OpTypeFunction %void
         %float = OpTypeFloat 32
        %vec4 = OpTypeVector %float 4
         %ptr_out = OpTypePointer Output %vec4
         
         %outColor = OpVariable %ptr_out Output
         
         ; Constants
         %f1 = OpConstant %float 1.0
         %f0 = OpConstant %float 0.0
         %color = OpConstantComposite %vec4 %f1 %f0 %f0 %f1
         
         ; Main function
         %main = OpFunction %void None %fn_void
         %lbl = OpLabel
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
