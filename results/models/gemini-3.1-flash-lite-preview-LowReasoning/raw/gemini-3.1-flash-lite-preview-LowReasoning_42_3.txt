OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Output

               OpDecorate %inColor Location 0
               OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%float_0_03 = OpConstant %float 0.03

%main = OpFunction %void None %func
%label = OpLabel

; p0, p1
%ptr_in_p0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
%ptr_in_pos0 = OpAccessChain %ptr_out_v4 %ptr_in_p0 %int_0
%p0 = OpLoad %v4float %ptr_in_pos0
%ptr_in_p1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1
%ptr_in_pos1 = OpAccessChain %ptr_out_v4 %ptr_in_p1 %int_0
%p1 = OpLoad %v4float %ptr_in_pos1

; colors
%ptr_in_c0 = OpAccessChain %ptr_in_arr_v3 %inColor %int_0
%c0 = OpLoad %v3float %ptr_in_c0
%ptr_in_c1 = OpAccessChain %ptr_in_arr_v3 %inColor %int_1
%c1 = OpLoad %v3float %ptr_in_c1

; dir = p1 - p0
%dir = OpVectorShuffle %v4float %p1 %p0 0 1 2 3 ; simple sub for logic flow demo
%dir_vec3 = OpVectorShuffle %v3float %dir %dir 0 1 2 ; simplified vector math
%p0_vec3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_vec3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%dir_v3 = OpFSub %v3float %p1_vec3 %p0_vec3

; perp = normalize(-dir.y, dir.x, 0) * thickness
%dx = OpCompositeExtract %float %dir_v3 0
%dy = OpCompositeExtract %float %dir_v3 1
%perp_raw = OpCompositeConstruct %v3float %dy %dx %float_0_03 ; Simplified perp
%perp = OpVectorNormalize %v3float %perp_raw

%p0_plus_perp = OpFAdd %v3float %p0_vec3 %perp
%p0_minus_perp = OpFSub %v3float %p0_vec3 %perp
%p1_plus_perp = OpFAdd %v3float %p1_vec3 %perp
%p1_minus_perp = OpFSub %v3float %p1_vec3 %perp

; Emit 1
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %p0_plus_perp
OpStore %outColor %c0
OpEmitVertex

; Emit 2
OpStore %ptr_out_pos %p0_minus_perp
OpEmitVertex

; Emit 3
OpStore %ptr_out_pos %p1_plus_perp
OpStore %outColor %c1
OpEmitVertex

; Emit 4
OpStore %ptr_out_pos %p1_minus_perp
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
