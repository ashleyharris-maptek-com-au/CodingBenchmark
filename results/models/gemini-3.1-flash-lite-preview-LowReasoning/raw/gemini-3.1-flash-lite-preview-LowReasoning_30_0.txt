import sys
import random

def solve():
    # Fast I/O
    input = sys.stdin.read().split()
    if not input:
        return
    
    num_vars = int(input[0])
    num_clauses = int(input[1])
    
    clauses = []
    idx = 2
    for _ in range(num_clauses):
        clauses.append([int(input[idx]), int(input[idx+1]), int(input[idx+2])])
        idx += 3
        
    # WalkSAT
    # Initial assignment
    assignment = [random.choice([True, False]) for _ in range(num_vars + 1)]
    
    max_flips = 1000000
    max_tries = 10
    
    for _ in range(max_tries):
        assignment = [random.choice([True, False]) for _ in range(num_vars + 1)]
        
        for _ in range(max_flips):
            unsatisfied = []
            for i, clause in enumerate(clauses):
                satisfied = False
                for lit in clause:
                    var = abs(lit)
                    val = assignment[var]
                    if (lit > 0 and val) or (lit < 0 and not val):
                        satisfied = True
                        break
                if not satisfied:
                    unsatisfied.append(i)
            
            if not unsatisfied:
                print("SAT")
                print(*(1 if assignment[i] else 0 for i in range(1, num_vars + 1)))
                return
            
            # Pick a random unsatisfied clause
            clause_idx = random.choice(unsatisfied)
            clause = clauses[clause_idx]
            
            # Flip a random variable in the clause
            lit_to_flip = random.choice(clause)
            var_to_flip = abs(lit_to_flip)
            assignment[var_to_flip] = not assignment[var_to_flip]
            
    # If we reach here, we didn't find a solution (should not happen for guaranteed satisfiable)
    # But for robustness, we could try a different approach or just output something
    pass

if __name__ == "__main__":
    solve()
