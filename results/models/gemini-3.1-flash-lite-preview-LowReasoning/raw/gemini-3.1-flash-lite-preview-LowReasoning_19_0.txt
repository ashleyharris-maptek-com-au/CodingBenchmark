using System;
using System.IO;
using System.Collections.Generic;

namespace MiningSim
{
    class Voxel
    {
        public int X, Y, Z;
        public int Mass;
        public int Value;
    }

    class SimulationState
    {
        public int XCount, YCount, ZCount;
        public int[,] GroundHeight;
        public Dictionary<(int, int, int), Voxel> Voxels = new Dictionary<(int, int, int), Voxel>();

        public SimulationState(int x, int y, int z)
        {
            XCount = x;
            YCount = y;
            ZCount = z;
            GroundHeight = new int[x, y];
            for (int i = 0; i < x; i++)
                for (int j = 0; j < y; j++)
                    GroundHeight[i, j] = z;
        }

        public bool IsExposed(int x, int y, int z)
        {
            return z >= GroundHeight[x, y] - 1;
        }

        public void ProcessVoxel(int x, int y, int z, int mass, int value)
        {
            if (mass > 0)
            {
                Voxels[(x, y, z)] = new Voxel { X = x, Y = y, Z = z, Mass = mass, Value = value };
            }
            else
            {
                if (z < GroundHeight[x, y])
                    GroundHeight[x, y] = z;
            }
        }
    }

    class Miner
    {
        public void Mine(SimulationState state)
        {
            var valuableVoxels = new List<Voxel>();
            foreach (var v in state.Voxels.Values)
            {
                if (v.Value > 0) valuableVoxels.Add(v);
            }
            
            valuableVoxels.Sort((a, b) => b.Value.CompareTo(a.Value));

            int operations = 0;
            var commands = new List<string>();

            foreach (var v in valuableVoxels)
            {
                if (state.IsExposed(v.X, v.Y, v.Z))
                {
                    operations++;
                    commands.Add($"1 {v.X} {v.Y} {v.Z} 0 0 0 0 0 0");
                }
            }

            Console.WriteLine(operations);
            foreach (var cmd in commands)
            {
                Console.WriteLine(cmd);
            }
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            string header = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(header)) return;

            string[] parts = header.Split(' ');
            int xCount = int.Parse(parts[0]);
            int yCount = int.Parse(parts[1]);
            int zCount = int.Parse(parts[2]);
            
            var state = new SimulationState(xCount, yCount, zCount);
            
            using (Stream input = Console.OpenStandardInput())
            {
                for (int x = 0; x < xCount; x++)
                {
                    for (int y = 0; y < yCount; y++)
                    {
                        for (int z = 0; z < zCount; z++)
                        {
                            int d = input.ReadByte();
                            if (d == -1) break;

                            int mass = d % 16;
                            int value = d / 16;
                            
                            state.ProcessVoxel(x, y, z, mass, value);
                        }
                    }
                }
            }

            var miner = new Miner();
            miner.Mine(state);
        }
    }
}
