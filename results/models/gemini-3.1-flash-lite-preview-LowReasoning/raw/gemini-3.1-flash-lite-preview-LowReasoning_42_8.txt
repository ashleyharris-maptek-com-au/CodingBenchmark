; SPIR-V Geometry Shader: Shrink Triangles
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %color_out
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorators
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %color_in Location 0
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Output
               OpDecorate %color_out Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%ptr_int = OpTypePointer Function %int

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_333333 = OpConstant %float 0.333333
%float_0_7 = OpConstant %float 0.7

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%color_out = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Function
%main = OpFunction %void None %func
%label_entry = OpLabel

; Read positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; Centroid C = (p0 + p1 + p2) * 0.333333
%sum_p0_p1 = OpFAdd %v4float %p0 %p1
%sum_p = OpFAdd %v4float %sum_p0_p1 %p2
%centroid = OpVectorTimesScalar %v4float %sum_p %float_0_333333

; Helper to emit vertex
%i = OpVariable %ptr_int Function
OpStore %i %int_0
OpBranch %loop
%loop = OpLabel
%idx = OpLoad %int %i
%cond = OpSLessThan %bool %idx %uint_3
OpLoopMerge %exit %loop None
OpBranchConditional %cond %body %exit
%body = OpLabel

; Calculate vertex: new_pos = C + 0.7 * (pi - C)
%pi_ptr = OpAccessChain %ptr_in_v4 %gl_in %idx %int_0
%pi = OpLoad %v4float %pi_ptr
%diff = OpFSub %v4float %pi %centroid
%scaled = OpVectorTimesScalar %v4float %diff %float_0_7
%new_pos = OpFAdd %v4float %centroid %scaled

; Emit vertex
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %new_pos

%ci_ptr = OpAccessChain %ptr_in_v3 %color_in %idx
%ci = OpLoad %v3float %ci_ptr
OpStore %color_out %ci

OpEmitVertex
%next_i = OpIAdd %int %idx %int_1
OpStore %i %next_i
OpBranch %loop

%exit = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
