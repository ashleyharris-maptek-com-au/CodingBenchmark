; N.y * 0.5 + 0.5

%blend = OpFAdd %float %ny_blend %float_0_5

; Hemisphere blend: mix(ground, sky, blend)

; Final color: amb + 0.3 (diffuse)

%final_rgb = OpFAdd %vec3 %amb %diffuse

%out_v4 = OpCompositeConstruct %vec4 %final_rgb %float_1_0
OpStore %out_color %out_v4
