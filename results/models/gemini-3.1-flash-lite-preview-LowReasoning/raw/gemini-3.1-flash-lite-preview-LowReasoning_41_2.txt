; SPIR-V Fragment Shader
; Simple fragment shader that outputs a solid red color

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpDecorate %outColor Location 0

; Types, Constants
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
%ptr_out_vec = OpTypePointer Output %v4float
  %outColor = OpVariable %ptr_out_vec Output

; Constants
     %float_1 = OpConstant %float 1.0
     %float_0 = OpConstant %float 0.0
   %color_red = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

; Main Function
       %main = OpFunction %void None %func
         %lb = OpLabel
               OpStore %outColor %color_red
               OpReturn
               OpFunctionEnd
