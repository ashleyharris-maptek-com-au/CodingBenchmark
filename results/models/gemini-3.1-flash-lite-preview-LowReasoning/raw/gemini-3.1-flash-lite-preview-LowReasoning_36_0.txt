#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>

// Use a more efficient MIS algorithm: branching with pruning.
// For larger graphs, we need better pruning and representation.
// Using bitsets to represent the graph and neighborhoods.

int n, m;
std::vector<std::bitset<1000>> adj; // Assume n <= 1000 for now.
std::vector<int> best_set;

void solve(std::bitset<1000> candidates, std::vector<int>& current_set) {
    if (candidates.none()) {
        if (current_set.size() > best_set.size()) {
            best_set = current_set;
        }
        return;
    }

    if (current_set.size() + candidates.count() <= best_set.size()) {
        return;
    }

    // Heuristic: pick vertex with max degree in the candidate subgraph?
    // For now, pick the first vertex.
    int v = -1;
    for (int i = 0; i < n; ++i) {
        if (candidates.test(i)) {
            v = i;
            break;
        }
    }

    if (v == -1) return;

    // Option 1: Include v
    std::vector<int> next_set = current_set;
    next_set.push_back(v);
    std::bitset<1000> next_candidates = candidates;
    next_candidates.reset(v);
    next_candidates &= ~adj[v];
    solve(next_candidates, next_set);

    // Option 2: Exclude v
    candidates.reset(v);
    solve(candidates, current_set);
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    if (!(std::cin >> n >> m)) return 0;

    adj.resize(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].set(v);
            adj[v].set(u);
        }
    }

    std::bitset<1000> initial_candidates;
    for (int i = 0; i < n; ++i) initial_candidates.set(i);

    std::vector<int> current_set;
    solve(initial_candidates, current_set);

    std::cout << best_set.size() << std::endl;
    for (size_t i = 0; i < best_set.size(); ++i) {
        std::cout << best_set[i] << (i == best_set.size() - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
