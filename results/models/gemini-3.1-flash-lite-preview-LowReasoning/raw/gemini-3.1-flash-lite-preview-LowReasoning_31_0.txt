use std::io::{self, BufRead, BufWriter, Write};

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    let mut first_line = String::new();
    if reader.read_line(&mut first_line)? == 0 {
        return Ok(());
    }
    
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();

    if parts.len() < 2 {
        return Ok(());
    }

    let _num_numbers = parts[0];
    let target = parts[1];

    let mut second_line = String::new();
    reader.read_line(&mut second_line)?;
    let nums: Vec<usize> = second_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();

    // parent[sum] = (index_in_nums, previous_sum)
    // To save memory, we can use a simpler structure if needed,
    // but this is the most direct way to reconstruct the indices.
    let mut parent = vec![None; target + 1];
    parent[0] = Some((0, 0)); // (index, prev_sum)

    for (idx, &num) in nums.iter().enumerate() {
        if num == 0 || num > target { continue; }
        for j in (num..=target).rev() {
            if parent[j].is_none() && parent[j - num].is_some() {
                parent[j] = Some((idx, j - num));
            }
        }
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    if let Some(mut curr) = parent[target] {
        writeln!(writer, "YES")?;
        let mut subset_indices = Vec::new();
        while curr.1 != 0 || (curr.0 != 0 || target == 0) {
            let (idx, prev_sum) = curr;
            subset_indices.push(idx);
            if prev_sum == 0 { break; }
            if let Some(next) = parent[prev_sum] {
                curr = next;
            } else {
                break;
            }
        }
        
        write!(writer, "[")?;
        for (i, &idx) in subset_indices.iter().enumerate() {
            write!(writer, "{}", idx)?;
            if i < subset_indices.len() - 1 {
                write!(writer, " ")?;
            }
        }
        writeln!(writer, "]")?;
    } else {
        writeln!(writer, "NO")?;
    }

    writer.flush()?;
    Ok(())
}
