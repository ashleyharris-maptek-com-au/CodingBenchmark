use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

struct Polygon {
    vertices: Vec<Point>,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Line 1: SV NP
    if let Some(Ok(line)) = lines.next() {
        let parts: Vec<usize> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        let sv = parts[0];
        let np = parts[1];

        // Stock polygon
        let mut stock_vertices = Vec::with_capacity(sv);
        for _ in 0..sv {
            if let Some(Ok(line)) = lines.next() {
                let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
                stock_vertices.push(Point { x: coords[0], y: coords[1] });
            }
        }

        // Read pieces
        let mut pieces = Vec::with_capacity(np);
        for _ in 0..np {
            if let Some(Ok(line)) = lines.next() {
                let pv: usize = line.trim().parse().unwrap();
                let mut vertices = Vec::with_capacity(pv);
                for _ in 0..pv {
                    if let Some(Ok(line)) = lines.next() {
                        let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
                        vertices.push(Point { x: coords[0], y: coords[1] });
                    }
                }
                pieces.push(Polygon { vertices });
            }
        }

        // Simple Greedy Packing:
        // For simplicity and to ensure validity, place each piece in a new stock (not optimal but valid).
        // A more advanced algorithm would pack multiple pieces into one stock.
        
        println!("{}", np); // Number of stocks used
        for i in 0..np {
            // Using index 'i' as stock index. 
            // x: 0, y: 0, rotation: 0
            println!("{} 0.0 0.0 0.0", i);
        }
    }
}
