using System;
using System.Collections.Generic;
using System.Linq;

namespace OrbitalTSP
{
    class Program
    {
        static double MU = 398600.4418;

        struct State
        {
            public double X, Y, Z, VX, VY, VZ;
        }

        static void Main(string[] args)
        {
            string[] line1 = Console.ReadLine().Split();
            int N = int.Parse(line1[0]);
            double mu = double.Parse(line1[1]);

            string[] startLine = Console.ReadLine().Split();
            State start = new State { X = double.Parse(startLine[0]), Y = double.Parse(startLine[1]), Z = double.Parse(startLine[2]), VX = double.Parse(startLine[3]), VY = double.Parse(startLine[4]), VZ = double.Parse(startLine[5]) };

            List<State> stations = new List<State>();
            for (int i = 0; i < N; i++)
            {
                string[] sLine = Console.ReadLine().Split();
                stations.Add(new State { X = double.Parse(sLine[0]), Y = double.Parse(sLine[1]), Z = double.Parse(sLine[2]), VX = double.Parse(sLine[3]), VY = double.Parse(sLine[4]), VZ = double.Parse(sLine[5]) });
            }

            // Simplistic TSP: Nearest Neighbor based on Euclidean distance of state vectors
            List<int> order = new List<int>();
            bool[] visited = new bool[N];
            State current = start;

            for (int i = 0; i < N; i++)
            {
                int bestIdx = -1;
                double minDistance = double.MaxValue;

                for (int j = 0; j < N; j++)
                {
                    if (!visited[j])
                    {
                        double dist = Distance(current, stations[j]);
                        if (dist < minDistance)
                        {
                            minDistance = dist;
                            bestIdx = j;
                        }
                    }
                }

                visited[bestIdx] = true;
                order.Add(bestIdx);
                current = stations[bestIdx];
            }

            Console.WriteLine(string.Join(" ", order));
        }

        static double Distance(State a, State b)
        {
            return Math.Sqrt(Math.Pow(a.X - b.X, 2) + Math.Pow(a.Y - b.Y, 2) + Math.Pow(a.Z - b.Z, 2));
        }
    }
}
