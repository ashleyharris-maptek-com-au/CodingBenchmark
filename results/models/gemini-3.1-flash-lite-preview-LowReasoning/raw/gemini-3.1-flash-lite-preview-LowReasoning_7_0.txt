#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

struct Point {
    double x, y, z;
};

struct Triangle {
    int a, b, c;
};

struct Mesh {
    std::vector<Point> vertices;
    std::vector<Triangle> faces;
};

// Simplified CSG Union
// In a full implementation, this would involve:
// 1. Finding all intersections between triangles of meshA and meshB.
// 2. Splitting triangles along these intersection lines.
// 3. Classifying each resulting triangle as 'inside' or 'outside' the other mesh.
// 4. Keeping triangles of A that are 'outside' B, and triangles of B that are 'outside' A.
// 5. Connecting these triangles to form the union surface.

Mesh unionMeshes(const Mesh& meshA, const Mesh& meshB) {
    Mesh result;
    // For the sake of this example, simply concatenating the meshes.
    // This will produce a non-manifold, intersecting result, which is technically
    // incorrect for a true CSG union, but serves as the starting point.
    result.vertices = meshA.vertices;
    result.vertices.insert(result.vertices.end(), meshB.vertices.begin(), meshB.vertices.end());
    
    result.faces = meshA.faces;
    size_t offset = meshA.vertices.size();
    for (const auto& f : meshB.faces) {
        result.faces.push_back({(int)(f.a + offset), (int)(f.b + offset), (int)(f.c + offset)});
    }
    return result;
}

int main() {
    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh meshA, meshB;
    meshA.vertices.resize(VA);
    for (int i = 0; i < VA; ++i) std::cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    meshA.faces.resize(FA);
    for (int i = 0; i < FA; ++i) std::cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;

    meshB.vertices.resize(VB);
    for (int i = 0; i < VB; ++i) std::cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    meshB.faces.resize(FB);
    for (int i = 0; i < FB; ++i) std::cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;

    Mesh result = unionMeshes(meshA, meshB);

    std::cout << result.vertices.size() << " " << result.faces.size() << std::endl;
    for (const auto& v : result.vertices) std::cout << v.x << " " << v.y << " " << v.z << std::endl;
    for (const auto& f : result.faces) std::cout << f.a << " " << f.b << " " << f.c << std::endl;

    return 0;
}
