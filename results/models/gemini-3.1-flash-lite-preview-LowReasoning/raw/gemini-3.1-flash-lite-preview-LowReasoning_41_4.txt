; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %out_color %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_uniform_float = OpTypePointer Uniform %float
%func = OpTypeFunction %void

; Variables
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %func
%label = OpLabel

; Load normal and worldPos
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos

; Normalize N
%n_norm = OpExtInst %vec3 %1 GLSL.std450 Normalize %normal

; Get lightPos
%lightPosPtr = OpAccessChain %ptr_uniform_vec4 %ubo OpConstant 3
%lightPos = OpLoad %vec4 %lightPosPtr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; L = lightPos - worldPos
%l_vec = OpFSub %vec3 %lightPos_xyz %worldPos
%l_norm = OpExtInst %vec3 %1 GLSL.std450 Normalize %l_vec

; Dot(N, L)
%dot_val = OpDot %float %n_norm %l_norm

; Intensity = max(dot, 0.0)
%zero = OpConstant %float 0.0
%intensity = OpExtInst %float %1 GLSL.std450 FMax %dot_val %zero

; Color = vec4(intensity, intensity, intensity, 1.0)
%one = OpConstant %float 1.0
%final_color = OpCompositeConstruct %vec4 %intensity %intensity %intensity %one

OpStore %out_color %final_color
OpReturn
OpFunctionEnd

%1 = OpExtInstImport "GLSL.std450"
