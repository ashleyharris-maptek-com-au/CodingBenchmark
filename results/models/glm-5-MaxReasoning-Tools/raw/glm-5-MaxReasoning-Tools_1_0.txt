Here's the complete, optimized C++ solution for the TSP problem that scales from 10 to 10,000,000+ cities:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>

using namespace std;

// Fast squared distance for comparisons
inline double sqDist(double x1, double y1, double x2, double y2) {
    double dx = x1 - x2;
    double dy = y1 - y2;
    return dx * dx + dy * dy;
}

// Hilbert curve rotation
void hilbertRot(int n, int* x, int* y, int rx, int ry) {
    if (ry == 0) {
        if (rx == 1) { *x = n - 1 - *x; *y = n - 1 - *y; }
        int t = *x; *x = *y; *y = t;
    }
}

// Convert (x,y) to Hilbert distance
int64_t xy2d(int n, int x, int y) {
    int rx, ry, s;
    int64_t d = 0;
    for (s = n / 2; s > 0; s /= 2) {
        rx = (x & s) ? 1 : 0;
        ry = (y & s) ? 1 : 0;
        d += (int64_t)s * s * ((3 * rx) ^ ry);
        hilbertRot(s, &x, &y, rx, ry);
    }
    return d;
}

// Hilbert curve ordering - O(n log n) for massive instances
vector<int> hilbertSolve(int n, const vector<double>& px, const vector<double>& py) {
    if (n <= 1) return vector<int>(1, 0);
    
    double minX = px[0], maxX = px[0], minY = py[0], maxY = py[0];
    for (int i = 1; i < n; i++) {
        if (px[i] < minX) minX = px[i]; if (px[i] > maxX) maxX = px[i];
        if (py[i] < minY) minY = py[i]; if (py[i] > maxY) maxY = py[i];
    }
    
    double scale = max(maxX - minX, maxY - minY);
    if (scale < 1e-9) scale = 1.0;
    
    const int BITS = 20, HSIZE = 1 << BITS;
    double invScale = (HSIZE - 1) / scale;
    
    vector<pair<int64_t, int>> order(n);
    for (int i = 0; i < n; i++) {
        int hx = max(0, min(HSIZE - 1, (int)((px[i] - minX) * invScale + 0.5)));
        int hy = max(0, min(HSIZE - 1, (int)((py[i] - minY) * invScale + 0.5)));
        order[i] = make_pair(xy2d(HSIZE, hx, hy), i);
    }
    
    sort(order.begin(), order.end());
    
    int startIdx = 0;
    for (int i = 0; i < n; i++) if (order[i].second == 0) { startIdx = i; break; }
    
    vector<int> result(n);
    for (int i = 0; i < n; i++) result[i] = order[(startIdx + i) % n].second;
    return result;
}

// Simple nearest neighbor - O(n^2)
vector<int> nearestNeighbor(int n, const vector<double>& px, const vector<double>& py) {
    if (n <= 1) return vector<int>(1, 0);
    
    vector<int> result(1, 0);
    vector<char> used(n, 0);
    used[0] = 1;
    int cur = 0;
    
    for (int step = 1; step < n; step++) {
        double bestD = 1e30; int bestI = -1;
        for (int i = 0; i < n; i++) {
            if (!used[i]) {
                double d = sqDist(px[cur], py[cur], px[i], py[i]);
                if (d < bestD) { bestD = d; bestI = i; }
            }
        }
        result.push_back(bestI);
        used[bestI] = 1;
        cur = bestI;
    }
    return result;
}

// Grid-based nearest neighbor for medium-large
vector<int> gridNN(int n, const vector<double>& px, const vector<double>& py) {
    if (n <= 1) return vector<int>(1, 0);
    
    double minX = px[0], maxX = px[0], minY = py[0], maxY = py[0];
    for (int i = 1; i < n; i++) {
        minX = min(minX, px[i]); maxX = max(maxX, px[i]);
        minY = min(minY, py[i]); maxY = max(maxY, py[i]);
    }
    
    double rangeX = maxX - minX + 1e-9, rangeY = maxY - minY + 1e-9;
    int gridSize = max(1, min(300, (int)sqrt((double)n)));
    double cellW = rangeX / gridSize + 1e-9, cellH = rangeY / gridSize + 1e-9;
    
    vector<vector<int>> buckets(gridSize * gridSize);
    for (int i = 0; i < n; i++) {
        int cx = max(0, min(gridSize - 1, (int)((px[i] - minX) / cellW)));
        int cy = max(0, min(gridSize - 1, (int)((py[i] - minY) / cellH)));
        buckets[cy * gridSize + cx].push_back(i);
    }
    
    vector<int> result(1, 0);
    vector<char> used(n, 0);
    used[0] = 1;
    int cur = 0;
    
    for (int step = 1; step < n; step++) {
        int curCx = max(0, min(gridSize - 1, (int)((px[cur] - minX) / cellW)));
        int curCy = max(0, min(gridSize - 1, (int)((py[cur] - minY) / cellH)));
        double bestD = 1e30; int bestI = -1;
        
        for (int ring = 0; ring <= gridSize && bestI < 0; ring++) {
            for (int dx = -ring; dx <= ring; dx++) {
                for (int dy = -ring; dy <= ring; dy++) {
                    if (abs(dx) != ring && abs(dy) != ring) continue;
                    int nx = curCx + dx, ny = curCy + dy;
                    if (nx < 0 || nx >= gridSize || ny < 0 || ny >= gridSize) continue;
                    for (size_t k = 0; k < buckets[ny * gridSize + nx].size(); k++) {
                        int i = buckets[ny * gridSize + nx][k];
                        if (!used[i]) {
                            double d = sqDist(px[cur], py[cur], px[i], py[i]);
                            if (d < bestD) { bestD = d; bestI = i; }
                        }
                    }
                }
            }
        }
        
        if (bestI < 0) { for (int i = 0; i < n; i++) if (!used[i]) { bestI = i; break; } }
        result.push_back(bestI);
        used[bestI] = 1;
        cur = bestI;
    }
    return result;
}

// 2-opt improvement
void twoOpt(vector<int>& tour, const vector<double>& px, const vector<double>& py, int maxIter) {
    int n = (int)tour.size();
    if (n < 4) return;
    
    bool improved = true;
    for (int iter = 0; improved && iter < maxIter; iter++) {
        improved = false;
        for (int i = 0; i < n - 1 && !improved; i++) {
            for (int j = i + 2; j < n && !improved; j++) {
                int a = tour[i], b = tour[i + 1], c = tour[j], d = tour[(j + 1) % n];
                double before = sqDist(px[a],py[a],px[b],py[b]) + sqDist(px[c],py[c],px[d],py[d]);
                double after = sqDist(px[a],py[a],px[c],py[c]) + sqDist(px[b],py[b],px[d],py[d]);
                if (after < before - 1e-9) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Held-Karp exact for tiny n
vector<int> heldKarp(int n, const vector<double>& px, const vector<double>& py) {
    if (n == 1) return vector<int>(1, 0);
    
    int fullMask = (1 << n) - 1;
    vector<vector<double>> dp(1 << n, vector<double>(n, 1e18));
    vector<vector<int>> parent(1 << n, vector<int>(n, -1));
    dp[1][0] = 0;
    
    for (int mask = 1; mask <= fullMask; mask++) {
        if (!(mask & 1)) continue;
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i)) || dp[mask][i] > 1e17) continue;
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;
                double cost = dp[mask][i] + sqrt(sqDist(px[i],py[i],px[j],py[j]));
                if (cost < dp[mask | (1 << j)][j]) {
                    dp[mask | (1 << j)][j] = cost;
                    parent[mask | (1 << j)][j] = i;
                }
            }
        }
    }
    
    int last = 1;
    for (int i = 2; i < n; i++) if (dp[fullMask][i] < dp[fullMask][last]) last = i;
    
    vector<int> result;
    int mask = fullMask;
    while (last != -1) {
        result.push_back(last);
        int p = parent[mask][last];
        mask ^= (1 << last);
        last = p;
    }
    reverse(result.begin(), result.end());
    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    if (n <= 0) return 0;
    
    vector<double> px(n), py(n);
    for (int i = 0; i < n; i++) cin >> px[i] >> py[i];
    
    vector<int> result;
    if (n <= 11) result = heldKarp(n, px, py);
    else if (n <= 80) { result = nearestNeighbor(n, px, py); twoOpt(result, px, py, 300); }
    else if (n <= 2000) { result = nearestNeighbor(n, px, py); twoOpt(result, px, py, 50); }
    else if (n <= 50000) result = gridNN(n, px, py);
    else result = hilbertSolve(n, px, py);
    
    cout << result[0];
    for (size_t i = 1; i < result.size(); i++) cout << " " << result[i];
    cout << "\n";
    return 0;
}
```

## Algorithm Summary

| Size Range | Algorithm | Time Complexity | Quality |
|------------|-----------|-----------------|---------|
| n ≤ 11 | Held-Karp exact DP | O(2ⁿ·n²) | Optimal |
| 12-80 | NN + 2-opt (300 iter) | O(n²) | Very good |
| 81-2000 | NN + 2-opt (50 iter) | O(n²) | Good |
| 2001-50000 | Grid-accelerated NN | O(n·√n) avg | Reasonable |
| n > 50000 | Hilbert curve ordering | O(n log n) | Locality-preserving |

## Key Features

- **Exact solution** for tiny instances (Held-Karp DP)
- **Hilbert curve ordering** for massive instances - provides excellent locality preservation with O(n log n) sorting
- **Grid-based spatial indexing** for medium-large instances to accelerate nearest neighbor search
- **2-opt local search** improves solution quality for smaller instances
- **Squared distances** for comparisons (avoids sqrt overhead)
- **Fast I/O** with `sync_with_stdio(false)`
