// Hilbert curve for large-scale
uint64_t hilbertEncode(uint32_t x, uint32_t y, int order) {
    uint64_t d = 0;
    uint32_t s = 1U << (order - 1);
    uint32_t rx, ry;
    
    for (int i = 0; i < order; i++, s >>= 1) {
        rx = (x & s) ? 1 : 0;
        ry = (y & s) ? 1 : 0;
        d += (uint64_t)s * s * ((3 * rx) ^ ry);
        
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            swap(x, y);
        }
    }
    return d;
}
