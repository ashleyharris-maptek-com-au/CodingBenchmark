#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * This program solves the Maximum Weight Independent Set (MWIS) problem.
 * Given a graph with weights on vertices, find an independent set (no two 
 * vertices are connected by an edge) such that the sum of weights is maximized.
 * 
 * We use a recursive backtracking approach with pruning (Branch and Bound).
 */

struct Solver {
    int n;
    vector<long long> weights;
    vector<vector<int>> adj;
    
    long long max_weight = 0;
    vector<int> best_assignment;
    vector<int> current_assignment;
    long long current_weight = 0;

    Solver(int n, const vector<long long>& w, const vector<vector<int>>& a) 
        : n(n), weights(w), adj(a) {
        best_assignment.assign(n, 0);
        current_assignment.assign(n, 0);
    }

    void solve(int idx, long long remaining_possible_weight) {
        if (idx == n) {
            if (current_weight > max_weight) {
                max_weight = current_weight;
                best_assignment = current_assignment;
            }
            return;
        }

        // Pruning: if even by picking all remaining vertices we can't beat max_weight, stop.
        if (current_weight + remaining_possible_weight <= max_weight) {
            return;
        }

        // Branching strategy: Try including vertex idx if possible, then try not including it.
        // Usually, trying to include first helps find a better lower bound faster for pruning.

        // Try including vertex idx if possible
        bool can_include = true;
        for (int neighbor : adj[idx]) {
            if (neighbor < idx && current_assignment[neighbor] == 1) {
                can_include = false;
                break;
            }
        }

        if (can_include) {
            current_assignment[idx] = 1;
            current_weight += weights[idx];
            solve(idx + 1, remaining_possible_weight - weights[idx]);
            current_weight -= weights[idx];
            current_assignment[idx] = 0;
        }

        // Try NOT including vertex idx
        solve(idx + 1, remaining_possible_weight - weights[idx]);
    }

    void run() {
        long long total_weight = 0;
        for (long long w : weights) total_weight += w;
        solve(0, total_weight);
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) {
        cin >> c[i];
    }

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    Solver solver(n, c, adj);
    solver.run();

    cout << solver.max_weight << "\n";
    for (int i = 0; i < n; ++i) {
        cout << solver.best_assignment[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
