const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    if (!input) return;
    const data = JSON.parse(input);

    const { grid, robots, jobs } = data;
    const rows = grid.length;
    const cols = grid[0].length;

    // Helper to run BFS from a starting point (x, y) and return distances to all other points
    function bfs(startX, startY) {
        const distances = Array.from({ length: rows }, () => new Int32Array(cols).fill(-1));
        const queue = [startX, startY];
        distances[startY][startX] = 0;
        let head = 0;

        const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];

        while (head < queue.length) {
            const cx = queue[head++];
            const cy = queue[head++];
            const d = distances[cy][cx];

            for (let i = 0; i < 4; i++) {
                const nx = cx + dirs[i][0];
                const ny = cy + dirs[i][1];

                if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.' && distances[ny][nx] === -1) {
                    distances[ny][nx] = d + 1;
                    queue.push(nx, ny);
                }
            }
        }
        return distances;
    }

    // Sort jobs by priority (descending) then by ID (ascending)
    const sortedJobs = [...jobs].sort((a, b) => {
        if (b.priority !== a.priority) return b.priority - a.priority;
        if (a.id < b.id) return -1;
        if (a.id > b.id) return 1;
        return 0;
    });

    const assignments = Array.from({ length: robots.length }, () => []);
    const robotCurrentPositions = robots.map(r => ({ x: r.x, y: r.y }));

    // For each job, find the best robot
    // The "deterministic greedy reference" usually means:
    // for each job (in priority order): assign to the robot that can reach it fastest from its CURRENT position.
    for (const job of sortedJobs) {
        let bestRobotIdx = -1;
        let minDistance = Infinity;

        // Since we need distances from robot's CURRENT position to the job,
        // and robot positions change, we might need to run BFS from each robot position
        // OR from the job position to all points.
        // Running BFS from the job position ONCE is more efficient.
        const jobDistances = bfs(job.x, job.y);

        for (let i = 0; i < robots.length; i++) {
            const pos = robotCurrentPositions[i];
            const d = jobDistances[pos.y][pos.x];

            if (d !== -1) {
                if (d < minDistance) {
                    minDistance = d;
                    bestRobotIdx = i;
                } else if (d === minDistance) {
                    // Tie-break: robot with lower index in input robots array (which is usually the order of robot IDs if they are 0-indexed)
                    // The prompt says "array index is the robot id". This implies robots[i].id === i.
                    // Let's tie-break by index `i`.
                    if (bestRobotIdx === -1 || i < bestRobotIdx) {
                        bestRobotIdx = i;
                    }
                }
            }
        }

        if (bestRobotIdx !== -1) {
            assignments[bestRobotIdx].push(job.id);
            robotCurrentPositions[bestRobotIdx] = { x: job.x, y: job.y };
        }
    }

    process.stdout.write(JSON.stringify({ assignments }));
}

solve();
