#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <set>
#include <queue>

using namespace std;

struct Graph {
    int num_vertices;
    int num_edges;
    int num_colors;
    vector<vector<int>> adj;

    Graph(int n, int m, int k) : num_vertices(n), num_edges(m), num_colors(k), adj(n) {}

    void add_edge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
};

struct NodeInfo {
    int u;
    int sat;
    int deg;

    bool operator<(const NodeInfo& other) const {
        if (sat != other.sat) return sat < other.sat;
        return deg < other.deg;
    }
};

bool solve_dsatur_optimized(const Graph& g, vector<int>& colors) {
    int n = g.num_vertices;
    colors.assign(n, -1);
    
    vector<int> degree(n);
    vector<int> saturation(n, 0);
    vector<set<int>> neighbor_colors(n);
    priority_queue<NodeInfo> pq;

    for (int i = 0; i < n; ++i) {
        degree[i] = (int)g.adj[i].size();
        pq.push({i, 0, degree[i]});
    }

    vector<bool> used(g.num_colors);
    int colored_count = 0;
    while (!pq.empty()) {
        NodeInfo top = pq.top();
        pq.pop();

        int u = top.u;
        if (colors[u] != -1) continue;

        if (top.sat < saturation[u]) continue;

        fill(used.begin(), used.end(), false);
        for (int v : g.adj[u]) {
            if (colors[v] != -1) {
                used[colors[v]] = true;
            }
        }

        int color = 0;
        while (color < g.num_colors && used[color]) color++;

        if (color == g.num_colors) return false;

        colors[u] = color;
        colored_count++;

        for (int v : g.adj[u]) {
            if (colors[v] == -1) {
                if (neighbor_colors[v].find(color) == neighbor_colors[v].end()) {
                    neighbor_colors[v].insert(color);
                    saturation[v] = (int)neighbor_colors[v].size();
                    pq.push({v, saturation[v], degree[v]});
                }
            }
        }
    }

    return colored_count == n;
}

bool solve_welsh_powell(const Graph& g, vector<int>& colors) {
    int n = g.num_vertices;
    colors.assign(n, -1);
    vector<int> nodes(n);
    for (int i = 0; i < n; ++i) nodes[i] = i;

    sort(nodes.begin(), nodes.end(), [&](int a, int b) {
        return g.adj[a].size() > g.adj[b].size();
    });

    vector<bool> used(g.num_colors);
    for (int u : nodes) {
        fill(used.begin(), used.end(), false);
        for (int v : g.adj[u]) {
            if (colors[v] != -1) {
                used[colors[v]] = true;
            }
        }

        int c = 0;
        while (c < g.num_colors && used[c]) c++;
        if (c == g.num_colors) return false;
        colors[u] = c;
    }
    return true;
}

bool solve_iterative_greedy(const Graph& g, vector<int>& colors, int max_time_ms) {
    auto start_time = chrono::steady_clock::now();
    int n = g.num_vertices;
    vector<int> nodes(n);
    for (int i = 0; i < n; ++i) nodes[i] = i;

    mt19937 rng(1337);

    if (solve_welsh_powell(g, colors)) return true;

    vector<bool> used(g.num_colors);
    while (true) {
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(chrono::steady_clock::now() - start_time).count();
        if (elapsed > max_time_ms) break;

        shuffle(nodes.begin(), nodes.end(), rng);
        vector<int> current_colors(n, -1);
        bool possible = true;

        for (int u : nodes) {
            fill(used.begin(), used.end(), false);
            for (int v : g.adj[u]) {
                if (current_colors[v] != -1) {
                    used[current_colors[v]] = true;
                }
            }
            int c = 0;
            while (c < g.num_colors && used[c]) c++;
            if (c == g.num_colors) {
                possible = false;
                break;
            }
            current_colors[u] = c;
        }

        if (possible) {
            colors = current_colors;
            return true;
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    Graph g(n, m, k);
    for (int i = 0; i < m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            g.add_edge(u, v);
        }
    }

    vector<int> colors(n, -1);
    bool success = false;
    const int TIME_LIMIT_MS = 270000;

    if (n < 20000) {
        success = solve_dsatur_optimized(g, colors);
        if (!success) {
            success = solve_iterative_greedy(g, colors, 30000);
        }
    } else {
        success = solve_iterative_greedy(g, colors, TIME_LIMIT_MS);
    }

    if (!success) {
        vector<int> nodes(n);
        for (int i = 0; i < n; ++i) nodes[i] = i;
        sort(nodes.begin(), nodes.end(), [&](int a, int b) {
            return g.adj[a].size() > g.adj[b].size();
        });
        vector<bool> used(k);
        for (int u : nodes) {
            if (colors[u] != -1) continue;
            fill(used.begin(), used.end(), false);
            for (int v : g.adj[u]) {
                if (colors[v] != -1) used[colors[v]] = true;
            }
            int c = 0;
            while (c < k && used[c]) c++;
            colors[u] = (c < k) ? c : 0;
        }
    }

    for (int i = 0; i < n; ++i) {
        cout << (colors[i] == -1 ? 0 : colors[i]) << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
