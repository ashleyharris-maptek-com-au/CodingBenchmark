#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <cmath>
#include <algorithm>
#include <memory>

using namespace std;

// VectorND represents a point in N-dimensional space
struct VectorND {
    vector<int> coords;

    bool operator==(const VectorND& other) const {
        return coords == other.coords;
    }

    bool operator!=(const VectorND& other) const {
        return !(*this == other);
    }
};

// Custom hash for VectorND to use in unordered_set and unordered_map
struct VectorNDHash {
    size_t operator()(const VectorND& v) const {
        size_t h = 0;
        for (int x : v.coords) {
            h ^= hash<int>{}(x) + 0x9e3779b9 + (h << 6) + (h >> 2);
        }
        return h;
    }
};

// Manhattan distance for grid-based pathfinding
int manhattan_distance(const VectorND& a, const VectorND& b) {
    int dist = 0;
    for (size_t i = 0; i < a.coords.size(); ++i) {
        dist += abs(a.coords[i] - b.coords[i]);
    }
    return dist;
}

// GameState holds the entire environment and snake data
struct GameState {
    int D;
    int max_turns;
    vector<int> bounds;
    deque<VectorND> snake;
    unordered_set<VectorND, VectorNDHash> obstacles;
    vector<VectorND> food;

    // Checks if a position is within bounds and doesn't collide with obstacles or snake body
    bool is_valid(const VectorND& pos) const {
        for (int i = 0; i < D; ++i) {
            if (pos.coords[i] < 0 || pos.coords[i] >= bounds[i]) return false;
        }
        if (obstacles.count(pos)) return false;
        for (const auto& segment : snake) {
            if (pos == segment) return false;
        }
        return true;
    }
};

// Node structure for A* pathfinding
struct AStarNode {
    VectorND pos;
    int g;
    int h;
    int axis;
    int dir;
    shared_ptr<AStarNode> parent;

    int f() const { return g + h; }
};

// Comparison for the priority queue in A*
struct AStarNodeCmp {
    bool operator()(const shared_ptr<AStarNode>& a, const shared_ptr<AStarNode>& b) {
        return a->f() > b->f();
    }
};

// Finds the next move for the snake
pair<int, int> find_move(GameState& state) {
    if (state.food.empty()) return {-1, 0};

    // Find nearest food item to target
    VectorND target_food = state.food[0];
    int min_dist = manhattan_distance(state.snake.front(), target_food);
    for (size_t i = 1; i < state.food.size(); ++i) {
        int d = manhattan_distance(state.snake.front(), state.food[i]);
        if (d < min_dist) {
            min_dist = d;
            target_food = state.food[i];
        }
    }

    // A* Pathfinding to the target food
    priority_queue<shared_ptr<AStarNode>, vector<shared_ptr<AStarNode>>, AStarNodeCmp> open_list;
    unordered_map<VectorND, int, VectorNDHash> closed_list;

    auto start_node = make_shared<AStarNode>();
    start_node->pos = state.snake.front();
    start_node->g = 0;
    start_node->h = manhattan_distance(start_node->pos, target_food);
    start_node->axis = -1;
    start_node->dir = 0;
    start_node->parent = nullptr;

    open_list.push(start_node);

    while (!open_list.empty()) {
        auto current = open_list.top();
        open_list.pop();

        if (current->pos == target_food) {
            // Found a path, trace back to the very first move
            auto p = current;
            while (p->parent && p->parent->parent) {
                p = p->parent;
            }
            return {p->axis, p->dir};
        }

        if (closed_list.count(current->pos) && closed_list[current->pos] <= current->g) {
            continue;
        }
        closed_list[current->pos] = current->g;

        // Depth limit to maintain responsiveness in extreme dimensionalities
        if (current->g > 150) continue; 

        for (int i = 0; i < state.D; ++i) {
            for (int d : {-1, 1}) {
                VectorND next_pos = current->pos;
                next_pos.coords[i] += d;

                if (state.is_valid(next_pos)) {
                    auto next_node = make_shared<AStarNode>();
                    next_node->pos = next_pos;
                    next_node->g = current->g + 1;
                    next_node->h = manhattan_distance(next_pos, target_food);
                    next_node->axis = i;
                    next_node->dir = d;
                    next_node->parent = current;

                    if (!closed_list.count(next_pos) || closed_list[next_pos] > next_node->g) {
                        open_list.push(next_node);
                    }
                }
            }
        }
    }

    // Survival logic: If A* fails, take any valid move that keeps the snake alive
    for (int i = 0; i < state.D; ++i) {
        for (int d : {-1, 1}) {
            VectorND next_pos = state.snake.front();
            next_pos.coords[i] += d;
            if (state.is_valid(next_pos)) {
                return {i, d};
            }
        }
    }

    return {-1, 0}; // No valid moves left
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    GameState state;
    if (!(cin >> state.D >> state.max_turns)) return 0;

    state.bounds.resize(state.D);
    for (int i = 0; i < state.D; ++i) cin >> state.bounds[i];

    int snake_length;
    cin >> snake_length;
    VectorND head;
    head.coords.resize(state.D);
    for (int i = 0; i < state.D; ++i) cin >> head.coords[i];
    // Initialize snake body as a stack of segments at the start position
    for (int i = 0; i < snake_length; ++i) state.snake.push_back(head);

    int F;
    cin >> F;
    state.food.resize(F);
    for (int i = 0; i < F; ++i) {
        state.food[i].coords.resize(state.D);
        for (int j = 0; j < state.D; ++j) cin >> state.food[i].coords[j];
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        VectorND obs;
        obs.coords.resize(state.D);
        for (int j = 0; j < state.D; ++j) cin >> obs.coords[j];
        state.obstacles.insert(obs);
    }

    // Game Loop
    for (int turn = 0; turn < state.max_turns; ++turn) {
        pair<int, int> move = find_move(state);
        if (move.first == -1) break;

        // Output move: axis direction
        cout << move.first << " " << move.second << endl;

        // Update local state to reflect the move
        VectorND next_head = state.snake.front();
        next_head.coords[move.first] += move.second;
        state.snake.push_front(next_head);

        bool ate_food = false;
        for (auto it = state.food.begin(); it != state.food.end(); ++it) {
            if (*it == next_head) {
                state.food.erase(it);
                ate_food = true;
                break;
            }
        }

        // If no food was eaten, the tail moves forward
        if (!ate_food) {
            state.snake.pop_back();
        }

        if (state.food.empty()) break;
    }

    return 0;
}
