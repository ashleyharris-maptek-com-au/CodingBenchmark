use std::io::{self, BufRead, BufWriter, Write};

fn solve() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lock = stdin.lock();
    let mut line = String::new();
    
    // Read the first line for number of vertices and edges
    lock.read_line(&mut line)?;
    let mut parts = line.split_whitespace();
    let num_vertices: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return Ok(()),
    };
    let num_edges: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => 0,
    };

    if num_vertices == 0 {
        println!("0");
        println!();
        return Ok(());
    }

    // Temporary storage for edges to build the CSR structure
    let mut u_edges = Vec::with_capacity(num_edges);
    let mut v_edges = Vec::with_capacity(num_edges);
    let mut degrees = vec![0u32; num_vertices];

    // Read edges and calculate degrees
    for _ in 0..num_edges {
        line.clear();
        if lock.read_line(&mut line)? == 0 { break; }
        let mut parts = line.split_whitespace();
        let u: usize = match parts.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => continue,
        };
        let v: usize = match parts.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => continue,
        };
        if u < num_vertices && v < num_vertices {
            u_edges.push(u as u32);
            v_edges.push(v as u32);
            degrees[u] += 1;
            degrees[v] += 1;
        }
    }

    // Calculate offsets for CSR structure
    let mut offsets = vec![0usize; num_vertices + 1];
    for i in 0..num_vertices {
        offsets[i+1] = offsets[i] + degrees[i] as usize;
    }

    // Build CSR adjacency list and edge ID map
    let mut adj = vec![0u32; 2 * num_edges];
    let mut edge_ids = vec![0u32; 2 * num_edges];
    let mut current_offset = offsets.clone();

    for i in 0..u_edges.len() {
        let u = u_edges[i] as usize;
        let v = v_edges[i] as usize;
        
        adj[current_offset[u]] = v as u32;
        edge_ids[current_offset[u]] = i as u32;
        current_offset[u] += 1;

        adj[current_offset[v]] = u as u32;
        edge_ids[current_offset[v]] = i as u32;
        current_offset[v] += 1;
    }

    // Free memory of temporary vectors
    drop(u_edges);
    drop(v_edges);
    drop(current_offset);

    let mut current_degrees = degrees.clone();
    let mut is_covered = vec![false; num_edges];
    let mut vertex_in_cover = vec![false; num_vertices];
    let mut cover = Vec::new();

    // Use a linked-list style bucket system for priority management (highest degree first)
    // This is more memory-efficient than a BinaryHeap for large graphs.
    let mut heads = vec![u32::MAX; num_vertices + 1];
    let mut nexts = vec![u32::MAX; num_vertices];
    let mut max_degree = 0usize;

    for i in 0..num_vertices {
        let d = current_degrees[i] as usize;
        if d > 0 {
            nexts[i] = heads[d];
            heads[d] = i as u32;
            if d > max_degree {
                max_degree = d;
            }
        }
    }

    let mut remaining_edges = num_edges;
    let mut d = max_degree;
    
    // Greedy heuristic: repeatedly pick the vertex with the highest remaining degree
    while remaining_edges > 0 && d > 0 {
        while heads[d] != u32::MAX {
            let u = heads[d] as usize;
            heads[d] = nexts[u];
            
            // If vertex is already in cover or its degree has decreased, skip or re-bucket it
            if vertex_in_cover[u] || (current_degrees[u] as usize) < d {
                if !vertex_in_cover[u] && current_degrees[u] > 0 {
                    let d_new = current_degrees[u] as usize;
                    nexts[u] = heads[d_new];
                    heads[d_new] = u as u32;
                }
                continue;
            }

            // Add vertex to the cover
            vertex_in_cover[u] = true;
            cover.push(u as u32);

            // Mark all incident edges as covered and update neighbor degrees
            for idx in offsets[u]..offsets[u+1] {
                let e_id = edge_ids[idx] as usize;
                if !is_covered[e_id] {
                    is_covered[e_id] = true;
                    remaining_edges -= 1;
                    let v = adj[idx] as usize;
                    current_degrees[v] -= 1;
                }
            }

            if remaining_edges == 0 {
                break;
            }
        }
        if d > 0 {
            d -= 1;
        }
    }

    // Fast output using BufWriter
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", cover.len())?;
    for (i, &v) in cover.iter().enumerate() {
        if i > 0 {
            write!(out, " ")?;
        }
        write!(out, "{}", v)?;
    }
    writeln!(out)?;

    Ok(())
}

fn main() {
    if let Err(e) = solve() {
        eprintln!("Error: {}", e);
    }
}
