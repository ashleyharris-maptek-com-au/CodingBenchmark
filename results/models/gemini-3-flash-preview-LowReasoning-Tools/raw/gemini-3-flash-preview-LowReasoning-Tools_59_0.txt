const fs = require('fs');

/**
 * Grid Path Planner for Search Drones
 * 
 * This program reads a JSON document from stdin containing a probability grid,
 * drone starting positions, and the maximum steps per drone. It outputs a JSON
 * document containing paths for each drone that maximize the unique probability
 * covered while avoiding obstacles and staying within the grid.
 * 
 * The algorithm uses a greedy BFS-based look-ahead to find the nearest cell
 * with a positive probability that has not yet been visited.
 */

function solve() {
    let input;
    try {
        // Read the entire input from stdin (fd 0)
        input = fs.readFileSync(0, 'utf8');
    } catch (err) {
        return;
    }
    if (!input) return;

    let data;
    try {
        data = JSON.parse(input);
    } catch (err) {
        return;
    }

    const { probability, drones, maxStepsPerDrone } = data;

    if (!probability || !probability.length || !drones) return;

    const rows = probability.length;
    const cols = probability[0].length;
    
    // Track visited cells globally across all drones to maximize unique coverage
    const visited = Array.from({ length: rows }, () => new Uint8Array(cols));

    const paths = drones.map(drone => {
        let currX = drone.x;
        let currY = drone.y;
        const path = [[currX, currY]];
        
        // Initial cell is marked as visited if not an obstacle
        if (probability[currY] && probability[currY][currX] !== -1) {
            visited[currY][currX] = 1;
        }

        for (let step = 0; step < maxStepsPerDrone; step++) {
            // BFS to find the nearest unvisited cell with probability > 0
            let queue = [{x: currX, y: currY, p: null, d: 0}];
            let bfsVisited = new Uint8Array(rows * cols);
            bfsVisited[currY * cols + currX] = 1;
            
            let head = 0;
            let firstMoveToTarget = [currX, currY];
            let foundTarget = false;
            
            // Limit BFS depth to balance performance and path quality
            const maxBfsDepth = 150; 

            while (head < queue.length) {
                const node = queue[head++];
                const {x: tx, y: ty, d: depth} = node;
                
                if (depth > maxBfsDepth) break;

                // If we found a cell with probability that hasn't been visited yet
                if (!visited[ty][tx] && probability[ty][tx] > 0) {
                    // Backtrack to find the first move of the path to this target
                    let curr = node;
                    while (curr.p && curr.p.p) {
                        curr = curr.p;
                    }
                    firstMoveToTarget = [curr.x, curr.y];
                    foundTarget = true;
                    break; 
                }

                // Standard 4-directional moves (up, down, left, right)
                const dx = [1, -1, 0, 0];
                const dy = [0, 0, 1, -1];
                for (let i = 0; i < 4; i++) {
                    const nx = tx + dx[i];
                    const ny = ty + dy[i];
                    if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                        const idx = ny * cols + nx;
                        if (!bfsVisited[idx]) {
                            bfsVisited[idx] = 1;
                            queue.push({x: nx, y: ny, p: node, d: depth + 1});
                        }
                    }
                }
            }

            if (foundTarget) {
                currX = firstMoveToTarget[0];
                currY = firstMoveToTarget[1];
            } else {
                // Fallback: If no positive probability target is found, move to the best neighboring cell
                const neighbors = [[currX + 1, currY], [currX - 1, currY], [currX, currY + 1], [currX, currY - 1], [currX, currY]];
                let bestN = [currX, currY];
                let bestScore = -Infinity;
                for (const [nx, ny] of neighbors) {
                    if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                        // Heuristic score: prefer unvisited cells
                        let score = (visited[ny][nx] ? -1 : 0);
                        if (score > bestScore) {
                            bestScore = score;
                            bestN = [nx, ny];
                        }
                    }
                }
                currX = bestN[0];
                currY = bestN[1];
            }
            
            path.push([currX, currY]);
            visited[currY][currX] = 1;
        }
        return path;
    });

    // Output the resulting paths as a single JSON document
    process.stdout.write(JSON.stringify({ paths }));
}

// Execute the solver
solve();
