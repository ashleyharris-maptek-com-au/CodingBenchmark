import sys
import math

def solve():
    # Read input
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    it = iter(input_data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAPACITY = int(next(it))
        
        depot_x = float(next(it))
        depot_y = float(next(it))
        
        customers = []
        for i in range(N):
            x = float(next(it))
            y = float(next(it))
            demand = int(next(it))
            customers.append({'id': i, 'x': x, 'y': y, 'demand': demand})
    except StopIteration:
        pass

    def dist(p1, p2):
        return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

    # Initial routes: each customer has their own route (depot -> i -> depot)
    # routes[i] = [customer_index]
    routes = [[i] for i in range(N)]
    
    # Precompute distances to depot
    dist_to_depot = [dist((c['x'], c['y']), (depot_x, depot_y)) for c in customers]
    
    # Precompute distances between customers
    dist_matrix = [[0.0] * N for _ in range(N)]
    for i in range(N):
        for j in range(i + 1, N):
            d = dist((customers[i]['x'], customers[i]['y']), (customers[j]['x'], customers[j]['y']))
            dist_matrix[i][j] = dist_matrix[j][i] = d
            
    # Calculate savings: S(i, j) = d(0, i) + d(0, j) - d(i, j)
    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            s = dist_to_depot[i] + dist_to_depot[j] - dist_matrix[i][j]
            savings.append((s, i, j))
            
    # Sort savings in descending order
    savings.sort(key=lambda x: x[0], reverse=True)
    
    # Map customer to the route they belong to
    # customer_to_route[i] = index of route in routes list
    customer_to_route = list(range(N))
    route_demands = [c['demand'] for c in customers]
    
    for s, i, j in savings:
        ri_idx = customer_to_route[i]
        rj_idx = customer_to_route[j]
        
        if ri_idx == rj_idx:
            continue
            
        ri = routes[ri_idx]
        rj = routes[rj_idx]
        
        # Check if i and j are at the ends of their respective routes
        # i must be first or last in ri, j must be first or last in rj
        is_i_end = (ri[0] == i or ri[-1] == i)
        is_j_end = (rj[0] == j or rj[-1] == j)
        
        if is_i_end and is_j_end:
            # Check capacity constraint
            if route_demands[ri_idx] + route_demands[rj_idx] <= CAPACITY:
                # Merge routes
                # Four cases based on which ends are connected
                if ri[-1] == i and rj[0] == j:
                    new_route = ri + rj
                elif ri[0] == i and rj[-1] == j:
                    new_route = rj + ri
                elif ri[-1] == i and rj[-1] == j:
                    new_route = ri + rj[::-1]
                elif ri[0] == i and rj[0] == j:
                    new_route = ri[::-1] + rj
                else:
                    continue # Should not happen
                
                # Update routes
                routes[ri_idx] = new_route
                routes[rj_idx] = None
                
                # Update route demands
                route_demands[ri_idx] += route_demands[rj_idx]
                
                # Update customer_to_route map
                for customer_id in rj:
                    customer_to_route[customer_id] = ri_idx

    # Filter out merged routes
    final_routes = [r for r in routes if r is not None]
    
    # Calculate total distance
    total_dist = 0
    for route in final_routes:
        if not route:
            continue
        # From depot to first customer
        total_dist += dist_to_depot[route[0]]
        # Between customers
        for k in range(len(route) - 1):
            total_dist += dist_matrix[route[k]][route[k+1]]
        # From last customer back to depot
        total_dist += dist_to_depot[route[-1]]
        
    # Output
    print(f"{total_dist:.4f}")
    print(len(final_routes))
    for route in final_routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    solve()
