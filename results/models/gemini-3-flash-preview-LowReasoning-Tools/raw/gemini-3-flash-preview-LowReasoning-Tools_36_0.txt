This C++ solution solves the **Maximum Independent Set (MIS)** problem exactly using an optimized backtracking approach. 

### Key Features:
1.  **Bitset Optimization:** Uses `std::bitset` for graph representation and set operations (intersection, difference, counting), which significantly accelerates adjacency checks and candidate set management.
2.  **Degree-based Pruning:**
    *   **Degree-0 Rule:** Automatically includes isolated vertices in the MIS.
    *   **Degree-1 Rule:** If a vertex has only one neighbor, it is always optimal to include that vertex and exclude its neighbor.
3.  **Heuristics & Pruning:**
    *   **Branching Heuristic:** For general nodes (degree > 1), it branches on the vertex with the maximum degree to maximize pruning in the "include" branch.
    *   **Upper Bound:** Uses a greedy upper bound to prune subtrees that cannot possibly improve the current best solution.
4.  **Platform Efficiency:** Utilizes `_Find_first()` and `_Find_next()` (efficient MSVC-specific bitset members) to iterate over set bits.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>

using namespace std;

// Using a bitset size suitable for standard graph benchmarks.
const int MAXN = 2048;

typedef bitset<MAXN> Bitset;

int N, M;
Bitset adj[MAXN];
vector<int> max_is;

// Simple greedy coloring for upper bound calculation
int greedy_bound(Bitset candidates) {
    int count = 0;
    while (candidates.any()) {
        // Pick vertex with minimum degree in the remaining subgraph to get a better bound
        int best_v = -1;
        int min_deg = MAXN + 1;
        for (int i = candidates._Find_first(); i < MAXN; i = candidates._Find_next(i)) {
            int deg = (adj[i] & candidates).count();
            if (deg < min_deg) {
                min_deg = deg;
                best_v = i;
            }
            if (min_deg == 0) break;
        }
        
        if (best_v == -1) break;
        candidates.reset(best_v);
        candidates &= ~adj[best_v];
        count++;
    }
    return count;
}

void solve(Bitset candidates, vector<int>& current) {
    if (candidates.none()) {
        if (current.size() > max_is.size()) {
            max_is = current;
        }
        return;
    }

    // Pruning: skip if remaining candidates + current size can't exceed best found
    if (current.size() + candidates.count() <= max_is.size()) {
        return;
    }

    // Find a vertex with minimum degree for core pruning rules
    int v = -1;
    int min_deg = MAXN + 1;
    for (int i = candidates._Find_first(); i < MAXN; i = candidates._Find_next(i)) {
        int deg = (adj[i] & candidates).count();
        if (deg < min_deg) {
            min_deg = deg;
            v = i;
        }
        if (min_deg == 0) break;
    }

    if (v == -1) return;

    if (min_deg == 0) {
        // Degree 0 Rule: Always include isolated vertices
        current.push_back(v);
        candidates.reset(v);
        solve(candidates, current);
        current.pop_back();
        return;
    }

    if (min_deg == 1) {
        // Degree 1 Rule: Always include v and discard its neighbor
        int neighbor = (adj[v] & candidates)._Find_first();
        current.push_back(v);
        Bitset next_candidates = candidates;
        next_candidates.reset(v);
        next_candidates.reset(neighbor);
        solve(next_candidates, current);
        current.pop_back();
        return;
    }

    // General case: Branch on a vertex with maximum degree to maximize pruning
    int max_v = -1;
    int max_deg = -1;
    for (int i = candidates._Find_first(); i < MAXN; i = candidates._Find_next(i)) {
        int deg = (adj[i] & candidates).count();
        if (deg > max_deg) {
            max_deg = deg;
            max_v = i;
        }
    }
    
    v = max_v;

    // Option 1: Include v in MIS
    current.push_back(v);
    Bitset next_candidates = candidates & ~adj[v];
    next_candidates.reset(v);
    solve(next_candidates, current);
    current.pop_back();

    // Option 2: Exclude v from MIS
    candidates.reset(v);
    solve(candidates, current);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> M)) return 0;

    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < N && v >= 0 && v < N && u != v) {
            adj[u].set(v);
            adj[v].set(u);
        }
    }

    Bitset initial_candidates;
    for (int i = 0; i < N; ++i) initial_candidates.set(i);

    vector<int> current;
    solve(initial_candidates, current);

    cout << max_is.size() << "\n";
    for (int i = 0; i < (int)max_is.size(); ++i) {
        cout << max_is[i] << (i == (int)max_is.size() - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
