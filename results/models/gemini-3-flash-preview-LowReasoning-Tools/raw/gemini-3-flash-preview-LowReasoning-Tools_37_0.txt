The provided C# code implements an optimal solver for the **Directed Feedback Vertex Set (DFVS)** problem. It employs several advanced techniques to handle the NP-hard nature of the problem efficiently:

1.  **Graph Reductions:** It applies standard graph reduction rules such as source/sink removal, self-loop identification, and simple path bypassing (Rule 1 and Rule 2).
2.  **SCC Decomposition:** The problem is broken down into its Strongly Connected Components (SCCs), which are solved independently to reduce the search space.
3.  **Iterative Deepening Search:** For each non-trivial SCC, the solver uses iterative deepening to find the smallest possible feedback vertex set.
4.  **Smallest Cycle Branching:** During branching, it identifies the shortest cycle and branches on all vertices within that cycle, prioritizing vertices with higher degrees.
5.  **Lower Bound Pruning:** It uses a lower bound based on the maximum number of vertex-disjoint 2-cycles to prune the search tree.

using System;
using System.Collections.Generic;
using System.Linq;

#nullable enable

public class DFVSSolver
{
    public static void Main(string[] args)
    {
        var inputReader = new InputReader(Console.OpenStandardInput());
        int n = inputReader.NextInt();
        int m = inputReader.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var adj = new HashSet<int>[n];
        var revAdj = new HashSet<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new HashSet<int>();
            revAdj[i] = new HashSet<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = inputReader.NextInt();
            int v = inputReader.NextInt();
            if (u >= 0 && u < n && v >= 0 && v < n)
            {
                adj[u].Add(v);
                revAdj[v].Add(u);
            }
        }

        var solver = new Solver(n, adj, revAdj);
        var result = solver.Solve();

        Console.WriteLine(result.Count);
        Console.WriteLine(string.Join(" ", result));
    }
}

public class Solver
{
    private int _n;
    private HashSet<int>?[] _adj;
    private HashSet<int>?[] _revAdj;

    public Solver(int n, HashSet<int>[] adj, HashSet<int>[] revAdj)
    {
        _n = n;
        _adj = adj;
        _revAdj = revAdj;
    }

    public List<int> Solve()
    {
        List<int> dfvs = new List<int>();
        ApplyGlobalReductions(dfvs);

        var activeNodes = new HashSet<int>();
        for (int i = 0; i < _n; i++)
        {
            if (_adj[i] != null) activeNodes.Add(i);
        }

        var sccs = GetSCCs(activeNodes);
        foreach (var scc in sccs)
        {
            var sccDfvs = SolveSCC(scc);
            dfvs.AddRange(sccDfvs);
        }

        return dfvs;
    }

    private void ApplyGlobalReductions(List<int> dfvs)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int i = 0; i < _n; i++)
            {
                var curAdj = _adj[i];
                var curRevAdj = _revAdj[i];
                if (curAdj == null || curRevAdj == null) continue;

                if (curAdj.Contains(i))
                {
                    dfvs.Add(i);
                    RemoveNode(i);
                    changed = true;
                    continue;
                }

                if (curAdj.Count == 0 || curRevAdj.Count == 0)
                {
                    RemoveNode(i);
                    changed = true;
                    continue;
                }

                if (curAdj.Count == 1)
                {
                    int v = curAdj.First();
                    foreach (int u in curRevAdj.ToList())
                    {
                        if (u != v)
                        {
                            var uAdj = _adj[u];
                            if (uAdj != null && !uAdj.Contains(v))
                            {
                                uAdj.Add(v);
                                _revAdj[v]?.Add(u);
                            }
                        }
                    }
                    RemoveNode(i);
                    changed = true;
                    continue;
                }
                
                if (curRevAdj.Count == 1)
                {
                    int u = curRevAdj.First();
                    foreach (int v in curAdj.ToList())
                    {
                        if (u != v)
                        {
                            var uAdj = _adj[u];
                            if (uAdj != null && !uAdj.Contains(v))
                            {
                                uAdj.Add(v);
                                _revAdj[v]?.Add(u);
                            }
                        }
                    }
                    RemoveNode(i);
                    changed = true;
                    continue;
                }
            }
        }
    }

    private void RemoveNode(int u)
    {
        var curAdj = _adj[u];
        var curRevAdj = _revAdj[u];
        if (curAdj != null) foreach (int v in curAdj) _revAdj[v]?.Remove(u);
        if (curRevAdj != null) foreach (int v in curRevAdj) _adj[v]?.Remove(u);
        _adj[u] = null;
        _revAdj[u] = null;
    }

    private List<HashSet<int>> GetSCCs(HashSet<int> nodes)
    {
        var result = new List<HashSet<int>>();
        if (nodes.Count == 0) return result;

        var stack = new Stack<int>();
        var visited = new HashSet<int>();
        foreach (var node in nodes)
        {
            if (!visited.Contains(node)) DFS1(node, visited, stack, nodes);
        }

        visited.Clear();
        while (stack.Count > 0)
        {
            int u = stack.Pop();
            if (!visited.Contains(u))
            {
                var scc = new HashSet<int>();
                DFS2(u, visited, scc, nodes);
                if (scc.Count > 1 || (scc.Count == 1 && _adj[scc.First()]?.Contains(scc.First()) == true))
                    result.Add(scc);
            }
        }
        return result;
    }

    private void DFS1(int u, HashSet<int> visited, Stack<int> stack, HashSet<int> nodes)
    {
        visited.Add(u);
        var curAdj = _adj[u];
        if (curAdj != null)
        {
            foreach (int v in curAdj)
            {
                if (nodes.Contains(v) && !visited.Contains(v)) DFS1(v, visited, stack, nodes);
            }
        }
        stack.Push(u);
    }

    private void DFS2(int u, HashSet<int> visited, HashSet<int> scc, HashSet<int> nodes)
    {
        visited.Add(u);
        scc.Add(u);
        var curRevAdj = _revAdj[u];
        if (curRevAdj != null)
        {
            foreach (int v in curRevAdj)
            {
                if (nodes.Contains(v) && !visited.Contains(v)) DFS2(v, visited, scc, nodes);
            }
        }
    }

    private List<int> SolveSCC(HashSet<int> sccNodes)
    {
        if (sccNodes.Count == 0) return new List<int>();

        int k = 0;
        while (true)
        {
            var result = SolveRecursive(sccNodes, k);
            if (result != null) return result;
            k++;
        }
    }

    private List<int>? SolveRecursive(HashSet<int> nodes, int k)
    {
        if (k < 0) return null;

        var localAdj = new Dictionary<int, HashSet<int>>();
        var localRevAdj = new Dictionary<int, HashSet<int>>();
        foreach (var u in nodes)
        {
            localAdj[u] = new HashSet<int>();
            localRevAdj[u] = new HashSet<int>();
        }
        foreach (var u in nodes)
        {
            var curAdj = _adj[u];
            if (curAdj != null)
            {
                foreach (var v in curAdj)
                {
                    if (nodes.Contains(v))
                    {
                        localAdj[u].Add(v);
                        localRevAdj[v].Add(u);
                    }
                }
            }
        }

        var localDFVS = new List<int>();
        if (!ApplyLocalReductions(localAdj, localRevAdj, localDFVS, ref k)) return null;

        if (localAdj.Count == 0) return localDFVS;
        if (k == 0) return null;

        int lb = Get2CycleLowerBound(localAdj);
        if (lb > k) return null;

        var cycle = FindShortestCycle(localAdj);
        if (cycle == null) return localDFVS;

        var sortedCycle = cycle.OrderByDescending(v => localAdj[v].Count + localRevAdj[v].Count).ToList();

        foreach (int v in sortedCycle)
        {
            var nextNodes = new HashSet<int>(localAdj.Keys);
            nextNodes.Remove(v);
            var res = SolveRecursive(nextNodes, k - 1);
            if (res != null)
            {
                res.AddRange(localDFVS);
                res.Add(v);
                return res;
            }
        }

        return null;
    }

    private bool ApplyLocalReductions(Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj, List<int> dfvs, ref int k)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            var currentNodes = adj.Keys.ToList();
            foreach (var i in currentNodes)
            {
                if (!adj.TryGetValue(i, out var curAdj) || !revAdj.TryGetValue(i, out var curRevAdj)) continue;

                if (curAdj.Contains(i))
                {
                    dfvs.Add(i);
                    RemoveLocalNode(i, adj, revAdj);
                    k--;
                    if (k < 0) return false;
                    changed = true;
                    continue;
                }

                if (curAdj.Count == 0 || curRevAdj.Count == 0)
                {
                    RemoveLocalNode(i, adj, revAdj);
                    changed = true;
                    continue;
                }

                if (curAdj.Count == 1)
                {
                    int v = curAdj.First();
                    foreach (int u in curRevAdj.ToList())
                    {
                        if (u != v)
                        {
                            if (adj.TryGetValue(u, out var uAdj) && !uAdj.Contains(v))
                            {
                                uAdj.Add(v);
                                revAdj[v].Add(u);
                            }
                        }
                    }
                    RemoveLocalNode(i, adj, revAdj);
                    changed = true;
                    continue;
                }

                if (curRevAdj.Count == 1)
                {
                    int u = curRevAdj.First();
                    foreach (int v in curAdj.ToList())
                    {
                        if (u != v)
                        {
                            if (adj.TryGetValue(u, out var uAdj) && !uAdj.Contains(v))
                            {
                                uAdj.Add(v);
                                revAdj[v].Add(u);
                            }
                        }
                    }
                    RemoveLocalNode(i, adj, revAdj);
                    changed = true;
                    continue;
                }
            }
        }
        return true;
    }

    private void RemoveLocalNode(int u, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj)
    {
        if (adj.TryGetValue(u, out var curAdj))
            foreach (int v in curAdj) revAdj[v].Remove(u);
        if (revAdj.TryGetValue(u, out var curRevAdj))
            foreach (int v in curRevAdj) adj[v].Remove(u);
        adj.Remove(u);
        revAdj.Remove(u);
    }

    private int Get2CycleLowerBound(Dictionary<int, HashSet<int>> adj)
    {
        var visited = new HashSet<int>();
        int lb = 0;
        foreach (var u in adj.Keys)
        {
            if (visited.Contains(u)) continue;
            foreach (var v in adj[u])
            {
                if (!visited.Contains(v) && adj[v].Contains(u))
                {
                    lb++;
                    visited.Add(u);
                    visited.Add(v);
                    break;
                }
            }
        }
        return lb;
    }

    private List<int>? FindShortestCycle(Dictionary<int, HashSet<int>> adj)
    {
        int minLen = int.MaxValue;
        List<int>? bestCycle = null;

        foreach (var startNode in adj.Keys)
        {
            var q = new Queue<int>();
            q.Enqueue(startNode);
            var dist = new Dictionary<int, int> { { startNode, 0 } };
            var parent = new Dictionary<int, int>();

            while (q.Count > 0)
            {
                int u = q.Dequeue();
                if (dist[u] >= minLen) break;

                foreach (int v in adj[u])
                {
                    if (v == startNode)
                    {
                        var cycle = new List<int>();
                        int curr = u;
                        while (curr != startNode)
                        {
                            cycle.Add(curr);
                            curr = parent[curr];
                        }
                        cycle.Add(startNode);
                        if (cycle.Count < minLen)
                        {
                            minLen = cycle.Count;
                            bestCycle = cycle;
                        }
                        goto next_start;
                    }
                    if (!dist.ContainsKey(v))
                    {
                        dist[v] = dist[u] + 1;
                        parent[v] = u;
                        q.Enqueue(v);
                    }
                }
            }
            next_start:;
            if (minLen == 2) break;
        }

        return bestCycle;
    }
}

public class InputReader
{
    private System.IO.Stream stream;
    private byte[] buffer = new byte[65536];
    private int curChar;
    private int numChars;

    public InputReader(System.IO.Stream stream)
    {
        this.stream = stream;
    }

    private int Read()
    {
        if (numChars == -1) return -1;
        if (curChar >= numChars)
        {
            curChar = 0;
            try { numChars = stream.Read(buffer, 0, buffer.Length); }
            catch { numChars = -1; }
            if (numChars <= 0) return -1;
        }
        return buffer[curChar++];
    }

    public int NextInt()
    {
        int c = Read();
        while (c != -1 && c <= 32) c = Read();
        if (c == -1) return -1;
        int sgn = 1;
        if (c == '-') { sgn = -1; c = Read(); }
        int res = 0;
        do
        {
            res = res * 10 + (c - '0');
            c = Read();
        } while (c > 32);
        return res * sgn;
    }
}
