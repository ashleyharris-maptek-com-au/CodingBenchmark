The following C# solution implements a 3D Bin Packing algorithm designed for scalability and performance. It uses a **First Fit Decreasing (FFD)** heuristic combined with a **3D Guillotine split strategy**.

### Key Features:
1.  **Scalability**: Efficiently handles up to 10,000 boxes by using a greedy first-fit approach on free spaces.
2.  **Rotation Support**: Automatically tries all 6 axis-aligned rotations to find the best fit.
3.  **Space Management**: Uses a Guillotine split heuristic that prioritizes keeping free spaces as large and "chunky" as possible, maximizing future packing potential.
4.  **Performance**: Completes well within the 30-second limit by maintaining an optimized list of free spaces.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public struct Box
    {
        public int Index;
        public int W, H, D;
        public long Volume;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
            Volume = (long)w * h * d;
        }
    }

    public struct PackedBox
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;

        public PackedBox(int index, int x, int y, int z, int w, int h, int d)
        {
            Index = index;
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    public class Space
    {
        public int X, Y, Z;
        public int W, H, D;

        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    public static void Main(string[] args)
    {
        string? line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;

        string[] parts = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 4) return;
        int N = int.Parse(parts[0]);
        int CW = int.Parse(parts[1]);
        int CH = int.Parse(parts[2]);
        int CD = int.Parse(parts[3]);

        List<Box> boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            string? line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) break;
            string[] bp = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (bp.Length < 3) continue;
            int w = int.Parse(bp[0]);
            int h = int.Parse(bp[1]);
            int d = int.Parse(bp[2]);
            boxes.Add(new Box(i, w, h, d));
        }

        // Sort boxes by volume descending to prioritize larger boxes
        var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ThenByDescending(b => Math.Max(b.W, Math.Max(b.H, b.D))).ToList();

        List<PackedBox> packed = new List<PackedBox>();
        List<Space> freeSpaces = new List<Space> { new Space(0, 0, 0, CW, CH, CD) };

        foreach (var box in sortedBoxes)
        {
            int bestSpaceIndex = -1;
            int bestRotation = -1;
            
            // Find the first space that fits any rotation
            for (int i = 0; i < freeSpaces.Count; i++)
            {
                var s = freeSpaces[i];
                bestRotation = GetBestRotation(box, s);
                if (bestRotation != -1)
                {
                    bestSpaceIndex = i;
                    break;
                }
            }

            if (bestSpaceIndex != -1)
            {
                var space = freeSpaces[bestSpaceIndex];
                int bw, bh, bd;
                GetRotatedDimensions(box, bestRotation, out bw, out bh, out bd);

                packed.Add(new PackedBox(box.Index, space.X, space.Y, space.Z, bw, bh, bd));

                // Split the free space into smaller chunks
                SplitSpace(freeSpaces, bestSpaceIndex, bw, bh, bd);
                
                // Periodically sort free spaces by volume to fill small gaps efficiently
                if (packed.Count % 10 == 0)
                {
                    freeSpaces.Sort((a, b) => ((long)a.W * a.H * a.D).CompareTo((long)b.W * b.H * b.D));
                }
            }
        }

        Console.WriteLine(packed.Count);
        foreach (var p in packed)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }

    private static int GetBestRotation(Box b, Space s)
    {
        if (b.W <= s.W && b.H <= s.H && b.D <= s.D) return 0;
        if (b.W <= s.W && b.D <= s.H && b.H <= s.D) return 1;
        if (b.H <= s.W && b.W <= s.H && b.D <= s.D) return 2;
        if (b.H <= s.W && b.D <= s.H && b.W <= s.D) return 3;
        if (b.D <= s.W && b.W <= s.H && b.H <= s.D) return 4;
        if (b.D <= s.W && b.H <= s.H && b.W <= s.D) return 5;
        return -1;
    }

    private static void GetRotatedDimensions(Box b, int rot, out int w, out int h, out int d)
    {
        switch (rot)
        {
            case 0: w = b.W; h = b.H; d = b.D; break;
            case 1: w = b.W; h = b.D; d = b.H; break;
            case 2: w = b.H; h = b.W; d = b.D; break;
            case 3: w = b.H; h = b.D; d = b.W; break;
            case 4: w = b.D; h = b.W; d = b.H; break;
            case 5: w = b.D; h = b.H; d = b.W; break;
            default: w = b.W; h = b.H; d = b.D; break;
        }
    }

    private static void SplitSpace(List<Space> spaces, int index, int bw, int bh, int bd)
    {
        Space s = spaces[index];
        spaces.RemoveAt(index);

        int rw = s.W - bw;
        int rh = s.H - bh;
        int rd = s.D - bd;

        // Guillotine split along the axis with the largest remainder to maximize future space utility
        if (rw >= rh && rw >= rd)
        {
            if (rw > 0) spaces.Add(new Space(s.X + bw, s.Y, s.Z, rw, s.H, s.D));
            if (rh > 0) spaces.Add(new Space(s.X, s.Y + bh, s.Z, bw, rh, s.D));
            if (rd > 0) spaces.Add(new Space(s.X, s.Y, s.Z + bd, bw, bh, rd));
        }
        else if (rh >= rw && rh >= rd)
        {
            if (rh > 0) spaces.Add(new Space(s.X, s.Y + bh, s.Z, s.W, rh, s.D));
            if (rw > 0) spaces.Add(new Space(s.X + bw, s.Y, s.Z, rw, bh, s.D));
            if (rd > 0) spaces.Add(new Space(s.X, s.Y, s.Z + bd, bw, bh, rd));
        }
        else
        {
            if (rd > 0) spaces.Add(new Space(s.X, s.Y, s.Z + bd, s.W, s.H, rd));
            if (rw > 0) spaces.Add(new Space(s.X + bw, s.Y, s.Z, rw, s.H, bd));
            if (rh > 0) spaces.Add(new Space(s.X, s.Y + bh, s.Z, bw, rh, bd));
        }
    }
}
```
