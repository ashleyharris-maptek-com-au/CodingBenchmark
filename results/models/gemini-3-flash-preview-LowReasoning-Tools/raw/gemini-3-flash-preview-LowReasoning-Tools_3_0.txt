The solution implements a high-performance **Force-Directed Graph Layout** algorithm based on the Fruchterman-Reingold model, specifically optimized for extreme scalability (up to 200,000 nodes) and execution within 30 seconds.

### Key Features:
- **Parallel Processing**: Utilizes the `rayon` library to parallelize both repulsive and attractive force calculations across available CPU cores.
- **Spatial Indexing**: Implements a grid-based spatial partition for repulsive forces. This reduces the complexity from $O(N^2)$ to approximately $O(N)$ for local interactions, which is critical for handling large graphs.
- **Adaptive Iterations**: The number of iterations and the cooling schedule (temperature $t$) adapt based on the graph size to balance quality and performance.
- **Robustness**: Handles disconnected components and various graph densities by normalizing initial areas and force constants.

### Rust Implementation:

```rust
use std::io::{self, BufRead};
use rand::Rng;
use rayon::prelude::*;
use std::f64;

struct Point {
    x: f64,
    y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };

    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap_or(0))
        .collect();

    if parts.len() < 2 {
        return;
    }

    let n = parts[0];
    let m = parts[1];

    let mut adj = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let edge_parts: Vec<usize> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap_or(0))
                .collect();
            if edge_parts.len() == 2 {
                let u = edge_parts[0];
                let v = edge_parts[1];
                if u < n && v < n {
                    adj[u].push(v);
                    adj[v].push(u);
                }
            }
        }
    }

    let mut points: Vec<Point> = (0..n)
        .map(|_| {
            let mut rng = rand::thread_rng();
            Point {
                x: rng.gen_range(0.0..1000.0),
                y: rng.gen_range(0.0..1000.0),
            }
        })
        .collect();

    let iterations = if n < 100 {
        300
    } else if n < 1000 {
        100
    } else if n < 10000 {
        50
    } else if n < 100000 {
        30
    } else {
        15
    };

    let area = (n as f64).sqrt() * 100.0 * (n as f64).sqrt() * 100.0;
    let k = (area / n as f64).sqrt();
    let mut t = (n as f64).sqrt() * 10.0;
    let dt = t / iterations as f64;

    for _ in 0..iterations {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

        // Repulsive forces
        if n < 500 {
            // All-pairs repulsion for very small graphs
            let repulsive_forces: Vec<(f64, f64)> = (0..n).into_par_iter().map(|v| {
                let mut dx_total = 0.0;
                let mut dy_total = 0.0;
                for u in 0..n {
                    if u != v {
                        let dx = points[v].x - points[u].x;
                        let dy = points[v].y - points[u].y;
                        let dist_sq = dx * dx + dy * dy + 1e-9;
                        let dist = dist_sq.sqrt();
                        let f = k * k / dist;
                        dx_total += (dx / dist) * f;
                        dy_total += (dy / dist) * f;
                    }
                }
                (dx_total, dy_total)
            }).collect();
            disp = repulsive_forces;
        } else {
            // Grid-based repulsion for larger graphs
            let grid_size = ((n as f64).sqrt() / 2.0).max(5.0) as usize;
            let mut min_x = f64::MAX;
            let mut max_x = f64::MIN;
            let mut min_y = f64::MAX;
            let mut max_y = f64::MIN;

            for p in &points {
                min_x = min_x.min(p.x);
                max_x = max_x.max(p.x);
                min_y = min_y.min(p.y);
                max_y = max_y.max(p.y);
            }

            let range_x = (max_x - min_x).max(1e-9);
            let range_y = (max_y - min_y).max(1e-9);

            let mut grid: Vec<Vec<usize>> = vec![Vec::new(); grid_size * grid_size];
            for (i, p) in points.iter().enumerate() {
                let gx = (((p.x - min_x) / range_x) * (grid_size as f64 - 1.0)).floor() as usize;
                let gy = (((p.y - min_y) / range_y) * (grid_size as f64 - 1.0)).floor() as usize;
                grid[gy * grid_size + gx].push(i);
            }

            let repulsive_forces: Vec<(f64, f64)> = (0..n).into_par_iter().map(|v| {
                let mut dx_total = 0.0;
                let mut dy_total = 0.0;
                let p_v = &points[v];
                let gx = (((p_v.x - min_x) / range_x) * (grid_size as f64 - 1.0)).floor() as usize;
                let gy = (((p_v.y - min_y) / range_y) * (grid_size as f64 - 1.0)).floor() as usize;

                for dy_idx in -1..=1 {
                    for dx_idx in -1..=1 {
                        let ny = gy as isize + dy_idx;
                        let nx = gx as isize + dx_idx;
                        if ny >= 0 && ny < grid_size as isize && nx >= 0 && nx < grid_size as isize {
                            let cell_idx = (ny as usize) * grid_size + (nx as usize);
                            for &u in &grid[cell_idx] {
                                if u != v {
                                    let dx = points[v].x - points[u].x;
                                    let dy = points[v].y - points[u].y;
                                    let dist_sq = dx * dx + dy * dy + 1e-9;
                                    let dist = dist_sq.sqrt();
                                    let f = k * k / dist;
                                    dx_total += (dx / dist) * f;
                                    dy_total += (dy / dist) * f;
                                }
                            }
                        }
                    }
                }
                (dx_total, dy_total)
            }).collect();
            disp = repulsive_forces;
        }

        // Attractive forces
        let attractive_updates: Vec<Vec<(usize, f64, f64)>> = (0..n).into_par_iter().map(|v| {
            let mut updates = Vec::new();
            for &u in &adj[v] {
                if v < u {
                    let dx = points[v].x - points[u].x;
                    let dy = points[v].y - points[u].y;
                    let dist_sq = dx * dx + dy * dy + 1e-9;
                    let dist = dist_sq.sqrt();
                    let f = dist * dist / k;
                    let adx = (dx / dist) * f;
                    let ady = (dy / dist) * f;
                    updates.push((v, -adx, -ady));
                    updates.push((u, adx, ady));
                }
            }
            updates
        }).collect();

        for update_list in attractive_updates {
            for (idx, adx, ady) in update_list {
                disp[idx].0 += adx;
                disp[idx].1 += ady;
            }
        }

        // Apply displacement limited by temperature t
        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let dist_sq = dx * dx + dy * dy + 1e-9;
            let dist = dist_sq.sqrt();
            points[i].x += (dx / dist) * dist.min(t);
            points[i].y += (dy / dist) * dist.min(t);
        }

        t -= dt;
    }

    println!("{}", n);
    for p in points {
        println!("{:.6} {:.6}", p.x, p.y);
    }
}
```
