#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick pattern scale
    vec2 scale = vec2(8.0, 16.0);
    vec2 brickUV = uv * scale;

    // Staggered rows: offset X by 0.5 for every other row
    if (mod(floor(brickUV.y), 2.0) > 0.5) {
        brickUV.x += 0.5;
    }

    // Local coordinates within a brick [0, 1]
    vec2 localUV = fract(brickUV);

    // Mortar thickness
    vec2 mortar = vec2(0.05, 0.1);

    // Determine if we are in the brick or mortar
    // step(edge, x) returns 0.0 if x < edge, 1.0 otherwise
    // We want brick if localUV.x > mortar.x/2 and localUV.x < 1.0 - mortar.x/2
    // Wait, usually mortar is at the edges. 
    // If thickness is 0.05, and we center the brick, mortar is [0, 0.025] and [0.975, 1.0].
    // Simpler: use smoothstep or step to define the brick area.
    vec2 brickSelect = step(mortar, localUV);
    float isBrick = brickSelect.x * brickSelect.y;

    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 baseColor = mix(mortarColor, brickColor, isBrick);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.4 + 0.6 * diffuse;

    outColor = vec4(baseColor * shading, 1.0);
}
