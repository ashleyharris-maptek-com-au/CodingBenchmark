; SPIR-V
; Version: 1.0
; Generator: Khronos; 0
; Bound: 100
; Schema: 0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform

%_ptr_Input_v3float = OpTypePointer Input %v3float
   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input

%_ptr_Output_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %_ptr_Output_v4float Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1

; Main function
       %main = OpFunction %void None %3
          %5 = OpLabel

; Load normal and normalize
     %norm_v = OpLoad %v3float %normal
     %norm_n = OpExtInst %v3float %1 Normalize %norm_v

; Load worldPos
    %world_v = OpLoad %v3float %worldPos

; Load lightPos from UBO
      %lp_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
       %lp_v4 = OpLoad %v4float %lp_ptr
       %lp_v3 = OpVectorShuffle %v3float %lp_v4 %lp_v4 0 1 2

; Calculate L = normalize(lightPos.xyz - worldPos)
      %L_sub = OpFSub %v3float %lp_v3 %world_v
        %L_n = OpExtInst %v3float %1 Normalize %L_sub

; Calculate dot(N, L)
      %dotNL = OpDot %float %norm_n %L_n

; Calculate d = max(dot(N, L), 0)
          %d = OpExtInst %float %1 FMax %dotNL %float_0

; Construct outColor = vec4(d, d, d, 1.0)
      %res_v = OpCompositeConstruct %v4float %d %d %d %float_1

; Store result
               OpStore %outColor %res_v
               OpReturn
               OpFunctionEnd
