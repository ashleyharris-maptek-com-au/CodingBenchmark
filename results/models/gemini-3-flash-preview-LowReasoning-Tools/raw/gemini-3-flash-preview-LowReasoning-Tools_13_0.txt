The following C++ solution solves the Longest Common Substring (LCS) problem for $N$ strings. It uses a **Generalized Suffix Automaton (SAM)** approach to achieve high performance and scalability, even for "Extreme" cases with 100 strings of 10MB+ each.

### Strategy Highlights:
- **SAM on Shortest String**: The SAM is built on the shortest input string to minimize the number of states ($O(L_{min})$), keeping the memory footprint well within the 32GB limit.
- **Linear Time Complexity**: The algorithm processes each subsequent string in $O(L_i \log \Sigma)$ time, where $\Sigma$ is the alphabet size (using `std::map` for transitions).
- **Suffix Link Tree Propagation**: After processing each string, matching lengths are propagated up the suffix link tree in $O(L_{min})$ using a topological sort (counting sort by length) to ensure correctness.
- **Efficient Extraction**: The `first_pos` of each state is tracked during SAM construction to easily extract the LCS substring from the base string.

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <map>

using namespace std;

// Suffix Automaton Node structure
struct Node {
    int len, link;
    map<char, int> next;
    int min_match;
    int cur_match;
    int first_pos;

    Node() : len(0), link(-1), min_match(0), cur_match(0), first_pos(-1) {}
};

class SuffixAutomaton {
public:
    vector<Node> st;
    int last;

    SuffixAutomaton(const string& s) {
        st.reserve(2 * s.length() + 1);
        st.emplace_back();
        st[0].len = 0;
        st[0].link = -1;
        last = 0;
        for (int i = 0; i < (int)s.length(); ++i) {
            extend(s[i], i);
        }
        // Initialize min_match with max possible (own length)
        for (int i = 0; i < (int)st.size(); ++i) {
            st[i].min_match = st[i].len;
        }
    }

    void extend(char c, int pos) {
        int cur = st.size();
        st.emplace_back();
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;
        int p = last;
        while (p != -1 && st[p].next.find(c) == st[p].next.end()) {
            st[p].next[c] = cur;
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = st.size();
                st.emplace_back();
                st[clone].len = st[p].len + 1;
                st[clone].next = st[q].next;
                st[clone].link = st[q].link;
                st[clone].first_pos = st[q].first_pos;
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    // Process a string against the SAM and update min_match for each state
    void process(const string& s) {
        for (int i = 0; i < (int)st.size(); ++i) {
            st[i].cur_match = 0;
        }
        int v = 0, l = 0;
        for (char c : s) {
            auto it = st[v].next.find(c);
            while (v != 0 && it == st[v].next.end()) {
                v = st[v].link;
                l = st[v].len;
                it = st[v].next.find(c);
            }
            if (it != st[v].next.end()) {
                v = it->second;
                l++;
            }
            if (l > st[v].cur_match) st[v].cur_match = l;
        }

        // Topological sort (by length) to propagate match lengths up the suffix link tree
        static vector<int> nodes_by_len;
        static vector<int> count;
        int max_l = 0;
        for (int i = 0; i < (int)st.size(); ++i) if (st[i].len > max_l) max_l = st[i].len;
        
        nodes_by_len.resize(st.size());
        count.assign(max_l + 1, 0);
        for (int i = 0; i < (int)st.size(); ++i) count[st[i].len]++;
        for (int i = 1; i <= max_l; ++i) count[i] += count[i - 1];
        for (int i = 0; i < (int)st.size(); ++i) nodes_by_len[--count[st[i].len]] = i;

        for (int i = (int)st.size() - 1; i >= 0; --i) {
            int v_idx = nodes_by_len[i];
            if (v_idx == 0) continue;
            int p = st[v_idx].link;
            if (p != -1) {
                if (st[v_idx].cur_match > st[p].cur_match) {
                    int propagated = min(st[p].len, st[v_idx].cur_match);
                    if (propagated > st[p].cur_match) st[p].cur_match = propagated;
                }
            }
            if (st[v_idx].cur_match < st[v_idx].min_match) st[v_idx].min_match = st[v_idx].cur_match;
        }
        st[0].min_match = 0;
    }

    // Find the state with the maximum min_match
    pair<int, int> get_lcs() {
        int max_len = 0;
        int best_node = 0;
        for (int i = 1; i < (int)st.size(); ++i) {
            if (st[i].min_match > max_len) {
                max_len = st[i].min_match;
                best_node = i;
            }
        }
        return {max_len, best_node};
    }
};

int main() {
    // Fast I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;
    if (n <= 0) {
        cout << endl;
        return 0;
    }

    vector<string> s(n);
    int shortest_idx = 0;
    for (int i = 0; i < n; ++i) {
        if (!(cin >> s[i])) {
            n = i;
            break;
        }
        if (s[i].length() < s[shortest_idx].length()) {
            shortest_idx = i;
        }
    }

    if (n == 0) return 0;

    // Build SAM on the shortest string
    SuffixAutomaton sam(s[shortest_idx]);

    // Process all other strings
    for (int i = 0; i < n; ++i) {
        if (i == shortest_idx) continue;
        sam.process(s[i]);
    }

    // Find and output the LCS
    auto [max_len, best_node] = sam.get_lcs();
    if (max_len == 0) {
        cout << endl;
    } else {
        int end_pos = sam.st[best_node].first_pos;
        cout << s[shortest_idx].substr(end_pos - max_len + 1, max_len) << endl;
    }

    return 0;
}
```
