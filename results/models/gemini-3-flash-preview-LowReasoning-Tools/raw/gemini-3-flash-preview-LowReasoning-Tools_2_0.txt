using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

public class Solution
{
    public class Edge
    {
        public int To;
        public int Weight;
        public int Id;
        public bool Used;

        public Edge(int to, int weight, int id)
        {
            To = to;
            Weight = weight;
            Id = id;
            Used = false;
        }
    }

    private static int[] degrees;

    public static void Main(string[] args)
    {
        var reader = new StreamReader(Console.OpenStandardInput(), Encoding.UTF8, false, 1 << 16);
        var line = reader.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;

        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        int N = int.Parse(parts[0]);
        int M = int.Parse(parts[1]);

        List<Edge>[] adj = new List<Edge>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<Edge>();

        degrees = new int[N];
        for (int i = 0; i < M; i++)
        {
            line = reader.ReadLine();
            if (line == null) break;
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 3) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);

            adj[u].Add(new Edge(v, w, i));
            adj[v].Add(new Edge(u, w, i));
            degrees[u]++;
            degrees[v]++;
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degrees[i] % 2 != 0)
            {
                oddNodes.Add(i);
            }
        }

        if (oddNodes.Count > 0)
        {
            bool[] matched = new bool[N];
            int edgeIdCounter = M;

            foreach (var u in oddNodes)
            {
                if (matched[u]) continue;

                int v = FindNearestOddNode(u, matched, adj, N);
                if (v != -1)
                {
                    matched[u] = true;
                    matched[v] = true;

                    List<ValueTuple<int, int>> pathWithWeights = GetShortestPathWithWeights(u, v, adj, N);
                    for (int i = 0; i < pathWithWeights.Count; i++)
                    {
                        int curr = i == 0 ? u : pathWithWeights[i-1].Item1;
                        int next = pathWithWeights[i].Item1;
                        int weight = pathWithWeights[i].Item2;
                        
                        adj[curr].Add(new Edge(next, weight, edgeIdCounter));
                        adj[next].Add(new Edge(curr, weight, edgeIdCounter));
                        edgeIdCounter++;
                    }
                }
            }
        }

        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        int[] edgePtr = new int[N];
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            while (edgePtr[u] < adj[u].Count)
            {
                var edge = adj[u][edgePtr[u]++];
                if (edge.Used) continue;

                edge.Used = true;
                foreach (var backEdge in adj[edge.To])
                {
                    if (backEdge.Id == edge.Id && !backEdge.Used)
                    {
                        backEdge.Used = true;
                        break;
                    }
                }

                stack.Push(edge.To);
                found = true;
                break;
            }

            if (!found)
            {
                circuit.Add(stack.Pop());
            }
        }

        var stdout = new StreamWriter(Console.OpenStandardOutput(), Encoding.UTF8, 1 << 16);
        stdout.WriteLine(circuit.Count);
        for (int i = 0; i < circuit.Count; i++)
        {
            stdout.Write(circuit[i]);
            if (i < circuit.Count - 1) stdout.Write(' ');
        }
        stdout.WriteLine();
        stdout.Flush();
    }

    private static int FindNearestOddNode(int startNode, bool[] matched, List<Edge>[] adj, int N)
    {
        PriorityQueue<int, int> pq = new PriorityQueue<int, int>();
        int[] dist = new int[N];
        for (int i = 0; i < N; i++) dist[i] = int.MaxValue;
        HashSet<int> visited = new HashSet<int>();

        dist[startNode] = 0;
        pq.Enqueue(startNode, 0);

        int nodesSearched = 0;
        int maxNodes = N > 10000 ? 5000 : N; 

        while (pq.Count > 0 && nodesSearched < maxNodes)
        {
            pq.TryDequeue(out int u, out int d);
            if (visited.Contains(u)) continue;
            visited.Add(u);
            nodesSearched++;

            if (u != startNode && (degrees[u] % 2 != 0) && !matched[u])
            {
                return u;
            }

            foreach (var edge in adj[u])
            {
                int newDist = d + edge.Weight;
                if (!visited.Contains(edge.To) && newDist < dist[edge.To])
                {
                    dist[edge.To] = newDist;
                    pq.Enqueue(edge.To, newDist);
                }
            }
        }

        for (int i = 0; i < N; i++)
        {
            if (i != startNode && (degrees[i] % 2 != 0) && !matched[i]) return i;
        }

        return -1;
    }

    private static List<ValueTuple<int, int>> GetShortestPathWithWeights(int startNode, int endNode, List<Edge>[] adj, int N)
    {
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        long[] dist = new long[N];
        int[] parent = new int[N];
        int[] edgeWeight = new int[N];
        for (int i = 0; i < N; i++)
        {
            dist[i] = long.MaxValue;
            parent[i] = -1;
        }

        dist[startNode] = 0;
        pq.Enqueue(startNode, 0);

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            if (u == endNode) break;

            foreach (var edge in adj[u])
            {
                if (dist[u] + edge.Weight < dist[edge.To])
                {
                    dist[edge.To] = dist[u] + edge.Weight;
                    parent[edge.To] = u;
                    edgeWeight[edge.To] = edge.Weight;
                    pq.Enqueue(edge.To, dist[edge.To]);
                }
            }
        }

        List<ValueTuple<int, int>> path = new List<ValueTuple<int, int>>();
        int curr = endNode;
        while (curr != startNode && curr != -1)
        {
            path.Add(new ValueTuple<int, int>(curr, edgeWeight[curr]));
            curr = parent[curr];
        }
        path.Reverse();
        return path;
    }
}
