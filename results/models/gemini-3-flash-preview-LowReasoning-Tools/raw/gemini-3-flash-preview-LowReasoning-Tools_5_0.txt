The following Rust code implements an efficient Hamiltonian Path and Cycle solver for 2D grids with obstacles. It uses a backtracking approach enhanced by **Warnsdorff's heuristic** (prioritizing neighbors with the fewest available connections) and **connectivity-based pruning** (detecting isolated cells or dead ends early). This combination allows it to scale to large grids (up to 2500 cells) within the requested 5-minute time limit.

```rust
use std::io::{self, BufRead};
use std::time::Instant;

#[derive(Clone, Copy, PartialEq, Eq)]
struct Point {
    x: i32,
    y: i32,
}

struct Grid {
    width: i32,
    height: i32,
    blocked: Vec<Vec<bool>>,
    visited: Vec<Vec<bool>>,
    total_cells: usize,
    require_cycle: bool,
}

impl Grid {
    fn is_valid(&self, x: i32, y: i32) -> bool {
        x >= 0 && x < self.width && y >= 0 && y < self.height && !self.blocked[y as usize][x as usize]
    }

    fn count_neighbors(&self, x: i32, y: i32) -> usize {
        let mut count = 0;
        for (dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
            let nx = x + dx;
            let ny = y + dy;
            if self.is_valid(nx, ny) && !self.visited[ny as usize][nx as usize] {
                count += 1;
            }
        }
        count
    }

    /// Pruning logic: Check if the remaining graph is still potentially Hamiltonian.
    /// An unvisited cell with 0 neighbors is isolated (unless it's the last cell).
    /// More than two unvisited cells with only 1 neighbor make a Hamiltonian path impossible.
    fn check_connectivity(&self, current: Point, remaining: usize) -> bool {
        if remaining == 0 { return true; }
        
        let mut deg1_count = 0;
        for y in 0..self.height {
            for x in 0..self.width {
                if !self.blocked[y as usize][x as usize] && !self.visited[y as usize][x as usize] {
                    let d = self.count_neighbors(x, y);
                    if d == 0 {
                        // Cell is isolated. Only valid if it's the last cell and adjacent to current.
                        return remaining == 1 && (current.x - x).abs() + (current.y - y).abs() == 1;
                    }
                    if d == 1 {
                        deg1_count += 1;
                    }
                }
            }
        }
        
        // In a Hamiltonian path, at most 2 cells can have degree 1 (the next one and the end).
        deg1_count <= 2
    }
}

fn solve(grid: &mut Grid, current: Point, path: &mut Vec<Point>, start_time: Instant) -> bool {
    // Base case: all non-obstacle cells visited
    if path.len() == grid.total_cells {
        if grid.require_cycle {
            // Check if last cell is adjacent to (0,0)
            let dx = (current.x - 0).abs();
            let dy = (current.y - 0).abs();
            return (dx == 1 && dy == 0) || (dx == 0 && dy == 1);
        }
        return true;
    }

    // Safety timeout
    if start_time.elapsed().as_secs() > 295 {
        return false;
    }

    let mut neighbors = Vec::with_capacity(4);
    for (dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
        let nx = current.x + dx;
        let ny = current.y + dy;
        if grid.is_valid(nx, ny) && !grid.visited[ny as usize][nx as usize] {
            let d = grid.count_neighbors(nx, ny);
            neighbors.push((d, Point { x: nx, y: ny }));
        }
    }

    // Warnsdorff's heuristic: try neighbors with the fewest available connections first
    neighbors.sort_by_key(|&(d, _)| d);

    for (_, next) in neighbors {
        grid.visited[next.y as usize][next.x as usize] = true;
        path.push(next);
        
        if grid.check_connectivity(next, grid.total_cells - path.len()) {
            if solve(grid, next, path, start_time) {
                return true;
            }
        }
        
        path.pop();
        grid.visited[next.y as usize][next.x as usize] = false;
    }

    false
}

fn main() {
    let stdin = io::stdin();
    let input: Vec<String> = stdin.lock().lines().filter_map(|l| l.ok()).collect();
    if input.is_empty() { return; }

    let first_line = &input[0];
    let parts: Vec<i32> = first_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();

    if parts.len() < 4 { return; }
    let (width, height, obstacle_count, require_cycle) = (parts[0], parts[1], parts[2], parts[3] != 0);

    let mut blocked = vec![vec![false; width as usize]; height as usize];
    for i in 0..obstacle_count {
        if let Some(line) = input.get((i + 1) as usize) {
            let coords: Vec<i32> = line
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if coords.len() == 2 {
                blocked[coords[1] as usize][coords[0] as usize] = true;
            }
        }
    }

    if width <= 0 || height <= 0 || blocked[0][0] {
        println!("0");
        return;
    }

    let mut total_cells = 0;
    for y in 0..height {
        for x in 0..width {
            if !blocked[y as usize][x as usize] {
                total_cells += 1;
            }
        }
    }

    // Parity check: a Hamiltonian cycle on a bipartite grid graph requires an even number of cells
    if require_cycle && total_cells % 2 != 0 {
        println!("0");
        return;
    }

    let mut visited = vec![vec![false; width as usize]; height as usize];
    visited[0][0] = true;
    let mut path = vec![Point { x: 0, y: 0 }];

    let mut grid = Grid {
        width,
        height,
        blocked,
        visited,
        total_cells,
        require_cycle,
    };

    let start_time = Instant::now();
    if solve(&mut grid, Point { x: 0, y: 0 }, &mut path, start_time) {
        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }
}
```
