Here is the complete Rust code for the Lunar Lander controller. It uses a memory-efficient map parser to find the target coordinates and a PD control strategy to land the ship safely.

```rust
use std::io::{self, BufRead};
use std::time::Instant;

struct Config {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    let mut line = String::new();

    // 1. Parse Initial Map Config
    if stdin_lock.read_line(&mut line).is_err() { return; }
    let config_parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if config_parts.len() < 5 { return; }
    let (_width, height, gravity, max_thrust, _fuel) = (
        config_parts[0] as usize,
        config_parts[1] as usize,
        config_parts[2],
        config_parts[3],
        config_parts[4],
    );
    line.clear();

    if stdin_lock.read_line(&mut line).is_err() { return; }
    let start_parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if start_parts.len() < 4 { return; }
    let (_start_x, _start_y, mut target_x, mut target_y) = (
        start_parts[0],
        start_parts[1],
        start_parts[2],
        start_parts[3],
    );
    line.clear();

    // 2. Parse Map Rows (Memory Efficiently)
    // We scan for 'T' or 'L' to find the landing pad center.
    let mut found_target = false;
    let mut sum_tx = 0.0;
    let mut sum_ty = 0.0;
    let mut count_t = 0;

    for r in 0..height {
        line.clear();
        if stdin_lock.read_line(&mut line).is_err() { break; }
        for (c, char) in line.trim().chars().enumerate() {
            if char == 'T' || char == 'L' {
                sum_tx += c as f64;
                sum_ty += r as f64;
                count_t += 1;
                found_target = true;
            }
        }
    }

    if found_target {
        target_x = sum_tx / count_t as f64;
        target_y = sum_ty / count_t as f64;
    }

    let config = Config {
        gravity, max_thrust, target_x, target_y,
    };

    // 3. Continuous State Updates and Control Loop
    let mut _last_output = Instant::now();
    loop {
        line.clear();
        match stdin_lock.read_line(&mut line) {
            Ok(0) => break, // EOF
            Ok(_) => {
                let state_parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
                if state_parts.len() < 7 { continue; }
                
                let state = State {
                    x: state_parts[0],
                    y: state_parts[1],
                    vx: state_parts[2],
                    vy: state_parts[3],
                    angle: state_parts[4],
                    angular_vel: state_parts[5],
                };

                // Control Logic
                let (thrust, turn) = calculate_control(&config, &state);

                // Output commands for the engine (at least 20Hz)
                println!("{:.4} {:.4}", thrust, turn);
                _last_output = Instant::now();
            }
            Err(_) => break,
        }
    }
}

fn calculate_control(config: &Config, state: &State) -> (f64, f64) {
    // PD Controller Constants
    let kp_y = 0.8;
    let kp_x = 0.15;
    let kp_angle = 5.0;
    let kd_angle = 1.2;

    // 1. Vertical Control (Thrust)
    // Maintain a target vertical velocity based on distance to the landing zone.
    let dist_y = config.target_y - state.y;
    let target_vy = if dist_y > 30.0 {
        4.0 // Fast descent
    } else if dist_y > 10.0 {
        2.0 // Slow descent
    } else if dist_y > 1.0 {
        1.0 // Final approach
    } else if dist_y > -1.0 {
        0.2 // Very slow landing
    } else {
        -0.5 // Hover/Slow climb if below target
    };

    let error_vy = target_vy - state.vy;
    let mut required_accel_y = error_vy * kp_y + config.gravity;
    
    // 2. Horizontal Control (Target Angle)
    // Tilt the lander to move towards the target X coordinate.
    let dist_x = config.target_x - state.x;
    let target_vx = (dist_x * kp_x).clamp(-8.0, 8.0);
    let error_vx = target_vx - state.vx;
    
    // Target tilt angle (radians) based on horizontal velocity error.
    let mut target_angle = (error_vx * 0.15).clamp(-0.5, 0.5);

    // Prioritize landing upright when very close to the ground.
    if dist_y < 1.5 {
        target_angle = 0.0;
    }

    // 3. Rotation Control (Turn)
    // Adjust turning rate to reach and maintain the target angle.
    let error_angle = target_angle - state.angle;
    let turn = (error_angle * kp_angle - state.angular_vel * kd_angle).clamp(-1.0, 1.0);

    // Compensate thrust for tilt to maintain vertical lift.
    let cos_angle = state.angle.cos().max(0.1);
    required_accel_y /= cos_angle;
    
    let thrust = (required_accel_y / config.max_thrust).clamp(0.0, 1.0);

    (thrust, turn)
}
```
