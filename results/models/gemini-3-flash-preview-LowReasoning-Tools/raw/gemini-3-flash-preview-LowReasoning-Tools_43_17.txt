struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // 1. Brick Pattern Parameters
    float2 uvScale = float2(8.0, 16.0);
    float2 mortarThickness = float2(0.05, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    // 2. Procedural Brick Logic
    float2 st = input.uv * uvScale;
    
    // Stagger every other row
    if (frac(st.y * 0.5) > 0.5) {
        st.x += 0.5;
    }
    
    float2 brick_uv = frac(st);
    
    // Determine if we are in the mortar or brick region
    // step(edge, x) returns 1 if x >= edge, 0 otherwise.
    // We want to be in "brick" if uv is between [mortarThickness/2, 1 - mortarThickness/2]
    float2 halfMortar = mortarThickness * 0.5;
    float2 is_brick_v = step(halfMortar, brick_uv) - step(1.0 - halfMortar, brick_uv);
    float is_brick = is_brick_v.x * is_brick_v.y;
    
    float3 baseColor = lerp(mortarColor, brickColor, is_brick);

    // 3. Shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;
    
    float3 finalColor = baseColor * shadingFactor;

    return float4(finalColor, 1.0);
}
