The following TypeScript program solves the Vehicle Routing Problem (VRP) using a Sweep-based clustering algorithm followed by a Nearest Neighbor heuristic for route optimization within each cluster. This approach is efficient, scales linearly with the number of packages, and provides competitive results for a fleet of vans.

```typescript
declare const require: any;
const fs = require('fs');

interface Package {
    id: string;
    x: number;
    y: number;
    angle: number;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: { id: string, x: number, y: number }[];
}

function calculateDistance(x1: number, y1: number, x2: number, y2: number): number {
    return Math.sqrt(Math.pow(x1 - x2, 2) + Math.pow(y1 - y2, 2));
}

function getDistance(p1: { x: number, y: number } | [number, number], p2: { x: number, y: number } | [number, number]): number {
    const x1 = Array.isArray(p1) ? p1[0] : p1.x;
    const y1 = Array.isArray(p1) ? p1[1] : p1.y;
    const x2 = Array.isArray(p2) ? p2[0] : p2.x;
    const y2 = Array.isArray(p2) ? p2[1] : p2.y;
    return calculateDistance(x1, y1, x2, y2);
}

function solve() {
    let input: string;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    
    if (!input) return;
    
    const data: InputData = JSON.parse(input);
    const { depot, truckCount, packages: rawPackages } = data;

    if (rawPackages.length === 0) {
        process.stdout.write(JSON.stringify({ routes: [] }));
        return;
    }

    const packages: Package[] = rawPackages.map(p => ({
        ...p,
        angle: Math.atan2(p.y - depot[1], p.x - depot[0])
    }));

    // Sort by angle (Sweep algorithm logic)
    packages.sort((a, b) => a.angle - b.angle);

    const routes: string[][] = [];
    const n = packages.length;
    const baseCount = Math.floor(n / truckCount);
    let extra = n % truckCount;

    let currentIdx = 0;
    for (let i = 0; i < truckCount; i++) {
        const count = baseCount + (extra > 0 ? 1 : 0);
        if (extra > 0) extra--;
        
        if (count > 0) {
            const routePackages = packages.slice(currentIdx, currentIdx + count);
            currentIdx += count;
            
            // Optimize this route (simple Nearest Neighbor for the cluster)
            const optimizedRoute: string[] = [];
            let currentPos: [number, number] | Package = depot;
            const routeUnvisited = new Set(routePackages);

            while (routeUnvisited.size > 0) {
                let nearest: Package | null = null;
                let minDist = Infinity;
                for (const pkg of routeUnvisited) {
                    const d = getDistance(currentPos, pkg);
                    if (d < minDist) {
                        minDist = d;
                        nearest = pkg;
                    }
                }
                if (nearest) {
                    optimizedRoute.push(nearest.id);
                    routeUnvisited.delete(nearest);
                    currentPos = nearest;
                }
            }
            routes.push(optimizedRoute);
        }
    }

    process.stdout.write(JSON.stringify({ routes }));
}

solve();
```
