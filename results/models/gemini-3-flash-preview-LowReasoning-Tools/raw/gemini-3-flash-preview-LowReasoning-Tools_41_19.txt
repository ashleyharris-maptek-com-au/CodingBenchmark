               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0      ; model
               OpMemberDecorate %UBO 1 Offset 64     ; view
               OpMemberDecorate %UBO 2 Offset 128    ; projection
               OpMemberDecorate %UBO 3 Offset 192    ; lightPos
               OpMemberDecorate %UBO 4 Offset 208    ; cameraPos
               OpMemberDecorate %UBO 5 Offset 224    ; params
               OpDecorate %uboVar DescriptorSet 0
               OpDecorate %uboVar Binding 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
     %uboVar = OpVariable %_ptr_Uniform_UBO Uniform

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input
         %uv = OpVariable %_ptr_Input_v2float Input
    %tangent = OpVariable %_ptr_Input_v3float Input
      %color = OpVariable %_ptr_Input_v3float Input

%_ptr_Output_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %_ptr_Output_v4float Output

%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4

    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
  %float_0_1 = OpConstant %float 0.1
 %float_0_05 = OpConstant %float 0.05
   %float_32 = OpConstant %float 32

    %v3ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05
    %v3specCol = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Main
       %main = OpFunction %void None %3
          %5 = OpLabel

; Load Inputs
    %inWorldPos = OpLoad %v3float %worldPos
    %inNormal   = OpLoad %v3float %normal
    %inColor    = OpLoad %v3float %color

; Load Uniforms
 %ptrLightPos = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_3
    %lightPos4 = OpLoad %v4float %ptrLightPos
    %lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

 %ptrCameraPos = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_4
    %cameraPos4 = OpLoad %v4float %ptrCameraPos
    %cameraPos3 = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2

; Normalize Normal
          %N = OpExtInst %v3float %1 Normalize %inNormal

; Compute L (Light Vector)
       %Ldir = OpFSub %v3float %lightPos3 %inWorldPos
          %L = OpExtInst %v3float %1 Normalize %Ldir

; Compute V (View Vector)
       %Vdir = OpFSub %v3float %cameraPos3 %inWorldPos
          %V = OpExtInst %v3float %1 Normalize %Vdir

; Diffuse: max(dot(N, L), 0) * inColor
      %dotNL = OpDot %float %N %L
     %diffCo = OpExtInst %float %1 FMax %dotNL %float_0
    %diffuse = OpFMul %v3float %inColor %diffCo

; Specular: max(dot(R, V), 0)^32 * (1, 1, 1)
; R = reflect(-L, N)
     %negL = OpFNegate %v3float %L
        %R = OpExtInst %v3float %1 Reflect %negL %N
    %dotRV = OpDot %float %R %V
   %specCo_ = OpExtInst %float %1 FMax %dotRV %float_0
   %specCo  = OpExtInst %float %1 Pow %specCo_ %float_32
   %specular = OpFMul %v3float %v3specCol %specCo

; Combine: ambient + diffuse + specular
      %sum1 = OpFAdd %v3float %v3ambient %diffuse
      %sum2 = OpFAdd %v3float %sum1 %specular

; Clamp sum to [0, 1]
    %v3zero = OpConstantComposite %v3float %float_0 %float_0 %float_0
     %v3one = OpConstantComposite %v3float %float_1 %float_1 %float_1
     %final = OpExtInst %v3float %1 FClamp %sum2 %v3zero %v3one

; Output color with alpha 1.0
  %outColor3 = OpVectorShuffle %v4float %final %final 0 1 2
  %resColor  = OpCompositeInsert %v4float %float_1 %outColor3 3
               OpStore %outColor %resColor

               OpReturn
               OpFunctionEnd
