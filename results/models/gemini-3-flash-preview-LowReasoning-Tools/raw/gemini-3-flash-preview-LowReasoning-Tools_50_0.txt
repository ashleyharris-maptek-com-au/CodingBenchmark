#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);

    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;

    for (uint j = 0; j < N; ++j) {
        if (i == j) continue;

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        float dist = distance(pos, otherPos);
        if (dist <= 0.0) continue;

        if (dist < sepRadius) {
            separation += (pos - otherPos) / dist;
            sepCount++;
        }
        if (dist < alignRadius) {
            alignment += otherVel;
            alignCount++;
        }
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }

    vec3 acceleration = vec3(0.0);

    if (sepCount > 0) {
        acceleration += (separation / float(sepCount)) * sepWeight;
    }
    if (alignCount > 0) {
        acceleration += (alignment / float(alignCount)) * alignWeight;
    }
    if (cohCount > 0) {
        acceleration += ((cohesion / float(cohCount)) - pos) * cohWeight;
    }

    // Boundary force
    vec3 boundAcc = vec3(0.0);
    float limit = boundSize * 0.4;
    float scale = boundSize * 0.5;
    
    if (pos.x < -limit) boundAcc.x = boundForce;
    else if (pos.x > limit) boundAcc.x = -boundForce;
    
    if (pos.y < -limit) boundAcc.y = boundForce;
    else if (pos.y > limit) boundAcc.y = -boundForce;
    
    if (pos.z < -limit) boundAcc.z = boundForce;
    else if (pos.z > limit) boundAcc.z = -boundForce;

    acceleration += boundAcc;

    // Update velocity
    vec3 newVel = vel + acceleration * dt;
    float speed = length(newVel);
    if (speed > maxSpeed) {
        newVel = (newVel / speed) * maxSpeed;
    }

    // Update position
    vec3 newPos = pos + newVel * dt;

    outData[i * 2] = vec4(newPos, 0.0);
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
