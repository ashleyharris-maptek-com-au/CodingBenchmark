I have implemented a robust and efficient C++ solver for the Job-Shop Scheduling problem. The solver employs a simulation-based approach inspired by the **Giffler-Thompson algorithm** to generate active schedules, which are generally efficient for this class of problems.

### Key Features:
- **Heuristic Strategy**: Uses the **Most Work Remaining (MWR)** priority rule to break ties in the conflict set, which often leads to shorter makespans.
- **Scalability**: Designed to handle both trivial and extreme cases (up to 50,000+ tasks). The complexity is approximately $O(T \times J)$, where $T$ is the total number of tasks and $J$ is the number of jobs. For the extreme constraints provided, this results in about 50 million operations, which executes efficiently within 1-2 seconds in C++.
- **Format Compliance**: Adheres strictly to the specified input and output formats.
- **Robustness**: Correctness was verified against the provided example, ensuring that all constraints (job sequence, machine exclusivity, and non-preemption) are strictly followed.

### Final C++ Code:

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

/**
 * Job-Shop Scheduling Solver
 * 
 * Approach:
 * Uses a priority-rule based simulation.
 * Specifically, it employs the Giffler-Thompson algorithm logic to generate active schedules.
 * Tie-breaking is done using the 'Most Work Remaining' (MWR) priority rule.
 * 
 * Efficiency:
 * O(T * J) complexity where T is total tasks and J is number of jobs.
 * For T=50k, J=1000, this is ~50M operations, well within the 1-2 second limit for C++.
 */

struct Task {
    int job_id;
    int task_id;
    int machine_id;
    int duration;
    long long start_time = -1;
};

struct Job {
    std::vector<Task> tasks;
};

int main() {
    // Speed up I/O for large scale problems
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_jobs, num_machines;
    if (!(std::cin >> num_jobs >> num_machines)) return 0;

    std::vector<Job> jobs(num_jobs);
    int total_tasks = 0;
    for (int i = 0; i < num_jobs; ++i) {
        int num_tasks_in_job;
        if (!(std::cin >> num_tasks_in_job)) break;
        jobs[i].tasks.resize(num_tasks_in_job);
        total_tasks += num_tasks_in_job;
        for (int j = 0; j < num_tasks_in_job; ++j) {
            jobs[i].tasks[j].job_id = i;
            jobs[i].tasks[j].task_id = j;
            std::cin >> jobs[i].tasks[j].machine_id >> jobs[i].tasks[j].duration;
        }
    }

    std::vector<int> next_task_idx(num_jobs, 0);
    std::vector<long long> job_ready_at(num_jobs, 0);
    std::vector<long long> machine_ready_at(num_machines, 0);
    
    // Calculate total remaining duration for each job for the MWR priority rule
    std::vector<long long> job_remaining_duration(num_jobs, 0);
    for (int i = 0; i < num_jobs; ++i) {
        for (const auto& t : jobs[i].tasks) {
            job_remaining_duration[i] += t.duration;
        }
    }

    // Keep track of which jobs still have tasks to be scheduled
    std::vector<int> active_jobs;
    for (int i = 0; i < num_jobs; ++i) {
        if (!jobs[i].tasks.empty()) {
            active_jobs.push_back(i);
        }
    }

    int scheduled_count = 0;
    while (scheduled_count < total_tasks) {
        long long min_finish = std::numeric_limits<long long>::max();
        int m_star = -1;

        // 1. Find the machine m_star that can complete its earliest task among all available tasks first
        for (int i : active_jobs) {
            const Task& t = jobs[i].tasks[next_task_idx[i]];
            long long est = std::max(job_ready_at[i], machine_ready_at[t.machine_id]);
            long long ect = est + t.duration;
            if (ect < min_finish) {
                min_finish = ect;
                m_star = t.machine_id;
            }
        }

        if (m_star == -1) break;

        // 2. Identify the conflict set: all tasks ready for m_star that can start before min_finish
        int best_j = -1;
        long long max_rem = -1;
        int best_j_idx_in_active = -1;

        for (int idx = 0; idx < (int)active_jobs.size(); ++idx) {
            int i = active_jobs[idx];
            const Task& t = jobs[i].tasks[next_task_idx[i]];
            if (t.machine_id == m_star) {
                long long est = std::max(job_ready_at[i], machine_ready_at[m_star]);
                if (est < min_finish) {
                    // Selection using Most Work Remaining (MWR) rule
                    if (job_remaining_duration[i] > max_rem) {
                        max_rem = job_remaining_duration[i];
                        best_j = i;
                        best_j_idx_in_active = idx;
                    } else if (job_remaining_duration[i] == max_rem) {
                        if (best_j == -1 || i < best_j) {
                            best_j = i;
                            best_j_idx_in_active = idx;
                        }
                    }
                }
            }
        }

        if (best_j != -1) {
            int t_idx = next_task_idx[best_j];
            Task& t = jobs[best_j].tasks[t_idx];
            long long est = std::max(job_ready_at[best_j], machine_ready_at[m_star]);
            t.start_time = est;
            
            long long finish_t = est + t.duration;
            job_ready_at[best_j] = finish_t;
            machine_ready_at[m_star] = finish_t;
            job_remaining_duration[best_j] -= t.duration;
            
            next_task_idx[best_j]++;
            scheduled_count++;
            
            // Remove job from active list if all its tasks are scheduled
            if (next_task_idx[best_j] == (int)jobs[best_j].tasks.size()) {
                active_jobs.erase(active_jobs.begin() + best_j_idx_in_active);
            }
        }
    }

    // Output result according to specified format
    for (int i = 0; i < num_jobs; ++i) {
        for (int j = 0; j < (int)jobs[i].tasks.size(); ++j) {
            std::cout << jobs[i].tasks[j].start_time << " " << jobs[i].tasks[j].machine_id << (j == (int)jobs[i].tasks.size() - 1 ? "" : " ");
        }
        std::cout << "\n";
    }

    return 0;
}
```
