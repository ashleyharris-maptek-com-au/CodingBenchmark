I have implemented a robust Tetris player in C# that uses an AI heuristic to optimize piece placement. The player uses a search algorithm to evaluate all possible moves for a given piece and selects the one that maximizes a score based on board height, lines cleared, number of holes, and surface bumpiness.

### Key Features:
- **Optimal Strategy:** Uses the "El-Tetris" heuristic weights, known for effective marathon play.
- **Adaptive Logic:** The AI evaluates multiple board factors to prioritize line clears and maintain a flat, hole-free board.
- **Efficiency:** The simulator and board representation are optimized to ensure decisions are made well within the 5-second time limit.
- **Robustness:** Handles variable board widths and long sequences of pieces from standard input.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class TetrisPlayer
{
    public static void Main(string[] args)
    {
        // Read initial board parameters: width height num_pieces
        string input = Console.ReadLine();
        if (string.IsNullOrEmpty(input)) return;
        string[] initialParams = input.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (initialParams.Length < 3) return;

        int width = int.Parse(initialParams[0]);
        int height = int.Parse(initialParams[1]);
        int numPieces = int.Parse(initialParams[2]);

        Board board = new Board(width, height);
        AI ai = new AI();

        for (int i = 0; i < numPieces; i++)
        {
            // Read piece type (I, O, T, S, Z, J, L)
            string pieceTypeLine = Console.ReadLine();
            if (pieceTypeLine == null) break;
            string pieceType = pieceTypeLine.Trim();
            if (string.IsNullOrEmpty(pieceType)) break;

            // Determine the best move (rotation and column)
            Move bestMove = ai.GetBestMove(board, pieceType);
            Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");

            // Read the result from the game engine (ok, cleared_N, gameover)
            string result = Console.ReadLine()?.Trim();
            if (result == null || result == "gameover") break;

            // Update the local board state
            board.ApplyMove(pieceType, bestMove.Rotation, bestMove.Column);
        }
    }
}

public class Move
{
    public int Rotation { get; set; }
    public int Column { get; set; }
    public double Score { get; set; }
}

public class Board
{
    public int Width { get; }
    public int Height { get; }
    public bool[,] Grid { get; }

    public Board(int width, int height)
    {
        Width = width;
        Height = height;
        Grid = new bool[width, height];
    }

    public Board(Board other)
    {
        Width = other.Width;
        Height = other.Height;
        Grid = (bool[,])other.Grid.Clone();
    }

    public void ApplyMove(string pieceType, int rotation, int column)
    {
        int[,] shape = Piece.GetShape(pieceType, rotation);
        int rWidth = shape.GetLength(0);
        int rHeight = shape.GetLength(1);

        int row = FindDropRow(shape, column);

        for (int x = 0; x < rWidth; x++)
        {
            for (int y = 0; y < rHeight; y++)
            {
                if (shape[x, y] != 0)
                {
                    int targetX = column + x;
                    int targetY = row + y;
                    if (targetX >= 0 && targetX < Width && targetY >= 0 && targetY < Height)
                    {
                        Grid[targetX, targetY] = true;
                    }
                }
            }
        }

        ClearLines();
    }

    public int FindDropRow(int[,] shape, int column)
    {
        int rHeight = shape.GetLength(1);

        // Start from the top and drop down until collision
        for (int y = Height - rHeight; y >= 0; y--)
        {
            if (HasCollision(shape, column, y))
            {
                return y + 1;
            }
        }
        return 0;
    }

    public bool HasCollision(int[,] shape, int column, int row)
    {
        int rWidth = shape.GetLength(0);
        int rHeight = shape.GetLength(1);

        for (int dx = 0; dx < rWidth; dx++)
        {
            for (int dy = 0; dy < rHeight; dy++)
            {
                if (shape[dx, dy] != 0)
                {
                    int tx = column + dx;
                    int ty = row + dy;
                    if (tx < 0 || tx >= Width || ty < 0 || ty >= Height || (ty < Height && Grid[tx, ty]))
                    {
                        return true;
                    }
                }
            }
        }
        return false;
    }

    private void ClearLines()
    {
        for (int y = 0; y < Height; y++)
        {
            bool full = true;
            for (int x = 0; x < Width; x++)
            {
                if (!Grid[x, y])
                {
                    full = false;
                    break;
                }
            }

            if (full)
            {
                for (int ty = y; ty < Height - 1; ty++)
                {
                    for (int tx = 0; tx < Width; tx++)
                    {
                        Grid[tx, ty] = Grid[tx, ty + 1];
                    }
                }
                for (int tx = 0; tx < Width; tx++)
                {
                    Grid[tx, Height - 1] = false;
                }
                y--; // Re-check the same row index
            }
        }
    }
}

public static class Piece
{
    // Define the 4 rotations for each piece type
    private static readonly Dictionary<string, int[][,]> Shapes = new Dictionary<string, int[][,]>()
    {
        { "I", new int[][,] { 
            new int[,] {{1}, {1}, {1}, {1}}, 
            new int[,] {{1, 1, 1, 1}} 
        } },
        { "O", new int[][,] { 
            new int[,] {{1, 1}, {1, 1}} 
        } },
        { "T", new int[][,] { 
            new int[,] {{0, 1}, {1, 1}, {0, 1}}, 
            new int[,] {{1, 1, 1}, {0, 1, 0}},
            new int[,] {{1, 0}, {1, 1}, {1, 0}},
            new int[,] {{0, 1, 0}, {1, 1, 1}}
        } },
        { "S", new int[][,] { 
            new int[,] {{1, 0}, {1, 1}, {0, 1}}, 
            new int[,] {{0, 1, 1}, {1, 1, 0}}
        } },
        { "Z", new int[][,] { 
            new int[,] {{0, 1}, {1, 1}, {1, 0}}, 
            new int[,] {{1, 1, 0}, {0, 1, 1}}
        } },
        { "J", new int[][,] { 
            new int[,] {{1, 1}, {1, 0}, {1, 0}}, 
            new int[,] {{1, 1, 1}, {0, 0, 1}},
            new int[,] {{0, 1}, {0, 1}, {1, 1}},
            new int[,] {{1, 0, 0}, {1, 1, 1}}
        } },
        { "L", new int[][,] { 
            new int[,] {{1, 0}, {1, 0}, {1, 1}}, 
            new int[,] {{0, 0, 1}, {1, 1, 1}},
            new int[,] {{1, 1}, {0, 1}, {0, 1}},
            new int[,] {{1, 1, 1}, {1, 0, 0}}
        } }
    };

    public static int[,] GetShape(string type, int rotation)
    {
        if (!Shapes.ContainsKey(type)) return new int[,] {{1}};
        var variants = Shapes[type];
        return variants[rotation % variants.Length];
    }

    public static int GetRotationCount(string type)
    {
        if (!Shapes.ContainsKey(type)) return 1;
        return Shapes[type].Length;
    }
}

public class AI
{
    // Heuristic weights for board evaluation
    private const double HeightWeight = -0.510066;
    private const double LinesWeight = 0.760666;
    private const double HolesWeight = -0.35663;
    private const double BumpinessWeight = -0.184483;

    public Move GetBestMove(Board board, string pieceType)
    {
        Move bestMove = null;
        int rotations = Piece.GetRotationCount(pieceType);

        for (int r = 0; r < rotations; r++)
        {
            int[,] shape = Piece.GetShape(pieceType, r);
            int rWidth = shape.GetLength(0);

            // Try every possible column placement for the given rotation
            for (int c = 0; c <= board.Width - rWidth; c++)
            {
                Board nextBoard = new Board(board);
                int linesCleared = SimulateMove(nextBoard, pieceType, r, c);
                double score = EvaluateBoard(nextBoard, linesCleared);

                if (bestMove == null || score > bestMove.Score)
                {
                    bestMove = new Move { Rotation = r, Column = c, Score = score };
                }
            }
        }

        return bestMove ?? new Move { Rotation = 0, Column = 0 };
    }

    private int SimulateMove(Board board, string pieceType, int rotation, int column)
    {
        int[,] shape = Piece.GetShape(pieceType, rotation);
        int rWidth = shape.GetLength(0);
        int rHeight = shape.GetLength(1);

        int row = board.FindDropRow(shape, column);
        int linesCleared = 0;

        for (int x = 0; x < rWidth; x++)
        {
            for (int y = 0; y < rHeight; y++)
            {
                if (shape[x, y] != 0)
                {
                    int tx = column + x;
                    int ty = row + y;
                    if (tx >= 0 && tx < board.Width && ty >= 0 && ty < board.Height)
                    {
                        board.Grid[tx, ty] = true;
                    }
                }
            }
        }

        // Simulate line clearing and count cleared lines
        for (int y = 0; y < board.Height; y++)
        {
            bool full = true;
            for (int x = 0; x < board.Width; x++)
            {
                if (!board.Grid[x, y]) { full = false; break; }
            }
            if (full)
            {
                linesCleared++;
                for (int ty = y; ty < board.Height - 1; ty++)
                {
                    for (int tx = 0; tx < board.Width; tx++)
                        board.Grid[tx, ty] = board.Grid[tx, ty + 1];
                }
                for (int tx = 0; tx < board.Width; tx++)
                    board.Grid[tx, board.Height - 1] = false;
                y--;
            }
        }

        return linesCleared;
    }

    private double EvaluateBoard(Board board, int linesCleared)
    {
        int[] heights = GetHeights(board);
        int aggregateHeight = heights.Sum();
        int holes = GetHoles(board, heights);
        int bumpiness = GetBumpiness(heights);

        return (HeightWeight * aggregateHeight) +
               (LinesWeight * linesCleared) +
               (HolesWeight * holes) +
               (BumpinessWeight * bumpiness);
    }

    private int[] GetHeights(Board board)
    {
        int[] heights = new int[board.Width];
        for (int x = 0; x < board.Width; x++)
        {
            for (int y = board.Height - 1; y >= 0; y--)
            {
                if (board.Grid[x, y])
                {
                    heights[x] = y + 1;
                    break;
                }
            }
        }
        return heights;
    }

    private int GetHoles(Board board, int[] heights)
    {
        int holes = 0;
        for (int x = 0; x < board.Width; x++)
        {
            for (int y = 0; y < heights[x]; y++)
            {
                if (!board.Grid[x, y]) holes++;
            }
        }
        return holes;
    }

    private int GetBumpiness(int[] heights)
    {
        int bumpiness = 0;
        for (int x = 0; x < heights.Length - 1; x++)
        {
            bumpiness += Math.Abs(heights[x] - heights[x + 1]);
        }
        return bumpiness;
    }
}
```
