```python
import math

# Persistence across calls
class AutopilotState:
    def __init__(self):
        self.reset()

    def reset(self):
        self.prev_errors = {}
        self.integrals = {}
        self.last_t = None
        self.phase = "APPROACH" # INTERCEPT, APPROACH, FLARE, TOUCHDOWN, ROLLOUT
        self.go_around = False
        self.initialized = False

state_obj = AutopilotState()

def get_valid_value(state, keys, default=0.0):
    valid_values = []
    # Check 3 redundant sensors
    for i in range(1, 4):
        val_key = f"{keys}_{i}"
        valid_key = f"{keys}_{i}_valid"
        
        # Special case for keys that might have different suffixes
        if keys == "ias":
            val_key = f"ias_{i}_kt"
            valid_key = f"ias_{i}_valid"
        elif keys == "glideslope":
            val_key = f"glideslope_{i}_dots"
            valid_key = f"glideslope_{i}_valid"
        elif keys == "localizer":
            val_key = f"localizer_{i}_dots"
            valid_key = f"localizer_{i}_valid"
        elif keys == "irs_pitch":
            val_key = f"irs_{i}_pitch_deg"
            valid_key = f"irs_{i}_valid"
        elif keys == "irs_roll":
            val_key = f"irs_{i}_roll_deg"
            valid_key = f"irs_{i}_valid"
        elif keys == "irs_vs":
            val_key = f"irs_{i}_vs_fpm"
            valid_key = f"irs_{i}_valid"

        if val_key in state and state.get(valid_key, True):
            val = state[val_key]
            if math.isfinite(val):
                valid_values.append(val)

    if not valid_values:
        # Fallback to target or ground truth keys if they exist
        if keys == "glideslope" and "target_gs_dots" in state:
            return state["target_gs_dots"]
        if keys == "localizer" and "target_loc_dots" in state:
            return state["target_loc_dots"]
        if keys in state:
            return state[keys]
        return default

    # Return median for robustness against single-sensor failure/bias
    return sorted(valid_values)[len(valid_values) // 2]

def pid_control(name, target, current, dt, kp, ki, kd, limit=None):
    error = target - current

    if name not in state_obj.integrals:
        state_obj.integrals[name] = 0.0
    if name not in state_obj.prev_errors:
        state_obj.prev_errors[name] = 0.0

    state_obj.integrals[name] += error * dt
    # Anti-windup: limit integral contribution
    if limit is not None:
        state_obj.integrals[name] = max(-limit*2, min(limit*2, state_obj.integrals[name]))

    derivative = (error - state_obj.prev_errors[name]) / dt if dt > 0 else 0.0  
    state_obj.prev_errors[name] = error

    output = kp * error + ki * state_obj.integrals[name] + kd * derivative      

    if limit is not None:
        output = max(-limit, min(limit, output))

    return output

def autopilot_step(state: dict, dt: float) -> dict:
    global state_obj
    # Reset if first call or if there's a large jump in time (sim restart)      
    if not state_obj.initialized or dt > 1.0:
        state_obj.reset()
        state_obj.initialized = True

    # Sensor fusion
    radio_alt = state.get("radio_alt_ft", 5000)
    if radio_alt > 2500 or radio_alt < -5:
        # Fallback to baro or GPS
        radio_alt = state.get("baro_alt_1_ft", state.get("gps_1_alt_ft", 5000)) 

    gs_dots = get_valid_value(state, "glideslope", 0.0)
    loc_dots = get_valid_value(state, "localizer", 0.0)
    ias = get_valid_value(state, "ias", 140)
    target_ias = state.get("target_vref_kt", 137)

    pitch = get_valid_value(state, "irs_pitch", 0.0)
    roll = get_valid_value(state, "irs_roll", 0.0)
    vs = get_valid_value(state, "irs_vs", 0.0)

    on_ground = state.get("on_ground", False) or state.get("weight_on_wheels", False) or (radio_alt < 2 and vs > -50)

    commands = {}

    # Phase Transitions
    if state_obj.go_around:
        if radio_alt > 1400: # Near 1500ft goal
            state_obj.go_around = False
    elif on_ground:
        state_obj.phase = "ROLLOUT"
    elif radio_alt < 50:
        state_obj.phase = "FLARE"
    else:
        state_obj.phase = "APPROACH"

    # Go-Around Logic (Safety checks)
    if not state_obj.go_around and radio_alt < 500 and not on_ground:
        # Unstable approach criteria
        if abs(gs_dots) > 2.5 or abs(loc_dots) > 2.5:
            state_obj.go_around = True
        if radio_alt < 200 and vs < -1500:
            state_obj.go_around = True
        if ias < target_ias - 20: # Severe speed loss (e.g. windshear)
            state_obj.go_around = True

    if state_obj.go_around:
        # TOGA
        commands["throttle_1"] = 1.0
        commands["throttle_2"] = 1.0
        # Pitch up to 12 degrees
        commands["elevator"] = pid_control("pitch_ga", 12.0, pitch, dt, 0.1, 0.02, 0.05, 0.8)
        commands["aileron"] = pid_control("roll_ga", 0.0, roll, dt, 0.04, 0.01, 0.05, 0.5)
        commands["gear"] = 0.0
        commands["flaps"] = 15.0
        commands["speedbrake"] = 0.0
        return commands

    # Default common outputs for approach
    commands["gear"] = 1.0
    commands["flaps"] = 30.0
    commands["speedbrake"] = 0.0

    if state_obj.phase == "ROLLOUT":
        commands["throttle_1"] = 0.0
        commands["throttle_2"] = 0.0
        commands["reverse_1"] = 1.0
        commands["reverse_2"] = 1.0
        commands["spoiler"] = 1.0
        commands["autobrake"] = 1.0
        commands["elevator"] = -0.3 # Nose down force
        # Use rudder to stay on centerline (Localizer)
        commands["rudder"] = pid_control("rudder_rollout", 0.0, loc_dots, dt, 0.6, 0.05, 0.2, 1.0)

    elif state_obj.phase == "FLARE":
        commands["throttle_1"] = 0.0
        commands["throttle_2"] = 0.0
        # Target a gentle touchdown vertical speed
        target_vs = -150 # fpm
        commands["elevator"] = pid_control("vs_flare", target_vs, vs, dt, 0.0012, 0.0003, 0.0001, 0.9)
        # Keep wings level
        commands["aileron"] = pid_control("roll_flare", 0.0, roll, dt, 0.05, 0.01, 0.05, 0.5)

    else: # APPROACH or INTERCEPT
        # Lateral: Roll to track Localizer
        # If loc_dots > 0 (right of path), we need negative roll (left)
        target_roll = pid_control("loc", 0.0, loc_dots, dt, 20.0, 0.8, 12.0, 25.0)
        commands["aileron"] = pid_control("roll", -target_roll, roll, dt, 0.05, 0.01, 0.1, 0.6)

        # Vertical: Pitch to track Glideslope
        # If gs_dots > 0 (above path), we need more negative pitch (nose down)  
        target_pitch_adj = pid_control("gs", 0.0, gs_dots, dt, 6.0, 0.3, 3.0, 10.0)
        nominal_pitch = -2.5 # Typical pitch for 3 deg glide
        commands["elevator"] = pid_control("pitch", nominal_pitch - target_pitch_adj, pitch, dt, 0.12, 0.03, 0.05, 0.8)

        # Speed: Throttle to track target IAS
        thrust_cmd = pid_control("speed", target_ias, ias, dt, 0.15, 0.03, 0.05, 1.0)
        base_thrust = 0.55
        throttle = max(0.0, min(1.0, base_thrust + thrust_cmd))
        commands["throttle_1"] = throttle
        commands["throttle_2"] = throttle

        # Configuration based on altitude
        if radio_alt > 1500:
            commands["flaps"] = 15.0
            commands["gear"] = 0.0
        else:
            commands["flaps"] = 30.0
            commands["gear"] = 1.0

    return commands
```
