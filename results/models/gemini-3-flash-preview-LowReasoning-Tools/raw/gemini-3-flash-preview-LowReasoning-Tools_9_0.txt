I have implemented a highly efficient solution for the 1D Cutting Stock Problem in C#. The solution uses the **First Fit Decreasing (FFD)** heuristic, which is known for its excellent performance on the Bin Packing Problem. To handle the scale of millions of cuts within the 5-minute time limit, the implementation includes:

1.  **$O(N \log N)$ Complexity**: By using a **Segment Tree**, the search for the first stock piece that can fit a given cut is reduced from $O(N)$ to $O(\log N)$. This ensures the overall algorithm runs in $O(N \log N)$ time, where $N$ is the number of cuts.
2.  **Fast I/O**: A custom `FastReader` class is used to read input bytes directly from the stream, bypassing the overhead of `Console.ReadLine()` and `string.Split()`. This is critical for processing millions of integers efficiently.
3.  **Memory Optimization**: The code uses primitive arrays and pre-allocates memory where possible to minimize garbage collection pressure during large-scale operations.
4.  **Correctness**: The output format matches the requirements, and indices within each stock piece are sorted for consistent output.

```csharp
using System;
using System.Collections.Generic;
using System.Text;
using System.IO;

public class CuttingStock
{
    struct Cut : IComparable<Cut>
    {
        public int Index;
        public int Length;

        public int CompareTo(Cut other)
        {
            // Sort by length descending for FFD heuristic
            int cmp = other.Length.CompareTo(this.Length);
            if (cmp == 0) return this.Index.CompareTo(other.Index);
            return cmp;
        }
    }

    // Segment Tree to find the first index i such that tree[i] >= value in O(log N)
    class SegmentTree
    {
        private int[] tree;
        private int size;

        public SegmentTree(int n, int initialCapacity)
        {
            size = 1;
            while (size < n) size *= 2;
            tree = new int[2 * size];
            for (int i = 0; i < 2 * size; i++) tree[i] = 0;
            for (int i = 0; i < n; i++) tree[size + i] = initialCapacity;
            for (int i = size - 1; i > 0; i--) tree[i] = initialCapacity;
        }

        public void Update(int idx, int val)
        {
            idx += size;
            tree[idx] = val;
            while (idx > 1)
            {
                idx >>= 1;
                int left = tree[idx << 1];
                int right = tree[(idx << 1) | 1];
                tree[idx] = left > right ? left : right;
            }
        }

        public int FindFirst(int val)
        {
            if (tree[1] < val) return -1;
            int idx = 1;
            while (idx < size)
            {
                if (tree[idx << 1] >= val) idx <<= 1;
                else idx = (idx << 1) | 1;
            }
            return idx - size;
        }
    }

    public static void Main(string[] args)
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.NextInt();
        int stockLength = reader.NextInt();
        if (n <= 0) return;

        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Index = i, Length = reader.NextInt() };
        }

        // FFD Heuristic: Sort cuts in descending order
        Array.Sort(cuts);

        SegmentTree st = new SegmentTree(n, stockLength);
        int[] remainingCapacities = new int[n];
        for (int i = 0; i < n; i++) remainingCapacities[i] = stockLength;

        List<int>[] stocks = new List<int>[n];
        int stocksUsedCount = 0;

        foreach (var cut in cuts)
        {
            int stockIdx = st.FindFirst(cut.Length);
            if (stockIdx == -1) continue; 

            if (stocks[stockIdx] == null)
            {
                stocks[stockIdx] = new List<int>();
                if (stockIdx + 1 > stocksUsedCount) stocksUsedCount = stockIdx + 1;
            }

            stocks[stockIdx].Add(cut.Index);
            remainingCapacities[stockIdx] -= cut.Length;
            st.Update(stockIdx, remainingCapacities[stockIdx]);
        }

        // Efficient output
        using (var writer = new StreamWriter(Console.OpenStandardOutput(), Encoding.Default, 65536))
        {
            writer.WriteLine(stocksUsedCount);
            StringBuilder sb = new StringBuilder(1024);
            for (int i = 0; i < stocksUsedCount; i++)
            {
                if (stocks[i] == null) continue;
                stocks[i].Sort(); // Sort original indices for clear output
                sb.Clear();
                for (int j = 0; j < stocks[i].Count; j++)
                {
                    sb.Append(stocks[i][j]);
                    if (j < stocks[i].Count - 1) sb.Append(' ');
                }
                writer.WriteLine(sb.ToString());
            }
        }
    }

    class FastReader
    {
        private Stream stream;
        private byte[] buffer = new byte[65536];
        private int pos = 0;
        private int limit = 0;

        public FastReader(Stream s) { stream = s; }

        private int ReadByte()
        {
            if (pos >= limit)
            {
                pos = 0;
                limit = stream.Read(buffer, 0, buffer.Length);
                if (limit <= 0) return -1;
            }
            return buffer[pos++];
        }

        public int NextInt()
        {
            int c = ReadByte();
            while (c != -1 && c <= 32) c = ReadByte();
            if (c == -1) return 0;
            int res = 0;
            while (c > 32)
            {
                res = res * 10 + (c - '0');
                c = ReadByte();
            }
            return res;
        }
    }
}
```
