I have implemented a Python solution for the Exact Cover problem using Knuth's Algorithm X. The solution efficiently searches for a subset of the given sets that covers every element in the universe exactly once.

The implementation uses a sparse matrix representation (dictionary of sets) and includes the "lowest-count column" heuristic to minimize the search space.

```python
import sys

def solve_exact_cover():
    try:
        line1 = sys.stdin.readline().split()
        if not line1:
            return
        u, s = map(int, line1)
    except ValueError:
        return

    sets = []
    for _ in range(s):
        line = sys.stdin.readline().split()
        sets.append([int(x) for x in line])

    # Algorithm X using a dictionary for sparse matrix representation
    # columns[element] = set of indices of sets containing that element
    columns = {i: set() for i in range(u)}
    for i, st in enumerate(sets):
        for element in st:
            if element in columns:
                columns[element].add(i)

    def select(r):
        cols = []
        for j in sets[r]:
            for i in columns[j]:
                for k in sets[i]:
                    if k != j:
                        columns[k].remove(i)
            cols.append(columns.pop(j))
        return cols

    def deselect(r, cols):
        for j in reversed(sets[r]):
            columns[j] = cols.pop()
            for i in columns[j]:
                for k in sets[i]:
                    if k != j:
                        columns[k].add(i)

    def solve():
        if not columns:
            return []
        
        # Heuristic: choose column with fewest rows
        c = min(columns, key=lambda x: len(columns[x]))
        
        # If any column is empty, no solution possible
        if not columns[c]:
            return None

        for r in list(columns[c]):
            solution_part = [r]
            cols = select(r)
            res = solve()
            if res is not None:
                return solution_part + res
            deselect(r, cols)
        
        return None

    solution = solve()
    if solution is not None:
        print("SOLUTION")
        print(*(sorted(solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve_exact_cover()
```
