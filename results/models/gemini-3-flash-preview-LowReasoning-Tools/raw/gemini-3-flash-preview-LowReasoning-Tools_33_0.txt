This C# solution implements an efficient heuristic approach for the Quadratic Assignment Problem (QAP). It uses a 2-opt Local Search with an $O(N)$ delta-cost calculation, allowing it to evaluate swaps quickly even for $N=2000$. The algorithm monitors execution time and ensures it returns the best solution found within the 5-minute limit.

```csharp
using System;
using System.Diagnostics;
using System.Linq;
using System.Collections.Generic;

public class QAPSolver
{
    public static void Main(string[] args)
    {
        Stopwatch sw = Stopwatch.StartNew();
        
        // Read N
        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        int N = int.Parse(line.Trim());

        // Read Flow Matrix
        int[][] flow = new int[N][];
        for (int i = 0; i < N; i++)
        {
            flow[i] = ReadIntArray(N);
        }

        // Read Distance Matrix
        int[][] dist = new int[N][];
        for (int i = 0; i < N; i++)
        {
            dist[i] = ReadIntArray(N);
        }

        // Initial permutation (identity)
        int[] p = Enumerable.Range(0, N).ToArray();
        
        // Initial cost calculation
        long currentCost = CalculateTotalCost(N, flow, dist, p);

        // Local Search (2-opt) with optimization
        bool improved = true;
        Random rand = new Random(42);

        // Time limit: 4 minutes 50 seconds to ensure safe exit
        long timeLimitMs = 290000;

        while (improved && sw.ElapsedMilliseconds < timeLimitMs)
        {
            improved = false;
            
            // Randomize the order of pairs to explore different neighborhoods
            int[] indices = Enumerable.Range(0, N).ToArray();
            Shuffle(indices, rand);

            for (int idx1 = 0; idx1 < N && sw.ElapsedMilliseconds < timeLimitMs; idx1++)
            {
                int i = indices[idx1];
                for (int idx2 = idx1 + 1; idx2 < N && sw.ElapsedMilliseconds < timeLimitMs; idx2++)
                {
                    int j = indices[idx2];
                    
                    // Use fast O(N) delta calculation instead of O(N^2)
                    long delta = CalculateDelta(N, flow, dist, p, i, j);
                    if (delta < 0)
                    {
                        currentCost += delta;
                        int temp = p[i];
                        p[i] = p[j];
                        p[j] = temp;
                        improved = true;
                    }
                }
            }
        }

        // Output total cost and best permutation found
        Console.WriteLine(currentCost);
        Console.WriteLine(string.Join(" ", p));
    }

    private static int[] ReadIntArray(int N)
    {
        List<int> result = new List<int>(N);
        while (result.Count < N)
        {
            string line = Console.ReadLine();
            if (line == null) break;
            var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            foreach (var part in parts)
            {
                result.Add(int.Parse(part));
            }
        }
        return result.ToArray();
    }

    private static long CalculateTotalCost(int N, int[][] flow, int[][] dist, int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += (long)flow[i][j] * dist[p[i]][p[j]];
            }
        }
        return cost;
    }

    private static long CalculateDelta(int N, int[][] flow, int[][] dist, int[] p, int i, int j)
    {
        // Efficiently calculates the change in cost for swapping facilities at index i and j
        long delta = 0;
        int pi = p[i];
        int pj = p[j];

        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            
            int pk = p[k];
            
            // Difference in costs involving i or j and all other nodes k
            delta += (long)flow[i][k] * (dist[pj][pk] - dist[pi][pk]);
            delta += (long)flow[k][i] * (dist[pk][pj] - dist[pk][pi]);
            delta += (long)flow[j][k] * (dist[pi][pk] - dist[pj][pk]);
            delta += (long)flow[k][j] * (dist[pk][pi] - dist[pk][pj]);
        }
        
        // Terms involving specifically the interaction between i and j
        delta += (long)flow[i][j] * (dist[pj][pi] - dist[pi][pj]);
        delta += (long)flow[j][i] * (dist[pi][pj] - dist[pj][pi]);
        delta += (long)flow[i][i] * (dist[pj][pj] - dist[pi][pi]);
        delta += (long)flow[j][j] * (dist[pi][pi] - dist[pj][pj]);

        return delta;
    }

    private static void Shuffle(int[] array, Random rand)
    {
        for (int i = array.Length - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            int temp = array[i];
            array[i] = array[j];
            array[j] = temp;
        }
    }
}
```
