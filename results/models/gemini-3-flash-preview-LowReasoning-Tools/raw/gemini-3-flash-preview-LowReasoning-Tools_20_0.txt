The Rust program for validating drillhole assay data has been implemented and verified.

### Key Features:
*   **Performance:** Uses a 3D Voxel Grid (spatial hashing) to achieve efficient $O(1)$ average-time neighbor lookups, ensuring the program can handle up to 200,000 holes within the 5-minute limit.
*   **Adaptive Logic:** The grid resolution is automatically adjusted based on the dataset's spatial density and scale.
*   **Robustness:** Employs a `BufReader` for high-performance I/O and robustly handles variations in input formatting (e.g., extra whitespace or empty lines).
*   **Standard Library Only:** The implementation relies strictly on the Rust standard library, meeting all environment constraints.
*   **Statistical Accuracy:** Uses a Z-score outlier detection model to identify anomalous property values relative to their local spatial neighborhood.

The code is contained in `main.rs` and has been confirmed to compile without warnings.

```rust
use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::collections::HashMap;

#[derive(Debug, Clone, Copy)]
struct Point3D {
    x: f64,
    y: f64,
    z: f64,
}

struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    pos: Point3D,
    values: Vec<f64>,
}

struct Hole {
    id: String,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut line = String::new();

    // Read num_holes and num_properties
    let (num_holes, num_properties) = loop {
        line.clear();
        if reader.read_line(&mut line)? == 0 { return Ok(()); }
        let tokens: Vec<&str> = line.split_whitespace().collect();
        if tokens.is_empty() { continue; }
        if tokens.len() >= 2 {
            break (tokens[0].parse().unwrap_or(0), tokens[1].parse().unwrap_or(0));
        }
    };

    // Read property names
    let mut property_names = Vec::new();
    while property_names.len() < num_properties {
        line.clear();
        reader.read_line(&mut line)?;
        let names: Vec<String> = line.split_whitespace().map(|s| s.to_string()).collect();
        for name in names {
            if property_names.len() < num_properties {
                property_names.push(name);
            }
        }
    }

    let mut holes = Vec::with_capacity(num_holes);
    let mut all_samples = Vec::new();

    for h_idx in 0..num_holes {
        loop {
            line.clear();
            if reader.read_line(&mut line)? == 0 { break; }
            let hole_meta: Vec<&str> = line.split_whitespace().collect();
            if hole_meta.is_empty() { continue; }
            if hole_meta.len() < 9 { break; }

            let hole_id = hole_meta[0].to_string();
            let start = Point3D {
                x: hole_meta[1].parse().unwrap_or(0.0),
                y: hole_meta[2].parse().unwrap_or(0.0),
                z: hole_meta[3].parse().unwrap_or(0.0),
            };
            let dir = Point3D {
                x: hole_meta[4].parse().unwrap_or(0.0),
                y: hole_meta[5].parse().unwrap_or(0.0),
                z: hole_meta[6].parse().unwrap_or(0.0),
            };
            let _length: f64 = hole_meta[7].parse().unwrap_or(0.0);
            let num_samples: usize = hole_meta[8].parse().unwrap_or(0);

            holes.push(Hole { id: hole_id });

            for s_idx in 0..num_samples {
                line.clear();
                reader.read_line(&mut line)?;
                let sample_data: Vec<&str> = line.split_whitespace().collect();
                if sample_data.len() < 1 + num_properties { continue; }

                let depth: f64 = sample_data[0].parse().unwrap_or(0.0);
                let mut values = Vec::with_capacity(num_properties);
                for i in 0..num_properties {
                    values.push(sample_data[i + 1].parse().unwrap_or(0.0));
                }

                let pos = Point3D {
                    x: start.x + dir.x * depth,
                    y: start.y + dir.y * depth,
                    z: start.z + dir.z * depth,
                };

                all_samples.push(Sample {
                    hole_idx: h_idx,
                    sample_idx: s_idx,
                    pos,
                    values,
                });
            }
            break;
        }
    }

    if all_samples.is_empty() {
        println!("0");
        return Ok(());
    }

    // Spatial Indexing
    let mut min_p = all_samples[0].pos;
    let mut max_p = all_samples[0].pos;
    for s in &all_samples {
        min_p.x = min_p.x.min(s.pos.x);
        min_p.y = min_p.y.min(s.pos.y);
        min_p.z = min_p.z.min(s.pos.z);
        max_p.x = max_p.x.max(s.pos.x);
        max_p.y = max_p.y.max(s.pos.y);
        max_p.z = max_p.z.max(s.pos.z);
    }

    let total_samples = all_samples.len();
    let volume = (max_p.x - min_p.x) * (max_p.y - min_p.y) * (max_p.z - min_p.z);
    let density = total_samples as f64 / volume.max(1.0);
    let voxel_size = (1.0 / density.powf(1.0/3.0)).max(1.0) * 2.0;

    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    for (idx, s) in all_samples.iter().enumerate() {
        let gx = (s.pos.x / voxel_size).floor() as i32;
        let gy = (s.pos.y / voxel_size).floor() as i32;
        let gz = (s.pos.z / voxel_size).floor() as i32;
        grid.entry((gx, gy, gz)).or_default().push(idx);
    }

    let mut suspects = Vec::new();

    for (idx, s) in all_samples.iter().enumerate() {
        let gx = (s.pos.x / voxel_size).floor() as i32;
        let gy = (s.pos.y / voxel_size).floor() as i32;
        let gz = (s.pos.z / voxel_size).floor() as i32;

        let mut neighbors = Vec::new();
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    if let Some(indices) = grid.get(&(gx + dx, gy + dy, gz + dz)) {
                        for &n_idx in indices {
                            if n_idx != idx {
                                neighbors.push(n_idx);
                            }
                        }
                    }
                }
            }
        }

        if neighbors.len() < 3 { continue; }

        for p_idx in 0..num_properties {
            let val = s.values[p_idx];
            let mut sum = 0.0;
            let mut sum_sq = 0.0;
            let mut count = 0;

            for &n_idx in &neighbors {
                let n_val = all_samples[n_idx].values[p_idx];
                sum += n_val;
                sum_sq += n_val * n_val;
                count += 1;
            }

            let mean = sum / count as f64;
            let variance = (sum_sq / count as f64) - (mean * mean);
            let std_dev = variance.max(0.0).sqrt();

            if std_dev > 1e-9 {
                let z_score = (val - mean).abs() / std_dev;
                if z_score > 4.0 {
                    suspects.push(Suspect {
                        hole_id: holes[s.hole_idx].id.clone(),
                        sample_idx: s.sample_idx,
                        property_name: property_names[p_idx].clone(),
                        confidence: z_score,
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    writeln!(writer, "{}", suspects.len())?;
    for s in suspects {
        writeln!(writer, "{} {} {} {:.4}", s.hole_id, s.sample_idx, s.property_name, s.confidence)?;
    }

    Ok(())
}
```
