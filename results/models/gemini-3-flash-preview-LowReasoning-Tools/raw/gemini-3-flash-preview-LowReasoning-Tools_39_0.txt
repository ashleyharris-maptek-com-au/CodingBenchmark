I have implemented a Rust solution for the **Graph Bisection** problem.

### Algorithm Details
1.  **Exact Solver (N ≤ 30):**
    *   Uses a bitmask-based approach to iterate through all possible ways to choose `n/2` vertices.
    *   To reduce the search space by half, vertex `0` is fixed in partition A, and we choose `(n/2 - 1)` more vertices from the remaining `n-1` vertices.
    *   **Gosper's Hack** is used to generate combinations efficiently.
    *   The cut size is calculated using bitwise operations (`adj[i] & !full_mask`) and `count_ones()` for high performance.
2.  **Heuristic Solver (N > 30):**
    *   Employs a **Multi-start Local Search** (similar to Kernighan-Lin).
    *   Starts with multiple random initial bisections (using a self-contained LCG for randomness).
    *   Iteratively improves each bisection by swapping a pair of vertices (one from each partition) if it reduces the cut size.
    *   Runs as many iterations as possible within a 1.8-second time limit to maximize the chance of finding the global minimum.

### Code
```rust
use std::io::{self, BufRead};
use std::time::Instant;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    if parts.is_empty() {
        return;
    }
    let n = parts[0];
    let m = if parts.len() > 1 { parts[1] } else { 0 };

    let mut adj = vec![0u64; n];
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let edge: Vec<usize> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            if edge.len() == 2 {
                let u = edge[0];
                let v = edge[1];
                if u < n && v < n {
                    adj[u] |= 1 << v;
                    adj[v] |= 1 << u;
                }
            }
        }
    }

    if n == 0 {
        println!("0\n");
        return;
    }

    let mut min_cut = usize::MAX;
    let mut best_mask = 0u64;

    let half_n = n / 2;
    
    // We'll use a threshold for exact search. (30 choose 15) / 2 is ~77.5M
    if n <= 30 {
        let k = half_n - 1;
        if k == 0 {
            let full_mask = 1u64;
            let mut cut_size = 0;
            for i in 0..n {
                if (full_mask >> i) & 1 == 1 {
                    cut_size += (adj[i] & !full_mask).count_ones() as usize;
                }
            }
            min_cut = cut_size;
            best_mask = full_mask;
        } else {
            let mut mask = (1u64 << k) - 1;
            let limit = 1u64 << (n - 1);

            while mask < limit {
                let full_mask = (mask << 1) | 1;
                
                let mut cut_size = 0;
                let mut temp_mask = full_mask;
                while temp_mask > 0 {
                    let i = temp_mask.trailing_zeros() as usize;
                    cut_size += (adj[i] & !full_mask).count_ones() as usize;
                    temp_mask &= temp_mask - 1;
                }

                if cut_size < min_cut {
                    min_cut = cut_size;
                    best_mask = full_mask;
                    if min_cut == 0 {
                        break;
                    }
                }

                let c = mask & mask.wrapping_neg();
                let r = mask + c;
                if r == 0 { break; }
                mask = (((r ^ mask) >> 2) / c) | r;
            }
        }
    } else {
        // Fallback for N > 30: Multi-start Local Search (Kernighan-Lin style)
        let start_time = Instant::now();
        let mut rng_state = 12345u64;
        fn next_rng(state: &mut u64) -> u64 {
            *state = state.wrapping_mul(6364136223846793005).wrapping_add(1);
            *state
        }

        fn calculate_cut(mask: u64, _n: usize, adj: &[u64]) -> usize {
            let mut cut = 0;
            let mut temp_mask = mask;
            while temp_mask > 0 {
                let i = temp_mask.trailing_zeros() as usize;
                cut += (adj[i] & !mask).count_ones() as usize;
                temp_mask &= temp_mask - 1;
            }
            cut
        }

        min_cut = usize::MAX;
        best_mask = 0;

        while start_time.elapsed().as_secs_f32() < 1.8 {
            let mut mask = 0u64;
            let mut indices: Vec<usize> = (0..n).collect();
            // Simple shuffle
            for i in (1..n).rev() {
                let j = (next_rng(&mut rng_state) % (i as u64 + 1)) as usize;
                indices.swap(i, j);
            }
            for &idx in indices.iter().take(half_n) {
                mask |= 1 << idx;
            }

            let mut improved = true;
            while improved {
                improved = false;
                let mut best_gain = 0i32;
                let mut best_swap = None;

                let current_cut = calculate_cut(mask, n, &adj);
                
                for i in 0..n {
                    if (mask >> i) & 1 == 1 {
                        for j in 0..n {
                            if (mask >> j) & 1 == 0 {
                                let new_mask = (mask ^ (1 << i)) | (1 << j);
                                let new_cut = calculate_cut(new_mask, n, &adj);
                                let gain = current_cut as i32 - new_cut as i32;
                                if gain > best_gain {
                                    best_gain = gain;
                                    best_swap = Some((i, j));
                                }
                            }
                        }
                    }
                }

                if let Some((i, j)) = best_swap {
                    mask = (mask ^ (1 << i)) | (1 << j);
                    improved = true;
                }
            }

            let final_cut = calculate_cut(mask, n, &adj);
            if final_cut < min_cut {
                min_cut = final_cut;
                best_mask = mask;
                if min_cut == 0 { break; }
            }
        }
    }

    println!("{}", min_cut);
    let mut first = true;
    for i in 0..n {
        if (best_mask >> i) & 1 == 1 {
            if !first {
                print!(" ");
            }
            print!("{}", i);
            first = false;
        }
    }
    println!("");
}
```
