const fs = require('fs');

/**
 * Solves the Vehicle Routing Problem (VRP) using the Clarke-Wright Savings algorithm.
 * This algorithm is efficient and scales well for larger datasets.
 */
function solveVRP() {
    let input;
    try {
        // Read JSON input from stdin
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;

    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }
    
    const { depot, truckCount, packages } = data;

    // Handle edge case with no packages
    if (!packages || packages.length === 0) {
        process.stdout.write(JSON.stringify({ routes: [] }) + '\n');
        return;
    }

    const depotX = depot[0];
    const depotY = depot[1];

    /**
     * Calculates the Euclidean distance between two points.
     */
    function getDist(p1, p2) {
        const dx = p1.x - p2.x;
        const dy = p1.y - p2.y;
        return Math.sqrt(dx * dx + dy * dy);
    }

    const n = packages.length;
    const distToDepot = new Float64Array(n);
    for (let i = 0; i < n; i++) {
        distToDepot[i] = getDist(packages[i], { x: depotX, y: depotY });
    }

    // Savings Calculation: S(i, j) = d(D, i) + d(D, j) - d(i, j)
    // This represents the distance saved by connecting packages i and j instead of going to the depot for each.
    const savings = [];
    for (let i = 0; i < n; i++) {
        for (let j = i + 1; j < n; j++) {
            const d_ij = getDist(packages[i], packages[j]);
            const s = distToDepot[i] + distToDepot[j] - d_ij;
            if (s > 0) {
                savings.push({ i, j, s });
            }
        }
    }

    // Sort savings in descending order to prioritize the best merges
    savings.sort((a, b) => b.s - a.s);

    // Initial state: Each package is in its own route
    let routes = packages.map((_, i) => [i]);
    const packageToRoute = new Int32Array(n);
    for (let i = 0; i < n; i++) packageToRoute[i] = i;

    // Merge routes based on savings
    for (let k = 0; k < savings.length; k++) {
        const { i, j } = savings[k];
        const rIIdx = packageToRoute[i];
        const rJIdx = packageToRoute[j];

        // Already in the same route
        if (rIIdx === rJIdx) continue;

        const routeI = routes[rIIdx];
        const routeJ = routes[rJIdx];

        // Packages can only be merged if they are at the ends of their respective routes
        const iIsStart = routeI[0] === i;
        const iIsEnd = routeI[routeI.length - 1] === i;
        const jIsStart = routeJ[0] === j;
        const jIsEnd = routeJ[routeJ.length - 1] === j;

        if ((iIsStart || iIsEnd) && (jIsStart || jIsEnd)) {
            let merged;
            // Determine the correct orientation for merging the routes
            if (iIsEnd && jIsStart) {
                merged = routeI.concat(routeJ);
            } else if (iIsStart && jIsEnd) {
                merged = routeJ.concat(routeI);
            } else if (iIsEnd && jIsEnd) {
                merged = routeI.concat(routeJ.slice().reverse());
            } else if (iIsStart && jIsStart) {
                merged = routeI.slice().reverse().concat(routeJ);
            }

            if (merged) {
                routes[rIIdx] = merged;
                routes[rJIdx] = null;
                // Update pointers for all packages in the new merged route
                for (let m = 0; m < merged.length; m++) {
                    packageToRoute[merged[m]] = rIIdx;
                }
            }
        }
    }

    // Filter out null entries (merged routes)
    let finalRoutes = routes.filter(r => r !== null);

    // If the number of routes exceeds truckCount, merge the shortest routes until within limit
    while (finalRoutes.length > truckCount) {
        finalRoutes.sort((a, b) => a.length - b.length);
        const shortest = finalRoutes.shift();
        finalRoutes[0] = finalRoutes[0].concat(shortest);
    }

    // Map package indices back to their IDs for the final output
    const result = {
        routes: finalRoutes.map(r => r.map(idx => packages[idx].id))
    };

    // Output final JSON document to stdout
    process.stdout.write(JSON.stringify(result) + '\n');
}

// Execute the solver
solveVRP();
