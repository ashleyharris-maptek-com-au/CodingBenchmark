using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x};
    }
    double lengthSq() const { return x * x + y * y + z * z; }
    double length() const { return sqrt(lengthSq()); }
    Vec3 normalize() const {
        double l = length();
        return (l > 1e-9) ? (*this * (1.0 / l)) : Vec3{0, 0, 0};
    }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz;
    array<Vec3, 4> vertices;

    void computeVertices(double e) {
        double R = e * sqrt(3.0 / 8.0);
        double r = e / sqrt(24.0);

        array<Vec3, 4> base_v = {
            Vec3{0, R, 0},
            Vec3{-e / 2.0, -r, -e / (2.0 * sqrt(3.0))},
            Vec3{e / 2.0, -r, -e / (2.0 * sqrt(3.0))},
            Vec3{0, -r, e / sqrt(3.0)}
        };

        double cx = cos(rx), sx = sin(rx);
        double cy = cos(ry), sy = sin(ry);
        double cz = cos(rz), sz = sin(rz);

        for (int i = 0; i < 4; ++i) {
            Vec3 v = base_v[i];
            double x1 = v.x * cz - v.y * sz;
            double y1 = v.x * sz + v.y * cz;
            double z1 = v.z;
            double x2 = x1 * cy + z1 * sy;
            double y2 = y1;
            double z2 = -x1 * sy + z1 * cy;
            double x3 = x2;
            double y3 = y2 * cx - z2 * sx;
            double z3 = y2 * sx + z2 * cx;
            vertices[i] = Vec3{x3, y3, z3} + center;
        }
    }
};

bool sat_overlap(const Tetrahedron& t1, const Tetrahedron& t2) {
    auto project = [](const Tetrahedron& t, const Vec3& axis) {
        double min_p = axis.dot(t.vertices[0]);
        double max_p = min_p;
        for (int i = 1; i < 4; ++i) {
            double p = axis.dot(t.vertices[i]);
            min_p = min(min_p, p);
            max_p = max(max_p, p);
        }
        return make_pair(min_p, max_p);
    };

    array<Vec3, 4> axes1, axes2;
    for (int i = 0; i < 4; ++i) {
        axes1[i] = (t1.vertices[(i + 1) % 4] - t1.vertices[i]).cross(t1.vertices[(i + 2) % 4] - t1.vertices[i]).normalize();
        axes2[i] = (t2.vertices[(i + 1) % 4] - t2.vertices[i]).cross(t2.vertices[(i + 2) % 4] - t2.vertices[i]).normalize();
    }

    for (const auto& axis : axes1) {
        auto [min1, max1] = project(t1, axis);
        auto [min2, max2] = project(t2, axis);
        if (max1 < min2 - 1e-7 || max2 < min1 - 1e-7) return false;
    }
    for (const auto& axis : axes2) {
        auto [min1, max1] = project(t1, axis);
        auto [min2, max2] = project(t2, axis);
        if (max1 < min2 - 1e-7 || max2 < min1 - 1e-7) return false;
    }

    for (int i = 0; i < 4; ++i) {
        Vec3 e1 = t1.vertices[i] - t1.vertices[(i + 1) % 4];
        for (int j = 0; j < 4; ++j) {
            Vec3 e2 = t2.vertices[j] - t2.vertices[(j + 1) % 4];
            Vec3 axis = e1.cross(e2);
            if (axis.lengthSq() < 1e-9) continue;
            axis = axis.normalize();
            auto [min1, max1] = project(t1, axis);
            auto [min2, max2] = project(t2, axis);
            if (max1 < min2 - 1e-7 || max2 < min1 - 1e-7) return false;
        }
    }
    return true;
}

struct SpatialGrid {
    double cell_size;
    Vec3 min_bound;
    vector<vector<int>> cells;
    int nx, ny, nz;

    SpatialGrid(double cs, Vec3 min_b, Vec3 max_b) : cell_size(cs), min_bound(min_b) {
        nx = (int)ceil((max_b.x - min_b.x) / cell_size) + 1;
        ny = (int)ceil((max_b.y - min_b.y) / cell_size) + 1;
        nz = (int)ceil((max_b.z - min_b.z) / cell_size) + 1;
        cells.resize((size_t)nx * ny * nz);
    }

    int get_index(const Vec3& p) {
        int ix = (int)((p.x - min_bound.x) / cell_size);
        int iy = (int)((p.y - min_bound.y) / cell_size);
        int iz = (int)((p.z - min_bound.z) / cell_size);
        if (ix < 0 || ix >= nx || iy < 0 || iy >= ny || iz < 0 || iz >= nz) return -1;
        return ix * ny * nz + iy * nz + iz;
    }

    void add(const Vec3& p, int id) {
        int idx = get_index(p);
        if (idx != -1) cells[idx].push_back(id);
    }

    vector<int> get_neighbors(const Vec3& p) {
        vector<int> neighbors;
        int ix = (int)((p.x - min_bound.x) / cell_size);
        int iy = (int)((p.y - min_bound.y) / cell_size);
        int iz = (int)((p.z - min_bound.z) / cell_size);
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    int nix = ix + dx, niy = iy + dy, niz = iz + dz;
                    if (nix >= 0 && nix < nx && niy >= 0 && niy < ny && niz >= 0 && niz < nz) {
                        int idx = nix * ny * nz + niy * nz + niz;
                        neighbors.insert(neighbors.end(), cells[idx].begin(), cells[idx].end());
                    }
                }
            }
        }
        return neighbors;
    }
};

struct Polyhedron {
    vector<Vec3> vertices;
    Vec3 min_bound, max_bound;

    void computeBounds() {
        if (vertices.empty()) return;
        min_bound = max_bound = vertices[0];
        for (const auto& v : vertices) {
            min_bound.x = min(min_bound.x, v.x); min_bound.y = min(min_bound.y, v.y); min_bound.z = min(min_bound.z, v.z);
            max_bound.x = max(max_bound.x, v.x); max_bound.y = max(max_bound.y, v.y); max_bound.z = max(max_bound.z, v.z);
        }
    }

    bool contains(const Vec3& p) const {
        if (p.x < min_bound.x || p.x > max_bound.x || p.y < min_bound.y || p.y > max_bound.y || p.z < min_bound.z || p.z > max_bound.z) return false;
        return true; 
    }
};

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int V_count; double edge_length;
    if (!(cin >> V_count >> edge_length)) return 0;
    Polyhedron container;
    for (int i = 0; i < V_count; ++i) {
        double x, y, z; cin >> x >> y >> z;
        container.vertices.push_back({x, y, z});
    }
    container.computeBounds();

    vector<Tetrahedron> packed;
    SpatialGrid grid(edge_length * 1.5, container.min_bound, container.max_bound);
    auto start_time = chrono::steady_clock::now();
    double step_x = edge_length * 0.8165; 
    double step_y = edge_length * 0.866;
    double step_z = edge_length * 0.943;

    for (double x = container.min_bound.x + edge_length; x < container.max_bound.x - edge_length; x += step_x) {
        for (double y = container.min_bound.y + edge_length; y < container.max_bound.y - edge_length; y += step_y) {
            for (double z = container.min_bound.z + edge_length; z < container.max_bound.z - edge_length; z += step_z) {
                if (chrono::duration_cast<chrono::seconds>(chrono::steady_clock::now() - start_time).count() > 27) goto end_packing;
                
                Tetrahedron t; t.center = {x, y, z}; t.rx = 0; t.ry = 0; t.rz = 0;
                t.computeVertices(edge_length);
                bool can_place = true;
                for (const auto& v : t.vertices) { if (!container.contains(v)) { can_place = false; break; } }
                if (can_place) {
                    vector<int> neighbors = grid.get_neighbors(t.center);
                    for (int id : neighbors) { if (sat_overlap(t, packed[id])) { can_place = false; break; } }
                }
                if (can_place) {
                    grid.add(t.center, (int)packed.size());
                    packed.push_back(t);
                }
            }
        }
    }

end_packing:
    cout << packed.size() << endl;
    for (const auto& t : packed) {
        cout << t.center.x << " " << t.center.y << " " << t.center.z << " " << t.rx << " " << t.ry << " " << t.rz << endl;
    }
    return 0;
}
