using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

public class AsteroidInterception
{
    public const double DayToSec = 86400.0;
    public const double SunMu = 1.32712440018e20;

    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
        public static Vector3 operator /(Vector3 a, double b) => new Vector3(a.X / b, a.Y / b, a.Z / b);
        public double Length => Math.Sqrt(X * X + Y * Y + Z * Z);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public Vector3 Normalized() { double l = Length; return l > 0 ? this / l : new Vector3(0, 0, 0); }
        public override string ToString() => $"{X} {Y} {Z}";
    }

    public class Body
    {
        public string Name = "";
        public double Mass;
        public Vector3 Position;
        public Vector3 Velocity;
    }

    public static void Main(string[] args)
    {
        string? line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts1.Length < 6) return;
        int numBodies = int.Parse(parts1[0]);
        double warningDays = double.Parse(parts1[1]);
        double impactorMass = double.Parse(parts1[2]);
        double dvBudget = double.Parse(parts1[3]);
        double asteroidMass = double.Parse(parts1[4]);
        long t0Unix = long.Parse(parts1[5]);

        List<Body> bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            string? bodyLine = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(bodyLine)) continue;
            string[] bp = bodyLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            bodies.Add(new Body
            {
                Name = bp[0],
                Mass = double.Parse(bp[1]),
                Position = new Vector3(double.Parse(bp[2]), double.Parse(bp[3]), double.Parse(bp[4])),
                Velocity = new Vector3(double.Parse(bp[5]), double.Parse(bp[6]), double.Parse(bp[7]))
            });
        }
        string? bspPath = Console.ReadLine();

        Body? earth = bodies.Find(b => b.Name == "Earth");
        Body? asteroid = bodies.Find(b => b.Name == "Asteroid");
        if (earth == null || asteroid == null) return;

        double bestDeflection = -1;
        double bestLaunchTime = 0;
        double bestImpactTime = warningDays;
        Vector3 bestDv1 = new Vector3(0, 0, 0);

        int tLSteps = 40;
        int tISteps = 40;
        
        for (int i = 0; i <= tLSteps; i++)
        {
            double tL = (warningDays * 0.8) * i / tLSteps;
            Vector3 rL = Propagate(earth, tL);
            Vector3 vL_earth = PropagateVel(earth, tL);

            for (int j = 1; j <= tISteps; j++)
            {
                double tI = tL + 50 + (warningDays - (tL + 50)) * j / tISteps;
                if (tI > warningDays) continue;

                Vector3 rI = Propagate(asteroid, tI);
                double dt = (tI - tL) * DayToSec;
                
                Vector3 v1, v2;
                if (SolveLambert(rL, rI, dt, SunMu, out v1, out v2))
                {
                    Vector3 dv1 = v1 - vL_earth;
                    if (dv1.Length <= dvBudget)
                    {
                        Vector3 vAst = PropagateVel(asteroid, tI);
                        Vector3 vRel = v2 - vAst;
                        // Maximize deflection proportional to momentum transfer and time until Earth impact
                        double deflection = vRel.Length * (warningDays - tI);
                        if (deflection > bestDeflection)
                        {
                            bestDeflection = deflection;
                            bestLaunchTime = tL;
                            bestImpactTime = tI;
                            bestDv1 = dv1;
                        }
                    }
                }
            }
        }

        // If no intercept found within budget, try a basic one
        if (bestDeflection < 0)
        {
            double tL = 0;
            double tI = warningDays * 0.5;
            Vector3 rL = earth.Position;
            Vector3 rI = Propagate(asteroid, tI);
            Vector3 v1, v2;
            if (SolveLambert(rL, rI, tI * DayToSec, SunMu, out v1, out v2))
            {
                bestLaunchTime = tL;
                bestImpactTime = tI;
                bestDv1 = v1 - earth.Velocity;
                if (bestDv1.Length > dvBudget) bestDv1 = bestDv1.Normalized() * dvBudget;
            }
        }

        // Output launch and "impact" (final approach) burns
        Console.WriteLine(2);
        Console.WriteLine($"{bestLaunchTime:F8} {bestDv1.X:F8} {bestDv1.Y:F8} {bestDv1.Z:F8}");
        Console.WriteLine($"{bestImpactTime:F8} 0.00000000 0.00000000 0.00000000");
    }

    // Simple 2nd order propagation for initial state estimation
    static Vector3 Propagate(Body b, double days)
    {
        double t = days * DayToSec;
        Vector3 r = b.Position;
        Vector3 v = b.Velocity;
        double r3 = Math.Pow(r.Length, 3);
        Vector3 acc = r * (-SunMu / r3);
        return r + v * t + acc * (0.5 * t * t);
    }

    static Vector3 PropagateVel(Body b, double days)
    {
        double t = days * DayToSec;
        Vector3 r = b.Position;
        Vector3 v = b.Velocity;
        double r3 = Math.Pow(r.Length, 3);
        Vector3 acc = r * (-SunMu / r3);
        return v + acc * t;
    }

    // Lambert Solver using Universal Variables (BMW algorithm)
    static bool SolveLambert(Vector3 r1, Vector3 r2, double dt, double mu, out Vector3 v1, out Vector3 v2)
    {
        v1 = v2 = new Vector3(0, 0, 0);
        double r1m = r1.Length;
        double r2m = r2.Length;
        double dot = Vector3.Dot(r1, r2);
        double cosDeltaNu = dot / (r1m * r2m);
        cosDeltaNu = Math.Max(-1.0, Math.Min(1.0, cosDeltaNu));

        double A = Math.Sqrt(r1m * r2m * (1 + cosDeltaNu));
        if (A == 0) return false;

        double z = 0;
        double zUpper = 4 * Math.PI * Math.PI;
        double zLower = -4 * Math.PI * Math.PI;

        for (int i = 0; i < 60; i++)
        {
            double c, s;
            Stumpff(z, out c, out s);
            double y = r1m + r2m + A * (z * s - 1) / Math.Sqrt(c);
            
            if (A > 0 && y < 0)
            {
                z += 0.05;
                continue;
            }

            double x = Math.Sqrt(y / c);
            double dtCalc = (Math.Pow(x, 3) * s + A * Math.Sqrt(y)) / Math.Sqrt(mu);

            if (Math.Abs(dt - dtCalc) < 1.0)
            {
                double f = 1 - y / r1m;
                double g = A * Math.Sqrt(y / mu);
                double gDot = 1 - y / r2m;
                v1 = (r2 - r1 * f) / g;
                v2 = (r2 * gDot - r1) / g;
                return true;
            }

            if (dtCalc < dt) zLower = z;
            else zUpper = z;
            z = (zUpper + zLower) / 2.0;
        }
        return false;
    }

    // Stumpff functions for universal variable iteration
    static void Stumpff(double z, out double c, out double s)
    {
        if (z > 1e-7)
        {
            double sqrtZ = Math.Sqrt(z);
            c = (1 - Math.Cos(sqrtZ)) / z;
            s = (sqrtZ - Math.Sin(sqrtZ)) / Math.Pow(sqrtZ, 3);
        }
        else if (z < -1e-7)
        {
            double sqrtZ = Math.Sqrt(-z);
            c = (Math.Exp(sqrtZ) + Math.Exp(-sqrtZ) - 2) / (2 * (-z));
            s = ( (Math.Exp(sqrtZ) - Math.Exp(-sqrtZ)) / 2 - sqrtZ) / Math.Pow(sqrtZ, 3);
        }
        else
        {
            c = 0.5;
            s = 1.0 / 6.0;
        }
    }
}
