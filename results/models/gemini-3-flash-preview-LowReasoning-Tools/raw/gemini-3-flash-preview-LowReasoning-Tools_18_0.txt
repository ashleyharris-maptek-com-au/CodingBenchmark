#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <limits>
#include <random>
#include <chrono>

using namespace std;

// Stoer-Wagner algorithm for global minimum cut in undirected graphs
class StoerWagner {
public:
    StoerWagner(int n) : n(n), adj(n, vector<int>(n, 0)), exists(n, true), in_a(n), w(n) {}

    void add_edge(int u, int v, int weight = 1) {
        if (u == v) return;
        adj[u][v] += weight;
        adj[v][u] += weight;
    }

    pair<int, vector<int>> solve() {
        int min_cut = numeric_limits<int>::max();
        vector<int> best_phase_cut;
        vector<vector<int>> components(n);
        for (int i = 0; i < n; ++i) components[i].push_back(i);

        for (int phase = 0; phase < n - 1; ++phase) {
            fill(in_a.begin(), in_a.end(), false);
            fill(w.begin(), w.end(), 0);
            int prev = -1, curr = -1;
            for (int i = 0; i < n - phase; ++i) {
                int next = -1;
                for (int j = 0; j < n; ++j) {
                    if (exists[j] && !in_a[j]) {
                        if (next == -1 || w[j] > w[next]) next = j;
                    }
                }
                in_a[next] = true;
                if (i == n - phase - 1) {
                    if (w[next] < min_cut) {
                        min_cut = w[next];
                        best_phase_cut = components[next];
                    }
                    if (curr != -1) {
                        for (int j = 0; j < n; ++j) {
                            if (exists[j] && j != next) {
                                adj[curr][j] += adj[next][j];
                                adj[j][curr] += adj[next][j];
                            }
                        }
                        components[curr].insert(components[curr].end(), components[next].begin(), components[next].end());
                    }
                    exists[next] = false;
                } else {
                    prev = curr;
                    curr = next;
                    for (int j = 0; j < n; ++j) {
                        if (exists[j] && !in_a[j]) w[j] += adj[curr][j];
                    }
                }
            }
            if (min_cut == 0) break;
        }
        return {min_cut, best_phase_cut};
    }

private:
    int n;
    vector<vector<int>> adj;
    vector<bool> exists;
    vector<bool> in_a;
    vector<int> w;
};

// Dinic algorithm for Max Flow / Min Cut
struct Dinic {
    struct Edge {
        int to, rev, cap, flow;
    };
    int n;
    vector<vector<Edge>> adj;
    vector<int> level, ptr;

    Dinic(int n) : n(n), adj(n), level(n), ptr(n) {}

    void add_edge(int from, int to, int cap) {
        adj[from].push_back({to, (int)adj[to].size(), cap, 0});
        adj[to].push_back({from, (int)adj[from].size() - 1, cap, 0});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& edge : adj[v]) {
                if (edge.cap - edge.flow > 0 && level[edge.to] == -1) {
                    level[edge.to] = level[v] + 1;
                    q.push(edge.to);
                }
            }
        }
        return level[t] != -1;
    }

    int dfs(int v, int t, int pushed) {
        if (pushed == 0) return 0;
        if (v == t) return pushed;
        for (int& cid = ptr[v]; cid < adj[v].size(); ++cid) {
            auto& edge = adj[v][cid];
            int tr = edge.to;
            if (level[v] + 1 != level[tr] || edge.cap - edge.flow == 0) continue;
            int push = dfs(tr, t, min(pushed, edge.cap - edge.flow));
            if (push == 0) continue;
            edge.flow += push;
            adj[tr][edge.rev].flow -= push;
            return push;
        }
        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(ptr.begin(), ptr.end(), 0);
            while (int pushed = dfs(s, t, numeric_limits<int>::max())) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<int> get_cut(int s) {
        vector<int> res;
        vector<bool> visited(n, false);
        queue<int> q;
        q.push(s);
        visited[s] = true;
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            res.push_back(v);
            for (auto& edge : adj[v]) {
                if (edge.cap - edge.flow > 0 && !visited[edge.to]) {
                    visited[edge.to] = true;
                    q.push(edge.to);
                }
            }
        }
        return res;
    }
};

void solve() {
    int N, M;
    if (!(cin >> N >> M)) return;

    if (N < 2) {
        cout << 0 << endl;
        return;
    }

    if (N <= 400) {
        StoerWagner sw(N);
        vector<pair<int, int>> all_edges;
        all_edges.reserve(M);
        for (int i = 0; i < M; ++i) {
            int u, v;
            cin >> u >> v;
            sw.add_edge(u, v);
            all_edges.push_back({u, v});
        }
        auto [min_cut_val, component] = sw.solve();
        cout << min_cut_val << "\n";
        vector<bool> in_comp(N, false);
        for (int node : component) in_comp[node] = true;
        for (auto& p : all_edges) {
            if (in_comp[p.first] != in_comp[p.second]) {
                cout << p.first << " " << p.second << "\n";
            }
        }
    } else {
        vector<pair<int, int>> all_edges;
        // 32GB memory allows for storing roughly 2 billion edges (16GB for vector of pairs)
        if (M < 1000000000) { 
            all_edges.reserve(M);
        }
        vector<int> degree(N, 0);
        for (int i = 0; i < M; ++i) {
            int u, v;
            cin >> u >> v;
            if (all_edges.capacity() > 0) all_edges.push_back({u, v});
            if (u < N) degree[u]++;
            if (v < N) degree[v]++;
        }

        int min_deg = numeric_limits<int>::max();
        int min_deg_node = -1;
        for (int i = 0; i < N; ++i) {
            if (degree[i] < min_deg) {
                min_deg = degree[i];
                min_deg_node = i;
            }
        }

        int best_cut_val = min_deg;
        vector<int> best_component = {min_deg_node};

        // Try randomized Dinic if memory and graph size allow
        if (N <= 200000 && !all_edges.empty() && all_edges.size() < 2000000) {
            mt19937 rng(1337);
            uniform_int_distribution<int> dist(0, N - 1);
            auto start_time = chrono::steady_clock::now();
            int s = dist(rng);
            for (int i = 0; i < 3; ++i) {
                int t = dist(rng);
                while (t == s) t = dist(rng);
                Dinic dinic(N);
                for (auto& p : all_edges) dinic.add_edge(p.first, p.second, 1);
                int flow = dinic.max_flow(s, t);
                if (flow < best_cut_val) {
                    best_cut_val = flow;
                    best_component = dinic.get_cut(s);
                }
                if (best_cut_val <= 1) break;
                if (chrono::duration_cast<chrono::seconds>(chrono::steady_clock::now() - start_time).count() > 20) break;
            }
        }

        cout << best_cut_val << "\n";
        vector<bool> in_comp(N, false);
        for (int node : best_component) in_comp[node] = true;
        
        if (!all_edges.empty()) {
            for (auto& p : all_edges) {
                if (in_comp[p.first] != in_comp[p.second]) {
                    cout << p.first << " " << p.second << "\n";
                }
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
