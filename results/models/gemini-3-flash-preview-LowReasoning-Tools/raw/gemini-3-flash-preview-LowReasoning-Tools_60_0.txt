const fs = require('fs');

function solve() {
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    const data = JSON.parse(input);
    const { machineCount, tasks } = data;

    const taskMap = new Map();
    const adj = new Map();
    const inDegree = new Map();

    for (const task of tasks) {
        taskMap.set(task.id, task);
        adj.set(task.id, []);
        inDegree.set(task.id, 0);
    }

    for (const task of tasks) {
        for (const depId of task.deps) {
            // Check if dependency exists in task list to avoid errors with malformed inputs
            if (adj.has(depId)) {
                adj.get(depId).push(task.id);
                inDegree.set(task.id, inDegree.get(task.id) + 1);
            }
        }
    }

    const readyQueue = [];
    for (const task of tasks) {
        if (inDegree.get(task.id) === 0) {
            readyQueue.push(task.id);
        }
    }

    /**
     * Priority heuristic for the ready queue:
     * 1. Earliest Deadline First (EDF) - crucial for minimizing lateness.
     * 2. Highest Weight/Duration Ratio (Smith's Rule) - optimal for minimizing weighted flow time.
     */
    const sortReadyQueue = (queue) => {
        queue.sort((a, b) => {
            const taskA = taskMap.get(a);
            const taskB = taskMap.get(b);
            if (taskA.deadline !== taskB.deadline) {
                return taskA.deadline - taskB.deadline;
            }
            // Use ratio to prioritize high-value, short-duration tasks
            const ratioA = taskA.weight / (taskA.duration || 1);
            const ratioB = taskB.weight / (taskB.duration || 1);
            return ratioB - ratioA;
        });
    };

    const schedule = [];
    const machineAvailableAt = new Array(machineCount).fill(0);
    const taskFinishedAt = new Map();
    let finishedCount = 0;

    while (finishedCount < tasks.length) {
        sortReadyQueue(readyQueue);

        let bestTaskIdx = -1;
        let bestMachineIdx = -1;
        let minStartTime = Infinity;

        // Find the earliest possible start time across all ready tasks and all machines
        // This greedy approach respects both dependencies and machine availability.
        for (let i = 0; i < readyQueue.length; i++) {
            const taskId = readyQueue[i];
            const task = taskMap.get(taskId);
            
            let depFinishedAt = 0;
            for (const depId of task.deps) {
                depFinishedAt = Math.max(depFinishedAt, taskFinishedAt.get(depId) || 0);
            }

            for (let m = 0; m < machineCount; m++) {
                const startTime = Math.max(machineAvailableAt[m], depFinishedAt);
                if (startTime < minStartTime) {
                    minStartTime = startTime;
                    bestTaskIdx = i;
                    bestMachineIdx = m;
                    // Optimization: if we find a machine starting at 0 or current time, it's a strong candidate
                    if (minStartTime === 0) break;
                }
            }
            if (minStartTime === 0) break;
        }

        if (bestTaskIdx !== -1) {
            const taskId = readyQueue.splice(bestTaskIdx, 1)[0];
            const task = taskMap.get(taskId);
            const startTime = minStartTime;
            const finishTime = startTime + task.duration;

            schedule.push({ id: taskId, start: startTime, machine: bestMachineIdx });
            machineAvailableAt[bestMachineIdx] = finishTime;
            taskFinishedAt.set(taskId, finishTime);
            finishedCount++;

            // Release dependencies
            for (const succId of adj.get(taskId)) {
                inDegree.set(succId, inDegree.get(succId) - 1);
                if (inDegree.get(succId) === 0) {
                    readyQueue.push(succId);
                }
            }
        } else {
            // Should not happen in a DAG
            break;
        }
    }

    process.stdout.write(JSON.stringify({ schedule }));
}

solve();
