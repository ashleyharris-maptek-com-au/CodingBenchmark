#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <stdint.h>

/**
 * TSP Solver using Morton Curve (Z-order) Sorting and Local 2-opt Refinement.
 * 
 * Scalability Strategy:
 * 1. Morton Curve: Maps 2D points to a 1D sequence that preserves spatial locality.
 *    Extremely fast (O(N log N) for sorting) and memory-efficient.
 * 2. Adaptive 2-opt:
 *    - For small N (< 1,000): Thorough refinement.
 *    - For medium N (1,000 - 100,000): Limited window and iteration refinement.
 *    - For large N (> 100,000): Skip refinement to maintain speed for millions of cities.
 */

struct Point {
    float x, y;
    int id;
};

// Morton curve code (Z-order) for fast spatial sorting
// Interleaves bits of X and Y coordinates to produce a 1D spatial key.
uint64_t splitBy2(uint32_t a) {
    uint64_t x = a & 0x00000000FFFFFFFF;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFF;
    x = (x | (x << 8))  & 0x00FF00FF00FF00FF;
    x = (x | (x << 4))  & 0x0F0F0F0F0F0F0F0F;
    x = (x | (x << 2))  & 0x3333333333333333;
    x = (x | (x << 1))  & 0x5555555555555555;
    return x;
}

uint64_t mortonEncode(uint32_t x, uint32_t y) {
    return (splitBy2(x) << 1) | splitBy2(y);
}

int main() {
    // Fast I/O for large datasets
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    if (!(std::cin >> N)) return 0;

    if (N <= 0) return 0;
    if (N == 1) {
        std::cout << "0\n";
        return 0;
    }

    std::vector<Point> points(N);
    float minX = 1e30f, minY = 1e30f, maxX = -1e30f, maxY = -1e30f;
    for (int i = 0; i < N; ++i) {
        if (!(std::cin >> points[i].x >> points[i].y)) break;
        points[i].id = i;
        if (points[i].x < minX) minX = points[i].x;
        if (points[i].y < minY) minY = points[i].y;
        if (points[i].x > maxX) maxX = points[i].x;
        if (points[i].y > maxY) maxY = points[i].y;
    }

    // Scale coordinates to [0, 2^32 - 1] for Morton encoding
    float rangeX = maxX - minX;
    float rangeY = maxY - minY;
    if (rangeX <= 0) rangeX = 1.0f;
    if (rangeY <= 0) rangeY = 1.0f;

    struct MortonPoint {
        uint64_t code;
        int id;
    };
    std::vector<MortonPoint> morton(N);
    for (int i = 0; i < N; ++i) {
        uint32_t ux = (uint32_t)(((points[i].x - minX) / rangeX) * 4294967295.0f);
        uint32_t uy = (uint32_t)(((points[i].y - minY) / rangeY) * 4294967295.0f);
        morton[i] = { mortonEncode(ux, uy), points[i].id };
    }

    // Sort by Morton code to get a good initial spatial tour
    std::sort(morton.begin(), morton.end(), [](const MortonPoint& a, const MortonPoint& b) {
        return a.code < b.code;
    });

    // Requirement: Output must start with city index 0
    int startIdx = 0;
    for (int i = 0; i < N; ++i) {
        if (morton[i].id == 0) {
            startIdx = i;
            break;
        }
    }
    std::rotate(morton.begin(), morton.begin() + startIdx, morton.end());

    // Adaptive 2-opt refinement
    if (N < 100000) {
        std::vector<int> tour(N);
        for(int i=0; i<N; ++i) tour[i] = morton[i].id;
        
        std::vector<Point> sortedPoints(N);
        for(int i=0; i<N; ++i) {
            sortedPoints[i] = points[tour[i]];
        }

        auto distSq = [](const Point& a, const Point& b) {
            float dx = a.x - b.x;
            float dy = a.y - b.y;
            return dx*dx + dy*dy;
        };

        bool improved = true;
        // Limit iterations based on scale
        int maxIter = (N < 1000) ? 100 : (N < 10000 ? 10 : 1);
        for (int iter = 0; iter < maxIter && improved; ++iter) {
            improved = false;
            // Limit the search window for larger scales
            int window = (N < 2000) ? N : 500;
            for (int i = 0; i < N - 2; ++i) {
                int limit = std::min(i + window, N);
                for (int j = i + 2; j < limit; ++j) {
                    int k = (j + 1) % N;
                    // Check if reversing the segment [i+1, j] reduces total distance
                    float currentDist = distSq(sortedPoints[i], sortedPoints[i+1]) + distSq(sortedPoints[j], sortedPoints[k]);
                    float newDist = distSq(sortedPoints[i], sortedPoints[j]) + distSq(sortedPoints[i+1], sortedPoints[k]);
                    if (newDist < currentDist) {
                        std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        std::reverse(sortedPoints.begin() + i + 1, sortedPoints.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
        for (int i = 0; i < N; ++i) {
            std::cout << tour[i] << (i == N - 1 ? "" : " ");
        }
        std::cout << "\n";
    } else {
        // Fast path for very large N: Output Morton order directly
        for (int i = 0; i < N; ++i) {
            std::cout << morton[i].id << (i == N - 1 ? "" : " ");
        }
        std::cout << "\n";
    }

    return 0;
}
