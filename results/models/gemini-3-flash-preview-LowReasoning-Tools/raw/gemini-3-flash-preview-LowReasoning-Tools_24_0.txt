Here is the complete C++ controller for the 3D Lunar Lander.

The controller uses a multithreaded architecture:
1.  **Main Thread**: Handles real-time I/O (reading `STATE` from `stdin` and writing commands to `stdout`).
2.  **Terrain Thread**: Parses the binary terrain file in the background to avoid blocking the initial simulation start.
3.  **Control Logic**: Uses PD loops for both position and attitude control. It calculates a desired acceleration vector to move towards the target while maintaining a safe altitude above the terrain, then converts that into `thrust`, `pitch`, and `roll` commands.

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <fstream>
#include <cmath>
#include <thread>
#include <mutex>
#include <atomic>
#include <iomanip>
#include <algorithm>

using namespace std;

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

// State update from stdin
struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

// Command to stdout
struct Command {
    double thrust;
    double pitch_cmd;
    double yaw_cmd;
    double roll_cmd;
};

// Terrain storage
struct TerrainColumn {
    uint16_t a, b, c;
};

class Terrain {
public:
    int width, depth, height;
    vector<TerrainColumn> columns;
    atomic<bool> ready{false};

    void load(const string& path, int w, int d, int h) {
        width = w;
        depth = d;
        height = h;
        columns.resize(width * depth);

        ifstream fs(path, ios::binary);
        if (!fs) return;

        for (int i = 0; i < width * depth; ++i) {
            fs.read(reinterpret_cast<char*>(&columns[i].a), 2);
            fs.read(reinterpret_cast<char*>(&columns[i].b), 2);
            fs.read(reinterpret_cast<char*>(&columns[i].c), 2);
        }
        ready = true;
    }

    double get_ground_height(double x, double y) const {
        if (!ready) return 0;
        int ix = static_cast<int>(floor(x));
        int iy = static_cast<int>(floor(y));
        if (ix < 0 || ix >= width || iy < 0 || iy >= depth) return 0;
        return columns[ix * depth + iy].a;
    }
};

struct Config {
    int width, depth, height;
    double gravity;
    double max_thrust;
    double fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string terrain_path;
};

class Controller {
public:
    Config config;
    Terrain terrain;
    mutex state_mutex;
    State current_state;
    atomic<bool> state_received{false};

    Controller(const Config& cfg) : config(cfg) {}

    void update_state(const State& s) {
        lock_guard<mutex> lock(state_mutex);
        current_state = s;
        state_received = true;
    }

    Command compute_command() {
        State s;
        {
            lock_guard<mutex> lock(state_mutex);
            s = current_state;
        }

        if (!state_received) return {0, 0, 0, 0};

        double dx = config.target_x - s.x;
        double dy = config.target_y - s.y;
        double dz = config.target_z - s.z;
        double dist_sq = dx * dx + dy * dy;
        double dist = sqrt(dist_sq);

        // Target velocity planning
        double target_vx = 0;
        double target_vy = 0;
        double target_vz = 0;

        double safe_z = config.target_z + 20.0;
        if (terrain.ready) {
            double ground = terrain.get_ground_height(s.x, s.y);
            safe_z = max(safe_z, ground + 10.0);
        }

        if (dist > 5.0) {
            // Horizontal approach
            double speed = min(20.0, dist * 0.5);
            target_vx = (dx / dist) * speed;
            target_vy = (dy / dist) * speed;
            
            // Maintain safe altitude
            double z_err = safe_z - s.z;
            target_vz = z_err * 0.5;
        } else {
            // Final vertical descent
            target_vx = dx * 1.0;
            target_vy = dy * 1.0;
            target_vz = dz * 0.5;
            target_vz = max(-2.0, min(2.0, target_vz)); // Keep vertical speed safe (< 3m/s)
        }

        // Acceleration control (PD on velocity)
        double ax = (target_vx - s.vx) * 1.0;
        double ay = (target_vy - s.vy) * 1.0;
        double az = (target_vz - s.vz) * 1.0 + config.gravity;

        // Rotate world acceleration to local frame (assuming roll/pitch are small for yaw calc)
        double cos_yaw = cos(s.yaw);
        double sin_yaw = sin(s.yaw);
        double local_ax = ax * cos_yaw + ay * sin_yaw;
        double local_ay = -ax * sin_yaw + ay * cos_yaw;

        double thrust_mag = sqrt(local_ax * local_ax + local_ay * local_ay + az * az);
        double thrust = thrust_mag / config.max_thrust;
        thrust = max(0.0, min(1.0, thrust));

        // Desired orientation to achieve acceleration
        double target_pitch = -atan2(local_ax, az);
        double target_roll = atan2(local_ay, sqrt(local_ax * local_ax + az * az));
        
        // Clamp angles to safe limits
        target_pitch = max(-M_PI/4, min(M_PI/4, target_pitch));
        target_roll = max(-M_PI/4, min(M_PI/4, target_roll));

        // Attitude control (PD)
        Command cmd;
        cmd.thrust = thrust;
        cmd.pitch_cmd = (target_pitch - s.pitch) * 10.0 - s.pitch_rate * 5.0;
        cmd.yaw_cmd = (0.0 - s.yaw) * 2.0 - s.yaw_rate * 2.0;
        cmd.roll_cmd = (target_roll - s.roll) * 10.0 - s.roll_rate * 5.0;

        cmd.pitch_cmd = max(-1.0, min(1.0, cmd.pitch_cmd));
        cmd.yaw_cmd = max(-1.0, min(1.0, cmd.yaw_cmd));
        cmd.roll_cmd = max(-1.0, min(1.0, cmd.roll_cmd));

        return cmd;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    Config cfg;
    if (!(cin >> cfg.width >> cfg.depth >> cfg.height >> cfg.gravity >> cfg.max_thrust >> cfg.fuel)) return 0;
    if (!(cin >> cfg.start_x >> cfg.start_y >> cfg.start_z >> cfg.target_x >> cfg.target_y >> cfg.target_z)) return 0;
    
    string dummy;
    getline(cin, dummy); // consume newline
    getline(cin, cfg.terrain_path);
    getline(cin, dummy); // consume "STATE"

    Controller controller(cfg);
    thread terrain_thread([&]() {
        controller.terrain.load(cfg.terrain_path, cfg.width, cfg.depth, cfg.height);
    });
    terrain_thread.detach();

    string line;
    while (getline(cin, line)) {
        if (line.empty()) continue;
        stringstream ss(line);
        State s;
        if (!(ss >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz >> s.pitch >> s.yaw >> s.roll >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) continue;
        
        controller.update_state(s);
        Command cmd = controller.compute_command();

        cout << fixed << setprecision(6) 
             << cmd.thrust << " " 
             << cmd.pitch_cmd << " " 
             << cmd.yaw_cmd << " " 
             << cmd.roll_cmd << endl;
    }

    return 0;
}
```
