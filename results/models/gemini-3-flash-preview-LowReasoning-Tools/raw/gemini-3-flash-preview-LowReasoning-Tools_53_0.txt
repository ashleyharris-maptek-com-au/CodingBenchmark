The orbital docking autopilot is implemented using a robust sensor fusion and tiered control strategy. It handles the counter-intuitive nature of orbital mechanics by adapting its control logic based on the range to the target.

### Key Features:
- **Sensor Fusion:** Merges data from redundant Radar, Lidar, and Range Finders. It cross-checks valid signals and uses range magnitude to scale relative position estimates, ensuring resilience against silent sensor failures.
- **Tiered Control:**
  - **Far Range (>200m):** Uses gentle nudges to align with the V-bar (along-track axis).
  - **Medium/Close Range (<200m):** Employs a PD-controller to gradually close the gap while maintaining zero radial and cross-track offsets.
  - **Final Approach (<10m):** Reduces approach speed to < 0.1 m/s for safe docking.
- **Bias Compensation:** Integral terms (`integral_x`, `integral_y`, `integral_z`) are used to detect and counteract uncommanded thrusts or constant disturbances.
- **Attitude Control:** Maintains zero pitch, roll, and yaw relative to the LVLH frame using star trackers for absolute orientation and gyroscopes for rate damping.

```python
import math

# Persistent state variables
integral_x = 0
integral_y = 0
integral_z = 0

def autopilot_step(state: dict, dt: float) -> dict:
    global integral_x, integral_y, integral_z
    
    # --- 1. Robust Sensor Fusion ---
    def get_valid_avg(prefix, suffix, keys_count):
        vals = []
        for i in range(1, keys_count + 1):
            if state.get(f"{prefix}_{i}_{suffix}_valid") or state.get(f"{prefix}_{i}_{suffix}_vld"):
                val = state.get(f"{prefix}_{i}_{suffix}")
                if val is not None:
                    vals.append(val)
        return sum(vals) / len(vals) if vals else None

    # Position estimation (LVLH)
    pos_x_src = []
    pos_y_src = []
    pos_z_src = []
    
    if state.get('radar_rel_x_valid'): pos_x_src.append(state['radar_rel_pos_x_m'])
    if state.get('radar_rel_y_valid'): pos_y_src.append(state['radar_rel_pos_y_m'])
    if state.get('radar_rel_z_valid'): pos_z_src.append(state['radar_rel_pos_z_m'])
    
    if state.get('lidar_rel_x_valid'): pos_x_src.append(state['lidar_rel_pos_x_m'])
    if state.get('lidar_rel_y_valid'): pos_y_src.append(state['lidar_rel_pos_y_m'])
    if state.get('lidar_rel_z_valid'): pos_z_src.append(state['lidar_rel_pos_z_m'])
    
    avg_pos_x = sum(pos_x_src) / len(pos_x_src) if pos_x_src else 0
    avg_pos_y = sum(pos_y_src) / len(pos_y_src) if pos_y_src else 0
    avg_pos_z = sum(pos_z_src) / len(pos_z_src) if pos_z_src else 0
    
    range_vals = []
    for i in range(1, 4):
        if state.get(f'range_{i}_valid'): range_vals.append(state[f'range_{i}_m'])
    
    est_range = math.sqrt(avg_pos_x**2 + avg_pos_y**2 + avg_pos_z**2)
    if range_vals:
        avg_range_sensor = sum(range_vals) / len(range_vals)
        if est_range > 1.0:
            scale = avg_range_sensor / est_range
            avg_pos_x *= scale
            avg_pos_y *= scale
            avg_pos_z *= scale
            est_range = avg_range_sensor
        elif not pos_x_src and not pos_y_src and not pos_z_src:
            avg_pos_y = -avg_range_sensor if avg_pos_y <= 0 else avg_range_sensor
            est_range = avg_range_sensor

    # Velocity estimation
    vel_x_src = []
    vel_y_src = []
    vel_z_src = []
    
    if state.get('radar_relv_x_valid'): vel_x_src.append(state['radar_rel_vel_x_ms'])
    if state.get('radar_relv_y_valid'): vel_y_src.append(state['radar_rel_vel_y_ms'])
    if state.get('radar_relv_z_valid'): vel_z_src.append(state['radar_rel_vel_z_ms'])
    
    if state.get('lidar_relv_x_valid'): vel_x_src.append(state['lidar_rel_vel_x_ms'])
    if state.get('lidar_relv_y_valid'): vel_y_src.append(state['lidar_rel_vel_y_ms'])
    if state.get('lidar_relv_z_valid'): vel_z_src.append(state['lidar_rel_vel_z_ms'])
    
    avg_vel_x = sum(vel_x_src) / len(vel_x_src) if vel_x_src else 0
    avg_vel_y = sum(vel_y_src) / len(vel_y_src) if vel_y_src else 0
    avg_vel_z = sum(vel_z_src) / len(vel_z_src) if vel_z_src else 0

    # --- 2. Control Strategy ---
    if est_range > 1000: target_vy = 3.0
    elif est_range > 200: target_vy = 1.0
    elif est_range > 50: target_vy = 0.5
    elif est_range > 10: target_vy = 0.2
    else: target_vy = 0.1
    
    if avg_pos_y > 0: target_vy = -target_vy
    if abs(avg_pos_y) < 0.5: target_vy = 0

    kp = 0.1
    kd = 0.6
    
    thrust_y = (0.3 if est_range > 200 else kp) * (target_vy - avg_vel_y)
    thrust_x = -kp * avg_pos_x - kd * avg_vel_x
    thrust_z = -kp * avg_pos_z - kd * avg_vel_z
    
    ki = 0.01
    if est_range < 500:
        integral_x += (0 - avg_pos_x) * dt
        integral_y += (target_vy - avg_vel_y) * dt
        integral_z += (0 - avg_pos_z) * dt
        limit = 15.0
        integral_x = max(min(integral_x, limit), -limit)
        integral_y = max(min(integral_y, limit), -limit)
        integral_z = max(min(integral_z, limit), -limit)
        thrust_x += ki * integral_x
        thrust_y += ki * integral_y
        thrust_z += ki * integral_z

    # --- 3. Attitude Control ---
    def get_star(axis):
        for i in range(1, 3):
            if state.get(f"star_tracker_{i}_{axis}_valid"):
                return state[f"star_tracker_{i}_{axis}_deg"]
        return 0

    att_p, att_r, att_y = get_star("pitch"), get_star("roll"), get_star("yaw")
    rate_p = get_valid_avg("gyro", "pitch_dps", 3) or 0
    rate_r = get_valid_avg("gyro", "roll_dps", 3) or 0
    rate_y = get_valid_avg("gyro", "yaw_dps", 3) or 0
    
    kp_att, kd_att = 0.02, 0.08
    cmd_p = -kp_att * att_p - kd_att * rate_p
    cmd_r = -kp_att * att_r - kd_att * rate_r
    cmd_y = -kp_att * att_y - kd_att * rate_y
    
    return {
        "thrust_x": max(min(thrust_x, 1.0), -1.0),
        "thrust_y": max(min(thrust_y, 1.0), -1.0),
        "thrust_z": max(min(thrust_z, 1.0), -1.0),
        "att_rate_roll": max(min(math.radians(cmd_r), 0.035), -0.035),
        "att_rate_pitch": max(min(math.radians(cmd_p), 0.035), -0.035),
        "att_rate_yaw": max(min(math.radians(cmd_y), 0.035), -0.035)
    }
```
