declare const require: any;
const fs = require('fs');

interface Drone {
    id: string | number;
    x: number;
    y: number;
}

interface InputData {
    probability: number[][];
    drones: Drone[];
    maxStepsPerDrone: number;
}

function solve() {
    const input: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(input);

    const { probability, drones, maxStepsPerDrone } = data;
    const rows = probability.length;
    const cols = probability[0].length;

    const paths: [number, number][][] = drones.map(d => [[d.x, d.y]]);
    const currentPositions = drones.map(d => ({ x: d.x, y: d.y }));
    
    const visited = new Set<string>();
    for (const pos of currentPositions) {
        visited.add(`${pos.x},${pos.y}`);
    }

    const dx = [0, 0, 1, -1, 0]; // Right, Left, Down, Up, Stay
    const dy = [1, -1, 0, 0, 0];

    for (let step = 0; step < maxStepsPerDrone; step++) {
        for (let i = 0; i < drones.length; i++) {
            const curr = currentPositions[i];
            let bestMove = { x: curr.x, y: curr.y };
            let maxScore = -Infinity;

            for (let m = 0; m < 5; m++) {
                const nx = curr.x + dx[m];
                const ny = curr.y + dy[m];

                if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && probability[nx][ny] !== -1) {
                    let score = 0;
                    if (!visited.has(`${nx},${ny}`)) {
                        score = probability[nx][ny];
                    } else {
                        score = -0.000001; 
                    }

                    if (score > maxScore) {
                        maxScore = score;
                        bestMove = { x: nx, y: ny };
                    }
                }
            }

            if (maxScore <= 0) {
                const target = findNearestTarget(curr.x, curr.y, rows, cols, probability, visited);
                if (target) {
                    let bestDist = Infinity;
                    for (let m = 0; m < 4; m++) {
                        const nx = curr.x + dx[m];
                        const ny = curr.y + dy[m];
                        if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && probability[nx][ny] !== -1) {
                            const dist = Math.abs(nx - target.x) + Math.abs(ny - target.y);
                            if (dist < bestDist) {
                                bestDist = dist;
                                bestMove = { x: nx, y: ny };
                            }
                        }
                    }
                }
            }

            currentPositions[i] = bestMove;
            paths[i].push([bestMove.x, bestMove.y]);
            visited.add(`${bestMove.x},${bestMove.y}`);
        }
    }

    console.log(JSON.stringify({ paths }));
}

function findNearestTarget(sx: number, sy: number, rows: number, cols: number, prob: number[][], visited: Set<string>) {
    const queue: [number, number][] = [[sx, sy]];
    const seen = new Set<string>();
    seen.add(`${sx},${sy}`);
    
    let head = 0;
    const maxSearch = 40; 
    let dist = 0;
    let levelSize = 1;

    while (head < queue.length && dist < maxSearch) {
        const [cx, cy] = queue[head++];
        levelSize--;

        if (prob[cx][cy] > 0 && !visited.has(`${cx},${cy}`)) {
            return { x: cx, y: cy };
        }

        const dx = [0, 0, 1, -1];
        const dy = [1, -1, 0, 0];

        for (let i = 0; i < 4; i++) {
            const nx = cx + dx[i];
            const ny = cy + dy[i];
            if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && prob[nx][ny] !== -1 && !seen.has(`${nx},${ny}`)) {
                seen.add(`${nx},${ny}`);
                queue.push([nx, ny]);
            }
        }

        if (levelSize === 0) {
            dist++;
            levelSize = queue.length - head;
        }
    }
    return null;
}

solve();
