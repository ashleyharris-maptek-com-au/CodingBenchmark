```spirv
; SPIR-V
Version 1.0
Generator 0
Bound 100
Schema 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for gl_PerVertex_in
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for gl_PerVertex_out
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations for user inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2

; Decorations for user outputs
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0_05 = OpConstant %float 0.05
%float_1 = OpConstant %float 1
%float_0 = OpConstant %float 0

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User inputs arrays (size 1 for InputPoints)
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User outputs
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types for AccessChain
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load center position (from worldPos) and color
%ptr_pos = OpAccessChain %ptr_in_v3 %worldPos %int_0
%pos = OpLoad %v3float %ptr_pos

%ptr_col = OpAccessChain %ptr_in_v3 %inColor %int_0
%col = OpLoad %v3float %ptr_col

; Extract x, y, z from the position
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2

; Calculate half-size offsets
%h = OpCopyObject %float %float_0_05

; Top vertex: (x, y+h)
%top_y = OpFAdd %float %y %h
%top_pos = OpCompositeConstruct %v3float %x %top_y %z
%top_v4 = OpCompositeConstruct %v4float %top_pos %float_1

; Right vertex: (x+h, y)
%right_x = OpFAdd %float %x %h
%right_pos = OpCompositeConstruct %v3float %right_x %y %z
%right_v4 = OpCompositeConstruct %v4float %right_pos %float_1

; Left vertex: (x-h, y)
%left_x = OpFSub %float %x %h
%left_pos = OpCompositeConstruct %v3float %left_x %y %z
%left_v4 = OpCompositeConstruct %v4float %left_pos %float_1

; Bottom vertex: (x, y-h)
%bot_y = OpFSub %float %y %h
%bot_pos = OpCompositeConstruct %v3float %x %bot_y %z
%bot_v4 = OpCompositeConstruct %v4float %bot_pos %float_1

; Emit Top Vertex
%ptr_gl_out_top = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_gl_out_top %top_v4
OpStore %outColor %col
OpEmitVertex

; Emit Right Vertex
%ptr_gl_out_right = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_gl_out_right %right_v4
OpStore %outColor %col
OpEmitVertex

; Emit Left Vertex
%ptr_gl_out_left = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_gl_out_left %left_v4
OpStore %outColor %col
OpEmitVertex

; Emit Bottom Vertex
%ptr_gl_out_bot = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_gl_out_bot %bot_v4
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
