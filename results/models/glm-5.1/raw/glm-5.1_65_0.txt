declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
    id: number;
    duration: number;
    deps: number[];
    deadline: number;
    weight: number;
}

interface InputData {
    machineCount: number;
    tasks: Task[];
}

const { machineCount, tasks } = data as InputData;
const n = tasks.length;

const idToIndex = new Map<number, number>();
for (let i = 0; i < n; i++) {
    idToIndex.set(tasks[i].id, i);
}

const duration = new Float64Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);
const numDeps = new Int32Array(n);
const adjList: number[][] = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
    duration[i] = tasks[i].duration;
    deadline[i] = tasks[i].deadline;
    weight[i] = tasks[i].weight;
    numDeps[i] = tasks[i].deps.length;
    for (const dep of tasks[i].deps) {
        const j = idToIndex.get(dep)!;
        adjList[j].push(i);
    }
}

// Reverse graph for longest path to sink
const revAdjList: number[][] = Array.from({ length: n }, () => []);
const outDegree = new Int32Array(n);
for (let i = 0; i < n; i++) {
    for (const j of adjList[i]) {
        revAdjList[j].push(i);
        outDegree[i]++;
    }
}

// Kahn's algorithm to compute longest path to sink (topological order on reversed graph)
const Q: number[] = [];
const distToSink = new Float64Array(n);
let head = 0;

for (let i = 0; i < n; i++) {
    if (outDegree[i] === 0) {
        Q.push(i);
        distToSink[i] = duration[i];
    }
}

while (head < Q.length) {
    const u = Q[head++];
    for (const v of revAdjList[u]) {
        const d = distToSink[u] + duration[v];
        if (d > distToSink[v]) {
            distToSink[v] = d;
        }
        outDegree[v]--;
        if (outDegree[v] === 0) {
            Q.push(v);
        }
    }
}

// Compute longest path from source (forward pass)
const distFromSource = new Float64Array(n);
const inDegree = new Int32Array(n);
const fwdQ: number[] = [];
let fwdHead = 0;

for (let i = 0; i < n; i++) {
    inDegree[i] = numDeps[i];
    if (inDegree[i] === 0) {
        fwdQ.push(i);
        distFromSource[i] = 0;
    }
}

while (fwdHead < fwdQ.length) {
    const u = fwdQ[fwdHead++];
    for (const v of adjList[u]) {
        const d = distFromSource[u] + duration[u];
        if (d > distFromSource[v]) {
            distFromSource[v] = d;
        }
        inDegree[v]--;
        if (inDegree[v] === 0) {
            fwdQ.push(v);
        }
    }
}

// Calculate heuristic priority
const criticalPath = Math.max(...distToSink);

const priority = new Float64Array(n);
for (let i = 0; i < n; i++) {
    if (criticalPath === 0) {
        priority[i] = 0;
    } else {
        // Weighted combination of earliest start time and critical path length to sink
        const startFactor = distFromSource[i] / criticalPath;
        const sinkFactor = distToSink[i] / criticalPath;
        // Tasks that are urgent (large weight, tight deadline) should also be prioritized
        const urgency = (weight[i] * duration[i]) / Math.max(1, deadline[i] - distFromSource[i] + 1);
        const urgencyNorm = urgency / (criticalPath * 10);
        priority[i] = 0.4 * sinkFactor + 0.4 * startFactor + 0.2 * Math.min(1, urgencyNorm);
    }
}

// Event-driven parallel list scheduler
const machineFreeAt = new Float64Array(machineCount).fill(0);
const earliestStart = new Float64Array(n);
const finishedAt = new Float64Array(n);
const remainingDeps = new Int32Array(numDeps);

// Priority queue for available tasks (max-heap by priority)
const heap: number[] = [];
let heapSize = 0;

function heapPush(val: number) {
    heap[heapSize] = val;
    let i = heapSize++;
    while (i > 0) {
        const parent = (i - 1) >> 1;
        if (priority[heap[parent]] < priority[heap[i]]) {
            const tmp = heap[parent];
            heap[parent] = heap[i];
            heap[i] = tmp;
            i = parent;
        } else {
            break;
        }
    }
}

function heapPop(): number {
    const top = heap[0];
    heap[0] = heap[--heapSize];
    let i = 0;
    while (true) {
        let largest = i;
        const l = 2 * i + 1;
        const r = 2 * i + 2;
        if (l < heapSize && priority[heap[l]] > priority[heap[largest]]) largest = l;
        if (r < heapSize && priority[heap[r]] > priority[heap[largest]]) largest = r;
        if (largest !== i) {
            const tmp = heap[largest];
            heap[largest] = heap[i];
            heap[i] = tmp;
            i = largest;
        } else {
            break;
        }
    }
    return top;
}

// Events priority queue (min-heap by time)
interface Event {
    time: number;
    taskIndex: number;
}

const events: Event[] = [];
let eventSize = 0;

function eventPush(time: number, taskIndex: number) {
    events[eventSize] = { time, taskIndex };
    let i = eventSize++;
    while (i > 0) {
        const parent = (i - 1) >> 1;
        if (events[parent].time > events[i].time) {
            const tmp = events[parent];
            events[parent] = events[i];
            events[i] = tmp;
            i = parent;
        } else {
            break;
        }
    }
}

function eventPop(): Event {
    const top = events[0];
    events[0] = events[--eventSize];
    let i = 0;
    while (true) {
        let smallest = i;
        const l = 2 * i + 1;
        const r = 2 * i + 2;
        if (l < eventSize && events[l].time < events[smallest].time) smallest = l;
        if (r < eventSize && events[r].time < events[smallest].time) smallest = r;
        if (smallest !== i) {
            const tmp = events[smallest];
            events[smallest] = events[i];
            events[i] = tmp;
            i = smallest;
        } else {
            break;
        }
    }
    return top;
}

// Initialize earliest start and push root tasks
for (let i = 0; i < n; i++) {
    if (remainingDeps[i] === 0) {
        heapPush(i);
        earliestStart[i] = 0;
    }
}

const schedule: { id: number; start: number; machine: number }[] = new Array(n);
let scheduledCount = 0;

// Process available tasks greedily
function scheduleAvailable() {
    while (heapSize > 0) {
        // Find earliest available machine
        let minTime = Infinity;
        let minM = -1;
        for (let m = 0; m < machineCount; m++) {
            if (machineFreeAt[m] < minTime) {
                minTime = machineFreeAt[m];
                minM = m;
            }
        }

        const task = heapPop();
        const start = Math.max(minTime, earliestStart[task]);
        const end = start + duration[task];

        schedule[task] = { id: tasks[task].id, start, machine: minM };
        scheduledCount++;

        machineFreeAt[minM] = end;
        finishedAt[task] = end;

        // Release dependents
        for (const dep of adjList[task]) {
            remainingDeps[dep]--;
            if (finishedAt[task] > earliestStart[dep]) {
                earliestStart[dep] = finishedAt[task];
            }
            if (remainingDeps[dep] === 0) {
                heapPush(dep);
            }
        }

        // Schedule completion event
        eventPush(end, task);
    }
}

// Add initial tasks
scheduleAvailable();

while (scheduledCount < n) {
    // Find the next moment a task could become available
    // We need to find the earliest machine free time that is >= current earliestStart of some pending task
    // Actually, we advance to the next event time which is a task completion
    const event = eventPop();
    const t = event.time;

    // Since we schedule greedily, when a machine becomes free we should try to fill it
    // But wait, our eventPop might pop events from tasks we already scheduled.
    // Let's just process time steps when tasks finish.
    
    // Re-evaluate machine availability at this time
    // Actually, a simpler and correct approach is:
    // When we pop an event, it means a task finished. We already released its dependents in scheduleAvailable.
    // So the only new tasks are those whose dependencies all finished.
    // But we already push them to the heap when their remainingDeps hit 0.
    // So if the heap is empty, we MUST jump to the next event time to free up machines and possibly unlock tasks.
    
    // The issue is: we need a machine to become free to schedule the next task.
    // We jump to the time when the earliest machine becomes free.
    // But tasks only become available when their deps finish.
    
    // Let's reconsider. We want to find the minimum time T such that:
    // 1. There is at least one task in the heap
    // 2. There is at least one machine free at T
    // If the heap is empty, we must advance time to when the next task finishes.
    
    // When an event pops, it means a task finished at time `t`. But we already handled the dependent releases.
    // Wait, no! If the heap was empty, it means no task was ready. When an event pops, we might have unlocked a task.
    // Let's check if any task became ready.
    // This logic is slightly tricky because we released dependents immediately in scheduleAvailable.
    // So if an event pops and the heap is STILL empty, it means we need to wait for MORE events.
    
    // If the heap is NOT empty, we can just call scheduleAvailable() again.
    // But wait, if we call scheduleAvailable(), it might assign a task to a machine that is currently busy,
    // scheduling it to start in the future. That's perfectly fine! The start time will be when the machine is free.
    
    // Is there a case where jumping to events is bad?
    // Suppose heap is empty. We MUST wait for an event. But we already processed the event's side effects (unlocking tasks)!
    // So if the event didn't unlock anything, the heap is still empty. We just discard the event and pop the next one.
}

// Wait, the above loop is wrong. We don't actually need the event queue for unlocking tasks because we unlock them immediately upon scheduling.
// The event queue is only needed if we want to simulate time.
// But we CAN just schedule everything greedily without simulating time!
// Let's rewrite the scheduler to be simpler and correctly greedy.

// Simpler greedy scheduler:
// 1. Maintain a set of ready tasks (heap).
// 2. Maintain machine free times.
// 3. While there are unscheduled tasks:
//    a. If heap is empty, advance time to the next finishing time of a scheduled task.
//       This will unlock some tasks. Add them to the heap.
//    b. Pop a task from the heap.
//    c. Find the earliest free machine. Start time = max(earliestStart[task], machineFreeAt[machine]).
//    d. Schedule the task.

// Let's reset and implement this simpler logic.
// We already have `schedule`, `scheduledCount`, `earliestStart`, `remainingDeps`, `machineFreeAt`.
// But we need to correctly handle advancing time.

// Let's just restart the scheduling logic with a cleaner state.
const machineFreeAt2 = new Float64Array(machineCount).fill(0);
const earliestStart2 = new Float64Array(n);
const remainingDeps2 = new Int32Array(numDeps);
const finishedAt2 = new Float64Array(n);
const schedule2: { id: number; start: number; machine: number }[] = new Array(n);
let scheduledCount2 = 0;

const heap2: number[] = [];
let heapSize2 = 0;

function heapPush2(val: number) {
    heap2[heapSize2] = val;
    let i = heapSize2++;
    while (i > 0) {
        const parent = (i - 1) >> 1;
        if (priority[heap2[parent]] < priority[heap2[i]]) {
            const tmp = heap2[parent]; heap2[parent] = heap2[i]; heap2[i] = tmp; i = parent;
        } else break;
    }
}

function heapPop2(): number {
    const top = heap2[0];
    heap2[0] = heap2[--heapSize2];
    let i = 0;
    while (true) {
        let largest = i;
        const l = 2 * i + 1, r = 2 * i + 2;
        if (l < heapSize2 && priority[heap2[l]] > priority[heap2[largest]]) largest = l;
        if (r < heapSize2 && priority[heap2[r]] > priority[heap2[largest]]) largest = r;
        if (largest !== i) {
            const tmp = heap2[largest]; heap2[largest] = heap2[i]; heap2[i] = tmp; i = largest;
        } else break;
    }
    return top;
}

for (let i = 0; i < n; i++) {
    if (remainingDeps2[i] === 0) {
        heapPush2(i);
    }
}

while (scheduledCount2 < n) {
    if (heapSize2 === 0) {
        // No tasks are ready. We must wait for some scheduled tasks to finish.
        // Find the minimum finishedAt2 among scheduled but not yet "processed for unlocking" tasks.
        // Wait, we unlock tasks immediately when we schedule them!
        // So if heapSize2 is 0, it means ALL tasks that depend on currently scheduled tasks are already in the heap (which is empty, so contradiction?)
        // NO! If we unlock tasks immediately, their earliestStart2 is set. They are pushed to the heap.
        // If the heap is empty, it means ALL tasks have been scheduled! (Because a DAG always has at least one ready task unless it's empty).
        // Wait, is this true? Yes, if we pop a task, schedule it, and unlock its children, we push them to the heap.
        // So the heap only becomes empty if there are no more tasks to schedule.
        // So we DON'T need an event queue! The greedy scheduler works naturally:
        
        // Pop task -> Schedule it -> Unlock children -> Push children to heap.
        // We just need to pick the best machine for the task.
        break; // Should naturally break when scheduledCount2 === n
    }

    const task = heapPop2();

    // Find the machine that allows the earliest start for this task
    let minStartTime = Infinity;
    let bestMachine = -1;
    for (let m = 0; m < machineCount; m++) {
        const start = Math.max(earliestStart2[task], machineFreeAt2[m]);
        if (start < minStartTime) {
            minStartTime = start;
            bestMachine = m;
        }
    }

    const start = minStartTime;
    const end = start + duration[task];

    schedule2[scheduledCount2++] = { id: tasks[task].id, start, machine: bestMachine };
    machineFreeAt2[bestMachine] = end;
    finishedAt2[task] = end;

    for (const dep of adjList[task]) {
        remainingDeps2[dep]--;
        if (finishedAt2[task] > earliestStart2[dep]) {
            earliestStart2[dep] = finishedAt2[task];
        }
        if (remainingDeps2[dep] === 0) {
            heapPush2(dep);
        }
    }
}

const output = { schedule: schedule2 };
