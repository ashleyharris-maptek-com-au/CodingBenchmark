; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fn_main = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0
%float_1_0 = OpConstant %float 1.0

%v3_base_r = OpConstant %float 0.8
%v3_base_g = OpConstant %float 0.2
%v3_base_b = OpConstant %float 0.2
%base_color = OpConstantComposite %vec3 %v3_base_r %v3_base_g %v3_base_b

%v3_fog_r = OpConstant %float 0.7
%v3_fog_g = OpConstant %float 0.7
%v3_fog_b = OpConstant %float 0.8
%fog_color = OpConstantComposite %vec3 %v3_fog_r %v3_fog_g %v3_fog_b

%main = OpFunction %void None %fn_main
%entry = OpLabel

; Load worldPos and normal
%worldPos = OpLoad %vec3 %in_worldPos
%raw_normal = OpLoad %vec3 %in_normal
%normal = OpNormalize %vec3 %raw_normal

; Load lightPos (offset 192)
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos_v4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_v4 %lightPos_v4 0 1 2

; Load cameraPos (offset 208)
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%cameraPos_v4 = OpLoad %vec4 %cameraPos_ptr
%cameraPos = OpVectorShuffle %vec3 %cameraPos_v4 %cameraPos_v4 0 1 2

; Compute light direction and distance
%lightDir = OpFSub %vec3 %lightPos %worldPos
%lightDist = OpLength %float %lightDir
%lightDirNorm = OpFNegate %vec3 %lightDir
%L = OpFDiv %vec3 %lightDirNorm %lightDist

; Compute Lambertian diffuse factor
%NdotL = OpDot %float %normal %L
%diffuse = OpFMax %float %NdotL %float_0_0

; Compute base color with Lambertian factor (0.3 + 0.7*diffuse)
%ambient = OpFMul %float %float_0_7 %diffuse
%intensity = OpFAdd %float %float_0_3 %ambient
%lit_base = OpVectorTimesScalar %vec3 %base_color %intensity

; Compute distance from camera
%viewDir = OpFSub %vec3 %cameraPos %worldPos
%dist = OpLength %float %viewDir

; Compute fog factor: clamp((dist - 1.5) / 3.0)
%dist_minus_min = OpFSub %float %dist %float_1_5
%fog_unclamped = OpFDiv %float %dist_minus_min %float_3_0
%fog_factor = OpFClamp %float %fog_unclamped %float_0_0 %float_1_0

; Mix lit base color and fog color
%mixed = OpFMix %vec3 %lit_base %fog_color %fog_factor

; Construct final output color with alpha 1.0
%final_v4 = OpCompositeConstruct %vec4 %mixed %float_1_0
OpStore %outColor %final_v4

OpReturn
OpFunctionEnd
