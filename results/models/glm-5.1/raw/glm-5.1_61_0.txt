```typescript
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Point {
    x: number;
    y: number;
}

interface Package extends Point {
    id: number;
}

function dist(a: Point, b: Point): number {
    const dx = a.x - b.x;
    const dy = a.y - b.y;
    return Math.sqrt(dx * dx + dy * dy);
}

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const truckCount: number = data.truckCount;
const packages: Package[] = data.packages;

if (packages.length === 0) {
    process.stdout.write(JSON.stringify({ routes: [] }));
    process.exit(0);
}

// Calculate polar angle from depot for sweep algorithm
const angles = new Float64Array(packages.length);
for (let i = 0; i < packages.length; i++) {
    angles[i] = Math.atan2(packages[i].y - depot.y, packages[i].x - depot.x);
}

const indices = Array.from({ length: packages.length }, (_, i) => i);
indices.sort((a, b) => angles[a] - angles[b]);

const sortedPackages = indices.map(i => packages[i]);

// Divide the sweep into truckCount routes
const routes: number[][] = [];
const packagesPerRoute = Math.ceil(packages.length / truckCount);

for (let r = 0; r < truckCount; r++) {
    const start = r * packagesPerRoute;
    const end = Math.min(start + packagesPerRoute, packages.length);
    if (start >= packages.length) break;
    
    const routePackages = sortedPackages.slice(start, end);
    const routeIds = routePackages.map(p => p.id);
    
    // Optimize the single route using 2-opt local search
    const n = routeIds.length;
    if (n <= 1) {
        routes.push(routeIds);
        continue;
    }
    
    const routeDist = new Float64Array(n + 1);
    const calcRouteDist = () => {
        routeDist[0] = dist(depot, routePackages[0]);
        for (let i = 1; i < n; i++) {
            routeDist[i] = routeDist[i - 1] + dist(routePackages[i - 1], routePackages[i]);
        }
        routeDist[n] = routeDist[n - 1] + dist(routePackages[n - 1], depot);
    };
    
    calcRouteDist();
    let improved = true;
    
    // 2-opt optimization loop
    while (improved) {
        improved = false;
        for (let i = 0; i < n - 1; i++) {
            for (let j = i + 1; j < n; j++) {
                // Evaluate the distance change of reversing the segment from i to j
                let prevI = (i === 0) ? depot : routePackages[i - 1];
                let nextJ = (j === n - 1) ? depot : routePackages[j + 1];
                
                const d1 = dist(prevI, routePackages[i]) + dist(routePackages[j], nextJ);
                const d2 = dist(prevI, routePackages[j]) + dist(routePackages[i], nextJ);
                
                if (d2 < d1 - 1e-10) {
                    // Reverse the sub-array from i to j in-place
                    let left = i;
                    let right = j;
                    while (left < right) {
                        const tmpPkg = routePackages[left];
                        routePackages[left] = routePackages[right];
                        routePackages[right] = tmpPkg;
                        
                        const tmpId = routeIds[left];
                        routeIds[left] = routeIds[right];
                        routeIds[right] = tmpId;
                        
                        left++;
                        right--;
                    }
                    
                    calcRouteDist();
                    improved = true;
                }
            }
        }
    }
    
    routes.push(routeIds);
}

process.stdout.write(JSON.stringify({ routes }));
```
