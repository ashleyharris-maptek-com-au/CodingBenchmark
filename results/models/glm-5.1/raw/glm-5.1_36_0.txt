#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

// Globals for algorithm state
vector<vector<int>> adj;
vector<int> deg;
vector<bool> active;
vector<int> current_set;
int n, m;

// Mask of available vertices for branch-and-bound
vector<unsigned long long> avail;
int avail_cnt;

// Bit manipulation helpers
inline int which(int v) { return v >> 6; }
inline unsigned long long bit(int v) { return 1ULL << (v & 63); }

// Branch and bound search
// curr_size: current independent set size
// best: reference to the best size found so far
void dfs(int curr_size, int& best) {
    if (avail_cnt == 0) {
        if (curr_size > best) best = curr_size;
        return;
    }

    // Upper bound: cannot improve best
    if (curr_size + avail_cnt <= best) return;

    // Find vertex with maximum degree among available
    int v = -1, max_deg = -1;
    for (int i = 0; i <= which(n - 1); ++i) {
        unsigned long long mask = avail[i];
        while (mask) {
            int j = __builtin_ctzll(mask);
            int u = (i << 6) + j;
            if (deg[u] > max_deg) {
                max_deg = deg[u];
                v = u;
            }
            mask &= mask - 1;
        }
    }

    // Save state
    vector<unsigned long long> saved = avail;
    int saved_cnt = avail_cnt;

    // Branch 1: Exclude v
    avail[which(v)] &= ~bit(v);
    avail_cnt--;
    dfs(curr_size, best);

    // Restore state
    avail = saved;
    avail_cnt = saved_cnt;

    // Branch 2: Include v
    avail[which(v)] &= ~bit(v);
    avail_cnt--;
    for (int nb : adj[v]) {
        if (avail[which(nb)] & bit(nb)) {
            avail[which(nb)] &= ~bit(nb);
            avail_cnt--;
        }
    }
    dfs(curr_size + 1, best);

    // Restore state for parent
    avail = saved;
    avail_cnt = saved_cnt;
}

// Iterative basic reductions
void basic_reduce() {
    bool changed = true;
    while (changed) {
        changed = false;
        for (int v = 0; v < n; ++v) {
            if (!active[v]) continue;
            if (deg[v] == 0) {
                current_set.push_back(v);
                active[v] = false;
                for (int nb : adj[v]) deg[nb]--;
                changed = true;
            } else if (deg[v] == 1) {
                int nb = -1;
                for (int u : adj[v]) if (active[u]) { nb = u; break; }
                current_set.push_back(nb);
                active[nb] = false;
                active[v] = false;
                deg[nb]--;
                for (int u : adj[nb]) if (active[u]) deg[u]--;
                changed = true;
            }
        }
    }
}

// LP relaxation reduction using maximum matching
vector<int> lp_reduce() {
    int N = 0;
    vector<int> map_vertex(n, -1);
    for (int v = 0; v < n; ++v) {
        if (active[v]) map_vertex[v] = N++;
    }
    if (N == 0) return {};

    vector<vector<int>> G(N);
    for (int v = 0; v < n; ++v) {
        if (!active[v]) continue;
        for (int nb : adj[v]) {
            if (active[nb] && nb > v) {
                G[map_vertex[v]].push_back(map_vertex[nb]);
                G[map_vertex[nb]].push_back(map_vertex[v]);
            }
        }
    }

    vector<int> match(N, -1);
    for (int u = 0; u < N; ++u) {
        vector<bool> visited(N, false);
        auto augment = [&](auto& self, int u) -> bool {
            for (int v : G[u]) {
                if (!visited[v]) {
                    visited[v] = true;
                    if (match[v] == -1 || self(self, match[v])) {
                        match[v] = u;
                        match[u] = v;
                        return true;
                    }
                }
            }
            return false;
        };
        augment(augment, u);
    }

    int matching_size = 0;
    for (int i = 0; i < N; ++i) if (match[i] != -1) matching_size++;
    matching_size /= 2;

    vector<bool> covered(N, false);
    for (int i = 0; i < N; ++i) if (match[i] != -1) covered[i] = true;

    // Alternate paths from unmatched vertices
    vector<int> q;
    for (int i = 0; i < N; ++i) if (match[i] == -1) q.push_back(i);
    for (int i = 0; i < (int)q.size(); ++i) {
        int u = q[i];
        for (int v : G[u]) {
            if (!covered[v] && match[v] != -1) {
                covered[v] = true;
                covered[match[v]] = true;
                q.push_back(match[v]);
            }
        }
    }

    // Cycles in matched edges
    vector<bool> seen(N, false);
    for (int i = 0; i < N; ++i) {
        if (match[i] != -1 && !seen[i]) {
            vector<int> cycle;
            int u = i;
            while (u != -1 && !seen[u]) {
                seen[u] = true;
                cycle.push_back(u);
                u = match[u];
            }
            for (int v : cycle) {
                if (covered[v]) break;
                for (int nb : G[v]) {
                    if (!covered[nb]) {
                        covered[nb] = true;
                        if (match[nb] != -1 && !covered[match[nb]]) {
                            covered[match[nb]] = true;
                            q.push_back(match[nb]);
                        }
                    }
                }
            }
        }
    }

    for (int v : q) {
        for (int nb : G[v]) {
            if (!covered[nb]) {
                covered[nb] = true;
                if (match[nb] != -1 && !covered[match[nb]]) {
                    covered[match[nb]] = true;
                    q.push_back(match[nb]);
                }
            }
        }
    }

    // Map back to original vertices
    vector<int> remove;
    for (int v = 0; v < n; ++v) {
        if (active[v] && !covered[map_vertex[v]]) remove.push_back(v);
    }
    return remove;
}

void apply_lp_reduce() {
    vector<int> remove = lp_reduce();
    for (int v : remove) {
        active[v] = false;
        for (int nb : adj[v]) deg[nb]--;
    }
}

void apply_crown_reduce() {
    while (true) {
        basic_reduce();
        vector<int> I, H;
        for (int v = 0; v < n; ++v) {
            if (active[v]) {
                if (deg[v] <= 1) I.push_back(v);
                else H.push_back(v);
            }
        }
        if (I.empty()) break;

        int NI = I.size(), NH = H.size();
        vector<int> I_map(NI), H_map(NH);
        for (int i = 0; i < NI; ++i) I_map[i] = i;
        for (int i = 0; i < NH; ++i) H_map[i] = i;

        vector<vector<int>> G(NI + NH);
        for (int i = 0; i < NI; ++i) {
            for (int nb : adj[I[i]]) {
                if (active[nb]) {
                    auto it = lower_bound(H.begin(), H.end(), nb);
                    int j = it - H.begin();
                    G[i].push_back(NI + j);
                    G[NI + j].push_back(i);
                }
            }
        }

        vector<int> match(NI + NH, -1);
        for (int u = 0; u < NI; ++u) {
            vector<bool> visited(NI + NH, false);
            auto augment = [&](auto& self, int u) -> bool {
                for (int v : G[u]) {
                    if (!visited[v]) {
                        visited[v] = true;
                        if (match[v] == -1 || self(self, match[v])) {
                            match[v] = u;
                            match[u] = v;
                            return true;
                        }
                    }
                }
                return false;
            };
            augment(augment, u);
        }

        vector<bool> covered(NI + NH, false);
        for (int u = 0; u < NI; ++u) {
            if (match[u] == -1) {
                vector<int> q;
                q.push_back(u);
                covered[u] = true;
                for (int i = 0; i < (int)q.size(); ++i) {
                    int x = q[i];
                    for (int v : G[x]) {
                        if (!covered[v]) {
                            covered[v] = true;
                            if (match[v] != -1 && !covered[match[v]]) {
                                covered[match[v]] = true;
                                q.push_back(match[v]);
                            }
                        }
                    }
                }
            }
        }

        bool changed = false;
        for (int i = 0; i < NI; ++i) {
            if (covered[i] && match[i] != -1) {
                int v = I[i];
                current_set.push_back(v);
                active[v] = false;
                for (int nb : adj[v]) deg[nb]--;
                int j = match[i] - NI;
                int h = H[j];
                active[h] = false;
                for (int nb : adj[h]) deg[nb]--;
                changed = true;
            }
        }
        if (!changed) break;
    }
}

// Main solver combining reductions and BnB
int solve() {
    basic_reduce();
    apply_lp_reduce();
    basic_reduce();
    apply_crown_reduce();
    basic_reduce();

    int remaining = 0;
    for (int v = 0; v < n; ++v) if (active[v]) remaining++;

    if (remaining == 0) return current_set.size();

    avail.assign((n >> 6) + 1, 0);
    avail_cnt = 0;
    for (int v = 0; v < n; ++v) {
        if (active[v]) {
            avail[which(v)] |= bit(v);
            avail_cnt++;
        }
    }

    int best = 0;
    dfs(0, best);
    return current_set.size() + best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    adj.resize(n);
    deg.resize(n, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        deg[u]++;
        deg[v]++;
    }

    active.assign(n, true);
    int result = solve();

    cout << result << "\n";
    for (int i = 0; i < result; ++i) {
        if (i > 0) cout << " ";
        cout << current_set[i];
    }
    if (result > 0) cout << "\n";

    return 0;
}
