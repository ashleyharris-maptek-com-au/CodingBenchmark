```typescript
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Town {
    id: string | number;
    x: number;
    y: number;
    value: number;
}

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions;
const towns: Town[] = data.towns;
const cutBudget: number = data.cutBudget;

const N = width * height;
const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];

// Initialize grid: 0 = burnable, 1 = non-burnable, 2 = ignition
const init_grid = new Uint8Array(N);
for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
        const u = y * width + x;
        if (grid[y][x] === '~') {
            init_grid[u] = 1;
        }
    }
}

// Precompute town values at each cell
const townValues = new Float64Array(N);
for (const t of towns) {
    const u = t.y * width + t.x;
    townValues[u] += t.value;
}

// Frontier BFS from ignitions
const fireTime = new Int32Array(N);
fireTime.fill(-1);
const queue = new Int32Array(N);
let qHead = 0;
let qTail = 0;

for (const [x, y] of ignitions) {
    const u = y * width + x;
    if (init_grid[u] === 0) {
        init_grid[u] = 2;
        fireTime[u] = 0;
        queue[qTail++] = u;
    }
}

while (qHead < qTail) {
    const u = queue[qHead++];
    const ux = u % width;
    const uy = (u - ux) / width;
    const t = fireTime[u] + 1;
    for (let d = 0; d < 4; d++) {
        const vx = ux + dx[d];
        const vy = uy + dy[d];
        if (vx >= 0 && vx < width && vy >= 0 && vy < height) {
            const v = vy * width + vx;
            if (init_grid[v] === 0) {
                init_grid[v] = 2;
                fireTime[v] = t;
                queue[qTail++] = v;
            }
        }
    }
}

// Calculate value saved by cutting each cell (exact simulation with cuts)
const savedVals = new Float64Array(N);
const valQueue = new Int32Array(N);
const valParent = new Int32Array(N).fill(-1);
const valBfsId = new Uint32Array(N);
let bfsCounter = 0;

for (let u = 0; u < N; u++) {
    if (init_grid[u] !== 0) continue;
    const cutT = fireTime[u];

    let totalSaved = 0;
    let bfsSize = 0;
    bfsCounter++;

    // Start BFS from u
    let localHead = 0;
    let localTail = 0;
    valBfsId[u] = bfsCounter;
    valParent[u] = -1;
    valQueue[localTail++] = u;

    while (localHead < localTail) {
        const v = valQueue[localHead++];
        bfsSize++;
        totalSaved += townValues[v];
        const vx = v % width;
        const vy = (v - vx) / width;
        for (let d = 0; d < 4; d++) {
            const wx = vx + dx[d];
            const wy = vy + dy[d];
            if (wx >= 0 && wx < width && wy >= 0 && wy < height) {
                const w = wy * width + wx;
                if (init_grid[w] === 0 && valBfsId[w] !== bfsCounter) {
                    const wt = fireTime[w];
                    if (wt < 0 || wt > cutT) {
                        valBfsId[w] = bfsCounter;
                        valParent[w] = v;
                        valQueue[localTail++] = w;
                    }
                }
            }
        }
    }

    if (totalSaved > 0) {
        // Backtrack: assign value proportionally based on depth
        for (let i = bfsSize - 1; i >= 0; i--) {
            const v = valQueue[i];
            const p = valParent[v];
            if (p >= 0) {
                savedVals[p] += savedVals[v] * (2 / 3);
            }
        }
        savedVals[u] += totalSaved;
    }
}

// Greedy selection of cuts
const cuts: number[][] = [];
const isCut = new Uint8Array(N);

for (let iter = 0; iter < cutBudget; iter++) {
    let bestCell = -1;
    let bestVal = 0;

    for (let u = 0; u < N; u++) {
        if (init_grid[u] === 0 && !isCut[u] && savedVals[u] > bestVal) {
            bestVal = savedVals[u];
            bestCell = u;
        }
    }

    if (bestCell < 0) break;
    isCut[bestCell] = 1;
    const bx = bestCell % width;
    const by = (bestCell - bx) / width;
    cuts.push([bx, by]);

    // Recalculate exact saved value for remaining adjacent uncut cells
    for (let d = 0; d < 4; d++) {
        const nx = bx + dx[d];
        const ny = by + dy[d];
        if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
            const n = ny * width + nx;
            if (init_grid[n] === 0 && !isCut[n]) {
                let newSaved = 0;
                bfsCounter++;
                let localHead = 0;
                let localTail = 0;
                valBfsId[n] = bfsCounter;
                valQueue[localTail++] = n;

                while (localHead < localTail) {
                    const v = valQueue[localHead++];
                    newSaved += townValues[v];
                    const vx = v % width;
                    const vy = (v - vx) / width;
                    for (let dd = 0; dd < 4; dd++) {
                        const wx = vx + dx[dd];
                        const wy = vy + dy[dd];
                        if (wx >= 0 && wx < width && wy >= 0 && wy < height) {
                            const w = wy * width + wx;
                            if (init_grid[w] === 0 && !isCut[w] && valBfsId[w] !== bfsCounter) {
                                const wt = fireTime[w];
                                const ct = fireTime[n];
                                if (wt < 0 || wt > ct) {
                                    valBfsId[w] = bfsCounter;
                                    valQueue[localTail++] = w;
                                }
                            }
                        }
                    }
                }

                // Apply backtracking for proportional distribution for future candidates
                if (newSaved > 0) {
                    savedVals[n] = 0;
                    const cbs = localTail;
                    for (let i = cbs - 1; i >= 0; i--) {
                        const v = valQueue[i];
                        const vx = v % width;
                        const vy = (v - vx) / width;
                        for (let dd = 0; dd < 4; dd++) {
                            const wx = vx + dx[dd];
                            const wy = vy + dy[dd];
                            if (wx >= 0 && wx < width && wy >= 0 && wy < height) {
                                const w = wy * width + wx;
                                if (init_grid[w] === 0 && !isCut[w] && valBfsId[w] === bfsCounter) {
                                    if (fireTime[w] < 0 || fireTime[w] > fireTime[v]) {
                                        savedVals[v] += savedVals[w] * (2 / 3);
                                    }
                                }
                            }
                        }
                    }
                    savedVals[n] += newSaved;
                } else {
                    savedVals[n] = 0;
                }
            }
        }
    }
    // Zero out the selected cell so it's ignored
    savedVals[bestCell] = 0;
}

console.log(JSON.stringify({ firebreaks: cuts }));
```
