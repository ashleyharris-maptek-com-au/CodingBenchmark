```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float NdotL = max(dot(N, L), 0.0);
    float diffuseFactor = 0.4 + 0.6 * NdotL;

    vec2 scaledUV = uv * vec2(8.0, 16.0);

    // Apply staggered offset for odd rows
    if (mod(floor(scaledUV.y), 2.0) > 0.5) {
        scaledUV.x += 0.5;
    }

    // Calculate fractional part for mortar boundaries
    vec2 brickUV = fract(scaledUV);

    // Mortar thickness: 0.05 in X, 0.1 in Y
    // A pixel is mortar if it's within the bottom 0.05 in X OR within the bottom 0.1 in Y
    bool isMortar = brickUV.x < 0.05 || brickUV.y < 0.1;

    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor = vec3(0.7, 0.2, 0.1);

    vec3 baseColor = isMortar ? mortarColor : brickColor;

    outColor = vec4(baseColor * diffuseFactor, 1.0);
}
```
