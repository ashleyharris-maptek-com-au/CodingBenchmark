; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %UBO
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0
OpDecorate %UBO Block

OpMemberDecorate %UBO_type 0 Offset 0
OpMemberDecorate %UBO_type 1 Offset 64
OpMemberDecorate %UBO_type 2 Offset 128
OpMemberDecorate %UBO_type 3 Offset 192
OpMemberDecorate %UBO_type 4 Offset 208
OpMemberDecorate %UBO_type 5 Offset 224

OpDecorate %mat4v4 ColMajor
OpMemberDecorate %mat4v4 0 Offset 0
OpMemberDecorate %mat4v4 1 Offset 16
OpMemberDecorate %mat4v4 2 Offset 32
OpMemberDecorate %mat4v4 3 Offset 48

%void = OpTypeVoid
%fn_main_type = OpTypeFunction %void

%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2

%mat4v4 = OpTypeMatrix %v4float 4

%UBO_type = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float

%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO_type
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

%in_worldPos = OpVariable %ptr_input_v3float Input
%in_normal = OpVariable %ptr_input_v3float Input
%in_uv = OpVariable %ptr_input_v2float Input
%in_tangent = OpVariable %ptr_input_v3float Input
%in_color = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%UBO = OpVariable %ptr_uniform_UBO Uniform

%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_8 = OpConstant %float 8
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_0_2 = OpConstant %float 0.2
%float_1 = OpConstant %float 1

%main = OpFunction %void None %fn_main_type
%entry = OpLabel

; Load UV
%uv = OpLoad %v2float %in_uv

; Checkerboard
%scaled_uv = OpVectorTimesScalar %v2float %uv %float_8
%floor_uv = OpFloor %v2float %scaled_uv
%checker = OpDPdx %float %floor_uv ; trick: DPdx of integer-valued float is 0. Let's use proper bitwise logic instead

; Re-doing checker properly:
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

%sx = OpFMul %float %uv_x %float_8
%sy = OpFMul %float %uv_y %float_8

%fx = OpFloor %float %sx
%fy = OpFloor %float %sy

; Convert to int for XOR-like parity checking
%int_32 = OpTypeInt 32 0
%uint_fx = OpConvertFToU %int_32 %fx
%uint_fy = OpConvertFToU %int_32 %fy

; Parity of sum: (fx + fy) % 2
%sum = OpIAdd %int_32 %uint_fx %uint_fy
%one = OpConstant %int_32 1
%parity = OpBitwiseAnd %int_32 %sum %one

; 0 -> 0.9, 1 -> 0.2
%is_zero = OpIEqual %bool %parity %one
%bool = OpTypeBool

%color_val = OpSelect %float %is_zero %float_0_2 %float_0_9

; Load Normal and Light Position
%norm = OpLoad %v3float %in_normal
%n_norm = OpNormalize %v3float %norm

%light_ptr = OpAccessChain %ptr_uniform_v4float %UBO %int_3
%light_v4 = OpLoad %v4float %light_ptr
%light_pos = OpVectorShuffle %v3float %light_v4 %light_v4 0 1 2

; Calculate Light Direction
%pos = OpLoad %v3float %in_worldPos
%to_light = OpFSub %v3float %light_pos %pos
%L = OpNormalize %v3float %to_light

; Diffuse factor
%NdotL = OpDot %float %n_norm %L
%NdotL_clamped = OpFMax %float %NdotL %float_0

; Shading factor: 0.5 + 0.5 * diffuse
%shading = OpFMul %float %float_0_5 %NdotL_clamped
%shaded_val = OpFAdd %float %float_0_5 %shading

; Final color
%final_r = OpFMul %float %color_val %shaded_val
%final_g = OpFMul %float %color_val %shaded_val
%final_b = OpFMul %float %color_val %shaded_val

%out_vec = OpCompositeConstruct %v4float %final_r %final_g %final_b %float_1
OpStore %outColor %out_vec

OpReturn
OpFunctionEnd
