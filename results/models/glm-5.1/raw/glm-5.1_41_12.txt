; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%main = OpFunction %void None %func
%entry = OpLabel

%normal_load = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %normal_load

%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%worldPos_load = OpLoad %vec3 %in_worldPos
%L_unnorm = OpFSub %vec3 %lightPos_xyz %worldPos_load
%L = OpExtInst %vec3 %1 Normalize %L_unnorm

%NdotL = OpDot %float %N %L
%max_NdotL = OpExtInst %float %1 FMax %NdotL %float_0
%diffuse_factor = OpFMul %float %float_0_3 %max_NdotL
%diffuse_color = OpCompositeConstruct %vec3 %diffuse_factor %diffuse_factor %diffuse_factor

%Ny = OpCompositeExtract %float %N 1
%blend = OpFMul %float %Ny %float_0_5
%blend_biased = OpFAdd %float %blend %float_0_5

%skyColor = OpCompositeConstruct %vec3 %float_0_4 %float_0_6 %float_1_0
%groundColor = OpCompositeConstruct %vec3 %float_0_3 %float_0_15 %float_0_05

%ground_blend = OpFSub %float %float_1 %blend_biased
%ground_term = OpVectorTimesScalar %vec3 %groundColor %ground_blend
%sky_term = OpVectorTimesScalar %vec3 %skyColor %blend_biased
%ambient_color = OpFAdd %vec3 %ground_term %sky_term

%final_rgb = OpFAdd %vec3 %ambient_color %diffuse_color
%final_rgba = OpCompositeConstruct %vec4 %final_rgb %float_1

OpStore %outColor %final_rgba
OpReturn
OpFunctionEnd
