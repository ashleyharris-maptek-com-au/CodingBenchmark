```spirv
; SPIR-V
Version 1.0
Generator 0
Bound 200
Schema 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %inColor "inColor"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fun_type = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0 = OpConstant %float 0.0

%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %fun_type
%entry = OpLabel

; Load main vertex positions (clip space)
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p0 = OpLoad %v3float %p0_ptr
%p2 = OpLoad %v3float %p2_ptr
%p4 = OpLoad %v3float %p4_ptr

; Compute face normal
%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0

%e1_x = OpCompositeExtract %float %e1 0
%e1_y = OpCompositeExtract %float %e1 1
%e1_z = OpCompositeExtract %float %e1 2

%e2_x = OpCompositeExtract %float %e2 0
%e2_y = OpCompositeExtract %float %e2 1
%e2_z = OpCompositeExtract %float %e2 2

%cross_x = OpFMul %float %e1_y %e2_z
%cross_x2 = OpFMul %float %e1_z %e2_y
%nx = OpFSub %float %cross_x %cross_x2

%cross_y = OpFMul %float %e1_z %e2_x
%cross_y2 = OpFMul %float %e1_x %e2_z
%ny = OpFSub %float %cross_y %cross_y2

%cross_z = OpFMul %float %e1_x %e2_y
%cross_z2 = OpFMul %float %e1_y %e2_x
%nz = OpFSub %float %cross_z %cross_z2

%cross_vec = OpCompositeConstruct %v3float %nx %ny %nz
%N = OpExtInst %v3float %glsl450 Normalize %cross_vec

; Light direction
%float_1 = OpConstant %float 1.0
%L = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%L_norm = OpExtInst %v3float %glsl450 Normalize %L

; Diffuse = max(dot(N, L_norm), 0.0)
%dot_NL = OpDot %float %N %L_norm
%diffuse = OpExtInst %float %glsl450 FMax %dot_NL %float_0

; Base color = average of color[0], color[2], color[4]
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr

%c02 = OpFAdd %v3float %c0 %c2
%c024 = OpFAdd %v3float %c02 %c4
%float_3 = OpConstant %float 3.0
%base_color = OpCompositeConstruct %v3float %float_3 %float_3 %float_3
%base = OpFDiv %v3float %c024 %base_color

; Final color = base * (0.2 + 0.8 * diffuse)
%diff_coeff = OpFMul %float %float_0_8 %diffuse
%light = OpFAdd %float %float_0_2 %diff_coeff
%light_v3 = OpCompositeConstruct %v3float %light %light %light
%final_color = OpFMul %v3float %base %light_v3

; Emit vertex 0
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%out_pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0_ptr %pos0
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 2
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%out_pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2_ptr %pos2
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 4
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%out_pos4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos4_ptr %pos4
OpStore %outColor %final_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
