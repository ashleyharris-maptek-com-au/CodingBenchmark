; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_32 = OpConstant %float 32
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func_type
%label = OpLabel

%normal_load = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %normal_load

%light_pos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_var %int_3
%light_pos = OpLoad %vec4 %light_pos_ptr
%L_unnorm = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2

%world_pos_load = OpLoad %vec3 %in_worldPos

%L_vec = OpFSub %vec3 %L_unnorm %world_pos_load
%L = OpExtInst %vec3 %1 Normalize %L_vec

%NdotL = OpDot %float %N %L
%NdotL_clamped = OpExtInst %float %1 FMax %NdotL %float_0

%camera_pos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_var %int_4
%camera_pos = OpLoad %vec4 %camera_pos_ptr
%camera_pos_xyz = OpVectorShuffle %vec3 %camera_pos %camera_pos 0 1 2

%V_unnorm = OpFSub %vec3 %camera_pos_xyz %world_pos_load
%V = OpExtInst %vec3 %1 Normalize %V_unnorm

%neg_L = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %1 Reflect %neg_L %N

%RdotV = OpDot %float %R %V
%RdotV_clamped = OpExtInst %float %1 FMax %RdotV %float_0
%spec = OpExtInst %float %1 Pow %RdotV_clamped %float_32

%diffuse_term = OpFMul %float %float_0_5 %NdotL_clamped
%specular_term = OpFMul %float %float_0_8 %spec

%intensity = OpFAdd %float %float_0_2 %diffuse_term
%intensity = OpFAdd %float %intensity %specular_term

%out_vec4 = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %out_vec4

OpReturn
OpFunctionEnd
