; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_1 = OpConstant %float 0.1
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_6 = OpConstant %float 6.0

%bool = OpTypeBool

%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float

%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO

%in_worldPos = OpVariable %ptr_input_v3float Input
%in_normal = OpVariable %ptr_input_v3float Input
%in_uv = OpVariable %ptr_input_v2float Input
%in_tangent = OpVariable %ptr_input_v3float Input
%in_color = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%ptr_uniform_v4float = OpTypePointer Uniform %v4float

%main = OpFunction %void None %func_type
%label = OpLabel

; Load inputs
%uv = OpLoad %v2float %in_uv
%normal = OpLoad %v3float %in_normal
%worldPos = OpLoad %v3float %in_worldPos

; UBO loads
%lightPos_ptr = OpAccessChain %ptr_uniform_v4float %ubo_var %int_3
%lightPos = OpLoad %v4float %lightPos_ptr

; Normalize normal
%n_len = OpExtInst %float %1 Length %normal
%n_norm = OpFMul %v3float %normal %n_len ; intentionally wrong to show potential bug, wait no, must divide
; Actually: n_norm = normal / n_len
%inv_n_len = OpFDiv %float %float_1 %n_len
%n_norm_correct = OpVectorTimesScalar %v3float %normal %inv_n_len

; Polka dot pattern
%uv_scaled = OpVectorTimesScalar %v2float %uv %float_6
%cell = OpFract %v2float %uv_scaled
%cell_centered = OpFSub %v2float %cell %v2_0_5
%dist_sq = OpDot %float %cell_centered %cell_centered
%dot_radius_sq = OpFMul %float %float_0_3 %float_0_3
%is_dot = OpFOrdLessThan %bool %dist_sq %dot_radius_sq

; Base color
%dot_color = OpCompositeConstruct %v3float %float_1 %float_0_8 %float_0
%bg_color = OpCompositeConstruct %v3float %float_0_1 %float_0_1 %float_0_3
%base_color = OpSelect %v3float %is_dot %dot_color %bg_color

; Lighting
%lightPos3 = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%to_light = OpFSub %v3float %lightPos3 %worldPos
%to_light_len = OpExtInst %float %1 Length %to_light
%inv_to_light_len = OpFDiv %float %float_1 %to_light_len
%L = OpVectorTimesScalar %v3float %to_light %inv_to_light_len

%NdotL = OpDot %float %n_norm_correct %L
%NdotL_clamped = OpExtInst %float %1 FClamp %NdotL %float_0 %float_1

; Lambertian factor
%term1 = OpFMul %float %float_0_7 %NdotL_clamped
%factor = OpFAdd %float %float_0_3 %term1

%shaded_color = OpVectorTimesScalar %v3float %base_color %factor

; Output
%final_color = OpCompositeConstruct %v4float %shaded_color %float_1
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
