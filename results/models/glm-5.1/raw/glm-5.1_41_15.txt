; SPIR-V
; Version: 1.0
Generator: 0
Bound: 100
Schema: 0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float

%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%tangent = OpVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output

%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3

%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0

%cool = OpConstantComposite %v3float %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_4

%main = OpFunction %void None %3
%5 = OpLabel

%ld_normal = OpLoad %v3float %normal
%n_norm = OpExtInst %v3float %1 Normalize %ld_normal

%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %uboVar %int_3
%ld_lightPos = OpLoad %v4float %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %v3float %ld_lightPos %ld_lightPos 0 1 2

%ld_worldPos = OpLoad %v3float %worldPos
%L_dir = OpFSub %v3float %lightPos_xyz %ld_worldPos
%L_norm = OpExtInst %v3float %1 Normalize %L_dir

%NdotL = OpDot %float %n_norm %L_norm
%NdotL_scaled = OpFMul %float %NdotL %float_0_5
%NdotL_biased = OpFAdd %float %NdotL_scaled %float_0_5

%one_minus_t = OpFSub %float %float_1 %NdotL_biased
%cool_contrib = OpVectorTimesScalar %v3float %cool %one_minus_t
%warm_contrib = OpVectorTimesScalar %v3float %warm %NdotL_biased
%gooch_color = OpFAdd %v3float %cool_contrib %warm_contrib

%out_rgba = OpCompositeExtract %float %gooch_color 0
%out_rgbg = OpCompositeExtract %float %gooch_color 1
%out_rgbb = OpCompositeExtract %float %gooch_color 2
%final_color = OpCompositeConstruct %v4float %out_rgba %out_rgbg %out_rgbb %float_1

OpStore %outColor %final_color

OpReturn
OpFunctionEnd
