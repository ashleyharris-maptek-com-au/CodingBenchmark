#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstdint>
#include <limits>

using namespace std;

const int64_t INF = 1e18;

struct Edge {
    int to;
    int64_t weight;
    int id;
};

struct State {
    int v;
    int64_t d;
    bool operator>(const State& other) const {
        return d > other.d;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> adj(n);
    vector<pair<int, int>> edge_info(m);

    for (int i = 0; i < m; ++i) {
        int u, v;
        int64_t w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w, i});
        adj[v].push_back({u, w, i});
        edge_info[i] = {u, v};
    }

    vector<int> term(k);
    for (int i = 0; i < k; ++i) {
        cin >> term[i];
    }

    // Dreyfus-Wagner algorithm for Steiner Tree
    vector<vector<int64_t>> dp(n, vector<int64_t>(1 << k, INF));

    for (int i = 0; i < k; ++i) {
        dp[term[i]][1 << i] = 0;
    }

    for (int mask = 1; mask < (1 << k); ++mask) {
        for (int v = 0; v < n; ++v) {
            for (int sub = mask; sub > 0; sub = (sub - 1) & mask) {
                int64_t val = dp[v][sub] + dp[v][mask ^ sub];
                if (val < dp[v][mask]) {
                    dp[v][mask] = val;
                }
            }
        }

        priority_queue<State, vector<State>, greater<State>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[v][mask] < INF) {
                pq.push({v, dp[v][mask]});
            }
        }

        while (!pq.empty()) {
            State top = pq.top();
            pq.pop();
            int u = top.v;
            int64_t d = top.d;

            if (d > dp[u][mask]) continue;

            for (const Edge& e : adj[u]) {
                if (dp[u][mask] + e.weight < dp[e.to][mask]) {
                    dp[e.to][mask] = dp[u][mask] + e.weight;
                    pq.push({e.to, dp[e.to][mask]});
                }
            }
        }
    }

    int full_mask = (1 << k) - 1;
    int best_root = -1;
    int64_t best_cost = INF;
    for (int v = 0; v < n; ++v) {
        if (dp[v][full_mask] < best_cost) {
            best_cost = dp[v][full_mask];
            best_root = v;
        }
    }

    if (best_root == -1) {
        cout << "0 0\n";
        return 0;
    }

    // Reconstruct the Steiner Tree using backtracking
    vector<bool> used(m, false);
    
    auto backtrack = [&](auto& self, int v, int mask) -> void {
        if (mask == 0) return;
        if (__popcount(mask) == 1) return; // Base case: single terminal

        // Try splitting the mask
        bool split = false;
        int sub = (mask - 1) & mask;
        while (sub > 0) {
            int64_t val = dp[v][sub] + dp[v][mask ^ sub];
            if (val == dp[v][mask]) {
                self(self, v, sub);
                self(self, v, mask ^ sub);
                split = true;
                break;
            }
            sub = (sub - 1) & mask;
        }

        if (!split) {
            // Shortest path part: relax along incoming edges
            for (const Edge& e : adj[v]) {
                if (dp[e.to][mask] + e.weight == dp[v][mask]) {
                    used[e.id] = true;
                    self(self, e.to, mask);
                    return;
                }
            }
        }
    };

    backtrack(backtrack, best_root, full_mask);

    // Collect edges, verify it is a tree (connected, acyclic, K-1 edges)
    vector<int> result_edges;
    for (int i = 0; i < m; ++i) {
        if (used[i]) {
            result_edges.push_back(i);
        }
    }

    cout << best_cost << " " << result_edges.size() << "\n";
    for (int idx : result_edges) {
        cout << edge_info[idx].first << " " << edge_info[idx].second << "\n";
    }

    return 0;
}
