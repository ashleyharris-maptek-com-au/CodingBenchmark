Looking at this problem, I need to build a real-time Lunar Lander controller that:
1. Continuously reads state updates without blocking or stalling
2. Navigates around obstacles using a pre-computed potential field
3. Controls descent using a PD controller for smooth landing

Here's my approach:
- **Async I/O with two threads**: Reader thread fills a shared state, control thread runs the PD loop at 20+ Hz
- **Obstacle navigation**: Pre-compute a distance field (BFS from target) to know if direct paths are blocked, plus a simple vector field pushing away from nearby obstacles
- **PD control**: Target angle and thrust computed from desired horizontal/vertical accelerations, with phase-dependent fuel budgets

        prev_x = x;
        prev_y = y;
    }

fn normalize_angle(a: f64) -> f64 {
    let mut a = a;
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }
    a
}

fn compute_obstacle_avoidance(state: &SharedState, _field: &Vec<(f64, f64)>) -> (f64, f64) {
    if !state.has_obstacles || state.width == 0 || state.height == 0 {
        return (0.0, 0.0);
    }

    let mut ax = 0.0;
    let mut ay = 0.0;
    let radius = 15.0;
    let ix = state.x as i32;
    let iy = state.y as i32;
    let r = radius as i32;

    // Check a grid of points around the lander for obstacles
    // Use larger step for efficiency
    let step = 2;
    for dy in -r..=r step step {
        for dx in -r..=r step step {
            let cx = ix + dx;
            let cy = iy + dy;
            if cx < 0 || cy < 0 || cx >= state.width as i32 || cy >= state.height as i32 {
                // Treat out of bounds as wall
                let dist = ((dx * dx + dy * dy) as f64).sqrt().max(1.0);
                if dist < radius {
                    let strength = (1.0 - dist / radius).powi(2);
                    ax -= (dx as f64 / dist) * strength * 8.0;
                    ay -= (dy as f64 / dist) * strength * 8.0;
                }
                continue;
            }
            // We can't check the actual map here since we didn't store it
            // Instead, use the distance fields if available
            let idx = (cy as usize) * state.width + (cx as usize);
            if idx < state.obstacle_clear.len() {
                let clearance = state.obstacle_clear[idx];
                if clearance < 8 {
                    let dist = ((dx * dx + dy * dy) as f64).sqrt().max(1.0);
                    let strength = (1.0 - clearance as f64 / 8.0).powi(2);
                    ax -= (dx as f64 / dist) * strength * 6.0;
                    ay -= (dy as f64 / dist) * strength * 6.0;
                }
            }
        }
    }

    (ax, ay)
}

fn check_ground_proximity(state: &SharedState) -> f64 {
    // Check how far below the lander is from the nearest surface
    if state.width == 0 || state.height == 0 {
        // No map data, estimate from target position
        return (state.target_y - state.y).max(0.0);
    }

    let ix = state.x as usize;
    if ix >= state.width {
        return 10.0;
    }

    // Check obstacle_clear field below lander
    if !state.obstacle_clear.is_empty() {
        let mut min_dist = f64::MAX;
        for dy in 0..30.min(state.height - state.y as usize) {
            let cy = state.y as usize + dy;
            let idx = cy * state.width + ix;
            if idx < state.obstacle_clear.len() && state.obstacle_clear[idx] == 0 {
                min_dist = dy as f64;
                break;
            }
            // Also check adjacent columns
            for &dx in &[0usize, 1] {
                let cx = (ix + dx).min(state.width - 1);
                let idx2 = cy * state.width + cx;
                if idx2 < state.obstacle_clear.len() && state.obstacle_clear[idx2] == 0 {
                    min_dist = min_dist.min(dy as f64);
                    break;
                }
            }
            if min_dist < f64::MAX { break; }
        }
        if min_dist < f64::MAX { return min_dist; }
    }

    // Fallback
    (state.target_y - state.y).max(1.0)
}

fn compute_control(s: &Snapshot) -> (f64, f64) {
    let g = s.gravity;
    let max_t = s.max_thrust;
    let fuel = s.fuel;
    let (x, y, vx, vy) = (s.x, s.y, s.vx, s.vy);
    let angle = s.angle;
    let angular_vel = s.angular_vel;
    let (tx, ty) = (s.target_x, s.target_y);

    let dx = tx - x;
    let dy = ty - y;
    let dist = (dx * dx + dy * dy).sqrt();
    let speed = (vx * vx + vy * vy).sqrt();

    // --- Obstacle avoidance and navigation ---
    let (mut nav_ax, mut nav_ay, ground_clear) = compute_navigation(s);

    // --- Determine desired velocities ---
    let desired_vx;
    let desired_vy;

    if dist < 5.0 && speed < 3.0 {
        // Final landing
        desired_vx = dx * 0.3;
        desired_vy = dy * 0.3;
    } else {
        // Navigate toward target
        let max_h_speed = 8.0;
        let max_v_speed = 10.0;

        desired_vx = (dx * 0.5).clamp(-max_h_speed, max_h_speed);

        // Vertical: descend toward target, but control rate
        if dy > 20.0 {
            desired_vy = (dy * 0.2).clamp(-max_v_speed, -2.0);
        } else if dy > 5.0 {
            desired_vy = (dy * 0.3).clamp(-8.0, -1.0);
        } else if dy > 0.0 {
            desired_vy = dy * 0.4;
        } else {
            // Above target (dy < 0)
            desired_vy = (dy * 0.3).clamp(-3.0, 3.0);
        }

        // Slow down as we approach horizontally
        let h_dist = dx.abs();
        if h_dist < 20.0 {
            desired_vx *= h_dist / 20.0;
        }
    }

    // --- Compute desired accelerations ---
    let kp = 0.8;
    let kd = 2.0;
    let mut desired_ax = kp * (desired_vx - vx) + nav_ax;
    let mut desired_ay = g + kp * (desired_vy - vy) + nav_ay;

    // Ground proximity braking
    if ground_clear < 15.0 && vy > 2.0 {
        let brake_ay = g + (vy * vy) / (2.0 * ground_clear.max(0.5)) * 1.5;
        desired_ay = desired_ay.max(brake_ay);
    }
    if ground_clear < 5.0 && vy > 1.5 {
        desired_ay = desired_ay.max(g + max_t * 0.9);
        desired_ax *= 0.2; // Focus on braking
    }

    // Fuel budgeting
    let h_to_land = dy.max(5.0);
    let fuel_needed = h_to_land / 3.0 + 3.0;
    let fuel_margin = fuel - fuel_needed;
    if fuel_margin < 3.0 {
        let scale = if fuel_margin < 0.0 { 0.05 } else { 0.15 + 0.85 * (fuel_margin / 3.0) };
        desired_ax *= scale;
        desired_ay = g + (desired_ay - g) * scale;
    }

    // --- Convert to angle + thrust ---
    let rel_ax = desired_ax;
    let rel_ay = desired_ay - g; // Thrust vertical component
    let target_angle = rel_ax.atan2(-rel_ay);
    let thrust_mag = (rel_ax * rel_ax + rel_ay * rel_ay).sqrt();
    let thrust_frac = (thrust_mag / max_t).clamp(0.0, 1.0);

    // --- Angle PD control ---
    let angle_err = normalize_angle(target_angle - angle);
    let desired_av = 6.0 * angle_err - 4.5 * angular_vel;
    let turn = desired_av.clamp(-1.0, 1.0);

    (thrust_frac, turn)
}

fn compute_navigation(s: &Snapshot) -> (f64, f64, f64) {
    let mut nav_ax = 0.0;
    let mut nav_ay = 0.0;
    let mut ground_clear = 100.0;

    let map = match &s.map {
        Some(m) => m,
        None => return (0.0, 0.0, 100.0),
    };

    let sw = map.width;
    let sh = map.height;
    let scale = map.scale;

    let ci = (s.y as usize / scale).min(sh.saturating_sub(1));
    let cj = (s.x as usize / scale).min(sw.saturating_sub(1));

    // Obstacle repulsion
    let radius = 15i32;
    for dr in -radius..=radius {
        for dc in -radius..=radius {
            let nr = ci as i32 + dr;
            let nc = cj as i32 + dc;
            if nr < 0 || nr >= sh as i32 || nc < 0 || nc >= sw as i32 {
                let d = ((dr * dr + dc * dc) as f64).sqrt().max(1.0);
                if d < radius as f64 {
                    let str = (1.0 - d / radius as f64).powi(2) * 4.0;
                    nav_ax -= (dc as f64 / d) * str;
                    nav_ay -= (dr as f64 / d) * str;
                }
                continue;
            }
            let nr = nr as usize;
            let nc = nc as usize;

                    nav_ax -= (dc as f64 / d) * str;
                    nav_ay -= (dr as f64 / d) * str;
                }
            }
        }
    }

    let my_idx = ci * sw + cj;
    if my_idx < map.nav_dist.len() && map.nav_dist[my_idx] > 3 {
        let mut gx = 0.0f64;
        let mut gy = 0.0f64;
        for &dc in &[-2i32, -1, 1, 2] {
            let nc = cj as i32 + dc;

                let ni = ci * sw + nc as usize;
                let dd = map.nav_dist[my_idx] as i32 - map.nav_dist[ni] as i32;
                gx += dd as f64 * dc.signum() as f64;
            }
        }
        for &dr in &[-2i32, -1, 1, 2] {
            let nr = ci as i32 + dr;
            if nr >= 0 && nr < sh as i32 {
                let ni = nr as usize * sw + cj;
                let dd = map.nav_dist[my_idx] as i32 - map.nav_dist[ni] as i32;
                gy += dd as f64 * dr.signum() as f64;
            }
        }
        let gm = (gx * gx + gy * gy).sqrt().max(0.001);
        let nav_str = (map.nav_dist[my_idx] as f64 * 0.2).min(5.0);
        nav_ax += gx / gm * nav_str;
        nav_ay += gy / gm * nav_str;
    }

    // Ground clearance
    for dy in 0..80.min(sh - ci) {
        let idx = (ci + dy) * sw + cj;
        if idx < map.obstacles.len() && map.obstacles[idx] {
            ground_clear = dy as f64 * scale as f64;
            break;
        }
    }

    (nav_ax, nav_ay, ground_clear)
}
