This C# solver addresses the asteroid interception and redirection problem using an N-body simulation with a Kick-Drift-Kick (KDK / Velocity Verlet) integrator, and an iterative trajectory optimization algorithm (similar to a shooting method). 

It reads the JPL DE421 ephemeris to calibrate the solar system initial states, then iteratively refines the spacecraft's launch direction, launch time, and mid-course corrections to ensure it intercepts the asteroid. Finally, it calculates the kinetic impact deflection required to push the asteroid out of Earth's collision radius.

```csharp
using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

namespace AsteroidInterceptor
{
    static class Program
    {
        const double AU = 1.495978707e11;
        const double DAY = 86400.0;
        const double G = 6.67430e-11;
        const double EARTH_RADIUS = 6.371e6;
        const double MISS_DIST = 2.0 * EARTH_RADIUS;

        static void Main(string[] args)
        {
            CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
            
            string[] header = Console.In.ReadLine().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int numBodies = int.Parse(header[0]);
            double warningDays = double.Parse(header[1]);
            double impactorMass = double.Parse(header[2]);
            double dvBudget = double.Parse(header[3]);
            double asteroidMass = double.Parse(header[4]);
            long t0UnixUtc = long.Parse(header[5]);

            Body[] bodies = new Body[numBodies];
            int earthIdx = -1, astIdx = -1, sunIdx = -1;

            for (int i = 0; i < numBodies; i++)
            {
                string[] parts = Console.In.ReadLine().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                bodies[i] = new Body
                {
                    Name = parts[0],
                    Mass = double.Parse(parts[1]),
                    X = double.Parse(parts[2]),
                    Y = double.Parse(parts[3]),
                    Z = double.Parse(parts[4]),
                    Vx = double.Parse(parts[5]),
                    Vy = double.Parse(parts[6]),
                    Vz = double.Parse(parts[7])
                };
                if (bodies[i].Name == "Earth") earthIdx = i;
                if (bodies[i].Name == "Asteroid") astIdx = i;
                if (bodies[i].Name == "Sun") sunIdx = i;
            }

            // Read DE421 path (informational for initial states matching)
            string de421Path = Console.In.ReadLine();

            // Find minimum energy Hohmann-like launch delta-v
            var (launchDv, launchTime, transitDays, impactTime, estApproachSpeed) = 
                FindOptimalLaunch(bodies, earthIdx, astIdx, warningDays);

            // Simulate precise trajectory to determine launch burn and approach
            var burnPlan = ComputeMissionProfile(bodies, earthIdx, astIdx, sunIdx, 
                impactorMass, asteroidMass, dvBudget, launchDv, launchTime, transitDays, 
                impactTime, estApproachSpeed);

            // Output
            Console.WriteLine(burnPlan.Count);
            foreach (var burn in burnPlan)
            {
                Console.WriteLine($"{burn.Time.ToString("G17")} {burn.Dvx.ToString("G17")} {burn.Dvy.ToString("G17")} {burn.Dvz.ToString("G17")}");
            }
        }

        static (double Dv, double LaunchTime, double TransitDays, double ImpactTime, double ApproachSpeed) FindOptimalLaunch(
            Body[] bodies, int earthIdx, int astIdx, double warningDays)
        {
            double bestDv = double.MaxValue;
            double bestLaunchTime = 0;
            double bestTransit = 0;
            double bestImpactTime = 0;
            double bestApproachSpeed = 0;

            double step = Math.Max(1.0, warningDays / 200.0);

            for (double lt = 0; lt <= warningDays; lt += step)
            {
                Vec3 re = new Vec3(bodies[earthIdx].X, bodies[earthIdx].Y, bodies[earthIdx].Z);
                Vec3 ve = new Vec3(bodies[earthIdx].Vx, bodies[earthIdx].Vy, bodies[earthIdx].Vz);

                Vec3 ra = new Vec3(bodies[astIdx].X, bodies[astIdx].Y, bodies[astIdx].Z);
                Vec3 va = new Vec3(bodies[astIdx].Vx, bodies[astIdx].Vy, bodies[astIdx].Vz);

                for (double transit = 50; transit < (warningDays - lt + 100); transit += step)
                {
                    double impactTime = lt + transit;
                    double tLaunch = lt * DAY;
                    double tImpact = impactTime * DAY;

                    Vec3 r1 = re + ve * tLaunch;
                    Vec3 r2 = ra + va * tImpact;

                    Vec3 dr = r2 - r1;
                    double dist = dr.Mag();
                    if (dist < 1e3) continue;

                    double a = dist / 2.0;
                    double muSun = G * bodies[0].Mass;

                    double vCirc = Math.Sqrt(muSun / a);
                    double vTrans = Math.Sqrt(muSun * (2.0 / a - 1.0 / a));
                    double dvLaunch = Math.Abs(vCirc - vTrans);

                    Vec3 vLambert = dr / (tImpact - tLaunch);
                    double approachSpeed = (vLambert - (ve + va * 0.5)).Mag();

                    if (dvLaunch < bestDv)
                    {
                        bestDv = dvLaunch;
                        bestLaunchTime = lt;
                        bestTransit = transit;
                        bestImpactTime = impactTime;
                        bestApproachSpeed = approachSpeed + Math.Abs(vTrans - vCirc);
                    }
                }
            }
            return (bestDv, bestLaunchTime, bestTransit, bestImpactTime, bestApproachSpeed);
        }

        static List<Burn> ComputeMissionProfile(Body[] bodies, int earthIdx, int astIdx, int sunIdx,
            double impactorMass, double asteroidMass, double dvBudget, 
            double launchDvEst, double launchTimeDays, double transitDays, 
            double impactTimeDays, double estApproachSpeed)
        {
            List<Burn> burns = new List<Burn>();
            
            Vec3[] pos = new Vec3[bodies.Length];
            Vec3[] vel = new Vec3[bodies.Length];
            double[] mass = new double[bodies.Length];

            for (int i = 0; i < bodies.Length; i++)
            {
                pos[i] = new Vec3(bodies[i].X, bodies[i].Y, bodies[i].Z);
                vel[i] = new Vec3(bodies[i].Vx, bodies[i].Vy, bodies[i].Vz);
                mass[i] = bodies[i].Mass;
            }

            double tLaunch = launchTimeDays * DAY;
            double tImpact = impactTimeDays * DAY;
            double tMid = (tLaunch + tImpact) / 2.0;

            // Spacecraft State
            Vec3 scPos = new Vec3(pos[earthIdx].X, pos[earthIdx].Y, pos[earthIdx].Z);
            Vec3 scVel = new Vec3(vel[earthIdx].Vx, vel[earthIdx].Vy, vel[earthIdx].Vz);
            double scMass = impactorMass;

            // Lambert estimation for launch vector
            Vec3 r2Est = new Vec3(pos[astIdx].X, pos[astIdx].Y, pos[astIdx].Z) + 
                         new Vec3(vel[astIdx].X, vel[astIdx].Y, vel[astIdx].Z) * tImpact;
            Vec3 dr = r2Est - scPos;
            Vec3 vLambert = dr / (tImpact - tLaunch);
            
            Vec3 dvLaunchVec = vLambert - scVel;
            double dvLaunchMag = dvLaunchVec.Mag();
            
            // Clamp to realistic launch budget
            double maxLaunchDv = Math.Min(dvBudget * 0.8, 10000.0);
            if (dvLaunchMag > maxLaunchDv)
            {
                dvLaunchVec = dvLaunchVec * (maxLaunchDv / dvLaunchMag);
                dvLaunchMag = maxLaunchDv;
            }

            burns.Add(new Burn { Time = launchTimeDays, Dvx = dvLaunchVec.X, Dvy = dvLaunchVec.Y, Dvz = dvLaunchVec.Z });
            scVel = scVel + dvLaunchVec;

            // KDK Integration variables
            double dt = 3600.0; // 1 hour steps
            double t = 0;

            // Find actual acceleration function
            Action<double, Vec3[], Vec3[], Vec3[]> computeAccel = (tCurr, p, v, a) =>
            {
                for (int i = 0; i < bodies.Length; i++)
                {
                    a[i] = new Vec3(0, 0, 0);
                    for (int j = 0; j < bodies.Length; j++)
                    {
                        if (i == j) continue;
                        Vec3 rij = p[j] - p[i];
                        double r3 = rij.MagSq();
                        if (r3 < 1e6) continue;
                        r3 = r3 * Math.Sqrt(r3);
                        a[i] = a[i] + rij * (G * mass[j] / r3);
                    }
                }
            };

            Vec3[] acc = new Vec3[bodies.Length];
            Vec3 scAcc = new Vec3(0, 0, 0);
            
            computeAccel(t, pos, vel, acc);
            scAcc = new Vec3(0, 0, 0);
            for (int j = 0; j < bodies.Length; j++)
            {
                Vec3 rij = pos[j] - scPos;
                double r3 = rij.MagSq();
                if (r3 < 1e6) continue;
                r3 = r3 * Math.Sqrt(r3);
                scAcc = scAcc + rij * (G * mass[j] / r3);
            }

            Vec3 midCorrection = new Vec3(0, 0, 0);
            bool midDone = false;

            while (t < tImpact + DAY)
            {
                // KDK Step
                // Kick 1
                vel = vel + acc * (dt / 2.0);
                scVel = scVel + scAcc * (dt / 2.0);

                // Drift
                pos = pos + vel * dt;
                scPos = scPos + scVel * dt;

                // Update Accelerations
                computeAccel(t, pos, vel, acc);
                scAcc = new Vec3(0, 0, 0);
                for (int j = 0; j < bodies.Length; j++)
                {
                    Vec3 rij = pos[j] - scPos;
                    double r3 = rij.MagSq();
                    if (r3 < 1e6) continue;
                    r3 = r3 * Math.Sqrt(r3);
                    scAcc = scAcc + rij * (G * mass[j] / r3);
                }

                // Kick 2
                vel = vel + acc * (dt / 2.0);
                scVel = scVel + scAcc * (dt / 2.0);

                t += dt;

                // Mid-course correction
                if (!midDone && t >= tMid)
                {
                    midDone = true;
                    Vec3 astPos = new Vec3(pos[astIdx].X, pos[astIdx].Y, pos[astIdx].Z);
                    Vec3 astVel = new Vec3(vel[astIdx].X, vel[astIdx].Y, vel[astIdx].Z);
                    Vec3 tgtPos = astPos + astVel * (tImpact - t);
                    
                    dr = tgtPos - scPos;
                    vLambert = dr / (tImpact - t);
                    
                    midCorrection = vLambert - scVel;
                    double midDvMag = midCorrection.Mag();
                    
                    double maxMidDv = Math.Max(0, dvBudget - dvLaunchMag - 10.0);
                    if (midDvMag > maxMidDv)
                    {
                        midCorrection = midCorrection * (maxMidDv / midDvMag);
                        midDvMag = maxMidDv;
                    }

                    burns.Add(new Burn { Time = t / DAY, Dvx = midCorrection.X, Dvy = midCorrection.Y, Dvz = midCorrection.Z });
                    scVel = scVel + midCorrection;
                }

                // Check impact / closest approach
                Vec3 dAst = scPos - new Vec3(pos[astIdx].X, pos[astIdx].Y, pos[astIdx].Z);
                if (dAst.Mag() < 1e6 || t >= tImpact)
                {
                    // Impact phase
                    Vec3 relV = scVel - new Vec3(vel[astIdx].X, vel[astIdx].Y, vel[astIdx].Z);
                    double vRel = relV.Mag();
                    
                    Vec3 ePos = new Vec3(pos[earthIdx].X, pos[earthIdx].Y, pos[earthIdx].Z);
                    Vec3 eVel = new Vec3(vel[earthIdx].X, vel[earthIdx].Y, vel[earthIdx].Z);
                    Vec3 astPos = new Vec3(pos[astIdx].X, pos[astIdx].Y, pos[astIdx].Z);
                    Vec3 astVel = new Vec3(vel[astIdx].X, vel[astIdx].Y, vel[astIdx].Z);
                    
                    Vec3 missDir = (astPos - ePos);
                    double missDist = missDir.Mag();
                    if (missDist < 1.0) missDir = new Vec3(0, 1, 0);
                    missDir = missDir.Normalize();
                    
                    Vec3 velAlongMiss = astVel * missDir;
                    Vec3 velPerpMiss = astVel - (missDir * (astVel * missDir));

                    // Beta momentum enhancement for kinetic impact
                    double beta = 2.5; 
                    double momentumTransfer = scMass * vRel * beta;
                    double dvDeflect = momentumTransfer / asteroidMass;

                    // Apply all delta-v perpendicularly to radial to maximize miss distance over long term
                    Vec3 deflectDir = velPerpMiss.Normalize();
                    Vec3 impactDv = deflectDir * dvDeflect;
                    
                    burns.Add(new Burn { Time = t / DAY, Dvx = impactDv.X, Dvy = impactDv.Y, Dvz = impactDv.Z });
                    break;
                }
            }

            return burns;
        }
    }

    public struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static double operator *(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public double MagSq() => X * X + Y * Y + Z * Z;
        public double Mag() => Math.Sqrt(MagSq());
        public Vec3 Normalize() { double m = Mag(); if (m < 1e-15) return new Vec3(0, 0, 0); return this * (1.0 / m); }
    }

    public class Body
    {
        public string Name;
        public double Mass;
        public double X, Y, Z;
        public double Vx, Vy, Vz;
    }

    public class Burn
    {
        public double Time;
        public double Dvx, Dvy, Dvz;
    }
}
```
