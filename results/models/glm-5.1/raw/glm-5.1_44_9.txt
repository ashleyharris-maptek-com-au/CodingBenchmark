```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Lambertian diffuse factor
    float diffuse = max(dot(N, L), 0.0);
    
    // UV checkerboard at 8x8 frequency
    vec2 checkerUV = floor(uv * 8.0);
    float checker = mod(checkerUV.x + checkerUV.y, 2.0);
    
    // Alternate colors 0.9 and 0.2 by parity
    float shade = mix(0.2, 0.9, checker);
    
    // Apply Lambertian shading with factor (0.5 + 0.5*diffuse)
    float lighting = 0.5 + 0.5 * diffuse;
    
    outColor = vec4(vec3(shade * lighting), 1.0);
}
```
