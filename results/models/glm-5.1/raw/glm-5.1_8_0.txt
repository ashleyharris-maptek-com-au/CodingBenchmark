```rust
use std::io::{self, Read, Write, BufWriter};

const NULL_POS: u32 = u32::MAX;
const UNVISITED: u32 = u32::MAX;

fn main() {
    let mut input = Vec::with_capacity(1 << 30);
    io::stdin().lock().read_to_end(&mut input).unwrap();
    let mut it = input.split(|&c| c == b' ' || c == b'\n');

    let mut next_usize = || {
        let mut num = 0usize;
        while let Some(slice) = it.next() {
            if slice.is_empty() { continue; }
            for &b in slice {
                num = num * 10 + (b - b'0') as usize;
            }
            return num;
        }
        unreachable!()
    };

    let h = next_usize();
    let w = next_usize();

    // Read all raw rows into a contiguous vector (W * H)
    let mut maze = vec![0u8; w * h];
    let mut idx = 0;
    let mut remaining = h;
    for &b in &input {
        if b == b'\n' || b == b'\r' {
            if idx % w == 0 && idx > 0 {
                remaining -= 1;
                if remaining == 0 { break; }
            }
            continue;
        }
        if idx < maze.len() {
            maze[idx] = b;
            idx += 1;
        }
    }

    // Locate start and end
    let mut start: u32 = NULL_POS;
    let mut end: u32 = NULL_POS;
    for i in 0..(w * h) {
        match maze[i] {
            b'A' => start = i as u32,
            b'B' => end = i as u32,
            _ => {}
        }
    }

    if start == NULL_POS || end == NULL_POS {
        let stdout = io::stdout();
        let mut out = BufWriter::new(stdout.lock());
        writeln!(out, "0").unwrap();
        return;
    }

    let total_cells = w * h;
    let mut visited = vec![UNVISITED; total_cells];

    // Allocate frontier buffers (used alternately as current/next frontier)
    let mut frontier_a: Vec<u32> = Vec::with_capacity(1024);
    let mut frontier_b: Vec<u32> = Vec::with_capacity(1024);

    visited[start as usize] = start;
    visited[end as usize] = end;
    frontier_a.push(start);
    frontier_b.push(end);

    // Bidirectional BFS offsets
    let offsets: [i32; 4] = [-(w as i32), -1, w as i32, 1];
    let col_offsets: [usize; 4] = [0, usize::MAX, 0, 0]; // MAX disables boundary check for L/R
    let col_checks: [i32; 4] = [0, -1, 0, 1]; // -1 = enforce left boundary, +1 = enforce right

    let mut meet_point: u32 = NULL_POS;
    let mut meet_dir: i32 = -1; // +1 = start-side meets end-side, -1 = end-side meets start-side

    // Main BFS loop
    // Always expand the smaller frontier to guarantee optimality
    while !frontier_a.is_empty() && !frontier_b.is_empty() {
        let (cur_frontier, oth_frontier, dir) = if frontier_a.len() <= frontier_b.len() {
            (&mut frontier_a, &mut frontier_b, 1i32)
        } else {
            (&mut frontier_b, &mut frontier_a, -1i32)
        };

        let mut next_frontier = Vec::with_capacity(cur_frontier.len() * 2);
        for &pos in cur_frontier.iter() {
            let pos_i = pos as usize;
            let pos_row = pos_i / w;
            let pos_col = pos_i % w;
            let parent = visited[pos_i]; // ancestor in the same BFS tree

            for d in 0..4 {
                // Fast boundary check
                if pos_col == 0 && col_checks[d] < 0 { continue; }
                if pos_col == w - 1 && col_checks[d] > 0 { continue; }

                let n = (pos as i32 + offsets[d]) as u32;
                let n_i = n as usize;
                let n_parent = visited[n_i];

                if n_parent != UNVISITED {
                    // If neighbor visited from the OTHER BFS tree, we found a meeting point
                    if dir == 1 {
                        if n_parent >= end { meet_point = pos; meet_dir = 1; break; }
                    } else {
                        if n_parent < end { meet_point = pos; meet_dir = -1; break; }
                    }
                    continue;
                }

                // Wall / valid cell check
                let b = maze[n_i];
                if b == b'#' { continue; }

                visited[n_i] = parent; // mark with ancestor
                next_frontier.push(n);
            }
            if meet_point != NULL_POS { break; }
        }

        if meet_point != NULL_POS { break; }

        // Swap buffers (next_frontier becomes cur_frontier for the next cycle)
        *cur_frontier = next_frontier;

        // Recalculate references after swapping
        let (cur_frontier, oth_frontier, dir) = if frontier_a.len() <= frontier_b.len() {
            (&mut frontier_a, &mut frontier_b, 1i32)
        } else {
            (&mut frontier_b, &mut frontier_a, -1i32)
        };

        let mut next_frontier = Vec::with_capacity(cur_frontier.len() * 2);
        for &pos in cur_frontier.iter() {
            let pos_i = pos as usize;
            let pos_col = pos_i % w;
            let parent = visited[pos_i];

            for d in 0..4 {
                if pos_col == 0 && col_checks[d] < 0 { continue; }
                if pos_col == w - 1 && col_checks[d] > 0 { continue; }

                let n = (pos as i32 + offsets[d]) as u32;
                let n_i = n as usize;
                let n_parent = visited[n_i];

                if n_parent != UNVISITED {
                    if dir == 1 {
                        if n_parent >= end { meet_point = pos; meet_dir = 1; break; }
                    } else {
                        if n_parent < end { meet_point = pos; meet_dir = -1; break; }
                    }
                    continue;
                }

                let b = maze[n_i];
                if b == b'#' { continue; }

                visited[n_i] = parent;
                next_frontier.push(n);
            }
            if meet_point != NULL_POS { break; }
        }

        if meet_point != NULL_POS { break; }
        *cur_frontier = next_frontier;
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if meet_point == NULL_POS {
        writeln!(out, "0").unwrap();
        return;
    }

    // Reconstruct path back to start
    let mut path_a = Vec::new();
    let mut cur = meet_point;
    loop {
        path_a.push(cur);
        let p = visited[cur as usize];
        if p == cur { break; }
        cur = p;
    }
    path_a.reverse(); // now from start -> meet_point

    // Find exact bridge neighbor to properly reconstruct end-side path
    let mp_i = meet_point as usize;
    let mp_row = mp_i / w;
    let mp_col = mp_i % w;
    let mut bridge: u32 = NULL_POS;

    for d in 0..4 {
        if mp_col == 0 && col_checks[d] < 0 { continue; }
        if mp_col == w - 1 && col_checks[d] > 0 { continue; }
        let n = (meet_point as i32 + offsets[d]) as u32;
        let n_parent = visited[n as usize];
        if n_parent != UNVISITED {
            if meet_dir == 1 && n_parent >= end { bridge = n; break; }
            if meet_dir == -1 && n_parent < end { bridge = n; break; }
        }
    }

    // Reconstruct path back to end
    let mut path_b = Vec::new();
    if bridge != NULL_POS {
        cur = bridge;
        loop {
            path_b.push(cur);
            let p = visited[cur as usize];
            if p == cur { break; }
            cur = p;
        }
    }

    // If meet_dir == 1: path_a is start->meet, path_b is bridge->end (already iterated reverse)
    // If meet_dir == -1: path_b is start->bridge, path_a is meet->end (reversed)
    // We always want to output start -> end.
    
    let (path_start, path_end) = if meet_dir == 1 {
        (&path_a, &path_b)
    } else {
        (&path_b, &path_a)
    };

    // Output
    let total_steps = path_start.len() + path_end.len();
    writeln!(out, "{}", total_steps).unwrap();

    for &pos in path_start.iter() {
        let r = pos as usize / w;
        let c = pos as usize % w;
        writeln!(out, "{} {}", c, r).unwrap();
    }
    for &pos in path_end.iter() {
        let r = pos as usize / w;
        let c = pos as usize % w;
        writeln!(out, "{} {}", c, r).unwrap();
    }
}
```
