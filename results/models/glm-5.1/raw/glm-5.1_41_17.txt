; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo_var = OpVariable %ptr_ubo Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_8 = OpConstant %float 8
%float_16 = OpConstant %float 16
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_5 = OpConstant %float 0.5
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_7 = OpConstant %float 0.7
%float_0_2 = OpConstant %float 0.2
%float_0_1_color = OpConstant %float 0.1

%vec3_mortar = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_7
%vec3_brick = OpConstantComposite %vec3 %float_0_7 %float_0_2 %float_0_1_color

%main = OpFunction %void None %functype
%label = OpLabel

%uv = OpLoad %vec2 %in_uv
%n_load = OpLoad %vec3 %in_normal
%n = OpExtInst %vec3 %1 Normalize %n_load

; Tile UV
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1
%tile_x = OpFMul %float %uv_x %float_8
%tile_y = OpFMul %float %uv_y %float_16

; Stagger every other row
%row = OpFloor %float %tile_y
%half_brick = OpFMul %float %float_0_5 %float_8 ; 4.0
%offset_x = OpFMul %float %row %half_brick
%stagger_x = OpFAdd %float %tile_x %offset_x

; Compute fractional coordinates within the brick
%frac_x = OpFract %float %stagger_x
%frac_y = OpFract %float %tile_y

; Mortar test X: frac_x >= 0.05
%cond_x = OpFOrdGreaterThanEqual %bool %frac_x %float_0_05
; Mortar test Y: frac_y >= 0.1
%cond_y = OpFOrdGreaterThanEqual %bool %frac_y %float_0_1

; Both true -> brick, else mortar
%is_brick = OpLogicalAnd %bool %cond_x %cond_y
%base_color = OpSelect %vec3 %is_brick %vec3_brick %vec3_mortar

; Light Pos (offset 192 = 3 * 64)
%light_ptr = OpAccessChain %ptr_ubo_vec4 %ubo_var %int_3
%lightPos_v4 = OpLoad %vec4 %light_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_v4 %lightPos_v4 0 1 2

; Compute light direction and diffuse
%worldPos = OpLoad %vec3 %in_worldPos
%to_light = OpFSub %vec3 %lightPos %worldPos
%to_light_norm = OpExtInst %vec3 %1 Normalize %to_light
%diffuse = OpDot %float %n %to_light_norm
%diffuse_clamped = OpExtInst %float %1 FMax %diffuse %float_0

; Shading factor: 0.4 + 0.6 * diffuse
%factor = OpFMul %float %float_0_6 %diffuse_clamped
%factor_final = OpFAdd %float %float_0_4 %factor

; Final color
%shaded_color = OpVectorTimesScalar %vec3 %base_color %factor_final
%final_vec4 = OpCompositeConstruct %vec4 %shaded_color %float_1

OpStore %outColor %final_vec4

OpReturn
OpFunctionEnd
