use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = Vec::new();
    io::stdin().lock().read_to_end(&mut input).unwrap();
    let mut ptr = 0;
    
    let next = || -> usize {
        while ptr < input.len() && input[ptr] == b' ' { ptr += 1; }
        let sign = if ptr < input.len() && input[ptr] == b'-' { ptr += 1; true } else { false };
        let mut num = 0usize;
        while ptr < input.len() && input[ptr].is_ascii_digit() {
            num = num * 10 + (input[ptr] - b'0') as usize;
            ptr += 1;
        }
        if sign { num.wrapping_neg() } else { num }
    };

    let n = next();
    let m = next();

    let mut degrees = vec![0u32; n];
    let mut edges = Vec::with_capacity(m);

    for _ in 0..m {
        let u = next();
        let v = next();
        edges.push((u, v));
        degrees[u] += 1;
        degrees[v] += 1;
    }

    // 2-phase matching algorithm for Minimum Vertex Cover (MVC) approximation
    let mut in_cover = vec![false; n];
    let mut matched = vec![false; n];

    // Phase 1: Greedy maximum weight matching prioritizing high-degree vertices.
    // This maximizes the degree reduction per selected edge early on.
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by(|&a, &b| degrees[b].cmp(&degrees[a]));

    for &u in &order {
        if !matched[u] {
            let mut best_v = None;
            let mut best_deg = 0;
            
            // Search for the unmatched neighbor with the highest degree
            let start = if u & 1 == 0 { 0 } else { 1 };
            let mut k = start;
            let base_idx = u * 2;
            while k < m * 2 {
                let offset = if k < m { k } else { k - m };
                let (a, b) = edges[offset];
                let v = if a == u { b } else if b == u { a } else { continue; };
                
                if !matched[v] && degrees[v] > best_deg {
                    best_deg = degrees[v];
                    best_v = Some(v);
                }
                k += 2;
            }
            
            if let Some(v) = best_v {
                matched[u] = true;
                matched[v] = true;
                in_cover[u] = true;
                in_cover[v] = true;
                
                // Decrement logical degrees to refine future matching choices
                degrees[u] = degrees[u].saturating_sub(1);
                degrees[v] = degrees[v].saturating_sub(1);
            }
        }
    }

    // Phase 2: Additional standard matching passes to capture any remaining disjoint edges.
    // Since degree adjustments occurred, a second pass guarantees no trivially matchable 
    // high-degree pairs were missed due to initial degree ties or ordering artifacts.
    for &u in &order {
        if !matched[u] {
            let mut best_v = None;
            let mut best_deg = 0;
            
            let start = if u & 1 == 0 { 0 } else { 1 };
            let mut k = start;
            while k < m * 2 {
                let offset = if k < m { k } else { k - m };
                let (a, b) = edges[offset];
                let v = if a == u { b } else if b == u { a } else { continue; };
                
                if !matched[v] && degrees[v] > best_deg {
                    best_deg = degrees[v];
                    best_v = Some(v);
                }
                k += 2;
            }
            
            if let Some(v) = best_v {
                matched[u] = true;
                matched[v] = true;
                in_cover[u] = true;
                in_cover[v] = true;
                degrees[u] = degrees[u].saturating_sub(1);
                degrees[v] = degrees[v].saturating_sub(1);
            }
        }
    }

    // Prune redundant vertices: Remove a vertex from the cover if all its incident 
    // edges are already covered by other vertices in the cover.
    // We check low-degree vertices first as they are more likely to be redundant.
    for &u in &order.iter().rev().copied().collect::<Vec<_>>() {
        if !in_cover[u] { continue; }
        
        let mut needed = false;
        let start = if u & 1 == 0 { 0 } else { 1 };
        let mut k = start;
        while k < m * 2 {
            let offset = if k < m { k } else { k - m };
            let (a, b) = edges[offset];
            let v = if a == u { b } else if b == u { a } else { continue; };
            
            if !in_cover[v] {
                needed = true;
                break;
            }
            k += 2;
        }
        
        if !needed {
            in_cover[u] = false;
        }
    }

    // Output generation
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let count = in_cover.iter().filter(|&&x| x).count();
    writeln!(out, "{}", count).unwrap();

    let mut first = true;
    for u in 0..n {
        if in_cover[u] {
            if !first {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", u).unwrap();
            first = false;
        }
    }
    writeln!(out).unwrap();
}
