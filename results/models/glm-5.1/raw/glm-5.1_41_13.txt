; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr = OpTypePointer Uniform %UBO

%in_ptr = OpTypePointer Input %vec3
%out_ptr = OpTypePointer Output %vec4

%in_worldPos = OpVariable %in_ptr Input
%outColor = OpVariable %out_ptr Output
%ubo_var = OpVariable %ubo_ptr Uniform

%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0 = OpConstant %float 0.0

%main = OpFunction %void None %func_type
%entry = OpLabel

%pos = OpLoad %vec3 %in_worldPos
%y = OpCompositeExtract %float %pos 1
%scaled = OpFMul %float %y %float_0_5
%shifted = OpFAdd %float %scaled %float_0_5
%t = OpFClamp %float %shifted %float_0 %float_1

%bottom_r = OpFMul %float %t %float_0_1
%bottom_g = OpFMul %float %t %float_0_4
%bottom_b = OpFMul %float %t %float_0_1

%inv_t = OpFSub %float %float_1 %t
%top_r = OpFMul %float %inv_t %float_1
%top_g = OpFMul %float %inv_t %float_1
%top_b = OpFMul %float %inv_t %float_1

%r = OpFAdd %float %bottom_r %top_r
%g = OpFAdd %float %bottom_g %top_g
%b = OpFAdd %float %bottom_b %top_b

%color = OpCompositeConstruct %vec4 %r %g %b %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
