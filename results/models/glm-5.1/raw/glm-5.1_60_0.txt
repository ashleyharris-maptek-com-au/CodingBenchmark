```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount;
const tasks = data.tasks;

const n = tasks.length;
if (n === 0) {
    process.stdout.write(JSON.stringify({ schedule: [] }));
    process.exit(0);
}

// Preprocess tasks
for (let i = 0; i < n; i++) {
    tasks[i].depsSet = new Set(tasks[i].deps);
}

const taskMap = new Map();
for (const t of tasks) {
    taskMap.set(t.id, t);
}

// 1. Compute bottom level (longest path from task to exit)
const bottomLevel = new Map();
const memoBL = new Map();

function getBottomLevel(task) {
    if (memoBL.has(task.id)) return memoBL.get(task.id);
    let maxBL = task.duration;
    for (const depId of task.deps) {
        const dep = taskMap.get(depId);
        const bl = getBottomLevel(dep);
        if (bl + task.duration > maxBL) {
            maxBL = bl + task.duration;
        }
    }
    memoBL.set(task.id, maxBL);
    return maxBL;
}

for (const t of tasks) {
    getBottomLevel(t);
}

// 2. Compute top level (longest path from entry to task's start)
const topLevel = new Map();
topLevel.set(tasks[0].id, 0); // dummy

const sortedTasks = [...tasks].sort((a, b) => (topLevel.get(a.id) || 0) + memoBL.get(a.id) - ((topLevel.get(b.id) || 0) + memoBL.get(b.id)));

for (const t of sortedTasks) {
    for (const depId of t.deps) {
        const dep = taskMap.get(depId);
        const newTop = (topLevel.get(depId) || 0) + dep.duration;
        if (newTop > (topLevel.get(t.id) || 0)) {
            topLevel.set(t.id, newTop);
        }
    }
}

// 3. Compute urgency (earliest possible start time / deadline slack)
const urgency = new Map();
for (const t of tasks) {
    const slack = (t.deadline || Infinity) - ((topLevel.get(t.id) || 0) + t.duration);
    urgency.set(t.id, slack);
}

// 4. Compute transitive dependency count for tie-breaking
const transitiveDeps = new Map();

function getTransitiveDeps(taskId) {
    if (transitiveDeps.has(taskId)) return transitiveDeps.get(taskId);
    const task = taskMap.get(taskId);
    const visited = new Set();
    const stack = [...task.deps];
    while (stack.length > 0) {
        const curr = stack.pop();
        if (!visited.has(curr)) {
            visited.add(curr);
            const currTask = taskMap.get(curr);
            for (const d of currTask.deps) {
                stack.push(d);
            }
        }
    }
    transitiveDeps.set(taskId, visited.size);
    return visited.size;
}

for (const t of tasks) {
    getTransitiveDeps(t.id);
}

// 5. Compute ready times from direct dependencies
const readyTimes = new Map();
for (const t of tasks) {
    let maxReady = 0;
    for (const depId of t.deps) {
        const dep = taskMap.get(depId);
        const depReady = (readyTimes.get(depId) || 0) + dep.duration;
        if (depReady > maxReady) {
            maxReady = depReady;
        }
    }
    readyTimes.set(t.id, maxReady);
}

// Sort tasks for the greedy schedule
tasks.sort((a, b) => {
    // Primary: urgency (earliest deadline slack first)
    const uA = urgency.get(a.id);
    const uB = urgency.get(b.id);
    if (uA !== uB) return uA - uB;

    // Secondary: bottom level (longest critical path first)
    const blA = memoBL.get(a.id);
    const blB = memoBL.get(b.id);
    if (blA !== blB) return blB - blA;

    // Tertiary: transitive dependency count (more dependencies first)
    const tdA = transitiveDeps.get(a.id);
    const tdB = transitiveDeps.get(b.id);
    if (tdA !== tdB) return tdB - tdA;

    // Quaternary: ready time (earliest ready first)
    const rA = readyTimes.get(a.id);
    const rB = readyTimes.get(b.id);
    if (rA !== rB) return rA - rB;

    // Final tie-breaker: ID string numeric value
    const idA = parseInt(a.id, 10) || 0;
    const idB = parseInt(b.id, 10) || 0;
    return idA - idB;
});

// Extract sorted IDs for the output
const order = tasks.map(t => t.id);

process.stdout.write(JSON.stringify({ order }));
```
