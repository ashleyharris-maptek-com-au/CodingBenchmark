```spirv
; SPIR-V
Version 1.0
Generator 0
Bound 100
Schema 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 3

; Types
%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
%float_1 = OpConstant %float 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_2
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr

; Load p1
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1 = OpLoad %v3float %p1_ptr

; Load c0
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr

; Load c1
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

; Load clip positions
%clip0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%clip0 = OpLoad %v4float %clip0_ptr
%clip1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%clip1 = OpLoad %v4float %clip1_ptr

; Compute line direction (p1 - p0)
%dir = OpFSub %v3float %p1 %p0

; Compute perpendicular in XY plane: (-dir.y, dir.x, 0)
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_xy = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0

; Normalize perpendicular
%len_sq = OpDot %float %perp_xy %perp_xy
%len = OpFMod %float %len_sq %len_sq ; Avoid sqrt(0) by relying on valid lines; normal robust approach:
%len_rt = OpExtInst %float %1 Sqrt %len_sq
%perp_norm = OpFDiv %v3float %perp_xy %len_rt
%is_zero = OpFOrdEqual %bool %len_rt %float_0 ; Need bool type
%bool = OpTypeBool
%perp = OpSelect %v3float %is_zero %v3float_0_0_0 %perp_norm ; fallback to zero if degenerate
%v3float_0_0_0 = OpConstantNull %v3float

; Compute thick offsets
%thick_start_v = OpVectorTimesScalar %v3float %perp %float_0_04
%thick_end_v = OpVectorTimesScalar %v3float %perp %float_0_005

; Vertex 1: p0 + perp*thick_start, c0
%v1_pos = OpFAdd %v3float %p0 %thick_start_v
%v1_clip = OpCompositeConstruct %v4float %v1_pos %float_1 ; Assuming simple projection to clip
%gl_out_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr1 %clip0 ; Store actual clip position for correctness
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p0 - perp*thick_start, c0
%v2_pos = OpFSub %v3float %p0 %thick_start_v
%gl_out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr2 %clip0
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: p1 + perp*thick_end, c1
%v3_pos = OpFAdd %v3float %p1 %thick_end_v
%gl_out_pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr3 %clip1
OpStore %outColor %c1
OpEmitVertex

; Vertex 4: p1 - perp*thick_end, c1
%v4_pos = OpFSub %v3float %p1 %thick_end_v
%gl_out_pos_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr4 %clip1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
