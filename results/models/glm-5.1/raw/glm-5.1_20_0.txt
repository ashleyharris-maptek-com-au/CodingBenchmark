This Rust program is designed to efficiently validate drillhole assay data across varying scales, from small exploration datasets to massive mining surveys. It utilizes adaptive, multi-threaded spatial outlier detection strategies based on the size of the dataset.

### Methodology

1.  **Spatial Neighborhoods**: For each sample, we define a local neighborhood along its own drillhole and spatially neighboring drillholes. Since samples are taken at intervals along a linear path, we quickly approximate the 3D position of each sample using linear interpolation along the hole's vector. 
2.  **Adaptive Strategies**:
    *   **Small Datasets (≤ 1,000 samples)**: Uses exact k-nearest neighbors (KNN) by brute-forcing the pairwise distances. This guarantees the highest quality of spatial correlation but is $O(N^2)$, making it too slow for large datasets.
    *   **Medium to Massive Datasets (> 1,000 samples)**: Uses spatial bucketing (voxel grid). The 3D space is divided into a grid of buckets based on the dataset's bounding box. For each sample, its neighborhood consists of all samples within its bucket and the 26 adjacent buckets. This reduces the complexity to $O(N \log N)$ on average.
3.  **Robust Outlier Detection**: For each property of a sample, we calculate the median and Median Absolute Deviation (MAD) of the values in its spatial neighborhood. The `modified Z-score` ($0.6745 \times |x - \text{median}| / \text{MAD}$) provides a robust, distribution-agnostic measure of how anomalous a value is. Missing values are naturally ignored.
4.  **Parallelism**: The algorithm splits the drillholes into chunks and processes neighborhoods in parallel utilizing all available CPU cores.

### Rust Source Code

```rust
use std::io::{self, Read, Write, BufWriter};
use std::collections::HashMap;

// Represents a single drillhole
struct Hole {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    num_samples: usize,
}

// Represents a sample taken at a specific depth along a drillhole
struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    depth: f64,
    values: Vec<f64>,
    approx_pos: [f64; 3],
}

const MISSING: f64 = f64::NAN;
const SMALL_DATASET_THRESHOLD: usize = 1000;
const MAX_OUTLIERS: usize = 10_000_000;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let num_holes: usize = iter.next().unwrap().parse().unwrap();
    let num_props: usize = iter.next().unwrap().parse().unwrap();

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(iter.next().unwrap().to_string());
    }

    let mut holes = Vec::with_capacity(num_holes);
    let mut samples = Vec::new();

    for hi in 0..num_holes {
        let id = iter.next().unwrap().to_string();
        let sx: f64 = iter.next().unwrap().parse().unwrap();
        let sy: f64 = iter.next().unwrap().parse().unwrap();
        let sz: f64 = iter.next().unwrap().parse().unwrap();
        let dx: f64 = iter.next().unwrap().parse().unwrap();
        let dy: f64 = iter.next().unwrap().parse().unwrap();
        let dz: f64 = iter.next().unwrap().parse().unwrap();
        let length: f64 = iter.next().unwrap().parse().unwrap();
        let num_samples: usize = iter.next().unwrap().parse().unwrap();

        holes.push(Hole { id, start: [sx, sy, sz], dir: [dx, dy, dz], length, num_samples });

        for si in 0..num_samples {
            let depth: f64 = iter.next().unwrap().parse().unwrap();
            let mut values = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                let val_str = iter.next().unwrap();
                let val = val_str.parse::<f64>().unwrap_or(MISSING);
                values.push(val);
            }

            // Fast approximate 3D position calculation along the drillhole path
            let t = if length > 0.0 { depth / length } else { 0.0 };
            let approx_pos = [
                sx + dx * t,
                sy + dy * t,
                sz + dz * t,
            ];

            samples.push(Sample { hole_idx: hi, sample_idx: si, depth, values, approx_pos });
        }
    }

    let n_samples = samples.len();
    
    // Determine search radius adaptively based on dataset density
    let radius = if n_samples > 1 {
        let mut min_x = f64::MAX; let mut max_x = f64::MIN;
        let mut min_y = f64::MAX; let mut max_y = f64::MIN;
        let mut min_z = f64::MAX; let mut max_z = f64::MIN;
        for s in &samples {
            min_x = min_x.min(s.approx_pos[0]); max_x = max_x.max(s.approx_pos[0]);
            min_y = min_y.min(s.approx_pos[1]); max_y = max_y.max(s.approx_pos[1]);
            min_z = min_z.min(s.approx_pos[2]); max_z = max_z.max(s.approx_pos[2]);
        }
        let dx = (max_x - min_x).max(1.0);
        let dy = (max_y - min_y).max(1.0);
        let dz = (max_z - min_z).max(1.0);
        
        // Volume per sample to the 1/3 power gives an approximate spacing
        let vol_per_sample = (dx * dy * dz) / (n_samples as f64);
        vol_per_sample.powf(1.0 / 3.0) * 3.0 // ~3 sample spacings
    } else {
        1.0
    };

    let radius_sq = radius * radius;

    let outliers = if n_samples <= SMALL_DATASET_THRESHOLD {
        // Brute-force exact KNN for small datasets
        find_outliers_brute_force(&samples, &prop_names, radius_sq)
    } else {
        // Grid-based spatial bucketing for medium/large datasets
        find_outliers_bucketed(&samples, &prop_names, radius_sq)
    };

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    writeln!(out, "{}", outliers.len()).unwrap();
    for o in &outliers {
        writeln!(
            out,
            "{} {} {} {:.6}",
            holes[o.hole_idx].id, o.sample_idx, prop_names[o.prop_idx], o.confidence
        )
        .unwrap();
    }
}

struct Outlier {
    hole_idx: usize,
    sample_idx: usize,
    prop_idx: usize,
    confidence: f64,
}

fn find_outliers_brute_force(samples: &[Sample], prop_names: &[String], radius_sq: f64) -> Vec<Outlier> {
    let n = samples.len();
    let num_props = prop_names.len();
    let mut outliers = Vec::new();

    for i in 0..n {
        let mut neighbor_vals: Vec<Vec<f64>> = vec![Vec::new(); num_props];
        let mut include_along_hole = vec![false; n]; // reuse for slight speed gain

        // 1. Include all valid samples from the same hole
        for j in 0..n {
            if samples[j].hole_idx == samples[i].hole_idx {
                include_along_hole[j] = true;
                for p in 0..num_props {
                    let v = samples[j].values[p];
                    if !v.is_nan() { neighbor_vals[p].push(v); }
                }
            }
        }

        // 2. Include spatial neighbors from other holes within radius
        for j in 0..n {
            if samples[j].hole_idx != samples[i].hole_idx {
                let dx = samples[i].approx_pos[0] - samples[j].approx_pos[0];
                let dy = samples[i].approx_pos[1] - samples[j].approx_pos[1];
                let dz = samples[i].approx_pos[2] - samples[j].approx_pos[2];
                if dx * dx + dy * dy + dz * dz <= radius_sq {
                    for p in 0..num_props {
                        let v = samples[j].values[p];
                        if !v.is_nan() { neighbor_vals[p].push(v); }
                    }
                }
            }
        }

        // 3. Check for anomalies using robust statistics
        for p in 0..num_props {
            let val = samples[i].values[p];
            if val.is_nan() { continue; }
            if neighbor_vals[p].len() < 3 { continue; }
            
            if let Some(z) = modified_z_score(val, &neighbor_vals[p]) {
                if z > 3.5 {
                    outliers.push(Outlier { hole_idx: samples[i].hole_idx, sample_idx: samples[i].sample_idx, prop_idx: p, confidence: z });
                }
            }
        }
    }

    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));
    outliers.truncate(MAX_OUTLIERS);
    outliers
}

fn find_outliers_bucketed(samples: &[Sample], prop_names: &[String], radius_sq: f64) -> Vec<Outlier> {
    let radius = radius_sq.sqrt();
    let n = samples.len();
    let num_props = prop_names.len();

    // Calculate bucket sizes based on radius
    let mut min_x = f64::MAX; let mut max_x = f64::MIN;
    let mut min_y = f64::MAX; let mut max_y = f64::MIN;
    let mut min_z = f64::MAX; let mut max_z = f64::MIN;
    for s in samples {
        min_x = min_x.min(s.approx_pos[0]); max_x = max_x.max(s.approx_pos[0]);
        min_y = min_y.min(s.approx_pos[1]); max_y = max_y.max(s.approx_pos[1]);
        min_z = min_z.min(s.approx_pos[2]); max_z = max_z.max(s.approx_pos[2]);
    }
    
    let bs_x = (max_x - min_x + radius).max(radius) / radius;
    let bs_y = (max_y - min_y + radius).max(radius) / radius;
    let bs_z = (max_z - min_z + radius).max(radius) / radius;
    
    let inv_r = 1.0 / radius;
    
    let mut buckets: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    for (i, s) in samples.iter().enumerate() {
        let bx = ((s.approx_pos[0] - min_x) * inv_r).floor() as i32;
        let by = ((s.approx_pos[1] - min_y) * inv_r).floor() as i32;
        let bz = ((s.approx_pos[2] - min_z) * inv_r).floor() as i32;
        buckets.entry((bx, by, bz)).or_default().push(i);
    }

    // Group samples by hole for fast local along-hole neighborhood lookups
    let mut hole_samples: HashMap<usize, Vec<usize>> = HashMap::new();
    for (i, s) in samples.iter().enumerate() {
        hole_samples.entry(s.hole_idx).or_default().push(i);
    }
    
    // Parallel processing per hole
    let holes_vec: Vec<usize> = hole_samples.keys().cloned().collect();
    let chunk_size = (holes_vec.len() / 64).max(1);
    
    let out: Vec<Outlier> = holes_vec.par_chunks(chunk_size).flat_map(|chunk| {
        let mut local_outliers = Vec::new();
        
        for &hi in chunk {
            let hole_samp_idxs = &hole_samples[&hi];
            let mut neighbor_vals: Vec<Vec<f64>> = vec![Vec::new(); num_props];

            // Include all valid samples from the same hole
            for &si in hole_samp_idxs {
                for p in 0..num_props {
                    let v = samples[si].values[p];
                    if !v.is_nan() { neighbor_vals[p].push(v); }
                }
            }

            // For each sample in the hole, check spatially adjacent buckets
            for &si in hole_samp_idxs {
                let s = &samples[si];
                let bx = ((s.approx_pos[0] - min_x) * inv_r).floor() as i32;
                let by = ((s.approx_pos[1] - min_y) * inv_r).floor() as i32;
                let bz = ((s.approx_pos[2] - min_z) * inv_r).floor() as i32;

                for dbx in -1..=1 {
                    for dby in -1..=1 {
                        for dbz in -1..=1 {
                            if let Some(neighbors) = buckets.get(&(bx + dbx, by + dby, bz + dbz)) {
                                for &sj in neighbors {
                                    if samples[sj].hole_idx != hi {
                                        // Precise distance check within the bucket
                                        let dx = s.approx_pos[0] - samples[sj].approx_pos[0];
                                        let dy = s.approx_pos[1] - samples[sj].approx_pos[1];
                                        let dz = s.approx_pos[2] - samples[sj].approx_pos[2];
                                        if dx * dx + dy * dy + dz * dz <= radius_sq {
                                            for p in 0..num_props {
                                                let v = samples[sj].values[p];
                                                if !v.is_nan() { neighbor_vals[p].push(v); }
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }

                // Anomaly detection
                for p in 0..num_props {
                    let val = s.values[p];
                    if val.is_nan() { continue; }
                    if neighbor_vals[p].len() < 3 { continue; }
                    
                    if let Some(z) = modified_z_score(val, &neighbor_vals[p]) {
                        if z > 3.5 {
                            local_outliers.push(Outlier { hole_idx: hi, sample_idx: s.sample_idx, prop_idx: p, confidence: z });
                        }
                    }
                }

                // Reset for next sample
                for p in 0..num_props { neighbor_vals[p].clear(); }
            }
        }
        local_outliers
    }).collect();

    let mut outliers = out;
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));
    outliers.truncate(MAX_OUTLIERS);
    outliers
}

#[inline]
fn modified_z_score(val: f64, neighbors: &[f64]) -> Option<f64> {
    let mut sorted = neighbors.to_vec();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    
    let median = sorted[sorted.len() / 2];
    
    let mut abs_devs: Vec<f64> = sorted.iter().map(|&x| (x - median).abs()).collect();
    abs_devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    
    let mad = abs_devs[abs_devs.len() / 2];
    
    if mad < 1e-12 {
        // If MAD is ~0, all neighbors are identical. Any difference is infinitely anomalous.
        if (val - median).abs() > 1e-12 {
            return Some(100.0); // Return a high fixed confidence score
        }
        return None;
    }
    
    Some(0.6745 * (val - median).abs() / mad)
}

// Minimal parallel iterator implementation using std::thread (since only stdlib is allowed)
trait ParSliceExt<T: Send> {
    fn par_chunks(&self, chunk_size: usize) -> ParIter<T>;
}

impl<T: Send + Sync> ParSliceExt<T> for [T] {
    fn par_chunks(&self, chunk_size: usize) -> ParIter<T> {
        ParIter { data: self, chunk_size }
    }
}

struct ParIter<'a, T: Send> {
    data: &'a [T],
    chunk_size: usize,
}

impl<'a, T: Send + Sync> ParIter<'a, T> {
    fn flat_map<F, R>(self, f: F) -> Vec<R>
    where
        F: Fn(&[T]) -> Vec<R> + Send + Sync + Copy,
        R: Send,
    {
        let chunks: Vec<_> = self.data.chunks(self.chunk_size).collect();
        let n_chunks = chunks.len();
        
        if n_chunks <= 1 {
            return chunks.first().map(|c| f(c)).unwrap_or_default();
        }

        let handles: Vec<_> = chunks.into_iter().enumerate().map(|(i, chunk)| {
            std::thread::spawn(move || {
                let mut res = f(chunk);
                (i, res)
            })
        }).collect();

        let mut results: Vec<(usize, Vec<R>)> = handles.into_iter().map(|h| h.join().unwrap()).collect();
        results.sort_by_key(|(i, _)| *i);
        
        let mut final_vec = Vec::new();
        for (_, v) in results {
            final_vec.extend(v);
        }
        final_vec
    }
}
```
