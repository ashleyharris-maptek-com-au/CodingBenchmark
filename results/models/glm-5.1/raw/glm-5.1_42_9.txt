; SPIR-V
Version: 1.0
Generator: 0
Bound: 200
Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_025 = OpConstant %float 0.025
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %functype
%label = OpLabel

; Load clip-space positions (XY) for all 4 vertices
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1

%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

; Compute tangent at start (t0): normalize(p2_xy - p0_xy)
%t0_unnorm = OpFSub %v2float %p2_xy %p0_xy
%t0_len = OpExtInst %float %1 Length %t0_unnorm
%t0_len_safe = OpFMax %float %t0_len %float_025
%t0 = OpFDiv %v2float %t0_unnorm %t0_len_safe

; Compute tangent at end (t1): normalize(p3_xy - p1_xy)
%t1_unnorm = OpFSub %v2float %p3_xy %p1_xy
%t1_len = OpExtInst %float %1 Length %t1_unnorm
%t1_len_safe = OpFMax %float %t1_len %float_025
%t1 = OpFDiv %v2float %t1_unnorm %t1_len_safe

; Compute perpendicular at start (perp0): vec2(-t0.y, t0.x)
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

; Compute perpendicular at end (perp1): vec2(-t1.y, t1.x)
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Offset vectors = perp * 0.025
%offset0 = OpVectorTimesScalar %v2float %perp0 %float_025
%offset1 = OpVectorTimesScalar %v2float %perp1 %float_025

; Load colors
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

; --- Emit Vertex 0: start + perp0*thickness ---
%v0_xy = OpFAdd %v2float %p1_xy %offset0
%gl_pos0 = OpCompositeConstruct %v4float %v0_x %v0_y %float_0 %float_1
%out_pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0_ptr %gl_pos0
OpStore %outColor %c1
OpEmitVertex

; --- Emit Vertex 1: start - perp0*thickness ---
%neg_offset0 = OpFNegate %v2float %offset0
%v1_xy = OpFAdd %v2float %p1_xy %neg_offset0
%gl_pos1 = OpCompositeConstruct %v4float %v1_x %v1_y %float_0 %float_1
%out_pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1_ptr %gl_pos1
OpStore %outColor %c1
OpEmitVertex

; --- Emit Vertex 2: end + perp1*thickness ---
%v2_xy = OpFAdd %v2float %p2_xy %offset1
%gl_pos2 = OpCompositeConstruct %v4float %v2_x %v2_y %float_0 %float_1
%out_pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2_ptr %gl_pos2
OpStore %outColor %c2
OpEmitVertex

; --- Emit Vertex 3: end - perp1*thickness ---
%neg_offset1 = OpFNegate %v2float %offset1
%v3_xy = OpFAdd %v2float %p2_xy %neg_offset1
%gl_pos3 = OpCompositeConstruct %v4float %v3_x %v3_y %float_0 %float_1
%out_pos3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos3_ptr %gl_pos3
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
