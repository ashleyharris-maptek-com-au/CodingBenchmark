This is a 1D Cutting Stock Problem. To solve it efficiently for millions of cuts, we use a heuristic inspired by the **First Fit Decreasing (FFD)** algorithm. 

The FFD algorithm works by sorting the cuts in descending order of length and then greedily assigning each cut to the first stock that has enough remaining capacity. To make this process extremely fast (close to $O(N \log N)$), we maintain the stocks in a **Max-Heap** (Priority Queue) based on their remaining capacity. This allows us to instantly find the best-fit stock for each cut and re-insert it in logarithmic time. 

To achieve maximum performance:
1. We read input using a highly-optimized custom `FastReader` that minimizes allocations and avoids `string.Split()`.
2. We use a custom struct-based Max-Heap instead of the built-in `PriorityQueue` to avoid costly interface dispatches, `Tuple` allocations, and property accesses.
3. Since we only need the remaining capacity to determine fit, we don't store the cut indices inside the heap; instead, we maintain a simple `stockRemaining` array.
4. We track the assignments using a native `int[][]` with pre-allocated `int[]` buffers to completely avoid `List<int>` resizing and memory overhead.

```csharp
using System;
using System.IO;

public class Program
{
    // Custom Max-Heap struct for ultimate performance (no interface dispatch, no Tuple allocations)
    private struct MaxHeap
    {
        private int[] _heap;
        private int _size;

        public MaxHeap(int capacity)
        {
            _heap = new int[capacity];
            _size = 0;
        }

        [System.Runtime.CompilerServices.MethodImpl(System.Runtime.CompilerServices.MethodImplOptions.AggressiveInlining)]
        public void Push(int value)
        {
            _heap[_size] = value;
            int i = _size;
            _size++;

            while (i > 0)
            {
                int parent = (i - 1) >> 1;
                if (_heap[parent] < _heap[i])
                {
                    int temp = _heap[parent];
                    _heap[parent] = _heap[i];
                    _heap[i] = temp;
                    i = parent;
                }
                else
                {
                    break;
                }
            }
        }

        [System.Runtime.CompilerServices.MethodImpl(System.Runtime.CompilerServices.MethodImplOptions.AggressiveInlining)]
        public int PopMax()
        {
            int max = _heap[0];
            _size--;
            _heap[0] = _heap[_size];

            int i = 0;
            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= _size) break;

                int right = left + 1;
                int largest = left;

                if (right < _size && _heap[right] > _heap[left])
                {
                    largest = right;
                }

                if (_heap[largest] > _heap[i])
                {
                    int temp = _heap[largest];
                    _heap[largest] = _heap[i];
                    _heap[i] = temp;
                    i = largest;
                }
                else
                {
                    break;
                }
            }

            return max;
        }
    }

    public static void Main()
    {
        var reader = new FastReader();

        int n = reader.ReadInt();
        int stockLength = reader.ReadInt();

        int[] cuts = new int[n];
        int[] indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = reader.ReadInt();
            indices[i] = i;
        }

        // Sort cuts in descending order of length while keeping track of original indices
        Array.Sort(cuts, indices);
        Array.Reverse(cuts);
        Array.Reverse(indices);

        // Stores the remaining capacity of each stock piece
        int[] stockRemaining = new int[n];
        // Stores the cuts assigned to each stock piece
        int[][] stockCuts = new int[n][];
        // Tracks how many cuts are assigned to each stock piece
        int[] stockCutCount = new int[n];

        // Pre-allocate arrays for assignments to avoid List<int> resizing overhead
        for (int i = 0; i < n; i++)
        {
            stockCuts[i] = new int[4]; // Start with a small capacity, will resize if needed
        }

        MaxHeap heap = new MaxHeap(n);
        int numStocks = 0;

        // First Fit Decreasing using Max-Heap
        for (int i = 0; i < n; i++)
        {
            int cut = cuts[i];
            int originalIndex = indices[i];
            int stockIdx = -1;

            if (heap._size > 0 && heap._heap[0] >= cut)
            {
                // A stock has enough remaining capacity
                int maxRemaining = heap.PopMax();

                // Find which stock actually has this remaining capacity
                // A simple linear scan is extremely fast because it operates on contiguous memory
                // and fully benefits from CPU cache lines and hardware prefetching.
                for (int j = 0; j < numStocks; j++)
                {
                    if (stockRemaining[j] == maxRemaining)
                    {
                        stockIdx = j;
                        break;
                    }
                }

                // Assign the cut to this stock
                int count = stockCutCount[stockIdx];
                if (count == stockCuts[stockIdx].Length)
                {
                    Array.Resize(ref stockCuts[stockIdx], count * 2);
                }
                stockCuts[stockIdx][count] = originalIndex;
                stockCutCount[stockIdx] = count + 1;
                stockRemaining[stockIdx] -= cut;

                // Re-insert updated stock back into the heap
                heap.Push(stockRemaining[stockIdx]);
            }
            else
            {
                // Need to open a new stock piece
                stockIdx = numStocks;
                numStocks++;

                stockRemaining[stockIdx] = stockLength - cut;
                stockCuts[stockIdx][0] = originalIndex;
                stockCutCount[stockIdx] = 1;

                heap.Push(stockRemaining[stockIdx]);
            }
        }

        // Fast output using StreamWriter buffered writing
        using (var writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.UTF8, 8192))
        {
            writer.WriteLine(numStocks);
            for (int i = 0; i < numStocks; i++)
            {
                int count = stockCutCount[i];
                // Build string manually to avoid string.Concat/Join overhead on small arrays
                for (int j = 0; j < count; j++)
                {
                    if (j > 0) writer.Write(' ');
                    writer.Write(stockCuts[i][j]);
                }
                writer.WriteLine();
            }
        }
    }

    // Highly optimized reader for fast stdin parsing
    private class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _index;
        private int _count;

        public FastReader()
        {
            _stream = Console.OpenStandardInput();
            _buffer = new byte[1 << 20]; // 1 MB buffer
            _index = 0;
            _count = 0;
        }

        [System.Runtime.CompilerServices.MethodImpl(System.Runtime.CompilerServices.MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_index >= _count)
            {
                _count = _stream.Read(_buffer, 0, _buffer.Length);
                if (_count <= 0) return -1;
                _index = 0;
            }
            return _buffer[_index++];
        }

        public int ReadInt()
        {
            int b;
            // Skip leading whitespace
            do
            {
                b = ReadByte();
            } while (b >= 0 && b <= ' ');

            int sign = 1;
            if (b == '-')
            {
                sign = -1;
                b = ReadByte();
            }

            int value = 0;
            while (b > ' ')
            {
                value = value * 10 + (b - '0');
                b = ReadByte();
            }

            return value * sign;
        }
    }
}
```
