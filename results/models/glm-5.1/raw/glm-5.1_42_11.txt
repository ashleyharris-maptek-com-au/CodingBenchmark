```spirv
; SPIR-V
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0
OpDecorate %UBO Block

OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpMemberDecorate %UBO 6 Offset 228
OpMemberDecorate %UBO 7 Offset 232

%void = OpTypeVoid
%fn_main = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%mat4v4float = OpTypeMatrix %v4float 4

%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_0_02 = OpConstant %float 0.02
%float_1_3 = OpConstant %float 0.333333
%float_2_3 = OpConstant %float 0.666667

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v3float %v3float %v3float %v3float %float
%ptr_ubo = OpTypePointer Uniform %UBO
%UBO_var = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %fn_main
%label = OpLabel

; Read start (gl_in[1]) and end (gl_in[2]) positions
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; Read adjacency positions for tangent smoothing
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

; Read colors
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

; Compute direction vectors
%dir1 = OpFSub %v4float %p1 %p0
%dir2 = OpFSub %v4float %p2 %p1
%dir3 = OpFSub %v4float %p3 %p2

; Smooth tangents (average of adjacent segments)
%t_start = OpFAdd %v4float %dir1 %dir2
%t_end = OpFAdd %v4float %dir2 %dir3

; Extract XY components
%t_start_xy = OpVectorShuffle %v2float %t_start %t_start 0 1
%t_end_xy = OpVectorShuffle %v2float %t_end %t_end 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1

; Calculate perpendicular vectors (rotated 90 degrees)
%n_start_xy = OpCompositeExtract %float %t_start_xy 1
%n_start_xx = OpCompositeExtract %float %t_start_xy 0
%n_start_y = OpFNegate %float %n_start_xx
%n_start = OpCompositeConstruct %v2float %n_start_xy %n_start_y

%n_end_xy = OpCompositeExtract %float %t_end_xy 1
%n_end_xx = OpCompositeExtract %float %t_end_xy 0
%n_end_y = OpFNegate %float %n_end_xx
%n_end = OpCompositeConstruct %v2float %n_end_xy %n_end_y

; Normalize perpendiculars
%len_start = OpDot %float %n_start %n_start
%len_end = OpDot %float %n_end %n_end
%safe_len_start = OpFMax %float %len_start %float_0
%safe_len_end = OpFMax %float %len_end %float_0
%inv_len_start = OpInverseSqrt %float %safe_len_start
%inv_len_end = OpInverseSqrt %float %safe_len_end
%norm_start = OpVectorTimesScalar %v2float %n_start %inv_len_start
%norm_end = OpVectorTimesScalar %v2float %n_end %inv_len_end

; Scale perpendiculars by thickness
%offset_start = OpVectorTimesScalar %v2float %norm_start %float_0_02
%offset_end = OpVectorTimesScalar %v2float %norm_end %float_0_02

; Compute interpolated positions on the segment (Z and W from p1 and p2)
%p1_zw = OpVectorShuffle %v2float %p1 %p1 2 3
%p2_zw = OpVectorShuffle %v2float %p2 %p2 2 3

%frac_1_3 = OpFMul %v2float %p1_zw %float_1_3
%frac_2_3 = OpFMul %v2float %p2_zw %float_2_3
%zw_1_3 = OpFAdd %v2float %frac_1_3 %frac_2_3

%frac_2_3_p1 = OpFMul %v2float %p1_zw %float_2_3
%frac_1_3_p2 = OpFMul %v2float %p2_zw %float_1_3
%zw_2_3 = OpFAdd %v2float %frac_2_3_p1 %frac_1_3_p2

; Interpolate offsets for the internal points
%off_1_3 = OpFMul %v2float %offset_start %float_2_3
%off_2_3_tmp = OpFMul %v2float %offset_end %float_1_3
%off_1_3_final = OpFAdd %v2float %off_1_3 %off_2_3_tmp

%off_2_3_tmp2 = OpFMul %v2float %offset_start %float_1_3
%off_2_3_tmp3 = OpFMul %v2float %offset_end %float_2_3
%off_2_3_final = OpFAdd %v2float %off_2_3_tmp2 %off_2_3_tmp3

; Interpolate colors
%c_1_3_tmp = OpFMul %v3float %c1 %float_2_3
%c_1_3_tmp2 = OpFMul %v3float %c2 %float_1_3
%c_1_3 = OpFAdd %v3float %c_1_3_tmp %c_1_3_tmp2

%c_2_3_tmp = OpFMul %v3float %c1 %float_1_3
%c_2_3_tmp2 = OpFMul %v3float %c2 %float_2_3
%c_2_3 = OpFAdd %v3float %c_2_3_tmp %c_2_3_tmp2

; --- QUAD 1 (0.0 to 0.3333) ---

; Vertex 0 (Start, -offset)
%xy_0 = OpFSub %v2float %p1_xy %offset_start
%pos_0 = OpCompositeConstruct %v4float %xy_0 %p1_zw
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %pos_0
%out_col_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_col_ptr %c1
OpEmitVertex

; Vertex 1 (Start, +offset)
%xy_1 = OpFAdd %v2float %p1_xy %offset_start
%pos_1 = OpCompositeConstruct %v4float %xy_1 %p1_zw
OpStore %out_pos_ptr %pos_1
OpStore %out_col_ptr %c1
OpEmitVertex

; Vertex 2 (1/3, -offset)
%xy_2 = OpFSub %v2float %p1_xy %off_1_3_final
%pos_2 = OpCompositeConstruct %v4float %xy_2 %zw_1_3
OpStore %out_pos_ptr %pos_2
OpStore %out_col_ptr %c_1_3
OpEmitVertex

; Vertex 3 (1/3, +offset)
%xy_3 = OpFAdd %v2float %p1_xy %off_1_3_final
%pos_3 = OpCompositeConstruct %v4float %xy_3 %zw_1_3
OpStore %out_pos_ptr %pos_3
OpStore %out_col_ptr %c_1_3
OpEmitVertex

OpEndPrimitive

; --- QUAD 2 (0.6667 to 1.0) ---

; Vertex 4 (2/3, -offset)
%xy_4 = OpFSub %v2float %p1_xy %off_2_3_final
%pos_4 = OpCompositeConstruct %v4float %xy_4 %zw_2_3
OpStore %out_pos_ptr %pos_4
OpStore %out_col_ptr %c_2_3
OpEmitVertex

; Vertex 5 (2/3, +offset)
%xy_5 = OpFAdd %v2float %p1_xy %off_2_3_final
%pos_5 = OpCompositeConstruct %v4float %xy_5 %zw_2_3
OpStore %out_pos_ptr %pos_5
OpStore %out_col_ptr %c_2_3
OpEmitVertex

; Vertex 6 (End, -offset)
%xy_6 = OpFSub %v2float %p2_xy %offset_end
%pos_6 = OpCompositeConstruct %v4float %xy_6 %p2_zw
OpStore %out_pos_ptr %pos_6
OpStore %out_col_ptr %c2
OpEmitVertex

; Vertex 7 (End, +offset)
%xy_7 = OpFAdd %v2float %p2_xy %offset_end
%pos_7 = OpCompositeConstruct %v4float %xy_7 %p2_zw
OpStore %out_pos_ptr %pos_7
OpStore %out_col_ptr %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
