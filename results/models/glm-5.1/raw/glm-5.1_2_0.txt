```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace ChinesePostman
{
    class Program
    {
        // Graph data
        static int N, M;
        static List<int[]>[] Adj;
        static int[] Degree;
        static long[,] ShortestDist;
        static int[,] ShortestNext;
        static List<int> OddNodes;

        // Hierholzer state
        static int[] EdgePtr;
        static List<int> EulerPath;

        static void Main()
        {
            ReadInput();
            var oddNodes = FindOddDegreeNodes();
            if (oddNodes.Count > 0)
            {
                FloydWarshallAllPairs();
                var matchings = SolveMinWeightPerfectMatching(oddNodes);
                AddMatchedEdges(matchings);
            }
            FindEulerTour();
            PrintOutput();
        }

        static void ReadInput()
        {
            var tokens = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            N = int.Parse(tokens[0]);
            M = int.Parse(tokens[1]);

            Adj = new List<int[]>[N];
            for (int i = 0; i < N; i++) Adj[i] = new List<int[]>();
            Degree = new int[N];

            for (int i = 0; i < M; i++)
            {
                tokens = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                int u = int.Parse(tokens[0]);
                int v = int.Parse(tokens[1]);
                int w = int.Parse(tokens[2]);

                Adj[u].Add(new int[] { v, w });
                Adj[v].Add(new int[] { u, w });
                Degree[u]++;
                Degree[v]++;
            }
        }

        static List<int> FindOddDegreeNodes()
        {
            OddNodes = new List<int>();
            for (int i = 0; i < N; i++)
            {
                if ((Degree[i] & 1) != 0)
                {
                    OddNodes.Add(i);
                }
            }
            return OddNodes;
        }

        static void FloydWarshallAllPairs()
        {
            ShortestDist = new long[N, N];
            ShortestNext = new int[N, N];
            long INF = 4_000_000_000L;

            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    ShortestDist[i, j] = (i == j) ? 0 : INF;
                    ShortestNext[i, j] = -1;
                }
            }

            for (int u = 0; u < N; u++)
            {
                foreach (var edge in Adj[u])
                {
                    int v = edge[0];
                    int w = edge[1];
                    if (w < ShortestDist[u, v])
                    {
                        ShortestDist[u, v] = w;
                        ShortestNext[u, v] = v;
                    }
                }
            }

            for (int k = 0; k < N; k++)
            {
                long dkj, dki;
                int skj;
                for (int i = 0; i < N; i++)
                {
                    dki = ShortestDist[i, k];
                    if (dki == INF) continue;
                    var row_i = i; // local copy helps range check elimination
                    for (int j = 0; j < N; j++)
                    {
                        dkj = ShortestDist[k, j];
                        if (dkj == INF) continue;
                        long newDist = dki + dkj;
                        if (newDist < ShortestDist[row_i, j])
                        {
                            ShortestDist[row_i, j] = newDist;
                            ShortestNext[row_i, j] = ShortestNext[row_i, k];
                        }
                    }
                }
            }
        }

        static List<(int u, int v)> SolveMinWeightPerfectMatching(List<int> oddNodes)
        {
            int K = oddNodes.Count;
            int n = K / 2;

            if (K == 0) return new List<(int, int)>();

            // DP Bitmask for K <= 22
            if (K <= 22)
            {
                int maxState = 1 << K;
                long[] dp = new long[maxState];
                int[] dpPrev = new int[maxState];
                long INF = 4_000_000_000L;

                for (int i = 0; i < maxState; i++) dp[i] = INF;
                dp[0] = 0;

                for (int mask = 0; mask < maxState; mask++)
                {
                    if (dp[mask] == INF) continue;
                    int i = 0;
                    while (i < K && (mask & (1 << i)) != 0) i++;
                    if (i >= K) continue;

                    int newMaskBase = mask | (1 << i);
                    for (int j = i + 1; j < K; j++)
                    {
                        if ((mask & (1 << j)) != 0) continue;
                        int newMask = newMaskBase | (1 << j);
                        long newCost = dp[mask] + ShortestDist[oddNodes[i], oddNodes[j]];
                        if (newCost < dp[newMask])
                        {
                            dp[newMask] = newCost;
                            dpPrev[newMask] = i * K + j;
                        }
                    }
                }

                int currMask = maxState - 1;
                var result = new List<(int, int)>(n);
                while (currMask > 0)
                {
                    int ij = dpPrev[currMask];
                    int i = ij / K;
                    int j = ij % K;
                    result.Add((oddNodes[i], oddNodes[j]));
                    currMask ^= (1 << i) | (1 << j);
                }
                return result;
            }

            // Heuristic Greedy Matching for K > 22
            var matched = new bool[K];
            var greedyResult = new List<(int, int)>(n);
            for (int iter = 0; iter < K / 2; iter++)
            {
                long minCost = long.MaxValue;
                int bestI = -1, bestJ = -1;
                int firstUnmatched = -1;
                for (int i = 0; i < K; i++)
                {
                    if (!matched[i]) { firstUnmatched = i; break; }
                }

                for (int j = firstUnmatched + 1; j < K; j++)
                {
                    if (matched[j]) continue;
                    long cost = ShortestDist[oddNodes[firstUnmatched], oddNodes[j]];
                    if (cost < minCost)
                    {
                        minCost = cost;
                        bestI = firstUnmatched;
                        bestJ = j;
                    }
                }

                matched[bestI] = true;
                matched[bestJ] = true;
                greedyResult.Add((oddNodes[bestI], oddNodes[bestJ]));
            }
            return greedyResult;
        }

        static void AddMatchedEdges(List<(int u, int v)> matchings)
        {
            foreach (var (u, v) in matchings)
            {
                int curr = u;
                while (curr != v)
                {
                    int next = ShortestNext[curr, v];
                    int w = (int)ShortestDist[curr, next];

                    Adj[curr].Add(new int[] { next, w });
                    Adj[next].Add(new int[] { curr, w });

                    Degree[curr]++;
                    Degree[next]++;

                    curr = next;
                }
            }
        }

        static void FindEulerTour()
        {
            EulerPath = new List<int>();
            EdgePtr = new int[N];

            var stack = new Stack<int>();
            stack.Push(0);

            while (stack.Count > 0)
            {
                int u = stack.Peek();
                if (EdgePtr[u] < Adj[u].Count)
                {
                    var edge = Adj[u][EdgePtr[u]];
                    int v = edge[0];
                    EdgePtr[u]++;
                    stack.Push(v);
                }
                else
                {
                    EulerPath.Add(u);
                    stack.Pop();
                }
            }
        }

        static void PrintOutput()
        {
            Console.WriteLine(EulerPath.Count);
            Console.WriteLine(string.Join(" ", EulerPath));
        }
    }
}
```
