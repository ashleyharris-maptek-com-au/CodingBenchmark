use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    if n == 0 {
        return;
    }

    let mut adj = vec![vec![]; n];
    let mut edge_list = Vec::with_capacity(m);

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut iter = line.split_whitespace();
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
            edge_list.push((u, v));
        }
    }

    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    let mut pos = vec![[0.0f64; 2]; n];
    let mut component_id = vec![0usize; n];
    let mut components: Vec<Vec<usize>> = Vec::new();

    // 1. Find Connected Components
    {
        let mut visited = vec![false; n];
        let mut queue = std::collections::VecDeque::new();
        let mut cid = 0;
        for i in 0..n {
            if !visited[i] {
                let mut comp = Vec::new();
                visited[i] = true;
                queue.push_back(i);
                while let Some(u) = queue.pop_front() {
                    comp.push(u);
                    component_id[u] = cid;
                    for &v in &adj[u] {
                        if !visited[v] {
                            visited[v] = true;
                            queue.push_back(v);
                        }
                    }
                }
                components.push(comp);
                cid += 1;
            }
        }
    }

    // 2. Layout each component
    for (cid, comp) in components.iter().enumerate() {
        let cn = comp.len();
        if cn == 1 {
            pos[comp[0]] = [0.0, 0.0];
            continue;
        }

        let mut c_deg = Vec::with_capacity(cn);
        let mut new_index = vec![0usize; n];
        for (i, &u) in comp.iter().enumerate() {
            c_deg.push(adj[u].len());
            new_index[u] = i;
        }

        let mut c_adj = vec![vec![]; cn];
        for &u in comp {
            let ci = new_index[u];
            for &v in &adj[u] {
                c_adj[ci].push(new_index[v]);
            }
        }

        // Centrality approximation for initial ordering
        let mut centrality = vec![0.0f64; cn];
        if cn <= 3000 {
            let mut bfs_dist = vec![usize::MAX; cn];
            for s in 0..cn {
                let mut q = std::collections::VecDeque::new();
                bfs_dist[s] = 0;
                q.push_back(s);
                let mut sum = 0;
                let mut cnt = 0;
                while let Some(u) = q.pop_front() {
                    sum += bfs_dist[u];
                    cnt += 1;
                    for &v in &c_adj[u] {
                        if bfs_dist[v] == usize::MAX {
                            bfs_dist[v] = bfs_dist[u] + 1;
                            q.push_back(v);
                        }
                    }
                }
                centrality[s] = if sum > 0 { cnt as f64 / sum as f64 } else { 0.0 };
                for &v in &comp { bfs_dist[new_index[v]] = usize::MAX; }
            }
        } else {
            for i in 0..cn {
                centrality[i] = c_deg[i] as f64;
            }
        }

        // Sort nodes by centrality (highest first)
        let mut order: Vec<usize> = (0..cn).collect();
        order.sort_unstable_by(|&a, &b| centrality[b].partial_cmp(&centrality[a]).unwrap_or(std::cmp::Ordering::Equal));

        let is_planar = cn <= 2000 && m < (cn as f64 * 3.3).min(cn * (cn - 1) as f64 / 2) as usize;

        let c_pos = if cn <= 10 || is_planar {
            layout_spring(&c_adj, &c_deg, cn, 800)
        } else if cn <= 250 {
            layout_sgd(&c_adj, &c_deg, cn, 200)
        } else {
            layout_multilevel(&c_adj, &c_deg, cn, &order)
        };

        for (i, &u) in comp.iter().enumerate() {
            pos[u] = c_pos[i];
        }

        // Scale component based on its size to avoid overlaps
        let scale = (cn as f64).sqrt().max(2.0);
        for &u in comp {
            pos[u][0] *= scale;
            pos[u][1] *= scale;
        }
    }

    // 3. Arrange disconnected components in a grid
    if components.len() > 1 {
        let mut comp_bounds = Vec::with_capacity(components.len());
        for comp in &components {
            if comp.is_empty() {
                comp_bounds.push([0.0, 0.0, 0.0, 0.0]);
                continue;
            }
            let mut min_x = f64::MAX;
            let mut min_y = f64::MAX;
            let mut max_x = f64::MIN;
            let mut max_y = f64::MIN;
            for &u in comp {
                min_x = min_x.min(pos[u][0]);
                min_y = min_y.min(pos[u][1]);
                max_x = max_x.max(pos[u][0]);
                max_y = max_y.max(pos[u][1]);
            }
            comp_bounds.push([min_x, min_y, max_x, max_y]);
        }

        let cols = (components.len() as f64).sqrt().ceil() as usize;
        let mut row_x = 0.0;
        let mut row_y = 0.0;
        let mut max_h = 0.0;
        let spacing = 20.0;

        for (i, comp) in components.iter().enumerate() {
            let col = i % cols;
            if col == 0 && i > 0 {
                row_y += max_h + spacing;
                row_x = 0.0;
                max_h = 0.0;
            }
            let [min_x, min_y, max_x, max_y] = comp_bounds[i];
            let dx = row_x - min_x;
            let dy = row_y - min_y;
            for &u in comp {
                pos[u][0] += dx;
                pos[u][1] += dy;
            }
            let w = max_x - min_x;
            let h = max_y - min_y;
            row_x += w + spacing;
            max_h = max_h.max(h);
        }
    }

    // 4. Output
    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", pos[i][0], pos[i][1]).unwrap();
    }
}

fn layout_spring(c_adj: &[Vec<usize>], c_deg: &[usize], cn: usize, iters: usize) -> Vec<[f64; 2]> {
    let mut pos = vec![[0.0f64; 2]; cn];
    let mut rng = Rng::new(cn as u64);

    let area = (cn as f64).max(1.0);
    let k = area.sqrt();

    for p in pos.iter_mut() {
        p[0] = rng.gen() * area;
        p[1] = rng.gen() * area;
    }

    let mut disp = vec![[0.0f64; 2]; cn];
    let mut temp = area.sqrt() / 2.0;
    let cooling = temp / (iters as f64 + 1.0);

    for _ in 0..iters {
        for d in disp.iter_mut() { d[0] = 0.0; d[1] = 0.0; }
        
        for u in 0..cn {
            for v in (u + 1)..cn {
                let dx = pos[u][0] - pos[v][0];
                let dy = pos[u][1] - pos[v][1];
                let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                let force = (k * k) / dist;
                let fx = (dx / dist) * force;
                let fy = (dy / dist) * force;
                disp[u][0] += fx; disp[u][1] += fy;
                disp[v][0] -= fx; disp[v][1] -= fy;
            }
            for &v in &c_adj[u] {
                if v > u {
                    let dx = pos[u][0] - pos[v][0];
                    let dy = pos[u][1] - pos[v][1];
                    let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                    let force = (dist * dist) / k;
                    let fx = (dx / dist) * force;
                    let fy = (dy / dist) * force;
                    disp[u][0] += fx; disp[u][1] += fy;
                    disp[v][0] -= fx; disp[v][1] -= fy;
                }
            }
        }

        for u in 0..cn {
            let dist = disp[u][0].hypot(disp[u][1]).max(0.01);
            let lim = dist.min(temp);
            pos[u][0] += (disp[u][0] / dist) * lim;
            pos[u][1] += (disp[u][1] / dist) * lim;
        }
        temp -= cooling;
        if temp <= 0.0 { break; }
    }
    pos
}

fn layout_sgd(c_adj: &[Vec<usize>], c_deg: &[usize], cn: usize, iters: usize) -> Vec<[f64; 2]> {
    let mut pos = vec![[0.0f64; 2]; cn];
    let mut rng = Rng::new(cn as u64);

    let mut dist = vec![u16::MAX; cn * cn];
    for s in 0..cn {
        dist[s * cn + s] = 0;
        let mut q = std::collections::VecDeque::new();
        q.push_back(s);
        while let Some(u) = q.pop_front() {
            let du = dist[s * cn + u] as usize;
            for &v in &c_adj[u] {
                if dist[s * cn + v] == u16::MAX {
                    dist[s * cn + v] = (du + 1) as u16;
                    q.push_back(v);
                }
            }
        }
    }

    let mut weights = vec![0.0f64; cn * cn];
    let mut eps = 0.1;
    for u in 0..cn {
        for v in 0..cn {
            let d = dist[u * cn + v];
            if d != u16::MAX && d > 0 {
                weights[u * cn + v] = 1.0 / (d as f64 * d as f64);
            } else {
                weights[u * cn + v] = eps;
            }
        }
    }

    let area = cn as f64;
    let k = area.sqrt();
    for u in 0..cn {
        pos[u][0] = rng.gen() * area;
        pos[u][1] = rng.gen() * area;
    }

    let eta0 = k / 2.0;
    let mut eta = eta0;
    let eta_min = 0.01;
    let decay = 0.95;

    for _ in 0..iters {
        let mut order: Vec<usize> = (0..cn * cn).collect();
        order.sort_unstable_by(|&a, &b| weights[a].partial_cmp(&weights[b]).unwrap_or(std::cmp::Ordering::Equal));

        for &idx in &order {
            let u = idx / cn;
            let v = idx % cn;
            if u == v { continue; }
            
            let dx = pos[u][0] - pos[v][0];
            let dy = pos[u][1] - pos[v][1];
            let mu = (dx * dx + dy * dy).sqrt().max(0.01);
            
            let w = weights[idx];
            let d = dist[idx];
            let ideal = if d != u16::MAX { d as f64 * k } else { k * 3.0 };
            
            let delta = (mu - ideal) / mu;
            let step = eta * delta * w.max(0.01);
            
            let rx = (dx / mu) * step;
            let ry = (dy / mu) * step;
            
            pos[u][0] -= rx;
            pos[u][1] -= ry;
        }
        eta = (eta * decay).max(eta_min);
    }
    pos
}

fn layout_multilevel(c_adj: &[Vec<usize>], c_deg: &[usize], cn: usize, order: &[usize]) -> Vec<[f64; 2]> {
    let max_coarse = 100;
    let mut levels = vec![c_adj.to_vec()];
    let mut mapping = Vec::new();
    let mut cur_adj = c_adj.to_vec();

    loop {
        let n = cur_adj.len();
        if n <= max_coarse { break; }
        
        let mut matched = vec![false; n];
        let mut new_idx = vec![usize::MAX; n];
        let mut coarser_adj = Vec::new();
        let mut cur_map = vec![Vec::new(); n];
        for i in 0..n { cur_map[i].push(i); }

        let mut num_new = 0;
        for &u in order {
            if u >= n || matched[u] { continue; }
            let mut best_v = usize::MAX;
            let mut best_w = -1.0f64;
            for &v in &cur_adj[u] {
                if !matched[v] {
                    let w = cur_adj[u].len().min(cur_adj[v].len()) as f64;
                    if w > best_w {
                        best_w = w;
                        best_v = v;
                    }
                }
            }
            new_idx[u] = num_new;
            matched[u] = true;
            if best_v != usize::MAX {
                new_idx[best_v] = num_new;
                matched[best_v] = true;
            }
            num_new += 1;
        }
        for i in 0..n {
            if !matched[i] {
                new_idx[i] = num_new;
                num_new += 1;
            }
        }

        let mut c_edges = vec![std::collections::HashSet::new(); num_new];
        for u in 0..n {
            let cu = new_idx[u];
            for &v in &cur_adj[u] {
                let cv = new_idx[v];
                if cu != cv {
                    c_edges[cu].insert(cv);
                }
            }
        }
        
        cur_adj = vec![Vec::new(); num_new];
        for i in 0..num_new {
            cur_adj[i] = c_edges[i].into_iter().collect();
        }

        let mut rev_map = vec![Vec::new(); num_new];
        for i in 0..n {
            rev_map[new_idx[i]].push(i);
        }
        
        mapping.push((new_idx, rev_map));
        levels.push(cur_adj.clone());
    }

    let coarse_n = cur_adj.len();
    let mut coarse_pos = vec![[0.0f64; 2]; coarse_n];
    let mut rng = Rng::new(cn as u64);
    for p in coarse_pos.iter_mut() {
        p[0] = rng.gen() * (coarse_n as f64).sqrt();
        p[1] = rng.gen() * (coarse_n as f64).sqrt();
    }

    // Perform SGD on the coarsest level to get a good initial layout
    let mut c_deg = vec![0; coarse_n];
    for i in 0..coarse_n { c_deg[i] = cur_adj[i].len(); }
    coarse_pos = layout_spring(&cur_adj, &c_deg, coarse_n, 200);

    let mut pos = coarse_pos;
    let mut cur_level = levels.len() - 1;

    while cur_level > 0 {
        cur_level -= 1;
        let (_, rev_map) = &mapping[cur_level];
        let fine_n = levels[cur_level].len();
        let mut new_pos = vec![[0.0f64; 2]; fine_n];

        for (c, nodes) in rev_map.iter().enumerate() {
            for &f in nodes {
                new_pos[f][0] = pos[c][0] + rng.gen() * 0.5 - 0.25;
                new_pos[f][1] = pos[c][1] + rng.gen() * 0.5 - 0.25;
            }
        }
        
        let fine_adj = &levels[cur_level];
        let mut f_deg = vec![0; fine_n];
        for i in 0..fine_n { f_deg[i] = fine_adj[i].len(); }
        
        let iters = if fine_n < 500 { 100 } else if fine_n < 2000 { 60 } else { 30 };
        pos = layout_spring(fine_adj, &f_deg, fine_n, iters);
    }

    pos
}

// Simple deterministic RNG
struct Rng { state: u64 }

impl Rng {
    fn new(seed: u64) -> Self {
        let mut r = Rng { state: seed ^ 0x1234567890ABCDEF };
        for _ in 0..10 { r.gen(); }
        r
    }
    fn gen(&mut self) -> f64 {
        self.state = self.state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        (self.state >> 33) as f64 / (1u64 << 31) as f64
    }
}
