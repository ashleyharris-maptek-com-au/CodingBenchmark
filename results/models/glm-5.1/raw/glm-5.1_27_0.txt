```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <cstring>

using namespace std;

// --- Configuration ---
static const int MAX_COLORS = 64;
static const int TIMEOUT_MS = 280000;
static const int ITER_LIMIT_NO_IMPROVE = 1000;

// --- Global State ---
int n, m, k;
vector<vector<int>> adj;
vector<int> initial_deg;

// Tabu Search Data
vector<int> color;
vector<vector<int>> freq;
int total_conflicts;
chrono::steady_clock::time_point start_time;

// Pseudo-random number generator
mt19937 rng(1337);

inline bool is_timeout() {
    auto now = chrono::steady_clock::now();
    return chrono::duration_cast<chrono::milliseconds>(now - start_time).count() > TIMEOUT_MS;
}

void compute_conflicts() {
    total_conflicts = 0;
    for (int u = 0; u < n; ++u) {
        for (int v : adj[u]) {
            if (color[u] == color[v]) {
                total_conflicts++;
            }
        }
    }
    total_conflicts /= 2;
}

void build_freq() {
    freq.assign(n, vector<int>(k, 0));
    for (int u = 0; u < n; ++u) {
        for (int v : adj[u]) {
            freq[u][color[v]]++;
        }
    }
}

bool tabu_search() {
    vector<int> tabu_list(n, 0);
    int iter = 0;
    int best_conflicts = total_conflicts;
    int iters_no_improve = 0;

    vector<int> best_cands;

    while (true) {
        if (total_conflicts == 0) return true;
        if (is_timeout()) return false;
        if (iters_no_improve > ITER_LIMIT_NO_IMPROVE) return false;

        iter++;
        iters_no_improve++;

        int min_delta = n;
        best_cands.clear();

        for (int u = 0; u < n; ++u) {
            if (freq[u][color[u]] == 0) continue;

            int old_c = color[u];
            for (int c = 0; c < k; ++c) {
                if (c == old_c) continue;

                int delta = freq[u][c] - freq[u][old_c];
                bool is_tabu = (tabu_list[u] > iter);

                if (delta < min_delta) {
                    if (!is_tabu || (total_conflicts + delta < best_conflicts)) {
                        min_delta = delta;
                        best_cands.clear();
                        best_cands.push_back((u << 6) | c);
                    }
                }
                else if (delta == min_delta && best_cands.size() < 32) {
                    if (!is_tabu || (total_conflicts + delta < best_conflicts)) {
                        best_cands.push_back((u << 6) | c);
                    }
                }
            }
        }

        if (!best_cands.empty()) {
            int choice = best_cands[rng() % best_cands.size()];
            int u = choice >> 6;
            int c = choice & 63;

            int old_c = color[u];
            color[u] = c;

            for (int v : adj[u]) {
                freq[v][old_c]--;
                freq[v][c]++;
            }

            total_conflicts += min_delta;
            tabu_list[u] = iter + rng() % 10 + 5;

            if (total_conflicts < best_conflicts) {
                best_conflicts = total_conflicts;
                iters_no_improve = 0;
            }
        }
        else {
            // Stuck, trigger perturbation
            for (int i = 0; i < max(1, n / 50); ++i) {
                int u = rng() % n;
                int c = rng() % k;
                int old_c = color[u];
                if (old_c == c) continue;

                color[u] = c;
                for (int v : adj[u]) {
                    freq[v][old_c]--;
                    freq[v][c]++;
                }
                int diff = freq[u][c] - freq[u][old_c];
                total_conflicts += diff;
                tabu_list[u] = iter + rng() % 20 + 10;
            }
        }
    }
}

void dsatur_color() {
    vector<int> sat(n, 0);
    vector<bool> used(k, false);
    vector<bool> vis(n, false);

    for (int i = 0; i < n; ++i) {
        int u = -1;
        int max_sat = -1, max_deg = -1;

        for (int v = 0; v < n; ++v) {
            if (vis[v]) continue;
            if (sat[v] > max_sat || (sat[v] == max_sat && initial_deg[v] > max_deg)) {
                max_sat = sat[v];
                max_deg = initial_deg[v];
                u = v;
            }
        }

        if (u == -1) break;
        vis[u] = true;

        for (int v : adj[u]) used[color[v]] = true;

        int c = 0;
        while (c < k && used[c]) c++;
        if (c == k) c = rng() % k;

        color[u] = c;
        fill(used.begin(), used.end(), false);

        for (int v : adj[u]) {
            if (color[v] == c) {
                sat[v]--;
            }
            if (color[v] == -1) {
                sat[v]++;
            }
        }
    }
}

void process_graph() {
    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);
    sort(verts.begin(), verts.end(), [](int a, int b) {
        return initial_deg[a] > initial_deg[b];
    });

    vector<vector<int>> radj(n);
    for (int u : verts) {
        for (int v : adj[u]) {
            radj[u].push_back(v);
        }
    }
    adj = move(radj);

    if (n <= 200) {
        for (int u = 0; u < n; ++u) {
            sort(adj[u].begin(), adj[u].end());
            adj[u].erase(unique(adj[u].begin(), adj[u].end()), adj[u].end());
        }
    }
    else {
        for (int u = 0; u < n; ++u) {
            if (adj[u].size() > 200) {
                sort(adj[u].begin(), adj[u].end());
                adj[u].erase(unique(adj[u].begin(), adj[u].end()), adj[u].end());
            }
        }
    }
}

void read_input() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    if (!(cin >> n >> m >> k)) return;

    adj.assign(n, {});
    initial_deg.assign(n, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
        initial_deg[u]++;
        initial_deg[v]++;
    }
}

void greedy_assign() {
    vector<bool> used(k, false);
    for (int u = 0; u < n; ++u) {
        for (int v : adj[u]) {
            if (color[v] != -1) used[color[v]] = true;
        }
        int c = 0;
        while (c < k && used[c]) c++;
        if (c == k) c = 0;
        color[u] = c;
        for (int v : adj[u]) {
            if (color[v] != -1) used[color[v]] = false;
        }
    }
}

void random_restart() {
    for (int i = 0; i < n; ++i) {
        color[i] = rng() % k;
    }
}

int main() {
    start_time = chrono::steady_clock::now();
    read_input();

    if (n == 0) return 0;
    if (k == 1) {
        bool has_edge = false;
        for (int i = 0; i < n; ++i) {
            if (!adj[i].empty()) {
                has_edge = true;
                break;
            }
        }
        if (has_edge) {
            for (int i = 0; i < n; ++i) cout << 0 << (i == n - 1 ? "" : " ");
            cout << "\n";
            return 1;
        }
        for (int i = 0; i < n; ++i) cout << 0 << (i == n - 1 ? "" : " ");
        cout << "\n";
        return 0;
    }

    process_graph();

    color.assign(n, -1);
    dsatur_color();
    build_freq();
    compute_conflicts();

    if (total_conflicts == 0) {
        for (int i = 0; i < n; ++i) cout << color[i] << (i == n - 1 ? "" : " ");
        cout << "\n";
        return 0;
    }

    if (tabu_search()) {
        for (int i = 0; i < n; ++i) cout << color[i] << (i == n - 1 ? "" : " ");
        cout << "\n";
        return 0;
    }

    while (!is_timeout()) {
        random_restart();
        build_freq();
        compute_conflicts();

        if (tabu_search()) {
            for (int i = 0; i < n; ++i) cout << color[i] << (i == n - 1 ? "" : " ");
            cout << "\n";
            return 0;
        }
    }

    greedy_assign();
    for (int i = 0; i < n; ++i) cout << color[i] << (i == n - 1 ? "" : " ");
    cout << "\n";
    return 1;
}
```
