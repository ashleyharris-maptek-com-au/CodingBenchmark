; SPIR-V

OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %UBO Binding 0

OpMemberDecorate %Matrices 0 ColMajor

OpMemberDecorate %Matrices 0 MatrixStride 16
OpMemberDecorate %Matrices 1 ColMajor
OpMemberDecorate %Matrices 1 Offset 64

OpMemberDecorate %Matrices 2 Offset 128

OpMemberDecorate %Matrices 5 Offset 216
OpMemberDecorate %Matrices 6 Offset 228
OpMemberDecorate %Matrices 7 Offset 232

%void = OpTypeVoid
%fn_main = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3

%uint = OpTypeInt 32 0

; Output variables

%outColor = OpVariable %ptr_out_v3 Output

; UBO
%Matrices = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v3float %v3float %v3float %float %int %int

%ptr_uniform_v3 = OpTypePointer Uniform %v3float

%main = OpFunction %void None %fn_main
%label = OpLabel

; Access UBO
%proj = OpAccessChain %ptr_uniform_mat %UBO %int_0
%load_proj = OpLoad %mat4v4float %proj

; Read input position and color
%in_pos_ptr = OpAccessChain %ptr_out_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr

%in_color_ptr = OpAccessChain %ptr_in_arr_v3 %inColor %int_0

; Calculate darker color for edges
%dark_color = OpVectorTimesScalar %v3float %in_color %float_0_6

; Project center position
%center_clip = OpMatrixTimesVector %v4float %load_proj %in_pos

; --- TRIANGLE 0 (Edge 0 to Edge 1) ---
; Emit center

OpStore %out_col_ptr0 %in_color
OpEmitVertex

%out_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr1 %e0_clip

%out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpEndPrimitive

OpStore %out_col_ptr6 %in_color

%out_pos_ptr7 = OpAccessChain %ptr_out_v4 %gl_out %int_0

%sin3 = OpCopyObject %float %float_0

%out_pos_ptr8 = OpAccessChain %ptr_out_v4 %gl_out %int_0

%out_col_ptr8 = OpAccessChain %ptr_out_v3 %outColor

; --- TRIANGLE 3 (Edge 3 to Edge 4) ---
%out_pos_ptr9 = OpAccessChain %ptr_out_v4 %gl_out %int_0
