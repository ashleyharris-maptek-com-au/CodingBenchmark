; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 101
; Schema: 0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%fun = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0_03 = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

OpDecorate %worldPos Location 0
%worldPos = OpVariable %ptr_in_arr_v3 Input

OpDecorate %normal Location 1
%normal = OpVariable %ptr_in_arr_v3 Input

OpDecorate %color Location 2
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

OpDecorate %outColor Location 0
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fun
%entry = OpLabel

; Load p0 and p1 (as vec2 clip space)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p0 = OpVectorShuffle %v2float %p0_v4 %p0_v4 0 1

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_v4 = OpLoad %v4float %p1_ptr
%p1 = OpVectorShuffle %v2float %p1_v4 %p1_v4 0 1

; Load c0 and c1
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute perpendicular vector
%dir = OpFSub %v2float %p1 %p0
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y
%perp_un = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%len = OpLength %float %perp_un
%inv_len = OpFDiv %float %float_0_03 %len
%perp = OpVectorTimesScalar %v2float %perp_un %inv_len

; Compute offsets
%p0_plus  = OpFAdd %v2float %p0 %perp
%p0_minus = OpFSub %v2float %p0 %perp
%p1_plus  = OpFAdd %v2float %p1 %perp
%p1_minus = OpFSub %v2float %p1 %perp

; Reconstruct vec4 positions (z=0.0, w=1.0)
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

%v0_xy = OpVectorShuffle %v4float %p0_plus %p0_plus 0 1 0 1
%v0_zw = OpCompositeConstruct %v4float %float_0 %float_1 %float_0 %float_1
%pos0 = OpVectorShuffle %v4float %v0_xy %v0_zw 0 1 4 5

%v1_xy = OpVectorShuffle %v4float %p0_minus %p0_minus 0 1 0 1
%pos1 = OpVectorShuffle %v4float %v1_xy %v0_zw 0 1 4 5

%v2_xy = OpVectorShuffle %v4float %p1_plus %p1_plus 0 1 0 1
%pos2 = OpVectorShuffle %v4float %v2_xy %v0_zw 0 1 4 5

%v3_xy = OpVectorShuffle %v4float %p1_minus %p1_minus 0 1 0 1
%pos3 = OpVectorShuffle %v4float %v3_xy %v0_zw 0 1 4 5

; Emit Vertex 0: p0 + perp, color c0
%out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %pos0
OpStore %outColor %c0
OpEmitVertex

; Emit Vertex 1: p0 - perp, color c0
%out_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr1 %pos1
OpStore %outColor %c0
OpEmitVertex

; Emit Vertex 2: p1 + perp, color c1
%out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr2 %pos2
OpStore %outColor %c1
OpEmitVertex

; Emit Vertex 3: p1 - perp, color c1
%out_pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr3 %pos3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
