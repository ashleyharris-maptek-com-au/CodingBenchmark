```spirv
; SPIR-V
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %functype
%label = OpLabel

; Read p0 and p1
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr

; Read c0 and c1
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; Compute dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0

; Compute dir.x and dir.y for perpendicular
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; Compute perp components: (-dir.y, dir.x)
%neg_dir_y = OpFNegate %float %dir_y
%perp_xy = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize perpendicular
%dot_perp = OpDot %float %perp_xy %perp_xy
%len_perp = OpExtInst %float %1 Sqrt %dot_perp
%inv_len = OpFDiv %float %float_1 %len_perp
%perp_norm_xy = OpVectorTimesScalar %v2float %perp_xy %inv_len

; Extend perpendicular to 3D (z = 0.0)
%perp_z = OpCompositeExtract %float %perp_norm_xy 0
%perp_y = OpCompositeExtract %float %perp_norm_xy 1
%perp = OpCompositeConstruct %v3float %perp_z %perp_y %float_0

; Compute mid = p0 + 0.75 * (p1 - p0)
%scaled_dir = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %scaled_dir

; --- SHAFT (4 vertices) ---
; Shaft Offset = perp * 0.015
%shaft_offset = OpVectorTimesScalar %v3float %perp %float_0_015

; Vertex 0: p0 + perp * 0.015
%s_v0_pos = OpFAdd %v3float %p0 %shaft_offset
%s_v0_pos4 = OpCompositeConstruct %v4float %s_v0_pos %float_1
%out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_ptr %s_v0_pos4
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0 - perp * 0.015
%s_v1_pos = OpFSub %v3float %p0 %shaft_offset
%s_v1_pos4 = OpCompositeConstruct %v4float %s_v1_pos %float_1
OpStore %out_ptr %s_v1_pos4
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: mid + perp * 0.015
%s_v2_pos = OpFAdd %v3float %mid %shaft_offset
%s_v2_pos4 = OpCompositeConstruct %v4float %s_v2_pos %float_1
OpStore %out_ptr %s_v2_pos4
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: mid - perp * 0.015
%s_v3_pos = OpFSub %v3float %mid %shaft_offset
%s_v3_pos4 = OpCompositeConstruct %v4float %s_v3_pos %float_1
OpStore %out_ptr %s_v3_pos4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; --- ARROWHEAD (3 vertices) ---
; Arrow Offset = perp * 0.05
%arrow_offset = OpVectorTimesScalar %v3float %perp %float_0_05

; Vertex 4: mid + perp * 0.05
%a_v4_pos = OpFAdd %v3float %mid %arrow_offset
%a_v4_pos4 = OpCompositeConstruct %v4float %a_v4_pos %float_1
OpStore %out_ptr %a_v4_pos4
OpStore %outColor %c1
OpEmitVertex

; Vertex 5: mid - perp * 0.05
%a_v5_pos = OpFSub %v3float %mid %arrow_offset
%a_v5_pos4 = OpCompositeConstruct %v4float %a_v5_pos %float_1
OpStore %out_ptr %a_v5_pos4
OpStore %outColor %c1
OpEmitVertex

; Vertex 6: p1 (tip)
%p1_pos4 = OpCompositeConstruct %v4float %p1 %float_1
OpStore %out_ptr %p1_pos4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
