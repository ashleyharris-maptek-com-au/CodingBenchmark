; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%fn_type = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%ubo_var = OpVariable %ptr_ubo Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32

%ambient_rgb = OpConstantComposite %vec3 %float_0_1 %float_0_05 %float_0_05
%specular_rgb = OpConstantComposite %vec3 %float_1 %float_1 %float_1
%float3_0 = OpConstantComposite %vec3 %float_0 %float_0 %float_0
%float3_1 = OpConstantComposite %vec3 %float_1 %float_1 %float_1

%main = OpFunction %void None %fn_type
%entry = OpLabel

%n_raw = OpLoad %vec3 %in_normal
%n = OpExtInst %vec3 %1 Normalize %n_raw
%frag_color = OpLoad %vec3 %in_color
%frag_pos = OpLoad %vec3 %in_worldPos

%ptr_lightPos = OpTypePointer Uniform %vec4
%lightPos_access = OpAccessChain %ptr_lightPos %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_access
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%ptr_cameraPos = OpTypePointer Uniform %vec4
%cameraPos_access = OpAccessChain %ptr_cameraPos %ubo_var %int_4
%cameraPos = OpLoad %vec4 %cameraPos_access
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

%L_unnorm = OpFSub %vec3 %lightPos_xyz %frag_pos
%dist = OpExtInst %float %1 Length %L_unnorm
%L = OpFDiv %vec3 %L_unnorm %dist

%V_unnorm = OpFSub %vec3 %cameraPos_xyz %frag_pos
%V = OpExtInst %vec3 %1 Normalize %V_unnorm

%negL = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %1 Reflect %negL %n

%NdotL = OpDot %float %n %L
%NdotL_clamped = OpExtInst %float %1 FMax %NdotL %float_0

%diffuse = OpVectorTimesScalar %vec3 %frag_color %NdotL_clamped

%RdotV = OpDot %float %R %V
%RdotV_clamped = OpExtInst %float %1 FMax %RdotV %float_0
%spec_factor = OpExtInst %float %1 Pow %RdotV_clamped %float_32
%specular = OpVectorTimesScalar %vec3 %specular_rgb %spec_factor

%sum = OpFAdd %vec3 %ambient_rgb %diffuse
%sum2 = OpFAdd %vec3 %sum %specular

%clamped = OpExtInst %vec3 %1 FMax %sum2 %float3_0
%final_rgb = OpExtInst %vec3 %1 FMin %clamped %float3_1

%final_rgba = OpCompositeConstruct %vec4 %final_rgb %float_1

OpStore %outColor %final_rgba

OpReturn
OpFunctionEnd
