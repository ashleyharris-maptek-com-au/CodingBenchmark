```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Box
{
    public int Id;
    public int W, H, D;
    public int Volume;
}

public class PackedBox
{
    public int Id, X, Y, Z, W, H, D;
}

public class Shelf
{
    public int StartH;
    public int EndH;
    public int AvailableDepth;
    public int StartX;
}

public class Program
{
    public static void Main()
    {
        Solve();
    }

    public static void Solve()
    {
        string firstLine = Console.ReadLine();
        while (string.IsNullOrWhiteSpace(firstLine)) firstLine = Console.ReadLine();
        string[] headerParts = firstLine.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(headerParts[0]);
        int W = int.Parse(headerParts[1]);
        int H = int.Parse(headerParts[2]);
        int D = int.Parse(headerParts[3]);

        Box[] boxes = new Box[N];
        long totalBoxVolume = 0;
        int maxBoxDim = 0;

        for (int i = 0; i < N; i++)
        {
            string line = Console.ReadLine();
            while (string.IsNullOrWhiteSpace(line)) line = Console.ReadLine();
            string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int w = int.Parse(parts[0]);
            int h = int.Parse(parts[1]);
            int d = int.Parse(parts[2]);
            boxes[i] = new Box { Id = i, W = w, H = h, D = d, Volume = w * h * d };
            totalBoxVolume += w * h * d;
            maxBoxDim = Math.Max(maxBoxDim, Math.Max(w, Math.Max(h, d)));
        }

        long containerVolume = (long)W * H * D;
        
        // If a single box is larger than the container in every dimension, it can never fit.
        // Filter them out early to save sorting and processing time.
        List<Box> validBoxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            if (boxes[i].W <= W || boxes[i].H <= H || boxes[i].D <= D)
            {
                validBoxes.Add(boxes[i]);
            }
        }
        boxes = validBoxes.ToArray();
        N = boxes.Length;

        if (N == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Sort by decreasing Base Area (W * D) as primary, decreasing Height as secondary.
        // This maximizes footprint usage per layer, yielding excellent packing density.
        Array.Sort(boxes, (a, b) =>
        {
            int areaA = a.W * a.D;
            int areaB = b.W * b.D;
            int cmp = areaB.CompareTo(areaA);
            if (cmp != 0) return cmp;
            cmp = b.H.CompareTo(a.H);
            if (cmp != 0) return cmp;
            return a.Id.CompareTo(b.Id); // Tie-breaker for stable sort
        });

        List<PackedBox> packed = new List<PackedBox>();
        List<Shelf> shelves = new List<Shelf>();
        long usedVolume = 0;
        long maxCapacity = (long)Math.Ceiling(containerVolume * 0.98); // slight margin for fragmentation
        int curShelfIdx = 0;
        int curY = 0;
        
        // Initialize the first shelf spanning the entire container width
        shelves.Add(new Shelf { StartH = 0, EndH = H, AvailableDepth = D, StartX = 0 });

        for (int i = 0; i < N; i++)
        {
            if (usedVolume + boxes[i].Volume > maxCapacity) break;

            int bW = boxes[i].W;
            int bH = boxes[i].H;
            int bD = boxes[i].D;

            int bestFitIdx = -1;
            int bestFitLeftover = int.MaxValue;

            // Find the best shelf for the current box (First Fit Decreasing with Best Fit heuristic)
            for (int s = 0; s < shelves.Count; s++)
            {
                Shelf shelf = shelves[s];
                int shelfAvailH = shelf.EndH - shelf.StartH;
                if (shelfAvailH < bH) continue; // Box doesn't fit vertically
                if (shelf.AvailableDepth < bD) continue; // Box doesn't fit depth-wise
                if (W - shelf.StartX < bW) continue; // Box doesn't fit width-wise

                int leftover = shelfAvailH - bH;
                if (leftover < bestFitLeftover)
                {
                    bestFitLeftover = leftover;
                    bestFitIdx = s;
                    if (leftover == 0) break; // Perfect height fit, can't do better
                }
            }

            // If no existing shelf can fit the box, create a new one
            if (bestFitIdx == -1)
            {
                // Ensure there is enough vertical space left in the container
                if (bH > H - curY)
                {
                    continue; // Cannot fit this box, skip to next
                }

                int newEndH;
                // Dynamic shelf height: If boxes are small relative to container, make thinner slices
                if (bH <= H / 4 && H > 20)
                {
                    newEndH = curY + Math.Max(bH, H / 8);
                    newEndH = Math.Min(newEndH, H); // Clamp to container bounds
                }
                else
                {
                    newEndH = H; // Fill the rest of the container height
                }

                shelves.Add(new Shelf
                {
                    StartH = curY,
                    EndH = newEndH,
                    AvailableDepth = D,
                    StartX = 0
                });
                bestFitIdx = shelves.Count - 1;
                curY = newEndH; // Next new shelf will start from here
            }

            Shelf bestShelf = shelves[bestFitIdx];

            // Place the box
            packed.Add(new PackedBox
            {
                Id = boxes[i].Id,
                X = bestShelf.StartX,
                Y = bestShelf.StartH,
                Z = 0,
                W = bW,
                H = bH,
                D = bD
            });

            // Update the shelf's width occupancy and remaining depth
            bestShelf.StartX += bW;
            if (bD < bestShelf.AvailableDepth)
            {
                bestShelf.AvailableDepth = bD;
            }

            usedVolume += boxes[i].Volume;
        }

        // Output results
        Console.WriteLine(packed.Count);
        var output = new System.Text.StringBuilder();
        for (int i = 0; i < packed.Count; i++)
        {
            output.Append(packed[i].Id);
            output.Append(' ');
            output.Append(packed[i].X);
            output.Append(' ');
            output.Append(packed[i].Y);
            output.Append(' ');
            output.Append(packed[i].Z);
            output.AppendLine();
        }
        Console.Write(output.ToString());
    }
}
```
