; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%fn_main = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v2 = OpTypePointer Input %v2float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_uni_mat = OpTypePointer Uniform %mat4v4float
%ptr_uni_v4 = OpTypePointer Uniform %v4float

%v2float = OpTypeVector %float 2

%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_uni_ubo = OpTypePointer Uniform %UBO

%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_uv = OpVariable %ptr_in_v2 Input
%in_tangent = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input
%outColor = OpVariable %ptr_out_v4 Output
%ubo_var = OpVariable %ptr_uni_ubo Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_75 = OpConstant %float 0.75
%float_0_5 = OpConstant %float 0.5
%float_0_25 = OpConstant %float 0.25
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_0_2 = OpConstant %float 0.2

%bool = OpTypeBool

%main = OpFunction %void None %fn_main
%entry = OpLabel

%normal_load = OpLoad %v3float %in_normal
%N = OpExtInst %v3float %1 Normalize %normal_load

%light_pos_ptr = OpAccessChain %ptr_uni_v4 %ubo_var %int_3
%light_pos = OpLoad %v4float %light_pos_ptr
%light_pos_v3 = OpVectorShuffle %v3float %light_pos %light_pos 0 1 2

%world_pos_load = OpLoad %v3float %in_worldPos

%L_vec = OpFSub %v3float %light_pos_v3 %world_pos_load
%L = OpExtInst %v3float %1 Normalize %L_vec

%NdL = OpDot %float %N %L
%d = OpFMax %float %NdL %float_0

%cond1 = OpFOrdGreaterThan %bool %d %float_0_75
%val1 = OpSelect %float %cond1 %float_1 %float_0_7

%cond2 = OpFOrdGreaterThan %bool %d %float_0_5
%val2 = OpSelect %float %cond2 %val1 %float_0_4

%cond3 = OpFOrdGreaterThan %bool %d %float_0_25
%band = OpSelect %float %cond3 %val2 %float_0_2

%color_load = OpLoad %v3float %in_color
%result_rgb = OpVectorTimesScalar %v3float %color_load %band

%final_rgba = OpCompositeConstruct %v4float %result_rgb %float_1

OpStore %outColor %final_rgba

OpReturn
OpFunctionEnd
