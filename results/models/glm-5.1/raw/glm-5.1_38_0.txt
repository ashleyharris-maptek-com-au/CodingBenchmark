import sys

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    U = int(input_data[0])
    S = int(input_data[1])
    
    sets = []
    idx = 2
    for _ in range(S):
        sets.append(set(input_data[idx:idx+U])) # Using U as upper bound for slice, no harm if smaller
        idx += len(sets[-1])
        
    # We use Donald Knuth's Algorithm X with dancing links to solve Exact Cover efficiently.
    # Given the strict requirement that the output must EXACTLY match the planted solution,
    # standard Algorithm X will deterministically find the same unique solution as planted.
    
    # Header and element nodes
    class Node:
        __slots__ = ['L', 'R', 'U', 'D', 'C', 'row']
        def __init__(self):
            self.L = self.R = self.U = self.D = self
            self.C = None
            self.row = -1

    # Initialize column headers for elements 0 to U-1
    header = Node()
    column_nodes = [Node() for _ in range(U)]
    
    # Link column nodes together horizontally
    prev = header
    for j in range(U):
        curr = column_nodes[j]
        curr.C = curr  # Column node points to itself as the column header
        curr.U = curr
        curr.D = curr
        curr.L = prev
        prev.R = curr
        prev = curr
    prev.R = header
    header.L = prev
    
    # To track column sizes for the heuristic (choose column with fewest ones)
    size = [0] * U
    
    # Add rows to the dancing links structure
    for i, s in enumerate(sets):
        # We must insert elements in sorted order to ensure determinism matches the planted solution's generation
        elements = sorted(int(x) for x in s)
        first_node = None
        for e in elements:
            col = column_nodes[e]
            node = Node()
            node.C = col
            node.row = i
            
            # Link vertically (insert above the column header)
            node.U = col.U
            node.D = col
            col.U.D = node
            col.U = node
            
            # Link horizontally
            if first_node is None:
                first_node = node
                node.L = node
                node.R = node
            else:
                node.L = first_node.L
                node.R = first_node
                first_node.L.R = node
                first_node.L = node
                
            size[e] += 1

    # Cover a column
    def cover(col):
        col.R.L = col.L
        col.L.R = col.R
        i = col.D
        while i != col:
            j = i.R
            while j != i:
                j.D.U = j.U
                j.U.D = j.D
                size[j.C.C - column_nodes[0]] -= 1
                j = j.R
            i = i.D

    # Uncover a column
    def uncover(col):
        i = col.U
        while i != col:
            j = i.L
            while j != i:
                size[j.C.C - column_nodes[0]] += 1
                j.D.U = j
                j.U.D = j
                j = j.L
            i = i.U
        col.R.L = col
        col.L.R = col

    solution = []
    
    # Recursive search function (Algorithm X)
    def search():
        # If header.R == header, all columns are covered -> solution found
        if header.R == header:
            return True
            
        # Choose column with the smallest size (S-heuristic)
        min_size = float('inf')
        chosen_col = None
        curr = header.R
        while curr != header:
            c_idx = curr.C - column_nodes[0]
            if size[c_idx] < min_size:
                min_size = size[c_idx]
                chosen_col = curr
                if min_size <= 1:
                    break
            curr = curr.R
            
        # If chosen column has no rows, this path fails
        if min_size == 0:
            return False
            
        cover(chosen_col)
        
        # Try each row in the chosen column
        r = chosen_col.D
        while r != chosen_col:
            solution.append(r.row)
            
            # Cover all columns intersecting this row
            j = r.R
            while j != r:
                cover(j.C)
                j = j.R
                
            if search():
                return True
                
            # Backtrack
            solution.pop()
            j = r.L
            while j != r:
                uncover(j.C)
                j = j.L
                
            r = r.D
            
        uncover(chosen_col)
        return False

    # Execute and output the exact matched solution
    if search():
        print("SOLUTION")
        print(" ".join(map(str, sorted(solution))))
    else:
        print("NO SOLUTION")

if __name__ == '__main__':
    solve()
