; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel

%N_raw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %N_raw

%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%worldPos = OpLoad %vec3 %in_worldPos
%L = OpExtInst %vec3 %1 Normalize %lightPos_xyz

%NdL = OpDot %float %N %L
%NdL_clamped = OpExtInst %float %1 FClamp %NdL %float_0 %float_1

%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

%V_unnorm = OpFSub %vec3 %cameraPos_xyz %worldPos
%V = OpExtInst %vec3 %1 Normalize %V_unnorm

%NdV = OpDot %float %N %V
%NdV_clamped = OpExtInst %float %1 FClamp %NdV %float_0 %float_1
%one_minus_NdV = OpFSub %float %float_1 %NdV_clamped
%rim = OpExtInst %float %1 Pow %one_minus_NdV %float_3

%diffuse_mod = OpCompositeConstruct %vec3 %NdL_clamped %NdL_clamped %NdL_clamped
%base_mod_val = OpCompositeConstruct %vec3 %float_0_3 %float_0_3 %float_0_3
%scale_val = OpCompositeConstruct %vec3 %float_0_5 %float_0_5 %float_0_5
%scaled_diff = OpFMul %vec3 %scale_val %diffuse_mod
%light_mod = OpFAdd %vec3 %base_mod_val %scaled_diff

%base_color = OpCompositeConstruct %vec3 %float_0_2 %float_0_3 %float_0_8
%lit_base = OpFMul %vec3 %base_color %light_mod

%rim_color = OpCompositeConstruct %vec3 %float_1 %float_0_8 %float_0_5
%rim_val_vec = OpCompositeConstruct %vec3 %rim %rim %rim
%rim_contrib = OpFMul %vec3 %rim_color %rim_val_vec

%final_color = OpFAdd %vec3 %lit_base %rim_contrib
%out_vec = OpCompositeConstruct %vec4 %final_color %float_1

OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
