```spirv
; SPIR-V
Version: 1.0
Generator: 0
Bound: 150
Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %vertColor %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %vertColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%float_0_08 = OpConstant %float 0x1.47ae14p-4 ; 0.08
%float_0 = OpConstant %float 0.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_3

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%vertColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %functype
%entry = OpLabel

; Load gl_in[0].gl_Position, gl_in[1].gl_Position, gl_in[2].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

; Extract xyz components
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute face center: center = (p0 + p1 + p2) / 3
%sum_01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum_012 = OpFAdd %v3float %sum_01 %p2_xyz
%center = OpCompositeConstruct %v3float %float_0_08 %float_0_08 %float_0_08 ; temporarily borrow 0.08 for div trick
%center_scaled = OpFMul %v3float %sum_012 %center ; multiply by 0.08 (approx 1/12.5)

; Actually 1/3 is ~0.333, let's manually do 1/3 using multiply for precision:
%one_third = OpConstant %float 0x1.555556p-2 ; 1/3
%third_vec = OpCompositeConstruct %v3float %one_third %one_third %one_third
%center_correct = OpFMul %v3float %sum_012 %third_vec

; Compute edges: e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; Compute cross product: N = cross(e1, e2)
; e1 = (a, b, c), e2 = (d, e, f)
; N = (bf-ce, cd-af, ae-bd)
%e1_x = OpCompositeExtract %float %e1 0
%e1_y = OpCompositeExtract %float %e1 1
%e1_z = OpCompositeExtract %float %e1 2
%e2_x = OpCompositeExtract %float %e2 0
%e2_y = OpCompositeExtract %float %e2 1
%e2_z = OpCompositeExtract %float %e2 2

%bf = OpFMul %float %e1_y %e2_z
%ce = OpFMul %float %e1_z %e2_y
%N_x = OpFSub %float %bf %ce

%cd = OpFMul %float %e1_z %e2_x
%af = OpFMul %float %e1_x %e2_z
%N_y = OpFSub %float %cd %af

%ae = OpFMul %float %e1_x %e2_y
%bd = OpFMul %float %e1_y %e2_x
%N_z = OpFSub %float %ae %bd

%N_raw = OpCompositeConstruct %v3float %N_x %N_y %N_z

; Normalize N manually: N / length(N)
; length(N) = sqrt(dot(N, N))
%dot_NN = OpDot %float %N_raw %N_raw
%len_N = OpExtInst %float %1 Sqrt %dot_NN
%inv_len_N = OpCompositeConstruct %v3float %len_N %len_N %len_N
%N = OpFDiv %v3float %N_raw %inv_len_N

; Offset = N * 0.08
%offset_scale = OpCompositeConstruct %v3float %float_0_08 %float_0_08 %float_0_08
%offset = OpFMul %v3float %N %offset_scale

; --- Emit Vertex 0 ---
%v0_color_ptr = OpAccessChain %ptr_in_v3 %vertColor %int_0
%c0 = OpLoad %v3float %v0_color_ptr
%p0_shifted = OpFAdd %v3float %p0_xyz %offset
%p0_new = OpCompositeConstruct %v4float %p0_shifted %float_0

%out_p0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_p0_ptr %p0_new
OpStore %outColor %c0
OpEmitVertex

; --- Emit Vertex 1 ---
%v1_color_ptr = OpAccessChain %ptr_in_v3 %vertColor %int_1
%c1 = OpLoad %v3float %v1_color_ptr
%p1_shifted = OpFAdd %v3float %p1_xyz %offset
%p1_new = OpCompositeConstruct %v4float %p1_shifted %float_0

%out_p1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_p1_ptr %p1_new
OpStore %outColor %c1
OpEmitVertex

; --- Emit Vertex 2 ---
%v2_color_ptr = OpAccessChain %ptr_in_v3 %vertColor %int_2
%c2 = OpLoad %v3float %v2_color_ptr
%p2_shifted = OpFAdd %v3float %p2_xyz %offset
%p2_new = OpCompositeConstruct %v4float %p2_shifted %float_0

%out_p2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_p2_ptr %p2_new
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
