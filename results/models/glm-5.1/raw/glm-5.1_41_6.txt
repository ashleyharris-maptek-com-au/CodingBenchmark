; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fn_main = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec4 = OpTypePointer Input %vec4
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec3 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_64 = OpConstant %float 64
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8

%main = OpFunction %void None %fn_main
%label = OpLabel

; Load Normal and normalize it
%normal_raw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %normal_raw

; Load Light Position (xyz)
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Load Camera Position (xyz)
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

; Load World Position
%worldPos = OpLoad %vec3 %in_worldPos

; Compute L = normalize(lightPos - worldPos)
%L_vec = OpFSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %1 Normalize %L_vec

; Compute V = normalize(cameraPos - worldPos)
%V_vec = OpFSub %vec3 %cameraPos_xyz %worldPos
%V = OpExtInst %vec3 %1 Normalize %V_vec

; Compute H = normalize(L + V)
%H_vec = OpFAdd %vec3 %L %V
%H = OpExtInst %vec3 %1 Normalize %H_vec

; Diffuse = max(dot(N, L), 0)
%NdotL = OpDot %float %N %L
%diffuse = OpFMax %float %NdotL %float_0

; Specular = pow(max(dot(N, H), 0), 64)
%NdotH = OpDot %float %N %H
%NdotH_clamped = OpFMax %float %NdotH %float_0
%specular = OpExtInst %float %1 Pow %NdotH_clamped %float_64

; Intensity = 0.1 + 0.6 * diffuse + 0.8 * specular
%term_diff = OpFMul %float %float_0_6 %diffuse
%term_spec = OpFMul %float %float_0_8 %specular
%intensity = OpFAdd %float %float_0_1 %term_diff
%intensity_final = OpFAdd %float %intensity %term_spec

; Output as vec4(intensity, intensity, intensity, 1.0)
%out_rgb = OpCompositeConstruct %vec3 %intensity_final %intensity_final %intensity_final
%out_rgba = OpCompositeConstruct %vec4 %out_rgb %float_1

OpStore %outColor %out_rgba

OpReturn
OpFunctionEnd
