```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocation
    if (i >= N) {
        return;
    }

    // Read current state for body i
    vec4 pos_mass_i = inData[i * 2];
    vec4 vel_i = inData[i * 2 + 1];

    vec3 pos_i = pos_mass_i.xyz;
    float mass_i = pos_mass_i.w;
    vec3 vel = vel_i.xyz;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Compute acceleration from all other bodies
    vec3 acc = vec3(0.0);
    float softening2 = softening * softening;

    for (uint j = 0; j < N; ++j) {
        // Skip self-interaction
        if (i == j) {
            continue;
        }

        vec4 pos_mass_j = inData[j * 2];
        vec3 pos_j = pos_mass_j.xyz;
        float mass_j = pos_mass_j.w;

        vec3 r = pos_j - pos_i;
        float dist2 = dot(r, r) + softening2;
        float inv_dist = 1.0 / sqrt(dist2);
        float inv_dist3 = inv_dist * inv_dist * inv_dist;

        acc += G * mass_j * r * inv_dist3;
    }

    // Advance velocity and position
    vel += acc * dt;
    vec3 pos = pos_i + vel * dt;

    // Write updated state to output
    outData[i * 2]     = vec4(pos, mass_i);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
