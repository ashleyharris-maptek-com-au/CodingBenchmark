; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fn_main = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_ubo Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_16 = OpConstant %float 16
%float_pi = OpConstant %float 3.14159

%main = OpFunction %void None %fn_main
%entry = OpLabel

; Load UV
%uv = OpLoad %vec2 %in_uv
%uv_y = OpCompositeExtract %float %uv 1

; Compute sin(uv.y * 3.14159 * 16.0)
%mul1 = OpFMul %float %uv_y %float_pi
%mul2 = OpFMul %float %mul1 %float_16
%sin_val = OpExtInst %float %1 Sin %mul2

; Determine stripe color based on sin sign
%cond = OpFOrdGreaterThan %bool %sin_val %float_0
%red_vec = OpCompositeConstruct %vec3 %float_0_9 %float_0_1 %float_0_1
%blue_vec = OpCompositeConstruct %vec3 %float_0_1 %float_0_1 %float_0_9
%stripe_color = OpSelect %vec3 %cond %red_vec %blue_vec

; Load normal and normalize
%normal_raw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %normal_raw

; Load light position (w=1.0)
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr

; Load world position
%worldPos = OpLoad %vec3 %in_worldPos

; Compute light direction (point light)
%diff_vec = OpSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %1 Normalize %diff_vec

; Compute Lambertian diffuse
%NdotL = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %NdotL %float_0

; Compute shading factor (0.4 + 0.6 * diffuse)
%term = OpFMul %float %float_0_6 %diffuse
%shading = OpFAdd %float %float_0_4 %term

; Final color
%shaded_color = OpVectorTimesScalar %vec3 %stripe_color %shading
%final_alpha = OpCompositeConstruct %vec4 %shaded_color %float_1
OpStore %outColor %final_alpha

OpReturn
OpFunctionEnd
