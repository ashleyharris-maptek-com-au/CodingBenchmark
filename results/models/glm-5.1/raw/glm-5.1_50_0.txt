```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds invocation
    if (idx >= N) {
        return;
    }

    // Read current position and velocity
    vec3 pos = inData[idx * 2].xyz;
    vec3 vel = inData[idx * 2 + 1].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Accumulators for flocking forces
    vec3 sepVec = vec3(0.0);
    vec3 alignVec = vec3(0.0);
    vec3 cohVec = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Loop over all other boids
    for (uint j = 0; j < N; ++j) {
        if (j == idx) continue; // Skip self

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        // Skip zero-distance neighbors to avoid NaNs
        if (dist <= 0.0) continue;

        // Separation: average push-away vector from neighbors
        if (dist < sepRadius) {
            sepVec += diff / dist; // Direction away, normalized magnitude
            sepCount++;
        }

        // Alignment: average velocity of neighbors
        if (dist < alignRadius) {
            alignVec += otherVel;
            alignCount++;
        }

        // Cohesion: average position of neighbors
        if (dist < cohRadius) {
            cohVec += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    // Average separation and apply weight
    if (sepCount > 0) {
        accel += (sepVec / float(sepCount)) * sepWeight;
    }

    // Average alignment (steer towards average velocity) and apply weight
    if (alignCount > 0) {
        vec3 avgAlign = alignVec / float(alignCount);
        accel += (avgAlign - vel) * alignWeight;
    }

    // Average cohesion (steer towards average position) and apply weight
    if (cohCount > 0) {
        vec3 avgCohPos = cohVec / float(cohCount);
        accel += (avgCohPos - pos) * cohWeight;
    }

    // Apply boundary force to keep boids contained
    float threshold = boundSize * 0.4;
    float scale = boundSize * 0.5;
    vec3 boundAccel = vec3(0.0);
    if (pos.x > threshold)  boundAccel.x -= boundForce * (pos.x - threshold) / scale;
    if (pos.x < -threshold) boundAccel.x += boundForce * (-threshold - pos.x) / scale;
    if (pos.y > threshold)  boundAccel.y -= boundForce * (pos.y - threshold) / scale;
    if (pos.y < -threshold) boundAccel.y += boundForce * (-threshold - pos.y) / scale;
    if (pos.z > threshold)  boundAccel.z -= boundForce * (pos.z - threshold) / scale;
    if (pos.z < -threshold) boundAccel.z += boundForce * (-threshold - pos.z) / scale;
    
    accel += boundAccel;

    // Update velocity by acceleration * dt
    vel += accel * dt;

    // Clamp velocity magnitude to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }

    // Update position by velocity * dt
    pos += vel * dt;

    // Write outputs maintaining the w components as 0.0
    outData[idx * 2] = vec4(pos, 0.0);
    outData[idx * 2 + 1] = vec4(vel, 0.0);
}
```
