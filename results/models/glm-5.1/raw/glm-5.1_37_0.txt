```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        var adj = new List<int>[n];
        var radj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            radj[v].Add(u);
        }

        var size = new int[n];
        var sizeToVerts = new List<int>[n + 1];
        for (int i = 0; i <= n; i++) sizeToVerts[i] = new List<int>();
        var inQ = new bool[n];
        var q = new Queue<int>();

        for (int i = 0; i < n; i++)
        {
            size[i] = Math.Max(adj[i].Count, radj[i].Count);
            sizeToVerts[size[i]].Add(i);
            if (size[i] <= 1)
            {
                q.Enqueue(i);
                inQ[i] = true;
            }
        }

        var removed = new bool[n];
        var order = new int[n];
        int p = n;
        int l = 0;

        while (q.Count > 0)
        {
            int u = q.Dequeue();
            order[p--] = u;
            removed[u] = true;

            foreach (int v in adj[u])
            {
                if (!removed[v])
                {
                    sizeToVerts[size[v]].Remove(v);
                    size[v]--;
                    sizeToVerts[size[v]].Add(v);
                    if (size[v] <= 1 && !inQ[v])
                    {
                        q.Enqueue(v);
                        inQ[v] = true;
                    }
                }
            }

            foreach (int v in radj[u])
            {
                if (!removed[v])
                {
                    sizeToVerts[size[v]].Remove(v);
                    size[v]--;
                    sizeToVerts[size[v]].Add(v);
                    if (size[v] <= 1 && !inQ[v])
                    {
                        q.Enqueue(v);
                        inQ[v] = true;
                    }
                }
            }
        }

        int w = p;

        if (w > 0)
        {
            int[] compId = new int[n];
            int nextCompId = 0;

            for (int i = w + 1; i <= n; i++)
                compId[order[i]] = -1;

            int[] stack = new int[w + 1];
            int[] sp = new int[w + 1];
            bool[] inStack = new bool[w + 1];
            int[] index = new int[w + 1];
            int[] lowlink = new int[w + 1];
            int idx = 0;

            int[] inS2 = new int[w + 1];
            var s2 = new Stack<int>();

            var gsucc = new List<int>[w + 1];
            var gpred = new List<int>[w + 1];
            var nodes = new List<int>[w + 1];
            var treeSucc = new List<int>[w + 1];
            var treePred = new List<int>[w + 1];
            var treeChild = new List<int>[w + 1];
            int[] treeParent = new int[w + 1];
            int[] treeDepth = new int[w + 1];
            int[] subtreeV = new int[w + 1];
            bool[] isTreeCut = new bool[w + 1];
            int[] isTreeCutR = new int[w + 1];

            for (int i = 1; i <= w; i++)
            {
                gsucc[i] = new List<int>();
                gpred[i] = new List<int>();
                nodes[i] = new List<int>();
                treeSucc[i] = new List<int>();
                treePred[i] = new List<int>();
                treeChild[i] = new List<int>();
                treeParent[i] = 0;
                treeDepth[i] = 0;
                index[i] = -1;
            }

            for (int i = 1; i <= w; i++)
            {
                int u = order[i];
                if (compId[u] != -1) continue;
                int stackTop = 0;
                stack[0] = i;
                sp[0] = 0;
                inStack[0] = true;
                int stackPtr = 0;

                while (stackPtr >= 0)
                {
                    int curr = stack[stackPtr];
                    int v = order[curr];
                    if (sp[curr] == 0)
                    {
                        index[curr] = lowlink[curr] = idx++;
                        s2.Push(curr);
                        inS2[curr] = 1;
                    }
                    bool done = true;
                    for (; sp[curr] < adj[v].Count; sp[curr]++)
                    {
                        int to = adj[v][sp[curr]];
                        if (compId[to] != -1) continue;
                        int j = Array.BinarySearch(order, w + 1, w - w, to);
                        if (j < 1 || j > w) continue;
                        if (index[j] == -1)
                        {
                            stackPtr++;
                            stack[stackPtr] = j;
                            sp[j] = 0;
                            inStack[stackPtr] = true;
                            done = false;
                            break;
                        }
                        else if (inS2[j] == 1)
                        {
                            lowlink[curr] = Math.Min(lowlink[curr], index[j]);
                        }
                    }
                    if (done)
                    {
                        if (lowlink[curr] == index[curr])
                        {
                            int cid = nextCompId++;
                            int top;
                            do
                            {
                                top = s2.Pop();
                                inS2[top] = 0;
                                compId[order[top]] = cid;
                                nodes[cid].Add(order[top]);
                            } while (top != curr);
                        }
                        if (stackPtr > 0)
                        {
                            lowlink[stack[stackPtr - 1]] = Math.Min(lowlink[stack[stackPtr - 1]], lowlink[curr]);
                        }
                        stackPtr--;
                    }
                }
            }

            for (int i = 1; i <= w; i++)
            {
                int u = order[i];
                foreach (int v in adj[u])
                {
                    if (compId[v] != -1) continue;
                    int j = Array.BinarySearch(order, w + 1, w - w, v);
                    if (j < 1 || j > w) continue;
                    int cu = compId[u], cv = compId[v];
                    if (cu != cv)
                    {
                        gsucc[cu].Add(cv);
                        gpred[cv].Add(cu);
                    }
                }
            }

            var topo = new int[nextCompId];
            int tIdx = 0;
            int[] inDeg = new int[nextCompId];
            var tQ = new Queue<int>();
            for (int i = 0; i < nextCompId; i++)
            {
                inDeg[i] = gpred[i].Count;
                if (inDeg[i] == 0) tQ.Enqueue(i);
            }
            while (tQ.Count > 0)
            {
                int u = tQ.Dequeue();
                topo[tIdx++] = u;
                foreach (int v in gsucc[u])
                {
                    inDeg[v]--;
                    if (inDeg[v] == 0) tQ.Enqueue(v);
                }
            }

            for (int i = 0; i < nextCompId; i++)
            {
                int u = topo[i];
                int maxD = -1, best = -1;
                foreach (int v in gpred[u])
                {
                    if (treeDepth[v] > maxD)
                    {
                        maxD = treeDepth[v];
                        best = v;
                    }
                }
                treeParent[u] = best;
                treeDepth[u] = maxD + 1;
                if (best != -1) treeChild[best].Add(u);
            }

            var leaves = new Stack<int>();
            for (int i = 0; i < nextCompId; i++)
                if (treeChild[i].Count == 0) leaves.Push(i);

            p = nextCompId;
            int[] remap = new int[nextCompId];
            while (leaves.Count > 0)
            {
                int u = leaves.Pop();
                remap[u] = p--;
                if (treeParent[u] != -1)
                {
                    treeChild[treeParent[u]].Remove(u);
                    if (treeChild[treeParent[u]].Count == 0)
                        leaves.Push(treeParent[u]);
                }
            }

            int[] hPred = new int[nextCompId];
            int[] hSucc = new int[nextCompId];
            for (int i = 0; i < nextCompId; i++)
            {
                int ri = remap[i];
                int maxP = -1, minS = nextCompId;
                foreach (int v in gpred[i])
                    if (remap[v] > maxP) maxP = remap[v];
                foreach (int v in gsucc[i])
                    if (remap[v] < minS) minS = remap[v];
                hPred[i] = maxP;
                hSucc[i] = minS;
                if (treeParent[i] != -1) isTreeCut[i] = true;
            }

            for (int i = nextCompId - 1; i >= 0; i--)
            {
                int u = topo[i];
                subtreeV[u] = nodes[u].Count;
                foreach (int v in gsucc[u])
                    if (isTreeCut[v] && treeParent[v] == u)
                        subtreeV[u] += subtreeV[v];
            }

            int[] rSubtreeV = new int[nextCompId];
            for (int i = 0; i < nextCompId; i++)
            {
                int u = topo[i];
                int rem = 0;
                foreach (int v in gpred[u])
                    if (isTreeCut[v] && treeParent[v] == u)
                        rem += subtreeV[v];
                rSubtreeV[u] = nodes[u].Count + rem;
            }

            for (int i = 1; i < nextCompId; i++)
            {
                int u = topo[i];
                if (hPred[u] > remap[treeParent[u]])
                    isTreeCutR[u] = 1;
            }

            for (int i = 0; i < nextCompId; i++)
            {
                int u = topo[i];
                int cSize = nodes[u].Count;
                int b1 = 0, b2 = 0;
                if (isTreeCut[u]) b1 = subtreeV[u];
                if (isTreeCutR[u] == 1) b2 = rSubtreeV[u];
                if (b1 + b2 > cSize)
                {
                    int sc = cSize - b2;
                    int rc = cSize - b1;
                    var cuts = new List<int>();
                    for (int x = remap[u] + 1; x < remap[u] + sc; x++)
                    {
                        int cu = Array.FindIndex(remap, r => r == x);
                        if (cu != -1) cuts.AddRange(nodes[cu]);
                    }
                    for (int x = remap[u] - rc; x < remap[u]; x++)
                    {
                        int cu = Array.FindIndex(remap, r => r == x);
                        if (cu != -1) cuts.AddRange(nodes[cu]);
                    }
                    int rem = cSize - cuts.Count;
                    for (int x = 0; x < nodes[u].Count && rem > 0; x++)
                    {
                        if (!cuts.Contains(nodes[u][x]))
                        {
                            cuts.Add(nodes[u][x]);
                            rem--;
                        }
                    }
                    foreach (int v in cuts) order[++l] = v;
                }
                else
                {
                    foreach (int v in nodes[u]) order[++l] = v;
                }
            }
        }

        Console.WriteLine(l);
        if (l > 0)
        {
            var sb = new System.Text.StringBuilder();
            for (int i = 1; i <= l; i++)
            {
                if (i > 1) sb.Append(' ');
                sb.Append(order[i]);
            }
            Console.WriteLine(sb.ToString());
        }
        else
        {
            Console.WriteLine();
        }
    }
}
```
