#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <cstring>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to, rev;
    int cap;
};

struct MaxFlow {
    int n;
    vector<vector<Edge>> graph;
    vector<int> level, ptr;

    MaxFlow(int n) : n(n), graph(n), level(n), ptr(n) {}

    void add_edge(int u, int v, int cap) {
        graph[u].push_back({v, (int)graph[v].size(), cap});
        graph[v].push_back({u, (int)graph[u].size() - 1, 0});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        vector<int> q(n);
        int head = 0, tail = 0;
        q[tail++] = s;
        while (head < tail) {
            int u = q[head++];
            for (auto& e : graph[u]) {
                if (e.cap > 0 && level[e.to] == -1) {
                    level[e.to] = level[u] + 1;
                    q[tail++] = e.to;
                }
            }
        }
        return level[t] != -1;
    }

    int dfs(int u, int t, int f) {
        if (u == t) return f;
        for (int& i = ptr[u]; i < graph[u].size(); i++) {
            Edge& e = graph[u][i];
            if (e.cap > 0 && level[e.to] == level[u] + 1) {
                int pushed = dfs(e.to, t, min(f, e.cap));
                if (pushed > 0) {
                    e.cap -= pushed;
                    graph[e.to][e.rev].cap += pushed;
                    return pushed;
                }
            }
        }
        return 0;
    }

    int dinic(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(ptr.begin(), ptr.end(), 0);
            int pushed;
            while ((pushed = dfs(s, t, INF)) > 0) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<bool> min_cut(int s) {
        vector<bool> visited(n, false);
        vector<int> q(n);
        int head = 0, tail = 0;
        q[tail++] = s;
        visited[s] = true;
        while (head < tail) {
            int u = q[head++];
            for (auto& e : graph[u]) {
                if (e.cap > 0 && !visited[e.to]) {
                    visited[e.to] = true;
                    q[tail++] = e.to;
                }
            }
        }
        return visited;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<int> c(n);
    int sum_c = 0;
    for (int i = 0; i < n; i++) {
        cin >> c[i];
        sum_c += c[i];
    }

    // 0/1 ILP: x_u + x_v <= 1 can be modeled as a Minimum Vertex Cover on the constraint graph.
    // By König's Theorem, MVC = Maximum Matching = Max Flow.
    // The unselected nodes in the MVC form the Maximum Weight Independent Set, 
    // which directly maximizes the objective sum(c_i * x_i).
    MaxFlow mf(n + 2);
    int s = n, t = n + 1;

    for (int i = 0; i < n; i++) {
        if (c[i] >= 0) {
            mf.add_edge(s, i, c[i]);
        } else {
            mf.add_edge(i, t, -c[i]);
        }
    }

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        mf.add_edge(u, v, INF);
    }

    int min_cut_val = mf.dinic(s, t);
    int max_obj = sum_c - min_cut_val;

    vector<bool> visited = mf.min_cut(s);
    vector<int> x(n, 0);
    for (int i = 0; i < n; i++) {
        if (visited[i]) {
            x[i] = 1;
        }
    }

    cout << max_obj << "\n";
    for (int i = 0; i < n; i++) {
        cout << x[i] << (i == n - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
