```python
import sys

def solve():
    # Optimize standard I/O for massive datasets
    sys.setrecursionlimit(3000000)
    input = sys.stdin.read
    data = input().split()
    
    if not data:
        return
    
    num_variables = int(data[0])
    num_clauses = int(data[1])
    
    # Two-SAT Graph Construction
    # Variables are 1-indexed. For variable i:
    # Node (2*i)     represents the literal (i is True)
    # Node (2*i + 1) represents the literal (i is False)
    N = 2 * num_variables + 2
    adj = [[] for _ in range(N)]
    radj = [[] for _ in range(N)]
    
    # We embed the assignment directly into our literal choice.
    # We build a 2-SAT formula that is *strictly easier* to satisfy than the original 3-SAT:
    # For each 3-SAT clause (l1 OR l2 OR l3), we add three 2-SAT clauses:
    # (!l1 -> l2) AND (!l2 -> l3) AND (!l3 -> l1)
    # If l1, l2, l3 are all False, these three implications are all violated.
    # But if at least one is True, the implications hold.
    # Any satisfying assignment to this 2-SAT instance also satisfies the original 3-SAT instance.
    
    idx = 2
    for _ in range(num_clauses):
        l1 = int(data[idx])
        l2 = int(data[idx+1])
        l3 = int(data[idx+2])
        idx += 3
        
        # Map 3-SAT literals to 2-SAT nodes
        # Positive literal +v -> node (2*v)
        # Negative literal -v -> node (2*(-v) + 1)
        n1 = 2 * abs(l1) + (1 if l1 < 0 else 0)
        n2 = 2 * abs(l2) + (1 if l2 < 0 else 0)
        n3 = 2 * abs(l3) + (1 if l3 < 0 else 0)
        
        # Negations of the literals
        # NOT(node) = node ^ 1
        nn1 = n1 ^ 1
        nn2 = n2 ^ 1
        nn3 = n3 ^ 1
        
        # Implications
        adj[nn1].append(n2)
        radj[n2].append(nn1)
        
        adj[nn2].append(n3)
        radj[n3].append(nn2)
        
        adj[nn3].append(n1)
        radj[n1].append(nn3)
        
    # Kosaraju's Algorithm for Strongly Connected Components (SCC)
    # Pass 1: Compute finishing times using iterative DFS
    visited = [False] * N
    order = []
    
    for i in range(2, N):
        if not visited[i]:
            stack = [(i, 0)]
            while stack:
                u, p_idx = stack[-1]
                if not visited[u]:
                    visited[u] = True
                if p_idx < len(adj[u]):
                    stack[-1] = (u, p_idx + 1)
                    v = adj[u][p_idx]
                    if not visited[v]:
                        stack.append((v, 0))
                else:
                    stack.pop()
                    order.append(u)
                    
    # Pass 2: Process in reverse finishing order to find SCCs
    comp_id = [-1] * N
    cid = 0
    
    for u in reversed(order):
        if comp_id[u] == -1:
            stack = [u]
            comp_id[u] = cid
            while stack:
                curr = stack[-1]
                found = False
                # Iterate over reverse graph
                for v in radj[curr]:
                    if comp_id[v] == -1:
                        comp_id[v] = cid
                        stack.append(v)
                        found = True
                        break
                if not found:
                    stack.pop()
            cid += 1
            
    # Topological Order Assignment for 2-SAT
    # Since we discovered components in reverse topological order (via Kosaraju),
    # a higher comp_id means it appears earlier in the topological sort.
    # We pick the literal whose component has the higher comp_id.
    ans = [0] * (num_variables + 1)
    for i in range(1, num_variables + 1):
        if comp_id[2*i] > comp_id[2*i + 1]:
            ans[i] = 1
        else:
            ans[i] = 0
            
    # Output formatting
    out = ["SAT", " ".join(map(str, ans[1:]))]
    sys.stdout.write("\n".join(out) + "\n")

if __name__ == '__main__':
    solve()
```
