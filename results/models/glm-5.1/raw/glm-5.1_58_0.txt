const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const R = grid.length;
const C = R > 0 ? grid[0].length : 0;
const numRobots = robots.length;
const numJobs = jobs.length;

const INF = 1e9;

// Initialize grid distance array
const dist = new Array(R);
for (let i = 0; i < R; i++) {
    dist[i] = new Int32Array(C).fill(INF);
}

// Multi-source BFS from all robot start positions
const queueR = new Int32Array(numRobots);
const queueC = new Int32Array(numRobots);
let head = 0, tail = 0;

for (let i = 0; i < numRobots; i++) {
    const r = robots[i].y;
    const c = robots[i].x;
    if (r >= 0 && r < R && c >= 0 && c < C && grid[r][c] !== '#') {
        if (dist[r][c] === INF) {
            dist[r][c] = 0;
            queueR[tail] = r;
            queueC[tail] = c;
            tail++;
        }
    }
}

const dr = [-1, 1, 0, 0];
const dc = [0, 0, -1, 1];

while (head < tail) {
    const r = queueR[head];
    const c = queueC[head];
    head++;
    const d = dist[r][c];
    for (let k = 0; k < 4; k++) {
        const nr = r + dr[k];
        const nc = c + dc[k];
        if (nr >= 0 && nr < R && nc >= 0 && nc < C && grid[nr][nc] !== '#' && dist[nr][nc] === INF) {
            dist[nr][nc] = d + 1;
            queueR[tail] = nr;
            queueC[tail] = nc;
            tail++;
        }
    }
}

// Sort jobs by priority (descending), then by distance to nearest robot (ascending)
const sortedJobs = jobs.map((job, i) => ({
    id: job.id,
    x: job.x,
    y: job.y,
    priority: job.priority,
    dist: (job.y >= 0 && job.y < R && job.x >= 0 && job.x < C) ? dist[job.y][job.x] : INF,
    index: i
}));

sortedJobs.sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    if (a.dist !== b.dist) return a.dist - b.dist;
    return a.id - b.id;
});

// Initialize robot states for assignment simulation
const robotX = new Int32Array(numRobots);
const robotY = new Int32Array(numRobots);
const robotDist = new Int32Array(numRobots);
const robotAssignments = new Array(numRobots);
const localDist = new Array(R);

for (let i = 0; i < numRobots; i++) {
    robotX[i] = robots[i].x;
    robotY[i] = robots[i].y;
    robotDist[i] = 0;
    robotAssignments[i] = [];
}

// Assign jobs greedily
for (const job of sortedJobs) {
    const jx = job.x;
    const jy = job.y;

    if (jx < 0 || jx >= C || jy < 0 || jy >= R || grid[jy][jx] === '#') {
        continue; // Skip unreachable jobs
    }

    // Local BFS from the job position to find distances to all robot current positions
    for (let r = 0; r < R; r++) {
        localDist[r] = new Int32Array(C).fill(INF);
    }
    localDist[jy][jx] = 0;

    head = 0;
    tail = 0;
    const lQR = new Int32Array(R * C);
    const lQC = new Int32Array(R * C);
    lQR[tail] = jy;
    lQC[tail] = jx;
    tail++;

    while (head < tail) {
        const r = lQR[head];
        const c = lQC[head];
        head++;
        const d = localDist[r][c];
        if (d >= robotDist[bestR] + bestLocal) break; // Pruning (safe because BFS explores in order)
        for (let k = 0; k < 4; k++) {
            const nr = r + dr[k];
            const nc = c + dc[k];
            if (nr >= 0 && nr < R && nc >= 0 && nc < C && grid[nr][nc] !== '#' && localDist[nr][nc] === INF) {
                localDist[nr][nc] = d + 1;
                lQR[tail] = nr;
                lQC[tail] = nc;
                tail++;
            }
        }
    }

    let bestR = -1;
    let bestTotal = INF;
    let bestLocal = INF;

    for (let i = 0; i < numRobots; i++) {
        const d = localDist[robotY[i]][robotX[i]];
        if (d < INF) {
            const total = robotDist[i] + d;
            if (total < bestTotal || (total === bestTotal && i < bestR)) {
                bestTotal = total;
                bestLocal = d;
                bestR = i;
            }
        }
    }

    if (bestR !== -1) {
        robotAssignments[bestR].push(job.id);
        robotDist[bestR] += bestLocal;
        robotX[bestR] = jx;
        robotY[bestR] = jy;
    }
}

const result = { assignments: robotAssignments };
fs.writeFileSync(1, JSON.stringify(result));
