; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %functype
%label = OpLabel
%uv_load = OpLoad %vec2 %in_uv
%uv_x = OpCompositeExtract %float %uv_load 0
%uv_y = OpCompositeExtract %float %uv_load 1
%color = OpCompositeConstruct %vec4 %uv_x %uv_y %float_0 %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
