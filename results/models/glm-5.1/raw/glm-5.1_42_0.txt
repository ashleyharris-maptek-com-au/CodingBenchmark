```spirv
; SPIR-V
Version 1.0
Generator 0
Bound 80
Schema 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations - Input
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2

; Decorations - Output
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.04

; Input Block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

; Input Arrays
%arr_v3 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

; Output Block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

; Output Variables
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %funct
%entry = OpLabel

; Read gl_in[0].gl_Position
%ptr_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %ptr_pos
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

; Read color[0]
%ptr_color = OpAccessChain %ptr_in_v3 %inColor %int_0
%color = OpLoad %v3float %ptr_color

; Calculate offsets
%xh = OpFAdd %float %x %float_h
%xl = OpFSub %float %x %float_h
%yh = OpFAdd %float %y %float_h
%yl = OpFSub %float %y %float_h

; Emit Vertex 0: (xl, yl)
%v0 = OpCompositeConstruct %v4float %xl %yl %float_0 %float_1
%ptr_out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos0 %v0
OpStore %outColor %color
OpEmitVertex

; Emit Vertex 1: (xh, yl)
%v1 = OpCompositeConstruct %v4float %xh %yl %float_0 %float_1
%ptr_out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos1 %v1
OpStore %outColor %color
OpEmitVertex

; Emit Vertex 2: (xl, yh)
%v2 = OpCompositeConstruct %v4float %xl %yh %float_0 %float_1
%ptr_out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos2 %v2
OpStore %outColor %color
OpEmitVertex

; Emit Vertex 3: (xh, yh)
%v3 = OpCompositeConstruct %v4float %xh %yh %float_0 %float_1
%ptr_out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos3 %v3
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
