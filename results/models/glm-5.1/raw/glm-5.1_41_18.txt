; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fn_main = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%inWorldPos = OpVariable %ptr_input_vec3 Input
%inNormal = OpVariable %ptr_input_vec3 Input
%inUV = OpVariable %ptr_input_vec2 Input
%inTangent = OpVariable %ptr_input_vec3 Input
%inColor = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2

%main = OpFunction %void None %fn_main
%label_entry = OpLabel

%normal_raw = OpLoad %vec3 %inNormal
%N = OpExtInst %vec3 %1 Normalize %normal_raw

%camera_pos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%camera_pos_data = OpLoad %vec4 %camera_pos_ptr
%camera_pos = OpVectorShuffle %vec3 %camera_pos_data %camera_pos_data 0 1 2

%world_pos = OpLoad %vec3 %inWorldPos
%V_unnorm = OpFSub %vec3 %camera_pos %world_pos
%V = OpExtInst %vec3 %1 Normalize %V_unnorm

%NdotV = OpDot %float %N %V
%NdotV_clamped = OpExtInst %float %1 FMax %NdotV %float_0

%is_low = OpFOrdLessThan %bool %NdotV_clamped %float_0_5
%NdotV_times_2 = OpFMul %float %NdotV_clamped %float_2
%g_low = OpFSub %float %NdotV_times_2 %float_0 ; just NdotV_times_2

%half_range = OpFSub %float %NdotV_clamped %float_0_5
%half_range_times_2 = OpFMul %float %half_range %float_2
%one_minus_half_range_times_2 = OpFSub %float %float_1 %half_range_times_2
%r_high = OpCopyObject %float %one_minus_half_range_times_2
%g_high = OpCopyObject %float %one_minus_half_range_times_2
%b_high = OpCopyObject %float %half_range_times_2

%r = OpSelect %float %is_low %float_1 %r_high
%g = OpSelect %float %is_low %g_low %g_high
%b = OpSelect %float %is_low %float_0 %b_high

%final_color = OpCompositeConstruct %vec4 %r %g %b %float_1
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
