; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%fn_type = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float

%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO

%in_uv = OpVariable %ptr_input_v2float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%float_0_5 = OpConstant %float 0.5
%float_20 = OpConstant %float 20.0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

%c_float_1 = OpConstant %float -1.0
%c_float_2 = OpConstant %float -2.0

%main = OpFunction %void None %fn_type
%label_entry = OpLabel

%uv = OpLoad %v2float %in_uv
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

; c.x = uv.x * 3 - 2
%cx = OpFMul %float %uv_x %float_3
%cx_off = OpFAdd %float %cx %c_float_2

; c.y = uv.y * 2 - 1
%cy = OpFMul %float %uv_y %float_2
%cy_off = OpFAdd %float %cy %c_float_1

%z_init = OpCompositeConstruct %v2float %float_0 %float_0
%c_init = OpCompositeConstruct %v2float %cx_off %cy_off

OpBranch %loop_header
%loop_header = OpLabel
%z = OpPhi %v2float %z_init %label_entry %z_next %loop_body
%c = OpPhi %v2float %c_init %label_entry %c %loop_body
%i = OpPhi %int %int_0 %label_entry %i_next %loop_body

%zx = OpCompositeExtract %float %z 0
%zy = OpCompositeExtract %float %z 1

; length squared = zx^2 + zy^2
%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%len2 = OpFAdd %float %zx2 %zy2

%escape = OpFOrdGreaterThan %bool %len2 %float_4
%max_iter = OpIEqual %bool %i %int_20

%cond_break = OpLogicalOr %bool %escape %max_iter
OpLoopMerge %loop_merge %loop_body None
OpBranchConditional %cond_break %loop_merge %loop_body

%loop_body = OpLabel
; z_new.x = zx^2 - zy^2 + c.x
%zx2_minus_zy2 = OpFSub %float %zx2 %zy2
%z_new_x = OpFAdd %float %zx2_minus_zy2 %cx_off

; z_new.y = 2 * zx * zy + c.y
%zx_zy = OpFMul %float %zx %zy
%two_zx_zy = OpFMul %float %zx_zy %float_2
%z_new_y = OpFAdd %float %two_zx_zy %cy_off

%z_next = OpCompositeConstruct %v2float %z_new_x %z_new_y
%i_next = OpIAdd %int %i %int_1
OpBranch %loop_header

%loop_merge = OpLabel
%fi = OpConvertSToF %float %i
%t = OpFDiv %float %fi %float_20

; color: (t, t*0.5, 1.0-t, 1.0)
%r = OpCopyObject %float %t
%g = OpFMul %float %t %float_0_5
%b = OpFSub %float %float_1 %t
%final_color = OpCompositeConstruct %v4float %r %g %b %float_1

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
