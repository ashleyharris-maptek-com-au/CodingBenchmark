```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Threading;
using System.Threading.Tasks;

public class MaximumClique
{
    private static int n;
    private static int[][] adj;
    private static int[] deg;
    private static int[] colorOrder;
    private static int maxCliqueSize;
    private static int[] maxCliqueVertices;
    private static bool timeUp;
    private static Stopwatch sw;

    private static readonly TimeSpan TimeLimit = TimeSpan.FromMinutes(4.5);

    public static void Main()
    {
        sw = Stopwatch.StartNew();
        ReadGraph();

        // Initialize data structures
        deg = new int[n];
        for (int i = 0; i < n; i++)
        {
            deg[i] = adj[i].Length;
        }

        int[] initial = new int[n];
        for (int i = 0; i < n; i++) initial[i] = i;

        colorOrder = new int[n];
        maxCliqueSize = 0;
        maxCliqueVertices = new int[0];
        timeUp = false;

        // Run greedy heuristic in parallel to find a good initial clique
        RunGreedyHeuristicParallel();

        // If a clique of size k is found, we only need to start branches from vertices
        // with degree >= k - 1. Also process highest degree vertices first.
        Array.Sort(initial, (a, b) => deg[b].CompareTo(deg[a]));

        // Process initial vertices in parallel using multiple threads
        Parallel.ForEach(initial, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, v =>
        {
            if (timeUp) return;
            if (deg[v] + 1 <= maxCliqueSize) return;

            // Prune initial candidate set
            List<int> cand = new List<int>();
            for (int i = 0; i < adj[v].Length; i++)
            {
                int u = adj[v][i];
                if (deg[u] + 2 > maxCliqueSize)
                {
                    cand.Add(u);
                }
            }

            if (cand.Count + 1 <= maxCliqueSize) return;

            int[] subAdjMap = new int[n];
            BuildSubAdj(cand, subAdjMap);

            int[] currentClique = new int[n];
            currentClique[0] = v;
            Expand(1, cand, subAdjMap, currentClique);
        });

        PrintResult();
    }

    private static void RunGreedyHeuristicParallel()
    {
        Parallel.For(0, n, i =>
        {
            if (timeUp) return;
            int size = 1;
            int[] tempClique = new int[n];
            tempClique[0] = i;
            int count = 1;

            int[] isAdj = new int[n];
            isAdj[i] = 1;
            for (int j = 0; j < adj[i].Length; j++) isAdj[adj[i][j]] = 1;

            for (int iter = 0; iter < 50; iter++)
            {
                int best = -1;
                int bestDeg = -1;
                for (int j = 0; j < n; j++)
                {
                    if (isAdj[j] == 1)
                    {
                        int d = 0;
                        for (int k = 0; k < adj[j].Length; k++)
                        {
                            if (isAdj[adj[j][k]] == 1) d++;
                        }
                        if (d > bestDeg)
                        {
                            bestDeg = d;
                            best = j;
                        }
                    }
                }
                if (best == -1 || bestDeg <= size) break;

                size++;
                tempClique[count++] = best;
                isAdj[best] = 2; // Mark as in clique
                for (int j = 0; j < adj[best].Length; j++)
                {
                    if (isAdj[adj[best][j]] == 1) isAdj[adj[best][j]] = 2;
                }
                for (int j = 0; j < n; j++) if (isAdj[j] == 1) isAdj[j] = 0;
            }

            if (size > maxCliqueSize)
            {
                lock (maxCliqueVertices) // Lock on the array to avoid concurrency issues
                {
                    if (size > maxCliqueSize)
                    {
                        maxCliqueSize = size;
                        maxCliqueVertices = new int[size];
                        Array.Copy(tempClique, maxCliqueVertices, size);
                    }
                }
            }
        });
    }

    private static void BuildSubAdj(List<int> cand, int[] subAdjMap)
    {
        for (int i = 0; i < cand.Count; i++)
        {
            subAdjMap[cand[i]] = i + 1; // 1-indexed for fast check
        }
    }

    private static void ClearSubAdj(List<int> cand, int[] subAdjMap)
    {
        for (int i = 0; i < cand.Count; i++)
        {
            subAdjMap[cand[i]] = 0;
        }
    }

    private static void Expand(int size, List<int> cand, int[] subAdjMap, int[] currentClique)
    {
        if (timeUp) return;
        if (sw.Elapsed > TimeLimit)
        {
            timeUp = true;
            return;
        }

        int maxAllowed = maxCliqueSize - size;
        int colors = ColorSort(cand, subAdjMap, maxAllowed);

        for (int i = cand.Count - 1; i >= 0; i--)
        {
            if (timeUp) return;
            int v = colorOrder[i];
            if (size + colors <= maxCliqueSize)
            {
                return;
            }

            currentClique[size] = v;

            // Prune candidate set for the next level
            List<int> newCand = new List<int>();
            int[] vAdj = adj[v];
            for (int j = 0; j < vAdj.Length; j++)
            {
                if (subAdjMap[vAdj[j]] > 0)
                {
                    newCand.Add(vAdj[j]);
                }
            }

            if (size + 1 > maxCliqueSize)
            {
                maxCliqueSize = size + 1;
                maxCliqueVertices = new int[maxCliqueSize];
                Array.Copy(currentClique, maxCliqueVertices, maxCliqueSize);
            }

            if (newCand.Count > 0)
            {
                int[] newSubAdjMap = new int[n];
                BuildSubAdj(newCand, newSubAdjMap);
                Expand(size + 1, newCand, newSubAdjMap, currentClique);
                ClearSubAdj(newCand, newSubAdjMap);
            }

            cand.RemoveAt(i); // Remove v from candidates
            colors--; // Since we colored in order, removing last reduces max color
        }
    }

    private static int ColorSort(List<int> cand, int[] subAdjMap, int maxAllowed)
    {
        int count = cand.Count;
        int maxColor = 0;
        int[] color = new int[count];

        // Sort candidates by degree descending to maximize coloring tightness
        cand.Sort((a, b) => deg[b].CompareTo(deg[a]));

        for (int i = 0; i < count; i++)
        {
            int v = cand[i];
            bool[] used = new bool[maxColor + 1];
            
            for (int j = 0; j < adj[v].Length; j++)
            {
                int u = adj[v][j];
                int uIdx = subAdjMap[u];
                if (uIdx > 0)
                {
                    int c = color[uIdx - 1];
                    if (c > 0) used[c - 1] = true;
                }
            }

            int c1 = 1;
            while (c1 <= maxColor && used[c1 - 1]) c1++;

            if (c1 > maxColor)
            {
                maxColor = c1;
                if (maxColor > maxAllowed) maxColor = maxAllowed + 1; // Limit for bounding
            }

            color[i] = c1;
            colorOrder[i] = v;
        }

        // Reorder candidates by color ascending so highest colors are popped last
        Array.Sort(color, colorOrder, 0, count);

        // Write back to list
        for (int i = 0; i < count; i++) cand[i] = colorOrder[i];

        return maxColor;
    }

    private static void ReadGraph()
    {
        byte[] buffer = new byte[1 << 20];
        int bytesRead = Console.OpenStandardInput().Read(buffer, 0, buffer.Length);
        int pos = 0;

        int ReadInt()
        {
            while (pos < bytesRead && buffer[pos] < '0') pos++;
            int val = 0;
            while (pos < bytesRead && buffer[pos] >= '0')
            {
                val = val * 10 + (buffer[pos] - '0');
                pos++;
            }
            return val;
        }

        n = ReadInt();
        int m = ReadInt();

        List<int>[] adjList = new List<int>[n];
        for (int i = 0; i < n; i++) adjList[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = ReadInt();
            int v = ReadInt();
            adjList[u].Add(v);
            adjList[v].Add(u);
        }

        adj = new int[n][];
        for (int i = 0; i < n; i++) adj[i] = adjList[i].ToArray();
    }

    private static void PrintResult()
    {
        Console.WriteLine(maxCliqueSize);
        if (maxCliqueSize > 0)
        {
            Console.WriteLine(string.Join(" ", maxCliqueVertices));
        }
    }
}
```
