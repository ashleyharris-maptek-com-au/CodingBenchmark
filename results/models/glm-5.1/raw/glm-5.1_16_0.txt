This C++ solver implements an adaptive Job-Shop Scheduling approach that dynamically selects its strategy based on the problem size.

### Strategy Selection:
1. **Trivial / Small Scale (≤ 50k tasks, ≤ 1000 machines)**: Uses an **Active Schedule Construction (Non-Delay Schedule)**. At each step, it resolves all currently schedulable earliest operations (respecting both job precedence and machine availability). If there are conflicts on a machine, it uses a **Longest Processing Time (LPT)** heuristic to prioritize operations. This builds a high-quality active schedule in `O(N log N)` time.
2. **Extreme Scale (> 50k tasks or > 1000 machines)**: Uses a **Fast Greedy Dispatch**. It builds a priority queue over all available next tasks for each job, always picking the one with the earliest possible start. Ties are broken by shortest processing time. It completely avoids tracking machine availability per-slot, falling back to tracking just the `last_time` per machine, yielding strict `O(N log N)` performance and minimal memory overhead perfectly suited for 50k+ tasks.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstdint>

using namespace std;

// Structure to hold task information
struct Task {
    int machine;
    int duration;
    int start_time;
};

// Structure for priority queue elements
struct PQEntry {
    int earliest_start;
    int duration;
    int job_id;
    int task_idx;
    
    // Comparator for priority queue
    // For LPT (used in active scheduling): prioritize by LARGER duration first
    // For Greedy: prioritize by SMALLER duration first
    // Ties broken by job_id, then task_idx for deterministic output
    bool operator>(const PQEntry& other) const {
        if (earliest_start != other.earliest_start) return earliest_start > other.earliest_start;
        if (duration != other.duration) return duration < other.duration; // Will be flipped dynamically if needed
        if (job_id != other.job_id) return job_id > other.job_id;
        return task_idx > other.task_idx;
    }
};

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    int total_tasks = 0;
    int max_tasks_per_job = 0;

    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            cin >> jobs[j][t].machine >> jobs[j][t].duration;
            jobs[j][t].start_time = -1;
        }
        total_tasks += T;
        max_tasks_per_job = max(max_tasks_per_job, T);
    }

    // Machine states tracking
    vector<int> machine_next_available(M, 0);

    // Job states tracking
    vector<int> job_next_task(J, 0);
    vector<int> job_next_time(J, 0);

    // Adaptive threshold based on problem scale
    const int SMALL_SCALE_TASKS = 50000;
    const int SMALL_SCALE_MACHINES = 1000;

    if (total_tasks <= SMALL_SCALE_TASKS && M <= SMALL_SCALE_MACHINES) {
        // --- Active Schedule Construction (High Quality) ---
        // Uses LPT (Longest Processing Time) conflict resolution for high density
        PQEntry cmp_entry;
        auto cmp = [&cmp_entry](const PQEntry& a, const PQEntry& b) {
            if (a.earliest_start != b.earliest_start) return a.earliest_start > b.earliest_start;
            // Flip LPT logic: we want LARGER duration first, hence we sort by LESS duration first in the max-heap comparator
            if (a.duration != b.duration) return a.duration < b.duration; 
            if (a.job_id != b.job_id) return a.job_id > b.job_id;
            return a.task_idx > b.task_idx;
        };

        priority_queue<PQEntry, vector<PQEntry>, decltype(cmp)> pq(cmp);

        for (int j = 0; j < J; ++j) {
            if (!jobs[j].empty()) {
                pq.push({0, jobs[j][0].duration, j, 0});
            }
        }

        while (!pq.empty()) {
            int min_start = pq.top().earliest_start;
            vector<PQEntry> current;
            
            // Extract all tasks that can start at exactly min_start
            while (!pq.empty() && pq.top().earliest_start == min_start) {
                current.push_back(pq.top());
                pq.pop();
            }

            vector<bool> machine_used(M, false);

            // Schedule tasks resolving conflicts via LPT
            for (const auto& entry : current) {
                int m = jobs[entry.job_id][entry.task_idx].machine;
                if (machine_used[m]) {
                    // Conflict: defer this task
                    int est = max(machine_next_available[m], entry.earliest_start);
                    pq.push({est, entry.duration, entry.job_id, entry.task_idx});
                } else {
                    // Schedule it
                    machine_used[m] = true;
                    jobs[entry.job_id][entry.task_idx].start_time = entry.earliest_start;
                    int end_time = entry.earliest_start + entry.duration;
                    machine_next_available[m] = end_time;
                    job_next_time[entry.job_id] = end_time;
                    job_next_task[entry.job_id] = entry.task_idx + 1;
                }
            }

            // Push next tasks for jobs that had a task scheduled just now
            for (const auto& entry : current) {
                if (jobs[entry.job_id][entry.task_idx].start_time != -1) {
                    int j = entry.job_id;
                    int next_t = job_next_task[j];
                    if (next_t < jobs[j].size()) {
                        int est = max(machine_next_available[jobs[j][next_t].machine], job_next_time[j]);
                        pq.push({est, jobs[j][next_t].duration, j, next_t});
                    }
                }
            }
        }
    } else {
        // --- Fast Greedy Dispatch (Extreme Scale) ---
        // Uses SPT (Shortest Processing Time) tie-breaking, bypasses machine collision checks
        auto cmp = [](const PQEntry& a, const PQEntry& b) {
            if (a.earliest_start != b.earliest_start) return a.earliest_start > b.earliest_start;
            if (a.duration != b.duration) return a.duration > b.duration;
            if (a.job_id != b.job_id) return a.job_id > b.job_id;
            return a.task_idx > b.task_idx;
        };

        priority_queue<PQEntry, vector<PQEntry>, decltype(cmp)> pq(cmp);

        for (int j = 0; j < J; ++j) {
            if (!jobs[j].empty()) {
                int m = jobs[j][0].machine;
                pq.push({max(machine_next_available[m], 0), jobs[j][0].duration, j, 0});
            }
        }

        while (!pq.empty()) {
            auto entry = pq.top();
            pq.pop();

            int j = entry.job_id;
            int t = entry.task_idx;
            int m = jobs[j][t].machine;

            // Task starts at the max of previously computed earliest start and current machine availability
            int start = max(entry.earliest_start, machine_next_available[m]);
            int end = start + jobs[j][t].duration;

            jobs[j][t].start_time = start;
            machine_next_available[m] = end;

            // Push next task for the job
            if (t + 1 < jobs[j].size()) {
                int next_m = jobs[j][t + 1].machine;
                int next_est = max(machine_next_available[next_m], end);
                pq.push({next_est, jobs[j][t + 1].duration, j, t + 1});
            }
        }
    }

    // Fast Output
    // Estimate maximum possible characters required (buffer safety margin applied)
    size_t out_buf_size = max((size_t)(total_tasks * 25), (size_t)65536);
    vector<char> out_buf(out_buf_size);
    
    // Set custom buffer for stdout
    setvbuf(stdout, out_buf.data(), _IOFBF, out_buf.size());

    for (int j = 0; j < J; ++j) {
        for (int t = 0; t < (int)jobs[j].size(); ++t) {
            if (t > 0) cout << ' ';
            cout << jobs[j][t].start_time << ' ' << jobs[j][t].machine;
        }
        cout << '\n';
    }

    return 0;
}
```
