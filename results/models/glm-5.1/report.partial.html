#include <cmath>
#include <cstdio>
#include <cstdlib>

struct Point { double x, y; };

static inline double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x, dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Thresholds for algorithm transitions
const int THRESH_2OPT = 5000;
const int THRESH_GRID = 200000;

// ---------------------------------------------------------
// 1. Trivial/Medium Scale: NN + Full 2-opt
// ---------------------------------------------------------
void solve_small(const std::vector<Point>& pts, std::vector<int>& tour) {
    int N = pts.size();
    std::vector<bool> visited(N, false);
    tour[0] = 0;
    visited[0] = true;
    int curr = 0;
    for (int i = 1; i < N; ++i) {
        double best_d2 = 1e300;
        int best_j = -1;
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d2 = dist2(pts[curr], pts[j]);
                if (d2 < best_d2) { best_d2 = d2; best_j = j; }
            }
        }
        tour[i] = best_j;
        visited[best_j] = true;
        curr = best_j;
    }

    // Full 2-opt local search
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 1; ++i) {
            int t1 = tour[i], t2 = tour[i + 1];
            double d1 = dist2(pts[t1], pts[t2]);
            for (int j = i + 2; j < N; ++j) {
                int t3 = tour[j], t4 = tour[(j + 1) % N];
                if (t1 == t4) break;
                double d2 = dist2(pts[t3], pts[t4]);
                double d3 = dist2(pts[t1], pts[t3]);
                double d4 = dist2(pts[t2], pts[t4]);
                if (d3 + d4 < d1 + d2 - 1e-9) {
                    std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    t2 = t3; d1 = d3;
                }
            }
        }
    }
}

// 2. Large Scale: NN with Grid
// ---------------------------------------------------------
void solve_grid_nn(const std::vector<Point>& pts, std::vector<int>& tour) {
    int N = pts.size();
    double min_x = 1e300, max_x = -1e300, min_y = 1e300, max_y = -1e300;
    for (int i = 0; i < N; ++i) {
        min_x = std::min(min_x, pts[i].x); max_x = std::max(max_x, pts[i].x);
        min_y = std::min(min_y, pts[i].y); max_y = std::max(max_y, pts[i].y);
    }
    int G = std::max(1, (int)std::sqrt(N / 4.0));
    double cw = (max_x - min_x) / G + 1e-9;
    double ch = (max_y - min_y) / G + 1e-9;
    std::vector<std::vector<int>> grid(G * G);
    for (int i = 0; i < N; ++i) {
        int gx = std::min(G - 1, (int)((pts[i].x - min_x) / cw));
        int gy = std::min(G - 1, (int)((pts[i].y - min_y) / ch));
        grid[gx * G + gy].push_back(i);
    }
    std::vector<bool> visited(N, false);
    tour[0] = 0; visited[0] = true;
    int curr = 0;
    for (int i = 1; i < N; ++i) {
        int cgx = std::min(G - 1, (int)((pts[curr].x - min_x) / cw));
        int cgy = std::min(G - 1, (int)((pts[curr].y - min_y) / ch));
        int best_j = -1; double best_d2 = 1e300;
        for (int r = 0; r <= G; ++r) {
            bool found = false;
            int x1 = std::max(0, cgx - r), x2 = std::min(G - 1, cgx + r);
            int y1 = std::max(0, cgy - r), y2 = std::min(G - 1, cgy + r);
            for (int gx = x1; gx <= x2; ++gx) {
                for (int gy = y1; gy <= y2; ++gy) {
                    if (gx != x1 && gx != x2 && gy != y1 && gy != y2) continue;
                    for (int j : grid[gx * G + gy]) {
                        if (!visited[j]) {
                            double d2 = dist2(pts[curr], pts[j]);
                            if (d2 < best_d2) { best_d2 = d2; best_j = j; found = true; }
                        }
                    }
                }
            }
            if (found) break;
        }
        tour[i] = best_j; visited[best_j] = true; curr = best_j;
    }
}

// 3. Extreme/Epic Scale: Parallel Multi-Start Grid-NN + Fast 2-opt

struct Node {
    int prev, next;
};

void nn_search(int start, int N, const std::vector<Point>& pts, int G, double cw, double ch, 
               double min_x, double min_y, const std::vector<std::vector<int>>& grid, 
               std::vector<Node>& path) {
    std::vector<bool> visited(N, false);
    visited[start] = true;
    path[start].prev = start; path[start].next = start;
    int curr = start;
    for (int k = 1; k < N; ++k) {
        int cgx = std::min(G - 1, (int)((pts[curr].x - min_x) / cw));
        int cgy = std::min(G - 1, (int)((pts[curr].y - min_y) / ch));
        int best_j = -1; double best_d2 = 1e300;
        for (int r = 0; r <= G; ++r) {
            bool found = false;
            int x1 = std::max(0, cgx - r), x2 = std::min(G - 1, cgx + r);
            int y1 = std::max(0, cgy - r), y2 = std::min(G - 1, cgy + r);
            for (int gx = x1; gx <= x2; ++gx) {
                for (int gy = y1; gy <= y2; ++gy) {
                    if (gx != x1 && gx != x2 && gy != y1 && gy != y2) continue;
                    for (int j : grid[gx * G + gy]) {
                        if (!visited[j]) {
                            double d2 = dist2(pts[curr], pts[j]);
                            if (d2 < best_d2) { best_d2 = d2; best_j = j; found = true; }
                        }
                    }
                }
            }
            if (found) break;
        }
        visited[best_j] = true;
        path[best_j].prev = curr; path[best_j].next = path[curr].next;
        path[path[curr].next].prev = best_j; path[curr].next = best_j;
        curr = best_j;
    }
}

// Bounded 2-opt for massive scale targeting worst kinks
void bounded_2opt(int N, const std::vector<Point>& pts, std::vector<Node>& path, double R2) {
    bool improved = true;
    while (improved) {
        improved = false;
        for (int a = 0; a < N; ++a) {
            int b = path[a].next;
            if (dist2(pts[a], pts[b]) > R2) {
                int c = path[b].next;
                int d = path[c].next;
                if (dist2(pts[a], pts[c]) + dist2(pts[b], pts[d]) < 
                    dist2(pts[a], pts[b]) + dist2(pts[c], pts[d]) - 1e-9) {
                    path[a].next = c; path[c].prev = a;
                    path[b].next = d; path[d].prev = b;
                    path[b].prev = b; path[c].next = c; // Detach b-c segment safely
                    improved = true;
                }
            }
        }
    }
}

void solve_massive(const std::vector<Point>& pts, std::vector<int>& tour) {
    int N = pts.size();
    double min_x = 1e300, max_x = -1e300, min_y = 1e300, max_y = -1e300;
    for (int i = 0; i < N; ++i) {
        min_x = std::min(min_x, pts[i].x); max_x = std::max(max_x, pts[i].x);
        min_y = std::min(min_y, pts[i].y); max_y = std::max(max_y, pts[i].y);
    }
    int G = std::max(1, (int)std::sqrt(N / 4.0));
    double cw = (max_x - min_x) / G + 1e-9;
    double ch = (max_y - min_y) / G + 1e-9;
    std::vector<std::vector<int>> grid(G * G);
    for (int i = 0; i < N; ++i) {
        int gx = std::min(G - 1, (int)((pts[i].x - min_x) / cw));
        int gy = std::min(G - 1, (int)((pts[i].y - min_y) / ch));
        grid[gx * G + gy].push_back(i);
    }
    
    int num_threads = std::thread::hardware_concurrency();
    if (num_threads == 0) num_threads = 8;
    int K = num_threads;
    
    std::vector<std::vector<Node>> paths(K, std::vector<Node>(N));
    std::vector<double> lens(K, 1e300);
    std::vector<int> starts(K);
    for (int k = 0; k < K; ++k) {
        starts[k] = (k * N) / K; // Well-spread out multi-starts
    }
    
    std::vector<std::thread> threads;
    for (int k = 0; k < K; ++k) {
        threads.emplace_back([&, k]() {
            nn_search(starts[k], N, pts, G, cw, ch, min_x, min_y, grid, paths[k]);
            double len = 0; int c = starts[k];
            for (int i = 0; i < N; ++i) { len += std::sqrt(dist2(pts[c], pts[paths[k][c].next])); c = paths[k][c].next; }
            lens[k] = len;
        });
    }
    for (auto& t : threads) t.join();
    
    int best_k = 0;
    for (int k = 1; k < K; ++k) if (lens[k] < lens[best_k]) best_k = k;
    
    double avg_len = lens[best_k] / N;
    double R2 = (avg_len * avg_len) * 9.0; // 3x expected NN edge length bounding
    
    bounded_2opt(N, pts, paths[best_k], R2);
    
    // Extract tour back from path
    tour[0] = 0;
    int c = 0;
    for (int i = 1; i < N; ++i) {
        c = paths[best_k][c].next;
        tour[i] = c;
    }
}

int main() {
    int N;
    if (scanf("%d", &N) != 1) return 0;
    
    std::vector<Point> pts(N);
    for (int i = 0; i < N; ++i) scanf("%lf %lf", &pts[i].x, &pts[i].y);
    
    std::vector<int> tour(N);
    
    if (N <= THRESH_2OPT) {
        solve_small(pts, tour);
    } else if (N <= THRESH_GRID) {
        solve_grid_nn(pts, tour);
    } else {
        solve_massive(pts, tour);
    }
    
    // Ensure the tour explicitly starts at 0 by rotating the cycle if necessary
    int idx0 = 0;
    for (int i = 0; i < N; ++i) {
        if (tour[i] == 0) { idx0 = i; break; }
    }
    std::rotate(tour.begin(), tour.begin() + idx0, tour.end());
    
    // Fast stdout write
    char* buf = new char[N * 12];
    int pos = 0;
    for (int i = 0; i < N; ++i) {
        int v = tour[i];
        if (v < 0) v = 0; // Safety check
        char tmp[12]; int l = 0;
        if (v == 0) tmp[l++] = '0';
        else { while (v > 0) { tmp[l++] = '0' + v % 10; v /= 10; } }
        for (int j = l - 1; j >= 0; --j) buf[pos++] = tmp[j];
        if (i < N - 1) buf[pos++] = ' ';
    }
    buf[pos++] = '\n';
    fwrite(buf, 1, pos, stdout);
    delete[] buf;
    
    return 0;
}
