fn greedy_with_bias(adj: &[Vec<usize>], biases: &[u64]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }
    let mut dominated = vec![false; n];
    let mut in_solution = vec![false; n];
    let mut undominated = n;
    let mut heap = BinaryHeap::with_capacity(n);
    for v in 0..n {
        heap.push(HeapEntry {
            score: adj[v].len(),
            bias: biases[v],
            vertex: v,
        });
    }
    let mut solution = Vec::new();
    while undominated > 0 {
        let entry = match heap.pop() {
            Some(e) => e,
            None => {
                if let Some(v) = dominated.iter().position(|&d| !d) {
                    if !in_solution[v] {
                        in_solution[v] = true;
                        solution.push(v);
                    }
                    for &u in &adj[v] {
                        if !dominated[u] {
                            dominated[u] = true;
                            undominated -= 1;
                        }
                    }
                    continue;
                } else {
                    break;
                }
            }
        };
        let v = entry.vertex;
        if in_solution[v] {
            continue;
        }
        let actual = adj[v].iter().filter(|&&u| !dominated[u]).count();
        if actual == 0 {
            continue;
        }
        if actual < entry.score {
            heap.push(HeapEntry {
                score: actual,
                bias: biases[v],
                vertex: v,

        }

        }

fn prune_solution(solution: &[usize], adj: &[Vec<usize>], n: usize) -> Vec<usize> {
    let mut coverage = vec![0u32; n];
    for &v in solution {
        for &u in &adj[v] {
            coverage[u] += 1;
        }
    }
    let mut kept_rev = Vec::with_capacity(solution.len());
    for &v in solution.iter().rev() {
        let mut removable = true;
        for &u in &adj[v] {
            if coverage[u] <= 1 {
                removable = false;
                break;
            }
        }
        if removable {
            for &u in &adj[v] {
                coverage[u] -= 1;
            }
        } else {
            kept_rev.push(v);
        }
    }
    kept_rev.reverse();
    kept_rev
}

fn prune_until_stable(mut solution: Vec<usize>, adj: &[Vec<usize>], n: usize) -> Vec<usize> {
    loop {
        let pruned = prune_solution(&solution, adj, n);
        if pruned.len() == solution.len() {
            return pruned;
        }
        solution = pruned;
    }
}

/// Runs multiple biased greedy passes followed by pruning and keeps the best result.
fn solve(adj: &[Vec<usize>]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }
    let (max_iterations, time_limit) = iteration_budget(n);
    let mut biases = vec![0u64; n];
    let mut seed_rng = SplitMix64::new(
        (n as u64)
            .wrapping_mul(0x9e3779b97f4a7c15)
            .wrapping_add(0x1234_5678_9abc_def0),
    );
    let mut best_solution = Vec::new();
    let mut best_len = usize::MAX;
    let timer = Instant::now();

    for iter in 0..max_iterations {
        if iter > 0 && timer.elapsed() >= time_limit {
            break;
        }
        let seed = seed_rng.next();
        fill_biases(&mut biases, seed);
        let mut candidate = greedy_with_bias(adj, &biases);
        candidate = prune_until_stable(candidate, adj, n);
        if candidate.len() < best_len {
            best_len = candidate.len();
            best_solution = candidate;
        }
    }

    if best_solution.is_empty() && n > 0 {
        fill_biases(&mut biases, 0);
        let mut candidate = greedy_with_bias(adj, &biases);
        candidate = prune_until_stable(candidate, adj, n);
        best_solution = candidate;
    }

    best_solution
}
