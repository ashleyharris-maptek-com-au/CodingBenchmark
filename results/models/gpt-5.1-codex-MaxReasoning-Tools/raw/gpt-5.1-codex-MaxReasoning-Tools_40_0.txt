#include <bits/stdc++.h>
using namespace std;

using ll = long long;

struct Dinic {
    struct Edge {
        int to;
        long long cap;
        int rev;
    };
    int N;
    vector<vector<Edge>> G;
    vector<int> level, prog;

    Dinic(int n = 0) { init(n); }

    void init(int n) {
        N = n;
        G.assign(n, {});
        level.assign(n, -1);
        prog.assign(n, 0);
    }

    void addEdge(int fr, int to, long long cap) {
        Edge f{to, cap, (int)G[to].size()};
        Edge b{fr, 0, (int)G[fr].size()};
        G[fr].push_back(f);
        G[to].push_back(b);
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto &e : G[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    if (e.to == t) return true;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int &i = prog[v]; i < (int)G[v].size(); ++i) {
            Edge &e = G[v][i];
            if (e.cap <= 0 || level[e.to] != level[v] + 1) continue;
            long long pushed = dfs(e.to, t, min(f, e.cap));
            if (pushed > 0) {
                e.cap -= pushed;
                G[e.to][e.rev].cap += pushed;
                return pushed;
            }
        }
        return 0;
    }

    long long max_flow(int s, int t) {
        long long flow = 0;
        const long long FLOW_INF = numeric_limits<long long>::max();
        while (bfs(s, t)) {
            fill(prog.begin(), prog.end(), 0);
            long long pushed;
            while ((pushed = dfs(s, t, FLOW_INF)) > 0) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<char> minCut(int s) {
        vector<char> vis(N, false);
        stack<int> st;
        st.push(s);
        vis[s] = true;
        while (!st.empty()) {
            int v = st.top();
            st.pop();
            for (const auto &e : G[v]) {
                if (e.cap > 0 && !vis[e.to]) {
                    vis[e.to] = true;
                    st.push(e.to);
                }
            }
        }
        return vis;
    }
};

struct BranchAndBoundSolver {
    int n;
    const vector<ll> &w;
    const vector<vector<int>> &adj;
    vector<char> best, cur;
    vector<int> stamp, neighborStamp;
    int stampCur = 1, neighborCur = 1;
    ll bestWeight = 0;

    BranchAndBoundSolver(const vector<ll> &weights,
                         const vector<vector<int>> &adjacency)
        : n((int)weights.size()),
          w(weights),
          adj(adjacency),
          best(n, 0),
          cur(n, 0),
          stamp(n, 0),
          neighborStamp(n, 0) {}

    vector<char> solve() {
        if (n == 0) return {};
        fill(best.begin(), best.end(), 0);
        fill(cur.begin(), cur.end(), 0);
        fill(stamp.begin(), stamp.end(), 0);
        fill(neighborStamp.begin(), neighborStamp.end(), 0);
        stampCur = neighborCur = 1;
        bestWeight = 0;

        vector<char> greedySol(n, 0);
        ll greedyVal = greedy(greedySol);
        if (greedyVal > bestWeight) {
            bestWeight = greedyVal;
            best = greedySol;
        }

        vector<int> nodes(n);
        iota(nodes.begin(), nodes.end(), 0);
        ll total = 0;
        for (ll val : w) total += val;
        dfs(std::move(nodes), 0, total);
        return best;
    }

    void advanceStamp() {
        ++stampCur;
        if (stampCur == INT_MAX) {
            stampCur = 1;
            fill(stamp.begin(), stamp.end(), 0);
        }
    }

    void advanceNeighborStamp() {
        ++neighborCur;
        if (neighborCur == INT_MAX) {
            neighborCur = 1;
            fill(neighborStamp.begin(), neighborStamp.end(), 0);
        }
    }

    void markNodes(const vector<int> &nodes) {
        advanceStamp();
        for (int u : nodes) stamp[u] = stampCur;
    }

    vector<int> applyIsolates(vector<int> &nodes, ll &curWeight, ll &remSum) {
        vector<int> forced;
        if (nodes.empty()) return forced;
        markNodes(nodes);
        vector<int> filtered;
        filtered.reserve(nodes.size());
        for (int u : nodes) {
            bool hasNeighbor = false;
            for (int v : adj[u]) {
                if (stamp[v] == stampCur) {
                    hasNeighbor = true;
                    break;
                }
            }
            if (!hasNeighbor) {
                forced.push_back(u);
                cur[u] = 1;
                curWeight += w[u];
                remSum -= w[u];
            } else {
                filtered.push_back(u);
            }
        }
        nodes.swap(filtered);
        return forced;
    }

    int choosePivot(const vector<int> &nodes) {
        markNodes(nodes);
        int pivot = nodes[0];
        int bestDeg = -1;
        ll bestW = -1;
        for (int u : nodes) {
            int deg = 0;
            for (int v : adj[u]) {
                if (stamp[v] == stampCur) ++deg;
            }
            if (deg > bestDeg || (deg == bestDeg && w[u] > bestW)) {
                bestDeg = deg;
                bestW = w[u];
                pivot = u;
            }
        }
        return pivot;
    }

    void markPivotNeighbors(int pivot) {
        advanceNeighborStamp();
        for (int v : adj[pivot]) neighborStamp[v] = neighborCur;
    }

    void dfs(vector<int> nodes, ll curWeight, ll remSum) {
        auto forced = applyIsolates(nodes, curWeight, remSum);

        if (curWeight > bestWeight) {
            bestWeight = curWeight;
            best = cur;
        }

        if (nodes.empty()) {
            for (int u : forced) cur[u] = 0;
            return;
        }

        __int128 potential = (__int128)curWeight + (__int128)remSum;
        if (potential <= bestWeight) {
            for (int u : forced) cur[u] = 0;
            return;
        }

        int pivot = choosePivot(nodes);
        markPivotNeighbors(pivot);

        vector<int> nodesInc;
        nodesInc.reserve(nodes.size());
        vector<int> nodesExc;
        nodesExc.reserve(nodes.size() - 1);

        ll remSumExc = remSum - w[pivot];
        ll remSumInc = remSum - w[pivot];

        for (int v : nodes) {
            if (v == pivot) continue;
            nodesExc.push_back(v);
            if (neighborStamp[v] == neighborCur) {
                remSumInc -= w[v];
            } else {
                nodesInc.push_back(v);
            }
        }

        cur[pivot] = 1;
        dfs(std::move(nodesInc), curWeight + w[pivot], remSumInc);
        cur[pivot] = 0;

        dfs(std::move(nodesExc), curWeight, remSumExc);

        for (int u : forced) cur[u] = 0;
    }

    ll greedy(vector<char> &pick) {
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        vector<int> deg(n);
        for (int i = 0; i < n; ++i) deg[i] = (int)adj[i].size();
        auto cmp = [&](int a, int b) {
            __int128 lhs = (__int128)w[a] * (deg[b] + 1);
            __int128 rhs = (__int128)w[b] * (deg[a] + 1);
            if (lhs != rhs) return lhs > rhs;
            if (w[a] != w[b]) return w[a] > w[b];
            return deg[a] < deg[b];
        };
        sort(order.begin(), order.end(), cmp);
        vector<char> forbidden(n, 0);
        ll value = 0;
        fill(pick.begin(), pick.end(), 0);
        for (int u : order) {
            if (forbidden[u]) continue;
            pick[u] = 1;
            value += w[u];
            forbidden[u] = 1;
            for (int v : adj[u]) forbidden[v] = 1;
        }
        return value;
    }
};

vector<char> solveBipartite(const vector<ll> &weight,
                            const vector<vector<int>> &adj,
                            const vector<int> &color) {
    int k = (int)weight.size();
    int leftCount = 0;
    for (int c : color) if (c == 0) ++leftCount;
    int rightCount = k - leftCount;
    int source = 0;
    int sink = 1 + leftCount + rightCount;

    Dinic dinic(sink + 1);
    vector<int> id(k, -1);
    int nextId = 1;
    for (int i = 0; i < k; ++i) {
        if (color[i] == 0) id[i] = nextId++;
    }
    for (int i = 0; i < k; ++i) {
        if (color[i] == 1) id[i] = nextId++;
    }

    long long totalWeight = 0;
    for (ll w : weight) totalWeight += w;
    __int128 inf128 = (__int128)totalWeight + 1;
    long long INF = (inf128 > numeric_limits<long long>::max())
                        ? numeric_limits<long long>::max()
                        : (long long)inf128;
    if (INF <= totalWeight) INF = numeric_limits<long long>::max();

    for (int i = 0; i < k; ++i) {
        if (color[i] == 0) {
            dinic.addEdge(source, id[i], weight[i]);
        } else {
            dinic.addEdge(id[i], sink, weight[i]);
        }
    }
    for (int u = 0; u < k; ++u) {
        if (color[u] != 0) continue;
        for (int v : adj[u]) {
            if (color[v] == 1) dinic.addEdge(id[u], id[v], INF);
        }
    }

    dinic.max_flow(source, sink);
    vector<char> reachable = dinic.minCut(source);

    vector<char> take(k, 0);
    for (int i = 0; i < k; ++i) {
        if (color[i] == 0) {
            if (reachable[id[i]]) take[i] = 1;
        } else {
            if (!reachable[id[i]]) take[i] = 1;
        }
    }
    return take;
}

vector<char> solveComponent(const vector<int> &comp,
                            const vector<vector<int>> &g,
                            const vector<ll> &weights,
                            vector<int> &indexMap) {
    int k = (int)comp.size();
    for (int i = 0; i < k; ++i) indexMap[comp[i]] = i;

    vector<ll> localWeight(k);
    vector<vector<int>> adj(k);
    for (int i = 0; i < k; ++i) {
        int u = comp[i];
        localWeight[i] = weights[u];
        for (int v : g[u]) {
            int idx = (v >= 0 && v < (int)indexMap.size()) ? indexMap[v] : -1;
            if (idx == -1 || idx == i) continue;
            adj[i].push_back(idx);
        }
    }
    for (auto &nbrs : adj) {
        sort(nbrs.begin(), nbrs.end());
        nbrs.erase(unique(nbrs.begin(), nbrs.end()), nbrs.end());
    }
    for (int node : comp) indexMap[node] = -1;

    vector<int> color(k, -1);
    bool isBip = true;
    queue<int> q;
    for (int i = 0; i < k && isBip; ++i) {
        if (color[i] != -1) continue;
        color[i] = 0;
        q.push(i);
        while (!q.empty() && isBip) {
            int u = q.front();
            q.pop();
            for (int v : adj[u]) {
                if (color[v] == -1) {
                    color[v] = color[u] ^ 1;
                    q.push(v);
                } else if (color[v] == color[u]) {
                    isBip = false;
                    break;
                }
            }
        }
    }

    if (isBip) {
        return solveBipartite(localWeight, adj, color);
    } else {
        BranchAndBoundSolver solver(localWeight, adj);
        return solver.solve();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<ll> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<vector<int>> g(n);
    vector<char> forcedZero(n, 0);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            forcedZero[u] = 1;
            continue;
        }
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vector<int> assignment(n, 0);
    vector<char> active(n, 0);
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0 && !forcedZero[i]) active[i] = 1;
    }

    vector<char> visited(n, 0);
    vector<int> indexMap(n, -1);

    for (int start = 0; start < n; ++start) {
        if (!active[start] || visited[start]) continue;
        vector<int> comp;
        queue<int> q;
        q.push(start);
        visited[start] = 1;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            comp.push_back(u);
            for (int v : g[u]) {
                if (!active[v] || visited[v]) continue;
                visited[v] = 1;
                q.push(v);
            }
        }

        vector<char> local = solveComponent(comp, g, c, indexMap);
        for (int i = 0; i < (int)comp.size(); ++i) {
            assignment[comp[i]] = local[i];
        }
    }

    ll objective = 0;
    for (int i = 0; i < n; ++i) {
        if (assignment[i]) objective += c[i];
    }

    cout << objective << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << assignment[i];
    }
    cout << "\n";
    return 0;
}
