#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <optional>
#include <string>
#include <unordered_map>
#include <vector>

struct RollingHash {
    static constexpr uint64_t BASE1 = 1315423911ull;
    static constexpr uint64_t BASE2 = 11400714819323198485ull;

    struct HashPair {
        uint64_t h1;
        uint64_t h2;
        bool operator==(const HashPair& other) const noexcept {
            return h1 == other.h1 && h2 == other.h2;
        }
    };

    std::vector<uint64_t> pref1, pref2, pow1, pow2;

    void build(const std::string& s) {
        const size_t n = s.size();
        pref1.assign(n + 1, 0);
        pref2.assign(n + 1, 0);
        pow1.assign(n + 1, 1);
        pow2.assign(n + 1, 1);
        for (size_t i = 0; i < n; ++i) {
            uint64_t val = static_cast<unsigned char>(s[i]) + 1;
            pref1[i + 1] = pref1[i] * BASE1 + val;
            pref2[i + 1] = pref2[i] * BASE2 + val;
            pow1[i + 1] = pow1[i] * BASE1;
            pow2[i + 1] = pow2[i] * BASE2;
        }
    }

    HashPair slice(size_t l, size_t r) const {
        HashPair res;
        res.h1 = pref1[r] - pref1[l] * pow1[r - l];
        res.h2 = pref2[r] - pref2[l] * pow2[r - l];
        return res;
    }
};

struct HashPairHasher {
    size_t operator()(const RollingHash::HashPair& p) const noexcept {
        return static_cast<size_t>(p.h1 ^ (p.h2 << 1));
    }
};

struct CandidateInfo {
    size_t pos;
    int mark;
};

std::optional<size_t> hash_check(
    int len,
    const std::vector<std::string>& strings,
    int refIdx,
    const std::vector<RollingHash>& hashes
) {
    if (len == 0) {
        return size_t{0};
    }
    const std::string& ref = strings[refIdx];
    if (len > static_cast<int>(ref.size())) {
        return std::nullopt;
    }
    const size_t L = static_cast<size_t>(len);
    std::unordered_map<RollingHash::HashPair, CandidateInfo, HashPairHasher> candidates;
    const size_t total = ref.size() - L + 1;
    candidates.reserve(total * 2 + 1);

    for (size_t i = 0; i + L <= ref.size(); ++i) {
        auto h = hashes[refIdx].slice(i, i + L);
        auto [it, inserted] = candidates.emplace(h, CandidateInfo{i, 0});
        if (!inserted && i < it->second.pos) {
            it->second.pos = i;
        }
    }

    int marker = 0;
    for (size_t idx = 0; idx < strings.size(); ++idx) {
        if (static_cast<int>(idx) == refIdx) {
            continue;
        }
        const std::string& s = strings[idx];
        if (s.size() < L) {
            return std::nullopt;
        }
        ++marker;
        for (size_t i = 0; i + L <= s.size(); ++i) {
            auto h = hashes[idx].slice(i, i + L);
            auto it = candidates.find(h);
            if (it != candidates.end()) {
                it->second.mark = marker;
            }
        }
        for (auto it = candidates.begin(); it != candidates.end();) {
            if (it->second.mark != marker) {
                it = candidates.erase(it);
            } else {
                ++it;
            }
        }
        if (candidates.empty()) {
            return std::nullopt;
        }
    }

    if (!candidates.empty()) {
        return candidates.begin()->second.pos;
    }
    return std::nullopt;
}

std::string solve_hash(const std::vector<std::string>& strings, int refIdx) {
    const std::string& ref = strings[refIdx];
    std::vector<RollingHash> hashes(strings.size());
    for (size_t i = 0; i < strings.size(); ++i) {
        hashes[i].build(strings[i]);
    }

    int low = 0;
    int high = static_cast<int>(ref.size());
    int bestLen = 0;
    size_t bestPos = 0;
    while (low <= high) {
        int mid = low + (high - low) / 2;
        auto pos = hash_check(mid, strings, refIdx, hashes);
        if (pos.has_value()) {
            bestLen = mid;
            bestPos = pos.value();
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    return ref.substr(bestPos, static_cast<size_t>(bestLen));
}

struct SuffixAutomaton {
    struct Node {
        int link;
        int len;
        int first_pos;
        int head;
    };
    struct Edge {
        int to;
        unsigned char ch;
        int next;
    };

    std::vector<Node> st;
    std::vector<Edge> edges;
    int last;

    explicit SuffixAutomaton(size_t reserve) {
        st.reserve(reserve * 2 + 1);
        edges.reserve(reserve * 3 + 1);
        st.push_back(Node{-1, 0, -1, -1});
        last = 0;
    }

    void extend(unsigned char c, int pos) {
        int cur = new_state();
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;
        int p = last;
        while (p != -1 && get_transition(p, c) == -1) {
            add_transition(p, c, cur);
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = get_transition(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = new_state();
                st[clone].len = st[p].len + 1;
                st[clone].first_pos = st[q].first_pos;
                st[clone].link = st[q].link;
                copy_transitions(q, clone);
                while (p != -1 && get_transition(p, c) == q) {
                    set_transition(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = clone;
                st[cur].link = clone;
            }
        }
        last = cur;
    }

    int next_state(int state, unsigned char c) const {
        return get_transition(state, c);
    }

private:
    int new_state() {
        st.push_back(Node{-1, 0, -1, -1});
        return static_cast<int>(st.size()) - 1;
    }

    inline int get_transition(int state, unsigned char c) const {
        for (int e = st[state].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                return edges[e].to;
            }
        }
        return -1;
    }

    inline void add_transition(int state, unsigned char c, int dest) {
        edges.push_back(Edge{dest, c, st[state].head});
        st[state].head = static_cast<int>(edges.size()) - 1;
    }

    inline void set_transition(int state, unsigned char c, int dest) {
        for (int e = st[state].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = dest;
                return;
            }
        }
        add_transition(state, c, dest);
    }

    inline void copy_transitions(int src, int dst) {
        st[dst].head = -1;
        for (int e = st[src].head; e != -1; e = edges[e].next) {
            edges.push_back(Edge{edges[e].to, edges[e].ch, st[dst].head});
            st[dst].head = static_cast<int>(edges.size()) - 1;
        }
    }
};

std::string solve_sam(const std::vector<std::string>& strings, int refIdx) {
    const std::string& reference = strings[refIdx];
    if (reference.empty()) {
        return "";
    }

    SuffixAutomaton sam(reference.size());
    for (size_t i = 0; i < reference.size(); ++i) {
        sam.extend(static_cast<unsigned char>(reference[i]), static_cast<int>(i));
    }

    const size_t stateCount = sam.st.size();
    std::vector<int> best(stateCount);
    int maxLen = 0;
    for (size_t i = 0; i < stateCount; ++i) {
        best[i] = sam.st[i].len;
        if (sam.st[i].len > maxLen) {
            maxLen = sam.st[i].len;
        }
    }

    std::vector<int> cnt(maxLen + 1, 0);
    for (size_t i = 0; i < stateCount; ++i) {
        cnt[sam.st[i].len]++;
    }
    for (int i = 1; i <= maxLen; ++i) {
        cnt[i] += cnt[i - 1];
    }
    std::vector<int> order(stateCount);
    for (int i = static_cast<int>(stateCount) - 1; i >= 0; --i) {
        int len = sam.st[i].len;
        order[--cnt[len]] = i;
    }

    std::vector<int> tmp(stateCount, 0);
    for (size_t idx = 0; idx < strings.size(); ++idx) {
        if (static_cast<int>(idx) == refIdx) {
            continue;
        }
        const std::string& s = strings[idx];
        std::fill(tmp.begin(), tmp.end(), 0);
        int v = 0;
        int l = 0;
        for (char raw : s) {
            unsigned char c = static_cast<unsigned char>(raw);
            int nxt = sam.next_state(v, c);
            while (v && nxt == -1) {
                v = sam.st[v].link;
                l = sam.st[v].len;
                nxt = sam.next_state(v, c);
            }
            if (nxt != -1) {
                v = nxt;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (tmp[v] < l) {
                tmp[v] = l;
            }
        }
        for (size_t oi = stateCount; oi-- > 0;) {
            int state = order[oi];
            int link = sam.st[state].link;
            if (link >= 0) {
                int val = tmp[state];
                if (val > sam.st[link].len) {
                    val = sam.st[link].len;
                }
                if (val > tmp[link]) {
                    tmp[link] = val;
                }
            }
        }
        int postMax = 0;
        for (size_t i = 0; i < stateCount; ++i) {
            if (tmp[i] < best[i]) {
                best[i] = tmp[i];
            }
            if (best[i] > postMax) {
                postMax = best[i];
            }
        }
        if (postMax == 0) {
            break;
        }
    }

    int answerLen = 0;
    int answerState = 0;
    for (size_t i = 0; i < stateCount; ++i) {
        if (best[i] > answerLen) {
            answerLen = best[i];
            answerState = static_cast<int>(i);
        } else if (best[i] == answerLen && answerLen > 0) {
            int currentStart = sam.st[i].first_pos - answerLen + 1;
            int bestStart = sam.st[answerState].first_pos - answerLen + 1;
            if (currentStart < bestStart) {
                answerState = static_cast<int>(i);
            }
        }
    }

    if (answerLen <= 0) {
        return "";
    }
    int endPos = sam.st[answerState].first_pos;
    int startPos = endPos - answerLen + 1;
    if (startPos < 0) {
        startPos = 0;
    }
    return reference.substr(static_cast<size_t>(startPos), static_cast<size_t>(answerLen));
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n)) {
        std::cout << "\n";
        return 0;
    }
    std::string dummy;
    std::getline(std::cin, dummy);  // consume trailing newline

    std::vector<std::string> strings;
    strings.reserve(n);
    for (int i = 0; i < n; ++i) {
        std::string line;
        if (!std::getline(std::cin, line)) {
            line.clear();
        }
        if (!line.empty() && line.back() == '\r') {
            line.pop_back();
        }
        strings.push_back(std::move(line));
    }

    if (n == 0) {
        std::cout << "\n";
        return 0;
    }
    if (n == 1) {
        std::cout << strings[0] << "\n";
        return 0;
    }

    uint64_t totalLen = 0;
    size_t maxLen = 0;
    size_t minLen = std::numeric_limits<size_t>::max();
    int refIdx = 0;
    bool hasEmpty = false;

    for (int i = 0; i < n; ++i) {
        size_t len = strings[i].size();
        totalLen += len;
        if (len > maxLen) {
            maxLen = len;
        }
        if (len < minLen) {
            minLen = len;
            refIdx = i;
        }
        if (len == 0) {
            hasEmpty = true;
            break;
        }
    }

    if (hasEmpty) {
        std::cout << "\n";
        return 0;
    }

    bool useHashStrategy = (totalLen <= 200000ull && maxLen <= 50000ull && n <= 10);

    // Adaptive strategy selection:
    // - Small inputs: binary search + rolling hash set intersections.
    // - Large inputs: suffix automaton over the shortest string with streaming passes.
    std::string answer = useHashStrategy ? solve_hash(strings, refIdx)
                                         : solve_sam(strings, refIdx);

    std::cout << answer << "\n";
    return 0;
}
