```ts
declare const require: any;
declare const process: any;

const fs = require('fs');

type JobIdentifier = string | number | null;

interface ParsedRobot {
    id: number;
    x: number;
    y: number;
}

interface Job {
    id: JobIdentifier;
    x: number;
    y: number;
    priority: number;
    index: number;
    cellIndex: number;
    nearestRobotEstimate: number;
}

interface RobotState {
    id: number;
    x: number;
    y: number;
    cellIndex: number;
    assignments: JobIdentifier[];
    travel: number;
}

function main(): void {
    const input: string = fs.readFileSync(0, 'utf8');
    const data = input.length ? JSON.parse(input) : {};

    const rawRobots: any[] = Array.isArray(data.robots) ? data.robots : [];
    const rawJobs: any[] = Array.isArray(data.jobs) ? data.jobs : [];
    const rawGrid: any[] = Array.isArray(data.grid) ? data.grid : [];

    const grid: string[] = rawGrid.map((row) => (typeof row === 'string' ? row : ''));
    const height: number = grid.length;
    const width: number = grid.reduce((max, row) => Math.max(max, row.length), 0);

    if (height === 0 || width === 0) {
        outputRoundRobinAssignments(rawRobots, rawJobs);
        return;
    }

    const robots: ParsedRobot[] = normalizeRobots(rawRobots, width, height);
    if (robots.length === 0) {
        outputRoundRobinAssignments(rawRobots, rawJobs);
        return;
    }

    const maxRobotId: number = robots.reduce((max, robot) => Math.max(max, robot.id), -1);
    if (maxRobotId < 0) {
        outputRoundRobinAssignments(rawRobots, rawJobs);
        return;
    }

    const assignments: JobIdentifier[][] = [];
    for (let i = 0; i <= maxRobotId; i++) {
        assignments[i] = [];
    }

    const jobs: Job[] = normalizeJobs(rawJobs, width, height);
    const jobCount = jobs.length;
    if (jobCount === 0) {
        process.stdout.write(JSON.stringify({ assignments }));
        return;
    }

    const uniqueRobots: ParsedRobot[] = deduplicateRobots(robots);
    if (uniqueRobots.length === 0) {
        outputRoundRobinAssignments(rawRobots, rawJobs);
        return;
    }

    const robotStates: RobotState[] = uniqueRobots.map((robot) => ({
        id: robot.id,
        x: robot.x,
        y: robot.y,
        cellIndex: robot.y * width + robot.x,
        assignments: assignments[robot.id],
        travel: 0,
    }));

    const totalCells = width * height;
    const openMask = buildOpenMask(grid, width, height);
    const jobIndexAtCell = buildJobIndexLookup(totalCells, jobs);
    const visitMarks = new Uint32Array(totalCells);
    const distCells = new Int32Array(totalCells);
    const bfsQueue = new Int32Array(totalCells);
    const distanceCache = new Map<number, Int32Array>();
    let visitToken = 1;
    const unreachablePenalty = totalCells + jobCount + robotStates.length + 1000;

    computeNearestRobotEstimates(jobs, robotStates);

    const jobOrder = jobs.map((job) => job.index);
    jobOrder.sort((aIdx, bIdx) => compareJobs(jobs[aIdx], jobs[bIdx]));

    function nextVisitToken(): number {
        visitToken++;
        if (visitToken === 0) {
            visitMarks.fill(0);
            visitToken = 1;
        }
        return visitToken;
    }

    function computeDistanceArray(sourceIdx: number): Int32Array {
        const distances = new Int32Array(jobCount);
        distances.fill(-1);
        if (jobCount === 0) {
            return distances;
        }

        const token = nextVisitToken();
        let head = 0;
        let tail = 0;

        bfsQueue[tail++] = sourceIdx;
        visitMarks[sourceIdx] = token;
        distCells[sourceIdx] = 0;

        let found = 0;

        while (head < tail && found < jobCount) {
            const idx = bfsQueue[head++];
            const d = distCells[idx];

            const jobIndices = jobIndexAtCell[idx];
            if (jobIndices) {
                for (let i = 0; i < jobIndices.length; i++) {
                    const jobIdx = jobIndices[i];
                    if (distances[jobIdx] === -1) {
                        distances[jobIdx] = d;
                        found++;
                    }
                }
            }

            const x = idx % width;
            const y = (idx - x) / width;
            const nextDist = d + 1;

            if (x > 0) {
                const leftIdx = idx - 1;
                if (openMask[leftIdx] === 1 && visitMarks[leftIdx] !== token) {
                    visitMarks[leftIdx] = token;
                    distCells[leftIdx] = nextDist;
                    bfsQueue[tail++] = leftIdx;
                }
            }
            if (x + 1 < width) {
                const rightIdx = idx + 1;
                if (openMask[rightIdx] === 1 && visitMarks[rightIdx] !== token) {
                    visitMarks[rightIdx] = token;
                    distCells[rightIdx] = nextDist;
                    bfsQueue[tail++] = rightIdx;
                }
            }
            if (y > 0) {
                const upIdx = idx - width;
                if (openMask[upIdx] === 1 && visitMarks[upIdx] !== token) {
                    visitMarks[upIdx] = token;
                    distCells[upIdx] = nextDist;
                    bfsQueue[tail++] = upIdx;
                }
            }
            if (y + 1 < height) {
                const downIdx = idx + width;
                if (openMask[downIdx] === 1 && visitMarks[downIdx] !== token) {
                    visitMarks[downIdx] = token;
                    distCells[downIdx] = nextDist;
                    bfsQueue[tail++] = downIdx;
                }
            }
        }

        return distances;
    }

    function getDistanceFrom(sourceIdx: number, jobIdx: number): number {
        let distances = distanceCache.get(sourceIdx);
        if (!distances) {
            distances = computeDistanceArray(sourceIdx);
            distanceCache.set(sourceIdx, distances);
        }
        const value = distances[jobIdx];
        return value >= 0 ? value : Number.POSITIVE_INFINITY;
    }

    function assignJob(job: Job): void {
        let bestRobot: RobotState | null = null;
        let bestCost = Number.POSITIVE_INFINITY;
        let bestActualDist = Number.POSITIVE_INFINITY;
        let bestApprox = Number.POSITIVE_INFINITY;

        for (const robot of robotStates) {
            const approx = manhattan(robot.x, robot.y, job.x, job.y);
            const actualDist = getDistanceFrom(robot.cellIndex, job.index);
            const effective = Number.isFinite(actualDist) ? actualDist : approx + unreachablePenalty;
            if (effective < bestCost) {
                bestCost = effective;
                bestRobot = robot;
                bestActualDist = actualDist;
                bestApprox = approx;
            } else if (effective === bestCost && bestRobot) {
                const loadDiff = robot.assignments.length - bestRobot.assignments.length;
                if (loadDiff < 0) {
                    bestRobot = robot;
                    bestActualDist = actualDist;
                    bestApprox = approx;
                } else if (loadDiff === 0) {
                    if (robot.travel < bestRobot.travel) {
                        bestRobot = robot;
                        bestActualDist = actualDist;
                        bestApprox = approx;
                    } else if (robot.travel === bestRobot.travel && robot.id < bestRobot.id) {
                        bestRobot = robot;
                        bestActualDist = actualDist;
                        bestApprox = approx;
                    }
                }
            }
        }

        if (!bestRobot) {
            return;
        }

        const distanceContribution = Number.isFinite(bestActualDist) ? bestActualDist : bestApprox;
        bestRobot.travel += distanceContribution;
        bestRobot.assignments.push(job.id);
        bestRobot.x = job.x;
        bestRobot.y = job.y;
        bestRobot.cellIndex = job.cellIndex;
    }

    for (const jobIdx of jobOrder) {
        const job = jobs[jobIdx];
        if (job) {
            assignJob(job);
        }
    }

    process.stdout.write(JSON.stringify({ assignments }));
}

main();

function normalizeRobots(list: any[], width: number, height: number): ParsedRobot[] {
    const robots: ParsedRobot[] = [];
    for (const entry of list) {
        if (entry == null || typeof entry !== 'object') {
            continue;
        }
        const id = sanitizeRobotId((entry as any).id);
        if (id < 0) {
            continue;
        }
        const x = sanitizeCoord((entry as any).x, width);
        const y = sanitizeCoord((entry as any).y, height);
        robots.push({ id, x, y });
    }
    return robots;
}

function normalizeJobs(list: any[], width: number, height: number): Job[] {
    const jobs: Job[] = [];
    for (let idx = 0; idx < list.length; idx++) {
        const entry = list[idx];
        const record: any = entry != null && typeof entry === 'object' ? entry : {};
        const x = sanitizeCoord(record.x, width);
        const y = sanitizeCoord(record.y, height);
        const priority = toNumber(record.priority, 0);
        const rawId = record.id;
        const idValue: JobIdentifier =
            typeof rawId === 'string' || typeof rawId === 'number' ? rawId : null;
        jobs.push({
            id: idValue,
            x,
            y,
            priority,
            index: idx,
            cellIndex: y * width + x,
            nearestRobotEstimate: 0,
        });
    }
    return jobs;
}

function deduplicateRobots(robots: ParsedRobot[]): ParsedRobot[] {
    const seen = new Set<number>();
    const unique: ParsedRobot[] = [];
    for (const robot of robots) {
        if (!seen.has(robot.id)) {
            seen.add(robot.id);
            unique.push(robot);
        }
    }
    unique.sort((a, b) => a.id - b.id);
    return unique;
}

function buildOpenMask(grid: string[], width: number, height: number): Uint8Array {
    const mask = new Uint8Array(width * height);
    for (let y = 0; y < height; y++) {
        const row = grid[y];
        for (let x = 0; x < width; x++) {
            const ch = x < row.length ? row[x] : '#';
            mask[y * width + x] = ch === '.' ? 1 : 0;
        }
    }
    return mask;
}

function buildJobIndexLookup(totalCells: number, jobList: Job[]): (number[] | null)[] {
    const lookup: (number[] | null)[] = new Array(totalCells);
    for (let i = 0; i < totalCells; i++) {
        lookup[i] = null;
    }
    for (const job of jobList) {
        const idx = job.cellIndex;
        if (idx < 0 || idx >= totalCells) {
            continue;
        }
        let bucket = lookup[idx];
        if (bucket === null) {
            bucket = [];
            lookup[idx] = bucket;
        }
        bucket.push(job.index);
    }
    return lookup;
}

function computeNearestRobotEstimates(jobList: Job[], robotStates: RobotState[]): void {
    for (const job of jobList) {
        let best = Number.POSITIVE_INFINITY;
        for (const robot of robotStates) {
            const dist = manhattan(robot.x, robot.y, job.x, job.y);
            if (dist < best) {
                best = dist;
            }
        }
        job.nearestRobotEstimate = Number.isFinite(best) ? best : Number.POSITIVE_INFINITY;
    }
}

function compareJobs(a: Job, b: Job): number {
    if (a.priority !== b.priority) {
        return b.priority - a.priority;
    }
    if (a.nearestRobotEstimate !== b.nearestRobotEstimate) {
        return a.nearestRobotEstimate - b.nearestRobotEstimate;
    }
    return a.index - b.index;
}

function sanitizeCoord(value: any, upperBound: number): number {
    if (upperBound <= 0) {
        return 0;
    }
    const num = Number(value);
    if (!Number.isFinite(num)) {
        return 0;
    }
    const integer = Math.trunc(num);
    if (integer < 0) {
        return 0;
    }
    if (integer >= upperBound) {
        return upperBound - 1;
    }
    return integer;
}

function sanitizeRobotId(value: any): number {
    const num = Number(value);
    if (!Number.isFinite(num)) {
        return -1;
    }
    const id = Math.trunc(num);
    return id >= 0 ? id : -1;
}

function toNumber(value: any, fallback = 0): number {
    const num = Number(value);
    return Number.isFinite(num) ? num : fallback;
}

function manhattan(ax: number, ay: number, bx: number, by: number): number {
    return Math.abs(ax - bx) + Math.abs(ay - by);
}

function outputRoundRobinAssignments(rawRobots: any[], rawJobs: any[]): void {
    const robotIds: number[] = [];
    const seen = new Set<number>();
    for (const entry of rawRobots) {
        if (entry == null || typeof entry !== 'object') {
            continue;
        }
        const id = sanitizeRobotId((entry as any).id);
        if (id >= 0 && !seen.has(id)) {
            seen.add(id);
            robotIds.push(id);
        }
    }
    robotIds.sort((a, b) => a - b);

    if (robotIds.length === 0) {
        const assignments: JobIdentifier[][] = [];
        process.stdout.write(JSON.stringify({ assignments }));
        return;
    }

    const maxId = robotIds[robotIds.length - 1];
    const assignments: JobIdentifier[][] = [];
    for (let i = 0; i <= maxId; i++) {
        assignments[i] = [];
    }

    const jobEntries = rawJobs.map((entry, index) => {
        const record: any = entry != null && typeof entry === 'object' ? entry : {};
        const priority = toNumber(record.priority, 0);
        const rawId = record.id;
        const idValue: JobIdentifier =
            typeof rawId === 'string' || typeof rawId === 'number' ? rawId : null;
        return { id: idValue, priority, index };
    });

    jobEntries.sort((a, b) => {
        if (a.priority !== b.priority) {
            return b.priority - a.priority;
        }
        return a.index - b.index;
    });

    let pointer = 0;
    for (const job of jobEntries) {
        const robotId = robotIds[pointer % robotIds.length];
        assignments[robotId].push(job.id);
        pointer++;
    }

    process.stdout.write(JSON.stringify({ assignments }));
}
```
