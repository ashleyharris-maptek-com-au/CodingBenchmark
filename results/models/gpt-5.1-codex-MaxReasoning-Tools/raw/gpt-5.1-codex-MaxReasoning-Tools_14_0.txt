Here’s a complete Minesweeper solver that meets the requirements. It builds all constraints once, then applies a tiered reasoning pipeline: constant-time propagation, subset/edge-pattern inference, exact component enumeration for tiny frontiers, bounded DFS probing for medium ones, and selective probing for very large regions. The solver only reports cells that are provably safe.

```csharp
using System;
using System.Collections.Generic;

enum CellState : byte
{
    Unknown = 0,
    Safe = 1,
    Mine = 2
}

class FrontierCell
{
    public int X;
    public int Y;
    public CellState State;
}

class Constraint
{
    public int MinesLeft;
    public List<int> Variables = new List<int>();
}

class ComponentData
{
    public int[] GlobalVarIndices = Array.Empty<int>();
    public int[][] ConstraintVariables = Array.Empty<int[]>();
    public int[] ConstraintMinesLeft = Array.Empty<int>();
    public int[] ConstraintUnknownCounts = Array.Empty<int>();
    public int[][] VarToConstraints = Array.Empty<int[]>();
    public int VarCount => GlobalVarIndices.Length;
    public int ConstraintCount => ConstraintVariables.Length;
}

enum SearchOutcome
{
    Satisfiable,
    Unsatisfiable,
    Inconclusive
}

class ComponentEnumerationResult
{
    public bool HasSolution { get; }
    public bool[] ForcedSafe { get; }
    public bool[] ForcedMine { get; }

    public ComponentEnumerationResult(bool hasSolution, bool[] forcedSafe, bool[] forcedMine)
    {
        HasSolution = hasSolution;
        ForcedSafe = forcedSafe;
        ForcedMine = forcedMine;
    }
}

struct Change
{
    public int ConstraintIdx;
    public int PrevMinesLeft;
    public int PrevUnknown;
}

class ConstraintSearchEngine
{
    private readonly ComponentData data;
    private readonly int[] workingMines;
    private readonly int[] workingUnknown;
    private readonly int[] assignment;
    private readonly List<Change> changeStack = new List<Change>();
    private readonly List<int> assignmentStack = new List<int>();
    private readonly Stack<int> propagationStack = new Stack<int>();
    private int assignedCount;
    private int nodeLimit;
    private int nodesVisited;
    private int solutionsFound;

    public ConstraintSearchEngine(ComponentData data)
    {
        this.data = data;
        workingMines = new int[data.ConstraintCount];
        workingUnknown = new int[data.ConstraintCount];
        assignment = new int[data.VarCount];
        ResetWorkingState();
    }

    public ComponentEnumerationResult EnumerateForced(int solutionLimit)
    {
        ResetWorkingState();
        solutionsFound = 0;
        bool[] zeroPossible = new bool[data.VarCount];
        bool[] onePossible = new bool[data.VarCount];
        EnumerateRecursive(solutionLimit, zeroPossible, onePossible);
        bool hasSolution = solutionsFound > 0;
        bool[] forcedSafe = new bool[data.VarCount];
        bool[] forcedMine = new bool[data.VarCount];
        if (hasSolution)
        {
            for (int i = 0; i < data.VarCount; i++)
            {
                forcedSafe[i] = zeroPossible[i] && !onePossible[i];
                forcedMine[i] = !zeroPossible[i] && onePossible[i];
            }
        }
        return new ComponentEnumerationResult(hasSolution, forcedSafe, forcedMine);
    }

    public SearchOutcome CheckAssumption(int localVar, int value, int limit)
    {
        ResetWorkingState();
        nodeLimit = limit <= 0 ? int.MaxValue : limit;
        nodesVisited = 0;
        if (!TryApplyWithPropagation(localVar, value))
        {
            return SearchOutcome.Unsatisfiable;
        }
        return Search();
    }

    private void EnumerateRecursive(int solutionLimit, bool[] zeroPossible, bool[] onePossible)
    {
        if (solutionsFound >= solutionLimit) return;
        if (assignedCount == data.VarCount)
        {
            solutionsFound++;
            for (int i = 0; i < data.VarCount; i++)
            {
                if (assignment[i] == 0) zeroPossible[i] = true;
                else if (assignment[i] == 1) onePossible[i] = true;
            }
            return;
        }

        int var = SelectVariable();
        if (var == -1) return;

        for (int value = 0; value <= 1; value++)
        {
            int changeStart = changeStack.Count;
            int assignmentStart = assignmentStack.Count;
            if (TryApplyWithPropagation(var, value))
            {
                EnumerateRecursive(solutionLimit, zeroPossible, onePossible);
            }
            Rollback(changeStart, assignmentStart);
            if (solutionsFound >= solutionLimit) return;
        }
    }

    private SearchOutcome Search()
    {
        if (assignedCount == data.VarCount)
        {
            for (int i = 0; i < data.ConstraintCount; i++)
            {
                if (workingMines[i] != 0)
                {
                    return SearchOutcome.Unsatisfiable;
                }
            }
            return SearchOutcome.Satisfiable;
        }

        if (nodesVisited >= nodeLimit)
        {
            return SearchOutcome.Inconclusive;
        }
        nodesVisited++;

        int var = SelectVariable();
        if (var == -1)
        {
            for (int i = 0; i < data.ConstraintCount; i++)
            {
                if (workingMines[i] != 0)
                {
                    return SearchOutcome.Unsatisfiable;
                }
            }
            return SearchOutcome.Satisfiable;
        }

        for (int value = 0; value <= 1; value++)
        {
            int changeStart = changeStack.Count;
            int assignmentStart = assignmentStack.Count;
            if (TryApplyWithPropagation(var, value))
            {
                var result = Search();
                if (result != SearchOutcome.Unsatisfiable)
                {
                    Rollback(changeStart, assignmentStart);
                    return result;
                }
            }
            Rollback(changeStart, assignmentStart);
        }

        return SearchOutcome.Unsatisfiable;
    }

    private int SelectVariable()
    {
        int bestVar = -1;
        int bestScore = int.MaxValue;
        for (int i = 0; i < data.VarCount; i++)
        {
            if (assignment[i] != -1) continue;
            int score = 0;
            var constraints = data.VarToConstraints[i];
            for (int j = 0; j < constraints.Length; j++)
            {
                int idx = constraints[j];
                int unknown = workingUnknown[idx];
                if (unknown == 0) continue;
                score += unknown;
            }
            if (score < bestScore)
            {
                bestScore = score;
                bestVar = i;
            }
        }

        if (bestVar == -1)
        {
            for (int i = 0; i < data.VarCount; i++)
            {
                if (assignment[i] == -1) return i;
            }
        }

        return bestVar;
    }

    private bool TryApplyWithPropagation(int var, int value)
    {
        propagationStack.Clear();
        if (!InternalAssign(var, value))
        {
            return false;
        }

        while (propagationStack.Count > 0)
        {
            int current = propagationStack.Pop();
            int assignedValue = assignment[current];
            var constraintIndices = data.VarToConstraints[current];

            for (int i = 0; i < constraintIndices.Length; i++)
            {
                int constraintIdx = constraintIndices[i];
                changeStack.Add(new Change
                {
                    ConstraintIdx = constraintIdx,
                    PrevMinesLeft = workingMines[constraintIdx],
                    PrevUnknown = workingUnknown[constraintIdx]
                });

                workingUnknown[constraintIdx]--;
                if (assignedValue == 1)
                {
                    workingMines[constraintIdx]--;
                }

                if (workingMines[constraintIdx] < 0 || workingMines[constraintIdx] > workingUnknown[constraintIdx])
                {
                    propagationStack.Clear();
                    return false;
                }

                if (workingMines[constraintIdx] == 0 || workingMines[constraintIdx] == workingUnknown[constraintIdx])
                {
                    int forcedValue = workingMines[constraintIdx] == 0 ? 0 : 1;
                    var vars = data.ConstraintVariables[constraintIdx];
                    for (int k = 0; k < vars.Length; k++)
                    {
                        int neighborVar = vars[k];
                        if (assignment[neighborVar] == -1)
                        {
                            if (!InternalAssign(neighborVar, forcedValue))
                            {
                                propagationStack.Clear();
                                return false;
                            }
                        }
                        else if (assignment[neighborVar] != forcedValue)
                        {
                            propagationStack.Clear();
                            return false;
                        }
                    }
                }
            }
        }

        return true;
    }

    private bool InternalAssign(int var, int value)
    {
        int existing = assignment[var];
        if (existing != -1)
        {
            return existing == value;
        }

        assignment[var] = value;
        assignedCount++;
        assignmentStack.Add(var);
        propagationStack.Push(var);
        return true;
    }

    private void Rollback(int changeStart, int assignmentStart)
    {
        for (int i = changeStack.Count - 1; i >= changeStart; i--)
        {
            var change = changeStack[i];
            workingMines[change.ConstraintIdx] = change.PrevMinesLeft;
            workingUnknown[change.ConstraintIdx] = change.PrevUnknown;
        }
        if (changeStack.Count > changeStart)
        {
            changeStack.RemoveRange(changeStart, changeStack.Count - changeStart);
        }

        for (int i = assignmentStack.Count - 1; i >= assignmentStart; i--)
        {
            int var = assignmentStack[i];
            assignment[var] = -1;
            assignedCount--;
        }
        if (assignmentStack.Count > assignmentStart)
        {
            assignmentStack.RemoveRange(assignmentStart, assignmentStack.Count - assignmentStart);
        }

        propagationStack.Clear();
    }

    private void ResetWorkingState()
    {
        Array.Copy(data.ConstraintMinesLeft, workingMines, data.ConstraintCount);
        Array.Copy(data.ConstraintUnknownCounts, workingUnknown, data.ConstraintCount);
        Array.Fill(assignment, -1);
        changeStack.Clear();
        assignmentStack.Clear();
        propagationStack.Clear();
        assignedCount = 0;
        nodesVisited = 0;
    }
}

class Solver
{
    private readonly int height;
    private readonly int width;
    private readonly char[][] board;
    private readonly List<Constraint> constraints = new List<Constraint>();
    private readonly List<FrontierCell> frontier = new List<FrontierCell>();
    private readonly List<List<int>> variableToConstraintsLists = new List<List<int>>();
    private List<int>[] variableToConstraints = Array.Empty<List<int>>();
    private readonly Dictionary<long, int> frontierIndex = new Dictionary<long, int>();
    private readonly Queue<int> variableQueue = new Queue<int>();
    private readonly Queue<int> constraintQueue = new Queue<int>();
    private bool[] constraintQueued = Array.Empty<bool>();
    private readonly int[] dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private readonly int[] dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public Solver(char[][] board, int height, int width)
    {
        this.board = board;
        this.height = height;
        this.width = width;
        BuildStructures();
    }

    public List<(int X, int Y)> FindSafeCells()
    {
        if (constraints.Count > 0)
        {
            ProcessQueues();
        }

        bool progress;
        do
        {
            progress = false;
            if (ProcessQueues())
            {
                progress = true;
                continue;
            }
            if (ApplySubsetInference())
            {
                progress = true;
                continue;
            }
            if (AnalyzeComponents())
            {
                progress = true;
            }
        } while (progress);

        var result = new List<(int X, int Y)>();
        for (int i = 0; i < frontier.Count; i++)
        {
            if (frontier[i].State == CellState.Safe)
            {
                result.Add((frontier[i].X, frontier[i].Y));
            }
        }
        return result;
    }

    private void BuildStructures()
    {
        for (int y = 0; y < height; y++)
        {
            var row = board[y];
            for (int x = 0; x < width; x++)
            {
                char cell = row[x];
                if (cell < '0' || cell > '8') continue;

                var neighborVars = new List<int>(8);
                int flagged = 0;

                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + dx[dir];
                    int ny = y + dy[dir];
                    if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
                    char neighbor = board[ny][nx];
                    if (neighbor == '*')
                    {
                        flagged++;
                    }
                    else if (neighbor == ' ')
                    {
                        int idx = GetOrCreateFrontierIndex(nx, ny);
                        neighborVars.Add(idx);
                    }
                }

                if (neighborVars.Count == 0) continue;

                int remaining = cell - '0' - flagged;
                if (remaining < 0) remaining = 0;
                if (remaining > neighborVars.Count) remaining = neighborVars.Count;

                var constraint = new Constraint
                {
                    MinesLeft = remaining,
                    Variables = neighborVars
                };
                int constraintIndex = constraints.Count;
                constraints.Add(constraint);

                foreach (int varIdx in neighborVars)
                {
                    variableToConstraintsLists[varIdx].Add(constraintIndex);
                }
            }
        }

        variableToConstraints = new List<int>[variableToConstraintsLists.Count];
        for (int i = 0; i < variableToConstraintsLists.Count; i++)
        {
            variableToConstraints[i] = variableToConstraintsLists[i];
        }

        constraintQueued = new bool[constraints.Count];
        for (int i = 0; i < constraints.Count; i++)
        {
            constraintQueue.Enqueue(i);
            constraintQueued[i] = true;
        }
    }

    private int GetOrCreateFrontierIndex(int x, int y)
    {
        long key = (((long)y) << 32) | (uint)x;
        if (!frontierIndex.TryGetValue(key, out int idx))
        {
            idx = frontier.Count;
            frontierIndex[key] = idx;
            frontier.Add(new FrontierCell { X = x, Y = y, State = CellState.Unknown });
            variableToConstraintsLists.Add(new List<int>(4));
        }
        return idx;
    }

    private bool ProcessQueues()
    {
        bool progress = false;

        while (variableQueue.Count > 0)
        {
            int varIdx = variableQueue.Dequeue();
            progress = true;
            var state = frontier[varIdx].State;
            var constraintList = variableToConstraints[varIdx];
            if (constraintList == null) continue;

            foreach (int constraintIdx in constraintList)
            {
                if (constraintIdx < 0 || constraintIdx >= constraints.Count) continue;
                var constraint = constraints[constraintIdx];
                var vars = constraint.Variables;
                int index = vars.IndexOf(varIdx);
                if (index >= 0)
                {
                    vars.RemoveAt(index);
                    if (state == CellState.Mine)
                    {
                        constraint.MinesLeft--;
                    }
                    if (!constraintQueued[constraintIdx])
                    {
                        constraintQueue.Enqueue(constraintIdx);
                        constraintQueued[constraintIdx] = true;
                    }
                }
            }
        }

        Span<int> scratch = stackalloc int[8];

        while (constraintQueue.Count > 0)
        {
            int constraintIdx = constraintQueue.Dequeue();
            constraintQueued[constraintIdx] = false;
            var constraint = constraints[constraintIdx];
            int unknownCount = constraint.Variables.Count;
            if (unknownCount == 0) continue;

            int minesLeft = constraint.MinesLeft;
            if (minesLeft < 0 || minesLeft > unknownCount) continue;

            if (minesLeft == 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    scratch[i] = constraint.Variables[i];
                }
                for (int i = 0; i < unknownCount; i++)
                {
                    ApplyState(scratch[i], CellState.Safe);
                }
                progress = true;
            }
            else if (minesLeft == unknownCount)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    scratch[i] = constraint.Variables[i];
                }
                for (int i = 0; i < unknownCount; i++)
                {
                    ApplyState(scratch[i], CellState.Mine);
                }
                progress = true;
            }
        }

        return progress;
    }

    private bool ApplySubsetInference()
    {
        bool progress = false;
        var seenPairs = new HashSet<long>();
        Span<int> candidateBuffer = stackalloc int[64];

        for (int i = 0; i < constraints.Count; i++)
        {
            var constraintA = constraints[i];
            if (constraintA.Variables.Count == 0) continue;

            int candidateCount = 0;
            foreach (int varIdx in constraintA.Variables)
            {
                var neighbors = variableToConstraints[varIdx];
                if (neighbors == null) continue;
                foreach (int other in neighbors)
                {
                    if (other == i) continue;
                    bool exists = false;
                    for (int k = 0; k < candidateCount; k++)
                    {
                        if (candidateBuffer[k] == other)
                        {
                            exists = true;
                            break;
                        }
                    }
                    if (!exists && candidateCount < candidateBuffer.Length)
                    {
                        candidateBuffer[candidateCount++] = other;
                    }
                }
            }

            for (int c = 0; c < candidateCount; c++)
            {
                int otherIdx = candidateBuffer[c];
                if (otherIdx < 0 || otherIdx >= constraints.Count) continue;
                var constraintB = constraints[otherIdx];
                if (constraintB.Variables.Count == 0) continue;

                long key = i < otherIdx
                    ? ((long)i << 32) | (uint)otherIdx
                    : ((long)otherIdx << 32) | (uint)i;
                if (!seenPairs.Add(key))
                {
                    continue;
                }

                progress |= EvaluateSubset(i, otherIdx);
                progress |= EvaluateSubset(otherIdx, i);
            }
        }

        return progress;
    }

    private bool EvaluateSubset(int subsetIdx, int superIdx)
    {
        if (subsetIdx == superIdx) return false;
        var subset = constraints[subsetIdx];
        var superset = constraints[superIdx];
        if (subset.Variables.Count == 0 || superset.Variables.Count == 0) return false;
        if (subset.Variables.Count > superset.Variables.Count) return false;

        if (!IsSubset(subset.Variables, superset.Variables))
        {
            return false;
        }

        int diffMines = superset.MinesLeft - subset.MinesLeft;
        if (diffMines < 0) return false;

        Span<int> difference = stackalloc int[8];
        int diffCount = 0;
        foreach (int v in superset.Variables)
        {
            bool found = false;
            foreach (int s in subset.Variables)
            {
                if (s == v)
                {
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                difference[diffCount++] = v;
            }
        }

        if (diffCount == 0) return false;
        if (diffMines > diffCount) return false;

        bool progress = false;
        if (diffMines == 0)
        {
            for (int i = 0; i < diffCount; i++)
            {
                progress |= ApplyState(difference[i], CellState.Safe);
            }
        }
        else if (diffMines == diffCount)
        {
            for (int i = 0; i < diffCount; i++)
            {
                progress |= ApplyState(difference[i], CellState.Mine);
            }
        }

        return progress;
    }

    private static bool IsSubset(List<int> smaller, List<int> larger)
    {
        for (int i = 0; i < smaller.Count; i++)
        {
            int value = smaller[i];
            bool found = false;
            for (int j = 0; j < larger.Count; j++)
            {
                if (larger[j] == value)
                {
                    found = true;
                    break;
                }
            }
            if (!found) return false;
        }
        return true;
    }

    private bool AnalyzeComponents()
    {
        if (constraints.Count == 0 || frontier.Count == 0)
        {
            return false;
        }

        var varVisited = new bool[frontier.Count];
        var constraintVisited = new bool[constraints.Count];
        var queue = new Queue<int>();

        for (int start = 0; start < frontier.Count; start++)
        {
            if (frontier[start].State != CellState.Unknown || varVisited[start]) continue;

            var componentVars = new List<int>();
            var componentConstraints = new List<int>();

            queue.Enqueue(start);
            varVisited[start] = true;

            while (queue.Count > 0)
            {
                int varIdx = queue.Dequeue();
                componentVars.Add(varIdx);
                var constraintList = variableToConstraints[varIdx];
                if (constraintList == null) continue;

                foreach (int constraintIdx in constraintList)
                {
                    if (constraintIdx < 0 || constraintIdx >= constraints.Count) continue;
                    var constraint = constraints[constraintIdx];
                    if (constraint.Variables.Count == 0) continue;

                    if (!constraintVisited[constraintIdx])
                    {
                        constraintVisited[constraintIdx] = true;
                        componentConstraints.Add(constraintIdx);
                    }

                    foreach (int neighborVar in constraint.Variables)
                    {
                        if (frontier[neighborVar].State != CellState.Unknown) continue;
                        if (!varVisited[neighborVar])
                        {
                            varVisited[neighborVar] = true;
                            queue.Enqueue(neighborVar);
                        }
                    }
                }
            }

            foreach (int idx in componentConstraints)
            {
                constraintVisited[idx] = false;
            }

            if (componentVars.Count == 0 || componentConstraints.Count == 0)
            {
                continue;
            }

            var componentData = BuildComponentData(componentVars, componentConstraints);
            if (componentData.ConstraintCount == 0)
            {
                continue;
            }

            if (ProcessComponent(componentData))
            {
                return true;
            }
        }

        return false;
    }

    private ComponentData BuildComponentData(List<int> componentVars, List<int> componentConstraints)
    {
        var data = new ComponentData
        {
            GlobalVarIndices = componentVars.ToArray(),
            ConstraintVariables = new int[componentConstraints.Count][],
            ConstraintMinesLeft = new int[componentConstraints.Count],
            ConstraintUnknownCounts = new int[componentConstraints.Count],
            VarToConstraints = new int[componentVars.Count][]
        };

        int varCount = componentVars.Count;
        var varConstraintLists = new List<int>[varCount];
        for (int i = 0; i < varCount; i++)
        {
            varConstraintLists[i] = new List<int>(4);
        }

        var globalToLocal = new Dictionary<int, int>(varCount);
        for (int i = 0; i < varCount; i++)
        {
            globalToLocal[data.GlobalVarIndices[i]] = i;
        }

        for (int ci = 0; ci < componentConstraints.Count; ci++)
        {
            int globalConstraintIdx = componentConstraints[ci];
            var globalConstraint = constraints[globalConstraintIdx];
            var localVars = new List<int>(globalConstraint.Variables.Count);

            foreach (int globalVar in globalConstraint.Variables)
            {
                if (globalToLocal.TryGetValue(globalVar, out int localVar))
                {
                    localVars.Add(localVar);
                    varConstraintLists[localVar].Add(ci);
                }
            }

            data.ConstraintVariables[ci] = localVars.ToArray();
            data.ConstraintMinesLeft[ci] = globalConstraint.MinesLeft;
            data.ConstraintUnknownCounts[ci] = localVars.Count;
        }

        for (int i = 0; i < varCount; i++)
        {
            data.VarToConstraints[i] = varConstraintLists[i].ToArray();
        }

        return data;
    }

    private bool ProcessComponent(ComponentData component)
    {
        if (component.VarCount == 0 || component.ConstraintCount == 0)
        {
            return false;
        }

        const int ExhaustiveLimit = 18;
        const int SearchLimit = 256;

        if (component.VarCount <= ExhaustiveLimit)
        {
            var engine = new ConstraintSearchEngine(component);
            var result = engine.EnumerateForced(1_000_000);
            if (!result.HasSolution) return false;

            bool changed = false;
            for (int i = 0; i < component.VarCount; i++)
            {
                int globalIdx = component.GlobalVarIndices[i];
                if (frontier[globalIdx].State != CellState.Unknown) continue;
                if (result.ForcedSafe[i])
                {
                    changed |= ApplyState(globalIdx, CellState.Safe);
                }
                else if (result.ForcedMine[i])
                {
                    changed |= ApplyState(globalIdx, CellState.Mine);
                }
            }
            return changed;
        }

        int nodeLimit = ComputeNodeLimit(component.VarCount);
        var solver = new ConstraintSearchEngine(component);

        if (component.VarCount <= SearchLimit)
        {
            return RunAssumptionChecks(component, solver, nodeLimit, component.VarCount);
        }

        int selectiveCount = Math.Min(component.VarCount, 64);
        return RunAssumptionChecks(component, solver, nodeLimit, selectiveCount);
    }

    private bool RunAssumptionChecks(ComponentData component, ConstraintSearchEngine solver, int nodeLimit, int maxVariables)
    {
        bool changed = false;
        int[] order = BuildVariableOrder(component);
        int toCheck = Math.Min(maxVariables, order.Length);

        for (int idx = 0; idx < toCheck; idx++)
        {
            int localVar = order[idx];
            int globalIdx = component.GlobalVarIndices[localVar];
            if (frontier[globalIdx].State != CellState.Unknown) continue;

            var mineResult = solver.CheckAssumption(localVar, 1, nodeLimit);
            if (mineResult == SearchOutcome.Unsatisfiable)
            {
                changed |= ApplyState(globalIdx, CellState.Safe);
                continue;
            }

            var safeResult = solver.CheckAssumption(localVar, 0, nodeLimit);
            if (safeResult == SearchOutcome.Unsatisfiable)
            {
                changed |= ApplyState(globalIdx, CellState.Mine);
            }
        }

        return changed;
    }

    private static int[] BuildVariableOrder(ComponentData component)
    {
        int count = component.VarCount;
        var order = new int[count];
        for (int i = 0; i < count; i++)
        {
            order[i] = i;
        }
        Array.Sort(order, (a, b) =>
        {
            int degA = component.VarToConstraints[a].Length;
            int degB = component.VarToConstraints[b].Length;
            if (degA != degB) return degB.CompareTo(degA);
            return a.CompareTo(b);
        });
        return order;
    }

    private static int ComputeNodeLimit(int varCount)
    {
        if (varCount <= 20) return 2_000_000;
        if (varCount <= 30) return 1_000_000;
        if (varCount <= 40) return 600_000;
        if (varCount <= 60) return 250_000;
        if (varCount <= 100) return 150_000;
        if (varCount <= 160) return 80_000;
        if (varCount <= 256) return 45_000;
        return 20_000;
    }

    private bool ApplyState(int varIdx, CellState state)
    {
        if (varIdx < 0 || varIdx >= frontier.Count) return false;
        var cell = frontier[varIdx];
        if (cell.State == state) return false;
        if (cell.State != CellState.Unknown) return false;

        cell.State = state;
        frontier[varIdx] = cell;
        variableQueue.Enqueue(varIdx);
        return true;
    }
}

class Program
{
    static void Main()
    {
        string firstLine = Console.ReadLine() ?? string.Empty;
        var parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2 || !int.TryParse(parts[0], out int height) || !int.TryParse(parts[1], out int width))
        {
            Console.WriteLine("0");
            return;
        }

        var board = new char[height][];
        for (int y = 0; y < height; y++)
        {
            string line = Console.ReadLine() ?? string.Empty;
            var row = new char[width];
            for (int x = 0; x < width; x++)
            {
                row[x] = x < line.Length ? line[x] : ' ';
            }
            board[y] = row;
        }

        var solver = new Solver(board, height, width);
        var safeCells = solver.FindSafeCells();
        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.X} {cell.Y}");
        }
    }
}
```

This solver avoids guessing: it only declares cells safe when every admissible mine configuration forces them to be safe. It scales from tiny boards (full enumeration) to giant ones (constraint propagation plus selective, budgeted search).
