Here’s a complete compute shader that implements one timestep of 3D boid flocking with the interface you specified:

```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 radii;     // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;   // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;    // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint boidCount = counts.x;
    if (idx >= boidCount) {
        return;
    }

    uint baseIndex = idx * 2;
    vec3 pos = inData[baseIndex].xyz;
    vec3 vel = inData[baseIndex + 1].xyz;

    float dt          = radii.x;
    float sepRadius   = radii.y;
    float alignRadius = radii.z;
    float cohRadius   = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize   = bounds.x;
    float boundForce  = bounds.y;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint j = 0; j < boidCount; ++j) {
        if (j == idx) {
            continue;
        }

        uint otherBase = j * 2;
        vec3 otherPos = inData[otherBase].xyz;
        vec3 offset = otherPos - pos;
        float dist = length(offset);
        if (dist <= 0.0) {
            continue;
        }

        // Separation
        if (dist < sepRadius) {
            sepAccum -= offset / dist;
            ++sepCount;
        }

        // Alignment
        if (dist < alignRadius) {
            vec3 otherVel = inData[otherBase + 1].xyz;
            alignAccum += otherVel;
            ++alignCount;
        }

        // Cohesion
        if (dist < cohRadius) {
            cohAccum += otherPos;
            ++cohCount;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        vec3 sepForce = sepAccum / float(sepCount);
        accel += sepForce * sepWeight;
    }

    if (alignCount > 0) {
        vec3 alignAvg = alignAccum / float(alignCount);
        vec3 alignForce = alignAvg - vel;
        accel += alignForce * alignWeight;
    }

    if (cohCount > 0) {
        vec3 cohCenter = cohAccum / float(cohCount);
        vec3 cohForce = cohCenter - pos;
        accel += cohForce * cohWeight;
    }

    // Boundary forces
    float innerLimit = boundSize * 0.4;
    float scaleDiv = max(boundSize * 0.5, 0.0001);

    for (int axis = 0; axis < 3; ++axis) {
        float component = pos[axis];
        float absComp = abs(component);
        if (absComp > innerLimit) {
            float direction = component > 0.0 ? -1.0 : 1.0;
            float over = absComp - innerLimit;
            float scale = clamp(over / scaleDiv, 0.0, 1.0);
            accel[axis] += direction * boundForce * scale;
        }
    }

    // Integrate velocity and clamp speed
    vec3 newVel = vel + accel * dt;
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 0.0) {
        newVel = (newVel / speed) * maxSpeed;
    }

    // Integrate position
    vec3 newPos = pos + newVel * dt;

    outData[baseIndex] = vec4(newPos, 0.0);
    outData[baseIndex + 1] = vec4(newVel, 0.0);
}
```

This shader applies the Reynolds steering rules per invocation, clamps velocity to the supplied max speed, nudges boids back toward the center when they drift near the bounds, and writes the updated position/velocity back to the output buffer for the next timestep.
