```rust
use std::fmt::Write;
use std::io::{self, Read};

const PRE_LOCAL_PRUNE_PASSES: usize = 3;
const POST_LOCAL_PRUNE_PASSES: usize = 2;
const FINAL_PRUNE_PASSES: usize = 1;
const LOCAL_SEARCH_MAX_ATTEMPTS: usize = 50_000;
const LOCAL_SEARCH_CANDIDATE_DEGREE_CAP: u32 = 64;
const LOCAL_SEARCH_NEIGHBOR_DEGREE_CAP: u32 = 512;
const LOCAL_SEARCH_MIN_REMOVALS: usize = 2;
const LOCAL_SEARCH_MAX_REMOVALS: usize = 6;

fn main() {
    let mut input = FastInput::new();
    let vertex_count = match input.next_usize() {
        Some(value) => value,
        None => return,
        };
    let edge_count = input
        .next_usize()
        .expect("input must provide edge count after vertex count");
    if vertex_count > u32::MAX as usize {
        panic!("vertex count exceeds supported range");
    }

    let mut edges = Vec::with_capacity(edge_count);
    let mut degree = vec![0u32; vertex_count];
    for _ in 0..edge_count {
        let u = input.next_usize().expect("missing edge endpoint (u)");
        let v = input.next_usize().expect("missing edge endpoint (v)");
        if u >= vertex_count || v >= vertex_count {
            panic!("edge endpoint {u} or {v} is out of bounds");
        }
        edges.push((u as u32, v as u32));
        degree[u] += 1;
        degree[v] += 1;
    }
    drop(input);

    let (offsets, adjacency) = build_adjacency(vertex_count, &degree, &edges);
    let mut cover_flags = build_initial_cover(vertex_count, &edges);

    prune_cover(
        &mut cover_flags,
        &offsets,
        &adjacency,
        PRE_LOCAL_PRUNE_PASSES,
    );
    local_exchange(
        &mut cover_flags,
        &offsets,
        &adjacency,
        &degree,
        LOCAL_SEARCH_MAX_ATTEMPTS,
        LOCAL_SEARCH_CANDIDATE_DEGREE_CAP,
        LOCAL_SEARCH_NEIGHBOR_DEGREE_CAP,
    );
    prune_cover(
        &mut cover_flags,
        &offsets,
        &adjacency,
        POST_LOCAL_PRUNE_PASSES,
    );
    ensure_coverage(&mut cover_flags, &edges, &degree);
    prune_cover(
        &mut cover_flags,
        &offsets,
        &adjacency,
        FINAL_PRUNE_PASSES,
    );
    ensure_coverage(&mut cover_flags, &edges, &degree);

    drop(adjacency);
    drop(offsets);
    drop(degree);

    #[cfg(debug_assertions)]
    validate_cover(&cover_flags, &edges);

    drop(edges);

    let mut solution = Vec::with_capacity(cover_flags.len());
    for (idx, &flag) in cover_flags.iter().enumerate() {
        if flag != 0 {
            solution.push(idx as u32);
        }
    }
    solution.sort_unstable();

    let mut output = String::with_capacity(solution.len() * 12 + 32);
    writeln!(&mut output, "{}", solution.len()).unwrap();
    if !solution.is_empty() {
        for (i, vertex) in solution.iter().enumerate() {
            if i > 0 {
                output.push(' ');
            }
            write!(&mut output, "{}", vertex).unwrap();
        }
    }
    output.push('\n');
    print!("{output}");
}

fn build_adjacency(
    vertex_count: usize,
    degree: &[u32],
    edges: &[(u32, u32)],
) -> (Vec<usize>, Vec<u32>) {
    let mut offsets = vec![0usize; vertex_count + 1];
    for i in 0..vertex_count {
        offsets[i + 1] = offsets[i] + degree[i] as usize;
    }
    let mut adjacency = vec![0u32; offsets[vertex_count]];
    let mut cursor = offsets.clone();
    for &(u, v) in edges {
        let ui = u as usize;
        let vi = v as usize;
        adjacency[cursor[ui]] = v;
        cursor[ui] += 1;
        adjacency[cursor[vi]] = u;
        cursor[vi] += 1;
    }
    (offsets, adjacency)
}

fn build_initial_cover(vertex_count: usize, edges: &[(u32, u32)]) -> Vec<u8> {
    let mut matched = vec![0u8; vertex_count];
    let mut cover = vec![0u8; vertex_count];
    for &(u, v) in edges {
        let ui = u as usize;
        let vi = v as usize;
        if matched[ui] == 0 && matched[vi] == 0 {
            matched[ui] = 1;
            matched[vi] = 1;
            cover[ui] = 1;
            cover[vi] = 1;
        }
    }
    cover
}

fn prune_cover(
    cover_flags: &mut [u8],
    offsets: &[usize],
    adjacency: &[u32],
    passes: usize,
) {
    if passes == 0 {
        return;
    }
    let mut order: Vec<u32> = cover_flags
        .iter()
        .enumerate()
        .filter_map(|(idx, &flag)| if flag != 0 { Some(idx as u32) } else { None })
        .collect();
    if order.is_empty() {
        return;
    }
    order.sort_unstable_by_key(|&vertex| offsets[vertex as usize + 1] - offsets[vertex as usize]);
    for pass in 0..passes {
        if pass % 2 == 0 {
            for &vertex in &order {
                try_remove_vertex(vertex as usize, cover_flags, offsets, adjacency);
            }
        } else {
            for &vertex in order.iter().rev() {
                try_remove_vertex(vertex as usize, cover_flags, offsets, adjacency);
            }
        }
    }
}

fn try_remove_vertex(
    vertex: usize,
    cover_flags: &mut [u8],
    offsets: &[usize],
    adjacency: &[u32],
) -> bool {
    if cover_flags[vertex] == 0 {
        return false;
    }
    let start = offsets[vertex];
    let end = offsets[vertex + 1];
    for idx in start..end {
        let neighbor = adjacency[idx] as usize;
        if neighbor == vertex || cover_flags[neighbor] == 0 {
            return false;
        }
    }
    cover_flags[vertex] = 0;
    true
}

fn local_exchange(
    cover_flags: &mut [u8],
    offsets: &[usize],
    adjacency: &[u32],
    degree: &[u32],
    max_attempts: usize,
    candidate_degree_cap: u32,
    neighbor_degree_cap: u32,
) {
    if cover_flags.is_empty() || max_attempts == 0 {
        return;
    }
    let min_candidate_degree = LOCAL_SEARCH_MIN_REMOVALS as u32;
    let candidates: Vec<usize> = degree
        .iter()
        .enumerate()
        .filter_map(|(idx, &deg_value)| {
            if deg_value >= min_candidate_degree && deg_value <= candidate_degree_cap {
                Some(idx)
            } else {
                None
            }
        })
        .collect();
    if candidates.is_empty() {
        return;
    }
    let mut attempts_limit = max_attempts.min(cover_flags.len().saturating_mul(2));
    attempts_limit = attempts_limit.min(candidates.len().saturating_mul(4));
    if attempts_limit == 0 {
        return;
    }
    let mut attempts = 0usize;
    let mut buffer = Vec::with_capacity(LOCAL_SEARCH_MAX_REMOVALS);
    while attempts < attempts_limit {
        let mut progress = false;
        for &candidate in &candidates {
            if cover_flags[candidate] != 0 {
                continue;
            }
            if attempts >= attempts_limit {
                break;
            }
            attempts += 1;
            cover_flags[candidate] = 1;
            buffer.clear();
            let start = offsets[candidate];
            let end = offsets[candidate + 1];
            for idx in start..end {
                let neighbor = adjacency[idx] as usize;
                if cover_flags[neighbor] == 0 {
                    continue;
                }
                if degree[neighbor] > neighbor_degree_cap {
                    continue;
                }
                if try_remove_vertex(neighbor, cover_flags, offsets, adjacency) {
                    buffer.push(neighbor);
                    if buffer.len() >= LOCAL_SEARCH_MAX_REMOVALS {
                        break;
                    }
                }
            }
            if buffer.len() >= LOCAL_SEARCH_MIN_REMOVALS {
                progress = true;
                break;
            } else {
                for &vertex in &buffer {
                    cover_flags[vertex] = 1;
                }
                cover_flags[candidate] = 0;
            }
        }
        if !progress {
            break;
        }
    }
}

fn ensure_coverage(cover_flags: &mut [u8], edges: &[(u32, u32)], degree: &[u32]) {
    for &(u, v) in edges {
        let ui = u as usize;
        let vi = v as usize;
        if cover_flags[ui] == 0 && cover_flags[vi] == 0 {
            if degree[ui] >= degree[vi] {
                cover_flags[ui] = 1;
            } else {
                cover_flags[vi] = 1;
            }
        }
    }
}

#[cfg(debug_assertions)]
fn validate_cover(cover_flags: &[u8], edges: &[(u32, u32)]) {
    for &(u, v) in edges {
        let ui = u as usize;
        let vi = v as usize;
        assert!(
            cover_flags.get(ui).copied().unwrap_or(0) != 0
                || cover_flags.get(vi).copied().unwrap_or(0) != 0,
            "edge ({u}, {v}) is uncovered"
        );
    }
}

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin()
            .read_to_end(&mut data)
            .expect("failed to read stdin");
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let len = self.data.len();
        while self.pos < len && !self.data[self.pos].is_ascii_digit() {
            self.pos += 1;
        }
        if self.pos >= len {
            return None;
        }
        let mut value = 0usize;
        while self.pos < len {
            let byte = self.data[self.pos];
            if !byte.is_ascii_digit() {
                break;
            }
            value = value * 10 + (byte - b'0') as usize;
            self.pos += 1;
        }
        Some(value)
    }
}
```
