fn is_connected(adj: &[Vec<usize>], start: usize) -> bool {
    let mut seen = vec![false; adj.len()];
    let mut q = VecDeque::with_capacity(adj.len());
    seen[start] = true;
    q.push_back(start);
    let mut count = 1usize;
    while let Some(v) = q.pop_front() {
        for &nb in &adj[v] {
            if !seen[nb] {
                seen[nb] = true;
                count += 1;
                q.push_back(nb);
            }
        }
    }
    count == adj.len()
}

fn bfs_layers(adj: &[Vec<usize>], start: usize) -> Vec<i32> {
    let mut dist = vec![-1i32; adj.len()];
    let mut q = VecDeque::with_capacity(adj.len());
    dist[start] = 0;
    q.push_back(start);
    while let Some(v) = q.pop_front() {
        let dv = dist[v];
        for &nb in &adj[v] {
            if dist[nb] == -1 {
                dist[nb] = dv + 1;
                q.push_back(nb);
            }
        }
    }
    dist
}

fn compute_base_seed(width: usize, height: usize, obstacles: usize, total: usize) -> u64 {
    let mut seed = 0x9E37_79B9_7F4A_7C15u64;
    seed ^= (width as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9);
    seed ^= (height as u64).wrapping_mul(0x94D0_49BB_1331_11EB);
    seed ^= (obstacles as u64).wrapping_mul(0xF135_7AEA_2E62_4EAF);
    seed ^= (total as u64).wrapping_mul(0x4F99_8A0A_F0F4_0D87);
    seed
}

struct Precomp {

    height: usize,
    coords: Vec<(i32, i32)>,
    adj: Vec<Vec<usize>>,
    start: usize,

    depth_hint: Vec<i32>,
    border_hint: Vec<i32>,
    neighbor_of_start: Vec<bool>,

    total: usize,
}

fn solve(pre: &Precomp, base_seed: u64) -> Option<Vec<usize>> {
    const STRATEGIES: [Strategy; 6] = [
        Strategy {
            border_bias: 1,
            depth_bias: 1,
            cycle_bias: true,
            shuffle: false,
        },
        Strategy {
            border_bias: 1,
            depth_bias: 1,
            cycle_bias: true,
            shuffle: true,
        },
        Strategy {
            border_bias: -1,
            depth_bias: 1,
            cycle_bias: true,
            shuffle: true,
        },
        Strategy {
            border_bias: 1,
            depth_bias: 0,
            cycle_bias: false,
            shuffle: true,
        },
        Strategy {
            border_bias: -1,
            depth_bias: 0,
            cycle_bias: true,
            shuffle: true,
        },
        Strategy {
            border_bias: 0,
            depth_bias: 1,
            cycle_bias: true,
            shuffle: true,
        },
    ];

    let repeats = match pre.total {
        0..=64 => 12,
        65..=128 => 10,
        129..=512 => 8,
        513..=2048 => 6,
        2049..=8192 => 5,
        _ => 4,
    };

    let mut attempt_idx = 0usize;
    for _ in 0..repeats {
        for strat in STRATEGIES {
            attempt_idx += 1;
            let seed = mix64(
                base_seed
                    ^ ((attempt_idx as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9)),
            );
            let mut engine = Engine::new(pre, seed, strat);
            if engine.run() {
                return Some(engine.path);
            }
        }
    }
    None
}

#[derive(Clone, Copy)]
struct Strategy {
    border_bias: i8,
    depth_bias: i8,
    cycle_bias: bool,
    shuffle: bool,
}

struct Engine<'a> {
    pre: &'a Precomp,
    visited: BitSet,
    path: Vec<usize>,
    visited_count: usize,
    target_len: usize,
    rng: SplitMix64,
    marker: Marker,
    queue: VecDeque<usize>,
    strategy: Strategy,
}

impl<'a> Engine<'a> {
    fn new(pre: &'a Precomp, seed: u64, strategy: Strategy) -> Self {
        let mut visited = BitSet::new(pre.total);
        visited.set(pre.start, true);
        let mut path = Vec::with_capacity(pre.total);
        path.push(pre.start);
        Self {
            pre,
            visited,
            path,
            visited_count: 1,
            target_len: pre.total,
            rng: SplitMix64::new(seed),
            marker: Marker::new(pre.total),
            queue: VecDeque::with_capacity(pre.total),
            strategy,
        }
    }

    fn run(&mut self) -> bool {
        self.dfs(self.pre.start)
    }

    fn dfs(&mut self, current: usize) -> bool {
        if self.visited_count == self.target_len {
            return self.finish_check(current);
        }
        let candidates = self.gather_candidates(current);
        if candidates.is_empty() {
            return false;
        }
        for nb in candidates {
            if !self.advance(nb) {
                continue;
            }
            if self.dfs(nb) {
                return true;
            }
            self.undo(nb);
        }
        false
    }

    fn finish_check(&self, current: usize) -> bool {
        if let Some(last) = self.pre.forced_last {
            if current != last {
                return false;
            }
        }
        if self.pre.require_cycle {
            return self.pre.adj[current]
                .iter()
                .any(|&nb| nb == self.pre.start);
        }
        true
    }

    fn advance(&mut self, node: usize) -> bool {
        self.visited.set(node, true);
        self.visited_count += 1;
        self.path.push(node);

        if self.visited_count == self.target_len {
            return true;
        }

        if let Some(last) = self.pre.forced_last {
            if node == last {
                self.visited.set(node, false);
                self.visited_count -= 1;
                self.path.pop();
                return false;
            }
        }

        if !self.connectivity_ok(node) {
            self.visited.set(node, false);
            self.visited_count -= 1;
            self.path.pop();
            return false;
        }
        true
    }

    fn undo(&mut self, node: usize) {
        self.visited.set(node, false);
        self.visited_count -= 1;
        self.path.pop();
    }

    fn gather_candidates(&mut self, current: usize) -> Vec<usize> {
        let mut entries: Vec<(usize, CandidateScore)> = Vec::with_capacity(4);
        let remaining_total = self.target_len - self.visited_count;
        for &nb in &self.pre.adj[current] {
            if self.visited.get(nb) {
                continue;
            }
            if let Some(last) = self.pre.forced_last {
                if nb == last && remaining_total > 1 {
                    continue;
                }
            }
            let rem_deg = self.remaining_degree(nb);
            if rem_deg == 0 && remaining_total > 1 {
                continue;
            }
            let forced_flag = if rem_deg <= 1 { 0 } else { 1 };
            let border_metric = match self.strategy.border_bias {
                1 => -self.pre.border_hint[nb],
                -1 => self.pre.border_hint[nb],
                _ => 0,
            };
            let depth_metric = match self.strategy.depth_bias {
                1 => -self.pre.depth_hint[nb],
                -1 => self.pre.depth_hint[nb],
                _ => 0,
            };
            let cycle_penalty = if self.strategy.cycle_bias
                && self.pre.require_cycle
                && remaining_total <= 6
            {
                if self.pre.neighbor_of_start[nb] {
                    0
                } else {
                    1
                }
            } else {
                0
            };
            let rand = if self.strategy.shuffle {
                self.rng.next_u32()
            } else {
                nb as u32
            };
            entries.push((
                nb,
                CandidateScore {
                    cycle_penalty,
                    forced_flag,
                    rem_deg: rem_deg as u8,
                    border_metric,
                    depth_metric,
                    rand,
                },
            ));
        }
        entries.sort_unstable_by(|a, b| a.1.cmp(&b.1));
        entries.into_iter().map(|(nb, _)| nb).collect()
    }

    fn remaining_degree(&self, node: usize) -> usize {
        self.pre.adj[node]
            .iter()
            .filter(|&&nb| !self.visited.get(nb))
            .count()
    }

    fn connectivity_ok(&mut self, anchor: usize) -> bool {
        let remaining = self.target_len - self.visited_count;
        if remaining == 0 {
            return true;
        }
        self.queue.clear();
        self.marker.reset();
        self.marker.mark(anchor);
        self.queue.push_back(anchor);
        let mut seen = 0usize;
        while let Some(v) = self.queue.pop_front() {
            for &nb in &self.pre.adj[v] {
                if self.visited.get(nb) && nb != anchor {
                    continue;
                }
                if self.marker.is_marked(nb) {
                    continue;
                }
                self.marker.mark(nb);
                self.queue.push_back(nb);
                if !self.visited.get(nb) {
                    seen += 1;
                }
            }
        }
        seen == remaining
    }
}

#[derive(Clone, Copy)]
struct CandidateScore {
    cycle_penalty: u8,
    forced_flag: u8,
    rem_deg: u8,
    border_metric: i32,
    depth_metric: i32,
    rand: u32,
}

impl CandidateScore {
    fn cmp(&self, other: &Self) -> Ordering {
        self.cycle_penalty
            .cmp(&other.cycle_penalty)
            .then(self.forced_flag.cmp(&other.forced_flag))
            .then(self.rem_deg.cmp(&other.rem_deg))
            .then(self.border_metric.cmp(&other.border_metric))
            .then(self.depth_metric.cmp(&other.depth_metric))
            .then(self.rand.cmp(&other.rand))
    }
}

struct BitSet {
    bits: Vec<u64>,
}

impl BitSet {
    fn new(size: usize) -> Self {
        Self {
            bits: vec![0; (size + 63) / 64],
        }
    }
    #[inline]
    fn set(&mut self, idx: usize, val: bool) {
        let block = idx / 64;
        let bit = idx % 64;
        if val {
            self.bits[block] |= 1u64 << bit;
        } else {
            self.bits[block] &= !(1u64 << bit);
        }
    }
    #[inline]
    fn get(&self, idx: usize) -> bool {
        let block = idx / 64;
        let bit = idx % 64;
        ((self.bits[block] >> bit) & 1) != 0
    }
}

struct Marker {
    marks: Vec<u32>,
    epoch: u32,
}

impl Marker {
    fn new(size: usize) -> Self {
        Self {
            marks: vec![0; size],
            epoch: 1,
        }
    }
    fn reset(&mut self) {
        self.epoch = self.epoch.wrapping_add(1);
        if self.epoch == 0 {
            self.marks.fill(0);
            self.epoch = 1;
        }
    }
    fn mark(&mut self, idx: usize) {
        self.marks[idx] = self.epoch;
    }
    fn is_marked(&self, idx: usize) -> bool {
        self.marks[idx] == self.epoch
    }
}

fn mix64(mut z: u64) -> u64 {
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
    z ^ (z >> 31)
}

struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E37_79B9_7F4A_7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
        z ^ (z >> 31)
    }
    fn next_u32(&mut self) => u32 {
        self.next_u64() as u32
    }
}
