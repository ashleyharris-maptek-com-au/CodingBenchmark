struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal : TEXCOORD1;
    [[vk::location(2)]] float2 uv : TEXCOORD2;
    [[vk::location(3)]] float3 tangent : TEXCOORD3;
    [[vk::location(4)]] float3 color : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float2 scaledUV = input.uv * float2(8.0, 16.0);
    float rowIndex = floor(scaledUV.y);
    float offset = fmod(rowIndex, 2.0) * 0.5;
    float2 brickUV = float2(scaledUV.x + offset, scaledUV.y);
    float2 cellUV = frac(brickUV);

    const float mortarX = 0.05;
    const float mortarY = 0.1;

    float insideX = (cellUV.x > mortarX && cellUV.x < 1.0 - mortarX) ? 1.0 : 0.0;
    float insideY = (cellUV.y > mortarY && cellUV.y < 1.0 - mortarY) ? 1.0 : 0.0;
    float brickMask = insideX * insideY;

    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = lerp(mortarColor, brickColor, brickMask);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    float3 shadedColor = baseColor * shade;
    return float4(shadedColor, 1.0);
}
