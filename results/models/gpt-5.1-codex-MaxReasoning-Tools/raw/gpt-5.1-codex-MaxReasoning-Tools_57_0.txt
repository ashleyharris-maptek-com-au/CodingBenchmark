const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8');
  if (!input.trim()) {
    outputResult([]);
    return;
  }
  let data;
  try {
    data = JSON.parse(input);
  } catch {
    outputResult([]);
    return;
  }

  const width = toInt(data.width);
  const height = toInt(data.height);
  let cutBudget = toInt(data.cutBudget);
  const grid = Array.isArray(data.grid) ? data.grid : [];

  if (!width || !height || cutBudget <= 0 || grid.length === 0) {
    outputResult([]);
    return;
  }

  const size = width * height;
  const burnable = new Uint8Array(size);
  for (let y = 0; y < height; y++) {
    const row = grid[y] || '';
    for (let x = 0; x < width; x++) {
      if (row[x] === '.') {
        burnable[y * width + x] = 1;
      }
    }
  }

  const rawTowns = Array.isArray(data.towns) ? data.towns : [];
  const townValuePerCell = new Float64Array(size);
  const townCells = [];
  for (const town of rawTowns) {
    if (!town) continue;
    const x = toInt(town.x);
    const y = toInt(town.y);
    const value = Number(town.value);
    if (!Number.isFinite(x) || !Number.isFinite(y) || !Number.isFinite(value) || value <= 0) continue;
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const id = y * width + x;
    if (!burnable[id]) continue;
    townValuePerCell[id] += value;
    townCells.push({ id, value });
  }

  const ignitionInput = Array.isArray(data.ignitions) ? data.ignitions : [];
  const ignitionIds = [];
  const isIgnitionCell = new Uint8Array(size);
  for (const ign of ignitionInput) {
    if (!Array.isArray(ign) || ign.length < 2) continue;
    const x = toInt(ign[0]);
    const y = toInt(ign[1]);
    if (!Number.isFinite(x) || !Number.isFinite(y)) continue;
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const id = y * width + x;
    if (!burnable[id]) continue;
    if (!isIgnitionCell[id]) {
      isIgnitionCell[id] = 1;
      ignitionIds.push(id);
    }
  }

  if (ignitionIds.length === 0) {
    outputResult([]);
    return;
  }

  const reachable = new Uint8Array(size);
  const dist = new Int32Array(size);
  dist.fill(-1);
  const bfsQueue = new Int32Array(size);
  let qh = 0;
  let qt = 0;
  for (const id of ignitionIds) {
    reachable[id] = 1;
    dist[id] = 0;
    bfsQueue[qt++] = id;
  }
  while (qh < qt) {
    const id = bfsQueue[qh++];
    const x = id % width;
    const y = (id / width) | 0;
    const nextDist = dist[id] + 1;
    if (y > 0) enqueue(id - width, nextDist);
    if (y + 1 < height) enqueue(id + width, nextDist);
    if (x > 0) enqueue(id - 1, nextDist);
    if (x + 1 < width) enqueue(id + 1, nextDist);
  }
  function enqueue(nid, d) {
    if (!burnable[nid] || reachable[nid]) return;
    reachable[nid] = 1;
    dist[nid] = d;
    bfsQueue[qt++] = nid;
  }

  const threatenedTownCells = [];
  let threatenedValueTotal = 0;
  for (const town of townCells) {
    if (reachable[town.id]) {
      threatenedTownCells.push(town);
      threatenedValueTotal += town.value;
    }
  }
  if (threatenedTownCells.length === 0 || threatenedValueTotal <= 0) {
    outputResult([]);
    return;
  }

  const candidateScore = new Float64Array(size);

  const dirX = [0, 1, 0, -1];
  const dirY = [-1, 0, 1, 0];

  const addCandidate = (id, score) => {
    if (score <= 0) return;
    if (!burnable[id] || !reachable[id]) return;
    if (townValuePerCell[id] > 0) return;
    if (score > candidateScore[id]) candidateScore[id] = score;
  };

  addArticulationCandidates();
  addAdjacentCandidates();
  addPathCandidates(25);
  addRingCandidates(40, 3);

  let candidateList = [];
  for (let id = 0; id < size; id++) {
    const score = candidateScore[id];
    if (score > 0) candidateList.push({ id, score });
  }

  if (candidateList.length === 0) {
    const fallback = buildFallbackLine(Math.min(cutBudget, 50));
    outputResult(fallback.map(idToCoord));
    return;
  }

  candidateList.sort((a, b) => (b.score - a.score) || (a.id - b.id));
  const limit = Math.min(candidateList.length, Math.max(200, cutBudget * 30));
  if (candidateList.length > limit) candidateList.length = limit;

  const cutMask = new Uint8Array(size);
  const selectedSet = new Set();
  let nextCandidateIdx = 0;

  const visited = new Uint32Array(size);
  let visitToken = 1;
  const simQueue = new Int32Array(size);

  const fillToBudget = () => {
    let added = 0;
    while (selectedSet.size < cutBudget && nextCandidateIdx < candidateList.length) {
      const { id } = candidateList[nextCandidateIdx++];
      if (cutMask[id]) continue;
      cutMask[id] = 1;
      selectedSet.add(id);
      added++;
    }
    return added;
  };

  const simulateCuts = () => {
    if (visitToken >= 0xfffffff0) {
      visited.fill(0);
      visitToken = 1;
    }
    const token = ++visitToken;
    let head = 0;
    let tail = 0;
    for (const id of ignitionIds) {
      if (!burnable[id] || cutMask[id]) continue;
      if (visited[id] === token) continue;
      visited[id] = token;
      simQueue[tail++] = id;
    }
    while (head < tail) {
      const id = simQueue[head++];
      const x = id % width;
      const y = (id / width) | 0;
      if (y > 0) spread(id - width, token);
      if (y + 1 < height) spread(id + width, token);
      if (x > 0) spread(id - 1, token);
      if (x + 1 < width) spread(id + 1, token);
    }
    let burned = 0;
    for (const town of threatenedTownCells) {
      if (visited[town.id] === token) burned += town.value;
    }
    return threatenedValueTotal - burned;
  };

  const spread = (nid, token) => {
    if (!burnable[nid] || cutMask[nid]) return;
    if (visited[nid] === token) return;
    visited[nid] = token;
    simQueue[simQueueTail++] = nid;
  };

  // Overriding spread to access tail pointer
  let simQueueTail = 0;
  function spread(nid, token) {
    if (!burnable[nid] || cutMask[nid]) return;
    if (visited[nid] === token) return;
    visited[nid] = token;
    simQueue[simQueueTail++] = nid;
  }

  fillToBudget();
  let bestSaved = selectedSet.size ? simulateWrapper() : 0;

  const maxIterations = 3;
  for (let iter = 0; iter < maxIterations; iter++) {
    let changed = false;
    const current = Array.from(selectedSet);
    for (const id of current) {
      cutMask[id] = 0;
      const saved = simulateWrapper();
      if (saved + 1e-6 >= bestSaved) {
        selectedSet.delete(id);
        bestSaved = saved;
        changed = true;
      } else {
        cutMask[id] = 1;
      }
    }
    const added = fillToBudget();
    if (added > 0) {
      bestSaved = simulateWrapper();
      changed = true;
    }
    if (!changed) break;
  }

  if ((bestSaved <= 0 || selectedSet.size === 0) && cutBudget > 0) {
    const fallbackIds = buildFallbackLine(Math.min(cutBudget, 100));
    selectedSet.clear();
    cutMask.fill(0);
    for (const id of fallbackIds) {
      cutMask[id] = 1;
      selectedSet.add(id);
      if (selectedSet.size >= cutBudget) break;
    }
    if (selectedSet.size) bestSaved = simulateWrapper();
  }

  const firebreaks = Array.from(selectedSet).sort((a, b) => a - b).map(idToCoord);
  outputResult(firebreaks);

  function simulateWrapper() {
    simQueueTail = 0;
    return simulateCuts();
  }

  function addArticulationCandidates() {
    const disc = new Int32Array(size);
    disc.fill(-1);
    const low = new Int32Array(size);
    const subTown = new Float64Array(size);
    const subIgn = new Int32Array(size);
    const savedPotential = new Float64Array(size);
    const stack = [];
    let time = 0;

    for (let start = 0; start < size; start++) {
      if (!reachable[start] || !burnable[start] || disc[start] !== -1) continue;
      stack.push({ id: start, parent: -1, nextDir: 0, entered: false, x: 0, y: 0 });
      while (stack.length) {
        const frame = stack[stack.length - 1];
        const id = frame.id;
        if (!frame.entered) {
          frame.entered = true;
          frame.x = id % width;
          frame.y = (id / width) | 0;
          disc[id] = low[id] = time++;
          subTown[id] = townValuePerCell[id];
          subIgn[id] = isIgnitionCell[id];
          savedPotential[id] = 0;
        }

        if (frame.nextDir < 4) {
          const dirIdx = frame.nextDir++;
          const nx = frame.x + dirX[dirIdx];
          const ny = frame.y + dirY[dirIdx];
          if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
          const nid = ny * width + nx;
          if (!reachable[nid] || !burnable[nid]) continue;
          if (disc[nid] === -1) {
            stack.push({ id: nid, parent: id, nextDir: 0, entered: false, x: 0, y: 0 });
          } else if (nid !== frame.parent) {
            if (disc[nid] < low[id]) low[id] = disc[nid];
          }
          continue;
        }

        stack.pop();
        const potential = savedPotential[id];
        if (potential > 0 && townValuePerCell[id] === 0) addCandidate(id, potential);

        if (frame.parent !== -1) {
          const parentId = frame.parent;
          if (low[id] < low[parentId]) low[parentId] = low[id];
          subTown[parentId] += subTown[id];
          subIgn[parentId] += subIgn[id];
          if (low[id] >= disc[parentId] && subIgn[id] === 0) {
            savedPotential[parentId] += subTown[id];
          }
        }
      }
    }
  }

  function addAdjacentCandidates() {
    for (const town of threatenedTownCells) {
      const id = town.id;
      const x = id % width;
      const y = (id / width) | 0;
      const weight = town.value * 0.35;
      if (y > 0) addCandidate(id - width, weight);
      if (y + 1 < height) addCandidate(id + width, weight);
      if (x > 0) addCandidate(id - 1, weight);
      if (x + 1 < width) addCandidate(id + 1, weight);
    }
  }

  function addPathCandidates(maxStepsPerTown) {
    for (const town of threatenedTownCells) {
      const townId = town.id;
      let current = townId;
      const maxSteps = Math.min(maxStepsPerTown, dist[townId]);
      if (maxSteps <= 0) continue;
      let step = 0;
      while (step < maxSteps) {
        const currentDist = dist[current];
        if (currentDist <= 0) break;
        const cx = current % width;
        const cy = (current / width) | 0;
        let nextId = -1;
        const dirShift = (cx + cy + step) & 3;
        for (let i = 0; i < 4; i++) {
          const dirIdx = (dirShift + i) & 3;
          const nx = cx + dirX[dirIdx];
          const ny = cy + dirY[dirIdx];
          if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
          const nid = ny * width + nx;
          if (!reachable[nid] || !burnable[nid]) continue;
          if (dist[nid] === currentDist - 1) {
            nextId = nid;
            break;
          }
        }
        if (nextId === -1) break;
        if (townValuePerCell[nextId] === 0) {
          const score = (town.value * 0.5) / (step + 1);
          addCandidate(nextId, score);
        }
        current = nextId;
        step++;
      }
    }
  }

  function addRingCandidates(maxTowns, maxRing) {
    const sorted = threatenedTownCells.slice().sort((a, b) => b.value - a.value);
    const limit = Math.min(sorted.length, maxTowns);
    for (let i = 0; i < limit; i++) {
      const town = sorted[i];
      const baseDist = dist[town.id];
      const ringLimit = Math.min(maxRing, baseDist);
      if (ringLimit <= 0) continue;
      const x0 = town.id % width;
      const y0 = (town.id / width) | 0;
      for (let r = 1; r <= ringLimit; r++) {
        const weight = (town.value * 0.4) / (r + 1);
        for (let dx = -r; dx <= r; dx++) {
          const dy = r - Math.abs(dx);
          const xs = x0 + dx;
          if (xs < 0 || xs >= width) continue;
          const positions = dy === 0 ? [[xs, y0]] : [[xs, y0 + dy], [xs, y0 - dy]];
          for (const [cx, cy] of positions) {
            if (cy < 0 || cy >= height) continue;
            const cid = cy * width + cx;
            if (!reachable[cid] || !burnable[cid]) continue;
            if (townValuePerCell[cid] > 0) continue;
            addCandidate(cid, weight);
          }
        }
      }
    }
  }

  function buildFallbackLine(limit) {
    const result = [];
    if (!threatenedTownCells.length || limit <= 0) return result;
    let topTown = threatenedTownCells[0];
    for (const town of threatenedTownCells) if (town.value > topTown.value) topTown = town;
    const seen = new Set();
    let current = topTown.id;
    let safety = 0;
    while (result.length < limit && dist[current] > 0 && safety < limit * 4) {
      safety++;
      const currentDist = dist[current];
      const cx = current % width;
      const cy = (current / width) | 0;
      let nextId = -1;
      const dirShift = (cx + cy + safety) & 3;
      for (let i = 0; i < 4; i++) {
        const dirIdx = (dirShift + i) & 3;
        const nx = cx + dirX[dirIdx];
        const ny = cy + dirY[dirIdx];
        if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
        const nid = ny * width + nx;
        if (!reachable[nid] || !burnable[nid]) continue;
        if (dist[nid] === currentDist - 1) {
          nextId = nid;
          break;
        }
      }
      if (nextId === -1) break;
      if (townValuePerCell[nextId] === 0 && !seen.has(nextId)) {
        result.push(nextId);
        seen.add(nextId);
      }
      current = nextId;
    }
    if (result.length < limit) {
      const extras = [];
      for (const cell of result) {
        const x = cell % width;
        const y = (cell / width) | 0;
        const neighbors = [
          y > 0 ? cell - width : -1,
          y + 1 < height ? cell + width : -1,
          x > 0 ? cell - 1 : -1,
          x + 1 < width ? cell + 1 : -1
        ];
        for (const nid of neighbors) {
          if (nid === -1) continue;
          if (!reachable[nid] || !burnable[nid]) continue;
          if (townValuePerCell[nid] > 0 || seen.has(nid)) continue;
          extras.push(nid);
          seen.add(nid);
          if (result.length + extras.length >= limit) break;
        }
        if (result.length + extras.length >= limit) break;
      }
      result.push(...extras);
    }
    if (result.length < limit) {
      const localVisited = new Set([topTown.id]);
      const queue = [topTown.id];
      let idx = 0;
      while (idx < queue.length && result.length < limit && idx < 2000) {
        const id = queue[idx++];
        const x = id % width;
        const y = (id / width) | 0;
        for (let d = 0; d < 4; d++) {
          const nx = x + dirX[d];
          const ny = y + dirY[d];
          if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
          const nid = ny * width + nx;
          if (!reachable[nid] || !burnable[nid]) continue;
          if (localVisited.has(nid)) continue;
          localVisited.add(nid);
          queue.push(nid);
          if (townValuePerCell[nid] === 0 && !seen.has(nid)) {
            result.push(nid);
            seen.add(nid);
            if (result.length >= limit) break;
          }
        }
      }
    }
    if (result.length > limit) result.length = limit;
    return result;
  }

  function idToCoord(id) {
    return [id % width, (id / width) | 0];
  }

  function toInt(value) {
    const num = Number(value);
    if (!Number.isFinite(num)) return 0;
    return Math.floor(num);
  }

  function outputResult(firebreaks) {
    console.log(JSON.stringify({ firebreaks }));
  }
}

main();
