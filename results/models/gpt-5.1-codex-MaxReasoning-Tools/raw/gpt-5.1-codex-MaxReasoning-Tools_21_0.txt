```cpp
#include <array>
#include <vector>
#include <deque>
#include <unordered_map>
#include <unordered_set>
#include <deque>
#include <iostream>
#include <algorithm>
#include <utility>
#include <cstdint>
#include <cstddef>

constexpr int MAX_DIM = 20;
inline int G_DIMS = 0;

struct Coord {
    std::array<int, MAX_DIM> v{};
    bool operator==(const Coord& other) const noexcept {
        for (int i = 0; i < G_DIMS; ++i) {
            if (v[i] != other.v[i]) {
                return false;
            }
        }
        return true;
    }
};

struct CoordHash {
    std::size_t operator()(const Coord& c) const noexcept {
        std::size_t h = 0;
        for (int i = 0; i < G_DIMS; ++i) {
            std::size_t part = static_cast<std::size_t>(c.v[i]) +
                               0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            h ^= part;
        }
        return h;
    }
};

class SnakeSolver {
public:
    SnakeSolver(int dims,
                long long maxTurns,
                std::array<int, MAX_DIM> bounds,
                std::deque<Coord> snake,
                std::unordered_map<Coord, int, CoordHash> occupancy,
                std::unordered_set<Coord, CoordHash> foods,
                std::unordered_set<Coord, CoordHash> obstacles)
        : dims_(dims),
          maxTurns_(maxTurns),
          bounds_(bounds),
          snake_(std::move(snake)),
          occupancy_(std::move(occupancy)),
          foods_(std::move(foods)),
          obstacles_(std::move(obstacles)),
          axisPreference_(dims, 1) {
        moves_.reserve(static_cast<std::size_t>(std::min<long long>(maxTurns_, 200000LL)));
    }

    void solve() {
        std::vector<Coord> foodPath;
        std::vector<Coord> tailPath;

        while (turnsUsed_ < maxTurns_ && !foods_.empty()) {
            bool moved = false;

            foodPath.clear();
            if (findFoodPath(foodPath) && foodPath.size() >= 2) {
                if (executeMove(foodPath[1])) {
                    moved = true;
                }
            }

            if (!moved) {
                tailPath.clear();
                if (findTailPath(tailPath) && tailPath.size() >= 2) {
                    if (executeMove(tailPath[1])) {
                        moved = true;
                    }
                }
            }

            if (!moved) {
                if (fallbackMove()) {
                    moved = true;
                }
            }

            if (!moved) {
                break;
            }
        }

        for (const auto& [axis, dir] : moves_) {
            std::cout << axis << ' ' << dir << '\n';
        }
    }

private:
    bool inBounds(const Coord& c) const {
        for (int i = 0; i < dims_; ++i) {
            if (c.v[i] < 0 || c.v[i] >= bounds_[i]) {
                return false;
            }
        }
        return true;
    }

    bool hasFood(const Coord& c) const {
        return foods_.find(c) != foods_.end();
    }

    bool isMoveLegal(const Coord& candidate, bool grows) const {
        if (!inBounds(candidate)) {
            return false;
        }
        if (obstacles_.find(candidate) != obstacles_.end()) {
            return false;
        }
        auto occ = occupancy_.find(candidate);
        if (occ != occupancy_.end()) {
            const Coord& tail = snake_.back();
            if (!grows && candidate == tail && occ->second == 1) {
                return true;
            }
            return false;
        }
        return true;
    }

    bool executeMove(const Coord& nextHead) {
        if (turnsUsed_ >= maxTurns_) {
            return false;
        }
        bool grows = hasFood(nextHead);
        if (!isMoveLegal(nextHead, grows)) {
            return false;
        }
        if (!ensureTailReachable(nextHead, grows)) {
            return false;
        }
        Coord prevHead = snake_.front();
        commitMove(nextHead, grows);
        auto diff = diffAxis(prevHead, nextHead);
        if (diff.first < 0) {
            return false;
        }
        moves_.push_back(diff);
        ++turnsUsed_;
        return true;
    }

    void commitMove(const Coord& nextHead, bool grows) {
        snake_.push_front(nextHead);
        ++occupancy_[nextHead];
        if (grows) {
            foods_.erase(nextHead);
        } else if (!snake_.empty()) {
            Coord tail = snake_.back();
            snake_.pop_back();
            auto it = occupancy_.find(tail);
            if (it != occupancy_.end()) {
                --(it->second);
                if (it->second == 0) {
                    occupancy_.erase(it);
                }
            }
        }
    }

    std::pair<int, int> diffAxis(const Coord& from, const Coord& to) const {
        for (int i = 0; i < dims_; ++i) {
            int delta = to.v[i] - from.v[i];
            if (delta != 0) {
                return {i, (delta > 0) ? 1 : -1};
            }
        }
        return {-1, 0};
    }

    bool ensureTailReachable(const Coord& candidateHead, bool grows) const {
        std::vector<Coord> futureBody;
        futureBody.reserve(snake_.size() + 1);
        futureBody.push_back(candidateHead);
        for (const auto& seg : snake_) {
            futureBody.push_back(seg);
        }
        if (!grows && !futureBody.empty()) {
            futureBody.pop_back();
        }

        std::unordered_map<Coord, int, CoordHash> futureOcc;
        futureOcc.reserve(futureBody.size() * 2 + 1);
        for (const auto& seg : futureBody) {
            ++futureOcc[seg];
        }

        const Coord& futureTail = futureBody.back();
        bool tailUnique = futureOcc[futureTail] == 1;

        auto isGoal = [&](const Coord& cell) {
            return cell == futureTail;
        };
        auto isBlocked = [&](const Coord& cell) {
            if (obstacles_.find(cell) != obstacles_.end()) {
                return true;
            }
            auto it = futureOcc.find(cell);
            if (it != futureOcc.end()) {
                if (tailUnique && cell == futureTail) {
                    return false;
                }
                return true;
            }
            return false;
        };

        std::vector<Coord> dummy;
        Coord reached{};
        return bfsGeneric(candidateHead, isGoal, isBlocked, dummy, reached, connectCap_);
    }

    bool findFoodPath(std::vector<Coord>& path) {
        if (foods_.empty()) {
            return false;
        }
        Coord tail = snake_.back();
        auto tailIt = occupancy_.find(tail);
        bool tailUnique = (tailIt != occupancy_.end() && tailIt->second == 1);

        auto isGoal = [&](const Coord& cell) {
            return foods_.find(cell) != foods_.end();
        };
        auto isBlocked = [&](const Coord& cell) {
            if (obstacles_.find(cell) != obstacles_.end()) {
                return true;
            }
            auto it = occupancy_.find(cell);
            if (it != occupancy_.end()) {
                if (tailUnique && cell == tail) {
                    return false;
                }
                return true;
            }
            return false;
        };

        Coord found{};
        return bfsGeneric(snake_.front(), isGoal, isBlocked, path, found, pathSearchCap_);
    }

    bool findTailPath(std::vector<Coord>& path) {
        if (snake_.size() <= 1) {
            return false;
        }
        Coord tail = snake_.back();
        auto tailIt = occupancy_.find(tail);
        if (tailIt == occupancy_.end() || tailIt->second != 1) {
            return false;
        }

        auto isGoal = [&](const Coord& cell) { return cell == tail; };
        auto isBlocked = [&](const Coord& cell) {
            if (obstacles_.find(cell) != obstacles_.end()) {
                return true;
            }
            auto it = occupancy_.find(cell);
            if (it != occupancy_.end()) {
                if (cell == tail) {
                    return false;
                }
                return true;
            }
            return false;
        };

        Coord found{};
        if (!bfsGeneric(snake_.front(), isGoal, isBlocked, path, found, tailSearchCap_)) {
            return false;
        }
        return path.size() >= 2;
    }

    template <typename GoalPredicate, typename BlockPredicate>
    bool bfsGeneric(const Coord& start,
                    GoalPredicate&& isGoal,
                    BlockPredicate&& isBlocked,
                    std::vector<Coord>& outPath,
                    Coord& goalOut,
                    std::size_t limit) const {
        if (!inBounds(start)) {
            return false;
        }
        std::deque<Coord> q;
        q.push_back(start);
        std::unordered_map<Coord, Coord, CoordHash> parent;
        parent.reserve(4096);
        parent.emplace(start, start);

        std::size_t expansions = 0;
        while (!q.empty() && expansions < limit) {
            Coord cur = q.front();
            q.pop_front();

            if (isGoal(cur)) {
                goalOut = cur;
                outPath.clear();
                Coord step = cur;
                while (!(step == start)) {
                    outPath.push_back(step);
                    step = parent.at(step);
                }
                outPath.push_back(start);
                std::reverse(outPath.begin(), outPath.end());
                return true;
            }

            ++expansions;

            for (int axis = 0; axis < dims_; ++axis) {
                if (bounds_[axis] <= 1) {
                    continue;
                }
                for (int dir = -1; dir <= 1; dir += 2) {
                    Coord next = cur;
                    next.v[axis] += dir;
                    if (!inBounds(next)) {
                        continue;
                    }
                    if (parent.find(next) != parent.end()) {
                        continue;
                    }
                    if (isBlocked(next) && !isGoal(next)) {
                        continue;
                    }
                    parent.emplace(next, cur);
                    q.push_back(next);
                }
            }
        }
        return false;
    }

    bool fallbackMove() {
        if (dims_ == 0 || turnsUsed_ >= maxTurns_) {
            return false;
        }
        Coord head = snake_.front();
        for (int offset = 0; offset < dims_; ++offset) {
            int axis = (axisCursor_ + offset) % dims_;
            if (bounds_[axis] <= 1) {
                continue;
            }
            int preferred = axisPreference_[axis];
            for (int attempt = 0; attempt < 2; ++attempt) {
                int dir = (attempt == 0) ? preferred : -preferred;
                Coord next = head;
                next.v[axis] += dir;
                if (executeMove(next)) {
                    axisPreference_[axis] = dir;
                    axisCursor_ = (axisCursor_ + 1) % dims_;
                    return true;
                }
            }
            axisPreference_[axis] = -axisPreference_[axis];
        }
        return false;
    }

private:
    int dims_;
    long long maxTurns_;
    std::array<int, MAX_DIM> bounds_;
    std::deque<Coord> snake_;
    std::unordered_map<Coord, int, CoordHash> occupancy_;
    std::unordered_set<Coord, CoordHash> foods_;
    std::unordered_set<Coord, CoordHash> obstacles_;
    std::vector<int> axisPreference_;
    int axisCursor_ = 0;
    long long turnsUsed_ = 0;
    std::vector<std::pair<int, int>> moves_;
    const std::size_t pathSearchCap_ = 2'000'000;
    const std::size_t tailSearchCap_ = 800'000;
    const std::size_t connectCap_ = 400'000;
};

Coord readCoord(int dims) {
    Coord c;
    for (int i = 0; i < dims; ++i) {
        std::cin >> c.v[i];
    }
    for (int i = dims; i < MAX_DIM; ++i) {
        c.v[i] = 0;
    }
    return c;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D;
    long long maxTurns;
    if (!(std::cin >> D >> maxTurns)) {
        return 0;
    }
    G_DIMS = D;

    std::array<int, MAX_DIM> bounds{};
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }
    for (int i = D; i < MAX_DIM; ++i) {
        bounds[i] = 1;
    }

    int snakeLen;
    std::cin >> snakeLen;
    Coord head = readCoord(D);
    int initialLen = std::max(1, snakeLen);

    std::deque<Coord> snake;
    snake.resize(initialLen, head);

    std::unordered_map<Coord, int, CoordHash> occupancy;
    occupancy.reserve(static_cast<std::size_t>(initialLen) * 2 + 16);
    for (const auto& seg : snake) {
        ++occupancy[seg];
    }

    int foodCount;
    std::cin >> foodCount;
    std::unordered_set<Coord, CoordHash> foods;
    foods.reserve(static_cast<std::size_t>(foodCount) * 2 + 1);
    for (int i = 0; i < foodCount; ++i) {
        Coord f = readCoord(D);
        foods.insert(f);
    }

    int obstacleCount;
    std::cin >> obstacleCount;
    std::unordered_set<Coord, CoordHash> obstacles;
    obstacles.reserve(static_cast<std::size_t>(obstacleCount) * 2 + 1);
    for (int i = 0; i < obstacleCount; ++i) {
        Coord o = readCoord(D);
        obstacles.insert(o);
    }

    SnakeSolver solver(D, maxTurns, bounds, std::move(snake),
                       std::move(occupancy), std::move(foods), std::move(obstacles));
    solver.solve();
    return 0;
}
```
