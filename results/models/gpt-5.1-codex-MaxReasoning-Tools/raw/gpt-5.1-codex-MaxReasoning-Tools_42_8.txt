OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_Input_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_pv Output
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%color_in = OpVariable %ptr_Input_arr_v3 Input
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_oneThird = OpConstant %float 0.333333343
%float_shrink = OpConstant %float 0.699999988
%main = OpFunction %void None %func
%entry = OpLabel
%ptr_p0 = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%tmp0 = OpFAdd %v4float %p0 %p1
%sum = OpFAdd %v4float %tmp0 %p2
%centroid = OpVectorTimesScalar %v4float %sum %float_oneThird
%ptr_out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%diff0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %diff0 %float_shrink
%new0 = OpFAdd %v4float %centroid %scaled0
OpStore %ptr_out_pos %new0
%ptr_color0 = OpAccessChain %ptr_Input_v3 %color_in %int_0
%color0 = OpLoad %v3float %ptr_color0
OpStore %outColor %color0
OpEmitVertex
%diff1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %diff1 %float_shrink
%new1 = OpFAdd %v4float %centroid %scaled1
OpStore %ptr_out_pos %new1
%ptr_color1 = OpAccessChain %ptr_Input_v3 %color_in %int_1
%color1 = OpLoad %v3float %ptr_color1
OpStore %outColor %color1
OpEmitVertex
%diff2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %diff2 %float_shrink
%new2 = OpFAdd %v4float %centroid %scaled2
OpStore %ptr_out_pos %new2
%ptr_color2 = OpAccessChain %ptr_Input_v3 %color_in %int_2
%color2 = OpLoad %v3float %ptr_color2
OpStore %outColor %color2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
