const fs = require('fs');

const NEAREST_NEIGHBOR_THRESHOLD = 1000;
const FULL_TWO_OPT_THRESHOLD = 1200;
const MEDIUM_TWO_OPT_THRESHOLD = 4000;
const DIST_EPSILON = 1e-9;
const TAU = Math.PI * 2;

(function main() {
    let rawInput = '';
    try {
        rawInput = fs.readFileSync(0, 'utf8');
    } catch (err) {
        rawInput = '';
    }

    if (!rawInput || !rawInput.trim()) {
        console.log(JSON.stringify({ routes: [] }));
        return;
    }

    let data;
    try {
        data = JSON.parse(rawInput);
    } catch (err) {
        console.log(JSON.stringify({ routes: [] }));
        return;
    }

    const depotArray = Array.isArray(data.depot) ? data.depot : [0, 0];
    const depot = {
        x: toNumberSafe(depotArray[0]),
        y: toNumberSafe(depotArray[1])
    };

    const packagesInput = Array.isArray(data.packages) ? data.packages : [];
    const normalizedPackages = packagesInput.map((pkg, idx) => ({
        id: pkg && pkg.id !== undefined ? pkg.id : idx,
        x: toNumberSafe(pkg && pkg.x),
        y: toNumberSafe(pkg && pkg.y),
        _seq: idx
    }));

    if (normalizedPackages.length === 0) {
        console.log(JSON.stringify({ routes: [] }));
        return;
    }

    let truckLimit = Number(data.truckCount);
    if (!Number.isFinite(truckLimit) || truckLimit < 1) {
        truckLimit = 1;
    } else {
        truckLimit = Math.floor(truckLimit);
        if (truckLimit < 1) {
            truckLimit = 1;
        }
    }

    const routeCount = Math.max(1, Math.min(truckLimit, normalizedPackages.length));

    const enriched = normalizedPackages.map(pkg => enrichPackage(pkg, depot));
    enriched.sort(compareByAngleAndRadius);
    const sweepOrdered = rotateByLargestGap(enriched);

    const groups = splitIntoGroups(sweepOrdered, routeCount);
    let objectRoutes = [];

    for (const group of groups) {
        if (group.length === 0) {
            continue;
        }
        const plannedRoute = planRoute(group, depot);
        if (plannedRoute.length > 0) {
            objectRoutes.push(plannedRoute);
        }
    }

    let coverageOk = objectRoutes.length > 0 && allPackagesCovered(objectRoutes, normalizedPackages.length);
    if (!coverageOk) {
        objectRoutes = createFallbackRoutes(sweepOrdered, routeCount);
        coverageOk = allPackagesCovered(objectRoutes, normalizedPackages.length);
    }
    if (!coverageOk) {
        objectRoutes = [sweepOrdered.slice()];
    }

    const routes = objectRoutes.map(route => route.map(node => node.id));
    console.log(JSON.stringify({ routes }));
})();

function toNumberSafe(value) {
    const num = Number(value);
    return Number.isFinite(num) ? num : 0;
}

function enrichPackage(pkg, depot) {
    const dx = pkg.x - depot.x;
    const dy = pkg.y - depot.y;
    return {
        id: pkg.id,
        x: pkg.x,
        y: pkg.y,
        _seq: pkg._seq,
        angle: Math.atan2(dy, dx),
        radial: dx * dx + dy * dy
    };
}

function compareByAngleAndRadius(a, b) {
    if (a.angle === b.angle) {
        if (a.radial === b.radial) {
            return a._seq - b._seq;
        }
        return a.radial - b.radial;
    }
    return a.angle - b.angle;
}

function rotateByLargestGap(points) {
    if (points.length <= 1) {
        return points.slice();
    }
    let maxGap = -Infinity;
    let splitIndex = 0;
    for (let i = 0; i < points.length; i++) {
        const currentAngle = points[i].angle;
        const nextAngle = i === points.length - 1 ? points[0].angle + TAU : points[i + 1].angle;
        const gap = nextAngle - currentAngle;
        if (gap > maxGap) {
            maxGap = gap;
            splitIndex = (i + 1) % points.length;
        }
    }
    if (splitIndex === 0) {
        return points.slice();
    }
    return points.slice(splitIndex).concat(points.slice(0, splitIndex));
}

function splitIntoGroups(list, groupCount) {
    const groups = [];
    if (groupCount <= 0) {
        return groups;
    }
    const baseSize = Math.floor(list.length / groupCount);
    const remainder = list.length % groupCount;
    let index = 0;
    for (let i = 0; i < groupCount; i++) {
        const size = baseSize + (i < remainder ? 1 : 0);
        const group = [];
        for (let j = 0; j < size && index < list.length; j++, index++) {
            group.push(list[index]);
        }
        if (group.length > 0) {
            groups.push(group);
        }
    }
    while (index < list.length && groups.length > 0) {
        groups[groups.length - 1].push(list[index++]);
    }
    return groups;
}

function planRoute(group, depot) {
    if (group.length <= 1) {
        return group.slice();
    }

    let ordered;
    if (group.length <= NEAREST_NEIGHBOR_THRESHOLD) {
        ordered = nearestNeighborOrder(group, depot);
    } else {
        ordered = group.slice();
    }

    const size = ordered.length;
    let options;
    if (size <= FULL_TWO_OPT_THRESHOLD) {
        options = undefined;
    } else if (size <= MEDIUM_TWO_OPT_THRESHOLD) {
        options = {
            maxPasses: 4,
            evaluationBudget: Math.max(20000, size * 25),
            maxSegmentSpan: 300
        };
    } else {
        options = {
            maxPasses: 2,
            evaluationBudget: Math.max(20000, size * 15),
            maxSegmentSpan: 200
        };
    }

    return twoOptImprove(ordered, depot, options);
}

function nearestNeighborOrder(group, depot) {
    const remaining = group.slice();
    const route = [];
    let currentPoint = depot;
    while (remaining.length > 0) {
        let bestIdx = 0;
        let bestDist = Infinity;
        for (let i = 0; i < remaining.length; i++) {
            const node = remaining[i];
            const dist = squaredDistance(node, currentPoint);
            if (dist < bestDist) {
                bestDist = dist;
                bestIdx = i;
            } else if (dist === bestDist && node._seq < remaining[bestIdx]._seq) {
                bestIdx = i;
            }
        }
        const lastIndex = remaining.length - 1;
        const nextNode = remaining[bestIdx];
        if (bestIdx !== lastIndex) {
            remaining[bestIdx] = remaining[lastIndex];
        }
        remaining.pop();
        route.push(nextNode);
        currentPoint = nextNode;
    }
    return route;
}

function squaredDistance(a, b) {
    const dx = a.x - b.x;
    const dy = a.y - b.y;
    return dx * dx + dy * dy;
}

function twoOptImprove(route, depot, options) {
    const n = route.length;
    if (n < 3) {
        return route;
    }

    options = options || {};

    let passes = Number.isFinite(options.maxPasses) ? Math.floor(options.maxPasses) : 6;
    if (passes < 1) {
        passes = 1;
    }

    let budget = Number.isFinite(options.evaluationBudget) ? Math.floor(options.evaluationBudget) : Math.max(10000, n * 60);
    if (budget < 1) {
        budget = 1;
    }

    let spanValue = Number.isFinite(options.maxSegmentSpan) ? Math.floor(options.maxSegmentSpan) : n;
    if (spanValue < 1) {
        spanValue = 1;
    }
    const spanLimit = spanValue;

    let evaluations = 0;

    for (let pass = 0; pass < passes; pass++) {
        let improvedInPass = false;
        for (let i = 0; i < n - 1 && evaluations < budget; i++) {
            let prev = i === 0 ? depot : route[i - 1];
            let current = route[i];
            let prevCurrentDist = euclidean(prev, current);
            const limit = spanLimit >= n ? n - 1 : Math.min(n - 1, i + spanLimit);
            let k = i + 1;
            while (k <= limit && evaluations < budget) {
                const next = route[k];
                const nextAfter = k === n - 1 ? depot : route[k + 1];
                evaluations++;
                const existing = prevCurrentDist + euclidean(next, nextAfter);
                const candidate = euclidean(prev, next) + euclidean(current, nextAfter);
                if (candidate + DIST_EPSILON < existing) {
                    reverseSegment(route, i, k);
                    improvedInPass = true;
                    current = route[i];
                    prev = i === 0 ? depot : route[i - 1];
                    prevCurrentDist = euclidean(prev, current);
                    k = i + 1;
                    continue;
                }
                k++;
            }
        }
        if (!improvedInPass || evaluations >= budget) {
            break;
        }
    }

    return route;
}

function reverseSegment(route, left, right) {
    while (left < right) {
        const temp = route[left];
        route[left] = route[right];
        route[right] = temp;
        left++;
        right--;
    }
}

function euclidean(a, b) {
    const dx = a.x - b.x;
    const dy = a.y - b.y;
    return Math.sqrt(dx * dx + dy * dy);
}

function createFallbackRoutes(list, routeCount) {
    if (list.length === 0) {
        return [];
    }
    const safeCount = Math.max(1, Math.min(routeCount, list.length));
    const groups = splitIntoGroups(list, safeCount);
    return groups.map(group => group.slice());
}

function allPackagesCovered(routes, totalCount) {
    if (routes.length === 0 && totalCount > 0) {
        return false;
    }
    const seen = new Set();
    for (const route of routes) {
        for (const node of route) {
            if (!seen.has(node._seq)) {
                seen.add(node._seq);
                if (seen.size === totalCount) {
                    return true;
                }
            }
        }
    }
    return seen.size === totalCount;
}
