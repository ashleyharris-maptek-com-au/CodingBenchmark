OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inUV %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%ptrInput_v2float = OpTypePointer Input %v2float
%ptrOutput_v4float = OpTypePointer Output %v4float
%inUV = OpVariable %ptrInput_v2float Input
%outColor = OpVariable %ptrOutput_v4float Output
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%main = OpFunction %void None %func
%entry = OpLabel
%uv = OpLoad %v2float %inUV
%ux = OpCompositeExtract %float %uv 0
%uy = OpCompositeExtract %float %uv 1
%color = OpCompositeConstruct %v4float %ux %uy %float_0 %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
