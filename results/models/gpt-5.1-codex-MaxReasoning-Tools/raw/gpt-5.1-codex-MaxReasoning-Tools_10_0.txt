Below is a scalable MaxRects-based solver that adapts its heuristics to problem size (multiple ordering strategies for small inputs, aggressive free-rectangle pruning for large ones). Board indices in the output are zero-based as requested.

```cpp
#include <algorithm>
#include <cstddef>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <utility>
#include <vector>

struct Item {
    int idx;
    int w;
    int h;
    long long area;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x;
    int y;
    int w;
    int h;
};

struct FitChoice {
    bool valid = false;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    bool rotated = false;
    long long areaFit = 0;
    int shortFit = 0;
    int longFit = 0;
};

struct CandidatePlacement {
    bool valid = false;
    int board = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    bool rotated = false;
    long long areaFit = 0;
    int shortFit = 0;
    int longFit = 0;
};

struct PackResult {
    int boardsUsed = 0;
    long long residualArea = 0;
    std::vector<Placement> placements;
};

static bool betterFit(const FitChoice &a, const FitChoice &b) {
    if (!a.valid) return false;
    if (!b.valid) return true;
    if (a.areaFit != b.areaFit) return a.areaFit < b.areaFit;
    if (a.longFit != b.longFit) return a.longFit < b.longFit;
    if (a.shortFit != b.shortFit) return a.shortFit < b.shortFit;
    if (a.y != b.y) return a.y < b.y;
    return a.x < b.x;
}

static bool betterPlacement(const CandidatePlacement &a, const CandidatePlacement &b) {
    if (!a.valid) return false;
    if (!b.valid) return true;
    if (a.areaFit != b.areaFit) return a.areaFit < b.areaFit;
    if (a.longFit != b.longFit) return a.longFit < b.longFit;
    if (a.shortFit != b.shortFit) return a.shortFit < b.shortFit;
    if (a.y != b.y) return a.y < b.y;
    if (a.x != b.x) return a.x < b.x;
    return a.board < b.board;
}

static std::size_t determineFreeRectLimit(int nRect) {
    if (nRect <= 32) return 512;
    if (nRect <= 128) return 384;
    if (nRect <= 512) return 256;
    if (nRect <= 2000) return 160;
    if (nRect <= 10000) return 120;
    return 96;
}

static std::vector<int> makeLongestOrder(const std::vector<Item> &items) {
    std::vector<int> order(items.size());
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        int lhsMax = std::max(items[lhs].w, items[lhs].h);
        int rhsMax = std::max(items[rhs].w, items[rhs].h);
        if (lhsMax != rhsMax) return lhsMax > rhsMax;
        if (items[lhs].area != items[rhs].area) return items[lhs].area > items[rhs].area;
        int lhsMin = std::min(items[lhs].w, items[lhs].h);
        int rhsMin = std::min(items[rhs].w, items[rhs].h);
        if (lhsMin != rhsMin) return lhsMin > rhsMin;
        return lhs < rhs;
    });
    return order;
}

static std::vector<int> makeAreaOrder(const std::vector<Item> &items) {
    std::vector<int> order(items.size());
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        if (items[lhs].area != items[rhs].area) return items[lhs].area > items[rhs].area;
        int lhsMax = std::max(items[lhs].w, items[lhs].h);
        int rhsMax = std::max(items[rhs].w, items[rhs].h);
        if (lhsMax != rhsMax) return lhsMax > rhsMax;
        return lhs < rhs;
    });
    return order;
}

static std::vector<int> makeDimensionOrder(const std::vector<Item> &items, bool widthFirst) {
    std::vector<int> order(items.size());
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        int lhsKey = widthFirst ? items[lhs].w : items[lhs].h;
        int rhsKey = widthFirst ? items[rhs].w : items[rhs].h;
        if (lhsKey != rhsKey) return lhsKey > rhsKey;
        if (items[lhs].area != items[rhs].area) return items[lhs].area > items[rhs].area;
        return lhs < rhs;
    });
    return order;
}

class Board {
public:
    Board() = default;
    Board(int w, int h) { init(w, h); }

    void init(int w, int h) {
        freeRects.clear();
        freeRects.reserve(64);
        freeRects.push_back({0, 0, w, h});
        recompute();
    }

    bool canFitCandidate(int rectW, int rectH, long long rectArea) const {
        if (freeRects.empty()) return false;
        if (rectArea > freeArea) return false;
        bool fitsNormal = (rectW <= maxFreeW && rectH <= maxFreeH);
        bool fitsRot = (rectH <= maxFreeW && rectW <= maxFreeH);
        return fitsNormal || fitsRot;
    }

    bool findBestFit(int rectW, int rectH, FitChoice &out) const {
        FitChoice best;
        for (const auto &fre : freeRects) {
            if (rectW <= fre.w && rectH <= fre.h) {
                FitChoice candidate;
                candidate.valid = true;
                candidate.x = fre.x;
                candidate.y = fre.y;
                candidate.w = rectW;
                candidate.h = rectH;
                candidate.rotated = false;
                candidate.areaFit = 1LL * fre.w * fre.h - 1LL * rectW * rectH;
                int leftoverW = fre.w - rectW;
                int leftoverH = fre.h - rectH;
                candidate.shortFit = std::min(leftoverW, leftoverH);
                candidate.longFit = std::max(leftoverW, leftoverH);
                if (betterFit(candidate, best)) best = candidate;
            }
            if (rectH <= fre.w && rectW <= fre.h && rectW != rectH) {
                FitChoice candidate;
                candidate.valid = true;
                candidate.x = fre.x;
                candidate.y = fre.y;
                candidate.w = rectH;
                candidate.h = rectW;
                candidate.rotated = true;
                candidate.areaFit = 1LL * fre.w * fre.h - 1LL * rectW * rectH;
                int leftoverW = fre.w - candidate.w;
                int leftoverH = fre.h - candidate.h;
                candidate.shortFit = std::min(leftoverW, leftoverH);
                candidate.longFit = std::max(leftoverW, leftoverH);
                if (betterFit(candidate, best)) best = candidate;
            }
        }
        if (best.valid) {
            out = best;
            return true;
        }
        return false;
    }

    void placeRectangle(int x, int y, int w, int h, std::size_t limit) {
        if (limit == 0) limit = 1;
        if (freeRects.capacity() < limit * 2) {
            freeRects.reserve(limit * 2);
        }
        FreeRect used{x, y, w, h};
        splitFreeRectangles(used);
        pruneFreeRects();
        enforceLimit(limit);
        recompute();
    }

    long long getFreeArea() const { return freeArea; }

private:
    std::vector<FreeRect> freeRects;
    long long freeArea = 0;
    int maxFreeW = 0;
    int maxFreeH = 0;

    void splitFreeRectangles(const FreeRect &used) {
        for (std::size_t i = 0; i < freeRects.size(); ++i) {
            FreeRect node = freeRects[i];
            if (splitFreeNode(node, used)) {
                freeRects[i] = freeRects.back();
                freeRects.pop_back();
                if (i > 0) --i;
            }
        }
    }

    bool splitFreeNode(const FreeRect &freeRect, const FreeRect &used) {
        if (used.x >= freeRect.x + freeRect.w || used.x + used.w <= freeRect.x ||
            used.y >= freeRect.y + freeRect.h || used.y + used.h <= freeRect.y) {
            return false;
        }

        const int freeRight = freeRect.x + freeRect.w;
        const int freeTop = freeRect.y + freeRect.h;
        const int usedRight = used.x + used.w;
        const int usedTop = used.y + used.h;

        if (used.x < freeRight && usedRight > freeRect.x) {
            if (used.y > freeRect.y && used.y < freeTop) {
                FreeRect newRect = freeRect;
                newRect.h = used.y - freeRect.y;
                addFreeRect(newRect);
            }
            if (usedTop < freeTop) {
                FreeRect newRect = freeRect;
                newRect.y = usedTop;
                newRect.h = freeTop - usedTop;
                addFreeRect(newRect);
            }
        }

        if (used.y < freeTop && usedTop > freeRect.y) {
            if (used.x > freeRect.x && used.x < freeRight) {
                FreeRect newRect = freeRect;
                newRect.w = used.x - freeRect.x;
                addFreeRect(newRect);
            }
            if (usedRight < freeRight) {
                FreeRect newRect = freeRect;
                newRect.x = usedRight;
                newRect.w = freeRight - usedRight;
                addFreeRect(newRect);
            }
        }
        return true;
    }

    void addFreeRect(const FreeRect &rect) {
        if (rect.w <= 0 || rect.h <= 0) return;
        freeRects.push_back(rect);
    }

    static bool isContained(const FreeRect &a, const FreeRect &b) {
        return a.x >= b.x && a.y >= b.y &&
               a.x + a.w <= b.x + b.w &&
               a.y + a.h <= b.y + b.h;
    }

    void pruneFreeRects() {
        std::size_t i = 0;
        while (i < freeRects.size()) {
            bool erased = false;
            std::size_t j = i + 1;
            while (j < freeRects.size()) {
                if (isContained(freeRects[i], freeRects[j])) {
                    freeRects[i] = freeRects.back();
                    freeRects.pop_back();
                    erased = true;
                    break;
                }
                if (isContained(freeRects[j], freeRects[i])) {
                    freeRects[j] = freeRects.back();
                    freeRects.pop_back();
                } else {
                    ++j;
                }
            }
            if (!erased) ++i;
        }
    }

    void enforceLimit(std::size_t limit) {
        if (limit == 0 || freeRects.size() <= limit) return;
        auto nth = freeRects.begin() + limit;
        std::nth_element(freeRects.begin(), nth, freeRects.end(),
                         [](const FreeRect &lhs, const FreeRect &rhs) {
                             long long lhsArea = 1LL * lhs.w * lhs.h;
                             long long rhsArea = 1LL * rhs.w * rhs.h;
                             if (lhsArea == rhsArea) {
                                 int lhsMax = std::max(lhs.w, lhs.h);
                                 int rhsMax = std::max(rhs.w, rhs.h);
                                 return lhsMax > rhsMax;
                             }
                             return lhsArea > rhsArea;
                         });
        freeRects.resize(limit);
    }

    void recompute() {
        freeArea = 0;
        maxFreeW = 0;
        maxFreeH = 0;
        for (const auto &rect : freeRects) {
            freeArea += 1LL * rect.w * rect.h;
            if (rect.w > maxFreeW) maxFreeW = rect.w;
            if (rect.h > maxFreeH) maxFreeH = rect.h;
        }
    }
};

class PackerCore {
public:
    PackerCore(int boardW, int boardH, const std::vector<Item> &items,
               long long totalArea, std::size_t freeLimit)
        : boardW(boardW),
          boardH(boardH),
          items(items),
          freeLimit(freeLimit),
          totalArea(totalArea) {
        boardArea = 1LL * boardW * boardH;
        if (boardArea <= 0) boardArea = 1;
    }

    PackResult run(const std::vector<int> &order) const {
        PackResult result;
        std::size_t n = items.size();
        result.placements.assign(n, Placement{});

        if (n == 0) {
            result.boardsUsed = 0;
            result.residualArea = 0;
            return result;
        }

        std::vector<Board> boards;
        std::size_t estimated = static_cast<std::size_t>(totalArea / boardArea + 2);
        estimated = std::min<std::size_t>(estimated, n + 1);
        boards.reserve(std::max<std::size_t>(1, estimated));

        for (int idx : order) {
            const Item &item = items[idx];
            CandidatePlacement chosen = placeItem(item, boards);
            if (chosen.board < 0 || static_cast<std::size_t>(chosen.board) >= boards.size()) {
                result.placements[item.idx] = {-1, 0, 0, 0};
                continue;
            }
            boards[chosen.board].placeRectangle(chosen.x, chosen.y, chosen.w, chosen.h, freeLimit);
            result.placements[item.idx] = {chosen.board, chosen.x, chosen.y, chosen.rotated ? 1 : 0};
        }

        result.boardsUsed = static_cast<int>(boards.size());
        long long freeSum = 0;
        for (const auto &board : boards) freeSum += board.getFreeArea();
        result.residualArea = freeSum;
        return result;
    }

private:
    CandidatePlacement placeItem(const Item &item, std::vector<Board> &boards) const {
        CandidatePlacement best;
        bool perfect = false;
        for (std::size_t b = 0; b < boards.size() && !perfect; ++b) {
            Board &board = boards[b];
            if (!board.canFitCandidate(item.w, item.h, item.area)) continue;
            FitChoice fit;
            if (!board.findBestFit(item.w, item.h, fit)) continue;
            CandidatePlacement candidate = convert(fit, static_cast<int>(b));
            if (!best.valid || betterPlacement(candidate, best)) {
                best = candidate;
                if (candidate.areaFit == 0 && candidate.longFit == 0 && candidate.shortFit == 0) {
                    perfect = true;
                }
            }
        }
        if (!best.valid) {
            boards.emplace_back(boardW, boardH);
            Board &board = boards.back();
            FitChoice fit;
            if (board.findBestFit(item.w, item.h, fit)) {
                best = convert(fit, static_cast<int>(boards.size() - 1));
            } else {
                best = forcePlace(item, static_cast<int>(boards.size() - 1));
            }
        }
        best.valid = true;
        return best;
    }

    CandidatePlacement convert(const FitChoice &fit, int boardIndex) const {
        CandidatePlacement out;
        out.valid = fit.valid;
        out.board = boardIndex;
        out.x = fit.x;
        out.y = fit.y;
        out.w = fit.w;
        out.h = fit.h;
        out.rotated = fit.rotated;
        out.areaFit = fit.areaFit;
        out.shortFit = fit.shortFit;
        out.longFit = fit.longFit;
        return out;
    }

    CandidatePlacement forcePlace(const Item &item, int boardIndex) const {
        CandidatePlacement fallback;
        fallback.valid = true;
        fallback.board = boardIndex;
        fallback.x = 0;
        fallback.y = 0;
        bool canRotate = (item.h <= boardW && item.w <= boardH);
        bool canUpright = (item.w <= boardW && item.h <= boardH);
        if (!canUpright && canRotate) {
            fallback.rotated = true;
            fallback.w = std::min(item.h, boardW);
            fallback.h = std::min(item.w, boardH);
        } else {
            fallback.rotated = false;
            fallback.w = std::min(item.w, boardW);
            fallback.h = std::min(item.h, boardH);
        }
        fallback.areaFit = 1LL * boardW * boardH - 1LL * fallback.w * fallback.h;
        fallback.shortFit = std::min(boardW - fallback.w, boardH - fallback.h);
        fallback.longFit = std::max(boardW - fallback.w, boardH - fallback.h);
        return fallback;
    }

    int boardW;
    int boardH;
    const std::vector<Item> &items;
    std::size_t freeLimit;
    long long totalArea;
    long long boardArea;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    int boardW, boardH;
    if (!(std::cin >> n >> boardW >> boardH)) {
        return 0;
    }

    std::vector<Item> items;
    if (n > 0) items.reserve(static_cast<std::size_t>(n));
    long long totalArea = 0;

    for (int i = 0; i < n; ++i) {
        int w, h;
        std::cin >> w >> h;
        Item item{i, w, h, 1LL * w * h};
        items.push_back(item);
        totalArea += item.area;
    }

    if (n == 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::size_t freeLimit = determineFreeRectLimit(n);
    PackerCore solver(boardW, boardH, items, totalArea, freeLimit);

    std::vector<std::vector<int>> candidateOrders;
    candidateOrders.reserve(3);
    auto addOrder = [&](std::vector<int> order) {
        for (const auto &existing : candidateOrders) {
            if (order == existing) return;
        }
        candidateOrders.push_back(std::move(order));
    };

    addOrder(makeLongestOrder(items));
    if (n <= 128) {
        addOrder(makeAreaOrder(items));
        bool widthFirst = boardW >= boardH;
        addOrder(makeDimensionOrder(items, widthFirst));
    }

    if (candidateOrders.empty()) {
        candidateOrders.push_back(makeLongestOrder(items));
    }

    long long boardAreaLL = 1LL * boardW * boardH;
    if (boardAreaLL <= 0) boardAreaLL = 1;
    long long lowerBoundBoards = (totalArea == 0) ? 0 : (totalArea + boardAreaLL - 1) / boardAreaLL;

    PackResult best;
    bool haveBest = false;

    for (const auto &order : candidateOrders) {
        PackResult res = solver.run(order);
        if (!haveBest ||
            res.boardsUsed < best.boardsUsed ||
            (res.boardsUsed == best.boardsUsed && res.residualArea < best.residualArea)) {
            best = std::move(res);
            haveBest = true;
        }
        if (haveBest && best.boardsUsed <= lowerBoundBoards) {
            break;
        }
    }

    if (!haveBest) {
        best.boardsUsed = 0;
        best.placements.assign(static_cast<std::size_t>(n), Placement{});
    }

    std::cout << best.boardsUsed << '\n';
    for (int i = 0; i < n; ++i) {
        const Placement &p = best.placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
```
