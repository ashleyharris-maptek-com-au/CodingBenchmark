OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%gl_PerVertex_in = OpTypeStruct %v4float
%arr3_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_pv = OpTypePointer Input %arr3_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%arr3_v3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3 = OpTypePointer Input %arr3_v3
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.0800000007
%float_0_333333343 = OpConstant %float 0.333333343
%gl_in = OpVariable %ptr_Input_arr_pv Input
%in_worldPos = OpVariable %ptr_Input_arr_v3 Input
%in_normal = OpVariable %ptr_Input_arr_v3 Input
%in_color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_glpv Output
%outColor = OpVariable %ptr_Output_v3 Output
%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%center = OpVectorTimesScalar %v4float %sum012 %float_0_333333343
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%normal = OpExtInst %v3float %glsl Normalize %cross
%offset = OpVectorTimesScalar %v3float %normal %float_0_08
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset_v4_init = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0
%center_zero = OpVectorTimesScalar %v4float %center %float_0
%offset_v4 = OpFAdd %v4float %offset_v4_init %center_zero
%p0_shift = OpFAdd %v4float %p0 %offset_v4
%p1_shift = OpFAdd %v4float %p1 %offset_v4
%p2_shift = OpFAdd %v4float %p2 %offset_v4
%color0_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_0
%color0 = OpLoad %v3float %color0_ptr
%color1_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_2
%color2 = OpLoad %v3float %color2_ptr
%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr %p0_shift
OpStore %outColor %color0
OpEmitVertex
OpStore %out_pos_ptr %p1_shift
OpStore %outColor %color1
OpEmitVertex
OpStore %out_pos_ptr %p2_shift
OpStore %outColor %color2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
