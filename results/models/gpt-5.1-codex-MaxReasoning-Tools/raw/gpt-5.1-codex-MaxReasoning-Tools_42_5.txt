OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpSource GLSL 450
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
%arr_worldPos = OpTypeArray %v3float %uint_2
%arr_normal = OpTypeArray %v3float %uint_2
%arr_color = OpTypeArray %v3float %uint_2
%ptr_Input_arr_worldPos = OpTypePointer Input %arr_worldPos
%ptr_Input_arr_normal = OpTypePointer Input %arr_normal
%ptr_Input_arr_color = OpTypePointer Input %arr_color
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_eps = OpConstant %float 0.00001
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_worldPos Input
%normal = OpVariable %ptr_Input_arr_normal Input
%color = OpVariable %ptr_Input_arr_color Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3 Output
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_worldPos ArrayStride 16
OpDecorate %arr_normal ArrayStride 16
OpDecorate %arr_color ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%main = OpFunction %void None %func
%entry = OpLabel
%outPosPtr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%gl_p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%gl_p1 = OpLoad %v4float %p1_ptr
%world0_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_0
%world0 = OpLoad %v3float %world0_ptr
%world1_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_1
%world1 = OpLoad %v3float %world1_ptr
%norm0_ptr = OpAccessChain %ptr_Input_v3 %normal %int_0
%norm0 = OpLoad %v3float %norm0_ptr
%norm1_ptr = OpAccessChain %ptr_Input_v3 %normal %int_1
%norm1 = OpLoad %v3float %norm1_ptr
%color0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%c0 = OpLoad %v3float %color0_ptr
%color1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c1 = OpLoad %v3float %color1_ptr
%world0_x = OpCompositeExtract %float %world0 0
%world0_y = OpCompositeExtract %float %world0 1
%world0_z = OpCompositeExtract %float %world0 2
%w0_w = OpCompositeExtract %float %gl_p0 3
%p0 = OpCompositeConstruct %v4float %world0_x %world0_y %world0_z %w0_w
%world1_x = OpCompositeExtract %float %world1 0
%world1_y = OpCompositeExtract %float %world1 1
%world1_z = OpCompositeExtract %float %world1 2
%w1_w = OpCompositeExtract %float %gl_p1 3
%p1 = OpCompositeConstruct %v4float %world1_x %world1_y %world1_z %w1_w
%dir = OpFSub %v4float %p1 %p0
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%dx_sq = OpFMul %float %dx %dx
%dy_sq = OpFMul %float %dy %dy
%len2 = OpFAdd %float %dx_sq %dy_sq
%dotnorm = OpDot %float %norm0 %norm1
%zeroDot = OpFMul %float %dotnorm %float_0
%len2_adj = OpFAdd %float %len2 %zeroDot
%len = OpExtInst %float %glsl Sqrt %len2_adj
%lenSafe = OpExtInst %float %glsl FMax %len %float_eps
%neg_dy = OpFNegate %float %dy
%perpX = OpFDiv %float %neg_dy %lenSafe
%perpY = OpFDiv %float %dx %lenSafe
%perp = OpCompositeConstruct %v2float %perpX %perpY
%offsetS2 = OpVectorTimesScalar %v2float %perp %float_0_015
%offsetA2 = OpVectorTimesScalar %v2float %perp %float_0_05
%offsetSx = OpCompositeExtract %float %offsetS2 0
%offsetSy = OpCompositeExtract %float %offsetS2 1
%offsetS4 = OpCompositeConstruct %v4float %offsetSx %offsetSy %float_0 %float_0
%offsetAx = OpCompositeExtract %float %offsetA2 0
%offsetAy = OpCompositeExtract %float %offsetA2 1
%offsetA4 = OpCompositeConstruct %v4float %offsetAx %offsetAy %float_0 %float_0
%dirScaled = OpVectorTimesScalar %v4float %dir %float_0_75
%mid = OpFAdd %v4float %p0 %dirScaled
%p0_plus = OpFAdd %v4float %p0 %offsetS4
%p0_minus = OpFSub %v4float %p0 %offsetS4
%mid_plus = OpFAdd %v4float %mid %offsetS4
%mid_minus = OpFSub %v4float %mid %offsetS4
%mid_arrow_plus = OpFAdd %v4float %mid %offsetA4
%mid_arrow_minus = OpFSub %v4float %mid %offsetA4
OpStore %outPosPtr %p0_plus
OpStore %outColor %c0
OpEmitVertex
OpStore %outPosPtr %p0_minus
OpStore %outColor %c0
OpEmitVertex
OpStore %outPosPtr %mid_plus
OpStore %outColor %c0
OpEmitVertex
OpStore %outPosPtr %mid_minus
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive
OpStore %outPosPtr %mid_arrow_plus
OpStore %outColor %c1
OpEmitVertex
OpStore %outPosPtr %mid_arrow_minus
OpStore %outColor %c1
OpEmitVertex
OpStore %outPosPtr %p1
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
