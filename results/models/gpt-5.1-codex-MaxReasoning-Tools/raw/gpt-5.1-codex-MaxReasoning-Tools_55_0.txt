Here's a self-contained C++20 program that parses a FEN string, runs an iterative-deepening alpha-beta search (with quiescence search, history/killer move ordering, basic evaluation, and full legality checking), and prints the best move it can find within the provided timeout:

inline uint64_t bit(int sq) { return 1ULL << sq; }

inline int popLSB(uint64_t &bb) {
    int sq = std::countr_zero(bb);
    bb &= bb - 1;
    return sq;
}

inline int mirrorSquare(int sq) { return sq ^ 56; }

inline int encodePiece(int color, int piece) { return color == WHITE ? (piece + 1) : -(piece + 1); }
inline int decodePieceType(int code) { return std::abs(code) - 1; }
inline int decodePieceColor(int code) { return code > 0 ? WHITE : BLACK; }

struct Move {
    uint8_t from = 0;
    uint8_t to = 0;
    uint8_t piece = NO_PIECE;
    uint8_t promotion = NO_PIECE;
    uint8_t flags = 0;
    uint8_t captured = NO_PIECE;

    bool isValid() const { return piece != NO_PIECE; }
    bool isCapture() const { return (flags & FLAG_CAPTURE) != 0; }
    bool isPromotion() const { return (flags & FLAG_PROMOTION) != 0; }
    bool isCastle() const { return (flags & FLAG_CASTLING) != 0; }
    bool isEnPassant() const { return (flags & FLAG_EN_PASSANT) != 0; }
    bool isDoublePush() const { return (flags & FLAG_DOUBLE_PAWN) != 0; }

    bool operator==(const Move &other) const {
        return from == other.from && to == other.to &&
               piece == other.piece && promotion == other.promotion &&
               flags == other.flags;
    }
};

uint64_t rookAttacks(int square, uint64_t occ) {
    uint64_t attacks = 0ULL;
    int rank = square / 8;
    int file = square % 8;
    for (int r = rank + 1; r <= 7; ++r) {
        int target = r * 8 + file;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    for (int r = rank - 1; r >= 0; --r) {
        int target = r * 8 + file;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    for (int f = file + 1; f <= 7; ++f) {
        int target = rank * 8 + f;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    for (int f = file - 1; f >= 0; --f) {
        int target = rank * 8 + f;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    return attacks;
}

uint64_t bishopAttacks(int square, uint64_t occ) {
    uint64_t attacks = 0ULL;
    int rank = square / 8;
    int file = square % 8;
    for (int r = rank + 1, f = file + 1; r <= 7 && f <= 7; ++r, ++f) {
        int target = r * 8 + f;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    for (int r = rank + 1, f = file - 1; r <= 7 && f >= 0; ++r, --f) {
        int target = r * 8 + f;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    for (int r = rank - 1, f = file + 1; r >= 0 && f <= 7; --r, ++f) {
        int target = r * 8 + f;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    for (int r = rank - 1, f = file - 1; r >= 0 && f >= 0; --r, --f) {
        int target = r * 8 + f;
        attacks |= bit(target);
        if (occ & bit(target)) break;
    }
    return attacks;
}

void initAttackTables() {
    for (int sq = 0; sq < 64; ++sq) {
        int rank = sq / 8;
        int file = sq % 8;
        uint64_t knightMask = 0ULL;
        static const int knightOffsets[8][2] = {
            {1, 2}, {2, 1}, {2, -1}, {1, -2},
            {-1,-2}, {-2,-1}, {-2, 1}, {-1, 2}
        };
        for (const auto &off : knightOffsets) {
            int r = rank + off[1];
            int f = file + off[0];
            if (r >= 0 && r < 8 && f >= 0 && f < 8) {
                knightMask |= bit(r * 8 + f);
            }
        }
        knightAttacks[sq] = knightMask;

        uint64_t kingMask = 0ULL;
        for (int dr = -1; dr <= 1; ++dr) {
            for (int df = -1; df <= 1; ++df) {
                if (dr == 0 && df == 0) continue;
                int r = rank + dr;
                int f = file + df;
                if (r >= 0 && r < 8 && f >= 0 && f < 8) {
                    kingMask |= bit(r * 8 + f);
                }
            }
        }
        kingAttacks[sq] = kingMask;
    }
}

class Board {

    int halfmoveClock;
    int fullmoveNumber;

    array<uint64_t, 2> occ;
    uint64_t occAll;
    array<int, 64> mailbox;
    array<int, 2> kingSq;

private:

Board::Board() { clear(); }

void Board::clear() {
    for (auto &colorPieces : pieces) {
        colorPieces.fill(0ULL);
    }
    occ.fill(0ULL);
    occAll = 0ULL;
    mailbox.fill(0);
    kingSq[WHITE] = kingSq[BLACK] = -1;
    whiteToMove = true;
    castlingRights = 0;
    enPassant = -1;
    halfmoveClock = 0;
    fullmoveNumber = 1;
}

void Board::updateOccupancies() {
    for (int color = 0; color < 2; ++color) {
        occ[color] = 0ULL;
        for (int type = 0; type < PIECE_TYPES; ++type) {
            occ[color] |= pieces[color][type];
        }
    }
    occAll = occ[WHITE] | occ[BLACK];
}

bool Board::loadFEN(const string &fen) {
    clear();
    stringstream ss(fen);
    string boardPart, active, castling, ep, halfmoveStr, fullmoveStr;
    if (!(ss >> boardPart >> active >> castling >> ep >> halfmoveStr >> fullmoveStr)) {
        return false;
    }
    int rank = 7;
    int file = 0;
    for (char ch : boardPart) {
        if (ch == '/') {
            --rank;
            file = 0;
            continue;
        }
        if (isdigit(static_cast<unsigned char>(ch))) {
            file += ch - '0';
            continue;
        }
        int color = isupper(static_cast<unsigned char>(ch)) ? WHITE : BLACK;
        char upper = toupper(static_cast<unsigned char>(ch));
        int type = -1;
        switch (upper) {
            case 'P': type = PAWN; break;
            case 'N': type = KNIGHT; break;
            case 'B': type = BISHOP; break;
            case 'R': type = ROOK; break;
            case 'Q': type = QUEEN; break;
            case 'K': type = KING; break;
            default: return false;
        }
        if (rank < 0 || file >= 8) return false;
        int sq = rank * 8 + file;
        pieces[color][type] |= bit(sq);
        mailbox[sq] = encodePiece(color, type);
        if (type == KING) kingSq[color] = sq;
        ++file;
    }
    whiteToMove = (active == "w");
    castlingRights = 0;
    if (castling.find('K') != string::npos) castlingRights |= 1;
    if (castling.find('Q') != string::npos) castlingRights |= 2;
    if (castling.find('k') != string::npos) castlingRights |= 4;
    if (castling.find('q') != string::npos) castlingRights |= 8;
    enPassant = -1;
    if (ep != "-") {
        if (ep.size() == 2) {
            int fileIdx = ep[0] - 'a';
            int rankIdx = ep[1] - '1';
            if (fileIdx >= 0 && fileIdx < 8 && rankIdx >= 0 && rankIdx < 8) {
                enPassant = rankIdx * 8 + fileIdx;
            }
        }
    }
    try {
        halfmoveClock = stoi(halfmoveStr);
    } catch (...) {
        halfmoveClock = 0;
    }
    try {
        int fm = stoi(fullmoveStr);
        fullmoveNumber = fm > 0 ? fm : 1;
    } catch (...) {
        fullmoveNumber = 1;
    }
    updateOccupancies();
    return true;
}

vector<Move> Board::generateMoves(bool capturesOnly) const {
    vector<Move> moves;
    moves.reserve(capturesOnly ? 64 : 196);
    int side = whiteToMove ? WHITE : BLACK;
    int opp = side ^ 1;
    static const array<int, 4> promotionPieces = {QUEEN, ROOK, BISHOP, KNIGHT};

    auto addMove = [&](int from, int to, int piece, int capturedType,
                       bool capture, bool doublePush, bool enPassantFlag,
                       bool castling, int promotion) {
        Move mv;
        mv.from = static_cast<uint8_t>(from);
        mv.to = static_cast<uint8_t>(to);
        mv.piece = static_cast<uint8_t>(piece);
        mv.captured = capturedType >= 0 ? static_cast<uint8_t>(capturedType) : NO_PIECE;
        mv.flags = 0;
        if (capture) mv.flags |= FLAG_CAPTURE;
        if (doublePush) mv.flags |= FLAG_DOUBLE_PAWN;
        if (enPassantFlag) mv.flags |= FLAG_EN_PASSANT;
        if (castling) mv.flags |= FLAG_CASTLING;
        if (promotion >= 0) {
            mv.flags |= FLAG_PROMOTION;
            mv.promotion = static_cast<uint8_t>(promotion);
        } else {
            mv.promotion = NO_PIECE;
        }
        if (capturesOnly && !(mv.flags & FLAG_CAPTURE) && !(mv.flags & FLAG_PROMOTION)) return;
        moves.push_back(mv);
    };

    if (side == WHITE) {
        uint64_t pawns = pieces[WHITE][PAWN];
        while (pawns) {
            int from = popLSB(pawns);
            int rank = from / 8;
            int file = from & 7;
            int forward = from + 8;
            if (!capturesOnly && forward < 64 && mailbox[forward] == 0) {
                if (rank == 6) {
                    for (int promo : promotionPieces) {
                        addMove(from, forward, PAWN, -1, false, false, false, false, promo);
                    }
                } else {
                    addMove(from, forward, PAWN, -1, false, false, false, false, -1);
                    if (rank == 1) {
                        int dbl = from + 16;
                        if (dbl < 64 && mailbox[dbl] == 0) {
                            addMove(from, dbl, PAWN, -1, false, true, false, false, -1);
                        }
                    }
                }
            }
            auto handleCapture = [&](int to, bool left) {
                if (to < 0 || to >= 64) return;
                int targetCode = mailbox[to];
                if (targetCode != 0 && decodePieceColor(targetCode) == opp) {
                    int capType = decodePieceType(targetCode);
                    if (rank == 6) {
                        for (int promo : promotionPieces) {
                            addMove(from, to, PAWN, capType, true, false, false, false, promo);
                        }
                    } else {
                        addMove(from, to, PAWN, capType, true, false, false, false, -1);
                    }
                } else if (enPassant == to) {
                    addMove(from, to, PAWN, PAWN, true, false, true, false, -1);
                }
            };
            if (file > 0) handleCapture(from + 7, true);
            if (file < 7) handleCapture(from + 9, false);
        }
    } else {
        uint64_t pawns = pieces[BLACK][PAWN];
        while (pawns) {
            int from = popLSB(pawns);
            int rank = from / 8;
            int file = from & 7;
            int forward = from - 8;
            if (!capturesOnly && forward >= 0 && mailbox[forward] == 0) {
                if (rank == 1) {
                    for (int promo : promotionPieces) {
                        addMove(from, forward, PAWN, -1, false, false, false, false, promo);
                    }
                } else {
                    addMove(from, forward, PAWN, -1, false, false, false, false, -1);
                    if (rank == 6) {
                        int dbl = from - 16;
                        if (dbl >= 0 && mailbox[dbl] == 0) {
                            addMove(from, dbl, PAWN, -1, false, true, false, false, -1);
                        }
                    }
                }
            }
            auto handleCapture = [&](int to) {
                if (to < 0 || to >= 64) return;
                int targetCode = mailbox[to];
                if (targetCode != 0 && decodePieceColor(targetCode) == opp) {
                    int capType = decodePieceType(targetCode);
                    if (rank == 1) {
                        for (int promo : promotionPieces) {
                            addMove(from, to, PAWN, capType, true, false, false, false, promo);
                        }
                    } else {
                        addMove(from, to, PAWN, capType, true, false, false, false, -1);
                    }
                } else if (enPassant == to) {
                    addMove(from, to, PAWN, PAWN, true, false, true, false, -1);
                }
            };
            if (file > 0) handleCapture(from - 9);
            if (file < 7) handleCapture(from - 7);
        }
    }

    auto addLeaperMoves = [&](uint64_t piecesBB, const array<uint64_t, 64> &attacksTable, int pieceType) {
        uint64_t bb = piecesBB;
        while (bb) {
            int from = popLSB(bb);
            uint64_t targets = attacksTable[from] & ~occ[side];
            while (targets) {
                int to = popLSB(targets);
                int targetCode = mailbox[to];
                if (targetCode != 0 && decodePieceColor(targetCode) == side) continue;
                int capturedType = (targetCode != 0 && decodePieceColor(targetCode) == opp)
                                       ? decodePieceType(targetCode)
                                       : -1;
                bool capture = capturedType >= 0;
                addMove(from, to, pieceType, capturedType, capture, false, false, false, -1);
            }
        }
    };

    addLeaperMoves(pieces[side][KNIGHT], knightAttacks, KNIGHT);
    addLeaperMoves(pieces[side][KING] & bit(kingSq[side]), kingAttacks, KING);

    auto addSliderMoves = [&](uint64_t piecesBB, auto attackFunc, int pieceType) {
        uint64_t bb = piecesBB;
        while (bb) {
            int from = popLSB(bb);
            uint64_t targets = attackFunc(from, occAll) & ~occ[side];
            while (targets) {
                int to = popLSB(targets);
                int targetCode = mailbox[to];
                if (targetCode != 0 && decodePieceColor(targetCode) == side) continue;
                int capturedType = (targetCode != 0 && decodePieceColor(targetCode) == opp)
                                       ? decodePieceType(targetCode)
                                       : -1;
                bool capture = capturedType >= 0;
                addMove(from, to, pieceType, capturedType, capture, false, false, false, -1);
            }
        }
    };

    addSliderMoves(pieces[side][BISHOP], bishopAttacks, BISHOP);
    addSliderMoves(pieces[side][ROOK], rookAttacks, ROOK);
    uint64_t queens = pieces[side][QUEEN];
    while (queens) {
        int from = popLSB(queens);
        uint64_t targets = (rookAttacks(from, occAll) | bishopAttacks(from, occAll)) & ~occ[side];
        while (targets) {
            int to = popLSB(targets);
            int targetCode = mailbox[to];
            if (targetCode != 0 && decodePieceColor(targetCode) == side) continue;
            int capturedType = (targetCode != 0 && decodePieceColor(targetCode) == opp)
                                   ? decodePieceType(targetCode)
                                   : -1;
            bool capture = capturedType >= 0;
            addMove(from, to, QUEEN, capturedType, capture, false, false, false, -1);
        }
    }

    if (!capturesOnly) {
        int kingSquare = kingSq[side];
        if (kingSquare >= 0 && !inCheck(side)) {
            if (side == WHITE && kingSquare == 4) {
                if ((castlingRights & 1) && mailbox[5] == 0 && mailbox[6] == 0 &&
                    mailbox[7] == encodePiece(WHITE, ROOK) &&
                    !isSquareAttacked(5, opp) && !isSquareAttacked(6, opp)) {
                    addMove(4, 6, KING, -1, false, false, false, true, -1);
                }
                if ((castlingRights & 2) && mailbox[3] == 0 && mailbox[2] == 0 && mailbox[1] == 0 &&
                    mailbox[0] == encodePiece(WHITE, ROOK) &&
                    !isSquareAttacked(3, opp) && !isSquareAttacked(2, opp)) {
                    addMove(4, 2, KING, -1, false, false, false, true, -1);
                }
            }
            if (side == BLACK && kingSquare == 60) {
                if ((castlingRights & 4) && mailbox[61] == 0 && mailbox[62] == 0 &&
                    mailbox[63] == encodePiece(BLACK, ROOK) &&
                    !isSquareAttacked(61, WHITE) && !isSquareAttacked(62, WHITE)) {
                    addMove(60, 62, KING, -1, false, false, false, true, -1);
                }
                if ((castlingRights & 8) && mailbox[59] == 0 && mailbox[58] == 0 && mailbox[57] == 0 &&
                    mailbox[56] == encodePiece(BLACK, ROOK) &&
                    !isSquareAttacked(59, WHITE) && !isSquareAttacked(58, WHITE)) {
                    addMove(60, 58, KING, -1, false, false, false, true, -1);
                }
            }
        }
    }

    return moves;
}

bool Board::isSquareAttacked(int square, int attackerColor) const {
    if (square < 0 || square >= 64) return false;
    if (attackerColor == WHITE) {
        uint64_t pawns = pieces[WHITE][PAWN];
        uint64_t attacks = ((pawns & ~FILE_H) << 9) | ((pawns & ~FILE_A) << 7);
        if (attacks & bit(square)) return true;
    } else {
        uint64_t pawns = pieces[BLACK][PAWN];
        uint64_t attacks = ((pawns & ~FILE_A) >> 9) | ((pawns & ~FILE_H) >> 7);
        if (attacks & bit(square)) return true;
    }
    if (knightAttacks[square] & pieces[attackerColor][KNIGHT]) return true;
    if (kingAttacks[square] & pieces[attackerColor][KING]) return true;
    uint64_t bishopsQueens = pieces[attackerColor][BISHOP] | pieces[attackerColor][QUEEN];
    if (bishopAttacks(square, occAll) & bishopsQueens) return true;
    uint64_t rooksQueens = pieces[attackerColor][ROOK] | pieces[attackerColor][QUEEN];
    if (rookAttacks(square, occAll) & rooksQueens) return true;
    return false;
}

bool Board::inCheck(int color) const {
    int kingSquare = kingSq[color];
    if (kingSquare < 0) return false;
    return isSquareAttacked(kingSquare, color ^ 1);
}

bool Board::makeMove(const Move &mv) {
    if (!mv.isValid()) return false;
    int side = whiteToMove ? WHITE : BLACK;
    int opp = side ^ 1;
    int from = mv.from;
    int to = mv.to;
    int movingType = mv.piece;
    if (movingType >= PIECE_TYPES) return false;
    int fromCode = mailbox[from];
    if (fromCode == 0 || decodePieceColor(fromCode) != side || decodePieceType(fromCode) != movingType) {
        return false;
    }

    mailbox[from] = 0;
    pieces[side][movingType] &= ~bit(from);

    auto clearCastleBySquare = [&](int square) {
        switch (square) {
            case 0: castlingRights &= ~2; break;
            case 7: castlingRights &= ~1; break;
            case 56: castlingRights &= ~8; break;
            case 63: castlingRights &= ~4; break;
            default: break;
        }
    };

    if (movingType == ROOK) {
        clearCastleBySquare(from);
    }
    if (movingType == KING) {
        if (side == WHITE) castlingRights &= ~(1 | 2);
        else castlingRights &= ~(4 | 8);
    }

    if (mv.isCapture()) {
        if (mv.isEnPassant()) {
            int capSq = side == WHITE ? to - 8 : to + 8;
            pieces[opp][PAWN] &= ~bit(capSq);
            mailbox[capSq] = 0;
        } else {
            int targetCode = mailbox[to];
            if (targetCode == 0 || decodePieceColor(targetCode) != opp) return false;
            int capturedType = decodePieceType(targetCode);
            pieces[opp][capturedType] &= ~bit(to);
            mailbox[to] = 0;
            if (capturedType == KING) kingSq[opp] = -1;
            clearCastleBySquare(to);
        }
    } else if (!mv.isCastle() && mailbox[to] != 0) {
        return false;
    }

    bool handledCastle = false;
    if (mv.isCastle()) {
        handledCastle = true;
        if (side == WHITE) {
            if (to == 6) {
                pieces[WHITE][KING] |= bit(6);
                mailbox[6] = encodePiece(WHITE, KING);
                kingSq[WHITE] = 6;
                pieces[WHITE][ROOK] &= ~bit(7);
                pieces[WHITE][ROOK] |= bit(5);
                mailbox[7] = 0;
                mailbox[5] = encodePiece(WHITE, ROOK);
            } else if (to == 2) {
                pieces[WHITE][KING] |= bit(2);
                mailbox[2] = encodePiece(WHITE, KING);
                kingSq[WHITE] = 2;
                pieces[WHITE][ROOK] &= ~bit(0);
                pieces[WHITE][ROOK] |= bit(3);
                mailbox[0] = 0;
                mailbox[3] = encodePiece(WHITE, ROOK);
            } else {
                return false;
            }
        } else {
            if (to == 62) {
                pieces[BLACK][KING] |= bit(62);
                mailbox[62] = encodePiece(BLACK, KING);
                kingSq[BLACK] = 62;
                pieces[BLACK][ROOK] &= ~bit(63);
                pieces[BLACK][ROOK] |= bit(61);
                mailbox[63] = 0;
                mailbox[61] = encodePiece(BLACK, ROOK);
            } else if (to == 58) {
                pieces[BLACK][KING] |= bit(58);
                mailbox[58] = encodePiece(BLACK, KING);
                kingSq[BLACK] = 58;
                pieces[BLACK][ROOK] &= ~bit(56);
                pieces[BLACK][ROOK] |= bit(59);
                mailbox[56] = 0;
                mailbox[59] = encodePiece(BLACK, ROOK);
            } else {
                return false;
            }
        }
    }

    if (!handledCastle) {
        int placeType = mv.isPromotion() ? mv.promotion : movingType;
        pieces[side][placeType] |= bit(to);
        mailbox[to] = encodePiece(side, placeType);
        if (placeType == KING) kingSq[side] = to;
    }

    if (mv.isDoublePush()) {
        enPassant = side == WHITE ? to - 8 : to + 8;
    } else {
        enPassant = -1;
    }

    if (movingType == PAWN || mv.isCapture()) halfmoveClock = 0;
    else ++halfmoveClock;

    updateOccupancies();

    if (isSquareAttacked(kingSq[side], opp)) {
        return false;
    }

    whiteToMove = !whiteToMove;
    if (side == BLACK) ++fullmoveNumber;
    return true;
}

struct ScoredMove {
    Move move;
    int score;
};

class SearchEngine {
public:
    Move findBestMove(const Board &board, double timeLimitSeconds);

private:
    double timeLimit = 0.0;
    chrono::steady_clock::time_point startTime;
    bool stop = false;
    long long nodes = 0;
    Move pvMove;
    Move rootBestMove;
    Move bestCompletedMove;
    int history[2][64][64]{};
    Move killerMoves[MAX_PLY][2]{};
    int lastScore = 0;

    bool timeExceeded();
    int negamax(const Board &board, int depth, int ply, int alpha, int beta);
    int quiescence(const Board &board, int alpha, int beta, int ply);
    int evaluate(const Board &board) const;
    int scoreMove(const Move &move, int ply, int side) const;
    void updateHistory(int side, const Move &move, int depth);
    void updateKillers(const Move &move, int ply);
};

bool SearchEngine::timeExceeded() {
    if (stop) return true;
    double elapsed = chrono::duration<double>(chrono::steady_clock::now() - startTime).count();
    if (elapsed >= timeLimit) {
        stop = true;
        return true;
    }
    return false;
}

Move SearchEngine::findBestMove(const Board &board, double timeLimitSeconds) {
    timeLimit = max(0.01, timeLimitSeconds * 0.95);
    startTime = chrono::steady_clock::now();
    stop = false;
    nodes = 0;
    pvMove = Move();
    rootBestMove = Move();
    bestCompletedMove = Move();
    lastScore = 0;

    for (int c = 0; c < 2; ++c)
        for (int f = 0; f < 64; ++f)
            for (int t = 0; t < 64; ++t)
                history[c][f][t] = 0;
    for (int i = 0; i < MAX_PLY; ++i) {
        killerMoves[i][0] = Move();
        killerMoves[i][1] = Move();
    }

    Move previousBest;

    for (int depth = 1; depth <= MAX_DEPTH; ++depth) {
        pvMove = previousBest;
        rootBestMove = Move();
        int score = negamax(board, depth, 0, -INF_SCORE, INF_SCORE);
        if (stop) break;
        if (rootBestMove.isValid()) {
            bestCompletedMove = rootBestMove;
            previousBest = rootBestMove;
            lastScore = score;
        }
        if (abs(score) >= MATE_VALUE - 100) break;
    }

    if (bestCompletedMove.isValid()) return bestCompletedMove;

    vector<Move> fallback = board.generateMoves(false);
    for (const Move &mv : fallback) {
        Board child = board;
        if (child.makeMove(mv)) return mv;
    }
    return Move();
}

int SearchEngine::scoreMove(const Move &move, int ply, int side) const {
    if (ply == 0 && pvMove.isValid() && move == pvMove) return 3'000'000;
    if (move.isCapture()) {
        int capturedValue = (move.captured < PIECE_TYPES) ? PIECE_VALUE[move.captured] : 0;
        int moverValue = PIECE_VALUE[move.piece];
        return 2'000'000 + capturedValue * 16 - moverValue;
    }
    if (move.isPromotion()) {
        int promoValue = (move.promotion < PIECE_TYPES) ? PIECE_VALUE[move.promotion] : 0;
        return 1'800'000 + promoValue;
    }
    int idx = min(ply, MAX_PLY - 1);
    if (killerMoves[idx][0].isValid() && move == killerMoves[idx][0]) return 1'600'000;
    if (killerMoves[idx][1].isValid() && move == killerMoves[idx][1]) return 1'500'000;
    return history[side][move.from][move.to];
}

void SearchEngine::updateHistory(int side, const Move &move, int depth) {
    int &entry = history[side][move.from][move.to];
    entry = min(HISTORY_LIMIT, entry + depth * depth);
}

void SearchEngine::updateKillers(const Move &move, int ply) {
    int idx = min(ply, MAX_PLY - 1);
    if (!killerMoves[idx][0].isValid() || !(killerMoves[idx][0] == move)) {
        killerMoves[idx][1] = killerMoves[idx][0];
        killerMoves[idx][0] = move;
    }
}

int SearchEngine::evaluate(const Board &board) const {
    int eval = 0;
    int nonPawnMaterial[2] = {0, 0};
    for (int color = 0; color < 2; ++color) {
        for (int type = KNIGHT; type <= QUEEN; ++type) {
            nonPawnMaterial[color] += static_cast<int>(std::popcount(board.pieces[color][type])) * PIECE_VALUE[type];
        }
    }
    bool endgame[2] = {
        nonPawnMaterial[WHITE] <= 1300,
        nonPawnMaterial[BLACK] <= 1300
    };

    for (int color = 0; color < 2; ++color) {
        int sign = (color == WHITE) ? 1 : -1;
        for (int type = PAWN; type <= KING; ++type) {
            uint64_t bb = board.pieces[color][type];
            uint64_t temp = bb;
            while (temp) {
                int sq = popLSB(temp);
                int mirrored = (color == WHITE) ? sq : mirrorSquare(sq);
                int pst = 0;
                int base = 0;
                switch (type) {
                    case PAWN:
                        base = PIECE_VALUE[PAWN];
                        pst = PAWN_TABLE[mirrored];
                        break;
                    case KNIGHT:
                        base = PIECE_VALUE[KNIGHT];
                        pst = KNIGHT_TABLE[mirrored];
                        break;
                    case BISHOP:
                        base = PIECE_VALUE[BISHOP];
                        pst = BISHOP_TABLE[mirrored];
                        break;
                    case ROOK:
                        base = PIECE_VALUE[ROOK];
                        pst = ROOK_TABLE[mirrored];
                        break;
                    case QUEEN:
                        base = PIECE_VALUE[QUEEN];
                        pst = QUEEN_TABLE[mirrored];
                        break;
                    case KING:
                        pst = (endgame[color] ? KING_END_TABLE[mirrored] : KING_MID_TABLE[mirrored]);
                        break;
                }
                eval += sign * (base + pst);
            }
        }
        if (std::popcount(board.pieces[color][BISHOP]) >= 2) eval += sign * 30;
    }
    eval += board.whiteToMove ? 10 : -10;
    return eval;
}

int SearchEngine::quiescence(const Board &board, int alpha, int beta, int ply) {
    if (stop) return 0;
    if ((nodes & 0x3FF) == 0 && timeExceeded()) return 0;
    ++nodes;

    int side = board.whiteToMove ? WHITE : BLACK;
    bool inChk = board.inCheck(side);
    int standPat = evaluate(board);

    if (!inChk) {
        if (standPat >= beta) return standPat;
        if (standPat > alpha) alpha = standPat;
    }

    vector<Move> moves = board.generateMoves(inChk ? false : true);
    vector<ScoredMove> scored;
    scored.reserve(moves.size());
    for (const Move &mv : moves) {
        if (!inChk && !mv.isCapture() && !mv.isPromotion()) continue;
        int sc = scoreMove(mv, ply, side);
        scored.push_back({mv, sc});
    }
    sort(scored.begin(), scored.end(), [](const ScoredMove &a, const ScoredMove &b) {
        return a.score > b.score;
    });

    bool legal = false;
    for (const auto &entry : scored) {
        Board child = board;
        if (!child.makeMove(entry.move)) continue;
        legal = true;
        int score = -quiescence(child, -beta, -alpha, ply + 1);
        if (stop) return 0;
        if (score >= beta) return score;
        if (score > alpha) alpha = score;
    }

    if (inChk && !legal) return -MATE_VALUE + ply;
    return alpha;
}

int SearchEngine::negamax(const Board &board, int depth, int ply, int alpha, int beta) {
    if (stop) return 0;
    if ((nodes & 0x3FF) == 0 && timeExceeded()) return 0;
    ++nodes;

    int side = board.whiteToMove ? WHITE : BLACK;

    if (depth <= 0) {
        if (board.inCheck(side)) depth = 1;
        else return quiescence(board, alpha, beta, ply);
    }

    vector<Move> moves = board.generateMoves(false);
    vector<ScoredMove> scored;
    scored.reserve(moves.size());
    for (const Move &mv : moves) {
        int sc = scoreMove(mv, ply, side);
        scored.push_back({mv, sc});
    }
    sort(scored.begin(), scored.end(), [](const ScoredMove &a, const ScoredMove &b) {
        return a.score > b.score;
    });

    bool legal = false;
    int bestScore = -INF_SCORE;

    for (const auto &entry : scored) {
        Board child = board;
        if (!child.makeMove(entry.move)) continue;
        legal = true;
        int score = -negamax(child, depth - 1, ply + 1, -beta, -alpha);
        if (stop) return 0;
        if (score > bestScore) bestScore = score;
        if (score > alpha) {
            alpha = score;
            if (ply == 0) rootBestMove = entry.move;
        }
        if (alpha >= beta) {
            if (!entry.move.isCapture() && !entry.move.isPromotion()) {
                updateKillers(entry.move, ply);
                updateHistory(side, entry.move, depth);
            }
            return alpha;
        }
    }

    if (!legal) {
        if (board.inCheck(side)) return -MATE_VALUE + ply;
        return 0;
    }
    return alpha;
}

string formatMove(const Board &board, const Move &move) {
    if (!move.isValid()) return "resign";
    if (move.isCastle()) return move.to > move.from ? "O-O" : "O-O-O";
    string result;
    result += pieceLetterForMove(move.piece);
    result += squareToString(move.from);
    if (move.isCapture()) result += 'x';
    result += squareToString(move.to);
    if (move.isPromotion()) {
        result += '/';
        result += pieceLetterForMove(move.promotion);
    }
    if (move.isEnPassant()) result += "ep";

    Board after = board;
    if (after.makeMove(move)) {
        int opponent = after.whiteToMove ? WHITE : BLACK;
        if (after.inCheck(opponent)) {
            bool hasReply = false;
            vector<Move> replies = after.generateMoves(false);
            for (const Move &reply : replies) {
                Board tmp = after;
                if (tmp.makeMove(reply)) {
                    hasReply = true;
                    break;
                }
            }
            result += hasReply ? '+' : '#';
        }
    }
    return result;
}
