Here's a complete C++20 implementation that adapts its strategy to the problem size, from brute force for tiny instances to space-filling-curve heuristics with limited local search for massive ones:

```cpp
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct Point {
    float x;
    float y;
};

struct DistanceCalculator {
    const std::vector<Point>& pts;
    double operator()(int a, int b) const noexcept {
        double dx = static_cast<double>(pts[a].x) - static_cast<double>(pts[b].x);
        double dy = static_cast<double>(pts[a].y) - static_cast<double>(pts[b].y);
        return std::sqrt(dx * dx + dy * dy);
    }
};

inline uint64_t part1by1(uint32_t n) noexcept {
    uint64_t x = n;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2)) & 0x3333333333333333ULL;
    x = (x | (x << 1)) & 0x5555555555555555ULL;
    return x;
}

inline uint64_t mortonFromInts(uint32_t x, uint32_t y) noexcept {
    return (part1by1(y) << 1) | part1by1(x);
}

uint64_t computeMorton(double x, double y,
                       double minX, double minY,
                       double invRangeX, double invRangeY) noexcept {
    constexpr double scale = 4294967295.0;
    double nx = invRangeX > 0.0 ? (x - minX) * invRangeX : 0.5;
    double ny = invRangeY > 0.0 ? (y - minY) * invRangeY : 0.5;
    if (nx < 0.0) nx = 0.0;
    if (nx > 1.0) nx = 1.0;
    if (ny < 0.0) ny = 0.0;
    if (ny > 1.0) ny = 1.0;
    uint64_t rawX = static_cast<uint64_t>(nx * scale);
    uint64_t rawY = static_cast<uint64_t>(ny * scale);
    if (rawX > 0xFFFFFFFFULL) rawX = 0xFFFFFFFFULL;
    if (rawY > 0xFFFFFFFFULL) rawY = 0xFFFFFFFFULL;
    return mortonFromInts(static_cast<uint32_t>(rawX), static_cast<uint32_t>(rawY));
}

void radixSortIndices64(std::vector<int>& order, const std::vector<uint64_t>& keys) {
    const size_t n = order.size();
    if (n <= 1) {
        return;
    }
    std::vector<int> buffer(n);
    constexpr int bitsPerPass = 8;
    constexpr int buckets = 1 << bitsPerPass;
    constexpr int passes = (64 + bitsPerPass - 1) / bitsPerPass;
    size_t count[buckets];
    for (int pass = 0; pass < passes; ++pass) {
        std::fill(count, count + buckets, static_cast<size_t>(0));
        const int shift = pass * bitsPerPass;
        for (size_t i = 0; i < n; ++i) {
            unsigned bucket = static_cast<unsigned>((keys[order[i]] >> shift) & (buckets - 1));
            ++count[bucket];
        }
        size_t sum = 0;
        for (int b = 0; b < buckets; ++b) {
            size_t tmp = count[b];
            count[b] = sum;
            sum += tmp;
        }
        for (size_t i = 0; i < n; ++i) {
            unsigned bucket = static_cast<unsigned>((keys[order[i]] >> shift) & (buckets - 1));
            buffer[count[bucket]++] = order[i];
        }
        order.swap(buffer);
    }
}

std::vector<int> spaceFillingOrder(const std::vector<Point>& pts,
                                   double minX, double maxX,
                                   double minY, double maxY) {
    const size_t n = pts.size();
    std::vector<int> order;
    if (n == 0) {
        return order;
    }
    std::vector<uint64_t> keys(n);
    const double invRangeX = (maxX > minX) ? 1.0 / (maxX - minX) : 0.0;
    const double invRangeY = (maxY > minY) ? 1.0 / (maxY - minY) : 0.0;

    for (size_t i = 0; i < n; ++i) {
        keys[i] = computeMorton(static_cast<double>(pts[i].x),
                                static_cast<double>(pts[i].y),
                                minX, minY, invRangeX, invRangeY);
    }

    order.resize(n);
    std::iota(order.begin(), order.end(), 0);

    constexpr size_t radixThreshold = 1'000'000;
    if (n >= radixThreshold) {
        radixSortIndices64(order, keys);
    } else {
        std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
            const uint64_t kl = keys[lhs];
            const uint64_t kr = keys[rhs];
            if (kl == kr) {
                return lhs < rhs;
            }
            return kl < kr;
        });
    }

    auto zeroIt = std::find(order.begin(), order.end(), 0);
    if (zeroIt != order.begin() && zeroIt != order.end()) {
        std::rotate(order.begin(), zeroIt, order.end());
    }
    return order;
}

std::vector<int> solveTinyBrute(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n == 0) {
        return {};
    }
    if (n == 1) {
        return std::vector<int>{0};
    }
    std::vector<int> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1);
    DistanceCalculator dist{pts};
    double best = std::numeric_limits<double>::infinity();
    std::vector<int> bestSeq = perm;
    do {
        double total = 0.0;
        int prev = 0;
        for (int idx : perm) {
            total += dist(prev, idx);
            prev = idx;
            if (total >= best) {
                break;
            }
        }
        if (total < best) {
            best = total;
            bestSeq = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));
    std::vector<int> order;
    order.reserve(n);
    order.push_back(0);
    order.insert(order.end(), bestSeq.begin(), bestSeq.end());
    return order;
}

std::vector<int> nearestNeighbor(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> order;
    order.reserve(n);
    if (n == 0) {
        return order;
    }
    order.push_back(0);
    if (n == 1) {
        return order;
    }
    std::vector<uint8_t> used(n, 0);
    used[0] = 1;
    DistanceCalculator dist{pts};
    for (int step = 1; step < n; ++step) {
        int current = order.back();
        int bestIdx = -1;
        double bestDist = std::numeric_limits<double>::infinity();
        for (int candidate = 0; candidate < n; ++candidate) {
            if (used[candidate]) {
                continue;
            }
            double d = dist(current, candidate);
            if (d < bestDist) {
                bestDist = d;
                bestIdx = candidate;
            }
        }
        if (bestIdx == -1) {
            for (int candidate = 0; candidate < n; ++candidate) {
                if (!used[candidate]) {
                    bestIdx = candidate;
                    break;
                }
            }
        }
        used[bestIdx] = 1;
        order.push_back(bestIdx);
    }
    return order;
}

void boundedTwoOpt(std::vector<int>& order,
                   const std::vector<Point>& pts,
                   int passes,
                   int window,
                   int stride) {
    const int n = static_cast<int>(order.size());
    if (n < 4 || passes <= 0) {
        return;
    }
    if (window < 2) {
        window = 2;
    }
    if (stride < 1) {
        stride = 1;
    }
    DistanceCalculator dist{pts};
    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        int i = 1;
        while (i < n - 2) {
            const int maxJ = std::min(n - 1, i + window);
            bool flipped = false;
            for (int j = i + 1; j < maxJ; ++j) {
                if (j + 1 >= n) {
                    break;
                }
                const int a = order[i - 1];
                const int b = order[i];
                const int c = order[j];
                const int d = order[j + 1];
                const double before = dist(a, b) + dist(c, d);
                const double after = dist(a, c) + dist(b, d);
                if (after + 1e-9 < before) {
                    std::reverse(order.begin() + i, order.begin() + j + 1);
                    improved = true;
                    flipped = true;
                    break;
                }
            }
            if (flipped && stride > 1) {
                i = std::max(1, i - stride);
            } else {
                i += stride;
            }
        }
        if (!improved) {
            break;
        }
    }
}

void refineRoute(std::vector<int>& order, const std::vector<Point>& pts) {
    const size_t n = order.size();
    if (n < 4) {
        return;
    }
    if (n <= 400) {
        boundedTwoOpt(order, pts, 40, 128, 1);
    } else if (n <= 2000) {
        boundedTwoOpt(order, pts, 30, 110, 1);
    } else if (n <= 4000) {
        boundedTwoOpt(order, pts, 20, 96, 1);
    } else if (n <= 20000) {
        boundedTwoOpt(order, pts, 12, 96, 1);
    } else if (n <= 200000) {
        boundedTwoOpt(order, pts, 5, 64, 2);
    } else if (n <= 5000000) {
        boundedTwoOpt(order, pts, 2, 32, 8);
    }
}

std::vector<int> buildRoute(const std::vector<Point>& pts,
                            double minX, double maxX,
                            double minY, double maxY) {
    const size_t n = pts.size();
    if (n == 0) {
        return {};
    }
    if (n == 1) {
        return std::vector<int>{0};
    }
    if (n <= 11) {
        return solveTinyBrute(pts);
    }
    std::vector<int> order;
    if (n <= 4000) {
        order = nearestNeighbor(pts);
        refineRoute(order, pts);
    } else {
        order = spaceFillingOrder(pts, minX, maxX, minY, maxY);
        refineRoute(order, pts);
    }
    if (!order.empty() && order[0] != 0) {
        auto it = std::find(order.begin(), order.end(), 0);
        if (it != order.end()) {
            std::rotate(order.begin(), it, order.end());
        }
    }
    return order;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::size_t n;
    if (!(std::cin >> n)) {
        return 0;
    }
    if (n == 0) {
        return 0;
    }

    std::vector<Point> pts(n);
    double minX = std::numeric_limits<double>::infinity();
    double maxX = -std::numeric_limits<double>::infinity();
    double minY = std::numeric_limits<double>::infinity();
    double maxY = -std::numeric_limits<double>::infinity();

    for (std::size_t i = 0; i < n; ++i) {
        double x = 0.0;
        double y = 0.0;
        if (!(std::cin >> x >> y)) {
            x = 0.0;
            y = 0.0;
        }
        pts[i].x = static_cast<float>(x);
        pts[i].y = static_cast<float>(y);
        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
    }

    std::vector<int> order = buildRoute(pts, minX, maxX, minY, maxY);
    if (order.empty()) {
        order.resize(n);
        std::iota(order.begin(), order.end(), 0);
    }

    for (std::size_t i = 0; i < order.size(); ++i) {
        if (i) {
            std::cout << ' ';
        }
        std::cout << order[i];
    }
    std::cout << '\n';
    return 0;
}
```

This single executable automatically switches among:

- brute force enumeration for N ≤ 11
- nearest-neighbor + intensive 2-opt for small/medium inputs
- Morton-order (space-filling curve) layouts, radix-sorted for huge N, plus localized 2-opt smoothing where it’s still affordable

It always outputs a 0-indexed permutation starting at city 0 without explicitly closing the tour.
