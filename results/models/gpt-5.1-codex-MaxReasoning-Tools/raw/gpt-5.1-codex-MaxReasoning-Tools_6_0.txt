Here is a complete C# implementation that adapts automatically from brute force on tiny instances to sophisticated heuristics (multi-start insertion/nearest-neighbor seeds, 2-opt, Or-opt, and perturbation-driven iterated improvement) on extreme cases, while using an orbital-aware delta‑V approximation to weight the TSP edges:

using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

struct Vector3d
{
    public double X;
    public double Y;
    public double Z;

    public Vector3d(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public double LengthSquared => X * X + Y * Y + Z * Z;

    public double Length()
    {
        return Math.Sqrt(LengthSquared);
    }

    public Vector3d Normalize()
    {
        double len = Length();
        if (len < 1e-9) return new Vector3d(0, 0, 0);
        double inv = 1.0 / len;
        return new Vector3d(X * inv, Y * inv, Z * inv);
    }

    public static Vector3d operator +(Vector3d a, Vector3d b) => new Vector3d(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3d operator -(Vector3d a, Vector3d b) => new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3d operator *(Vector3d a, double s) => new Vector3d(a.X * s, a.Y * s, a.Z * s);
    public static Vector3d operator *(double s, Vector3d a) => a * s;
    public static Vector3d operator /(Vector3d a, double s) => new Vector3d(a.X / s, a.Y / s, a.Z / s);

    public static double Dot(Vector3d a, Vector3d b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

    public static Vector3d Cross(Vector3d a, Vector3d b)
    {
        return new Vector3d(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    public static double AngleBetween(Vector3d a, Vector3d b)
    {
        double denom = Math.Sqrt(a.LengthSquared * b.LengthSquared);
        if (denom < 1e-12) return 0.0;
        double cos = Dot(a, b) / denom;
        cos = Clamp(cos, -1.0, 1.0);
        return Math.Acos(cos);
    }

    public static double Clamp(double value, double min, double max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }
}

class OrbitNode
{
    public Vector3d Position { get; }
    public Vector3d Velocity { get; }
    public double Radius { get; }
    public double Speed { get; }
    public Vector3d PlaneNormal { get; }
    public double SemiMajorAxis { get; }
    public double Eccentricity { get; }
    public Vector3d EccentricityVector { get; }
    public double SpecificEnergy { get; }
    public Vector3d UnitPosition { get; }

    private OrbitNode(Vector3d position, Vector3d velocity, double radius, double speed,
                      Vector3d planeNormal, double semiMajorAxis, double eccentricity,
                      Vector3d eccentricityVector, double specificEnergy, Vector3d unitPosition)
    {
        Position = position;
        Velocity = velocity;
        Radius = radius;
        Speed = speed;
        PlaneNormal = planeNormal;
        SemiMajorAxis = semiMajorAxis;
        Eccentricity = eccentricity;
        EccentricityVector = eccentricityVector;
        SpecificEnergy = specificEnergy;
        UnitPosition = unitPosition;
    }

    public static OrbitNode FromState(Vector3d r, Vector3d v, double mu)
    {
        double radius = Math.Max(r.Length(), 1e-6);
        double speed = v.Length();
        Vector3d h = Vector3d.Cross(r, v);
        double hMag = h.Length();
        Vector3d planeNormal = hMag > 1e-9 ? h / hMag : new Vector3d(0, 0, 1);
        double energy = 0.5 * speed * speed - mu / radius;
        double semiMajor = Math.Abs(energy) < 1e-9 ? double.PositiveInfinity : -mu / (2.0 * energy);
        Vector3d eVec = (Vector3d.Cross(v, h) / mu) - (r / radius);
        double ecc = eVec.Length();
        Vector3d unitPos = r / radius;
        return new OrbitNode(r, v, radius, speed, planeNormal, semiMajor, ecc, eVec, energy, unitPos);
    }
}

class OrbitalTspSolver
{
    private readonly OrbitNode[] nodes;
    private readonly OrbitNode startNode;
    private readonly double mu;
    private readonly int n;
    private readonly double[][] costMatrix;
    private readonly double[] startCost;
    private readonly Random random;

    public OrbitalTspSolver(OrbitNode[] nodes, OrbitNode startNode, double mu)
    {
        this.nodes = nodes;
        this.startNode = startNode;
        this.mu = mu;
        this.n = nodes.Length;
        costMatrix = new double[n][];
        for (int i = 0; i < n; ++i)
        {
            costMatrix[i] = new double[n];
        }
        startCost = new double[n];
        random = new Random(1337 + n);
        BuildCosts();
    }

    public int[] Solve()
    {
        if (n == 0) return Array.Empty<int>();
        if (n <= 8) return SolveBruteForce();
        if (n <= 13) return SolveHeldKarp();
        return SolveHeuristically();
    }

    private void BuildCosts()
    {
        for (int i = 0; i < n; ++i)
        {
            startCost[i] = TransferCost(startNode, nodes[i]);
        }

        for (int i = 0; i < n; ++i)
        {
            costMatrix[i][i] = 0.0;
            for (int j = i + 1; j < n; ++j)
            {
                double c = TransferCost(nodes[i], nodes[j]);
                costMatrix[i][j] = c;
                costMatrix[j][i] = c;
            }
        }
    }

    private double TransferCost(OrbitNode from, OrbitNode to)
    {
        double r1 = Math.Max(from.Radius, 1.0);
        double r2 = Math.Max(to.Radius, 1.0);
        double hohmann = HohmannCost(r1, r2);
        double planeAngle = Vector3d.AngleBetween(from.PlaneNormal, to.PlaneNormal);
        double planeSpeed = Math.Max(0.01, Math.Min(from.Speed, to.Speed));
        double planeCost = 2.0 * planeSpeed * Math.Sin(planeAngle * 0.5);

        double eccCost = Math.Abs(from.Eccentricity - to.Eccentricity) * 0.5 * (from.Speed + to.Speed);
        double orientationAngle = Vector3d.AngleBetween(from.UnitPosition, to.UnitPosition);
        double orientationCost = 0.25 * (from.Speed + to.Speed) * orientationAngle;

        double direct = (from.Velocity - to.Velocity).Length();
        double energyDiff = Math.Abs(from.SpecificEnergy - to.SpecificEnergy);
        double energyCost = energyDiff / Math.Max(planeSpeed, 0.1);

        double combined = 0.35 * (hohmann + planeCost) + 0.35 * direct + 0.2 * (eccCost + orientationCost) + 0.1 * energyCost;
        if (!double.IsFinite(combined) || combined < 0)
        {
            combined = direct;
        }
        return combined;
    }

    private double HohmannCost(double r1, double r2)
    {
        if (Math.Abs(r1 - r2) < 1e-6) return 0.0;
        double aTransfer = 0.5 * (r1 + r2);
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double vPeri = Math.Sqrt(Math.Max(0.0, mu * (2.0 / r1 - 1.0 / aTransfer)));
        double vApo = Math.Sqrt(Math.Max(0.0, mu * (2.0 / r2 - 1.0 / aTransfer)));
        return Math.Abs(vPeri - v1) + Math.Abs(v2 - vApo);
    }

    private int[] SolveBruteForce()
    {
        int[] perm = Enumerable.Range(0, n).ToArray();
        int[] best = new int[n];
        double bestCost = double.PositiveInfinity;

        void Permute(int idx)
        {
            if (idx == n)
            {
                double cost = RouteCost(perm);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    Array.Copy(perm, best, n);
                }
                return;
            }

            for (int i = idx; i < n; ++i)
            {
                Swap(perm, idx, i);
                Permute(idx + 1);
                Swap(perm, idx, i);
            }
        }

        Permute(0);
        return best;
    }

    private int[] SolveHeldKarp()
    {
        int size = 1 << n;
        double[][] dp = new double[size][];
        int[][] parent = new int[size][];
        for (int mask = 0; mask < size; ++mask)
        {
            dp[mask] = Enumerable.Repeat(double.PositiveInfinity, n).ToArray();
            parent[mask] = Enumerable.Repeat(-1, n).ToArray();
        }

        for (int j = 0; j < n; ++j)
        {
            int mask = 1 << j;
            dp[mask][j] = startCost[j];
        }

        for (int mask = 1; mask < size; ++mask)
        {
            for (int last = 0; last < n; ++last)
            {
                if ((mask & (1 << last)) == 0) continue;
                double current = dp[mask][last];
                if (!double.IsFinite(current)) continue;
                int remainder = ((1 << n) - 1) ^ mask;
                for (int next = 0; next < n; ++next)
                {
                    if ((remainder & (1 << next)) == 0) continue;
                    int nextMask = mask | (1 << next);
                    double candidate = current + costMatrix[last][next];
                    if (candidate < dp[nextMask][next])
                    {
                        dp[nextMask][next] = candidate;
                        parent[nextMask][next] = last;
                    }
                }
            }
        }

        int fullMask = size - 1;
        double bestCost = double.PositiveInfinity;
        int bestLast = -1;
        for (int last = 0; last < n; ++last)
        {
            if (dp[fullMask][last] < bestCost)
            {
                bestCost = dp[fullMask][last];
                bestLast = last;
            }
        }

        int[] route = new int[n];
        int maskState = fullMask;
        int idx = n - 1;
        int currentNode = bestLast;
        while (idx >= 0 && currentNode != -1)
        {
            route[idx--] = currentNode;
            int prev = parent[maskState][currentNode];
            maskState &= ~(1 << currentNode);
            currentNode = prev;
        }

        if (idx >= 0)
        {
            var remaining = new HashSet<int>(Enumerable.Range(0, n));
            foreach (var val in route)
            {
                remaining.Remove(val);
            }

            foreach (var rem in remaining)
            {
                route[idx--] = rem;
            }
        }

        return route;
    }

    private int[] SolveHeuristically()
    {
        var candidates = BuildInitialCandidates();
        if (candidates.Count == 0)
        {
            return Enumerable.Range(0, n).ToArray();
        }

        List<int> bestRoute = null;
        double bestCost = double.PositiveInfinity;

        foreach (var candidate in candidates)
        {
            var routeList = new List<int>(candidate);
            LocalOptimize(routeList);
            double cost = RouteCost(routeList);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = new List<int>(routeList);
            }

            if (n >= 25)
            {
                var working = new List<int>(routeList);
                double workingCost = cost;
                int iterations = n >= 60 ? 40 : (n >= 40 ? 30 : 15);
                for (int iter = 0; iter < iterations; ++iter)
                {
                    var perturbed = Perturb(working);
                    LocalOptimize(perturbed);
                    double newCost = RouteCost(perturbed);
                    if (newCost + 1e-9 < workingCost || random.NextDouble() < 0.15)
                    {
                        working = perturbed;
                        workingCost = newCost;
                    }
                    if (newCost < bestCost)
                    {
                        bestCost = newCost;
                        bestRoute = new List<int>(perturbed);
                    }
                }
            }
        }

        if (bestRoute == null)
        {
            bestRoute = new List<int>(candidates[0]);
        }

        return bestRoute.ToArray();
    }

    private List<int[]> BuildInitialCandidates()
    {
        var result = new List<int[]>();
        if (n == 0) return result;
        var indices = Enumerable.Range(0, n).ToArray();

        int[] byStart = indices.OrderBy(i => startCost[i]).ToArray();
        result.Add(BuildRouteByOrderedInsertion(byStart));
        result.Add(BuildRouteByOrderedInsertion(byStart.Reverse().ToArray()));

        int[] byRadius = indices.OrderBy(i => nodes[i].Radius).ToArray();
        result.Add(BuildRouteByOrderedInsertion(byRadius));
        result.Add(BuildRouteByOrderedInsertion(byRadius.Reverse().ToArray()));

        int[] bySpeed = indices.OrderBy(i => nodes[i].Speed).ToArray();
        result.Add(BuildRouteByOrderedInsertion(bySpeed));
        result.Add(BuildRouteByOrderedInsertion(bySpeed.Reverse().ToArray()));

        int[] byPlane = indices.OrderBy(i => Vector3d.AngleBetween(startNode.PlaneNormal, nodes[i].PlaneNormal)).ToArray();
        result.Add(BuildRouteByOrderedInsertion(byPlane));

        int[] byEnergy = indices.OrderBy(i => nodes[i].SpecificEnergy).ToArray();
        result.Add(BuildRouteByOrderedInsertion(byEnergy));

        var seedSet = new HashSet<int>();
        foreach (var idx in byStart.Take(Math.Min(5, n))) seedSet.Add(idx);
        foreach (var idx in byStart.Reverse().Take(Math.Min(3, n))) seedSet.Add(idx);
        foreach (var idx in byRadius.Take(Math.Min(3, n))) seedSet.Add(idx);
        foreach (var idx in byRadius.Reverse().Take(Math.Min(3, n))) seedSet.Add(idx);
        while (seedSet.Count < Math.Min(10, n))
        {
            seedSet.Add(random.Next(n));
        }
        foreach (var seed in seedSet)
        {
            result.Add(BuildNearestNeighborRoute(seed));
        }

        int randomCount = Math.Min(4 + n / 25, 8);
        for (int i = 0; i < randomCount; ++i)
        {
            result.Add(BuildRandomRoute());
        }

        return Deduplicate(result);
    }

    private List<int[]> Deduplicate(List<int[]> routes)
    {
        var seen = new HashSet<string>();
        var filtered = new List<int[]>();
        foreach (var route in routes)
        {
            if (route.Length != n) continue;
            string key = string.Join(",", route);
            if (seen.Add(key))
            {
                filtered.Add(route);
            }
        }
        return filtered;
    }

    private int[] BuildRouteByOrderedInsertion(int[] orderedNodes)
    {
        var route = new List<int>();
        foreach (var node in orderedNodes)
        {
            InsertNode(route, node);
        }
        return route.ToArray();
    }

    private void InsertNode(List<int> route, int node)
    {
        if (route.Count == 0)
        {
            route.Add(node);
            return;
        }

        int bestPosition = 0;
        double bestDelta = double.PositiveInfinity;
        for (int pos = 0; pos <= route.Count; ++pos)
        {
            int left = pos == 0 ? -1 : route[pos - 1];
            int right = pos == route.Count ? -2 : route[pos];
            double delta = ConnectionCost(left, node) + ConnectionCost(node, right) - ConnectionCost(left, right);
            if (delta + 1e-9 < bestDelta || (Math.Abs(delta - bestDelta) <= 1e-9 && pos < bestPosition))
            {
                bestDelta = delta;
                bestPosition = pos;
            }
        }
        route.Insert(bestPosition, node);
    }

    private int[] BuildNearestNeighborRoute(int firstNode)
    {
        var route = new List<int>();
        var used = new bool[n];
        route.Add(firstNode);
        used[firstNode] = true;
        int current = firstNode;

        for (int count = 1; count < n; ++count)
        {
            int bestNext = -1;
            double bestCost = double.PositiveInfinity;
            for (int candidate = 0; candidate < n; ++candidate)
            {
                if (used[candidate]) continue;
                double c = costMatrix[current][candidate];
                if (c + 1e-9 < bestCost)
                {
                    bestCost = c;
                    bestNext = candidate;
                }
            }

            if (bestNext == -1)
            {
                for (int candidate = 0; candidate < n; ++candidate)
                {
                    if (!used[candidate])
                    {
                        bestNext = candidate;
                        break;
                    }
                }
            }

            route.Add(bestNext);
            used[bestNext] = true;
            current = bestNext;
        }

        return route.ToArray();
    }

    private int[] BuildRandomRoute()
    {
        int[] perm = Enumerable.Range(0, n).ToArray();
        for (int i = perm.Length - 1; i > 0; --i)
        {
            int j = random.Next(i + 1);
            Swap(perm, i, j);
        }
        return perm;
    }

    private void LocalOptimize(List<int> route)
    {
        if (route.Count <= 1) return;
        int maxPasses = n < 30 ? 40 : (n < 60 ? 30 : 20);
        int passes = 0;
        bool improved;
        do
        {
            improved = false;
            while (TwoOpt(route)) improved = true;
            while (OrOpt(route, 1)) improved = true;
            if (n >= 6)
            {
                while (OrOpt(route, 2)) improved = true;
            }
            if (n >= 10)
            {
                int seg = Math.Min(3, n - 1);
                while (seg >= 3 && OrOpt(route, seg)) improved = true;
            }
            passes++;
        }
        while (improved && passes < maxPasses);
    }

    private bool TwoOpt(List<int> route)
    {
        int count = route.Count;
        for (int i = 0; i < count - 1; ++i)
        {
            int prev = i == 0 ? -1 : route[i - 1];
            for (int j = i + 1; j < count; ++j)
            {
                int next = j + 1 < count ? route[j + 1] : -2;
                double before = ConnectionCost(prev, route[i]) + ConnectionCost(route[j], next);
                double after = ConnectionCost(prev, route[j]) + ConnectionCost(route[i], next);
                if (after + 1e-9 < before)
                {
                    int length = j - i + 1;
                    route.Reverse(i, length);
                    return true;
                }
            }
        }
        return false;
    }

    private bool OrOpt(List<int> route, int segmentLength)
    {
        int count = route.Count;
        if (segmentLength <= 0 || segmentLength >= count) return false;
        for (int start = 0; start <= count - segmentLength; ++start)
        {
            int end = start + segmentLength - 1;
            int prev = start == 0 ? -1 : route[start - 1];
            int next = end + 1 < count ? route[end + 1] : -2;
            double removalDelta = ConnectionCost(prev, next) - ConnectionCost(prev, route[start]) - ConnectionCost(route[end], next);

            int compressedSize = count - segmentLength;
            var compressed = new int[compressedSize];
            int idx = 0;
            for (int i = 0; i < count; ++i)
            {
                if (i < start || i > end)
                {
                    compressed[idx++] = route[i];
                }
            }

            for (int insertPos = 0; insertPos <= compressedSize; ++insertPos)
            {
                int left = insertPos == 0 ? -1 : compressed[insertPos - 1];
                int right = insertPos == compressedSize ? -2 : compressed[insertPos];
                double insertionDelta = ConnectionCost(left, route[start]) + ConnectionCost(route[end], right) - ConnectionCost(left, right);
                double totalDelta = removalDelta + insertionDelta;

                if (totalDelta < -1e-9)
                {
                    var segment = route.GetRange(start, segmentLength);
                    route.RemoveRange(start, segmentLength);
                    route.InsertRange(insertPos, segment);
                    return true;
                }
            }
        }
        return false;
    }

    private List<int> Perturb(List<int> route)
    {
        int count = route.Count;
        if (count < 8) return new List<int>(route);
        int a = random.Next(0, count - 3);
        int b = random.Next(a + 1, count - 2);
        int c = random.Next(b + 1, count - 1);
        int d = random.Next(c + 1, count);

        var perturbed = new List<int>(count);
        perturbed.AddRange(route.GetRange(0, a));
        perturbed.AddRange(route.GetRange(c, d - c));
        perturbed.AddRange(route.GetRange(b, c - b));
        perturbed.AddRange(route.GetRange(a, b - a));
        perturbed.AddRange(route.GetRange(d, count - d));
        return perturbed;
    }

    private double RouteCost(IReadOnlyList<int> route)
    {
        if (route == null || route.Count == 0) return 0.0;
        double total = startCost[route[0]];
        for (int i = 0; i < route.Count - 1; ++i)
        {
            total += costMatrix[route[i]][route[i + 1]];
        }
        return total;
    }

    private double ConnectionCost(int from, int to)
    {
        if (to == -2) return 0.0;
        if (from == -1) return startCost[to];
        return costMatrix[from][to];
    }

    private static void Swap(int[] arr, int i, int j)
    {
        if (i == j) return;
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
}

**Key features:**

- **Physics-aware costs:** Pairwise edge weights combine Hohmann transfer, plane-change penalties, eccentricity differences, orientation/phase adjustment, and direct velocity matching.
- **Scale-adaptive solving:** Up to 8 nodes uses exact brute force, up to 13 uses Held–Karp DP, larger instances leverage multi-start heuristics (ordered insertion, nearest-neighbor, random seeds) with powerful local search (2-opt, Or-opt) and perturbation-driven refinement.
- **Path support:** The solver treats the mission as a start-to-finish path (no return leg) while respecting the mandated output of visiting indices `0..N-1`.
