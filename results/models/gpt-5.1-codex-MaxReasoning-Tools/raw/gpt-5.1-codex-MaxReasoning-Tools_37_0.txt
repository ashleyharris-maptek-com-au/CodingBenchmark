Below is a complete C# program that reads the directed graph, decomposes it into strongly connected components, and for every cyclic component computes an exact minimum directed feedback vertex set. Small components (≤20 vertices) are solved by exhaustive subset enumeration, while larger components use a branch‑and‑bound search with aggressive reductions (self-loop removal, sink/source peeling, SCC splitting, and a simple disjoint-cycle lower bound) to stay exact yet practical.

```csharp
using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        int m = scanner.NextInt();

        var adj = new List<int>[n];
        var rev = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            rev[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u < 0 || u >= n || v < 0 || v >= n) continue;
            adj[u].Add(v);
            rev[v].Add(u);
        }

        var components = GraphUtils.GetStronglyConnectedComponents(adj, rev);
        var answer = new List<int>();

        foreach (var comp in components)
        {
            if (comp.Count == 1)
            {
                int node = comp[0];
                bool selfLoop = false;
                var edges = adj[node];
                for (int j = 0; j < edges.Count; j++)
                {
                    if (edges[j] == node)
                    {
                        selfLoop = true;
                        break;
                    }
                }
                if (selfLoop) answer.Add(node);
                continue;
            }

            var solver = new ComponentSolver(adj, rev, comp);
            var subset = solver.Solve();
            answer.AddRange(subset);
        }

        answer.Sort();
        Console.WriteLine(answer.Count);
        if (answer.Count > 0)
            Console.WriteLine(string.Join(" ", answer));
        else
            Console.WriteLine();
    }
}

static class GraphUtils
{
    public static List<List<int>> GetStronglyConnectedComponents(List<int>[] adj, List<int>[] rev)
    {
        int n = adj.Length;
        var visited = new bool[n];
        var order = new Stack<int>(n);

        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
                Dfs1(i, adj, visited, order);
        }

        var visited2 = new bool[n];
        var result = new List<List<int>>();

        while (order.Count > 0)
        {
            int node = order.Pop();
            if (visited2[node]) continue;

            var comp = new List<int>();
            var stack = new Stack<int>();
            stack.Push(node);
            visited2[node] = true;

            while (stack.Count > 0)
            {
                int v = stack.Pop();
                comp.Add(v);
                var edges = rev[v];
                for (int i = 0; i < edges.Count; i++)
                {
                    int to = edges[i];
                    if (!visited2[to])
                    {
                        visited2[to] = true;
                        stack.Push(to);
                    }
                }
            }

            result.Add(comp);
        }

        return result;
    }

    private static void Dfs1(int start, List<int>[] adj, bool[] visited, Stack<int> order)
    {
        var stack = new Stack<(int node, int index)>();
        stack.Push((start, 0));
        visited[start] = true;

        while (stack.Count > 0)
        {
            var (node, idx) = stack.Pop();
            if (idx < adj[node].Count)
            {
                int to = adj[node][idx];
                stack.Push((node, idx + 1));
                if (!visited[to])
                {
                    visited[to] = true;
                    stack.Push((to, 0));
                }
            }
            else
            {
                order.Push(node);
            }
        }
    }
}

class ComponentSolver
{
    private readonly List<int>[] graph;
    private readonly List<int>[] reverse;
    private readonly int size;
    private readonly int[] nodes;
    private readonly Dictionary<int, int> globalToLocal;
    private readonly int[][] outEdges;
    private readonly int[][] inEdges;
    private readonly bool[] hasSelfLoop;
    private readonly List<int> selfLoopVertices;
    private readonly List<int> current;
    private List<int> bestSolutionLocal;
    private int bestSize;

    private const int BitmaskLimit = 20;
    private const int CyclePackingBoundLimit = 12;

    public ComponentSolver(List<int>[] graph, List<int>[] reverse, List<int> componentNodes)
    {
        this.graph = graph;
        this.reverse = reverse;
        size = componentNodes.Count;
        nodes = componentNodes.ToArray();
        globalToLocal = new Dictionary<int, int>(size * 2);
        for (int i = 0; i < size; i++)
            globalToLocal[nodes[i]] = i;

        outEdges = new int[size][];
        inEdges = new int[size][];
        hasSelfLoop = new bool[size];
        selfLoopVertices = new List<int>();
        current = new List<int>();
        bestSolutionLocal = new List<int>();

        for (int i = 0; i < size; i++)
        {
            int g = nodes[i];
            var outs = new List<int>();
            var edges = graph[g];
            for (int j = 0; j < edges.Count; j++)
            {
                int to = edges[j];
                if (to == g) hasSelfLoop[i] = true;
                if (globalToLocal.TryGetValue(to, out int local))
                    outs.Add(local);
            }
            outEdges[i] = outs.ToArray();
            if (hasSelfLoop[i]) selfLoopVertices.Add(i);
        }

        for (int i = 0; i < size; i++)
        {
            int g = nodes[i];
            var ins = new List<int>();
            var edges = reverse[g];
            for (int j = 0; j < edges.Count; j++)
            {
                int from = edges[j];
                if (globalToLocal.TryGetValue(from, out int local))
                    ins.Add(local);
            }
            inEdges[i] = ins.ToArray();
        }
    }

    public List<int> Solve()
    {
        if (size == 0) return new List<int>();
        if (size <= BitmaskLimit) return SolveByEnumeration();
        return SolveBySearch();
    }

    private List<int> SolveByEnumeration()
    {
        int n = size;
        var removed = new bool[n];
        var selection = new int[n];
        var indegTemp = new int[n];
        List<int> bestLocal = null;

        for (int k = 0; k <= n; k++)
        {
            Array.Clear(removed, 0, n);
            if (Enumerate(0, 0, k))
            {
                var result = new List<int>(bestLocal.Count);
                for (int i = 0; i < bestLocal.Count; i++)
                    result.Add(nodes[bestLocal[i]]);
                return result;
            }
        }

        var fallback = new List<int>(nodes);
        return fallback;

        bool Enumerate(int depth, int start, int target)
        {
            if (depth == target)
            {
                if (IsAcyclicSubset(removed, indegTemp))
                {
                    bestLocal = new List<int>(target);
                    for (int i = 0; i < target; i++)
                        bestLocal.Add(selection[i]);
                    return true;
                }
                return false;
            }

            for (int i = start; i <= n - (target - depth); i++)
            {
                removed[i] = true;
                selection[depth] = i;
                if (Enumerate(depth + 1, i + 1, target))
                    return true;
                removed[i] = false;
            }

            return false;
        }
    }

    private bool IsAcyclicSubset(bool[] removed, int[] indegTemp)
    {
        int remaining = 0;
        for (int i = 0; i < size; i++)
        {
            if (removed[i])
            {
                indegTemp[i] = 0;
                continue;
            }
            remaining++;
            int indegree = 0;
            var ins = inEdges[i];
            for (int j = 0; j < ins.Length; j++)
                if (!removed[ins[j]])
                    indegree++;
            indegTemp[i] = indegree;
        }

        if (remaining == 0) return true;

        var queue = new Queue<int>();
        for (int i = 0; i < size; i++)
            if (!removed[i] && indegTemp[i] == 0)
                queue.Enqueue(i);

        int visited = 0;
        while (queue.Count > 0)
        {
            int node = queue.Dequeue();
            visited++;
            var outs = outEdges[node];
            for (int j = 0; j < outs.Length; j++)
            {
                int to = outs[j];
                if (removed[to]) continue;
                indegTemp[to]--;
                if (indegTemp[to] == 0)
                    queue.Enqueue(to);
            }
        }

        return visited == remaining;
    }

    private List<int> SolveBySearch()
    {
        bestSize = size;
        bestSolutionLocal = new List<int>();
        current.Clear();

        bool[] active = new bool[size];
        for (int i = 0; i < size; i++)
            active[i] = true;

        int[] indeg = new int[size];
        int[] outdeg = new int[size];
        for (int i = 0; i < size; i++)
        {
            indeg[i] = inEdges[i].Length;
            outdeg[i] = outEdges[i].Length;
        }

        int activeCount = size;
        Recurse(active, indeg, outdeg, activeCount);

        var result = new List<int>(bestSolutionLocal.Count);
        for (int i = 0; i < bestSolutionLocal.Count; i++)
            result.Add(nodes[bestSolutionLocal[i]]);
        return result;
    }

    private void Recurse(bool[] active, int[] indeg, int[] outdeg, int activeCount)
    {
        if (current.Count >= bestSize) return;

        int forcedAdded = 0;
        if (!Simplify(active, indeg, outdeg, ref activeCount, ref forcedAdded))
        {
            PopForced(forcedAdded);
            return;
        }

        if (current.Count >= bestSize)
        {
            PopForced(forcedAdded);
            return;
        }

        if (activeCount == 0)
        {
            UpdateBest(current);
            PopForced(forcedAdded);
            return;
        }

        var sccs = GetActiveSccs(active);
        var cyclic = new List<List<int>>();
        for (int i = 0; i < sccs.Count; i++)
            if (sccs[i].Count > 1)
                cyclic.Add(sccs[i]);

        if (cyclic.Count == 0)
        {
            UpdateBest(current);
            PopForced(forcedAdded);
            return;
        }

        if (cyclic.Count > 1)
        {
            var candidate = new List<int>(current);
            for (int i = 0; i < cyclic.Count; i++)
            {
                var comp = cyclic[i];
                var subNodes = new List<int>(comp.Count);
                for (int j = 0; j < comp.Count; j++)
                    subNodes.Add(nodes[comp[j]]);
                var subSolver = new ComponentSolver(graph, reverse, subNodes);
                var subset = subSolver.Solve();
                for (int j = 0; j < subset.Count; j++)
                    if (globalToLocal.TryGetValue(subset[j], out int mapped))
                        candidate.Add(mapped);
            }
            UpdateBest(candidate);
            PopForced(forcedAdded);
            return;
        }

        if (ShouldPruneByCyclePacking(active))
        {
            PopForced(forcedAdded);
            return;
        }

        var cycle = FindCycle(active);
        if (cycle == null || cycle.Count == 0)
        {
            UpdateBest(current);
            PopForced(forcedAdded);
            return;
        }

        cycle.Sort((a, b) =>
        {
            int da = indeg[a] + outdeg[a];
            int db = indeg[b] + outdeg[b];
            return db.CompareTo(da);
        });

        for (int i = 0; i < cycle.Count; i++)
        {
            int v = cycle[i];
            if (!active[v]) continue;
            if (current.Count + 1 >= bestSize) continue;

            bool[] nextActive = (bool[])active.Clone();
            int[] nextIndeg = (int[])indeg.Clone();
            int[] nextOutdeg = (int[])outdeg.Clone();
            int nextActiveCount = activeCount;

            RemoveVertexPlain(v, nextActive, nextIndeg, nextOutdeg, ref nextActiveCount);

            current.Add(v);
            Recurse(nextActive, nextIndeg, nextOutdeg, nextActiveCount);
            current.RemoveAt(current.Count - 1);
        }

        PopForced(forcedAdded);
    }

    private bool Simplify(bool[] active, int[] indeg, int[] outdeg, ref int activeCount, ref int forcedAdded)
    {
        var queue = new Queue<int>();

        for (int i = 0; i < selfLoopVertices.Count; i++)
        {
            int v = selfLoopVertices[i];
            if (!active[v]) continue;
            if (current.Count + forcedAdded + 1 >= bestSize) return false;
            current.Add(v);
            forcedAdded++;
            RemoveVertexWithQueue(v, active, indeg, outdeg, ref activeCount, queue);
        }

        for (int v = 0; v < size; v++)
            if (active[v] && (indeg[v] == 0 || outdeg[v] == 0))
                queue.Enqueue(v);

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!active[v]) continue;
            RemoveVertexWithQueue(v, active, indeg, outdeg, ref activeCount, queue);
        }

        return true;
    }

    private void RemoveVertexWithQueue(int v, bool[] active, int[] indeg, int[] outdeg, ref int activeCount, Queue<int> queue)
    {
        if (!active[v]) return;
        active[v] = false;
        activeCount--;

        var outs = outEdges[v];
        for (int i = 0; i < outs.Length; i++)
        {
            int to = outs[i];
            if (!active[to]) continue;
            indeg[to]--;
            if (indeg[to] == 0 || outdeg[to] == 0)
                queue.Enqueue(to);
        }

        var ins = inEdges[v];
        for (int i = 0; i < ins.Length; i++)
        {
            int from = ins[i];
            if (!active[from]) continue;
            outdeg[from]--;
            if (indeg[from] == 0 || outdeg[from] == 0)
                queue.Enqueue(from);
        }

        indeg[v] = 0;
        outdeg[v] = 0;
    }

    private void RemoveVertexPlain(int v, bool[] active, int[] indeg, int[] outdeg, ref int activeCount)
    {
        if (!active[v]) return;
        active[v] = false;
        activeCount--;

        var outs = outEdges[v];
        for (int i = 0; i < outs.Length; i++)
        {
            int to = outs[i];
            if (active[to]) indeg[to]--;
        }

        var ins = inEdges[v];
        for (int i = 0; i < ins.Length; i++)
        {
            int from = ins[i];
            if (active[from]) outdeg[from]--;
        }

        indeg[v] = 0;
        outdeg[v] = 0;
    }

    private List<List<int>> GetActiveSccs(bool[] active)
    {
        var result = new List<List<int>>();
        var visited = new bool[size];
        var order = new Stack<int>();

        for (int i = 0; i < size; i++)
        {
            if (!active[i] || visited[i]) continue;
            var stack = new Stack<Frame>();
            stack.Push(new Frame(i, 0));
            visited[i] = true;

            while (stack.Count > 0)
            {
                var frame = stack.Pop();
                int node = frame.Node;
                int idx = frame.Index;

                if (idx < outEdges[node].Length)
                {
                    int to = outEdges[node][idx];
                    stack.Push(new Frame(node, idx + 1));
                    if (!active[to] || visited[to]) continue;
                    visited[to] = true;
                    stack.Push(new Frame(to, 0));
                }
                else
                {
                    order.Push(node);
                }
            }
        }

        var visited2 = new bool[size];
        while (order.Count > 0)
        {
            int node = order.Pop();
            if (!active[node] || visited2[node]) continue;

            var comp = new List<int>();
            var stack = new Stack<int>();
            stack.Push(node);
            visited2[node] = true;

            while (stack.Count > 0)
            {
                int v = stack.Pop();
                comp.Add(v);
                var edges = inEdges[v];
                for (int i = 0; i < edges.Length; i++)
                {
                    int to = edges[i];
                    if (!active[to] || visited2[to]) continue;
                    visited2[to] = true;
                    stack.Push(to);
                }
            }

            result.Add(comp);
        }

        return result;
    }

    private List<int> FindCycle(bool[] active)
    {
        var visited = new bool[size];
        var onStack = new bool[size];
        var parent = new int[size];
        for (int i = 0; i < size; i++) parent[i] = -1;
        var stack = new Stack<Frame>();

        for (int start = 0; start < size; start++)
        {
            if (!active[start] || visited[start]) continue;
            stack.Push(new Frame(start, 0));
            visited[start] = true;
            onStack[start] = true;
            parent[start] = -1;

            while (stack.Count > 0)
            {
                var frame = stack.Pop();
                int node = frame.Node;
                int idx = frame.Index;

                if (idx < outEdges[node].Length)
                {
                    int to = outEdges[node][idx];
                    stack.Push(new Frame(node, idx + 1));

                    if (!active[to]) continue;

                    if (!visited[to])
                    {
                        visited[to] = true;
                        parent[to] = node;
                        onStack[to] = true;
                        stack.Push(new Frame(to, 0));
                    }
                    else if (onStack[to])
                    {
                        var cycle = new List<int> { to };
                        int cur = node;
                        while (cur != to && cur != -1)
                        {
                            cycle.Add(cur);
                            cur = parent[cur];
                        }
                        cycle.Reverse();
                        return cycle;
                    }
                }
                else
                {
                    onStack[node] = false;
                }
            }
        }

        return null;
    }

    private bool ShouldPruneByCyclePacking(bool[] active)
    {
        int limit = bestSize - current.Count;
        if (limit <= 0) return true;
        if (limit > CyclePackingBoundLimit) return false;

        bool[] temp = (bool[])active.Clone();
        int found = 0;

        while (found < limit)
        {
            var cycle = FindCycle(temp);
            if (cycle == null || cycle.Count == 0) break;
            found++;
            for (int i = 0; i < cycle.Count; i++)
                temp[cycle[i]] = false;
        }

        return found >= limit;
    }

    private void UpdateBest(List<int> candidate)
    {
        if (candidate.Count < bestSize)
        {
            bestSize = candidate.Count;
            bestSolutionLocal = new List<int>(candidate);
        }
    }

    private void PopForced(int count)
    {
        for (int i = 0; i < count; i++)
            current.RemoveAt(current.Count - 1);
    }

    private readonly struct Frame
    {
        public int Node { get; }
        public int Index { get; }
        public Frame(int node, int index)
        {
            Node = node;
            Index = index;
        }
    }
}

class FastScanner
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr, len;

    public FastScanner(Stream stream) => this.stream = stream;

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0) return -1;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return 0;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > ' ')
        {
            val = val * 10 + c - '0';
            c = Read();
        }
        return val * sign;
    }
}
```
