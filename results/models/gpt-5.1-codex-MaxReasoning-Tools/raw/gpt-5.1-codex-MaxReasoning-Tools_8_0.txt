Key ideas implemented below:

- Parse the grid dimensions, then stream the maze with a `BufReader`, marking every walkable cell (anything except `#`) inside a compact bitset. This avoids storing the entire ASCII maze even for 50 000×50 000 instances.
- Run a bidirectional BFS: one frontier from `A`, one from `B`. Each side keeps a queue of cell indices (stored as `u32` to halve memory) and a `Vec<u8>` with just 4 direction codes plus a root mark, so we can reconstruct a simple, loop-free path.
- As soon as the two frontiers touch, follow the parent chains from the meeting cell back to `A` and to `B`, merge those halves, and print the coordinates. If the fronts exhaust without meeting, print `0`.

This keeps both time and memory scalable across tiny and massive mazes while respecting the required output format.

```rust
use std::io::{self, BufRead, BufReader, BufWriter, Write};

const TO_PARENT_UP: u8 = 0;
const TO_PARENT_DOWN: u8 = 1;
const TO_PARENT_LEFT: u8 = 2;
const TO_PARENT_RIGHT: u8 = 3;
const ROOT_MARK: u8 = 0xFE;
const UNVISITED: u8 = 0xFF;

fn main() -> io::Result<()> {
    run()
}

fn run() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let (height, width) = read_dimensions(&mut reader)?;
    let total_cells = height
        .checked_mul(width)
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Grid dimensions overflow"))?;

    if total_cells == 0 {
        let stdout = io::stdout();
        let mut out = BufWriter::new(stdout.lock());
        writeln!(out, "0")?;
        return Ok(());
    }

    if total_cells > u32::MAX as usize {
        return Err(io::Error::new(
            io::ErrorKind::InvalidInput,
            "Maze too large for 32-bit indexing",
        ));
    }

    let mut passable = vec![0u64; (total_cells + 63) / 64];
    let mut parents_start = vec![UNVISITED; total_cells];
    let mut parents_goal = vec![UNVISITED; total_cells];
    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    read_maze(
        &mut reader,
        height,
        width,
        &mut passable,
        &mut start_idx,
        &mut goal_idx,
    )?;

    let start_idx = start_idx.ok_or_else(|| {
        io::Error::new(io::ErrorKind::InvalidData, "Start cell 'A' not found in maze")
    })?;
    let goal_idx = goal_idx.ok_or_else(|| {
        io::Error::new(io::ErrorKind::InvalidData, "Goal cell 'B' not found in maze")
    })?;

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if start_idx == goal_idx {
        writeln!(out, "1")?;
        writeln!(out, "{} {}", start_idx / width, start_idx % width)?;
        return Ok(());
    }

    parents_start[start_idx] = ROOT_MARK;
    parents_goal[goal_idx] = ROOT_MARK;

    let initial_capacity = total_cells.min(1usize << 20);
    let mut queue_start: Vec<u32> = Vec::with_capacity(initial_capacity);
    let mut queue_goal: Vec<u32> = Vec::with_capacity(initial_capacity);
    queue_start.push(start_idx as u32);
    queue_goal.push(goal_idx as u32);
    let mut head_start = 0usize;
    let mut head_goal = 0usize;
    let mut meeting: Option<usize> = None;

    loop {
        let remaining_start = queue_start.len() - head_start;
        let remaining_goal = queue_goal.len() - head_goal;

        if remaining_start == 0 && remaining_goal == 0 {
            break;
        }

        let expand_start = if remaining_start == 0 {
            false
        } else if remaining_goal == 0 {
            true
        } else {
            remaining_start <= remaining_goal
        };

        if expand_start {
            if let Some(idx) = expand_frontier(
                &mut queue_start,
                &mut head_start,
                &mut parents_start,
                &parents_goal,
                &passable,
                width,
                height,
            ) {
                meeting = Some(idx);
                break;
            }
        } else if let Some(idx) = expand_frontier(
            &mut queue_goal,
            &mut head_goal,
            &mut parents_goal,
            &parents_start,
            &passable,
            width,
            height,
        ) {
            meeting = Some(idx);
            break;
        }
    }

    if let Some(meet_idx) = meeting {
        let (left, right) = collect_path_segments(
            meet_idx,
            start_idx,
            goal_idx,
            width,
            &parents_start,
            &parents_goal,
        );
        let total_len = left.len() + right.len();
        writeln!(out, "{}", total_len)?;
        for &idx in left.iter().rev() {
            let cell = idx as usize;
            writeln!(out, "{} {}", cell / width, cell % width)?;
        }
        for &idx in &right {
            let cell = idx as usize;
            writeln!(out, "{} {}", cell / width, cell % width)?;
        }
    } else {
        writeln!(out, "0")?;
    }

    Ok(())
}

fn read_dimensions<R: BufRead>(reader: &mut R) -> io::Result<(usize, usize)> {
    let mut line = String::new();
    loop {
        line.clear();
        let bytes = reader.read_line(&mut line)?;
        if bytes == 0 {
            return Err(io::Error::new(
                io::ErrorKind::UnexpectedEof,
                "Missing maze dimensions",
            ));
        }
        let trimmed = line.trim();
        if trimmed.is_empty() {
            continue;
        }
        let trimmed = trimmed.trim_start_matches('\u{feff}');
        let mut parts = trimmed.split_whitespace();
        let h = parts
            .next()
            .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Missing height value"))?
            .parse::<usize>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "Invalid height value"))?;
        let w = parts
            .next()
            .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Missing width value"))?
            .parse::<usize>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "Invalid width value"))?;
        if h == 0 || w == 0 {
            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                "Maze dimensions must be positive",
            ));
        }
        return Ok((h, w));
    }
}

fn read_maze<R: BufRead>(
    reader: &mut R,
    height: usize,
    width: usize,
    passable: &mut [u64],
    start_idx: &mut Option<usize>,
    goal_idx: &mut Option<usize>,
) -> io::Result<()> {
    let mut line = String::with_capacity(width + 8);
    for row in 0..height {
        loop {
            line.clear();
            let bytes = reader.read_line(&mut line)?;
            if bytes == 0 {
                return Err(io::Error::new(
                    io::ErrorKind::UnexpectedEof,
                    format!("Expected {height} rows but reached EOF at row {row}"),
                ));
            }
            trim_line_endings(&mut line);
            if line.is_empty() {
                continue;
            }
            break;
        }

        if line.len() < width {
            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                format!(
                    "Row {} is too short ({} characters, expected {})",
                    row,
                    line.len(),
                    width
                ),
            ));
        }

        let bytes = line.as_bytes();
        for col in 0..width {
            let idx = row * width + col;
            let ch = bytes[col];
            match ch {
                b'#' => {}
                b'A' => {
                    if start_idx.is_some() {
                        return Err(io::Error::new(
                            io::ErrorKind::InvalidData,
                            "Multiple 'A' cells found",
                        ));
                    }
                    set_open(passable, idx);
                    *start_idx = Some(idx);
                }
                b'B' => {
                    if goal_idx.is_some() {
                        return Err(io::Error::new(
                            io::ErrorKind::InvalidData,
                            "Multiple 'B' cells found",
                        ));
                    }
                    set_open(passable, idx);
                    *goal_idx = Some(idx);
                }
                _ => set_open(passable, idx),
            }
        }
    }
    Ok(())
}

fn expand_frontier(
    queue: &mut Vec<u32>,
    head: &mut usize,
    parents_self: &mut [u8],
    parents_other: &[u8],
    passable: &[u64],
    width: usize,
    height: usize,
) -> Option<usize> {
    if *head >= queue.len() {
        return None;
    }
    let current = queue[*head] as usize;
    *head += 1;

    if parents_other[current] != UNVISITED {
        return Some(current);
    }

    let row = current / width;
    let col = current % width;

    if row > 0 {
        let next = current - width;
        if is_open(passable, next) && parents_self[next] == UNVISITED {
            parents_self[next] = TO_PARENT_DOWN;
            queue.push(next as u32);
            if parents_other[next] != UNVISITED {
                return Some(next);
            }
        }
    }
    if row + 1 < height {
        let next = current + width;
        if is_open(passable, next) && parents_self[next] == UNVISITED {
            parents_self[next] = TO_PARENT_UP;
            queue.push(next as u32);
            if parents_other[next] != UNVISITED {
                return Some(next);
            }
        }
    }
    if col > 0 {
        let next = current - 1;
        if is_open(passable, next) && parents_self[next] == UNVISITED {
            parents_self[next] = TO_PARENT_RIGHT;
            queue.push(next as u32);
            if parents_other[next] != UNVISITED {
                return Some(next);
            }
        }
    }
    if col + 1 < width {
        let next = current + 1;
        if is_open(passable, next) && parents_self[next] == UNVISITED {
            parents_self[next] = TO_PARENT_LEFT;
            queue.push(next as u32);
            if parents_other[next] != UNVISITED {
                return Some(next);
            }
        }
    }

    None
}

fn collect_path_segments(
    meeting: usize,
    start_idx: usize,
    goal_idx: usize,
    width: usize,
    parents_start: &[u8],
    parents_goal: &[u8],
) -> (Vec<u32>, Vec<u32>) {
    let mut left = Vec::<u32>::new();
    let mut current = meeting;
    left.push(current as u32);
    while current != start_idx {
        let dir = parents_start[current];
        current = step_to_parent(current, dir, width);
        left.push(current as u32);
    }

    let mut right = Vec::<u32>::new();
    current = meeting;
    while current != goal_idx {
        let dir = parents_goal[current];
        current = step_to_parent(current, dir, width);
        right.push(current as u32);
    }

    (left, right)
}

fn step_to_parent(idx: usize, dir: u8, width: usize) -> usize {
    match dir {
        TO_PARENT_UP => idx.checked_sub(width).expect("Broken parent link (up)"),
        TO_PARENT_DOWN => idx.checked_add(width).expect("Broken parent link (down)"),
        TO_PARENT_LEFT => idx.checked_sub(1).expect("Broken parent link (left)"),
        TO_PARENT_RIGHT => idx.checked_add(1).expect("Broken parent link (right)"),
        _ => panic!("Invalid parent direction {dir}"),
    }
}

#[inline(always)]
fn set_open(bits: &mut [u64], idx: usize) {
    bits[idx >> 6] |= 1u64 << (idx & 63);
}

#[inline(always)]
fn is_open(bits: &[u64], idx: usize) -> bool {
    ((bits[idx >> 6] >> (idx & 63)) & 1u64) != 0
}

fn trim_line_endings(line: &mut String) {
    loop {
        let remove = matches!(line.as_bytes().last(), Some(b) if *b == b'\n' || *b == b'\r');
        if remove {
            line.pop();
        } else {
            break;
        }
    }
}
```

This program reads from stdin, prints either the full 0-indexed path from `A` to `B` (including both endpoints) or `0` when no route exists, and scales cleanly all the way up to 50 000×50 000 mazes.
