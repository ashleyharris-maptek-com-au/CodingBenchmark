@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16u>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4u>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(value : u32, shift : u32) -> u32 {
    return (value << shift) | (value >> (32u - shift));
}

fn count_trailing_zeros32(value : u32) -> u32 {
    if (value == 0u) {
        return 32u;
    }
    var v = value;
    var count = 0u;
    loop {
        if ((v & 1u) != 0u) {
            break;
        }
        count = count + 1u;
        v = v >> 1u;
    }
    return count;
}

fn count_trailing_zeros64(lo : u32, hi : u32) -> u32 {
    if (lo != 0u) {
        return count_trailing_zeros32(lo);
    }
    if (hi != 0u) {
        return 32u + count_trailing_zeros32(hi);
    }
    return 64u;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let idx = gid.x;
    if (idx >= params.z) {
        return;
    }

    let nonce = params.y + idx;
    var state : array<u32, 16u>;

    for (var i = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round = 0u; round < 8u; round = round + 1u) {
        // state[0..3]
        state[0] = state[0] + state[1];
        state[3] = rotl(state[3] ^ state[0], 16u);
        state[2] = state[2] + state[3];
        state[1] = rotl(state[1] ^ state[2], 12u);
        state[0] = state[0] + state[1];
        state[3] = rotl(state[3] ^ state[0], 8u);
        state[2] = state[2] + state[3];
        state[1] = rotl(state[1] ^ state[2], 7u);

        // state[4..7]
        state[4] = state[4] + state[5];
        state[7] = rotl(state[7] ^ state[4], 16u);
        state[6] = state[6] + state[7];
        state[5] = rotl(state[5] ^ state[6], 12u);
        state[4] = state[4] + state[5];
        state[7] = rotl(state[7] ^ state[4], 8u);
        state[6] = state[6] + state[7];
        state[5] = rotl(state[5] ^ state[6], 7u);

        // state[8..11]
        state[8]  = state[8]  + state[9];
        state[11] = rotl(state[11] ^ state[8], 16u);
        state[10] = state[10] + state[11];
        state[9]  = rotl(state[9]  ^ state[10], 12u);
        state[8]  = state[8]  + state[9];
        state[11] = rotl(state[11] ^ state[8], 8u);
        state[10] = state[10] + state[11];
        state[9]  = rotl(state[9]  ^ state[10], 7u);

        // state[12..15]
        state[12] = state[12] + state[13];
        state[15] = rotl(state[15] ^ state[12], 16u);
        state[14] = state[14] + state[15];
        state[13] = rotl(state[13] ^ state[14], 12u);
        state[12] = state[12] + state[13];
        state[15] = rotl(state[15] ^ state[12], 8u);
        state[14] = state[14] + state[15];
        state[13] = rotl(state[13] ^ state[14], 7u);
    }

    var h0 = state[0] ^ state[2] ^ state[4] ^ state[6];
    h0 = h0 ^ state[8] ^ state[10] ^ state[12] ^ state[14];

    var h1 = state[1] ^ state[3] ^ state[5] ^ state[7];
    h1 = h1 ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    let tz = count_trailing_zeros64(h0, h1);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
