#include <algorithm>

using namespace std;

struct Graph {
    int n = 0;
    vector<int> offsets;
    vector<int> adj;
};

void ensureColored(vector<int>& colors, int k, mt19937& rng) {
    if (k <= 0) return;
    uniform_int_distribution<int> dist(0, k - 1);
    for (int& c : colors) {
        if (c < 0 || c >= k) {
            c = dist(rng);
        }
    }
}

size_t countConflicts(const vector<pair<int, int>>& edges, const vector<int>& color) {
    size_t conflicts = 0;
    for (const auto& e : edges) {
        int u = e.first;
        int v = e.second;
        int cu = (u < (int)color.size()) ? color[u] : -1;
        int cv = (v < (int)color.size()) ? color[v] : -1;
        if (cu >= 0 && cu == cv) {
            ++conflicts;
        }
    }
    return conflicts;
}

bool greedyColor(const Graph& g, const vector<int>& order, int k, vector<int>& color) {
    if (k <= 0 || (int)order.size() != g.n) return false;
    vector<int> seen(k, -1);
    int marker = 0;
    for (int idx = 0; idx < g.n; ++idx) {
        int u = order[idx];
        ++marker;
        for (int e = g.offsets[u]; e < g.offsets[u + 1]; ++e) {
            int v = g.adj[e];
            int c = color[v];
            if (c >= 0 && c < k) {
                seen[c] = marker;
            }
        }
        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (seen[c] != marker) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            return false;
        }
        color[u] = chosen;
    }
    return true;
}

vector<int> buildDegeneracyOrder(const Graph& g, const vector<int>& degree) {
    const int n = g.n;
    vector<int> order;
    order.reserve(n);
    vector<int> deg = degree;
    vector<char> removed(n, 0);
    using Node = pair<int, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    for (int i = 0; i < n; ++i) {
        pq.emplace(deg[i], i);
    }
    while ((int)order.size() < n && !pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (removed[u]) continue;
        removed[u] = 1;
        order.push_back(u);
        for (int e = g.offsets[u]; e < g.offsets[u + 1]; ++e) {
            int v = g.adj[e];
            if (!removed[v]) {
                --deg[v];
                pq.emplace(deg[v], v);
            }
        }
    }
    reverse(order.begin(), order.end());
    return order;
}

bool dsaturColor(const Graph& g, int k, const vector<int>& degree, vector<int>& color) {
    if (k <= 0) return false;
    const int n = g.n;
    color.assign(n, -1);
    vector<int> sat(n, 0);
    vector<uint64_t> used(n, 0);

    struct Node {
        int sat;
        int deg;
        int idx;
    };
    auto cmp = [](const Node& a, const Node& b) {
        if (a.sat != b.sat) return a.sat < b.sat;
        if (a.deg != b.deg) return a.deg < b.deg;
        return a.idx > b.idx;
    };
    priority_queue<Node, vector<Node>, decltype(cmp)> pq(cmp);
    for (int i = 0; i < n; ++i) {
        pq.push({0, degree[i], i});
    }

    const uint64_t allowedMask = (k >= 64) ? numeric_limits<uint64_t>::max() : ((uint64_t(1) << k) - 1ull);

    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();
        int u = cur.idx;
        if (color[u] != -1) continue;
        if (cur.sat != sat[u]) continue;

        uint64_t mask = used[u];
        if (((~mask) & allowedMask) == 0ull) {
            return false;
        }

        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if ((mask & (uint64_t(1) << c)) == 0ull) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) return false;

        color[u] = chosen;
        for (int e = g.offsets[u]; e < g.offsets[u + 1]; ++e) {
            int v = g.adj[e];
            if (color[v] != -1) continue;
            uint64_t bit = (uint64_t(1) << chosen);
            if ((used[v] & bit) == 0ull) {
                used[v] |= bit;
                ++sat[v];
                pq.push({sat[v], degree[v], v});
            }
        }
    }
    return true;
}

bool localRepair(const Graph& g,
                 const vector<pair<int, int>>& edges,
                 int k,
                 vector<int>& color,
                 mt19937& rng,
                 const chrono::steady_clock::time_point& start,
                 chrono::milliseconds limit) {
    if (k <= 0) return false;
    const int n = g.n;
    vector<int> conflictCount(n, 0);
    vector<int> conflictPos(n, -1);
    vector<int> conflictList;
    conflictList.reserve(n / 4 + 1);
    long long totalConflicts = 0;

    for (const auto& e : edges) {
        int u = e.first;
        int v = e.second;
        if (color[u] >= 0 && color[u] == color[v]) {
            ++conflictCount[u];
            ++conflictCount[v];
            ++totalConflicts;
        }
    }

    auto addConflict = [&](int v) {
        if (conflictCount[v] > 0 && conflictPos[v] == -1) {
            conflictPos[v] = static_cast<int>(conflictList.size());
            conflictList.push_back(v);
        }
    };
    auto removeConflict = [&](int v) {
        if (conflictCount[v] == 0 && conflictPos[v] != -1) {
            int pos = conflictPos[v];
            int last = conflictList.back();
            conflictList[pos] = last;
            conflictPos[last] = pos;
            conflictList.pop_back();
            conflictPos[v] = -1;
        }
    };

    for (int i = 0; i < n; ++i) {
        if (conflictCount[i] > 0) addConflict(i);
    }
    if (totalConflicts == 0) return true;

    auto timeExceeded = [&]() {
        return chrono::steady_clock::now() - start > limit;
    };

    long long iterLimit = 2000000LL;
    long long scaled = static_cast<long long>(edges.size()) * 2LL;
    if (scaled > 0) {
        iterLimit = min(iterLimit, max(200000LL, scaled));
    }

    vector<int> freq(k, 0);
    vector<int> freqStamp(k, -1);
    vector<int> bestColors;
    bestColors.reserve(k);
    int freqIteration = 0;

    for (long long iter = 0; iter < iterLimit && totalConflicts > 0; ++iter) {
        if (timeExceeded() || conflictList.empty()) break;
        size_t poolSize = conflictList.size();
        int idx = static_cast<int>(rng() % poolSize);
        int u = conflictList[idx];
        if (conflictCount[u] == 0) {
            int last = conflictList.back();
            conflictList[idx] = last;
            conflictPos[last] = idx;
            conflictList.pop_back();
            conflictPos[u] = -1;
            continue;
        }

        ++freqIteration;
        if (freqIteration == INT_MAX) {
            fill(freqStamp.begin(), freqStamp.end(), -1);
            freqIteration = 1;
        }
        for (int e = g.offsets[u]; e < g.offsets[u + 1]; ++e) {
            int v = g.adj[e];
            int cv = color[v];
            if (cv >= 0 && cv < k) {
                if (freqStamp[cv] != freqIteration) {
                    freqStamp[cv] = freqIteration;
                    freq[cv] = 0;
                }
                ++freq[cv];
            }
        }

        int minConf = INT_MAX;
        bestColors.clear();
        for (int c = 0; c < k; ++c) {
            int val = (freqStamp[c] == freqIteration) ? freq[c] : 0;
            if (val < minConf) {
                minConf = val;
                bestColors.clear();
                bestColors.push_back(c);
            } else if (val == minConf) {
                bestColors.push_back(c);
            }
        }
        if (bestColors.empty()) continue;

        int oldColor = color[u];
        int newColor = bestColors[rng() % bestColors.size()];
        if (newColor == oldColor) {
            if (bestColors.size() > 1) {
                for (int t = 0; t < 3 && newColor == oldColor; ++t) {
                    newColor = bestColors[rng() % bestColors.size()];
                }
            }
            if (newColor == oldColor) {
                if (k > 1) {
                    newColor = (oldColor + 1) % k;
                } else {
                    continue;
                }
            }
        }
        if (newColor == oldColor) continue;

        for (int e = g.offsets[u]; e < g.offsets[u + 1]; ++e) {
            int v = g.adj[e];
            if (color[v] == oldColor) {
                --conflictCount[v];
                if (conflictCount[v] == 0) removeConflict(v);
                --conflictCount[u];
                --totalConflicts;
            }
        }

        color[u] = newColor;

        for (int e = g.offsets[u]; e < g.offsets[u + 1]; ++e) {
            int v = g.adj[e];
            if (color[v] == newColor) {
                ++conflictCount[v];
                if (conflictCount[v] == 1) addConflict(v);
                ++conflictCount[u];
                ++totalConflicts;
            }
        }

        if (conflictCount[u] == 0) {
            removeConflict(u);
        } else {
            addConflict(u);
        }
    }

    return totalConflicts == 0;
}

struct ColoringResult {
    bool success = false;
    vector<int> colors;
    size_t conflicts = numeric_limits<size_t>::max();
};

ColoringResult runHeuristicColoring(const Graph& g,
                                    const vector<pair<int, int>>& edges,
                                    const vector<int>& degree,
                                    int k,
                                    mt19937& rng,
                                    const chrono::steady_clock::time_point& start,
                                    chrono::milliseconds limit) {
    ColoringResult res;
    if (g.n == 0) {
        res.success = true;
        res.conflicts = 0;
        return res;
    }
    if (k <= 0) {
        res.colors.assign(g.n, 0);
        res.conflicts = edges.empty() ? 0 : edges.size();
        res.success = (res.conflicts == 0);
        return res;
    }

    auto timeExceeded = [&]() {
        return chrono::steady_clock::now() - start > limit;
    };

    vector<int> base(g.n);
    iota(base.begin(), base.end(), 0);

    vector<int> working(g.n, -1);
    vector<int> bestColors;
    size_t bestConflicts = numeric_limits<size_t>::max();
    bool hasBest = false;

    auto attempt = [&](const vector<int>& order) -> bool {
        if ((int)order.size() != g.n) return false;
        fill(working.begin(), working.end(), -1);
        bool ok = greedyColor(g, order, k, working);
        ensureColored(working, k, rng);
        size_t conflicts = countConflicts(edges, working);
        if (!hasBest || conflicts < bestConflicts) {
            bestColors = working;
            bestConflicts = conflicts;
            hasBest = true;
        }
        return ok && conflicts == 0;
    };

    vector<int> order;
    order.reserve(g.n);

    if (!timeExceeded()) {
        order = base;
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            return a < b;
        });
        if (attempt(order)) {
            res.success = true;
            res.colors = bestColors;
            res.conflicts = 0;
            return res;
        }
    }

    if (!timeExceeded()) {
        if (attempt(base)) {
            res.success = true;
            res.colors = bestColors;
            res.conflicts = 0;
            return res;
        }
    }

    if (!timeExceeded() && g.n <= 200000 && edges.size() <= 2000000ull) {
        vector<int> degeneracy = buildDegeneracyOrder(g, degree);
        if (!degeneracy.empty() && attempt(degeneracy)) {
            res.success = true;
            res.colors = bestColors;
            res.conflicts = 0;
            return res;
        }
    }

    if (!timeExceeded()) {
        order = base;
        shuffle(order.begin(), order.end(), rng);
        if (attempt(order)) {
            res.success = true;
            res.colors = bestColors;
            res.conflicts = 0;
            return res;
        }
    }

    if (!timeExceeded()) {
        order = base;
        shuffle(order.begin(), order.end(), rng);
        if (attempt(order)) {
            res.success = true;
            res.colors = bestColors;
            res.conflicts = 0;
            return res;
        }
    }

    if (!hasBest) {
        bestColors = base;
        ensureColored(bestColors, k, rng);
        bestConflicts = countConflicts(edges, bestColors);
        hasBest = true;
    }

    if (hasBest && bestConflicts > 0 && !timeExceeded()) {
        vector<int> candidate = bestColors;
        bool repaired = localRepair(g, edges, k, candidate, rng, start, limit);
        size_t conflicts = countConflicts(edges, candidate);
        if (conflicts < bestConflicts) {
            bestColors = candidate;
            bestConflicts = conflicts;
        }
        if (repaired && conflicts == 0) {
            res.success = true;
            res.colors = bestColors;
            res.conflicts = 0;
            return res;
        }
    }

    res.colors = bestColors;
    res.conflicts = bestConflicts;
    res.success = (bestConflicts == 0);
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    Graph g;
    g.n = n;
    g.offsets.assign(n + 1, 0);

    vector<pair<int, int>> edges;
    edges.reserve(m);
    vector<int> degree(n, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        edges.emplace_back(u, v);
        ++degree[u];
        ++degree[v];
    }

    for (int i = 0; i < n; ++i) {
        g.offsets[i + 1] = g.offsets[i] + degree[i];
    }
    g.adj.assign(g.offsets.back(), 0);
    vector<int> cursor = g.offsets;
    for (const auto& e : edges) {
        int u = e.first;
        int v = e.second;
        g.adj[cursor[u]++] = v;
        g.adj[cursor[v]++] = u;
    }

    auto startTime = chrono::steady_clock::now();
    chrono::milliseconds softLimit(270000);
    random_device rd;
    mt19937 rng(rd());

    vector<int> colors(n, 0);
    bool solved = false;

    if (k > 0 && n <= 5000 && k <= 64) {
        vector<int> dsColor;
        if (dsaturColor(g, k, degree, dsColor)) {
            colors = dsColor;
            solved = true;
        } else {
            auto result = runHeuristicColoring(g, edges, degree, k, rng, startTime, softLimit);
            colors = result.colors;
            solved = result.success;
        }
    } else {
        auto result = runHeuristicColoring(g, edges, degree, k, rng, startTime, softLimit);
        colors = result.colors;
        solved = result.success;
    }

    if ((int)colors.size() != n) {
        colors.assign(n, 0);
    }
    ensureColored(colors, k, rng);

    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        if (k > 0) {
            cout << colors[i];
        } else {
            cout << 0;
        }
    }
    cout << '\n';
    return 0;
}
