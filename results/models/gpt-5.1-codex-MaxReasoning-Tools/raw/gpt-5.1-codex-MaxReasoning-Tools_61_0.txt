declare const require: any;
const fs = require('fs');

type PackageId = string | number;

interface PackageNode {
    id: PackageId;
    x: number;
    y: number;
    angle: number;
    dist2: number;
}

const rawInput: string = fs.readFileSync(0, 'utf8');
const trimmedInput = rawInput.trim();

if (!trimmedInput) {
    console.log('{"routes":[]}');
} else {
    const data = JSON.parse(trimmedInput);
    main(data);
}

function main(data: any): void {
    const depotArray = Array.isArray(data.depot) ? data.depot : [0, 0];
    const depotX = toFiniteNumber(depotArray[0], 0);
    const depotY = toFiniteNumber(depotArray[1], 0);

    const rawPackages = Array.isArray(data.packages) ? data.packages : [];
    const packages: PackageNode[] = rawPackages.map((pkg: any, index: number) => {
        const x = toFiniteNumber(pkg?.x, 0);
        const y = toFiniteNumber(pkg?.y, 0);
        const dx = x - depotX;
        const dy = y - depotY;
        let angle = Math.atan2(dy, dx);
        if (angle < 0) {
            angle += Math.PI * 2;
        }
        const dist2 = dx * dx + dy * dy;
        const id: PackageId = pkg && pkg.id !== undefined ? pkg.id : index;
        return { id, x, y, angle, dist2 };
    });

    const totalPackages = packages.length;
    if (totalPackages === 0) {
        console.log('{"routes":[]}');
        return;
    }

    let truckCount = Math.max(0, Math.floor(toFiniteNumber(data.truckCount, 1)));
    if (truckCount <= 0) {
        truckCount = 1;
    }
    const maxUsableRoutes = Math.max(1, Math.min(truckCount, totalPackages));

    const angleOrder = packages
        .map((_, idx) => idx)
        .sort((a, b) => {
            const angleDiff = packages[a].angle - packages[b].angle;
            if (angleDiff !== 0) return angleDiff;
            const distDiff = packages[a].dist2 - packages[b].dist2;
            if (distDiff !== 0) return distDiff;
            return a - b;
        });

    const candidateCountsSet = new Set<number>();
    candidateCountsSet.add(maxUsableRoutes);
    if (maxUsableRoutes > 1) {
        candidateCountsSet.add(Math.max(1, Math.ceil(maxUsableRoutes / 2)));
        candidateCountsSet.add(1);
    }
    if (maxUsableRoutes > 2) {
        candidateCountsSet.add(maxUsableRoutes - 1);
    }
    const candidateCounts = Array.from(candidateCountsSet).map((count) =>
        Math.max(1, Math.min(count, totalPackages))
    );
    candidateCounts.sort((a, b) => b - a);

    interface BuiltPlan {
        indexRoutes: number[][];
        idRoutes: PackageId[][];
        distance: number;
    }

    const plans: BuiltPlan[] = [];
    for (const count of candidateCounts) {
        const sweepClusters = createSweepClusters(count);
        plans.push(buildPlanFromClusters(sweepClusters));
        if (shouldUseKMeans(count)) {
            const kMeansClusters = createKMeansClusters(count);
            plans.push(buildPlanFromClusters(kMeansClusters));
        }
    }

    if (plans.length === 0) {
        const fallbackClusters = [packages.map((_, idx) => idx)];
        plans.push(buildPlanFromClusters(fallbackClusters));
    }

    let bestPlan: BuiltPlan | null = null;
    const EPS = 1e-6;
    for (const plan of plans) {
        if (!bestPlan || plan.distance + EPS < bestPlan.distance) {
            bestPlan = plan;
        } else if (Math.abs(plan.distance - bestPlan.distance) <= EPS) {
            if (plan.indexRoutes.length < bestPlan.indexRoutes.length) {
                bestPlan = plan;
            }
        }
    }

    if (!bestPlan) {
        const fallbackClusters = [packages.map((_, idx) => idx)];
        bestPlan = buildPlanFromClusters(fallbackClusters);
    }

    const output = JSON.stringify({ routes: bestPlan.idRoutes });
    console.log(output);

    function shouldUseKMeans(clusterCount: number): boolean {
        if (clusterCount <= 1) return false;
        if (clusterCount >= totalPackages) return false;
        if (totalPackages > 100000) return false;
        if (totalPackages > 200 && clusterCount > totalPackages * 0.8) return false;
        const complexity = clusterCount * totalPackages;
        if (complexity > 5000000) return false;
        return true;
    }

    function createSweepClusters(clusterCount: number): number[][] {
        const clusters: number[][] = [];
        const baseSize = Math.floor(totalPackages / clusterCount);
        let remainder = totalPackages % clusterCount;
        let cursor = 0;

        for (let i = 0; i < clusterCount; i++) {
            const currentSize = baseSize + (remainder > 0 ? 1 : 0);
            if (remainder > 0) remainder--;
            const group: number[] = [];
            for (let j = 0; j < currentSize && cursor < angleOrder.length; j++) {
                group.push(angleOrder[cursor++]);
            }
            clusters.push(group);
        }

        if (cursor < angleOrder.length && clusters.length > 0) {
            const lastGroup = clusters[clusters.length - 1];
            while (cursor < angleOrder.length) {
                lastGroup.push(angleOrder[cursor++]);
            }
        }

        return clusters;
    }

    function createKMeansClusters(clusterCount: number): number[][] {
        const n = totalPackages;
        const k = Math.max(1, Math.min(clusterCount, n));
        if (k === 1) {
            return [packages.map((_, idx) => idx)];
        }

        const centroids: { x: number; y: number }[] = [];
        const chosen = new Array(n).fill(false);

        let firstIndex = 0;
        let maxDist = -1;
        for (let i = 0; i < n; i++) {
            if (packages[i].dist2 > maxDist) {
                maxDist = packages[i].dist2;
                firstIndex = i;
            }
        }
        centroids.push({ x: packages[firstIndex].x, y: packages[firstIndex].y });
        chosen[firstIndex] = true;

        while (centroids.length < k) {
            let bestPoint = -1;
            let bestScore = -1;
            for (let i = 0; i < n; i++) {
                if (chosen[i]) continue;
                let minDist = Infinity;
                for (let c = 0; c < centroids.length; c++) {
                    const dx = packages[i].x - centroids[c].x;
                    const dy = packages[i].y - centroids[c].y;
                    const dist = dx * dx + dy * dy;
                    if (dist < minDist) {
                        minDist = dist;
                    }
                }
                if (minDist > bestScore) {
                    bestScore = minDist;
                    bestPoint = i;
                }
            }
            if (bestPoint === -1) {
                break;
            }
            centroids.push({ x: packages[bestPoint].x, y: packages[bestPoint].y });
            chosen[bestPoint] = true;
            if (centroids.length === n) {
                break;
            }
        }

        while (centroids.length < k) {
            const fallbackIndex = centroids.length % n;
            centroids.push({ x: packages[fallbackIndex].x, y: packages[fallbackIndex].y });
        }

        const assignments = new Array(n).fill(0);
        const maxIterations = totalPackages > 40000 ? 15 : totalPackages > 10000 ? 20 : 30;

        for (let iter = 0; iter < maxIterations; iter++) {
            let changed = false;

            for (let i = 0; i < n; i++) {
                let bestCluster = 0;
                let bestDist = Infinity;
                for (let c = 0; c < k; c++) {
                    const dx = packages[i].x - centroids[c].x;
                    const dy = packages[i].y - centroids[c].y;
                    const dist = dx * dx + dy * dy;
                    if (dist < bestDist) {
                        bestDist = dist;
                        bestCluster = c;
                    }
                }
                if (assignments[i] !== bestCluster) {
                    assignments[i] = bestCluster;
                    changed = true;
                }
            }

            const counts = new Array(k).fill(0);
            for (let i = 0; i < n; i++) {
                counts[assignments[i]]++;
            }

            const empties: number[] = [];
            for (let c = 0; c < k; c++) {
                if (counts[c] === 0) {
                    empties.push(c);
                }
            }

            if (empties.length > 0) {
                for (const emptyCluster of empties) {
                    let donorPoint = -1;
                    let donorCluster = -1;
                    let farthestDist = -1;
                    for (let i = 0; i < n; i++) {
                        const clusterIndex = assignments[i];
                        if (counts[clusterIndex] <= 1) continue;
                        const dx = packages[i].x - centroids[clusterIndex].x;
                        const dy = packages[i].y - centroids[clusterIndex].y;
                        const dist = dx * dx + dy * dy;
                        if (dist > farthestDist) {
                            farthestDist = dist;
                            donorPoint = i;
                            donorCluster = clusterIndex;
                        }
                    }
                    if (donorPoint === -1) {
                        donorPoint = 0;
                        donorCluster = assignments[0];
                    }
                    assignments[donorPoint] = emptyCluster;
                    counts[donorCluster]--;
                    counts[emptyCluster]++;
                    changed = true;
                }
            }

            const sumX = new Array(k).fill(0);
            const sumY = new Array(k).fill(0);
            for (let i = 0; i < n; i++) {
                const clusterIndex = assignments[i];
                sumX[clusterIndex] += packages[i].x;
                sumY[clusterIndex] += packages[i].y;
            }
            for (let c = 0; c < k; c++) {
                if (counts[c] > 0) {
                    centroids[c].x = sumX[c] / counts[c];
                    centroids[c].y = sumY[c] / counts[c];
                }
            }

            if (!changed) {
                break;
            }
        }

        const clusters = Array.from({ length: k }, () => [] as number[]);
        for (let i = 0; i < n; i++) {
            clusters[assignments[i]].push(i);
        }
        return clusters;
    }

    function buildPlanFromClusters(clusters: number[][]): BuiltPlan {
        const routeOrders: number[][] = [];
        for (const cluster of clusters) {
            if (cluster.length === 0) continue;
            const order = orderCluster(cluster);
            if (order.length === 0) continue;
            improveRoute(order);
            routeOrders.push(order);
        }

        if (routeOrders.length === 0 && totalPackages > 0) {
            const fallbackOrder = orderCluster(packages.map((_, idx) => idx));
            improveRoute(fallbackOrder);
            routeOrders.push(fallbackOrder);
        }

        const sanitized = ensureCoverage(routeOrders);
        const indexRoutes: number[][] = [];
        const idRoutes: PackageId[][] = [];
        let totalDistance = 0;

        for (const route of sanitized) {
            indexRoutes.push(route.slice());
            idRoutes.push(route.map((idx) => packages[idx].id));
            totalDistance += computeRouteDistance(route);
        }

        return { indexRoutes, idRoutes, distance: totalDistance };
    }

    function ensureCoverage(routes: number[][]): number[][] {
        const seen = new Array(totalPackages).fill(false);
        const sanitized: number[][] = [];

        for (const route of routes) {
            const filtered: number[] = [];
            for (const idx of route) {
                if (idx < 0 || idx >= totalPackages) continue;
                if (seen[idx]) continue;
                seen[idx] = true;
                filtered.push(idx);
            }
            if (filtered.length > 0) {
                sanitized.push(filtered);
            }
        }

        const missing: number[] = [];
        for (let i = 0; i < totalPackages; i++) {
            if (!seen[i]) {
                missing.push(i);
            }
        }

        if (missing.length > 0) {
            if (sanitized.length === 0) {
                sanitized.push([]);
            }
            let pointer = 0;
            for (const idx of missing) {
                sanitized[pointer].push(idx);
                pointer++;
                if (pointer >= sanitized.length) {
                    pointer = 0;
                }
            }
        }

        return sanitized;
    }

    function orderCluster(indices: number[]): number[] {
        const NEAREST_NEIGHBOR_LIMIT = 1500;
        if (indices.length === 0) return [];
        if (indices.length <= NEAREST_NEIGHBOR_LIMIT) {
            return orderByNearestNeighbor(indices);
        }
        return orderByAngle(indices);
    }

    function orderByNearestNeighbor(indices: number[]): number[] {
        const remaining = indices.slice();
        const order: number[] = [];
        let currentX = depotX;
        let currentY = depotY;

        while (remaining.length > 0) {
            let bestPos = 0;
            let bestDist = Infinity;
            for (let i = 0; i < remaining.length; i++) {
                const idx = remaining[i];
                const pkg = packages[idx];
                const dist = distanceSqCoords(currentX, currentY, pkg.x, pkg.y);
                if (dist < bestDist) {
                    bestDist = dist;
                    bestPos = i;
                }
            }
            const chosenIndex = remaining[bestPos];
            const lastPos = remaining.length - 1;
            const lastValue = remaining[lastPos];
            remaining[bestPos] = lastValue;
            remaining.pop();
            order.push(chosenIndex);
            currentX = packages[chosenIndex].x;
            currentY = packages[chosenIndex].y;
        }

        return order;
    }

    function orderByAngle(indices: number[]): number[] {
        const arr = indices.slice();
        arr.sort((a, b) => {
            const diff = packages[a].angle - packages[b].angle;
            if (diff !== 0) return diff;
            const distDiff = packages[a].dist2 - packages[b].dist2;
            if (distDiff !== 0) return distDiff;
            return a - b;
        });

        if (arr.length === 0) return arr;

        let bestStart = 0;
        let bestDist = Infinity;
        for (let i = 0; i < arr.length; i++) {
            const dist = packages[arr[i]].dist2;
            if (dist < bestDist) {
                bestDist = dist;
                bestStart = i;
            }
        }

        if (bestStart === 0) {
            return arr;
        }

        return arr.slice(bestStart).concat(arr.slice(0, bestStart));
    }

    function improveRoute(order: number[]): void {
        const n = order.length;
        if (n < 4) return;
        if (n > 1200) return;

        const extended: number[] = new Array(n + 2);
        extended[0] = -1;
        for (let i = 0; i < n; i++) {
            extended[i + 1] = order[i];
        }
        extended[n + 1] = -1;

        const maxSwaps = Math.min(10000, Math.max(200, n * 5));
        const iterations = n > 400 ? 6 : 10;
        let swaps = 0;

        for (let iter = 0; iter < iterations; iter++) {
            let improved = false;
            for (let i = 0; i < extended.length - 3; i++) {
                for (let j = i + 2; j < extended.length - 1; j++) {
                    if (i === 0 && j === extended.length - 2) continue;
                    const a = extended[i];
                    const b = extended[i + 1];
                    const c = extended[j];
                    const d = extended[j + 1];
                    const current = distanceBetween(a, b) + distanceBetween(c, d);
                    const proposed = distanceBetween(a, c) + distanceBetween(b, d);
                    if (proposed + 1e-9 < current) {
                        reverseSection(extended, i + 1, j);
                        swaps++;
                        improved = true;
                        if (swaps >= maxSwaps) break;
                    }
                }
                if (swaps >= maxSwaps) break;
            }
            if (!improved || swaps >= maxSwaps) {
                break;
            }
        }

        for (let i = 1; i < extended.length - 1; i++) {
            order[i - 1] = extended[i];
        }
    }

    function computeRouteDistance(route: number[]): number {
        if (route.length === 0) return 0;
        let total = distanceBetween(-1, route[0]);
        for (let i = 0; i < route.length - 1; i++) {
            total += distanceBetween(route[i], route[i + 1]);
        }
        total += distanceBetween(route[route.length - 1], -1);
        return total;
    }

    function distanceBetween(aIndex: number, bIndex: number): number {
        const ax = aIndex === -1 ? depotX : packages[aIndex].x;
        const ay = aIndex === -1 ? depotY : packages[aIndex].y;
        const bx = bIndex === -1 ? depotX : packages[bIndex].x;
        const by = bIndex === -1 ? depotY : packages[bIndex].y;
        const dx = ax - bx;
        const dy = ay - by;
        return Math.sqrt(dx * dx + dy * dy);
    }

    function distanceSqCoords(ax: number, ay: number, bx: number, by: number): number {
        const dx = ax - bx;
        const dy = ay - by;
        return dx * dx + dy * dy;
    }

    function reverseSection(arr: number[], start: number, end: number): void {
        while (start < end) {
            const temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }
}

function toFiniteNumber(value: any, fallback: number = 0): number {
    const num = Number(value);
    return Number.isFinite(num) ? num : fallback;
}
