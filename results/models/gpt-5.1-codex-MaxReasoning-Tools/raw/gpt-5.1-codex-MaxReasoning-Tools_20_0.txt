fn detect_outliers<'a>(
    samples: &[SampleMeta],
    holes: &[HoleInfo],
    prop_values: &[Vec<f32>],
    global_stats: &[RunningStat],
    hole_prop_stats: &[RunningStat],
    scale: Scale,
    spatial: &SpatialStrategy<'a>,
) -> Vec<Suspect> {
    let num_props = prop_values.len();
    if samples.is_empty() || num_props == 0 {
        return Vec::new();
    }

    let threshold = scale.threshold();
    let local_window = scale.local_window();
    let local_min_needed = scale.local_min();
    let hole_min_needed = scale.hole_min();
    let global_min_needed = scale.global_min();
    let max_suspects = scale.max_suspects(samples.len(), num_props);
    let mut heap: BinaryHeap<Reverse<Suspect>> = BinaryHeap::new();

    for (sample_idx, sample) in samples.iter().enumerate() {
        let hole_idx = sample.hole_idx;
        if hole_idx >= holes.len() {
            continue;
        }
        let hole = &holes[hole_idx];
        let hole_start = hole.start_index;
        let hole_count = hole.sample_count;

        for prop_idx in 0..num_props {
            let raw_value = prop_values[prop_idx][sample_idx];
            if !raw_value.is_finite() {
                continue;
            }
            let value = raw_value as f64;

            let mut local_stat = RunningStat::new();
            let mut prev_val = None;
            let mut next_val = None;
            for step in 1..=local_window {
                if sample.sample_idx_in_hole >= step {
                    let neighbor_idx = hole_start + sample.sample_idx_in_hole - step;
                    if neighbor_idx < prop_values[prop_idx].len() {
                        let neighbor_val = prop_values[prop_idx][neighbor_idx];
                        if neighbor_val.is_finite() {
                            let v = neighbor_val as f64;
                            local_stat.push(v);
                            if prev_val.is_none() {
                                prev_val = Some(v);
                            }
                        }
                    }
                }
                if sample.sample_idx_in_hole + step < hole_count {
                    let neighbor_idx = hole_start + sample.sample_idx_in_hole + step;
                    if neighbor_idx < prop_values[prop_idx].len() {
                        let neighbor_val = prop_values[prop_idx][neighbor_idx];
                        if neighbor_val.is_finite() {
                            let v = neighbor_val as f64;
                            local_stat.push(v);
                            if next_val.is_none() {
                                next_val = Some(v);
                            }
                        }
                    }
                }
            }

            spatial.accumulate(
                sample_idx,
                prop_idx,
                value,
                &mut local_stat,
                samples,
                prop_values,
            );

            let mut ref_mean = None;
            let mut ref_std = None;
            let mut reliability = 1.0;

            if local_stat.count >= local_min_needed {
                ref_mean = Some(local_stat.mean);
                ref_std = local_stat.stddev().or(Some(MIN_STD_DEV));
            }

            let mut hole_stat = if let Some(idx) = hole_prop_index(hole_idx, prop_idx, num_props) {
                hole_prop_stats.get(idx).copied().unwrap_or_default()
            } else {
                RunningStat::new()
            };
            if hole_stat.count > 0 {
                hole_stat = hole_stat.without(value);
            }
            if ref_mean.is_none() && hole_stat.count >= hole_min_needed {
                ref_mean = Some(hole_stat.mean);
                ref_std = hole_stat.stddev().or(Some(MIN_STD_DEV));
                reliability = 0.92;
            }

            let mut global_stat = global_stats.get(prop_idx).copied().unwrap_or_default();
            if global_stat.count > 0 {
                global_stat = global_stat.without(value);
            }
            if ref_mean.is_none() && global_stat.count >= global_min_needed {
                ref_mean = Some(global_stat.mean);
                ref_std = global_stat.stddev().or(Some(MIN_STD_DEV));
                reliability = 0.85;
            }

            let Some(mean) = ref_mean else {
                continue;
            };
            let std = ref_std.unwrap_or(MIN_STD_DEV).max(MIN_STD_DEV);
            if !std.is_finite() || std <= 0.0 {
                continue;
            }

            let diff = (value - mean).abs();
            let mut confidence = (diff / std).min(1e6) * reliability;
            if !confidence.is_finite() {
                continue;
            }

            let std_denom = std + MIN_STD_DEV;
            let mut gradient_component = 0.0;
            if let Some(prev) = prev_val {
                gradient_component = gradient_component.max((value - prev).abs() / std_denom);
            }
            if let Some(next) = next_val {
                gradient_component = gradient_component.max((value - next).abs() / std_denom);
            }
            if let (Some(prev), Some(next)) = (prev_val, next_val) {
                let interp = (prev + next) * 0.5;
                gradient_component = gradient_component.max((value - interp).abs() / std_denom);
            }
            gradient_component = gradient_component.min(12.0);
            if local_stat.count < local_min_needed {
                confidence += 0.35 * gradient_component;
                confidence *= 0.95;
            } else {
                confidence += 0.25 * gradient_component;
            }

            if confidence >= threshold {
                push_suspect(
                    &mut heap,
                    Suspect {
                        hole_idx,
                        sample_idx_in_hole: sample.sample_idx_in_hole,
                        property_idx: prop_idx,
                        confidence,
                    },
                    max_suspects,
                );
            }
        }
    }

    let mut suspects: Vec<Suspect> = heap.into_iter().map(|rev| rev.0).collect();
    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_idx.cmp(&b.hole_idx))
            .then_with(|| a.sample_idx_in_hole.cmp(&b.sample_idx_in_hole))
            .then_with(|| a.property_idx.cmp(&b.property_idx))
    });
    suspects
}

struct Scanner<'a> {
    data: &'a [u8],
    pos: usize,
    len: usize,
}

impl<'a> Scanner<'a> {
    fn new(input: &'a str) -> Self {
        Self {
            data: input.as_bytes(),
            pos: 0,
            len: input.len(),
        }
    }

    fn next_token(&mut self) -> Option<&'a str> {
        while self.pos < self.len && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.len {
            return None;
        }
        let start = self.pos;
        while self.pos < self.len && !self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let slice = &self.data[start..self.pos];
        Some(unsafe { std::str::from_utf8_unchecked(slice) })
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_token()?.parse().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_token()?.parse().ok()
    }
}

#[derive(Clone, Copy, Default)]
struct RunningStat {
    count: u64,
    mean: f64,
    m2: f64,
}

impl RunningStat {
    fn new() -> Self {
        Self {
            count: 0,
            mean: 0.0,
            m2: 0.0,
        }
    }

    fn push(&mut self, value: f64) {
        if !value.is_finite() {
            return;
        }
        self.count += 1;
        let delta = value - self.mean;
        self.mean += delta / self.count as f64;
        let delta2 = value - self.mean;
        self.m2 += delta * delta2;
    }

    fn absorb(&mut self, other: &RunningStat) {
        if other.count == 0 {
            return;
        }
        if self.count == 0 {
            self.count = other.count;
            self.mean = other.mean;
            self.m2 = other.m2;
            return;
        }
        let total = self.count + other.count;
        let delta = other.mean - self.mean;
        self.mean += delta * (other.count as f64 / total as f64);
        self.m2 += other.m2
            + delta * delta * (self.count as f64 * other.count as f64 / total as f64);
        self.count = total;
    }

    fn without(&self, value: f64) -> RunningStat {
        if self.count == 0 {
            return *self;
        }
        if self.count == 1 {
            return RunningStat::new();
        }
        let count = self.count - 1;
        let total = self.mean * self.count as f64 - value;
        let mean = total / count as f64;
        let delta1 = value - self.mean;
        let delta2 = value - mean;
        let mut m2 = self.m2 - delta1 * delta2;
        if m2 < 0.0 {
            m2 = 0.0;
        }
        RunningStat { count, mean, m2 }
    }

    fn stddev(&self) -> Option<f64> {
        if self.count < 2 {
            None
        } else {
            Some((self.m2.max(0.0) / (self.count - 1) as f64).sqrt())
        }
    }
}

#[derive(Hash, Eq, PartialEq, Clone, Copy)]
struct GridKey {
    x: i64,
    y: i64,
    z: i64,
}

struct CellAgg {
    stats: Vec<RunningStat>,
}

impl CellAgg {
    fn new(num_props: usize) -> Self {
        Self {
            stats: vec![RunningStat::new(); num_props],
        }
    }
}

enum SpatialStrategy<'a> {
    None,
    Small {
        grid: &'a HashMap<GridKey, Vec<usize>>,
        origin: [f64; 3],
        inv_cell: f64,
        search_radius2: f64,
        max_neighbors: usize,
    },
    Medium {
        grid: &'a HashMap<GridKey, CellAgg>,
        origin: [f64; 3],
        inv_cell: f64,
    },
}

impl<'a> SpatialStrategy<'a> {
    fn accumulate(
        &self,
        sample_idx: usize,
        prop_idx: usize,
        sample_value: f64,
        stat: &mut RunningStat,
        samples: &[SampleMeta],
        prop_values: &[Vec<f32>],
    ) {
        match self {
            SpatialStrategy::None => {}
            SpatialStrategy::Small {
                grid,
                origin,
                inv_cell,
                search_radius2,
                max_neighbors,
            } => {
                if prop_idx >= prop_values.len() {
                    return;
                }
                let sample = match samples.get(sample_idx) {
                    Some(s) => s,
                    None => return,
                };
                if !is_valid_position(&sample.pos) {
                    return;
                }
                let base_key = cell_key(&sample.pos, origin, *inv_cell);
                let mut added = 0usize;
                for dx in -1..=1 {
                    for dy in -1..=1 {
                        for dz in -1..=1 {
                            let neighbor_key = GridKey {
                                x: base_key.x + dx,
                                y: base_key.y + dy,
                                z: base_key.z + dz,
                            };
                            if let Some(indices) = grid.get(&neighbor_key) {
                                for &neighbor_idx in indices {
                                    if neighbor_idx == sample_idx {
                                        continue;
                                    }
                                    if neighbor_idx >= samples.len() {
                                        continue;
                                    }
                                    let neighbor_sample = &samples[neighbor_idx];
                                    if !is_valid_position(&neighbor_sample.pos) {
                                        continue;
                                    }
                                    let neighbor_value =
                                        prop_values[prop_idx][neighbor_idx];
                                    if !neighbor_value.is_finite() {
                                        continue;
                                    }
                                    let dist2 = squared_distance(
                                        &sample.pos,
                                        &neighbor_sample.pos,
                                    );
                                    if dist2 > *search_radius2 {
                                        continue;
                                    }
                                    stat.push(neighbor_value as f64);
                                    added += 1;
                                    if added >= *max_neighbors {
                                        return;
                                    }
                                }
                            }
                        }
                    }
                }
            }
            SpatialStrategy::Medium {
                grid,
                origin,
                inv_cell,
            } => {
                if prop_idx >= prop_values.len() {
                    return;
                }
                let sample = match samples.get(sample_idx) {
                    Some(s) => s,
                    None => return,
                };
                if !is_valid_position(&sample.pos) {
                    return;
                }
                let base_key = cell_key(&sample.pos, origin, *inv_cell);
                for dx in -1..=1 {
                    for dy in -1..=1 {
                        for dz in -1..=1 {
                            let neighbor_key = GridKey {
                                x: base_key.x + dx,
                                y: base_key.y + dy,
                                z: base_key.z + dz,
                            };
                            if let Some(cell) = grid.get(&neighbor_key) {
                                if prop_idx >= cell.stats.len() {
                                    continue;
                                }
                                let mut cell_stat = cell.stats[prop_idx];
                                if neighbor_key == base_key
                                    && cell_stat.count > 0
                                    && sample_value.is_finite()
                                {
                                    cell_stat = cell_stat.without(sample_value);
                                }
                                if cell_stat.count > 0 {
                                    stat.absorb(&cell_stat);
                                }
                            }
                        }
                    }
                }
            }
        }
    }
}

#[derive(Clone)]
struct Suspect {
    hole_idx: usize,
    sample_idx_in_hole: usize,
    property_idx: usize,
    confidence: f64,
}

impl PartialEq for Suspect {
    fn eq(&self, other: &Self) -> bool {
        self.confidence == other.confidence
    }
}
impl Eq for Suspect {}
impl PartialOrd for Suspect {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        self.confidence.partial_cmp(&other.confidence)
    }
}
impl Ord for Suspect {
    fn cmp(&self, other: &Self) -> Ordering {
        self.partial_cmp(other).unwrap_or(Ordering::Equal)
    }
}

#[derive(Clone, Copy)]
enum Scale {
    Small,
    Medium,
    Large,
}

impl Scale {
    fn threshold(self) -> f64 {
        match self {
            Scale::Small => 3.25,
            Scale::Medium => 3.75,
            Scale::Large => 4.25,
        }
    }
    fn local_window(self) -> usize {
        match self {
            Scale::Small => 2,
            Scale::Medium => 2,
            Scale::Large => 1,
        }
    }
    fn local_min(self) -> u64 {
        match self {
            Scale::Small => 4,
            Scale::Medium => 5,
            Scale::Large => 2,
        }
    }
    fn hole_min(self) -> u64 {
        match self {
            Scale::Small => 4,
            Scale::Medium => 5,
            Scale::Large => 5,
        }
    }
    fn global_min(self) -> u64 {
        match self {
            Scale::Small => 12,
            Scale::Medium => 15,
            Scale::Large => 20,
        }
    }
    fn max_suspects(self, total_samples: usize, num_props: usize) -> usize {
        let possible = total_samples.saturating_mul(num_props);
        if possible == 0 {
            0
        } else {
            let base = match self {
                Scale::Small => 4_000,
                Scale::Medium => 8_000,
                Scale::Large => 12_000,
            };
            base.min(possible).max(512.min(possible))
        }
    }
}

fn parse_optional_f64(token: &str) -> Option<f64> {
    if token.is_empty() {
        return None;
    }
    if token.eq_ignore_ascii_case("na")
        || token.eq_ignore_ascii_case("nan")
        || token.eq_ignore_ascii_case("null")
        || token.eq_ignore_ascii_case("missing")
        || token == "-"
        || token == "*"
    {
        return None;
    }
    token.parse().ok()
}

fn is_valid_position(pos: &[f32; 3]) -> bool {
    pos.iter().all(|v| v.is_finite())
}

fn squared_distance(a: &[f32; 3], b: &[f32; 3]) -> f64 {
    let dx = (a[0] - b[0]) as f64;
    let dy = (a[1] - b[1]) as f64;
    let dz = (a[2] - b[2]) as f64;
    dx * dx + dy * dy + dz * dz
}

fn cell_key(pos: &[f32; 3], origin: &[f64; 3], inv_cell: f64) -> GridKey {
    GridKey {
        x: ((pos[0] as f64 - origin[0]) * inv_cell).floor() as i64,
        y: ((pos[1] as f64 - origin[1]) * inv_cell).floor() as i64,
        z: ((pos[2] as f64 - origin[2]) * inv_cell).floor() as i64,
    }
}

fn build_point_grid(
    samples: &[SampleMeta],
    origin: [f64; 3],
    inv_cell: f64,
) -> HashMap<GridKey, Vec<usize>> {
    let mut grid: HashMap<GridKey, Vec<usize>> = HashMap::new();
    for (idx, sample) in samples.iter().enumerate() {
        if !is_valid_position(&sample.pos) {
            continue;
        }
        let key = cell_key(&sample.pos, &origin, inv_cell);
        grid.entry(key).or_insert_with(Vec::new).push(idx);
    }
    grid
}

fn build_cell_aggs(
    samples: &[SampleMeta],
    prop_values: &[Vec<f32>],
    origin: [f64; 3],
    inv_cell: f64,
) -> HashMap<GridKey, CellAgg> {
    let num_props = prop_values.len();
    let mut grid: HashMap<GridKey, CellAgg> = HashMap::new();
    for (idx, sample) in samples.iter().enumerate() {
        if !is_valid_position(&sample.pos) {
            continue;
        }
        let key = cell_key(&sample.pos, &origin, inv_cell);
        let cell = grid.entry(key).or_insert_with(|| CellAgg::new(num_props));
        for (prop_idx, column) in prop_values.iter().enumerate() {
            let value = column[idx];
            if value.is_finite() {
                cell.stats[prop_idx].push(value as f64);
            }
        }
    }
    grid
}

fn push_suspect(
    heap: &mut BinaryHeap<Reverse<Suspect>>,
    suspect: Suspect,
    max_keep: usize,
) {
    if max_keep == 0 {
        return;
    }
    if heap.len() < max_keep {
        heap.push(Reverse(suspect));
    } else if let Some(current_min) = heap.peek() {
        if suspect.confidence > current_min.0.confidence {
            heap.pop();
            heap.push(Reverse(suspect));
        }
    }
}

fn hole_prop_index(hole_idx: usize, prop_idx: usize, num_props: usize) -> Option<usize> {
    hole_idx.checked_mul(num_props)?.checked_add(prop_idx)
}
