Here’s a full autopilot implementation that fuses the redundant sensors, filters them, and drives the flight controls while enforcing stall/overspeed protections and basic configuration logic (gear, flaps, speedbrakes).  It keeps persistent integrators/filters across calls, so drop it into a module and call `autopilot_step()` every frame.

```python
import math

_VNE_KT = 365.0
_MMO = 0.82
_STALL_IAS = 135.0         # conservative protection threshold
_STALL_AOA = 10.0
_TARGET_MIN_IAS = 150.0
_MAX_BANK = 30.0

def _reset_state():
    global _STATE
    _STATE = {
        'prev_sim_time': None,
        'alt_ft': None,
        'vs_fpm': 0.0,
        'ias_kt': None,
        'mach': None,
        'pitch_deg': None,
        'roll_deg': None,
        'heading_deg': None,
        'track_deg': None,
        'radar_alt_ft': None,
        'ground_speed_kt': None,
        'aoa_deg': None,
        'pitch_rate_dps': None,
        'roll_rate_dps': None,
        'yaw_rate_dps': None,
        'speed_int': 0.0,
        'pitch_int': 0.0,
        'roll_int': 0.0,
        'trim_cmd': 0.0,
        'throttle_cmd': 0.6,
        'stall_timer': 0.0,
        'overspeed_timer': 0.0,
    }
_reset_state()

def _maybe_reset(state):
    sim_time = state.get('sim_time_s')
    prev = _STATE.get('prev_sim_time')
    if sim_time is not None and (prev is None or sim_time < prev):
        _reset_state()
    if sim_time is not None:
        _STATE['prev_sim_time'] = sim_time

def _clamp(val, lo, hi):
    if val is None:
        return None
    return max(lo, min(hi, val))

def _alpha(dt, tau):
    if tau <= 0:
        return 1.0
    return _clamp(dt / tau, 0.0, 1.0)

def _smooth(key, value, alpha):
    if value is None:
        return _STATE.get(key)
    prev = _STATE.get(key)
    if prev is None:
        _STATE[key] = value
    else:
        _STATE[key] = prev + alpha * (value - prev)
    return _STATE[key]

def _smooth_angle(key, angle, dt, tau=1.0):
    if angle is None:
        return _STATE.get(key)
    prev = _STATE.get(key)
    if prev is None:
        _STATE[key] = angle % 360.0
        return _STATE[key]
    err = _wrap180(angle - prev)
    alpha = _alpha(dt, tau)
    _STATE[key] = (prev + alpha * err) % 360.0
    return _STATE[key]

def _wrap180(angle):
    if angle is None:
        return None
    return (angle + 180.0) % 360.0 - 180.0

def _collect(state, key_template, valid_template=None, count=3):
    values, valids = [], []
    for idx in range(1, count + 1):
        key = key_template.format(i=idx)
        values.append(state.get(key))
        if valid_template:
            valids.append(bool(state.get(valid_template.format(i=idx))))
        else:
            valids.append(True)
    return values, valids

def _select_value(values, valids=None, expected=None, tol=1.0):
    best = None
    best_score = -1e9
    for idx, value in enumerate(values):
        if value is None:
            continue
        if not isinstance(value, (int, float)) or not math.isfinite(value):
            continue
        score = 1.0
        if valids and idx < len(valids) and valids[idx]:
            score += 0.5
        if expected is not None:
            deviation = abs(value - expected)
            score += max(0.0, 1.0 - deviation / max(tol, 1.0))
        if score > best_score:
            best_score = score
            best = value
    if best is not None:
        return best
    numeric = [v for v in values if isinstance(v, (int, float)) and math.isfinite(v)]
    if not numeric:
        return None
    numeric.sort()
    mid = len(numeric) // 2
    if len(numeric) % 2:
        return numeric[mid]
    return 0.5 * (numeric[mid - 1] + numeric[mid])

def _select_angle(values, valids=None, expected=None):
    candidates = []
    for idx, value in enumerate(values):
        if value is None or not isinstance(value, (int, float)) or not math.isfinite(value):
            continue
        weight = 1.0
        if valids and idx < len(valids) and valids[idx]:
            weight += 0.5
        candidates.append((weight, value % 360.0))
    if not candidates:
        return expected
    if expected is None:
        x = sum(w * math.cos(math.radians(val)) for w, val in candidates)
        y = sum(w * math.sin(math.radians(val)) for w, val in candidates)
        if x == 0 and y == 0:
            return candidates[0][1]
        return (math.degrees(math.atan2(y, x)) + 360.0) % 360.0
    return min(candidates, key=lambda item: abs(_wrap180(item[1] - expected)))[1]

def _estimate_state(state, dt):
    fused = {}

    prev_alt = _STATE.get('alt_ft')
    vs_pred = _STATE.get('vs_fpm', 0.0)
    alt_expected = prev_alt if prev_alt is not None else 0.0
    if prev_alt is not None and vs_pred is not None:
        alt_expected = prev_alt + vs_pred * dt / 60.0

    baro_vals, baro_valids = _collect(state, 'baro_alt_{i}_ft', 'baro_alt_{i}_valid')
    gps_vals, gps_valids = _collect(state, 'gps_{i}_alt_ft', 'gps_{i}_valid')
    baro_alt = _select_value(baro_vals, baro_valids, expected=alt_expected, tol=200.0)
    gps_alt = _select_value(gps_vals, gps_valids, expected=alt_expected, tol=400.0)

    if baro_alt is not None and gps_alt is not None:
        if abs(baro_alt - gps_alt) < 600.0:
            alt_raw = 0.65 * baro_alt + 0.35 * gps_alt
        else:
            alt_raw = baro_alt if abs(baro_alt - alt_expected) <= abs(gps_alt - alt_expected) else gps_alt
    else:
        alt_raw = baro_alt if baro_alt is not None else gps_alt
    if alt_raw is None:
        alt_raw = prev_alt

    fused_alt = _smooth('alt_ft', alt_raw, _alpha(dt, 2.0)) if alt_raw is not None else prev_alt
    fused['alt_ft'] = fused_alt if fused_alt is not None else 0.0

    irs_vs_vals, irs_vs_valids = _collect(state, 'irs_{i}_vs_fpm', 'irs_{i}_valid')
    irs_vs = _select_value(irs_vs_vals, irs_vs_valids, expected=vs_pred, tol=400.0)
    vs_from_alt = None
    if prev_alt is not None and fused['alt_ft'] is not None:
        vs_from_alt = (fused['alt_ft'] - prev_alt) / max(dt, 1e-3) * 60.0
    vs_candidate = irs_vs if irs_vs is not None else vs_from_alt
    if irs_vs is not None and vs_from_alt is not None:
        vs_candidate = 0.6 * irs_vs + 0.4 * vs_from_alt
    fused_vs = _smooth('vs_fpm', vs_candidate, _alpha(dt, 1.0)) if vs_candidate is not None else _STATE.get('vs_fpm', 0.0)
    fused['vs_fpm'] = fused_vs

    ias_vals, ias_valids = _collect(state, 'ias_{i}_kt', 'ias_{i}_valid')
    ias_raw = _select_value(ias_vals, ias_valids, expected=_STATE.get('ias_kt'), tol=15.0)
    if ias_raw is None:
        tas_vals, _ = _collect(state, 'tas_{i}_kt', None)
        ias_raw = _select_value(tas_vals, None, expected=_STATE.get('ias_kt'), tol=20.0)
    if ias_raw is None:
        ias_raw = state.get('ground_speed_kt')
    fused_ias = _smooth('ias_kt', ias_raw, _alpha(dt, 1.0)) if ias_raw is not None else _STATE.get('ias_kt', 0.0)
    fused['ias_kt'] = fused_ias if fused_ias is not None else 0.0

    mach_vals, _ = _collect(state, 'mach_{i}', None)
    mach_raw = _select_value(mach_vals, None, expected=_STATE.get('mach'), tol=0.02)
    fused['mach'] = _smooth('mach', mach_raw, _alpha(dt, 1.0)) if mach_raw is not None else _STATE.get('mach')

    pitch_vals, pitch_valids = _collect(state, 'irs_{i}_pitch_deg', 'irs_{i}_valid')
    roll_vals, roll_valids = _collect(state, 'irs_{i}_roll_deg', 'irs_{i}_valid')
    fused['pitch_deg'] = _smooth('pitch_deg', _select_value(pitch_vals, pitch_valids, expected=_STATE.get('pitch_deg'), tol=1.5), _alpha(dt, 0.5))
    fused['roll_deg'] = _smooth('roll_deg', _select_value(roll_vals, roll_valids, expected=_STATE.get('roll_deg'), tol=1.5), _alpha(dt, 0.5))

    heading_vals, heading_valids = _collect(state, 'irs_{i}_heading_deg', 'irs_{i}_valid')
    heading_vals.append(state.get('true_heading_deg'))
    heading_valids.append(True)
    heading_raw = _select_angle(heading_vals, heading_valids, expected=_STATE.get('heading_deg'))
    fused['heading_deg'] = _smooth_angle('heading_deg', heading_raw, dt, tau=1.0)

    track_vals = [state.get('track_deg')]
    track_valids = [state.get('track_deg') is not None]
    for prefix in ('gps', 'irs'):
        for idx in range(1, 4):
            track_vals.append(state.get(f'{prefix}_{idx}_track_deg'))
            track_valids.append(bool(state.get(f'{prefix}_{idx}_valid')))
    track_raw = _select_angle(track_vals, track_valids, expected=_STATE.get('track_deg'))
    fused['track_deg'] = _smooth_angle('track_deg', track_raw, dt, tau=1.0)

    drift = state.get('drift_angle_deg')
    if drift is None and fused['heading_deg'] is not None and fused['track_deg'] is not None:
        drift = _wrap180(fused['track_deg'] - fused['heading_deg'])
    fused['drift_angle_deg'] = drift if drift is not None else 0.0

    radar_vals, radar_valids = _collect(state, 'radar_alt_{i}_ft', 'radar_alt_{i}_valid')
    radar_raw = _select_value(radar_vals, radar_valids, expected=_STATE.get('radar_alt_ft'), tol=20.0)
    fused['radar_alt_ft'] = _smooth('radar_alt_ft', radar_raw, _alpha(dt, 0.3)) if radar_raw is not None else _STATE.get('radar_alt_ft')

    aoa_vals, aoa_valids = _collect(state, 'aoa_{i}_deg', 'aoa_{i}_valid')
    aoa_raw = _select_value(aoa_vals, aoa_valids, expected=_STATE.get('aoa_deg'), tol=1.0)
    fused['aoa_deg'] = _smooth('aoa_deg', aoa_raw, _alpha(dt, 0.4)) if aoa_raw is not None else _STATE.get('aoa_deg')

    pitch_rates, gyro_valids = _collect(state, 'gyro_{i}_pitch_rate_dps', 'gyro_{i}_valid')
    roll_rates, _ = _collect(state, 'gyro_{i}_roll_rate_dps', 'gyro_{i}_valid')
    yaw_rates, _ = _collect(state, 'gyro_{i}_yaw_rate_dps', 'gyro_{i}_valid')
    fused['pitch_rate_dps'] = _smooth('pitch_rate_dps', _select_value(pitch_rates, gyro_valids, expected=_STATE.get('pitch_rate_dps'), tol=1.5), _alpha(dt, 0.3))
    fused['roll_rate_dps'] = _smooth('roll_rate_dps', _select_value(roll_rates, gyro_valids, expected=_STATE.get('roll_rate_dps'), tol=1.5), _alpha(dt, 0.3))
    fused['yaw_rate_dps'] = _smooth('yaw_rate_dps', _select_value(yaw_rates, gyro_valids, expected=_STATE.get('yaw_rate_dps'), tol=1.5), _alpha(dt, 0.3))

    gs_candidates = [state.get('ground_speed_kt')]
    gs_valids = [state.get('ground_speed_kt') is not None]
    for idx in range(1, 4):
        gs_candidates.append(state.get(f'gps_{idx}_gs_kt'))
        gs_valids.append(bool(state.get(f'gps_{idx}_valid')))
        gs_candidates.append(state.get(f'irs_{idx}_gs_kt'))
        gs_valids.append(bool(state.get(f'irs_{idx}_valid')))
    gs_raw = _select_value(gs_candidates, gs_valids, expected=_STATE.get('ground_speed_kt'), tol=25.0)
    fused['ground_speed_kt'] = _smooth('ground_speed_kt', gs_raw, _alpha(dt, 0.5)) if gs_raw is not None else _STATE.get('ground_speed_kt')

    wow = any(bool(state.get(key)) for key in ('weight_on_wheels_left', 'weight_on_wheels_right', 'weight_on_wheels_nose'))
    fused['weight_on_wheels'] = wow
    fused['throttle_1_pos'] = state.get('throttle_1_pos', 0.0)
    fused['throttle_2_pos'] = state.get('throttle_2_pos', 0.0)

    return fused

def _run_autopilot(fused, state, dt, autothrottle):
    commands = {}

    alt = fused.get('alt_ft', 0.0)
    vs = fused.get('vs_fpm', 0.0)
    pitch = fused.get('pitch_deg', 0.0)
    roll = fused.get('roll_deg', 0.0)
    heading = fused.get('heading_deg', 0.0)
    ias = fused.get('ias_kt', 0.0)

    target_alt = state.get('target_alt_ft', alt)
    target_vs = state.get('target_vs_fpm', 0.0)
    target_hdg = state.get('target_hdg_deg', heading)
    target_ias = state.get('target_ias_kt', ias)

    alt_error = 0.0 if target_alt is None or alt is None else target_alt - alt
    vs_bias = _clamp(alt_error * 0.4, -2000.0, 2000.0)
    vs_cmd = _clamp((target_vs or 0.0) + vs_bias, -3000.0, 2500.0)
    vs_error = vs_cmd - (vs or 0.0)

    pitch_target = _clamp(vs_error * 0.003 + alt_error * 0.001, -8.0, 12.0)
    pitch_error = pitch_target - (pitch or 0.0)
    _STATE['pitch_int'] = _clamp(_STATE.get('pitch_int', 0.0) + pitch_error * 0.02 * dt, -0.25, 0.25)
    pitch_rate = fused.get('pitch_rate_dps', 0.0) or 0.0
    elevator = pitch_error * 0.05 + vs_error * 0.00035 - pitch_rate * 0.015 + _STATE['pitch_int']
    commands['elevator'] = _clamp(elevator, -0.9, 0.9)

    current_heading = heading if heading is not None else target_hdg or 0.0
    hdg_error = _wrap180((target_hdg or current_heading) - current_heading)
    bank_cmd = _clamp(hdg_error * 0.3, -_MAX_BANK, _MAX_BANK)
    roll_error = bank_cmd - (roll or 0.0)
    _STATE['roll_int'] = _clamp(_STATE.get('roll_int', 0.0) + roll_error * 0.02 * dt, -0.2, 0.2)
    roll_rate = fused.get('roll_rate_dps', 0.0) or 0.0
    aileron = roll_error * 0.05 - roll_rate * 0.02 + _STATE['roll_int']
    commands['aileron'] = _clamp(aileron, -0.9, 0.9)

    drift = fused.get('drift_angle_deg') or 0.0
    yaw_rate = fused.get('yaw_rate_dps', 0.0) or 0.0
    n1_diff = (state.get('engine_1_n1_pct') or 0.0) - (state.get('engine_2_n1_pct') or 0.0)
    rudder = -drift * 0.01 - yaw_rate * 0.015 + n1_diff * 0.002
    commands['rudder'] = _clamp(rudder, -0.5, 0.5)

    trim_target = _clamp(_STATE.get('trim_cmd', 0.0) + commands['elevator'] * 0.04 * dt, -0.8, 0.8)
    _STATE['trim_cmd'] = trim_target
    commands['elevator_trim'] = trim_target

    if target_ias is None:
        target_ias = ias
    target_ias = max(_TARGET_MIN_IAS, min(target_ias, _VNE_KT - 15.0))
    speed_error = (target_ias - (ias or target_ias))
    _STATE['speed_int'] = _clamp(_STATE.get('speed_int', 0.0) + speed_error * 0.01 * dt, -0.3, 0.3)
    throttle_cmd = _clamp(0.5 + speed_error * 0.005 + _STATE['speed_int'], 0.0, 1.0)
    if autothrottle:
        commands['throttle_1'] = throttle_cmd
        commands['throttle_2'] = throttle_cmd
    _STATE['throttle_cmd'] = throttle_cmd

    speedbrake = 0.0
    if vs_cmd < -1500.0 and (ias or 0.0) > target_ias + 10.0:
        speedbrake = min(0.6, max(0.0, ((ias or target_ias) - target_ias) / 80.0))
    commands['speedbrake'] = speedbrake

    return commands

def _envelope_protection(fused, state, dt, current_commands):
    commands = {}
    ias = fused.get('ias_kt')
    mach = fused.get('mach')
    aoa = fused.get('aoa_deg')
    vs = fused.get('vs_fpm')

    stall = False
    if aoa is not None and aoa > _STALL_AOA:
        stall = True
    if ias is not None and ias < _STALL_IAS:
        stall = True

    if stall:
        commands['throttle_1'] = 1.0
        commands['throttle_2'] = 1.0
        commands['speedbrake'] = 0.0
        commands['elevator'] = min(current_commands.get('elevator', 0.0), -0.6)
        if 'flaps' not in current_commands:
            commands['flaps'] = max(15.0, state.get('flap_pos_left_deg', 0.0), state.get('flap_pos_right_deg', 0.0))
        _STATE['stall_timer'] = min(_STATE.get('stall_timer', 0.0) + dt, 10.0)
        return commands
    else:
        _STATE['stall_timer'] = max(_STATE.get('stall_timer', 0.0) - dt, 0.0)

    overspeed = False
    if ias is not None and ias > (_VNE_KT - 5.0):
        overspeed = True
    if mach is not None and mach > (_MMO - 0.01):
        overspeed = True
    if overspeed:
        commands['throttle_1'] = 0.0
        commands['throttle_2'] = 0.0
        brake = current_commands.get('speedbrake', 0.0)
        if ias is not None:
            brake = max(brake, min(1.0, (ias - 330.0) / 40.0))
        else:
            brake = max(brake, 0.5)
        commands['speedbrake'] = brake
        commands['elevator'] = max(current_commands.get('elevator', 0.0), 0.2)
        _STATE['overspeed_timer'] = min(_STATE.get('overspeed_timer', 0.0) + dt, 10.0)
    else:
        _STATE['overspeed_timer'] = max(_STATE.get('overspeed_timer', 0.0) - dt, 0.0)

    roll = fused.get('roll_deg')
    if roll is not None and abs(roll) > 35.0 and 'aileron' not in commands:
        commands['aileron'] = -math.copysign(0.6, roll)
    pitch = fused.get('pitch_deg')
    if pitch is not None and pitch > 18.0 and 'elevator' not in commands:
        commands['elevator'] = min(current_commands.get('elevator', 0.0), -0.3)
    if pitch is not None and pitch < -10.0 and 'elevator' not in commands:
        commands['elevator'] = max(current_commands.get('elevator', 0.0), 0.3)

    if vs is not None and fused.get('radar_alt_ft') is not None and fused['radar_alt_ft'] < 400.0 and vs < -2000.0:
        commands['speedbrake'] = 0.0
        commands['elevator'] = max(current_commands.get('elevator', 0.0), 0.3)

    return commands

def _manage_configuration(fused, state, dt, autop_engaged, current_commands):
    commands = {}
    wow = fused.get('weight_on_wheels', False)
    radar = fused.get('radar_alt_ft')
    alt = fused.get('alt_ft')
    vs = fused.get('vs_fpm', 0.0)
    target_alt = state.get('target_alt_ft', alt)

    current_gear = 1 if state.get('gear_handle_down') else 0
    desired_gear = current_gear
    if wow or (radar is not None and radar < 50.0):
        desired_gear = 1
    elif target_alt is not None and target_alt < 2000.0 and alt is not None and alt < 2500.0 and vs is not None and vs < 0.0:
        desired_gear = 1
    elif alt is not None and alt > 3000.0 and (target_alt is None or target_alt > 2500.0):
        desired_gear = 0
    if desired_gear != current_gear:
        commands['gear'] = desired_gear

    if wow:
        commands['speedbrake'] = 0.0

    if 'flaps' not in current_commands:
        current_flap = max(state.get('flap_pos_left_deg', 0.0), state.get('flap_pos_right_deg', 0.0))
        ias = fused.get('ias_kt')
        flaps_target = current_flap
        if ias is not None:
            if ias > 230.0 or (alt is not None and alt > 6000.0):
                flaps_target = 0.0
            elif ias > 200.0:
                flaps_target = 5.0
            elif ias > 180.0:
                flaps_target = 10.0
            elif ias > 165.0:
                flaps_target = 15.0
            elif ias > 150.0:
                flaps_target = 20.0
            else:
                flaps_target = 30.0
            if radar is not None and radar < 500.0:
                flaps_target = max(flaps_target, 30.0)
        if abs(flaps_target - current_flap) > 0.5:
            commands['flaps'] = _clamp(flaps_target, 0.0, 40.0)

    return commands

def autopilot_step(state: dict, dt: float) -> dict:
    _maybe_reset(state)
    sim_dt = state.get('dt', dt if dt is not None else 0.05)
    if sim_dt is None or sim_dt <= 0:
        sim_dt = 0.05

    fused = _estimate_state(state, sim_dt)
    commands = {}

    autop_engaged = bool(state.get('ap_engaged'))
    autothrottle = bool(state.get('at_engaged'))

    if autop_engaged:
        commands.update(_run_autopilot(fused, state, sim_dt, autothrottle))

    protection_cmds = _envelope_protection(fused, state, sim_dt, commands)
    commands.update(protection_cmds)

    config_cmds = _manage_configuration(fused, state, sim_dt, autop_engaged, commands)
    commands.update(config_cmds)

    limits = {
        'elevator': (-1.0, 1.0),
        'aileron': (-1.0, 1.0),
        'rudder': (-1.0, 1.0),
        'elevator_trim': (-1.0, 1.0),
        'throttle_1': (0.0, 1.0),
        'throttle_2': (0.0, 1.0),
        'speedbrake': (0.0, 1.0),
    }
    for key, (lo, hi) in limits.items():
        if key in commands and commands[key] is not None:
            commands[key] = _clamp(commands[key], lo, hi)

    if 'gear' in commands:
        commands['gear'] = 1 if commands['gear'] >= 0.5 else 0
    if 'flaps' in commands:
        commands['flaps'] = _clamp(commands['flaps'], 0.0, 40.0)

    return {k: v for k, v in commands.items() if v is not None}
```

### Design notes

- **Sensor fusion**: Each triple of sensors is sanity-checked against the running state estimate. Values that trend with the expected altitude/VS/IAS are favored; others fall back to weighted medians. Radar altitude is only trusted close to the ground and isn’t mixed with barometric altitude to avoid reference-frame mistakes.
- **State filters**: Low-pass filters (with tunable time constants) smooth every critical state: altitude, vertical speed, IAS, angle data, gyro rates, Mach, etc. Angles use wrap-safe interpolation so heading/track stay continuous through 0°/360°.
- **Autopilot loops**: 
  - Altitude loop: altitude error → desired VS → pitch target → elevator (with integral and rate damping).
  - Lateral loop: heading error → bank target → aileron (with integral and roll-rate damping).
  - Rudder loop: coordinates turns using drift/yaw-rate plus asymmetric thrust compensation.
  - Autothrottle loop: PI controller tracks target IAS and feeds both engines (unless AT disengaged, in which case throttle commands are only updated for protections).
  - Trim follows elevator demand so steady-state forces stay small.
- **Envelope protection**: Independent stall and overspeed guards override whatever the AP is doing, commanding full thrust/nose-down for stalls and idle thrust + speedbrake + gentle nose-up for overspeed. Excessive bank/pitch and steep sink close to the ground also trigger corrective commands.
- **Configuration logic**: Gear/flaps extend automatically for landing (low targets, low radar altitude, or WOW). Gear retracts above 3000 ft when landing isn’t imminent. Flaps obey a simple speed/altitude schedule, and speedbrakes auto-retract when on the ground or during flare.
- **Persistence**: `_STATE` keeps all filters and integrators across calls; resets happen automatically when the simulator’s time goes backward (new scenario).
