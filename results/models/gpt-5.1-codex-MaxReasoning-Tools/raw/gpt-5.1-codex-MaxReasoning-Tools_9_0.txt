Here’s a complete C# implementation that uses a best-fit-decreasing heuristic backed by a treap (randomized balanced BST) to keep residual capacities searchable in O(log m) time, plus fast I/O to cope with very large inputs:

```csharp
using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.ReadInt();
        long stockLength = reader.ReadLong();

        if (stockLength == long.MinValue || stockLength <= 0)
        {
            Console.WriteLine("-1");
            return;
        }

        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            long length = reader.ReadLong();
            if (length == long.MinValue || length < 0 || length > stockLength)
            {
                Console.WriteLine("-1");
                return;
            }

            cuts[i] = new Cut(length, i);
        }

        Array.Sort(cuts, CutLengthDescendingComparer.Instance);

        var treap = new StockTreap();
        var assignments = new List<List<int>>(Math.Min(n, 32768));

        foreach (var cut in cuts)
        {
            StockNode node;
            if (treap.TryGetFirstFit(cut.Length, out node))
            {
                treap.Remove(node);
                assignments[node.Id].Add(cut.Index);

                long remaining = node.Remaining - cut.Length;
                if (remaining > 0)
                {
                    treap.Insert(new StockNode(remaining, node.Id));
                }
            }
            else
            {
                int stockId = assignments.Count;
                var bucket = new List<int> { cut.Index };
                assignments.Add(bucket);

                long remaining = stockLength - cut.Length;
                if (remaining > 0)
                {
                    treap.Insert(new StockNode(remaining, stockId));
                }
            }
        }

        using (var outputStream = new BufferedStream(Console.OpenStandardOutput(), 1 << 16))
        using (var writer = new StreamWriter(outputStream))
        {
            writer.WriteLine(assignments.Count);
            foreach (var bucket in assignments)
            {
                if (bucket.Count == 0)
                {
                    writer.WriteLine();
                    continue;
                }

                writer.Write(bucket[0]);
                for (int i = 1; i < bucket.Count; i++)
                {
                    writer.Write(' ');
                    writer.Write(bucket[i]);
                }

                writer.WriteLine();
            }
        }
    }
}

readonly struct Cut
{
    public Cut(long length, int index)
    {
        Length = length;
        Index = index;
    }

    public long Length { get; }
    public int Index { get; }
}

readonly struct StockNode
{
    public StockNode(long remaining, int id)
    {
        Remaining = remaining;
        Id = id;
    }

    public long Remaining { get; }
    public int Id { get; }
}

sealed class CutLengthDescendingComparer : IComparer<Cut>
{
    public static readonly CutLengthDescendingComparer Instance = new CutLengthDescendingComparer();

    public int Compare(Cut x, Cut y)
    {
        return y.Length.CompareTo(x.Length);
    }
}

sealed class StockNodeComparer : IComparer<StockNode>
{
    public static readonly StockNodeComparer Instance = new StockNodeComparer();

    public int Compare(StockNode x, StockNode y)
    {
        int cmp = x.Remaining.CompareTo(y.Remaining);
        if (cmp != 0)
        {
            return cmp;
        }

        return x.Id.CompareTo(y.Id);
    }
}

sealed class StockTreap
{
    private sealed class Node
    {
        public Node(StockNode value, int priority)
        {
            Value = value;
            Priority = priority;
        }

        public StockNode Value;
        public int Priority;
        public Node Left;
        public Node Right;
    }

    private Node _root;
    private uint _seed = 0x9E3779B9u;
    private static readonly StockNodeComparer Comparer = StockNodeComparer.Instance;

    public void Insert(StockNode value)
    {
        _root = Insert(_root, value);
    }

    public bool Remove(StockNode value)
    {
        bool removed;
        _root = Remove(_root, value, out removed);
        return removed;
    }

    public bool TryGetFirstFit(long needed, out StockNode result)
    {
        Node current = _root;
        Node best = null;
        var target = new StockNode(needed, int.MinValue);

        while (current != null)
        {
            if (Comparer.Compare(current.Value, target) < 0)
            {
                current = current.Right;
            }
            else
            {
                best = current;
                current = current.Left;
            }
        }

        if (best != null)
        {
            result = best.Value;
            return true;
        }

        result = default(StockNode);
        return false;
    }

    private Node Insert(Node node, StockNode value)
    {
        if (node == null)
        {
            return new Node(value, NextPriority());
        }

        if (Comparer.Compare(value, node.Value) < 0)
        {
            node.Left = Insert(node.Left, value);
            if (node.Left != null && node.Left.Priority < node.Priority)
            {
                node = RotateRight(node);
            }
        }
        else
        {
            node.Right = Insert(node.Right, value);
            if (node.Right != null && node.Right.Priority < node.Priority)
            {
                node = RotateLeft(node);
            }
        }

        return node;
    }

    private Node Remove(Node node, StockNode value, out bool removed)
    {
        if (node == null)
        {
            removed = false;
            return null;
        }

        int cmp = Comparer.Compare(value, node.Value);
        if (cmp < 0)
        {
            node.Left = Remove(node.Left, value, out removed);
        }
        else if (cmp > 0)
        {
            node.Right = Remove(node.Right, value, out removed);
        }
        else
        {
            removed = true;
            if (node.Left == null) return node.Right;
            if (node.Right == null) return node.Left;

            if (node.Left.Priority < node.Right.Priority)
            {
                node = RotateRight(node);
                bool dummy;
                node.Right = Remove(node.Right, value, out dummy);
            }
            else
            {
                node = RotateLeft(node);
                bool dummy;
                node.Left = Remove(node.Left, value, out dummy);
            }

            return node;
        }

        return node;
    }

    private static Node RotateLeft(Node node)
    {
        Node right = node.Right;
        node.Right = right.Left;
        right.Left = node;
        return right;
    }

    private static Node RotateRight(Node node)
    {
        Node left = node.Left;
        node.Left = left.Right;
        left.Right = node;
        return left;
    }

    private int NextPriority()
    {
        uint x = _seed;
        x ^= x << 13;
        x ^= x >> 17;
        x ^= x << 5;
        _seed = x;
        return (int)(x & 0x7fffffff);
    }
}

sealed class FastReader
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _length;
    private int _position;

    public FastReader(Stream stream, int bufferSize = 1 << 20)
    {
        _stream = stream;
        _buffer = new byte[bufferSize];
    }

    public int ReadInt()
    {
        return (int)ReadLong();
    }

    public long ReadLong()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return long.MinValue;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        long value = 0;
        while (c > ' ')
        {
            value = value * 10 + (c - '0');
            c = Read();
        }

        return sign == 1 ? value : -value;
    }

    private int Read()
    {
        if (_position >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _position = 0;
            if (_length == 0) return -1;
        }

        return _buffer[_position++];
    }
}
```
