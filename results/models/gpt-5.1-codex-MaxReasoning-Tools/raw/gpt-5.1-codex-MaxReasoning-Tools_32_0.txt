Here's a complete C++20 solution that combines an exact Dreyfus–Wagner dynamic program for small terminal sets with a fast shortest-path heuristic (and a final MST fallback) for larger inputs. It adheres to the given Windows/MSVC toolchain and avoids non-standard headers.

```cpp
#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

struct Edge {
    int u;
    int v;
    long long w;
};

struct AdjEdge {
    int to;
    long long w;
    int id;
};

class DSU {
public:
    explicit DSU(int n = 0) { init(n); }
    void init(int n) {
        parent.resize(n);
        size.assign(n, 1);
        std::iota(parent.begin(), parent.end(), 0);
    }
    int find(int x) {
        while (parent[x] != x) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (size[a] < size[b]) std::swap(a, b);
        parent[b] = a;
        size[a] += size[b];
        return true;
    }
private:
    std::vector<int> parent;
    std::vector<int> size;
};

struct SteinerResult {
    bool success = false;
    long long total_weight = 0;
    std::vector<int> edges;
};

constexpr long long INF = std::numeric_limits<long long>::max() / 4;

bool validate_tree(int n,
                   const std::vector<Edge>& edges,
                   const std::vector<int>& terminals,
                   const std::vector<int>& edge_ids) {
    if (terminals.empty()) return edge_ids.empty();
    if (terminals.size() == 1) return edge_ids.empty();
    DSU dsu(n);
    for (int id : edge_ids) {
        const auto& e = edges[id];
        if (!dsu.unite(e.u, e.v)) return false; // cycle detected
    }
    int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root) return false;
    }
    return true;
}

SteinerResult solve_exact_dp(int n,
                             const std::vector<Edge>& edges,
                             const std::vector<std::vector<AdjEdge>>& adj,
                             const std::vector<int>& terminals) {
    SteinerResult res;
    int k = static_cast<int>(terminals.size());
    if (k == 0 || n == 0) {
        res.success = (k == 0);
        return res;
    }
    int total_masks = 1 << k;
    size_t total_states = static_cast<size_t>(total_masks) * static_cast<size_t>(n);
    std::vector<long long> dp(total_states, INF);
    std::vector<int> parent_vertex(total_states, -1);
    std::vector<int> parent_edge(total_states, -1);
    std::vector<int> split_state(total_states, 0);

    for (int i = 0; i < k; ++i) {
        size_t mask_val = static_cast<size_t>(1) << i;
        size_t idx = mask_val * static_cast<size_t>(n) + static_cast<size_t>(terminals[i]);
        dp[idx] = 0;
    }

    using PQNode = std::pair<long long, int>;

    for (int mask = 1; mask < total_masks; ++mask) {
        size_t base_mask = static_cast<size_t>(mask) * static_cast<size_t>(n);

        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int other = mask ^ sub;
            if (other == 0) continue;
            if (sub > other) continue; // avoid duplicate splits

            size_t base_sub = static_cast<size_t>(sub) * static_cast<size_t>(n);
            size_t base_other = static_cast<size_t>(other) * static_cast<size_t>(n);

            for (int v = 0; v < n; ++v) {
                long long val1 = dp[base_sub + v];
                long long val2 = dp[base_other + v];
                if (val1 >= INF || val2 >= INF) continue;
                long long cand = (val1 > INF - val2) ? INF : val1 + val2;
                size_t idx = base_mask + v;
                if (cand < dp[idx]) {
                    dp[idx] = cand;
                    parent_vertex[idx] = -1;
                    parent_edge[idx] = -1;
                    split_state[idx] = sub;
                }
            }
        }

        std::priority_queue<PQNode, std::vector<PQNode>, std::greater<PQNode>> pq;
        bool has_finite = false;
        for (int v = 0; v < n; ++v) {
            long long d = dp[base_mask + v];
            if (d < INF) {
                has_finite = true;
                pq.emplace(d, v);
            }
        }
        if (!has_finite) continue;

        while (!pq.empty()) {
            auto [dist_u, u] = pq.top();
            pq.pop();
            size_t idx_u = base_mask + u;
            if (dist_u != dp[idx_u]) continue;

            for (const auto& edge : adj[u]) {
                long long cand = (dist_u > INF - edge.w) ? INF : dist_u + edge.w;
                size_t idx_v = base_mask + edge.to;
                if (cand < dp[idx_v]) {
                    dp[idx_v] = cand;
                    parent_vertex[idx_v] = u;
                    parent_edge[idx_v] = edge.id;
                    split_state[idx_v] = 0;
                    pq.emplace(cand, edge.to);
                }
            }
        }
    }

    int full_mask = total_masks - 1;
    size_t base_full = static_cast<size_t>(full_mask) * static_cast<size_t>(n);
    long long best_value = INF;
    int best_vertex = -1;
    for (int v = 0; v < n; ++v) {
        long long d = dp[base_full + v];
        if (d < best_value) {
            best_value = d;
            best_vertex = v;
        }
    }
    if (best_vertex == -1 || best_value >= INF) {
        return res; // failed
    }

    std::vector<char> visited(total_states, 0);
    std::vector<char> used_edge(edges.size(), 0);
    std::vector<int> selected_edges;
    selected_edges.reserve(n);

    std::vector<std::pair<int, int>> stack;
    stack.emplace_back(full_mask, best_vertex);

    while (!stack.empty()) {
        auto [mask, v] = stack.back();
        stack.pop_back();
        size_t idx = static_cast<size_t>(mask) * static_cast<size_t>(n) + static_cast<size_t>(v);
        if (visited[idx]) continue;
        visited[idx] = 1;

        int split = split_state[idx];
        if (split != 0) {
            int other = mask ^ split;
            stack.emplace_back(split, v);
            stack.emplace_back(other, v);
        } else {
            int parent = parent_vertex[idx];
            int edge_id = parent_edge[idx];
            if (parent != -1 && edge_id != -1) {
                if (!used_edge[edge_id]) {
                    used_edge[edge_id] = 1;
                    selected_edges.push_back(edge_id);
                }
                stack.emplace_back(mask, parent);
            }
        }
    }

    long long total_weight = 0;
    for (int id : selected_edges) {
        total_weight += edges[id].w;
    }
    res.success = true;
    res.total_weight = total_weight;
    res.edges = std::move(selected_edges);
    return res;
}

SteinerResult solve_shortest_path_heuristic(int n,
                                            const std::vector<Edge>& edges,
                                            const std::vector<std::vector<AdjEdge>>& adj,
                                            const std::vector<int>& terminals) {
    SteinerResult res;
    int k = static_cast<int>(terminals.size());
    if (k == 0) {
        res.success = true;
        return res;
    }
    if (k == 1) {
        res.success = true;
        return res;
    }

    std::vector<int> terminal_index(n, -1);
    for (int i = 0; i < k; ++i) {
        terminal_index[terminals[i]] = i;
    }

    std::vector<char> in_tree(n, 0);
    std::vector<int> tree_nodes;
    tree_nodes.reserve(n);

    std::vector<char> terminal_in_tree(k, 0);
    std::vector<char> used_edge(edges.size(), 0);

    int root = terminals[0];
    in_tree[root] = 1;
    tree_nodes.push_back(root);
    terminal_in_tree[0] = 1;
    int connected = 1;

    std::vector<long long> dist(n, 0);
    std::vector<int> parent_v(n, -1);
    std::vector<int> parent_edge(n, -1);
    std::vector<int> seen(n, -1);
    int iter_token = 0;

    using PQNode = std::pair<long long, int>;

    while (connected < k) {
        ++iter_token;
        std::priority_queue<PQNode, std::vector<PQNode>, std::greater<PQNode>> pq;
        for (int node : tree_nodes) {
            seen[node] = iter_token;
            dist[node] = 0;
            parent_v[node] = -1;
            parent_edge[node] = -1;
            pq.emplace(0, node);
        }

        int target = -1;
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (seen[u] != iter_token || d != dist[u]) continue;

            int t_idx = terminal_index[u];
            if (t_idx != -1 && !terminal_in_tree[t_idx]) {
                target = u;
                break;
            }

            for (const auto& edge : adj[u]) {
                long long cand = (d > INF - edge.w) ? INF : d + edge.w;
                if (seen[edge.to] != iter_token || cand < dist[edge.to]) {
                    seen[edge.to] = iter_token;
                    dist[edge.to] = cand;
                    parent_v[edge.to] = u;
                    parent_edge[edge.to] = edge.id;
                    pq.emplace(cand, edge.to);
                }
            }
        }

        if (target == -1) {
            res.success = false;
            res.edges.clear();
            res.total_weight = 0;
            return res;
        }

        if (in_tree[target]) {
            int idx = terminal_index[target];
            if (idx != -1 && !terminal_in_tree[idx]) {
                terminal_in_tree[idx] = 1;
                ++connected;
            }
            continue;
        }

        std::vector<int> path_nodes;
        int cur = target;
        while (!in_tree[cur]) {
            path_nodes.push_back(cur);
            int prev = parent_v[cur];
            if (prev == -1) {
                res.success = false;
                res.edges.clear();
                res.total_weight = 0;
                return res;
            }
            cur = prev;
        }

        for (int node : path_nodes) {
            int eid = parent_edge[node];
            if (eid == -1) {
                res.success = false;
                res.edges.clear();
                res.total_weight = 0;
                return res;
            }
            if (!used_edge[eid]) {
                used_edge[eid] = 1;
                res.edges.push_back(eid);
                res.total_weight += edges[eid].w;
            }
        }

        for (int node : path_nodes) {
            if (!in_tree[node]) {
                in_tree[node] = 1;
                tree_nodes.push_back(node);
                int idx = terminal_index[node];
                if (idx != -1 && !terminal_in_tree[idx]) {
                    terminal_in_tree[idx] = 1;
                    ++connected;
                }
            }
        }

        int meet_idx = terminal_index[cur];
        if (meet_idx != -1 && !terminal_in_tree[meet_idx]) {
            terminal_in_tree[meet_idx] = 1;
            ++connected;
        }
    }

    res.success = true;
    return res;
}

SteinerResult build_fallback_mst(int n,
                                 const std::vector<Edge>& edges,
                                 const std::vector<int>& terminals) {
    SteinerResult res;
    if (terminals.empty()) {
        res.success = true;
        return res;
    }
    DSU dsu(n);
    std::vector<int> order(edges.size());
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });

    std::vector<int> forest_edges;
    forest_edges.reserve(n ? n - 1 : 0);
    for (int id : order) {
        if (dsu.unite(edges[id].u, edges[id].v)) {
            forest_edges.push_back(id);
        }
    }

    std::vector<std::vector<std::pair<int, int>>> tree_adj(n);
    for (int id : forest_edges) {
        const auto& e = edges[id];
        tree_adj[e.u].push_back({e.v, id});
        tree_adj[e.v].push_back({e.u, id});
    }

    std::vector<char> visited(n, 0);
    std::vector<int> stack;
    std::vector<int> component_edges;
    int root = terminals[0];
    stack.push_back(root);
    visited[root] = 1;

    while (!stack.empty()) {
        int u = stack.back();
        stack.pop_back();
        for (const auto& [v, eid] : tree_adj[u]) {
            if (!visited[v]) {
                visited[v] = 1;
                component_edges.push_back(eid);
                stack.push_back(v);
            }
        }
    }

    for (int t : terminals) {
        if (t < 0 || t >= n || !visited[t]) {
            res.success = false;
            res.edges.clear();
            res.total_weight = 0;
            return res;
        }
    }

    long long total_weight = 0;
    for (int id : component_edges) total_weight += edges[id].w;

    res.success = true;
    res.total_weight = total_weight;
    res.edges = std::move(component_edges);
    return res;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    if (!(std::cin >> n >> m >> k)) return 0;

    if (n <= 0) {
        std::cout << "0 0\n";
        return 0;
    }

    std::vector<Edge> edges(m);
    std::vector<std::vector<AdjEdge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        std::cin >> u >> v >> w;
        edges[i] = {u, v, w};
        adj[u].push_back({v, w, i});
        adj[v].push_back({u, w, i});
    }

    std::vector<int> raw_terms(k);
    for (int i = 0; i < k; ++i) std::cin >> raw_terms[i];

    std::vector<int> terminals;
    terminals.reserve(k);
    std::vector<char> seen_vertex(n, 0);
    for (int v : raw_terms) {
        if (v < 0 || v >= n) continue;
        if (!seen_vertex[v]) {
            seen_vertex[v] = 1;
            terminals.push_back(v);
        }
    }

    if (terminals.empty()) {
        std::cout << "0 0\n";
        return 0;
    }
    if (terminals.size() == 1) {
        std::cout << "0 0\n";
        return 0;
    }

    const int MAX_DP_TERMINALS = 17;
    const unsigned long long MAX_DP_STATES = 8'000'000ull;

    bool try_dp = false;
    int k_unique = static_cast<int>(terminals.size());
    if (k_unique <= MAX_DP_TERMINALS) {
        unsigned long long total_masks = 1ull << k_unique;
        unsigned long long states = total_masks * static_cast<unsigned long long>(n);
        if (states <= MAX_DP_STATES) try_dp = true;
    }

    SteinerResult result;
    bool solved = false;

    if (try_dp) {
        result = solve_exact_dp(n, edges, adj, terminals);
        if (result.success && validate_tree(n, edges, terminals, result.edges)) {
            solved = true;
        }
    }

    if (!solved) {
        result = solve_shortest_path_heuristic(n, edges, adj, terminals);
        if (result.success && validate_tree(n, edges, terminals, result.edges)) {
            solved = true;
        }
    }

    if (!solved) {
        result = build_fallback_mst(n, edges, terminals);
        if (result.success && validate_tree(n, edges, terminals, result.edges)) {
            solved = true;
        }
    }

    if (!solved) {
        std::cout << "0 0\n";
        return 0;
    }

    if (result.total_weight == 0) {
        long long recompute = 0;
        for (int id : result.edges) recompute += edges[id].w;
        result.total_weight = recompute;
    }

    std::cout << result.total_weight << ' ' << result.edges.size() << '\n';
    for (int id : result.edges) {
        std::cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
    return 0;
}
```

**Highlights:**

1. **Exact solver when feasible:** For up to 17 terminals (and manageable state counts) the Dreyfus–Wagner DP yields an optimal tree.
2. **Scalable heuristic:** For larger instances, the multi-source shortest-path heuristic incrementally connects terminals using true shortest paths, delivering a valid Steiner tree quickly.
3. **Safety net MST:** If needed, a final Kruskal/DFS combination produces a spanning tree for the component containing the terminals.
4. **Validation:** Every candidate tree is checked for acyclicity and terminal connectivity before output.

This approach balances optimality and scalability, ensuring a valid tree is always emitted under typical contest constraints.
