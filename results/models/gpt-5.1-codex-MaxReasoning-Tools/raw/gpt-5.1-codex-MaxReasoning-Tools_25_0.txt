using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public class Program
{
    private static readonly CultureInfo Culture = CultureInfo.InvariantCulture;
    private const double SecondsPerDay = 86400.0;
    private const double LaunchAltitude = 6.771e6;      // ~400 km circular LEO
    private const double ImpactToleranceMeters = 250.0; // target intercept accuracy
    private const double CorrectionToleranceMeters = 1500.0;
    private const double DvPerturb = 1.0;               // m/s, finite-difference step

    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        string firstToken = reader.Next();
        if (firstToken == null)
        {
            return;
        }

        int bodyCount = int.Parse(firstToken, Culture);
        double warningDays = reader.NextDouble();
        double impactorMass = reader.NextDouble();
        double deltaVBudget = reader.NextDouble();
        double asteroidMass = reader.NextDouble();
        double t0UnixUtc = reader.NextDouble();

        var bodies = new BodyDefinition[bodyCount];
        int earthIndex = -1;
        int asteroidIndex = -1;
        int sunIndex = -1;

        for (int i = 0; i < bodyCount; i++)
        {
            string name = reader.Next();
            double mass = reader.NextDouble();
            double x = reader.NextDouble();
            double y = reader.NextDouble();
            double z = reader.NextDouble();
            double vx = reader.NextDouble();
            double vy = reader.NextDouble();
            double vz = reader.NextDouble();

            bodies[i] = new BodyDefinition
            {
                Name = name,
                Mass = mass,
                Position = new Vector3d(x, y, z),
                Velocity = new Vector3d(vx, vy, vz)
            };

            if (name.Equals("Earth", StringComparison.OrdinalIgnoreCase))
            {
                earthIndex = i;
            }

            if (name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase))
            {
                asteroidIndex = i;
            }

            if (name.Equals("Sun", StringComparison.OrdinalIgnoreCase))
            {
                sunIndex = i;
            }
        }

        string _ = reader.Next(); // de421 path (unused but consumed)

        if (earthIndex < 0 || asteroidIndex < 0)
        {
            OutputFallback(warningDays);
            return;
        }

        double baseEphemStep = warningDays / 4000.0;
        if (warningDays <= 0)
        {
            warningDays = 1.0;
        }

        if (baseEphemStep < 1.0 / 96.0) baseEphemStep = 1.0 / 96.0;
        if (baseEphemStep > 1.0) baseEphemStep = 1.0;
        if (baseEphemStep > warningDays) baseEphemStep = warningDays / 4.0;

        var ephemeris = Ephemeris.Build(bodies, warningDays, baseEphemStep);
        double spacecraftStep = baseEphemStep * 0.5;
        if (spacecraftStep > 0.5) spacecraftStep = 0.5;
        if (spacecraftStep < 1.0 / 240.0) spacecraftStep = 1.0 / 240.0;

        var impactInfo = ImpactAnalyzer.Analyze(ephemeris, earthIndex, asteroidIndex, warningDays);
        Vector3d deflectionDirection =
            Vector3d.Normalize(ephemeris.SamplePosition(asteroidIndex, impactInfo.Time) -
                               ephemeris.SamplePosition(earthIndex, impactInfo.Time));
        if (deflectionDirection.LengthSquared < 1e-12)
        {
            deflectionDirection = new Vector3d(1, 0, 0);
        }

        var simulator = new TrajectorySimulator(ephemeris, earthIndex, asteroidIndex, sunIndex, spacecraftStep, LaunchAltitude);

        var context = new MissionContext
        {
            Ephemeris = ephemeris,
            Simulator = simulator,
            WarningDays = warningDays,
            DeltaVBudget = deltaVBudget,
            ImpactorMass = impactorMass,
            AsteroidMass = asteroidMass,
            PredictedImpactTime = impactInfo.Time,
            DeflectionDirection = deflectionDirection,
            ImpactTolerance = ImpactToleranceMeters,
            CorrectionTolerance = CorrectionToleranceMeters,
            EarthIndex = earthIndex,
            AsteroidIndex = asteroidIndex,
            SunIndex = sunIndex
        };

        MissionPlan plan = MissionPlanner.PlanMission(context, DvPerturb);
        if (plan == null)
        {
            plan = MissionPlanner.CreateFallback(warningDays);
        }

        EmitPlan(plan);
    }

    private static void EmitPlan(MissionPlan plan)
    {
        var burns = new List<Burn>
        {
            new Burn(plan.LaunchTime, plan.LaunchDeltaV)
        };

        if (plan.Corrections != null && plan.Corrections.Count > 0)
        {
            burns.AddRange(plan.Corrections);
        }

        burns.Add(new Burn(plan.ImpactTime, plan.ImpactDeltaV));
        burns.Sort((a, b) => a.Time.CompareTo(b.Time));

        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine(string.Format(Culture,
                "{0:G17} {1:G17} {2:G17} {3:G17}",
                burn.Time,
                burn.DeltaV.X,
                burn.DeltaV.Y,
                burn.DeltaV.Z));
        }
    }

    private static void OutputFallback(double warningDays)
    {
        double launchTime = Math.Min(1.0, warningDays * 0.1);
        double correctionTime = Math.Min(warningDays * 0.5, warningDays - 1.0);
        double impactTime = Math.Max(correctionTime + 1.0, warningDays * 0.9);

        Console.WriteLine(3);
        Console.WriteLine(string.Format(Culture, "{0:G17} 0 0 0", launchTime));
        Console.WriteLine(string.Format(Culture, "{0:G17} 0 0 0", correctionTime));
        Console.WriteLine(string.Format(Culture, "{0:G17} 0 0 0", impactTime));
    }
}

internal sealed class MissionPlanner
{
    public static MissionPlan PlanMission(MissionContext context, double dvPerturb)
    {
        var candidates = DetermineInterceptCandidates(context.WarningDays, context.PredictedImpactTime);
        MissionPlan bestPlan = null;
        double bestScore = double.PositiveInfinity;

        foreach (double intercept in candidates)
        {
            if (intercept <= 0 || intercept > context.WarningDays) continue;
            var options = BuildFlightOptions(context, intercept);
            if (options.Count == 0) continue;

            foreach (var option in options)
            {
                var plan = BuildPlanForOption(context, option, intercept, dvPerturb);
                if (plan == null) continue;

                double score = plan.FinalSeparation + 0.0001 * plan.TotalDeltaV;
                if (score < bestScore)
                {
                    bestScore = score;
                    bestPlan = plan;
                }

                if (plan.FinalSeparation <= context.ImpactTolerance && plan.TotalDeltaV <= context.DeltaVBudget + 1e-3)
                {
                    return plan;
                }
            }
        }

        return bestPlan;
    }

    public static MissionPlan CreateFallback(double warningDays)
    {
        double launchTime = Math.Min(5.0, warningDays * 0.1);
        double correctionTime = Math.Min(launchTime + 10.0, warningDays * 0.5);
        double impactTime = Math.Max(correctionTime + 10.0, warningDays * 0.9);

        return new MissionPlan
        {
            LaunchTime = launchTime,
            LaunchDeltaV = Vector3d.Zero,
            Corrections = new List<Burn> { new Burn(correctionTime, Vector3d.Zero) },
            ImpactTime = impactTime,
            ImpactDeltaV = Vector3d.Zero,
            TotalDeltaV = 0,
            FinalSeparation = double.PositiveInfinity
        };
    }

    private static MissionPlan BuildPlanForOption(MissionContext context, FlightOption option, double interceptTime, double dvPerturb)
    {
        double flightTime = interceptTime - option.LaunchTime;
        if (flightTime <= 0.5) return null;

        StageBudget budgets = ComputeStageBudgets(context.DeltaVBudget);
        Vector3d launchDv = option.InitialDv;

        if (launchDv.Length > budgets.Launch)
        {
            launchDv = launchDv.Normalized() * budgets.Launch;
        }

        var corrections = InitializeCorrections(option.LaunchTime, interceptTime, context.WarningDays);
        if (corrections.Count == 0)
        {
            double mid = option.LaunchTime + flightTime * 0.6;
            if (mid < interceptTime - 0.5)
                corrections.Add(new Correction { Time = mid, DeltaV = Vector3d.Zero });
        }

        var simulator = context.Simulator;

        OptimizeLaunch(simulator, context, interceptTime, corrections, budgets.Launch, ref launchDv, dvPerturb);

        OptimizeCorrections(simulator, context, interceptTime, launchDv, corrections, budgets.MidCourse, dvPerturb);

        FineTuneLaunch(simulator, context, interceptTime, corrections, budgets.Launch, ref launchDv, dvPerturb);

        var burnList = ComposeBurns(option.LaunchTime, launchDv, corrections, -1, Vector3d.Zero);
        var finalResult = simulator.Run(burnList, interceptTime, null);

        if (!finalResult.SpacecraftActive)
        {
            return null;
        }

        double totalDvUsed = launchDv.Length;
        foreach (var correction in corrections)
        {
            totalDvUsed += correction.DeltaV.Length;
        }

        Vector3d impactDv = ComputeImpactBurn(context, finalResult, interceptTime, totalDvUsed);
        totalDvUsed += impactDv.Length;

        double finalSeparation = finalResult.FinalSeparation;
        var correctionBurns = new List<Burn>();
        foreach (var correction in corrections)
        {
            correctionBurns.Add(new Burn(correction.Time, correction.DeltaV));
        }

        return new MissionPlan
        {
            LaunchTime = option.LaunchTime,
            LaunchDeltaV = launchDv,
            Corrections = correctionBurns,
            ImpactTime = interceptTime,
            ImpactDeltaV = impactDv,
            TotalDeltaV = totalDvUsed,
            FinalSeparation = finalSeparation,
            ClosestApproach = finalResult.ClosestApproach,
            TimeOfClosestApproach = finalResult.TimeOfClosestApproach
        };
    }

    private static Vector3d ComputeImpactBurn(MissionContext context, SimulationResult result, double interceptTime, double dvUsed)
    {
        double remaining = Math.Max(0.0, context.DeltaVBudget - dvUsed);
        if (!result.SpacecraftActive || remaining <= 1e-3) return Vector3d.Zero;

        Vector3d relVel = result.SpacecraftVelocity - result.AsteroidVelocity;
        Vector3d desiredDir = context.DeflectionDirection.LengthSquared > 1e-12
            ? context.DeflectionDirection
            : relVel.Normalized();

        if (desiredDir.LengthSquared < 1e-12)
        {
            desiredDir = new Vector3d(1, 0, 0);
        }

        double desiredSpeed = relVel.Length + remaining * 0.5;
        if (desiredSpeed < relVel.Length) desiredSpeed = relVel.Length;

        Vector3d targetRelative = desiredDir.Normalized() * desiredSpeed;
        Vector3d delta = targetRelative - relVel;
        double deltaMag = delta.Length;

        if (deltaMag <= remaining)
        {
            return delta;
        }

        return delta / deltaMag * remaining;
    }

    private static void OptimizeLaunch(TrajectorySimulator simulator, MissionContext context, double interceptTime,
        List<Correction> corrections, double maxLaunchDv, ref Vector3d launchDv, double dvPerturb)
    {
        Vector3d bestDv = launchDv;
        double bestError = double.MaxValue;

        for (int iter = 0; iter < 4; iter++)
        {
            var burns = ComposeBurns(launchDvTime: context, launchDv, corrections, -1, Vector3d.Zero);
            var result = simulator.Run(burns, interceptTime, null);

            if (!result.SpacecraftActive)
            {
                break;
            }

            Vector3d error = result.SpacecraftPosition - result.AsteroidPosition;
            double errorNorm = error.Length;

            if (errorNorm < bestError)
            {
                bestError = errorNorm;
                bestDv = launchDv;
            }

            if (errorNorm <= context.ImpactTolerance * 5)
            {
                break;
            }

            Vector3d[] columns = new Vector3d[3];
            for (int axis = 0; axis < 3; axis++)
            {
                Vector3d axisVec = AxisVector(axis) * dvPerturb;
                var perturbedBurns = ComposeBurns(launchDvTime: context, launchDv + axisVec, corrections, -1, Vector3d.Zero);
                var pertResult = simulator.Run(perturbedBurns, interceptTime, null);
                Vector3d pertError = pertResult.SpacecraftPosition - pertResult.AsteroidPosition;
                columns[axis] = (pertError - error) / dvPerturb;
            }

            Vector3d update = SolveForDelta(columns, error, interceptTime - context.LaunchWindowStart);
            double damping = 0.6;
            launchDv += update * damping;

            double mag = launchDv.Length;
            if (mag > maxLaunchDv && mag > 1e-9)
            {
                launchDv = launchDv / mag * maxLaunchDv;
            }
        }

        launchDv = bestDv;
    }

    private static void OptimizeCorrections(TrajectorySimulator simulator, MissionContext context, double interceptTime,
        Vector3d launchDv, List<Correction> corrections, double maxTotalCorrectionDv, double dvPerturb)
    {
        for (int idx = 0; idx < corrections.Count; idx++)
        {
            for (int iter = 0; iter < 3; iter++)
            {
                var burns = ComposeBurns(context.LaunchWindowStart, launchDv, corrections, -1, Vector3d.Zero);
                var result = simulator.Run(burns, interceptTime, null);

                if (!result.SpacecraftActive) return;

                Vector3d error = result.SpacecraftPosition - result.AsteroidPosition;
                if (error.Length <= context.ImpactTolerance) break;

                Vector3d[] columns = new Vector3d[3];
                for (int axis = 0; axis < 3; axis++)
                {
                    Vector3d axisVec = AxisVector(axis) * dvPerturb;
                    var pertBurns = ComposeBurns(context.LaunchWindowStart, launchDv, corrections, idx, axisVec);
                    var pertResult = simulator.Run(pertBurns, interceptTime, null);
                    Vector3d pertError = pertResult.SpacecraftPosition - pertResult.AsteroidPosition;
                    columns[axis] = (pertError - error) / dvPerturb;
                }

                Vector3d update = SolveForDelta(columns, error, interceptTime - corrections[idx].Time);
                Vector3d newDv = corrections[idx].DeltaV + update * 0.7;
                double limit = AvailableCorrectionBudget(corrections, idx, maxTotalCorrectionDv);

                double mag = newDv.Length;
                if (mag > limit && limit > 1e-6)
                {
                    newDv = newDv / mag * limit;
                }
                else if (limit <= 1e-6)
                {
                    break;
                }

                corrections[idx] = new Correction { Time = corrections[idx].Time, DeltaV = newDv };
            }
        }
    }

    private static void FineTuneLaunch(TrajectorySimulator simulator, MissionContext context, double interceptTime,
        List<Correction> corrections, double maxLaunchDv, ref Vector3d launchDv, double dvPerturb)
    {
        for (int iter = 0; iter < 2; iter++)
        {
            var burns = ComposeBurns(context.LaunchWindowStart, launchDv, corrections, -1, Vector3d.Zero);
            var result = simulator.Run(burns, interceptTime, null);
            if (!result.SpacecraftActive) return;

            Vector3d error = result.SpacecraftPosition - result.AsteroidPosition;
            if (error.Length <= context.ImpactTolerance) break;

            Vector3d[] columns = new Vector3d[3];
            for (int axis = 0; axis < 3; axis++)
            {
                Vector3d axisVec = AxisVector(axis) * dvPerturb;
                var pertBurns = ComposeBurns(context.LaunchWindowStart, launchDv + axisVec, corrections, -1, Vector3d.Zero);
                var pertResult = simulator.Run(pertBurns, interceptTime, null);
                Vector3d pertError = pertResult.SpacecraftPosition - pertResult.AsteroidPosition;
                columns[axis] = (pertError - error) / dvPerturb;
            }

            Vector3d update = SolveForDelta(columns, error, interceptTime - context.LaunchWindowStart);
            launchDv += update * 0.5;

            double mag = launchDv.Length;
            if (mag > maxLaunchDv && mag > 1e-9)
            {
                launchDv = launchDv / mag * maxLaunchDv;
            }
        }
    }

    private static Vector3d SolveForDelta(Vector3d[] columns, Vector3d error, double timeSpanDays)
    {
        double[,] matrix = new double[3, 3];
        for (int col = 0; col < 3; col++)
        {
            matrix[0, col] = columns[col].X;
            matrix[1, col] = columns[col].Y;
            matrix[2, col] = columns[col].Z;
        }

        double[] rhs = { -error.X, -error.Y, -error.Z };
        if (LinearSolver.Solve3x3(matrix, rhs, out Vector3d solution))
        {
            return solution;
        }

        double scale = 1.0 / Math.Max(SecondsPerDay * Math.Max(timeSpanDays, 1.0), 1.0);
        return error * -scale;
    }

    private static Vector3d AxisVector(int axis)
    {
        return axis switch
        {
            0 => new Vector3d(1, 0, 0),
            1 => new Vector3d(0, 1, 0),
            _ => new Vector3d(0, 0, 1),
        };
    }

    private static StageBudget ComputeStageBudgets(double budget)
    {
        if (budget <= 0)
        {
            return new StageBudget(0, 0, 0);
        }

        double reserved = Math.Min(budget * 0.4, Math.Max(0.05 * budget + 0.15 * budget, 100.0));
        double launch = budget - reserved;
        if (launch < budget * 0.3)
        {
            launch = budget * 0.3;
            reserved = budget - launch;
        }

        double finalReserve = reserved * 0.3;
        double midReserve = reserved - finalReserve;
        return new StageBudget(launch, midReserve, finalReserve);
    }

    private static List<Correction> InitializeCorrections(double launchTime, double interceptTime, double warningDays)
    {
        var corrections = new List<Correction>();
        double flight = interceptTime - launchTime;
        double[] fractions = flight > 120.0
            ? new[] { 0.35, 0.65, 0.85 }
            : flight > 30.0
                ? new[] { 0.4, 0.75 }
                : new[] { 0.5 };

        foreach (double fraction in fractions)
        {
            double time = launchTime + flight * fraction;
            if (time - launchTime < 0.5) continue;
            if (interceptTime - time < 0.5) continue;
            if (time > warningDays) continue;
            corrections.Add(new Correction { Time = time, DeltaV = Vector3d.Zero });
        }

        if (corrections.Count == 0)
        {
            double fallback = launchTime + flight * 0.6;
            if (fallback < interceptTime - 0.25)
            {
                corrections.Add(new Correction { Time = fallback, DeltaV = Vector3d.Zero });
            }
        }

        return corrections;
    }

    private static double AvailableCorrectionBudget(List<Correction> corrections, int targetIndex, double maxTotal)
    {
        double used = 0.0;
        for (int i = 0; i < corrections.Count; i++)
        {
            if (i == targetIndex) continue;
            used += corrections[i].DeltaV.Length;
        }

        double left = maxTotal - used;
        return left > 0 ? left : 0;
    }

    private static List<double> DetermineInterceptCandidates(double warningDays, double predictedImpactTime)
    {
        var set = new SortedSet<double>();

        double margin = Math.Max(10.0, 0.05 * warningDays);
        double[] offsets = { margin, margin * 2.0, margin * 3.5 };

        foreach (double offset in offsets)
        {
            double t = predictedImpactTime - offset;
            if (t > 1.0) set.Add(Math.Min(t, warningDays - 0.5));
        }

        double[] scale = { 0.9, 0.7, 0.5, 0.35 };
        foreach (double factor in scale)
        {
            double t = predictedImpactTime * factor;
            if (t > 1.0) set.Add(Math.Min(t, warningDays - 0.5));
        }

        if (set.Count == 0)
        {
            set.Add(Math.Max(1.0, warningDays * 0.6));
        }

        var list = new List<double>(set);
        list.Reverse();
        return list;
    }

    private static List<FlightOption> BuildFlightOptions(MissionContext context, double interceptTime)
    {
        var options = new List<FlightOption>();
        double minFlight = Math.Max(5.0, interceptTime * 0.15);
        double maxFlight = Math.Max(minFlight + 1.0, interceptTime - Math.Max(0.5, interceptTime * 0.02));

        if (maxFlight <= 0.5) return options;
        if (minFlight >= maxFlight) minFlight = Math.Max(0.5, maxFlight * 0.5);

        int samples = 8;
        for (int i = 0; i < samples; i++)
        {
            double flight = minFlight + (maxFlight - minFlight) * i / Math.Max(samples - 1, 1);
            double launch = interceptTime - flight;
            if (launch < 0 || launch > context.WarningDays) continue;

            var earthState = context.Ephemeris.SampleState(context.EarthIndex, launch);
            var asteroidState = context.Ephemeris.SampleState(context.AsteroidIndex, interceptTime);
            double flightSeconds = flight * SecondsPerDay;
            if (flightSeconds <= 0) continue;

            Vector3d requiredVelocity = (asteroidState.Position - earthState.Position) / flightSeconds;
            Vector3d dv = requiredVelocity - earthState.Velocity;
            options.Add(new FlightOption
            {
                LaunchTime = launch,
                FlightTime = flight,
                InitialDv = dv,
                InitialDvMag = dv.Length
            });
        }

        options.Sort((a, b) => a.InitialDvMag.CompareTo(b.InitialDvMag));
        var selected = new List<FlightOption>();
        foreach (var option in options)
        {
            selected.Add(option);
            if (selected.Count >= 3) break;
        }

        FlightOption? longest = null;
        foreach (var option in options)
        {
            if (longest == null || option.FlightTime > longest.Value.FlightTime)
            {
                longest = option;
            }
        }

        if (longest.HasValue)
        {
            bool already = false;
            foreach (var sel in selected)
            {
                if (Math.Abs(sel.LaunchTime - longest.Value.LaunchTime) < 0.1)
                {
                    already = true;
                    break;
                }
            }

            if (!already)
            {
                selected.Add(longest.Value);
            }
        }

        return selected;
    }

    private static List<Burn> ComposeBurns(MissionContext context, Vector3d launchDv, List<Correction> corrections, int overrideIndex, Vector3d overrideDelta)
    {
        return ComposeBurns(context.LaunchWindowStart, launchDv, corrections, overrideIndex, overrideDelta);
    }

    private static List<Burn> ComposeBurns(double launchTime, Vector3d launchDv, List<Correction> corrections, int overrideIndex, Vector3d overrideDelta)
    {
        var list = new List<Burn> { new Burn(launchTime, launchDv) };
        for (int i = 0; i < corrections.Count; i++)
        {
            Vector3d dv = corrections[i].DeltaV;
            if (i == overrideIndex)
            {
                dv += overrideDelta;
            }

            if (dv.LengthSquared > 1e-10)
            {
                list.Add(new Burn(corrections[i].Time, dv));
            }
        }

        list.Sort((a, b) => a.Time.CompareTo(b.Time));
        var merged = new List<Burn>();
        const double timeTol = 1e-9;

        foreach (var burn in list)
        {
            if (merged.Count > 0 && Math.Abs(merged[^1].Time - burn.Time) < timeTol)
            {
                var last = merged[^1];
                last.DeltaV += burn.DeltaV;
                merged[^1] = last;
            }
            else
            {
                merged.Add(burn);
            }
        }

        return merged;
    }
}

internal readonly struct StageBudget
{
    public StageBudget(double launch, double midCourse, double impactReserve)
    {
        Launch = launch;
        MidCourse = midCourse;
        ImpactReserve = impactReserve;

internal sealed class TrajectorySimulator
{
    private readonly Ephemeris _ephemeris;
    private readonly int _earthIndex;
    private readonly int _asteroidIndex;
    private readonly int _sunIndex;
    private readonly double _stepDays;
    private readonly double _launchAltitude;
    private readonly int _bodyCount;
    private readonly double[] _masses;

    private const double TimeTolerance = 1e-9;
    private const double Softening = 1.0;

    public TrajectorySimulator(Ephemeris ephemeris, int earthIndex, int asteroidIndex, int sunIndex, double stepDays, double launchAltitude)
    {
        _ephemeris = ephemeris;
        _earthIndex = earthIndex;
        _asteroidIndex = asteroidIndex;
        _sunIndex = sunIndex;
        _stepDays = Math.Max(stepDays, 1e-4);
        _launchAltitude = launchAltitude;
        _bodyCount = ephemeris.BodyCount;
        _masses = ephemeris.Masses;
    }

    public SimulationResult Run(IReadOnlyList<Burn> burns, double endTimeDays, IReadOnlyList<double> sampleTimes)
    {
        var result = new SimulationResult();
        double currentTime = 0.0;
        bool spacecraftActive = false;
        Vector3d scPos = Vector3d.Zero;
        Vector3d scVel = Vector3d.Zero;
        double totalDv = 0.0;

        var burnList = burns != null ? new List<Burn>(burns) : new List<Burn>();
        burnList.Sort((a, b) => a.Time.CompareTo(b.Time));
        int burnIndex = 0;

        List<double> sampleList = null;
        if (sampleTimes != null && sampleTimes.Count > 0)
        {
            sampleList = new List<double>();
            foreach (double t in sampleTimes)
            {
                if (t >= 0 && t <= endTimeDays + 1e-6)
                {
                    sampleList.Add(t);
                }
            }

            sampleList.Sort();
            for (int i = sampleList.Count - 1; i > 0; i--)
            {
                if (Math.Abs(sampleList[i] - sampleList[i - 1]) < 1e-9)
                {
                    sampleList.RemoveAt(i);
                }
            }
        }

        int sampleIndex = 0;
        double closest = double.MaxValue;
        double timeClosest = 0.0;

        var sampler = _ephemeris.CreateSampler();
        var startPositions = new Vector3d[_bodyCount];
        var endPositions = new Vector3d[_bodyCount];

        while (currentTime < endTimeDays - TimeTolerance)
        {
            while (burnIndex < burnList.Count && Math.Abs(burnList[burnIndex].Time - currentTime) <= TimeTolerance)
            {
                var burn = burnList[burnIndex];
                if (!spacecraftActive)
                {
                    var earthState = _ephemeris.SampleState(_earthIndex, burn.Time);
                    Vector3d offset = BuildLaunchOffset(burn.Time, earthState.Position, earthState.Velocity);
                    scPos = earthState.Position + offset;
                    scVel = earthState.Velocity;
                    spacecraftActive = true;
                }

                scVel += burn.DeltaV;
                totalDv += burn.DeltaV.Length;
                burnIndex++;
            }

            if (sampleList != null)
            {
                while (sampleIndex < sampleList.Count && Math.Abs(sampleList[sampleIndex] - currentTime) <= TimeTolerance)
                {
                    if (spacecraftActive)
                    {
                        result.RecordSample(sampleList[sampleIndex], scPos, scVel);
                    }

                    sampleIndex++;
                }
            }

            double nextEvent = endTimeDays;
            if (burnIndex < burnList.Count)
                nextEvent = Math.Min(nextEvent, burnList[burnIndex].Time);
            if (sampleList != null && sampleIndex < sampleList.Count)
                nextEvent = Math.Min(nextEvent, sampleList[sampleIndex]);

            double step = _stepDays;
            if (currentTime + step > nextEvent - TimeTolerance)
            {
                step = Math.Max(nextEvent - currentTime, 1e-6);
            }

            if (currentTime + step > endTimeDays)
            {
                step = endTimeDays - currentTime;
            }

            if (step <= 1e-9)
            {
                currentTime += step;
                continue;
            }

            if (!spacecraftActive)
            {
                sampler.Evaluate(currentTime + step, startPositions);
                currentTime += step;
                continue;
            }

            double stepSeconds = step * SecondsPerDay;
            sampler.Evaluate(currentTime, startPositions);
            Vector3d accStart = ComputeAcceleration(scPos, startPositions);
            scVel += accStart * (0.5 * stepSeconds);
            scPos += scVel * stepSeconds;

            sampler.Evaluate(currentTime + step, endPositions);
            Vector3d accEnd = ComputeAcceleration(scPos, endPositions);
            scVel += accEnd * (0.5 * stepSeconds);

            Vector3d asteroidPos = endPositions[_asteroidIndex];
            double dist = (scPos - asteroidPos).Length;
            if (dist < closest)
            {
                closest = dist;
                timeClosest = currentTime + step;
            }

            currentTime += step;
        }

        while (burnIndex < burnList.Count && Math.Abs(burnList[burnIndex].Time - currentTime) <= TimeTolerance)
        {
            var burn = burnList[burnIndex];
            if (!spacecraftActive)
            {
                var earthState = _ephemeris.SampleState(_earthIndex, burn.Time);
                Vector3d offset = BuildLaunchOffset(burn.Time, earthState.Position, earthState.Velocity);
                scPos = earthState.Position + offset;
                scVel = earthState.Velocity;
                spacecraftActive = true;
            }

            scVel += burn.DeltaV;
            totalDv += burn.DeltaV.Length;
            burnIndex++;
        }

        result.SpacecraftActive = spacecraftActive;
        if (spacecraftActive)
        {
            result.SpacecraftPosition = scPos;
            result.SpacecraftVelocity = scVel;
        }

        var asteroidState = _ephemeris.SampleState(_asteroidIndex, endTimeDays);
        result.AsteroidPosition = asteroidState.Position;
        result.AsteroidVelocity = asteroidState.Velocity;
        result.FinalSeparation = spacecraftActive ? (scPos - asteroidState.Position).Length : double.PositiveInfinity;
        result.TotalDeltaV = totalDv;
        result.ClosestApproach = closest;
        result.TimeOfClosestApproach = timeClosest;

        return result;
    }

    private Vector3d ComputeAcceleration(Vector3d scPos, Vector3d[] bodyPositions)
    {
        Vector3d acc = Vector3d.Zero;
        for (int i = 0; i < _bodyCount; i++)
        {
            Vector3d diff = bodyPositions[i] - scPos;
            double distSqr = diff.LengthSquared + Softening;
            double invDist = 1.0 / Math.Sqrt(distSqr);
            double invDist3 = invDist * invDist * invDist;
            double scale = Ephemeris.GravitationalConstant * _masses[i] * invDist3;
            acc += diff * scale;
        }

        return acc;
    }

    private Vector3d BuildLaunchOffset(double time, Vector3d earthPos, Vector3d earthVel)
    {
        Vector3d dir = earthVel.LengthSquared > 1e-12 ? earthVel : new Vector3d(1, 0, 0);
        if (_sunIndex >= 0)
        {
            Vector3d sunPos = _ephemeris.SamplePosition(_sunIndex, time);
            Vector3d radial = earthPos - sunPos;
            if (radial.LengthSquared > 1e-6)
            {
                dir = radial;
            }
        }

        return dir.Normalized() * _launchAltitude;
    }
}

internal sealed class Ephemeris
{
    public const double GravitationalConstant = 6.67430e-11;

    public double[] Times { get; private set; }
    public Vector3d[][] Positions { get; private set; }
    public Vector3d[][] Velocities { get; private set; }
    public double[] Masses { get; private set; }
    public int BodyCount => Masses.Length;

    public static Ephemeris Build(BodyDefinition[] bodies, double durationDays, double stepDays)
    {
        if (stepDays <= 0) stepDays = 0.25;
        var working = new BodyDefinition[bodies.Length];
        for (int i = 0; i < bodies.Length; i++)
        {
            working[i] = bodies[i].Clone();
        }

        var timeSamples = new List<double> { 0.0 };
        var posHistory = new List<Vector3d>[bodies.Length];
        var velHistory = new List<Vector3d>[bodies.Length];
        for (int i = 0; i < bodies.Length; i++)
        {
            posHistory[i] = new List<Vector3d> { working[i].Position };
            velHistory[i] = new List<Vector3d> { working[i].Velocity };
        }

        double currentTime = 0.0;
        double[] masses = new double[bodies.Length];
        for (int i = 0; i < bodies.Length; i++)
        {
            masses[i] = bodies[i].Mass;
        }

        var acc = new Vector3d[bodies.Length];
        while (currentTime < durationDays - 1e-9)
        {
            double step = Math.Min(stepDays, durationDays - currentTime);
            double stepSeconds = step * SecondsPerDay;
            Array.Clear(acc, 0, acc.Length);
            ComputeAccelerations(working, masses, acc);

            for (int i = 0; i < working.Length; i++)
            {
                working[i].Velocity += acc[i] * (0.5 * stepSeconds);
            }

            for (int i = 0; i < working.Length; i++)
            {
                working[i].Position += working[i].Velocity * stepSeconds;
            }

            Array.Clear(acc, 0, acc.Length);
            ComputeAccelerations(working, masses, acc);

            for (int i = 0; i < working.Length; i++)
            {
                working[i].Velocity += acc[i] * (0.5 * stepSeconds);
            }

            currentTime += step;
            timeSamples.Add(currentTime);
            for (int i = 0; i < working.Length; i++)
            {
                posHistory[i].Add(working[i].Position);
                velHistory[i].Add(working[i].Velocity);
            }
        }

        var ephemeris = new Ephemeris
        {
            Times = timeSamples.ToArray(),
            Positions = new Vector3d[bodies.Length][],
            Velocities = new Vector3d[bodies.Length][],
            Masses = masses
        };

        for (int i = 0; i < bodies.Length; i++)
        {
            ephemeris.Positions[i] = posHistory[i].ToArray();
            ephemeris.Velocities[i] = velHistory[i].ToArray();
        }

        return ephemeris;
    }

    public BodyState SampleState(int bodyIndex, double time)
    {
        var times = Times;
        if (times.Length == 1)
        {
            return new BodyState
            {
                Position = Positions[bodyIndex][0],
                Velocity = Velocities[bodyIndex][0]
            };
        }

        if (time <= times[0])
        {
            return new BodyState
            {
                Position = Positions[bodyIndex][0],
                Velocity = Velocities[bodyIndex][0]
            };
        }

        if (time >= times[^1])
        {
            int last = times.Length - 1;
            return new BodyState
            {
                Position = Positions[bodyIndex][last],
                Velocity = Velocities[bodyIndex][last]
            };
        }

        int low = 0;
        int high = times.Length - 1;
        while (high - low > 1)
        {
            int mid = (low + high) / 2;
            if (times[mid] <= time) low = mid;
            else high = mid;
        }

        double t0 = times[low];
        double t1 = times[high];
        double alpha = (time - t0) / (t1 - t0);

        Vector3d pos = Vector3d.Lerp(Positions[bodyIndex][low], Positions[bodyIndex][high], alpha);
        Vector3d vel = Vector3d.Lerp(Velocities[bodyIndex][low], Velocities[bodyIndex][high], alpha);

        return new BodyState { Position = pos, Velocity = vel };
    }

    public Vector3d SamplePosition(int bodyIndex, double time) => SampleState(bodyIndex, time).Position;

    public Vector3d SampleVelocity(int bodyIndex, double time) => SampleState(bodyIndex, time).Velocity;

    public EphemerisSampler CreateSampler() => new EphemerisSampler(this);

    private static void ComputeAccelerations(BodyDefinition[] bodies, double[] masses, Vector3d[] acc)
    {
        for (int i = 0; i < bodies.Length; i++)
        {
            acc[i] = Vector3d.Zero;
        }

        for (int i = 0; i < bodies.Length; i++)
        {
            for (int j = i + 1; j < bodies.Length; j++)
            {
                Vector3d diff = bodies[j].Position - bodies[i].Position;
                double distSqr = diff.LengthSquared + 100.0;
                double dist = Math.Sqrt(distSqr);
                double invDist3 = 1.0 / (distSqr * dist);
                Vector3d force = diff * (GravitationalConstant * invDist3);
                acc[i] += force * masses[j];
                acc[j] -= force * masses[i];
            }
        }
    }
}

internal sealed class EphemerisSampler
{
    private readonly Ephemeris _ephemeris;
    private int _index;

    public EphemerisSampler(Ephemeris ephemeris)
    {
        _ephemeris = ephemeris;
        _index = 0;
    }

    public void Evaluate(double time, Vector3d[] output)
    {
        var times = _ephemeris.Times;
        if (times.Length == 1)
        {
            for (int i = 0; i < output.Length; i++)
            {
                output[i] = _ephemeris.Positions[i][0];
            }

            return;
        }

        if (time <= times[0])
            _index = 0;
        else
        {
            while (_index + 1 < times.Length && times[_index + 1] <= time)
                _index++;
            while (_index > 0 && times[_index] > time)
                _index--;
        }

        int nextIndex = Math.Min(_index + 1, times.Length - 1);
        double t0 = times[_index];
        double t1 = times[nextIndex];
        double alpha = t1 > t0 ? (time - t0) / (t1 - t0) : 0.0;

        for (int body = 0; body < _ephemeris.BodyCount; body++)
        {
            output[body] = Vector3d.Lerp(_ephemeris.Positions[body][_index], _ephemeris.Positions[body][nextIndex], alpha);
        }
    }
}

internal static class ImpactAnalyzer
{
    public static ImpactInfo Analyze(Ephemeris ephemeris, int earthIndex, int asteroidIndex, double warningDays)
    {
        var times = ephemeris.Times;
        var earthPositions = ephemeris.Positions[earthIndex];
        var asteroidPositions = ephemeris.Positions[asteroidIndex];

        double minDist = double.MaxValue;
        int bestIndex = 0;
        for (int i = 0; i < times.Length; i++)
        {
            double dist = (asteroidPositions[i] - earthPositions[i]).Length;
            if (dist < minDist)
            {
                minDist = dist;
                bestIndex = i;
            }
        }

        double baseTime = times[Math.Max(0, Math.Min(bestIndex, times.Length - 1))];
        double window = Math.Max(5.0, warningDays * 0.02);
        double start = Math.Max(0.0, baseTime - window);
        double end = Math.Min(warningDays, baseTime + window);

        double refinedTime = baseTime;
        for (int i = 0; i <= 200; i++)
        {
            double t = start + (end - start) * i / 200.0;
            var earthState = ephemeris.SampleState(earthIndex, t);
            var asteroidState = ephemeris.SampleState(asteroidIndex, t);
            double dist = (asteroidState.Position - earthState.Position).Length;
            if (dist < minDist)
            {
                minDist = dist;
                refinedTime = t;
            }
        }

        var earthAtImpact = ephemeris.SampleState(earthIndex, refinedTime);
        var asteroidAtImpact = ephemeris.SampleState(asteroidIndex, refinedTime);
        Vector3d relVel = asteroidAtImpact.Velocity - earthAtImpact.Velocity;

        return new ImpactInfo
        {
            Time = refinedTime,
            Distance = minDist,
            RelativeSpeed = relVel.Length
        };
    }
}

    {
        double len = Length;
        if (len < 1e-12) return Zero;
        return this / len;
    }

    public static Vector3d operator -(Vector3d a, Vector3d b) => new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3d operator *(Vector3d a, double scalar) => new Vector3d(a.X * scalar, a.Y * scalar, a.Z * scalar);
    public static Vector3d operator *(double scalar, Vector3d a) => a * scalar;
    public static Vector3d operator /(Vector3d a, double scalar) => new Vector3d(a.X / scalar, a.Y / scalar, a.Z / scalar);

    public static double Dot(Vector3d a, Vector3d b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

    public static Vector3d Cross(Vector3d a, Vector3d b)
    {
        return new Vector3d(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    public static Vector3d Lerp(Vector3d a, Vector3d b, double t)
    {
        return new Vector3d(
            a.X + (b.X - a.X) * t,
            a.Y + (b.Y - a.Y) * t,
            a.Z + (b.Z - a.Z) * t
        );
    }
}

internal static class LinearSolver
{
    public static bool Solve3x3(double[,] matrix, double[] rhs, out Vector3d solution)
    {
        double[,] m = new double[3, 4];
        for (int r = 0; r < 3; r++)
        {
            for (int c = 0; c < 3; c++)
            {
                m[r, c] = matrix[r, c];
            }

            m[r, 3] = rhs[r];
        }

        for (int pivot = 0; pivot < 3; pivot++)
        {
            int maxRow = pivot;
            double maxVal = Math.Abs(m[pivot, pivot]);
            for (int row = pivot + 1; row < 3; row++)
            {
                double val = Math.Abs(m[row, pivot]);
                if (val > maxVal)
                {
                    maxVal = val;
                    maxRow = row;
                }
            }

            if (maxVal < 1e-9)
            {
                solution = Vector3d.Zero;
                return false;
            }

            if (maxRow != pivot)
            {
                SwapRows(m, pivot, maxRow);
            }

            double pivotVal = m[pivot, pivot];
            for (int col = pivot; col < 4; col++)
            {
                m[pivot, col] /= pivotVal;
            }

            for (int row = 0; row < 3; row++)
            {
                if (row == pivot) continue;
                double factor = m[row, pivot];
                for (int col = pivot; col < 4; col++)
                {
                    m[row, col] -= factor * m[pivot, col];
                }
            }
        }

        solution = new Vector3d(m[0, 3], m[1, 3], m[2, 3]);
        return true;
    }

    private static void SwapRows(double[,] matrix, int rowA, int rowB)
    {
        for (int col = 0; col < 4; col++)
        {
            double tmp = matrix[rowA, col];
            matrix[rowA, col] = matrix[rowB, col];
            matrix[rowB, col] = tmp;
        }
    }
}

internal sealed class FastReader
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _length;
    private int _position;

    public FastReader(Stream stream)
    {
        _stream = stream;
    }

    public string Next()
    {
        var sb = new StringBuilder();
        int c;
        while ((c = Read()) != -1 && char.IsWhiteSpace((char)c)) { }
        if (c == -1) return null;
        do
        {
            sb.Append((char)c);
        } while ((c = Read()) != -1 && !char.IsWhiteSpace((char)c));
        return sb.ToString();
    }

    public double NextDouble()
    {
        string token = Next();
        return token == null ? 0.0 : double.Parse(token, CultureInfo.InvariantCulture);
    }

    private int Read()
    {
        if (_position >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _position = 0;
            if (_length <= 0) return -1;
        }

        return _buffer[_position++];
    }
}
