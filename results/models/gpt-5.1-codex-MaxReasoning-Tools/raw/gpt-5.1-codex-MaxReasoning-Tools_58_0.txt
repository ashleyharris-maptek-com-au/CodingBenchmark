```javascript
'use strict';

const fs = require('fs');

(function main() {
    const input = fs.readFileSync(0, 'utf8');
    const data = input.trim().length ? JSON.parse(input) : {};

    const rawGrid = Array.isArray(data.grid) ? data.grid : [];
    const rawRobots = Array.isArray(data.robots) ? data.robots : [];
    const rawJobs = Array.isArray(data.jobs) ? data.jobs : [];

    const robots = [];
    const seenRobotIds = new Set();
    for (const entry of rawRobots) {
        const robot = normalizeRobot(entry);
        if (!robot || seenRobotIds.has(robot.id)) continue;
        seenRobotIds.add(robot.id);
        robots.push(robot);
    }

    const jobs = rawJobs.map(normalizeJob);

    if (robots.length === 0) {
        process.stdout.write(JSON.stringify({ assignments: [] }));
        return;
    }

    const width = rawGrid.reduce(
        (max, row) => Math.max(max, typeof row === 'string' ? row.length : 0),
        0
    );
    const height = rawGrid.length;

    if (width === 0 || height === 0) {
        const assignmentsFallback = assignWithoutGrid(robots, jobs);
        process.stdout.write(JSON.stringify({ assignments: assignmentsFallback }));
        return;
    }

    const openMask = buildOpenMask(rawGrid, width, height);
    markPointsAsOpen(robots, openMask, width, height);
    markPointsAsOpen(jobs, openMask, width, height);

    const { owner, dist } = buildVoronoi(width, height, openMask, robots);

    let maxRobotId = -1;
    for (const robot of robots) {
        if (robot.id > maxRobotId) maxRobotId = robot.id;
    }

    const assignments = new Array(maxRobotId + 1);
    const jobsPerRobot = new Array(maxRobotId + 1);
    const robotInfo = new Array(maxRobotId + 1);
    for (let i = 0; i <= maxRobotId; i++) {
        assignments[i] = [];
        jobsPerRobot[i] = [];
        robotInfo[i] = null;
    }

    const hilbertSize = nextPowerOfTwo(Math.max(width, height));
    const hilbertRange = Math.max(1, hilbertSize * hilbertSize);
    const shiftBase = hilbertRange + 1;

    for (const robot of robots) {
        const hx = clamp(robot.x, 0, hilbertSize - 1);
        const hy = clamp(robot.y, 0, hilbertSize - 1);
        const hilbertValue = hilbertIndex(hx, hy, hilbertSize);
        robotInfo[robot.id] = { id: robot.id, x: robot.x, y: robot.y, hilbert: hilbertValue };
    }

    const priorityStats = computePriorityStats(jobs);
    const bucketCount = 12;

    for (const job of jobs) {
        const hx = clamp(job.x, 0, hilbertSize - 1);
        const hy = clamp(job.y, 0, hilbertSize - 1);
        job._hilbert = hilbertIndex(hx, hy, hilbertSize);
        job._bucket = computeBucket(
            job.priority,
            priorityStats.minPriority,
            priorityStats.range,
            bucketCount
        );

        const inBounds = job.x >= 0 && job.x < width && job.y >= 0 && job.y < height;
        const idx = inBounds ? job.y * width + job.x : -1;
        let ownerId = idx >= 0 ? owner[idx] : -1;
        let distVal = idx >= 0 ? dist[idx] : -1;

        if (ownerId < 0 || ownerId > maxRobotId || !robotInfo[ownerId]) {
            ownerId = selectNearestRobot(job, robots);
            distVal = -1;
        }

        let distanceScore = distVal >= 0 ? distVal : Number.MAX_SAFE_INTEGER;
        if (distanceScore === Number.MAX_SAFE_INTEGER) {
            const ref = robotInfo[ownerId];
            if (ref) {
                distanceScore = Math.abs(ref.x - job.x) + Math.abs(ref.y - job.y);
            }
        }
        job._dist = distanceScore;
        jobsPerRobot[ownerId].push(job);
    }

    for (let robotId = 0; robotId <= maxRobotId; robotId++) {
        const list = jobsPerRobot[robotId];
        if (!list || list.length === 0) {
            assignments[robotId] = [];
            continue;
        }
        const info = robotInfo[robotId];
        const anchor = info ? info.hilbert : 0;
        list.sort((a, b) => {
            const relA = relativeHilbert(a._hilbert, anchor, hilbertRange);
            const relB = relativeHilbert(b._hilbert, anchor, hilbertRange);
            const scoreA = a._bucket * shiftBase + relA;
            const scoreB = b._bucket * shiftBase + relB;
            if (scoreA !== scoreB) return scoreA - scoreB;
            if (a._dist !== b._dist) return a._dist - b._dist;
            if (a.priority !== b.priority) return b.priority - a.priority;
            return compareIds(a.id, b.id);
        });
        assignments[robotId] = list.map(job => job.id);
    }

    process.stdout.write(JSON.stringify({ assignments }));
})();

function normalizeRobot(robot) {
    if (!robot || typeof robot !== 'object') return null;
    const idNum = Number(robot.id);
    if (!Number.isFinite(idNum)) return null;
    const id = Math.trunc(idNum);
    if (id < 0) return null;
    return { id, x: toCoord(robot.x), y: toCoord(robot.y) };
}

function normalizeJob(job) {
    if (!job || typeof job !== 'object') {
        return { id: undefined, x: 0, y: 0, priority: 0 };
    }
    return {
        id: job.id,
        x: toCoord(job.x),
        y: toCoord(job.y),
        priority: toNumber(job.priority)
    };
}

function toCoord(value) {
    const num = Number(value);
    if (!Number.isFinite(num)) return 0;
    return Math.trunc(num);
}

function toNumber(value) {
    const num = Number(value);
    return Number.isFinite(num) ? num : 0;
}

function buildOpenMask(grid, width, height) {
    const mask = new Uint8Array(width * height);
    for (let y = 0; y < height; y++) {
        const row = typeof grid[y] === 'string' ? grid[y] : '';
        for (let x = 0; x < width; x++) {
            const ch = row.charCodeAt(x);
            if (ch === 46) {
                mask[y * width + x] = 1;
            }
        }
    }
    return mask;
}

function markPointsAsOpen(points, mask, width, height) {
    if (!points) return;
    for (const point of points) {
        if (!point) continue;
        const x = point.x;
        const y = point.y;
        if (x >= 0 && x < width && y >= 0 && y < height) {
            mask[y * width + x] = 1;
        }
    }
}

function buildVoronoi(width, height, openMask, robots) {
    const total = width * height;
    const owner = new Int32Array(total);
    const dist = new Int32Array(total);
    owner.fill(-1);
    dist.fill(-1);
    if (total === 0 || robots.length === 0) return { owner, dist };

    const queue = new Int32Array(total);
    let head = 0;
    let tail = 0;

    const sorted = robots.slice().sort((a, b) => a.id - b.id);
    for (const robot of sorted) {
        const x = robot.x;
        const y = robot.y;
        if (x < 0 || x >= width || y < 0 || y >= height) continue;
        const idx = y * width + x;
        if (!openMask[idx] || dist[idx] !== -1) continue;
        owner[idx] = robot.id;
        dist[idx] = 0;
        queue[tail++] = idx;
    }

    const dirs = [1, 0, -1, 0, 1];
    while (head < tail) {
        const current = queue[head++];
        const cx = current % width;
        const cy = (current / width) | 0;
        const baseOwner = owner[current];
        const baseDist = dist[current];
        for (let d = 0; d < 4; d++) {
            const nx = cx + dirs[d];
            const ny = cy + dirs[d + 1];
            if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
            const nIdx = ny * width + nx;
            if (!openMask[nIdx] || dist[nIdx] !== -1) continue;
            owner[nIdx] = baseOwner;
            dist[nIdx] = baseDist + 1;
            queue[tail++] = nIdx;
        }
    }

    return { owner, dist };
}

function nextPowerOfTwo(n) {
    let size = 1;
    while (size < n) size *= 2;
    return size;
}

function clamp(value, min, max) {
    if (max < min) return min;
    if (value < min) return min;
    if (value > max) return max;
    return value;
}

function hilbertIndex(x, y, size) {
    let index = 0;
    for (let s = size >> 1; s > 0; s >>= 1) {
        const rx = (x & s) ? 1 : 0;
        const ry = (y & s) ? 1 : 0;
        index += s * s * ((3 * rx) ^ ry);
        if (ry === 0) {
            if (rx === 1) {
                x = s - 1 - x;
                y = s - 1 - y;
            }
            const temp = x;
            x = y;
            y = temp;
        }
    }
    return index;
}

function relativeHilbert(value, anchor, range) {
    let diff = value - anchor;
    if (diff < 0) diff += range;
    return diff;
}

function computePriorityStats(jobs) {
    if (jobs.length === 0) return { minPriority: 0, maxPriority: 0, range: 0 };
    let min = Infinity;
    let max = -Infinity;
    for (const job of jobs) {
        const p = job.priority;
        if (p < min) min = p;
        if (p > max) max = p;
    }
    if (!Number.isFinite(min)) min = 0;
    if (!Number.isFinite(max)) max = 0;
    return { minPriority: min, maxPriority: max, range: max - min };
}

function computeBucket(value, min, range, bucketCount) {
    if (bucketCount <= 1 || !Number.isFinite(range) || range === 0) return 0;
    let normalized = (value - min) / range;
    if (normalized < 0) normalized = 0;
    if (normalized > 0.999999) normalized = 0.999999;
    let bucket = bucketCount - 1 - Math.floor(normalized * bucketCount);
    if (bucket < 0) bucket = 0;
    if (bucket >= bucketCount) bucket = bucketCount - 1;
    return bucket;
}

function compareIds(a, b) {
    const aStr = a === undefined ? '' : String(a);
    const bStr = b === undefined ? '' : String(b);
    if (aStr < bStr) return -1;
    if (aStr > bStr) return 1;
    return 0;
}

function selectNearestRobot(job, robots) {
    if (robots.length === 0) return 0;
    let bestId = robots[0].id;
    let bestDist = Infinity;
    for (const robot of robots) {
        const d = Math.abs(robot.x - job.x) + Math.abs(robot.y - job.y);
        if (d < bestDist || (d === bestDist && robot.id < bestId)) {
            bestDist = d;
            bestId = robot.id;
        }
    }
    return bestId;
}

function assignWithoutGrid(robots, jobs) {
    let maxId = -1;
    for (const robot of robots) {
        if (robot.id > maxId) maxId = robot.id;
    }
    if (maxId < 0) return [];
    const assignments = new Array(maxId + 1);
    for (let i = 0; i <= maxId; i++) assignments[i] = [];
    if (robots.length === 0) return assignments;

    const robotIds = robots.map(r => r.id).sort((a, b) => a - b);
    const orderedJobs = jobs.slice().sort((a, b) => {
        if (a.priority !== b.priority) return b.priority - a.priority;
        return compareIds(a.id, b.id);
    });

    for (let i = 0; i < orderedJobs.length; i++) {
        const rid = robotIds[i % robotIds.length];
        assignments[rid].push(orderedJobs[i].id);
    }
    return assignments;
}
```
