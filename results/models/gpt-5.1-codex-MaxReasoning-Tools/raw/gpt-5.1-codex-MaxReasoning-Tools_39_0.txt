Here's a complete Rust program that finds an exact minimum bisection: it first builds a good initial partition via greedy + Kernighan–Lin style swaps (to get a tight upper bound) and then runs an exact branch-and-bound search with symmetry breaking and admissible lower bounds to guarantee optimality.

```rust
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut iter = input.split_whitespace();
    let n: usize = match iter.next() {
        Some(token) => token.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = iter.next().and_then(|t| t.parse().ok()).unwrap_or(0);

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut adjacency = vec![Vec::<usize>::new(); n];
    let mut adj_matrix = vec![vec![0u32; n]; n];
    let mut edges: Vec<(usize, usize)> = Vec::new();

    for _ in 0..m {
        if let (Some(us), Some(vs)) = (iter.next(), iter.next()) {
            let u = match us.parse::<usize>() {
                Ok(val) => val,
                Err(_) => continue,
            };
            let v = match vs.parse::<usize>() {
                Ok(val) => val,
                Err(_) => continue,
            };
            if u >= n || v >= n || u == v {
                continue;
            }
            adjacency[u].push(v);
            adjacency[v].push(u);
            adj_matrix[u][v] += 1;
            adj_matrix[v][u] += 1;
            edges.push((u, v));
        } else {
            break;
        }
    }

    let (heur_assignment, heur_cut) = run_heuristics(n, &adjacency, &adj_matrix, &edges);

    if heur_cut == 0 {
        #[cfg(debug_assertions)]
        {
            let check = compute_cut(&heur_assignment, &edges);
            debug_assert_eq!(check, heur_cut);
        }
        output_result(heur_cut, &heur_assignment);
        return;
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| {
        adjacency[b]
            .len()
            .cmp(&adjacency[a].len())
            .then_with(|| a.cmp(&b))
    });

    let mut solver = BranchAndBound::new(adjacency, order, heur_cut, heur_assignment);
    solver.solve();
    let (final_cut, final_assignment) = solver.into_result();

    #[cfg(debug_assertions)]
    {
        let check = compute_cut(&final_assignment, &edges);
        debug_assert_eq!(check, final_cut);
    }

    output_result(final_cut, &final_assignment);
}

fn run_heuristics(
    n: usize,
    adjacency: &Vec<Vec<usize>>,
    adj_matrix: &Vec<Vec<u32>>,
    edges: &[(usize, usize)],
) -> (Vec<i8>, usize) {
    if n == 0 {
        return (Vec::new(), 0);
    }
    let half = n / 2;
    let mut orders: Vec<Vec<usize>> = Vec::new();

    let mut deg_order: Vec<usize> = (0..n).collect();
    deg_order.sort_by(|&a, &b| {
        adjacency[b]
            .len()
            .cmp(&adjacency[a].len())
            .then_with(|| a.cmp(&b))
    });
    add_order(&mut orders, deg_order.clone());
    let mut deg_order_rev = deg_order.clone();
    deg_order_rev.reverse();
    add_order(&mut orders, deg_order_rev);

    let natural: Vec<usize> = (0..n).collect();
    add_order(&mut orders, natural.clone());
    let mut natural_rev = natural.clone();
    natural_rev.reverse();
    add_order(&mut orders, natural_rev);

    if n > 2 {
        let mut rotated = deg_order.clone();
        rotated.rotate_left(n / 2);
        add_order(&mut orders, rotated);
    }

    if orders.is_empty() {
        orders.push((0..n).collect());
    }

    let mut best_side = vec![0i8; n];
    let mut best_cut = usize::MAX;
    let mut found = false;

    for order in orders.iter() {
        let mut candidate = greedy_from_order(order, adjacency, half);
        candidate = local_optimize(candidate, adjacency, adj_matrix);
        let cut = compute_cut(&candidate, edges);
        if cut < best_cut {
            best_cut = cut;
            best_side = candidate;
            found = true;
        }
    }

    if !found {
        let mut fallback = vec![0i8; n];
        for i in 0..half {
            fallback[i] = 1;
        }
        let cut = compute_cut(&fallback, edges);
        return (fallback, cut);
    }

    (best_side, best_cut)
}

fn add_order(orders: &mut Vec<Vec<usize>>, order: Vec<usize>) {
    if !orders.iter().any(|existing| existing == &order) {
        orders.push(order);
    }
}

fn greedy_from_order(order: &[usize], adjacency: &Vec<Vec<usize>>, half: usize) -> Vec<i8> {
    let n = adjacency.len();
    let mut side = vec![-1i8; n];
    let mut count_a = 0usize;
    let mut count_b = 0usize;
    for &v in order {
        if v >= n || side[v] != -1 {
            continue;
        }
        if count_a == half {
            side[v] = 0;
            count_b += 1;
            continue;
        }
        if count_b == half {
            side[v] = 1;
            count_a += 1;
            continue;
        }
        let mut to_a = 0usize;
        let mut to_b = 0usize;
        for &nbr in &adjacency[v] {
            match side[nbr] {
                1 => to_a += 1,
                0 => to_b += 1,
                _ => {}
            }
        }
        if to_a > to_b {
            side[v] = 0;
            count_b += 1;
        } else if to_b > to_a {
            side[v] = 1;
            count_a += 1;
        } else if count_a <= count_b {
            side[v] = 1;
            count_a += 1;
        } else {
            side[v] = 0;
            count_b += 1;
        }
    }
    for v in 0..n {
        if side[v] == -1 {
            if count_a < half {
                side[v] = 1;
                count_a += 1;
            } else {
                side[v] = 0;
                count_b += 1;
            }
        }
    }
    side
}

fn local_optimize(
    mut side: Vec<i8>,
    adjacency: &Vec<Vec<usize>>,
    adj_matrix: &Vec<Vec<u32>>,
) -> Vec<i8> {
    let n = side.len();
    if n == 0 {
        return side;
    }
    loop {
        let mut d = vec![0isize; n];
        for v in 0..n {
            let mut same = 0usize;
            let mut diff = 0usize;
            for &nbr in &adjacency[v] {
                if side[nbr] == side[v] {
                    same += 1;
                } else {
                    diff += 1;
                }
            }
            d[v] = diff as isize - same as isize;
        }
        let mut best_gain = 0isize;
        let mut best_pair: Option<(usize, usize)> = None;
        let mut has_a = false;
        let mut has_b = false;
        for a in 0..n {
            if side[a] != 1 {
                continue;
            }
            has_a = true;
            let da = d[a];
            for b in 0..n {
                if side[b] != 0 {
                    continue;
                }
                has_b = true;
                let db = d[b];
                let w = adj_matrix[a][b] as isize;
                let gain = da + db - 2 * w;
                if gain > best_gain {
                    best_gain = gain;
                    best_pair = Some((a, b));
                }
            }
        }
        if !has_a || !has_b {
            break;
        }
        if best_gain > 0 {
            if let Some((a, b)) = best_pair {
                side[a] = 0;
                side[b] = 1;
                continue;
            }
        }
        break;
    }
    side
}

fn compute_cut(partition: &[i8], edges: &[(usize, usize)]) -> usize {
    edges
        .iter()
        .filter(|&&(u, v)| partition[u] != partition[v])
        .count()
}

fn output_result(cut: usize, assignment: &[i8]) {
    println!("{}", cut);
    let mut first = true;
    let mut count = 0usize;
    for (idx, &side) in assignment.iter().enumerate() {
        if side == 1 {
            if !first {
                print!(" ");
            }
            first = false;
            print!("{}", idx);
            count += 1;
        }
    }
    println!();
    debug_assert_eq!(count, assignment.len() / 2);
}

struct BranchAndBound {
    n: usize,
    half: usize,
    adjacency: Vec<Vec<usize>>,
    order: Vec<usize>,
    assigned: Vec<i8>,
    deg_to_a: Vec<usize>,
    deg_to_b: Vec<usize>,
    best_cut: usize,
    best_assignment: Vec<i8>,
}

impl BranchAndBound {
    fn new(
        adjacency: Vec<Vec<usize>>,
        order: Vec<usize>,
        best_cut: usize,
        best_assignment: Vec<i8>,
    ) -> Self {
        let n = adjacency.len();
        let mut best_assignment = best_assignment;
        if best_assignment.len() != n {
            best_assignment = vec![0i8; n];
        }
        Self {
            n,
            half: n / 2,
            adjacency,
            order,
            assigned: vec![-1; n],
            deg_to_a: vec![0; n],
            deg_to_b: vec![0; n],
            best_cut,
            best_assignment,
        }
    }

    fn solve(&mut self) {
        if self.n == 0 || self.best_cut == 0 || self.order.is_empty() {
            return;
        }
        let first = self.order[0];
        let delta = self.place_vertex(first, 1);
        self.dfs(1, 1, 1, delta);
        self.remove_vertex(first, 1);
    }

    fn into_result(self) -> (usize, Vec<i8>) {
        (self.best_cut, self.best_assignment)
    }

    fn place_vertex(&mut self, v: usize, side: i8) -> usize {
        self.assigned[v] = side;
        let delta = if side == 1 {
            self.deg_to_b[v]
        } else {
            self.deg_to_a[v]
        };
        for &nbr in &self.adjacency[v] {
            if self.assigned[nbr] == -1 {
                if side == 1 {
                    self.deg_to_a[nbr] += 1;
                } else {
                    self.deg_to_b[nbr] += 1;
                }
            }
        }
        delta
    }

    fn remove_vertex(&mut self, v: usize, side: i8) {
        for &nbr in &self.adjacency[v] {
            if self.assigned[nbr] == -1 {
                if side == 1 {
                    self.deg_to_a[nbr] -= 1;
                } else {
                    self.deg_to_b[nbr] -= 1;
                }
            }
        }
        self.assigned[v] = -1;
    }

    fn dfs(&mut self, pos: usize, chosen_a: usize, assigned_count: usize, current_cut: usize) {
        if self.best_cut == 0 || chosen_a > self.half {
            return;
        }
        let chosen_b = assigned_count - chosen_a;
        if chosen_b > self.half || current_cut >= self.best_cut {
            return;
        }
        if assigned_count == self.n {
            if current_cut < self.best_cut {
                self.best_cut = current_cut;
                self.best_assignment = self.assigned.clone();
            }
            return;
        }
        if pos >= self.order.len() {
            return;
        }
        let rem_a = self.half - chosen_a;
        if self.best_cut != usize::MAX {
            let lb_extra = self.lower_bound_extra(rem_a);
            let projected = current_cut.saturating_add(lb_extra);
            if projected >= self.best_cut {
                return;
            }
        }
        let v = self.order[pos];
        let can_a = chosen_a < self.half;
        let can_b = chosen_b < self.half;

        if can_a && can_b {
            if self.deg_to_b[v] <= self.deg_to_a[v] {
                self.branch(v, 1, pos, chosen_a, assigned_count, current_cut);
                self.branch(v, 0, pos, chosen_a, assigned_count, current_cut);
            } else {
                self.branch(v, 0, pos, chosen_a, assigned_count, current_cut);
                self.branch(v, 1, pos, chosen_a, assigned_count, current_cut);
            }
        } else if can_a {
            self.branch(v, 1, pos, chosen_a, assigned_count, current_cut);
        } else if can_b {
            self.branch(v, 0, pos, chosen_a, assigned_count, current_cut);
        }
    }

    fn branch(
        &mut self,
        v: usize,
        side: i8,
        pos: usize,
        chosen_a: usize,
        assigned_count: usize,
        current_cut: usize,
    ) {
        let delta = self.place_vertex(v, side);
        let new_cut = current_cut + delta;
        let new_chosen_a = if side == 1 { chosen_a + 1 } else { chosen_a };
        let new_assigned = assigned_count + 1;
        self.dfs(pos + 1, new_chosen_a, new_assigned, new_cut);
        self.remove_vertex(v, side);
    }

    fn lower_bound_extra(&self, rem_a: usize) -> usize {
        if rem_a == 0 {
            let mut cost = 0usize;
            for v in 0..self.n {
                if self.assigned[v] == -1 {
                    cost += self.deg_to_a[v];
                }
            }
            return cost;
        }
        let mut deltas: Vec<isize> = Vec::new();
        let mut base = 0usize;
        for v in 0..self.n {
            if self.assigned[v] == -1 {
                let cost_b = self.deg_to_a[v];
                base += cost_b;
                let cost_a = self.deg_to_b[v];
                deltas.push(cost_a as isize - cost_b as isize);
            }
        }
        if rem_a > deltas.len() {
            return usize::MAX / 4;
        }
        deltas.sort();
        let mut adjustment = 0isize;
        for i in 0..rem_a {
            adjustment += deltas[i];
        }
        let total = base as isize + adjustment;
        if total < 0 {
            0
        } else {
            total as usize
        }
    }
}
