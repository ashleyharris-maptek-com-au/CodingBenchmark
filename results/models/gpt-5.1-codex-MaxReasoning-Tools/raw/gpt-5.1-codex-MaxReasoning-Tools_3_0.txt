Below is a hybrid graph-layout solver that adapts to the graph size/density: it runs a grid-approximated Fruchterman-Reingold force computation for graphs up to ~20K nodes (with iteration budgets chosen per scale/density), and automatically falls back to a fast multi-component radial/spiral placement for larger or extremely dense instances. Both modes normalize the final coordinates for numerical stability, and disconnected components, multi-edges, and self-loops are handled gracefully.

```rust
use std::collections::{HashMap, VecDeque};
use std::f64::consts::TAU;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut iter = input.split_whitespace();
    let n: usize = match iter.next().and_then(|s| s.parse().ok()) {
        Some(value) => value,
        None => return,
    };
    let m: usize = match iter.next().and_then(|s| s.parse().ok()) {
        Some(value) => value,
        None => return,
    };

    let mut adj = vec![Vec::<u32>::new(); n];
    for _ in 0..m {
        let u_opt = iter.next();
        let v_opt = iter.next();
        if u_opt.is_none() || v_opt.is_none() {
            break;
        }
        let u = match u_opt.unwrap().parse::<usize>() {
            Ok(val) => val,
            Err(_) => continue,
        };
        let v = match v_opt.unwrap().parse::<usize>() {
            Ok(val) => val,
            Err(_) => continue,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // Ignore self-loops for layout forces; they do not change geometry.
            continue;
        }
        adj[u].push(v as u32);
        adj[v].push(u as u32);
    }

    if n == 0 {
        return;
    }

    let avg_degree = if n > 0 {
        (2.0 * m as f64) / (n as f64)
    } else {
        0.0
    };
    let iterations = choose_iterations(n, avg_degree);

    let mut positions = vec![[0.0f64, 0.0f64]; n];
    if iterations == 0 {
        fallback_layout(&adj, &mut positions);
    } else {
        force_layout(&adj, &mut positions, iterations);
    }
    normalize_positions(&mut positions);

    let mut output = String::with_capacity(n * 32);
    for pos in &positions {
        output.push_str(&format!("{:.6} {:.6}\n", pos[0], pos[1]));
    }
    print!("{output}");
}

fn choose_iterations(n: usize, avg_degree: f64) -> usize {
    if n == 0 {
        0
    } else if n <= 15 {
        900
    } else if n <= 50 {
        700
    } else if n <= 150 {
        520
    } else if n <= 400 {
        360
    } else if n <= 1200 {
        240
    } else if n <= 4000 {
        160
    } else if n <= 12000 {
        if avg_degree > 40.0 { 100 } else { 140 }
    } else if n <= 20000 {
        if avg_degree > 25.0 { 50 } else { 80 }
    } else {
        0
    }
}

fn force_layout(adj: &[Vec<u32>], positions: &mut Vec<[f64; 2]>, iterations: usize) {
    let n = positions.len();
    if n == 0 {
        return;
    }

    let mut rng = SplitMix64::new(
        0x9E37_79B9_7F4A_7C15u64 ^ (n as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9),
    );
    let side = (n as f64).sqrt().max(1.0) * 10.0;
    for pos in positions.iter_mut() {
        let rx = rng.next_f64();
        let ry = rng.next_f64();
        pos[0] = (rx - 0.5) * side;
        pos[1] = (ry - 0.5) * side;
    }

    let area = side * side;
    let mut k = (area / (n as f64).max(1.0)).sqrt();
    if k < 0.1 {
        k = 0.1;
    }

    let mut displacements = vec![[0.0f64, 0.0f64]; n];
    for iter_idx in 0..iterations {
        let snapshot = positions.clone();
        let cell_size = k * 1.5 + 1e-9;
        let mut grid: HashMap<(i64, i64), Vec<usize>> = HashMap::with_capacity(n * 2 + 1);
        for (idx, pos) in snapshot.iter().enumerate() {
            let cx = (pos[0] / cell_size).floor() as i64;
            let cy = (pos[1] / cell_size).floor() as i64;
            grid.entry((cx, cy)).or_insert_with(Vec::new).push(idx);
        }

        for disp in &mut displacements {
            disp[0] = 0.0;
            disp[1] = 0.0;
        }

        for (i, pos) in snapshot.iter().enumerate() {
            let cx = (pos[0] / cell_size).floor() as i64;
            let cy = (pos[1] / cell_size).floor() as i64;
            let mut disp = [0.0f64, 0.0f64];
            for dx in -1..=1 {
                for dy in -1..=1 {
                    if let Some(list) = grid.get(&(cx + dx, cy + dy)) {
                        for &j in list {
                            if j == i {
                                continue;
                            }
                            let dxp = pos[0] - snapshot[j][0];
                            let dyp = pos[1] - snapshot[j][1];
                            let dist_sq = dxp * dxp + dyp * dyp + 1e-9;
                            let dist = dist_sq.sqrt();
                            let force = (k * k) / dist;
                            disp[0] += dxp / dist * force;
                            disp[1] += dyp / dist * force;
                        }
                    }
                }
            }
            displacements[i] = disp;
        }

        for (u, neighbors) in adj.iter().enumerate() {
            let pu = snapshot[u];
            for &v_u32 in neighbors {
                let v = v_u32 as usize;
                if v <= u {
                    continue;
                }
                let pv = snapshot[v];
                let dx = pu[0] - pv[0];
                let dy = pu[1] - pv[1];
                let dist_sq = dx * dx + dy * dy + 1e-9;
                let dist = dist_sq.sqrt();
                let force = dist_sq / k;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                displacements[u][0] -= fx;
                displacements[u][1] -= fy;
                displacements[v][0] += fx;
                displacements[v][1] += fy;
            }
        }

        let mut temperature = side / (1.0 + iter_idx as f64).sqrt();
        let min_temp = k * 0.05;
        if temperature < min_temp {
            temperature = min_temp;
        }
        for (pos, disp) in positions.iter_mut().zip(displacements.iter()) {
            let dx = disp[0];
            let dy = disp[1];
            let dist = (dx * dx + dy * dy).sqrt();
            if dist > temperature && dist > 0.0 {
                pos[0] += dx / dist * temperature;
                pos[1] += dy / dist * temperature;
            } else {
                pos[0] += dx;
                pos[1] += dy;
            }
        }
    }
}

fn fallback_layout(adj: &[Vec<u32>], positions: &mut Vec<[f64; 2]>) {
    let n = positions.len();
    if n == 0 {
        return;
    }
    let components = compute_components(adj);
    if components.is_empty() {
        return;
    }
    let spacing = (n as f64).max(1.0).sqrt() * 8.0 + 10.0;
    let golden_angle = 2.399963229728653;
    for (idx, nodes) in components.iter().enumerate() {
        if nodes.is_empty() {
            continue;
        }
        let ring = (idx as f64 + 1.0).sqrt();
        let angle = golden_angle * idx as f64;
        let center_x = ring * spacing * angle.cos();
        let center_y = ring * spacing * angle.sin();
        layout_component(nodes, positions, center_x, center_y);
    }
}

fn layout_component(nodes: &[usize], positions: &mut Vec<[f64; 2]>, center_x: f64, center_y: f64) {
    let size = nodes.len();
    if size == 0 {
        return;
    }
    if size == 1 {
        positions[nodes[0]][0] = center_x;
        positions[nodes[0]][1] = center_y;
    } else if size >= 4000 {
        let base_radius = (size as f64).sqrt() * 0.3 + 20.0;
        for (idx, &node) in nodes.iter().enumerate() {
            let theta = idx as f64 * 0.12;
            let r = base_radius + theta * 0.25;
            positions[node][0] = center_x + r * theta.cos();
            positions[node][1] = center_y + r * theta.sin();
        }
    } else {
        let local_radius = (size as f64).sqrt() * 6.0 + 5.0;
        for (idx, &node) in nodes.iter().enumerate() {
            let theta = TAU * idx as f64 / size as f64;
            positions[node][0] = center_x + local_radius * theta.cos();
            positions[node][1] = center_y + local_radius * theta.sin();
        }
    }
}

fn compute_components(adj: &[Vec<u32>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut visited = vec![false; n];
    let mut components = Vec::new();
    let mut queue = VecDeque::new();

    for start in 0..n {
        if visited[start] {
            continue;
        }
        visited[start] = true;
        queue.push_back(start);
        let mut nodes = Vec::new();
        while let Some(u) = queue.pop_front() {
            nodes.push(u);
            for &v_u32 in &adj[u] {
                let v = v_u32 as usize;
                if v == u {
                    continue;
                }
                if !visited[v] {
                    visited[v] = true;
                    queue.push_back(v);
                }
            }
        }
        components.push(nodes);
    }
    components
}

fn normalize_positions(positions: &mut Vec<[f64; 2]>) {
    if positions.is_empty() {
        return;
    }
    let mut min_x = positions[0][0];
    let mut max_x = positions[0][0];
    let mut min_y = positions[0][1];
    let mut max_y = positions[0][1];
    for pos in positions.iter() {
        if pos[0] < min_x {
            min_x = pos[0];
        }
        if pos[0] > max_x {
            max_x = pos[0];
        }
        if pos[1] < min_y {
            min_y = pos[1];
        }
        if pos[1] > max_y {
            max_y = pos[1];
        }
    }
    let center_x = 0.5 * (min_x + max_x);
    let center_y = 0.5 * (min_y + max_y);
    let span = (max_x - min_x).abs().max((max_y - min_y).abs()).max(1e-6);
    let scale = 1000.0 / span;
    for pos in positions.iter_mut() {
        pos[0] = (pos[0] - center_x) * scale;
        pos[1] = (pos[1] - center_y) * scale;
    }
}

#[derive(Clone)]
struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E37_79B9_7F4A_7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
        z ^ (z >> 31)
    }

    fn next_f64(&mut self) -> f64 {
        const SCALE: f64 = 1.0 / ((1u64 << 53) as f64);
        ((self.next_u64() >> 11) as f64) * SCALE
    }
}
```

This single binary reads from stdin, adapts the layout strategy to the graph’s size/density, and writes one `(x, y)` pair per node, ensuring good performance from tiny planar graphs up through extreme-scale inputs with hundreds of thousands of vertices.
