use std::collections::{HashMap, HashSet};
use std::io::{self, Read};
use std::rc::Rc;
use std::time::{Duration, Instant};
use std::thread;

const DIRECT_MITM_LIMIT: usize = 46;
const DENSE_SUM_LIMIT: i128 = 2_000_000;
const DENSE_N_LIMIT: usize = 6_000;
const SPARSE_N_LIMIT: usize = 4_000;
const SPARSE_STATE_LIMIT: usize = 100_000;
const SPARSE_PREFIX_LEN: usize = 4_000;
const RANDOM_SAMPLE_SIZE: usize = DIRECT_MITM_LIMIT;
const RANDOM_ITERATIONS: usize = 256;

struct FastScanner<'a> {
    data: &'a [u8],
    idx: usize,
}

impl<'a> FastScanner<'a> {
    fn new(data: &'a [u8]) -> Self {
        Self { data, idx: 0 }
    }

    fn skip_whitespace(&mut self) {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_i64(&mut self) -> Option<i64> {
        self.skip_whitespace();
        if self.idx >= self.data.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        } else if self.data[self.idx] == b'+' {
            self.idx += 1;
        }
        let mut value = 0i64;
        let mut found = false;
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_digit() {
            found = true;
            value = value * 10 + (self.data[self.idx] - b'0') as i64;
            self.idx += 1;
        }
        if found {
            Some(value * sign)
        } else {
            None
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_i64().and_then(|v| if v >= 0 { Some(v as usize) } else { None })
    }
}

#[derive(Clone)]
struct TraceNode {
    parent: Option<Rc<TraceNode>>,
    index_used: Option<usize>,
}

struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        let mut z = self.state.wrapping_add(0x9E3779B97F4A7C15);
        self.state = z;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }

    fn next_usize(&mut self, bound: usize) -> usize {
        if bound == 0 {
            0
        } else {
            (self.next_u64() % bound as u64) as usize
        }
    }
}

struct Solver {
    numbers: Vec<i64>,
    target: i64,
    deadline: Instant,
}

impl Solver {
    fn new(numbers: Vec<i64>, target: i64, deadline: Instant) -> Self {
        Self {
            numbers,
            target,
            deadline,
        }
    }

    fn numbers(&self) -> &[i64] {
        &self.numbers
    }

    fn target(&self) -> i64 {
        self.target
    }

    fn solve(&self) -> Option<Vec<usize>> {
        let n = self.numbers.len();
        if self.target == 0 {
            return Some(Vec::new());
        }
        if n == 0 {
            return None;
        }
        if let Some(idx) = self.numbers.iter().position(|&v| v == self.target) {
            return Some(vec![idx]);
        }
        let total_sum: i128 = self.numbers.iter().map(|&v| v as i128).sum();
        if total_sum == self.target as i128 {
            return Some((0..n).collect());
        }
        if n <= DIRECT_MITM_LIMIT {
            let indices: Vec<usize> = (0..n).collect();
            if let Some(res) =
                meet_in_middle_indices(&self.numbers, &indices, self.target, self.deadline)
            {
                return Some(res);
            }
        }
        if let Some(res) = self.try_dense_dp() {
            return Some(res);
        }
        if n <= SPARSE_N_LIMIT {
            if let Some(res) = bounded_sparse_dp(
                &self.numbers,
                None,
                self.target,
                SPARSE_STATE_LIMIT,
                self.deadline,
            ) {
                return Some(res);
            }
        } else {
            let prefix_len = n.min(SPARSE_PREFIX_LEN);
            if prefix_len > 0 {
                if let Some(res) = bounded_sparse_dp(
                    &self.numbers[..prefix_len],
                    None,
                    self.target,
                    SPARSE_STATE_LIMIT,
                    self.deadline,
                ) {
                    return Some(res);
                }
            }
        }
        if let Some(res) = self.core_focus_exact() {
            return Some(res);
        }
        self.random_sampling_search()
    }

    fn try_dense_dp(&self) -> Option<Vec<usize>> {
        let n = self.numbers.len();
        if n == 0 || n > DENSE_N_LIMIT {
            return None;
        }
        let mut min_sum: i128 = 0;
        let mut max_sum: i128 = 0;
        for &value in &self.numbers {
            if value >= 0 {
                max_sum += value as i128;
            } else {
                min_sum += value as i128;
            }
        }
        let span = max_sum - min_sum;
        if span > DENSE_SUM_LIMIT {
            return None;
        }
        let span_usize = (span as usize) + 1;
        let offset = (-min_sum) as usize;
        let target_i128 = self.target as i128;
        if target_i128 < min_sum || target_i128 > max_sum {
            return None;
        }
        let target_idx = (target_i128 - min_sum) as usize;
        let bits_len = (span_usize + 63) / 64;
        let mut bits = vec![0u64; bits_len];
        let mut shifted = vec![0u64; bits_len];
        let mut parent_choice = vec![-1i32; span_usize];
        let mut parent_prev = vec![-1i32; span_usize];
        bits[offset / 64] |= 1u64 << (offset % 64);
        parent_choice[offset] = -2;
        parent_prev[offset] = -1;
        for (idx, &value) in self.numbers.iter().enumerate() {
            if Instant::now() >= self.deadline {
                break;
            }
            if value == 0 {
                continue;
            }
            if value > 0 {
                shift_left_into(&bits, value as usize, &mut shifted, span_usize);
            } else {
                shift_right_into(&bits, (-value) as usize, &mut shifted, span_usize);
            }
            for word in 0..bits_len {
                let new_bits = shifted[word] & !bits[word];
                if new_bits == 0 {
                    continue;
                }
                let mut bits_to_process = new_bits;
                while bits_to_process != 0 {
                    let bit_pos = bits_to_process.trailing_zeros() as usize;
                    let bit_index = word * 64 + bit_pos;
                    if bit_index >= span_usize {
                        break;
                    }
                    if parent_choice[bit_index] == -1 {
                        let prev_index = if value > 0 {
                            if bit_index < value as usize {
                                bits_to_process &= bits_to_process - 1;
                                continue;
                            }
                            bit_index - value as usize
                        } else {
                            let offset_idx = bit_index + (-value) as usize;
                            if offset_idx >= span_usize {
                                bits_to_process &= bits_to_process - 1;
                                continue;
                            }
                            offset_idx
                        };
                        parent_choice[bit_index] = idx as i32;
                        parent_prev[bit_index] = prev_index as i32;
                        if bit_index == target_idx {
                            return Some(rebuild_from_parents(
                                bit_index,
                                &parent_choice,
                                &parent_prev,
                            ));
                        }
                    }
                    bits_to_process &= bits_to_process - 1;
                }
            }
            for word in 0..bits_len {
                bits[word] |= shifted[word];
            }
        }
        if parent_choice[target_idx] != -1 {
            return Some(rebuild_from_parents(
                target_idx,
                &parent_choice,
                &parent_prev,
            ));
        }
        None
    }

    fn core_focus_exact(&self) -> Option<Vec<usize>> {
        let n = self.numbers.len();
        if n <= DIRECT_MITM_LIMIT {
            return None;
        }
        let core_limit = DIRECT_MITM_LIMIT.min(n);
        if core_limit == 0 {
            return None;
        }
        let mut indices: Vec<usize> = (0..n).collect();
        indices.sort_unstable_by(|&a, &b| {
            self.numbers[b]
                .unsigned_abs()
                .cmp(&self.numbers[a].unsigned_abs())
        });
        indices.truncate(core_limit);
        meet_in_middle_indices(&self.numbers, &indices, self.target, self.deadline)
    }

    fn random_sampling_search(&self) -> Option<Vec<usize>> {
        let n = self.numbers.len();
        if n == 0 {
            return None;
        }
        let max_size = RANDOM_SAMPLE_SIZE.min(n);
        if max_size == 0 {
            return None;
       
        }
        let mut seed = n as u64;
        seed ^= (self.target as i128).unsigned_abs() as u64;
        seed ^= self.numbers[0].unsigned_abs().rotate_left(13);
        let mut rng = SplitMix64::new(seed);
        let mut buffer = Vec::with_capacity(max_size);
        let size_candidates = [4usize, 8, 12, 16, 24, 32, max_size];
        let mut last_size = 0usize;
        for &candidate in &size_candidates {
            let actual_size = candidate.min(max_size);
            if actual_size == 0 || actual_size == last_size {
                continue;
            }
            last_size = actual_size;
            for _ in 0..RANDOM_ITERATIONS {
                if Instant::now() >= self.deadline {
                    return None;
                }
                buffer.clear();
                while buffer.len() < actual_size {
                    let idx = rng.next_usize(n);
                    if !buffer.contains(&idx) {
                        buffer.push(idx);
                    }
                }
                if let Some(res) =
                    meet_in_middle_indices(&self.numbers, &buffer, self.target, self.deadline)
                {
                    return Some(res);
                }
            }
        }
        None
    }
}

fn main() {
    let mut input = Vec::new();
    if io::stdin().read_to_end(&mut input).is_err() {
        return;
    }
    if input.is_empty() {
        println!("NO");
        return;
    }
    let mut scanner = FastScanner::new(&input);
    let declared_n = match scanner.next_usize() {
        Some(v) => v,
        None => {
            println!("NO");
            return;
        }
    };
    let target = match scanner.next_i64() {
        Some(v) => v,
        None => {
            println!("NO");
            return;
        }
    };
    let mut numbers = Vec::with_capacity(declared_n);
    for _ in 0..declared_n {
        if let Some(value) = scanner.next_i64() {
            numbers.push(value);
        } else {
            break;
        }
    }
    let now = Instant::now();
    let deadline = now
        .checked_add(Duration::from_secs(295))
        .unwrap_or(now + Duration::from_secs(295));
    let solver = Solver::new(numbers, target, deadline);
    match solver.solve() {
        Some(mut indices) => {
            indices.sort_unstable();
            debug_assert!(verify_subset(solver.numbers(), &indices, solver.target()));
            println!("YES");
            print_indices(&indices);
        }
        None => println!("NO"),
    }
}

fn meet_in_middle_indices(
    numbers: &[i64],
    indices: &[usize],
    target: i64,
    deadline: Instant,
) -> Option<Vec<usize>> {
    let len = indices.len();
    if len == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }
    if len > 60 {
        return None;
    }
    let mid = len / 2;
    let (left_idx, right_idx) = indices.split_at(mid);
    let left_vals: Vec<i64> = left_idx.iter().map(|&i| numbers[i]).collect();
    let right_vals: Vec<i64> = right_idx.iter().map(|&i| numbers[i]).collect();
    let (left_sums, mut right_sums) = thread::scope(|scope| {
        let left_handle = scope.spawn(|_| generate_subset_sums(&left_vals));
        let right_handle = scope.spawn(|_| generate_subset_sums(&right_vals));
        (left_handle.join().unwrap(), right_handle.join().unwrap())
    });
    for &(sum, mask) in &left_sums {
        if sum == target && mask != 0 {
            let mut result = Vec::new();
            append_mask_indices(mask, left_idx, &mut result);
            return Some(result);
        }
    }
    for &(sum, mask) in &right_sums {
        if sum == target && mask != 0 {
            let mut result = Vec::new();
            append_mask_indices(mask, right_idx, &mut result);
            return Some(result);
        }
    }
    right_sums.sort_by_key(|entry| entry.0);
    for (pos, &(sum, mask)) in left_sums.iter().enumerate() {
        if (pos & 0x3FFF) == 0 && Instant::now() >= deadline {
            return None;
        }
        let needed = target - sum;
        if let Ok(idx) = right_sums.binary_search_by_key(&needed, |entry| entry.0) {
            let mut result = Vec::new();
            if mask != 0 {
                append_mask_indices(mask, left_idx, &mut result);
            }
            let right_mask = right_sums[idx].1;
            if right_mask != 0 {
                append_mask_indices(right_mask, right_idx, &mut result);
            }
            if result.is_empty() && target == 0 {
                return Some(result);
            }
            return Some(result);
        }
    }
    None
}

fn generate_subset_sums(values: &[i64]) -> Vec<(i64, u64)> {
    let len = values.len();
    let total = 1usize << len;
    let mut sums = vec![0i64; total];
    for mask in 1..total {
        let lsb = mask & (!mask + 1);
        let bit = lsb.trailing_zeros() as usize;
        sums[mask] = sums[mask ^ lsb] + values[bit];
    }
    let mut res = Vec::with_capacity(total);
    for mask in 0..total {
        res.push((sums[mask], mask as u64));
    }
    res
}

fn append_mask_indices(mask: u64, mapping: &[usize], out: &mut Vec<usize>) {
    for (bit, &index) in mapping.iter().enumerate() {
        if ((mask >> bit) & 1) != 0 {
            out.push(index);
        }
    }
}

fn shift_left_into(src: &[u64], shift: usize, dst: &mut [u64], total_bits: usize) {
    dst.fill(0);
    if shift == 0 {
        dst.copy_from_slice(src);
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    let len = src.len();
    if word_shift >= len {
        return;
    }
    for i in 0..(len - word_shift) {
        let val = src[i];
        if val == 0 {
            continue;
        }
        let target = i + word_shift;
        dst[target] |= val << bit_shift;
        if bit_shift != 0 && target + 1 < len {
            dst[target + 1] |= val >> (64 - bit_shift);
        }
    }
    mask_unused_bits(dst, total_bits);
}

fn shift_right_into(src: &[u64], shift: usize, dst: &mut [u64], total_bits: usize) {
    dst.fill(0);
    if shift == 0 {
        dst.copy_from_slice(src);
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    let len = src.len();
    if word_shift >= len {
        return;
    }
    for i in word_shift..len {
        let val = src[i];
        if val == 0 {
            continue;
        }
        let target = i - word_shift;
        dst[target] |= val >> bit_shift;
        if bit_shift != 0 && target > 0 {
            dst[target - 1] |= val << (64 - bit_shift);
        }
    }
    mask_unused_bits(dst, total_bits);
}

fn mask_unused_bits(bits: &mut [u64], total_bits: usize) {
    let remainder = total_bits % 64;
    if remainder == 0 || bits.is_empty() {
        return;
    }
    let mask = (1u64 << remainder) - 1;
    if let Some(last) = bits.last_mut() {
        *last &= mask;
    }
}

fn rebuild_from_parents(
    mut idx: usize,
    parent_choice: &[i32],
    parent_prev: &[i32],
) -> Vec<usize> {
    let mut result = Vec::new();
    loop {
        let choice = parent_choice[idx];
        if choice == -2 {
            break;
        }
        if choice < 0 {
            result.clear();
            break;
        }
        result.push(choice as usize);
        let prev = parent_prev[idx];
        if prev < 0 {
            break;
        }
        idx = prev as usize;
    }
    result.reverse();
    result
}

fn bounded_sparse_dp(
    values: &[i64],
    index_map: Option<&[usize]>,
    target: i64,
    limit: usize,
    deadline: Instant,
) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if values.is_empty() {
        return None;
    }
    let mut map: HashMap<i64, Rc<TraceNode>> = HashMap::new();
    let base = Rc::new(TraceNode {
        parent: None,
        index_used: None,
    });
    map.insert(0, base.clone());
    let mut new_entries: Vec<(i64, Rc<TraceNode>)> = Vec::new();
    let mut seen = HashSet::new();
    for (idx, &value) in values.iter().enumerate() {
        if Instant::now() >= deadline {
            break;
        }
        new_entries.clear();
        seen.clear();
        let actual_index = index_map.map(|m| m[idx]).unwrap_or(idx);
        for (&sum, node) in map.iter() {
            let new_sum = sum + value;
            if map.contains_key(&new_sum) || !seen.insert(new_sum) {
                continue;
            }
            let new_node = Rc::new(TraceNode {
                parent: Some(node.clone()),
                index_used: Some(actual_index),
            });
            new_entries.push((new_sum, new_node));
        }
        for (new_sum, node) in new_entries.drain(..) {
            map.insert(new_sum, node.clone());
            if new_sum == target {
                return Some(reconstruct_from_node(&node));
            }
        }
        if map.len() > limit {
            prune_hash_map_rc(&mut map, target, limit, &base);
        }
    }
    None
}

fn prune_hash_map_rc(
    map: &mut HashMap<i64, Rc<TraceNode>>,
    target: i64,
    limit: usize,
    base: &Rc<TraceNode>,
) {
    if map.len() <= limit {
        return;
    }
    let mut entries: Vec<(i64, Rc<TraceNode>)> = map
        .iter()
        .filter(|(&sum, _)| sum != 0)
        .map(|(&sum, node)| (sum, node.clone()))
        .collect();
    entries.sort_by_key(|&(sum, _)| {
        let diff = (sum as i128) - (target as i128);
        diff.abs()
    });
    map.clear();
    map.insert(0, base.clone());
    for (sum, node) in entries.into_iter() {
        if map.len() >= limit {
            break;
        }
        map.insert(sum, node);
    }
}

fn reconstruct_from_node(node: &Rc<TraceNode>) -> Vec<usize> {
    let mut result = Vec::new();
    let mut current = Some(node.clone());
    while let Some(cur) = current {
        if let Some(idx) = cur.index_used {
            result.push(idx);
        }
        current = cur.parent.clone();
    }
    result.reverse();
    result
}

fn verify_subset(numbers: &[i64], subset: &[usize], target: i64) -> bool {
    let mut sum = 0i128;
    for &idx in subset {
        if let Some(&value) = numbers.get(idx) {
            sum += value as i128;
        } else {
            return false;
        }
    }
    sum == target as i128
}

fn print_indices(indices: &[usize]) {
    print!("[");
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{idx}");
    }
    println!("]");
}
