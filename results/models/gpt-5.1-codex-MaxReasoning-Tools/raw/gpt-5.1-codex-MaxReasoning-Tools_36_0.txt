#include <algorithm>
#include <cstdint>
#include <iostream>
#include <queue>
#include <vector>

#ifdef _MSC_VER
#include <intrin.h>
#endif

struct Component {
    std::vector<int> nodes;
    bool bipartite = true;
};

// Branch-and-bound maximum clique solver (Tomita-style) operating on complement graphs.
class MaxClique {
public:
    MaxClique(int n, std::vector<std::vector<uint64_t>> adjacency)
        : n_(n),
          words_(n > 0 ? static_cast<int>(adjacency[0].size()) : 0),
          adj_(std::move(adjacency)),
          temp_(words_, 0),
          scratch_(words_, 0),
          orderBuf_(n_ + 1),
          colorBuf_(n_ + 1) {
        curr_.reserve(n_);
        best_.reserve(n_);
    }

    std::vector<int> solveWithMask(const std::vector<uint64_t>& mask) {
        if (words_ == 0 || isZero(mask)) {
            return {};
        }
        best_.clear();
        curr_.clear();
        std::vector<uint64_t> cand = mask;
        expand(cand, 0);
        return best_;
    }

private:
    int n_;
    int words_;
    std::vector<std::vector<uint64_t>> adj_;
    std::vector<int> curr_;
    std::vector<int> best_;
    std::vector<std::vector<int>> orderBuf_;
    std::vector<std::vector<int>> colorBuf_;
    std::vector<uint64_t> temp_;
    std::vector<uint64_t> scratch_;

    static bool isZero(const std::vector<uint64_t>& bits) {
        for (uint64_t x : bits) {
            if (x) {
                return false;
            }
        }
        return true;
    }

    void clearBit(std::vector<uint64_t>& bits, int v) const {
        bits[v >> 6] &= ~(1ULL << (v & 63));
    }

    int popLSB(std::vector<uint64_t>& bits) const {
        for (int i = 0; i < words_; ++i) {
            uint64_t word = bits[i];
            if (word) {
                uint64_t lsb = word & -word;
                bits[i] = word ^ lsb;
#ifdef _MSC_VER
                unsigned long idx;
                _BitScanForward64(&idx, lsb);
                return static_cast<int>((i << 6) + idx);
#else
                return (i << 6) + __builtin_ctzll(lsb);
#endif
            }
        }
        return -1;
    }

    void colorOrder(const std::vector<uint64_t>& cand,
                    std::vector<int>& order,
                    std::vector<int>& colors) {
        order.clear();
        colors.clear();
        if (words_ == 0) {
            return;
        }
        std::copy(cand.begin(), cand.end(), temp_.begin());
        int color = 0;
        while (!isZero(temp_)) {
            ++color;
            std::copy(temp_.begin(), temp_.end(), scratch_.begin());
            while (!isZero(scratch_)) {
                int v = popLSB(scratch_);
                order.push_back(v);
                colors.push_back(color);
                clearBit(temp_, v);
                const auto& neigh = adj_[v];
                for (int i = 0; i < words_; ++i) {
                    scratch_[i] &= ~neigh[i];
                }
            }
        }
    }

    void expand(std::vector<uint64_t>& cand, int depth) {
        auto& order = orderBuf_[depth];
        auto& colors = colorBuf_[depth];
        colorOrder(cand, order, colors);
        int orderSize = static_cast<int>(order.size());
        if (orderSize == 0) {
            if (curr_.size() > best_.size()) {
                best_ = curr_;
            }
            return;
        }
        std::vector<uint64_t> next(words_);
        for (int idx = orderSize - 1; idx >= 0; --idx) {
            int v = order[idx];
            if (static_cast<int>(curr_.size()) + colors[idx] <= static_cast<int>(best_.size())) {
                return;
            }
            curr_.push_back(v);
            if (curr_.size() > best_.size()) {
                best_ = curr_;
            }
            bool empty = true;
            const auto& neigh = adj_[v];
            for (int i = 0; i < words_; ++i) {
                uint64_t val = cand[i] & neigh[i];
                next[i] = val;
                if (val) {
                    empty = false;
                }
            }
            if (!empty) {
                expand(next, depth + 1);
            }
            curr_.pop_back();
            clearBit(cand, v);
        }
    }
};

std::vector<int> solveBipartiteComponent(const std::vector<int>& nodes,
                                         const std::vector<std::vector<int>>& adjList,
                                         const std::vector<int>& color,
                                         std::vector<int>& rightIndexBuffer) {
    std::vector<int> left, right;
    left.reserve(nodes.size());
    right.reserve(nodes.size());
    for (int v : nodes) {
        if (color[v] == 0) {
            left.push_back(v);
        } else {
            right.push_back(v);
        }
    }
    if (left.empty()) {
        return right;
    }
    if (right.empty()) {
        return left;
    }

    for (int i = 0; i < static_cast<int>(right.size()); ++i) {
        rightIndexBuffer[right[i]] = i;
    }

    std::vector<std::vector<int>> graph(left.size());
    for (int i = 0; i < static_cast<int>(left.size()); ++i) {
        int u = left[i];
        auto& edges = graph[i];
        edges.reserve(adjList[u].size());
        for (int nb : adjList[u]) {
            int idx = rightIndexBuffer[nb];
            if (idx != -1) {
                edges.push_back(idx);
            }
        }
    }

    std::vector<int> dist(left.size());
    std::vector<int> matchL(left.size(), -1);
    std::vector<int> matchR(right.size(), -1);

    auto bfs = [&]() -> bool {
        std::queue<int> q;
        bool found = false;
        for (int i = 0; i < static_cast<int>(left.size()); ++i) {
            if (matchL[i] == -1) {
                dist[i] = 0;
                q.push(i);
            } else {
                dist[i] = -1;
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : graph[u]) {
                int mate = matchR[v];
                if (mate == -1) {
                    found = true;
                } else if (dist[mate] == -1) {
                    dist[mate] = dist[u] + 1;
                    q.push(mate);
                }
            }
        }
        return found;
    };

    auto dfs = [&](auto&& self, int u) -> bool {
        for (int v : graph[u]) {
            int mate = matchR[v];
            if (mate == -1 || (dist[mate] == dist[u] + 1 && self(self, mate))) {
                matchL[u] = v;
                matchR[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    };

    while (bfs()) {
        for (int i = 0; i < static_cast<int>(left.size()); ++i) {
            if (matchL[i] == -1) {
                dfs(dfs, i);
            }
        }
    }

    std::vector<int> visitedLeft(left.size(), 0);
    std::vector<int> visitedRight(right.size(), 0);
    std::queue<int> q;
    for (int i = 0; i < static_cast<int>(left.size()); ++i) {
        if (matchL[i] == -1) {
            visitedLeft[i] = 1;
            q.push(i);
        }
    }
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : graph[u]) {
            if (matchL[u] == v) {
                continue;
            }
            if (visitedRight[v]) {
                continue;
            }
            visitedRight[v] = 1;
            int mate = matchR[v];
            if (mate != -1 && !visitedLeft[mate]) {
                visitedLeft[mate] = 1;
                q.push(mate);
            }
        }
    }

    std::vector<int> independent;
    independent.reserve(nodes.size());
    for (int i = 0; i < static_cast<int>(left.size()); ++i) {
        if (visitedLeft[i]) {
            independent.push_back(left[i]);
        }
    }
    for (int j = 0; j < static_cast<int>(right.size()); ++j) {
        if (!visitedRight[j]) {
            independent.push_back(right[j]);
        }
    }

    for (int v : right) {
        rightIndexBuffer[v] = -1;
    }

    return independent;
}

std::vector<int> solveNonBipartiteComponent(const std::vector<int>& nodes,
                                            const std::vector<std::vector<int>>& adjList,
                                            std::vector<int>& vertexToLocal) {
    const int k = static_cast<int>(nodes.size());
    if (k == 0) {
        return {};
    }
    const int words = (k + 63) / 64;
    std::vector<std::vector<uint64_t>> adjacency(k, std::vector<uint64_t>(words, 0));
    for (int i = 0; i < k; ++i) {
        vertexToLocal[nodes[i]] = i;
    }
    for (int i = 0; i < k; ++i) {
        int v = nodes[i];
        for (int nb : adjList[v]) {
            int idx = vertexToLocal[nb];
            if (idx != -1) {
                adjacency[i][idx >> 6] |= (1ULL << (idx & 63));
            }
        }
    }
    std::vector<uint64_t> fullMask(words, ~0ULL);
    if (words > 0) {
        uint64_t tail = (k % 64 == 0) ? ~0ULL : ((1ULL << (k % 64)) - 1ULL);
        fullMask.back() = tail;
    }
    for (int i = 0; i < k; ++i) {
        for (int w = 0; w < words; ++w) {
            adjacency[i][w] = fullMask[w] & ~adjacency[i][w];
        }
        adjacency[i][i >> 6] &= ~(1ULL << (i & 63));
    }
    MaxClique solver(k, std::move(adjacency));
    auto clique = solver.solveWithMask(fullMask);

    std::vector<int> independent;
    independent.reserve(clique.size());
    for (int idx : clique) {
        independent.push_back(nodes[idx]);
    }
    for (int v : nodes) {
        vertexToLocal[v] = -1;
    }
    return independent;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m;
    if (!(std::cin >> n >> m)) {
        return 0;
    }

    std::vector<std::vector<int>> adjList(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) {
            continue;
        }
        if (u == v) {
            continue;
        }
        adjList[u].push_back(v);
        adjList[v].push_back(u);
    }

    std::vector<Component> components;
    components.reserve(n);
    std::vector<int> compId(n, -1);
    std::vector<int> color(n, -1);

    for (int start = 0; start < n; ++start) {
        if (compId[start] != -1) {
            continue;
        }
        Component comp;
        comp.bipartite = true;
        std::vector<int> stack;
        stack.push_back(start);
        int compIdx = static_cast<int>(components.size());
        compId[start] = compIdx;
        color[start] = 0;
        while (!stack.empty()) {
            int v = stack.back();
            stack.pop_back();
            comp.nodes.push_back(v);
            for (int nb : adjList[v]) {
                if (compId[nb] == -1) {
                    compId[nb] = compIdx;
                    color[nb] = color[v] ^ 1;
                    stack.push_back(nb);
                } else if (color[nb] == color[v]) {
                    comp.bipartite = false;
                }
            }
        }
        components.push_back(std::move(comp));
    }

    std::vector<int> result;
    result.reserve(n);
    std::vector<int> rightIndex(n, -1);
    std::vector<int> vertexToLocal(n, -1);

    for (const auto& comp : components) {
        const auto& nodes = comp.nodes;
        if (nodes.empty()) {
            continue;
        }
        if (nodes.size() == 1) {
            result.push_back(nodes[0]);
        } else if (comp.bipartite) {
            auto part = solveBipartiteComponent(nodes, adjList, color, rightIndex);
            result.insert(result.end(), part.begin(), part.end());
        } else {
            auto part = solveNonBipartiteComponent(nodes, adjList, vertexToLocal);
            result.insert(result.end(), part.begin(), part.end());
        }
    }

    std::sort(result.begin(), result.end());
    result.erase(std::unique(result.begin(), result.end()), result.end());

    std::cout << result.size() << "\n";
    for (std::size_t i = 0; i < result.size(); ++i) {
        if (i) {
            std::cout << ' ';
        }
        std::cout << result[i];
    }
    std::cout << "\n";
    return 0;
}
