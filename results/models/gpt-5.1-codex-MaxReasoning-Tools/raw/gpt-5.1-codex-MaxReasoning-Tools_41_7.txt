OpCapability Shader

OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo

OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16

OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%_ptr_Input_v3 = OpTypePointer Input %v3
%_ptr_Output_v4 = OpTypePointer Output %v4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4 = OpTypePointer Uniform %v4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1_0 = OpConstant %float 1
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%baseColor = OpConstantComposite %v3 %float_0_2 %float_0_3 %float_0_8
%rimColorConst = OpConstantComposite %v3 %float_1_0 %float_0_8 %float_0_5
%worldPos = OpVariable %_ptr_Input_v3 Input
%normal = OpVariable %_ptr_Input_v3 Input
%outColor = OpVariable %_ptr_Output_v4 Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%main = OpFunction %void None %func
%entry = OpLabel
%normalVal = OpLoad %v3 %normal
%normN = OpExtInst %v3 %glsl Normalize %normalVal
%worldVal = OpLoad %v3 %worldPos
%lightPtr = OpAccessChain %_ptr_Uniform_v4 %ubo %uint_3
%lightVec4 = OpLoad %v4 %lightPtr
%lightVec3 = OpVectorShuffle %v3 %lightVec4 %lightVec4 0 1 2
%lightDirRaw = OpFSub %v3 %lightVec3 %worldVal
%lightDir = OpExtInst %v3 %glsl Normalize %lightDirRaw
%camPtr = OpAccessChain %_ptr_Uniform_v4 %ubo %uint_4
%camVec4 = OpLoad %v4 %camPtr
%camVec3 = OpVectorShuffle %v3 %camVec4 %camVec4 0 1 2
%viewDirRaw = OpFSub %v3 %camVec3 %worldVal
%viewDir = OpExtInst %v3 %glsl Normalize %viewDirRaw
%dotNL = OpDot %float %normN %lightDir
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0
%diffuseScaled = OpFMul %float %float_0_5 %diffuse
%baseScale = OpFAdd %float %diffuseScaled %float_0_3
%baseLit = OpVectorTimesScalar %v3 %baseColor %baseScale
%dotNV = OpDot %float %normN %viewDir
%dotNVClamped = OpExtInst %float %glsl FMax %dotNV %float_0
%oneMinus = OpFSub %float %float_1 %dotNVClamped
%rimSq = OpFMul %float %oneMinus %oneMinus
%rim = OpFMul %float %rimSq %oneMinus
%rimColor = OpVectorTimesScalar %v3 %rimColorConst %rim
%finalRGB = OpFAdd %v3 %baseLit %rimColor
%finalColor = OpCompositeConstruct %v4 %finalRGB %float_1
OpStore %outColor %finalColor
OpReturn
