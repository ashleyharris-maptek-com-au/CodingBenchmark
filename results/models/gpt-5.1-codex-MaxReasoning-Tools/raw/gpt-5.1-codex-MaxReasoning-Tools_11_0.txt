use std::cmp::Ordering;
use std::io::{self, Read};

const EPS: f64 = 1e-9;
const FIT_EPS: f64 = 1e-7;
const BASE_ANGLE_STEPS: usize = 8;
const MAX_ORIENTATION_SET: usize = 16;
const OFFSET_CAP: usize = 5;
const BASE_GRID_STEPS: usize = 3;
const DENSE_GRID_STEPS: usize = 10;
const FORCE_GRID_STEPS: usize = 20;
const GRID_POS_CAP: usize = FORCE_GRID_STEPS + 1;
const MAX_FREE_RECTS: usize = 512;
const RANDOM_ATTEMPTS: usize = 96;
const TWO_PI: f64 = std::f64::consts::PI * 2.0;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Rect {
    x: f64,
    y: f64,
    width: f64,
    height: f64,
}

impl Rect {
    fn right(self) -> f64 {
        self.x + self.width
    }
    fn top(self) -> f64 {
        self.y + self.height
    }
    fn area(self) -> f64 {
        self.width.max(0.0) * self.height.max(0.0)
    }
    fn overlaps(&self, other: &Rect) -> bool {
        self.x + self.width - FIT_EPS > other.x
            && other.x + other.width - FIT_EPS > self.x
            && self.y + self.height - FIT_EPS > other.y
            && other.y + other.height - FIT_EPS > self.y
    }
    fn intersects_inclusive(&self, other: &Rect) -> bool {
        self.x < other.right() + FIT_EPS
            && self.right() > other.x - FIT_EPS
            && self.y < other.top() + FIT_EPS
            && self.top() > other.y - FIT_EPS
    }
    fn contains(&self, other: &Rect) -> bool {
        other.x + FIT_EPS >= self.x
            && other.y + FIT_EPS >= self.y
            && other.right() <= self.right() + FIT_EPS
            && other.top() <= self.top() + FIT_EPS
    }
    fn valid(&self) -> bool {
        self.width > FIT_EPS && self.height > FIT_EPS
    }
}

#[derive(Clone, Copy, Debug)]
struct Aabb {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

impl Aabb {
    fn new_empty() -> Self {
        Self {
            min_x: f64::INFINITY,
            min_y: f64::INFINITY,
            max_x: f64::NEG_INFINITY,
            max_y: f64::NEG_INFINITY,
        }
    }
    fn include(&mut self, p: Point) {
        if p.x < self.min_x {
            self.min_x = p.x;
        }
        if p.y < self.min_y {
            self.min_y = p.y;
        }
        if p.x > self.max_x {
            self.max_x = p.x;
        }
        if p.y > self.max_y {
            self.max_y = p.y;
        }
    }
    fn width(&self) -> f64 {
        (self.max_x - self.min_x).max(0.0)
    }
    fn height(&self) -> f64 {
        (self.max_y - self.min_y).max(0.0)
    }
}

impl From<Aabb> for Rect {
    fn from(aabb: Aabb) -> Self {
        Rect {
            x: aabb.min_x,
            y: aabb.min_y,
            width: aabb.width(),
            height: aabb.height(),
        }
    }
}

struct QueryMarker {
    marks: Vec<u32>,
    token: u32,
}

impl QueryMarker {
    fn new(size: usize) -> Self {
        Self {
            marks: vec![0; size],
            token: 1,
        }
    }
    fn next(&mut self) {
        self.token = self.token.wrapping_add(1);
        if self.token == 0 {
            self.marks.fill(0);
            self.token = 1;
        }
    }
    fn mark(&mut self, idx: usize) -> bool {
        if self.marks[idx] != self.token {
            self.marks[idx] = self.token;
            true
        } else {
            false
        }
    }
}

struct SpatialGrid {
    origin_x: f64,
    origin_y: f64,
    cell_w: f64,
    cell_h: f64,
    cols: usize,
    rows: usize,
    buckets: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(bbox: &Rect, edges: &[Edge]) -> Self {
        let width = bbox.width.max(FIT_EPS);
        let height = bbox.height.max(FIT_EPS);
        let base = ((edges.len().max(1) as f64).sqrt().ceil() as usize).clamp(1, 64);
        let cols = base;
        let rows = base;
        let cell_w = (width / cols as f64).max(FIT_EPS);
        let cell_h = (height / rows as f64).max(FIT_EPS);
        let mut buckets = vec![Vec::new(); cols * rows];
        for (idx, edge) in edges.iter().enumerate() {
            let (sx, ex) = Self::range(edge.bbox.min_x, edge.bbox.max_x, bbox.x, cell_w, cols);
            let (sy, ey) = Self::range(edge.bbox.min_y, edge.bbox.max_y, bbox.y, cell_h, rows);
            for ix in sx..=ex {
                for iy in sy..=ey {
                    buckets[iy * cols + ix].push(idx);
                }
            }
        }
        Self {
            origin_x: bbox.x,
            origin_y: bbox.y,
            cell_w,
            cell_h,
            cols,
            rows,
            buckets,
        }
    }

    fn range(min_v: f64, max_v: f64, origin: f64, cell: f64, count: usize) -> (usize, usize) {
        if count == 0 {
            return (0, 0);
        }
        let mut a = min_v.min(max_v);
        let mut b = min_v.max(max_v);
        if (b - a).abs() < EPS {
            b = a;
        }
        let idx_min = (((a - origin) / cell).floor() as isize).clamp(0, count as isize - 1);
        let idx_max = (((b - origin) / cell).floor() as isize).clamp(0, count as isize - 1);
        let (s, e) = if idx_min <= idx_max {
            (idx_min, idx_max)
        } else {
            (idx_max, idx_min)
        };
        (s as usize, e as usize)
    }

    fn query(
        &self,
        min_x: f64,
        min_y: f64,
        max_x: f64,
        max_y: f64,
        marker: &mut QueryMarker,
        buffer: &mut Vec<usize>,
    ) {
        buffer.clear();
        if self.cols == 0 || self.rows == 0 {
            return;
        }
        let (sx, ex) = Self::range(min_x, max_x, self.origin_x, self.cell_w, self.cols);
        let (sy, ey) = Self::range(min_y, max_y, self.origin_y, self.cell_h, self.rows);
        marker.next();
        for ix in sx..=ex {
            for iy in sy..=ey {
                let bucket = &self.buckets[iy * self.cols + ix];
                for &edge_idx in bucket {
                    if marker.mark(edge_idx) {
                        buffer.push(edge_idx);
                    }
                }
            }
        }
    }
}

struct StockGeometry {
    polygon: Vec<Point>,
    bbox: Rect,
    edges: Vec<Edge>,
    grid: SpatialGrid,
    centroid: Point,
}

impl StockGeometry {
    fn new(vertices: Vec<Point>) -> Self {
        let bbox_aabb = compute_aabb(&vertices);
        let bbox = Rect::from(bbox_aabb);
        let mut edges = Vec::with_capacity(vertices.len());
        if !vertices.is_empty() {
            for i in 0..vertices.len() {
                let a = vertices[i];
                let b = vertices[(i + 1) % vertices.len()];
                let edge_bbox = Aabb {
                    min_x: a.x.min(b.x),
                    min_y: a.y.min(b.y),
                    max_x: a.x.max(b.x),
                    max_y: a.y.max(b.y),
                };
                edges.push(Edge {
                    start: a,
                    end: b,
                    bbox: edge_bbox,
                });
            }
        }
        let grid = SpatialGrid::new(&bbox, &edges);
        let centroid = polygon_centroid(&vertices);
        Self {
            polygon: vertices,
            bbox,
            edges,
            grid,
            centroid,
        }
    }

    fn bounding_rect(&self) -> Rect {
        self.bbox
    }

    fn centroid(&self) -> Point {
        self.centroid
    }

    fn polygon_points(&self) -> &[Point] {
        &self.polygon
    }

    fn point_inside(&self, point: Point) -> bool {
        let n = self.polygon.len();
        if n == 0 {
            return false;
        }
        if n == 1 {
            return ((self.polygon[0].x - point.x).abs() < FIT_EPS)
                && ((self.polygon[0].y - point.y).abs() < FIT_EPS);
        }
        if n == 2 {
            return point_on_segment(point, self.polygon[0], self.polygon[1]);
        }
        let mut inside = false;
        let mut j = n - 1;
        for i in 0..n {
            let pi = self.polygon[i];
            let pj = self.polygon[j];
            if point_on_segment(point, pi, pj) {
                return true;
            }
            let intersect = ((pi.y > point.y) != (pj.y > point.y))
                && (point.x
                    < (pj.x - pi.x) * (point.y - pi.y) / (pj.y - pi.y + EPS) + pi.x);
            if intersect {
                inside = !inside;
            }
            j = i;
        }
        inside
    }

    fn piece_fits(
        &self,
        rotated: &[Point],
        tx: f64,
        ty: f64,
        marker: &mut QueryMarker,
        buffer: &mut Vec<usize>,
    ) -> bool {
        for v in rotated.iter() {
            let p = Point {
                x: v.x + tx,
                y: v.y + ty,
            };
            if !self.point_inside(p) {
                return false;
            }
        }
        if rotated.len() < 2 {
            return true;
        }
        for i in 0..rotated.len() {
            let a = rotated[i];
            let b = rotated[(i + 1) % rotated.len()];
            let pa = Point {
                x: a.x + tx,
                y: a.y + ty,
            };
            let pb = Point {
                x: b.x + tx,
                y: b.y + ty,
            };
            if self.segment_crosses_boundary(pa, pb, marker, buffer) {
                return false;
            }
        }
        true
    }

    fn segment_crosses_boundary(
        &self,
        a: Point,
        b: Point,
        marker: &mut QueryMarker,
        buffer: &mut Vec<usize>,
    ) -> bool {
        if (a.x - b.x).abs() < EPS && (a.y - b.y).abs() < EPS {
            return false;
        }
        let min_x = a.x.min(b.x);
        let max_x = a.x.max(b.x);
        let min_y = a.y.min(b.y);
        let max_y = a.y.max(b.y);
        self.grid
            .query(min_x, min_y, max_x, max_y, marker, buffer);
        for &edge_idx in buffer.iter() {
            let edge = &self.edges[edge_idx];
            if segments_strictly_intersect(a, b, edge.start, edge.end) {
                return true;
            }
        }
        false
    }
}

struct GeometryHelper {
    marker: QueryMarker,
    buffer: Vec<usize>,
}

impl GeometryHelper {
    fn new(edge_count: usize) -> Self {
        Self {
            marker: QueryMarker::new(edge_count),
            buffer: Vec::new(),
        }
    }

    fn fits(
        &mut self,
        geom: &StockGeometry,
        rotated: &[Point],
        tx: f64,
        ty: f64,
    ) -> bool {
        geom.piece_fits(rotated, tx, ty, &mut self.marker, &mut self.buffer)
    }
}

struct PlacementResult {
    stock_index: usize,
    translation_x: f64,
    translation_y: f64,
    rotation: f64,
}

struct Candidate {
    rect: Rect,
    orient_idx: usize,
    tx: f64,
    ty: f64,
    leftover_short: f64,
    leftover_long: f64,
    y: f64,
    x: f64,
}

struct StockState {
    index: usize,
    bounding_rect: Rect,
    total_area: f64,
    free_rects: Vec<Rect>,
    used_rects: Vec<Rect>,
    area_used: f64,
}

impl StockState {
    fn new(index: usize, bounding_rect: Rect) -> Self {
        let mut free_rects = Vec::new();
        if bounding_rect.valid() || (bounding_rect.width >= 0.0 && bounding_rect.height >= 0.0) {
            free_rects.push(bounding_rect);
        }
        Self {
            index,
            bounding_rect,
            total_area: bounding_rect.area().max(EPS),
            free_rects,
            used_rects: Vec::new(),
            area_used: 0.0,
        }
    }

    fn fill_ratio(&self) -> f64 {
        if self.total_area <= 0.0 {
            0.0
        } else {
            (self.area_used / self.total_area).clamp(0.0, 1.0)
        }
    }

    fn try_place_piece(
        &mut self,
        piece: &Piece,
        geom: &StockGeometry,
        helper: &mut GeometryHelper,
        rng: &mut Rng,
    ) -> Option<PlacementResult> {
        if piece.orientations.is_empty() {
            return None;
        }
        if let Some(res) = self.try_maxrects(piece, geom, helper) {
            return Some(res);
        }
        if let Some(res) =
            self.try_align_to_point(piece, geom, helper, geom.centroid())
        {
            return Some(res);
        }
        let rect_center = Point {
            x: self.bounding_rect.x + self.bounding_rect.width * 0.5,
            y: self.bounding_rect.y + self.bounding_rect.height * 0.5,
        };
        if let Some(res) = self.try_align_to_point(piece, geom, helper, rect_center) {
            return Some(res);
        }
        if let Some(res) = self.try_grid(piece, geom, helper, BASE_GRID_STEPS) {
            return Some(res);
        }
        if self.used_rects.is_empty() {
            if let Some(res) = self.try_grid(piece, geom, helper, DENSE_GRID_STEPS) {
                return Some(res);
            }
            if let Some(res) =
                self.try_vertex_alignment(piece, geom, helper, 64)
            {
                return Some(res);
            }
            if let Some(res) =
                self.try_random(piece, geom, helper, rng, RANDOM_ATTEMPTS * 8)
            {
                return Some(res);
            }
        }
        if let Some(res) =
            self.try_random(piece, geom, helper, rng, RANDOM_ATTEMPTS)
        {
            return Some(res);
        }
        None
    }

    fn force_place_piece(
        &mut self,
        piece: &Piece,
        geom: &StockGeometry,
        helper: &mut GeometryHelper,
        rng: &mut Rng,
    ) -> Option<PlacementResult> {
        if let Some(res) = self.try_grid(piece, geom, helper, FORCE_GRID_STEPS) {
            return Some(res);
        }
        if let Some(res) =
            self.try_vertex_alignment(piece, geom, helper, geom.polygon_points().len().max(1))
        {
            return Some(res);
        }
        self.try_random(
            piece,
            geom,
            helper,
            rng,
            RANDOM_ATTEMPTS * 400,
        )
    }

    fn finalize(
        &mut self,
        piece: &Piece,
        orient_idx: usize,
        rect: Rect,
        tx: f64,
        ty: f64,
    ) -> PlacementResult {
        self.area_used += rect.area();
        self.used_rects.push(rect);
        self.split_free_rects(rect);
        self.prune_free_rects();
        if self.free_rects.len() > MAX_FREE_RECTS {
            self.free_rects
                .sort_by(|a, b| b.area().partial_cmp(&a.area()).unwrap_or(Ordering::Equal));
            self.free_rects.truncate(MAX_FREE_RECTS);
        }
        PlacementResult {
            stock_index: self.index,
            translation_x: tx,
            translation_y: ty,
            rotation: piece.orientations[orient_idx].angle,
        }
    }

    fn try_maxrects(
        &mut self,
        piece: &Piece,
        geom: &StockGeometry,
        helper: &mut GeometryHelper,
    ) -> Option<PlacementResult> {
        let mut best: Option<Candidate> = None;
        for free_rect in self.free_rects.iter() {
            for (orient_idx, variant) in piece.orientations.iter().enumerate() {
                if variant.width - free_rect.width > FIT_EPS
                    || variant.height - free_rect.height > FIT_EPS
                {
                    continue;
                }
                let (off_x, count_x) = sample_offsets(free_rect.width, variant.width);
                let (off_y, count_y) =
                    sample_offsets(free_rect.height, variant.height);
                for ix in 0..count_x {
                    for iy in 0..count_y {
                        let max_x = free_rect.x + free_rect.width - variant.width;
                        let max_y = free_rect.y + free_rect.height - variant.height;
                        let mut place_x = free_rect.x + off_x[ix];
                        let mut place_y = free_rect.y + off_y[iy];
                        if max_x > free_rect.x {
                            place_x = clamp(place_x, free_rect.x, max_x);
                        } else {
                            place_x = free_rect.x;
                        }
                        if max_y > free_rect.y {
                            place_y = clamp(place_y, free_rect.y, max_y);
                        } else {
                            place_y = free_rect.y;
                        }
                        let rect = Rect {
                            x: place_x,
                            y: place_y,
                            width: variant.width,
                            height: variant.height,
                        };
                        if !self.is_area_free(&rect) {
                            continue;
                        }
                        let tx = place_x - variant.bbox.min_x;
                        let ty = place_y - variant.bbox.min_y;
                        if !helper.fits(geom, &variant.rotated, tx, ty) {
                            continue;
                        }
                        let leftover_w = free_rect.width - variant.width;
                        let leftover_h = free_rect.height - variant.height;
                        let candidate = Candidate {
                            rect,
                            orient_idx,
                            tx,
                            ty,
                            leftover_short: leftover_w.min(leftover_h),
                            leftover_long: leftover_w.max(leftover_h),
                            y: place_y,
                            x: place_x,
                        };
                        if best
                            .as_ref()
                            .map_or(true, |b| candidate_better(&candidate, b))
                        {
                            best = Some(candidate);
                        }
                    }
                }
            }
        }
        best.map(|c| self.finalize(piece, c.orient_idx, c.rect, c.tx, c.ty))
    }

    fn try_align_to_point(
        &mut self,
        piece: &Piece,
        geom: &StockGeometry,
        helper: &mut GeometryHelper,
        anchor: Point,
    ) -> Option<PlacementResult> {
        for (orient_idx, variant) in piece.orientations.iter().enumerate() {
            let x_min = self.bounding_rect.x;
            let x_max = self.bounding_rect.right() - variant.width;
            if x_max + FIT_EPS < x_min {
                continue;
            }
            let y_min = self.bounding_rect.y;
            let y_max = self.bounding_rect.top() - variant.height;
            if y_max + FIT_EPS < y_min {
                continue;
           
        }
        None
    }
}

fn candidate_better(a: &Candidate, b: &Candidate) -> bool {
    if (a.leftover_short - b.leftover_short).abs() > FIT_EPS {
        return a.leftover_short < b.leftover_short;
    }
    if (a.leftover_long - b.leftover_long).abs() > FIT_EPS {
        return a.leftover_long < b.leftover_long;
    }
    if (a.y - b.y).abs() > FIT_EPS {
        return a.y < b.y;
    }
    if (a.x - b.x).abs() > FIT_EPS {
        return a.x < b.x;
    }
    a.orient_idx < b.orient_idx
}

fn sample_offsets(span: f64, size: f64) -> ([f64; OFFSET_CAP], usize) {
    let mut arr = [0.0; OFFSET_CAP];
    let mut count = 0;
    if span + FIT_EPS < size {
        return (arr, 0);
    }
    arr[count] = 0.0;
    count += 1;
    let slack = (span - size).max(0.0);
    if slack <= FIT_EPS {
        return (arr, count);
    }
    add_offset(&mut arr, &mut count, slack);
    add_offset(&mut arr, &mut count, slack * 0.5);
    if slack > size * 0.5 {
        add_offset(&mut arr, &mut count, slack * 0.25);
        add_offset(&mut arr, &mut count, slack * 0.75);
    }
    (arr, count)
}

fn add_offset(arr: &mut [f64; OFFSET_CAP], count: &mut usize, value: f64) {
    if *count >= OFFSET_CAP {
        return;
    }
    for i in 0..*count {
        if (arr[i] - value).abs() < 1e-6 {
            return;
        }
    }
    arr[*count] = value;
    *count += 1;
}

fn generate_positions(min_v: f64, max_v: f64, steps: usize) -> ([f64; GRID_POS_CAP], usize) {
    let mut arr = [0.0; GRID_POS_CAP];
    if steps == 0 || max_v <= min_v + FIT_EPS {
        arr[0] = min_v.min(max_v);
        return (arr, 1);
    }
    let steps = steps.min(GRID_POS_CAP - 1).max(1);
    for i in 0..=steps {
        let t = i as f64 / steps as f64;
        arr[i] = min_v + (max_v - min_v) * t;
    }
    (arr, steps + 1)
}

fn clamp(value: f64, min_v: f64, max_v: f64) -> f64 {
    if value < min_v {
        min_v
    } else if value > max_v {
        max_v
    } else {
        value
    }
}

fn orientation(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let cross = orientation(a, b, p);
    if cross.abs() > EPS {
        return false;
    }
    let min_x = a.x.min(b.x) - FIT_EPS;
    let max_x = a.x.max(b.x) + FIT_EPS;
    let min_y = a.y.min(b.y) - FIT_EPS;
    let max_y = a.y.max(b.y) + FIT_EPS;
    p.x >= min_x && p.x <= max_x && p.y >= min_y && p.y <= max_y
}

fn segments_strictly_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orientation(a1, a2, b1);
    let o2 = orientation(a1, a2, b2);
    let o3 = orientation(b1, b2, a1);
    let o4 = orientation(b1, b2, a2);
    ((o1 > EPS && o2 < -EPS) || (o1 < -EPS && o2 > EPS))
        && ((o3 > EPS && o4 < -EPS) || (o3 < -EPS && o4 > EPS))
}

fn compute_aabb(vertices: &[Point]) -> Aabb {
    let mut bbox = Aabb::new_empty();
    if vertices.is_empty() {
        bbox.min_x = 0.0;
        bbox.min_y = 0.0;
        bbox.max_x = 0.0;
        bbox.max_y = 0.0;
        return bbox;
    }
    for &v in vertices {
        bbox.include(v);
    }
    bbox
}

fn polygon_area(vertices: &[Point]) -> f64 {
    if vertices.len() < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        sum += (a.x * b.y) - (b.x * a.y);
    }
    0.5 * sum
}

fn polygon_centroid(vertices: &[Point]) -> Point {
    if vertices.is_empty() {
        return Point { x: 0.0, y: 0.0 };
    }
    if vertices.len() < 3 {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for v in vertices {
            sx += v.x;
            sy += v.y;
        }
        let k = vertices.len() as f64;
        return Point {
            x: sx / k,
            y: sy / k,
        };
    }
    let mut area = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        let cross = a.x * b.y - b.x * a.y;
        area += cross;
        cx += (a.x + b.x) * cross;
        cy += (a.y + b.y) * cross;
    }
    area *= 0.5;
    if area.abs() < EPS {
        return Point {
            x: vertices.iter().map(|p| p.x).sum::<f64>() / vertices.len() as f64,
            y: vertices.iter().map(|p| p.y).sum::<f64>() / vertices.len() as f64,
        };
    }
    cx /= 6.0 * area;
    cy /= 6.0 * area;
    Point { x: cx, y: cy }
}

fn longest_edge_angle(vertices: &[Point]) -> Option<f64> {
    if vertices.len() < 2 {
        return None;
    }
    let mut max_len = 0.0;
    let mut angle = None;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        let dx = b.x - a.x;
        let dy = b.y - a.y;
        let len = dx * dx + dy * dy;
        if len > max_len {
            max_len = len;
            angle = Some(dy.atan2(dx));
        }
    }
    angle
}

fn normalize_angle(angle: f64) -> f64 {
    let mut a = angle % TWO_PI;
    if a < 0.0 {
        a += TWO_PI;
    }
    a
}

fn build_orientations(vertices: &[Point]) -> Vec<OrientedVariant> {
    let mut angles = Vec::new();
    for i in 0..BASE_ANGLE_STEPS {
        let step =
            i as f64 * (TWO_PI / BASE_ANGLE_STEPS as f64);
        angles.push(normalize_angle(step));
    }
    if let Some(edge_angle) = longest_edge_angle(vertices) {
        angles.push(normalize_angle(edge_angle));
        angles.push(normalize_angle(edge_angle + std::f64::consts::FRAC_PI_2));
    }
    angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mut filtered = Vec::new();
    for angle in angles {
        if filtered.iter().any(|&v| (v - angle).abs() < 1e-4) {
            continue;
        }
        filtered.push(angle);
    }
    if filtered.is_empty() {
        filtered.push(0.0);
    }
    if filtered.len() > MAX_ORIENTATION_SET {
        let mut trimmed = Vec::new();
        for i in 0..MAX_ORIENTATION_SET {
            let idx = i * filtered.len() / MAX_ORIENTATION_SET;
            let angle = filtered[idx];
            if trimmed.iter().any(|&v| (v - angle).abs() < 1e-4) {
                continue;
            }
            trimmed.push(angle);
        }
        filtered = trimmed;
    }
    filtered
        .into_iter()
        .map(|angle| OrientedVariant::from_angle(vertices, angle))
        .collect()
}

impl OrientedVariant {
    fn from_angle(vertices: &[Point], angle: f64) -> Self {
        let cos_t = angle.cos();
        let sin_t = angle.sin();
        let mut rotated = Vec::with_capacity(vertices.len());
        let mut bbox = Aabb::new_empty();
        if vertices.is_empty() {
            rotated.push(Point { x: 0.0, y: 0.0 });
            bbox = Aabb {
                min_x: 0.0,
                min_y: 0.0,
                max_x: 0.0,
                max_y: 0.0,
            };
        } else {
            for &p in vertices {
                let x = p.x * cos_t - p.y * sin_t;
                let y = p.x * sin_t + p.y * cos_t;
                let point = Point { x, y };
                bbox.include(point);
                rotated.push(point);
            }
        }
        let width = (bbox.max_x - bbox.min_x).max(0.0);
        let height = (bbox.max_y - bbox.min_y).max(0.0);
        Self {
            angle,
            rotated,
            bbox,
            width,
            height,
        }
    }
}

impl Piece {
    fn new(index: usize, vertices: Vec<Point>) -> Self {
        let bbox = compute_aabb(&vertices);
        let mut score = bbox.width() * bbox.height();
        if score <= 0.0 {
            score = polygon_area(&vertices).abs();
        }
        let orientations = build_orientations(&vertices);
        Self {
            index,
            vertices,
            score,
            orientations,
        }
    }
}

struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        self.state = self
            .state
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1);
        self.state
    }
    fn next_f64(&mut self) -> f64 {
        const SCALE: f64 = 1.0 / ((1u64 << 53) as f64);
        ((self.next_u64() >> 11) as f64) * SCALE
    }
    fn mix(&mut self, value: u64) {
        self.state ^= value.wrapping_mul(0x9E3779B97F4A7C15);
        let _ = self.next_u64();
    }
}

fn read_usize(iter: &mut std::str::SplitWhitespace) -> usize {
    iter.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0)
}

fn read_f64(iter: &mut std::str::SplitWhitespace) -> f64 {
    iter.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    let stock_vertices = read_usize(&mut iter);
    let piece_count = read_usize(&mut iter);

    let mut stock_points = Vec::with_capacity(stock_vertices);
    for _ in 0..stock_vertices {
        let x = read_f64(&mut iter);
        let y = read_f64(&mut iter);
        stock_points.push(Point { x, y });
    }
    let geometry = StockGeometry::new(stock_points);
    let mut pieces = Vec::with_capacity(piece_count);
    for idx in 0..piece_count {
        let pv = read_usize(&mut iter);
        let mut vertices = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = read_f64(&mut iter);
            let y = read_f64(&mut iter);
            vertices.push(Point { x, y });
        }
        pieces.push(Piece::new(idx, vertices));
    }

    let mut order: Vec<usize> = (0..pieces.len()).collect();
    order.sort_by(|&a, &b| {
        pieces[b]
            .score
            .partial_cmp(&pieces[a].score)
            .unwrap_or(Ordering::Equal)
    });

    let mut placements = vec![None; pieces.len()];
    let mut stocks: Vec<StockState> = Vec::new();
    let mut helper = GeometryHelper::new(geometry.edges.len());
    let mut rng = Rng::new(0x1234_5678_9ABC_DEF0);

    for &piece_idx in &order {
        let piece = &pieces[piece_idx];
        let mut placed = None;

        if !stocks.is_empty() {
            let mut stock_order: Vec<usize> = (0..stocks.len()).collect();
            stock_order.sort_by(|&a, &b| {
                stocks[b]
                    .fill_ratio()
                    .partial_cmp(&stocks[a].fill_ratio())
                    .unwrap_or(Ordering::Equal)
            });
            for idx in stock_order {
                let res = {
                    let stock = &mut stocks[idx];
                    stock.try_place_piece(piece, &geometry, &mut helper, &mut rng)
                };
                if let Some(r) = res {
                    placed = Some(r);
                    break;
                }
            }
        }

        if placed.is_none() {
            let new_index = stocks.len();
            let mut new_stock = StockState::new(new_index, geometry.bounding_rect());
            let mut attempt =
                new_stock.try_place_piece(piece, &geometry, &mut helper, &mut rng);
            if attempt.is_none() {
                attempt =
                    new_stock.force_place_piece(piece, &geometry, &mut helper, &mut rng);
            }
            if let Some(res) = attempt {
                placements[piece_idx] = Some(res);
                stocks.push(new_stock);
                continue;
            } else {
                // As a last resort, continue forcing with much larger random search.
                let mut rescue = None;
                for _ in 0..5 {
                    rescue = new_stock.force_place_piece(
                        piece,
                        &geometry,
                        &mut helper,
                        &mut rng,
                    );
                    if rescue.is_some() {
                        break;
                    }
                }
                if let Some(res) = rescue {
                    placements[piece_idx] = Some(res);
                    stocks.push(new_stock);
                    continue;
                } else {
                    panic!("Unable to place piece {}", piece_idx);
                }
            }
        } else {
            placements[piece_idx] = placed;
        }
    }

    println!("{}", stocks.len());
    for idx in 0..pieces.len() {
        if let Some(p) = &placements[idx] {
            println!(
                "{} {:.9} {:.9} {:.9}",
                p.stock_index, p.translation_x, p.translation_y, p.rotation
            );
        } else {
            println!("0 0.0 0.0 0.0");
        }
    }
}
