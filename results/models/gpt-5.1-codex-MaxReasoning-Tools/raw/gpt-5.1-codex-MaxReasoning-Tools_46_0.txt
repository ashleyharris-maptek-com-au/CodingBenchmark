```glsl
#version 450

layout(local_size_x = 256) in;

// Each body occupies two vec4 elements in the buffers, as described.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 physics;   // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    uint baseIdx = i * 2u;
    vec4 posMass = inData[baseIdx];
    vec4 velVec  = inData[baseIdx + 1u];

    vec3 pos = posMass.xyz;
    float mass = posMass.w;
    vec3 vel = velVec.xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint jBase = j * 2u;
        vec4 otherPosMass = inData[jBase];
        vec3 r = otherPosMass.xyz - pos;
        float distSqr = dot(r, r) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * otherPosMass.w * r * invDist3;
    }

    vel += acc * dt;
    pos += vel * dt;

    outData[baseIdx]     = vec4(pos, mass);
    outData[baseIdx + 1u] = vec4(vel, 0.0);
}
```
