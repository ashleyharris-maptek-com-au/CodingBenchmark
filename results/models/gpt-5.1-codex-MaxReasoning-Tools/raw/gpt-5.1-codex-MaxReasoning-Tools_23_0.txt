```rust
use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, BufWriter, ErrorKind, Write};
use std::sync::mpsc::{self, TryRecvError};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct WorldParams {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start: Point,
    target: Point,
}

#[derive(Clone, Copy, Debug)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
    received_at: Instant,
}

#[derive(Clone, Copy, Debug)]
struct Command {
    thrust: f64,
    turn: f64,
}

struct Controller {
    params: WorldParams,
}

impl Controller {
    fn new(params: WorldParams) -> Self {
        Self { params }
    }

    fn command(&mut self, state: &LanderState) -> Command {
        let altitude = (self.params.target.y - state.y).max(-200.0);
        let horizontal_error = self.params.target.x - state.x;

        let initial_fuel = self.params.fuel.max(1.0);
        let fuel_ratio = (state.fuel_remaining / initial_fuel).clamp(0.0, 1.0);

        let tilt_limit = self
            .max_tilt_for_altitude(altitude)
            * (0.6 + 0.4 * fuel_ratio);

        let desired_vx = (horizontal_error * 0.05 * (0.5 + fuel_ratio)).clamp(-8.0, 8.0);
        let vx_error = desired_vx - state.vx;
        let mut desired_angle = (vx_error * 0.18).clamp(-tilt_limit, tilt_limit);

        if altitude < 25.0 {
            desired_angle = desired_angle.clamp(-0.3, 0.3);
        }
        if altitude < 10.0 {
            desired_angle *= 0.5;
        }
        if altitude < 3.0 {
            desired_angle = 0.0;
        }

        let angle_error = normalize_angle(desired_angle - state.angle);
        let turn = (angle_error * 3.0 - state.angular_vel * 0.8).clamp(-1.0, 1.0);

        let target_vy = self.desired_descent_speed(altitude);
        let vy_error = target_vy - state.vy;
        let altitude_term = (altitude * 0.015).clamp(-2.0, 2.0);
        let desired_ay = vy_error * 0.9 + altitude_term;

        let cosine = state.angle.cos().abs().max(0.1);
        let mut thrust = (self.params.gravity - desired_ay) / (self.params.max_thrust * cosine);
        thrust = thrust.clamp(0.0, 1.0);

        if altitude <= 2.0 {
            thrust = thrust.max(0.65);
        } else if altitude <= 5.0 && state.vy > 1.5 {
            thrust = thrust.max(0.75);
        }

        Command { thrust, turn }
    }

    fn desired_descent_speed(&self, altitude: f64) -> f64 {
        if altitude <= -5.0 {
            -2.3
        } else if altitude <= 0.0 {
            -0.8
        } else if altitude > 200.0 {
            12.0
        } else if altitude > 120.0 {
            9.0
        } else if altitude > 80.0 {
            7.0
        } else if altitude > 50.0 {
            5.0
        } else if altitude > 30.0 {
            3.5
        } else if altitude > 15.0 {
            2.2
        } else if altitude > 7.0 {
            1.4
        } else if altitude > 3.0 {
            0.9
        } else {
            0.3
        }
    }

    fn max_tilt_for_altitude(&self, altitude: f64) -> f64 {
        if altitude > 150.0 {
            0.8
        } else if altitude > 80.0 {
            0.65
        } else if altitude > 40.0 {
            0.45
        } else if altitude > 15.0 {
            0.30
        } else {
            0.18
        }
    }
}

fn normalize_angle(mut angle: f64) -> f64 {
    while angle > PI {
        angle -= 2.0 * PI;
    }
    while angle < -PI {
        angle += 2.0 * PI;
    }
    angle
}

fn main() {
    if let Err(err) = run() {
        eprintln!("controller terminated: {err}");
    }
}

fn run() -> io::Result<()> {
    let (world_tx, world_rx) = mpsc::channel();
    let (state_tx, state_rx) = mpsc::channel();

    let reader_handle = thread::spawn(move || {
        let stdin = io::stdin();
        let mut reader = BufReader::new(stdin.lock());
        if let Err(err) = stream_input(&mut reader, world_tx, state_tx) {
            eprintln!("input reader error: {err}");
        }
    });

    let world = match world_rx.recv() {
        Ok(world) => world,
        Err(_) => {
            let _ = reader_handle.join();
            return Ok(());
        }
    };

    let first_state = match state_rx.recv() {
        Ok(state) => state,
        Err(_) => {
            let _ = reader_handle.join();
            return Ok(());
        }
    };

    let mut controller = Controller::new(world);
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    let mut latest_state = first_state;
    let control_period = Duration::from_micros(25_000); // 40 Hz
    let mut next_tick = Instant::now();

    loop {
        let mut disconnected = false;
        loop {
            match state_rx.try_recv() {
                Ok(state) => {
                    latest_state = state;
                }
                Err(TryRecvError::Empty) => break,
                Err(TryRecvError::Disconnected) => {
                    disconnected = true;
                    break;
                }
            }
        }

        let cmd = controller.command(&latest_state);
        writeln!(
            writer,
            "{:.4} {:.4}",
            cmd.thrust.clamp(0.0, 1.0),
            cmd.turn.clamp(-1.0, 1.0)
        )?;
        writer.flush()?;

        if disconnected {
            break;
        }

        next_tick = next_tick + control_period;
        let now = Instant::now();
        if let Some(wait) = next_tick.checked_duration_since(now) {
            thread::sleep(wait);
        } else {
            next_tick = now;
        }
    }

    let _ = reader_handle.join();
    Ok(())
}

fn stream_input<R: BufRead>(
    reader: &mut R,
    world_tx: mpsc::Sender<WorldParams>,
    state_tx: mpsc::Sender<LanderState>,
) -> io::Result<()> {
    let world = read_initial_configuration(reader)?;
    let _ = world_tx.send(world.clone());

    let mut line = String::new();
    loop {
        line.clear();
        let read = reader.read_line(&mut line)?;
        if read == 0 {
            break;
        }
        if line.trim().is_empty() {
            continue;
        }
        if let Some(state) = parse_state_line(&line) {
            if state_tx.send(state).is_err() {
                break;
            }
        }
    }
    Ok(())
}

fn parse_state_line(line: &str) -> Option<LanderState> {
    let mut parts = line.split_ascii_whitespace();
    let x = parts.next()?.parse().ok()?;
    let y = parts.next()?.parse().ok()?;
    let vx = parts.next()?.parse().ok()?;
    let vy = parts.next()?.parse().ok()?;
    let angle = parts.next()?.parse().ok()?;
    let angular_vel = parts.next()?.parse().ok()?;
    let fuel_remaining = parts.next()?.parse().ok()?;
    Some(LanderState {
        x,
        y,
        vx,
        vy,
        angle,
        angular_vel,
        fuel_remaining,
        received_at: Instant::now(),
    })
}

fn read_initial_configuration<R: BufRead>(reader: &mut R) -> io::Result<WorldParams> {
    let mut line = String::new();

    if read_non_empty_line(reader, &mut line)? == 0 {
        return Err(io::Error::new(ErrorKind::UnexpectedEof, "missing header"));
    }
    let mut header = line.split_whitespace();
    let width = parse_dimension(header.next())?;
    let height = parse_dimension(header.next())?;
    let gravity = parse_float(header.next())?;
    let max_thrust = parse_float(header.next())?;
    let fuel = parse_float(header.next())?;

    if read_non_empty_line(reader, &mut line)? == 0 {
        return Err(io::Error::new(
            ErrorKind::UnexpectedEof,
            "missing start/target line",
        ));
    }
    let mut coords = line.split_whitespace();
    let start_x = parse_float(coords.next())?;
    let start_y = parse_float(coords.next())?;
    let target_x = parse_float(coords.next())?;
    let target_y = parse_float(coords.next())?;

    skip_map(reader, height)?;

    Ok(WorldParams {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start: Point { x: start_x, y: start_y },
        target: Point { x: target_x, y: target_y },
    })
}

fn read_non_empty_line<R: BufRead>(reader: &mut R, buf: &mut String) -> io::Result<usize> {
    loop {
        buf.clear();
        let bytes = reader.read_line(buf)?;
        if bytes == 0 || !buf.trim().is_empty() {
            return Ok(bytes);
        }
    }
}

fn parse_float(token: Option<&str>) -> io::Result<f64> {
    let raw = token.ok_or_else(|| io::Error::new(ErrorKind::UnexpectedEof, "missing value"))?;
    raw.parse::<f64>()
        .map_err(|_| io::Error::new(ErrorKind::InvalidData, "invalid number"))
}

fn parse_dimension(token: Option<&str>) -> io::Result<usize> {
    let value = parse_float(token)?;
    if value < 0.0 {
        return Err(io::Error::new(
            ErrorKind::InvalidData,
            "negative dimension",
        ));
    }
    Ok(value.round() as usize)
}

fn skip_map<R: BufRead>(reader: &mut R, rows: usize) -> io::Result<()> {
    for _ in 0..rows {
        loop {
            let buffer = reader.fill_buf()?;
            if buffer.is_empty() {
                return Err(io::Error::new(
                    ErrorKind::UnexpectedEof,
                    "map truncated before state stream",
                ));
            }
            if let Some(pos) = buffer.iter().position(|&b| b == b'\n') {
                reader.consume(pos + 1);
                break;
            } else {
                let len = buffer.len();
                reader.consume(len);
            }
        }
    }
    Ok(())
}
```
