declare const require: any;
const fs = require('fs');

interface Town {
    id: string;
    x: number;
    y: number;
    value: number;
    idx: number;
}

interface CandidatePlan {
    cost: number;
    savedValue: number;
    boundary: number[];
    towns: number[];
    score: number;
}

const EPS = 1e-9;

function main(): void {
    const rawInput: string = fs.readFileSync(0, 'utf8');
    const trimmed = rawInput.trim();
    if (!trimmed) {
        console.log(JSON.stringify({ firebreaks: [] }));
        return;
    }
    const data = JSON.parse(trimmed);
    const width = Math.max(0, Math.floor(Number(data.width) || 0));
    const height = Math.max(0, Math.floor(Number(data.height) || 0));
    const totalCells = width * height;
    if (width <= 0 || height <= 0 || totalCells <= 0) {
        console.log(JSON.stringify({ firebreaks: [] }));
        return;
    }
    const gridRows: string[] = Array.isArray(data.grid) ? data.grid.map((row: any) => (typeof row === 'string' ? row : '')) : [];
    const FOREST = 0;
    const BLOCKED = 1;
    const baseBlocked = new Uint8Array(totalCells);
    for (let y = 0; y < height; y++) {
        const row = gridRows[y] || '';
        for (let x = 0; x < width; x++) {
            const idx = y * width + x;
            const ch = row.charCodeAt(x);
            baseBlocked[idx] = ch === 46 ? FOREST : BLOCKED;
        }
    }
    const towns: Town[] = [];
    const townsByCell = new Map<number, number[]>();
    const cellHasTown = new Uint8Array(totalCells);
    if (Array.isArray(data.towns)) {
        for (const rawTown of data.towns) {
            if (!rawTown) continue;
            const rawX = Number(rawTown.x);
            const rawY = Number(rawTown.y);
            const rawValue = Number(rawTown.value);
            if (!Number.isFinite(rawX) || !Number.isFinite(rawY)) continue;
            const x = Math.floor(rawX);
            const y = Math.floor(rawY);
            if (x < 0 || x >= width || y < 0 || y >= height) continue;
            const idx = y * width + x;
            const safeValue = Number.isFinite(rawValue) && rawValue > 0 ? rawValue : 0;
            const town: Town = {
                id: rawTown.id !== undefined ? String(rawTown.id) : String(towns.length),
                x,
                y,
                value: safeValue,
                idx,
            };
            towns.push(town);
            cellHasTown[idx] = 1;
            if (!townsByCell.has(idx)) {
                townsByCell.set(idx, []);
            }
            townsByCell.get(idx)!.push(towns.length - 1);
        }
    }
    const townCount = towns.length;
    const cutBudgetRaw = Number(data.cutBudget);
    let remainingBudget = Number.isFinite(cutBudgetRaw) ? Math.max(0, Math.floor(cutBudgetRaw)) : 0;
    const isCut = new Uint8Array(totalCells);
    const fireSources: number[] = [];
    if (Array.isArray(data.ignitions)) {
        const seenSources = new Uint8Array(totalCells);
        for (const src of data.ignitions) {
            if (!Array.isArray(src) || src.length < 2) continue;
            const rawX = Number(src[0]);
            const rawY = Number(src[1]);
            if (!Number.isFinite(rawX) || !Number.isFinite(rawY)) continue;
            const x = Math.floor(rawX);
            const y = Math.floor(rawY);
            if (x < 0 || x >= width || y < 0 || y >= height) continue;
            const idx = y * width + x;
            if (baseBlocked[idx] !== FOREST) continue;
            if (seenSources[idx]) continue;
            seenSources[idx] = 1;
            fireSources.push(idx);
        }
    }
    const fireDist = new Int32Array(totalCells);
    for (let i = 0; i < totalCells; i++) {
        fireDist[i] = -1;
    }
    if (fireSources.length > 0) {
        const queue = new Int32Array(totalCells);
        let head = 0;
        let tail = 0;
        for (const src of fireSources) {
            fireDist[src] = 0;
            queue[tail++] = src;
        }
        while (head < tail) {
            const idx = queue[head++];
            const dist = fireDist[idx];
            const x = idx % width;
            const y = (idx / width) | 0;
            if (x + 1 < width) {
                const nb = idx + 1;
                if (baseBlocked[nb] === FOREST && fireDist[nb] === -1) {
                    fireDist[nb] = dist + 1;
                    queue[tail++] = nb;
                }
            }
            if (x - 1 >= 0) {
                const nb = idx - 1;
                if (baseBlocked[nb] === FOREST && fireDist[nb] === -1) {
                    fireDist[nb] = dist + 1;
                    queue[tail++] = nb;
                }
            }
            if (y + 1 < height) {
                const nb = idx + width;
                if (baseBlocked[nb] === FOREST && fireDist[nb] === -1) {
                    fireDist[nb] = dist + 1;
                    queue[tail++] = nb;
                }
            }
            if (y - 1 >= 0) {
                const nb = idx - width;
                if (baseBlocked[nb] === FOREST && fireDist[nb] === -1) {
                    fireDist[nb] = dist + 1;
                    queue[tail++] = nb;
                }
            }
        }
    }
    const secured = new Array<boolean>(townCount).fill(false);
    const atRisk = new Array<boolean>(townCount).fill(false);
    const relevantTowns: number[] = [];
    for (let i = 0; i < townCount; i++) {
        const town = towns[i];
        if (town.value <= 0) {
            secured[i] = true;
            continue;
        }
        const idx = town.idx;
        if (idx < 0 || idx >= totalCells) {
            secured[i] = true;
            continue;
        }
        if (baseBlocked[idx] !== FOREST) {
            secured[i] = true;
            continue;
        }
        if (fireSources.length === 0) {
            secured[i] = true;
            continue;
        }
        const dist = fireDist[idx];
        if (dist === -1) {
            secured[i] = true;
            continue;
        }
        if (dist === 0) {
            secured[i] = true;
            continue;
        }
        atRisk[i] = true;
        relevantTowns.push(i);
    }
    if (remainingBudget <= 0 || fireSources.length === 0 || relevantTowns.length === 0) {
        console.log(JSON.stringify({ firebreaks: [] }));
        return;
    }
    const townOrder = relevantTowns.slice();
    townOrder.sort((a, b) => {
        const townA = towns[a];
        const townB = towns[b];
        const distA = fireDist[townA.idx];
        const distB = fireDist[townB.idx];
        const scoreA = townA.value / Math.max(1, distA);
        const scoreB = townB.value / Math.max(1, distB);
        if (scoreB !== scoreA) {
            return scoreB - scoreA;
        }
        if (townB.value !== townA.value) {
            return townB.value - townA.value;
        }
        return distA - distB;
    });

    const regionSeen = new Uint32Array(totalCells);
    const boundarySeen = new Uint32Array(totalCells);
    const townSeen = new Uint32Array(Math.max(1, townCount));
    const regionTokenCounter = { value: 0 };
    const boundaryTokenCounter = { value: 0 };
    const townTokenCounter = { value: 0 };
    const selectedCuts: number[] = [];

    const nextToken = (counter: { value: number }, arr: Uint32Array): number => {
        counter.value++;
        if (counter.value >= 0xfffffffe) {
            counter.value = 1;
            arr.fill(0);
        }
        return counter.value;
    };

    const betterCandidate = (current: CandidatePlan | null, candidate: CandidatePlan): CandidatePlan => {
        if (!current) return candidate;
        if (candidate.score > current.score + EPS) return candidate;
        if (current.score > candidate.score + EPS) return current;
        if (candidate.savedValue > current.savedValue + EPS) return candidate;
        if (current.savedValue > candidate.savedValue + EPS) return current;
        if (candidate.cost < current.cost) return candidate;
        return current;
    };

    const evaluateTownRegion = (townIndex: number, budget: number): CandidatePlan | null => {
        if (secured[townIndex] || !atRisk[townIndex]) {
            return null;
        }
        const startIdx = towns[townIndex].idx;
        if (startIdx < 0 || startIdx >= totalCells) {
            return null;
        }
        if (baseBlocked[startIdx] !== FOREST || isCut[startIdx]) {
            return null;
        }
        const regionToken = nextToken(regionTokenCounter, regionSeen);
        const townToken = nextToken(townTokenCounter, townSeen);
        let boundaryToken = nextToken(boundaryTokenCounter, boundarySeen);
        const pendingList: number[] = [];
        const regionTownList: number[] = [];
        let regionValue = 0;
        let regionCellsCount = 0;
        let containsIgnition = false;
        const fireDistance = fireDist[startIdx];
        const EXTRA_DEPTH = 6;
        const MAX_DEPTH_CAP = 80;
        const maxDepth = Math.min(MAX_DEPTH_CAP, fireDistance > 0 ? fireDistance + EXTRA_DEPTH : MAX_DEPTH_CAP);
        const MAX_REGION_ABS = 20000;
        const MIN_REGION_LIMIT = 1000;
        const dynamicLimit = 2000 + budget * 40;
        const maxRegionCells = Math.min(totalCells, Math.min(MAX_REGION_ABS, Math.max(MIN_REGION_LIMIT, dynamicLimit)));
        let depth = 0;

        const includeCell = (idx: number): void => {
            if (regionSeen[idx] === regionToken) {
                return;
            }
            regionSeen[idx] = regionToken;
            regionCellsCount++;
            if (fireDist[idx] === 0) {
                containsIgnition = true;
            }
            if (cellHasTown[idx]) {
                const list = townsByCell.get(idx);
                if (list) {
                    for (const tIdx of list) {
                        if (secured[tIdx]) continue;
                        if (townSeen[tIdx] === townToken) continue;
                        townSeen[tIdx] = townToken;
                        regionTownList.push(tIdx);
                        if (atRisk[tIdx]) {
                            const val = towns[tIdx].value;
                            if (val > 0) {
                                regionValue += val;
                            }
                        }
                    }
                }
            }
        };

        const addBoundaryNeighbor = (idx: number): void => {
            if (regionSeen[idx] === regionToken) return;
            if (baseBlocked[idx] !== FOREST) return;
            if (isCut[idx]) return;
            if (boundarySeen[idx] === boundaryToken) return;
            boundarySeen[idx] = boundaryToken;
            pendingList.push(idx);
        };

        const addNeighborsToBoundary = (idx: number): void => {
            const x = idx % width;
            const y = (idx / width) | 0;
            if (x + 1 < width) addBoundaryNeighbor(idx + 1);
            if (x - 1 >= 0) addBoundaryNeighbor(idx - 1);
            if (y + 1 < height) addBoundaryNeighbor(idx + width);
            if (y - 1 >= 0) addBoundaryNeighbor(idx - width);
        };

        includeCell(startIdx);
        addNeighborsToBoundary(startIdx);
        let best: CandidatePlan | null = null;

        const evaluateBoundary = (): void => {
            if (regionValue <= 0) return;
            if (containsIgnition) return;
            const boundarySize = pendingList.length;
            if (boundarySize === 0) {
                const candidate: CandidatePlan = {
                    cost: 0,
                    savedValue: regionValue,
                    boundary: [],
                    towns: regionTownList.slice(),
                    score: regionValue / 0.5,
                };
                best = betterCandidate(best, candidate);
                return;
            }
            if (boundarySize > budget) {
                return;
            }
            for (const cell of pendingList) {
                if (cellHasTown[cell]) {
                    return;
                }
            }
            const candidate: CandidatePlan = {
                cost: boundarySize,
                savedValue: regionValue,
                boundary: pendingList.slice(),
                towns: regionTownList.slice(),
                score: regionValue / (boundarySize + 0.5),
            };
            best = betterCandidate(best, candidate);
        };

        while (true) {
            evaluateBoundary();
            if (pendingList.length === 0) {
                break;
            }
            if (containsIgnition) {
                break;
            }
            if (depth >= maxDepth) {
                break;
            }
            if (regionCellsCount + pendingList.length > maxRegionCells) {
                break;
            }
            depth++;
            const promoted = pendingList.slice();
            boundaryToken = nextToken(boundaryTokenCounter, boundarySeen);
            pendingList.length = 0;
            for (const idx of promoted) {
                includeCell(idx);
            }
            if (containsIgnition) {
                break;
            }
            for (const idx of promoted) {
                addNeighborsToBoundary(idx);
            }
        }

        return best;
    };

    const applyCandidate = (candidate: CandidatePlan): void => {
        if (candidate.cost > 0) {
            let applied = 0;
            for (const cell of candidate.boundary) {
                if (isCut[cell]) continue;
                isCut[cell] = 1;
                selectedCuts.push(cell);
                applied++;
            }
            if (applied > 0) {
                remainingBudget -= applied;
                if (remainingBudget < 0) {
                    remainingBudget = 0;
                }
            }
        }
        for (const tIdx of candidate.towns) {
            if (!secured[tIdx]) {
                secured[tIdx] = true;
                atRisk[tIdx] = false;
            }
        }
    };

    const tracePathTowardsFire = (startIdx: number, maxSteps: number): number[] => {
        const path: number[] = [];
        if (startIdx < 0 || startIdx >= totalCells) {
            return path;
        }
        let current = startIdx;
        let steps = 0;
        while (steps < maxSteps) {
            const currentDist = fireDist[current];
            if (currentDist <= 0) {
                break;
            }
            let nextIdx = -1;
            const x = current % width;
            const y = (current / width) | 0;
            if (x + 1 < width) {
                const nb = current + 1;
                if (fireDist[nb] === currentDist - 1) {
                    nextIdx = nb;
                }
            }
            if (nextIdx === -1 && x - 1 >= 0) {
                const nb = current - 1;
                if (fireDist[nb] === currentDist - 1) {
                    nextIdx = nb;
                }
            }
            if (nextIdx === -1 && y + 1 < height) {
                const nb = current + width;
                if (fireDist[nb] === currentDist - 1) {
                    nextIdx = nb;
                }
            }
            if (nextIdx === -1 && y - 1 >= 0) {
                const nb = current - width;
                if (fireDist[nb] === currentDist - 1) {
                    nextIdx = nb;
                }
            }
            if (nextIdx === -1) {
                break;
            }
            path.push(nextIdx);
            current = nextIdx;
            steps++;
        }
        return path;
    };

    const fallbackCuts = (): void => {
        const fallbackCandidates: { cell: number; score: number }[] = [];
        const PATH_LIMIT = 12;
        const STEPS_PER_TOWN = 4;
        for (const townIdx of townOrder) {
            if (remainingBudget <= 0) break;
            if (!atRisk[townIdx]) continue;
            if (secured[townIdx]) continue;
            const town = towns[townIdx];
            const path = tracePathTowardsFire(town.idx, PATH_LIMIT);
            if (path.length === 0) continue;
            let stepsUsed = 0;
            for (const cell of path) {
                stepsUsed++;
                if (cellHasTown[cell]) continue;
                if (baseBlocked[cell] !== FOREST) continue;
                if (isCut[cell]) continue;
                const score = town.value / Math.max(1, stepsUsed);
                fallbackCandidates.push({ cell, score });
                if (stepsUsed >= STEPS_PER_TOWN) {
                    break;
                }
            }
        }
        fallbackCandidates.sort((a, b) => b.score - a.score);
        for (const candidate of fallbackCandidates) {
            if (remainingBudget <= 0) break;
            const cell = candidate.cell;
            if (isCut[cell]) continue;
            if (cellHasTown[cell]) continue;
            if (baseBlocked[cell] !== FOREST) continue;
            isCut[cell] = 1;
            selectedCuts.push(cell);
            remainingBudget--;
        }
    };

    while (remainingBudget > 0) {
        let bestCandidate: CandidatePlan | null = null;
        for (const townIdx of townOrder) {
            if (secured[townIdx]) continue;
            const candidate = evaluateTownRegion(townIdx, remainingBudget);
            if (!candidate) continue;
            bestCandidate = betterCandidate(bestCandidate, candidate);
        }
        if (!bestCandidate) {
            break;
        }
        if (bestCandidate.cost > remainingBudget) {
            break;
        }
        applyCandidate(bestCandidate);
    }

    if (remainingBudget > 0) {
        fallbackCuts();
    }

    const firebreaks: number[][] = [];
    for (const idx of selectedCuts) {
        const x = idx % width;
        const y = (idx / width) | 0;
        firebreaks.push([x, y]);
    }
    console.log(JSON.stringify({ firebreaks }));
}

main();
