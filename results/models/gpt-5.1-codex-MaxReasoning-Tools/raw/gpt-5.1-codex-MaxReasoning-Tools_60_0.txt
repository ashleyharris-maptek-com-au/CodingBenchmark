class BinaryHeap {
    constructor(compare) {

    size() {

    peek() {
        return this.data[0];

    push(item) {

    pop() {

            return undefined;
        }
        const top = data[0];
        const last = data.pop();
        if (data.length > 0) {
            data[0] = last;
            this._siftDown(0);
        }
        return top;
    }
    _siftUp(index) {
        const data = this.data;
        const compare = this.compare;
        const item = data[index];
        while (index > 0) {
            const parent = (index - 1) >> 1;
            const parentItem = data[parent];
            if (compare(item, parentItem) < 0) {
                data[index] = parentItem;
                index = parent;
            } else {
                break;
            }
        }
        data[index] = item;
    }
    _siftDown(index) {
        const data = this.data;
        const compare = this.compare;
        const length = data.length;
        const item = data[index];
        while (true) {
            const left = index * 2 + 1;
            if (left >= length) {
                break;
            }
            const right = left + 1;
            let smallest = left;
            if (right < length && compare(data[right], data[left]) < 0) {

                index = smallest;
            } else {
                break;
            }

        data[index] = item;
    }
}
