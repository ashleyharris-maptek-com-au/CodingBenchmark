```cpp
#include <cstdint>
#include <cstdio>
#include <vector>
#include <unordered_map>
#include <queue>
#include <limits>
#include <algorithm>
#include <iostream>
#include <utility>
#include <new>

struct Edge {
    uint32_t u;
    uint32_t v;
};

class FastInput {
public:
    FastInput() : idx(0), len(0) {}
    template <typename T>
    bool readUnsigned(T &out) {
        static_assert(std::is_unsigned_v<T>);
        char c = read();
        while (c <= ' ') {
            if (!c) return false;
            c = read();
        }
        T value = 0;
        while (c > ' ') {
            value = value * 10 + static_cast<T>(c - '0');
            c = read();
        }
        out = value;
        return true;
    }
private:
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t idx;
    size_t len;
    char read() {
        if (idx >= len) {
            len = std::fread(buffer, 1, BUFSIZE, stdin);
            idx = 0;
            if (len == 0) return 0;
        }
        return buffer[idx++];
    }
};

template <typename EdgeIndexT>
bool findFirstBridge(const std::vector<Edge> &edges,
                     const std::vector<uint64_t> &offset,
                     const std::vector<EdgeIndexT> &incident,
                     uint32_t n,
                     uint64_t &bridgeIdx) {
    if (edges.empty()) return false;
    std::vector<int> disc(n, -1);
    std::vector<int> low(n, -1);
    std::vector<int> parent(n, -1);
    std::vector<uint64_t> parentEdge(n, std::numeric_limits<uint64_t>::max());
    struct Frame {
        uint32_t node;
        uint64_t it;
    };
    size_t reserveHint = static_cast<size_t>(std::min<uint32_t>(n, static_cast<uint32_t>(1 << 18)));
    std::vector<Frame> stack;
    stack.reserve(reserveHint);
    int timer = 0;
    for (uint32_t start = 0; start < n; ++start) {
        if (disc[start] != -1) continue;
        stack.clear();
        stack.push_back({start, offset[start]});
        parent[start] = -1;
        disc[start] = low[start] = timer++;
        while (!stack.empty()) {
            Frame &frame = stack.back();
            uint32_t u = frame.node;
            uint64_t &it = frame.it;
            if (it < offset[u + 1]) {
                uint64_t idx = static_cast<uint64_t>(incident[static_cast<size_t>(it++)]);
                const Edge &e = edges[static_cast<size_t>(idx)];
                uint32_t v = (e.u == u) ? e.v : e.u;
                if (disc[v] == -1) {
                    parent[v] = static_cast<int>(u);
                    parentEdge[v] = idx;
                    disc[v] = low[v] = timer++;
                    stack.push_back({v, offset[v]});
                } else if (static_cast<int>(v) != parent[u]) {
                    low[u] = std::min(low[u], disc[v]);
                }
            } else {
                stack.pop_back();
                if (parent[u] != -1) {
                    int p = parent[u];
                    low[p] = std::min(low[p], low[u]);
                    if (low[u] > disc[p]) {
                        bridgeIdx = parentEdge[u];
                        return true;
                    }
                }
            }
        }
    }
    return false;
}

struct CutResult {
    uint64_t value;
    std::vector<uint32_t> side;
};

void mergeSparse(int a,
                 int b,
                 std::vector<std::unordered_map<uint32_t, uint64_t>> &adj,
                 std::vector<std::vector<uint32_t>> &members,
                 std::vector<char> &active,
                 std::vector<int> &alive,
                 std::vector<int> &where) {
    if (a == b) return;
    int keep = a;
    int remove = b;
    if (adj[keep].size() < adj[remove].size()) {
        std::swap(keep, remove);
    }
    auto it = adj[keep].find(static_cast<uint32_t>(remove));
    if (it != adj[keep].end()) {
        adj[keep].erase(it);
    }
    for (const auto &entry : adj[remove]) {
        int nei = static_cast<int>(entry.first);
        uint64_t w = entry.second;
        if (!active[nei] || nei == keep) continue;
        uint64_t &forward = adj[keep][static_cast<uint32_t>(nei)];
        forward += w;
        uint64_t &backward = adj[nei][static_cast<uint32_t>(keep)];
        backward += w;
        auto itBack = adj[nei].find(static_cast<uint32_t>(remove));
        if (itBack != adj[nei].end()) {
            adj[nei].erase(itBack);
        }
    }
    adj[remove].clear();
    active[remove] = false;
    if (members[keep].size() < members[remove].size()) {
        members[keep].swap(members[remove]);
    }
    members[keep].insert(members[keep].end(), members[remove].begin(), members[remove].end());
    members[remove].clear();
    int idx = where[remove];
    int last = alive.back();
    alive[idx] = last;
    where[last] = idx;
    alive.pop_back();
    where[remove] = -1;
}

CutResult stoerWagnerSparse(std::vector<std::unordered_map<uint32_t, uint64_t>> &adj) {
    const size_t n = adj.size();
    CutResult result{std::numeric_limits<uint64_t>::max(), {}};
    if (n == 0) {
        result.value = 0;
        result.side.clear();
        return result;
    }
    std::vector<std::vector<uint32_t>> members(n);
    for (uint32_t i = 0; i < n; ++i) {
        members[i].push_back(i);
    }
    std::vector<char> active(n, true);
    std::vector<int> alive;
    alive.reserve(n);
    std::vector<int> where(n, -1);
    for (uint32_t i = 0; i < n; ++i) {
        alive.push_back(static_cast<int>(i));
        where[i] = static_cast<int>(i);
    }
    std::vector<int> addedVersion(n, -1);
    std::vector<int> weightVersion(n, -1);
    std::vector<uint64_t> weights(n, 0);

    int phaseId = 0;
    while (alive.size() > 1) {
        ++phaseId;
        size_t phaseSize = alive.size();
        std::priority_queue<std::pair<uint64_t, int>> pq;
        int start = alive[0];
        weightVersion[start] = phaseId;
        weights[start] = 0;
        pq.emplace(0, start);
        int prev = -1;
        for (size_t iter = 0; iter < phaseSize; ++iter) {
            int sel = -1;
            uint64_t selWeight = 0;
            while (true) {
                if (pq.empty()) {
                    int candidate = -1;
                    for (int v : alive) {
                        if (!active[v]) continue;
                        if (addedVersion[v] == phaseId) continue;
                        candidate = v;
                        break;
                    }
                    if (candidate == -1) break;
                    uint64_t val = (weightVersion[candidate] == phaseId) ? weights[candidate] : 0;
                    pq.emplace(val, candidate);
                }
                auto top = pq.top();
                pq.pop();
                int vtx = top.second;
                if (!active[vtx]) continue;
                if (addedVersion[vtx] == phaseId) continue;
                uint64_t actual = (weightVersion[vtx] == phaseId) ? weights[vtx] : 0;
                if (top.first != actual) continue;
                sel = vtx;
                selWeight = actual;
                break;
            }
            if (sel == -1) break;
            if (iter == phaseSize - 1) {
                if (selWeight < result.value) {
                    result.value = selWeight;
                    result.side = members[sel];
                }
                if (prev == -1) {
                    active[sel] = false;
                    alive.pop_back();
                    where[sel] = -1;
                    break;
                }
                mergeSparse(prev, sel, adj, members, active, alive, where);
                break;
            }
            addedVersion[sel] = phaseId;
            prev = sel;
            for (const auto &entry : adj[sel]) {
                int nei = static_cast<int>(entry.first);
                uint64_t w = entry.second;
                if (!active[nei]) continue;
                if (addedVersion[nei] == phaseId) continue;
                if (weightVersion[nei] != phaseId) {
                    weightVersion[nei] = phaseId;
                    weights[nei] = 0;
                }
                weights[nei] += w;
                pq.emplace(weights[nei], nei);
            }
        }
    }
    if (result.side.empty()) {
        result.side = members[alive.front()];
        result.value = 0;
    }
    return result;
}

CutResult stoerWagnerDense(uint32_t n, std::vector<uint64_t> &mat) {
    CutResult result{std::numeric_limits<uint64_t>::max(), {}};
    if (n == 0) {
        result.value = 0;
        return result;
    }
    std::vector<int> order(n);
    for (uint32_t i = 0; i < n; ++i) order[i] = static_cast<int>(i);
    std::vector<std::vector<uint32_t>> members(n);
    for (uint32_t i = 0; i < n; ++i) members[i].push_back(i);
    std::vector<uint64_t> w(n, 0);
    std::vector<char> added(n, false);
    int current = static_cast<int>(n);
    size_t nSize = static_cast<size_t>(n);
    while (current > 1) {
        std::fill(added.begin(), added.end(), false);
        std::fill(w.begin(), w.end(), 0);
        int prev = -1;
        for (int i = 0; i < current; ++i) {
            int sel = -1;
            for (int j = 0; j < current; ++j) {
                int node = order[j];
                if (added[node]) continue;
                if (sel == -1 || w[node] > w[sel]) sel = node;
            }
            if (sel == -1) break;
            if (i == current - 1) {
                if (w[sel] < result.value) {
                    result.value = w[sel];
                    result.side = members[sel];
                }
                if (prev == -1) {
                    return result;
                }
                for (int j = 0; j < current; ++j) {
                    int node = order[j];
                    if (node == prev) continue;
                    size_t idxPrev = static_cast<size_t>(prev) * nSize + static_cast<size_t>(node);
                    size_t idxSel = static_cast<size_t>(sel) * nSize + static_cast<size_t>(node);
                    mat[idxPrev] += mat[idxSel];
                    mat[static_cast<size_t>(node) * nSize + static_cast<size_t>(prev)] = mat[idxPrev];
                }
                if (members[prev].size() < members[sel].size()) members[prev].swap(members[sel]);
                members[prev].insert(members[prev].end(), members[sel].begin(), members[sel].end());
                members[sel].clear();
                for (int j = 0; j < current; ++j) {
                    if (order[j] == sel) {
                        order[j] = order[current - 1];
                        break;
                    }
                }
                --current;
                break;
            }
            added[sel] = true;
            prev = sel;
            for (int j = 0; j < current; ++j) {
                int node = order[j];
                if (added[node]) continue;
                w[node] += mat[static_cast<size_t>(sel) * nSize + static_cast<size_t>(node)];
            }
        }
    }
    if (result.side.empty()) {
        result.side = members[order[0]];
        if (result.value == std::numeric_limits<uint64_t>::max()) result.value = 0;
    }
    return result;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastInput in;
    uint32_t N = 0;
    uint64_t M = 0;
    if (!in.readUnsigned(N) || !in.readUnsigned(M)) {
        return 0;
    }
    if (N <= 1) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Edge> edges;
    try {
        if (M <= static_cast<uint64_t>(std::numeric_limits<size_t>::max())) {
            edges.reserve(static_cast<size_t>(M));
        }
    } catch (...) {
        // ignore reserve failure
    }
    std::vector<uint64_t> degree(N, 0);
    for (uint64_t i = 0; i < M; ++i) {
        uint32_t u = 0, v = 0;
        if (!in.readUnsigned(u) || !in.readUnsigned(v)) {
            return 0;
        }
        if (u >= N || v >= N) continue;
        if (u == v) continue;
        edges.push_back({u, v});
        ++degree[u];
        ++degree[v];
    }

    if (edges.empty()) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<uint64_t> offset(N + 1, 0);
    for (uint32_t i = 0; i < N; ++i) {
        offset[i + 1] = offset[i] + degree[i];
    }
    uint64_t totalAdj = offset.back();

    uint64_t bridgeIdx = std::numeric_limits<uint64_t>::max();
    bool hasBridge = false;
    if (edges.size() <= static_cast<size_t>(std::numeric_limits<uint32_t>::max())) {
        std::vector<uint32_t> incident(static_cast<size_t>(totalAdj));
        std::vector<uint64_t> cursor = offset;
        for (size_t idx = 0; idx < edges.size(); ++idx) {
            const Edge &e = edges[idx];
            incident[static_cast<size_t>(cursor[e.u]++)] = static_cast<uint32_t>(idx);
            incident[static_cast<size_t>(cursor[e.v]++)] = static_cast<uint32_t>(idx);
        }
        hasBridge = findFirstBridge(edges, offset, incident, N, bridgeIdx);
    } else {
        std::vector<uint64_t> incident(static_cast<size_t>(totalAdj));
        std::vector<uint64_t> cursor = offset;
        for (size_t idx = 0; idx < edges.size(); ++idx) {
            const Edge &e = edges[idx];
            incident[static_cast<size_t>(cursor[e.u]++)] = static_cast<uint64_t>(idx);
            incident[static_cast<size_t>(cursor[e.v]++)] = static_cast<uint64_t>(idx);
        }
        hasBridge = findFirstBridge(edges, offset, incident, N, bridgeIdx);
    }

    if (hasBridge) {
        const Edge &e = edges[static_cast<size_t>(bridgeIdx)];
        std::cout << 1 << '\n';
        std::cout << e.u << ' ' << e.v << '\n';
        return 0;
    }

    std::vector<uint64_t>().swap(offset);

    constexpr uint32_t DENSE_NODE_LIMIT = 4000;
    bool useDense = (N <= DENSE_NODE_LIMIT);
    CutResult result;

    if (useDense) {
        std::vector<uint64_t>().swap(degree);
        size_t nSize = static_cast<size_t>(N);
        size_t matrixSize = nSize * nSize;
        std::vector<uint64_t> mat(matrixSize, 0);
        for (const auto &e : edges) {
            size_t idx1 = static_cast<size_t>(e.u) * nSize + static_cast<size_t>(e.v);
            size_t idx2 = static_cast<size_t>(e.v) * nSize + static_cast<size_t>(e.u);
            ++mat[idx1];
            ++mat[idx2];
        }
        result = stoerWagnerDense(N, mat);
    } else {
        std::vector<std::unordered_map<uint32_t, uint64_t>> adj(N);
        for (uint32_t i = 0; i < N; ++i) {
            adj[i].max_load_factor(0.7f);
            if (degree[i] > 0) {
                uint64_t uniqueHint = std::min<uint64_t>(degree[i], static_cast<uint64_t>(N ? N : 1));
                size_t bucketHint = static_cast<size_t>(std::max<uint64_t>(uniqueHint, 1));
                adj[i].reserve(bucketHint);
            }
        }
        for (const auto &e : edges) {
            ++adj[e.u][e.v];
            ++adj[e.v][e.u];
        }
        std::vector<uint64_t>().swap(degree);
        result = stoerWagnerSparse(adj);
    }

    if (result.side.empty()) {
        result.side.push_back(0);
    }

    std::vector<uint8_t> sideFlag(N, 0);
    for (uint32_t node : result.side) {
        if (node < N) sideFlag[node] = 1;
    }

    uint64_t cutCount = 0;
    for (const auto &e : edges) {
        if (sideFlag[e.u] != sideFlag[e.v]) {
            ++cutCount;
        }
    }
    if (result.value == std::numeric_limits<uint64_t>::max() || result.value != cutCount) {
        result.value = cutCount;
    }

    std::cout << result.value << '\n';
    for (const auto &e : edges) {
        if (sideFlag[e.u] != sideFlag[e.v]) {
            std::cout << e.u << ' ' << e.v << '\n';
        }
    }
    return 0;
}
```

**Key design choices**

1. **Memory-aware ingestion** – edges are stored as compact 64-bit pairs, self-loops are discarded, and degrees are counted in one pass. CSR-style offsets plus either 32-bit or 64-bit incident arrays are built (depending on graph size) to enable fast Tarjan bridge detection without quadratic memory.

2. **Adaptive solver pipeline**
   - **Phase 1:** Tarjan DFS quickly returns when a bridge (cut of size 1) exists.
   - **Phase 2:** If `N ≤ 4000`, a dense Stoer–Wagner implementation backed by a flat matrix exploits spatial locality on smaller graphs.
   - **Phase 3:** Otherwise, a sparse Stoer–Wagner variant with hash-based adjacency, phase-local heaps, and small-to-large membership merging handles large/sparse or massive instances.

3. **Scalability safeguards** – load-factor tuning, size hints, iterative DFS (no recursion), and avoiding repeated global resets keep the solver practical from toy graphs to multi-gigabyte inputs.

4. **Cut reconstruction** – whenever a better cut is found, the exact vertex subset is persisted. At the end, edges crossing that subset are streamed once to emit both the cut size and the explicit edge list, guaranteeing a valid disconnecting set.
