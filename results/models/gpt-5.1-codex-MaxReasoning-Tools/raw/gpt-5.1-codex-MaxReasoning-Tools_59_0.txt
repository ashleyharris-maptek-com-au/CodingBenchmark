```javascript
'use strict';

const fs = require('fs');

function main() {
    let input = '';
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (err) {
        input = '';
    }

    let data;
    try {
        data = input ? JSON.parse(input) : {};
    } catch (err) {
        data = {};
    }

    const grid = Array.isArray(data.probability) ? data.probability : [];
    const dronesInput = Array.isArray(data.drones) ? data.drones : [];
    let maxSteps = Number(data.maxStepsPerDrone);
    if (!Number.isFinite(maxSteps)) maxSteps = 0;
    maxSteps = Math.floor(maxSteps);
    if (maxSteps < 0) maxSteps = 0;
    const MAX_INT_STEP = 2147483646;
    if (maxSteps > MAX_INT_STEP) maxSteps = MAX_INT_STEP;

    const rows = grid.length;
    let cols = 0;
    for (let y = 0; y < rows; y++) {
        const row = Array.isArray(grid[y]) ? grid[y] : [];
        if (row.length > cols) cols = row.length;
    }

    if (rows === 0 || cols === 0) {
        const fallbackPaths = dronesInput.map(drone => {
            const x = clampCoord(toNumber(drone && drone.x, 0), Math.max(cols, 1));
            const y = clampCoord(toNumber(drone && drone.y, 0), Math.max(rows, 1));
            return [[x, y]];
        });
        process.stdout.write(JSON.stringify({paths: fallbackPaths}));
        return;
    }

    const totalCells = rows * cols;
    const probFlat = new Float64Array(totalCells);
    const isObstacle = new Uint8Array(totalCells);
    let remainingPositive = 0;
    let firstFreeIdx = -1;

    for (let y = 0; y < rows; y++) {
        const row = Array.isArray(grid[y]) ? grid[y] : [];
        for (let x = 0; x < cols; x++) {
            const idx = y * cols + x;
            let cellValue = -1;
            if (x < row.length) {
                const numeric = Number(row[x]);
                if (Number.isFinite(numeric)) {
                    cellValue = numeric;
                }
            }
            if (cellValue < 0) {
                isObstacle[idx] = 1;
                probFlat[idx] = 0;
            } else {
                isObstacle[idx] = 0;
                probFlat[idx] = cellValue;
                if (cellValue > 0) remainingPositive++;
                if (firstFreeIdx === -1) firstFreeIdx = idx;
            }
        }
    }

    if (firstFreeIdx === -1) {
        const fallbackPaths = dronesInput.map(drone => {
            const x = clampCoord(toNumber(drone && drone.x, 0), cols);
            const y = clampCoord(toNumber(drone && drone.y, 0), rows);
            return [[x, y]];
        });
        process.stdout.write(JSON.stringify({paths: fallbackPaths}));
        return;
    }

    const dirX = [1, -1, 0, 0];
    const dirY = [0, 0, 1, -1];

    const sanitizedStarts = dronesInput.map(drone => {
        const rawX = toNumber(drone && drone.x, 0);
        const rawY = toNumber(drone && drone.y, 0);
        let x = clampCoord(rawX, cols);
        let y = clampCoord(rawY, rows);
        let idx = y * cols + x;
        if (isObstacle[idx]) {
            idx = findNearestPassableCell(x, y);
            if (idx === -1) {
                return null;
            }
            y = Math.floor(idx / cols);
            x = idx - y * cols;
        }
        return {x, y, idx};
    });

    const voronoiOwner = buildVoronoi(sanitizedStarts);

    const covered = new Uint8Array(totalCells);
    const bfsQueue = new Int32Array(totalCells);
    const workVisited = new Int32Array(totalCells);
    const workParent = new Int32Array(totalCells);
    const workDist = new Int32Array(totalCells);
    let visitToken = 1;
    const EPS = 1e-9;

    const resultPaths = [];

    for (let i = 0; i < dronesInput.length; i++) {
        const startInfo = sanitizedStarts[i];
        if (!startInfo || startInfo.idx < 0) {
            const fallbackX = clampCoord(toNumber(dronesInput[i] && dronesInput[i].x, 0), cols);
            const fallbackY = clampCoord(toNumber(dronesInput[i] && dronesInput[i].y, 0), rows);
            resultPaths.push([[fallbackX, fallbackY]]);
            continue;
        }

        const path = [[startInfo.x, startInfo.y]];
        resultPaths.push(path);

        if (probFlat[startInfo.idx] > 0 && covered[startInfo.idx] === 0) {
            covered[startInfo.idx] = 1;
            remainingPositive--;
        }

        if (maxSteps === 0 || remainingPositive <= 0) {
            continue;
        }

        let stepsLeft = maxSteps;
        let currentIdx = startInfo.idx;

        while (stepsLeft > 0 && remainingPositive > 0) {
            const plan = findBestPath(currentIdx, stepsLeft, i);
            if (!plan) break;
            const {indices, distance} = plan;
            if (distance <= 0) break;

            for (let p = 1; p < indices.length; p++) {
                const idx = indices[p];
                const y = Math.floor(idx / cols);
                const x = idx - y * cols;
                path.push([x, y]);
            }

            for (let p = 0; p < indices.length; p++) {
                const idx = indices[p];
                if (probFlat[idx] > 0 && covered[idx] === 0) {
                    covered[idx] = 1;
                    remainingPositive--;
                }
            }

            stepsLeft -= distance;
            currentIdx = indices[indices.length - 1];
        }
    }

    process.stdout.write(JSON.stringify({paths: resultPaths}));

    function findBestPath(currentIdx, stepsAllowed, droneIdx) {
        if (stepsAllowed <= 0 || remainingPositive <= 0) {
            return null;
        }
        const token = nextVisitToken();
        let head = 0;
        let tail = 0;

        bfsQueue[tail++] = currentIdx;
        workVisited[currentIdx] = token;
        workParent[currentIdx] = -1;
        workDist[currentIdx] = 0;

        let bestIdx = -1;
        let bestScore = -Infinity;
        let bestProb = 0;
        let bestDistValue = Infinity;

        while (head < tail) {
            const idx = bfsQueue[head++];
            const distHere = workDist[idx];

            if (probFlat[idx] > 0 && covered[idx] === 0) {
                const assignedBonus = (voronoiOwner[idx] === droneIdx && voronoiOwner[idx] !== -1) ? 1.5 : 1.0;
                const score = (probFlat[idx] * assignedBonus) / (1 + distHere);
                if (score > bestScore + EPS ||
                    (Math.abs(score - bestScore) <= EPS && (
                        probFlat[idx] > bestProb + EPS ||
                        (Math.abs(probFlat[idx] - bestProb) <= EPS && (
                            distHere < bestDistValue ||
                            (distHere === bestDistValue && (bestIdx === -1 || idx < bestIdx))
                        ))
                    ))
                ) {
                    bestScore = score;
                    bestIdx = idx;
                    bestProb = probFlat[idx];
                    bestDistValue = distHere;
                }
            }

            if (distHere >= stepsAllowed) continue;

            const y = Math.floor(idx / cols);
            const x = idx - y * cols;

            for (let d = 0; d < 4; d++) {
                const nx = x + dirX[d];
                const ny = y + dirY[d];
                if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
                const nIdx = ny * cols + nx;
                if (isObstacle[nIdx]) continue;
                if (workVisited[nIdx] === token) continue;
                workVisited[nIdx] = token;
                workParent[nIdx] = idx;
                workDist[nIdx] = distHere + 1;
                bfsQueue[tail++] = nIdx;
            }
        }

        if (bestIdx === -1) {
            return null;
        }

        const indices = [];
        let cursor = bestIdx;
        while (cursor !== -1) {
            indices.push(cursor);
            cursor = workParent[cursor];
        }
        indices.reverse();
        const distance = indices.length - 1;
        return {indices, distance};
    }

    function nextVisitToken() {
        visitToken++;
        if (visitToken === 0x7fffffff) {
            workVisited.fill(0);
            visitToken = 1;
        }
        return visitToken;
    }

    function buildVoronoi(starts) {
        const owner = new Int32Array(totalCells);
        owner.fill(-1);
        const dist = new Int32Array(totalCells);
        dist.fill(-1);
        const queue = new Int32Array(totalCells);
        let head = 0;
        let tail = 0;

        for (let i = 0; i < starts.length; i++) {
            const info = starts[i];
            if (!info || info.idx < 0) continue;
            const idx = info.idx;
            if (isObstacle[idx]) continue;
            if (dist[idx] === -1) {
                dist[idx] = 0;
                owner[idx] = i;
                queue[tail++] = idx;
            } else if (i < owner[idx]) {
                owner[idx] = i;
            }
        }

        while (head < tail) {
            const idx = queue[head++];
            const baseOwner = owner[idx];
            const baseDist = dist[idx];
            const y = Math.floor(idx / cols);
            const x = idx - y * cols;

            for (let d = 0; d < 4; d++) {
                const nx = x + dirX[d];
                const ny = y + dirY[d];
                if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
                const nIdx = ny * cols + nx;
                if (isObstacle[nIdx]) continue;
                if (dist[nIdx] === -1) {
                    dist[nIdx] = baseDist + 1;
                    owner[nIdx] = baseOwner;
                    queue[tail++] = nIdx;
                } else if (dist[nIdx] === baseDist + 1 && baseOwner !== -1) {
                    if (owner[nIdx] === -1 || baseOwner < owner[nIdx]) {
                        owner[nIdx] = baseOwner;
                    }
                }
            }
        }

        return owner;
    }

    function findNearestPassableCell(targetX, targetY) {
        let bestIdx = -1;
        let bestDist = Infinity;
        for (let idx = 0; idx < totalCells; idx++) {
            if (isObstacle[idx]) continue;
            const y = Math.floor(idx / cols);
            const x = idx - y * cols;
            const dist = Math.abs(x - targetX) + Math.abs(y - targetY);
            if (dist < bestDist) {
                bestDist = dist;
                bestIdx = idx;
                if (dist === 0) break;
            }
        }
        return bestIdx;
    }

    function clampCoord(value, limit) {
        if (limit <= 0) return 0;
        const num = Number(value);
        let val = Number.isFinite(num) ? Math.floor(num) : 0;
        if (val < 0) val = 0;
        if (val >= limit) val = limit - 1;
        return val;
    }

    function toNumber(value, fallback) {
        const num = Number(value);
        return Number.isFinite(num) ? num : fallback;
    }
}

main();
```
