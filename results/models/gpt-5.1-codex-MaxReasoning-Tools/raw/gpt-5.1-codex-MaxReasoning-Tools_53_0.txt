```python
import math

# --- Constants & configuration ------------------------------------------------
MU_EARTH = 3.986004418e14        # m^3/s^2
R_EARTH = 6_371_000.0            # m
DEFAULT_ALT = 400_000.0          # m
DEFAULT_A = R_EARTH + DEFAULT_ALT
DEFAULT_N = math.sqrt(MU_EARTH / (DEFAULT_A ** 3))
ACCEL_PER_THRUST = 0.10          # m/s^2 commanded per unit thrust (tunable)

_PHASE_CONFIG = {
    "far":   dict(kp_x=0.008, kd_x=0.08, kp_y=0.0025, kd_y=0.05,
                  kp_z=0.015, kd_z=0.08, vy_gain=0.010,
                  speed_limit=0.6, accel_limit=0.060),
    "mid":   dict(kp_x=0.015, kd_x=0.11, kp_y=0.006, kd_y=0.06,
                  kp_z=0.025, kd_z=0.11, vy_gain=0.020,
                  speed_limit=0.35, accel_limit=0.040),
    "close": dict(kp_x=0.032, kd_x=0.16, kp_y=0.015, kd_y=0.08,
                  kp_z=0.040, kd_z=0.16, vy_gain=0.045,
                  speed_limit=0.15, accel_limit=0.022),
    "final": dict(kp_x=0.055, kd_x=0.22, kp_y=0.030, kd_y=0.10,
                  kp_z=0.055, kd_z=0.22, vy_gain=0.070,
                  speed_limit=0.08, accel_limit=0.012),
}

_REL_SENSORS = (
    dict(
        pos=("lidar_rel_pos_x_m", "lidar_rel_pos_y_m", "lidar_rel_pos_z_m"),
        vel=("lidar_rel_vel_x_ms", "lidar_rel_vel_y_ms", "lidar_rel_vel_z_ms"),
        pos_valid=("lidar_rel_x_valid", "lidar_rel_y_valid", "lidar_rel_z_valid"),
        vel_valid=("lidar_relv_x_valid", "lidar_relv_y_valid", "lidar_relv_z_valid"),
    ),
    dict(
        pos=("radar_rel_pos_x_m", "radar_rel_pos_y_m", "radar_rel_pos_z_m"),
        vel=("radar_rel_vel_x_ms", "radar_rel_vel_y_ms", "radar_rel_vel_z_ms"),
        pos_valid=("radar_rel_x_valid", "radar_rel_y_valid", "radar_rel_z_valid"),
        vel_valid=("radar_relv_x_valid", "radar_relv_y_valid", "radar_relv_z_valid"),
    ),
)

_AUTOPILOT_STATE = {
    "nav_pos": [0.0, 0.0, 0.0],
    "nav_vel": [0.0, 0.0, 0.0],
    "pos_age": [1e9, 1e9, 1e9],
    "vel_age": [1e9, 1e9, 1e9],
    "n": DEFAULT_N,
    "range_filt": None,
    "phase": "far",
}


# --- Utility helpers ----------------------------------------------------------
def _clamp(value, lo, hi):
    return max(lo, min(hi, value))


def _lowpass(prev, new, alpha):
    if prev is None:
        return new
    return prev + alpha * (new - prev)


def _bool(state, key):
    return bool(state.get(key, False))


def _median(values):
    ordered = sorted(values)
    n = len(ordered)
    mid = n // 2
    if n % 2:
        return ordered[mid]
    return 0.5 * (ordered[mid - 1] + ordered[mid])


# --- Navigation & sensing -----------------------------------------------------
def _update_mean_motion(state):
    a = None
    if _bool(state, "ground_semi_major_axis_m_valid"):
        a = state.get("ground_semi_major_axis_m")
    elif _bool(state, "ground_altitude_km_valid"):
        a = R_EARTH + 1000.0 * state.get("ground_altitude_km", 0.0)
    if a and a > 0:
        n_meas = math.sqrt(MU_EARTH / (a ** 3))
        _AUTOPILOT_STATE["n"] = _lowpass(_AUTOPILOT_STATE["n"], n_meas, 0.1)
    return _AUTOPILOT_STATE["n"]


def _read_rel_measurements(state):
    pos_meas = [None, None, None]
    vel_meas = [None, None, None]
    for axis in range(3):
        for sensor in _REL_SENSORS:
            if _bool(state, sensor["pos_valid"][axis]):
                pos_meas[axis] = state.get(sensor["pos"][axis])
                break
        for sensor in _REL_SENSORS:
            if _bool(state, sensor["vel_valid"][axis]):
                vel_meas[axis] = state.get(sensor["vel"][axis])
                break
    return pos_meas, vel_meas


def _fuse_accel(state):
    axes = {"x": [], "y": [], "z": []}
    for idx in (1, 2, 3):
        for axis in ("x", "y", "z"):
            valid = _bool(state, f"accel_{idx}_{axis}_valid")
            if valid:
                axes[axis].append(state.get(f"accel_{idx}_{axis}_ms2", 0.0))
    avg = []
    for axis in ("x", "y", "z"):
        avg.append(sum(axes[axis]) / len(axes[axis]) if axes[axis] else 0.0)
    return avg


def _update_navigation(pos_meas, vel_meas, accel, dt):
    pos = _AUTOPILOT_STATE["nav_pos"]
    vel = _AUTOPILOT_STATE["nav_vel"]
    pos_age = _AUTOPILOT_STATE["pos_age"]
    vel_age = _AUTOPILOT_STATE["vel_age"]

    for i in range(3):
        if pos_meas[i] is not None:
            pos[i] = _lowpass(pos[i], pos_meas[i], 0.35)
            pos_age[i] = 0.0
        else:
            pos[i] += vel[i] * dt
            pos_age[i] += dt

        if vel_meas[i] is not None:
            vel[i] = _lowpass(vel[i], vel_meas[i], 0.30)
            vel_age[i] = 0.0
        else:
            vel[i] += accel[i] * dt
            vel_age[i] += dt
    return pos[:], vel[:]


def _estimate_range(state, pos):
    candidates = []
    norm = math.sqrt(sum(p * p for p in pos))
    candidates.append(norm)

    for idx in (1, 2, 3):
        if _bool(state, f"range_{idx}_valid"):
            rng = state.get(f"range_{idx}_m")
            if rng and rng > 0:
                candidates.append(rng)

    if state.get("dock_camera_active"):
        cam_rng = state.get("dock_camera_range_m")
        if cam_rng and cam_rng > 0:
            candidates.append(cam_rng)

    tgt_rng = state.get("target_dock_range_m")
    if tgt_rng and tgt_rng > 0:
        candidates.append(tgt_rng)

    if candidates:
        rng_est = _median(candidates)
        _AUTOPILOT_STATE["range_filt"] = _lowpass(
            _AUTOPILOT_STATE["range_filt"], rng_est, 0.25
        )
    return _AUTOPILOT_STATE["range_filt"] or norm


def _select_phase(rng):
    if rng > 200.0:
        return "far"
    if rng > 50.0:
        return "mid"
    if rng > 10.0:
        return "close"
    return "final"


# --- Guidance logic -----------------------------------------------------------
def _build_targets(pos, rng, phase):
    y = pos[1]

    if rng > 600:
        y_target = -0.7 * rng
    elif rng > 300:
        y_target = -0.55 * rng
    elif rng > 120:
        y_target = -40.0
    elif rng > 60:
        y_target = -8.0
    elif rng > 35:
        y_target = -3.0
    else:
        y_target = 0.0

    # Force trailing geometry if currently ahead
    if y > 5.0 and rng > 40.0:
        y_target = min(y_target, -5.0)

    y_error = y - y_target
    if rng > 80.0:
        bias = min(35.0, max(5.0, 0.05 * abs(y_error)))
        if y_error < -15.0:
            x_target = -bias  # drop orbit (retrograde catch-up)
        elif y_error > 15.0:
            x_target = bias   # raise orbit to slow down
        else:
            x_target = 0.0
    else:
        x_target = 0.0

    z_target = 0.0

    config = _PHASE_CONFIG[phase]
    v_limit = min(config["speed_limit"], 0.012 * max(rng, 1.0))
    if phase in ("close", "final"):
        v_limit = min(v_limit, 0.12 if phase == "close" else 0.06)
    v_limit = max(0.04, v_limit)

    vy_target = _clamp(-config["vy_gain"] * y_error, -v_limit, v_limit)

    return dict(
        x_target=x_target,
        y_target=y_target,
        z_target=z_target,
        vy_target=vy_target,
    )


def _cw_guidance(pos, vel, targets, n, phase):
    cfg = _PHASE_CONFIG[phase]

    x_err = pos[0] - targets["x_target"]
    vx_err = vel[0]
    x_ddot_des = -cfg["kp_x"] * x_err - cfg["kd_x"] * vx_err
    ax = x_ddot_des + 3.0 * (n ** 2) * pos[0] + 2.0 * n * vel[1]

    y_err = pos[1] - targets["y_target"]
    vy_err = vel[1] - targets["vy_target"]
    y_ddot_des = -cfg["kp_y"] * y_err - cfg["kd_y"] * vy_err
    ay = y_ddot_des + 2.0 * n * vel[0]

    z_err = pos[2] - targets["z_target"]
    vz_err = vel[2]
    z_ddot_des = -cfg["kp_z"] * z_err - cfg["kd_z"] * vz_err
    az = z_ddot_des - (n ** 2) * pos[2]

    return [ax, ay, az]


def _apply_los_brake(pos, vel, rng, accel_cmd, phase):
    if rng <= 0.5:
        return accel_cmd
    los = [p / rng for p in pos]
    closing_speed = sum(los[i] * vel[i] for i in range(3))
    limit = min(_PHASE_CONFIG[phase]["speed_limit"], 0.02 * max(rng, 1.0))
    if phase == "close":
        limit = min(limit, 0.10)
    if phase == "final":
        limit = min(limit, 0.05)
    if closing_speed < -limit:
        excess = (-limit - closing_speed)
        brake_gain = 0.5 if phase in ("close", "final") else 0.3
        correction = brake_gain * excess
        accel_cmd = [accel_cmd[i] + los[i] * correction for i in range(3)]
    return accel_cmd


def _map_to_thrust(accel_cmd, phase, state, rng):
    accel_limit = _PHASE_CONFIG[phase]["accel_limit"]
    fuel_pct = state.get("fuel_remaining_pct")
    fuel_kg = state.get("fuel_remaining_kg")

    fuel_scale = 1.0
    if fuel_pct is not None:
        if fuel_pct < 20.0:
            fuel_scale = 0.7
        if fuel_pct < 10.0:
            fuel_scale = 0.5
    if fuel_kg is not None and fuel_kg < 120.0:
        fuel_scale = min(fuel_scale, 0.6)

    accel_limit *= fuel_scale
    if rng < 50.0:
        accel_limit = min(accel_limit, 0.018)
    if rng < 10.0:
        accel_limit = min(accel_limit, 0.010)
    if rng < 5.0:
        accel_limit = min(accel_limit, 0.008)

    thrust = {}
    for axis, cmd in zip(("x", "y", "z"), accel_cmd):
        limited = _clamp(cmd, -accel_limit, accel_limit)
        thrust[axis] = _clamp(limited / ACCEL_PER_THRUST, -1.0, 1.0)
    return thrust


# --- Attitude hold ------------------------------------------------------------
def _attitude_hold(state):
    desired = dict(roll=0.0, pitch=0.0, yaw=0.0)
    att = {}
    for axis in ("roll", "pitch", "yaw"):
        value = None
        for tracker in (1, 2):
            if _bool(state, f"star_tracker_{tracker}_{axis}_valid"):
                deg_val = state.get(f"star_tracker_{tracker}_{axis}_deg", 0.0)
                value = math.radians(deg_val)
                break
        att[axis] = value

    rates = {}
    for axis in ("roll", "pitch", "yaw"):
        rate = None
        for gyro in (1, 2, 3):
            if _bool(state, f"gyro_{gyro}_{axis}_valid"):
                deg_per_s = state.get(f"gyro_{gyro}_{axis}_dps", 0.0)
                rate = math.radians(deg_per_s)
                break
        rates[axis] = rate

    cmd = {}
    kp = 0.6
    kd = 0.5
    for axis in ("roll", "pitch", "yaw"):
        err = (att[axis] - desired[axis]) if att[axis] is not None else 0.0
        rate = rates[axis] if rates[axis] is not None else 0.0
        cmd[axis] = _clamp(-kp * err - kd * rate, -0.035, 0.035)
    return cmd


# --- Main autopilot -----------------------------------------------------------
def autopilot_step(state: dict, dt: float) -> dict:
    """
    Rendezvous & docking translational autopilot using CW guidance with
    sensor-fused relative navigation. Handles far/mid/close/final regimes,
    enforces approach corridors, and limits thrust for fuel preservation.
    """
    dt = max(float(dt), 1e-3)

    n = _update_mean_motion(state)
    pos_meas, vel_meas = _read_rel_measurements(state)
    accel = _fuse_accel(state)
    pos, vel = _update_navigation(pos_meas, vel_meas, accel, dt)
    rng = _estimate_range(state, pos)
    phase = _select_phase(rng)
    _AUTOPILOT_STATE["phase"] = phase

    targets = _build_targets(pos, rng, phase)
    accel_cmd = _cw_guidance(pos, vel, targets, n, phase)
    accel_cmd = _apply_los_brake(pos, vel, rng, accel_cmd, phase)
    thrust = _map_to_thrust(accel_cmd, phase, state, rng)
    attitude = _attitude_hold(state)

    return {
        "thrust_x": float(thrust["x"]),
        "thrust_y": float(thrust["y"]),
        "thrust_z": float(thrust["z"]),
        "att_rate_roll": float(attitude["roll"]),
        "att_rate_pitch": float(attitude["pitch"]),
        "att_rate_yaw": float(attitude["yaw"]),
    }
```
