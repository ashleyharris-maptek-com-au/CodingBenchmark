```python
import sys
import math


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    idx = 0
    N = int(data[idx])
    idx += 1
    K = int(data[idx])
    idx += 1
    capacity = float(data[idx])
    idx += 1
    depot_x = float(data[idx])
    idx += 1
    depot_y = float(data[idx])
    idx += 1

    coords = []
    demands = []
    for _ in range(N):
        x = float(data[idx])
        y = float(data[idx + 1])
        demand = float(data[idx + 2])
        idx += 3
        coords.append((x, y))
        demands.append(demand)

    if N == 0:
        print("0.000000")
        print("0")
        return

    distance_matrix = [[0.0] * N for _ in range(N)]
    for i in range(N):
        xi, yi = coords[i]
        for j in range(i + 1, N):
            xj, yj = coords[j]
            d = math.hypot(xi - xj, yi - yj)
            distance_matrix[i][j] = distance_matrix[j][i] = d

    dist_depot = [
        math.hypot(coords[i][0] - depot_x, coords[i][1] - depot_y) for i in range(N)
    ]
    EPS = 1e-9
    vehicle_limit = K if K > 0 else 1

    def edge_length(a, b):
        if a == -1 and b == -1:
            return 0.0
        if a == -1:
            return dist_depot[b]
        if b == -1:
            return dist_depot[a]
        return distance_matrix[a][b]

    def route_distance(nodes):
        if not nodes:
            return 0.0
        total = dist_depot[nodes[0]]
        for u, v in zip(nodes, nodes[1:]):
            total += distance_matrix[u][v]
        total += dist_depot[nodes[-1]]
        return total

    def node_position_in_route(nodes, node):
        if not nodes:
            return None
        if nodes[0] == node:
            return 'start'
        if nodes[-1] == node:
            return 'end'
        return 'middle'

    def join_routes(nodes_a, nodes_b, pos_a, pos_b):
        options = []
        oriented_a = nodes_a[::-1] if pos_a == 'start' else nodes_a[:]
        oriented_b = nodes_b[::-1] if pos_b == 'end' else nodes_b[:]
        candidate1 = oriented_a + oriented_b
        options.append((route_distance(candidate1), candidate1))
        oriented_a2 = nodes_a[::-1] if pos_a == 'end' else nodes_a[:]
        oriented_b2 = nodes_b[::-1] if pos_b == 'start' else nodes_b[:]
        candidate2 = oriented_b2 + oriented_a2
        options.append((route_distance(candidate2), candidate2))
        options.sort(key=lambda x: x[0])
        return options[0][1], options[0][0]

    def best_insertion_position(route_nodes, node):
        best_pos = None
        best_delta = None
        length = len(route_nodes)
        for pos in range(length + 1):
            prev_idx = route_nodes[pos - 1] if pos > 0 else -1
            next_idx = route_nodes[pos] if pos < length else -1
            delta = (
                edge_length(prev_idx, node)
                + edge_length(node, next_idx)
                - edge_length(prev_idx, next_idx)
            )
            if best_delta is None or delta < best_delta:
                best_delta = delta
                best_pos = pos
        return best_pos, best_delta

    def two_opt_route(nodes):
        if len(nodes) < 4:
            return nodes[:]
        best = nodes[:]
        max_iter = min(8, len(best))
        iteration = 0
        while iteration < max_iter:
            improved = False
            m = len(best)
            for i in range(-1, m - 1):
                for j in range(i + 1, m):
                    if j - i < 2:
                        continue
                    a = -1 if i == -1 else best[i]
                    b = best[i + 1] if i + 1 < m else -1
                    c = best[j]
                    d = best[j + 1] if j + 1 < m else -1
                    delta = (
                        edge_length(a, c)
                        + edge_length(b, d)
                        - edge_length(a, b)
                        - edge_length(c, d)
                    )
                    if delta < -1e-7:
                        segment = best[i + 1 : j + 1]
                        segment.reverse()
                        best = best[: i + 1] + segment + best[j + 1 :]
                        improved = True
                        break
                if improved:
                    break
            if not improved:
                break
            iteration += 1
        return best

    def merge_best_pair(route_list):
        best = None
        for i in range(len(route_list)):
            for j in range(i + 1, len(route_list)):
                load_sum = route_list[i]['load'] + route_list[j]['load']
                if load_sum > capacity + EPS:
                    continue
                current_cost = route_list[i]['cost'] + route_list[j]['cost']
                nodes_i = route_list[i]['nodes']
                nodes_j = route_list[j]['nodes']
                for first, second in ((nodes_i, nodes_j), (nodes_j, nodes_i)):
                    for rev_first in (False, True):
                        oriented_first = first[::-1] if rev_first else first
                        for rev_second in (False, True):
                            oriented_second = second[::-1] if rev_second else second
                            candidate = oriented_first + oriented_second
                            new_cost = route_distance(candidate)
                            delta = new_cost - current_cost
                            if best is None or delta < best[0]:
                                best = (delta, i, j, candidate, new_cost, load_sum)
        if best is None:
            return False
        _, i, j, candidate_nodes, new_cost, load_sum = best
        route_list[i]['nodes'] = candidate_nodes
        route_list[i]['load'] = load_sum
        route_list[i]['cost'] = new_cost
        route_list.pop(j)
        return True

    def try_remove_route(route_list, remove_idx):
        route = route_list[remove_idx]
        nodes = route['nodes']
        if not nodes:
            return False
        orders = [
            tuple(sorted(nodes, key=lambda n: demands[n], reverse=True)),
            tuple(nodes),
            tuple(reversed(nodes)),
        ]
        unique_orders = []
        seen = set()
        for order in orders:
            if order not in seen:
                unique_orders.append(list(order))
                seen.add(order)
        for order in unique_orders:
            others = []
            for idx_r, other in enumerate(route_list):
                if idx_r == remove_idx:
                    continue
                others.append(
                    {'ref': other, 'nodes': other['nodes'][:], 'load': other['load']}
                )
            success = True
            for node in order:
                best_choice = None
                for data in others:
                    if data['load'] + demands[node] > capacity + EPS:
                        continue
                    pos, delta = best_insertion_position(data['nodes'], node)
                    if pos is None:
                        continue
                    if best_choice is None or delta < best_choice[0]:
                        best_choice = (delta, data, pos)
                if best_choice is None:
                    success = False
                    break
                _, target, pos = best_choice
                target['nodes'].insert(pos, node)
                target['load'] += demands[node]
            if success:
                route_list.pop(remove_idx)
                for data in others:
                    ref_route = data['ref']
                    ref_route['nodes'] = data['nodes']
                    ref_route['load'] = data['load']
                    ref_route['cost'] = route_distance(ref_route['nodes'])
                return True
        return False

    def relocate_remove_route(route_list):
        order = sorted(
            range(len(route_list)),
            key=lambda idx: (route_list[idx]['load'], len(route_list[idx]['nodes'])),
        )
        for idx in order:
            if try_remove_route(route_list, idx):
                return True
        return False

    def enforce_vehicle_limit(route_list):
        while len(route_list) > vehicle_limit:
            if merge_best_pair(route_list):
                continue
            if relocate_remove_route(route_list):
                continue
            break

    def build_fallback_routes():
        remaining = set(range(N))
        fallback = []
        while remaining:
            route_nodes = []
            route_load = 0.0
            current = -1
            while True:
                best_candidate = None
                for node in remaining:
                    if route_load + demands[node] > capacity + EPS:
                        continue
                    distance = edge_length(current, node)
                    if best_candidate is None or distance < best_candidate[0]:
                        best_candidate = (distance, node)
                if best_candidate is None:
                    break
                _, chosen = best_candidate
                route_nodes.append(chosen)
                route_load += demands[chosen]
                remaining.remove(chosen)
                current = chosen
            if not route_nodes:
                chosen = remaining.pop()
                route_nodes.append(chosen)
                route_load = demands[chosen]
            fallback.append(
                {'nodes': route_nodes, 'load': route_load, 'cost': route_distance(route_nodes)}
            )
        return fallback

    routes = []
    node_route = [None] * N
    for node_idx in range(N):
        route = {
            'nodes': [node_idx],
            'load': demands[node_idx],
            'cost': dist_depot[node_idx] * 2.0,
        }
        routes.append(route)
        node_route[node_idx] = route

    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            saving = dist_depot[i] + dist_depot[j] - distance_matrix[i][j]
            if saving > 0:
                savings.append((saving, i, j))
    savings.sort(reverse=True)

    for _, i, j in savings:
        ri = node_route[i]
        rj = node_route[j]
        if ri is None or rj is None or ri is rj:
            continue
        if not ri['nodes'] or not rj['nodes']:
            continue
        if ri['load'] + rj['load'] > capacity + EPS:
            continue
        pos_i = node_position_in_route(ri['nodes'], i)
        pos_j = node_position_in_route(rj['nodes'], j)
        if pos_i == 'middle' or pos_j == 'middle':
            continue
        new_nodes, new_cost = join_routes(ri['nodes'], rj['nodes'], pos_i, pos_j)
        old_nodes_rj = rj['nodes'][:]
        ri['nodes'] = new_nodes
        ri['load'] += rj['load']
        ri['cost'] = new_cost
        rj['nodes'] = []
        rj['load'] = 0.0
        rj['cost'] = 0.0
        for node in old_nodes_rj:
            node_route[node] = ri

    routes = [route for route in routes if route['nodes']]
    enforce_vehicle_limit(routes)

    for route in routes:
        optimized = two_opt_route(route['nodes'])
        route['nodes'] = optimized
        route['cost'] = route_distance(optimized)

    assigned = sorted(node for route in routes for node in route['nodes'])
    if len(assigned) != N or any(idx != assigned[idx] for idx in range(N)):
        routes = build_fallback_routes()
        enforce_vehicle_limit(routes)
        for route in routes:
            optimized = two_opt_route(route['nodes'])
            route['nodes'] = optimized
            route['cost'] = route_distance(optimized)

    routes.sort(key=lambda r: r['nodes'][0] if r['nodes'] else -1)
    total_distance = sum(route['cost'] for route in routes)

    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(str(node) for node in route['nodes']))


if __name__ == "__main__":
    main()
```
