namespace {
constexpr double kPi = 3.14159265358979323846;

double wrapAngle(double angle) {
    constexpr double two_pi = 2.0 * kPi;
    angle = std::fmod(angle + kPi, two_pi);
    if (angle < 0.0) {
        angle += two_pi;
    }
    return angle - kPi;
}

struct Vec2 {
    double x = 0.0;
    double y = 0.0;
};

double magnitude(const Vec2& v) {
    return std::hypot(v.x, v.y);
}

Vec2 normalize(const Vec2& v) {
    double len = magnitude(v);
    if (len < 1e-9) {
        return {0.0, 0.0};
    }
    return {v.x / len, v.y / len};
}

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
};

double magnitude(const Vec3& v) {
    return std::sqrt(v.x * v.x + v.y * v.y + v.z * v.z);
}

struct TerrainColumn {
    double ground = 0.0;
    double float_bottom = 0.0;
    double float_top = 0.0;
};

class TerrainSampler {
public:
    TerrainSampler(int w, int d, std::string path)
        : width_(w), depth_(d), path_(std::move(path)) {}

    std::optional<TerrainColumn> sampleAt(double x, double y) const {
        if (width_ <= 0 || depth_ <= 0 || path_.empty()) {
            return std::nullopt;
        }
        int xi = clampIndex(x, width_);
        int yi = clampIndex(y, depth_);
        return readColumn(xi, yi);
    }

private:
    int width_;
    int depth_;
    std::string path_;

    static int clampIndex(double value, int max_value) {
        if (max_value <= 0) {
            return 0;
        }
        int idx = static_cast<int>(std::floor(value));
        if (idx < 0) {
            idx = 0;
        }
        if (idx >= max_value) {
            idx = max_value - 1;
        }
        return idx;
    }

    std::optional<TerrainColumn> readColumn(int x, int y) const {
        std::ifstream in(path_, std::ios::binary);
        if (!in) {
            return std::nullopt;
        }
        uint64_t idx = static_cast<uint64_t>(x) * static_cast<uint64_t>(depth_) + static_cast<uint64_t>(y);
        uint64_t byte_offset = idx * 3ull * sizeof(uint16_t);
        in.seekg(static_cast<std::streamoff>(byte_offset), std::ios::beg);
        if (!in) {
            return std::nullopt;
        }
        uint16_t raw[3] = {0, 0, 0};
        in.read(reinterpret_cast<char*>(raw), sizeof(raw));
        if (!in) {
            return std::nullopt;
        }
        TerrainColumn column;
        column.ground = static_cast<double>(raw[0]);
        column.float_bottom = static_cast<double>(raw[1]);
        column.float_top = static_cast<double>(raw[2]);
        return column;
    }
};

struct SimulationConfig {
    int width = 0;
    int depth = 0;
    int height = 0;
    double gravity = 1.62;
    double max_thrust = 0.0;
    double initial_fuel = 0.0;
    Vec3 start;
    Vec3 target;
    double landing_radius = 0.0;
};

struct State {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
    double vx = 0.0;
    double vy = 0.0;
    double vz = 0.0;
    double pitch = 0.0;
    double yaw = 0.0;
    double roll = 0.0;
    double pitch_rate = 0.0;
    double yaw_rate = 0.0;
    double roll_rate = 0.0;
    double fuel = 0.0;
    double timestamp = 0.0;
};

struct ControlOutput {
    double thrust = 0.0;
    double pitch_cmd = 0.0;
    double yaw_cmd = 0.0;
    double roll_cmd = 0.0;
};

struct SharedState {
    State latest{};
    std::mutex mutex;
    std::atomic<uint64_t> counter{0};
    std::atomic<bool> has_state{false};
    std::atomic<bool> stream_alive{true};
};

class Autopilot {
public:
    Autopilot(const SimulationConfig& cfg, double landing_altitude)
        : cfg_(cfg),
          landing_altitude_(landing_altitude),
          landing_radius_(std::max(5.0, cfg.landing_radius)),
          max_horizontal_speed_(std::max(15.0, 0.015 * std::hypot(static_cast<double>(cfg.width), static_cast<double>(cfg.depth)))),
          max_tilt_rad_(1.05),
          thrust_limit_(cfg.max_thrust > 0.0 ? cfg.max_thrust : std::max(cfg.gravity + 1.0, 1.0)) {}

    ControlOutput update(const State& state) {
        ControlOutput output;

        Vec2 pos_err_xy{cfg_.target.x - state.x, cfg_.target.y - state.y};
        double dist_xy = magnitude(pos_err_xy);
        Vec2 vel_xy{state.vx, state.vy};

        double desired_speed_xy = std::clamp(dist_xy * 0.3, 0.0, max_horizontal_speed_);
        if (dist_xy < landing_radius_ * 0.75) {
            desired_speed_xy = std::min(desired_speed_xy, 2.0);
        }
        Vec2 dir_xy = dist_xy > 1e-3 ? normalize(pos_err_xy) : Vec2{0.0, 0.0};
        Vec2 desired_vel_xy{dir_xy.x * desired_speed_xy, dir_xy.y * desired_speed_xy};
        Vec2 vel_error_xy{desired_vel_xy.x - vel_xy.x, desired_vel_xy.y - vel_xy.y};

        constexpr double kPosXY = 0.02;
        constexpr double kVelXY = 0.85;
        Vec2 acc_xy{
            kPosXY * pos_err_xy.x + kVelXY * vel_error_xy.x,
            kPosXY * pos_err_xy.y + kVelXY * vel_error_xy.y};

        double horizontal_limit = thrust_limit_ * 0.7;
        double acc_xy_mag = magnitude(acc_xy);
        if (acc_xy_mag > horizontal_limit && acc_xy_mag > 1e-6) {
            double scale = horizontal_limit / acc_xy_mag;
            acc_xy.x *= scale;
            acc_xy.y *= scale;
        }

        double extra_altitude = std::clamp(dist_xy * 0.1, 0.0, std::max(80.0, 0.05 * static_cast<double>(cfg_.height)));
        double desired_altitude = std::min(static_cast<double>(cfg_.height) - 1.0, landing_altitude_ + extra_altitude);
        double pos_err_z = desired_altitude - state.z;

        double base_down_speed = 0.5 + 0.015 * std::max(0.0, state.z - landing_altitude_);
        double max_down_speed = std::clamp(base_down_speed, 1.0, 12.0);
        if (dist_xy < landing_radius_ * 2.0) {
            max_down_speed = std::min(max_down_speed, 3.0);
        }
        if (state.z - landing_altitude_ < 8.0) {
            max_down_speed = 1.0;
        }
        double max_up_speed = 6.0;
        double desired_vz = std::clamp(pos_err_z * 0.4, -max_down_speed, max_up_speed);
        double vel_err_z = desired_vz - state.vz;

        constexpr double kPosZ = 0.08;
        constexpr double kVelZ = 1.8;
        double acc_z = kPosZ * pos_err_z + kVelZ * vel_err_z;

        Vec3 thrust_vec{acc_xy.x, acc_xy.y, acc_z + cfg_.gravity};
        if (thrust_vec.z < 0.0) {
            thrust_vec.z = 0.0;
        }

        if (thrust_vec.z > 1e-3) {
            double horizontal = std::hypot(thrust_vec.x, thrust_vec.y);
            double max_horizontal = thrust_vec.z * std::tan(max_tilt_rad_);
            if (horizontal > max_horizontal && horizontal > 1e-6) {
                double scale = max_horizontal / horizontal;
                thrust_vec.x *= scale;
                thrust_vec.y *= scale;
            }
        } else {
            thrust_vec.z = 1e-3;
            thrust_vec.x = 0.0;
            thrust_vec.y = 0.0;
        }

        double thr_mag = magnitude(thrust_vec);
        if (thr_mag > thrust_limit_) {
            double scale = thrust_limit_ / (thr_mag + 1e-9);
            thrust_vec.x *= scale;
            thrust_vec.y *= scale;
            thrust_vec.z *= scale;
            thr_mag = thrust_limit_;
        }
        if (thr_mag < 1e-6) {
            thrust_vec = {0.0, 0.0, 1e-6};
            thr_mag = 1e-6;
        }

        Vec3 dir{thrust_vec.x / thr_mag, thrust_vec.y / thr_mag, thrust_vec.z / thr_mag};

        double horiz = std::hypot(dir.x, dir.y);
        if (horiz > 1e-4) {
            yaw_target_ = std::atan2(-dir.y, -dir.x);
        }
        double sin_pitch = -(dir.x * std::cos(yaw_target_) + dir.y * std::sin(yaw_target_));
        sin_pitch = std::clamp(sin_pitch, -1.0, 1.0);
        double pitch_des = std::asin(sin_pitch);
        pitch_des = std::clamp(pitch_des, -max_tilt_rad_, max_tilt_rad_);

        constexpr double ang_acc_per_cmd = 2.0;
        constexpr double kp_pitch = 3.8;
        constexpr double kd_pitch = 1.1;
        constexpr double kp_yaw = 2.4;
        constexpr double kd_yaw = 0.9;
        constexpr double kp_roll = 2.0;
        constexpr double kd_roll = 0.7;

        double pitch_error = wrapAngle(pitch_des - state.pitch);
        double yaw_error = wrapAngle(yaw_target_ - state.yaw);

        double pitch_acc = kp_pitch * pitch_error - kd_pitch * state.pitch_rate;
        double yaw_acc = kp_yaw * yaw_error - kd_yaw * state.yaw_rate;
        double roll_acc = -(kp_roll * state.roll + kd_roll * state.roll_rate);

        output.pitch_cmd = std::clamp(pitch_acc / ang_acc_per_cmd, -1.0, 1.0);
        output.yaw_cmd = std::clamp(yaw_acc / ang_acc_per_cmd, -1.0, 1.0);
        output.roll_cmd = std::clamp(roll_acc / ang_acc_per_cmd, -1.0, 1.0);
        output.thrust = std::clamp(thr_mag / thrust_limit_, 0.0, 1.0);

        return output;
    }

private:
    SimulationConfig cfg_;
    double landing_altitude_;
    double landing_radius_;
    double max_horizontal_speed_;
    double max_tilt_rad_;
    double thrust_limit_;
    double yaw_target_ = 0.0;
};

void stateReader(SharedState& shared) {
    auto start_time = std::chrono::steady_clock::now();
    while (true) {
        State incoming;
        if (!(std::cin >> incoming.x >> incoming.y >> incoming.z >> incoming.vx >> incoming.vy >> incoming.vz
                      >> incoming.pitch >> incoming.yaw >> incoming.roll
                      >> incoming.pitch_rate >> incoming.yaw_rate >> incoming.roll_rate >> incoming.fuel)) {
            break;
        }
        incoming.timestamp = std::chrono::duration<double>(std::chrono::steady_clock::now() - start_time).count();
        {
            std::lock_guard<std::mutex> lock(shared.mutex);
            shared.latest = incoming;
        }
        shared.counter.fetch_add(1, std::memory_order_release);
        shared.has_state.store(true, std::memory_order_release);
    }
    shared.stream_alive.store(false, std::memory_order_release);
}

void controlLoop(SimulationConfig cfg, double landing_altitude, SharedState& shared, std::atomic<bool>& keep_running) {
    Autopilot controller(cfg, landing_altitude);
    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(6);

    auto next_tick = std::chrono::steady_clock::now();
    const auto period = std::chrono::milliseconds(50);
    State local{};
    bool have_state = false;
    uint64_t last_counter = 0;

    while (keep_running.load(std::memory_order_acquire) || shared.stream_alive.load(std::memory_order_acquire)) {
        if (!shared.has_state.load(std::memory_order_acquire)) {
            std::this_thread::sleep_for(std::chrono::milliseconds(2));
            next_tick = std::chrono::steady_clock::now();
            continue;
        }

        uint64_t counter = shared.counter.load(std::memory_order_acquire);
        if (!have_state || counter != last_counter) {
            std::lock_guard<std::mutex> lock(shared.mutex);
            local = shared.latest;
            have_state = true;
            last_counter = counter;
        }

        if (have_state) {
            ControlOutput cmd = controller.update(local);
            std::cout << cmd.thrust << ' ' << cmd.pitch_cmd << ' ' << cmd.yaw_cmd << ' ' << cmd.roll_cmd << '\n';
            std::cout.flush();
        }

        next_tick += period;
        auto now = std::chrono::steady_clock::now();
        if (next_tick > now) {
            std::this_thread::sleep_until(next_tick);
        } else {
            next_tick = now;
        }
    }
}
}  // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    SimulationConfig cfg;
    if (!(std::cin >> cfg.width >> cfg.depth >> cfg.height >> cfg.gravity >> cfg.max_thrust >> cfg.initial_fuel)) {
        return 0;
    }
    if (!(std::cin >> cfg.start.x >> cfg.start.y >> cfg.start.z >> cfg.target.x >> cfg.target.y >> cfg.target.z)) {
        return 0;
    }

    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    std::string marker;
    if (!(std::cin >> marker)) {
        return 0;
    }

    cfg.landing_radius = std::max(5.0, static_cast<double>(std::min(cfg.width, cfg.depth)) * 0.02);

    TerrainSampler sampler(cfg.width, cfg.depth, terrain_path);
    double landing_altitude = cfg.target.z;
    if (auto column = sampler.sampleAt(cfg.target.x, cfg.target.y)) {
        double min_alt = column->ground + 0.5;
        double max_alt = static_cast<double>(cfg.height) - 1.0;
        if (column->float_bottom < column->float_top) {
            max_alt = std::min(max_alt, column->float_bottom - 1.0);
        }
        if (max_alt < min_alt) {
            max_alt = min_alt;
        }
        landing_altitude = std::clamp(landing_altitude, min_alt, max_alt);
    }

    SharedState shared;
    shared.stream_alive.store(true, std::memory_order_release);

    std::atomic<bool> keep_running{true};

    std::thread reader_thread(stateReader, std::ref(shared));
    std::thread control_thread(controlLoop, cfg, landing_altitude, std::ref(shared), std::ref(keep_running));

    reader_thread.join();
    keep_running.store(false, std::memory_order_release);
    control_thread.join();

    return 0;
}
