using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        int m = scanner.NextInt();

        if (n < 0) n = 0;
        if (m < 0) m = 0;

        var adjacencyLists = new List<int>[n];
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();

            if (u < 0 || v < 0 || u >= n || v >= n || u == v)
            {
                continue;
            }

            adjacencyLists[u] ??= new List<int>();
            adjacencyLists[u].Add(v);

            adjacencyLists[v] ??= new List<int>();
            adjacencyLists[v].Add(u);
        }

        var adjacency = new int[n][];
        var degrees = new int[n];
        long edgeCount = 0;

        for (int i = 0; i < n; i++)
        {
            var list = adjacencyLists[i];
            if (list == null || list.Count == 0)
            {
                adjacency[i] = Array.Empty<int>();
                continue;
            }

            list.Sort();
            int writeIndex = 0;
            int prev = -1;
            for (int j = 0; j < list.Count; j++)
            {
                int neighbor = list[j];
                if (neighbor == i || neighbor == prev)
                {
                    continue;
                }
                list[writeIndex++] = neighbor;
                prev = neighbor;
            }

            if (writeIndex == 0)
            {
                adjacency[i] = Array.Empty<int>();
                continue;
            }

            var arr = new int[writeIndex];
            for (int j = 0; j < writeIndex; j++)
            {
                arr[j] = list[j];
            }

            adjacency[i] = arr;
            degrees[i] = arr.Length;
            edgeCount += arr.Length;
        }

        edgeCount /= 2;

        var solver = new MaxCliqueSolver(adjacency, degrees, edgeCount, TimeSpan.FromMilliseconds(295000));
        var clique = solver.Solve();
        clique.Sort();

        var output = new StringBuilder();
        output.AppendLine(clique.Count.ToString());
        if (clique.Count > 0)
        {
            for (int i = 0; i < clique.Count; i++)
            {
                if (i > 0) output.Append(' ');
                output.Append(clique[i]);
            }
            output.AppendLine();
        }
        else
        {
            output.AppendLine();
        }

        Console.Write(output.ToString());
    }
}

internal sealed class FastScanner
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr;
    private int length;

    public FastScanner(Stream stream)
    {
        this.stream = stream;
    }

    private int Read()
    {
        if (ptr >= length)
        {
            length = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (length <= 0) return -1;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return 0;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int value = 0;
        while (c > ' ')
        {
            value = value * 10 + (c - '0');
            c = Read();
        }

        return value * sign;
    }
}

internal sealed class MaxCliqueSolver
{
    private readonly int[][] adjacency;
    private readonly int[] degrees;
    private readonly int n;
    private readonly long edgeCount;
    private readonly Stopwatch stopwatch;
    private readonly long timeLimitMs;

    private readonly bool[] neighborMark;
    private readonly bool[] colorMark;

    private readonly List<int> colorWorkA = new List<int>();
    private readonly List<int> colorWorkB = new List<int>();
    private readonly List<int> colorTouched = new List<int>();

    private readonly IntListPool listPool = new IntListPool(4096);
    private readonly Random rng = new Random(0x5F3759DF);

    private readonly List<int> heuristicBuffer = new List<int>();
    private readonly List<int> currentClique = new List<int>();
    private readonly List<int> bestClique = new List<int>();

    private readonly DegreeComparer degreeComparer;
    private int[] degreeOrder;
    private bool timeExceeded;

    public MaxCliqueSolver(int[][] adjacency, int[] degrees, long edgeCount, TimeSpan timeLimit)
    {
        this.adjacency = adjacency;
        this.degrees = degrees;
        this.edgeCount = edgeCount;
        this.n = adjacency.Length;
        this.stopwatch = Stopwatch.StartNew();
        this.timeLimitMs = Math.Max(1L, (long)timeLimit.TotalMilliseconds);
        this.neighborMark = new bool[n];
        this.colorMark = new bool[n];
        this.degreeComparer = new DegreeComparer(degrees);
    }

    public List<int> Solve()
    {
        if (n == 0)
        {
            return new List<int>();
        }

        RunHeuristics();

        if (!timeExceeded && ShouldRunExact())
        {
            var initialCandidates = listPool.Rent();
            try
            {
                for (int i = 0; i < n; i++)
                {
                    if (degrees[i] + 1 > bestClique.Count)
                    {
                        initialCandidates.Add(i);
                    }
                }

                if (initialCandidates.Count > 0)
                {
                    currentClique.Clear();
                    Expand(initialCandidates, currentClique);
                }
            }
            finally
            {
                listPool.Return(initialCandidates);
            }
        }

        if (bestClique.Count == 0 && n > 0)
        {
            bestClique.Add(0);
        }

        return new List<int>(bestClique);
    }

    private void RunHeuristics()
    {
        if (n == 0) return;
        EnsureDegreeOrder();

        int deterministicSeeds = DetermineDeterministicSeedLimit();
        for (int i = 0; i < deterministicSeeds && i < degreeOrder.Length; i++)
        {
            if (IsTimeLimitReached()) return;

            int vertex = degreeOrder[i];
            if (degrees[vertex] + 1 <= bestClique.Count)
            {
                break;
            }

            BuildCliqueFromSeed(vertex, heuristicBuffer);
            if (heuristicBuffer.Count > bestClique.Count)
            {
                UpdateBest(heuristicBuffer);
            }
        }

        int randomTrials = DetermineRandomTrialCount();
        for (int i = 0; i < randomTrials; i++)
        {
            if (IsTimeLimitReached()) return;

            int vertex = rng.Next(n);
            if (degrees[vertex] + 1 <= bestClique.Count)
            {
                continue;
            }

            BuildCliqueFromSeed(vertex, heuristicBuffer);
            if (heuristicBuffer.Count > bestClique.Count)
            {
                UpdateBest(heuristicBuffer);
            }
        }
    }

    private void BuildCliqueFromSeed(int seed, List<int> output)
    {
        var clique = listPool.Rent();
        var candidate = listPool.Rent();
        var next = listPool.Rent();

        try
        {
            clique.Add(seed);
            var neighbors = adjacency[seed];
            if (neighbors.Length > 0)
            {
                candidate.AddRange(neighbors);
            }

            while (candidate.Count > 0)
            {
                if (IsTimeLimitReached())
                {
                    break;
                }

                int chosenIndex = 0;
                int chosenVertex = candidate[0];
                int bestDegree = degrees[chosenVertex];
                for (int i = 1; i < candidate.Count; i++)
                {
                    int v = candidate[i];
                    int d = degrees[v];
                    if (d > bestDegree)
                    {
                        bestDegree = d;
                        chosenVertex = v;
                        chosenIndex = i;
                    }
                }

                int last = candidate.Count - 1;
                candidate[chosenIndex] = candidate[last];
                candidate.RemoveAt(last);

                clique.Add(chosenVertex);

                next.Clear();
                IntersectWithNeighbors(candidate, chosenVertex, next);

                var temp = candidate;
                candidate = next;
                next = temp;
            }

            output.Clear();
            output.AddRange(clique);
        }
        finally
        {
            listPool.Return(candidate);
            listPool.Return(next);
            listPool.Return(clique);
        }
    }

    private void Expand(List<int> candidates, List<int> current)
    {
        if (IsTimeLimitReached())
        {
            if (current.Count > bestClique.Count)
            {
                UpdateBest(current);
            }
            return;
        }

        if (candidates.Count == 0)
        {
            if (current.Count > bestClique.Count)
            {
                UpdateBest(current);
            }
            return;
        }

        if (current.Count + candidates.Count <= bestClique.Count)
        {
            return;
        }

        var order = listPool.Rent();
        var colorBounds = listPool.Rent();

        try
        {
            ColorSort(candidates, order, colorBounds);

            for (int idx = order.Count - 1; idx >= 0; idx--)
            {
                if (IsTimeLimitReached())
                {
                    if (current.Count > bestClique.Count)
                    {
                        UpdateBest(current);
                    }
                    break;
                }

                int bound = colorBounds[idx];
                if (current.Count + bound <= bestClique.Count)
                {
                    break;
                }

                int vertex = order[idx];
                if (degrees[vertex] + 1 <= bestClique.Count)
                {
                    continue;
                }

                var nextCandidates = listPool.Rent();
                try
                {
                    BuildNextCandidates(order, idx, vertex, nextCandidates);

                    current.Add(vertex);
                    if (nextCandidates.Count == 0)
                    {
                        if (current.Count > bestClique.Count)
                        {
                            UpdateBest(current);
                        }
                    }
                    else
                    {
                        Expand(nextCandidates, current);
                    }
                    current.RemoveAt(current.Count - 1);
                }
                finally
                {
                    listPool.Return(nextCandidates);
                }
            }
        }
        finally
        {
            listPool.Return(order);
            listPool.Return(colorBounds);
        }
    }

    private void ColorSort(List<int> candidates, List<int> orderOut, List<int> colorOut)
    {
        orderOut.Clear();
        colorOut.Clear();

        if (candidates.Count == 0)
        {
            return;
        }

        var nodes = colorWorkA;
        var next = colorWorkB;
        var touched = colorTouched;

        nodes.Clear();
        nodes.AddRange(candidates);
        nodes.Sort(degreeComparer);
        next.Clear();

        int color = 0;
        while (nodes.Count > 0)
        {
            color++;
            touched.Clear();

            for (int i = 0; i < nodes.Count; i++)
            {
                int vertex = nodes[i];
                if (colorMark[vertex])
                {
                    next.Add(vertex);
                    continue;
                }

                orderOut.Add(vertex);
                colorOut.Add(color);
                colorMark[vertex] = true;
                touched.Add(vertex);

                var neighbors = adjacency[vertex];
                for (int j = 0; j < neighbors.Length; j++)
                {
                    int neighbor = neighbors[j];
                    if (!colorMark[neighbor])
                    {
                        colorMark[neighbor] = true;
                        touched.Add(neighbor);
                    }
                }
            }

            foreach (int t in touched)
            {
                colorMark[t] = false;
            }

            nodes.Clear();
            var temp = nodes;
            nodes = next;
            next = temp;
            next.Clear();
        }
    }

    private void BuildNextCandidates(List<int> order, int limitExclusive, int vertex, List<int> dest)
    {
        dest.Clear();
        var neighbors = adjacency[vertex];
        if (neighbors.Length == 0)
        {
            return;
        }

        for (int i = 0; i < neighbors.Length; i++)
        {
            neighborMark[neighbors[i]] = true;
        }

        for (int i = 0; i < limitExclusive; i++)
        {
            int candidate = order[i];
            if (neighborMark[candidate])
            {
                dest.Add(candidate);
            }
        }

        for (int i = 0; i < neighbors.Length; i++)
        {
            neighborMark[neighbors[i]] = false;
        }
    }

    private void IntersectWithNeighbors(List<int> source, int vertex, List<int> dest)
    {
        dest.Clear();
        if (source.Count == 0)
        {
            return;
        }

        var neighbors = adjacency[vertex];
        if (neighbors.Length == 0)
        {
            return;
        }

        for (int i = 0; i < neighbors.Length; i++)
        {
            neighborMark[neighbors[i]] = true;
        }

        for (int i = 0; i < source.Count; i++)
        {
            int candidate = source[i];
            if (neighborMark[candidate])
            {
                dest.Add(candidate);
            }
        }

        for (int i = 0; i < neighbors.Length; i++)
        {
            neighborMark[neighbors[i]] = false;
        }
    }

    private bool ShouldRunExact()
    {
        if (n == 0)
        {
            return false;
        }

        if (n <= 2000)
        {
            return true;
        }

        if (n <= 4000 && edgeCount <= 6_000_000)
        {
            return true;
        }

        if (n <= 6000 && edgeCount <= 4_000_000)
        {
            return true;
        }

        return false;
    }

    private void EnsureDegreeOrder()
    {
        if (degreeOrder != null)
        {
            return;
        }

        degreeOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            degreeOrder[i] = i;
        }

        Array.Sort(degreeOrder, degreeComparer);
    }

    private int DetermineDeterministicSeedLimit()
    {
        if (n <= 2000) return n;
        if (n <= 8000) return 4000;
        if (n <= 20000) return 3000;
        if (n <= 100000) return 1500;
        if (n <= 300000) return 800;
        return 400;
    }

    private int DetermineRandomTrialCount()
    {
        if (n <= 10000) return 120;
        if (n <= 100000) return 200;
        return 320;
    }

    private bool IsTimeLimitReached()
    {
        if (timeExceeded)
        {
            return true;
        }

        if (stopwatch.ElapsedMilliseconds >= timeLimitMs)
        {
            timeExceeded = true;
            return true;
        }

        return false;
    }

    private void UpdateBest(IList<int> clique)
    {
        bestClique.Clear();
        for (int i = 0; i < clique.Count; i++)
        {
            bestClique.Add(clique[i]);
        }
    }

    private sealed class DegreeComparer : IComparer<int>
    {
        private readonly int[] degrees;
        public DegreeComparer(int[] degrees) => this.degrees = degrees;

        public int Compare(int x, int y)
        {
            int dx = degrees[x];
            int dy = degrees[y];
            if (dx == dy) return x.CompareTo(y);
            return dy.CompareTo(dx);
        }
    }

    private sealed class IntListPool
    {
        private readonly Stack<List<int>> pool = new Stack<List<int>>();
        private readonly int maxSize;

        public IntListPool(int maxSize)
        {
            this.maxSize = Math.Max(4, maxSize);
        }

        public List<int> Rent()
        {
            if (pool.Count == 0)
            {
                return new List<int>();
            }

            var list = pool.Pop();
            list.Clear();
            return list;
        }

        public void Return(List<int> list)
        {
            if (list == null)
            {
                return;
            }

            list.Clear();
            if (pool.Count < maxSize)
            {
                pool.Push(list);
            }
        }
    }
}
