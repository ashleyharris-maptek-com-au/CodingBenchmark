Here’s a complete Python program that reads a 3‑SAT instance from stdin, uses a WalkSAT-style local-search solver tailored for very large formulas, and prints a satisfying assignment that meets the required format:

```python
import sys
import time
import random
from array import array


def int_stream():
    buf = sys.stdin.buffer
    for chunk in buf:
        for token in chunk.split():
            yield int(token)


def read_instance():
    ints = int_stream()
    try:
        num_vars = next(ints)
        num_clauses = next(ints)
    except StopIteration:
        return 0, 0, array('i'), array('I', [0]), array('I')

    clauses = array('i')
    if num_vars > 0:
        occ_counts = array('I', [0]) * num_vars
    else:
        occ_counts = array('I')

    actual_clauses = 0
    for _ in range(num_clauses):
        try:
            a = next(ints)
            b = next(ints)
            c = next(ints)
        except StopIteration:
            break
        clauses.append(a)
        clauses.append(b)
        clauses.append(c)
        actual_clauses += 1
        if num_vars:
            for lit in (a, b, c):
                var = abs(lit) - 1
                if 0 <= var < num_vars:
                    occ_counts[var] += 1

    num_clauses = actual_clauses
    if num_vars > 0:
        occ_offsets = array('I', [0]) * (num_vars + 1)
        total = 0
        for i in range(num_vars):
            occ_offsets[i] = total
            total += occ_counts[i]
        occ_offsets[num_vars] = total
        occ_data = array('I', [0]) * total
        occ_cursor = array('I', occ_offsets)
        for clause_idx in range(num_clauses):
            base = clause_idx * 3
            for j in range(3):
                lit = clauses[base + j]
                var = abs(lit) - 1
                if 0 <= var < num_vars:
                    pos = occ_cursor[var]
                    occ_data[pos] = clause_idx
                    occ_cursor[var] = pos + 1
        del occ_cursor
    else:
        occ_offsets = array('I', [0])
        occ_data = array('I')

    return num_vars, num_clauses, clauses, occ_offsets, occ_data


class WalkSATSolver:
    __slots__ = (
        'n', 'm', 'clauses', 'occ_offsets', 'occ_data', 'assignment',
        'clause_status', 'unsat_pool', 'unsat_total', 'rng', 'noise'
    )

    def __init__(self, n, m, clauses, occ_offsets, occ_data):
        self.n = n
        self.m = m
        self.clauses = clauses
        self.occ_offsets = occ_offsets
        self.occ_data = occ_data
        self.assignment = bytearray(n)
        self.clause_status = bytearray(m)
        self.unsat_pool = array('I')
        self.unsat_total = 0
        seed = (time.time_ns() ^ (n << 32) ^ m) & ((1 << 64) - 1)
        self.rng = random.Random(seed)
        self.noise = 0.3

    def clause_value(self, clause_idx):
        base = clause_idx * 3
        data = self.clauses
        assign = self.assignment
        n = self.n

        lit = data[base]
        if lit > 0:
            idx = lit - 1
            if 0 <= idx < n and assign[idx]:
                return 1
        elif lit < 0:
            idx = -lit - 1
            if 0 <= idx < n and not assign[idx]:
                return 1

        lit = data[base + 1]
        if lit > 0:
            idx = lit - 1
            if 0 <= idx < n and assign[idx]:
                return 1
        elif lit < 0:
            idx = -lit - 1
            if 0 <= idx < n and not assign[idx]:
                return 1

        lit = data[base + 2]
        if lit > 0:
            idx = lit - 1
            if 0 <= idx < n and assign[idx]:
                return 1
        elif lit < 0:
            idx = -lit - 1
            if 0 <= idx < n and not assign[idx]:
                return 1

        return 0

    def randomize_assignment(self):
        n = self.n
        if n == 0:
            return
        assign = self.assignment
        rng = self.rng
        idx = 0
        full_chunks = n // 64
        for _ in range(full_chunks):
            bits = rng.getrandbits(64)
            for _ in range(64):
                assign[idx] = bits & 1
                bits >>= 1
                idx += 1
        remaining = n - idx
        if remaining:
            bits = rng.getrandbits(remaining)
            for _ in range(remaining):
                assign[idx] = bits & 1
                bits >>= 1
                idx += 1

    def initialize_state(self):
        self.unsat_pool = array('I')
        pool = self.unsat_pool
        clause_status = self.clause_status
        unsat_total = 0
        clause_value = self.clause_value
        for idx in range(self.m):
            val = clause_value(idx)
            clause_status[idx] = val
            if not val:
                pool.append(idx)
                unsat_total += 1
        self.unsat_total = unsat_total

    def refill_unsat_pool(self):
        pool = self.unsat_pool
        status = self.clause_status
        pool.extend(idx for idx in range(self.m) if not status[idx])

    def pick_unsatisfied_clause(self):
        pool = self.unsat_pool
        status = self.clause_status
        rng = self.rng
        while True:
            if not pool:
                self.refill_unsat_pool()
                if not pool:
                    return None
            idx = rng.randrange(len(pool))
            clause_idx = pool[idx]
            pool[idx] = pool[-1]
            pool.pop()
            if clause_idx < self.m and not status[clause_idx]:
                return clause_idx

    def compute_delta(self, var):
        if not (0 <= var < self.n):
            return 0
        assign = self.assignment
        offsets = self.occ_offsets
        data = self.occ_data
        clause_status = self.clause_status
        clause_value = self.clause_value

        assign[var] ^= 1
        delta = 0
        start = offsets[var]
        end = offsets[var + 1] if var + 1 < len(offsets) else start
        for idx in range(start, end):
            clause_idx = data[idx]
            new_val = clause_value(clause_idx)
            old_val = clause_status[clause_idx]
            if new_val and not old_val:
                delta -= 1
            elif not new_val and old_val:
                delta += 1
        assign[var] ^= 1
        return delta

    def flip_variable(self, var):
        if not (0 <= var < self.n):
            return
        assign = self.assignment
        offsets = self.occ_offsets
        data = self.occ_data
        clause_status = self.clause_status
        pool = self.unsat_pool
        clause_value = self.clause_value

        assign[var] ^= 1
        unsat_total = self.unsat_total
        start = offsets[var]
        end = offsets[var + 1] if var + 1 < len(offsets) else start
        for idx in range(start, end):
            clause_idx = data[idx]
            new_val = clause_value(clause_idx)
            old_val = clause_status[clause_idx]
            if new_val != old_val:
                clause_status[clause_idx] = new_val
                if new_val:
                    unsat_total -= 1
                else:
                    unsat_total += 1
                    pool.append(clause_idx)
        self.unsat_total = unsat_total

    def choose_variable(self, clause_idx):
        base = clause_idx * 3
        data = self.clauses
        rng = self.rng
        if self.n == 0:
            return None

        if rng.random() < self.noise:
            for _ in range(3):
                lit = data[base + rng.randrange(3)]
                var = abs(lit) - 1
                if 0 <= var < self.n:
                    return var
            return None

        seen = []
        best_delta = None
        best_vars = []
        for j in range(3):
            lit = data[base + j]
            var = abs(lit) - 1
            if not (0 <= var < self.n):
                continue
            if var in seen:
                continue
            seen.append(var)
            delta = self.compute_delta(var)
            if best_delta is None or delta < best_delta:
                best_delta = delta
                best_vars = [var]
                if best_delta < 0:
                    break
            elif delta == best_delta:
                best_vars.append(var)

        if best_vars:
            return rng.choice(best_vars)

        for j in range(3):
            lit = data[base + j]
            var = abs(lit) - 1
            if 0 <= var < self.n:
                return var
        return None

    def solve(self, deadline):
        if self.m == 0:
            return True

        size_hint = max(1, self.n, self.m)
        base_limit = max(10000, min(size_hint * 4, 8_000_000))
        restart = 0
        time_check_interval = 4096

        while True:
            if time.time() >= deadline:
                return self.unsat_total == 0
            self.randomize_assignment()
            self.initialize_state()
            if self.unsat_total == 0:
                return True

            flips = 0
            local_limit = min(base_limit + restart * 20000, 20_000_000)
            checks_left = time_check_interval

            while self.unsat_total and flips < local_limit:
                clause_idx = self.pick_unsatisfied_clause()
                if clause_idx is None:
                    break
                var = self.choose_variable(clause_idx)
                if var is None:
                    continue
                self.flip_variable(var)
                if clause_idx < self.m and not self.clause_status[clause_idx]:
                    self.unsat_pool.append(clause_idx)
                flips += 1
                checks_left -= 1
                if checks_left == 0:
                    if time.time() >= deadline:
                        return self.unsat_total == 0
                    checks_left = time_check_interval

            if self.unsat_total == 0:
                return True
            restart += 1


def write_assignment(bits):
    out = sys.stdout.write
    if not bits:
        out("\n")
        return
    first = True
    for val in bits:
        if first:
            out('1' if val else '0')
            first = False
        else:
            out(' ')
            out('1' if val else '0')
    out("\n")


def main():
    n, m, clauses, occ_offsets, occ_data = read_instance()

    if n == 0:
        print("SAT")
        sys.stdout.write("\n")
        return

    if m == 0:
        print("SAT")
        write_assignment(bytearray(n))
        return

    solver = WalkSATSolver(n, m, clauses, occ_offsets, occ_data)
    deadline = time.time() + 295.0
    success = solver.solve(deadline)

    if not success or solver.unsat_total != 0:
        extra_deadline = min(deadline, time.time() + 1.0)
        while time.time() < extra_deadline and solver.unsat_total:
            solver.randomize_assignment()
            solver.initialize_state()
            if solver.unsat_total == 0:
                success = True
                break

    if solver.unsat_total != 0:
        raise SystemExit("Failed to find satisfying assignment within time limit")

    print("SAT")
    write_assignment(solver.assignment)


if __name__ == "__main__":
    main()
```

This solver uses memory-efficient arrays, keeps per-variable occurrence lists, and performs randomized WalkSAT with adaptive restarts to handle instances scaling up to tens of millions of variables and clauses within the required runtime.
