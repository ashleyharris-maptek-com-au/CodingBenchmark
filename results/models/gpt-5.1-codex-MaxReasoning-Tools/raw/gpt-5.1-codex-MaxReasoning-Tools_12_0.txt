Here is a complete, compilable C# solver that adapts its strategy (space selection, orientation scoring, tiling density, and memory trimming) to the detected scale of the problem while respecting the required input/output formats and quaternion conventions:

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Text;

internal static class Program
{
    private static void Main()
    {
        var scanner = new FastScanner(Console.In);
        string? firstToken = scanner.Next();
        if (firstToken == null)
        {
            Console.WriteLine("0");
            return;
        }

        int vertexCount = int.Parse(firstToken, CultureInfo.InvariantCulture);
        int faceCount = scanner.NextInt();

        Vector3d min = new Vector3d(double.MaxValue, double.MaxValue, double.MaxValue);
        Vector3d max = new Vector3d(double.MinValue, double.MinValue, double.MinValue);

        for (int i = 0; i < vertexCount; i++)
        {
            double x = scanner.NextDouble();
            double y = scanner.NextDouble();
            double z = scanner.NextDouble();
            var v = new Vector3d(x, y, z);
            min = Vector3d.Min(min, v);
            max = Vector3d.Max(max, v);
        }

        for (int i = 0; i < faceCount; i++)
        {
            scanner.NextInt();
            scanner.NextInt();
            scanner.NextInt();
        }

        double width = scanner.NextDouble();
        double height = scanner.NextDouble();
        double depth = scanner.NextDouble();

        if (vertexCount == 0)
        {
            min = Vector3d.Zero;
            max = new Vector3d(1.0, 1.0, 1.0);
        }

        if (width <= 0.0 || height <= 0.0 || depth <= 0.0)
        {
            Console.WriteLine("0");
            return;
        }

        var orientations = OrientationGenerator.Generate(min, max);
        if (orientations.Count == 0)
        {
            orientations.Add(OrientationGenerator.CreateFallback());
        }

        var axisMinimums = OrientationGenerator.ComputeAxisMinimums(orientations);
        var scale = ScaleClassifier.Classify(width, height, depth);
        var container = new Space { Origin = Vector3d.Zero, Size = new Vector3d(width, height, depth) };

        var solver = new PackingSolver(container, orientations, scale, axisMinimums);
        var placements = solver.Execute();

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < placements.Count; i++)
        {
            Placement p = placements[i];
            writer.Write(p.Translation.X.ToString("G17", CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.Translation.Y.ToString("G17", CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.Translation.Z.ToString("G17", CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.Rotation.W.ToString("G17", CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.Rotation.X.ToString("G17", CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.Rotation.Y.ToString("G17", CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.WriteLine(p.Rotation.Z.ToString("G17", CultureInfo.InvariantCulture));
        }
    }
}

internal sealed class FastScanner
{
    private readonly TextReader _reader;

    public FastScanner(TextReader reader)
    {
        _reader = reader;
    }

    public string? Next()
    {
        var builder = new StringBuilder();
        int ch;
        do
        {
            ch = _reader.Read();
            if (ch == -1)
            {
                return null;
            }
        } while (char.IsWhiteSpace((char)ch));

        do
        {
            builder.Append((char)ch);
            ch = _reader.Read();
        } while (ch != -1 && !char.IsWhiteSpace((char)ch));

        return builder.ToString();
    }

    public int NextInt()
    {
        string? token = Next();
        if (token == null)
        {
            throw new InvalidOperationException("Unexpected end of input while reading int.");
        }

        return int.Parse(token, CultureInfo.InvariantCulture);
    }

    public double NextDouble()
    {
        string? token = Next();
        if (token == null)
        {
            throw new InvalidOperationException("Unexpected end of input while reading double.");
        }

        return double.Parse(token, CultureInfo.InvariantCulture);
    }
}

internal enum Scale
{
    Trivial,
    Medium,
    Large,
    Extreme
}

internal static class ScaleClassifier
{
    public static Scale Classify(double width, double height, double depth)
    {
        double maxEdge = Math.Max(Math.Abs(width), Math.Max(Math.Abs(height), Math.Abs(depth)));
        if (maxEdge <= 5.0)
        {
            return Scale.Trivial;
        }

        if (maxEdge <= 20.0)
        {
            return Scale.Medium;
        }

        if (maxEdge <= 100.0)
        {
            return Scale.Large;
        }

        return Scale.Extreme;
    }
}

internal readonly struct Vector3d
{
    public readonly double X;
    public readonly double Y;
    public readonly double Z;

    public Vector3d(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public static Vector3d operator +(Vector3d a, Vector3d b) => new Vector3d(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3d operator -(Vector3d a, Vector3d b) => new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3d operator *(Vector3d v, double scalar) => new Vector3d(v.X * scalar, v.Y * scalar, v.Z * scalar);
    public static Vector3d operator *(double scalar, Vector3d v) => v * scalar;

    public static Vector3d Min(Vector3d a, Vector3d b) => new Vector3d(Math.Min(a.X, b.X), Math.Min(a.Y, b.Y), Math.Min(a.Z, b.Z));
    public static Vector3d Max(Vector3d a, Vector3d b) => new Vector3d(Math.Max(a.X, b.X), Math.Max(a.Y, b.Y), Math.Max(a.Z, b.Z));

    public static double Dot(Vector3d a, Vector3d b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

    public static Vector3d Cross(Vector3d a, Vector3d b)
    {
        return new Vector3d(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    public double LengthSquared => X * X + Y * Y + Z * Z;

    public static Vector3d Zero => new Vector3d(0.0, 0.0, 0.0);
}

internal readonly struct Matrix3x3
{
    public readonly Vector3d Column0;
    public readonly Vector3d Column1;
    public readonly Vector3d Column2;

    public Matrix3x3(Vector3d column0, Vector3d column1, Vector3d column2)
    {
        Column0 = column0;
        Column1 = column1;
        Column2 = column2;
    }

    public Vector3d Multiply(Vector3d v)
    {
        return new Vector3d(
            Column0.X * v.X + Column1.X * v.Y + Column2.X * v.Z,
            Column0.Y * v.X + Column1.Y * v.Y + Column2.Y * v.Z,
            Column0.Z * v.X + Column1.Z * v.Y + Column2.Z * v.Z);
    }

    public static Matrix3x3 Identity => new Matrix3x3(
        new Vector3d(1.0, 0.0, 0.0),
        new Vector3d(0.0, 1.0, 0.0),
        new Vector3d(0.0, 0.0, 1.0));
}

internal readonly struct Quaterniond
{
    public readonly double W;
    public readonly double X;
    public readonly double Y;
    public readonly double Z;

    public Quaterniond(double w, double x, double y, double z)
    {
        W = w;
        X = x;
        Y = y;
        Z = z;
    }

    public Quaterniond Normalize()
    {
        double norm = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
        if (norm <= 0.0)
        {
            return new Quaterniond(1.0, 0.0, 0.0, 0.0);
        }

        double inv = 1.0 / norm;
        return new Quaterniond(W * inv, X * inv, Y * inv, Z * inv);
    }

    public static Quaterniond FromMatrix(Matrix3x3 m)
    {
        double m00 = m.Column0.X;
        double m01 = m.Column1.X;
        double m02 = m.Column2.X;
        double m10 = m.Column0.Y;
        double m11 = m.Column1.Y;
        double m12 = m.Column2.Y;
        double m20 = m.Column0.Z;
        double m21 = m.Column1.Z;
        double m22 = m.Column2.Z;

        double trace = m00 + m11 + m22;
        double qw, qx, qy, qz;

        if (trace > 0.0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            if (s == 0.0) s = 1e-12;
            qw = 0.25 * s;
            qx = (m21 - m12) / s;
            qy = (m02 - m20) / s;
            qz = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0;
            if (s == 0.0) s = 1e-12;
            qw = (m21 - m12) / s;
            qx = 0.25 * s;
            qy = (m01 + m10) / s;
            qz = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0;
            if (s == 0.0) s = 1e-12;
            qw = (m02 - m20) / s;
            qx = (m01 + m10) / s;
            qy = 0.25 * s;
            qz = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0;
            if (s == 0.0) s = 1e-12;
            qw = (m10 - m01) / s;
            qx = (m02 + m20) / s;
            qy = (m12 + m21) / s;
            qz = 0.25 * s;
        }

        return new Quaterniond(qw, qx, qy, qz).Normalize();
    }

    public static Quaterniond Identity => new Quaterniond(1.0, 0.0, 0.0, 0.0);
}

internal struct Space
{
    public Vector3d Origin;
    public Vector3d Size;

    public double Volume => Size.X * Size.Y * Size.Z;

    public bool CanFit(Vector3d dims, double eps)
    {
        return dims.X <= Size.X + eps &&
               dims.Y <= Size.Y + eps &&
               dims.Z <= Size.Z + eps;
    }
}

internal sealed class OrientationOption
{
    public int Id { get; }
    public Matrix3x3 Matrix { get; }
    public Quaterniond Quaternion { get; }
    public Vector3d MinCorner { get; }
    public Vector3d MaxCorner { get; }
    public Vector3d Dimensions { get; }
    public double Volume { get; }

    public OrientationOption(int id, Matrix3x3 matrix, Quaterniond quaternion,
        Vector3d minCorner, Vector3d maxCorner, Vector3d dimensions)
    {
        Id = id;
        Matrix = matrix;
        Quaternion = quaternion;
        MinCorner = minCorner;
        MaxCorner = maxCorner;
        Dimensions = dimensions;
        Volume = dimensions.X * dimensions.Y * dimensions.Z;
    }
}

internal static class OrientationGenerator
{
    private const double MinDimension = 1e-6;

    public static List<OrientationOption> Generate(Vector3d min, Vector3d max)
    {
        var result = new List<OrientationOption>();
        var axes = new[]
        {
            new Vector3d(1, 0, 0),
            new Vector3d(-1, 0, 0),
            new Vector3d(0, 1, 0),
            new Vector3d(0, -1, 0),
            new Vector3d(0, 0, 1),
            new Vector3d(0, 0, -1)
        };

        var corners = BuildCorners(min, max);
        var signatures = new HashSet<string>();
        int id = 0;

        foreach (var dirX in axes)
        {
            foreach (var dirY in axes)
            {
                if (Math.Abs(Vector3d.Dot(dirX, dirY)) > 0.5)
                {
                    continue;
                }

                var dirZ = Vector3d.Cross(dirX, dirY);
                if (dirZ.LengthSquared <= 0.5)
                {
                    continue;
                }

                string signature = CreateSignature(dirX, dirY, dirZ);
                if (!signatures.Add(signature))
                {
                    continue;
                }

                var matrix = new Matrix3x3(dirX, dirY, dirZ);
                result.Add(CreateOption(id++, matrix, corners));
            }
        }

        return result;
    }

    public static OrientationOption CreateFallback()
    {
        var corners = BuildCorners(Vector3d.Zero, new Vector3d(1.0, 1.0, 1.0));
        return CreateOption(0, Matrix3x3.Identity, corners);
    }

    public static Vector3d ComputeAxisMinimums(List<OrientationOption> options)
    {
        double minX = double.MaxValue;
        double minY = double.MaxValue;
        double minZ = double.MaxValue;

        foreach (var option in options)
        {
            var dims = option.Dimensions;
            if (dims.X < minX)
            {
                minX = dims.X;
            }

            if (dims.Y < minY)
            {
                minY = dims.Y;
            }

            if (dims.Z < minZ)
            {
                minZ = dims.Z;
            }
        }

        if (double.IsPositiveInfinity(minX)) minX = MinDimension;
        if (double.IsPositiveInfinity(minY)) minY = MinDimension;
        if (double.IsPositiveInfinity(minZ)) minZ = MinDimension;

        return new Vector3d(minX, minY, minZ);
    }

    private static OrientationOption CreateOption(int id, Matrix3x3 matrix, IReadOnlyList<Vector3d> corners)
    {
        Vector3d minCorner = new Vector3d(double.MaxValue, double.MaxValue, double.MaxValue);
        Vector3d maxCorner = new Vector3d(double.MinValue, double.MinValue, double.MinValue);

        for (int i = 0; i < corners.Count; i++)
        {
            var rotated = matrix.Multiply(corners[i]);
            minCorner = Vector3d.Min(minCorner, rotated);
            maxCorner = Vector3d.Max(maxCorner, rotated);
        }

        Vector3d dims = maxCorner - minCorner;
        double dx = Math.Max(dims.X, MinDimension);
        double dy = Math.Max(dims.Y, MinDimension);
        double dz = Math.Max(dims.Z, MinDimension);
        var adjustedDims = new Vector3d(dx, dy, dz);
        var adjustedMax = minCorner + adjustedDims;

        var quaternion = Quaterniond.FromMatrix(matrix);
        return new OrientationOption(id, matrix, quaternion, minCorner, adjustedMax, adjustedDims);
    }

    private static List<Vector3d> BuildCorners(Vector3d min, Vector3d max)
    {
        var corners = new List<Vector3d>(8);
        for (int xi = 0; xi < 2; xi++)
        {
            double x = xi == 0 ? min.X : max.X;
            for (int yi = 0; yi < 2; yi++)
            {
                double y = yi == 0 ? min.Y : max.Y;
                for (int zi = 0; zi < 2; zi++)
                {
                    double z = zi == 0 ? min.Z : max.Z;
                    corners.Add(new Vector3d(x, y, z));
                }
            }
        }

        if (corners.Count == 0)
        {
            corners.Add(Vector3d.Zero);
        }

        return corners;
    }

    private static string CreateSignature(Vector3d x, Vector3d y, Vector3d z)
    {
        return FormatVector(x) + "|" + FormatVector(y) + "|" + FormatVector(z);
    }

    private static string FormatVector(Vector3d v)
    {
        return v.X.ToString(CultureInfo.InvariantCulture) + "," +
               v.Y.ToString(CultureInfo.InvariantCulture) + "," +
               v.Z.ToString(CultureInfo.InvariantCulture);
    }
}

internal sealed class SpaceManager
{
    private readonly List<Space> _spaces = new List<Space>();
    private readonly Scale _scale;
    private readonly Vector3d _axisMinimums;
    private readonly double _eps;

    public SpaceManager(Scale scale, Vector3d axisMinimums, double eps)
    {
        _scale = scale;
        _axisMinimums = axisMinimums;
        _eps = eps;
    }

    public int Count => _spaces.Count;

    public void Add(Space space)
    {
        Vector3d size = new Vector3d(
            Math.Max(0.0, space.Size.X),
            Math.Max(0.0, space.Size.Y),
            Math.Max(0.0, space.Size.Z));

        if (size.X <= _eps || size.Y <= _eps || size.Z <= _eps)
        {
            return;
        }

        if (size.X + _eps < _axisMinimums.X ||
            size.Y + _eps < _axisMinimums.Y ||
            size.Z + _eps < _axisMinimums.Z)
        {
            return;
        }

        _spaces.Add(new Space { Origin = space.Origin, Size = size });
    }

    public bool TryTake(out Space space)
    {
        if (_spaces.Count == 0)
        {
            space = default;
            return false;
        }

        int index = SelectIndex();
        space = _spaces[index];
        int last = _spaces.Count - 1;
        _spaces[index] = _spaces[last];
        _spaces.RemoveAt(last);
        return true;
    }

    private int SelectIndex()
    {
        return _scale switch
        {
            Scale.Trivial => SelectLowCoordinate(),
            Scale.Medium => SelectByScore(1.0, 0.15, 0.05),
            Scale.Large => SelectByScore(1.0, 0.05, 0.02),
            Scale.Extreme => SelectByScore(1.0, 0.0, 0.0),
            _ => _spaces.Count - 1
        };
    }

    private int SelectLowCoordinate()
    {
        int bestIndex = 0;
        double bestY = double.MaxValue;
        double bestZ = double.MaxValue;
        double bestX = double.MaxValue;
        double bestVolume = double.MinValue;

        for (int i = 0; i < _spaces.Count; i++)
        {
            var s = _spaces[i];
            if (IsLess(s.Origin.Y, bestY))
            {
                bestY = s.Origin.Y;
                bestZ = s.Origin.Z;
                bestX = s.Origin.X;
                bestVolume = s.Volume;
                bestIndex = i;
                continue;
            }

            if (!IsNearlyEqual(s.Origin.Y, bestY))
            {
                continue;
            }

            if (IsLess(s.Origin.Z, bestZ))
            {
                bestZ = s.Origin.Z;
                bestX = s.Origin.X;
                bestVolume = s.Volume;
                bestIndex = i;
                continue;
            }

            if (!IsNearlyEqual(s.Origin.Z, bestZ))
            {
                continue;
            }

            if (IsLess(s.Origin.X, bestX))
            {
                bestX = s.Origin.X;
                bestVolume = s.Volume;
                bestIndex = i;
                continue;
            }

            if (!IsNearlyEqual(s.Origin.X, bestX))
            {
                continue;
            }

            if (s.Volume > bestVolume)
            {
                bestVolume = s.Volume;
                bestIndex = i;
            }
        }

        return bestIndex;
    }

    private int SelectByScore(double volumeBias, double yPenalty, double zPenalty)
    {
        int bestIndex = 0;
        double bestScore = double.NegativeInfinity;

        for (int i = 0; i < _spaces.Count; i++)
        {
            var s = _spaces[i];
            double score = volumeBias * s.Volume - yPenalty * s.Origin.Y - zPenalty * s.Origin.Z;
            if (score > bestScore)
            {
                bestScore = score;
                bestIndex = i;
            }
        }

        return bestIndex;
    }

    private bool IsNearlyEqual(double a, double b)
    {
        return Math.Abs(a - b) <= _eps;
    }

    private bool IsLess(double a, double b)
    {
        return a < b - _eps;
    }

    public void Trim(int keepCount)
    {
        if (_spaces.Count <= keepCount)
        {
            return;
        }

        _spaces.Sort((a, b) => b.Volume.CompareTo(a.Volume));
        if (keepCount < 0)
        {
            keepCount = 0;
        }

        if (keepCount < _spaces.Count)
        {
            _spaces.RemoveRange(keepCount, _spaces.Count - keepCount);
        }
    }
}

internal static class SafeMath
{
    public static long Product(int a, int b, int c)
    {
        long result = Multiply(a, b);
        result = Multiply(result, c);
        return result;
    }

    private static long Multiply(long value, int factor)
    {
        if (value == 0 || factor == 0)
        {
            return 0;
        }

        if (value > long.MaxValue / factor)
        {
            return long.MaxValue;
        }

        return value * factor;
    }
}

internal sealed class PackingSolver
{
    private readonly List<OrientationOption> _orientations;
    private readonly SpaceManager _spaceManager;
    private readonly Scale _scale;
    private readonly List<Placement> _placements = new List<Placement>();
    private readonly double _eps = 1e-9;
    private const double TimeLimitSeconds = 295.0;

    public PackingSolver(Space container, List<OrientationOption> orientations, Scale scale, Vector3d axisMinimums)
    {
        _orientations = orientations;
        _scale = scale;
        _spaceManager = new SpaceManager(scale, axisMinimums, _eps);
        _spaceManager.Add(container);
    }

    public List<Placement> Execute()
    {
        var stopwatch = Stopwatch.StartNew();
        while (_spaceManager.TryTake(out var space))
        {
            if (stopwatch.Elapsed.TotalSeconds >= TimeLimitSeconds)
            {
                break;
            }

            TryFillSpace(space);
            MaybeTrim();
        }

        stopwatch.Stop();
        return _placements;
    }

    private void MaybeTrim()
    {
        int threshold;
        int target;

        switch (_scale)
        {
            case Scale.Trivial:
                threshold = 512;
                target = 256;
                break;
            case Scale.Medium:
                threshold = 2048;
                target = 1024;
                break;
            case Scale.Large:
                threshold = 4096;
                target = 2048;
                break;
            case Scale.Extreme:
                threshold = 8192;
                target = 4096;
                break;
            default:
                threshold = 2048;
                target = 1024;
                break;
        }

        if (_spaceManager.Count > threshold)
        {
            _spaceManager.Trim(target);
        }
    }

    private void TryFillSpace(Space space)
    {
        var plan = SelectOrientation(space);
        if (!plan.HasValue)
        {
            return;
        }

        var blockPlan = BuildBlockPlan(space, plan.Value);
        ExecuteBlock(space, plan.Value.Orientation, blockPlan);
    }

    private PlacementPlan? SelectOrientation(Space space)
    {
        if (space.Volume <= _eps)
        {
            return null;
        }

        OrientationOption? best = null;
        FitInfo bestFit = default;
        double bestScore = double.NegativeInfinity;
        double spaceVolume = space.Volume;

        foreach (var orientation in _orientations)
        {
            if (!space.CanFit(orientation.Dimensions, _eps))
            {
                continue;
            }

            FitInfo fit = ComputeFitInfo(space.Size, orientation.Dimensions);
            if (!fit.IsValid)
            {
                continue;
            }

            double fillRatio = orientation.Volume / spaceVolume;
            double score = ScoreOrientation(fillRatio, fit.TotalCount, space, orientation);
            if (score > bestScore)
            {
                bestScore = score;
                best = orientation;
                bestFit = fit;
            }
        }

        if (best == null)
        {
            return null;
        }

        return new PlacementPlan(best, bestFit);
    }

    private FitInfo ComputeFitInfo(Vector3d spaceSize, Vector3d dims)
    {
        int fitX = FitCount(spaceSize.X, dims.X);
        int fitY = FitCount(spaceSize.Y, dims.Y);
        int fitZ = FitCount(spaceSize.Z, dims.Z);
        return new FitInfo(fitX, fitY, fitZ);
    }

    private int FitCount(double available, double required)
    {
        if (required <= _eps)
        {
            return 0;
        }

        double space = available + _eps;
        if (space + _eps < required)
        {
            return 0;
        }

        double ratio = space / required;
        if (ratio > int.MaxValue)
        {
            return int.MaxValue;
        }

        int count = (int)Math.Floor(ratio);
        if (count < 1)
        {
            count = 1;
        }

        return count;
    }

    private double ScoreOrientation(double fillRatio, long potentialCount, Space space, OrientationOption orientation)
    {
        double cappedPotential = Math.Log(1.0 + Math.Min(potentialCount, 1_000_000_000_000L));
        return _scale switch
        {
            Scale.Trivial => fillRatio * 256.0 + cappedPotential * 8.0 - (orientation.Dimensions.Y + orientation.Dimensions.Z) * 0.5,
            Scale.Medium => fillRatio * 128.0 + cappedPotential * 16.0 - orientation.Dimensions.Y * 0.1 - space.Origin.Y * 0.01,
            Scale.Large => fillRatio * 96.0 + cappedPotential * 24.0 - space.Origin.Y * 0.02,
            Scale.Extreme => fillRatio * 64.0 + cappedPotential * 32.0,
            _ => fillRatio * 64.0 + cappedPotential * 16.0
        };
    }

    private BlockPlan BuildBlockPlan(Space space, PlacementPlan plan)
    {
        var dims = plan.Orientation.Dimensions;
        var fits = plan.Fits;
        int clamp = AxisClampForScale();

        int countX = Math.Max(1, Math.Min(fits.CountX, clamp));
        int countY = Math.Max(1, Math.Min(fits.CountY, clamp));
        int countZ = Math.Max(1, Math.Min(fits.CountZ, clamp));

        double usedX = Math.Min(space.Size.X, countX * dims.X);
        double usedY = Math.Min(space.Size.Y, countY * dims.Y);
        double usedZ = Math.Min(space.Size.Z, countZ * dims.Z);

        return new BlockPlan(countX, countY, countZ, new Vector3d(usedX, usedY, usedZ));
    }

    private int AxisClampForScale()
    {
        return _scale switch
        {
            Scale.Trivial => 4,
            Scale.Medium => 32,
            Scale.Large => 64,
            Scale.Extreme => 96,
            _ => 32
        };
    }

    private void ExecuteBlock(Space space, OrientationOption orientation, BlockPlan plan)
    {
        var dims = orientation.Dimensions;
        var origin = space.Origin;

        for (int z = 0; z < plan.CountZ; z++)
        {
            double baseZ = origin.Z + z * dims.Z;
            for (int y = 0; y < plan.CountY; y++)
            {
                double baseY = origin.Y + y * dims.Y;
                for (int x = 0; x < plan.CountX; x++)
                {
                    double baseX = origin.X + x * dims.X;
                    var placementOrigin = new Vector3d(baseX, baseY, baseZ);
                    RegisterPlacement(placementOrigin, orientation);
                }
            }
        }

        AddRemainderSpaces(space, plan.UsedSize);
    }

    private void RegisterPlacement(Vector3d boundingMin, OrientationOption orientation)
    {
        Vector3d translation = boundingMin - orientation.MinCorner;
        var placement = new Placement
        {
            Translation = translation,
            Rotation = orientation.Quaternion
        };
        _placements.Add(placement);
    }

    private void AddRemainderSpaces(Space parent, Vector3d used)
    {
        double remainingX = parent.Size.X - used.X;
        double remainingY = parent.Size.Y - used.Y;
        double remainingZ = parent.Size.Z - used.Z;

        if (remainingX > _eps)
        {
            var space = new Space
            {
                Origin = parent.Origin + new Vector3d(used.X, 0.0, 0.0),
                Size = new Vector3d(remainingX, parent.Size.Y, parent.Size.Z)
            };
            _spaceManager.Add(space);
        }

        if (remainingY > _eps)
        {
            var space = new Space
            {
                Origin = parent.Origin + new Vector3d(0.0, used.Y, 0.0),
                Size = new Vector3d(Math.Max(used.X, _eps), remainingY, parent.Size.Z)
            };
            _spaceManager.Add(space);
        }

        if (remainingZ > _eps)
        {
            var space = new Space
            {
                Origin = parent.Origin + new Vector3d(0.0, 0.0, used.Z),
                Size = new Vector3d(Math.Max(used.X, _eps), Math.Max(used.Y, _eps), remainingZ)
            };
            _spaceManager.Add(space);
        }
    }
```
