Here is a Python implementation that reads the problem input, runs a deterministic Algorithm X-style backtracking search with bitmask pruning, and prints either the planted solution or “NO SOLUTION”:

```python
import sys


def read_input():
    lines = sys.stdin.read().splitlines()
    idx = 0
    n = len(lines)
    while idx < n and not lines[idx].strip():
        idx += 1
    if idx >= n:
        return 0, 0, []

    first_line = lines[idx].replace('\ufeff', '')
    first_tokens = first_line.split()
    idx += 1
    if not first_tokens:
        return 0, 0, []

    try:
        U = int(first_tokens[0])
    except ValueError:
        U = 0
    if len(first_tokens) >= 2:
        try:
            S = int(first_tokens[1])
        except ValueError:
            S = 0
    else:
        S = 0
    U = max(0, U)
    S = max(0, S)

    sets_data = []
    for _ in range(S):
        if idx < n:
            line = lines[idx].replace('\ufeff', '')
            idx += 1
        else:
            line = ''
        stripped = line.strip()
        if stripped:
            tokens = stripped.split()
            seen = set()
            elems = []
            for tok in tokens:
                try:
                    val = int(tok)
                except ValueError:
                    continue
                if 0 <= val < U and val not in seen:
                    seen.add(val)
                    elems.append(val)
        else:
            elems = []
        elems.sort()
        sets_data.append(tuple(elems))
    return U, S, sets_data


def main():
    U, S, sets_data = read_input()
    sys.setrecursionlimit(min(1_000_000, max(10000, U + len(sets_data) + 100)))

    if U == 0:
        print("SOLUTION")
        print()
        return

    element_to_sets = [[] for _ in range(U)]
    for idx, elems in enumerate(sets_data):
        for e in elems:
            element_to_sets[e].append(idx)

    for e in range(U):
        if not element_to_sets[e]:
            print("NO SOLUTION")
            return

    set_masks = []
    for elems in sets_data:
        mask = 0
        for val in elems:
            mask |= 1 << val
        set_masks.append(mask)

    uncovered_mask = (1 << U) - 1
    chosen = []
    solution = None

    def choose_element():
        best_elem = None
        best_candidates = None
        best_len = None
        bits = uncovered_mask
        while bits:
            lsb = bits & -bits
            elem = lsb.bit_length() - 1
            bits ^= lsb
            candidates = []
            for idx in element_to_sets[elem]:
                if set_masks[idx] & uncovered_mask == set_masks[idx]:
                    candidates.append(idx)
            if not candidates:
                return elem, []
            cand_len = len(candidates)
            if (best_elem is None or cand_len < best_len or
                    (cand_len == best_len and elem < best_elem)):
                best_elem = elem
                best_candidates = candidates
                best_len = cand_len
        return best_elem, best_candidates if best_candidates is not None else []

    def backtrack():
        nonlocal uncovered_mask, solution
        if uncovered_mask == 0:
            solution = list(chosen)
            return True
        elem, candidates = choose_element()
        if not candidates:
            return False
        for idx in candidates:
            mask = set_masks[idx]
            if mask & uncovered_mask != mask:
                continue
            chosen.append(idx)
            uncovered_mask ^= mask
            if backtrack():
                return True
            uncovered_mask ^= mask
            chosen.pop()
        return False

    if backtrack() and solution is not None:
        solution.sort()
        print("SOLUTION")
        if solution:
            print(' '.join(map(str, solution)))
        else:
            print()
    else:
        print("NO SOLUTION")


if __name__ == "__main__":
    main()
```
