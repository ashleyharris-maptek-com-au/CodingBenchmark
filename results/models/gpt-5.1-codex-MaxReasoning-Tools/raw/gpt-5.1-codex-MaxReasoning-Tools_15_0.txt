impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
    fn dot(self, rhs: Vec3) -> f64 {
        self.x * rhs.x + self.y * rhs.y + self.z * rhs.z
    }
    fn cross(self, rhs: Vec3) -> Vec3 {
        Vec3::new(
            self.y * rhs.z - self.z * rhs.y,
            self.z * rhs.x - self.x * rhs.z,
            self.x * rhs.y - self.y * rhs.x,
        )
    }
    fn length(self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }
    fn length_squared(self) -> f64 {
        self.x * self.x + self.y * self.y + self.z * self.z
    }
    fn normalize(self) -> Vec3 {
        let len = self.length();
        if len < 1e-9 {
            Self::new(0.0, 0.0, 0.0)
        } else {
            self / len
        }
    }
}

#[derive(Clone, Copy)]
struct Mat3 {
    rows: [Vec3; 3],
}
impl Mat3 {
    fn mul_vec(&self, v: Vec3) -> Vec3 {
        Vec3::new(
            self.rows[0].dot(v),
            self.rows[1].dot(v),
            self.rows[2].dot(v),
        )
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}
impl Quaternion {
    fn from_matrix(m: &Mat3) -> Self {
        let m00 = m.rows[0].x;
        let m11 = m.rows[1].y;
        let m22 = m.rows[2].z;
        let trace = m00 + m11 + m22;
        let (w, x, y, z) = if trace > 0.0 {
            let s = (trace + 1.0).sqrt() * 2.0;
            (
                0.25 * s,
                (m.rows[2].y - m.rows[1].z) / s,
                (m.rows[0].z - m.rows[2].x) / s,
                (m.rows[1].x - m.rows[0].y) / s,
            )
        } else if m00 > m11 && m00 > m22 {
            let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
            (
                (m.rows[2].y - m.rows[1].z) / s,
                0.25 * s,
                (m.rows[0].y + m.rows[1].x) / s,
                (m.rows[0].z + m.rows[2].x) / s,
            )
        } else if m11 > m22 {
            let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
            (
                (m.rows[0].z - m.rows[2].x) / s,
                (m.rows[0].y + m.rows[1].x) / s,
                0.25 * s,
                (m.rows[1].z + m.rows[2].y) / s,
            )
        } else {
            let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
            (
                (m.rows[1].x - m.rows[0].y) / s,
                (m.rows[0].z + m.rows[2].x) / s,
                (m.rows[1].z + m.rows[2].y) / s,
                0.25 * s,
            )
        };
        Quaternion { w, x, y, z }.normalized()
    }
    fn from_yaw(angle: f64) -> Self {
        let half = angle * 0.5;
        Quaternion {
            w: half.cos(),
            x: 0.0,
            y: 0.0,
            z: half.sin(),
        }
    }
    fn normalized(self) -> Self {
        let len = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if len < 1e-9 {
            Self {
                w: 1.0,
                x: 0.0,
                y: 0.0,
                z: 0.0,
            }
        } else {
            Self {
                w: self.w / len,
                x: self.x / len,
                y: self.y / len,
                z: self.z / len,
            }
        }
    }
}
impl Mul<Quaternion> for Quaternion {
    type Output = Quaternion;
    fn mul(self, rhs: Quaternion) -> Quaternion {
        Quaternion {
            w: self.w * rhs.w - self.x * rhs.x - self.y * rhs.y - self.z * rhs.z,
            x: self.w * rhs.x + self.x * rhs.w + self.y * rhs.z - self.z * rhs.y,
            y: self.w * rhs.y - self.x * rhs.z + self.y * rhs.w + self.z * rhs.x,
            z: self.w * rhs.z + self.x * rhs.y - self.y * rhs.x + self.z * rhs.w,
        }
    }
}

const REF_VERTICES: [Vec3; 4] = [
    Vec3 {
        x: 1.0,
        y: 0.0,
        z: -0.707,
    },
    Vec3 {
        x: -1.0,
        y: 0.0,
        z: -0.707,
    },
    Vec3 {
        x: 0.0,
        y: 1.0,
        z: 0.707,
    },
    Vec3 {
        x: 0.0,
        y: -1.0,
        z: 0.707,
    },
];

struct BaseFrame {
    quat: Quaternion,
    face_centroid: Vec3,
    min_z: f64,
}
fn build_base_frame() -> BaseFrame {
    let a = REF_VERTICES[0];
    let b = REF_VERTICES[1];
    let c = REF_VERTICES[2];

    let u = (a - b).normalize();
    let n = (c - a).cross(b - a).normalize();
    let v = n.cross(u);

    let rot = Mat3 {
        rows: [u, v, n],
    };
    let quat = Quaternion::from_matrix(&rot);

    let mut rotated = [Vec3::default(); 4];
    for (i, rv) in rotated.iter_mut().enumerate() {
        *rv = rot.mul_vec(REF_VERTICES[i]);
    }

    let face_centroid = (rotated[0] + rotated[1] + rotated[2]) / 3.0;
    let mut min_z = f64::INFINITY;
    for p in rotated.iter() {
        if p.z < min_z {
            min_z = p.z;
        }
    }
    BaseFrame {
        quat,
        face_centroid,
        min_z,
    }
}

#[derive(Clone, Copy)]
struct EdgeRecord {
    y_min: f64,
    y_max: f64,
    x_at_y_min: f64,
    inv_slope: f64,
}
fn build_edge_records(poly: &[Vec2]) -> Vec<EdgeRecord> {
    let mut edges = Vec::new();
    if poly.len() < 2 {
        return edges;
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let dy = b.y - a.y;
        if dy.abs() < 1e-9 {
            continue;
        }
        if a.y < b.y {
            edges.push(EdgeRecord {
                y_min: a.y,
                y_max: b.y,
                x_at_y_min: a.x,
                inv_slope: (b.x - a.x) / (b.y - a.y),
            });
        } else {
            edges.push(EdgeRecord {
                y_min: b.y,
                y_max: a.y,
                x_at_y_min: b.x,
                inv_slope: (a.x - b.x) / (a.y - b.y),
            });
        }
    }
    edges
}
fn scanline_intervals(
    y: f64,
    edges: &[EdgeRecord],
    xs: &mut Vec<f64>,
    intervals: &mut Vec<(f64, f64)>,
) {
    xs.clear();
    for e in edges {
        if y >= e.y_min && y < e.y_max {
            xs.push(e.x_at_y_min + (y - e.y_min) * e.inv_slope);
        }
    }
    xs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    intervals.clear();
    let mut i = 0;
    while i + 1 < xs.len() {
        let x0 = xs[i];
        let x1 = xs[i + 1];
        intervals.push((x0.min(x1), x0.max(x1)));
        i += 2;
    }
}
fn expand_and_merge_intervals(
    src: &[(f64, f64)],
    pad: f64,
    dst: &mut Vec<(f64, f64)>,
) {
    dst.clear();
    if src.is_empty() {
        return;
    }
    let mut current = (src[0].0 - pad, src[0].1 + pad);
    for &(s, e) in src.iter().skip(1) {
        let start = s - pad;
        let end = e + pad;
        if start <= current.1 {
            if end > current.1 {
                current.1 = end;
            }
        } else {
            dst.push(current);
            current = (start, end);
        }
    }
    dst.push(current);
}
fn point_within_intervals(x: f64, intervals: &[(f64, f64)], margin: f64) -> bool {
    for &(s, e) in intervals {
        if x >= s - margin && x <= e + margin {
            return true;
        }
    }
    false
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    if poly.len() < 2 {
        return 0.0;
    }
    let mut acc = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        acc += a.x * b.y - b.x * a.y;
    }
    acc * 0.5
}
fn polygon_perimeter(poly: &[Vec2]) -> f64 {
    if poly.len() < 2 {
        return 0.0;
    }
    let mut perim = 0.0;
    for i in 0..poly.len() {
        perim += (poly[i] - poly[(i + 1) % poly.len()]).length();
    }
    perim
}
fn polygon_centroid(poly: &[Vec2]) -> Vec2 {
    let area = polygon_area(poly);
    if area.abs() < 1e-9 {
        let mut sum = Vec2::new(0.0, 0.0);
        for p in poly {
            sum = sum + *p;
        }
        if poly.is_empty() {
            Vec2::new(0.0, 0.0)
        } else {
            sum / poly.len() as f64
        }
    } else {
        let mut cx = 0.0;
        let mut cy = 0.0;
        for i in 0..poly.len() {
            let a = poly[i];
            let b = poly[(i + 1) % poly.len()];
            let cross = a.x * b.y - b.x * a.y;
            cx += (a.x + b.x) * cross;
            cy += (a.y + b.y) * cross;
        }
        let factor = 1.0 / (6.0 * area);
        Vec2::new(cx * factor, cy * factor)
    }
}
fn principal_angle(poly: &[Vec2]) -> f64 {
    if poly.is_empty() {
        return 0.0;
    }
    let centroid = polygon_centroid(poly);
    let mut cxx = 0.0;
    let mut cxy = 0.0;
    let mut cyy = 0.0;
    for p in poly {
        let dx = p.x - centroid.x;
        let dy = p.y - centroid.y;
        cxx += dx * dx;
        cxy += dx * dy;
        cyy += dy * dy;
    }
    if (cxx + cyy).abs() < 1e-9 {
        0.0
    } else {
        0.5 * (2.0 * cxy).atan2(cxx - cyy)
    }
}
fn rotate_world_to_local(p: Vec2, cos_a: f64, sin_a: f64) -> Vec2 {
    Vec2::new(p.x * cos_a + p.y * sin_a, -p.x * sin_a + p.y * cos_a)
}
fn rotate_local_to_world(p: Vec2, cos_a: f64, sin_a: f64) -> Vec2 {
    Vec2::new(p.x * cos_a - p.y * sin_a, p.x * sin_a + p.y * cos_a)
}

#[derive(Clone, Copy)]
enum Strategy {
    Dense,
    Balanced,
    Sweep,
}
fn choose_strategy(diag: f64, area: f64, nv: usize) -> Strategy {
    if diag < 120.0 && area < 50_000.0 && nv < 512 {
        Strategy::Dense
    } else if diag < 1500.0 || nv < 5000 {
        Strategy::Balanced
    } else {
        Strategy::Sweep
    }
}

struct Tetra {
    center: Vec3,
    quat: Quaternion,
    scale: f64,
}

struct TetraEmitter {
    data: Vec<Tetra>,
    base_quat: Quaternion,
    scale: f64,
    face_centroid_scaled: Vec3,
    base_height: f64,
    sep_dir: Vec3,
    clearance: f64,
    frame_cos: f64,
    frame_sin: f64,
}
impl TetraEmitter {
    fn new(
        capacity: usize,
        base_quat: Quaternion,
        scale: f64,
        face_centroid_scaled: Vec3,
        base_height: f64,
        sep_dir: Vec3,
        clearance: f64,
        frame_cos: f64,
        frame_sin: f64,
    ) -> Self {
        Self {
            data: Vec::with_capacity(capacity),
            base_quat,
            scale,
            face_centroid_scaled,
            base_height,
            sep_dir,
            clearance,
            frame_cos,
            frame_sin,
        }
    }
    fn emit_triangle(&mut self, a_local: Vec2, b_local: Vec2, c_local: Vec2) {
        let centroid_local = (a_local + b_local + c_local) / 3.0;
        let centroid_world = rotate_local_to_world(centroid_local, self.frame_cos, self.frame_sin);

        let edge_local = b_local - a_local;
        let edge_world = rotate_local_to_world(edge_local, self.frame_cos, self.frame_sin);
        if edge_world.length_squared() < 1e-12 {
            return;
        }
        let theta = edge_world.y.atan2(edge_world.x);
        let q_z = Quaternion::from_yaw(theta);
        let mut quat = q_z * self.base_quat;
        quat = quat.normalized();

        let (s, c) = theta.sin_cos();
        let rotated_face = Vec3::new(
            self.face_centroid_scaled.x * c - self.face_centroid_scaled.y * s,
            self.face_centroid_scaled.x * s + self.face_centroid_scaled.y * c,
            self.face_centroid_scaled.z,
        );

        let mut center = Vec3::new(
            centroid_world.x - rotated_face.x,
            centroid_world.y - rotated_face.y,
            self.base_height - rotated_face.z,
        );
        if self.clearance > 0.0 {
            let offset = self.sep_dir * (self.data.len() as f64 * self.clearance);
            center = center + offset;
        }

        self.data.push(Tetra {
            center,
            quat,
            scale: self.scale,
        });
    }
}

fn lattice_point(i: i32, j: i32, side: f64, height: f64, origin_x: f64, origin_y: f64) -> Vec2 {
    Vec2::new(
        origin_x + (i as f64) * side + (j as f64) * (side * 0.5),
        origin_y + (j as f64) * height,
    )
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let n: usize = match iter.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => 0,
    };
    let mut polygon = Vec::with_capacity(n);
    for _ in 0..n {
        let x: f64 = iter.next().unwrap().parse().unwrap();
        let y: f64 = iter.next().unwrap().parse().unwrap();
        polygon.push(Vec2::new(x, y));
    }
    let sx: f64 = iter.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sy: f64 = iter.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sz: f64 = iter.next().unwrap_or("-1").parse().unwrap_or(-1.0);

    if polygon.len() < 3 {
        println!("1");
        println!("0 0 5 1 0 0 0 1");
        return;
    }

    let base_frame = build_base_frame();

    let min_x = polygon.iter().map(|p| p.x).fold(f64::INFINITY, f64::min);
    let max_x = polygon.iter().map(|p| p.x).fold(f64::NEG_INFINITY, f64::max);
    let min_y = polygon.iter().map(|p| p.y).fold(f64::INFINITY, f64::min);
    let max_y = polygon.iter().map(|p| p.y).fold(f64::NEG_INFINITY, f64::max);
    let width = (max_x - min_x).max(1e-6);
    let height = (max_y - min_y).max(1e-6);
    let diag = (width * width + height * height).sqrt();
    let area = polygon_area(&polygon).abs().max(1e-6);
    let perimeter = polygon_perimeter(&polygon).max(width + height);

    let strategy = choose_strategy(diag, area, polygon.len());
    let raw_angle = principal_angle(&polygon);
    let angle = match strategy {
        Strategy::Dense | Strategy::Balanced => raw_angle,
        Strategy::Sweep => {
            if diag > 5000.0 {
                0.0
            } else {
                raw_angle
            }
        }
    };
    let cos_a = angle.cos();
    let sin_a = angle.sin();

    let mut poly_local = Vec::with_capacity(polygon.len());
    for &p in &polygon {
        poly_local.push(rotate_world_to_local(p, cos_a, sin_a));
    }

    let min_local_x = poly_local.iter().map(|p| p.x).fold(f64::INFINITY, f64::min);
    let max_local_x = poly_local.iter().map(|p| p.x).fold(f64::NEG_INFINITY, f64::max);
    let min_local_y = poly_local.iter().map(|p| p.y).fold(f64::INFINITY, f64::min);
    let max_local_y = poly_local.iter().map(|p| p.y).fold(f64::NEG_INFINITY, f64::max);
    let local_width = (max_local_x - min_local_x).max(1e-6);
    let local_height = (max_local_y - min_local_y).max(1e-6);

    let mut side = (local_width.max(local_height) / match strategy {
        Strategy::Dense => 10.0,
        Strategy::Balanced => 14.0,
        Strategy::Sweep => 18.0,
    })
    .clamp(MIN_SIDE, 20.0);
    let mean_edge = (perimeter / polygon.len() as f64).clamp(MIN_SIDE, 20.0);
    side = side.min(mean_edge * 1.2).clamp(MIN_SIDE, 20.0);

    let budget = match strategy {
        Strategy::Dense => 40_000.0,
        Strategy::Balanced => 120_000.0,
        Strategy::Sweep => 250_000.0,
    };

    let pad_factor = match strategy {
        Strategy::Dense => 0.4,
        Strategy::Balanced => 0.8,
        Strategy::Sweep => 1.2,
    };

    let mut cover_min_x;
    let mut cover_max_x;
    let mut cover_min_y;
    let mut cover_max_y;
    let mut tri_area;
    loop {
        let tri_height = side * SQRT_3_OVER_2;
        let pad_x = side * pad_factor + 1e-6;
        let pad_y = tri_height * pad_factor + 1e-6;
        cover_min_x = min_local_x - pad_x;
        cover_max_x = max_local_x + pad_x;
        cover_min_y = min_local_y - pad_y;
        cover_max_y = max_local_y + pad_y;
        let rect_area = ((cover_max_x - cover_min_x).max(1e-6))
            * ((cover_max_y - cover_min_y).max(1e-6));
        tri_area = TRI_AREA_FACTOR * side * side;
        let approx_triangles = rect_area / tri_area;
        if approx_triangles <= budget || side >= 20.0 {
            break;
        }
        let factor = (approx_triangles / budget).sqrt();
        let new_side = (side * factor).min(20.0);
        if (new_side - side).abs() < 1e-6 {
            break;
        }
        side = new_side;
    }

    let tri_height = side * SQRT_3_OVER_2;
    let origin_x = cover_min_x;
    let origin_y = cover_min_y;

    let scale = (side / 2.0).min(10.0);
    let face_centroid_scaled = base_frame.face_centroid * scale;
    let min_rel = base_frame.min_z - base_frame.face_centroid.z;
    let safe_ground = 0.4;
    let base_height = (safe_ground - scale * min_rel).max(0.8);

    let mut sep_dir = Vec3::new(-sx, -sy, -sz);
    if sz >= 0.0 || sep_dir.length_squared() < 1e-9 {
        sep_dir = Vec3::new(0.0, 0.0, 1.0);
    }
    sep_dir = sep_dir.normalize();
    let clearance = REF_CIRCUMRADIUS * scale * 2.4 + 0.25;

    let rect_area = ((cover_max_x - cover_min_x).max(1e-6))
        * ((cover_max_y - cover_min_y).max(1e-6));
    let approx_triangles = (rect_area / tri_area).ceil() as usize;
    let reserve = approx_triangles.min(1_000_000);

    let mut emitter = TetraEmitter::new(
        reserve,
        base_frame.quat,
        scale,
        face_centroid_scaled,
        base_height,
        sep_dir,
        clearance,
        cos_a,
        sin_a,
    );

    let cover_margin_x = side;
    let cover_margin_y = tri_height;

    match strategy {
        Strategy::Dense | Strategy::Balanced => {
            let j_end = ((cover_max_y - origin_y) / tri_height).ceil() as i32 + 4;
            for j in -2..j_end {
                let y_low = origin_y + j as f64 * tri_height;
                if y_low > cover_max_y + tri_height {
                    break;
                }
                let base_x_j = origin_x + j as f64 * side * 0.5;
                let base_x_next = origin_x + (j + 1) as f64 * side * 0.5;
                let min_base = base_x_j.min(base_x_next);
                let max_base = base_x_j.max(base_x_next);
                let i_min =
                    ((cover_min_x - min_base) / side).floor() as i32 - 2;
                let i_max =
                    ((cover_max_x - max_base) / side).ceil() as i32 + 2;
                if i_min > i_max {
                    continue;
                }
                for i in i_min..=i_max {
                    let a = lattice_point(i, j, side, tri_height, origin_x, origin_y);
                    let b = lattice_point(i + 1, j, side, tri_height, origin_x, origin_y);
                    let c = lattice_point(i, j + 1, side, tri_height, origin_x, origin_y);
                    let centroid = (a + b + c) / 3.0;
                    if centroid.x < cover_min_x - cover_margin_x
                        || centroid.x > cover_max_x + cover_margin_x
                        || centroid.y < cover_min_y - cover_margin_y
                        || centroid.y > cover_max_y + cover_margin_y
                    {
                        continue;
                    }
                    emitter.emit_triangle(a, b, c);

                    let d = lattice_point(i + 1, j + 1, side, tri_height, origin_x, origin_y);
                    let centroid2 = (d + c + b) / 3.0;
                    if centroid2.x < cover_min_x - cover_margin_x
                        || centroid2.x > cover_max_x + cover_margin_x
                        || centroid2.y < cover_min_y - cover_margin_y
                        || centroid2.y > cover_max_y + cover_margin_y
                    {
                        continue;
                    }
                    emitter.emit_triangle(d, c, b);
                }
            }
        }
        Strategy::Sweep => {
            let edges = build_edge_records(&poly_local);
            let mut xs_buf = Vec::new();
            let mut raw_intervals = Vec::new();
            let mut merged_intervals = Vec::new();
            let j_end = ((cover_max_y - origin_y) / tri_height).ceil() as i32 + 4;

            for j in -2..j_end {
                let y_low = origin_y + j as f64 * tri_height;
                if y_low > cover_max_y + tri_height {
                    break;
                }
                let y_mid = y_low + 0.5 * tri_height;
                raw_intervals.clear();
                scanline_intervals(y_mid, &edges, &mut xs_buf, &mut raw_intervals);
                if raw_intervals.is_empty() {
                    scanline_intervals(y_mid + 0.25 * tri_height, &edges, &mut xs_buf, &mut raw_intervals);
                }
                if raw_intervals.is_empty() {
                    scanline_intervals(y_mid - 0.25 * tri_height, &edges, &mut xs_buf, &mut raw_intervals);
                }
                if raw_intervals.is_empty() {
                    continue;
                }
                expand_and_merge_intervals(&raw_intervals, cover_margin_x, &mut merged_intervals);
                if merged_intervals.is_empty() {
                    continue;
                }

                let first_start = merged_intervals.first().unwrap().0;
                let last_end = merged_intervals.last().unwrap().1;

                let base_x_j = origin_x + j as f64 * side * 0.5;
                let base_x_next = origin_x + (j + 1) as f64 * side * 0.5;
                let min_base = base_x_j.min(base_x_next);
                let max_base = base_x_j.max(base_x_next);
                let i_min =
                    ((first_start - min_base) / side).floor() as i32 - 2;
                let i_max =
                    ((last_end - max_base) / side).ceil() as i32 + 2;
                if i_min > i_max {
                    continue;
                }

                for i in i_min..=i_max {
                    let a = lattice_point(i, j, side, tri_height, origin_x, origin_y);
                    let b = lattice_point(i + 1, j, side, tri_height, origin_x, origin_y);
                    let c = lattice_point(i, j + 1, side, tri_height, origin_x, origin_y);
                    let centroid = (a + b + c) / 3.0;
                    if !point_within_intervals(
                        centroid.x,
                        &merged_intervals,
                        side * 0.5,
                    ) {
                        continue;
                    }
                    emitter.emit_triangle(a, b, c);

                    let d = lattice_point(i + 1, j + 1, side, tri_height, origin_x, origin_y);
                    let centroid2 = (d + c + b) / 3.0;
                    if !point_within_intervals(
                        centroid2.x,
                        &merged_intervals,
                        side * 0.5,
                    ) {
                        continue;
                    }
                    emitter.emit_triangle(d, c, b);
                }
            }
        }
    }

    if emitter.data.is_empty() {
        let tri_center = Vec2::new(
            (cover_min_x + cover_max_x) * 0.5,
            (cover_min_y + cover_max_y) * 0.5,
        );
        let half = side * 0.5;
        let h = tri_height;
        let a = Vec2::new(tri_center.x - half, tri_center.y - h / 3.0);
        let b = Vec2::new(tri_center.x + half, tri_center.y - h / 3.0);
        let c = Vec2::new(tri_center.x, tri_center.y + 2.0 * h / 3.0);
        emitter.emit_triangle(a, b, c);
    }

    println!("{}", emitter.data.len());
    for t in &emitter.data {
        println!(
            "{:.8} {:.8} {:.8} {:.8} {:.8} {:.8} {:.8} {:.8}",
            t.center.x,
            t.center.y,
            t.center.z,
            t.quat.w,
            t.quat.x,
            t.quat.y,
            t.quat.z,
            t.scale
        );
    }
}
