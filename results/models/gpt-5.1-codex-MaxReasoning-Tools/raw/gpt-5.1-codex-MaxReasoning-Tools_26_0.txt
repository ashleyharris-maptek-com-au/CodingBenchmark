#![allow(clippy::needless_range_loop)]

use std::cmp::Ordering;
use std::env;
use std::fs::{remove_file, File};
use std::io::{self, BufRead, BufReader, BufWriter, ErrorKind, Read, Write};
use std::path::PathBuf;
use std::process;
use std::time::{SystemTime, UNIX_EPOCH};

#[cfg(target_endian = "big")]
compile_error!("This solver expects a little-endian target so it can stream binary f32 data efficiently.");

const INPUT_BUF_BYTES: usize = 16 * 1024 * 1024;
const OUTPUT_BUF_BYTES: usize = 4 * 1024 * 1024;
const INGEST_CHUNK_POINTS: usize = 1 << 15; // 32K points (~1.5 MB)
const DISK_CHUNK_POINTS: usize = 1 << 15;
const STREAM_CHUNK_POINTS_LARGE: usize = 1 << 16; // 65K points
const STREAM_CHUNK_POINTS_EXTREME: usize = 1 << 18; // 262K points
const TREE_THRESHOLD: usize = 32;

const FULL_BATCH_POINT_LIMIT: u64 = 15_000_000;
const MINI_BATCH_POINT_LIMIT: u64 = 150_000_000;

const POINT_BYTES: usize = std::mem::size_of::<Point3>();
const _: [u8; POINT_BYTES] = [0; 12];
const _: [u8; 12] = [0; POINT_BYTES];

#[repr(C)]
#[derive(Clone, Copy, Default)]
struct Point3 {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3 {
    #[inline]
    fn to_f64(self) -> [f64; 3] {
        [self.x as f64, self.y as f64, self.z as f64]
    }
}

#[inline]
fn points_to_bytes(points: &[Point3]) -> &[u8] {
    unsafe {
        std::slice::from_raw_parts(points.as_ptr() as *const u8, points.len() * POINT_BYTES)
    }
}

#[inline]
fn points_to_bytes_mut(points: &mut [Point3]) -> &mut [u8] {
    unsafe {
        std::slice::from_raw_parts_mut(points.as_mut_ptr() as *mut u8, points.len() * POINT_BYTES)
    }
}

struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E3779B97F4A7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
    #[inline]
    fn next_usize(&mut self, bound: usize) -> usize {
        if bound <= 1 {
            0
        } else {
            (self.next_u64() % bound as u64) as usize
        }
    }
}

struct KdNode {
    index: usize,
    axis: usize,
    split: f64,
    left: Option<usize>,
    right: Option<usize>,
}

struct KdTree {
    nodes: Vec<KdNode>,
    root: Option<usize>,
}

impl KdTree {
    fn build(centroids: &[[f64; 3]]) -> Self {
        if centroids.is_empty() {
            return Self {
                nodes: Vec::new(),
                root: None,
            };
        }
        let mut indices: Vec<usize> = (0..centroids.len()).collect();
        let mut nodes = Vec::with_capacity(centroids.len());
        let root = Self::build_recursive(centroids, &mut indices, 0, &mut nodes);
        Self {
            nodes,
            root: Some(root),
        }
    }

    fn build_recursive(
        centroids: &[[f64; 3]],
        indices: &mut [usize],
        depth: usize,
        nodes: &mut Vec<KdNode>,
    ) -> usize {
        let axis = depth % 3;
        let mid = indices.len() / 2;
        indices.select_nth_unstable_by(mid, |a, b| {
            centroids[*a][axis]
                .partial_cmp(&centroids[*b][axis])
                .unwrap_or(Ordering::Equal)
        });
        let (left_slice, rest) = indices.split_at_mut(mid);
        let (median_slice, right_slice) = rest.split_first_mut().unwrap();
        let current_index = *median_slice;

        let node_idx = nodes.len();
        nodes.push(KdNode {
            index: current_index,
            axis,
            split: centroids[current_index][axis],
            left: None,
            right: None,
        });

        if !left_slice.is_empty() {
            let left_child = Self::build_recursive(centroids, left_slice, depth + 1, nodes);
            nodes[node_idx].left = Some(left_child);
        }
        if !right_slice.is_empty() {
            let right_child = Self::build_recursive(centroids, right_slice, depth + 1, nodes);
            nodes[node_idx].right = Some(right_child);
        }
        node_idx
    }

    fn nearest(&self, point: [f64; 3], centroids: &[[f64; 3]]) -> usize {
        let root = match self.root {
            Some(idx) => idx,
            None => return 0,
        };
        let mut best_idx = self.nodes[root].index;
        let mut best_dist = distance_squared(centroids[best_idx], point);
        self.search(root, point, centroids, &mut best_idx, &mut best_dist);
        best_idx
    }

    fn search(
        &self,
        node_idx: usize,
        point: [f64; 3],
        centroids: &[[f64; 3]],
        best_idx: &mut usize,
        best_dist: &mut f64,
    ) {
        let node = &self.nodes[node_idx];
        let idx = node.index;
        let dist = distance_squared(centroids[idx], point);
        if dist < *best_dist {
            *best_dist = dist;
            *best_idx = idx;
        }

        let axis = node.axis;
        let diff = point[axis] - node.split;
        let (primary, secondary) = if diff <= 0.0 {
            (node.left, node.right)
        } else {
            (node.right, node.left)
        };

        if let Some(child) = primary {
            self.search(child, point, centroids, best_idx, best_dist);
        }
        if diff * diff < *best_dist {
            if let Some(child) = secondary {
                self.search(child, point, centroids, best_idx, best_dist);
            }
        }
    }
}

enum SearchMethod {
    Linear,
    Tree(KdTree),
}

struct NearestSearcher {
    method: SearchMethod,
}

impl NearestSearcher {
    fn new(centroids: &[[f64; 3]]) -> Self {
        if centroids.len() < TREE_THRESHOLD {
            Self {
                method: SearchMethod::Linear,
            }
        } else {
            Self {
                method: SearchMethod::Tree(KdTree::build(centroids)),
            }
        }
    }

    #[inline]
    fn nearest_idx(&self, point: [f64; 3], centroids: &[[f64; 3]]) -> usize {
        match &self.method {
            SearchMethod::Linear => linear_nearest(centroids, point),
            SearchMethod::Tree(tree) => tree.nearest(point, centroids),
        }
    }
}

#[inline]
fn distance_squared(a: [f64; 3], b: [f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn linear_nearest(centroids: &[[f64; 3]], point: [f64; 3]) -> usize {
    let mut best = 0;
    let mut best_dist = f64::INFINITY;
    for (idx, centroid) in centroids.iter().enumerate() {
        let dist = distance_squared(*centroid, point);
        if dist < best_dist {
            best_dist = dist;
            best = idx;
        }
    }
    best
}

struct DiskWriter {
    path: PathBuf,
    writer: BufWriter<File>,
    count: u64,
}

impl DiskWriter {
    fn new() -> io::Result<Self> {
        let mut path = env::temp_dir();
        let nanos = SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap_or_default()
            .as_nanos();
        path.push(format!(
            "rust_stream_kmeans_{}_{}.bin",
            process::id(),
            nanos
        ));
        let file = File::create(&path)?;
        Ok(Self {
            path,
            writer: BufWriter::with_capacity(4 * 1024 * 1024, file),
            count: 0,
        })
    }

    fn write_chunk(&mut self, points: &[Point3]) -> io::Result<()> {
        if points.is_empty() {
            return Ok(());
        }
        let bytes = points_to_bytes(points);
        self.writer.write_all(bytes)?;
        self.count += points.len() as u64;
        Ok(())
    }

    fn finish(mut self) -> io::Result<DiskStore> {
        self.writer.flush()?;
        Ok(DiskStore {
            path: self.path,
            num_points: self.count,
        })
    }
}

struct DiskStore {
    path: PathBuf,
    num_points: u64,
}

impl DiskStore {
    fn scan_points<F>(&self, mut f: F) -> io::Result<()>
    where
        F: FnMut(Point3) -> io::Result<()>,
    {
        let mut reader = BufReader::with_capacity(4 * 1024 * 1024, File::open(&self.path)?);
        let mut buffer = vec![Point3::default(); DISK_CHUNK_POINTS];
        let mut remaining = self.num_points;
        while remaining > 0 {
            let chunk = remaining.min(buffer.len() as u64) as usize;
            let slice = &mut buffer[..chunk];
            read_points(&mut reader, slice)?;
            for &pt in slice {
                f(pt)?;
            }
            remaining -= chunk as u64;
        }
        Ok(())
    }
}

impl Drop for DiskStore {
    fn drop(&mut self) {
        let _ = remove_file(&self.path);
    }
}

enum DatasetStorage {

struct IngestResult {

enum Strategy {
    FullBatch,
    MiniBatch,
    StreamingLarge,
    StreamingExtreme,
}

enum StorageTarget {
    Memory,
    Disk,
}

fn select_strategy(num_points: u64, num_clusters: u64) -> Strategy {
    if num_points <= FULL_BATCH_POINT_LIMIT && num_clusters <= 4_000 {
        Strategy::FullBatch
    } else if num_points <= MINI_BATCH_POINT_LIMIT {
        Strategy::MiniBatch
    } else if num_points <= 1_000_000_000 {
        Strategy::StreamingLarge
    } else {
        Strategy::StreamingExtreme
    }
}

fn read_header<R: BufRead>(reader: &mut R) -> io::Result<(u64, u64)> {
    let mut line = String::new();
    loop {
        line.clear();
        let bytes = reader.read_line(&mut line)?;
        if bytes == 0 {
            return Err(io::Error::new(
                ErrorKind::UnexpectedEof,
                "header missing",
            ));
        }
        if !line.trim().is_empty() {
            break;
        }
    }
    let mut parts = line.split_whitespace();
    let num_points = parts
        .next()
        .ok_or_else(|| io::Error::new(ErrorKind::InvalidData, "missing num_points"))?
        .parse::<u64>()
        .map_err(|_| io::Error::new(ErrorKind::InvalidData, "invalid num_points"))?;
    let num_clusters = parts
        .next()
        .ok_or_else(|| io::Error::new(ErrorKind::InvalidData, "missing num_clusters"))?
        .parse::<u64>()
        .map_err(|_| io::Error::new(ErrorKind::InvalidData, "invalid num_clusters"))?;
    Ok((num_points, num_clusters))
}

fn read_points<R: Read>(reader: &mut R, out: &mut [Point3]) -> io::Result<()> {
    if out.is_empty() {
        return Ok(());
    }
    let bytes = points_to_bytes_mut(out);
    reader.read_exact(bytes)
}

fn drain_points<R: Read>(reader: &mut R, num_points: u64) -> io::Result<()> {
    let mut buffer = vec![Point3::default(); INGEST_CHUNK_POINTS];
    let mut remaining = num_points;
    while remaining > 0 {
        let chunk = remaining.min(buffer.len() as u64) as usize;
        read_points(reader, &mut buffer[..chunk])?;
        remaining -= chunk as u64;
    }
    Ok(())
}

fn seed_bank_target(k: usize) -> usize {
    let base = k.max(1024);
    base.saturating_mul(4).min(100_000).max(base)
}

fn ingest_dataset<R: Read>(
    reader: &mut R,
    num_points: u64,
    num_clusters: usize,
    target: StorageTarget,
    rng: &mut SplitMix64,
) -> io::Result<IngestResult> {
    if num_clusters == 0 {
        return Ok(IngestResult {
            storage: match target {
                StorageTarget::Memory => DatasetStorage::Memory(Vec::new()),
                StorageTarget::Disk => {
                    let writer = DiskWriter::new()?;
                    DatasetStorage::Disk(writer.finish()?)
                }
            },
            centroids: Vec::new(),
            seed_bank: Vec::new(),
        });
    }

    if matches!(target, StorageTarget::Memory) && num_points > usize::MAX as u64 {
        return Err(io::Error::new(
            ErrorKind::InvalidInput,
            "dataset too large for memory mode",
        ));
    }

    let mut buffer = vec![Point3::default(); INGEST_CHUNK_POINTS];
    let mut memory_points = match target {
        StorageTarget::Memory => Some(Vec::with_capacity(num_points as usize)),
        StorageTarget::Disk => None,
    };
    let mut disk_writer = match target {
        StorageTarget::Memory => None,
        StorageTarget::Disk => Some(DiskWriter::new()?),
    };

    let mut centroids = Vec::with_capacity(num_clusters.min(num_points as usize).max(1));
    let mut seed_bank = Vec::with_capacity(seed_bank_target(num_clusters));

    let mut processed = 0u64;
    while processed < num_points {
        let chunk = ((num_points - processed).min(buffer.len() as u64)) as usize;
        let slice = &mut buffer[..chunk];
        read_points(reader, slice)?;
        if let Some(points) = memory_points.as_mut() {
            points.extend_from_slice(slice);
        }
        if let Some(writer) = disk_writer.as_mut() {
            writer.write_chunk(slice)?;
        }
        for (offset, point) in slice.iter().enumerate() {
            let pf = point.to_f64();
            let global_idx = processed + offset as u64;

            if centroids.len() < num_clusters {
                centroids.push(pf);
            } else {
                let r = rng.next_u64() % (global_idx + 1);
                if r < num_clusters as u64 {
                    centroids[r as usize] = pf;
                }
            }

            if seed_bank.len() < seed_bank.capacity() {
                seed_bank.push(pf);
            } else {
                let r = rng.next_u64() % (global_idx + 1);
                if r < seed_bank.len() as u64 {
                    seed_bank[r as usize] = pf;
                }
            }
        }
        processed += chunk as u64;
    }

    if centroids.is_empty() {
        centroids.push([0.0; 3]);
    }
    while centroids.len() < num_clusters {
        let idx = rng.next_usize(centroids.len());
        centroids.push(centroids[idx]);
    }
    if seed_bank.is_empty() {
        seed_bank.extend_from_slice(&centroids);
    }

    let storage = match target {
        StorageTarget::Memory => DatasetStorage::Memory(memory_points.unwrap()),
        StorageTarget::Disk => DatasetStorage::Disk(disk_writer.unwrap().finish()?),
    };

    Ok(IngestResult {
        storage,
        centroids,
        seed_bank,
    })
}

fn choose_replacement(
    fallback_points: Option<&[Point3]>,
    seed_bank: &[[f64; 3]],
    centroids: &[[f64; 3]],
    rng: &mut SplitMix64,
) -> [f64; 3] {
    if let Some(points) = fallback_points {
        if !points.is_empty() {
            return points[rng.next_usize(points.len())].to_f64();
        }
    }
    if !seed_bank.is_empty() {
        seed_bank[rng.next_usize(seed_bank.len())]
    } else if !centroids.is_empty() {
        centroids[rng.next_usize(centroids.len())]
    } else {
        [0.0; 3]
    }
}

fn update_centroids(
    centroids: &mut [[f64; 3]],
    sums: &[[f64; 3]],
    counts: &[u64],
    fallback_points: Option<&[Point3]>,
    seed_bank: &[[f64; 3]],
    rng: &mut SplitMix64,
) -> f64 {
    let mut max_shift = 0.0;
    for i in 0..centroids.len() {
        if counts[i] > 0 {
            let inv = 1.0 / counts[i] as f64;
            let new_c = [
                sums[i][0] * inv,
                sums[i][1] * inv,
                sums[i][2] * inv,
            ];
            let shift = distance_squared(centroids[i], new_c).sqrt();
            if shift > max_shift {
                max_shift = shift;
            }
            centroids[i] = new_c;
        } else {
            centroids[i] = choose_replacement(fallback_points, seed_bank, centroids, rng);
        }
    }
    max_shift
}

fn run_full_batch(
    points: &[Point3],
    centroids: &mut Vec<[f64; 3]>,
    seed_bank: &[[f64; 3]],
    rng: &mut SplitMix64,
) -> Vec<u32> {
    let n = points.len();
    if centroids.is_empty() {
        return vec![0u32; n];
    }
    let mut assignments = vec![0u32; n];
    if n == 0 {
        return assignments;
    }

    let k = centroids.len();
    let mut sums = vec![[0f64; 3]; k];
    let mut counts = vec![0u64; k];
    let mut prev_shift = f64::MAX;

    for _ in 0..50 {
        sums.iter_mut().for_each(|s| *s = [0.0; 3]);
        counts.iter_mut().for_each(|c| *c = 0);
        let searcher = NearestSearcher::new(centroids);
        let mut changes = 0usize;

        for (idx, point) in points.iter().enumerate() {
            let pf = point.to_f64();
            let cluster = searcher.nearest_idx(pf, centroids);
            if assignments[idx] != cluster as u32 {
                assignments[idx] = cluster as u32;
                changes += 1;
            }
            sums[cluster][0] += pf[0];
            sums[cluster][1] += pf[1];
            sums[cluster][2] += pf[2];
            counts[cluster] += 1;
        }

        let shift =
            update_centroids(centroids, &sums, &counts, Some(points), seed_bank, rng);
        if shift < 1e-5 || (changes as f64 / n as f64) < 5e-4 || (prev_shift - shift).abs() < 1e-7
        {
            break;
        }
        prev_shift = shift;
    }

    assignments
}

fn train_multi_pass(
    store: &DiskStore,
    centroids: &mut Vec<[f64; 3]>,
    passes: usize,
    tolerance: f64,
    seed_bank: &[[f64; 3]],
    rng: &mut SplitMix64,
) -> io::Result<()> {
    if centroids.is_empty() {
        return Ok(());
    }
    let k = centroids.len();
    let mut sums = vec![[0f64; 3]; k];
    let mut counts = vec![0u64; k];

    for _ in 0..passes {
        sums.iter_mut().for_each(|s| *s = [0.0; 3]);
        counts.iter_mut().for_each(|c| *c = 0);

        let searcher = NearestSearcher::new(centroids);
        store.scan_points(|pt| {
            let pf = pt.to_f64();
            let idx = searcher.nearest_idx(pf, centroids);
            sums[idx][0] += pf[0];
            sums[idx][1] += pf[1];
            sums[idx][2] += pf[2];
            counts[idx] += 1;
            Ok(())
        })?;

        let shift = update_centroids(centroids, &sums, &counts, None, seed_bank, rng);
        if shift < tolerance {
            break;
        }
    }
    Ok(())
}

fn write_assignments_vec(assignments: &[u32]) -> io::Result<()> {
    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(OUTPUT_BUF_BYTES, stdout.lock());
    for &idx in assignments {
        writer.write_all(&idx.to_le_bytes())?;
    }
    writer.flush()
}

fn assign_and_write_disk(store: &DiskStore, centroids: &[[f64; 3]]) -> io::Result<()> {
    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(OUTPUT_BUF_BYTES, stdout.lock());
    if centroids.is_empty() {
        writer.flush()?;
        return Ok(());
    }
    let searcher = NearestSearcher::new(centroids);
    store.scan_points(|pt| {
        let idx = searcher.nearest_idx(pt.to_f64(), centroids);
        writer.write_all(&(idx as u32).to_le_bytes())?;
        Ok(())
    })?;
    writer.flush()
}

fn run_streaming<R: Read>(
    reader: &mut R,
    num_points: u64,
    num_clusters: usize,
    chunk_points: usize,
) -> io::Result<()> {
    if num_clusters == 0 {
        return drain_points(reader, num_points);
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(OUTPUT_BUF_BYTES, stdout.lock());
    let mut buffer = vec![Point3::default(); chunk_points];

    let mut centroids = vec![[0.0; 3]; num_clusters];
    let mut running_sums = vec![[0f64; 3]; num_clusters];
    let mut running_counts = vec![0f64; num_clusters];
    let mut chunk_sums = vec![[0f64; 3]; num_clusters];
    let mut chunk_counts = vec![0f64; num_clusters];

    let mut active = 0usize;
    let mut remaining = num_points;

    while remaining > 0 {
        let chunk = remaining.min(buffer.len() as u64) as usize;
        let slice = &mut buffer[..chunk];
        read_points(reader, slice)?;
        remaining -= chunk as u64;

        if active > 0 {
            for i in 0..active {
                chunk_counts[i] = 0.0;
                chunk_sums[i] = [0.0; 3];
            }
        }

        let mut searcher = if active >= TREE_THRESHOLD {
            Some(NearestSearcher::new(&centroids[..active]))
        } else {
            None
        };

        for &point in slice {
            if active < num_clusters {
                let pf = point.to_f64();
                centroids[active] = pf;
                running_sums[active] = pf;
                running_counts[active] = 1.0;
                chunk_counts[active] = 0.0;
                chunk_sums[active] = [0.0; 3];
                writer.write_all(&(active as u32).to_le_bytes())?;
                active += 1;
                if active >= TREE_THRESHOLD {
                    searcher = Some(NearestSearcher::new(&centroids[..active]));
                }
                continue;
            }

            let pf = point.to_f64();
            let idx = match &searcher {
                Some(tree) => tree.nearest_idx(pf, &centroids[..active]),
                None => linear_nearest(&centroids[..active], pf),
            };
            chunk_counts[idx] += 1.0;
            chunk_sums[idx][0] += pf[0];
            chunk_sums[idx][1] += pf[1];
            chunk_sums[idx][2] += pf[2];
            writer.write_all(&(idx as u32).to_le_bytes())?;
        }

        for i in 0..active {
            if chunk_counts[i] > 0.0 {
                running_counts[i] += chunk_counts[i];
                running_sums[i][0] += chunk_sums[i][0];
                running_sums[i][1] += chunk_sums[i][1];
                running_sums[i][2] += chunk_sums[i][2];
                let inv = 1.0 / running_counts[i];
                centroids[i][0] = running_sums[i][0] * inv;
                centroids[i][1] = running_sums[i][1] * inv;
                centroids[i][2] = running_sums[i][2] * inv;
            }
        }
    }

    writer.flush()
}

fn streaming_chunk_size(strategy: &Strategy) -> usize {
    match strategy {
        Strategy::StreamingExtreme => STREAM_CHUNK_POINTS_EXTREME,
        _ => STREAM_CHUNK_POINTS_LARGE,
    }
}

fn derive_seed(num_points: u64, num_clusters: u64) -> u64 {
    num_points
        .wrapping_mul(0x9E3779B97F4A7C15)
        .rotate_left(13)
        ^ num_clusters.wrapping_mul(0xD2B74407B1CE6E93).rotate_left(7)
        ^ 0xA4093822299F31D0
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(INPUT_BUF_BYTES, stdin.lock());
    let (num_points, num_clusters_raw) = read_header(&mut reader)?;
    if num_clusters_raw == 0 {
        drain_points(&mut reader, num_points)?;
        return Ok(());
    }
    let num_clusters = usize::try_from(num_clusters_raw)
        .map_err(|_| io::Error::new(ErrorKind::InvalidInput, "cluster count too large"))?;

    let strategy = select_strategy(num_points, num_clusters_raw);
    let seed = derive_seed(num_points, num_clusters_raw);
    let mut rng = SplitMix64::new(seed);

    match strategy {
        Strategy::StreamingLarge | Strategy::StreamingExtreme => {
            run_streaming(
                &mut reader,
                num_points,
                num_clusters,
                streaming_chunk_size(&strategy),
            )?;
        }
        Strategy::FullBatch | Strategy::MiniBatch => {
            let target = if matches!(strategy, Strategy::FullBatch) {
                StorageTarget::Memory
            } else {
                StorageTarget::Disk
            };
            let IngestResult {
                storage,
                mut centroids,
                mut seed_bank,
            } = ingest_dataset(&mut reader, num_points, num_clusters, target, &mut rng)?;
            if seed_bank.is_empty() && !centroids.is_empty() {
                seed_bank.extend_from_slice(&centroids);
            }

            match (strategy, storage) {
                (Strategy::FullBatch, DatasetStorage::Memory(points)) => {
                    let assignments = run_full_batch(&points, &mut centroids, &seed_bank, &mut rng);
                    write_assignments_vec(&assignments)?;
                }
                (Strategy::MiniBatch, DatasetStorage::Disk(store)) => {
                    train_multi_pass(&store, &mut centroids, 8, 5e-4, &seed_bank, &mut rng)?;
                    assign_and_write_disk(&store, &centroids)?;
                }
                _ => {
                    return Err(io::Error::new(
                        ErrorKind::Other,
                        "internal storage/strategy mismatch",
                    ))
                }
            }
        }
    }

    Ok(())
}
