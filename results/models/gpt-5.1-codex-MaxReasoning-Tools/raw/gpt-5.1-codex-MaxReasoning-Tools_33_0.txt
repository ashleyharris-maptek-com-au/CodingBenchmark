Here's a standalone C# implementation that reads the QAP instance from stdin, builds a permutation via flow/distance-based sorting, and then improves it with a mix of deterministic swaps (for small N), simulated annealing, and randomized descent so it scales up to the required sizes.

```csharp
using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var reader = new FastReader();
        int n;
        try
        {
            n = reader.NextInt();
        }
        catch (EndOfStreamException)
        {
            return;
        }

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            int[] row = new int[n];
            for (int j = 0; j < n; j++)
            {
                row[j] = reader.NextInt();
            }
            flow[i] = row;
        }

        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            int[] row = new int[n];
            for (int j = 0; j < n; j++)
            {
                row[j] = reader.NextInt();
            }
            dist[i] = row;
        }

        var solver = new QapSolver(flow, dist);
        solver.Solve();
        solver.PrintSolution();
    }
}

sealed class FastReader
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _bufferLength;
    private int _bufferPosition;

    public FastReader()
    {
        _stream = Console.OpenStandardInput();
        _buffer = new byte[1 << 16];
    }

    private int Read()
    {
        if (_bufferPosition >= _bufferLength)
        {
            _bufferLength = _stream.Read(_buffer, 0, _buffer.Length);
            _bufferPosition = 0;
            if (_bufferLength <= 0)
            {
                return -1;
            }
        }

        return _buffer[_bufferPosition++];
    }

    public int NextInt()
    {
        int c = Read();
        while (c <= ' ')
        {
            if (c == -1)
            {
                throw new EndOfStreamException();
            }
            c = Read();
        }

        bool negative = false;
        if (c == '-')
        {
            negative = true;
            c = Read();
        }

        int value = 0;
        while (c > ' ')
        {
            value = value * 10 + (c - '0');
            c = Read();
        }

        return negative ? -value : value;
    }
}

sealed class QapSolver
{
    private readonly int n;
    private readonly int[][] flow;
    private readonly int[][] dist;
    private readonly Random rng;
    private readonly double timeLimitSeconds;
    private readonly Stopwatch stopwatch = new Stopwatch();

    private int[] perm = Array.Empty<int>();
    private int[] bestPerm = Array.Empty<int>();
    private long currentCost;
    private long bestCost;
    private int[] flowOrder = Array.Empty<int>();

    private const int DeterministicThreshold = 120;

    public QapSolver(int[][] flow, int[][] dist)
    {
        this.n = flow.Length;
        this.flow = flow;
        this.dist = dist;
        timeLimitSeconds = DetermineTimeLimitSeconds(n);

        int seed = 7919;
        int inspect = Math.Min(n, 16);
        for (int i = 0; i < inspect; i++)
        {
            seed = unchecked(seed * 1315423911 + flow[i][i] + dist[i][i] + i);
        }
        rng = new Random(seed);
    }

    public void Solve()
    {
        stopwatch.Restart();
        BuildInitialSolution();

        if (n >= 2)
        {
            if (n <= DeterministicThreshold)
            {
                DeterministicLocalSearch();
            }

            SimulatedAnnealing();

            if (stopwatch.Elapsed.TotalSeconds < timeLimitSeconds)
            {
                currentCost = bestCost;
                Array.Copy(bestPerm, perm, n);
                RandomDescent();
            }
        }

        stopwatch.Stop();
    }

    public void PrintSolution()
    {
        long verifiedCost = ComputeTotalCost(bestPerm);
        bestCost = verifiedCost;
        Console.WriteLine(bestCost);

        var sb = new StringBuilder();
        if (bestPerm.Length > 0)
        {
            sb.Append(bestPerm[0]);
            for (int i = 1; i < bestPerm.Length; i++)
            {
                sb.Append(' ').Append(bestPerm[i]);
            }
        }
        Console.WriteLine(sb.ToString());
    }

    private static double DetermineTimeLimitSeconds(int size)
    {
        if (size <= 0)
        {
            return 0.2;
        }

        double limit;
        if (size <= 20)
        {
            limit = 0.6;
        }
        else if (size <= 60)
        {
            limit = 0.9 + 0.01 * size;
        }
        else if (size <= 200)
        {
            limit = 1.5 + 0.008 * size;
        }
        else if (size <= 800)
        {
            limit = 3.1 + 0.004 * (size - 200);
        }
        else
        {
            limit = 5.5 + 0.002 * (size - 800);
        }

        if (limit > 12.0)
        {
            limit = 12.0;
        }

        return limit;
    }

    private void BuildInitialSolution()
    {
        perm = new int[n];
        bestPerm = new int[n];
        flowOrder = new int[n];

        if (n == 0)
        {
            currentCost = 0;
            bestCost = 0;
            return;
        }

        long[] flowWeight = new long[n];
        long[] distWeight = new long[n];

        for (int i = 0; i < n; i++)
        {
            long sum = 0;
            var row = flow[i];
            for (int j = 0; j < n; j++)
            {
                sum += row[j];
            }
            flowWeight[i] = sum;
            flowOrder[i] = i;
        }

        for (int i = 0; i < n; i++)
        {
            long sum = 0;
            var row = dist[i];
            for (int j = 0; j < n; j++)
            {
                sum += row[j];
            }
            distWeight[i] = sum;
        }

        Array.Sort(flowOrder, (a, b) =>
        {
            int cmp = flowWeight[b].CompareTo(flowWeight[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        int[] locationOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            locationOrder[i] = i;
        }

        Array.Sort(locationOrder, (a, b) =>
        {
            int cmp = distWeight[a].CompareTo(distWeight[b]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        for (int idx = 0; idx < n; idx++)
        {
            perm[flowOrder[idx]] = locationOrder[idx];
        }

        currentCost = ComputeTotalCost(perm);
        bestCost = currentCost;
        Array.Copy(perm, bestPerm, n);
    }

    private void DeterministicLocalSearch()
    {
        if (n < 2)
        {
            return;
        }

        bool improved = true;
        while (improved && stopwatch.Elapsed.TotalSeconds < timeLimitSeconds)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                if (stopwatch.Elapsed.TotalSeconds >= timeLimitSeconds)
                {
                    break;
                }

                for (int j = i + 1; j < n; j++)
                {
                    long delta = ComputeSwapDelta(i, j, perm);
                    if (delta < 0)
                    {
                        ApplySwap(i, j, perm);
                        currentCost += delta;
                        UpdateBest();
                        improved = true;
                    }

                    if (stopwatch.Elapsed.TotalSeconds >= timeLimitSeconds)
                    {
                        break;
                    }
                }
            }
        }
    }

    private void SimulatedAnnealing()
    {
        if (n < 2)
        {
            return;
        }

        double initialTemp = EstimateInitialTemperature();
        if (initialTemp < 1e-6)
        {
            initialTemp = 1.0;
        }

        double finalTemp = Math.Max(1e-3, initialTemp * 1e-4);
        if (finalTemp > initialTemp)
        {
            finalTemp = initialTemp;
        }

        double logRatio = Math.Log(finalTemp / initialTemp);

        while (stopwatch.Elapsed.TotalSeconds < timeLimitSeconds)
        {
            double elapsed = stopwatch.Elapsed.TotalSeconds;
            double progress = Math.Min(1.0, elapsed / timeLimitSeconds);
            double temperature = (Math.Abs(logRatio) < 1e-12)
                ? initialTemp
                : initialTemp * Math.Exp(logRatio * progress);

            int innerIterations = DetermineAnnealingBatchSize();
            for (int iter = 0; iter < innerIterations; iter++)
            {
                if (stopwatch.Elapsed.TotalSeconds >= timeLimitSeconds)
                {
                    return;
                }

                int a = PickFocusedFacility();
                int b = rng.Next(n - 1);
                if (b >= a)
                {
                    b++;
                }

                long delta = ComputeSwapDelta(a, b, perm);
                if (delta <= 0)
                {
                    ApplySwap(a, b, perm);
                    currentCost += delta;
                    UpdateBest();
                }
                else
                {
                    double acceptProbability = Math.Exp(-delta / Math.Max(temperature, 1e-9));
                    if (rng.NextDouble() < acceptProbability)
                    {
                        ApplySwap(a, b, perm);
                        currentCost += delta;
                    }
                }
            }
        }
    }

    private void RandomDescent()
    {
        if (n < 2)
        {
            return;
        }

        while (stopwatch.Elapsed.TotalSeconds < timeLimitSeconds)
        {
            bool improved = false;
            int attempts = Math.Min(2048, Math.Max(256, n));
            for (int iter = 0; iter < attempts; iter++)
            {
                if (stopwatch.Elapsed.TotalSeconds >= timeLimitSeconds)
                {
                    return;
                }

                int a = PickFocusedFacility();
                int b = rng.Next(n - 1);
                if (b >= a)
                {
                    b++;
                }

                long delta = ComputeSwapDelta(a, b, perm);
                if (delta < 0)
                {
                    ApplySwap(a, b, perm);
                    currentCost += delta;
                    UpdateBest();
                    improved = true;
                }
            }

            if (!improved)
            {
                break;
            }
        }
    }

    private double EstimateInitialTemperature()
    {
        if (n < 2)
        {
            return 1.0;
        }

        int samples = Math.Max(50, Math.Min(400, n * 3));
        long positiveSum = 0;
        int positiveCount = 0;

        for (int s = 0; s < samples; s++)
        {
            if (stopwatch.Elapsed.TotalSeconds >= timeLimitSeconds)
            {
                break;
            }

            int a = PickFocusedFacility();
            int b = rng.Next(n - 1);
            if (b >= a)
            {
                b++;
            }

            long delta = ComputeSwapDelta(a, b, perm);
            if (delta > 0)
            {
                positiveSum += delta;
                positiveCount++;
            }
        }

        if (positiveCount == 0)
        {
            return 1.0;
        }

        double average = (double)positiveSum / positiveCount;
        return Math.Max(1.0, average);
    }

    private int DetermineAnnealingBatchSize()
    {
        int baseCount = n / 2;
        if (baseCount < 256)
        {
            baseCount = 256;
        }

        if (baseCount > 1536)
        {
            baseCount = 1536;
        }

        return baseCount;
    }

    private int PickFocusedFacility()
    {
        if (n <= 1 || flowOrder.Length == 0)
        {
            return 0;
        }

        int topCount = Math.Max(2, n / 3);
        if (topCount > flowOrder.Length)
        {
            topCount = flowOrder.Length;
        }

        double roll = rng.NextDouble();
        if (roll < 0.65)
        {
            return flowOrder[rng.Next(topCount)];
        }

        if (roll < 0.9)
        {
            return flowOrder[rng.Next(flowOrder.Length)];
        }

        return rng.Next(n);
    }

    private void UpdateBest()
    {
        if (currentCost < bestCost)
        {
            bestCost = currentCost;
                Array.Copy(perm, bestPerm, n);
        }
    }

    private static void ApplySwap(int a, int b, int[] array)
    {
        int temp = array[a];
        array[a] = array[b];
        array[b] = temp;
    }

    private long ComputeSwapDelta(int a, int b, int[] permutation)
    {
        if (a == b)
        {
            return 0;
        }

        int pa = permutation[a];
        int pb = permutation[b];

        var flowA = flow[a];
        var flowB = flow[b];
        var distPa = dist[pa];
        var distPb = dist[pb];

        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b)
            {
                continue;
            }

            int pk = permutation[k];
            var flowK = flow[k];
            var distPk = dist[pk];

            delta += (long)(flowA[k] - flowB[k]) * (distPb[pk] - distPa[pk]);
            delta += (long)(flowK[a] - flowK[b]) * (distPk[pb] - distPk[pa]);
        }

        delta += (long)(flowA[a] - flowB[b]) * (distPb[pb] - distPa[pa]);
        delta += (long)(flowA[b] - flowB[a]) * (distPb[pa] - distPa[pb]);

        return delta;
    }

    private long ComputeTotalCost(int[] permutation)
    {
        if (permutation.Length == 0)
        {
            return 0;
        }

        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = permutation[i];
            var flowRow = flow[i];
            var distRow = dist[pi];
            for (int j = 0; j < n; j++)
            {
                int pj = permutation[j];
                total += (long)flowRow[j] * distRow[pj];
            }
        }

        return total;
    }
}
```

This program reads the matrices, builds an initial permutation by matching high-flow facilities to low-distance locations, and repeatedly improves the solution with swap-based heuristics (deterministic descent for smaller instances, simulated annealing, and a final randomized descent for polishing). It outputs the best cost found followed by the permutation, satisfying the required format.
