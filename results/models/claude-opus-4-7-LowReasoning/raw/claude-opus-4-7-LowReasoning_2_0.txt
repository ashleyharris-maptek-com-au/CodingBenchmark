using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int N, M;
    static int[] eu, ev;
    static long[] ew;
    static List<(int to, int eid)>[] adj;

    static void Main()
    {
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        N = int.Parse(tokens[p++]);
        M = int.Parse(tokens[p++]);
        eu = new int[M];
        ev = new int[M];
        ew = new long[M];
        adj = new List<(int, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int)>();
        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(tokens[p++]);
            int v = int.Parse(tokens[p++]);
            long w = long.Parse(tokens[p++]);
            eu[i] = u; ev[i] = v; ew[i] = w;
            adj[u].Add((v, i));
            if (u != v) adj[v].Add((u, i));
        }

        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            if (eu[i] == ev[i]) continue;
            deg[eu[i]]++; deg[ev[i]]++;
        }
        var odd = new List<int>();
        for (int i = 0; i < N; i++) if ((deg[i] & 1) == 1) odd.Add(i);

        var extra = new List<(int u, int v, int origEid)>();
        if (odd.Count > 0)
        {
            PairOddVertices(odd, extra);
        }

        int totalEdges = M + extra.Count;
        int[] meu = new int[totalEdges];
        int[] mev = new int[totalEdges];
        for (int i = 0; i < M; i++) { meu[i] = eu[i]; mev[i] = ev[i]; }
        for (int i = 0; i < extra.Count; i++)
        {
            meu[M + i] = extra[i].u;
            mev[M + i] = extra[i].v;
        }

        int[] degM = new int[N];
        for (int i = 0; i < totalEdges; i++)
        {
            degM[meu[i]]++;
            if (meu[i] != mev[i]) degM[mev[i]]++;
        }
        int[] head = new int[N + 1];
        for (int i = 0; i < N; i++) head[i + 1] = head[i] + degM[i];
        int[] nextSlot = new int[N];
        int[] adjE = new int[head[N]];
        int[] adjV = new int[head[N]];
        for (int i = 0; i < totalEdges; i++)
        {
            int u = meu[i], v = mev[i];
            int su = head[u] + nextSlot[u]++; adjE[su] = i; adjV[su] = v;
            if (u != v) { int sv = head[v] + nextSlot[v]++; adjE[sv] = i; adjV[sv] = u; }
        }

        bool[] used = new bool[totalEdges];
        int[] iter = new int[N];
        var stack = new Stack<int>();
        var circuit = new List<int>();
        stack.Push(0);
        while (stack.Count > 0)
        {
            int v = stack.Peek();
            int start = head[v];
            int end = head[v + 1];
            int idx = start + iter[v];
            while (idx < end && used[adjE[idx]]) { iter[v]++; idx++; }
            if (idx < end)
            {
                int eid = adjE[idx];
                int nv = adjV[idx];
                used[eid] = true;
                iter[v]++;
                stack.Push(nv);
            }
            else
            {
                circuit.Add(v);
                stack.Pop();
            }
        }
        circuit.Reverse();

        var sb = new StringBuilder();
        sb.Append(circuit.Count);
        sb.Append('\n');
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.Out.Write(sb.ToString());
    }

    static void PairOddVertices(List<int> odd, List<(int u, int v, int origEid)> extra)
    {
        bool[] isOdd = new bool[N];
        bool[] matched = new bool[N];
        foreach (var o in odd) isOdd[o] = true;

        var remaining = new HashSet<int>(odd);

        long[] dist = new long[N];
        int[] prevNode = new int[N];
        int[] prevEdge = new int[N];

        while (remaining.Count > 0)
        {
            int src = -1;
            foreach (var x in remaining) { src = x; break; }
            remaining.Remove(src);

            for (int i = 0; i < N; i++) { dist[i] = long.MaxValue; prevNode[i] = -1; prevEdge[i] = -1; }
            dist[src] = 0;
            var pq = new PriorityQueue<int, long>();
            pq.Enqueue(src, 0);
            int found = -1;
            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[u]) continue;
                if (u != src && isOdd[u] && !matched[u] && remaining.Contains(u))
                {
                    found = u;
                    break;
                }
                foreach (var (to, eid) in adj[u])
                {
                    long nd = d + ew[eid];
                    if (nd < dist[to])
                    {
                        dist[to] = nd;
                        prevNode[to] = u;
                        prevEdge[to] = eid;
                        pq.Enqueue(to, nd);
                    }
                }
            }
            if (found == -1) continue;
            remaining.Remove(found);
            matched[src] = true;
            matched[found] = true;
            int cur = found;
            while (cur != src)
            {
                int pe = prevEdge[cur];
                int pn = prevNode[cur];
                extra.Add((eu[pe], ev[pe], pe));
                cur = pn;
            }
        }
    }
}
