using System;
using System.Collections.Generic;
using System.Globalization;
using System.Diagnostics;

class Program
{
    static double[][] P;
    static double[][] V;
    static int N;
    static double mu;

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        var ci = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd();
        var lines = input.Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        var hdr = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        N = int.Parse(hdr[0], ci);
        mu = double.Parse(hdr[1], ci);

        var start = ParseVec(lines[1]);
        P = new double[N + 1][];
        V = new double[N + 1][];
        P[0] = new[] { start[0], start[1], start[2] };
        V[0] = new[] { start[3], start[4], start[5] };
        for (int i = 0; i < N; i++)
        {
            var v = ParseVec(lines[2 + i]);
            P[i + 1] = new[] { v[0], v[1], v[2] };
            V[i + 1] = new[] { v[3], v[4], v[5] };
        }

        int M = N + 1;
        double[,] C = new double[M, M];
        for (int i = 0; i < M; i++)
            for (int j = 0; j < M; j++)
                if (i != j) C[i, j] = Cost(i, j);

        var bestTour = NearestNeighbor(C, M);
        double bestLen = TourLength(bestTour, C);

        if (N <= 30)
        {
            for (int s = 1; s <= N && sw.Elapsed.TotalSeconds < 20; s++)
            {
                var t = NearestNeighborFromForcedFirst(C, M, s);
                double L = TourLength(t, C);
                if (L < bestLen) { bestLen = L; bestTour = t; }
            }
        }

        double budget = N <= 20 ? 30.0 : (N <= 50 ? 90.0 : 220.0);
        TwoOpt(bestTour, C, sw, budget);
        OrOpt(bestTour, C, sw, budget + 30);
        TwoOpt(bestTour, C, sw, budget + 60);

        var sbOut = new System.Text.StringBuilder();
        for (int i = 1; i < bestTour.Length; i++)
        {
            if (i > 1) sbOut.Append(' ');
            sbOut.Append(bestTour[i] - 1);
        }
        Console.WriteLine(sbOut.ToString());
    }

    static double[] ParseVec(string line)
    {
        var idx = line.IndexOf(':');
        if (idx >= 0) line = line.Substring(idx + 1);
        var parts = line.Split(new[] { ' ', '\t', ',' }, StringSplitOptions.RemoveEmptyEntries);
        var ci = CultureInfo.InvariantCulture;
        var res = new double[parts.Length];
        for (int i = 0; i < parts.Length; i++) res[i] = double.Parse(parts[i], ci);
        return res;
    }

    static double Cost(int i, int j)
    {
        double dx = P[i][0] - P[j][0];
        double dy = P[i][1] - P[j][1];
        double dz = P[i][2] - P[j][2];
        double d = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double dvx = V[i][0] - V[j][0];
        double dvy = V[i][1] - V[j][1];
        double dvz = V[i][2] - V[j][2];
        double dv = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);
        return d + 500.0 * dv;
    }

    static int[] NearestNeighbor(double[,] C, int M)
    {
        var tour = new int[M];
        var visited = new bool[M];
        tour[0] = 0; visited[0] = true;
        for (int k = 1; k < M; k++)
        {
            int cur = tour[k - 1];
            int best = -1; double bd = double.MaxValue;
            for (int j = 0; j < M; j++)
            {
                if (visited[j]) continue;
                if (C[cur, j] < bd) { bd = C[cur, j]; best = j; }
            }
            tour[k] = best; visited[best] = true;
        }
        return tour;
    }

    static int[] NearestNeighborFromForcedFirst(double[,] C, int M, int firstAfterStart)
    {
        var tour = new int[M];
        var visited = new bool[M];
        tour[0] = 0; visited[0] = true;
        tour[1] = firstAfterStart; visited[firstAfterStart] = true;
        for (int k = 2; k < M; k++)
        {
            int cur = tour[k - 1];
            int best = -1; double bd = double.MaxValue;
            for (int j = 0; j < M; j++)
            {
                if (visited[j]) continue;
                if (C[cur, j] < bd) { bd = C[cur, j]; best = j; }
            }
            tour[k] = best; visited[best] = true;
        }
        return tour;
    }

    static double TourLength(int[] t, double[,] C)
    {
        double s = 0;
        for (int i = 0; i < t.Length - 1; i++) s += C[t[i], t[i + 1]];
        return s;
    }

    static void TwoOpt(int[] tour, double[,] C, Stopwatch sw, double budget)
    {
        int n = tour.Length;
        bool improved = true;
        while (improved && sw.Elapsed.TotalSeconds < budget)
        {
            improved = false;
            for (int i = 1; i < n - 1; i++)
            {
                if (sw.Elapsed.TotalSeconds >= budget) return;
                for (int j = i + 1; j < n; j++)
                {
                    int a = tour[i - 1], b = tour[i], c = tour[j];
                    int d = (j + 1 < n) ? tour[j + 1] : -1;
                    double before = C[a, b] + (d >= 0 ? C[c, d] : 0);
                    double after = C[a, c] + (d >= 0 ? C[b, d] : 0);
                    if (after + 1e-9 < before)
                    {
                        Array.Reverse(tour, i, j - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static void OrOpt(int[] tour, double[,] C, Stopwatch sw, double budget)
    {
        int n = tour.Length;
        bool improved = true;
        int[] segLens = { 1, 2, 3 };
        while (improved && sw.Elapsed.TotalSeconds < budget)
        {
            improved = false;
            foreach (int L in segLens)
            {
                for (int i = 1; i + L <= n; i++)
                {
                    int prev = tour[i - 1];
                    int segStart = tour[i];
                    int segEnd = tour[i + L - 1];
                    int next = (i + L < n) ? tour[i + L] : -1;
                    double removeCost = C[prev, segStart] + (next >= 0 ? C[segEnd, next] : 0)
                                        - (next >= 0 ? C[prev, next] : 0);
                    for (int j = 0; j < n - 1; j++)
                    {
                        if (j >= i - 1 && j < i + L) continue;
                        int a = tour[j];
                        int b = tour[j + 1];
                        if (j + 1 >= i && j + 1 < i + L) continue;
                        double insertCost = C[a, segStart] + C[segEnd, b] - C[a, b];
                        if (insertCost + 1e-9 < removeCost)
                        {
                            var seg = new int[L];
                            Array.Copy(tour, i, seg, 0, L);
                            var tmp = new List<int>(tour);
                            tmp.RemoveRange(i, L);
                            int insIdx = j + 1;
                            if (insIdx > i) insIdx -= L;
                            tmp.InsertRange(insIdx, seg);
                            for (int k = 0; k < n; k++) tour[k] = tmp[k];
                            improved = true;
                            goto restart;
                        }
                    }
                }
            }
        restart:;
            if (sw.Elapsed.TotalSeconds >= budget) return;
        }
    }
}
