// CSG Union of two manifold, watertight meshes.
//
// Strategy:
//  1. Read meshes A,B.
//  2. Build AABB BVHs for both.
//  3. Find candidate intersecting triangle pairs via BVH overlap.
//  4. For each pair compute the triangle-triangle intersection segment
//     (Moller plane-interval method) and accumulate per-triangle constraints.
//  5. For every triangle with constraints, project to 2D (drop dominant
//     normal axis), Delaunay-triangulate corner+intersection points, clip
//     to the original triangle, and emit sub-triangles in 3D.
//  6. Classify each sub-triangle's centroid against the OTHER mesh via parity
//     ray-casting on the other mesh's BVH. Keep only sub-triangles whose
//     centroid is OUTSIDE the other mesh.
//  7. Weld all output vertices on a quantization grid (eps = 1e-9 * diag)
//     so the seam closes and the result is connected/watertight.
//
// Compiles with MSVC /std:c++20.

#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <limits>
#include <unordered_map>
#include <vector>

using std::size_t;

struct V3 {
    double x, y, z;
    V3() : x(0), y(0), z(0) {}
    V3(double a, double b, double c) : x(a), y(b), z(c) {}
    V3 operator+(const V3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    V3 operator-(const V3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    V3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const V3& o) const { return x*o.x + y*o.y + z*o.z; }
    V3 cross(const V3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double norm2() const { return x*x + y*y + z*z; }
    double norm()  const { return std::sqrt(norm2()); }
};

struct Tri { int a, b, c; };

struct AABB {
    V3 mn, mx;
    AABB() {
        const double inf = std::numeric_limits<double>::infinity();
        mn = { inf,  inf,  inf};
        mx = {-inf, -inf, -inf};
    }
    void expand(const V3& p) {
        if (p.x < mn.x) mn.x = p.x; if (p.x > mx.x) mx.x = p.x;
        if (p.y < mn.y) mn.y = p.y; if (p.y > mx.y) mx.y = p.y;
        if (p.z < mn.z) mn.z = p.z; if (p.z > mx.z) mx.z = p.z;
    }
    void expand(const AABB& b) { expand(b.mn); expand(b.mx); }
    bool overlaps(const AABB& b, double eps = 0) const {
        return !(mx.x + eps < b.mn.x || b.mx.x + eps < mn.x ||
                 mx.y + eps < b.mn.y || b.mx.y + eps < mn.y ||
                 mx.z + eps < b.mn.z || b.mx.z + eps < mn.z);
    }
};

struct BVHNode {
    AABB box;
    int left = -1, right = -1;
    int first = 0, count = 0;
};

struct BVH {
    std::vector<BVHNode> nodes;
    std::vector<int> prims;
    const std::vector<V3>*  V = nullptr;
    const std::vector<Tri>* T = nullptr;

    void build(const std::vector<V3>& verts, const std::vector<Tri>& tris) {
        V = &verts; T = &tris;
        prims.resize(tris.size());
        for (size_t i = 0; i < tris.size(); ++i) prims[i] = (int)i;
        nodes.clear();
        nodes.reserve(tris.size() * 2 + 1);
        if (!tris.empty()) buildRec(0, (int)tris.size(), 0);
    }

    AABB triBox(int ti) const {
        const Tri& t = (*T)[ti];
        AABB b;
        b.expand((*V)[t.a]);
        b.expand((*V)[t.b]);
        b.expand((*V)[t.c]);
        return b;
    }

    int buildRec(int begin, int end, int depth) {
        int idx = (int)nodes.size();
        nodes.push_back({});
        AABB box;
        for (int i = begin; i < end; ++i) box.expand(triBox(prims[i]));
        nodes[idx].box = box;
        int n = end - begin;
        if (n <= 4 || depth > 40) {
            nodes[idx].first = begin;
            nodes[idx].count = n;
            return idx;
        }
        V3 ext = box.mx - box.mn;
        int axis = 0;
        if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;
        auto cent = [&](int ti) {
            AABB b = triBox(ti);
            V3 c = (b.mn + b.mx) * 0.5;
            return axis == 0 ? c.x : axis == 1 ? c.y : c.z;
        };
        int mid = begin + n / 2;
        std::nth_element(prims.begin() + begin, prims.begin() + mid,
                         prims.begin() + end,
                         [&](int a, int b) { return cent(a) < cent(b); });
        int l = buildRec(begin, mid, depth + 1);
        int r = buildRec(mid,   end, depth + 1);
        nodes[idx].left  = l;
        nodes[idx].right = r;
        return idx;
    }

    static bool rayBox(const V3& o, const V3& invD, const AABB& b) {
        double t1 = (b.mn.x - o.x) * invD.x;
        double t2 = (b.mx.x - o.x) * invD.x;
        double tmn = std::min(t1, t2), tmx = std::max(t1, t2);
        t1 = (b.mn.y - o.y) * invD.y;
        t2 = (b.mx.y - o.y) * invD.y;
        tmn = std::max(tmn, std::min(t1, t2));
        tmx = std::min(tmx, std::max(t1, t2));
        t1 = (b.mn.z - o.z) * invD.z;
        t2 = (b.mx.z - o.z) * invD.z;
        tmn = std::max(tmn, std::min(t1, t2));
        tmx = std::min(tmx, std::max(t1, t2));
        return tmx >= std::max(tmn, 0.0);
    }

    bool insideByParity(const V3& o, const V3& dir) const {
        if (nodes.empty()) return false;
        V3 invD{ 1.0 / (dir.x != 0 ? dir.x : 1e-300),
                1.0 / (dir.y != 0 ? dir.y : 1e-300),
                1.0 / (dir.z != 0 ? dir.z : 1e-300) };
        int crossings = 0;
        int stk[128]; int sp = 0;
        stk[sp++] = 0;
        while (sp) {
            int ni = stk[--sp];
            const BVHNode& n = nodes[ni];
            if (!rayBox(o, invD, n.box)) continue;
            if (n.left < 0) {
                for (int k = 0; k < n.count; ++k) {
                    int ti = prims[n.first + k];
                    const Tri& tr = (*T)[ti];
                    const V3& A = (*V)[tr.a];
                    const V3& B = (*V)[tr.b];
                    const V3& C = (*V)[tr.c];
                    V3 e1 = B - A, e2 = C - A;
                    V3 p = dir.cross(e2);
                    double det = e1.dot(p);
                    if (std::fabs(det) < 1e-30) continue;
                    double inv = 1.0 / det;
                    V3 s = o - A;
                    double u = s.dot(p) * inv;
                    if (u < 0 || u > 1) continue;
                    V3 q = s.cross(e1);
                    double v = dir.dot(q) * inv;
                    if (v < 0 || u + v > 1) continue;
                    double t = e2.dot(q) * inv;
                    if (t > 1e-12) crossings++;
                }
            } else {
                if (sp + 2 < 128) {
                    stk[sp++] = n.left;
                    stk[sp++] = n.right;
                }
            }
        }
        return (crossings & 1) != 0;
    }
};

// Triangle-Triangle intersection segment (Moller).
// Returns true and fills outP/outQ when triangles intersect on a segment.
static bool triTriSegment(const V3& A0, const V3& A1, const V3& A2,
                          const V3& B0, const V3& B1, const V3& B2,
                          V3& outP, V3& outQ)
{
    V3 nA = (A1 - A0).cross(A2 - A0);
    double dB0 = nA.dot(B0 - A0);
    double dB1 = nA.dot(B1 - A0);
    double dB2 = nA.dot(B2 - A0);
    double epsA = 1e-14 * (1.0 + nA.norm());
    if ((dB0 >  epsA && dB1 >  epsA && dB2 >  epsA) ||
        (dB0 < -epsA && dB1 < -epsA && dB2 < -epsA)) return false;

    V3 nB = (B1 - B0).cross(B2 - B0);
    double dA0 = nB.dot(A0 - B0);
    double dA1 = nB.dot(A1 - B0);
    double dA2 = nB.dot(A2 - B0);
    double epsB = 1e-14 * (1.0 + nB.norm());
    if ((dA0 >  epsB && dA1 >  epsB && dA2 >  epsB) ||
        (dA0 < -epsB && dA1 < -epsB && dA2 < -epsB)) return false;

    V3 D = nA.cross(nB);
    if (D.norm2() < 1e-30) return false; // coplanar - skip

    auto interval = [&](const V3& P0, const V3& P1, const V3& P2,
                        double d0, double d1, double d2,
                        double& tlo, double& thi)
    {
        double p0 = D.dot(P0), p1 = D.dot(P1), p2 = D.dot(P2);
        // unique-sign vertex moves to position A
        const V3* PA = &P0; const V3* PB = &P1; const V3* PC = &P2;
        double dA = d0, dB = d1, dC = d2;
        double pA = p0, pB = p1, pC = p2;
        bool s01 = (d0 > 0) == (d1 > 0);
        bool s02 = (d0 > 0) == (d2 > 0);
        if (s01) { PA = &P2; PB = &P0; PC = &P1; dA=d2; dB=d0; dC=d1; pA=p2; pB=p0; pC=p1; }
        else if (s02) { PA = &P1; PB = &P0; PC = &P2; dA=d1; dB=d0; dC=d2; pA=p1; pB=p0; pC=p2; }
        double tA = pA + (pB - pA) * (dA / (dA - dB));
        double tB = pA + (pC - pA) * (dA / (dA - dC));
        if (tA > tB) std::swap(tA, tB);
        tlo = tA; thi = tB;
        (void)PA; (void)PB; (void)PC;
    };

    double a0, a1, b0, b1;
    interval(A0, A1, A2, dA0, dA1, dA2, a0, a1);
    interval(B0, B1, B2, dB0, dB1, dB2, b0, b1);
    double lo = std::max(a0, b0);
    double hi = std::min(a1, b1);
    if (hi < lo) return false;

    double denom = D.norm2();
    double cA = -nA.dot(A0);
    double cB = -nB.dot(B0);
    V3 base = (D.cross(nB) * cA + nA.cross(D) * cB) * (1.0 / denom);
    double tLo = (lo - D.dot(base)) / denom;
    double tHi = (hi - D.dot(base)) / denom;
    outP = base + D * tLo;
    outQ = base + D * tHi;
    return true;
}

// 2D primitives
struct P2 { double x, y; int id; };
static double orient2d(const P2& a, const P2& b, const P2& c) {
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}
static bool inCircle(const P2& a, const P2& b, const P2& c, const P2& d) {
    double ax = a.x - d.x, ay = a.y - d.y;
    double bx = b.x - d.x, by = b.y - d.y;
    double cx = c.x - d.x, cy = c.y - d.y;
    double det = (ax*ax+ay*ay) * (bx*cy - by*cx)
               - (bx*bx+by*by) * (ax*cy - ay*cx)
               + (cx*cx+cy*cy) * (ax*by - ay*bx);
    return det > 0;
}

// Bowyer-Watson Delaunay (small point sets).
static std::vector<std::array<int,3>> delaunay2D(std::vector<P2> pts) {
    int n = (int)pts.size();
    double mnx = 1e300, mny = 1e300, mxx = -1e300, mxy = -1e300;
    for (auto& p : pts) {
        if (p.x < mnx) mnx = p.x; if (p.x > mxx) mxx = p.x;
        if (p.y < mny) mny = p.y; if (p.y > mxy) mxy = p.y;
    }
    double dm = std::max(mxx - mnx, mxy - mny) * 20 + 1.0;
    double midx = (mnx + mxx) * 0.5, midy = (mny + mxy) * 0.5;
    pts.push_back({midx - dm, midy - dm * 0.5, -1});
    pts.push_back({midx + dm, midy - dm * 0.5, -2});
    pts.push_back({midx,      midy + dm,       -3});
    int s0 = n, s1 = n + 1, s2 = n + 2;

    std::vector<std::array<int,3>> tris;
    tris.push_back({s0, s1, s2});
    for (int i = 0; i < n; ++i) {
        std::vector<int> bad;
        for (int t = 0; t < (int)tris.size(); ++t) {
            auto& tr = tris[t];
            if (inCircle(pts[tr[0]], pts[tr[1]], pts[tr[2]], pts[i]))
                bad.push_back(t);
        }
        std::vector<std::array<int,2>> poly;
        for (int b : bad) {
            auto& tr = tris[b];
            std::array<std::array<int,2>,3> es = {{{tr[0],tr[1]},{tr[1],tr[2]},{tr[2],tr[0]}}};
            for (auto& e : es) {
                bool shared = false;
                for (int b2 : bad) {
                    if (b2 == b) continue;
                    auto& tr2 = tris[b2];
                    std::array<std::array<int,2>,3> es2 = {{{tr2[0],tr2[1]},{tr2[1],tr2[2]},{tr2[2],tr2[0]}}};
                    for (auto& e2 : es2) {
                        if ((e[0]==e2[1] && e[1]==e2[0]) ||
                            (e[0]==e2[0] && e[1]==e2[1])) { shared = true; break; }
                    }
                    if (shared) break;
                }
                if (!shared) poly.push_back(e);
            }
        }
        std::sort(bad.begin(), bad.end(), std::greater<int>());
        for (int b : bad) { tris[b] = tris.back(); tris.pop_back(); }
        for (auto& e : poly) tris.push_back({e[0], e[1], i});
    }
    std::vector<std::array<int,3>> out;
    out.reserve(tris.size());
    for (auto& tr : tris) {
        if (tr[0]==s0||tr[0]==s1||tr[0]==s2) continue;
        if (tr[1]==s0||tr[1]==s1||tr[1]==s2) continue;
        if (tr[2]==s0||tr[2]==s1||tr[2]==s2) continue;
        if (orient2d(pts[tr[0]], pts[tr[1]], pts[tr[2]]) < 0)
            std::swap(tr[1], tr[2]);
        out.push_back(tr);
    }
    return out;
}

struct Mesh {
    std::vector<V3>  V;
    std::vector<Tri> F;
};

static void readMesh(Mesh& m, int nv, int nf) {
    m.V.resize(nv);
    for (int i = 0; i < nv; ++i)
        std::scanf("%lf %lf %lf", &m.V[i].x, &m.V[i].y, &m.V[i].z);
    m.F.resize(nf);
    for (int i = 0; i < nf; ++i)
        std::scanf("%d %d %d", &m.F[i].a, &m.F[i].b, &m.F[i].c);
}

static int dominantAxis(const V3& n) {
    double ax = std::fabs(n.x), ay = std::fabs(n.y), az = std::fabs(n.z);
    if (ax >= ay && ax >= az) return 0;
    if (ay >= az) return 1;
    return 2;
}
static P2 project2D(const V3& p, int axis, int id) {
    if (axis == 0) return {p.y, p.z, id};
    if (axis == 1) return {p.z, p.x, id};
    return            {p.x, p.y, id};
}

int main() {
    int VA, FA, VB, FB;
    if (std::scanf("%d %d %d %d", &VA, &FA, &VB, &FB) != 4) return 0;
    Mesh A, B;
    readMesh(A, VA, FA);
    readMesh(B, VB, FB);

    BVH bvhA, bvhB;
    bvhA.build(A.V, A.F);
    bvhB.build(B.V, B.F);

    AABB world;
    for (auto& v : A.V) world.expand(v);
    for (auto& v : B.V) world.expand(v);
    double diag = (world.mx - world.mn).norm();
    if (diag <= 0) diag = 1.0;
    double snapEps = diag * 1e-9;

    std::vector<std::vector<std::pair<V3,V3>>> segA(A.F.size()), segB(B.F.size());

    // BVH-vs-BVH traversal for triangle pair candidates.
    if (!A.F.empty() && !B.F.empty()) {
        struct Pair { int a, b; };
        std::vector<Pair> stk;
        stk.reserve(1024);
        stk.push_back({0, 0});
        while (!stk.empty()) {
            Pair pr = stk.back(); stk.pop_back();
            const BVHNode& na = bvhA.nodes[pr.a];
            const BVHNode& nb = bvhB.nodes[pr.b];
            if (!na.box.overlaps(nb.box)) continue;
            bool leafA = na.left < 0;
            bool leafB = nb.left < 0;
            if (leafA && leafB) {
                for (int i = 0; i < na.count; ++i) {
                    int ai = bvhA.prims[na.first + i];
                    const Tri& ta = A.F[ai];
                    const V3& A0 = A.V[ta.a]; const V3& A1 = A.V[ta.b]; const V3& A2 = A.V[ta.c];
                    AABB tba; tba.expand(A0); tba.expand(A1); tba.expand(A2);
                    for (int j = 0; j < nb.count; ++j) {
                        int bi = bvhB.prims[nb.first + j];
                        const Tri& tb = B.F[bi];
                        const V3& B0 = B.V[tb.a]; const V3& B1 = B.V[tb.b]; const V3& B2 = B.V[tb.c];
                        AABB tbb; tbb.expand(B0); tbb.expand(B1); tbb.expand(B2);
                        if (!tba.overlaps(tbb)) continue;
                        V3 p, q;
                        if (triTriSegment(A0,A1,A2,B0,B1,B2,p,q)) {
                            if ((p - q).norm2() > snapEps * snapEps * 1e-4) {
                                segA[ai].push_back({p, q});
                                segB[bi].push_back({p, q});
                            }
                        }
                    }
                }
            } else if (leafA) {
                stk.push_back({pr.a, nb.left});
                stk.push_back({pr.a, nb.right});
            } else if (leafB) {
                stk.push_back({na.left,  pr.b});
                stk.push_back({na.right, pr.b});
            } else {
                stk.push_back({na.left,  nb.left});
                stk.push_back({na.left,  nb.right});
                stk.push_back({na.right, nb.left});
                stk.push_back({na.right, nb.right});
            }
        }
    }

    // Vertex welder.
    std::vector<V3> outV;
    outV.reserve(A.V.size() + B.V.size());
    struct K { int64_t x, y, z;
        bool operator==(const K& o) const { return x==o.x && y==o.y && z==o.z; } };
    struct KH { size_t operator()(const K& k) const {
        uint64_t h = (uint64_t)(k.x * 73856093) ^ (uint64_t)(k.y * 19349663)
                   ^ (uint64_t)(k.z * 83492791);
        h ^= h >> 33; h *= 0xff51afd7ed558ccdULL; h ^= h >> 33; return (size_t)h;
    } };
    std::unordered_map<K, int, KH> weld;
    weld.reserve(2 * (A.V.size() + B.V.size()));
    auto quant = [&](const V3& p) {
        double inv = 1.0 / snapEps;
        return K{ (int64_t)std::llround(p.x * inv),
                  (int64_t)std::llround(p.y * inv),
                  (int64_t)std::llround(p.z * inv) };
    };
    auto addVert = [&](const V3& p) {
        K k = quant(p);
        auto it = weld.find(k);
        if (it != weld.end()) return it->second;
        int idx = (int)outV.size();
        outV.push_back(p);
        weld.emplace(k, idx);
        return idx;
    };

    std::vector<Tri> outF;
    outF.reserve(A.F.size() + B.F.size());

    auto processMesh = [&](const Mesh& M,
                           const std::vector<std::vector<std::pair<V3,V3>>>& segs,
                           const BVH& otherBVH,
                           bool otherEmpty) {
        V3 rayDir{0.5773, 0.5774, 0.5775};
        for (size_t fi = 0; fi < M.F.size(); ++fi) {
            const Tri& f = M.F[fi];
            const V3& P0 = M.V[f.a];
            const V3& P1 = M.V[f.b];
            const V3& P2 = M.V[f.c];
            V3 n = (P1 - P0).cross(P2 - P0);
            if (n.norm2() == 0) continue;

            auto classifyKeep = [&](const V3& c) {
                if (otherEmpty) return true;
                return !otherBVH.insideByParity(c, rayDir);
            };

            if (segs[fi].empty()) {
                V3 c = (P0 + P1 + P2) * (1.0 / 3.0);
                if (!classifyKeep(c)) continue;
                int ia = addVert(P0);
                int ib = addVert(P1);
                int ic = addVert(P2);
                if (ia == ib || ib == ic || ic == ia) continue;
                outF.push_back({ia, ib, ic});
                continue;
            }

            int axis = dominantAxis(n);
            bool flip = false;
            {
                P2 a = project2D(P0, axis, 0);
                P2 b = project2D(P1, axis, 1);
                P2 c = project2D(P2, axis, 2);
                if (orient2d(a,b,c) < 0) flip = true;
            }

            std::vector<V3> pts3;
            std::vector<P2> pts2;
            pts3.reserve(3 + 2 * segs[fi].size());
            pts2.reserve(3 + 2 * segs[fi].size());
            auto pushPt = [&](const V3& p) {
                K k = quant(p);
                for (size_t i = 0; i < pts3.size(); ++i) {
                    K k2 = quant(pts3[i]);
                    if (k == k2) return (int)i;
                }
                pts3.push_back(p);
                P2 pp = project2D(p, axis, (int)pts3.size() - 1);
                if (flip) std::swap(pp.x, pp.y);
                pts2.push_back(pp);
                return (int)pts3.size() - 1;
            };
            int i0 = pushPt(P0);
            int i1 = pushPt(P1);
            int i2 = pushPt(P2);
            for (auto& s : segs[fi]) { pushPt(s.first); pushPt(s.second); }

            auto tris2d = delaunay2D(pts2);

            P2 q0 = pts2[i0], q1 = pts2[i1], q2 = pts2[i2];
            double mainArea = orient2d(q0, q1, q2);
            int sgnMain = mainArea > 0 ? 1 : -1;
            auto insideMain = [&](double x, double y) {
                P2 p{x, y, -1};
                double s0 = orient2d(q0, q1, p) * sgnMain;
                double s1 = orient2d(q1, q2, p) * sgnMain;
                double s2 = orient2d(q2, q0, p) * sgnMain;
                double tol = 1e-12 * (std::fabs(mainArea) + 1.0);
                return s0 >= -tol && s1 >= -tol && s2 >= -tol;
            };

            for (auto& t : tris2d) {
                P2 a = pts2[t[0]], b = pts2[t[1]], c = pts2[t[2]];
                double cx = (a.x + b.x + c.x) / 3.0;
                double cy = (a.y + b.y + c.y) / 3.0;
                if (!insideMain(cx, cy)) continue;
                V3 A3 = pts3[t[0]], B3 = pts3[t[1]], C3 = pts3[t[2]];
                V3 cent3 = (A3 + B3 + C3) * (1.0 / 3.0);
                if (!classifyKeep(cent3)) continue;
                V3 nn = (B3 - A3).cross(C3 - A3);
                if (nn.dot(n) < 0) std::swap(B3, C3);
                int ia = addVert(A3);
                int ib = addVert(B3);
                int ic = addVert(C3);
                if (ia == ib || ib == ic || ic == ia) continue;
                outF.push_back({ia, ib, ic});
            }
        }
    };

    processMesh(A, segA, bvhB, B.F.empty());
    processMesh(B, segB, bvhA, A.F.empty());

    std::printf("%zu %zu\n", outV.size(), outF.size());
    for (auto& v : outV)
        std::printf("%.17g %.17g %.17g\n", v.x, v.y, v.z);
    for (auto& f : outF)
        std::printf("%d %d %d\n", f.a, f.b, f.c);
    return 0;
}
