// Adaptive TSP solver for N from 10 to 10,000,000+
// Strategy:
//   - N <= 12         : exact Held-Karp DP
//   - N <= 2000       : grid nearest-neighbor + 2-opt with neighbor list
//   - N <= 200000     : grid nearest-neighbor + 2-opt with neighbor list (smaller K, time-limited)
//   - N >  200000     : Hilbert space-filling curve ordering (+ optional light 2-opt for mid-large)
//
// MSVC /std:c++20 compatible.

#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cmath>
#include <vector>
#include <algorithm>
#include <chrono>
#include <numeric>
#include <string>
#include <cstring>

using namespace std;
using Clock = chrono::steady_clock;

static double elapsed(const Clock::time_point& s) {
    return chrono::duration<double>(Clock::now() - s).count();
}

struct In {
    static constexpr size_t BUF = 1 << 20;
    char buf[BUF];
    size_t pos = 0, len = 0;
    int gc() {
        if (pos >= len) {
            len = fread(buf, 1, BUF, stdin);
            pos = 0;
            if (!len) return -1;
        }
        return (unsigned char)buf[pos++];
    }
    bool readInt(long long& x) {
        int c = gc();
        while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = gc();
        if (c == -1) return false;
        int sign = 1;
        if (c == '-') { sign = -1; c = gc(); }
        x = 0;
        while (c >= '0' && c <= '9') { x = x * 10 + (c - '0'); c = gc(); }
        x *= sign;
        return true;
    }
    bool readDouble(double& x) {
        int c = gc();
        while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = gc();
        if (c == -1) return false;
        int sign = 1;
        if (c == '-') { sign = -1; c = gc(); }
        else if (c == '+') c = gc();
        double val = 0.0;
        while (c >= '0' && c <= '9') { val = val * 10 + (c - '0'); c = gc(); }
        if (c == '.') {
            c = gc();
            double f = 0.0, d = 1.0;
            while (c >= '0' && c <= '9') { f = f * 10 + (c - '0'); d *= 10; c = gc(); }
            val += f / d;
        }
        if (c == 'e' || c == 'E') {
            c = gc();
            int es = 1;
            if (c == '-') { es = -1; c = gc(); } else if (c == '+') c = gc();
            int e = 0;
            while (c >= '0' && c <= '9') { e = e * 10 + (c - '0'); c = gc(); }
            val *= pow(10.0, es * e);
        }
        x = sign * val;
        return true;
    }
};

static inline double dist(const float* X, const float* Y, int a, int b) {
    double dx = X[a] - X[b], dy = Y[a] - Y[b];
    return sqrt(dx * dx + dy * dy);
}

static vector<int> heldKarp(int N, const float* X, const float* Y) {
    vector<vector<double>> D(N, vector<double>(N));
    for (int i = 0; i < N; i++) for (int j = 0; j < N; j++) D[i][j] = dist(X, Y, i, j);
    int M = 1 << N;
    vector<vector<double>> dp(M, vector<double>(N, 1e18));
    vector<vector<int>> par(M, vector<int>(N, -1));
    dp[1][0] = 0;
    for (int mask = 1; mask < M; mask++) {
        if (!(mask & 1)) continue;
        for (int u = 0; u < N; u++) {
            if (!(mask & (1 << u))) continue;
            double cur = dp[mask][u];
            if (cur >= 1e17) continue;
            for (int v = 1; v < N; v++) {
                if (mask & (1 << v)) continue;
                int nm = mask | (1 << v);
                double nd = cur + D[u][v];
                if (nd < dp[nm][v]) { dp[nm][v] = nd; par[nm][v] = u; }
            }
        }
    }
    int full = M - 1;
    double best = 1e18; int bestEnd = -1;
    for (int u = 1; u < N; u++) {
        double v = dp[full][u] + D[u][0];
        if (v < best) { best = v; bestEnd = u; }
    }
    vector<int> tour;
    if (bestEnd < 0) { for (int i = 0; i < N; i++) tour.push_back(i); return tour; }
    int mask = full, u = bestEnd;
    while (u != -1) {
        tour.push_back(u);
        int p = par[mask][u];
        mask ^= (1 << u);
        u = p;
    }
    reverse(tour.begin(), tour.end());
    return tour;
}

struct Grid {
    int gx = 1, gy = 1;
    double minX = 0, minY = 0, cell = 1;
    vector<vector<int>> cells;
    void build(int N, const float* X, const float* Y, int target) {
        double mnx = 1e30, mny = 1e30, mxx = -1e30, mxy = -1e30;
        for (int i = 0; i < N; i++) {
            if (X[i] < mnx) mnx = X[i];
            if (Y[i] < mny) mny = Y[i];
            if (X[i] > mxx) mxx = X[i];
            if (Y[i] > mxy) mxy = Y[i];
        }
        double w = max(1e-9, mxx - mnx), h = max(1e-9, mxy - mny);
        double area = w * h;
        double per = sqrt(area / max(1, target));
        if (per <= 0) per = 1;
        gx = max(1, (int)ceil(w / per));
        gy = max(1, (int)ceil(h / per));
        if ((long long)gx * gy > 4000000) {
            double sc = sqrt((double)gx * gy / 4000000.0);
            gx = max(1, (int)(gx / sc));
            gy = max(1, (int)(gy / sc));
        }
        minX = mnx; minY = mny;
        cell = max(w / gx, h / gy);
        gx = max(1, (int)ceil(w / cell));
        gy = max(1, (int)ceil(h / cell));
        cells.assign((size_t)gx * gy, {});
        for (int i = 0; i < N; i++) {
            int cx = min(gx - 1, max(0, (int)((X[i] - minX) / cell)));
            int cy = min(gy - 1, max(0, (int)((Y[i] - minY) / cell)));
            cells[(size_t)cy * gx + cx].push_back(i);
        }
    }
    inline void coord(double x, double y, int& cx, int& cy) const {
        cx = min(gx - 1, max(0, (int)((x - minX) / cell)));
        cy = min(gy - 1, max(0, (int)((y - minY) / cell)));
    }
};

static vector<int> nnTour(int N, const float* X, const float* Y) {
    Grid g; g.build(N, X, Y, N);
    vector<char> used(N, 0);
    vector<int> tour; tour.reserve(N);
    int cur = 0;
    tour.push_back(0); used[0] = 1;
    int maxR = max(g.gx, g.gy);
    for (int step = 1; step < N; step++) {
        double cx = X[cur], cy = Y[cur];
        int gcx, gcy; g.coord(cx, cy, gcx, gcy);
        int best = -1; double bestD = 1e30;
        int rFound = -1;
        for (int r = 0; r <= maxR; r++) {
            int x0 = max(0, gcx - r), x1 = min(g.gx - 1, gcx + r);
            int y0 = max(0, gcy - r), y1 = min(g.gy - 1, gcy + r);
            for (int y = y0; y <= y1; y++) {
                for (int x = x0; x <= x1; x++) {
                    if (r > 0 && x != gcx - r && x != gcx + r && y != gcy - r && y != gcy + r) continue;
                    const auto& vec = g.cells[(size_t)y * g.gx + x];
                    for (int id : vec) {
                        if (used[id]) continue;
                        double dx = X[id] - cx, dy = Y[id] - cy;
                        double d = dx * dx + dy * dy;
                        if (d < bestD) { bestD = d; best = id; }
                    }
                }
            }
            if (best >= 0 && rFound < 0) rFound = r;
            if (best >= 0 && r >= rFound + 1) break;
        }
        if (best < 0) {
            for (int i = 0; i < N; i++) if (!used[i]) { best = i; break; }
        }
        tour.push_back(best); used[best] = 1; cur = best;
    }
    return tour;
}

static inline void hrot(uint64_t n, uint64_t& x, uint64_t& y, uint64_t rx, uint64_t ry) {
    if (ry == 0) {
        if (rx == 1) { x = n - 1 - x; y = n - 1 - y; }
        uint64_t t = x; x = y; y = t;
    }
}
static uint64_t hilbertD(uint64_t n, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = n / 2; s > 0; s /= 2) {
        uint64_t rx = (x & s) > 0;
        uint64_t ry = (y & s) > 0;
        d += s * s * ((3 * rx) ^ ry);
        hrot(s, x, y, rx, ry);
    }
    return d;
}

static vector<int> hilbertTour(int N, const float* X, const float* Y) {
    double mnx = 1e30, mny = 1e30, mxx = -1e30, mxy = -1e30;
    for (int i = 0; i < N; i++) {
        if (X[i] < mnx) mnx = X[i];
        if (Y[i] < mny) mny = Y[i];
        if (X[i] > mxx) mxx = X[i];
        if (Y[i] > mxy) mxy = Y[i];
    }
    double w = max(1e-9, mxx - mnx), h = max(1e-9, mxy - mny);
    uint64_t order = 1ULL << 20;
    vector<pair<uint64_t, int>> keys(N);
    for (int i = 0; i < N; i++) {
        uint64_t xi = (uint64_t)((double)(X[i] - mnx) / w * (double)(order - 1));
        uint64_t yi = (uint64_t)((double)(Y[i] - mny) / h * (double)(order - 1));
        if (xi >= order) xi = order - 1;
        if (yi >= order) yi = order - 1;
        keys[i] = { hilbertD(order, xi, yi), i };
    }
    sort(keys.begin(), keys.end(),
         [](const pair<uint64_t,int>& a, const pair<uint64_t,int>& b){ return a.first < b.first; });
    vector<int> tour(N);
    for (int i = 0; i < N; i++) tour[i] = keys[i].second;
    int p0 = 0;
    for (int i = 0; i < N; i++) if (tour[i] == 0) { p0 = i; break; }
    if (p0 != 0) rotate(tour.begin(), tour.begin() + p0, tour.end());
    return tour;
}

static void twoOptNeighbors(vector<int>& tour, const float* X, const float* Y, int K,
                            double timeLimit, Clock::time_point start) {
    int N = (int)tour.size();
    if (N < 4) return;

    Grid g; g.build(N, X, Y, N);
    vector<vector<int>> nbr(N);
    {
        vector<pair<double,int>> tmp;
        for (int i = 0; i < N; i++) {
            tmp.clear();
            double cx = X[i], cy = Y[i];
            int gcx, gcy; g.coord(cx, cy, gcx, gcy);
            int r = 1;
            int maxR = max(g.gx, g.gy);
            while ((int)tmp.size() < K + 1 && r <= maxR) {
                tmp.clear();
                int x0 = max(0, gcx - r), x1 = min(g.gx - 1, gcx + r);
                int y0 = max(0, gcy - r), y1 = min(g.gy - 1, gcy + r);
                for (int y = y0; y <= y1; y++) for (int x = x0; x <= x1; x++) {
                    const auto& vec = g.cells[(size_t)y * g.gx + x];
                    for (int id : vec) {
                        if (id == i) continue;
                        double dx = X[id] - cx, dy = Y[id] - cy;
                        tmp.push_back({ dx * dx + dy * dy, id });
                    }
                }
                r++;
                if (elapsed(start) > timeLimit * 0.4) break;
            }
            if ((int)tmp.size() > K) {
                nth_element(tmp.begin(), tmp.begin() + K, tmp.end(),
                            [](const pair<double,int>& a, const pair<double,int>& b){ return a.first < b.first; });
                tmp.resize(K);
            }
            sort(tmp.begin(), tmp.end(),
                 [](const pair<double,int>& a, const pair<double,int>& b){ return a.first < b.first; });
            nbr[i].reserve(tmp.size());
            for (auto& p : tmp) nbr[i].push_back(p.second);
            if (elapsed(start) > timeLimit * 0.6) {
                // bail early on neighbor building, keep what we have
                for (int j = i + 1; j < N; j++) nbr[j].clear();
                break;
            }
        }
    }

    vector<int> pos(N);
    for (int i = 0; i < N; i++) pos[tour[i]] = i;

    auto D = [&](int a, int b) { return dist(X, Y, a, b); };

    bool improved = true;
    int passes = 0;
    while (improved && elapsed(start) < timeLimit) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            if ((i & 8191) == 0 && elapsed(start) >= timeLimit) break;
            int a = tour[i];
            int b = tour[(i + 1) % N];
            double dab = D(a, b);
            for (int c : nbr[a]) {
                if (c == b || c == a) continue;
                double dac = D(a, c);
                if (dac >= dab) break;
                int pc = pos[c];
                int d = tour[(pc + 1) % N];
                if (d == a) continue;
                double dcd = D(c, d);
                double dbd = D(b, d);
                if (dab + dcd > dac + dbd + 1e-12) {
                    int l = (i + 1) % N, r2 = pc;
                    int segLen = (r2 - l + N) % N + 1;
                    if (segLen > N / 2) {
                        l = (pc + 1) % N;
                        r2 = i;
                        segLen = N - segLen;
                    }
                    int li = l, ri = r2;
                    int steps = segLen / 2;
                    for (int s = 0; s < steps; s++) {
                        int tv = tour[li]; tour[li] = tour[ri]; tour[ri] = tv;
                        pos[tour[li]] = li; pos[tour[ri]] = ri;
                        li = (li + 1) % N; ri = (ri - 1 + N) % N;
                    }
                    improved = true;
                    break;
                }
            }
        }
        if (passes > 50) break;
    }
}

int main() {
    auto start = Clock::now();
    In in;
    long long Nll;
    if (!in.readInt(Nll)) return 0;
    int N = (int)Nll;
    if (N <= 0) return 0;

    vector<float> X(N), Y(N);
    for (int i = 0; i < N; i++) {
        double x, y;
        in.readDouble(x); in.readDouble(y);
        X[i] = (float)x; Y[i] = (float)y;
    }

    vector<int> tour;

    if (N == 1) {
        tour = { 0 };
    } else if (N <= 12) {
        tour = heldKarp(N, X.data(), Y.data());
    } else if (N <= 2000) {
        tour = nnTour(N, X.data(), Y.data());
        twoOptNeighbors(tour, X.data(), Y.data(), 20, 30.0, start);
    } else if (N <= 200000) {
        tour = nnTour(N, X.data(), Y.data());
        int K = (N <= 20000) ? 15 : 10;
        double tl = (N <= 20000) ? 60.0 : 150.0;
        twoOptNeighbors(tour, X.data(), Y.data(), K, tl, start);
    } else {
        tour = hilbertTour(N, X.data(), Y.data());
        if (N <= 2000000 && elapsed(start) < 60.0) {
            twoOptNeighbors(tour, X.data(), Y.data(), 5,
                            min(220.0, 260.0 - elapsed(start)), start);
        }
    }

    if (!tour.empty() && tour[0] != 0) {
        int p = 0;
        for (int i = 0; i < (int)tour.size(); i++) if (tour[i] == 0) { p = i; break; }
        rotate(tour.begin(), tour.begin() + p, tour.end());
    }

    string out;
    out.reserve(tour.size() * 8);
    char buf[16];
    for (size_t i = 0; i < tour.size(); i++) {
        int len = snprintf(buf, sizeof(buf), "%d", tour[i]);
        if (i) out.push_back(' ');
        out.append(buf, len);
    }
    out.push_back('\n');
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
