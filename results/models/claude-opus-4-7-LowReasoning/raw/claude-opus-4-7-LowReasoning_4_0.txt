#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int V;
    double e;
    if (scanf("%d %lf", &V, &e) != 2) return 0;
    vector<double> xs(V), ys(V), zs(V);
    double minx=1e300,miny=1e300,minz=1e300;
    double maxx=-1e300,maxy=-1e300,maxz=-1e300;
    for (int i = 0; i < V; i++) {
        scanf("%lf %lf %lf", &xs[i], &ys[i], &zs[i]);
        minx = min(minx, xs[i]); maxx = max(maxx, xs[i]);
        miny = min(miny, ys[i]); maxy = max(maxy, ys[i]);
        minz = min(minz, zs[i]); maxz = max(maxz, zs[i]);
    }

    // Conservative packing: use circumscribed-sphere grid inside the AABB.
    // Regular tetrahedron with edge e has circumradius R = e*sqrt(3/8).
    // Placing tetrahedra centers on a cubic grid with spacing 2*R guarantees
    // no overlap (spheres are tangent). Restrict centers so spheres fit in AABB.
    double R = e * sqrt(3.0/8.0);
    double s = 2.0 * R * 1.0005;

    double lx = minx + R, hx = maxx - R;
    double ly = miny + R, hy = maxy - R;
    double lz = minz + R, hz = maxz - R;

    auto countAxis = [&](double lo, double hi) -> long long {
        if (hi < lo) return 0;
        return (long long)floor((hi - lo) / s) + 1;
    };

    long long nx = countAxis(lx, hx);
    long long ny = countAxis(ly, hy);
    long long nz = countAxis(lz, hz);

    // Cap total output to remain well within 30s and reasonable output size.
    const long long CAP = 1500000;
    long long total = nx * ny * nz;
    while (total > CAP) {
        s *= 1.05;
        nx = countAxis(lx, hx);
        ny = countAxis(ly, hy);
        nz = countAxis(lz, hz);
        total = nx * ny * nz;
    }

    if (nx < 0) nx = 0;
    if (ny < 0) ny = 0;
    if (nz < 0) nz = 0;
    total = nx * ny * nz;

    // Fast buffered output
    static char buf[1 << 16];
    setvbuf(stdout, buf, _IOFBF, sizeof(buf));

    printf("%lld\n", total);
    for (long long iz = 0; iz < nz; iz++) {
        double cz = lz + iz * s;
        for (long long iy = 0; iy < ny; iy++) {
            double cy = ly + iy * s;
            for (long long ix = 0; ix < nx; ix++) {
                double cx = lx + ix * s;
                printf("%.6f %.6f %.6f 0 0 0\n", cx, cy, cz);
            }
        }
    }
    return 0;
}
