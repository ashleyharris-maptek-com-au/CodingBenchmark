use std::io::{self, Read, Write, BufWriter};

struct Solver {
    w: i32,
    h: i32,
    blocked: Vec<bool>,
    visited: Vec<bool>,
    total: usize,
    require_cycle: bool,
    deadline: std::time::Instant,
    timed_out: bool,
}

impl Solver {
    #[inline]
    fn idx(&self, x: i32, y: i32) -> usize { (y * self.w + x) as usize }

    #[inline]
    fn in_bounds(&self, x: i32, y: i32) -> bool {
        x >= 0 && x < self.w && y >= 0 && y < self.h
    }

    #[inline]
    fn is_free(&self, x: i32, y: i32) -> bool {
        let i = self.idx(x, y);
        !self.blocked[i] && !self.visited[i]
    }

    fn degree(&self, x: i32, y: i32) -> i32 {
        let dirs = [(1i32,0i32),(-1,0),(0,1),(0,-1)];
        let mut d = 0;
        for (dx,dy) in dirs {
            let nx = x+dx; let ny = y+dy;
            if self.in_bounds(nx,ny) && self.is_free(nx,ny) { d += 1; }
        }
        d
    }

    fn prune(&self, end_x: i32, end_y: i32, remaining: usize) -> bool {
        if remaining == 0 { return false; }

        let mut dead_ends = 0;
        for y in 0..self.h {
            for x in 0..self.w {
                if !self.is_free(x,y) { continue; }
                let d = self.degree(x,y);
                if d == 0 {
                    let adj = (x-end_x).abs() + (y-end_y).abs() == 1;
                    if !adj { return true; }
                    if remaining > 1 { return true; }
                }
                if d == 1 { dead_ends += 1; }
            }
        }

        if self.require_cycle {
            if dead_ends > 0 { return true; }
        } else {
            if dead_ends > 1 { return true; }
        }

        let mut stack: Vec<(i32,i32)> = Vec::with_capacity(remaining);
        let mut seen = vec![false; (self.w*self.h) as usize];
        let start_i = self.idx(end_x, end_y);
        seen[start_i] = true;
        let dirs = [(1i32,0i32),(-1,0),(0,1),(0,-1)];
        for (dx,dy) in dirs {
            let nx = end_x+dx; let ny = end_y+dy;
            if self.in_bounds(nx,ny) && self.is_free(nx,ny) {
                let i = self.idx(nx,ny);
                if !seen[i] { seen[i] = true; stack.push((nx,ny)); }
            }
        }
        let mut count = 0usize;
        while let Some((x,y)) = stack.pop() {
            count += 1;
            for (dx,dy) in dirs {
                let nx = x+dx; let ny = y+dy;
                if self.in_bounds(nx,ny) && self.is_free(nx,ny) {
                    let i = self.idx(nx,ny);
                    if !seen[i] { seen[i] = true; stack.push((nx,ny)); }
                }
            }
        }
        if count != remaining { return true; }
        false
    }

    fn dfs(&mut self, x: i32, y: i32, depth: usize, coord: &mut Vec<(i32,i32)>) -> bool {
        if self.timed_out { return false; }
        if depth & 0xFFF == 0 {
            if std::time::Instant::now() >= self.deadline {
                self.timed_out = true;
                return false;
            }
        }

        if depth == self.total {
            if self.require_cycle {
                return (x.abs() + y.abs()) == 1;
            }
            return true;
        }

        let dirs = [(1i32,0i32),(-1,0),(0,1),(0,-1)];
        let mut cands: [(i32,i32,i32); 4] = [(0,0,99); 4];
        let mut n = 0;
        for (dx,dy) in dirs {
            let nx = x+dx; let ny = y+dy;
            if self.in_bounds(nx,ny) && self.is_free(nx,ny) {
                let d = self.degree(nx,ny);
                cands[n] = (nx,ny,d);
                n += 1;
            }
        }
        if n == 0 { return false; }

        for i in 1..n {
            let mut j = i;
            while j > 0 && cands[j-1].2 > cands[j].2 {
                cands.swap(j-1, j);
                j -= 1;
            }
        }

        let do_prune = self.total >= 30 && (depth + 8 < self.total);
        let prune_period: usize = if self.total > 400 { 1 } else if self.total > 100 { 2 } else { 4 };

        for i in 0..n {
            let (nx, ny, _) = cands[i];
            let idx = self.idx(nx, ny);
            self.visited[idx] = true;
            coord.push((nx,ny));

            let remaining = self.total - depth - 1;
            let skip = if do_prune && remaining > 0 && (depth % prune_period == 0) {
                self.prune(nx, ny, remaining)
            } else { false };

            if !skip && self.dfs(nx, ny, depth + 1, coord) {
                return true;
            }
            self.visited[idx] = false;
            coord.pop();
            if self.timed_out { return false; }
        }
        false
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let w: i32 = it.next().unwrap().parse().unwrap();
    let h: i32 = it.next().unwrap().parse().unwrap();
    let oc: usize = it.next().unwrap().parse().unwrap();
    let rc: i32 = it.next().unwrap().parse().unwrap();
    let require_cycle = rc != 0;

    let n = (w * h) as usize;
    let mut blocked = vec![false; n];
    for _ in 0..oc {
        let x: i32 = it.next().unwrap().parse().unwrap();
        let y: i32 = it.next().unwrap().parse().unwrap();
        if x >= 0 && x < w && y >= 0 && y < h {
            blocked[(y * w + x) as usize] = true;
        }
    }

    if w <= 0 || h <= 0 || blocked[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let total = n - oc;

    let mut black = 0i64;
    let mut white = 0i64;
    for y in 0..h {
        for x in 0..w {
            if !blocked[(y*w+x) as usize] {
                if (x+y) & 1 == 0 { black += 1; } else { white += 1; }
            }
        }
    }

    if require_cycle {
        if black != white || total < 4 {
            if !(total == 1 && !require_cycle) {
                writeln!(out, "0").unwrap();
                return;
            }
        }
    } else {
        let t = total as i64;
        let need_black = (t + 1) / 2;
        let need_white = t / 2;
        if black != need_black || white != need_white {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    if total == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(280);

    let mut solver = Solver {
        w, h,
        blocked,
        visited: vec![false; n],
        total,
        require_cycle,
        deadline,
        timed_out: false,
    };

    solver.visited[0] = true;
    let mut coord_path: Vec<(i32,i32)> = Vec::with_capacity(total);
    coord_path.push((0,0));

    let found = solver.dfs(0, 0, 1, &mut coord_path);

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    writeln!(out, "{}", coord_path.len()).unwrap();
    for (x,y) in &coord_path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
