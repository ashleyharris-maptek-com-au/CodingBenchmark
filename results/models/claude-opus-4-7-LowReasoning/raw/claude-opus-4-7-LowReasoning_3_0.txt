use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v || u >= n || v >= n { continue; }
        adj[u].push(v as u32);
        adj[v].push(u as u32);
    }
    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    let mut x: Vec<f64> = vec![0.0; n];
    let mut y: Vec<f64> = vec![0.0; n];

    let area = (n as f64).max(1.0);
    let side = area.sqrt() * 10.0;
    let k = side / (n as f64).sqrt().max(1.0);
    let k = k.max(1.0);

    let mut seed: u64 = 0x9E3779B97F4A7C15;
    let mut rnd = || {
        seed ^= seed << 13;
        seed ^= seed >> 7;
        seed ^= seed << 17;
        (seed as f64) / (u64::MAX as f64)
    };

    for i in 0..n {
        x[i] = (rnd() - 0.5) * side;
        y[i] = (rnd() - 0.5) * side;
    }

    let iters: usize = if n <= 50 { 300 }
                else if n <= 200 { 250 }
                else if n <= 1000 { 150 }
                else if n <= 5000 { 80 }
                else if n <= 20000 { 50 }
                else { 30 };

    let mut t = side / 10.0;
    let cool = t / (iters as f64);

    let k2 = k * k;

    let use_grid = n > 200;
    let cell = (2.0 * k).max(1.0);

    let mut dx: Vec<f64> = vec![0.0; n];
    let mut dy: Vec<f64> = vec![0.0; n];

    for _iter in 0..iters {
        for i in 0..n { dx[i] = 0.0; dy[i] = 0.0; }

        let mut minx = f64::INFINITY; let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY; let mut maxy = f64::NEG_INFINITY;
        for i in 0..n {
            if x[i] < minx { minx = x[i]; }
            if y[i] < miny { miny = y[i]; }
            if x[i] > maxx { maxx = x[i]; }
            if y[i] > maxy { maxy = y[i]; }
        }
        if !minx.is_finite() { minx = 0.0; maxx = 0.0; miny = 0.0; maxy = 0.0; }

        if use_grid {
            let w = (((maxx - minx) / cell).ceil() as i64 + 1).max(1) as usize;
            let h = (((maxy - miny) / cell).ceil() as i64 + 1).max(1) as usize;
            let mut grid: Vec<Vec<u32>> = vec![Vec::new(); w * h];
            let mut cx: Vec<i32> = vec![0; n];
            let mut cy: Vec<i32> = vec![0; n];
            for i in 0..n {
                let gx = (((x[i] - minx) / cell) as i64).clamp(0, w as i64 - 1) as usize;
                let gy = (((y[i] - miny) / cell) as i64).clamp(0, h as i64 - 1) as usize;
                cx[i] = gx as i32;
                cy[i] = gy as i32;
                grid[gy * w + gx].push(i as u32);
            }
            for i in 0..n {
                let gx = cx[i] as i64;
                let gy = cy[i] as i64;
                for ay in (gy-1).max(0)..=(gy+1).min(h as i64 - 1) {
                    for ax in (gx-1).max(0)..=(gx+1).min(w as i64 - 1) {
                        let cellv = &grid[ay as usize * w + ax as usize];
                        for &ju in cellv {
                            let j = ju as usize;
                            if j == i { continue; }
                            let mut ddx = x[i] - x[j];
                            let mut ddy = y[i] - y[j];
                            let mut d2 = ddx*ddx + ddy*ddy;
                            if d2 < 1e-6 {
                                ddx = (rnd() - 0.5) * 0.01;
                                ddy = (rnd() - 0.5) * 0.01;
                                d2 = ddx*ddx + ddy*ddy + 1e-6;
                            }
                            let f = k2 / d2;
                            dx[i] += ddx * f;
                            dy[i] += ddy * f;
                        }
                    }
                }
            }
        } else {
            for i in 0..n {
                for j in (i+1)..n {
                    let mut ddx = x[i] - x[j];
                    let mut ddy = y[i] - y[j];
                    let mut d2 = ddx*ddx + ddy*ddy;
                    if d2 < 1e-6 {
                        ddx = (rnd() - 0.5) * 0.01;
                        ddy = (rnd() - 0.5) * 0.01;
                        d2 = ddx*ddx + ddy*ddy + 1e-6;
                    }
                    let f = k2 / d2;
                    dx[i] += ddx * f;
                    dy[i] += ddy * f;
                    dx[j] -= ddx * f;
                    dy[j] -= ddy * f;
                }
            }
        }

        for u in 0..n {
            for &vu in &adj[u] {
                let v = vu as usize;
                if v <= u { continue; }
                let ddx = x[u] - x[v];
                let ddy = y[u] - y[v];
                let d2 = ddx*ddx + ddy*ddy;
                let d = d2.sqrt().max(1e-6);
                let f = d2 / k;
                let fx = ddx / d * f;
                let fy = ddy / d * f;
                dx[u] -= fx;
                dy[u] -= fy;
                dx[v] += fx;
                dy[v] += fy;
            }
        }

        for i in 0..n {
            let d = (dx[i]*dx[i] + dy[i]*dy[i]).sqrt().max(1e-6);
            let lim = d.min(t);
            x[i] += dx[i] / d * lim;
            y[i] += dy[i] / d * lim;
        }

        t = (t - cool).max(k * 0.01);
    }

    writeln!(out, "{}", n).ok();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).ok();
    }
}
