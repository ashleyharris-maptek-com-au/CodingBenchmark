#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self { Self { x, y } }
    fn len(&self) -> f64 { (self.x * self.x + self.y * self.y).sqrt() }
    fn normalize(&self) -> Self {
        let l = self.len();
        if l > 1e-6 { Self { x: self.x / l, y: self.y / l } } else { *self }
    }
    fn dist(&self, other: &Vec2) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        (dx*dx + dy*dy).sqrt()
    }
}

#[derive(Copy, Clone, PartialEq)]
struct Node {
    cost: f64,
    est_total: f64,
    idx: usize,
}

impl Eq for Node {}

impl Ord for Node {
    fn cmp(&self, other: &Self) -> Ordering {
        other.est_total.partial_cmp(&self.est_total).unwrap_or(Ordering::Equal)
    }
}

impl PartialOrd for Node {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }

fn find_path(map: &MapGrid, start: Vec2, target: Vec2) -> Vec<Vec2> {
    let start_idx = (start.y as usize) * map.width + (start.x as usize);
    let target_idx = (target.y as usize) * map.width + (target.x as usize);
    
    // Limits to prevent OOM/Timeout on massive maps if path is complex
    // On 64-core 128GB machine, we can go deep, but for real-time start, strict limit.
    let max_nodes = 5_000_000;
    
    let mut dist = vec![f64::INFINITY; map.cells.len()];
    let mut came_from = vec![usize::MAX; map.cells.len()];
    let mut heap = BinaryHeap::new();

    dist[start_idx] = 0.0;
    heap.push(Node {
        cost: 0.0,
        est_total: start.dist(&target),
        idx: start_idx,
    });

    let mut visited_count = 0;

    while let Some(Node { cost, idx, .. }) = heap.pop() {
        visited_count += 1;
        if idx == target_idx || visited_count > max_nodes {
            break;
        }

        if cost > dist[idx] { continue; }

        let cx = (idx % map.width) as i32;
        let cy = (idx / map.width) as i32;

        // 8-way connectivity
        for dy in -1..=1 {
            for dx in -1..=1 {
                if dx == 0 && dy == 0 { continue; }
                let nx = cx + dx;
                let ny = cy + dy;

                if nx >= 0 && nx < map.width as i32 && ny >= 0 && ny < map.height as i32 {
                    let nidx = (ny as usize) * map.width + (nx as usize);
                    if map.cells[nidx] { continue; }

                    let step_cost = if dx == 0 || dy == 0 { 1.0 } else { 1.414 };
                    // Penalty for being next to walls (safety margin)
                    // This is expensive to check perfectly, so we skip for speed or do simple check
                    let new_cost = cost + step_cost;

                    if new_cost < dist[nidx] {
                        dist[nidx] = new_cost;
                        came_from[nidx] = idx;
                        let h = ((nx as f64 - target.x).powi(2) + (ny as f64 - target.y).powi(2)).sqrt();
                        heap.push(Node {
                            cost: new_cost,
                            est_total: new_cost + h,
                            idx: nidx,
                        });
                    }
                }
            }
        }
    }

    // Reconstruct
    let mut path = Vec::new();
    let mut curr = target_idx;
    if dist[curr] == f64::INFINITY {
        // Pathfinding failed (e.g. enclosed or timeout), return direct target
        return vec![target];
    }

    while curr != usize::MAX {
        let x = (curr % map.width) as f64 + 0.5;
        let y = (curr / map.width) as f64 + 0.5;
        path.push(Vec2::new(x, y));
        if curr == start_idx { break; }
        curr = came_from[curr];
    }
    path.reverse();
    path
}

            v_desired.x = v_desired.x.clamp(-5.0, 5.0);

            // dy is usually negative if we are above? No, Y increases downward.
            // If we are above, pos.y < target.y. dy > 0.
            // We want positive velocity (downward) to approach.
            // Close to ground (dy small), we want small vel.
            let desired_descent = if dy < 5.0 { 0.5 } else { 2.0 };
            v_desired.y = desired_descent;
        } else {
            // Cruise logic: fly towards waypoint
            let mut dir = Vec2::new(target_wp.x - current_state.pos.x, target_wp.y - current_state.pos.y);
            let dist = dir.len();
            dir = dir.normalize();
            
            let cruise_speed = 30.0;
            // Slow down for turns or end of path
            let speed_ref = cruise_speed.min(dist * 2.0).max(5.0);
            v_desired.x = dir.x * speed_ref;
            v_desired.y = dir.y * speed_ref;

        // Acceleration Control
        // F = m*a. a_req = k_p * (v_des - v). 
        // Thrust_vec + Gravity_vec = a_req
        // Thrust_vec = a_req - Gravity_vec
        // G is (0, g). -G is (0, -g)
        
        let a_x = k_vel_p * (v_desired.x - current_state.vel.x);
        let a_y = k_vel_p * (v_desired.y - current_state.vel.y);

        let req_force_y = a_y - config.gravity; // To counteract gravity G, we need -G force component
        
        // Convert force vector to Thrust Magnitude and Angle
        // Force applied in direction of ship angle (Up-vector rotated)
        // Ship Angle 0 => Up (-Y). Force Vector (-sin(0), -cos(0)) * T = (0, -T)
        // We want (req_fx, req_fy).
        // T = sqrt(fx^2 + fy^2)
        // angle = atan2(fx, -fy)
        
        let t_req = (req_force_x.powi(2) + req_force_y.powi(2)).sqrt();
        let mut thrust_cmd = t_req / config.max_thrust;
        
        let angle_target = req_force_x.atan2(-req_force_y);
        
        // Clamp thrust
        if thrust_cmd > 1.0 { thrust_cmd = 1.0; }
        if thrust_cmd < 0.0 { thrust_cmd = 0.0; }

        // Safety override: If landing and very close, ensure upright
        let target_angle_clamped = if is_landing && dist_to_target < 5.0 {
            0.0 
        } else {
            angle_target.clamp(-1.0, 1.0) // Don't tilt too crazy
        };

        // Angular Control (PD)
        let mut angle_err = target_angle_clamped - current_state.angle;
        // Normalize angle error to -PI..PI
        while angle_err > std::f64::consts::PI { angle_err -= 2.0 * std::f64::consts::PI; }
        while angle_err < -std::f64::consts::PI { angle_err += 2.0 * std::f64::consts::PI; }

        let turn_cmd_raw = k_ang_p * angle_err - k_ang_d * current_state.ang_vel;
        let turn_cmd = turn_cmd_raw.clamp(-1.0, 1.0);

        // 3. Output
        // "thrust turn"
        println!("{:.4} {:.4}", thrust_cmd, turn_cmd);
        io::stdout().flush().ok();

        // Sleep remainder
        let elapsed = loop_start.elapsed();
        if elapsed < dt_target {
            thread::sleep(dt_target - elapsed);
        }
    }

                        heap.push(Node {
                            cost: new_cost,
                            est_total: new_cost + h,
                            idx: nidx,
                        });
                    }
                }
            }
        }
    }

    // Reconstruct
    let mut path = Vec::new();
    let mut curr = target_idx;
    if dist[curr] == f64::INFINITY {
        // Pathfinding failed (e.g. enclosed or timeout), return direct target
        return vec![target];
    }

    while curr != usize::MAX {
        let x = (curr % map.width) as f64 + 0.5;
        let y = (curr / map.width) as f64 + 0.5;
        path.push(Vec2::new(x, y));
        if curr == start_idx { break; }
        curr = came_from[curr];
    }
    path.reverse();
    path
}

// --- Main Control Logic ---

fn main() {
    // 1. Initialization and Map Parsing
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    // Read Header
    let mut buffer = String::new();
    handle.read_line(&mut buffer).unwrap(); // First line
    let params: Vec<f64> = buffer.split_whitespace().map(|s| s.parse().unwrap()).collect();
    
    // If params are split across lines, this simple parse might fail, but prompt format implies structure.
    // Handle case where start/target are on second line as per prompt description
    // Input: "width height ... fuel \n start_x ... target_y \n map..."
    let (width, height, gravity, max_thrust, fuel) = (
        params[0] as usize, params[1] as usize, params[2], params[3], params[4]
    );
    
    buffer.clear();
    handle.read_line(&mut buffer).unwrap();
    let coords: Vec<f64> = buffer.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let start = Vec2::new(coords[0], coords[1]);
    let target = Vec2::new(coords[2], coords[3]);

    let config = GameConfig {
        width, height, gravity, max_thrust, fuel, start, target
    };

    // Read Map
    // Optimizing for speed: Read lines, push to bool vec
    let mut cells = Vec::with_capacity(width * height);
    for _ in 0..height {
        buffer.clear();
        handle.read_line(&mut buffer).unwrap();
        for c in buffer.trim().chars() {
            cells.push(c == '#');
        }
    }

    let map = MapGrid { width, height, cells };

    // 2. Pathfinding (Blocking, before state updates start)
    let path = find_path(&map, start, target);
    
    // Shared State
    let state = Arc::new(Mutex::new(ShipState {
        pos: start, vel: Vec2::new(0.0, 0.0), angle: 0.0, ang_vel: 0.0,
        fuel,
        timestamp: Instant::now(),
    }));

    // 3. Spawn Input Thread
    let state_clone = state.clone();
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut handle = stdin.lock();
        let mut buf = String::new();
        loop {
            buf.clear();
            match handle.read_line(&mut buf) {
                Ok(0) => break, // EOF
                Ok(_) => {
                    let parts: Vec<f64> = buf.split_whitespace()
                        .filter_map(|s| s.parse().ok())
                        .collect();
                    if parts.len() >= 7 {
                        let mut s = state_clone.lock().unwrap();
                        s.pos.x = parts[0];
                        s.pos.y = parts[1];
                        s.vel.x = parts[2];
                        s.vel.y = parts[3];
                        s.angle = parts[4];
                        s.ang_vel = parts[5];
                        s.fuel = parts[6];
                        s.timestamp = Instant::now();
                    }
                }
                Err(_) => break,
            }
        }
    });

    // 4. Control Loop (Main Thread)
    // Run at ~60Hz to be safe above 20Hz req
    let dt_target = Duration::from_millis(16);
    let mut path_idx = 0;
    
    // PID Controllers
    // Angle PD
    let k_ang_p = 4.0;
    let k_ang_d = 2.0;
    
    // Velocity PIDs
    let k_vel_p = 2.0;
    
    loop {
        let loop_start = Instant::now();

        // 1. Get State & Dead Reckoning
        let current_state = {
            let s = state.lock().unwrap();
            let now = Instant::now();
            let dt = (now - s.timestamp).as_secs_f64();
            
            // Simple prediction to smooth out lag/jitters
            let mut pred = *s;
            if dt > 0.0 && dt < 1.0 {
                pred.pos.x += pred.vel.x * dt;
                pred.pos.y += pred.vel.y * dt;
                // Simple gravity integration (ignoring thrust for prediction to be safe)
                pred.vel.y += config.gravity * dt;
                pred.angle += pred.ang_vel * dt;
            }
            pred
        };

        // 2. Logic
        
        // Find current target waypoint
        // Advance path index if close to current waypoint
        while path_idx < path.len() - 1 {
            if current_state.pos.dist(&path[path_idx]) < 5.0 {
                path_idx += 1;
            } else {
                break;
            }
        }
        
        let target_wp = if path_idx < path.len() { path[path_idx] } else { config.target };
        let dist_to_target = current_state.pos.dist(&config.target);
        let is_landing = dist_to_target < 20.0 || path_idx >= path.len() - 1;

        // Desired Velocity Vector
        let mut v_desired = Vec2::new(0.0, 0.0);
        if is_landing {
            // Landing logic: align with target x, descend slowly
            let dx = config.target.x - current_state.pos.x;
            v_desired.x = dx * 1.0; // P-control for X position
            // Clamp horizontal speed near ground
            v_desired.x = v_desired.x.clamp(-5.0, 5.0);
            
            // Vertical: descend at 1.5 m/s, slow to 0.5 near ground
            let dy = config.target.y - current_state.pos.y;
            // dy is usually negative if we are above? No, Y increases downward.
            // If we are above, pos.y < target.y. dy > 0.
            // We want positive velocity (downward) to approach.
            // Close to ground (dy small), we want small vel.
            let desired_descent = if dy < 5.0 { 0.5 } else { 2.0 };
            v_desired.y = desired_descent;
        } else {
            // Cruise logic: fly towards waypoint
            let mut dir = Vec2::new(target_wp.x - current_state.pos.x, target_wp.y - current_state.pos.y);
            let dist = dir.len();
            dir = dir.normalize();
            
            let cruise_speed = 30.0;
            // Slow down for turns or end of path
            let speed_ref = cruise_speed.min(dist * 2.0).max(5.0);
            v_desired.x = dir.x * speed_ref;
            v_desired.y = dir.y * speed_ref;
        }

        // Acceleration Control
        // F = m*a. a_req = k_p * (v_des - v). 
        // Thrust_vec + Gravity_vec = a_req
        // Thrust_vec = a_req - Gravity_vec
        // G is (0, g). -G is (0, -g)
        
        let a_x = k_vel_p * (v_desired.x - current_state.vel.x);
        let a_y = k_vel_p * (v_desired.y - current_state.vel.y);
        
        let req_force_x = a_x;
        let req_force_y = a_y - config.gravity; // To counteract gravity G, we need -G force component
        
        // Convert force vector to Thrust Magnitude and Angle
        // Force applied in direction of ship angle (Up-vector rotated)
        // Ship Angle 0 => Up (-Y). Force Vector (-sin(0), -cos(0)) * T = (0, -T)
        // We want (req_fx, req_fy).
        // T = sqrt(fx^2 + fy^2)
        // angle = atan2(fx, -fy)
        
        let t_req = (req_force_x.powi(2) + req_force_y.powi(2)).sqrt();
        let mut thrust_cmd = t_req / config.max_thrust;
        
        let angle_target = req_force_x.atan2(-req_force_y);
        
        // Clamp thrust
        if thrust_cmd > 1.0 { thrust_cmd = 1.0; }
        if thrust_cmd < 0.0 { thrust_cmd = 0.0; }

        // Safety override: If landing and very close, ensure upright
        let target_angle_clamped = if is_landing && dist_to_target < 5.0 {
            0.0 
        } else {
            angle_target.clamp(-1.0, 1.0) // Don't tilt too crazy
        };

        // Angular Control (PD)
        let mut angle_err = target_angle_clamped - current_state.angle;
        // Normalize angle error to -PI..PI
        while angle_err > std::f64::consts::PI { angle_err -= 2.0 * std::f64::consts::PI; }
        while angle_err < -std::f64::consts::PI { angle_err += 2.0 * std::f64::consts::PI; }

        let turn_cmd_raw = k_ang_p * angle_err - k_ang_d * current_state.ang_vel;
        let turn_cmd = turn_cmd_raw.clamp(-1.0, 1.0);

        // 3. Output
        // "thrust turn"
        println!("{:.4} {:.4}", thrust_cmd, turn_cmd);
        io::stdout().flush().ok();

        // Sleep remainder
        let elapsed = loop_start.elapsed();
        if elapsed < dt_target {
            thread::sleep(dt_target - elapsed);
        }
    }
}
