using namespace std;

static const int IO_BUF_SIZE = 1 << 24;
char in_buf[IO_BUF_SIZE];
char out_buf[IO_BUF_SIZE];

struct FastIO {
    int buf_pos = 0;
    int buf_len = 0;
    
    inline char get_char() {
        if (buf_pos == buf_len) {
            buf_pos = 0;
            buf_len = std::fread(in_buf, 1, IO_BUF_SIZE, stdin);
            if (buf_len == 0) return EOF;
        }
        return in_buf[buf_pos++];
    }
    
    inline int read_int() {
        char c = get_char();
        while (c < '0' || c > '9') {
            if (c == EOF) return -1;
            c = get_char();
        }
        int val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = get_char();
        }
        return val;
    }
    
    inline float read_float() {
        char c = get_char();
        bool neg = false;
        while ((c < '0' || c > '9') && c != '-' && c != '.') {
            if (c == EOF) return 0.0f;
            c = get_char();
        }
        if (c == '-') {
            neg = true;
            c = get_char();
        }
        float val = 0.0f;
        while (c >= '0' && c <= '9') {
            val = val * 10.0f + (c - '0');
            c = get_char();
        }
        if (c == '.') {
            c = get_char();
            float fraction = 0.1f;
            while (c >= '0' && c <= '9') {
                val += (c - '0') * fraction;
                fraction *= 0.1f;
                c = get_char();
            }
        }
        if (c == 'e' || c == 'E') {
            c = get_char();
            bool exp_neg = false;
            if (c == '-') {
                exp_neg = true;
                c = get_char();
            } else if (c == '+') {
                c = get_char();
            }
            int exp_val = 0;
            while (c >= '0' && c <= '9') {
                exp_val = exp_val * 10 + (c - '0');
                c = get_char();
            }
            if (exp_neg) {
                val *= (float)std::pow(10.0, -exp_val);
            } else {
                val *= (float)std::pow(10.0, exp_val);
            }
        }
        return neg ? -val : val;
    }
};

struct FastOut {
    int buf_pos = 0;
    
    inline void flush() {
        if (buf_pos > 0) {
            std::fwrite(out_buf, 1, buf_pos, stdout);
            buf_pos = 0;
        }
    }
    
    inline void write_int(int val) {
        if (val == 0) {
            if (buf_pos + 2 > IO_BUF_SIZE) flush();
            out_buf[buf_pos++] = '0';
            out_buf[buf_pos++] = ' ';
            return;
        }
        char temp[12];
        int pos = 0;
        while (val > 0) {
            temp[pos++] = '0' + (val % 10);
            val /= 10;
        }
        if (buf_pos + pos + 1 > IO_BUF_SIZE) flush();
        while (pos > 0) {
            out_buf[buf_pos++] = temp[--pos];
        }
        out_buf[buf_pos++] = ' ';
    }
};

FastIO fastIO;
FastOut fastOut;

struct HNode {
    uint64_t h;
    int id;
    bool operator<(const HNode& other) const {
        return h < other.h;
    }
};

uint64_t hilbert_code(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    uint32_t rx, ry;
    for (int s = 31; s >= 0; s--) {
        rx = (x >> s) & 1;
        ry = (y >> s) & 1;
        uint64_t quad = ((3ULL * rx) ^ ry);
        d = (d << 2) | quad;
        
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

uint32_t float_to_ordered_int(float f) {
    uint32_t ui;
    std::memcpy(&ui, &f, sizeof(float));
    if (ui & 0x80000000) {
        ui = ~ui;
    } else {
        ui |= 0x80000000;
    }
    return ui;
}

inline float dist(float x1, float y1, float x2, float y2) {
    float dx = x1 - x2;
    float dy = y1 - y2;
    return std::sqrt(dx*dx + dy*dy);
}

void optimize_2opt(vector<int>& tour, const float* x_arr, const float* y_arr, int window, double time_limit, chrono::time_point<chrono::high_resolution_clock> start_time) {
    int n = tour.size();
    bool improved = true;
    int max_iters = 50;
    int iter = 0;
    
    while (improved && iter++ < max_iters) {
        improved = false;
        
        auto now = chrono::high_resolution_clock::now();
        double elapsed = chrono::duration_cast<chrono::duration<double>>(now - start_time).count();
        if (elapsed > time_limit) break;
        
        for (int i = 0; i < n - 1; ++i) {
            int max_j = min(n - 1, i + window);
            for (int j = i + 2; j <= max_j; ++j) {
                int a = tour[i];
                int b = tour[i+1];
                int c = tour[j];
                int d = tour[(j+1)%n];
                
                float d1 = dist(x_arr[a], y_arr[a], x_arr[b], y_arr[b]) + dist(x_arr[c], y_arr[c], x_arr[d], y_arr[d]);
                float d2 = dist(x_arr[a], y_arr[a], x_arr[c], y_arr[c]) + dist(x_arr[b], y_arr[b], x_arr[d], y_arr[d]);
                
                if (d2 < d1) {
                    std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

int main() {
    auto start_time = chrono::high_resolution_clock::now();
    
    int N = fastIO.read_int();
    if (N <= 0) return 0;
    
    if (N >= 100000000) {
        HNode* nodes = new HNode[N];
        for (int i = 0; i < N; ++i) {
            float x = fastIO.read_float();
            float y = fastIO.read_float();
            uint32_t ix = float_to_ordered_int(x);
            uint32_t iy = float_to_ordered_int(y);
            nodes[i].h = hilbert_code(ix, iy);
            nodes[i].id = i;
        }
        std::sort(std::execution::par_unseq, nodes, nodes + N);
        
        int start_idx = 0;
        for (int i = 0; i < N; ++i) {
            if (nodes[i].id == 0) {
                start_idx = i;
                break;
            }
        }
        for (int i = start_idx; i < N; ++i) fastOut.write_int(nodes[i].id);
        for (int i = 0; i < start_idx; ++i) fastOut.write_int(nodes[i].id);
        
        delete[] nodes;
        fastOut.flush();
        return 0;
    }
    
    float* x_arr = new float[N];
    float* y_arr = new float[N];
    float min_x = 1e30f, max_x = -1e30f;
    float min_y = 1e30f, max_y = -1e30f;
    for (int i = 0; i < N; ++i) {
        x_arr[i] = fastIO.read_float();
        y_arr[i] = fastIO.read_float();
        if (x_arr[i] < min_x) min_x = x_arr[i];
        if (x_arr[i] > max_x) max_x = x_arr[i];
        if (y_arr[i] < min_y) min_y = y_arr[i];
        if (y_arr[i] > max_y) max_y = y_arr[i];
    }
    
    double dx = max_x - min_x;
    double dy = max_y - min_y;
    double scale = (dx > dy ? dx : dy);
    if (scale == 0.0) scale = 1.0;
    
    HNode* nodes = new HNode[N];
    for (int i = 0; i < N; ++i) {
        double nx_d = (x_arr[i] - min_x) / scale * 4294967295.0;
        double ny_d = (y_arr[i] - min_y) / scale * 4294967295.0;
        uint32_t nx = (uint32_t)(std::min(nx_d, 4294967295.0));
        uint32_t ny = (uint32_t)(std::min(ny_d, 4294967295.0));
        nodes[i].h = hilbert_code(nx, ny);
        nodes[i].id = i;
    }
    
    std::sort(std::execution::par_unseq, nodes, nodes + N);
    
    vector<int> tour(N);
    for (int i = 0; i < N; ++i) tour[i] = nodes[i].id;
    delete[] nodes;
    
    int window = 0;
    if (N <= 5000) window = N;
    else if (N <= 20000) window = 500;
    else if (N <= 100000) window = 100;
    else if (N <= 1000000) window = 50;
    else if (N <= 10000000) window = 20;
    else window = 10;
    
    if (N <= 1000) {
        vector<int> best_tour = tour;
        float best_len = 1e30f;
        
        auto get_len = [&](const vector<int>& t) {
            float l = 0;
            for (int i=0; i<N; ++i) {
                int a = t[i];
                int b = t[(i+1)%N];
                l += dist(x_arr[a], y_arr[a], x_arr[b], y_arr[b]);
            }
            return l;
        };
        
        best_len = get_len(tour);
        auto local_start = chrono::high_resolution_clock::now();
        srand(42);
        
        while (true) {
            auto now = chrono::high_resolution_clock::now();
            double el = chrono::duration_cast<chrono::duration<double>>(now - local_start).count();
            if (el > 0.5) break; 
            
            vector<int> current_tour = tour;
            for(int i = N - 1; i > 0; --i) {
                int j = rand() % (i + 1);
                std::swap(current_tour[i], current_tour[j]);
            }
            
            optimize_2opt(current_tour, x_arr, y_arr, N, 0.5, local_start);
            
            float current_len = get_len(current_tour);
            if (current_len < best_len) {
                best_len = current_len;
                best_tour = current_tour;
            }
        }
        tour = best_tour;
    } else if (window > 0) {
        optimize_2opt(tour, x_arr, y_arr, window, 250.0, start_time);
    }
    
    int start_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (tour[i] == 0) {
            start_idx = i;
            break;
        }
    }
    
    for (int i = start_idx; i < N; ++i) fastOut.write_int(tour[i]);
    for (int i = 0; i < start_idx; ++i) fastOut.write_int(tour[i]);
    
    fastOut.flush();
    delete[] x_arr;
    delete[] y_arr;
    return 0;
}
