using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    class Edge
    {
        public int To;
        public int Id;
        public Edge(int to, int id) { To = to; Id = id; }
    }

    struct EdgeData : IComparable<EdgeData>
    {
        public int u, v, w;
        public int CompareTo(EdgeData other) => w.CompareTo(other.w);
    }

    class DSU
    {
        int[] parent;
        public DSU(int n)
        {
            parent = new int[n];
            for (int i = 0; i < n; i++) parent[i] = i;
        }
        public int Find(int i)
        {
            if (parent[i] == i) return i;
            return parent[i] = Find(parent[i]);
        }
        public bool Union(int i, int j)
        {
            int rootI = Find(i);
            int rootJ = Find(j);
            if (rootI != rootJ)
            {
                parent[rootI] = rootJ;
                return true;
            }
            return false;
        }
    }

    class Scanner
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 64];
        private int head, tail;

        public Scanner(Stream s)
        {
            stream = s;
        }

        public int NextInt()
        {
            int c = Read();
            while (c <= ' ')
            {
                if (c == -1) return -1;
                c = Read();
            }
            int res = 0;
            bool neg = false;
            if (c == '-')
            {
                neg = true;
                c = Read();
            }
            while (c > ' ')
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return neg ? -res : res;
        }

        private int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }

    static List<Edge>[] multigraph;
    static List<bool> usedEdges;
    static int nextEdgeId = 0;

    static void AddEdgeToMultigraph(int u, int v)
    {
        multigraph[u].Add(new Edge(v, nextEdgeId));
        multigraph[v].Add(new Edge(u, nextEdgeId));
        usedEdges.Add(false);
        nextEdgeId++;
    }

    static void Main(string[] args)
    {
        using var stream = Console.OpenStandardInput();
        var scanner = new Scanner(stream);

        int N = scanner.NextInt();
        if (N == -1) return; // EOF
        int M = scanner.NextInt();

        multigraph = new List<Edge>[N];
        List<(int to, int w)>[] adj = new List<(int, int)>[N];
        for (int i = 0; i < N; i++)
        {
            multigraph[i] = new List<Edge>();
            adj[i] = new List<(int, int)>();
        }

        usedEdges = new List<bool>(M);
        int[] deg = new int[N];
        EdgeData[] edges = new EdgeData[M];

        for (int i = 0; i < M; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt();
            edges[i] = new EdgeData { u = u, v = v, w = w };
            
            AddEdgeToMultigraph(u, v);
            adj[u].Add((v, w));
            adj[v].Add((u, w));
            deg[u]++;
            deg[v]++;
        }

        if (M == 0)
        {
            Console.WriteLine("1\n0");
            return;
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (deg[i] % 2 != 0) oddNodes.Add(i);
        }

        int K = oddNodes.Count;

        if (K > 0)
        {
            long memoryRequired = (long)K * N * 8L;
            long computeRequired = (long)K * M;

            if (K <= 16)
            {
                // EXACT DP
                int[][] distMatrix = new int[K][];
                int[][] prevMatrix = new int[K][];
                var pq = new PriorityQueue<int, int>();

                for (int i = 0; i < K; i++)
                {
                    distMatrix[i] = new int[N];
                    prevMatrix[i] = new int[N];
                    Array.Fill(distMatrix[i], int.MaxValue / 2);
                    Array.Fill(prevMatrix[i], -1);

                    int start = oddNodes[i];
                    distMatrix[i][start] = 0;
                    pq.Enqueue(start, 0);

                    while (pq.Count > 0)
                    {
                        pq.TryDequeue(out int u, out int d);
                        if (d > distMatrix[i][u]) continue;

                        foreach (var edge in adj[u])
                        {
                            int nxt = edge.to;
                            int nxtD = d + edge.w;
                            if (nxtD < distMatrix[i][nxt])
                            {
                                distMatrix[i][nxt] = nxtD;
                                prevMatrix[i][nxt] = u;
                                pq.Enqueue(nxt, nxtD);
                            }
                        }
                    }
                }

                int[,] distK = new int[K, K];
                for (int i = 0; i < K; i++)
                {
                    for (int j = 0; j < K; j++)
                    {
                        distK[i, j] = distMatrix[i][oddNodes[j]];
                    }
                }

                int[] dp = new int[1 << K];
                int[] parentMask = new int[1 << K];
                Array.Fill(dp, int.MaxValue / 2);
                dp[0] = 0;

                for (int mask = 0; mask < (1 << K); mask++)
                {
                    if (dp[mask] >= int.MaxValue / 2) continue;
                    int i = -1;
                    for (int b = 0; b < K; b++)
                    {
                        if ((mask & (1 << b)) == 0) { i = b; break; }
                    }
                    if (i == -1) continue;

                    for (int j = i + 1; j < K; j++)
                    {
                        if ((mask & (1 << j)) == 0)
                        {
                            int nxtMask = mask | (1 << i) | (1 << j);
                            int cost = dp[mask] + distK[i, j];
                            if (cost < dp[nxtMask])
                            {
                                dp[nxtMask] = cost;
                                parentMask[nxtMask] = mask;
                            }
                        }
                    }
                }

                int currM = (1 << K) - 1;
                while (currM > 0)
                {
                    int pM = parentMask[currM];
                    int diff = currM ^ pM;
                    int uIdx = -1, vIdx = -1;
                    for (int b = 0; b < K; b++)
                    {
                        if ((diff & (1 << b)) != 0)
                        {
                            if (uIdx == -1) uIdx = b; else vIdx = b;
                        }
                    }

                    int curr = oddNodes[vIdx];
                    int startNode = oddNodes[uIdx];
                    while (curr != startNode && curr != -1)
                    {
                        int pNode = prevMatrix[uIdx][curr];
                        if (pNode == -1) break;
                        AddEdgeToMultigraph(curr, pNode);
                        curr = pNode;
                    }

                    currM = pM;
                }
            }
            else if (computeRequired <= 100_000_000L && memoryRequired <= 500_000_000L)
            {
                // GREEDY + 2-OPT
                int[][] distMatrix = new int[K][];
                int[][] prevMatrix = new int[K][];
                var pq = new PriorityQueue<int, int>();

                for (int i = 0; i < K; i++)
                {
                    distMatrix[i] = new int[N];
                    prevMatrix[i] = new int[N];
                    Array.Fill(distMatrix[i], int.MaxValue / 2);
                    Array.Fill(prevMatrix[i], -1);

                    int start = oddNodes[i];
                    distMatrix[i][start] = 0;
                    pq.Enqueue(start, 0);

                    while (pq.Count > 0)
                    {
                        pq.TryDequeue(out int u, out int d);
                        if (d > distMatrix[i][u]) continue;

                        foreach (var edge in adj[u])
                        {
                            int nxt = edge.to;
                            int nxtD = d + edge.w;
                            if (nxtD < distMatrix[i][nxt])
                            {
                                distMatrix[i][nxt] = nxtD;
                                prevMatrix[i][nxt] = u;
                                pq.Enqueue(nxt, nxtD);
                            }
                        }
                    }
                }

                var pairs = new List<(int i, int j, int dist)>();
                for (int i = 0; i < K; i++)
                {
                    for (int j = i + 1; j < K; j++)
                    {
                        pairs.Add((i, j, distMatrix[i][oddNodes[j]]));
                    }
                }
                pairs.Sort((a, b) => a.dist.CompareTo(b.dist));

                var matched = new bool[K];
                var currentMatch = new int[K];
                Array.Fill(currentMatch, -1);

                foreach (var p in pairs)
                {
                    if (!matched[p.i] && !matched[p.j])
                    {
                        matched[p.i] = true;
                        matched[p.j] = true;
                        currentMatch[p.i] = p.j;
                        currentMatch[p.j] = p.i;
                    }
                }

                bool improved = true;
                while (improved)
                {
                    improved = false;
                    for (int i = 0; i < K; i++)
                    {
                        for (int j = i + 1; j < K; j++)
                        {
                            int mi = currentMatch[i];
                            int mj = currentMatch[j];
                            if (mi == j || mi == -1 || mj == -1) continue;

                            int costCurrent = distMatrix[i][oddNodes[mi]] + distMatrix[j][oddNodes[mj]];
                            int costAlt1 = distMatrix[i][oddNodes[j]] + distMatrix[mi][oddNodes[mj]];
                            int costAlt2 = distMatrix[i][oddNodes[mj]] + distMatrix[j][oddNodes[mi]];

                            if (costAlt1 < costCurrent && costAlt1 <= costAlt2)
                            {
                                currentMatch[i] = j;
                                currentMatch[j] = i;
                                currentMatch[mi] = mj;
                                currentMatch[mj] = mi;
                                improved = true;
                            }
                            else if (costAlt2 < costCurrent)
                            {
                                currentMatch[i] = mj;
                                currentMatch[mj] = i;
                                currentMatch[j] = mi;
                                currentMatch[mi] = j;
                                improved = true;
                            }
                        }
                    }
                }

                var added = new bool[K];
                for (int i = 0; i < K; i++)
                {
                    if (!added[i])
                    {
                        int uIdx = i;
                        int vIdx = currentMatch[i];

                        int startNode = oddNodes[uIdx];
                        int curr = oddNodes[vIdx];
                        while (curr != startNode && curr != -1)
                        {
                            int pNode = prevMatrix[uIdx][curr];
                            if (pNode == -1) break;
                            AddEdgeToMultigraph(curr, pNode);
                            curr = pNode;
                        }

                        added[i] = true;
                        added[currentMatch[i]] = true;
                    }
                }
            }
            else
            {
                // MST-BASED MATCHING
                Array.Sort(edges);
                var dsu = new DSU(N);
                List<int>[] mstAdj = new List<int>[N];
                for (int i = 0; i < N; i++) mstAdj[i] = new List<int>();

                foreach (var e in edges)
                {
                    if (dsu.Union(e.u, e.v))
                    {
                        mstAdj[e.u].Add(e.v);
                        mstAdj[e.v].Add(e.u);
                    }
                }

                int[] parent = new int[N];
                int[] order = new int[N];
                int orderIdx = 0;
                bool[] vis = new bool[N];
                Queue<int> q = new Queue<int>();

                q.Enqueue(0);
                vis[0] = true;

                while (q.Count > 0)
                {
                    int curr = q.Dequeue();
                    order[orderIdx++] = curr;
                    foreach (var nxt in mstAdj[curr])
                    {
                        if (!vis[nxt])
                        {
                            vis[nxt] = true;
                            parent[nxt] = curr;
                            q.Enqueue(nxt);
                        }
                    }
                }

                int[] subtreeOddCount = new int[N];
                for (int i = 0; i < N; i++)
                {
                    if (deg[i] % 2 != 0) subtreeOddCount[i] = 1;
                }

                for (int i = orderIdx - 1; i > 0; i--)
                {
                    int u = order[i];
                    int p = parent[u];
                    if (subtreeOddCount[u] % 2 != 0)
                    {
                        AddEdgeToMultigraph(u, p);
                        subtreeOddCount[p] += 1;
                    }
                }
            }
        }

        // HIERHOLZER'S ALGORITHM
        int[] headArr = new int[N];
        var currPath = new Stack<int>();
        var circuit = new List<int>();

        currPath.Push(0);

        while (currPath.Count > 0)
        {
            int u = currPath.Peek();
            bool found = false;
            while (headArr[u] < multigraph[u].Count)
            {
                var e = multigraph[u][headArr[u]];
                headArr[u]++;
                if (!usedEdges[e.Id])
                {
                    usedEdges[e.Id] = true;
                    currPath.Push(e.To);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                circuit.Add(u);
                currPath.Pop();
            }
        }

        circuit.Reverse();

        using (var sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536))
        {
            sw.WriteLine(circuit.Count);
            for (int i = 0; i < circuit.Count; i++)
            {
                sw.Write(circuit[i]);
                if (i < circuit.Count - 1) sw.Write(' ');
            }
            sw.WriteLine();
        }
    }
}
