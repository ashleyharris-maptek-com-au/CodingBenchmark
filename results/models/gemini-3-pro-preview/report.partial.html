#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self { Self { x, y } }
    fn dist(&self, other: &Vec2) -> f64 {
        ((self.x - other.x).powi(2) + (self.y - other.y).powi(2)).sqrt()
    }
    fn len(&self) -> f64 {
        (self.x.powi(2) + self.y.powi(2)).sqrt()
    }
}

// A coarse grid for pathfinding on large maps
struct NavigationMesh {
    block_size: usize,
    cols: usize,
    rows: usize,
    // flattened grid: true = blocked (obstacle), false = free
    grid: Vec<bool>,
}

impl NavigationMesh {
    fn new(width: usize, height: usize) -> Self {
        // Adaptive block size: aim for roughly 100-200 blocks wide max to keep A* fast
        let block_size = (width / 100).max(1).max(height / 100);
        let cols = (width + block_size - 1) / block_size;
        let rows = (height + block_size - 1) / block_size;
        Self {
            block_size,
            cols,
            rows,
            grid: vec![false; cols * rows],
        }
    }

    fn mark_obstacle(&mut self, x: usize, y: usize) {
        let bx = x / self.block_size;
        let by = y / self.block_size;
        if bx < self.cols && by < self.rows {
            self.grid[by * self.cols + bx] = true;
        }
    }

    fn is_blocked(&self, bx: usize, by: usize) -> bool {
        if bx >= self.cols || by >= self.rows { return true; }
        self.grid[by * self.cols + bx]
    }

    // Convert world pos to block coords
    fn to_block(&self, pos: Vec2) -> (usize, usize) {
        let bx = (pos.x as usize / self.block_size).min(self.cols - 1);
        let by = (pos.y as usize / self.block_size).min(self.rows - 1);
        (bx, by)
    }

    // Get center of block in world coords
    fn to_world(&self, bx: usize, by: usize) -> Vec2 {
        Vec2 {
            x: (bx * self.block_size + self.block_size / 2) as f64,
            y: (by * self.block_size + self.block_size / 2) as f64,
        }
    }
}

// --- Pathfinding ---

#[derive(Copy, Clone, Eq, PartialEq)]
struct Node {
    bx: usize,
    by: usize,
    cost: i64, // scaled float
    heuristic: i64,
}

impl Ord for Node {
    fn cmp(&self, other: &Self) -> Ordering {
        // Min-heap: reverse comparison
        (other.cost + other.heuristic).cmp(&(self.cost + self.heuristic))
    }
}

impl PartialOrd for Node {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn find_path(nav: &NavigationMesh, start: Vec2, target: Vec2) -> Vec<Vec2> {
    let (sx, sy) = nav.to_block(start);
    let (tx, ty) = nav.to_block(target);

    let mut open_set = std::collections::BinaryHeap::new();
    let mut came_from: std::collections::HashMap<(usize, usize), (usize, usize)> = std::collections::HashMap::new();
    let mut g_score: std::collections::HashMap<(usize, usize), i64> = std::collections::HashMap::new();

    g_score.insert((sx, sy), 0);
    open_set.push(Node { bx: sx, by: sy, cost: 0, heuristic: 0 });

    let mut visited = HashSet::new();

    // Limit search to prevent timeout on weird maps
    let max_nodes = 50_000;
    let mut nodes_checked = 0;

    while let Some(node) = open_set.pop() {
        if node.bx == tx && node.by == ty {
            // Reconstruct
            let mut path = Vec::new();
            let mut curr = (tx, ty);
            while curr != (sx, sy) {
                path.push(nav.to_world(curr.0, curr.1));
                curr = *came_from.get(&curr).unwrap();
            }
            path.reverse();
            return path;
        }

        if !visited.insert((node.bx, node.by)) { continue; }
        nodes_checked += 1;
        if nodes_checked > max_nodes { break; }

        // Neighbors (8-way)
        for dx in -1..=1 {
            for dy in -1..=1 {
                if dx == 0 && dy == 0 { continue; }
                let nx = node.bx as isize + dx;
                let ny = node.by as isize + dy;

                if nx >= 0 && ny >= 0 && nx < nav.cols as isize && ny < nav.rows as isize {
                    let nx = nx as usize;
                    let ny = ny as usize;
                    if nav.is_blocked(nx, ny) { continue; }

                    let dist_cost = if dx == 0 || dy == 0 { 10 } else { 14 }; // 1.0 vs 1.414
                    let new_g = g_score.get(&(node.bx, node.by)).unwrap() + dist_cost;

                    if new_g < *g_score.get(&(nx, ny)).unwrap_or(&i64::MAX) {
                        came_from.insert((nx, ny), (node.bx, node.by));
                        g_score.insert((nx, ny), new_g);
                        let h = (((nx as i64 - tx as i64).abs() + (ny as i64 - ty as i64).abs()) * 10);
                        open_set.push(Node { bx: nx, by: ny, cost: new_g, heuristic: h });
                    }
                }
            }
        }
    }
    
    // Fallback: direct line to target
    vec![target]
}

            target_angle = target_angle.clamp(-0.2, 0.2);
        } else {
            target_angle = target_angle.clamp(-MAX_TILT, MAX_TILT);
        }

        let angle_err = target_angle - current_state.angle;
        let mut turn_cmd = angle_err * kp_ang - current_state.angular_vel * kd_ang;
        turn_cmd = turn_cmd.clamp(-1.0, 1.0);

        // If we are tilted correctly, apply thrust. If strictly turning, reduce thrust?

        let dist_ground = config.height as f64 - pos.y; // rough approx
        if dist_ground < 50.0 && vel.vy > 5.0 {
           // thrust_cmd = 1.0; // Can be risky if angled wrong
        }

        // Use writeln! to buffer? stdout is line buffered usually.
        println!("{:.4} {:.4}", thrust_cmd, turn_cmd);
    }

        let block_size = (width / 100).max(1).max(height / 100);
        let cols = (width + block_size - 1) / block_size;
        let rows = (height + block_size - 1) / block_size;
        Self {
            block_size,
            cols,
            rows,
            grid: vec![false; cols * rows],
        }
    }

    fn mark_obstacle(&mut self, x: usize, y: usize) {
        let bx = x / self.block_size;
        let by = y / self.block_size;
        if bx < self.cols && by < self.rows {
            self.grid[by * self.cols + bx] = true;
        }
    }

    fn is_blocked(&self, bx: usize, by: usize) -> bool {
        if bx >= self.cols || by >= self.rows { return true; }
        self.grid[by * self.cols + bx]
    }

    // Convert world pos to block coords
    fn to_block(&self, pos: Vec2) -> (usize, usize) {
        let bx = (pos.x as usize / self.block_size).min(self.cols - 1);
        let by = (pos.y as usize / self.block_size).min(self.rows - 1);
        (bx, by)
    }

    // Get center of block in world coords
    fn to_world(&self, bx: usize, by: usize) -> Vec2 {
        Vec2 {
            x: (bx * self.block_size + self.block_size / 2) as f64,
            y: (by * self.block_size + self.block_size / 2) as f64,
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    last_update: Option<Instant>,
}

impl Default for LanderState {
    fn default() -> Self {
        Self {
            x: 0.0, y: 0.0, vx: 0.0, vy: 0.0,
            angle: 0.0, angular_vel: 0.0, fuel: 0.0,
            last_update: None,
        }
    }
}

// --- Pathfinding ---

#[derive(Copy, Clone, Eq, PartialEq)]
struct Node {
    bx: usize,
    by: usize,
    cost: i64, // scaled float
    heuristic: i64,
}

impl Ord for Node {
    fn cmp(&self, other: &Self) -> Ordering {
        // Min-heap: reverse comparison
        (other.cost + other.heuristic).cmp(&(self.cost + self.heuristic))
    }
}

impl PartialOrd for Node {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn find_path(nav: &NavigationMesh, start: Vec2, target: Vec2) -> Vec<Vec2> {
    let (sx, sy) = nav.to_block(start);
    let (tx, ty) = nav.to_block(target);

    let mut open_set = std::collections::BinaryHeap::new();
    let mut came_from: std::collections::HashMap<(usize, usize), (usize, usize)> = std::collections::HashMap::new();
    let mut g_score: std::collections::HashMap<(usize, usize), i64> = std::collections::HashMap::new();

    g_score.insert((sx, sy), 0);
    open_set.push(Node { bx: sx, by: sy, cost: 0, heuristic: 0 });

    let mut visited = HashSet::new();

    // Limit search to prevent timeout on weird maps
    let max_nodes = 50_000;
    let mut nodes_checked = 0;

    while let Some(node) = open_set.pop() {
        if node.bx == tx && node.by == ty {
            // Reconstruct
            let mut path = Vec::new();
            let mut curr = (tx, ty);
            while curr != (sx, sy) {
                path.push(nav.to_world(curr.0, curr.1));
                curr = *came_from.get(&curr).unwrap();
            }
            path.reverse();
            return path;
        }

        if !visited.insert((node.bx, node.by)) { continue; }
        nodes_checked += 1;
        if nodes_checked > max_nodes { break; }

        // Neighbors (8-way)
        for dx in -1..=1 {
            for dy in -1..=1 {
                if dx == 0 && dy == 0 { continue; }
                let nx = node.bx as isize + dx;
                let ny = node.by as isize + dy;

                if nx >= 0 && ny >= 0 && nx < nav.cols as isize && ny < nav.rows as isize {
                    let nx = nx as usize;
                    let ny = ny as usize;
                    if nav.is_blocked(nx, ny) { continue; }

                    let dist_cost = if dx == 0 || dy == 0 { 10 } else { 14 }; // 1.0 vs 1.414
                    let new_g = g_score.get(&(node.bx, node.by)).unwrap() + dist_cost;

                    if new_g < *g_score.get(&(nx, ny)).unwrap_or(&i64::MAX) {
                        came_from.insert((nx, ny), (node.bx, node.by));
                        g_score.insert((nx, ny), new_g);
                        let h = (((nx as i64 - tx as i64).abs() + (ny as i64 - ty as i64).abs()) * 10);
                        open_set.push(Node { bx: nx, by: ny, cost: new_g, heuristic: h });
                    }
                }
            }
        }
    }
    
    // Fallback: direct line to target
    vec![target]
}

// --- Main --- //

fn main() {
    // 1. Setup IO
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    // 2. Read Config
    let mut line = String::new();
    handle.read_line(&mut line).expect("Failed to read config");
    let params: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if params.len() < 5 {
        panic!("Invalid config");
    }
    let (width, height, gravity, max_thrust, fuel) = (params[0], params[1], params[2], params[3], params[4]);
    let width = width as usize;
    let height = height as usize;

    line.clear();
    handle.read_line(&mut line).expect("Failed to read coords");
    let coords: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    let start = Vec2::new(coords[0], coords[1]);
    let target = Vec2::new(coords[2], coords[3]);

    let config = Arc::new(WorldConfig {
        width, height, gravity, max_thrust, fuel, start, target
    });

    // 3. Read Map & Build NavMesh
    let mut nav = NavigationMesh::new(width, height);
    // Read `height` lines
    for y in 0..height {
        line.clear();
        handle.read_line(&mut line).expect("Failed to read map row");
        for (x, c) in line.chars().enumerate() {
            if x >= width { break; }
            if c == '#' {
                nav.mark_obstacle(x, y);
            }
        }
    }

    // 4. Pre-calculate path (simplistic version for now)
    let mut waypoints = find_path(&nav, start, target);
    // Ensure target is the final point
    if let Some(last) = waypoints.last() {
        if last.dist(&target) > 5.0 {
            waypoints.push(target);
        }
    } else {
        waypoints.push(target);
    }
    let waypoints = Arc::new(RwLock::new(VecDeque::from(waypoints)));

    // 5. Shared State Setup
    let shared_state = Arc::new(RwLock::new(LanderState::default()));
    
    // 6. Spawn Input Thread
    let state_writer = shared_state.clone();
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut handle = stdin.lock();
        let mut buffer = String::new();
        
        loop {
            buffer.clear();
            match handle.read_line(&mut buffer) {
                Ok(0) => break, // EOF
                Ok(_) => {
                    let parts: Vec<&str> = buffer.split_whitespace().collect();
                    if parts.len() >= 7 {
                        if let (Ok(x), Ok(y), Ok(vx), Ok(vy), Ok(a), Ok(av), Ok(f)) = (
                            parts[0].parse(), parts[1].parse(), parts[2].parse(),
                            parts[3].parse(), parts[4].parse(), parts[5].parse(), parts[6].parse()
                        ) {
                            if let Ok(mut st) = state_writer.write() {
                                *st = LanderState {
                                    x, y, vx, vy, angle: a, angular_vel: av, fuel: f, 
                                    last_update: Some(Instant::now())
                                };
                            }
                        }
                    }
                }
                Err(_) => break,
            }
        }
    });

    // 7. Control Loop (Main Thread)
    let mut stdout = io::stdout();
    let mut last_cmd_time = Instant::now();

    // PID Constants
    // Position (Outer Loop) -> Desired Force
    let kp_pos = 0.5;
    let kd_pos = 1.2;
    
    // Attitude (Inner Loop) -> Turn Command
    let kp_ang = 5.0;
    let kd_ang = 10.0;

    loop {
        let now = Instant::now();
        if now.duration_since(last_cmd_time).as_secs_f64() < DT {
            // Small sleep to prevent busy loop, but maintain >20Hz
            thread::sleep(Duration::from_millis(5));
            continue;
        }
        last_cmd_time = now;

        let current_state = {
            let r = shared_state.read().unwrap();
            *r
        };

        // Wait for first update
        if current_state.last_update.is_none() {
            continue;
        }

        let pos = Vec2::new(current_state.x, current_state.y);
        let vel = Vec2::new(current_state.vx, current_state.vy);
        
        // -- Logic --
        // 1. Determine Target Point
        let mut wp_lock = waypoints.write().unwrap();
        let mut target_point = config.target;
        
        // Simple waypoint following
        if !wp_lock.is_empty() {
            target_point = *wp_lock.front().unwrap();
            if pos.dist(&target_point) < 20.0 && wp_lock.len() > 1 {
                wp_lock.pop_front();
                target_point = *wp_lock.front().unwrap();
            }
        }

        // 2. Control Mode
        let dist_to_final = pos.dist(&config.target);
        let landing_mode = dist_to_final < LANDING_dist || (pos.x - config.target.x).abs() < 5.0 && pos.y < config.target.y && pos.y > config.target.y - 50.0;

        // Desired Acceleration calculation
        let mut ax_des;
        let mut ay_des;

        if landing_mode {
            // Landing: Kill Horizontal V, Controlled Descent
            ax_des = (config.target.x - pos.x) * 1.0 - vel.x * 2.0;
            // Desired descent speed decreases as we get closer
            // Target Vy should be positive (down) but small. e.g. 1.0 m/s
            let target_vy = 1.0;
            ay_des = (target_vy - vel.vy) * 2.0; 
            // Don't fight gravity too hard if we are above target, just drift down
        } else {
            // Cruise: PD to target point
            ax_des = (target_point.x - pos.x) * kp_pos - vel.x * kd_pos;
            ay_des = (target_point.y - pos.y) * kp_pos - vel.vy * kd_pos;
            
            // Speed limit
            let max_speed = 40.0;
            let v_norm = (vel.x*vel.x + vel.vy*vel.vy).sqrt();
            if v_norm > max_speed {
                ax_des *= 0.1;
                ay_des *= 0.1;
            }
        }

        // Gravity compensation
        // We need total force Fy to result in (ay_des + g)
        // Thrust provides force vector F. Fy = -F * cos(angle). Fx = F * sin(angle).
        // Required Fy_total = ay_des. F_engine_y + g = ay_des => F_engine_y = ay_des - g.
        // Wait, gravity is positive down. 
        // Engine pushes Up (negative Y) when angle is 0.
        // Accel_y = Gravity + Thrust_y.
        // Thrust_y_req = ay_des - Gravity.
        let req_acc_y = ay_des - config.gravity;
        let req_acc_x = ax_des;

        // Calculate desired angle and thrust magnitude
        // Engine force vector F_eng should be (req_acc_x, req_acc_y)
        // But engine thrusts along local -Y. 
        // Rotate local (0, -T) by angle theta -> (-T sin theta, -T cos theta) ?
        // Check: Angle 0 -> (0, -T). Correct (Up).
        // Check: Angle 90 (PI/2) -> (-T, 0). Left? 
        // Prompt: "angle is in radians, 0 is facing up. + is clockwise".
        // Vector rotated clockwise:
        // x' = x cos a - y sin a 
        // y' = x sin a + y cos a
        // Local thrust is (0, -T). 
        // x_world = 0 - (-T) sin a = T sin a
        // y_world = 0 + (-T) cos a = -T cos a
        // So F = (T sin a, -T cos a)
        
        // We want F = (req_acc_x, req_acc_y)
        // tan(a) = Fx / -Fy = req_acc_x / -req_acc_y
        // T = sqrt(Fx^2 + Fy^2)
        
        let thrust_accel_req = (req_acc_x.powi(2) + req_acc_y.powi(2)).sqrt();
        let mut target_angle = f64::atan2(req_acc_x, -req_acc_y);

        // Clamp angle for safety (don't flip over)
        // Unless we are high up and need to turn around, but simplistic clamp is safer
        if landing_mode {
            target_angle = target_angle.clamp(-0.2, 0.2);
        } else {
            target_angle = target_angle.clamp(-MAX_TILT, MAX_TILT);
        }

        // 3. Attitude Control
        let angle_err = target_angle - current_state.angle;
        let mut turn_cmd = angle_err * kp_ang - current_state.angular_vel * kd_ang;
        turn_cmd = turn_cmd.clamp(-1.0, 1.0);

        // 4. Thrust Control
        // If we are tilted correctly, apply thrust. If strictly turning, reduce thrust?
        // Simple: apply requested thrust scaled by max.
        let mut thrust_cmd = thrust_accel_req / config.max_thrust;
        
        // If we are pointing wrong way, reduce thrust to avoid accelerating in wrong direction
        if angle_err.abs() > 0.5 {
            thrust_cmd *= 0.1;
        } 

        thrust_cmd = thrust_cmd.clamp(0.0, 1.0);

        // Emergency override: if very close to ground and falling fast, full blast
        let dist_ground = config.height as f64 - pos.y; // rough approx
        if dist_ground < 50.0 && vel.vy > 5.0 {
           // thrust_cmd = 1.0; // Can be risky if angled wrong
        }

        // Output
        // Use writeln! to buffer? stdout is line buffered usually.
        println!("{:.4} {:.4}", thrust_cmd, turn_cmd);
    }
}
