#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

using namespace std;

struct Point {
    double x, y, z;
};

Point cross(Point a, Point b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
Point sub(Point a, Point b) {
    return {a.x-b.x, a.y-b.y, a.z-b.z};
}
double dot(Point a, Point b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}
double norm(Point a) {
    return sqrt(a.x*a.x + a.y*a.y + a.z*a.z);
}

struct Face {
    int v[3];
    Point normal;
    double d;
    bool visible(const Point& p) const {
        return dot(normal, p) > d + 1e-7;
    }
};

Face make_face(int v0, int v1, int v2, const vector<Point>& pts) {
    Face f;
    f.v[0] = v0; f.v[1] = v1; f.v[2] = v2;
    f.normal = cross(sub(pts[v1], pts[v0]), sub(pts[v2], pts[v0]));
    double len = norm(f.normal);
    if (len > 1e-12) {
        f.normal.x /= len; f.normal.y /= len; f.normal.z /= len;
    }
    f.d = dot(f.normal, pts[v0]);
    return f;
}

struct BaseTetra {
    Point v[4];
    Point edges[6];
    Point faces[4];
};
BaseTetra base_tet[4];

void init_bases(double e) {
    double R = e * sqrt(3.0 / 8.0);
    Point v0[4] = {
        { sqrt(8.0/9.0)*R, 0, -R/3.0 },
        { -sqrt(2.0/9.0)*R, sqrt(2.0/3.0)*R, -R/3.0 },
        { -sqrt(2.0/9.0)*R, -sqrt(2.0/3.0)*R, -R/3.0 },
        { 0, 0, R }
    };
    
    for(int k=0; k<4; ++k) base_tet[0].v[k] = v0[k];
    for(int k=0; k<4; ++k) base_tet[1].v[k] = {v0[k].x, -v0[k].y, -v0[k].z};
    for(int k=0; k<4; ++k) base_tet[2].v[k] = {-v0[k].x, v0[k].y, -v0[k].z};
    for(int k=0; k<4; ++k) base_tet[3].v[k] = {-v0[k].x, -v0[k].y, v0[k].z};
    
    for(int i=0; i<4; ++i) {
        base_tet[i].edges[0] = sub(base_tet[i].v[1], base_tet[i].v[0]);
        base_tet[i].edges[1] = sub(base_tet[i].v[2], base_tet[i].v[0]);
        base_tet[i].edges[2] = sub(base_tet[i].v[3], base_tet[i].v[0]);
        base_tet[i].edges[3] = sub(base_tet[i].v[2], base_tet[i].v[1]);
        base_tet[i].edges[4] = sub(base_tet[i].v[3], base_tet[i].v[1]);
        base_tet[i].edges[5] = sub(base_tet[i].v[3], base_tet[i].v[2]);
        
        base_tet[i].faces[0] = cross(base_tet[i].edges[0], base_tet[i].edges[1]);
        base_tet[i].faces[1] = cross(base_tet[i].edges[1], base_tet[i].edges[2]);
        base_tet[i].faces[2] = cross(base_tet[i].edges[2], base_tet[i].edges[0]);
        base_tet[i].faces[3] = cross(base_tet[i].edges[3], base_tet[i].edges[4]);
        
        for(int j=0; j<4; ++j) {
            double len = norm(base_tet[i].faces[j]);
            if(len > 1e-12) {
                base_tet[i].faces[j].x /= len;
                base_tet[i].faces[j].y /= len;
                base_tet[i].faces[j].z /= len;
            }
        }
    }
}

struct AxisData {
    Point axis;
    double min1, max1;
    double min2, max2;
};
vector<AxisData> precomputed_axes[4][4];

void precompute_axes() {
    for(int i=0; i<4; ++i) {
        for(int j=i; j<4; ++j) {
            vector<Point> raw_axes;
            for(int k=0; k<4; ++k) raw_axes.push_back(base_tet[i].faces[k]);
            for(int k=0; k<4; ++k) raw_axes.push_back(base_tet[j].faces[k]);
            for(int k1=0; k1<6; ++k1) {
                for(int k2=0; k2<6; ++k2) {
                    Point c = cross(base_tet[i].edges[k1], base_tet[j].edges[k2]);
                    double len = norm(c);
                    if(len > 1e-7) {
                        c.x /= len; c.y /= len; c.z /= len;
                        raw_axes.push_back(c);
                    }
                }
            }
            
            vector<Point> uniq;
            for(auto ax : raw_axes) {
                bool dup = false;
                for(auto u : uniq) {
                    if(norm(sub(ax, u)) < 1e-5 || norm(sub(ax, {-u.x, -u.y, -u.z})) < 1e-5) {
                        dup = true; break;
                    }
                }
                if(!dup) uniq.push_back(ax);
            }
            
            for(auto ax : uniq) {
                AxisData ad;
                ad.axis = ax;
                ad.min1 = 1e9; ad.max1 = -1e9;
                for(int k=0; k<4; ++k) {
                    double p = dot(base_tet[i].v[k], ax);
                    ad.min1 = min(ad.min1, p);
                    ad.max1 = max(ad.max1, p);
                }
                ad.min2 = 1e9; ad.max2 = -1e9;
                for(int k=0; k<4; ++k) {
                    double p = dot(base_tet[j].v[k], ax);
                    ad.min2 = min(ad.min2, p);
                    ad.max2 = max(ad.max2, p);
                }
                precomputed_axes[i][j].push_back(ad);
            }
        }
    }
}

bool overlap(int type1, Point c1, int type2, Point c2) {
    if(type1 > type2) {
        swap(type1, type2);
        swap(c1, c2);
    }
    for(const auto& ad : precomputed_axes[type1][type2]) {
        double shift1 = dot(c1, ad.axis);
        double shift2 = dot(c2, ad.axis);
        if(ad.max1 + shift1 <= ad.min2 + shift2 + 1e-7 || 
           ad.max2 + shift2 <= ad.min1 + shift1 + 1e-7) {
            return false;
        }
    }
    return true;
}

struct Plane {
    Point normal;
    double d;
    double offset[4];
};
vector<Plane> poly_planes;

void prepare_planes(const vector<Face>& hull) {
    for(const auto& f : hull) {
        Plane p;
        p.normal = f.normal;
        p.d = f.d;
        for(int type=0; type<4; ++type) {
            double mx = -1e9;
            for(int k=0; k<4; ++k) {
                mx = max(mx, dot(f.normal, base_tet[type].v[k]));
            }
            p.offset[type] = mx;
        }
        poly_planes.push_back(p);
    }
}

bool is_inside(int type, Point C) {
    for(const auto& p : poly_planes) {
        if(dot(p.normal, C) > p.d - p.offset[type] + 1e-7) return false;
    }
    return true;
}

struct Placed {
    int type;
    Point c;
};
vector<Placed> placed_tets;

struct Cell {
    int x, y, z;
    bool operator==(const Cell& o) const { return x==o.x && y==o.y && z==o.z; }
};
struct CellHash {
    size_t operator()(const Cell& c) const {
        return (size_t)(c.x * 73856093) ^ (size_t)(c.y * 19349663) ^ (size_t)(c.z * 83492791);
    }
};
unordered_map<Cell, vector<int>, CellHash> grid_map;

bool can_place(int type, Point C, double cs) {
    if(!is_inside(type, C)) return false;
    
    int cx = floor(C.x / cs);
    int cy = floor(C.y / cs);
    int cz = floor(C.z / cs);
    
    for(int dx=-1; dx<=1; ++dx) {
        for(int dy=-1; dy<=1; ++dy) {
            for(int dz=-1; dz<=1; ++dz) {
                Cell cell = {cx+dx, cy+dy, cz+dz};
                auto it = grid_map.find(cell);
                if(it != grid_map.end()) {
                    for(int idx : it->second) {
                        if(overlap(type, C, placed_tets[idx].type, placed_tets[idx].c)) {
                            return false;
                        }
                    }
                }
            }
        }
    }
    return true;
}

void place(int type, Point C, double cs) {
    int idx = placed_tets.size();
    placed_tets.push_back({type, C});
    int cx = floor(C.x / cs);
    int cy = floor(C.y / cs);
    int cz = floor(C.z / cs);
    grid_map[{cx, cy, cz}].push_back(idx);
}

struct GridPt {
    int x, y, z;
    bool operator==(const GridPt& o) const { return x==o.x && y==o.y && z==o.z; }
};
struct GridHash {
    size_t operator()(const GridPt& pt) const {
        return (size_t)(pt.x * 73856093) ^ (size_t)(pt.y * 19349663) ^ (size_t)(pt.z * 83492791);
    }
};
unordered_set<GridPt, GridHash> visited;

struct PQNode {
    GridPt pt;
    double z;
    bool operator<(const PQNode& o) const {
        return z > o.z; 
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int V;
    double edge_length;
    if (!(cin >> V >> edge_length)) return 0;

    vector<Point> pts(V);
    for(int i=0; i<V; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
    }

    if(V < 4) {
        cout << 0 << "\n";
        return 0;
    }

    int v0=0, v1=-1, v2=-1, v3=-1;
    for(int i=1; i<V; ++i) {
        if(norm(sub(pts[0], pts[i])) > 1e-7) { v1 = i; break; }
    }
    if(v1 != -1) {
        for(int i=v1+1; i<V; ++i) {
            if(norm(cross(sub(pts[v1], pts[v0]), sub(pts[i], pts[v0]))) > 1e-7) { v2 = i; break; }
        }
    }
    if(v2 != -1) {
        Point n = cross(sub(pts[v1], pts[v0]), sub(pts[v2], pts[v0]));
        for(int i=v2+1; i<V; ++i) {
            if(abs(dot(n, sub(pts[i], pts[v0]))) > 1e-7) { v3 = i; break; }
        }
    }
    if(v3 == -1) {
        cout << 0 << "\n";
        return 0;
    }

    Point cent = {0,0,0};
    for(auto p : pts) { cent.x += p.x; cent.y += p.y; cent.z += p.z; }
    cent.x /= V; cent.y /= V; cent.z /= V;

    auto make_face_outward = [&](int a, int b, int c) {
        Face f = make_face(a, b, c, pts);
        if(dot(f.normal, cent) > f.d) {
            return make_face(a, c, b, pts);
        }
        return f;
    };

    vector<Face> hull;
    hull.push_back(make_face_outward(v0, v1, v2));
    hull.push_back(make_face_outward(v0, v2, v3));
    hull.push_back(make_face_outward(v0, v3, v1));
    hull.push_back(make_face_outward(v1, v2, v3));

    vector<bool> is_used(V, false);
    is_used[v0] = is_used[v1] = is_used[v2] = is_used[v3] = true;
    vector<int> rem;
    for(int i=0; i<V; ++i) if(!is_used[i]) rem.push_back(i);

    mt19937 g(1337);
    shuffle(rem.begin(), rem.end(), g);

    for(int p_idx : rem) {
        vector<Face> next_hull;
        vector<pair<int, int>> bnd;
        for(const auto& f : hull) {
            if(f.visible(pts[p_idx])) {
                bnd.push_back({f.v[0], f.v[1]});
                bnd.push_back({f.v[1], f.v[2]});
                bnd.push_back({f.v[2], f.v[0]});
            } else {
                next_hull.push_back(f);
            }
        }
        if(bnd.empty()) continue; 
        
        vector<bool> keep(bnd.size(), true);
        for(size_t i=0; i<bnd.size(); ++i) {
            if(!keep[i]) continue;
            for(size_t j=i+1; j<bnd.size(); ++j) {
                if(keep[j] && bnd[i].first == bnd[j].second && bnd[i].second == bnd[j].first) {
                    keep[i] = false;
                    keep[j] = false;
                    break;
                }
            }
        }
        
        for(size_t i=0; i<bnd.size(); ++i) {
            if(keep[i]) {
                next_hull.push_back(make_face(bnd[i].first, bnd[i].second, p_idx, pts));
            }
        }
        hull = next_hull;
    }

    init_bases(edge_length);
    precompute_axes();
    prepare_planes(hull);

    double cs = edge_length;
    double step = edge_length * 0.3;

    priority_queue<PQNode> pq;

    for(auto p : pts) {
        Point dp = sub(cent, p);
        double dl = norm(dp);
        Point sc = p;
        if(dl > 1e-7) {
            sc.x += dp.x / dl * step * 2.0;
            sc.y += dp.y / dl * step * 2.0;
            sc.z += dp.z / dl * step * 2.0;
        }
        GridPt gp = { (int)round(sc.x/step), (int)round(sc.y/step), (int)round(sc.z/step) };
        if(visited.find(gp) == visited.end()) {
            visited.insert(gp);
            pq.push({gp, gp.z * step});
        }
    }

    auto start_time = chrono::steady_clock::now();
    int iters = 0;

    while(!pq.empty()) {
        PQNode node = pq.top();
        pq.pop();
        
        Point C = { node.pt.x * step, node.pt.y * step, node.pt.z * step };
        
        bool center_inside = true;
        for(const auto& p : poly_planes) {
            if(dot(p.normal, C) > p.d + 1e-7) {
                center_inside = false;
                break;
            }
        }
        if(!center_inside) continue; 

        for(int type=0; type<4; ++type) {
            if(can_place(type, C, cs)) {
                place(type, C, cs);
                break;
            }
        }

        int dx[] = {1, -1, 0, 0, 0, 0};
        int dy[] = {0, 0, 1, -1, 0, 0};
        int dz[] = {0, 0, 0, 0, 1, -1};
        for(int i=0; i<6; ++i) {
            GridPt npt = {node.pt.x+dx[i], node.pt.y+dy[i], node.pt.z+dz[i]};
            if(visited.find(npt) == visited.end()) {
                visited.insert(npt);
                pq.push({npt, npt.z * step});
            }
        }

        if((++iters & 2047) == 0) {
            auto now = chrono::steady_clock::now();
            if(chrono::duration_cast<chrono::milliseconds>(now - start_time).count() > 28000) {
                break;
            }
        }
    }

    double euler[4][3] = {
        {0.0, 0.0, 0.0},
        {M_PI, 0.0, 0.0},
        {0.0, M_PI, 0.0},
        {0.0, 0.0, M_PI}
    };

    cout << placed_tets.size() << "\n";
    for(const auto& t : placed_tets) {
        cout << fixed << setprecision(6) 
             << t.c.x << " " << t.c.y << " " << t.c.z << " "
             << euler[t.type][0] << " " << euler[t.type][1] << " " << euler[t.type][2] << "\n";
    }

    return 0;
}
