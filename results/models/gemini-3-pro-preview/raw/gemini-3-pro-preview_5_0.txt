use std::io::{self, BufRead};

struct Solver {
    w: usize,
    h: usize,
    require_cycle: bool,
    adj: Vec<Vec<usize>>,
    unvisited: Vec<bool>,
    degree: Vec<usize>,
    unvisited_count: usize,
    path: Vec<usize>,
    
    bfs_visited: Vec<u32>,
    bfs_queue: Vec<usize>,
    bfs_generation: u32,
    
    deg1_count: usize,
    deg0_count: usize,
}

impl Solver {
    fn solve(&mut self, u: usize) -> bool {
        if self.unvisited_count == 0 {
            if self.require_cycle {
                return self.adj[u].contains(&0);
            }
            return true;
        }
        
        if self.require_cycle {
            let mut has_cycle_neighbor = false;
            for &v in &self.adj[0] {
                if self.unvisited[v] {
                    has_cycle_neighbor = true;
                    break;
                }
            }
            if !has_cycle_neighbor {
                return false;
            }
        }
        
        if self.deg0_count > 0 && self.unvisited_count > 1 {
            return false;
        }
        
        if self.deg1_count > 2 {
            return false;
        }
        
        let mut neighbors = [0; 4];
        let mut n_count = 0;
        for &v in &self.adj[u] {
            if self.unvisited[v] {
                neighbors[n_count] = v;
                n_count += 1;
            }
        }
        
        if n_count == 0 {
            return false;
        }
        
        if !self.check_connectivity(neighbors[0]) {
            return false;
        }
        
        neighbors[0..n_count].sort_by_key(|&v| self.degree[v]);
        
        for i in 0..n_count {
            let v = neighbors[i];
            
            self.unvisited[v] = false;
            self.unvisited_count -= 1;
            self.path.push(v);
            
            let mut changed = [(0, 0, 0); 4];
            let mut changed_count = 0;
            for &n in &self.adj[v] {
                if self.unvisited[n] {
                    let old_deg = self.degree[n];
                    self.degree[n] -= 1;
                    let new_deg = self.degree[n];
                    changed[changed_count] = (n, old_deg, new_deg);
                    changed_count += 1;
                    
                    if old_deg == 1 { self.deg1_count -= 1; }
                    if old_deg == 2 { self.deg1_count += 1; }
                    
                    if new_deg == 0 { self.deg0_count += 1; }
                }
            }
            
            if self.solve(v) {
                return true;
            }
            
            for j in 0..changed_count {
                let (n, old_deg, new_deg) = changed[j];
                self.degree[n] = old_deg;
                if old_deg == 1 { self.deg1_count += 1; }
                if old_deg == 2 { self.deg1_count -= 1; }
                
                if new_deg == 0 { self.deg0_count -= 1; }
            }
            self.path.pop();
            self.unvisited_count += 1;
            self.unvisited[v] = true;
        }
        
        false
    }
    
    fn check_connectivity(&mut self, start: usize) -> bool {
        self.bfs_generation += 1;
        self.bfs_visited[start] = self.bfs_generation;
        self.bfs_queue.clear();
        self.bfs_queue.push(start);
        let mut head = 0;
        let mut count = 1;
        
        while head < self.bfs_queue.len() {
            let v = self.bfs_queue[head];
            head += 1;
            for &n in &self.adj[v] {
                if self.unvisited[n] && self.bfs_visited[n] != self.bfs_generation {
                    self.bfs_visited[n] = self.bfs_generation;
                    self.bfs_queue.push(n);
                    count += 1;
                }
            }
        }
        
        count == self.unvisited_count
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());
    
    let mut w = 0;
    let mut h = 0;
    let mut obs_count = 0;
    let mut require_cycle = false;
    
    while let Some(line) = lines.next() {
        let parts: Vec<usize> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
        if parts.len() >= 4 {
            w = parts[0];
            h = parts[1];
            obs_count = parts[2];
            require_cycle = parts[3] == 1;
            break;
        }
    }
    
    if w == 0 || h == 0 { return; }
    
    let mut grid = vec![false; w * h];
    let mut obs_read = 0;
    while obs_read < obs_count {
        if let Some(line) = lines.next() {
            let p: Vec<usize> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
            if p.len() >= 2 {
                grid[p[1] * w + p[0]] = true;
                obs_read += 1;
            }
        } else {
            break;
        }
    }
    
    if grid[0] {
        println!("0");
        return;
    }
    
    let mut n0 = 0;
    let mut n1 = 0;
    for y in 0..h {
        for x in 0..w {
            if !grid[y * w + x] {
                if (x + y) % 2 == 0 {
                    n0 += 1;
                } else {
                    n1 += 1;
                }
            }
        }
    }
    
    if require_cycle {
        if n0 != n1 {
            println!("0");
            return;
        }
    } else {
        if n0 != n1 && n0 != n1 + 1 {
            println!("0");
            return;
        }
    }
    
    let mut adj = vec![Vec::new(); w * h];
    for y in 0..h {
        for x in 0..w {
            let u = y * w + x;
            if grid[u] { continue; }
            
            if x > 0 && !grid[y * w + x - 1] { adj[u].push(y * w + x - 1); }
            if x + 1 < w && !grid[y * w + x + 1] { adj[u].push(y * w + x + 1); }
            if y > 0 && !grid[(y - 1) * w + x] { adj[u].push((y - 1) * w + x); }
            if y + 1 < h && !grid[(y + 1) * w + x] { adj[u].push((y + 1) * w + x); }
        }
    }
    
    let mut unvisited = vec![false; w * h];
    let mut unvisited_count = 0;
    for y in 0..h {
        for x in 0..w {
            if !grid[y * w + x] {
                unvisited[y * w + x] = true;
                unvisited_count += 1;
            }
        }
    }
    
    unvisited[0] = false;
    unvisited_count -= 1;
    
    let mut degree = vec![0; w * h];
    for y in 0..h {
        for x in 0..w {
            let u = y * w + x;
            if unvisited[u] {
                for &v in &adj[u] {
                    if unvisited[v] {
                        degree[u] += 1;
                    }
                }
            }
        }
    }
    
    let mut deg1_count = 0;
    let mut deg0_count = 0;
    for i in 0..w * h {
        if unvisited[i] {
            if degree[i] == 1 { deg1_count += 1; }
            if degree[i] == 0 { deg0_count += 1; }
        }
    }
    
    let mut path = Vec::with_capacity(w * h);
    path.push(0);
    
    let mut solver = Solver {
        w,
        h,
        require_cycle,
        adj,
        unvisited,
        degree,
        unvisited_count,
        path,
        bfs_visited: vec![0; w * h],
        bfs_queue: Vec::with_capacity(w * h),
        bfs_generation: 0,
        deg1_count,
        deg0_count,
    };
    
    if solver.solve(0) {
        println!("{}", solver.path.len());
        for &u in &solver.path {
            let x = u % w;
            let y = u / w;
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
