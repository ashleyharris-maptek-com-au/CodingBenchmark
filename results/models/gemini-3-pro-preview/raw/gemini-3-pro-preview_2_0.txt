using System;
using System.IO;
using System.Collections.Generic;

class Program {
    struct OriginalEdge {
        public int u, v, w, id;
    }

    class Edge {
        public int to;
        public int id;
        public Edge rev;
        public bool used;
    }

    struct Pair : IComparable<Pair> {
        public int i, j;
        public long dist;
        public int CompareTo(Pair other) => dist.CompareTo(other.dist);
    }

    static int N, M;
    static OriginalEdge[] originalEdges;
    static List<OriginalEdge>[] originalAdj;
    static List<Edge>[] adj;
    
    static int[] oddNodes;
    static int K;
    
    static List<int> duplicatedEdges = new List<int>();

    static long[] min_dist;
    static int[] dist_id;
    static int current_search_id = 0;
    static int[] parent_edge;
    static int[] parent_node;

    class Scanner {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 64];
        private int head, tail;

        public Scanner(Stream stream) {
            this.stream = stream;
        }

        public int NextInt() {
            int c = Read();
            while (c <= ' ') {
                if (c == -1) return -1;
                c = Read();
            }
            int sign = 1;
            if (c == '-') {
                sign = -1;
                c = Read();
            }
            int res = 0;
            while (c > ' ') {
                res = res * 10 + c - '0';
                c = Read();
            }
            return res * sign;
        }

        private int Read() {
            if (head >= tail) {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }

    static void Main(string[] args) {
        Solve();
    }

    static void Solve() {
        using Stream inputStream = Console.OpenStandardInput();
        Scanner scanner = new Scanner(inputStream);
        
        int nVal = scanner.NextInt();
        if (nVal == -1) return;
        N = nVal;
        M = scanner.NextInt();
        
        if (M == 0) {
            Console.WriteLine("0");
            return;
        }
        
        originalEdges = new OriginalEdge[M];
        originalAdj = new List<OriginalEdge>[N];
        for(int i = 0; i < N; i++) originalAdj[i] = new List<OriginalEdge>();
        
        int[] deg = new int[N];
        for(int i = 0; i < M; i++) {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt();
            
            originalEdges[i] = new OriginalEdge { u = u, v = v, w = w, id = i };
            originalAdj[u].Add(originalEdges[i]);
            originalAdj[v].Add(originalEdges[i]);
            deg[u]++;
            deg[v]++;
        }
        
        List<int> oddList = new List<int>();
        for(int i = 0; i < N; i++) {
            if (deg[i] % 2 != 0) {
                oddList.Add(i);
            }
        }
        oddNodes = oddList.ToArray();
        K = oddNodes.Length;
        
        min_dist = new long[N];
        dist_id = new int[N];
        parent_edge = new int[N];
        parent_node = new int[N];
        
        if (K > 0) {
            if (K <= 20) {
                SolveDP();
            } else if (K <= 500) {
                SolveGreedyAllPairs();
            } else {
                SolveGreedyNearest();
            }
        }
        
        adj = new List<Edge>[N];
        for(int i = 0; i < N; i++) adj[i] = new List<Edge>();
        
        for(int i = 0; i < M; i++) {
            AddEdgeToAdj(originalEdges[i].u, originalEdges[i].v, i);
        }
        int newId = M;
        foreach(int eId in duplicatedEdges) {
            AddEdgeToAdj(originalEdges[eId].u, originalEdges[eId].v, newId++);
        }
        
        List<int> circuit = new List<int>();
        int[] edgeIndex = new int[N];
        Stack<int> stack = new Stack<int>();
        stack.Push(0); 
        
        while(stack.Count > 0) {
            int u = stack.Peek();
            bool found = false;
            while(edgeIndex[u] < adj[u].Count) {
                Edge e = adj[u][edgeIndex[u]++];
                if (!e.used) {
                    e.used = true;
                    e.rev.used = true;
                    stack.Push(e.to);
                    found = true;
                    break;
                }
            }
            if (!found) {
                circuit.Add(stack.Pop());
            }
        }
        circuit.Reverse();
        
        using StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
        for (int i = 0; i < circuit.Count; i++) {
            writer.Write(circuit[i]);
            if (i < circuit.Count - 1) writer.Write(' ');
        }
        writer.WriteLine();
    }

    static void AddEdgeToAdj(int u, int v, int id) {
        Edge e1 = new Edge { to = v, id = id };
        Edge e2 = new Edge { to = u, id = id };
        e1.rev = e2;
        e2.rev = e1;
        adj[u].Add(e1);
        adj[v].Add(e2);
    }

    static void SolveDP() {
        long[,] oddDist = new long[K, K];
        int[,][] oddPathEdges = new int[K, K][];

        for(int i = 0; i < K; i++) {
            ComputeAllPathsFrom(oddNodes[i], out long[] dists, out int[] pEdge, out int[] pNode);
            for(int j = i + 1; j < K; j++) {
                int target = oddNodes[j];
                oddDist[i, j] = dists[target];
                oddDist[j, i] = dists[target];
                
                if (dists[target] != long.MaxValue) {
                    List<int> edges = new List<int>();
                    int curr = target;
                    while(curr != oddNodes[i]) {
                        edges.Add(pEdge[curr]);
                        curr = pNode[curr];
                    }
                    edges.Reverse();
                    oddPathEdges[i, j] = edges.ToArray();
                }
            }
        }

        long[] dp = new long[1 << K];
        int[] dp_prev = new int[1 << K];
        int[] dp_pair = new int[1 << K];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;

        for(int mask = 0; mask < (1 << K); mask++) {
            if (dp[mask] >= long.MaxValue / 2) continue;
            
            int i = -1;
            for(int b = 0; b < K; b++) {
                if ((mask & (1 << b)) == 0) {
                    i = b;
                    break;
                }
            }
            if (i == -1) continue;
            
            for(int j = i + 1; j < K; j++) {
                if ((mask & (1 << j)) == 0) {
                    if (oddDist[i, j] == long.MaxValue) continue;
                    int nextMask = mask | (1 << i) | (1 << j);
                    long nCost = dp[mask] + oddDist[i, j];
                    if (nCost < dp[nextMask]) {
                        dp[nextMask] = nCost;
                        dp_prev[nextMask] = mask;
                        dp_pair[nextMask] = j;
                    }
                }
            }
        }

        int currMask = (1 << K) - 1;
        while(currMask > 0) {
            int prevMask = dp_prev[currMask];
            int diff = currMask ^ prevMask;
            int i = -1, j = -1;
            for(int b = 0; b < K; b++) {
                if ((diff & (1 << b)) != 0) {
                    if (i == -1) i = b;
                    else j = b;
                }
            }
            
            if (oddPathEdges[i, j] != null) {
                foreach(int eId in oddPathEdges[i, j]) {
                    duplicatedEdges.Add(eId);
                }
            }
            
            currMask = prevMask;
        }
    }

    static void SolveGreedyAllPairs() {
        long[,] oddDist = new long[K, K];
        int[,][] oddPathEdges = new int[K, K][];

        for(int i = 0; i < K; i++) {
            ComputeAllPathsFrom(oddNodes[i], out long[] dists, out int[] pEdge, out int[] pNode);
            for(int j = i + 1; j < K; j++) {
                int target = oddNodes[j];
                oddDist[i, j] = dists[target];
                oddDist[j, i] = dists[target];
                
                if (dists[target] != long.MaxValue) {
                    List<int> edges = new List<int>();
                    int curr = target;
                    while(curr != oddNodes[i]) {
                        edges.Add(pEdge[curr]);
                        curr = pNode[curr];
                    }
                    edges.Reverse();
                    oddPathEdges[i, j] = edges.ToArray();
                }
            }
        }

        List<Pair> pairs = new List<Pair>();
        for(int i = 0; i < K; i++) {
            for(int j = i + 1; j < K; j++) {
                if (oddDist[i, j] != long.MaxValue) {
                    pairs.Add(new Pair { i = i, j = j, dist = oddDist[i, j] });
                }
            }
        }
        pairs.Sort();

        bool[] matched = new bool[K];
        foreach(var p in pairs) {
            if (!matched[p.i] && !matched[p.j]) {
                matched[p.i] = true;
                matched[p.j] = true;
                foreach(int eId in oddPathEdges[p.i, p.j]) {
                    duplicatedEdges.Add(eId);
                }
            }
        }
    }

    static void SolveGreedyNearest() {
        bool[] isUnmatchedOdd = new bool[N];
        foreach (int u in oddNodes) isUnmatchedOdd[u] = true;

        for (int i = 0; i < K; i++) {
            int start = oddNodes[i];
            if (!isUnmatchedOdd[start]) continue;
            
            int target = FindNearestUnmatched(start, isUnmatchedOdd, out List<int> pathEdges);
            
            if (target != -1) {
                isUnmatchedOdd[start] = false;
                isUnmatchedOdd[target] = false;
                foreach (int eId in pathEdges) {
                    duplicatedEdges.Add(eId);
                }
            }
        }
    }

    static void ComputeAllPathsFrom(int start, out long[] outDist, out int[] outPEdge, out int[] outPNode) {
        outDist = new long[N];
        outPEdge = new int[N];
        outPNode = new int[N];
        Array.Fill(outDist, long.MaxValue);
        
        current_search_id++;
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        pq.Enqueue(start, 0);
        dist_id[start] = current_search_id;
        min_dist[start] = 0;
        outDist[start] = 0;
        
        while(pq.Count > 0) {
            pq.TryDequeue(out int u_curr, out long d);
            if (min_dist[u_curr] < d) continue;
            
            foreach(var edge in originalAdj[u_curr]) {
                int v = edge.u == u_curr ? edge.v : edge.u;
                long newDist = d + edge.w;
                bool update = false;
                
                if (dist_id[v] != current_search_id) {
                    update = true;
                } else if (newDist < min_dist[v]) {
                    update = true;
                }
                
                if (update) {
                    dist_id[v] = current_search_id;
                    min_dist[v] = newDist;
                    outDist[v] = newDist;
                    outPEdge[v] = edge.id;
                    outPNode[v] = u_curr;
                    pq.Enqueue(v, newDist);
                }
            }
        }
    }

    static int FindNearestUnmatched(int start, bool[] isUnmatchedOdd, out List<int> pathEdges) {
        current_search_id++;
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        
        pq.Enqueue(start, 0);
        dist_id[start] = current_search_id;
        min_dist[start] = 0;
        
        int target = -1;
        
        while (pq.Count > 0) {
            pq.TryDequeue(out int u_curr, out long d);
            if (min_dist[u_curr] < d) continue;
            
            if (u_curr != start && isUnmatchedOdd[u_curr]) {
                target = u_curr;
                break;
            }
            
            foreach (var edge in originalAdj[u_curr]) {
                int v = edge.u == u_curr ? edge.v : edge.u;
                long newDist = d + edge.w;
                bool update = false;
                
                if (dist_id[v] != current_search_id) {
                    update = true;
                } else if (newDist < min_dist[v]) {
                    update = true;
                }
                
                if (update) {
                    dist_id[v] = current_search_id;
                    min_dist[v] = newDist;
                    parent_edge[v] = edge.id;
                    parent_node[v] = u_curr;
                    pq.Enqueue(v, newDist);
                }
            }
        }
        
        pathEdges = new List<int>();
        if (target != -1) {
            int curr = target;
            while (curr != start) {
                pathEdges.Add(parent_edge[curr]);
                curr = parent_node[curr];
            }
            pathEdges.Reverse();
        }
        return target;
    }
}
