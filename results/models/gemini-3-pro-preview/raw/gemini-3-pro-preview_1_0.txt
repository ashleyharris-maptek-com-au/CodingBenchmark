#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstdint>
#include <chrono>

using namespace std;

struct Point {
    float x, y;
    int id;
};

inline float distSq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

inline float dist(const Point& a, const Point& b) {
    return std::sqrt(distSq(a, b));
}

uint64_t hilbert_code(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int i = 31; i >= 0; i--) {
        uint32_t rx = (x >> i) & 1;
        uint32_t ry = (y >> i) & 1;
        uint64_t quad = ((3 * rx) ^ ry);
        d = (d << 2) | quad;
        
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

struct CityNode {
    int id;
    uint64_t hcode;
    bool operator<(const CityNode& o) const {
        return hcode < o.hcode;
    }
};

vector<int> generate_hilbert_tour(const vector<Point>& pts) {
    int n = pts.size();
    float min_x = pts[0].x, max_x = pts[0].x;
    float min_y = pts[0].y, max_y = pts[0].y;
    for (int i = 1; i < n; i++) {
        if (pts[i].x < min_x) min_x = pts[i].x;
        if (pts[i].x > max_x) max_x = pts[i].x;
        if (pts[i].y < min_y) min_y = pts[i].y;
        if (pts[i].y > max_y) max_y = pts[i].y;
    }
    
    float max_len = max_x - min_x;
    if (max_y - min_y > max_len) max_len = max_y - min_y;
    if (max_len == 0.0f) max_len = 1.0f;
    
    vector<CityNode> nodes(n);
    for (int i = 0; i < n; i++) {
        uint32_t nx = (uint32_t)(((pts[i].x - min_x) / max_len) * 4294967000.0f);
        uint32_t ny = (uint32_t)(((pts[i].y - min_y) / max_len) * 4294967000.0f);
        nodes[i].id = i;
        nodes[i].hcode = hilbert_code(nx, ny);
    }
    
    sort(nodes.begin(), nodes.end());
    
    vector<int> tour(n);
    for (int i = 0; i < n; i++) {
        tour[i] = nodes[i].id;
    }
    return tour;
}

vector<int> generate_nn_tour(const vector<Point>& pts, int start_idx = 0) {
    int n = pts.size();
    vector<int> tour;
    tour.reserve(n);
    
    vector<int> unvisited;
    unvisited.reserve(n - 1);
    for(int i = 0; i < n; i++) {
        if (i != start_idx) unvisited.push_back(i);
    }
    
    int curr = start_idx;
    tour.push_back(curr);
    
    for (int step = 1; step < n; step++) {
        int best_idx_in_unvisited = -1;
        float best_dist = 1e30f;
        float cx = pts[curr].x;
        float cy = pts[curr].y;
        
        int unv_size = unvisited.size();
        for (int i = 0; i < unv_size; i++) {
            int uid = unvisited[i];
            float dx = cx - pts[uid].x;
            float dy = cy - pts[uid].y;
            float d = dx * dx + dy * dy;
            if (d < best_dist) {
                best_dist = d;
                best_idx_in_unvisited = i;
            }
        }
        curr = unvisited[best_idx_in_unvisited];
        tour.push_back(curr);
        
        unvisited[best_idx_in_unvisited] = unvisited.back();
        unvisited.pop_back();
    }
    return tour;
}

template<typename TimePoint>
void apply_2opt(vector<int>& tour, const vector<Point>& pts, int max_window, int max_passes, TimePoint start_time) {
    int n = tour.size();
    if (n < 4) return;
    
    vector<float> edge_len(n);
    for (int k = 0; k < n; k++) {
        edge_len[k] = dist(pts[tour[k]], pts[tour[(k+1)%n]]);
    }
    
    bool improved = true;
    int passes = 0;
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < n - 2; i++) {
            if ((i & 1023) == 0) {
                auto now = chrono::steady_clock::now();
                if (chrono::duration_cast<chrono::seconds>(now - start_time).count() > 280) {
                    return;
                }
            }
            int j_end = n;
            if (max_window > 0) j_end = min(n, i + max_window);
            
            int t_i = tour[i];
            int t_i1 = tour[i+1];
            float d1 = edge_len[i];
            
            for (int j = i + 2; j < j_end; j++) {
                int t_j = tour[j];
                int t_j1 = tour[(j+1)%n];
                
                float d2 = edge_len[j];
                float current_dist = d1 + d2;
                float cd_sq = current_dist * current_dist;
                
                float d3_sq = distSq(pts[t_i], pts[t_j]);
                if (d3_sq >= cd_sq) continue;
                
                float d4_sq = distSq(pts[t_i1], pts[t_j1]);
                if (d4_sq >= cd_sq) continue;
                
                float d3 = std::sqrt(d3_sq);
                if (d3 >= current_dist) continue;
                
                float d4 = std::sqrt(d4_sq);
                float improvement = current_dist - (d3 + d4);
                
                if (improvement > 0.0f && improvement > current_dist * 1e-6f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    reverse(edge_len.begin() + i + 1, edge_len.begin() + j);
                    edge_len[i] = d3;
                    edge_len[j] = d4;
                    improved = true;
                    break;
                }
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    auto global_start = chrono::steady_clock::now();
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<Point> pts(n);
    for (int i = 0; i < n; i++) {
        cin >> pts[i].x >> pts[i].y;
        pts[i].id = i;
    }
    
    vector<int> best_tour;
    double best_tour_len = 1e30;
    
    auto calc_tour_len = [&](const vector<int>& t) {
        double len = 0;
        for (int i = 0; i < n; i++) {
            len += dist(pts[t[i]], pts[t[(i+1)%n]]);
        }
        return len;
    };

    if (n <= 100) {
        for (int start = 0; start < n; start++) {
            vector<int> tour = generate_nn_tour(pts, start);
            apply_2opt(tour, pts, 0, 100, global_start);
            double len = calc_tour_len(tour);
            if (len < best_tour_len) {
                best_tour_len = len;
                best_tour = tour;
            }
            auto now = chrono::steady_clock::now();
            if (chrono::duration_cast<chrono::seconds>(now - global_start).count() > 280) break;
        }
    } else if (n <= 1000) {
        int num_starts = min(n, 40);
        int step = n / num_starts;
        for (int i = 0; i < num_starts; i++) {
            int start = i * step;
            vector<int> tour = generate_nn_tour(pts, start);
            apply_2opt(tour, pts, 0, 50, global_start);
            double len = calc_tour_len(tour);
            if (len < best_tour_len) {
                best_tour_len = len;
                best_tour = tour;
            }
            auto now = chrono::steady_clock::now();
            if (chrono::duration_cast<chrono::seconds>(now - global_start).count() > 280) break;
        }
    } else if (n <= 50000) {
        best_tour = generate_nn_tour(pts, 0);
        apply_2opt(best_tour, pts, 0, 30, global_start);
    } else {
        best_tour = generate_hilbert_tour(pts);
        
        int window = 2000;
        if (n > 100000) window = 1000;
        if (n > 500000) window = 500;
        if (n > 1000000) window = 200;
        if (n > 5000000) window = 100;
        
        apply_2opt(best_tour, pts, window, 5, global_start);
    }
    
    int zero_idx = -1;
    for (int i = 0; i < n; i++) {
        if (best_tour[i] == 0) {
            zero_idx = i;
            break;
        }
    }
    
    for (int i = 0; i < n; i++) {
        cout << best_tour[(zero_idx + i) % n] << (i == n - 1 ? "" : " ");
    }
    cout << "\n";
    
    return 0;
}
