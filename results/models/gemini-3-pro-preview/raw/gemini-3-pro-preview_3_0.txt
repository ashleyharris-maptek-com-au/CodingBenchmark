use std::io::{self, BufRead, Read, Write, BufWriter};
use std::time::Instant;

struct QuadNode {
    mass: f32,
    cx: f32,
    cy: f32,
    is_leaf: bool,
    body: Option<usize>,
    children: [usize; 4],
}

impl QuadNode {
    fn new() -> Self {
        QuadNode {
            mass: 0.0,
            cx: 0.0,
            cy: 0.0,
            is_leaf: true,
            body: None,
            children: [0; 4],
        }
    }
}

fn get_quadrant(px: f32, py: f32, xmin: f32, ymin: f32, size: f32) -> usize {
    let half = size / 2.0;
    let left = px < xmin + half;
    let bottom = py < ymin + half;
    match (left, bottom) {
        (true, true) => 0,
        (false, true) => 1,
        (true, false) => 2,
        (false, false) => 3,
    }
}

fn child_bounds(xmin: f32, ymin: f32, size: f32, q: usize) -> (f32, f32, f32) {
    let half = size / 2.0;
    match q {
        0 => (xmin, ymin, half),
        1 => (xmin + half, ymin, half),
        2 => (xmin, ymin + half, half),
        3 => (xmin + half, ymin + half, half),
        _ => unreachable!(),
    }
}

fn insert(
    tree: &mut Vec<QuadNode>,
    b_id: usize,
    x: &[f32],
    y: &[f32],
    mut node_idx: usize,
    mut xmin: f32,
    mut ymin: f32,
    mut size: f32,
    mut depth: usize,
) {
    loop {
        if tree[node_idx].is_leaf {
            if tree[node_idx].body.is_none() {
                tree[node_idx].body = Some(b_id);
                return;
            }
            if depth >= 60 {
                return; // Max depth reached, safely drop this node from layout tree for this iteration
            }

            let old_body = tree[node_idx].body.unwrap();
            tree[node_idx].body = None;
            tree[node_idx].is_leaf = false;

            let base = tree.len();
            tree.push(QuadNode::new());
            tree.push(QuadNode::new());
            tree.push(QuadNode::new());
            tree.push(QuadNode::new());
            tree[node_idx].children = [base, base + 1, base + 2, base + 3];

            let ox = x[old_body];
            let oy = y[old_body];
            let o_q = get_quadrant(ox, oy, xmin, ymin, size);
            let child_idx = tree[node_idx].children[o_q];
            let (cx, cy, csize) = child_bounds(xmin, ymin, size, o_q);
            insert(tree, old_body, x, y, child_idx, cx, cy, csize, depth + 1);
        }

        let q = get_quadrant(x[b_id], y[b_id], xmin, ymin, size);
        node_idx = tree[node_idx].children[q];
        let (nx, ny, ns) = child_bounds(xmin, ymin, size, q);
        xmin = nx;
        ymin = ny;
        size = ns;
        depth += 1;
    }
}

fn compute_mass(
    tree: &mut Vec<QuadNode>,
    node_idx: usize,
    x: &[f32],
    y: &[f32],
    masses: &[f32],
) {
    if tree[node_idx].is_leaf {
        if let Some(b) = tree[node_idx].body {
            tree[node_idx].mass = masses[b];
            tree[node_idx].cx = x[b];
            tree[node_idx].cy = y[b];
        } else {
            tree[node_idx].mass = 0.0;
            tree[node_idx].cx = 0.0;
            tree[node_idx].cy = 0.0;
        }
    } else {
        let mut total_mass = 0.0;
        let mut sum_x = 0.0;
        let mut sum_y = 0.0;
        let children = tree[node_idx].children;
        for &c in &children {
            compute_mass(tree, c, x, y, masses);
            let cmass = tree[c].mass;
            total_mass += cmass;
            sum_x += tree[c].cx * cmass;
            sum_y += tree[c].cy * cmass;
        }
        tree[node_idx].mass = total_mass;
        if total_mass > 0.0 {
            tree[node_idx].cx = sum_x / total_mass;
            tree[node_idx].cy = sum_y / total_mass;
        }
    }
}

fn repulse(
    node_idx: usize,
    tree: &[QuadNode],
    px: f32,
    py: f32,
    kr: f32,
    theta: f32,
    size: f32,
    fx: &mut f32,
    fy: &mut f32,
) {
    let node = &tree[node_idx];
    if node.mass == 0.0 {
        return;
    }

    let mut dx = px - node.cx;
    let mut dy = py - node.cy;
    let mut dist_sq = dx * dx + dy * dy;

    if dist_sq < 1e-6 {
        dx = 1e-3;
        dy = 1e-3;
        dist_sq = dx * dx + dy * dy;
    }

    if node.is_leaf {
        let dist = dist_sq.sqrt();
        let force = kr * node.mass / dist;
        *fx += (dx / dist) * force;
        *fy += (dy / dist) * force;
    } else {
        let dist = dist_sq.sqrt();
        if size / dist < theta {
            let force = kr * node.mass / dist;
            *fx += (dx / dist) * force;
            *fy += (dy / dist) * force;
        } else {
            let half = size / 2.0;
            for &c in &node.children {
                repulse(c, tree, px, py, kr, theta, half, fx, fy);
            }
        }
    }
}

fn next_f32(state: &mut u32) -> f32 {
    let mut x = *state;
    x ^= x << 13;
    x ^= x >> 17;
    x ^= x << 5;
    *state = x;
    (x as f32) / (u32::MAX as f32)
}

fn main() {
    let start_time = Instant::now();

    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    let mut buffer = String::new();

    if stdin_lock.read_line(&mut buffer).is_err() {
        return;
    }
    let mut parts = buffer.split_whitespace();
    let n_str = parts.next();
    let m_str = parts.next();
    if n_str.is_none() || m_str.is_none() {
        return;
    }
    let n: usize = n_str.unwrap().parse().unwrap();
    let m: usize = m_str.unwrap().parse().unwrap();

    if n == 0 {
        return;
    }

    let mut edges = Vec::with_capacity(m);
    buffer.clear();
    stdin_lock.read_to_string(&mut buffer).unwrap();
    let mut parts = buffer.split_whitespace();
    while let (Some(u_str), Some(v_str)) = (parts.next(), parts.next()) {
        let mut u: usize = u_str.parse().unwrap();
        let mut v: usize = v_str.parse().unwrap();
        if u != v {
            if u > v {
                std::mem::swap(&mut u, &mut v);
            }
            edges.push((u, v));
        }
    }

    // Deduplicate to avoid excessive forces from parallel edges
    edges.sort_unstable();
    edges.dedup();

    let mut degree = vec![0.0; n];
    for &(u, v) in &edges {
        degree[u] += 1.0;
        degree[v] += 1.0;
    }

    let mass: Vec<f32> = degree.into_iter().map(|d| d + 1.0).collect();

    let radius = (n as f32).sqrt().max(10.0);
    let mut rng_state = 123456789u32;
    let mut pos_x: Vec<f32> = (0..n)
        .map(|_| (next_f32(&mut rng_state) * 2.0 - 1.0) * radius)
        .collect();
    let mut pos_y: Vec<f32> = (0..n)
        .map(|_| (next_f32(&mut rng_state) * 2.0 - 1.0) * radius)
        .collect();

    let num_threads = std::thread::available_parallelism()
        .map(|n| n.get())
        .unwrap_or(4);

    let kr = 2.0;
    let kg = 0.05;
    let theta = 0.8;

    let target_iters = match n {
        0..=100 => 1200,
        101..=1000 => 1000,
        1001..=10000 => 600,
        10001..=50000 => 400,
        _ => 200,
    };

    let initial_temp = radius * 5.0;
    let min_temp = 0.1;
    let mut temp = initial_temp;
    let cool_factor = (min_temp / initial_temp).powf(1.0 / target_iters as f32);

    let mut force_x = vec![0.0; n];
    let mut force_y = vec![0.0; n];
    let mut tree = Vec::with_capacity(n * 4);
    let chunk_size = (n + num_threads - 1) / num_threads;

    let mut iter = 0;
    while iter < target_iters {
        if start_time.elapsed().as_secs() >= 28 {
            break;
        }

        let mut min_x = pos_x[0];
        let mut max_x = pos_x[0];
        let mut min_y = pos_y[0];
        let mut max_y = pos_y[0];

        for i in 1..n {
            if pos_x[i] < min_x { min_x = pos_x[i]; }
            if pos_x[i] > max_x { max_x = pos_x[i]; }
            if pos_y[i] < min_y { min_y = pos_y[i]; }
            if pos_y[i] > max_y { max_y = pos_y[i]; }
        }

        let root_size = (max_x - min_x).max(max_y - min_y) + 1e-4;

        tree.clear();
        tree.push(QuadNode::new());

        for i in 0..n {
            insert(&mut tree, i, &pos_x, &pos_y, 0, min_x, min_y, root_size, 0);
        }

        compute_mass(&mut tree, 0, &pos_x, &pos_y, &mass);

        force_x.fill(0.0);
        force_y.fill(0.0);

        std::thread::scope(|s| {
            for (i, (fx_chunk, fy_chunk)) in force_x
                .chunks_mut(chunk_size)
                .zip(force_y.chunks_mut(chunk_size))
                .enumerate()
            {
                let start_idx = i * chunk_size;
                let px = &pos_x;
                let py = &pos_y;
                let m = &mass;
                let tr = &tree;

                s.spawn(move || {
                    for j in 0..fx_chunk.len() {
                        let u = start_idx + j;
                        let mut fx = 0.0;
                        let mut fy = 0.0;
                        repulse(0, tr, px[u], py[u], kr, theta, root_size, &mut fx, &mut fy);
                        fx *= m[u];
                        fy *= m[u];

                        fx += -kg * m[u] * px[u];
                        fy += -kg * m[u] * py[u];

                        fx_chunk[j] = fx;
                        fy_chunk[j] = fy;
                    }
                });
            }
        });

        for &(u, v) in &edges {
            let dx = pos_x[v] - pos_x[u];
            let dy = pos_y[v] - pos_y[u];
            force_x[u] += dx;
            force_y[u] += dy;
            force_x[v] -= dx;
            force_y[v] -= dy;
        }

        for u in 0..n {
            let dx = force_x[u] / mass[u];
            let dy = force_y[u] / mass[u];
            if dx.is_nan() || dy.is_nan() {
                continue;
            }
            let d = (dx * dx + dy * dy).sqrt();
            if d > 0.0 && !d.is_nan() {
                let step = d.min(temp);
                pos_x[u] += (dx / d) * step;
                pos_y[u] += (dy / d) * step;
            }
        }

        temp *= cool_factor;
        iter += 1;
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", pos_x[i], pos_y[i]).unwrap();
    }
}
