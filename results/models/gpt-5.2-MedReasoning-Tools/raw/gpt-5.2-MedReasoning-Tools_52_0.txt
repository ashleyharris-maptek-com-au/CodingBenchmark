# Autopilot implementation for benchmark.
# Function signature:
#   def autopilot_step(state: dict, dt: float) -> dict:

import math
from typing import Dict, Any, List, Optional, Tuple

# Persistent memory across calls
_mem: Dict[str, Any] = {
    "init": False,
    "t": 0.0,
    "speed_mps": 0.0,
    "yaw_rate_dps": 0.0,
    "heading_deg": None,
    "pos": None,  # (x, y)
    "lane_offset_m": 0.0,
    "curvature_1pm": 0.0,
    "speed_i": 0.0,
    "steer_i": 0.0,
    "last_cmd": {"steering": 0.0, "throttle": 0.0, "brake": 0.0},
    "speed_hist": [],  # recent speeds for estimating acceleration
    "fault": {
        "brake_weak": False,
        "throttle_stuck": False,
        "steer_weak": False,
    },
}


def _clamp(x: float, lo: float, hi: float) -> float:
    return lo if x < lo else hi if x > hi else x


def _wrap_deg(a: float) -> float:
    # wrap to [-180, 180)
    a = (a + 180.0) % 360.0 - 180.0
    return a


def _median(xs: List[float]) -> Optional[float]:
    if not xs:
        return None
    xs2 = sorted(xs)
    n = len(xs2)
    mid = n // 2
    if n % 2 == 1:
        return xs2[mid]
    return 0.5 * (xs2[mid - 1] + xs2[mid])


def _robust_fuse(
    candidates: List[float],
    prev: Optional[float],
    max_jump: float,
    hard_min: float,
    hard_max: float,
) -> Optional[float]:
    # Filter grossly implausible values; then take median; then limit jump.
    vals = [v for v in candidates if v is not None and hard_min <= v <= hard_max]
    if not vals:
        return prev
    m = _median(vals)
    if m is None:
        return prev
    if prev is None:
        return m
    return _clamp(m, prev - max_jump, prev + max_jump)


def _ema(prev: float, meas: float, alpha: float) -> float:
    return prev + alpha * (meas - prev)


def _get(state: Dict[str, Any], key: str, default=None):
    try:
        return state.get(key, default)
    except Exception:
        return default


def _vision_measurements(state: Dict[str, Any]) -> Dict[str, List[float]]:
    lane_offsets: List[float] = []
    curvatures: List[float] = []
    speed_limits: List[float] = []
    obs_dists: List[float] = []
    obs_rel_speeds: List[float] = []

    for i in (1, 2, 3):
        lv = bool(_get(state, f"vision_{i}_left_line_valid", False))
        rv = bool(_get(state, f"vision_{i}_right_line_valid", False))
        ov = bool(_get(state, f"vision_{i}_obstacle_valid", False))
        slv = bool(_get(state, f"vision_{i}_speed_limit_valid", False))

        # Lane offset direct
        lo = _get(state, f"vision_{i}_lane_offset_m", None)
        if isinstance(lo, (int, float)) and abs(lo) <= 3.0:
            lane_offsets.append(float(lo))

        # Infer offset from line distances when both valid
        ld = _get(state, f"vision_{i}_left_line_dist_m", None)
        rd = _get(state, f"vision_{i}_right_line_dist_m", None)
        if lv and rv and isinstance(ld, (int, float)) and isinstance(rd, (int, float)):
            ld = float(ld)
            rd = float(rd)
            if 0.5 <= ld <= 5.0 and 0.5 <= rd <= 5.0:
                # If car centered, ld ~= rd. Offset positive means car is right of center.
                lane_offsets.append(0.5 * (rd - ld))

        # Curvature
        c = _get(state, f"vision_{i}_road_curvature_1pm", None)
        if isinstance(c, (int, float)) and abs(c) <= 0.05:  # up to 1/20m radius
            curvatures.append(float(c))

        # Speed limit
        sl = _get(state, f"vision_{i}_speed_limit_kmh", None)
        if slv and isinstance(sl, (int, float)) and 0.0 < float(sl) <= 160.0:
            speed_limits.append(float(sl))

        # Obstacles
        od = _get(state, f"vision_{i}_obstacle_dist_m", None)
        orr = _get(state, f"vision_{i}_obstacle_rel_speed_kmh", None)
        detected = bool(_get(state, f"vision_{i}_obstacle_detected", False))
        if detected and isinstance(od, (int, float)):
            od = float(od)
            if 0.5 <= od <= 250.0:
                obs_dists.append(od)
                if isinstance(orr, (int, float)):
                    obs_rel_speeds.append(float(orr))

    return {
        "lane_offsets": lane_offsets,
        "curvatures": curvatures,
        "speed_limits": speed_limits,
        "obs_dists": obs_dists,
        "obs_rel_speeds": obs_rel_speeds,
    }


def _lidar_front_range(state: Dict[str, Any]) -> Optional[float]:
    # Use near-forward beams: +/-5, +/-15, +/-30 degrees
    angles = [
        "+005",
        "-005",
        "+015",
        "-015",
        "+030",
        "-030",
    ]
    ranges: List[float] = []
    for a in angles:
        rk = f"lidar_{a}_deg_range_m"
        vk = f"lidar_{a}_deg_valid"
        v = _get(state, rk, None)
        ok = _get(state, vk, None)
        if isinstance(v, (int, float)):
            v = float(v)
            if 0.2 <= v <= 300.0:
                # Do not trust valid flag fully; just treat it as a mild gate.
                if ok is True or ok is False:
                    if ok:
                        ranges.append(v)
                else:
                    ranges.append(v)
    if not ranges:
        return None
    return min(ranges)


def _estimate_mu(state: Dict[str, Any]) -> float:
    mu = _get(state, "estimated_mu", None)
    if isinstance(mu, (int, float)):
        mu = float(mu)
        return _clamp(mu, 0.05, 1.2)
    return 1.0


def _fuse_speed_kmh(state: Dict[str, Any], dt: float) -> Optional[float]:
    cands: List[float] = []

    # Wheel speeds
    for corner in ("fl", "fr", "rl", "rr"):
        v = _get(state, f"wheel_speed_{corner}_kmh", None)
        if isinstance(v, (int, float)):
            v = float(v)
            if -5.0 <= v <= 260.0:
                cands.append(max(0.0, v))

    # Vehicle speed sensors
    for k in ("vehicle_speed_1_kmh", "vehicle_speed_2_kmh"):
        v = _get(state, k, None)
        if isinstance(v, (int, float)):
            v = float(v)
            if -5.0 <= v <= 260.0:
                cands.append(max(0.0, v))

    # GPS speed
    for i in (1, 2):
        if bool(_get(state, f"gps_{i}_valid", False)):
            v = _get(state, f"gps_{i}_speed_kmh", None)
            if isinstance(v, (int, float)):
                v = float(v)
                if -5.0 <= v <= 260.0:
                    cands.append(max(0.0, v))

    # Speed from GPS position delta (even if gps valid flickers)
    pos_cands: List[Tuple[float, float]] = []
    for i in (1, 2):
        x = _get(state, f"gps_{i}_x_m", None)
        y = _get(state, f"gps_{i}_y_m", None)
        if isinstance(x, (int, float)) and isinstance(y, (int, float)):
            pos_cands.append((float(x), float(y)))
    if pos_cands and _mem.get("pos") is not None and dt > 1e-3:
        # Use median position to reduce outliers
        xs = [p[0] for p in pos_cands]
        ys = [p[1] for p in pos_cands]
        x = _median(xs)
        y = _median(ys)
        if x is not None and y is not None:
            px, py = _mem["pos"]
            dx = x - px
            dy = y - py
            sp_mps = math.hypot(dx, dy) / dt
            sp_kmh = sp_mps * 3.6
            if 0.0 <= sp_kmh <= 260.0:
                cands.append(sp_kmh)

    if not cands:
        return None

    prev_kmh = _mem.get("speed_mps", 0.0) * 3.6
    fused = _robust_fuse(
        cands,
        prev=prev_kmh,
        max_jump=25.0 * dt + 2.0,  # allow moderate accel and sensor corrections
        hard_min=0.0,
        hard_max=260.0,
    )
    return fused


def _fuse_yaw_rate_dps(state: Dict[str, Any], dt: float) -> Optional[float]:
    cands: List[float] = []

    for i in (1, 2, 3):
        v = _get(state, f"gyro_{i}_yaw_rate_dps", None)
        if isinstance(v, (int, float)):
            v = float(v)
            if abs(v) <= 200.0:
                cands.append(v)

    # Derive from GPS heading change
    headings: List[float] = []
    for i in (1, 2):
        if bool(_get(state, f"gps_{i}_valid", False)):
            h = _get(state, f"gps_{i}_heading_deg", None)
            if isinstance(h, (int, float)):
                headings.append(float(h) % 360.0)
    if headings and _mem.get("heading_deg") is not None and dt > 1e-3:
        h = _median(headings)
        if h is not None:
            dh = _wrap_deg(h - float(_mem["heading_deg"]))
            cands.append(dh / dt)

    if not cands:
        return None

    prev = _mem.get("yaw_rate_dps", 0.0)
    fused = _robust_fuse(
        cands,
        prev=prev,
        max_jump=120.0 * dt + 5.0,
        hard_min=-200.0,
        hard_max=200.0,
    )
    return fused


def autopilot_step(state: dict, dt: float) -> dict:
    # Defensive dt
    if not isinstance(dt, (int, float)):
        dt = _get(state, "dt", 0.05)
    dt = float(dt)
    dt = _clamp(dt, 0.01, 0.2)

    # Initialize memory on first call
    if not _mem["init"]:
        _mem["init"] = True
        _mem["t"] = float(_get(state, "sim_time_s", 0.0) or 0.0)

    _mem["t"] += dt

    mu = _estimate_mu(state)
    rain = float(_get(state, "rain_sensor", 0.0) or 0.0)
    temp_c = float(_get(state, "temperature_c", 20.0) or 20.0)

    # Fuse GPS position + heading for dead-reckoning
    pos_cands: List[Tuple[float, float]] = []
    head_cands: List[float] = []
    for i in (1, 2):
        x = _get(state, f"gps_{i}_x_m", None)
        y = _get(state, f"gps_{i}_y_m", None)
        if isinstance(x, (int, float)) and isinstance(y, (int, float)):
            pos_cands.append((float(x), float(y)))
        if bool(_get(state, f"gps_{i}_valid", False)):
            h = _get(state, f"gps_{i}_heading_deg", None)
            if isinstance(h, (int, float)):
                head_cands.append(float(h) % 360.0)

    if pos_cands:
        xs = [p[0] for p in pos_cands]
        ys = [p[1] for p in pos_cands]
        x = _median(xs)
        y = _median(ys)
        if x is not None and y is not None:
            _mem["pos"] = (x, y)

    if head_cands:
        h = _median(head_cands)
        if h is not None:
            _mem["heading_deg"] = h

    # --- Sensor fusion: speed, yaw ---
    sp_kmh = _fuse_speed_kmh(state, dt)
    if sp_kmh is None:
        sp_kmh = _mem["speed_mps"] * 3.6
    sp_mps_meas = sp_kmh / 3.6

    yaw_dps = _fuse_yaw_rate_dps(state, dt)
    if yaw_dps is None:
        yaw_dps = _mem["yaw_rate_dps"]

    # Smooth with EMA (more smoothing on low mu / bad visibility)
    alpha_sp = _clamp(dt / (0.35 + dt), 0.05, 0.4)
    alpha_yaw = _clamp(dt / (0.25 + dt), 0.07, 0.5)
    alpha_sp *= _clamp(mu, 0.2, 1.0)

    _mem["speed_mps"] = max(0.0, _ema(_mem["speed_mps"], sp_mps_meas, alpha_sp))
    _mem["yaw_rate_dps"] = _ema(_mem["yaw_rate_dps"], float(yaw_dps), alpha_yaw)

    speed_mps = _mem["speed_mps"]

    # --- Vision + lane/curvature ---
    vm = _vision_measurements(state)
    lo_meas = _robust_fuse(vm["lane_offsets"], prev=_mem.get("lane_offset_m", 0.0), max_jump=0.8 * dt + 0.02, hard_min=-3.0, hard_max=3.0)
    if lo_meas is None:
        lo_meas = _mem.get("lane_offset_m", 0.0)

    curv_meas = _robust_fuse(vm["curvatures"], prev=_mem.get("curvature_1pm", 0.0), max_jump=0.04 * dt + 0.001, hard_min=-0.05, hard_max=0.05)
    if curv_meas is None:
        curv_meas = _mem.get("curvature_1pm", 0.0)

    # Extra smoothing when visibility reduced / sensors flicker
    vis_penalty = 1.0
    if rain > 0.6:
        vis_penalty *= 0.7
    if temp_c < 0.0:
        vis_penalty *= 0.8

    alpha_lane = _clamp(dt / (0.25 + dt), 0.05, 0.35) * vis_penalty
    alpha_curv = _clamp(dt / (0.45 + dt), 0.04, 0.25) * vis_penalty

    _mem["lane_offset_m"] = _ema(_mem["lane_offset_m"], float(lo_meas), alpha_lane)
    _mem["curvature_1pm"] = _ema(_mem["curvature_1pm"], float(curv_meas), alpha_curv)

    lane_offset_m = _mem["lane_offset_m"]
    curvature_1pm = _mem["curvature_1pm"]

    # --- Targets and constraints ---
    target_lane = int(_get(state, "target_lane", _get(state, "current_lane", 1)) or 1)
    current_lane = int(_get(state, "current_lane", target_lane) or target_lane)
    target_lane = max(0, min(2, target_lane))
    current_lane = max(0, min(2, current_lane))

    target_speed_kmh = _get(state, "target_speed_kmh", 100.0)
    if not isinstance(target_speed_kmh, (int, float)):
        target_speed_kmh = 100.0
    target_speed_kmh = float(target_speed_kmh)

    speed_limit_kmh = _median(vm["speed_limits"]) if vm["speed_limits"] else None
    if speed_limit_kmh is not None:
        target_speed_kmh = min(target_speed_kmh, float(speed_limit_kmh))

    # Traction management: reduce target on low mu
    # Also reduce a bit on heavy rain.
    mu_speed_scale = _clamp(0.35 + 0.65 * mu, 0.35, 1.0)
    if rain > 0.7:
        mu_speed_scale *= 0.85
    target_speed_kmh *= mu_speed_scale

    # --- Obstacle assessment ---
    lidar_front = _lidar_front_range(state)
    obs_dist = None
    if vm["obs_dists"]:
        obs_dist = min(vm["obs_dists"])
    if lidar_front is not None:
        obs_dist = lidar_front if obs_dist is None else min(obs_dist, lidar_front)

    # Relative speed: treat negative as closing, positive as opening; be conservative if unknown
    rel_kmh = _median(vm["obs_rel_speeds"]) if vm["obs_rel_speeds"] else None
    closing_mps = 0.0
    if rel_kmh is not None:
        # rel = obstacle - ego; negative => we are faster (closing)
        closing_mps = max(0.0, -float(rel_kmh) / 3.6)
    else:
        closing_mps = 0.0

    # Emergency logic (TTC)
    ttc = None
    if obs_dist is not None:
        # If we don't know closing speed, assume modest closing at (ego*0.5)
        assumed_close = max(closing_mps, 0.5 * speed_mps)
        denom = max(0.1, assumed_close)
        ttc = float(obs_dist) / denom

    # Decide whether to request a lane change due to obstacle
    want_lane_change = False
    if obs_dist is not None:
        # If obstacle within 45m or TTC small, prefer lane change (if target lane differs, we obey target)
        if obs_dist < 45.0 or (ttc is not None and ttc < 3.5):
            want_lane_change = True

    # --- Fault detection (actuator effectiveness) ---
    _mem["speed_hist"].append(speed_mps)
    if len(_mem["speed_hist"]) > 10:
        _mem["speed_hist"].pop(0)

    # Estimate longitudinal accel from speed history
    accel_est = 0.0
    if len(_mem["speed_hist"]) >= 3:
        # simple slope
        accel_est = (_mem["speed_hist"][-1] - _mem["speed_hist"][-3]) / (2.0 * dt)

    last_cmd = _mem["last_cmd"]
    # Throttle stuck: we command low throttle, but still see sustained positive acceleration.
    if last_cmd["throttle"] < 0.05 and last_cmd["brake"] < 0.05 and accel_est > 0.6 and speed_mps > 8.0:
        _mem["fault"]["throttle_stuck"] = True
    # Brake weak: we command strong brake but decel is small.
    if last_cmd["brake"] > 0.6 and accel_est > -0.5 and speed_mps > 10.0:
        _mem["fault"]["brake_weak"] = True
    # Steering weak: large steer command but yaw rate remains tiny at speed
    if abs(last_cmd["steering"]) > 0.5 and abs(_mem["yaw_rate_dps"]) < 1.0 and speed_mps > 15.0:
        _mem["fault"]["steer_weak"] = True

    # --- Lateral control (lane keeping + lane change) ---
    # Lane change: shift desired offset by +/- lane width if target lane differs.
    lane_width = 3.6
    lane_delta = target_lane - current_lane
    desired_offset = 0.0
    if lane_delta != 0:
        # Move toward adjacent lane center smoothly
        desired_offset = -lane_delta * lane_width

    # If obstacle ahead and we're not in target lane already, bias toward target lane earlier.
    # If obstacle ahead and target lane equals current lane, make a small offset bias away from obstacle.
    if want_lane_change and lane_delta == 0:
        # Bias within lane: drift slightly to create buffer
        desired_offset = _clamp(desired_offset + (0.4 if current_lane < 1 else -0.4), -1.2, 1.2)

    # Lateral error (positive means car right of desired center)
    e_lat = lane_offset_m - desired_offset

    # Desired yaw rate from curvature (deg/s)
    desired_yaw_dps = (speed_mps * curvature_1pm) * (180.0 / math.pi)
    e_yaw = _mem["yaw_rate_dps"] - desired_yaw_dps

    # Steering command: PD with gentle integral to remove bias
    # Gains reduced on low mu and if steering is weak.
    k_lat = 0.35
    k_yaw = 0.05
    k_i = 0.02

    traction_gain = _clamp(mu, 0.2, 1.0)
    k_lat *= traction_gain
    k_yaw *= traction_gain
    k_i *= traction_gain
    if _mem["fault"]["steer_weak"]:
        k_lat *= 0.7
        k_yaw *= 0.7

    _mem["steer_i"] = _clamp(_mem["steer_i"] + (-e_lat) * dt, -1.0, 1.0)
    steer_cmd = (-k_lat * e_lat) + (-k_yaw * e_yaw) + (k_i * _mem["steer_i"])

    # Crosswind/gust compensation using lateral accel when valid
    lat_accels: List[float] = []
    for i in (1, 2, 3):
        a = _get(state, f"accel_{i}_lateral_ms2", None)
        if isinstance(a, (int, float)) and abs(float(a)) <= 15.0:
            lat_accels.append(float(a))
    a_lat = _median(lat_accels) if lat_accels else 0.0
    # Small correction: if lateral accel indicates drift, counter-steer
    steer_cmd += _clamp(-0.03 * a_lat, -0.15, 0.15)

    # Limit steering rate and magnitude (gentler on low mu)
    steer_limit = 0.9 * _clamp(0.5 + 0.5 * mu, 0.4, 1.0)
    max_steer_rate = 1.8 * _clamp(mu, 0.25, 1.0)  # units per second
    prev_steer = float(last_cmd.get("steering", 0.0))
    steer_cmd = _clamp(steer_cmd, -steer_limit, steer_limit)
    steer_cmd = _clamp(steer_cmd, prev_steer - max_steer_rate * dt, prev_steer + max_steer_rate * dt)

    # --- Longitudinal control (speed/obstacles/traction) ---
    target_speed_mps = max(0.0, target_speed_kmh / 3.6)

    # Safe following / emergency braking
    brake_cmd = 0.0
    throttle_cmd = 0.0

    # Dynamic safety distance based on speed and mu
    base_headway = 1.4 + (1.5 if mu < 0.35 else 0.0)  # seconds
    safety_dist = 6.0 + speed_mps * base_headway

    urgent = False
    if obs_dist is not None:
        d = float(obs_dist)
        # If very close, urgent
        if d < max(8.0, 0.6 * safety_dist):
            urgent = True
        # If TTC short, urgent
        if ttc is not None and ttc < (2.2 if mu > 0.5 else 2.8):
            urgent = True

        if urgent:
            # Compute required decel to stop within distance with margin
            v = speed_mps
            # d_eff: leave some buffer
            d_eff = max(1.0, d - 2.5)
            req_decel = (v * v) / (2.0 * d_eff)
            # Map to brake command with traction-limited decel
            max_decel = 7.0 * _clamp(mu, 0.15, 1.0)
            brake_cmd = _clamp(req_decel / max_decel, 0.2, 1.0)
            throttle_cmd = 0.0
        else:
            # If within following distance, gently slow
            if d < safety_dist:
                brake_cmd = _clamp((safety_dist - d) / max(1.0, safety_dist), 0.0, 0.6)
                throttle_cmd = 0.0

    # If brake weak and urgent obstacle, attempt lane change bias and cut throttle
    if urgent and _mem["fault"]["brake_weak"]:
        throttle_cmd = 0.0
        brake_cmd = 1.0
        # Encourage steering toward target lane (or away from center if no lane change command)
        steer_cmd = _clamp(steer_cmd * 1.2, -steer_limit, steer_limit)

    # Normal speed PID when not braking for obstacle
    if brake_cmd < 0.05:
        v_err = target_speed_mps - speed_mps

        # Gains
        kp = 0.35
        ki = 0.12
        kd = 0.05
        kp *= _clamp(mu, 0.25, 1.0)
        ki *= _clamp(mu, 0.25, 1.0)
        kd *= _clamp(mu, 0.25, 1.0)

        # Estimate accel from history for derivative damping
        v_dot = accel_est
        _mem["speed_i"] = _clamp(_mem["speed_i"] + v_err * dt, -8.0, 8.0)

        u = kp * v_err + ki * _mem["speed_i"] - kd * v_dot

        if u >= 0.0:
            throttle_cmd = _clamp(u, 0.0, 1.0)
            brake_cmd = 0.0
        else:
            throttle_cmd = 0.0
            brake_cmd = _clamp(-u * 0.6, 0.0, 0.7)

    # Traction management: cap throttle and brake on low mu
    throttle_cap = _clamp(0.25 + 0.75 * mu, 0.15, 1.0)
    brake_cap = _clamp(0.35 + 0.65 * mu, 0.2, 1.0)
    throttle_cmd = min(throttle_cmd, throttle_cap)
    brake_cmd = min(brake_cmd, brake_cap)

    # If throttle stuck suspected, always bias to braking to control speed
    if _mem["fault"]["throttle_stuck"]:
        throttle_cmd = 0.0
        # use mild brake to counter unexpected accel when above target
        if speed_mps > target_speed_mps + 0.5:
            brake_cmd = max(brake_cmd, _clamp((speed_mps - target_speed_mps) * 0.12, 0.1, 0.7))

    # If total brake failure scenario: brake pressure sensors might show no response.
    # Detect mismatch between commanded brake and pressure (if available); if so, reduce speed target aggressively.
    bp = []
    for i in (1, 2):
        v = _get(state, f"brake_pressure_{i}_pct", None)
        if isinstance(v, (int, float)):
            bp.append(float(v))
    bp_med = _median(bp) if bp else None
    if bp_med is not None and brake_cmd > 0.6 and bp_med < 5.0:
        _mem["fault"]["brake_weak"] = True
        throttle_cmd = 0.0
        # Strongly reduce target via engine braking (throttle 0) and avoid obstacles via steering
        brake_cmd = 1.0

    # Final smoothing / rate limiting
    prev_th = float(last_cmd.get("throttle", 0.0))
    prev_br = float(last_cmd.get("brake", 0.0))
    max_th_rate = 2.5 * _clamp(mu, 0.25, 1.0)
    max_br_rate = 3.5 * _clamp(mu, 0.25, 1.0)

    throttle_cmd = _clamp(throttle_cmd, prev_th - max_th_rate * dt, prev_th + max_th_rate * dt)
    brake_cmd = _clamp(brake_cmd, prev_br - max_br_rate * dt, prev_br + max_br_rate * dt)

    # Prevent simultaneous high throttle and brake
    if throttle_cmd > 0.2 and brake_cmd > 0.2:
        if throttle_cmd > brake_cmd:
            brake_cmd *= 0.4
        else:
            throttle_cmd *= 0.4

    out = {
        "steering": float(_clamp(steer_cmd, -1.0, 1.0)),
        "throttle": float(_clamp(throttle_cmd, 0.0, 1.0)),
        "brake": float(_clamp(brake_cmd, 0.0, 1.0)),
    }

    _mem["last_cmd"] = out
    return out
