// Warehouse robot-job assignment benchmark
// Complete runnable TypeScript program (no external deps).

declare const require: any;
declare const process: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8").trim();
const data = input.length ? JSON.parse(input) : {};

type Grid = string[];
type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: Grid = data.grid || [];
const robots: Robot[] = data.robots || [];
const jobs: Job[] = data.jobs || [];

const H = grid.length | 0;
const W = H ? (grid[0]?.length | 0) : 0;
const N = (W * H) | 0;

function idx(x: number, y: number): number {
  return (y * W + x) | 0;
}

// Passability map
const passable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    const c = row.charCodeAt(x);
    // '.' open, '#' blocked; treat everything else as open by default.
    passable[idx(x, y)] = c === 35 /*#*/ ? 0 : 1;
  }
}

const INF = 0x3fffffff;

// BFS shortest path distances from a single start over 4-neighborhood.
// Returns dist array with -1 for unreachable.
function bfsFrom(start: number): Int32Array {
  const dist = new Int32Array(N);
  dist.fill(-1);
  if (start < 0 || start >= N || passable[start] === 0) return dist;

  // Queue arrays sized N for speed.
  const q = new Int32Array(N);
  let qh = 0;
  let qt = 0;
  dist[start] = 0;
  q[qt++] = start;

  while (qh < qt) {
    const cur = q[qh++];
    const d = dist[cur] + 1;

    const x = cur % W;
    const y = (cur / W) | 0;

    // neighbor: left
    if (x > 0) {
      const ni = cur - 1;
      if (passable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
    // right
    if (x + 1 < W) {
      const ni = cur + 1;
      if (passable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
    // up
    if (y > 0) {
      const ni = cur - W;
      if (passable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
    // down
    if (y + 1 < H) {
      const ni = cur + W;
      if (passable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
  }

  return dist;
}

function manhattan(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);
}

// Sort jobs deterministically: priority desc, then id asc.
const sortedJobs = jobs.slice().sort((a, b) => {
  const pd = (b.priority | 0) - (a.priority | 0);
  if (pd) return pd;
  return (a.id | 0) - (b.id | 0);
});

// Ensure robots are deterministic: id asc (tie by y,x).
const sortedRobots = robots.slice().sort((a, b) => {
  const idd = (a.id | 0) - (b.id | 0);
  if (idd) return idd;
  const yd = (a.y | 0) - (b.y | 0);
  if (yd) return yd;
  return (a.x | 0) - (b.x | 0);
});

// If no robots, still must output something; but every job must appear exactly once.
// In that edge case, put all jobs into a single bucket.
if (sortedRobots.length === 0) {
  const out = { assignments: [sortedJobs.map((j) => j.id)] };
  process.stdout.write(JSON.stringify(out));
  process.exit(0);
}

// Precompute BFS from each robot start once (scales with R * gridSize).
const robotStartIndex: number[] = new Array(sortedRobots.length);
const robotBfs: Int32Array[] = new Array(sortedRobots.length);
for (let r = 0; r < sortedRobots.length; r++) {
  const rb = sortedRobots[r];
  const s = idx(rb.x | 0, rb.y | 0);
  robotStartIndex[r] = s;
  robotBfs[r] = bfsFrom(s);
}

// Assign jobs to robots based on start-distance, with deterministic tie-breaking.
const assignedByRobot: Job[][] = Array.from({ length: sortedRobots.length }, () => []);
for (let ji = 0; ji < sortedJobs.length; ji++) {
  const job = sortedJobs[ji];
  const jIndex = idx(job.x | 0, job.y | 0);

  let bestR = 0;
  let bestD = INF;
  for (let r = 0; r < sortedRobots.length; r++) {
    const d = robotBfs[r][jIndex];
    const dd = d >= 0 ? d : INF;
    if (dd < bestD) {
      bestD = dd;
      bestR = r;
    } else if (dd === bestD) {
      // tie-break by robot id to be deterministic
      if ((sortedRobots[r].id | 0) < (sortedRobots[bestR].id | 0)) bestR = r;
    }
  }

  // If unreachable for all robots, fall back to Manhattan to choose nearest robot.
  if (bestD === INF) {
    bestR = 0;
    let bestM = INF;
    for (let r = 0; r < sortedRobots.length; r++) {
      const rb = sortedRobots[r];
      const m = manhattan(rb.x | 0, rb.y | 0, job.x | 0, job.y | 0);
      if (m < bestM) {
        bestM = m;
        bestR = r;
      } else if (m === bestM) {
        if ((sortedRobots[r].id | 0) < (sortedRobots[bestR].id | 0)) bestR = r;
      }
    }
  }

  assignedByRobot[bestR].push(job);
}

// Sequence jobs per robot to reduce travel:
// - For smaller lists, do BFS-based nearest-neighbor from current position.
// - For very large lists, use Manhattan nearest-neighbor for speed.
function orderJobsForRobot(robot: Robot, jobsForRobot: Job[]): number[] {
  if (jobsForRobot.length === 0) return [];

  const remaining = jobsForRobot.slice().sort((a, b) => {
    // deterministic initial order: priority desc then id asc
    const pd = (b.priority | 0) - (a.priority | 0);
    if (pd) return pd;
    return (a.id | 0) - (b.id | 0);
  });

  const out: number[] = [];
  let curX = robot.x | 0;
  let curY = robot.y | 0;
  let curIndex = idx(curX, curY);

  const BFS_LIMIT = 120; // cap BFS-based NN steps for scalability
  let steps = 0;

  while (remaining.length) {
    let bestK = 0;
    let bestScore = INF;

    if (steps < BFS_LIMIT) {
      const dist = bfsFrom(curIndex);
      for (let k = 0; k < remaining.length; k++) {
        const j = remaining[k];
        const di = dist[idx(j.x | 0, j.y | 0)];
        const dd = di >= 0 ? di : INF;
        if (dd < bestScore) {
          bestScore = dd;
          bestK = k;
        } else if (dd === bestScore) {
          // tie-break: higher priority first, then id
          const ap = remaining[bestK].priority | 0;
          const bp = j.priority | 0;
          if (bp > ap) bestK = k;
          else if (bp === ap && ((j.id | 0) < (remaining[bestK].id | 0))) bestK = k;
        }
      }
    } else {
      for (let k = 0; k < remaining.length; k++) {
        const j = remaining[k];
        const dd = manhattan(curX, curY, j.x | 0, j.y | 0);
        if (dd < bestScore) {
          bestScore = dd;
          bestK = k;
        } else if (dd === bestScore) {
          const ap = remaining[bestK].priority | 0;
          const bp = j.priority | 0;
          if (bp > ap) bestK = k;
          else if (bp === ap && ((j.id | 0) < (remaining[bestK].id | 0))) bestK = k;
        }
      }
    }

    const chosen = remaining.splice(bestK, 1)[0];
    out.push(chosen.id);
    curX = chosen.x | 0;
    curY = chosen.y | 0;
    curIndex = idx(curX, curY);
    steps++;
  }

  return out;
}

// Build output assignments array indexed by robot id.
let maxRobotId = 0;
for (const r of sortedRobots) if ((r.id | 0) > maxRobotId) maxRobotId = r.id | 0;
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

for (let r = 0; r < sortedRobots.length; r++) {
  const rb = sortedRobots[r];
  const route = orderJobsForRobot(rb, assignedByRobot[r]);
  assignments[rb.id | 0] = route;
}

process.stdout.write(JSON.stringify({ assignments }));
