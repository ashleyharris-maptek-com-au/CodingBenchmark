declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : { depot: [0, 0], truckCount: 0, packages: [] };

type Pkg = { id: string | number; x: number; y: number };

const depotX: number = Array.isArray(data.depot) ? Number(data.depot[0]) : 0;
const depotY: number = Array.isArray(data.depot) ? Number(data.depot[1]) : 0;
const packages: Pkg[] = Array.isArray(data.packages) ? data.packages : [];
const n: number = packages.length | 0;
const truckCountRaw: number = Number(data.truckCount);
const maxRoutes: number = Math.max(1, Math.min((truckCountRaw | 0) || 1, n || 1));

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const xs = new Float64Array(n);
const ys = new Float64Array(n);
const ids: Array<string | number> = new Array(n);
const angles = new Float64Array(n);

for (let i = 0; i < n; i++) {
  const p = packages[i];
  const x = Number(p.x);
  const y = Number(p.y);
  xs[i] = x;
  ys[i] = y;
  ids[i] = p.id;
  angles[i] = Math.atan2(y - depotY, x - depotX);
}

function distIdx(a: number, b: number): number {
  const ax = a === -1 ? depotX : xs[a];
  const ay = a === -1 ? depotY : ys[a];
  const bx = b === -1 ? depotX : xs[b];
  const by = b === -1 ? depotY : ys[b];
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

function routeDistance(route: number[]): number {
  const m = route.length;
  if (m === 0) return 0;
  let total = distIdx(-1, route[0]);
  for (let i = 1; i < m; i++) total += distIdx(route[i - 1], route[i]);
  total += distIdx(route[m - 1], -1);
  return total;
}

function reverseInPlace(arr: number[]): void {
  let i = 0;
  let j = arr.length - 1;
  while (i < j) {
    const t = arr[i];
    arr[i] = arr[j];
    arr[j] = t;
    i++;
    j--;
  }
}

function partitionSweep(sorted: number[], k: number): number[][] {
  const parts: number[][] = new Array(k);
  const base = (n / k) | 0;
  const rem = n - base * k;
  let offset = 0;
  for (let r = 0; r < k; r++) {
    const size = base + (r < rem ? 1 : 0);
    const seg = sorted.slice(offset, offset + size);
    offset += size;
    parts[r] = seg;
  }
  return parts;
}

function chooseBetterDirection(route: number[]): number[] {
  if (route.length <= 1) return route;
  const forward = route;
  const backward = route.slice();
  reverseInPlace(backward);
  const df = routeDistance(forward);
  const db = routeDistance(backward);
  return db < df ? backward : forward;
}

function clampInt(v: number, lo: number, hi: number): number {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v | 0;
}

function buildCandidates(maxK: number): number[] {
  const set = new Set<number>();
  set.add(1);
  set.add(maxK);
  let k = 1;
  while (k < maxK) {
    k *= 2;
    set.add(k > maxK ? maxK : k);
  }
  const root = Math.round(Math.sqrt(n));
  for (let d = -1; d <= 1; d++) set.add(clampInt(root + d, 1, maxK));
  if (maxK >= 3) set.add(clampInt(Math.round(maxK / 2), 1, maxK));
  const arr = Array.from(set);
  arr.sort((a, b) => a - b);
  return arr;
}

function twoOptFull(route: number[]): void {
  const m = route.length;
  if (m < 4) return;
  let improved = true;
  while (improved) {
    improved = false;
    for (let i = 0; i < m - 1; i++) {
      const a = i === 0 ? -1 : route[i - 1];
      const b = route[i];
      for (let j = i + 1; j < m; j++) {
        const c = route[j];
        const d = j === m - 1 ? -1 : route[j + 1];
        const delta = distIdx(a, c) + distIdx(b, d) - (distIdx(a, b) + distIdx(c, d));
        if (delta < -1e-9) {
          // reverse segment [i..j]
          let l = i;
          let r = j;
          while (l < r) {
            const t = route[l];
            route[l] = route[r];
            route[r] = t;
            l++;
            r--;
          }
          improved = true;
        }
      }
    }
  }
}

function twoOptRandom(route: number[], iters: number): void {
  const m = route.length;
  if (m < 4) return;
  for (let t = 0; t < iters; t++) {
    let i = (Math.random() * (m - 1)) | 0;
    let j = (Math.random() * (m - 1)) | 0;
    if (i === j) continue;
    if (i > j) {
      const tmp = i;
      i = j;
      j = tmp;
    }
    if (j <= i) continue;
    const a = i === 0 ? -1 : route[i - 1];
    const b = route[i];
    const c = route[j];
    const d = j === m - 1 ? -1 : route[j + 1];
    const delta = distIdx(a, c) + distIdx(b, d) - (distIdx(a, b) + distIdx(c, d));
    if (delta < -1e-9) {
      let l = i;
      let r = j;
      while (l < r) {
        const swap = route[l];
        route[l] = route[r];
        route[r] = swap;
        l++;
        r--;
      }
    }
  }
}

function optimizeRoute(route: number[]): number[] {
  const m = route.length;
  if (m <= 3) return route;
  // Quick win: try both orientations again after any optimization.
  if (m <= 200) twoOptFull(route);
  else twoOptRandom(route, Math.min(5000, (m * 30) | 0));
  return chooseBetterDirection(route);
}

// Sweep sort by angle (then radius to reduce crossings).
const idxs: number[] = new Array(n);
for (let i = 0; i < n; i++) idxs[i] = i;
idxs.sort((a, b) => {
  const da = angles[a] - angles[b];
  if (da !== 0) return da < 0 ? -1 : 1;
  const ra = (xs[a] - depotX) * (xs[a] - depotX) + (ys[a] - depotY) * (ys[a] - depotY);
  const rb = (xs[b] - depotX) * (xs[b] - depotX) + (ys[b] - depotY) * (ys[b] - depotY);
  return ra - rb;
});

const candidates = buildCandidates(maxRoutes);
let bestParts: number[][] = [];
let bestTotal = Number.POSITIVE_INFINITY;

for (const k of candidates) {
  const parts = partitionSweep(idxs, k);
  let total = 0;
  for (let r = 0; r < parts.length; r++) {
    if (parts[r].length === 0) continue;
    parts[r] = chooseBetterDirection(parts[r]);
    total += routeDistance(parts[r]);
    if (total >= bestTotal) break;
  }
  if (total < bestTotal) {
    bestTotal = total;
    bestParts = parts;
  }
}

const routesOut: Array<Array<string | number>> = [];
for (let r = 0; r < bestParts.length; r++) {
  const routeIdx = bestParts[r];
  if (routeIdx.length === 0) continue;
  const optimized = optimizeRoute(routeIdx.slice());
  const out: Array<string | number> = new Array(optimized.length);
  for (let i = 0; i < optimized.length; i++) out[i] = ids[optimized[i]];
  routesOut.push(out);
}

process.stdout.write(JSON.stringify({ routes: routesOut }));
