using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly System.IO.TextReader _reader;
        private readonly char[] _buf = new char[1 << 16];
        private int _len, _pos;

        public FastScanner(System.IO.TextReader reader) => _reader = reader;

        private int ReadChar()
        {
            if (_pos >= _len)
            {
                _len = _reader.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public string? Next()
        {
            int c;
            do
            {
                c = ReadChar();
                if (c < 0) return null;
            } while (c <= ' ' || c == 0xFEFF); // whitespace / BOM

            var sb = new System.Text.StringBuilder(8);
            while (c > ' ')
            {
                sb.Append((char)c);
                c = ReadChar();
                if (c < 0) break;
            }
            string t = sb.ToString();
            if (t.Length > 0 && t[0] == '\uFEFF') t = t.Substring(1);
            // Defensive: handle UTF-8 BOM bytes mis-decoded as Latin-1: ï»¿
            if (t.Length >= 3 && t[0] == '\u00EF' && t[1] == '\u00BB' && t[2] == '\u00BF') t = t.Substring(3);
            return t;
        }

        public int NextInt()
        {
            var s = Next();
            if (s == null) throw new InvalidOperationException("Unexpected EOF");
            return int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
        }
    }

    private readonly struct PieceRot
    {
        public readonly int W;
        public readonly int H;
        public readonly ulong[] Rows; // bottom->top, each already aligned to x=0

        public PieceRot(int w, int h, ulong[] rows)
        {
            W = w;
            H = h;
            Rows = rows;
        }
    }

    private sealed class Board
    {
        public readonly int Width;
        public readonly int Height;
        public readonly ulong FullMask;
        public readonly ulong[] Rows; // bottom->top
        public readonly int[] ColHeights;

        public int MaxHeight { get; private set; }

        public Board(int width, int height)
        {
            if (width <= 0 || width > 64) throw new ArgumentOutOfRangeException(nameof(width), "Supported width: 1..64");
            Width = width;
            Height = height;
            FullMask = width == 64 ? ulong.MaxValue : ((1UL << width) - 1UL);
            Rows = new ulong[height];
            ColHeights = new int[width];
            MaxHeight = 0;
        }

        public Board Clone()
        {
            var b = new Board(Width, Height);
            Array.Copy(Rows, b.Rows, Rows.Length);
            Array.Copy(ColHeights, b.ColHeights, ColHeights.Length);
            b.MaxHeight = MaxHeight;
            return b;
        }

        private void RecomputeHeights()
        {
            Array.Clear(ColHeights, 0, ColHeights.Length);
            int maxH = 0;
            for (int x = 0; x < Width; x++)
            {
                int h = 0;
                for (int y = Height - 1; y >= 0; y--)
                {
                    if (((Rows[y] >> x) & 1UL) != 0)
                    {
                        h = y + 1;
                        break;
                    }
                }
                ColHeights[x] = h;
                if (h > maxH) maxH = h;
            }
            MaxHeight = maxH;
        }

        public bool CanPlace(in PieceRot p, int x, int y)
        {
            if (x < 0 || x + p.W > Width) return false;
            if (y < 0 || y + p.H > Height) return false;

            for (int i = 0; i < p.H; i++)
            {
                ulong m = p.Rows[i] << x;
                if ((Rows[y + i] & m) != 0) return false;
            }
            return true;
        }

        public int FindDropY(in PieceRot p, int x)
        {
            int y = Height - p.H;
            while (y > 0 && CanPlace(p, x, y - 1)) y--;
            return y;
        }

        public bool PlaceAndClear(in PieceRot p, int x, out int clearedLines)
        {
            clearedLines = 0;
            int y = FindDropY(p, x);
            if (!CanPlace(p, x, y)) return false;

            for (int i = 0; i < p.H; i++)
            {
                Rows[y + i] |= (p.Rows[i] << x);
            }

            // Clear full lines.
            int write = 0;
            for (int read = 0; read < Height; read++)
            {
                if ((Rows[read] & FullMask) == FullMask)
                {
                    clearedLines++;
                    continue;
                }
                Rows[write++] = Rows[read];
            }
            while (write < Height) Rows[write++] = 0UL;

            RecomputeHeights();
            return true;
        }

        public void GetColumnHeights(int[] dst)
        {
            Array.Copy(ColHeights, dst, ColHeights.Length);
        }

        public int CountHoles()
        {
            int holes = 0;
            for (int x = 0; x < Width; x++)
            {
                bool seenBlock = false;
                for (int y = Height - 1; y >= 0; y--)
                {
                    bool filled = ((Rows[y] >> x) & 1UL) != 0;
                    if (filled) seenBlock = true;
                    else if (seenBlock) holes++;
                }
            }
            return holes;
        }

        public int AggregateHeight()
        {
            int sum = 0;
            for (int x = 0; x < Width; x++) sum += ColHeights[x];
            return sum;
        }

        public int Bumpiness()
        {
            int b = 0;
            for (int x = 0; x < Width - 1; x++) b += Math.Abs(ColHeights[x] - ColHeights[x + 1]);
            return b;
        }

        public int WellSums()
        {
            // Sum of well depths (columns lower than both neighbors; edges use one neighbor).
            int sum = 0;
            for (int x = 0; x < Width; x++)
            {
                int left = x == 0 ? int.MaxValue : ColHeights[x - 1];
                int right = x == Width - 1 ? int.MaxValue : ColHeights[x + 1];
                int minNeighbor = Math.Min(left, right);
                if (minNeighbor > ColHeights[x])
                {
                    sum += (minNeighbor - ColHeights[x]);
                }
            }
            return sum;
        }

        public int RowTransitions()
        {
            // Count transitions filled<->empty per row including borders.
            int trans = 0;
            for (int y = 0; y < Height; y++)
            {
                ulong row = Rows[y] & FullMask;
                int last = 1; // left border treated as filled
                for (int x = 0; x < Width; x++)
                {
                    int cur = (int)((row >> x) & 1UL);
                    if (cur != last) trans++;
                    last = cur;
                }
                if (last == 0) trans++; // right border filled
            }
            return trans;
        }

        public int ColTransitions()
        {
            int trans = 0;
            for (int x = 0; x < Width; x++)
            {
                int last = 1; // bottom border filled
                for (int y = 0; y < Height; y++)
                {
                    int cur = (int)((Rows[y] >> x) & 1UL);
                    if (cur != last) trans++;
                    last = cur;
                }
                if (last == 0) trans++; // top border filled
            }
            return trans;
        }
    }

    private static readonly Dictionary<char, PieceRot[]> Pieces = BuildPieces();

    private static Dictionary<char, PieceRot[]> BuildPieces()
    {
        // Each rotation stored as bottom->top row bitmasks in minimal bounding box.
        static PieceRot RotFrom(params string[] rowsTopToBottom)
        {
            int h = rowsTopToBottom.Length;
            int w = 0;
            for (int i = 0; i < h; i++) w = Math.Max(w, rowsTopToBottom[i].Length);
            // Normalize: treat 'X' as filled.
            var rows = new ulong[h];
            for (int ty = 0; ty < h; ty++)
            {
                string r = rowsTopToBottom[ty];
                ulong m = 0;
                for (int x = 0; x < r.Length; x++)
                {
                    if (r[x] == 'X') m |= 1UL << x;
                }
                rows[h - 1 - ty] = m;
            }
            // Trim empty top rows and empty bottom rows and empty left/right columns.
            int bottom = 0;
            int top = h - 1;
            while (bottom <= top && rows[bottom] == 0) bottom++;
            while (top >= bottom && rows[top] == 0) top--;
            if (bottom > top) return new PieceRot(0, 0, Array.Empty<ulong>());

            int newH = top - bottom + 1;
            var cut = new ulong[newH];
            Array.Copy(rows, bottom, cut, 0, newH);

            ulong all = 0;
            for (int i = 0; i < newH; i++) all |= cut[i];
            int leftShift = 0;
            while (((all >> leftShift) & 1UL) == 0) leftShift++;
            int rightMost = 0;
            for (int bit = 0; bit < 64; bit++) if (((all >> bit) & 1UL) != 0) rightMost = bit;
            int newW = rightMost - leftShift + 1;

            for (int i = 0; i < newH; i++) cut[i] = (cut[i] >> leftShift);
            return new PieceRot(newW, newH, cut);
        }

        var d = new Dictionary<char, PieceRot[]>();

        d['I'] = new[]
        {
            RotFrom("XXXX"),
            RotFrom("X","X","X","X"),
            RotFrom("XXXX"),
            RotFrom("X","X","X","X"),
        };
        d['O'] = new[]
        {
            RotFrom("XX","XX"),
            RotFrom("XX","XX"),
            RotFrom("XX","XX"),
            RotFrom("XX","XX"),
        };
        d['T'] = new[]
        {
            RotFrom("XXX",".X."),
            RotFrom(".X","XX",".X"),
            RotFrom(".X.","XXX"),
            RotFrom("X.","XX","X."),
        };
        d['S'] = new[]
        {
            RotFrom(".XX","XX."),
            RotFrom("X.","XX",".X"),
            RotFrom(".XX","XX."),
            RotFrom("X.","XX",".X"),
        };
        d['Z'] = new[]
        {
            RotFrom("XX.",".XX"),
            RotFrom(".X","XX","X."),
            RotFrom("XX.",".XX"),
            RotFrom(".X","XX","X."),
        };
        d['J'] = new[]
        {
            RotFrom("X..","XXX"),
            RotFrom("XX","X.","X."),
            RotFrom("XXX","..X"),
            RotFrom(".X",".X","XX"),
        };
        d['L'] = new[]
        {
            RotFrom("..X","XXX"),
            RotFrom("X.","X.","XX"),
            RotFrom("XXX","X.."),
            RotFrom("XX",".X",".X"),
        };

        return d;
    }

    private sealed class Evaluator
    {
        // Base weights: classic well-performing hand-tuned heuristic.
        // These are adapted based on danger level (stack height).
        public double Score(Board b, int clearedLines)
        {
            double danger = b.MaxHeight / (double)Math.Max(1, b.Height); // 0..1+
            if (danger < 0) danger = 0;
            if (danger > 1.5) danger = 1.5;

            // Blend two regimes: early (flat/efficient) vs late (survival).
            double t = Math.Clamp((danger - 0.35) / 0.45, 0.0, 1.0); // start shifting ~35%, finish ~80%

            // Early-game weights
            double wLinesE = 1.00;
            double wAggE = -0.50;
            double wMaxE = -0.90;
            double wHolesE = -4.50;
            double wBumpE = -0.35;
            double wWellE = -0.18;
            double wTransE = -0.02;

            // Late-game weights
            double wLinesL = 1.10;
            double wAggL = -0.40;
            double wMaxL = -3.20;
            double wHolesL = -6.50;
            double wBumpL = -0.28;
            double wWellL = -0.25;
            double wTransL = -0.03;

            double wLines = Lerp(wLinesE, wLinesL, t);
            double wAgg = Lerp(wAggE, wAggL, t);
            double wMax = Lerp(wMaxE, wMaxL, t);
            double wHoles = Lerp(wHolesE, wHolesL, t);
            double wBump = Lerp(wBumpE, wBumpL, t);
            double wWell = Lerp(wWellE, wWellL, t);
            double wTrans = Lerp(wTransE, wTransL, t);

            // Reward multi-line clears superlinearly to prefer tetrises.
            double lineReward = clearedLines switch
            {
                0 => 0.0,
                1 => 1.0,
                2 => 2.4,
                3 => 4.2,
                _ => 6.8
            };

            int holes = b.CountHoles();
            int agg = b.AggregateHeight();
            int bump = b.Bumpiness();
            int well = b.WellSums();
            int rowT = b.RowTransitions();
            int colT = b.ColTransitions();
            int trans = rowT + colT;

            // Small extra penalty for being near the top, to avoid sudden death.
            double topPenalty = 0.0;
            if (b.MaxHeight >= b.Height - 2) topPenalty = 1000.0; // essentially dead
            else if (b.MaxHeight >= b.Height - 4) topPenalty = 50.0;

            return
                wLines * lineReward +
                wAgg * agg +
                wMax * b.MaxHeight +
                wHoles * holes +
                wBump * bump +
                wWell * well +
                wTrans * trans -
                topPenalty;
        }

        private static double Lerp(double a, double b, double t) => a + (b - a) * t;
    }

    private sealed class Player
    {
        private readonly Board _board;
        private readonly Evaluator _eval = new Evaluator();

        public Player(int width, int height) => _board = new Board(width, height);

        public (int rot, int col) ChooseMove(char pieceType)
        {
            var rots = Pieces[pieceType];
            double bestScore = double.NegativeInfinity;
            int bestRot = 0, bestCol = 0;

            // Complexity-adaptive search: widen exploration only when safe.
            // When the stack is low, consider a "riskier" set of placements that create wells.
            int maxH = _board.MaxHeight;
            double danger = maxH / (double)Math.Max(1, _board.Height);
            bool allowWellBuilding = danger < 0.35;

            // Avoid duplicated rotations for symmetric pieces by using shape signatures.
            var seen = new HashSet<ulong>();
            for (int r = 0; r < 4; r++)
            {
                var p = rots[r];
                if (p.W == 0) continue;
                ulong sig = Signature(p);
                if (!seen.Add(sig)) continue;

                int maxX = _board.Width - p.W;
                for (int x = 0; x <= maxX; x++)
                {
                    var b2 = _board.Clone();
                    if (!b2.PlaceAndClear(p, x, out int cleared)) continue;

                    double s = _eval.Score(b2, cleared);
                    if (!allowWellBuilding)
                    {
                        // Late game: discourage creating deep wells unless it also clears lines.
                        // (This is a heuristic; the evaluator already penalizes wells, but we sharpen it here.)
                        if (cleared == 0) s -= 0.25 * b2.WellSums();
                    }

                    // Tie-breakers: prefer leftmost for determinism and slightly prefer line clears.
                    if (s > bestScore || (Math.Abs(s - bestScore) < 1e-9 && (cleared > 0 || x < bestCol)))
                    {
                        bestScore = s;
                        bestRot = r;
                        bestCol = x;
                    }
                }
            }

            return (bestRot, bestCol);
        }

        public void ApplyMove(char pieceType, int rot, int col)
        {
            var p = Pieces[pieceType][rot & 3];
            _board.PlaceAndClear(p, col, out _);
        }

        private static ulong Signature(in PieceRot p)
        {
            // Combine dims and first few rows into a signature.
            ulong sig = (ulong)p.W | ((ulong)p.H << 8);
            int take = Math.Min(p.H, 6);
            for (int i = 0; i < take; i++)
            {
                sig ^= (p.Rows[i] + 0x9E3779B97F4A7C15UL) * (ulong)(i + 1);
                sig = (sig << 7) | (sig >> (64 - 7));
            }
            return sig;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.In);
        string? wTok = fs.Next();
        string? hTok = fs.Next();
        string? nTok = fs.Next();
        if (wTok == null || hTok == null || nTok == null) return;
        if (!int.TryParse(wTok, NumberStyles.Integer, CultureInfo.InvariantCulture, out int width)) return;
        if (!int.TryParse(hTok, NumberStyles.Integer, CultureInfo.InvariantCulture, out int height)) return;
        if (!int.TryParse(nTok, NumberStyles.Integer, CultureInfo.InvariantCulture, out int numPieces)) return;

        var player = new Player(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string? tok = fs.Next();
            if (tok == null) break;
            char pieceType = tok[0];

            int rot = 0, col = 0;
            if (Pieces.ContainsKey(pieceType))
            {
                try
                {
                    (rot, col) = player.ChooseMove(pieceType);
                }
                catch
                {
                    rot = 0;
                    col = 0;
                }
            }

            Console.WriteLine(rot.ToString(CultureInfo.InvariantCulture) + " " + col.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            if (Pieces.ContainsKey(pieceType))
            {
                try { player.ApplyMove(pieceType, rot, col); } catch { }
            }

            string? result = fs.Next();
            if (result == null) break;
            if (result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;
        }
    }
}
