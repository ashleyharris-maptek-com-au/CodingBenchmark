using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private struct Box
    {
        public int W, H, D;
        public int Index;
        public long Volume => (long)W * H * D;
    }

    private struct Placed
    {
        public int X, Y, Z;
        public int W, H, D;
        public int BoxIndex;
    }

    private struct Point3
    {
        public int X, Y, Z;
        public Point3(int x, int y, int z) { X = x; Y = y; Z = z; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _pos;
        public FastScanner(Stream stream) { _stream = stream; }
        private byte Read()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_pos++];
        }
        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= 32 && c != 0);
            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }

    private sealed class SpatialGrid
    {
        private readonly int _cell;
        private readonly Dictionary<long, List<int>> _cells;
        private readonly List<int> _large;
        private int[] _mark;
        private int _stamp;

        public SpatialGrid(int cellSize, int initialPlacedCapacity)
        {
            _cell = Math.Max(1, cellSize);
            _cells = new Dictionary<long, List<int>>(capacity: Math.Max(1024, initialPlacedCapacity * 2));
            _large = new List<int>();
            _mark = new int[Math.Max(16, initialPlacedCapacity)];
            _stamp = 1;
        }

        public void EnsureMarkCapacity(int placedCount)
        {
            if (_mark.Length < placedCount)
            {
                int newSize = _mark.Length;
                while (newSize < placedCount) newSize *= 2;
                Array.Resize(ref _mark, newSize);
            }
        }

        private long Key(int cx, int cy, int cz)
        {
            // Pack 3 signed-ish ints into a long (assumes cx,cy,cz are non-negative and < 2^21 in typical cases).
            return ((long)cx << 42) ^ ((long)cy << 21) ^ (uint)cz;
        }

        public void Insert(int placedIndex, in Placed b)
        {
            int cx0 = b.X / _cell;
            int cy0 = b.Y / _cell;
            int cz0 = b.Z / _cell;
            int cx1 = (b.X + b.W - 1) / _cell;
            int cy1 = (b.Y + b.H - 1) / _cell;
            int cz1 = (b.Z + b.D - 1) / _cell;

            long span = (long)(cx1 - cx0 + 1) * (cy1 - cy0 + 1) * (cz1 - cz0 + 1);
            if (span > 4096)
            {
                _large.Add(placedIndex);
                return;
            }

            for (int cx = cx0; cx <= cx1; cx++)
            for (int cy = cy0; cy <= cy1; cy++)
            for (int cz = cz0; cz <= cz1; cz++)
            {
                long k = Key(cx, cy, cz);
                if (!_cells.TryGetValue(k, out var list))
                {
                    list = new List<int>(4);
                    _cells[k] = list;
                }
                list.Add(placedIndex);
            }
        }

        public bool AnyOverlap(int x, int y, int z, int w, int h, int d, List<Placed> placed)
        {
            EnsureMarkCapacity(placed.Count);
            int stamp = ++_stamp;
            if (stamp == int.MaxValue)
            {
                Array.Clear(_mark, 0, _mark.Length);
                _stamp = 1;
                stamp = 1;
            }

            int cx0 = x / _cell;
            int cy0 = y / _cell;
            int cz0 = z / _cell;
            int cx1 = (x + w - 1) / _cell;
            int cy1 = (y + h - 1) / _cell;
            int cz1 = (z + d - 1) / _cell;

            long span = (long)(cx1 - cx0 + 1) * (cy1 - cy0 + 1) * (cz1 - cz0 + 1);
            if (span > 4096)
            {
                for (int i = 0; i < placed.Count; i++)
                    if (Overlap(x, y, z, w, h, d, placed[i])) return true;
                return false;
            }

            for (int cx = cx0; cx <= cx1; cx++)
            for (int cy = cy0; cy <= cy1; cy++)
            for (int cz = cz0; cz <= cz1; cz++)
            {
                long k = Key(cx, cy, cz);
                if (!_cells.TryGetValue(k, out var list)) continue;
                for (int i = 0; i < list.Count; i++)
                {
                    int idx = list[i];
                    if (_mark[idx] == stamp) continue;
                    _mark[idx] = stamp;
                    if (Overlap(x, y, z, w, h, d, placed[idx])) return true;
                }
            }

            for (int i = 0; i < _large.Count; i++)
            {
                int idx = _large[i];
                if (_mark[idx] == stamp) continue;
                _mark[idx] = stamp;
                if (Overlap(x, y, z, w, h, d, placed[idx])) return true;
            }

            return false;
        }

        private static bool Overlap(int ax, int ay, int az, int aw, int ah, int ad, in Placed b)
        {
            // Strict overlap test (touching faces/edges is allowed).
            return ax < b.X + b.W && ax + aw > b.X &&
                   ay < b.Y + b.H && ay + ah > b.Y &&
                   az < b.Z + b.D && az + ad > b.Z;
        }
    }

    private static int[][] Orientations(int w, int h, int d)
    {
        // Up to 6 unique permutations, de-duplicated for equal edges.
        int[][] raw =
        {
            new[] { w, h, d },
            new[] { w, d, h },
            new[] { h, w, d },
            new[] { h, d, w },
            new[] { d, w, h },
            new[] { d, h, w },
        };
        var list = new List<int[]>(6);
        for (int i = 0; i < raw.Length; i++)
        {
            bool dup = false;
            for (int j = 0; j < list.Count; j++)
            {
                var o = list[j];
                if (o[0] == raw[i][0] && o[1] == raw[i][1] && o[2] == raw[i][2]) { dup = true; break; }
            }
            if (!dup) list.Add(raw[i]);
        }
        return list.ToArray();
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();

        var boxes = new Box[n];
        long sumEdges = 0;
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            boxes[i] = new Box { W = w, H = h, D = d, Index = i };
            sumEdges += (long)w + h + d;
        }

        Array.Sort(boxes, (a, b) =>
        {
            int c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            int am = Math.Max(a.W, Math.Max(a.H, a.D));
            int bm = Math.Max(b.W, Math.Max(b.H, b.D));
            c = bm.CompareTo(am);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        });

        int avgEdge = (int)Math.Max(1, sumEdges / (3L * Math.Max(1, n)));
        int cellSize = avgEdge;
        if (n >= 5000) cellSize = Math.Max(1, (int)(avgEdge * 0.75));
        if (n <= 200) cellSize = Math.Max(1, (int)(avgEdge * 0.5));

        var placed = new List<Placed>(Math.Min(n, 16384));
        var grid = new SpatialGrid(cellSize, Math.Min(n, 16384));

        int pointCap = n <= 200 ? 20000 : (n <= 2000 ? 8000 : 4500);
        var points = new List<Point3>(Math.Min(pointCap, 8192)) { new Point3(0, 0, 0) };

        // Reusable buffers for pruning
        var pointKeySet = new HashSet<long>();
        var pointTmp = new List<Point3>();

        for (int bi = 0; bi < n; bi++)
        {
            Box box = boxes[bi];
            var ors = Orientations(box.W, box.H, box.D);

            int bestPi = -1, bestOw = 0, bestOh = 0, bestOd = 0;
            long bestScore = long.MaxValue;

            // Prefer lower Z, then Y, then X. Add small tie-break based on how well it "fills" the corner.
            for (int pi = 0; pi < points.Count; pi++)
            {
                var p = points[pi];
                if (p.X > W || p.Y > H || p.Z > D) continue;

                for (int oi = 0; oi < ors.Length; oi++)
                {
                    int ow = ors[oi][0], oh = ors[oi][1], od = ors[oi][2];
                    if (p.X + ow > W || p.Y + oh > H || p.Z + od > D) continue;
                    if (grid.AnyOverlap(p.X, p.Y, p.Z, ow, oh, od, placed)) continue;

                    int rx = W - (p.X + ow);
                    int ry = H - (p.Y + oh);
                    int rz = D - (p.Z + od);
                    long fillPenalty = (long)rx + ry + rz; // smaller is better
                    long score = ((long)p.Z << 42) + ((long)p.Y << 21) + (uint)p.X;
                    score = score * 1024 + fillPenalty;

                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestPi = pi;
                        bestOw = ow; bestOh = oh; bestOd = od;
                    }
                }
            }

            if (bestPi < 0) continue;

            var bp = points[bestPi];
            var pl = new Placed
            {
                X = bp.X, Y = bp.Y, Z = bp.Z,
                W = bestOw, H = bestOh, D = bestOd,
                BoxIndex = box.Index
            };
            int placedIndex = placed.Count;
            placed.Add(pl);
            grid.Insert(placedIndex, pl);

            // Add new extreme points from this placement.
            points.Add(new Point3(bp.X + bestOw, bp.Y, bp.Z));
            points.Add(new Point3(bp.X, bp.Y + bestOh, bp.Z));
            points.Add(new Point3(bp.X, bp.Y, bp.Z + bestOd));

            // Remove the used point (swap-remove).
            int last = points.Count - 4; // before the 3 adds, last point index was Count-4
            points[bestPi] = points[last];
            points.RemoveAt(last);

            // Prune points to keep runtime bounded.
            if (points.Count > pointCap)
            {
                // 1) Dedup + bounds filter.
                pointKeySet.Clear();
                pointTmp.Clear();
                for (int i = 0; i < points.Count; i++)
                {
                    var p = points[i];
                    if (p.X < 0 || p.Y < 0 || p.Z < 0 || p.X >= W || p.Y >= H || p.Z >= D) continue;
                    long key = ((long)p.X << 42) ^ ((long)p.Y << 21) ^ (uint)p.Z;
                    if (pointKeySet.Add(key)) pointTmp.Add(p);
                }

                pointTmp.Sort((a, b) =>
                {
                    int c = a.Z.CompareTo(b.Z);
                    if (c != 0) return c;
                    c = a.Y.CompareTo(b.Y);
                    if (c != 0) return c;
                    return a.X.CompareTo(b.X);
                });

                // 2) Keep the best subset (low corners).
                int keep = pointCap;
                if (pointTmp.Count > keep) pointTmp.RemoveRange(keep, pointTmp.Count - keep);

                points.Clear();
                points.AddRange(pointTmp);
            }
        }

        var sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 1 << 16);
        sw.WriteLine(placed.Count);
        for (int i = 0; i < placed.Count; i++)
        {
            var p = placed[i];
            sw.Write(p.BoxIndex);
            sw.Write(' ');
            sw.Write(p.X);
            sw.Write(' ');
            sw.Write(p.Y);
            sw.Write(' ');
            sw.WriteLine(p.Z);
        }
        sw.Flush();
    }
}
