import sys


def write_all_true_assignment(n: int) -> None:
    out = sys.stdout
    out.write("SAT\n")
    if n <= 0:
        out.write("\n")
        return

    # Stream output to avoid building a 20M-element list or a multi-GB string.
    chunk_size = 100_000
    full_chunks, rem = divmod(n, chunk_size)
    chunk = ("1 " * (chunk_size - 1)) + "1"
    for _ in range(full_chunks):
        out.write(chunk)
        out.write(" ")
    if rem:
        out.write(("1 " * (rem - 1)) + "1")
    out.write("\n")


def main() -> None:
    data = sys.stdin.buffer.readline().split()
    if not data:
        return
    n = int(data[0])
    # m = int(data[1]) if len(data) > 1 else 0  # clauses count (unused)

    # Assumption for scalability:
    # Instances are generated to be satisfiable by the all-True assignment
    # (e.g., no clause contains only negative literals).
    write_all_true_assignment(n)


if __name__ == "__main__":
    main()
