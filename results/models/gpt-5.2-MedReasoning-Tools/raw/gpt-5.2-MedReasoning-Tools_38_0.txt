import sys


def _parse_int_token(tok: str) -> int:
    # Be permissive about BOM / stray Unicode at start of stream.
    tok = tok.strip().lstrip("\ufeff")
    filtered = []
    for ch in tok:
        if ch.isdigit() or ch == "-":
            filtered.append(ch)
    return int("".join(filtered))


def solve(u: int, sets_list: list[tuple[int, ...]]) -> list[int] | None:
    if u == 0:
        return []

    # Normalize rows and build column -> rows incidence.
    rows = [tuple(sorted(set(r))) for r in sets_list]
    col_rows: list[set[int]] = [set() for _ in range(u)]
    for r_idx, elems in enumerate(rows):
        for c in elems:
            if 0 <= c < u:
                col_rows[c].add(r_idx)

    # Quick infeasibility: any column with no rows.
    for c in range(u):
        if not col_rows[c]:
            return None

    active_cols: set[int] = set(range(u))
    partial: list[int] = []

    def search() -> list[int] | None:
        if not active_cols:
            return list(partial)

        # Choose column with fewest options (Algorithm X heuristic).
        col = min(active_cols, key=lambda c: (len(col_rows[c]), c))
        if not col_rows[col]:
            return None

        # Deterministic row iteration: increasing input index.
        for r in sorted(col_rows[col]):
            removed_cols: list[int] = []
            removed_pairs: list[tuple[int, int]] = []  # (column, row) removals

            # Cover all columns in this row.
            ok = True
            for c in rows[r]:
                if c not in active_cols:
                    continue
                removed_cols.append(c)
                active_cols.remove(c)
                # For each row that has c, remove it from all other active columns it touches.
                for r2 in list(col_rows[c]):
                    for c2 in rows[r2]:
                        if c2 in active_cols and r2 in col_rows[c2]:
                            col_rows[c2].remove(r2)
                            removed_pairs.append((c2, r2))

            if ok:
                partial.append(r)
                res = search()
                if res is not None:
                    return res
                partial.pop()

            # Uncover in reverse order.
            for c2, r2 in reversed(removed_pairs):
                col_rows[c2].add(r2)
            for c in reversed(removed_cols):
                active_cols.add(c)

        return None

    return search()


def main() -> None:
    data = sys.stdin.read().splitlines()
    if not data:
        return

    # Parse header.
    header_i = 0
    while header_i < len(data) and not data[header_i].strip():
        header_i += 1
    if header_i >= len(data):
        return
    u_s = data[header_i].split()
    u = _parse_int_token(u_s[0])
    s = _parse_int_token(u_s[1])

    sets_list: list[tuple[int, ...]] = []
    for line in data[header_i + 1 : header_i + 1 + s]:
        line = line.strip()
        if not line:
            sets_list.append(tuple())
            continue
        sets_list.append(tuple(map(int, line.split())))

    sol = solve(u, sets_list)
    if sol is None:
        sys.stdout.write("NO SOLUTION\n")
        return

    sol_sorted = sorted(sol)
    sys.stdout.write("SOLUTION\n")
    sys.stdout.write(" ".join(map(str, sol_sorted)) + "\n")


if __name__ == "__main__":
    main()
