// Wildfire firebreak placement heuristic (Node.js TypeScript)
// Reads one JSON document from stdin and outputs {"firebreaks":[[x,y],...]} to stdout.

declare const require: any;
declare const process: any;
const fs = require("fs");

type Town = { id: string | number; x: number; y: number; value: number };

function clampAdd(a: number, b: number, cap: number): number {
  const s = a + b;
  return s >= cap ? cap : s;
}

class MinHeap {
  private scores: number[] = [];
  private idxs: number[] = [];

  get size(): number {
    return this.scores.length;
  }

  peekScore(): number {
    return this.scores.length === 0 ? -Infinity : this.scores[0];
  }

  push(score: number, idx: number): void {
    const scores = this.scores;
    const idxs = this.idxs;
    let i = scores.length;
    scores.push(score);
    idxs.push(idx);
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (scores[p] <= score) break;
      scores[i] = scores[p];
      idxs[i] = idxs[p];
      i = p;
    }
    scores[i] = score;
    idxs[i] = idx;
  }

  pop(): [number, number] | null {
    const scores = this.scores;
    const idxs = this.idxs;
    const n = scores.length;
    if (n === 0) return null;
    const minScore = scores[0];
    const minIdx = idxs[0];
    const lastScore = scores.pop() as number;
    const lastIdx = idxs.pop() as number;
    if (n > 1) {
      let i = 0;
      const half = (n - 1) >> 1;
      while (i < half) {
        let l = (i << 1) + 1;
        let r = l + 1;
        let c = r < n - 1 && scores[r] < scores[l] ? r : l;
        if (scores[c] >= lastScore) break;
        scores[i] = scores[c];
        idxs[i] = idxs[c];
        i = c;
      }
      scores[i] = lastScore;
      idxs[i] = lastIdx;
    }
    return [minScore, minIdx];
  }

  toArray(): Array<[number, number]> {
    const out: Array<[number, number]> = [];
    for (let i = 0; i < this.scores.length; i++) out.push([this.scores[i], this.idxs[i]]);
    return out;
  }
}

function main(): void {
  const input: string = fs.readFileSync(0, "utf8");
  const data = JSON.parse(input);

  const width: number = data.width | 0;
  const height: number = data.height | 0;
  const grid: string[] = data.grid as string[];
  const ignitions: Array<[number, number]> = data.ignitions as Array<[number, number]>;
  const towns: Town[] = data.towns as Town[];
  const cutBudget: number = Math.max(0, data.cutBudget | 0);

  const n = width * height;
  if (width <= 0 || height <= 0 || !Array.isArray(grid) || grid.length !== height || cutBudget <= 0) {
    process.stdout.write(JSON.stringify({ firebreaks: [] }));
    return;
  }

  const burnable = new Uint8Array(n);
  for (let y = 0; y < height; y++) {
    const row = grid[y] ?? "";
    for (let x = 0; x < width; x++) {
      const c = row.charCodeAt(x);
      // '.' = 46
      burnable[y * width + x] = c === 46 ? 1 : 0;
    }
  }

  const isTown = new Uint8Array(n);
  const isSource = new Uint8Array(n);
  const townVal = new Float64Array(n);

  if (Array.isArray(towns)) {
    for (let i = 0; i < towns.length; i++) {
      const t = towns[i];
      const x = t?.x | 0;
      const y = t?.y | 0;
      if (x < 0 || x >= width || y < 0 || y >= height) continue;
      const idx = y * width + x;
      // If the town isn't on burnable terrain, it's already safe.
      if (burnable[idx] === 0) continue;
      isTown[idx] = 1;
      const v = typeof t?.value === "number" ? t.value : Number(t?.value ?? 0);
      if (Number.isFinite(v) && v > 0) townVal[idx] += v;
    }
  }

  const dist = new Int32Array(n);
  dist.fill(-1);
  const queue = new Int32Array(n);
  let qh = 0;
  let qt = 0;

  if (Array.isArray(ignitions)) {
    for (let i = 0; i < ignitions.length; i++) {
      const p = ignitions[i];
      const x = p?.[0] | 0;
      const y = p?.[1] | 0;
      if (x < 0 || x >= width || y < 0 || y >= height) continue;
      const idx = y * width + x;
      if (burnable[idx] === 0) continue;
      if (dist[idx] !== -1) continue;
      dist[idx] = 0;
      isSource[idx] = 1;
      queue[qt++] = idx;
    }
  }

  if (qt === 0) {
    process.stdout.write(JSON.stringify({ firebreaks: [] }));
    return;
  }

  const order = new Int32Array(n);
  let orderLen = 0;

  while (qh < qt) {
    const v = queue[qh++];
    order[orderLen++] = v;
    const dv = dist[v];
    const x = v % width;
    const y = (v / width) | 0;

    // 4-neighbors
    let ni: number;
    if (x > 0) {
      ni = v - 1;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = dv + 1;
        queue[qt++] = ni;
      }
    }
    if (x + 1 < width) {
      ni = v + 1;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = dv + 1;
        queue[qt++] = ni;
      }
    }
    if (y > 0) {
      ni = v - width;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = dv + 1;
        queue[qt++] = ni;
      }
    }
    if (y + 1 < height) {
      ni = v + width;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = dv + 1;
        queue[qt++] = ni;
      }
    }
  }

  const ways = new Float64Array(n);
  // Initialize source ways
  for (let i = 0; i < ignitions.length; i++) {
    const p = ignitions[i];
    const x = p?.[0] | 0;
    const y = p?.[1] | 0;
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const idx = y * width + x;
    if (dist[idx] === 0) ways[idx] = 1;
  }

  const WAYS_CAP = 1e9;
  for (let oi = 0; oi < orderLen; oi++) {
    const v = order[oi];
    const dv = dist[v];
    const wv = ways[v];
    if (wv <= 0) continue;
    const x = v % width;
    const y = (v / width) | 0;

    let ni: number;
    if (x > 0) {
      ni = v - 1;
      if (dist[ni] === dv + 1) ways[ni] = clampAdd(ways[ni], wv, WAYS_CAP);
    }
    if (x + 1 < width) {
      ni = v + 1;
      if (dist[ni] === dv + 1) ways[ni] = clampAdd(ways[ni], wv, WAYS_CAP);
    }
    if (y > 0) {
      ni = v - width;
      if (dist[ni] === dv + 1) ways[ni] = clampAdd(ways[ni], wv, WAYS_CAP);
    }
    if (y + 1 < height) {
      ni = v + width;
      if (dist[ni] === dv + 1) ways[ni] = clampAdd(ways[ni], wv, WAYS_CAP);
    }
  }

  const g = new Float64Array(n);
  for (let oi = 0; oi < orderLen; oi++) {
    const v = order[oi];
    const tv = townVal[v];
    if (tv > 0) g[v] = tv;
  }

  const G_CAP = 1e12;
  for (let oi = orderLen - 1; oi >= 0; oi--) {
    const v = order[oi];
    const dv = dist[v];
    let gv = g[v];
    const x = v % width;
    const y = (v / width) | 0;

    let ni: number;
    if (x > 0) {
      ni = v - 1;
      if (dist[ni] === dv + 1) gv = clampAdd(gv, g[ni], G_CAP);
    }
    if (x + 1 < width) {
      ni = v + 1;
      if (dist[ni] === dv + 1) gv = clampAdd(gv, g[ni], G_CAP);
    }
    if (y > 0) {
      ni = v - width;
      if (dist[ni] === dv + 1) gv = clampAdd(gv, g[ni], G_CAP);
    }
    if (y + 1 < height) {
      ni = v + width;
      if (dist[ni] === dv + 1) gv = clampAdd(gv, g[ni], G_CAP);
    }
    g[v] = gv;
  }

  // Select top-k cells by a heuristic "importance" score.
  // Score favors cutting earlier (closer to sources) by dividing by (1 + dist).
  const heap = new MinHeap();
  const SCORE_CAP = 1e18;

  for (let oi = 0; oi < orderLen; oi++) {
    const v = order[oi];
    if (isSource[v] || isTown[v]) continue;
    if (burnable[v] === 0) continue;
    const gv = g[v];
    if (gv <= 0) continue; // no valuable towns downstream on shortest-path DAG
    const wv = ways[v];
    if (wv <= 0) continue;
    const d = dist[v];
    let score = (wv * gv) / (1 + d);
    if (!Number.isFinite(score) || score <= 0) continue;
    if (score > SCORE_CAP) score = SCORE_CAP;

    if (heap.size < cutBudget) {
      heap.push(score, v);
    } else if (score > heap.peekScore()) {
      heap.pop();
      heap.push(score, v);
    }
  }

  const picked = heap.toArray();
  // Sort descending for determinism
  picked.sort((a, b) => b[0] - a[0]);

  const firebreaks: number[][] = [];
  const used = new Uint8Array(n);
  const maxOut = Math.min(cutBudget, picked.length);
  for (let i = 0; i < maxOut; i++) {
    const idx = picked[i][1];
    if (used[idx]) continue;
    used[idx] = 1;
    const x = idx % width;
    const y = (idx / width) | 0;
    firebreaks.push([x, y]);
  }

  process.stdout.write(JSON.stringify({ firebreaks }));
}

main();
