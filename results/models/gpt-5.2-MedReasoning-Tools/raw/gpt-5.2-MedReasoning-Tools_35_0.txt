use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};
use std::time::{SystemTime, UNIX_EPOCH};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9e37_79b9_7f4a_7c15;
        }
        Self { state: s }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    fn gen_usize(&mut self, upper_exclusive: usize) -> usize {
        if upper_exclusive <= 1 {
            return 0;
        }
        (self.next_u64() as usize) % upper_exclusive
    }
    fn shuffle<T>(&mut self, a: &mut [T]) {
        for i in (1..a.len()).rev() {
            let j = self.gen_usize(i + 1);
            a.swap(i, j);
        }
    }
}

struct Scanner {
    input: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { input, idx: 0 }
    }
    fn next_usize(&mut self) -> Option<usize> {
        // Be tolerant of non-ASCII/UTF-16 pipeline bytes (e.g., NULs) by skipping to the next digit.
        while self.idx < self.input.len() && !self.input[self.idx].is_ascii_digit() {
            self.idx += 1;
        }
        if self.idx >= self.input.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.idx < self.input.len() {
            let b = self.input[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            x = x * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        Some(x)
    }
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapItem {
    score: usize,
    degree: usize,
    v: usize,
}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        // Max-heap: prefer higher score, then higher degree, then lower vertex id (stable-ish).
        self.score
            .cmp(&other.score)
            .then_with(|| self.degree.cmp(&other.degree))
            .then_with(|| other.v.cmp(&self.v))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn build_closed_neighbors(adj: &mut [Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }
    let mut closed = Vec::with_capacity(n);
    for v in 0..n {
        let mut nb = adj[v].clone();
        nb.push(v);
        nb.sort_unstable();
        nb.dedup();
        closed.push(nb);
    }
    closed
}

fn current_score(v: usize, dominated: &[bool], closed: &[Vec<usize>]) -> usize {
    closed[v].iter().filter(|&&u| !dominated[u]).count()
}

fn greedy_dominating_set(closed: &[Vec<usize>]) -> (Vec<usize>, Vec<u32>, Vec<bool>) {
    let n = closed.len();
    let mut dominated = vec![false; n];
    let mut dom_count = vec![0u32; n];
    let mut in_set = vec![false; n];
    let mut set = Vec::new();

    let mut undom = n;
    let mut heap = BinaryHeap::with_capacity(n);
    for v in 0..n {
        heap.push(HeapItem {
            score: closed[v].len(),
            degree: closed[v].len(),
            v,
        });
    }

    while undom > 0 {
        let Some(item) = heap.pop() else { break };
        let v = item.v;
        if in_set[v] {
            continue;
        }
        let real = current_score(v, &dominated, closed);
        if real == 0 {
            continue;
        }
        if real < item.score {
            heap.push(HeapItem {
                score: real,
                degree: item.degree,
                v,
            });
            continue;
        }

        in_set[v] = true;
        set.push(v);
        for &u in &closed[v] {
            if dom_count[u] == 0 {
                dominated[u] = true;
                undom -= 1;
            }
            dom_count[u] += 1;
        }
    }

    // Safety repair (should rarely trigger): add any still-undominated vertices.
    for v in 0..n {
        if !dominated[v] {
            in_set[v] = true;
            set.push(v);
            for &u in &closed[v] {
                if dom_count[u] == 0 {
                    dominated[u] = true;
                }
                dom_count[u] += 1;
            }
        }
    }

    (set, dom_count, in_set)
}

fn prune_redundant(
    set: &mut Vec<usize>,
    dom_count: &mut [u32],
    in_set: &mut [bool],
    closed: &[Vec<usize>],
) {
    if set.is_empty() {
        return;
    }
    let seed = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_nanos() as u64;
    let mut rng = XorShift64::new(seed ^ ((set.len() as u64) << 1));
    rng.shuffle(set);

    for &v in set.iter() {
        if !in_set[v] {
            continue;
        }
        let mut removable = true;
        for &u in &closed[v] {
            if dom_count[u] <= 1 {
                removable = false;
                break;
            }
        }
        if !removable {
            continue;
        }
        in_set[v] = false;
        for &u in &closed[v] {
            dom_count[u] -= 1;
        }
    }

    set.clear();
    for (v, &flag) in in_set.iter().enumerate() {
        if flag {
            set.push(v);
        }
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(x) => x,
        None => {
            println!("0\n");
            return;
        }
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let Some(u) = sc.next_usize() else { break };
        let Some(v) = sc.next_usize() else { break };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    let closed = build_closed_neighbors(&mut adj);
    let (mut set, mut dom_count, mut in_set) = greedy_dominating_set(&closed);
    prune_redundant(&mut set, &mut dom_count, &mut in_set, &closed);

    print!("{}", set.len());
    println!();
    if !set.is_empty() {
        for (i, v) in set.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{v}");
        }
    }
    println!();
}
