using System;
using System.Diagnostics;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int x = 0;
            while (c > 32)
            {
                x = unchecked(x * 10 + (c - '0'));
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private static long ComputeCost(int n, int[] flow, int[] dist, int[] p)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = p[i];
            int fi = i * n;
            int di = pi * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[fi + j] * dist[di + p[j]];
            }
        }
        return cost;
    }

    // Delta formula for swapping facilities a and b (asymmetric general case)
    private static long DeltaSwap(int n, int[] flow, int[] dist, int[] p, int a, int b)
    {
        if (a == b) return 0;
        int pa = p[a];
        int pb = p[b];
        if (pa == pb) return 0;

        long delta = 0;

        // base terms (u,v)
        delta += (long)(flow[a * n + a] - flow[b * n + b]) * (dist[pb * n + pb] - dist[pa * n + pa]);
        delta += (long)(flow[a * n + b] - flow[b * n + a]) * (dist[pb * n + pa] - dist[pa * n + pb]);

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = p[k];

            delta += (long)(flow[a * n + k] - flow[b * n + k]) * (dist[pb * n + pk] - dist[pa * n + pk]);
            delta += (long)(flow[k * n + a] - flow[k * n + b]) * (dist[pk * n + pb] - dist[pk * n + pa]);
        }

        return delta;
    }

    private static void Swap(int[] p, int a, int b)
    {
        int t = p[a];
        p[a] = p[b];
        p[b] = t;
    }

    private static int[] GreedyStart(int n, int[] flow, int[] dist)
    {
        long[] flowScore = new long[n];
        long[] distScore = new long[n];

        for (int i = 0; i < n; i++)
        {
            long fs = 0;
            int row = i * n;
            for (int j = 0; j < n; j++)
                fs += flow[row + j] + flow[j * n + i];
            flowScore[i] = fs;

            long ds = 0;
            int drow = i * n;
            for (int j = 0; j < n; j++)
                ds += dist[drow + j] + dist[j * n + i];
            distScore[i] = ds;
        }

        int[] fac = new int[n];
        int[] loc = new int[n];
        for (int i = 0; i < n; i++) { fac[i] = i; loc[i] = i; }

        Array.Sort(fac, (a, b) => flowScore[b].CompareTo(flowScore[a]));
        Array.Sort(loc, (a, b) => distScore[a].CompareTo(distScore[b]));

        int[] p = new int[n];
        for (int i = 0; i < n; i++)
            p[fac[i]] = loc[i];
        return p;
    }

    private static int[] RandomPermutation(int n, Random rng)
    {
        int[] p = new int[n];
        for (int i = 0; i < n; i++) p[i] = i;
        for (int i = n - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int t = p[i];
            p[i] = p[j];
            p[j] = t;
        }
        return p;
    }

    private static void ImproveRandomSwaps(int n, int[] flow, int[] dist, int[] p, ref long cost, Random rng, Stopwatch sw, long endMs)
    {
        // For large n: randomized first-improvement hillclimb
        // For small n: add periodic intensification passes on a subset
        int attemptsWithoutImprove = 0;
        int maxNoImprove = Math.Max(50_000, n * 200);

        while (sw.ElapsedMilliseconds < endMs && attemptsWithoutImprove < maxNoImprove)
        {
            int a = rng.Next(n);
            int b = rng.Next(n - 1);
            if (b >= a) b++;

            long delta = DeltaSwap(n, flow, dist, p, a, b);
            if (delta < 0)
            {
                Swap(p, a, b);
                cost += delta;
                attemptsWithoutImprove = 0;
            }
            else
            {
                attemptsWithoutImprove++;
            }

            // Occasional diversification (very light)
            if (attemptsWithoutImprove > 0 && (attemptsWithoutImprove % (n * 50 + 1)) == 0 && sw.ElapsedMilliseconds < endMs)
            {
                int x = rng.Next(n);
                int y = rng.Next(n - 1);
                if (y >= x) y++;
                long d2 = DeltaSwap(n, flow, dist, p, x, y);
                if (d2 < 0)
                {
                    Swap(p, x, y);
                    cost += d2;
                    attemptsWithoutImprove = 0;
                }
            }
        }

        if (n <= 200 && sw.ElapsedMilliseconds < endMs)
        {
            // Small intensification: a few best-improvement rounds with limited neighborhood sampling
            for (int round = 0; round < 5 && sw.ElapsedMilliseconds < endMs; round++)
            {
                bool improved = false;
                int samples = Math.Min(40_000, n * n);
                long bestDelta = 0;
                int bestA = -1, bestB = -1;

                for (int s = 0; s < samples && sw.ElapsedMilliseconds < endMs; s++)
                {
                    int a = rng.Next(n);
                    int b = rng.Next(n - 1);
                    if (b >= a) b++;
                    long delta = DeltaSwap(n, flow, dist, p, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestA = a;
                        bestB = b;
                    }
                }

                if (bestDelta < 0)
                {
                    Swap(p, bestA, bestB);
                    cost += bestDelta;
                    improved = true;
                }

                if (!improved) break;
            }
        }
    }

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        var fs = new FastScanner(Console.OpenStandardInput());

        int n;
        try
        {
            n = fs.NextInt();
        }
        catch
        {
            return;
        }

        int nn = checked(n * n);
        int[] flow = new int[nn];
        int[] dist = new int[nn];

        for (int i = 0; i < nn; i++) flow[i] = fs.NextInt();
        for (int i = 0; i < nn; i++) dist[i] = fs.NextInt();

        var rng = new Random(1234567);

        int[] bestP = GreedyStart(n, flow, dist);
        long bestCost = ComputeCost(n, flow, dist, bestP);

        // Time budget: leave some margin for output
        long hardEnd = 290_000; // ~4m50s
        if (n <= 200) hardEnd = 295_000;
        if (n <= 50) hardEnd = 298_000;
        long endMs = hardEnd;

        // Multi-start: greedy + a few randomized starts (count depends on n)
        int starts = 1;
        if (n <= 300) starts = 6;
        else if (n <= 800) starts = 4;
        else starts = 3;

        for (int s = 0; s < starts && sw.ElapsedMilliseconds < endMs; s++)
        {
            int[] p;
            if (s == 0) p = (int[])bestP.Clone();
            else p = RandomPermutation(n, rng);

            long cost = (s == 0) ? bestCost : ComputeCost(n, flow, dist, p);

            // Allocate a time slice per start
            long remaining = endMs - sw.ElapsedMilliseconds;
            if (remaining <= 0) break;
            long slice = remaining / (starts - s);
            long sliceEnd = sw.ElapsedMilliseconds + slice;

            ImproveRandomSwaps(n, flow, dist, p, ref cost, rng, sw, sliceEnd);

            if (cost < bestCost)
            {
                bestCost = cost;
                bestP = p;
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(bestCost);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) writer.Write(' ');
            writer.Write(bestP[i]);
        }
        writer.WriteLine();
    }
}
