use std::collections::HashMap;
use std::io::{self, Read, Write};

// Fast buffered binary/text reader for stdin.
struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn skip_ws(&mut self) {
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' {
                self.idx += 1;
            } else {
                break;
            }
        }
    }

    fn read_u64_ascii(&mut self) -> u64 {
        self.skip_ws();
        let mut v: u64 = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if (b'0'..=b'9').contains(&b) {
                v = v * 10 + (b - b'0') as u64;
                self.idx += 1;
            } else {
                break;
            }
        }
        v
    }

    #[inline]
    fn read_f32_le(&mut self) -> f32 {
        let b0 = self.buf[self.idx];
        let b1 = self.buf[self.idx + 1];
        let b2 = self.buf[self.idx + 2];
        let b3 = self.buf[self.idx + 3];
        self.idx += 4;
        f32::from_le_bytes([b0, b1, b2, b3])
    }

    #[inline]
    fn read_point3(&mut self) -> [f32; 3] {
        [self.read_f32_le(), self.read_f32_le(), self.read_f32_le()]
    }
}

#[inline]
fn dist2(a: [f32; 3], b: [f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

// Simple, fast PRNG (xorshift64*).
#[derive(Clone)]
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed | 1 }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.state = x;
        x.wrapping_mul(0x2545F4914F6CDD1D)
    }
    #[inline]
    fn next_f32(&mut self) -> f32 {
        // [0,1)
        let v = (self.next_u64() >> 40) as u32;
        (v as f32) * (1.0 / (1u32 << 24) as f32)
    }
    #[inline]
    fn gen_range_usize(&mut self, n: usize) -> usize {
        (self.next_u64() % (n as u64)) as usize
    }
}

fn kmeanspp_init(points: &[[f32; 3]], k: usize, rng: &mut XorShift64) -> Vec<[f32; 3]> {
    let n = points.len();
    if k == 0 {
        return Vec::new();
    }
    if n == 0 {
        return vec![[0.0, 0.0, 0.0]; k];
    }
    let mut centroids = Vec::with_capacity(k);
    centroids.push(points[rng.gen_range_usize(n)]);

    let mut d2 = vec![0f32; n];
    for i in 0..n {
        d2[i] = dist2(points[i], centroids[0]);
    }

    while centroids.len() < k {
        let mut sum = 0f64;
        for &v in &d2 {
            sum += v as f64;
        }
        if !sum.is_finite() || sum <= 0.0 {
            // Degenerate: pick random remaining.
            centroids.push(points[rng.gen_range_usize(n)]);
            continue;
        }
        let mut r = (rng.next_f32() as f64) * sum;
        let mut pick = 0usize;
        for (i, &v) in d2.iter().enumerate() {
            r -= v as f64;
            if r <= 0.0 {
                pick = i;
                break;
            }
        }
        centroids.push(points[pick]);
        let cnew = *centroids.last().unwrap();
        for i in 0..n {
            let dv = dist2(points[i], cnew);
            if dv < d2[i] {
                d2[i] = dv;
            }
        }
    }
    centroids
}

fn assign_bruteforce(p: [f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best = 0usize;
    let mut bestd = dist2(p, centroids[0]);
    for (i, &c) in centroids.iter().enumerate().skip(1) {
        let d = dist2(p, c);
        if d < bestd {
            bestd = d;
            best = i;
        }
    }
    best
}

#[derive(Clone, Copy)]
struct GridParams {
    inv_cell: f32,
    origin: [f32; 3],
}

#[inline]
fn cell_coord(x: f32, origin: f32, inv_cell: f32) -> i32 {
    ((x - origin) * inv_cell).floor() as i32
}

#[inline]
fn cell_key(cx: i32, cy: i32, cz: i32) -> u64 {
    // Pack three signed 21-bit ints into u64 (bias to unsigned).
    const BIAS: i64 = 1 << 20;
    let ux = (cx as i64 + BIAS) as u64;
    let uy = (cy as i64 + BIAS) as u64;
    let uz = (cz as i64 + BIAS) as u64;
    (ux & 0x1F_FFFF) | ((uy & 0x1F_FFFF) << 21) | ((uz & 0x1F_FFFF) << 42)
}

fn build_grid(
    centroids: &[[f32; 3]],
    gp: GridParams,
    grid: &mut HashMap<u64, Vec<u32>>,
) {
    grid.clear();
    grid.reserve(centroids.len() * 2);
    for (i, &c) in centroids.iter().enumerate() {
        let cx = cell_coord(c[0], gp.origin[0], gp.inv_cell);
        let cy = cell_coord(c[1], gp.origin[1], gp.inv_cell);
        let cz = cell_coord(c[2], gp.origin[2], gp.inv_cell);
        let key = cell_key(cx, cy, cz);
        grid.entry(key).or_default().push(i as u32);
    }
}

fn estimate_grid_params(centroids: &[[f32; 3]], k: usize) -> GridParams {
    // Choose a cell size to keep ~O(1) centroids per cell on average.
    let mut minv = [f32::INFINITY; 3];
    let mut maxv = [f32::NEG_INFINITY; 3];
    for &c in centroids {
        for d in 0..3 {
            minv[d] = minv[d].min(c[d]);
            maxv[d] = maxv[d].max(c[d]);
        }
    }
    let spanx = (maxv[0] - minv[0]).abs().max(1e-3);
    let spany = (maxv[1] - minv[1]).abs().max(1e-3);
    let spanz = (maxv[2] - minv[2]).abs().max(1e-3);
    let volume = (spanx as f64) * (spany as f64) * (spanz as f64);
    let target_cells = (k as f64).max(8.0);
    let mut cell = (volume / target_cells).cbrt() as f32;
    // Slightly larger cells reduce empty-cell misses.
    cell *= 1.6;
    if !cell.is_finite() || cell <= 0.0 {
        cell = 1.0;
    }
    GridParams {
        inv_cell: 1.0 / cell,
        origin: minv,
    }
}

fn assign_with_grid(
    p: [f32; 3],
    centroids: &[[f32; 3]],
    gp: GridParams,
    grid: &HashMap<u64, Vec<u32>>,
    rng: &mut XorShift64,
) -> usize {
    let cx = cell_coord(p[0], gp.origin[0], gp.inv_cell);
    let cy = cell_coord(p[1], gp.origin[1], gp.inv_cell);
    let cz = cell_coord(p[2], gp.origin[2], gp.inv_cell);

    let mut best = None::<(usize, f32)>;

    // Search neighbor rings; cap to keep runtime bounded.
    for r in 0i32..=2i32 {
        for dx in -r..=r {
            for dy in -r..=r {
                for dz in -r..=r {
                    if r > 0 && dx.abs().max(dy.abs()).max(dz.abs()) != r {
                        continue; // only outer shell for this r
                    }
                    let key = cell_key(cx + dx, cy + dy, cz + dz);
                    if let Some(list) = grid.get(&key) {
                        for &idx in list {
                            let i = idx as usize;
                            let d = dist2(p, centroids[i]);
                            match best {
                                None => best = Some((i, d)),
                                Some((_, bd)) if d < bd => best = Some((i, d)),
                                _ => {}
                            }
                        }
                    }
                }
            }
        }
        if best.is_some() {
            return best.unwrap().0;
        }
    }

    // Fallback: sample a small random subset.
    let k = centroids.len();
    let trials = 32usize.min(k);
    let mut besti = 0usize;
    let mut bestd = dist2(p, centroids[0]);
    for _ in 0..trials {
        let i = rng.gen_range_usize(k);
        let d = dist2(p, centroids[i]);
        if d < bestd {
            bestd = d;
            besti = i;
        }
    }
    besti
}

fn main() {
    let mut input = FastInput::new();
    let num_points = input.read_u64_ascii() as usize;
    let num_clusters = input.read_u64_ascii() as usize;

    let mut out = io::BufWriter::new(io::stdout().lock());

    if num_points == 0 {
        return;
    }
    if num_clusters == 0 {
        // Nothing sensible; assign all to 0 (still a u32 stream).
        for _ in 0..num_points {
            out.write_all(&0u32.to_le_bytes()).unwrap();
        }
        return;
    }

    // Initialize using a small prefix buffer, then stream.
    // Keep this small enough for memory but large enough for decent init.
    let init_cap = 200_000usize;
    let init_points = num_points.min(init_cap.max(num_clusters.saturating_mul(32)).min(num_points));

    let seed = (num_points as u64)
        .wrapping_mul(0x9E3779B97F4A7C15)
        ^ (num_clusters as u64).rotate_left(17);
    let mut rng = XorShift64::new(seed);

    let mut buf_points: Vec<[f32; 3]> = Vec::with_capacity(init_points);
    for _ in 0..init_points {
        buf_points.push(input.read_point3());
    }

    // If clusters exceed buffered points, allow empty clusters by repeating points.
    let mut centroids = if num_clusters <= buf_points.len() {
        kmeanspp_init(&buf_points, num_clusters, &mut rng)
    } else {
        let mut c = Vec::with_capacity(num_clusters);
        for i in 0..num_clusters {
            let p = buf_points[i % buf_points.len()];
            c.push(p);
        }
        c
    };

    // Small: do a few Lloyd iterations on the init buffer for better start.
    // Keep iterations low and adaptive.
    if buf_points.len() >= num_clusters && num_clusters <= 2048 && buf_points.len() <= 1_000_000 {
        let iters = if num_clusters <= 64 { 10 } else { 6 };
        let k = num_clusters;
        let mut sums = vec![[0f64; 3]; k];
        let mut counts = vec![0u32; k];
        for _ in 0..iters {
            for s in sums.iter_mut() {
                *s = [0.0, 0.0, 0.0];
            }
            counts.fill(0);
            for &p in &buf_points {
                let ci = assign_bruteforce(p, &centroids);
                sums[ci][0] += p[0] as f64;
                sums[ci][1] += p[1] as f64;
                sums[ci][2] += p[2] as f64;
                counts[ci] += 1;
            }
            for i in 0..k {
                if counts[i] > 0 {
                    let inv = 1.0 / (counts[i] as f64);
                    centroids[i] = [
                        (sums[i][0] * inv) as f32,
                        (sums[i][1] * inv) as f32,
                        (sums[i][2] * inv) as f32,
                    ];
                } else {
                    centroids[i] = buf_points[rng.gen_range_usize(buf_points.len())];
                }
            }
        }
    }

    // Prepare optional grid acceleration for larger k.
    let use_grid = num_clusters > 256;
    let mut gp = estimate_grid_params(&centroids, num_clusters);
    let mut grid: HashMap<u64, Vec<u32>> = HashMap::new();
    if use_grid {
        build_grid(&centroids, gp, &mut grid);
    }

    // Online update state: counts per centroid.
    let mut counts: Vec<u32> = vec![0; num_clusters];

    // First, assign buffered init points and do online updates.
    // Buffer their labels (so we can write once, sequentially).
    let mut init_labels: Vec<u32> = Vec::with_capacity(init_points);
    for &p in &buf_points {
        let ci = if use_grid {
            assign_with_grid(p, &centroids, gp, &grid, &mut rng)
        } else {
            assign_bruteforce(p, &centroids)
        };
        init_labels.push(ci as u32);
        let cnt = counts[ci].wrapping_add(1);
        counts[ci] = cnt;
        let lr = 1.0f32 / (cnt as f32);
        let c = &mut centroids[ci];
        c[0] += (p[0] - c[0]) * lr;
        c[1] += (p[1] - c[1]) * lr;
        c[2] += (p[2] - c[2]) * lr;
    }

    // Write init labels.
    for &lab in &init_labels {
        out.write_all(&lab.to_le_bytes()).unwrap();
    }

    // Stream remaining points.
    // Periodically refresh grid parameters for drifting centroids.
    let mut since_rebuild: usize = 0;
    let rebuild_every = if num_clusters <= 10_000 { 200_000 } else { 500_000 };
    let mut remaining = num_points - init_points;

    while remaining > 0 {
        let p = input.read_point3();
        let ci = if use_grid {
            assign_with_grid(p, &centroids, gp, &grid, &mut rng)
        } else {
            assign_bruteforce(p, &centroids)
        };

        out.write_all(&(ci as u32).to_le_bytes()).unwrap();

        let cnt = counts[ci].wrapping_add(1);
        counts[ci] = cnt;
        // Diminishing step size. Clamp to avoid vanishing on extreme streams.
        let lr = (1.0f32 / (cnt as f32)).max(0.001);
        let c = &mut centroids[ci];
        c[0] += (p[0] - c[0]) * lr;
        c[1] += (p[1] - c[1]) * lr;
        c[2] += (p[2] - c[2]) * lr;

        since_rebuild += 1;
        if use_grid && since_rebuild >= rebuild_every {
            gp = estimate_grid_params(&centroids, num_clusters);
            build_grid(&centroids, gp, &mut grid);
            since_rebuild = 0;
        }

        remaining -= 1;
    }

    out.flush().unwrap();
}
