from __future__ import annotations

import math
from typing import Any, Dict, Iterable, List, Optional, Tuple


# Persistent controller state (reset between scenarios by the harness).
_mode: str = "intercept"  # intercept | approach | flare | rollout | go_around
_t: float = 0.0
_go_around_time: float = -1.0

_gs_i: float = 0.0
_loc_i: float = 0.0
_spd_i: float = 0.0

_gs_f: Optional[float] = None
_loc_f: Optional[float] = None
_ra_f: Optional[float] = None
_ias_f: Optional[float] = None


def _clamp(x: float, lo: float, hi: float) -> float:
    return lo if x < lo else hi if x > hi else x


def _isfinite(x: Any) -> bool:
    return isinstance(x, (int, float)) and math.isfinite(float(x))


def _wrap_deg(a: float) -> float:
    a = (a + 180.0) % 360.0 - 180.0
    return a


def _median(xs: List[float]) -> float:
    xs = sorted(xs)
    n = len(xs)
    if n == 0:
        raise ValueError("median of empty list")
    mid = n // 2
    return xs[mid] if (n % 2) else 0.5 * (xs[mid - 1] + xs[mid])


def _robust_vote(readings: Iterable[Tuple[float, Any]]) -> Optional[float]:
    """
    readings: iterable of (value, valid_flag)
    Returns a robust estimate or None if no valid readings.
    """
    vals: List[float] = []
    for v, ok in readings:
        if ok and _isfinite(v):
            vals.append(float(v))
    if not vals:
        return None
    if len(vals) == 1:
        return vals[0]
    m = _median(vals)
    # Drop gross outliers (e.g., stuck receiver) and re-median.
    kept = [v for v in vals if abs(v - m) <= 2.5]
    if kept:
        return _median(kept)
    return m


def _lowpass(prev: Optional[float], x: Optional[float], alpha: float) -> Optional[float]:
    if x is None:
        return prev
    if prev is None:
        return x
    return prev + alpha * (x - prev)


def _get(state: Dict[str, Any], key: str, default: Any = None) -> Any:
    return state.get(key, default)


def _pick_speed(state: Dict[str, Any]) -> Optional[float]:
    ias = _robust_vote(
        [
            (_get(state, "ias_1_kt"), _get(state, "ias_1_valid")),
            (_get(state, "ias_2_kt"), _get(state, "ias_2_valid")),
            (_get(state, "ias_3_kt"), _get(state, "ias_3_valid")),
        ]
    )
    if ias is not None:
        return ias
    # Fallback: groundspeed as a weak proxy if all pitots failed.
    gs = _get(state, "groundspeed_kt")
    return float(gs) if _isfinite(gs) else None


def _pick_att(state: Dict[str, Any], key: str) -> Optional[float]:
    return _robust_vote(
        [
            (_get(state, f"irs_1_{key}"), _get(state, "irs_1_valid")),
            (_get(state, f"irs_2_{key}"), _get(state, "irs_2_valid")),
            (_get(state, f"irs_3_{key}"), _get(state, "irs_3_valid")),
        ]
    )


def autopilot_step(state: dict, dt: float) -> dict:
    """
    ILS autoland autopilot step.
    Returns control surface / configuration commands as a dict.
    """
    global _mode, _t, _go_around_time
    global _gs_i, _loc_i, _spd_i
    global _gs_f, _loc_f, _ra_f, _ias_f

    dt = float(dt) if _isfinite(dt) and dt > 0.0 else 0.02
    dt = _clamp(dt, 0.01, 0.2)
    _t += dt

    # --- Sensor fusion ---
    gs = _robust_vote(
        [
            (_get(state, "glideslope_1_dots"), _get(state, "glideslope_1_valid")),
            (_get(state, "glideslope_2_dots"), _get(state, "glideslope_2_valid")),
            (_get(state, "glideslope_3_dots"), _get(state, "glideslope_3_valid")),
        ]
    )
    loc = _robust_vote(
        [
            (_get(state, "localizer_1_dots"), _get(state, "localizer_1_valid")),
            (_get(state, "localizer_2_dots"), _get(state, "localizer_2_valid")),
            (_get(state, "localizer_3_dots"), _get(state, "localizer_3_valid")),
        ]
    )

    ra = _get(state, "radio_alt_ft")
    ra = float(ra) if _isfinite(ra) and ra >= -50 else None
    if ra is not None:
        ra = max(0.0, ra)

    ias = _pick_speed(state)
    pitch = _pick_att(state, "pitch_deg")
    roll = _pick_att(state, "roll_deg")
    track = _pick_att(state, "track_deg")
    heading = _pick_att(state, "heading_deg")
    vs = _pick_att(state, "vs_fpm")

    # Light filtering to ride through flicker.
    _gs_f = _lowpass(_gs_f, gs, alpha=_clamp(dt / 0.6, 0.02, 0.25))
    _loc_f = _lowpass(_loc_f, loc, alpha=_clamp(dt / 0.6, 0.02, 0.25))
    _ra_f = _lowpass(_ra_f, ra, alpha=_clamp(dt / 0.3, 0.05, 0.35))
    _ias_f = _lowpass(_ias_f, ias, alpha=_clamp(dt / 0.8, 0.02, 0.2))

    gs = _gs_f
    loc = _loc_f
    ra = _ra_f
    ias = _ias_f

    # --- Phase selection ---
    on_ground = bool(_get(state, "on_ground")) or bool(_get(state, "weight_on_wheels"))
    below_dh = bool(_get(state, "below_decision_height"))
    dme = _get(state, "dme_1_nm")
    if not _isfinite(dme):
        dme = _get(state, "dme_2_nm")
    dme = float(dme) if _isfinite(dme) else None

    if on_ground:
        _mode = "rollout"
    else:
        if _mode != "go_around":
            if ra is not None:
                if ra < 50.0:
                    _mode = "flare"
                elif ra < 1000.0:
                    _mode = "approach"
                else:
                    _mode = "intercept"
            else:
                # Radar alt failed: use DME to approximate phase.
                if dme is not None and dme <= 3.0:
                    _mode = "approach"
                else:
                    _mode = "intercept"

    # --- Core safety / go-around trigger ---
    target_hdg = _get(state, "target_hdg_deg")
    target_hdg = float(target_hdg) if _isfinite(target_hdg) else (heading if heading is not None else 0.0)

    target_vref = _get(state, "target_vref_kt")
    target_vref = float(target_vref) if _isfinite(target_vref) else 137.0

    # Basic wind/gust buffer: don't chase exact Vref in turbulence.
    spd_target = target_vref + (10.0 if _mode in ("intercept", "approach") else 0.0)
    if _isfinite(_get(state, "target_ias_kt")):
        spd_target = float(_get(state, "target_ias_kt"))

    # Unstable approach criteria near DH: large ILS errors or extreme speed.
    unstable = False
    if below_dh:
        if gs is None or loc is None:
            unstable = True
        else:
            if abs(gs) > 2.0 or abs(loc) > 2.0:
                unstable = True
        if ias is not None and (ias < target_vref - 10.0 or ias > target_vref + 25.0):
            unstable = True
        if vs is not None and vs < -1200.0:
            unstable = True

    # Also go-around if we're very low and clearly not tracking.
    if ra is not None and ra < 150.0 and gs is not None and loc is not None:
        if abs(gs) > 3.0 or abs(loc) > 3.0:
            unstable = True

    if unstable and _mode not in ("go_around", "rollout"):
        _mode = "go_around"
        _go_around_time = _t
        # Reset integrators to avoid fighting the recovery.
        _gs_i = 0.0
        _loc_i = 0.0
        _spd_i = 0.0

    # --- Configuration commands ---
    # Use radio altitude if available; otherwise use DME for conservative configuration.
    conf_ref = ra if ra is not None else (max(0.0, (dme or 10.0) * 300.0))
    gear = 1.0 if conf_ref < 2500.0 else 0.0
    if _mode == "go_around":
        gear = 0.0 if (ra is not None and ra > 200.0) else 1.0

    if conf_ref > 2000.0:
        flaps = 5.0
    elif conf_ref > 1200.0:
        flaps = 10.0
    elif conf_ref > 700.0:
        flaps = 15.0
    elif conf_ref > 300.0:
        flaps = 25.0
    else:
        flaps = 35.0
    if _mode == "go_around":
        flaps = 15.0

    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    # --- Lateral control (localizer + track/heading) ---
    # Desired bank is primarily from LOC error, with a small track-to-runway correction.
    loc_err = float(loc) if loc is not None else 0.0
    track_err = 0.0
    if track is not None:
        track_err = _wrap_deg(float(track) - float(target_hdg))

    # Bank command: negative loc means left of centerline => bank left (negative roll).
    bank_cmd = _clamp(-(9.0 * loc_err) - 0.15 * track_err, -25.0, 25.0)
    roll_now = float(roll) if roll is not None else 0.0
    aileron = _clamp(0.06 * (bank_cmd - roll_now), -1.0, 1.0)

    # Rudder: damp crab (heading-track) and help with asymmetric thrust.
    crab = 0.0
    if heading is not None and track is not None:
        crab = _wrap_deg(float(heading) - float(track))
    thrust1 = _get(state, "engine_1_thrust_pct")
    thrust2 = _get(state, "engine_2_thrust_pct")
    thrust_delta = 0.0
    if _isfinite(thrust1) and _isfinite(thrust2):
        thrust_delta = float(thrust2) - float(thrust1)
    rudder = _clamp(-0.02 * crab + 0.004 * thrust_delta, -1.0, 1.0)

    # Integrator only when tracking ILS (prevents bias / stuck receiver problems).
    if _mode in ("intercept", "approach") and loc is not None and abs(loc_err) < 2.0:
        _loc_i = _clamp(_loc_i + (-0.15 * loc_err) * dt, -0.35, 0.35)
    else:
        _loc_i *= 0.98
    aileron = _clamp(aileron + _loc_i, -1.0, 1.0)

    # --- Vertical control (glideslope + flare shaping) ---
    gs_err = float(gs) if gs is not None else 0.0
    pitch_now = float(pitch) if pitch is not None else 0.0
    vs_now = float(vs) if vs is not None else 0.0

    # Convert GS dots to vertical speed correction.
    # Positive GS dots = above path -> command more descent (more negative vs).
    target_vs = float(_get(state, "target_vs_fpm")) if _isfinite(_get(state, "target_vs_fpm")) else -700.0
    if _mode in ("intercept", "approach") and gs is not None:
        target_vs = target_vs - 350.0 * gs_err
    target_vs = _clamp(target_vs, -1400.0, -200.0)

    # Flare: progressively reduce sink, small pitch-up.
    if _mode == "flare":
        h = ra if ra is not None else 30.0
        # 50ft -> -300fpm, 20ft -> -120fpm, 5ft -> -60fpm
        if h > 20.0:
            target_vs = -300.0 + (h - 20.0) * (-10.0)  # gentle trend
        elif h > 5.0:
            target_vs = -120.0
        else:
            target_vs = -60.0
        target_vs = _clamp(target_vs, -450.0, -30.0)

    # Pitch command from VS error (simple outer-loop).
    vs_err = target_vs - vs_now
    pitch_cmd = _clamp(2.5 + 0.0035 * vs_err, -2.0, 8.0)
    if _mode == "go_around":
        pitch_cmd = 10.0
    elevator = _clamp(0.10 * (pitch_cmd - pitch_now) - 0.0003 * vs_now, -1.0, 1.0)

    # Trim integrator to reduce steady-state ILS errors (helps jam/partial authority).
    if _mode in ("intercept", "approach") and gs is not None and abs(gs_err) < 2.0:
        _gs_i = _clamp(_gs_i + (-0.10 * gs_err) * dt, -0.4, 0.4)
    else:
        _gs_i *= 0.98
    elevator_trim = _gs_i

    # --- Throttle control (speed) ---
    engine1_running = bool(_get(state, "engine_1_running", True))
    engine2_running = bool(_get(state, "engine_2_running", True))
    both_dead = (not engine1_running) and (not engine2_running)

    throttle = 0.55
    if both_dead:
        throttle = 0.0
    else:
        # Outer-loop speed PI -> throttle (with some anti-windshear aggressiveness).
        if ias is not None:
            spd_err = spd_target - float(ias)
        else:
            spd_err = 0.0
        # Integrate only when airborne and not flaring.
        if _mode in ("intercept", "approach") and ias is not None:
            _spd_i = _clamp(_spd_i + 0.03 * spd_err * dt, -0.25, 0.25)
        else:
            _spd_i *= 0.98

        p = 0.012 * spd_err
        throttle = _clamp(0.55 + p + _spd_i, 0.0, 1.0)

        # Flare/idle.
        if _mode == "flare":
            throttle = 0.05

        if _mode == "go_around":
            throttle = 1.0

    throttle_1 = throttle
    throttle_2 = throttle

    # --- Go-around completion logic ---
    if _mode == "go_around":
        speedbrake = 0.0
        spoiler = 0.0
        if ra is not None and ra >= 1500.0 and (_t - _go_around_time) > 8.0:
            # Hand back to intercept; sim resets shortly after go-around completion.
            _mode = "intercept"

    # --- Rollout logic ---
    if _mode == "rollout":
        # Immediately deploy ground spoilers and begin braking / reverse.
        spoiler = 1.0
        speedbrake = 1.0
        gs_kt = _get(state, "groundspeed_kt")
        gs_kt = float(gs_kt) if _isfinite(gs_kt) else 0.0
        autobrake = 1.0 if gs_kt > 80.0 else 0.7 if gs_kt > 50.0 else 0.4 if gs_kt > 20.0 else 0.0
        if engine1_running and gs_kt > 60.0:
            reverse_1 = 1.0
        if engine2_running and gs_kt > 60.0:
            reverse_2 = 1.0
        throttle_1 = 0.0
        throttle_2 = 0.0
        # Stay on centerline using LOC if available, else track runway heading.
        if loc is not None:
            bank_cmd = _clamp(-(5.0 * float(loc)), -10.0, 10.0)
            aileron = _clamp(0.08 * (bank_cmd - roll_now), -1.0, 1.0)
            rudder = _clamp(-0.25 * float(loc), -1.0, 1.0)
        else:
            if heading is not None:
                hdg_err = _wrap_deg(float(heading) - float(target_hdg))
                rudder = _clamp(-0.02 * hdg_err, -1.0, 1.0)
                aileron = 0.0
        elevator = _clamp(-0.2, -0.4, 0.0)  # keep nose down lightly
        elevator_trim = 0.0

    # --- Output ---
    return {
        "elevator": float(_clamp(elevator, -1.0, 1.0)),
        "aileron": float(_clamp(aileron, -1.0, 1.0)),
        "rudder": float(_clamp(rudder, -1.0, 1.0)),
        "throttle_1": float(_clamp(throttle_1, 0.0, 1.0)),
        "throttle_2": float(_clamp(throttle_2, 0.0, 1.0)),
        "elevator_trim": float(_clamp(elevator_trim, -1.0, 1.0)),
        "flaps": float(_clamp(flaps, 0.0, 40.0)),
        "gear": float(1.0 if gear >= 0.5 else 0.0),
        "speedbrake": float(_clamp(speedbrake, 0.0, 1.0)),
        "spoiler": float(1.0 if spoiler >= 0.5 else 0.0),
        "autobrake": float(_clamp(autobrake, 0.0, 1.0)),
        "reverse_1": float(_clamp(reverse_1, 0.0, 1.0)),
        "reverse_2": float(_clamp(reverse_2, 0.0, 1.0)),
    }
