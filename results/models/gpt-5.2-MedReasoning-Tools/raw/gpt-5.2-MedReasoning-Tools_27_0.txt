// Heuristic k-coloring solver.
// Assumes input instances are k-colorable; uses adaptive greedy + local repair.

namespace gc {

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    unsigned char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline unsigned char read() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        unsigned char c;
        do {
            c = read();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = read();
        }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = read();
        }
        out = neg ? -val : val;
        return true;
    }
};

struct Graph {
    int n = 0;
    int64_t m = 0;
    std::vector<int> off;      // size n+1
    std::vector<int> adj;      // size 2m
    std::vector<int> deg;      // size n
};

static inline uint64_t splitmix64(uint64_t &x) {
    x += 0x9e3779b97f4a7c15ull;
    uint64_t z = x;
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ull;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebull;
    return z ^ (z >> 31);
}

Graph read_graph(FastInput &in, int &k) {
    int n, m, kk;
    if (!in.readInt(n)) std::exit(0);
    in.readInt(m);
    in.readInt(kk);
    k = kk;

    Graph g;
    g.n = n;
    g.m = m;
    g.deg.assign(n, 0);

    std::vector<int> eu;
    std::vector<int> ev;
    eu.reserve(static_cast<size_t>(m));
    ev.reserve(static_cast<size_t>(m));

    for (int i = 0; i < m; ++i) {
        int u, v;
        in.readInt(u);
        in.readInt(v);
        if (u == v) continue; // ignore self loops
        eu.push_back(u);
        ev.push_back(v);
        g.deg[u]++;
        g.deg[v]++;
    }

    g.off.assign(n + 1, 0);
    for (int i = 0; i < n; ++i) g.off[i + 1] = g.off[i] + g.deg[i];
    g.adj.assign(g.off[n], 0);

    std::vector<int> cur = g.off;
    for (size_t i = 0; i < eu.size(); ++i) {
        int u = eu[i], v = ev[i];
        g.adj[cur[u]++] = v;
        g.adj[cur[v]++] = u;
    }

    // Optional: sort adjacency for cache/consistency on smaller graphs
    if (n <= 200000 && g.off[n] <= 4000000) {
        for (int v = 0; v < n; ++v) {
            int b = g.off[v], e = g.off[v + 1];
            std::sort(g.adj.begin() + b, g.adj.begin() + e);
        }
    }

    return g;
}

struct Solver {
    const Graph *g = nullptr;
    int k = 0;
    std::vector<int> color;

    // Working buffers
    std::vector<int> order;
    std::vector<int> conflicts;
    std::vector<int> visited;
    std::vector<int> q;
    std::vector<int> tmp;
    std::vector<int> cnt_buf;
    std::vector<int> present_buf;

    explicit Solver(const Graph &graph, int kk) : g(&graph), k(kk) {
        color.assign(g->n, -1);
        order.resize(g->n);
        visited.assign(g->n, 0);
        q.reserve(1024);
        tmp.reserve(256);
        if (k > 64) cnt_buf.assign(k, 0);
        present_buf.reserve(std::min(k, 64));
    }

    inline void clear_colors() {
        std::fill(color.begin(), color.end(), -1);
    }

    inline bool can_use(int v, int c) const {
        int b = g->off[v], e = g->off[v + 1];
        for (int i = b; i < e; ++i) {
            int u = g->adj[i];
            if (color[u] == c) return false;
        }
        return true;
    }

    int first_available_color64(int v) const {
        uint64_t forbidden = 0;
        int b = g->off[v], e = g->off[v + 1];
        for (int i = b; i < e; ++i) {
            int u = g->adj[i];
            int cu = color[u];
            if (cu >= 0) forbidden |= (1ull << cu);
        }
        uint64_t avail = (~forbidden);
        uint64_t mask = (k == 64) ? ~0ull : ((1ull << k) - 1ull);
        avail &= mask;
        if (!avail) return -1;
#if defined(_MSC_VER)
        unsigned long idx;
        _BitScanForward64(&idx, avail);
        return static_cast<int>(idx);
#else
        return __builtin_ctzll(avail);
#endif
    }

    int first_available_color_general(int v, std::vector<uint64_t> &bits) const {
        std::fill(bits.begin(), bits.end(), 0ull);
        int b = g->off[v], e = g->off[v + 1];
        for (int i = b; i < e; ++i) {
            int u = g->adj[i];
            int cu = color[u];
            if (cu >= 0) bits[cu >> 6] |= (1ull << (cu & 63));
        }
        int words = (k + 63) >> 6;
        for (int w = 0; w < words; ++w) {
            uint64_t forbidden = bits[w];
            uint64_t avail = ~forbidden;
            if (w == words - 1) {
                int rem = k & 63;
                if (rem) avail &= ((1ull << rem) - 1ull);
            }
            if (avail) {
#if defined(_MSC_VER)
                unsigned long idx;
                _BitScanForward64(&idx, avail);
                return (w << 6) + static_cast<int>(idx);
#else
                return (w << 6) + __builtin_ctzll(avail);
#endif
            }
        }
        return -1;
    }

    void build_order_degree(bool random_ties, uint64_t seed) {
        std::iota(order.begin(), order.end(), 0);
        if (random_ties) {
            // stable-ish: shuffle then sort by degree
            uint64_t x = seed;
            for (int i = (int)order.size() - 1; i > 0; --i) {
                uint64_t r = splitmix64(x);
                int j = (int)(r % (uint64_t)(i + 1));
                std::swap(order[i], order[j]);
            }
        }
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            int da = g->deg[a], db = g->deg[b];
            if (da != db) return da > db;
            return a < b;
        });
    }

    void build_order_shuffle(uint64_t seed) {
        std::iota(order.begin(), order.end(), 0);
        uint64_t x = seed;
        for (int i = (int)order.size() - 1; i > 0; --i) {
            uint64_t r = splitmix64(x);
            int j = (int)(r % (uint64_t)(i + 1));
            std::swap(order[i], order[j]);
        }
    }

    bool greedy_color() {
        conflicts.clear();
        if (k <= 64) {
            for (int v : order) {
                int c = first_available_color64(v);
                if (c < 0) {
                    conflicts.push_back(v);
                } else {
                    color[v] = c;
                }
            }
        } else {
            std::vector<uint64_t> bits((k + 63) >> 6);
            for (int v : order) {
                int c = first_available_color_general(v, bits);
                if (c < 0) conflicts.push_back(v);
                else color[v] = c;
            }
        }
        return conflicts.empty();
    }

    // Try to assign a color to v by moving neighbors of a target color.
    bool kempe_repair_vertex(int v, uint64_t &seed) {
        // Build forbidden set and neighbor lists by color (only for colors used among neighbors).
        // Prefer low-degree vertices for repair.
        tmp.clear();

        // Count conflicts per color around v.
        // Use preallocated buffers to avoid per-call allocations.
        present_buf.clear();

        int b = g->off[v], e = g->off[v + 1];
        if (k <= 64) {
            int cnt_small[64];
            std::memset(cnt_small, 0, sizeof(cnt_small));

            for (int i = b; i < e; ++i) {
                int u = g->adj[i];
                int cu = color[u];
                if (cu >= 0 && cnt_small[cu]++ == 0) present_buf.push_back(cu);
            }

            // Missing color => direct assignment.
            if ((int)present_buf.size() < k) {
                uint64_t forbidden = 0;
                for (int c : present_buf) forbidden |= (1ull << c);
                uint64_t mask = (k == 64) ? ~0ull : ((1ull << k) - 1ull);
                uint64_t avail = (~forbidden) & mask;
                if (avail) {
#if defined(_MSC_VER)
                    unsigned long idx;
                    _BitScanForward64(&idx, avail);
                    color[v] = (int)idx;
#else
                    color[v] = __builtin_ctzll(avail);
#endif
                    return true;
                }
            }

            std::sort(present_buf.begin(), present_buf.end(), [&](int a, int b2) {
                if (cnt_small[a] != cnt_small[b2]) return cnt_small[a] < cnt_small[b2];
                return a < b2;
            });

            int attempts = std::min<int>((int)present_buf.size(), 6);
            for (int t = 0; t < attempts; ++t) {
                int target = present_buf[t];
                tmp.clear();
                for (int i = b; i < e; ++i) {
                    int u = g->adj[i];
                    if (color[u] == target) tmp.push_back(u);
                }

                std::vector<std::pair<int, int>> moves;
                moves.reserve(tmp.size());
                bool ok = true;
                for (int u : tmp) {
                    int old = color[u];
                    color[u] = -1;

                    int newc = -1;
                    uint64_t x = seed;
                    for (int tries = 0; tries < std::min(12, k); ++tries) {
                        int c = (int)(splitmix64(x) % (uint64_t)k);
                        if (c == old) continue;
                        if (can_use(u, c)) {
                            newc = c;
                            break;
                        }
                    }
                    if (newc < 0) {
                        for (int c = 0; c < k; ++c) {
                            if (c == old) continue;
                            if (can_use(u, c)) {
                                newc = c;
                                break;
                            }
                        }
                    }
                    if (newc < 0) {
                        ok = false;
                        color[u] = old;
                        break;
                    }
                    moves.push_back({u, newc});
                    color[u] = old;
                }
                if (!ok) continue;

                for (auto &mv : moves) color[mv.first] = mv.second;
                if (can_use(v, target)) {
                    color[v] = target;
                    seed ^= 0x9e3779b97f4a7c15ull;
                    return true;
                }
            }
            return false;
        }

        std::fill(cnt_buf.begin(), cnt_buf.end(), 0);
        for (int i = b; i < e; ++i) {
            int u = g->adj[i];
            int cu = color[u];
            if (cu >= 0 && cnt_buf[cu]++ == 0) present_buf.push_back(cu);
        }

        if ((int)present_buf.size() < k) {
            for (int c = 0; c < k; ++c) {
                if (cnt_buf[c] == 0) {
                    color[v] = c;
                    return true;
                }
            }
        }

        std::sort(present_buf.begin(), present_buf.end(), [&](int a, int b2) {
            if (cnt_buf[a] != cnt_buf[b2]) return cnt_buf[a] < cnt_buf[b2];
            return a < b2;
        });

        int attempts = std::min<int>((int)present_buf.size(), 6);
        for (int t = 0; t < attempts; ++t) {
            int target = present_buf[t];
            // Gather neighbors of v with this target color.
            tmp.clear();
            for (int i = b; i < e; ++i) {
                int u = g->adj[i];
                if (color[u] == target) tmp.push_back(u);
            }

            // For each such neighbor u, attempt to move u to some alternative color not used by its neighbors (excluding v).
            // Do this greedily; if all can move, free 'target'.
            std::vector<std::pair<int,int>> moves;
            moves.reserve(tmp.size());
            bool ok = true;
            for (int u : tmp) {
                int old = color[u];
                // Temporarily uncolor u to evaluate.
                color[u] = -1;

                int newc = -1;
                // Randomized color probe
                uint64_t x = seed;
                for (int tries = 0; tries < std::min(12, k); ++tries) {
                    int c = (int)(splitmix64(x) % (uint64_t)k);
                    if (c == old) continue;
                    if (can_use(u, c)) { newc = c; break; }
                }
                if (newc < 0) {
                    for (int c = 0; c < k; ++c) {
                        if (c == old) continue;
                        if (can_use(u, c)) { newc = c; break; }
                    }
                }
                if (newc < 0) {
                    ok = false;
                    color[u] = old;
                    break;
                }
                moves.push_back({u, newc});
                color[u] = old;
            }
            if (!ok) continue;

            // Apply moves.
            for (auto &mv : moves) color[mv.first] = mv.second;

            // Now 'target' should be free for v.
            if (can_use(v, target)) {
                color[v] = target;
                seed ^= 0x9e3779b97f4a7c15ull;
                return true;
            }

            // If something went wrong, revert by trying again (but keep moves; next attempts may still succeed).
        }

        return false;
    }

    bool local_repair(std::chrono::steady_clock::time_point deadline) {
        uint64_t seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
        seed ^= (uint64_t)g->n * 0x9e3779b97f4a7c15ull;

        // Try to color conflicts; if still stuck, uncolor a small neighborhood and retry.
        int rounds = 0;
        while (!conflicts.empty() && std::chrono::steady_clock::now() < deadline) {
            ++rounds;
            // Process conflicts in a pseudo-random order.
            for (int i = (int)conflicts.size() - 1; i > 0; --i) {
                int j = (int)(splitmix64(seed) % (uint64_t)(i + 1));
                std::swap(conflicts[i], conflicts[j]);
            }

            std::vector<int> next;
            next.reserve(conflicts.size());

            for (int v : conflicts) {
                if (color[v] >= 0) continue;

                // First: try direct assignment.
                int c = -1;
                if (k <= 64) c = first_available_color64(v);
                else {
                    std::vector<uint64_t> bits((k + 63) >> 6);
                    c = first_available_color_general(v, bits);
                }
                if (c >= 0) {
                    color[v] = c;
                    continue;
                }

                // Second: try Kempe-style repair.
                if (kempe_repair_vertex(v, seed)) continue;

                // Third: perturb neighborhood (uncolor a few neighbors with high degree to reshuffle).
                int b = g->off[v], e = g->off[v + 1];
                int take = 0;
                for (int it = b; it < e && take < 6; ++it) {
                    int u = g->adj[it];
                    if (color[u] >= 0) {
                        color[u] = -1;
                        next.push_back(u);
                        ++take;
                    }
                }
                next.push_back(v);
            }

            // Re-run greedy on the subset by giving them priority in the order.
            // For performance, just attempt to recolor uncolored vertices by scanning order.
            conflicts.clear();
            if (k <= 64) {
                for (int v : order) {
                    if (color[v] >= 0) continue;
                    int c = first_available_color64(v);
                    if (c < 0) conflicts.push_back(v);
                    else color[v] = c;
                }
            } else {
                std::vector<uint64_t> bits((k + 63) >> 6);
                for (int v : order) {
                    if (color[v] >= 0) continue;
                    int c = first_available_color_general(v, bits);
                    if (c < 0) conflicts.push_back(v);
                    else color[v] = c;
                }
            }

            // Prevent infinite loops on pathological cases.
            if (rounds > 40) break;
        }

        return conflicts.empty();
    }

    bool verify() const {
        for (int v = 0; v < g->n; ++v) {
            int cv = color[v];
            if (cv < 0 || cv >= k) return false;
            int b = g->off[v], e = g->off[v + 1];
            for (int i = b; i < e; ++i) {
                int u = g->adj[i];
                if (u > v && color[u] == cv) return false;
            }
        }
        return true;
    }

    bool solve(std::chrono::steady_clock::time_point deadline) {
        // Trivial cases
        if (g->n == 0) return true;
        if (k <= 0) return false;
        if (k == 1) {
            // Only valid if no edges.
            if (g->off[g->n] != 0) return false;
            std::fill(color.begin(), color.end(), 0);
            return true;
        }

        uint64_t base_seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
        base_seed ^= (uint64_t)g->off[g->n] * 0x9e3779b97f4a7c15ull;

        // Adaptive: for smaller graphs, do multiple restarts with different orders.
        int restarts;
        if (g->n <= 50000) restarts = 18;
        else if (g->n <= 250000) restarts = 8;
        else restarts = 3;

        for (int r = 0; r < restarts && std::chrono::steady_clock::now() < deadline; ++r) {
            clear_colors();
            uint64_t seed = base_seed + (uint64_t)r * 0x9e3779b97f4a7c15ull;

            if (g->n <= 80000) {
                // Mix shuffled-degree and pure-degree.
                if ((r % 3) == 0) build_order_degree(false, seed);
                else if ((r % 3) == 1) build_order_degree(true, seed);
                else build_order_shuffle(seed);
            } else {
                // Large: degree order once, then a little randomization.
                if (r == 0) build_order_degree(false, seed);
                else build_order_degree(true, seed);
            }

            bool ok = greedy_color();
            if (!ok) {
                ok = local_repair(deadline);
            }
            if (ok && verify()) return true;
        }

        return false;
    }
};

} // namespace gc

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    gc::FastInput in;
    int k = 0;
    gc::Graph g = gc::read_graph(in, k);

    auto start = std::chrono::steady_clock::now();
    auto deadline = start + std::chrono::minutes(5) - std::chrono::seconds(2);

    gc::Solver solver(g, k);
    bool ok = solver.solve(deadline);

    if (!ok) {
        // As a last resort, output something (judge will fail, but avoid crashing).
        for (int i = 0; i < g.n; ++i) {
            if (i) std::cout << ' ';
            std::cout << (i % std::max(1, k));
        }
        std::cout << '\n';
        return 0;
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) std::cout << ' ';
        std::cout << solver.color[i];
    }
    std::cout << '\n';
    return 0;
}
