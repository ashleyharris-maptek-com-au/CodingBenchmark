// C++17 exact solver for:
//   maximize sum(c_i * x_i)
//   s.t. x_u + x_v <= 1 for each (u,v), x_i in {0,1}
// This is a maximum weight independent set problem on the conflict graph.
// We solve it exactly via branch-and-bound maximum weight clique on the complement graph,
// using a greedy coloring upper bound (sum of max weights per color class).

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using std::int64_t;

#if defined(_MSC_VER)
#include <intrin.h>
#endif

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
  return (int)__popcnt64(x);
#else
  return __builtin_popcountll(x);
#endif
}

struct BitGraph {
  int n = 0;
  int words = 0;
  std::vector<std::vector<uint64_t>> adj;  // adj[v][w] bitset

  explicit BitGraph(int n_) : n(n_), words((n_ + 63) / 64), adj(n_, std::vector<uint64_t>(words, 0)) {}

  inline void set_edge(int u, int v) {
    adj[u][v >> 6] |= (uint64_t(1) << (v & 63));
  }
  inline bool has_edge(int u, int v) const {
    return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
  }
};

struct MaxWeightClique {
  const BitGraph& g;
  const std::vector<int64_t>& w;
  std::vector<int> best_clique;
  int64_t best_weight = 0;

  std::vector<int> cur_clique;

  explicit MaxWeightClique(const BitGraph& graph, const std::vector<int64_t>& weights)
      : g(graph), w(weights) {}

  void greedy_initial(const std::vector<int>& vertices) {
    std::vector<int> order = vertices;
    std::sort(order.begin(), order.end(), [&](int a, int b) {
      if (w[a] != w[b]) return w[a] > w[b];
      // tie-break by degree
      int da = 0, db = 0;
      for (int k = 0; k < g.words; ++k) {
        da += popcount64(g.adj[a][k]);
        db += popcount64(g.adj[b][k]);
      }
      return da > db;
    });
    std::vector<int> clique;
    int64_t total = 0;
    for (int v : order) {
      bool ok = true;
      for (int u : clique) {
        if (!g.has_edge(v, u)) {
          ok = false;
          break;
        }
      }
      if (ok) {
        clique.push_back(v);
        total += w[v];
      }
    }
    if (total > best_weight) {
      best_weight = total;
      best_clique = clique;
    }
  }

  struct ColorResult {
    std::vector<int> order;       // vertices in non-decreasing color order (highest color last)
    std::vector<int64_t> bound;   // bound[i] = sum of color-class maxima up to color(order[i])
  };

  ColorResult color_sort(const std::vector<int>& vertices) const {
    std::vector<int> vtx = vertices;
    // Heuristic: order by weight desc then degree desc for better coloring.
    std::sort(vtx.begin(), vtx.end(), [&](int a, int b) {
      if (w[a] != w[b]) return w[a] > w[b];
      int da = 0, db = 0;
      for (int k = 0; k < g.words; ++k) {
        da += popcount64(g.adj[a][k]);
        db += popcount64(g.adj[b][k]);
      }
      return da > db;
    });

    std::vector<std::vector<uint64_t>> color_bits;
    std::vector<int64_t> color_max;
    std::vector<std::vector<int>> color_vertices;

    std::vector<int> color_of(vtx.size(), -1);
    for (size_t i = 0; i < vtx.size(); ++i) {
      int v = vtx[i];
      int chosen = -1;
      for (int c = 0; c < (int)color_bits.size(); ++c) {
        bool conflict = false;
        for (int k = 0; k < g.words; ++k) {
          if (g.adj[v][k] & color_bits[c][k]) {
            conflict = true;
            break;
          }
        }
        if (!conflict) {
          chosen = c;
          break;
        }
      }
      if (chosen == -1) {
        chosen = (int)color_bits.size();
        color_bits.push_back(std::vector<uint64_t>(g.words, 0));
        color_max.push_back(0);
        color_vertices.push_back({});
      }
      color_of[i] = chosen;
      color_bits[chosen][v >> 6] |= (uint64_t(1) << (v & 63));
      color_max[chosen] = std::max<int64_t>(color_max[chosen], w[v]);
      color_vertices[chosen].push_back(v);
    }

    // Build order grouped by increasing color.
    ColorResult res;
    res.order.reserve(vtx.size());
    std::vector<int64_t> prefix(color_bits.size() + 1, 0);
    for (size_t c = 0; c < color_bits.size(); ++c) prefix[c + 1] = prefix[c] + color_max[c];

    res.bound.reserve(vtx.size());
    for (size_t c = 0; c < color_vertices.size(); ++c) {
      for (int v : color_vertices[c]) {
        res.order.push_back(v);
        res.bound.push_back(prefix[c + 1]);
      }
    }
    return res;
  }

  void expand(const std::vector<int>& candidates, int64_t cur_weight) {
    if (candidates.empty()) {
      if (cur_weight > best_weight) {
        best_weight = cur_weight;
        best_clique = cur_clique;
      }
      return;
    }

    auto colored = color_sort(candidates);
    const auto& order = colored.order;
    const auto& bound = colored.bound;

    for (int i = (int)order.size() - 1; i >= 0; --i) {
      if (cur_weight + bound[i] <= best_weight) return;

      int v = order[i];
      cur_clique.push_back(v);
      int64_t new_weight = cur_weight + w[v];
      if (new_weight > best_weight) {
        best_weight = new_weight;
        best_clique = cur_clique;
      }

      std::vector<int> new_candidates;
      new_candidates.reserve(i);
      for (int j = 0; j < i; ++j) {
        int u = order[j];
        if (g.has_edge(v, u)) new_candidates.push_back(u);
      }
      expand(new_candidates, new_weight);
      cur_clique.pop_back();
    }
  }

  std::vector<int> solve() {
    std::vector<int> vertices(g.n);
    std::iota(vertices.begin(), vertices.end(), 0);
    greedy_initial(vertices);
    expand(vertices, 0);
    return best_clique;
  }
};

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n, m;
  if (!(std::cin >> n >> m)) return 0;
  std::vector<int64_t> c(n);
  for (int i = 0; i < n; ++i) std::cin >> c[i];

  std::vector<std::pair<int, int>> edges;
  edges.reserve(m);
  std::vector<char> forbidden(n, 0);
  for (int i = 0; i < m; ++i) {
    int u, v;
    std::cin >> u >> v;
    edges.emplace_back(u, v);
    if (u == v && 0 <= u && u < n) forbidden[u] = 1;  // x_u + x_u <= 1 => x_u = 0
  }

  // Keep only strictly positive-weight, non-forbidden variables (others are always 0 in an optimum).
  std::vector<int> old_to_new(n, -1);
  std::vector<int> new_to_old;
  for (int i = 0; i < n; ++i) {
    if (!forbidden[i] && c[i] > 0) {
      old_to_new[i] = (int)new_to_old.size();
      new_to_old.push_back(i);
    }
  }

  int N = (int)new_to_old.size();
  std::vector<int> x(n, 0);
  if (N == 0) {
    std::cout << 0 << "\n";
    for (int i = 0; i < n; ++i) {
      if (i) std::cout << ' ';
      std::cout << 0;
    }
    std::cout << "\n";
    return 0;
  }

  // Build conflict graph among kept vertices.
  BitGraph conflict(N);
  for (auto [u0, v0] : edges) {
    if (u0 < 0 || u0 >= n || v0 < 0 || v0 >= n) continue;
    if (u0 == v0) continue;
    int u = old_to_new[u0];
    int v = old_to_new[v0];
    if (u < 0 || v < 0) continue;
    conflict.set_edge(u, v);
    conflict.set_edge(v, u);
  }

  // Build complement graph for clique search (edges mean "can be together").
  BitGraph clique(N);
  int words = clique.words;
  std::vector<uint64_t> all(words, ~uint64_t(0));
  if (N % 64) all[words - 1] = (uint64_t(1) << (N % 64)) - 1;

  for (int v = 0; v < N; ++v) {
    for (int k = 0; k < words; ++k) clique.adj[v][k] = all[k] & ~conflict.adj[v][k];
    // clear self-loop
    clique.adj[v][v >> 6] &= ~(uint64_t(1) << (v & 63));
  }

  std::vector<int64_t> w(N);
  for (int i = 0; i < N; ++i) w[i] = c[new_to_old[i]];

  MaxWeightClique mwc(clique, w);
  std::vector<int> best = mwc.solve();

  int64_t obj = 0;
  for (int v : best) {
    int orig = new_to_old[v];
    x[orig] = 1;
    obj += c[orig];
  }

  std::cout << obj << "\n";
  for (int i = 0; i < n; ++i) {
    if (i) std::cout << ' ';
    std::cout << x[i];
  }
  std::cout << "\n";
  return 0;
}
