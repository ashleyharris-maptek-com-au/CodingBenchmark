using System;
using System.Buffers;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private readonly struct Voxel
    {
        public readonly byte Raw;
        public readonly int Mass;   // 0..15
        public readonly int Value;  // 0..15
        public Voxel(byte raw)
        {
            Raw = raw;
            Mass = raw & 0x0F;
            Value = (raw >> 4) & 0x0F;
        }
        public bool IsAir => Mass == 0;
        public bool IsValuable => Mass > 0 && Value > 0;
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _pos;
        public FastScanner(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }
        private int ReadByteInternal()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }
        public string NextToken()
        {
            int c;
            do
            {
                c = ReadByteInternal();
                if (c < 0) return "";
            } while (c <= 32);
            var sb = new System.Text.StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByteInternal();
                if (c < 0) break;
            }
            return sb.ToString();
        }
        public int NextInt()
        {
            var t = NextToken();
            if (t.Length == 0) return 0;
            return int.Parse(t, CultureInfo.InvariantCulture);
        }
        public double NextDouble()
        {
            var t = NextToken();
            if (t.Length == 0) return 0;
            return double.Parse(t, CultureInfo.InvariantCulture);
        }
        public bool TryReadVoxelByte(out byte b)
        {
            int c = ReadByteInternal();
            if (c < 0) { b = 0; return false; }
            b = (byte)c;
            return true;
        }
    }

    private sealed class Op
    {
        public int DigX, DigY, DigZ;
        public List<(int x, int y)> Steps = new List<(int x, int y)>(64); // absolute coords per step
        public int DestX, DestY;
    }

    public static void Main()
    {
        Console.OutputEncoding = System.Text.Encoding.ASCII;
        var fs = new FastScanner(Console.OpenStandardInput());

        int xCount = fs.NextInt();
        int yCount = fs.NextInt();
        int zCount = fs.NextInt();
        _ = fs.NextDouble(); // block sizes not needed for planning (all costs defined in voxel units)
        _ = fs.NextDouble();
        _ = fs.NextDouble();

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        // Adaptive strategy:
        // - For large worlds: do nothing (guaranteed valid, avoids O(X*Y*Z) memory/time).
        // - For smaller worlds: load full grid and mine top-surface valuable ore reachable to plant.
        //
        // Threshold tuned to keep memory reasonable (byte grid + some arrays).
        long total = (long)xCount * yCount * zCount;
        const long LoadThreshold = 25_000_000; // 25 MB grid

        if (total > LoadThreshold)
        {
            // Consume input quickly (required in many judge harnesses) without storing.
            // We still must read the voxel bytes to completion.
            byte tmp;
            for (long i = 0; i < total; i++)
                if (!fs.TryReadVoxelByte(out tmp)) break;
            Console.WriteLine("0");
            return;
        }

        // Load grid as [x][y][z] flattened: idx = ((x*yCount)+y)*zCount + z
        int xyCount = checked(xCount * yCount);
        int zC = zCount;
        int totalInt = checked((int)total);
        byte[] grid = new byte[totalInt];

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int baseIdx = ((x * yCount) + y) * zC;
                for (int z = 0; z < zC; z++)
                {
                    if (!fs.TryReadVoxelByte(out grid[baseIdx + z]))
                    {
                        // Incomplete input; output nothing.
                        Console.WriteLine("0");
                        return;
                    }
                }
            }
        }

        // Compute surface height and top voxel per column.
        // Height = highest z with mass>0, else -1.
        int[] height = new int[xyCount];
        byte[] topRaw = new byte[xyCount];
        for (int i = 0; i < xyCount; i++)
        {
            height[i] = -1;
            topRaw[i] = 0;
        }

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int col = x * yCount + y;
                int baseIdx = col * zC;
                int h = -1;
                byte tr = 0;
                for (int z = zC - 1; z >= 0; z--)
                {
                    byte raw = grid[baseIdx + z];
                    if ((raw & 0x0F) != 0)
                    {
                        h = z;
                        tr = raw;
                        break;
                    }
                }
                height[col] = h;
                topRaw[col] = tr;
            }
        }

        // Build a quick reachability map from plant (0,0) over surface with slope constraint.
        // Only moves in X/Y; allowed if abs(height[next]-height[cur]) <= 1.
        // Treat missing columns (height==-1) as impassable.
        bool[] reachable = new bool[xyCount];
        int[] prev = new int[xyCount];
        for (int i = 0; i < xyCount; i++) prev[i] = -2; // -2 means unvisited

        int plant = 0; // (0,0)
        if (height[plant] < 0)
        {
            // No ground at plant; cannot dump safely under our surface-path assumption.
            Console.WriteLine("0");
            return;
        }

        var q = new int[xyCount];
        int qh = 0, qt = 0;
        q[qt++] = plant;
        prev[plant] = -1;
        reachable[plant] = true;

        while (qh < qt)
        {
            int cur = q[qh++];
            int cx = cur / yCount;
            int cy = cur - cx * yCount;
            int ch = height[cur];
            if (ch < 0) continue;

            void TryEdge(int nx, int ny)
            {
                if ((uint)nx >= (uint)xCount || (uint)ny >= (uint)yCount) return;
                int ni = nx * yCount + ny;
                if (prev[ni] != -2) return;
                int nh = height[ni];
                if (nh < 0) return;
                int dh = nh - ch;
                if (dh > 1 || dh < -1) return;
                prev[ni] = cur;
                reachable[ni] = true;
                q[qt++] = ni;
            }

            TryEdge(cx - 1, cy);
            TryEdge(cx + 1, cy);
            TryEdge(cx, cy - 1);
            TryEdge(cx, cy + 1);
        }

        // Collect candidate top-surface valuable ore that is reachable.
        // We will mine a limited number to keep output manageable and runtime bounded.
        // Score ~ net profit lower bound: revenue - flat-move-cost.
        var candidates = new List<(long score, int col, int x, int y)>();
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int col = x * yCount + y;
                if (!reachable[col]) continue;
                int h = height[col];
                if (h < 0) continue;
                var v = new Voxel(topRaw[col]);
                if (!v.IsValuable) continue;

                // Path length lower bound is Manhattan distance.
                int dist = x + y;
                long revenue = (long)v.Mass * v.Value * 1000L;
                long cost = (long)v.Mass * dist; // optimistic (downhill free, uphill may be higher)
                long score = revenue - cost;
                if (score <= 0) continue;
                candidates.Add((score, col, x, y));
            }
        }

        candidates.Sort((a, b) => b.score.CompareTo(a.score));

        // Limit operations: avoid huge output.
        int maxOps = Math.Min(5000, candidates.Count);
        var ops = new List<Op>(maxOps);

        // Build path per candidate by backtracking prev[] (which is shortest path in steps under slope constraint)
        // from column to plant. We output steps as a sequence of absolute (x,y) coords.
        for (int i = 0; i < maxOps; i++)
        {
            var c = candidates[i];
            int col = c.col;
            int h = height[col];
            if (h < 0) continue;
            var v = new Voxel(topRaw[col]);
            if (!v.IsValuable) continue;

            // Reconstruct path: col -> ... -> plant
            // prev[] contains a BFS tree rooted at plant; to get path from col to plant, follow prev back to plant.
            // That yields col->...->plant. For steps, we want successive positions along the path.
            if (prev[col] == -2) continue;

            var pathCols = new List<int>(64);
            int t = col;
            while (t != -1)
            {
                pathCols.Add(t);
                t = prev[t];
            }
            // pathCols currently ends at -1 after adding plant; reverse to plant->...->col,
            // then we will walk from dig site to plant, so use col->...->plant.
            pathCols.Reverse(); // plant -> ... -> col
            // Convert to col->...->plant
            pathCols.Reverse(); // col -> ... -> plant

            var op = new Op
            {
                DigX = c.x,
                DigY = c.y,
                DigZ = h,
                DestX = 0,
                DestY = 0
            };

            // Steps should list intermediate points. We will include every move as (x,y) absolute.
            // Exclude the starting cell (dig location) to avoid a zero-length step.
            for (int pi = 1; pi < pathCols.Count; pi++)
            {
                int pc = pathCols[pi];
                int px = pc / yCount;
                int py = pc - px * yCount;
                op.Steps.Add((px, py));
            }

            // If no movement (dig at plant), it's still valid: N=0 steps; dump at plant.
            ops.Add(op);

            // Update world model: digging removes top voxel in that column.
            // We only remove the single top voxel; if there is more material below, surface lowers.
            int baseIdx = col * zC;
            grid[baseIdx + h] = 0;
            // Recompute new height/topRaw for this column (scan downward from h-1).
            int nh = -1;
            byte nr = 0;
            for (int z = h - 1; z >= 0; z--)
            {
                byte raw = grid[baseIdx + z];
                if ((raw & 0x0F) != 0)
                {
                    nh = z;
                    nr = raw;
                    break;
                }
            }
            height[col] = nh;
            topRaw[col] = nr;
            // Note: we do not update reachability or BFS tree after digs; this is a fast heuristic.
        }

        // Emit output.
        // Format:
        // M
        // For each op: N digX digY digZ stepX1 stepY1 ... destX destY
        // Where N is number of steps (each step is absolute coordinate of truck position after that step).
        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };
        sw.WriteLine(ops.Count);
        for (int i = 0; i < ops.Count; i++)
        {
            var op = ops[i];
            sw.Write(op.Steps.Count);
            sw.Write(' ');
            sw.Write(op.DigX);
            sw.Write(' ');
            sw.Write(op.DigY);
            sw.Write(' ');
            sw.Write(op.DigZ);
            for (int s = 0; s < op.Steps.Count; s++)
            {
                sw.Write(' ');
                sw.Write(op.Steps[s].x);
                sw.Write(' ');
                sw.Write(op.Steps[s].y);
            }
            sw.Write(' ');
            sw.Write(op.DestX);
            sw.Write(' ');
            sw.Write(op.DestY);
            sw.WriteLine();
        }
    }
}
