using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    // Simple kinetic-impact trajectory solver.
    // Implements a basic N-body KDK integrator for the provided bodies and
    // uses a finite-difference sensitivity solve to pick a single launch burn
    // that targets the asteroid at a chosen intercept time.

    private const double G = 6.67430e-11;
    private const double Day = 86400.0;

    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R;
        public Vec3 V;
    }

    private readonly struct Vec3
    {
        public readonly double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm() => Math.Sqrt(Dot(this));
        public Vec3 Normalized()
        {
            var n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    private sealed class Ephemeris
    {
        public readonly double DtDays;
        public readonly int Steps;
        public readonly Vec3[] EarthR;
        public readonly Vec3[] EarthV;
        public readonly Vec3[] AstR;
        public readonly Vec3[] AstV;

        public Ephemeris(double dtDays, int steps)
        {
            DtDays = dtDays;
            Steps = steps;
            EarthR = new Vec3[steps + 1];
            EarthV = new Vec3[steps + 1];
            AstR = new Vec3[steps + 1];
            AstV = new Vec3[steps + 1];
        }

        public void Sample(double tDays, out Vec3 earthR, out Vec3 earthV, out Vec3 astR, out Vec3 astV)
        {
            if (tDays <= 0)
            {
                earthR = EarthR[0]; earthV = EarthV[0];
                astR = AstR[0]; astV = AstV[0];
                return;
            }
            var maxT = Steps * DtDays;
            if (tDays >= maxT)
            {
                earthR = EarthR[Steps]; earthV = EarthV[Steps];
                astR = AstR[Steps]; astV = AstV[Steps];
                return;
            }
            var f = tDays / DtDays;
            int i = (int)Math.Floor(f);
            double a = f - i;
            earthR = Lerp(EarthR[i], EarthR[i + 1], a);
            earthV = Lerp(EarthV[i], EarthV[i + 1], a);
            astR = Lerp(AstR[i], AstR[i + 1], a);
            astV = Lerp(AstV[i], AstV[i + 1], a);
        }

        private static Vec3 Lerp(Vec3 a, Vec3 b, double t) => a * (1.0 - t) + b * t;
    }

    private static Vec3[] ComputeAccelerations(List<Body> bodies, Vec3[] r)
    {
        int n = bodies.Count;
        var a = new Vec3[n];
        for (int i = 0; i < n; i++) a[i] = new Vec3(0, 0, 0);

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                var rij = r[j] - r[i];
                double d2 = rij.Dot(rij) + 1e6; // softening (1 km)^2 to avoid singularity
                double invD = 1.0 / Math.Sqrt(d2);
                double invD3 = invD * invD * invD;
                double s = G * invD3;
                var aij = rij * s;
                a[i] = a[i] + aij * bodies[j].Mass;
                a[j] = a[j] - aij * bodies[i].Mass;
            }
        }
        return a;
    }

    private static Ephemeris BuildEphemeris(List<Body> initialBodies, int earthIdx, int astIdx, double maxDays, double dtDays)
    {
        int steps = (int)Math.Ceiling(maxDays / dtDays);
        var eph = new Ephemeris(dtDays, steps);

        int n = initialBodies.Count;
        var r = new Vec3[n];
        var v = new Vec3[n];
        for (int i = 0; i < n; i++) { r[i] = initialBodies[i].R; v[i] = initialBodies[i].V; }

        eph.EarthR[0] = r[earthIdx]; eph.EarthV[0] = v[earthIdx];
        eph.AstR[0] = r[astIdx]; eph.AstV[0] = v[astIdx];

        double dt = dtDays * Day;
        for (int k = 0; k < steps; k++)
        {
            var a0 = ComputeAccelerations(initialBodies, r);
            for (int i = 0; i < n; i++) v[i] = v[i] + a0[i] * (0.5 * dt);
            for (int i = 0; i < n; i++) r[i] = r[i] + v[i] * dt;
            var a1 = ComputeAccelerations(initialBodies, r);
            for (int i = 0; i < n; i++) v[i] = v[i] + a1[i] * (0.5 * dt);

            eph.EarthR[k + 1] = r[earthIdx]; eph.EarthV[k + 1] = v[earthIdx];
            eph.AstR[k + 1] = r[astIdx]; eph.AstV[k + 1] = v[astIdx];
        }

        return eph;
    }

    private static Vec3 PropagateSpacecraftTo(List<Body> bodies, Ephemeris eph, int earthIdx, double tLaunchDays, Vec3 dvLaunch, double tTargetDays, double dtDays, out Vec3 scVEnd)
    {
        eph.Sample(tLaunchDays, out var earthR0, out var earthV0, out _, out _);
        var scR = earthR0;
        var scV = earthV0 + dvLaunch;

        // Integrate spacecraft under gravity of (fixed) propagated bodies using KDK with body interpolation.
        double dt = dtDays * Day;
        int steps = (int)Math.Ceiling((tTargetDays - tLaunchDays) / dtDays);
        double t = tLaunchDays;
        for (int k = 0; k < steps; k++)
        {
            double tNext = Math.Min(tTargetDays, t + dtDays);
            double hDays = tNext - t;
            double h = hDays * Day;

            // Kick
            var a0 = AccelFromBodies(bodies, eph, t, scR, earthIdx);
            scV = scV + a0 * (0.5 * h);
            // Drift
            scR = scR + scV * h;
            // Kick
            var a1 = AccelFromBodies(bodies, eph, tNext, scR, earthIdx);
            scV = scV + a1 * (0.5 * h);
            t = tNext;
        }
        scVEnd = scV;
        return scR;
    }

    private static Vec3 AccelFromBodies(List<Body> bodies, Ephemeris eph, double tDays, Vec3 scR, int earthIdx)
    {
        // Use ephemeris positions for all bodies by integrating their stored states?
        // We only stored Earth+Asteroid; for other bodies use their initial Keplerian drift (fallback):
        // In practice for benchmark scenarios this still provides a reasonable baseline; dominant body is Sun.
        // If only Sun/Earth/Asteroid are present, this is exact w.r.t our ephemeris storage.

        // Build positions for each body at time tDays.
        // Earth/Asteroid from eph; others: linear extrapolation from t=0.
        eph.Sample(tDays, out var earthR, out _, out var astR, out _);

        var a = new Vec3(0, 0, 0);
        for (int i = 0; i < bodies.Count; i++)
        {
            Vec3 rBody;
            if (i == earthIdx) rBody = earthR;
            else if (bodies[i].Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase)) rBody = astR;
            else rBody = bodies[i].R + bodies[i].V * (tDays * Day);

            var dr = rBody - scR;
            double d2 = dr.Dot(dr) + 1e6;
            double invD = 1.0 / Math.Sqrt(d2);
            double invD3 = invD * invD * invD;
            a = a + dr * (G * bodies[i].Mass * invD3);
        }
        return a;
    }

    private static bool Solve3x3(double[,] A, double[] b, out double[] x)
    {
        // Gaussian elimination with partial pivot.
        x = new double[3];
        var M = new double[3, 4];
        for (int i = 0; i < 3; i++)
        {
            for (int j = 0; j < 3; j++) M[i, j] = A[i, j];
            M[i, 3] = b[i];
        }

        for (int col = 0; col < 3; col++)
        {
            int piv = col;
            double best = Math.Abs(M[col, col]);
            for (int r = col + 1; r < 3; r++)
            {
                double v = Math.Abs(M[r, col]);
                if (v > best) { best = v; piv = r; }
            }
            if (best < 1e-18) return false;
            if (piv != col)
            {
                for (int c = col; c < 4; c++)
                {
                    var tmp = M[col, c];
                    M[col, c] = M[piv, c];
                    M[piv, c] = tmp;
                }
            }

            double diag = M[col, col];
            for (int c = col; c < 4; c++) M[col, c] /= diag;
            for (int r = 0; r < 3; r++)
            {
                if (r == col) continue;
                double f = M[r, col];
                for (int c = col; c < 4; c++) M[r, c] -= f * M[col, c];
            }
        }

        x[0] = M[0, 3];
        x[1] = M[1, 3];
        x[2] = M[2, 3];
        return true;
    }

    public static void Main()
    {
        var tokens = ReadAllTokens(Console.In);
        int p = 0;

        if (tokens.Count < 7)
        {
            // Output a trivial plan if input is malformed.
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        int numBodies = ParseInt(tokens[p++]);
        double warningDays = ParseDouble(tokens[p++]);
        double impactorMass = ParseDouble(tokens[p++]);
        double dvBudget = ParseDouble(tokens[p++]);
        double asteroidMass = ParseDouble(tokens[p++]); // may duplicate body mass; keep for momentum reasoning if needed
        _ = ParseDouble(tokens[p++]); // t0_unix_utc (unused for dynamics)

        var bodies = new List<Body>(numBodies);
        int earthIdx = -1, astIdx = -1;
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[p++];
            b.Mass = ParseDouble(tokens[p++]);
            b.R = new Vec3(ParseDouble(tokens[p++]), ParseDouble(tokens[p++]), ParseDouble(tokens[p++]));
            b.V = new Vec3(ParseDouble(tokens[p++]), ParseDouble(tokens[p++]), ParseDouble(tokens[p++]));
            if (b.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase)) earthIdx = i;
            if (b.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase)) astIdx = i;
            bodies.Add(b);
        }

        // de421 path is provided but not required for this baseline solver.
        if (p < tokens.Count) _ = tokens[p++]; // bsp path token

        if (earthIdx < 0 || astIdx < 0)
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        // Build ephemeris for Earth+Asteroid over the warning period.
        double bodyDtDays = warningDays <= 800 ? 0.25 : 0.5; // coarser for very long runs
        var eph = BuildEphemeris(bodies, earthIdx, astIdx, warningDays, bodyDtDays);

        // Candidate intercept time: early enough to be before impact, but late enough for transfer.
        double tImpactDays = warningDays; // common interpretation: warning time until impact
        double tInterceptDays = Math.Max(1.0, Math.Min(tImpactDays - 5.0, 0.7 * tImpactDays));
        if (tInterceptDays <= 0) tInterceptDays = 0.8 * tImpactDays;

        // Search over a few launch times; pick the one yielding the smallest miss distance under dvBudget.
        double bestLaunch = 0;
        Vec3 bestDv = new Vec3(0, 0, 0);
        double bestMiss = double.PositiveInfinity;

        // If the budget is extremely small, don't waste time on a big grid.
        int launchSamples = warningDays <= 50 ? 21 : (warningDays <= 500 ? 31 : 41);
        if (dvBudget < 20) launchSamples = Math.Min(launchSamples, 15);

        for (int s = 0; s < launchSamples; s++)
        {
            double tLaunch = (launchSamples == 1) ? 0 : (warningDays * s / (launchSamples - 1));
            if (tLaunch >= tInterceptDays - 0.5) continue;

            // Finite difference sensitivity of final position vs dv.
            double scDtDays = warningDays <= 800 ? 0.1 : 0.25;

            var r0 = PropagateSpacecraftTo(bodies, eph, earthIdx, tLaunch, new Vec3(0, 0, 0), tInterceptDays, scDtDays, out _);
            eph.Sample(tInterceptDays, out _, out _, out var astR, out _);
            var err = r0 - astR;
            double miss0 = err.Norm();

            // If already close without dv (rare), accept.
            if (miss0 < bestMiss)
            {
                bestMiss = miss0;
                bestLaunch = tLaunch;
                bestDv = new Vec3(0, 0, 0);
            }

            double eps = Math.Max(0.1, Math.Min(10.0, dvBudget * 0.02 + 0.1)); // m/s perturb
            var basis = new Vec3[] { new Vec3(eps, 0, 0), new Vec3(0, eps, 0), new Vec3(0, 0, eps) };
            var J = new double[3, 3];

            for (int j = 0; j < 3; j++)
            {
                var r1 = PropagateSpacecraftTo(bodies, eph, earthIdx, tLaunch, basis[j], tInterceptDays, scDtDays, out _);
                var de = (r1 - astR) - err;
                // Column j = d(err)/d(dv_j)
                J[0, j] = de.X / eps;
                J[1, j] = de.Y / eps;
                J[2, j] = de.Z / eps;
            }

            var bvec = new double[] { -err.X, -err.Y, -err.Z };
            if (!Solve3x3(J, bvec, out var sol)) continue;

            var dv = new Vec3(sol[0], sol[1], sol[2]);
            double dvMag = dv.Norm();
            if (double.IsNaN(dvMag) || double.IsInfinity(dvMag)) continue;

            if (dvMag > dvBudget && dvMag > 0)
                dv = dv * (dvBudget / dvMag);

            var rEnd = PropagateSpacecraftTo(bodies, eph, earthIdx, tLaunch, dv, tInterceptDays, scDtDays, out _);
            var miss = (rEnd - astR).Norm();

            if (miss < bestMiss)
            {
                bestMiss = miss;
                bestLaunch = tLaunch;
                bestDv = dv;
            }
        }

        // If the found miss is still huge, bias towards just matching asteroid direction from Earth.
        if (double.IsInfinity(bestMiss) || bestMiss > 5e7) // 50,000 km
        {
            double tLaunch = 0;
            double scDtDays = warningDays <= 800 ? 0.1 : 0.25;
            eph.Sample(tLaunch, out var earthR0, out var earthV0, out var astR0, out var astV0);
            var relR = astR0 - earthR0;
            var relV = astV0 - earthV0;
            var dir = (relR + relV * (Math.Max(1.0, tInterceptDays - tLaunch) * Day)).Normalized();
            bestLaunch = tLaunch;
            bestDv = dir * Math.Min(dvBudget, Math.Max(0.0, dvBudget));
        }

        // Output burns: single launch burn + impact entry (0 dv) at intercept time.
        // Times in days since scenario start.
        Console.WriteLine("2");
        Console.WriteLine(Form(bestLaunch) + " " + Form(bestDv.X) + " " + Form(bestDv.Y) + " " + Form(bestDv.Z));
        Console.WriteLine(Form(tInterceptDays) + " 0 0 0");
    }

    private static List<string> ReadAllTokens(TextReader tr)
    {
        var list = new List<string>(4096);
        string? line;
        while ((line = tr.ReadLine()) != null)
        {
            int i = 0;
            while (i < line.Length)
            {
                while (i < line.Length && char.IsWhiteSpace(line[i])) i++;
                if (i >= line.Length) break;
                int j = i;
                while (j < line.Length && !char.IsWhiteSpace(line[j])) j++;
                list.Add(line.Substring(i, j - i));
                i = j;
            }
        }
        return list;
    }

    private static double ParseDouble(string s) => double.Parse(s, NumberStyles.Float, CultureInfo.InvariantCulture);
    private static int ParseInt(string s) => int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
    private static string Form(double x) => x.ToString("R", CultureInfo.InvariantCulture);
}
