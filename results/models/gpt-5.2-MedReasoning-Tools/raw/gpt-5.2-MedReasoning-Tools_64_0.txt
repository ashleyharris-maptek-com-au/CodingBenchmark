// Grid drone path planner benchmark solution (TypeScript, no deps).
// Heuristic multi-agent greedy with local lookahead potential and light coordination.

declare const require: any;
declare const process: any;

type Drone = { id: string | number; x: number; y: number };

const fs = require("fs");
let input: string = fs.readFileSync(0, "utf8");
// Tolerate UTF-8 BOM and leading whitespace.
if (input.length > 0 && input.charCodeAt(0) === 0xfeff) input = input.slice(1);
input = input.trimStart();
const data = JSON.parse(input);

const prob: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone | 0;

const H = prob.length | 0;
const W = (H > 0 ? prob[0].length : 0) | 0;

function idx(x: number, y: number): number {
  return y * W + x;
}

const obstacle = new Uint8Array(W * H);
const value = new Float64Array(W * H);
for (let y = 0; y < H; y++) {
  const row = prob[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    const i = idx(x, y);
    if (v === -1) {
      obstacle[i] = 1;
      value[i] = 0;
    } else {
      obstacle[i] = 0;
      value[i] = v > 0 ? v : 0;
    }
  }
}

// Shared visited set for unique coverage scoring.
const visited = new Uint8Array(W * H);

// Simple RNG (deterministic from input size + drone count).
let seed = (((W * 73856093) ^ (H * 19349663) ^ (drones.length * 83492791)) >>> 0) || 1;
function randU32(): number {
  // xorshift32
  seed ^= seed << 13;
  seed >>>= 0;
  seed ^= seed >>> 17;
  seed >>>= 0;
  seed ^= seed << 5;
  seed >>>= 0;
  return seed;
}
function randFloat(): number {
  return (randU32() >>> 0) / 0xffffffff;
}

function clampInt(v: number, lo: number, hi: number): number {
  return v < lo ? lo : v > hi ? hi : v;
}

// Precompute manhattan-neighborhood offsets with weights.
const R = 4;
type Off = { dx: number; dy: number; w: number; d: number };
const offs: Off[] = [];
for (let dy = -R; dy <= R; dy++) {
  for (let dx = -R; dx <= R; dx++) {
    const d = Math.abs(dx) + Math.abs(dy);
    if (d === 0 || d > R) continue;
    // Discount with distance; slightly superlinear to focus closer cells.
    const w = 1 / (d * d + 1);
    offs.push({ dx, dy, w, d });
  }
}

// Movement options: stay, then 4-neighbors.
const moves = [
  { dx: 0, dy: 0 },
  { dx: 1, dy: 0 },
  { dx: -1, dy: 0 },
  { dx: 0, dy: 1 },
  { dx: 0, dy: -1 },
];

// Per-drone state.
const curX = new Int32Array(drones.length);
const curY = new Int32Array(drones.length);
const paths: number[][][] = new Array(drones.length);
// Short history to discourage tight loops.
const histLen = 8;
const hist: Int32Array[] = new Array(drones.length);
const histPos = new Int32Array(drones.length);

function isValidCell(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < W && y < H && obstacle[idx(x, y)] === 0;
}

function markVisited(x: number, y: number): void {
  const i = idx(x, y);
  if (value[i] > 0) visited[i] = 1;
}

for (let k = 0; k < drones.length; k++) {
  const sx = clampInt(drones[k].x | 0, 0, W - 1);
  const sy = clampInt(drones[k].y | 0, 0, H - 1);
  // If start is on obstacle, try to nudge to nearest valid by local spiral.
  let x = sx, y = sy;
  if (!isValidCell(x, y)) {
    let found = false;
    for (let r = 1; r <= 8 && !found; r++) {
      for (let t = 0; t < 20 && !found; t++) {
        const ang = randFloat() * Math.PI * 2;
        const nx = sx + Math.round(Math.cos(ang) * r);
        const ny = sy + Math.round(Math.sin(ang) * r);
        if (isValidCell(nx, ny)) {
          x = nx; y = ny; found = true;
        }
      }
    }
    if (!found) {
      // Fallback: scan a small window.
      for (let yy = Math.max(0, sy - 4); yy <= Math.min(H - 1, sy + 4) && !found; yy++) {
        for (let xx = Math.max(0, sx - 4); xx <= Math.min(W - 1, sx + 4) && !found; xx++) {
          if (isValidCell(xx, yy)) { x = xx; y = yy; found = true; }
        }
      }
    }
  }
  curX[k] = x;
  curY[k] = y;
  paths[k] = [[x, y]];
  hist[k] = new Int32Array(histLen);
  histPos[k] = 0;
  hist[k][0] = idx(x, y);
  markVisited(x, y);
}

function loopPenalty(droneIdx: number, cellIndex: number): number {
  const h = hist[droneIdx];
  // Penalize if in recent history; heavier for more recent.
  let p = 0;
  for (let t = 0; t < histLen; t++) {
    const hi = h[t];
    if (hi === 0 && t !== 0) break;
    if (hi === cellIndex) {
      // recent entries are near histPos
      const dist = (histPos[droneIdx] - t + histLen) % histLen;
      p += 0.3 + 0.1 * (histLen - dist);
    }
  }
  return p;
}

function nearbyPotential(x: number, y: number): number {
  let s = 0;
  for (let j = 0; j < offs.length; j++) {
    const o = offs[j];
    const nx = x + o.dx;
    const ny = y + o.dy;
    if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
    const ni = idx(nx, ny);
    if (obstacle[ni]) continue;
    const v = value[ni];
    if (v <= 0 || visited[ni]) continue;
    s += v * o.w;
  }
  return s;
}

function separationPenalty(x: number, y: number, self: number): number {
  // Encourage drones to spread out a bit (soft).
  let p = 0;
  for (let k = 0; k < drones.length; k++) {
    if (k === self) continue;
    const dx = Math.abs(curX[k] - x);
    const dy = Math.abs(curY[k] - y);
    const d = dx + dy;
    if (d === 0) p += 1.5;
    else if (d === 1) p += 0.7;
    else if (d === 2) p += 0.25;
  }
  return p;
}

// Interleaved planning: at each global tick, move drones one by one.
for (let step = 0; step < maxStepsPerDrone; step++) {
  // Slightly reshuffle drone order each tick for fairness.
  const order = new Int32Array(drones.length);
  for (let i = 0; i < drones.length; i++) order[i] = i;
  for (let i = drones.length - 1; i > 0; i--) {
    const j = randU32() % (i + 1);
    const tmp = order[i]; order[i] = order[j]; order[j] = tmp;
  }

  for (let oi = 0; oi < order.length; oi++) {
    const k = order[oi];
    const x0 = curX[k], y0 = curY[k];
    let bestX = x0, bestY = y0;
    let bestScore = -1e100;

    for (let mi = 0; mi < moves.length; mi++) {
      const nx = x0 + moves[mi].dx;
      const ny = y0 + moves[mi].dy;
      if (!isValidCell(nx, ny)) continue;
      const ni = idx(nx, ny);

      const immediate = (value[ni] > 0 && visited[ni] === 0) ? value[ni] : 0;
      // Local potential: sums surrounding unvisited high-value cells.
      const pot = nearbyPotential(nx, ny);
      const sep = separationPenalty(nx, ny, k);
      const loop = loopPenalty(k, ni);

      // Small preference to move (breaks long "stay" streaks when tied).
      const moveBonus = (mi === 0 ? 0 : 0.01);

      // Weighted objective.
      const score =
        immediate * 2.0 +
        pot * 1.0 +
        moveBonus -
        sep * 0.35 -
        loop * 0.9 +
        (randFloat() - 0.5) * 1e-6; // deterministic tie-break.

      if (score > bestScore) {
        bestScore = score;
        bestX = nx;
        bestY = ny;
      }
    }

    curX[k] = bestX;
    curY[k] = bestY;
    paths[k].push([bestX, bestY]);

    const h = hist[k];
    histPos[k] = (histPos[k] + 1) % histLen;
    h[histPos[k]] = idx(bestX, bestY);

    markVisited(bestX, bestY);
  }
}

process.stdout.write(JSON.stringify({ paths }));
