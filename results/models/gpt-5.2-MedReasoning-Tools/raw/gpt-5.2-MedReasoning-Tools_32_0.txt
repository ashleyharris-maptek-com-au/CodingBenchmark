using std::cin;
using std::cout;
using std::pair;
using std::priority_queue;
using std::vector;

static constexpr int64_t INF64 = (std::numeric_limits<int64_t>::max)() / 4;

struct Edge {
  int to;
  int id;
  int64_t w;
};

struct InputEdge {
  int u;
  int v;
  int64_t w;
};

struct DSU {
  vector<int> parent;
  vector<int> rankv;
  explicit DSU(int n) : parent(n), rankv(n, 0) { std::iota(parent.begin(), parent.end(), 0); }
  int find(int a) {
    while (parent[a] != a) {
      parent[a] = parent[parent[a]];
      a = parent[a];
    }
    return a;
  }
  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (rankv[a] < rankv[b]) std::swap(a, b);
    parent[b] = a;
    if (rankv[a] == rankv[b]) rankv[a]++;
    return true;
  }
};

static void dijkstra(int n, const vector<vector<Edge>>& g, int src, vector<int64_t>& dist, vector<int>& parent_v,
                     vector<int>& parent_e) {
  dist.assign(n, INF64);
  parent_v.assign(n, -1);
  parent_e.assign(n, -1);

  using State = pair<int64_t, int>;
  priority_queue<State, vector<State>, std::greater<State>> pq;
  dist[src] = 0;
  pq.push({0, src});

  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != dist[v]) continue;
    for (const auto& e : g[v]) {
      int nv = e.to;
      int64_t nd = d + e.w;
      if (nd < dist[nv]) {
        dist[nv] = nd;
        parent_v[nv] = v;
        parent_e[nv] = e.id;
        pq.push({nd, nv});
      }
    }
  }
}

static void reconstruct_path_edges(int src, int dst, const vector<int>& parent_v, const vector<int>& parent_e,
                                   vector<int>& out_edge_ids) {
  int v = dst;
  while (v != -1 && v != src) {
    int pe = parent_e[v];
    int pv = parent_v[v];
    if (pe < 0 || pv < 0) break;
    out_edge_ids.push_back(pe);
    v = pv;
  }
}

static vector<pair<int, int>> build_tree_from_edge_ids(int n, const vector<InputEdge>& edges, const vector<int>& edge_ids,
                                                       int64_t& total_weight_out) {
  // Deduplicate edge ids (a path-union can repeat edges).
  vector<int> uniq = edge_ids;
  std::sort(uniq.begin(), uniq.end());
  uniq.erase(std::unique(uniq.begin(), uniq.end()), uniq.end());

  // Kruskal on the subgraph consisting of these edges.
  vector<int> order = uniq;
  std::sort(order.begin(), order.end(), [&](int a, int b) { return edges[a].w < edges[b].w; });
  DSU dsu(n);
  vector<pair<int, int>> tree_edges;
  total_weight_out = 0;
  for (int id : order) {
    int u = edges[id].u, v = edges[id].v;
    if (dsu.unite(u, v)) {
      tree_edges.push_back({u, v});
      total_weight_out += edges[id].w;
    }
  }
  return tree_edges;
}

static bool is_terminal_covered(const vector<int>& terminals, const vector<int>& comp) {
  int root = comp[terminals[0]];
  for (int t : terminals) {
    if (comp[t] != root) return false;
  }
  return true;
}

static void compute_components(int n, const vector<pair<int, int>>& tree_edges, vector<int>& comp_out) {
  DSU dsu(n);
  for (auto [u, v] : tree_edges) dsu.unite(u, v);
  comp_out.resize(n);
  for (int i = 0; i < n; i++) comp_out[i] = dsu.find(i);
}

static vector<pair<int, int>> heuristic_steiner_tree(int n, const vector<vector<Edge>>& g, const vector<InputEdge>& edges,
                                                     const vector<int>& terminals, int64_t& total_weight_out) {
  const int k = static_cast<int>(terminals.size());
  if (k <= 1) {
    total_weight_out = 0;
    return {};
  }

  // Run Dijkstra from each terminal; keep parents to reconstruct each MST edge path.
  vector<vector<int64_t>> dist(k);
  vector<vector<int>> parent_v(k), parent_e(k);
  for (int i = 0; i < k; i++) {
    dijkstra(n, g, terminals[i], dist[i], parent_v[i], parent_e[i]);
  }

  // Prim's MST on the complete graph over terminals (metric closure).
  vector<int64_t> key(k, INF64);
  vector<int> parent(k, -1);
  vector<int> parent_src(k, -1); // which Dijkstra tree to use (source terminal index)
  vector<char> in_mst(k, 0);
  key[0] = 0;
  for (int it = 0; it < k; it++) {
    int u = -1;
    int64_t best = INF64;
    for (int i = 0; i < k; i++) {
      if (!in_mst[i] && key[i] < best) {
        best = key[i];
        u = i;
      }
    }
    if (u == -1) break;
    in_mst[u] = 1;
    for (int v = 0; v < k; v++) {
      if (in_mst[v]) continue;
      int64_t w = dist[u][terminals[v]];
      if (w < key[v]) {
        key[v] = w;
        parent[v] = u;
        parent_src[v] = u;
      }
    }
  }

  // Collect all edges used by expanded MST paths.
  vector<int> used_edge_ids;
  used_edge_ids.reserve(k * 16);
  for (int v = 1; v < k; v++) {
    int u = parent[v];
    if (u < 0) continue;
    int src_idx = parent_src[v];
    int src_vertex = terminals[src_idx];
    int dst_vertex = terminals[v];
    reconstruct_path_edges(src_vertex, dst_vertex, parent_v[src_idx], parent_e[src_idx], used_edge_ids);
  }

  // Build an acyclic subset (tree/forest) of that union via Kruskal.
  auto tree_edges = build_tree_from_edge_ids(n, edges, used_edge_ids, total_weight_out);

  // If disconnected (can happen in disconnected graph), fall back to connecting terminals via a simple star on best reachable center.
  vector<int> comp;
  compute_components(n, tree_edges, comp);
  if (!is_terminal_covered(terminals, comp)) {
    // Choose a center among terminals with minimal sum distances.
    int best_center = 0;
    int64_t best_sum = INF64;
    for (int c = 0; c < k; c++) {
      int64_t sum = 0;
      for (int i = 0; i < k; i++) {
        int64_t d = dist[c][terminals[i]];
        if (d >= INF64 / 2) {
          sum = INF64;
          break;
        }
        sum += d;
        if (sum >= best_sum) break;
      }
      if (sum < best_sum) {
        best_sum = sum;
        best_center = c;
      }
    }
    used_edge_ids.clear();
    for (int i = 0; i < k; i++) {
      if (i == best_center) continue;
      reconstruct_path_edges(terminals[best_center], terminals[i], parent_v[best_center], parent_e[best_center], used_edge_ids);
    }
    tree_edges = build_tree_from_edge_ids(n, edges, used_edge_ids, total_weight_out);
  }
  return tree_edges;
}

static vector<pair<int, int>> exact_steiner_tree_small_k(int n, const vector<vector<Edge>>& g, const vector<InputEdge>& edges,
                                                         const vector<int>& terminals, int64_t& total_weight_out) {
  const int k = static_cast<int>(terminals.size());
  if (k <= 1) {
    total_weight_out = 0;
    return {};
  }

  const int full = (1 << k) - 1;
  const int64_t INF = INF64;

  // dp[mask*n + v] = min cost of Steiner tree connecting terminals in mask and "ending" at v.
  vector<int64_t> dp(static_cast<size_t>(1 << k) * static_cast<size_t>(n), INF);
  vector<int> prev_mask(static_cast<size_t>(1 << k) * static_cast<size_t>(n), -1); // split info: previous mask used
  vector<int> prev_v(static_cast<size_t>(1 << k) * static_cast<size_t>(n), -1);    // previous vertex (for dijkstra step)
  vector<int> prev_e(static_cast<size_t>(1 << k) * static_cast<size_t>(n), -1);    // previous edge id (for dijkstra step)

  auto idx = [n](int mask, int v) -> size_t { return static_cast<size_t>(mask) * static_cast<size_t>(n) + static_cast<size_t>(v); };

  for (int i = 0; i < k; i++) {
    dp[idx(1 << i, terminals[i])] = 0;
  }

  // For each mask: combine submasks, then run multi-source Dijkstra to propagate along edges.
  vector<int> in_queue(n, 0);
  vector<int64_t> dist(n, INF);
  vector<int> parentv(n, -1), parente(n, -1);

  for (int mask = 1; mask <= full; mask++) {
    // Subset combination: dp[mask][v] = min(dp[sub][v] + dp[mask^sub][v])
    for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
      int other = mask ^ sub;
      if (other == 0) continue;
      // To avoid double-work, only consider sub < other.
      if (sub > other) continue;
      for (int v = 0; v < n; v++) {
        int64_t a = dp[idx(sub, v)];
        int64_t b = dp[idx(other, v)];
        if (a >= INF / 2 || b >= INF / 2) continue;
        int64_t cand = a + b;
        size_t idm = idx(mask, v);
        if (cand < dp[idm]) {
          dp[idm] = cand;
          prev_mask[idm] = sub; // indicates a split at same v
          prev_v[idm] = v;
          prev_e[idm] = -1;
        }
      }
    }

    // Multi-source Dijkstra using dp[mask][*] as initial distances.
    std::fill(dist.begin(), dist.end(), INF);
    std::fill(parentv.begin(), parentv.end(), -1);
    std::fill(parente.begin(), parente.end(), -1);
    using State = pair<int64_t, int>;
    priority_queue<State, vector<State>, std::greater<State>> pq;
    for (int v = 0; v < n; v++) {
      int64_t base = dp[idx(mask, v)];
      if (base < INF / 2) {
        dist[v] = base;
        pq.push({base, v});
        parentv[v] = v;
      }
    }

    while (!pq.empty()) {
      auto [d, v] = pq.top();
      pq.pop();
      if (d != dist[v]) continue;
      for (const auto& e : g[v]) {
        int nv = e.to;
        int64_t nd = d + e.w;
        if (nd < dist[nv]) {
          dist[nv] = nd;
          parentv[nv] = v;
          parente[nv] = e.id;
          pq.push({nd, nv});
        }
      }
    }
    for (int v = 0; v < n; v++) {
      size_t idm = idx(mask, v);
      if (dist[v] < dp[idm]) {
        dp[idm] = dist[v];
        prev_mask[idm] = -2; // indicates shortest-path relaxation
        prev_v[idm] = parentv[v];
        prev_e[idm] = parente[v];
      }
    }
  }

  // Choose best root v for full mask.
  int best_v = 0;
  int64_t best_cost = INF;
  for (int v = 0; v < n; v++) {
    int64_t cost = dp[idx(full, v)];
    if (cost < best_cost) {
      best_cost = cost;
      best_v = v;
    }
  }

  // Reconstruct by traversing backpointers and collecting edges.
  vector<int> used_edge_ids;
  used_edge_ids.reserve(n);

  std::function<void(int, int)> rec = [&](int mask, int v) {
    if ((mask & (mask - 1)) == 0) return; // single terminal: no edges necessarily
    size_t idm = idx(mask, v);
    int pm = prev_mask[idm];
    if (pm == -2) {
      int pv = prev_v[idm];
      int pe = prev_e[idm];
      if (pe >= 0 && pv >= 0 && pv != v) {
        used_edge_ids.push_back(pe);
        rec(mask, pv);
      }
      return;
    }
    if (pm > 0) {
      int other = mask ^ pm;
      int split_v = prev_v[idm];
      rec(pm, split_v);
      rec(other, split_v);
      return;
    }
    // If no backpointer recorded, nothing to do.
  };

  rec(full, best_v);

  auto tree_edges = build_tree_from_edge_ids(n, edges, used_edge_ids, total_weight_out);
  return tree_edges;
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n = 0, m = 0, k = 0;
  if (!(cin >> n >> m >> k)) return 0;

  vector<InputEdge> edges;
  edges.reserve(m);
  vector<vector<Edge>> g(n);
  for (int i = 0; i < m; i++) {
    int u, v;
    int64_t w;
    cin >> u >> v >> w;
    edges.push_back({u, v, w});
    g[u].push_back({v, i, w});
    g[v].push_back({u, i, w});
  }

  vector<int> terminals(k);
  for (int i = 0; i < k; i++) cin >> terminals[i];
  std::sort(terminals.begin(), terminals.end());
  terminals.erase(std::unique(terminals.begin(), terminals.end()), terminals.end());
  k = static_cast<int>(terminals.size());

  if (k <= 1) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  // Decide exact vs heuristic based on memory budget.
  const int max_k_for_exact = 18;
  const int64_t max_states = 5'000'000; // (1<<k) * n
  bool use_exact = (k <= max_k_for_exact) && (static_cast<int64_t>(1LL << k) * static_cast<int64_t>(n) <= max_states);

  int64_t total_weight = 0;
  vector<pair<int, int>> tree_edges;
  if (use_exact) {
    tree_edges = exact_steiner_tree_small_k(n, g, edges, terminals, total_weight);
  } else {
    tree_edges = heuristic_steiner_tree(n, g, edges, terminals, total_weight);
  }

  // Final validity adjustments: ensure connected over terminals by adding shortest paths if needed.
  // If heuristic produced a forest, connect components via extra shortest paths between components.
  {
    vector<int> comp;
    compute_components(n, tree_edges, comp);
    if (!is_terminal_covered(terminals, comp)) {
      // Precompute dijkstra from each terminal until connected by unioning paths between components.
      // We'll greedily connect components using terminal-to-terminal shortest paths.
      const int kk = k;
      vector<vector<int64_t>> dist(kk);
      vector<vector<int>> parent_v(kk), parent_e(kk);
      for (int i = 0; i < kk; i++) dijkstra(n, g, terminals[i], dist[i], parent_v[i], parent_e[i]);

      vector<int> used_edge_ids;
      used_edge_ids.reserve(tree_edges.size() * 2 + kk * 16);
      // Add existing edges into used set.
      std::unordered_map<uint64_t, int> edge_map;
      edge_map.reserve(edges.size() * 2);
      for (int id = 0; id < (int)edges.size(); id++) {
        int a = edges[id].u, b = edges[id].v;
        if (a > b) std::swap(a, b);
        uint64_t key = (static_cast<uint64_t>(a) << 32) | static_cast<uint32_t>(b);
        edge_map.emplace(key, id);
      }
      for (auto [u, v] : tree_edges) {
        int a = u, b = v;
        if (a > b) std::swap(a, b);
        uint64_t key = (static_cast<uint64_t>(a) << 32) | static_cast<uint32_t>(b);
        auto it = edge_map.find(key);
        if (it != edge_map.end()) used_edge_ids.push_back(it->second);
      }

      compute_components(n, tree_edges, comp);
      while (!is_terminal_covered(terminals, comp)) {
        // Pick two terminals in different components with minimal distance.
        int best_i = -1, best_j = -1;
        int64_t best_d = INF64;
        for (int i = 0; i < kk; i++) {
          for (int j = i + 1; j < kk; j++) {
            if (comp[terminals[i]] == comp[terminals[j]]) continue;
            int64_t d = dist[i][terminals[j]];
            if (d < best_d) {
              best_d = d;
              best_i = i;
              best_j = j;
            }
          }
        }
        if (best_i < 0 || best_d >= INF64 / 2) break;
        reconstruct_path_edges(terminals[best_i], terminals[best_j], parent_v[best_i], parent_e[best_i], used_edge_ids);
        tree_edges = build_tree_from_edge_ids(n, edges, used_edge_ids, total_weight);
        compute_components(n, tree_edges, comp);
      }
    }
  }

  cout << total_weight << " " << tree_edges.size() << "\n";
  for (auto [u, v] : tree_edges) cout << u << " " << v << "\n";
  return 0;
}
