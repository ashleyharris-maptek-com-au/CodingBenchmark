using i64 = long long;

struct VecHash {
  size_t operator()(const std::vector<int>& v) const noexcept {
    // FNV-1a 64-bit
    uint64_t h = 1469598103934665603ull;
    for (int x : v) {
      uint64_t y = static_cast<uint64_t>(static_cast<uint32_t>(x)) + 0x9e3779b97f4a7c15ull;
      h ^= y;
      h *= 1099511628211ull;
    }
    return static_cast<size_t>(h);
  }
};

static bool in_bounds(const std::vector<int>& p, const std::vector<int>& bounds) {
  for (size_t i = 0; i < bounds.size(); ++i) {
    if (p[i] < 0 || p[i] >= bounds[i]) return false;
  }
  return true;
}

static std::vector<int> add_step(const std::vector<int>& p, int axis, int dir) {
  std::vector<int> q = p;
  q[axis] += dir;
  return q;
}

static i64 manhattan(const std::vector<int>& a, const std::vector<int>& b) {
  i64 s = 0;
  for (size_t i = 0; i < a.size(); ++i) s += std::llabs((i64)a[i] - (i64)b[i]);
  return s;
}

// Compute the set of cells that are occupied at the *time of arriving* at a state with a given depth,
// approximating snake movement (tail moves unless we eat at that step).
// For planning, we treat body positions as occupied but allow stepping into the current tail if it will move away.
struct SnakeState {
  std::deque<std::vector<int>> body; // front = head, back = tail
  std::unordered_set<std::vector<int>, VecHash> occ;

  explicit SnakeState(std::deque<std::vector<int>> b) : body(std::move(b)) {
    for (const auto& x : body) occ.insert(x);
  }

  const std::vector<int>& head() const { return body.front(); }
  const std::vector<int>& tail() const { return body.back(); }
  size_t length() const { return body.size(); }

  bool contains(const std::vector<int>& p) const { return occ.find(p) != occ.end(); }

  void step_to(const std::vector<int>& next_head, bool grow) {
    body.push_front(next_head);
    occ.insert(next_head);
    if (!grow) {
      auto t = body.back();
      body.pop_back();
      occ.erase(t);
    }
  }
};

struct Move {
  int axis = 0;
  int dir = +1;
};

// A* with a conservative collision model: disallow stepping into any occupied cell except the current tail
// (because tail will move if we don't grow on that step).
static bool astar_path_to_target(
    const SnakeState& snake,
    const std::vector<int>& bounds,
    const std::unordered_set<std::vector<int>, VecHash>& obstacles,
    const std::unordered_set<std::vector<int>, VecHash>& food_set,
    const std::vector<int>& target,
    int max_expand,
    std::vector<Move>& out_path) {
  struct Node {
    std::vector<int> pos;
    i64 g = 0;
    i64 f = 0;
  };
  struct Cmp {
    bool operator()(const Node& a, const Node& b) const { return a.f > b.f; }
  };

  const int D = (int)bounds.size();
  std::priority_queue<Node, std::vector<Node>, Cmp> pq;
  std::unordered_map<std::vector<int>, std::pair<std::vector<int>, Move>, VecHash> parent;
  std::unordered_map<std::vector<int>, i64, VecHash> best_g;

  auto start = snake.head();
  pq.push(Node{start, 0, manhattan(start, target)});
  best_g[start] = 0;

  int expanded = 0;
  while (!pq.empty() && expanded < max_expand) {
    Node cur = pq.top();
    pq.pop();
    auto itg = best_g.find(cur.pos);
    if (itg == best_g.end() || itg->second != cur.g) continue;

    if (cur.pos == target) {
      // Reconstruct
      std::vector<Move> rev;
      std::vector<int> p = cur.pos;
      while (p != start) {
        auto it = parent.find(p);
        if (it == parent.end()) break;
        rev.push_back(it->second.second);
        p = it->second.first;
      }
      std::reverse(rev.begin(), rev.end());
      out_path = std::move(rev);
      return true;
    }

    ++expanded;
    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, +1}) {
        std::vector<int> nxt = add_step(cur.pos, axis, dir);
        if (!in_bounds(nxt, bounds)) continue;
        if (obstacles.find(nxt) != obstacles.end()) continue;

        // Collision check (conservative): allow stepping into current tail (only if we don't eat there).
        bool is_tail = (nxt == snake.tail());
        bool is_food = (food_set.find(nxt) != food_set.end());
        if (snake.contains(nxt) && !(is_tail && !is_food)) continue;

        i64 ng = cur.g + 1;
        auto it = best_g.find(nxt);
        if (it == best_g.end() || ng < it->second) {
          best_g[nxt] = ng;
          parent[nxt] = {cur.pos, Move{axis, dir}};
          i64 nf = ng + manhattan(nxt, target);
          pq.push(Node{std::move(nxt), ng, nf});
        }
      }
    }
  }
  return false;
}

static std::vector<int> choose_food_target(
    const std::vector<std::vector<int>>& food,
    const SnakeState& snake) {
  // Nearest by Manhattan (ties stable).
  int best = 0;
  i64 bestd = std::numeric_limits<i64>::max();
  for (int i = 0; i < (int)food.size(); ++i) {
    i64 d = manhattan(food[i], snake.head());
    if (d < bestd) {
      bestd = d;
      best = i;
    }
  }
  return food[best];
}

static bool greedy_safe_step(
    const SnakeState& snake,
    const std::vector<int>& bounds,
    const std::unordered_set<std::vector<int>, VecHash>& obstacles,
    const std::unordered_set<std::vector<int>, VecHash>& food_set,
    const std::vector<int>& target,
    Move& chosen) {
  const int D = (int)bounds.size();
  struct Cand {
    Move m;
    i64 score;
  };
  std::vector<Cand> cands;
  cands.reserve(2 * D);

  for (int axis = 0; axis < D; ++axis) {
    for (int dir : {-1, +1}) {
      std::vector<int> nxt = add_step(snake.head(), axis, dir);
      if (!in_bounds(nxt, bounds)) continue;
      if (obstacles.find(nxt) != obstacles.end()) continue;

      bool is_tail = (nxt == snake.tail());
      bool is_food = (food_set.find(nxt) != food_set.end());
      if (snake.contains(nxt) && !(is_tail && !is_food)) continue;

      // Prefer moves that get closer to target, and keep options open (simple degree count).
      i64 dist = manhattan(nxt, target);
      int degree = 0;
      for (int ax2 = 0; ax2 < D; ++ax2) {
        for (int dr2 : {-1, +1}) {
          auto nn = add_step(nxt, ax2, dr2);
          if (!in_bounds(nn, bounds)) continue;
          if (obstacles.find(nn) != obstacles.end()) continue;
          bool nn_is_tail = (nn == snake.tail());
          bool nn_is_food = (food_set.find(nn) != food_set.end());
          if (snake.contains(nn) && !(nn_is_tail && !nn_is_food)) continue;
          ++degree;
        }
      }
      // Lower score is better.
      i64 score = dist * 1000 - degree;
      cands.push_back(Cand{Move{axis, dir}, score});
    }
  }
  if (cands.empty()) return false;
  std::sort(cands.begin(), cands.end(), [](const Cand& a, const Cand& b) { return a.score < b.score; });
  chosen = cands.front().m;
  return true;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int D = 0;
  int max_turns = 0;
  if (!(std::cin >> D >> max_turns)) return 0;

  std::vector<int> bounds(D);
  for (int i = 0; i < D; ++i) std::cin >> bounds[i];

  int snake_length = 1;
  std::vector<int> head(D);
  std::cin >> snake_length;
  for (int i = 0; i < D; ++i) std::cin >> head[i];

  // Input doesn't provide body coordinates; assume all segments start at head (degenerate).
  // This is safe for movement simulation because it behaves like a length-1 snake until it starts moving.
  std::deque<std::vector<int>> body;
  body.push_back(head);
  for (int i = 1; i < snake_length; ++i) body.push_back(head);
  SnakeState snake(std::move(body));

  int F = 0;
  std::cin >> F;
  std::vector<std::vector<int>> food;
  food.reserve(std::max(0, F));
  std::unordered_set<std::vector<int>, VecHash> food_set;
  for (int i = 0; i < F; ++i) {
    std::vector<int> p(D);
    for (int j = 0; j < D; ++j) std::cin >> p[j];
    food.push_back(p);
    food_set.insert(p);
  }

  int O = 0;
  std::cin >> O;
  std::unordered_set<std::vector<int>, VecHash> obstacles;
  obstacles.reserve((size_t)std::max(0, O) * 2 + 16);
  for (int i = 0; i < O; ++i) {
    std::vector<int> p(D);
    for (int j = 0; j < D; ++j) std::cin >> p[j];
    obstacles.insert(p);
  }

  // If no food, just output arbitrary safe moves (or nothing).
  if (F == 0) {
    for (int t = 0; t < max_turns; ++t) {
      Move m;
      if (!greedy_safe_step(snake, bounds, obstacles, food_set, snake.head(), m)) break;
      std::cout << m.axis << " " << m.dir << "\n";
      auto nxt = add_step(snake.head(), m.axis, m.dir);
      bool grow = (food_set.find(nxt) != food_set.end());
      snake.step_to(nxt, grow);
    }
    return 0;
  }

  // Main loop: plan to nearest food via bounded A*; fallback to greedy safe step.
  for (int turn = 0; turn < max_turns && !food.empty(); ++turn) {
    std::vector<int> target = choose_food_target(food, snake);

    std::vector<Move> path;
    // Budget expands with dimension and snake length, but remains bounded.
    int max_expand = 4000 + 200 * D + (int)snake.length() * 5;
    bool ok = astar_path_to_target(snake, bounds, obstacles, food_set, target, max_expand, path);

    Move m;
    if (ok && !path.empty()) {
      m = path.front();
    } else {
      if (!greedy_safe_step(snake, bounds, obstacles, food_set, target, m)) break;
    }

    std::cout << m.axis << " " << m.dir << "\n";

    auto nxt = add_step(snake.head(), m.axis, m.dir);
    bool grow = false;
    auto itf = food_set.find(nxt);
    if (itf != food_set.end()) {
      grow = true;
      food_set.erase(itf);
      // Remove from vector list
      for (size_t i = 0; i < food.size(); ++i) {
        if (food[i] == nxt) {
          food.erase(food.begin() + (ptrdiff_t)i);
          break;
        }
      }
    }
    snake.step_to(nxt, grow);
  }

  return 0;
}
