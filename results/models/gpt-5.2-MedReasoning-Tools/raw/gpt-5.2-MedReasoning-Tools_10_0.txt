// 2D Cutting Stock / Rectangle Bin Packing solver
// Heuristic, scalable: uses Skyline (better quality) for small/medium instances,
// and a fast Shelf packer for very large instances.
//
// Input:
//   N boardW boardH
//   N lines: w h
// Output:
//   numBoards
//   N lines: board_index x y rotated
//
// Notes:
// - board_index is 0-based.
// - rotated=1 means the rectangle was rotated 90 degrees (w/h swapped).

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

using std::int64_t;

struct RectIn {
  int id;
  int w;
  int h;
  int64_t area() const { return int64_t(w) * int64_t(h); }
};

struct Placement {
  int board = 0;
  int x = 0;
  int y = 0;
  int rotated = 0;
};

// -------------------- Shelf packer (very fast) --------------------

struct ShelfBoard {
  int W = 0, H = 0;
  int curX = 0;
  int curY = 0;
  int shelfH = 0;
  int64_t usedArea = 0;

  explicit ShelfBoard(int w, int h) : W(w), H(h) {}

  bool tryPlace(int rw, int rh, int& outX, int& outY) {
    // Transactional placement: do not mutate state unless successful.
    int nextCurX = curX;
    int nextCurY = curY;
    int nextShelfH = shelfH;
    int x = 0, y = 0;

    // Try current shelf.
    if (nextCurY + rh <= H && nextCurX + rw <= W) {
      x = nextCurX;
      y = nextCurY;
      nextCurX += rw;
      nextShelfH = std::max(nextShelfH, rh);
    } else {
      // Try starting a new shelf (only if we've placed something on the current shelf).
      if (nextCurX == 0) return false;
      nextCurY += nextShelfH;
      nextCurX = 0;
      nextShelfH = 0;
      if (nextCurY + rh > H || rw > W) return false;
      x = 0;
      y = nextCurY;
      nextCurX = rw;
      nextShelfH = rh;
    }

    outX = x;
    outY = y;
    curX = nextCurX;
    curY = nextCurY;
    shelfH = nextShelfH;
    usedArea += int64_t(rw) * int64_t(rh);
    return true;
  }
};

// -------------------- Skyline packer (better quality) --------------------

struct SkylineNode {
  int x = 0;
  int y = 0;
  int width = 0;
};

struct SkylineBoard {
  int W = 0, H = 0;
  std::vector<SkylineNode> nodes;
  int64_t usedArea = 0;

  explicit SkylineBoard(int w, int h) : W(w), H(h) {
    nodes.push_back(SkylineNode{0, 0, W});
  }

  int computeYForNode(int index, int rw, int rh) const {
    const int x = nodes[index].x;
    if (x + rw > W) return -1;
    int widthLeft = rw;
    int y = nodes[index].y;
    int i = index;
    while (widthLeft > 0) {
      if (i >= (int)nodes.size()) return -1;
      y = std::max(y, nodes[i].y);
      if (y + rh > H) return -1;
      widthLeft -= nodes[i].width;
      ++i;
    }
    return y;
  }

  void mergeSkylines() {
    for (int i = 0; i + 1 < (int)nodes.size(); ) {
      if (nodes[i].y == nodes[i + 1].y) {
        nodes[i].width += nodes[i + 1].width;
        nodes.erase(nodes.begin() + (i + 1));
      } else {
        ++i;
      }
    }
  }

  void addSkylineLevel(int index, int x, int y, int rw, int rh) {
    SkylineNode newNode{x, y + rh, rw};
    nodes.insert(nodes.begin() + index, newNode);

    // Fix overlaps to the right.
    for (int i = index + 1; i < (int)nodes.size(); ) {
      SkylineNode& prev = nodes[i - 1];
      SkylineNode& cur = nodes[i];
      const int overlap = (prev.x + prev.width) - cur.x;
      if (overlap <= 0) break;
      cur.x += overlap;
      cur.width -= overlap;
      if (cur.width <= 0) {
        nodes.erase(nodes.begin() + i);
        continue;
      }
      break;
    }
    mergeSkylines();
  }

  bool tryPlace(int rw, int rh, int& outX, int& outY, int& outIndex) const {
    int bestYPlusH = std::numeric_limits<int>::max();
    int bestY = std::numeric_limits<int>::max();
    int bestX = 0;
    int bestIndex = -1;

    for (int i = 0; i < (int)nodes.size(); ++i) {
      const int y = computeYForNode(i, rw, rh);
      if (y < 0) continue;
      const int yPlusH = y + rh;
      const int x = nodes[i].x;
      if (yPlusH < bestYPlusH || (yPlusH == bestYPlusH && (y < bestY || (y == bestY && x < bestX)))) {
        bestYPlusH = yPlusH;
        bestY = y;
        bestX = x;
        bestIndex = i;
      }
    }
    if (bestIndex < 0) return false;
    outX = bestX;
    outY = bestY;
    outIndex = bestIndex;
    return true;
  }

  bool place(int rw, int rh, int& outX, int& outY) {
    int index = -1;
    if (!tryPlace(rw, rh, outX, outY, index)) return false;
    addSkylineLevel(index, outX, outY, rw, rh);
    usedArea += int64_t(rw) * int64_t(rh);
    return true;
  }
};

// -------------------- Multi-board orchestrators --------------------

static inline bool canFitEitherOrientation(int W, int H, int w, int h) {
  return (w <= W && h <= H) || (h <= W && w <= H);
}

static Placement packWithShelves(int W, int H, const RectIn& r, std::vector<ShelfBoard>& boards) {
  // Try last few boards first.
  const int scan = std::min<int>((int)boards.size(), 8);
  int bestBoard = -1;
  int bestX = 0, bestY = 0, bestRot = 0;

  auto tryBoard = [&](int bi, int rw, int rh, int rotFlag, int& outX, int& outY) -> bool {
    // We need a non-const attempt because it mutates. We'll "simulate" by checking feasibility
    // with a cheap heuristic: only allow actual placement on first matching board from our search,
    // preferring lower y then lower x.
    (void)rotFlag;
    return boards[bi].tryPlace(rw, rh, outX, outY);
  };

  // For shelves, actually trying mutates, so we do greedy: first-fit on a short list.
  // To preserve the "try last few first" behavior, we iterate accordingly.
  for (int t = 0; t < scan; ++t) {
    const int bi = (int)boards.size() - 1 - t;
    int x = 0, y = 0;
    if (r.w <= W && r.h <= H && tryBoard(bi, r.w, r.h, 0, x, y)) return Placement{bi, x, y, 0};
    if (r.h <= W && r.w <= H && tryBoard(bi, r.h, r.w, 1, x, y)) return Placement{bi, x, y, 1};
  }

  // Try all boards only if count is small.
  if ((int)boards.size() <= 64) {
    for (int bi = 0; bi < (int)boards.size(); ++bi) {
      int x = 0, y = 0;
      if (r.w <= W && r.h <= H && boards[bi].tryPlace(r.w, r.h, x, y)) return Placement{bi, x, y, 0};
      if (r.h <= W && r.w <= H && boards[bi].tryPlace(r.h, r.w, x, y)) return Placement{bi, x, y, 1};
    }
  }

  // New board.
  boards.emplace_back(W, H);
  int x = 0, y = 0;
  if (r.w <= W && r.h <= H && boards.back().tryPlace(r.w, r.h, x, y)) return Placement{(int)boards.size() - 1, x, y, 0};
  // must fit by earlier feasibility
  (void)boards.back().tryPlace(r.h, r.w, x, y);
  return Placement{(int)boards.size() - 1, x, y, 1};
}

static Placement packWithSkylines(int W, int H, const RectIn& r, std::vector<SkylineBoard>& boards) {
  const int64_t boardArea = int64_t(W) * int64_t(H);
  // Candidate set: last few boards + best free-area board.
  int bestFreeIdx = -1;
  int64_t bestFree = -1;
  for (int i = 0; i < (int)boards.size(); ++i) {
    const int64_t freeA = boardArea - boards[i].usedArea;
    if (freeA > bestFree) {
      bestFree = freeA;
      bestFreeIdx = i;
    }
  }

  std::vector<int> candidates;
  candidates.reserve(16);
  if (bestFreeIdx >= 0) candidates.push_back(bestFreeIdx);
  for (int t = 0; t < 8 && t < (int)boards.size(); ++t) {
    const int bi = (int)boards.size() - 1 - t;
    if (bi != bestFreeIdx) candidates.push_back(bi);
  }
  if ((int)boards.size() <= 32) {
    candidates.clear();
    for (int i = 0; i < (int)boards.size(); ++i) candidates.push_back(i);
  }

  int chosenBoard = -1;
  int chosenX = 0, chosenY = 0, chosenRot = 0;
  int bestScore = std::numeric_limits<int>::max(); // lower is better: y+h

  for (int bi : candidates) {
    const int64_t freeA = boardArea - boards[bi].usedArea;
    if (freeA < r.area()) continue;
    int x = 0, y = 0, idx = -1;
    if (r.w <= W && r.h <= H && boards[bi].tryPlace(r.w, r.h, x, y, idx)) {
      const int score = y + r.h;
      if (score < bestScore) {
        bestScore = score;
        chosenBoard = bi;
        chosenX = x;
        chosenY = y;
        chosenRot = 0;
      }
    }
    if (r.h <= W && r.w <= H && boards[bi].tryPlace(r.h, r.w, x, y, idx)) {
      const int score = y + r.w;
      if (score < bestScore) {
        bestScore = score;
        chosenBoard = bi;
        chosenX = x;
        chosenY = y;
        chosenRot = 1;
      }
    }
  }

  if (chosenBoard >= 0) {
    if (chosenRot == 0) {
      int x = 0, y = 0;
      (void)boards[chosenBoard].place(r.w, r.h, x, y);
      return Placement{chosenBoard, x, y, 0};
    } else {
      int x = 0, y = 0;
      (void)boards[chosenBoard].place(r.h, r.w, x, y);
      return Placement{chosenBoard, x, y, 1};
    }
  }

  boards.emplace_back(W, H);
  int x = 0, y = 0;
  if (r.w <= W && r.h <= H && boards.back().place(r.w, r.h, x, y)) return Placement{(int)boards.size() - 1, x, y, 0};
  (void)boards.back().place(r.h, r.w, x, y);
  return Placement{(int)boards.size() - 1, x, y, 1};
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int N = 0;
  int W = 0, H = 0;
  if (!(std::cin >> N >> W >> H)) return 0;

  std::vector<RectIn> rects;
  rects.reserve(std::max(0, N));

  for (int i = 0; i < N; ++i) {
    int w = 0, h = 0;
    std::cin >> w >> h;
    rects.push_back(RectIn{i, w, h});
  }

  // Sort by descending "difficulty": max side, then area.
  std::vector<int> order(N);
  std::iota(order.begin(), order.end(), 0);
  std::sort(order.begin(), order.end(), [&](int a, int b) {
    const RectIn& ra = rects[a];
    const RectIn& rb = rects[b];
    const int ma = std::max(ra.w, ra.h);
    const int mb = std::max(rb.w, rb.h);
    if (ma != mb) return ma > mb;
    const int64_t aa = ra.area();
    const int64_t ab = rb.area();
    if (aa != ab) return aa > ab;
    return ra.id < rb.id;
  });

  // If any rectangle cannot fit, we still must output something; put it at (0,0) in its own board.
  // Benchmark instances are expected to be feasible.
  std::vector<Placement> out(N);

  const bool useSkyline = (N <= 8000);

  if (useSkyline) {
    std::vector<SkylineBoard> boards;
    boards.reserve(64);
    for (int idx : order) {
      const RectIn& r = rects[idx];
      if (!canFitEitherOrientation(W, H, r.w, r.h)) {
        // Fallback: dedicate a board and clamp to (0,0)
        boards.emplace_back(W, H);
        out[r.id] = Placement{(int)boards.size() - 1, 0, 0, 0};
        continue;
      }
      if (boards.empty()) boards.emplace_back(W, H);
      out[r.id] = packWithSkylines(W, H, r, boards);
    }

    std::cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
      const Placement& p = out[i];
      std::cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }
    return 0;
  }

  // Large instances: shelf first-fit for speed.
  std::vector<ShelfBoard> boards;
  boards.reserve(256);
  for (int idx : order) {
    const RectIn& r = rects[idx];
    if (!canFitEitherOrientation(W, H, r.w, r.h)) {
      boards.emplace_back(W, H);
      out[r.id] = Placement{(int)boards.size() - 1, 0, 0, 0};
      continue;
    }
    if (boards.empty()) boards.emplace_back(W, H);
    out[r.id] = packWithShelves(W, H, r, boards);
  }

  std::cout << boards.size() << "\n";
  for (int i = 0; i < N; ++i) {
    const Placement& p = out[i];
    std::cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
  }
  return 0;
}
