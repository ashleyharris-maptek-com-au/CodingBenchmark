const fs = require('fs');

class BinaryHeap {
  constructor(less) {
    this.a = [];
    this.less = less; // less(x,y) => true if x has higher priority than y for root
  }
  size() {
    return this.a.length;
  }
  peek() {
    return this.a[0];
  }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      this._down(0);
    }
    return top;
  }
  _down(i) {
    const a = this.a;
    const n = a.length;
    while (true) {
      let best = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && this.less(a[l], a[best])) best = l;
      if (r < n && this.less(a[r], a[best])) best = r;
      if (best === i) break;
      [a[i], a[best]] = [a[best], a[i]];
      i = best;
    }
  }
}

function safeNum(x, dflt) {
  const v = Number(x);
  return Number.isFinite(v) ? v : dflt;
}

function main() {
  const input = fs.readFileSync(0, 'utf8').trim();
  const data = input ? JSON.parse(input) : {};
  const machineCount = Math.max(0, (data.machineCount | 0) || 0);
  const tasksIn = Array.isArray(data.tasks) ? data.tasks : [];

  const n = tasksIn.length;
  const idToIdx = new Map();
  const ids = new Array(n);
  for (let i = 0; i < n; i++) {
    const id = String(tasksIn[i]?.id);
    ids[i] = id;
    idToIdx.set(id, i);
  }

  const duration = new Array(n);
  const deadline = new Array(n);
  const weight = new Array(n);
  const deps = new Array(n);
  const succ = Array.from({ length: n }, () => []);
  const indeg = new Array(n).fill(0);

  for (let i = 0; i < n; i++) {
    const t = tasksIn[i] || {};
    duration[i] = Math.max(0, safeNum(t.duration, 0));
    deadline[i] = safeNum(t.deadline, Number.POSITIVE_INFINITY);
    weight[i] = Math.max(0, safeNum(t.weight, 1));
    const depList = Array.isArray(t.deps) ? t.deps : [];
    const mapped = [];
    for (let k = 0; k < depList.length; k++) {
      const did = String(depList[k]);
      const di = idToIdx.get(did);
      if (di === undefined) continue; // ignore unknown deps
      mapped.push(di);
    }
    deps[i] = mapped;
  }

  for (let i = 0; i < n; i++) {
    const dl = deps[i];
    indeg[i] = dl.length;
    for (let k = 0; k < dl.length; k++) {
      succ[dl[k]].push(i);
    }
  }

  // Kahn topo; also used for cycle detection and reverse-topo DP.
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  let qh = 0;
  const topo = [];
  const indegTmp = indeg.slice();
  while (qh < q.length) {
    const v = q[qh++];
    topo.push(v);
    const s = succ[v];
    for (let i = 0; i < s.length; i++) {
      const u = s[i];
      indegTmp[u]--;
      if (indegTmp[u] === 0) q.push(u);
    }
  }

  if (topo.length !== n || machineCount === 0) {
    // Cycle or no machines: return a deterministic order fallback.
    // If cycle, topo is partial; append remaining in input order.
    const seen = new Array(n).fill(false);
    for (let i = 0; i < topo.length; i++) seen[topo[i]] = true;
    const order = topo.map((i) => ids[i]);
    for (let i = 0; i < n; i++) if (!seen[i]) order.push(ids[i]);
    process.stdout.write(JSON.stringify({ order }));
    return;
  }

  // Critical path length to exit (upward rank / longest path).
  const crit = new Array(n).fill(0);
  for (let ti = n - 1; ti >= 0; ti--) {
    const v = topo[ti];
    let bestSucc = 0;
    const s = succ[v];
    for (let i = 0; i < s.length; i++) {
      const u = s[i];
      if (crit[u] > bestSucc) bestSucc = crit[u];
    }
    crit[v] = duration[v] + bestSucc;
  }

  // Ready-task max-heap: higher crit, higher weight, earlier deadline, longer duration, id.
  const readyLess = (a, b) => {
    if (crit[a] !== crit[b]) return crit[a] > crit[b];
    if (weight[a] !== weight[b]) return weight[a] > weight[b];
    if (deadline[a] !== deadline[b]) return deadline[a] < deadline[b];
    if (duration[a] !== duration[b]) return duration[a] > duration[b];
    return ids[a] < ids[b];
  };
  const ready = new BinaryHeap(readyLess);

  const depLeft = indeg.slice();
  const readyTime = new Array(n).fill(0);
  for (let i = 0; i < n; i++) if (depLeft[i] === 0) ready.push(i);

  // Machine min-heap by availability, then index.
  const machAvail = new Array(machineCount).fill(0);
  const machLess = (a, b) => {
    const ta = machAvail[a];
    const tb = machAvail[b];
    if (ta !== tb) return ta < tb;
    return a < b;
  };
  const machines = new BinaryHeap(machLess);
  for (let m = 0; m < machineCount; m++) machines.push(m);

  const schedule = [];
  let scheduledCount = 0;

  while (scheduledCount < n) {
    if (ready.size() === 0) {
      // Shouldn't happen for DAG, but keep deterministic fallback.
      process.stdout.write(JSON.stringify({ order: topo.map((i) => ids[i]) }));
      return;
    }

    const task = ready.pop();
    const machine = machines.pop();

    const start = Math.max(machAvail[machine], readyTime[task]);
    const finish = start + duration[task];
    machAvail[machine] = finish;
    machines.push(machine);

    schedule.push({ id: ids[task], start: Math.trunc(start), machine });
    scheduledCount++;

    const s = succ[task];
    for (let i = 0; i < s.length; i++) {
      const u = s[i];
      if (finish > readyTime[u]) readyTime[u] = finish;
      depLeft[u]--;
      if (depLeft[u] === 0) ready.push(u);
    }
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
