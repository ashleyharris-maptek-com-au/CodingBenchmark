const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const grid = Array.isArray(data.grid) ? data.grid : [];
const robots = Array.isArray(data.robots) ? data.robots : [];
const jobs = Array.isArray(data.jobs) ? data.jobs : [];

const h = grid.length;
const w = h ? grid[0].length : 0;
const nCells = w * h;

function idx(x, y) {
  return y * w + x;
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

// Precompute passability
const passable = new Uint8Array(nCells);
for (let y = 0; y < h; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < w; x++) {
    passable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
  }
}

// Determine assignments array size based on max robot id (robot ids are used as indices)
let maxRobotId = -1;
for (const r of robots) {
  const rid = (r && typeof r.id === 'number') ? r.id : -1;
  if (rid > maxRobotId) maxRobotId = rid;
}
const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

// If no robots, still must output valid JSON; put all jobs under index 0.
if (robots.length === 0) {
  const out = { assignments: [jobs.map((j) => j.id)] };
  process.stdout.write(JSON.stringify(out));
  process.exit(0);
}

// Stable robot ordering by id for deterministic tie-breaks
const robotsSorted = robots
  .map((r) => ({
    id: r.id,
    x: r.x | 0,
    y: r.y | 0,
  }))
  .sort((a, b) => a.id - b.id);

// Multi-source BFS from robot start positions to label each cell with nearest robot id and distance.
const dist = new Int32Array(nCells);
dist.fill(-1);
const owner = new Int32Array(nCells);
owner.fill(-1);

const q = new Int32Array(nCells);
let qh = 0;
let qt = 0;

for (const r of robotsSorted) {
  if (!inBounds(r.x, r.y)) continue;
  const start = idx(r.x, r.y);
  if (!passable[start]) continue;
  if (dist[start] === -1) {
    dist[start] = 0;
    owner[start] = r.id;
    q[qt++] = start;
  } else if (dist[start] === 0 && r.id < owner[start]) {
    owner[start] = r.id; // same cell start; pick smaller id
  }
}

while (qh < qt) {
  const v = q[qh++];
  const vd = dist[v];
  const vo = owner[v];
  const x = v % w;
  const y = (v / w) | 0;

  // Explore 4-neighborhood
  // Inline to reduce overhead
  let n;

  // left
  if (x > 0) {
    n = v - 1;
    if (passable[n]) {
      const nd = vd + 1;
      const cd = dist[n];
      if (cd === -1 || nd < cd || (nd === cd && vo < owner[n])) {
        dist[n] = nd;
        owner[n] = vo;
        if (cd === -1) q[qt++] = n;
      }
    }
  }
  // right
  if (x + 1 < w) {
    n = v + 1;
    if (passable[n]) {
      const nd = vd + 1;
      const cd = dist[n];
      if (cd === -1 || nd < cd || (nd === cd && vo < owner[n])) {
        dist[n] = nd;
        owner[n] = vo;
        if (cd === -1) q[qt++] = n;
      }
    }
  }
  // up
  if (y > 0) {
    n = v - w;
    if (passable[n]) {
      const nd = vd + 1;
      const cd = dist[n];
      if (cd === -1 || nd < cd || (nd === cd && vo < owner[n])) {
        dist[n] = nd;
        owner[n] = vo;
        if (cd === -1) q[qt++] = n;
      }
    }
  }
  // down
  if (y + 1 < h) {
    n = v + w;
    if (passable[n]) {
      const nd = vd + 1;
      const cd = dist[n];
      if (cd === -1 || nd < cd || (nd === cd && vo < owner[n])) {
        dist[n] = nd;
        owner[n] = vo;
        if (cd === -1) q[qt++] = n;
      }
    }
  }
}

// Helper: round-robin fallback robot id (smallest id) when unreachable
const fallbackRobotId = robotsSorted[0].id;

// Sort jobs by priority desc then id asc for determinism (used in tie cases / fallback behavior)
const jobsSorted = jobs
  .map((j) => ({
    id: j.id,
    x: j.x | 0,
    y: j.y | 0,
    priority: typeof j.priority === 'number' ? j.priority : 0,
  }))
  .sort((a, b) => (b.priority - a.priority) || ((a.id < b.id) ? -1 : (a.id > b.id ? 1 : 0)));

// Group jobs by nearest robot (by true grid distance from robot starts via multi-source BFS)
const perRobot = new Map(); // robotId -> array of job objects
for (const r of robotsSorted) perRobot.set(r.id, []);

for (const j of jobsSorted) {
  let rid = fallbackRobotId;
  let d = -1;
  if (inBounds(j.x, j.y)) {
    const p = idx(j.x, j.y);
    if (passable[p]) {
      d = dist[p];
      if (d !== -1) rid = owner[p];
    }
  }
  const arr = perRobot.get(rid);
  if (arr) {
    arr.push({ ...j, dFromStart: d });
  } else {
    // If robot id is unexpected, fall back to smallest id
    perRobot.get(fallbackRobotId).push({ ...j, dFromStart: d });
  }
}

// For each robot, produce an order: nearer-to-start first, then higher priority, then id
for (const r of robotsSorted) {
  const list = perRobot.get(r.id) || [];
  list.sort((a, b) => {
    const ad = a.dFromStart === -1 ? 0x3fffffff : a.dFromStart;
    const bd = b.dFromStart === -1 ? 0x3fffffff : b.dFromStart;
    if (ad !== bd) return ad - bd;
    if (a.priority !== b.priority) return b.priority - a.priority;
    return (a.id < b.id) ? -1 : (a.id > b.id ? 1 : 0);
  });
  if (r.id >= 0) {
    if (!assignments[r.id]) assignments[r.id] = [];
    for (const j of list) assignments[r.id].push(j.id);
  }
}

// Ensure every job appears exactly once:
// - If some robot ids were missing in [0..maxId], keep their assignment arrays as empty.
process.stdout.write(JSON.stringify({ assignments }));
