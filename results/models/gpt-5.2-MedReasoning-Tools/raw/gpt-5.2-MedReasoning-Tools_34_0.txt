import sys
import math
from typing import List, Tuple, Optional


def solve() -> None:
    raw = sys.stdin.read()
    raw = raw.lstrip()
    if raw.startswith("\ufeff"):
        raw = raw[1:]
    data = raw.strip().split()
    if not data:
        return
    # Be robust to BOM bytes being decoded as Latin-1 (ï»¿).
    def _strip_bom_bytes(s: str) -> str:
        return s.lstrip("\ufeff\u00ef\u00bb\u00bf")

    data = [_strip_bom_bytes(tok) for tok in data]
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))

    xs = [0.0] * n
    ys = [0.0] * n
    dem = [0.0] * n
    for i in range(n):
        xs[i] = float(next(it))
        ys[i] = float(next(it))
        dem[i] = float(next(it))

    def dist_idx(a: int, b: int) -> float:
        dx = xs[a] - xs[b]
        dy = ys[a] - ys[b]
        return math.hypot(dx, dy)

    def dist_depot(i: int) -> float:
        return math.hypot(xs[i] - depot_x, ys[i] - depot_y)

    # Precompute depot distances and (optionally) pairwise distances for speed.
    dd = [dist_depot(i) for i in range(n)]
    # Pairwise distance matrix can be large; compute lazily with a cache.
    dist_cache = {}

    def d(a: int, b: int) -> float:
        if a == -1 and b == -1:
            return 0.0
        if a == -1:
            return dd[b]
        if b == -1:
            return dd[a]
        if a == b:
            return 0.0
        if a > b:
            a, b = b, a
        key = (a, b)
        val = dist_cache.get(key)
        if val is None:
            val = dist_idx(a, b)
            dist_cache[key] = val
        return val

    total_demand = sum(dem)
    if total_demand > k * cap + 1e-9:
        # Infeasible with the given K; fall back to a feasible (capacity-respecting)
        # solution using as many routes as needed.
        k = n

    # --- Clarke-Wright savings heuristic (start with singletons then merge) ---
    routes: List[List[int]] = [[i] for i in range(n)]
    route_demand = [dem[i] for i in range(n)]
    route_of = list(range(n))  # customer -> route index

    # Savings list
    savings: List[Tuple[float, int, int]] = []
    for i in range(n):
        for j in range(i + 1, n):
            s = dd[i] + dd[j] - d(i, j)
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])

    def try_merge(i: int, j: int) -> bool:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            return False
        r1 = routes[ri]
        r2 = routes[rj]
        if not r1 or not r2:
            return False
        # i must be at an end of r1, j at an end of r2
        i_at_start = (r1[0] == i)
        i_at_end = (r1[-1] == i)
        j_at_start = (r2[0] == j)
        j_at_end = (r2[-1] == j)
        if not (i_at_start or i_at_end) or not (j_at_start or j_at_end):
            return False
        if route_demand[ri] + route_demand[rj] > cap + 1e-9:
            return False

        # Merge with correct orientation:
        # r1 + r2, where i is at end of r1 and j at start of r2, etc.
        if i_at_end and j_at_start:
            merged = r1 + r2
        elif i_at_start and j_at_end:
            merged = r2 + r1
        elif i_at_start and j_at_start:
            merged = list(reversed(r1)) + r2
        else:  # i_at_end and j_at_end
            merged = r1 + list(reversed(r2))

        # Apply merge: keep ri, clear rj
        routes[ri] = merged
        route_demand[ri] += route_demand[rj]
        routes[rj] = []
        route_demand[rj] = 0.0
        for c in merged:
            route_of[c] = ri
        return True

    for _, i, j in savings:
        try_merge(i, j)

    routes = [r for r in routes if r]

    def route_load(route: List[int]) -> float:
        return sum(dem[c] for c in route)

    def route_len(route: List[int]) -> float:
        if not route:
            return 0.0
        length = d(-1, route[0])
        for a, b in zip(route, route[1:]):
            length += d(a, b)
        length += d(route[-1], -1)
        return length

    def best_insertion(route: List[int], c: int) -> Tuple[float, int]:
        # returns (delta_cost, position)
        best_delta = float("inf")
        best_pos = 0
        m = len(route)
        for pos in range(m + 1):
            prev = -1 if pos == 0 else route[pos - 1]
            nxt = -1 if pos == m else route[pos]
            delta = d(prev, c) + d(c, nxt) - d(prev, nxt)
            if delta < best_delta:
                best_delta = delta
                best_pos = pos
        return best_delta, best_pos

    def two_opt(route: List[int], max_passes: int = 2) -> None:
        m = len(route)
        if m < 4:
            return
        # Keep it cheap for larger routes
        if m > 250:
            max_passes = 1
        for _ in range(max_passes):
            improved = False
            for i in range(m - 3):
                a = -1 if i == 0 else route[i - 1]
                b = route[i]
                for j in range(i + 2, m - 1):
                    c = route[j]
                    dnode = route[j + 1]
                    before = d(a, b) + d(c, dnode)
                    after = d(a, c) + d(b, dnode)
                    if after + 1e-12 < before:
                        route[i : j + 1] = reversed(route[i : j + 1])
                        improved = True
            if not improved:
                break

    # --- Enforce <= K routes by dissolving extra routes into others (if needed) ---
    def enforce_k(current_routes: List[List[int]]) -> Optional[List[List[int]]]:
        routes2 = [r[:] for r in current_routes]
        # While too many routes, dissolve the smallest one.
        while len(routes2) > k:
            routes2.sort(key=len)
            victim = routes2[0]
            rest = routes2[1:]
            loads = [route_load(r) for r in rest]
            caps_left = [cap - ld for ld in loads]
            ok = True
            for c in victim:
                best = (float("inf"), -1, 0)
                for idx, r in enumerate(rest):
                    if dem[c] > caps_left[idx] + 1e-9:
                        continue
                    delta, pos = best_insertion(r, c)
                    if delta < best[0]:
                        best = (delta, idx, pos)
                if best[1] == -1:
                    ok = False
                    break
                _, idx, pos = best
                rest[idx].insert(pos, c)
                caps_left[idx] -= dem[c]
            if not ok:
                return None
            routes2 = rest
        return routes2

    routes_k = enforce_k(routes)

    # Fallback construction: greedy cheapest insertion into exactly <= K routes.
    def build_greedy() -> List[List[int]]:
        customers = list(range(n))
        customers.sort(key=lambda i: dem[i], reverse=True)
        rs: List[List[int]] = [[] for _ in range(k)]
        caps_left = [cap] * k
        for c in customers:
            best = (float("inf"), -1, 0)
            for ridx in range(k):
                if dem[c] > caps_left[ridx] + 1e-9:
                    continue
                r = rs[ridx]
                if not r:
                    delta = 2.0 * dd[c]
                    pos = 0
                else:
                    delta, pos = best_insertion(r, c)
                if delta < best[0]:
                    best = (delta, ridx, pos)
            if best[1] == -1:
                # Should not happen if feasible; open a new route slot by expanding k.
                rs.append([c])
                caps_left.append(cap - dem[c])
            else:
                _, ridx, pos = best
                rs[ridx].insert(pos, c)
                caps_left[ridx] -= dem[c]
        rs = [r for r in rs if r]
        return rs

    if routes_k is None:
        routes = build_greedy()
    else:
        routes = routes_k

    # Local improvement: 2-opt per route
    for r in routes:
        two_opt(r)

    total = sum(route_len(r) for r in routes)

    out_lines = []
    out_lines.append(f"{total:.6f}")
    out_lines.append(str(len(routes)))
    for r in routes:
        out_lines.append(" ".join(map(str, r)))
    sys.stdout.write("\n".join(out_lines))


if __name__ == "__main__":
    solve()
