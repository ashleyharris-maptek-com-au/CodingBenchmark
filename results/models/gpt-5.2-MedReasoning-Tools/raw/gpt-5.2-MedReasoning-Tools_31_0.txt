use std::cmp::Ordering;
use std::io::{self, Read};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn skip_ws(&mut self) {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_i64(&mut self) -> Option<i64> {
        self.skip_ws();
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        }
        let mut v: i64 = 0;
        while self.idx < self.buf.len() {
            let c = self.buf[self.idx];
            if c.is_ascii_digit() {
                v = v * 10 + (c - b'0') as i64;
                self.idx += 1;
            } else {
                break;
            }
        }
        Some(v * sign)
    }
}

fn output_yes(indices: &[usize]) {
    let mut out = String::new();
    out.push_str("YES\n");
    if !indices.is_empty() {
        for (k, &i) in indices.iter().enumerate() {
            if k > 0 {
                out.push(' ');
            }
            out.push_str(&i.to_string());
        }
    }
    out.push('\n');
    print!("{out}");
}

fn output_no() {
    print!("NO\n");
}

fn meet_in_middle(nums: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    let left_len = left.len();
    let right_len = right.len();
    let left_cnt: usize = 1usize << left_len;
    let right_cnt: usize = 1usize << right_len;

    let mut left_sums: Vec<(i64, u64)> = Vec::with_capacity(left_cnt);
    left_sums.push((0, 0));
    for i in 0..left_len {
        let (val, _) = left[i];
        let cur_len = left_sums.len();
        for j in 0..cur_len {
            let (s, mask) = left_sums[j];
            left_sums.push((s + val, mask | (1u64 << i)));
        }
    }

    let mut right_sums: Vec<(i64, u64)> = Vec::with_capacity(right_cnt);
    right_sums.push((0, 0));
    for i in 0..right_len {
        let (val, _) = right[i];
        let cur_len = right_sums.len();
        for j in 0..cur_len {
            let (s, mask) = right_sums[j];
            right_sums.push((s + val, mask | (1u64 << i)));
        }
    }
    right_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    for (ls, lmask) in left_sums {
        let need = target - ls;
        let mut lo = 0usize;
        let mut hi = right_sums.len();
        while lo < hi {
            let m = (lo + hi) / 2;
            match right_sums[m].0.cmp(&need) {
                Ordering::Less => lo = m + 1,
                Ordering::Greater => hi = m,
                Ordering::Equal => {
                    let rmask = right_sums[m].1;
                    let mut res = Vec::new();
                    for i in 0..left_len {
                        if ((lmask >> i) & 1) != 0 {
                            res.push(left[i].1);
                        }
                    }
                    for i in 0..right_len {
                        if ((rmask >> i) & 1) != 0 {
                            res.push(right[i].1);
                        }
                    }
                    return Some(res);
                }
            }
        }
    }
    None
}

fn bitset_dp_nonneg(nums: &[(i64, usize)], target: usize) -> Option<Vec<usize>> {
    let words = target / 64 + 1;
    let mut bits = vec![0u64; words];
    bits[0] = 1u64;
    let mut parent_sum = vec![-1i32; target + 1];
    let mut parent_idx = vec![-1i32; target + 1];
    parent_sum[0] = 0;
    parent_idx[0] = -2;

    for (i, &(v64, _orig_idx)) in nums.iter().enumerate() {
        if v64 <= 0 {
            continue;
        }
        let v = v64 as usize;
        if v > target {
            continue;
        }
        let word_shift = v / 64;
        let bit_shift = v % 64;
        let mut j = words;
        while j > 0 {
            j -= 1;
            if j < word_shift {
                continue;
            }
            let src = j - word_shift;
            let mut shifted = bits[src] << bit_shift;
            if bit_shift != 0 && src > 0 {
                shifted |= bits[src - 1] >> (64 - bit_shift);
            }
            if shifted == 0 {
                continue;
            }
            let new_bits = shifted & !bits[j];
            if new_bits != 0 {
                let base = j * 64;
                let mut x = new_bits;
                while x != 0 {
                    let tz = x.trailing_zeros() as usize;
                    let s = base + tz;
                    if s <= target && parent_idx[s] == -1 {
                        parent_idx[s] = i as i32;
                        parent_sum[s] = (s - v) as i32;
                    }
                    x &= x - 1;
                }
            }
            bits[j] |= shifted;
        }
        if ((bits[target / 64] >> (target % 64)) & 1) != 0 {
            break;
        }
    }

    if ((bits[target / 64] >> (target % 64)) & 1) == 0 {
        return None;
    }

    let mut res = Vec::new();
    let mut s = target as i32;
    while s != 0 {
        let idx = parent_idx[s as usize];
        if idx < 0 {
            return None;
        }
        let idx_usize = idx as usize;
        res.push(nums[idx_usize].1);
        s = parent_sum[s as usize];
        if s < 0 {
            return None;
        }
    }
    Some(res)
}

fn small_hash_dp(nums: &[(i64, usize)], target: i64, limit_states: usize) -> Option<Vec<usize>> {
    use std::collections::HashMap;
    let mut map: HashMap<i64, (i64, usize)> = HashMap::new(); // sum -> (prev_sum, used_index_in_nums)
    map.insert(0, (0, usize::MAX));

    for (i, &(v, _)) in nums.iter().enumerate() {
        if v == 0 {
            continue;
        }
        if map.contains_key(&target) {
            break;
        }
        if map.len() > limit_states {
            break;
        }
        let mut additions: Vec<(i64, (i64, usize))> = Vec::new();
        additions.reserve(map.len().min(8192));
        for (&s, _) in map.iter() {
            let ns = s + v;
            if !map.contains_key(&ns) {
                additions.push((ns, (s, i)));
            }
        }
        for (ns, back) in additions {
            map.insert(ns, back);
        }
    }
    if !map.contains_key(&target) {
        return None;
    }
    let mut res = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev, idx) = *map.get(&cur)?;
        if idx == usize::MAX {
            break;
        }
        res.push(nums[idx].1);
        cur = prev;
    }
    Some(res)
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_i64() {
        Some(v) => v as usize,
        None => return,
    };
    let target = match input.next_i64() {
        Some(v) => v,
        None => return,
    };

    let mut nums: Vec<(i64, usize)> = Vec::with_capacity(n);
    let mut has_neg = false;
    let mut has_pos = false;
    let mut sum_pos: i128 = 0;
    let mut sum_neg: i128 = 0;
    let mut gcd_all: u64 = 0;
    for i in 0..n {
        let v = match input.next_i64() {
            Some(x) => x,
            None => 0,
        };
        if v < 0 {
            has_neg = true;
            sum_neg += v as i128;
        }
        if v > 0 {
            has_pos = true;
            sum_pos += v as i128;
        }
        let av = (v as i128).abs() as u64;
        if av != 0 {
            gcd_all = if gcd_all == 0 { av } else { gcd_u64(gcd_all, av) };
        }
        nums.push((v, i));
    }

    if target == 0 {
        output_yes(&[]);
        return;
    }

    for &(v, idx) in nums.iter() {
        if v == target {
            output_yes(&[idx]);
            return;
        }
    }

    // Safe pruning by reachable range and gcd.
    // For any subset sum, gcd(all numbers) must divide target.
    if gcd_all != 0 && ((target as i128).rem_euclid(gcd_all as i128)) != 0 {
        output_no();
        return;
    }
    let t128 = target as i128;
    if !has_neg {
        if t128 > sum_pos {
            output_no();
            return;
        }
    } else {
        if t128 < sum_neg || t128 > sum_pos {
            output_no();
            return;
        }
    }

    // Exact for small n: meet-in-the-middle handles negatives too.
    if n <= 46 {
        if let Some(sol) = meet_in_middle(&nums, target) {
            output_yes(&sol);
        } else {
            output_no();
        }
        return;
    }

    // If all numbers are non-negative and target is non-negative, use bitset DP when target is not too large.
    if !has_neg && target > 0 && has_pos {
        let t = target as u64;
        // Memory: parents (8 bytes per sum) + bitset (~1/8 byte per sum). Keep <= ~1.5GB.
        let max_target: u64 = 30_000_000;
        if t <= max_target {
            let t_usize = t as usize;
            if let Some(sol) = bitset_dp_nonneg(&nums, t_usize) {
                output_yes(&sol);
            } else {
                output_no();
            }
            return;
        }
    }

    // Moderate fallback: exact hash-DP with a cap on number of reachable sums.
    // Works well for instances with small effective state-space.
    let limit_states = 2_000_000usize;
    if let Some(sol) = small_hash_dp(&nums, target, limit_states) {
        output_yes(&sol);
    } else {
        output_no();
    }
}

fn gcd_u64(mut a: u64, mut b: u64) -> u64 {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}
