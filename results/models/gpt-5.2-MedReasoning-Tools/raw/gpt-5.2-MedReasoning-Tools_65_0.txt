// Mission-control workstation scheduler (TypeScript, no deps)
// Reads one JSON document from stdin and prints one JSON document to stdout.

declare const require: any;
declare const process: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8").trim();
const data = input.length ? JSON.parse(input) : {};

type TaskIn = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
};

type ScheduleItem = { id: string | number; start: number; machine: number };

class MinHeap<T> {
  private a: T[] = [];
  constructor(private less: (x: T, y: T) => boolean) {}
  size() {
    return this.a.length;
  }
  peek(): T | undefined {
    return this.a[0];
  }
  push(v: T) {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length) {
      a[0] = last;
      this.down(0);
    }
    return top;
  }
  private down(i: number) {
    const a = this.a;
    for (;;) {
      const l = i * 2 + 1;
      const r = l + 1;
      let m = i;
      if (l < a.length && this.less(a[l], a[m])) m = l;
      if (r < a.length && this.less(a[r], a[m])) m = r;
      if (m === i) break;
      const tmp = a[i];
      a[i] = a[m];
      a[m] = tmp;
      i = m;
    }
  }
}

class MaxHeap<T> {
  private a: T[] = [];
  constructor(private greater: (x: T, y: T) => boolean) {}
  size() {
    return this.a.length;
  }
  push(v: T) {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.greater(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length) {
      a[0] = last;
      this.down(0);
    }
    return top;
  }
  private down(i: number) {
    const a = this.a;
    for (;;) {
      const l = i * 2 + 1;
      const r = l + 1;
      let m = i;
      if (l < a.length && this.greater(a[l], a[m])) m = l;
      if (r < a.length && this.greater(a[r], a[m])) m = r;
      if (m === i) break;
      const tmp = a[i];
      a[i] = a[m];
      a[m] = tmp;
      i = m;
    }
  }
}

function clamp01(x: number) {
  if (x <= 0) return 0;
  if (x >= 1) return 1;
  return x;
}

const machineCount: number = Math.max(1, (data.machineCount | 0) || 1);
const tasksIn: TaskIn[] = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksIn.length;

// Map task id -> index
const idToIdx = new Map<string, number>();
const ids: Array<string | number> = new Array(n);
for (let i = 0; i < n; i++) {
  const idRaw = tasksIn[i]?.id;
  const idStr = String(idRaw);
  ids[i] = idRaw;
  idToIdx.set(idStr, i);
}

const dur = new Array<number>(n);
const deadline = new Array<number>(n);
const weight = new Array<number>(n);
const deps: number[][] = new Array(n);
const outs: number[][] = new Array(n);
for (let i = 0; i < n; i++) outs[i] = [];

for (let i = 0; i < n; i++) {
  const t = tasksIn[i] || ({} as TaskIn);
  const d = typeof t.duration === "number" && isFinite(t.duration) ? t.duration : 0;
  dur[i] = d > 0 ? d : 0;
  const dl = typeof t.deadline === "number" && isFinite(t.deadline) ? t.deadline : Infinity;
  deadline[i] = dl;
  const w = typeof t.weight === "number" && isFinite(t.weight) ? t.weight : 1;
  weight[i] = w > 0 ? w : 0;
  const depList = Array.isArray(t.deps) ? t.deps : [];
  const di: number[] = [];
  for (let k = 0; k < depList.length; k++) {
    const j = idToIdx.get(String(depList[k]));
    if (j !== undefined && j !== i) di.push(j);
  }
  deps[i] = di;
  for (const j of di) outs[j].push(i);
}

// Topological order (Kahn). If cycles exist, we still produce some order by breaking ties.
const indeg0 = new Array<number>(n);
for (let i = 0; i < n; i++) indeg0[i] = deps[i].length;
const q: number[] = [];
for (let i = 0; i < n; i++) if (indeg0[i] === 0) q.push(i);
let qh = 0;
const topo: number[] = [];
while (qh < q.length) {
  const u = q[qh++];
  topo.push(u);
  for (const v of outs[u]) {
    indeg0[v]--;
    if (indeg0[v] === 0) q.push(v);
  }
}
if (topo.length < n) {
  // Cycle or missing deps; append remaining nodes deterministically.
  const seen = new Array<boolean>(n).fill(false);
  for (const x of topo) seen[x] = true;
  for (let i = 0; i < n; i++) if (!seen[i]) topo.push(i);
}

// Compute a critical-path-like downstream length (including self).
const downLen = new Array<number>(n).fill(0);
for (let ii = topo.length - 1; ii >= 0; ii--) {
  const u = topo[ii];
  let best = 0;
  for (const v of outs[u]) if (downLen[v] > best) best = downLen[v];
  downLen[u] = dur[u] + best;
}

// Precompute some normalization stats.
let sumDur = 0;
let maxDur = 0;
let minDeadline = Infinity;
let maxDeadline = -Infinity;
let maxDown = 0;
let maxW = 0;
for (let i = 0; i < n; i++) {
  sumDur += dur[i];
  if (dur[i] > maxDur) maxDur = dur[i];
  if (deadline[i] < minDeadline) minDeadline = deadline[i];
  if (deadline[i] > maxDeadline && isFinite(deadline[i])) maxDeadline = deadline[i];
  if (downLen[i] > maxDown) maxDown = downLen[i];
  if (weight[i] > maxW) maxW = weight[i];
}
const avgDur = n ? sumDur / n : 1;
const dlRange = isFinite(minDeadline) && isFinite(maxDeadline) ? Math.max(1, maxDeadline - minDeadline) : 1;

// Scheduling simulation.
const remaining = new Array<number>(n);
for (let i = 0; i < n; i++) remaining[i] = deps[i].length;

const readyTime = new Array<number>(n).fill(0);
const finish = new Array<number>(n).fill(0);
const scheduled = new Array<boolean>(n).fill(false);

type TaskKey = { idx: number; key: number };

const ready = new MaxHeap<TaskKey>((a, b) => a.key > b.key);
const future = new MinHeap<{ t: number; idx: number }>((a, b) => a.t < b.t);

function priority(idx: number, now: number): number {
  const d = dur[idx];
  const dl = deadline[idx];
  const w = weight[idx];
  const rt = readyTime[idx];
  const start = now < rt ? rt : now;
  const estFinish = start + d;

  // Core ideas:
  // - Higher weight -> earlier
  // - Higher critical downstream length -> earlier
  // - Less slack (deadline - estFinish) -> earlier
  // - Slight bias to shorter tasks when equal (through ratio terms)
  const wN = maxW > 0 ? w / maxW : 0;
  const cpN = maxDown > 0 ? downLen[idx] / maxDown : 0;

  let slackN = 0;
  if (isFinite(dl)) {
    const slack = dl - estFinish;
    // negative slack => urgent; map to [0,1] where 1 is very urgent
    slackN = clamp01((avgDur - slack) / (avgDur + dlRange));
  }

  const shortN = avgDur > 0 ? clamp01(avgDur / (d + avgDur)) : 0.5;

  // Weighted blend; keep in a stable numeric range.
  const key = 3.0 * wN + 2.0 * slackN + 1.5 * cpN + 0.3 * shortN;
  return key;
}

for (let i = 0; i < n; i++) {
  if (remaining[i] === 0) {
    if (readyTime[i] <= 0) ready.push({ idx: i, key: priority(i, 0) });
    else future.push({ t: readyTime[i], idx: i });
  }
}

type MachineState = { t: number; m: number };
const machines = new MinHeap<MachineState>((a, b) => (a.t !== b.t ? a.t < b.t : a.m < b.m));
for (let m = 0; m < machineCount; m++) machines.push({ t: 0, m });

const schedule: ScheduleItem[] = [];
let scheduledCount = 0;

while (scheduledCount < n) {
  // If cycle/blocked, break it by releasing the smallest index unscheduled task.
  if (ready.size() === 0 && future.size() === 0) {
    let pick = -1;
    for (let i = 0; i < n; i++) {
      if (!scheduled[i]) {
        pick = i;
        break;
      }
    }
    if (pick >= 0) {
      // Force it ready (ignoring deps) to avoid infinite loop on invalid input.
      remaining[pick] = 0;
      readyTime[pick] = 0;
      ready.push({ idx: pick, key: priority(pick, 0) });
    } else {
      break;
    }
  }

  const ms = machines.pop()!;
  let now = ms.t;

  // Promote any tasks that become ready by time 'now'
  while (future.size()) {
    const top = future.peek()!;
    if (top.t > now) break;
    future.pop();
    if (!scheduled[top.idx] && remaining[top.idx] === 0) ready.push({ idx: top.idx, key: priority(top.idx, now) });
  }

  let chosen: number | undefined;
  if (ready.size()) {
    chosen = ready.pop()!.idx;
    // Ensure chosen is actually startable; skip stale heap entries.
    while (chosen !== undefined && (scheduled[chosen] || remaining[chosen] !== 0 || readyTime[chosen] > now)) {
      chosen = ready.size() ? ready.pop()!.idx : undefined;
    }
  }

  if (chosen === undefined) {
    // No task can start now; idle until the next future-ready task time (or keep machine time).
    const nxt = future.peek();
    if (nxt) now = Math.max(now, nxt.t);
    machines.push({ t: now, m: ms.m });
    continue;
  }

  const idx = chosen;
  const start = now < readyTime[idx] ? readyTime[idx] : now;
  const end = start + dur[idx];

  scheduled[idx] = true;
  finish[idx] = end;
  scheduledCount++;
  schedule.push({ id: ids[idx], start: start | 0, machine: ms.m });

  // Release dependents.
  for (const v of outs[idx]) {
    if (scheduled[v]) continue;
    remaining[v]--;
    if (remaining[v] <= 0) {
      let rt = 0;
      // max finish of deps; compute lazily by scanning deps[v] (usually small).
      for (const p of deps[v]) if (finish[p] > rt) rt = finish[p];
      readyTime[v] = rt;
      if (rt <= end) ready.push({ idx: v, key: priority(v, end) });
      else future.push({ t: rt, idx: v });
    }
  }

  machines.push({ t: end, m: ms.m });
}

// Output a schedule; the grader will validate precedence.
process.stdout.write(JSON.stringify({ schedule }));
