#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0, std430) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1, std430) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (id >= N) {
        return;
    }

    vec3 pos = inData[id * 2u + 0u].xyz;
    vec3 vel = inData[id * 2u + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    float sepR2 = sepRadius * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2 = cohRadius * cohRadius;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == id) {
            continue;
        }

        vec3 otherPos = inData[j * 2u + 0u].xyz;
        vec3 otherVel = inData[j * 2u + 1u].xyz;

        vec3 delta = pos - otherPos;
        float d2 = dot(delta, delta);
        if (d2 <= 1e-12) {
            continue; // zero-distance neighbor
        }

        if (d2 < sepR2) {
            // Repel more strongly when closer (approximately 1 / dist)
            float invDist = inversesqrt(d2);
            sepAccum += delta * invDist;
            sepCount++;
        }

        if (d2 < alignR2) {
            alignAccum += otherVel;
            alignCount++;
        }

        if (d2 < cohR2) {
            cohAccum += otherPos;
            cohCount++;
        }
    }

    vec3 sepForce = vec3(0.0);
    if (sepCount > 0u) {
        sepForce = sepAccum / float(sepCount);
    }

    vec3 alignForce = vec3(0.0);
    if (alignCount > 0u) {
        vec3 avgVel = alignAccum / float(alignCount);
        alignForce = avgVel - vel;
    }

    vec3 cohForce = vec3(0.0);
    if (cohCount > 0u) {
        vec3 avgPos = cohAccum / float(cohCount);
        cohForce = avgPos - pos;
    }

    vec3 accel = sepForce * sepWeight + alignForce * alignWeight + cohForce * cohWeight;

    // Boundary force: if a component exceeds boundSize*0.4, push toward center.
    float limit = boundSize * 0.4;
    float scale = boundSize * 0.5;
    if (pos.x > limit) {
        accel.x -= boundForce * ((pos.x - limit) / max(scale, 1e-6));
    } else if (pos.x < -limit) {
        accel.x += boundForce * ((-limit - pos.x) / max(scale, 1e-6));
    }
    if (pos.y > limit) {
        accel.y -= boundForce * ((pos.y - limit) / max(scale, 1e-6));
    } else if (pos.y < -limit) {
        accel.y += boundForce * ((-limit - pos.y) / max(scale, 1e-6));
    }
    if (pos.z > limit) {
        accel.z -= boundForce * ((pos.z - limit) / max(scale, 1e-6));
    } else if (pos.z < -limit) {
        accel.z += boundForce * ((-limit - pos.z) / max(scale, 1e-6));
    }

    vel += accel * dt;

    float speed2 = dot(vel, vel);
    if (speed2 > maxSpeed * maxSpeed) {
        float invSpeed = inversesqrt(speed2);
        vel *= maxSpeed * invSpeed;
    }

    pos += vel * dt;

    outData[id * 2u + 0u] = vec4(pos, 0.0);
    outData[id * 2u + 1u] = vec4(vel, 0.0);
}
