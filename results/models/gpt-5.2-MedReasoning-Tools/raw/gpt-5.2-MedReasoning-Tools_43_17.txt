struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float3 BrickPattern(float2 uv)
{
    float2 scaled = uv * float2(8.0, 16.0);
    float row = floor(scaled.y);
    if (fmod(row, 2.0) >= 1.0) {
        scaled.x += 0.5;
    }

    float2 cell = frac(scaled);
    float mortarX = 0.05;
    float mortarY = 0.1;

    bool isMortar =
        (cell.x < mortarX) || (cell.x > (1.0 - mortarX)) ||
        (cell.y < mortarY) || (cell.y > (1.0 - mortarY));

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);
    return isMortar ? mortarColor : brickColor;
}

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float shade = 0.4 + 0.6 * diffuse;

    float3 baseColor = BrickPattern(input.uv);
    float3 outColor = baseColor * shade;
    return float4(outColor, 1.0);
}
