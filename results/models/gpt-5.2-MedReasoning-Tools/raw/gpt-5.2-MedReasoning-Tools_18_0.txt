// Minimum Edge Cut (Global Min-Cut) solver for undirected unweighted graphs.
// Adaptive strategy:
//   - For smaller graphs: deterministic Stoer–Wagner (exact).
//   - For larger graphs: Karger–Stein randomized contraction (high probability),
//     with a small exact brute-force base case.
// Input can be huge; edges are stored either in memory or in a temp file.

struct FastScanner {
  static constexpr size_t BUFSZ = 1u << 20;
  unsigned char buf[BUFSZ];
  size_t idx = 0, size = 0;

  inline int readByte() {
    if (idx >= size) {
      size = std::fread(buf, 1, BUFSZ, stdin);
      idx = 0;
      if (size == 0) return -1;
    }
    return buf[idx++];
  }

  template <class T>
  bool readInt(T &out) {
    int c;
    do {
      c = readByte();
      if (c < 0) return false;
    } while (c <= ' ');  // skip whitespace
    bool neg = false;
    if (c == '-') {
      neg = true;
      c = readByte();
    }
    T val = 0;
    while (c > ' ') {
      val = static_cast<T>(val * 10 + (c - '0'));
      c = readByte();
    }
    out = neg ? -val : val;
    return true;
  }
};

struct SplitMix64 {
  uint64_t x;
  explicit SplitMix64(uint64_t seed) : x(seed) {}
  uint64_t next() {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
  }
  uint64_t operator()() { return next(); }
};

struct DSU {
  std::vector<int> parent;
  std::vector<int> sz;
  int comps;
  explicit DSU(int n) : parent(n), sz(n, 1), comps(n) {
    std::iota(parent.begin(), parent.end(), 0);
  }
  int find(int a) {
    while (parent[a] != a) {
      parent[a] = parent[parent[a]];
      a = parent[a];
    }
    return a;
  }
  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (sz[a] < sz[b]) std::swap(a, b);
    parent[b] = a;
    sz[a] += sz[b];
    comps--;
    return true;
  }
};

struct Edge {
  uint32_t u;
  uint32_t v;
};

struct IEdgeStore {
  virtual ~IEdgeStore() = default;
  virtual uint64_t m() const = 0;
  virtual Edge get(uint64_t i) const = 0;  // random access
  virtual void rewindIter() const = 0;
  virtual bool readNextBlock(std::vector<Edge> &out, size_t maxEdges) const = 0;
};

struct MemEdgeStore final : IEdgeStore {
  std::vector<Edge> edges;
  mutable uint64_t it = 0;
  explicit MemEdgeStore(uint64_t m) { edges.reserve(static_cast<size_t>(m)); }
  uint64_t m() const override { return static_cast<uint64_t>(edges.size()); }
  Edge get(uint64_t i) const override { return edges[static_cast<size_t>(i)]; }
  void rewindIter() const override { it = 0; }
  bool readNextBlock(std::vector<Edge> &out, size_t maxEdges) const override {
    out.clear();
    if (it >= edges.size()) return false;
    size_t take = std::min(maxEdges, static_cast<size_t>(edges.size() - it));
    out.insert(out.end(), edges.begin() + static_cast<size_t>(it),
               edges.begin() + static_cast<size_t>(it + take));
    it += take;
    return true;
  }
};

struct FileEdgeStore final : IEdgeStore {
  mutable std::FILE *f = nullptr;
  uint64_t cnt = 0;
  mutable uint64_t it = 0;
  explicit FileEdgeStore(std::FILE *file) : f(file) {}
  ~FileEdgeStore() override {
    if (f) std::fclose(f);
  }
  uint64_t m() const override { return cnt; }
  Edge get(uint64_t i) const override {
    Edge e{0, 0};
    const uint64_t off = i * sizeof(Edge);
#if defined(_MSC_VER)
    _fseeki64(f, static_cast<__int64>(off), SEEK_SET);
#else
    std::fseek(f, static_cast<long>(off), SEEK_SET);
#endif
    std::fread(&e, sizeof(Edge), 1, f);
    return e;
  }
  void rewindIter() const override {
    std::fflush(f);
#if defined(_MSC_VER)
    _fseeki64(f, 0, SEEK_SET);
#else
    std::fseek(f, 0, SEEK_SET);
#endif
    it = 0;
  }
  bool readNextBlock(std::vector<Edge> &out, size_t maxEdges) const override {
    out.clear();
    if (it >= cnt) return false;
    out.resize(maxEdges);
    size_t got = std::fread(out.data(), sizeof(Edge), maxEdges, f);
    out.resize(got);
    it += got;
    return got > 0;
  }
};

struct Graph {
  int n = 0;
  std::vector<std::pair<int, int>> edges;       // multiedges allowed
  std::vector<std::vector<int>> super;          // super[i] = original vertices
};

struct CutResult {
  int64_t cut = std::numeric_limits<int64_t>::max();
  std::vector<int> side;  // list of original vertices in one component
};

static CutResult streamingKarger(const IEdgeStore &store, int n, SplitMix64 &rng, int reps) {
  // Karger's contraction using only random edge access + DSU.
  // Uses O(n) memory; success is probabilistic.
  CutResult best;
  best.cut = std::numeric_limits<int64_t>::max();
  const uint64_t m = store.m();
  if (n <= 1 || m == 0) {
    best.cut = 0;
    best.side = {0};
    return best;
  }

  std::vector<Edge> block;
  for (int r = 0; r < reps; ++r) {
    DSU dsu(n);
    while (dsu.comps > 2) {
      const uint64_t idx = rng() % m;
      Edge e = store.get(idx);
      int u = static_cast<int>(e.u);
      int v = static_cast<int>(e.v);
      if (u == v) continue;
      dsu.unite(u, v);
    }

    // Choose side as the component containing vertex 0.
    const int root0 = dsu.find(0);
    std::vector<unsigned char> inSide(static_cast<size_t>(n), 0);
    for (int i = 0; i < n; ++i) {
      if (dsu.find(i) == root0) inSide[static_cast<size_t>(i)] = 1;
    }

    // Compute cut size by streaming all edges.
    int64_t cut = 0;
    store.rewindIter();
    while (store.readNextBlock(block, 1u << 20)) {
      for (const auto &ed : block) {
        if (ed.u >= static_cast<uint32_t>(n) || ed.v >= static_cast<uint32_t>(n)) continue;
        if (inSide[static_cast<size_t>(ed.u)] ^ inSide[static_cast<size_t>(ed.v)]) cut++;
      }
    }

    if (cut < best.cut) {
      best.cut = cut;
      best.side.clear();
      for (int i = 0; i < n; ++i) {
        if (inSide[static_cast<size_t>(i)]) best.side.push_back(i);
      }
    }
  }
  return best;
}

static CutResult bruteForceMinCut(const Graph &g) {
  // Exact for small n by enumerating subsets; force vertex 0 on "left" side.
  const int n = g.n;
  CutResult best;
  if (n <= 1) {
    best.cut = 0;
    best.side.clear();
    return best;
  }
  const uint64_t totalMasks = 1ULL << n;
  for (uint64_t mask = 1; mask < totalMasks - 1; ++mask) {
    if ((mask & 1ULL) == 0) continue;  // ensure non-empty and avoid duplicates
    int64_t c = 0;
    for (const auto &e : g.edges) {
      const bool a = (mask >> e.first) & 1ULL;
      const bool b = (mask >> e.second) & 1ULL;
      c += (a != b);
    }
    if (c < best.cut) {
      best.cut = c;
      best.side.clear();
      for (int i = 0; i < n; ++i) {
        if ((mask >> i) & 1ULL) {
          best.side.insert(best.side.end(), g.super[i].begin(), g.super[i].end());
        }
      }
    }
  }
  return best;
}

static Graph contractTo(const Graph &g, int target, SplitMix64 &rng) {
  const int n = g.n;
  DSU dsu(n);
  std::vector<std::vector<int>> groups = g.super;  // will be merged by DSU root

  if (n <= target) return g;

  // Contract edges uniformly at random (with replacement) until #components == target.
  while (dsu.comps > target) {
    const uint64_t idx = rng() % static_cast<uint64_t>(g.edges.size());
    int a = g.edges[static_cast<size_t>(idx)].first;
    int b = g.edges[static_cast<size_t>(idx)].second;
    a = dsu.find(a);
    b = dsu.find(b);
    if (a == b) continue;
    // Union by size and merge groups: append smaller into larger.
    if (dsu.sz[a] < dsu.sz[b]) std::swap(a, b);
    dsu.parent[b] = a;
    dsu.sz[a] += dsu.sz[b];
    dsu.comps--;
    if (groups[a].size() < groups[b].size()) groups[a].swap(groups[b]);
    groups[a].insert(groups[a].end(), groups[b].begin(), groups[b].end());
    groups[b].clear();
  }

  // Reindex components to 0..k-1.
  std::vector<int> rootToIdx(n, -1);
  int k = 0;
  for (int i = 0; i < n; ++i) {
    if (dsu.find(i) == i) rootToIdx[i] = k++;
  }
  Graph out;
  out.n = k;
  out.super.assign(k, {});
  for (int r = 0; r < n; ++r) {
    if (rootToIdx[r] >= 0) out.super[rootToIdx[r]] = std::move(groups[r]);
  }

  out.edges.reserve(g.edges.size());
  for (const auto &e : g.edges) {
    int a = rootToIdx[dsu.find(e.first)];
    int b = rootToIdx[dsu.find(e.second)];
    if (a == b) continue;
    out.edges.emplace_back(a, b);
  }
  return out;
}

static CutResult kargerStein(const Graph &g, SplitMix64 &rng) {
  const int n = g.n;
  if (n <= 12) return bruteForceMinCut(g);

  const double invSqrt2 = 1.0 / std::sqrt(2.0);
  int t = static_cast<int>(std::ceil(n * invSqrt2));
  if (t < 2) t = 2;

  Graph g1 = contractTo(g, t, rng);
  Graph g2 = contractTo(g, t, rng);

  CutResult c1 = kargerStein(g1, rng);
  CutResult c2 = kargerStein(g2, rng);
  return (c1.cut <= c2.cut) ? std::move(c1) : std::move(c2);
}

static CutResult stoerWagnerExact(int n, const IEdgeStore &store) {
  // Deterministic Stoer–Wagner using adjacency hash-maps.
  // Exact; intended for moderate sizes.
  std::vector<std::unordered_map<int, int>> adj(n);

  // Build adjacency lists (with parallel edges).
  std::vector<Edge> block;
  store.rewindIter();
  while (store.readNextBlock(block, 1u << 20)) {
    for (const auto &e : block) {
      int u = static_cast<int>(e.u);
      int v = static_cast<int>(e.v);
      if (u == v) continue;
      adj[u][v] += 1;
      adj[v][u] += 1;
    }
  }

  std::vector<int> active(n);
  std::iota(active.begin(), active.end(), 0);

  std::vector<std::vector<int>> super(n);
  for (int i = 0; i < n; ++i) super[i] = {i};

  CutResult best;
  best.cut = std::numeric_limits<int64_t>::max();

  // We'll maintain "merged" marker by keeping active list and swapping removed vertex.
  for (int phaseN = n; phaseN > 1; --phaseN) {
    // Maximum adjacency search (MA-ordering).
    std::vector<int64_t> w(n, 0);
    std::vector<char> inA(n, 0);

    int prev = -1;
    int last = -1;
    int start = active[0];
    last = start;
    inA[start] = 1;

    // Initialize weights from start.
    for (const auto &kv : adj[start]) w[kv.first] += kv.second;

    for (int it = 1; it < phaseN; ++it) {
      int sel = -1;
      int64_t bestW = -1;
      for (int i = 0; i < phaseN; ++i) {
        int v = active[i];
        if (!inA[v] && w[v] > bestW) {
          bestW = w[v];
          sel = v;
        }
      }
      prev = last;
      last = sel;
      inA[last] = 1;
      for (const auto &kv : adj[last]) {
        if (!inA[kv.first]) w[kv.first] += kv.second;
      }
    }

    // Cut-of-the-phase is separating "last" from rest, with weight w[last].
    if (w[last] < best.cut) {
      best.cut = w[last];
      best.side = super[last];
    }

    // Merge prev and last into prev. Update only incident edges (no global scans).
    super[prev].insert(super[prev].end(), super[last].begin(), super[last].end());
    super[last].clear();

    // For each neighbor x of 'last' with weight wlx, add it to prev-x and fix x adjacency.
    for (const auto &kv : adj[last]) {
      const int x = kv.first;
      const int wlx = kv.second;
      if (x == prev) continue;
      adj[prev][x] += wlx;
      adj[x][prev] += wlx;
      // Remove references to 'last' from x.
      auto it2 = adj[x].find(last);
      if (it2 != adj[x].end()) adj[x].erase(it2);
    }
    // Remove self-loop and clear last.
    adj[prev].erase(prev);
    adj[last].clear();
    // Remove prev<->last edge if present (now merged).
    adj[prev].erase(last);
    adj[last].erase(prev);

    // Remove "last" from active set.
    auto it = std::find(active.begin(), active.begin() + phaseN, last);
    if (it != active.begin() + phaseN) {
      std::swap(*it, active[phaseN - 1]);
    }
  }

  // Convert best.side (in terms of original vertices within this routine's numbering)
  // into vertex list; here original == 0..n-1.
  return best;
}

static uint64_t pickSeed() {
  uint64_t s = 0;
  s ^= static_cast<uint64_t>(std::chrono::high_resolution_clock::now().time_since_epoch().count());
  s ^= (static_cast<uint64_t>(std::uintptr_t(&s)) << 1);
  return s;
}

}  // namespace

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  FastScanner fs;
  int64_t N64 = 0, M64 = 0;
  if (!fs.readInt(N64) || !fs.readInt(M64)) return 0;
  if (N64 <= 0 || M64 < 0) return 0;
  const int N = static_cast<int>(N64);
  const uint64_t M = static_cast<uint64_t>(M64);

  std::unique_ptr<IEdgeStore> store;
  MemEdgeStore *memStore = nullptr;
  std::FILE *tmp = nullptr;

  // Try to store edges in memory; if allocation fails, fall back to temp file.
  try {
    auto ms = std::make_unique<MemEdgeStore>(M);
    memStore = ms.get();
    store = std::move(ms);
  } catch (const std::bad_alloc &) {
    tmp = std::tmpfile();
    if (!tmp) return 0;
    store = std::make_unique<FileEdgeStore>(tmp);
  }

  // Read edges.
  for (uint64_t i = 0; i < M; ++i) {
    int64_t u64 = 0, v64 = 0;
    if (!fs.readInt(u64) || !fs.readInt(v64)) break;
    uint32_t u = static_cast<uint32_t>(u64);
    uint32_t v = static_cast<uint32_t>(v64);
    if (memStore) {
      memStore->edges.push_back(Edge{u, v});
    } else {
      Edge e{u, v};
      auto *fs2 = static_cast<FileEdgeStore *>(store.get());
      std::fwrite(&e, sizeof(Edge), 1, fs2->f);
      fs2->cnt++;
    }
  }

  // Decide algorithm.
  // 1) If N small-ish and we can afford deterministic exact Stoer-Wagner, use it.
  // 2) Otherwise, run Karger-Stein multiple times for high probability.
  CutResult best;

  bool usedStoerWagner = false;
  const bool storeIsFile = (dynamic_cast<FileEdgeStore *>(store.get()) != nullptr);

  if (N <= 2500 && !storeIsFile) {
    // For this threshold, adjacency-list Stoer-Wagner is usually fine.
    usedStoerWagner = true;
    best = stoerWagnerExact(N, *store);
  } else {
    SplitMix64 rng(pickSeed());
    const int reps = std::max(8, static_cast<int>(4 * std::ceil(std::log2(std::max(2, N)))));

    if (storeIsFile || store->m() > 60000000ULL) {
      // Streaming-friendly fallback: no need to hold all edges in RAM.
      best = streamingKarger(*store, N, rng, reps);
    } else {
      // Karger-Stein on an in-memory multigraph.
      Graph g;
      g.n = N;
      g.super.resize(N);
      for (int i = 0; i < N; ++i) g.super[i] = {i};
      g.edges.reserve(static_cast<size_t>(store->m()));

      std::vector<Edge> block;
      store->rewindIter();
      while (store->readNextBlock(block, 1u << 20)) {
        for (const auto &e : block) {
          int u = static_cast<int>(e.u);
          int v = static_cast<int>(e.v);
          if (u == v) continue;
          g.edges.emplace_back(u, v);
        }
      }

      best.cut = std::numeric_limits<int64_t>::max();
      for (int r = 0; r < reps; ++r) {
        CutResult cur = kargerStein(g, rng);
        if (cur.cut < best.cut) best = std::move(cur);
      }
    }
  }

  // Build membership array for best.side.
  std::vector<unsigned char> inSide(static_cast<size_t>(N), 0);
  for (int v : best.side) {
    if (0 <= v && v < N) inSide[static_cast<size_t>(v)] = 1;
  }
  // Ensure non-trivial cut; if algorithm produced empty/full side, fix by trivial split.
  size_t sideCount = 0;
  for (auto b : inSide) sideCount += (b != 0);
  if (sideCount == 0 || sideCount == static_cast<size_t>(N)) {
    std::fill(inSide.begin(), inSide.end(), 0);
    inSide[0] = 1;
  }

  // Collect cut edges from the ORIGINAL input edge list.
  std::vector<std::pair<uint32_t, uint32_t>> cutEdges;
  cutEdges.reserve(static_cast<size_t>(std::min<int64_t>(best.cut, 10000000)));
  std::vector<Edge> block;
  store->rewindIter();
  while (store->readNextBlock(block, 1u << 20)) {
    for (const auto &e : block) {
      uint32_t u = e.u, v = e.v;
      if (u >= static_cast<uint32_t>(N) || v >= static_cast<uint32_t>(N)) continue;
      if (inSide[static_cast<size_t>(u)] ^ inSide[static_cast<size_t>(v)]) {
        cutEdges.emplace_back(u, v);
      }
    }
  }

  // Output.
  std::cout << cutEdges.size() << "\n";
  for (const auto &e : cutEdges) {
    std::cout << e.first << " " << e.second << "\n";
  }

  (void)usedStoerWagner;
  return 0;
}
