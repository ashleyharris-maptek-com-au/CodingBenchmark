using System;
using System.Buffers;
using System.Collections.Generic;
using System.Diagnostics;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _pos;
        private readonly System.IO.Stream _stream = Console.OpenStandardInput();

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c != -1);
            if (c == -1) return int.MinValue;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private sealed class Graph
    {
        public readonly int N;
        public readonly int[] Off;
        public readonly int[] To;
        public readonly int[] Deg;

        public Graph(int n, int[] off, int[] to)
        {
            N = n;
            Off = off;
            To = to;
            Deg = new int[n];
            for (int i = 0; i < n; i++) Deg[i] = off[i + 1] - off[i];
        }

        public bool Adjacent(int u, int v)
        {
            int lo = Off[u], hi = Off[u + 1] - 1;
            while (lo <= hi)
            {
                int mid = (lo + hi) >>> 1;
                int x = To[mid];
                if (x == v) return true;
                if (x < v) lo = mid + 1;
                else hi = mid - 1;
            }
            return false;
        }
    }

    private static long _deadlineTicks;
    private static volatile int _bestSize;
    private static int[] _bestClique = Array.Empty<int>();
    private static readonly object _bestLock = new object();

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        _deadlineTicks = Stopwatch.GetTimestamp() + (long)(Stopwatch.Frequency * 290.0); // ~4m50s safety

        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine("0\n");
            return;
        }

        int[] eu = new int[m];
        int[] ev = new int[m];
        int[] deg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v)
            {
                eu[i] = 0; ev[i] = 0;
                continue;
            }
            eu[i] = u; ev[i] = v;
            deg[u]++; deg[v]++;
        }

        int[] off = new int[n + 1];
        for (int i = 0; i < n; i++) off[i + 1] = off[i] + deg[i];
        int[] cur = new int[n];
        Array.Copy(off, 0, cur, 0, n);
        int[] to = new int[off[n]];
        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            if (u == v) continue;
            int pu = cur[u]++; to[pu] = v;
            int pv = cur[v]++; to[pv] = u;
        }

        // Sort adjacency lists for binary search.
        for (int v = 0; v < n; v++)
        {
            int start = off[v];
            int len = off[v + 1] - start;
            if (len > 1) Array.Sort(to, start, len);
        }

        var g = new Graph(n, off, to);

        // Quick heuristic: greedy by degree, then try to expand.
        int[] initial = GreedyHeuristic(g);
        UpdateBest(initial);

        // Exact search based on degeneracy ordering (good for large sparse graphs).
        ComputeDegeneracyOrdering(g, out int[] order, out int[] rank, out int[] core, out int degeneracy);

        // If degeneracy is huge, exact search may explode; still run but with time cap.
        RunExactSearch(g, order, rank, core, degeneracy);

        // Output best found (always a clique). If time cap hit, this may be non-optimal.
        int[] best;
        lock (_bestLock)
        {
            best = new int[_bestClique.Length];
            Array.Copy(_bestClique, best, best.Length);
        }

        Array.Sort(best);
        Console.WriteLine(best.Length);
        if (best.Length == 0)
        {
            Console.WriteLine();
            return;
        }
        var sb = new StringBuilder(best.Length * 2);
        sb.Append(best[0]);
        for (int i = 1; i < best.Length; i++)
        {
            sb.Append(' ');
            sb.Append(best[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static bool TimeUp() => Stopwatch.GetTimestamp() >= _deadlineTicks;

    private static void UpdateBest(int[] clique)
    {
        int sz = clique.Length;
        while (true)
        {
            int cur = _bestSize;
            if (sz <= cur) return;
            if (Interlocked.CompareExchange(ref _bestSize, sz, cur) == cur) break;
        }
        lock (_bestLock)
        {
            if (clique.Length > _bestClique.Length)
            {
                _bestClique = new int[clique.Length];
                Array.Copy(clique, _bestClique, clique.Length);
            }
        }
    }

    private static int[] GreedyHeuristic(Graph g)
    {
        int n = g.N;
        // Start from a high-degree vertex (O(n)).
        int start = 0;
        int bestDeg0 = -1;
        for (int v = 0; v < n; v++)
        {
            int d = g.Deg[v];
            if (d > bestDeg0) { bestDeg0 = d; start = v; }
        }

        var clique = new List<int>(64) { start };
        int degStart = g.Off[start + 1] - g.Off[start];
        if (degStart == 0) return clique.ToArray();

        var cand = new List<int>(degStart);
        for (int e = g.Off[start]; e < g.Off[start + 1]; e++) cand.Add(g.To[e]);

        // Greedy: pick vertex with highest degree within candidate set, intersect candidates.
        while (cand.Count > 0 && !TimeUp())
        {
            int bestIdx = 0;
            int bestDeg = -1;
            for (int i = 0; i < cand.Count; i++)
            {
                int v = cand[i];
                int d = g.Deg[v];
                if (d > bestDeg) { bestDeg = d; bestIdx = i; }
            }

            int pick = cand[bestIdx];
            clique.Add(pick);

            int write = 0;
            for (int i = 0; i < cand.Count; i++)
            {
                int u = cand[i];
                if (u == pick) continue;
                if (g.Adjacent(pick, u)) cand[write++] = u;
            }
            if (write < cand.Count) cand.RemoveRange(write, cand.Count - write);
        }

        return clique.ToArray();
    }

    private static void ComputeDegeneracyOrdering(Graph g, out int[] order, out int[] rank, out int[] core, out int degeneracy)
    {
        int n = g.N;
        int[] d = new int[n];
        int maxDeg = 0;
        for (int i = 0; i < n; i++)
        {
            int di = g.Deg[i];
            d[i] = di;
            if (di > maxDeg) maxDeg = di;
        }

        int[] bin = new int[maxDeg + 1];
        for (int i = 0; i < n; i++) bin[d[i]]++;
        int start = 0;
        for (int i = 0; i <= maxDeg; i++)
        {
            int num = bin[i];
            bin[i] = start;
            start += num;
        }

        int[] vert = new int[n];
        int[] pos = new int[n];
        for (int v = 0; v < n; v++)
        {
            int p = bin[d[v]]++;
            pos[v] = p;
            vert[p] = v;
        }

        // Restore bin to point to starts.
        for (int i = maxDeg; i >= 1; i--) bin[i] = bin[i - 1];
        bin[0] = 0;

        order = new int[n];
        core = new int[n];
        rank = new int[n];
        degeneracy = 0;

        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            order[i] = v;
            rank[v] = i;
            int dv = d[v];
            core[v] = dv;
            if (dv > degeneracy) degeneracy = dv;

            for (int e = g.Off[v]; e < g.Off[v + 1]; e++)
            {
                int u = g.To[e];
                if (d[u] <= dv) continue;
                int du = d[u];
                int pu = pos[u];
                int pw = bin[du];
                int w = vert[pw];
                if (u != w)
                {
                    vert[pu] = w;
                    pos[w] = pu;
                    vert[pw] = u;
                    pos[u] = pw;
                }
                bin[du]++;
                d[u] = du - 1;
            }
        }
    }

    private static void RunExactSearch(Graph g, int[] order, int[] rank, int[] core, int degeneracy)
    {
        int n = g.N;
        int startBest = _bestSize;

        // Parallelize by splitting the top-level vertices.
        int workers = Math.Max(1, Math.Min(Environment.ProcessorCount, 12));
        var opts = new ParallelOptions { MaxDegreeOfParallelism = workers };

        // Process in reverse ordering (later vertices first tends to give larger cliques earlier).
        Parallel.For(0, n, opts, idx =>
        {
            int i = n - 1 - idx;
            if (TimeUp()) return;
            int v = order[i];

            // Upper bound: clique containing v is at most core[v] + 1.
            int localBest = _bestSize;
            if (core[v] + 1 <= localBest) return;

            // Candidates: neighbors after v in degeneracy ordering.
            int degv = g.Off[v + 1] - g.Off[v];
            if (degv == 0)
            {
                if (localBest < 1) UpdateBest(new[] { v });
                return;
            }

            int[] cand = ArrayPool<int>.Shared.Rent(degv);
            int candCount = 0;
            for (int e = g.Off[v]; e < g.Off[v + 1]; e++)
            {
                int u = g.To[e];
                if (rank[u] > rank[v] && core[u] + 1 > localBest) cand[candCount++] = u;
            }
            if (candCount == 0)
            {
                ArrayPool<int>.Shared.Return(cand);
                if (localBest < 1) UpdateBest(new[] { v });
                return;
            }

            // Small optimisation: if 1 + candCount <= best, skip.
            if (1 + candCount <= _bestSize)
            {
                ArrayPool<int>.Shared.Return(cand);
                return;
            }

            int[] clique = new int[Math.Min(degeneracy + 1, 1 + candCount)];
            clique[0] = v;
            Expand(g, clique, 1, cand, candCount);

            ArrayPool<int>.Shared.Return(cand);
        });
    }

    private static void Expand(Graph g, int[] clique, int cliqueLen, int[] cand, int candCount)
    {
        if (TimeUp()) return;
        if (candCount == 0)
        {
            if (cliqueLen > _bestSize)
            {
                int[] sol = new int[cliqueLen];
                Array.Copy(clique, sol, cliqueLen);
                UpdateBest(sol);
            }
            return;
        }

        // Coloring-based bound and ordering.
        int[] order = new int[candCount];
        int[] colors = new int[candCount];
        ColorSort(g, cand, candCount, order, colors);

        // Explore in reverse (higher color bound first).
        for (int i = candCount - 1; i >= 0; i--)
        {
            if (TimeUp()) return;
            int bound = cliqueLen + colors[i];
            if (bound <= _bestSize) return;

            int v = order[i];
            clique[cliqueLen] = v;

            // Intersect candidates with N(v), considering only order[0..i-1].
            int[] next = new int[i];
            int nextCount = 0;
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (g.Adjacent(v, u)) next[nextCount++] = u;
            }

            Expand(g, clique, cliqueLen + 1, next, nextCount);
        }
    }

    // Greedy sequential coloring by repeatedly extracting independent sets.
    // Produces 'order' in nondecreasing color, and colors[i] = colorIndex+1 for order[i].
    private static void ColorSort(Graph g, int[] cand, int candCount, int[] order, int[] colors)
    {
        // Work on a local copy; candCount is usually small (<= degeneracy).
        int[] rem = new int[candCount];
        Array.Copy(cand, rem, candCount);

        // Slightly better coloring bound: process high-degree first.
        Array.Sort(rem, 0, candCount, Comparer<int>.Create((a, b) => g.Deg[b].CompareTo(g.Deg[a])));

        int outCount = 0;
        int color = 0;
        int[] selected = new int[rem.Length];

        int remCount = candCount;
        while (remCount > 0)
        {
            color++;
            int selCount = 0;
            int nextRemCount = 0;

            // Single pass: either put v in this color class, or keep it for future colors.
            for (int i = 0; i < remCount; i++)
            {
                int v = rem[i];
                bool ok = true;
                for (int j = 0; j < selCount; j++)
                {
                    if (g.Adjacent(v, selected[j])) { ok = false; break; }
                }
                if (ok)
                {
                    selected[selCount++] = v;
                    order[outCount] = v;
                    colors[outCount] = color;
                    outCount++;
                }
                else
                {
                    rem[nextRemCount++] = v;
                }
            }

            remCount = nextRemCount;
        }
    }
}
