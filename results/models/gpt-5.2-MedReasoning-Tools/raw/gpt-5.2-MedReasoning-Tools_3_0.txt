use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f32,
    y: f32,
}

struct CsrGraph {
    n: usize,
    offs: Vec<u32>,
    adj: Vec<u32>,
}

impl CsrGraph {
    #[inline]
    fn neighbors(&self, u: usize) -> &[u32] {
        let a = self.offs[u] as usize;
        let b = self.offs[u + 1] as usize;
        &self.adj[a..b]
    }
}

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    #[inline]
    fn skip_ws(&mut self) {
        while self.idx < self.buf.len() {
            let b = unsafe { *self.buf.get_unchecked(self.idx) };
            if b > b' ' {
                break;
            }
            self.idx += 1;
        }
    }

    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        self.skip_ws();
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut v: usize = 0;
        while self.idx < self.buf.len() {
            let b = unsafe { *self.buf.get_unchecked(self.idx) };
            if b <= b' ' {
                break;
            }
            v = v * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        Some(v)
    }
}

#[inline]
fn hash32(mut x: u32) -> u32 {
    // xorshift / avalanching-ish
    x ^= x >> 16;
    x = x.wrapping_mul(0x7feb352d);
    x ^= x >> 15;
    x = x.wrapping_mul(0x846ca68b);
    x ^= x >> 16;
    x
}

fn build_graph(n: usize, edges: &[(u32, u32)]) -> CsrGraph {
    let mut deg = vec![0u32; n];
    for &(a, b) in edges {
        if a as usize >= n || b as usize >= n {
            continue;
        }
        if a == b {
            continue;
        }
        deg[a as usize] = deg[a as usize].saturating_add(1);
        deg[b as usize] = deg[b as usize].saturating_add(1);
    }
    let mut offs = vec![0u32; n + 1];
    for i in 0..n {
        offs[i + 1] = offs[i].saturating_add(deg[i]);
    }
    let m2 = offs[n] as usize;
    let mut adj = vec![0u32; m2];
    let mut cur = offs[..n].to_vec();

    for &(a, b) in edges {
        if a as usize >= n || b as usize >= n {
            continue;
        }
        if a == b {
            continue;
        }
        let ia = a as usize;
        let ib = b as usize;

        let pa = cur[ia] as usize;
        adj[pa] = b;
        cur[ia] += 1;

        let pb = cur[ib] as usize;
        adj[pb] = a;
        cur[ib] += 1;
    }

    // Sort neighbor lists (helps with determinism and cache-locality a bit)
    for u in 0..n {
        let a = offs[u] as usize;
        let b = offs[u + 1] as usize;
        adj[a..b].sort_unstable();
    }

    CsrGraph { n, offs, adj }
}

fn connected_components(g: &CsrGraph) -> Vec<Vec<u32>> {
    let mut comps: Vec<Vec<u32>> = Vec::new();
    let mut vis = vec![0u8; g.n];
    let mut q: VecDeque<u32> = VecDeque::new();

    for s in 0..g.n {
        if vis[s] != 0 {
            continue;
        }
        vis[s] = 1;
        q.clear();
        q.push_back(s as u32);
        let mut comp: Vec<u32> = Vec::new();
        while let Some(u32u) = q.pop_front() {
            comp.push(u32u);
            let u = u32u as usize;
            for &v in g.neighbors(u) {
                let vi = v as usize;
                if vis[vi] == 0 {
                    vis[vi] = 1;
                    q.push_back(v);
                }
            }
        }
        comps.push(comp);
    }

    comps
}

fn layout_small_force(g: &CsrGraph, nodes: &[u32], pos: &mut [Pt]) {
    let n = nodes.len();
    if n <= 1 {
        return;
    }

    // Map global node id -> local index
    let mut local = vec![u32::MAX; g.n];
    for (i, &u) in nodes.iter().enumerate() {
        local[u as usize] = i as u32;
    }

    // Init on a circle with light deterministic jitter
    let radius = (n as f32).sqrt().max(1.0);
    for (i, &u) in nodes.iter().enumerate() {
        let t = (i as f32) * (std::f32::consts::TAU / n as f32);
        let h = hash32(u.wrapping_mul(2654435761));
        let jx = ((h & 1023) as f32) / 1023.0 - 0.5;
        let jy = (((h >> 10) & 1023) as f32) / 1023.0 - 0.5;
        pos[u as usize] = Pt {
            x: radius * t.cos() + 0.1 * jx,
            y: radius * t.sin() + 0.1 * jy,
        };
    }

    // Build local edge list (dedup-ish by u<v and neighbor scan)
    let mut edges: Vec<(u32, u32)> = Vec::new();
    edges.reserve(n * 2);
    for &u32u in nodes {
        let u = u32u as usize;
        let lu = local[u] as u32;
        for &v32 in g.neighbors(u) {
            let v = v32 as usize;
            let lv = local[v];
            if lv == u32::MAX {
                continue;
            }
            if (lu as u32) < lv {
                edges.push((lu as u32, lv));
            }
        }
    }

    // Fruchterman-Reingold with O(n^2) repulsion (n <= ~2000)
    let area = (n as f32).max(1.0).powf(1.15);
    let k = (area / (n as f32)).sqrt().max(1e-3);

    let iters = if n <= 80 {
        250
    } else if n <= 250 {
        140
    } else if n <= 800 {
        80
    } else {
        50
    };

    let mut disp = vec![Pt::default(); n];
    let mut temperature = (k * 2.0).min(10.0);

    for _ in 0..iters {
        for d in disp.iter_mut() {
            *d = Pt { x: 0.0, y: 0.0 };
        }

        // Repulsion
        for i in 0..n {
            let ui = nodes[i] as usize;
            let pi = pos[ui];
            for j in (i + 1)..n {
                let uj = nodes[j] as usize;
                let pj = pos[uj];
                let mut dx = pi.x - pj.x;
                let mut dy = pi.y - pj.y;
                let dist2 = dx * dx + dy * dy + 1e-6;
                let inv = 1.0 / dist2.sqrt();
                dx *= inv;
                dy *= inv;
                let f = (k * k) * inv;
                disp[i].x += dx * f;
                disp[i].y += dy * f;
                disp[j].x -= dx * f;
                disp[j].y -= dy * f;
            }
        }

        // Attraction along edges
        for &(a, b) in &edges {
            let ia = a as usize;
            let ib = b as usize;
            let ua = nodes[ia] as usize;
            let ub = nodes[ib] as usize;
            let pa = pos[ua];
            let pb = pos[ub];
            let mut dx = pa.x - pb.x;
            let mut dy = pa.y - pb.y;
            let dist = (dx * dx + dy * dy + 1e-6).sqrt();
            dx /= dist;
            dy /= dist;
            let f = (dist * dist) / k;
            disp[ia].x -= dx * f;
            disp[ia].y -= dy * f;
            disp[ib].x += dx * f;
            disp[ib].y += dy * f;
        }

        // Apply displacement with cooling
        for i in 0..n {
            let u = nodes[i] as usize;
            let mut dx = disp[i].x;
            let mut dy = disp[i].y;
            let len = (dx * dx + dy * dy + 1e-9).sqrt();
            dx = dx / len * len.min(temperature);
            dy = dy / len * len.min(temperature);
            pos[u].x += dx;
            pos[u].y += dy;
        }

        temperature *= 0.95;
        if temperature < 1e-3 {
            break;
        }
    }
}

fn layout_large_layered(g: &CsrGraph, nodes: &[u32], pos: &mut [Pt]) {
    let n = nodes.len();
    if n == 0 {
        return;
    }
    if n == 1 {
        pos[nodes[0] as usize] = Pt { x: 0.0, y: 0.0 };
        return;
    }

    // Map global node id -> local index
    let mut local = vec![u32::MAX; g.n];
    for (i, &u) in nodes.iter().enumerate() {
        local[u as usize] = i as u32;
    }

    // Pick a root: highest degree in component (fast heuristic)
    let mut root = nodes[0] as usize;
    let mut best_deg = 0usize;
    for &u32u in nodes {
        let u = u32u as usize;
        let d = g.neighbors(u).len();
        if d > best_deg {
            best_deg = d;
            root = u;
        }
    }

    // BFS levels on the component
    let mut level = vec![u32::MAX; n];
    let mut q: VecDeque<u32> = VecDeque::new();
    let rloc = local[root] as usize;
    level[rloc] = 0;
    q.push_back(root as u32);

    while let Some(u32u) = q.pop_front() {
        let u = u32u as usize;
        let lu = local[u] as usize;
        let du = level[lu];
        for &v32 in g.neighbors(u) {
            let v = v32 as usize;
            let lv = local[v];
            if lv == u32::MAX {
                continue;
            }
            let lvi = lv as usize;
            if level[lvi] == u32::MAX {
                level[lvi] = du + 1;
                q.push_back(v32);
            }
        }
    }

    // Bucket nodes by level
    let mut max_level = 0u32;
    for &lv in &level {
        if lv != u32::MAX {
            max_level = max_level.max(lv);
        }
    }
    let layers = (max_level as usize) + 1;
    let mut buckets: Vec<Vec<u32>> = vec![Vec::new(); layers];
    for (i, &u32u) in nodes.iter().enumerate() {
        let lv = level[i];
        let lv = if lv == u32::MAX { max_level + 1 } else { lv };
        if (lv as usize) >= buckets.len() {
            buckets.push(Vec::new());
        }
        buckets[lv as usize].push(u32u);
    }

    // Initial order inside each layer: by hashed id (deterministic)
    for b in buckets.iter_mut() {
        b.sort_unstable_by(|&a, &b| hash32(a).cmp(&hash32(b)));
    }

    // A couple of barycenter sweeps to reduce crossings
    let sweeps = 2;
    let mut x_index = vec![0f32; g.n];
    for _ in 0..sweeps {
        // forward
        for l in 1..buckets.len() {
            let prev = &buckets[l - 1];
            for (i, &u) in prev.iter().enumerate() {
                x_index[u as usize] = i as f32;
            }
            let cur = &mut buckets[l];
            cur.sort_unstable_by(|&a, &b| {
                let ba = barycenter(g, a, &local, &x_index);
                let bb = barycenter(g, b, &local, &x_index);
                ba.partial_cmp(&bb).unwrap_or(Ordering::Equal)
            });
        }
        // backward
        for l in (0..buckets.len().saturating_sub(1)).rev() {
            let next = &buckets[l + 1];
            for (i, &u) in next.iter().enumerate() {
                x_index[u as usize] = i as f32;
            }
            let cur = &mut buckets[l];
            cur.sort_unstable_by(|&a, &b| {
                let ba = barycenter(g, a, &local, &x_index);
                let bb = barycenter(g, b, &local, &x_index);
                ba.partial_cmp(&bb).unwrap_or(Ordering::Equal)
            });
        }
    }

    // Emit coordinates: x by layer, y by within-layer order
    let x_gap = 1.5f32;
    let y_gap = 1.0f32;
    for (l, layer_nodes) in buckets.iter().enumerate() {
        let y0 = -0.5 * (layer_nodes.len() as f32 - 1.0) * y_gap;
        for (i, &u) in layer_nodes.iter().enumerate() {
            pos[u as usize] = Pt {
                x: (l as f32) * x_gap,
                y: y0 + (i as f32) * y_gap,
            };
        }
    }

    // Clear local map (not strictly necessary)
    for &u in nodes {
        local[u as usize] = u32::MAX;
    }
}

#[inline]
fn barycenter(g: &CsrGraph, u: u32, local: &[u32], x_index: &[f32]) -> f32 {
    let ui = u as usize;
    let mut sum = 0.0f32;
    let mut cnt = 0.0f32;
    for &v in g.neighbors(ui) {
        if local[v as usize] == u32::MAX {
            continue;
        }
        sum += x_index[v as usize];
        cnt += 1.0;
    }
    if cnt > 0.0 {
        sum / cnt
    } else {
        // isolated in that layer: deterministic fallback
        (hash32(u) & 0x3fff) as f32
    }
}

fn pack_components(comps: &mut [Vec<u32>], pos: &mut [Pt]) {
    // Simple skyline packing: place each component in a row, wrapping as needed.
    // Components are expected to already have relative coordinates around origin.
    let mut boxes: Vec<(usize, f32, f32, f32, f32)> = Vec::with_capacity(comps.len());
    for (ci, comp) in comps.iter().enumerate() {
        let mut minx = f32::INFINITY;
        let mut maxx = f32::NEG_INFINITY;
        let mut miny = f32::INFINITY;
        let mut maxy = f32::NEG_INFINITY;
        for &u in comp {
            let p = pos[u as usize];
            minx = minx.min(p.x);
            maxx = maxx.max(p.x);
            miny = miny.min(p.y);
            maxy = maxy.max(p.y);
        }
        if !minx.is_finite() {
            minx = 0.0;
            maxx = 0.0;
            miny = 0.0;
            maxy = 0.0;
        }
        boxes.push((ci, minx, maxx, miny, maxy));
    }

    // Sort by descending area so big comps place first
    boxes.sort_unstable_by(|a, b| {
        let aw = (a.2 - a.1).abs();
        let ah = (a.4 - a.3).abs();
        let bw = (b.2 - b.1).abs();
        let bh = (b.4 - b.3).abs();
        let aa = aw * ah;
        let ba = bw * bh;
        ba.partial_cmp(&aa).unwrap_or(Ordering::Equal)
    });

    let gap = 5.0f32;
    let max_row_width = 500.0f32;
    let mut cx = 0.0f32;
    let mut cy = 0.0f32;
    let mut row_h = 0.0f32;

    for (ci, minx, maxx, miny, maxy) in boxes {
        let w = (maxx - minx).abs() + gap;
        let h = (maxy - miny).abs() + gap;
        if cx + w > max_row_width {
            cx = 0.0;
            cy += row_h;
            row_h = 0.0;
        }
        row_h = row_h.max(h);
        let dx = cx - minx;
        let dy = cy - miny;
        for &u in &comps[ci] {
            pos[u as usize].x += dx;
            pos[u as usize].y += dy;
        }
        cx += w;
    }
}

fn main() {
    let mut inp = FastInput::new();
    let n = match inp.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = match inp.next_usize() {
        Some(v) => v,
        None => 0,
    };

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let a = match inp.next_usize() {
            Some(v) => v,
            None => break,
        };
        let b = match inp.next_usize() {
            Some(v) => v,
            None => break,
        };
        edges.push((a as u32, b as u32));
    }

    let g = build_graph(n, &edges);
    let mut pos = vec![Pt { x: 0.0, y: 0.0 }; n];

    // Component-wise layout with size-based strategy
    let mut comps = connected_components(&g);
    for comp in comps.iter() {
        let sz = comp.len();
        if sz <= 2000 {
            layout_small_force(&g, comp, &mut pos);
        } else {
            layout_large_layered(&g, comp, &mut pos);
        }
    }

    // Pack disconnected components so they don't overlap
    if comps.len() > 1 {
        pack_components(&mut comps, &mut pos);
    }

    // Output N lines: x y
    // (Optionally prefixing with N is allowed, but we omit for simplicity.)
    let mut out = String::with_capacity(n * 24);
    for i in 0..n {
        out.push_str(&format!("{:.6} {:.6}\n", pos[i].x as f64, pos[i].y as f64));
    }
    print!("{}", out);
}
