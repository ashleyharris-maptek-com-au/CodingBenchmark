import math
from typing import Dict


# Module-level persistent autopilot memory (survives across calls in a scenario)
_AP = {
    "init": False,
    "x": 0.0,
    "y": 0.0,
    "z": 0.0,
    "vx": 0.0,
    "vy": 0.0,
    "vz": 0.0,
    "have_state": False,
    "last_dir": (0.0, 1.0, 0.0),  # default: target "ahead" along +Y
    "last_dt": 0.1,
    "last_cmd": (0.0, 0.0, 0.0),
}


def _clamp(v: float, lo: float, hi: float) -> float:
    if v < lo:
        return lo
    if v > hi:
        return hi
    return v


def _norm3(x: float, y: float, z: float) -> float:
    return math.sqrt(x * x + y * y + z * z)


def _median(vals):
    vals = [v for v in vals if v is not None]
    if not vals:
        return None
    vals.sort()
    n = len(vals)
    mid = n // 2
    if n % 2 == 1:
        return vals[mid]
    return 0.5 * (vals[mid - 1] + vals[mid])


def _read_vector_measurement(state: Dict) -> Dict:
    """
    Returns:
      {
        "pos": (x,y,z) or None,
        "vel": (vx,vy,vz) or None,
        "range": float or None,
        "quality": float in [0,1]
      }
    """
    # Radar
    rpos_ok = bool(state.get("radar_rel_x_valid")) and bool(state.get("radar_rel_y_valid")) and bool(state.get("radar_rel_z_valid"))
    rvel_ok = bool(state.get("radar_relv_x_valid")) and bool(state.get("radar_relv_y_valid")) and bool(state.get("radar_relv_z_valid"))
    radar_pos = None
    radar_vel = None
    if rpos_ok:
        radar_pos = (
            float(state.get("radar_rel_pos_x_m", 0.0)),
            float(state.get("radar_rel_pos_y_m", 0.0)),
            float(state.get("radar_rel_pos_z_m", 0.0)),
        )
    if rvel_ok:
        radar_vel = (
            float(state.get("radar_rel_vel_x_ms", 0.0)),
            float(state.get("radar_rel_vel_y_ms", 0.0)),
            float(state.get("radar_rel_vel_z_ms", 0.0)),
        )

    # Lidar
    lpos_ok = bool(state.get("lidar_rel_x_valid")) and bool(state.get("lidar_rel_y_valid")) and bool(state.get("lidar_rel_z_valid"))
    lvel_ok = bool(state.get("lidar_relv_x_valid")) and bool(state.get("lidar_relv_y_valid")) and bool(state.get("lidar_relv_z_valid"))
    lidar_pos = None
    lidar_vel = None
    if lpos_ok:
        lidar_pos = (
            float(state.get("lidar_rel_pos_x_m", 0.0)),
            float(state.get("lidar_rel_pos_y_m", 0.0)),
            float(state.get("lidar_rel_pos_z_m", 0.0)),
        )
    if lvel_ok:
        lidar_vel = (
            float(state.get("lidar_rel_vel_x_ms", 0.0)),
            float(state.get("lidar_rel_vel_y_ms", 0.0)),
            float(state.get("lidar_rel_vel_z_ms", 0.0)),
        )

    # Docking camera (bearing + range)
    cam_active = bool(state.get("dock_camera_active"))
    cam_pos = None
    if cam_active:
        bx_ok = bool(state.get("dock_camera_bearing_x_valid"))
        by_ok = bool(state.get("dock_camera_bearing_y_valid"))
        bz_ok = bool(state.get("dock_camera_bearing_z_valid"))
        if bx_ok and by_ok and bz_ok:
            bx = float(state.get("dock_camera_bearing_x", 0.0))
            by = float(state.get("dock_camera_bearing_y", 0.0))
            bz = float(state.get("dock_camera_bearing_z", 0.0))
            br = _norm3(bx, by, bz)
            if br > 1e-6:
                rng = float(state.get("dock_camera_range_m", 0.0))
                cam_pos = (bx / br * rng, by / br * rng, bz / br * rng)

    # Rangefinders (scalar)
    ranges = []
    for i in (1, 2, 3):
        if bool(state.get(f"range_{i}_valid")):
            ranges.append(float(state.get(f"range_{i}_m", 0.0)))
    range_scalar = _median(ranges)

    # Fuse pos/vel: prefer lidar close, radar far; camera when active; fallback to scalar
    pos = None
    vel = None
    quality = 0.0

    if radar_pos is not None and lidar_pos is not None:
        # Weight by range: lidar stronger <300m, radar stronger otherwise
        r_radar = _norm3(*radar_pos)
        w_l = _clamp((300.0 - r_radar) / 300.0, 0.0, 1.0)  # 1 close, 0 far
        w_r = 1.0 - w_l
        pos = (
            w_r * radar_pos[0] + w_l * lidar_pos[0],
            w_r * radar_pos[1] + w_l * lidar_pos[1],
            w_r * radar_pos[2] + w_l * lidar_pos[2],
        )
        quality = 0.95
    elif lidar_pos is not None:
        pos = lidar_pos
        quality = 0.9
    elif radar_pos is not None:
        pos = radar_pos
        quality = 0.85
    elif cam_pos is not None:
        pos = cam_pos
        quality = 0.75

    if radar_vel is not None and lidar_vel is not None:
        r_pos = pos if pos is not None else radar_pos
        r_rng = _norm3(*r_pos) if r_pos is not None else 500.0
        w_l = _clamp((300.0 - r_rng) / 300.0, 0.0, 1.0)
        w_r = 1.0 - w_l
        vel = (
            w_r * radar_vel[0] + w_l * lidar_vel[0],
            w_r * radar_vel[1] + w_l * lidar_vel[1],
            w_r * radar_vel[2] + w_l * lidar_vel[2],
        )
        quality = max(quality, 0.95)
    elif lidar_vel is not None:
        vel = lidar_vel
        quality = max(quality, 0.9)
    elif radar_vel is not None:
        vel = radar_vel
        quality = max(quality, 0.85)

    # If only scalar range available, reconstruct position along last known direction
    if pos is None and range_scalar is not None and _AP["have_state"]:
        dx, dy, dz = _AP["last_dir"]
        pos = (dx * range_scalar, dy * range_scalar, dz * range_scalar)
        quality = max(quality, 0.4)

    # If we have pos but not range_scalar, compute range
    rng = None
    if pos is not None:
        rng = _norm3(*pos)
    elif range_scalar is not None:
        rng = float(range_scalar)

    return {"pos": pos, "vel": vel, "range": rng, "quality": quality}


def _mean_motion(state: Dict) -> float:
    # Prefer semi-major axis when valid. Otherwise approximate from altitude.
    mu = 3.986004418e14  # m^3/s^2
    a = None
    if bool(state.get("ground_semi_major_axis_m_valid")):
        a = float(state.get("ground_semi_major_axis_m", 0.0))
        if not (6.0e6 < a < 9.0e6):
            a = None
    if a is None and bool(state.get("ground_altitude_km_valid")):
        alt_km = float(state.get("ground_altitude_km", 0.0))
        if 100.0 < alt_km < 2000.0:
            a = 6371e3 + alt_km * 1e3
    if a is None:
        # Typical LEO-ish fallback (~400 km)
        a = 6371e3 + 400e3
    return math.sqrt(mu / (a ** 3))


def _propagate_cw(x, y, z, vx, vy, vz, n, dt):
    # Simple Euler integration of unforced HCW dynamics.
    ax = 3.0 * n * n * x + 2.0 * n * vy
    ay = -2.0 * n * vx
    az = -1.0 * n * n * z
    vx2 = vx + ax * dt
    vy2 = vy + ay * dt
    vz2 = vz + az * dt
    x2 = x + vx2 * dt
    y2 = y + vy2 * dt
    z2 = z + vz2 * dt
    return x2, y2, z2, vx2, vy2, vz2


def _alpha_beta_update(x, v, z, dt, alpha, beta):
    # 1D alpha-beta filter update with position measurement z.
    r = z - x
    x2 = x + alpha * r
    v2 = v + (beta / max(dt, 1e-3)) * r
    return x2, v2


def _attitude_rate_cmd(state: Dict) -> Dict:
    # Stabilize angular rates using median of valid gyros.
    # Gyros are deg/s; output rad/s with max ±0.035.
    def gyro_axis(axis):
        vals = []
        for i in (1, 2, 3):
            if bool(state.get(f"gyro_{i}_{axis}_valid")):
                vals.append(float(state.get(f"gyro_{i}_{axis}_dps", 0.0)))
        m = _median(vals)
        if m is None:
            return 0.0
        return m * math.pi / 180.0

    roll_rate = gyro_axis("roll")
    pitch_rate = gyro_axis("pitch")
    yaw_rate = gyro_axis("yaw")

    k = 0.8  # damping gain
    cmd_roll = _clamp(-k * roll_rate, -0.035, 0.035)
    cmd_pitch = _clamp(-k * pitch_rate, -0.035, 0.035)
    cmd_yaw = _clamp(-k * yaw_rate, -0.035, 0.035)
    return {
        "att_rate_roll": cmd_roll,
        "att_rate_pitch": cmd_pitch,
        "att_rate_yaw": cmd_yaw,
    }


def autopilot_step(state: dict, dt: float) -> dict:
    # Defensive dt
    if dt is None or not (dt > 0.0) or dt > 2.0:
        dt = _AP.get("last_dt", 0.1)
    _AP["last_dt"] = float(dt)

    n = _mean_motion(state)
    meas = _read_vector_measurement(state)
    pos = meas["pos"]
    vel = meas["vel"]
    rng = meas["range"]
    q = meas["quality"]

    # Initialize state estimate
    if not _AP["init"]:
        if pos is not None:
            _AP["x"], _AP["y"], _AP["z"] = pos
            if vel is not None:
                _AP["vx"], _AP["vy"], _AP["vz"] = vel
            else:
                _AP["vx"] = _AP["vy"] = _AP["vz"] = 0.0
            _AP["have_state"] = True
            _AP["init"] = True
        else:
            # No measurements yet: output safe zeros
            out = {"thrust_x": 0.0, "thrust_y": 0.0, "thrust_z": 0.0}
            out.update(_attitude_rate_cmd(state))
            return out

    # Propagate estimate using CW dynamics
    if _AP["have_state"]:
        _AP["x"], _AP["y"], _AP["z"], _AP["vx"], _AP["vy"], _AP["vz"] = _propagate_cw(
            _AP["x"], _AP["y"], _AP["z"], _AP["vx"], _AP["vy"], _AP["vz"], n, dt
        )

    # Update estimate with measurements (alpha-beta)
    # Higher quality -> higher alpha/beta
    alpha_p = 0.35 + 0.45 * q
    beta_p = 0.05 + 0.25 * q
    alpha_v = 0.4 + 0.5 * q

    if pos is not None:
        _AP["x"], _AP["vx"] = _alpha_beta_update(_AP["x"], _AP["vx"], pos[0], dt, alpha_p, beta_p)
        _AP["y"], _AP["vy"] = _alpha_beta_update(_AP["y"], _AP["vy"], pos[1], dt, alpha_p, beta_p)
        _AP["z"], _AP["vz"] = _alpha_beta_update(_AP["z"], _AP["vz"], pos[2], dt, alpha_p, beta_p)
        rr = _norm3(pos[0], pos[1], pos[2])
        if rr > 1e-6:
            _AP["last_dir"] = (pos[0] / rr, pos[1] / rr, pos[2] / rr)

    if vel is not None:
        _AP["vx"] = (1.0 - alpha_v) * _AP["vx"] + alpha_v * vel[0]
        _AP["vy"] = (1.0 - alpha_v) * _AP["vy"] + alpha_v * vel[1]
        _AP["vz"] = (1.0 - alpha_v) * _AP["vz"] + alpha_v * vel[2]

    x, y, z = _AP["x"], _AP["y"], _AP["z"]
    vx, vy, vz = _AP["vx"], _AP["vy"], _AP["vz"]

    # Estimate range/speed
    if rng is None:
        rng = _norm3(x, y, z)
    speed = _norm3(vx, vy, vz)

    # Fuel-aware thrust limiting
    fuel_pct = float(state.get("fuel_remaining_pct", 100.0) or 0.0)
    fuel_kg = float(state.get("fuel_remaining_kg", 500.0) or 0.0)
    low_fuel = (fuel_kg < 80.0) or (fuel_pct < 15.0)
    very_low_fuel = (fuel_kg < 30.0) or (fuel_pct < 7.0)

    # Normalized max thrust per axis (dimensionless output). Keep conservative.
    max_u = 0.45
    if low_fuel:
        max_u = 0.28
    if very_low_fuel:
        max_u = 0.18

    # Desired docking thresholds
    target_range = float(state.get("target_dock_range_m", 2.0) or 2.0)
    target_speed = float(state.get("target_dock_speed_ms", 0.3) or 0.3)

    # Approach speed schedule (positive means allowed closing speed magnitude)
    if rng > 200.0:
        v_allow = 1.2
    elif rng > 50.0:
        v_allow = 0.5
    elif rng > 10.0:
        v_allow = 0.18
    else:
        v_allow = 0.08

    # Control modes
    # FAR: leverage "counter-intuitive" along-track control: ay ~ +k*y (y<0 -> retrograde)
    # MED/CLOSE: PD to zero with speed limiting along line-of-sight
    if rng > 200.0:
        # Primary goal: reduce along-track separation without excessive fuel.
        k_y = 0.0009
        kd_y = 0.08
        # Counter-intuitive sign: command retrograde when behind (y<0), prograde when ahead (y>0)
        u_y = _clamp(k_y * y + kd_y * vy, -max_u, max_u)

        # Keep radial/cross-track bounded gently (avoid oscillatory forcing)
        k_x = 0.0022
        kd_x = 0.12
        k_z = 0.0020
        kd_z = 0.12
        u_x = _clamp(-(k_x * x + kd_x * vx), -max_u, max_u)
        u_z = _clamp(-(k_z * z + kd_z * vz), -max_u, max_u)
    elif rng > 50.0:
        k_p = 0.010
        k_d = 0.22
        u_x = -(k_p * x + k_d * vx)
        u_y = -(k_p * y + k_d * vy)
        u_z = -(k_p * z + k_d * vz)

        # Prefer a V-bar-ish approach: discourage large radial motion when already close in y
        if abs(y) < 80.0:
            u_x *= 0.6
            u_z *= 0.8
    else:
        # CLOSE: regulate closing speed along the line-of-sight + hold x/z small.
        r = _norm3(x, y, z)
        if r < 1e-6:
            r_hat = (0.0, 0.0, 0.0)
        else:
            r_hat = (x / r, y / r, z / r)
        v_rad = vx * r_hat[0] + vy * r_hat[1] + vz * r_hat[2]  # positive = opening

        # Desired radial velocity: slightly negative (closing), decreasing with range
        v_des = -min(v_allow, max(0.05, 0.02 * r))
        # If already within docking limits, stop pushing.
        if r <= target_range and speed <= target_speed:
            v_des = 0.0

        # Acceleration-like command along r_hat from radial velocity error
        k_v = 0.9
        a_los = _clamp(k_v * (v_des - v_rad), -0.18, 0.18)

        # Lateral (x/z) hold (soft near final)
        k_lat_p = 0.06 if r > 10.0 else 0.035
        k_lat_d = 0.35 if r > 10.0 else 0.25
        a_x_lat = -(k_lat_p * x + k_lat_d * vx)
        a_z_lat = -(k_lat_p * z + k_lat_d * vz)

        u_x = a_x_lat + a_los * r_hat[0]
        u_y = -(0.04 * y + 0.22 * vy) + a_los * r_hat[1]
        u_z = a_z_lat + a_los * r_hat[2]

        # Very close: avoid over-control
        if r < 10.0:
            u_x *= 0.7
            u_y *= 0.7
            u_z *= 0.7

    # Global speed limit: if closing too fast, push opposite velocity
    if rng is not None and rng < 120.0 and speed > (v_allow * 1.35):
        # Brake more aggressively when very close.
        k_brake = 0.22 if rng < 25.0 else 0.14
        u_x += -k_brake * vx
        u_y += -k_brake * vy
        u_z += -k_brake * vz

    # Clamp and apply fuel-aware limits
    u_x = _clamp(u_x, -max_u, max_u)
    u_y = _clamp(u_y, -max_u, max_u)
    u_z = _clamp(u_z, -max_u, max_u)

    # Deadband near docking to prevent chattering
    if rng is not None and rng < 3.0:
        if abs(u_x) < 0.05:
            u_x = 0.0
        if abs(u_y) < 0.05:
            u_y = 0.0
        if abs(u_z) < 0.05:
            u_z = 0.0

    _AP["last_cmd"] = (u_x, u_y, u_z)

    out = {"thrust_x": float(u_x), "thrust_y": float(u_y), "thrust_z": float(u_z)}
    out.update(_attitude_rate_cmd(state))
    return out
