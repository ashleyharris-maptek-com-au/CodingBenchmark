use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn main() {
    let (width, height, gravity, max_thrust, initial_fuel,
         start_x, start_y, target_x, target_y) = {
        let stdin = io::stdin();
        let mut handle = stdin.lock();

        let mut line = String::new();
        handle.read_line(&mut line).unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let width: i32 = parts[0].parse().unwrap();
        let height: i32 = parts[1].parse().unwrap();
        let gravity: f64 = parts[2].parse().unwrap();
        let max_thrust: f64 = parts[3].parse().unwrap();
        let initial_fuel: f64 = parts[4].parse().unwrap();

        line.clear();
        handle.read_line(&mut line).unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let start_x: f64 = parts[0].parse().unwrap();
        let start_y: f64 = parts[1].parse().unwrap();
        let target_x: f64 = parts[2].parse().unwrap();
        let target_y: f64 = parts[3].parse().unwrap();

        for _ in 0..height {
            line.clear();
            handle.read_line(&mut line).unwrap();
        }

        (width, height, gravity, max_thrust, initial_fuel,
         start_x, start_y, target_x, target_y)
    };

    let state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: initial_fuel,
    }));

    let state_reader = Arc::clone(&state);
    thread::spawn(move || {
        let stdin = io::stdin();
        let reader = stdin.lock();
        for line_result in reader.lines() {
            if let Ok(line) = line_result {
                let parts: Vec<&str> = line.split_whitespace().collect();
                if parts.len() >= 7 {
                    if let (Ok(x), Ok(y), Ok(vx), Ok(vy), Ok(angle), Ok(angular_vel), Ok(fuel)) = (
                        parts[0].parse::<f64>(),
                        parts[1].parse::<f64>(),
                        parts[2].parse::<f64>(),
                        parts[3].parse::<f64>(),
                        parts[4].parse::<f64>(),
                        parts[5].parse::<f64>(),
                        parts[6].parse::<f64>(),
                    ) {
                        let mut s = state_reader.lock().unwrap();
                        s.x = x;
                        s.y = y;
                        s.vx = vx;
                        s.vy = vy;
                        s.angle = angle;
                        s.angular_vel = angular_vel;
                        s.fuel = fuel;
                    }
                }
            }
        }
    });

    let mut stdout = io::stdout();
    let mut last_output = Instant::now();
    let output_interval = Duration::from_millis(50);

    loop {
        let now = Instant::now();
        if now.duration_since(last_output) >= output_interval {
            let s = state.lock().unwrap();

            let dx = target_x - s.x;
            let dy = target_y - s.y;
            let dist = (dx * dx + dy * dy).sqrt();

            let target_descent = 1.0;
            let thrust = if s.vy > target_descent {
                1.0
            } else if s.vy < 0.0 && dist > 5.0 {
                0.0
            } else {
                0.5
            };

            let target_angle = if dist > 2.0 {
                dx.atan2(-dy)
            } else {
                0.0
            };

            let mut angle_diff = target_angle - s.angle;
            const PI: f64 = std::f64::consts::PI;
            while angle_diff > PI { angle_diff -= 2.0 * PI; }
            while angle_diff < -PI { angle_diff += 2.0 * PI; }

            let turn = angle_diff.clamp(-1.0, 1.0);

            writeln!(stdout, "{} {}", thrust, turn).ok();
            stdout.flush().ok();

            last_output = now;
            drop(s);
        }

        thread::sleep(Duration::from_millis(1));
    }
}
