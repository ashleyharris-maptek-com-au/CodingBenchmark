#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate view direction
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

    // Calculate half vector for Blinn-Phong
    vec3 H = normalize(L + V);

    // Calculate diffuse component
    float diffuse = max(dot(N, L), 0.0);

    // Calculate specular component
    float specular = pow(max(dot(N, H), 0.0), 64.0);

    // Calculate grayscale intensity
    float intensity = 0.1 + 0.6 * diffuse + 0.8 * specular;

    // Output as grayscale with alpha 1.0
    outColor = vec4(vec3(intensity), 1.0);
}
