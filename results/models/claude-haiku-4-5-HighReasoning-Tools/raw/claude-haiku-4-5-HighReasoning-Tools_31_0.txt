use std::io::{self, BufRead};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: n and target
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let target: i64 = parts.next().unwrap().parse().unwrap();

    // Handle empty set
    if n == 0 {
        if target == 0 {
            println!("YES");
            println!("[]");
        } else {
            println!("NO");
        }
        return;
    }

    // Read second line: the numbers
    let numbers_line = lines.next().unwrap().unwrap();
    let numbers: Vec<i64> = numbers_line
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();

    // Choose algorithm based on problem size
    if n <= 20 {
        solve_brute_force(&numbers, target);
    } else if n <= 40 {
        solve_meet_in_middle(&numbers, target);
    } else {
        solve_dp_or_approximation(&numbers, target);
    }
}

fn solve_brute_force(numbers: &[i64], target: i64) {
    let n = numbers.len();

    for mask in 0..(1u64 << n) {
        let mut sum = 0i64;
        let mut indices = Vec::new();

        for i in 0..n {
            if (mask >> i) & 1 == 1 {
                sum += numbers[i];
                indices.push(i);
            }
        }

        if sum == target {
            println!("YES");
            print_indices(&indices);
            return;
        }
    }

    println!("NO");
}

fn solve_meet_in_middle(numbers: &[i64], target: i64) {
    let n = numbers.len();
    let mid = n / 2;

    // Generate all subset sums for first half with their masks
    let mut left_sums: HashMap<i64, Vec<u32>> = HashMap::new();
    for mask in 0..(1u32 << mid) {
        let mut sum = 0i64;

        for i in 0..mid {
            if (mask >> i) & 1 == 1 {
                sum += numbers[i];
            }
        }

        left_sums.entry(sum).or_insert_with(Vec::new).push(mask);
    }

    // Generate subset sums for second half and find matches
    let right_len = n - mid;
    for mask in 0..(1u32 << right_len) {
        let mut sum = 0i64;

        for i in 0..right_len {
            if (mask >> i) & 1 == 1 {
                sum += numbers[mid + i];
            }
        }

        let needed = target - sum;
        if let Some(left_masks) = left_sums.get(&needed) {
            if let Some(&left_mask) = left_masks.first() {
                // Reconstruct solution
                let mut indices = Vec::new();

                for i in 0..mid {
                    if (left_mask >> i) & 1 == 1 {
                        indices.push(i);
                    }
                }

                for i in 0..right_len {
                    if (mask >> i) & 1 == 1 {
                        indices.push(mid + i);
                    }
                }

                println!("YES");
                print_indices(&indices);
                return;
            }
        }
    }

    println!("NO");
}

fn solve_dp_or_approximation(numbers: &[i64], target: i64) {
    // For larger sets, try DP if target is reasonable
    if target >= 0 && target <= 100_000_000 {
        if solve_dp_bounded(numbers, target) {
            return;
        }
    }

    // Fallback: use randomized/greedy approximation
    solve_greedy_approximation(numbers, target);
}

fn solve_dp_bounded(numbers: &[i64], target: i64) -> bool {
    let target_u = target as usize;

    // Use bitset-like approach with boolean array
    let mut dp: Vec<bool> = vec![false; target_u + 1];
    let mut parent: Vec<Option<usize>> = vec![None; target_u + 1];
    dp[0] = true;

    for (idx, &num) in numbers.iter().enumerate() {
        if num <= 0 {
            continue;
        }
        let num_u = num as usize;

        if num_u > target_u {
            continue;
        }

        for j in (num_u..=target_u).rev() {
            if dp[j - num_u] && !dp[j] {
                dp[j] = true;
                parent[j] = Some(idx);
            }
        }
    }

    if !dp[target_u] {
        return false;
    }

    // Reconstruct solution
    let mut indices = Vec::new();
    let mut current = target_u;

    while current > 0 {
        if let Some(idx) = parent[current] {
            indices.push(idx);
            current -= numbers[idx] as usize;
        } else {
            break;
        }
    }

    println!("YES");
    print_indices(&indices);
    true
}

fn solve_greedy_approximation(numbers: &[i64], target: i64) {
    let n = numbers.len();

    // Try to find solution by greedy selection
    let mut indexed: Vec<(usize, i64)> = numbers
        .iter()
        .enumerate()
        .map(|(i, &x)| (i, x))
        .collect();

    // Sort by absolute value descending for better greedy
    indexed.sort_by_key(|x| -x.1.abs());

    let mut current_sum = 0i64;
    let mut selected = Vec::new();

    // Greedy: select elements that get us closer to target
    for &(idx, num) in &indexed {
        let new_sum = current_sum + num;
        if new_sum == target {
            selected.push(idx);
            println!("YES");
            selected.sort();
            print_indices(&selected);
            return;
        }

        // Take element if it gets us closer or doesn't overshoot too much
        if (new_sum - target).abs() < (current_sum - target).abs() {
            current_sum = new_sum;
            selected.push(idx);
        }
    }

    // Try subset of solutions with backtracking for small number of remaining elements
    if n <= 100 {
        if backtrack_solve(&numbers, target, &mut vec![], 0, 0) {
            return;
        }
    }

    println!("NO");
}

fn backtrack_solve(numbers: &[i64], target: i64, current: &mut Vec<usize>, idx: usize, sum: i64) -> bool {
    if sum == target {
        println!("YES");
        print_indices(current);
        return true;
    }

    if idx >= numbers.len() {
        return false;
    }

    // Prune if sum too large
    if sum > target && numbers.iter().all(|&x| x >= 0) {
        return false;
    }

    // Try including current element
    current.push(idx);
    if backtrack_solve(numbers, target, current, idx + 1, sum + numbers[idx]) {
        return true;
    }
    current.pop();

    // Try not including current element
    if backtrack_solve(numbers, target, current, idx + 1, sum) {
        return true;
    }

    false
}

fn print_indices(indices: &[usize]) {
    print!("[");
    for (i, &idx) in indices.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", idx);
    }
    println!("]");
}
