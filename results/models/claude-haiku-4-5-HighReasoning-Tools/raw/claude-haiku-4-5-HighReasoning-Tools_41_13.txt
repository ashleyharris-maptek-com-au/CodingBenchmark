OpCapability Shader
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos_in %outColor_out
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos_in Location 0
OpDecorate %outColor_out Location 0

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Function type for main
%void_fn = OpTypeFunction %void

; Pointer types
%Input_v3float_ptr = OpTypePointer Input %v3float
%Output_v4float_ptr = OpTypePointer Output %v4float

; Variable declarations
%worldPos_in = OpVariable %Input_v3float_ptr Input
%outColor_out = OpVariable %Output_v4float_ptr Output

; Constants
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4

%v3_bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%v3_top = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Extended instructions
%glsl_ext = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load worldPos
%worldPos = OpLoad %v3float %worldPos_in

; Extract y component
%y = OpCompositeExtract %float %worldPos 1

; Calculate t = worldPos.y * 0.5 + 0.5
%temp1 = OpFMul %float %y %float_0_5
%t_unclamped = OpFAdd %float %temp1 %float_0_5

; Clamp t to [0, 1]
%t = OpExtInst %float %glsl_ext FClamp %t_unclamped %float_0 %float_1

; Mix colors: lerp(bottom, top, t)
%mixed_color = OpExtInst %v3float %glsl_ext Mix %v3_bottom %v3_top %t

; Create output color with alpha = 1.0
%outColor = OpCompositeConstruct %v4float %mixed_color %float_1

; Store result
OpStore %outColor_out %outColor

OpReturn
OpFunctionEnd
