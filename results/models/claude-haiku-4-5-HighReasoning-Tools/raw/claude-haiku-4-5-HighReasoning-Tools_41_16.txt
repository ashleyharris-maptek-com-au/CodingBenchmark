%void = OpTypeVoid

%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptr_uniform_ubo = OpTypePointer Uniform %ubo_type
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_input_v2 = OpTypePointer Input %v2float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_function_v2 = OpTypePointer Function %v2float
%ptr_function_i = OpTypePointer Function %int

%fn_void = OpTypeFunction %void

%in_worldPos = OpVariable %ptr_input_v3 Input
%in_normal = OpVariable %ptr_input_v3 Input
%in_uv = OpVariable %ptr_input_v2 Input
%in_tangent = OpVariable %ptr_input_v3 Input
%in_color = OpVariable %ptr_input_v3 Input
%out_color = OpVariable %ptr_output_v4 Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%fc0 = OpConstant %float 0.0
%fc1 = OpConstant %float 1.0
%fc2 = OpConstant %float 2.0
%fc3 = OpConstant %float 3.0
%fc4 = OpConstant %float 4.0
%fc0_5 = OpConstant %float 0.5
%fc20 = OpConstant %float 20.0

%ic0 = OpConstant %int 0
%ic1 = OpConstant %int 1
%ic20 = OpConstant %int 20

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load UV coordinates
%uv = OpLoad %v2float %in_uv
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

; Map UV to Mandelbrot space: c = (uv.x*3-2, uv.y*2-1)
%tmp1 = OpFMul %float %uv_x %fc3
%c_x = OpFSub %float %tmp1 %fc2
%tmp2 = OpFMul %float %uv_y %fc2
%c_y = OpFSub %float %tmp2 %fc1
%c = OpCompositeConstruct %v2float %c_x %c_y

; Initialize z = 0 and iteration counter = 0
%z_ptr = OpVariable %ptr_function_v2 Function
%z_init = OpCompositeConstruct %v2float %fc0 %fc0
OpStore %z_ptr %z_init
%iter_ptr = OpVariable %ptr_function_i Function
OpStore %iter_ptr %ic0

; Loop header
OpBranch %loop_header
%loop_header = OpLabel
%iter_val = OpLoad %int %iter_ptr
%loop_cond = OpSLessThan %bool %iter_val %ic20
OpLoopMerge %loop_merge %loop_continue None
OpBranchConditional %loop_cond %loop_body %loop_merge

; Loop body
%loop_body = OpLabel
%z_val = OpLoad %v2float %z_ptr

; Extract z components
%z_x = OpCompositeExtract %float %z_val 0
%z_y = OpCompositeExtract %float %z_val 1

; Calculate z = z*z + c
; z.x' = z.x*z.x - z.y*z.y + c.x
%z_x_sq = OpFMul %float %z_x %z_x
%z_y_sq = OpFMul %float %z_y %z_y
%z_x_part = OpFSub %float %z_x_sq %z_y_sq
%z_x_new = OpFAdd %float %z_x_part %c_x

; z.y' = 2*z.x*z.y + c.y
%z_2x = OpFMul %float %fc2 %z_x
%z_2xy = OpFMul %float %z_2x %z_y
%z_y_new = OpFAdd %float %z_2xy %c_y

%z_new = OpCompositeConstruct %v2float %z_x_new %z_y_new
OpStore %z_ptr %z_new

; Check escape condition: |z|^2 > 4
%z_x_chk = OpCompositeExtract %float %z_new 0
%z_y_chk = OpCompositeExtract %float %z_new 1
%z_x_sq_chk = OpFMul %float %z_x_chk %z_x_chk
%z_y_sq_chk = OpFMul %float %z_y_chk %z_y_chk
%z_mag_sq = OpFAdd %float %z_x_sq_chk %z_y_sq_chk
%escaped = OpFOrdGreaterThan %bool %z_mag_sq %fc4

OpSelectionMerge %check_done None
OpBranchConditional %escaped %loop_merge %check_done

%check_done = OpLabel
OpBranch %loop_continue

; Loop continue - increment iteration counter
%loop_continue = OpLabel
%next_iter = OpIAdd %int %iter_val %ic1
OpStore %iter_ptr %next_iter
OpBranch %loop_header

; End of loop
%loop_merge = OpLabel
%iter_final = OpLoad %int %iter_ptr

; Calculate color: t = iterations/20, color = (t, t*0.5, 1.0-t, 1.0)
%iter_float = OpConvertSToF %float %iter_final
%t = OpFDiv %float %iter_float %fc20
%color_r = %t
%color_g = OpFMul %float %t %fc0_5
%color_b = OpFSub %float %fc1 %t
%result = OpCompositeConstruct %v4float %color_r %color_g %color_b %fc1

; Write output color
OpStore %out_color %result
OpReturn
OpFunctionEnd
