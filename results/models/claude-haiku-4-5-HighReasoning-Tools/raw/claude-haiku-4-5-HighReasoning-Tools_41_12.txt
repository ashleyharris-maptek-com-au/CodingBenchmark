; Sky and Ground Colors
%sky = OpConstantComposite %vec3 %c_0_4 %c_0_6 %c_1_0

%normal = OpLoad %vec3 %in_normal

%tangent = OpLoad %vec3 %in_tangent

; Normalize the world-space normal
%N = OpExtInst %vec3 %glsl_450 Normalize %normal

; Extract Y component of normalized normal
%N_y = OpCompositeExtract %float %N 1

; Calculate hemisphere blend factor: N.y * 0.5 + 0.5
%N_y_scaled = OpFMul %float %N_y %c_0_5
%blend_factor = OpFAdd %float %N_y_scaled %c_0_5

; Calculate inverse blend for ground contribution
%inv_blend = OpFSub %float %c_1_0 %blend_factor

; Hemisphere ambient: ground * (1 - blend) + sky * blend
%ground_part = OpVectorTimesScalar %vec3 %ground %inv_blend

%light_pos_v4 = OpLoad %vec4 %light_ptr

; Extract XYZ from light position (discard W)
%light_pos = OpVectorShuffle %vec3 %light_pos_v4 %light_pos_v4 0 1 2

; Calculate light direction vector
%to_light = OpFSub %vec3 %light_pos %worldPos

; Normalize light direction
%L = OpExtInst %vec3 %glsl_450 Normalize %to_light

; Calculate diffuse: max(0, dot(N, L))
%ndotl = OpDot %float %N %L
%diffuse_factor = OpExtInst %float %glsl_450 FMax %ndotl %c_0_0

; Apply diffuse term: 0.3 * diffuse_factor
%diffuse_scale = OpVectorTimesScalar %vec3 %sky %c_0_3
%diffuse = OpVectorTimesScalar %vec3 %diffuse_scale %diffuse_factor

; Combine ambient and diffuse
%final_rgb = OpFAdd %vec3 %ambient %diffuse

; Pack RGB into vec4 with alpha = 1.0
%final_color = OpCompositeConstruct %vec4 %final_rgb %c_1_0

; Store output
OpStore %out_color %final_color
