use std::io::{self, BufRead};
use std::time::Instant;
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next()
        .and_then(|r| r.ok())
        .unwrap_or_default();

    let first_line = first_line.trim_start_matches('\u{FEFF}').to_string();
    let parts: Vec<&str> = first_line.trim().split_whitespace().collect();

    if parts.len() < 4 {
        println!("0");
        return;
    }

    let width: usize = parts[0].parse().unwrap_or(0);
    let height: usize = parts[1].parse().unwrap_or(0);
    let obstacle_count: usize = parts[2].parse().unwrap_or(0);
    let require_cycle: bool = parts[3].parse::<u8>().unwrap_or(0) == 1;

    if width == 0 || height == 0 {
        println!("0");
        return;
    }

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        if let Some(Ok(line)) = lines.next() {
            let coords: Vec<usize> = line.trim()
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if coords.len() >= 2 {
                obstacles.insert((coords[0], coords[1]));
            }
        }
    }

    let total_cells = width * height;
    let target_visits = total_cells - obstacles.len();

    let start_time = Instant::now();
    const TIME_LIMIT_SECS: u64 = 290;

    if let Some(path) = solve(width, height, &obstacles, require_cycle, target_visits, &start_time, TIME_LIMIT_SECS) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn solve(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target: usize,
    start_time: &Instant,
    time_limit: u64,
) -> Option<Vec<(usize, usize)>> {
    if obstacles.contains(&(0, 0)) || target == 0 {
        return None;
    }

    let mut visited = vec![false; width * height];
    let mut path = vec![(0, 0)];
    visited[0] = true;

    if target == 1 {
        return if require_cycle {
            if has_neighbor(width, height, 0, 0, obstacles) {
                Some(path)
            } else {
                None
            }
        } else {
            Some(path)
        };
    }

    let degrees = compute_degrees(width, height, obstacles);

    if backtrack(
        &mut path,
        &mut visited,
        width,
        height,
        obstacles,
        &degrees,
        target,
        require_cycle,
        start_time,
        time_limit,
    ) {
        Some(path)
    } else {
        None
    }
}

fn has_neighbor(width: usize, height: usize, x: usize, y: usize, obstacles: &HashSet<(usize, usize)>) -> bool {
    for (dx, dy) in &[(0, 1), (1, 0), (0, -1), (-1, 0)] {
        let nx = (x as i32 + dx) as usize;
        let ny = (y as i32 + dy) as usize;
        if nx < width && ny < height && !obstacles.contains(&(nx, ny)) {
            return true;
        }
    }
    false
}

fn compute_degrees(width: usize, height: usize, obstacles: &HashSet<(usize, usize)>) -> Vec<usize> {
    let mut degrees = vec![0; width * height];
    for x in 0..width {
        for y in 0..height {
            if !obstacles.contains(&(x, y)) {
                let mut deg = 0;
                for (dx, dy) in &[(0, 1), (1, 0), (0, -1), (-1, 0)] {
                    let nx = (x as i32 + dx) as usize;
                    let ny = (y as i32 + dy) as usize;
                    if nx < width && ny < height && !obstacles.contains(&(nx, ny)) {
                        deg += 1;
                    }
                }
                degrees[x + y * width] = deg;
            }
        }
    }
    degrees
}

fn backtrack(
    path: &mut Vec<(usize, usize)>,
    visited: &mut Vec<bool>,
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    degrees: &[usize],
    target: usize,
    require_cycle: bool,
    start_time: &Instant,
    time_limit: u64,
) -> bool {
    if start_time.elapsed().as_secs() > time_limit {
        return false;
    }

    if path.len() == target {
        if require_cycle {
            let (x, y) = path[target - 1];
            return (x == 0 && y == 1) || (x == 1 && y == 0);
        }
        return true;
    }

    let (x, y) = path[path.len() - 1];
    let mut neighbors = Vec::new();

    for (dx, dy) in &[(0, 1), (1, 0), (0, -1), (-1, 0)] {
        let nx = (x as i32 + dx) as usize;
        let ny = (y as i32 + dy) as usize;

        if nx < width && ny < height {
            let idx = nx + ny * width;
            if !obstacles.contains(&(nx, ny)) && !visited[idx] {
                neighbors.push((nx, ny, degrees[idx]));
            }
        }
    }

    neighbors.sort_by_key(|(_x, _y, deg)| *deg);

    if neighbors.is_empty() && path.len() < target {
        return false;
    }

    for (nx, ny, _deg) in neighbors {
        let idx = nx + ny * width;
        visited[idx] = true;
        path.push((nx, ny));

        if backtrack(path, visited, width, height, obstacles, degrees, target, require_cycle, start_time, time_limit) {
            return true;
        }

        path.pop();
        visited[idx] = false;
    }

    false
}
