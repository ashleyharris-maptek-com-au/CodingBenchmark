#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<int>> machines(J);
    vector<vector<int>> durations(J);

    // Parse input
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        machines[j].resize(T);
        durations[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> machines[j][t] >> durations[j][t];
        }
    }

    // Track when each machine will be free
    vector<long long> machine_free_time(M, 0);

    // Track start times for each task
    vector<vector<long long>> start_times(J);

    // Greedy scheduling: process jobs sequentially
    // For each job, schedule tasks as early as possible
    for (int j = 0; j < J; j++) {
        int T = machines[j].size();
        start_times[j].resize(T);

        long long job_complete_time = 0;

        for (int t = 0; t < T; t++) {
            int m = machines[j][t];
            int d = durations[j][t];

            // Task can start when both:
            // 1. Previous task in this job is done (job_complete_time)
            // 2. Machine is free (machine_free_time[m])
            long long start_time = max(job_complete_time, machine_free_time[m]);
            start_times[j][t] = start_time;

            // Update machine free time
            machine_free_time[m] = start_time + d;

            // Update job completion time
            job_complete_time = start_time + d;
        }
    }

    // Output results
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)start_times[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << start_times[j][t] << " " << machines[j][t];
        }
        cout << "\n";
    }

    return 0;
}
