OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %outColor %gl_out %worldPos %normal %color

OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

; Type definitions
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_0_05 = OpConstant %float 0.05

; gl_PerVertex input struct
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

; Input array types
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3f = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3f Input
%normal = OpVariable %ptr_in_arr_v3f Input
%color = OpVariable %ptr_in_arr_v3f Input
%ptr_in_v3float = OpTypePointer Input %v3float

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

; Output color variable
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; Function type
%func_void = OpTypeFunction %void

; Main function definition
%main = OpFunction %void None %func_void
%label_entry = OpLabel

; Load input position gl_in[0].gl_Position
%pos_access = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%input_pos = OpLoad %v4float %pos_access

; Load input color color[0]
%col_access = OpAccessChain %ptr_in_v3float %color %int_0
%input_col = OpLoad %v3float %col_access

; Extract x, y, z, w components
%x = OpCompositeExtract %float %input_pos 0
%y = OpCompositeExtract %float %input_pos 1
%z = OpCompositeExtract %float %input_pos 2
%w = OpCompositeExtract %float %input_pos 3

; h = 0.05
%h = %float_0_05

; Calculate vertex positions for diamond
; Top: (x, y+h, z, w)
%y_plus_h = OpFAdd %float %y %h
%vert_top = OpCompositeConstruct %v4float %x %y_plus_h %z %w

; Right: (x+h, y, z, w)
%x_plus_h = OpFAdd %float %x %h
%vert_right = OpCompositeConstruct %v4float %x_plus_h %y %z %w

; Left: (x-h, y, z, w)
%x_minus_h = OpFSub %float %x %h
%vert_left = OpCompositeConstruct %v4float %x_minus_h %y %z %w

; Bottom: (x, y-h, z, w)
%y_minus_h = OpFSub %float %y %h
%vert_bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w

; Get the output position pointer
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

; Emit top vertex
OpStore %out_pos_ptr %vert_top
OpStore %outColor %input_col
OpEmitVertex

; Emit right vertex
OpStore %out_pos_ptr %vert_right
OpStore %outColor %input_col
OpEmitVertex

; Emit left vertex
OpStore %out_pos_ptr %vert_left
OpStore %outColor %input_col
OpEmitVertex

; Emit bottom vertex
OpStore %out_pos_ptr %vert_bottom
OpStore %outColor %input_col
OpEmitVertex

; End the primitive
OpEndPrimitive

; Return from main
OpReturn
OpFunctionEnd
