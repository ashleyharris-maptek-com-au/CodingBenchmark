OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor_out
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor_out Location 0
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0p05 = OpConstant %float 0.05
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor_out = OpVariable %ptr_out_v3 Output
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%color = OpLoad %v3float %color_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3
%y_plus_h = OpFAdd %float %y %float_0p05
%y_minus_h = OpFSub %float %y %float_0p05
%x_plus_h = OpFAdd %float %x %float_0p05
%x_minus_h = OpFSub %float %x %float_0p05
%top = OpCompositeConstruct %v4float %x %y_plus_h %z %w
%right = OpCompositeConstruct %v4float %x_plus_h %y %z %w
%left = OpCompositeConstruct %v4float %x_minus_h %y %z %w
%bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %top
OpStore %outColor_out %color
OpEmitVertex
OpStore %out_pos_ptr %right
OpStore %outColor_out %color
OpEmitVertex
OpStore %out_pos_ptr %left
OpStore %outColor_out %color
OpEmitVertex
OpStore %out_pos_ptr %bottom
OpStore %outColor_out %color
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
