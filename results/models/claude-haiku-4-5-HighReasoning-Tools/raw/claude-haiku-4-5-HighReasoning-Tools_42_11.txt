OpCapability Geometry
OpCapability Shader
OpExtension "GLSL.std.450"

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %color_in %outColor %gl_out

OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_0_02 = OpConstant %float 0.02
%float_1_3 = OpConstant %float 0.333333
%float_2_3 = OpConstant %float 0.666667

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_4
%ptr_in_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr Input

%arr_v3_4 = OpTypeArray %v3float %int_4
%ptr_in_v3_arr = OpTypePointer Input %arr_v3_4
%color_in = OpVariable %ptr_in_v3_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%func_type = OpTypeFunction %void
%ext_glsl_std_450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

; Load colors
%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

; Extract XYZ
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

; Compute tangent directions for perpendicular
%tan_start = OpFSub %v3float %p2_xyz %p0_xyz
%tan_end = OpFSub %v3float %p3_xyz %p1_xyz
%tan_start_norm = OpExtInst %v3float %ext_glsl_std_450 Normalize %tan_start
%tan_end_norm = OpExtInst %v3float %ext_glsl_std_450 Normalize %tan_end

; Create perpendicular vectors (rotate tangent 90° in XY plane)
%perp_start = OpCompositeConstruct %v3float (OpFNegate %float (OpCompositeExtract %float %tan_start_norm 1)) (OpCompositeExtract %float %tan_start_norm 0) %float_0
%perp_start_norm = OpExtInst %v3float %ext_glsl_std_450 Normalize %perp_start

%perp_end = OpCompositeConstruct %v3float (OpFNegate %float (OpCompositeExtract %float %tan_end_norm 1)) (OpCompositeExtract %float %tan_end_norm 0) %float_0
%perp_end_norm = OpExtInst %v3float %ext_glsl_std_450 Normalize %perp_end

; Scale perpendiculars by thickness
%perp_start_thick = OpVectorTimesScalar %v3float %perp_start_norm %float_0_02
%perp_end_thick = OpVectorTimesScalar %v3float %perp_end_norm %float_0_02

; Segment 1 (0 to 1/3): positions from p1 towards p2
%seg1_start_xyz = %p1_xyz
%seg1_end_xyz = OpFAdd %v3float %p1_xyz (OpVectorTimesScalar %v3float (OpFSub %v3float %p2_xyz %p1_xyz) %float_1_3)

; Use interpolated perpendiculars for segment 1
%perp_seg1_start = %perp_start_thick
%perp_seg1_end = OpVectorTimesScalar %v3float %perp_start_norm %float_0_02

; Emit quad for segment 1
%v00 = OpFSub %v3float %seg1_start_xyz %perp_seg1_start
%v10 = OpFAdd %v3float %seg1_start_xyz %perp_seg1_start
%v01 = OpFSub %v3float %seg1_end_xyz %perp_seg1_end
%v11 = OpFAdd %v3float %seg1_end_xyz %perp_seg1_end

; Quad 1, Vertex 0
%v00_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v00 0) (OpCompositeExtract %float %v00 1) (OpCompositeExtract %float %v00 2) %float_1
%gp_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gp_0 %v00_4
%color_interp_1 = OpExtInst %v3float %ext_glsl_std_450 Mix %c1 %c2 %float_0
OpStore %outColor %color_interp_1
OpEmitVertex

; Quad 1, Vertex 1
%v10_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v10 0) (OpCompositeExtract %float %v10 1) (OpCompositeExtract %float %v10 2) %float_1
OpStore %gp_0 %v10_4
OpStore %outColor %color_interp_1
OpEmitVertex

; Quad 1, Vertex 2
%v01_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v01 0) (OpCompositeExtract %float %v01 1) (OpCompositeExtract %float %v01 2) %float_1
OpStore %gp_0 %v01_4
OpStore %outColor %color_interp_1
OpEmitVertex

; Quad 1, Vertex 3
%v11_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v11 0) (OpCompositeExtract %float %v11 1) (OpCompositeExtract %float %v11 2) %float_1
OpStore %gp_0 %v11_4
OpStore %outColor %color_interp_1
OpEmitVertex

OpEndPrimitive

; Segment 3 (2/3 to 1): positions from 2/3 point towards p2
%seg3_start_xyz = OpFAdd %v3float %p1_xyz (OpVectorTimesScalar %v3float (OpFSub %v3float %p2_xyz %p1_xyz) %float_2_3)
%seg3_end_xyz = %p2_xyz

; Use interpolated perpendiculars for segment 3
%perp_seg3_start = OpVectorTimesScalar %v3float %perp_end_norm %float_0_02
%perp_seg3_end = %perp_end_thick

; Emit quad for segment 3
%v20 = OpFSub %v3float %seg3_start_xyz %perp_seg3_start
%v30 = OpFAdd %v3float %seg3_start_xyz %perp_seg3_start
%v21 = OpFSub %v3float %seg3_end_xyz %perp_seg3_end
%v31 = OpFAdd %v3float %seg3_end_xyz %perp_seg3_end

; Quad 2, Vertex 0
%v20_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v20 0) (OpCompositeExtract %float %v20 1) (OpCompositeExtract %float %v20 2) %float_1
OpStore %gp_0 %v20_4
%color_interp_2 = OpExtInst %v3float %ext_glsl_std_450 Mix %c1 %c2 %float_1
OpStore %outColor %color_interp_2
OpEmitVertex

; Quad 2, Vertex 1
%v30_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v30 0) (OpCompositeExtract %float %v30 1) (OpCompositeExtract %float %v30 2) %float_1
OpStore %gp_0 %v30_4
OpStore %outColor %color_interp_2
OpEmitVertex

; Quad 2, Vertex 2
%v21_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v21 0) (OpCompositeExtract %float %v21 1) (OpCompositeExtract %float %v21 2) %float_1
OpStore %gp_0 %v21_4
OpStore %outColor %color_interp_2
OpEmitVertex

; Quad 2, Vertex 3
%v31_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v31 0) (OpCompositeExtract %float %v31 1) (OpCompositeExtract %float %v31 2) %float_1
OpStore %gp_0 %v31_4
OpStore %outColor %color_interp_2
OpEmitVertex

OpEndPrimitive

OpFunctionEnd
