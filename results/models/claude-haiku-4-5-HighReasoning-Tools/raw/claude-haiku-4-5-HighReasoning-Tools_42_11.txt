OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0_333 = OpConstant %float 0.333333333
%float_0_667 = OpConstant %float 0.666666667
%float_0_02 = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %int_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%glsl = OpExtInstImport "GLSL.std.450"

%func_main = OpTypeFunction %void
%main = OpFunction %void None %func_main
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

%tan1 = OpFSub %v3float %p2_xyz %p0_xyz
%tan1_half = OpVectorTimesScalar %v3float %tan1 %float_0_5

%tan2 = OpFSub %v3float %p3_xyz %p1_xyz
%tan2_half = OpVectorTimesScalar %v3float %tan2 %float_0_5

%z_axis = OpConstantComposite %v3float %float_0 %float_0 %float_1
%perp1 = OpCrossProduct %v3float %tan1_half %z_axis
%perp2 = OpCrossProduct %v3float %tan2_half %z_axis

%perp1_norm = OpExtInst %v3float %glsl OpNormalize %perp1
%perp2_norm = OpExtInst %v3float %glsl OpNormalize %perp2
%thickness = %float_0_02

%perp1_scaled = OpVectorTimesScalar %v3float %perp1_norm %thickness
%perp2_scaled = OpVectorTimesScalar %v3float %perp2_norm %thickness

%seg_dir = OpFSub %v3float %p2_xyz %p1_xyz

%pt1_3_off = OpVectorTimesScalar %v3float %seg_dir %float_0_333
%pt1_3 = OpFAdd %v3float %p1_xyz %pt1_3_off

%pt2_3_off = OpVectorTimesScalar %v3float %seg_dir %float_0_667
%pt2_3 = OpFAdd %v3float %p1_xyz %pt2_3_off

%col_1_3_a = OpVectorTimesScalar %v3float %c1 %float_0_667
%col_1_3_b = OpVectorTimesScalar %v3float %c2 %float_0_333
%col_1_3 = OpFAdd %v3float %col_1_3_a %col_1_3_b

%col_2_3_a = OpVectorTimesScalar %v3float %c1 %float_0_333
%col_2_3_b = OpVectorTimesScalar %v3float %c2 %float_0_667
%col_2_3 = OpFAdd %v3float %col_2_3_a %col_2_3_b

%pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%v0_pos = OpFSub %v3float %p1_xyz %perp1_scaled
%v0_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v0_pos_4 = OpCompositeInsert %v4float %v0_pos %v0_pos_4 0
%v0_pos_4 = OpCompositeInsert %v4float %v0_pos %v0_pos_4 1
%v0_pos_4 = OpCompositeInsert %v4float %v0_pos %v0_pos_4 2

OpStore %pos_ptr %v0_pos_4
OpStore %outColor %c1
OpEmitVertex

%v1_pos = OpFAdd %v3float %p1_xyz %perp1_scaled
%v1_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v1_pos_4 = OpCompositeInsert %v4float %v1_pos %v1_pos_4 0
%v1_pos_4 = OpCompositeInsert %v4float %v1_pos %v1_pos_4 1
%v1_pos_4 = OpCompositeInsert %v4float %v1_pos %v1_pos_4 2

OpStore %pos_ptr %v1_pos_4
OpStore %outColor %c1
OpEmitVertex

%v2_pos = OpFSub %v3float %pt1_3 %perp1_scaled
%v2_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v2_pos_4 = OpCompositeInsert %v4float %v2_pos %v2_pos_4 0
%v2_pos_4 = OpCompositeInsert %v4float %v2_pos %v2_pos_4 1
%v2_pos_4 = OpCompositeInsert %v4float %v2_pos %v2_pos_4 2

OpStore %pos_ptr %v2_pos_4
OpStore %outColor %col_1_3
OpEmitVertex

%v3_pos = OpFAdd %v3float %pt1_3 %perp1_scaled
%v3_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v3_pos_4 = OpCompositeInsert %v4float %v3_pos %v3_pos_4 0
%v3_pos_4 = OpCompositeInsert %v4float %v3_pos %v3_pos_4 1
%v3_pos_4 = OpCompositeInsert %v4float %v3_pos %v3_pos_4 2

OpStore %pos_ptr %v3_pos_4
OpStore %outColor %col_1_3
OpEmitVertex

OpEndPrimitive

%v4_pos = OpFSub %v3float %pt2_3 %perp2_scaled
%v4_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v4_pos_4 = OpCompositeInsert %v4float %v4_pos %v4_pos_4 0
%v4_pos_4 = OpCompositeInsert %v4float %v4_pos %v4_pos_4 1
%v4_pos_4 = OpCompositeInsert %v4float %v4_pos %v4_pos_4 2

OpStore %pos_ptr %v4_pos_4
OpStore %outColor %col_2_3
OpEmitVertex

%v5_pos = OpFAdd %v3float %pt2_3 %perp2_scaled
%v5_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v5_pos_4 = OpCompositeInsert %v4float %v5_pos %v5_pos_4 0
%v5_pos_4 = OpCompositeInsert %v4float %v5_pos %v5_pos_4 1
%v5_pos_4 = OpCompositeInsert %v4float %v5_pos %v5_pos_4 2

OpStore %pos_ptr %v5_pos_4
OpStore %outColor %col_2_3
OpEmitVertex

%v6_pos = OpFSub %v3float %p2_xyz %perp2_scaled
%v6_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v6_pos_4 = OpCompositeInsert %v4float %v6_pos %v6_pos_4 0
%v6_pos_4 = OpCompositeInsert %v4float %v6_pos %v6_pos_4 1
%v6_pos_4 = OpCompositeInsert %v4float %v6_pos %v6_pos_4 2

OpStore %pos_ptr %v6_pos_4
OpStore %outColor %c2
OpEmitVertex

%v7_pos = OpFAdd %v3float %p2_xyz %perp2_scaled
%v7_pos_4 = OpCompositeConstruct %v4float 0 0 0 1
%v7_pos_4 = OpCompositeInsert %v4float %v7_pos %v7_pos_4 0
%v7_pos_4 = OpCompositeInsert %v4float %v7_pos %v7_pos_4 1
%v7_pos_4 = OpCompositeInsert %v4float %v7_pos %v7_pos_4 2

OpStore %pos_ptr %v7_pos_4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
