OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %color Centroid
               OpDecorate %color Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
      %float = OpTypeFloat 32
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%fn_type = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_type
      %entry = OpLabel
               ; Emit vertex 0
        %pos_ptr_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %pos_0 = OpLoad %v4float %pos_ptr_0
        %out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr_0 %pos_0
        %color_ptr_0 = OpAccessChain %ptr_in_v3 %color %int_0
        %col_0 = OpLoad %v3float %color_ptr_0
               OpStore %outColor %col_0
               OpEmitVertex

               ; Emit vertex 1
        %pos_ptr_1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %pos_1 = OpLoad %v4float %pos_ptr_1
               OpStore %out_pos_ptr_0 %pos_1
        %color_ptr_1 = OpAccessChain %ptr_in_v3 %color %int_1
        %col_1 = OpLoad %v3float %color_ptr_1
               OpStore %outColor %col_1
               OpEmitVertex

               ; Emit vertex 2
        %pos_ptr_2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %pos_2 = OpLoad %v4float %pos_ptr_2
               OpStore %out_pos_ptr_0 %pos_2
        %color_ptr_2 = OpAccessChain %ptr_in_v3 %color %int_2
        %col_2 = OpLoad %v3float %color_ptr_2
               OpStore %outColor %col_2
               OpEmitVertex

               ; End primitive
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
