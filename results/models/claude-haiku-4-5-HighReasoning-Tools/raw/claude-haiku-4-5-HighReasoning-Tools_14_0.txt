using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    private int height;
    private int width;
    private char[,] board;
    private bool[,] isSafe;
    private bool[,] isMine;

    static void Main()
    {
        var solver = new MinesweeperSolver();
        solver.Solve();
    }

    void Solve()
    {
        ReadInput();
        SolveMinesweeper();
        OutputResults();
    }

    void ReadInput()
    {
        var firstLine = Console.ReadLine().TrimStart('﻿').Split();
        height = int.Parse(firstLine[0]);
        width = int.Parse(firstLine[1]);

        board = new char[height, width];
        isSafe = new bool[height, width];
        isMine = new bool[height, width];

        for (int i = 0; i < height; i++)
        {
            string line = Console.ReadLine();
            for (int j = 0; j < width; j++)
            {
                board[i, j] = j < line.Length ? line[j] : ' ';

                if (board[i, j] == '*')
                    isMine[i, j] = true;
                else if (board[i, j] != ' ')
                    isSafe[i, j] = true;
            }
        }
    }

    void SolveMinesweeper()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;

            for (int i = 0; i < height; i++)
            {
                for (int j = 0; j < width; j++)
                {
                    if (char.IsDigit(board[i, j]))
                    {
                        int mineCount = int.Parse(board[i, j].ToString());
                        ApplyConstraint(i, j, mineCount, ref changed);
                    }
                }
            }
        }
    }

    void ApplyConstraint(int row, int col, int mineCount, ref bool changed)
    {
        var neighbors = GetNeighbors(row, col);

        int flaggedMines = neighbors.Count(n => isMine[n.Item1, n.Item2]);
        int unknownCount = neighbors.Count(n => !isSafe[n.Item1, n.Item2] && !isMine[n.Item1, n.Item2]);

        if (flaggedMines == mineCount)
        {
            foreach (var (r, c) in neighbors)
            {
                if (!isSafe[r, c] && !isMine[r, c])
                {
                    isSafe[r, c] = true;
                    changed = true;
                }
            }
        }
        else if (flaggedMines + unknownCount == mineCount)
        {
            foreach (var (r, c) in neighbors)
            {
                if (!isSafe[r, c] && !isMine[r, c])
                {
                    isMine[r, c] = true;
                    changed = true;
                }
            }
        }
    }

    List<(int, int)> GetNeighbors(int row, int col)
    {
        var neighbors = new List<(int, int)>();
        for (int dr = -1; dr <= 1; dr++)
        {
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int nr = row + dr;
                int nc = col + dc;
                if (nr >= 0 && nr < height && nc >= 0 && nc < width)
                    neighbors.Add((nr, nc));
            }
        }
        return neighbors;
    }

    void OutputResults()
    {
        var safeCells = new List<(int, int)>();

        for (int i = 0; i < height; i++)
        {
            for (int j = 0; j < width; j++)
            {
                if (isSafe[i, j] && board[i, j] == ' ')
                {
                    safeCells.Add((j, i));
                }
            }
        }

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
        {
            Console.WriteLine($"{x} {y}");
        }
    }
}
