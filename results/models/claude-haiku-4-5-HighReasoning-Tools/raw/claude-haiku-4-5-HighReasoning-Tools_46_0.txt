#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out of bounds invocations
    if (idx >= N) {
        return;
    }

    // Extract parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Load current body data (position+mass and velocity)
    uint bodyIdx = idx * 2;
    vec4 posAndMass = inData[bodyIdx];
    vec4 velData = inData[bodyIdx + 1];

    vec3 pos = posAndMass.xyz;
    float mass = posAndMass.w;
    vec3 vel = velData.xyz;

    // Compute acceleration from gravitational forces
    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        // Skip self-interaction
        if (j == idx) {
            continue;
        }

        // Load other body data
        uint jBodyIdx = j * 2;
        vec4 jPosAndMass = inData[jBodyIdx];

        vec3 jPos = jPosAndMass.xyz;
        float jMass = jPosAndMass.w;

        // Compute relative position vector
        vec3 r = jPos - pos;

        // Apply softening to avoid singularities
        // acc += G * mass_j * r / (|r|^2 + softening^2)^(3/2)
        float rSqPlusSoftening = dot(r, r) + softening * softening;
        float denominator = rSqPlusSoftening * sqrt(rSqPlusSoftening);

        acc += G * jMass * r / denominator;
    }

    // Update velocity using acceleration and dt
    vel += acc * dt;

    // Update position using updated velocity and dt
    pos += vel * dt;

    // Write updated body data to output buffers
    outData[bodyIdx] = vec4(pos, mass);
    outData[bodyIdx + 1] = vec4(vel, 0.0);
}
