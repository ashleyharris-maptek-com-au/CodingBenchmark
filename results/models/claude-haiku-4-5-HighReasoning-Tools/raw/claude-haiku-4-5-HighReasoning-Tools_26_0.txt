use std::io::{self, Read, Write};
use std::f32;

const MAX_ITERATIONS: usize = 100;
const CONVERGENCE_THRESHOLD: f32 = 1e-4;
const STREAMING_BATCH_SIZE: usize = 100_000;
const MINI_BATCH_SIZE: usize = 10_000;

#[derive(Clone)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

fn read_binary_f32(reader: &mut io::BufReader<io::Stdin>, count: usize) -> io::Result<Vec<f32>> {
    let mut buffer = vec![0u8; count * 4];
    reader.read_exact(&mut buffer)?;

    let mut values = Vec::with_capacity(count);
    for i in 0..count {
        let bytes = [buffer[i*4], buffer[i*4+1], buffer[i*4+2], buffer[i*4+3]];
        values.push(f32::from_le_bytes(bytes));
    }
    Ok(values)
}

fn write_binary_u32(values: &[u32]) -> io::Result<()> {
    let mut stdout = io::stdout().lock();
    for &val in values {
        stdout.write_all(&val.to_le_bytes())?;
    }
    stdout.flush()?;
    Ok(())
}

fn initialize_centroids(points: &[Point], k: usize) -> Vec<Point> {
    let mut centroids = Vec::with_capacity(k);
    let n = points.len();

    if n < k {
        for i in 0..n {
            centroids.push(points[i].clone());
        }
        for _ in n..k {
            centroids.push(Point { x: 0.0, y: 0.0, z: 0.0 });
        }
    } else {
        let step = n / k;
        for i in 0..k {
            centroids.push(points[i * step].clone());
        }
    }
    centroids
}

fn assign_clusters(points: &[Point], centroids: &[Point]) -> Vec<u32> {
    points
        .iter()
        .map(|p| {
            let mut best_cluster = 0u32;
            let mut best_dist = f32::INFINITY;

            for (i, c) in centroids.iter().enumerate() {
                let dist = p.distance_squared(c);
                if dist < best_dist {
                    best_dist = dist;
                    best_cluster = i as u32;
                }
            }
            best_cluster
        })
        .collect()
}

fn update_centroids(points: &[Point], assignments: &[u32], k: usize) -> (Vec<Point>, f32) {
    let mut new_centroids = vec![Point { x: 0.0, y: 0.0, z: 0.0 }; k];
    let mut counts = vec![0u32; k];

    for (p, &cluster) in points.iter().zip(assignments.iter()) {
        let c = cluster as usize;
        if c < k {
            new_centroids[c].x += p.x;
            new_centroids[c].y += p.y;
            new_centroids[c].z += p.z;
            counts[c] += 1;
        }
    }

    let mut movement = 0.0f32;
    for i in 0..k {
        if counts[i] > 0 {
            let count = counts[i] as f32;
            new_centroids[i].x /= count;
            new_centroids[i].y /= count;
            new_centroids[i].z /= count;
        } else {
            new_centroids[i] = Point { x: f32::NAN, y: f32::NAN, z: f32::NAN };
        }
    }

    (new_centroids, movement)
}

fn kmeans_full(points: Vec<Point>, k: usize) -> Vec<u32> {
    let mut centroids = initialize_centroids(&points, k);

    for _iter in 0..MAX_ITERATIONS {
        let assignments = assign_clusters(&points, &centroids);
        let (new_centroids, _movement) = update_centroids(&points, &assignments, k);
        centroids = new_centroids;
    }

    assign_clusters(&points, &centroids)
}

fn kmeans_mini_batch(reader: &mut io::BufReader<io::Stdin>, num_points: usize, k: usize) -> io::Result<Vec<u32>> {
    let mut centroids = vec![Point { x: 0.0, y: 0.0, z: 0.0 }; k];
    let mut centroid_counts = vec![0u32; k];
    let mut assignments = Vec::with_capacity(num_points);

    let mut points_read = 0;
    while points_read < num_points {
        let batch_size = std::cmp::min(MINI_BATCH_SIZE, num_points - points_read);
        let coords = read_binary_f32(reader, batch_size * 3)?;

        let mut batch_points = Vec::with_capacity(batch_size);
        for i in 0..batch_size {
            batch_points.push(Point {
                x: coords[i * 3],
                y: coords[i * 3 + 1],
                z: coords[i * 3 + 2],
            });
        }

        let batch_assignments = assign_clusters(&batch_points, &centroids);

        for (p, &cluster) in batch_points.iter().zip(batch_assignments.iter()) {
            let c = cluster as usize;
            if c < k {
                centroids[c].x += p.x;
                centroids[c].y += p.y;
                centroids[c].z += p.z;
                centroid_counts[c] += 1;
            }
            assignments.push(cluster);
        }

        for i in 0..k {
            if centroid_counts[i] > 0 {
                let count = centroid_counts[i] as f32;
                centroids[i].x /= count;
                centroids[i].y /= count;
                centroids[i].z /= count;
            }
        }

        points_read += batch_size;
    }

    Ok(assignments)
}

fn kmeans_streaming(reader: &mut io::BufReader<io::Stdin>, num_points: usize, k: usize) -> io::Result<Vec<u32>> {
    let mut centroids = vec![Point { x: 0.0, y: 0.0, z: 0.0 }; k];
    let mut centroid_counts = vec![0u32; k];
    let mut assignments = Vec::with_capacity(num_points);

    let mut points_read = 0;
    while points_read < num_points {
        let batch_size = std::cmp::min(STREAMING_BATCH_SIZE, num_points - points_read);
        let coords = read_binary_f32(reader, batch_size * 3)?;

        let mut batch_points = Vec::with_capacity(batch_size);
        for i in 0..batch_size {
            batch_points.push(Point {
                x: coords[i * 3],
                y: coords[i * 3 + 1],
                z: coords[i * 3 + 2],
            });
        }

        let batch_assignments = assign_clusters(&batch_points, &centroids);

        for (p, &cluster) in batch_points.iter().zip(batch_assignments.iter()) {
            let c = cluster as usize;
            if c < k {
                let alpha = 1.0 / (centroid_counts[c] as f32 + 1.0);
                centroids[c].x = centroids[c].x * (1.0 - alpha) + p.x * alpha;
                centroids[c].y = centroids[c].y * (1.0 - alpha) + p.y * alpha;
                centroids[c].z = centroids[c].z * (1.0 - alpha) + p.z * alpha;
                centroid_counts[c] += 1;
            }
            assignments.push(cluster);
        }

        points_read += batch_size;
    }

    Ok(assignments)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = io::BufReader::with_capacity(1024 * 1024, stdin.lock());

    let mut header = String::new();
    let mut temp_reader = io::BufReader::new(io::stdin());
    temp_reader.read_line(&mut header)?;
    drop(temp_reader);

    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    let num_points: usize = parts[0].parse().expect("Invalid num_points");
    let k: usize = parts[1].parse().expect("Invalid k");

    let assignments = if num_points <= 10_000_000 {
        let coords = read_binary_f32(&mut reader, num_points * 3)?;
        let mut points = Vec::with_capacity(num_points);
        for i in 0..num_points {
            points.push(Point {
                x: coords[i * 3],
                y: coords[i * 3 + 1],
                z: coords[i * 3 + 2],
            });
        }
        kmeans_full(points, k)
    } else if num_points <= 100_000_000 {
        kmeans_mini_batch(&mut reader, num_points, k)?
    } else {
        kmeans_streaming(&mut reader, num_points, k)?
    };

    write_binary_u32(&assignments)?;
    Ok(())
}
