[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    // UV checkerboard at 8x8 frequency
    int2 checkerCell = int(input.uv * 8.0);
    float checkerColor = ((checkerCell.x + checkerCell.y) % 2 == 0) ? 0.9 : 0.2;

    // Normalize the normal vector
    float3 normal = normalize(input.normal);

    // Light direction from world position to light
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse component
    float diffuse = max(0.0, dot(normal, lightDir));

    // Shading factor: (0.5 + 0.5*diffuse)
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Apply shading to checkerboard pattern
    float finalColor = checkerColor * shadingFactor;

    return float4(finalColor, finalColor, finalColor, 1.0);
}
