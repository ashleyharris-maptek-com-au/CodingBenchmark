OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor

OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_3 = OpConstant %uint 3

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%float_0_333333 = OpConstant %float 0.333333
%float_0_7 = OpConstant %float 0.7

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr_3_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%arr_3_v3 = OpTypeArray %v3float %uint_3

%ptr_in_arr_pv = OpTypePointer Input %arr_3_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_3_v3

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_v4 = OpTypePointer Output %v4float

%func_type = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func_type
%label = OpLabel

%pos_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos_0 = OpLoad %v4float %pos_0_ptr

%pos_1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos_1 = OpLoad %v4float %pos_1_ptr

%pos_2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos_2 = OpLoad %v4float %pos_2_ptr

%color_0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%color_0 = OpLoad %v3float %color_0_ptr

%color_1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%color_1 = OpLoad %v3float %color_1_ptr

%color_2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%color_2 = OpLoad %v3float %color_2_ptr

%sum_01 = OpFAdd %v4float %pos_0 %pos_1
%sum = OpFAdd %v4float %sum_01 %pos_2
%centroid = OpFMul %v4float %sum %float_0_333333

%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%diff_0 = OpFSub %v4float %pos_0 %centroid
%scaled_0 = OpFMul %v4float %diff_0 %float_0_7
%new_pos_0 = OpFAdd %v4float %centroid %scaled_0
OpStore %gl_out_pos_ptr %new_pos_0
OpStore %outColor %color_0
OpEmitVertex

%diff_1 = OpFSub %v4float %pos_1 %centroid
%scaled_1 = OpFMul %v4float %diff_1 %float_0_7
%new_pos_1 = OpFAdd %v4float %centroid %scaled_1
OpStore %gl_out_pos_ptr %new_pos_1
OpStore %outColor %color_1
OpEmitVertex

%diff_2 = OpFSub %v4float %pos_2 %centroid
%scaled_2 = OpFMul %v4float %diff_2 %float_0_7
%new_pos_2 = OpFAdd %v4float %centroid %scaled_2
OpStore %gl_out_pos_ptr %new_pos_2
OpStore %outColor %color_2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
