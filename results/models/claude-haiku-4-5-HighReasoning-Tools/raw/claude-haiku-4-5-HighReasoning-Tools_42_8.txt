OpCapability Geometry

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor

OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Location decorations for user inputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2

; Location decoration for user output
OpDecorate %outColor Location 0

; Type definitions
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0p7 = OpConstant %float 0.7
%float_3p0 = OpConstant %float 3.0
%v4float_3 = OpConstantComposite %v4float %float_3p0 %float_3p0 %float_3p0 %float_3p0

; gl_PerVertex structures
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array types
%array_3_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %int_3
%array_3_v3float = OpTypeArray %v3float %int_3

; Pointer types for inputs
%ptr_Input_array_gl_PerVertex_in = OpTypePointer Input %array_3_gl_PerVertex_in
%ptr_Input_array_v3float = OpTypePointer Input %array_3_v3float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

; Pointer types for outputs
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

; Pointer types for function variables
%ptr_Function_v4float = OpTypePointer Function %v4float

; Function type
%func_void_type = OpTypeFunction %void

; Variable declarations
%gl_in = OpVariable %ptr_Input_array_gl_PerVertex_in Input
%inWorldPos = OpVariable %ptr_Input_array_v3float Input
%inNormal = OpVariable %ptr_Input_array_v3float Input
%inColor = OpVariable %ptr_Input_array_v3float Input

%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

; Main function
%main = OpFunction %void None %func_void_type
%entry_label = OpLabel

; Allocate local variables for positions and centroid
%p0_ptr = OpVariable %ptr_Function_v4float Function
%p1_ptr = OpVariable %ptr_Function_v4float Function
%p2_ptr = OpVariable %ptr_Function_v4float Function
%centroid_ptr = OpVariable %ptr_Function_v4float Function

; Load positions from gl_in[0..2].gl_Position
%pos0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
OpStore %p0_ptr %pos0

%pos1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
OpStore %p1_ptr %pos1

%pos2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
OpStore %p2_ptr %pos2

; Compute centroid: C = (p0 + p1 + p2) / 3
%p0_load = OpLoad %v4float %p0_ptr
%p1_load = OpLoad %v4float %p1_ptr
%sum1 = OpFAdd %v4float %p0_load %p1_load
%p2_load = OpLoad %v4float %p2_ptr
%sum2 = OpFAdd %v4float %sum1 %p2_load
%centroid = OpFDiv %v4float %sum2 %v4float_3
OpStore %centroid_ptr %centroid

; --- Emit Vertex 0 ---
; new_pos = C + 0.7 * (p0 - C)
%centroid0 = OpLoad %v4float %centroid_ptr
%p0_emit0 = OpLoad %v4float %p0_ptr
%diff0 = OpFSub %v4float %p0_emit0 %centroid0
%scaled0 = OpVectorTimesScalar %v4float %diff0 %float_0p7
%new_pos0 = OpFAdd %v4float %centroid0 %scaled0

; Store output position
%out_pos_ptr0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr0 %new_pos0

; Store output color
%color0_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%color0 = OpLoad %v3float %color0_ptr
OpStore %outColor %color0

OpEmitVertex

; --- Emit Vertex 1 ---
; new_pos = C + 0.7 * (p1 - C)
%centroid1 = OpLoad %v4float %centroid_ptr
%p1_emit1 = OpLoad %v4float %p1_ptr
%diff1 = OpFSub %v4float %p1_emit1 %centroid1
%scaled1 = OpVectorTimesScalar %v4float %diff1 %float_0p7
%new_pos1 = OpFAdd %v4float %centroid1 %scaled1

; Store output position
%out_pos_ptr1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr1 %new_pos1

; Store output color
%color1_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1

OpEmitVertex

; --- Emit Vertex 2 ---
; new_pos = C + 0.7 * (p2 - C)
%centroid2 = OpLoad %v4float %centroid_ptr
%p2_emit2 = OpLoad %v4float %p2_ptr
%diff2 = OpFSub %v4float %p2_emit2 %centroid2
%scaled2 = OpVectorTimesScalar %v4float %diff2 %float_0p7
%new_pos2 = OpFAdd %v4float %centroid2 %scaled2

; Store output position
%out_pos_ptr2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr2 %new_pos2

; Store output color
%color2_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2

OpEmitVertex

; End the primitive strip
OpEndPrimitive

; Return from function
OpReturn

OpFunctionEnd
