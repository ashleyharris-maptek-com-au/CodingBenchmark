#include <iostream>
#include <fstream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <vector>
#include <cmath>
#include <chrono>
#include <atomic>
#include <queue>
#include <iomanip>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Command {
    double thrust, pitch_cmd, yaw_cmd, roll_cmd;
};

struct TerrainCell {
    uint16_t ground_top, floating_bottom, floating_top;
};

int width, depth, height;
double gravity, max_thrust;
double target_x, target_y, target_z;
vector<vector<TerrainCell>> terrain;

mutex state_mutex;
State current_state;
bool state_updated = false;
condition_variable state_cv;

atomic<bool> running(true);
atomic<bool> terrain_loaded(false);

void read_terrain(const string& filepath) {
    ifstream file(filepath, ios::binary);
    if (!file) return;

    terrain.resize(width, vector<TerrainCell>(depth));

    for (int x = 0; x < width; x++) {
        for (int y = 0; y < depth; y++) {
            uint16_t a, b, c;
            file.read((char*)&a, 2);
            file.read((char*)&b, 2);
            file.read((char*)&c, 2);
            terrain[x][y] = {a, b, c};
        }
    }
    terrain_loaded = true;
}

bool check_collision(double x, double y, double z) {
    if (x < 0 || x >= width || y < 0 || y >= depth || z < 0 || z >= height) {
        return true;
    }

    int xi = (int)x;
    int yi = (int)y;
    if (xi >= width) xi = width - 1;
    if (yi >= depth) yi = depth - 1;

    auto& cell = terrain[xi][yi];

    if (z < cell.ground_top) return true;
    if (z >= cell.floating_bottom && z < cell.floating_top) return true;

    return false;
}

bool is_landed(const State& s) {
    double dist = sqrt((s.x - target_x) * (s.x - target_x) +
                       (s.y - target_y) * (s.y - target_y));
    double landing_radius = max(5.0, min(width, depth) * 0.02);
    double speed = sqrt(s.vx * s.vx + s.vy * s.vy + s.vz * s.vz);

    return dist < landing_radius && speed < 3.0 && s.z <= target_z + 1.0;
}

void input_thread_fn() {
    string line;
    while (running && getline(cin, line)) {
        if (line == "STATE") {
            break;
        }
    }

    while (running) {
        State s;
        if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                   >> s.pitch >> s.yaw >> s.roll
                   >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
            running = false;
            state_cv.notify_all();
            break;
        }

        {
            lock_guard<mutex> lock(state_mutex);
            current_state = s;
            state_updated = true;
        }
        state_cv.notify_all();
    }
}

Command compute_control(const State& s) {
    Command cmd = {0, 0, 0, 0};

    if (!terrain_loaded || is_landed(s)) {
        return cmd;
    }

    // Distance to target
    double dx = target_x - s.x;
    double dy = target_y - s.y;
    double dz = target_z - s.z;
    double horiz_dist = sqrt(dx*dx + dy*dy);

    // Altitude control
    double descent_rate = 0.5;
    if (horiz_dist > 50) {
        descent_rate = 1.0;
    }

    double desired_vz = -descent_rate;
    double vz_error = s.vz - desired_vz;
    double thrust_needed = (gravity + vz_error * 0.3) / (gravity * 10.0);
    thrust_needed = max(0.0, min(1.0, thrust_needed));
    cmd.thrust = thrust_needed;

    // Lateral control
    if (horiz_dist > 2.0) {
        double target_angle = atan2(dy, dx);
        double yaw_error = target_angle - s.yaw;
        while (yaw_error > M_PI) yaw_error -= 2 * M_PI;
        while (yaw_error < -M_PI) yaw_error += 2 * M_PI;
        cmd.yaw_cmd = max(-1.0, min(1.0, yaw_error * 0.5));

        double pitch_cmd = max(-1.0, min(1.0, horiz_dist * 0.05));
        cmd.pitch_cmd = pitch_cmd * cos(yaw_error);
    } else {
        cmd.pitch_cmd = max(-1.0, min(1.0, s.vx * 0.5));
        cmd.yaw_cmd = max(-1.0, min(1.0, s.vy * 0.5));
    }

    // Damping for landing
    if (horiz_dist < 20) {
        cmd.thrust *= (1.0 - 0.03 * (20 - horiz_dist));
        cmd.pitch_cmd *= 0.7;
        cmd.yaw_cmd *= 0.7;
    }

    return cmd;
}

void control_loop() {
    auto last_output = chrono::steady_clock::now();

    while (running) {
        {
            unique_lock<mutex> lock(state_mutex);
            state_cv.wait_for(lock, chrono::milliseconds(150), [] { return state_updated; });
            state_updated = false;
        }

        if (!running) break;

        State s;
        {
            lock_guard<mutex> lock(state_mutex);
            s = current_state;
        }

        Command cmd = compute_control(s);

        cout << fixed << setprecision(6)
             << cmd.thrust << " "
             << cmd.pitch_cmd << " "
             << cmd.yaw_cmd << " "
             << cmd.roll_cmd << "\n";
        cout.flush();

        auto now = chrono::steady_clock::now();
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(now - last_output).count();
        if (elapsed < 100) {
            this_thread::sleep_for(chrono::milliseconds(100 - elapsed));
        }
        last_output = chrono::steady_clock::now();
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    double start_x, start_y, start_z;
    string terrain_path;

    cin >> width >> depth >> height >> gravity >> max_thrust;
    double fuel;
    cin >> fuel;
    cin >> start_x >> start_y >> start_z;
    cin >> target_x >> target_y >> target_z;
    cin.ignore();
    getline(cin, terrain_path);

    current_state = {start_x, start_y, start_z, 0, 0, 0, 0, 0, 0, 0, 0, 0, fuel};

    // Load terrain in background
    thread terrain_thread(read_terrain, terrain_path);

    // Start input and control threads
    thread input_thread(input_thread_fn);
    thread control_thread(control_loop);

    // Wait for threads
    input_thread.join();
    control_thread.join();
    terrain_thread.join();

    return 0;
}
