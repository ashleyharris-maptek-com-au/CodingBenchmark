OpCapability Geometry
               OpCapability Float64
               OpCapability GeometryStreams
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %arr_pv_in ArrayStride 16
               OpDecorate %gl_in Binding 0
               OpDecorate %gl_in DescriptorSet 0
               OpDecorate %color_in Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
     %float = OpTypeFloat 32
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0
      %v2float = OpTypeVector %float 2
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4
   %gl_PerVertex_in = OpTypeStruct %v4float
    %uint_4 = OpConstant %uint 4
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
   %gl_in = OpVariable %ptr_in_arr_pv Input
%arr_color_in = OpTypeArray %v3float %uint_4
%ptr_in_color = OpTypePointer Input %arr_color_in
 %color_in = OpVariable %ptr_in_color Input
  %gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
   %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
 %outColor = OpVariable %ptr_out_v3 Output
%func_type = OpTypeFunction %void
   %int_0 = OpConstant %int 0
   %int_1 = OpConstant %int 1
   %int_2 = OpConstant %int 2
   %int_3 = OpConstant %int 3
   %float_0 = OpConstant %float 0.0
   %float_1 = OpConstant %float 1.0
   %float_0p025 = OpConstant %float 0.025
   %float_neg1 = OpConstant %float -1.0
   %uint_0 = OpConstant %uint 0
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%glsl_ext_inst = OpExtInstImport "GLSL.std.450"
       %main = OpFunction %void None %func_type
    %entry = OpLabel
    ; Load all 4 input positions
    %pos0_chain = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %pos0 = OpLoad %v4float %pos0_chain
    %pos1_chain = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
    %pos1 = OpLoad %v4float %pos1_chain
    %pos2_chain = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
    %pos2 = OpLoad %v4float %pos2_chain
    %pos3_chain = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
    %pos3 = OpLoad %v4float %pos3_chain

    ; Extract xyz components
    %pos0_xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
    %pos1_xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
    %pos2_xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
    %pos3_xyz = OpVectorShuffle %v3float %pos3 %pos3 0 1 2

    ; Compute tangent at start: t0 = normalize(pos2 - pos0)
    %vec_t0 = OpFSub %v3float %pos2_xyz %pos0_xyz
    %t0 = OpExtInst %v3float %glsl_ext_inst Normalize %vec_t0

    ; Compute tangent at end: t1 = normalize(pos3 - pos1)
    %vec_t1 = OpFSub %v3float %pos3_xyz %pos1_xyz
    %t1 = OpExtInst %v3float %glsl_ext_inst Normalize %vec_t1

    ; Load input colors
    %color1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
    %color1 = OpLoad %v3float %color1_ptr
    %color2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
    %color2 = OpLoad %v3float %color2_ptr

    ; Compute perpendiculars from tangents (2D: perp = (-y, x))
    %t0_y = OpCompositeExtract %float %t0 1
    %t0_x = OpCompositeExtract %float %t0 0
    %perp0_x = OpFNegate %float %t0_y
    %perp0_2d = OpCompositeConstruct %v2float %perp0_x %t0_x
    %perp0 = OpCompositeConstruct %v3float %perp0_2d %float_0

    %t1_y = OpCompositeExtract %float %t1 1
    %t1_x = OpCompositeExtract %float %t1 0
    %perp1_x = OpFNegate %float %t1_y
    %perp1_2d = OpCompositeConstruct %v2float %perp1_x %t1_x
    %perp1 = OpCompositeConstruct %v3float %perp1_2d %float_0

    ; Scale perpendiculars by thickness
    %perp0_thick = OpVectorTimesScalar %v3float %perp0 %float_0p025
    %perp1_thick = OpVectorTimesScalar %v3float %perp1 %float_0p025

    ; Get output position pointer
    %out_pos_chain = OpAccessChain %ptr_out_v4 %gl_out %int_0

    ; Emit vertex 0: start + perp0*thickness
    %v0_pos = OpFAdd %v3float %pos1_xyz %perp0_thick
    %v0_pos4 = OpCompositeConstruct %v4float %v0_pos %float_1
    OpStore %out_pos_chain %v0_pos4
    OpStore %outColor %color1
    OpEmitVertex

    ; Emit vertex 1: start - perp0*thickness
    %neg_perp0 = OpVectorTimesScalar %v3float %perp0_thick %float_neg1
    %v1_pos = OpFAdd %v3float %pos1_xyz %neg_perp0
    %v1_pos4 = OpCompositeConstruct %v4float %v1_pos %float_1
    OpStore %out_pos_chain %v1_pos4
    OpStore %outColor %color1
    OpEmitVertex

    ; Emit vertex 2: end + perp1*thickness
    %v2_pos = OpFAdd %v3float %pos2_xyz %perp1_thick
    %v2_pos4 = OpCompositeConstruct %v4float %v2_pos %float_1
    OpStore %out_pos_chain %v2_pos4
    OpStore %outColor %color2
    OpEmitVertex

    ; Emit vertex 3: end - perp1*thickness
    %neg_perp1 = OpVectorTimesScalar %v3float %perp1_thick %float_neg1
    %v3_pos = OpFAdd %v3float %pos2_xyz %neg_perp1
    %v3_pos4 = OpCompositeConstruct %v4float %v3_pos %float_1
    OpStore %out_pos_chain %v3_pos4
    OpStore %outColor %color2
    OpEmitVertex

    ; End primitive
    OpEndPrimitive
    OpReturn
               OpFunctionEnd
