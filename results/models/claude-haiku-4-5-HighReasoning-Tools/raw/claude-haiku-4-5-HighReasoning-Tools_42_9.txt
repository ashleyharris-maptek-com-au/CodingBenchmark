OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport %glsl "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %inWorldPos %inNormal %inColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for built-in structures
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input/output variable decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Type definitions
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0_025 = OpConstant %float 0.025

; Built-in structures
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array types
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_4
%arr_v3_in = OpTypeArray %v3float %int_4

; Pointer types
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

; Variables
%gl_in = OpVariable %ptr_Input_arr_pv Input
%gl_out = OpVariable %ptr_Output_pv Output
%inWorldPos = OpVariable %ptr_Input_arr_v3 Input
%inNormal = OpVariable %ptr_Input_arr_v3 Input
%inColor = OpVariable %ptr_Input_arr_v3 Input
%outColor = OpVariable %ptr_Output_v3 Output

; Function type
%func_void = OpTypeFunction %void

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load all 4 input positions
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

; Load colors
%col1_ptr = OpAccessChain %ptr_Input_v3 %inColor %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_Input_v3 %inColor %int_2
%col2 = OpLoad %v3float %col2_ptr

; Extract 2D positions
%p0_2d = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_2d = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_2d = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_2d = OpVectorShuffle %v2float %p3 %p3 0 1

; Compute t0 = normalize(p2.xy - p0.xy)
%t0_diff = OpFSub %v2float %p2_2d %p0_2d
%t0 = OpExtInst %v2float %glsl Normalize %t0_diff

; Compute t1 = normalize(p3.xy - p1.xy)
%t1_diff = OpFSub %v2float %p3_2d %p1_2d
%t1 = OpExtInst %v2float %glsl Normalize %t1_diff

; Compute perpendiculars: perp0 = vec2(-t0.y, t0.x)
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%perp0_x = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %perp0_x %t0_x

; Compute perpendiculars: perp1 = vec2(-t1.y, t1.x)
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%perp1_x = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %perp1_x %t1_x

; Scale perpendiculars by thickness
%perp0_thick = OpVectorTimesScalar %v2float %perp0 %float_0_025
%perp1_thick = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Emit vertex 0: start + perp0 * thickness
%v0_xy = OpFAdd %v2float %p1_2d %perp0_thick
%p1_zw = OpVectorShuffle %v2float %p1 %p1 2 3
%v0 = OpCompositeConstruct %v4float %v0_xy %p1_zw
%v0_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v0_ptr %v0
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 1: start - perp0 * thickness
%perp0_neg = OpFNegate %v2float %perp0_thick
%v1_xy = OpFAdd %v2float %p1_2d %perp0_neg
%v1 = OpCompositeConstruct %v4float %v1_xy %p1_zw
%v1_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v1_ptr %v1
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 2: end + perp1 * thickness
%v2_xy = OpFAdd %v2float %p2_2d %perp1_thick
%p2_zw = OpVectorShuffle %v2float %p2 %p2 2 3
%v2 = OpCompositeConstruct %v4float %v2_xy %p2_zw
%v2_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v2_ptr %v2
OpStore %outColor %col2
OpEmitVertex

; Emit vertex 3: end - perp1 * thickness
%perp1_neg = OpFNegate %v2float %perp1_thick
%v3_xy = OpFAdd %v2float %p2_2d %perp1_neg
%v3 = OpCompositeConstruct %v4float %v3_xy %p2_zw
%v3_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v3_ptr %v3
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
