struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Map UV coordinates to complex plane: c = (uv.x*3-2, uv.y*2-1)
    float2 c = float2(input.uv.x * 3.0 - 2.0, input.uv.y * 2.0 - 1.0);

    // Start with z = 0
    float2 z = float2(0.0, 0.0);

    // Count iterations
    int iterations = 0;

    // Iterate up to 20 steps
    for (int i = 0; i < 20; i++)
    {
        // Check escape condition: |z|^2 > 4
        float z_mag_sq = z.x * z.x + z.y * z.y;
        if (z_mag_sq > 4.0)
        {
            break;
        }

        // z = z^2 + c
        // (a + bi)^2 = a^2 - b^2 + 2abi
        float z_real = z.x * z.x - z.y * z.y + c.x;
        float z_imag = 2.0 * z.x * z.y + c.y;

        z = float2(z_real, z_imag);
        iterations++;
    }

    // Normalize iterations to [0, 1]
    float t = float(iterations) / 20.0;

    // Color: (t, t*0.5, 1.0-t), alpha 1.0
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
