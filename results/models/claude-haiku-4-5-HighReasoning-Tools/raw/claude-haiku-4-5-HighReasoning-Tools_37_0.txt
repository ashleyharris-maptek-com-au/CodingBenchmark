using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            line = Console.ReadLine().Split();
            int u = int.Parse(line[0]);
            int v = int.Parse(line[1]);
            adj[u].Add(v);
        }

        var fvs = FindFVS(n, adj);

        Console.WriteLine(fvs.Count);
        if (fvs.Count > 0)
            Console.WriteLine(string.Join(" ", fvs.OrderBy(x => x)));
    }

    static List<int> FindFVS(int n, List<int>[] adj)
    {
        var fvs = new List<int>();
        bool[] removed = new bool[n];

        while (true)
        {
            var backEdges = FindBackEdges(n, adj, removed);

            if (backEdges.Count == 0)
                break;

            // Count frequency of vertices in back edges
            int[] count = new int[n];
            foreach (var (u, v) in backEdges)
            {
                count[u]++;
                count[v]++;
            }

            // Find vertex to remove (highest count)
            int toRemove = -1;
            int maxCount = -1;
            for (int i = 0; i < n; i++)
            {
                if (!removed[i] && count[i] > maxCount)
                {
                    maxCount = count[i];
                    toRemove = i;
                }
            }

            if (toRemove == -1)
                break;

            fvs.Add(toRemove);
            removed[toRemove] = true;
        }

        return fvs;
    }

    static List<(int, int)> FindBackEdges(int n, List<int>[] adj, bool[] removed)
    {
        var backEdges = new List<(int, int)>();
        int[] color = new int[n]; // 0: white, 1: gray, 2: black

        for (int i = 0; i < n; i++)
        {
            if (!removed[i] && color[i] == 0)
                DfsBackEdges(i, adj, color, removed, backEdges);
        }

        return backEdges;
    }

    static void DfsBackEdges(int u, List<int>[] adj, int[] color, bool[] removed, List<(int, int)> backEdges)
    {
        color[u] = 1; // gray

        foreach (int v in adj[u])
        {
            if (removed[v])
                continue;

            if (color[v] == 1)
                backEdges.Add((u, v)); // back edge found
            else if (color[v] == 0)
                DfsBackEdges(v, adj, color, removed, backEdges);
        }

        color[u] = 2; // black
    }
}
