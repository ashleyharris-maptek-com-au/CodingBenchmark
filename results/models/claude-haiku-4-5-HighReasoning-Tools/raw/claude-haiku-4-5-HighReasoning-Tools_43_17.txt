[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv;

    // Scale UV by 8x16
    float2 scaledUV = uv * float2(8.0, 16.0);

    // Get row for staggering
    float row = floor(scaledUV.y);

    // Apply half-brick offset to odd rows
    float offset = fmod(row, 2.0) > 0.5 ? 0.5 : 0.0;
    float adjustedX = scaledUV.x + offset;

    // Get fractional parts
    float fracX = frac(adjustedX);
    float fracY = frac(scaledUV.y);

    // Detect mortar regions (0.05 thickness in X, 0.1 in Y)
    float mortarMaskX = step(fracX, 0.05) + step(0.95, fracX);
    float mortarMaskY = step(fracY, 0.1) + step(0.9, fracY);
    float isMortar = max(mortarMaskX, mortarMaskY);

    // Colors
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    float3 baseColor = lerp(brickColor, mortarColor, isMortar);

    // Lambertian shading
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(normal, lightDir));
    float shadingFactor = 0.4 + 0.6 * diffuse;

    float3 finalColor = baseColor * shadingFactor;

    return float4(finalColor, 1.0);
}
