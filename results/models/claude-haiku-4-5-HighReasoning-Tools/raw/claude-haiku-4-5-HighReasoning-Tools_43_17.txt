[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    float2 uv = input.uv;

    // Scale UV for brick pattern: 8x16 means 8 bricks wide, 16 tall
    float2 scaled = uv * float2(8.0, 16.0);

    // Get row for staggering (every other row offset by 0.5)
    float row = floor(scaled.y);
    float offsetX = fmod(row, 2.0) * 0.5;

    // Apply offset and get brick coordinates
    float2 brickCoord = scaled + float2(offsetX, 0.0);

    // Get position within the brick cell (0-1)
    float2 cellPos = frac(brickCoord);

    // Mortar thickness: 0.05 in X (vertical lines), 0.1 in Y (horizontal lines)
    float mortarX = 0.05;
    float mortarY = 0.1;

    // Check if in mortar (at the start of each cell)
    bool inVerticalMortar = cellPos.x < mortarX;
    bool inHorizontalMortar = cellPos.y < mortarY;
    bool inMortar = inVerticalMortar || inHorizontalMortar;

    // Colors
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 baseColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading: factor = 0.4 + 0.6*diffuse
    float3 norm = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(norm, lightDir), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    float3 finalColor = baseColor * shadingFactor;

    return float4(finalColor, 1.0);
}
