; SPIR-V

; Generator: Custom
; Bound: 90
; Schema: 0
OpCapability Shader
OpExtInstImport "GLSL.std.450" %glslExt
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBOBlock Block
OpMemberDecorate %UBOBlock 0 Offset 0
OpMemberDecorate %UBOBlock 1 Offset 64
OpMemberDecorate %UBOBlock 2 Offset 128
OpMemberDecorate %UBOBlock 3 Offset 192
OpMemberDecorate %UBOBlock 4 Offset 208
OpMemberDecorate %UBOBlock 5 Offset 224
OpDecorate %UBOVar DescriptorSet 0
OpDecorate %UBOVar Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBOBlock = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%UBOPtr = OpTypePointer Uniform %UBOBlock
%vec3InputPtr = OpTypePointer Input %vec3
%vec2InputPtr = OpTypePointer Input %vec2
%vec4OutputPtr = OpTypePointer Output %vec4
%vec4UniformPtr = OpTypePointer Uniform %vec4

%worldPos = OpVariable %vec3InputPtr Input
%normal = OpVariable %vec3InputPtr Input
%uv = OpVariable %vec2InputPtr Input
%tangent = OpVariable %vec3InputPtr Input
%color = OpVariable %vec3InputPtr Input
%outColor = OpVariable %vec4OutputPtr Output

%UBOVar = OpVariable %UBOPtr Uniform

%const0i = OpConstant %float 0.0
%const1f = OpConstant %float 1.0

%voidFn = OpTypeFunction %void

%main = OpFunction %void None %voidFn
%entry = OpLabel

; Load input variables
%worldPosVal = OpLoad %vec3 %worldPos
%normalVal = OpLoad %vec3 %normal

; Access lightPos from UBO (member 3)
%int0 = OpConstant %float 0
%int3 = OpConstant %float 3
%lightPosPtr = OpAccessChain %vec4UniformPtr %UBOVar %int0 %int3
%lightPosFull = OpLoad %vec4 %lightPosPtr
%lightPos = OpVectorShuffle %vec3 %lightPosFull %lightPosFull 0 1 2

; Calculate N = normalize(normal)
%N = OpExtInst %vec3 %glslExt Normalize %normalVal

; Calculate L = normalize(lightPos.xyz - worldPos)
%lightDir = OpFSub %vec3 %lightPos %worldPosVal
%L = OpExtInst %vec3 %glslExt Normalize %lightDir

; Calculate d = max(dot(N, L), 0)
%dotNL = OpDot %float %N %L
%intensity = OpExtInst %float %glslExt FMax %dotNL %const0i

; Construct output color: grayscale intensity with alpha 1.0
%outColorVal = OpCompositeConstruct %vec4 %intensity %intensity %intensity %const1f
OpStore %outColor %outColorVal

OpReturn
OpFunctionEnd
