import sys
import math
from typing import List, Tuple

def read_input():
    line = input().split()
    N, K, capacity = int(line[0]), int(line[1]), int(line[2])

    depot = list(map(int, input().split()))

    customers = []
    for _ in range(N):
        x, y, demand = list(map(int, input().split()))
        customers.append((x, y, demand))

    return N, K, capacity, depot, customers

def distance(p1: Tuple[int, int], p2: Tuple[int, int]) -> float:
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def route_distance(route: List[int], depot: Tuple[int, int], customers: List[Tuple[int, int, int]]) -> float:
    total = 0
    current = depot
    for idx in route:
        customer = customers[idx]
        total += distance(current, (customer[0], customer[1]))
        current = (customer[0], customer[1])
    total += distance(current, depot)
    return total

def solve_cvrp(N: int, K: int, capacity: int, depot: Tuple[int, int], customers: List[Tuple[int, int, int]]) -> Tuple[List[List[int]], float]:
    routes = []
    unvisited = set(range(N))

    # Greedy nearest neighbor construction with capacity constraints
    while unvisited:
        route = []
        current_load = 0
        current_pos = depot

        # Start a new route with the nearest unvisited customer
        best_next = None
        best_dist = float('inf')
        for idx in unvisited:
            d = distance(current_pos, (customers[idx][0], customers[idx][1]))
            if d < best_dist:
                best_dist = d
                best_next = idx

        if best_next is None:
            break

        route.append(best_next)
        current_load = customers[best_next][2]
        current_pos = (customers[best_next][0], customers[best_next][1])
        unvisited.remove(best_next)

        # Greedily add more customers to this route
        while unvisited:
            best_next = None
            best_dist = float('inf')

            for idx in unvisited:
                if current_load + customers[idx][2] <= capacity:
                    d = distance(current_pos, (customers[idx][0], customers[idx][1]))
                    if d < best_dist:
                        best_dist = d
                        best_next = idx

            if best_next is None:
                break

            route.append(best_next)
            current_load += customers[best_next][2]
            current_pos = (customers[best_next][0], customers[best_next][1])
            unvisited.remove(best_next)

        routes.append(route)

        if len(routes) >= K:
            # Force remaining customers into the last route if needed
            if unvisited and K > 0:
                for idx in list(unvisited):
                    if current_load + customers[idx][2] <= capacity:
                        routes[-1].append(idx)
                        current_load += customers[idx][2]
                        unvisited.remove(idx)
            break

    # If there are still unvisited customers, add them to existing routes or create new ones
    while unvisited:
        best_route_idx = -1
        best_customer_idx = -1
        best_position = -1
        best_increase = float('inf')

        for customer_idx in unvisited:
            for route_idx, route in enumerate(routes):
                current_load = sum(customers[idx][2] for idx in route)
                if current_load + customers[customer_idx][2] <= capacity:
                    # Try inserting at each position
                    for pos in range(len(route) + 1):
                        # Calculate distance increase
                        if pos == 0:
                            prev_pos = depot
                        else:
                            prev_pos = (customers[route[pos-1]][0], customers[route[pos-1]][1])

                        if pos == len(route):
                            next_pos = depot
                        else:
                            next_pos = (customers[route[pos]][0], customers[route[pos]][1])

                        curr_pos = (customers[customer_idx][0], customers[customer_idx][1])

                        increase = distance(prev_pos, curr_pos) + distance(curr_pos, next_pos) - distance(prev_pos, next_pos)

                        if increase < best_increase:
                            best_increase = increase
                            best_route_idx = route_idx
                            best_customer_idx = customer_idx
                            best_position = pos

        if best_route_idx >= 0:
            routes[best_route_idx].insert(best_position, best_customer_idx)
            unvisited.remove(best_customer_idx)
        elif len(routes) < K:
            # Create a new route for the remaining customer
            new_route = [unvisited.pop()]
            routes.append(new_route)
        else:
            # Force it into the largest route with most remaining capacity
            best_route_idx = 0
            for i, route in enumerate(routes):
                current_load = sum(customers[idx][2] for idx in route)
                best_load = sum(customers[idx][2] for idx in routes[best_route_idx])
                if capacity - current_load > capacity - best_load:
                    best_route_idx = i
            routes[best_route_idx].append(unvisited.pop())

    # Calculate total distance
    total_dist = sum(route_distance(route, depot, customers) for route in routes)

    return routes, total_dist

def main():
    N, K, capacity, depot, customers = read_input()
    routes, total_dist = solve_cvrp(N, K, capacity, depot, customers)

    print(f"{total_dist:.1f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    main()
