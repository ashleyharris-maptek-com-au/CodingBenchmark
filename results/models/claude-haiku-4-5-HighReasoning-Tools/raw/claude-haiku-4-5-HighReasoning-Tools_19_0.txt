using System;
using System.Collections.Generic;
using System.Linq;

class Mining
{
    struct Node
    {
        public int X, Y, Cost;
    }

    static void Main()
    {
        string header = Console.ReadLine();
        var parts = header.Split(' ');
        int xv = int.Parse(parts[0]);
        int yv = int.Parse(parts[1]);
        int zv = int.Parse(parts[2]);
        double xb = double.Parse(parts[3]);
        double yb = double.Parse(parts[4]);
        double zb = double.Parse(parts[5]);

        byte[,,] voxels = new byte[xv, yv, zv];

        // Read voxel data from stdin
        for (int x = 0; x < xv; x++)
        {
            for (int y = 0; y < yv; y++)
            {
                for (int z = 0; z < zv; z++)
                {
                    int val = Console.Read();
                    if (val == -1) break;
                    voxels[x, y, z] = (byte)val;
                }
            }
        }

        var operations = new List<string>();

        // Find all accessible voxels with value
        var oreList = new List<(int x, int y, int z, int mass, int value)>();

        for (int x = 0; x < xv; x++)
        {
            for (int y = 0; y < yv; y++)
            {
                for (int z = 0; z < zv; z++)
                {
                    byte b = voxels[x, y, z];
                    if (b > 0)
                    {
                        int mass = b % 16;
                        int value = b / 16;
                        if (value > 0 && CanDig(voxels, x, y, z, xv, yv, zv))
                        {
                            oreList.Add((x, y, z, mass, value));
                        }
                    }
                }
            }
        }

        // Sort by total value descending (value * mass)
        oreList.Sort((a, b) => (b.value * b.mass).CompareTo(a.value * a.mass));

        // Process ore - limit to avoid timeout
        int processed = 0;
        int maxOps = Math.Min(200, oreList.Count);

        foreach (var ore in oreList.Take(maxOps))
        {
            int x = ore.x, y = ore.y, z = ore.z;

            if (!CanDig(voxels, x, y, z, xv, yv, zv))
                continue;

            // Try to find path to processing plant
            var path = FindPath(voxels, x, y, z, 0, 0, xv, yv, zv);

            if (path != null && path.Count > 0)
            {
                string op = $"{path.Count} {x} {y} {z}";
                foreach (var p in path)
                {
                    op += $" {p.Item1} {p.Item2}";
                }
                op += " 0 0";
                operations.Add(op);
                voxels[x, y, z] = 0;
                processed++;

                if (processed >= 100)
                    break;
            }
        }

        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }

    static bool CanDig(byte[,,] voxels, int x, int y, int z, int xv, int yv, int zv)
    {
        if (voxels[x, y, z] == 0) return false;

        // Exposed from above
        if (z == zv - 1 || voxels[x, y, z + 1] == 0) return true;

        // Exposed from side
        if ((x > 0 && voxels[x - 1, y, z] == 0) || (x < xv - 1 && voxels[x + 1, y, z] == 0) ||
            (y > 0 && voxels[x, y - 1, z] == 0) || (y < yv - 1 && voxels[x, y + 1, z] == 0))
            return true;

        return false;
    }

    static List<Tuple<int, int>> FindPath(byte[,,] voxels, int fromX, int fromY, int fromZ,
        int toX, int toY, int xv, int yv, int zv)
    {
        var cameFrom = new Dictionary<Tuple<int, int>, Tuple<int, int>>();
        var gScore = new Dictionary<Tuple<int, int>, int>();
        var visited = new HashSet<Tuple<int, int>>();
        var open = new List<Node>();

        var start = Tuple.Create(fromX, fromY);
        open.Add(new Node { X = fromX, Y = fromY, Cost = 0 });
        gScore[start] = 0;

        int iterations = 0;
        int maxIterations = 50000;

        while (open.Count > 0 && iterations < maxIterations)
        {
            iterations++;

            // Get node with lowest cost
            int minIdx = 0;
            for (int i = 1; i < open.Count; i++)
            {
                if (open[i].Cost < open[minIdx].Cost)
                    minIdx = i;
            }
            var current = open[minIdx];
            open.RemoveAt(minIdx);

            var currTuple = Tuple.Create(current.X, current.Y);
            if (visited.Contains(currTuple))
                continue;
            visited.Add(currTuple);

            if (current.X == toX && current.Y == toY)
            {
                var path = new List<Tuple<int, int>> { currTuple };
                var curr = currTuple;
                while (cameFrom.ContainsKey(curr))
                {
                    curr = cameFrom[curr];
                    path.Add(curr);
                }
                path.Reverse();
                return path.ConvertAll(t => new Tuple<int, int>(t.Item1, t.Item2));
            }

            int h1 = GetHeight(voxels, current.X, current.Y, zv);

            int[][] dirs = new int[][] { new int[] { 0, 1 }, new int[] { 0, -1 }, new int[] { 1, 0 }, new int[] { -1, 0 } };
            foreach (var dir in dirs)
            {
                int nx = current.X + dir[0];
                int ny = current.Y + dir[1];

                if (nx < 0 || nx >= xv || ny < 0 || ny >= yv) continue;

                var nTuple = Tuple.Create(nx, ny);
                if (visited.Contains(nTuple)) continue;

                int h2 = GetHeight(voxels, nx, ny, zv);

                // Check slope constraint
                if (Math.Abs(h2 - h1) > 1) continue;

                int tentativeG = gScore[currTuple] + 1;

                if (!gScore.ContainsKey(nTuple) || tentativeG < gScore[nTuple])
                {
                    cameFrom[nTuple] = currTuple;
                    gScore[nTuple] = tentativeG;
                    int h = Math.Abs(nx - toX) + Math.Abs(ny - toY);
                    int priority = tentativeG + h;
                    open.Add(new Node { X = nx, Y = ny, Cost = priority });
                }
            }
        }

        return null;
    }

    static int GetHeight(byte[,,] voxels, int x, int y, int zv)
    {
        for (int z = zv - 1; z >= 0; z--)
        {
            if (voxels[x, y, z] != 0)
                return z + 1;
        }
        return 0;
    }
}
