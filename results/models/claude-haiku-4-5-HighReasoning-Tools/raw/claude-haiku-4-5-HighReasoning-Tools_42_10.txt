OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor

OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %gl_in BuiltIn
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %UniformBlock Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UniformBlock 0 Offset 0
OpMemberDecorate %UniformBlock 1 Offset 64
OpMemberDecorate %UniformBlock 2 Offset 128
OpMemberDecorate %UniformBlock 3 Offset 192
OpMemberDecorate %UniformBlock 4 Offset 208
OpMemberDecorate %UniformBlock 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4x4f = OpTypeMatrix %v4float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_neg1 = OpConstant %float -1.0
%float_epsilon = OpConstant %float 0.0001

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr4_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr4_v3float = OpTypeArray %v3float %uint_4

%ptr_in_arr4_pv = OpTypePointer Input %arr4_pv_in
%ptr_in_arr4_v3 = OpTypePointer Input %arr4_v3float
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3float = OpTypePointer Output %v3float
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_function_v3 = OpTypePointer Function %v3float
%ptr_function_v4 = OpTypePointer Function %v4float
%ptr_function_float = OpTypePointer Function %float
%ptr_function_int = OpTypePointer Function %int

%UniformBlock = OpTypeStruct %mat4x4f %mat4x4f %mat4x4f %v3float %v3float %v3float
%ptr_uniform_UniformBlock = OpTypePointer Uniform %UniformBlock

%gl_in = OpVariable %ptr_in_arr4_pv Input
%inWorldPos = OpVariable %ptr_in_arr4_v3 Input
%inNormal = OpVariable %ptr_in_arr4_v3 Input
%inColor = OpVariable %ptr_in_arr4_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output
%ubo = OpVariable %ptr_uniform_UniformBlock Uniform

%func_type = OpTypeFunction %void
%func_v3_float_v3_v3_v3_to_v3 = OpTypeFunction %v3float %v3float %float %v3float %v3float %v3float

%catmull_rom = OpFunction %v3float None %func_v3_float_v3_v3_v3_to_v3
%p1_param = OpFunctionParameter %v3float
%t_param = OpFunctionParameter %float
%p2_param = OpFunctionParameter %v3float
%p3_param = OpFunctionParameter %v3float
%p0_param = OpFunctionParameter %v3float
%cr_block = OpLabel

%p2_p1 = OpFSub %v3float %p2_param %p1_param
%scaled = OpVectorTimesScalar %v3float %p2_p1 %t_param
%result = OpFAdd %v3float %p1_param %scaled
OpReturnValue %result
OpFunctionEnd

%main = OpFunction %void None %func_type
%entry = OpLabel

%i = OpVariable %ptr_function_int Function
%t = OpVariable %ptr_function_float Function
%j = OpVariable %ptr_function_float Function
%pos = OpVariable %ptr_function_v3 Function
%tangent = OpVariable %ptr_function_v3 Function
%perp = OpVariable %ptr_function_v3 Function
%vertex_pos = OpVariable %ptr_function_v3 Function
%color_t = OpVariable %ptr_function_v3 Function
%p0_val = OpVariable %ptr_function_v3 Function
%p1_val = OpVariable %ptr_function_v3 Function
%p2_val = OpVariable %ptr_function_v3 Function
%p3_val = OpVariable %ptr_function_v3 Function

; Load control points
%gl_in_0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%gl_in_0 = OpLoad %v4float %gl_in_0_ptr
%p0_xyz = OpVectorShuffle %v3float %gl_in_0 %gl_in_0 0 1 2
OpStore %p0_val %p0_xyz

%gl_in_1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%gl_in_1 = OpLoad %v4float %gl_in_1_ptr
%p1_xyz = OpVectorShuffle %v3float %gl_in_1 %gl_in_1 0 1 2
OpStore %p1_val %p1_xyz

%gl_in_2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%gl_in_2 = OpLoad %v4float %gl_in_2_ptr
%p2_xyz = OpVectorShuffle %v3float %gl_in_2 %gl_in_2 0 1 2
OpStore %p2_val %p2_xyz

%gl_in_3_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%gl_in_3 = OpLoad %v4float %gl_in_3_ptr
%p3_xyz = OpVectorShuffle %v3float %gl_in_3 %gl_in_3 0 1 2
OpStore %p3_val %p3_xyz

%color_1_ptr = OpAccessChain %ptr_in_v3float %inColor %int_1
%color_1 = OpLoad %v3float %color_1_ptr
%color_2_ptr = OpAccessChain %ptr_in_v3float %inColor %int_2
%color_2 = OpLoad %v3float %color_2_ptr

OpStore %i %int_0
%loop_header = OpLabel
%i_val = OpLoad %int %i
%cond = OpSLessThanEqual %bool %i_val %int_4
OpBranchConditional %cond %loop_body %loop_end

%loop_body = OpLabel
%i_val2 = OpLoad %int %i
%t_val = OpConvertSToF %float %i_val2
%t_normalized = OpFMul %float %t_val %float_0_25
OpStore %t %t_normalized

; Get interpolated position via lerp(P1, P2, t)
%p0_l = OpLoad %v3float %p0_val
%p1_l = OpLoad %v3float %p1_val
%p2_l = OpLoad %v3float %p2_val
%p3_l = OpLoad %v3float %p3_val
%interp_pos = OpFunctionCall %v3float %catmull_rom %p1_l %t_normalized %p2_l %p3_l %p0_l
OpStore %pos %interp_pos

; Compute tangent = P2 - P0
%p2_p0 = OpFSub %v3float %p2_l %p0_l
OpStore %tangent %p2_p0

; Compute perpendicular: [-t_y, t_x, 0]
%tangent_l = OpLoad %v3float %tangent
%t_x = OpCompositeExtract %float %tangent_l 0
%t_y = OpCompositeExtract %float %tangent_l 1
%neg_t_y = OpFNegate %float %t_y
%perp_unnorm = OpCompositeConstruct %v3float %neg_t_y %t_x %float_0
%perp_len_sq = OpDot %float %perp_unnorm %perp_unnorm
%perp_len = OpSqrt %float %perp_len_sq
%perp_len_safe = OpFMax %float %perp_len %float_epsilon
%inv_len = OpFDiv %float %float_1 %perp_len_safe
%perp_unit = OpVectorTimesScalar %v3float %perp_unnorm %inv_len
%perp_scaled = OpVectorTimesScalar %v3float %perp_unit %float_0_02
OpStore %perp %perp_scaled

; Emit quads: for j = -1, 1 (two sides)
OpStore %j %float_neg1
%emit_loop = OpLabel
%j_val = OpLoad %float %j
%emit_cond = OpFOrdLessThanEqual %bool %j_val %float_1
OpBranchConditional %emit_cond %emit_body %emit_end

%emit_body = OpLabel
%pos_l = OpLoad %v3float %pos
%perp_l = OpLoad %v3float %perp
%j_val2 = OpLoad %float %j
%perp_offset = OpVectorTimesScalar %v3float %perp_l %j_val2
%vertex = OpFAdd %v3float %pos_l %perp_offset
OpStore %vertex_pos %vertex

; Interpolate color
%color_1_l = OpLoad %v3float %color_1
%color_2_l = OpLoad %v3float %color_2
%color_diff = OpFSub %v3float %color_2_l %color_1_l
%t_l = OpLoad %float %t
%color_delta = OpVectorTimesScalar %v3float %color_diff %t_l
%color_interp = OpFAdd %v3float %color_1_l %color_delta
OpStore %color_t %color_interp

; Output vertex
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
%vertex_v = OpLoad %v3float %vertex_pos
%vertex_v4 = OpCompositeConstruct %v4float %vertex_v %float_1
OpStore %out_pos_ptr %vertex_v4

%color_v = OpLoad %v3float %color_t
OpStore %outColor %color_v

OpEmitVertex

%j_next = OpFAdd %float %j_val2 %float_1
OpStore %j %j_next
OpBranch %emit_loop

%emit_end = OpLabel

; Increment i
%i_next = OpIAdd %int %i_val2 %int_1
OpStore %i %i_next
OpBranch %loop_header

%loop_end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
