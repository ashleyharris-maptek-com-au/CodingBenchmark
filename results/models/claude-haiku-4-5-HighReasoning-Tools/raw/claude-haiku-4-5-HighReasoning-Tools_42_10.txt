OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0p25 = OpConstant %float 0.25
%float_0p5 = OpConstant %float 0.5
%float_0p75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0
%float_0p02 = OpConstant %float 0.02
%float_2 = OpConstant %float 2.0

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr4_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr4_v3 = OpTypeArray %v3float %uint_4

%ptr_in_arr4_pv = OpTypePointer Input %arr4_pv_in
%ptr_in_arr4_v3 = OpTypePointer Input %arr4_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr4_pv Input
%worldPos = OpVariable %ptr_in_arr4_v3 Input
%normal = OpVariable %ptr_in_arr4_v3 Input
%color = OpVariable %ptr_in_arr4_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%ptr_Function_v3 = OpTypePointer Function %v3float
%ptr_Function_v4 = OpTypePointer Function %v4float
%ptr_Function_float = OpTypePointer Function %float

%func_ty = OpTypeFunction %void
%main = OpFunction %void None %func_ty
%entry = OpLabel

%p0 = OpVariable %ptr_Function_v3 Function
%p1 = OpVariable %ptr_Function_v3 Function
%p2 = OpVariable %ptr_Function_v3 Function
%p3 = OpVariable %ptr_Function_v3 Function
%color_start = OpVariable %ptr_Function_v3 Function
%color_end = OpVariable %ptr_Function_v3 Function

; Load control points from worldPos array
%ac_p0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0_val = OpLoad %v3float %ac_p0
OpStore %p0 %p0_val

%ac_p1 = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1_val = OpLoad %v3float %ac_p1
OpStore %p1 %p1_val

%ac_p2 = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p2_val = OpLoad %v3float %ac_p2
OpStore %p2 %p2_val

%ac_p3 = OpAccessChain %ptr_in_v3 %worldPos %int_3
%p3_val = OpLoad %v3float %ac_p3
OpStore %p3 %p3_val

; Load colors
%ac_col1 = OpAccessChain %ptr_in_v3 %color %int_1
%col1 = OpLoad %v3float %ac_col1
OpStore %color_start %col1

%ac_col2 = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %ac_col2
OpStore %color_end %col2

; Helper macro to emit vertex at position with perpendicular offset
; We need to compute tangent and perpendicular for each interpolation point

; Get tangent direction (P3 - P0)
%p0_l = OpLoad %v3float %p0
%p3_l = OpLoad %v3float %p3
%tangent = OpFSub %v3float %p3_l %p0_l

; Point 0: t=0 (at P1)
%p1_l = OpLoad %v3float %p1
%perp_raw = OpCompositeConstruct %v3float %tangent %int_1 %int_2
%perp_y = OpCompositeExtract %float %tangent 1
%perp_z = OpCompositeExtract %float %tangent 2
%perp_x_off = OpCompositeExtract %float %tangent 0
%perp_x_neg = OpFNegate %float %perp_x_off
%perp_dir = OpCompositeConstruct %v3float %perp_y %perp_x_neg %float_0
%perp_scaled = OpVectorTimesScalar %v3float %perp_dir %float_0p02

%v0_add = OpFAdd %v3float %p1_l %perp_scaled
%v0_add_v4 = OpCompositeConstruct %v4float %v0_add %float_1
%ac_out_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_out_0 %v0_add_v4
%col_0 = OpLoad %v3float %color_start
OpStore %outColor %col_0
OpEmitVertex

%v0_sub = OpFSub %v3float %p1_l %perp_scaled
%v0_sub_v4 = OpCompositeConstruct %v4float %v0_sub %float_1
OpStore %ac_out_0 %v0_sub_v4
OpStore %outColor %col_0
OpEmitVertex

; Point 1: t=0.25
%p1_l2 = OpLoad %v3float %p1
%p2_l = OpLoad %v3float %p2
%seg = OpFSub %v3float %p2_l %p1_l2
%pt = OpVectorTimesScalar %v3float %seg %float_0p25
%pos1 = OpFAdd %v3float %p1_l2 %pt
OpStore %p1 %pos1

%v1_add = OpFAdd %v3float %pos1 %perp_scaled
%v1_add_v4 = OpCompositeConstruct %v4float %v1_add %float_1
OpStore %ac_out_0 %v1_add_v4
%col_1 = OpFMul %v3float %color_start %float_0p75
%col_1_b = OpFMul %v3float %color_end %float_0p25
%col_1_final = OpFAdd %v3float %col_1 %col_1_b
OpStore %outColor %col_1_final
OpEmitVertex

%v1_sub = OpFSub %v3float %pos1 %perp_scaled
%v1_sub_v4 = OpCompositeConstruct %v4float %v1_sub %float_1
OpStore %ac_out_0 %v1_sub_v4
OpStore %outColor %col_1_final
OpEmitVertex

; Point 2: t=0.5
%p1_l3 = OpLoad %v3float %p1
%p2_l2 = OpLoad %v3float %p2
%seg2 = OpFSub %v3float %p2_l2 %p1_l3
%pt2 = OpVectorTimesScalar %v3float %seg2 %float_0p5
%pos2 = OpFAdd %v3float %p1_l3 %pt2
OpStore %p2 %pos2

%v2_add = OpFAdd %v3float %pos2 %perp_scaled
%v2_add_v4 = OpCompositeConstruct %v4float %v2_add %float_1
OpStore %ac_out_0 %v2_add_v4
%col_2 = OpFMul %v3float %color_start %float_0p5
%col_2_b = OpFMul %v3float %color_end %float_0p5
%col_2_final = OpFAdd %v3float %col_2 %col_2_b
OpStore %outColor %col_2_final
OpEmitVertex

%v2_sub = OpFSub %v3float %pos2 %perp_scaled
%v2_sub_v4 = OpCompositeConstruct %v4float %v2_sub %float_1
OpStore %ac_out_0 %v2_sub_v4
OpStore %outColor %col_2_final
OpEmitVertex

; Point 3: t=0.75
%p1_l4 = OpLoad %v3float %p1
%p2_l3 = OpLoad %v3float %p2
%seg3 = OpFSub %v3float %p2_l3 %p1_l4
%pt3 = OpVectorTimesScalar %v3float %seg3 %float_0p75
%pos3 = OpFAdd %v3float %p1_l4 %pt3
OpStore %p3 %pos3

%v3_add = OpFAdd %v3float %pos3 %perp_scaled
%v3_add_v4 = OpCompositeConstruct %v4float %v3_add %float_1
OpStore %ac_out_0 %v3_add_v4
%col_3 = OpFMul %v3float %color_start %float_0p25
%col_3_b = OpFMul %v3float %color_end %float_0p75
%col_3_final = OpFAdd %v3float %col_3 %col_3_b
OpStore %outColor %col_3_final
OpEmitVertex

%v3_sub = OpFSub %v3float %pos3 %perp_scaled
%v3_sub_v4 = OpCompositeConstruct %v4float %v3_sub %float_1
OpStore %ac_out_0 %v3_sub_v4
OpStore %outColor %col_3_final
OpEmitVertex

; Point 4: t=1.0 (at P2)
%p1_l5 = OpLoad %v3float %p1
%p2_l4 = OpLoad %v3float %p2
%pos4 = %p2_l4

%v4_add = OpFAdd %v3float %pos4 %perp_scaled
%v4_add_v4 = OpCompositeConstruct %v4float %v4_add %float_1
OpStore %ac_out_0 %v4_add_v4
%col_4 = OpLoad %v3float %color_end
OpStore %outColor %col_4
OpEmitVertex

%v4_sub = OpFSub %v3float %pos4 %perp_scaled
%v4_sub_v4 = OpCompositeConstruct %v4float %v4_sub %float_1
OpStore %ac_out_0 %v4_sub_v4
OpStore %outColor %col_4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
