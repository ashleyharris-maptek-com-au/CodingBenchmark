OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpExtInstImport %glsl "GLSL.std.450"
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
       %bool = OpTypeBool
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %v2float = OpTypeVector %float 2
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4

      %uint_1 = OpConstant %uint 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_6 = OpConstant %int 6
      %float_0 = OpConstant %float 0.0
      %float_1 = OpConstant %float 1.0
      %float_0_5 = OpConstant %float 0.5
      %float_0_866 = OpConstant %float 0.866025404
      %float_0_04 = OpConstant %float 0.04
      %float_neg_0_04 = OpConstant %float -0.04
      %float_0_02 = OpConstant %float 0.02
      %float_neg_0_02 = OpConstant %float -0.02
      %float_0_0346 = OpConstant %float 0.0346410162
      %float_neg_0_0346 = OpConstant %float -0.0346410162
      %float_0_8 = OpConstant %float 0.8
      %float_pi_3 = OpConstant %float 1.047197551

      %gl_PerVertex_in = OpTypeStruct %v4float
      %arr_v4_in = OpTypeArray %gl_PerVertex_in %uint_1
      %ptr_in_arr_v4 = OpTypePointer Input %arr_v4_in

      %arr_v3_in = OpTypeArray %v3float %uint_1
      %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
      %ptr_in_v4 = OpTypePointer Input %v4float
      %ptr_in_v3 = OpTypePointer Input %v3float

      %gl_PerVertex_out = OpTypeStruct %v4float
      %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
      %ptr_out_v4 = OpTypePointer Output %v4float
      %ptr_out_v3 = OpTypePointer Output %v3float
      %ptr_fun_v3 = OpTypePointer Function %v3float

      %gl_in = OpVariable %ptr_in_arr_v4 Input
      %worldPos = OpVariable %ptr_in_arr_v3 Input
      %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

      %gl_out = OpVariable %ptr_out_pv Output
      %outColor = OpVariable %ptr_out_v3 Output

      %func_type = OpTypeFunction %void
      %func_type_v3 = OpTypeFunction %v3float %float

      %main = OpFunction %void None %func_type
      %entry = OpLabel

      ; Load center position
      %center_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %center = OpLoad %v4float %center_ptr

      ; Load center color
      %color_ptr = OpAccessChain %ptr_in_v3 %color %int_0
      %base_color = OpLoad %v3float %color_ptr

      ; Darker color for edges
      %edge_color = OpVectorTimesScalar %v3float %base_color %float_0_8

      ; Extract center xyz
      %center_xyz = OpVectorShuffle %v3float %center %center 0 1 2
      %center_x = OpCompositeExtract %float %center 0
      %center_y = OpCompositeExtract %float %center 1
      %center_z = OpCompositeExtract %float %center 2

      ; Triangle 1: center, edge0 (0°), edge1 (60°)
      ; Center
      OpStore %outColor %base_color
      %v1 = OpCompositeConstruct %v4float %center_x %center_y %center_z %float_1
      %pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %pos1_ptr %v1
      OpEmitVertex

      ; Edge0 at 0°: (r, 0)
      OpStore %outColor %edge_color
      %e0x = OpFAdd %float %center_x %float_0_04
      %e0y = %center_y
      %v2 = OpCompositeConstruct %v4float %e0x %e0y %center_z %float_1
      OpStore %pos1_ptr %v2
      OpEmitVertex

      ; Edge1 at 60°: (r*0.5, r*0.866)
      %e1x = OpFAdd %float %center_x %float_0_02
      %e1y = OpFAdd %float %center_y %float_0_0346
      %v3 = OpCompositeConstruct %v4float %e1x %e1y %center_z %float_1
      OpStore %pos1_ptr %v3
      OpEmitVertex

      OpEndPrimitive

      ; Triangle 2: center, edge1 (60°), edge2 (120°)
      OpStore %outColor %base_color
      OpStore %pos1_ptr %v1
      OpEmitVertex

      OpStore %outColor %edge_color
      OpStore %pos1_ptr %v3
      OpEmitVertex

      ; Edge2 at 120°: (-r*0.5, r*0.866)
      %e2x = OpFAdd %float %center_x %float_neg_0_02
      %e2y = OpFAdd %float %center_y %float_0_0346
      %v4 = OpCompositeConstruct %v4float %e2x %e2y %center_z %float_1
      OpStore %pos1_ptr %v4
      OpEmitVertex

      OpEndPrimitive

      ; Triangle 3: center, edge2 (120°), edge3 (180°)
      OpStore %outColor %base_color
      OpStore %pos1_ptr %v1
      OpEmitVertex

      OpStore %outColor %edge_color
      OpStore %pos1_ptr %v4
      OpEmitVertex

      ; Edge3 at 180°: (-r, 0)
      %e3x = OpFAdd %float %center_x %float_neg_0_04
      %e3y = %center_y
      %v5 = OpCompositeConstruct %v4float %e3x %e3y %center_z %float_1
      OpStore %pos1_ptr %v5
      OpEmitVertex

      OpEndPrimitive

      ; Triangle 4: center, edge3 (180°), edge4 (240°)
      OpStore %outColor %base_color
      OpStore %pos1_ptr %v1
      OpEmitVertex

      OpStore %outColor %edge_color
      OpStore %pos1_ptr %v5
      OpEmitVertex

      ; Edge4 at 240°: (-r*0.5, -r*0.866)
      %e4x = OpFAdd %float %center_x %float_neg_0_02
      %e4y = OpFAdd %float %center_y %float_neg_0_0346
      %v6 = OpCompositeConstruct %v4float %e4x %e4y %center_z %float_1
      OpStore %pos1_ptr %v6
      OpEmitVertex

      OpEndPrimitive

      ; Triangle 5: center, edge4 (240°), edge5 (300°)
      OpStore %outColor %base_color
      OpStore %pos1_ptr %v1
      OpEmitVertex

      OpStore %outColor %edge_color
      OpStore %pos1_ptr %v6
      OpEmitVertex

      ; Edge5 at 300°: (r*0.5, -r*0.866)
      %e5x = OpFAdd %float %center_x %float_0_02
      %e5y = OpFAdd %float %center_y %float_neg_0_0346
      %v7 = OpCompositeConstruct %v4float %e5x %e5y %center_z %float_1
      OpStore %pos1_ptr %v7
      OpEmitVertex

      OpEndPrimitive

      ; Triangle 6: center, edge5 (300°), edge0 (0°) - closing
      OpStore %outColor %base_color
      OpStore %pos1_ptr %v1
      OpEmitVertex

      OpStore %outColor %edge_color
      OpStore %pos1_ptr %v7
      OpEmitVertex

      OpStore %pos1_ptr %v2
      OpEmitVertex

      OpEndPrimitive

      OpReturn
      OpFunctionEnd
