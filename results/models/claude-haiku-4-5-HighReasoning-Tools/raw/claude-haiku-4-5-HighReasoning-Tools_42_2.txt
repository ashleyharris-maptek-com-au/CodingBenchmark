OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport %glsl "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo_Block BufferBlock
OpMemberDecorate %ubo_Block 0 Offset 0
OpMemberDecorate %ubo_Block 1 Offset 64
OpMemberDecorate %ubo_Block 2 Offset 128
OpMemberDecorate %ubo_Block 3 Offset 192
OpMemberDecorate %ubo_Block 4 Offset 208
OpMemberDecorate %ubo_Block 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0.0
%float_0p5 = OpConstant %float 0.5
%float_0p866 = OpConstant %float 0.866025403
%float_1 = OpConstant %float 1.0
%float_neg0p5 = OpConstant %float -0.5
%float_neg0p866 = OpConstant %float -0.866025403
%float_neg1 = OpConstant %float -1.0
%float_04 = OpConstant %float 0.04

%v3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v4_0001 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ubo_Block = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
%ptr_ubo = OpTypePointer Uniform %ubo_Block
%ubo = OpVariable %ptr_ubo Uniform

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input point position and color
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%in_color_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%in_color = OpLoad %v3float %in_color_ptr

; Darken color for edge vertices
%edge_color = OpVectorTimesScalar %v3float %in_color %float_0p5

; Compute hexagon vertices
; Hexagon vertices at radius 0.04, angles 0°, 60°, 120°, 180°, 240°, 300°
; Extract xyz from input position
%center_xyz = OpVectorShuffle %v3float %in_pos %in_pos 0 1 2

; e0: 0° = (1, 0)
%e0_dir = OpCompositeConstruct %v3float %float_1 %float_0 %float_0
%e0_offset = OpVectorTimesScalar %v3float %e0_dir %float_04
%e0 = OpFAdd %v3float %center_xyz %e0_offset
%e0_v4 = OpCompositeConstruct %v4float %e0 3 %float_1

; e1: 60° = (0.5, 0.866)
%e1_dir = OpCompositeConstruct %v3float %float_0p5 %float_0p866 %float_0
%e1_offset = OpVectorTimesScalar %v3float %e1_dir %float_04
%e1 = OpFAdd %v3float %center_xyz %e1_offset
%e1_v4 = OpCompositeConstruct %v4float %e1 3 %float_1

; e2: 120° = (-0.5, 0.866)
%e2_dir = OpCompositeConstruct %v3float %float_neg0p5 %float_0p866 %float_0
%e2_offset = OpVectorTimesScalar %v3float %e2_dir %float_04
%e2 = OpFAdd %v3float %center_xyz %e2_offset
%e2_v4 = OpCompositeConstruct %v4float %e2 3 %float_1

; e3: 180° = (-1, 0)
%e3_dir = OpCompositeConstruct %v3float %float_neg1 %float_0 %float_0
%e3_offset = OpVectorTimesScalar %v3float %e3_dir %float_04
%e3 = OpFAdd %v3float %center_xyz %e3_offset
%e3_v4 = OpCompositeConstruct %v4float %e3 3 %float_1

; e4: 240° = (-0.5, -0.866)
%e4_dir = OpCompositeConstruct %v3float %float_neg0p5 %float_neg0p866 %float_0
%e4_offset = OpVectorTimesScalar %v3float %e4_dir %float_04
%e4 = OpFAdd %v3float %center_xyz %e4_offset
%e4_v4 = OpCompositeConstruct %v4float %e4 3 %float_1

; e5: 300° = (0.5, -0.866)
%e5_dir = OpCompositeConstruct %v3float %float_0p5 %float_neg0p866 %float_0
%e5_offset = OpVectorTimesScalar %v3float %e5_dir %float_04
%e5 = OpFAdd %v3float %center_xyz %e5_offset
%e5_v4 = OpCompositeConstruct %v4float %e5 3 %float_1

; Output pointer
%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit triangle 1: center, e0, e1
OpStore %out_pos %in_pos
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos %e0_v4
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos %e1_v4
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Emit triangle 2: center, e1, e2
OpStore %out_pos %in_pos
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos %e1_v4
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos %e2_v4
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Emit triangle 3: center, e2, e3
OpStore %out_pos %in_pos
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos %e2_v4
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos %e3_v4
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Emit triangle 4: center, e3, e4
OpStore %out_pos %in_pos
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos %e3_v4
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos %e4_v4
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Emit triangle 5: center, e4, e5
OpStore %out_pos %in_pos
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos %e4_v4
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos %e5_v4
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Emit triangle 6: center, e5, e0
OpStore %out_pos %in_pos
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos %e5_v4
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos %e0_v4
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

OpFunctionEnd
