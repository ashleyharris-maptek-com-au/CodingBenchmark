; Type definitions
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Pointer types
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Uniform_ubo_block = OpTypePointer Uniform %ubo_block

; UBO struct
%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Function types
%void_func = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_0_2 = OpConstant %float 0.2
%float_0_75 = OpConstant %float 0.75
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4

%vec3_cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%vec3_warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4

%int_3 = OpConstant %int 3

; Variables
%ubo_var = OpVariable %ptr_Uniform_ubo_block Uniform

%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input

%out_outColor = OpVariable %ptr_Output_vec4 Output

; Main function
%main = OpFunction %void None %void_func
%entry_block = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal

; Normalize normal
%N = OpNormalize %vec3 %normal

; Load light position from UBO offset 192 (member 3)
%light_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %int_3
%lightPos_vec4 = OpLoad %vec4 %light_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

; Calculate light direction
%lightDir_unnorm = OpFSub %vec3 %lightPos %worldPos
%L = OpNormalize %vec3 %lightDir_unnorm

; Calculate dot(N,L)
%dot_NL = OpDot %float %N %L

; Calculate t = dot(N,L) * 0.5 + 0.5
%t_scaled = OpFMul %float %dot_NL %float_0_5
%t = OpFAdd %float %t_scaled %float_0_5

; Interpolate: cool * (1-t) + warm * t
%one_minus_t = OpFSub %float %float_1 %t
%cool_contribution = OpVectorTimesScalar %vec3 %vec3_cool %one_minus_t
%warm_contribution = OpVectorTimesScalar %vec3 %vec3_warm %t
%shaded_color = OpFAdd %vec3 %cool_contribution %warm_contribution

; Create output vec4 with alpha = 1.0
%output_color = OpCompositeConstruct %vec4 %shaded_color %float_1

; Store output
OpStore %out_outColor %output_color

OpReturn
OpFunctionEnd
