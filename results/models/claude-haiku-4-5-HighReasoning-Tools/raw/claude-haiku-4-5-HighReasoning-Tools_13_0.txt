#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (strings.empty()) {
        cout << endl;
        return 0;
    }

    // Find the smallest string - the LCS cannot be longer than this
    size_t minIdx = 0;
    for (size_t i = 1; i < strings.size(); i++) {
        if (strings[i].length() < strings[minIdx].length()) {
            minIdx = i;
        }
    }

    const string& smallest = strings[minIdx];

    // Try all lengths from longest to shortest
    for (size_t len = smallest.length(); len > 0; len--) {
        // Build set of all substrings of length len from the smallest string
        unordered_set<string> candidates;
        for (size_t i = 0; i + len <= smallest.length(); i++) {
            candidates.insert(smallest.substr(i, len));
        }

        // For each other string, intersect candidates
        for (size_t idx = 0; idx < strings.size(); idx++) {
            if (idx == minIdx) continue;

            const string& str = strings[idx];
            unordered_set<string> common;

            // Find which candidates appear in this string
            for (size_t i = 0; i + len <= str.length(); i++) {
                string sub = str.substr(i, len);
                if (candidates.count(sub)) {
                    common.insert(sub);
                }
            }

            candidates = common;

            // Early exit if no common substrings at this length
            if (candidates.empty()) break;
        }

        // If any candidates remain, they are common to all strings
        if (!candidates.empty()) {
            cout << *candidates.begin() << endl;
            return 0;
        }
    }

    // No common substring found
    cout << endl;
    return 0;
}
