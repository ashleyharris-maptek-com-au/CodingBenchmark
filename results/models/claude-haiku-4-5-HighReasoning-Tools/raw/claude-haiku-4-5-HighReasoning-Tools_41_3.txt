; SPIR-V
; Version: 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_color %out_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%fn_void = OpTypeFunction %void
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%one = OpConstant %float 1.0

%in_color = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

%color = OpLoad %vec3 %in_color

%x = OpCompositeExtract %float %color 0
%y = OpCompositeExtract %float %color 1
%z = OpCompositeExtract %float %color 2

%result = OpCompositeConstruct %vec4 %x %y %z %one

OpStore %out_color %result

OpReturn
OpFunctionEnd
