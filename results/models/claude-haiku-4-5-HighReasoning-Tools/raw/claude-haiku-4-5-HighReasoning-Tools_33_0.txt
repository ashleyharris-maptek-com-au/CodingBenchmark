using System;
using System.Diagnostics;
using System.Linq;

class QAP
{
    static void Main()
    {
        int n = int.Parse(Console.ReadLine());

        int[][] flow = new int[n][];
        int[][] distance = new int[n][];

        for (int i = 0; i < n; i++)
        {
            flow[i] = Console.ReadLine().Split(' ').Select(int.Parse).ToArray();
        }

        for (int i = 0; i < n; i++)
        {
            distance[i] = Console.ReadLine().Split(' ').Select(int.Parse).ToArray();
        }

        var (perm, cost) = SolveQAP(flow, distance, n);

        Console.WriteLine(cost);
        Console.WriteLine(string.Join(" ", perm));
    }

    static (int[], long) SolveQAP(int[][] flow, int[][] distance, int n)
    {
        int[] bestPerm = new int[n];
        long bestCost = long.MaxValue;

        var sw = Stopwatch.StartNew();
        var rand = new Random(42);

        while (sw.ElapsedMilliseconds < 270000)
        {
            int[] perm = Enumerable.Range(0, n).ToArray();

            if (bestCost != long.MaxValue)
            {
                for (int k = n - 1; k > 0; k--)
                {
                    int idx = rand.Next(k + 1);
                    (perm[k], perm[idx]) = (perm[idx], perm[k]);
                }
            }

            long cost = CalculateCost(perm, flow, distance);
            TwoOptImprove(perm, flow, distance, ref cost);

            if (cost < bestCost)
            {
                bestCost = cost;
                bestPerm = (int[])perm.Clone();
            }
        }

        return (bestPerm, bestCost);
    }

    static void TwoOptImprove(int[] perm, int[][] flow, int[][] distance, ref long cost)
    {
        int n = perm.Length;
        bool improved = true;

        while (improved)
        {
            improved = false;

            for (int i = 0; i < n && !improved; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long delta = CalculateDeltaCost(perm, flow, distance, i, j);

                    if (delta < 0)
                    {
                        (perm[i], perm[j]) = (perm[j], perm[i]);
                        cost += delta;
                        improved = true;
                        break;
                    }
                }
            }
        }
    }

    static long CalculateDeltaCost(int[] perm, int[][] flow, int[][] distance, int i, int j)
    {
        int n = perm.Length;
        int x = perm[i];
        int y = perm[j];
        long delta = 0;

        delta += (long)(flow[i][i] - flow[j][j]) * (distance[y][y] - distance[x][x]);
        delta += (long)(flow[i][j] - flow[j][i]) * (distance[y][x] - distance[x][y]);

        for (int b = 0; b < n; b++)
        {
            if (b != i && b != j)
            {
                delta += (long)(flow[i][b] - flow[j][b]) * (distance[y][perm[b]] - distance[x][perm[b]]);
            }
        }

        for (int a = 0; a < n; a++)
        {
            if (a != i && a != j)
            {
                delta += (long)(flow[a][i] - flow[a][j]) * (distance[perm[a]][y] - distance[perm[a]][x]);
            }
        }

        return delta;
    }

    static long CalculateCost(int[] perm, int[][] flow, int[][] distance)
    {
        long cost = 0;
        for (int i = 0; i < perm.Length; i++)
        {
            for (int j = 0; j < perm.Length; j++)
            {
                cost += (long)flow[i][j] * distance[perm[i]][perm[j]];
            }
        }
        return cost;
    }
}
