OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations for built-in blocks
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Decorations for user-defined I/O
               OpDecorate %inColor Location 0
               OpDecorate %outColor Location 0

               ; Types
               %void = OpTypeVoid
               %int = OpTypeInt 32 1
               %uint = OpTypeInt 32 0
               %float = OpTypeFloat 32
               %v4float = OpTypeVector %float 4
               %v3float = OpTypeVector %float 3

               ; Constants
               %int_0 = OpConstant %int 0
               %float_0 = OpConstant %float 0.0
               %float_1 = OpConstant %float 1.0
               %float_h = OpConstant %float 0.04
               %uint_1 = OpConstant %uint 1

               ; Types - built-in blocks
               %gl_PerVertex_in = OpTypeStruct %v4float
               %gl_PerVertex_out = OpTypeStruct %v4float

               ; Types - arrays
               %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
               %arr_color = OpTypeArray %v3float %uint_1

               ; Pointer types - input
               %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
               %ptr_in_v4 = OpTypePointer Input %v4float
               %ptr_in_arr_color = OpTypePointer Input %arr_color
               %ptr_in_v3 = OpTypePointer Input %v3float

               ; Pointer types - output
               %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
               %ptr_out_v4 = OpTypePointer Output %v4float
               %ptr_out_v3 = OpTypePointer Output %v3float

               ; Function type
               %func_type = OpTypeFunction %void

               ; Variables
               %gl_in = OpVariable %ptr_in_arr_pv Input
               %inColor = OpVariable %ptr_in_arr_color Input
               %gl_out = OpVariable %ptr_out_pv Output
               %outColor = OpVariable %ptr_out_v3 Output

               ; Main function
               %main = OpFunction %void None %func_type
               %entry = OpLabel

               ; Load input position: gl_in[0].gl_Position
               %pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
               %pos = OpLoad %v4float %pos_ptr

               ; Load input color: color[0]
               %col_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
               %col = OpLoad %v3float %col_ptr

               ; Extract x, y from position
               %x = OpCompositeExtract %float %pos 0
               %y = OpCompositeExtract %float %pos 1

               ; Get output position pointer
               %out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; Emit vertex 0: (x-h, y-h)
               %x0 = OpFSub %float %x %float_h
               %y0 = OpFSub %float %y %float_h
               %v0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
               OpStore %out_pos %v0
               OpStore %outColor %col
               OpEmitVertex

               ; Emit vertex 1: (x+h, y-h)
               %x1 = OpFAdd %float %x %float_h
               %y1 = OpFSub %float %y %float_h
               %v1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
               OpStore %out_pos %v1
               OpStore %outColor %col
               OpEmitVertex

               ; Emit vertex 2: (x-h, y+h)
               %x2 = OpFSub %float %x %float_h
               %y2 = OpFAdd %float %y %float_h
               %v2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
               OpStore %out_pos %v2
               OpStore %outColor %col
               OpEmitVertex

               ; Emit vertex 3: (x+h, y+h)
               %x3 = OpFAdd %float %x %float_h
               %y3 = OpFAdd %float %y %float_h
               %v3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
               OpStore %out_pos %v3
               OpStore %outColor %col
               OpEmitVertex

               ; End primitive
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
