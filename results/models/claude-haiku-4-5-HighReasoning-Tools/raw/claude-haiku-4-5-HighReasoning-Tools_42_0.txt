OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0p04 = OpConstant %float 0.04
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_neg0p04 = OpConstant %float -0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load position from gl_in[0].gl_Position
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr

; Load color from color[0]
%col_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%col = OpLoad %v3float %col_ptr

; Extract x and y coordinates
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

; Calculate vertex positions
; v0: (x-h, y-h)
%v0_x = OpFAdd %float %x %float_neg0p04
%v0_y = OpFAdd %float %y %float_neg0p04
%v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %float_0 %float_1

; v1: (x+h, y-h)
%v1_x = OpFAdd %float %x %float_0p04
%v1_y = OpFAdd %float %y %float_neg0p04
%v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %float_0 %float_1

; v2: (x-h, y+h)
%v2_x = OpFAdd %float %x %float_neg0p04
%v2_y = OpFAdd %float %y %float_0p04
%v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %float_0 %float_1

; v3: (x+h, y+h)
%v3_x = OpFAdd %float %x %float_0p04
%v3_y = OpFAdd %float %y %float_0p04
%v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %float_0 %float_1

; Emit vertex 0
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos %v0_pos
OpStore %outColor %col
OpEmitVertex

; Emit vertex 1
OpStore %gl_out_pos %v1_pos
OpStore %outColor %col
OpEmitVertex

; Emit vertex 2
OpStore %gl_out_pos %v2_pos
OpStore %outColor %col
OpEmitVertex

; Emit vertex 3
OpStore %gl_out_pos %v3_pos
OpStore %outColor %col
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
