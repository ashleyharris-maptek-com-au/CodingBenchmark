%main = OpFunction %void None %func_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %in_worldPos

%color = OpLoad %v3float %in_color

; Load lightPos from UBO (member 3)
%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %UBOVar %int_3
%lightPos_v4 = OpLoad %v4float %lightPos_ptr
%lightPos = OpVectorShuffle %v3float %lightPos_v4 %lightPos_v4 0 1 2

; Calculate light direction: L = lightPos - worldPos
%L = OpFSub %v3float %lightPos %worldPos

; Normalize normal
%normal_norm = OpExtInst %v3float %1 Normalize %normal

; Normalize light direction
%L_norm = OpExtInst %v3float %1 Normalize %L

; Calculate diffuse: d = max(dot(N,L), 0)
%dot_NL = OpDot %float %normal_norm %L_norm
%diffuse = OpExtInst %float %1 FMax %dot_NL %float_0

; Quantize into bands: thresholds 0.75/0.5/0.25, values 1.0/0.7/0.4/0.2
; if d >= 0.75: band = 1.0
; else if d >= 0.5: band = 0.7
; else if d >= 0.25: band = 0.4
; else: band = 0.2

%cmp_0_75 = OpFOrdGreaterThanEqual %bool %diffuse %float_0_75
%cmp_0_5 = OpFOrdGreaterThanEqual %bool %diffuse %float_0_5
%cmp_0_25 = OpFOrdGreaterThanEqual %bool %diffuse %float_0_25

; Build band value using nested selects
%band_step1 = OpSelect %float %cmp_0_75 %float_1 %float_0_7
%band_step2 = OpSelect %float %cmp_0_5 %band_step1 %float_0_4
%band_value = OpSelect %float %cmp_0_25 %band_step2 %float_0_2

; Multiply vertex color by the band value
%shaded = OpVectorTimesScalar %v3float %color %band_value

; Output with alpha 1.0
%outColor = OpCompositeConstruct %v4float %shaded %float_1

; Store output
OpStore %out_outColor %outColor

OpReturn
