; SPIR-V

OpCapability Shader
OpMemoryModel Logical GLSL450

%glsl = OpExtInstImport "GLSL.std.450"

OpEntryPoint Fragment %main "main" %worldPos_in %normal_in %color_in %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 ColMajor
OpMemberDecorate %ubo_block 0 MatrixStride 16
OpMemberDecorate %ubo_block 1 ColMajor
OpMemberDecorate %ubo_block 1 MatrixStride 16
OpMemberDecorate %ubo_block 2 ColMajor
OpMemberDecorate %ubo_block 2 MatrixStride 16
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224

OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 4
OpDecorate %outColor Location 0

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool

%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4

%mat4 = OpTypeMatrix %vec4 4

%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_Uniform_ubo_block = OpTypePointer Uniform %ubo_block
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4

%func_void = OpTypeFunction %void

; Variable declarations
%ubo_var = OpVariable %ptr_Uniform_ubo_block Uniform
%worldPos_in = OpVariable %ptr_Input_vec3 Input
%normal_in = OpVariable %ptr_Input_vec3 Input
%color_in = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output

; Constants
%c_0 = OpConstant %float 0.0
%c_1 = OpConstant %float 1.0
%c_0_2 = OpConstant %float 0.2
%c_0_25 = OpConstant %float 0.25
%c_0_4 = OpConstant %float 0.4
%c_0_5 = OpConstant %float 0.5
%c_0_7 = OpConstant %float 0.7
%c_0_75 = OpConstant %float 0.75

%c_3_idx = OpConstant %int 3

; Main function
%main = OpFunction %void None %func_void
%entry_label = OpLabel

; Load inputs
%pos = OpLoad %vec3 %worldPos_in
%norm = OpLoad %vec3 %normal_in
%col = OpLoad %vec3 %color_in

; Normalize normal
%norm_n = OpExtInst %vec3 %glsl Normalize %norm

; Load light position from UBO
%light_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %c_3_idx
%light = OpLoad %vec4 %light_ptr

; Get light xyz
%light_xyz = OpVectorShuffle %vec3 %light %light 0 1 2

; Compute light direction (light - pos)
%light_d = OpFSub %vec3 %light_xyz %pos

; Normalize light direction
%light_d_n = OpExtInst %vec3 %glsl Normalize %light_d

; Compute dot product
%dot_prod = OpDot %float %norm_n %light_d_n

; Compute diffuse: max(dot, 0)
%diffuse = OpExtInst %float %glsl FMax %dot_prod %c_0

; Quantize: determine band value
%b075 = OpFOrdGreaterThanEqual %bool %diffuse %c_0_75
%b05 = OpFOrdGreaterThanEqual %bool %diffuse %c_0_5
%b025 = OpFOrdGreaterThanEqual %bool %diffuse %c_0_25

; Nested ternary: (d >= 0.75) ? 1.0 : ((d >= 0.5) ? 0.7 : ((d >= 0.25) ? 0.4 : 0.2))
%inner = OpSelect %float %b025 %c_0_4 %c_0_2
%middle = OpSelect %float %b05 %c_0_7 %inner
%band = OpSelect %float %b075 %c_1 %middle

; Multiply color by band value
%shaded = OpVectorTimesScalar %vec3 %col %band

; Create output vec4 (color, alpha=1.0)
%result = OpCompositeConstruct %vec4 %shaded %c_1

; Store result
OpStore %outColor %result

OpReturn
OpFunctionEnd
