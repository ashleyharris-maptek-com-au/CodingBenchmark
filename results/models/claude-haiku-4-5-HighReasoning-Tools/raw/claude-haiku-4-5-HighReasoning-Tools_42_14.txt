OpCapability Geometry
               OpCapability GLSL450
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpExtInstImport %glsl "GLSL.std.450"
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
               OpDecorate %ubo_type BufferBlock
               OpMemberDecorate %ubo_type 0 Offset 0
               OpMemberDecorate %ubo_type 1 Offset 64
               OpMemberDecorate %ubo_type 2 Offset 128
               OpMemberDecorate %ubo_type 3 Offset 144
               OpMemberDecorate %ubo_type 4 Offset 160
               OpMemberDecorate %ubo_type 5 Offset 176
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0
        %void = OpTypeVoid
        %bool = OpTypeBool
         %int = OpTypeInt 32 1
       %float = OpTypeFloat 32
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
   %mat4float = OpTypeMatrix %v4float 4
        %uint = OpTypeInt 32 0
       %uint_6 = OpConstant %uint 6
       %int_0 = OpConstant %int 0
       %int_1 = OpConstant %int 1
       %int_2 = OpConstant %int 2
       %int_4 = OpConstant %int 4
     %float_0 = OpConstant %float 0.0
   %float_0p2 = OpConstant %float 0.2
   %float_0p8 = OpConstant %float 0.8
     %float_1 = OpConstant %float 1.0
     %float_3 = OpConstant %float 3.0
 %v3float_111 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input
   %arr_v3_in = OpTypeArray %v3float %uint_6
   %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
     %worldPos = OpVariable %ptr_in_arr_v3 Input
      %normal = OpVariable %ptr_in_arr_v3 Input
       %color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
  %ptr_out_v3 = OpTypePointer Output %v3float
    %outColor = OpVariable %ptr_out_v3 Output
  %ptr_in_v4 = OpTypePointer Input %v4float
  %ptr_in_v3 = OpTypePointer Input %v3float
  %ptr_out_v4 = OpTypePointer Output %v4float
    %ubo_type = OpTypeStruct %mat4float %mat4float %mat4float %v3float %v3float %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_type
     %ubo_var = OpVariable %ptr_uniform_ubo Uniform
  %func_type = OpTypeFunction %void
        %main = OpFunction %void None %func_type
       %entry = OpLabel
        %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
           %p0 = OpLoad %v4float %ptr_p0
        %p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
        %ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
           %p2 = OpLoad %v4float %ptr_p2
        %p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
        %ptr_p4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
           %p4 = OpLoad %v4float %ptr_p4
        %p4_v3 = OpVectorShuffle %v3float %p4 %p4 0 1 2
       %edge1 = OpFSub %v3float %p2_v3 %p0_v3
       %edge2 = OpFSub %v3float %p4_v3 %p0_v3
      %cross = OpExtInst %v3float %glsl Cross %edge1 %edge2
  %normal_face = OpExtInst %v3float %glsl Normalize %cross
          %L = OpExtInst %v3float %glsl Normalize %v3float_111
        %dotNL = OpExtInst %float %glsl Dot %normal_face %L
     %diffuse = OpExtInst %float %glsl FMax %dotNL %float_0
      %ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %ptr_c0
      %ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %ptr_c2
      %ptr_c4 = OpAccessChain %ptr_in_v3 %color %int_4
         %c4 = OpLoad %v3float %ptr_c4
      %sum01 = OpFAdd %v3float %c0 %c2
     %sum_all = OpFAdd %v3float %sum01 %c4
   %inv_three = OpFDiv %float %float_1 %float_3
    %avg_color = OpVectorTimesScalar %v3float %sum_all %inv_three
 %lighting_mult = OpFMul %float %diffuse %float_0p8
   %intensity = OpFAdd %float %float_0p2 %lighting_mult
  %flat_color = OpVectorTimesScalar %v3float %avg_color %intensity
      %ptr_v0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %v0_pos = OpLoad %v4float %ptr_v0
  %ptr_out_v0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %ptr_out_v0 %v0_pos
              OpStore %outColor %flat_color
              OpEmitVertex
      %ptr_v2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %v2_pos = OpLoad %v4float %ptr_v2
  %ptr_out_v2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %ptr_out_v2 %v2_pos
              OpStore %outColor %flat_color
              OpEmitVertex
      %ptr_v4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
        %v4_pos = OpLoad %v4float %ptr_v4
  %ptr_out_v4e = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %ptr_out_v4e %v4_pos
              OpStore %outColor %flat_color
              OpEmitVertex
              OpEndPrimitive
              OpReturn
               OpFunctionEnd
