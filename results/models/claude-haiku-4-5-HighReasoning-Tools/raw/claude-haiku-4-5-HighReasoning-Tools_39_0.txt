use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj = vec![Vec::new(); n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge_parts: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = edge_parts[0];
        let v = edge_parts[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    let (partition, cut_size) = find_min_bisection(n, &adj);

    println!("{}", cut_size);
    let partition_a: Vec<usize> = partition.iter()
        .enumerate()
        .filter(|(_, &p)| p == 0)
        .map(|(i, _)| i)
        .collect();

    println!("{}", partition_a.iter().map(|i| i.to_string()).collect::<Vec<_>>().join(" "));
}

fn count_cut(partition: &[usize], adj: &[Vec<usize>]) -> usize {
    let mut cut = 0;
    let n = partition.len();

    for u in 0..n {
        for &v in &adj[u] {
            if u < v && partition[u] != partition[v] {
                cut += 1;
            }
        }
    }

    cut
}

fn swap_gain(u: usize, v: usize, partition: &[usize], adj: &[Vec<usize>]) -> i32 {
    let partition_u = partition[u];
    let partition_v = partition[v];

    let mut edges_u_internal = 0;
    let mut edges_u_external = 0;
    let mut edges_v_internal = 0;
    let mut edges_v_external = 0;

    for &neighbor in &adj[u] {
        if neighbor == v {
            continue;
        }
        if partition[neighbor] == partition_u {
            edges_u_internal += 1;
        } else {
            edges_u_external += 1;
        }
    }

    for &neighbor in &adj[v] {
        if neighbor == u {
            continue;
        }
        if partition[neighbor] == partition_v {
            edges_v_internal += 1;
        } else {
            edges_v_external += 1;
        }
    }

    (edges_u_external + edges_v_external) as i32 - (edges_u_internal + edges_v_internal) as i32
}

fn find_min_bisection(n: usize, adj: &[Vec<usize>]) -> (Vec<usize>, usize) {
    let mut partition = vec![0; n];
    let mut sizes = [0, 0];

    let mut vertices: Vec<usize> = (0..n).collect();
    vertices.sort_by_key(|&v| std::cmp::Reverse(adj[v].len()));

    for v in vertices {
        if sizes[0] < n / 2 {
            partition[v] = 0;
            sizes[0] += 1;
        } else {
            partition[v] = 1;
            sizes[1] += 1;
        }
    }

    loop {
        let mut improved = false;

        for u in 0..n {
            for v in (u + 1)..n {
                if partition[u] != partition[v] {
                    let gain = swap_gain(u, v, &partition, adj);
                    if gain > 0 {
                        partition.swap(u, v);
                        improved = true;
                        break;
                    }
                }
            }
            if improved {
                break;
            }
        }

        if !improved {
            break;
        }
    }

    let cut_size = count_cut(&partition, adj);
    (partition, cut_size)
}
