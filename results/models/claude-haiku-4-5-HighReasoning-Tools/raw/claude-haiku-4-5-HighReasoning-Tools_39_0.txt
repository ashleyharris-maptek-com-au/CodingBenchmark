use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut edges = Vec::new();
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<usize> = line.split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();
        edges.push((parts[0], parts[1]));
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for &(u, v) in &edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    let calc_cut = |partition: &[bool]| -> usize {
        let mut cut = 0;
        for &(u, v) in &edges {
            if partition[u] != partition[v] {
                cut += 1;
            }
        }
        cut
    };

    let mut best_partition = vec![false; n];
    let mut best_cut = usize::MAX;

    for trial in 0..1000 {
        let mut partition = vec![false; n];

        if trial == 0 {
            for i in 0..(n/2) {
                partition[i] = true;
            }
        } else {
            for i in 0..n {
                partition[i] = ((i + trial * 17) % n) < n / 2;
            }
        }

        let mut improved = true;
        while improved {
            improved = false;
            for i in 0..n {
                for j in (i+1)..n {
                    if partition[i] == partition[j] {
                        continue;
                    }

                    let mut delta = 0i32;

                    for &neighbor in &adj[i] {
                        if neighbor != j {
                            if partition[neighbor] == partition[i] {
                                delta += 1;
                            } else {
                                delta -= 1;
                            }
                        }
                    }

                    for &neighbor in &adj[j] {
                        if neighbor != i {
                            if partition[neighbor] == partition[j] {
                                delta += 1;
                            } else {
                                delta -= 1;
                            }
                        }
                    }

                    if delta < 0 {
                        partition[i] = !partition[i];
                        partition[j] = !partition[j];
                        improved = true;
                    }
                }
            }
        }

        let cut = calc_cut(&partition);
        if cut < best_cut {
            best_cut = cut;
            best_partition = partition.clone();
        }
    }

    let partition_a: Vec<usize> = (0..n)
        .filter(|&i| best_partition[i])
        .collect();

    println!("{}", best_cut);
    println!("{}", partition_a.iter()
        .map(|x| x.to_string())
        .collect::<Vec<_>>()
        .join(" "));
}
