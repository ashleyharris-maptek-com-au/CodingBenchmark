def solve_exact_cover():
    # Read input
    line = input().split()
    U = int(line[0])
    S = int(line[1])

    sets = []
    for i in range(S):
        s = list(map(int, input().split()))
        sets.append(frozenset(s))

    # Find exact cover using backtracking
    def backtrack(index, covered, used_indices):
        # If all elements are covered, we have a solution
        if covered == set(range(U)):
            return used_indices

        # If we've exhausted all sets without covering everything
        if index == S:
            return None

        # Try including the current set (only if it doesn't overlap with covered elements)
        if not (covered & sets[index]):
            new_covered = covered | sets[index]
            result = backtrack(index + 1, new_covered, used_indices + [index])
            if result is not None:
                return result

        # Try not including the current set
        result = backtrack(index + 1, covered, used_indices)
        if result is not None:
            return result

        return None

    solution = backtrack(0, set(), [])

    if solution is not None:
        print("SOLUTION")
        print(" ".join(map(str, solution)))
    else:
        print("NO SOLUTION")

solve_exact_cover()
