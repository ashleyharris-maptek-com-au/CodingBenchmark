import sys

def solve_exact_cover():
    lines = sys.stdin.read().strip().split('\n')

    # Parse first line
    U, S = map(int, lines[0].split())

    # Parse sets
    sets = []
    for i in range(1, S + 1):
        if i < len(lines):
            elements = list(map(int, lines[i].split())) if lines[i].strip() else []
        else:
            elements = []
        sets.append(frozenset(elements))

    # Backtracking to find exact cover
    def backtrack(covered, index, chosen):
        # All elements covered - found solution
        if len(covered) == U:
            return chosen[:]

        # No more sets to try
        if index >= S:
            return None

        # Pruning: check if remaining elements can be covered
        uncovered = set(range(U)) - covered
        remaining_coverage = set()
        for i in range(index, S):
            remaining_coverage |= sets[i]

        if not uncovered <= remaining_coverage:
            return None

        # Try including current set (if no overlap)
        if not (covered & sets[index]):
            chosen.append(index)
            result = backtrack(covered | sets[index], index + 1, chosen)
            if result is not None:
                return result
            chosen.pop()

        # Try not including current set
        result = backtrack(covered, index + 1, chosen)
        return result

    solution = backtrack(set(), 0, [])

    if solution is not None:
        print("SOLUTION")
        print(" ".join(map(str, solution)))
    else:
        print("NO SOLUTION")

solve_exact_cover()
