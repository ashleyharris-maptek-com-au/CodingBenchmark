OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0008 = OpConstant %float 0.008
%float_neg0008 = OpConstant %float -0.008
%float_0_5 = OpConstant %float 0.5

%gl_PerVertex_in = OpTypeStruct %v4float
%arr6_gl_pv = OpTypeArray %gl_PerVertex_in %int_6
%ptr_in_arr_pv = OpTypePointer Input %arr6_gl_pv
%ptr_in_v4 = OpTypePointer Input %v4float

%arr6_v3 = OpTypeArray %v3float %int_6
%ptr_in_arr_v3 = OpTypePointer Input %arr6_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main_func = OpTypeFunction %void

%main = OpFunction %void None %main_func
%entry = OpLabel

; Load main triangle vertices (indices 0, 2, 4)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4_ptr

; ===== EDGE 0->2 =====
; Average color for edge
%c01_s0 = OpVectorTimesScalar %v3float %c0 %float_0_5
%c01_s1 = OpVectorTimesScalar %v3float %c2 %float_0_5
%c01_avg = OpFAdd %v3float %c01_s0 %c01_s1

; Extract edge direction in NDC space
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%e02_dir = OpFSub %v2float %p2_xy %p0_xy

; Compute perpendicular (rotate 90 degrees: (x,y) -> (-y,x))
%e02_perp_x = OpFNegate %float %e02_dir.y
%e02_perp_y = OpFMul %float %e02_dir.x %float_1
%e02_perp = OpCompositeConstruct %v2float %e02_perp_x %e02_perp_y

; Normalize and scale by thickness
%e02_perp_len_sq = OpDot %float %e02_perp %e02_perp
%e02_perp_len = OpSqrt %float %e02_perp_len_sq
%e02_perp_norm = OpFDiv %v2float %e02_perp %e02_perp_len
%e02_offset = OpVectorTimesScalar %v2float %e02_perp_norm %float_0008

; Emit 4 vertices as triangle strip
; V0: p0 - offset
%p0_off_neg = OpFSub %v2float %p0_xy %e02_offset
%v0_pos = OpVectorShuffle %v4float %p0 %p0 2 3 2 3
%v0_xy = OpCompositeConstruct %v4float %p0_off_neg.x %p0_off_neg.y %float_0 %float_1
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %v0_xy
%out_col_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_col_ptr %c01_avg
OpEmitVertex

; V1: p0 + offset
%p0_off_pos = OpFAdd %v2float %p0_xy %e02_offset
%v1_xy = OpCompositeConstruct %v4float %p0_off_pos.x %p0_off_pos.y %float_0 %float_1
OpStore %out_pos_ptr %v1_xy
OpStore %out_col_ptr %c01_avg
OpEmitVertex

; V2: p2 + offset
%p2_off_pos = OpFAdd %v2float %p2_xy %e02_offset
%v2_xy = OpCompositeConstruct %v4float %p2_off_pos.x %p2_off_pos.y %float_0 %float_1
OpStore %out_pos_ptr %v2_xy
OpStore %out_col_ptr %c01_avg
OpEmitVertex

; V3: p2 - offset
%p2_off_neg = OpFSub %v2float %p2_xy %e02_offset
%v3_xy = OpCompositeConstruct %v4float %p2_off_neg.x %p2_off_neg.y %float_0 %float_1
OpStore %out_pos_ptr %v3_xy
OpStore %out_col_ptr %c01_avg
OpEmitVertex

OpEndPrimitive

; ===== EDGE 2->4 =====
; Average color for edge
%c24_s0 = OpVectorTimesScalar %v3float %c2 %float_0_5
%c24_s1 = OpVectorTimesScalar %v3float %c4 %float_0_5
%c24_avg = OpFAdd %v3float %c24_s0 %c24_s1

; Edge direction
%e24_dir = OpFSub %v2float %p4_xy %p2_xy

; Perpendicular
%e24_perp_x = OpFNegate %float %e24_dir.y
%e24_perp_y = OpFMul %float %e24_dir.x %float_1
%e24_perp = OpCompositeConstruct %v2float %e24_perp_x %e24_perp_y

; Normalize and scale
%e24_perp_len_sq = OpDot %float %e24_perp %e24_perp
%e24_perp_len = OpSqrt %float %e24_perp_len_sq
%e24_perp_norm = OpFDiv %v2float %e24_perp %e24_perp_len
%e24_offset = OpVectorTimesScalar %v2float %e24_perp_norm %float_0008

; Emit 4 vertices
%p4_xy = OpVectorShuffle %v2float %p4 %p4 0 1

%p2_off_neg_24 = OpFSub %v2float %p2_xy %e24_offset
%v4_xy = OpCompositeConstruct %v4float %p2_off_neg_24.x %p2_off_neg_24.y %float_0 %float_1
OpStore %out_pos_ptr %v4_xy
OpStore %out_col_ptr %c24_avg
OpEmitVertex

%p2_off_pos_24 = OpFAdd %v2float %p2_xy %e24_offset
%v5_xy = OpCompositeConstruct %v4float %p2_off_pos_24.x %p2_off_pos_24.y %float_0 %float_1
OpStore %out_pos_ptr %v5_xy
OpStore %out_col_ptr %c24_avg
OpEmitVertex

%p4_off_pos = OpFAdd %v2float %p4_xy %e24_offset
%v6_xy = OpCompositeConstruct %v4float %p4_off_pos.x %p4_off_pos.y %float_0 %float_1
OpStore %out_pos_ptr %v6_xy
OpStore %out_col_ptr %c24_avg
OpEmitVertex

%p4_off_neg = OpFSub %v2float %p4_xy %e24_offset
%v7_xy = OpCompositeConstruct %v4float %p4_off_neg.x %p4_off_neg.y %float_0 %float_1
OpStore %out_pos_ptr %v7_xy
OpStore %out_col_ptr %c24_avg
OpEmitVertex

OpEndPrimitive

; ===== EDGE 4->0 =====
; Average color for edge
%c40_s0 = OpVectorTimesScalar %v3float %c4 %float_0_5
%c40_s1 = OpVectorTimesScalar %v3float %c0 %float_0_5
%c40_avg = OpFAdd %v3float %c40_s0 %c40_s1

; Edge direction
%e40_dir = OpFSub %v2float %p0_xy %p4_xy

; Perpendicular
%e40_perp_x = OpFNegate %float %e40_dir.y
%e40_perp_y = OpFMul %float %e40_dir.x %float_1
%e40_perp = OpCompositeConstruct %v2float %e40_perp_x %e40_perp_y

; Normalize and scale
%e40_perp_len_sq = OpDot %float %e40_perp %e40_perp
%e40_perp_len = OpSqrt %float %e40_perp_len_sq
%e40_perp_norm = OpFDiv %v2float %e40_perp %e40_perp_len
%e40_offset = OpVectorTimesScalar %v2float %e40_perp_norm %float_0008

; Emit 4 vertices
%p4_off_neg_40 = OpFSub %v2float %p4_xy %e40_offset
%v8_xy = OpCompositeConstruct %v4float %p4_off_neg_40.x %p4_off_neg_40.y %float_0 %float_1
OpStore %out_pos_ptr %v8_xy
OpStore %out_col_ptr %c40_avg
OpEmitVertex

%p4_off_pos_40 = OpFAdd %v2float %p4_xy %e40_offset
%v9_xy = OpCompositeConstruct %v4float %p4_off_pos_40.x %p4_off_pos_40.y %float_0 %float_1
OpStore %out_pos_ptr %v9_xy
OpStore %out_col_ptr %c40_avg
OpEmitVertex

%p0_off_pos_40 = OpFAdd %v2float %p0_xy %e40_offset
%v10_xy = OpCompositeConstruct %v4float %p0_off_pos_40.x %p0_off_pos_40.y %float_0 %float_1
OpStore %out_pos_ptr %v10_xy
OpStore %out_col_ptr %c40_avg
OpEmitVertex

%p0_off_neg_40 = OpFSub %v2float %p0_xy %e40_offset
%v11_xy = OpCompositeConstruct %v4float %p0_off_neg_40.x %p0_off_neg_40.y %float_0 %float_1
OpStore %out_pos_ptr %v11_xy
OpStore %out_col_ptr %c40_avg
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
