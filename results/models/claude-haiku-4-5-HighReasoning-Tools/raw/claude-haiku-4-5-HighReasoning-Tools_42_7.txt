OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpExtInstImport "GLSL.std.450"
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Flatten
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
    %uint_3 = OpConstant %uint 3
%arr_in_pv = OpTypeArray %gl_PerVertex_in %uint_3
%arr_in_v3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_pv = OpTypePointer Input %arr_in_pv
%ptr_Input_arr_v3 = OpTypePointer Input %arr_in_v3
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
  %fn_void = OpTypeFunction %void
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_2 = OpConstant %int 2
%float_0p08 = OpConstant %float 0.08
  %float_0 = OpConstant %float 0.0
      %gl_in = OpVariable %ptr_Input_arr_pv Input
   %worldPos = OpVariable %ptr_Input_arr_v3 Input
     %normal = OpVariable %ptr_Input_arr_v3 Input
      %color = OpVariable %ptr_Input_arr_v3 Input
      %gl_out = OpVariable %ptr_Output_gl_PerVertex Output
    %outColor = OpVariable %ptr_Output_v3 Output
       %glsl = OpExtInstImport "GLSL.std.450"
       %main = OpFunction %void None %fn_void
      %entry = OpLabel
    %p0_addr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
     %p0_v4 = OpLoad %v4float %p0_addr
    %p1_addr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
     %p1_v4 = OpLoad %v4float %p1_addr
    %p2_addr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
     %p2_v4 = OpLoad %v4float %p2_addr
         %p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
         %p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
         %p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
         %e1 = OpFSub %v3float %p1 %p0
         %e2 = OpFSub %v3float %p2 %p0
      %cross = OpExtInst %v3float %glsl Cross %e1 %e2
   %face_norm = OpExtInst %v3float %glsl Normalize %cross
     %offset = OpVectorTimesScalar %v3float %face_norm %float_0p08
    %c0_addr = OpAccessChain %ptr_Input_v3 %color %int_0
         %c0 = OpLoad %v3float %c0_addr
    %c1_addr = OpAccessChain %ptr_Input_v3 %color %int_1
         %c1 = OpLoad %v3float %c1_addr
    %c2_addr = OpAccessChain %ptr_Input_v3 %color %int_2
         %c2 = OpLoad %v3float %c2_addr
      %off_x = OpCompositeExtract %float %offset 0
      %off_y = OpCompositeExtract %float %offset 1
      %off_z = OpCompositeExtract %float %offset 2
  %offset_v4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0
     %out_p0 = OpFAdd %v4float %p0_v4 %offset_v4
    %pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
               OpStore %pos_ptr %out_p0
               OpStore %outColor %c0
               OpEmitVertex
     %out_p1 = OpFAdd %v4float %p1_v4 %offset_v4
               OpStore %pos_ptr %out_p1
               OpStore %outColor %c1
               OpEmitVertex
     %out_p2 = OpFAdd %v4float %p2_v4 %offset_v4
               OpStore %pos_ptr %out_p2
               OpStore %outColor %c2
               OpEmitVertex
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
