; SPIR-V

; Bound: 120
; Schema: 0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%func_void = OpTypeFunction %void

; Constants
%c0f = OpConstant %float 0.0
%c1f = OpConstant %float 1.0
%c0p4f = OpConstant %float 0.4
%c0p6f = OpConstant %float 0.6
%c0p9f = OpConstant %float 0.9
%c0p1f = OpConstant %float 0.1
%cpif = OpConstant %float 3.14159
%c16f = OpConstant %float 16.0
%cred = OpConstantComposite %v3float %c0p9f %c0p1f %c0p1f
%cblue = OpConstantComposite %v3float %c0p1f %c0p1f %c0p9f

; Global variables
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%out_outColor = OpVariable %ptr_Output_v4float Output

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal
%uv = OpLoad %v2float %in_uv

; Normalize normal
%normal_norm = OpExtInst %v3float %1 Normalize %normal

; Load UBO and extract light position
%ubo = OpLoad %UBO %ubo_var
%lightv4 = OpCompositeExtract %v4float %ubo 3
%lightPos = OpVectorShuffle %v3float %lightv4 %lightv4 0 1 2

; Compute light direction
%lightDir = OpFSub %v3float %lightPos %worldPos
%lightDir_norm = OpExtInst %v3float %1 Normalize %lightDir

; Compute diffuse (N dot L, clamped to [0, 1])
%dotNL = OpDot %float %normal_norm %lightDir_norm
%diffuse = OpExtInst %float %1 FMax %c0f %dotNL

; Compute Lambertian factor: 0.4 + 0.6 * diffuse
%factor_term = OpFMul %float %c0p6f %diffuse
%lambertian = OpFAdd %float %c0p4f %factor_term

; Compute stripe pattern: sin(uv.y * pi * 16)
%uv_y = OpCompositeExtract %float %uv 1
%pi_times_16 = OpFMul %float %cpif %c16f
%angle = OpFMul %float %uv_y %pi_times_16
%sine_val = OpExtInst %float %1 Sin %angle

; Select color based on sine sign
%sine_positive = OpFOrdGreaterThan %bool %sine_val %c0f
%base_color = OpSelect %v3float %sine_positive %cred %cblue

; Apply Lambertian shading to color
%shaded_color = OpVectorTimesScalar %v3float %base_color %lambertian

; Extract components for output
%out_r = OpCompositeExtract %float %shaded_color 0
%out_g = OpCompositeExtract %float %shaded_color 1
%out_b = OpCompositeExtract %float %shaded_color 2

; Construct output color with alpha=1.0
%output = OpCompositeConstruct %v4float %out_r %out_g %out_b %c1f

; Store output
OpStore %out_outColor %output
OpReturn
OpFunctionEnd
