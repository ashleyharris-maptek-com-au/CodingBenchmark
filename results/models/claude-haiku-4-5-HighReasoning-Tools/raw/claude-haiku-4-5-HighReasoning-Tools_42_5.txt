OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_2_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_2_v3f = OpTypeArray %v3float %uint_2

%ptr_in_arr_pv = OpTypePointer Input %arr_2_pv_in
%ptr_in_v4f = OpTypePointer Input %v4float
%ptr_in_arr_v3f = OpTypePointer Input %arr_2_v3f
%ptr_in_v3f = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4f = OpTypePointer Output %v4float
%ptr_out_v3f = OpTypePointer Output %v3float

%func_void = OpTypeFunction %void

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3f Input
%normal_in = OpVariable %ptr_in_arr_v3f Input
%color_in = OpVariable %ptr_in_arr_v3f Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3f Output

%main = OpFunction %void None %func_void
%entry = OpLabel

%p0_addr = OpAccessChain %ptr_in_v4f %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_addr
%p1_addr = OpAccessChain %ptr_in_v4f %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_addr

%c0_addr = OpAccessChain %ptr_in_v3f %color_in %int_0
%c0 = OpLoad %v3float %c0_addr
%c1_addr = OpAccessChain %ptr_in_v3f %color_in %int_1
%c1 = OpLoad %v3float %c1_addr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

%dir = OpFSub %v3float %p1_xyz %p0_xyz

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%perp_x = OpFNegate %float %dir_y
%perp_y = %dir_x
%perp_unnorm = OpCompositeConstruct %v3float %perp_x %perp_y %float_0

%perp_len_sq = OpDot %float %perp_unnorm %perp_unnorm
%perp_len = OpSqrt %float %perp_len_sq
%perp_scale = OpFDiv %float %float_1 %perp_len
%perp = OpVectorTimesScalar %v3float %perp_unnorm %perp_scale

%dir_scaled = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0_xyz %dir_scaled

%perp_t = OpVectorTimesScalar %v3float %perp %float_0_015

%sv0 = OpFSub %v3float %p0_xyz %perp_t
%sv0_x = OpCompositeExtract %float %sv0 0
%sv0_y = OpCompositeExtract %float %sv0 1
%sv0_z = OpCompositeExtract %float %sv0 2
%sv0_4 = OpCompositeConstruct %v4float %sv0_x %sv0_y %sv0_z %float_1
%sv0_pos_addr = OpAccessChain %ptr_out_v4f %gl_out %int_0
OpStore %sv0_pos_addr %sv0_4
OpStore %outColor %c0
OpEmitVertex

%sv1 = OpFAdd %v3float %p0_xyz %perp_t
%sv1_x = OpCompositeExtract %float %sv1 0
%sv1_y = OpCompositeExtract %float %sv1 1
%sv1_z = OpCompositeExtract %float %sv1 2
%sv1_4 = OpCompositeConstruct %v4float %sv1_x %sv1_y %sv1_z %float_1
OpStore %sv0_pos_addr %sv1_4
OpStore %outColor %c0
OpEmitVertex

%sv2 = OpFSub %v3float %mid %perp_t
%sv2_x = OpCompositeExtract %float %sv2 0
%sv2_y = OpCompositeExtract %float %sv2 1
%sv2_z = OpCompositeExtract %float %sv2 2
%sv2_4 = OpCompositeConstruct %v4float %sv2_x %sv2_y %sv2_z %float_1
OpStore %sv0_pos_addr %sv2_4
OpStore %outColor %c0
OpEmitVertex

%sv3 = OpFAdd %v3float %mid %perp_t
%sv3_x = OpCompositeExtract %float %sv3 0
%sv3_y = OpCompositeExtract %float %sv3 1
%sv3_z = OpCompositeExtract %float %sv3 2
%sv3_4 = OpCompositeConstruct %v4float %sv3_x %sv3_y %sv3_z %float_1
OpStore %sv0_pos_addr %sv3_4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

%perp_a = OpVectorTimesScalar %v3float %perp %float_0_05

%av0 = OpFAdd %v3float %mid %perp_a
%av0_x = OpCompositeExtract %float %av0 0
%av0_y = OpCompositeExtract %float %av0 1
%av0_z = OpCompositeExtract %float %av0 2
%av0_4 = OpCompositeConstruct %v4float %av0_x %av0_y %av0_z %float_1
OpStore %sv0_pos_addr %av0_4
OpStore %outColor %c1
OpEmitVertex

%av1 = OpFSub %v3float %mid %perp_a
%av1_x = OpCompositeExtract %float %av1 0
%av1_y = OpCompositeExtract %float %av1 1
%av1_z = OpCompositeExtract %float %av1 2
%av1_4 = OpCompositeConstruct %v4float %av1_x %av1_y %av1_z %float_1
OpStore %sv0_pos_addr %av1_4
OpStore %outColor %c1
OpEmitVertex

%av2_4 = %p1
OpStore %sv0_pos_addr %av2_4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
