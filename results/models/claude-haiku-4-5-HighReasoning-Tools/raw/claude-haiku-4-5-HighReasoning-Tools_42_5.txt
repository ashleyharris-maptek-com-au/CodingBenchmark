OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inColor Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %vec4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_vec4 = OpTypePointer Input %vec4

%arr_vec3 = OpTypeArray %vec3 %int_2
%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3
%ptr_in_vec3 = OpTypePointer Input %vec3

%gl_PerVertex_out = OpTypeStruct %vec4
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_vec4 = OpTypePointer Output %vec4
%ptr_out_vec3 = OpTypePointer Output %vec3

%gl_in = OpVariable %ptr_in_arr_pv Input
%inColor = OpVariable %ptr_in_arr_vec3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_vec3 Output

%main_func_type = OpTypeFunction %void
%main = OpFunction %void None %main_func_type
%entry = OpLabel

; Load input positions from gl_in
%p0_chain = OpAccessChain %ptr_in_vec4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %vec4 %p0_chain
%p0 = OpVectorShuffle %vec3 %p0_v4 %p0_v4 0 1 2

%p1_chain = OpAccessChain %ptr_in_vec4 %gl_in %int_1 %int_0
%p1_v4 = OpLoad %vec4 %p1_chain
%p1 = OpVectorShuffle %vec3 %p1_v4 %p1_v4 0 1 2

; Load colors
%c0_chain = OpAccessChain %ptr_in_vec3 %inColor %int_0
%c0 = OpLoad %vec3 %c0_chain

%c1_chain = OpAccessChain %ptr_in_vec3 %inColor %int_1
%c1 = OpLoad %vec3 %c1_chain

; Compute direction: dir = p1 - p0
%dir = OpFSub %vec3 %p1 %p0

; Compute perpendicular: perp = normalize((-dir.y, dir.x, 0))
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %vec3 %neg_y %dir_x %float_0
%perp = OpExtInst %vec3 GLSL450 Normalize %perp_unnorm

; Compute mid: mid = p0 + 0.75 * dir
%dir_scaled = OpVectorTimesScalar %vec3 %dir %float_0_75
%mid = OpFAdd %vec3 %p0 %dir_scaled

; === SHAFT (4 vertices) ===
; Vertex 0: p0 - perp*0.015
%t1 = OpVectorTimesScalar %vec3 %perp %float_0_015
%sh0 = OpFSub %vec3 %p0 %t1
%sh0_v4 = OpCompositeConstruct %vec4 %sh0 %float_1
%out_pos = OpAccessChain %ptr_out_vec4 %gl_out %int_0
OpStore %out_pos %sh0_v4
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0 + perp*0.015
%sh1 = OpFAdd %vec3 %p0 %t1
%sh1_v4 = OpCompositeConstruct %vec4 %sh1 %float_1
OpStore %out_pos %sh1_v4
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: mid - perp*0.015
%sh2 = OpFSub %vec3 %mid %t1
%sh2_v4 = OpCompositeConstruct %vec4 %sh2 %float_1
OpStore %out_pos %sh2_v4
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: mid + perp*0.015
%sh3 = OpFAdd %vec3 %mid %t1
%sh3_v4 = OpCompositeConstruct %vec4 %sh3 %float_1
OpStore %out_pos %sh3_v4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; === ARROWHEAD (3 vertices) ===
; Vertex 4: mid + perp*0.05
%t2 = OpVectorTimesScalar %vec3 %perp %float_0_05
%ah0 = OpFAdd %vec3 %mid %t2
%ah0_v4 = OpCompositeConstruct %vec4 %ah0 %float_1
OpStore %out_pos %ah0_v4
OpStore %outColor %c1
OpEmitVertex

; Vertex 5: mid - perp*0.05
%ah1 = OpFSub %vec3 %mid %t2
%ah1_v4 = OpCompositeConstruct %vec4 %ah1 %float_1
OpStore %out_pos %ah1_v4
OpStore %outColor %c1
OpEmitVertex

; Vertex 6: p1 (tip)
%p1_final = OpCompositeConstruct %vec4 %p1 %float_1
OpStore %out_pos %p1_final
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
