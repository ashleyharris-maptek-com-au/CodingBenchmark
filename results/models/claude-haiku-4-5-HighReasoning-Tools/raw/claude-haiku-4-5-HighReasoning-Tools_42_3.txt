OpCapability Geometry
OpCapability Shader
OpExtension "GLSL.std.450"
OpMemoryModel Logical GLSL450

%glsl = OpExtInstImport "GLSL.std.450"

OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor

OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %color_in Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

; Type definitions
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Struct types
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array types
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in = OpTypeArray %v3float %uint_2

; Pointer types
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_input_v4f = OpTypePointer Input %v4float
%ptr_input_v3f = OpTypePointer Input %v3float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4f = OpTypePointer Output %v4float
%ptr_output_v3f = OpTypePointer Output %v3float

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_03 = OpConstant %float 0.03

; Variables
%gl_in = OpVariable %ptr_input_arr_pv Input
%color_in = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3f Output

; Function type
%func_type = OpTypeFunction %void

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input positions
%pos0_ptr = OpAccessChain %ptr_input_v4f %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr

%pos1_ptr = OpAccessChain %ptr_input_v4f %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr

; Load input colors
%color0_ptr = OpAccessChain %ptr_input_v3f %color_in %int_0
%color0 = OpLoad %v3float %color0_ptr

%color1_ptr = OpAccessChain %ptr_input_v3f %color_in %int_1
%color1 = OpLoad %v3float %color1_ptr

; Extract 2D positions
%pos0_2d = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_2d = OpVectorShuffle %v2float %pos1 %pos1 0 1

; Compute direction
%dir = OpFSub %v2float %pos1_2d %pos0_2d

; Compute perpendicular: (-dir.y, dir.x)
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize and scale by thickness
%perp_norm = OpExtInst %v2float %glsl Normalize %perp_unnorm
%perp = OpVectorTimesScalar %v2float %perp_norm %float_03

; Extract z and w components
%pos0_z = OpCompositeExtract %float %pos0 2
%pos0_w = OpCompositeExtract %float %pos0 3
%pos1_z = OpCompositeExtract %float %pos1 2
%pos1_w = OpCompositeExtract %float %pos1 3

; Emit vertex 1: pos0 + perp
%v1_2d = OpFAdd %v2float %pos0_2d %perp
%v1_x = OpCompositeExtract %float %v1_2d 0
%v1_y = OpCompositeExtract %float %v1_2d 1
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %pos0_z %pos0_w

%out_pos = OpAccessChain %ptr_output_v4f %gl_out %int_0
OpStore %out_pos %v1
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 2: pos0 - perp
%v2_2d = OpFSub %v2float %pos0_2d %perp
%v2_x = OpCompositeExtract %float %v2_2d 0
%v2_y = OpCompositeExtract %float %v2_2d 1
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %pos0_z %pos0_w

OpStore %out_pos %v2
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 3: pos1 + perp
%v3_2d = OpFAdd %v2float %pos1_2d %perp
%v3_x = OpCompositeExtract %float %v3_2d 0
%v3_y = OpCompositeExtract %float %v3_2d 1
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %pos1_z %pos1_w

OpStore %out_pos %v3
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 4: pos1 - perp
%v4_2d = OpFSub %v2float %pos1_2d %perp
%v4_x = OpCompositeExtract %float %v4_2d 0
%v4_y = OpCompositeExtract %float %v4_2d 1
%v4 = OpCompositeConstruct %v4float %v4_x %v4_y %pos1_z %pos1_w

OpStore %out_pos %v4
OpStore %outColor %color1
OpEmitVertex

; End the primitive strip
OpEndPrimitive

OpReturn
OpFunctionEnd
