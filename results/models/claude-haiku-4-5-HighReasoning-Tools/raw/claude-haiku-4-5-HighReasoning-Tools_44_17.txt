#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Scale UV to brick grid (8x16)
    vec2 scaledUv = uv * vec2(8.0, 16.0);

    // Get row index for staggering
    float rowIndex = floor(scaledUv.y);

    // Apply half-brick offset to odd rows
    float offsetX = mod(rowIndex, 2.0) * 0.5;
    scaledUv.x += offsetX;

    // Get brick coordinates and local position within brick
    vec2 brickPos = floor(scaledUv);
    vec2 localUv = fract(scaledUv);

    // Mortar thickness from edges of each brick
    float mortarX = 0.05;
    float mortarY = 0.1;

    // Determine if current fragment is in mortar or brick
    bool inMortarX = localUv.x < mortarX || localUv.x > (1.0 - mortarX);
    bool inMortarY = localUv.y < mortarY || localUv.y > (1.0 - mortarY);
    bool inMortar = inMortarX || inMortarY;

    // Apply appropriate color
    vec3 surfaceColor = inMortar ? vec3(0.7, 0.7, 0.7) : vec3(0.7, 0.2, 0.1);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Output final color with shading applied
    vec3 finalColor = surfaceColor * shadingFactor;
    outColor = vec4(finalColor, 1.0);
}
