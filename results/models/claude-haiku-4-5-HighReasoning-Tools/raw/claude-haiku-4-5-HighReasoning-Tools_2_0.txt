using System;
using System.Collections.Generic;
using System.Linq;

class Solution
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        var edges = new List<(int u, int v, int w)>();
        var adj = new List<(int v, int w, int idx)>[n];

        for (int i = 0; i < n; i++)
            adj[i] = new List<(int, int, int)>();

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);

            edges.Add((u, v, w));
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
        }

        // Find odd-degree vertices
        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (adj[i].Count % 2 == 1)
                oddVertices.Add(i);
        }

        // Augmented edges: store edge indices for each adjacency
        var augEdges = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            augEdges[i] = new List<int>();
            foreach (var (_, _, idx) in adj[i])
                augEdges[i].Add(idx);
        }

        if (oddVertices.Count > 0)
        {
            // Compute shortest paths and minimum matching
            int k = oddVertices.Count;
            var dist = new int[k][];
            var paths = new List<int>[k][];

            for (int i = 0; i < k; i++)
            {
                var (d, p) = Dijkstra(adj, oddVertices[i], n);
                dist[i] = new int[k];
                paths[i] = new List<int>[k];

                for (int j = 0; j < k; j++)
                {
                    dist[i][j] = d[oddVertices[j]];
                    paths[i][j] = p[oddVertices[j]];
                }
            }

            // Find minimum weight perfect matching
            var matching = (k <= 20) ?
                MinWeightMatchingDP(dist, k) :
                MinWeightMatchingGreedy(dist, k);

            // Add edges from shortest paths
            foreach (var (ui, vi) in matching)
            {
                var path = paths[ui][vi];
                for (int j = 0; j < path.Count - 1; j++)
                {
                    int a = path[j];
                    int b = path[j + 1];

                    // Find edge between a and b
                    int edgeIdx = -1;
                    foreach (var (nbr, _, idx) in adj[a])
                    {
                        if (nbr == b)
                        {
                            edgeIdx = idx;
                            break;
                        }
                    }

                    if (edgeIdx >= 0)
                    {
                        augEdges[a].Add(edgeIdx);
                        augEdges[b].Add(edgeIdx);
                    }
                }
            }
        }

        // Find Eulerian circuit
        var circuit = FindEulerianCircuit(augEdges, edges, n);

        Console.WriteLine(circuit.Count);
        Console.WriteLine(string.Join(" ", circuit));
    }

    static (int[], List<int>[]) Dijkstra(List<(int, int, int)>[] adj, int start, int n)
    {
        var dist = new int[n];
        var prev = new int[n];
        var path = new List<int>[n];

        Array.Fill(dist, int.MaxValue);
        Array.Fill(prev, -1);
        dist[start] = 0;

        var pq = new PriorityQueue<(int, int), int>();
        pq.Enqueue((0, start), 0);

        while (pq.Count > 0)
        {
            var (d, u) = pq.Dequeue();

            if (d > dist[u]) continue;

            foreach (var (v, w, _) in adj[u])
            {
                if (dist[u] != int.MaxValue && dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    prev[v] = u;
                    path[v] = null;
                    pq.Enqueue((dist[v], v), dist[v]);
                }
            }
        }

        // Reconstruct paths
        for (int i = 0; i < n; i++)
        {
            if (dist[i] != int.MaxValue)
            {
                path[i] = new List<int>();
                int curr = i;
                while (curr != -1)
                {
                    path[i].Add(curr);
                    curr = prev[curr];
                }
                path[i].Reverse();
            }
            else
            {
                path[i] = new List<int>();
            }
        }

        return (dist, path);
    }

    static List<(int, int)> MinWeightMatchingDP(int[][] dist, int k)
    {
        var dp = new int[1 << k];
        var edge = new (int, int)[1 << k];

        Array.Fill(dp, int.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == int.MaxValue) continue;

            int i = -1;
            for (int j = 0; j < k; j++)
            {
                if ((mask & (1 << j)) == 0)
                {
                    i = j;
                    break;
                }
            }

            if (i == -1) continue;

            for (int j = i + 1; j < k; j++)
            {
                if ((mask & (1 << j)) == 0)
                {
                    int newMask = mask | (1 << i) | (1 << j);
                    int cost = dist[i][j];
                    if (dp[mask] + cost < dp[newMask])
                    {
                        dp[newMask] = dp[mask] + cost;
                        edge[newMask] = (i, j);
                    }
                }
            }
        }

        var result = new List<(int, int)>();
        int curr = (1 << k) - 1;
        while (curr > 0)
        {
            var (i, j) = edge[curr];
            result.Add((i, j));
            curr ^= (1 << i) | (1 << j);
        }

        return result;
    }

    static List<(int, int)> MinWeightMatchingGreedy(int[][] dist, int k)
    {
        var edgeList = new List<(int, int, int)>();

        for (int i = 0; i < k; i++)
        {
            for (int j = i + 1; j < k; j++)
            {
                edgeList.Add((dist[i][j], i, j));
            }
        }

        edgeList.Sort();

        var matched = new bool[k];
        var result = new List<(int, int)>();

        foreach (var (cost, i, j) in edgeList)
        {
            if (!matched[i] && !matched[j])
            {
                matched[i] = matched[j] = true;
                result.Add((i, j));
            }
        }

        return result;
    }

    static List<int> FindEulerianCircuit(List<int>[] augEdges, List<(int, int, int)> edges, int n)
    {
        var stack = new Stack<int>();
        var path = new List<int>();
        var adj = new List<int>[n];

        for (int i = 0; i < n; i++)
            adj[i] = new List<int>(augEdges[i]);

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();

            if (adj[u].Count > 0)
            {
                int edgeIdx = adj[u][adj[u].Count - 1];
                adj[u].RemoveAt(adj[u].Count - 1);

                var (eu, ev, _) = edges[edgeIdx];
                int v = (eu == u) ? ev : eu;

                // Remove matching edge from v
                for (int i = adj[v].Count - 1; i >= 0; i--)
                {
                    if (adj[v][i] == edgeIdx)
                    {
                        adj[v].RemoveAt(i);
                        break;
                    }
                }

                stack.Push(v);
            }
            else
            {
                path.Add(stack.Pop());
            }
        }

        path.Reverse();
        return path;
    }
}
