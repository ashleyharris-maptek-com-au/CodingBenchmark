using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read input
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] cutsInput = Console.ReadLine().Split();
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(cutsInput[i]);
        }

        // Create list of (cut_length, original_index) and sort by length descending
        var sortedCuts = new List<(int length, int index)>();
        for (int i = 0; i < n; i++)
        {
            sortedCuts.Add((cuts[i], i));
        }
        sortedCuts.Sort((a, b) => b.length.CompareTo(a.length));

        // Bin packing using First Fit Decreasing
        var stocks = new List<List<int>>();
        var stockRemaining = new List<int>();

        foreach (var (length, index) in sortedCuts)
        {
            // Find first stock with enough space
            int stockIdx = -1;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockRemaining[i] >= length)
                {
                    stockIdx = i;
                    break;
                }
            }

            // If no suitable stock, create a new one
            if (stockIdx == -1)
            {
                stocks.Add(new List<int>());
                stockRemaining.Add(stockLength);
                stockIdx = stocks.Count - 1;
            }

            // Add cut to stock
            stocks[stockIdx].Add(index);
            stockRemaining[stockIdx] -= length;
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
