using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    private int[] columnHeights;
    private int width;
    private int boardHeight;
    private bool[,] fullBoard;

    static void Main()
    {
        var input = Console.ReadLine().Split();
        int width = int.Parse(input[0]);
        int height = int.Parse(input[1]);
        int numPieces = int.Parse(input[2]);

        var player = new TetrisPlayer(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            char pieceType = Console.ReadLine().Trim()[0];
            (int rotation, int column) = player.FindBestMove(pieceType);
            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();
            if (result.StartsWith("cleared_"))
            {
                int clearedLines = int.Parse(result.Substring(8));
                player.ProcessClearedLines(clearedLines);
            }
            else if (result == "gameover")
            {
                break;
            }
        }
    }

    public TetrisPlayer(int w, int h)
    {
        width = w;
        boardHeight = h;
        columnHeights = new int[width];
        fullBoard = new bool[boardHeight, width];
    }

    public (int rotation, int column) FindBestMove(char pieceType)
    {
        int bestRotation = 0;
        int bestColumn = 0;
        double bestScore = double.MinValue;

        for (int rotation = 0; rotation < 4; rotation++)
        {
            var (shape, w) = GetPieceShape(pieceType, rotation);

            for (int column = 0; column <= this.width - w; column++)
            {
                if (CanPlace(shape, column))
                {
                    double score = EvaluatePlacement(shape, column);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRotation = rotation;
                        bestColumn = column;
                    }
                }
            }
        }

        PlacePiece(GetPieceShape(pieceType, bestRotation).shape, bestColumn);
        return (bestRotation, bestColumn);
    }

    private (int[], int) GetPieceShape(char type, int rotation)
    {
        rotation = rotation % 4;

        return type switch
        {
            'I' => rotation switch
            {
                0 or 2 => (new[] { 0, 1, 2, 3 }, 4),
                _ => (new[] { 0, 0, 0, 0 }, 1)
            },
            'O' => (new[] { 0, 1, 0, 1 }, 2),
            'T' => rotation switch
            {
                0 => (new[] { 0, -1, 0, 1 }, 3),
                1 => (new[] { -1, 0, 0, 1 }, 2),
                2 => (new[] { -1, 0, 1, 0 }, 3),
                _ => (new[] { -1, 0, 1, 0 }, 2)
            },
            'S' => rotation switch
            {
                0 or 2 => (new[] { 0, 1, -1, 0 }, 3),
                _ => (new[] { -1, 0, 0, 1 }, 2)
            },
            'Z' => rotation switch
            {
                0 or 2 => (new[] { -1, 0, 0, 1 }, 3),
                _ => (new[] { 0, -1, 1, 0 }, 2)
            },
            'J' => rotation switch
            {
                0 => (new[] { -1, 0, 1, -1 }, 3),
                1 => (new[] { -1, -1, 0, 1 }, 2),
                2 => (new[] { 0, 1, -1, 1 }, 3),
                _ => (new[] { -1, 0, 1, 1 }, 2)
            },
            'L' => rotation switch
            {
                0 => (new[] { -1, 0, 1, 1 }, 3),
                1 => (new[] { -1, 1, 0, 1 }, 2),
                2 => (new[] { -1, -1, 0, 1 }, 3),
                _ => (new[] { -1, -1, 0, 1 }, 2)
            },
            _ => (new[] { 0 }, 1)
        };
    }

    private bool CanPlace(int[] shape, int startColumn)
    {
        int minCol = int.MaxValue, maxCol = int.MinValue;
        for (int i = 0; i < shape.Length; i++)
        {
            int col = startColumn + shape[i];
            minCol = Math.Min(minCol, col);
            maxCol = Math.Max(maxCol, col);
        }

        return minCol >= 0 && maxCol < width;
    }

    private double EvaluatePlacement(int[] shape, int startColumn)
    {
        var tempHeights = (int[])columnHeights.Clone();

        int maxHeight = 0;
        for (int i = 0; i < shape.Length; i++)
        {
            int col = startColumn + shape[i];
            int newHeight = tempHeights[col] + 1;
            tempHeights[col] = newHeight;
            maxHeight = Math.Max(maxHeight, newHeight);
        }

        double score = 0;
        score -= maxHeight * 50;

        int avgHeight = tempHeights.Sum() / width;
        int heightVariance = tempHeights.Sum(h => Math.Abs(h - avgHeight));
        score -= heightVariance * 2;

        int holes = 0;
        for (int col = 0; col < width; col++)
        {
            int filledCells = 0;
            int gapCells = 0;
            for (int row = 0; row < boardHeight; row++)
            {
                if (fullBoard[row, col])
                    filledCells++;
                else if (row > tempHeights[col])
                    gapCells++;
            }
            holes += gapCells;
        }
        score -= holes * 30;

        return score;
    }

    private void PlacePiece(int[] shape, int startColumn)
    {
        for (int i = 0; i < shape.Length; i++)
        {
            int col = startColumn + shape[i];
            columnHeights[col]++;

            if (columnHeights[col] <= boardHeight)
                fullBoard[columnHeights[col] - 1, col] = true;
        }
    }

    private void ProcessClearedLines(int clearedCount)
    {
        for (int i = 0; i < width; i++)
        {
            columnHeights[i] = Math.Max(0, columnHeights[i] - clearedCount);
        }

        for (int row = clearedCount; row < boardHeight; row++)
        {
            for (int col = 0; col < width; col++)
            {
                if (row - clearedCount >= 0)
                    fullBoard[row - clearedCount, col] = fullBoard[row, col];
                else
                    fullBoard[row - clearedCount, col] = false;
            }
        }

        for (int row = boardHeight - clearedCount; row < boardHeight; row++)
        {
            for (int col = 0; col < width; col++)
            {
                fullBoard[row, col] = false;
            }
        }
    }
}
