#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

const float EPSILON = 1e-6f;
const float INF = numeric_limits<float>::infinity();

struct Vec3 {
    float x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(float t) const { return Vec3(x * t, y * t, z * t); }
    float dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    float length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const {
        float len = length();
        return len > EPSILON ? (*this) * (1.0f / len) : Vec3();
    }
};

struct Face {
    int a, b, c;
};

struct AABB {
    Vec3 min, max;

    AABB() : min(INF, INF, INF), max(-INF, -INF, -INF) {}

    bool rayIntersects(const Vec3& origin, const Vec3& dir) const {
        float tmin = -INF, tmax = INF;
        for (int i = 0; i < 3; i++) {
            float coord_min = (&min.x)[i];
            float coord_max = (&max.x)[i];
            float origin_c = (&origin.x)[i];
            float dir_c = (&dir.x)[i];

            if (abs(dir_c) < EPSILON) {
                if (origin_c < coord_min || origin_c > coord_max) return false;
            } else {
                float t1 = (coord_min - origin_c) / dir_c;
                float t2 = (coord_max - origin_c) / dir_c;
                if (t1 > t2) swap(t1, t2);
                tmin = max(tmin, t1);
                tmax = min(tmax, t2);
                if (tmin > tmax) return false;
            }
        }
        return true;
    }
};

class Mesh {
public:
    vector<Vec3> vertices;
    vector<Face> faces;
    vector<AABB> faceAABBs;
    AABB meshAABB;

    void computeAABBs() {
        faceAABBs.resize(faces.size());
        meshAABB = AABB();

        for (size_t i = 0; i < faces.size(); i++) {
            const Face& f = faces[i];
            const Vec3& v0 = vertices[f.a];
            const Vec3& v1 = vertices[f.b];
            const Vec3& v2 = vertices[f.c];

            AABB& aabb = faceAABBs[i];
            aabb.min = Vec3(min({v0.x, v1.x, v2.x}), min({v0.y, v1.y, v2.y}), min({v0.z, v1.z, v2.z}));
            aabb.max = Vec3(max({v0.x, v1.x, v2.x}), max({v0.y, v1.y, v2.y}), max({v0.z, v1.z, v2.z}));

            meshAABB.min.x = min(meshAABB.min.x, aabb.min.x);
            meshAABB.min.y = min(meshAABB.min.y, aabb.min.y);
            meshAABB.min.z = min(meshAABB.min.z, aabb.min.z);
            meshAABB.max.x = max(meshAABB.max.x, aabb.max.x);
            meshAABB.max.y = max(meshAABB.max.y, aabb.max.y);
            meshAABB.max.z = max(meshAABB.max.z, aabb.max.z);
        }
    }
};

bool rayTriangleIntersect(const Vec3& rayOrigin, const Vec3& rayDir,
                          const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 h = rayDir.cross(edge2);
    float a = edge1.dot(h);

    if (abs(a) < EPSILON) return false;

    float f = 1.0f / a;
    Vec3 s = rayOrigin - v0;
    float u = f * s.dot(h);

    if (u < 0.0f || u > 1.0f) return false;

    Vec3 q = s.cross(edge1);
    float v = f * rayDir.dot(q);

    if (v < 0.0f || u + v > 1.0f) return false;

    float t = f * edge2.dot(q);
    return t > EPSILON;
}

bool pointInMesh(const Vec3& p, const Mesh& mesh) {
    if (!mesh.meshAABB.rayIntersects(p, Vec3(1, 0, 0))) {
        return false;
    }

    Vec3 rayDir(1.0f, 0.0f, 0.0f);
    int count = 0;

    for (size_t i = 0; i < mesh.faces.size(); i++) {
        if (!mesh.faceAABBs[i].rayIntersects(p, rayDir)) continue;

        const Face& face = mesh.faces[i];
        if (rayTriangleIntersect(p, rayDir,
                                mesh.vertices[face.a],
                                mesh.vertices[face.b],
                                mesh.vertices[face.c])) {
            count++;
        }
    }

    return count % 2 == 1;
}

Vec3 getTriangleCenter(const Mesh& mesh, const Face& f) {
    return (mesh.vertices[f.a] + mesh.vertices[f.b] + mesh.vertices[f.c]) * (1.0f / 3.0f);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int va, fa, vb, fb;
    cin >> va >> fa >> vb >> fb;

    Mesh meshA, meshB;
    meshA.vertices.resize(va);
    meshA.faces.resize(fa);
    meshB.vertices.resize(vb);
    meshB.faces.resize(fb);

    for (int i = 0; i < va; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }

    for (int i = 0; i < fa; i++) {
        cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;
    }

    for (int i = 0; i < vb; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }

    for (int i = 0; i < fb; i++) {
        cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;
    }

    meshA.computeAABBs();
    meshB.computeAABBs();

    vector<Vec3> resultVertices;
    vector<Face> resultFaces;
    vector<int> mapA(va, -1), mapB(vb, -1);

    for (int i = 0; i < fa; i++) {
        const Face& face = meshA.faces[i];
        Vec3 center = getTriangleCenter(meshA, face);
        bool centerInB = pointInMesh(center, meshB);

        if (!centerInB) {
            Face newFace;

            if (mapA[face.a] == -1) {
                mapA[face.a] = resultVertices.size();
                resultVertices.push_back(meshA.vertices[face.a]);
            }
            newFace.a = mapA[face.a];

            if (mapA[face.b] == -1) {
                mapA[face.b] = resultVertices.size();
                resultVertices.push_back(meshA.vertices[face.b]);
            }
            newFace.b = mapA[face.b];

            if (mapA[face.c] == -1) {
                mapA[face.c] = resultVertices.size();
                resultVertices.push_back(meshA.vertices[face.c]);
            }
            newFace.c = mapA[face.c];

            resultFaces.push_back(newFace);
        }
    }

    for (int i = 0; i < fb; i++) {
        const Face& face = meshB.faces[i];
        Vec3 center = getTriangleCenter(meshB, face);
        bool centerInA = pointInMesh(center, meshA);

        if (!centerInA) {
            Face newFace;

            if (mapB[face.a] == -1) {
                mapB[face.a] = resultVertices.size();
                resultVertices.push_back(meshB.vertices[face.a]);
            }
            newFace.a = mapB[face.a];

            if (mapB[face.b] == -1) {
                mapB[face.b] = resultVertices.size();
                resultVertices.push_back(meshB.vertices[face.b]);
            }
            newFace.b = mapB[face.b];

            if (mapB[face.c] == -1) {
                mapB[face.c] = resultVertices.size();
                resultVertices.push_back(meshB.vertices[face.c]);
            }
            newFace.c = mapB[face.c];

            resultFaces.push_back(newFace);
        }
    }

    cout << resultVertices.size() << " " << resultFaces.size() << "\n";

    for (const Vec3& v : resultVertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const Face& f : resultFaces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
