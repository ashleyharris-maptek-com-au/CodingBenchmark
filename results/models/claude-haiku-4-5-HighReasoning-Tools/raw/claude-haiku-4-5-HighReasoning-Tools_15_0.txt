use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Vec3 { x, y, z }
    }

    fn normalize(&self) -> Vec3 {
        let len = (self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if len > 1e-10 {
            Vec3 {
                x: self.x / len,
                y: self.y / len,
                z: self.z / len,
            }
        } else {
            *self
        }
    }
}

#[derive(Debug, Clone, Copy)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn identity() -> Self {
        Quaternion {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }
}

#[derive(Debug, Clone)]
struct Tetrahedron {
    position: Vec3,
    rotation: Quaternion,
    scale: f64,
}

fn main() {
    let stdin = io::stdin();
    let reader = stdin.lock();
    let mut lines = reader.lines();

    // Parse number of vertices
    let nv_line = lines.next().unwrap().unwrap();
    let nv: usize = nv_line.trim().parse().unwrap();

    // Parse polygon vertices
    let mut vertices = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        vertices.push(Vec3::new(x, y, 0.0));
    }

    // Parse sun direction
    let sun_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = sun_line.trim().split_whitespace().collect();
    let sx: f64 = parts[0].parse().unwrap();
    let sy: f64 = parts[1].parse().unwrap();
    let sz: f64 = parts[2].parse().unwrap();
    let _sun_dir = Vec3::new(sx, sy, sz).normalize();

    // Calculate bounding box of polygon
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for v in &vertices {
        min_x = min_x.min(v.x);
        max_x = max_x.max(v.x);
        min_y = min_y.min(v.y);
        max_y = max_y.max(v.y);
    }

    let width = (max_x - min_x).abs().max(0.001);
    let height = (max_y - min_y).abs().max(0.001);
    let area = width * height;

    // Adaptive grid sizing based on polygon area
    let grid_size = if area < 10.0 {
        1
    } else if area < 100.0 {
        1
    } else if area < 1000.0 {
        2
    } else if area < 10000.0 {
        3
    } else {
        ((area / 1000.0).sqrt().ceil() as usize).max(3)
    };

    let cell_width = width / grid_size as f64;
    let cell_height = height / grid_size as f64;

    let mut tetrahedrons = Vec::new();

    // Place tetrahedrons in grid pattern
    for i in 0..grid_size {
        for j in 0..grid_size {
            let center_x = min_x + (i as f64 + 0.5) * cell_width;
            let center_y = min_y + (j as f64 + 0.5) * cell_height;

            // Scale tetrahedron to cover the cell
            let base_scale = ((cell_width + cell_height) / 2.0 * 0.75).min(10.0);
            let scale = base_scale.max(0.1);

            // Position above ground to cast shadows
            let z_pos = (scale * 1.5).max(0.5);

            tetrahedrons.push(Tetrahedron {
                position: Vec3::new(center_x, center_y, z_pos),
                rotation: Quaternion::identity(),
                scale,
            });
        }
    }

    // Output tetrahedrons
    println!("{}", tetrahedrons.len());
    for tet in &tetrahedrons {
        println!(
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
            tet.position.x,
            tet.position.y,
            tet.position.z,
            tet.rotation.w,
            tet.rotation.x,
            tet.rotation.y,
            tet.rotation.z,
            tet.scale
        );
    }
}
