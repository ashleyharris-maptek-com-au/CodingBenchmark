use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Clone, Debug)]
struct Sample {
    depth: f64,
    properties: Vec<f64>,
}

#[derive(Clone, Debug)]
struct Drillhole {
    id: String,
    start: (f64, f64, f64),
    direction: (f64, f64, f64),
    length: f64,
    samples: Vec<Sample>,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_idx: usize,
    property_name: String,
    confidence: f64,
}

fn normalize_vector(v: (f64, f64, f64)) -> (f64, f64, f64) {
    let len = (v.0 * v.0 + v.1 * v.1 + v.2 * v.2).sqrt();
    if len > 0.0 {
        (v.0 / len, v.1 / len, v.2 / len)
    } else {
        (0.0, 0.0, 0.0)
    }
}

fn get_sample_position(hole: &Drillhole, depth: f64) -> (f64, f64, f64) {
    let dir = normalize_vector(hole.direction);
    (
        hole.start.0 + dir.0 * depth,
        hole.start.1 + dir.1 * depth,
        hole.start.2 + dir.2 * depth,
    )
}

fn distance(p1: (f64, f64, f64), p2: (f64, f64, f64)) -> f64 {
    let dx = p1.0 - p2.0;
    let dy = p1.1 - p2.1;
    let dz = p1.2 - p2.2;
    (dx * dx + dy * dy + dz * dz).sqrt()
}

fn detect_outliers_small(
    holes: &[Drillhole],
    property_names: &[String],
) -> Vec<Suspect> {
    let mut suspects = Vec::new();
    let k = 5.min((holes.len() * 10).min(50));

    for (hole_idx, hole) in holes.iter().enumerate() {
        for (sample_idx, sample) in hole.samples.iter().enumerate() {
            let sample_pos = get_sample_position(hole, sample.depth);

            for (prop_idx, &value) in sample.properties.iter().enumerate() {
                let mut neighbors = Vec::new();

                for (other_hole_idx, other_hole) in holes.iter().enumerate() {
                    for (other_sample_idx, other_sample) in other_hole.samples.iter().enumerate() {
                        if hole_idx == other_hole_idx && sample_idx == other_sample_idx {
                            continue;
                        }

                        let other_pos = get_sample_position(other_hole, other_sample.depth);
                        let dist = distance(sample_pos, other_pos);

                        if prop_idx < other_sample.properties.len() {
                            neighbors.push((
                                dist,
                                other_sample.properties[prop_idx],
                            ));
                        }
                    }
                }

                if neighbors.is_empty() {
                    continue;
                }

                neighbors.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
                let k_neighbors: Vec<f64> = neighbors.iter().take(k).map(|(_, v)| *v).collect();

                if k_neighbors.len() < 3 {
                    continue;
                }

                let mean: f64 = k_neighbors.iter().sum::<f64>() / k_neighbors.len() as f64;
                let variance: f64 = k_neighbors
                    .iter()
                    .map(|v| (v - mean).powi(2))
                    .sum::<f64>()
                    / k_neighbors.len() as f64;
                let stddev = variance.sqrt();

                if stddev > 0.0 {
                    let z_score = (value - mean).abs() / stddev;
                    if z_score > 2.5 {
                        let confidence = (z_score - 2.5) / (z_score.max(3.0) - 2.5);
                        suspects.push(Suspect {
                            hole_id: hole.id.clone(),
                            sample_idx,
                            property_idx: prop_idx,
                            property_name: property_names[prop_idx].clone(),
                            confidence: confidence.min(1.0),
                        });
                    }
                }
            }
        }
    }

    suspects
}

fn detect_outliers_large(
    holes: &[Drillhole],
    property_names: &[String],
) -> Vec<Suspect> {
    let mut suspects = Vec::new();

    for (hole_idx, hole) in holes.iter().enumerate() {
        for (sample_idx, sample) in hole.samples.iter().enumerate() {
            for (prop_idx, &value) in sample.properties.iter().enumerate() {
                let mut local_values = Vec::new();

                let depth = sample.depth;
                let search_radius = (hole.length / 20.0).max(1.0);

                for (check_sample_idx, check_sample) in hole.samples.iter().enumerate() {
                    if (check_sample.depth - depth).abs() <= search_radius {
                        local_values.push(check_sample.properties[prop_idx]);
                    }
                }

                if local_values.len() < 3 {
                    continue;
                }

                let mean: f64 = local_values.iter().sum::<f64>() / local_values.len() as f64;
                let variance: f64 = local_values
                    .iter()
                    .map(|v| (v - mean).powi(2))
                    .sum::<f64>()
                    / local_values.len() as f64;
                let stddev = variance.sqrt();

                if stddev > 1e-10 {
                    let z_score = (value - mean).abs() / stddev;
                    if z_score > 3.0 {
                        let confidence = ((z_score - 3.0) / (z_score.max(4.0) - 3.0)).min(1.0);
                        suspects.push(Suspect {
                            hole_id: hole.id.clone(),
                            sample_idx,
                            property_idx: prop_idx,
                            property_name: property_names[prop_idx].clone(),
                            confidence,
                        });
                    }
                }
            }
        }
    }

    suspects
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();

    reader.read_line(&mut line).unwrap();
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    line.clear();
    reader.read_line(&mut line).unwrap();
    let property_names: Vec<String> = line
        .trim()
        .split_whitespace()
        .take(num_properties)
        .map(|s| s.to_string())
        .collect();

    let mut holes = Vec::new();

    for _ in 0..num_holes {
        line.clear();
        reader.read_line(&mut line).unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();

        let id = parts[0].to_string();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let length: f64 = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        let mut samples = Vec::new();

        for _ in 0..num_samples {
            line.clear();
            reader.read_line(&mut line).unwrap();
            let sample_parts: Vec<&str> = line.trim().split_whitespace().collect();

            let depth: f64 = sample_parts[0].parse().unwrap();
            let properties: Vec<f64> = sample_parts[1..]
                .iter()
                .take(num_properties)
                .map(|s| s.parse().unwrap_or(0.0))
                .collect();

            samples.push(Sample { depth, properties });
        }

        holes.push(Drillhole {
            id,
            start: (start_x, start_y, start_z),
            direction: (dir_x, dir_y, dir_z),
            length,
            samples,
        });
    }

    let mut suspects = if holes.len() < 500 {
        detect_outliers_small(&holes, &property_names)
    } else {
        detect_outliers_large(&holes, &property_names)
    };

    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    println!("{}", suspects.len());
    for suspect in suspects {
        println!(
            "{} {} {} {:.6}",
            suspect.hole_id, suspect.sample_idx, suspect.property_name, suspect.confidence
        );
    }
}
