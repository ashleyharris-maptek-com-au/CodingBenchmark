#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

struct Edge {
    int to, cap, flow, rev;
};

class Dinic {
public:
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    int n;

    Dinic(int n) : n(n), graph(n), level(n), iter(n) {}

    void add_edge(int from, int to, int cap) {
        graph[from].push_back({to, cap, 0, (int)graph[to].size()});
        graph[to].push_back({from, 0, 0, (int)graph[from].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > e.flow && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t || pushed == 0) return pushed;
        for (int& cid = iter[v]; cid < (int)graph[v].size(); cid++) {
            Edge& e = graph[v][cid];
            if (level[v] + 1 != level[e.to] || e.cap <= e.flow)
                continue;

            int tr = dfs(e.to, t, min(pushed, e.cap - e.flow));
            if (tr > 0) {
                e.flow += tr;
                graph[e.to][e.rev].flow -= tr;
                return tr;
            }
        }
        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            while (int pushed = dfs(s, t, INT_MAX)) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<bool> get_source_side(int s) {
        vector<bool> reachable(n, false);
        queue<int> q;
        q.push(s);
        reachable[s] = true;

        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (auto& e : graph[u]) {
                if (e.cap > e.flow && !reachable[e.to]) {
                    reachable[e.to] = true;
                    q.push(e.to);
                }
            }
        }

        return reachable;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<pair<int,int>> edges(m);

    for (int i = 0; i < m; i++) {
        cin >> edges[i].first >> edges[i].second;
    }

    if (m == 0) {
        cout << 0 << "\n";
        return 0;
    }

    int min_cut = INT_MAX;
    vector<pair<int,int>> result_cut;

    // Try multiple vertex pairs to find minimum cut
    vector<pair<int,int>> candidates;

    // Use edge endpoints as candidates
    for (auto& edge : edges) {
        candidates.push_back(edge);
        if (candidates.size() >= min(10, (int)edges.size())) break;
    }

    for (auto& [s, t] : candidates) {
        Dinic mf(n);

        for (auto& [u, v] : edges) {
            mf.add_edge(u, v, 1);
            mf.add_edge(v, u, 1);
        }

        int flow = mf.max_flow(s, t);

        if (flow < min_cut) {
            min_cut = flow;

            vector<bool> source_side = mf.get_source_side(s);
            result_cut.clear();

            for (auto& [u, v] : edges) {
                if (source_side[u] && !source_side[v]) {
                    result_cut.push_back({u, v});
                } else if (!source_side[u] && source_side[v]) {
                    result_cut.push_back({v, u});
                }
            }
        }
    }

    cout << min_cut << "\n";
    for (auto& [u, v] : result_cut) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
