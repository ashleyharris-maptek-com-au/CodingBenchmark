OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor

OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %gl_PerVertex_out Block

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%float_0 = OpConstant %float 0.0
%float_0_04 = OpConstant %float 0.04
%float_m0_04 = OpConstant %float -0.04
%float_0_005 = OpConstant %float 0.005
%float_m0_005 = OpConstant %float -0.005

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%array_v4_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_array_v4_2 = OpTypePointer Input %array_v4_2
%gl_in = OpVariable %ptr_in_array_v4_2 Input

%array_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_array_v3_2 = OpTypePointer Input %array_v3_2
%inWorldPos = OpVariable %ptr_in_array_v3_2 Input
%inNormal = OpVariable %ptr_in_array_v3_2 Input
%inColor = OpVariable %ptr_in_array_v3_2 Input

%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%fn_void = OpTypeFunction %void

%ext = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %fn_void
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

%dir = OpFSub %v3float %p1_xyz %p0_xyz

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

%neg_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v3float %neg_y %dir_x %float_0

%perp_norm = OpExtInst %v3float %ext Normalize %perp_unnorm

%perp_start_pos = OpVectorTimesScalar %v3float %perp_norm %float_0_04
%v0_xyz = OpFAdd %v3float %p0_xyz %perp_start_pos
%v0_pos = OpVectorShuffle %v4float %v0_xyz %p0 0 1 2 6

%v0_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_out_ptr %v0_pos
OpStore %outColor %c0
OpEmitVertex

%perp_start_neg = OpVectorTimesScalar %v3float %perp_norm %float_m0_04
%v1_xyz = OpFAdd %v3float %p0_xyz %perp_start_neg
%v1_pos = OpVectorShuffle %v4float %v1_xyz %p0 0 1 2 6

OpStore %v0_out_ptr %v1_pos
OpStore %outColor %c0
OpEmitVertex

%perp_end_pos = OpVectorTimesScalar %v3float %perp_norm %float_0_005
%v2_xyz = OpFAdd %v3float %p1_xyz %perp_end_pos
%v2_pos = OpVectorShuffle %v4float %v2_xyz %p1 0 1 2 6

OpStore %v0_out_ptr %v2_pos
OpStore %outColor %c1
OpEmitVertex

%perp_end_neg = OpVectorTimesScalar %v3float %perp_norm %float_m0_005
%v3_xyz = OpFAdd %v3float %p1_xyz %perp_end_neg
%v3_pos = OpVectorShuffle %v4float %v3_xyz %p1 0 1 2 6

OpStore %v0_out_ptr %v3_pos
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
