OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor

               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0

               OpDecorate %ubo_block Block
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0
               OpMemberDecorate %ubo_block 0 Offset 0
               OpMemberDecorate %ubo_block 1 Offset 64
               OpMemberDecorate %ubo_block 2 Offset 128
               OpMemberDecorate %ubo_block 3 Offset 144
               OpMemberDecorate %ubo_block 4 Offset 160

       %void = OpTypeVoid
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %bool = OpTypeBool
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4

   %mat4float = OpTypeMatrix %v4float 4
      %uint_1 = OpConstant %int 1
      %uint_2 = OpConstant %int 2
      %uint_4 = OpConstant %int 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1

      %f_0_0 = OpConstant %float 0.0
      %f_1_0 = OpConstant %float 1.0
      %f_0_04 = OpConstant %float 0.04
      %f_0_005 = OpConstant %float 0.005

%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
  %ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
  %ptr_in_v4 = OpTypePointer Input %v4float
  %ptr_in_v3 = OpTypePointer Input %v3float

%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float

   %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
   %ptr_out_v3 = OpTypePointer Output %v3float
   %ptr_out_v4 = OpTypePointer Output %v4float

%ubo_block = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float
%ptr_uniform = OpTypePointer Uniform %ubo_block
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

%func_void = OpTypeFunction %void

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

        %gl_in = OpVariable %ptr_in_arr Input
   %worldPos_in = OpVariable %ptr_in_arr_v3 Input
    %normal_in = OpVariable %ptr_in_arr_v3 Input
     %color_in = OpVariable %ptr_in_arr_v3 Input

       %gl_out = OpVariable %ptr_out_pv Output
    %outColor = OpVariable %ptr_out_v3 Output

     %ubo_var = OpVariable %ptr_uniform Uniform

       %main = OpFunction %void None %func_void
%main_entry = OpLabel

               ; Load p0 and p1 from gl_in
      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
      %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %int_0
         %p1 = OpLoad %v4float %p1_ptr

               ; Load colors
      %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr
      %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %uint_2
         %c1 = OpLoad %v3float %c1_ptr

               ; Extract xyz from p0 and p1
     %p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
     %p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

               ; Compute line direction: p1 - p0
  %line_dir = OpFSub %v3float %p1_xyz %p0_xyz

               ; Compute perpendicular in xy plane (z=0)
               ; For 2D perpendicular: if dir=(x,y,z), perp=(-y,x,0) normalized
     %dir_x = OpCompositeExtract %float %line_dir 0
     %dir_y = OpCompositeExtract %float %line_dir 1
     %dir_z = OpCompositeExtract %float %line_dir 2

    %perp_x = OpFNegate %float %dir_y
    %perp_y = OpCompositeConstruct %v3float %perp_x %dir_x %f_0_0
               ; Normalize perpendicular vector
   %perp_len = OpExtInst %float %GLSL450 Length %perp_y
  %perp_norm = OpFDiv %v3float %perp_y %perp_len

               ; Scale by thickness at p0 and p1
%perp_thick0 = OpVectorTimesScalar %v3float %perp_norm %f_0_04
%perp_thick1 = OpVectorTimesScalar %v3float %perp_norm %f_0_005

               ; Compute the 4 vertices of the ribbon
  %v0_pos = OpFAdd %v3float %p0_xyz %perp_thick0
  %v1_pos = OpFSub %v3float %p0_xyz %perp_thick0
  %v2_pos = OpFAdd %v3float %p1_xyz %perp_thick1
  %v3_pos = OpFSub %v3float %p1_xyz %perp_thick1

               ; Convert back to homogeneous coords
  %v0_4 = OpCompositeConstruct %v4float %v0_pos %f_1_0
  %v1_4 = OpCompositeConstruct %v4float %v1_pos %f_1_0
  %v2_4 = OpCompositeConstruct %v4float %v2_pos %f_1_0
  %v3_4 = OpCompositeConstruct %v4float %v3_pos %f_1_0

               ; Emit vertex 0: p0+perp*thick, color c0
   %out0_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out0_pos %v0_4
   %out0_col = %outColor
               OpStore %out0_col %c0
               OpEmitVertex

               ; Emit vertex 1: p0-perp*thick, color c0
   %out1_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out1_pos %v1_4
   %out1_col = %outColor
               OpStore %out1_col %c0
               OpEmitVertex

               ; Emit vertex 2: p1+perp*thick_end, color c1
   %out2_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out2_pos %v2_4
   %out2_col = %outColor
               OpStore %out2_col %c1
               OpEmitVertex

               ; Emit vertex 3: p1-perp*thick_end, color c1
   %out3_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out3_pos %v3_4
   %out3_col = %outColor
               OpStore %out3_col %c1
               OpEmitVertex

               ; End the primitive
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
