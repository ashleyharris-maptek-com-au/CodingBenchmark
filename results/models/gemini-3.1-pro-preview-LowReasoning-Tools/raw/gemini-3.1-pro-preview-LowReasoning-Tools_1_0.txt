using namespace std;

class FastIO {
    static const int BUF_SIZE = 1 << 24;
    char buf[BUF_SIZE];
    int pos = 0, len = 0;
    FILE* f;
public:
    FastIO(FILE* f) : f(f) {
#ifdef _WIN32
        _setmode(_fileno(f), _O_BINARY);
#endif
    }
    inline char nextChar() {
        if (pos == len) {
            pos = 0;
            len = (int)fread(buf, 1, BUF_SIZE, f);
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }
    
    inline bool readInt(int &x) {
        char c = nextChar();
        while (c <= ' ' && c != EOF) c = nextChar();
        if (c == EOF) return false;
        x = 0;
        bool sign = false;
        if (c == '-') { sign = true; c = nextChar(); }
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = nextChar();
        }
        if (sign) x = -x;
        return true;
    }
    
    inline bool readFloat(float &x) {
        char c = nextChar();
        while (c <= ' ' && c != EOF) c = nextChar();
        if (c == EOF) return false;
        bool sign = false;
        if (c == '-') { sign = true; c = nextChar(); }
        double res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = nextChar();
        }
        if (c == '.') {
            c = nextChar();
            double frac = 1;
            while (c >= '0' && c <= '9') {
                frac *= 0.1;
                res += (c - '0') * frac;
                c = nextChar();
            }
        }
        x = (float)(sign ? -res : res);
        return true;
    }
};

class FastOut {
    static const int BUF_SIZE = 1 << 24;
    char buf[BUF_SIZE];
    int pos = 0;
    FILE* f;
public:
    FastOut(FILE* f) : f(f) {
#ifdef _WIN32
        _setmode(_fileno(f), _O_BINARY);
#endif
    }
    ~FastOut() { flush(); }
    inline void flush() {
        if (pos > 0) {
            fwrite(buf, 1, pos, f);
            pos = 0;
        }
    }
    inline void writeInt(int x) {
        if (x == 0) {
            writeChar('0');
            return;
        }
        if (x < 0) {
            writeChar('-');
            x = -x;
        }
        char temp[12];
        int idx = 0;
        while (x > 0) {
            temp[idx++] = (x % 10) + '0';
            x /= 10;
        }
        for (int i = idx - 1; i >= 0; i--) {
            writeChar(temp[i]);
        }
    }
    inline void writeChar(char c) {
        if (pos == BUF_SIZE) flush();
        buf[pos++] = c;
    }
};

inline uint64_t hilbert_d(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int i = 31; i >= 0; i--) {
        uint32_t rx = (x >> i) & 1;
        uint32_t ry = (y >> i) & 1;
        d = (d << 2) | ((rx * 3) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

int main() {
    FastIO fio(stdin);
    int N;
    if (!fio.readInt(N)) return 0;

    vector<float> X(N), Y(N);
    float min_x = 1e9, max_x = -1e9;
    float min_y = 1e9, max_y = -1e9;

    for (int i = 0; i < N; i++) {
        fio.readFloat(X[i]);
        fio.readFloat(Y[i]);
        if (X[i] < min_x) min_x = X[i];
        if (X[i] > max_x) max_x = X[i];
        if (Y[i] < min_y) min_y = Y[i];
        if (Y[i] > max_y) max_y = Y[i];
    }

    float range_x = max_x - min_x;
    float range_y = max_y - min_y;
    float range = max(range_x, range_y);
    if (range == 0) range = 1;

    vector<uint64_t> H(N);
    for (int i = 0; i < N; i++) {
        uint32_t nx = (uint32_t)((X[i] - min_x) / range * 4294967295.0);
        uint32_t ny = (uint32_t)((Y[i] - min_y) / range * 4294967295.0);
        H[i] = hilbert_d(nx, ny);
    }

    vector<int> p(N);
    iota(p.begin(), p.end(), 0);
    sort(p.begin(), p.end(), [&](int a, int b) {
        return H[a] < H[b];
    });

    auto dist = [&](int a, int b) {
        float dx = X[a] - X[b];
        float dy = Y[a] - Y[b];
        return sqrtf(dx * dx + dy * dy);
    };

    if (N <= 1000) {
        vector<int> best_p = p;
        float best_cost = 1e18f;
        
        auto calc_cost = [&](const vector<int>& tour) {
            float cost = 0;
            for (int i = 0; i < N; i++) {
                cost += dist(tour[i], tour[(i+1)%N]);
            }
            return cost;
        };

        int starts = (N <= 40) ? 2000 : (N <= 200 ? 200 : 20);
        std::mt19937 rng(1337);
        
        for (int s = 0; s < starts; s++) {
            vector<int> cur_p = p;
            if (s > 0) {
                if (s <= N) {
                    cur_p.clear();
                    vector<bool> vis(N, false);
                    int curr = s - 1;
                    cur_p.push_back(curr);
                    vis[curr] = true;
                    for (int i = 1; i < N; i++) {
                        int best_next = -1;
                        float min_d = 1e18f;
                        for (int j = 0; j < N; j++) {
                            if (!vis[j]) {
                                float d = dist(curr, j);
                                if (d < min_d) {
                                    min_d = d;
                                    best_next = j;
                                }
                            }
                        }
                        curr = best_next;
                        cur_p.push_back(curr);
                        vis[curr] = true;
                    }
                } else {
                    shuffle(cur_p.begin(), cur_p.end(), rng);
                }
            }
            
            bool improved = true;
            while (improved) {
                improved = false;
                for (int i = 0; i < N - 2; i++) {
                    for (int j = i + 2; j < N; j++) {
                        int a = cur_p[i], b = cur_p[i+1];
                        int c = cur_p[j], d = (j + 1 < N) ? cur_p[j+1] : cur_p[0];
                        if (dist(a, c) + dist(b, d) < dist(a, b) + dist(c, d) - 1e-5f) {
                            reverse(cur_p.begin() + i + 1, cur_p.begin() + j + 1);
                            improved = true;
                        }
                    }
                }
            }
            
            float c_cost = calc_cost(cur_p);
            if (c_cost < best_cost) {
                best_cost = c_cost;
                best_p = cur_p;
            }
        }
        p = best_p;
    } else {
        int window = 0;
        int max_iters = 0;
        if (N <= 10000) { window = 500; max_iters = 20; }
        else if (N <= 50000) { window = 200; max_iters = 10; }
        else if (N <= 200000) { window = 50; max_iters = 5; }
        else if (N <= 1000000) { window = 15; max_iters = 2; }
        else { window = 0; max_iters = 0; } 

        auto time_start = chrono::steady_clock::now();
        for (int iter = 0; iter < max_iters; iter++) {
            bool improved = false;
            for (int i = 0; i < N - 2; i++) {
                int max_j = min(N - 1, i + window);
                for (int j = i + 2; j <= max_j; j++) {
                    int a = p[i], b = p[i+1];
                    int c = p[j], d = (j + 1 < N) ? p[j+1] : p[0];
                    float d1 = dist(a, b) + dist(c, d);
                    float d2 = dist(a, c) + dist(b, d);
                    if (d2 < d1 - 1e-5f) {
                        reverse(p.begin() + i + 1, p.begin() + j + 1);
                        improved = true;
                    }
                }
            }
            auto time_now = chrono::steady_clock::now();
            double elapsed = chrono::duration<double>(time_now - time_start).count();
            if (elapsed > 280.0) break;
            if (!improved) break;
        }
    }

    auto it = find(p.begin(), p.end(), 0);
    if (it != p.begin()) {
        rotate(p.begin(), it, p.end());
    }

    FastOut fout(stdout);
    for (int i = 0; i < N; i++) {
        if (i > 0) fout.writeChar(' ');
        fout.writeInt(p[i]);
    }
    fout.writeChar('\n');

    return 0;
}
