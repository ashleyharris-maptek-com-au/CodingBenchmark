using System;
using System.Collections.Generic;
using System.Diagnostics;

class Program
{
    struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double d) => new Vector3(a.X * d, a.Y * d, a.Z * d);
        public static Vector3 operator /(Vector3 a, double d) => new Vector3(a.X / d, a.Y / d, a.Z / d);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    struct Orbit
    {
        public int Id;
        public Vector3 R, V;
        public Vector3 H, E, N;
        public double A;
        public double Vc;
    }

    static Orbit ParseOrbit(int id, string line, double mu)
    {
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        var r = new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2]));
        var v = new Vector3(double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5]));
        
        var h = Vector3.Cross(r, v);
        var e = Vector3.Cross(v, h) / mu - r / r.Length();
        double rMag = r.Length();
        double vMag = v.Length();
        double energy = vMag * vMag / 2 - mu / rMag;
        double a = -mu / (2 * energy);
        double vc = Math.Sqrt(mu / Math.Abs(a));

        double hLen = h.Length();
        return new Orbit { Id = id, R = r, V = v, H = h, E = e, N = hLen > 0 ? h / hLen : new Vector3(0,0,1), A = a, Vc = vc };
    }

    static double GetCost(Orbit o1, Orbit o2, double mu)
    {
        double dotN = Math.Max(-1, Math.Min(1, Vector3.Dot(o1.N, o2.N)));
        double delta_i = Math.Acos(dotN);
        double dv_plane = 2 * Math.Min(o1.Vc, o2.Vc) * Math.Sin(delta_i / 2.0);
        
        double dv_hohmann = 0;
        if (o1.A > 0 && o2.A > 0)
        {
            dv_hohmann = Math.Abs(Math.Sqrt(mu / o1.A) * (Math.Sqrt(2 * o2.A / (o1.A + o2.A)) - 1)) +
                         Math.Abs(Math.Sqrt(mu / o2.A) * (1 - Math.Sqrt(2 * o1.A / (o1.A + o2.A))));
        }
        else
        {
            dv_hohmann = Math.Abs(o1.Vc - o2.Vc);
        }
        
        double dv_ecc = (o1.Vc + o2.Vc) / 2.0 * (o1.E - o2.E).Length();
        
        return dv_plane + dv_hohmann + dv_ecc;
    }

    static void Reverse(int[] arr, int i, int j)
    {
        while (i < j)
        {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }
    }

    static double CalcCost(int[] tour, double[,] costMatrix)
    {
        double cost = 0;
        for (int i = 0; i < tour.Length - 1; i++)
        {
            cost += costMatrix[tour[i], tour[i + 1]];
        }
        return cost;
    }

    static void Main()
    {
        System.Threading.Thread.CurrentThread.CurrentCulture = System.Globalization.CultureInfo.InvariantCulture;
        string firstLine = null;
        while (string.IsNullOrWhiteSpace(firstLine = Console.ReadLine())) { if (firstLine == null) return; }
        
        var parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        double mu = double.Parse(parts[1]);

        if (N < 1) return;
        if (N == 1)
        {
            Console.WriteLine("0");
            return;
        }

        string startLine = null;
        while (string.IsNullOrWhiteSpace(startLine = Console.ReadLine())) { if (startLine == null) return; }
        Orbit startOrbit = ParseOrbit(-1, startLine, mu);

        Orbit[] stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            string line = null;
            while (string.IsNullOrWhiteSpace(line = Console.ReadLine())) { if (line == null) break; }
            stations[i] = ParseOrbit(i, line, mu);
        }

        double[,] costMatrix = new double[N + 1, N + 1];
        for (int i = 0; i < N; i++)
        {
            costMatrix[0, i + 1] = GetCost(startOrbit, stations[i], mu);
            costMatrix[i + 1, 0] = double.MaxValue;
            for (int j = 0; j < N; j++)
            {
                if (i == j) costMatrix[i + 1, j + 1] = 0;
                else costMatrix[i + 1, j + 1] = GetCost(stations[i], stations[j], mu);
            }
        }
        
        int[] currentTour = new int[N + 1];
        bool[] visited = new bool[N + 1];
        currentTour[0] = 0;
        visited[0] = true;
        
        for (int i = 1; i <= N; i++)
        {
            int lastNode = currentTour[i - 1];
            double minCost = double.MaxValue;
            int bestNext = -1;
            for (int j = 1; j <= N; j++)
            {
                if (!visited[j] && costMatrix[lastNode, j] < minCost)
                {
                    minCost = costMatrix[lastNode, j];
                    bestNext = j;
                }
            }
            currentTour[i] = bestNext;
            visited[bestNext] = true;
        }

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 1; i < N; i++)
            {
                for (int j = i + 1; j <= N; j++)
                {
                    double oldEdges = costMatrix[currentTour[i - 1], currentTour[i]];
                    if (j < N) oldEdges += costMatrix[currentTour[j], currentTour[j + 1]];
                    
                    double newEdges = costMatrix[currentTour[i - 1], currentTour[j]];
                    if (j < N) newEdges += costMatrix[currentTour[i], currentTour[j + 1]];
                    
                    if (newEdges < oldEdges - 1e-6)
                    {
                        Reverse(currentTour, i, j);
                        improved = true;
                    }
                }
            }
        }

        int[] globalBestTour = (int[])currentTour.Clone();
        double globalBestCost = CalcCost(globalBestTour, costMatrix);

        Random rnd = new Random(42);
        int numRestarts = 10;
        long maxMillis = 5000 / numRestarts; 
        
        for (int restart = 0; restart < numRestarts; restart++)
        {
            if (restart > 0)
            {
                for (int i = 1; i <= N; i++)
                {
                    int swapIdx = rnd.Next(1, N + 1);
                    int temp = currentTour[i];
                    currentTour[i] = currentTour[swapIdx];
                    currentTour[swapIdx] = temp;
                }
                
                bool imp = true;
                while (imp)
                {
                    imp = false;
                    for (int i = 1; i < N; i++)
                    {
                        for (int j = i + 1; j <= N; j++)
                        {
                            double oldE = costMatrix[currentTour[i - 1], currentTour[i]];
                            if (j < N) oldE += costMatrix[currentTour[j], currentTour[j + 1]];
                            
                            double newE = costMatrix[currentTour[i - 1], currentTour[j]];
                            if (j < N) newE += costMatrix[currentTour[i], currentTour[j + 1]];
                            
                            if (newE < oldE - 1e-6)
                            {
                                Reverse(currentTour, i, j);
                                imp = true;
                            }
                        }
                    }
                }
            }
            
            int[] workingTour = (int[])currentTour.Clone();
            double currentCost = CalcCost(workingTour, costMatrix);
            
            double T_start = currentCost * 0.1;
            if (T_start < 1) T_start = 100;
            double T_end = 1e-4;
            
            Stopwatch sw = Stopwatch.StartNew();
            int iter = 0;
            while (sw.ElapsedMilliseconds < maxMillis)
            {
                iter++;
                if (iter % 1000 == 0) currentCost = CalcCost(workingTour, costMatrix);
                
                int i = rnd.Next(1, N);
                int j = rnd.Next(i + 1, N + 1);
                
                double oldEdges = costMatrix[workingTour[i - 1], workingTour[i]];
                if (j < N) oldEdges += costMatrix[workingTour[j], workingTour[j + 1]];
                
                double newEdges = costMatrix[workingTour[i - 1], workingTour[j]];
                if (j < N) newEdges += costMatrix[workingTour[i], workingTour[j + 1]];
                
                double delta = newEdges - oldEdges;
                
                double fraction = (double)sw.ElapsedMilliseconds / maxMillis;
                if (fraction > 1.0) fraction = 1.0;
                double temp = T_start * Math.Pow(T_end / T_start, fraction);
                
                if (delta < 0 || Math.Exp(-delta / temp) > rnd.NextDouble())
                {
                    Reverse(workingTour, i, j);
                    currentCost += delta;
                    
                    if (currentCost < globalBestCost)
                    {
                        globalBestCost = currentCost;
                        Array.Copy(workingTour, globalBestTour, N + 1);
                    }
                }
            }
        }
        
        var resultIndices = new List<string>();
        for (int i = 1; i <= N; i++)
        {
            resultIndices.Add((globalBestTour[i] - 1).ToString());
        }
        Console.WriteLine(string.Join(" ", resultIndices));
    }
}
