using System;
using System.IO;
using System.Collections.Generic;

class Program
{
    class Edge
    {
        public int Id;
        public int To;
        public long Weight;
    }

    static List<(int u, int v, long w)> all_edges;
    static List<Edge>[] graph;

    static void DuplicateEdge(int edgeId)
    {
        var e = all_edges[edgeId];
        int newId = all_edges.Count;
        all_edges.Add(e);
        graph[e.u].Add(new Edge { Id = newId, To = e.v, Weight = e.w });
        graph[e.v].Add(new Edge { Id = newId, To = e.u, Weight = e.w });
    }

    static int GetOtherEndpoint(int edgeId, int node)
    {
        var e = all_edges[edgeId];
        return e.u == node ? e.v : e.u;
    }

    static void SolveExactDP(List<int> oddNodes, int N)
    {
        int K = oddNodes.Count;
        long[][] dist = new long[K][];
        int[][] parent_edge = new int[K][];

        for (int i = 0; i < K; i++)
        {
            int start = oddNodes[i];
            dist[i] = new long[N];
            Array.Fill(dist[i], long.MaxValue);
            parent_edge[i] = new int[N];
            Array.Fill(parent_edge[i], -1);

            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            dist[i][start] = 0;
            pq.Enqueue(start, 0);

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[i][u]) continue;

                int count = graph[u].Count;
                for (int idx = 0; idx < count; idx++)
                {
                    var edge = graph[u][idx];
                    int v = edge.To;
                    long w = edge.Weight;
                    if (dist[i][u] + w < dist[i][v])
                    {
                        dist[i][v] = dist[i][u] + w;
                        parent_edge[i][v] = edge.Id;
                        pq.Enqueue(v, dist[i][v]);
                    }
                }
            }
        }

        long[] dp = new long[1 << K];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;
        int[] dp_parent = new int[1 << K];
        Array.Fill(dp_parent, -1);
        int[] dp_pair = new int[1 << K];

        for (int mask = 0; mask < (1 << K); mask++)
        {
            if (dp[mask] >= long.MaxValue / 2) continue;

            int i = 0;
            while (i < K && ((mask >> i) & 1) == 1) i++;
            if (i == K) continue;

            for (int j = i + 1; j < K; j++)
            {
                if (((mask >> j) & 1) == 0)
                {
                    if (dist[i][oddNodes[j]] != long.MaxValue)
                    {
                        long cost = dp[mask] + dist[i][oddNodes[j]];
                        int nmask = mask | (1 << i) | (1 << j);
                        if (cost < dp[nmask])
                        {
                            dp[nmask] = cost;
                            dp_parent[nmask] = mask;
                            dp_pair[nmask] = (i << 16) | j;
                        }
                    }
                }
            }
        }

        int curr_mask = (1 << K) - 1;
        while (curr_mask > 0)
        {
            int pair = dp_pair[curr_mask];
            int a = pair >> 16;
            int b = pair & 0xFFFF;

            int curr = oddNodes[b];
            while (curr != oddNodes[a])
            {
                int e_id = parent_edge[a][curr];
                DuplicateEdge(e_id);
                curr = GetOtherEndpoint(e_id, curr);
            }

            curr_mask = dp_parent[curr_mask];
        }
    }

    static void SolveGreedy2Opt(List<int> oddNodes, int N)
    {
        int K = oddNodes.Count;
        long[][] dist = new long[K][];
        int[][] parent_edge = new int[K][];

        for (int i = 0; i < K; i++)
        {
            int start = oddNodes[i];
            dist[i] = new long[N];
            Array.Fill(dist[i], long.MaxValue);
            parent_edge[i] = new int[N];
            Array.Fill(parent_edge[i], -1);

            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            dist[i][start] = 0;
            pq.Enqueue(start, 0);

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[i][u]) continue;

                int count = graph[u].Count;
                for (int idx = 0; idx < count; idx++)
                {
                    var edge = graph[u][idx];
                    int v = edge.To;
                    long w = edge.Weight;
                    if (dist[i][u] + w < dist[i][v])
                    {
                        dist[i][v] = dist[i][u] + w;
                        parent_edge[i][v] = edge.Id;
                        pq.Enqueue(v, dist[i][v]);
                    }
                }
            }
        }

        var edgesList = new List<(long cost, int u, int v)>();
        for (int i = 0; i < K; i++)
        {
            for (int j = i + 1; j < K; j++)
            {
                if (dist[i][oddNodes[j]] != long.MaxValue)
                {
                    edgesList.Add((dist[i][oddNodes[j]], i, j));
                }
            }
        }
        edgesList.Sort((a, b) => a.cost.CompareTo(b.cost));

        bool[] matched = new bool[K];
        var matches = new List<(int, int)>();
        foreach (var e in edgesList)
        {
            if (!matched[e.u] && !matched[e.v])
            {
                matched[e.u] = true;
                matched[e.v] = true;
                matches.Add((e.u, e.v));
            }
        }

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < matches.Count; i++)
            {
                for (int j = i + 1; j < matches.Count; j++)
                {
                    int a = matches[i].Item1, b = matches[i].Item2;
                    int c = matches[j].Item1, d = matches[j].Item2;

                    long current_cost = dist[a][oddNodes[b]] + dist[c][oddNodes[d]];
                    long cost1 = dist[a][oddNodes[c]] + dist[b][oddNodes[d]];
                    long cost2 = dist[a][oddNodes[d]] + dist[b][oddNodes[c]];

                    if (cost1 < current_cost && cost1 <= cost2)
                    {
                        matches[i] = (a, c);
                        matches[j] = (b, d);
                        improved = true;
                    }
                    else if (cost2 < current_cost)
                    {
                        matches[i] = (a, d);
                        matches[j] = (b, c);
                        improved = true;
                    }
                }
            }
        }

        foreach (var match in matches)
        {
            int a = match.Item1;
            int b = match.Item2;
            int curr = oddNodes[b];
            while (curr != oddNodes[a])
            {
                int e_id = parent_edge[a][curr];
                DuplicateEdge(e_id);
                curr = GetOtherEndpoint(e_id, curr);
            }
        }
    }

    static void SolveMultiPhaseDijkstra(List<int> oddNodes, int N)
    {
        int K = oddNodes.Count;
        int unmatched_count = K;
        bool[] matched = new bool[N];

        while (unmatched_count > 0)
        {
            int initial_unmatched = unmatched_count;
            long[] dist = new long[N];
            Array.Fill(dist, long.MaxValue);
            int[] origin = new int[N];
            Array.Fill(origin, -1);
            int[] parent_edge = new int[N];
            Array.Fill(parent_edge, -1);

            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();

            for (int i = 0; i < K; i++)
            {
                int u = oddNodes[i];
                if (!matched[u])
                {
                    dist[u] = 0;
                    origin[u] = u;
                    pq.Enqueue(u, 0);
                }
            }

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[u]) continue;

                if (matched[origin[u]]) continue;

                int count = graph[u].Count;
                for (int idx = 0; idx < count; idx++)
                {
                    var edge = graph[u][idx];
                    int v = edge.To;
                    long w = edge.Weight;
                    int origU = origin[u];
                    int origV = origin[v];

                    if (origV != -1 && origV != origU)
                    {
                        if (!matched[origU] && !matched[origV])
                        {
                            matched[origU] = true;
                            matched[origV] = true;
                            unmatched_count -= 2;

                            int curr = u;
                            while (curr != origU)
                            {
                                int e_id = parent_edge[curr];
                                DuplicateEdge(e_id);
                                curr = GetOtherEndpoint(e_id, curr);
                            }

                            DuplicateEdge(edge.Id);

                            curr = v;
                            while (curr != origV)
                            {
                                int e_id = parent_edge[curr];
                                DuplicateEdge(e_id);
                                curr = GetOtherEndpoint(e_id, curr);
                            }

                            break;
                        }
                    }

                    if (dist[u] + w < dist[v])
                    {
                        dist[v] = dist[u] + w;
                        origin[v] = origU;
                        parent_edge[v] = edge.Id;
                        pq.Enqueue(v, dist[v]);
                    }
                }
            }

            if (unmatched_count == initial_unmatched) break;
        }
    }

    static List<int> FindEulerianCircuit(int N, int totalEdgesCount)
    {
        bool[] edge_used = new bool[totalEdgesCount];
        int[] edge_idx = new int[N];
        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            while (edge_idx[u] < graph[u].Count)
            {
                var edge = graph[u][edge_idx[u]];
                edge_idx[u]++;
                if (!edge_used[edge.Id])
                {
                    edge_used[edge.Id] = true;
                    stack.Push(edge.To);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                circuit.Add(stack.Pop());
            }
        }
        circuit.Reverse();
        return circuit;
    }

    public static void Main()
    {
        var scanner = new Scanner(Console.OpenStandardInput());
        int N = scanner.NextInt();
        if (N == -1) return;
        int M = scanner.NextInt();

        all_edges = new List<(int u, int v, long w)>(M);
        graph = new List<Edge>[N];
        for (int i = 0; i < N; i++) graph[i] = new List<Edge>();

        int[] degree = new int[N];

        for (int i = 0; i < M; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            long w = scanner.NextLong();
            all_edges.Add((u, v, w));
            graph[u].Add(new Edge { Id = i, To = v, Weight = w });
            graph[v].Add(new Edge { Id = i, To = u, Weight = w });
            degree[u]++;
            degree[v]++;
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0) oddNodes.Add(i);
        }

        int K = oddNodes.Count;
        if (K > 0)
        {
            if (K <= 20)
            {
                SolveExactDP(oddNodes, N);
            }
            else if (K <= 500 && (long)K * M <= 20000000L)
            {
                SolveGreedy2Opt(oddNodes, N);
            }
            else
            {
                SolveMultiPhaseDijkstra(oddNodes, N);
            }
        }

        var circuit = FindEulerianCircuit(N, all_edges.Count);

        using var sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
        for (int i = 0; i < circuit.Count; i++)
        {
            sw.Write(circuit[i]);
            if (i < circuit.Count - 1) sw.Write(" ");
        }
        sw.WriteLine();
    }

    class Scanner
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int head, tail;

        public Scanner(Stream stream)
        {
            this.stream = stream;
        }

        public long NextLong()
        {
            int c = Read();
            while (c <= 32)
            {
                if (c == -1) return -1;
                c = Read();
            }
            bool negative = false;
            if (c == '-')
            {
                negative = true;
                c = Read();
            }
            long res = 0;
            while (c > 32)
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return negative ? -res : res;
        }

        public int NextInt()
        {
            return (int)NextLong();
        }

        private int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }
}
