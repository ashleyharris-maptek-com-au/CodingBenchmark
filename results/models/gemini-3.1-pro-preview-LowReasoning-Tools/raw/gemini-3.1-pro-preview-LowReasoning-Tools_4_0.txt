#include <iostream>
#include <vector>
#include <cmath>
#include <queue>
#include <set>
#include <algorithm>
#include <tuple>
#include <iomanip>
#include <cstdlib>

using namespace std;

struct Point {
    double x, y, z;
};

Point operator-(const Point& a, const Point& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
Point operator+(const Point& a, const Point& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
Point operator*(const Point& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
Point operator/(const Point& a, double s) { return {a.x / s, a.y / s, a.z / s}; }
double dot(const Point& a, const Point& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
Point cross(const Point& a, const Point& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
double length(const Point& a) { return sqrt(dot(a, a)); }

struct Face { Point n; double d; };

struct Tet { double cx, cy, cz, rx, ry, rz; };

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if (!(cin >> V >> e)) return 0;

    vector<Point> pts(V);
    double min_x = 1e9, max_x = -1e9;
    double min_y = 1e9, max_y = -1e9;
    double min_z = 1e9, max_z = -1e9;

    for(int i=0; i<V; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
        pts[i].x += (rand() * 1.0 / RAND_MAX - 0.5) * 1e-7;
        pts[i].y += (rand() * 1.0 / RAND_MAX - 0.5) * 1e-7;
        pts[i].z += (rand() * 1.0 / RAND_MAX - 0.5) * 1e-7;

        if(pts[i].x < min_x) min_x = pts[i].x;
        if(pts[i].x > max_x) max_x = pts[i].x;
        if(pts[i].y < min_y) min_y = pts[i].y;
        if(pts[i].y > max_y) max_y = pts[i].y;
        if(pts[i].z < min_z) min_z = pts[i].z;
        if(pts[i].z > max_z) max_z = pts[i].z;
    }

    if (V < 4) {
        cout << 0 << "\n";
        return 0;
    }

    int p0 = 0;
    for(int i=1; i<V; ++i) if(pts[i].z < pts[p0].z) p0 = i;

    int p1 = -1;
    double min_sin = 2.0; 
    for(int i=0; i<V; ++i) {
        if(i == p0) continue;
        double len = length(pts[i] - pts[p0]);
        if(len < 1e-9) continue;
        double s = (pts[i].z - pts[p0].z) / len;
        if(s < min_sin) {
            min_sin = s;
            p1 = i;
        }
    }

    Point n_F = {0, 0, -1};
    Point A = pts[p0];
    Point B = pts[p1];
    Point u = B - A; u = u / length(u);

    int p2 = -1;
    double min_theta = 10.0;
    for(int i=0; i<V; ++i) {
        if(i == p0 || i == p1) continue;
        double Y = dot(pts[i] - A, n_F);
        double X = dot(pts[i] - A, cross(u, n_F));
        double theta = atan2(-Y, X);
        if(theta < min_theta) {
            min_theta = theta;
            p2 = i;
        }
    }

    vector<Face> hull;
    set<pair<int, int>> edges_done;
    queue<tuple<int, int, Point>> q;

    auto add_face = [&](int a, int b, int c) {
        Point n = cross(pts[b] - pts[a], pts[c] - pts[a]);
        double len = length(n);
        if(len < 1e-9) return;
        n = n / len;
        hull.push_back({n, dot(n, pts[a])});
        
        edges_done.insert({a, b});
        edges_done.insert({b, c});
        edges_done.insert({c, a});
        
        if (!edges_done.count({b, a})) q.push({b, a, n});
        if (!edges_done.count({c, b})) q.push({c, b, n});
        if (!edges_done.count({a, c})) q.push({a, c, n});
    };

    if (p1 != -1 && p2 != -1) {
        add_face(p1, p0, p2);
    }

    while(!q.empty()) {
        auto [a, b, nf] = q.front();
        q.pop();
        if (edges_done.count({a, b})) continue;

        Point pA = pts[a];
        Point pB = pts[b];
        Point dir = pB - pA; 
        double dirlen = length(dir);
        if(dirlen < 1e-9) continue;
        dir = dir / dirlen;

        int best_p = -1;
        double mt = 10.0;
        for(int i=0; i<V; ++i) {
            if(i == a || i == b) continue;
            double Y = dot(pts[i] - pA, nf);
            double X = dot(pts[i] - pA, cross(dir, nf));
            double theta = atan2(-Y, X);
            if(theta < mt) {
                mt = theta;
                best_p = i;
            }
        }

        if (best_p != -1) {
            add_face(b, a, best_p);
        }
    }

    double h = e * sqrt(2.0/3.0);
    double r = e / sqrt(24.0);
    double R = e * sqrt(3.0/8.0);
    
    int min_k = floor(min_z / h) - 1;
    int max_k = ceil(max_z / h) + 1;
    int min_j = floor(min_y / (e * sqrt(3.0)/2.0)) - 1;
    int max_j = ceil(max_y / (e * sqrt(3.0)/2.0)) + 1;
    int min_i = floor(min_x / e) - 1;
    int max_i = ceil(max_x / e) + 1;

    auto check_tet = [&](const Point& v0, const Point& v1, const Point& v2, const Point& v3) {
        for(const auto& f : hull) {
            if (dot(v0, f.n) > f.d + 1e-6) return false;
            if (dot(v1, f.n) > f.d + 1e-6) return false;
            if (dot(v2, f.n) > f.d + 1e-6) return false;
            if (dot(v3, f.n) > f.d + 1e-6) return false;
        }
        return true;
    };

    vector<Tet> result;

    for(int k = min_k; k <= max_k; ++k) {
        double cz = k * h;
        for(int j = min_j; j <= max_j; ++j) {
            for(int i = min_i; i <= max_i; ++i) {
                double cx_up = i * e + (j % 2 != 0 ? e/2.0 : 0.0);
                double cy_up = j * e * sqrt(3.0)/2.0;
                
                Point v0_up = {cx_up, cy_up, cz + R};
                Point v1_up = {cx_up, cy_up + e/sqrt(3.0), cz - r};
                Point v2_up = {cx_up + e/2.0, cy_up - e/(2.0*sqrt(3.0)), cz - r};
                Point v3_up = {cx_up - e/2.0, cy_up - e/(2.0*sqrt(3.0)), cz - r};

                if (check_tet(v0_up, v1_up, v2_up, v3_up)) {
                    result.push_back({cx_up, cy_up, cz, 0, 0, 0});
                }

                double cx_dn = cx_up + e/2.0;
                double cy_dn = cy_up + e/(2.0*sqrt(3.0));

                Point v0_dn = {cx_dn, cy_dn, cz + R};
                Point v1_dn = {cx_dn, cy_dn - e/sqrt(3.0), cz - r};
                Point v2_dn = {cx_dn - e/2.0, cy_dn + e/(2.0*sqrt(3.0)), cz - r};
                Point v3_dn = {cx_dn + e/2.0, cy_dn + e/(2.0*sqrt(3.0)), cz - r};

                if (check_tet(v0_dn, v1_dn, v2_dn, v3_dn)) {
                    result.push_back({cx_dn, cy_dn, cz, 0, 0, 3.14159265358979323846});
                }
            }
        }
    }

    cout << result.size() << "\n";
    cout << fixed << setprecision(6);
    for(const auto& t : result) {
        cout << t.cx << " " << t.cy << " " << t.cz << " " 
             << t.rx << " " << t.ry << " " << t.rz << "\n";
    }

    return 0;
}
