using System;
using System.Collections.Generic;
using System.Linq;
using CromulentBisgetti.ContainerPacking;
using CromulentBisgetti.ContainerPacking.Entities;

namespace BinPackingSolver
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Starting 3D Bin Packing Solver...");

            // Step 3: Define Input Data (Bins and Items)
            var containers = new List<Container>
            {
                // Example: One large container
                new Container(1, 100, 100, 100) // ID, Length, Width, Height
            };

            var items = new List<Item>
            {
                new Item(1, 20, 20, 20, 5),   // 5 items, 20x20x20
                new Item(2, 30, 30, 30, 3),   // 3 items, 30x30x30
                new Item(3, 10, 10, 50, 10),  // 10 items, 10x10x50
                new Item(4, 5, 5, 5, 20)      // 20 items, 5x5x5
            };

            // Step 4: Implement Packing Logic
            // The library expects a list of algorithms to try.
            // For this example, we'll just use EB_AFIT.
            var algorithms = new List<int> { (int)CromulentBisgetti.ContainerPacking.Enums.AlgorithmType.EB_AFIT };
            
            List<ContainerPackingResult> packingResults = PackingService.Pack(containers, items, algorithms);

            // Step 5: Process and Display Results
            Console.WriteLine("\n--- Packing Results ---");

            foreach (var containerResult in packingResults)
            {
                Console.WriteLine($"\n--- Container {containerResult.ContainerID} ---");
                // Find the original container to get its dimensions
                var originalContainer = containers.FirstOrDefault(c => c.ID == containerResult.ContainerID);
                if (originalContainer != null)
                {
                    Console.WriteLine($"  Dimensions: {originalContainer.Length}x{originalContainer.Width}x{originalContainer.Height}");
                }

                foreach (var algorithmResult in containerResult.AlgorithmPackingResults)
                {
                    Console.WriteLine($"\n  Algorithm Used: {((CromulentBisgetti.ContainerPacking.Enums.AlgorithmType)algorithmResult.AlgorithmID).ToString()}");
                    Console.WriteLine($"  Pack Time: {algorithmResult.PackTimeInMilliseconds}ms");
                    Console.WriteLine($"  Percent Volume Packed: {algorithmResult.PercentContainerVolumePacked:P}");

                    if (algorithmResult.PackedItems.Any())
                    {
                        Console.WriteLine("  Packed Items:");
                        foreach (var packedItem in algorithmResult.PackedItems)
                        {
                            Console.WriteLine($"    Item {packedItem.ID} ({packedItem.Coordinate.L}x{packedItem.Coordinate.W}x{packedItem.Coordinate.H}) at ({packedItem.StartCoordinate.X},{packedItem.StartCoordinate.Y},{packedItem.StartCoordinate.Z})");
                        }
                    }
                    else
                    {
                        Console.WriteLine("  No items packed in this container for this algorithm run.");
                    }

                    if (algorithmResult.UnpackedItems.Any())
                    {
                        Console.WriteLine("\n  Unpacked Items:");
                        foreach (var unpackedItem in algorithmResult.UnpackedItems)
                        {
                            Console.WriteLine($"    Item {unpackedItem.ID} (Original Quantity: {items.FirstOrDefault(i => i.ID == unpackedItem.ID)?.Quantity}) - {unpackedItem.Length}x{unpackedItem.Width}x{unpackedItem.Height}");
                        }
                    }
                    else
                    {
                        Console.WriteLine("\n  All items successfully packed for this algorithm run!");
                    }
                }
            }
            Console.WriteLine("\n3D Bin Packing Solver Finished.");
        }
    }
}
