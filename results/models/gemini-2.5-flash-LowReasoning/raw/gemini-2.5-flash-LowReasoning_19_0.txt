using System;
using System.IO;
using System.Globalization;
using System.Linq; // For Max
using System.Collections.Generic; // For List

public struct Voxel
{
    public byte MassInTonnes;
    public byte ValuePerTonne;
    public bool IsDug;

    public Voxel(byte mass, byte value)
    {
        MassInTonnes = mass;
        ValuePerTonne = value;
        IsDug = false;
    }

    public bool IsValuable => ValuePerTonne > 0;
    public bool IsSolid => MassInTonnes > 0 && !IsDug;
}

public class Grid
{
    public int XVoxelCount { get; private set; }
    public int YVoxelCount { get; private set; }
    public int ZVoxelCount { get; private set; }
    public decimal XBlockSize { get; private set; }
    public decimal YBlockSize { get; private set; }
    public decimal ZBlockSize { get; private set; }

    public Voxel[,,] Voxels { get; private set; }
    public int[,] GroundHeight { get; private set; } // Stores the highest Z-coordinate of a solid voxel at (x,y)

    public Grid(int xCount, int yCount, int zCount, decimal xBlock, decimal yBlock, decimal zBlock)
    {
        XVoxelCount = xCount;
        YVoxelCount = yCount;
        ZVoxelCount = zCount;
        XBlockSize = xBlock;
        YBlockSize = yBlock;
        ZBlockSize = zBlock;

        Voxels = new Voxel[xCount, yCount, zCount];
        GroundHeight = new int[xCount, yCount];
    }

    public void SetInitialVoxel(int x, int y, int z, Voxel voxel)
    {
        Voxels[x, y, z] = voxel;
    }

    public void InitializeGroundHeight()
    {
        for (int x = 0; x < XVoxelCount; x++)
        {
            for (int y = 0; y < YVoxelCount; y++)
            {
                GroundHeight[x, y] = -1; // Default to below ground

                for (int z = ZVoxelCount - 1; z >= 0; z--)
                {
                    if (Voxels[x, y, z].IsSolid)
                    {
                        GroundHeight[x, y] = z;
                        break; // Found the highest solid voxel in this column
                    }
                }
            }
        }
    }

    // Helper methods for bounds checking
    public bool IsInBounds(int x, int y)
    {
        return x >= 0 && x < XVoxelCount &&
               y >= 0 && y < YVoxelCount;
    }

    public bool IsInBounds(int x, int y, int z)
    {
        return IsInBounds(x, y) &&
               z >= 0 && z < ZVoxelCount;
    }

    public bool IsExposed(int x, int y, int z)
    {
        // A voxel is exposed if it's not dug and one of its 6 faces is exposed
        if (!IsInBounds(x, y, z) || Voxels[x, y, z].IsDug)
            return false;

        // Exposed from above
        if (z == ZVoxelCount - 1) return true; // Topmost layer
        if (IsInBounds(x, y, z + 1) && Voxels[x, y, z + 1].IsDug) return true; // Voxel above is dug

        // Exposed from sides (checking if neighbor is dug or out of bounds)
        // Check X-1
        if (!IsInBounds(x - 1, y) || Voxels[x - 1, y, z].IsDug) return true;
        // Check X+1
        if (!IsInBounds(x + 1, y) || Voxels[x + 1, y, z].IsDug) return true;
        // Check Y-1
        if (!IsInBounds(x, y - 1) || Voxels[x, y - 1, z].IsDug) return true;
        // Check Y+1
        if (!IsInBounds(x, y + 1) || Voxels[x, y + 1, z].IsDug) return true;
        
        return false;
    }

    public void DigVoxel(int x, int y, int z)
    {
        if (!IsInBounds(x, y, z) || Voxels[x, y, z].IsDug)
        {
            // Already dug or out of bounds
            return;
        }

        Voxel dugVoxel = Voxels[x, y, z];
        dugVoxel.IsDug = true;
        Voxels[x, y, z] = dugVoxel; // Update the voxel in the grid

        RecalculateGroundHeight(x, y);
    }

    public void DumpVoxel(int x, int y, Voxel voxelToDump)
    {
        if (!IsInBounds(x, y)) // Check if (x,y) is in bounds
        {
            // Cannot dump out of bounds
            return;
        }

        int targetZ = GroundHeight[x, y] + 1;

        if (targetZ >= ZVoxelCount)
        {
            // This is a critical case. If ZVoxelCount is truly a hard limit,
            // then we cannot dump here. For now, we will just ignore it.
            // A real solution might require resizing the grid or re-evaluating
            // the problem constraints.
            Console.Error.WriteLine($"Cannot dump at {x},{y}: ZVoxelCount limit reached.");
            return;
        }

        // Create a new solid voxel at the targetZ
        Voxels[x, y, targetZ] = new Voxel(voxelToDump.MassInTonnes, voxelToDump.ValuePerTonne);
        // The dumped voxel is "new" and thus not dug.
        // Also, since it's "dumped", it is solid. So MassInTonnes is not 0.
        // The IsSolid property in Voxel will correctly return true.
        
        GroundHeight[x, y] = targetZ; // Update ground height
    }

    private void RecalculateGroundHeight(int x, int y)
    {
        GroundHeight[x, y] = -1; // Reset for recalculation
        for (int z = ZVoxelCount - 1; z >= 0; z--)
        {
            if (Voxels[x, y, z].IsSolid)
            {
                GroundHeight[x, y] = z;
                break;
            }
        }
    }

    // New helper to get actual Z for pathfinding
    public int GetCurrentZ(int x, int y)
    {
        if (!IsInBounds(x,y)) return -1; // Or throw exception, depending on desired behavior
        return GroundHeight[x, y];
    }

    public Voxel GetVoxel(int x, int y, int z)
    {
        if (!IsInBounds(x,y,z))
        {
            return new Voxel(0,0); // Return an empty voxel for out-of-bounds access
        }
        return Voxels[x,y,z];
    }
}

public class MiningStep
{
    public int X { get; set; }
    public int Y { get; set; }
}

public class MiningOperation
{
    public int DigX { get; set; }
    public int DigY { get; set; }
    public int DigZ { get; set; }
    public List<MiningStep> Path { get; set; } = new List<MiningStep>();
    public int DestX { get; set; }
    public int DestY { get; set; }
}

public class MiningSolution
{
    public List<MiningOperation> Operations { get; set; } = new List<MiningOperation>();
    public long TotalProfit { get; private set; }

    public void AddOperation(MiningOperation op, Voxel dugVoxel, long fuelCost)
    {
        Operations.Add(op);

        long revenue = 0;
        if (op.DestX == 0 && op.DestY == 0) // Processing plant
        {
            if (dugVoxel.IsValuable)
            {
                revenue = dugVoxel.MassInTonnes * dugVoxel.ValuePerTonne * 1000;
            }
            else
            {
                revenue = -dugVoxel.MassInTonnes * 1000; // Damage to plant
            }
        }
        // If dumped elsewhere, no immediate revenue or cost beyond fuel, but it becomes an obstacle.

        TotalProfit += (revenue - fuelCost);
    }

    public void PrintSolution()
    {
        Console.WriteLine(Operations.Count);
        foreach (var op in Operations)
        {
            Console.Write($"{op.Path.Count} {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var step in op.Path)
            {
                Console.Write($" {step.X} {step.Y}");
            }
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }
}

// Represents a node in the A* search for pathfinding
public class PathNode
{
    public int X { get; }
    public int Y { get; }
    public int Z { get; } // Current Z-level of the ground at (X,Y)
    public long Cost { get; } // gCost - actual cost from start to this node
    public long Heuristic { get; } // hCost - estimated cost from this node to end
    public long TotalCost => Cost + Heuristic; // fCost
    public PathNode? Parent { get; } // Parent node to reconstruct path

    public PathNode(int x, int y, int z, long cost, long heuristic, PathNode? parent)
    {
        X = x;
        Y = y;
        Z = z;
        Cost = cost;
        Heuristic = heuristic;
        Parent = parent;
    }
}

public class Miner
{
    private Grid _grid;
    private MiningSolution _solution = new MiningSolution();
    private HashSet<(int x, int y, int z)> _exposedVoxels = new HashSet<(int x, int y, int z)>();

    public Miner(Grid grid)
    {
        _grid = grid;
    }

    public void PrintSolution()
    {
        _solution.PrintSolution();
    }

    public void InitializeExposedVoxels()
    {
        for (int x = 0; x < _grid.XVoxelCount; x++)
        {
            for (int y = 0; y < _grid.YVoxelCount; y++)
            {
                for (int z = 0; z < _grid.ZVoxelCount; z++)
                {
                    if (_grid.GetVoxel(x, y, z).IsSolid && _grid.IsExposed(x, y, z))
                    {
                        _exposedVoxels.Add((x, y, z));
                    }
                }
            }
        }
    }

    // Main mining loop
    public void Mine()
    {
        InitializeExposedVoxels(); // Populate the initial set of exposed voxels

        while (true)
        {
            long maxProfit = 0;
            MiningOperation? bestOperation = null;
            Voxel bestDugVoxel = new Voxel();
            long bestFuelCost = 0;

            // Iterate only over exposed voxels
            foreach (var voxelCoords in _exposedVoxels)
            {
                int x = voxelCoords.x;
                int y = voxelCoords.y;
                int z = voxelCoords.z;

                Voxel currentVoxel = _grid.GetVoxel(x, y, z);

                // Re-check IsSolid and IsExposed in case the grid state changed
                // (e.g., due to collapsing voxels, not yet implemented)
                if (!currentVoxel.IsSolid || !_grid.IsExposed(x, y, z))
                {
                    // This voxel is no longer exposed or solid, remove it and skip
                    // (This will be more efficient if we remove during updates, but safety check for now)
                    continue; 
                }

                // Try pathfinding to processing plant (0,0)
                (List<MiningStep>? path, long fuelCost) = FindPath(x, y, 0, 0, currentVoxel.MassInTonnes);

                if (path != null)
                {
                    long revenue = 0;
                    if (currentVoxel.IsValuable)
                    {
                        revenue = currentVoxel.MassInTonnes * currentVoxel.ValuePerTonne * 1000;
                    }
                    else
                    {
                        revenue = -currentVoxel.MassInTonnes * 1000; // Damage to plant
                    }

                    long netProfit = revenue - fuelCost;

                    if (netProfit > maxProfit)
                    {
                        maxProfit = netProfit;
                        bestOperation = new MiningOperation
                        {
                            DigX = x,
                            DigY = y,
                            DigZ = z,
                            Path = path,
                            DestX = 0,
                            DestY = 0
                        };
                        bestDugVoxel = currentVoxel;
                        bestFuelCost = fuelCost;
                    }
                }
            }
            
            // Cleanup any non-exposed or non-solid voxels found during iteration
            _exposedVoxels.RemoveWhere(coords => {
                Voxel v = _grid.GetVoxel(coords.x, coords.y, coords.z);
                return !v.IsSolid || !_grid.IsExposed(coords.x, coords.y, coords.z);
            });


            if (maxProfit > 0 && bestOperation != null)
            {
                // Execute the best operation
                _grid.DigVoxel(bestOperation.DigX, bestOperation.DigY, bestOperation.DigZ);
                // For now, if dumped at processing plant, it ceases to exist, no DumpVoxel call needed.
                _solution.AddOperation(bestOperation, bestDugVoxel, bestFuelCost);
                
                // Update _exposedVoxels based on the dig operation
                UpdateExposedVoxels(bestOperation.DigX, bestOperation.DigY, bestOperation.DigZ);
            }
            else
            {
                // No more profitable operations found
                break;
            }
        }
    }

    // Pathfinding function using A*
    private (List<MiningStep>? path, long fuelCost) FindPath(int startX, int startY, int endX, int endY, byte mass)
    {
        var openSet = new PriorityQueue<PathNode, long>();
        var gScore = new Dictionary<(int x, int y), long>(); // Using dictionary for gScore to easily retrieve min

        // Start node
        int startZ = _grid.GetCurrentZ(startX, startY);
        PathNode startNode = new PathNode(startX, startY, startZ, 0, CalculateHeuristic(startX, startY, endX, endY), null);
        openSet.Enqueue(startNode, startNode.TotalCost);
        gScore[(startX, startY)] = 0;

        int[] dx = { 0, 0, 1, -1 }; // Possible moves in X
        int[] dy = { 1, -1, 0, 0 }; // Possible moves in Y

        while (openSet.Count > 0)
        {
            PathNode current = openSet.Dequeue();

            // Optimization: If we already found a shorter path to 'current', ignore this one
            if (current.Cost > gScore[(current.X, current.Y)])
            {
                continue;
            }

            if (current.X == endX && current.Y == endY)
            {
                return ReconstructPath(current, mass);
            }

            for (int i = 0; i < 4; i++)
            {
                int neighborX = current.X + dx[i];
                int neighborY = current.Y + dy[i];

                if (!_grid.IsInBounds(neighborX, neighborY))
                {
                    continue;
                }

                int neighborZ = _grid.GetCurrentZ(neighborX, neighborY);

                // Check slope constraint: Max 1 Z voxel per X/Y voxel
                // Also check if neighborZ is -1 (below ground). If so, it's not a valid path.
                if (neighborZ == -1 || Math.Abs(current.Z - neighborZ) > 1)
                {
                    continue;
                }

                long movementCost = CalculateMovementCost(current.Z, neighborZ, mass);
                long tentative_gScore = current.Cost + movementCost;

                if (!gScore.ContainsKey((neighborX, neighborY)) || tentative_gScore < gScore[(neighborX, neighborY)])
                {
                    gScore[(neighborX, neighborY)] = tentative_gScore;
                    PathNode neighborNode = new PathNode(neighborX, neighborY, neighborZ, tentative_gScore, CalculateHeuristic(neighborX, neighborY, endX, endY), current);
                    openSet.Enqueue(neighborNode, neighborNode.TotalCost);
                }
            }
        }

        return (null, -1); // No path found
    }

    private long CalculateHeuristic(int x1, int y1, int x2, int y2)
    {
        // Manhattan distance heuristic
        return Math.Abs(x1 - x2) + Math.Abs(y1 - y2);
    }

    private long CalculateMovementCost(int currentZ, int neighborZ, byte mass)
    {
        long cost = 0;
        if (currentZ == neighborZ)
        {
            cost = 1; // Flat movement
        }
        else if (neighborZ > currentZ) // Uphill
        {
            cost = 5; 
        }
        // Downhill is free (cost = 0)
        
        return cost * mass;
    }

    private (List<MiningStep> path, long fuelCost) ReconstructPath(PathNode endNode, byte mass)
    {
        List<MiningStep> path = new List<MiningStep>();
        PathNode? current = endNode;
        while (current != null)
        {
            path.Insert(0, new MiningStep { X = current.X, Y = current.Y });
            current = current.Parent;
        }
        return (path, endNode.Cost);
    }

    private void UpdateExposedVoxels(int dugX, int dugY, int dugZ)
    {
        // Remove the dug voxel from the exposed set
        _exposedVoxels.Remove((dugX, dugY, dugZ));

        // Coordinates of potential neighbors
        (int x, int y, int z)[] neighborsToCheck = new (int x, int y, int z)[]
        {
            (dugX, dugY, dugZ + 1), // Above
            (dugX - 1, dugY, dugZ), // Left
            (dugX + 1, dugY, dugZ), // Right
            (dugX, dugY - 1, dugZ), // Front
            (dugX, dugY + 1, dugZ)  // Back
            // Note: We don't check below (dugZ - 1) for exposure, as per problem.
        };

        foreach (var coords in neighborsToCheck)
        {
            if (_grid.IsInBounds(coords.x, coords.y, coords.z))
            {
                if (_grid.GetVoxel(coords.x, coords.y, coords.z).IsSolid && _grid.IsExposed(coords.x, coords.y, coords.z))
                {
                    _exposedVoxels.Add(coords);
                }
            }
        }
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        // Set culture to invariant to ensure decimal parsing works consistently
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        // Read header
        string headerLine = Console.ReadLine();
        string[] headerParts = headerLine.Split(' ');

        int xVoxelCount = int.Parse(headerParts[0]);
        int yVoxelCount = int.Parse(headerParts[1]);
        int zVoxelCount = int.Parse(headerParts[2]);
        decimal xBlockSize = decimal.Parse(headerParts[3]);
        decimal yBlockSize = decimal.Parse(headerParts[4]);
        decimal zBlockSize = decimal.Parse(headerParts[5]);

        Grid grid = new Grid(xVoxelCount, yVoxelCount, zVoxelCount, xBlockSize, yBlockSize, zBlockSize);

        // Read voxel data
        using (var stdin = Console.OpenStandardInput())
        {
            for (int x = 0; x < xVoxelCount; x++)
            {
                for (int y = 0; y < yVoxelCount; y++)
                {
                    for (int z = 0; z < zVoxelCount; z++)
                    {
                        int d = stdin.ReadByte();
                        if (d == -1) // End of stream
                        {
                            Console.Error.WriteLine("Unexpected end of input stream while reading voxel data.");
                            return;
                        }

                        byte massInTonnes = (byte)(d % 16);
                        byte valuePerTonne = (byte)(d / 16);
                        grid.SetInitialVoxel(x, y, z, new Voxel(massInTonnes, valuePerTonne));
                    }
                }
            }
        }

        grid.InitializeGroundHeight(); // Initialize ground height after all voxels are set

        Console.Error.WriteLine($"Grid initialized with dimensions: {xVoxelCount}x{yVoxelCount}x{zVoxelCount}");
        
        Miner miner = new Miner(grid);
        miner.Mine(); // Start the mining process
        miner.PrintSolution(); // Print the final solution
    }
}
