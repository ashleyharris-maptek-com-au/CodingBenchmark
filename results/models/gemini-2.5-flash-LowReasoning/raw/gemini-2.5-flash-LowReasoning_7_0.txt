// Define a small epsilon for floating-point comparisons
const float EPSILON = 1e-6;

// --- Geometric Primitives ---

struct Vec3 {
    float x, y, z;

    // Default constructor
    Vec3() : x(0.0f), y(0.0f), z(0.0f) {}
    // Parameterized constructor
    Vec3(float x_in, float y_in, float z_in) : x(x_in), y(y_in), z(z_in) {}

    // Vector addition
    Vec3 operator+(const Vec3& other) const {
        return Vec3(x + other.x, y + other.y, z + other.z);
    }

    // Vector subtraction
    Vec3 operator-(const Vec3& other) const {
        return Vec3(x - other.x, y - other.y, z - other.z);
    }

    // Scalar multiplication
    Vec3 operator*(float scalar) const {
        return Vec3(x * scalar, y * scalar, z * scalar);
    }

    // Scalar division
    Vec3 operator/(float scalar) const {
        return Vec3(x / scalar, y / scalar, z / scalar);
    }

    // Dot product
    float dot(const Vec3& other) const {
        return x * other.x + y * other.y + z * other.z;
    }

    // Cross product
    Vec3 cross(const Vec3& other) const {
        return Vec3(y * other.z - z * other.y,
                    z * other.x - x * other.z,
                    x * other.y - y * other.x);
    }

    // Magnitude (length)
    float length() const {
        return std::sqrt(x * x + y * y + z * z);
    }

    // Normalized vector
    Vec3 normalize() const {
        float len = length();
        if (len > EPSILON) {
            return *this / len;
        }
        return Vec3(0.0f, 0.0f, 0.0f); // Return zero vector for zero-length vector
    }

    // Equality comparison with epsilon
    bool operator==(const Vec3& other) const {
        return std::abs(x - other.x) < EPSILON &&
               std::abs(y - other.y) < EPSILON &&
               std::abs(z - other.z) < EPSILON;
    }

    // Inequality comparison
    bool operator!=(const Vec3& other) const {
        return !(*this == other);
    }
};

struct Triangle {
    int v[3]; // Indices into the Mesh's vertex vector

    Triangle(int v1, int v2, int v3) {
        v[0] = v1;
        v[1] = v2;
        v[2] = v3;
    }
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<Triangle> faces;
};

// --- Input/Output Functions ---

Mesh read_mesh() {
    int V, F;
    std::cin >> V >> F;
    Mesh mesh;
    mesh.vertices.reserve(V);
    mesh.faces.reserve(F);

    for (int i = 0; i < V; ++i) {
        float x, y, z;
        std::cin >> x >> y >> z;
        mesh.vertices.emplace_back(x, y, z);
    }

    for (int i = 0; i < F; ++i) {
        int v1, v2, v3;
        std::cin >> v1 >> v2 >> v3;
        mesh.faces.emplace_back(v1, v2, v3);
    }
    return mesh;
}

void write_mesh(const Mesh& mesh) {
    std::cout << mesh.vertices.size() << " " << mesh.faces.size() << std::endl;
    std::cout << std::fixed << std::setprecision(6); // Set precision for float output

    for (const auto& v : mesh.vertices) {
        std::cout << v.x << " " << v.y << " " << v.z << std::endl;
    }

    for (const auto& f : mesh.faces) {
        std::cout << f.v[0] << " " << f.v[1] << " " << f.v[2] << std::endl;
    }
}
