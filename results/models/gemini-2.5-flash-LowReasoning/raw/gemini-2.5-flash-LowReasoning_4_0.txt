#include "CollisionDetector.h"
#include <limits> // For std::numeric_limits

namespace CollisionDetector {

    // Checks if a tetrahedron is entirely inside a polyhedron
    bool is_tetrahedron_inside_polyhedron(const Tetrahedron& t, const Polyhedron& p) {
        std::vector<Vector3> tet_vertices = t.get_vertices();
        for (const auto& vertex : tet_vertices) {
            if (!p.is_point_inside(vertex)) {
                return false;
            }
        }
        return true;
    }

    // Helper function for Separating Axis Theorem (SAT)
    // Projects all vertices of two objects onto an axis and checks for overlap
    bool check_separation_axis(const std::vector<Vector3>& vertices1, const std::vector<Vector3>& vertices2, const Vector3& axis) {
        if (axis.magnitude_sq() < 1e-9) { // Avoid division by zero for null axis
            return false; // Or throw an error, depending on desired behavior
        }
        
        // Normalize the axis to simplify projection
        Vector3 normalized_axis = axis.normalize();

        double min1 = std::numeric_limits<double>::max();
        double max1 = std::numeric_limits<double>::lowest();
        for (const auto& v : vertices1) {
            double projection = v.dot(normalized_axis);
            if (projection < min1) min1 = projection;
            if (projection > max1) max1 = projection;
        }

        double min2 = std::numeric_limits<double>::max();
        double max2 = std::numeric_limits<double>::lowest();
        for (const auto& v : vertices2) {
            double projection = v.dot(normalized_axis);
            if (projection < min2) min2 = projection;
            if (projection > max2) max2 = projection;
        }

        // Check for overlap
        // If max1 < min2 or max2 < min1, there is a gap and thus a separating axis.
        return (max1 >= min2 && max2 >= min1);
    }

    // Checks if two tetrahedrons are colliding using Separating Axis Theorem (SAT)
    bool are_tetrahedrons_colliding(const Tetrahedron& t1, const Tetrahedron& t2) {
        std::vector<Vector3> vertices1 = t1.get_vertices();
        std::vector<Vector3> vertices2 = t2.get_vertices();

        // Axes to check for separation:
        // 1. Face normals of t1 (4 axes)
        // 2. Face normals of t2 (4 axes)
        // 3. Cross products of one edge from t1 and one edge from t2 (6*6=36 axes)
        //    For tetrahedrons, it's 6 unique edge pairs (4 faces, 6 edges total) -> 6 axes
        //    Actually, number of unique edge combinations is 6. For example, edge (0,1) of T1 with edge (0,1) of T2, etc.

        std::vector<Vector3> axes;

        // 1. Face normals of t1
        // To get face normals, we need to know the face topology.
        // For a tetrahedron, let's define faces. This requires knowing which vertices form which face.
        // For simplicity for now, let's approximate or choose a fixed topology.
        // The local vertices method provides a fixed order.
        std::vector<Vector3> local_v1 = t1.get_local_vertices();
        std::vector<Vector3> local_v2 = t2.get_local_vertices();

        // Faces of tetrahedron (indices relative to get_vertices() output)
        // Assuming the order given in Tetrahedron::get_local_vertices()
        // V0, V1, V2, V3
        // Face 0: V0, V1, V2 (if base is downwards)
        // Face 1: V0, V1, V3
        // Face 2: V0, V2, V3
        // Face 3: V1, V2, V3
        
        // Calculate normals for t1 (transform local normals by t1.transform for world normals)
        // Normal 0: (V1-V0) x (V2-V0)
        axes.push_back((vertices1[1] - vertices1[0]).cross(vertices1[2] - vertices1[0]).normalize());
        // Normal 1: (V1-V0) x (V3-V0) -- This might be wrong, need to be careful with winding order
        axes.push_back((vertices1[1] - vertices1[0]).cross(vertices1[3] - vertices1[0]).normalize());
        axes.push_back((vertices1[2] - vertices1[0]).cross(vertices1[3] - vertices1[0]).normalize());
        axes.push_back((vertices1[2] - vertices1[1]).cross(vertices1[3] - vertices1[1]).normalize());


        // Calculate normals for t2
        axes.push_back((vertices2[1] - vertices2[0]).cross(vertices2[2] - vertices2[0]).normalize());
        axes.push_back((vertices2[1] - vertices2[0]).cross(vertices2[3] - vertices2[0]).normalize());
        axes.push_back((vertices2[2] - vertices2[0]).cross(vertices2[3] - vertices2[0]).normalize());
        axes.push_back((vertices2[2] - vertices2[1]).cross(vertices2[3] - vertices2[1]).normalize());
        
        // 2. Cross products of edges (6 edges per tetrahedron)
        // Edges for a tetrahedron: (0,1), (0,2), (0,3), (1,2), (1,3), (2,3)
        std::vector<Vector3> edges1 = {
            vertices1[1] - vertices1[0], vertices1[2] - vertices1[0], vertices1[3] - vertices1[0],
            vertices1[2] - vertices1[1], vertices1[3] - vertices1[1], vertices1[3] - vertices1[2]
        };
        std::vector<Vector3> edges2 = {
            vertices2[1] - vertices2[0], vertices2[2] - vertices2[0], vertices2[3] - vertices2[0],
            vertices2[2] - vertices2[1], vertices2[3] - vertices2[1], vertices2[3] - vertices2[2]
        };

        for (const auto& edge1 : edges1) {
            for (const auto& edge2 : edges2) {
                Vector3 axis_candidate = edge1.cross(edge2);
                if (axis_candidate.magnitude_sq() > 1e-9) { // Avoid adding null axes
                    axes.push_back(axis_candidate.normalize());
                }
            }
        }

        // Check for separation along all axes
        for (const auto& axis : axes) {
            if (!check_separation_axis(vertices1, vertices2, axis)) {
                return false; // Separating axis found, no collision
            }
        }

        return true; // No separating axis found, collision
    }

} // namespace CollisionDetector
