use std::io::{self, BufRead};
use std::collections::{HashSet, VecDeque};

#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy)]
struct Point {
    x: isize,
    y: isize,
}

struct Grid {
    width: isize,
    height: isize,
    obstacles: HashSet<Point>,
    total_cells_to_visit: usize,
}

impl Grid {
    fn new(width: isize, height: isize, obstacles: HashSet<Point>) -> Self {
        let total_cells = (width * height) as usize;
        let total_cells_to_visit = total_cells - obstacles.len();
        Grid {
            width,
            height,
            obstacles,
            total_cells_to_visit,
        }
    }

    fn is_valid(&self, p: &Point) -> bool {
        p.x >= 0 && p.x < self.width && p.y >= 0 && p.y < self.height && !self.obstacles.contains(p)
    }

    fn neighbors(&self, p: &Point) -> Vec<Point> {
        let mut n = Vec::new();
        let moves = [
            Point { x: p.x + 1, y: p.y },
            Point { x: p.x - 1, y: p.y },
            Point { x: p.x, y: p.y + 1 },
            Point { x: p.x, y: p.y - 1 },
        ];
        for m in moves.iter() {
            if self.is_valid(m) {
                n.push(*m);
            }
        }
        n
    }
}

fn solve_hamiltonian_path(
    grid: &Grid,
    start: Point,
    require_cycle: bool,
) -> Option<Vec<Point>> {
    let mut path = VecDeque::new();
    let mut visited = HashSet::new();

    path.push_back(start);
    visited.insert(start);

    if backtrack(grid, &mut path, &mut visited, require_cycle) {
        Some(path.into())
    } else {
        None
    }
}

fn backtrack(
    grid: &Grid,
    path: &mut VecDeque<Point>,
    visited: &mut HashSet<Point>,
    require_cycle: bool,
) -> bool {
    if path.len() == grid.total_cells_to_visit {
        if require_cycle {
            let start_node = Point { x: 0, y: 0 };
            let last_node = path.back().unwrap();
            let neighbors_of_last = grid.neighbors(last_node);
            neighbors_of_last.contains(&start_node)
        } else {
            true
        }
    } else {
        let current_pos = *path.back().unwrap();
        for neighbor in grid.neighbors(&current_pos) {
            if !visited.contains(&neighbor) {
                path.push_back(neighbor);
                visited.insert(neighbor);

                if backtrack(grid, path, visited, require_cycle) {
                    return true;
                }

                visited.remove(&neighbor);
                path.pop_back();
            }
        }
        false
    }
}


fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read width, height, obstacle_count, require_cycle
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<isize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2] as usize;
    let require_cycle = parts[3] == 1;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<isize> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.insert(Point {
            x: parts[0],
            y: parts[1],
        });
    }

    let grid = Grid::new(width, height, obstacles);
    let start_point = Point { x: 0, y: 0 };

    if grid.obstacles.contains(&start_point) {
        println!("0"); // Start point is an obstacle, no path possible
        return;
    }

    match solve_hamiltonian_path(&grid, start_point, require_cycle) {
        Some(path) => {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
        }
        None => {
            println!("0");
        }
    }
}
