using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public static void Main(string[] args)
    {
        // Read N and stock_length
        string[] firstLine = Console.ReadLine().Split(' ');
        int N = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        // Read cut lengths
        List<int> cutLengths = Console.ReadLine().Split(' ').Select(int.Parse).ToList();

        // Validate input
        if (cutLengths.Count != N)
        {
            Console.Error.WriteLine("Error: Number of cut lengths does not match N.");
            return;
        }

        // Store cuts with their original indices
        List<(int Length, int OriginalIndex)> cutsWithIndices = new List<(int, int)>();
        for (int i = 0; i < N; i++)
        {
            cutsWithIndices.Add((cutLengths[i], i));
        }

        // Sort cuts in decreasing order of length
        cutsWithIndices = cutsWithIndices.OrderByDescending(c => c.Length).ToList();

        List<Stock> stocks = new List<Stock>();

        foreach (var cut in cutsWithIndices)
        {
            bool placed = false;
            // Try to fit the cut into an existing stock
            foreach (var stock in stocks)
            {
                if (stock.CanFit(cut.Length))
                {
                    stock.AddCut(cut.Length, cut.OriginalIndex);
                    placed = true;
                    break;
                }
            }

            // If it doesn't fit in any existing stock, create a new one
            if (!placed)
            {
                Stock newStock = new Stock(stockLength);
                newStock.AddCut(cut.Length, cut.OriginalIndex);
                stocks.Add(newStock);
            }
        }

        // Output results
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock.GetCutOriginalIndices()));
        }
    }
}

public class Stock
{
    private int _capacity;
    private int _remainingSpace;
    private List<int> _cutOriginalIndices;

    public Stock(int capacity)
    {
        _capacity = capacity;
        _remainingSpace = capacity;
        _cutOriginalIndices = new List<int>();
    }

    public bool CanFit(int cutLength)
    {
        return cutLength <= _remainingSpace;
    }

    public void AddCut(int cutLength, int originalIndex)
    {
        if (CanFit(cutLength))
        {
            _remainingSpace -= cutLength;
            _cutOriginalIndices.Add(originalIndex);
        }
        else
        {
            throw new InvalidOperationException("Cut does not fit in this stock.");
        }
    }

    public IEnumerable<int> GetCutOriginalIndices()
    {
        return _cutOriginalIndices.OrderBy(i => i); // Output in ascending order of original index
    }
}
