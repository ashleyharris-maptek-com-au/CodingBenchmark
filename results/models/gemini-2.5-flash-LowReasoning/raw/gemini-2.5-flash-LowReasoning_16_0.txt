#include <iostream>
#include <vector>
#include <string>
#include <fstream>
#include <sstream>
#include <algorithm>
#include <limits> // Required for std::numeric_limits

#include "jobshop_types.h"

// Function to parse input from a file
bool parse_input(const std::string& filename, std::vector<Job>& jobs, int& num_machines) {
    std::ifstream infile(filename);
    if (!infile.is_open()) {
        std::cerr << "Error: Could not open input file " << filename << std::endl;
        return false;
    }

    int num_jobs_val;
    infile >> num_jobs_val >> num_machines;

    if (infile.fail()) {
        std::cerr << "Error: Failed to read num_jobs and num_machines." << std::endl;
        return false;
    }

    jobs.resize(num_jobs_val);

    std::string line;
    std::getline(infile, line); // Consume the rest of the first line

    for (int i = 0; i < num_jobs_val; ++i) {
        if (!std::getline(infile, line)) {
            std::cerr << "Error: Failed to read job data for job " << i << std::endl;
            return false;
        }
        std::stringstream ss(line);
        int machine_id, processing_time;
        int operation_counter = 0;
        Job current_job(i); // Assign job_id

        while (ss >> machine_id >> processing_time) {
            current_job.operations.emplace_back(machine_id, processing_time, operation_counter++);
        }
        jobs[i] = current_job;
    }

    infile.close();
    return true;
}

// Function to print the schedule
void print_schedule(const std::vector<Job>& scheduled_jobs, int makespan) {
    std::cout << "Job-Shop Schedule (Makespan: " << makespan << ")" << std::endl;
    for (const auto& job : scheduled_jobs) {
        std::cout << "Job " << job.job_id << ":" << std::endl;
        for (const auto& op : job.operations) {
            std::cout << "  Operation " << op.operation_id
                      << " (Machine " << op.machine_id
                      << "): Start=" << op.start_time
                      << ", End=" << op.end_time
                      << ", Duration=" << op.processing_time << std::endl;
        }
    }
}

int main() {
    std::vector<Job> jobs;
    int num_machines;
    const std::string input_filename = "input.txt";

    if (!parse_input(input_filename, jobs, num_machines)) {
        return 1;
    }

    std::cout << "Parsed " << jobs.size() << " jobs and " << num_machines << " machines." << std::endl;
    for (const auto& job : jobs) {
        std::cout << "Job " << job.job_id << " has " << job.operations.size() << " operations." << std::endl;
    }

    // Initialize machines
    std::vector<Machine> machines(num_machines);
    for (int i = 0; i < num_machines; ++i) {
        machines[i].machine_id = i;
    }

    int completed_operations_count = 0;
    int total_operations = 0;
    for (const auto& job : jobs) {
        total_operations += job.operations.size();
    }

    int makespan = 0;

    // Scheduling loop
    while (completed_operations_count < total_operations) {
        int best_job_idx = -1;
        int best_op_idx = -1;
        int min_finish_time = std::numeric_limits<int>::max();
        int potential_start_time = 0;

        // Find the next operation to schedule
        for (int i = 0; i < jobs.size(); ++i) {
            Job& current_job = jobs[i];
            if (current_job.current_operation_idx < current_job.operations.size()) {
                // This job still has unscheduled operations
                Operation& op_to_consider = current_job.operations[current_job.current_operation_idx];
                Machine& required_machine = machines[op_to_consider.machine_id];

                // Calculate the earliest possible start time for this operation
                // It must start after the job's previous operation finishes AND after the machine is free
                int current_op_start_time = std::max(current_job.earliest_start_time, required_machine.available_time);
                int current_op_finish_time = current_op_start_time + op_to_consider.processing_time;

                if (current_op_finish_time < min_finish_time) {
                    min_finish_time = current_op_finish_time;
                    best_job_idx = i;
                    best_op_idx = current_job.current_operation_idx;
                    potential_start_time = current_op_start_time;
                }
            }
        }

        if (best_job_idx == -1) {
            // This should not happen if total_operations > completed_operations_count
            // but is a safeguard against infinite loops if logic is flawed.
            std::cerr << "Error: No schedulable operation found, but not all operations completed." << std::endl;
            break;
        }

        // Schedule the chosen operation
        Job& scheduled_job = jobs[best_job_idx];
        Operation& scheduled_op = scheduled_job.operations[best_op_idx];
        Machine& used_machine = machines[scheduled_op.machine_id];

        scheduled_op.start_time = potential_start_time;
        scheduled_op.end_time = min_finish_time;

        used_machine.available_time = scheduled_op.end_time;
        scheduled_job.earliest_start_time = scheduled_op.end_time;
        scheduled_job.current_operation_idx++;
        completed_operations_count++;

        makespan = std::max(makespan, scheduled_op.end_time);
    }

    print_schedule(jobs, makespan);
    return 0;
}
