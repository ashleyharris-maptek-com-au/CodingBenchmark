#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <algorithm>

const int INF = std::numeric_limits<int>::max();

// Structure to represent an edge in the graph
struct Edge {
    int to;
    int capacity;
    int flow;
    int reverse_edge; // Index of the reverse edge in the 'to' node's adj list
};

// Adjacency list representation of the graph
std::vector<std::vector<Edge>> adj;
int N; // Number of nodes

// Function to add an edge to the graph
void add_edge(int u, int v, int capacity) {
    adj[u].push_back({v, capacity, 0, (int)adj[v].size()});
    adj[v].push_back({u, 0, 0, (int)adj[u].size() - 1}); // Residual edge for backflow
}

// BFS to find an augmenting path
bool bfs(int s, int t, std::vector<int>& parent_edge, std::vector<int>& parent_node) {
    std::fill(parent_node.begin(), parent_node.end(), -1);
    std::queue<int> q;

    q.push(s);
    parent_node[s] = s; // Mark source as visited

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (size_t i = 0; i < adj[u].size(); ++i) {
            Edge& e = adj[u][i];
            if (parent_node[e.to] == -1 && e.capacity - e.flow > 0) {
                parent_node[e.to] = u;
                parent_edge[e.to] = i; // Store the index of the edge in adj[u]
                q.push(e.to);
                if (e.to == t) {
                    return true;
                }
            }
        }
    }
    return false;
}

// Edmonds-Karp algorithm to find the maximum flow
int edmonds_karp(int s, int t) {
    int max_flow = 0;
    std::vector<int> parent_node(N);
    std::vector<int> parent_edge(N);

    while (bfs(s, t, parent_edge, parent_node)) {
        int path_flow = INF;
        // Find the minimum capacity in the augmenting path
        for (int v = t; v != s; v = parent_node[v]) {
            int u = parent_node[v];
            int edge_idx = parent_edge[v];
            path_flow = std::min(path_flow, adj[u][edge_idx].capacity - adj[u][edge_idx].flow);
        }

        // Update flow along the augmenting path
        for (int v = t; v != s; v = parent_node[v]) {
            int u = parent_node[v];
            int edge_idx = parent_edge[v];
            adj[u][edge_idx].flow += path_flow;
            int reverse_edge_idx = adj[u][edge_idx].reverse_edge;
            adj[v][reverse_edge_idx].flow -= path_flow;
        }
        max_flow += path_flow;
    }
    return max_flow;
}

// DFS to find reachable nodes from source in residual graph (for min-cut)
void dfs(int u, std::vector<bool>& visited) {
    visited[u] = true;
    for (const auto& edge : adj[u]) {
        if (!visited[edge.to] && edge.capacity - edge.flow > 0) {
            dfs(edge.to, visited);
        }
    }
}

int main() {
    // Example Usage:
    // N = number of nodes (0 to N-1)
    // s = source node
    // t = sink node
    N = 6;
    adj.resize(N);

    // Graph from example: s=0, t=5
    // Edges (u, v, capacity)
    add_edge(0, 1, 10);
    add_edge(0, 2, 10);
    add_edge(1, 2, 2);
    add_edge(1, 3, 4);
    add_edge(1, 4, 8);
    add_edge(2, 4, 9);
    add_edge(3, 5, 10);
    add_edge(4, 3, 6);
    add_edge(4, 5, 10);

    int s = 0;
    int t = 5;

    int max_flow = edmonds_karp(s, t);
    std::cout << "Max Flow: " << max_flow << std::endl;

    // Find the Min-Cut
    std::vector<bool> visited(N, false);
    dfs(s, visited);

    std::cout << "Min Cut (edges across the cut):" << std::endl;
    for (int u = 0; u < N; ++u) {
        for (const auto& edge : adj[u]) {
            if (visited[u] && !visited[edge.to] && edge.capacity > 0) {
                std::cout << u << " -> " << edge.to << " (Capacity: " << edge.capacity << ")" << std::endl;
            }
        }
    }

    return 0;
}
