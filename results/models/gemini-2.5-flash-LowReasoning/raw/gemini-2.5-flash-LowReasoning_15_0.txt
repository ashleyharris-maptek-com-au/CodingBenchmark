use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
pub struct Vector3D {
    pub x: f64,
    pub y: f64,
    pub z: f64,
}

impl Vector3D {
    pub fn new(x: f64, y: f64, z: f64) -> Self {
        Vector3D { x, y, z }
    }

    pub fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    pub fn cross(self, other: Self) -> Self {
        Vector3D::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    pub fn normalize(self) -> Self {
        let length = self.length();
        if length == 0.0 {
            self
        } else {
            Vector3D::new(self.x / length, self.y / length, self.z / length)
        }
    }

    pub fn length(self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

    pub fn scale(self, factor: f64) -> Self {
        Vector3D::new(self.x * factor, self.y * factor, self.z * factor)
    }
}

impl std::ops::Add for Vector3D {
    type Output = Self;
    fn add(self, other: Self) -> Self {
        Vector3D::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }
}

impl std::ops::Sub for Vector3D {
    type Output = Self;
    fn sub(self, other: Self) -> Self {
        Vector3D::new(self.x - other.x, self.y - other.y, self.z - other.z)
    }
}

const EPSILON: f64 = 1e-9;
const MAX_SCALE: f64 = 10.0;

fn project_point(point: Vector3D, sun_vector: Vector3D) -> Vector3D {
    // If the sun is directly above or below, or very close to it,
    // and the point is not on the z=0 plane, projection is undefined or a point directly below/above.
    // In this problem, we are projecting to z=0, so if sun_vector.z is 0, special care is needed.
    // However, the problem implies a non-zero sun_vector.z for typical shadows.
    // The formula is: x' = x - z * sx / sz, y' = y - z * sy / sz
    if sun_vector.z.abs() < EPSILON {
        // Handle cases where sun is parallel to z=0 plane.
        // If the point is on z=0, it's its own projection.
        // If not, and sun is parallel, shadow extends to infinity.
        // For this problem context, we assume sun_vector.z is not zero.
        // If it were, the projection would depend on the sign of z for points not on z=0.
        // A simpler approach for now is to just return the point itself if z=0,
        // or a very large coordinate if z!=0, but the problem implies a well-defined shadow.
        // For simplicity, let's assume sun_vector.z is never 0 based on problem context.
        // If point.z is 0, the projection is the point itself.
        if point.z.abs() < EPSILON {
            return point;
        }
        // If sun_vector.z is effectively zero and point.z is not, this case needs clarification
        // based on problem constraints or a more robust geometric handling.
        // For now, let's assume it won't happen for valid inputs.
        return Vector3D::new(f64::NAN, f64::NAN, 0.0); // Indicate an error state or unhandled case
    }

    let t = -point.z / sun_vector.z;
    Vector3D::new(
        point.x + t * sun_vector.x,
        point.y + t * sun_vector.y,
        0.0,
    )
}


// The function returns following values:
// 0 --> p, q and r are collinear
// 1 --> Clockwise
// 2 --> Counterclockwise
fn orientation(p: Vector3D, q: Vector3D, r: Vector3D) -> i32 {
    let val = (q.y - p.y) * (r.x - q.x) - (q.x - p.x) * (r.y - q.y);
    if val.abs() < EPSILON {
        0
    } else if val > 0.0 {
        1
    } else {
        2
    }
}

// Implements the Monotone Chain algorithm for 2D convex hull.
// Assumes points are already on the z=0 plane (or their z-coordinate is ignored).
fn convex_hull_2d(points: &mut [Vector3D]) -> Vec<Vector3D> {
    let n = points.len();
    if n <= 2 {
        points.to_vec()
    } else {
        // Sort points lexicographically (by x, then by y)
        points.sort_by(|a, b| {
            a.x.partial_cmp(&b.x)
                .unwrap_or_else(|| a.y.partial_cmp(&b.y).unwrap_or(std::cmp::Ordering::Equal))
        });

        let mut upper_hull = Vec::new();
        for i in 0..n {
            while upper_hull.len() >= 2
                && orientation(
                    upper_hull[upper_hull.len() - 2],
                    upper_hull[upper_hull.len() - 1],
                    points[i],
                ) != 2
            {
                upper_hull.pop();
            }
            upper_hull.push(points[i]);
        }

        let mut lower_hull = Vec::new();
        for i in (0..n).rev() {
            while lower_hull.len() >= 2
                && orientation(
                    lower_hull[lower_hull.len() - 2],
                    lower_hull[lower_hull.len() - 1],
                    points[i],
                ) != 2
            {
                lower_hull.pop();
            }
            lower_hull.push(points[i]);
        }

        // Remove the duplicate points where upper and lower hulls meet
        lower_hull.pop();
        upper_hull.pop();

        upper_hull.extend_from_slice(&lower_hull);
        upper_hull
    }
}

#[derive(Debug, Clone, Copy)
pub struct Quaternion {
    pub w: f64,
    pub x: f64,
    pub y: f64,
    pub z: f64,
}

impl Quaternion {
    pub fn new(w: f64, x: f64, y: f64, z: f64) -> Self {
        Quaternion { w, x, y, z }
    }

    pub fn identity() -> Self {
        Quaternion::new(1.0, 0.0, 0.0, 0.0)
    }

    pub fn normalize(self) -> Self {
        let magnitude = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if magnitude == 0.0 {
            self
        } else {
            Quaternion::new(
                self.w / magnitude,
                self.x / magnitude,
                self.y / magnitude,
                self.z / magnitude,
            )
        }
    }

    // Quaternion multiplication (Hamilton product)
    pub fn multiply(self, other: Self) -> Self {
        Quaternion::new(
            self.w * other.w - self.x * other.x - self.y * other.y - self.z * other.z,
            self.w * other.x + self.x * other.w + self.y * other.z - self.z * other.y,
            self.w * other.y - self.x * other.z + self.y * other.w + self.z * other.x,
            self.w * other.z + self.x * other.y - self.y * other.x + self.z * other.w,
        )
    }

    // Rotate a vector by this quaternion
    pub fn rotate_vector(self, vec: Vector3D) -> Vector3D {
        let p = Quaternion::new(0.0, vec.x, vec.y, vec.z);
        let q_conj = self.conjugate();
        let rotated_p = self.multiply(p).multiply(q_conj);
        Vector3D::new(rotated_p.x, rotated_p.y, rotated_p.z)
    }

    pub fn conjugate(self) -> Self {
        Quaternion::new(self.w, -self.x, -self.y, -self.z)
    }
}

#[derive(Debug, Clone, Copy)]
pub struct Tetrahedron {
    pub vertices: [Vector3D; 4],
}

impl Tetrahedron {
    pub fn new(scale: f64) -> Self {
        // Standard tetrahedron vertices (edge length 2)
        let mut base_vertices = [
            Vector3D::new(1.0, 0.0, -0.707),
            Vector3D::new(-1.0, 0.0, -0.707),
            Vector3D::new(0.0, 1.0, 0.707),
            Vector3D::new(0.0, -1.0, 0.707),
        ];

        // Apply scaling
        for v in &mut base_vertices {
            *v = v.scale(scale);
        }

        Tetrahedron {
            vertices: base_vertices,
        }
    }

    pub fn transform(self, position: Vector3D, rotation: Quaternion) -> Self {
        let mut transformed_vertices = [Vector3D::new(0.0, 0.0, 0.0); 4];
        for (i, &vertex) in self.vertices.iter().enumerate() {
            transformed_vertices[i] = rotation.rotate_vector(vertex) + position;
        }
        Tetrahedron {
            vertices: transformed_vertices,
        }
    }

    pub fn shadow_polygon(&self, sun_vector: Vector3D) -> Vec<Vector3D> {
        let mut projected_vertices: Vec<Vector3D> = self
            .vertices
            .iter()
            .map(|&v| project_point(v, sun_vector))
            .collect();
        convex_hull_2d(&mut projected_vertices)
    }
}

pub struct Placement {

// Helper function to check if a point is inside a triangle
fn is_point_in_triangle(p: Vector3D, t1: Vector3D, t2: Vector3D, t3: Vector3D) -> bool {
    let s = t1.y * t3.x - t1.x * t3.y + (t3.y - t1.y) * p.x + (t1.x - t3.x) * p.y;
    let a = t1.x * t2.y - t1.y * t2.x + (t1.y - t2.y) * p.x + (t2.x - t1.x) * p.y;

    if s < 0.0 != a < 0.0 && s.abs() > EPSILON && a.abs() > EPSILON {
        return false;
    }

    let b = t2.x * t3.y - t2.y * t3.x + (t2.y - t3.y) * p.x + (t3.x - t2.x) * p.y;
    if b < 0.0 != s < 0.0 && b.abs() > EPSILON && s.abs() > EPSILON {
        return false;
    }

    true
}

// Helper to check if a vertex is a convex ear
fn is_ear(
    p_prev: Vector3D,
    p_curr: Vector3D,
    p_next: Vector3D,
    remaining_vertices: &[Vector3D],
) -> bool {
    // Check if the ear tip is convex (orientation is counter-clockwise)
    if orientation(p_prev, p_curr, p_next) == 1 {
        return false; // Clockwise, not a convex ear
    }

    // Check if any other remaining vertex lies inside the triangle formed by the ear
    for &v in remaining_vertices {
        // Skip current ear vertices
        if (v.x - p_prev.x).abs() < EPSILON && (v.y - p_prev.y).abs() < EPSILON { continue; }
        if (v.x - p_curr.x).abs() < EPSILON && (v.y - p_curr.y).abs() < EPSILON { continue; }
        if (v.x - p_next.x).abs() < EPSILON && (v.y - p_next.y).abs() < EPSILON { continue; }

        if is_point_in_triangle(v, p_prev, p_curr, p_next) {
            return false; // Point inside, not an ear
        }
    }
    true
}

// Basic ear-clipping triangulation for simple polygons
// Assumes a simple polygon (no self-intersections) and counter-clockwise winding
fn triangulate_polygon(polygon_vertices: &[Vector3D]) -> Vec<(Vector3D, Vector3D, Vector3D)> {
    let n = polygon_vertices.len();
    if n < 3 {
        return Vec::new();
    }
    if n == 3 {
        return vec![(
            polygon_vertices[0],
            polygon_vertices[1],
            polygon_vertices[2],
        )];
    }

    let mut triangles = Vec::new();
    let mut indices: Vec<usize> = (0..n).collect(); // Store indices to original vertices

    while indices.len() > 3 {
        let mut ear_found = false;
        for i in 0..indices.len() {
            let prev_idx = (i + indices.len() - 1) % indices.len();
            let curr_idx = i;
            let next_idx = (i + 1) % indices.len();

            let p_prev = polygon_vertices[indices[prev_idx]];
            let p_curr = polygon_vertices[indices[curr_idx]];
            let p_next = polygon_vertices[indices[next_idx]];

            let mut remaining_poly_vertices = Vec::new();
            for j in 0..indices.len() {
                if j != prev_idx && j != curr_idx && j != next_idx {
                    remaining_poly_vertices.push(polygon_vertices[indices[j]]);
                }
            }

            if is_ear(p_prev, p_curr, p_next, &remaining_poly_vertices) {
                triangles.push((p_prev, p_curr, p_next));
                indices.remove(curr_idx);
                ear_found = true;
                break;
            }
        }
        // If no ear is found, something is wrong with the polygon (e.g., not simple or wrong winding)
        // For robustness, might need more advanced triangulation or error handling
        if !ear_found {
            eprintln!("Warning: No ear found during triangulation. Polygon might be complex or malformed.");
            break;
        }
    }

    // Add the last remaining triangle
    if indices.len() == 3 {
        triangles.push((
            polygon_vertices[indices[0]],
            polygon_vertices[indices[1]],
            polygon_vertices[indices[2]],
        ));
    }

    triangles
}
