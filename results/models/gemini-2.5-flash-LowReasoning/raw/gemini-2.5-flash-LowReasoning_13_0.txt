#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

std::string longestCommonSubstring(const std::string& s1, const std::string& s2) {
    int m = s1.length();
    int n = s2.length();
    
    std::vector<std::vector<int>> dp(m + 1, std::vector<int>(n + 1, 0));
    
    int maxLength = 0;
    int endIndex = 0;
    
    for (int i = 1; i <= m; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
                if (dp[i][j] > maxLength) {
                    maxLength = dp[i][j];
                    endIndex = i - 1;
                }
            } else {
                dp[i][j] = 0;
            }
        }
    }
    
    if (maxLength == 0) {
        return "";
    }
    
    return s1.substr(endIndex - maxLength + 1, maxLength);
}

int main() {
    std::string s1 = "abcdaf";
    std::string s2 = "zbcdf";
    
    std::cout << "String 1: " << s1 << std::endl;
    std::cout << "String 2: " << s2 << std::endl;
    std::cout << "Longest Common Substring: " << longestCommonSubstring(s1, s2) << std::endl;
    
    s1 = "abcdef";
    s2 = "abcde";
    std::cout << "String 1: " << s1 << std::endl;
    std::cout << "String 2: " << s2 << std::endl;
    std::cout << "Longest Common Substring: " << longestCommonSubstring(s1, s2) << std::endl;

    s1 = "programming";
    s2 = "coding";
    std::cout << "String 1: " << s1 << std::endl;
    std::cout << "String 2: " << s2 << std::endl;
    std::cout << "Longest Common Substring: " << longestCommonSubstring(s1, s2) << std::endl;

    s1 = "apple";
    s2 = "orange";
    std::cout << "String 1: " << s1 << std::endl;
    std::cout << "String 2: " << s2 << std::endl;
    std::cout << "Longest Common Substring: " << longestCommonSubstring(s1, s2) << std::endl;

    return 0;
}
