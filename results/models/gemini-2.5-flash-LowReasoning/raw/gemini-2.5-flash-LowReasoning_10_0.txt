#include "rectangle_packer.h"

// --- Bin Class Implementation ---

Bin::Bin(int w, int h) : width(w), height(h) {
    // Initialize with one large free rectangle representing the entire bin space
    freeRectangles.push_back({-1, w, h, 0, 0, false});
}

bool Bin::placeRectangle(Rectangle& rect) {
    int bestFreeRectIndex = findBestFreeRectangle(rect);

    if (bestFreeRectIndex != -1) {
        Rectangle& freeRect = freeRectangles[bestFreeRectIndex];

        // Place the rectangle
        rect.x = freeRect.x;
        rect.y = freeRect.y;
        rect.packed = true;
        packedRectangles.push_back(rect);

        // Update free rectangles by splitting the chosen freeRect
        splitFreeRectangle(bestFreeRectIndex, rect);

        // After splitting, merge overlapping/adjacent free rectangles
        mergeFreeRectangles();
        return true;
    }
    return false;
}

int Bin::findBestFreeRectangle(const Rectangle& rect) {
    // Implement Best Fit Decreasing for area heuristic
    // Find the free rectangle that the item fits into and that leaves the smallest area left over
    int bestIndex = -1;
    int minAreaLeft = -1;

    for (size_t i = 0; i < freeRectangles.size(); ++i) {
        const Rectangle& freeRect = freeRectangles[i];

        // Check if the rectangle fits into the free space
        if (rect.width <= freeRect.width && rect.height <= freeRect.height) {
            int areaLeft = freeRect.width * freeRect.height - rect.width * rect.height;
            if (bestIndex == -1 || areaLeft < minAreaLeft) {
                minAreaLeft = areaLeft;
                bestIndex = i;
            }
        }
    }
    return bestIndex;
}

void Bin::splitFreeRectangle(int freeRectIndex, const Rectangle& placedRect) {
    Rectangle originalFreeRect = freeRectangles[freeRectIndex];
    freeRectangles.erase(freeRectangles.begin() + freeRectIndex);

    // Create remaining free spaces. There can be up to two new free rectangles.
    // 1. Space to the right of the placed rectangle
    if (originalFreeRect.x + originalFreeRect.width > placedRect.x + placedRect.width) {
        freeRectangles.push_back({
            -1,
            originalFreeRect.width - placedRect.width,
            placedRect.height,
            placedRect.x + placedRect.width,
            placedRect.y,
            false
        });
    }

    // 2. Space above the placed rectangle
    if (originalFreeRect.y + originalFreeRect.height > placedRect.y + placedRect.height) {
        freeRectangles.push_back({
            -1,
            originalFreeRect.width, // This free space has the full width of the original free rect
            originalFreeRect.height - placedRect.height,
            placedRect.x, // Starts at the same X as the placed rect
            placedRect.y + placedRect.height,
            false
        });
    }

    // Handle the remaining space from the original free rectangle if the placed rectangle
    // did not cover its full width or height. This can be complex to do optimally.
    // For a simpler first approach, we'll assume a vertical/horizontal split that leaves
    // two new free rectangles, as implemented above.
    // A more robust algorithm would create up to 4 potential new free spaces from the original
    // free rectangle, effectively carving out the placed rectangle.

    // A common approach for maximal rectangles is to generate two new rectangles:
    // 1. The part of the original free rect to the right of the placed rect.
    // 2. The part of the original free rect above the placed rect.

    // This is a basic split. A more advanced algorithm would also consider splitting
    // the original freeRect into two new rectangles that share a border with the placed rectangle.
    // For example, if placedRect is (px, py, pw, ph) within originalFreeRect (fx, fy, fw, fh):
    // New free rect 1: (fx, fy, fw, py - fy) - if there's space below placedRect
    // New free rect 2: (fx, py, px - fx, ph) - if there's space to the left
    // New free rect 3: (px + pw, py, fx + fw - (px + pw), ph) - space to the right (implemented above)
    // New free rect 4: (fx, py + ph, fw, fy + fh - (py + ph)) - space above (implemented above)

    // To keep it simpler for now and match the plan's implicit splitting, we focus on space to the right and above.
    // However, the current implementation (Option 1 and Option 2) is splitting the *original free rect* in a way
    // that assumes the placed rect starts at its bottom-left. This isn't always true for BFD.

    // Let's refine the splitting logic. When a rect is placed, it carves out a section.
    // The remaining space can be seen as either:
    // 1. One rectangle to the right: (placedRect.x + placedRect.width, placedRect.y, originalFreeRect.x + originalFreeRect.width - (placedRect.x + placedRect.width), placedRect.height)
    // 2. One rectangle above: (placedRect.x, placedRect.y + placedRect.height, placedRect.width, originalFreeRect.y + originalFreeRect.height - (placedRect.y + placedRect.height))
    // AND
    // Remaining space from the original free rectangle:
    // This is where it gets tricky to split without overlaps.

    // Let's try a different splitting strategy, standard for maximal rectangles:
    // When a rectangle R is placed in a free rectangle F:
    // 1. New free rectangle F1: Portion of F to the right of R.
    // 2. New free rectangle F2: Portion of F above R.
    // This implies that the placedRect is exactly at the bottom-left of the originalFreeRect.
    // Since findBestFreeRectangle finds *any* best fit, it might not be bottom-left.

    // The current `findBestFreeRectangle` is simple "Best Area Fit".
    // Let's refine the splitting to generate all potential new free rectangles that are maximal.

    // Reverting to the simpler two-way split from the plan, but with correction for coordinates.
    // If rect was placed at (rect.x, rect.y) within originalFreeRect (fx, fy, fw, fh)
    // New free rectangle (horizontal split): (rect.x + rect.width, rect.y, fx + fw - (rect.x + rect.width), rect.height)
    // New free rectangle (vertical split): (rect.x, rect.y + rect.height, rect.width, fy + fh - (rect.y + rect.height))
    // This is incorrect. The remaining parts should cover the *entire* unused portion of the `originalFreeRect`.

    // The standard way to split a free rectangle F by a placed rectangle P:
    // F is defined by (fx, fy, fw, fh) and P by (px, py, pw, ph).
    // The new free spaces are derived from F by "cutting" P out.
    // This generates at most 4 new rectangles, but usually 2 or 3.

    // Let's simplify for now to focus on the overall structure, and then refine.
    // For a simple Best-Fit heuristic, we place the rectangle at the bottom-left of the chosen free space.
    // This means rect.x = freeRect.x and rect.y = freeRect.y.
    // My current `placeRectangle` already sets `rect.x = freeRect.x` and `rect.y = freeRect.y`.
    // So the previous split logic (Option 1 and Option 2) was nearly correct,
    // assuming `placedRect.x = originalFreeRect.x` and `placedRect.y = originalFreeRect.y`.

    // Refined split (assuming placedRect is at the bottom-left of originalFreeRect):
    // 1. Remainder to the right:
    if (originalFreeRect.width > placedRect.width) {
        freeRectangles.push_back({
            -1,
            originalFreeRect.width - placedRect.width,
            originalFreeRect.height, // This part extends to the full height of the original free rect
            originalFreeRect.x + placedRect.width,
            originalFreeRect.y,
            false
        });
    }

    // 2. Remainder above:
    if (originalFreeRect.height > placedRect.height) {
        freeRectangles.push_back({
            -1,
            placedRect.width, // This part has the width of the placed rect
            originalFreeRect.height - placedRect.height,
            originalFreeRect.x,
            originalFreeRect.y + placedRect.height,
            false
        });
    }
    // Note: If both remaining spaces exist, they will overlap.
    // The mergeFreeRectangles function is crucial to resolve this.
}

void Bin::mergeFreeRectangles() {
    // A simple merge: remove any free rectangles that are fully contained within another.
    // This is not a full "maximal rectangles" merge, but it helps.
    std::vector<Rectangle> newFreeRectangles;
    for (size_t i = 0; i < freeRectangles.size(); ++i) {
        bool contained = false;
        for (size_t j = 0; j < freeRectangles.size(); ++j) {
            if (i == j) continue;
            if (freeRectangles[j].contains(freeRectangles[i])) {
                contained = true;
                break;
            }
        }
        if (!contained) {
            newFreeRectangles.push_back(freeRectangles[i]);
        }
    }
    freeRectangles = newFreeRectangles;

    // Further merging: check for adjacent rectangles that can form a larger one.
    // This is more complex and often involves sorting and iterating.
    // For now, the "contained" check is a good start.
}

// --- Packer Class Implementation ---

Packer::Packer(int binW, int binH, const std::vector<Rectangle>& rects)
    : binWidth(binW), binHeight(binH), inputRectangles(rects) {
    // Assign IDs to input rectangles if not already present
    for (size_t i = 0; i < inputRectangles.size(); ++i) {
        inputRectangles[i].id = i;
        inputRectangles[i].packed = false;
    }
}

void Packer::pack() {
    // Sort input rectangles (e.g., by height descending, then width descending)
    // This helps in achieving better packing density with greedy algorithms.
    std::sort(inputRectangles.begin(), inputRectangles.end(), [](const Rectangle& a, const Rectangle& b) {
        if (a.height != b.height) {
            return a.height > b.height;
        }
        return a.width > b.width;
    });

    // Initialize the first bin
    bins.emplace_back(binWidth, binHeight);

    for (Rectangle& rect : inputRectangles) {
        if (rect.width > binWidth || rect.height > binHeight) {
            std::cerr << "Warning: Rectangle " << rect.id << " (" << rect.width << "x" << rect.height << ") is larger than bin (" << binWidth << "x" << binHeight << ") and cannot be packed." << std::endl;
            rect.packed = false; // Mark as not packed
            continue; // Skip to the next rectangle
        }

        bool placed = false;
        // Try to place in existing bins
        for (Bin& bin : bins) {
            if (bin.placeRectangle(rect)) {
                placed = true;
                break;
            }
        }

        // If not placed in any existing bin, create a new bin and try again
        if (!placed) {
            bins.emplace_back(binWidth, binHeight);
            if (!bins.back().placeRectangle(rect)) {
                // This case should ideally only happen if a rectangle is too big for an empty bin,
                // which was already checked above.
                std::cerr << "Error: Rectangle " << rect.id << " could not be placed even in a new bin. This indicates an issue in logic." << std::endl;
                // For robustness, if it truly can't be placed, mark it as such.
                rect.packed = false;
            }
        }
    }
}
