use serde::Deserialize;
use std::error::Error;
use csv;

#[derive(Deserialize, Debug)] // Added Debug for easy printing
struct AssayRecord {
    sample_id: String,
    gold: f64,
    silver: f64,
}

impl AssayRecord {
    fn validate(&self) -> Result<(), String> {
        if self.gold < 0.0 {
            return Err(format!("Gold value for sample {} is negative: {}", self.sample_id, self.gold));
        }
        if self.silver < 0.0 {
            return Err(format!("Silver value for sample {} is negative: {}", self.sample_id, self.silver));
        }
        if self.gold > 1000.0 {
            return Err(format!("Gold value for sample {} is unexpectedly high: {}", self.sample_id, self.gold));
        }
        if self.silver > 10000.0 {
            return Err(format!("Silver value for sample {} is unexpectedly high: {}", self.sample_id, self.silver));
        }
        Ok(())
    }
}

fn read_assay_data(file_path: &str) -> Result<Vec<AssayRecord>, Box<dyn Error>> {
    let mut rdr = csv::Reader::from_path(file_path)?;
    let mut records = Vec::new();
    for result in rdr.deserialize() {
        let record: AssayRecord = result?;
        records.push(record);
    }
    Ok(records)
}

fn main() {
    println!("Hello, world!");
    if let Err(err) = run() {
        eprintln!("{}", err);
        std::process::exit(1);
    }
}

fn run() -> Result<(), Box<dyn Error>> {
    let records = read_assay_data("data.csv")?;
    for record in records {
        match record.validate() {
            Ok(_) => println!("Sample {}: Validation successful.", record.sample_id),
            Err(e) => eprintln!("Sample {}: Validation error: {}", record.sample_id, e),
        }
    }
    Ok(())
}
