#include <algorithm>
#include <cstdint>
#include <iostream>
#include <vector>

using namespace std;

struct Rect {
    int w, h;
    int idx;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Shelf {
    int y = 0;
    int height = 0;
    int used = 0;
};

struct Board {
    int usedHeight = 0;
    vector<Shelf> shelves;
};

static inline bool chooseOrientation(int rw, int rh, int bw, int bh, int& ow, int& oh, int& rot) {
    bool a = (rw <= bw && rh <= bh);
    bool b = (rh <= bw && rw <= bh);
    if (!a && !b) return false;
    if (a && !b) {
        ow = rw; oh = rh; rot = 0; return true;
    }
    if (!a && b) {
        ow = rh; oh = rw; rot = 1; return true;
    }
    // Both fit: prefer the one with smaller height, then smaller width.
    if (rh < rw || (rh == rw && rh <= bh && rw <= bw)) {
        ow = rw; oh = rh; rot = 0;
    } else {
        ow = rh; oh = rw; rot = 1;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    vector<Placement> ans(N);

    // Sort by descending max side, then area. This helps the shelf heuristic.
    vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        int am = max(rects[a].w, rects[a].h);
        int bm = max(rects[b].w, rects[b].h);
        if (am != bm) return am > bm;
        long long aa = 1LL * rects[a].w * rects[a].h;
        long long ab = 1LL * rects[b].w * rects[b].h;
        if (aa != ab) return aa > ab;
        return rects[a].idx < rects[b].idx;
    });

    vector<Board> boards;
    boards.reserve(max(1, N / 8));
    boards.push_back(Board{});

    for (int id : order) {
        const Rect& r = rects[id];

        int bestW = 0, bestH = 0, bestRot = 0;
        if (!chooseOrientation(r.w, r.h, BW, BH, bestW, bestH, bestRot)) {
            // Problem guarantees feasibility in the benchmark; keep a fallback.
            cout << 0 << '\n';
            return 0;
        }

        Board& b = boards.back();
        bool placed = false;

        if (!b.shelves.empty()) {
            Shelf& s = b.shelves.back();
            // Try to fit into the current shelf with either orientation.
            auto tryPlace = [&](int w, int h, int rot) -> bool {
                if (h <= s.height && s.used + w <= BW) {
                    ans[r.idx] = {int(boards.size() - 1), s.used, s.y, rot};
                    s.used += w;
                    placed = true;
                    return true;
                }
                return false;
            };

            // Prefer the orientation chosen by chooseOrientation, then the other one if feasible.
            if (bestRot == 0) {
                tryPlace(r.w, r.h, 0) || tryPlace(r.h, r.w, 1);
            } else {
                tryPlace(r.h, r.w, 1) || tryPlace(r.w, r.h, 0);
            }
        }

        if (!placed) {
            // Start a new shelf in the current board if it fits vertically.
            int w1 = r.w, h1 = r.h, rot1 = 0;
            int w2 = r.h, h2 = r.w, rot2 = 1;
            bool ok1 = (w1 <= BW && h1 <= BH);
            bool ok2 = (w2 <= BW && h2 <= BH);
            if (!ok1 && !ok2) {
                cout << 0 << '\n';
                return 0;
            }
            if (!ok1 || (ok2 && (h2 < h1 || (h2 == h1 && w2 < w1)))) {
                swap(w1, w2);
                swap(h1, h2);
                swap(rot1, rot2);
            }

            if (b.usedHeight + h1 > BH) {
                boards.push_back(Board{});
                Board& nb = boards.back();
                nb.usedHeight = 0;
                nb.shelves.push_back(Shelf{0, h1, w1});
                ans[r.idx] = {int(boards.size() - 1), 0, 0, rot1};
            } else {
                b.shelves.push_back(Shelf{b.usedHeight, h1, w1});
                ans[r.idx] = {int(boards.size() - 1), 0, b.usedHeight, rot1};
                b.usedHeight += h1;
            }
        }
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }
    return 0;
}
