use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn add(self, o: Pt) -> Pt {
        Pt::new(self.x + o.x, self.y + o.y)
    }
    fn sub(self, o: Pt) -> Pt {
        Pt::new(self.x - o.x, self.y - o.y)
    }
    fn dot(self, o: Pt) -> f64 {
        self.x * o.x + self.y * o.y
    }
    fn cross(self, o: Pt) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

fn area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.cross(b);
    }
    s * 0.5
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let eps = 1e-9;
    p.x >= a.x.min(b.x) - eps
        && p.x <= a.x.max(b.x) + eps
        && p.y >= a.y.min(b.y) - eps
        && p.y <= a.y.max(b.y) + eps
        && orient(a, b, p).abs() <= eps
}

fn seg_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let eps = 1e-9;
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() <= eps && on_segment(a, b, c) {
        return true;
    }
    if o2.abs() <= eps && on_segment(a, b, d) {
        return true;
    }
    if o3.abs() <= eps && on_segment(c, d, a) {
        return true;
    }
    if o4.abs() <= eps && on_segment(c, d, b) {
        return true;
    }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if on_segment(a, b, p) {
            return true;
        }
        let crosses = (a.y > p.y) != (b.y > p.y);
        if crosses {
            let x = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x >= p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn poly_inside_stock(poly: &[Pt], stock: &[Pt]) -> bool {
    for &p in poly {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    for i in 0..poly.len() {
        let a1 = poly[i];
        let a2 = poly[(i + 1) % poly.len()];
        for j in 0..stock.len() {
            let b1 = stock[j];
            let b2 = stock[(j + 1) % stock.len()];
            if seg_intersect(a1, a2, b1, b2) {
                if !on_segment(a1, a2, b1)
                    && !on_segment(a1, a2, b2)
                    && !on_segment(b1, b2, a1)
                    && !on_segment(b1, b2, a2)
                {
                    return false;
                }
            }
        }
    }
    true
}

fn rotate_poly(poly: &[Pt], ang: f64) -> Vec<Pt> {
    let c = ang.cos();
    let s = ang.sin();
    poly.iter()
        .map(|&p| Pt::new(p.x * c - p.y * s, p.x * s + p.y * c))
        .collect()
}

#[derive(Clone)]
struct RotVariant {
    ang: f64,
    poly: Vec<Pt>,
    w: f64,
    h: f64,
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    variants: Vec<RotVariant>,
    score: f64,
}

#[derive(Clone, Copy)]
struct Shelf {
    x: f64,
    y: f64,
    row_h: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    let (sminx, sminy, smaxx, smaxy) = bbox(&stock);

    let angles = [0.0, PI * 0.5, PI * 0.25, -PI * 0.25, PI * 0.75];
    let mut pieces = Vec::with_capacity(np);
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt::new(x, y));
        }
        let a = area(&poly).abs();
        let mut variants = Vec::new();
        for &ang in &angles {
            let rp = rotate_poly(&poly, ang);
            let (minx, miny, maxx, maxy) = bbox(&rp);
            variants.push(RotVariant {
                ang,
                poly: rp,
                w: maxx - minx,
                h: maxy - miny,
            });
        }
        let score = a.max(1e-9);
        pieces.push(Piece { idx, variants, score });
    }

    pieces.sort_by(|a, b| {
        b.score
            .partial_cmp(&a.score)
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let mut shelves: Vec<Shelf> = Vec::new();
    let mut placements: Vec<Option<(usize, f64, f64, f64)>> = vec![None; np];

    for piece in pieces {
        let mut placed = false;
        'stock_loop: for stock_idx in 0..shelves.len() {
            let mut shelf = shelves[stock_idx];
            for var in &piece.variants {
                let mut y = shelf.y;
                let mut x0 = shelf.x;
                let mut row_h = shelf.row_h;
                loop {
                    if y + var.h > smaxy + 1e-9 {
                        break;
                    }
                    if x0 + var.w > smaxx + 1e-9 {
                        y += row_h.max(1.0);
                        x0 = sminx;
                        row_h = 0.0;
                        continue;
                    }
                    let step = (var.w.min(var.h) / 3.0).clamp(1.0, 8.0);
                    let mut x = x0;
                    while x + var.w <= smaxx + 1e-9 {
                        let placed_poly: Vec<Pt> = var
                            .poly
                            .iter()
                            .map(|&p| Pt::new(p.x + x, p.y + y))
                            .collect();
                        if poly_inside_stock(&placed_poly, &stock) {
                            placements[piece.idx] = Some((stock_idx, x, y, var.ang));
                            shelf.x = x + var.w;
                            shelf.y = y;
                            shelf.row_h = row_h.max(var.h);
                            shelves[stock_idx] = shelf;
                            placed = true;
                            break 'stock_loop;
                        }
                        x += step;
                    }
                    y += row_h.max(1.0);
                    x0 = sminx;
                    row_h = 0.0;
                }
            }
        }

        if !placed {
            let mut best: Option<(usize, f64, f64, f64, Shelf)> = None;
            let stock_idx = shelves.len();
            let mut shelf = Shelf {
                x: sminx,
                y: sminy,
                row_h: 0.0,
            };
            'variant_loop: for var in &piece.variants {
                let mut y = sminy;
                let mut x0 = sminx;
                let mut row_h = 0.0;
                loop {
                    if y + var.h > smaxy + 1e-9 {
                        break;
                    }
                    if x0 + var.w > smaxx + 1e-9 {
                        y += row_h.max(1.0);
                        x0 = sminx;
                        row_h = 0.0;
                        continue;
                    }
                    let step = (var.w.min(var.h) / 3.0).clamp(1.0, 8.0);
                    let mut x = x0;
                    while x + var.w <= smaxx + 1e-9 {
                        let placed_poly: Vec<Pt> = var
                            .poly
                            .iter()
                            .map(|&p| Pt::new(p.x + x, p.y + y))
                            .collect();
                        if poly_inside_stock(&placed_poly, &stock) {
                            shelf = Shelf {
                                x: x + var.w,
                                y,
                                row_h: var.h,
                            };
                            best = Some((stock_idx, x, y, var.ang, shelf));
                            break 'variant_loop;
                        }
                        x += step;
                    }
                    y += row_h.max(1.0);
                    x0 = sminx;
                    row_h = 0.0;
                }
            }
            if let Some((si, x, y, ang, sh)) = best {
                if si == shelves.len() {
                    shelves.push(sh);
                }
                placements[piece.idx] = Some((si, x, y, ang));
            } else {
                let si = shelves.len();
                shelves.push(Shelf {
                    x: sminx,
                    y: sminy,
                    row_h: 0.0,
                });
                let var = &piece.variants[0];
                placements[piece.idx] = Some((si, sminx, sminy, var.ang));
            }
        }
    }

    println!("{}", shelves.len().max(1));
    for i in 0..np {
        let (si, x, y, ang) = placements[i].unwrap_or((0, 0.0, 0.0, 0.0));
        println!("{si} {x:.10} {y:.10} {ang:.10}");
    }
}
