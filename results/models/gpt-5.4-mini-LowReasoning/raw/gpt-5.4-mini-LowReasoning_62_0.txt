declare const require: any;
const fs = require('fs');

type Town = { id: string | number; x: number; y: number; value: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid as string[];
const ignitions: number[][] = data.ignitions as number[][];
const towns: Town[] = data.towns as Town[];
const budget: number = data.cutBudget | 0;

const n = width * height;
const blocked = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    blocked[y * width + x] = row.charCodeAt(x) === 126 ? 1 : 0; // ~
  }
}

const dist = new Int32Array(n);
dist.fill(-1);
const parent = new Int32Array(n);
parent.fill(-1);

const q = new Int32Array(n);
let qh = 0, qt = 0;

for (const [sx, sy] of ignitions) {
  if (sx < 0 || sx >= width || sy < 0 || sy >= height) continue;
  const s = sy * width + sx;
  if (blocked[s] || dist[s] !== -1) continue;
  dist[s] = 0;
  parent[s] = s;
  q[qt++] = s;
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];
while (qh < qt) {
  const v = q[qh++];
  const x = v % width;
  const y = (v / width) | 0;
  const nd = dist[v] + 1;
  for (let k = 0; k < 4; k++) {
    const nx = x + dx[k];
    const ny = y + dy[k];
    if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
    const u = ny * width + nx;
    if (blocked[u] || dist[u] !== -1) continue;
    dist[u] = nd;
    parent[u] = v;
    q[qt++] = u;
  }
}

const degree = new Int32Array(n);
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const v = y * width + x;
    if (blocked[v]) continue;
    let d = 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dx[k];
      const ny = y + dy[k];
      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
      if (!blocked[ny * width + nx]) d++;
    }
    degree[v] = d;
  }
}

type Candidate = { idx: number; score: number };
const scores = new Float64Array(n);
const threatened: Town[] = [];

for (const town of towns) {
  if (town.x < 0 || town.x >= width || town.y < 0 || town.y >= height) continue;
  const t = town.y * width + town.x;
  if (blocked[t] || dist[t] === -1) continue;
  threatened.push(town);

  const path: number[] = [];
  let cur = t;
  let guard = 0;
  while (true) {
    path.push(cur);
    const p = parent[cur];
    if (p === cur || p === -1) break;
    cur = p;
    if (++guard > n) break;
  }

  const pathLen = path.length;
  for (let i = 1; i < pathLen; i++) {
    const cell = path[i];
    const dFromTown = pathLen - 1 - i;
    const bottleneck = Math.max(1, 5 - degree[cell]);
    const weight = town.value * bottleneck / (dFromTown + 1);
    scores[cell] += weight;
  }

  const tx = town.x;
  const ty = town.y;
  for (let k = 0; k < 4; k++) {
    const nx = tx + dx[k];
    const ny = ty + dy[k];
    if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
    const u = ny * width + nx;
    if (blocked[u]) continue;
    scores[u] += town.value * 2.0;
  }
}

const candidates: Candidate[] = [];
for (let i = 0; i < n; i++) {
  if (blocked[i]) continue;
  if (scores[i] > 0) candidates.push({ idx: i, score: scores[i] });
}

candidates.sort((a, b) => b.score - a.score || a.idx - b.idx);

const chosen: number[] = [];
const seen = new Uint8Array(n);
for (const c of candidates) {
  if (chosen.length >= budget) break;
  if (seen[c.idx]) continue;
  chosen.push(c.idx);
  seen[c.idx] = 1;
}

// If we still have budget, add high-degree cells near threatened routes to form
// a slightly thicker barrier around the selected chokepoints.
if (chosen.length < budget) {
  const extra: Candidate[] = [];
  const inChosen = new Uint8Array(n);
  for (const idx of chosen) inChosen[idx] = 1;
  for (const c of candidates) {
    if (inChosen[c.idx]) continue;
    const x = c.idx % width;
    const y = (c.idx / width) | 0;
    let adj = 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dx[k];
      const ny = y + dy[k];
      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
      const u = ny * width + nx;
      if (!blocked[u]) adj++;
    }
    extra.push({ idx: c.idx, score: c.score + adj * 0.1 });
  }
  extra.sort((a, b) => b.score - a.score || a.idx - b.idx);
  for (const c of extra) {
    if (chosen.length >= budget) break;
    if (seen[c.idx]) continue;
    chosen.push(c.idx);
    seen[c.idx] = 1;
  }
}

const firebreaks = chosen.map((idx) => [idx % width, (idx / width) | 0]);
process.stdout.write(JSON.stringify({ firebreaks }));
