const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

const H = grid.length;
const W = H ? grid[0].length : 0;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function cellValue(x, y) {
  if (!inBounds(x, y)) return -1;
  return grid[y][x];
}

function key(x, y) {
  return y * W + x;
}

// Build connected components of non-obstacle cells so drones do not chase
// unreachable targets.
const comp = Array.from({ length: H }, () => Array(W).fill(-1));
const comps = [];
const dirs = [
  [0, 1],
  [1, 0],
  [0, -1],
  [-1, 0],
];

for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    if (grid[y][x] === -1 || comp[y][x] !== -1) continue;
    const cid = comps.length;
    const cells = [];
    const stack = [[x, y]];
    comp[y][x] = cid;
    while (stack.length) {
      const [cx, cy] = stack.pop();
      cells.push([cx, cy]);
      for (const [dx, dy] of dirs) {
        const nx = cx + dx, ny = cy + dy;
        if (!inBounds(nx, ny) || grid[ny][nx] === -1 || comp[ny][nx] !== -1) continue;
        comp[ny][nx] = cid;
        stack.push([nx, ny]);
      }
    }
    comps.push(cells);
  }
}

const droneStates = drones.map((d, idx) => {
  const x = d.x | 0;
  const y = d.y | 0;
  return {
    id: d.id,
    idx,
    x,
    y,
    comp: inBounds(x, y) ? comp[y][x] : -1,
    path: [[x, y]],
  };
});

// Precompute all positive cells sorted by value, then by proximity to each drone
// during selection.
const positiveCells = [];
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    const v = grid[y][x];
    if (v > 0) positiveCells.push({ x, y, v, k: key(x, y), comp: comp[y][x] });
  }
}
positiveCells.sort((a, b) => b.v - a.v);

const visited = new Set();
for (const d of droneStates) {
  if (inBounds(d.x, d.y) && grid[d.y][d.x] > 0) visited.add(key(d.x, d.y));
}

function bfsNextStep(startX, startY, targetX, targetY) {
  if (startX === targetX && startY === targetY) return [startX, startY];
  const prev = Array.from({ length: H }, () => Array(W).fill(null));
  const qx = new Int32Array(W * H);
  const qy = new Int32Array(W * H);
  let qh = 0, qt = 0;
  qx[qt] = startX; qy[qt] = startY; qt++;
  prev[startY][startX] = [-2, -2];
  while (qh < qt) {
    const x = qx[qh];
    const y = qy[qh];
    qh++;
    for (const [dx, dy] of dirs) {
      const nx = x + dx, ny = y + dy;
      if (!inBounds(nx, ny) || grid[ny][nx] === -1 || prev[ny][nx] !== null) continue;
      prev[ny][nx] = [x, y];
      if (nx === targetX && ny === targetY) {
        let cx = nx, cy = ny;
        while (true) {
          const p = prev[cy][cx];
          if (p[0] === startX && p[1] === startY) return [cx, cy];
          cx = p[0]; cy = p[1];
        }
      }
      qx[qt] = nx; qy[qt] = ny; qt++;
    }
  }
  return [startX, startY];
}

function bfsDistances(sx, sy) {
  const dist = Array.from({ length: H }, () => Array(W).fill(-1));
  if (!inBounds(sx, sy) || grid[sy][sx] === -1) return dist;
  const qx = new Int32Array(W * H);
  const qy = new Int32Array(W * H);
  let qh = 0, qt = 0;
  qx[qt] = sx; qy[qt] = sy; qt++;
  dist[sy][sx] = 0;
  while (qh < qt) {
    const x = qx[qh];
    const y = qy[qh];
    qh++;
    const nd = dist[y][x] + 1;
    for (const [dx, dy] of dirs) {
      const nx = x + dx, ny = y + dy;
      if (!inBounds(nx, ny) || grid[ny][nx] === -1 || dist[ny][nx] !== -1) continue;
      dist[ny][nx] = nd;
      qx[qt] = nx; qy[qt] = ny; qt++;
    }
  }
  return dist;
}

function bestTargetForDrone(d, dist) {
  let best = null;
  let bestScore = -Infinity;
  for (const c of positiveCells) {
    if (visited.has(c.k)) continue;
    if (d.comp !== -1 && c.comp !== d.comp) continue;
    const dd = dist[c.y][c.x];
    if (dd < 0) continue;
    const score = c.v * 1000 - dd * 7;
    if (score > bestScore) {
      bestScore = score;
      best = c;
    }
  }
  return best;
}

for (let step = 0; step < maxSteps; step++) {
  // Reserve destinations to avoid all drones collapsing onto the same target.
  const claimed = new Set();
  const plans = [];
  const dists = droneStates.map(d => bfsDistances(d.x, d.y));
  for (const d of droneStates) {
    const target = bestTargetForDrone(d, dists[d.idx]);
    plans.push(target);
    if (target) claimed.add(target.k);
  }

  let anyMoved = false;
  for (let i = 0; i < droneStates.length; i++) {
    const d = droneStates[i];
    const target = plans[i];
    let nx = d.x, ny = d.y;

    if (target) {
      const [mx, my] = bfsNextStep(d.x, d.y, target.x, target.y);
      nx = mx; ny = my;
      if (claimed.has(key(nx, ny)) && !(nx === d.x && ny === d.y)) {
        // If two drones would converge on the same exact next cell, let the
        // lower-index drone keep it and keep the other drone in place.
        let conflict = false;
        for (let j = 0; j < i; j++) {
          const od = droneStates[j];
          const last = od.path[od.path.length - 1];
          if (last[0] === nx && last[1] === ny) {
            conflict = true;
            break;
          }
        }
        if (conflict) {
          nx = d.x; ny = d.y;
        }
      }
    }

    if (!(nx === d.x && ny === d.y)) anyMoved = true;
    d.x = nx; d.y = ny;
    d.path.push([nx, ny]);
    if (inBounds(nx, ny) && grid[ny][nx] > 0) visited.add(key(nx, ny));
  }

}

// Ensure every path has exactly maxSteps + 1 positions.
for (const d of droneStates) {
  while (d.path.length < maxSteps + 1) d.path.push([d.x, d.y]);
  if (d.path.length > maxSteps + 1) d.path.length = maxSteps + 1;
}

const output = {
  paths: droneStates.map(d => d.path),
};

process.stdout.write(JSON.stringify(output));
