#include <array>

#include <cstdint>

using namespace std;

struct Graph {
    int n = 0, m = 0, k = 0;
    vector<vector<int>> adj;
    vector<int> deg;
};

static uint64_t now_seed() {
    return chrono::high_resolution_clock::now().time_since_epoch().count();
}

static bool is_valid_coloring(const Graph& g, const vector<int>& color) {
    for (int u = 0; u < g.n; ++u) {
        if (color[u] < 0 || color[u] >= g.k) return false;
        for (int v : g.adj[u]) {
            if (u < v && color[u] == color[v]) return false;
        }
    }
    return true;
}

static vector<int> greedy_dsatur(const Graph& g, mt19937_64& rng, bool randomize_ties) {
    vector<int> color(g.n, -1);
    vector<int> sat(g.n, 0);
    vector<unordered_set<int>> used(g.n);
    vector<char> colored(g.n, 0);
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);

    vector<int> deg_order = order;
    sort(deg_order.begin(), deg_order.end(), [&](int a, int b) {
        if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
        return a < b;
    });

    auto pick_vertex = [&]() -> int {
        int best = -1;
        int best_sat = -1;
        int best_deg = -1;
        for (int v : order) {
            if (colored[v]) continue;
            int sv = (int)used[v].size();
            if (sv > best_sat || (sv == best_sat && g.deg[v] > best_deg)) {
                best = v;
                best_sat = sv;
                best_deg = g.deg[v];
            } else if (randomize_ties && sv == best_sat && g.deg[v] == best_deg && best != -1) {
                if ((rng() & 1ULL) == 0) best = v;
            }
        }
        return best;
    };

    for (int step = 0; step < g.n; ++step) {
        int v = pick_vertex();
        if (v == -1) break;

        vector<char> forbidden(g.k, 0);
        for (int u : g.adj[v]) {
            if (color[u] != -1 && color[u] < g.k) forbidden[color[u]] = 1;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (!forbidden[c]) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            return {};
        }
        color[v] = chosen;
        colored[v] = 1;
        for (int u : g.adj[v]) {
            if (!colored[u] && used[u].insert(chosen).second) {
                sat[u] = (int)used[u].size();
            }
        }
    }
    return color;
}

static bool exact_backtrack(const Graph& g, vector<int>& color, vector<int>& sat, vector<unordered_set<int>>& used, vector<char>& colored, int colored_cnt, int limit_nodes) {
    if (colored_cnt == g.n) return true;

    int best = -1, best_sat = -1, best_deg = -1;
    for (int v = 0; v < g.n; ++v) {
        if (colored[v]) continue;
        int sv = (int)used[v].size();
        if (sv > best_sat || (sv == best_sat && g.deg[v] > best_deg)) {
            best = v;
            best_sat = sv;
            best_deg = g.deg[v];
        }
    }
    if (best == -1) return true;
    if (colored_cnt > limit_nodes) return false;

    vector<char> forbidden(g.k, 0);
    for (int u : g.adj[best]) if (color[u] != -1) forbidden[color[u]] = 1;

    vector<int> colors;
    for (int c = 0; c < g.k; ++c) if (!forbidden[c]) colors.push_back(c);
    sort(colors.begin(), colors.end(), [&](int a, int b) {
        int ca = 0, cb = 0;
        for (int u : g.adj[best]) {
            if (!colored[u]) continue;
            if (color[u] == a) ++ca;
            if (color[u] == b) ++cb;
        }
        return ca < cb;
    });

    for (int c : colors) {
        color[best] = c;
        colored[best] = 1;
        vector<int> touched;
        for (int u : g.adj[best]) {
            if (!colored[u] && used[u].insert(c).second) touched.push_back(u);
        }
        bool ok = exact_backtrack(g, color, sat, used, colored, colored_cnt + 1, limit_nodes);
        if (ok) return true;
        for (int u : touched) {
            used[u].erase(c);
        }
        color[best] = -1;
        colored[best] = 0;
    }
    return false;
}

static vector<int> solve_graph(const Graph& g) {
    vector<int> deg_order(g.n);
    iota(deg_order.begin(), deg_order.end(), 0);
    sort(deg_order.begin(), deg_order.end(), [&](int a, int b) {
        if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
        return a < b;
    });

    mt19937_64 rng(now_seed());

    if (g.n == 0) return {};
    if (g.k <= 0) return vector<int>(g.n, 0);
    if (g.k == 1) return vector<int>(g.n, 0);

    if (g.n <= 120) {
        vector<int> color(g.n, -1), sat(g.n, 0);
        vector<unordered_set<int>> used(g.n);
        vector<char> colored(g.n, 0);
        if (exact_backtrack(g, color, sat, used, colored, 0, 120)) return color;
    }

    // Try a few randomized DSATUR passes first.
    for (int attempt = 0; attempt < 24; ++attempt) {
        vector<int> c = greedy_dsatur(g, rng, attempt > 0);
        if (!c.empty() && is_valid_coloring(g, c)) return c;
    }

    // Fallback: greedy with degree order and occasional recolor attempts.
    vector<int> color(g.n, -1);
    vector<vector<int>> forbidden_count(g.n, vector<int>(g.k, 0));
    for (int v : deg_order) {
        vector<char> bad(g.k, 0);
        for (int u : g.adj[v]) if (color[u] != -1) bad[color[u]] = 1;
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (!bad[c]) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            // Try to free a color by recoloring a single neighbor.
            for (int c = 0; c < g.k && chosen == -1; ++c) {
                int block = -1;
                for (int u : g.adj[v]) if (color[u] == c) { block = u; break; }
                if (block == -1) { chosen = c; break; }
                vector<char> bad2(g.k, 0);
                for (int w : g.adj[block]) if (w != v && color[w] != -1) bad2[color[w]] = 1;
                for (int c2 = 0; c2 < g.k; ++c2) {
                    if (!bad2[c2]) {
                        color[block] = c2;
                        chosen = c;
                        break;
                    }
                }
                if (chosen != -1) break;
            }
        }
        if (chosen == -1) {
            // Deterministic last resort: choose a color and repair later.
            chosen = 0;
        }
        color[v] = chosen;
    }

    // Local repair loop: repeatedly fix conflicts by recoloring the higher-degree endpoint.
    for (int iter = 0; iter < 8 * g.n + 1000; ++iter) {
        int conflict_u = -1, conflict_v = -1;
        for (int u = 0; u < g.n; ++u) {
            for (int v : g.adj[u]) {
                if (u < v && color[u] == color[v]) {
                    conflict_u = u;
                    conflict_v = v;
                    break;
                }
            }
            if (conflict_u != -1) break;
        }
        if (conflict_u == -1) break;

        int x = (g.deg[conflict_u] >= g.deg[conflict_v]) ? conflict_u : conflict_v;
        vector<char> bad(g.k, 0);
        for (int y : g.adj[x]) if (y != conflict_u && y != conflict_v && color[y] != -1) bad[color[y]] = 1;
        bool moved = false;
        for (int c = 0; c < g.k; ++c) {
            if (!bad[c] && c != color[x]) {
                color[x] = c;
                moved = true;
                break;
            }
        }
        if (!moved) {
            int y = (x == conflict_u ? conflict_v : conflict_u);
            for (int c = 0; c < g.k; ++c) {
                if (c == color[y]) continue;
                bool ok = true;
                for (int z : g.adj[y]) {
                    if (z != x && color[z] == c) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    color[y] = c;
                    moved = true;
                    break;
                }
            }
        }
        if (!moved) {
            color[x] = (color[x] + 1) % g.k;
        }
    }

    if (is_valid_coloring(g, color)) return color;

    // Final exact attempt on small/medium graphs with a bounded node cap.
    if (g.n <= 2500) {
        vector<int> exact_color(g.n, -1), sat(g.n, 0);
        vector<unordered_set<int>> used(g.n);
        vector<char> colored(g.n, 0);
        if (exact_backtrack(g, exact_color, sat, used, colored, 0, 400)) return exact_color;
    }

    // As a last resort, provide a deterministic coloring. This should be rare.
    vector<int> fallback(g.n, 0);
    for (int v : deg_order) {
        vector<char> bad(g.k, 0);
        for (int u : g.adj[v]) if (fallback[u] != -1) bad[fallback[u]] = 1;
        for (int c = 0; c < g.k; ++c) {
            if (!bad[c]) {
                fallback[v] = c;
                break;
            }
        }
    }
    return fallback;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    if (!(cin >> g.n >> g.m >> g.k)) return 0;
    g.adj.assign(g.n, {});
    g.deg.assign(g.n, 0);
    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= g.n || v < 0 || v >= g.n || u == v) continue;
        g.adj[u].push_back(v);
        g.adj[v].push_back(u);
    }
    for (int i = 0; i < g.n; ++i) {
        sort(g.adj[i].begin(), g.adj[i].end());
        g.adj[i].erase(unique(g.adj[i].begin(), g.adj[i].end()), g.adj[i].end());
        g.deg[i] = (int)g.adj[i].size();
    }

    vector<int> ans = solve_graph(g);
    if ((int)ans.size() != g.n) ans.assign(g.n, 0);
    for (int i = 0; i < g.n; ++i) {
        if (ans[i] < 0 || ans[i] >= g.k) ans[i] = 0;
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
