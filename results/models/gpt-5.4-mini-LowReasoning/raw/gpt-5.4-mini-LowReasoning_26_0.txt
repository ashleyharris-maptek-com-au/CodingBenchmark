use std::env;
use std::fs::{remove_file, File};
use std::io::{self, BufReader, BufWriter, Read, Write};
use std::path::PathBuf;

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone, Copy, Default)]
struct Centroid {
    x: f64,
    y: f64,
    z: f64,
}

impl Centroid {
    fn from_point(p: Point) -> Self {
        Self { x: p.x as f64, y: p.y as f64, z: p.z as f64 }
    }

    fn dist2_point(&self, p: Point) -> f64 {
        let dx = self.x - p.x as f64;
        let dy = self.y - p.y as f64;
        let dz = self.z - p.z as f64;
        dx * dx + dy * dy + dz * dz
    }
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 0x9E3779B97F4A7C15 } else { seed } }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }

    fn gen_range(&mut self, n: usize) -> usize {
        if n <= 1 { 0 } else { (self.next_u64() as usize) % n }
    }

    fn gen_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn read_header_and_spool_points() -> io::Result<(usize, usize, PathBuf, u64)> {
    let mut stdin = io::stdin().lock();
    let mut header = Vec::with_capacity(64);
    let mut buf = [0u8; 1];
    let mut values = Vec::with_capacity(2);
    let mut token = Vec::new();
    loop {
        stdin.read_exact(&mut buf)?;
        let b = buf[0];
        if b.is_ascii_whitespace() {
            if !token.is_empty() {
                values.push(String::from_utf8(token.clone()).unwrap());
                token.clear();
                if values.len() == 2 {
                    break;
                }
            }
        } else {
            token.push(b);
        }
        header.push(b);
    }
    let n: usize = values[0].parse().unwrap_or(0);
    let mut k: usize = values[1].parse().unwrap_or(1);
    if k == 0 {
        k = 1;
    }
    if n == 0 {
        let path = temp_path();
        File::create(&path)?;
        return Ok((0, k, path, 0));
    }
    let path = temp_path();
    let mut file = File::create(&path)?;
    if !token.is_empty() {
        panic!("unexpected extra token in header");
    }
    let mut total = 0u64;
    let mut copy_buf = [0u8; 8192];
    loop {
        let read = stdin.read(&mut copy_buf)?;
        if read == 0 {
            break;
        }
        file.write_all(&copy_buf[..read])?;
        total += read as u64;
    }
    Ok((n, k, path, total))
}

fn temp_path() -> PathBuf {
    let mut p = env::temp_dir();
    let unique = format!(
        "codex_cluster_{}_{}.bin",
        std::process::id(),
        XorShift64::new(0xD1B54A32D192ED03).next_u64()
    );
    p.push(unique);
    p
}

fn load_points(path: &PathBuf, n: usize) -> io::Result<Vec<Point>> {
    let mut f = BufReader::new(File::open(path)?);
    let mut points = Vec::with_capacity(n);
    let mut raw = [0u8; 12];
    for _ in 0..n {
        f.read_exact(&mut raw)?;
        points.push(Point {
            x: f32::from_le_bytes(raw[0..4].try_into().unwrap()),
            y: f32::from_le_bytes(raw[4..8].try_into().unwrap()),
            z: f32::from_le_bytes(raw[8..12].try_into().unwrap()),
        });
    }
    Ok(points)
}

fn read_point_at<R: Read>(r: &mut R) -> io::Result<Point> {
    let mut raw = [0u8; 12];
    r.read_exact(&mut raw)?;
    Ok(Point {
        x: f32::from_le_bytes(raw[0..4].try_into().unwrap()),
        y: f32::from_le_bytes(raw[4..8].try_into().unwrap()),
        z: f32::from_le_bytes(raw[8..12].try_into().unwrap()),
    })
}

fn kmeans_full(points: &[Point], k: usize, iters: usize, rng: &mut XorShift64) -> Vec<u32> {
    let n = points.len();
    let k = k.min(n.max(1));
    let mut centroids = init_kmeans_pp(points, k, rng);
    let mut labels = vec![0u32; n];
    for _ in 0..iters {
        let mut sums = vec![Centroid::default(); k];
        let mut counts = vec![0u32; k];
        let mut changed = 0usize;
        for (i, &p) in points.iter().enumerate() {
            let mut best = 0usize;
            let mut best_d = centroids[0].dist2_point(p);
            for c in 1..k {
                let d = centroids[c].dist2_point(p);
                if d < best_d {
                    best_d = d;
                    best = c;
                }
            }
            if labels[i] as usize != best {
                changed += 1;
            }
            labels[i] = best as u32;
            sums[best].x += p.x as f64;
            sums[best].y += p.y as f64;
            sums[best].z += p.z as f64;
            counts[best] += 1;
        }
        reinit_empty_clusters(points, &mut centroids, &counts, rng);
        for c in 0..k {
            if counts[c] > 0 {
                let inv = 1.0 / counts[c] as f64;
                centroids[c].x = sums[c].x * inv;
                centroids[c].y = sums[c].y * inv;
                centroids[c].z = sums[c].z * inv;
            }
        }
        if changed == 0 {
            break;
        }
    }
    labels
}

fn init_kmeans_pp(points: &[Point], k: usize, rng: &mut XorShift64) -> Vec<Centroid> {
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    centroids.push(Centroid::from_point(points[rng.gen_range(n)]));
    let mut dist2 = vec![0.0f64; n];
    for _ in 1..k {
        let mut total = 0.0;
        for (i, &p) in points.iter().enumerate() {
            let mut best = centroids[0].dist2_point(p);
            for c in 1..centroids.len() {
                best = best.min(centroids[c].dist2_point(p));
            }
            dist2[i] = best;
            total += best;
        }
        let mut r = rng.gen_f64() * total.max(1e-12);
        let mut idx = 0usize;
        for (i, &d) in dist2.iter().enumerate() {
            r -= d;
            if r <= 0.0 {
                idx = i;
                break;
            }
            idx = i;
        }
        centroids.push(Centroid::from_point(points[idx]));
    }
    centroids
}

fn reinit_empty_clusters(points: &[Point], centroids: &mut [Centroid], counts: &[u32], rng: &mut XorShift64) {
    let k = centroids.len();
    for c in 0..k {
        if counts[c] == 0 {
            centroids[c] = Centroid::from_point(points[rng.gen_range(points.len())]);
        }
    }
}

fn sample_points(path: &PathBuf, n: usize, sample_size: usize) -> io::Result<Vec<Point>> {
    let mut f = BufReader::new(File::open(path)?);
    let mut sample = Vec::with_capacity(sample_size.min(n));
    let mut rng = XorShift64::new(0xA24BAED4963EE407);
    for i in 0..n {
        let p = read_point_at(&mut f)?;
        if sample.len() < sample_size {
            sample.push(p);
        } else {
            let j = rng.gen_range(i + 1);
            if j < sample_size {
                sample[j] = p;
            }
        }
    }
    Ok(sample)
}

fn streaming_kmeans(path: &PathBuf, n: usize, k: usize, passes: usize) -> io::Result<Vec<Centroid>> {
    let k = k.min(n.max(1));
    let seed_sample = sample_points(path, n, (k * 8).clamp(1024, 50000))?;
    let mut rng = XorShift64::new(0xC6BC279692B5C323);
    let mut centroids = init_kmeans_pp(&seed_sample, k.min(seed_sample.len().max(1)), &mut rng);
    if centroids.len() < k {
        while centroids.len() < k {
            centroids.push(centroids[rng.gen_range(centroids.len())]);
        }
    }
    let mut lr0 = 0.5;
    for pass in 0..passes {
        let mut f = BufReader::new(File::open(path)?);
        let mut counts = vec![0u64; k];
        for _ in 0..n {
            let p = read_point_at(&mut f)?;
            let mut best = 0usize;
            let mut best_d = centroids[0].dist2_point(p);
            for c in 1..k {
                let d = centroids[c].dist2_point(p);
                if d < best_d {
                    best_d = d;
                    best = c;
                }
            }
            counts[best] += 1;
            let lr = lr0 / (1.0 + counts[best] as f64 * 1e-6 + pass as f64 * 0.25);
            centroids[best].x = (1.0 - lr) * centroids[best].x + lr * p.x as f64;
            centroids[best].y = (1.0 - lr) * centroids[best].y + lr * p.y as f64;
            centroids[best].z = (1.0 - lr) * centroids[best].z + lr * p.z as f64;
        }
        lr0 *= 0.7;
    }
    Ok(centroids)
}

fn assign_and_write(path: &PathBuf, n: usize, centroids: &[Centroid]) -> io::Result<()> {
    let mut f = BufReader::new(File::open(path)?);
    let mut out = BufWriter::new(io::stdout().lock());
    for _ in 0..n {
        let p = read_point_at(&mut f)?;
        let mut best = 0usize;
        let mut best_d = centroids[0].dist2_point(p);
        for c in 1..centroids.len() {
            let d = centroids[c].dist2_point(p);
            if d < best_d {
                best_d = d;
                best = c;
            }
        }
        out.write_all(&(best as u32).to_le_bytes())?;
    }
    out.flush()?;
    Ok(())
}

fn main() -> io::Result<()> {
    let (n, k, path, _bytes) = read_header_and_spool_points()?;
    if n == 0 {
        let _ = remove_file(&path);
        return Ok(());
    }

    let mut rng = XorShift64::new(0x123456789ABCDEF0);
    if n <= 2_000_000 {
        let points = load_points(&path, n)?;
        let labels = kmeans_full(&points, k, 20, &mut rng);
        let mut out = BufWriter::new(io::stdout().lock());
        for lab in labels {
            out.write_all(&lab.to_le_bytes())?;
        }
        out.flush()?;
    } else if n <= 100_000_000 {
        let centroids = streaming_kmeans(&path, n, k, 3)?;
        assign_and_write(&path, n, &centroids)?;
    } else {
        let centroids = streaming_kmeans(&path, n, k, 2)?;
        assign_and_write(&path, n, &centroids)?;
    }

    let _ = remove_file(&path);
    Ok(())
}
