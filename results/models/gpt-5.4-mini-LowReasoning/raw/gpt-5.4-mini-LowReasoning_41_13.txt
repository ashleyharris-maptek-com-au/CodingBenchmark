; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 30
OpCapability Shader
OpMemoryModel Logical GLSL450
%1 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%c_0_1 = OpConstant %float 0.100000001
%c_0_4 = OpConstant %float 0.400000006
%c_1_0 = OpConstant %float 1.0
%c_0_5 = OpConstant %float 0.5
%c_0_0 = OpConstant %float 0.0
%c_1_0b = OpConstant %float 1.0
%c_v4_1 = OpConstantComposite %v4float %c_1_0b %c_1_0b %c_1_0b %c_1_0b
%c_v4_bottom = OpConstantComposite %v4float %c_0_1 %c_0_4 %c_0_1 %c_1_0

%worldPos = OpVariable %ptr_in_v3float Input
%outColor = OpVariable %ptr_out_v4float Output

%main = OpFunction %void None %funct
%label = OpLabel
%wp = OpLoad %v3float %worldPos
%y = OpCompositeExtract %float %wp 1
%mul = OpFMul %float %y %c_0_5
%add = OpFAdd %float %mul %c_0_5
%t = OpExtInst %float %1 FClamp %add %c_0_0 %c_1_0
%tvec = OpCompositeConstruct %v4float %t %t %t %c_1_0
%mixed = OpExtInst %v4float %1 FMix %c_v4_bottom %c_v4_1 %tvec
OpStore %outColor %mixed
OpReturn
OpFunctionEnd
