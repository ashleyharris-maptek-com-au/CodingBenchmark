import sys


def solve_exact_cover(U, sets):
    # Map each element to the set indices that contain it.
    elem_to_sets = [[] for _ in range(U)]
    set_elems = [[] for _ in range(len(sets))]
    for idx, s in enumerate(sets):
        uniq = []
        seen = set()
        for x in s:
            if x not in seen:
                seen.add(x)
                uniq.append(x)
        set_elems[idx] = uniq
        for x in uniq:
            elem_to_sets[x].append(idx)

    # If any element is uncovered by all sets, no solution exists.
    for e in range(U):
        if not elem_to_sets[e]:
            return None

    chosen = []
    covered = [False] * U
    used_set = [False] * len(sets)

    def dfs():
        # Find an uncovered element with the fewest compatible sets.
        target = -1
        best_options = None
        best_count = 10**9
        for e in range(U):
            if not covered[e]:
                options = []
                for si in elem_to_sets[e]:
                    if used_set[si]:
                        continue
                    se = set_elems[si]
                    conflict = False
                    for x in se:
                        if covered[x]:
                            conflict = True
                            break
                    if not conflict:
                        options.append(si)
                cnt = len(options)
                if cnt == 0:
                    return False
                if cnt < best_count:
                    best_count = cnt
                    target = e
                    best_options = options
                    if cnt == 1:
                        break

        if target == -1:
            return True

        for si in best_options:
            se = set_elems[si]
            used_set[si] = True
            chosen.append(si)
            for x in se:
                covered[x] = True
            if dfs():
                return True
            for x in se:
                covered[x] = False
            chosen.pop()
            used_set[si] = False
        return False

    if dfs():
        return chosen
    return None


def main():
    data = sys.stdin.buffer.read().splitlines()
    if not data:
        return
    U, S = map(int, data[0].split())
    sets = []
    for i in range(1, 1 + S):
        if i < len(data) and data[i].strip():
            sets.append(list(map(int, data[i].split())))
        else:
            sets.append([])

    ans = solve_exact_cover(U, sets)
    out = []
    if ans is None:
        out.append("NO SOLUTION")
    else:
        out.append("SOLUTION")
        out.append(" ".join(map(str, ans)))
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
