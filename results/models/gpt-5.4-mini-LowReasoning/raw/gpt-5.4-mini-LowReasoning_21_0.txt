#include <algorithm>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <sstream>
#include <string>
#include <unordered_set>
#include <vector>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        size_t h = 0;
        for (int x : v) {
            size_t y = static_cast<size_t>(x) + 0x9e3779b97f4a7c15ULL;
            h ^= y + (h << 6) + (h >> 2);
        }
        return h;
    }
};

static long long manhattan(const vector<int>& a, const vector<int>& b) {
    long long d = 0;
    for (size_t i = 0; i < a.size(); ++i) d += llabs((long long)a[i] - (long long)b[i]);
    return d;
}

static bool inside(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static vector<int> apply_move(const vector<int>& p, int axis, int dir) {
    vector<int> q = p;
    q[axis] += dir;
    return q;
}

static long long score_move(const vector<int>& head, const vector<int>& next, const vector<int>& target) {
    long long before = manhattan(head, target);
    long long after = manhattan(next, target);
    return (before - after) * 1000 - after;
}

static int free_neighbour_count(
    const vector<int>& p,
    const vector<int>& bounds,
    const unordered_set<vector<int>, VecHash>& blocked
) {
    int cnt = 0;
    for (size_t axis = 0; axis < p.size(); ++axis) {
        for (int dir : {-1, 1}) {
            vector<int> q = p;
            q[axis] += dir;
            if (inside(q, bounds) && !blocked.count(q)) ++cnt;
        }
    }
    return cnt;
}

static bool limited_reachability(
    const vector<int>& start,
    const vector<int>& bounds,
    const unordered_set<vector<int>, VecHash>& blocked,
    int limit,
    int need
) {
    queue<vector<int>> q;
    unordered_set<vector<int>, VecHash> seen;
    q.push(start);
    seen.insert(start);
    int reached = 0;
    while (!q.empty() && reached < limit) {
        vector<int> cur = q.front();
        q.pop();
        ++reached;
        if (reached >= need) return true;
        for (size_t axis = 0; axis < cur.size(); ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> nxt = cur;
                nxt[axis] += dir;
                if (!inside(nxt, bounds) || blocked.count(nxt) || seen.count(nxt)) continue;
                seen.insert(nxt);
                q.push(move(nxt));
            }
        }
    }
    return reached >= need;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> foods(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) cin >> foods[i][j];
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash> obstacles;
    for (int i = 0; i < O; ++i) {
        vector<int> p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        obstacles.insert(move(p));
    }

    deque<vector<int>> snake;
    snake.push_front(head);
    for (int i = 1; i < snake_length; ++i) {
        vector<int> b = head;
        int axis = 0;
        if (b[axis] - i >= 0) b[axis] -= i;
        else if (b[axis] + i < bounds[axis]) b[axis] += i;
        else break;
        snake.push_back(move(b));
    }

    unordered_set<vector<int>, VecHash> body(snake.begin(), snake.end());

    int turns = 0;
    while (turns < max_turns && !foods.empty()) {
        int best_food = 0;
        long long best_food_dist = numeric_limits<long long>::max();
        for (int i = 0; i < (int)foods.size(); ++i) {
            long long d = manhattan(head, foods[i]);
            if (d < best_food_dist) {
                best_food_dist = d;
                best_food = i;
            }
        }
        vector<int> target = foods[best_food];

        vector<int> tail = snake.back();
        unordered_set<vector<int>, VecHash> blocked = obstacles;
        for (const auto& p : body) blocked.insert(p);

        bool ate = false;
        vector<pair<int, int>> candidates;
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> next = apply_move(head, axis, dir);
                if (!inside(next, bounds)) continue;
                bool will_eat = (next == target);
                bool tail_free = !will_eat;
                if (tail_free) {
                    blocked.erase(tail);
                }
                if (blocked.count(next)) {
                    if (tail_free) blocked.insert(tail);
                    continue;
                }
                if (tail_free) blocked.insert(tail);
                long long base = score_move(head, next, target);
                int adj = free_neighbour_count(next, bounds, blocked);
                long long sc = base + adj * 3;
                if (will_eat) sc += 1000000;
                candidates.push_back({(int)sc, axis * 2 + (dir > 0 ? 1 : 0)});
            }
        }

        int chosen_axis = -1, chosen_dir = 1;
        if (!candidates.empty()) {
            sort(candidates.begin(), candidates.end(), [](const auto& a, const auto& b) {
                if (a.first != b.first) return a.first > b.first;
                return a.second < b.second;
            });

            for (const auto& cand : candidates) {
                int code = cand.second;
                int axis = code / 2;
                int dir = (code % 2) ? 1 : -1;
                vector<int> next = apply_move(head, axis, dir);
                unordered_set<vector<int>, VecHash> test_blocked = obstacles;
                for (const auto& p : body) test_blocked.insert(p);
                bool will_eat = (next == target);
                if (!will_eat) test_blocked.erase(tail);
                if (test_blocked.count(next)) continue;
                int need = (int)snake.size() + (will_eat ? 1 : 0);
                if (limited_reachability(next, bounds, test_blocked, 5000, min(need + 2, 2000))) {
                    chosen_axis = axis;
                    chosen_dir = dir;
                    ate = will_eat;
                    break;
                }
            }
            if (chosen_axis == -1) {
                int code = candidates.front().second;
                chosen_axis = code / 2;
                chosen_dir = (code % 2) ? 1 : -1;
                ate = (apply_move(head, chosen_axis, chosen_dir) == target);
            }
        } else {
            break;
        }

        cout << chosen_axis << ' ' << chosen_dir << '\n';
        cout.flush();

        vector<int> next = apply_move(head, chosen_axis, chosen_dir);
        snake.push_front(next);
        body.insert(next);
        head = next;
        if (ate) {
            foods.erase(foods.begin() + best_food);
        } else {
            vector<int> old_tail = snake.back();
            snake.pop_back();
            body.erase(old_tail);
        }
        ++turns;
    }

    return 0;
}
