#include <bits/stdc++.h>
using namespace std;

static constexpr double EPS = 1e-9;

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}
static inline double norm2(const Vec3& a) { return dot(a, a); }
static inline Vec3 normalize(const Vec3& a) {
    double n = sqrt(max(0.0, norm2(a)));
    if (n < EPS) return {0, 0, 0};
    return a / n;
}

struct Plane {
    Vec3 n;
    double w{};
    Plane() = default;
    Plane(const Vec3& a, const Vec3& b, const Vec3& c) {
        n = normalize(cross(b - a, c - a));
        w = dot(n, a);
    }
    void flip() { n = n * -1.0; w = -w; }
};

struct Polygon {
    vector<Vec3> v;
    Plane plane;
};

struct BSPNode {
    vector<Polygon> polygons;
    unique_ptr<BSPNode> front, back;
    optional<Plane> plane;

    static int classifyPoint(const Plane& p, const Vec3& v) {
        double t = dot(p.n, v) - p.w;
        if (t < -EPS) return -1;
        if (t > EPS) return 1;
        return 0;
    }

    static void splitPolygon(const Plane& p, const Polygon& poly,
                             vector<Polygon>& coplanarFront,
                             vector<Polygon>& coplanarBack,
                             vector<Polygon>& frontPolys,
                             vector<Polygon>& backPolys) {
        int polygonType = 0;
        vector<int> types(poly.v.size());
        for (size_t i = 0; i < poly.v.size(); ++i) {
            types[i] = classifyPoint(p, poly.v[i]);
            polygonType |= (types[i] > 0 ? 1 : 0);
            polygonType |= (types[i] < 0 ? 2 : 0);
        }

        switch (polygonType) {
            case 0:
                (dot(p.n, poly.plane.n) > 0 ? coplanarFront : coplanarBack).push_back(poly);
                break;
            case 1:
                frontPolys.push_back(poly);
                break;
            case 2:
                backPolys.push_back(poly);
                break;
            default: {
                vector<Vec3> f, b;
                for (size_t i = 0; i < poly.v.size(); ++i) {
                    size_t j = (i + 1) % poly.v.size();
                    const Vec3& vi = poly.v[i];
                    const Vec3& vj = poly.v[j];
                    int ti = types[i], tj = types[j];
                    if (ti >= 0) f.push_back(vi);
                    if (ti <= 0) b.push_back(vi);
                    if ((ti < 0 && tj > 0) || (ti > 0 && tj < 0)) {
                        Vec3 d = vj - vi;
                        double t = (p.w - dot(p.n, vi)) / dot(p.n, d);
                        Vec3 v = vi + d * t;
                        f.push_back(v);
                        b.push_back(v);
                    }
                }
                if (f.size() >= 3) frontPolys.push_back({move(f), poly.plane});
                if (b.size() >= 3) backPolys.push_back({move(b), poly.plane});
            } break;
        }
    }

    void build(const vector<Polygon>& polys) {
        if (polys.empty()) return;
        if (!plane.has_value()) plane = polys[0].plane;

        vector<Polygon> frontList, backList;
        vector<Polygon> coplanarFront, coplanarBack;
        for (const auto& poly : polys) {
            splitPolygon(*plane, poly, coplanarFront, coplanarBack, frontList, backList);
        }
        polygons.insert(polygons.end(), coplanarFront.begin(), coplanarFront.end());
        polygons.insert(polygons.end(), coplanarBack.begin(), coplanarBack.end());
        if (!frontList.empty()) {
            if (!front) front = make_unique<BSPNode>();
            front->build(frontList);
        }
        if (!backList.empty()) {
            if (!back) back = make_unique<BSPNode>();
            back->build(backList);
        }
    }

    void allPolygons(vector<Polygon>& out) const {
        out.insert(out.end(), polygons.begin(), polygons.end());
        if (front) front->allPolygons(out);
        if (back) back->allPolygons(out);
    }

    void invert() {
        for (auto& p : polygons) {
            reverse(p.v.begin(), p.v.end());
            p.plane.flip();
        }
        if (plane) plane->flip();
        if (front) front->invert();
        if (back) back->invert();
        swap(front, back);
    }

    void clipPolygons(const vector<Polygon>& polys, vector<Polygon>& out) const {
        if (!plane) {
            out.insert(out.end(), polys.begin(), polys.end());
            return;
        }
        vector<Polygon> frontList, backList;
        vector<Polygon> cf, cb;
        for (const auto& p : polys) splitPolygon(*plane, p, cf, cb, frontList, backList);
        if (front) front->clipPolygons(frontList, out);
        else out.insert(out.end(), frontList.begin(), frontList.end());
        if (back) back->clipPolygons(backList, out);
    }

    void clipTo(const BSPNode& other) {
        vector<Polygon> clipped;
        other.clipPolygons(polygons, clipped);
        polygons.swap(clipped);
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }
};

static vector<Polygon> meshToPolys(const vector<Vec3>& verts, const vector<array<int,3>>& faces) {
    vector<Polygon> polys;
    polys.reserve(faces.size());
    for (auto f : faces) {
        const Vec3& a = verts[f[0]];
        const Vec3& b = verts[f[1]];
        const Vec3& c = verts[f[2]];
        if (norm2(cross(b - a, c - a)) < EPS * EPS) continue;
        Polygon p;
        p.v = {a, b, c};
        p.plane = Plane(a, b, c);
        polys.push_back(move(p));
    }
    return polys;
}

struct Key {
    long long x, y, z;
    bool operator==(const Key& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct KeyHash {
    size_t operator()(const Key& k) const noexcept {
        size_t h = std::hash<long long>{}(k.x);
        h ^= std::hash<long long>{}(k.y) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<long long>{}(k.z) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

static int addVertex(const Vec3& v, vector<Vec3>& out, unordered_map<Key,int,KeyHash>& mp) {
    const double S = 1e9;
    Key k{llround(v.x * S), llround(v.y * S), llround(v.z * S)};
    auto it = mp.find(k);
    if (it != mp.end()) return it->second;
    int id = (int)out.size();
    out.push_back(v);
    mp.emplace(k, id);
    return id;
}

static vector<array<int,3>> triangulateAndCollect(const vector<Polygon>& polys, vector<Vec3>& outVerts) {
    unordered_map<Key,int,KeyHash> mp;
    vector<array<int,3>> faces;
    for (const auto& p : polys) {
        if (p.v.size() < 3) continue;
        vector<int> ids;
        ids.reserve(p.v.size());
        for (const auto& v : p.v) ids.push_back(addVertex(v, outVerts, mp));
        for (size_t i = 1; i + 1 < ids.size(); ++i) faces.push_back({ids[0], ids[i], ids[i + 1]});
    }
    return faces;
}

static vector<Polygon> booleanUnion(vector<Polygon> A, vector<Polygon> B) {
    BSPNode a, b;
    a.build(A);
    b.build(B);
    a.clipTo(b);
    b.clipTo(a);
    b.invert();
    b.clipTo(a);
    b.invert();
    a.build(b.polygons);
    vector<Polygon> out;
    a.allPolygons(out);
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    vector<Vec3> va(VA), vb(VB);
    for (int i = 0; i < VA; ++i) cin >> va[i].x >> va[i].y >> va[i].z;
    vector<array<int,3>> fa(FA), fb(FB);
    for (int i = 0; i < FA; ++i) cin >> fa[i][0] >> fa[i][1] >> fa[i][2];
    for (int i = 0; i < VB; ++i) cin >> vb[i].x >> vb[i].y >> vb[i].z;
    for (int i = 0; i < FB; ++i) cin >> fb[i][0] >> fb[i][1] >> fb[i][2];

    auto polysA = meshToPolys(va, fa);
    auto polysB = meshToPolys(vb, fb);
    auto polys = booleanUnion(move(polysA), move(polysB));

    vector<Vec3> outVerts;
    auto outFaces = triangulateAndCollect(polys, outVerts);

    cout << outVerts.size() << ' ' << outFaces.size() << '\n';
    cout.setf(std::ios::fixed); cout << setprecision(10);
    for (const auto& v : outVerts) cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    for (const auto& f : outFaces) cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    return 0;
}
