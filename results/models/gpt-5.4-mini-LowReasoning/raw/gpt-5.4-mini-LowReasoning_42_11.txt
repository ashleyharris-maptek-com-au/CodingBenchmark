OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_8 = OpConstant %uint 8
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_33333334 = OpConstant %float 0.33333334
%float_6666667 = OpConstant %float 0.6666667
%float_002 = OpConstant %float 0.02
%float_001 = OpConstant %float 0.01

%v4float_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%v4float_1 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1
%v3float_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v3_out = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_v3_out Output

%mat4 = OpTypeMatrix %v4float 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 ColMajor
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 0 MatrixStride 16
OpMemberDecorate %ubo_struct 1 ColMajor
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 1 MatrixStride 16
OpMemberDecorate %ubo_struct 2 ColMajor
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 2 MatrixStride 16
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%main = OpFunction %void None %func
%entry = OpLabel

%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1 = OpLoad %v4float %p1ptr

%c0ptr = OpAccessChain %ptr_v3_in %color %uint_1
%c1ptr = OpAccessChain %ptr_v3_in %color %uint_2
%c0 = OpLoad %v3float %c0ptr
%c1 = OpLoad %v3float %c1ptr

%seg = OpFSub %v4float %p1 %p0
%seg3 = OpVectorShuffle %v3float %seg %seg 0 1 2
%segx = OpCompositeExtract %float %seg3 0
%segy = OpCompositeExtract %float %seg3 1
%segz = OpCompositeExtract %float %seg3 2
%segx2 = OpFMul %float %segx %segx
%segy2 = OpFMul %float %segy %segy
%segz2 = OpFMul %float %segz %segz
%len2a = OpFAdd %float %segx2 %segy2
%len2 = OpFAdd %float %len2a %segz2
%len = OpExtInst %float %glsl GLSLstd450Sqrt %len2
%invlen = OpFDiv %float %float_1 %len
%dirx = OpFMul %float %segx %invlen
%diry = OpFMul %float %segy %invlen
%halfw = OpFMul %float %float_001 %float_1
%offx = OpFMul %float %diry %halfw
%negdirx = OpFNegate %float %dirx
%offy = OpFMul %float %negdirx %halfw

; part 1: t = [0, 1/3]
%t1v = OpCompositeConstruct %v4float %float_33333334 %float_33333334 %float_33333334 %float_33333334
%d0 = OpFSub %v4float %p1 %p0
%d1 = OpFMul %v4float %d0 %t1v
%a1 = OpFAdd %v4float %p0 %d1
%ca1 = OpFSub %v3float %c1 %c0
%ca1s = OpCompositeConstruct %v3float %float_33333334 %float_33333334 %float_33333334
%cat = OpFMul %v3float %ca1 %ca1s
%ca = OpFAdd %v3float %c0 %cat

%a0l = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0
%a1l = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0
%a0r = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0
%a1r = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0

%v0 = OpFAdd %v4float %p0 %a0l
%v1 = OpFAdd %v4float %p0 %a0r
%v2 = OpFAdd %v4float %a1 %a1l
%v3 = OpFAdd %v4float %a1 %a1r

%v0p = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0p %v0
OpStore %outColor %c0
OpEmitVertex
%v1p = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v1p %v1
OpStore %outColor %c0
OpEmitVertex
%v2p = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v2p %v2
OpStore %outColor %ca
OpEmitVertex
%v3p = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v3p %v3
OpStore %outColor %ca
OpEmitVertex
OpEndPrimitive

; part 2: t = [2/3, 1]
%t2v = OpCompositeConstruct %v4float %float_6666667 %float_6666667 %float_6666667 %float_6666667
%b0d = OpFMul %v4float %d0 %t2v
%b0 = OpFAdd %v4float %p0 %b0d
%b1 = OpCopyObject %v4float %p1
%cb2s = OpCompositeConstruct %v3float %float_6666667 %float_6666667 %float_6666667
%cb2 = OpFMul %v3float %ca1 %cb2s
%cb = OpFAdd %v3float %c0 %cb2

%b0l = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0
%b1l = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0
%b0r = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0
%b1r = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0

%w0 = OpFAdd %v4float %b0 %b0l
%w1 = OpFAdd %v4float %b0 %b0r
%w2 = OpFAdd %v4float %b1 %b1l
%w3 = OpFAdd %v4float %b1 %b1r

OpStore %v0p %w0
OpStore %outColor %cb
OpEmitVertex
OpStore %v1p %w1
OpStore %outColor %cb
OpEmitVertex
OpStore %v2p %w2
OpStore %outColor %c1
OpEmitVertex
OpStore %v3p %w3
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
