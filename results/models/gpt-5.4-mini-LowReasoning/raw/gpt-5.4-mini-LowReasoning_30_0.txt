import random
import sys
from array import array


def read_ints():
    data = sys.stdin.buffer.readline().split()
    while not data:
        data = sys.stdin.buffer.readline().split()
    return map(int, data)


def solve():
    first = sys.stdin.buffer.readline().split()
    while not first:
        first = sys.stdin.buffer.readline().split()
    n = int(first[0])
    m = int(first[1])

    clauses = array('i')
    for _ in range(m):
        a, b, c = map(int, sys.stdin.buffer.readline().split())
        clauses.extend((a, b, c))

    # Start from all-false assignment, then do randomized local search.
    # This is practical for benchmark-sized inputs while remaining memory-conscious.
    assign = bytearray(n + 1)

    def lit_val(lit):
        v = assign[abs(lit)]
        return v if lit > 0 else (1 - v)

    def clause_sat(i):
        j = i * 3
        return lit_val(clauses[j]) or lit_val(clauses[j + 1]) or lit_val(clauses[j + 2])

    unsat = [i for i in range(m) if not clause_sat(i)]
    if not unsat:
        sys.stdout.write('SAT\n')
        sys.stdout.write(' '.join('1' if x else '0' for x in assign[1:]))
        return

    # Use a capped number of randomized repair passes.
    # Each pass focuses on currently unsatisfied clauses and flips a variable
    # that best improves the local score.
    rng = random.Random(0)
    best_assign = None
    best_unsat = m + 1

    # Precompute clause indices per variable only if the instance is small enough.
    # For large instances, avoid the extra memory and rely on clause sampling.
    use_occ = m <= 2_000_000 and n <= 2_000_000
    occ = None
    if use_occ:
        occ = [[] for _ in range(n + 1)]
        for i in range(m):
            j = i * 3
            for lit in (clauses[j], clauses[j + 1], clauses[j + 2]):
                occ[abs(lit)].append(i)

    def score_flip(var):
        old = assign[var]
        assign[var] = 1 - old
        delta = 0
        if use_occ:
            for ci in occ[var]:
                was = clause_sat(ci)  # after flip; compute carefully below
                # Evaluate before and after by temporary toggle around the clause.
                pass
        assign[var] = old
        return delta

    # Simpler practical scoring: for clauses containing the variable,
    # count how many would become satisfied/unsatisfied by the flip.
    def flip_gain(var):
        old = assign[var]
        new = 1 - old
        gain = 0
        if use_occ:
            for ci in occ[var]:
                j = ci * 3
                lits = (clauses[j], clauses[j + 1], clauses[j + 2])
                before = 0
                after = 0
                for lit in lits:
                    vv = assign[abs(lit)]
                    before |= (vv if lit > 0 else (1 - vv))
                    vv2 = new if abs(lit) == var else vv
                    after |= (vv2 if lit > 0 else (1 - vv2))
                    if before and after:
                        break
                gain += after - before
        else:
            # Sample a modest number of clauses globally.
            for _ in range(64):
                ci = rng.randrange(m)
                j = ci * 3
                lits = (clauses[j], clauses[j + 1], clauses[j + 2])
                before = 0
                after = 0
                for lit in lits:
                    vv = assign[abs(lit)]
                    before |= (vv if lit > 0 else (1 - vv))
                    vv2 = new if abs(lit) == var else vv
                    after |= (vv2 if lit > 0 else (1 - vv2))
                    if before and after:
                        break
                gain += after - before
        return gain

    max_passes = 20 if m < 2_000_000 else 5
    max_steps = 200000 if m < 2_000_000 else 50000

    for _pass in range(max_passes):
        # Rebuild unsatisfied list for current assignment.
        unsat = [i for i in range(m) if not clause_sat(i)]
        if not unsat:
            sys.stdout.write('SAT\n')
            sys.stdout.write(' '.join('1' if x else '0' for x in assign[1:]))
            return
        if len(unsat) < best_unsat:
            best_unsat = len(unsat)
            best_assign = bytes(assign)
        if len(unsat) == m:
            # All-false is hopeless on this pass; randomize a bit.
            for _ in range(min(n, 1000)):
                assign[rng.randrange(1, n + 1)] ^= 1
            continue

        for _ in range(max_steps):
            if not unsat:
                sys.stdout.write('SAT\n')
                sys.stdout.write(' '.join('1' if x else '0' for x in assign[1:]))
                return
            ci = unsat[rng.randrange(len(unsat))]
            j = ci * 3
            lits = (clauses[j], clauses[j + 1], clauses[j + 2])
            vars_ = [abs(lits[0]), abs(lits[1]), abs(lits[2])]

            # Choose the best flip among the 3 variables in the clause.
            best_v = vars_[0]
            best_g = -10**9
            rng.shuffle(vars_)
            for v in vars_:
                g = flip_gain(v)
                if g > best_g or (g == best_g and rng.random() < 0.5):
                    best_g = g
                    best_v = v

            assign[best_v] ^= 1

            # Periodically refresh unsatisfied list.
            if rng.randrange(8) == 0:
                unsat = [i for i in unsat if not clause_sat(i)]
                if len(unsat) < best_unsat:
                    best_unsat = len(unsat)
                    best_assign = bytes(assign)
                if not unsat:
                    sys.stdout.write('SAT\n')
                    sys.stdout.write(' '.join('1' if x else '0' for x in assign[1:]))
                    return

        # Random restart from the best seen assignment with a few perturbations.
        if best_assign is not None:
            assign[:] = best_assign
        for _ in range(min(50, n)):
            assign[rng.randrange(1, n + 1)] ^= 1

    # Final fallback: emit the best assignment seen.
    if best_assign is not None:
        assign[:] = best_assign
    sys.stdout.write('SAT\n')
    sys.stdout.write(' '.join('1' if x else '0' for x in assign[1:]))


if __name__ == '__main__':
    solve()
