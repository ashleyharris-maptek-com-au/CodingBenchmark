using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32 && c != 0)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    private sealed class Graph
    {
        public readonly int N;
        public readonly int M;
        public readonly int[][] Adj;
        public readonly int[] Deg;
        public Graph(int n, int m, int[][] adj, int[] deg)
        {
            N = n; M = m; Adj = adj; Deg = deg;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();
        var edges = new List<(int, int)>(m);
        var adjSets = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adjSets[i] = new HashSet<int>();
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == v) continue;
            if (u < 0 || u >= n || v < 0 || v >= n) continue;
            if (adjSets[u].Add(v))
            {
                adjSets[v].Add(u);
                edges.Add((u, v));
            }
        }

        var adj = new int[n][];
        var deg = new int[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = adjSets[i].ToArray();
            Array.Sort(adj[i]);
            deg[i] = adj[i].Length;
        }

        var g = new Graph(n, edges.Count, adj, deg);
        var best = Solve(g);

        var sb = new StringBuilder();
        sb.AppendLine(best.Count.ToString());
        if (best.Count > 0)
            sb.AppendLine(string.Join(" ", best));
        else
            sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private static List<int> Solve(Graph g)
    {
        if (g.N == 0) return new List<int>();
        if (g.N <= 80)
            return ExactClique(g);

        // Fast heuristic seed: high-degree greedy in several orderings.
        var best = new List<int>();
        TryHeuristic(g, ref best);

        // Exact search on a reduced candidate set built around the best heuristic clique.
        if (g.N <= 4000)
        {
            var exact = ExactClique(g);
            if (exact.Count > best.Count) best = exact;
        }
        else
        {
            // For very large graphs, strengthen the heuristic with local exact searches.
            best = ImproveLargeGraph(g, best);
        }

        best.Sort();
        return best;
    }

    private static void TryHeuristic(Graph g, ref List<int> best)
    {
        int n = g.N;
        int[] order1 = Enumerable.Range(0, n).OrderByDescending(i => g.Deg[i]).ToArray();
        int[] order2 = Enumerable.Range(0, n).OrderBy(i => g.Deg[i]).ToArray();
        int[] order3 = Enumerable.Range(0, n).OrderBy(i => (g.Deg[i] & 1) == 0 ? g.Deg[i] : -g.Deg[i]).ToArray();
        foreach (var order in new[] { order1, order2, order3 })
        {
            var clique = GreedyClique(g, order);
            if (clique.Count > best.Count) best = clique;
        }

        // Expand from the current best using random-ish degree-based restarts.
        if (best.Count == 0 && n > 0) best = new List<int> { 0 };
    }

    private static List<int> GreedyClique(Graph g, int[] order)
    {
        var clique = new List<int>();
        var inClique = new bool[g.N];
        foreach (int v in order)
        {
            bool ok = true;
            foreach (int u in clique)
            {
                if (Array.BinarySearch(g.Adj[v], u) < 0)
                {
                    ok = false;
                    break;
                }
            }
            if (ok)
            {
                clique.Add(v);
                inClique[v] = true;
            }
        }
        return clique;
    }

    private static List<int> ImproveLargeGraph(Graph g, List<int> best)
    {
        // Use a bounded exact search on neighborhoods of promising vertices.
        var candidates = Enumerable.Range(0, g.N)
            .OrderByDescending(i => g.Deg[i])
            .Take(Math.Min(g.N, 300))
            .ToArray();

        foreach (int v in candidates)
        {
            var sub = new List<int>();
            foreach (int u in g.Adj[v])
                if (g.Deg[u] >= best.Count - 1) sub.Add(u);
            sub.Sort((a, b) => g.Deg[b].CompareTo(g.Deg[a]));
            var clique = BkOnSubset(g, sub, new List<int> { v }, best.Count);
            if (clique.Count > best.Count) best = clique;
        }

        return best;
    }

    private static List<int> ExactClique(Graph g)
    {
        var p = Enumerable.Range(0, g.N).ToList();
        p.Sort((a, b) => g.Deg[b].CompareTo(g.Deg[a]));
        return BkOnSubset(g, p, new List<int>(), 0);
    }

    private static List<int> BkOnSubset(Graph g, List<int> p, List<int> r, int bestKnown)
    {
        List<int> best = new List<int>(r);
        Expand(g, r, p, ref best, bestKnown);
        return best;
    }

    private static void Expand(Graph g, List<int> r, List<int> p, ref List<int> best, int bestKnown)
    {
        if (r.Count + p.Count <= Math.Max(best.Count, bestKnown)) return;
        if (p.Count == 0)
        {
            if (r.Count > best.Count) best = new List<int>(r);
            return;
        }

        int pivot = ChoosePivot(g, p);
        var pNoNbr = new List<int>();
        var pivotNbr = new HashSet<int>(g.Adj[pivot]);
        foreach (var v in p)
            if (!pivotNbr.Contains(v)) pNoNbr.Add(v);

        pNoNbr.Sort((a, b) => g.Deg[b].CompareTo(g.Deg[a]));
        foreach (int v in pNoNbr)
        {
            if (r.Count + p.Count <= best.Count) return;
            var newR = new List<int>(r) { v };
            var newP = IntersectSortedWithSet(g.Adj[v], p);
            Expand(g, newR, newP, ref best, bestKnown);
            p.Remove(v);
        }

        if (r.Count > best.Count) best = new List<int>(r);
    }

    private static int ChoosePivot(Graph g, List<int> p)
    {
        int pivot = p[0];
        int bestScore = -1;
        var set = new HashSet<int>(p);
        foreach (int v in p)
        {
            int score = 0;
            foreach (int u in g.Adj[v])
                if (set.Contains(u)) score++;
            if (score > bestScore)
            {
                bestScore = score;
                pivot = v;
            }
        }
        return pivot;
    }

    private static List<int> IntersectSortedWithSet(int[] sorted, List<int> p)
    {
        var set = new HashSet<int>(p);
        var res = new List<int>();
        foreach (int x in sorted)
            if (set.Contains(x)) res.Add(x);
        return res;
    }
}
