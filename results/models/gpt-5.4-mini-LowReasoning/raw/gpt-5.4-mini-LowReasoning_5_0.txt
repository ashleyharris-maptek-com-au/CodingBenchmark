use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: usize,
    y: usize,
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn neighbors(p: Pt, w: usize, h: usize) -> impl Iterator<Item = Pt> {
    let mut v = Vec::with_capacity(4);
    if p.x > 0 {
        v.push(Pt { x: p.x - 1, y: p.y });
    }
    if p.x + 1 < w {
        v.push(Pt { x: p.x + 1, y: p.y });
    }
    if p.y > 0 {
        v.push(Pt { x: p.x, y: p.y - 1 });
    }
    if p.y + 1 < h {
        v.push(Pt { x: p.x, y: p.y + 1 });
    }
    v.into_iter()
}

fn full_snake(w: usize, h: usize, cycle: bool) -> Option<Vec<Pt>> {
    if w == 0 || h == 0 {
        return None;
    }
    let mut path = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push(Pt { x, y });
            }
        } else {
            for x in (0..w).rev() {
                path.push(Pt { x, y });
            }
        }
    }
    if !cycle {
        return Some(path);
    }
    if w == 1 || h == 1 {
        return None;
    }
    let n = path.len();
    let a = path[0];
    let b = path[n - 1];
    let adj = a.x.abs_diff(b.x) + a.y.abs_diff(b.y) == 1;
    if adj {
        Some(path)
    } else {
        None
    }
}

fn connected_all(free: &[bool], w: usize, h: usize) -> bool {
    let n = free.len();
    let mut start = None;
    let mut cnt = 0usize;
    for i in 0..n {
        if free[i] {
            cnt += 1;
            if start.is_none() {
                start = Some(i);
            }
        }
    }
    let Some(s) = start else { return false; };
    let mut q = VecDeque::new();
    let mut seen = vec![false; n];
    seen[s] = true;
    q.push_back(s);
    let mut got = 0usize;
    while let Some(v) = q.pop_front() {
        got += 1;
        let x = v % w;
        let y = v / w;
        let p = Pt { x, y };
        for nb in neighbors(p, w, h) {
            let ni = idx(nb.x, nb.y, w);
            if free[ni] && !seen[ni] {
                seen[ni] = true;
                q.push_back(ni);
            }
        }
    }
    got == cnt
}

fn parity_ok(free: &[bool], w: usize, h: usize, cycle: bool) -> bool {
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            if free[idx(x, y, w)] {
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }
    if cycle {
        black == white && black + white > 0
    } else {
        let start_black = (0 + 0) % 2 == 0;
        if start_black {
            black == white + 1 || black == white
        } else {
            white == black + 1 || white == black
        }
    }
}

fn degree_of(v: usize, free: &[bool], used: &[bool], w: usize, h: usize) -> usize {
    let x = v % w;
    let y = v / w;
    let mut d = 0usize;
    for nb in neighbors(Pt { x, y }, w, h) {
        let ni = idx(nb.x, nb.y, w);
        if free[ni] && !used[ni] {
            d += 1;
        }
    }
    d
}

fn solve_search(
    free: &[bool],
    w: usize,
    h: usize,
    cycle: bool,
    total: usize,
) -> Option<Vec<Pt>> {
    let start = idx(0, 0, w);
    if !free[start] {
        return None;
    }
    let mut used = vec![false; free.len()];
    let mut path = Vec::with_capacity(total);
    used[start] = true;
    path.push(Pt { x: 0, y: 0 });

    fn dfs(
        cur: usize,
        depth: usize,
        total: usize,
        free: &[bool],
        used: &mut [bool],
        path: &mut Vec<Pt>,
        w: usize,
        h: usize,
        cycle: bool,
    ) -> bool {
        if depth == total {
            if !cycle {
                return true;
            }
            let first = path[0];
            let last = path[path.len() - 1];
            return first.x.abs_diff(last.x) + first.y.abs_diff(last.y) == 1;
        }

        let cx = cur % w;
        let cy = cur / w;
        let mut cand = Vec::new();
        for nb in neighbors(Pt { x: cx, y: cy }, w, h) {
            let ni = idx(nb.x, nb.y, w);
            if free[ni] && !used[ni] {
                cand.push(ni);
            }
        }
        if cand.is_empty() {
            return false;
        }
        cand.sort_by_key(|&ni| degree_of(ni, free, used, w, h));

        for ni in cand {
            used[ni] = true;
            path.push(Pt { x: ni % w, y: ni / w });

            let mut isolated = false;
            for i in 0..free.len() {
                if free[i] && !used[i] {
                    let d = degree_of(i, free, used, w, h);
                    if d == 0 {
                        isolated = true;
                        break;
                    }
                }
            }

            if !isolated && dfs(ni, depth + 1, total, free, used, path, w, h, cycle) {
                return true;
            }
            path.pop();
            used[ni] = false;
        }
        false
    }

    if dfs(start, 1, total, free, &mut used, &mut path, w, h, cycle) {
        Some(path)
    } else {
        None
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let cycle_flag: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let cycle = cycle_flag != 0;

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut free = vec![true; w * h];
    for _ in 0..m {
        let x: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        let y: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        if x < w && y < h {
            free[idx(x, y, w)] = false;
        }
    }

    let total = free.iter().filter(|&&b| b).count();
    if total == 0 || !free[0] {
        println!("0");
        return;
    }

    if !connected_all(&free, w, h) || !parity_ok(&free, w, h, cycle) {
        println!("0");
        return;
    }

    if m == 0 {
        if let Some(path) = full_snake(w, h, cycle) {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
            return;
        }
    }

    if total <= 2500 {
        if let Some(path) = solve_search(&free, w, h, cycle, total) {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
            return;
        }
    }

    println!("0");
}
