declare const require: any;
const fs = require('fs');

type Cell = [number, number];

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob: number[][] = data.probability || [];
const drones: { id: any; x: number; y: number }[] = data.drones || [];
const maxSteps: number = Math.max(0, data.maxStepsPerDrone | 0);

const H = prob.length;
const W = H > 0 ? prob[0].length : 0;

const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

function inside(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}

function isFree(x: number, y: number): boolean {
  return inside(x, y) && prob[y][x] !== -1;
}

function key(x: number, y: number): number {
  return y * W + x;
}

function bfs(startX: number, startY: number): Int32Array {
  const dist = new Int32Array(H * W);
  dist.fill(-1);
  if (!isFree(startX, startY)) return dist;
  const qx = new Int32Array(H * W);
  const qy = new Int32Array(H * W);
  let head = 0;
  let tail = 0;
  const s = key(startX, startY);
  dist[s] = 0;
  qx[tail] = startX;
  qy[tail] = startY;
  tail++;
  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;
    const nd = dist[key(x, y)] + 1;
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (!isFree(nx, ny)) continue;
      const nk = key(nx, ny);
      if (dist[nk] !== -1) continue;
      dist[nk] = nd;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }
  return dist;
}

function reconstructPath(
  sx: number,
  sy: number,
  tx: number,
  ty: number,
  stepBudget: number,
): Cell[] {
  if (sx === tx && sy === ty) return [[sx, sy]];
  if (!isFree(sx, sy) || !isFree(tx, ty)) return [[sx, sy]];
  const prev = new Int32Array(H * W);
  prev.fill(-2);
  const qx = new Int32Array(H * W);
  const qy = new Int32Array(H * W);
  let head = 0;
  let tail = 0;
  qx[tail] = sx;
  qy[tail] = sy;
  tail++;
  prev[key(sx, sy)] = -1;
  let found = false;
  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;
    if (x === tx && y === ty) {
      found = true;
      break;
    }
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (!isFree(nx, ny)) continue;
      const nk = key(nx, ny);
      if (prev[nk] !== -2) continue;
      prev[nk] = key(x, y);
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }
  if (!found) return [[sx, sy]];
  const rev: Cell[] = [];
  let cur = key(tx, ty);
  while (cur !== -1) {
    const x = cur % W;
    const y = Math.floor(cur / W);
    rev.push([x, y]);
    cur = prev[cur];
  }
  rev.reverse();
  if (rev.length > stepBudget + 1) return rev.slice(0, stepBudget + 1);
  return rev;
}

const positiveCells: { x: number; y: number; v: number }[] = [];
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    const v = prob[y][x];
    if (v > 0) positiveCells.push({ x, y, v });
  }
}
positiveCells.sort((a, b) => b.v - a.v);

const droneCount = drones.length;
const startDists: Int32Array[] = [];
for (const d of drones) startDists.push(bfs(d.x, d.y));

const assigned: { x: number; y: number; v: number }[][] = Array.from({ length: droneCount }, () => []);
for (const cell of positiveCells) {
  let bestDrone = 0;
  let bestScore = -1e100;
  for (let i = 0; i < droneCount; i++) {
    const dist = startDists[i][key(cell.x, cell.y)];
    if (dist < 0) continue;
    const score = cell.v * 1000 - dist * 3;
    if (score > bestScore) {
      bestScore = score;
      bestDrone = i;
    }
  }
  if (bestScore > -1e99) assigned[bestDrone].push(cell);
}

function pathForDrone(idx: number): Cell[] {
  const start = drones[idx];
  let curX = start.x;
  let curY = start.y;
  const path: Cell[] = [[curX, curY]];
  const used = new Set<number>();
  const candidates = assigned[idx].slice();
  candidates.sort((a, b) => b.v - a.v);

  let stepsLeft = maxSteps;
  while (stepsLeft > 0) {
    const distMap = bfs(curX, curY);
    let bestIndex = -1;
    let bestMetric = -1e100;
    let bestDist = 1e9;
    for (let i = 0; i < candidates.length; i++) {
      const c = candidates[i];
      const k = key(c.x, c.y);
      if (used.has(k)) continue;
      const d = distMap[k];
      if (d < 0 || d > stepsLeft) continue;
      const metric = c.v * 1000 - d * 5;
      if (metric > bestMetric || (metric === bestMetric && d < bestDist)) {
        bestMetric = metric;
        bestDist = d;
        bestIndex = i;
      }
    }
    if (bestIndex === -1) {
      while (stepsLeft > 0) {
        path.push([curX, curY]);
        stepsLeft--;
      }
      break;
    }

    const target = candidates[bestIndex];
    const seg = reconstructPath(curX, curY, target.x, target.y, stepsLeft);
    if (seg.length <= 1) {
      used.add(key(target.x, target.y));
      continue;
    }
    for (let i = 1; i < seg.length; i++) {
      path.push(seg[i]);
      curX = seg[i][0];
      curY = seg[i][1];
      stepsLeft--;
      if (stepsLeft === 0) break;
    }
    used.add(key(target.x, target.y));
  }

  return path;
}

const paths: Cell[][] = [];
for (let i = 0; i < droneCount; i++) {
  const p = pathForDrone(i);
  paths.push(p);
}

process.stdout.write(JSON.stringify({ paths }));
