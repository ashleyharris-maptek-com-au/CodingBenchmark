use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input)?;
        Ok(Self { buf: input, idx: 0 })
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut val: usize = 0;
        let mut found = false;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            found = true;
            val = val.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.idx += 1;
        }
        if found { Some(val) } else { None }
    }
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;
    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    let m = scanner.next_usize().unwrap_or(0);

    let mut matched = vec![false; n];
    let mut selected = vec![false; n];
    let mut cover = Vec::with_capacity(n.min(m.saturating_mul(2)));

    for _ in 0..m {
        let u = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };

        if u >= n || v >= n || u == v {
            if u < n && !selected[u] {
                selected[u] = true;
                cover.push(u);
            }
            if v < n && !selected[v] {
                selected[v] = true;
                cover.push(v);
            }
            continue;
        }

        if !matched[u] && !matched[v] {
            matched[u] = true;
            matched[v] = true;
            if !selected[u] {
                selected[u] = true;
                cover.push(u);
            }
            if !selected[v] {
                selected[v] = true;
                cover.push(v);
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", cover.len())?;
    for (i, v) in cover.iter().enumerate() {
        if i > 0 {
            write!(out, " ")?;
        }
        write!(out, "{}", v)?;
    }
    writeln!(out)?;
    Ok(())
}
