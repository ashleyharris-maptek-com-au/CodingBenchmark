using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string Next()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return null;
            } while (char.IsWhiteSpace((char)c));

            var sb = new StringBuilder();
            while (c >= 0 && !char.IsWhiteSpace((char)c))
            {
                sb.Append((char)c);
                c = ReadByte();
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            return int.Parse(Next());
        }
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int X;
        public readonly int Y;
        public readonly int LinesCleared;
        public readonly double Score;
        public readonly ulong[] Rows;

        public Move(int rotation, int x, int y, int linesCleared, double score, ulong[] rows)
        {
            Rotation = rotation;
            X = x;
            Y = y;
            LinesCleared = linesCleared;
            Score = score;
            Rows = rows;
        }
    }

    private sealed class Board
    {
        public readonly int Width;
        public readonly int Height;
        public readonly ulong Mask;
        public readonly ulong[] Rows;

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            Mask = width == 64 ? ulong.MaxValue : ((1UL << width) - 1UL);
            Rows = new ulong[height];
        }

        public Board Clone()
        {
            var b = new Board(Width, Height);
            Array.Copy(Rows, b.Rows, Rows.Length);
            return b;
        }

        public int ColumnHeights(out int holes, out int aggregateHeight, out int bumpiness)
        {
            int[] heights = new int[Width];
            holes = 0;
            aggregateHeight = 0;
            bumpiness = 0;

            for (int x = 0; x < Width; x++)
            {
                bool seenBlock = false;
                int h = 0;
                for (int y = Height - 1; y >= 0; y--)
                {
                    bool filled = ((Rows[y] >> x) & 1UL) != 0;
                    if (filled)
                    {
                        if (!seenBlock)
                        {
                            h = y + 1;
                            seenBlock = true;
                        }
                    }
                    else if (seenBlock)
                    {
                        holes++;
                    }
                }
                heights[x] = h;
                aggregateHeight += h;
            }

            for (int x = 0; x < Width - 1; x++) bumpiness += Math.Abs(heights[x] - heights[x + 1]);
            return MaxHeight(heights);
        }

        private int MaxHeight(int[] heights)
        {
            int m = 0;
            for (int i = 0; i < heights.Length; i++) if (heights[i] > m) m = heights[i];
            return m;
        }

        public bool TryPlace(PieceShape shape, int rotation, int x, out Move move)
        {
            move = default;
            int pw = shape.Width(rotation);
            int ph = shape.Height(rotation);
            if (x < 0 || x + pw > Width) return false;

            int y = FindDropY(shape, rotation, x, ph);
            if (y < 0 || y + ph > Height) return false;

            var next = new ulong[Height];
            Array.Copy(Rows, next, Height);

            for (int dy = 0; dy < ph; dy++)
            {
                ulong rowBits = shape.RowBits(rotation, dy) << x;
                int yy = y + (ph - 1 - dy);
                if ((next[yy] & rowBits) != 0) return false;
                next[yy] |= rowBits;
            }

            int cleared = 0;
            int write = 0;
            for (int read = 0; read < Height; read++)
            {
                if ((next[read] & Mask) == Mask)
                {
                    cleared++;
                    continue;
                }
                next[write++] = next[read];
            }
            while (write < Height) next[write++] = 0;

            double score = Evaluate(next, cleared);
            move = new Move(rotation, x, y, cleared, score, next);
            return true;
        }

        private int FindDropY(PieceShape shape, int rotation, int x, int ph)
        {
            int maxY = Height - ph;
            for (int y = 0; y <= maxY; y++)
            {
                bool collides = false;
                for (int row = 0; row < ph; row++)
                {
                    ulong bits = shape.RowBits(rotation, row) << x;
                    int boardRow = y + (ph - 1 - row);
                    if ((Rows[boardRow] & bits) != 0)
                    {
                        collides = true;
                        break;
                    }
                }
                if (collides) return y - 1;
            }
            return maxY;
        }

        private double Evaluate(ulong[] rows, int cleared)
        {
            int[] heights = new int[Width];
            int holes = 0;
            int agg = 0;
            int bump = 0;
            int maxH = 0;
            int[] rowFill = new int[Height];

            for (int y = 0; y < Height; y++)
            {
                ulong r = rows[y] & Mask;
                rowFill[y] = PopCount(r);
            }

            for (int x = 0; x < Width; x++)
            {
                bool seen = false;
                int h = 0;
                for (int y = Height - 1; y >= 0; y--)
                {
                    bool filled = ((rows[y] >> x) & 1UL) != 0;
                    if (filled)
                    {
                        if (!seen)
                        {
                            h = y + 1;
                            seen = true;
                        }
                    }
                    else if (seen)
                    {
                        holes++;
                    }
                }
                heights[x] = h;
                agg += h;
                if (h > maxH) maxH = h;
            }

            for (int x = 0; x < Width - 1; x++) bump += Math.Abs(heights[x] - heights[x + 1]);

            int wells = 0;
            for (int x = 0; x < Width; x++)
            {
                int left = x == 0 ? Height : heights[x - 1];
                int right = x == Width - 1 ? Height : heights[x + 1];
                int minAdj = Math.Min(left, right);
                if (minAdj > heights[x]) wells += minAdj - heights[x];
            }

            int completeNear = 0;
            for (int y = 0; y < Height; y++)
            {
                int filled = rowFill[y];
                if (filled >= Width - 2) completeNear += filled;
            }

            return cleared * 1000.0 - holes * 45.0 - agg * 4.0 - bump * 3.0 - wells * 6.0 - maxH * 2.0 + completeNear * 1.2;
        }
    }

    private sealed class PieceShape
    {
        private readonly int[][] _rowBits = new int[4][];
        private readonly int[] _width = new int[4];
        private readonly int[] _height = new int[4];

        public PieceShape(params string[] rotations)
        {
            for (int r = 0; r < 4; r++)
            {
                ParseRotation(rotations[Math.Min(r, rotations.Length - 1)], out _rowBits[r], out _width[r], out _height[r]);
            }
        }

        private static void ParseRotation(string s, out int[] rows, out int w, out int h)
        {
            var lines = s.Split('\n');
            h = lines.Length;
            w = 0;
            for (int i = 0; i < lines.Length; i++) if (lines[i].Length > w) w = lines[i].Length;
            rows = new int[h];
            for (int y = 0; y < h; y++)
            {
                int bits = 0;
                for (int x = 0; x < lines[y].Length; x++)
                {
                    if (lines[y][x] != '.') bits |= 1 << x;
                }
                rows[y] = bits;
            }
        }

        public int Width(int rotation) => _width[rotation & 3];
        public int Height(int rotation) => _height[rotation & 3];
        public ulong RowBits(int rotation, int row) => (ulong)_rowBits[rotation & 3][row];
    }

    private static readonly Dictionary<char, PieceShape> Pieces = new Dictionary<char, PieceShape>
    {
        ['I'] = new PieceShape(
            "####",
            "#\n#\n#\n#",
            "####",
            "#\n#\n#\n#"),
        ['O'] = new PieceShape(
            "##\n##",
            "##\n##",
            "##\n##",
            "##\n##"),
        ['T'] = new PieceShape(
            ".#.\n###",
            "#.\n##\n#.",
            "###\n.#.",
            ".#\n##\n.#"),
        ['S'] = new PieceShape(
            ".##\n##.",
            "#.\n##\n.#",
            ".##\n##.",
            "#.\n##\n.#"),
        ['Z'] = new PieceShape(
            "##.\n.##",
            ".#\n##\n#.",
            "##.\n.##",
            ".#\n##\n#."),
        ['J'] = new PieceShape(
            "#..\n###",
            "##\n#.\n#.",
            "###\n..#",
            ".#\n.#\n##"),
        ['L'] = new PieceShape(
            "..#\n###",
            "#.\n#.\n##",
            "###\n#..",
            "##\n.#\n.#")
    };

    private static int PopCount(ulong x)
    {
        int c = 0;
        while (x != 0)
        {
            x &= x - 1;
            c++;
        }
        return c;
    }

    public static void Main()
    {
        var input = new FastScanner();
        string wTok = input.Next();
        if (wTok == null) return;
        int width = int.Parse(wTok);
        int height = input.NextInt();
        int numPieces = input.NextInt();

        var board = new Board(width, height);
        var output = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        for (int i = 0; i < numPieces; i++)
        {
            string tok = input.Next();
            if (tok == null) break;
            char pieceChar = tok[0];
            var piece = Pieces[pieceChar];

            bool complex = CountFilled(board.Rows) > width * height / 4 || i > 40;
            Move best = default;
            bool found = false;
            double bestScore = double.NegativeInfinity;

            int rotationLimit = 4;
            for (int r = 0; r < rotationLimit; r++)
            {
                int pw = piece.Width(r);
                if (pw > width) continue;
                for (int x = 0; x <= width - pw; x++)
                {
                    if (!board.TryPlace(piece, r, x, out var move)) continue;
                    double score = move.Score;

                    if (complex)
                    {
                        // Small lookahead on simple boards: prefer moves that leave more options.
                        score += LookaheadBonus(board, pieceChar, move.Rows, width, height);
                    }

                    if (!found || score > bestScore)
                    {
                        found = true;
                        bestScore = score;
                        best = new Move(move.Rotation, move.X, move.Y, move.LinesCleared, score, move.Rows);
                    }
                }
            }

            if (!found)
            {
                output.WriteLine("0 0");
                output.Flush();
                string result = input.Next();
                if (result == null || result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;
                continue;
            }

            output.WriteLine($"{best.Rotation} {best.X}");
            output.Flush();

            string response = input.Next();
            if (response == null) break;
            if (response.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;

            board = new Board(width, height);
            Array.Copy(best.Rows, board.Rows, height);
        }
    }

    private static int CountFilled(ulong[] rows)
    {
        int c = 0;
        for (int i = 0; i < rows.Length; i++) c += PopCount(rows[i]);
        return c;
    }

    private static double LookaheadBonus(Board board, char current, ulong[] nextRows, int width, int height)
    {
        // Evaluate the resulting board against all possible next pieces with a cheap average.
        var nextBoard = new Board(width, height);
        Array.Copy(nextRows, nextBoard.Rows, height);

        char[] ps = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
        double sum = 0;
        int count = 0;
        for (int p = 0; p < ps.Length; p++)
        {
            var shape = Pieces[ps[p]];
            double localBest = double.NegativeInfinity;
            for (int r = 0; r < 4; r++)
            {
                int pw = shape.Width(r);
                if (pw > width) continue;
                for (int x = 0; x <= width - pw; x++)
                {
                    if (!nextBoard.TryPlace(shape, r, x, out var mv)) continue;
                    if (mv.Score > localBest) localBest = mv.Score;
                }
            }
            if (localBest > double.NegativeInfinity / 2)
            {
                sum += localBest;
                count++;
            }
        }
        return count == 0 ? 0 : sum / count * 0.02;
    }
}
