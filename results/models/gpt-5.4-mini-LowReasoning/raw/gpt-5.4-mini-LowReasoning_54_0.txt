from __future__ import annotations

import math


_state = {
    "go_around": False,
    "last_flaps": 0.0,
    "last_gear": 1.0,
    "last_spoiler": 0.0,
}


def _valid_value(state: dict, names):
    vals = []
    for name in names:
        v = state.get(name)
        if isinstance(v, bool):
            if v:
                vals.append(1.0)
        elif isinstance(v, (int, float)) and math.isfinite(v):
            vals.append(float(v))
    if not vals:
        return None
    vals.sort()
    return vals[len(vals) // 2]


def _vote_bool(state: dict, names, default=False):
    votes = []
    for name in names:
        v = state.get(name)
        if isinstance(v, bool):
            votes.append(v)
        elif isinstance(v, (int, float)):
            votes.append(v != 0)
    if not votes:
        return default
    return sum(1 for v in votes if v) >= (len(votes) / 2)


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _avg_valid(state: dict, value_keys, valid_keys=None, default=None):
    vals = []
    for i, k in enumerate(value_keys):
        v = state.get(k)
        ok = True
        if valid_keys is not None:
            vk = state.get(valid_keys[i])
            ok = bool(vk)
        if ok and isinstance(v, (int, float)) and math.isfinite(v):
            vals.append(float(v))
    if not vals:
        return default
    vals.sort()
    return sum(vals) / len(vals)


def _median_valid(state: dict, value_keys, valid_keys=None, default=None):
    vals = []
    for i, k in enumerate(value_keys):
        v = state.get(k)
        ok = True
        if valid_keys is not None:
            vk = state.get(valid_keys[i])
            ok = bool(vk)
        if ok and isinstance(v, (int, float)) and math.isfinite(v):
            vals.append(float(v))
    if not vals:
        return default
    vals.sort()
    return vals[len(vals) // 2]


def _wrap_deg(x):
    return (x + 180.0) % 360.0 - 180.0


def _choose_alt_ft(state: dict):
    ra = state.get("radio_alt_ft")
    if isinstance(ra, (int, float)) and math.isfinite(ra):
        return float(ra)
    return _median_valid(
        state,
        ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"],
        ["baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"],
        default=5000.0,
    )


def _choose_speed(state: dict):
    ias = _median_valid(
        state,
        ["ias_1_kt", "ias_2_kt", "ias_3_kt"],
        ["ias_1_valid", "ias_2_valid", "ias_3_valid"],
        default=None,
    )
    if ias is not None:
        return ias
    tas = _median_valid(state, ["tas_1_kt", "tas_2_kt", "tas_3_kt"], default=None)
    return tas if tas is not None else 140.0


def _choose_heading(state: dict):
    h = _median_valid(
        state,
        ["irs_1_heading_deg", "irs_2_heading_deg", "irs_3_heading_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=None,
    )
    if h is not None:
        return h
    t = _median_valid(
        state,
        ["irs_1_track_deg", "irs_2_track_deg", "irs_3_track_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=0.0,
    )
    return t


def autopilot_step(state: dict, dt: float) -> dict:
    alt = _choose_alt_ft(state)
    spd = _choose_speed(state)
    hdg = _choose_heading(state)
    target_ias = float(state.get("target_ias_kt", 137.0) or 137.0)
    target_vref = float(state.get("target_vref_kt", target_ias - 5.0) or (target_ias - 5.0))
    target_hdg = float(state.get("target_hdg_deg", hdg) or hdg)
    loc = _median_valid(
        state,
        ["localizer_1_dots", "localizer_2_dots", "localizer_3_dots"],
        ["localizer_1_valid", "localizer_2_valid", "localizer_3_valid"],
        default=0.0,
    ) or 0.0
    gs = _median_valid(
        state,
        ["glideslope_1_dots", "glideslope_2_dots", "glideslope_3_dots"],
        ["glideslope_1_valid", "glideslope_2_valid", "glideslope_3_valid"],
        default=0.0,
    ) or 0.0
    wow = _vote_bool(state, ["weight_on_wheels", "on_ground"], default=False)
    below_dh = _vote_bool(state, ["below_decision_height"], default=alt < 200.0)
    gs_kts = _avg_valid(
        state,
        ["gps_1_gs_kt", "gps_2_gs_kt", "gps_3_gs_kt", "irs_1_gs_kt", "irs_2_gs_kt", "irs_3_gs_kt"],
    )
    if gs_kts is None:
        gs_kts = spd

    # Faults and instability detection.
    engine_fail = (
        (state.get("engine_1_running") is False) or
        (state.get("engine_2_running") is False) or
        any(bool(state.get(k)) for k in ["engine_1_fire", "engine_2_fire"])
    )
    severe_crosswind = abs(loc) > 1.5 and alt < 500.0
    unstable_speed = spd > target_ias + 15.0 or spd < target_vref - 20.0
    unstable_path = abs(gs) > 1.5 or abs(loc) > 2.0
    rough_controls = any(abs(float(state.get(k, 0.0) or 0.0)) > 8.0 for k in [
        "irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"
    ])
    go_around = (
        (alt > 50.0 and (unstable_speed or unstable_path or severe_crosswind)) or
        (alt <= 50.0 and (unstable_speed and spd > target_vref + 25.0)) or
        (rough_controls and alt < 500.0 and not wow)
    )

    if state.get("radio_alt_ft") is None and alt < 150.0 and (unstable_path or unstable_speed):
        go_around = True

    if go_around and not wow:
        _state["go_around"] = True

    out = {}

    if wow:
        _state["go_around"] = False
        out["throttle_1"] = 0.0
        out["throttle_2"] = 0.0
        out["spoiler"] = 1.0
        out["reverse_1"] = 1.0 if spd < 120.0 else 0.7
        out["reverse_2"] = 1.0 if spd < 120.0 else 0.7
        out["autobrake"] = 1.0 if spd > 80.0 else 0.4
        out["gear"] = 1.0
        out["flaps"] = 40.0
        out["speedbrake"] = 1.0
        out["elevator"] = 0.2 if spd > 60.0 else 0.0
        out["aileron"] = _clamp(-loc * 0.4, -0.5, 0.5)
        out["rudder"] = _clamp(-_wrap_deg(target_hdg - hdg) / 45.0, -0.5, 0.5)
        return out

    if _state["go_around"]:
        out["throttle_1"] = 1.0
        out["throttle_2"] = 1.0
        out["elevator"] = 0.35
        out["elevator_trim"] = 0.15
        out["aileron"] = _clamp(-loc * 0.25, -0.5, 0.5)
        out["rudder"] = _clamp(-_wrap_deg(target_hdg - hdg) / 60.0, -0.4, 0.4)
        out["gear"] = 1.0
        out["flaps"] = 15.0
        out["speedbrake"] = 0.0
        out["spoiler"] = 0.0
        out["autobrake"] = 0.0
        if alt > 1500.0:
            _state["go_around"] = False
        return out

    # Config: keep landing configuration progressing with altitude.
    if alt > 2000.0:
        flap = 5.0
        gear = 0.0
    elif alt > 1000.0:
        flap = 15.0
        gear = 1.0
    elif alt > 500.0:
        flap = 20.0
        gear = 1.0
    elif alt > 100.0:
        flap = 30.0
        gear = 1.0
    else:
        flap = 40.0
        gear = 1.0

    # Speed target adapts to configuration and engine failures.
    target = max(target_vref, target_ias - 8.0)
    if engine_fail:
        target += 5.0
    if state.get("asymmetric_flap_extension"):
        target += 5.0

    # Vertical and lateral guidance.
    gs_err = -gs
    loc_err = -loc
    pitch_cmd = _clamp(0.08 * gs_err + 0.002 * (target - spd), -0.25, 0.25)
    if alt < 50.0:
        pitch_cmd = _clamp(0.18 + 0.002 * (target - spd), -0.1, 0.3)
    elif alt < 20.0:
        pitch_cmd = _clamp(0.12 + 0.0015 * (target - spd), -0.05, 0.25)

    roll_cmd = _clamp(0.25 * loc_err + _wrap_deg(target_hdg - hdg) / 90.0, -0.6, 0.6)
    rudder_cmd = _clamp(0.03 * _wrap_deg(target_hdg - hdg) + 0.08 * roll_cmd * 10.0, -0.7, 0.7)

    # Thrust schedule.
    if alt > 150.0:
        thrust = _clamp(0.35 + 0.015 * (target - spd) + 0.05 * max(gs, 0.0), 0.15, 0.9)
    elif alt > 50.0:
        thrust = _clamp(0.25 + 0.02 * (target - spd), 0.08, 0.55)
    else:
        thrust = _clamp(0.1 + 0.01 * (target - spd), 0.0, 0.25)

    # If the approach is clearly unstable close to the ground, try to salvage it
    # only by going around before flare.
    if alt < 100.0 and (spd > target + 20.0 or abs(loc) > 2.5 or abs(gs) > 2.0):
        out["throttle_1"] = 1.0
        out["throttle_2"] = 1.0
        out["elevator"] = 0.3
        out["elevator_trim"] = 0.1
        out["aileron"] = _clamp(roll_cmd, -0.6, 0.6)
        out["rudder"] = _clamp(rudder_cmd, -0.7, 0.7)
        out["gear"] = 1.0
        out["flaps"] = 15.0
        out["speedbrake"] = 0.0
        out["spoiler"] = 0.0
        out["autobrake"] = 0.0
        _state["go_around"] = True
        return out

    out["elevator"] = _clamp(pitch_cmd, -1.0, 1.0)
    out["aileron"] = _clamp(roll_cmd, -1.0, 1.0)
    out["rudder"] = _clamp(rudder_cmd, -1.0, 1.0)
    out["throttle_1"] = _clamp(thrust, 0.0, 1.0)
    out["throttle_2"] = _clamp(thrust, 0.0, 1.0)
    out["elevator_trim"] = _clamp(0.4 * pitch_cmd, -1.0, 1.0)
    out["flaps"] = flap
    out["gear"] = gear
    out["speedbrake"] = 0.0 if alt > 30.0 else 0.2
    out["spoiler"] = 0.0
    out["autobrake"] = 0.0

    if alt < 15.0 and spd < target_vref + 10.0:
        out["throttle_1"] = min(out["throttle_1"], 0.12)
        out["throttle_2"] = min(out["throttle_2"], 0.12)
        out["elevator"] = _clamp(0.16 + 0.002 * (target_vref - spd), 0.0, 0.3)

    return out
