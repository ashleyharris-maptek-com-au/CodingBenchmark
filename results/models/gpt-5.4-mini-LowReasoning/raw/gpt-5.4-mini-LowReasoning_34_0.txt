import sys
import math


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, pts, depot):
    if not route:
        return 0.0
    total = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        total += dist(pts[route[i]], pts[route[i + 1]])
    total += dist(pts[route[-1]], depot)
    return total


def two_opt(route, pts, depot):
    n = len(route)
    if n < 4:
        return route

    def node_at(pos):
        if pos == -1:
            return depot
        if pos == n:
            return depot
        return pts[route[pos]]

    improved = True
    while improved:
        improved = False
        for i in range(n - 1):
            a = node_at(i - 1)
            b = node_at(i)
            for k in range(i + 1, n):
                c = node_at(k)
                d = node_at(k + 1)
                delta = dist(a, c) + dist(b, d) - dist(a, b) - dist(c, d)
                if delta < -1e-9:
                    route[i : k + 1] = reversed(route[i : k + 1])
                    improved = True
                    break
            if improved:
                break
    return route


def build_initial_routes(customers, pts, depot, demands, cap):
    routes = []
    current = []
    load = 0
    for c in customers:
        d = demands[c]
        if current and load + d > cap:
            routes.append(two_opt(current, pts, depot))
            current = [c]
            load = d
        else:
            current.append(c)
            load += d
    if current:
        routes.append(two_opt(current, pts, depot))
    return routes


def solve():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = int(next(it))
    depot = (float(next(it)), float(next(it)))
    pts = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        pts.append((x, y))
        demands.append(d)

    if n == 0:
        print("0")
        print("0")
        return

    customers = list(range(n))
    customers.sort(key=lambda i: (math.atan2(pts[i][1] - depot[1], pts[i][0] - depot[0]), dist(depot, pts[i])))

    routes = build_initial_routes(customers, pts, depot, demands, cap)

    total = 0.0
    if len(routes) > k:
        # This should not happen when the instance is feasible, but keep a safe fallback.
        routes = sorted(routes, key=len, reverse=True)
        merged = []
        for route in routes:
            placed = False
            for i in range(len(merged)):
                if sum(demands[c] for c in merged[i]) + sum(demands[c] for c in route) <= cap:
                    merged[i].extend(route)
                    placed = True
                    break
            if not placed:
                merged.append(route)
        routes = [two_opt(r, pts, depot) for r in merged]

    for route in routes:
        total += route_distance(route, pts, depot)

    print(f"{total:.10f}")
    print(len(routes))
    for route in routes:
        print(" ".join(str(c) for c in route))


if __name__ == "__main__":
    solve()
