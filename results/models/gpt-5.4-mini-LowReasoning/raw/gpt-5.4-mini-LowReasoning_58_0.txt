const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid || [];
const H = grid.length;
const W = H ? grid[0].length : 0;
const robots = data.robots || [];
const jobs = data.jobs || [];

function idx(x, y) {
  return y * W + x;
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function isOpen(x, y) {
  return inBounds(x, y) && grid[y].charCodeAt(x) !== 35; // '#'
}

const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

function bfs(startX, startY) {
  const n = W * H;
  const dist = new Int32Array(n);
  dist.fill(-1);
  if (!isOpen(startX, startY)) return dist;

  const qx = new Int32Array(n);
  const qy = new Int32Array(n);
  let qh = 0;
  let qt = 0;
  const s = idx(startX, startY);
  dist[s] = 0;
  qx[qt] = startX;
  qy[qt] = startY;
  qt++;

  while (qh < qt) {
    const x = qx[qh];
    const y = qy[qh];
    qh++;
    const base = dist[idx(x, y)] + 1;
    for (let i = 0; i < 4; i++) {
      const nx = x + dirs[i][0];
      const ny = y + dirs[i][1];
      if (!isOpen(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (dist[ni] !== -1) continue;
      dist[ni] = base;
      qx[qt] = nx;
      qy[qt] = ny;
      qt++;
    }
  }
  return dist;
}

const robotDists = robots.map(r => bfs(r.x, r.y));

const jobInfo = jobs.map((j, i) => ({
  index: i,
  id: j.id,
  x: j.x,
  y: j.y,
  priority: j.priority == null ? 0 : j.priority,
}));

const assignments = Array.from({ length: robots.length }, () => []);
const robotLoad = new Float64Array(robots.length);
const robotJobCounts = new Int32Array(robots.length);

// Assign higher priority jobs first, then by closeness to the best robot.
jobInfo.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.index - b.index;
});

for (const job of jobInfo) {
  let bestRobot = 0;
  let bestScore = Infinity;
  const jobCell = idx(job.x, job.y);

  for (let r = 0; r < robots.length; r++) {
    const d = robotDists[r][jobCell];
    if (d < 0) continue;
    const score = d + robotLoad[r] * 0.75 + robotJobCounts[r] * 0.5 - job.priority * 0.1;
    if (score < bestScore || (score === bestScore && r < bestRobot)) {
      bestScore = score;
      bestRobot = r;
    }
  }

  if (!isFinite(bestScore)) {
    // Fallback for unreachable jobs: keep deterministic placement.
    bestRobot = job.index % Math.max(1, robots.length);
  }

  assignments[bestRobot].push(job);
  robotJobCounts[bestRobot]++;
  if (isFinite(bestScore)) robotLoad[bestRobot] += bestScore;
}

function manhattan(a, b) {
  return Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
}

// Order each robot's jobs to reduce travel. Use a nearest-neighbor chain with
// priority as a secondary tie-breaker.
for (let r = 0; r < robots.length; r++) {
  const list = assignments[r];
  if (list.length <= 1) continue;
  const ordered = [];
  const remaining = list.slice();
  let cx = robots[r].x;
  let cy = robots[r].y;

  while (remaining.length) {
    let best = 0;
    let bestDist = Infinity;
    let bestPr = -Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const job = remaining[i];
      const d = Math.abs(job.x - cx) + Math.abs(job.y - cy);
      if (d < bestDist || (d === bestDist && (job.priority > bestPr || (job.priority === bestPr && job.id < remaining[best].id)))) {
        best = i;
        bestDist = d;
        bestPr = job.priority;
      }
    }
    const chosen = remaining.splice(best, 1)[0];
    ordered.push(chosen);
    cx = chosen.x;
    cy = chosen.y;
  }

  assignments[r] = ordered;
}

const output = {
  assignments: assignments.map(list => list.map(job => job.id)),
};

process.stdout.write(JSON.stringify(output));
