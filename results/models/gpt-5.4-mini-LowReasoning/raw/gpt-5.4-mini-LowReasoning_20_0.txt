use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct HoleMeta {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    sample_count: usize,
    sample_indices: Vec<usize>,
}

#[derive(Clone)]
struct Sample {
    hole: usize,
    idx_in_hole: usize,
    depth: f64,
    point: [f64; 3],
    values: Vec<Option<f64>>,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

struct Tokenizer<'a> {
    it: std::str::SplitWhitespace<'a>,
}

impl<'a> Tokenizer<'a> {
    fn new(s: &'a str) -> Self {
        Self { it: s.split_whitespace() }
    }

    fn next_str(&mut self) -> Option<&'a str> {
        self.it.next()
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_str()?.parse().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_str()?.parse().ok()
    }
}

fn median(sorted: &[f64]) -> f64 {
    let n = sorted.len();
    if n == 0 {
        return f64::NAN;
    }
    if n % 2 == 1 {
        sorted[n / 2]
    } else {
        (sorted[n / 2 - 1] + sorted[n / 2]) * 0.5
    }
}

fn quantile_sorted(sorted: &[f64], q: f64) -> f64 {
    if sorted.is_empty() {
        return f64::NAN;
    }
    let n = sorted.len();
    if n == 1 {
        return sorted[0];
    }
    let pos = q.clamp(0.0, 1.0) * (n as f64 - 1.0);
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        sorted[lo]
    } else {
        let t = pos - lo as f64;
        sorted[lo] * (1.0 - t) + sorted[hi] * t
    }
}

fn median_abs_dev(values: &[f64], center: f64) -> f64 {
    let mut devs = Vec::with_capacity(values.len());
    for &v in values {
        devs.push((v - center).abs());
    }
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    median(&devs)
}

fn robust_scale(values: &[f64], center: f64) -> f64 {
    let mad = median_abs_dev(values, center);
    if mad.is_finite() && mad > 1e-12 {
        1.4826 * mad
    } else {
        let mut sorted = values.to_vec();
        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        let q1 = quantile_sorted(&sorted, 0.25);
        let q3 = quantile_sorted(&sorted, 0.75);
        let iqr = q3 - q1;
        if iqr.is_finite() && iqr > 1e-12 {
            iqr / 1.349
        } else {
            let mean = values.iter().sum::<f64>() / values.len().max(1) as f64;
            let var = values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len().max(1) as f64;
            var.sqrt().max(1e-12)
        }
    }
}

fn confidence_from_score(score: f64) -> f64 {
    if !score.is_finite() || score <= 0.0 {
        0.0
    } else {
        1.0 - (-score / 3.0).exp()
    }
}

fn compute_local_score(
    samples: &[Sample],
    holes: &[HoleMeta],
    prop: usize,
    window: usize,
) -> Vec<f64> {
    let mut scores = vec![0.0; samples.len()];
    let mut neighborhood = Vec::with_capacity(window * 2 + 1);
    for hole in holes {
        let refs = &hole.sample_indices;
        if refs.len() < 3 {
            continue;
        }
        for (pos, &si) in refs.iter().enumerate() {
            neighborhood.clear();
            let start = pos.saturating_sub(window);
            let end = (pos + window + 1).min(refs.len());
            for j in start..end {
                if j == pos {
                    continue;
                }
                if let Some(v) = samples[refs[j]].values[prop] {
                    neighborhood.push(v);
                }
            }
            if neighborhood.len() < 2 {
                continue;
            }
            neighborhood.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
            let center = median(&neighborhood);
            let scale = robust_scale(&neighborhood, center).max(1e-12);
            if let Some(v) = samples[si].values[prop] {
                let z = (v - center).abs() / scale;
                scores[si] = z;
            }
        }
    }
    scores
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tk = Tokenizer::new(&input);

    let num_holes = match tk.next_usize() {
        Some(v) => v,
        None => return,
    };
    let num_props = match tk.next_usize() {
        Some(v) => v,
        None => return,
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(tk.next_str().unwrap_or("").to_string());
    }

    let mut holes: Vec<HoleMeta> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    let mut hole_lookup: HashMap<String, usize> = HashMap::new();

    for hole_idx in 0..num_holes {
        let hole_id = tk.next_str().unwrap_or("").to_string();
        let sx = tk.next_f64().unwrap_or(0.0);
        let sy = tk.next_f64().unwrap_or(0.0);
        let sz = tk.next_f64().unwrap_or(0.0);
        let dx = tk.next_f64().unwrap_or(0.0);
        let dy = tk.next_f64().unwrap_or(0.0);
        let dz = tk.next_f64().unwrap_or(0.0);
        let length = tk.next_f64().unwrap_or(0.0);
        let sample_count = tk.next_usize().unwrap_or(0);

        hole_lookup.insert(hole_id.clone(), hole_idx);
        holes.push(HoleMeta {
            id: hole_id,
            start: [sx, sy, sz],
            dir: [dx, dy, dz],
            length,
            sample_count,
            sample_indices: Vec::with_capacity(sample_count),
        });

        for sidx in 0..sample_count {
            let depth = tk.next_f64().unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                let tok = tk.next_str().unwrap_or("");
                let v = tok.parse::<f64>().ok().filter(|x| x.is_finite());
                values.push(v);
            }
            let h = &holes[hole_idx];
            let point = [
                h.start[0] + h.dir[0] * depth,
                h.start[1] + h.dir[1] * depth,
                h.start[2] + h.dir[2] * depth,
            ];
            samples.push(Sample {
                hole: hole_idx,
                idx_in_hole: sidx,
                depth,
                point,
                values,
            });
            holes[hole_idx].sample_indices.push(samples.len() - 1);
        }
    }

    for hole in &mut holes {
        hole.sample_indices.sort_by(|&a, &b| {
            samples[a]
                .depth
                .partial_cmp(&samples[b].depth)
                .unwrap_or(Ordering::Equal)
        });
    }

    let total_values = samples.len() * num_props;
    let large_scale = total_values > 1_500_000 || samples.len() > 250_000;
    let local_window = if large_scale { 3 } else { 5 };

    let mut suspects: Vec<Suspect> = Vec::new();

    for prop in 0..num_props {
        let mut vals = Vec::new();
        vals.reserve(samples.len().min(1_000_000));
        for s in &samples {
            if let Some(v) = s.values[prop] {
                vals.push(v);
            }
        }
        if vals.len() < 5 {
            continue;
        }
        vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        let center = median(&vals);
        let scale = robust_scale(&vals, center).max(1e-12);

        let mut global_scores = vec![0.0; samples.len()];
        for (i, s) in samples.iter().enumerate() {
            if let Some(v) = s.values[prop] {
                global_scores[i] = (v - center).abs() / scale;
            }
        }

        let mut local_scores = vec![0.0; samples.len()];
        local_scores = compute_local_score(&samples, &holes, prop, if large_scale { 2 } else { local_window });

        let mut prop_scores = Vec::new();
        for i in 0..samples.len() {
            let score = global_scores[i].max(local_scores[i]);
            if score >= 4.0 {
                prop_scores.push((i, score));
            }
        }

        if prop_scores.is_empty() {
            let mut top: Vec<(usize, f64)> = (0..samples.len())
                .filter_map(|i| {
                    let s = global_scores[i].max(local_scores[i]);
                    if s.is_finite() {
                        Some((i, s))
                    } else {
                        None
                    }
                })
                .collect();
            top.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(Ordering::Equal));
            top.truncate(top.len().min(3));
            prop_scores = top;
        }

        for (i, score) in prop_scores {
            let conf = confidence_from_score(score);
            if conf < 0.5 {
                continue;
            }
            let sample = &samples[i];
            suspects.push(Suspect {
                hole_id: holes[sample.hole].id.clone(),
                sample_idx: sample.idx_in_hole,
                property_name: prop_names[prop].clone(),
                confidence: conf,
            });
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    if suspects.len() > 10_000 {
        suspects.truncate(10_000);
    }

    println!("{}", suspects.len());
    for s in suspects {
        println!(
            "{} {} {} {:.6}",
            s.hole_id, s.sample_idx, s.property_name, s.confidence
        );
    }
}
