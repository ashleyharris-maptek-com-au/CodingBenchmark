OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_4 = OpConstant %float 4
%float_5 = OpConstant %float 5
%float_02 = OpConstant %float 0.02

%glsl = OpExtInstImport "GLSL.std.450"

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%_ptr_Function_v4float = OpTypePointer Function %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_v2float = OpTypePointer Function %v2float

%main = OpFunction %void None %func
%entry = OpLabel

%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1 = OpLoad %v4float %p1ptr
%p2 = OpLoad %v4float %p2ptr
%p3 = OpLoad %v4float %p3ptr

%c1ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1ptr
%c2 = OpLoad %v3float %c2ptr

%tan0_4 = OpFSub %v4float %p2 %p0
%tan1_4 = OpFSub %v4float %p3 %p1
%tan0_xy = OpVectorShuffle %v2float %tan0_4 %tan0_4 0 1
%tan1_xy = OpVectorShuffle %v2float %tan1_4 %tan1_4 0 1
%tan0_n = OpExtInst %v2float %glsl Normalize %tan0_xy
%tan1_n = OpExtInst %v2float %glsl Normalize %tan1_xy

%t0 = OpConstant %float 0
%t1 = OpConstant %float 0.25
%t2 = OpConstant %float 0.5
%t3 = OpConstant %float 0.75
%t4 = OpConstant %float 1

; point 0
%tanmix0 = OpExtInst %v2float %glsl FMix %tan0_n %tan1_n %t0
%tanmix0_n = OpExtInst %v2float %glsl Normalize %tanmix0
%perp0x = OpCompositeExtract %float %tanmix0_n 1
%perp0y = OpCompositeExtract %float %tanmix0_n 0
%perp0yneg = OpFNegate %float %perp0y
%perp0v2 = OpCompositeConstruct %v2float %perp0yneg %perp0x
%perp0s = OpVectorTimesScalar %v2float %perp0v2 %float_02
%pos0 = OpExtInst %v4float %glsl FMix %p1 %p2 %t0
%perp0 = OpCompositeConstruct %v4float %perp0s %float_0 %float_0
%pos0a = OpFAdd %v4float %pos0 %perp0
%pos0b = OpFSub %v4float %pos0 %perp0
%col0 = OpExtInst %v3float %glsl FMix %c1 %c2 %t0
%p0outptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p0outptr %pos0a
OpStore %outColor %col0
OpEmitVertex
OpStore %p0outptr %pos0b
OpStore %outColor %col0
OpEmitVertex

; point 1
%tanmix1 = OpExtInst %v2float %glsl FMix %tan0_n %tan1_n %t1
%tanmix1_n = OpExtInst %v2float %glsl Normalize %tanmix1
%perp1x = OpCompositeExtract %float %tanmix1_n 1
%perp1y = OpCompositeExtract %float %tanmix1_n 0
%perp1yneg = OpFNegate %float %perp1y
%perp1v2 = OpCompositeConstruct %v2float %perp1yneg %perp1x
%perp1s = OpVectorTimesScalar %v2float %perp1v2 %float_02
%pos1 = OpExtInst %v4float %glsl FMix %p1 %p2 %t1
%perp1 = OpCompositeConstruct %v4float %perp1s %float_0 %float_0
%pos1a = OpFAdd %v4float %pos1 %perp1
%pos1b = OpFSub %v4float %pos1 %perp1
%col1 = OpExtInst %v3float %glsl FMix %c1 %c2 %t1
OpStore %p0outptr %pos1a
OpStore %outColor %col1
OpEmitVertex
OpStore %p0outptr %pos1b
OpStore %outColor %col1
OpEmitVertex

; point 2
%tanmix2 = OpExtInst %v2float %glsl FMix %tan0_n %tan1_n %t2
%tanmix2_n = OpExtInst %v2float %glsl Normalize %tanmix2
%perp2x = OpCompositeExtract %float %tanmix2_n 1
%perp2y = OpCompositeExtract %float %tanmix2_n 0
%perp2yneg = OpFNegate %float %perp2y
%perp2v2 = OpCompositeConstruct %v2float %perp2yneg %perp2x
%perp2s = OpVectorTimesScalar %v2float %perp2v2 %float_02
%pos2 = OpExtInst %v4float %glsl FMix %p1 %p2 %t2
%perp2 = OpCompositeConstruct %v4float %perp2s %float_0 %float_0
%pos2a = OpFAdd %v4float %pos2 %perp2
%pos2b = OpFSub %v4float %pos2 %perp2
%col2 = OpExtInst %v3float %glsl FMix %c1 %c2 %t2
OpStore %p0outptr %pos2a
OpStore %outColor %col2
OpEmitVertex
OpStore %p0outptr %pos2b
OpStore %outColor %col2
OpEmitVertex

; point 3
%tanmix3 = OpExtInst %v2float %glsl FMix %tan0_n %tan1_n %t3
%tanmix3_n = OpExtInst %v2float %glsl Normalize %tanmix3
%perp3x = OpCompositeExtract %float %tanmix3_n 1
%perp3y = OpCompositeExtract %float %tanmix3_n 0
%perp3yneg = OpFNegate %float %perp3y
%perp3v2 = OpCompositeConstruct %v2float %perp3yneg %perp3x
%perp3s = OpVectorTimesScalar %v2float %perp3v2 %float_02
%pos3 = OpExtInst %v4float %glsl FMix %p1 %p2 %t3
%perp3 = OpCompositeConstruct %v4float %perp3s %float_0 %float_0
%pos3a = OpFAdd %v4float %pos3 %perp3
%pos3b = OpFSub %v4float %pos3 %perp3
%col3 = OpExtInst %v3float %glsl FMix %c1 %c2 %t3
OpStore %p0outptr %pos3a
OpStore %outColor %col3
OpEmitVertex
OpStore %p0outptr %pos3b
OpStore %outColor %col3
OpEmitVertex

; point 4
%tanmix4 = OpExtInst %v2float %glsl FMix %tan0_n %tan1_n %t4
%tanmix4_n = OpExtInst %v2float %glsl Normalize %tanmix4
%perp4x = OpCompositeExtract %float %tanmix4_n 1
%perp4y = OpCompositeExtract %float %tanmix4_n 0
%perp4yneg = OpFNegate %float %perp4y
%perp4v2 = OpCompositeConstruct %v2float %perp4yneg %perp4x
%perp4s = OpVectorTimesScalar %v2float %perp4v2 %float_02
%pos4 = OpExtInst %v4float %glsl FMix %p1 %p2 %t4
%perp4 = OpCompositeConstruct %v4float %perp4s %float_0 %float_0
%pos4a = OpFAdd %v4float %pos4 %perp4
%pos4b = OpFSub %v4float %pos4 %perp4
%col4 = OpExtInst %v3float %glsl FMix %c1 %c2 %t4
OpStore %p0outptr %pos4a
OpStore %outColor %col4
OpEmitVertex
OpStore %p0outptr %pos4b
OpStore %outColor %col4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
