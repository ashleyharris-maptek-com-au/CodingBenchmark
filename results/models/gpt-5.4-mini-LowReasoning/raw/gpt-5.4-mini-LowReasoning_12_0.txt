using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }
    }

    private readonly struct Mat3
    {
        public readonly int M00, M01, M02;
        public readonly int M10, M11, M12;
        public readonly int M20, M21, M22;

        public Mat3(int m00, int m01, int m02, int m10, int m11, int m12, int m20, int m21, int m22)
        {
            M00 = m00; M01 = m01; M02 = m02;
            M10 = m10; M11 = m11; M12 = m12;
            M20 = m20; M21 = m21; M22 = m22;
        }

        public Vec3 Transform(in Vec3 v)
        {
            return new Vec3(
                M00 * v.X + M01 * v.Y + M02 * v.Z,
                M10 * v.X + M11 * v.Y + M12 * v.Z,
                M20 * v.X + M21 * v.Y + M22 * v.Z);
        }
    }

    private readonly struct Candidate
    {
        public readonly Mat3 Matrix;
        public readonly double Qw, Qx, Qy, Qz;
        public readonly double W, H, D;
        public readonly long Count;
        public readonly double Waste;

        public Candidate(Mat3 matrix, double qw, double qx, double qy, double qz, double w, double h, double d, long count, double waste)
        {
            Matrix = matrix;
            Qw = qw; Qx = qx; Qy = qy; Qz = qz;
            W = w; H = h; D = d;
            Count = count;
            Waste = waste;
        }
    }

    public static void Main()
    {
        var tokens = ReadAllTokens();
        int p = 0;
        int vCount = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        int fCount = int.Parse(tokens[p++], CultureInfo.InvariantCulture);

        var vertices = new Vec3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            double x = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            double y = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            double z = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            vertices[i] = new Vec3(x, y, z);
        }

        // Faces are not needed for the conservative packing strategy.
        p += fCount * 3;

        double containerW = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double containerH = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double containerD = double.Parse(tokens[p++], CultureInfo.InvariantCulture);

        var rotations = BuildProperAxisRotations();
        var candidates = new List<Candidate>(24);

        foreach (var rot in rotations)
        {
            GetRotatedBounds(vertices, rot, out double bw, out double bh, out double bd);
            if (bw <= 0 || bh <= 0 || bd <= 0)
                continue;

            long nx = (long)Math.Floor(containerW / bw);
            long ny = (long)Math.Floor(containerH / bh);
            long nz = (long)Math.Floor(containerD / bd);
            long count = nx <= 0 || ny <= 0 || nz <= 0 ? 0 : nx * ny * nz;
            double waste = (containerW - nx * bw) + (containerH - ny * bh) + (containerD - nz * bd);
            var q = QuaternionFromMatrix(rot);
            candidates.Add(new Candidate(rot, q[0], q[1], q[2], q[3], bw, bh, bd, count, waste));
        }

        if (candidates.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        candidates.Sort((a, b) =>
        {
            int cmp = b.Count.CompareTo(a.Count);
            if (cmp != 0) return cmp;
            cmp = a.Waste.CompareTo(b.Waste);
            if (cmp != 0) return cmp;
            double av = a.W * a.H * a.D;
            double bv = b.W * b.H * b.D;
            return av.CompareTo(bv);
        });

        var best = candidates[0];
        long xCount = (long)Math.Floor(containerW / best.W);
        long yCount = (long)Math.Floor(containerH / best.H);
        long zCount = (long)Math.Floor(containerD / best.D);
        long total = xCount * yCount * zCount;

        Console.WriteLine(total.ToString(CultureInfo.InvariantCulture));
        if (total == 0)
            return;

        string q = Format(best.Qw) + " " + Format(best.Qx) + " " + Format(best.Qy) + " " + Format(best.Qz);
        for (long ix = 0; ix < xCount; ix++)
        {
            double tx = ix * best.W;
            for (long iy = 0; iy < yCount; iy++)
            {
                double ty = iy * best.H;
                for (long iz = 0; iz < zCount; iz++)
                {
                    double tz = iz * best.D;
                    Console.WriteLine(Format(tx) + " " + Format(ty) + " " + Format(tz) + " " + q);
                }
            }
        }
    }

    private static string[] ReadAllTokens()
    {
        var data = Console.In.ReadToEnd();
        return data.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
    }

    private static string Format(double value) => value.ToString("0.######", CultureInfo.InvariantCulture);

    private static void GetRotatedBounds(Vec3[] vertices, Mat3 rot, out double width, out double height, out double depth)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        for (int i = 0; i < vertices.Length; i++)
        {
            Vec3 v = rot.Transform(vertices[i]);
            if (v.X < minX) minX = v.X;
            if (v.Y < minY) minY = v.Y;
            if (v.Z < minZ) minZ = v.Z;
            if (v.X > maxX) maxX = v.X;
            if (v.Y > maxY) maxY = v.Y;
            if (v.Z > maxZ) maxZ = v.Z;
        }

        width = maxX - minX;
        height = maxY - minY;
        depth = maxZ - minZ;
    }

    private static List<Mat3> BuildProperAxisRotations()
    {
        var result = new List<Mat3>(24);
        int[][] perms =
        {
            new[] { 0, 1, 2 },
            new[] { 0, 2, 1 },
            new[] { 1, 0, 2 },
            new[] { 1, 2, 0 },
            new[] { 2, 0, 1 },
            new[] { 2, 1, 0 },
        };

        int[][] signs =
        {
            new[] { 1, 1, 1 },
            new[] { 1, -1, -1 },
            new[] { -1, 1, -1 },
            new[] { -1, -1, 1 },
        };

        foreach (var p in perms)
        {
            int parity = PermutationParity(p);
            foreach (var s in signs)
            {
                int detSign = parity * s[0] * s[1] * s[2];
                if (detSign != 1)
                    continue;

                var cols = new int[3][];
                for (int c = 0; c < 3; c++)
                {
                    cols[c] = new int[3];
                    cols[c][p[c]] = s[c];
                }

                result.Add(new Mat3(
                    cols[0][0], cols[1][0], cols[2][0],
                    cols[0][1], cols[1][1], cols[2][1],
                    cols[0][2], cols[1][2], cols[2][2]));
            }
        }

        return result;
    }

    private static int PermutationParity(int[] p)
    {
        int inversions = 0;
        for (int i = 0; i < p.Length; i++)
            for (int j = i + 1; j < p.Length; j++)
                if (p[i] > p[j])
                    inversions++;
        return (inversions % 2 == 0) ? 1 : -1;
    }

    private static double[] QuaternionFromMatrix(Mat3 m)
    {
        double trace = m.M00 + m.M11 + m.M22;
        double qw, qx, qy, qz;

        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            qw = 0.25 * s;
            qx = (m.M21 - m.M12) / s;
            qy = (m.M02 - m.M20) / s;
            qz = (m.M10 - m.M01) / s;
        }
        else if (m.M00 > m.M11 && m.M00 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
            qw = (m.M21 - m.M12) / s;
            qx = 0.25 * s;
            qy = (m.M01 + m.M10) / s;
            qz = (m.M02 + m.M20) / s;
        }
        else if (m.M11 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
            qw = (m.M02 - m.M20) / s;
            qx = (m.M01 + m.M10) / s;
            qy = 0.25 * s;
            qz = (m.M12 + m.M21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
            qw = (m.M10 - m.M01) / s;
            qx = (m.M02 + m.M20) / s;
            qy = (m.M12 + m.M21) / s;
            qz = 0.25 * s;
        }

        // Normalize to suppress tiny numeric drift.
        double norm = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        return new[] { qw / norm, qx / norm, qy / norm, qz / norm };
    }
}
