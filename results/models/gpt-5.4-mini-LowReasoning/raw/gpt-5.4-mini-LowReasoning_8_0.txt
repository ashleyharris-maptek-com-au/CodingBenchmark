use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.is_empty() {
        return;
    }

    let mut lines = input.lines();
    let first = match lines.next() {
        Some(v) => v,
        None => return,
    };
    let mut dims = first.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();

    let mut grid = vec![b'#'; h * w];
    let mut start = None;
    let mut goal = None;

    for r in 0..h {
        let line = match lines.next() {
            Some(v) => v.as_bytes(),
            None => break,
        };
        for c in 0..w.min(line.len()) {
            let ch = line[c];
            let pos = r * w + c;
            grid[pos] = ch;
            if ch == b'A' {
                start = Some(pos);
            } else if ch == b'B' {
                goal = Some(pos);
            }
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let n = h * w;
    let mut parent = vec![usize::MAX; n];
    let mut q = VecDeque::new();
    parent[start] = start;
    q.push_back(start);

    while let Some(v) = q.pop_front() {
        if v == goal {
            break;
        }
        let r = v / w;
        let c = v % w;

        if r > 0 {
            let nv = v - w;
            if parent[nv] == usize::MAX && grid[nv] != b'#' {
                parent[nv] = v;
                q.push_back(nv);
            }
        }
        if r + 1 < h {
            let nv = v + w;
            if parent[nv] == usize::MAX && grid[nv] != b'#' {
                parent[nv] = v;
                q.push_back(nv);
            }
        }
        if c > 0 {
            let nv = v - 1;
            if parent[nv] == usize::MAX && grid[nv] != b'#' {
                parent[nv] = v;
                q.push_back(nv);
            }
        }
        if c + 1 < w {
            let nv = v + 1;
            if parent[nv] == usize::MAX && grid[nv] != b'#' {
                parent[nv] = v;
                q.push_back(nv);
            }
        }
    }

    if parent[goal] == usize::MAX {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = parent[cur];
    }
    path.reverse();

    println!("{}", path.len());
    for p in path {
        println!("{} {}", p / w, p % w);
    }
}
