using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private struct Box
    {
        public int W, H, D, Index;
        public long Volume;
        public Box(int w, int h, int d, int index)
        {
            W = w; H = h; D = d; Index = index;
            Volume = (long)w * h * d;
        }
    }

    private struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public Placement(int index, int x, int y, int z)
        {
            Index = index; X = x; Y = y; Z = z;
        }
    }

    private struct Space
    {
        public int X, Y, Z, W, H, D;
        public long Volume;
        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x; Y = y; Z = z; W = w; H = h; D = d;
            Volume = (long)w * h * d;
        }
    }

    private readonly struct Orient
    {
        public readonly int W, H, D;
        public Orient(int w, int h, int d) { W = w; H = h; D = d; }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int cw = fs.NextInt();
        int ch = fs.NextInt();
        int cd = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            boxes[i] = new Box(w, h, d, i);
        }

        Array.Sort(boxes, CompareBoxes);

        var spaces = new List<Space>(Math.Max(4, n * 2));
        spaces.Add(new Space(0, 0, 0, cw, ch, cd));

        var placements = new List<Placement>(n);

        for (int bi = 0; bi < n; bi++)
        {
            ref readonly Box box = ref boxes[bi];
            int bestSpace = -1;
            Orient bestOrient = default;
            long bestWaste = long.MaxValue;
            int bestExtent = int.MaxValue;

            Span<Orient> orients = stackalloc Orient[6]
            {
                new Orient(box.W, box.H, box.D),
                new Orient(box.W, box.D, box.H),
                new Orient(box.H, box.W, box.D),
                new Orient(box.H, box.D, box.W),
                new Orient(box.D, box.W, box.H),
                new Orient(box.D, box.H, box.W)
            };

            for (int si = 0; si < spaces.Count; si++)
            {
                ref readonly Space sp = ref spaces[si];
                if (sp.Volume < box.Volume) continue;

                for (int oi = 0; oi < orients.Length; oi++)
                {
                    Orient o = orients[oi];
                    if (o.W > sp.W || o.H > sp.H || o.D > sp.D) continue;

                    long waste = sp.Volume - (long)o.W * o.H * o.D;
                    int extent = Math.Max(sp.W - o.W, Math.Max(sp.H - o.H, sp.D - o.D));

                    if (waste < bestWaste ||
                        (waste == bestWaste && extent < bestExtent))
                    {
                        bestWaste = waste;
                        bestExtent = extent;
                        bestSpace = si;
                        bestOrient = o;
                    }
                }
            }

            if (bestSpace < 0)
                continue;

            Space used = spaces[bestSpace];
            int bw = bestOrient.W;
            int bh = bestOrient.H;
            int bd = bestOrient.D;

            placements.Add(new Placement(box.Index, used.X, used.Y, used.Z));

            // Remove the used space and replace it with three disjoint residual spaces.
            spaces[bestSpace] = spaces[spaces.Count - 1];
            spaces.RemoveAt(spaces.Count - 1);

            if (used.W - bw > 0)
                spaces.Add(new Space(used.X + bw, used.Y, used.Z, used.W - bw, used.H, used.D));
            if (used.H - bh > 0)
                spaces.Add(new Space(used.X, used.Y + bh, used.Z, bw, used.H - bh, used.D));
            if (used.D - bd > 0)
                spaces.Add(new Space(used.X, used.Y, used.Z + bd, bw, bh, used.D - bd));
        }

        var output = new System.Text.StringBuilder();
        output.AppendLine(placements.Count.ToString());
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            output.Append(p.Index).Append(' ')
                  .Append(p.X).Append(' ')
                  .Append(p.Y).Append(' ')
                  .Append(p.Z).Append('\n');
        }
        Console.Write(output.ToString());
    }

    private static int CompareBoxes(Box a, Box b)
    {
        int c = a.Volume.CompareTo(b.Volume);
        if (c != 0) return c;
        int am = Math.Max(a.W, Math.Max(a.H, a.D));
        int bm = Math.Max(b.W, Math.Max(b.H, b.D));
        c = am.CompareTo(bm);
        if (c != 0) return c;
        c = Math.Min(a.W, Math.Min(a.H, a.D)).CompareTo(Math.Min(b.W, Math.Min(b.H, b.D)));
        if (c != 0) return c;
        return a.Index.CompareTo(b.Index);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            return v * sign;
        }
    }
}
