using System;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        int[] u = new int[m];
        int[] v = new int[m];
        int[] next = new int[m * 2];
        int[] to = new int[m * 2];
        int[] head = new int[n];
        Array.Fill(head, -1);

        for (int i = 0, e = 0; i < m; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            fs.NextInt(); // weight is irrelevant for constructing a valid walk

            u[i] = a;
            v[i] = b;

            to[e] = b;
            next[e] = head[a];
            head[a] = e++;

            to[e] = a;
            next[e] = head[b];
            head[b] = e++;
        }

        var used = new bool[m];
        var iter = new int[n];
        Array.Copy(head, iter, n);

        int[] stack = new int[Math.Max(1, m + 1)];
        int sp = 0;
        stack[sp++] = 0;

        var output = new FastOutput(Console.OpenStandardOutput());
        output.WriteInt(0);

        while (sp > 0)
        {
            int x = stack[sp - 1];
            int e = iter[x];
            while (e != -1 && used[e >> 1])
            {
                e = next[e];
            }
            iter[x] = e;

            if (e == -1)
            {
                sp--;
                if (sp > 0)
                {
                    output.WriteSpace();
                    output.WriteInt(stack[sp - 1]);
                }
                continue;
            }

            used[e >> 1] = true;
            iter[x] = next[e];

            int y = to[e];
            stack[sp++] = y;
            output.WriteSpace();
            output.WriteInt(y);
        }

        output.WriteLine();
        output.Flush();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - (byte)'0');
                c = Read();
            }
            return val * sign;
        }
    }

    private sealed class FastOutput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr;

        public FastOutput(Stream stream) => _stream = stream;

        public void WriteSpace() => WriteByte((byte)' ');
        public void WriteLine() => WriteByte((byte)'\n');

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[11];
            int len = 0;
            while (value > 0)
            {
                tmp[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }
            for (int i = len - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        private void WriteByte(byte b)
        {
            if (_ptr >= _buffer.Length) Flush();
            _buffer[_ptr++] = b;
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }
    }
}
