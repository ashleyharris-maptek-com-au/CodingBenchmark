#include <algorithm>
#include <chrono>
#include <cstdint>
#include <cstdio>
#include <numeric>
#include <random>
#include <utility>
#include <vector>
using namespace std;

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char getChar() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        char c;
        T sign = 1, val = 0;
        c = getChar();
        if (!c) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = getChar();
            if (!c) return false;
        }
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        for (; c >= '0' && c <= '9'; c = getChar()) val = val * 10 + (c - '0');
        out = val * sign;
        return true;
    }
};

struct DSU {
    vector<int> p, sz;
    int comps;
    DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        sz.assign(n, 1);
        iota(p.begin(), p.end(), 0);
        comps = n;
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (sz[a] < sz[b]) swap(a, b);
        p[b] = a;
        sz[a] += sz[b];
        --comps;
        return true;
    }
};

struct Edge { int u, v; };

static uint64_t seed_now() {
    return chrono::high_resolution_clock::now().time_since_epoch().count() ^ 0x9e3779b97f4a7c15ULL;
}

int main() {
    FastInput in;
    int N, M;
    if (!in.readInt(N)) return 0;
    in.readInt(M);

    vector<Edge> edges;
    edges.reserve(M);
    vector<vector<pair<int,int>>> adj(N);
    vector<int> deg(N, 0);

    for (int i = 0; i < M; ++i) {
        int u, v;
        in.readInt(u); in.readInt(v);
        edges.push_back({u, v});
        adj[u].push_back({v, i});
        adj[v].push_back({u, i});
        ++deg[u];
        ++deg[v];
    }

    auto emit_cut = [&](const vector<int> &side) {
        vector<pair<int,int>> cut;
        cut.reserve(M);
        for (const auto &e : edges) {
            if (side[e.u] != side[e.v]) cut.push_back({e.u, e.v});
        }
        printf("%d\n", (int)cut.size());
        for (auto &e : cut) printf("%d %d\n", e.first, e.second);
    };

    // Exact bridge detection: if a bridge exists, it is a minimum cut of size 1.
    vector<int> tin(N, -1), low(N, -1), parentEdge(N, -1), it(N, 0);
    int timer = 0;
    int bridgeEdge = -1;
    for (int start = 0; start < N && bridgeEdge == -1; ++start) {
        if (tin[start] != -1) continue;
        vector<int> st;
        st.push_back(start);
        parentEdge[start] = -1;
        while (!st.empty() && bridgeEdge == -1) {
            int v = st.back();
            if (tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }
            if (it[v] < (int)adj[v].size()) {
                auto [to, id] = adj[v][it[v]++];
                if (id == parentEdge[v]) continue;
                if (tin[to] == -1) {
                    parentEdge[to] = id;
                    st.push_back(to);
                } else {
                    low[v] = min(low[v], tin[to]);
                }
            } else {
                st.pop_back();
                if (parentEdge[v] != -1) {
                    int pe = parentEdge[v];
                    int p = edges[pe].u ^ edges[pe].v ^ v;
                    low[p] = min(low[p], low[v]);
                    if (low[v] > tin[p]) {
                        bridgeEdge = pe;
                        break;
                    }
                }
            }
        }
    }
    if (bridgeEdge != -1) {
        printf("1\n%d %d\n", edges[bridgeEdge].u, edges[bridgeEdge].v);
        return 0;
    }

    // Baseline: cut isolating a minimum-degree vertex.
    int bestSize = INT32_MAX;
    vector<int> bestSide(N, 0);
    int minDegVertex = 0;
    for (int i = 1; i < N; ++i) if (deg[i] < deg[minDegVertex]) minDegVertex = i;
    bestSize = deg[minDegVertex];
    bestSide[minDegVertex] = 1;

    // Randomized contraction search for smaller cuts.
    mt19937_64 rng(seed_now());
    auto one_run = [&]() {
        DSU dsu(N);
        while (dsu.comps > 2) {
            int idx = (int)(rng() % M);
            int a = dsu.find(edges[idx].u);
            int b = dsu.find(edges[idx].v);
            if (a == b) continue;
            dsu.unite(a, b);
        }
        vector<int> side(N, 0);
        int aroot = -1;
        for (int i = 0; i < N; ++i) {
            int r = dsu.find(i);
            if (aroot == -1) aroot = r;
            if (r == aroot) side[i] = 1;
        }
        int cut = 0;
        for (const auto &e : edges) if (side[e.u] != side[e.v]) ++cut;
        if (cut < bestSize && cut > 0) {
            bestSize = cut;
            bestSide = move(side);
        }
    };

    int runs;
    if (N <= 64) runs = 4000;
    else if (N <= 512) runs = 1200;
    else if (N <= 5000) runs = 300;
    else runs = 80;

    for (int i = 0; i < runs; ++i) one_run();

    emit_cut(bestSide);
    return 0;
}
