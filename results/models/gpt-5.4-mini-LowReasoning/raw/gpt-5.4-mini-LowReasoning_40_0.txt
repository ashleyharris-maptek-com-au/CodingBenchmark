#include <bits/stdc++.h>
using namespace std;

struct Solver {
    int n;
    vector<long long> w;
    vector<vector<int>> g;
    vector<int> best_take, cur_take;
    long long best_val = 0;

    struct BitSet {
        vector<uint64_t> a;
        int n = 0;
        BitSet() {}
        BitSet(int n_) : a((n_ + 63) / 64, 0), n(n_) {}
        void set(int i) { a[i >> 6] |= 1ULL << (i & 63); }
        void reset(int i) { a[i >> 6] &= ~(1ULL << (i & 63)); }
        bool test(int i) const { return (a[i >> 6] >> (i & 63)) & 1ULL; }
        bool any() const {
            for (uint64_t x : a) if (x) return true;
            return false;
        }
        int popcount() const {
            int s = 0;
            for (uint64_t x : a) s += __builtin_popcountll(x);
            return s;
        }
    };

    long long greedy_bound(const BitSet& cand, const vector<int>& verts) {
        vector<int> order = verts;
        sort(order.begin(), order.end(), [&](int i, int j) {
            int di = 0, dj = 0;
            for (int x : g[i]) if (cand.test(x)) ++di;
            for (int x : g[j]) if (cand.test(x)) ++dj;
            if (w[i] != w[j]) return w[i] > w[j];
            if (di != dj) return di < dj;
            return i < j;
        });

        vector<vector<int>> color_classes;
        vector<long long> color_best;
        for (int v : order) {
            int chosen = -1;
            for (int c = 0; c < (int)color_classes.size(); ++c) {
                bool ok = true;
                for (int u : color_classes[c]) {
                    if (binary_search(g[v].begin(), g[v].end(), u)) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    chosen = c;
                    break;
                }
            }
            if (chosen == -1) {
                color_classes.push_back({v});
                color_best.push_back(w[v]);
            } else {
                color_classes[chosen].push_back(v);
                color_best[chosen] = max(color_best[chosen], w[v]);
            }
        }
        long long ub = 0;
        for (long long x : color_best) ub += x;
        return ub;
    }

    int choose_vertex(const BitSet& cand, const vector<int>& verts) {
        int best = -1;
        int best_score = -1;
        for (int v : verts) {
            int deg = 0;
            for (int u : g[v]) if (cand.test(u)) ++deg;
            int score = deg * 1000000 + (int)min<long long>(w[v], 1000000LL);
            if (score > best_score) {
                best_score = score;
                best = v;
            }
        }
        return best;
    }

    void dfs(BitSet cand, long long cur_val, vector<int> verts) {
        if (!cand.any()) {
            if (cur_val > best_val) {
                best_val = cur_val;
                best_take = cur_take;
            }
            return;
        }
        long long bound = greedy_bound(cand, verts);
        if (cur_val + bound <= best_val) return;

        int v = choose_vertex(cand, verts);
        if (v == -1) return;

        BitSet cand_incl = cand;
        cand_incl.reset(v);
        vector<int> verts_incl;
        for (int u : verts) {
            if (u == v) continue;
            bool adj = false;
            for (int x : g[v]) if (x == u) { adj = true; break; }
            if (!adj) verts_incl.push_back(u);
        }
        for (int u : g[v]) cand_incl.reset(u);

        cur_take[v] = 1;
        dfs(cand_incl, cur_val + w[v], verts_incl);
        cur_take[v] = 0;

        BitSet cand_excl = cand;
        cand_excl.reset(v);
        vector<int> verts_excl;
        verts_excl.reserve(verts.size() - 1);
        for (int u : verts) if (u != v) verts_excl.push_back(u);
        dfs(cand_excl, cur_val, verts_excl);
    }

    pair<long long, vector<int>> solve_component(const vector<int>& comp) {
        vector<int> verts;
        for (int v : comp) if (w[v] > 0) verts.push_back(v);
        if (verts.empty()) return {0, vector<int>(n, 0)};

        BitSet cand(n);
        for (int v : verts) cand.set(v);
        best_val = 0;
        best_take.assign(n, 0);
        cur_take.assign(n, 0);
        dfs(cand, 0, verts);
        return {best_val, best_take};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (0 <= u && u < n && 0 <= v && v < n && u != v) {
            g[u].push_back(v);
            g[v].push_back(u);
        }
    }
    for (int i = 0; i < n; ++i) {
        sort(g[i].begin(), g[i].end());
        g[i].erase(unique(g[i].begin(), g[i].end()), g[i].end());
    }

    vector<int> comp_id(n, -1);
    vector<vector<int>> comps;
    for (int i = 0; i < n; ++i) {
        if (comp_id[i] != -1) continue;
        vector<int> comp;
        queue<int> q;
        q.push(i);
        comp_id[i] = (int)comps.size();
        while (!q.empty()) {
            int v = q.front(); q.pop();
            comp.push_back(v);
            for (int u : g[v]) if (comp_id[u] == -1) {
                comp_id[u] = comp_id[i];
                q.push(u);
            }
        }
        comps.push_back(comp);
    }

    vector<int> ans(n, 0);
    long long total = 0;
    Solver solver;
    solver.n = n;
    solver.w = c;
    solver.g = g;

    for (auto &comp : comps) {
        Solver sub;
        sub.n = n;
        sub.w = c;
        sub.g = g;
        auto [val, take] = sub.solve_component(comp);
        total += val;
        for (int i = 0; i < n; ++i) if (take[i]) ans[i] = 1;
    }

    long long check = 0;
    for (int i = 0; i < n; ++i) if (ans[i]) check += c[i];
    cout << check << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
