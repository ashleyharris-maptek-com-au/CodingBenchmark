use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let half = n / 2;

    let mut adj = vec![0u64; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u] |= 1u64 << v;
        adj[v] |= 1u64 << u;
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut best_cut = u64::MAX;
    let mut best_mask = 0u64;

    if n <= 63 {
        let limit = 1u64 << n;
        let mut mask = if half == 0 { 0 } else { (1u64 << half) - 1 };
        while mask < limit {
            let mut cut = 0u64;
            for i in 0..n {
                let in_a = ((mask >> i) & 1) != 0;
                let crossing = (adj[i] & mask).count_ones() as u64;
                if in_a {
                    cut += (adj[i].count_ones() as u64) - crossing;
                } else {
                    cut += crossing;
                }
            }
            cut /= 2;
            if cut < best_cut {
                best_cut = cut;
                best_mask = mask;
            }

            if mask == 0 {
                break;
            }
            let c = mask & (!mask + 1);
            let r = mask + c;
            mask = (((r ^ mask) >> 2) / c) | r;
        }
    } else {
        unreachable!("n too large for this implementation");
    }

    let mut part = Vec::with_capacity(half);
    for i in 0..n {
        if ((best_mask >> i) & 1) != 0 {
            part.push(i);
        }
    }

    println!("{}", best_cut);
    for (idx, v) in part.iter().enumerate() {
        if idx > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
