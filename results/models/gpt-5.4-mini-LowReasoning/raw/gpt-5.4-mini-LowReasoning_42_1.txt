OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%_arr_gl_PerVertex_in_uint_1 = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input__arr_gl_PerVertex_in_uint_1 = OpTypePointer Input %_arr_gl_PerVertex_in_uint_1
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_arr_v3float_uint_1 = OpTypeArray %v3float %uint_1
%_ptr_Input__arr_v3float_uint_1 = OpTypePointer Input %_arr_v3float_uint_1
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%_ptr_Function_v4float = OpTypePointer Function %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_05 = OpConstant %float 0.05

%gl_in = OpVariable %_ptr_Input__arr_gl_PerVertex_in_uint_1 Input
%inColor = OpVariable %_ptr_Input__arr_v3float_uint_1 Input
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output

%main = OpFunction %void None %void
%label = OpLabel
%in_pos_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1
%z = OpCompositeExtract %float %in_pos 2
%w = OpCompositeExtract %float %in_pos 3
%in_color_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_0
%color = OpLoad %v3float %in_color_ptr

%y_top = OpFAdd %float %y %float_05
%x_right = OpFAdd %float %x %float_05
%y_bottom = OpFSub %float %y %float_05
%x_left = OpFSub %float %x %float_05

%top = OpCompositeConstruct %v4float %x %y_top %z %w
%right = OpCompositeConstruct %v4float %x_right %y %z %w
%left = OpCompositeConstruct %v4float %x_left %y %z %w
%bottom = OpCompositeConstruct %v4float %x %y_bottom %z %w

%out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %top
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_ptr %right
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_ptr %left
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_ptr %bottom
OpStore %outColor %color
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
