const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget | 0;

if (cutBudget <= 0 || width <= 0 || height <= 0 || !grid || !grid.length) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const n = width * height;
const idx = (x, y) => y * width + x;
const inBounds = (x, y) => x >= 0 && x < width && y >= 0 && y < height;

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    burnable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const ignitionMask = new Uint8Array(n);
for (const [x, y] of ignitions) {
  if (inBounds(x, y)) ignitionMask[idx(x, y)] = 1;
}

const townAt = new Int32Array(n);
townAt.fill(-1);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (inBounds(t.x, t.y)) townAt[idx(t.x, t.y)] = i;
}

const dist = new Int32Array(n);
dist.fill(-1);
const parent = new Int32Array(n);
parent.fill(-1);
const q = new Int32Array(n);
let qh = 0, qt = 0;

for (const [x, y] of ignitions) {
  if (!inBounds(x, y)) continue;
  const i = idx(x, y);
  if (!burnable[i]) continue;
  if (dist[i] !== -1) continue;
  dist[i] = 0;
  q[qt++] = i;
}

while (qh < qt) {
  const cur = q[qh++];
  const cx = cur % width;
  const cy = (cur / width) | 0;
  const nd = dist[cur] + 1;

  if (cx > 0) {
    const ni = cur - 1;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = cur;
      q[qt++] = ni;
    }
  }
  if (cx + 1 < width) {
    const ni = cur + 1;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = cur;
      q[qt++] = ni;
    }
  }
  if (cy > 0) {
    const ni = cur - width;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = cur;
      q[qt++] = ni;
    }
  }
  if (cy + 1 < height) {
    const ni = cur + width;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = cur;
      q[qt++] = ni;
    }
  }
}

const score = new Float64Array(n);
for (const town of towns) {
  const i = inBounds(town.x, town.y) ? idx(town.x, town.y) : -1;
  if (i < 0) continue;
  if (dist[i] === -1) continue;
  let cur = i;
  let steps = 0;
  const value = Number(town.value) || 0;
  const pathLimit = Math.min(dist[i], 2000);
  while (parent[cur] !== -1 && steps < pathLimit) {
    cur = parent[cur];
    steps++;
    if (ignitionMask[cur]) break;
    if (!burnable[cur]) break;
    const d = dist[cur];
    if (d >= 0) {
      score[cur] += value / (steps + d + 1);
    }
  }
}

// Prefer cells near ignitions that help many towns, but also give weight to
// cells that are on many shortest paths from fires to valuable towns.
for (let i = 0; i < n; i++) {
  if (!burnable[i] || ignitionMask[i] || townAt[i] !== -1) {
    score[i] = -1;
    continue;
  }
  if (dist[i] === -1) score[i] = -1;
  else if (dist[i] <= 2) score[i] *= 1.8;
  else if (dist[i] <= 5) score[i] *= 1.3;
}

const chosen = [];
const chosenMask = new Uint8Array(n);

function pickBest() {
  let best = -1;
  let bestScore = 0;
  for (let i = 0; i < n; i++) {
    if (chosenMask[i] || score[i] <= bestScore) continue;
    bestScore = score[i];
    best = i;
  }
  return best;
}

function zeroNeighborhood(i) {
  score[i] = -1;
}

while (chosen.length < cutBudget) {
  const best = pickBest();
  if (best === -1 || score[best] <= 0) break;
  chosen.push([best % width, (best / width) | 0]);
  chosenMask[best] = 1;
  zeroNeighborhood(best);

  const bx = best % width;
  const by = (best / width) | 0;
  const decays = [
    [bx - 1, by],
    [bx + 1, by],
    [bx, by - 1],
    [bx, by + 1],
  ];
  for (const [x, y] of decays) {
    if (!inBounds(x, y)) continue;
    const ni = idx(x, y);
    if (score[ni] > 0) score[ni] *= 0.85;
  }
}

// Small refinement: if we have remaining budget, choose extra cells adjacent
// to already selected firebreaks to thicken the barrier around the best spot.
if (chosen.length < cutBudget) {
  const adjacencyScore = new Float64Array(n);
  for (let i = 0; i < n; i++) {
    if (!burnable[i] || chosenMask[i] || ignitionMask[i] || townAt[i] !== -1) continue;
    const x = i % width;
    const y = (i / width) | 0;
    let adj = 0;
    if (x > 0 && chosenMask[i - 1]) adj++;
    if (x + 1 < width && chosenMask[i + 1]) adj++;
    if (y > 0 && chosenMask[i - width]) adj++;
    if (y + 1 < height && chosenMask[i + width]) adj++;
    if (adj > 0) adjacencyScore[i] = score[i] + adj * 0.25;
  }
  while (chosen.length < cutBudget) {
    let best = -1;
    let bestScore = 0;
    for (let i = 0; i < n; i++) {
      if (chosenMask[i] || adjacencyScore[i] <= bestScore) continue;
      bestScore = adjacencyScore[i];
      best = i;
    }
    if (best === -1 || bestScore <= 0) break;
    chosen.push([best % width, (best / width) | 0]);
    chosenMask[best] = 1;
    adjacencyScore[best] = -1;
  }
}

process.stdout.write(JSON.stringify({ firebreaks: chosen }));
