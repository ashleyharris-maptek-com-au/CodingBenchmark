%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO

%worldPos = OpVariable %ptr_input_v3 Input

%v3_fog = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_8

%main = OpFunction %void None %func
%label = OpLabel
%pos = OpLoad %v3float %worldPos
%nrm = OpLoad %v3float %normal
%n = OpExtInst %v3float %1 Normalize %nrm
%camera = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%camera4 = OpLoad %v4float %camera
%camera3 = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%light = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light4 = OpLoad %v4float %light
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%toLight = OpFSub %v3float %light3 %pos
%toLightN = OpExtInst %v3float %1 Normalize %toLight
%diffuseDot = OpDot %float %n %toLightN
%diffuse = OpExtInst %float %1 FClamp %diffuseDot %float_0 %float_1
%lambertMul = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %lambertMul
%lit = OpVectorTimesScalar %v3float %v3_base %lambert
%toCam = OpFSub %v3float %pos %camera3
%dist = OpExtInst %float %1 Length %toCam
%fogNum = OpFSub %float %dist %float_1_5
%fogDiv = OpFDiv %float %fogNum %float_3
%fogFactor = OpExtInst %float %1 FClamp %fogDiv %float_0 %float_1
%fogMixA = OpVectorTimesScalar %v3float %lit %fogFactor
%oneMinusFog = OpFSub %float %float_1 %fogFactor
%fogMixB = OpVectorTimesScalar %v3float %v3_fog %oneMinusFog
%color = OpFAdd %v3float %fogMixA %fogMixB
%color_r = OpCompositeExtract %float %color 0
%color_g = OpCompositeExtract %float %color 1
%color_b = OpCompositeExtract %float %color 2
%color4 = OpCompositeConstruct %v4float %color_r %color_g %color_b %float_1
OpStore %outColor %color4
OpReturn
