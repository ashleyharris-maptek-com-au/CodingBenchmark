OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%glsl450 = OpExtInstImport "GLSL.std.450"

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_thickness = OpConstant %float 0.03

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_input_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_output_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_input_arr_pv Input
%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_input_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_input_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%dir = OpFSub %v4float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_src = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm = OpExtInst %v2float %glsl450 Normalize %perp_src
%perp = OpVectorTimesScalar %v2float %perp_norm %float_thickness

%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p0_xy = OpCompositeConstruct %v2float %p0_x %p0_y
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_xy = OpCompositeConstruct %v2float %p1_x %p1_y

%v0_xy = OpFAdd %v2float %p0_xy %perp
%v1_xy = OpFSub %v2float %p0_xy %perp
%v2_xy = OpFAdd %v2float %p1_xy %perp
%v3_xy = OpFSub %v2float %p1_xy %perp

%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1

%pos0 = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
%pos1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
%pos2 = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
%pos3 = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w

%out0_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %outColor %c0
OpStore %out0_ptr %pos0
OpEmitVertex

%out1_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %outColor %c0
OpStore %out1_ptr %pos1
OpEmitVertex

%out2_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %outColor %c1
OpStore %out2_ptr %pos2
OpEmitVertex

%out3_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %outColor %c1
OpStore %out3_ptr %pos3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
