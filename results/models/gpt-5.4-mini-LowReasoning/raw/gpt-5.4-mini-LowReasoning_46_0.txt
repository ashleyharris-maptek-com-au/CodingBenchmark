#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    uint baseI = i * 2u;
    vec3 pos_i = inData[baseI].xyz;
    float mass_i = inData[baseI].w;
    vec3 vel_i = inData[baseI + 1u].xyz;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softening2 = softening * softening;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint baseJ = j * 2u;
        vec3 pos_j = inData[baseJ].xyz;
        float mass_j = inData[baseJ].w;

        vec3 r = pos_j - pos_i;
        float dist2 = dot(r, r) + softening2;
        float invDist = inversesqrt(dist2);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    vec3 vel_out = vel_i + acc * dt;
    vec3 pos_out = pos_i + vel_out * dt;

    outData[baseI] = vec4(pos_out, mass_i);
    outData[baseI + 1u] = vec4(vel_out, 0.0);
}
