use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn parse_f64(s: &str) -> Option<f64> {
    s.parse::<f64>().ok()
}

fn parse_nums(line: &str) -> Vec<f64> {
    line.split_whitespace().filter_map(parse_f64).collect()
}

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut line = String::new();

    // Header line.
    line.clear();
    if reader.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
        return;
    }
    let header = parse_nums(&line);
    if header.len() < 5.0_f64 as usize {
        return;
    }
    let _width = header[0] as usize;
    let _height = header[1] as usize;
    let gravity = header[2];
    let max_thrust = header[3];
    let _fuel_capacity = header[4];

    // Second line: start and target coordinates.
    line.clear();
    if reader.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
        return;
    }
    let second = parse_nums(&line);
    if second.len() < 4 {
        return;
    }
    let _start_x = second[0];
    let _start_y = second[1];
    let target_x = second[2];
    let target_y = second[3];

    // Consume the map rows without storing them. If a target/landing mark exists,
    // this keeps the parser aligned even for large maps.
    for _ in 0.._height {
        line.clear();
        if reader.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
            break;
        }
        let _ = line.trim_end();
    }

    let mut state_line = String::new();
    let mut last_turn = 0.0f64;
    let pad_x = target_x + 0.5;
    let pad_y = target_y + 0.5;

    loop {
        state_line.clear();
        let n = match reader.read_line(&mut state_line) {
            Ok(n) => n,
            Err(_) => break,
        };
        if n == 0 {
            break;
        }
        let nums = parse_nums(&state_line);
        if nums.len() < 7 {
            continue;
        }
        let s = State {
            x: nums[0],
            y: nums[1],
            vx: nums[2],
            vy: nums[3],
            angle: nums[4],
            angular_vel: nums[5],
            fuel: nums[6],
        };

        let dx = pad_x - s.x;
        let dy = pad_y - s.y;

        // Horizontal guidance: lean toward the pad when far away, but level out
        // as we approach the landing zone.
        let dist_x = dx.abs();
        let mut desired_angle = if dist_x > 30.0 {
            clamp(dx * 0.03 + s.vx * 0.05, -0.75, 0.75)
        } else if dist_x > 10.0 {
            clamp(dx * 0.05 + s.vx * 0.08, -0.55, 0.55)
        } else {
            clamp(dx * 0.08 + s.vx * 0.12, -0.25, 0.25)
        };

        // If we are low and still far from the pad, prioritize a stronger
        // lateral correction so we do not drift out of range.
        if dy < 20.0 && dist_x > 6.0 {
            desired_angle = clamp(dx * 0.09 + s.vx * 0.15, -0.65, 0.65);
        }

        // Vertical profile: keep a safe descent rate that slows near the ground.
        let desired_vy = if dy > 120.0 {
            16.0
        } else if dy > 60.0 {
            10.0
        } else if dy > 25.0 {
            6.0
        } else if dy > 10.0 {
            3.0
        } else {
            1.0
        };

        let vertical_error = desired_vy - s.vy;
        let desired_down_accel = clamp(vertical_error * 0.75, -max_thrust * 0.95, max_thrust * 0.95);
        let cos_a = s.angle.cos().abs().max(0.15);
        let mut thrust = (gravity - desired_down_accel) / (max_thrust * cos_a);

        // When approaching the pad, bias toward full thrust for stability.
        if dy < 30.0 {
            thrust = thrust.max(0.75);
        }
        if dy < 12.0 && s.vy > 2.0 {
            thrust = 1.0;
        }
        if s.fuel < 1.0 {
            thrust = 0.0;
        }
        thrust = clamp(thrust, 0.0, 1.0);

        let turn = clamp((desired_angle - s.angle) * 3.0 - s.angular_vel * 0.9, -1.0, 1.0);
        last_turn = turn;

        if writeln!(out, "{:.3} {:.3}", thrust, turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }
    }

    let _ = PI; // Keep imports stable if the optimizer removes angle math paths.
    let _ = last_turn;
}
