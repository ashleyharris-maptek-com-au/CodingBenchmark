@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl32(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn quarter_round(a : ptr<function, u32>,
                 b : ptr<function, u32>,
                 c : ptr<function, u32>,
                 d : ptr<function, u32>) {
    (*a) = (*a) + (*b);
    (*d) = rotl32((*d) ^ (*a), 16u);

    (*c) = (*c) + (*d);
    (*b) = rotl32((*b) ^ (*c), 12u);

    (*a) = (*a) + (*b);
    (*d) = rotl32((*d) ^ (*a), 8u);

    (*c) = (*c) + (*d);
    (*b) = rotl32((*b) ^ (*c), 7u);
}

fn custom_hash(data : array<u32, 16>, nonce : u32) -> vec2<u32> {
    var state : array<u32, 16>;

    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = data[i] ^ (nonce + i);
    }

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        quarter_round(&state[0], &state[1], &state[2], &state[3]);
        quarter_round(&state[4], &state[5], &state[6], &state[7]);
        quarter_round(&state[8], &state[9], &state[10], &state[11]);
        quarter_round(&state[12], &state[13], &state[14], &state[15]);
    }

    var h0 : u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    var h1 : u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];
    return vec2<u32>(h0, h1);
}

fn trailing_zeros_64(lo : u32, hi : u32) -> u32 {
    if (lo != 0u) {
        return countTrailingZeros(lo);
    }
    return 32u + countTrailingZeros(hi);
}

@compute @workgroup_size(256u)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce : u32 = params.y + gid.x;
    let hash : vec2<u32> = custom_hash(baseData, nonce);
    let tz : u32 = trailing_zeros_64(hash.x, hash.y);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = hash.x;
        result[3] = hash.y;
    }
}
