// Longest Common Substring solver with adaptive strategy.
// - Small/medium shortest string: suffix automaton
// - Large inputs: binary search + 64-bit rolling hash

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <vector>

using namespace std;

static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct Sam {
    struct State {
        int link = -1;
        int len = 0;
        int firstpos = -1;
        unordered_map<unsigned char, int> next;
    };

    vector<State> st;
    int last = 0;

    explicit Sam(size_t reserve_states = 0) {
        st.reserve(reserve_states);
        st.push_back(State{});
    }

    void extend(unsigned char c, int pos) {
        int cur = (int)st.size();
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].firstpos = pos;

        int p = last;
        while (p != -1 && !st[p].next.count(c)) {
            st[p].next[c] = cur;
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                // firstpos of clone is fine for reconstruction.
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static string longest_common_substring_sam(const vector<string>& s) {
    int idx = 0;
    for (int i = 1; i < (int)s.size(); ++i) {
        if (s[i].size() < s[idx].size()) idx = i;
    }
    string base = s[idx];

    Sam sam(base.size() * 2 + 1);
    for (int i = 0; i < (int)base.size(); ++i) sam.extend((unsigned char)base[i], i);

    int m = (int)sam.st.size();
    vector<int> order(m), cnt(base.size() + 1);
    for (const auto& st : sam.st) ++cnt[st.len];
    for (int i = 1; i < (int)cnt.size(); ++i) cnt[i] += cnt[i - 1];
    for (int i = m - 1; i >= 0; --i) order[--cnt[sam.st[i].len]] = i;

    vector<int> best(m);
    for (int i = 0; i < m; ++i) best[i] = sam.st[i].len;

    vector<int> mx(m);
    for (int si = 0; si < (int)s.size(); ++si) {
        if (si == idx) continue;
        fill(mx.begin(), mx.end(), 0);

        int v = 0, l = 0;
        for (unsigned char c : s[si]) {
            while (v && !sam.st[v].next.count(c)) {
                v = sam.st[v].link;
                l = sam.st[v].len;
            }
            auto it = sam.st[v].next.find(c);
            if (it != sam.st[v].next.end()) {
                v = it->second;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            mx[v] = max(mx[v], l);
        }

        for (int i = m - 1; i > 0; --i) {
            int vtx = order[i];
            int p = sam.st[vtx].link;
            if (p >= 0) {
                mx[p] = max(mx[p], min(mx[vtx], sam.st[p].len));
            }
        }
        for (int i = 0; i < m; ++i) best[i] = min(best[i], mx[i]);
    }

    int ansLen = 0, ansState = 0;
    for (int i = 0; i < m; ++i) {
        if (best[i] > ansLen) {
            ansLen = best[i];
            ansState = i;
        }
    }
    if (ansLen == 0) return "";
    int endPos = sam.st[ansState].firstpos;
    return base.substr(endPos - ansLen + 1, ansLen);
}

struct RollingHash {
    static constexpr uint64_t B1 = 1469598103934665603ULL;
    static constexpr uint64_t B2 = 1099511628211ULL;
    vector<uint64_t> p1, p2, h1, h2;

    explicit RollingHash(const string& s) {
        size_t n = s.size();
        p1.resize(n + 1);
        p2.resize(n + 1);
        h1.resize(n + 1);
        h2.resize(n + 1);
        p1[0] = p2[0] = 1;
        for (size_t i = 0; i < n; ++i) {
            uint64_t x = (unsigned char)s[i] + 1;
            p1[i + 1] = p1[i] * B1;
            p2[i + 1] = p2[i] * B2;
            h1[i + 1] = h1[i] * B1 + x;
            h2[i + 1] = h2[i] * B2 + x;
        }
    }

    pair<uint64_t, uint64_t> get(size_t l, size_t r) const {
        return {h1[r] - h1[l] * p1[r - l], h2[r] - h2[l] * p2[r - l]};
    }
};

struct Key {
    uint64_t a, b;
    bool operator==(const Key& o) const { return a == o.a && b == o.b; }
};

struct KeyHash {
    size_t operator()(const Key& k) const noexcept {
        return (size_t)(splitmix64(k.a) ^ (splitmix64(k.b) << 1));
    }
};

static bool check_len_hash(const vector<string>& s, int baseIdx, int L, string& out) {
    if (L == 0) {
        out.clear();
        return true;
    }

    const string& base = s[baseIdx];
    RollingHash hb(base);

    unordered_map<Key, int, KeyHash> cand;
    cand.reserve(base.size() > (size_t)L ? base.size() - L + 1 : 1);
    for (size_t i = 0; i + (size_t)L <= base.size(); ++i) {
        auto [a, b] = hb.get(i, i + L);
        cand[{a, b}] = (int)i;
    }

    for (int si = 0; si < (int)s.size(); ++si) {
        if (si == baseIdx) continue;
        if (cand.empty()) return false;
        RollingHash hs(s[si]);
        unordered_set<Key, KeyHash> seen;
        seen.reserve(s[si].size() > (size_t)L ? s[si].size() - L + 1 : 1);
        for (size_t i = 0; i + (size_t)L <= s[si].size(); ++i) {
            auto [a, b] = hs.get(i, i + L);
            seen.insert({a, b});
        }
        for (auto it = cand.begin(); it != cand.end(); ) {
            if (!seen.count(it->first)) it = cand.erase(it);
            else ++it;
        }
    }

    if (cand.empty()) return false;
    out = base.substr(cand.begin()->second, L);
    return true;
}

static string longest_common_substring_hash(const vector<string>& s) {
    int idx = 0;
    for (int i = 1; i < (int)s.size(); ++i) {
        if (s[i].size() < s[idx].size()) idx = i;
    }

    int lo = 0, hi = (int)s[idx].size();
    string best, cur;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        if (check_len_hash(s, idx, mid, cur)) {
            best = cur;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }
    if (best.size() != (size_t)hi) {
        cur.clear();
        if (check_len_hash(s, idx, hi, cur)) best = cur;
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> s(N);
    for (int i = 0; i < N; ++i) getline(cin, s[i]);

    size_t total = 0, mn = numeric_limits<size_t>::max();
    for (const auto& x : s) {
        total += x.size();
        mn = min(mn, x.size());
    }

    string ans;
    if (mn <= 200000 && total <= 4000000) {
        ans = longest_common_substring_sam(s);
    } else {
        ans = longest_common_substring_hash(s);
    }

    cout << ans << '\n';
    return 0;
}
