using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            return v * sign;
        }

        public byte NextByte()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            return (byte)c;
        }
    }

    private static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int xCount = fs.NextInt();
        int yCount = fs.NextInt();
        int zCount = fs.NextInt();
        fs.NextInt();
        fs.NextInt();
        fs.NextInt();

        long cellCountLong = (long)xCount * yCount;
        if (cellCountLong <= 0 || cellCountLong > int.MaxValue)
        {
            Console.WriteLine(0);
            return;
        }

        int cellCount = (int)cellCountLong;
        byte[] top = new byte[cellCount];

        for (int x = 0; x < xCount; x++)
        {
            int xBase = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                byte last = 0;
                for (int z = 0; z < zCount; z++)
                {
                    byte d = fs.NextByte();
                    if (z == zCount - 1) last = d;
                }
                top[xBase + y] = last;
            }
        }

        int maxOps = 5000;
        var sb = new StringBuilder();
        var ops = new List<string>(Math.Min(maxOps, cellCount));

        int xLimit = Math.Min(xCount, 2000);
        int yLimit = Math.Min(yCount, 2000);

        for (int sum = 0; sum <= xLimit + yLimit - 2 && ops.Count < maxOps; sum++)
        {
            int xStart = Math.Max(0, sum - (yLimit - 1));
            int xEnd = Math.Min(xLimit - 1, sum);
            for (int x = xStart; x <= xEnd && ops.Count < maxOps; x++)
            {
                int y = sum - x;
                if (y < 0 || y >= yLimit) continue;
                byte d = top[x * yCount + y];
                int mass = d & 15;
                int value = d >> 4;
                if (mass == 0 || value == 0) continue;

                long profitPerTon = 1000L * value - (x + y);
                if (profitPerTon <= 0) continue;

                int steps = x + y;
                sb.Clear();
                sb.Append(steps).Append(' ').Append(x).Append(' ').Append(y).Append(' ').Append(zCount - 1);

                int cx = x;
                int cy = y;
                for (int i = 0; i < x; i++)
                {
                    cx--;
                    sb.Append(' ').Append(cx).Append(' ').Append(cy);
                }
                for (int i = 0; i < y; i++)
                {
                    cy--;
                    sb.Append(' ').Append(cx).Append(' ').Append(cy);
                }
                sb.Append(' ').Append(0).Append(' ').Append(0);
                ops.Add(sb.ToString());
            }
        }

        var output = new StringBuilder();
        output.AppendLine(ops.Count.ToString());
        for (int i = 0; i < ops.Count; i++)
            output.AppendLine(ops[i]);
        Console.Write(output.ToString());
    }
}
