from __future__ import annotations

import math
from statistics import median


_S = {
    "t_prev": None,
    "alt_i": 0.0,
    "hdg_i": 0.0,
    "spd_i": 0.0,
    "vs_i": 0.0,
    "roll_i": 0.0,
    "trim": 0.0,
    "last_alt": None,
    "last_hdg": None,
    "last_vs": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _wrap180(a):
    return (a + 180.0) % 360.0 - 180.0


def _wrap360(a):
    return a % 360.0


def _mw(vals, flags=None, max_dev=None):
    items = []
    if flags is None:
        flags = [True] * len(vals)
    for v, f in zip(vals, flags):
        if f and v is not None and not isinstance(v, bool) and math.isfinite(v):
            items.append(float(v))
    if not items:
        return None
    if len(items) == 1 or max_dev is None:
        return median(items)
    m = median(items)
    close = [v for v in items if abs(v - m) <= max_dev]
    return median(close) if close else m


def _pick_heading(state):
    vals = []
    for k in ("target_hdg_deg", "magnetic_heading_deg", "true_heading_deg", "track_deg"):
        v = state.get(k)
        if isinstance(v, (int, float)) and math.isfinite(v):
            vals.append(float(v))
    return vals[0] if vals else 0.0


def _voted_alt(state):
    # Prefer radar near the ground, otherwise a robust baro/GPS blend.
    rad = _mw(
        [state.get("radar_alt_1_ft"), state.get("radar_alt_2_ft"), state.get("radar_alt_3_ft")],
        [state.get("radar_alt_1_valid", True), state.get("radar_alt_2_valid", True), state.get("radar_alt_3_valid", True)],
        max_dev=50.0,
    )
    baro = _mw(
        [state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")],
        [state.get("baro_alt_1_valid", True), state.get("baro_alt_2_valid", True), state.get("baro_alt_3_valid", True)],
        max_dev=300.0,
    )
    gps = _mw(
        [state.get("gps_1_alt_ft"), state.get("gps_2_alt_ft"), state.get("gps_3_alt_ft")],
        [state.get("gps_1_valid", True), state.get("gps_2_valid", True), state.get("gps_3_valid", True)],
        max_dev=500.0,
    )
    if rad is not None and rad < 2500.0:
        if baro is None:
            return rad
        return 0.75 * rad + 0.25 * baro
    candidates = [v for v in [baro, gps] if v is not None]
    if candidates:
        return median(candidates)
    if rad is not None:
        return rad
    return float(state.get("target_alt_ft", 0.0))


def _voted_spd(state):
    ias = _mw(
        [state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")],
        [state.get("ias_1_valid", True), state.get("ias_2_valid", True), state.get("ias_3_valid", True)],
        max_dev=15.0,
    )
    gs = _mw(
        [state.get("gps_1_gs_kt"), state.get("gps_2_gs_kt"), state.get("gps_3_gs_kt")],
        [state.get("gps_1_valid", True), state.get("gps_2_valid", True), state.get("gps_3_valid", True)],
        max_dev=20.0,
    )
    tas = _mw([state.get("tas_1_kt"), state.get("tas_2_kt"), state.get("tas_3_kt")], max_dev=20.0)
    if ias is not None:
        if gs is not None and abs(gs - ias) > 80.0 and tas is not None:
            return 0.7 * ias + 0.3 * tas
        return ias
    if tas is not None:
        return tas
    if gs is not None:
        return gs
    return 200.0


def _voted_vs(state):
    vs = _mw(
        [state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm")],
        [state.get("irs_1_valid", True), state.get("irs_2_valid", True), state.get("irs_3_valid", True)],
        max_dev=400.0,
    )
    if vs is not None:
        return vs
    return float(state.get("vertical_speed_fpm", 0.0))


def _voted_attitude(state):
    pitch = _mw(
        [state.get("irs_1_pitch_deg"), state.get("irs_2_pitch_deg"), state.get("irs_3_pitch_deg")],
        [state.get("irs_1_valid", True), state.get("irs_2_valid", True), state.get("irs_3_valid", True)],
        max_dev=8.0,
    )
    roll = _mw(
        [state.get("irs_1_roll_deg"), state.get("irs_2_roll_deg"), state.get("irs_3_roll_deg")],
        [state.get("irs_1_valid", True), state.get("irs_2_valid", True), state.get("irs_3_valid", True)],
        max_dev=8.0,
    )
    yaw = _mw(
        [state.get("irs_1_track_deg"), state.get("irs_2_track_deg"), state.get("irs_3_track_deg")],
        [state.get("irs_1_valid", True), state.get("irs_2_valid", True), state.get("irs_3_valid", True)],
        max_dev=15.0,
    )
    if pitch is None:
        pitch = 0.0
    if roll is None:
        roll = 0.0
    if yaw is None:
        yaw = _pick_heading(state)
    return pitch, roll, yaw


def autopilot_step(state: dict, dt: float) -> dict:
    now = float(state.get("sim_time_s", 0.0))
    if _S["t_prev"] is None:
        _S["t_prev"] = now
    dt = float(dt) if dt and dt > 1e-4 else max(0.02, min(0.2, now - _S["t_prev"] if now > _S["t_prev"] else 0.05))
    _S["t_prev"] = now

    ap = bool(state.get("ap_engaged", 1))
    at = bool(state.get("at_engaged", 1))

    alt = _voted_alt(state)
    spd = _voted_spd(state)
    vs = _voted_vs(state)
    pitch, roll, track = _voted_attitude(state)
    tgt_alt = float(state.get("target_alt_ft", alt))
    tgt_hdg = float(state.get("target_hdg_deg", track))
    tgt_spd = float(state.get("target_ias_kt", spd))
    tgt_vs = float(state.get("target_vs_fpm", 0.0))

    # Smooth state derivatives from fused values.
    if _S["last_alt"] is None:
        _S["last_alt"] = alt
    if _S["last_hdg"] is None:
        _S["last_hdg"] = track
    alt_rate = (alt - _S["last_alt"]) / max(dt, 0.02)
    hdg_rate = _wrap180(track - _S["last_hdg"]) / max(dt, 0.02)
    _S["last_alt"] = alt
    _S["last_hdg"] = track

    alt_err = tgt_alt - alt
    spd_err = tgt_spd - spd
    hdg_err = _wrap180(tgt_hdg - track)
    vs_err = tgt_vs - vs
    roll_err = _wrap180(0.0 - roll)

    # Energy management: speed protection near the ground and at high speed.
    low_alt = alt < 3000.0
    high_speed = spd > 330.0
    stall_warn = spd < 1.18 * max(120.0, tgt_spd * 0.85) or (state.get("aoa_1_deg", 0.0) > 12.0 and state.get("aoa_2_deg", 0.0) > 12.0)

    # Pitch command from altitude, vertical speed, and speed protection.
    alt_capture = _clamp(alt_err / 1200.0, -1.0, 1.0)
    vs_capture = _clamp(vs_err / 1800.0, -1.0, 1.0)
    pitch_cmd = 0.55 * alt_capture + 0.35 * vs_capture
    if low_alt:
        pitch_cmd += _clamp((tgt_spd - spd) / 80.0, -0.6, 0.6)
    if stall_warn:
        pitch_cmd -= 0.9
        _S["trim"] -= 0.0025 * dt * 50.0
    if high_speed:
        pitch_cmd -= _clamp((spd - 330.0) / 60.0, 0.0, 0.8)
    pitch_cmd = _clamp(pitch_cmd - 0.02 * pitch - 0.0006 * alt_rate, -1.0, 1.0)

    # Bank command from heading error with turn coordination.
    turn_gain = 0.018 if spd > 160.0 else 0.022
    bank_cmd = _clamp(turn_gain * hdg_err - 0.02 * hdg_rate, -25.0, 25.0)
    if abs(hdg_err) > 90.0:
        bank_cmd = _clamp(turn_gain * _wrap180(hdg_err) * 1.35, -30.0, 30.0)
    roll_cmd = _clamp((bank_cmd - roll) / 22.0, -1.0, 1.0)

    # Rudder damps sideslip / asymmetric thrust / roll coupling.
    n1 = float(state.get("engine_1_n1_pct", 0.0))
    n2 = float(state.get("engine_2_n1_pct", 0.0))
    yaw_rate = 0.0
    for k in ("gyro_1_yaw_rate_dps", "gyro_2_yaw_rate_dps", "gyro_3_yaw_rate_dps"):
        v = state.get(k)
        if isinstance(v, (int, float)) and math.isfinite(v):
            yaw_rate += float(v)
    yaw_rate /= 3.0
    rudder_cmd = _clamp(-0.015 * hdg_err - 0.06 * yaw_rate + 0.01 * (n1 - n2) + 0.02 * roll, -1.0, 1.0)

    # Thrust control: maintain speed, protect against stalls and overspeed, and respect engine-out asymmetry.
    thrust_base = 0.58 + _clamp(-spd_err / 180.0, -0.35, 0.35) + _clamp(alt_err / 25000.0, -0.12, 0.12)
    if low_alt:
        thrust_base += _clamp((tgt_spd - spd) / 120.0, -0.25, 0.25)
    if stall_warn:
        thrust_base += 0.25
    if high_speed:
        thrust_base -= _clamp((spd - 330.0) / 90.0, 0.0, 0.35)
    if bool(state.get("engine_1_fire", False)):
        thrust_base = max(thrust_base, 0.2)
    if bool(state.get("engine_2_fire", False)):
        thrust_base = max(thrust_base, 0.2)
    thrust_base = _clamp(thrust_base, 0.05, 1.0)
    asym = _clamp(0.004 * (n2 - n1), -0.12, 0.12)
    throttle_1 = _clamp(thrust_base - asym, 0.0, 1.0)
    throttle_2 = _clamp(thrust_base + asym, 0.0, 1.0)

    # Trim slowly to unload sustained pitch command.
    _S["trim"] = _clamp(_S["trim"] + 0.0025 * _clamp(pitch_cmd, -0.7, 0.7) * dt * 20.0 - 0.0008 * pitch * dt * 20.0, -1.0, 1.0)

    # Flap schedule: conservative and retracted with speed.
    if spd < 140.0 and alt < 4000.0:
        flaps = 20
    elif spd < 175.0 and alt < 7000.0:
        flaps = 10
    elif spd < 215.0 and alt < 12000.0:
        flaps = 0
    else:
        flaps = 0
    if high_speed:
        flaps = 0

    speedbrake = 0.0
    if spd > tgt_spd + 20.0 and alt > 1500.0:
        speedbrake = _clamp((spd - tgt_spd - 20.0) / 60.0, 0.0, 1.0)
    if high_speed:
        speedbrake = max(speedbrake, _clamp((spd - 330.0) / 35.0, 0.0, 1.0))

    gear = 1 if alt < 800.0 and tgt_alt < 1500.0 else 0
    if bool(state.get("gear_handle_down", False)):
        gear = 1

    # Derive commanded elevator relative to trim.
    elevator = _clamp(pitch_cmd - _S["trim"] - 0.03 * pitch, -1.0, 1.0)
    aileron = _clamp(roll_cmd - 0.02 * roll, -1.0, 1.0)
    rudder = rudder_cmd

    # Extra stall recovery logic.
    if stall_warn:
        elevator = min(elevator, -0.15)
        aileron = _clamp(aileron, -0.4, 0.4)
        rudder = _clamp(rudder + 0.15 * math.copysign(1.0, n2 - n1 if abs(n2 - n1) > 1e-3 else 0.0), -1.0, 1.0)
        flaps = max(flaps, 10)
        gear = 1 if alt < 3000.0 else gear

    # Damper on strong turbulence.
    turb = float(state.get("turbulence_intensity", 0.0))
    if turb > 0.6:
        elevator *= 0.85
        aileron *= 0.85
        rudder *= 0.9

    out = {}
    if ap:
        out["elevator"] = float(_clamp(elevator, -1.0, 1.0))
        out["aileron"] = float(_clamp(aileron, -1.0, 1.0))
        out["rudder"] = float(_clamp(rudder, -1.0, 1.0))
        out["elevator_trim"] = float(_clamp(_S["trim"], -1.0, 1.0))
        out["flaps"] = int(_clamp(flaps, 0, 40))
        out["speedbrake"] = float(_clamp(speedbrake, 0.0, 1.0))
        out["gear"] = int(gear)
        if at:
            out["throttle_1"] = float(throttle_1)
            out["throttle_2"] = float(throttle_2)
    else:
        # If AP is off, still protect the aircraft a little when the state is clearly unsafe.
        if stall_warn:
            out["elevator"] = -0.2
            out["throttle_1"] = max(float(state.get("throttle_1_pos", 0.5)), 0.75)
            out["throttle_2"] = max(float(state.get("throttle_2_pos", 0.5)), 0.75)
            out["flaps"] = max(int(state.get("flaps", 0) or 0), 10)
            out["gear"] = 1 if alt < 3000.0 else int(state.get("gear_handle_down", 0))
    return out
