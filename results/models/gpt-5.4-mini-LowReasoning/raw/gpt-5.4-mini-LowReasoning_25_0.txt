using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private const double DaySeconds = 86400.0;
    private const double EarthRadius = 6371000.0;
    private const double EarthEscapeAltitude = 200000.0;
    private const double MinLaunchTauDays = 5.0 / 24.0;

    private readonly struct BodyState
    {
        public readonly string Name;
        public readonly double Mass;
        public readonly double X;
        public readonly double Y;
        public readonly double Z;
        public readonly double Vx;
        public readonly double Vy;
        public readonly double Vz;

        public BodyState(string name, double mass, double x, double y, double z, double vx, double vy, double vz)
        {
            Name = name;
            Mass = mass;
            X = x;
            Y = y;
            Z = z;
            Vx = vx;
            Vy = vy;
            Vz = vz;
        }

        public Vec3 Pos => new Vec3(X, Y, Z);
        public Vec3 Vel => new Vec3(Vx, Vy, Vz);
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double NormSq() => Dot(this);
        public double Norm() => Math.Sqrt(NormSq());
        public Vec3 Normalize()
        {
            var n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    private sealed class Tokens
    {
        private readonly string[] _tokens;
        private int _index;

        public Tokens(string input)
        {
            _tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            _index = 0;
        }

        public bool HasNext => _index < _tokens.Length;

        public string Next()
        {
            if (!HasNext) throw new EndOfStreamException();
            return _tokens[_index++];
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    public static void Main()
    {
        var input = Console.In.ReadToEnd();
        var tok = new Tokens(input);
        if (!tok.HasNext) return;

        int numBodies = tok.NextInt();
        double warningDays = tok.NextDouble();
        double impactorMass = tok.NextDouble();
        double dvBudget = tok.NextDouble();
        double asteroidMass = tok.NextDouble();
        double t0Unix = tok.NextDouble();

        var bodies = new Dictionary<string, BodyState>(StringComparer.OrdinalIgnoreCase);
        for (int i = 0; i < numBodies; i++)
        {
            string name = tok.Next();
            double mass = tok.NextDouble();
            double x = tok.NextDouble();
            double y = tok.NextDouble();
            double z = tok.NextDouble();
            double vx = tok.NextDouble();
            double vy = tok.NextDouble();
            double vz = tok.NextDouble();
            bodies[name] = new BodyState(name, mass, x, y, z, vx, vy, vz);
        }

        string bspPath = tok.HasNext ? tok.Next() : string.Empty;

        if (!bodies.TryGetValue("Earth", out var earth) || !bodies.TryGetValue("Asteroid", out var asteroid))
        {
            Console.WriteLine("0");
            return;
        }

        var best = FindBestTransfer(earth, asteroid, warningDays, dvBudget);

        var launchTime = best.LaunchTime;
        var impactTime = best.ImpactTime;
        var launchDv = best.LaunchDv;
        var correctionDv = best.CorrectionDv;
        var impactDv = best.ImpactDv;

        Console.WriteLine("3");
        WriteBurn(launchTime, launchDv);
        WriteBurn((launchTime + impactTime) * 0.5, correctionDv);
        WriteBurn(impactTime, impactDv);
    }

    private static void WriteBurn(double timeDays, Vec3 dv)
    {
        Console.WriteLine(string.Format(CultureInfo.InvariantCulture,
            "{0:R} {1:R} {2:R} {3:R}",
            timeDays, dv.X, dv.Y, dv.Z));
    }

    private sealed class Solution
    {
        public double LaunchTime;
        public double ImpactTime;
        public Vec3 LaunchDv;
        public Vec3 CorrectionDv;
        public Vec3 ImpactDv;
        public double Score;
    }

    private static Solution FindBestTransfer(BodyState earth, BodyState asteroid, double warningDays, double dvBudget)
    {
        var best = new Solution
        {
            Score = double.PositiveInfinity,
            LaunchTime = 0,
            ImpactTime = Math.Min(warningDays, 1.0),
            LaunchDv = new Vec3(0, 0, 0),
            CorrectionDv = new Vec3(0, 0, 0),
            ImpactDv = new Vec3(0, 0, 0)
        };

        var r0 = asteroid.Pos - earth.Pos;
        var v0 = asteroid.Vel - earth.Vel;
        double tEncounter = SolveClosestApproachTime(r0, v0, warningDays);
        if (double.IsNaN(tEncounter) || tEncounter < MinLaunchTauDays) tEncounter = Math.Min(warningDays, Math.Max(10.0, warningDays * 0.75));

        double interceptLead = Math.Max(2.0, Math.Min(tEncounter * 0.35, 60.0));
        double desiredImpact = Math.Max(MinLaunchTauDays * 2.0, tEncounter - interceptLead);
        desiredImpact = Math.Min(desiredImpact, warningDays);

        double launchGridStep = Math.Max(0.5, warningDays / 48.0);
        for (double launch = 0.0; launch <= warningDays; launch += launchGridStep)
        {
            double maxTau = Math.Max(MinLaunchTauDays, Math.Min(3650.0, warningDays - launch));
            double targetTau = Math.Max(MinLaunchTauDays, Math.Min(maxTau, desiredImpact - launch));

            foreach (double tau in SampleTauCandidates(targetTau, maxTau))
            {
                if (tau <= 0) continue;
                double impactTime = launch + tau;
                if (impactTime > warningDays + 1e-9) continue;

                var earthAtLaunch = PropagateLinear(earth, launch);
                var asteroidAtImpact = PropagateLinear(asteroid, impactTime);
                var desiredSpacecraftVel = (asteroidAtImpact.Pos - earthAtLaunch.Pos) / tau;
                var launchDv = desiredSpacecraftVel - earthAtLaunch.Vel;
                double launchDvMag = launchDv.Norm();
                if (launchDvMag > dvBudget * 1.5) continue;

                double remaining = Math.Max(0.0, dvBudget - launchDvMag);
                var corridor = asteroidAtImpact.Pos - (earthAtLaunch.Pos + earthAtLaunch.Vel * tau);
                double missLike = corridor.Norm();
                double penalty = missLike / (EarthRadius * 20.0);
                penalty += Math.Max(0.0, launchDvMag - dvBudget) * 1000.0;
                penalty += launch * 0.02;
                penalty -= tau * 0.0005;

                if (launchDvMag <= dvBudget)
                {
                    var corrTime = launch + tau * 0.5;
                    var correction = RemainingCorrection(earth, asteroid, launch, tau, launchDv, remaining);
                    var impactDv = FinalTrim(earth, asteroid, launch, tau, launchDv, correction, remaining);

                    double score = launchDvMag + correction.Norm() + impactDv.Norm() + penalty;
                    if (score < best.Score)
                    {
                        best.Score = score;
                        best.LaunchTime = launch;
                        best.ImpactTime = impactTime;
                        best.LaunchDv = launchDv;
                        best.CorrectionDv = correction;
                        best.ImpactDv = impactDv;
                    }
                }
            }
        }

        if (!double.IsFinite(best.Score))
        {
            var launch = Math.Min(Math.Max(0.0, warningDays * 0.05), warningDays);
            var impactTime = Math.Min(warningDays, Math.Max(launch + 20.0, warningDays * 0.9));
            var earthAtLaunch = PropagateLinear(earth, launch);
            var asteroidAtImpact = PropagateLinear(asteroid, impactTime);
            var desiredVel = (asteroidAtImpact.Pos - earthAtLaunch.Pos) / Math.Max(1e-6, impactTime - launch);
            var launchDv = desiredVel - earthAtLaunch.Vel;
            var launchMag = launchDv.Norm();
            if (launchMag > dvBudget && launchMag > 0) launchDv = launchDv * (dvBudget / launchMag);

            best.LaunchTime = launch;
            best.ImpactTime = impactTime;
            best.LaunchDv = launchDv;
            best.CorrectionDv = new Vec3(0, 0, 0);
            best.ImpactDv = new Vec3(0, 0, 0);
        }

        return best;
    }

    private static IEnumerable<double> SampleTauCandidates(double targetTau, double maxTau)
    {
        var seen = new HashSet<long>();
        double[] scales = { 0.15, 0.3, 0.5, 0.75, 1.0, 1.25, 1.6, 2.0 };
        foreach (var s in scales)
        {
            double tau = targetTau * s;
            if (tau > 0 && tau <= maxTau)
            {
                long key = (long)Math.Round(tau * 1000.0);
                if (seen.Add(key)) yield return tau;
            }
        }

        for (double tau = Math.Max(MinLaunchTauDays, targetTau * 0.5); tau <= maxTau; tau += Math.Max(0.5, maxTau / 30.0))
        {
            long key = (long)Math.Round(tau * 1000.0);
            if (seen.Add(key)) yield return tau;
        }
    }

    private static BodyState PropagateLinear(BodyState b, double days)
    {
        double s = days * DaySeconds;
        return new BodyState(b.Name, b.Mass,
            b.X + b.Vx * s,
            b.Y + b.Vy * s,
            b.Z + b.Vz * s,
            b.Vx, b.Vy, b.Vz);
    }

    private static double SolveClosestApproachTime(Vec3 r0, Vec3 v0, double maxDays)
    {
        double denom = v0.NormSq();
        if (denom <= 0) return double.NaN;
        double t = -r0.Dot(v0) / denom / DaySeconds;
        if (double.IsNaN(t) || double.IsInfinity(t)) return double.NaN;
        if (t < 0) t = 0;
        if (t > maxDays) t = maxDays;
        return t;
    }

    private static Vec3 RemainingCorrection(BodyState earth, BodyState asteroid, double launch, double tau, Vec3 launchDv, double remainingDv)
    {
        if (remainingDv <= 1e-9) return new Vec3(0, 0, 0);
        var earthAtLaunch = PropagateLinear(earth, launch);
        var spacecraftPosAtImpact = earthAtLaunch.Pos + (earthAtLaunch.Vel + launchDv) * (tau * DaySeconds);
        var asteroidAtImpact = PropagateLinear(asteroid, launch + tau);
        var miss = asteroidAtImpact.Pos - spacecraftPosAtImpact;
        var dir = miss.Normalize();
        if (dir.NormSq() <= 0) dir = asteroidAtImpact.Vel.Normalize();
        var corrMag = Math.Min(remainingDv * 0.35, miss.Norm() / Math.Max(1.0, tau * DaySeconds));
        return dir * corrMag;
    }

    private static Vec3 FinalTrim(BodyState earth, BodyState asteroid, double launch, double tau, Vec3 launchDv, Vec3 correction, double remainingDv)
    {
        if (remainingDv <= 1e-9) return new Vec3(0, 0, 0);
        var earthAtLaunch = PropagateLinear(earth, launch);
        var v = earthAtLaunch.Vel + launchDv + correction;
        var spacecraftPosAtImpact = earthAtLaunch.Pos + v * (tau * DaySeconds);
        var asteroidAtImpact = PropagateLinear(asteroid, launch + tau);
        var miss = asteroidAtImpact.Pos - spacecraftPosAtImpact;
        var dir = miss.Normalize();
        if (dir.NormSq() <= 0) dir = asteroidAtImpact.Vel.Normalize();
        double trim = Math.Min(remainingDv * 0.15, miss.Norm() / Math.Max(1.0, 0.25 * tau * DaySeconds));
        return dir * trim;
    }
}
