OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "proj"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%c0 = OpConstant %int 0
%c1 = OpConstant %int 1
%c2 = OpConstant %int 2
%c3 = OpConstant %int 3
%c4 = OpConstant %int 4
%c5 = OpConstant %int 5
%u6 = OpConstant %uint 6
%f0 = OpConstant %float 0
%f015 = OpConstant %float 0.015
%f03 = OpConstant %float 0.3
%f1 = OpConstant %float 1

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %u6
%arr_v3_in = OpTypeArray %v3float %u6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_ubo = OpTypePointer Uniform %UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %fn
%entry = OpLabel

%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %c0 %c0
%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %c2 %c0
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %c4 %c0
%a0ptr = OpAccessChain %ptr_in_v4 %gl_in %c1 %c0
%a1ptr = OpAccessChain %ptr_in_v4 %gl_in %c3 %c0
%a2ptr = OpAccessChain %ptr_in_v4 %gl_in %c5 %c0

%p0 = OpLoad %v4float %p0ptr
%p1 = OpLoad %v4float %p1ptr
%p2 = OpLoad %v4float %p2ptr
%a0 = OpLoad %v4float %a0ptr
%a1 = OpLoad %v4float %a1ptr
%a2 = OpLoad %v4float %a2ptr

%col0ptr = OpAccessChain %ptr_in_v3 %color %c0
%col2ptr = OpAccessChain %ptr_in_v3 %color %c2
%col4ptr = OpAccessChain %ptr_in_v3 %color %c4
%col0 = OpLoad %v3float %col0ptr
%col2 = OpLoad %v3float %col2ptr
%col4 = OpLoad %v3float %col4ptr
%fill0 = OpVectorTimesScalar %v3float %col0 %f03
%fill1 = OpVectorTimesScalar %v3float %col2 %f03
%fill2 = OpVectorTimesScalar %v3float %col4 %f03
%white = OpConstantComposite %v3float %f1 %f1 %f1

%outp = OpAccessChain %ptr_out_v4 %gl_out %c0

; Filled triangle
OpStore %outColor %fill0
OpStore %outp %p0
OpEmitVertex
OpStore %outColor %fill1
OpStore %outp %p1
OpEmitVertex
OpStore %outColor %fill2
OpStore %outp %p2
OpEmitVertex
OpEndPrimitive

; Edge 0 -> 2
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2
%p2w = OpCompositeExtract %float %p2 3
%p0yu = OpFAdd %float %p0y %f015
%p0yd = OpFSub %float %p0y %f015
%p2yu = OpFAdd %float %p2y %f015
%p2yd = OpFSub %float %p2y %f015
%q0a = OpCompositeConstruct %v4float %p0x %p0yu %p0z %p0w
%q0b = OpCompositeConstruct %v4float %p0x %p0yd %p0z %p0w
%q0c = OpCompositeConstruct %v4float %p2x %p2yu %p2z %p2w
%q0d = OpCompositeConstruct %v4float %p2x %p2yd %p2z %p2w
OpStore %outColor %white
OpStore %outp %q0a
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q0b
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q0c
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q0d
OpEmitVertex
OpEndPrimitive

; Edge 2 -> 4
%p4ptr = OpAccessChain %ptr_in_v4 %gl_in %c4 %c0
%p4 = OpLoad %v4float %p4ptr
%p4x = OpCompositeExtract %float %p4 0
%p4y = OpCompositeExtract %float %p4 1
%p4z = OpCompositeExtract %float %p4 2
%p4w = OpCompositeExtract %float %p4 3
%p4yu = OpFAdd %float %p4y %f015
%p4yd = OpFSub %float %p4y %f015
%q1a = OpCompositeConstruct %v4float %p2x %p2yu %p2z %p2w
%q1b = OpCompositeConstruct %v4float %p2x %p2yd %p2z %p2w
%q1c = OpCompositeConstruct %v4float %p4x %p4yu %p4z %p4w
%q1d = OpCompositeConstruct %v4float %p4x %p4yd %p4z %p4w
OpStore %outColor %white
OpStore %outp %q1a
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q1b
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q1c
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q1d
OpEmitVertex
OpEndPrimitive

; Edge 4 -> 0
%p0yu2 = OpFAdd %float %p0y %f015
%p0yd2 = OpFSub %float %p0y %f015
%q2a = OpCompositeConstruct %v4float %p4x %p4yu %p4z %p4w
%q2b = OpCompositeConstruct %v4float %p4x %p4yd %p4z %p4w
%q2c = OpCompositeConstruct %v4float %p0x %p0yu2 %p0z %p0w
%q2d = OpCompositeConstruct %v4float %p0x %p0yd2 %p0z %p0w
OpStore %outColor %white
OpStore %outp %q2a
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q2b
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q2c
OpEmitVertex
OpStore %outColor %white
OpStore %outp %q2d
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
