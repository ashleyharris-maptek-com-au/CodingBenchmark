#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <tuple>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u, v;
    long long w;
};

struct Adj {
    int to;
    int id;
    long long w;
};

struct DSU {
    vector<int> p, r;
    DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int x) { return p[x] == x ? x : p[x] = find(p[x]); }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static const long long INF = (1LL << 62);

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<Edge> edges(m);
    vector<vector<Adj>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        g[u].push_back({v, i, w});
        g[v].push_back({u, i, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];
    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    k = (int)terminals.size();

    if (k <= 1) {
        cout << 0 << " " << 0 << '\n';
        return 0;
    }

    vector<vector<long long>> dist(k, vector<long long>(n, INF));
    vector<vector<int>> parentEdge(k, vector<int>(n, -1));

    for (int si = 0; si < k; ++si) {
        int s = terminals[si];
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        dist[si][s] = 0;
        pq.push({0, s});
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d != dist[si][u]) continue;
            for (const auto &e : g[u]) {
                int v = e.to;
                long long nd = d + e.w;
                if (nd < dist[si][v]) {
                    dist[si][v] = nd;
                    parentEdge[si][v] = e.id;
                    pq.push({nd, v});
                }
            }
        }
    }

    // Prim on terminal metric closure.
    vector<long long> best(k, INF);
    vector<int> bestFrom(k, -1);
    vector<char> used(k, 0);
    best[0] = 0;
    for (int it = 0; it < k; ++it) {
        int x = -1;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && (x == -1 || best[i] < best[x])) x = i;
        }
        used[x] = 1;
        for (int y = 0; y < k; ++y) {
            if (used[y]) continue;
            long long d = dist[x][terminals[y]];
            if (d < best[y]) {
                best[y] = d;
                bestFrom[y] = x;
            }
        }
    }

    auto add_path_edges = [&](int fromTermIdx, int toVertex, unordered_set<int> &picked) {
        int cur = toVertex;
        while (cur != terminals[fromTermIdx]) {
            int eid = parentEdge[fromTermIdx][cur];
            if (eid < 0) break;
            picked.insert(eid);
            int a = edges[eid].u;
            int b = edges[eid].v;
            cur = (cur == a ? b : a);
        }
    };

    unordered_set<int> picked;
    picked.reserve((size_t)k * 32);

    for (int i = 1; i < k; ++i) {
        int a = i;
        int b = bestFrom[i];
        if (b < 0) continue;

        // Reconstruct the shortest path between terminal a and terminal b.
        int target = terminals[a];
        int source = terminals[b];

        // Path source -> target using parent edges from source terminal.
        vector<int> pathEdges;
        int cur = target;
        while (cur != source) {
            int eid = parentEdge[b][cur];
            if (eid < 0) break;
            pathEdges.push_back(eid);
            int uu = edges[eid].u;
            int vv = edges[eid].v;
            cur = (cur == uu ? vv : uu);
        }
        for (int eid : pathEdges) picked.insert(eid);
    }

    // If the closure MST missed anything due to disconnected terminals, fall back
    // to a simple star around the first terminal that can reach all others.
    if ((int)picked.empty()) {
        int s = terminals[0];
        for (int i = 1; i < k; ++i) {
            int cur = terminals[i];
            while (cur != s) {
                int eid = parentEdge[0][cur];
                if (eid < 0) break;
                picked.insert(eid);
                int uu = edges[eid].u;
                int vv = edges[eid].v;
                cur = (cur == uu ? vv : uu);
            }
        }
    }

    vector<int> candidate(picked.begin(), picked.end());
    sort(candidate.begin(), candidate.end());

    vector<int> verts;
    vector<char> inUsed(n, 0);
    for (int eid : candidate) {
        if (!inUsed[edges[eid].u]) {
            inUsed[edges[eid].u] = 1;
            verts.push_back(edges[eid].u);
        }
        if (!inUsed[edges[eid].v]) {
            inUsed[edges[eid].v] = 1;
            verts.push_back(edges[eid].v);
        }
    }
    for (int t : terminals) {
        if (!inUsed[t]) {
            inUsed[t] = 1;
            verts.push_back(t);
        }
    }

    DSU dsu(n);
    vector<int> treeEdges;
    long long total = 0;
    for (int eid : candidate) {
        int u = edges[eid].u, v = edges[eid].v;
        if (dsu.unite(u, v)) {
            treeEdges.push_back(eid);
            total += edges[eid].w;
        }
    }

    // If not all terminals are connected, add shortest-path edges from the first terminal.
    for (int ti = 1; ti < k; ++ti) {
        int a = terminals[0], b = terminals[ti];
        if (dsu.find(a) == dsu.find(b)) continue;
        vector<int> path;
        int cur = b;
        while (cur != a) {
            int eid = parentEdge[0][cur];
            if (eid < 0) break;
            path.push_back(eid);
            int uu = edges[eid].u;
            int vv = edges[eid].v;
            cur = (cur == uu ? vv : uu);
        }
        for (int eid : path) {
            if (dsu.unite(edges[eid].u, edges[eid].v)) {
                treeEdges.push_back(eid);
                total += edges[eid].w;
            }
        }
    }

    // Keep only a tree from the chosen subgraph.
    vector<int> finalEdges;
    DSU dsu2(n);
    sort(treeEdges.begin(), treeEdges.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });
    for (int eid : treeEdges) {
        if (dsu2.unite(edges[eid].u, edges[eid].v)) finalEdges.push_back(eid);
    }

    total = 0;
    for (int eid : finalEdges) total += edges[eid].w;

    cout << total << ' ' << finalEdges.size() << '\n';
    for (int eid : finalEdges) {
        cout << edges[eid].u << ' ' << edges[eid].v << '\n';
    }
    return 0;
}
