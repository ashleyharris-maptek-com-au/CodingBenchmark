float4 main(PSInput input) : SV_Target0
{
    float2 c = float2(input.uv.x * 3.0 - 2.0, input.uv.y * 2.0 - 1.0);
    float2 z = float2(0.0, 0.0);
    int iter = 0;

    [loop]
    for (int i = 0; i < 20; ++i)
    {
        float x = z.x * z.x - z.y * z.y + c.x;
        float y = 2.0 * z.x * z.y + c.y;
        z = float2(x, y);

        if (dot(z, z) > 4.0)
        {
            iter = i;
            break;
        }

        iter = i + 1;
    }

    float t = iter / 20.0;
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
