using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double PosWeight = 1.0;
    private const double VelWeight = 120.0;

    private struct Body
    {
        public double X, Y, Z, VX, VY, VZ;
    }

    public static void Main()
    {
        var tokens = ReadTokens();
        if (tokens.Count < 2) return;

        int p = 0;
        int n = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double mu = double.Parse(tokens[p++], CultureInfo.InvariantCulture);

        var start = ReadBody(tokens, ref p);
        var stations = new Body[n];
        for (int i = 0; i < n; i++) stations[i] = ReadBody(tokens, ref p);

        var order = Solve(stations, start, mu);
        Console.WriteLine(string.Join(" ", order));
    }

    private static List<string> ReadTokens()
    {
        var input = Console.In.ReadToEnd();
        var parts = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        return new List<string>(parts);
    }

    private static Body ReadBody(List<string> tokens, ref int p)
    {
        return new Body
        {
            X = double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            Y = double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            Z = double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            VX = double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            VY = double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            VZ = double.Parse(tokens[p++], CultureInfo.InvariantCulture)
        };
    }

    private static int[] Solve(Body[] stations, Body start, double mu)
    {
        int n = stations.Length;
        if (n <= 1) return Enumerable.Range(0, n).ToArray();

        double[] distFromStart = new double[n];
        for (int i = 0; i < n; i++) distFromStart[i] = Cost(start, stations[i], mu);

        int[] nearest = new int[n];
        double[] bestOut = new double[n];
        for (int i = 0; i < n; i++)
        {
            int best = -1;
            double bestCost = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                double c = Cost(stations[i], stations[j], mu);
                if (c < bestCost)
                {
                    bestCost = c;
                    best = j;
                }
            }
            nearest[i] = best;
            bestOut[i] = bestCost;
        }

        int startNode = 0;
        double bestStart = distFromStart[0];
        for (int i = 1; i < n; i++)
        {
            if (distFromStart[i] < bestStart)
            {
                bestStart = distFromStart[i];
                startNode = i;
            }
        }

        var order = GreedyOrder(stations, start, mu, startNode, distFromStart);

        if (n <= 20)
        {
            order = ImproveTwoOpt(order, stations, start, mu, 80);
        }
        else if (n <= 60)
        {
            order = ImproveTwoOpt(order, stations, start, mu, 40);
            order = LocalShuffleImprove(order, stations, start, mu, 25);
        }
        else
        {
            order = ImproveTwoOpt(order, stations, start, mu, 15);
        }

        return order;
    }

    private static int[] GreedyOrder(Body[] stations, Body start, double mu, int startNode, double[] distFromStart)
    {
        int n = stations.Length;
        var used = new bool[n];
        var order = new int[n];
        int curNode = startNode;
        Body cur = start;

        for (int step = 0; step < n; step++)
        {
            int best = -1;
            double bestScore = double.PositiveInfinity;

            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double score = step == 0 ? distFromStart[i] : Cost(cur, stations[i], mu);
                score += 0.05 * LookAheadPenalty(i, stations, used, mu);
                if (score < bestScore)
                {
                    bestScore = score;
                    best = i;
                }
            }

            order[step] = best;
            used[best] = true;
            cur = stations[best];
            curNode = best;
        }

        return order;
    }

    private static double LookAheadPenalty(int i, Body[] stations, bool[] used, double mu)
    {
        double best = double.PositiveInfinity;
        for (int j = 0; j < stations.Length; j++)
        {
            if (used[j] || j == i) continue;
            double c = Cost(stations[i], stations[j], mu);
            if (c < best) best = c;
        }
        return double.IsPositiveInfinity(best) ? 0.0 : best;
    }

    private static int[] ImproveTwoOpt(int[] order, Body[] stations, Body start, double mu, int passes)
    {
        int n = order.Length;
        if (n < 4) return order;

        double current = TotalCost(order, stations, start, mu);
        for (int pass = 0; pass < passes; pass++)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int k = i + 1; k < n; k++)
                {
                    double delta = DeltaTwoOpt(order, stations, start, mu, i, k);
                    if (delta < -1e-9)
                    {
                        Array.Reverse(order, i, k - i + 1);
                        current += delta;
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }
        return order;
    }

    private static int[] LocalShuffleImprove(int[] order, Body[] stations, Body start, double mu, int tries)
    {
        int n = order.Length;
        if (n < 8) return order;
        var rng = new Random(1234567);
        double best = TotalCost(order, stations, start, mu);
        var bestOrder = (int[])order.Clone();

        for (int t = 0; t < tries; t++)
        {
            var cand = (int[])bestOrder.Clone();
            int a = rng.Next(n);
            int b = rng.Next(n);
            if (a > b) (a, b) = (b, a);
            if (a == b) continue;
            Array.Reverse(cand, a, b - a + 1);
            cand = ImproveTwoOpt(cand, stations, start, mu, 8);
            double cost = TotalCost(cand, stations, start, mu);
            if (cost < best)
            {
                best = cost;
                bestOrder = cand;
            }
        }

        return bestOrder;
    }

    private static double TotalCost(int[] order, Body[] stations, Body start, double mu)
    {
        double sum = Cost(start, stations[order[0]], mu);
        for (int i = 1; i < order.Length; i++)
            sum += Cost(stations[order[i - 1]], stations[order[i]], mu);
        return sum;
    }

    private static double DeltaTwoOpt(int[] order, Body[] stations, Body start, double mu, int i, int k)
    {
        int n = order.Length;
        int a = order[i];
        int b = order[k];
        double before = 0, after = 0;

        if (i == 0)
        {
            before += Cost(start, stations[a], mu);
            after += Cost(start, stations[b], mu);
        }
        else
        {
            int p = order[i - 1];
            before += Cost(stations[p], stations[a], mu);
            after += Cost(stations[p], stations[b], mu);
        }

        if (k == n - 1)
        {
            // no outgoing edge
        }
        else
        {
            int q = order[k + 1];
            before += Cost(stations[b], stations[q], mu);
            after += Cost(stations[a], stations[q], mu);
        }

        return after - before;
    }

    private static double Cost(Body a, Body b, double mu)
    {
        double dx = a.X - b.X, dy = a.Y - b.Y, dz = a.Z - b.Z;
        double dvx = a.VX - b.VX, dvy = a.VY - b.VY, dvz = a.VZ - b.VZ;

        double pos = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double vel = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);

        double ra = Math.Sqrt(a.X * a.X + a.Y * a.Y + a.Z * a.Z);
        double rb = Math.Sqrt(b.X * b.X + b.Y * b.Y + b.Z * b.Z);
        double ca = CrossMag(a.X, a.Y, a.Z, a.VX, a.VY, a.VZ) / Math.Max(1e-9, ra);
        double cb = CrossMag(b.X, b.Y, b.Z, b.VX, b.VY, b.VZ) / Math.Max(1e-9, rb);
        double gravBias = Math.Abs(Math.Sqrt(mu / Math.Max(1e-9, ra)) - Math.Sqrt(mu / Math.Max(1e-9, rb)));

        return PosWeight * pos + VelWeight * vel + 0.1 * Math.Abs(ca - cb) + 0.01 * gravBias;
    }

    private static double CrossMag(double ax, double ay, double az, double bx, double by, double bz)
    {
        double cx = ay * bz - az * by;
        double cy = az * bx - ax * bz;
        double cz = ax * by - ay * bx;
        return Math.Sqrt(cx * cx + cy * cy + cz * cz);
    }
}
