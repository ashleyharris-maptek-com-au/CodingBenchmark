// C++20 lunar lander controller
#include <algorithm>
#include <atomic>
#include <chrono>
#include <cmath>
#include <condition_variable>
#include <cstdint>
#include <fstream>
#include <iostream>
#include <mutex>
#include <optional>
#include <string>
#include <thread>
#include <tuple>

using namespace std;

static constexpr double PI = 3.141592653589793238462643383279502884;

struct State {
    double x{}, y{}, z{};
    double vx{}, vy{}, vz{};
    double pitch{}, yaw{}, roll{};
    double pitch_rate{}, yaw_rate{}, roll_rate{};
    double fuel{};
    bool valid{false};
};

struct Vec3 {
    double x{}, y{}, z{};
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static Vec3 operator/(const Vec3& a, double s) { return {a.x / s, a.y / s, a.z / s}; }

static double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static double norm(const Vec3& a) { return sqrt(dot(a, a)); }

static Vec3 normalize(const Vec3& a) {
    double n = norm(a);
    if (n < 1e-9) return {0, 0, 1};
    return a / n;
}

static double clampd(double v, double lo, double hi) {
    return max(lo, min(hi, v));
}

struct SharedState {
    mutex m;
    condition_variable cv;
    State latest;
    bool has = false;
    bool done = false;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double width, depth, height, gravity, max_thrust, fuel0;
    if (!(cin >> width >> depth >> height >> gravity >> max_thrust >> fuel0)) return 0;

    double start_x, start_y, start_z, target_x, target_y, target_z;
    if (!(cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) return 0;

    string terrain_path;
    getline(cin >> ws, terrain_path);
    if (!terrain_path.empty()) {
        ifstream terrain(terrain_path, ios::binary);
        // Terrain is not required for the basic controller; consume the file if present.
    }

    SharedState shared;
    atomic<bool> reader_ok{true};

    thread reader([&] {
        while (true) {
            State s;
            if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                      >> s.pitch >> s.yaw >> s.roll
                      >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
                lock_guard<mutex> lk(shared.m);
                shared.done = true;
                shared.cv.notify_all();
                break;
            }
            s.valid = true;
            {
                lock_guard<mutex> lk(shared.m);
                shared.latest = s;
                shared.has = true;
            }
            shared.cv.notify_all();
        }
    });

    auto guidance = [&](const State& s) {
        const Vec3 pos{s.x, s.y, s.z};
        const Vec3 vel{s.vx, s.vy, s.vz};
        const Vec3 target{target_x, target_y, target_z};
        Vec3 to_target = target - pos;

        double dist_xy = hypot(to_target.x, to_target.y);
        double dist_z = to_target.z;

        // Two-stage guidance: approach laterally, then brake vertically near target.
        Vec3 acc{0, 0, 0};

        double kpos_xy = (dist_z > 30.0 ? 0.03 : 0.08);
        double kvel_xy = 0.35;
        double kpos_z = (dist_z > 30.0 ? 0.08 : 0.20);
        double kvel_z = 0.90;

        acc.x += kpos_xy * to_target.x - kvel_xy * vel.x;
        acc.y += kpos_xy * to_target.y - kvel_xy * vel.y;

        double desired_vz = 0.0;
        if (dist_z > 60.0) desired_vz = -min(12.0, dist_z * 0.15);
        else if (dist_z > 15.0) desired_vz = -min(6.0, dist_z * 0.25);
        else desired_vz = -max(0.8, min(3.0, dist_z * 0.25));
        acc.z += kpos_z * to_target.z + kvel_z * (desired_vz - vel.z) + gravity;

        // When close laterally, reduce sideways motion aggressively.
        if (dist_xy < max(5.0, min(width, depth) * 0.02 * 1.5)) {
            acc.x -= 0.6 * vel.x;
            acc.y -= 0.6 * vel.y;
        }

        Vec3 thrust_dir = normalize(acc);
        double thrust_mag = norm(acc) / max_thrust;
        thrust_mag = clampd(thrust_mag, 0.0, 1.0);

        // Rotate the lander so local up matches desired thrust direction.
        double yaw_cmd = 0.0;
        double pitch_cmd = 0.0;
        double roll_cmd = 0.0;

        double desired_yaw = atan2(thrust_dir.y, thrust_dir.x);
        double horiz = hypot(thrust_dir.x, thrust_dir.y);
        double desired_pitch = atan2(-horiz, thrust_dir.z);

        auto wrap_pi = [](double a) {
            while (a > PI) a -= 2.0 * PI;
            while (a < -PI) a += 2.0 * PI;
            return a;
        };

        double yaw_err = wrap_pi(desired_yaw - s.yaw);
        double pitch_err = desired_pitch - s.pitch;

        yaw_cmd = clampd(1.8 * yaw_err - 0.5 * s.yaw_rate, -1.0, 1.0);
        pitch_cmd = clampd(1.8 * pitch_err - 0.5 * s.pitch_rate, -1.0, 1.0);
        roll_cmd = clampd(-0.8 * s.roll - 0.3 * s.roll_rate, -1.0, 1.0);

        // Near touchdown, keep attitude simple and reduce oscillation.
        if (dist_z < 10.0 && dist_xy < max(8.0, min(width, depth) * 0.025)) {
            roll_cmd *= 0.2;
        }

        return tuple<double, double, double, double>{thrust_mag, pitch_cmd, yaw_cmd, roll_cmd};
    };

    unique_lock<mutex> lk(shared.m);
    while (true) {
        shared.cv.wait_for(lk, chrono::milliseconds(90), [&] { return shared.has || shared.done; });
        if (!shared.has && shared.done) break;
        if (!shared.has) continue;
        State s = shared.latest;
        shared.has = false;
        lk.unlock();

        auto [thrust, pitch_cmd, yaw_cmd, roll_cmd] = guidance(s);
        cout.setf(ios::fixed);
        cout.precision(6);
        cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n' << flush;

        lk.lock();
        if (shared.done) break;
    }

    if (reader.joinable()) reader.join();
    return 0;
}
