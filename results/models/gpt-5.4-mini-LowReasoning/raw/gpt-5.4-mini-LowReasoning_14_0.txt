using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    private struct Eq
    {
        public List<int> Vars;
        public int Need;
    }

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrEmpty(first))
        {
            return;
        }

        var header = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(header[0]);
        int w = int.Parse(header[1]);

        char[][] board = new char[h][];
        for (int r = 0; r < h; r++)
        {
            string? line = Console.ReadLine();
            if (line == null) line = string.Empty;
            if (line.Length < w) line = line.PadRight(w, ' ');
            board[r] = line.ToCharArray();
        }

        var state = new sbyte[h, w]; // 0 = unknown, 1 = safe, 2 = mine, 3 = revealed clue
        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                char ch = board[r][c];
                if (ch == ' ')
                {
                    state[r, c] = 0;
                }
                else if (ch == '*')
                {
                    state[r, c] = 2;
                }
                else
                {
                    state[r, c] = 3;
                }
            }
        }

        var safe = new bool[h, w];
        var mine = new bool[h, w];
        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                if (state[r, c] == 2) mine[r, c] = true;
            }
        }

        // Fast fixed-point deductions: if a clue has all mines accounted for, adjacent unknowns are safe.
        // If all remaining unknowns must be mines, mark them as such and iterate.
        bool changed;
        do
        {
            changed = false;
            for (int r = 0; r < h; r++)
            {
                for (int c = 0; c < w; c++)
                {
                    char ch = board[r][c];
                    if (ch < '0' || ch > '8') continue;

                    int need = ch - '0';
                    int unknownCount = 0;
                    List<(int rr, int cc)> unknowns = null!;

                    for (int dr = -1; dr <= 1; dr++)
                    {
                        int nr = r + dr;
                        if ((uint)nr >= (uint)h) continue;
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            if (dr == 0 && dc == 0) continue;
                            int nc = c + dc;
                            if ((uint)nc >= (uint)w) continue;
                            if (state[nr, nc] == 2) need--;
                            else if (state[nr, nc] == 0)
                            {
                                unknownCount++;
                                unknowns ??= new List<(int, int)>();
                                unknowns.Add((nr, nc));
                            }
                        }
                    }

                    if (need < 0 || need > unknownCount) continue;

                    if (need == 0 && unknowns != null)
                    {
                        foreach (var (nr, nc) in unknowns)
                        {
                            if (!safe[nr, nc])
                            {
                                safe[nr, nc] = true;
                                state[nr, nc] = 1;
                                changed = true;
                            }
                        }
                    }
                    else if (need == unknownCount && unknowns != null)
                    {
                        foreach (var (nr, nc) in unknowns)
                        {
                            if (!mine[nr, nc])
                            {
                                mine[nr, nc] = true;
                                state[nr, nc] = 2;
                                changed = true;
                            }
                        }
                    }
                }
            }
        } while (changed);

        // Build exact local constraints over remaining unknown frontier cells.
        var varId = new int[h, w];
        Array.Fill(varId, -1);
        var vars = new List<(int r, int c)>();
        var eqs = new List<Eq>();

        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                char ch = board[r][c];
                if (ch < '0' || ch > '8') continue;

                int need = ch - '0';
                var list = new List<int>(8);
                for (int dr = -1; dr <= 1; dr++)
                {
                    int nr = r + dr;
                    if ((uint)nr >= (uint)h) continue;
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        int nc = c + dc;
                        if ((uint)nc >= (uint)w) continue;
                        if (mine[nr, nc]) need--;
                        else if (state[nr, nc] == 0)
                        {
                            int id = varId[nr, nc];
                            if (id < 0)
                            {
                                id = vars.Count;
                                varId[nr, nc] = id;
                                vars.Add((nr, nc));
                            }
                            list.Add(id);
                        }
                    }
                }

                if (list.Count > 0)
                {
                    if (need < 0 || need > list.Count)
                    {
                        continue;
                    }
                    eqs.Add(new Eq { Vars = list, Need = need });
                }
            }
        }

        // Exact inference on small connected components of the frontier graph.
        int n = vars.Count;
        if (n > 0 && eqs.Count > 0)
        {
            var varToEqs = new List<int>[n];
            for (int i = 0; i < n; i++) varToEqs[i] = new List<int>();
            for (int ei = 0; ei < eqs.Count; ei++)
            {
                foreach (int v in eqs[ei].Vars)
                {
                    varToEqs[v].Add(ei);
                }
            }

            var seen = new bool[n];
            int threshold = 24;
            for (int start = 0; start < n; start++)
            {
                if (seen[start]) continue;

                var componentVars = new List<int>();
                var componentEqsSet = new HashSet<int>();
                var queue = new Queue<int>();
                queue.Enqueue(start);
                seen[start] = true;

                while (queue.Count > 0)
                {
                    int v = queue.Dequeue();
                    componentVars.Add(v);
                    foreach (int ei in varToEqs[v])
                    {
                        if (componentEqsSet.Add(ei))
                        {
                            foreach (int u in eqs[ei].Vars)
                            {
                                if (!seen[u])
                                {
                                    seen[u] = true;
                                    queue.Enqueue(u);
                                }
                            }
                        }
                    }
                }

                if (componentVars.Count > threshold)
                {
                    continue;
                }

                var componentEqs = componentEqsSet.Select(i => eqs[i]).ToList();
                SolveComponent(componentVars, componentEqs, vars, safe);
            }
        }

        var output = new List<(int x, int y)>();
        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                if (safe[r, c] && board[r][c] == ' ')
                {
                    output.Add((c, r));
                }
            }
        }

        Console.WriteLine(output.Count);
        foreach (var (x, y) in output)
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    private static void SolveComponent(List<int> componentVars, List<Eq> componentEqs, List<(int r, int c)> vars, bool[,] safe)
    {
        int k = componentVars.Count;
        if (k == 0) return;

        var localIndex = new Dictionary<int, int>(k);
        for (int i = 0; i < k; i++) localIndex[componentVars[i]] = i;

        var eqVars = new int[componentEqs.Count][];
        var need = new int[componentEqs.Count];
        var unassigned = new int[componentEqs.Count];
        var varToEqs = new List<int>[k];
        for (int i = 0; i < k; i++) varToEqs[i] = new List<int>();

        for (int ei = 0; ei < componentEqs.Count; ei++)
        {
            var src = componentEqs[ei];
            var arr = new int[src.Vars.Count];
            for (int j = 0; j < src.Vars.Count; j++)
            {
                int lv = localIndex[src.Vars[j]];
                arr[j] = lv;
                varToEqs[lv].Add(ei);
            }
            eqVars[ei] = arr;
            need[ei] = src.Need;
            unassigned[ei] = arr.Length;
        }

        var assign = new sbyte[k];
        var mustBeSafe = new bool[k];
        var canBeMine = new bool[k];

        bool ApplyDeterministic()
        {
            bool progress;
            do
            {
                progress = false;
                for (int ei = 0; ei < componentEqs.Count; ei++)
                {
                    if (need[ei] < 0 || need[ei] > unassigned[ei]) return false;
                    if (unassigned[ei] == 0) continue;
                    if (need[ei] != 0 && need[ei] != unassigned[ei]) continue;

                    int forcedVal = need[ei] == 0 ? 0 : 1;
                    foreach (int v in eqVars[ei])
                    {
                        if (assign[v] == -1)
                        {
                            assign[v] = (sbyte)forcedVal;
                            for (int j = 0; j < varToEqs[v].Count; j++)
                            {
                                int ej = varToEqs[v][j];
                                if (forcedVal == 1) need[ej]--;
                                unassigned[ej]--;
                            }
                            progress = true;
                        }
                        else if (assign[v] != forcedVal)
                        {
                            return false;
                        }
                    }
                }
            }
            while (progress);
            return true;
        }

        bool Search()
        {
            var savedNeed = (int[])need.Clone();
            var savedUnassigned = (int[])unassigned.Clone();
            var savedAssign = (sbyte[])assign.Clone();

            if (!ApplyDeterministic())
            {
                need = savedNeed;
                unassigned = savedUnassigned;
                assign = savedAssign;
                return false;
            }

            int chosen = -1;
            int bestDeg = -1;
            for (int i = 0; i < k; i++)
            {
                if (assign[i] != -1) continue;
                int deg = varToEqs[i].Count;
                if (deg > bestDeg)
                {
                    bestDeg = deg;
                    chosen = i;
                }
            }

            bool any = false;
            if (chosen == -1)
            {
                for (int i = 0; i < k; i++)
                {
                    if (assign[i] == 1) canBeMine[i] = true;
                    if (assign[i] == 0) mustBeSafe[i] = true;
                }
                any = true;
            }
            else
            {
                for (int branch = 0; branch < 2; branch++)
                {
                    var branchNeed = (int[])need.Clone();
                    var branchUnassigned = (int[])unassigned.Clone();
                    var branchAssign = (sbyte[])assign.Clone();
                    branchAssign[chosen] = (sbyte)branch;
                    need = branchNeed;
                    unassigned = branchUnassigned;
                    assign = branchAssign;
                    if (Search()) any = true;
                }
            }

            need = savedNeed;
            unassigned = savedUnassigned;
            assign = savedAssign;
            return any;
        }

        Search();

        for (int i = 0; i < k; i++)
        {
            var (r, c) = vars[componentVars[i]];
            if (mustBeSafe[i] && !canBeMine[i])
            {
                safe[r, c] = true;
            }
        }
    }
}
