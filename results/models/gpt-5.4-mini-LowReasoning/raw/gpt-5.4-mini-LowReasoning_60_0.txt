const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCountRaw = Number(data.machineCount);
const tasksRaw = Array.isArray(data.tasks) ? data.tasks : [];
const machineCount = Number.isFinite(machineCountRaw) && machineCountRaw > 0 ? Math.floor(machineCountRaw) : 1;

if (tasksRaw.length === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const n = tasksRaw.length;
const idToIndex = new Map();
for (let i = 0; i < n; i++) idToIndex.set(tasksRaw[i].id, i);

const duration = new Array(n);
const deadline = new Array(n);
const weight = new Array(n);
const deps = new Array(n);
const succ = Array.from({ length: n }, () => []);
const indeg = new Array(n).fill(0);

for (let i = 0; i < n; i++) {
  const t = tasksRaw[i] || {};
  duration[i] = Number.isFinite(t.duration) ? Math.max(0, Math.floor(t.duration)) : 0;
  deadline[i] = Number.isFinite(t.deadline) ? Math.floor(t.deadline) : 0;
  weight[i] = Number.isFinite(t.weight) ? Number(t.weight) : 0;
  const rawDeps = Array.isArray(t.deps) ? t.deps : [];
  const dlist = [];
  for (let j = 0; j < rawDeps.length; j++) {
    const depIdx = idToIndex.get(rawDeps[j]);
    if (depIdx !== undefined) {
      dlist.push(depIdx);
      succ[depIdx].push(i);
      indeg[i]++;
    }
  }
  deps[i] = dlist;
}

const topo = [];
const q = [];
for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  topo.push(u);
  const out = succ[u];
  for (let k = 0; k < out.length; k++) {
    const v = out[k];
    indeg[v]--;
    if (indeg[v] === 0) q.push(v);
  }
}

if (topo.length !== n) {
  const fallback = tasksRaw.slice().map(t => t.id);
  process.stdout.write(JSON.stringify({ order: fallback }));
  process.exit(0);
}

const crit = new Array(n).fill(0);
for (let i = topo.length - 1; i >= 0; i--) {
  const u = topo[i];
  let bestChild = 0;
  const out = succ[u];
  for (let k = 0; k < out.length; k++) {
    const v = out[k];
    if (crit[v] > bestChild) bestChild = crit[v];
  }
  crit[u] = duration[u] + bestChild;
}

const depDone = new Array(n).fill(0);
const remainingDeps = new Array(n);
for (let i = 0; i < n; i++) remainingDeps[i] = deps[i].length;

class MaxHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  size() { return this.a.length; }
  push(x) {
    const a = this.a;
    let i = a.length;
    a.push(x);
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[p], x) >= 0) break;
      a[i] = a[p];
      i = p;
    }
    a[i] = x;
  }
  pop() {
    const a = this.a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const x = a.pop();
    if (n > 1) {
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let best = l;
        if (r < a.length && this.cmp(a[r], a[l]) > 0) best = r;
        if (this.cmp(a[best], x) <= 0) break;
        a[i] = a[best];
        i = best;
      }
      a[i] = x;
    }
    return top;
  }
}

class MinHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  size() { return this.a.length; }
  push(x) {
    const a = this.a;
    let i = a.length;
    a.push(x);
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[p], x) <= 0) break;
      a[i] = a[p];
      i = p;
    }
    a[i] = x;
  }
  pop() {
    const a = this.a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const x = a.pop();
    if (n > 1) {
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let best = l;
        if (r < a.length && this.cmp(a[r], a[l]) < 0) best = r;
        if (this.cmp(a[best], x) >= 0) break;
        a[i] = a[best];
        i = best;
      }
      a[i] = x;
    }
    return top;
  }
}

function taskScore(i) {
  const dd = deadline[i];
  const w = weight[i];
  const d = duration[i];
  const cp = crit[i];
  const urgency = dd === 0 ? cp : 1e6 / (1 + Math.max(0, dd - cp));
  return w * 1e9 + cp * 1e6 + urgency * 1e3 - d;
}

const ready = new MaxHeap((a, b) => {
  const sa = a.score, sb = b.score;
  if (sa !== sb) return sa - sb;
  if (crit[a.i] !== crit[b.i]) return crit[a.i] - crit[b.i];
  if (weight[a.i] !== weight[b.i]) return weight[a.i] - weight[b.i];
  if (duration[a.i] !== duration[b.i]) return duration[b.i] - duration[a.i];
  return b.i - a.i;
});

for (let i = 0; i < n; i++) {
  if (remainingDeps[i] === 0) ready.push({ i, score: taskScore(i) });
}

const machines = new MinHeap((a, b) => {
  if (a.free !== b.free) return a.free - b.free;
  return a.idx - b.idx;
});
for (let m = 0; m < machineCount; m++) machines.push({ free: 0, idx: m });

const scheduled = new Array(n);
let doneCount = 0;

while (doneCount < n) {
  if (ready.size() === 0) break;
  const task = ready.pop();
  const i = task.i;
  const machine = machines.pop();
  const start = Math.max(machine.free, depDone[i]);
  const finish = start + duration[i];
  scheduled[i] = { id: tasksRaw[i].id, start, machine: machine.idx };
  machine.free = finish;
  machines.push(machine);
  doneCount++;

  const out = succ[i];
  for (let k = 0; k < out.length; k++) {
    const v = out[k];
    if (depDone[v] < finish) depDone[v] = finish;
    remainingDeps[v]--;
    if (remainingDeps[v] === 0) ready.push({ i: v, score: taskScore(v) });
  }
}

if (doneCount !== n) {
  const order = topo.map(i => tasksRaw[i].id);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

scheduled.sort((a, b) => {
  if (a.start !== b.start) return a.start - b.start;
  if (a.machine !== b.machine) return a.machine - b.machine;
  if (a.id < b.id) return -1;
  if (a.id > b.id) return 1;
  return 0;
});

process.stdout.write(JSON.stringify({ schedule: scheduled }));
