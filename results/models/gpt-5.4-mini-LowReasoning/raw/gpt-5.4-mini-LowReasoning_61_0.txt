declare const require: any;
const fs = require('fs');

type PackageItem = {
  id: string | number;
  x: number;
  y: number;
};

type InputData = {
  depot: [number, number];
  truckCount: number;
  packages: PackageItem[];
};

type Point = PackageItem & {
  angle: number;
  dist2: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as InputData;

const depotX = data.depot[0];
const depotY = data.depot[1];
const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

function sqr(v: number): number {
  return v * v;
}

function dist2Ax(x1: number, y1: number, x2: number, y2: number): number {
  return sqr(x1 - x2) + sqr(y1 - y2);
}

function angleFromDepot(p: PackageItem): number {
  return Math.atan2(p.y - depotY, p.x - depotX);
}

const pts: Point[] = packages.map((p) => ({
  id: p.id,
  x: p.x,
  y: p.y,
  angle: angleFromDepot(p),
  dist2: dist2Ax(p.x, p.y, depotX, depotY),
}));

pts.sort((a, b) => {
  if (a.angle !== b.angle) return a.angle - b.angle;
  if (a.dist2 !== b.dist2) return a.dist2 - b.dist2;
  const ai = String(a.id);
  const bi = String(b.id);
  return ai < bi ? -1 : ai > bi ? 1 : 0;
});

const routeCount = Math.max(0, Math.min(Math.floor(data.truckCount || 0), n));
const routes: (string | number)[][] = [];

if (n === 0 || routeCount === 0) {
  console.log(JSON.stringify({ routes }));
  process.exit(0);
}

const sizes: number[] = [];
const base = Math.floor(n / routeCount);
let extra = n % routeCount;
for (let i = 0; i < routeCount; i++) {
  sizes.push(base + (extra > 0 ? 1 : 0));
  if (extra > 0) extra--;
}

let index = 0;
for (let r = 0; r < routeCount; r++) {
  const size = sizes[r];
  const cluster = pts.slice(index, index + size);
  index += size;

  if (cluster.length === 0) {
    routes.push([]);
    continue;
  }

  if (cluster.length === 1) {
    routes.push([cluster[0].id]);
    continue;
  }

  let cx = 0;
  let cy = 0;
  for (const p of cluster) {
    cx += p.x;
    cy += p.y;
  }
  cx /= cluster.length;
  cy /= cluster.length;

  const ordered = cluster.slice().sort((a, b) => {
    const aa = Math.atan2(a.y - cy, a.x - cx);
    const ba = Math.atan2(b.y - cy, b.x - cx);
    if (aa !== ba) return aa - ba;
    const ad = a.dist2;
    const bd = b.dist2;
    if (ad !== bd) return ad - bd;
    const ai = String(a.id);
    const bi = String(b.id);
    return ai < bi ? -1 : ai > bi ? 1 : 0;
  });

  let bestStart = 0;
  let bestCost = Infinity;
  const m = ordered.length;
  for (let i = 0; i < m; i++) {
    const first = ordered[i];
    const last = ordered[(i + m - 1) % m];
    const cost = Math.sqrt(dist2Ax(depotX, depotY, first.x, first.y)) +
      Math.sqrt(dist2Ax(depotX, depotY, last.x, last.y));
    if (cost < bestCost) {
      bestCost = cost;
      bestStart = i;
    }
  }

  const route: (string | number)[] = [];
  for (let i = 0; i < m; i++) {
    route.push(ordered[(bestStart + i) % m].id);
  }
  routes.push(route);
}

console.log(JSON.stringify({ routes }));
