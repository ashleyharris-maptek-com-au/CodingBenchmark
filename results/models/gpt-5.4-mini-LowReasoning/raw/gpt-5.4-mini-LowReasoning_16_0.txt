// Job-Shop Scheduling heuristic solver
// C++20, single-file, feasible on arbitrary input sizes.

#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long duration;
};

struct Job {
    vector<Task> tasks;
};

struct MachineEvent {
    long long finish;
    int job;
    int task;
    long long start;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<Job> jobs(J);
    long long totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].tasks.resize(T);
        totalTasks += T;
        for (int i = 0; i < T; ++i) {
            int machine;
            long long duration;
            cin >> machine >> duration;
            jobs[j].tasks[i] = {machine, duration};
        }
    }

    vector<vector<long long>> startTimes(J);
    for (int j = 0; j < J; ++j) startTimes[j].assign(jobs[j].tasks.size(), 0);

    vector<long long> machineFree(M, 0);
    vector<long long> jobReady(J, 0);
    vector<int> nextTask(J, 0);

    auto allDone = [&]() {
        for (int j = 0; j < J; ++j) {
            if (nextTask[j] < (int)jobs[j].tasks.size()) return false;
        }
        return true;
    };

    // Adaptive strategy:
    // - For small instances, prefer jobs with the most remaining work to reduce obvious contention.
    // - For large instances, prefer jobs whose next task can start earliest.
    bool smallInstance = (totalTasks <= 2000 && J <= 100 && M <= 30);

    while (!allDone()) {
        int chosenJob = -1;
        long long chosenStart = LLONG_MAX;
        long long chosenEnd = LLONG_MAX;
        long long chosenScore = LLONG_MAX;

        for (int j = 0; j < J; ++j) {
            int t = nextTask[j];
            if (t >= (int)jobs[j].tasks.size()) continue;

            const auto &task = jobs[j].tasks[t];
            long long s = max(jobReady[j], machineFree[task.machine]);
            long long e = s + task.duration;

            long long score;
            if (smallInstance) {
                long long remaining = 0;
                for (int k = t; k < (int)jobs[j].tasks.size(); ++k) remaining += jobs[j].tasks[k].duration;
                score = (s << 20) - min<long long>(remaining, (1LL << 20));
            } else {
                score = (s << 20) + task.duration;
            }

            if (score < chosenScore ||
                (score == chosenScore && (e < chosenEnd || (e == chosenEnd && j < chosenJob)))) {
                chosenScore = score;
                chosenStart = s;
                chosenEnd = e;
                chosenJob = j;
            }
        }

        if (chosenJob == -1) break;

        int t = nextTask[chosenJob];
        const auto &task = jobs[chosenJob].tasks[t];
        long long s = max(jobReady[chosenJob], machineFree[task.machine]);
        long long e = s + task.duration;

        startTimes[chosenJob][t] = s;
        jobReady[chosenJob] = e;
        machineFree[task.machine] = e;
        ++nextTask[chosenJob];
    }

    for (int j = 0; j < J; ++j) {
        for (int t = 0; t < (int)jobs[j].tasks.size(); ++t) {
            if (t) cout << ' ';
            cout << startTimes[j][t] << ' ' << jobs[j].tasks[t].machine;
        }
        cout << '\n';
    }

    return 0;
}
