using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= ' ' && c != -1);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            long val = 0;
            while (c > ' ' && c != -1)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }

        public int NextInt() => (int)NextLong();
    }

    private struct Item
    {
        public long Score;
        public int Index;
    }

    private static void SortByScoreDesc(Item[] items)
    {
        Array.Sort(items, (a, b) =>
        {
            int cmp = b.Score.CompareTo(a.Score);
            if (cmp != 0) return cmp;
            return a.Index.CompareTo(b.Index);
        });
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long[][] flow = new long[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new long[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextLong();
        }

        long[][] dist = new long[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new long[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextLong();
        }

        var facility = new Item[n];
        var location = new Item[n];

        for (int i = 0; i < n; i++)
        {
            long sum = 0;
            for (int j = 0; j < n; j++) sum += flow[i][j] + flow[j][i];
            facility[i] = new Item { Score = sum, Index = i };
        }

        for (int i = 0; i < n; i++)
        {
            long sum = 0;
            for (int j = 0; j < n; j++) sum += dist[i][j] + dist[j][i];
            location[i] = new Item { Score = sum, Index = i };
        }

        SortByScoreDesc(facility);
        SortByScoreDesc(location);

        int[] perm = new int[n];
        for (int i = 0; i < n; i++) perm[facility[i].Index] = location[n - 1 - i].Index;

        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            long[] flowRow = flow[i];
            for (int j = 0; j < n; j++)
            {
                total += flowRow[j] * dist[pi][perm[j]];
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(total.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }
}
