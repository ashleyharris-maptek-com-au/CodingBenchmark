; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 20
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpName %main "main"
OpName %uv "uv"
OpName %outColor "outColor"
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%uv = OpVariable %ptr_input_v2float Input
%outColor = OpVariable %ptr_output_v4float Output
%main = OpFunction %void None %func
%label = OpLabel
%uv_val = OpLoad %v2float %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%rgb = OpCompositeConstruct %v4float %uv_x %uv_y %f0 %f1
OpStore %outColor %rgb
OpReturn
OpFunctionEnd
