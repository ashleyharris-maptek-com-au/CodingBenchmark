use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone, Copy, Eq, PartialEq)]
struct Item {
    deg: usize,
    v: usize,
}

impl Ord for Item {
    fn cmp(&self, other: &Self) -> Ordering {
        self.deg.cmp(&other.deg).then_with(|| self.v.cmp(&other.v))
    }
}

impl PartialOrd for Item {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for a in &mut adj {
        a.sort_unstable();
        a.dedup();
    }

    let mut heap = BinaryHeap::with_capacity(n);
    for v in 0..n {
        heap.push(Item { deg: adj[v].len(), v });
    }

    let mut selected = vec![false; n];
    let mut dominated = vec![false; n];
    let mut dom_count = vec![0u32; n];
    let mut ans = Vec::new();
    let mut undominated = n;

    while undominated > 0 {
        let v = loop {
            let item = heap.pop().unwrap();
            if !selected[item.v] && !dominated[item.v] {
                break item.v;
            }
        };

        selected[v] = true;
        ans.push(v);

        if !dominated[v] {
            dominated[v] = true;
            undominated -= 1;
        }
        for &to in &adj[v] {
            if !dominated[to] {
                dominated[to] = true;
                undominated -= 1;
            }
        }
    }

    for &v in &ans {
        dom_count[v] += 1;
        for &to in &adj[v] {
            dom_count[to] += 1;
        }
    }

    let mut keep = vec![true; n];
    for &v in ans.iter().rev() {
        let mut removable = true;
        if dom_count[v] == 1 {
            removable = false;
        } else {
            for &to in &adj[v] {
                if dom_count[to] == 1 {
                    removable = false;
                    break;
                }
            }
        }

        if removable {
            keep[v] = false;
            dom_count[v] -= 1;
            for &to in &adj[v] {
                dom_count[to] -= 1;
            }
        }
    }

    let mut out = Vec::new();
    let mut final_set = Vec::new();
    for &v in &ans {
        if keep[v] {
            final_set.push(v);
        }
    }

    out.push(final_set.len().to_string());
    if !final_set.is_empty() {
        let line = final_set.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" ");
        out.push(line);
    } else {
        out.push(String::new());
    }

    print!("{}", out.join("\n"));
}
