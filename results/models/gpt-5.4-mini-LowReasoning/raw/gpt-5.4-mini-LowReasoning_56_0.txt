const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};
const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const px = Number(depot[0]) || 0;
const py = Number(depot[1]) || 0;
const packages = Array.isArray(data.packages) ? data.packages : [];

function dist2(x1, y1, x2, y2) {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return dx * dx + dy * dy;
}

if (packages.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// A deterministic, scalable heuristic: order packages by polar angle around the depot.
// This tends to keep the route spatially coherent while remaining O(n log n).
const pts = packages.map((p, i) => {
  const x = Number(p.x) || 0;
  const y = Number(p.y) || 0;
  return {
    id: p.id,
    x,
    y,
    ang: Math.atan2(y - py, x - px),
    r2: dist2(x, y, px, py),
    idx: i,
  };
});

// Start the sweep near the farthest point from the depot to reduce an early long jump.
pts.sort((a, b) => {
  if (b.r2 !== a.r2) return b.r2 - a.r2;
  if (a.ang !== b.ang) return a.ang - b.ang;
  return a.idx - b.idx;
});
const startAngle = pts[0].ang;

function normAngle(a) {
  let x = a - startAngle;
  while (x < 0) x += Math.PI * 2;
  while (x >= Math.PI * 2) x -= Math.PI * 2;
  return x;
}

pts.sort((a, b) => {
  const da = normAngle(a.ang);
  const db = normAngle(b.ang);
  if (da !== db) return da - db;
  if (a.r2 !== b.r2) return a.r2 - b.r2;
  return a.idx - b.idx;
});

// Lightweight local improvement on adjacent edges within a fixed window.
// This is cheap and helps on clustered inputs without risking quadratic work.
const order = pts;
const n = order.length;
const window = Math.min(50, n);
for (let pass = 0; pass < 2; pass++) {
  for (let i = 0; i + 2 < n; i++) {
    const a = i === 0 ? { x: px, y: py } : order[i - 1];
    const b = order[i];
    let bestJ = -1;
    let bestGain = 0;
    const maxJ = Math.min(n - 1, i + window);
    for (let j = i + 1; j <= maxJ; j++) {
      const c = order[j];
      const d = j + 1 < n ? order[j + 1] : { x: px, y: py };
      const current = Math.sqrt(dist2(a.x, a.y, b.x, b.y)) + Math.sqrt(dist2(c.x, c.y, d.x, d.y));
      const swapped = Math.sqrt(dist2(a.x, a.y, c.x, c.y)) + Math.sqrt(dist2(b.x, b.y, d.x, d.y));
      const gain = current - swapped;
      if (gain > bestGain + 1e-12) {
        bestGain = gain;
        bestJ = j;
      }
    }
    if (bestJ > i) {
      for (let l = i, r = bestJ; l < r; l++, r--) {
        const tmp = order[l];
        order[l] = order[r];
        order[r] = tmp;
      }
    }
  }
}

const routes = [order.map(p => p.id)];
process.stdout.write(JSON.stringify({ routes }));
