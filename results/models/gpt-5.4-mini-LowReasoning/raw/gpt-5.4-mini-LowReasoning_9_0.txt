using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Cut
    {
        public int Length;
        public int Index;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Length = fs.NextInt(), Index = i };
        }

        Array.Sort(cuts, (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            return cmp != 0 ? cmp : a.Index.CompareTo(b.Index);
        });

        var bins = new List<List<int>>(Math.Max(1, n));
        var remaining = new List<int>(Math.Max(1, n));
        var tree = new SegmentTree(Math.Max(1, n));

        foreach (var cut in cuts)
        {
            int bin = tree.FindFirstAtLeast(cut.Length);
            if (bin < 0)
            {
                bin = bins.Count;
                bins.Add(new List<int>());
                remaining.Add(stockLength);
                tree.EnsureCapacity(bins.Count);
            }

            remaining[bin] -= cut.Length;
            bins[bin].Add(cut.Index);
            tree.Update(bin, remaining[bin]);
        }

        var sb = new StringBuilder();
        sb.AppendLine(bins.Count.ToString());
        for (int i = 0; i < bins.Count; i++)
        {
            var bin = bins[i];
            for (int j = 0; j < bin.Count; j++)
            {
                if (j != 0) sb.Append(' ');
                sb.Append(bin[j]);
            }
            sb.AppendLine();
        }

        Console.Write(sb.ToString());
    }

    private sealed class SegmentTree
    {
        private int size;
        private int[] tree;

        public SegmentTree(int capacity)
        {
            size = 1;
            while (size < capacity) size <<= 1;
            tree = new int[size << 1];
        }

        public void EnsureCapacity(int capacity)
        {
            if (capacity <= size) return;

            int newSize = size;
            while (newSize < capacity) newSize <<= 1;
            var newTree = new int[newSize << 1];
            Array.Copy(tree, 0, newTree, 0, tree.Length);
            tree = newTree;
            size = newSize;
        }

        public void Update(int index, int value)
        {
            int pos = index + size;
            tree[pos] = value;
            pos >>= 1;
            while (pos > 0)
            {
                int left = tree[pos << 1];
                int right = tree[(pos << 1) | 1];
                tree[pos] = left >= right ? left : right;
                pos >>= 1;
            }
        }

        public int FindFirstAtLeast(int need)
        {
            if (tree[1] < need) return -1;

            int pos = 1;
            int left = 0;
            int right = size;
            while (pos < size)
            {
                int leftChild = pos << 1;
                if (tree[leftChild] >= need)
                {
                    pos = leftChild;
                    right = (left + right) >> 1;
                }
                else
                {
                    pos = leftChild | 1;
                    left = (left + right) >> 1;
                }
            }

            int index = pos - size;
            return tree[pos] >= need ? index : -1;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len;
        private int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }

            return buffer[ptr++];
        }
    }
}
