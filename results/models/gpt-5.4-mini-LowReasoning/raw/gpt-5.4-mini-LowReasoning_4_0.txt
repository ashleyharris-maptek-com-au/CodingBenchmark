#include <bits/stdc++.h>
using namespace std;

struct Vec3 {
    long double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(long double x_, long double y_, long double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(long double s) const { return {x * s, y * s, z * s}; }
};

static inline long double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline long double norm2(const Vec3& a) { return dot(a, a); }

struct Plane {
    Vec3 n;
    long double d;
};

static long double eps = 1e-12L;

static bool same_plane(const Plane& a, const Plane& b) {
    long double dn = fabsl(a.n.x - b.n.x) + fabsl(a.n.y - b.n.y) + fabsl(a.n.z - b.n.z);
    return dn < 1e-10L && fabsl(a.d - b.d) < 1e-10L;
}

static vector<Plane> build_convex_hull_planes(const vector<Vec3>& p) {
    int n = (int)p.size();
    vector<Plane> planes;
    if (n < 4) return planes;

    for (int i = 0; i < n; ++i) for (int j = i + 1; j < n; ++j) for (int k = j + 1; k < n; ++k) {
        Vec3 a = p[j] - p[i], b = p[k] - p[i];
        Vec3 nrm = cross(a, b);
        long double m = sqrtl(norm2(nrm));
        if (m < 1e-18L) continue;
        nrm = nrm * (1.0L / m);
        long double d = -dot(nrm, p[i]);
        long double side = 0;
        bool valid = true;
        for (int t = 0; t < n; ++t) {
            long double s = dot(nrm, p[t]) + d;
            if (fabsl(s) <= 1e-10L) continue;
            if (side == 0) side = (s > 0 ? 1 : -1);
            else if (side * s < -1e-10L) { valid = false; break; }
        }
        if (!valid) continue;
        if (side > 0) {
            nrm = nrm * -1.0L;
            d = -d;
        }
        Plane pl{nrm, d};
        bool dup = false;
        for (const auto& q : planes) {
            if (same_plane(pl, q)) { dup = true; break; }
        }
        if (!dup) planes.push_back(pl);
    }
    return planes;
}

static bool inside_convex(const vector<Plane>& planes, const Vec3& q) {
    for (const auto& pl : planes) {
        if (dot(pl.n, q) + pl.d > 1e-9L) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    long double e;
    if (!(cin >> V >> e)) return 0;

    vector<Vec3> pts(V);
    for (int i = 0; i < V; ++i) cin >> pts[i].x >> pts[i].y >> pts[i].z;

    if (V < 4) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Plane> planes = build_convex_hull_planes(pts);
    if (planes.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    Vec3 mn = pts[0], mx = pts[0];
    for (auto& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }

    const long double sqrt8 = sqrtl(8.0L);
    const long double R = e * sqrtl(3.0L / 8.0L);
    const long double step = 2.0L * R;

    vector<Vec3> local = {
        Vec3( 1,  1,  1),
        Vec3( 1, -1, -1),
        Vec3(-1,  1, -1),
        Vec3(-1, -1,  1)
    };
    for (auto& v : local) v = v * (e / sqrt8);

    vector<array<long double, 6>> out;
    long double z0 = mn.z + step * 0.5L;
    for (long double z = z0; z <= mx.z - step * 0.5L + 1e-12L; z += step) {
        long double y0 = mn.y + step * 0.5L;
        for (long double y = y0; y <= mx.y - step * 0.5L + 1e-12L; y += step) {
            long double x0 = mn.x + step * 0.5L;
            for (long double x = x0; x <= mx.x - step * 0.5L + 1e-12L; x += step) {
                Vec3 c{x, y, z};
                bool ok = true;
                for (const auto& v : local) {
                    if (!inside_convex(planes, c + v)) { ok = false; break; }
                }
                if (!ok) continue;
                out.push_back({x, y, z, 0.0L, 0.0L, 0.0L});
            }
        }
    }

    cout << out.size() << '\n';
    cout.setf(std::ios::fixed);
    cout << setprecision(15);
    for (auto& t : out) {
        cout << t[0] << ' ' << t[1] << ' ' << t[2] << ' ' << t[3] << ' ' << t[4] << ' ' << t[5] << '\n';
    }
    return 0;
}
