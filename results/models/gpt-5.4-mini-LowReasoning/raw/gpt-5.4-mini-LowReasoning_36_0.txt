#include <bits/stdc++.h>
using namespace std;

struct Solver {
    int n;
    int W;
    vector<vector<uint64_t>> adj;
    vector<int> best;
    vector<int> cur;

    Solver(int n_, const vector<pair<int,int>>& edges) : n(n_) {
        W = (n + 63) >> 6;
        adj.assign(n, vector<uint64_t>(W, 0));
        for (auto [u, v] : edges) {
            if (u == v) continue;
            adj[u][v >> 6] |= 1ULL << (v & 63);
            adj[v][u >> 6] |= 1ULL << (u & 63);
        }
    }

    static inline int popcount64(uint64_t x) { return (int)__builtin_popcountll(x); }

    int countBits(const vector<uint64_t>& bs) const {
        int s = 0;
        for (uint64_t x : bs) s += popcount64(x);
        return s;
    }

    bool emptySet(const vector<uint64_t>& bs) const {
        for (uint64_t x : bs) if (x) return false;
        return true;
    }

    int firstVertex(const vector<uint64_t>& bs) const {
        for (int i = 0; i < W; ++i) {
            if (bs[i]) return (i << 6) + __builtin_ctzll(bs[i]);
        }
        return -1;
    }

    vector<int> extractVertices(const vector<uint64_t>& bs) const {
        vector<int> vs;
        for (int w = 0; w < W; ++w) {
            uint64_t x = bs[w];
            while (x) {
                int b = __builtin_ctzll(x);
                int v = (w << 6) + b;
                if (v < n) vs.push_back(v);
                x &= x - 1;
            }
        }
        return vs;
    }

    int greedyColorBound(const vector<uint64_t>& cand) const {
        vector<int> verts = extractVertices(cand);
        if (verts.empty()) return 0;
        sort(verts.begin(), verts.end(), [&](int a, int b) {
            int da = 0, db = 0;
            for (int i = 0; i < W; ++i) {
                da += popcount64(adj[a][i] & cand[i]);
                db += popcount64(adj[b][i] & cand[i]);
            }
            return da > db;
        });

        vector<vector<uint64_t>> colorMasks;
        for (int v : verts) {
            int c = 0;
            for (; c < (int)colorMasks.size(); ++c) {
                bool ok = true;
                for (int i = 0; i < W; ++i) {
                    if (adj[v][i] & colorMasks[c][i]) { ok = false; break; }
                }
                if (ok) break;
            }
            if (c == (int)colorMasks.size()) colorMasks.push_back(vector<uint64_t>(W, 0));
            colorMasks[c][v >> 6] |= 1ULL << (v & 63);
        }
        return (int)colorMasks.size();
    }

    vector<uint64_t> intersectWithNeighborhood(const vector<uint64_t>& cand, int v, bool includeV) const {
        vector<uint64_t> res(W, 0);
        for (int i = 0; i < W; ++i) {
            uint64_t x = includeV ? (cand[i] & ~adj[v][i]) : cand[i];
            res[i] = x;
        }
        if (includeV) res[v >> 6] &= ~(1ULL << (v & 63));
        else res[v >> 6] &= ~(1ULL << (v & 63));
        return res;
    }

    int chooseVertex(const vector<uint64_t>& cand) const {
        int bestV = -1, bestD = -1;
        for (int w = 0; w < W; ++w) {
            uint64_t x = cand[w];
            while (x) {
                int b = __builtin_ctzll(x);
                int v = (w << 6) + b;
                if (v >= n) break;
                int d = 0;
                for (int i = 0; i < W; ++i) d += popcount64(adj[v][i] & cand[i]);
                if (d > bestD) { bestD = d; bestV = v; }
                x &= x - 1;
            }
        }
        return bestV;
    }

    void dfs(vector<uint64_t>& cand) {
        if ((int)cur.size() + countBits(cand) <= (int)best.size()) return;
        if (emptySet(cand)) {
            if (cur.size() > best.size()) best = cur;
            return;
        }
        int ub = greedyColorBound(cand);
        if ((int)cur.size() + ub <= (int)best.size()) return;

        int v = chooseVertex(cand);
        if (v < 0) return;

        vector<uint64_t> withoutV = cand;
        withoutV[v >> 6] &= ~(1ULL << (v & 63));

        // Branch 1: include v
        vector<uint64_t> withV = cand;
        for (int i = 0; i < W; ++i) withV[i] &= ~adj[v][i];
        withV[v >> 6] &= ~(1ULL << (v & 63));
        cur.push_back(v);
        dfs(withV);
        cur.pop_back();

        // Branch 2: exclude v
        dfs(withoutV);
    }

    vector<int> solveComponent(const vector<int>& comp) {
        if (comp.empty()) return {};
        if ((int)comp.size() == 1) return {comp[0]};

        vector<int> idx = comp;
        sort(idx.begin(), idx.end());
        vector<int> mapOldToNew(n, -1);
        for (int i = 0; i < (int)idx.size(); ++i) mapOldToNew[idx[i]] = i;

        int k = (int)idx.size();
        int w = (k + 63) >> 6;
        vector<vector<uint64_t>> oldAdj = adj;
        vector<vector<uint64_t>> cadj(k, vector<uint64_t>(w, 0));
        for (int i = 0; i < k; ++i) {
            int u = idx[i];
            for (int j = 0; j < k; ++j) {
                int v = idx[j];
                if (u != v && ((oldAdj[u][v >> 6] >> (v & 63)) & 1ULL)) {
                    cadj[i][j >> 6] |= 1ULL << (j & 63);
                }
            }
        }

        Solver sub(k, {});
        sub.adj = move(cadj);
        sub.best.clear();
        sub.cur.clear();
        vector<uint64_t> cand(w, ~0ULL);
        if (k % 64) cand.back() &= ((1ULL << (k % 64)) - 1);
        sub.dfs(cand);

        vector<int> res;
        for (int x : sub.best) res.push_back(idx[x]);
        return res;
    }

    vector<int> solve() {
        vector<int> seen(n, 0);
        vector<int> answer;
        for (int s = 0; s < n; ++s) {
            if (seen[s]) continue;
            queue<int> q;
            q.push(s);
            seen[s] = 1;
            vector<int> comp;
            while (!q.empty()) {
                int u = q.front(); q.pop();
                comp.push_back(u);
                for (int v = 0; v < n; ++v) {
                    if (!seen[v] && ((adj[u][v >> 6] >> (v & 63)) & 1ULL)) {
                        seen[v] = 1;
                        q.push(v);
                    }
                }
            }
            vector<int> part = solveComponent(comp);
            answer.insert(answer.end(), part.begin(), part.end());
        }
        sort(answer.begin(), answer.end());
        return answer;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<pair<int,int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u > v) swap(u, v);
        edges.push_back({u, v});
    }

    Solver solver(n, edges);
    vector<int> ans = solver.solve();
    cout << ans.size() << '\n';
    for (size_t i = 0; i < ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
