declare const require: any;
const fs = require('fs');

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };
type Data = { grid: string[]; robots: Robot[]; jobs: Job[] };

const input: string = fs.readFileSync(0, 'utf8');
const data: Data = JSON.parse(input);

const grid = data.grid;
const h = grid.length;
const w = h === 0 ? 0 : grid[0].length;

const robots = data.robots.slice().sort((a, b) => a.id - b.id);
const jobs = data.jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const robotCount = robots.length;
const jobCount = jobs.length;

const cellCount = w * h;
const passable = new Uint8Array(cellCount);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    passable[y * w + x] = row.charCodeAt(x) === 35 ? 0 : 1;
  }
}

function idx(x: number, y: number): number {
  return y * w + x;
}

function bfsFrom(sx: number, sy: number): Int32Array {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);
  if (sx < 0 || sy < 0 || sx >= w || sy >= h) return dist;
  const start = idx(sx, sy);
  if (!passable[start]) return dist;

  const q = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;
  q[tail++] = start;
  dist[start] = 0;

  while (head < tail) {
    const v = q[head++];
    const d = dist[v] + 1;
    const x = v % w;
    const y = (v / w) | 0;

    if (x > 0) {
      const u = v - 1;
      if (passable[u] && dist[u] === -1) {
        dist[u] = d;
        q[tail++] = u;
      }
    }
    if (x + 1 < w) {
      const u = v + 1;
      if (passable[u] && dist[u] === -1) {
        dist[u] = d;
        q[tail++] = u;
      }
    }
    if (y > 0) {
      const u = v - w;
      if (passable[u] && dist[u] === -1) {
        dist[u] = d;
        q[tail++] = u;
      }
    }
    if (y + 1 < h) {
      const u = v + w;
      if (passable[u] && dist[u] === -1) {
        dist[u] = d;
        q[tail++] = u;
      }
    }
  }

  return dist;
}

const robotDistances: Int32Array[] = [];
for (const r of robots) {
  robotDistances.push(bfsFrom(r.x, r.y));
}

const assignments: number[][] = Array.from({ length: robotCount }, () => []);
const load = new Float64Array(robotCount);
const robotIndexById = new Map<number, number>();
for (let i = 0; i < robots.length; i++) robotIndexById.set(robots[i].id, i);

for (const job of jobs) {
  let bestRobot = 0;
  let bestScore = Number.POSITIVE_INFINITY;
  for (let i = 0; i < robotCount; i++) {
    const d = robotDistances[i][idx(job.x, job.y)];
    if (d < 0) continue;
    const score = d + load[i] * 0.35 - job.priority * 0.1;
    if (score < bestScore || (score === bestScore && robots[i].id < robots[bestRobot].id)) {
      bestScore = score;
      bestRobot = i;
    }
  }
  assignments[bestRobot].push(job.id);
  load[bestRobot] += 1 + Math.max(0, 6 - job.priority) * 0.15;
}

for (let i = 0; i < assignments.length; i++) {
  assignments[i].sort((a, b) => a - b);
}

process.stdout.write(JSON.stringify({ assignments }));
