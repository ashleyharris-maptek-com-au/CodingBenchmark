use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn quat_from_z_to_dir(dx: f64, dy: f64, dz: f64) -> (f64, f64, f64, f64) {
    let n = (dx * dx + dy * dy + dz * dz).sqrt().max(1e-12);
    let ux = dx / n;
    let uy = dy / n;
    let uz = dz / n;
    let dot = clamp(uz, -1.0, 1.0);
    if dot > 0.999_999 {
        return (1.0, 0.0, 0.0, 0.0);
    }
    if dot < -0.999_999 {
        return (0.0, 1.0, 0.0, 0.0);
    }
    let ax = -uy;
    let ay = ux;
    let az = 0.0;
    let s = ((1.0 + dot) * 2.0).sqrt();
    let invs = 1.0 / s;
    let qw = 0.5 * s;
    let qx = ax * invs;
    let qy = ay * invs;
    let qz = az * invs;
    (qw, qx, qy, qz)
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut a = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        a += p.x * q.y - q.x * p.y;
    }
    a * 0.5
}

fn centroid(poly: &[Pt]) -> Pt {
    let mut sx = 0.0;
    let mut sy = 0.0;
    for p in poly {
        sx += p.x;
        sy += p.y;
    }
    let n = poly.len().max(1) as f64;
    Pt { x: sx / n, y: sy / n }
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = it.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Pt { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    let (minx, miny, maxx, maxy) = bbox(&poly);
    let w = (maxx - minx).max(1e-9);
    let h = (maxy - miny).max(1e-9);
    let area = polygon_area(&poly).abs();
    let c = centroid(&poly);

    let q = quat_from_z_to_dir(sx, sy, sz);

    let mut instances: Vec<(f64, f64, f64, (f64, f64, f64, f64), f64)> = Vec::new();

    // Adaptive strategy:
    // Small or simple shapes: use one larger tetrahedron near the centroid.
    // Larger shapes: cover the bounding box with a grid of moderately sized tetrahedrons.
    let use_single = nv <= 8 || area < 0.02 * w * h;

    if use_single {
        let scale = (w.max(h) * 0.75).min(10.0).max(0.05);
        instances.push((c.x, c.y, 5.0 * scale + 1.0, q, scale));
    } else {
        let cell = (w.max(h) / ((nv as f64).sqrt().max(2.0) * 0.85)).clamp(0.5, 10.0);
        let nx = ((w / cell).ceil() as usize).max(1);
        let ny = ((h / cell).ceil() as usize).max(1);
        let scale = (cell * 0.42).min(10.0).max(0.05);
        let mut idx = 0usize;
        for iy in 0..ny {
            for ix in 0..nx {
                let px = minx + (ix as f64 + 0.5) * w / nx as f64;
                let py = miny + (iy as f64 + 0.5) * h / ny as f64;
                let pz = 2.0 + idx as f64 * 0.25 + scale * 2.0;
                instances.push((px, py, pz, q, scale));
                idx += 1;
            }
        }
    }

    println!("{}", instances.len());
    for (px, py, pz, (qw, qx, qy, qz), scale) in instances {
        println!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            px, py, pz, qw, qx, qy, qz, scale
        );
    }
}
