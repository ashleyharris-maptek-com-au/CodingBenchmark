OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0

%void = OpTypeVoid
%funct = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_half = OpConstant %float 0.5
%float_nhalf = OpConstant %float -0.5
%float_sqrt3_2 = OpConstant %float 0.8660254
%float_nsqrt3_2 = OpConstant %float -0.8660254
%float_n1 = OpConstant %float -1.0
%float_r = OpConstant %float 0.04
%float_dark = OpConstant %float 0.7

%dir0 = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_0
%dir1 = OpConstantComposite %v4float %float_half %float_sqrt3_2 %float_0 %float_0
%dir2 = OpConstantComposite %v4float %float_nhalf %float_sqrt3_2 %float_0 %float_0
%dir3 = OpConstantComposite %v4float %float_n1 %float_0 %float_0 %float_0
%dir4 = OpConstantComposite %v4float %float_nhalf %float_nsqrt3_2 %float_0 %float_0
%dir5 = OpConstantComposite %v4float %float_half %float_nsqrt3_2 %float_0 %float_0

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %funct
%entry = OpLabel

%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%base_color = OpLoad %v3float %color_ptr
%dark_color = OpVectorTimesScalar %v3float %base_color %float_dark

%off0 = OpVectorTimesScalar %v4float %dir0 %float_r
%off1 = OpVectorTimesScalar %v4float %dir1 %float_r
%off2 = OpVectorTimesScalar %v4float %dir2 %float_r
%off3 = OpVectorTimesScalar %v4float %dir3 %float_r
%off4 = OpVectorTimesScalar %v4float %dir4 %float_r
%off5 = OpVectorTimesScalar %v4float %dir5 %float_r

%edge0 = OpFAdd %v4float %center %off0
%edge1 = OpFAdd %v4float %center %off1
%edge2 = OpFAdd %v4float %center %off2
%edge3 = OpFAdd %v4float %center %off3
%edge4 = OpFAdd %v4float %center %off4
%edge5 = OpFAdd %v4float %center %off5

%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos %center
OpStore %outColor %base_color
OpEmitVertex

OpStore %out_pos %edge0
OpStore %outColor %dark_color
OpEmitVertex

OpStore %out_pos %edge1
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %out_pos %center
OpStore %outColor %base_color
OpEmitVertex

OpStore %out_pos %edge1
OpStore %outColor %dark_color
OpEmitVertex

OpStore %out_pos %edge2
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %out_pos %center
OpStore %outColor %base_color
OpEmitVertex

OpStore %out_pos %edge2
OpStore %outColor %dark_color
OpEmitVertex

OpStore %out_pos %edge3
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %out_pos %center
OpStore %outColor %base_color
OpEmitVertex

OpStore %out_pos %edge3
OpStore %outColor %dark_color
OpEmitVertex

OpStore %out_pos %edge4
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %out_pos %center
OpStore %outColor %base_color
OpEmitVertex

OpStore %out_pos %edge4
OpStore %outColor %dark_color
OpEmitVertex

OpStore %out_pos %edge5
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %out_pos %center
OpStore %outColor %base_color
OpEmitVertex

OpStore %out_pos %edge5
OpStore %outColor %dark_color
OpEmitVertex

OpStore %out_pos %edge0
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
