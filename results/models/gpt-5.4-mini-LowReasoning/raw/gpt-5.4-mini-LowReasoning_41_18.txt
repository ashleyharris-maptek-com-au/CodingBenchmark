%label = OpLabel

%b = OpFMul %float %t %float_2
%color_high = OpCompositeConstruct %v4float %rg %rg %b %float_1
OpBranch %merge

%merge = OpLabel
%result = OpPhi %v4float %color_low %low %color_high %high
