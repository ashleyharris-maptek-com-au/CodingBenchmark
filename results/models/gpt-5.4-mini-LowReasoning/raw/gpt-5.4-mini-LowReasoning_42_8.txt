OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor %gl_out
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_0333333 = OpConstant %float 0.33333334
%float_3 = OpConstant %float 3
%float_7 = OpConstant %float 0.7

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_3
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_input_gl_in = OpTypePointer Input %arr_gl_in
%ptr_input_v3arr = OpTypePointer Input %arr_v3
%ptr_output_v3 = OpTypePointer Output %v3float
%ptr_output_gl = OpTypePointer Output %gl_PerVertex_out
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_input_gl_in Input
%worldPos = OpVariable %ptr_input_v3arr Input
%normal = OpVariable %ptr_input_v3arr Input
%color = OpVariable %ptr_input_v3arr Input
%outColor = OpVariable %ptr_output_v3 Output
%gl_out = OpVariable %ptr_output_gl Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum012 %float_0333333

%c0_ptr = OpAccessChain %ptr_input_v3arr %color %int_0
%c1_ptr = OpAccessChain %ptr_input_v3arr %color %int_1
%c2_ptr = OpAccessChain %ptr_input_v3arr %color %int_2
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%d0 = OpFSub %v4float %p0 %centroid
%s0 = OpVectorTimesScalar %v4float %d0 %float_7
%n0 = OpFAdd %v4float %centroid %s0
%p0_out_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %p0_out_ptr %n0
OpStore %outColor %c0
OpEmitVertex

%d1 = OpFSub %v4float %p1 %centroid
%s1 = OpVectorTimesScalar %v4float %d1 %float_7
%n1 = OpFAdd %v4float %centroid %s1
%p1_out_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %p1_out_ptr %n1
OpStore %outColor %c1
OpEmitVertex

%d2 = OpFSub %v4float %p2 %centroid
%s2 = OpVectorTimesScalar %v4float %d2 %float_7
%n2 = OpFAdd %v4float %centroid %s2
%p2_out_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %p2_out_ptr %n2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
