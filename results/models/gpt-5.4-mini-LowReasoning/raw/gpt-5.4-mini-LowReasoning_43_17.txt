float4 main(PSInput input) : SV_Target0
{
    float2 scaledUV = input.uv * float2(8.0, 16.0);
    float row = floor(scaledUV.y);
    float2 cellUV = frac(scaledUV);

    // Half-brick stagger on alternating rows.
    if (fmod(row, 2.0) >= 1.0)
    {
        cellUV.x = frac(cellUV.x + 0.5);
    }

    float mortarX = 0.05;
    float mortarY = 0.10;
    float brickMask = 1.0;

    if (cellUV.x < mortarX || cellUV.x > 1.0 - mortarX || cellUV.y < mortarY || cellUV.y > 1.0 - mortarY)
    {
        brickMask = 0.0;
    }

    float3 baseColor = lerp(float3(0.7, 0.7, 0.7), float3(0.7, 0.2, 0.1), brickMask);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    return float4(baseColor * shade, 1.0);
}
