using namespace std;

struct Point {
    double x = 0.0;
    double y = 0.0;
};

static inline uint32_t clamp_u32(double v) {
    if (v < 0.0) return 0u;
    if (v > 4294967295.0) return 4294967295u;
    return static_cast<uint32_t>(v);
}

static inline uint64_t split_by_1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFull;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFull;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0Full;
    v = (v | (v << 2))  & 0x3333333333333333ull;
    v = (v | (v << 1))  & 0x5555555555555555ull;
    return v;
}

static inline uint64_t morton_code(double x, double y, double minx, double miny, double invx, double invy) {
    const double sx = (x - minx) * invx;
    const double sy = (y - miny) * invy;
    const uint32_t ix = clamp_u32(sx * 4294967295.0);
    const uint32_t iy = clamp_u32(sy * 4294967295.0);
    return (split_by_1(ix) << 1) | split_by_1(iy);
}

static inline double dist2(const Point& a, const Point& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static vector<int> exact_bruteforce(const vector<Point>& p) {
    const int n = static_cast<int>(p.size());
    vector<int> perm;
    perm.reserve(n);
    perm.push_back(0);
    for (int i = 1; i < n; ++i) perm.push_back(i);

    vector<int> best = perm;
    double bestLen = numeric_limits<double>::infinity();
    do {
        double len = 0.0;
        for (int i = 0; i + 1 < n; ++i) len += sqrt(dist2(p[perm[i]], p[perm[i + 1]]));
        if (n > 1) len += sqrt(dist2(p[perm.back()], p[perm[0]]));
        if (len < bestLen) {
            bestLen = len;
            best = perm;
        }
    } while (next_permutation(perm.begin() + 1, perm.end()));
    return best;
}

static vector<int> nearest_neighbor_tour(const vector<Point>& p) {
    const int n = static_cast<int>(p.size());
    vector<int> order;
    order.reserve(n);
    order.push_back(0);

    vector<char> used(n, 0);
    used[0] = 1;

    vector<int> byX(n);
    iota(byX.begin(), byX.end(), 0);
    sort(byX.begin(), byX.end(), [&](int a, int b) {
        if (p[a].x != p[b].x) return p[a].x < p[b].x;
        return p[a].y < p[b].y;
    });

    vector<int> rank(n);
    for (int i = 0; i < n; ++i) rank[byX[i]] = i;

    int cur = 0;
    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestD = numeric_limits<double>::infinity();

        const int pos = rank[cur];
        const int left = max(0, pos - 64);
        const int right = min(n - 1, pos + 64);
        for (int k = left; k <= right; ++k) {
            int cand = byX[k];
            if (used[cand]) continue;
            double d = dist2(p[cur], p[cand]);
            if (d < bestD) {
                bestD = d;
                best = cand;
            }
        }

        if (best == -1) {
            for (int i = 0; i < n; ++i) {
                if (!used[i]) {
                    best = i;
                    break;
                }
            }
        }

        used[best] = 1;
        order.push_back(best);
        cur = best;
    }
    return order;
}

static vector<int> morton_tour(const vector<Point>& p) {
    const int n = static_cast<int>(p.size());
    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (const auto& pt : p) {
        minx = min(minx, pt.x);
        maxx = max(maxx, pt.x);
        miny = min(miny, pt.y);
        maxy = max(maxy, pt.y);
    }
    const double dx = max(maxx - minx, 1e-12);
    const double dy = max(maxy - miny, 1e-12);
    const double invx = 1.0 / dx;
    const double invy = 1.0 / dy;

    vector<pair<uint64_t, int>> items;
    items.reserve(n);
    for (int i = 0; i < n; ++i) {
        items.emplace_back(morton_code(p[i].x, p[i].y, minx, miny, invx, invy), i);
    }
    sort(items.begin(), items.end(), [](const auto& a, const auto& b) {
        if (a.first != b.first) return a.first < b.first;
        return a.second < b.second;
    });

    vector<int> order;
    order.reserve(n);
    int pos0 = 0;
    while (pos0 < n && items[pos0].second != 0) ++pos0;
    for (int i = 0; i < n; ++i) order.push_back(items[(pos0 + i) % n].second);
    return order;
}

static void two_opt_local(vector<int>& tour, const vector<Point>& p, int passes) {
    const int n = static_cast<int>(tour.size());
    if (n < 4) return;
    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 3 < n; ++i) {
            for (int j = i + 2; j + 1 < n; ++j) {
                if (i == 0 && j + 1 == n) continue;
                int a = tour[i], b = tour[i + 1], c = tour[j], d = tour[j + 1];
                double before = sqrt(dist2(p[a], p[b])) + sqrt(dist2(p[c], p[d]));
                double after  = sqrt(dist2(p[a], p[c])) + sqrt(dist2(p[b], p[d]));
                if (after + 1e-12 < before) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    vector<Point> p(N);
    for (int i = 0; i < N; ++i) cin >> p[i].x >> p[i].y;

    vector<int> ans;
    if (N <= 10) {
        ans = exact_bruteforce(p);
    } else if (N <= 2500) {
        ans = nearest_neighbor_tour(p);
        two_opt_local(ans, p, 2);
    } else if (N <= 200000) {
        ans = morton_tour(p);
        if (N <= 20000) two_opt_local(ans, p, 1);
    } else {
        ans = morton_tour(p);
    }

    if (!ans.empty() && ans[0] != 0) {
        auto it = find(ans.begin(), ans.end(), 0);
        if (it != ans.end()) {
            rotate(ans.begin(), it, ans.end());
        }
    }

    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
