use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push(Edge { u, v });
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let coords = if n <= 1200 {
        force_layout(n, &adj, &edges)
    } else {
        fast_layout(n, &adj)
    };

    let mut out = String::with_capacity(n * 24);
    for (x, y) in coords {
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{}", out);
}

fn fast_layout(n: usize, adj: &Vec<Vec<usize>>) -> Vec<(f64, f64)> {
    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut q = VecDeque::new();

    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let cid = comps.len();
        comp_id[s] = cid;
        comps.push(Vec::new());
        q.push_back(s);
        while let Some(u) = q.pop_front() {
            comps[cid].push(u);
            for &v in &adj[u] {
                if comp_id[v] == usize::MAX {
                    comp_id[v] = cid;
                    q.push_back(v);
                }
            }
        }
    }

    let mut coords = vec![(0.0, 0.0); n];
    let mut offset_x = 0.0;
    let mut comp_sizes: Vec<usize> = comps.iter().map(|c| c.len()).collect();
    let total: usize = comp_sizes.iter().sum();
    if total == 0 {
        return coords;
    }

    for (cid, nodes) in comps.iter().enumerate() {
        let sz = nodes.len();
        let width = (sz as f64).sqrt().max(1.0) * 4.0;
        let root = choose_seed(nodes, adj);
        let level = bfs_levels(root, adj, nodes, n);
        let max_level = level.iter().copied().max().unwrap_or(0);

        let mut by_level: Vec<Vec<usize>> = vec![Vec::new(); max_level + 1];
        for &u in nodes {
            let l = level[u];
            by_level[l].push(u);
        }

        for l in 0..=max_level {
            by_level[l].sort_by(|&a, &b| {
                let da = adj[a].len();
                let db = adj[b].len();
                db.cmp(&da).then_with(|| a.cmp(&b))
            });
        }

        let comp_radius = (sz as f64).sqrt().max(2.0) * 1.8;
        let cx = offset_x + comp_radius;
        let cy = 0.0;
        offset_x += width + 8.0;

        for (l, layer) in by_level.iter().enumerate() {
            if layer.is_empty() {
                continue;
            }
            let r = comp_radius * (0.25 + 0.75 * (l as f64 + 1.0) / (max_level as f64 + 1.5));
            let step = 2.0 * PI / (layer.len() as f64);
            for (i, &u) in layer.iter().enumerate() {
                let ang = i as f64 * step;
                coords[u] = (cx + r * ang.cos(), cy + r * ang.sin());
            }
        }

        if max_level == 0 {
            let layer = &by_level[0];
            let step = 2.0 * PI / (layer.len() as f64);
            for (i, &u) in layer.iter().enumerate() {
                let ang = i as f64 * step;
                coords[u] = (cx + comp_radius * ang.cos(), cy + comp_radius * ang.sin());
            }
        }
    }

    coords
}

fn force_layout(n: usize, adj: &Vec<Vec<usize>>, edges: &Vec<Edge>) -> Vec<(f64, f64)> {
    let mut pos = Vec::with_capacity(n);
    let side = (n as f64).sqrt().ceil() as usize;
    for i in 0..n {
        let x = (i % side) as f64;
        let y = (i / side) as f64;
        pos.push((x, y));
    }

    let mut degree = vec![0usize; n];
    for u in 0..n {
        degree[u] = adj[u].len().max(1);
    }

    let iterations = if n <= 100 { 80 } else { 25 };
    let base_k = ((n as f64).max(1.0)).sqrt() * 0.8 + 1.0;

    for itn in 0..iterations {
        let temp = 0.15 * (1.0 - itn as f64 / iterations as f64) + 0.02;
        let mut disp = vec![(0.0f64, 0.0f64); n];

        let mut cell_size = base_k * 2.5;
        if cell_size < 1.0 {
            cell_size = 1.0;
        }
        let mut grid = Grid::new(cell_size);
        for i in 0..n {
            grid.insert(i, pos[i]);
        }

        for i in 0..n {
            let (xi, yi) = pos[i];
            let neigh = grid.neighbors((xi, yi));
            for j in neigh {
                if j <= i {
                    continue;
                }
                let (xj, yj) = pos[j];
                let dx = xi - xj;
                let dy = yi - yj;
                let mut dist2 = dx * dx + dy * dy + 1e-9;
                let dist = dist2.sqrt();
                let force = base_k * base_k / dist2;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        for e in edges {
            if e.u == e.v || e.u >= n || e.v >= n {
                continue;
            }
            let (xu, yu) = pos[e.u];
            let (xv, yv) = pos[e.v];
            let dx = xu - xv;
            let dy = yu - yv;
            let dist = (dx * dx + dy * dy).sqrt().max(1e-6);
            let force = (dist * dist) / base_k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[e.u].0 -= fx;
            disp[e.u].1 -= fy;
            disp[e.v].0 += fx;
            disp[e.v].1 += fy;
        }

        for i in 0..n {
            let (dx, dy) = disp[i];
            let len = (dx * dx + dy * dy).sqrt();
            if len > 0.0 {
                let step = temp / (1.0 + degree[i] as f64 * 0.05);
                pos[i].0 += dx / len * step;
                pos[i].1 += dy / len * step;
            }
        }

        let mut cx = 0.0;
        let mut cy = 0.0;
        for &(x, y) in &pos {
            cx += x;
            cy += y;
        }
        cx /= n as f64;
        cy /= n as f64;
        for p in &mut pos {
            p.0 -= cx;
            p.1 -= cy;
        }
    }

    pos
}

fn choose_seed(nodes: &Vec<usize>, adj: &Vec<Vec<usize>>) -> usize {
    let mut best = nodes[0];
    let mut best_deg = adj[best].len();
    for &u in nodes.iter().skip(1) {
        let d = adj[u].len();
        if d > best_deg {
            best_deg = d;
            best = u;
        }
    }
    best
}

fn bfs_levels(root: usize, adj: &Vec<Vec<usize>>, nodes: &Vec<usize>, n: usize) -> Vec<usize> {
    let mut in_comp = vec![false; n];
    for &u in nodes {
        in_comp[u] = true;
    }
    let mut level = vec![usize::MAX; n];
    let mut q = VecDeque::new();
    level[root] = 0;
    q.push_back(root);
    while let Some(u) = q.pop_front() {
        let lu = level[u];
        for &v in &adj[u] {
            if in_comp[v] && level[v] == usize::MAX {
                level[v] = lu + 1;
                q.push_back(v);
            }
        }
    }
    for &u in nodes {
        if level[u] == usize::MAX {
            level[u] = 0;
        }
    }
    level
}

struct Grid {
    cell: f64,
    buckets: std::collections::HashMap<(i64, i64), Vec<usize>>,
    points: Vec<(f64, f64)>,
}

impl Grid {
    fn new(cell: f64) -> Self {
        Self {
            cell,
            buckets: std::collections::HashMap::new(),
            points: Vec::new(),
        }
    }

    fn key(&self, p: (f64, f64)) -> (i64, i64) {
        let gx = (p.0 / self.cell).floor() as i64;
        let gy = (p.1 / self.cell).floor() as i64;
        (gx, gy)
    }

    fn insert(&mut self, idx: usize, p: (f64, f64)) {
        if idx == self.points.len() {
            self.points.push(p);
        } else {
            self.points[idx] = p;
        }
        let k = self.key(p);
        self.buckets.entry(k).or_default().push(idx);
    }

    fn neighbors(&self, p: (f64, f64)) -> Vec<usize> {
        let (cx, cy) = self.key(p);
        let mut out = Vec::new();
        for dx in -1..=1 {
            for dy in -1..=1 {
                if let Some(v) = self.buckets.get(&(cx + dx, cy + dy)) {
                    out.extend_from_slice(v);
                }
            }
        }
        out
    }
}
