static inline int sq(int f, int r) { return r * 8 + f; }
static inline int file_of(int s) { return s & 7; }
static inline int rank_of(int s) { return s >> 3; }
static inline string square_str(int s) {
    string t;
    t += char('a' + file_of(s));
    t += char('1' + rank_of(s));
    return t;
}

enum Piece { EMPTY=0, WP=1, WN, WB, WR, WQ, WK, BP, BN, BB, BR, BQ, BK };

struct Move {
    int from=-1, to=-1, promo=0;
    bool cap=false, ep=false, castle=false;
};

struct Undo {
    int captured=0;
    int ep=-1;
    int castling=0;
    int halfmove=0;
    int kingSqW=0, kingSqB=0;
};

struct Position {
    array<int, 64> b{};
    bool white = true;
    int castling = 0; // 1 K, 2 Q, 4 k, 8 q
    int ep = -1;
    int halfmove = 0;
    int fullmove = 1;
    int kingSqW = 60, kingSqB = 4;
};

static const int knightD[8] = {17,15,10,6,-17,-15,-10,-6};
static const int bishopD[4] = {9,7,-9,-7};
static const int rookD[4] = {8,-8,1,-1};
static const int kingD[8] = {8,-8,1,-1,9,7,-9,-7};

static inline bool on_board(int s) { return s >= 0 && s < 64; }
static inline bool same_line(int a, int b, int d) {
    if (d == 1 || d == -1) return rank_of(a) == rank_of(b);
    if (d == 9 || d == -9 || d == 7 || d == -7) return abs(file_of(a) - file_of(b)) == abs(rank_of(a) - rank_of(b));
    return true;
}

static bool is_attacked(const Position& p, int sqr, bool byWhite) {
    int f = file_of(sqr), r = rank_of(sqr);
    if (byWhite) {
        if (r > 0) {
            if (f > 0 && p.b[sqr - 9] == WP) return true;
            if (f < 7 && p.b[sqr - 7] == WP) return true;
        }
    } else {
        if (r < 7) {
            if (f > 0 && p.b[sqr + 7] == BP) return true;
            if (f < 7 && p.b[sqr + 9] == BP) return true;
        }
    }
    for (int d : knightD) {
        int t = sqr + d;
        if (!on_board(t)) continue;
        if (abs(file_of(t) - f) > 2 || abs(rank_of(t) - r) > 2) continue;
        int pc = p.b[t];
        if (byWhite && pc == WN) return true;
        if (!byWhite && pc == BN) return true;
    }
    for (int d : bishopD) {
        int t = sqr;
        while (true) {
            int nt = t + d;
            if (!on_board(nt) || !same_line(t, nt, d)) break;
            t = nt;
            int pc = p.b[t];
            if (pc) {
                if (byWhite && (pc == WB || pc == WQ)) return true;
                if (!byWhite && (pc == BB || pc == BQ)) return true;
                break;
            }
        }
    }
    for (int d : rookD) {
        int t = sqr;
        while (true) {
            int nt = t + d;
            if (!on_board(nt) || !same_line(t, nt, d)) break;
            t = nt;
            int pc = p.b[t];
            if (pc) {
                if (byWhite && (pc == WR || pc == WQ)) return true;
                if (!byWhite && (pc == BR || pc == BQ)) return true;
                break;
            }
        }
    }
    for (int d : kingD) {
        int t = sqr + d;
        if (!on_board(t) || abs(file_of(t)-f) > 1 || abs(rank_of(t)-r) > 1) continue;
        int pc = p.b[t];
        if (byWhite && pc == WK) return true;
        if (!byWhite && pc == BK) return true;
    }
    return false;
}

static int evaluate(const Position& p) {
    int score = 0;
    for (int s = 0; s < 64; ++s) {
        int pc = p.b[s];
        if (!pc) continue;
        int val = pieceVal[pc];
        switch (pc) {
            case WP: val += pstP[s]; break;
            case BP: val += pstP[63 - s]; break;
            case WN: val += pstN[s]; break;
            case BN: val += pstN[63 - s]; break;
            case WB: val += pstB[s]; break;
            case BB: val += pstB[63 - s]; break;
            case WR: val += pstR[s]; break;
            case BR: val += pstR[63 - s]; break;
            case WQ: val += pstQ[s]; break;
            case BQ: val += pstQ[63 - s]; break;
            case WK: val += pstK[s]; break;
            case BK: val += pstK[63 - s]; break;
        }
        if (pc <= WK) score += val; else score -= val;
    }
    score += p.white ? 10 : -10;
    return p.white ? score : -score;
}

static void add_move(vector<Move>& mv, int from, int to, bool cap=false, int promo=0, bool ep=false, bool castle=false) {
    Move m; m.from=from; m.to=to; m.cap=cap; m.promo=promo; m.ep=ep; m.castle=castle; mv.push_back(m);
}

static vector<Move> gen_moves(const Position& p) {
    vector<Move> mv;
    bool w = p.white;
    int usP = w ? WP : BP, usN = w ? WN : BN, usB = w ? WB : BB, usR = w ? WR : BR, usQ = w ? WQ : BQ, usK = w ? WK : BK;
    int themP = w ? BP : WP;
    int dir = w ? 8 : -8;
    int startRank = w ? 1 : 6;
    int promoRank = w ? 6 : 1;
    int kingSq = w ? p.kingSqW : p.kingSqB;

    for (int s = 0; s < 64; ++s) {
        int pc = p.b[s];
        if (pc != usP) continue;
        int r = rank_of(s), f = file_of(s);
        int one = s + dir;
        if (on_board(one) && !p.b[one]) {
            if (r == promoRank) {
                add_move(mv, s, one, false, w ? WQ : BQ);
                add_move(mv, s, one, false, w ? WR : BR);
                add_move(mv, s, one, false, w ? WB : BB);
                add_move(mv, s, one, false, w ? WN : BN);
            } else {
                add_move(mv, s, one);
                int two = s + 2 * dir;
                if (r == startRank && !p.b[two]) add_move(mv, s, two);
            }
        }
        for (int df : {-1, 1}) {
            int tf = f + df;
            if (tf < 0 || tf > 7) continue;
            int to = s + dir + df;
            if (!on_board(to)) continue;
            if (to == p.ep) {
                add_move(mv, s, to, true, 0, true);
            } else if (p.b[to] && ((w && p.b[to] >= BP) || (!w && p.b[to] <= WK && p.b[to] != 0))) {
                if (r == promoRank) {
                    add_move(mv, s, to, true, w ? WQ : BQ);
                    add_move(mv, s, to, true, w ? WR : BR);
                    add_move(mv, s, to, true, w ? WB : BB);
                    add_move(mv, s, to, true, w ? WN : BN);
                } else add_move(mv, s, to, true);
            }
        }
    }
    for (int s = 0; s < 64; ++s) {
        int pc = p.b[s];
        if (pc != usN) continue;
        int f = file_of(s), r = rank_of(s);
        for (int d : knightD) {
            int to = s + d;
            if (!on_board(to) || abs(file_of(to)-f) > 2 || abs(rank_of(to)-r) > 2) continue;
            int t = p.b[to];
            if (!t || (w ? t >= BP : t <= WK)) add_move(mv, s, to, t != 0);
        }
    }
    auto slide = [&](int s, const int* dirs, int cnt, int piece) {
        for (int i = 0; i < cnt; ++i) {
            int d = dirs[i], t = s;
            while (true) {
                int nt = t + d;
                if (!on_board(nt) || !same_line(t, nt, d)) break;
                t = nt;
                int q = p.b[t];
                if (!q) add_move(mv, s, t);
                else {
                    if (w ? q >= BP : q <= WK) add_move(mv, s, t, true);
                    break;
                }
            }
        }
    };
    for (int s = 0; s < 64; ++s) if (p.b[s] == usB) slide(s, bishopD, 4, usB);
    for (int s = 0; s < 64; ++s) if (p.b[s] == usR) slide(s, rookD, 4, usR);
    for (int s = 0; s < 64; ++s) if (p.b[s] == usQ) { slide(s, bishopD, 4, usQ); slide(s, rookD, 4, usQ); }
    for (int s = 0; s < 64; ++s) {
        if (p.b[s] != usK) continue;
        int f = file_of(s), r = rank_of(s);
        for (int d : kingD) {
            int to = s + d;
            if (!on_board(to) || abs(file_of(to)-f) > 1 || abs(rank_of(to)-r) > 1) continue;
            int t = p.b[to];
            if (!t || (w ? t >= BP : t <= WK)) add_move(mv, s, to, t != 0);
        }
        if (w && s == 60) {
            if ((p.castling & 1) && !p.b[61] && !p.b[62] && !is_attacked(p, 60, false) && !is_attacked(p,61,false) && !is_attacked(p,62,false))
                add_move(mv, 60, 62, false, 0, false, true);
            if ((p.castling & 2) && !p.b[59] && !p.b[58] && !p.b[57] && !is_attacked(p,60,false) && !is_attacked(p,59,false) && !is_attacked(p,58,false))
                add_move(mv, 60, 58, false, 0, false, true);
        }
        if (!w && s == 4) {
            if ((p.castling & 4) && !p.b[5] && !p.b[6] && !is_attacked(p,4,true) && !is_attacked(p,5,true) && !is_attacked(p,6,true))
                add_move(mv, 4, 6, false, 0, false, true);
            if ((p.castling & 8) && !p.b[3] && !p.b[2] && !p.b[1] && !is_attacked(p,4,true) && !is_attacked(p,3,true) && !is_attacked(p,2,true))
                add_move(mv, 4, 2, false, 0, false, true);
        }
    }
    vector<Move> legal;
    legal.reserve(mv.size());
    for (auto &m : mv) {
        Position c = p;
        Undo u;
        int pc = c.b[m.from];
        int cap = 0;
        c.b[m.from] = 0;
        if (m.ep) {
            cap = w ? BP : WP;
            c.b[w ? m.to - 8 : m.to + 8] = 0;
        } else if (m.cap) {
            cap = c.b[m.to];
            c.b[m.to] = 0;
        }
        if (m.castle) {
            if (m.to == 62) { c.b[63]=0; c.b[61]=WR; }
            else if (m.to == 58) { c.b[56]=0; c.b[59]=WR; }
            else if (m.to == 6) { c.b[7]=0; c.b[5]=BR; }
            else if (m.to == 2) { c.b[0]=0; c.b[3]=BR; }
        }
        int placed = m.promo ? m.promo : pc;
        c.b[m.to] = placed;
        if (pc == WK) c.kingSqW = m.to;
        if (pc == BK) c.kingSqB = m.to;
        int ksq = w ? c.kingSqW : c.kingSqB;
        if (!is_attacked(c, ksq, !w)) legal.push_back(m);
    }
    return legal;
}

static void make_move(Position& p, const Move& m, Undo& u) {
    u.captured = m.ep ? (p.white ? BP : WP) : p.b[m.to];
    u.ep = p.ep; u.castling = p.castling; u.halfmove = p.halfmove; u.kingSqW = p.kingSqW; u.kingSqB = p.kingSqB;
    int pc = p.b[m.from];
    p.ep = -1;
    if (pc == WP || pc == BP || u.captured) p.halfmove = 0; else ++p.halfmove;
    p.b[m.from] = 0;
    if (m.ep) p.b[p.white ? m.to - 8 : m.to + 8] = 0;
    else if (m.cap) p.b[m.to] = 0;
    if (m.castle) {
        if (m.to == 62) { p.b[63]=0; p.b[61]=WR; }
        else if (m.to == 58) { p.b[56]=0; p.b[59]=WR; }
        else if (m.to == 6) { p.b[7]=0; p.b[5]=BR; }
        else if (m.to == 2) { p.b[0]=0; p.b[3]=BR; }
    }
    int placed = m.promo ? m.promo : pc;
    p.b[m.to] = placed;
    if (pc == WK) p.kingSqW = m.to;
    if (pc == BK) p.kingSqB = m.to;
    if (pc == WK) p.castling &= ~3;
    if (pc == BK) p.castling &= ~12;
    if (m.from == 63 || m.to == 63) p.castling &= ~1;
    if (m.from == 56 || m.to == 56) p.castling &= ~2;
    if (m.from == 7 || m.to == 7) p.castling &= ~4;
    if (m.from == 0 || m.to == 0) p.castling &= ~8;
    if (pc == WP && m.to - m.from == 16) p.ep = m.from + 8;
    if (pc == BP && m.from - m.to == 16) p.ep = m.from - 8;
    p.white = !p.white;
    if (p.white) ++p.fullmove;
}

static void unmake_move(Position& p, const Move& m, const Undo& u) {
    p.white = !p.white;
    if (!p.white) --p.fullmove;
    p.castling = u.castling; p.ep = u.ep; p.halfmove = u.halfmove; p.kingSqW = u.kingSqW; p.kingSqB = u.kingSqB;
    int pc = p.b[m.to];
    p.b[m.to] = 0;
    if (m.castle) {
        if (m.to == 62) { p.b[63]=WR; p.b[61]=0; }
        else if (m.to == 58) { p.b[56]=WR; p.b[59]=0; }
        else if (m.to == 6) { p.b[7]=BR; p.b[5]=0; }
        else if (m.to == 2) { p.b[0]=BR; p.b[3]=0; }
    }
    p.b[m.from] = m.promo ? (p.white ? WP : BP) : pc;
    if (m.ep) p.b[p.white ? m.to - 8 : m.to + 8] = u.captured;
    else if (m.cap) p.b[m.to] = u.captured;
}

static int ttFlag=0;
struct TTEntry { uint64_t key; int depth; int score; int flag; Move best; };
static unordered_map<uint64_t, TTEntry> TT;

static uint64_t hash_pos(const Position& p) {
    uint64_t h = 1469598103934665603ull;
    auto mix = [&](uint64_t x){ h ^= x; h *= 1099511628211ull; };
    for (int i=0;i<64;++i) mix((uint64_t)p.b[i] + 17ull * i);
    mix(p.white); mix(p.castling); mix(p.ep + 2); mix(p.halfmove);
    return h;
}

static int quiesce(Position& p, int alpha, int beta, int ply);

static int search(Position& p, int depth, int alpha, int beta, int ply) {
    if (depth <= 0) return quiesce(p, alpha, beta, ply);
    uint64_t key = hash_pos(p);
    auto it = TT.find(key);
    if (it != TT.end() && it->second.depth >= depth) {
        if (it->second.flag == 0) return it->second.score;
        if (it->second.flag == 1 && it->second.score <= alpha) return alpha;
        if (it->second.flag == 2 && it->second.score >= beta) return beta;
    }
    vector<Move> mv = gen_moves(p);
    if (mv.empty()) {
        int ksq = p.white ? p.kingSqW : p.kingSqB;
        if (is_attacked(p, ksq, !p.white)) return -100000 + ply;
        return 0;
    }
    sort(mv.begin(), mv.end(), [&](const Move& a, const Move& b){
        int sa = (a.cap?10:0) + (a.promo?20:0) + (a.castle?2:0);
        int sb = (b.cap?10:0) + (b.promo?20:0) + (b.castle?2:0);
        return sa > sb;
    });
    int origAlpha = alpha;
    Move best;
    for (auto &m : mv) {
        Undo u;
        make_move(p, m, u);
        int score = -search(p, depth - 1, -beta, -alpha, ply + 1);
        unmake_move(p, m, u);
        if (score > alpha) { alpha = score; best = m; }
        if (alpha >= beta) break;
    }
    TT[key] = {key, depth, alpha, alpha <= origAlpha ? 1 : (alpha >= beta ? 2 : 0), best};
    return alpha;
}

static int quiesce(Position& p, int alpha, int beta, int ply) {
    int stand = evaluate(p);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;
    vector<Move> mv = gen_moves(p);
    sort(mv.begin(), mv.end(), [&](const Move& a, const Move& b){ return a.cap > b.cap; });
    for (auto &m : mv) {
        if (!m.cap && !m.promo && !m.ep) continue;
        Undo u;
        make_move(p, m, u);
        int score = -quiesce(p, -beta, -alpha, ply + 1);
        unmake_move(p, m, u);
        if (score > alpha) alpha = score;
        if (alpha >= beta) break;
    }
    return alpha;
}

static Position parse_fen(const string& fen) {
    Position p;
    fill(p.b.begin(), p.b.end(), 0);
    stringstream ss(fen);
    string board, turn, castle, epp;
    ss >> board >> turn >> castle >> epp >> p.halfmove >> p.fullmove;
    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') { --r; f = 0; continue; }
        if (isdigit((unsigned char)c)) { f += c - '0'; continue; }
        int s = sq(f, r);
        switch (c) {
            case 'P': p.b[s] = WP; break; case 'N': p.b[s] = WN; break; case 'B': p.b[s] = WB; break;
            case 'R': p.b[s] = WR; break; case 'Q': p.b[s] = WQ; break; case 'K': p.b[s] = WK; p.kingSqW = s; break;
            case 'p': p.b[s] = BP; break; case 'n': p.b[s] = BN; break; case 'b': p.b[s] = BB; break;
            case 'r': p.b[s] = BR; break; case 'q': p.b[s] = BQ; break; case 'k': p.b[s] = BK; p.kingSqB = s; break;
        }
        ++f;
    }
    p.white = (turn == "w");
    p.castling = 0;
    if (castle.find('K') != string::npos) p.castling |= 1;
    if (castle.find('Q') != string::npos) p.castling |= 2;
    if (castle.find('k') != string::npos) p.castling |= 4;
    if (castle.find('q') != string::npos) p.castling |= 8;
    p.ep = (epp == "-") ? -1 : sq(epp[0] - 'a', epp[1] - '1');
    return p;
}

static string format_move(Position p, const Move& m) {
    int pc = p.b[m.from];
    string s;
    char piece = 'p';
    if (pc == WN || pc == BN) piece = 'N';
    else if (pc == WB || pc == BB) piece = 'B';
    else if (pc == WR || pc == BR) piece = 'R';
    else if (pc == WQ || pc == BQ) piece = 'Q';
    else if (pc == WK || pc == BK) piece = 'K';
    s += piece;
    s += square_str(m.from);
    if (m.cap) s += 'x';
    s += square_str(m.to);
    if (m.promo) {
        s += '/';
        s += char(toupper((pc <= WK) ? (m.promo == WQ ? 'q' : m.promo == WR ? 'r' : m.promo == WB ? 'b' : 'n')
                                      : (m.promo == BQ ? 'q' : m.promo == BR ? 'r' : m.promo == BB ? 'b' : 'n')));
    }
    if (m.ep) s += "ep";
    Undo u;
    make_move(p, m, u);
    vector<Move> reply = gen_moves(p);
    bool mate = reply.empty() && is_attacked(p, p.white ? p.kingSqW : p.kingSqB, !p.white);
    bool check = !mate && is_attacked(p, p.white ? p.kingSqW : p.kingSqB, !p.white);
    unmake_move(p, m, u);
    if (mate) s += '#';
    else if (check) s += '+';
    if (m.castle) return (m.to == 62 || m.to == 6) ? "O-O" : "O-O-O";
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    double timeout;
    string fen;
    if (!(cin >> timeout)) return 0;
    getline(cin, fen);
    if (fen.empty()) getline(cin, fen);
    Position p = parse_fen(fen);
    auto start = chrono::steady_clock::now();
    auto limit = chrono::duration<double>(max(0.01, timeout * 0.95));
    Move best;
    string bestStr = "0000";
    TT.reserve(1 << 20);
    int prev = -1000000;
    for (int depth = 1; depth <= 64; ++depth) {
        int alpha = -1000000, beta = 1000000;
        vector<Move> mv = gen_moves(p);
        if (mv.empty()) break;
        sort(mv.begin(), mv.end(), [&](const Move& a, const Move& b){
            int sa = (a.cap?10:0) + (a.promo?20:0) + (a.castle?2:0);
            int sb = (b.cap?10:0) + (b.promo?20:0) + (b.castle?2:0);
            return sa > sb;
        });
        int bestScore = -1000000;
        Move localBest = mv[0];
        for (auto &m : mv) {
            if (chrono::steady_clock::now() - start > limit) break;
            Undo u;
            make_move(p, m, u);
            int score = -search(p, depth - 1, -beta, -alpha, 1);
            unmake_move(p, m, u);
            if (score > bestScore) { bestScore = score; localBest = m; }
            if (score > alpha) alpha = score;
        }
        best = localBest;
        bestStr = format_move(p, best);
        prev = bestScore;
        if (chrono::steady_clock::now() - start > limit) break;
    }
    cout << bestStr << '\n';
    return 0;
}
