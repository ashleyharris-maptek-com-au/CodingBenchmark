declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Task = {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
};

const machineCount: number = Number.isFinite(data.machineCount) ? data.machineCount : 1;
const tasks: Task[] = Array.isArray(data.tasks) ? data.tasks : [];

const n = tasks.length;
const idToIndex = new Map<string, number>();
for (let i = 0; i < n; i++) idToIndex.set(String(tasks[i].id), i);

const depsIdx: number[][] = Array.from({ length: n }, () => []);
const succ: number[][] = Array.from({ length: n }, () => []);
const indeg = new Array<number>(n).fill(0);

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (const depIdRaw of deps) {
    const depId = String(depIdRaw);
    const j = idToIndex.get(depId);
    if (j === undefined || j === i) continue;
    depsIdx[i].push(j);
    succ[j].push(i);
    indeg[i]++;
  }
}

// Longest path to a sink, used as a proxy for criticality.
const crit = new Array<number>(n).fill(0);
const outdeg = succ.map((s) => s.length);
const q: number[] = [];
for (let i = 0; i < n; i++) {
  if (outdeg[i] === 0) q.push(i);
}
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  crit[u] = Math.max(crit[u], Number(tasks[u].duration) || 0);
  for (const p of depsIdx[u]) {
    const cand = crit[u] + (Number(tasks[p].duration) || 0);
    if (cand > crit[p]) crit[p] = cand;
    outdeg[p]--;
    if (outdeg[p] === 0) q.push(p);
  }
}

class Heap {
  private a: number[] = [];
  constructor(private cmp: (i: number, j: number) => boolean) {}
  push(x: number) {
    const a = this.a;
    a.push(x);
    for (let i = a.length - 1; i > 0;) {
      const p = (i - 1) >> 1;
      if (!this.cmp(a[i], a[p])) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop(): number | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length) {
      a[0] = last;
      for (let i = 0;;) {
        let l = i * 2 + 1;
        let r = l + 1;
        let best = i;
        if (l < a.length && this.cmp(a[l], a[best])) best = l;
        if (r < a.length && this.cmp(a[r], a[best])) best = r;
        if (best === i) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
  get length() {
    return this.a.length;
  }
}

const cmp = (i: number, j: number): boolean => {
  const ti = tasks[i];
  const tj = tasks[j];
  const di = Number(ti.deadline);
  const dj = Number(tj.deadline);
  if (di !== dj) return di < dj;
  const wi = Number(ti.weight) || 0;
  const wj = Number(tj.weight) || 0;
  if (wi !== wj) return wi > wj;
  const ci = crit[i];
  const cj = crit[j];
  if (ci !== cj) return ci > cj;
  const duri = Number(ti.duration) || 0;
  const durj = Number(tj.duration) || 0;
  if (duri !== durj) return duri > durj;
  const idi = String(ti.id);
  const idj = String(tj.id);
  return idi < idj;
};

const heap = new Heap(cmp);
for (let i = 0; i < n; i++) {
  if (indeg[i] === 0) heap.push(i);
}

const order: string[] = [];
while (heap.length) {
  const u = heap.pop()!;
  order.push(String(tasks[u].id));
  for (const v of succ[u]) {
    indeg[v]--;
    if (indeg[v] === 0) heap.push(v);
  }
}

// If the input is cyclic, fall back to a stable listing of all ids.
if (order.length !== n) {
  const seen = new Set(order);
  for (const t of tasks) {
    const id = String(t.id);
    if (!seen.has(id)) order.push(id);
  }
}

process.stdout.write(JSON.stringify({ order }));
