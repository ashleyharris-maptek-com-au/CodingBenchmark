OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_p04 = OpConstant %float 0.04
%float_n04 = OpConstant %float -0.04
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%gl_in = OpVariable %ptr_in_arr_pv Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel
%p_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p = OpLoad %v4float %p_ptr
%c_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c = OpLoad %v3float %c_ptr

%px = OpCompositeExtract %float %p 0
%py = OpCompositeExtract %float %p 1

%x0 = OpFAdd %float %px %float_n04
%y0 = OpFAdd %float %py %float_n04
%x1 = OpFAdd %float %px %float_p04
%y1 = OpFAdd %float %py %float_n04
%x2 = OpFAdd %float %px %float_n04
%y2 = OpFAdd %float %py %float_p04
%x3 = OpFAdd %float %px %float_p04
%y3 = OpFAdd %float %py %float_p04

%pos0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%pos1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%pos2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
%pos3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1

%outp0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp0 %pos0
%outc0 = OpAccessChain %ptr_out_v3 %outColor %int_0
OpStore %outc0 %c
OpEmitVertex

%outp1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp1 %pos1
%outc1 = OpAccessChain %ptr_out_v3 %outColor %int_0
OpStore %outc1 %c
OpEmitVertex

%outp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp2 %pos2
%outc2 = OpAccessChain %ptr_out_v3 %outColor %int_0
OpStore %outc2 %c
OpEmitVertex

%outp3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp3 %pos3
%outc3 = OpAccessChain %ptr_out_v3 %outColor %int_0
OpStore %outc3 %c
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
