using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    private static int n, m;
    private static int[] from = Array.Empty<int>();
    private static int[] to = Array.Empty<int>();
    private static List<int>[] outEdges = Array.Empty<List<int>>();
    private static List<int>[] inEdges = Array.Empty<List<int>>();
    private static bool[] removed = Array.Empty<bool>();

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        n = fs.NextInt();
        m = fs.NextInt();
        from = new int[m];
        to = new int[m];
        outEdges = new List<int>[n];
        inEdges = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outEdges[i] = new List<int>();
            inEdges[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            from[i] = u;
            to[i] = v;
            outEdges[u].Add(v);
            inEdges[v].Add(u);
        }

        removed = new bool[n];
        var answer = SolveAll();

        answer.Sort();
        Console.WriteLine(answer.Count);
        if (answer.Count == 0)
        {
            Console.WriteLine();
        }
        else
        {
            for (int i = 0; i < answer.Count; i++)
            {
                if (i > 0) Console.Write(' ');
                Console.Write(answer[i]);
            }
            Console.WriteLine();
        }
    }

    private static List<int> SolveAll()
    {
        TrimAcyclic();
        var comps = GetSccComponents();
        if (comps.Count == 0) return new List<int>();

        List<int> best = null!;
        foreach (var comp in comps)
        {
            var sub = SolveComponent(comp);
            if (best == null) best = sub;
            else best.AddRange(sub);
        }
        return best ?? new List<int>();
    }

    private static List<int> SolveComponent(List<int> comp)
    {
        if (comp.Count <= 1)
        {
            if (comp.Count == 1)
            {
                int v = comp[0];
                foreach (var w in outEdges[v])
                {
                    if (w == v && !removed[v])
                    {
                        return new List<int> { v };
                    }
                }
            }
            return new List<int>();
        }

        var localRemoved = new bool[n];
        foreach (var v in comp) localRemoved[v] = true;

        var solver = new ExactSolver(comp, localRemoved);
        return solver.Solve();
    }

    private static void TrimAcyclic()
    {
        var q = new Queue<int>();
        int[] indeg = new int[n];
        int[] outdeg = new int[n];
        for (int i = 0; i < n; i++)
        {
            if (removed[i]) continue;
            int oi = 0, ii = 0;
            foreach (var v in outEdges[i]) if (!removed[v]) oi++;
            foreach (var v in inEdges[i]) if (!removed[v]) ii++;
            indeg[i] = ii;
            outdeg[i] = oi;
            if (ii == 0 || oi == 0) q.Enqueue(i);
        }

        while (q.Count > 0)
        {
            int v = q.Dequeue();
            if (removed[v]) continue;
            removed[v] = true;
            foreach (var u in inEdges[v])
            {
                if (removed[u]) continue;
                if (--outdeg[u] == 0) q.Enqueue(u);
            }
            foreach (var w in outEdges[v])
            {
                if (removed[w]) continue;
                if (--indeg[w] == 0) q.Enqueue(w);
            }
        }
    }

    private static List<List<int>> GetSccComponents()
    {
        int[] index = new int[n];
        int[] low = new int[n];
        bool[] onStack = new bool[n];
        Array.Fill(index, -1);
        var stack = new Stack<int>();
        int time = 0;
        var comps = new List<List<int>>();

        void Dfs(int v)
        {
            index[v] = low[v] = time++;
            stack.Push(v);
            onStack[v] = true;

            foreach (var w in outEdges[v])
            {
                if (removed[w]) continue;
                if (index[w] == -1)
                {
                    Dfs(w);
                    low[v] = Math.Min(low[v], low[w]);
                }
                else if (onStack[w])
                {
                    low[v] = Math.Min(low[v], index[w]);
                }
            }

            if (low[v] == index[v])
            {
                var comp = new List<int>();
                while (true)
                {
                    int x = stack.Pop();
                    onStack[x] = false;
                    comp.Add(x);
                    if (x == v) break;
                }
                if (comp.Count > 1 || HasSelfLoop(comp[0]))
                    comps.Add(comp);
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (!removed[i] && index[i] == -1) Dfs(i);
        }
        return comps;
    }

    private static bool HasSelfLoop(int v)
    {
        foreach (var w in outEdges[v]) if (w == v && !removed[v]) return true;
        return false;
    }

    private sealed class ExactSolver
    {
        private readonly List<int> vertices;
        private readonly bool[] active;

        public ExactSolver(List<int> comp, bool[] initialActive)
        {
            vertices = comp;
            active = new bool[n];
            foreach (var v in comp) active[v] = initialActive[v];
        }

        public List<int> Solve()
        {
            Reduce();
            if (!HasCycle()) return new List<int>();

            var cycle = FindCycle();
            List<int> best = null!;
            foreach (int v in cycle)
            {
                if (!active[v]) continue;
                var saved = (bool[])active.Clone();
                active[v] = false;
                Reduce();
                var sub = Solve();
                sub.Add(v);
                if (best == null || sub.Count < best.Count) best = sub;
                Array.Copy(saved, active, n);
            }
            return best ?? new List<int>();
        }

        private void Reduce()
        {
            var q = new Queue<int>();
            int[] indeg = new int[n];
            int[] outdeg = new int[n];
            foreach (var v in vertices)
            {
                if (!active[v]) continue;
                int ii = 0, oo = 0;
                foreach (var u in inEdges[v]) if (active[u]) ii++;
                foreach (var w in outEdges[v]) if (active[w]) oo++;
                indeg[v] = ii;
                outdeg[v] = oo;
                if (ii == 0 || oo == 0) q.Enqueue(v);
            }

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                if (!active[v]) continue;
                active[v] = false;
                foreach (var u in inEdges[v])
                {
                    if (!active[u]) continue;
                    if (--outdeg[u] == 0) q.Enqueue(u);
                }
                foreach (var w in outEdges[v])
                {
                    if (!active[w]) continue;
                    if (--indeg[w] == 0) q.Enqueue(w);
                }
            }
        }

        private bool HasCycle()
        {
            int[] color = new int[n];
            foreach (var v in vertices)
            {
                if (active[v] && color[v] == 0 && Dfs(v, color)) return true;
            }
            return false;
        }

        private bool Dfs(int v, int[] color)
        {
            color[v] = 1;
            foreach (var w in outEdges[v])
            {
                if (!active[w]) continue;
                if (color[w] == 1) return true;
                if (color[w] == 0 && Dfs(w, color)) return true;
            }
            color[v] = 2;
            return false;
        }

        private List<int> FindCycle()
        {
            int[] color = new int[n];
            int[] parent = new int[n];
            Array.Fill(parent, -1);
            int start = -1, end = -1;

            bool DfsFind(int v)
            {
                color[v] = 1;
                foreach (var w in outEdges[v])
                {
                    if (!active[w]) continue;
                    if (color[w] == 0)
                    {
                        parent[w] = v;
                        if (DfsFind(w)) return true;
                    }
                    else if (color[w] == 1)
                    {
                        start = w;
                        end = v;
                        return true;
                    }
                }
                color[v] = 2;
                return false;
            }

            foreach (var v in vertices)
            {
                if (active[v] && color[v] == 0 && DfsFind(v)) break;
            }

            var cycle = new List<int>();
            cycle.Add(start);
            for (int v = end; v != start; v = parent[v]) cycle.Add(v);
            return cycle;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream stream) => this.stream = stream;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }
    }
}
