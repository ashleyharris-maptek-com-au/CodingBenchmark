use std::io::{self, Read};
use std::collections::HashMap;
use std::time::{Instant, Duration};

#[derive(Clone)]
struct BitSet {
    words: Vec<u64>,
}

impl BitSet {
    fn new(size_bits: usize) -> Self {
        let num_words = (size_bits + 63) / 64;
        BitSet {
            words: vec![0; num_words],
        }
    }

    #[inline(always)]
    fn set_bit(&mut self, bit: usize) {
        let word = bit / 64;
        let pos = bit % 64;
        if word < self.words.len() {
            self.words[word] |= 1 << pos;
        }
    }

    #[inline(always)]
    fn get_bit(&self, bit: usize) -> bool {
        let word = bit / 64;
        let pos = bit % 64;
        if word < self.words.len() {
            (self.words[word] & (1 << pos)) != 0
        } else {
            false
        }
    }

    fn shift_or(&mut self, shift: usize) {
        let words = shift / 64;
        let bits = shift % 64;
        let n = self.words.len();
        if words >= n {
            return;
        }
        if bits == 0 {
            for i in (words..n).rev() {
                self.words[i] |= self.words[i - words];
            }
        } else {
            let shift_minus = 64 - bits;
            for i in (words + 1..n).rev() {
                self.words[i] |= (self.words[i - words] << bits) | (self.words[i - words - 1] >> shift_minus);
            }
            self.words[words] |= self.words[0] << bits;
        }
    }
}

struct Xorshift {
    state: u64,
}

impl Xorshift {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 { s = 1; }
        Xorshift { state: s }
    }
    
    #[inline(always)]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    
    #[inline(always)]
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() & 0xFFFFFFFFFFFFF) as f64 / 9007199254740992.0
    }
}

#[derive(Clone)]
struct DecomposedElement {
    value: i64,
    indices: Vec<usize>,
}

fn decompose_elements(elements: &[(i64, usize)]) -> Vec<DecomposedElement> {
    let mut groups: HashMap<i64, Vec<usize>> = HashMap::new();
    for &(val, idx) in elements {
        groups.entry(val).or_default().push(idx);
    }
    
    let mut decomposed = Vec::new();
    for (v, idxs) in groups {
        let mut c = idxs.len();
        let mut i = 0;
        let mut power = 1;
        while c > 0 {
            let count = if c >= power { power } else { c };
            let chunk = idxs[i..i+count].to_vec();
            decomposed.push(DecomposedElement {
                value: v * count as i64,
                indices: chunk,
            });
            i += count;
            c -= count;
            power *= 2;
        }
    }
    decomposed
}

fn solve_mitm(elements: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let n = elements.len();
    if n == 0 {
        if target == 0 { return Some(vec![]); }
        return None;
    }
    let n1 = n / 2;

    let mut left = Vec::with_capacity(1 << n1);
    left.push((0i64, 0u64));
    for i in 0..n1 {
        let val = elements[i].0;
        let len = left.len();
        for j in 0..len {
            let (s, mask) = left[j];
            left.push((s + val, mask | (1 << i)));
        }
    }
    left.sort_unstable_by_key(|x| x.0);

    struct MitmSearcher<'a> {
        elements: &'a [(i64, usize)],
        n1: usize,
        target: i64,
        left: &'a [(i64, u64)],
        result: Option<Vec<usize>>,
    }

    impl<'a> MitmSearcher<'a> {
        fn dfs(&mut self, idx: usize, current_sum: i64, current_mask: u64) -> bool {
            if self.result.is_some() {
                return true;
            }
            if idx == self.elements.len() {
                let needed = self.target - current_sum;
                if let Ok(pos) = self.left.binary_search_by_key(&needed, |x| x.0) {
                    let left_mask = self.left[pos].1;
                    let right_mask = current_mask;
                    let mut indices = Vec::new();
                    for i in 0..self.n1 {
                        if (left_mask & (1 << i)) != 0 {
                            indices.push(self.elements[i].1);
                        }
                    }
                    for i in self.n1..self.elements.len() {
                        if (right_mask & (1 << (i - self.n1))) != 0 {
                            indices.push(self.elements[i].1);
                        }
                    }
                    self.result = Some(indices);
                    return true;
                }
                return false;
            }

            if self.dfs(idx + 1, current_sum, current_mask) {
                return true;
            }
            if self.dfs(idx + 1, current_sum + self.elements[idx].0, current_mask | (1 << (idx - self.n1))) {
                return true;
            }
            false
        }
    }

    let mut searcher = MitmSearcher {
        elements,
        n1,
        target,
        left: &left,
        result: None,
    };
    searcher.dfs(n1, 0, 0);
    searcher.result
}

fn solve_dp_dc(elements: &[DecomposedElement], target: usize) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(vec![]);
    }
    if elements.is_empty() {
        return None;
    }
    if elements.len() == 1 {
        if elements[0].value == target as i64 {
            return Some(elements[0].indices.clone());
        } else {
            return None;
        }
    }
    
    let mid = elements.len() / 2;
    let left = &elements[0..mid];
    let right = &elements[mid..];
    
    let left_sum: i64 = left.iter().map(|e| e.value).sum();
    let left_limit = std::cmp::min(target, left_sum as usize);
    let mut dp_left = BitSet::new(left_limit + 1);
    dp_left.set_bit(0);
    for e in left {
        dp_left.shift_or(e.value as usize);
    }
    
    let right_sum: i64 = right.iter().map(|e| e.value).sum();
    let right_limit = std::cmp::min(target, right_sum as usize);
    let mut dp_right = BitSet::new(right_limit + 1);
    dp_right.set_bit(0);
    for e in right {
        dp_right.shift_or(e.value as usize);
    }
    
    for t1 in 0..=left_limit {
        let t2 = target - t1;
        if t2 <= right_limit {
            if dp_left.get_bit(t1) && dp_right.get_bit(t2) {
                let mut res_left = solve_dp_dc(left, t1)?;
                let res_right = solve_dp_dc(right, t2)?;
                res_left.extend(res_right);
                return Some(res_left);
            }
        }
    }
    
    None
}

fn dfs_pool(
    idx: usize,
    target: i64,
    pool: &[DecomposedElement],
    pos_sum: &[i64],
    neg_sum: &[i64],
    current_subset: &mut Vec<usize>,
    solutions: &mut Vec<usize>,
    start_time: Instant,
    timeout: Duration,
    backtracks: &mut u64,
    max_backtracks: u64,
) -> bool {
    if target == 0 {
        *solutions = current_subset.clone();
        return true;
    }
    if idx >= pool.len() {
        return false;
    }
    if target < neg_sum[idx] || target > pos_sum[idx] {
        return false;
    }
    if target == pos_sum[idx] {
        for i in idx..pool.len() {
            if pool[i].value > 0 {
                current_subset.push(i);
            }
        }
        *solutions = current_subset.clone();
        return true;
    }
    if target == neg_sum[idx] {
        for i in idx..pool.len() {
            if pool[i].value < 0 {
                current_subset.push(i);
            }
        }
        *solutions = current_subset.clone();
        return true;
    }
    if *backtracks > max_backtracks {
        return false;
    }
    if start_time.elapsed() > timeout {
        return false;
    }
    *backtracks += 1;
    
    let val = pool[idx].value;
    let same_sign = (val > 0 && target > 0) || (val < 0 && target < 0);
    if same_sign {
        current_subset.push(idx);
        if dfs_pool(idx + 1, target - val, pool, pos_sum, neg_sum, current_subset, solutions, start_time, timeout, backtracks, max_backtracks) {
            return true;
        }
        current_subset.pop();
        
        if dfs_pool(idx + 1, target, pool, pos_sum, neg_sum, current_subset, solutions, start_time, timeout, backtracks, max_backtracks) {
            return true;
        }
    } else {
        if dfs_pool(idx + 1, target, pool, pos_sum, neg_sum, current_subset, solutions, start_time, timeout, backtracks, max_backtracks) {
            return true;
        }
        
        current_subset.push(idx);
        if dfs_pool(idx + 1, target - val, pool, pos_sum, neg_sum, current_subset, solutions, start_time, timeout, backtracks, max_backtracks) {
            return true;
        }
        current_subset.pop();
    }
    
    false
}

fn solve_large_heuristic(nums: &[DecomposedElement], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        if target == 0 { return Some(vec![]); }
        return None;
    }
    
    let mut pool_elements = Vec::new();
    let mut pool_sum = 0i64;
    let mut pool_size = 0;
    
    for e in nums.iter().rev() {
        if pool_sum + e.value <= 20_000_000 && pool_size < 2000 {
            pool_elements.push(e.clone());
            pool_sum += e.value;
            pool_size += 1;
        } else {
            break;
        }
    }
    pool_elements.reverse();
    
    if pool_size < 5 {
        return None;
    }
    
    let candidates_len = n - pool_size;
    let candidates = &nums[0..candidates_len];
    
    let mut dp_pool = BitSet::new(pool_sum as usize + 1);
    dp_pool.set_bit(0);
    for e in &pool_elements {
        dp_pool.shift_or(e.value as usize);
    }
    
    let mut rng = Xorshift::new(123456789);
    
    for iter in 0..1000 {
        let mut t_rem = target;
        let mut chosen_indices = Vec::new();
        
        for e in candidates {
            if t_rem >= e.value {
                let include = if iter == 0 {
                    true
                } else {
                    rng.next_f64() < 0.8
                };
                if include {
                    t_rem -= e.value;
                    chosen_indices.extend(&e.indices);
                }
            }
        }
        
        if t_rem == 0 {
            return Some(chosen_indices);
        }
        
        if t_rem > 0 && t_rem <= pool_sum {
            if dp_pool.get_bit(t_rem as usize) {
                if let Some(pool_indices) = solve_dp_dc(&pool_elements, t_rem as usize) {
                    chosen_indices.extend(pool_indices);
                    return Some(chosen_indices);
                }
            }
        }
    }
    
    None
}

fn solve_mixed_heuristic(nums: &[DecomposedElement], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        if target == 0 { return Some(vec![]); }
        return None;
    }
    
    let pool_size = std::cmp::min(n / 2, 500);
    if pool_size == 0 {
        return None;
    }
    
    let candidates_len = n - pool_size;
    let candidates = &nums[0..candidates_len];
    let pool = &nums[candidates_len..];
    
    let pool_len = pool.len();
    let mut pos_sum = vec![0i64; pool_len + 1];
    let mut neg_sum = vec![0i64; pool_len + 1];
    for i in (0..pool_len).rev() {
        let val = pool[i].value;
        pos_sum[i] = pos_sum[i + 1] + if val > 0 { val } else { 0 };
        neg_sum[i] = neg_sum[i + 1] + if val < 0 { val } else { 0 };
    }
    
    let mut rng = Xorshift::new(987654321);
    
    for iter in 0..500 {
        let mut t_rem = target;
        let mut chosen_indices = Vec::new();
        
        for e in candidates {
            let val = e.value;
            if (t_rem - val).abs() < t_rem.abs() {
                let include = if iter == 0 {
                    true
                } else {
                    rng.next_f64() < 0.8
                };
                if include {
                    t_rem -= val;
                    chosen_indices.extend(&e.indices);
                }
            }
        }
        
        if t_rem == 0 {
            return Some(chosen_indices);
        }
        
        let mut current_subset = Vec::new();
        let mut pool_solution = Vec::new();
        let mut backtracks = 0;
        let start_time = Instant::now();
        let timeout = Duration::from_millis(5);
        
        let found = dfs_pool(
            0,
            t_rem,
            pool,
            &pos_sum,
            &neg_sum,
            &mut current_subset,
            &mut pool_solution,
            start_time,
            timeout,
            &mut backtracks,
            1000,
        );
        
        if found {
            for idx in pool_solution {
                chosen_indices.extend(&pool[idx].indices);
            }
            return Some(chosen_indices);
        }
    }
    
    None
}

fn solve_global_dfs(elements: &[DecomposedElement], target: i64, timeout: Duration) -> Option<Vec<usize>> {
    let mut sorted_elements = elements.to_vec();
    sorted_elements.sort_unstable_by_key(|e| e.value.abs());
    
    let limit = std::cmp::min(sorted_elements.len(), 1000);
    let search_elements = &sorted_elements[0..limit];
    
    let len = search_elements.len();
    let mut pos_sum = vec![0i64; len + 1];
    let mut neg_sum = vec![0i64; len + 1];
    for i in (0..len).rev() {
        let val = search_elements[i].value;
        pos_sum[i] = pos_sum[i + 1] + if val > 0 { val } else { 0 };
        neg_sum[i] = neg_sum[i + 1] + if val < 0 { val } else { 0 };
    }
    
    let mut current_subset = Vec::new();
    let mut solution = Vec::new();
    let mut backtracks = 0;
    let start_time = Instant::now();
    
    let found = dfs_pool(
        0,
        target,
        search_elements,
        &pos_sum,
        &neg_sum,
        &mut current_subset,
        &mut solution,
        start_time,
        timeout,
        &mut backtracks,
        1_000_000,
    );
    
    if found {
        let mut indices = Vec::new();
        for idx in solution {
            indices.extend(&search_elements[idx].indices);
        }
        Some(indices)
    } else {
        None
    }
}

fn parse_input() -> io::Result<(usize, i64, Vec<i64>)> {
    let mut buffer = Vec::new();
    io::stdin().read_to_end(&mut buffer)?;
    
    let mut cursor = 0;
    
    #[inline(always)]
    fn skip_whitespace(buffer: &[u8], cursor: &mut usize) {
        while *cursor < buffer.len() && buffer[*cursor] <= b' ' {
            *cursor += 1;
        }
    }
    
    #[inline(always)]
    fn parse_i64(buffer: &[u8], cursor: &mut usize) -> Option<i64> {
        skip_whitespace(buffer, cursor);
        if *cursor >= buffer.len() {
            return None;
        }
        let mut neg = false;
        if buffer[*cursor] == b'-' {
            neg = true;
            *cursor += 1;
        } else if buffer[*cursor] == b'+' {
            *cursor += 1;
        }
        if *cursor >= buffer.len() || !buffer[*cursor].is_ascii_digit() {
            return None;
        }
        let mut val = 0i64;
        while *cursor < buffer.len() && buffer[*cursor].is_ascii_digit() {
            val = val * 10 + (buffer[*cursor] - b'0') as i64;
            *cursor += 1;
        }
        if neg {
            val = -val;
        }
        Some(val)
    }

    let num_numbers = parse_i64(&buffer, &mut cursor)
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Failed to parse num_numbers"))? as usize;
    let target = parse_i64(&buffer, &mut cursor)
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Failed to parse target"))?;
    
    let mut nums = Vec::with_capacity(num_numbers);
    for _ in 0..num_numbers {
        if let Some(val) = parse_i64(&buffer, &mut cursor) {
            nums.push(val);
        } else {
            break;
        }
    }
    
    Ok((num_numbers, target, nums))
}

fn main() -> io::Result<()> {
    let (_num_numbers, target, nums) = match parse_input() {
        Ok(res) => res,
        Err(_) => {
            println!("NO");
            return Ok(());
        }
    };

    if target == 0 {
        println!("YES");
        println!("[]");
        return Ok(());
    }

    if nums.is_empty() {
        println!("NO");
        return Ok(());
    }

    let mut all_non_negative = true;
    let mut all_non_positive = true;
    for &x in &nums {
        if x < 0 {
            all_non_negative = false;
        }
        if x > 0 {
            all_non_positive = false;
        }
    }

    let (negated_nums, target_negated) = if all_non_positive && target <= 0 {
        let mut new_nums = nums.clone();
        for x in &mut new_nums {
            *x = -*x;
        }
        (new_nums, -target)
    } else {
        (nums.clone(), target)
    };

    let all_non_negative_now = all_non_negative || (all_non_positive && target <= 0);

    let mut elements = Vec::new();
    for (idx, &val) in negated_nums.iter().enumerate() {
        if val != 0 {
            if all_non_negative_now && val > target_negated {
                continue;
            }
            elements.push((val, idx));
        }
    }

    let pos_sum: i64 = elements.iter().filter(|&&(x, _)| x > 0).map(|&(x, _)| x).sum();
    let neg_sum: i64 = elements.iter().filter(|&&(x, _)| x < 0).map(|&(x, _)| x).sum();

    if target_negated < neg_sum || target_negated > pos_sum {
        println!("NO");
        return Ok(());
    }

    let n = elements.len();
    let result = if n <= 40 {
        solve_mitm(&elements, target_negated)
    } else {
        let mut decomposed = decompose_elements(&elements);
        decomposed.sort_unstable_by(|a, b| b.value.abs().cmp(&a.value.abs()));

        if all_non_negative_now {
            if target_negated <= 20_000_000 {
                solve_dp_dc(&decomposed, target_negated as usize)
            } else {
                if let Some(res) = solve_large_heuristic(&decomposed, target_negated) {
                    Some(res)
                } else {
                    solve_global_dfs(&decomposed, target_negated, Duration::from_secs(250))
                }
            }
        } else {
            if let Some(res) = solve_mixed_heuristic(&decomposed, target_negated) {
                Some(res)
            } else {
                solve_global_dfs(&decomposed, target_negated, Duration::from_secs(250))
            }
        }
    };

    if let Some(mut indices) = result {
        let mut sum = 0;
        for &idx in &indices {
            sum += nums[idx];
        }
        if sum == target {
            println!("YES");
            indices.sort_unstable();
            print!("[");
            for (i, &idx) in indices.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!("]");
        } else {
            println!("NO");
        }
    } else {
        println!("NO");
    }

    Ok(())
}
