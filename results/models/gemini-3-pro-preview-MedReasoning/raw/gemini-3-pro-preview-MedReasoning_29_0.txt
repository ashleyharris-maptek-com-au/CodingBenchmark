using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Threading;
using System.Threading.Tasks;

class Program {
    // Shared state for all search threads
    class SharedState {
        public volatile int GlobalBestSize;
        public int[] GlobalBestClique;
        public object CliqueLock = new object();
        public volatile bool CancelSearch = false;
        public Stopwatch Stopwatch = new Stopwatch();
    }

    // Fast reader for parsing large undirected graph inputs efficiently
    class FastReader {
        private const int BufferSize = 1 << 20;
        private Stream stream;
        private byte[] buffer = new byte[BufferSize];
        private int bufferPointer = 0;
        private int bytesRead = 0;

        public FastReader() {
            stream = Console.OpenStandardInput();
        }

        private void fillBuffer() {
            bufferPointer = 0;
            bytesRead = stream.Read(buffer, 0, BufferSize);
        }

        public int NextInt() {
            int ret = 0;
            byte c;
            while (true) {
                if (bufferPointer >= bytesRead) {
                    fillBuffer();
                    if (bytesRead == 0) return -1;
                }
                c = buffer[bufferPointer++];
                if (c > 32) break;
            }
            do {
                ret = ret * 10 + (c - '0');
                if (bufferPointer >= bytesRead) {
                    fillBuffer();
                    if (bytesRead == 0) return ret;
                }
                c = buffer[bufferPointer++];
            } while (c >= '0' && c <= '9');
            return ret;
        }
    }

    // High-performance bitset implementation for fast intersection and cardinality checks
    struct BitSet {
        public ulong[] Array;
        public int Offset;
        public int Words;

        public BitSet(ulong[] array, int offset, int words) {
            this.Array = array;
            this.Offset = offset;
            this.Words = words;
        }

        public bool IsEmpty() {
            for (int i = 0; i < Words; i++) {
                if (Array[Offset + i] != 0) return false;
            }
            return true;
        }

        public void Clear() {
            System.Array.Clear(Array, Offset, Words);
        }

        public void SetBit(int bit) {
            Array[Offset + (bit >> 6)] |= (1UL << (bit & 63));
        }

        public void ClearBit(int bit) {
            Array[Offset + (bit >> 6)] &= ~(1UL << (bit & 63));
        }

        public int GetFirstSetBit(ref int startWord) {
            for (int i = startWord; i < Words; i++) {
                ulong val = Array[Offset + i];
                if (val != 0) {
                    startWord = i;
                    return i * 64 + TrailingZeroCount(val);
                }
            }
            return -1;
        }

        public int IntersectAndCount(BitSet a, BitSet b) {
            int count = 0;
            for (int i = 0; i < Words; i++) {
                ulong val = a.Array[a.Offset + i] & b.Array[b.Offset + i];
                Array[Offset + i] = val;
                count += PopCount(val);
            }
            return count;
        }
    }

    [System.Runtime.CompilerServices.MethodImpl(System.Runtime.CompilerServices.MethodImplOptions.AggressiveInlining)]
    private static int PopCount(ulong x) {
        x -= (x >> 1) & 0x5555555555555555UL;
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        x = (x + (x >> 4)) & 0x0f0f0f0f0f0f0f0fUL;
        return (int)((x * 0x0101010101010101UL) >> 56);
    }

    [System.Runtime.CompilerServices.MethodImpl(System.Runtime.CompilerServices.MethodImplOptions.AggressiveInlining)]
    private static int TrailingZeroCount(ulong x) {
        if (x == 0) return 64;
        int n = 0;
        if ((x & 0xFFFFFFFFUL) == 0) { n += 32; x >>= 32; }
        if ((x & 0xFFFFUL) == 0) { n += 16; x >>= 16; }
        if ((x & 0xFFUL) == 0) { n += 8; x >>= 8; }
        if ((x & 0xFUL) == 0) { n += 4; x >>= 4; }
        if ((x & 3UL) == 0) { n += 2; x >>= 2; }
        if ((x & 1UL) == 0) { n += 1; }
        return n;
    }

    // Thread-local workspace to avoid memory allocations during search
    class ThreadLocalWorkspace {
        public ulong[] BitsetPool;
        public int WordsPerBitset;
        public int MaxDepth;
        public BitSet[] AdjMasks;
        public int[] CurrentClique;
        public int[] LocalVertices;
        public int Steps;

        // Recursion stacks for iterative DFS search
        public int[] pCountStack;
        public int[] remainingStack;
        public int[] startWordStack;
        public int[] uStack;

        public ThreadLocalWorkspace(int maxDegeneracy) {
            this.WordsPerBitset = (maxDegeneracy + 63) / 64;
            if (this.WordsPerBitset == 0) this.WordsPerBitset = 1;
            this.MaxDepth = maxDegeneracy + 2;
            this.BitsetPool = new ulong[MaxDepth * WordsPerBitset];
            this.AdjMasks = new BitSet[maxDegeneracy];
            ulong[] adjPool = new ulong[maxDegeneracy * WordsPerBitset];
            for (int i = 0; i < maxDegeneracy; i++) {
                this.AdjMasks[i] = new BitSet(adjPool, i * WordsPerBitset, WordsPerBitset);
            }
            this.CurrentClique = new int[MaxDepth];
            this.LocalVertices = new int[maxDegeneracy];
            this.Steps = 0;

            this.pCountStack = new int[MaxDepth];
            this.remainingStack = new int[MaxDepth];
            this.startWordStack = new int[MaxDepth];
            this.uStack = new int[MaxDepth];
        }

        public BitSet GetBitSetAtDepth(int depth) {
            return new BitSet(BitsetPool, depth * WordsPerBitset, WordsPerBitset);
        }

        public void InitializeSubproblem(int startVertex, int[] P, int[][] adj) {
            int p = P.Length;
            System.Array.Copy(P, LocalVertices, p);
            for (int j = 0; j < p; j++) {
                int u = P[j];
                int[] adjU = adj[u];
                BitSet mask = AdjMasks[j];
                mask.Clear();
                int ptr1 = 0;
                int ptr2 = 0;
                while (ptr1 < p && ptr2 < adjU.Length) {
                    if (P[ptr1] == adjU[ptr2]) {
                        mask.SetBit(ptr1);
                        ptr1++;
                        ptr2++;
                    } else if (P[ptr1] < adjU[ptr2]) {
                        ptr1++;
                    } else {
                        ptr2++;
                    }
                }
            }
        }

        public void SearchIterative(int startVertex, int initialPCount, SharedState state) {
            int depth = 0;
            pCountStack[0] = initialPCount;
            remainingStack[0] = initialPCount;
            startWordStack[0] = 0;

            BitSet P = GetBitSetAtDepth(0);
            uStack[0] = P.GetFirstSetBit(ref startWordStack[0]);

            while (true) {
                if ((Steps++ & 4095) == 0) {
                    if (state.Stopwatch.ElapsedMilliseconds > 280000) {
                        state.CancelSearch = true;
                    }
                }
                if (state.CancelSearch) return;

                int pCount = pCountStack[depth];
                int remaining = remainingStack[depth];
                int u = uStack[depth];

                if (depth + 1 + remaining <= state.GlobalBestSize || u == -1) {
                    if (depth == 0) {
                        break;
                    }
                    depth--;
                    P = GetBitSetAtDepth(depth);
                    int prevU = uStack[depth];
                    P.ClearBit(prevU);
                    remainingStack[depth]--;
                    uStack[depth] = P.GetFirstSetBit(ref startWordStack[depth]);
                    continue;
                }

                CurrentClique[depth] = u;

                BitSet P_next = GetBitSetAtDepth(depth + 1);
                int nextCount = P_next.IntersectAndCount(P, AdjMasks[u]);

                if (nextCount == 0) {
                    int newSize = depth + 2;
                    if (newSize > state.GlobalBestSize) {
                        lock (state.CliqueLock) {
                            if (newSize > state.GlobalBestSize) {
                                int[] newClique = new int[newSize];
                                newClique[0] = startVertex;
                                for (int d = 0; d <= depth; d++) {
                                    newClique[d + 1] = LocalVertices[CurrentClique[d]];
                                }
                                state.GlobalBestSize = newSize;
                                state.GlobalBestClique = newClique;
                            }
                        }
                    }

                    P.ClearBit(u);
                    remainingStack[depth]--;
                    uStack[depth] = P.GetFirstSetBit(ref startWordStack[depth]);
                } else {
                    depth++;
                    pCountStack[depth] = nextCount;
                    remainingStack[depth] = nextCount;
                    startWordStack[depth] = 0;
                    P = P_next;
                    uStack[depth] = P.GetFirstSetBit(ref startWordStack[depth]);
                }
            }
        }
    }

    // Greedy heuristic to find a high-quality initial clique to enable massive branch pruning
    static int[] FindHeuristicClique(int V, int[][] adj, int[] degrees) {
        int[] bestClique = new int[0];
        
        int[] sortedVertices = new int[V];
        for (int i = 0; i < V; i++) sortedVertices[i] = i;
        Array.Sort(sortedVertices, (a, b) => degrees[b].CompareTo(degrees[a]));
        
        int trials = Math.Min(100, V);
        for (int t = 0; t < trials; t++) {
            List<int> clique = new List<int>();
            int start = sortedVertices[t];
            clique.Add(start);
            
            int[] neighbors = adj[start];
            List<int> candidates = new List<int>(neighbors);
            candidates.Sort((a, b) => degrees[b].CompareTo(degrees[a]));
            
            foreach (int c in candidates) {
                bool ok = true;
                foreach (int v in clique) {
                    if (v == start) continue;
                    if (Array.BinarySearch(adj[v], c) < 0) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    clique.Add(c);
                }
            }
            
            if (clique.Count > bestClique.Length) {
                bestClique = clique.ToArray();
            }
        }
        
        Random rand = new Random(42);
        int randomTrials = Math.Min(200, V);
        for (int t = 0; t < randomTrials; t++) {
            List<int> clique = new List<int>();
            int limit = Math.Max(1, V / 10);
            int start = sortedVertices[rand.Next(limit)];
            clique.Add(start);
            
            int[] neighbors = adj[start];
            List<int> candidates = new List<int>(neighbors);
            
            while (candidates.Count > 0) {
                int nextIdx = 0;
                if (rand.NextDouble() < 0.2) {
                    nextIdx = rand.Next(candidates.Count);
                } else {
                    int maxDeg = -1;
                    for (int i = 0; i < candidates.Count; i++) {
                        int d = degrees[candidates[i]];
                        if (d > maxDeg) {
                            maxDeg = d;
                            nextIdx = i;
                        }
                    }
                }
                
                int nextVertex = candidates[nextIdx];
                clique.Add(nextVertex);
                
                int[] nextNeighbors = adj[nextVertex];
                List<int> nextCandidates = new List<int>();
                foreach (int c in candidates) {
                    if (c == nextVertex) continue;
                    if (Array.BinarySearch(nextNeighbors, c) >= 0) {
                        nextCandidates.Add(c);
                    }
                }
                candidates = nextCandidates;
            }
            
            if (clique.Count > bestClique.Length) {
                bestClique = clique.ToArray();
            }
        }
        
        return bestClique;
    }

    // Matula-Beck algorithm to compute degeneracy ordering in O(V + E) time
    static int[] ComputeDegeneracyOrdering(int V, int[][] adj, int[] degrees, out int degeneracy) {
        int[] order = new int[V];
        int[] deg = (int[])degrees.Clone();
        
        int[] head = new int[V];
        for (int i = 0; i < V; i++) head[i] = -1;
        int[] prev = new int[V];
        int[] next = new int[V];
        
        for (int i = 0; i < V; i++) {
            int d = deg[i];
            next[i] = head[d];
            if (head[d] != -1) {
                prev[head[d]] = i;
            }
            prev[i] = -1;
            head[d] = i;
        }
        
        int maxDeg = 0;
        int minDegPointer = 0;
        bool[] removed = new bool[V];
        
        for (int i = 0; i < V; i++) {
            while (minDegPointer < V && head[minDegPointer] == -1) {
                minDegPointer++;
            }
            
            int u = head[minDegPointer];
            head[minDegPointer] = next[u];
            if (next[u] != -1) {
                prev[next[u]] = -1;
            }
            
            order[i] = u;
            removed[u] = true;
            maxDeg = Math.Max(maxDeg, minDegPointer);
            
            foreach (int v in adj[u]) {
                if (!removed[v]) {
                    int vd = deg[v];
                    if (prev[v] != -1) {
                        next[prev[v]] = next[v];
                    } else {
                        head[vd] = next[v];
                    }
                    if (next[v] != -1) {
                        prev[next[v]] = prev[v];
                    }
                    
                    deg[v]--;
                    vd--;
                    
                    next[v] = head[vd];
                    if (head[vd] != -1) {
                        prev[head[vd]] = v;
                    }
                    prev[v] = -1;
                    head[vd] = v;
                    
                    if (vd < minDegPointer) {
                        minDegPointer = vd;
                    }
                }
            }
        }
        
        degeneracy = maxDeg;
        return order;
    }

    static void Main(string[] args) {
        FastReader reader = new FastReader();
        int V = reader.NextInt();
        if (V <= 0) return;
        int E = reader.NextInt();
        if (E < 0) return;

        int[] edgeU = new int[E];
        int[] edgeV = new int[E];
        int actualE = 0;
        for (int i = 0; i < E; i++) {
            int u = reader.NextInt();
            int v = reader.NextInt();
            if (u < 0 || u >= V || v < 0 || v >= V) continue;
            if (u == v) continue;
            edgeU[actualE] = u;
            edgeV[actualE] = v;
            actualE++;
        }
        E = actualE;

        int[] degrees = new int[V];
        for (int i = 0; i < E; i++) {
            degrees[edgeU[i]]++;
            degrees[edgeV[i]]++;
        }

        int[] adjFlat = new int[2 * E];
        int[] head = new int[V + 1];
        for (int i = 0; i < V; i++) {
            head[i + 1] = head[i] + degrees[i];
        }

        int[] currentHead = (int[])head.Clone();
        for (int i = 0; i < E; i++) {
            int u = edgeU[i];
            int v = edgeV[i];
            adjFlat[currentHead[u]++] = v;
            adjFlat[currentHead[v]++] = u;
        }

        int[][] adj = new int[V][];
        for (int i = 0; i < V; i++) {
            int start = head[i];
            int end = head[i + 1];
            int count = end - start;
            if (count > 1) {
                Array.Sort(adjFlat, start, count);
                int writePtr = start;
                for (int readPtr = start; readPtr < end; readPtr++) {
                    if (readPtr == start || adjFlat[readPtr] != adjFlat[readPtr - 1]) {
                        adjFlat[writePtr++] = adjFlat[readPtr];
                    }
                }
                degrees[i] = writePtr - start;
            } else {
                degrees[i] = count;
            }
            adj[i] = new int[degrees[i]];
            Array.Copy(adjFlat, head[i], adj[i], 0, degrees[i]);
        }

        // Free memory early
        edgeU = null;
        edgeV = null;
        adjFlat = null;
        head = null;
        currentHead = null;
        GC.Collect();

        // 1. Find a high-quality initial heuristic clique
        int[] initialClique = FindHeuristicClique(V, adj, degrees);

        SharedState state = new SharedState();
        state.GlobalBestSize = initialClique.Length;
        state.GlobalBestClique = initialClique;
        state.Stopwatch.Start();

        // 2. Compute degeneracy ordering and graph degeneracy
        int degeneracy;
        int[] order = ComputeDegeneracyOrdering(V, adj, degrees, out degeneracy);

        if (degeneracy == 0) {
            PrintClique(state.GlobalBestClique);
            return;
        }

        // 3. Compute forward neighbors for each vertex in the degeneracy ordering
        int[] pos = new int[V];
        for (int i = 0; i < V; i++) {
            pos[order[i]] = i;
        }

        int[][] forwardNeighbors = new int[V][];
        for (int u = 0; u < V; u++) {
            int count = 0;
            foreach (int v in adj[u]) {
                if (pos[v] > pos[u]) count++;
            }
            forwardNeighbors[u] = new int[count];
            int idx = 0;
            foreach (int v in adj[u]) {
                if (pos[v] > pos[u]) {
                    forwardNeighbors[u][idx++] = v;
                }
            }
        }

        // 4. Identify active indices (vertices that can potentially yield a larger clique)
        List<int> activeIndices = new List<int>();
        for (int i = 0; i < V; i++) {
            int u = order[i];
            if (forwardNeighbors[u].Length >= state.GlobalBestSize - 1) {
                activeIndices.Add(i);
            }
        }

        // Sort active indices by forward neighbors length descending (heaviest and most promising first)
        activeIndices.Sort((a, b) => forwardNeighbors[order[b]].Length.CompareTo(forwardNeighbors[order[a]].Length));

        // 5. Parallel search over all active subproblems
        Parallel.ForEach(activeIndices, 
            () => new ThreadLocalWorkspace(degeneracy), 
            (idx, loopState, workspace) => {
                if (state.CancelSearch) {
                    loopState.Stop();
                    return workspace;
                }

                int startVertex = order[idx];
                int[] P = forwardNeighbors[startVertex];
                int p = P.Length;

                if (p + 1 <= state.GlobalBestSize) {
                    return workspace;
                }

                workspace.InitializeSubproblem(startVertex, P, adj);

                BitSet P_initial = workspace.GetBitSetAtDepth(0);
                P_initial.Clear();
                for (int j = 0; j < p; j++) {
                    P_initial.SetBit(j);
                }

                workspace.SearchIterative(startVertex, p, state);

                return workspace;
            }, 
            (workspace) => {}
        );

        PrintClique(state.GlobalBestClique);
    }

    static void PrintClique(int[] clique) {
        Console.WriteLine(clique.Length);
        for (int i = 0; i < clique.Length; i++) {
            if (i > 0) Console.Write(" ");
            Console.Write(clique[i]);
        }
        Console.WriteLine();
    }
}
