#include <iostream>
#include <string>
#include <vector>
#include <cmath>
#include <algorithm>
#include <windows.h>

using namespace std;

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

struct Point3D { double x, y, z; };

struct Terrain {
    uint16_t* data = nullptr;
    int width, depth, height;
    bool valid = false;
    HANDLE hFile = INVALID_HANDLE_VALUE;
    HANDLE hMapping = NULL;

    void load(const string& path, int w, int d, int h) {
        width = w; depth = d; height = h;
        hFile = CreateFileA(path.c_str(), GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
        if (hFile == INVALID_HANDLE_VALUE) return;
        hMapping = CreateFileMappingA(hFile, NULL, PAGE_READONLY, 0, 0, NULL);
        if (!hMapping) return;
        data = (uint16_t*)MapViewOfFile(hMapping, FILE_MAP_READ, 0, 0, 0);
        if (data) valid = true;
    }

    void get_cell(int x, int y, uint16_t& a, uint16_t& b, uint16_t& c) {
        if (!valid || x < 0 || x >= width || y < 0 || y >= depth) {
            a = b = c = 0;
            return;
        }
        size_t idx = ((size_t)x * (size_t)depth + (size_t)y) * 3;
        a = data[idx];
        b = data[idx+1];
        c = data[idx+2];
    }

    ~Terrain() {
        if (data) UnmapViewOfFile(data);
        if (hMapping) CloseHandle(hMapping);
        if (hFile != INVALID_HANDLE_VALUE) CloseHandle(hFile);
    }
};

Terrain terrain;

bool is_line_clear(double x0, double y0, double z0, double x1, double y1, double z1) {
    double dist = sqrt((x1-x0)*(x1-x0) + (y1-y0)*(y1-y0) + (z1-z0)*(z1-z0));
    int steps = max(10, (int)(dist / 2.0));
    for (int i = 0; i <= steps; ++i) {
        double t = (double)i / steps;
        double x = x0 + t*(x1-x0);
        double y = y0 + t*(y1-y0);
        double z = z0 + t*(z1-z0);
        
        if (x < 0 || x >= terrain.width || y < 0 || y >= terrain.depth || z < 0 || z >= terrain.height) {
            return false;
        }
        
        uint16_t a, b, c;
        terrain.get_cell((int)x, (int)y, a, b, c);
        
        if (z < a) return false;
        if (z >= b && z < c) return false;
    }
    return true;
}

struct RRTNode {
    double x, y, z;
    int parent;
};

vector<Point3D> find_path_rrt(double sx, double sy, double sz, double tx, double ty, double tz) {
    vector<RRTNode> nodes;
    nodes.push_back({sx, sy, sz, -1});
    
    auto dist_sq = [](const RRTNode& a, const RRTNode& b) {
        return (a.x-b.x)*(a.x-b.x) + (a.y-b.y)*(a.y-b.y) + (a.z-b.z)*(a.z-b.z);
    };
    
    bool found = false;
    int best_node = -1;
    
    srand(42);
    
    for (int iter = 0; iter < 50000; ++iter) {
        RRTNode sample;
        if (rand() % 100 < 10) {
            sample = {tx, ty, tz, -1};
        } else {
            sample.x = ((double)rand() / RAND_MAX) * terrain.width;
            sample.y = ((double)rand() / RAND_MAX) * terrain.depth;
            sample.z = ((double)rand() / RAND_MAX) * terrain.height;
        }
        
        int nearest = 0;
        double min_d = dist_sq(nodes[0], sample);
        for (int i = 1; i < (int)nodes.size(); ++i) {
            double d = dist_sq(nodes[i], sample);
            if (d < min_d) {
                min_d = d;
                nearest = i;
            }
        }
        
        double step_size = 100.0;
        double d = sqrt(min_d);
        if (d > step_size) {
            sample.x = nodes[nearest].x + (sample.x - nodes[nearest].x) / d * step_size;
            sample.y = nodes[nearest].y + (sample.y - nodes[nearest].y) / d * step_size;
            sample.z = nodes[nearest].z + (sample.z - nodes[nearest].z) / d * step_size;
        }
        
        if (is_line_clear(nodes[nearest].x, nodes[nearest].y, nodes[nearest].z, sample.x, sample.y, sample.z)) {
            sample.parent = nearest;
            nodes.push_back(sample);
            
            if (is_line_clear(sample.x, sample.y, sample.z, tx, ty, tz)) {
                nodes.push_back({tx, ty, tz, (int)nodes.size() - 1});
                best_node = (int)nodes.size() - 1;
                found = true;
                break;
            }
        }
    }
    
    if (found) {
        vector<Point3D> path;
        int curr = best_node;
        while (curr != -1) {
            path.push_back({nodes[curr].x, nodes[curr].y, nodes[curr].z});
            curr = nodes[curr].parent;
        }
        reverse(path.begin(), path.end());
        
        vector<Point3D> smooth_path;
        smooth_path.push_back(path[0]);
        int current_idx = 0;
        while (current_idx < (int)path.size() - 1) {
            int furthest = current_idx + 1;
            for (int j = (int)path.size() - 1; j > current_idx + 1; --j) {
                if (is_line_clear(path[current_idx].x, path[current_idx].y, path[current_idx].z,
                                  path[j].x, path[j].y, path[j].z)) {
                    furthest = j;
                    break;
                }
            }
            smooth_path.push_back(path[furthest]);
            current_idx = furthest;
        }
        return smooth_path;
    }
    
    return {{tx, ty, tz}};
}

vector<Point3D> find_path(double sx, double sy, double sz, double tx, double ty, double tz) {
    if (is_line_clear(sx, sy, sz, tx, ty, tz)) {
        return {{tx, ty, tz}};
    }
    
    for (double z = max(sz, tz); z < terrain.height; z += 50) {
        if (is_line_clear(sx, sy, sz, sx, sy, z) &&
            is_line_clear(sx, sy, z, tx, ty, z) &&
            is_line_clear(tx, ty, z, tx, ty, tz)) {
            return {{sx, sy, z}, {tx, ty, z}, {tx, ty, tz}};
        }
    }
    
    for (double z = tz; z < terrain.height; z += 50) {
        for (int angle = 0; angle < 8; ++angle) {
            double a = angle * M_PI / 4.0;
            double r = 100;
            double app_x = tx + r * cos(a);
            double app_y = ty + r * sin(a);
            
            if (app_x < 0 || app_x >= terrain.width || app_y < 0 || app_y >= terrain.depth) continue;
            
            if (is_line_clear(sx, sy, sz, sx, sy, z) &&
                is_line_clear(sx, sy, z, app_x, app_y, z) &&
                is_line_clear(app_x, app_y, z, app_x, app_y, tz + 20) &&
                is_line_clear(app_x, app_y, tz + 20, tx, ty, tz)) {
                return {
                    {sx, sy, z},
                    {app_x, app_y, z},
                    {app_x, app_y, tz + 20},
                    {tx, ty, tz}
                };
            }
        }
    }
    
    return find_path_rrt(sx, sy, sz, tx, ty, tz);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int width, depth, height;
    double gravity, max_thrust, fuel;
    double start_x, start_y, start_z, target_x, target_y, target_z;
    
    if (!(cin >> width >> depth >> height >> gravity >> max_thrust >> fuel
              >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 0;
    }
    
    string terrain_path;
    getline(cin, terrain_path);
    while (terrain_path.empty() || terrain_path.find_first_not_of(" \r\n\t") == string::npos) {
        getline(cin, terrain_path);
    }
    terrain_path.erase(0, terrain_path.find_first_not_of(" \r\n\t"));
    terrain_path.erase(terrain_path.find_last_not_of(" \r\n\t") + 1);
    
    string state_marker;
    cin >> state_marker;
    
    terrain.load(terrain_path, width, depth, height);
    
    vector<Point3D> waypoints = find_path(start_x, start_y, start_z, target_x, target_y, target_z);
    int wp_index = 0;
    
    double x, y, z, vx, vy, vz, pitch, yaw, roll, pitch_rate, yaw_rate, roll_rate, fuel_rem;
    while (cin >> x >> y >> z >> vx >> vy >> vz >> pitch >> yaw >> roll 
               >> pitch_rate >> yaw_rate >> roll_rate >> fuel_rem) {
        
        if (wp_index < (int)waypoints.size() - 1) {
            double dx = waypoints[wp_index].x - x;
            double dy = waypoints[wp_index].y - y;
            double dz = waypoints[wp_index].z - z;
            double dist = sqrt(dx*dx + dy*dy + dz*dz);
            if (dist < 15.0) {
                wp_index++;
            }
        }
        
        Point3D wp = waypoints[wp_index];
        bool is_final_wp = (wp_index == (int)waypoints.size() - 1);
        
        double target_wp_z = wp.z;
        if (is_final_wp) {
            target_wp_z = target_z - 0.5;
        }
        
        double ex = wp.x - x;
        double ey = wp.y - y;
        double ez = target_wp_z - z;
        double dist_horiz = sqrt(ex*ex + ey*ey);
        
        double a_max_horiz = sqrt(max(0.1, max_thrust*max_thrust - gravity*gravity)) * 0.4;
        double a_max_up = (max_thrust - gravity) * 0.5;
        double a_max_down = gravity * 0.8;
        
        double max_v_horiz = 200.0;
        double max_v_up = 50.0;
        double max_v_down = 20.0;
        
        if (is_final_wp) {
            if (dist_horiz < 50.0) max_v_horiz = 10.0;
            if (dist_horiz < 10.0) max_v_horiz = 2.0;
            
            double dist_z_to_target = abs(z - target_z);
            if (dist_z_to_target < 50.0) max_v_down = 10.0;
            if (dist_z_to_target < 20.0) max_v_down = 2.0;
            if (dist_z_to_target < 5.0)  max_v_down = 1.0;
        }
        
        double v_des_horiz = sqrt(2 * a_max_horiz * dist_horiz);
        if (dist_horiz < 10.0) v_des_horiz = dist_horiz * sqrt(2 * a_max_horiz * 10.0) / 10.0;
        v_des_horiz = min(v_des_horiz, max_v_horiz);
        
        double vx_des = 0, vy_des = 0;
        if (dist_horiz > 0.1) {
            vx_des = (ex / dist_horiz) * v_des_horiz;
            vy_des = (ey / dist_horiz) * v_des_horiz;
        }
        
        double vz_des = 0;
        if (ez > 0) {
            vz_des = sqrt(2 * a_max_up * ez);
            if (ez < 10.0) vz_des = ez * sqrt(2 * a_max_up * 10.0) / 10.0;
            vz_des = min(vz_des, max_v_up);
        } else {
            double d_down = -ez;
            vz_des = -sqrt(2 * a_max_down * d_down);
            if (d_down < 10.0) vz_des = -d_down * sqrt(2 * a_max_down * 10.0) / 10.0;
            vz_des = max(vz_des, -max_v_down);
        }
        
        double K_vel = 1.0;
        double ax = K_vel * (vx_des - vx);
        double ay = K_vel * (vy_des - vy);
        double az = K_vel * (vz_des - vz) + gravity;
        
        az = max(0.0, az);
        
        if (az > max_thrust) {
            az = max_thrust;
            ax = 0;
            ay = 0;
        } else {
            double max_h = sqrt(max_thrust*max_thrust - az*az);
            double a_horiz = sqrt(ax*ax + ay*ay);
            if (a_horiz > max_h * 0.95) {
                ax = ax / a_horiz * max_h * 0.95;
                ay = ay / a_horiz * max_h * 0.95;
            }
        }
        
        double pitch_des1 = atan2(sqrt(ax*ax + ay*ay), az);
        double yaw_des1 = yaw;
        if (sqrt(ax*ax + ay*ay) > 0.01) {
            yaw_des1 = atan2(-ay, -ax);
        }
        
        double pitch_des2 = -pitch_des1;
        double yaw_des2 = yaw_des1 + M_PI;
        while(yaw_des2 > M_PI) yaw_des2 -= 2*M_PI;
        while(yaw_des2 < -M_PI) yaw_des2 += 2*M_PI;
        
        auto wrap = [](double a) {
            while(a > M_PI) a -= 2*M_PI;
            while(a < -M_PI) a += 2*M_PI;
            return a;
        };
        
        double cost1 = abs(wrap(yaw_des1 - yaw)) + abs(pitch_des1 - pitch);
        double cost2 = abs(wrap(yaw_des2 - yaw)) + abs(pitch_des2 - pitch);
        
        double final_pitch_des, final_yaw_des;
        if (cost1 < cost2) {
            final_pitch_des = pitch_des1;
            final_yaw_des = yaw_des1;
        } else {
            final_pitch_des = pitch_des2;
            final_yaw_des = yaw_des2;
        }
        
        double yaw_err = wrap(final_yaw_des - yaw);
        double yaw_cmd = 2.0 * yaw_err - 1.5 * yaw_rate;
        yaw_cmd = max(-1.0, min(1.0, yaw_cmd));
        
        double pitch_err = final_pitch_des - pitch;
        double pitch_cmd = 2.0 * pitch_err - 1.5 * pitch_rate;
        pitch_cmd = max(-1.0, min(1.0, pitch_cmd));
        
        double roll_err = 0 - roll;
        double roll_cmd = 2.0 * roll_err - 1.5 * roll_rate;
        roll_cmd = max(-1.0, min(1.0, roll_cmd));
        
        double a_mag = sqrt(ax*ax + ay*ay + az*az);
        double thrust_cmd = a_mag / max_thrust;
        
        double cx = -sin(pitch) * cos(yaw);
        double cy = -sin(pitch) * sin(yaw);
        double cz = cos(pitch);
        
        double dx_dir = 0, dy_dir = 0, dz_dir = 1;
        if (a_mag > 0.001) {
            dx_dir = ax / a_mag;
            dy_dir = ay / a_mag;
            dz_dir = az / a_mag;
        }
        
        double dot = cx*dx_dir + cy*dy_dir + cz*dz_dir;
        
        if (dot < 0.5) {
            thrust_cmd *= max(0.0, dot * 2.0);
        }
        if (dot < 0) thrust_cmd = 0;
        
        if (z < terrain.height) {
            double ground_z = 0;
            uint16_t a, b, c;
            terrain.get_cell((int)x, (int)y, a, b, c);
            if (z >= c) ground_z = c;
            else ground_z = a;
            
            if (z - ground_z < 20.0 && vz < -5.0 && cz > 0.5) {
                thrust_cmd = 1.0;
            }
        }
        
        thrust_cmd = max(0.0, min(1.0, thrust_cmd));
        
        cout << thrust_cmd << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();
    }
    
    return 0;
}
