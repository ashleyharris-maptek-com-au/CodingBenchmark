#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cstdint>

using namespace std;

// Fast I/O is crucial for large inputs
void setup_io() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
}

// O(N * M^2) simple DP/search for trivial and medium sizes
string solve_trivial(const vector<string>& strings) {
    int low = 1, high = (int)strings[0].length();
    string best = "";
    while (low <= high) {
        int mid = low + (high - low) / 2;
        string found = "";
        for (size_t i = 0; i <= strings[0].length() - mid; ++i) {
            string sub = strings[0].substr(i, mid);
            bool ok = true;
            for (size_t j = 1; j < strings.size(); ++j) {
                if (strings[j].find(sub) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                found = sub;
                break;
            }
        }
        if (found != "") {
            best = found;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    return best;
}

// 128-bit combined hash using 3 parallel hashes for collision avoidance
struct Hash128 {
    uint64_t h1;
    uint32_t h2, h3;
    bool operator==(const Hash128& o) const {
        return h1 == o.h1 && h2 == o.h2 && h3 == o.h3;
    }
    bool operator<(const Hash128& o) const {
        if (h1 != o.h1) return h1 < o.h1;
        if (h2 != o.h2) return h2 < o.h2;
        return h3 < o.h3;
    }
};

struct Candidate {
    Hash128 hash;
    uint32_t pos;
};

// Global static allocations to avoid repeating memory overhead
const uint32_t MASK = (1 << 29) - 1;
vector<uint32_t> bloom;
vector<Candidate> candidates;
vector<uint8_t> visited_cand;
vector<uint64_t> p1_pow, p2_pow, p3_pow;

const uint64_t P1 = 257;
const uint64_t P2 = 263;
const uint64_t M2 = 1000000007;
const uint64_t P3 = 269;
const uint64_t M3 = 1000000009;

void precompute(size_t max_len) {
    p1_pow.resize(max_len + 1);
    p2_pow.resize(max_len + 1);
    p3_pow.resize(max_len + 1);
    p1_pow[0] = 1; p2_pow[0] = 1; p3_pow[0] = 1;
    for(size_t i = 1; i <= max_len; ++i) {
        p1_pow[i] = p1_pow[i-1] * P1;
        p2_pow[i] = (p2_pow[i-1] * P2) % M2;
        p3_pow[i] = (p3_pow[i-1] * P3) % M3;
    }
}

// Highly optimized O(N log M) Binary Search + Rolling Hash for extreme scale
string solve_hash(const vector<string>& strings) {
    size_t shortest_idx = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].length() < strings[shortest_idx].length()) {
            shortest_idx = i;
        }
    }
    
    const string& shortest_s = strings[shortest_idx];
    size_t min_len = shortest_s.length();
    if (min_len == 0) return "";
    
    precompute(min_len);
    bloom.assign((MASK + 1) / 32, 0);
    visited_cand.resize(min_len + 1);
    candidates.reserve(min_len + 1);
    
    int low = 1, high = (int)min_len;
    int best_L = 0;
    string best_str = "";
    
    while(low <= high) {
        int L = low + (high - low) / 2;
        
        candidates.clear();
        uint64_t remove_term1[256], remove_term2[256], remove_term3[256];
        for(int c = 0; c < 256; ++c) {
            remove_term1[c] = c * p1_pow[L];
            remove_term2[c] = (c * p2_pow[L]) % M2;
            remove_term3[c] = (c * p3_pow[L]) % M3;
        }
        
        uint64_t h1 = 0, h2 = 0, h3 = 0;
        for(int i = 0; i < L; ++i) {
            uint8_t c = shortest_s[i];
            h1 = h1 * P1 + c;
            h2 = (h2 * P2 + c) % M2;
            h3 = (h3 * P3 + c) % M3;
        }
        candidates.push_back({{h1, (uint32_t)h2, (uint32_t)h3}, 0});
        
        for (size_t i = 1; i <= shortest_s.length() - L; ++i) {
            uint8_t old_c = shortest_s[i - 1];
            uint8_t new_c = shortest_s[i + L - 1];
            h1 = h1 * P1 + new_c - remove_term1[old_c];
            h2 = (h2 * P2 + new_c + M2 - remove_term2[old_c]) % M2;
            h3 = (h3 * P3 + new_c + M3 - remove_term3[old_c]) % M3;
            candidates.push_back({{h1, (uint32_t)h2, (uint32_t)h3}, (uint32_t)i});
        }
        
        sort(candidates.begin(), candidates.end(), [](const Candidate& a, const Candidate& b) {
            if (a.hash == b.hash) return a.pos < b.pos;
            return a.hash < b.hash;
        });
        
        auto it = unique(candidates.begin(), candidates.end(), [](const Candidate& a, const Candidate& b) {
            return a.hash == b.hash;
        });
        candidates.erase(it, candidates.end());
        
        fill(bloom.begin(), bloom.end(), 0);
        for(const auto& c : candidates) {
            uint32_t b = c.hash.h1 & MASK;
            bloom[b >> 5] |= (1u << (b & 31));
        }
        
        bool possible = true;
        for (size_t i = 0; i < strings.size(); ++i) {
            if (i == shortest_idx) continue;
            
            const string& s = strings[i];
            if (s.length() < (size_t)L) { possible = false; break; }
            
            fill(visited_cand.begin(), visited_cand.begin() + candidates.size(), 0);
            
            uint64_t cur_h1 = 0, cur_h2 = 0, cur_h3 = 0;
            for(int j = 0; j < L; ++j) {
                uint8_t c = s[j];
                cur_h1 = cur_h1 * P1 + c;
                cur_h2 = (cur_h2 * P2 + c) % M2;
                cur_h3 = (cur_h3 * P3 + c) % M3;
            }
            
            auto check_hash = [&](uint64_t _h1, uint64_t _h2, uint64_t _h3) {
                uint32_t b = _h1 & MASK;
                if (bloom[b >> 5] & (1u << (b & 31))) {
                    Hash128 dummy = {_h1, (uint32_t)_h2, (uint32_t)_h3};
                    Candidate cd; cd.hash = dummy; cd.pos = 0;
                    auto it2 = lower_bound(candidates.begin(), candidates.end(), cd, 
                        [](const Candidate& a, const Candidate& b) { return a.hash < b.hash; });
                    if (it2 != candidates.end() && it2->hash == dummy) {
                        visited_cand[it2 - candidates.begin()] = 1;
                    }
                }
            };
            
            check_hash(cur_h1, cur_h2, cur_h3);
            
            for (size_t j = 1; j <= s.length() - L; ++j) {
                uint8_t old_c = s[j - 1];
                uint8_t new_c = s[j + L - 1];
                cur_h1 = cur_h1 * P1 + new_c - remove_term1[old_c];
                cur_h2 = (cur_h2 * P2 + new_c + M2 - remove_term2[old_c]) % M2;
                cur_h3 = (cur_h3 * P3 + new_c + M3 - remove_term3[old_c]) % M3;
                
                check_hash(cur_h1, cur_h2, cur_h3);
            }
            
            size_t new_size = 0;
            for(size_t k = 0; k < candidates.size(); ++k) {
                if (visited_cand[k]) {
                    candidates[new_size++] = candidates[k];
                }
            }
            candidates.resize(new_size);
            if (candidates.empty()) {
                possible = false;
                break;
            }
            
            if (i + 1 < strings.size()) {
                fill(bloom.begin(), bloom.end(), 0);
                for(const auto& c : candidates) {
                    uint32_t b = c.hash.h1 & MASK;
                    bloom[b >> 5] |= (1u << (b & 31));
                }
            }
        }
        
        if (possible && !candidates.empty()) {
            best_L = L;
            int earliest_idx = -1;
            string current_best_str = "";
            
            // To be completely deterministic and return the lexicographically earliest 
            // occurrence relative to strings[0].
            for (const auto& c : candidates) {
                string sub = shortest_s.substr(c.pos, L);
                size_t idx = strings[0].find(sub);
                if (idx != string::npos) {
                    if (earliest_idx == -1 || (int)idx < earliest_idx) {
                        earliest_idx = (int)idx;
                        current_best_str = move(sub);
                    }
                }
            }
            best_str = current_best_str;
            low = L + 1;
        } else {
            high = L - 1;
        }
    }
    
    return best_str;
}

int main() {
    setup_io();
    int N;
    if (!(cin >> N)) return 0;
    string line;
    getline(cin, line); // consume newline
    
    vector<string> strings;
    strings.reserve(N);
    size_t total_len = 0;
    for (int i = 0; i < N; ++i) {
        getline(cin, line);
        if (!line.empty() && line.back() == '\r') line.pop_back();
        total_len += line.length();
        strings.push_back(move(line));
    }
    
    if (strings.empty()) return 0;
    
    // Adaptive logic threshold
    if (total_len <= 2000) {
        cout << solve_trivial(strings) << "\n";
    } else {
        cout << solve_hash(strings) << "\n";
    }
    
    return 0;
}
