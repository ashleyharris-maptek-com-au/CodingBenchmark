use std::collections::VecDeque;
use std::io::{self, BufRead, Write};

struct FastOut<W: Write> {
    inner: W,
    buf: Vec<u8>,
}

impl<W: Write> FastOut<W> {
    fn new(inner: W) -> Self {
        Self { inner, buf: Vec::with_capacity(1024 * 1024) }
    }
    
    #[inline]
    fn write_u32(&mut self, mut n: u32) {
        if n == 0 {
            self.push(b'0');
            return;
        }
        let mut tmp = [0u8; 10];
        let mut i = 10;
        while n > 0 {
            i -= 1;
            tmp[i] = (n % 10) as u8 + b'0';
            n /= 10;
        }
        self.extend_from_slice(&tmp[i..]);
    }
    
    #[inline]
    fn push(&mut self, b: u8) {
        if self.buf.len() >= self.buf.capacity() {
            self.flush_buf();
        }
        self.buf.push(b);
    }
    
    #[inline]
    fn extend_from_slice(&mut self, slice: &[u8]) {
        if self.buf.len() + slice.len() > self.buf.capacity() {
            self.flush_buf();
        }
        self.buf.extend_from_slice(slice);
    }
    
    #[inline]
    fn flush_buf(&mut self) {
        if !self.buf.is_empty() {
            let _ = self.inner.write_all(&self.buf);
            self.buf.clear();
        }
    }
}

impl<W: Write> Drop for FastOut<W> {
    fn drop(&mut self) {
        self.flush_buf();
    }
}

#[inline(always)]
fn pack(x: u32, y: u32) -> u32 {
    (x << 16) | y
}

#[inline(always)]
fn unpack(val: u32) -> (u32, u32) {
    (val >> 16, val & 0xFFFF)
}

fn main() {
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    
    let mut first_line = String::new();
    if stdin_lock.read_line(&mut first_line).is_err() { return; }
    
    let parts: Vec<usize> = first_line.trim().split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if parts.len() < 2 { return; }
    let h = parts[0];
    let w = parts[1];
    
    if h == 0 || w == 0 {
        return;
    }
    
    let h_u32 = h as u32;
    let w_u32 = w as u32;
    
    if h_u32 > 65535 || w_u32 > 65535 {
        panic!("Dimensions too large for u16 packing");
    }
    
    let mut maze = vec![0u8; h * w];
    let mut write_idx = 0;
    let mut start_pos = (0, 0);
    let mut end_pos = (0, 0);
    
    let mut x = 0;
    let mut y = 0;
    
    loop {
        let buf = stdin_lock.fill_buf().unwrap();
        let len = buf.len();
        if len == 0 { break; }
        
        for &b in buf {
            if b == b'\n' || b == b'\r' {
                continue;
            }
            if write_idx < maze.len() {
                if b == b'A' {
                    start_pos = (x, y);
                } else if b == b'B' {
                    end_pos = (x, y);
                }
                maze[write_idx] = b;
                write_idx += 1;
                
                x += 1;
                if x == w_u32 {
                    x = 0;
                    y += 1;
                }
            }
        }
        stdin_lock.consume(len);
    }
    
    let mut q: VecDeque<u32> = VecDeque::with_capacity(1024 * 1024);
    q.push_back(pack(start_pos.0, start_pos.1));
    
    let mut found = false;
    
    while let Some(val) = q.pop_front() {
        let (cx, cy) = unpack(val);
        if (cx, cy) == end_pos {
            found = true;
            break;
        }
        
        if cy > 0 {
            let nx = cx;
            let ny = cy - 1;
            let idx = (ny * w_u32 + nx) as usize;
            let b = maze[idx];
            if b == b' ' || b == b'B' {
                maze[idx] = b'D'; 
                q.push_back(pack(nx, ny));
                if b == b'B' { found = true; break; }
            }
        }
        if cy + 1 < h_u32 {
            let nx = cx;
            let ny = cy + 1;
            let idx = (ny * w_u32 + nx) as usize;
            let b = maze[idx];
            if b == b' ' || b == b'B' {
                maze[idx] = b'U';
                q.push_back(pack(nx, ny));
                if b == b'B' { found = true; break; }
            }
        }
        if cx > 0 {
            let nx = cx - 1;
            let ny = cy;
            let idx = (ny * w_u32 + nx) as usize;
            let b = maze[idx];
            if b == b' ' || b == b'B' {
                maze[idx] = b'R';
                q.push_back(pack(nx, ny));
                if b == b'B' { found = true; break; }
            }
        }
        if cx + 1 < w_u32 {
            let nx = cx + 1;
            let ny = cy;
            let idx = (ny * w_u32 + nx) as usize;
            let b = maze[idx];
            if b == b' ' || b == b'B' {
                maze[idx] = b'L';
                q.push_back(pack(nx, ny));
                if b == b'B' { found = true; break; }
            }
        }
    }
    
    if !found {
        println!("0");
        return;
    }
    
    let mut curr = end_pos;
    let mut next_forward_dir = b'E'; 
    
    while curr != start_pos {
        let (cx, cy) = curr;
        let idx = (cy * w_u32 + cx) as usize;
        let b = maze[idx];
        
        maze[idx] = next_forward_dir;
        
        let parent = match b {
            b'U' => { next_forward_dir = b'D'; (cx, cy - 1) },
            b'D' => { next_forward_dir = b'U'; (cx, cy + 1) },
            b'L' => { next_forward_dir = b'R'; (cx - 1, cy) },
            b'R' => { next_forward_dir = b'L'; (cx + 1, cy) },
            _ => panic!("invalid direction code during backtrace"),
        };
        
        curr = parent;
    }
    
    let mut len = 1;
    let mut count_curr = start_pos;
    let mut count_dir = next_forward_dir;
    
    while count_curr != end_pos {
        len += 1;
        count_curr = match count_dir {
            b'U' => (count_curr.0, count_curr.1 - 1),
            b'D' => (count_curr.0, count_curr.1 + 1),
            b'L' => (count_curr.0 - 1, count_curr.1),
            b'R' => (count_curr.0 + 1, count_curr.1),
            _ => unreachable!(),
        };
        if count_curr != end_pos {
            count_dir = maze[(count_curr.1 * w_u32 + count_curr.0) as usize];
        }
    }
    
    let stdout = io::stdout();
    let mut fast_out = FastOut::new(stdout.lock());
    
    fast_out.write_u32(len);
    fast_out.push(b'\n');
    
    let mut fcurr = start_pos;
    let mut fdir = next_forward_dir;
    
    fast_out.write_u32(fcurr.0);
    fast_out.push(b' ');
    fast_out.write_u32(fcurr.1);
    fast_out.push(b'\n');
    
    while fcurr != end_pos {
        fcurr = match fdir {
            b'U' => (fcurr.0, fcurr.1 - 1),
            b'D' => (fcurr.0, fcurr.1 + 1),
            b'L' => (fcurr.0 - 1, fcurr.1),
            b'R' => (fcurr.0 + 1, fcurr.1),
            _ => unreachable!(),
        };
        
        fast_out.write_u32(fcurr.0);
        fast_out.push(b' ');
        fast_out.write_u32(fcurr.1);
        fast_out.push(b'\n');
        
        if fcurr != end_pos {
            fdir = maze[(fcurr.1 * w_u32 + fcurr.0) as usize];
        }
    }
}
