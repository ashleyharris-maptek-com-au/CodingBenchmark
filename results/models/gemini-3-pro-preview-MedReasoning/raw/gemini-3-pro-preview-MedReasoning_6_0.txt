using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;

class Program
{
    static void Main(string[] args)
    {
        string line1 = ReadNextLine();
        if (line1 == null) return;
        var parts = line1.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;

        int N = int.Parse(parts[0]);
        double mu = double.Parse(parts[1], CultureInfo.InvariantCulture);

        string startLine = ReadNextLine();
        if (startLine == null) return;
        double[] startSv = ParseLine(startLine);
        Orbit startOrbit = new Orbit(0, startSv, mu);

        Orbit[] orbits = new Orbit[N + 1];
        orbits[0] = startOrbit;
        for (int i = 0; i < N; i++)
        {
            string stLine = ReadNextLine();
            if (stLine == null) break;
            orbits[i + 1] = new Orbit(i + 1, ParseLine(stLine), mu);
        }

        double[,] dist = new double[N + 1, N + 1];
        for (int i = 0; i <= N; i++)
        {
            for (int j = 0; j <= N; j++)
            {
                if (i == j) dist[i, j] = 0;
                else dist[i, j] = GetCost(orbits[i], orbits[j], mu);
            }
        }

        int[] bestPath = null;
        double bestCost = double.MaxValue;
        Random rnd = new Random(42);

        Stopwatch sw = Stopwatch.StartNew();
        bool first = true;
        int maxIterations = N <= 10 ? 10000 : 200000;
        int iter = 0;

        while (sw.ElapsedMilliseconds < 4000 && iter < maxIterations)
        {
            iter++;
            int[] path = new int[N + 1];
            bool[] visited = new bool[N + 1];
            path[0] = 0;
            visited[0] = true;

            for (int step = 1; step <= N; step++)
            {
                int curr = path[step - 1];
                List<int> cands = new List<int>();
                for (int j = 1; j <= N; j++)
                {
                    if (!visited[j]) cands.Add(j);
                }

                int bestNext = -1;
                if (first)
                {
                    double minD = double.MaxValue;
                    foreach (int c in cands)
                    {
                        if (dist[curr, c] < minD)
                        {
                            minD = dist[curr, c];
                            bestNext = c;
                        }
                    }
                }
                else
                {
                    cands.Sort((a, b) => dist[curr, a].CompareTo(dist[curr, b]));
                    int topK = Math.Max(1, Math.Min(4, cands.Count));
                    if (cands.Count > 4 && rnd.NextDouble() < 0.1) topK = Math.Min(8, cands.Count);
                    bestNext = cands[rnd.Next(topK)];
                }

                path[step] = bestNext;
                visited[bestNext] = true;
            }

            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 1; i < N; i++)
                {
                    for (int j = i + 1; j <= N; j++)
                    {
                        double currentCost = dist[path[i - 1], path[i]];
                        if (j < N) currentCost += dist[path[j], path[j + 1]];

                        double newCost = dist[path[i - 1], path[j]];
                        if (j < N) newCost += dist[path[i], path[j + 1]];

                        if (newCost < currentCost - 1e-9)
                        {
                            int l = i, r = j;
                            while (l < r)
                            {
                                int temp = path[l];
                                path[l] = path[r];
                                path[r] = temp;
                                l++; r--;
                            }
                            improved = true;
                        }
                    }
                }
            }

            double totalCost = 0;
            for (int i = 0; i < N; i++) totalCost += dist[path[i], path[i + 1]];

            if (totalCost < bestCost)
            {
                bestCost = totalCost;
                bestPath = (int[])path.Clone();
            }

            first = false;
        }

        List<int> result = new List<int>();
        for (int i = 1; i <= N; i++)
        {
            result.Add(bestPath[i] - 1);
        }
        Console.WriteLine(string.Join(" ", result));
    }

    static string ReadNextLine()
    {
        while (true)
        {
            string line = Console.ReadLine();
            if (line == null) return null;
            if (!string.IsNullOrWhiteSpace(line)) return line;
        }
    }

    static double[] ParseLine(string line)
    {
        if (string.IsNullOrWhiteSpace(line)) return new double[6];
        var parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] res = new double[6];
        for (int i = 0; i < Math.Min(6, parts.Length); i++)
        {
            double.TryParse(parts[i], NumberStyles.Any, CultureInfo.InvariantCulture, out res[i]);
        }
        return res;
    }

    static double GetCost(Orbit o1, Orbit o2, double mu)
    {
        if (o1.a <= 0 || o2.a <= 0 || o1.h < 1e-6 || o2.h < 1e-6)
        {
            double dx = o1.x - o2.x; double dy = o1.y - o2.y; double dz = o1.z - o2.z;
            double dvx = o1.vx - o2.vx; double dvy = o1.vy - o2.vy; double dvz = o1.vz - o2.vz;
            return Math.Sqrt(dx * dx + dy * dy + dz * dz) * 0.001 + Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);
        }

        double v_ref = Math.Sqrt(mu / o1.a);

        double dv_a = Math.Abs(Math.Sqrt(mu / o1.a) - Math.Sqrt(mu / o2.a));

        double cos_di = (o1.hx * o2.hx + o1.hy * o2.hy + o1.hz * o2.hz) / (o1.h * o2.h);
        cos_di = Math.Max(-1.0, Math.Min(1.0, cos_di));
        double di = Math.Acos(cos_di);
        double dv_i = 2 * v_ref * Math.Sin(di / 2);

        double dex = o1.ex - o2.ex;
        double dey = o1.ey - o2.ey;
        double dez = o1.ez - o2.ez;
        double de = Math.Sqrt(dex * dex + dey * dey + dez * dez);
        double dv_e = 0.5 * v_ref * de;

        double dv_phase = Math.Sqrt(Math.Pow(o1.vx - o2.vx, 2) + Math.Pow(o1.vy - o2.vy, 2) + Math.Pow(o1.vz - o2.vz, 2));

        return dv_a + dv_i + dv_e + 0.05 * dv_phase;
    }
}

class Orbit
{
    public int id;
    public double x, y, z, vx, vy, vz;
    public double a, e;
    public double hx, hy, hz, h;
    public double ex, ey, ez;

    public Orbit(int id, double[] sv, double mu)
    {
        this.id = id;
        x = sv[0]; y = sv[1]; z = sv[2];
        vx = sv[3]; vy = sv[4]; vz = sv[5];

        double r = Math.Sqrt(x * x + y * y + z * z);
        double v2 = vx * vx + vy * vy + vz * vz;

        hx = y * vz - z * vy;
        hy = z * vx - x * vz;
        hz = x * vy - y * vx;
        h = Math.Sqrt(hx * hx + hy * hy + hz * hz);

        double r_dot_v = x * vx + y * vy + z * vz;
        ex = ((v2 - mu / r) * x - r_dot_v * vx) / mu;
        ey = ((v2 - mu / r) * y - r_dot_v * vy) / mu;
        ez = ((v2 - mu / r) * z - r_dot_v * vz) / mu;
        e = Math.Sqrt(ex * ex + ey * ey + ez * ez);

        double E = v2 / 2 - mu / r;
        if (Math.Abs(E) > 1e-9)
        {
            a = -mu / (2 * E);
        }
        else
        {
            a = -1;
        }
    }
}
