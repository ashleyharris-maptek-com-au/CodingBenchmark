use std::io::{self, BufRead};

const EPSILON: f64 = 1e-7;

#[derive(Clone, Debug)]
struct Point { x: f64, y: f64 }

#[derive(Clone, Debug)]
struct AABB {
    min_x: f64, max_x: f64,
    min_y: f64, max_y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
    aabb: AABB,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        for v in &vertices {
            if v.x < min_x { min_x = v.x; }
            if v.x > max_x { max_x = v.x; }
            if v.y < min_y { min_y = v.y; }
            if v.y > max_y { max_y = v.y; }
        }
        Polygon {
            vertices,
            aabb: AABB { min_x, max_x, min_y, max_y },
        }
    }
}

fn polygon_area(poly: &Polygon) -> f64 {
    let mut area = 0.0;
    let n = poly.vertices.len();
    if n < 3 { return 0.0; }
    for i in 0..n {
        let j = (i + 1) % n;
        area += poly.vertices[i].x * poly.vertices[j].y;
        area -= poly.vertices[j].x * poly.vertices[i].y;
    }
    (area / 2.0).abs()
}

fn polygon_centroid(poly: &Polygon) -> Point {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut area = 0.0;
    let n = poly.vertices.len();
    if n < 3 {
        return Point { x: poly.vertices[0].x, y: poly.vertices[0].y };
    }
    for i in 0..n {
        let j = (i + 1) % n;
        let cross = poly.vertices[i].x * poly.vertices[j].y - poly.vertices[j].x * poly.vertices[i].y;
        cx += (poly.vertices[i].x + poly.vertices[j].x) * cross;
        cy += (poly.vertices[i].y + poly.vertices[j].y) * cross;
        area += cross;
    }
    area /= 2.0;
    if area.abs() < EPSILON {
        return Point { x: poly.vertices[0].x, y: poly.vertices[0].y };
    }
    cx /= 6.0 * area;
    cy /= 6.0 * area;
    Point { x: cx, y: cy }
}

fn aabb_intersect(a: &AABB, b: &AABB) -> bool {
    if a.max_x - b.min_x <= EPSILON || b.max_x - a.min_x <= EPSILON { return false; }
    if a.max_y - b.min_y <= EPSILON || b.max_y - a.min_y <= EPSILON { return false; }
    true
}

fn aabb_inside(a: &AABB, b: &AABB) -> bool {
    a.min_x >= b.min_x - EPSILON && a.max_x <= b.max_x + EPSILON &&
    a.min_y >= b.min_y - EPSILON && a.max_y <= b.max_y + EPSILON
}

fn cross_product(p1: &Point, p2: &Point, p3: &Point) -> f64 {
    (p2.x - p1.x) * (p3.y - p1.y) - (p2.y - p1.y) * (p3.x - p1.x)
}

fn on_segment(p: &Point, q: &Point, r: &Point) -> bool {
    q.x <= p.x.max(r.x) + EPSILON && q.x >= p.x.min(r.x) - EPSILON &&
    q.y <= p.y.max(r.y) + EPSILON && q.y >= p.y.min(r.y) - EPSILON
}

fn point_on_segment_exact(p: &Point, a: &Point, b: &Point) -> bool {
    let cross = cross_product(a, b, p);
    if cross.abs() > EPSILON { return false; }
    on_segment(a, p, b)
}

fn segments_intersect(p1: &Point, q1: &Point, p2: &Point, q2: &Point) -> bool {
    let cp1 = cross_product(p1, q1, p2);
    let cp2 = cross_product(p1, q1, q2);
    let cp3 = cross_product(p2, q2, p1);
    let cp4 = cross_product(p2, q2, q1);

    if ((cp1 > EPSILON && cp2 < -EPSILON) || (cp1 < -EPSILON && cp2 > EPSILON)) &&
       ((cp3 > EPSILON && cp4 < -EPSILON) || (cp3 < -EPSILON && cp4 > EPSILON)) {
        return true;
    }
    false
}

fn point_in_polygon_inclusive(p: &Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();
    for i in 0..n {
        let vi = &poly.vertices[i];
        let vj = &poly.vertices[(i + 1) % n];
        
        if point_on_segment_exact(p, vi, vj) {
            return true; 
        }

        if (vi.y > p.y) != (vj.y > p.y) {
            let intersect_x = vi.x + (p.y - vi.y) * (vj.x - vi.x) / (vj.y - vi.y);
            if p.x < intersect_x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_in_polygon_strict(p: &Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();
    for i in 0..n {
        let vi = &poly.vertices[i];
        let vj = &poly.vertices[(i + 1) % n];
        
        if point_on_segment_exact(p, vi, vj) {
            return false;
        }

        if (vi.y > p.y) != (vj.y > p.y) {
            let intersect_x = vi.x + (p.y - vi.y) * (vj.x - vi.x) / (vj.y - vi.y);
            if p.x < intersect_x {
                inside = !inside;
            }
        }
    }
    inside
}

fn polygon_inside(a: &Polygon, s: &Polygon) -> bool {
    if !aabb_inside(&a.aabb, &s.aabb) { return false; }
    
    let na = a.vertices.len();
    let ns = s.vertices.len();
    for i in 0..na {
        for j in 0..ns {
            if segments_intersect(&a.vertices[i], &a.vertices[(i+1)%na],
                                  &s.vertices[j], &s.vertices[(j+1)%ns]) {
                return false;
            }
        }
    }
    
    for v in &a.vertices {
        if !point_in_polygon_inclusive(v, s) { return false; }
    }
    true
}

fn polygons_overlap(a: &Polygon, b: &Polygon) -> bool {
    if !aabb_intersect(&a.aabb, &b.aabb) { return false; }
    
    let na = a.vertices.len();
    let nb = b.vertices.len();
    for i in 0..na {
        for j in 0..nb {
            if segments_intersect(&a.vertices[i], &a.vertices[(i+1)%na],
                                  &b.vertices[j], &b.vertices[(j+1)%nb]) {
                return true;
            }
        }
    }
    
    for v in &a.vertices {
        if point_in_polygon_strict(v, b) { return true; }
    }
    for v in &b.vertices {
        if point_in_polygon_strict(v, a) { return true; }
    }
    
    let centroid_a = polygon_centroid(a);
    if point_in_polygon_strict(&centroid_a, b) { return true; }
    let centroid_b = polygon_centroid(b);
    if point_in_polygon_strict(&centroid_b, a) { return true; }
    
    false
}

fn transform_polygon(poly: &Polygon, tx: f64, ty: f64, rot: f64) -> Polygon {
    let cos_r = rot.cos();
    let sin_r = rot.sin();
    
    let mut new_verts = Vec::with_capacity(poly.vertices.len());
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    
    for v in &poly.vertices {
        let nx = v.x * cos_r - v.y * sin_r + tx;
        let ny = v.x * sin_r + v.y * cos_r + ty;
        if nx < min_x { min_x = nx; }
        if nx > max_x { max_x = nx; }
        if ny < min_y { min_y = ny; }
        if ny > max_y { max_y = ny; }
        new_verts.push(Point { x: nx, y: ny });
    }
    
    Polygon {
        vertices: new_verts,
        aabb: AABB { min_x, max_x, min_y, max_y },
    }
}

fn update_translated_polygon(src: &Polygon, dest: &mut Polygon, tx: f64, ty: f64) {
    for (i, v) in src.vertices.iter().enumerate() {
        dest.vertices[i].x = v.x + tx;
        dest.vertices[i].y = v.y + ty;
    }
    dest.aabb.min_x = src.aabb.min_x + tx;
    dest.aabb.max_x = src.aabb.max_x + tx;
    dest.aabb.min_y = src.aabb.min_y + ty;
    dest.aabb.max_y = src.aabb.max_y + ty;
}

struct SpatialGrid {
    cell_size_x: f64,
    cell_size_y: f64,
    cols: usize,
    rows: usize,
    min_x: f64,
    min_y: f64,
    cells: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(aabb: &AABB, grid_size: usize) -> Self {
        let w = aabb.max_x - aabb.min_x;
        let h = aabb.max_y - aabb.min_y;
        let w = if w <= EPSILON { 1.0 } else { w };
        let h = if h <= EPSILON { 1.0 } else { h };
        
        SpatialGrid {
            cell_size_x: w / grid_size as f64,
            cell_size_y: h / grid_size as f64,
            cols: grid_size,
            rows: grid_size,
            min_x: aabb.min_x,
            min_y: aabb.min_y,
            cells: vec![Vec::new(); grid_size * grid_size],
        }
    }
    
    fn get_cell_range(&self, aabb: &AABB) -> (usize, usize, usize, usize) {
        let min_col = ((aabb.min_x - self.min_x) / self.cell_size_x).floor() as isize;
        let max_col = ((aabb.max_x - self.min_x) / self.cell_size_x).floor() as isize;
        let min_row = ((aabb.min_y - self.min_y) / self.cell_size_y).floor() as isize;
        let max_row = ((aabb.max_y - self.min_y) / self.cell_size_y).floor() as isize;
        
        (
            min_col.clamp(0, self.cols as isize - 1) as usize,
            max_col.clamp(0, self.cols as isize - 1) as usize,
            min_row.clamp(0, self.rows as isize - 1) as usize,
            max_row.clamp(0, self.rows as isize - 1) as usize,
        )
    }
    
    fn insert(&mut self, piece_idx: usize, aabb: &AABB) {
        let (min_c, max_c, min_r, max_r) = self.get_cell_range(aabb);
        for r in min_r..=max_r {
            for c in min_c..=max_c {
                self.cells[r * self.cols + c].push(piece_idx);
            }
        }
    }
    
    fn get_candidates(&self, aabb: &AABB) -> Vec<usize> {
        let (min_c, max_c, min_r, max_r) = self.get_cell_range(aabb);
        let mut cands = Vec::new();
        for r in min_r..=max_r {
            for c in min_c..=max_c {
                cands.extend_from_slice(&self.cells[r * self.cols + c]);
            }
        }
        cands.sort_unstable();
        cands.dedup();
        cands
    }
}

struct Stock {
    poly: Polygon,
    placed_pieces: Vec<Polygon>,
    grid: SpatialGrid,
}

impl Stock {
    fn new(poly: Polygon) -> Self {
        let grid = SpatialGrid::new(&poly.aabb, 20);
        Stock {
            poly,
            placed_pieces: Vec::new(),
            grid,
        }
    }
    
    fn can_place(&self, cand: &Polygon) -> bool {
        if !polygon_inside(cand, &self.poly) { return false; }
        
        let cands = self.grid.get_candidates(&cand.aabb);
        for &idx in &cands {
            if polygons_overlap(cand, &self.placed_pieces[idx]) {
                return false;
            }
        }
        true
    }
    
    fn place(&mut self, cand: Polygon) {
        let idx = self.placed_pieces.len();
        self.grid.insert(idx, &cand.aabb);
        self.placed_pieces.push(cand);
    }
}

#[derive(Clone)]
struct Output {
    stock_idx: usize,
    tx: f64,
    ty: f64,
    rot: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());
    
    let first_line = lines.next().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();
    
    let mut stock_verts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        stock_verts.push(Point { x, y });
    }
    let stock_poly = Polygon::new(stock_verts);
    
    let mut pieces = Vec::with_capacity(np);
    for i in 0..np {
        let line = lines.next().unwrap();
        let pv: usize = line.trim().parse().unwrap();
        let mut verts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let parts: Vec<&str> = line.split_whitespace().collect();
            let x: f64 = parts[0].parse().unwrap();
            let y: f64 = parts[1].parse().unwrap();
            verts.push(Point { x, y });
        }
        pieces.push((i, Polygon::new(verts)));
    }
    
    pieces.sort_by(|a, b| polygon_area(&b.1).partial_cmp(&polygon_area(&a.1)).unwrap());
    
    let mut stocks: Vec<Stock> = vec![Stock::new(stock_poly.clone())];
    let mut outputs = vec![None; np];
    
    let grid_res = if np < 100 { 80 } else if np < 1000 { 50 } else { 30 };
    let stock_w = stock_poly.aabb.max_x - stock_poly.aabb.min_x;
    let stock_h = stock_poly.aabb.max_y - stock_poly.aabb.min_y;
    let step_size = (stock_w.max(stock_h)) / (grid_res as f64);
    let step_size = step_size.max(EPSILON);
    
    let angles = [0.0, std::f64::consts::PI / 2.0, std::f64::consts::PI, 3.0 * std::f64::consts::PI / 2.0];
    
    for (orig_idx, piece) in pieces {
        let mut placed = false;
        
        for stock_idx in 0..stocks.len() {
            let stock = &stocks[stock_idx];
            let mut best_place = None;
            
            for &rot in &angles {
                let rotated_piece = transform_polygon(&piece, 0.0, 0.0, rot);
                let start_x = stock.poly.aabb.min_x - rotated_piece.aabb.min_x;
                let end_x = stock.poly.aabb.max_x - rotated_piece.aabb.max_x;
                let start_y = stock.poly.aabb.min_y - rotated_piece.aabb.min_y;
                let end_y = stock.poly.aabb.max_y - rotated_piece.aabb.max_y;

                if end_x < start_x - EPSILON || end_y < start_y - EPSILON {
                    continue;
                }
                
                let res_x = (((end_x - start_x) / step_size).ceil() as usize + 1).max(2);
                let res_y = (((end_y - start_y) / step_size).ceil() as usize + 1).max(2);
                
                let mut cand = rotated_piece.clone();
                let mut found = false;
                
                for i in 0..res_y {
                    let ty = start_y + (i as f64) * (end_y - start_y) / ((res_y - 1) as f64);
                    for j in 0..res_x {
                        let tx = start_x + (j as f64) * (end_x - start_x) / ((res_x - 1) as f64);
                        
                        update_translated_polygon(&rotated_piece, &mut cand, tx, ty);
                        if stock.can_place(&cand) {
                            best_place = Some((stock_idx, tx, ty, rot, cand.clone()));
                            found = true;
                            break;
                        }
                    }
                    if found { break; }
                }
                if found { break; }
            }
            
            if let Some((s_idx, tx, ty, rot, final_poly)) = best_place {
                stocks[s_idx].place(final_poly);
                outputs[orig_idx] = Some(Output { stock_idx: s_idx, tx, ty, rot });
                placed = true;
                break;
            }
        }
        
        if !placed {
            let mut new_stock = Stock::new(stock_poly.clone());
            let stock_idx = stocks.len();
            
            let mut best_place = None;
            for &rot in &angles {
                let rotated_piece = transform_polygon(&piece, 0.0, 0.0, rot);
                let start_x = new_stock.poly.aabb.min_x - rotated_piece.aabb.min_x;
                let end_x = new_stock.poly.aabb.max_x - rotated_piece.aabb.max_x;
                let start_y = new_stock.poly.aabb.min_y - rotated_piece.aabb.min_y;
                let end_y = new_stock.poly.aabb.max_y - rotated_piece.aabb.max_y;

                if end_x < start_x - EPSILON || end_y < start_y - EPSILON {
                    continue;
                }
                
                let res_x = (((end_x - start_x) / step_size).ceil() as usize + 1).max(2);
                let res_y = (((end_y - start_y) / step_size).ceil() as usize + 1).max(2);
                
                let mut cand = rotated_piece.clone();
                let mut found = false;
                
                for i in 0..res_y {
                    let ty = start_y + (i as f64) * (end_y - start_y) / ((res_y - 1) as f64);
                    for j in 0..res_x {
                        let tx = start_x + (j as f64) * (end_x - start_x) / ((res_x - 1) as f64);
                        
                        update_translated_polygon(&rotated_piece, &mut cand, tx, ty);
                        if new_stock.can_place(&cand) {
                            best_place = Some((stock_idx, tx, ty, rot, cand.clone()));
                            found = true;
                            break;
                        }
                    }
                    if found { break; }
                }
                
                if !found {
                    let dense_res_x = 100;
                    let dense_res_y = 100;
                    for i in 0..dense_res_y {
                        let ty = start_y + (i as f64) * (end_y - start_y) / ((dense_res_y - 1) as f64);
                        for j in 0..dense_res_x {
                            let tx = start_x + (j as f64) * (end_x - start_x) / ((dense_res_x - 1) as f64);
                            update_translated_polygon(&rotated_piece, &mut cand, tx, ty);
                            if new_stock.can_place(&cand) {
                                best_place = Some((stock_idx, tx, ty, rot, cand.clone()));
                                found = true;
                                break;
                            }
                        }
                        if found { break; }
                    }
                }
                
                if found { break; }
            }
            
            if let Some((s_idx, tx, ty, rot, final_poly)) = best_place {
                new_stock.place(final_poly);
                stocks.push(new_stock);
                outputs[orig_idx] = Some(Output { stock_idx: s_idx, tx, ty, rot });
            } else {
                outputs[orig_idx] = Some(Output { stock_idx, tx: 0.0, ty: 0.0, rot: 0.0 });
                stocks.push(new_stock);
            }
        }
    }
    
    println!("{}", stocks.len());
    for out in outputs {
        let o = out.unwrap();
        println!("{} {:.6} {:.6} {:.6}", o.stock_idx, o.tx, o.ty, o.rot);
    }
}
