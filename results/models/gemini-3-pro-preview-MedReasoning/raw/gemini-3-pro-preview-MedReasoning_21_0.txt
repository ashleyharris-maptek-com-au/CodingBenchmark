#include <iostream>
#include <vector>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <deque>
#include <cmath>
#include <cstring>
#include <algorithm>

int D;
int max_turns;

struct Point {
    int16_t c[24];
    Point() { memset(c, 0, sizeof(c)); }
    bool operator==(const Point& o) const {
        for(int i = 0; i < D; ++i) if(c[i] != o.c[i]) return false;
        return true;
    }
};

struct PointHash {
    size_t operator()(const Point& p) const {
        size_t hash = 14695981039346656037ULL;
        for (int i = 0; i < D; ++i) {
            hash ^= (p.c[i] & 0xFF);
            hash *= 1099511628211ULL;
            hash ^= ((p.c[i] >> 8) & 0xFF);
            hash *= 1099511628211ULL;
        }
        return hash;
    }
};

struct Node {
    Point p;
    int g;
    int f;
    int id;
    int parent_id;
    int move_axis;
    int move_dir;
};

bool find_path(const Point& start, 
               const std::vector<Point>& foods, 
               const std::deque<Point>& old_snake, 
               const std::unordered_set<Point, PointHash>& obstacles,
               const std::vector<int>& bounds,
               std::vector<std::pair<int, int>>& out_moves,
               Point& out_food_eaten) {
               
    std::unordered_map<Point, int, PointHash> min_g;
    
    struct PQElement {
        int f, g, id;
        bool operator<(const PQElement& o) const {
            if (f != o.f) return f > o.f;
            return g < o.g;
        }
    };
    std::priority_queue<PQElement> pq;
    std::vector<Node> nodes;
    
    nodes.reserve(100000); 
    
    auto get_h = [&](const Point& p) {
        int min_h = 1e9;
        for (const auto& f : foods) {
            int h = 0;
            for (int i = 0; i < D; ++i) {
                h += std::abs(p.c[i] - f.c[i]);
            }
            if (h < min_h) min_h = h;
        }
        return min_h;
    };
    
    nodes.push_back({start, 0, get_h(start), 0, -1, -1, 0});
    pq.push({nodes[0].f, 0, 0});
    min_g[start] = 0;
    
    int L = old_snake.size();
    std::unordered_map<Point, int, PointHash> old_snake_idx;
    for (int i = 0; i < L; ++i) {
        if (old_snake_idx.find(old_snake[i]) == old_snake_idx.end()) {
            old_snake_idx[old_snake[i]] = i; 
        }
    }
    
    while (!pq.empty()) {
        auto top = pq.top();
        pq.pop();
        
        Node curr = nodes[top.id];
        
        if (curr.g > min_g[curr.p]) continue;
        
        bool is_food = false;
        for (const auto& f : foods) {
            if (curr.p == f) {
                is_food = true;
                out_food_eaten = f;
                break;
            }
        }
        
        if (is_food) {
            int curr_id = top.id;
            while (nodes[curr_id].parent_id != -1) {
                out_moves.push_back({nodes[curr_id].move_axis, nodes[curr_id].move_dir});
                curr_id = nodes[curr_id].parent_id;
            }
            std::reverse(out_moves.begin(), out_moves.end());
            return true;
        }
        
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Point nxt = curr.p;
                nxt.c[axis] += dir;
                
                if (nxt.c[axis] < 0 || nxt.c[axis] >= bounds[axis]) continue;
                if (obstacles.count(nxt)) continue;
                
                int t = curr.g + 1;
                
                if (min_g.count(nxt) && min_g[nxt] <= t) continue;
                
                int remaining_old = L - 1 - t;
                if (remaining_old >= 0) {
                    auto it = old_snake_idx.find(nxt);
                    if (it != old_snake_idx.end() && it->second <= remaining_old) {
                        continue;
                    }
                }
                
                min_g[nxt] = t;
                int new_id = nodes.size();
                int h = get_h(nxt);
                nodes.push_back({nxt, t, t + h, new_id, top.id, axis, dir});
                pq.push({t + h, t, new_id});
            }
        }
    }
    
    return false;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    if (!(std::cin >> D >> max_turns)) return 0;
    
    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];
    
    int snake_length;
    std::cin >> snake_length;
    
    Point start_pos;
    for (int i = 0; i < D; ++i) {
        int val;
        std::cin >> val;
        start_pos.c[i] = val;
    }
    
    int F;
    std::cin >> F;
    std::vector<Point> foods;
    for (int i = 0; i < F; ++i) {
        Point p;
        for (int j = 0; j < D; ++j) {
            int val;
            std::cin >> val;
            p.c[j] = val;
        }
        foods.push_back(p);
    }
    
    int O;
    std::cin >> O;
    std::unordered_set<Point, PointHash> obstacles;
    for (int i = 0; i < O; ++i) {
        Point p;
        for (int j = 0; j < D; ++j) {
            int val;
            std::cin >> val;
            p.c[j] = val;
        }
        obstacles.insert(p);
    }
    
    std::deque<Point> snake_body;
    for (int i = 0; i < snake_length; ++i) {
        snake_body.push_back(start_pos);
    }
    
    std::vector<std::pair<int, int>> all_moves;
    all_moves.reserve(max_turns);
    
    while (!foods.empty() && (int)all_moves.size() < max_turns) {
        std::vector<std::pair<int, int>> out_moves;
        Point eaten;
        
        bool found = find_path(snake_body.front(), foods, snake_body, obstacles, bounds, out_moves, eaten);
        
        if (found) {
            for (size_t i = 0; i < out_moves.size(); ++i) {
                if ((int)all_moves.size() >= max_turns) break;
                
                int axis = out_moves[i].first;
                int dir = out_moves[i].second;
                Point new_head = snake_body.front();
                new_head.c[axis] += dir;
                
                snake_body.push_front(new_head);
                if (i == out_moves.size() - 1) {
                    // Ate food, do not pop tail
                } else {
                    snake_body.pop_back();
                }
                
                all_moves.push_back({axis, dir});
            }
            
            auto it = std::find(foods.begin(), foods.end(), eaten);
            if (it != foods.end()) {
                foods.erase(it);
            }
            
        } else {
            // Fallback: try to make one valid move
            bool fallback = false;
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    Point nxt = snake_body.front();
                    nxt.c[axis] += dir;
                    
                    if (nxt.c[axis] < 0 || nxt.c[axis] >= bounds[axis]) continue;
                    if (obstacles.count(nxt)) continue;
                    
                    bool hit = false;
                    for (size_t i = 0; i < snake_body.size() - 1; ++i) {
                        if (snake_body[i] == nxt) {
                            hit = true; break;
                        }
                    }
                    if (!hit) {
                        snake_body.push_front(nxt);
                        snake_body.pop_back();
                        all_moves.push_back({axis, dir});
                        fallback = true;
                        break;
                    }
                }
                if (fallback) break;
            }
            
            if (!fallback) {
                break;
            }
        }
    }
    
    for (const auto& m : all_moves) {
        std::cout << m.first << " " << m.second << "\n";
    }
    
    return 0;
}
