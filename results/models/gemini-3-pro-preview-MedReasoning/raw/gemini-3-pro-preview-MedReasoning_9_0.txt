using System;
using System.IO;

class Program
{
    static void Main()
    {
        using var stream = Console.OpenStandardInput();
        var scanner = new Scanner(stream);
        
        long nLong = scanner.NextLong();
        if (nLong <= 0) return;
        int N = (int)nLong;
        long stockLength = scanner.NextLong();
        
        var cuts = new Cut[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = new Cut { Length = scanner.NextLong(), Id = i };
        }
        
        // Sort cuts in descending order for First Fit Decreasing (FFD) heuristic
        Array.Sort(cuts);
        
        // Segment tree to find the leftmost bin with enough capacity in O(log N)
        long[] tree = new long[4 * N];
        Build(tree, 1, 0, N - 1, stockLength);
        
        // Linked list arrays to store cuts assigned to each bin without object allocations
        int[] head = new int[N];
        int[] tail = new int[N];
        int[] next = new int[N];
        Array.Fill(head, -1);
        
        int maxBinUsed = -1;
        
        for (int i = 0; i < N; i++)
        {
            int binIndex = FindAndReduce(tree, 1, 0, N - 1, cuts[i].Length);
            if (binIndex > maxBinUsed)
            {
                maxBinUsed = binIndex;
            }
            
            int id = cuts[i].Id;
            next[id] = -1;
            
            if (head[binIndex] == -1)
            {
                head[binIndex] = id;
                tail[binIndex] = id;
            }
            else
            {
                next[tail[binIndex]] = id;
                tail[binIndex] = id;
            }
        }
        
        int numStocks = maxBinUsed + 1;
        
        // Fast output
        using var outStream = Console.OpenStandardOutput();
        using var writer = new StreamWriter(outStream, System.Text.Encoding.ASCII, 1024 * 1024);
        writer.WriteLine(numStocks);
        for (int i = 0; i < numStocks; i++)
        {
            int curr = head[i];
            bool first = true;
            while (curr != -1)
            {
                if (!first) writer.Write(" ");
                writer.Write(curr);
                first = false;
                curr = next[curr];
            }
            writer.WriteLine();
        }
    }

    struct Cut : IComparable<Cut>
    {
        public long Length;
        public int Id;

        public int CompareTo(Cut other)
        {
            // Descending order
            return other.Length.CompareTo(this.Length);
        }
    }
    
    static void Build(long[] tree, int node, int l, int r, long val)
    {
        tree[node] = val;
        if (l == r) return;
        int mid = l + (r - l) / 2;
        Build(tree, 2 * node, l, mid, val);
        Build(tree, 2 * node + 1, mid + 1, r, val);
    }
    
    static int FindAndReduce(long[] tree, int node, int l, int r, long val)
    {
        if (tree[node] < val) return -1;
        if (l == r)
        {
            tree[node] -= val;
            return l;
        }
        int mid = l + (r - l) / 2;
        if (tree[2 * node] >= val)
        {
            int res = FindAndReduce(tree, 2 * node, l, mid, val);
            tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
            return res;
        }
        else
        {
            int res = FindAndReduce(tree, 2 * node + 1, mid + 1, r, val);
            tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
            return res;
        }
    }
}

class Scanner
{
    private Stream stream;
    private byte[] buffer = new byte[1024 * 1024];
    private int head = 0;
    private int tail = 0;

    public Scanner(Stream stream)
    {
        this.stream = stream;
    }

    public long NextLong()
    {
        int c = Read();
        while (c <= ' ')
        {
            if (c == -1) return -1;
            c = Read();
        }
        long res = 0;
        while (c > ' ')
        {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }

    private int Read()
    {
        if (head >= tail)
        {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}
