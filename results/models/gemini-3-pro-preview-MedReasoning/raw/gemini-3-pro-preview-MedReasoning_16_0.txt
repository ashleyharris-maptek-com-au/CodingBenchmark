#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>

using namespace std;
using namespace std::chrono;

struct Task {
    int machine;
    int duration;
    int start_time = -1;
};

struct Job {
    vector<Task> tasks;
    int current_task = 0;
    int end_time = 0;
    long long remaining_work = 0;
    int remaining_tasks = 0;
};

struct Interval {
    int start;
    int end;
    bool operator<(const Interval& other) const {
        return start < other.start;
    }
};

struct Machine {
    vector<Interval> intervals;

    int get_earliest_start(int min_start, int duration) const {
        if (duration == 0) return min_start;
        int current_start = min_start;
        for (const auto& iv : intervals) {
            if (current_start + duration <= iv.start) {
                return current_start;
            }
            if (current_start < iv.end) {
                current_start = iv.end;
            }
        }
        return current_start;
    }

    void add_task(int start, int duration) {
        if (duration == 0) return;
        intervals.push_back({start, start + duration});
        
        int i = (int)intervals.size() - 2;
        Interval temp = intervals.back();
        while (i >= 0 && intervals[i].start > temp.start) {
            intervals[i + 1] = intervals[i];
            i--;
        }
        intervals[i + 1] = temp;
    }
    
    void clear() {
        intervals.clear();
        intervals.reserve(1024);
    }
};

struct AvailableTask {
    bool valid = false;
    int machine_id;
    int duration;
    int earliest_start;
    int earliest_completion;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> original_jobs(J);
    int total_tasks = 0;
    for (int i = 0; i < J; ++i) {
        int T;
        cin >> T;
        original_jobs[i].resize(T);
        total_tasks += T;
        for (int j = 0; j < T; ++j) {
            cin >> original_jobs[i][j].machine >> original_jobs[i][j].duration;
        }
    }

    int max_iterations = 1;
    long long time_limit_ms = 1900; 
    vector<int> strategies; 
    
    if (total_tasks <= 50) {
        max_iterations = 50000;
        strategies = {0, 1, 2, 3, 4}; 
    } else if (total_tasks <= 1000) {
        max_iterations = 5000;
        strategies = {0, 1, 2, 3, 4};
    } else if (total_tasks <= 10000) {
        max_iterations = 500;
        strategies = {0, 1, 2, 3, 4};
    } else {
        max_iterations = 10;
        strategies = {0, 1, 3};
    }

    mt19937 rng(1337);

    int best_makespan = 2e9 + 7;
    vector<vector<Task>> best_jobs;

    auto start_time_clock = high_resolution_clock::now();

    for (int iter = 0; iter < max_iterations; ++iter) {
        vector<Job> jobs(J);
        for (int i = 0; i < J; ++i) {
            jobs[i].tasks = original_jobs[i];
            jobs[i].remaining_tasks = jobs[i].tasks.size();
            for (const auto& t : jobs[i].tasks) {
                jobs[i].remaining_work += t.duration;
            }
        }
        vector<Machine> machines(M);
        for(auto& m : machines) m.clear();
        
        vector<AvailableTask> available(J);
        auto update_available = [&](int job_idx) {
            if (jobs[job_idx].current_task < jobs[job_idx].tasks.size()) {
                Task& t = jobs[job_idx].tasks[jobs[job_idx].current_task];
                available[job_idx].valid = true;
                available[job_idx].machine_id = t.machine;
                available[job_idx].duration = t.duration;
                available[job_idx].earliest_start = machines[t.machine].get_earliest_start(jobs[job_idx].end_time, t.duration);
                available[job_idx].earliest_completion = available[job_idx].earliest_start + t.duration;
            } else {
                available[job_idx].valid = false;
            }
        };

        for (int i = 0; i < J; ++i) {
            update_available(i);
        }

        int scheduled_tasks = 0;
        int current_makespan = 0;
        
        int strategy = (iter < strategies.size()) ? strategies[iter] : 5; 

        while (scheduled_tasks < total_tasks) {
            int min_completion_time = 2e9 + 7;
            int selected_machine = -1;
            
            for (int i = 0; i < J; ++i) {
                if (available[i].valid) {
                    if (available[i].earliest_completion < min_completion_time) {
                        min_completion_time = available[i].earliest_completion;
                        selected_machine = available[i].machine_id;
                    }
                }
            }
            
            vector<int> conflict_set;
            for (int i = 0; i < J; ++i) {
                if (available[i].valid && available[i].machine_id == selected_machine) {
                    if (available[i].earliest_start < min_completion_time) {
                        conflict_set.push_back(i);
                    }
                }
            }
            if (conflict_set.empty()) {
                for (int i = 0; i < J; ++i) {
                    if (available[i].valid && available[i].machine_id == selected_machine) {
                        if (available[i].earliest_completion == min_completion_time) {
                            conflict_set.push_back(i);
                        }
                    }
                }
            }
            
            int chosen_job_idx = -1;
            if (strategy == 0) {
                long long max_val = -1;
                for (int idx : conflict_set) {
                    if (jobs[idx].remaining_work > max_val) {
                        max_val = jobs[idx].remaining_work;
                        chosen_job_idx = idx;
                    }
                }
            } else if (strategy == 1) {
                int min_val = 2e9 + 7;
                for (int idx : conflict_set) {
                    if (available[idx].duration < min_val) {
                        min_val = available[idx].duration;
                        chosen_job_idx = idx;
                    }
                }
            } else if (strategy == 2) {
                int max_val = -1;
                for (int idx : conflict_set) {
                    if (available[idx].duration > max_val) {
                        max_val = available[idx].duration;
                        chosen_job_idx = idx;
                    }
                }
            } else if (strategy == 3) {
                int min_val = 2e9 + 7;
                for (int idx : conflict_set) {
                    if (available[idx].earliest_start < min_val) {
                        min_val = available[idx].earliest_start;
                        chosen_job_idx = idx;
                    }
                }
            } else if (strategy == 4) {
                int max_val = -1;
                for (int idx : conflict_set) {
                    if (jobs[idx].remaining_tasks > max_val) {
                        max_val = jobs[idx].remaining_tasks;
                        chosen_job_idx = idx;
                    }
                }
            } else {
                uniform_int_distribution<int> dist(0, (int)conflict_set.size() - 1);
                chosen_job_idx = conflict_set[dist(rng)];
            }
            
            Task& t = jobs[chosen_job_idx].tasks[jobs[chosen_job_idx].current_task];
            t.start_time = available[chosen_job_idx].earliest_start;
            
            machines[t.machine].add_task(t.start_time, t.duration);
            jobs[chosen_job_idx].end_time = t.start_time + t.duration;
            jobs[chosen_job_idx].remaining_work -= t.duration;
            jobs[chosen_job_idx].remaining_tasks--;
            jobs[chosen_job_idx].current_task++;
            scheduled_tasks++;
            
            current_makespan = max(current_makespan, jobs[chosen_job_idx].end_time);
            
            update_available(chosen_job_idx);
            for (int i = 0; i < J; ++i) {
                if (i != chosen_job_idx && available[i].valid && available[i].machine_id == selected_machine) {
                    update_available(i);
                }
            }
        }
        
        if (current_makespan < best_makespan) {
            best_makespan = current_makespan;
            best_jobs.resize(J);
            for(int i = 0; i < J; ++i) {
                best_jobs[i] = jobs[i].tasks;
            }
        }
        
        auto now = high_resolution_clock::now();
        if (duration_cast<milliseconds>(now - start_time_clock).count() > time_limit_ms) {
            break;
        }
    }
    
    for (int i = 0; i < J; ++i) {
        for (int j = 0; j < (int)best_jobs[i].size(); ++j) {
            cout << best_jobs[i][j].start_time << " " << best_jobs[i][j].machine;
            if (j < (int)best_jobs[i].size() - 1) cout << " ";
        }
        cout << "\n";
    }

    return 0;
}
