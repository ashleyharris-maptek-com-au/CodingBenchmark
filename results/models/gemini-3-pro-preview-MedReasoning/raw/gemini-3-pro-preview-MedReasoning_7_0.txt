const double INF = 1e30;

struct vec3 {
    double x, y, z;
    vec3() : x(0), y(0), z(0) {}
    vec3(double x, double y, double z) : x(x), y(y), z(z) {}
    vec3(double v) : x(v), y(v), z(v) {}
    vec3 operator+(const vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    vec3 operator-(const vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
};

double dot(const vec3& a, const vec3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
vec3 cross(const vec3& a, const vec3& b) { return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x}; }
double length(const vec3& a) { return sqrt(dot(a, a)); }
vec3 normalize(const vec3& a) { double l = length(a); return l>0 ? a*(1.0/l) : vec3(0); }
vec3 min(const vec3& a, const vec3& b) { return {std::min(a.x, b.x), std::min(a.y, b.y), std::min(a.z, b.z)}; }
vec3 max(const vec3& a, const vec3& b) { return {std::max(a.x, b.x), std::max(a.y, b.y), std::max(a.z, b.z)}; }

struct Face { int v[3]; };

struct Mesh {
    std::vector<vec3> vertices;
    std::vector<Face> faces;
};

struct BVHNode {
    vec3 bmin, bmax;
    int left, right;
    int first, count;
};

struct BVH {
    std::vector<BVHNode> nodes;
    std::vector<int> tri_indices;
    
    int build(std::vector<int>& indices, int start, int end, const Mesh& mesh, const std::vector<vec3>& centroids) {
        int nodeIdx = nodes.size();
        nodes.push_back(BVHNode());
        
        vec3 bmin(INF), bmax(-INF);
        vec3 cmin(INF), cmax(-INF);
        
        for (int i = start; i < end; ++i) {
            int t = indices[i];
            for (int j = 0; j < 3; ++j) {
                bmin = min(bmin, mesh.vertices[mesh.faces[t].v[j]]);
                bmax = max(bmax, mesh.vertices[mesh.faces[t].v[j]]);
            }
            cmin = min(cmin, centroids[t]);
            cmax = max(cmax, centroids[t]);
        }
        
        nodes[nodeIdx].bmin = bmin - vec3(1e-6);
        nodes[nodeIdx].bmax = bmax + vec3(1e-6);
        
        if (end - start <= 4) {
            nodes[nodeIdx].left = nodes[nodeIdx].right = -1;
            nodes[nodeIdx].first = start;
            nodes[nodeIdx].count = end - start;
            return nodeIdx;
        }
        
        int axis = 0;
        vec3 ext = cmax - cmin;
        if (ext.y > ext.x) axis = 1;
        if (ext.z > ext.y && ext.z > ext.x) axis = 2;
        
        double split_val = (cmin.x * (axis==0) + cmin.y * (axis==1) + cmin.z * (axis==2) +
                            cmax.x * (axis==0) + cmax.y * (axis==1) + cmax.z * (axis==2)) * 0.5;
        
        int mid = start;
        for (int i = start; i < end; ++i) {
            double cv = centroids[indices[i]].x * (axis==0) +
                        centroids[indices[i]].y * (axis==1) +
                        centroids[indices[i]].z * (axis==2);
            if (cv < split_val) {
                std::swap(indices[i], indices[mid]);
                mid++;
            }
        }
        
        if (mid == start || mid == end) mid = start + (end - start) / 2;
        
        nodes[nodeIdx].count = 0;
        nodes[nodeIdx].left = build(indices, start, mid, mesh, centroids);
        nodes[nodeIdx].right = build(indices, mid, end, mesh, centroids);
        return nodeIdx;
    }
    
    void build_mesh(const Mesh& mesh) {
        if(mesh.faces.empty()) return;
        tri_indices.resize(mesh.faces.size());
        std::iota(tri_indices.begin(), tri_indices.end(), 0);
        std::vector<vec3> centroids(mesh.faces.size());
        for (size_t i = 0; i < mesh.faces.size(); ++i) {
            vec3 c(0);
            for (int j=0; j<3; ++j) c = c + mesh.vertices[mesh.faces[i].v[j]];
            centroids[i] = c * (1.0/3.0);
        }
        build(tri_indices, 0, tri_indices.size(), mesh, centroids);
    }
};

bool rayIntersectsAABB(const vec3& orig, const vec3& dir, const vec3& bmin, const vec3& bmax, double& tmin, double& tmax) {
    tmin = -INF; tmax = INF;
    for (int i = 0; i < 3; ++i) {
        double o = (i==0) ? orig.x : ((i==1) ? orig.y : orig.z);
        double d = (i==0) ? dir.x : ((i==1) ? dir.y : dir.z);
        double min_v = (i==0) ? bmin.x : ((i==1) ? bmin.y : bmin.z);
        double max_v = (i==0) ? bmax.x : ((i==1) ? bmax.y : bmax.z);
        
        if (std::abs(d) < 1e-9) {
            if (o < min_v || o > max_v) return false;
        } else {
            double t1 = (min_v - o) / d;
            double t2 = (max_v - o) / d;
            if (t1 > t2) std::swap(t1, t2);
            if (t1 > tmin) tmin = t1;
            if (t2 < tmax) tmax = t2;
            if (tmin > tmax) return false;
        }
    }
    return tmax > 0;
}

bool aabbOverlap(const vec3& min1, const vec3& max1, const vec3& min2, const vec3& max2) {
    return (max1.x >= min2.x && min1.x <= max2.x &&
            max1.y >= min2.y && min1.y <= max2.y &&
            max1.z >= min2.z && min1.z <= max2.z);
}

std::vector<int> getIntersectingTriangles(const vec3& bmin, const vec3& bmax, const BVH& bvh) {
    std::vector<int> result;
    if (bvh.nodes.empty()) return result;
    int stack[64];
    int stack_ptr = 0;
    stack[stack_ptr++] = 0;
    
    while(stack_ptr > 0) {
        int nodeIdx = stack[--stack_ptr];
        const BVHNode& node = bvh.nodes[nodeIdx];
        if (!aabbOverlap(bmin, bmax, node.bmin, node.bmax)) continue;
        
        if (node.count > 0) {
            for(int i=0; i<node.count; ++i) {
                result.push_back(bvh.tri_indices[node.first + i]);
            }
        } else {
            stack[stack_ptr++] = node.left;
            stack[stack_ptr++] = node.right;
        }
    }
    return result;
}

bool overlap(const vec3& axis, const vec3 p1[3], const vec3 p2[3]) {
    double min1 = INF, max1 = -INF, min2 = INF, max2 = -INF;
    for(int i=0; i<3; ++i) {
        double d1 = dot(axis, p1[i]);
        if(d1 < min1) min1 = d1;
        if(d1 > max1) max1 = d1;
        double d2 = dot(axis, p2[i]);
        if(d2 < min2) min2 = d2;
        if(d2 > max2) max2 = d2;
    }
    return (max1 >= min2 && max2 >= min1);
}

bool triTriIntersect(const vec3& a0, const vec3& a1, const vec3& a2,
                     const vec3& b0, const vec3& b1, const vec3& b2) {
    vec3 p1[3] = {a0, a1, a2};
    vec3 p2[3] = {b0, b1, b2};
    vec3 e1[3] = {p1[1]-p1[0], p1[2]-p1[1], p1[0]-p1[2]};
    vec3 e2[3] = {p2[1]-p2[0], p2[2]-p2[1], p2[0]-p2[2]};
    vec3 n1 = cross(e1[0], e1[1]);
    vec3 n2 = cross(e2[0], e2[1]);
    
    if(!overlap(n1, p1, p2)) return false;
    if(!overlap(n2, p1, p2)) return false;
    
    for(int i=0; i<3; ++i) {
        for(int j=0; j<3; ++j) {
            vec3 axis = cross(e1[i], e2[j]);
            if(length(axis) > 1e-8) {
                if(!overlap(axis, p1, p2)) return false;
            }
        }
    }
    return true;
}

bool rayTriIntersect(const vec3& orig, const vec3& dir, const vec3& v0, const vec3& v1, const vec3& v2, double& t, double& u, double& v) {
    vec3 edge1 = v1 - v0;
    vec3 edge2 = v2 - v0;
    vec3 h = cross(dir, edge2);
    double a = dot(edge1, h);
    if (a > -1e-8 && a < 1e-8) return false;
    double f = 1.0 / a;
    vec3 s = orig - v0;
    u = f * dot(s, h);
    if (u < 0.0 || u > 1.0) return false;
    vec3 q = cross(s, edge1);
    v = f * dot(dir, q);
    if (v < 0.0 || u + v > 1.0) return false;
    t = f * dot(edge2, q);
    return t > 1e-8;
}

vec3 getRandomDir(int attempt) {
    double x = std::sin(attempt * 12.9898) * std::cos(attempt * 78.233);
    double y = std::sin(attempt * 12.9898) * std::sin(attempt * 78.233);
    double z = std::cos(attempt * 12.9898);
    return normalize(vec3(x, y, z));
}

bool isSpaceOutside(const vec3& orig, vec3 D, const Mesh& mesh, const BVH& bvh) {
    if (bvh.nodes.empty()) return true;
    for (int attempt = 1; attempt <= 20; ++attempt) {
        vec3 rayD = D;
        if (attempt > 1) {
            vec3 randD = getRandomDir(attempt * 137);
            if (dot(randD, D) < 0) randD = randD * -1.0;
            rayD = normalize(D + randD * 0.15);
        }
        
        int hit_tri = -1;
        double min_t = INF;
        bool edge_hit = false;
        
        int stack[64];
        int stack_ptr = 0;
        stack[stack_ptr++] = 0;
        
        while (stack_ptr > 0) {
            int nodeIdx = stack[--stack_ptr];
            const BVHNode& node = bvh.nodes[nodeIdx];
            double tmin_aabb, tmax_aabb;
            if (!rayIntersectsAABB(orig, rayD, node.bmin, node.bmax, tmin_aabb, tmax_aabb)) continue;
            if (tmin_aabb > min_t) continue;
            
            if (node.count > 0) {
                for (int i = 0; i < node.count; ++i) {
                    int triIdx = bvh.tri_indices[node.first + i];
                    vec3 v0 = mesh.vertices[mesh.faces[triIdx].v[0]];
                    vec3 v1 = mesh.vertices[mesh.faces[triIdx].v[1]];
                    vec3 v2 = mesh.vertices[mesh.faces[triIdx].v[2]];
                    double t, u, v;
                    if (rayTriIntersect(orig, rayD, v0, v1, v2, t, u, v)) {
                        if (t > 1e-8 && t < min_t) {
                            if (u < 1e-4 || v < 1e-4 || (1 - u - v) < 1e-4) {
                                edge_hit = true;
                                min_t = t;
                            } else {
                                edge_hit = false;
                                min_t = t;
                                hit_tri = triIdx;
                            }
                        }
                    }
                }
            } else {
                stack[stack_ptr++] = node.left;
                stack[stack_ptr++] = node.right;
            }
        }
        
        if (edge_hit) continue;
        
        if (hit_tri != -1) {
            vec3 v0 = mesh.vertices[mesh.faces[hit_tri].v[0]];
            vec3 v1 = mesh.vertices[mesh.faces[hit_tri].v[1]];
            vec3 v2 = mesh.vertices[mesh.faces[hit_tri].v[2]];
            vec3 N = normalize(cross(v1 - v0, v2 - v1));
            double d = dot(N, rayD);
            if (std::abs(d) < 1e-4) continue;
            return d < 0;
        } else {
            return true;
        }
    }
    return true;
}

void splitPolygon(const std::vector<vec3>& poly, const vec3& n, double d, std::vector<vec3>& front, std::vector<vec3>& back) {
    front.clear(); back.clear();
    if (poly.empty()) return;
    std::vector<double> dists(poly.size());
    int frontCount = 0, backCount = 0;
    for (size_t i = 0; i < poly.size(); i++) {
        dists[i] = dot(n, poly[i]) + d;
        if (std::abs(dists[i]) < 1e-9) dists[i] = 0.0;
        if (dists[i] > 0) frontCount++;
        else if (dists[i] < 0) backCount++;
    }
    if (frontCount == 0 && backCount == 0) {
        front = poly; return;
    }
    if (frontCount == 0) {
        back = poly; return;
    }
    if (backCount == 0) {
        front = poly; return;
    }
    for (size_t i = 0; i < poly.size(); i++) {
        size_t j = (i + 1) % poly.size();
        double d1 = dists[i], d2 = dists[j];
        vec3 p1 = poly[i], p2 = poly[j];

        if (d1 >= 0) front.push_back(p1);
        if (d1 <= 0) back.push_back(p1);

        if ((d1 > 0 && d2 < 0) || (d1 < 0 && d2 > 0)) {
            double t = d1 / (d1 - d2);
            vec3 p = p1 + (p2 - p1) * t;
            front.push_back(p);
            back.push_back(p);
        }
    }
}

struct Plane { vec3 n; double d; };

std::vector<vec3> processTriangle(const Mesh& meshA, int faceIdx, const Mesh& meshB, const BVH& bvhB, bool isMeshA) {
    std::vector<vec3> result_tris;
    
    vec3 v0 = meshA.vertices[meshA.faces[faceIdx].v[0]];
    vec3 v1 = meshA.vertices[meshA.faces[faceIdx].v[1]];
    vec3 v2 = meshA.vertices[meshA.faces[faceIdx].v[2]];
    vec3 NA = normalize(cross(v1 - v0, v2 - v1));
    if (length(NA) < 1e-8) return result_tris;
    
    vec3 bmin = min(min(v0, v1), v2) - vec3(1e-5);
    vec3 bmax = max(max(v0, v1), v2) + vec3(1e-5);
    
    std::vector<int> potential = getIntersectingTriangles(bmin, bmax, bvhB);
    std::vector<Plane> planes;
    
    auto addPlane = [&](const vec3& n, double d) {
        for (const auto& p : planes) {
            if (std::abs(dot(p.n, n)) > 0.9999 && std::abs(p.d - d) < 1e-5) return;
            if (std::abs(dot(p.n, n)) < -0.9999 && std::abs(p.d + d) < 1e-5) return;
        }
        planes.push_back({n, d});
    };
    
    for (int tb_idx : potential) {
        vec3 u0 = meshB.vertices[meshB.faces[tb_idx].v[0]];
        vec3 u1 = meshB.vertices[meshB.faces[tb_idx].v[1]];
        vec3 u2 = meshB.vertices[meshB.faces[tb_idx].v[2]];
        
        if (triTriIntersect(v0, v1, v2, u0, u1, u2)) {
            vec3 NB = normalize(cross(u1 - u0, u2 - u1));
            if (std::abs(dot(NA, NB)) > 0.9999) {
                for (int i=0; i<3; ++i) {
                    vec3 e_start = (i==0)?u0 : ((i==1)?u1:u2);
                    vec3 e_end   = (i==0)?u1 : ((i==1)?u2:u0);
                    vec3 edge_dir = normalize(e_end - e_start);
                    vec3 edge_normal = cross(edge_dir, NB);
                    if (length(edge_normal) > 1e-5) {
                        edge_normal = normalize(edge_normal);
                        addPlane(edge_normal, -dot(edge_normal, e_start));
                    }
                }
            } else {
                addPlane(NB, -dot(NB, u0));
            }
        }
    }
    
    std::vector<std::vector<vec3>> fragments;
    fragments.push_back({v0, v1, v2});
    
    for (const auto& plane : planes) {
        std::vector<std::vector<vec3>> next_frags;
        for (const auto& frag : fragments) {
            std::vector<vec3> front, back;
            splitPolygon(frag, plane.n, plane.d, front, back);
            if (front.size() >= 3) next_frags.push_back(front);
            if (back.size() >= 3) next_frags.push_back(back);
        }
        fragments = next_frags;
    }
    
    for (const auto& frag : fragments) {
        if (frag.size() < 3) continue;
        double area = 0;
        for(size_t i=1; i<frag.size()-1; ++i) {
            area += length(cross(frag[i]-frag[0], frag[i+1]-frag[0]));
        }
        if (area < 1e-12) continue;
        
        vec3 C(0,0,0);
        for(auto v : frag) C = C + v;
        C = C * (1.0 / frag.size());
        
        bool out_pos = isSpaceOutside(C, NA, meshB, bvhB);
        bool out_neg = isSpaceOutside(C, NA * -1.0, meshB, bvhB);
        
        bool keep = false;
        if (out_pos && out_neg) keep = true;
        else if (!out_pos && !out_neg) keep = false;
        else if (out_pos && !out_neg) keep = isMeshA;
        else keep = false;
        
        if (keep) {
            for (size_t i = 1; i < frag.size() - 1; ++i) {
                result_tris.push_back(frag[0]);
                result_tris.push_back(frag[i]);
                result_tris.push_back(frag[i+1]);
            }
        }
    }
    return result_tris;
}

struct ivec3 {
    long long x, y, z;
    bool operator==(const ivec3& o) const { return x==o.x && y==o.y && z==o.z; }
};

namespace std {
    template<> struct hash<ivec3> {
        size_t operator()(const ivec3& v) const {
            return (v.x * 73856093) ^ (v.y * 19349663) ^ (v.z * 83492791);
        }
    };
}

ivec3 quantize(const vec3& v) {
    const double SCALE = 1e6;
    return { (long long)(std::round(v.x * SCALE)),
             (long long)(std::round(v.y * SCALE)),
             (long long)(std::round(v.z * SCALE)) };
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.vertices.resize(VA); A.faces.resize(FA);
    B.vertices.resize(VB); B.faces.resize(FB);

    for (int i = 0; i < VA; ++i) std::cin >> A.vertices[i].x >> A.vertices[i].y >> A.vertices[i].z;
    for (int i = 0; i < FA; ++i) std::cin >> A.faces[i].v[0] >> A.faces[i].v[1] >> A.faces[i].v[2];
    for (int i = 0; i < VB; ++i) std::cin >> B.vertices[i].x >> B.vertices[i].y >> B.vertices[i].z;
    for (int i = 0; i < FB; ++i) std::cin >> B.faces[i].v[0] >> B.faces[i].v[1] >> B.faces[i].v[2];

    BVH bvhA, bvhB;
    bvhA.build_mesh(A);
    bvhB.build_mesh(B);

    std::vector<int> indicesA(FA), indicesB(FB);
    std::iota(indicesA.begin(), indicesA.end(), 0);
    std::iota(indicesB.begin(), indicesB.end(), 0);

    std::vector<std::vector<vec3>> outputsA(FA), outputsB(FB);

    std::for_each(std::execution::par, indicesA.begin(), indicesA.end(), [&](int i) {
        outputsA[i] = processTriangle(A, i, B, bvhB, true);
    });

    std::for_each(std::execution::par, indicesB.begin(), indicesB.end(), [&](int i) {
        outputsB[i] = processTriangle(B, i, A, bvhA, false);
    });

    std::unordered_map<ivec3, int> vertex_map;
    std::vector<vec3> out_vertices;
    std::vector<Face> out_faces;

    auto addVertex = [&](const vec3& v) -> int {
        ivec3 q = quantize(v);
        auto it = vertex_map.find(q);
        if (it != vertex_map.end()) return it->second;
        int idx = out_vertices.size();
        out_vertices.push_back(v);
        vertex_map[q] = idx;
        return idx;
    };

    for (const auto& list : outputsA) {
        for (size_t i = 0; i < list.size(); i += 3) {
            int v0 = addVertex(list[i]);
            int v1 = addVertex(list[i+1]);
            int v2 = addVertex(list[i+2]);
            if (v0 != v1 && v1 != v2 && v2 != v0) {
                out_faces.push_back({v0, v1, v2});
            }
        }
    }
    for (const auto& list : outputsB) {
        for (size_t i = 0; i < list.size(); i += 3) {
            int v0 = addVertex(list[i]);
            int v1 = addVertex(list[i+1]);
            int v2 = addVertex(list[i+2]);
            if (v0 != v1 && v1 != v2 && v2 != v0) {
                out_faces.push_back({v0, v1, v2});
            }
        }
    }

    std::cout << out_vertices.size() << " " << out_faces.size() << "\n";
    for (const auto& v : out_vertices) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : out_faces) {
        std::cout << f.v[0] << " " << f.v[1] << " " << f.v[2] << "\n";
    }

    return 0;
}
