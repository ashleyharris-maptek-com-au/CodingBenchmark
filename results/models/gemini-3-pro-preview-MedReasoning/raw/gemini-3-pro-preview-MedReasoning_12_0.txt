using System;
using System.IO;
using System.Collections.Generic;
using System.Text;
using System.Globalization;

namespace BinPacking
{
    class Program
    {
        struct Vector3
        {
            public double x, y, z;
            public Vector3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
            public Vector3 Rotate(Quaternion q)
            {
                Quaternion p = new Quaternion(0, x, y, z);
                Quaternion qInv = q.Inverse();
                Quaternion qp = Quaternion.Multiply(q, p);
                Quaternion qpqInv = Quaternion.Multiply(qp, qInv);
                return new Vector3(qpqInv.x, qpqInv.y, qpqInv.z);
            }
        }

        struct Quaternion
        {
            public double w, x, y, z;
            public Quaternion(double w, double x, double y, double z) { this.w = w; this.x = x; this.y = y; this.z = z; }
            public static Quaternion Multiply(Quaternion q1, Quaternion q2)
            {
                return new Quaternion(
                    q1.w * q2.w - q1.x * q2.x - q1.y * q2.y - q1.z * q2.z,
                    q1.w * q2.x + q1.x * q2.w + q1.y * q2.z - q1.z * q2.y,
                    q1.w * q2.y - q1.x * q2.z + q1.y * q2.w + q1.z * q2.x,
                    q1.w * q2.z + q1.x * q2.y - q1.y * q2.x + q1.z * q2.w
                );
            }
            public Quaternion Inverse() { return new Quaternion(w, -x, -y, -z); }
        }

        static Quaternion FromEuler(double pitch, double yaw, double roll)
        {
            double cy = Math.Cos(yaw * 0.5);
            double sy = Math.Sin(yaw * 0.5);
            double cp = Math.Cos(pitch * 0.5);
            double sp = Math.Sin(pitch * 0.5);
            double cr = Math.Cos(roll * 0.5);
            double sr = Math.Sin(roll * 0.5);

            Quaternion q;
            q.w = cr * cp * cy + sr * sp * sy;
            q.x = sr * cp * cy - cr * sp * sy;
            q.y = cr * sp * cy + sr * cp * sy;
            q.z = cr * cp * sy - sr * sp * cy;
            return q;
        }

        class Scanner
        {
            private readonly TextReader _reader;
            private char[] _buffer = new char[65536];
            private int _head = 0;
            private int _tail = 0;

            public Scanner(TextReader reader)
            {
                _reader = reader;
            }

            public string NextToken()
            {
                StringBuilder sb = new StringBuilder();
                int c = ReadChar();
                while (c != -1 && char.IsWhiteSpace((char)c))
                {
                    c = ReadChar();
                }
                if (c == -1) return null;
                while (c != -1 && !char.IsWhiteSpace((char)c))
                {
                    sb.Append((char)c);
                    c = ReadChar();
                }
                return sb.ToString();
            }

            private int ReadChar()
            {
                if (_head >= _tail)
                {
                    _head = 0;
                    _tail = _reader.Read(_buffer, 0, _buffer.Length);
                    if (_tail == 0) return -1;
                }
                return _buffer[_head++];
            }
        }

        enum AdaptiveStrategy { Trivial, Medium, Large, Extreme }

        static void Main()
        {
            Scanner scanner = new Scanner(Console.In);
            string vStr = scanner.NextToken();
            if (vStr == null) return;
            
            int V = int.Parse(vStr);
            int F = int.Parse(scanner.NextToken());

            List<Vector3> vertices = new List<Vector3>(V);
            for (int i = 0; i < V; i++)
            {
                double x = double.Parse(scanner.NextToken(), CultureInfo.InvariantCulture);
                double y = double.Parse(scanner.NextToken(), CultureInfo.InvariantCulture);
                double z = double.Parse(scanner.NextToken(), CultureInfo.InvariantCulture);
                vertices.Add(new Vector3(x, y, z));
            }

            // Skip face indices as AABB packing doesn't strictly need them
            for (int i = 0; i < F; i++)
            {
                scanner.NextToken();
                scanner.NextToken();
                scanner.NextToken();
            }

            double ContainerW = double.Parse(scanner.NextToken(), CultureInfo.InvariantCulture);
            double ContainerH = double.Parse(scanner.NextToken(), CultureInfo.InvariantCulture);
            double ContainerD = double.Parse(scanner.NextToken(), CultureInfo.InvariantCulture);

            // Generate 24 orthogonal rotations
            List<Quaternion> uniqueOrientations = new List<Quaternion>();
            List<string> seenAxes = new List<string>();

            double[] angles = { 0, Math.PI / 2, Math.PI, 3 * Math.PI / 2 };
            foreach (var ax in angles)
            {
                foreach (var ay in angles)
                {
                    foreach (var az in angles)
                    {
                        Quaternion q = FromEuler(ax, ay, az);
                        
                        double len = Math.Sqrt(q.w * q.w + q.x * q.x + q.y * q.y + q.z * q.z);
                        q.w /= len; q.x /= len; q.y /= len; q.z /= len;

                        Vector3 vx = (new Vector3(1, 0, 0)).Rotate(q);
                        Vector3 vy = (new Vector3(0, 1, 0)).Rotate(q);
                        Vector3 vz = (new Vector3(0, 0, 1)).Rotate(q);
                        
                        int xx = (int)Math.Round(vx.x); int xy = (int)Math.Round(vx.y); int xz = (int)Math.Round(vx.z);
                        int yx = (int)Math.Round(vy.x); int yy = (int)Math.Round(vy.y); int yz = (int)Math.Round(vy.z);
                        int zx = (int)Math.Round(vz.x); int zy = (int)Math.Round(vz.y); int zz = (int)Math.Round(vz.z);
                        
                        string hash = $"{xx},{xy},{xz},{yx},{yy},{yz},{zx},{zy},{zz}";
                        if (!seenAxes.Contains(hash))
                        {
                            seenAxes.Add(hash);
                            uniqueOrientations.Add(q);
                        }
                    }
                }
            }

            long maxCount = -1;
            Quaternion bestQ = new Quaternion(1, 0, 0, 0);
            double bestW = 0, bestH = 0, bestD = 0;
            double bestMinX = 0, bestMinY = 0, bestMinZ = 0;
            long bestCountX = 0, bestCountY = 0, bestCountZ = 0;

            foreach (var q in uniqueOrientations)
            {
                double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
                double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

                foreach (var v in vertices)
                {
                    Vector3 rv = v.Rotate(q);
                    if (rv.x < minX) minX = rv.x;
                    if (rv.y < minY) minY = rv.y;
                    if (rv.z < minZ) minZ = rv.z;
                    if (rv.x > maxX) maxX = rv.x;
                    if (rv.y > maxY) maxY = rv.y;
                    if (rv.z > maxZ) maxZ = rv.z;
                }

                double w = Math.Round((maxX - minX) * 1000000) / 1000000.0;
                double h = Math.Round((maxY - minY) * 1000000) / 1000000.0;
                double d = Math.Round((maxZ - minZ) * 1000000) / 1000000.0;
                
                w = Math.Max(w, 1e-6);
                h = Math.Max(h, 1e-6);
                d = Math.Max(d, 1e-6);

                long countX = (long)Math.Floor((ContainerW + 1e-7) / w);
                long countY = (long)Math.Floor((ContainerH + 1e-7) / h);
                long countZ = (long)Math.Floor((ContainerD + 1e-7) / d);
                long total = countX * countY * countZ;

                if (total > maxCount)
                {
                    maxCount = total;
                    bestQ = q;
                    bestW = w; bestH = h; bestD = d;
                    bestMinX = minX; bestMinY = minY; bestMinZ = minZ;
                    bestCountX = countX; bestCountY = countY; bestCountZ = countZ;
                }
            }

            // Adaptive Strategy Logic
            AdaptiveStrategy strategy = AdaptiveStrategy.Trivial;
            double vol = ContainerW * ContainerH * ContainerD;
            if (vol >= 1000000) strategy = AdaptiveStrategy.Extreme;
            else if (vol >= 8000) strategy = AdaptiveStrategy.Large;
            else if (vol >= 1000) strategy = AdaptiveStrategy.Medium;

            Console.Error.WriteLine($"Adaptive Strategy Mode: {strategy}");
            Console.Error.WriteLine($"Best pack count: {maxCount} (Grid: {bestCountX}x{bestCountY}x{bestCountZ})");

            // Extreme output optimization ensures < 5 min limit is easily met.
            using (var stdout = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1048576))
            {
                stdout.WriteLine(maxCount);
                if (maxCount == 0) return;

                string qStr = $"{bestQ.w.ToString("F5", CultureInfo.InvariantCulture)} {bestQ.x.ToString("F5", CultureInfo.InvariantCulture)} {bestQ.y.ToString("F5", CultureInfo.InvariantCulture)} {bestQ.z.ToString("F5", CultureInfo.InvariantCulture)}";
                
                string[] txStrs = new string[bestCountX];
                for (long ix = 0; ix < bestCountX; ix++)
                {
                    double tx = ix * bestW - bestMinX;
                    txStrs[ix] = tx.ToString("F5", CultureInfo.InvariantCulture);
                }

                string[] tyStrs = new string[bestCountY];
                for (long iy = 0; iy < bestCountY; iy++)
                {
                    double ty = iy * bestH - bestMinY;
                    tyStrs[iy] = ty.ToString("F5", CultureInfo.InvariantCulture);
                }

                string[] tzStrs = new string[bestCountZ];
                for (long iz = 0; iz < bestCountZ; iz++)
                {
                    double tz = iz * bestD - bestMinZ;
                    tzStrs[iz] = tz.ToString("F5", CultureInfo.InvariantCulture);
                }

                for (long iz = 0; iz < bestCountZ; iz++)
                {
                    string sz = tzStrs[iz];
                    for (long iy = 0; iy < bestCountY; iy++)
                    {
                        string sy = tyStrs[iy];
                        for (long ix = 0; ix < bestCountX; ix++)
                        {
                            stdout.Write(txStrs[ix]);
                            stdout.Write(' ');
                            stdout.Write(sy);
                            stdout.Write(' ');
                            stdout.Write(sz);
                            stdout.Write(' ');
                            stdout.WriteLine(qStr);
                        }
                    }
                }
            }
        }
    }
}
