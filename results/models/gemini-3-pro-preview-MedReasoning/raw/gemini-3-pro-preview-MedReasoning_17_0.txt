using System;
using System.Collections.Generic;
using System.Diagnostics;

class Program
{
    struct Box
    {
        public int Id;
        public long Volume;
        public int w, h, d;
        public int MaxDim, MinDim, MidDim;
    }

    struct PlacedBox
    {
        public int Id;
        public int x, y, z;
    }

    struct Space
    {
        public int x, y, z;
        public int w, h, d;
        public long Volume;
    }

    static Space MakeSpace(int x, int y, int z, int w, int h, int d)
    {
        return new Space { x = x, y = y, z = z, w = w, h = h, d = d, Volume = (long)w * h * d };
    }

    static bool CanFit(Box box, Space s, out int fw, out int fh, out int fd)
    {
        fw = 0; fh = 0; fd = 0;
        int bestScore = -1;
        bool found = false;
        
        int w0 = box.w, h0 = box.h, d0 = box.d;
        
        // P1
        if (w0 <= s.w && h0 <= s.h && d0 <= s.d) {
            int score = Math.Max(s.w - w0, Math.Max(s.h - h0, s.d - d0));
            if (score > bestScore) { bestScore = score; fw = w0; fh = h0; fd = d0; found = true; }
        }
        // P2
        if (w0 <= s.w && d0 <= s.h && h0 <= s.d) {
            int score = Math.Max(s.w - w0, Math.Max(s.h - d0, s.d - h0));
            if (score > bestScore) { bestScore = score; fw = w0; fh = d0; fd = h0; found = true; }
        }
        // P3
        if (h0 <= s.w && w0 <= s.h && d0 <= s.d) {
            int score = Math.Max(s.w - h0, Math.Max(s.h - w0, s.d - d0));
            if (score > bestScore) { bestScore = score; fw = h0; fh = w0; fd = d0; found = true; }
        }
        // P4
        if (h0 <= s.w && d0 <= s.h && w0 <= s.d) {
            int score = Math.Max(s.w - h0, Math.Max(s.h - d0, s.d - w0));
            if (score > bestScore) { bestScore = score; fw = h0; fh = d0; fd = w0; found = true; }
        }
        // P5
        if (d0 <= s.w && w0 <= s.h && h0 <= s.d) {
            int score = Math.Max(s.w - d0, Math.Max(s.h - w0, s.d - h0));
            if (score > bestScore) { bestScore = score; fw = d0; fh = w0; fd = h0; found = true; }
        }
        // P6
        if (d0 <= s.w && h0 <= s.h && w0 <= s.d) {
            int score = Math.Max(s.w - d0, Math.Max(s.h - h0, s.d - w0));
            if (score > bestScore) { bestScore = score; fw = d0; fh = h0; fd = w0; found = true; }
        }
        
        return found;
    }

    static void SplitSpace(Space s, int w, int h, int d, List<Space> targetSpaces)
    {
        int rw = s.w - w;
        int rh = s.h - h;
        int rd = s.d - d;
        
        long v0 = Math.Max(Math.Max((long)rw*h*d, (long)s.w*rh*d), (long)s.w*s.h*rd); // P1
        long v1 = Math.Max(Math.Max((long)w*rh*d, (long)rw*s.h*d), (long)s.w*s.h*rd); // P2
        long v2 = Math.Max(Math.Max((long)rw*h*d, (long)s.w*h*rd), (long)s.w*rh*s.d); // P3
        long v3 = Math.Max(Math.Max((long)w*h*rd, (long)rw*h*s.d), (long)s.w*rh*s.d); // P4
        long v4 = Math.Max(Math.Max((long)w*rh*d, (long)w*s.h*rd), (long)rw*s.h*s.d); // P5
        long v5 = Math.Max(Math.Max((long)w*h*rd, (long)w*rh*s.d), (long)rw*s.h*s.d); // P6
        
        long maxV = v0; int best = 0;
        if (v1 > maxV) { maxV = v1; best = 1; }
        if (v2 > maxV) { maxV = v2; best = 2; }
        if (v3 > maxV) { maxV = v3; best = 3; }
        if (v4 > maxV) { maxV = v4; best = 4; }
        if (v5 > maxV) { maxV = v5; best = 5; }
        
        if (best == 0) {
            if (rw > 0 && h > 0 && d > 0) targetSpaces.Add(MakeSpace(s.x + w, s.y, s.z, rw, h, d));
            if (s.w > 0 && rh > 0 && d > 0) targetSpaces.Add(MakeSpace(s.x, s.y + h, s.z, s.w, rh, d));
            if (s.w > 0 && s.h > 0 && rd > 0) targetSpaces.Add(MakeSpace(s.x, s.y, s.z + d, s.w, s.h, rd));
        } else if (best == 1) {
            if (w > 0 && rh > 0 && d > 0) targetSpaces.Add(MakeSpace(s.x, s.y + h, s.z, w, rh, d));
            if (rw > 0 && s.h > 0 && d > 0) targetSpaces.Add(MakeSpace(s.x + w, s.y, s.z, rw, s.h, d));
            if (s.w > 0 && s.h > 0 && rd > 0) targetSpaces.Add(MakeSpace(s.x, s.y, s.z + d, s.w, s.h, rd));
        } else if (best == 2) {
            if (rw > 0 && h > 0 && d > 0) targetSpaces.Add(MakeSpace(s.x + w, s.y, s.z, rw, h, d));
            if (s.w > 0 && h > 0 && rd > 0) targetSpaces.Add(MakeSpace(s.x, s.y, s.z + d, s.w, h, rd));
            if (s.w > 0 && rh > 0 && s.d > 0) targetSpaces.Add(MakeSpace(s.x, s.y + h, s.z, s.w, rh, s.d));
        } else if (best == 3) {
            if (w > 0 && h > 0 && rd > 0) targetSpaces.Add(MakeSpace(s.x, s.y, s.z + d, w, h, rd));
            if (rw > 0 && h > 0 && s.d > 0) targetSpaces.Add(MakeSpace(s.x + w, s.y, s.z, rw, h, s.d));
            if (s.w > 0 && rh > 0 && s.d > 0) targetSpaces.Add(MakeSpace(s.x, s.y + h, s.z, s.w, rh, s.d));
        } else if (best == 4) {
            if (w > 0 && rh > 0 && d > 0) targetSpaces.Add(MakeSpace(s.x, s.y + h, s.z, w, rh, d));
            if (w > 0 && s.h > 0 && rd > 0) targetSpaces.Add(MakeSpace(s.x, s.y, s.z + d, w, s.h, rd));
            if (rw > 0 && s.h > 0 && s.d > 0) targetSpaces.Add(MakeSpace(s.x + w, s.y, s.z, rw, s.h, s.d));
        } else if (best == 5) {
            if (w > 0 && h > 0 && rd > 0) targetSpaces.Add(MakeSpace(s.x, s.y, s.z + d, w, h, rd));
            if (w > 0 && rh > 0 && s.d > 0) targetSpaces.Add(MakeSpace(s.x, s.y + h, s.z, w, rh, s.d));
            if (rw > 0 && s.h > 0 && s.d > 0) targetSpaces.Add(MakeSpace(s.x + w, s.y, s.z, rw, s.h, s.d));
        }
    }

    static void Main()
    {
        string header = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(header)) return;
        string[] firstLine = header.Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        if (firstLine.Length < 4) return;
        
        int N = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        int H = int.Parse(firstLine[2]);
        int D = int.Parse(firstLine[3]);
        
        List<Box> boxes = new List<Box>(N);
        int readBoxes = 0;
        while (readBoxes < N)
        {
            string lineStr = Console.ReadLine();
            if (lineStr == null) break;
            if (string.IsNullOrWhiteSpace(lineStr)) continue;
            string[] line = lineStr.Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
            int w = int.Parse(line[0]);
            int h = int.Parse(line[1]);
            int d = int.Parse(line[2]);
            
            int max = Math.Max(w, Math.Max(h, d));
            int min = Math.Min(w, Math.Min(h, d));
            int mid = w + h + d - max - min;
            
            boxes.Add(new Box { Id = readBoxes, w = w, h = h, d = d, Volume = (long)w * h * d, MaxDim = max, MinDim = min, MidDim = mid });
            readBoxes++;
        }
        
        boxes.Sort((a, b) => a.Volume.CompareTo(b.Volume));
        
        List<Space> spaces = new List<Space>(8192);
        spaces.Add(MakeSpace(0, 0, 0, W, H, D));
        
        List<PlacedBox> placedBoxes = new List<PlacedBox>(N);
        
        Stopwatch sw = Stopwatch.StartNew();
        
        foreach (var box in boxes)
        {
            if (sw.ElapsedMilliseconds > 28000) break;
            
            int newCount = 0;
            long bestSpaceVolume = long.MaxValue;
            int bestSpaceIndex = -1;
            int bestSpaceX = int.MaxValue;
            int bestSpaceY = int.MaxValue;
            int bestSpaceZ = int.MaxValue;
            int bestW = 0, bestH = 0, bestD = 0;
            
            int maxBox = box.MaxDim;
            int minBox = box.MinDim;
            int midBox = box.MidDim;
            
            for (int i = 0; i < spaces.Count; i++)
            {
                var s = spaces[i];
                if (s.Volume < box.Volume) continue;
                
                bool potentialBest = false;
                if (s.Volume < bestSpaceVolume) potentialBest = true;
                else if (s.Volume == bestSpaceVolume) {
                    if (s.z < bestSpaceZ) potentialBest = true;
                    else if (s.z == bestSpaceZ && s.y < bestSpaceY) potentialBest = true;
                    else if (s.z == bestSpaceZ && s.y == bestSpaceY && s.x < bestSpaceX) potentialBest = true;
                }
                
                if (potentialBest)
                {
                    int maxSpace = Math.Max(s.w, Math.Max(s.h, s.d));
                    int minSpace = Math.Min(s.w, Math.Min(s.h, s.d));
                    int midSpace = s.w + s.h + s.d - maxSpace - minSpace;
                    
                    if (maxBox <= maxSpace && minBox <= minSpace && midBox <= midSpace)
                    {
                        if (CanFit(box, s, out int fw, out int fh, out int fd))
                        {
                            bestSpaceVolume = s.Volume;
                            bestSpaceIndex = newCount;
                            bestSpaceX = s.x;
                            bestSpaceY = s.y;
                            bestSpaceZ = s.z;
                            bestW = fw; bestH = fh; bestD = fd;
                        }
                    }
                }
                
                spaces[newCount++] = s;
            }
            spaces.RemoveRange(newCount, spaces.Count - newCount);
            
            if (bestSpaceIndex != -1)
            {
                var s = spaces[bestSpaceIndex];
                spaces.RemoveAt(bestSpaceIndex);
                
                placedBoxes.Add(new PlacedBox { Id = box.Id, x = s.x, y = s.y, z = s.z });
                
                SplitSpace(s, bestW, bestH, bestD, spaces);
            }
        }
        
        Console.WriteLine(placedBoxes.Count);
        foreach (var pb in placedBoxes)
        {
            Console.WriteLine($"{pb.Id} {pb.x} {pb.y} {pb.z}");
        }
    }
}
