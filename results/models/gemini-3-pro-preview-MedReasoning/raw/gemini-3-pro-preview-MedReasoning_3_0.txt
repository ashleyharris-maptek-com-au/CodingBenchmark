use std::io::{self, Read, Write};
use std::time::Instant;

#[derive(Default)]
struct QuadNode {
    center_x: f32,
    center_y: f32,
    mass: f32,
    width: f32,
    node_id: usize,
    children: [usize; 4],
}

fn get_components(n: usize, adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let mut visited = vec![false; n];
    let mut components = Vec::new();
    for i in 0..n {
        if !visited[i] {
            let mut comp = Vec::new();
            let mut q = vec![i];
            visited[i] = true;
            let mut head = 0;
            while head < q.len() {
                let u = q[head];
                head += 1;
                comp.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        q.push(v);
                    }
                }
            }
            components.push(comp);
        }
    }
    components
}

fn build_tree(
    tree: &mut Vec<QuadNode>,
    nodes: &mut [usize],
    x: &[f32],
    y: &[f32],
    min_x: f32,
    max_x: f32,
    min_y: f32,
    max_y: f32,
) -> usize {
    let idx = tree.len();
    tree.push(QuadNode {
        center_x: 0.0,
        center_y: 0.0,
        mass: nodes.len() as f32,
        width: (max_x - min_x).max(max_y - min_y),
        node_id: usize::MAX,
        children: [usize::MAX; 4],
    });

    if nodes.is_empty() {
        tree[idx].mass = 0.0;
        return idx;
    }

    if nodes.len() == 1 {
        let n = nodes[0];
        tree[idx].center_x = x[n];
        tree[idx].center_y = y[n];
        tree[idx].node_id = n;
        return idx;
    }

    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut all_same = true;
    let first_x = x[nodes[0]];
    let first_y = y[nodes[0]];

    for &n in nodes.iter() {
        cx += x[n];
        cy += y[n];
        if (x[n] - first_x).abs() > 1e-5 || (y[n] - first_y).abs() > 1e-5 {
            all_same = false;
        }
    }

    if all_same {
        tree[idx].center_x = cx / nodes.len() as f32;
        tree[idx].center_y = cy / nodes.len() as f32;
        tree[idx].node_id = nodes[0];
        return idx;
    }

    tree[idx].center_x = cx / nodes.len() as f32;
    tree[idx].center_y = cy / nodes.len() as f32;

    let mut mid_x = (min_x + max_x) / 2.0;
    let mut mid_y = (min_y + max_y) / 2.0;
    if mid_x <= min_x || mid_x >= max_x { mid_x = min_x + 1e-5; }
    if mid_y <= min_y || mid_y >= max_y { mid_y = min_y + 1e-5; }

    let mut i = 0;
    let mut j = nodes.len();
    while i < j {
        if x[nodes[i]] >= mid_x {
            j -= 1;
            nodes.swap(i, j);
        } else {
            i += 1;
        }
    }
    let mid_lr = i;

    let mut i = 0;
    let mut j = mid_lr;
    while i < j {
        if y[nodes[i]] >= mid_y {
            j -= 1;
            nodes.swap(i, j);
        } else {
            i += 1;
        }
    }
    let mid_q02 = i;

    let mut i = mid_lr;
    let mut j = nodes.len();
    while i < j {
        if y[nodes[i]] >= mid_y {
            j -= 1;
            nodes.swap(i, j);
        } else {
            i += 1;
        }
    }
    let mid_q13 = i;

    let (left, right) = nodes.split_at_mut(mid_lr);
    let (q0, q2) = left.split_at_mut(mid_q02);
    let (q1, q3) = right.split_at_mut(mid_q13 - mid_lr);

    let c0 = build_tree(tree, q0, x, y, min_x, mid_x, min_y, mid_y);
    let c1 = build_tree(tree, q1, x, y, mid_x, max_x, min_y, mid_y);
    let c2 = build_tree(tree, q2, x, y, min_x, mid_x, mid_y, max_y);
    let c3 = build_tree(tree, q3, x, y, mid_x, max_x, mid_y, max_y);

    tree[idx].children = [c0, c1, c2, c3];
    idx
}

fn compute_repulsion(
    u: usize,
    node_idx: usize,
    tree: &[QuadNode],
    x: &[f32],
    y: &[f32],
    disp_x: &mut f32,
    disp_y: &mut f32,
    theta: f32,
) {
    let node = &tree[node_idx];
    if node.mass == 0.0 {
        return;
    }
    if node.node_id == u {
        return;
    }

    let dx = x[u] - node.center_x;
    let dy = y[u] - node.center_y;
    let mut dist_sq = dx * dx + dy * dy;

    if dist_sq < 1e-6 {
        dist_sq = 1e-6;
    }

    let dist = dist_sq.sqrt();

    if node.children[0] == usize::MAX || node.width / dist < theta {
        let f = node.mass / dist_sq;
        *disp_x += f * dx;
        *disp_y += f * dy;
    } else {
        for &child in &node.children {
            if child != usize::MAX {
                compute_repulsion(u, child, tree, x, y, disp_x, disp_y, theta);
            }
        }
    }
}

fn layout_component(
    comp: &[usize],
    adj: &[Vec<usize>],
    x: &mut [f32],
    y: &mut [f32],
    max_time: &Instant,
    global_to_local: &mut [usize],
) {
    let nc = comp.len();
    if nc == 1 {
        x[comp[0]] = 0.0;
        y[comp[0]] = 0.0;
        return;
    }
    if nc == 2 {
        x[comp[0]] = -0.5;
        y[comp[0]] = 0.0;
        x[comp[1]] = 0.5;
        y[comp[1]] = 0.0;
        return;
    }

    for (idx, &u) in comp.iter().enumerate() {
        global_to_local[u] = idx;
    }

    let mut d = vec![0.0; nc];
    let mut comp_adj = vec![Vec::new(); nc];
    for (idx, &u) in comp.iter().enumerate() {
        let mut degree = 0.0;
        for &v in &adj[u] {
            let v_idx = global_to_local[v];
            if v_idx != usize::MAX {
                comp_adj[idx].push(v_idx);
                degree += 1.0;
            }
        }
        d[idx] = degree.max(1.0);
    }

    let mut cx = vec![0.0; nc];
    let mut cy = vec![0.0; nc];

    for i in 0..nc {
        cx[i] = ((i * 137) % 1000) as f32 / 1000.0 * 2.0 - 1.0;
        cy[i] = ((i * 317) % 1000) as f32 / 1000.0 * 2.0 - 1.0;
    }

    let spectral_iters = if nc > 50000 {
        100
    } else if nc > 10000 {
        200
    } else {
        300
    };

    for _ in 0..spectral_iters {
        if max_time.elapsed().as_secs_f32() > 28.5 {
            break;
        }
        let mut nx = vec![0.0; nc];
        let mut ny = vec![0.0; nc];
        for i in 0..nc {
            let mut sx = 0.0;
            let mut sy = 0.0;
            for &j in &comp_adj[i] {
                sx += cx[j];
                sy += cy[j];
            }
            nx[i] = 0.5 * cx[i] + 0.5 * sx / d[i];
            ny[i] = 0.5 * cy[i] + 0.5 * sy / d[i];
        }

        let mut sum_d = 0.0;
        let mut sum_dx = 0.0;
        for i in 0..nc {
            sum_d += d[i];
            sum_dx += d[i] * nx[i];
        }
        let mean_x = sum_dx / sum_d;
        let mut norm_x = 0.0;
        for i in 0..nc {
            nx[i] -= mean_x;
            norm_x += d[i] * nx[i] * nx[i];
        }
        norm_x = norm_x.sqrt();
        if norm_x > 1e-6 {
            for i in 0..nc { nx[i] /= norm_x; }
        } else {
            for i in 0..nc {
                nx[i] = ((i * 137) % 1000) as f32 / 1000.0 * 2.0 - 1.0;
            }
        }

        let mut sum_dy = 0.0;
        for i in 0..nc {
            sum_dy += d[i] * ny[i];
        }
        let mean_y = sum_dy / sum_d;
        for i in 0..nc {
            ny[i] -= mean_y;
        }

        let mut dot = 0.0;
        for i in 0..nc {
            dot += d[i] * ny[i] * nx[i];
        }
        let mut norm_y = 0.0;
        for i in 0..nc {
            ny[i] -= dot * nx[i];
            norm_y += d[i] * ny[i] * ny[i];
        }
        norm_y = norm_y.sqrt();
        if norm_y > 1e-6 {
            for i in 0..nc { ny[i] /= norm_y; }
        } else {
            for i in 0..nc {
                ny[i] = ((i * 317) % 1000) as f32 / 1000.0 * 2.0 - 1.0;
            }
        }

        cx = nx;
        cy = ny;
    }

    let ideal_area = nc as f32;
    let side = ideal_area.sqrt();
    let mut min_x = f32::MAX;
    let mut max_x = f32::MIN;
    let mut min_y = f32::MAX;
    let mut max_y = f32::MIN;
    for i in 0..nc {
        if cx[i] < min_x { min_x = cx[i]; }
        if cx[i] > max_x { max_x = cx[i]; }
        if cy[i] < min_y { min_y = cy[i]; }
        if cy[i] > max_y { max_y = cy[i]; }
    }
    let wx = (max_x - min_x).max(1e-6);
    let wy = (max_y - min_y).max(1e-6);

    for i in 0..nc {
        cx[i] = (cx[i] - min_x) / wx * side - side / 2.0;
        cy[i] = (cy[i] - min_y) / wy * side - side / 2.0;
        cx[i] += ((i as f32 * 0.123).fract() - 0.5) * 1e-3;
        cy[i] += ((i as f32 * 0.321).fract() - 0.5) * 1e-3;
    }

    let bh_iters = if nc > 50000 { 150 } else { 300 };
    let t_init = side / 2.0;

    let mut disp_x = vec![0.0; nc];
    let mut disp_y = vec![0.0; nc];

    let mut edges = Vec::new();
    for i in 0..nc {
        for &j in &comp_adj[i] {
            if i < j {
                edges.push((i, j));
            }
        }
    }

    let mut nodes_idx: Vec<usize> = (0..nc).collect();
    let mut tree = Vec::with_capacity(nc * 4);

    for iter in 0..bh_iters {
        if max_time.elapsed().as_secs_f32() > 28.5 {
            break;
        }

        for i in 0..nc {
            disp_x[i] = 0.0;
            disp_y[i] = 0.0;
        }

        let mut min_x = f32::MAX;
        let mut max_x = f32::MIN;
        let mut min_y = f32::MAX;
        let mut max_y = f32::MIN;
        for i in 0..nc {
            if cx[i].is_nan() { cx[i] = 0.0; }
            if cy[i].is_nan() { cy[i] = 0.0; }
            if cx[i] < min_x { min_x = cx[i]; }
            if cx[i] > max_x { max_x = cx[i]; }
            if cy[i] < min_y { min_y = cy[i]; }
            if cy[i] > max_y { max_y = cy[i]; }
        }
        if max_x - min_x < 1e-5 { max_x += 1e-5; min_x -= 1e-5; }
        if max_y - min_y < 1e-5 { max_y += 1e-5; min_y -= 1e-5; }

        tree.clear();
        build_tree(
            &mut tree,
            &mut nodes_idx,
            &cx,
            &cy,
            min_x,
            max_x,
            min_y,
            max_y,
        );

        for i in 0..nc {
            let mut dx = 0.0;
            let mut dy = 0.0;
            compute_repulsion(i, 0, &tree, &cx, &cy, &mut dx, &mut dy, 1.2);
            disp_x[i] += dx;
            disp_y[i] += dy;
        }

        for &(u, v) in &edges {
            let dx = cx[v] - cx[u];
            let dy = cy[v] - cy[u];
            let dist_sq = dx * dx + dy * dy;
            let dist = dist_sq.sqrt().max(1e-6);

            let f_disp_x = dist * dx;
            let f_disp_y = dist * dy;

            disp_x[u] += f_disp_x;
            disp_y[u] += f_disp_y;
            disp_x[v] -= f_disp_x;
            disp_y[v] -= f_disp_y;
        }

        let t = t_init * (1.0 - iter as f32 / bh_iters as f32).max(0.01);

        for i in 0..nc {
            let dx = disp_x[i];
            let dy = disp_y[i];
            if dx.is_nan() || dy.is_nan() { continue; }

            let len = (dx * dx + dy * dy).sqrt().max(1e-6);
            if len > t {
                cx[i] += (dx / len) * t;
                cy[i] += (dy / len) * t;
            } else {
                cx[i] += dx;
                cy[i] += dy;
            }
        }
    }

    for (idx, &u) in comp.iter().enumerate() {
        x[u] = cx[idx];
        y[u] = cy[idx];
    }

    for &u in comp {
        global_to_local[u] = usize::MAX;
    }
}

fn pack_components(components: &[Vec<usize>], x: &mut [f32], y: &mut [f32]) {
    let mut comps = components.iter().collect::<Vec<_>>();
    comps.sort_by_key(|c| -(c.len() as isize));

    let mut current_x = 0.0;
    let mut current_y = 0.0;
    let mut row_height = 0.0;
    let mut max_width = 0.0;
    for c in &comps {
        max_width += c.len() as f32;
    }
    let target_width = max_width.sqrt() * 1.5;

    for c in comps {
        if c.is_empty() {
            continue;
        }
        let mut min_x = f32::MAX;
        let mut max_x = f32::MIN;
        let mut min_y = f32::MAX;
        let mut max_y = f32::MIN;
        for &u in c.iter() {
            if x[u] < min_x { min_x = x[u]; }
            if x[u] > max_x { max_x = x[u]; }
            if y[u] < min_y { min_y = y[u]; }
            if y[u] > max_y { max_y = y[u]; }
        }
        let w = (max_x - min_x).max(1.0);
        let h = (max_y - min_y).max(1.0);

        if current_x + w > target_width && current_x > 0.0 {
            current_x = 0.0;
            current_y += row_height + 2.0;
            row_height = 0.0;
        }

        let offset_x = current_x - min_x;
        let offset_y = current_y - min_y;

        for &u in c.iter() {
            x[u] += offset_x;
            y[u] += offset_y;
        }

        current_x += w + 2.0;
        if h > row_height {
            row_height = h;
        }
    }
}

fn main() {
    let start_time = Instant::now();
    let mut stdin = io::stdin();
    let mut buf = String::new();
    if stdin.read_to_string(&mut buf).is_err() {
        return;
    }
    let mut tokens = buf.split_whitespace();

    let n_str = tokens.next();
    if n_str.is_none() {
        return;
    }
    let n: usize = n_str.unwrap().parse().unwrap();
    let m_str = tokens.next();
    if m_str.is_none() {
        return;
    }
    let m: usize = m_str.unwrap().parse().unwrap();

    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        if let (Some(u_str), Some(v_str)) = (tokens.next(), tokens.next()) {
            if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                if u != v {
                    adj[u].push(v);
                    adj[v].push(u);
                }
            }
        }
    }

    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let components = get_components(n, &adj);
    let mut x = vec![0.0; n];
    let mut y = vec![0.0; n];

    let mut global_to_local = vec![usize::MAX; n];

    for comp in &components {
        layout_component(&comp, &adj, &mut x, &mut y, &start_time, &mut global_to_local);
    }

    pack_components(&components, &mut x, &mut y);

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    for i in 0..n {
        let _ = writeln!(out, "{:.6} {:.6}", x[i], y[i]);
    }
}
