class FastIO {
    static const int BUF_SIZE = 1 << 26; // 64 MB
    char* buf;
    int pos, len;
public:
    FastIO() : pos(0), len(0) {
        buf = new char[BUF_SIZE];
    }
    ~FastIO() {
        delete[] buf;
    }
    inline char nextChar() {
        if (pos >= len) {
            pos = 0;
            len = fread(buf, 1, BUF_SIZE, stdin);
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }
    inline bool readUint64(uint64_t &x) {
        char c;
        while ((c = nextChar()) != EOF && c <= ' ');
        if (c == EOF) return false;
        x = 0;
        do {
            x = x * 10 + (c - '0');
        } while ((c = nextChar()) != EOF && c > ' ');
        return true;
    }
    inline bool readUint32(uint32_t &x) {
        char c;
        while ((c = nextChar()) != EOF && c <= ' ');
        if (c == EOF) return false;
        x = 0;
        do {
            x = x * 10 + (c - '0');
        } while ((c = nextChar()) != EOF && c > ' ');
        return true;
    }
};

struct Edge {
    uint32_t u, v;
};

struct KargerDSU {
    vector<int> parent;
    int components;
    KargerDSU(int n) {
        parent.resize(n);
        for(int i=0; i<n; ++i) parent[i] = i;
        components = n;
    }
    inline int find(int i) {
        int root = i;
        while (root != parent[root]) root = parent[root];
        int curr = i;
        while (curr != root) {
            int nxt = parent[curr];
            parent[curr] = root;
            curr = nxt;
        }
        return root;
    }
    inline bool unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
            components--;
            return true;
        }
        return false;
    }
};

int run_karger(int N, const vector<Edge>& edges, vector<int>& best_cut_set, double time_limit_sec) {
    auto start = chrono::steady_clock::now();
    int min_cut = INT32_MAX;
    
    mt19937 rng(1337);
    uniform_int_distribution<int> dist(0, (int)edges.size() - 1);
    
    while (true) {
        auto now = chrono::steady_clock::now();
        chrono::duration<double> elapsed = now - start;
        if (elapsed.count() > time_limit_sec) break;
        
        KargerDSU dsu(N);
        int steps = 0;
        bool timeout = false;
        
        while (dsu.components > 2) {
            int idx = dist(rng);
            dsu.unite(edges[idx].u, edges[idx].v);
            if ((++steps & 0xFFFFF) == 0) {
                now = chrono::steady_clock::now();
                elapsed = now - start;
                if (elapsed.count() > time_limit_sec) {
                    timeout = true;
                    break;
                }
            }
        }
        
        if (timeout) break;
        
        for(int i=0; i<N; ++i) dsu.parent[i] = dsu.find(i);
        
        int cut_size = 0;
        for (const auto& e : edges) {
            if (dsu.parent[e.u] != dsu.parent[e.v]) {
                cut_size++;
            }
        }
        
        if (cut_size < min_cut) {
            min_cut = cut_size;
            best_cut_set.clear();
            int comp = dsu.parent[0];
            for (int i = 0; i < N; ++i) {
                if (dsu.parent[i] == comp) {
                    best_cut_set.push_back(i);
                }
            }
        }
    }
    return min_cut;
}

uint64_t stoer_wagner(int n, vector<vector<uint64_t>>& mat, vector<int>& best_cut_set) {
    uint64_t min_cut = UINT64_MAX;
    vector<vector<int>> nodes(n);
    for (int i = 0; i < n; ++i) {
        nodes[i].push_back(i);
    }
    
    while (n > 1) {
        vector<uint64_t> w(n, 0);
        vector<bool> added(n, false);
        int prev = -1, curr = -1;
        for (int i = 0; i < n; ++i) {
            uint64_t max_w = 0;
            int best_j = -1;
            for (int j = 0; j < n; ++j) {
                if (!added[j] && (best_j == -1 || w[j] >= max_w)) {
                    max_w = w[j];
                    best_j = j;
                }
            }
            if (best_j == -1) break;
            if (i == n - 1) {
                if (max_w < min_cut) {
                    min_cut = max_w;
                    best_cut_set = nodes[best_j];
                }
                
                for (int j = 0; j < n; ++j) {
                    if (j != best_j && j != prev) {
                        mat[prev][j] += mat[best_j][j];
                        mat[j][prev] = mat[prev][j];
                    }
                }
                mat[prev][prev] = 0;
                nodes[prev].insert(nodes[prev].end(), nodes[best_j].begin(), nodes[best_j].end());
                
                if (best_j != n - 1) {
                    for (int j = 0; j < n; ++j) {
                        mat[best_j][j] = mat[n - 1][j];
                        mat[j][best_j] = mat[j][n - 1];
                    }
                    nodes[best_j] = std::move(nodes[n - 1]);
                }
                break;
            }
            added[best_j] = true;
            prev = curr;
            curr = best_j;
            for (int j = 0; j < n; ++j) {
                if (!added[j]) w[j] += mat[curr][j];
            }
        }
        n--;
    }
    return min_cut;
}

struct BridgeFinder {
    int N;
    const vector<int>& head;
    const vector<int>& to;
    const vector<int>& nxt;
    
    vector<int> tin, low;
    int timer;
    bool found;
    Edge bridge;
    
    BridgeFinder(int n, const vector<int>& h, const vector<int>& t, const vector<int>& nx)
        : N(n), head(h), to(t), nxt(nx), tin(n, -1), low(n, -1), timer(0), found(false) {}
        
    void run_iterative(int start_node) {
        struct Frame {
            int v;
            int p_edge;
            int e;
            int state;
            int to_node;
        };
        vector<Frame> stack;
        stack.reserve(N);
        
        stack.push_back({start_node, -1, head[start_node], 0, -1});
        tin[start_node] = low[start_node] = timer++;
        
        while (!stack.empty()) {
            auto& f = stack.back();
            if (f.state == 0) {
                if (f.e != -1) {
                    int e = f.e;
                    int to_node = to[e];
                    f.e = nxt[e];
                    
                    if ((e ^ 1) == f.p_edge) continue;
                    
                    if (tin[to_node] != -1) {
                        low[f.v] = min(low[f.v], tin[to_node]);
                    } else {
                        f.state = 1;
                        f.to_node = to_node;
                        tin[to_node] = low[to_node] = timer++;
                        stack.push_back({to_node, e, head[to_node], 0, -1});
                    }
                } else {
                    stack.pop_back();
                }
            } else if (f.state == 1) {
                if (found) return;
                low[f.v] = min(low[f.v], low[f.to_node]);
                if (low[f.to_node] > tin[f.v]) {
                    found = true;
                    bridge = {(uint32_t)f.v, (uint32_t)f.to_node};
                    return;
                }
                f.state = 0;
            }
        }
    }
};

int main() {
#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
#endif
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    FastIO fio;
    
    uint64_t N_val, M_val;
    if (!fio.readUint64(N_val)) return 0;
    if (!fio.readUint64(M_val)) return 0;
    
    if (N_val <= 1) {
        cout << "0\n";
        return 0;
    }
    
    uint32_t N = (uint32_t)N_val;
    uint64_t M = M_val;
    
    vector<uint32_t> degree(N, 0);
    
    bool use_disk = false;
    vector<Edge> edges;
    FILE* temp_file = nullptr;
    
    if (M > 1000000000ULL) { 
        use_disk = true;
    } else {
        try {
            edges.reserve(M);
        } catch (...) {
            use_disk = true;
        }
    }
    
    if (use_disk) {
        temp_file = fopen("temp_edges.bin", "wb+");
        if (!temp_file) {
            return 1; // Unlikely to reach unless disk fails
        }
    }
    
    for (uint64_t i = 0; i < M; ++i) {
        uint32_t u, v;
        fio.readUint32(u);
        fio.readUint32(v);
        degree[u]++;
        degree[v]++;
        if (!use_disk) {
            edges.push_back({u, v});
        } else {
            Edge e = {u, v};
            fwrite(&e, sizeof(Edge), 1, temp_file);
        }
    }
    
    uint32_t min_node = 0;
    uint32_t min_deg = UINT32_MAX;
    for (uint32_t i = 0; i < N; ++i) {
        if (degree[i] < min_deg) {
            min_deg = degree[i];
            min_node = i;
        }
    }
    
    if (min_deg == 0) {
        cout << "0\n";
        if (use_disk) {
            fclose(temp_file);
            remove("temp_edges.bin");
        }
        return 0;
    }
    
    vector<Edge> final_cut;
    
    if (!use_disk && N <= 3000) {
        vector<vector<uint64_t>> mat(N, vector<uint64_t>(N, 0));
        for (const auto& e : edges) {
            mat[e.u][e.v]++;
            mat[e.v][e.u]++;
        }
        vector<int> best_cut_set;
        uint64_t sw_cut = stoer_wagner(N, mat, best_cut_set);
        
        vector<bool> in_set(N, false);
        for (int node : best_cut_set) in_set[node] = true;
        for (const auto& e : edges) {
            if (in_set[e.u] != in_set[e.v]) {
                final_cut.push_back(e);
            }
        }
    } else if (!use_disk && M <= 500000000ULL) {
        vector<int> head(N, -1), to, nxt;
        try {
            to.reserve(M * 2);
            nxt.reserve(M * 2);
            for (const auto& e : edges) {
                to.push_back(e.v);
                nxt.push_back(head[e.u]);
                head[e.u] = to.size() - 1;
                
                to.push_back(e.u);
                nxt.push_back(head[e.v]);
                head[e.v] = to.size() - 1;
            }
            
            BridgeFinder bf(N, head, to, nxt);
            bf.run_iterative(0);
            
            if (bf.found) {
                final_cut.push_back(bf.bridge);
            } else if (min_deg == 2) {
                for (const auto& e : edges) {
                    if (e.u == min_node || e.v == min_node) {
                        final_cut.push_back(e);
                    }
                }
            } else {
                vector<int> best_cut_set;
                int k_cut = run_karger(N, edges, best_cut_set, 1.5);
                if (k_cut < (int)min_deg) {
                    vector<bool> in_set(N, false);
                    for (int node : best_cut_set) in_set[node] = true;
                    for (const auto& e : edges) {
                        if (in_set[e.u] != in_set[e.v]) {
                            final_cut.push_back(e);
                        }
                    }
                } else {
                    for (const auto& e : edges) {
                        if (e.u == min_node || e.v == min_node) {
                            final_cut.push_back(e);
                        }
                    }
                }
            }
        } catch (...) {
            for (const auto& e : edges) {
                if (e.u == min_node || e.v == min_node) {
                    final_cut.push_back(e);
                }
            }
        }
    } else {
        if (use_disk) {
            fseek(temp_file, 0, SEEK_SET);
            Edge e;
            while (fread(&e, sizeof(Edge), 1, temp_file) == 1) {
                if (e.u == min_node || e.v == min_node) {
                    final_cut.push_back(e);
                }
            }
            fclose(temp_file);
            remove("temp_edges.bin");
        } else {
            for (const auto& e : edges) {
                if (e.u == min_node || e.v == min_node) {
                    final_cut.push_back(e);
                }
            }
        }
    }
    
    cout << final_cut.size() << "\n";
    for (const auto& e : final_cut) {
        cout << e.u << " " << e.v << "\n";
    }
    
    return 0;
}
