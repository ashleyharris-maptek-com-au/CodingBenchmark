use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Clone, Debug)]
struct Sample {
    hole_idx: u32,
    sample_idx: u32,
    pos: [f32; 3],
}

struct Outlier {
    hole_idx: u32,
    sample_idx: u32,
    prop_idx: usize,
    confidence: f32,
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();

    while reader.read_line(&mut line).unwrap_or(0) > 0 {
        if !line.trim().is_empty() {
            break;
        }
        line.clear();
    }
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 2 { return; }
    let num_holes: usize = parts[0].parse().unwrap_or(0);
    let num_properties: usize = parts[1].parse().unwrap_or(0);

    line.clear();
    while reader.read_line(&mut line).unwrap_or(0) > 0 {
        if !line.trim().is_empty() {
            break;
        }
        line.clear();
    }
    let mut property_names: Vec<String> = line.split_whitespace().map(|s| s.to_string()).collect();
    while property_names.len() < num_properties {
        property_names.push(format!("prop_{}", property_names.len()));
    }

    let mut samples: Vec<Sample> = Vec::new();
    let mut all_props: Vec<f32> = Vec::new();
    let mut hole_names: Vec<String> = Vec::new();

    let mut hole_idx = 0;
    while hole_idx < num_holes {
        line.clear();
        if reader.read_line(&mut line).unwrap_or(0) == 0 { break; }
        if line.trim().is_empty() { continue; }

        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() < 9 { continue; }
        
        let hole_id = parts[0].to_string();
        hole_names.push(hole_id);
        
        let start_x: f32 = parts[1].parse().unwrap_or(0.0);
        let start_y: f32 = parts[2].parse().unwrap_or(0.0);
        let start_z: f32 = parts[3].parse().unwrap_or(0.0);
        let dir_x: f32 = parts[4].parse().unwrap_or(0.0);
        let dir_y: f32 = parts[5].parse().unwrap_or(0.0);
        let dir_z: f32 = parts[6].parse().unwrap_or(0.0);
        let num_samples: usize = parts[8].parse().unwrap_or(0);

        let len = (dir_x*dir_x + dir_y*dir_y + dir_z*dir_z).sqrt();
        let (nx, ny, nz) = if len > 1e-9 {
            (dir_x/len, dir_y/len, dir_z/len)
        } else {
            (0.0, 0.0, 0.0)
        };

        let mut s_idx = 0;
        while s_idx < num_samples {
            line.clear();
            if reader.read_line(&mut line).unwrap_or(0) == 0 { break; }
            if line.trim().is_empty() { continue; }

            let sparts: Vec<&str> = line.split_whitespace().collect();
            if sparts.is_empty() { continue; }
            
            let depth: f32 = sparts[0].parse().unwrap_or(0.0);
            let pos = [
                start_x + nx * depth,
                start_y + ny * depth,
                start_z + nz * depth,
            ];
            
            samples.push(Sample {
                hole_idx: hole_idx as u32,
                sample_idx: s_idx as u32,
                pos,
            });
            
            for i in 0..num_properties {
                if 1 + i < sparts.len() {
                    all_props.push(sparts[1 + i].parse().unwrap_or(f32::NAN));
                } else {
                    all_props.push(f32::NAN);
                }
            }
            s_idx += 1;
        }
        hole_idx += 1;
    }

    if samples.is_empty() {
        println!("0");
        return;
    }

    let mut outliers = Vec::new();
    let n = samples.len();
    let num_threads = 12;

    if n < 5000 {
        let chunk_size = (n + num_threads - 1) / num_threads;
        let mut thread_results = Vec::new();

        std::thread::scope(|s| {
            let mut handles = Vec::new();
            
            for t in 0..num_threads {
                let start_idx = t * chunk_size;
                if start_idx >= n { break; }
                let end_idx = ((t + 1) * chunk_size).min(n);
                
                let samples_ref = &samples;
                let all_props_ref = &all_props;
                
                let handle = s.spawn(move || {
                    let mut local_outliers = Vec::new();
                    for i in start_idx..end_idx {
                        let s_i = &samples_ref[i];
                        for p in 0..num_properties {
                            let val_i = all_props_ref[i * num_properties + p];
                            if val_i.is_nan() { continue; }
                            let mut neighbors = Vec::new();
                            for j in 0..n {
                                if i == j { continue; }
                                let val_j = all_props_ref[j * num_properties + p];
                                if val_j.is_nan() { continue; }
                                let dist_sq = (s_i.pos[0] - samples_ref[j].pos[0]).powi(2)
                                            + (s_i.pos[1] - samples_ref[j].pos[1]).powi(2)
                                            + (s_i.pos[2] - samples_ref[j].pos[2]).powi(2);
                                neighbors.push((dist_sq, val_j));
                            }
                            
                            if neighbors.len() < 3 { continue; }
                            neighbors.sort_unstable_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
                            let k = 10.min(neighbors.len());
                            
                            let mut sum = 0.0;
                            for j in 0..k { sum += neighbors[j].1; }
                            let mean = sum / k as f32;
                            
                            let mut var = 0.0;
                            for j in 0..k { var += (neighbors[j].1 - mean).powi(2); }
                            var /= k as f32;
                            let std_dev = var.sqrt();
                            
                            if std_dev > 1e-6 {
                                let z_score = (val_i - mean).abs() / std_dev;
                                if z_score > 3.0 {
                                    local_outliers.push(Outlier {
                                        hole_idx: s_i.hole_idx,
                                        sample_idx: s_i.sample_idx,
                                        prop_idx: p,
                                        confidence: z_score,
                                    });
                                }
                            } else if (val_i - mean).abs() > 1e-3 {
                                local_outliers.push(Outlier {
                                    hole_idx: s_i.hole_idx,
                                    sample_idx: s_i.sample_idx,
                                    prop_idx: p,
                                    confidence: 999.0,
                                });
                            }
                        }
                    }
                    local_outliers
                });
                handles.push(handle);
            }
            
            for handle in handles {
                thread_results.push(handle.join().unwrap());
            }
        });
        
        for mut local_outliers in thread_results {
            outliers.append(&mut local_outliers);
        }
    } else {
        let mut min_pos = [f32::MAX; 3];
        let mut max_pos = [f32::MIN; 3];
        for s in &samples {
            for i in 0..3 {
                if s.pos[i] < min_pos[i] { min_pos[i] = s.pos[i]; }
                if s.pos[i] > max_pos[i] { max_pos[i] = s.pos[i]; }
            }
        }
        
        let target_samples_per_cell = 20.0;
        let total_cells = (n as f32) / target_samples_per_cell;
        let mut grid_cells = total_cells.powf(1.0/3.0).ceil() as i32;
        if grid_cells < 1 { grid_cells = 1; }
        
        let mut cell_size = [0.0; 3];
        for i in 0..3 {
            cell_size[i] = (max_pos[i] - min_pos[i]) / grid_cells as f32;
            if cell_size[i] < 1e-5 { cell_size[i] = 1.0; }
        }
        
        let mut grid: HashMap<[i32; 3], Vec<usize>> = HashMap::new();
        for (idx, s) in samples.iter().enumerate() {
            let cx = ((s.pos[0] - min_pos[0]) / cell_size[0]).floor() as i32;
            let cy = ((s.pos[1] - min_pos[1]) / cell_size[1]).floor() as i32;
            let cz = ((s.pos[2] - min_pos[2]) / cell_size[2]).floor() as i32;
            grid.entry([cx, cy, cz]).or_insert_with(Vec::new).push(idx);
        }
        
        let chunk_size = (n + num_threads - 1) / num_threads;
        let mut thread_results = Vec::new();

        std::thread::scope(|s| {
            let mut handles = Vec::new();
            
            for t in 0..num_threads {
                let start_idx = t * chunk_size;
                if start_idx >= n { break; }
                let end_idx = ((t + 1) * chunk_size).min(n);
                
                let grid_ref = &grid;
                let samples_ref = &samples;
                let all_props_ref = &all_props;
                let min_pos_ref = &min_pos;
                let cell_size_ref = &cell_size;
                
                let handle = s.spawn(move || {
                    let mut local_outliers = Vec::new();
                    for i in start_idx..end_idx {
                        let s_i = &samples_ref[i];
                        let cx = ((s_i.pos[0] - min_pos_ref[0]) / cell_size_ref[0]).floor() as i32;
                        let cy = ((s_i.pos[1] - min_pos_ref[1]) / cell_size_ref[1]).floor() as i32;
                        let cz = ((s_i.pos[2] - min_pos_ref[2]) / cell_size_ref[2]).floor() as i32;
                        
                        for p in 0..num_properties {
                            let val_i = all_props_ref[i * num_properties + p];
                            if val_i.is_nan() { continue; }
                            
                            let mut neighbors = Vec::new();
                            for dx in -1..=1 {
                                for dy in -1..=1 {
                                    for dz in -1..=1 {
                                        if let Some(cell_samples) = grid_ref.get(&[cx + dx, cy + dy, cz + dz]) {
                                            for &j in cell_samples {
                                                if i == j { continue; }
                                                let val_j = all_props_ref[j * num_properties + p];
                                                if val_j.is_nan() { continue; }
                                                let dist_sq = (s_i.pos[0] - samples_ref[j].pos[0]).powi(2)
                                                            + (s_i.pos[1] - samples_ref[j].pos[1]).powi(2)
                                                            + (s_i.pos[2] - samples_ref[j].pos[2]).powi(2);
                                                neighbors.push((dist_sq, val_j));
                                            }
                                        }
                                    }
                                }
                            }
                            
                            if neighbors.len() < 5 { continue; }
                            neighbors.sort_unstable_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
                            let k = 15.min(neighbors.len());
                            
                            let mut sum = 0.0;
                            for j in 0..k { sum += neighbors[j].1; }
                            let mean = sum / k as f32;
                            
                            let mut var = 0.0;
                            for j in 0..k { var += (neighbors[j].1 - mean).powi(2); }
                            var /= k as f32;
                            let std_dev = var.sqrt();
                            
                            if std_dev > 1e-6 {
                                let z_score = (val_i - mean).abs() / std_dev;
                                if z_score > 3.5 {
                                    local_outliers.push(Outlier {
                                        hole_idx: s_i.hole_idx,
                                        sample_idx: s_i.sample_idx,
                                        prop_idx: p,
                                        confidence: z_score,
                                    });
                                }
                            } else if (val_i - mean).abs() > 1e-3 {
                                local_outliers.push(Outlier {
                                    hole_idx: s_i.hole_idx,
                                    sample_idx: s_i.sample_idx,
                                    prop_idx: p,
                                    confidence: 999.0,
                                });
                            }
                        }
                    }
                    local_outliers
                });
                handles.push(handle);
            }
            
            for handle in handles {
                thread_results.push(handle.join().unwrap());
            }
        });
        
        for mut local_outliers in thread_results {
            outliers.append(&mut local_outliers);
        }
    }

    outliers.sort_unstable_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());
    outliers.truncate(100000);

    println!("{}", outliers.len());
    for o in outliers {
        println!("{} {} {} {:.4}", 
            hole_names[o.hole_idx as usize], 
            o.sample_idx, 
            property_names[o.prop_idx], 
            o.confidence
        );
    }
}
