use std::io::{self, BufRead, Write, BufWriter};
use std::collections::HashSet;

fn get_rotation_to_d(d: [f64; 3]) -> [f64; 4] {
    let mut half = [d[0], d[1], d[2] + 1.0];
    let len = (half[0] * half[0] + half[1] * half[1] + half[2] * half[2]).sqrt();
    if len < 1e-6 {
        return [0.0, 1.0, 0.0, 0.0];
    }
    half[0] /= len;
    half[1] /= len;
    half[2] /= len;
    [half[2], -half[1], half[0], 0.0]
}

fn quat_to_mat(q: [f64; 4]) -> [[f64; 3]; 3] {
    let (w, x, y, z) = (q[0], q[1], q[2], q[3]);
    [
        [1.0 - 2.0 * y * y - 2.0 * z * z, 2.0 * x * y - 2.0 * z * w, 2.0 * x * z + 2.0 * y * w],
        [2.0 * x * y + 2.0 * z * w, 1.0 - 2.0 * x * x - 2.0 * z * z, 2.0 * y * z - 2.0 * x * w],
        [2.0 * x * z - 2.0 * y * w, 2.0 * y * z + 2.0 * x * w, 1.0 - 2.0 * x * x - 2.0 * y * y],
    ]
}

#[derive(Clone, Copy)]
struct Point {
    u: f64,
    v: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(|l| l.ok()).filter(|l| !l.trim().is_empty());

    let nv_str = lines.next().expect("Missing NV");
    let nv: usize = nv_str.trim().parse().expect("Invalid NV");

    let mut vertices = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().expect("Missing vertex");
        let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        vertices.push((parts[0], parts[1]));
    }

    let sun_line = lines.next().expect("Missing sun vector");
    let sun_parts: Vec<f64> = sun_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let sun = [sun_parts[0], sun_parts[1], sun_parts[2]];

    let mut d = sun;
    let len = (d[0] * d[0] + d[1] * d[1] + d[2] * d[2]).sqrt();
    d[0] /= len;
    d[1] /= len;
    d[2] /= len;

    let dz = if d[2].abs() < 1e-6 {
        if d[2] < 0.0 { -1e-6 } else { 1e-6 }
    } else {
        d[2]
    };

    let q = get_rotation_to_d(d);
    let r = quat_to_mat(q);

    let u_vec = [r[0][0], r[1][0], r[2][0]];
    let v_vec = [r[0][1], r[1][1], r[2][1]];

    let u_prime = [u_vec[0] - (u_vec[2] / dz) * d[0], u_vec[1] - (u_vec[2] / dz) * d[1]];
    let v_prime = [v_vec[0] - (v_vec[2] / dz) * d[0], v_vec[1] - (v_vec[2] / dz) * d[1]];

    let a = u_prime;
    let b = v_prime;

    let t1 = [9.9 * (a[0] + b[0]), 9.9 * (a[1] + b[1])];
    let t2 = [9.9 * (a[0] - b[0]), 9.9 * (a[1] - b[1])];

    let det = t1[0] * t2[1] - t2[0] * t1[1];
    if det.abs() < 1e-12 {
        panic!("Degenerate transformation matrix");
    }

    let mut u_verts = Vec::with_capacity(nv);
    let mut min_u = f64::MAX;
    let mut max_u = f64::MIN;
    let mut min_v = f64::MAX;
    let mut max_v = f64::MIN;

    for &(px, py) in &vertices {
        let u = (px * t2[1] - py * t2[0]) / det;
        let v = (-px * t1[1] + py * t1[0]) / det;
        min_u = min_u.min(u);
        max_u = max_u.max(u);
        min_v = min_v.min(v);
        max_v = max_v.max(v);
        u_verts.push(Point { u, v });
    }

    let w = max_u - min_u;
    let h = max_v - min_v;
    let s_unit = 10.0 / 19.8;

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    // Adaptive logic: if polygon is small enough, cover with a single tetrahedron.
    if w <= 2.0 * s_unit && h <= 2.0 * s_unit {
        let mut s_scale = 19.8 * (w / 2.0).max(h / 2.0);
        s_scale *= 1.01;
        s_scale = s_scale.clamp(0.001, 10.0);

        let cu = (min_u + max_u) / 2.0;
        let cv = (min_v + max_v) / 2.0;

        let cx = cu * t1[0] + cv * t2[0];
        let cy = cu * t1[1] + cv * t2[1];

        let dz_abs = dz.abs().max(1e-6);
        let offset_k = 25.0 / dz_abs;
        let px = cx - offset_k * d[0];
        let py = cy - offset_k * d[1];
        let pz = 0.0 - offset_k * d[2];

        writeln!(writer, "1").unwrap();
        writeln!(
            writer,
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
            px, py, pz, q[0], q[1], q[2], q[3], s_scale
        )
        .unwrap();
        return;
    }

    let mut cells = HashSet::new();
    let s = 10.0 / 19.8;

    let mut edges = Vec::new();
    for i in 0..nv {
        edges.push((&u_verts[i], &u_verts[(i + 1) % nv]));
    }

    for &(p1, p2) in &edges {
        let du = p2.u - p1.u;
        let dv = p2.v - p1.v;
        let len = (du * du + dv * dv).sqrt();
        let steps = (len / 0.1).ceil() as usize;
        let steps = steps.max(1);

        for k in 0..=steps {
            let t = k as f64 / steps as f64;
            let pu = p1.u + t * du;
            let pv = p1.v + t * dv;

            let i_min = (pu - s).ceil() as i64;
            let i_max = (pu + s).floor() as i64;
            let j_min = (pv - s).ceil() as i64;
            let j_max = (pv + s).floor() as i64;

            for i in i_min..=i_max {
                for j in j_min..=j_max {
                    cells.insert((i, j));
                }
            }
        }
    }

    let j_start = min_v.floor() as i64 - 1;
    let j_end = max_v.ceil() as i64 + 1;

    for j in j_start..=j_end {
        let v_test = j as f64 + 0.5;
        let mut crossings = Vec::new();
        for &(p1, p2) in &edges {
            if (p1.v < v_test && p2.v >= v_test) || (p2.v < v_test && p1.v >= v_test) {
                let u = p1.u + (p2.u - p1.u) * (v_test - p1.v) / (p2.v - p1.v);
                crossings.push(u);
            }
        }

        crossings.sort_by(|a, b| a.partial_cmp(b).unwrap());

        for chunk in crossings.chunks(2) {
            if chunk.len() == 2 {
                let u1 = chunk[0];
                let u2 = chunk[1];
                let i_start = (u1 - s).ceil() as i64;
                let i_end = (u2 + s).floor() as i64;
                for i in i_start..=i_end {
                    cells.insert((i, j));
                }
            }
        }
    }

    writeln!(writer, "{}", cells.len()).unwrap();

    let dz_abs = dz.abs().max(1e-6);
    let offset_step = 25.0 / dz_abs;

    for (idx, &(i, j)) in cells.iter().enumerate() {
        let cx = i as f64 * t1[0] + j as f64 * t2[0];
        let cy = i as f64 * t1[1] + j as f64 * t2[1];

        let k = (idx as f64 + 1.0) * offset_step;
        let px = cx - k * d[0];
        let py = cy - k * d[1];
        let pz = 0.0 - k * d[2];

        writeln!(
            writer,
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
            px, py, pz, q[0], q[1], q[2], q[3], 10.0
        )
        .unwrap();
    }
}
