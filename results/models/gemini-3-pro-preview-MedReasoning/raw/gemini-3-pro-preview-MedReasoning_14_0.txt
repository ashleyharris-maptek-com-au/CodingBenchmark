using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static int H, W;
    static byte[] board;
    static int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

    static Queue<int> safeQueue = new Queue<int>();
    static Queue<int> mineQueue = new Queue<int>();
    static bool[] inActiveQueue;
    static Queue<int> activeQueue = new Queue<int>();
    static List<int> resultSafeCells = new List<int>();

    static int[] tempUnkA = new int[8];
    static int[] tempUnkB = new int[8];
    static List<int> boundaryNumbers = new List<int>();
    static int[] varId;
    static List<int> resetVarIds = new List<int>();

    static void Main()
    {
        FastScanner scanner = new FastScanner(Console.OpenStandardInput());
        H = scanner.NextInt();
        if (H == -1) return;
        W = scanner.NextInt();

        board = new byte[H * W];
        scanner.ReadBoard(board, H, W);

        inActiveQueue = new bool[H * W];
        varId = new int[H * W];
        Array.Fill(varId, -1);

        for (int idx = 0; idx < board.Length; idx++)
        {
            if (board[idx] <= 8)
            {
                EnqueueActive(idx);
            }
        }

        bool changed = true;
        while (changed)
        {
            changed = false;
            if (ProcessQueues()) changed = true;
            if (!changed)
            {
                if (PairwiseSubsets()) changed = true;
            }
            if (!changed)
            {
                if (SolveCSP()) changed = true;
            }
        }

        resultSafeCells.Sort((a, b) => {
            int ya = a / W, xa = a % W;
            int yb = b / W, xb = b % W;
            if (ya != yb) return ya.CompareTo(yb);
            return xa.CompareTo(xb);
        });

        Console.WriteLine(resultSafeCells.Count);
        foreach (int idx in resultSafeCells)
        {
            Console.WriteLine($"{idx % W} {idx / W}");
        }
    }

    static void EnqueueActive(int idx)
    {
        if (!inActiveQueue[idx])
        {
            inActiveQueue[idx] = true;
            activeQueue.Enqueue(idx);
        }
    }

    static void GetUnk(int idx, int[] unk, out int count, out int req)
    {
        count = 0;
        req = board[idx];
        int r = idx / W;
        int c = idx % W;
        for (int i = 0; i < 8; i++)
        {
            int nr = r + dr[i];
            int nc = c + dc[i];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
            {
                int nidx = nr * W + nc;
                if (board[nidx] == 10) req--;
                else if (board[nidx] == 9) unk[count++] = nidx;
            }
        }
    }

    static void WakeUpNeighbors(int idx)
    {
        int r = idx / W;
        int c = idx % W;
        for (int i = 0; i < 8; i++)
        {
            int nr = r + dr[i];
            int nc = c + dc[i];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
            {
                int nidx = nr * W + nc;
                if (board[nidx] <= 8)
                {
                    EnqueueActive(nidx);
                }
            }
        }
    }

    static bool ProcessQueues()
    {
        bool changed = false;
        while (activeQueue.Count > 0 || safeQueue.Count > 0 || mineQueue.Count > 0)
        {
            while (safeQueue.Count > 0)
            {
                int idx = safeQueue.Dequeue();
                if (board[idx] != 9) continue;
                board[idx] = 11;
                resultSafeCells.Add(idx);
                changed = true;
                WakeUpNeighbors(idx);
            }

            while (mineQueue.Count > 0)
            {
                int idx = mineQueue.Dequeue();
                if (board[idx] != 9) continue;
                board[idx] = 10;
                changed = true;
                WakeUpNeighbors(idx);
            }

            while (activeQueue.Count > 0)
            {
                int idx = activeQueue.Dequeue();
                inActiveQueue[idx] = false;

                if (board[idx] > 8) continue;

                GetUnk(idx, tempUnkA, out int count, out int req);
                if (count == 0) continue;

                if (req == 0)
                {
                    for (int i = 0; i < count; i++)
                    {
                        if (board[tempUnkA[i]] == 9) safeQueue.Enqueue(tempUnkA[i]);
                    }
                }
                else if (req == count)
                {
                    for (int i = 0; i < count; i++)
                    {
                        if (board[tempUnkA[i]] == 9) mineQueue.Enqueue(tempUnkA[i]);
                    }
                }
            }
        }
        return changed;
    }

    static void UpdateBoundaryNumbers()
    {
        boundaryNumbers.Clear();
        int idx = 0;
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[idx] <= 8)
                {
                    bool hasUnk = false;
                    for (int i = 0; i < 8; i++)
                    {
                        int nr = r + dr[i];
                        int nc = c + dc[i];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            if (board[nr * W + nc] == 9)
                            {
                                hasUnk = true;
                                break;
                            }
                        }
                    }
                    if (hasUnk) boundaryNumbers.Add(idx);
                }
                idx++;
            }
        }
    }

    static bool PairwiseSubsets()
    {
        bool changed = false;
        UpdateBoundaryNumbers();

        for (int i = 0; i < boundaryNumbers.Count; i++)
        {
            int idxA = boundaryNumbers[i];
            int rA = idxA / W;
            int cA = idxA % W;

            GetUnk(idxA, tempUnkA, out int countA, out int reqA);
            if (countA == 0) continue;

            for (int drB = -2; drB <= 2; drB++)
            {
                for (int dcB = -2; dcB <= 2; dcB++)
                {
                    if (drB == 0 && dcB == 0) continue;

                    int rB = rA + drB;
                    int cB = cA + dcB;

                    if (rB >= 0 && rB < H && cB >= 0 && cB < W)
                    {
                        int idxB = rB * W + cB;
                        if (board[idxB] <= 8)
                        {
                            GetUnk(idxB, tempUnkB, out int countB, out int reqB);
                            if (countB == 0) continue;

                            if (countA >= countB) continue;

                            bool isSubset = true;
                            for (int a = 0; a < countA; a++)
                            {
                                bool found = false;
                                for (int b = 0; b < countB; b++)
                                {
                                    if (tempUnkA[a] == tempUnkB[b])
                                    {
                                        found = true;
                                        break;
                                    }
                                }
                                if (!found)
                                {
                                    isSubset = false;
                                    break;
                                }
                            }

                            if (isSubset)
                            {
                                int diffReq = reqB - reqA;
                                int diffCount = countB - countA;

                                if (diffReq == 0)
                                {
                                    for (int b = 0; b < countB; b++)
                                    {
                                        bool inA = false;
                                        for (int a = 0; a < countA; a++)
                                        {
                                            if (tempUnkB[b] == tempUnkA[a]) { inA = true; break; }
                                        }
                                        if (!inA)
                                        {
                                            if (board[tempUnkB[b]] == 9)
                                            {
                                                safeQueue.Enqueue(tempUnkB[b]);
                                                changed = true;
                                            }
                                        }
                                    }
                                }
                                else if (diffReq == diffCount)
                                {
                                    for (int b = 0; b < countB; b++)
                                    {
                                        bool inA = false;
                                        for (int a = 0; a < countA; a++)
                                        {
                                            if (tempUnkB[b] == tempUnkA[a]) { inA = true; break; }
                                        }
                                        if (!inA)
                                        {
                                            if (board[tempUnkB[b]] == 9)
                                            {
                                                mineQueue.Enqueue(tempUnkB[b]);
                                                changed = true;
                                            }
                                        }
                                    }
                                }

                                if (changed) return true;
                            }
                        }
                    }
                }
            }
        }
        return changed;
    }

    static bool SolveCSP()
    {
        List<int> vars = new List<int>();
        List<int> constraints = boundaryNumbers;

        foreach (int cIdx in constraints)
        {
            GetUnk(cIdx, tempUnkA, out int count, out int req);
            for (int i = 0; i < count; i++)
            {
                int u = tempUnkA[i];
                if (varId[u] == -1)
                {
                    varId[u] = vars.Count;
                    vars.Add(u);
                    resetVarIds.Add(u);
                }
            }
        }

        if (vars.Count == 0) return false;

        List<int>[] adj = new List<int>[vars.Count];
        for (int i = 0; i < vars.Count; i++) adj[i] = new List<int>();

        foreach (int cIdx in constraints)
        {
            GetUnk(cIdx, tempUnkA, out int count, out int req);
            for (int i = 0; i < count; i++)
            {
                for (int j = i + 1; j < count; j++)
                {
                    int u1 = varId[tempUnkA[i]];
                    int u2 = varId[tempUnkA[j]];
                    adj[u1].Add(u2);
                    adj[u2].Add(u1);
                }
            }
        }

        bool[] visited = new bool[vars.Count];
        List<List<int>> components = new List<List<int>>();

        for (int i = 0; i < vars.Count; i++)
        {
            if (!visited[i])
            {
                List<int> comp = new List<int>();
                Queue<int> q = new Queue<int>();
                q.Enqueue(i);
                visited[i] = true;

                while (q.Count > 0)
                {
                    int curr = q.Dequeue();
                    comp.Add(curr);
                    foreach (int nxt in adj[curr])
                    {
                        if (!visited[nxt])
                        {
                            visited[nxt] = true;
                            q.Enqueue(nxt);
                        }
                    }
                }
                components.Add(comp);
            }
        }

        bool globalChanged = false;

        foreach (var comp in components)
        {
            int maxCompSize = 50;
            if (comp.Count > maxCompSize) continue;

            if (SolveComponent(comp, vars, constraints))
            {
                globalChanged = true;
            }
        }

        foreach (int u in resetVarIds) varId[u] = -1;
        resetVarIds.Clear();

        return globalChanged;
    }

    static bool SolveComponent(List<int> comp, List<int> vars, List<int> constraints)
    {
        int n = comp.Count;
        int[] localVarIdx = new int[vars.Count];
        Array.Fill(localVarIdx, -1);
        for (int i = 0; i < n; i++) localVarIdx[comp[i]] = i;

        List<int> compConstraints = new List<int>();
        foreach (int cIdx in constraints)
        {
            GetUnk(cIdx, tempUnkB, out int count, out int req);
            bool inComp = false;
            for (int i = 0; i < count; i++)
            {
                if (localVarIdx[varId[tempUnkB[i]]] != -1)
                {
                    inComp = true;
                    break;
                }
            }
            if (inComp) compConstraints.Add(cIdx);
        }

        int m = compConstraints.Count;
        int[] reqs = new int[m];
        int[] counts = new int[m];
        List<int>[] varConstraints = new List<int>[n];
        for (int i = 0; i < n; i++) varConstraints[i] = new List<int>();

        for (int j = 0; j < m; j++)
        {
            int cIdx = compConstraints[j];
            GetUnk(cIdx, tempUnkB, out int count, out int req);
            reqs[j] = req;
            counts[j] = count;
            for (int i = 0; i < count; i++)
            {
                int localU = localVarIdx[varId[tempUnkB[i]]];
                if (localU != -1) 
                {
                    varConstraints[localU].Add(j);
                }
            }
        }

        long validConfigurations = 0;
        int[] minesInValid = new int[n];
        int[] currentMines = new int[m];
        int[] currentZeros = new int[m];
        int[] assignment = new int[n];
        bool aborted = false;
        long maxConfigs = 20000;

        void Backtrack(int varIndex)
        {
            if (aborted) return;
            if (varIndex == n)
            {
                for (int j = 0; j < m; j++)
                {
                    if (currentMines[j] != reqs[j]) return;
                }
                validConfigurations++;
                if (validConfigurations > maxConfigs) aborted = true;

                for (int i = 0; i < n; i++)
                {
                    if (assignment[i] == 1) minesInValid[i]++;
                }
                return;
            }

            bool canBeZero = true;
            foreach (int j in varConstraints[varIndex])
            {
                if (currentZeros[j] + 1 > counts[j] - reqs[j])
                {
                    canBeZero = false;
                    break;
                }
            }

            if (canBeZero)
            {
                assignment[varIndex] = 0;
                foreach (int j in varConstraints[varIndex]) currentZeros[j]++;

                Backtrack(varIndex + 1);

                foreach (int j in varConstraints[varIndex]) currentZeros[j]--;
            }

            bool canBeOne = true;
            foreach (int j in varConstraints[varIndex])
            {
                if (currentMines[j] + 1 > reqs[j])
                {
                    canBeOne = false;
                    break;
                }
            }

            if (canBeOne)
            {
                assignment[varIndex] = 1;
                foreach (int j in varConstraints[varIndex]) currentMines[j]++;

                Backtrack(varIndex + 1);

                foreach (int j in varConstraints[varIndex]) currentMines[j]--;
            }
        }

        Backtrack(0);

        bool changed = false;
        if (!aborted && validConfigurations > 0)
        {
            for (int i = 0; i < n; i++)
            {
                int globalU = vars[comp[i]];
                if (minesInValid[i] == 0)
                {
                    if (board[globalU] == 9)
                    {
                        safeQueue.Enqueue(globalU);
                        changed = true;
                    }
                }
                else if (minesInValid[i] == validConfigurations)
                {
                    if (board[globalU] == 9)
                    {
                        mineQueue.Enqueue(globalU);
                        changed = true;
                    }
                }
            }
        }

        return changed;
    }
}

class FastScanner
{
    private Stream stream;
    private byte[] buffer = new byte[1024 * 1024];
    private int head = 0;
    private int tail = 0;

    public FastScanner(Stream stream)
    {
        this.stream = stream;
    }

    private int Read()
    {
        if (head >= tail)
        {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }

    public int NextInt()
    {
        int c = Read();
        while (c <= 32)
        {
            if (c == -1) return -1;
            c = Read();
        }
        int res = 0;
        while (c > 32)
        {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }

    public void ReadBoard(byte[] board, int H, int W)
    {
        int idx = 0;
        int total = H * W;
        while (idx < total)
        {
            int c = Read();
            if (c == -1) break;
            if (c == ' ' || c == '*' || (c >= '0' && c <= '8'))
            {
                if (c == ' ') board[idx] = 9;
                else if (c == '*') board[idx] = 10;
                else board[idx] = (byte)(c - '0');
                idx++;
            }
        }
    }
}
