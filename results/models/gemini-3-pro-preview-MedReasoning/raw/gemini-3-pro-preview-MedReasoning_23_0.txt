use std::collections::{BinaryHeap, HashMap};
use std::cmp::Ordering;
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Clone)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    gravity: f64,
    max_thrust: f64,
    
    waypoints: Vec<Point>,
    pathfinding_done: bool,
    running: bool,
}

#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug)]
struct Point {
    x: i32,
    y: i32,
}

impl Ord for AStarState {
    fn cmp(&self, other: &Self) -> Ordering {
        other.cost.cmp(&self.cost)
    }
}

impl PartialOrd for AStarState {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

#[derive(Clone, Eq, PartialEq)]
struct AStarState {
    cost: i32,
    position: Point,
}

fn heuristic(a: Point, b: Point) -> i32 {
    let dx = (a.x - b.x).abs();
    let dy = (a.y - b.y).abs();
    if dx > dy {
        14 * dy + 10 * (dx - dy)
    } else {
        14 * dx + 10 * (dy - dx)
    }
}

fn is_safe(nx: i32, ny: i32, map_bits: &[u8], width: i32, height: i32, start: Point, target: Point) -> bool {
    if (nx - target.x).abs() <= 2 && (ny - target.y).abs() <= 2 {
        if nx < 0 || ny < 0 || nx >= width || ny >= height { return false; }
        let idx = (ny as usize) * (width as usize) + (nx as usize);
        return (map_bits[idx / 8] & (1 << (idx % 8))) == 0;
    }
    if (nx - start.x).abs() <= 2 && (ny - start.y).abs() <= 2 {
        if nx < 0 || ny < 0 || nx >= width || ny >= height { return false; }
        let idx = (ny as usize) * (width as usize) + (nx as usize);
        return (map_bits[idx / 8] & (1 << (idx % 8))) == 0;
    }
    
    for sx in -1..=1 {
        for sy in -1..=1 {
            let cx = nx + sx;
            let cy = ny + sy;
            if cx >= 0 && cy >= 0 && cx < width && cy < height {
                let idx = (cy as usize) * (width as usize) + (cx as usize);
                if (map_bits[idx / 8] & (1 << (idx % 8))) != 0 {
                    return false;
                }
            } else {
                return false;
            }
        }
    }
    true
}

fn has_line_of_sight(p1: Point, p2: Point, map_bits: &[u8], width: usize, height: usize, start: Point, target: Point) -> bool {
    let mut x0 = p1.x;
    let mut y0 = p1.y;
    let x1 = p2.x;
    let y1 = p2.y;
    
    let dx = (x1 - x0).abs();
    let sx = if x0 < x1 { 1 } else { -1 };
    let dy = -(y1 - y0).abs();
    let sy = if y0 < y1 { 1 } else { -1 };
    let mut err = dx + dy;
    
    loop {
        if !is_safe(x0, y0, map_bits, width as i32, height as i32, start, target) {
            return false;
        }
        
        if x0 == x1 && y0 == y1 { break; }
        let e2 = 2 * err;
        if e2 >= dy { err += dy; x0 += sx; }
        if e2 <= dx { err += dx; y0 += sy; }
    }
    true
}

fn find_path(map_bits: &[u8], width: usize, height: usize, start_x: i32, start_y: i32, target_x: i32, target_y: i32) -> Vec<Point> {
    let start = Point { x: start_x, y: start_y };
    let target = Point { x: target_x, y: target_y };
    
    let mut open = BinaryHeap::new();
    let mut cost_so_far = HashMap::new();
    let mut came_from = HashMap::new();
    
    open.push(AStarState { cost: 0, position: start });
    cost_so_far.insert(start, 0);
    
    let mut nodes_evaluated = 0;
    let mut path = Vec::new();
    
    while let Some(AStarState { position, .. }) = open.pop() {
        if position == target {
            let mut curr = target;
            while curr != start {
                path.push(curr);
                curr = came_from[&curr];
            }
            path.push(start);
            path.reverse();
            break;
        }
        
        nodes_evaluated += 1;
        if nodes_evaluated > 2_000_000 {
            break; // give up A*, use fallback
        }
        
        for dx in -1..=1 {
            for dy in -1..=1 {
                if dx == 0 && dy == 0 { continue; }
                let nx = position.x + dx;
                let ny = position.y + dy;
                
                if !is_safe(nx, ny, map_bits, width as i32, height as i32, start, target) {
                    continue;
                }
                
                let move_cost = if dx == 0 || dy == 0 { 10 } else { 14 };
                let new_cost = cost_so_far[&position] + move_cost;
                
                let next_p = Point { x: nx, y: ny };
                if new_cost < *cost_so_far.get(&next_p).unwrap_or(&i32::MAX) {
                    cost_so_far.insert(next_p, new_cost);
                    let priority = new_cost + heuristic(next_p, target);
                    open.push(AStarState { cost: priority, position: next_p });
                    came_from.insert(next_p, position);
                }
            }
        }
    }
    path
}

fn smooth_path(path: &[Point], map_bits: &[u8], width: usize, height: usize, start: Point, target: Point) -> Vec<Point> {
    if path.is_empty() { return vec![]; }
    let mut smooth = vec![path[0]];
    let mut curr = 0;
    
    while curr < path.len() - 1 {
        let mut furthest = curr + 1;
        for i in (curr + 2)..path.len() {
            if has_line_of_sight(path[curr], path[i], map_bits, width, height, start, target) {
                furthest = i;
            } else {
                break;
            }
        }
        smooth.push(path[furthest]);
        curr = furthest;
    }
    smooth
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();
    
    // Read line 1
    reader.read_line(&mut line).unwrap();
    let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let width = parts[0] as usize;
    let height = parts[1] as usize;
    let gravity = parts[2];
    let max_thrust = parts[3];
    
    // Read line 2
    line.clear();
    reader.read_line(&mut line).unwrap();
    let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let start_x = parts[0];
    let start_y = parts[1];
    let target_x = parts[2];
    let target_y = parts[3];
    
    // Read map
    let bit_size = (width * height + 7) / 8;
    let mut map_bits = vec![0u8; bit_size];
    
    let mut buf = Vec::new();
    for y in 0..height {
        buf.clear();
        reader.read_until(b'\n', &mut buf).unwrap();
        let row_offset = y * width;
        for x in 0..width {
            if x < buf.len() {
                let b = buf[x];
                if b == b'#' {
                    let idx = row_offset + x;
                    map_bits[idx / 8] |= 1 << (idx % 8);
                }
            }
        }
    }
    
    let state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: 0.0,
        start_x,
        start_y,
        target_x,
        target_y,
        gravity,
        max_thrust,
        waypoints: Vec::new(),
        pathfinding_done: false,
        running: true,
    }));
    
    // Launch pathfinding thread
    let state_clone = Arc::clone(&state);
    let map_arc = Arc::new(map_bits);
    thread::spawn(move || {
        let start_p = Point { x: start_x as i32, y: start_y as i32 };
        let target_p = Point { x: target_x as i32, y: target_y as i32 };
        
        let path = find_path(&map_arc, width, height, start_p.x, start_p.y, target_p.x, target_p.y);
        let mut smoothed = smooth_path(&path, &map_arc, width, height, start_p, target_p);
        
        let mut st = state_clone.lock().unwrap();
        if smoothed.is_empty() {
            // Fallback strategy: fly high
            smoothed.push(Point { x: start_x as i32, y: (start_y as i32 - 40).max(10) });
            smoothed.push(Point { x: target_x as i32, y: (start_y as i32 - 40).max(10) });
            smoothed.push(Point { x: target_x as i32, y: target_y as i32 });
        }
        st.waypoints = smoothed;
        st.pathfinding_done = true;
    });
    
    // Control loop thread
    let state_clone2 = Arc::clone(&state);
    thread::spawn(move || {
        loop {
            let st = {
                let lock = state_clone2.lock().unwrap();
                if !lock.running { break; }
                lock.clone()
            };
            
            let mut current_tx = st.start_x;
            let mut current_ty = st.start_y;
            let mut is_final = false;
            
            if !st.pathfinding_done {
                // Keep hovering at start
                current_tx = st.start_x;
                current_ty = st.start_y;
            } else {
                if !st.waypoints.is_empty() {
                    let wp = st.waypoints[0];
                    current_tx = wp.x as f64;
                    current_ty = wp.y as f64;
                    is_final = st.waypoints.len() == 1;
                    
                    let dx = current_tx - st.x;
                    let dy = current_ty - st.y;
                    if (dx*dx + dy*dy).sqrt() < 4.0 && !is_final {
                        state_clone2.lock().unwrap().waypoints.remove(0);
                    }
                } else {
                    current_tx = st.target_x;
                    current_ty = st.target_y;
                    is_final = true;
                }
            }
            
            if is_final {
                current_ty += 0.5; // Aim slightly below ground to ensure touchdown
            }
            
            let dx = current_tx - st.x;
            let dy = current_ty - st.y;
            let dist = (dx*dx + dy*dy).sqrt();
            
            let max_speed = 15.0;
            let speed_mult = if is_final { 0.8 } else { 1.5 };
            let speed_limit = (dist * speed_mult).clamp(0.0, max_speed);
            
            let desired_vx = if dist > 0.0 { (dx / dist) * speed_limit } else { 0.0 };
            let desired_vy = if dist > 0.0 { (dy / dist) * speed_limit } else { 0.0 };
            
            let time_constant = 0.5;
            let ax = (desired_vx - st.vx) / time_constant;
            let ay = (desired_vy - st.vy) / time_constant;
            
            let req_thrust_x = ax;
            let req_thrust_y = ay - st.gravity; 
            
            let max_tilt = 1.0; 
            let target_angle = req_thrust_x.atan2(-req_thrust_y).clamp(-max_tilt, max_tilt);
            
            let required_thrust_mag = (req_thrust_x.powi(2) + req_thrust_y.powi(2)).sqrt();
            let thrust = (required_thrust_mag / st.max_thrust).clamp(0.0, 1.0);
            
            let mut err_angle = target_angle - st.angle;
            while err_angle > std::f64::consts::PI { err_angle -= 2.0 * std::f64::consts::PI; }
            while err_angle < -std::f64::consts::PI { err_angle += 2.0 * std::f64::consts::PI; }
            
            let desired_angular_vel = err_angle * 3.0;
            let turn = desired_angular_vel.clamp(-1.0, 1.0);
            
            println!("{:.4} {:.4}", thrust, turn);
            let _ = io::stdout().flush();
            
            thread::sleep(Duration::from_millis(50));
        }
    });
    
    // State update loop
    loop {
        line.clear();
        if reader.read_line(&mut line).unwrap() == 0 {
            break;
        }
        let parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
        if parts.len() >= 7 {
            let mut st = state.lock().unwrap();
            st.x = parts[0];
            st.y = parts[1];
            st.vx = parts[2];
            st.vy = parts[3];
            st.angle = parts[4];
            st.angular_vel = parts[5];
            st.fuel = parts[6];
        }
    }
    
    state.lock().unwrap().running = false;
}
