#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Node {
    int x, y, w;
};

struct Rect {
    int id;
    int w, h;
};

struct Candidate {
    int x = -1, y = -1, w = -1, h = -1;
    bool rotated = false;
    bool valid = false;
    long long score = 4000000000000000000LL; // large enough
    
    bool operator<(const Candidate& other) const {
        if (!valid) return false;
        if (!other.valid) return true;
        if (score != other.score) return score < other.score;
        return w > other.w;
    }
};

int get_y_max(int nx, int nw, const vector<Node>& nodes) {
    int y_max = 0;
    int l = 0, r = (int)nodes.size() - 1;
    int start_idx = r + 1;
    while (l <= r) {
        int mid = l + (r - l) / 2;
        if (nodes[mid].x + nodes[mid].w > nx) {
            start_idx = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    
    for (int i = start_idx; i < (int)nodes.size(); ++i) {
        if (nodes[i].x >= nx + nw) break;
        if (nodes[i].y > y_max) y_max = nodes[i].y;
    }
    return y_max;
}

Candidate find_best_placement(int bw, int bh, int rw, int rh, const vector<Node>& nodes) {
    Candidate best;
    auto evaluate = [&](int w, int h, bool rot) {
        for (int i = 0; i < (int)nodes.size(); ++i) {
            // Left align
            int x1 = nodes[i].x;
            if (x1 + w <= bw) {
                int y1 = get_y_max(x1, w, nodes);
                if (y1 + h <= bh) {
                    long long score = (long long)(y1 + h) * (bw + 1) + x1;
                    Candidate cand{x1, y1, w, h, rot, true, score};
                    if (cand < best) best = cand;
                }
            }
            // Right align
            int x2 = nodes[i].x + nodes[i].w - w;
            if (x2 >= 0 && x2 + w <= bw) {
                int y2 = get_y_max(x2, w, nodes);
                if (y2 + h <= bh) {
                    long long score = (long long)(y2 + h) * (bw + 1) + x2;
                    Candidate cand{x2, y2, w, h, rot, true, score};
                    if (cand < best) best = cand;
                }
            }
        }
    };
    evaluate(rw, rh, false);
    if (rw != rh) evaluate(rh, rw, true);
    return best;
}

void update_skyline(int nx, int ny, int nw, int nh, vector<Node>& nodes) {
    vector<Node> new_nodes;
    new_nodes.reserve(nodes.size() + 2);
    for (const auto& n : nodes) {
        if (n.x + n.w <= nx || n.x >= nx + nw) {
            new_nodes.push_back(n);
        } else {
            if (n.x < nx) {
                new_nodes.push_back({n.x, n.y, nx - n.x});
            }
            if (n.x + n.w > nx + nw) {
                new_nodes.push_back({nx + nw, n.y, (n.x + n.w) - (nx + nw)});
            }
        }
    }
    new_nodes.push_back({nx, ny + nh, nw});
    
    sort(new_nodes.begin(), new_nodes.end(), [](const Node& a, const Node& b) {
        return a.x < b.x;
    });
    
    nodes.clear();
    for (const auto& n : new_nodes) {
        if (!nodes.empty() && nodes.back().y == n.y && nodes.back().x + nodes.back().w == n.x) {
            nodes.back().w += n.w;
        } else {
            nodes.push_back(n);
        }
    }
}

struct Placement {
    int board_idx;
    int x, y;
    bool rotated;
};

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, bw, bh;
    if (!(cin >> N >> bw >> bh)) return 0;
    
    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        cin >> rects[i].w >> rects[i].h;
    }
    
    // Sort rectangles by max dimension descending, then area descending
    vector<Rect> sorted_rects = rects;
    sort(sorted_rects.begin(), sorted_rects.end(), [](const Rect& a, const Rect& b) {
        long long area_a = (long long)a.w * a.h;
        long long area_b = (long long)b.w * b.h;
        int max_a = max(a.w, a.h);
        int max_b = max(b.w, b.h);
        if (max_a != max_b) return max_a > max_b;
        if (area_a != area_b) return area_a > area_b;
        return a.id < b.id;
    });
    
    vector<Placement> results(N);
    vector<bool> packed(N, false);
    
    int num_packed = 0;
    int num_boards = 0;
    
    while (num_packed < N) {
        int current_board = num_boards++;
        vector<Node> nodes;
        nodes.push_back({0, 0, bw});
        
        // This array ensures we don't repeatedly try to fit a rectangle that has already failed
        // to fit in the current board, vastly improving worst-case performance.
        vector<bool> failed(N, false);
        
        long long remaining_area = (long long)bw * bh;
        int packed_this_board = 0;
        
        for (int i = 0; i < N; ++i) {
            if (packed[i] || failed[i]) continue;
            
            const Rect& r = sorted_rects[i];
            
            long long rect_area = (long long)r.w * r.h;
            if (rect_area > remaining_area) {
                failed[i] = true;
                continue;
            }
            if ((r.w > bw || r.h > bh) && (r.h > bw || r.w > bh)) {
                failed[i] = true;
                continue;
            }
            
            Candidate cand = find_best_placement(bw, bh, r.w, r.h, nodes);
            
            if (cand.valid) {
                update_skyline(cand.x, cand.y, cand.w, cand.h, nodes);
                results[r.id] = {current_board, cand.x, cand.y, cand.rotated};
                packed[i] = true;
                num_packed++;
                packed_this_board++;
                remaining_area -= rect_area;
            } else {
                failed[i] = true;
            }
        }
        
        if (packed_this_board == 0) {
            // Emergency break to avoid infinite loop if no rectangle can fit
            break;
        }
    }
    
    cout << num_boards << "\n";
    for (int i = 0; i < N; ++i) {
        cout << results[i].board_idx << " " << results[i].x << " " << results[i].y << " " << (results[i].rotated ? 1 : 0) << "\n";
    }
    
    return 0;
}
