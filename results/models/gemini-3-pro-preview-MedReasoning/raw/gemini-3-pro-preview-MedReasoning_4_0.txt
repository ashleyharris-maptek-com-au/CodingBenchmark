#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double length() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const {
        double l = length();
        if (l < 1e-9) return {0,0,0};
        return {x/l, y/l, z/l};
    }
};

bool update_simplex(vector<Vec3>& simplex, Vec3& d) {
    int sz = simplex.size();
    if (sz == 2) {
        Vec3 A = simplex[1];
        Vec3 B = simplex[0];
        Vec3 AB = B - A;
        Vec3 AO = A * -1.0;
        
        if (AB.dot(AO) > 0) {
            d = AB.cross(AO).cross(AB);
            if (d.length() < 1e-7) return true;
        } else {
            simplex = {A};
            d = AO;
        }
        return false;
    }
    else if (sz == 3) {
        Vec3 A = simplex[2];
        Vec3 B = simplex[1];
        Vec3 C = simplex[0];
        Vec3 AB = B - A;
        Vec3 AC = C - A;
        Vec3 AO = A * -1.0;
        
        Vec3 ABC = AB.cross(AC);
        
        if (ABC.cross(AC).dot(AO) > 0) {
            if (AC.dot(AO) > 0) {
                simplex = {C, A};
                d = AC.cross(AO).cross(AC);
                if (d.length() < 1e-7) return true;
            } else {
                if (AB.dot(AO) > 0) {
                    simplex = {B, A};
                    d = AB.cross(AO).cross(AB);
                    if (d.length() < 1e-7) return true;
                } else {
                    simplex = {A};
                    d = AO;
                }
            }
        } else {
            if (AB.cross(ABC).dot(AO) > 0) {
                if (AB.dot(AO) > 0) {
                    simplex = {B, A};
                    d = AB.cross(AO).cross(AB);
                    if (d.length() < 1e-7) return true;
                } else {
                    simplex = {A};
                    d = AO;
                }
            } else {
                if (ABC.dot(AO) > 0) {
                    simplex = {C, B, A};
                    d = ABC;
                } else {
                    simplex = {B, C, A};
                    d = ABC * -1.0;
                }
            }
        }
        return false;
    }
    else if (sz == 4) {
        Vec3 A = simplex[3];
        Vec3 B = simplex[2];
        Vec3 C = simplex[1];
        Vec3 D = simplex[0];
        
        Vec3 AB = B - A;
        Vec3 AC = C - A;
        Vec3 AD = D - A;
        Vec3 AO = A * -1.0;
        
        Vec3 ABC = AB.cross(AC);
        if (ABC.dot(AD) > 0) ABC = ABC * -1.0;
        
        Vec3 ACD = AC.cross(AD);
        if (ACD.dot(AB) > 0) ACD = ACD * -1.0;
        
        Vec3 ADB = AD.cross(AB);
        if (ADB.dot(AC) > 0) ADB = ADB * -1.0;
        
        if (ABC.dot(AO) > 0) {
            simplex = {C, B, A};
            return update_simplex(simplex, d);
        }
        if (ACD.dot(AO) > 0) {
            simplex = {D, C, A};
            return update_simplex(simplex, d);
        }
        if (ADB.dot(AO) > 0) {
            simplex = {B, D, A};
            return update_simplex(simplex, d);
        }
        
        return true;
    }
    return false;
}

bool GJK_is_inside(const Vec3& p, const vector<Vec3>& S) {
    Vec3 d = S[0] - p;
    if (d.length() < 1e-7) return true;
    
    vector<Vec3> simplex;
    simplex.push_back(S[0] - p);
    d = d * -1.0;
    
    for (int iter = 0; iter < 64; ++iter) {
        double max_dot = -1e9;
        Vec3 support = S[0] - p;
        for (const auto& v : S) {
            Vec3 v_shifted = v - p;
            double dt = v_shifted.dot(d);
            if (dt > max_dot) {
                max_dot = dt;
                support = v_shifted;
            }
        }
        
        if (max_dot < 1e-7) {
            return false; 
        }
        
        for (const auto& sv : simplex) {
            if ((sv - support).length() < 1e-5) {
                return false; 
            }
        }
        
        simplex.push_back(support);
        
        if (update_simplex(simplex, d)) {
            return true;
        }
    }
    return false; 
}

struct Tetrahedron {
    Vec3 pts[4];
};

struct BBox {
    double min_x, max_x, min_y, max_y, min_z, max_z;
    bool intersects(const BBox& o) const {
        return !(max_x < o.min_x || min_x > o.max_x ||
                 max_y < o.min_y || min_y > o.max_y ||
                 max_z < o.min_z || min_z > o.max_z);
    }
};

BBox get_bbox(const Tetrahedron& t) {
    BBox b = {1e9, -1e9, 1e9, -1e9, 1e9, -1e9};
    for(int i=0; i<4; ++i) {
        b.min_x = min(b.min_x, t.pts[i].x); b.max_x = max(b.max_x, t.pts[i].x);
        b.min_y = min(b.min_y, t.pts[i].y); b.max_y = max(b.max_y, t.pts[i].y);
        b.min_z = min(b.min_z, t.pts[i].z); b.max_z = max(b.max_z, t.pts[i].z);
    }
    return b;
}

bool check_overlap(const Tetrahedron& t1, const Tetrahedron& t2) {
    vector<Vec3> axes;
    int faces[4][3] = {{0,1,2}, {0,2,3}, {0,3,1}, {1,3,2}};
    for(int i=0; i<4; ++i) {
        Vec3 a = t1.pts[faces[i][0]];
        Vec3 b = t1.pts[faces[i][1]];
        Vec3 c = t1.pts[faces[i][2]];
        axes.push_back((b-a).cross(c-a).normalized());
    }
    for(int i=0; i<4; ++i) {
        Vec3 a = t2.pts[faces[i][0]];
        Vec3 b = t2.pts[faces[i][1]];
        Vec3 c = t2.pts[faces[i][2]];
        axes.push_back((b-a).cross(c-a).normalized());
    }
    vector<Vec3> e1, e2;
    for(int i=0; i<3; ++i) {
        for(int j=i+1; j<4; ++j) {
            e1.push_back((t1.pts[i]-t1.pts[j]).normalized());
            e2.push_back((t2.pts[i]-t2.pts[j]).normalized());
        }
    }
    for(auto& a : e1) {
        for(auto& b : e2) {
            Vec3 cross = a.cross(b);
            if(cross.length() > 1e-7) {
                axes.push_back(cross.normalized());
            }
        }
    }

    for(auto& axis : axes) {
        double min1 = 1e9, max1 = -1e9;
        double min2 = 1e9, max2 = -1e9;
        for(int i=0; i<4; ++i) {
            double p1 = t1.pts[i].dot(axis);
            min1 = min(min1, p1); max1 = max(max1, p1);
            double p2 = t2.pts[i].dot(axis);
            min2 = min(min2, p2); max2 = max(max2, p2);
        }
        if(max1 <= min2 + 1e-7 || max2 <= min1 + 1e-7) {
            return false; 
        }
    }
    return true; 
}

Tetrahedron get_base_tet(double L) {
    double r = L / sqrt(24.0);
    double R = L * sqrt(3.0 / 8.0);
    Tetrahedron t;
    t.pts[0] = {0, 0, R};
    t.pts[1] = {L/sqrt(3.0), 0, -r};
    t.pts[2] = {-L/(2.0*sqrt(3.0)), L/2.0, -r};
    t.pts[3] = {-L/(2.0*sqrt(3.0)), -L/2.0, -r};
    return t;
}

Tetrahedron transform_tet(const Tetrahedron& base, Vec3 center, double rx, double ry, double rz) {
    Tetrahedron t;
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    for(int i=0; i<4; ++i) {
        double x = base.pts[i].x;
        double y = base.pts[i].y;
        double z = base.pts[i].z;

        double x1 = x;
        double y1 = y*cx - z*sx;
        double z1 = y*sx + z*cx;

        double x2 = x1*cy + z1*sy;
        double y2 = y1;
        double z2 = -x1*sy + z1*cy;

        double x3 = x2*cz - y2*sz;
        double y3 = x2*sz + y2*cz;
        double z3 = z2;

        t.pts[i] = {x3 + center.x, y3 + center.y, z3 + center.z};
    }
    return t;
}

struct Placed {
    Tetrahedron t;
    BBox b;
    Vec3 center;
    Vec3 rot;
};

struct SpatialHash {
    double cell_size;
    unordered_map<int64_t, vector<int>> grid;
    
    int64_t hash(int x, int y, int z) {
        return ((int64_t)(uint32_t)x << 42) ^ ((int64_t)(uint32_t)y << 21) ^ (int64_t)(uint32_t)z;
    }
    
    void insert(int id, const BBox& b) {
        int min_x = floor(b.min_x / cell_size);
        int max_x = floor(b.max_x / cell_size);
        int min_y = floor(b.min_y / cell_size);
        int max_y = floor(b.max_y / cell_size);
        int min_z = floor(b.min_z / cell_size);
        int max_z = floor(b.max_z / cell_size);
        
        for(int x = min_x; x <= max_x; ++x) {
            for(int y = min_y; y <= max_y; ++y) {
                for(int z = min_z; z <= max_z; ++z) {
                    grid[hash(x,y,z)].push_back(id);
                }
            }
        }
    }
    
    vector<int> query(const BBox& b) {
        vector<int> res;
        int min_x = floor(b.min_x / cell_size);
        int max_x = floor(b.max_x / cell_size);
        int min_y = floor(b.min_y / cell_size);
        int max_y = floor(b.max_y / cell_size);
        int min_z = floor(b.min_z / cell_size);
        int max_z = floor(b.max_z / cell_size);
        
        for(int x = min_x; x <= max_x; ++x) {
            for(int y = min_y; y <= max_y; ++y) {
                for(int z = min_z; z <= max_z; ++z) {
                    auto it = grid.find(hash(x,y,z));
                    if(it != grid.end()) {
                        for(int id : it->second) res.push_back(id);
                    }
                }
            }
        }
        sort(res.begin(), res.end());
        res.erase(unique(res.begin(), res.end()), res.end());
        return res;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    auto start_time = chrono::steady_clock::now();
    
    int V;
    double edge_length;
    if (!(cin >> V >> edge_length)) return 0;
    
    vector<Vec3> container(V);
    double min_x = 1e9, max_x = -1e9;
    double min_y = 1e9, max_y = -1e9;
    double min_z = 1e9, max_z = -1e9;
    
    for (int i = 0; i < V; ++i) {
        cin >> container[i].x >> container[i].y >> container[i].z;
        min_x = min(min_x, container[i].x); max_x = max(max_x, container[i].x);
        min_y = min(min_y, container[i].y); max_y = max(max_y, container[i].y);
        min_z = min(min_z, container[i].z); max_z = max(max_z, container[i].z);
    }
    
    if (V < 4) {
        cout << 0 << "\n";
        return 0;
    }

    vector<Vec3> S_hull;
    if (container.size() > 100) {
        int num_dirs = 1000;
        mt19937 rng(42);
        uniform_real_distribution<double> dist(-1.0, 1.0);
        for(int i=0; i<num_dirs; ++i) {
            Vec3 d;
            do {
                d = {dist(rng), dist(rng), dist(rng)};
            } while(d.length() < 0.01);
            d = d.normalized();
            
            double max_dot = -1e9;
            Vec3 best_p = container[0];
            for(auto& p : container) {
                double dp = p.dot(d);
                if(dp > max_dot) {
                    max_dot = dp;
                    best_p = p;
                }
            }
            
            bool exists = false;
            for(auto& sp : S_hull) {
                if((sp - best_p).length() < 1e-5) { exists = true; break; }
            }
            if(!exists) S_hull.push_back(best_p);
        }
    } else {
        S_hull = container;
    }
    
    double vol = (max_x - min_x) * (max_y - min_y) * (max_z - min_z);
    double step = edge_length * 0.4;
    double cells = vol / pow(step, 3);
    if (cells > 500000.0) {
        step = cbrt(vol / 500000.0);
    }
    
    vector<Vec3> rotations = {
        {0, 0, 0},
        {M_PI, 0, 0},
        {0, M_PI, 0},
        {0, 0, M_PI},
        {M_PI/2, M_PI/4, 0},
        {0, M_PI/2, M_PI/4},
        {M_PI/4, 0, M_PI/2},
        {M_PI/4, M_PI/4, M_PI/4}
    };
    
    Tetrahedron base_tet = get_base_tet(edge_length);
    vector<Tetrahedron> base_tets_rot;
    for(auto& rot : rotations) {
        base_tets_rot.push_back(transform_tet(base_tet, {0,0,0}, rot.x, rot.y, rot.z));
    }
    
    vector<Placed> placed_tets;
    SpatialHash shash;
    shash.cell_size = edge_length * 2.0;
    
    for (int ix = 0; ; ix++) {
        double x = min_x + ix * step;
        if (x > max_x) break;
        for (int iy = 0; ; iy++) {
            double y = min_y + iy * step + ((ix % 2 == 0) ? 0 : step * 0.5);
            if (y > max_y) break;
            for (int iz = 0; ; iz++) {
                double z = min_z + iz * step + ((iy % 2 == 0) ? 0 : step * 0.5);
                if (z > max_z) break;
                
                auto current_time = chrono::steady_clock::now();
                double elapsed = chrono::duration<double>(current_time - start_time).count();
                if (elapsed > 28.0) {
                    goto end_search;
                }
                
                Vec3 center = {x, y, z};
                
                for(size_t rot_idx=0; rot_idx < rotations.size(); ++rot_idx) {
                    Tetrahedron t = base_tets_rot[rot_idx];
                    for(int i=0; i<4; ++i) t.pts[i] = t.pts[i] + center;
                    
                    BBox tb = get_bbox(t);
                    if (tb.min_x < min_x || tb.max_x > max_x ||
                        tb.min_y < min_y || tb.max_y > max_y ||
                        tb.min_z < min_z || tb.max_z > max_z) continue;
                    
                    bool inside = true;
                    for (int i=0; i<4; ++i) {
                        if (!GJK_is_inside(t.pts[i], S_hull)) {
                            inside = false;
                            break;
                        }
                    }
                    
                    if (inside) {
                        bool overlap = false;
                        vector<int> candidates = shash.query(tb);
                        for (int id : candidates) {
                            if (tb.intersects(placed_tets[id].b)) {
                                if (check_overlap(t, placed_tets[id].t)) {
                                    overlap = true;
                                    break;
                                }
                            }
                        }
                        
                        if (!overlap) {
                            int new_id = placed_tets.size();
                            placed_tets.push_back({t, tb, center, rotations[rot_idx]});
                            shash.insert(new_id, tb);
                            break; 
                        }
                    }
                }
            }
        }
    }
end_search:

    cout << placed_tets.size() << "\n";
    for (const auto& p : placed_tets) {
        cout << p.center.x << " " << p.center.y << " " << p.center.z << " "
             << p.rot.x << " " << p.rot.y << " " << p.rot.z << "\n";
    }

    return 0;
}
