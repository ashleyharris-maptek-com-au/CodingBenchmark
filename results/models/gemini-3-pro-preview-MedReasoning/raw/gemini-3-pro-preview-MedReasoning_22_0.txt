using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisPlayer
{
    class Program
    {
        static Dictionary<(char, int), List<(int x, int y)>> pieceCache = new Dictionary<(char, int), List<(int x, int y)>>();

        static string[] GetBaseShape(char p) {
            switch(p) {
                case 'I': return new string[] { "....", "####", "....", "...." };
                case 'O': return new string[] { "##", "##" };
                case 'T': return new string[] { ".#.", "###", "..." };
                case 'J': return new string[] { "#..", "###", "..." };
                case 'L': return new string[] { "..#", "###", "..." };
                case 'S': return new string[] { ".##", "##.", "..." };
                case 'Z': return new string[] { "##.", ".##", "..." };
                default: throw new Exception("Unknown piece " + p);
            }
        }

        static List<(int x, int y)> GetPiece(char p, int r) {
            if (pieceCache.ContainsKey((p, r))) return pieceCache[(p, r)];
            
            string[] shape = GetBaseShape(p);
            int N = shape.Length;
            char[,] grid = new char[N, N];
            for(int i = 0; i < N; i++)
                for(int j = 0; j < N; j++)
                    grid[i, j] = shape[i][j];

            for(int rot = 0; rot < r; rot++) {
                char[,] next = new char[N, N];
                for(int i = 0; i < N; i++)
                    for(int j = 0; j < N; j++)
                        next[j, N - 1 - i] = grid[i, j];
                grid = next;
            }

            List<(int x, int y)> blocks = new List<(int, int)>();
            for(int i = 0; i < N; i++) {
                for(int j = 0; j < N; j++) {
                    if (grid[i, j] == '#') {
                        blocks.Add((j, N - 1 - i));
                    }
                }
            }

            int min_x = blocks.Min(b => b.x);
            int min_y = blocks.Min(b => b.y);

            for(int i = 0; i < blocks.Count; i++) {
                blocks[i] = (blocks[i].x - min_x, blocks[i].y - min_y);
            }
            
            pieceCache[(p, r)] = blocks;
            return blocks;
        }

        class State {
            public bool[,] board;
            public int[] heights;
            public int width;
            public int height;

            public State Clone() {
                State s = new State();
                s.width = width;
                s.height = height;
                s.board = (bool[,])board.Clone();
                s.heights = (int[])heights.Clone();
                return s;
            }
        }

        static double Evaluate(bool[,] board, int[] heights, int width, int height, int linesCleared, int maxH, int dy, int pieceHeight,
                               double wLanding, double wLines, double wRowTrans, double wColTrans, double wHoles, double wWells) {
            int evalHeight = Math.Max(height, heights.Max());

            int rowTransitions = 0;
            for (int y = 0; y < evalHeight; y++) {
                bool last = true; 
                for (int x = 0; x < width; x++) {
                    bool current = board[x, y];
                    if (last != current) rowTransitions++;
                    last = current;
                }
                if (!last) rowTransitions++; 
            }

            int colTransitions = 0;
            for (int x = 0; x < width; x++) {
                bool last = true; 
                for (int y = 0; y < evalHeight; y++) {
                    bool current = board[x, y];
                    if (last != current) colTransitions++;
                    last = current;
                }
                if (last) colTransitions++; 
            }

            int holes = 0;
            for (int x = 0; x < width; x++) {
                for (int y = 0; y < heights[x] - 1; y++) {
                    if (!board[x, y]) holes++;
                }
            }

            int wellSums = 0;
            for (int x = 0; x < width; x++) {
                int lHeight = (x == 0) ? evalHeight : heights[x - 1];
                int rHeight = (x == width - 1) ? evalHeight : heights[x + 1];
                int minAdj = Math.Min(lHeight, rHeight);
                if (minAdj > heights[x]) {
                    int depth = minAdj - heights[x];
                    wellSums += depth * (depth + 1) / 2;
                }
            }

            double landingHeight = dy + pieceHeight / 2.0;

            return wLanding * landingHeight
                + wLines * linesCleared
                + wRowTrans * rowTransitions
                + wColTrans * colTransitions
                + wHoles * holes
                + wWells * wellSums;
        }

        static void Main(string[] args)
        {
            string header = Console.ReadLine();
            if (header == null) return;
            string[] parts = header.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 3) return;

            int width = int.Parse(parts[0]);
            int height = int.Parse(parts[1]);
            int numPieces = int.Parse(parts[2]);

            State gameState = new State();
            gameState.width = width;
            gameState.height = height;
            gameState.board = new bool[width, height + 8]; 
            gameState.heights = new int[width];

            for (int p = 0; p < numPieces; p++) {
                string pieceLine = Console.ReadLine();
                if (pieceLine == null) break;
                pieceLine = pieceLine.Trim();
                if (pieceLine.Length == 0) continue;
                
                char pieceType = pieceLine[0];

                double bestScore = double.NegativeInfinity;
                int bestRot = 0;
                int bestCol = 0;
                State bestNextState = null;

                int currentMaxH = gameState.heights.Max();
                
                double wLanding = -4.500158;
                double wLines = 3.418126;
                double wRowTrans = -3.217888;
                double wColTrans = -9.348695;
                double wHoles = -7.899265;
                double wWells = -3.385597;

                // Adaptive logic adjusting based on game complexity
                if (currentMaxH > height * 0.7) {
                    wLanding = -10.0;
                    wLines = 10.0;
                    wRowTrans = -3.0;
                    wColTrans = -3.0;
                    wHoles = -15.0;
                    wWells = -1.0;
                } else if (p < 50 && currentMaxH < height * 0.3) {
                    wLines = 1.0; 
                    wWells = -2.0;
                }

                for (int rot = 0; rot < 4; rot++) {
                    var blocks = GetPiece(pieceType, rot);
                    int pieceWidth = blocks.Max(b => b.x) + 1;
                    int pieceHeight = blocks.Max(b => b.y) + 1;

                    for (int col = 0; col <= width - pieceWidth; col++) {
                        State nextState = gameState.Clone();
                        int dy = 0;
                        foreach (var b in blocks) {
                            int px = col + b.x;
                            int by = b.y;
                            int req = nextState.heights[px] - by;
                            if (req > dy) dy = req;
                        }

                        bool outOfBounds = false;
                        foreach (var b in blocks) {
                            int py = dy + b.y;
                            if (py >= nextState.board.GetLength(1)) {
                                outOfBounds = true;
                                break;
                            }
                        }
                        if (outOfBounds) continue;

                        foreach (var b in blocks) {
                            int px = col + b.x;
                            int py = dy + b.y;
                            nextState.board[px, py] = true;
                        }

                        int linesCleared = 0;
                        var rowsToCheck = blocks.Select(b => dy + b.y).Distinct().OrderByDescending(y => y).ToList();

                        foreach (int y in rowsToCheck) {
                            bool full = true;
                            for (int x = 0; x < width; x++) {
                                if (!nextState.board[x, y]) {
                                    full = false;
                                    break;
                                }
                            }
                            if (full) {
                                linesCleared++;
                                for (int row = y; row < nextState.board.GetLength(1) - 1; row++) {
                                    for (int x = 0; x < width; x++) {
                                        nextState.board[x, row] = nextState.board[x, row + 1];
                                    }
                                }
                                for (int x = 0; x < width; x++) {
                                    nextState.board[x, nextState.board.GetLength(1) - 1] = false;
                                }
                            }
                        }

                        for (int x = 0; x < width; x++) {
                            nextState.heights[x] = 0;
                            for (int y = nextState.board.GetLength(1) - 1; y >= 0; y--) {
                                if (nextState.board[x, y]) {
                                    nextState.heights[x] = y + 1;
                                    break;
                                }
                            }
                        }

                        int maxH = nextState.heights.Max();
                        double score = Evaluate(nextState.board, nextState.heights, width, height, linesCleared, maxH, dy, pieceHeight,
                                                wLanding, wLines, wRowTrans, wColTrans, wHoles, wWells);

                        if (score > bestScore) {
                            bestScore = score;
                            bestRot = rot;
                            bestCol = col;
                            bestNextState = nextState;
                        }
                    }
                }

                if (bestNextState != null) {
                    gameState = bestNextState;
                }

                Console.WriteLine($"{bestRot} {bestCol}");
                
                string resultLine = Console.ReadLine();
                if (resultLine == null || resultLine.Trim() == "gameover") {
                    break;
                }
            }
        }
    }
}
