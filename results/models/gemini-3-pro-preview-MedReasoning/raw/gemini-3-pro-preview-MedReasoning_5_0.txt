use std::io::{self, BufRead, Write, BufWriter};
use std::time::{Instant, Duration};

struct SolverState {
    width: usize,
    height: usize,
    empty_count: usize,
    require_cycle: bool,
    start_time: Instant,
    visited_gen: usize,
    visited: Vec<usize>,
    q: Vec<usize>,
}

impl SolverState {
    fn is_connected(&mut self, grid: &Vec<Vec<bool>>, cx: usize, cy: usize, remaining: usize) -> bool {
        if remaining == 0 {
            return true;
        }
        self.visited_gen += 1;
        self.q.clear();
        let mut count = 0;

        let mut start_node = None;
        let dx = [0, 0, 1, -1];
        let dy = [1, -1, 0, 0];

        for i in 0..4 {
            let nx = cx as isize + dx[i];
            let ny = cy as isize + dy[i];
            if nx >= 0 && nx < self.width as isize && ny >= 0 && ny < self.height as isize {
                let nx = nx as usize;
                let ny = ny as usize;
                if !grid[ny][nx] {
                    start_node = Some((nx, ny));
                    break;
                }
            }
        }

        if let Some((sx, sy)) = start_node {
            let start_idx = sy * self.width + sx;
            self.q.push(start_idx);
            self.visited[start_idx] = self.visited_gen;
            count += 1;

            let mut head = 0;
            while head < self.q.len() {
                let curr = self.q[head];
                head += 1;
                let x = curr % self.width;
                let y = curr / self.width;

                for i in 0..4 {
                    let nx = x as isize + dx[i];
                    let ny = y as isize + dy[i];
                    if nx >= 0 && nx < self.width as isize && ny >= 0 && ny < self.height as isize {
                        let nx = nx as usize;
                        let ny = ny as usize;
                        let idx = ny * self.width + nx;
                        if !grid[ny][nx] && self.visited[idx] != self.visited_gen {
                            self.visited[idx] = self.visited_gen;
                            self.q.push(idx);
                            count += 1;
                        }
                    }
                }
            }
        }

        count == remaining
    }
}

fn check_parity(cx: usize, cy: usize, rem_even: usize, rem_odd: usize, require_cycle: bool) -> bool {
    let is_even = (cx + cy) % 2 == 0;
    
    if require_cycle {
        if is_even {
            rem_odd == rem_even + 1
        } else {
            rem_even == rem_odd
        }
    } else {
        if is_even {
            rem_odd == rem_even || rem_odd == rem_even + 1
        } else {
            rem_even == rem_odd || rem_even == rem_odd + 1
        }
    }
}

fn solve(
    grid: &mut Vec<Vec<bool>>,
    degrees: &mut Vec<Vec<i32>>,
    state: &mut SolverState,
    path: &mut Vec<(usize, usize)>,
    rem_even: usize,
    rem_odd: usize,
    start_neighbors: usize,
) -> bool {
    if state.start_time.elapsed() > Duration::from_secs(295) {
        return false;
    }

    let (cx, cy) = *path.last().unwrap();
    let remaining = state.empty_count - path.len();

    if remaining == 0 {
        if state.require_cycle {
            if cx + cy == 1 { // adjacent to (0,0)
                return true;
            }
            return false;
        }
        return true;
    }

    if !check_parity(cx, cy, rem_even, rem_odd, state.require_cycle) {
        return false;
    }

    if !state.is_connected(grid, cx, cy, remaining) {
        return false;
    }

    let mut neighbors = Vec::new();
    let dx = [0, 0, 1, -1];
    let dy = [1, -1, 0, 0];

    let mut forced_neighbor = None;
    let mut deg0_count_here = 0;

    for i in 0..4 {
        let nx = cx as isize + dx[i];
        let ny = cy as isize + dy[i];

        if nx >= 0 && nx < state.width as isize && ny >= 0 && ny < state.height as isize {
            let nx = nx as usize;
            let ny = ny as usize;
            if !grid[ny][nx] {
                let deg = degrees[ny][nx];
                if deg == 0 {
                    deg0_count_here += 1;
                    forced_neighbor = Some((deg, nx, ny));
                }
                neighbors.push((deg, nx, ny));
            }
        }
    }

    if deg0_count_here > 1 {
        return false;
    }

    if let Some(forced) = forced_neighbor {
        neighbors.clear();
        neighbors.push(forced);
    } else {
        neighbors.sort_by_key(|k| k.0);
    }

    for (_, nx, ny) in neighbors {
        let is_start_neighbor = (nx + ny) == 1;
        let next_start_neighbors = if is_start_neighbor { start_neighbors.saturating_sub(1) } else { start_neighbors };

        if state.require_cycle && remaining > 1 && next_start_neighbors == 0 {
            continue;
        }

        let mut deg0_count = 0;
        for i in 0..4 {
            let nnx = nx as isize + dx[i];
            let nny = ny as isize + dy[i];
            if nnx >= 0 && nnx < state.width as isize && nny >= 0 && nny < state.height as isize {
                let nnx = nnx as usize;
                let nny = nny as usize;
                if !grid[nny][nnx] {
                    if degrees[nny][nnx] == 1 {
                        deg0_count += 1;
                    }
                }
            }
        }
        
        if deg0_count > 1 {
            continue;
        }

        grid[ny][nx] = true;
        path.push((nx, ny));
        let is_nx_even = (nx + ny) % 2 == 0;
        let next_rem_even = if is_nx_even { rem_even - 1 } else { rem_even };
        let next_rem_odd = if is_nx_even { rem_odd } else { rem_odd - 1 };
        
        for i in 0..4 {
            let nnx = nx as isize + dx[i];
            let nny = ny as isize + dy[i];
            if nnx >= 0 && nnx < state.width as isize && nny >= 0 && nny < state.height as isize {
                let nnx = nnx as usize;
                let nny = nny as usize;
                if !grid[nny][nnx] {
                    degrees[nny][nnx] -= 1;
                }
            }
        }

        if solve(grid, degrees, state, path, next_rem_even, next_rem_odd, next_start_neighbors) {
            return true;
        }

        for i in 0..4 {
            let nnx = nx as isize + dx[i];
            let nny = ny as isize + dy[i];
            if nnx >= 0 && nnx < state.width as isize && nny >= 0 && nny < state.height as isize {
                let nnx = nnx as usize;
                let nny = nny as usize;
                if !grid[nny][nnx] {
                    degrees[nny][nnx] += 1;
                }
            }
        }

        path.pop();
        grid[ny][nx] = false;
    }

    false
}

fn main() {
    let start_time = Instant::now();
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = if let Some(Ok(line)) = lines.next() {
        line
    } else {
        return;
    };
    
    let mut parts = first_line.split_whitespace();
    let width: usize = match parts.next() {
        Some(w) => w.parse().unwrap(),
        None => return,
    };
    let height: usize = parts.next().unwrap().parse().unwrap();
    let obstacle_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle == 1;

    let mut grid = vec![vec![false; width]; height];
    for _ in 0..obstacle_count {
        if let Some(Ok(line)) = lines.next() {
            let mut parts = line.split_whitespace();
            let x: usize = parts.next().unwrap().parse().unwrap();
            let y: usize = parts.next().unwrap().parse().unwrap();
            grid[y][x] = true;
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if grid[0][0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut empty_count = 0;
    let mut even_count = 0;
    let mut odd_count = 0;

    let mut degrees = vec![vec![0; width]; height];

    for y in 0..height {
        for x in 0..width {
            if !grid[y][x] {
                empty_count += 1;
                if (x + y) % 2 == 0 {
                    even_count += 1;
                } else {
                    odd_count += 1;
                }
            }
        }
    }

    let mut start_neighbors = 0;

    for y in 0..height {
        for x in 0..width {
            if !grid[y][x] {
                let mut deg = 0;
                let dx = [0, 0, 1, -1];
                let dy = [1, -1, 0, 0];
                for i in 0..4 {
                    let nx = x as isize + dx[i];
                    let ny = y as isize + dy[i];
                    if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
                        if !grid[ny as usize][nx as usize] {
                            deg += 1;
                            if x == 0 && y == 0 {
                                start_neighbors += 1;
                            }
                        }
                    }
                }
                degrees[y][x] = deg;
            }
        }
    }

    if require_cycle && empty_count % 2 != 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    if even_count > odd_count + 1 || odd_count > even_count + 1 {
        writeln!(out, "0").unwrap();
        return;
    }
    
    if require_cycle && even_count != odd_count {
        writeln!(out, "0").unwrap();
        return;
    }

    if require_cycle && start_neighbors == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path = Vec::with_capacity(empty_count);
    path.push((0, 0));
    grid[0][0] = true;

    let dx = [0, 0, 1, -1];
    let dy = [1, -1, 0, 0];
    for i in 0..4 {
        let nx = 0isize + dx[i];
        let ny = 0isize + dy[i];
        if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
            let nx = nx as usize;
            let ny = ny as usize;
            if !grid[ny][nx] {
                degrees[ny][nx] -= 1;
            }
        }
    }

    let mut state = SolverState {
        width,
        height,
        empty_count,
        require_cycle,
        start_time,
        visited_gen: 0,
        visited: vec![0; width * height],
        q: Vec::with_capacity(width * height),
    };

    if solve(&mut grid, &mut degrees, &mut state, &mut path, even_count - 1, odd_count, start_neighbors) {
        writeln!(out, "{}", path.len()).unwrap();
        for (x, y) in path {
            writeln!(out, "{} {}", x, y).unwrap();
        }
    } else {
        writeln!(out, "0").unwrap();
    }
}
