#[derive(Clone, Eq, PartialEq)]
struct AStarState {
    cost: i32,
    position: Point,
}

fn heuristic(a: Point, b: Point) -> i32 {
    let dx = (a.x - b.x).abs();
    let dy = (a.y - b.y).abs();
    if dx > dy {
        14 * dy + 10 * (dx - dy)
    } else {
        14 * dx + 10 * (dy - dx)
    }
}

fn is_safe(nx: i32, ny: i32, map_bits: &[u8], width: i32, height: i32, start: Point, target: Point) -> bool {
    if (nx - target.x).abs() <= 2 && (ny - target.y).abs() <= 2 {
        if nx < 0 || ny < 0 || nx >= width || ny >= height { return false; }
        let idx = (ny as usize) * (width as usize) + (nx as usize);
        return (map_bits[idx / 8] & (1 << (idx % 8))) == 0;
    }
    if (nx - start.x).abs() <= 2 && (ny - start.y).abs() <= 2 {
        if nx < 0 || ny < 0 || nx >= width || ny >= height { return false; }
        let idx = (ny as usize) * (width as usize) + (nx as usize);
        return (map_bits[idx / 8] & (1 << (idx % 8))) == 0;
    }
    
    for sx in -1..=1 {
        for sy in -1..=1 {
            let cx = nx + sx;
            let cy = ny + sy;
            if cx >= 0 && cy >= 0 && cx < width && cy < height {
                let idx = (cy as usize) * (width as usize) + (cx as usize);
                if (map_bits[idx / 8] & (1 << (idx % 8))) != 0 {
                    return false;
                }
            } else {
                return false;
            }
        }
    }
    true
}

fn has_line_of_sight(p1: Point, p2: Point, map_bits: &[u8], width: usize, height: usize, start: Point, target: Point) -> bool {
    let mut x0 = p1.x;
    let mut y0 = p1.y;
    let x1 = p2.x;
    let y1 = p2.y;
    
    let dx = (x1 - x0).abs();
    let sx = if x0 < x1 { 1 } else { -1 };
    let dy = -(y1 - y0).abs();
    let sy = if y0 < y1 { 1 } else { -1 };
    let mut err = dx + dy;
    
    loop {
        if !is_safe(x0, y0, map_bits, width as i32, height as i32, start, target) {
            return false;
        }
        
        if x0 == x1 && y0 == y1 { break; }
        let e2 = 2 * err;
        if e2 >= dy { err += dy; x0 += sx; }
        if e2 <= dx { err += dx; y0 += sy; }
    }
    true
}

fn find_path(map_bits: &[u8], width: usize, height: usize, start_x: i32, start_y: i32, target_x: i32, target_y: i32) -> Vec<Point> {
    let start = Point { x: start_x, y: start_y };
    let target = Point { x: target_x, y: target_y };
    
    let mut open = BinaryHeap::new();
    let mut cost_so_far = HashMap::new();
    let mut came_from = HashMap::new();
    
    open.push(AStarState { cost: 0, position: start });
    cost_so_far.insert(start, 0);
    
    let mut nodes_evaluated = 0;
    let mut path = Vec::new();
    
    while let Some(AStarState { position, .. }) = open.pop() {
        if position == target {
            let mut curr = target;
            while curr != start {
                path.push(curr);
                curr = came_from[&curr];
            }
            path.push(start);
            path.reverse();
            break;
        }
        
        nodes_evaluated += 1;
        if nodes_evaluated > 2_000_000 {
            break; // give up A*, use fallback
        }
        
        for dx in -1..=1 {
            for dy in -1..=1 {
                if dx == 0 && dy == 0 { continue; }
                let nx = position.x + dx;
                let ny = position.y + dy;
                
                if !is_safe(nx, ny, map_bits, width as i32, height as i32, start, target) {
                    continue;
                }
                
                let move_cost = if dx == 0 || dy == 0 { 10 } else { 14 };
                let new_cost = cost_so_far[&position] + move_cost;
                
                let next_p = Point { x: nx, y: ny };
                if new_cost < *cost_so_far.get(&next_p).unwrap_or(&i32::MAX) {
                    cost_so_far.insert(next_p, new_cost);
                    let priority = new_cost + heuristic(next_p, target);
                    open.push(AStarState { cost: priority, position: next_p });
                    came_from.insert(next_p, position);
                }
            }
        }
    }
    path
}

fn smooth_path(path: &[Point], map_bits: &[u8], width: usize, height: usize, start: Point, target: Point) -> Vec<Point> {
    if path.is_empty() { return vec![]; }
    let mut smooth = vec![path[0]];
    let mut curr = 0;
    
    while curr < path.len() - 1 {
        let mut furthest = curr + 1;
        for i in (curr + 2)..path.len() {
            if has_line_of_sight(path[curr], path[i], map_bits, width, height, start, target) {
                furthest = i;
            } else {
                break;
            }
        }
        smooth.push(path[furthest]);
        curr = furthest;
    }
    smooth
}
