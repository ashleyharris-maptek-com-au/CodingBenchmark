using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class Program
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 v, double s) => new Vec3(v.X * s, v.Y * s, v.Z * s);
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalized() { var l = Length(); return l > 1e-10 ? new Vec3(X / l, Y / l, Z / l) : new Vec3(0, 0, 0); }
    }

    struct Quaternion
    {
        public double W, X, Y, Z;
        public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        public static Quaternion Identity => new Quaternion(1, 0, 0, 0);

        public static Quaternion FromAxisAngle(Vec3 axis, double angle)
        {
            var a = axis.Normalized();
            var s = Math.Sin(angle / 2);
            return new Quaternion(Math.Cos(angle / 2), a.X * s, a.Y * s, a.Z * s);
        }

        public Vec3 Rotate(Vec3 p)
        {
            double tx = 2 * (Y * p.Z - Z * p.Y);
            double ty = 2 * (Z * p.X - X * p.Z);
            double tz = 2 * (X * p.Y - Y * p.X);
            return new Vec3(
                p.X + W * tx + Y * tz - Z * ty,
                p.Y + W * ty + Z * tx - X * tz,
                p.Z + W * tz + X * ty - Y * tx);
        }
    }

    struct AABB
    {
        public Vec3 Min, Max;
        public AABB(Vec3 min, Vec3 max) { Min = min; Max = max; }
        public Vec3 Size => Max - Min;

        public bool Intersects(AABB other)
        {
            return Min.X < other.Max.X && Max.X > other.Min.X &&
                   Min.Y < other.Max.Y && Max.Y > other.Min.Y &&
                   Min.Z < other.Max.Z && Max.Z > other.Min.Z;
        }

        public bool ContainedIn(AABB container)
        {
            return Min.X >= container.Min.X - 1e-9 && Max.X <= container.Max.X + 1e-9 &&
                   Min.Y >= container.Min.Y - 1e-9 && Max.Y <= container.Max.Y + 1e-9 &&
                   Min.Z >= container.Min.Z - 1e-9 && Max.Z <= container.Max.Z + 1e-9;
        }
    }

    static List<Vec3> vertices = new List<Vec3>();
    static List<int[]> faces = new List<int[]>();
    static double containerW, containerH, containerD;

    static readonly Quaternion[] rotations = GenerateRotations();

    static Quaternion[] GenerateRotations()
    {
        var rots = new List<Quaternion>();
        rots.Add(Quaternion.Identity);

        double[] angles = { 0, Math.PI / 2, Math.PI, 3 * Math.PI / 2 };
        Vec3[] axes = { new Vec3(1, 0, 0), new Vec3(0, 1, 0), new Vec3(0, 0, 1) };

        foreach (var axis in axes)
        {
            foreach (var angle in angles)
            {
                if (angle != 0)
                    rots.Add(Quaternion.FromAxisAngle(axis, angle));
            }
        }

        rots.Add(Quaternion.FromAxisAngle(new Vec3(1, 0, 0), Math.PI / 2));
        rots.Add(Quaternion.FromAxisAngle(new Vec3(1, 0, 0), -Math.PI / 2));
        rots.Add(Quaternion.FromAxisAngle(new Vec3(0, 1, 0), Math.PI / 2));
        rots.Add(Quaternion.FromAxisAngle(new Vec3(0, 1, 0), -Math.PI / 2));

        return rots.Distinct(new QuaternionComparer()).ToArray();
    }

    class QuaternionComparer : IEqualityComparer<Quaternion>
    {
        public bool Equals(Quaternion a, Quaternion b)
        {
            return Math.Abs(a.W - b.W) < 0.01 && Math.Abs(a.X - b.X) < 0.01 &&
                   Math.Abs(a.Y - b.Y) < 0.01 && Math.Abs(a.Z - b.Z) < 0.01;
        }
        public int GetHashCode(Quaternion q) => 0;
    }

    static AABB ComputeAABB(Quaternion rotation, Vec3 translation)
    {
        Vec3 min = new Vec3(double.MaxValue, double.MaxValue, double.MaxValue);
        Vec3 max = new Vec3(double.MinValue, double.MinValue, double.MinValue);

        foreach (var v in vertices)
        {
            var rv = rotation.Rotate(v);
            var tv = rv + translation;
            min = new Vec3(Math.Min(min.X, tv.X), Math.Min(min.Y, tv.Y), Math.Min(min.Z, tv.Z));
            max = new Vec3(Math.Max(max.X, tv.X), Math.Max(max.Y, tv.Y), Math.Max(max.Z, tv.Z));
        }
        return new AABB(min, max);
    }

    static AABB ComputeRotatedBounds(Quaternion rotation)
    {
        Vec3 min = new Vec3(double.MaxValue, double.MaxValue, double.MaxValue);
        Vec3 max = new Vec3(double.MinValue, double.MinValue, double.MinValue);

        foreach (var v in vertices)
        {
            var rv = rotation.Rotate(v);
            min = new Vec3(Math.Min(min.X, rv.X), Math.Min(min.Y, rv.Y), Math.Min(min.Z, rv.Z));
            max = new Vec3(Math.Max(max.X, rv.X), Math.Max(max.Y, rv.Y), Math.Max(max.Z, rv.Z));
        }
        return new AABB(min, max);
    }

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        var line = Console.ReadLine().Split();
        int V = int.Parse(line[0]);
        int F = int.Parse(line[1]);

        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Split();
            vertices.Add(new Vec3(
                double.Parse(parts[0], CultureInfo.InvariantCulture),
                double.Parse(parts[1], CultureInfo.InvariantCulture),
                double.Parse(parts[2], CultureInfo.InvariantCulture)));
        }

        for (int i = 0; i < F; i++)
        {
            var parts = Console.ReadLine().Split();
            faces.Add(new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) });
        }

        var containerLine = Console.ReadLine().Split();
        containerW = double.Parse(containerLine[0], CultureInfo.InvariantCulture);
        containerH = double.Parse(containerLine[1], CultureInfo.InvariantCulture);
        containerD = double.Parse(containerLine[2], CultureInfo.InvariantCulture);

        AABB container = new AABB(new Vec3(0, 0, 0), new Vec3(containerW, containerH, containerD));

        Quaternion bestRotation = Quaternion.Identity;
        AABB bestBounds = ComputeRotatedBounds(Quaternion.Identity);
        double bestVolume = double.MaxValue;

        foreach (var rot in rotations)
        {
            var bounds = ComputeRotatedBounds(rot);
            var size = bounds.Size;
            double vol = size.X * size.Y * size.Z;

            if (size.X <= containerW + 1e-9 && size.Y <= containerH + 1e-9 && size.Z <= containerD + 1e-9)
            {
                if (vol < bestVolume)
                {
                    bestVolume = vol;
                    bestRotation = rot;
                    bestBounds = bounds;
                }
            }
        }

        var placements = new List<(Vec3 translation, Quaternion rotation)>();

        var objSize = bestBounds.Size;
        var objMin = bestBounds.Min;

        if (objSize.X > containerW || objSize.Y > containerH || objSize.Z > containerD)
        {
            Console.WriteLine("0");
            return;
        }

        double containerVolume = containerW * containerH * containerD;
        bool useFastPacking = containerVolume > 1e6 || (containerW > 100 && containerH > 100 && containerD > 100);

        double spacing = 1e-6;

        if (useFastPacking)
        {
            int countX = (int)((containerW + spacing) / (objSize.X + spacing));
            int countY = (int)((containerH + spacing) / (objSize.Y + spacing));
            int countZ = (int)((containerD + spacing) / (objSize.Z + spacing));

            countX = Math.Max(1, countX);
            countY = Math.Max(1, countY);
            countZ = Math.Max(1, countZ);

            for (int iz = 0; iz < countZ; iz++)
            {
                for (int iy = 0; iy < countY; iy++)
                {
                    for (int ix = 0; ix < countX; ix++)
                    {
                        double tx = ix * (objSize.X + spacing) - objMin.X;
                        double ty = iy * (objSize.Y + spacing) - objMin.Y;
                        double tz = iz * (objSize.Z + spacing) - objMin.Z;

                        Vec3 translation = new Vec3(tx, ty, tz);
                        AABB placedBounds = ComputeAABB(bestRotation, translation);

                        if (placedBounds.ContainedIn(container))
                        {
                            placements.Add((translation, bestRotation));
                        }
                    }
                }
            }
        }
        else
        {
            var placedAABBs = new List<AABB>();

            var rotsToTry = rotations.Where(r =>
            {
                var b = ComputeRotatedBounds(r);
                return b.Size.X <= containerW + 1e-9 && b.Size.Y <= containerH + 1e-9 && b.Size.Z <= containerD + 1e-9;
            }).ToList();

            if (rotsToTry.Count == 0)
            {
                Console.WriteLine("0");
                return;
            }

            double stepX = Math.Max(0.5, objSize.X / 4);
            double stepY = Math.Max(0.5, objSize.Y / 4);
            double stepZ = Math.Max(0.5, objSize.Z / 4);

            if (containerVolume > 1e4)
            {
                stepX = objSize.X;
                stepY = objSize.Y;
                stepZ = objSize.Z;
            }

            for (double z = 0; z <= containerD; z += stepZ)
            {
                for (double y = 0; y <= containerH; y += stepY)
                {
                    for (double x = 0; x <= containerW; x += stepX)
                    {
                        foreach (var rot in rotsToTry)
                        {
                            var bounds = ComputeRotatedBounds(rot);
                            Vec3 translation = new Vec3(x - bounds.Min.X, y - bounds.Min.Y, z - bounds.Min.Z);
                            AABB placedBounds = ComputeAABB(rot, translation);

                            if (!placedBounds.ContainedIn(container))
                                continue;

                            bool collision = false;
                            foreach (var existing in placedAABBs)
                            {
                                if (placedBounds.Intersects(existing))
                                {
                                    collision = true;
                                    break;
                                }
                            }

                            if (!collision)
                            {
                                placements.Add((translation, rot));
                                placedAABBs.Add(placedBounds);
                                break;
                            }
                        }
                    }
                }
            }
        }

        Console.WriteLine(placements.Count);
        foreach (var (t, q) in placements)
        {
            Console.WriteLine($"{t.X:G17} {t.Y:G17} {t.Z:G17} {q.W:G17} {q.X:G17} {q.Y:G17} {q.Z:G17}");
        }
    }
}
