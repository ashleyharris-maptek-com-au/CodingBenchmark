use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Default)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    updated: bool,
}

struct MapInfo {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

fn parse_initial_map<R: BufRead>(reader: &mut R) -> io::Result<MapInfo> {
    let mut first_line = String::new();
    reader.read_line(&mut first_line)?;
    let parts: Vec<f64> = first_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();

    let height = parts[1] as usize;
    let gravity = parts[2];
    let max_thrust = parts[3];

    let mut second_line = String::new();
    reader.read_line(&mut second_line)?;
    let parts2: Vec<f64> = second_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();

    let target_x = parts2[2];
    let target_y = parts2[3];

    for _ in 0..height {
        let mut line = String::new();
        reader.read_line(&mut line)?;
    }

    Ok(MapInfo {
        gravity,
        max_thrust,
        target_x,
        target_y,
    })
}

fn normalize_angle(a: f64) -> f64 {
    let pi = std::f64::consts::PI;
    let mut result = a;
    while result > pi { result -= 2.0 * pi; }
    while result < -pi { result += 2.0 * pi; }
    result
}

fn main() {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    let map_info = match parse_initial_map(&mut handle) {
        Ok(info) => info,
        Err(_) => return,
    };

    drop(handle);

    let gravity = map_info.gravity;
    let max_thrust = map_info.max_thrust;
    let target_x = map_info.target_x;
    let target_y = map_info.target_y;

    let state = Arc::new(Mutex::new(LanderState::default()));
    let state_clone = Arc::clone(&state);

    thread::spawn(move || {
        let stdin = io::stdin();
        let handle = stdin.lock();

        for line in handle.lines() {
            if let Ok(line) = line {
                let parts: Vec<f64> = line
                    .split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();

                if parts.len() >= 7 {
                    let mut s = state_clone.lock().unwrap();
                    s.x = parts[0];
                    s.y = parts[1];
                    s.vx = parts[2];
                    s.vy = parts[3];
                    s.angle = parts[4];
                    s.angular_vel = parts[5];
                    s.fuel = parts[6];
                    s.updated = true;
                }
            }
        }
    });

    let mut stdout = io::stdout();
    let control_interval = Duration::from_millis(20);

    loop {
        let start = Instant::now();

        let s = {
            let guard = state.lock().unwrap();
            guard.clone()
        };

        if !s.updated {
            let _ = writeln!(stdout, "0.0 0.0");
            let _ = stdout.flush();
            thread::sleep(control_interval);
            continue;
        }

        let dx = target_x - s.x;
        let dy = target_y - s.y;
        let dist = (dx * dx + dy * dy).sqrt();
        let altitude = dy;

        let max_hvel = 8.0;
        let desired_vx = (dx * 0.4).clamp(-max_hvel, max_hvel);

        let safe_vy = if altitude > 100.0 {
            12.0
        } else if altitude > 50.0 {
            8.0
        } else if altitude > 20.0 {
            4.0
        } else if altitude > 10.0 {
            2.5
        } else if altitude > 3.0 {
            1.5
        } else {
            0.8
        };

        let desired_vy = if altitude > 0.0 {
            safe_vy.min(altitude * 0.3 + 0.5)
        } else {
            -1.0
        };

        let vx_err = desired_vx - s.vx;
        let vy_err = desired_vy - s.vy;

        let kp_h = 1.5;
        let kp_v = 2.5;

        let ax = vx_err * kp_h;
        let ay = vy_err * kp_v - gravity;

        let desired_angle = if ay.abs() < 0.001 && ax.abs() < 0.001 {
            0.0
        } else {
            ax.atan2(-ay)
        };

        let accel_mag = (ax * ax + ay * ay).sqrt();
        let base_thrust = if max_thrust > 0.0 {
            accel_mag / max_thrust
        } else {
            0.0
        };

        let angle_err = normalize_angle(desired_angle - s.angle);
        let kp_ang = 4.0;
        let kd_ang = 2.0;
        let turn = (angle_err * kp_ang - s.angular_vel * kd_ang).clamp(-1.0, 1.0);

        let thrust_scale = (1.0 - angle_err.abs() / std::f64::consts::PI).max(0.3);
        let thrust = (base_thrust * thrust_scale).clamp(0.0, 1.0);

        let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
        let near = dist < 2.0 && altitude.abs() < 1.5;

        let (final_thrust, final_turn) = if near && speed < 2.0 {
            (0.0, 0.0)
        } else if s.fuel <= 0.0 {
            (0.0, turn)
        } else {
            (thrust, turn)
        };

        let _ = writeln!(stdout, "{:.4} {:.4}", final_thrust, final_turn);
        let _ = stdout.flush();

        let elapsed = start.elapsed();
        if elapsed < control_interval {
            thread::sleep(control_interval - elapsed);
        }
    }
}
