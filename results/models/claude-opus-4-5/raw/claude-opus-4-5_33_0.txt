using System;
using System.Diagnostics;

class Program
{
    static int N;
    static int[,] flow;
    static int[,] dist;
    static Random rnd = new Random(42);

    static void Main()
    {
        N = int.Parse(Console.ReadLine().Trim());
        flow = new int[N, N];
        dist = new int[N, N];

        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N; j++)
                flow[i, j] = int.Parse(parts[j]);
        }

        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N; j++)
                dist[i, j] = int.Parse(parts[j]);
        }

        var sw = Stopwatch.StartNew();
        var timeLimit = TimeSpan.FromSeconds(270);

        int[] bestP = GreedyConstruction();
        long bestCost = CalculateCost(bestP);

        int[] p = (int[])bestP.Clone();
        long cost = bestCost;

        // Local search improvement
        bool improved = true;
        while (improved && sw.Elapsed < timeLimit)
        {
            improved = false;
            for (int i = 0; i < N - 1 && sw.Elapsed < timeLimit; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    long delta = SwapDelta(p, i, j);
                    if (delta < 0)
                    {
                        Swap(p, i, j);
                        cost += delta;
                        improved = true;
                        if (cost < bestCost)
                        {
                            bestCost = cost;
                            Array.Copy(p, bestP, N);
                        }
                    }
                }
            }
        }

        // Simulated annealing for remaining time
        double temp = Math.Max(1, bestCost / (double)(N * N + 1));
        double cooling = N <= 100 ? 0.9999 : (N <= 500 ? 0.99995 : 0.999995);

        Array.Copy(bestP, p, N);
        cost = bestCost;

        while (sw.Elapsed < timeLimit)
        {
            for (int iter = 0; iter < 1000 && sw.Elapsed < timeLimit; iter++)
            {
                int i = rnd.Next(N);
                int j = rnd.Next(N);
                if (i == j) continue;

                long delta = SwapDelta(p, i, j);

                if (delta < 0 || rnd.NextDouble() < Math.Exp(-delta / temp))
                {
                    Swap(p, i, j);
                    cost += delta;

                    if (cost < bestCost)
                    {
                        bestCost = cost;
                        Array.Copy(p, bestP, N);
                    }
                }
                temp *= cooling;
            }
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestP));
    }

    static int[] GreedyConstruction()
    {
        int[] p = new int[N];
        bool[] usedLoc = new bool[N];
        bool[] assigned = new bool[N];

        // Sort facilities by total flow (descending)
        int[] order = new int[N];
        long[] totalFlow = new long[N];
        for (int i = 0; i < N; i++)
        {
            order[i] = i;
            for (int j = 0; j < N; j++)
                totalFlow[i] += flow[i, j] + flow[j, i];
        }
        Array.Sort(order, (a, b) => totalFlow[b].CompareTo(totalFlow[a]));

        // Sort locations by total distance (ascending)
        int[] locOrder = new int[N];
        long[] totalDist = new long[N];
        for (int i = 0; i < N; i++)
        {
            locOrder[i] = i;
            for (int j = 0; j < N; j++)
                totalDist[i] += dist[i, j] + dist[j, i];
        }
        Array.Sort(locOrder, (a, b) => totalDist[a].CompareTo(totalDist[b]));

        // Assign high-flow facilities to central (low total distance) locations
        for (int k = 0; k < N; k++)
        {
            int fac = order[k];
            long bestContrib = long.MaxValue;
            int bestLoc = -1;

            foreach (int loc in locOrder)
            {
                if (usedLoc[loc]) continue;

                long contrib = 0;
                for (int other = 0; other < N; other++)
                {
                    if (assigned[other])
                    {
                        contrib += (long)flow[fac, other] * dist[loc, p[other]];
                        contrib += (long)flow[other, fac] * dist[p[other], loc];
                    }
                }

                if (contrib < bestContrib)
                {
                    bestContrib = contrib;
                    bestLoc = loc;
                }
            }

            p[fac] = bestLoc;
            usedLoc[bestLoc] = true;
            assigned[fac] = true;
        }

        return p;
    }

    static long CalculateCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                cost += (long)flow[i, j] * dist[p[i], p[j]];
        return cost;
    }

    static long SwapDelta(int[] p, int i, int j)
    {
        long delta = 0;
        int pi = p[i], pj = p[j];

        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];

            delta -= (long)flow[i, k] * dist[pi, pk];
            delta -= (long)flow[k, i] * dist[pk, pi];
            delta -= (long)flow[j, k] * dist[pj, pk];
            delta -= (long)flow[k, j] * dist[pk, pj];

            delta += (long)flow[i, k] * dist[pj, pk];
            delta += (long)flow[k, i] * dist[pk, pj];
            delta += (long)flow[j, k] * dist[pi, pk];
            delta += (long)flow[k, j] * dist[pk, pi];
        }

        delta -= (long)flow[i, j] * dist[pi, pj];
        delta -= (long)flow[j, i] * dist[pj, pi];
        delta += (long)flow[i, j] * dist[pj, pi];
        delta += (long)flow[j, i] * dist[pi, pj];

        return delta;
    }

    static void Swap(int[] p, int i, int j)
    {
        int t = p[i];
        p[i] = p[j];
        p[j] = t;
    }
}
