using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    const double MU = 398600.4418; // Earth gravitational parameter km³/s²
    const double EARTH_RADIUS = 6371.0;

    static void Main()
    {
        var lines = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
            lines.Add(line);

        var firstLine = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(firstLine[0]);

        var startOrbit = ParseOrbit(lines[1]);
        var stations = new Orbit[n];
        for (int i = 0; i < n; i++)
            stations[i] = ParseOrbit(lines[2 + i]);

        double[,] costMatrix = ComputeCostMatrix(startOrbit, stations);
        int[] route = SolveTSP(n, costMatrix);

        Console.WriteLine(string.Join(" ", route));
    }

    static Orbit ParseOrbit(string line)
    {
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        return new Orbit
        {
            X = double.Parse(parts[0]),
            Y = double.Parse(parts[1]),
            Z = double.Parse(parts[2]),
            VX = double.Parse(parts[3]),
            VY = double.Parse(parts[4]),
            VZ = double.Parse(parts[5])
        };
    }

    static double[,] ComputeCostMatrix(Orbit start, Orbit[] stations)
    {
        int n = stations.Length;
        double[,] cost = new double[n + 1, n + 1];

        var orbitalParams = new OrbitalParams[n + 1];
        orbitalParams[0] = ComputeOrbitalParams(start);
        for (int i = 0; i < n; i++)
            orbitalParams[i + 1] = ComputeOrbitalParams(stations[i]);

        for (int i = 0; i <= n; i++)
        {
            for (int j = 0; j <= n; j++)
            {
                if (i == j)
                    cost[i, j] = 0;
                else
                    cost[i, j] = EstimateTransferDeltaV(orbitalParams[i], orbitalParams[j]);
            }
        }
        return cost;
    }

    static OrbitalParams ComputeOrbitalParams(Orbit o)
    {
        double r = Math.Sqrt(o.X * o.X + o.Y * o.Y + o.Z * o.Z);
        double v = Math.Sqrt(o.VX * o.VX + o.VY * o.VY + o.VZ * o.VZ);

        double energy = v * v / 2 - MU / r;
        double a = -MU / (2 * energy);

        double hx = o.Y * o.VZ - o.Z * o.VY;
        double hy = o.Z * o.VX - o.X * o.VZ;
        double hz = o.X * o.VY - o.Y * o.VX;
        double h = Math.Sqrt(hx * hx + hy * hy + hz * hz);

        double inc = Math.Acos(Math.Max(-1, Math.Min(1, hz / h)));

        double p = h * h / MU;
        double ecc = Math.Sqrt(Math.Max(0, 1 - p / a));

        double raan = 0;
        double nMag = Math.Sqrt(hx * hx + hy * hy);
        if (nMag > 1e-10)
        {
            raan = Math.Atan2(hx, -hy);
            if (raan < 0) raan += 2 * Math.PI;
        }

        return new OrbitalParams { A = a, E = ecc, Inc = inc, RAAN = raan, H = h };
    }

    static double EstimateTransferDeltaV(OrbitalParams from, OrbitalParams to)
    {
        if (from.A <= 0 || to.A <= 0) return 1e9;

        double r1 = from.A * (1 - from.E * from.E) / (1 + from.E);
        double r2 = to.A * (1 - to.E * to.E) / (1 + to.E);

        if (r1 <= EARTH_RADIUS || r2 <= EARTH_RADIUS) return 1e9;

        double v1 = Math.Sqrt(MU / r1);
        double v2 = Math.Sqrt(MU / r2);

        double aTransfer = (r1 + r2) / 2;
        double vTransfer1 = Math.Sqrt(MU * (2 / r1 - 1 / aTransfer));
        double vTransfer2 = Math.Sqrt(MU * (2 / r2 - 1 / aTransfer));

        double dv1 = Math.Abs(vTransfer1 - v1);
        double dv2 = Math.Abs(v2 - vTransfer2);
        double dvHohmann = dv1 + dv2;

        double incChange = Math.Abs(from.Inc - to.Inc);
        double raanChange = Math.Abs(from.RAAN - to.RAAN);
        if (raanChange > Math.PI) raanChange = 2 * Math.PI - raanChange;

        double planeChange = Math.Sqrt(incChange * incChange +
            Math.Pow(raanChange * Math.Sin((from.Inc + to.Inc) / 2), 2));

        double avgV = (v1 + v2) / 2;
        double dvPlane = 2 * avgV * Math.Sin(planeChange / 2);

        double eccPenalty = Math.Abs(from.E - to.E) * avgV * 0.5;

        return dvHohmann + dvPlane + eccPenalty;
    }

    static int[] SolveTSP(int n, double[,] cost)
    {
        if (n == 1) return new[] { 0 };
        if (n == 2) return cost[1, 2] + cost[2, 1] < cost[2, 1] + cost[1, 2] ? new[] { 0, 1 } : new[] { 0, 1 };

        if (n <= 10)
            return SolveTSPBruteForce(n, cost);
        else
            return SolveTSPHeuristic(n, cost);
    }

    static int[] SolveTSPBruteForce(int n, double[,] cost)
    {
        int[] best = null;
        double bestCost = double.MaxValue;

        var perm = Enumerable.Range(0, n).ToArray();
        do
        {
            double c = cost[0, perm[0] + 1];
            for (int i = 0; i < n - 1; i++)
                c += cost[perm[i] + 1, perm[i + 1] + 1];

            if (c < bestCost)
            {
                bestCost = c;
                best = (int[])perm.Clone();
            }
        } while (NextPermutation(perm));

        return best;
    }

    static bool NextPermutation(int[] arr)
    {
        int i = arr.Length - 2;
        while (i >= 0 && arr[i] >= arr[i + 1]) i--;
        if (i < 0) return false;

        int j = arr.Length - 1;
        while (arr[j] <= arr[i]) j--;

        (arr[i], arr[j]) = (arr[j], arr[i]);
        Array.Reverse(arr, i + 1, arr.Length - i - 1);
        return true;
    }

    static int[] SolveTSPHeuristic(int n, double[,] cost)
    {
        int[] route = NearestNeighbor(n, cost);
        route = TwoOpt(n, cost, route);

        if (n <= 50)
        {
            for (int iter = 0; iter < 5; iter++)
            {
                var newRoute = RandomNearestNeighbor(n, cost, iter);
                newRoute = TwoOpt(n, cost, newRoute);
                if (RouteCost(n, cost, newRoute) < RouteCost(n, cost, route))
                    route = newRoute;
            }
        }

        if (n >= 20)
            route = SimulatedAnnealing(n, cost, route);

        return route;
    }

    static int[] NearestNeighbor(int n, double[,] cost)
    {
        var route = new List<int>();
        var visited = new bool[n];
        int current = 0;

        for (int i = 0; i < n; i++)
        {
            double minCost = double.MaxValue;
            int next = -1;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j])
                {
                    double c = (i == 0) ? cost[0, j + 1] : cost[current + 1, j + 1];
                    if (c < minCost)
                    {
                        minCost = c;
                        next = j;
                    }
                }
            }

            route.Add(next);
            visited[next] = true;
            current = next;
        }

        return route.ToArray();
    }

    static int[] RandomNearestNeighbor(int n, double[,] cost, int seed)
    {
        var rng = new Random(seed * 12345);
        var route = new List<int>();
        var visited = new bool[n];
        int current = 0;

        for (int i = 0; i < n; i++)
        {
            var candidates = new List<(int idx, double c)>();

            for (int j = 0; j < n; j++)
            {
                if (!visited[j])
                {
                    double c = (i == 0) ? cost[0, j + 1] : cost[current + 1, j + 1];
                    candidates.Add((j, c));
                }
            }

            candidates.Sort((a, b) => a.c.CompareTo(b.c));
            int pickRange = Math.Min(3, candidates.Count);
            int pick = rng.Next(pickRange);
            int next = candidates[pick].idx;

            route.Add(next);
            visited[next] = true;
            current = next;
        }

        return route.ToArray();
    }

    static int[] TwoOpt(int n, double[,] cost, int[] route)
    {
        var best = (int[])route.Clone();
        bool improved = true;

        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double delta = TwoOptDelta(n, cost, best, i, j);
                    if (delta < -1e-10)
                    {
                        Array.Reverse(best, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }

        return best;
    }

    static double TwoOptDelta(int n, double[,] cost, int[] route, int i, int j)
    {
        int a = (i == 0) ? 0 : route[i - 1] + 1;
        int b = route[i] + 1;
        int c = route[j] + 1;
        int d = (j == n - 1) ? 0 : route[j + 1] + 1;

        if (i == 0) a = 0;

        double oldCost, newCost;

        if (i == 0)
        {
            oldCost = cost[0, b] + cost[c, d == 0 ? 0 : d];
            newCost = cost[0, c] + cost[b, d == 0 ? 0 : d];
        }
        else
        {
            oldCost = cost[a, b] + cost[c, d == 0 ? 0 : d];
            newCost = cost[a, c] + cost[b, d == 0 ? 0 : d];
        }

        return newCost - oldCost;
    }

    static double RouteCost(int n, double[,] cost, int[] route)
    {
        double c = cost[0, route[0] + 1];
        for (int i = 0; i < n - 1; i++)
            c += cost[route[i] + 1, route[i + 1] + 1];
        return c;
    }

    static int[] SimulatedAnnealing(int n, double[,] cost, int[] initial)
    {
        var rng = new Random(42);
        var current = (int[])initial.Clone();
        var best = (int[])initial.Clone();
        double currentCost = RouteCost(n, cost, current);
        double bestCost = currentCost;

        double temp = bestCost * 0.1;
        double cooling = 0.9995;
        int iterations = Math.Min(n * n * 1000, 500000);

        for (int iter = 0; iter < iterations; iter++)
        {
            int i = rng.Next(n);
            int j = rng.Next(n);
            if (i > j) (i, j) = (j, i);
            if (i == j || j - i < 2) continue;

            double delta = TwoOptDelta(n, cost, current, i, j);

            if (delta < 0 || rng.NextDouble() < Math.Exp(-delta / temp))
            {
                Array.Reverse(current, i + 1, j - i);
                currentCost += delta;

                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    best = (int[])current.Clone();
                }
            }

            temp *= cooling;
        }

        return best;
    }

    struct Orbit
    {
        public double X, Y, Z, VX, VY, VZ;
    }

    struct OrbitalParams
    {
        public double A, E, Inc, RAAN, H;
    }
}
