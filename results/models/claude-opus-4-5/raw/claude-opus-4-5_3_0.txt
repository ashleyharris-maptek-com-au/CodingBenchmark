use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    if n == 0 {
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut parts = line.split_whitespace();
            let u: usize = parts.next().unwrap().parse().unwrap();
            let v: usize = parts.next().unwrap().parse().unwrap();
            if u != v && u < n && v < n {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let mut pos: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
    let area = (n as f64).sqrt() * 100.0;
    let k = (area / (n as f64).sqrt()).max(1.0);

    let mut visited = vec![false; n];
    let mut component_start_x = 0.0;

    for start in 0..n {
        if visited[start] {
            continue;
        }

        let mut queue = VecDeque::new();
        let mut component = Vec::new();
        queue.push_back(start);
        visited[start] = true;

        while let Some(node) = queue.pop_front() {
            component.push(node);
            for &neighbor in &adj[node] {
                if !visited[neighbor] {
                    visited[neighbor] = true;
                    queue.push_back(neighbor);
                }
            }
        }

        let comp_size = component.len();
        let cols = (comp_size as f64).sqrt().ceil() as usize;

        for (idx, &node) in component.iter().enumerate() {
            let row = idx / cols;
            let col = idx % cols;
            let jitter_x = ((node * 7) % 100) as f64 / 500.0;
            let jitter_y = ((node * 13) % 100) as f64 / 500.0;
            pos[node] = (
                component_start_x + col as f64 * k + jitter_x,
                row as f64 * k + jitter_y,
            );
        }

        component_start_x += (cols as f64 + 2.0) * k;
    }

    let iterations = if n <= 50 {
        300
    } else if n <= 500 {
        150
    } else if n <= 5000 {
        80
    } else if n <= 50000 {
        30
    } else {
        15
    };

    let mut temp = area / 10.0;
    let cooling = temp / (iterations as f64 + 1.0);

    let use_grid = n > 500;

    for _ in 0..iterations {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

        if use_grid {
            let cell_size = k * 2.0;
            let mut grid: std::collections::HashMap<(i64, i64), Vec<usize>> = std::collections::HashMap::new();

            for i in 0..n {
                let cx = (pos[i].0 / cell_size).floor() as i64;
                let cy = (pos[i].1 / cell_size).floor() as i64;
                grid.entry((cx, cy)).or_default().push(i);
            }

            for i in 0..n {
                let cx = (pos[i].0 / cell_size).floor() as i64;
                let cy = (pos[i].1 / cell_size).floor() as i64;

                for dx in -2i64..=2 {
                    for dy in -2i64..=2 {
                        if let Some(cell) = grid.get(&(cx + dx, cy + dy)) {
                            for &j in cell {
                                if i >= j {
                                    continue;
                                }
                                let (dx, dy) = (pos[i].0 - pos[j].0, pos[i].1 - pos[j].1);
                                let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                                let force = k * k / dist;
                                let fx = dx / dist * force;
                                let fy = dy / dist * force;
                                disp[i].0 += fx;
                                disp[i].1 += fy;
                                disp[j].0 -= fx;
                                disp[j].1 -= fy;
                            }
                        }
                    }
                }
            }
        } else {
            for i in 0..n {
                for j in (i + 1)..n {
                    let (dx, dy) = (pos[i].0 - pos[j].0, pos[i].1 - pos[j].1);
                    let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                    let force = k * k / dist;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }
        }

        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let (dx, dy) = (pos[u].0 - pos[v].0, pos[u].1 - pos[v].1);
                    let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                    let force = dist * dist / k;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[u].0 -= fx;
                    disp[u].1 -= fy;
                    disp[v].0 += fx;
                    disp[v].1 += fy;
                }
            }
        }

        for i in 0..n {
            let (dx, dy) = disp[i];
            let dist = (dx * dx + dy * dy).sqrt().max(0.01);
            let capped = dist.min(temp);
            pos[i].0 += dx / dist * capped;
            pos[i].1 += dy / dist * capped;
        }

        temp -= cooling;
        if temp <= 0.0 {
            break;
        }
    }

    for (x, y) in pos {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}
