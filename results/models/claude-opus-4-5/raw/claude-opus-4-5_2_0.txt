using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static int N, M;
    static List<(int to, int weight, int idx)>[] adj;
    static List<(int u, int v, int w)> edges;
    static int[] edgeCount;

    static void Main()
    {
        ReadInput();

        var oddVerts = new List<int>();
        for (int i = 0; i < N; i++)
            if (Degree(i) % 2 == 1)
                oddVerts.Add(i);

        if (oddVerts.Count > 0)
        {
            var dist = new long[oddVerts.Count][];
            var paths = new List<int>[oddVerts.Count][];

            for (int i = 0; i < oddVerts.Count; i++)
            {
                dist[i] = new long[oddVerts.Count];
                paths[i] = new List<int>[oddVerts.Count];
                var (d, p) = Dijkstra(oddVerts[i], oddVerts);
                for (int j = 0; j < oddVerts.Count; j++)
                {
                    dist[i][j] = d[j];
                    paths[i][j] = p[j];
                }
            }

            var matching = MinMatching(dist, oddVerts.Count);

            foreach (var (i, j) in matching)
            {
                var path = paths[i][j];
                for (int k = 0; k < path.Count - 1; k++)
                {
                    int u = path[k], v = path[k + 1];
                    int minW = int.MaxValue;
                    foreach (var e in adj[u])
                        if (e.to == v && edges[e.idx].w < minW)
                            minW = edges[e.idx].w;
                    int idx = edges.Count;
                    edges.Add((u, v, minW));
                    adj[u].Add((v, minW, idx));
                    adj[v].Add((u, minW, idx));
                    edgeCount[idx] = 1;
                }
            }
        }

        var circuit = Hierholzer(0);
        Console.WriteLine(string.Join(" ", circuit));
    }

    static void ReadInput()
    {
        var line = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        N = int.Parse(line[0]);
        M = int.Parse(line[1]);

        adj = new List<(int, int, int)>[N];
        edges = new List<(int, int, int)>(M);
        edgeCount = new int[M + N * 2];

        for (int i = 0; i < N; i++)
            adj[i] = new List<(int, int, int)>();

        for (int i = 0; i < M; i++)
        {
            line = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(line[0]);
            int v = int.Parse(line[1]);
            int w = int.Parse(line[2]);
            edges.Add((u, v, w));
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
            edgeCount[i] = 1;
        }
    }

    static int Degree(int v)
    {
        int d = 0;
        foreach (var e in adj[v])
            d += edgeCount[e.idx];
        return d;
    }

    static (long[], List<int>[]) Dijkstra(int src, List<int> targets)
    {
        var targetSet = new HashSet<int>(targets);
        var dist = new long[N];
        var prev = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(prev, -1);
        dist[src] = 0;

        var pq = new SortedSet<(long d, int v)> { (0, src) };

        while (pq.Count > 0)
        {
            var (d, u) = pq.Min;
            pq.Remove(pq.Min);
            if (d > dist[u]) continue;

            foreach (var (v, w, _) in adj[u])
            {
                if (dist[u] + w < dist[v])
                {
                    pq.Remove((dist[v], v));
                    dist[v] = dist[u] + w;
                    prev[v] = u;
                    pq.Add((dist[v], v));
                }
            }
        }

        var result = new long[targets.Count];
        var paths = new List<int>[targets.Count];

        for (int i = 0; i < targets.Count; i++)
        {
            result[i] = dist[targets[i]];
            paths[i] = new List<int>();
            for (int v = targets[i]; v != -1; v = prev[v])
                paths[i].Add(v);
            paths[i].Reverse();
        }

        return (result, paths);
    }

    static List<(int, int)> MinMatching(long[][] dist, int n)
    {
        if (n <= 20)
            return DPMatching(dist, n);
        return GreedyMatching(dist, n);
    }

    static List<(int, int)> DPMatching(long[][] dist, int n)
    {
        int full = (1 << n) - 1;
        var dp = new long[1 << n];
        var parent = new int[1 << n];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << n); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int first = -1;
            for (int i = 0; i < n; i++)
                if ((mask & (1 << i)) == 0) { first = i; break; }
            if (first == -1) continue;

            for (int second = first + 1; second < n; second++)
            {
                if ((mask & (1 << second)) != 0) continue;
                int newMask = mask | (1 << first) | (1 << second);
                long newCost = dp[mask] + dist[first][second];
                if (newCost < dp[newMask])
                {
                    dp[newMask] = newCost;
                    parent[newMask] = mask | (first << 10) | (second << 20);
                }
            }
        }

        var result = new List<(int, int)>();
        int cur = full;
        while (cur != 0)
        {
            int p = parent[cur];
            int first = (p >> 10) & 0x3FF;
            int second = (p >> 20) & 0x3FF;
            result.Add((first, second));
            cur = p & 0x3FF;
        }
        return result;
    }

    static List<(int, int)> GreedyMatching(long[][] dist, int n)
    {
        var used = new bool[n];
        var result = new List<(int, int)>();
        var candidates = new List<(long d, int i, int j)>();

        for (int i = 0; i < n; i++)
            for (int j = i + 1; j < n; j++)
                candidates.Add((dist[i][j], i, j));

        candidates.Sort();

        foreach (var (_, i, j) in candidates)
        {
            if (used[i] || used[j]) continue;
            used[i] = used[j] = true;
            result.Add((i, j));
        }
        return result;
    }

    static List<int> Hierholzer(int start)
    {
        var localCount = new int[edgeCount.Length];
        Array.Copy(edgeCount, localCount, edgeCount.Length);

        var stack = new Stack<int>();
        var circuit = new List<int>();
        stack.Push(start);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;

            foreach (var (v, _, idx) in adj[u])
            {
                if (localCount[idx] > 0)
                {
                    localCount[idx]--;
                    stack.Push(v);
                    found = true;
                    break;
                }
            }

            if (!found)
                circuit.Add(stack.Pop());
        }

        circuit.Reverse();
        return circuit;
    }
}
