#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

using namespace std;

struct Rect { int w, h, idx; };
struct Placement { int board, x, y, rotated; };
struct FreeRect { int x, y, w, h; };

constexpr size_t MAX_FREE = 100;

class Bin {
public:
    vector<FreeRect> freeList;

    Bin(int w, int h) { freeList.push_back({0, 0, w, h}); }

    int findBest(int w, int h, int& px, int& py, bool& rot) const {
        int best = INT32_MAX;
        for (int r = 0; r <= (w != h ? 1 : 0); r++) {
            int rw = r ? h : w, rh = r ? w : h;
            for (const auto& f : freeList) {
                if (rw <= f.w && rh <= f.h) {
                    int s = min(f.w - rw, f.h - rh);
                    if (s < best) {
                        best = s;
                        px = f.x; py = f.y;
                        rot = (r == 1);
                    }
                }
            }
        }
        return best;
    }

    void place(int w, int h, int x, int y) {
        vector<FreeRect> newList;
        for (const auto& f : freeList) {
            if (x >= f.x + f.w || x + w <= f.x || y >= f.y + f.h || y + h <= f.y) {
                newList.push_back(f);
            } else {
                if (x > f.x) newList.push_back({f.x, f.y, x - f.x, f.h});
                if (x + w < f.x + f.w) newList.push_back({x + w, f.y, f.x + f.w - x - w, f.h});
                if (y > f.y) newList.push_back({f.x, f.y, f.w, y - f.y});
                if (y + h < f.y + f.h) newList.push_back({f.x, y + h, f.w, f.y + f.h - y - h});
            }
        }
        freeList = move(newList);

        if (freeList.size() > MAX_FREE) {
            nth_element(freeList.begin(), freeList.begin() + MAX_FREE, freeList.end(),
                       [](const FreeRect& a, const FreeRect& b) {
                           return (int64_t)a.w * a.h > (int64_t)b.w * b.h;
                       });
            freeList.resize(MAX_FREE);
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, W, H;
    cin >> n >> W >> H;

    if (n == 0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        int64_t aa = (int64_t)a.w * a.h, ab = (int64_t)b.w * b.h;
        return aa != ab ? aa > ab : max(a.w, a.h) > max(b.w, b.h);
    });

    vector<Bin> bins;
    vector<Placement> result(n);

    for (const auto& r : rects) {
        int bestBin = -1, bestScore = INT32_MAX;
        int bx = 0, by = 0; bool brot = false;

        int threshold = min(r.w, r.h) / 4;

        for (size_t b = 0; b < bins.size(); b++) {
            int x, y; bool rot;
            int s = bins[b].findBest(r.w, r.h, x, y, rot);
            if (s < bestScore) {
                bestScore = s;
                bestBin = static_cast<int>(b);
                bx = x; by = y; brot = rot;
                if (s <= threshold) break;
            }
        }

        if (bestBin < 0) {
            bins.emplace_back(W, H);
            bestBin = static_cast<int>(bins.size()) - 1;
            bins[bestBin].findBest(r.w, r.h, bx, by, brot);
        }

        int pw = brot ? r.h : r.w, ph = brot ? r.w : r.h;
        bins[bestBin].place(pw, ph, bx, by);

        result[r.idx] = {bestBin, bx, by, brot ? 1 : 0};
    }

    cout << bins.size() << '\n';
    for (const auto& p : result)
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';

    return 0;
}
