#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <set>
#include <map>
#include <tuple>
#include <bit>

using namespace std;

typedef long long ll;
typedef pair<ll, int> pli;

const ll INF = 1e18;
const int MAXN = 5005;
const int MAXK = 11;

int n, m, k;
vector<pair<int, ll>> adj[MAXN];
int terminals[MAXN];
bool isTerminal[MAXN];
map<pair<int,int>, ll> edgeWeight;

ll dp[MAXN][1 << MAXK];
pair<int, int> from_dp[MAXN][1 << MAXK];

void dijkstra(int mask) {
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    for (int v = 0; v < n; v++) {
        if (dp[v][mask] < INF) {
            pq.push({dp[v][mask], v});
        }
    }
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d > dp[u][mask]) continue;
        for (auto& [v, w] : adj[u]) {
            if (dp[u][mask] + w < dp[v][mask]) {
                dp[v][mask] = dp[u][mask] + w;
                from_dp[v][mask] = {u, 0};
                pq.push({dp[v][mask], v});
            }
        }
    }
}

set<pair<int,int>> resultEdges;

void reconstruct(int v, int mask) {
    if (mask == 0) return;
    if (std::popcount(static_cast<unsigned>(mask)) == 1) return;
    auto [type_val, sub] = from_dp[v][mask];
    if (type_val == -1) {
        reconstruct(v, sub);
        reconstruct(v, mask ^ sub);
    } else if (type_val >= 0) {
        resultEdges.insert({min(type_val, v), max(type_val, v)});
        reconstruct(type_val, mask);
    }
}

ll dist_arr[MAXN];
int pred_arr[MAXN];

void dijkstraSingle(int src) {
    fill(dist_arr, dist_arr + n, INF);
    fill(pred_arr, pred_arr + n, -1);
    dist_arr[src] = 0;
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d > dist_arr[u]) continue;
        for (auto& [v, w] : adj[u]) {
            if (dist_arr[u] + w < dist_arr[v]) {
                dist_arr[v] = dist_arr[u] + w;
                pred_arr[v] = u;
                pq.push({dist_arr[v], v});
            }
        }
    }
}

int uf[MAXN];
int find(int x) { return uf[x] == x ? x : uf[x] = find(uf[x]); }
void unite(int x, int y) { uf[find(x)] = find(y); }

void solveHeuristic() {
    vector<vector<ll>> termDist(k, vector<ll>(k));
    vector<vector<int>> termPred(k, vector<int>(n));
    for (int i = 0; i < k; i++) {
        dijkstraSingle(terminals[i]);
        for (int j = 0; j < k; j++) termDist[i][j] = dist_arr[terminals[j]];
        for (int v = 0; v < n; v++) termPred[i][v] = pred_arr[v];
    }
    vector<tuple<ll, int, int>> mstEdges;
    for (int i = 0; i < k; i++)
        for (int j = i + 1; j < k; j++)
            mstEdges.push_back({termDist[i][j], i, j});
    sort(mstEdges.begin(), mstEdges.end());
    for (int i = 0; i < k; i++) uf[i] = i;
    set<pair<int,int>> usedEdges;
    for (auto& [w, i, j] : mstEdges) {
        if (find(i) != find(j)) {
            unite(i, j);
            int cur = terminals[j];
            while (cur != terminals[i]) {
                int p = termPred[i][cur];
                if (p == -1) break;
                usedEdges.insert({min(cur, p), max(cur, p)});
                cur = p;
            }
        }
    }
    vector<tuple<ll, int, int>> subEdges;
    for (auto& [u, v] : usedEdges) subEdges.push_back({edgeWeight[{u, v}], u, v});
    sort(subEdges.begin(), subEdges.end());
    for (int i = 0; i < n; i++) uf[i] = i;
    set<pair<int,int>> finalEdges;
    ll totalWeight = 0;
    for (auto& [w, u, v] : subEdges) {
        if (find(u) != find(v)) {
            unite(u, v);
            finalEdges.insert({u, v});
            totalWeight += w;
        }
    }
    vector<set<int>> subAdj(n);
    for (auto& [u, v] : finalEdges) { subAdj[u].insert(v); subAdj[v].insert(u); }
    bool changed = true;
    while (changed) {
        changed = false;
        for (int v = 0; v < n; v++) {
            if (!isTerminal[v] && subAdj[v].size() == 1) {
                int u = *subAdj[v].begin();
                ll w = edgeWeight[{min(u, v), max(u, v)}];
                subAdj[u].erase(v);
                subAdj[v].clear();
                totalWeight -= w;
                finalEdges.erase({min(u, v), max(u, v)});
                changed = true;
            }
        }
    }
    cout << totalWeight << " " << finalEdges.size() << "\n";
    for (auto& [u, v] : finalEdges) cout << u << " " << v << "\n";
}

void solveDP() {
    int fullMask = (1 << k) - 1;
    for (int v = 0; v < n; v++)
        for (int mask = 0; mask <= fullMask; mask++) {
            dp[v][mask] = INF;
            from_dp[v][mask] = {-2, 0};
        }
    for (int i = 0; i < k; i++) dp[terminals[i]][1 << i] = 0;
    for (int mask = 1; mask <= fullMask; mask++) {
        for (int v = 0; v < n; v++) {
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (other > sub) continue;
                if (dp[v][sub] < INF && dp[v][other] < INF) {
                    ll nc = dp[v][sub] + dp[v][other];
                    if (nc < dp[v][mask]) {
                        dp[v][mask] = nc;
                        from_dp[v][mask] = {-1, sub};
                    }
                }
            }
        }
        dijkstra(mask);
    }
    ll minCost = INF;
    int bestV = -1;
    for (int v = 0; v < n; v++)
        if (dp[v][fullMask] < minCost) { minCost = dp[v][fullMask]; bestV = v; }
    resultEdges.clear();
    reconstruct(bestV, fullMask);
    cout << minCost << " " << resultEdges.size() << "\n";
    for (auto& [u, v] : resultEdges) cout << u << " " << v << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cin >> n >> m >> k;
    for (int i = 0; i < m; i++) {
        int u, v; ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
        auto key = make_pair(min(u, v), max(u, v));
        if (edgeWeight.find(key) == edgeWeight.end() || edgeWeight[key] > w)
            edgeWeight[key] = w;
    }
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
        isTerminal[terminals[i]] = true;
    }
    if (k == 1) { cout << 0 << " " << 0 << "\n"; return 0; }
    if (k <= 10) solveDP();
    else solveHeuristic();
    return 0;
}
