using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int n;
    static List<int>[] adj, radj;

    static void Main()
    {
        var parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        adj = new List<int>[n];
        radj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        var selfLoops = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);

            if (u == v)
                selfLoops.Add(u);
            else
            {
                adj[u].Add(v);
                radj[v].Add(u);
            }
        }

        var result = SolveDFVS(selfLoops);

        Console.WriteLine(result.Count);
        if (result.Count > 0)
            Console.WriteLine(string.Join(" ", result));
        else
            Console.WriteLine();
    }

    static List<int> SolveDFVS(HashSet<int> mustRemove)
    {
        var result = new List<int>(mustRemove);
        var removed = new HashSet<int>(mustRemove);

        while (true)
        {
            Reduce(removed);
            var scc = FindCyclicSCC(removed);
            if (scc == null || scc.Count == 0)
                break;

            var fvs = MinFVS(scc);
            foreach (int v in fvs)
            {
                result.Add(v);
                removed.Add(v);
            }
        }

        return result;
    }

    static void Reduce(HashSet<int> removed)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int v = 0; v < n; v++)
            {
                if (removed.Contains(v)) continue;

                bool hasIn = false, hasOut = false;
                foreach (int u in radj[v])
                    if (!removed.Contains(u)) { hasIn = true; break; }
                foreach (int w in adj[v])
                    if (!removed.Contains(w)) { hasOut = true; break; }

                if (!hasIn || !hasOut)
                {
                    removed.Add(v);
                    changed = true;
                }
            }
        }
    }

    static HashSet<int> FindCyclicSCC(HashSet<int> removed)
    {
        var active = new HashSet<int>();
        for (int i = 0; i < n; i++)
            if (!removed.Contains(i))
                active.Add(i);

        if (active.Count == 0) return null;

        var visited = new HashSet<int>();
        var order = new List<int>();

        foreach (int v in active)
            if (!visited.Contains(v))
                DFS1(v, active, visited, order);

        var assigned = new HashSet<int>();

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!assigned.Contains(v))
            {
                var scc = new HashSet<int>();
                DFS2(v, active, assigned, scc);

                if (scc.Count > 1)
                    return scc;
            }
        }

        return null;
    }

    static void DFS1(int v, HashSet<int> active, HashSet<int> visited, List<int> order)
    {
        var stack = new Stack<(int, bool)>();
        stack.Push((v, false));

        while (stack.Count > 0)
        {
            var (u, done) = stack.Pop();

            if (done)
            {
                order.Add(u);
                continue;
            }

            if (visited.Contains(u)) continue;
            visited.Add(u);
            stack.Push((u, true));

            foreach (int w in adj[u])
                if (active.Contains(w) && !visited.Contains(w))
                    stack.Push((w, false));
        }
    }

    static void DFS2(int v, HashSet<int> active, HashSet<int> assigned, HashSet<int> scc)
    {
        var stack = new Stack<int>();
        stack.Push(v);

        while (stack.Count > 0)
        {
            int u = stack.Pop();
            if (assigned.Contains(u)) continue;
            assigned.Add(u);
            scc.Add(u);

            foreach (int w in radj[u])
                if (active.Contains(w) && !assigned.Contains(w))
                    stack.Push(w);
        }
    }

    static List<int> MinFVS(HashSet<int> scc)
    {
        var vertices = scc.ToList();
        int k = vertices.Count;

        var localAdj = new List<int>[k];
        var idx = new Dictionary<int, int>();

        for (int i = 0; i < k; i++)
        {
            idx[vertices[i]] = i;
            localAdj[i] = new List<int>();
        }

        foreach (int u in scc)
            foreach (int v in adj[u])
                if (scc.Contains(v))
                    localAdj[idx[u]].Add(idx[v]);

        for (int maxSize = 1; maxSize < k; maxSize++)
        {
            var result = new List<int>();
            var removed = new bool[k];

            if (TryFindFVS(localAdj, k, maxSize, removed, result))
                return result.Select(i => vertices[i]).ToList();
        }

        return vertices.Take(k - 1).ToList();
    }

    static bool TryFindFVS(List<int>[] adj, int n, int maxSize, bool[] removed, List<int> current)
    {
        var cycle = FindCycle(adj, n, removed);
        if (cycle == null)
            return true;

        if (current.Count >= maxSize)
            return false;

        var ordered = cycle.OrderByDescending(v =>
            adj[v].Count(w => !removed[w])
        ).ToList();

        foreach (int v in ordered)
        {
            removed[v] = true;
            current.Add(v);

            if (TryFindFVS(adj, n, maxSize, removed, current))
                return true;

            current.RemoveAt(current.Count - 1);
            removed[v] = false;
        }

        return false;
    }

    static List<int> FindCycle(List<int>[] adj, int n, bool[] removed)
    {
        var color = new int[n];
        var parent = new int[n];
        Array.Fill(parent, -1);

        for (int start = 0; start < n; start++)
        {
            if (removed[start] || color[start] != 0) continue;

            var stack = new Stack<(int v, int idx)>();
            stack.Push((start, 0));
            color[start] = 1;

            while (stack.Count > 0)
            {
                var (v, idx) = stack.Pop();

                bool foundNext = false;
                for (int i = idx; i < adj[v].Count; i++)
                {
                    int w = adj[v][i];
                    if (removed[w]) continue;

                    if (color[w] == 1)
                    {
                        var cycle = new List<int> { w };
                        int curr = v;
                        while (curr != w && curr != -1)
                        {
                            cycle.Add(curr);
                            curr = parent[curr];
                        }
                        return cycle;
                    }

                    if (color[w] == 0)
                    {
                        stack.Push((v, i + 1));
                        parent[w] = v;
                        color[w] = 1;
                        stack.Push((w, 0));
                        foundNext = true;
                        break;
                    }
                }

                if (!foundNext)
                    color[v] = 2;
            }
        }

        return null;
    }
}
