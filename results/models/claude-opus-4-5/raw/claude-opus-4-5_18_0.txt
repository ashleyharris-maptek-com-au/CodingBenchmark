#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <climits>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m;
    cin >> n >> m;

    if (n <= 1 || m == 0) {
        cout << 0 << "\n";
        return 0;
    }

    vector<pair<int,int>> edges;
    edges.reserve(m);
    vector<int> degree(n, 0);

    for (long long i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        edges.emplace_back(u, v);
        degree[u]++;
        degree[v]++;
    }

    int minDegVertex = 0;
    for (int i = 1; i < n; i++) {
        if (degree[i] < degree[minDegVertex]) {
            minDegVertex = i;
        }
    }

    if (degree[minDegVertex] == 1) {
        for (auto& [u, v] : edges) {
            if (u == minDegVertex || v == minDegVertex) {
                cout << 1 << "\n" << u << " " << v << "\n";
                return 0;
            }
        }
    }

    vector<unordered_map<int, int>> adj(n);
    for (auto& [u, v] : edges) {
        adj[u][v]++;
        adj[v][u]++;
    }

    vector<bool> alive(n, true);
    vector<vector<int>> group(n);
    for (int i = 0; i < n; i++) group[i] = {i};

    int minCut = INT_MAX;
    vector<int> bestGroup;

    for (int phase = 0; phase < n - 1; phase++) {
        int start = 0;
        while (!alive[start]) start++;

        vector<int> key(n, 0);
        vector<bool> inSet(n, false);
        priority_queue<pair<int,int>> pq;
        pq.emplace(0, start);

        int prev = -1, last = -1;

        while (!pq.empty()) {
            auto [k, v] = pq.top();
            pq.pop();

            if (inSet[v] || !alive[v] || k < key[v]) continue;

            prev = last;
            last = v;
            inSet[v] = true;

            for (auto& [neighbor, weight] : adj[v]) {
                if (alive[neighbor] && !inSet[neighbor]) {
                    key[neighbor] += weight;
                    pq.emplace(key[neighbor], neighbor);
                }
            }
        }

        if (key[last] < minCut) {
            minCut = key[last];
            bestGroup = group[last];
        }

        if (minCut <= 1) break;

        alive[last] = false;
        for (int v : group[last]) group[prev].push_back(v);

        for (auto& [neighbor, weight] : adj[last]) {
            if (neighbor != prev && alive[neighbor]) {
                adj[prev][neighbor] += weight;
                adj[neighbor][prev] += weight;
            }
        }
        adj[prev].erase(last);
        adj[last].clear();
    }

    unordered_set<int> cutSet(bestGroup.begin(), bestGroup.end());
    vector<pair<int,int>> cutEdges;

    for (auto& [u, v] : edges) {
        if (cutSet.count(u) != cutSet.count(v)) {
            cutEdges.emplace_back(u, v);
        }
    }

    cout << cutEdges.size() << "\n";
    for (auto& [u, v] : cutEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
