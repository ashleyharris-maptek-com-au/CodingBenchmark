using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var sr = new StreamReader(Console.OpenStandardInput(), Encoding.UTF8, false, 1 << 20);
        string[] tokens = sr.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);

        int p = 0;
        int n = int.Parse(tokens[p++]);
        int L = int.Parse(tokens[p++]);

        if (n == 0)
        {
            Console.WriteLine("0");
            return;
        }

        int[] cuts = new int[n];
        int[] orig = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(tokens[p++]);
            orig[i] = i;
        }

        Array.Sort(cuts, orig);
        Array.Reverse(cuts);
        Array.Reverse(orig);

        var binRem = new List<int>();
        var assign = new int[n];
        var bins = new SortedSet<(int rem, int idx)>();

        for (int i = 0; i < n; i++)
        {
            int c = cuts[i];
            int o = orig[i];

            var view = bins.GetViewBetween((c, int.MinValue), (L + 1, int.MaxValue));
            var it = view.GetEnumerator();

            if (it.MoveNext())
            {
                var best = it.Current;
                it.Dispose();
                bins.Remove(best);
                int newRem = best.rem - c;
                binRem[best.idx] = newRem;
                assign[o] = best.idx;
                if (newRem > 0) bins.Add((newRem, best.idx));
            }
            else
            {
                it.Dispose();
                int b = binRem.Count;
                binRem.Add(L - c);
                assign[o] = b;
                if (L - c > 0) bins.Add((L - c, b));
            }
        }

        int numBins = binRem.Count;
        var contents = new List<int>[numBins];
        for (int i = 0; i < numBins; i++) contents[i] = new List<int>();
        for (int i = 0; i < n; i++) contents[assign[i]].Add(i);

        var sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.UTF8, 1 << 20);
        sw.WriteLine(numBins);
        for (int i = 0; i < numBins; i++)
        {
            var list = contents[i];
            if (list.Count > 0)
            {
                sw.Write(list[0]);
                for (int j = 1; j < list.Count; j++)
                {
                    sw.Write(' ');
                    sw.Write(list[j]);
                }
            }
            sw.WriteLine();
        }
        sw.Flush();
    }
}
