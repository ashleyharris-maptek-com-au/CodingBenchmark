use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut input = String::new();
    stdin.lock().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    if m == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut in_cover = vec![false; n];
    let mut degree: Vec<usize> = adj.iter().map(|x| x.len()).collect();

    // Greedy by degree using max-heap with lazy updates
    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new();
    for v in 0..n {
        if degree[v] > 0 {
            heap.push((degree[v], v));
        }
    }

    while let Some((d, v)) = heap.pop() {
        if in_cover[v] || degree[v] != d || d == 0 {
            continue;
        }

        in_cover[v] = true;

        for &u in &adj[v] {
            if !in_cover[u] && degree[u] > 0 {
                degree[u] -= 1;
                if degree[u] > 0 {
                    heap.push((degree[u], u));
                }
            }
        }
    }

    // Post-processing: remove redundant vertices
    for v in 0..n {
        if in_cover[v] {
            let can_remove = adj[v].iter().all(|&u| in_cover[u]);
            if can_remove {
                in_cover[v] = false;
            }
        }
    }

    let cover: Vec<usize> = (0..n).filter(|&i| in_cover[i]).collect();

    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        write!(out, "{}", cover[0]).unwrap();
        for &v in &cover[1..] {
            write!(out, " {}", v).unwrap();
        }
        writeln!(out).unwrap();
    }
}
