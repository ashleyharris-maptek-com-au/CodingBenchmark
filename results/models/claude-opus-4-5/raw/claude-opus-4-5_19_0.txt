using System;
using System.Collections.Generic;
using System.Diagnostics;

class Mining
{
    static int XS, YS, ZS;
    static byte[,,] vox;
    static int[,] ht;
    static Stopwatch sw;

    public static void Main()
    {
        sw = Stopwatch.StartNew();

        var hdr = Console.ReadLine().Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        XS = int.Parse(hdr[0]);
        YS = int.Parse(hdr[1]);
        ZS = int.Parse(hdr[2]);

        vox = new byte[XS, YS, ZS];
        ht = new int[XS, YS];

        using (var s = Console.OpenStandardInput())
        {
            for (int x = 0; x < XS; x++)
            {
                for (int y = 0; y < YS; y++)
                {
                    ht[x, y] = 0;
                    for (int z = 0; z < ZS; z++)
                    {
                        int b = s.ReadByte();
                        if (b < 0) b = 0;
                        vox[x, y, z] = (byte)b;
                        if (b > 0) ht[x, y] = z + 1;
                    }
                }
            }
        }

        long totalVoxels = (long)XS * YS * ZS;

        List<string> ops;
        if (totalVoxels <= 1000000)
            ops = SolveSmall();
        else
            ops = SolveLarge();

        Console.WriteLine(ops.Count);
        foreach (var op in ops)
            Console.WriteLine(op);
    }

    static bool TimedOut => sw.Elapsed.TotalMinutes > 4;

    static List<string> SolveSmall()
    {
        var ops = new List<string>();

        bool changed = true;
        while (changed && !TimedOut)
        {
            changed = false;
            int bestX = -1, bestY = -1, bestZ = -1;
            double bestProfit = 0;
            List<int> bestPath = null;

            for (int x = 0; x < XS && !TimedOut; x++)
            {
                for (int y = 0; y < YS; y++)
                {
                    int z = ht[x, y] - 1;
                    if (z < 0) continue;

                    int mass = vox[x, y, z] & 15;
                    int value = vox[x, y, z] >> 4;
                    if (value == 0 || mass == 0) continue;

                    var path = FindPath(x, y, z);
                    if (path == null) continue;

                    double rev = 1000.0 * mass * value;
                    double cost = CalcCost(path, z, mass);
                    double profit = rev - cost;

                    if (profit > bestProfit)
                    {
                        bestProfit = profit;
                        bestX = x; bestY = y; bestZ = z;
                        bestPath = path;
                    }
                }
            }

            if (bestX >= 0)
            {
                changed = true;
                vox[bestX, bestY, bestZ] = 0;
                UpdateHeight(bestX, bestY);
                ops.Add(FormatOp(bestX, bestY, bestZ, bestPath));
            }
        }

        return ops;
    }

    static List<string> SolveLarge()
    {
        var ops = new List<string>();
        var candidates = new List<(int x, int y, double estProfit)>();

        for (int x = 0; x < XS; x++)
        {
            for (int y = 0; y < YS; y++)
            {
                int z = ht[x, y] - 1;
                if (z < 0) continue;

                int mass = vox[x, y, z] & 15;
                int value = vox[x, y, z] >> 4;
                if (value == 0 || mass == 0) continue;

                double rev = 1000.0 * mass * value;
                double cost = mass * (x + y) * 2.0;
                double profit = rev - cost;

                if (profit > 0)
                    candidates.Add((x, y, profit));
            }
        }

        candidates.Sort((a, b) => b.estProfit.CompareTo(a.estProfit));

        foreach (var (x, y, _) in candidates)
        {
            if (TimedOut) break;

            int z = ht[x, y] - 1;
            if (z < 0 || vox[x, y, z] == 0) continue;

            int mass = vox[x, y, z] & 15;
            int value = vox[x, y, z] >> 4;
            if (value == 0 || mass == 0) continue;

            var path = FindPath(x, y, z);
            if (path == null) continue;

            double rev = 1000.0 * mass * value;
            double cost = CalcCost(path, z, mass);
            if (rev - cost <= 0) continue;

            vox[x, y, z] = 0;
            UpdateHeight(x, y);
            ops.Add(FormatOp(x, y, z, path));
        }

        return ops;
    }

    static void UpdateHeight(int x, int y)
    {
        while (ht[x, y] > 0 && vox[x, y, ht[x, y] - 1] == 0)
            ht[x, y]--;
    }

    static List<int> FindPath(int sx, int sy, int sz)
    {
        var path = new List<int>();
        int cx = sx, cy = sy;
        int prevZ = sz;

        int[] dx = { -1, 1, 0, 0 };
        int[] dy = { 0, 0, -1, 1 };

        while (cx != 0 || cy != 0)
        {
            int bestNx = -1, bestNy = -1;
            int bestDist = int.MaxValue;

            for (int d = 0; d < 4; d++)
            {
                int nx = cx + dx[d];
                int ny = cy + dy[d];

                if (nx < 0 || ny < 0 || nx >= XS || ny >= YS) continue;
                int nz = ht[nx, ny];
                if (nz > prevZ + 1) continue;

                int dist = nx + ny;
                if (dist < bestDist)
                {
                    bestDist = dist;
                    bestNx = nx;
                    bestNy = ny;
                }
            }

            if (bestNx < 0) return null;

            path.Add(bestNx);
            path.Add(bestNy);
            prevZ = ht[bestNx, bestNy];
            cx = bestNx;
            cy = bestNy;
        }

        return path;
    }

    static double CalcCost(List<int> path, int startZ, int mass)
    {
        double cost = 0;
        int prevZ = startZ;

        for (int i = 0; i < path.Count; i += 2)
        {
            int nz = ht[path[i], path[i + 1]];
            cost += (nz > prevZ) ? 5.0 * mass : 1.0 * mass;
            prevZ = nz;
        }

        return cost;
    }

    static string FormatOp(int digX, int digY, int digZ, List<int> path)
    {
        var sb = new System.Text.StringBuilder();

        int n = (path.Count / 2) - 1;
        if (n < 0) n = 0;

        sb.Append(n);
        sb.Append(' ');
        sb.Append(digX);
        sb.Append(' ');
        sb.Append(digY);
        sb.Append(' ');
        sb.Append(digZ);

        for (int i = 0; i < path.Count - 2; i += 2)
        {
            sb.Append(' ');
            sb.Append(path[i]);
            sb.Append(' ');
            sb.Append(path[i + 1]);
        }

        sb.Append(" 0 0");

        return sb.ToString();
    }
}
