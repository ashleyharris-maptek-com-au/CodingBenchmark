using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class AsteroidInterceptor
{
    const double G = 6.67430e-11;
    const double AU = 1.495978707e11;
    const double DAY = 86400.0;
    const double EARTH_RADIUS = 6.371e6;
    const double SUN_MU = 1.32712440018e20;

    static void Main(string[] args)
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        var input = Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        int numBodies = int.Parse(input[idx++]);
        double warningDays = double.Parse(input[idx++]);
        double impactorMass = double.Parse(input[idx++]);
        double deltaVBudget = double.Parse(input[idx++]);
        double asteroidMass = double.Parse(input[idx++]);
        double t0Unix = double.Parse(input[idx++]);

        var bodies = new List<Body>();
        Body asteroid = null, earth = null, sun = null;

        for (int i = 0; i < numBodies; i++)
        {
            string name = input[idx++];
            double mass = double.Parse(input[idx++]);
            double x = double.Parse(input[idx++]);
            double y = double.Parse(input[idx++]);
            double z = double.Parse(input[idx++]);
            double vx = double.Parse(input[idx++]);
            double vy = double.Parse(input[idx++]);
            double vz = double.Parse(input[idx++]);

            var body = new Body(name, mass, new Vec3(x, y, z), new Vec3(vx, vy, vz));
            bodies.Add(body);

            if (name == "Asteroid") asteroid = body;
            else if (name == "Earth") earth = body;
            else if (name == "Sun") sun = body;
        }

        string bspPath = input[idx++];

        double mu = sun != null ? G * sun.Mass : SUN_MU;

        var solution = FindInterceptTrajectory(bodies, asteroid, earth, sun, warningDays,
                                                deltaVBudget, asteroidMass, impactorMass, mu);

        Console.WriteLine(solution.Burns.Count);
        foreach (var burn in solution.Burns)
        {
            Console.WriteLine($"{burn.Time:G17} {burn.DV.X:G17} {burn.DV.Y:G17} {burn.DV.Z:G17}");
        }
    }

    static Solution FindInterceptTrajectory(List<Body> bodies, Body asteroid, Body earth,
                                            Body sun, double warningDays, double dvBudget,
                                            double asteroidMass, double impactorMass, double mu)
    {
        double maxT = warningDays * DAY;
        double earthImpactTime = EstimateImpactTime(asteroid, earth, bodies, maxT);
        double interceptDeadline = earthImpactTime * 0.95;

        Solution bestSolution = null;
        double bestScore = double.MaxValue;

        int launchSamples = Math.Min(60, (int)(warningDays / 4) + 15);
        int tofSamples = 40;

        for (int li = 0; li < launchSamples; li++)
        {
            double launchFrac = li / (double)(launchSamples - 1);
            double launchTime = launchFrac * Math.Min(warningDays * 0.6, interceptDeadline * 0.4) * DAY;

            var earthAtLaunch = PropagateBody(earth, bodies, launchTime);

            for (int ti = 0; ti < tofSamples; ti++)
            {
                double minTof = 20 * DAY;
                double maxTof = interceptDeadline - launchTime - 5 * DAY;
                if (maxTof < minTof) continue;

                double tofFrac = ti / (double)(tofSamples - 1);
                double tof = minTof + tofFrac * (maxTof - minTof);
                double arrivalTime = launchTime + tof;

                var asteroidAtArrival = PropagateBody(asteroid, bodies, arrivalTime);

                foreach (bool longWay in new[] { false, true })
                {
                    var lambert = SolveLambert(earthAtLaunch.Pos, asteroidAtArrival.Pos, tof, mu, longWay);
                    if (lambert == null) continue;

                    Vec3 launchDv = lambert.V1 - earthAtLaunch.Vel;
                    double launchDvMag = launchDv.Mag();

                    if (launchDvMag > dvBudget * 0.98) continue;

                    Vec3 relVel = lambert.V2 - asteroidAtArrival.Vel;
                    double impactSpeed = relVel.Mag();

                    double deflectionTime = earthImpactTime - arrivalTime;
                    if (deflectionTime < DAY) continue;

                    double momentumTransfer = impactorMass * impactSpeed * 1.5;
                    double dvAsteroid = momentumTransfer / asteroidMass;
                    double deflectionDist = dvAsteroid * deflectionTime;
                    double minSafeDist = EARTH_RADIUS * 5;

                    double deflectionScore = Math.Min(1.0, deflectionDist / minSafeDist);
                    double timeScore = (earthImpactTime - arrivalTime) / earthImpactTime;
                    double dvScore = launchDvMag / dvBudget;

                    double score = dvScore * 0.3 + (1 - deflectionScore) * 0.5 + (1 - timeScore) * 0.2;

                    if (score < bestScore && deflectionScore > 0.5)
                    {
                        bestScore = score;
                        bestSolution = new Solution();
                        bestSolution.Burns.Add(new Burn(launchTime / DAY, launchDv));

                        Vec3 impactDir = relVel.Normalized() * (-1);
                        double impactDvMag = Math.Min(impactSpeed, dvBudget - launchDvMag);
                        bestSolution.Burns.Add(new Burn(arrivalTime / DAY, impactDir * impactDvMag));
                    }
                }
            }
        }

        if (bestSolution == null)
        {
            bestSolution = CreateFallbackSolution(asteroid, earth, bodies, warningDays, dvBudget, mu, earthImpactTime);
        }

        return bestSolution ?? CreateMinimalSolution(warningDays);
    }

    static Solution CreateFallbackSolution(Body asteroid, Body earth, List<Body> bodies,
                                           double warningDays, double dvBudget, double mu, double impactTime)
    {
        double launchTime = Math.Min(10 * DAY, warningDays * 0.1 * DAY);
        double interceptTime = impactTime * 0.75;

        var earthAtLaunch = PropagateBody(earth, bodies, launchTime);
        var asteroidAtIntercept = PropagateBody(asteroid, bodies, interceptTime);

        double tof = interceptTime - launchTime;
        if (tof < DAY) return null;

        var lambert = SolveLambert(earthAtLaunch.Pos, asteroidAtIntercept.Pos, tof, mu, false);

        if (lambert != null)
        {
            Vec3 launchDv = lambert.V1 - earthAtLaunch.Vel;
            double launchDvMag = launchDv.Mag();

            if (launchDvMag > dvBudget)
            {
                launchDv = launchDv.Normalized() * (dvBudget * 0.95);
            }

            Vec3 relVel = lambert.V2 - asteroidAtIntercept.Vel;

            var solution = new Solution();
            solution.Burns.Add(new Burn(launchTime / DAY, launchDv));
            solution.Burns.Add(new Burn(interceptTime / DAY, relVel.Normalized() * Math.Min(relVel.Mag(), dvBudget * 0.04)));
            return solution;
        }

        return null;
    }

    static Solution CreateMinimalSolution(double warningDays)
    {
        var solution = new Solution();
        solution.Burns.Add(new Burn(1.0, new Vec3(5000, 0, 0)));
        solution.Burns.Add(new Burn(warningDays * 0.7, new Vec3(0, 100, 0)));
        return solution;
    }

    static double EstimateImpactTime(Body asteroid, Body earth, List<Body> bodies, double maxT)
    {
        double dt = DAY * 0.5;
        double minDist = double.MaxValue;
        double impactTime = maxT;

        var astState = new State(asteroid.Pos, asteroid.Vel);
        var earthState = new State(earth.Pos, earth.Vel);

        for (double t = 0; t <= maxT; t += dt)
        {
            if (t > 0)
            {
                astState = PropagateState(astState, asteroid, bodies, dt);
                earthState = PropagateState(earthState, earth, bodies, dt);
            }

            double dist = (astState.Pos - earthState.Pos).Mag();
            if (dist < minDist)
            {
                minDist = dist;
                impactTime = t;
            }

            if (dist < EARTH_RADIUS * 50 && dt > DAY * 0.1)
            {
                dt = DAY * 0.1;
            }
        }

        return impactTime;
    }

    static State PropagateBody(Body body, List<Body> allBodies, double time)
    {
        if (Math.Abs(time) < 1) return new State(body.Pos, body.Vel);

        double dt = Math.Min(3600, Math.Abs(time) / 100);
        int steps = (int)(Math.Abs(time) / dt);
        if (steps < 1) steps = 1;
        dt = time / steps;

        Vec3 pos = body.Pos;
        Vec3 vel = body.Vel;

        for (int i = 0; i < steps; i++)
        {
            Vec3 acc = ComputeAcceleration(pos, body, allBodies);
            vel = vel + acc * (dt * 0.5);
            pos = pos + vel * dt;
            acc = ComputeAcceleration(pos, body, allBodies);
            vel = vel + acc * (dt * 0.5);
        }

        return new State(pos, vel);
    }

    static State PropagateState(State state, Body self, List<Body> bodies, double dt)
    {
        Vec3 pos = state.Pos;
        Vec3 vel = state.Vel;

        Vec3 acc = ComputeAccelerationPos(pos, self, bodies);
        vel = vel + acc * (dt * 0.5);
        pos = pos + vel * dt;
        acc = ComputeAccelerationPos(pos, self, bodies);
        vel = vel + acc * (dt * 0.5);

        return new State(pos, vel);
    }

    static Vec3 ComputeAcceleration(Vec3 pos, Body self, List<Body> bodies)
    {
        Vec3 acc = new Vec3(0, 0, 0);

        foreach (var body in bodies)
        {
            if (body == self || body.Mass < 1e10) continue;

            Vec3 r = body.Pos - pos;
            double rMag = r.Mag();
            if (rMag < 1000) continue;

            double aMag = G * body.Mass / (rMag * rMag);
            acc = acc + r.Normalized() * aMag;
        }

        return acc;
    }

    static Vec3 ComputeAccelerationPos(Vec3 pos, Body self, List<Body> bodies)
    {
        return ComputeAcceleration(pos, self, bodies);
    }

    static LambertResult SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu, bool longWay)
    {
        if (tof <= 0 || mu <= 0) return null;

        double r1Mag = r1.Mag();
        double r2Mag = r2.Mag();
        if (r1Mag < 1e6 || r2Mag < 1e6) return null;

        Vec3 c = r2 - r1;
        double cMag = c.Mag();
        if (cMag < 1e6) return null;

        double s = (r1Mag + r2Mag + cMag) / 2;
        double lambda2 = 1 - cMag / s;
        if (lambda2 < 0) lambda2 = 0;
        double lambda = Math.Sqrt(lambda2);

        Vec3 ir1 = r1.Normalized();
        Vec3 ir2 = r2.Normalized();
        Vec3 ih = ir1.Cross(ir2);
        double ihMag = ih.Mag();

        if (ihMag < 1e-10)
        {
            ih = new Vec3(0, 0, 1);
        }
        else
        {
            ih = ih.Normalized();
        }

        Vec3 it1, it2;
        if (ih.Z < 0)
        {
            lambda = -lambda;
            it1 = ir1.Cross(ih);
            it2 = ir2.Cross(ih);
        }
        else
        {
            it1 = ih.Cross(ir1);
            it2 = ih.Cross(ir2);
        }

        if (longWay)
        {
            lambda = -lambda;
            it1 = it1 * -1;
            it2 = it2 * -1;
        }

        double T = Math.Sqrt(2 * mu / (s * s * s)) * tof;

        double x = SolveLambertX(lambda, T);
        if (double.IsNaN(x) || double.IsInfinity(x)) return null;

        double y2 = 1 - lambda * lambda * (1 - x * x);
        if (y2 < 0) y2 = 0;
        double y = Math.Sqrt(y2);

        double gamma = Math.Sqrt(mu * s / 2);
        double rho = (r1Mag - r2Mag) / cMag;
        double sigma2 = 1 - rho * rho;
        if (sigma2 < 0) sigma2 = 0;
        double sigma = Math.Sqrt(sigma2);

        double lyx = lambda * y - x;
        double lypx = lambda * y + x;
        double ylx = y + lambda * x;

        double vr1 = gamma * (lyx - rho * lypx) / r1Mag;
        double vr2 = -gamma * (lyx + rho * lypx) / r2Mag;
        double vt1 = gamma * sigma * ylx / r1Mag;
        double vt2 = gamma * sigma * ylx / r2Mag;

        Vec3 v1 = ir1 * vr1 + it1 * vt1;
        Vec3 v2 = ir2 * vr2 + it2 * vt2;

        if (double.IsNaN(v1.Mag()) || double.IsNaN(v2.Mag())) return null;

        return new LambertResult(v1, v2);
    }

    static double SolveLambertX(double lambda, double T)
    {
        if (T <= 0) return double.NaN;

        double lambdaClamped = Math.Max(-0.99999, Math.Min(0.99999, lambda));

        double x = 0;
        double lower = -0.9999;
        double upper = 0.9999;

        for (int iter = 0; iter < 60; iter++)
        {
            double y2 = 1 - lambdaClamped * lambdaClamped * (1 - x * x);
            if (y2 < 0) y2 = 1e-10;
            double y = Math.Sqrt(y2);

            double Tx = ComputeTFromX(x, y, lambdaClamped);

            if (Math.Abs(Tx - T) < 1e-12) return x;

            if (Tx > T)
                lower = x;
            else
                upper = x;

            x = (lower + upper) / 2;
        }

        return x;
    }

    static double ComputeTFromX(double x, double y, double lambda)
    {
        double eta = x * x - 1;

        if (Math.Abs(eta) < 1e-10)
        {
            return 2.0 / 3.0 * (1 - lambda * lambda * lambda);
        }
        else if (eta < 0)
        {
            double S = Math.Sqrt(-eta);
            double psi = Math.Acos(x * y + lambda * (-eta));
            return (psi - lambda * S) / (S * S * S);
        }
        else
        {
            double S = Math.Sqrt(eta);
            double F = Math.Log(Math.Abs(y * S + x + lambda));
            return (lambda * S - F) / (S * S * S);
        }
    }

    class Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public double Mag() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalized()
        {
            double m = Mag();
            return m > 1e-20 ? new Vec3(X / m, Y / m, Z / m) : new Vec3(0, 0, 0);
        }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public Vec3 Cross(Vec3 b) => new Vec3(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
    }

    class Body
    {
        public string Name;
        public double Mass;
        public Vec3 Pos, Vel;
        public Body(string name, double mass, Vec3 pos, Vec3 vel)
        {
            Name = name; Mass = mass; Pos = pos; Vel = vel;
        }
    }

    class State
    {
        public Vec3 Pos, Vel;
        public State(Vec3 pos, Vec3 vel) { Pos = pos; Vel = vel; }
    }

    class Burn
    {
        public double Time;
        public Vec3 DV;
        public Burn(double time, Vec3 dv) { Time = time; DV = dv; }
    }

    class Solution
    {
        public List<Burn> Burns = new List<Burn>();
    }

    class LambertResult
    {
        public Vec3 V1, V2;
        public LambertResult(Vec3 v1, Vec3 v2) { V1 = v1; V2 = v2; }
    }
}
