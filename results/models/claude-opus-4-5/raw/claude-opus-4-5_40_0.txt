#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;
long long best_obj;
vector<int> best_x;

void backtrack(int idx, vector<int>& x, long long cur_obj, vector<int>& status) {
    // Upper bound pruning: current + all positive undecided weights
    long long ub = cur_obj;
    for (int i = 0; i < n; i++) {
        if (status[i] == 0 && c[i] > 0) {
            ub += c[i];
        }
    }
    if (ub <= best_obj) return;

    // Find next undecided variable
    while (idx < n && status[idx] != 0) {
        idx++;
    }

    if (idx == n) {
        if (cur_obj > best_obj) {
            best_obj = cur_obj;
            best_x = x;
        }
        return;
    }

    // Try including first if weight is positive, else exclude first
    int first_val = (c[idx] > 0) ? 1 : 0;

    for (int trial = 0; trial < 2; trial++) {
        int val = (trial == 0) ? first_val : (1 - first_val);

        x[idx] = val;
        status[idx] = (val == 1) ? 1 : 2;

        vector<int> excluded;
        if (val == 1) {
            for (int neighbor : adj[idx]) {
                if (status[neighbor] == 0) {
                    status[neighbor] = 2;
                    x[neighbor] = 0;
                    excluded.push_back(neighbor);
                }
            }
        }

        backtrack(idx + 1, x, cur_obj + (val == 1 ? c[idx] : 0), status);

        for (int neighbor : excluded) {
            status[neighbor] = 0;
        }
        status[idx] = 0;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    c.resize(n);
    adj.resize(n);

    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    best_obj = 0;
    best_x.assign(n, 0);

    vector<int> x(n, 0);
    vector<int> status(n, 0);
    backtrack(0, x, 0, status);

    cout << best_obj << "\n";
    for (int i = 0; i < n; i++) {
        cout << best_x[i];
        if (i < n - 1) cout << " ";
    }
    cout << "\n";

    return 0;
}
