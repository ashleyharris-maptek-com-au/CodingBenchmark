use std::io::{self, BufRead, Write, BufWriter};

struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    x: f64,
    y: f64,
    z: f64,
    values: Vec<f64>,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse header
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let num_holes: usize = parts.next().unwrap().parse().unwrap();
    let num_properties: usize = parts.next().unwrap().parse().unwrap();

    let props_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = props_line.split_whitespace().map(String::from).collect();

    // Parse all holes and samples
    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut all_samples: Vec<Sample> = Vec::new();

    for hole_idx in 0..num_holes {
        let hole_line = match lines.next() {
            Some(Ok(line)) => line,
            _ => break,
        };
        let mut parts = hole_line.split_whitespace();

        let hole_id = parts.next().unwrap().to_string();
        let start_x: f64 = parts.next().unwrap().parse().unwrap();
        let start_y: f64 = parts.next().unwrap().parse().unwrap();
        let start_z: f64 = parts.next().unwrap().parse().unwrap();
        let dir_x: f64 = parts.next().unwrap().parse().unwrap();
        let dir_y: f64 = parts.next().unwrap().parse().unwrap();
        let dir_z: f64 = parts.next().unwrap().parse().unwrap();
        let _length: f64 = parts.next().unwrap().parse().unwrap();
        let num_samples: usize = parts.next().unwrap().parse().unwrap();

        hole_ids.push(hole_id);

        // Normalize direction vector
        let dir_len = (dir_x*dir_x + dir_y*dir_y + dir_z*dir_z).sqrt();
        let (dx, dy, dz) = if dir_len > 1e-10 {
            (dir_x / dir_len, dir_y / dir_len, dir_z / dir_len)
        } else {
            (0.0, 0.0, -1.0)
        };

        for sample_idx in 0..num_samples {
            let sample_line = match lines.next() {
                Some(Ok(line)) => line,
                _ => break,
            };
            let mut parts = sample_line.split_whitespace();

            let depth: f64 = parts.next().unwrap().parse().unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_properties);

            for _ in 0..num_properties {
                let val: f64 = parts.next()
                    .and_then(|s| s.parse().ok())
                    .unwrap_or(f64::NAN);
                values.push(val);
            }

            // Calculate 3D position of sample
            let x = start_x + depth * dx;
            let y = start_y + depth * dy;
            let z = start_z + depth * dz;

            all_samples.push(Sample {
                hole_idx,
                sample_idx,
                x, y, z,
                values,
            });
        }
    }

    if all_samples.is_empty() || num_properties == 0 {
        println!("0");
        return;
    }

    // Compute global stats for each property
    let mut prop_stats: Vec<(f64, f64)> = Vec::with_capacity(num_properties);
    for prop_idx in 0..num_properties {
        let vals: Vec<f64> = all_samples.iter()
            .map(|s| s.values[prop_idx])
            .filter(|v| v.is_finite())
            .collect();

        if vals.is_empty() {
            prop_stats.push((0.0, 1.0));
            continue;
        }

        let mean = vals.iter().sum::<f64>() / vals.len() as f64;
        let variance = vals.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / vals.len() as f64;
        let std = variance.sqrt().max(1e-10);
        prop_stats.push((mean, std));
    }

    // Adaptive strategy based on dataset size
    let total_samples = all_samples.len();
    let k_neighbors = if total_samples < 100 {
        (total_samples / 4).max(3).min(20)
    } else if total_samples < 10000 {
        15
    } else {
        10
    };

    // Build spatial grid for large datasets
    let use_grid = total_samples > 1000;
    let mut grid: std::collections::HashMap<(i64, i64, i64), Vec<usize>> = std::collections::HashMap::new();
    let cell_size = 50.0;

    if use_grid {
        for (idx, sample) in all_samples.iter().enumerate() {
            let cx = (sample.x / cell_size).floor() as i64;
            let cy = (sample.y / cell_size).floor() as i64;
            let cz = (sample.z / cell_size).floor() as i64;
            grid.entry((cx, cy, cz)).or_default().push(idx);
        }
    }

    // Find outliers
    let mut suspects: Vec<Suspect> = Vec::new();

    for (i, sample) in all_samples.iter().enumerate() {
        // Find k nearest neighbors
        let neighbors: Vec<usize> = if use_grid {
            find_neighbors_grid(&all_samples, i, &grid, cell_size, k_neighbors)
        } else {
            find_neighbors_brute(&all_samples, i, k_neighbors)
        };

        if neighbors.len() < 3 {
            continue;
        }

        // Check each property for outliers
        for prop_idx in 0..num_properties {
            let val = sample.values[prop_idx];
            if !val.is_finite() {
                continue;
            }

            // Compute local statistics from neighbors
            let neighbor_vals: Vec<f64> = neighbors.iter()
                .map(|&n| all_samples[n].values[prop_idx])
                .filter(|v| v.is_finite())
                .collect();

            if neighbor_vals.len() < 3 {
                continue;
            }

            let local_mean = neighbor_vals.iter().sum::<f64>() / neighbor_vals.len() as f64;
            let local_var = neighbor_vals.iter()
                .map(|v| (v - local_mean).powi(2))
                .sum::<f64>() / neighbor_vals.len() as f64;
            let local_std = local_var.sqrt();

            // Skip if no variation in neighbors
            if local_std < 1e-10 {
                let (global_mean, global_std) = prop_stats[prop_idx];
                let global_z = ((val - global_mean) / global_std).abs();
                if global_z > 4.0 && (val - local_mean).abs() > global_std * 0.5 {
                    suspects.push(Suspect {
                        hole_id: hole_ids[sample.hole_idx].clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence: global_z,
                    });
                }
                continue;
            }

            // Local z-score
            let local_z = (val - local_mean) / local_std;
            let local_z_abs = local_z.abs();

            // Also compute global z-score for comparison
            let (global_mean, global_std) = prop_stats[prop_idx];
            let global_z = ((val - global_mean) / global_std).abs();

            // Use MAD (Median Absolute Deviation) for robust outlier detection
            let mut sorted_vals = neighbor_vals.clone();
            sorted_vals.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let median = if sorted_vals.len() % 2 == 0 {
                (sorted_vals[sorted_vals.len()/2 - 1] + sorted_vals[sorted_vals.len()/2]) / 2.0
            } else {
                sorted_vals[sorted_vals.len()/2]
            };

            let mad: f64 = {
                let mut abs_devs: Vec<f64> = neighbor_vals.iter()
                    .map(|v| (v - median).abs())
                    .collect();
                abs_devs.sort_by(|a, b| a.partial_cmp(&b).unwrap());
                if abs_devs.len() % 2 == 0 {
                    (abs_devs[abs_devs.len()/2 - 1] + abs_devs[abs_devs.len()/2]) / 2.0
                } else {
                    abs_devs[abs_devs.len()/2]
                }
            };

            let modified_z = if mad > 1e-10 {
                0.6745 * (val - median).abs() / mad
            } else {
                local_z_abs
            };

            // Outlier criteria: high local deviation that's also significant globally
            let confidence = if modified_z > 3.5 && local_z_abs > 2.5 {
                modified_z * 0.6 + local_z_abs * 0.3 + global_z.min(5.0) * 0.1
            } else if local_z_abs > 3.5 && global_z > 2.0 {
                local_z_abs * 0.7 + global_z.min(5.0) * 0.3
            } else {
                continue;
            };

            suspects.push(Suspect {
                hole_id: hole_ids[sample.hole_idx].clone(),
                sample_idx: sample.sample_idx,
                property_name: property_names[prop_idx].clone(),
                confidence,
            });
        }
    }

    // Sort by confidence descending
    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Deduplicate (keep highest confidence for same hole/sample/property)
    let mut seen = std::collections::HashSet::new();
    suspects.retain(|s| {
        let key = (s.hole_id.clone(), s.sample_idx, s.property_name.clone());
        seen.insert(key)
    });

    // Output
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    writeln!(out, "{}", suspects.len()).unwrap();
    for s in &suspects {
        writeln!(out, "{} {} {} {:.4}", s.hole_id, s.sample_idx, s.property_name, s.confidence).unwrap();
    }
}

fn find_neighbors_brute(samples: &[Sample], idx: usize, k: usize) -> Vec<usize> {
    let sample = &samples[idx];
    let mut dists: Vec<(usize, f64)> = samples.iter()
        .enumerate()
        .filter(|(i, _)| *i != idx)
        .map(|(i, s)| {
            let dx = s.x - sample.x;
            let dy = s.y - sample.y;
            let dz = s.z - sample.z;
            (i, dx*dx + dy*dy + dz*dz)
        })
        .collect();

    dists.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
    dists.truncate(k);
    dists.into_iter().map(|(i, _)| i).collect()
}

fn find_neighbors_grid(
    samples: &[Sample],
    idx: usize,
    grid: &std::collections::HashMap<(i64, i64, i64), Vec<usize>>,
    cell_size: f64,
    k: usize,
) -> Vec<usize> {
    let sample = &samples[idx];
    let cx = (sample.x / cell_size).floor() as i64;
    let cy = (sample.y / cell_size).floor() as i64;
    let cz = (sample.z / cell_size).floor() as i64;

    let mut candidates: Vec<(usize, f64)> = Vec::new();

    // Search in expanding radius until we have enough neighbors
    for radius in 0..=5 {
        for dx in -radius..=radius {
            for dy in -radius..=radius {
                for dz in -radius..=radius {
                    if radius == 0 || dx.abs() == radius || dy.abs() == radius || dz.abs() == radius {
                        if let Some(indices) = grid.get(&(cx + dx, cy + dy, cz + dz)) {
                            for &i in indices {
                                if i != idx {
                                    let s = &samples[i];
                                    let dist_sq = (s.x - sample.x).powi(2)
                                        + (s.y - sample.y).powi(2)
                                        + (s.z - sample.z).powi(2);
                                    candidates.push((i, dist_sq));
                                }
                            }
                        }
                    }
                }
            }
        }

        if candidates.len() >= k * 2 {
            break;
        }
    }

    // Deduplicate and sort
    candidates.sort_by(|a, b| a.0.cmp(&b.0));
    candidates.dedup_by(|a, b| a.0 == b.0);
    candidates.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
    candidates.truncate(k);
    candidates.into_iter().map(|(i, _)| i).collect()
}
