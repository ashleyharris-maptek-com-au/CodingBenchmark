using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using System.Diagnostics;
using System.Runtime.CompilerServices;

class Program
{
    static int n, m;
    static List<int>[] adj;
    static HashSet<int>[] adjSet;
    static int[] degree;
    static int[] bestClique;
    static int bestSize = 0;
    static readonly object lockObj = new object();
    static Stopwatch sw;
    static readonly long TimeLimitMs = 270000;
    static volatile bool timeOut = false;

    static void Main()
    {
        sw = Stopwatch.StartNew();
        ReadInput();

        Task.Run(() =>
        {
            while (sw.ElapsedMilliseconds < TimeLimitMs)
                Thread.Sleep(50);
            timeOut = true;
        });

        double density = n > 1 ? (2.0 * m) / (n * (n - 1.0)) : 0;

        if (n <= 200)
            ExactBronKerbosch();
        else if (n <= 5000 && density < 0.9)
            BranchAndBoundMaxClique();
        else
            ParallelHeuristic();

        if (bestSize == 0)
            GreedyClique(0);

        Console.WriteLine(bestSize);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }

    static void ReadInput()
    {
        string line = Console.ReadLine();
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(parts[0]);
        m = int.Parse(parts[1]);

        adj = new List<int>[n];
        adjSet = new HashSet<int>[n];
        degree = new int[n];

        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            adjSet[i] = new HashSet<int>();
        }

        for (int i = 0; i < m; i++)
        {
            line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) { i--; continue; }
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v && !adjSet[u].Contains(v))
            {
                adj[u].Add(v);
                adj[v].Add(u);
                adjSet[u].Add(v);
                adjSet[v].Add(u);
                degree[u]++;
                degree[v]++;
            }
        }

        for (int i = 0; i < n; i++)
            adj[i].Sort((a, b) => degree[b].CompareTo(degree[a]));
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static bool HasEdge(int u, int v)
    {
        return adjSet[u].Contains(v);
    }

    static void UpdateBest(List<int> clique)
    {
        lock (lockObj)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                bestClique = clique.ToArray();
            }
        }
    }

    static void UpdateBest(int[] clique, int size)
    {
        lock (lockObj)
        {
            if (size > bestSize)
            {
                bestSize = size;
                bestClique = new int[size];
                Array.Copy(clique, bestClique, size);
            }
        }
    }

    static void ExactBronKerbosch()
    {
        var R = new List<int>();
        var P = new HashSet<int>(Enumerable.Range(0, n).Where(i => degree[i] > 0));
        var X = new HashSet<int>();

        if (P.Count == 0)
        {
            bestSize = 1;
            bestClique = new int[] { 0 };
            return;
        }

        BronKerboschPivot(R, P, X);
    }

    static void BronKerboschPivot(List<int> R, HashSet<int> P, HashSet<int> X)
    {
        if (timeOut) return;

        if (P.Count == 0 && X.Count == 0)
        {
            UpdateBest(R);
            return;
        }

        if (R.Count + P.Count <= bestSize) return;

        int pivot = -1;
        int maxConn = -1;
        foreach (int u in P.Concat(X))
        {
            int conn = adjSet[u].Count(v => P.Contains(v));
            if (conn > maxConn)
            {
                maxConn = conn;
                pivot = u;
            }
        }

        var candidates = P.Where(v => !adjSet[pivot].Contains(v)).OrderByDescending(v => degree[v]).ToList();

        foreach (int v in candidates)
        {
            if (timeOut) return;

            var newR = new List<int>(R) { v };
            var newP = new HashSet<int>(P.Where(u => adjSet[v].Contains(u)));
            var newX = new HashSet<int>(X.Where(u => adjSet[v].Contains(u)));

            BronKerboschPivot(newR, newP, newX);

            P.Remove(v);
            X.Add(v);
        }
    }

    static void BranchAndBoundMaxClique()
    {
        var order = Enumerable.Range(0, n).OrderByDescending(i => degree[i]).ToArray();
        var position = new int[n];
        for (int i = 0; i < n; i++) position[order[i]] = i;

        var colorBound = new int[n];
        ComputeColorBounds(order, colorBound);

        GreedyClique(order[0]);

        int numThreads = Math.Min(Environment.ProcessorCount, Math.Min(n, 12));
        var tasks = new List<Task>();

        for (int t = 0; t < numThreads && t < n; t++)
        {
            int startIdx = t;
            tasks.Add(Task.Run(() =>
            {
                for (int i = startIdx; i < n && !timeOut; i += numThreads)
                {
                    int v = order[i];
                    if (colorBound[i] <= bestSize) continue;

                    var clique = new int[n];
                    clique[0] = v;
                    var candidates = adj[v].Where(u => position[u] > i).OrderByDescending(u => degree[u]).ToList();

                    BranchAndBound(clique, 1, candidates, colorBound, position);
                }
            }));
        }

        Task.WaitAll(tasks.ToArray());
    }

    static void ComputeColorBounds(int[] order, int[] colorBound)
    {
        var color = new int[n];
        Array.Fill(color, -1);
        int maxColor = 0;

        for (int i = n - 1; i >= 0; i--)
        {
            int v = order[i];
            var usedColors = new HashSet<int>();
            foreach (int u in adj[v])
            {
                if (color[u] >= 0)
                    usedColors.Add(color[u]);
            }

            int c = 0;
            while (usedColors.Contains(c)) c++;
            color[v] = c;
            maxColor = Math.Max(maxColor, c + 1);
            colorBound[i] = maxColor;
        }
    }

    static void BranchAndBound(int[] clique, int cliqueSize, List<int> candidates, int[] colorBound, int[] position)
    {
        if (timeOut) return;

        if (candidates.Count == 0)
        {
            UpdateBest(clique, cliqueSize);
            return;
        }

        if (cliqueSize + candidates.Count <= bestSize) return;

        var candColors = new int[candidates.Count];
        var usedColors = new bool[candidates.Count + 1];
        for (int i = candidates.Count - 1; i >= 0; i--)
        {
            Array.Clear(usedColors, 0, usedColors.Length);
            int v = candidates[i];
            for (int j = i + 1; j < candidates.Count; j++)
            {
                if (HasEdge(v, candidates[j]))
                    usedColors[candColors[j]] = true;
            }
            int c = 1;
            while (usedColors[c]) c++;
            candColors[i] = c;
        }

        int maxColorInCand = candColors.Length > 0 ? candColors.Max() : 0;
        if (cliqueSize + maxColorInCand <= bestSize) return;

        for (int i = 0; i < candidates.Count && !timeOut; i++)
        {
            if (cliqueSize + candColors[i] <= bestSize) continue;

            int v = candidates[i];
            clique[cliqueSize] = v;

            var newCandidates = new List<int>();
            for (int j = i + 1; j < candidates.Count; j++)
            {
                if (HasEdge(v, candidates[j]))
                    newCandidates.Add(candidates[j]);
            }

            BranchAndBound(clique, cliqueSize + 1, newCandidates, colorBound, position);
        }
    }

    static void ParallelHeuristic()
    {
        int numThreads = Environment.ProcessorCount;
        var tasks = new Task[numThreads];
        var byDegree = Enumerable.Range(0, n).OrderByDescending(i => degree[i]).ToArray();

        for (int t = 0; t < numThreads; t++)
        {
            int threadId = t;
            tasks[t] = Task.Run(() =>
            {
                var random = new Random(threadId * 12345);

                for (int iter = 0; iter < 1000 && !timeOut; iter++)
                {
                    int startIdx = (threadId + iter * numThreads) % Math.Min(n, 1000);
                    int start = byDegree[startIdx];
                    GreedyCliqueLocal(start, random);

                    if (iter % 10 == 0)
                    {
                        int randomStart = byDegree[random.Next(Math.Min(n, 100))];
                        GreedyCliqueLocal(randomStart, random);
                    }
                }

                for (int iter = 0; iter < 100 && !timeOut; iter++)
                {
                    LocalSearch(random);
                }
            });
        }

        Task.WaitAll(tasks);
    }

    static void GreedyClique(int start)
    {
        var clique = new List<int> { start };
        var candidates = new HashSet<int>(adj[start]);

        while (candidates.Count > 0)
        {
            int best = -1;
            int bestScore = -1;
            foreach (int v in candidates)
            {
                int score = candidates.Count(u => HasEdge(v, u));
                if (score > bestScore || (score == bestScore && degree[v] > degree[best]))
                {
                    bestScore = score;
                    best = v;
                }
            }

            clique.Add(best);
            candidates.RemoveWhere(v => !HasEdge(best, v));
        }

        UpdateBest(clique);
    }

    static void GreedyCliqueLocal(int start, Random random)
    {
        var clique = new List<int> { start };
        var candidates = new List<int>(adj[start]);

        while (candidates.Count > 0)
        {
            int bestIdx = 0;
            int bestScore = -1;

            int checkCount = Math.Min(candidates.Count, 50);
            for (int i = 0; i < checkCount; i++)
            {
                int idx = i < 10 ? i : random.Next(candidates.Count);
                int v = candidates[idx];
                int score = 0;
                foreach (int u in candidates)
                {
                    if (HasEdge(v, u)) score++;
                }
                if (score > bestScore)
                {
                    bestScore = score;
                    bestIdx = idx;
                }
            }

            int chosen = candidates[bestIdx];
            clique.Add(chosen);
            candidates = candidates.Where(v => HasEdge(chosen, v)).ToList();
        }

        UpdateBest(clique);
    }

    static void LocalSearch(Random random)
    {
        if (bestSize == 0) return;

        int[] currentClique;
        int currentSize;
        lock (lockObj)
        {
            if (bestClique == null) return;
            currentClique = (int[])bestClique.Clone();
            currentSize = bestSize;
        }

        var cliqueSet = new HashSet<int>(currentClique);

        for (int iter = 0; iter < 100 && !timeOut; iter++)
        {
            var nearClique = new List<int>();
            for (int v = 0; v < n; v++)
            {
                if (cliqueSet.Contains(v)) continue;
                int missing = 0;
                foreach (int u in cliqueSet)
                {
                    if (!HasEdge(v, u))
                    {
                        missing++;
                        if (missing > 1) break;
                    }
                }
                if (missing == 1)
                    nearClique.Add(v);
            }

            if (nearClique.Count >= 2)
            {
                for (int i = 0; i < Math.Min(nearClique.Count, 20) && !timeOut; i++)
                {
                    int v1 = nearClique[i];
                    for (int j = i + 1; j < Math.Min(nearClique.Count, 20); j++)
                    {
                        int v2 = nearClique[j];
                        if (!HasEdge(v1, v2)) continue;

                        int toRemove = -1;
                        foreach (int u in cliqueSet)
                        {
                            if (!HasEdge(v1, u) || !HasEdge(v2, u))
                            {
                                if (toRemove == -1) toRemove = u;
                                else { toRemove = -2; break; }
                            }
                        }

                        if (toRemove >= 0)
                        {
                            cliqueSet.Remove(toRemove);
                            cliqueSet.Add(v1);
                            cliqueSet.Add(v2);
                            UpdateBest(cliqueSet.ToList());
                            return;
                        }
                    }
                }
            }

            if (iter % 20 == 19)
            {
                int remove = currentClique[random.Next(currentSize)];
                cliqueSet.Remove(remove);

                var addCandidates = Enumerable.Range(0, n)
                    .Where(v => !cliqueSet.Contains(v) && cliqueSet.All(u => HasEdge(v, u)))
                    .OrderByDescending(v => degree[v])
                    .ToList();

                foreach (int v in addCandidates)
                {
                    if (cliqueSet.All(u => HasEdge(v, u)))
                        cliqueSet.Add(v);
                }

                if (cliqueSet.Count > bestSize)
                    UpdateBest(cliqueSet.ToList());

                lock (lockObj)
                {
                    if (bestClique != null)
                        cliqueSet = new HashSet<int>(bestClique);
                }
            }
        }
    }
}
