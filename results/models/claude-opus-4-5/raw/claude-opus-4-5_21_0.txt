#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <deque>
#include <map>
#include <climits>

using namespace std;

int D;
vector<int> bounds;
deque<vector<int>> snake;
set<vector<int>> obstacles;
set<vector<int>> food;
set<vector<int>> snakeBody;

bool inBounds(const vector<int>& p) {
    for (int i = 0; i < D; i++)
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    return true;
}

int manhattan(const vector<int>& a, const vector<int>& b) {
    int d = 0;
    for (int i = 0; i < D; i++) d += abs(a[i] - b[i]);
    return d;
}

const vector<int>* nearestFood(const vector<int>& h) {
    const vector<int>* best = nullptr;
    int bestD = INT_MAX;
    for (const auto& f : food) {
        int dist = manhattan(h, f);
        if (dist < bestD) { bestD = dist; best = &f; }
    }
    return best;
}

bool canMove(const vector<int>& next, bool eating) {
    if (!inBounds(next) || obstacles.count(next)) return false;
    if (snakeBody.count(next)) {
        return !eating && snake.size() > 1 && next == snake.back();
    }
    return true;
}

pair<int,int> greedyMove(const vector<int>& head, const vector<int>& target) {
    int bestScore = INT_MIN;
    pair<int,int> best = {0, 1};
    bool found = false;

    for (int ax = 0; ax < D; ax++) {
        for (int dr : {-1, 1}) {
            vector<int> nxt = head;
            nxt[ax] += dr;
            bool eats = food.count(nxt) > 0;
            if (!canMove(nxt, eats)) continue;
            found = true;

            int score = (manhattan(head, target) - manhattan(nxt, target)) * 1000;

            for (int a2 = 0; a2 < D; a2++) {
                for (int d2 : {-1, 1}) {
                    vector<int> fut = nxt;
                    fut[a2] += d2;
                    if (inBounds(fut) && !obstacles.count(fut) && !snakeBody.count(fut))
                        score += 10;
                }
            }

            if (score > bestScore) { bestScore = score; best = {ax, dr}; }
        }
    }

    if (!found) {
        for (int ax = 0; ax < D; ax++)
            for (int dr : {-1, 1}) {
                vector<int> nxt = head;
                nxt[ax] += dr;
                if (inBounds(nxt)) return {ax, dr};
            }
    }
    return best;
}

pair<int,int> bfsMove(const vector<int>& head, const vector<int>& target, int limit) {
    map<vector<int>, pair<int,int>> firstMove;
    queue<vector<int>> q;
    set<vector<int>> visited;
    visited.insert(head);

    for (int ax = 0; ax < D; ax++) {
        for (int dr : {-1, 1}) {
            vector<int> nxt = head;
            nxt[ax] += dr;
            bool eats = food.count(nxt) > 0;
            if (!canMove(nxt, eats)) continue;
            if (nxt == target) return {ax, dr};
            firstMove[nxt] = {ax, dr};
            q.push(nxt);
            visited.insert(nxt);
        }
    }

    int cnt = 0;
    while (!q.empty() && cnt++ < limit) {
        vector<int> cur = q.front();
        q.pop();
        for (int ax = 0; ax < D; ax++) {
            for (int dr : {-1, 1}) {
                vector<int> nxt = cur;
                nxt[ax] += dr;
                if (visited.count(nxt)) continue;
                if (!inBounds(nxt) || obstacles.count(nxt)) continue;

                if (nxt == target) return firstMove[cur];
                firstMove[nxt] = firstMove[cur];
                q.push(nxt);
                visited.insert(nxt);
            }
        }
    }

    return greedyMove(head, target);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int maxT;
    cin >> D >> maxT;
    bounds.resize(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    int len;
    cin >> len;
    vector<int> head(D);
    for (int i = 0; i < D; i++) cin >> head[i];

    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        vector<int> p(D);
        for (int j = 0; j < D; j++) cin >> p[j];
        food.insert(p);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        vector<int> p(D);
        for (int j = 0; j < D; j++) cin >> p[j];
        obstacles.insert(p);
    }

    snake.push_back(head);
    snakeBody.insert(head);
    for (int i = 1; i < len; i++) {
        vector<int> b = head;
        b[0] -= i;
        if (inBounds(b) && !obstacles.count(b) && !snakeBody.count(b)) {
            snake.push_back(b);
            snakeBody.insert(b);
        }
    }

    int bfsLimit = D <= 4 ? 100000 : D <= 8 ? 10000 : 1000;

    for (int t = 0; t < maxT && !food.empty(); t++) {
        vector<int> h = snake.front();
        const vector<int>* tgt = nearestFood(h);
        if (!tgt) break;

        pair<int,int> move = (D <= 10) ? bfsMove(h, *tgt, bfsLimit) : greedyMove(h, *tgt);
        cout << move.first << " " << move.second << "\n";

        vector<int> nh = h;
        nh[move.first] += move.second;
        snake.push_front(nh);
        snakeBody.insert(nh);

        if (food.count(nh)) {
            food.erase(nh);
        } else if (snake.size() > 1) {
            snakeBody.erase(snake.back());
            snake.pop_back();
        }
    }

    return 0;
}
