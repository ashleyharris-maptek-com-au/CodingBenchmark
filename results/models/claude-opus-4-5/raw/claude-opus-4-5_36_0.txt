#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2048;

int n, m;
int best = 0;
vector<int> bestSet;
bitset<MAXN> adj[MAXN];

void search(vector<int>& current, bitset<MAXN> mask) {
    int cnt = (int)mask.count();
    if ((int)current.size() + cnt <= best) return;

    if (cnt == 0) {
        best = (int)current.size();
        bestSet = current;
        return;
    }

    // Find vertex with min degree in masked subgraph (greedy heuristic for MIS)
    int minDeg = n + 1, v = -1;
    for (int i = 0; i < n; i++) {
        if (mask[i]) {
            int deg = (int)(mask & adj[i]).count();
            if (deg < minDeg) {
                minDeg = deg;
                v = i;
            }
        }
    }

    mask[v] = 0;

    // Include v (try this first to find good solutions quickly)
    bitset<MAXN> newMask = mask & ~adj[v];
    current.push_back(v);
    search(current, newMask);
    current.pop_back();

    // Pruning before exclude
    if ((int)current.size() + (int)mask.count() <= best) return;

    // Exclude v
    search(current, mask);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u][v] = adj[v][u] = 1;
    }

    // Greedy init - prioritize low-degree vertices
    vector<pair<int,int>> byDegree(n);
    for (int i = 0; i < n; i++) {
        byDegree[i] = {(int)adj[i].count(), i};
    }
    sort(byDegree.begin(), byDegree.end());

    bitset<MAXN> used;
    for (int j = 0; j < n; j++) {
        int i = byDegree[j].second;
        if (!used[i]) {
            bestSet.push_back(i);
            used[i] = 1;
            used |= adj[i];
        }
    }
    best = (int)bestSet.size();

    bitset<MAXN> mask;
    for (int i = 0; i < n; i++) mask[i] = 1;

    vector<int> current;
    search(current, mask);

    cout << best << "\n";
    for (int i = 0; i < best; i++) {
        if (i > 0) cout << " ";
        cout << bestSet[i];
    }
    cout << "\n";

    return 0;
}
