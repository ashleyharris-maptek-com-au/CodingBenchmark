import sys

def solve_exact_cover(universe_size, sets):
    """Solve exact cover using backtracking (Algorithm X)."""
    uncovered = set(range(universe_size))
    covered = set()

    def backtrack(chosen):
        if not uncovered:
            return chosen[:]

        # Choose element that appears in fewest remaining valid sets (MRV heuristic)
        target = min(uncovered)

        # Try each set that contains this element
        for i in range(len(sets)):
            if i in chosen:
                continue
            s = sets[i]
            if target not in s:
                continue
            # Check if this set overlaps with already covered elements
            if s & covered:
                continue

            # Choose this set
            chosen.append(i)
            covered.update(s)
            uncovered.difference_update(s)

            result = backtrack(chosen)
            if result is not None:
                return result

            # Unchoose
            chosen.pop()
            covered.difference_update(s)
            uncovered.update(s)

        return None

    return backtrack([])

def main():
    lines = sys.stdin.read().strip().split('\n')
    U, S = map(int, lines[0].split())

    sets = []
    for i in range(1, S + 1):
        if i < len(lines) and lines[i].strip():
            s = set(map(int, lines[i].split()))
        else:
            s = set()
        sets.append(s)

    result = solve_exact_cover(U, sets)

    if result is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(' '.join(map(str, sorted(result))))

if __name__ == "__main__":
    main()
