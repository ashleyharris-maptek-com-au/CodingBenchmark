use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut input = String::new();
    stdin.lock().read_line(&mut input).unwrap();
    let p: Vec<usize> = input.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let (w, h, obs_count, cycle) = (p[0], p[1], p[2], p[3] == 1);

    let mut blocked = vec![vec![false; h]; w];
    for _ in 0..obs_count {
        input.clear();
        stdin.lock().read_line(&mut input).unwrap();
        let c: Vec<usize> = input.split_whitespace().map(|s| s.parse().unwrap()).collect();
        blocked[c[0]][c[1]] = true;
    }

    let n = w * h - obs_count;

    if n == 0 || blocked[0][0] {
        writeln!(out, "0").unwrap();
        return;
    }

    if n == 1 {
        writeln!(out, "{}", if cycle { "0" } else { "1\n0 0" }).unwrap();
        return;
    }

    if !is_connected(&blocked, w, h) {
        writeln!(out, "0").unwrap();
        return;
    }

    if cycle {
        let (b, wh) = count_colors(&blocked, w, h);
        if b != wh {
            writeln!(out, "0").unwrap();
            return;
        }
        if !has_neighbor(0, 0, &blocked, w, h) {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    if obs_count == 0 {
        if let Some(path) = snake_pattern(w, h, cycle) {
            output(&mut out, &path);
            return;
        }
    }

    let mut visited = vec![vec![false; h]; w];
    let mut seen = vec![vec![0u32; h]; w];
    let mut gen = 0u32;
    visited[0][0] = true;
    let mut path = vec![(0usize, 0usize)];

    if search(&blocked, &mut visited, &mut seen, &mut gen, &mut path, w, h, n, cycle) {
        output(&mut out, &path);
    } else {
        writeln!(out, "0").unwrap();
    }
}

fn output<W: Write>(out: &mut W, path: &[(usize, usize)]) {
    writeln!(out, "{}", path.len()).unwrap();
    for &(x, y) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}

fn is_connected(blocked: &[Vec<bool>], w: usize, h: usize) -> bool {
    let mut start = None;
    let mut total = 0;
    for x in 0..w {
        for y in 0..h {
            if !blocked[x][y] {
                total += 1;
                if start.is_none() { start = Some((x, y)); }
            }
        }
    }
    let (sx, sy) = match start { Some(s) => s, None => return true };
    let mut seen = vec![vec![false; h]; w];
    let mut queue = VecDeque::new();
    queue.push_back((sx, sy));
    seen[sx][sy] = true;
    let mut count = 1;
    while let Some((x, y)) = queue.pop_front() {
        for (nx, ny) in adj(x, y, w, h) {
            if !blocked[nx][ny] && !seen[nx][ny] {
                seen[nx][ny] = true;
                count += 1;
                queue.push_back((nx, ny));
            }
        }
    }
    count == total
}

fn count_colors(blocked: &[Vec<bool>], w: usize, h: usize) -> (usize, usize) {
    let (mut b, mut wh) = (0, 0);
    for x in 0..w {
        for y in 0..h {
            if !blocked[x][y] {
                if (x + y) % 2 == 0 { wh += 1; } else { b += 1; }
            }
        }
    }
    (b, wh)
}

fn has_neighbor(x: usize, y: usize, blocked: &[Vec<bool>], w: usize, h: usize) -> bool {
    adj(x, y, w, h).iter().any(|&(nx, ny)| !blocked[nx][ny])
}

fn adj(x: usize, y: usize, w: usize, h: usize) -> Vec<(usize, usize)> {
    let mut r = Vec::with_capacity(4);
    if x > 0 { r.push((x-1, y)); }
    if x+1 < w { r.push((x+1, y)); }
    if y > 0 { r.push((x, y-1)); }
    if y+1 < h { r.push((x, y+1)); }
    r
}

fn snake_pattern(w: usize, h: usize, cycle: bool) -> Option<Vec<(usize, usize)>> {
    let n = w * h;
    if cycle && n % 2 != 0 { return None; }
    let mut path = Vec::with_capacity(n);
    if h % 2 == 0 {
        for x in 0..w {
            if x % 2 == 0 { for y in 0..h { path.push((x, y)); } }
            else { for y in (0..h).rev() { path.push((x, y)); } }
        }
    } else if w % 2 == 0 {
        for y in 0..h {
            if y % 2 == 0 { for x in 0..w { path.push((x, y)); } }
            else { for x in (0..w).rev() { path.push((x, y)); } }
        }
    } else {
        if cycle { return None; }
        for x in 0..w {
            if x % 2 == 0 { for y in 0..h { path.push((x, y)); } }
            else { for y in (0..h).rev() { path.push((x, y)); } }
        }
    }
    if cycle {
        let (lx, ly) = *path.last().unwrap();
        if !((lx == 0 && ly == 1) || (lx == 1 && ly == 0)) { return None; }
    }
    Some(path)
}

fn search(
    blocked: &[Vec<bool>], visited: &mut [Vec<bool>], seen: &mut [Vec<u32>], gen: &mut u32,
    path: &mut Vec<(usize, usize)>, w: usize, h: usize, n: usize, cycle: bool
) -> bool {
    if path.len() == n {
        if !cycle { return true; }
        let (x, y) = *path.last().unwrap();
        return (x == 0 && y == 1) || (x == 1 && y == 0);
    }
    let (cx, cy) = *path.last().unwrap();
    let mut nbrs: Vec<(usize, usize)> = adj(cx, cy, w, h)
        .into_iter().filter(|&(nx, ny)| !blocked[nx][ny] && !visited[nx][ny]).collect();
    nbrs.sort_by_key(|&(nx, ny)| degree(nx, ny, blocked, visited, w, h));
    for (nx, ny) in nbrs {
        visited[nx][ny] = true;
        path.push((nx, ny));
        let remaining = n - path.len();
        if connected(blocked, visited, seen, gen, w, h, remaining) {
            if search(blocked, visited, seen, gen, path, w, h, n, cycle) { return true; }
        }
        path.pop();
        visited[nx][ny] = false;
    }
    false
}

fn degree(x: usize, y: usize, blocked: &[Vec<bool>], visited: &[Vec<bool>], w: usize, h: usize) -> usize {
    adj(x, y, w, h).iter().filter(|&&(nx, ny)| !blocked[nx][ny] && !visited[nx][ny]).count()
}

fn connected(blocked: &[Vec<bool>], visited: &[Vec<bool>], seen: &mut [Vec<u32>], gen: &mut u32, w: usize, h: usize, expect: usize) -> bool {
    if expect == 0 { return true; }
    *gen += 1;
    let g = *gen;
    let mut start = None;
    'outer: for x in 0..w {
        for y in 0..h {
            if !blocked[x][y] && !visited[x][y] { start = Some((x, y)); break 'outer; }
        }
    }
    let (sx, sy) = match start { Some(s) => s, None => return false };
    let mut queue = VecDeque::new();
    queue.push_back((sx, sy));
    seen[sx][sy] = g;
    let mut count = 1;
    while let Some((x, y)) = queue.pop_front() {
        for (nx, ny) in adj(x, y, w, h) {
            if !blocked[nx][ny] && !visited[nx][ny] && seen[nx][ny] != g {
                seen[nx][ny] = g;
                count += 1;
                if count == expect { return true; }
                queue.push_back((nx, ny));
            }
        }
    }
    count == expect
}
