struct Point {
    float x, y;
    int idx;
};

inline float dist(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

inline float distSq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx*dx + dy*dy;
}

vector<int> nearestNeighbor(const vector<Point>& cities) {
    int n = (int)cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    int current = 0;
    tour.push_back(current);
    visited[current] = true;

    for (int i = 1; i < n; i++) {
        float bestDist = numeric_limits<float>::max();
        int bestNext = -1;
        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = distSq(cities[current], cities[j]);
                if (d < bestDist) {
                    bestDist = d;
                    bestNext = j;
                }
            }
        }
        tour.push_back(bestNext);
        visited[bestNext] = true;
        current = bestNext;
    }
    return tour;
}

void twoOptImprove(vector<int>& tour, const vector<Point>& cities, int maxIter = 1000000) {
    int n = (int)tour.size();
    if (n < 4) return;

    bool improved = true;
    int iter = 0;
    while (improved && iter < maxIter) {
        improved = false;
        iter++;
        for (int i = 0; i < n - 2 && !improved; i++) {
            for (int j = i + 2; j < n; j++) {
                if (i == 0 && j == n - 1) continue;

                int i1 = tour[i], i2 = tour[i+1];
                int j1 = tour[j], j2 = tour[(j+1) % n];

                float d1 = dist(cities[i1], cities[i2]) + dist(cities[j1], cities[j2]);
                float d2 = dist(cities[i1], cities[j1]) + dist(cities[i2], cities[j2]);

                if (d2 < d1 - 1e-6f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    break;
                }
            }
        }
    }
}

void limitedTwoOpt(vector<int>& tour, const vector<Point>& cities, int passes) {
    int n = (int)tour.size();
    if (n < 4) return;

    for (int pass = 0; pass < passes; pass++) {
        bool improved = false;
        for (int i = 0; i < n - 2; i++) {
            int jEnd = min(i + 50, n);
            for (int j = i + 2; j < jEnd; j++) {
                if (i == 0 && j == n - 1) continue;

                int i1 = tour[i], i2 = tour[i+1];
                int j1 = tour[j], j2 = tour[(j+1) % n];

                float d1 = dist(cities[i1], cities[i2]) + dist(cities[j1], cities[j2]);
                float d2 = dist(cities[i1], cities[j1]) + dist(cities[i2], cities[j2]);

                if (d2 < d1 - 1e-6f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

struct GridCell {
    vector<int> points;
};

vector<int> gridBasedTour(vector<Point>& cities) {
    int n = (int)cities.size();

    float minX = cities[0].x, maxX = cities[0].x;
    float minY = cities[0].y, maxY = cities[0].y;
    for (const auto& c : cities) {
        minX = min(minX, c.x); maxX = max(maxX, c.x);
        minY = min(minY, c.y); maxY = max(maxY, c.y);
    }

    int gridSize = max(1, (int)sqrt((double)n / 10.0));
    float cellW = (maxX - minX + 1e-6f) / gridSize;
    float cellH = (maxY - minY + 1e-6f) / gridSize;

    vector<vector<GridCell>> grid(gridSize, vector<GridCell>(gridSize));

    for (int i = 0; i < n; i++) {
        int gx = min(gridSize - 1, (int)((cities[i].x - minX) / cellW));
        int gy = min(gridSize - 1, (int)((cities[i].y - minY) / cellH));
        grid[gx][gy].points.push_back(i);
    }

    for (int gx = 0; gx < gridSize; gx++) {
        for (int gy = 0; gy < gridSize; gy++) {
            auto& pts = grid[gx][gy].points;
            if (pts.size() > 1) {
                sort(pts.begin(), pts.end(), [&](int a, int b) {
                    return cities[a].x + cities[a].y < cities[b].x + cities[b].y;
                });
            }
        }
    }

    vector<int> tour;
    tour.reserve(n);
    vector<bool> visitedCell(gridSize * gridSize, false);

    int startGx = min(gridSize - 1, (int)((cities[0].x - minX) / cellW));
    int startGy = min(gridSize - 1, (int)((cities[0].y - minY) / cellH));

    auto addCell = [&](int gx, int gy, bool reverse_order) {
        auto& pts = grid[gx][gy].points;
        if (reverse_order) {
            for (int i = (int)pts.size() - 1; i >= 0; i--) {
                tour.push_back(pts[i]);
            }
        } else {
            for (int p : pts) tour.push_back(p);
        }
    };

    bool leftToRight = true;
    for (int gy = 0; gy < gridSize; gy++) {
        if (leftToRight) {
            for (int gx = 0; gx < gridSize; gx++) {
                addCell(gx, gy, false);
            }
        } else {
            for (int gx = gridSize - 1; gx >= 0; gx--) {
                addCell(gx, gy, true);
            }
        }
        leftToRight = !leftToRight;
    }

    int startPos = 0;
    for (int i = 0; i < (int)tour.size(); i++) {
        if (tour[i] == 0) { startPos = i; break; }
    }

    vector<int> result;
    result.reserve(n);
    for (int i = 0; i < n; i++) {
        result.push_back(tour[(startPos + i) % n]);
    }

    return result;
}

vector<int> stripeTour(vector<Point>& cities) {
    int n = (int)cities.size();

    vector<int> indices(n);
    for (int i = 0; i < n; i++) indices[i] = i;

    float minX = cities[0].x, maxX = cities[0].x;
    for (const auto& c : cities) {
        minX = min(minX, c.x); maxX = max(maxX, c.x);
    }

    int numStripes = max(1, (int)sqrt((double)n / 100.0));
    float stripeW = (maxX - minX + 1e-6f) / numStripes;

    sort(indices.begin(), indices.end(), [&](int a, int b) {
        int sa = min(numStripes - 1, (int)((cities[a].x - minX) / stripeW));
        int sb = min(numStripes - 1, (int)((cities[b].x - minX) / stripeW));
        if (sa != sb) return sa < sb;
        if (sa % 2 == 0) return cities[a].y < cities[b].y;
        return cities[a].y > cities[b].y;
    });

    int startPos = 0;
    for (int i = 0; i < n; i++) {
        if (indices[i] == 0) { startPos = i; break; }
    }

    vector<int> result;
    result.reserve(n);
    for (int i = 0; i < n; i++) {
        result.push_back(indices[(startPos + i) % n]);
    }

    return result;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].idx = i;
    }

    vector<int> tour;

    if (n <= 1) {
        tour.push_back(0);
    } else if (n <= 2000) {
        tour = nearestNeighbor(cities);
        twoOptImprove(tour, cities, n * n);
    } else if (n <= 20000) {
        tour = nearestNeighbor(cities);
        limitedTwoOpt(tour, cities, 5);
    } else if (n <= 200000) {
        tour = gridBasedTour(cities);
        limitedTwoOpt(tour, cities, 2);
    } else {
        tour = stripeTour(cities);
    }

    if (tour[0] != 0) {
        int startPos = 0;
        for (int i = 0; i < n; i++) {
            if (tour[i] == 0) { startPos = i; break; }
        }
        vector<int> result;
        result.reserve(n);
        for (int i = 0; i < n; i++) {
            result.push_back(tour[(startPos + i) % n]);
        }
        tour = result;
    }

    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';

    return 0;
}
