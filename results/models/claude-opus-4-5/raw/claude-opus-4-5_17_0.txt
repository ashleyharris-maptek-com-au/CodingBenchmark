using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int W, H, D;
    static List<(int x, int y, int z, int w, int h, int d)> placed = new List<(int, int, int, int, int, int)>();
    static int cellSize = 10;
    static Dictionary<(int, int, int), List<int>> grid = new Dictionary<(int, int, int), List<int>>();

    static void Main()
    {
        string line = Console.ReadLine();
        if (string.IsNullOrEmpty(line)) { Console.WriteLine(0); return; }

        var tokens = line.Split(new[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(tokens[0]);
        W = int.Parse(tokens[1]);
        H = int.Parse(tokens[2]);
        D = int.Parse(tokens[3]);

        if (N == 0) { Console.WriteLine(0); return; }

        double avgVol = (double)W * H * D / N;
        cellSize = Math.Max(1, (int)Math.Ceiling(Math.Pow(avgVol, 1.0/3.0)));
        cellSize = Math.Min(cellSize, 100);

        var boxes = new List<(int idx, int w, int h, int d)>(N);
        for (int i = 0; i < N; i++)
        {
            var dims = Console.ReadLine().Split(new[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
            boxes.Add((i, int.Parse(dims[0]), int.Parse(dims[1]), int.Parse(dims[2])));
        }

        boxes.Sort((a, b) => ((long)b.w * b.h * b.d).CompareTo((long)a.w * a.h * a.d));

        var eps = new SortedSet<(int x, int y, int z)>(
            Comparer<(int x, int y, int z)>.Create((a, b) => {
                int c = a.y.CompareTo(b.y);
                if (c != 0) return c;
                c = a.x.CompareTo(b.x);
                if (c != 0) return c;
                return a.z.CompareTo(b.z);
            }));
        eps.Add((0, 0, 0));

        var results = new List<(int idx, int x, int y, int z)>();

        foreach (var box in boxes)
        {
            var rotations = GetRotations(box.w, box.h, box.d);

            (int x, int y, int z, int w, int h, int d)? placement = null;
            (int x, int y, int z) usedEp = default;

            foreach (var ep in eps)
            {
                if (placement.HasValue) break;

                foreach (var rot in rotations)
                {
                    if (CanPlace(ep.x, ep.y, ep.z, rot.w, rot.h, rot.d))
                    {
                        placement = (ep.x, ep.y, ep.z, rot.w, rot.h, rot.d);
                        usedEp = ep;
                        break;
                    }
                }
            }

            if (placement.HasValue)
            {
                var p = placement.Value;
                PlaceBox(p.x, p.y, p.z, p.w, p.h, p.d);
                results.Add((box.idx, p.x, p.y, p.z));

                eps.Remove(usedEp);

                if (p.x + p.w < W) eps.Add((p.x + p.w, p.y, p.z));
                if (p.y + p.h < H) eps.Add((p.x, p.y + p.h, p.z));
                if (p.z + p.d < D) eps.Add((p.x, p.y, p.z + p.d));

                while (eps.Count > 10000)
                    eps.Remove(eps.Max);
            }
        }

        Console.WriteLine(results.Count);
        foreach (var r in results)
            Console.WriteLine($"{r.idx} {r.x} {r.y} {r.z}");
    }

    static List<(int w, int h, int d)> GetRotations(int w, int h, int d)
    {
        var set = new HashSet<(int, int, int)>
        {
            (w, h, d), (w, d, h), (h, w, d), (h, d, w), (d, w, h), (d, h, w)
        };
        return set.OrderBy(r => r.Item2).ToList();
    }

    static bool CanPlace(int x, int y, int z, int w, int h, int d)
    {
        if (x < 0 || y < 0 || z < 0) return false;
        if (x + w > W || y + h > H || z + d > D) return false;

        if (placed.Count < 100)
        {
            foreach (var p in placed)
            {
                if (Overlaps(x, y, z, w, h, d, p.x, p.y, p.z, p.w, p.h, p.d))
                    return false;
            }
            return true;
        }

        int cx1 = x / cellSize, cx2 = (x + w - 1) / cellSize;
        int cy1 = y / cellSize, cy2 = (y + h - 1) / cellSize;
        int cz1 = z / cellSize, cz2 = (z + d - 1) / cellSize;

        long numCells = (long)(cx2 - cx1 + 1) * (cy2 - cy1 + 1) * (cz2 - cz1 + 1);
        if (numCells > 2000)
        {
            foreach (var p in placed)
            {
                if (Overlaps(x, y, z, w, h, d, p.x, p.y, p.z, p.w, p.h, p.d))
                    return false;
            }
            return true;
        }

        var seen = new HashSet<int>();
        for (int cx = cx1; cx <= cx2; cx++)
        {
            for (int cy = cy1; cy <= cy2; cy++)
            {
                for (int cz = cz1; cz <= cz2; cz++)
                {
                    if (grid.TryGetValue((cx, cy, cz), out var list))
                    {
                        foreach (int idx in list)
                        {
                            if (seen.Add(idx))
                            {
                                var p = placed[idx];
                                if (Overlaps(x, y, z, w, h, d, p.x, p.y, p.z, p.w, p.h, p.d))
                                    return false;
                            }
                        }
                    }
                }
            }
        }
        return true;
    }

    static bool Overlaps(int x1, int y1, int z1, int w1, int h1, int d1,
                         int x2, int y2, int z2, int w2, int h2, int d2)
    {
        return x1 < x2 + w2 && x2 < x1 + w1 &&
               y1 < y2 + h2 && y2 < y1 + h1 &&
               z1 < z2 + d2 && z2 < z1 + d1;
    }

    static void PlaceBox(int x, int y, int z, int w, int h, int d)
    {
        int idx = placed.Count;
        placed.Add((x, y, z, w, h, d));

        int cx1 = x / cellSize, cx2 = (x + w - 1) / cellSize;
        int cy1 = y / cellSize, cy2 = (y + h - 1) / cellSize;
        int cz1 = z / cellSize, cz2 = (z + d - 1) / cellSize;

        long numCells = (long)(cx2 - cx1 + 1) * (cy2 - cy1 + 1) * (cz2 - cz1 + 1);
        if (numCells > 2000) return;

        for (int cx = cx1; cx <= cx2; cx++)
        {
            for (int cy = cy1; cy <= cy2; cy++)
            {
                for (int cz = cz1; cz <= cz2; cz++)
                {
                    var key = (cx, cy, cz);
                    if (!grid.ContainsKey(key))
                        grid[key] = new List<int>();
                    grid[key].Add(idx);
                }
            }
        }
    }
}
