#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <cmath>
#include <thread>
#include <mutex>
#include <atomic>
#include <chrono>
#include <limits>

constexpr double PI = 3.14159265358979323846;

struct State {
    double x = 0, y = 0, z = 0;
    double vx = 0, vy = 0, vz = 0;
    double pitch = 0, yaw = 0, roll = 0;
    double pitch_rate = 0, yaw_rate = 0, roll_rate = 0;
    double fuel = 100;
    bool valid = false;
};

std::mutex g_mutex;
State g_state;
std::atomic<bool> g_running{true};

double g_gravity, g_max_thrust;
double g_target_x, g_target_y, g_target_z;
double g_width, g_depth;

inline double clamp_d(double v, double lo, double hi) {
    return v < lo ? lo : (v > hi ? hi : v);
}

inline double wrap_angle(double a) {
    while (a > PI) a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

void input_reader() {
    std::string line;
    while (g_running && std::getline(std::cin, line)) {
        if (line.empty()) continue;
        std::istringstream iss(line);
        State s;
        if (iss >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                >> s.pitch >> s.yaw >> s.roll
                >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel) {
            s.valid = true;
            std::lock_guard<std::mutex> lock(g_mutex);
            g_state = s;
        }
    }
    g_running = false;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    double height, fuel, start_x, start_y, start_z;
    std::cin >> g_width >> g_depth >> height >> g_gravity >> g_max_thrust >> fuel;
    std::cin >> start_x >> start_y >> start_z >> g_target_x >> g_target_y >> g_target_z;
    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    std::string terrain_path;
    std::getline(std::cin, terrain_path);

    std::string state_marker;
    std::getline(std::cin, state_marker);

    {
        std::lock_guard<std::mutex> lock(g_mutex);
        g_state.x = start_x;
        g_state.y = start_y;
        g_state.z = start_z;
        g_state.fuel = fuel;
        g_state.valid = true;
    }

    std::thread terrain_thread([&terrain_path]() {
        std::ifstream file(terrain_path, std::ios::binary);
    });
    terrain_thread.detach();

    std::thread reader(input_reader);

    double hover_thrust = g_gravity / g_max_thrust;
    double map_scale = std::min(g_width, g_depth);
    double max_cruise_speed = clamp_d(map_scale * 0.12, 12.0, 250.0);

    while (g_running) {
        State s;
        {
            std::lock_guard<std::mutex> lock(g_mutex);
            s = g_state;
        }

        double thrust = hover_thrust;
        double pitch_cmd = 0, yaw_cmd = 0, roll_cmd = 0;

        if (s.valid) {
            double dx = g_target_x - s.x;
            double dy = g_target_y - s.y;
            double dz = g_target_z - s.z;

            double horiz_dist = std::sqrt(dx*dx + dy*dy);
            double dist_3d = std::sqrt(dx*dx + dy*dy + dz*dz);
            double speed = std::sqrt(s.vx*s.vx + s.vy*s.vy + s.vz*s.vz);
            double horiz_speed = std::sqrt(s.vx*s.vx + s.vy*s.vy);

            double max_decel = g_max_thrust * 0.55;
            double stopping_dist = speed * speed / (2.0 * max_decel) * 2.2 + 20.0;

            double speed_limit = max_cruise_speed;
            if (dist_3d < stopping_dist) {
                speed_limit = std::sqrt(2.0 * max_decel * dist_3d * 0.35);
            }
            if (dist_3d < 50.0) speed_limit = std::min(speed_limit, 8.0);
            if (dist_3d < 15.0) speed_limit = std::min(speed_limit, 4.0);
            if (dist_3d < 5.0) speed_limit = std::min(speed_limit, 2.0);
            speed_limit = std::max(speed_limit, 0.5);

            double desired_vx = 0, desired_vy = 0, desired_vz = 0;

            if (horiz_dist > 2.0) {
                double h_scale = std::min(speed_limit / horiz_dist, 0.6);
                desired_vx = dx * h_scale;
                desired_vy = dy * h_scale;

                double safe_alt = g_target_z + std::min(50.0, horiz_dist * 0.3);
                if (s.z < safe_alt) {
                    desired_vz = std::min(5.0, (safe_alt - s.z) * 0.3);
                } else {
                    desired_vz = -std::min(2.0, (s.z - safe_alt) * 0.2);
                }
            } else {
                if (s.z > g_target_z + 1.0) {
                    double descent_rate = std::min(2.0, (s.z - g_target_z) * 0.2);
                    desired_vz = -clamp_d(descent_rate, 0.3, 2.0);
                } else if (s.z < g_target_z - 0.5) {
                    desired_vz = std::min(1.0, (g_target_z - s.z) * 0.3);
                } else {
                    desired_vz = -0.3;
                }

                desired_vx = dx * 0.3;
                desired_vy = dy * 0.3;
            }

            double k_vel = 1.3;
            double ax = (desired_vx - s.vx) * k_vel;
            double ay = (desired_vy - s.vy) * k_vel;
            double az = (desired_vz - s.vz) * k_vel + g_gravity;

            double horiz_accel = std::sqrt(ax*ax + ay*ay);
            double total_accel = std::sqrt(ax*ax + ay*ay + az*az);

            double pitch_desired = 0, yaw_desired = s.yaw;

            if (az > 0.01) {
                thrust = clamp_d(total_accel / g_max_thrust, 0.0, 1.0);

                if (horiz_accel > 0.01) {
                    yaw_desired = std::atan2(-ay, -ax);
                    pitch_desired = std::atan2(horiz_accel, az);
                }
                pitch_desired = clamp_d(pitch_desired, -1.3, 1.3);
            } else {
                thrust = 0.0;
                pitch_desired = 0.0;
            }

            double k_p = 2.8, k_d = 1.4;
            double pitch_err = wrap_angle(pitch_desired - s.pitch);
            double yaw_err = wrap_angle(yaw_desired - s.yaw);

            pitch_cmd = k_p * pitch_err - k_d * s.pitch_rate;
            yaw_cmd = k_p * yaw_err - k_d * s.yaw_rate;
            roll_cmd = -2.0 * s.roll_rate - 0.5 * s.roll;

            if (dist_3d < 3.0 && speed < 4.0) {
                if (s.vz < -2.0) {
                    thrust = clamp_d(thrust + 0.15, 0.0, 1.0);
                }
                pitch_cmd = clamp_d(-s.pitch * 4.0 - s.pitch_rate * 2.0, -1.0, 1.0);
                yaw_cmd *= 0.5;
            }

            thrust = clamp_d(thrust, 0.0, 1.0);
            pitch_cmd = clamp_d(pitch_cmd, -1.0, 1.0);
            yaw_cmd = clamp_d(yaw_cmd, -1.0, 1.0);
            roll_cmd = clamp_d(roll_cmd, -1.0, 1.0);
        }

        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();

        std::this_thread::sleep_for(std::chrono::milliseconds(40));
    }

    reader.join();
    return 0;
}
