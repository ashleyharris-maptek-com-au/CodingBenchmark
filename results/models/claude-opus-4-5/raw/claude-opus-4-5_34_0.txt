import sys
import math

def dist(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def route_distance(depot, customers, route):
    if not route:
        return 0
    d = dist(depot, customers[route[0]])
    for i in range(len(route) - 1):
        d += dist(customers[route[i]], customers[route[i+1]])
    d += dist(customers[route[-1]], depot)
    return d

def two_opt(depot, customers, route):
    if len(route) < 4:
        return route
    improved = True
    while improved:
        improved = False
        best_delta = 0
        best_i, best_j = -1, -1
        for i in range(len(route) - 1):
            for j in range(i + 2, len(route)):
                if i == 0 and j == len(route) - 1:
                    continue
                p1 = depot if i == 0 else customers[route[i-1]]
                p2 = customers[route[i]]
                p3 = customers[route[j]]
                p4 = depot if j == len(route) - 1 else customers[route[j+1]]
                old_cost = dist(p1, p2) + dist(p3, p4)
                new_cost = dist(p1, p3) + dist(p2, p4)
                delta = new_cost - old_cost
                if delta < best_delta:
                    best_delta = delta
                    best_i, best_j = i, j
        if best_delta < -1e-9:
            route = route[:best_i] + route[best_i:best_j+1][::-1] + route[best_j+1:]
            improved = True
    return route

def solve():
    lines = sys.stdin.read().strip().split('\n')
    n, k, capacity = map(int, lines[0].split())
    depot = tuple(map(float, lines[1].split()))

    customers = []
    demands = []
    for i in range(n):
        parts = list(map(float, lines[2 + i].split()))
        customers.append((parts[0], parts[1]))
        demands.append(parts[2])

    # Clarke-Wright Savings Algorithm
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            s = dist(depot, customers[i]) + dist(depot, customers[j]) - dist(customers[i], customers[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    # Each customer starts in own route
    route_id = list(range(n))
    routes = [[i] for i in range(n)]
    route_demand = demands[:]

    for s, i, j in savings:
        ri, rj = route_id[i], route_id[j]
        if ri == rj:
            continue
        if route_demand[ri] + route_demand[rj] > capacity:
            continue

        # Check if i and j are at endpoints of their routes
        r1, r2 = routes[ri], routes[rj]
        if not r1 or not r2:
            continue

        i_at_start = r1[0] == i
        i_at_end = r1[-1] == i
        j_at_start = r2[0] == j
        j_at_end = r2[-1] == j

        if not (i_at_start or i_at_end) or not (j_at_start or j_at_end):
            continue

        # Merge routes
        if i_at_end and j_at_start:
            merged = r1 + r2
        elif i_at_end and j_at_end:
            merged = r1 + r2[::-1]
        elif i_at_start and j_at_start:
            merged = r1[::-1] + r2
        else:
            merged = r2 + r1

        routes[ri] = merged
        route_demand[ri] += route_demand[rj]
        routes[rj] = []
        route_demand[rj] = 0
        for c in merged:
            route_id[c] = ri

    # Filter empty routes
    final_routes = [r for r in routes if r]

    # Apply 2-opt to each route
    final_routes = [two_opt(depot, customers, r) for r in final_routes]

    # Calculate total distance
    total = sum(route_distance(depot, customers, r) for r in final_routes)

    print(f"{total:.6f}")
    print(len(final_routes))
    for r in final_routes:
        print(' '.join(map(str, r)))

if __name__ == "__main__":
    solve()
