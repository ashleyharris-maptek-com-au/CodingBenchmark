use std::io::{self, BufRead, BufReader, BufWriter, Read, Write};

struct Rng(u64);

impl Rng {
    fn new(seed: u64) -> Self { Rng(seed) }

    fn next(&mut self) -> u64 {
        self.0 = self.0.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        self.0
    }

    fn usize(&mut self, max: usize) -> usize {
        if max == 0 { 0 } else { (self.next() as usize) % max }
    }

    fn f64(&mut self) -> f64 {
        (self.next() >> 11) as f64 / (1u64 << 53) as f64
    }
}

type Point = [f32; 3];

#[inline(always)]
fn read_point<R: Read>(r: &mut R) -> io::Result<Point> {
    let mut buf = [0u8; 12];
    r.read_exact(&mut buf)?;
    Ok([
        f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
        f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
        f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
    ])
}

#[inline(always)]
fn dist_sq(a: &Point, b: &Point) -> f32 {
    let (dx, dy, dz) = (a[0] - b[0], a[1] - b[1], a[2] - b[2]);
    dx * dx + dy * dy + dz * dz
}

#[inline(always)]
fn nearest(p: &Point, cs: &[Point]) -> usize {
    let mut best = 0;
    let mut best_d = f32::MAX;
    for (i, c) in cs.iter().enumerate() {
        let d = dist_sq(p, c);
        if d < best_d { best_d = d; best = i; }
    }
    best
}

fn kpp_init(pts: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    if pts.is_empty() || k == 0 { return vec![]; }
    let k = k.min(pts.len());

    let mut cs = vec![pts[rng.usize(pts.len())]];
    let mut d: Vec<f32> = pts.iter().map(|p| dist_sq(p, &cs[0])).collect();

    for _ in 1..k {
        let total: f64 = d.iter().map(|&x| x as f64).sum();
        if total <= 0.0 {
            cs.push(pts[rng.usize(pts.len())]);
        } else {
            let mut r = rng.f64() * total;
            let mut chosen = pts.len() - 1;
            for (i, &di) in d.iter().enumerate() {
                r -= di as f64;
                if r <= 0.0 { chosen = i; break; }
            }
            cs.push(pts[chosen]);
            let nc = cs.last().unwrap();
            for (i, p) in pts.iter().enumerate() {
                d[i] = d[i].min(dist_sq(p, nc));
            }
        }
    }
    cs
}

fn kmeans<R: Read, W: Write>(r: &mut R, w: &mut W, n: usize, k: usize, rng: &mut Rng) -> io::Result<()> {
    let mut pts = Vec::with_capacity(n);
    for _ in 0..n { pts.push(read_point(r)?); }

    let k = k.min(n);
    if k == 0 { return Ok(()); }
    if k >= n {
        for i in 0..n as u32 { w.write_all(&i.to_le_bytes())?; }
        return Ok(());
    }

    let mut cs = kpp_init(&pts, k, rng);
    let mut asgn = vec![0usize; n];

    let max_iter = if n > 10_000_000 { 15 } else { 30 };

    for _ in 0..max_iter {
        let mut changed = false;
        for (i, p) in pts.iter().enumerate() {
            let c = nearest(p, &cs);
            if c != asgn[i] { changed = true; asgn[i] = c; }
        }
        if !changed { break; }

        let mut sums = vec![[0.0f64; 3]; k];
        let mut cnts = vec![0usize; k];
        for (i, p) in pts.iter().enumerate() {
            let c = asgn[i];
            sums[c][0] += p[0] as f64;
            sums[c][1] += p[1] as f64;
            sums[c][2] += p[2] as f64;
            cnts[c] += 1;
        }
        for (i, s) in sums.iter().enumerate() {
            if cnts[i] > 0 {
                let cnt = cnts[i] as f64;
                cs[i] = [(s[0]/cnt) as f32, (s[1]/cnt) as f32, (s[2]/cnt) as f32];
            } else {
                cs[i] = pts[rng.usize(n)];
            }
        }
    }

    for &a in &asgn { w.write_all(&(a as u32).to_le_bytes())?; }
    Ok(())
}

fn minibatch<R: Read, W: Write>(r: &mut R, w: &mut W, n: usize, k: usize, rng: &mut Rng) -> io::Result<()> {
    let mut pts = Vec::with_capacity(n);
    for _ in 0..n { pts.push(read_point(r)?); }

    let k = k.min(n);
    if k == 0 { return Ok(()); }
    if k >= n {
        for i in 0..n as u32 { w.write_all(&i.to_le_bytes())?; }
        return Ok(());
    }

    let sample_size = (k * 50).min(500_000).min(n);
    let sample: Vec<Point> = (0..sample_size).map(|_| pts[rng.usize(n)]).collect();
    let mut cs = kpp_init(&sample, k, rng);

    let batch_size = 50_000.min(n);
    let epochs = 5.min(50_000_000 / n.max(1));

    for _ in 0..epochs {
        for batch_start in (0..n).step_by(batch_size) {
            let batch_end = (batch_start + batch_size).min(n);
            let mut sums = vec![[0.0f64; 3]; k];
            let mut cnts = vec![0usize; k];

            for i in batch_start..batch_end {
                let c = nearest(&pts[i], &cs);
                sums[c][0] += pts[i][0] as f64;
                sums[c][1] += pts[i][1] as f64;
                sums[c][2] += pts[i][2] as f64;
                cnts[c] += 1;
            }

            let lr = 1.0 / (1.0 + (batch_start / batch_size) as f64 * 0.1);
            for (i, s) in sums.iter().enumerate() {
                if cnts[i] > 0 {
                    let cnt = cnts[i] as f64;
                    let avg = [(s[0]/cnt) as f32, (s[1]/cnt) as f32, (s[2]/cnt) as f32];
                    cs[i][0] = (cs[i][0] as f64 * (1.0 - lr) + avg[0] as f64 * lr) as f32;
                    cs[i][1] = (cs[i][1] as f64 * (1.0 - lr) + avg[1] as f64 * lr) as f32;
                    cs[i][2] = (cs[i][2] as f64 * (1.0 - lr) + avg[2] as f64 * lr) as f32;
                }
            }
        }
    }

    for p in &pts { w.write_all(&(nearest(p, &cs) as u32).to_le_bytes())?; }
    Ok(())
}

fn streaming<R: Read, W: Write>(r: &mut R, w: &mut W, n: usize, k: usize, rng: &mut Rng) -> io::Result<()> {
    let k = k.min(n);
    if k == 0 { return Ok(()); }
    if k >= n {
        for i in 0..n { let _ = read_point(r)?; w.write_all(&(i as u32).to_le_bytes())?; }
        return Ok(());
    }

    let warmup = (k * 10).min(200_000).min(n);
    let mut sample = Vec::with_capacity(warmup);
    for _ in 0..warmup { sample.push(read_point(r)?); }

    let mut cs = kpp_init(&sample, k, rng);

    for _ in 0..5 {
        let mut sums = vec![[0.0f64; 3]; k];
        let mut cnts = vec![0usize; k];
        for p in &sample {
            let c = nearest(p, &cs);
            sums[c][0] += p[0] as f64;
            sums[c][1] += p[1] as f64;
            sums[c][2] += p[2] as f64;
            cnts[c] += 1;
        }
        for (i, s) in sums.iter().enumerate() {
            if cnts[i] > 0 {
                let cnt = cnts[i] as f64;
                cs[i] = [(s[0]/cnt) as f32, (s[1]/cnt) as f32, (s[2]/cnt) as f32];
            }
        }
    }

    let mut cnts = vec![0usize; k];
    for p in &sample { cnts[nearest(p, &cs)] += 1; }

    for p in &sample { w.write_all(&(nearest(p, &cs) as u32).to_le_bytes())?; }
    drop(sample);

    for _ in warmup..n {
        let p = read_point(r)?;
        let c = nearest(&p, &cs);
        w.write_all(&(c as u32).to_le_bytes())?;

        cnts[c] += 1;
        let alpha = 1.0 / cnts[c] as f32;
        cs[c][0] += alpha * (p[0] - cs[c][0]);
        cs[c][1] += alpha * (p[1] - cs[c][1]);
        cs[c][2] += alpha * (p[2] - cs[c][2]);
    }
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut r = BufReader::with_capacity(1 << 20, stdin.lock());
    let mut w = BufWriter::with_capacity(1 << 20, stdout.lock());

    let mut hdr = String::new();
    r.read_line(&mut hdr)?;
    let mut it = hdr.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let k: usize = it.next().unwrap().parse().unwrap();

    let mut rng = Rng::new(42);

    if n <= 10_000_000 {
        kmeans(&mut r, &mut w, n, k, &mut rng)?;
    } else if n <= 200_000_000 {
        minibatch(&mut r, &mut w, n, k, &mut rng)?;
    } else {
        streaming(&mut r, &mut w, n, k, &mut rng)?;
    }

    w.flush()
}
