#include <iostream>
#include <string>
#include <vector>
#include <unordered_map>
#include <unordered_set>
#include <cstdint>

using namespace std;

using ull = uint64_t;

constexpr ull BASE = 257;
constexpr ull MOD1 = 1000000007ULL;
constexpr ull MOD2 = 998244353ULL;

vector<ull> pow1, pow2;

void precomputePowers(size_t maxLen) {
    pow1.resize(maxLen + 1);
    pow2.resize(maxLen + 1);
    pow1[0] = pow2[0] = 1;
    for (size_t i = 1; i <= maxLen; i++) {
        pow1[i] = (pow1[i-1] * BASE) % MOD1;
        pow2[i] = (pow2[i-1] * BASE) % MOD2;
    }
}

struct PairHash {
    size_t operator()(const pair<ull, ull>& p) const {
        return static_cast<size_t>(p.first ^ (p.second << 1));
    }
};

using HashPair = pair<ull, ull>;
using HashSet = unordered_set<HashPair, PairHash>;
using HashMap = unordered_map<HashPair, vector<size_t>, PairHash>;

HashMap getHashesWithPos(const string& s, size_t len) {
    HashMap result;
    size_t n = s.size();
    if (len > n || len == 0) return result;

    ull h1 = 0, h2 = 0;
    for (size_t i = 0; i < len; i++) {
        h1 = (h1 * BASE + static_cast<unsigned char>(s[i])) % MOD1;
        h2 = (h2 * BASE + static_cast<unsigned char>(s[i])) % MOD2;
    }
    result[{h1, h2}].push_back(0);

    for (size_t i = len; i < n; i++) {
        ull c_out = static_cast<unsigned char>(s[i - len]);
        ull c_in = static_cast<unsigned char>(s[i]);
        h1 = ((h1 + MOD1 - c_out * pow1[len - 1] % MOD1) * BASE + c_in) % MOD1;
        h2 = ((h2 + MOD2 - c_out * pow2[len - 1] % MOD2) * BASE + c_in) % MOD2;
        result[{h1, h2}].push_back(i - len + 1);
    }

    return result;
}

HashSet getHashes(const string& s, size_t len) {
    HashSet result;
    size_t n = s.size();
    if (len > n || len == 0) return result;

    result.reserve(n - len + 1);

    ull h1 = 0, h2 = 0;
    for (size_t i = 0; i < len; i++) {
        h1 = (h1 * BASE + static_cast<unsigned char>(s[i])) % MOD1;
        h2 = (h2 * BASE + static_cast<unsigned char>(s[i])) % MOD2;
    }
    result.emplace(h1, h2);

    for (size_t i = len; i < n; i++) {
        ull c_out = static_cast<unsigned char>(s[i - len]);
        ull c_in = static_cast<unsigned char>(s[i]);
        h1 = ((h1 + MOD1 - c_out * pow1[len - 1] % MOD1) * BASE + c_in) % MOD1;
        h2 = ((h2 + MOD2 - c_out * pow2[len - 1] % MOD2) * BASE + c_in) % MOD2;
        result.emplace(h1, h2);
    }

    return result;
}

bool hasCommon(const vector<string>& strs, size_t len, string& result) {
    if (len == 0) {
        result = "";
        return true;
    }

    size_t shortestIdx = 0;
    for (size_t i = 1; i < strs.size(); i++) {
        if (strs[i].size() < strs[shortestIdx].size()) {
            shortestIdx = i;
        }
    }

    if (len > strs[shortestIdx].size()) {
        return false;
    }

    HashMap hashMap = getHashesWithPos(strs[shortestIdx], len);
    if (hashMap.empty()) return false;

    for (size_t i = 0; i < strs.size(); i++) {
        if (i == shortestIdx) continue;

        HashSet hashes = getHashes(strs[i], len);

        for (auto it = hashMap.begin(); it != hashMap.end(); ) {
            if (hashes.find(it->first) == hashes.end()) {
                it = hashMap.erase(it);
            } else {
                ++it;
            }
        }

        if (hashMap.empty()) return false;
    }

    for (const auto& [hash, positions] : hashMap) {
        for (size_t pos : positions) {
            string candidate = strs[shortestIdx].substr(pos, len);
            bool valid = true;
            for (size_t i = 0; i < strs.size(); i++) {
                if (strs[i].find(candidate) == string::npos) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                result = candidate;
                return true;
            }
        }
    }

    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    if (n == 0) {
        cout << endl;
        return 0;
    }

    vector<string> strs(n);
    size_t minLen = SIZE_MAX;
    for (int i = 0; i < n; i++) {
        getline(cin, strs[i]);
        if (!strs[i].empty() && strs[i].back() == '\r') {
            strs[i].pop_back();
        }
        if (strs[i].size() < minLen) {
            minLen = strs[i].size();
        }
    }

    if (n == 1) {
        cout << strs[0] << endl;
        return 0;
    }

    if (minLen == 0) {
        cout << endl;
        return 0;
    }

    precomputePowers(minLen);

    size_t lo = 0, hi = minLen;
    string answer = "";

    while (lo <= hi) {
        size_t mid = lo + (hi - lo) / 2;
        string candidate;
        if (hasCommon(strs, mid, candidate)) {
            answer = candidate;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    cout << answer << endl;
    return 0;
}
