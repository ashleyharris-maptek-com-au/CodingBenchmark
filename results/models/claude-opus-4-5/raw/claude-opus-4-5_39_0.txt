use std::io::{self, BufRead, Write, BufWriter};

fn solve(
    vertex: usize,
    n: usize,
    half: usize,
    count_a: usize,
    current_cut: usize,
    in_a: &mut Vec<bool>,
    adj: &[Vec<usize>],
    best_cut: &mut usize,
    best_partition: &mut Vec<usize>,
) {
    let count_b = vertex - count_a;
    let remaining = n - vertex;

    if count_a > half || count_b > half {
        return;
    }

    if half - count_a > remaining || half - count_b > remaining {
        return;
    }

    if current_cut >= *best_cut {
        return;
    }

    if vertex == n {
        *best_cut = current_cut;
        *best_partition = (0..n).filter(|&i| in_a[i]).collect();
        return;
    }

    let mut neighbors_in_a = 0;
    let mut neighbors_in_b = 0;
    for &neighbor in &adj[vertex] {
        if neighbor < vertex {
            if in_a[neighbor] {
                neighbors_in_a += 1;
            } else {
                neighbors_in_b += 1;
            }
        }
    }

    let try_a_first = neighbors_in_a >= neighbors_in_b;

    if try_a_first {
        if count_a < half {
            in_a[vertex] = true;
            solve(vertex + 1, n, half, count_a + 1, current_cut + neighbors_in_b,
                  in_a, adj, best_cut, best_partition);
        }
        if count_b < half {
            in_a[vertex] = false;
            solve(vertex + 1, n, half, count_a, current_cut + neighbors_in_a,
                  in_a, adj, best_cut, best_partition);
        }
    } else {
        if count_b < half {
            in_a[vertex] = false;
            solve(vertex + 1, n, half, count_a, current_cut + neighbors_in_a,
                  in_a, adj, best_cut, best_partition);
        }
        if count_a < half {
            in_a[vertex] = true;
            solve(vertex + 1, n, half, count_a + 1, current_cut + neighbors_in_b,
                  in_a, adj, best_cut, best_partition);
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let half = n / 2;
    let mut best_cut = usize::MAX;
    let mut best_partition: Vec<usize> = vec![];
    let mut in_a = vec![false; n];

    in_a[0] = true;
    solve(1, n, half, 1, 0, &mut in_a, &adj, &mut best_cut, &mut best_partition);

    writeln!(out, "{}", best_cut).unwrap();
    let partition_str: Vec<String> = best_partition.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", partition_str.join(" ")).unwrap();
}
