#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <chrono>
#include <random>
#include <tuple>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<int>> adj(n);
    vector<int> color(n, -1);
    vector<int> deg(n, 0);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        deg[u]++;
        deg[v]++;
    }

    auto getAvailColor = [&](int v) -> int {
        long long used = 0;
        for (int u : adj[v]) {
            if (color[u] >= 0 && color[u] < 64) {
                used |= (1LL << color[u]);
            }
        }
        for (int c = 0; c < k; c++) {
            if (!(used & (1LL << c))) return c;
        }
        return -1;
    };

    auto dsatur = [&]() -> bool {
        fill(color.begin(), color.end(), -1);
        vector<int> sat(n, 0);
        vector<long long> mask(n, 0);

        priority_queue<tuple<int,int,int>> pq;
        for (int i = 0; i < n; i++) {
            pq.push(make_tuple(0, deg[i], i));
        }

        int cnt = 0;
        while (!pq.empty() && cnt < n) {
            int s, d, v;
            tie(s, d, v) = pq.top();
            pq.pop();

            if (color[v] >= 0 || s != sat[v]) continue;

            int c = getAvailColor(v);
            if (c < 0) return false;

            color[v] = c;
            cnt++;

            for (int u : adj[v]) {
                if (color[u] < 0 && !(mask[u] & (1LL << c))) {
                    mask[u] |= (1LL << c);
                    sat[u]++;
                    pq.push(make_tuple(sat[u], deg[u], u));
                }
            }
        }
        return cnt == n;
    };

    auto greedyDeg = [&]() -> bool {
        fill(color.begin(), color.end(), -1);
        vector<int> ord(n);
        for (int i = 0; i < n; i++) ord[i] = i;
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            return deg[a] > deg[b];
        });

        for (int v : ord) {
            int c = getAvailColor(v);
            if (c < 0) return false;
            color[v] = c;
        }
        return true;
    };

    auto localSearch = [&]() -> bool {
        mt19937 rng(42);
        auto start = chrono::steady_clock::now();

        for (int v = 0; v < n; v++) {
            if (color[v] < 0) color[v] = rng() % k;
        }

        vector<int> conflictCount(n, 0);
        for (int v = 0; v < n; v++) {
            for (int u : adj[v]) {
                if (color[v] == color[u]) conflictCount[v]++;
            }
        }

        vector<int> conflicting;
        for (int v = 0; v < n; v++) {
            if (conflictCount[v] > 0) conflicting.push_back(v);
        }

        for (int iter = 0; iter < 50000000 && !conflicting.empty(); iter++) {
            int idx = rng() % conflicting.size();
            int cv = conflicting[idx];

            if (conflictCount[cv] == 0) {
                conflicting[idx] = conflicting.back();
                conflicting.pop_back();
                continue;
            }

            int oldC = color[cv];
            int bc = oldC, bconf = conflictCount[cv];

            vector<int> colorConf(k, 0);
            for (int u : adj[cv]) {
                colorConf[color[u]]++;
            }

            for (int c = 0; c < k; c++) {
                if (colorConf[c] < bconf) {
                    bconf = colorConf[c];
                    bc = c;
                }
            }

            if (bc != oldC) {
                for (int u : adj[cv]) {
                    if (color[u] == oldC) {
                        conflictCount[cv]--;
                        conflictCount[u]--;
                    }
                    if (color[u] == bc) {
                        conflictCount[cv]++;
                        conflictCount[u]++;
                        if (conflictCount[u] == 1) {
                            conflicting.push_back(u);
                        }
                    }
                }
                color[cv] = bc;

                if (conflictCount[cv] == 0) {
                    conflicting[idx] = conflicting.back();
                    conflicting.pop_back();
                }
            }

            if (iter % 500000 == 0) {
                auto elapsed = chrono::steady_clock::now() - start;
                if (chrono::duration_cast<chrono::seconds>(elapsed).count() > 270) {
                    break;
                }

                conflicting.clear();
                for (int v = 0; v < n; v++) {
                    if (conflictCount[v] > 0) conflicting.push_back(v);
                }
            }
        }

        for (int v = 0; v < n; v++) {
            for (int u : adj[v]) {
                if (color[v] == color[u]) return false;
            }
        }
        return true;
    };

    bool solved = dsatur() || greedyDeg() || localSearch();

    if (solved) {
        for (int i = 0; i < n; i++) {
            if (i > 0) cout << ' ';
            cout << color[i];
        }
        cout << '\n';
    }

    return 0;
}
