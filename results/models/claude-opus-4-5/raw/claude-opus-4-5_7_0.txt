#include <iostream>
#include <vector>
#include <array>
#include <cmath>
#include <algorithm>
#include <unordered_map>
#include <unordered_set>
#include <map>
#include <set>
#include <iomanip>
#include <functional>

using namespace std;

const double EPS = 1e-12;
const double MERGE_EPS = 1e-10;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& v) const { return {x+v.x, y+v.y, z+v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x-v.x, y-v.y, z-v.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    Vec3 operator/(double s) const { return {x/s, y/s, z/s}; }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return {y*v.z-z*v.y, z*v.x-x*v.z, x*v.y-y*v.x}; }
    double len2() const { return x*x + y*y + z*z; }
    double len() const { return sqrt(len2()); }
    Vec3 norm() const { double l=len(); return l>EPS ? *this/l : Vec3(1,0,0); }
    bool eq(const Vec3& v) const { return fabs(x-v.x)<MERGE_EPS && fabs(y-v.y)<MERGE_EPS && fabs(z-v.z)<MERGE_EPS; }
};

struct AABB {
    Vec3 lo{1e30,1e30,1e30}, hi{-1e30,-1e30,-1e30};
    void add(const Vec3& p) {
        lo.x=min(lo.x,p.x); lo.y=min(lo.y,p.y); lo.z=min(lo.z,p.z);
        hi.x=max(hi.x,p.x); hi.y=max(hi.y,p.y); hi.z=max(hi.z,p.z);
    }
    bool hit(const AABB& b) const {
        return lo.x<=b.hi.x+EPS && hi.x>=b.lo.x-EPS &&
               lo.y<=b.hi.y+EPS && hi.y>=b.lo.y-EPS &&
               lo.z<=b.hi.z+EPS && hi.z>=b.lo.z-EPS;
    }
    Vec3 mid() const { return (lo+hi)*0.5; }
};

struct BNode { AABB box; int L=-1,R=-1,s=0,n=0; bool leaf() const { return n>0; } };

class Mesh {
public:
    vector<Vec3> V;
    vector<array<int,3>> T;
    vector<BNode> bvh;
    vector<int> idx;
    AABB bbox;

    void build() {
        if (T.empty()) return;
        bbox = AABB();
        for (auto& v : V) bbox.add(v);
        idx.resize(T.size());
        for (size_t i = 0; i < T.size(); i++) idx[i] = (int)i;
        bvh.reserve(T.size()*2);
        mkNode(0, (int)T.size());
    }

    AABB tbox(int i) const {
        AABB b; b.add(V[T[i][0]]); b.add(V[T[i][1]]); b.add(V[T[i][2]]); return b;
    }

    int mkNode(int s, int e) {
        int id = (int)bvh.size();
        bvh.push_back({});
        for (int i = s; i < e; i++) bvh[id].box.add(tbox(idx[i]).lo), bvh[id].box.add(tbox(idx[i]).hi);
        if (e-s <= 4) { bvh[id].s=s; bvh[id].n=e-s; return id; }
        Vec3 ext = bvh[id].box.hi - bvh[id].box.lo;
        int ax = ext.y>ext.x ? (ext.z>ext.y?2:1) : (ext.z>ext.x?2:0);
        int m = (s+e)/2;
        nth_element(idx.begin()+s, idx.begin()+m, idx.begin()+e, [&](int a, int b) {
            Vec3 ca=tbox(a).mid(), cb=tbox(b).mid();
            return (ax==0?ca.x:ax==1?ca.y:ca.z) < (ax==0?cb.x:ax==1?cb.y:cb.z);
        });
        bvh[id].L = mkNode(s, m);
        bvh[id].R = mkNode(m, e);
        return id;
    }
};

bool rayTri(const Vec3& o, const Vec3& d, const Vec3& a, const Vec3& b, const Vec3& c, double& t) {
    Vec3 e1=b-a, e2=c-a, h=d.cross(e2);
    double det = e1.dot(h);
    if (fabs(det) < EPS) return false;
    double f = 1.0/det;
    Vec3 s = o - a;
    double u = f * s.dot(h);
    if (u < -EPS || u > 1+EPS) return false;
    Vec3 q = s.cross(e1);
    double v = f * d.dot(q);
    if (v < -EPS || u+v > 1+EPS) return false;
    t = f * e2.dot(q);
    return t > EPS;
}

int rayCnt(const Vec3& p, const Vec3& d, const Mesh& m, int ni) {
    if (ni < 0) return 0;
    auto& nd = m.bvh[ni];
    Vec3 inv(1.0/(fabs(d.x)>EPS?d.x:EPS), 1.0/(fabs(d.y)>EPS?d.y:EPS), 1.0/(fabs(d.z)>EPS?d.z:EPS));
    double tmin=-1e30, tmax=1e30;
    for (int i = 0; i < 3; i++) {
        double lo=i==0?nd.box.lo.x:i==1?nd.box.lo.y:nd.box.lo.z;
        double hi=i==0?nd.box.hi.x:i==1?nd.box.hi.y:nd.box.hi.z;
        double pp=i==0?p.x:i==1?p.y:p.z;
        double iv=i==0?inv.x:i==1?inv.y:inv.z;
        double t1=(lo-pp)*iv, t2=(hi-pp)*iv;
        if (t1>t2) swap(t1,t2);
        tmin=max(tmin,t1); tmax=min(tmax,t2);
    }
    if (tmax<-EPS || tmin>tmax+EPS) return 0;
    if (nd.leaf()) {
        int c = 0;
        for (int i = 0; i < nd.n; i++) {
            int fi = m.idx[nd.s+i];
            auto& tri = m.T[fi];
            double t;
            if (rayTri(p, d, m.V[tri[0]], m.V[tri[1]], m.V[tri[2]], t)) c++;
        }
        return c;
    }
    return rayCnt(p, d, m, nd.L) + rayCnt(p, d, m, nd.R);
}

bool inside(const Vec3& p, const Mesh& m) {
    if (m.bvh.empty()) return false;
    if (p.x<m.bbox.lo.x-EPS || p.x>m.bbox.hi.x+EPS ||
        p.y<m.bbox.lo.y-EPS || p.y>m.bbox.hi.y+EPS ||
        p.z<m.bbox.lo.z-EPS || p.z>m.bbox.hi.z+EPS) return false;
    Vec3 dirs[] = {{1,0.1371,0.0419}, {0.0321,1,0.0853}, {0.0617,0.0258,1}};
    int v = 0;
    for (auto& d : dirs) if (rayCnt(p, d.norm(), m, 0) % 2 == 1) v++;
    return v >= 2;
}

double planeDist(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 n = (b-a).cross(c-a);
    double l = n.len();
    if (l < EPS) return 0;
    return n.dot(p-a) / l;
}

bool segPlane(const Vec3& p1, const Vec3& p2, const Vec3& a, const Vec3& b, const Vec3& c, double& t, Vec3& pt) {
    double d1 = planeDist(p1, a, b, c);
    double d2 = planeDist(p2, a, b, c);
    if ((d1 > EPS && d2 > EPS) || (d1 < -EPS && d2 < -EPS)) return false;
    if (fabs(d1-d2) < EPS) return false;
    t = d1 / (d1 - d2);
    if (t < -EPS || t > 1+EPS) return false;
    pt = p1 + (p2-p1) * t;
    return true;
}

bool ptInTri(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 n = (b-a).cross(c-a);
    double area = n.len();
    if (area < EPS) return false;
    Vec3 n1 = (b-a).cross(p-a);
    Vec3 n2 = (c-b).cross(p-b);
    Vec3 n3 = (a-c).cross(p-c);
    double d1 = n.dot(n1), d2 = n.dot(n2), d3 = n.dot(n3);
    return d1 >= -EPS*area && d2 >= -EPS*area && d3 >= -EPS*area;
}

struct Seg { Vec3 a, b; };

bool triTriSeg(const Vec3& a0, const Vec3& a1, const Vec3& a2,
               const Vec3& b0, const Vec3& b1, const Vec3& b2, Seg& seg) {
    vector<Vec3> pts;
    Vec3 av[] = {a0,a1,a2}, bv[] = {b0,b1,b2};
    for (int i = 0; i < 3; i++) {
        double t; Vec3 pt;
        if (segPlane(av[i], av[(i+1)%3], b0, b1, b2, t, pt)) {
            if (ptInTri(pt, b0, b1, b2)) pts.push_back(pt);
        }
        if (segPlane(bv[i], bv[(i+1)%3], a0, a1, a2, t, pt)) {
            if (ptInTri(pt, a0, a1, a2)) pts.push_back(pt);
        }
    }
    if (pts.size() < 2) return false;
    seg.a = pts[0]; seg.b = pts[1];
    double maxD = (seg.b-seg.a).len2();
    for (size_t i = 0; i < pts.size(); i++) {
        for (size_t j = i+1; j < pts.size(); j++) {
            double d = (pts[i]-pts[j]).len2();
            if (d > maxD) { maxD = d; seg.a = pts[i]; seg.b = pts[j]; }
        }
    }
    return maxD > EPS*EPS;
}

void findAllSegs(const Mesh& A, const Mesh& B, map<int, vector<Seg>>& segsA, map<int, vector<Seg>>& segsB) {
    function<void(int, int)> check = [&](int ai, int bi) {
        if (ai < 0 || bi < 0) return;
        auto& na = A.bvh[ai], &nb = B.bvh[bi];
        if (!na.box.hit(nb.box)) return;
        if (na.leaf() && nb.leaf()) {
            for (int i = 0; i < na.n; i++) {
                int fi = A.idx[na.s+i];
                auto& ta = A.T[fi];
                for (int j = 0; j < nb.n; j++) {
                    int fj = B.idx[nb.s+j];
                    auto& tb = B.T[fj];
                    Seg seg;
                    if (triTriSeg(A.V[ta[0]], A.V[ta[1]], A.V[ta[2]],
                                  B.V[tb[0]], B.V[tb[1]], B.V[tb[2]], seg)) {
                        segsA[fi].push_back(seg);
                        segsB[fj].push_back(seg);
                    }
                }
            }
            return;
        }
        if (na.leaf()) { check(ai, B.bvh[bi].L); check(ai, B.bvh[bi].R); }
        else if (nb.leaf()) { check(A.bvh[ai].L, bi); check(A.bvh[ai].R, bi); }
        else {
            check(A.bvh[ai].L, B.bvh[bi].L);
            check(A.bvh[ai].L, B.bvh[bi].R);
            check(A.bvh[ai].R, B.bvh[bi].L);
            check(A.bvh[ai].R, B.bvh[bi].R);
        }
    };
    if (!A.bvh.empty() && !B.bvh.empty()) check(0, 0);
}

void clipTri(const Vec3& v0, const Vec3& v1, const Vec3& v2, const vector<Seg>& segs,
             const Mesh& other, vector<array<Vec3,3>>& out) {
    Vec3 n = (v1-v0).cross(v2-v0);
    double area = n.len();
    if (area < EPS) return;

    vector<Vec3> pts;
    pts.push_back(v0); pts.push_back(v1); pts.push_back(v2);
    for (auto& s : segs) {
        if (ptInTri(s.a, v0, v1, v2)) pts.push_back(s.a);
        if (ptInTri(s.b, v0, v1, v2)) pts.push_back(s.b);
    }

    vector<Vec3> unique;
    for (auto& p : pts) {
        bool found = false;
        for (auto& q : unique) if (p.eq(q)) { found = true; break; }
        if (!found) unique.push_back(p);
    }

    if (unique.size() <= 3) {
        Vec3 ctr = (v0+v1+v2)/3.0;
        Vec3 test = ctr + n.norm() * EPS * 1000;
        if (!inside(test, other)) out.push_back({v0, v1, v2});
        return;
    }

    int axis = 0;
    if (fabs(n.y) > fabs(n.x)) axis = 1;
    if (fabs(n.z) > fabs(axis==0?n.x:n.y)) axis = 2;

    auto proj = [axis](const Vec3& p) -> pair<double,double> {
        if (axis == 0) return {p.y, p.z};
        if (axis == 1) return {p.x, p.z};
        return {p.x, p.y};
    };

    Vec3 ctr(0,0,0);
    for (auto& p : unique) ctr = ctr + p;
    ctr = ctr / (double)unique.size();

    auto [cx, cy] = proj(ctr);
    sort(unique.begin(), unique.end(), [&](const Vec3& a, const Vec3& b) {
        auto [ax, ay] = proj(a);
        auto [bx, by] = proj(b);
        return atan2(ay-cy, ax-cx) < atan2(by-cy, bx-cx);
    });

    for (size_t i = 1; i + 1 < unique.size(); i++) {
        Vec3 a = unique[0], b = unique[i], c = unique[i+1];
        Vec3 tn = (b-a).cross(c-a);
        if (tn.len2() < EPS*EPS) continue;
        if (tn.dot(n) < 0) swap(b, c);
        Vec3 tctr = (a+b+c)/3.0;
        Vec3 test = tctr + n.norm() * EPS * 1000;
        if (!inside(test, other)) out.push_back({a, b, c});
    }
}

struct VHash {
    size_t operator()(const Vec3& v) const {
        auto h=[](double d){return hash<long long>()((long long)(d/MERGE_EPS+(d>=0?0.5:-0.5)));};
        return h(v.x) ^ (h(v.y)*31) ^ (h(v.z)*997);
    }
};
struct VEq { bool operator()(const Vec3& a, const Vec3& b) const { return a.eq(b); } };

class CSG {
    vector<Vec3> outV;
    vector<array<int,3>> outT;
    unordered_map<Vec3, int, VHash, VEq> vmap;
public:
    int addV(const Vec3& v) {
        auto it = vmap.find(v);
        if (it != vmap.end()) return it->second;
        int id = (int)outV.size();
        outV.push_back(v);
        vmap[v] = id;
        return id;
    }

    void addT(const Vec3& a, const Vec3& b, const Vec3& c) {
        Vec3 n = (b-a).cross(c-a);
        if (n.len2() < EPS*EPS) return;
        int i0=addV(a), i1=addV(b), i2=addV(c);
        if (i0==i1 || i1==i2 || i2==i0) return;
        outT.push_back({i0,i1,i2});
    }

    void processMesh(const Mesh& src, const Mesh& other, const map<int, vector<Seg>>& fsegs) {
        for (int fi = 0; fi < (int)src.T.size(); fi++) {
            auto& t = src.T[fi];
            Vec3 v0 = src.V[t[0]], v1 = src.V[t[1]], v2 = src.V[t[2]];

            auto it = fsegs.find(fi);
            if (it != fsegs.end() && !it->second.empty()) {
                vector<array<Vec3,3>> sub;
                clipTri(v0, v1, v2, it->second, other, sub);
                for (auto& tri : sub) addT(tri[0], tri[1], tri[2]);
            } else {
                Vec3 ctr = (v0+v1+v2)/3.0;
                Vec3 n = (v1-v0).cross(v2-v0).norm();
                Vec3 test = ctr + n * EPS * 1000;
                if (!inside(test, other)) addT(v0, v1, v2);
            }
        }
    }

    void compute(Mesh& A, Mesh& B) {
        A.build();
        B.build();

        map<int, vector<Seg>> segsA, segsB;
        findAllSegs(A, B, segsA, segsB);

        if (segsA.empty() && segsB.empty()) {
            Vec3 sA = (A.V[A.T[0][0]] + A.V[A.T[0][1]] + A.V[A.T[0][2]]) / 3.0;
            Vec3 sB = (B.V[B.T[0][0]] + B.V[B.T[0][1]] + B.V[B.T[0][2]]) / 3.0;
            bool aInB = inside(sA, B);
            bool bInA = inside(sB, A);
            if (aInB) {
                for (auto& t : B.T) addT(B.V[t[0]], B.V[t[1]], B.V[t[2]]);
            } else if (bInA) {
                for (auto& t : A.T) addT(A.V[t[0]], A.V[t[1]], A.V[t[2]]);
            } else {
                for (auto& t : A.T) addT(A.V[t[0]], A.V[t[1]], A.V[t[2]]);
                for (auto& t : B.T) addT(B.V[t[0]], B.V[t[1]], B.V[t[2]]);
            }
            return;
        }

        processMesh(A, B, segsA);
        processMesh(B, A, segsB);
    }

    void output() {
        cout << outV.size() << " " << outT.size() << "\n";
        for (auto& v : outV) cout << v.x << " " << v.y << " " << v.z << "\n";
        for (auto& t : outT) cout << t[0] << " " << t[1] << " " << t[2] << "\n";
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout << fixed << setprecision(9);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh A, B;
    A.V.resize(VA); A.T.resize(FA);
    B.V.resize(VB); B.T.resize(FB);

    for (int i = 0; i < VA; i++) cin >> A.V[i].x >> A.V[i].y >> A.V[i].z;
    for (int i = 0; i < FA; i++) cin >> A.T[i][0] >> A.T[i][1] >> A.T[i][2];
    for (int i = 0; i < VB; i++) cin >> B.V[i].x >> B.V[i].y >> B.V[i].z;
    for (int i = 0; i < FB; i++) cin >> B.T[i][0] >> B.T[i][1] >> B.T[i][2];

    CSG csg;
    csg.compute(A, B);
    csg.output();

    return 0;
}
