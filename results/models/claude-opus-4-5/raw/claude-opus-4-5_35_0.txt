use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut input = String::new();
    stdin.lock().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    // Greedy algorithm with lazy heap updates
    // Select vertex that covers most uncovered vertices
    let mut dominated = vec![false; n];
    let mut result: Vec<usize> = Vec::new();
    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new();

    for v in 0..n {
        heap.push((1 + adj[v].len(), v));
    }

    let mut remaining = n;

    while remaining > 0 && !heap.is_empty() {
        let (old_cov, v) = heap.pop().unwrap();

        // Calculate actual current coverage
        let mut cov = if dominated[v] { 0 } else { 1 };
        for &u in &adj[v] {
            if !dominated[u] {
                cov += 1;
            }
        }

        if cov == 0 {
            continue;
        }

        if cov < old_cov {
            // Coverage decreased, re-insert with updated value
            heap.push((cov, v));
            continue;
        }

        // Add v to dominating set
        result.push(v);

        if !dominated[v] {
            dominated[v] = true;
            remaining -= 1;
        }
        for &u in &adj[v] {
            if !dominated[u] {
                dominated[u] = true;
                remaining -= 1;
            }
        }
    }

    // Cleanup phase: try to remove redundant vertices
    let mut in_ds = vec![false; n];
    for &v in &result {
        in_ds[v] = true;
    }

    // dom_count[v] = number of DS vertices dominating v
    let mut dom_count = vec![0usize; n];
    for v in 0..n {
        if in_ds[v] {
            dom_count[v] += 1;
            for &u in &adj[v] {
                dom_count[u] += 1;
            }
        }
    }

    // Sort by degree descending to try removing high-degree vertices first
    result.sort_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));

    let mut final_result: Vec<usize> = Vec::new();

    for &v in &result {
        // Check if v can be removed from DS
        // v must still be covered (has another dominator)
        let mut can_remove = dom_count[v] >= 2;

        // All non-DS neighbors must still be covered
        if can_remove {
            for &u in &adj[v] {
                if !in_ds[u] && dom_count[u] < 2 {
                    can_remove = false;
                    break;
                }
            }
        }

        if can_remove {
            in_ds[v] = false;
            dom_count[v] -= 1;
            for &u in &adj[v] {
                dom_count[u] -= 1;
            }
        } else {
            final_result.push(v);
        }
    }

    // Output
    writeln!(out, "{}", final_result.len()).unwrap();
    if final_result.is_empty() {
        writeln!(out, "").unwrap();
    } else {
        let s: Vec<String> = final_result.iter().map(|x| x.to_string()).collect();
        writeln!(out, "{}", s.join(" ")).unwrap();
    }
}
