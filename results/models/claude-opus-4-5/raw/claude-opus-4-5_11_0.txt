use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<(f64, f64)>,
}

impl Polygon {
    fn new(vertices: Vec<(f64, f64)>) -> Self {
        Polygon { vertices }
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        for &(x, y) in &self.vertices {
            min_x = min_x.min(x);
            min_y = min_y.min(y);
            max_x = max_x.max(x);
            max_y = max_y.max(y);
        }
        (min_x, min_y, max_x, max_y)
    }

    fn centroid(&self) -> (f64, f64) {
        let n = self.vertices.len() as f64;
        let sum: (f64, f64) = self.vertices.iter().fold((0.0, 0.0), |acc, &(x, y)| (acc.0 + x, acc.1 + y));
        (sum.0 / n, sum.1 / n)
    }

    fn area(&self) -> f64 {
        let n = self.vertices.len();
        let mut area = 0.0;
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].0 * self.vertices[j].1;
            area -= self.vertices[j].0 * self.vertices[i].1;
        }
        area.abs() / 2.0
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices: Vec<(f64, f64)> = self.vertices.iter()
            .map(|&(x, y)| (x * cos_a - y * sin_a, x * sin_a + y * cos_a))
            .collect();
        Polygon::new(vertices)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let vertices: Vec<(f64, f64)> = self.vertices.iter()
            .map(|&(x, y)| (x + dx, y + dy))
            .collect();
        Polygon::new(vertices)
    }

    fn normalize_to_origin(&self) -> (Polygon, f64, f64) {
        let (min_x, min_y, _, _) = self.bounding_box();
        (self.translate(-min_x, -min_y), min_x, min_y)
    }
}

fn point_in_polygon(px: f64, py: f64, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly.vertices[i];
        let (xj, yj) = poly.vertices[j];
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn segments_intersect(p1: (f64, f64), p2: (f64, f64), p3: (f64, f64), p4: (f64, f64)) -> bool {
    let d1 = direction(p3, p4, p1);
    let d2 = direction(p3, p4, p2);
    let d3 = direction(p1, p2, p3);
    let d4 = direction(p1, p2, p4);

    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }

    let eps = 1e-10;
    if d1.abs() < eps && on_segment(p3, p4, p1) { return true; }
    if d2.abs() < eps && on_segment(p3, p4, p2) { return true; }
    if d3.abs() < eps && on_segment(p1, p2, p3) { return true; }
    if d4.abs() < eps && on_segment(p1, p2, p4) { return true; }

    false
}

fn direction(pi: (f64, f64), pj: (f64, f64), pk: (f64, f64)) -> f64 {
    (pk.0 - pi.0) * (pj.1 - pi.1) - (pj.0 - pi.0) * (pk.1 - pi.1)
}

fn on_segment(pi: (f64, f64), pj: (f64, f64), pk: (f64, f64)) -> bool {
    pk.0.min(pi.0.max(pj.0)) <= pk.0.max(pi.0.min(pj.0)) + 1e-10 &&
    pk.1.min(pi.1.max(pj.1)) <= pk.1.max(pi.1.min(pj.1)) + 1e-10
}

fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    let (min1x, min1y, max1x, max1y) = poly1.bounding_box();
    let (min2x, min2y, max2x, max2y) = poly2.bounding_box();

    let eps = 1e-9;
    if max1x < min2x - eps || max2x < min1x - eps || max1y < min2y - eps || max2y < min1y - eps {
        return false;
    }

    let n1 = poly1.vertices.len();
    let n2 = poly2.vertices.len();

    for i in 0..n1 {
        let p1 = poly1.vertices[i];
        let p2 = poly1.vertices[(i + 1) % n1];
        for j in 0..n2 {
            let p3 = poly2.vertices[j];
            let p4 = poly2.vertices[(j + 1) % n2];
            if segments_intersect(p1, p2, p3, p4) {
                return true;
            }
        }
    }

    let c1 = poly1.centroid();
    if point_in_polygon(c1.0, c1.1, poly2) {
        return true;
    }
    let c2 = poly2.centroid();
    if point_in_polygon(c2.0, c2.1, poly1) {
        return true;
    }

    if !poly1.vertices.is_empty() && point_in_polygon(poly1.vertices[0].0, poly1.vertices[0].1, poly2) {
        return true;
    }
    if !poly2.vertices.is_empty() && point_in_polygon(poly2.vertices[0].0, poly2.vertices[0].1, poly1) {
        return true;
    }

    false
}

fn polygon_in_stock(piece: &Polygon, stock: &Polygon) -> bool {
    let (pmin_x, pmin_y, pmax_x, pmax_y) = piece.bounding_box();
    let (smin_x, smin_y, smax_x, smax_y) = stock.bounding_box();

    let eps = 1e-9;
    if pmin_x < smin_x - eps || pmax_x > smax_x + eps || pmin_y < smin_y - eps || pmax_y > smax_y + eps {
        return false;
    }

    for &(x, y) in &piece.vertices {
        if !point_in_polygon(x, y, stock) {
            let on_boundary = is_on_boundary(x, y, stock);
            if !on_boundary {
                return false;
            }
        }
    }

    true
}

fn is_on_boundary(px: f64, py: f64, poly: &Polygon) -> bool {
    let n = poly.vertices.len();
    let eps = 1e-7;
    for i in 0..n {
        let (x1, y1) = poly.vertices[i];
        let (x2, y2) = poly.vertices[(i + 1) % n];
        let d = point_to_segment_dist(px, py, x1, y1, x2, y2);
        if d < eps {
            return true;
        }
    }
    false
}

fn point_to_segment_dist(px: f64, py: f64, x1: f64, y1: f64, x2: f64, y2: f64) -> f64 {
    let dx = x2 - x1;
    let dy = y2 - y1;
    let len_sq = dx * dx + dy * dy;
    if len_sq < 1e-14 {
        return ((px - x1).powi(2) + (py - y1).powi(2)).sqrt();
    }
    let t = ((px - x1) * dx + (py - y1) * dy) / len_sq;
    let t = t.max(0.0).min(1.0);
    let proj_x = x1 + t * dx;
    let proj_y = y1 + t * dy;
    ((px - proj_x).powi(2) + (py - proj_y).powi(2)).sqrt()
}

struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

struct Stock {
    polygon: Polygon,
    placed_pieces: Vec<Polygon>,
}

impl Stock {
    fn new(polygon: Polygon) -> Self {
        Stock { polygon, placed_pieces: Vec::new() }
    }

    fn can_place(&self, piece: &Polygon) -> bool {
        if !polygon_in_stock(piece, &self.polygon) {
            return false;
        }
        for placed in &self.placed_pieces {
            if polygons_overlap(piece, placed) {
                return false;
            }
        }
        true
    }

    fn place(&mut self, piece: Polygon) {
        self.placed_pieces.push(piece);
    }
}

fn try_place_piece(piece: &Polygon, stock: &Stock, stock_bbox: (f64, f64, f64, f64), num_pieces: usize) -> Option<(f64, f64, f64)> {
    let (smin_x, smin_y, smax_x, smax_y) = stock_bbox;
    let stock_w = smax_x - smin_x;
    let stock_h = smax_y - smin_y;

    let (num_rotations, grid_step) = if num_pieces > 2000 {
        (4, 0.08)
    } else if num_pieces > 500 {
        (8, 0.05)
    } else if num_pieces > 100 {
        (12, 0.03)
    } else {
        (24, 0.02)
    };

    for rot_idx in 0..num_rotations {
        let angle = (rot_idx as f64) * PI / (num_rotations as f64 / 2.0);
        let rotated = piece.rotate(angle);
        let (normalized, _, _) = rotated.normalize_to_origin();
        let (_, _, pw, ph) = normalized.bounding_box();

        if pw > stock_w + 1e-9 || ph > stock_h + 1e-9 {
            continue;
        }

        let step_x = (stock_w * grid_step).max(1.0);
        let step_y = (stock_h * grid_step).max(1.0);

        let mut y = smin_y;
        while y <= smax_y - ph + 1e-9 {
            let mut x = smin_x;
            while x <= smax_x - pw + 1e-9 {
                let placed = normalized.translate(x, y);
                if stock.can_place(&placed) {
                    return Some((x, y, angle));
                }
                x += step_x;
            }
            y += step_y;
        }
    }

    None
}

fn solve(stock_poly: Polygon, pieces: Vec<Polygon>) -> Vec<Placement> {
    let num_pieces = pieces.len();
    let mut placements = vec![None; num_pieces];

    let mut indexed_pieces: Vec<(usize, f64, Polygon)> = pieces.into_iter()
        .enumerate()
        .map(|(i, p)| (i, p.area(), p))
        .collect();
    indexed_pieces.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());

    let mut stocks: Vec<Stock> = Vec::new();
    let stock_bbox = stock_poly.bounding_box();

    for (original_idx, _, piece) in indexed_pieces {
        let mut placed = false;

        for (stock_idx, stock) in stocks.iter_mut().enumerate() {
            if let Some((x, y, rotation)) = try_place_piece(&piece, stock, stock_bbox, num_pieces) {
                let rotated = piece.rotate(rotation);
                let (normalized, _, _) = rotated.normalize_to_origin();
                let final_piece = normalized.translate(x, y);
                stock.place(final_piece);
                placements[original_idx] = Some(Placement { stock_index: stock_idx, x, y, rotation });
                placed = true;
                break;
            }
        }

        if !placed {
            let mut new_stock = Stock::new(stock_poly.clone());
            if let Some((x, y, rotation)) = try_place_piece(&piece, &new_stock, stock_bbox, num_pieces) {
                let rotated = piece.rotate(rotation);
                let (normalized, _, _) = rotated.normalize_to_origin();
                let final_piece = normalized.translate(x, y);
                new_stock.place(final_piece);
                let stock_idx = stocks.len();
                stocks.push(new_stock);
                placements[original_idx] = Some(Placement { stock_index: stock_idx, x, y, rotation });
            } else {
                let (smin_x, smin_y, _, _) = stock_bbox;
                let stock_idx = stocks.len();
                stocks.push(Stock::new(stock_poly.clone()));
                placements[original_idx] = Some(Placement { stock_index: stock_idx, x: smin_x, y: smin_y, rotation: 0.0 });
            }
        }
    }

    placements.into_iter().map(|p| p.unwrap()).collect()
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let sv: usize = parts.next().unwrap().parse().unwrap();
    let np: usize = parts.next().unwrap().parse().unwrap();

    let mut stock_vertices = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let mut coords = line.split_whitespace();
        let x: f64 = coords.next().unwrap().parse().unwrap();
        let y: f64 = coords.next().unwrap().parse().unwrap();
        stock_vertices.push((x, y));
    }
    let stock_poly = Polygon::new(stock_vertices);

    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        let mut piece_vertices = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let mut coords = line.split_whitespace();
            let x: f64 = coords.next().unwrap().parse().unwrap();
            let y: f64 = coords.next().unwrap().parse().unwrap();
            piece_vertices.push((x, y));
        }
        pieces.push(Polygon::new(piece_vertices));
    }

    let placements = solve(stock_poly, pieces);

    let num_stocks = placements.iter().map(|p| p.stock_index).max().map(|m| m + 1).unwrap_or(0);

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", num_stocks).unwrap();
    for p in &placements {
        writeln!(out, "{} {} {} {}", p.stock_index, p.x, p.y, p.rotation).unwrap();
    }
}
