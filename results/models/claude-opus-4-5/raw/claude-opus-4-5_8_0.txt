use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut stdin = stdin.lock();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut line = String::new();
    stdin.read_line(&mut line).unwrap();
    let mut dims = line.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();

    let mut maze: Vec<u8> = vec![b'#'; h * w];
    let mut start: u32 = 0;
    let mut end: u32 = 0;

    for y in 0..h {
        line.clear();
        stdin.read_line(&mut line).unwrap();
        let bytes = line.as_bytes();
        for x in 0..w.min(bytes.len()) {
            let c = bytes[x];
            let idx = y * w + x;
            maze[idx] = c;
            if c == b'A' { start = idx as u32; }
            if c == b'B' { end = idx as u32; }
        }
    }

    let mut parent: Vec<u8> = vec![0; h * w];
    let mut queue: VecDeque<u32> = VecDeque::new();

    queue.push_back(start);
    parent[start as usize] = 5;

    let (w32, h32) = (w as u32, h as u32);

    while let Some(idx) = queue.pop_front() {
        if idx == end { break; }

        let x = idx % w32;
        let y = idx / w32;

        if x + 1 < w32 {
            let ni = (idx + 1) as usize;
            if parent[ni] == 0 && maze[ni] != b'#' {
                parent[ni] = 1;
                queue.push_back(idx + 1);
            }
        }
        if x > 0 {
            let ni = (idx - 1) as usize;
            if parent[ni] == 0 && maze[ni] != b'#' {
                parent[ni] = 2;
                queue.push_back(idx - 1);
            }
        }
        if y + 1 < h32 {
            let ni = (idx + w32) as usize;
            if parent[ni] == 0 && maze[ni] != b'#' {
                parent[ni] = 3;
                queue.push_back(idx + w32);
            }
        }
        if y > 0 {
            let ni = (idx - w32) as usize;
            if parent[ni] == 0 && maze[ni] != b'#' {
                parent[ni] = 4;
                queue.push_back(idx - w32);
            }
        }
    }

    if parent[end as usize] == 0 && start != end {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path: Vec<u32> = Vec::new();
    let mut curr = end;
    loop {
        path.push(curr);
        if curr == start { break; }
        curr = match parent[curr as usize] {
            1 => curr - 1,
            2 => curr + 1,
            3 => curr - w32,
            4 => curr + w32,
            _ => unreachable!(),
        };
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for idx in path {
        writeln!(out, "{} {}", idx % w32, idx / w32).unwrap();
    }
}
