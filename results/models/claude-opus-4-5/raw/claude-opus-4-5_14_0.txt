using System;
using System.Collections.Generic;

class Program
{
    static int H, W;
    static char[,] board;
    static bool[,] safe, mine;
    static readonly int[] dx = {-1,-1,-1,0,0,1,1,1};
    static readonly int[] dy = {-1,0,1,-1,1,-1,0,1};

    public static void Main()
    {
        string[] p = Console.ReadLine().Split();
        H = int.Parse(p[0]); W = int.Parse(p[1]);
        board = new char[H, W];
        safe = new bool[H, W];
        mine = new bool[H, W];

        for (int y = 0; y < H; y++)
        {
            string ln = Console.ReadLine() ?? "";
            for (int x = 0; x < W; x++)
            {
                board[y, x] = x < ln.Length ? ln[x] : ' ';
                if (board[y, x] == '*') mine[y, x] = true;
            }
        }

        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int y = 0; y < H; y++)
                for (int x = 0; x < W; x++)
                    if (board[y, x] >= '0' && board[y, x] <= '8')
                        if (ProcessCell(x, y)) changed = true;
        }

        SolveCSP();

        var res = new List<string>();
        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
                if (board[y, x] == ' ' && safe[y, x] && !mine[y, x])
                    res.Add($"{x} {y}");

        Console.WriteLine(res.Count);
        foreach (var s in res) Console.WriteLine(s);
    }

    static bool ProcessCell(int x, int y)
    {
        int need = board[y, x] - '0';
        int mines = 0;
        var unk = new List<(int, int)>();

        for (int d = 0; d < 8; d++)
        {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                if (mine[ny, nx]) mines++;
                else if (board[ny, nx] == ' ' && !safe[ny, nx]) unk.Add((nx, ny));
            }
        }

        bool ch = false;
        if (mines == need)
            foreach (var (ux, uy) in unk)
                if (!safe[uy, ux]) { safe[uy, ux] = true; ch = true; }
        else if (mines + unk.Count == need)
            foreach (var (ux, uy) in unk)
                if (!mine[uy, ux]) { mine[uy, ux] = true; ch = true; }
        return ch;
    }

    static void SolveCSP()
    {
        var frontier = new HashSet<(int, int)>();
        var constraints = new List<(HashSet<(int, int)>, int)>();

        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
                if (board[y, x] >= '0' && board[y, x] <= '8')
                {
                    int need = board[y, x] - '0', mines = 0;
                    var unk = new HashSet<(int, int)>();

                    for (int d = 0; d < 8; d++)
                    {
                        int nx = x + dx[d], ny = y + dy[d];
                        if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                        {
                            if (mine[ny, nx]) mines++;
                            else if (board[ny, nx] == ' ' && !safe[ny, nx]) unk.Add((nx, ny));
                        }
                    }

                    int rem = need - mines;
                    if (unk.Count > 0 && rem >= 0 && rem <= unk.Count)
                    {
                        constraints.Add((unk, rem));
                        foreach (var u in unk) frontier.Add(u);
                    }
                }

        if (frontier.Count == 0 || frontier.Count > 500) return;

        var c2c = new Dictionary<(int, int), List<int>>();
        for (int i = 0; i < constraints.Count; i++)
            foreach (var c in constraints[i].Item1)
            {
                if (!c2c.ContainsKey(c)) c2c[c] = new List<int>();
                c2c[c].Add(i);
            }

        var visited = new HashSet<(int, int)>();
        foreach (var start in frontier)
        {
            if (visited.Contains(start)) continue;

            var cells = new List<(int, int)>();
            var cons = new List<(HashSet<(int, int)>, int)>();
            var used = new HashSet<int>();
            var q = new Queue<(int, int)>();

            q.Enqueue(start); visited.Add(start);
            while (q.Count > 0)
            {
                var cell = q.Dequeue();
                cells.Add(cell);
                if (!c2c.TryGetValue(cell, out var cis)) continue;
                foreach (int ci in cis)
                    if (used.Add(ci))
                    {
                        cons.Add(constraints[ci]);
                        foreach (var o in constraints[ci].Item1)
                            if (visited.Add(o)) q.Enqueue(o);
                    }
            }

            if (cells.Count <= 22)
                SolveComponent(cells, cons);
        }
    }

    static void SolveComponent(List<(int, int)> cells, List<(HashSet<(int, int)>, int)> constraints)
    {
        int n = cells.Count;
        var idx = new Dictionary<(int, int), int>();
        for (int i = 0; i < n; i++) idx[cells[i]] = i;

        var bc = new List<(int, int)>();
        foreach (var (cs, cnt) in constraints)
        {
            int m = 0;
            foreach (var c in cs)
                if (idx.TryGetValue(c, out int j)) m |= 1 << j;
            bc.Add((m, cnt));
        }

        int alwaysSafe = (1 << n) - 1;
        int validCount = 0;

        for (int s = 0; s < (1 << n); s++)
        {
            bool ok = true;
            foreach (var (m, cnt) in bc)
                if (BitCount(s & m) != cnt) { ok = false; break; }
            if (ok) { validCount++; alwaysSafe &= ~s; }
        }

        if (validCount > 0)
            for (int i = 0; i < n; i++)
                if ((alwaysSafe & (1 << i)) != 0)
                    safe[cells[i].Item2, cells[i].Item1] = true;
    }

    static int BitCount(int x)
    {
        x -= (x >> 1) & 0x55555555;
        x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
        return (((x + (x >> 4)) & 0x0F0F0F0F) * 0x01010101) >> 24;
    }
}
