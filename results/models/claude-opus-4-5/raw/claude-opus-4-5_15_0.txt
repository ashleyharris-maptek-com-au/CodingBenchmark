use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut polygon: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let mut it = line.trim().split_whitespace();
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        polygon.push((x, y));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let sun: Vec<f64> = sun_line.trim().split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let dx = sun[0] / sun[2];
    let dy = sun[1] / sun[2];

    let triangles = triangulate(&polygon);

    let mut final_tris: Vec<((f64, f64), (f64, f64), (f64, f64))> = Vec::new();
    for tri in &triangles {
        let t0 = polygon[tri.0];
        let t1 = polygon[tri.1];
        let t2 = polygon[tri.2];
        subdivide(&mut final_tris, t0, t1, t2);
    }

    let stdout = io::stdout();
    let mut out = stdout.lock();

    writeln!(out, "{}", final_tris.len()).unwrap();

    for (i, &(t0, t1, t2)) in final_tris.iter().enumerate() {
        let cx = (t0.0 + t1.0 + t2.0) / 3.0;
        let cy = (t0.1 + t1.1 + t2.1) / 3.0;

        let r0 = ((t0.0 - cx).powi(2) + (t0.1 - cy).powi(2)).sqrt();
        let r1 = ((t1.0 - cx).powi(2) + (t1.1 - cy).powi(2)).sqrt();
        let r2 = ((t2.0 - cx).powi(2) + (t2.1 - cy).powi(2)).sqrt();
        let r = r0.max(r1).max(r2);

        let scale = (r * 4.0 + 0.2).clamp(0.1, 10.0);
        let pz = 1.0 + (i as f64) * 15.0;
        let px = cx + pz * dx;
        let py = cy + pz * dy;

        writeln!(out, "{} {} {} 1 0 0 0 {}", px, py, pz, scale).unwrap();
    }
}

fn subdivide(result: &mut Vec<((f64, f64), (f64, f64), (f64, f64))>,
             t0: (f64, f64), t1: (f64, f64), t2: (f64, f64)) {
    let cx = (t0.0 + t1.0 + t2.0) / 3.0;
    let cy = (t0.1 + t1.1 + t2.1) / 3.0;

    let r0 = ((t0.0 - cx).powi(2) + (t0.1 - cy).powi(2)).sqrt();
    let r1 = ((t1.0 - cx).powi(2) + (t1.1 - cy).powi(2)).sqrt();
    let r2 = ((t2.0 - cx).powi(2) + (t2.1 - cy).powi(2)).sqrt();
    let r = r0.max(r1).max(r2);

    if r * 4.0 + 0.2 <= 10.0 {
        result.push((t0, t1, t2));
    } else {
        let m01 = ((t0.0 + t1.0) / 2.0, (t0.1 + t1.1) / 2.0);
        let m12 = ((t1.0 + t2.0) / 2.0, (t1.1 + t2.1) / 2.0);
        let m20 = ((t2.0 + t0.0) / 2.0, (t2.1 + t0.1) / 2.0);

        subdivide(result, t0, m01, m20);
        subdivide(result, m01, t1, m12);
        subdivide(result, m20, m12, t2);
        subdivide(result, m01, m12, m20);
    }
}

fn triangulate(polygon: &[(f64, f64)]) -> Vec<(usize, usize, usize)> {
    let n = polygon.len();
    if n < 3 { return vec![]; }
    if n == 3 { return vec![(0, 1, 2)]; }

    let mut result = Vec::new();
    let mut rem: Vec<usize> = (0..n).collect();

    if signed_area(polygon) < 0.0 {
        rem.reverse();
    }

    while rem.len() > 3 {
        let m = rem.len();
        let mut found = false;
        for i in 0..m {
            if is_ear(polygon, &rem, i) {
                let prev = rem[(i + m - 1) % m];
                let curr = rem[i];
                let next = rem[(i + 1) % m];
                result.push((prev, curr, next));
                rem.remove(i);
                found = true;
                break;
            }
        }
        if !found {
            result.push((rem[0], rem[1], rem[2]));
            rem.remove(1);
        }
    }

    if rem.len() == 3 {
        result.push((rem[0], rem[1], rem[2]));
    }

    result
}

fn signed_area(polygon: &[(f64, f64)]) -> f64 {
    let n = polygon.len();
    let mut area = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        area += polygon[i].0 * polygon[j].1 - polygon[j].0 * polygon[i].1;
    }
    area / 2.0
}

fn is_ear(polygon: &[(f64, f64)], rem: &[usize], idx: usize) -> bool {
    let n = rem.len();
    let pi = rem[(idx + n - 1) % n];
    let ci = rem[idx];
    let ni = rem[(idx + 1) % n];
    let a = polygon[pi];
    let b = polygon[ci];
    let c = polygon[ni];

    if (b.0 - a.0) * (c.1 - a.1) - (b.1 - a.1) * (c.0 - a.0) <= 0.0 {
        return false;
    }

    for (i, &vi) in rem.iter().enumerate() {
        if i == (idx + n - 1) % n || i == idx || i == (idx + 1) % n {
            continue;
        }
        if point_in_tri(polygon[vi], a, b, c) {
            return false;
        }
    }
    true
}

fn point_in_tri(p: (f64, f64), a: (f64, f64), b: (f64, f64), c: (f64, f64)) -> bool {
    let sign = |p1: (f64, f64), p2: (f64, f64), p3: (f64, f64)| {
        (p1.0 - p3.0) * (p2.1 - p3.1) - (p2.0 - p3.0) * (p1.1 - p3.1)
    };
    let d1 = sign(p, a, b);
    let d2 = sign(p, b, c);
    let d3 = sign(p, c, a);
    !((d1 < 0.0 || d2 < 0.0 || d3 < 0.0) && (d1 > 0.0 || d2 > 0.0 || d3 > 0.0))
}
