using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    static int width, height, numPieces;
    static bool[,] board;
    static int[] columnHeights;

    static readonly Dictionary<char, int[,][]> Pieces = new Dictionary<char, int[,][]>
    {
        ['I'] = new int[,][] {
            new int[,] {{1,1,1,1}},
            new int[,] {{1},{1},{1},{1}},
            new int[,] {{1,1,1,1}},
            new int[,] {{1},{1},{1},{1}}
        },
        ['O'] = new int[,][] {
            new int[,] {{1,1},{1,1}},
            new int[,] {{1,1},{1,1}},
            new int[,] {{1,1},{1,1}},
            new int[,] {{1,1},{1,1}}
        },
        ['T'] = new int[,][] {
            new int[,] {{1,1,1},{0,1,0}},
            new int[,] {{0,1},{1,1},{0,1}},
            new int[,] {{0,1,0},{1,1,1}},
            new int[,] {{1,0},{1,1},{1,0}}
        },
        ['S'] = new int[,][] {
            new int[,] {{0,1,1},{1,1,0}},
            new int[,] {{1,0},{1,1},{0,1}},
            new int[,] {{0,1,1},{1,1,0}},
            new int[,] {{1,0},{1,1},{0,1}}
        },
        ['Z'] = new int[,][] {
            new int[,] {{1,1,0},{0,1,1}},
            new int[,] {{0,1},{1,1},{1,0}},
            new int[,] {{1,1,0},{0,1,1}},
            new int[,] {{0,1},{1,1},{1,0}}
        },
        ['J'] = new int[,][] {
            new int[,] {{1,0,0},{1,1,1}},
            new int[,] {{1,1},{1,0},{1,0}},
            new int[,] {{1,1,1},{0,0,1}},
            new int[,] {{0,1},{0,1},{1,1}}
        },
        ['L'] = new int[,][] {
            new int[,] {{0,0,1},{1,1,1}},
            new int[,] {{1,0},{1,0},{1,1}},
            new int[,] {{1,1,1},{1,0,0}},
            new int[,] {{1,1},{0,1},{0,1}}
        }
    };

    static void Main()
    {
        string[] header = Console.ReadLine().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        width = int.Parse(header[0]);
        height = int.Parse(header[1]);
        numPieces = int.Parse(header[2]);

        board = new bool[height, width];
        columnHeights = new int[width];

        for (int i = 0; i < numPieces; i++)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) { i--; continue; }
            char pieceType = line.Trim()[0];

            var (bestRot, bestCol) = FindBestPlacement(pieceType);
            Console.WriteLine($"{bestRot} {bestCol}");
            Console.Out.Flush();

            string result = Console.ReadLine();
            if (result == null || result.Contains("gameover")) break;

            ApplyPiece(pieceType, bestRot, bestCol);
            if (result.StartsWith("cleared_"))
            {
                int cleared = int.Parse(result.Substring(8));
                ClearLines(cleared);
            }
        }
    }

    static (int rotation, int column) FindBestPlacement(char pieceType)
    {
        double bestScore = double.MinValue;
        int bestRot = 0, bestCol = 0;

        for (int rot = 0; rot < 4; rot++)
        {
            int[,] piece = Pieces[pieceType][rot];
            int pieceWidth = piece.GetLength(1);
            int pieceHeight = piece.GetLength(0);

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                if (!CanPlace(piece, col, out int landingRow)) continue;

                double score = EvaluatePlacement(piece, col, landingRow);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }
        return (bestRot, bestCol);
    }

    static bool CanPlace(int[,] piece, int col, out int landingRow)
    {
        int pieceWidth = piece.GetLength(1);
        int pieceHeight = piece.GetLength(0);
        landingRow = 0;

        for (int c = 0; c < pieceWidth; c++)
        {
            int lowestPieceRow = -1;
            for (int r = pieceHeight - 1; r >= 0; r--)
            {
                if (piece[r, c] == 1) { lowestPieceRow = r; break; }
            }
            if (lowestPieceRow == -1) continue;

            int targetRow = columnHeights[col + c] - (pieceHeight - 1 - lowestPieceRow);
            landingRow = Math.Max(landingRow, targetRow);
        }

        if (landingRow + pieceHeight > height) return false;

        for (int r = 0; r < pieceHeight; r++)
        {
            for (int c = 0; c < pieceWidth; c++)
            {
                if (piece[r, c] == 1 && board[landingRow + r, col + c])
                    return false;
            }
        }
        return true;
    }

    static double EvaluatePlacement(int[,] piece, int col, int landingRow)
    {
        int pieceWidth = piece.GetLength(1);
        int pieceHeight = piece.GetLength(0);

        bool[,] tempBoard = (bool[,])board.Clone();
        int[] tempHeights = (int[])columnHeights.Clone();

        for (int r = 0; r < pieceHeight; r++)
        {
            for (int c = 0; c < pieceWidth; c++)
            {
                if (piece[r, c] == 1)
                {
                    tempBoard[landingRow + r, col + c] = true;
                    tempHeights[col + c] = Math.Max(tempHeights[col + c], landingRow + r + 1);
                }
            }
        }

        int linesCleared = 0;
        for (int r = 0; r < height; r++)
        {
            bool full = true;
            for (int c = 0; c < width; c++)
            {
                if (!tempBoard[r, c]) { full = false; break; }
            }
            if (full) linesCleared++;
        }

        int holes = 0;
        for (int c = 0; c < width; c++)
        {
            bool foundBlock = false;
            for (int r = height - 1; r >= 0; r--)
            {
                if (tempBoard[r, c]) foundBlock = true;
                else if (foundBlock) holes++;
            }
        }

        int aggregateHeight = tempHeights.Sum();
        int maxHeight = tempHeights.Max();

        int bumpiness = 0;
        for (int c = 0; c < width - 1; c++)
            bumpiness += Math.Abs(tempHeights[c] - tempHeights[c + 1]);

        int wells = 0;
        for (int c = 0; c < width; c++)
        {
            int left = c > 0 ? tempHeights[c - 1] : height;
            int right = c < width - 1 ? tempHeights[c + 1] : height;
            int wellDepth = Math.Min(left, right) - tempHeights[c];
            if (wellDepth > 0) wells += wellDepth;
        }

        double score = 0;
        score += linesCleared * 1000;
        score -= holes * 150;
        score -= aggregateHeight * 5;
        score -= bumpiness * 10;
        score -= maxHeight * 8;
        score -= wells * 5;

        if (maxHeight > height * 0.7)
        {
            score -= (maxHeight - height * 0.7) * 50;
            score += linesCleared * 500;
        }

        return score;
    }

    static void ApplyPiece(char pieceType, int rot, int col)
    {
        int[,] piece = Pieces[pieceType][rot];
        int pieceWidth = piece.GetLength(1);
        int pieceHeight = piece.GetLength(0);

        if (!CanPlace(piece, col, out int landingRow)) return;

        for (int r = 0; r < pieceHeight; r++)
        {
            for (int c = 0; c < pieceWidth; c++)
            {
                if (piece[r, c] == 1)
                {
                    board[landingRow + r, col + c] = true;
                    columnHeights[col + c] = Math.Max(columnHeights[col + c], landingRow + r + 1);
                }
            }
        }
    }

    static void ClearLines(int count)
    {
        for (int cleared = 0; cleared < count;)
        {
            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++)
                {
                    if (!board[r, c]) { full = false; break; }
                }
                if (full)
                {
                    for (int row = r; row < height - 1; row++)
                    {
                        for (int c = 0; c < width; c++)
                            board[row, c] = board[row + 1, c];
                    }
                    for (int c = 0; c < width; c++)
                        board[height - 1, c] = false;
                    cleared++;
                    break;
                }
            }
        }

        for (int c = 0; c < width; c++)
        {
            columnHeights[c] = 0;
            for (int r = height - 1; r >= 0; r--)
            {
                if (board[r, c]) { columnHeights[c] = r + 1; break; }
            }
        }
    }
}
