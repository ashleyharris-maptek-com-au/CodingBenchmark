use std::io::{self, Read, Write, BufWriter};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut input = String::new();
    stdin.lock().read_to_string(&mut input).unwrap();

    let mut iter = input.split_ascii_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let target: i64 = iter.next().unwrap().parse().unwrap();

    let nums: Vec<i64> = iter.take(n).map(|s| s.parse().unwrap()).collect();

    if let Some(indices) = solve_subset_sum(&nums, target) {
        writeln!(out, "YES").unwrap();
        if indices.is_empty() {
            writeln!(out, "").unwrap();
        } else {
            let idx_str: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            writeln!(out, "{}", idx_str.join(" ")).unwrap();
        }
    } else {
        writeln!(out, "NO").unwrap();
    }
}

fn solve_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();

    if target == 0 {
        return Some(vec![]);
    }

    if n == 0 {
        return None;
    }

    for (i, &num) in nums.iter().enumerate() {
        if num == target {
            return Some(vec![i]);
        }
    }

    let total_sum: i64 = nums.iter().sum();
    if total_sum == target {
        return Some((0..n).collect());
    }

    let positive_sum: i64 = nums.iter().filter(|&&x| x > 0).sum();
    let negative_sum: i64 = nums.iter().filter(|&&x| x < 0).sum();

    if target > positive_sum || target < negative_sum {
        return None;
    }

    if n <= 24 {
        return brute_force(nums, target);
    }

    if n <= 40 {
        return meet_in_middle(nums, target);
    }

    let all_positive = nums.iter().all(|&x| x >= 0);
    let all_negative = nums.iter().all(|&x| x <= 0);

    if all_positive && target > 0 && target <= 50_000_000 {
        return bitset_dp_positive(nums, target);
    }

    if all_negative && target < 0 && -target <= 50_000_000 {
        let negated: Vec<i64> = nums.iter().map(|&x| -x).collect();
        return bitset_dp_positive(&negated, -target);
    }

    if n <= 1000 {
        if let Some(result) = hashset_dp(nums, target) {
            return Some(result);
        }
    }

    if let Some(result) = greedy_with_refinement(nums, target) {
        return Some(result);
    }

    random_search(nums, target)
}

fn brute_force(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    for mask in 1u32..(1u32 << n) {
        let mut sum: i64 = 0;
        for i in 0..n {
            if mask & (1u32 << i) != 0 {
                sum += nums[i];
            }
        }
        if sum == target {
            let indices: Vec<usize> = (0..n).filter(|&i| mask & (1u32 << i) != 0).collect();
            return Some(indices);
        }
    }
    None
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;

    let mut first_half: HashMap<i64, u64> = HashMap::new();
    for mask in 0u64..(1u64 << mid) {
        let mut sum: i64 = 0;
        for i in 0..mid {
            if mask & (1u64 << i) != 0 {
                sum += nums[i];
            }
        }
        first_half.entry(sum).or_insert(mask);
    }

    let second_len = n - mid;
    for mask2 in 0u64..(1u64 << second_len) {
        let mut sum: i64 = 0;
        for i in 0..second_len {
            if mask2 & (1u64 << i) != 0 {
                sum += nums[mid + i];
            }
        }
        let needed = target - sum;
        if let Some(&mask1) = first_half.get(&needed) {
            if mask1 == 0 && mask2 == 0 {
                continue;
            }
            let mut indices = Vec::new();
            for i in 0..mid {
                if mask1 & (1u64 << i) != 0 {
                    indices.push(i);
                }
            }
            for i in 0..second_len {
                if mask2 & (1u64 << i) != 0 {
                    indices.push(mid + i);
                }
            }
            return Some(indices);
        }
    }
    None
}

fn bitset_dp_positive(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let target_usize = target as usize;
    let num_words = (target_usize + 64) / 64;
    let mut dp: Vec<u64> = vec![0; num_words];
    dp[0] = 1;

    let mut parent: Vec<Option<(usize, usize)>> = vec![None; target_usize + 1];

    for (idx, &num) in nums.iter().enumerate() {
        if num <= 0 || num > target {
            continue;
        }
        let num_usize = num as usize;

        for s in (num_usize..=target_usize).rev() {
            let prev = s - num_usize;
            let prev_word = prev / 64;
            let prev_bit = prev % 64;
            let curr_word = s / 64;
            let curr_bit = s % 64;

            if (dp[prev_word] >> prev_bit) & 1 == 1 && (dp[curr_word] >> curr_bit) & 1 == 0 {
                dp[curr_word] |= 1u64 << curr_bit;
                parent[s] = Some((prev, idx));
            }
        }

        let target_word = target_usize / 64;
        let target_bit = target_usize % 64;
        if (dp[target_word] >> target_bit) & 1 == 1 {
            let mut indices = Vec::new();
            let mut curr = target_usize;
            while let Some((prev, idx)) = parent[curr] {
                indices.push(idx);
                curr = prev;
            }
            indices.sort();
            return Some(indices);
        }
    }

    None
}

fn hashset_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for (idx, &num) in nums.iter().enumerate() {
        let entries: Vec<(i64, Vec<usize>)> = dp.iter()
            .filter(|&(s, _)| !dp.contains_key(&(s + num)))
            .map(|(s, indices)| {
                let mut new_indices = indices.clone();
                new_indices.push(idx);
                (*s + num, new_indices)
            })
            .collect();

        for (new_sum, indices) in entries {
            if new_sum == target {
                return Some(indices);
            }
            if dp.len() < 1_000_000 {
                dp.insert(new_sum, indices);
            }
        }
    }

    dp.get(&target).cloned()
}

fn greedy_with_refinement(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut indexed: Vec<(usize, i64)> = nums.iter().enumerate().map(|(i, &v)| (i, v)).collect();
    indexed.sort_by(|a, b| b.1.abs().cmp(&a.1.abs()));

    let mut current_sum: i64 = 0;
    let mut selected: Vec<usize> = Vec::new();

    for &(idx, val) in &indexed {
        if current_sum + val == target {
            selected.push(idx);
            selected.sort();
            return Some(selected);
        }

        let diff_with = (current_sum + val - target).abs();
        let diff_without = (current_sum - target).abs();

        if diff_with < diff_without {
            current_sum += val;
            selected.push(idx);
        }
    }

    if current_sum == target && !selected.is_empty() {
        selected.sort();
        return Some(selected);
    }

    let remaining: Vec<(usize, i64)> = indexed.iter()
        .filter(|(idx, _)| !selected.contains(idx))
        .copied()
        .collect();

    for &(idx, val) in &remaining {
        if current_sum + val == target {
            selected.push(idx);
            selected.sort();
            return Some(selected);
        }
    }

    for i in 0..selected.len() {
        let removed_idx = selected[i];
        let removed_val = nums[removed_idx];
        let new_sum = current_sum - removed_val;

        for &(idx, val) in &remaining {
            if new_sum + val == target {
                let mut result: Vec<usize> = selected.iter()
                    .filter(|&&x| x != removed_idx)
                    .copied()
                    .collect();
                result.push(idx);
                result.sort();
                return Some(result);
            }
        }
    }

    None
}

fn random_search(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return None;
    }

    let mut seed: u64 = 12345;
    let iterations = std::cmp::min(n * 1000, 10_000_000);

    for _ in 0..iterations {
        seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
        let start = (seed as usize) % n;

        let mut current_sum: i64 = 0;
        let mut selected: Vec<usize> = Vec::new();

        for offset in 0..n {
            let idx = (start + offset) % n;
            let val = nums[idx];

            if current_sum + val == target {
                selected.push(idx);
                selected.sort();
                return Some(selected);
            }

            let diff_with = (current_sum + val - target).abs();
            let diff_without = (current_sum - target).abs();

            if diff_with <= diff_without || (seed >> 32) % 10 < 3 {
                current_sum += val;
                selected.push(idx);
                seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
            }
        }

        if current_sum == target && !selected.is_empty() {
            selected.sort();
            return Some(selected);
        }
    }

    None
}
