```javascript
// Node.js program: Plan grid paths for search drones
// Reads one JSON from stdin (see problem statement) and writes one JSON to stdout.
// No external libraries. Designed to scale reasonably for large grids.

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Input fields
const probability = data.probability || [];
const drones = data.drones || [];
const maxStepsPerDrone = Number.isInteger(data.maxStepsPerDrone) ? data.maxStepsPerDrone : 0;

// Grid dimensions
const H = probability.length;
const W = H > 0 ? (probability[0] ? probability[0].length : 0) : 0;

// Helper: clone probability values into mutable remaining-probability grid
const rem = new Array(H);
for (let y = 0; y < H; y++) {
  rem[y] = new Array(W);
  for (let x = 0; x < W; x++) {
    // Keep obstacles as -1, otherwise positive values (or zero)
    rem[y][x] = probability[y][x];
  }
}

// Directions: [dx,dy]
const DIRS = [
  [0, -1], // up
  [1, 0],  // right
  [0, 1],  // down
  [-1, 0], // left
];

// BFS that returns distance grid and parent encoding from start, limited by maxDepth
function bfsDistances(startX, startY, maxDepth) {
  // Initialize distance with -1 (unvisited)
  const dist = Array.from({ length: H }, () => new Int32Array(W).fill(-1));
  const parent = Array.from({ length: H }, () => new Int32Array(W).fill(-1)); // encode prev as py*W + px
  if (startX < 0 || startX >= W || startY < 0 || startY >= H) return { dist, parent };
  if (rem[startY][startX] === -1) return { dist, parent }; // start on obstacle -> unreachable

  const qx = new Int32Array((maxDepth + 1) * (W * H) + 4); // allocate large enough (safe)
  const qy = new Int32Array((maxDepth + 1) * (W * H) + 4);
  let qh = 0, qt = 0;
  qx[qt] = startX; qy[qt] = startY; qt++;
  dist[startY][startX] = 0;
  parent[startY][startX] = -1;

  while (qh < qt) {
    const x = qx[qh], y = qy[qh]; qh++;
    const d = dist[y][x];
    if (d >= maxDepth) continue; // do not expand further than maxDepth
    for (let k = 0; k < 4; k++) {
      const nx = x + DIRS[k][0];
      const ny = y + DIRS[k][1];
      if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
      if (rem[ny][nx] === -1) continue; // obstacle
      if (dist[ny][nx] !== -1) continue; // visited
      dist[ny][nx] = d + 1;
      parent[ny][nx] = y * W + x;
      qx[qt] = nx; qy[qt] = ny; qt++;
    }
  }
  return { dist, parent };
}

// Reconstruct path (list of [x,y]) from start to target using parent grid.
// Includes start and target positions.
function reconstructPath(parent, startX, startY, tx, ty) {
  const path = [];
  let cx = tx, cy = ty;
  // If target unreachable (parent -1 and not start), handle accordingly
  if (cx === startX && cy === startY) {
    path.push([cx, cy]);
    return path;
  }
  while (!(cx === startX && cy === startY)) {
    path.push([cx, cy]);
    const p = parent[cy][cx];
    if (p === -1) {
      // not reachable; return empty to signal failure
      return [];
    }
    cx = p % W;
    cy = Math.floor(p / W);
  }
  path.push([startX, startY]);
  path.reverse();
  return path;
}

// Main planning
const paths = [];

for (let di = 0; di < drones.length; di++) {
  const drone = drones[di];
  let cx = Math.max(0, Math.min(W - 1, Math.floor(drone.x)));
  let cy = Math.max(0, Math.min(H - 1, Math.floor(drone.y)));

  // If initial cell is an obstacle, try to find nearest non-obstacle within small radius; otherwise stay in place but cannot move.
  if (H > 0 && W > 0 && rem[cy][cx] === -1) {
    // simple local search for nearest free cell (Manhattan up to 10)
    let found = false;
    const maxTry = Math.max(H + W, 10);
    for (let r = 1; r <= maxTry && !found; r++) {
      for (let dx = -r; dx <= r && !found; dx++) {
        const dy = r - Math.abs(dx);
        for (const sy of [dy, -dy]) {
          const nx = cx + dx;
          const ny = cy + sy;
          if (nx >= 0 && nx < W && ny >= 0 && ny < H && rem[ny][nx] !== -1) {
            cx = nx; cy = ny; found = true; break;
          }
          if (dy === 0) break;
        }
      }
    }
    // if not found, keep original (will be stuck).
  }

  const path = [];
  path.push([cx, cy]); // include starting cell

  // If start cell has positive probability, mark it as covered
  if (H > 0 && W > 0 && rem[cy][cx] > 0) {
    rem[cy][cx] = 0;
  }

  let stepsRemaining = maxStepsPerDrone;

  while (stepsRemaining > 0) {
    // BFS distances up to stepsRemaining
    const { dist, parent } = bfsDistances(cx, cy, stepsRemaining);
    // Choose best target cell maximizing score = remVal / (dist+1)
    let bestScore = 0;
    let bestX = -1, bestY = -1, bestDist = -1, bestVal = 0;
    for (let y = 0; y < H; y++) {
      for (let x = 0; x < W; x++) {
        const v = rem[y][x];
        if (!v || v <= 0) continue;
        const d = dist[y][x];
        if (d === -1) continue; // unreachable within remaining steps
        // score heuristic: value / (distance + 1)
        const score = v / (d + 1);
        if (score > bestScore || (score === bestScore && (v > bestVal || d < bestDist))) {
          bestScore = score;
          bestX = x; bestY = y; bestDist = d; bestVal = v;
        }
      }
    }

    if (bestScore <= 0 || bestX === -1) {
      // No beneficial target reachable: stay in place for remaining steps
      for (let s = 0; s < stepsRemaining; s++) {
        path.push([cx, cy]);
      }
      stepsRemaining = 0;
      break;
    }

    // Reconstruct path from current to best target
    const subpath = reconstructPath(parent, cx, cy, bestX, bestY);
    if (subpath.length === 0) {
      // unreachable unexpectedly, break and stay
      for (let s = 0; s < stepsRemaining; s++) path.push([cx, cy]);
      stepsRemaining = 0;
      break;
    }

    // subpath includes current position at index 0; we should consume moves (excluding the current)
    // Apply moves one by one, marking covered cells and decrementing stepsRemaining
    for (let i = 1; i < subpath.length && stepsRemaining > 0; i++) {
      const [nx, ny] = subpath[i];
      // double-check valid
      if (nx < 0 || nx >= W || ny < 0 || ny >= H) break;
      if (rem[ny][nx] === -1) {
        // obstacle encountered unexpectedly, stop pathing
        stepsRemaining = 0;
        break;
      }
      path.push([nx, ny]);
      // mark covered unique probability
      if (rem[ny][nx] > 0) rem[ny][nx] = 0;
      // update current position
      cx = nx; cy = ny;
      stepsRemaining--;
    }

    // if we reached the target and still have steps, loop again to seek next target
    // else if stepsRemaining == 0 loop ends automatically
  }

  // Ensure the path length equals maxStepsPerDrone + 1 (start + moves). Pad by staying if needed.
  const targetLen = Math.max(1, maxStepsPerDrone + 1);
  while (path.length < targetLen) {
    const last = path[path.length - 1] || [cx, cy];
    path.push([last[0], last[1]]);
  }
  // If path longer (rare), truncate to allowed length
  if (path.length > targetLen) path.length = targetLen;

  paths.push(path);
}

// Output JSON
const output = { paths: paths };
console.log(JSON.stringify(output));
```
