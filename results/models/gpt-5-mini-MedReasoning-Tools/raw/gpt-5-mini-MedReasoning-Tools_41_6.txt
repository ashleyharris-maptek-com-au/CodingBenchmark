; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColorVar
OpExecutionMode %main OriginUpperLeft

; Names
OpName %main "main"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %outColorVar "outColorVar"
OpName %ubo "ubo"
OpName %uboStruct "UBO"

; Decorations for inputs/outputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColorVar Location 0

; UBO struct and members
OpDecorate %uboStruct Block
OpMemberDecorate %uboStruct 0 Offset 0
OpMemberDecorate %uboStruct 1 Offset 64
OpMemberDecorate %uboStruct 2 Offset 128
OpMemberDecorate %uboStruct 3 Offset 192
OpMemberDecorate %uboStruct 4 Offset 208
OpMemberDecorate %uboStruct 5 Offset 224

; Descriptor for uniform buffer variable
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Import GLSL std450 for Normalize, Pow, etc.
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%f32 = OpTypeFloat 32
%v2f = OpTypeVector %f32 2
%v3f = OpTypeVector %f32 3
%v4f = OpTypeVector %f32 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%ptr_input_v3 = OpTypePointer Input %v3f
%ptr_input_v2 = OpTypePointer Input %v2f
%ptr_output_v4 = OpTypePointer Output %v4f

%ptr_uniform_v4 = OpTypePointer Uniform %v4f
%ptr_uniform_mat4 = OpTypePointer Uniform %mat4

%uboStruct = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_uniform_ubo = OpTypePointer Uniform %uboStruct

; Input/Output variable declarations
%inWorldPos = OpVariable %ptr_input_v3 Input
%inNormal   = OpVariable %ptr_input_v3 Input
%inUV       = OpVariable %ptr_input_v2 Input
%inTangent  = OpVariable %ptr_input_v3 Input
%inColor    = OpVariable %ptr_input_v3 Input

%outColorVar = OpVariable %ptr_output_v4 Output

; Uniform variable
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%f32_0 = OpConstant %f32 0
%f32_0p1 = OpConstant %f32 0.100000001
%f32_0p6 = OpConstant %f32 0.600000024
%f32_0p8 = OpConstant %f32 0.800000012
%f32_1 = OpConstant %f32 1
%f32_64 = OpConstant %f32 64

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %inWorldPos
%normalIn = OpLoad %v3f %inNormal
%uvVal = OpLoad %v2f %inUV
%tangentVal = OpLoad %v3f %inTangent
%colorIn = OpLoad %v3f %inColor

; Load lightPos (member 3) and cameraPos (member 4) from UBO
%lightPtr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%lightVec4 = OpLoad %v4f %lightPtr
%cameraPtr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%cameraVec4 = OpLoad %v4f %cameraPtr

; Extract xyz components
%lightXYZ = OpVectorShuffle %v3f %lightVec4 %lightVec4 0 1 2
%cameraXYZ = OpVectorShuffle %v3f %cameraVec4 %cameraVec4 0 1 2

; Normalize normal
%n_norm = OpExtInst %v3f %glsl Normalize %normalIn

; Compute and normalize light vector L = normalize(lightPos - worldPos)
%Lpre = OpFSub %v3f %lightXYZ %worldPos
%L = OpExtInst %v3f %glsl Normalize %Lpre

; Compute and normalize view vector V = normalize(cameraPos - worldPos)
%Vpre = OpFSub %v3f %cameraXYZ %worldPos
%V = OpExtInst %v3f %glsl Normalize %Vpre

; Diffuse = max(dot(N, L), 0)
%dotNL = OpDot %f32 %n_norm %L
%diffuse = OpFMax %f32 %dotNL %f32_0

; Half vector H = normalize(L + V)
%LplusV = OpFAdd %v3f %L %V
%H = OpExtInst %v3f %glsl Normalize %LplusV

; Specular = pow(max(dot(N, H), 0), 64)
%dotNH = OpDot %f32 %n_norm %H
%dotNHpos = OpFMax %f32 %dotNH %f32_0
%specular = OpExtInst %f32 %glsl Pow %dotNHpos %f32_64

; intensity = 0.1 + 0.6*diffuse + 0.8*specular
%dTerm = OpFMul %f32 %f32_0p6 %diffuse
%sTerm = OpFMul %f32 %f32_0p8 %specular
%sum1 = OpFAdd %f32 %dTerm %sTerm
%intensity = OpFAdd %f32 %f32_0p1 %sum1

; Construct output color vec4(intensity, intensity, intensity, 1.0)
%outVec = OpCompositeConstruct %v4f %intensity %intensity %intensity %f32_1
OpStore %outColorVar %outVec

OpReturn
OpFunctionEnd
