Here's a complete, self-contained Node.js TypeScript program that implements a deterministic, scalable route planner using a k-means clustering (deterministic kmeans++) followed by nearest-neighbor + 2-opt route improvement per cluster. It reads the JSON from stdin using the exact pattern you provided and prints only the required JSON output.

Save as a .ts file and compile/run with your TypeScript toolchain.

```ts
declare const require: any;
const fs = require('fs');

type Point = { id: string; x: number; y: number };
type Vec = { x: number; y: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot: [number, number] = data.depot;
const truckCount: number = Math.max(0, Math.floor(data.truckCount || 0));
const packages: Point[] = (data.packages || []).map((p: any) => ({
  id: String(p.id),
  x: Number(p.x),
  y: Number(p.y),
}));

// Utility functions
function dist2(a: Vec, b: Vec): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return dx * dx + dy * dy;
}
function dist(a: Vec, b: Vec): number {
  return Math.sqrt(dist2(a, b));
}

function pointVec(p: Point): Vec {
  return { x: p.x, y: p.y };
}

// If no trucks, produce 0 routes (but ensure each package appears exactly once -> not possible).
// The problem states truckCount is maximum number of vans/routes. If truckCount is 0, we must still produce 0 routes,
// but then packages can't be assigned. To be safe, if truckCount <= 0, produce 1 route containing all packages.
const K = truckCount <= 0 ? 1 : truckCount;

// If no packages, output K empty routes
if (packages.length === 0) {
  const emptyRoutes: string[][] = [];
  for (let i = 0; i < K; i++) emptyRoutes.push([]);
  console.log(JSON.stringify({ routes: emptyRoutes }));
  process.exit(0);
}

// Deterministic kmeans++ initialization
function kmeans(points: Point[], k: number, maxIters = 50): number[] {
  const n = points.length;
  k = Math.min(k, n); // at most one center per point
  const centers: Vec[] = [];
  const centerIdx: number[] = [];

  // Choose first center deterministically: smallest x, then smallest y, then smallest index
  let firstIdx = 0;
  for (let i = 1; i < n; i++) {
    if (
      points[i].x < points[firstIdx].x ||
      (points[i].x === points[firstIdx].x && points[i].y < points[firstIdx].y)
    ) {
      firstIdx = i;
    }
  }
  centers.push(pointVec(points[firstIdx]));
  centerIdx.push(firstIdx);

  // Choose remaining centers: pick point maximizing min distance^2 to existing centers (deterministic tie break by index)
  while (centers.length < k) {
    let bestIdx = -1;
    let bestD = -1;
    for (let i = 0; i < n; i++) {
      // skip if already chosen
      if (centerIdx.indexOf(i) >= 0) continue;
      let md = Infinity;
      const v = pointVec(points[i]);
      for (const c of centers) {
        const d = dist2(v, c);
        if (d < md) md = d;
      }
      if (md > bestD || (md === bestD && i < bestIdx)) {
        bestD = md;
        bestIdx = i;
      }
    }
    if (bestIdx === -1) break;
    centers.push(pointVec(points[bestIdx]));
    centerIdx.push(bestIdx);
  }

  // Run Lloyd's algorithm (k-means)
  const assignments = new Array(n).fill(0);
  for (let iter = 0; iter < maxIters; iter++) {
    let changed = false;
    // Assignment step
    for (let i = 0; i < n; i++) {
      const v = pointVec(points[i]);
      let best = 0;
      let bestd = dist2(v, centers[0]);
      for (let j = 1; j < centers.length; j++) {
        const d = dist2(v, centers[j]);
        if (d < bestd) {
          bestd = d;
          best = j;
        }
      }
      if (assignments[i] !== best) {
        assignments[i] = best;
        changed = true;
      }
    }
    // Update step
    const sums: Vec[] = [];
    const counts: number[] = [];
    for (let j = 0; j < centers.length; j++) {
      sums.push({ x: 0, y: 0 });
      counts.push(0);
    }
    for (let i = 0; i < n; i++) {
      const a = assignments[i];
      sums[a].x += points[i].x;
      sums[a].y += points[i].y;
      counts[a]++;
    }
    for (let j = 0; j < centers.length; j++) {
      if (counts[j] > 0) {
        centers[j] = { x: sums[j].x / counts[j], y: sums[j].y / counts[j] };
      } else {
        // empty cluster: reinitialize to the point farthest from current centers
        let farIdx = -1;
        let farD = -1;
        for (let i = 0; i < n; i++) {
          const v = pointVec(points[i]);
          let md = Infinity;
          for (let c of centers) {
            const d = dist2(v, c);
            if (d < md) md = d;
          }
          if (md > farD) {
            farD = md;
            farIdx = i;
          }
        }
        if (farIdx >= 0) centers[j] = pointVec(points[farIdx]);
      }
    }

    if (!changed) break;
  }

  return assignments; // index per point -> cluster index in [0..k-1]
}

// Build clusters
const k = Math.min(K, packages.length);
const assignments = kmeans(packages, k, 50);
const clusters: Point[][] = [];
for (let i = 0; i < k; i++) clusters.push([]);
for (let i = 0; i < packages.length; i++) {
  const a = Math.max(0, Math.min(k - 1, assignments[i] || 0));
  clusters[a].push(packages[i]);
}

// If K > k, add empty clusters to make exactly K routes
while (clusters.length < K) clusters.push([]);

// TSP heuristics per cluster: nearest neighbor from depot then 2-opt
const depotVec = { x: depot[0], y: depot[1] };

function routeDistanceOrder(order: number[], points: Point[]): number {
  // order contains indices into points array (0..points.length-1)
  if (order.length === 0) return 0;
  let d = 0;
  let prev = depotVec;
  for (const idx of order) {
    d += dist(prev, pointVec(points[idx]));
    prev = pointVec(points[idx]);
  }
  d += dist(prev, depotVec);
  return d;
}

function nearestNeighborOrder(points: Point[]): number[] {
  const n = points.length;
  if (n === 0) return [];
  const unvisited = new Array(n).fill(true);
  const order: number[] = [];
  // start with point closest to depot (deterministic tie-break by index)
  let start = 0;
  let bestd = dist2(pointVec(points[0]), depotVec);
  for (let i = 1; i < n; i++) {
    const d = dist2(pointVec(points[i]), depotVec);
    if (d < bestd || (d === bestd && i < start)) {
      bestd = d;
      start = i;
    }
  }
  let cur = start;
  order.push(cur);
  unvisited[cur] = false;
  for (let step = 1; step < n; step++) {
    let best = -1;
    let bestDist = Infinity;
    const curVec = pointVec(points[cur]);
    for (let j = 0; j < n; j++) {
      if (!unvisited[j]) continue;
      const d = dist2(curVec, pointVec(points[j]));
      if (d < bestDist || (d === bestDist && j < best)) {
        bestDist = d;
        best = j;
      }
    }
    if (best === -1) break;
    order.push(best);
    unvisited[best] = false;
    cur = best;
  }
  return order;
}

function twoOptImprove(order: number[], points: Point[], timeLimitMillis = 200): number[] {
  // 2-opt improvement with time cap per cluster to keep runtime bounded.
  const n = order.length;
  if (n <= 2) return order;
  const startTime = Date.now();
  let improved = true;
  // Precompute point vectors
  const pvecs = points.map(pointVec);

  function edgeCost(aIdx: number | 'depot', bIdx: number | 'depot'): number {
    const a = aIdx === 'depot' ? depotVec : pvecs[aIdx];
    const b = bIdx === 'depot' ? depotVec : pvecs[bIdx];
    return dist(a, b);
  }

  // Convert order indices into indices of points array for easier referencing
  while (improved) {
    improved = false;
    for (let i = 0; i < n - 1; i++) {
      for (let k = i + 1; k < n; k++) {
        // Consider reversing (i..k)
        // Edges affected:
        // prev -> i
        // k -> next
        const prevIdx = i - 1;
        const nextIdx = k + 1;
        const a = prevIdx >= 0 ? order[prevIdx] : 'depot';
        const b = order[i];
        const c = order[k];
        const d = nextIdx < n ? order[nextIdx] : 'depot';

        const before = edgeCost(a, b) + edgeCost(c, d);
        const after = edgeCost(a, c) + edgeCost(b, d);
        if (after + 1e-12 < before) {
          // perform reversal
          let l = i, r = k;
          while (l < r) {
            const tmp = order[l];
            order[l] = order[r];
            order[r] = tmp;
            l++; r--;
          }
          improved = true;
        }
        if (Date.now() - startTime > timeLimitMillis) return order;
      }
    }
  }
  return order;
}

// Build final routes as lists of package ids
const routes: string[][] = [];

for (let ci = 0; ci < clusters.length; ci++) {
  const cluster = clusters[ci];
  if (cluster.length === 0) {
    routes.push([]);
    continue;
  }
  // Build nearest neighbor order (returns indexes into cluster array)
  let order = nearestNeighborOrder(cluster);

  // Improve with 2-opt. Limit time per cluster depending on cluster size.
  // Use up to 2 ms per point, capped to 500 ms per cluster.
  const timeLimit = Math.min(500, Math.max(10, cluster.length * 2));
  order = twoOptImprove(order, cluster, timeLimit);

  // Convert to package IDs
  const routeIds = order.map(idx => cluster[idx].id);
  routes.push(routeIds);
}

// Guarantee each package appears exactly once: if any packages missing due to unexpected logic, append them to routes[0]
const seen = new Set<string>();
for (const r of routes) for (const id of r) seen.add(id);
const missing: string[] = [];
for (const p of packages) if (!seen.has(p.id)) missing.push(p.id);
if (missing.length > 0) {
  if (routes.length === 0) routes.push([]);
  routes[0].push(...missing);
}

// Ensure routes length is exactly K (truckCount or 1 if truckCount <= 0)
while (routes.length < K) routes.push([]);
if (routes.length > K) routes.length = K;

// Output JSON only
console.log(JSON.stringify({ routes }));
```

Notes:
- Deterministic behavior: initialization and all tie-breaking are deterministic (no RNG).
- Scales: uses k-means clustering to split large problems and limits per-cluster 2-opt time to keep runtime bounded for big inputs.
- Produces exactly truckCount routes (or 1 route when provided truckCount <= 0) and ensures every package id appears exactly once.
