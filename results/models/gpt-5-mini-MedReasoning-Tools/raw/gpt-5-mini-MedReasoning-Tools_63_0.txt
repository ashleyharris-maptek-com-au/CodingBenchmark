declare const require: any;
const fs = require('fs');

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority?: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robotsInput: Robot[] = data.robots || [];
const jobsInput: Job[] = data.jobs || [];

// Basic validations
const height = grid.length;
const width = height > 0 ? grid[0].length : 0;

// Convert grid to boolean walkable array for faster checks
const walkable: Uint8Array = new Uint8Array(width * height);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    walkable[y * width + x] = (row.charAt(x) !== '#') ? 1 : 0;
  }
}

// Helper: encode/decode positions
const enc = (x: number, y: number) => y * width + x;
const decX = (pos: number) => pos % width;
const decY = (pos: number) => Math.floor(pos / width);

// Validate and normalize robots and jobs
const robots: Robot[] = robotsInput.map(r => ({ id: Math.floor(r.id), x: Math.floor(r.x), y: Math.floor(r.y) }));
const jobs: Job[] = jobsInput.map(j => ({ id: j.id, x: Math.floor(j.x), y: Math.floor(j.y), priority: j.priority ?? 0 }));

// Determine assignment array size (index == robot id)
let maxRobotId = -1;
for (const r of robots) if (r.id > maxRobotId) maxRobotId = r.id;
const robotCountIndex = Math.max(maxRobotId + 1, 0);
const assignments: any[] = new Array(robotCountIndex);
for (let i = 0; i < robotCountIndex; i++) assignments[i] = [];

// Quick helper to check if a coordinate is inside and walkable
function inBounds(x: number, y: number) {
  return x >= 0 && x < width && y >= 0 && y < height;
}
function isWalkablePos(x: number, y: number) {
  return inBounds(x, y) && walkable[enc(x, y)] === 1;
}

// BFS that starts from a single start position and stops when we've found all target positions (if provided).
// Returns a Map from encoded target position to distance. If targetPositions is omitted, returns full distance array.
function bfsDistancesFrom(startX: number, startY: number, targetSet?: Set<number>): Map<number, number> | Int32Array {
  const total = width * height;
  const dist = new Int32Array(total);
  for (let i = 0; i < total; i++) dist[i] = -1;
  const q: number[] = [];
  const s = enc(startX, startY);
  if (!inBounds(startX, startY) || walkable[s] === 0) {
    // Unwalkable start: return empty map or full dist with -1
    if (targetSet) return new Map<number, number>();
    return dist;
  }
  dist[s] = 0;
  q.push(s);
  let qi = 0;
  const resultMap = new Map<number, number>();
  if (targetSet && targetSet.has(s)) {
    resultMap.set(s, 0);
    if (resultMap.size === targetSet.size) return resultMap;
  }
  while (qi < q.length) {
    const cur = q[qi++];
    const dcur = dist[cur];
    const cx = decX(cur);
    const cy = decY(cur);
    // neighbors: up, down, left, right
    const nx1 = cx, ny1 = cy - 1;
    const nx2 = cx, ny2 = cy + 1;
    const nx3 = cx - 1, ny3 = cy;
    const nx4 = cx + 1, ny4 = cy;
    const nlist = [nx1, ny1, nx2, ny2, nx3, ny3, nx4, ny4];
    for (let k = 0; k < 4; k++) {
      const nx = nlist[k * 2], ny = nlist[k * 2 + 1];
      if (!inBounds(nx, ny)) continue;
      const npos = enc(nx, ny);
      if (walkable[npos] === 0) continue;
      if (dist[npos] !== -1) continue;
      dist[npos] = dcur + 1;
      q.push(npos);
      if (targetSet && targetSet.has(npos)) {
        resultMap.set(npos, dist[npos]);
        if (resultMap.size === targetSet.size) return resultMap;
      }
    }
  }
  if (targetSet) return resultMap;
  return dist;
}

// Build mapping of job index and robot index for internal arrays
const jobCount = jobs.length;
const robotCount = robots.length;
const jobIndexById = new Map<any, number>();
for (let i = 0; i < jobCount; i++) jobIndexById.set(jobs[i].id, i);
const robotIndexById = new Map<number, number>();
for (let i = 0; i < robotCount; i++) robotIndexById.set(robots[i].id, i);

// Precompute distances from robots to jobs. To scale, choose BFS from smaller side (robots or jobs).
// We'll compute a matrix distRtoJ[robotIndex][jobIndex] = distance or Infinity
const INF = 1 << 28;
const distRtoJ: number[][] = new Array(robotCount);
for (let i = 0; i < robotCount; i++) {
  distRtoJ[i] = new Array(jobCount).fill(INF);
}

if (robotCount <= jobCount) {
  // BFS from each robot, stop when we've found all jobs or exhausted
  const jobPosSet = new Set<number>();
  for (const j of jobs) jobPosSet.add(enc(j.x, j.y));
  for (let ri = 0; ri < robotCount; ri++) {
    const r = robots[ri];
    const found = bfsDistancesFrom(r.x, r.y, jobPosSet) as Map<number, number>;
    // Map from encoded pos to distance
    for (let ji = 0; ji < jobCount; ji++) {
      const jpos = enc(jobs[ji].x, jobs[ji].y);
      const d = found.get(jpos);
      if (typeof d === 'number') distRtoJ[ri][ji] = d;
      else distRtoJ[ri][ji] = INF;
    }
  }
} else {
  // BFS from each job to find distances to robots (fewer robots); then fill matrix by transposing
  const robotPosSet = new Set<number>();
  for (const r of robots) robotPosSet.add(enc(r.x, r.y));
  const distJtoR: number[][] = new Array(jobCount);
  for (let ji = 0; ji < jobCount; ji++) {
    const j = jobs[ji];
    const found = bfsDistancesFrom(j.x, j.y, robotPosSet) as Map<number, number>;
    distJtoR[ji] = new Array(robotCount).fill(INF);
    for (let ri = 0; ri < robotCount; ri++) {
      const rpos = enc(robots[ri].x, robots[ri].y);
      const d = found.get(rpos);
      if (typeof d === 'number') distJtoR[ji][ri] = d;
      else distJtoR[ji][ri] = INF;
    }
  }
  // Transpose into distRtoJ
  for (let ri = 0; ri < robotCount; ri++) {
    for (let ji = 0; ji < jobCount; ji++) {
      distRtoJ[ri][ji] = distJtoR[ji][ri];
    }
  }
}

// Assign each job to its nearest robot (ties by robot id)
const assignedJobsPerRobot: number[][] = new Array(robotCount);
for (let ri = 0; ri < robotCount; ri++) assignedJobsPerRobot[ri] = [];

for (let ji = 0; ji < jobCount; ji++) {
  let bestRi = -1;
  let bestDist = INF;
  for (let ri = 0; ri < robotCount; ri++) {
    const d = distRtoJ[ri][ji];
    if (d < bestDist) {
      bestDist = d;
      bestRi = ri;
    } else if (d === bestDist && d !== INF) {
      // tie-break by robot id (original id)
      const ridA = robots[bestRi].id;
      const ridB = robots[ri].id;
      if (ridB < ridA) bestRi = ri;
    }
  }
  // If unreachable by all robots (bestRi stays -1), assign deterministically to robot index 0 if exists else skip
  if (bestRi === -1) {
    if (robotCount > 0) assignedJobsPerRobot[0].push(ji);
    else {
      // No robots: still must output assignments array though; we'll skip
    }
  } else {
    assignedJobsPerRobot[bestRi].push(ji);
  }
}

// For each robot, produce an ordered visiting sequence using greedy nearest-neighbor (on-grid distances).
// We'll compute distances on the fly with BFS from current position to remaining job targets.
for (let ri = 0; ri < robotCount; ri++) {
  const assigned = assignedJobsPerRobot[ri];
  if (assigned.length === 0) {
    // Ensure the output array has an entry for this robot id (we'll fill later)
    continue;
  }
  // We'll keep remaining set of encoded job positions
  const remainingSet = new Set<number>();
  const jobPosByJi = new Map<number, number>();
  for (const ji of assigned) {
    const pos = enc(jobs[ji].x, jobs[ji].y);
    remainingSet.add(pos);
    jobPosByJi.set(pos, ji);
  }
  // Starting position is robot location
  let curX = robots[ri].x;
  let curY = robots[ri].y;
  const orderJi: number[] = [];
  while (remainingSet.size > 0) {
    const distancesFound = bfsDistancesFrom(curX, curY, remainingSet) as Map<number, number>;
    // Choose the closest remaining job; tie-break by job priority desc, then job id
    let chosenPos: number | null = null;
    let chosenDist = INF;
    let chosenJi = -1;
    for (const pos of remainingSet) {
      const d = distancesFound.get(pos);
      const ji = jobPosByJi.get(pos)!;
      const job = jobs[ji];
      const prio = job.priority ?? 0;
      // If BFS didn't reach (d undefined), treat as INF
      const distVal = (typeof d === 'number') ? d : INF;
      if (chosenPos === null) {
        chosenPos = pos;
        chosenDist = distVal;
        chosenJi = ji;
      } else {
        if (distVal < chosenDist) {
          chosenPos = pos;
          chosenDist = distVal;
          chosenJi = ji;
        } else if (distVal === chosenDist) {
          // Tie-break: higher priority first
          const curJob = jobs[chosenJi];
          if ((job.priority ?? 0) > (curJob.priority ?? 0)) {
            chosenPos = pos;
            chosenDist = distVal;
            chosenJi = ji;
          } else if ((job.priority ?? 0) === (curJob.priority ?? 0)) {
            // Finally tie-break by job id (deterministic)
            if (String(job.id) < String(curJob.id)) {
              chosenPos = pos;
              chosenDist = distVal;
              chosenJi = ji;
            }
          }
        }
      }
    }
    if (chosenPos === null) break; // nothing reachable
    // append chosen job index to order
    orderJi.push(chosenJi);
    remainingSet.delete(chosenPos);
    // update current location to chosen job
    curX = jobs[chosenJi].x;
    curY = jobs[chosenJi].y;
  }
  // If some jobs remained unreachable and not appended (rare), append them in deterministic order
  if (remainingSet.size > 0) {
    const rest: number[] = [];
    for (const pos of remainingSet) rest.push(jobPosByJi.get(pos)!);
    rest.sort((a, b) => {
      // higher priority first, then job id string order
      const pa = jobs[a].priority ?? 0;
      const pb = jobs[b].priority ?? 0;
      if (pa !== pb) return pb - pa;
      return String(jobs[a].id).localeCompare(String(jobs[b].id));
    });
    for (const ji of rest) orderJi.push(ji);
  }
  // Save ordered job IDs into assignments at index robots[ri].id
  const rid = robots[ri].id;
  // Ensure assignments array large enough
  if (rid >= assignments.length) {
    for (let k = assignments.length; k <= rid; k++) assignments.push([]);
  }
  assignments[rid] = orderJi.map(ji => jobs[ji].id);
}

// For robot IDs that were not present in input, ensure they are empty arrays (already initialized).
// Also if there are robots in input but some with id < 0 or missing, we've handled robot ids as integer keys.

// Finally: if there are zero robots, we must still output an assignments array of length 0, but every job must appear exactly once.
// The problem statement says array index is robot id; with no robots this is ambiguous. We'll output an array of length 0.
if (robotCount === 0) {
  // If there are jobs but no robots, we must still output something. We'll output a single robot slot (index 0) containing all job ids.
  // This prevents losing jobs. (Deterministic)
  if (jobs.length > 0) {
    const allJobIds = jobs.map(j => j.id);
    const out = { assignments: [allJobIds] };
    process.stdout.write(JSON.stringify(out));
    process.exit(0);
  } else {
    const out = { assignments: [] };
    process.stdout.write(JSON.stringify(out));
    process.exit(0);
  }
}

// Ensure every job appears exactly once in output. If any job missing (shouldn't happen), assign to robot 0.
const seenJobs = new Set<any>();
for (let rid = 0; rid < assignments.length; rid++) {
  const arr = assignments[rid] || [];
  for (const jid of arr) seenJobs.add(jid);
}
for (const j of jobs) {
  if (!seenJobs.has(j.id)) {
    // assign missing job to robot with smallest id (0..)
    let targetRid = 0;
    if (assignments.length === 0) {
      assignments.push([j.id]);
    } else {
      assignments[targetRid].push(j.id);
    }
    seenJobs.add(j.id);
  }
}

// Convert any undefined slots to empty arrays to keep indices consistent
for (let i = 0; i < assignments.length; i++) {
  if (!Array.isArray(assignments[i])) assignments[i] = [];
}

// Output JSON
const output = { assignments: assignments };
process.stdout.write(JSON.stringify(output));
