; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %ubo %outColor
OpExecutionMode %main OriginUpperLeft

OpExtInstImport "GLSL.std.450"

; Names
OpName %main "main"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %ubo "ubo"
OpName %outColor "outColor"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Pointer types
%_ptr_Input_v3f = OpTypePointer Input %v3f
%_ptr_Input_v2f = OpTypePointer Input %v2f
%_ptr_Input_v4f = OpTypePointer Input %v4f
%_ptr_Output_v4f = OpTypePointer Output %v4f
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_v4f = OpTypePointer Uniform %v4f
%_ptr_Uniform_ubo = OpTypePointer Uniform %_type_ubo

; Constant declarations
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_8 = OpConstant %float 8.0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

; UBO struct type (define forward pointer then actual struct to use pointer type above)
%_type_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%_ptr_Uniform_ubo = OpTypePointer Uniform %_type_ubo

; Decorate UBO struct members for std140 layout offsets and matrix columns
OpMemberDecorate %_type_ubo 0 ColMajor
OpMemberDecorate %_type_ubo 0 MatrixStride 16
OpMemberDecorate %_type_ubo 0 Offset 0
OpMemberDecorate %_type_ubo 1 ColMajor
OpMemberDecorate %_type_ubo 1 MatrixStride 16
OpMemberDecorate %_type_ubo 1 Offset 64
OpMemberDecorate %_type_ubo 2 ColMajor
OpMemberDecorate %_type_ubo 2 MatrixStride 16
OpMemberDecorate %_type_ubo 2 Offset 128
OpMemberDecorate %_type_ubo 3 Offset 192
OpMemberDecorate %_type_ubo 4 Offset 208
OpMemberDecorate %_type_ubo 5 Offset 224
OpDecorate %_type_ubo Block

; Inputs
%inWorldPos = OpVariable %_ptr_Input_v3f Input
OpDecorate %inWorldPos Location 0

%inNormal = OpVariable %_ptr_Input_v3f Input
OpDecorate %inNormal Location 1

%inUV = OpVariable %_ptr_Input_v2f Input
OpDecorate %inUV Location 2

%inTangent = OpVariable %_ptr_Input_v3f Input
OpDecorate %inTangent Location 3

%inColor = OpVariable %_ptr_Input_v3f Input
OpDecorate %inColor Location 4

; Uniform block variable
%ubo = OpVariable %_ptr_Uniform_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Output
%outColor = OpVariable %_ptr_Output_v4f Output
OpDecorate %outColor Location 0

; Function main
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %inWorldPos
%normal = OpLoad %v3f %inNormal
%uv = OpLoad %v2f %inUV

; Normalize normal
%norm = OpExtInst %v3f %GLSLstd450 Normalize %normal

; Get light position (member index 3)
%ptr_light = OpAccessChain %_ptr_Uniform_v4f %ubo %int_3
%lightVec4 = OpLoad %v4f %ptr_light
; extract xyz
%light_x = OpCompositeExtract %float %lightVec4 0
%light_y = OpCompositeExtract %float %lightVec4 1
%light_z = OpCompositeExtract %float %lightVec4 2
%lightPos3 = OpCompositeConstruct %v3f %light_x %light_y %light_z

; Compute light direction and normalize
%neg_world = OpFNegate %v3f %worldPos
%lightDir = OpFSub %v3f %lightPos3 %worldPos
%lightDirNorm = OpExtInst %v3f %GLSLstd450 Normalize %lightDir

; Lambert diffuse (clamped)
%dotNL = OpDot %float %norm %lightDirNorm
%diffuse = OpExtInst %float %GLSLstd450 FMax %dotNL %float_0

; lambert factor = 0.5 + 0.5 * diffuse
%halfDiffuse = OpFMul %float %float_0_5 %diffuse
%lambert = OpFAdd %float %float_0_5 %halfDiffuse

; Checkerboard: uv * 8, floor, convert to int, parity by sum & 1
%scaled = OpVectorTimesScalar %v2f %uv %float_8
%floored = OpExtInst %v2f %GLSLstd450 Floor %scaled
%u_f = OpCompositeExtract %float %floored 0
%v_f = OpCompositeExtract %float %floored 1
%u_i = OpConvertFToS %int %u_f
%v_i = OpConvertFToS %int %v_f
%sum_i = OpIAdd %int %u_i %v_i
%parity = OpIAnd %int %sum_i %int_1
%isOdd = OpINotEqual %bool %parity %int_0

; Select checker color 0.9 for odd, 0.2 for even
%checkerVal = OpSelect %float %isOdd %float_0_9 %float_0_2

; Make vec3 checker color
%checkerVec = OpCompositeConstruct %v3f %checkerVal %checkerVal %checkerVal

; final RGB = checkerVec * lambert
%finalRGB = OpVectorTimesScalar %v3f %checkerVec %lambert

; Compose vec4 with alpha 1.0
%finalColor = OpCompositeConstruct %v4f %finalRGB %float_1

; Store to output
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd
