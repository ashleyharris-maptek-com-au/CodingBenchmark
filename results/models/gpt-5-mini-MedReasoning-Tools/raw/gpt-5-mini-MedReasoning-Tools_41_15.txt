; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %100 "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %uboVar
OpExecutionMode %100 OriginUpperLeft
OpExtInstImport %1 "GLSL.std.450"

OpName %100 "main"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %outColor "outColor"
OpName %uboVar "uboVar"
OpMemberName %_struct_ubo 0 "model"
OpMemberName %_struct_ubo 1 "view"
OpMemberName %_struct_ubo 2 "projection"
OpMemberName %_struct_ubo 3 "lightPos"
OpMemberName %_struct_ubo 4 "cameraPos"
OpMemberName %_struct_ubo 5 "params"
OpName %_struct_ubo "UBO"

; Decorations for inputs/outputs/uniforms
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %_struct_ubo Block
OpMemberDecorate %_struct_ubo 0 Offset 0
OpMemberDecorate %_struct_ubo 1 Offset 64
OpMemberDecorate %_struct_ubo 2 Offset 128
OpMemberDecorate %_struct_ubo 3 Offset 192
OpMemberDecorate %_struct_ubo 4 Offset 208
OpMemberDecorate %_struct_ubo 5 Offset 224

; Types
%2 = OpTypeVoid
%3 = OpTypeFunction %2
%4 = OpTypeFloat 32
%5 = OpTypeVector %4 2
%6 = OpTypeVector %4 3
%7 = OpTypeVector %4 4
%8 = OpTypeMatrix %7 4
%_struct_ubo = OpTypeStruct %8 %8 %8 %7 %7 %7
%9 = OpTypePointer Input %6
%10 = OpTypePointer Input %5
%11 = OpTypePointer Output %7
%12 = OpTypeInt 32 0
%13 = OpConstant %12 0
%14 = OpConstant %12 3
%15 = OpTypePointer Uniform %_struct_ubo
%16 = OpTypePointer Uniform %7
%17 = OpTypePointer Uniform %6

; Constants (floats)
%f_0_5 = OpConstant %4 0.5
%f_1 = OpConstant %4 1.0
%f_0 = OpConstant %4 0.0
%f_0_2 = OpConstant %4 0.2
%f_0_75 = OpConstant %4 0.75
%f_0_7 = OpConstant %4 0.7
%f_0_4 = OpConstant %4 0.4

; Constant vectors for cool/warm
%cool_const = OpConstantComposite %6 %f_0_2 %f_0_2 %f_0_75
%warm_const = OpConstantComposite %6 %f_0_7 %f_0_7 %f_0_4

; Variables (inputs, output, uniform)
%inWorldPos = OpVariable %9 Input
%inNormal = OpVariable %9 Input
%inUV = OpVariable %10 Input
%inTangent = OpVariable %9 Input
%inColor = OpVariable %9 Input
%outColor = OpVariable %11 Output
%uboVar = OpVariable %15 Uniform

; Function main
%100 = OpFunction %2 None %3
%101 = OpLabel

; Load interpolated inputs
%wp = OpLoad %6 %inWorldPos
%n_in = OpLoad %6 %inNormal
%vcolor = OpLoad %6 %inColor

; Normalize normal
%n = OpExtInst %6 %1 Normalize %n_in

; Load light position from UBO (member 3) and make vec3
%light_ptr = OpAccessChain %16 %uboVar %14
%light_v4 = OpLoad %7 %light_ptr
%light = OpVectorShuffle %6 %light_v4 %light_v4 0 1 2

; Compute light direction and normalize
%Lpre = OpFSub %6 %light %wp
%L = OpExtInst %6 %1 Normalize %Lpre

; Dot(N,L)
%ndotl = OpExtInst %4 %1 Dot %n %L

; t = ndotl * 0.5 + 0.5
%ndotl_mul = OpFMul %4 %ndotl %f_0_5
%t = OpFAdd %4 %ndotl_mul %f_0_5

; oneMinus = 1.0 - t
%oneMinus = OpFSub %4 %f_1 %t

; Splat scalars to vec3
%t_vec = OpCompositeConstruct %6 %t %t %t
%oneMinus_vec = OpCompositeConstruct %6 %oneMinus %oneMinus %oneMinus

; mix = cool*(1-t) + warm*t
%cool_part = OpFMul %6 %cool_const %oneMinus_vec
%warm_part = OpFMul %6 %warm_const %t_vec
%mix_rgb = OpFAdd %6 %cool_part %warm_part

; Modulate by vertex color
%final_rgb = OpFMul %6 %mix_rgb %vcolor

; Build vec4(outColor) with alpha = 1.0
%c0 = OpCompositeExtract %4 %final_rgb 0
%c1 = OpCompositeExtract %4 %final_rgb 1
%c2 = OpCompositeExtract %4 %final_rgb 2
%outv = OpCompositeConstruct %7 %c0 %c1 %c2 %f_1

; Store to output
OpStore %outColor %outv

OpReturn
OpFunctionEnd
