; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %uboVar
OpExecutionMode %main OriginUpperLeft
OpExtInstImport %GLSL "GLSL.std.450"
OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %in_tangent "tangent"
OpName %in_color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %uboVar "ubo"
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%v4_4 = OpTypeMatrix %v4 4

%UBO = OpTypeStruct %v4_4 %v4_4 %v4_4 %v4 %v4 %v4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO

%_ptr_Input_v3 = OpTypePointer Input %v3
%_ptr_Input_v2 = OpTypePointer Input %v2
%_ptr_Output_v4 = OpTypePointer Output %v4

; Uniform variable
%uboVar = OpVariable %_ptr_Uniform_UBO Uniform

; Input variables
%in_worldPos = OpVariable %_ptr_Input_v3 Input
%in_normal = OpVariable %_ptr_Input_v3 Input
%in_uv = OpVariable %_ptr_Input_v2 Input
%in_tangent = OpVariable %_ptr_Input_v3 Input
%in_color = OpVariable %_ptr_Input_v3 Input

; Output variable
%outColor = OpVariable %_ptr_Output_v4 Output

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_half = OpConstant %float 0.5
%float_0p1 = OpConstant %float 0.1
%float_0p4 = OpConstant %float 0.4

%const_bottom = OpConstantComposite %v3 %float_0p1 %float_0p4 %float_0p1
%const_top = OpConstantComposite %v3 %float_1 %float_1 %float_1

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel
%wp = OpLoad %v3 %in_worldPos
%wy = OpCompositeExtract %float %wp 1
%mul = OpFMul %float %wy %float_half
%add = OpFAdd %float %mul %float_half
%t = OpExtInst %float %GLSL FClamp %add %float_0 %float_1

%t_vec = OpCompositeConstruct %v3 %t %t %t
%diff = OpFSub %v3 %const_top %const_bottom
%scaled = OpVectorTimesScalar %v3 %diff %t
%color_v3 = OpFAdd %v3 %const_bottom %scaled

%c0 = OpCompositeExtract %float %color_v3 0
%c1 = OpCompositeExtract %float %color_v3 1
%c2 = OpCompositeExtract %float %color_v3 2
%color_v4 = OpCompositeConstruct %v4 %c0 %c1 %c2 %float_1

OpStore %outColor %color_v4
OpReturn
OpFunctionEnd
