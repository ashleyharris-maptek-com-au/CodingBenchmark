#version 450

layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal (interpolated, may need normalize)
layout(location = 2) in vec2 uv;         // texture coordinates in [0,1]
layout(location = 3) in vec3 tangent;    // tangent vector
layout(location = 4) in vec3 color;      // per-vertex color

layout(location = 0) out vec4 outColor;  // RGBA float, stored as rgba8unorm

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

void main()
{
    // Brick pattern parameters
    const vec3 brickColor = vec3(0.7, 0.2, 0.1);
    const vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    const float mortarX = 0.05; // fraction of brick width
    const float mortarY = 0.1;  // fraction of brick height
    const vec2 scaleUV = vec2(8.0, 16.0); // 8x16 UV scale

    // Build brick cell coordinates with staggered rows (half-brick offset every other row)
    vec2 scaled = uv * scaleUV;
    float row = floor(scaled.y);
    float rowMod = mod(row, 2.0);
    // Apply half-brick offset on odd rows
    if (rowMod > 0.5) {
        scaled.x += 0.5;
    }
    // Local cell coordinates in [0,1]
    vec2 cell = fract(scaled);

    // Compute derivative-based smoothing amounts to anti-alias mortar edges
    float ax = max(fwidth(scaled.x), 0.0001);
    float ay = max(fwidth(scaled.y), 0.0001);

    // Vertical mortar mask (left or right edges of the cell)
    float leftMask  = 1.0 - smoothstep(mortarX - ax, mortarX + ax, cell.x);
    float rightMask = smoothstep(1.0 - mortarX - ax, 1.0 - mortarX + ax, cell.x);
    float verticalMortar = max(leftMask, rightMask);

    // Horizontal mortar mask (top or bottom edges of the cell)
    float bottomMask = 1.0 - smoothstep(mortarY - ay, mortarY + ay, cell.y);
    float topMask    = smoothstep(1.0 - mortarY - ay, 1.0 - mortarY + ay, cell.y);
    float horizontalMortar = max(bottomMask, topMask);

    // Combined mortar mask (1.0 = mortar, 0.0 = brick)
    float mortarMask = clamp(max(verticalMortar, horizontalMortar), 0.0, 1.0);

    // Base color based on mortar/brick
    vec3 baseColor = mix(brickColor, mortarColor, mortarMask);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse; // provided factor

    vec3 shaded = baseColor * shade;

    outColor = vec4(shaded, 1.0);
}
