#version 450

layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal (interpolated, may need normalize)
layout(location = 2) in vec2 uv;         // texture coordinates in [0,1]
layout(location = 3) in vec3 tangent;    // tangent vector
layout(location = 4) in vec3 color;      // per-vertex color

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Light vector (point light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // View vector (from fragment to camera)
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

    // Ambient term
    vec3 ambient = vec3(0.1, 0.05, 0.05);

    // Diffuse term: use vertex color modulated by Lambertian cosine
    float NdotL = max(dot(N, L), 0.0);
    vec3 diffuse = color * NdotL;

    // Specular term: Phong using reflection vector R = reflect(-L, N)
    vec3 R = reflect(-L, N);
    float specFactor = pow(max(dot(R, V), 0.0), 32.0);
    vec3 specular = vec3(1.0) * specFactor;

    // Sum components and clamp to [0,1]
    vec3 result = clamp(ambient + diffuse + specular, 0.0, 1.0);

    outColor = vec4(result, 1.0);
}
