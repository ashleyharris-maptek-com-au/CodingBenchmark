struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // View vector (from surface point to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Light vector (from surface point to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);

    // Rim (Fresnel-like): pow(1 - max(dot(N,V),0), 3)
    float NdotV = max(dot(N, V), 0.0);
    float rim = pow(1.0 - NdotV, 3.0);

    // Base color modulated by (0.3 + 0.5*diffuse)
    float3 baseColor = float3(0.2, 0.3, 0.8);
    float3 litBase = baseColor * (0.3 + 0.5 * diffuse);

    // Rim tint
    float3 rimTint = float3(1.0, 0.8, 0.5);

    float3 color = litBase + rimTint * rim;
    color = clamp(color, 0.0, 1.0);

    return float4(color, 1.0);
}
