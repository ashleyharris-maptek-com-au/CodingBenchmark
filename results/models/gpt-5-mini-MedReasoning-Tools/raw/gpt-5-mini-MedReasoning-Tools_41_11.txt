; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %uboVar
OpExecutionMode %main OriginUpperLeft
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool

; vectors and matrices
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%f_0 = OpConstant %float 0.0
%f_0_1 = OpConstant %float 0.1
%f_0_3 = OpConstant %float 0.3
%f_0_5 = OpConstant %float 0.5
%f_0_7 = OpConstant %float 0.7
%f_0_8 = OpConstant %float 0.8
%f_1 = OpConstant %float 1.0
%f_6 = OpConstant %float 6.0

; Vector constants
%v2_center = OpConstantComposite %v2f %f_0_5 %f_0_5
%yellow_v3 = OpConstantComposite %v3f %f_1 %f_0_8 %f_0
%bg_v3 = OpConstantComposite %v3f %f_0_1 %f_0_1 %f_0_3
%alpha_v4 = OpConstantComposite %v4f %f_1 %f_1 %f_1 %f_1
%yellow_v4 = OpConstantComposite %v4f %f_1 %f_0_8 %f_0 %f_1
%bg_v4 = OpConstantComposite %v4f %f_0_1 %f_0_1 %f_0_3 %f_1

; Pointer types
%ptr_input_v3 = OpTypePointer Input %v3f
%ptr_input_v2 = OpTypePointer Input %v2f
%ptr_input_v4 = OpTypePointer Input %v4f
%ptr_output_v4 = OpTypePointer Output %v4f
%ptr_uniform_v4 = OpTypePointer Uniform %v4f

; Uniform buffer struct type
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct

; Variables (inputs, outputs, uniform)
%inWorldPos = OpVariable %ptr_input_v3 Input
%inNormal = OpVariable %ptr_input_v3 Input
%inUV = OpVariable %ptr_input_v2 Input
%inTangent = OpVariable %ptr_input_v3 Input
%inColor = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%uboVar = OpVariable %ptr_uniform_ubo Uniform

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224

; Function main
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load inputs
%uv = OpLoad %v2f %inUV
%wpos = OpLoad %v3f %inWorldPos
%n_in = OpLoad %v3f %inNormal

; uv * 6.0
%uv6 = OpFMul %v2f %uv (OpConstantComposite %v2f %f_6 %f_6)

; fract(uv6)
%cell = OpExtInst %v2f %glsl Fract %uv6

; diff = cell - center
%diff2 = OpFSub %v2f %cell %v2_center

; dist = length(diff)
%dist = OpExtInst %float %glsl Length %diff2

; compare dist <= 0.3
%cond = OpFOrdLessThanEqual %bool %dist %f_0_3

; choose color between yellow and background
%baseColor = OpSelect %v4f %cond %yellow_v4 %bg_v4

; Load light position from UBO (member index 3)
%light_ptr = OpAccessChain %ptr_uniform_v4 %uboVar %int_3
%light_v4 = OpLoad %v4f %light_ptr
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

; compute light direction = normalize(light_xyz - worldPos)
%ldiff = OpFSub %v3f %light_xyz %wpos
%ldir = OpExtInst %v3f %glsl Normalize %ldiff

; normalize normal
%n = OpExtInst %v3f %glsl Normalize %n_in

; diffuse = max(dot(n, ldir), 0.0)
%dotnl = OpExtInst %float %glsl Dot %n %ldir
%diff_clamped = OpExtInst %float %glsl Max %dotnl %f_0

; lambert = 0.3 + 0.7 * diffuse
%tmp_mul = OpFMul %float %f_0_7 %diff_clamped
%lambert = OpFAdd %float %f_0_3 %tmp_mul

; replicate lambert to vec3
%lambert_v3 = OpCompositeConstruct %v3f %lambert %lambert %lambert

; extract base rgb
%base_rgb = OpVectorShuffle %v3f %baseColor %baseColor 0 1 2

; final rgb = base_rgb * lambert_v3
%final_rgb = OpFMul %v3f %base_rgb %lambert_v3

; compose final vec4 (alpha = 1.0)
%final_color = OpCompositeConstruct %v4f %final_rgb.x %final_rgb.y %final_rgb.z %f_1

; store to output
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
