Below is a single-file, self-contained Python autopilot implementation that will run inside the simulated environment you described. It uses simple but robust logic, sensor voting (median-of-valids), PID-like controllers, mode/state management across calls (module-level persistence), and explicit go-around handling. It is written for a simulator (not for real aircraft) — do not use for actual flight operations.

Paste this into your simulation environment. Module-level variables (AP_STATE) persist across calls within a scenario and are reset between scenarios and go-around retries per your environment description.

```python
# Autoland autopilot for the described simulator.
# Usage: call autopilot_step(state: dict, dt: float) -> dict repeatedly.
# WARNING: For simulation/testing only. Not safe or certified for real aircraft.

from math import tan, radians

# Module-level persistent state (will persist across calls within a scenario)
AP_STATE = {
    "phase": "intercept",        # intercept / approach / flare / touchdown / rollout / goaround
    "go_around_count": 0,
    "ias_int": 0.0,
    "gs_int": 0.0,
    "vs_int": 0.0,
    "last_radio_alt": None,
    "last_time": 0.0,
    "last_alt_ft": None,
    "last_vs_fpm": 0.0,
    "integrator_limits": {"ias": 50.0, "gs": 500.0, "vs": 1000.0},
    "prev_loc_err": 0.0,
    "prev_gs_err": 0.0,
    "time_since_touch": 0.0,
    "go_around_active": False,
    "flap_target": 0.0,
    "gear_target": 0.0,
    "throttle_cmd": 0.0,
    "last_outputs": {},
}

# Helpers: robust median-of-valids for triple-redundant sensors where each sensor has an associated valid flag
def median_of_valids(vals, valids=None):
    """
    vals: list of numeric sensor readings (len 1-3)
    valids: optional list of booleans of same length indicating validity (True means valid)
    Returns median of valid readings, or None if no valids. If valids is None assume all valid.
    """
    if valids is None:
        valid_vals = [v for v in vals if v is not None]
    else:
        valid_vals = [v for v, ok in zip(vals, valids) if ok and v is not None]
    if not valid_vals:
        return None
    # median
    s = sorted(valid_vals)
    n = len(s)
    return s[n // 2] if n % 2 == 1 else 0.5 * (s[n//2 - 1] + s[n//2])

def choose_first_valid(vals, valids=None):
    if valids is None:
        for v in vals:
            if v is not None:
                return v
    else:
        for v, ok in zip(vals, valids):
            if ok and v is not None:
                return v
    return None

def clamp(x, a, b):
    return max(a, min(b, x))

def safe_div(a, b, default=0.0):
    try:
        return a / b
    except Exception:
        return default

def get_bool(state, key):
    v = state.get(key)
    return bool(v)

# Main autopilot step
def autopilot_step(state: dict, dt: float) -> dict:
    global AP_STATE

    # ---------- Read robust/aggregated sensors ----------
    # IAS
    ias = median_of_valids(
        [state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")],
        [state.get("ias_1_valid"), state.get("ias_2_valid"), state.get("ias_3_valid")]
    )
    # If no valid IAS, fallback to IRS/GPS groundspeed as approximate airspeed (not ideal)
    if ias is None:
        ias = median_of_valids(
            [state.get("irs_1_gs_kt"), state.get("irs_2_gs_kt"), state.get("irs_3_gs_kt")],
            [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")]
        )
    if ias is None:
        ias = state.get("target_ias_kt", 140.0)

    # Radio altitude (single sensor)
    radio_alt = state.get("radio_alt_ft", None)

    # Barometric altitude (median of 3)
    baro_alt = median_of_valids(
        [state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")],
        [state.get("baro_alt_1_valid"), state.get("baro_alt_2_valid"), state.get("baro_alt_3_valid")]
    )

    # Vertical speed from IRS if available (median)
    irs_vs = median_of_valids(
        [state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm")],
        [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")]
    )
    vs_fpm = irs_vs if irs_vs is not None else AP_STATE["last_vs_fpm"]

    # Localizer and glideslope (median-of-valids)
    loc_err = median_of_valids(
        [state.get("localizer_1_dots"), state.get("localizer_2_dots"), state.get("localizer_3_dots")],
        [state.get("localizer_1_valid"), state.get("localizer_2_valid"), state.get("localizer_3_valid")]
    )
    gs_err = median_of_valids(
        [state.get("glideslope_1_dots"), state.get("glideslope_2_dots"), state.get("glideslope_3_dots")],
        [state.get("glideslope_1_valid"), state.get("glideslope_2_valid"), state.get("glideslope_3_valid")]
    )

    # Groundspeed (sim provides groundspeed_kt in ground group)
    groundspeed = state.get("groundspeed_kt", None)
    # If not present, use IRS median GS
    if groundspeed is None:
        groundspeed = median_of_valids(
            [state.get("irs_1_gs_kt"), state.get("irs_2_gs_kt"), state.get("irs_3_gs_kt")],
            [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")]
        )
    if groundspeed is None:
        groundspeed = max(1.0, ias)  # fallback

    # Weight on wheels (touchdown)
    wow = bool(state.get("weight_on_wheels", False) or state.get("on_ground", False) or state.get("below_decision_height", False) and state.get("radio_alt_ft", 9999) < 2)

    # Engines running
    eng1_run = bool(state.get("engine_1_running", True))
    eng2_run = bool(state.get("engine_2_running", True))

    # Targets and setpoints
    target_vref = state.get("target_vref_kt", 137.0)
    target_ias = state.get("target_ias_kt", target_vref + 5.0)
    decision_height = 200.0  # CAT I
    flare_height = 50.0

    # Update last knowns
    if radio_alt is not None:
        AP_STATE["last_radio_alt"] = radio_alt
    AP_STATE["last_vs_fpm"] = vs_fpm
    AP_STATE["last_alt_ft"] = baro_alt if baro_alt is not None else AP_STATE["last_alt_ft"]

    # Phase transitions based on altitude and state
    phase = AP_STATE["phase"]
    if AP_STATE["go_around_active"]:
        phase = "goaround"
        AP_STATE["phase"] = "goaround"
    else:
        if wow:
            # landed
            phase = "touchdown" if AP_STATE["phase"] != "rollout" else AP_STATE["phase"]
            AP_STATE["phase"] = "touchdown"
        else:
            # decide by radio altitude if available
            ra = radio_alt if radio_alt is not None else (baro_alt if baro_alt is not None else 99999.0)
            if ra <= flare_height:
                phase = "flare"
            elif ra <= 1000.0:
                phase = "approach"
            else:
                phase = "intercept"
            AP_STATE["phase"] = phase

    # Reset integrators if go-around triggered or new approach start
    if phase == "intercept":
        # gentle trim/integral decay
        AP_STATE["ias_int"] *= 0.95
        AP_STATE["gs_int"] *= 0.95
        AP_STATE["vs_int"] *= 0.95

    # ---------- Safety / go-around detection ----------
    unstable = False
    # Criteria for unstable approach below 1000 ft AGL
    ra_for_check = radio_alt if radio_alt is not None else baro_alt if baro_alt is not None else 99999.0
    # Use thresholds (tunable)
    if ra_for_check <= 1000.0:
        if loc_err is not None and abs(loc_err) > 1.2:  # big localizer deviation
            unstable = True
        if gs_err is not None and abs(gs_err) > 0.8:   # large glideslope deviation
            unstable = True
        if abs(ias - target_ias) > 12.0:
            unstable = True
        if vs_fpm is not None and vs_fpm < -1200 and ra_for_check < 500.0:
            unstable = True
    # Engine failure -> go around
    if not eng1_run or not eng2_run:
        unstable = True

    # If below decision height and unstable -> go-around
    go_around = False
    if (ra_for_check <= decision_height and unstable) or AP_STATE["go_around_active"]:
        go_around = True

    # If user-specified missed approach or explicit condition (not provided) - we use above
    if go_around and not AP_STATE["go_around_active"]:
        AP_STATE["go_around_active"] = True
        AP_STATE["go_around_count"] += 1
        # reset integrators aggressively
        AP_STATE["ias_int"] = 0.0
        AP_STATE["gs_int"] = 0.0
        AP_STATE["vs_int"] = 0.0

    # If go-around count > 10, leave autopilot in safe mode (cut throttle to idle) - simulator will mark fail
    if AP_STATE["go_around_count"] > 10:
        # Produce a safe, non-operational output
        return {
            "throttle_1": 0.0,
            "throttle_2": 0.0,
            "elevator": 0.0,
            "aileron": 0.0,
            "rudder": 0.0,
            "flaps": AP_STATE.get("flap_target", 25.0),
            "gear": 1.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "autobrake": 0.0,
        }

    # ---------- Lateral control (localizer) ----------
    # If localizer available, steer to zero dots. Otherwise use target_hdg_deg if provided.
    hdg_target = state.get("target_hdg_deg", None)
    # Gains (tunable)
    loc_gain = 0.6  # how aggressively to command roll from localizer dots -> degrees
    max_roll_deg = 18.0
    if loc_err is not None:
        desired_roll_deg = clamp(-loc_err * 14.0 * loc_gain, -max_roll_deg, max_roll_deg)
    else:
        # Use heading tracking fallback: compute heading error
        heading = median_of_valids(
            [state.get("irs_1_heading_deg"), state.get("irs_2_heading_deg"), state.get("irs_3_heading_deg")],
            [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")]
        )
        if heading is None or hdg_target is None:
            desired_roll_deg = 0.0
        else:
            # shortest difference
            err = ((hdg_target - heading + 180) % 360) - 180
            desired_roll_deg = clamp(err * 0.4, -max_roll_deg, max_roll_deg)

    aileron_cmd = clamp(desired_roll_deg / max_roll_deg, -1.0, 1.0)

    # Rudder: small assist to maintain yaw/directional control; use localizer yaw correction if crosswind present
    rudder_cmd = 0.0
    # If aileron is used heavily, apply small rudder in same sign to coordinate (simple coordination)
    rudder_cmd = clamp(aileron_cmd * 0.25, -1.0, 1.0)

    # ---------- Vertical control (glideslope) ----------
    # Compute nominal 3-degree descent rate for current groundspeed
    # 1 kt = 101.27 ft/min; vertical speed = -gs_kt * 101.27 * tan(3deg)
    tan3 = tan(radians(3.0))
    nominal_vs = -groundspeed * 101.27 * tan3  # negative = descent fpm

    # If glideslope available, use it to bias desired_vs. Each dot shifts desired_vs by a gain factor.
    if gs_err is not None:
        # GS error positive = above (per your doc), so need to descend more if above (positive error)
        desired_vs = nominal_vs - gs_err * 450.0  # tune: 1 dot -> 450 fpm correction
    else:
        # if no GS, try to capture by target vertical speed based on distance (if DME available)
        # fallback to nominal_vs
        desired_vs = nominal_vs

    # Add safety: limit descent rate
    desired_vs = clamp(desired_vs, -2200.0, 200.0)

    # VS controller (PD-ish)
    Kp_vs = 0.0018  # maps fpm error to elevator command (tunable)
    Kd_vs = 0.0008
    vs_error = (desired_vs - (vs_fpm if vs_fpm is not None else AP_STATE["last_vs_fpm"]))
    d_vs_error = (vs_error - (AP_STATE.get("prev_vs_err", 0.0))) / max(dt, 1e-6)
    AP_STATE["prev_vs_err"] = vs_error
    elevator_from_vs = clamp(Kp_vs * vs_error + Kd_vs * d_vs_error, -0.9, 0.9)

    # Also include pitch trim towards modest pitch angle baseline to maintain speed
    # Speed error influences pitch trim/integrator: reduce speed by pitching up, increase by pitching down
    speed_error = target_ias - ias
    AP_STATE["ias_int"] += speed_error * dt * 0.02
    AP_STATE["ias_int"] = clamp(AP_STATE["ias_int"], -AP_STATE["integrator_limits"]["ias"], AP_STATE["integrator_limits"]["ias"])
    pitch_trim_from_speed = clamp(0.002 * speed_error + 0.0002 * AP_STATE["ias_int"], -0.6, 0.6)

    # Combine to form elevator command (relative to trim)
    elevator_cmd = clamp(elevator_from_vs + pitch_trim_from_speed, -1.0, 1.0)

    # ---------- Speed / thrust management ----------
    # Simple PI throttle to maintain target_ias (with schedule for approach/flare)
    throttle = AP_STATE.get("throttle_cmd", 0.5)
    # Approach target speed is target_ias; add small margin above VREF
    targ_speed = target_ias

    # When flaring (< flare_height) throttle commanded to idle (to allow flare) unless go-around
    if AP_STATE["go_around_active"]:
        throttle = 1.0
    elif phase == "flare":
        # idle thrust during flare
        throttle = 0.05
    elif phase == "touchdown" or phase == "rollout":
        # after touchdown, engines may idle and reverse applied separately
        throttle = 0.05
    else:
        # PI control (tunable)
        Kp_th = 0.015
        Ki_th = 0.004
        throttle += (Kp_th * (targ_speed - ias) + Ki_th * AP_STATE["ias_int"]) * dt
        throttle = clamp(throttle, 0.0, 1.0)

    AP_STATE["throttle_cmd"] = throttle

    # ---------- Configuration (flaps, gear, speedbrake) ----------
    flap_target = AP_STATE.get("flap_target", 15.0)
    gear_target = AP_STATE.get("gear_target", 0.0)
    # Flaps schedule: intercept >1000ft -> 15, approach <1000 -> 30, final <200 -> 40 (if available)
    if phase == "intercept":
        flap_target = 15.0
    elif phase == "approach":
        flap_target = 30.0
    elif phase == "flare":
        flap_target = 40.0
    elif phase in ("touchdown", "rollout"):
        flap_target = 40.0
    elif AP_STATE["go_around_active"]:
        # go-around initial: flaps to go-around setting (15)
        flap_target = 15.0

    # Gear: lower when below 1000ft and on approach, or in touchdown/rollout
    if phase in ("approach", "flare", "touchdown", "rollout"):
        gear_target = 1.0
    elif AP_STATE["go_around_active"]:
        # Raise gear after positive climb (use radio_alt increase)
        # If radio alt known and climbing above 400ft, gear up
        ra = radio_alt if radio_alt is not None else baro_alt if baro_alt is not None else 99999.0
        if ra > 400.0:
            gear_target = 0.0
        else:
            gear_target = 1.0
    else:
        gear_target = 0.0

    AP_STATE["flap_target"] = flap_target
    AP_STATE["gear_target"] = gear_target

    # Speedbrake: retracted in approach, can be used after touchdown
    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    # ---------- Flare, touchdown, rollout, and go-around logic ----------
    outputs = {}
    # Default outputs
    outputs["aileron"] = aileron_cmd
    outputs["rudder"] = rudder_cmd
    outputs["elevator"] = elevator_cmd
    outputs["elevator_trim"] = clamp(pitch_trim_from_speed, -1.0, 1.0)
    outputs["flaps"] = clamp(flap_target, 0.0, 40.0)
    outputs["gear"] = 1.0 if gear_target >= 0.5 else 0.0

    # Throttle split with engine failure handling
    if not eng1_run and not eng2_run:
        # both engines failed -> idle, try glide (no thrust)
        outputs["throttle_1"] = 0.0
        outputs["throttle_2"] = 0.0
    elif not eng1_run:
        outputs["throttle_1"] = 0.0
        outputs["throttle_2"] = throttle
    elif not eng2_run:
        outputs["throttle_1"] = throttle
        outputs["throttle_2"] = 0.0
    else:
        outputs["throttle_1"] = throttle
        outputs["throttle_2"] = throttle

    # Flare behavior
    if phase == "flare" and not AP_STATE["go_around_active"]:
        # Reduce descent rate progressively by commanding pitch-up (elevator) and idling thrust
        # Aim to reduce sink to ~ -100 fpm before touchdown
        # Use a small elevator command bias
        elevator_flare = 0.25 + 0.5 * clamp( (flare_height - (radio_alt if radio_alt is not None else flare_height)) / max(flare_height,1.0), 0.0, 1.0)
        # Reduce aggressiveness if vertical speed already low
        if vs_fpm is not None and vs_fpm > -300.0:
            elevator_flare *= 0.5
        outputs["elevator"] = clamp(outputs["elevator"] + elevator_flare, -1.0, 1.0)
        # Throttle kept near idle (already set above)
        outputs["throttle_1"] = clamp(outputs["throttle_1"], 0.0, 0.15)
        outputs["throttle_2"] = clamp(outputs["throttle_2"], 0.0, 0.15)

    # Touchdown detection: either weight_on_wheels or radio_alt < 1.0 and small vs
    touchdown_detected = False
    if wow:
        touchdown_detected = True
    else:
        if radio_alt is not None:
            # If radio alt almost zero and descent nearing zero, consider touchdown
            if radio_alt <= 1.0 and (vs_fpm is None or abs(vs_fpm) < 400):
                touchdown_detected = True

    if touchdown_detected:
        # Enter rollout
        AP_STATE["phase"] = "rollout"
        # Deploy ground devices
        spoiler = 1.0
        autobrake = 1.0
        # Cut engines to idle and apply reverse if engines available and speed > 40 kt
        groundspeed_now = groundspeed
        if eng1_run:
            reverse_1 = 0.6 if groundspeed_now > 40.0 else 0.0
        if eng2_run:
            reverse_2 = 0.6 if groundspeed_now > 40.0 else 0.0
        outputs["spoiler"] = 1.0
        outputs["autobrake"] = 1.0
        outputs["reverse_1"] = reverse_1
        outputs["reverse_2"] = reverse_2
        outputs["throttle_1"] = clamp(outputs["throttle_1"], 0.0, 0.15)
        outputs["throttle_2"] = clamp(outputs["throttle_2"], 0.0, 0.15)
        outputs["flaps"] = clamp(flap_target, 0.0, 40.0)
        outputs["gear"] = 1.0
        # Return rollout outputs
        AP_STATE["time_since_touch"] = AP_STATE.get("time_since_touch", 0.0) + dt
        return outputs

    # Go-around handling
    if AP_STATE["go_around_active"]:
        # Execute go-around: TOGA thrust, pitch up, flaps to go-around setting, retract speedbrake
        outputs["throttle_1"] = 1.0 if eng1_run else 0.0
        outputs["throttle_2"] = 1.0 if eng2_run else 0.0
        outputs["elevator"] = clamp(0.6, -1.0, 1.0)  # pitch up command
        outputs["flaps"] = 15.0
        outputs["gear"] = 1.0  # keep gear down until positive climb and safe altitude for retract
        outputs["speedbrake"] = 0.0
        outputs["spoiler"] = 0.0
        outputs["autobrake"] = 0.0
        outputs["reverse_1"] = 0.0
        outputs["reverse_2"] = 0.0

        # Determine when to stop go-around: climb to 1500ft AGL
        # Use radio_alt if available, otherwise baro_alt is ambiguous; prefer radio
        if radio_alt is not None and radio_alt >= 1500.0:
            # end go-around and return to intercept/approach mode
            AP_STATE["go_around_active"] = False
            AP_STATE["phase"] = "intercept"
            # reset integrators so next approach starts clean
            AP_STATE["ias_int"] = 0.0
            AP_STATE["gs_int"] = 0.0
            AP_STATE["vs_int"] = 0.0
        return outputs

    # If not touchdown or go-around, normal approach/intercept outputs:
    outputs["throttle_1"] = clamp(outputs["throttle_1"], 0.0, 1.0)
    outputs["throttle_2"] = clamp(outputs["throttle_2"], 0.0, 1.0)
    outputs["speedbrake"] = 0.0
    outputs["spoiler"] = 0.0
    outputs["autobrake"] = 0.0
    outputs["reverse_1"] = 0.0
    outputs["reverse_2"] = 0.0

    # Update previous errors for filters
    AP_STATE["prev_loc_err"] = loc_err if loc_err is not None else AP_STATE["prev_loc_err"]
    AP_STATE["prev_gs_err"] = gs_err if gs_err is not None else AP_STATE["prev_gs_err"]

    # Save and return
    AP_STATE["last_outputs"] = outputs.copy()
    return outputs
```

Notes and tuning guidance
- The code uses simple median voting and fallbacks for missing/failed sensors.
- Controller gains and thresholds are intentionally conservative and should be tuned against your simulator's dynamics.
- Go-around logic is conservative: poor alignment, large airspeed deviations, high sink rates, or engine failures near decision height trigger a go-around.
- The autopilot attempts to flare and idle thrust for touchdown; on touchdown it deploys spoilers, autobrake, and reversers.
- This is a simulation autopilot. Do not deploy or use this code on real aircraft.
