OpCapability Shader
OpCapability Geometry
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; Entry point and interface (list ALL interface variables)
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; -------------------------
; Types, constants, pointers
; -------------------------
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_3 = OpConstant %float 3.0
%float_third = OpConstant %float 0.33333334
%float_off = OpConstant %float 0.08

; pointer types for Input/Output accesses
%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v4_out = OpTypePointer Output %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v3_out = OpTypePointer Output %v3float

; -------------------------
; gl_PerVertex (input)
; -------------------------
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; -------------------------
; user inputs (arrays, per-vertex)
; layout(location = X) in vec3 ...
; -------------------------
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3

%in_worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; -------------------------
; gl_PerVertex (output)
; -------------------------
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; -------------------------
; Simple UBO (set=0 binding=0) - declared but unused
; (keeps a compatible uniform available)
; -------------------------
%mat4 = OpTypeMatrix %v4float 4
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float %v4float
OpDecorate %ubo_t Block
OpMemberDecorate %ubo_t 0 Offset 0
OpMemberDecorate %ubo_t 1 Offset 64
OpMemberDecorate %ubo_t 2 Offset 128
OpMemberDecorate %ubo_t 3 Offset 192
OpMemberDecorate %ubo_t 4 Offset 208
OpMemberDecorate %ubo_t 5 Offset 224
OpMemberDecorate %ubo_t 6 Offset 240
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; -------------------------
; Main function
; -------------------------
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load the three input positions: gl_in[0..2].gl_Position
%p0_ptr = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_v4_in %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; extract xyz as vec3
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; center = (p0 + p1 + p2) / 3.0
%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum = OpFAdd %v3float %sum01 %p2_xyz
%center = OpVectorTimesScalar %v3float %sum %float_third

; e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; face normal: cross(e1, e2) then normalize
%cross = OpExtInst %v3float %GLSL.std.450 Cross %e1 %e2
%normal = OpExtInst %v3float %GLSL.std.450 Normalize %cross

; offset = normal * 0.08
%offset = OpVectorTimesScalar %v3float %normal %float_off

; build vec4 offset (w = 0)
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset_v4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

; --- Emit vertex 0 ---
%outpos0 = OpFAdd %v4float %p0 %offset_v4
%gl_out_pos_ptr = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %gl_out_pos_ptr %outpos0

; load color[0] and store to outColor
%col0_ptr = OpAccessChain %ptr_v3_in %in_color %int_0
%col0 = OpLoad %v3float %col0_ptr
OpStore %outColor %col0

OpEmitVertex

; --- Emit vertex 1 ---
%outpos1 = OpFAdd %v4float %p1 %offset_v4
OpStore %gl_out_pos_ptr %outpos1

%col1_ptr = OpAccessChain %ptr_v3_in %in_color %int_1
%col1 = OpLoad %v3float %col1_ptr
OpStore %outColor %col1

OpEmitVertex

; --- Emit vertex 2 ---
%outpos2 = OpFAdd %v4float %p2 %offset_v4
OpStore %gl_out_pos_ptr %outpos2

%col2_ptr = OpAccessChain %ptr_v3_in %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr
OpStore %outColor %col2

; finish primitive
OpEndPrimitive

OpReturn
