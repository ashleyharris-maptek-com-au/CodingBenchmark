Here's a single-file, complete C# program that tries to find a maximum clique. It uses an exact, highly optimized bitset-based branch-and-bound (Tomita-style) for small-to-moderate graphs (n ≤ 2000) and falls back to several fast heuristics with repeated randomized greedy expansion for larger graphs. It respects a global time limit so it will return the best found clique before the time runs out. The program reads the graph from stdin and writes the clique size and the vertices (0-indexed) to stdout.

Notes:
- For small graphs (<= 2000 vertices) it attempts an exact search and will return the provably maximum clique if the search finishes within the time limit.
- For larger graphs it uses heuristics (greedy expansion with randomized restarts and degree ordering). These heuristics are fast and robust across sparse/dense graphs but do not guarantee optimality for very large inputs (that's unavoidable for NP-hard problems).
- The program validates the result before printing and falls back to a trivial solution (single vertex) if something went wrong.

Copy/paste the code into a .cs file and compile with csc.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Threading;
using System.Threading.Tasks;

class MaxCliqueFinder
{
    static Stopwatch sw;
    static int timeLimitMs = 280000; // 4 minutes 40 seconds (safe margin under 5 minutes)

    // Graph representations
    static int n;
    static long m;
    static List<int>[] adjList;      // adjacency lists (sorted)
    static ulong[][] adjBitsets;     // used for small graphs exact algorithm, bitset words length = (n + 63) / 64

    // Exact algorithm (Tomita-style) results
    static int bestSizeExact;
    static int[] bestCliqueExact;    // store vertex indices
    static int words;

    // Heuristic result
    static int bestSizeHeur;
    static int[] bestCliqueHeur;
    static Random rng = new Random(1234567);

    static void Main(string[] args)
    {
        sw = Stopwatch.StartNew();

        // Read input
        var tokens = ReadAllTokens();
        if (tokens.Count == 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        int idx = 0;
        n = int.Parse(tokens[idx++]);
        m = long.Parse(tokens[idx++]);

        adjList = new List<int>[n];
        for (int i = 0; i < n; i++) adjList[i] = new List<int>();

        for (long e = 0; e < m; e++)
        {
            if (idx + 1 >= tokens.Count) break;
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            if (u >= 0 && u < n && v >= 0 && v < n && u != v)
            {
                adjList[u].Add(v);
                adjList[v].Add(u);
            }
        }

        if (n == 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        // Sort adjacency lists and remove duplicates if any
        for (int i = 0; i < n; i++)
        {
            if (adjList[i].Count > 1)
            {
                adjList[i].Sort();
                int write = 0;
                int prev = -1;
                for (int j = 0; j < adjList[i].Count; j++)
                {
                    if (adjList[i][j] != prev)
                    {
                        adjList[i][write++] = adjList[i][j];
                        prev = adjList[i][j];
                    }
                }
                if (write != adjList[i].Count)
                    adjList[i].RemoveRange(write, adjList[i].Count - write);
            }
        }

        // Quick trivial cases
        int maxDegree = 0;
        for (int i = 0; i < n; i++) if (adjList[i].Count > maxDegree) maxDegree = adjList[i].Count;
        if (maxDegree == 0)
        {
            // empty graph, maximum clique is any single vertex
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        // If graph small enough, build bitsets and run exact algorithm (Tomita-style).
        if (n <= 2000)
        {
            BuildBitsets();
            // Run exact algorithm but do it with time awareness.
            bestSizeExact = 0;
            bestCliqueExact = new int[0];

            // Choose time budget for exact stage (a portion of global limit)
            int exactTimeBudget = Math.Min(200000, timeLimitMs - 1000); // at most ~200s, leave margin
            long endTime = sw.ElapsedMilliseconds + exactTimeBudget;

            // Call exact search
            var allP = new ulong[words];
            for (int i = 0; i < n; i++)
            {
                int w = i >> 6;
                int b = i & 63;
                allP[w] |= (1UL << b);
            }

            // Order vertices by degeneracy/degree may help; we perform Tomita with coloring
            try
            {
                TomitaExpand(new List<int>(), allP, endTime);
            }
            catch (TimeUpException)
            {
                // time up for exact search - continue with heuristic
            }

            // If exact found something good and time remains to validate and print
            if (bestSizeExact > 0 && sw.ElapsedMilliseconds < timeLimitMs)
            {
                // Output exact result
                OutputClique(bestCliqueExact);
                return;
            }
            // else fall through to heuristics (maybe exact timed out or found nothing)
        }

        // Heuristic stage (handles all sizes)
        bestSizeHeur = 0;
        bestCliqueHeur = new int[0];

        int heurTimeBudget = timeLimitMs - (int)sw.ElapsedMilliseconds - 200; // leave a little margin
        if (heurTimeBudget < 50) heurTimeBudget = 50;
        long heurEndTime = sw.ElapsedMilliseconds + heurTimeBudget;

        RunHeuristics(heurEndTime);

        // Pick best found (if exact ran earlier and produced solution, prefer exact; otherwise heuristic)
        if (bestSizeExact > 0 && bestSizeExact >= bestSizeHeur)
            OutputClique(bestCliqueExact);
        else if (bestSizeHeur > 0)
            OutputClique(bestCliqueHeur);
        else
        {
            // Fallback: pick single vertex with highest degree
            int best = 0;
            for (int i = 1; i < n; i++) if (adjList[i].Count > adjList[best].Count) best = i;
            Console.WriteLine(1);
            Console.WriteLine(best);
        }
    }

    static void OutputClique(int[] clique)
    {
        if (clique == null) clique = new int[0];
        // Validate clique; if invalid, fallback to single vertex
        if (!IsClique(clique))
        {
            if (clique.Length == 0)
            {
                Console.WriteLine("0");
                Console.WriteLine();
                return;
            }
            // find a valid clique subset (naive) -> return single vertex
            Console.WriteLine(1);
            Console.WriteLine(clique[0]);
            return;
        }
        Console.WriteLine(clique.Length);
        if (clique.Length > 0)
        {
            for (int i = 0; i < clique.Length; i++)
            {
                if (i > 0) Console.Write(" ");
                Console.Write(clique[i]);
            }
            Console.WriteLine();
        }
    }

    static bool IsClique(int[] clique)
    {
        if (clique == null) return false;
        int k = clique.Length;
        for (int i = 0; i < k; i++)
        {
            int a = clique[i];
            for (int j = i + 1; j < k; j++)
            {
                int b = clique[j];
                if (!AreAdjacentSorted(a, b)) return false;
            }
        }
        return true;
    }

    static bool AreAdjacentSorted(int a, int b)
    {
        // adjacency lists sorted -> binary search
        var list = adjList[a];
        int lo = 0, hi = list.Count - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int val = list[mid];
            if (val == b) return true;
            if (val < b) lo = mid + 1; else hi = mid - 1;
        }
        return false;
    }

    // ---------- Exact bitset-based Tomita with coloring ----------
    class TimeUpException : Exception { }

    static void BuildBitsets()
    {
        words = (n + 63) >> 6;
        adjBitsets = new ulong[n][];
        for (int i = 0; i < n; i++)
        {
            adjBitsets[i] = new ulong[words];
            foreach (int v in adjList[i])
            {
                int w = v >> 6;
                int b = v & 63;
                adjBitsets[i][w] |= (1UL << b);
            }
        }
    }

    static void TomitaExpand(List<int> currentClique, ulong[] P, long timeLimitEnd)
    {
        if (sw.ElapsedMilliseconds > timeLimitEnd || sw.ElapsedMilliseconds > timeLimitMs) throw new TimeUpException();
        int pCount = BitCount(P);
        if (pCount == 0)
        {
            // no more candidates
            if (currentClique.Count > bestSizeExact)
            {
                bestSizeExact = currentClique.Count;
                bestCliqueExact = currentClique.ToArray();
            }
            return;
        }

        // Greedy coloring on P to get upper bounds
        var (vertOrder, colors) = GreedyColoring(P);

        // iterate vertices in descending color order (last to first)
        for (int idx = vertOrder.Count - 1; idx >= 0; idx--)
        {
            if (sw.ElapsedMilliseconds > timeLimitEnd || sw.ElapsedMilliseconds > timeLimitMs) throw new TimeUpException();

            int v = vertOrder[idx];
            int color = colors[idx];

            if (currentClique.Count + color <= bestSizeExact) // pruning
                return;

            // Add v to clique
            currentClique.Add(v);

            // P' = P ∩ N(v)
            var newP = BitsetAnd(P, adjBitsets[v]);

            TomitaExpand(currentClique, newP, timeLimitEnd);

            // remove v
            currentClique.RemoveAt(currentClique.Count - 1);

            // P = P \ {v}
            ClearBit(P, v);
            if (BitCount(P) == 0)
            {
                if (currentClique.Count > bestSizeExact)
                {
                    bestSizeExact = currentClique.Count;
                    bestCliqueExact = currentClique.ToArray();
                }
                return;
            }
        }
    }

    static (List<int> vertOrder, int[] colors) GreedyColoring(ulong[] P)
    {
        // Build list of vertices in P
        var verts = BitsetToList(P);
        int m = verts.Count;
        var colors = new int[m];
        var colorMasks = new List<ulong[]>();

        // We pick vertices in non-increasing degree order inside P to improve coloring
        verts.Sort((a, b) => adjList[b].Count.CompareTo(adjList[a].Count));

        for (int i = 0; i < m; i++)
        {
            int v = verts[i];
            bool assigned = false;
            for (int c = 0; c < colorMasks.Count; c++)
            {
                if (!BitsetIntersects(adjBitsets[v], colorMasks[c]))
                {
                    // can place v in color c
                    colors[i] = c + 1;
                    SetBitInMask(colorMasks[c], v);
                    assigned = true;
                    break;
                }
            }
            if (!assigned)
            {
                // create new color class
                var mask = new ulong[words];
                SetBitInMask(mask, v);
                colorMasks.Add(mask);
                colors[i] = colorMasks.Count;
            }
        }
        return (verts, colors);
    }

    static void SetBitInMask(ulong[] mask, int v)
    {
        int w = v >> 6;
        int b = v & 63;
        mask[w] |= (1UL << b);
    }

    static bool BitsetIntersects(ulong[] a, ulong[] b)
    {
        for (int i = 0; i < words; i++)
            if ((a[i] & b[i]) != 0) return true;
        return false;
    }

    static ulong[] BitsetAnd(ulong[] a, ulong[] b)
    {
        var r = new ulong[words];
        for (int i = 0; i < words; i++) r[i] = a[i] & b[i];
        return r;
    }

    static int BitCount(ulong[] a)
    {
        int sum = 0;
        for (int i = 0; i < words; i++)
            sum += BitOperations.PopCount(a[i]);
        return sum;
    }

    static List<int> BitsetToList(ulong[] a)
    {
        var list = new List<int>();
        for (int w = 0; w < words; w++)
        {
            ulong x = a[w];
            while (x != 0)
            {
                int tz = BitOperations.TrailingZeroCount(x);
                int v = (w << 6) + tz;
                if (v < n) list.Add(v);
                x &= x - 1;
            }
        }
        return list;
    }

    static void ClearBit(ulong[] a, int v)
    {
        int w = v >> 6;
        int b = v & 63;
        a[w] &= ~(1UL << b);
    }

    // ---------- Heuristics (greedy + randomized restarts) ----------
    static void RunHeuristics(long heurEndTime)
    {
        // Prepare adjacency arrays (already have sorted List<int> arrays)
        var adjArr = new int[n][];
        for (int i = 0; i < n; i++) adjArr[i] = adjList[i].ToArray();

        // initial greedy solutions:
        var degOrder = Enumerable.Range(0, n).OrderByDescending(v => adjArr[v].Length).ToArray();
        var clique1 = GreedyExpandFromOrder(degOrder, adjArr);
        UpdateBestHeur(clique1);

        // try from highest degree vertices individually for a few
        int tries = Math.Min(n, 300);
        for (int t = 0; t < tries && sw.ElapsedMilliseconds < heurEndTime && sw.ElapsedMilliseconds < timeLimitMs; t++)
        {
            int seed = degOrder[t];
            var c = GreedyExpandFromSeed(seed, adjArr);
            UpdateBestHeur(c);
        }

        // Randomized restarts until time runs out
        int iter = 0;
        while (sw.ElapsedMilliseconds < heurEndTime && sw.ElapsedMilliseconds < timeLimitMs)
        {
            iter++;
            // choose random seed biased by degree
            int seed = WeightedRandomByDegree();
            var clique = GreedyExpandFromSeed(seed, adjArr, randomness: true);
            UpdateBestHeur(clique);

            // occasionally try expansion from a random ordering of high-degree vertices
            if ((iter & 7) == 0)
            {
                int[] perm = degOrder.Take(Math.Min(n, 500)).ToArray();
                Shuffle(perm);
                var c2 = GreedyExpandFromOrder(perm, adjArr);
                UpdateBestHeur(c2);
            }
        }
    }

    static int[] GreedyExpandFromOrder(int[] order, int[][] adjArr)
    {
        var best = new List<int>();
        foreach (int v0 in order)
        {
            var c = GreedyExpandFromSeed(v0, adjArr);
            if (c.Length > best.Count) best = new List<int>(c);
            // small speed optimization: stop early if clique equals max possible (degree + 1)
            if (best.Count >= adjArr[v0].Length + 1) break;
        }
        return best.ToArray();
    }

    static int[] GreedyExpandFromSeed(int seed, int[][] adjArr, bool randomness = false)
    {
        var clique = new List<int>();
        clique.Add(seed);

        // candidates = neighbors(seed)
        var cand = new List<int>(adjArr[seed]);

        // To ensure deterministic behavior, keep candidates sorted by descending degree
        cand.Sort((a, b) => adjArr[b].Length.CompareTo(adjArr[a].Length));

        while (cand.Count > 0)
        {
            // select next candidate
            int pickIndex = 0;
            if (randomness)
            {
                // sample top-K and pick random among them weighted by degree
                int K = Math.Min(5, cand.Count);
                int r = rng.Next(K);
                pickIndex = r;
            }
            int v = cand[pickIndex];

            // include v
            clique.Add(v);

            // new candidates = intersection(cand, neighbors(v))
            var newCand = IntersectSortedLists(cand, adjArr[v]);
            // remove already included vertices (clique members) - but newCand is from cand so fine
            // reorder by degree descending for next iteration
            newCand.Sort((a, b) => adjArr[b].Length.CompareTo(adjArr[a].Length));
            cand = newCand;
        }

        return clique.ToArray();
    }

    static List<int> IntersectSortedLists(List<int> aList, int[] bArr)
    {
        var res = new List<int>();
        int i = 0, j = 0;
        int an = aList.Count, bn = bArr.Length;
        while (i < an && j < bn)
        {
            int av = aList[i], bv = bArr[j];
            if (av == bv)
            {
                res.Add(av);
                i++; j++;
            }
            else if (av < bv) i++;
            else j++;
        }
        return res;
    }

    static int WeightedRandomByDegree()
    {
        // pick vertex weighted by (deg+1)
        long total = 0;
        // build cumulative using small sample of vertices to speed up for large n
        int sample = Math.Min(n, 1000);
        int[] idxArr = Enumerable.Range(0, n).ToArray();
        // sample uniformly
        int s = rng.Next(n);
        int v0 = s;

        // compute degrees cumulative on the fly (smaller memory pressure)
        long[] prefix = new long[sample];
        int[] sampleIdx = new int[sample];
        for (int i = 0; i < sample; i++)
        {
            int vi = (v0 + i) % n;
            sampleIdx[i] = vi;
            prefix[i] = (i == 0 ? (adjList[vi].Count + 1) : prefix[i - 1] + adjList[vi].Count + 1);
        }
        long r = (long)(rng.NextDouble() * prefix[sample - 1]);
        int pos = Array.BinarySearch(prefix, r);
        if (pos < 0) pos = ~pos;
        if (pos >= sample) pos = sample - 1;
        return sampleIdx[pos];
    }

    static void UpdateBestHeur(int[] clique)
    {
        if (clique == null) return;
        if (clique.Length > bestSizeHeur)
        {
            // validate quickly
            if (IsClique(clique))
            {
                bestSizeHeur = clique.Length;
                bestCliqueHeur = clique.ToArray();
            }
        }
    }

    // ---------- Utilities ----------
    static void Shuffle<T>(T[] arr)
    {
        for (int i = arr.Length - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            T t = arr[i]; arr[i] = arr[j]; arr[j] = t;
        }
    }

    // Fast input reading
    static List<string> ReadAllTokens()
    {
        var list = new List<string>();
        using (var sr = new StreamReader(Console.OpenStandardInput()))
        {
            string s;
            while ((s = sr.ReadLine()) != null)
            {
                if (s.Length == 0) continue;
                var parts = s.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                foreach (var p in parts) list.Add(p);
            }
        }
        return list;
    }
}
```
