; SPIR-V
; Geometry shader: expands a 2-vertex line (InputLines) into a thick ribbon (triangle strip of 4 vertices)
; Emits vertices: p0+perp, p0-perp, p1+perp, p1-perp
; outColor set per-vertex (c0 for p0 vertices, c1 for p1 vertices)
; Assembles with spirv-as (Vulkan 1.0)

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color_in %outColor %gl_out %ubo
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Import GLSL extended instruction set
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_0_03 = OpConstant %float 0.03

; gl_PerVertex input struct (array of 2 for InputLines)
%_struct_gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %_struct_gl_PerVertex_in Block
OpMemberDecorate %_struct_gl_PerVertex_in 0 BuiltIn Position

%arr_gl_PerVertex_in = OpTypeArray %_struct_gl_PerVertex_in %uint_2
%ptr_input_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_input_arr_gl_PerVertex Input

%ptr_input_v4 = OpTypePointer Input %v4float

; user-defined input arrays (location = 0,1,2)
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in

%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color_in = OpVariable %ptr_input_arr_v3 Input

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color_in Location 2

; Output gl_PerVertex struct
%_struct_gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %_struct_gl_PerVertex_out Block
OpMemberDecorate %_struct_gl_PerVertex_out 0 BuiltIn Position

%ptr_output_gl_PerVertex = OpTypePointer Output %_struct_gl_PerVertex_out
%gl_out = OpVariable %ptr_output_gl_PerVertex Output

%ptr_output_v4 = OpTypePointer Output %v4float

; Output color (location = 0)
%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output
OpDecorate %outColor Location 0

; Uniform buffer (set=0 binding=0) - declared but not used here (available)
%ubo_struct = OpTypeStruct %v4float %v4float %v4float
OpDecorate %ubo_struct Block
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_uniform_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Function main
%main = OpFunction %void None %func_void
%entry = OpLabel

; Access gl_in[0].gl_Position
%idx0 = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %idx0

; Access gl_in[1].gl_Position
%idx1 = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %idx1

; Load colors c0,c1 from color_in[0], color_in[1]
%ptr_color0 = OpAccessChain %ptr_input_v3 %color_in %int_0
%c0 = OpLoad %v3float %ptr_color0
%ptr_color1 = OpAccessChain %ptr_input_v3 %color_in %int_1
%c1 = OpLoad %v3float %ptr_color1

; dir = p1 - p0  (vec4)
%dir = OpFSub %v4float %p1 %p0

; Extract dir.x and dir.y
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; perp2 = normalize( vec2(-dir.y, dir.x) ) * thickness
%neg_dir_y = OpFNegate %float %dir_y
%tmp_perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm = OpExtInst %v2float %GLSL450 Normalize %tmp_perp
%perp_scaled = OpVectorTimesScalar %v2float %perp_norm %float_0_03

; Build vec4 perp (perp.x, perp.y, 0, 0)
%perp_x = OpCompositeExtract %float %perp_scaled 0
%perp_y = OpCompositeExtract %float %perp_scaled 1
%perp4 = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0

; Compute four positions: p0+perp, p0-perp, p1+perp, p1-perp
%p0_plus = OpFAdd %v4float %p0 %perp4
%p0_minus = OpFSub %v4float %p0 %perp4
%p1_plus = OpFAdd %v4float %p1 %perp4
%p1_minus = OpFSub %v4float %p1 %perp4

; Prepare pointer to gl_out position member
%ptr_gl_out_pos = OpAccessChain %ptr_output_v4 %gl_out %int_0

; Emit vertex 0: p0 + perp, color = c0
OpStore %ptr_gl_out_pos %p0_plus
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1: p0 - perp, color = c0
OpStore %ptr_gl_out_pos %p0_minus
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2: p1 + perp, color = c1
OpStore %ptr_gl_out_pos %p1_plus
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3: p1 - perp, color = c1
OpStore %ptr_gl_out_pos %p1_minus
OpStore %outColor %c1
OpEmitVertex

; End primitive (triangle strip complete)
OpEndPrimitive

OpReturn
OpFunctionEnd
