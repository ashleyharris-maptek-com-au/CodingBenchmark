; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %uboStruct Block
MemberDecorate %uboStruct 0 Offset 0
MemberDecorate %uboStruct 1 Offset 64
MemberDecorate %uboStruct 2 Offset 128
MemberDecorate %uboStruct 3 Offset 192
MemberDecorate %uboStruct 4 Offset 208
MemberDecorate %uboStruct 5 Offset 224

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%uboStruct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptr_uniform_ubo = OpTypePointer Uniform %uboStruct
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

; Constants
%float_1 = OpConstant %float 1.0

; Variables
%uboVar = OpVariable %ptr_uniform_ubo Uniform
%inColor = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output

; Function
%main = OpFunction %void None %func
%entry = OpLabel
%col = OpLoad %v3float %inColor
%c0 = OpCompositeExtract %float %col 0
%c1 = OpCompositeExtract %float %col 1
%c2 = OpCompositeExtract %float %col 2
%col4 = OpCompositeConstruct %v4float %c0 %c1 %c2 %float_1
OpStore %outColor %col4
OpReturn
OpFunctionEnd
