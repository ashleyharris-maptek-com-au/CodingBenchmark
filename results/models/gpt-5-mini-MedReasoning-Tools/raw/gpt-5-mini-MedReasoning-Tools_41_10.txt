; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor %uboVar
OpExecutionMode %main OriginUpperLeft
OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inUV "uv"
OpName %outColor "outColor"
OpName %uboVar "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpExtInstImport %GLSL "GLSL.std.450"
OpTypeVoid %void
OpTypeFunction %void %void
OpTypeFloat %f32 32
OpTypeInt %i32 32 1
OpTypeBool %bool
OpTypeVector %v2f %f32 2
OpTypeVector %v3f %f32 3
OpTypeVector %v4f %f32 4
OpTypeMatrix %mat4 %v4f 4
OpTypePointer Input %v3f
OpTypePointer Input %v2f
OpTypePointer Output %v4f
OpTypePointer Uniform %UBO
OpTypePointer Uniform %v4f
OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
; IDs for constants
OpConstant %f32 0.0
OpConstant %f32 1.0
OpConstant %f32 3.14159
OpConstant %f32 16.0
OpConstant %f32 50.26544
OpConstant %f32 0.4
OpConstant %f32 0.6
OpConstant %f32 0.9
OpConstant %f32 0.1
OpConstant %f32 0.0
; composite constants for colors
OpConstantComposite %v3f %_c9 %_c10 %_c10
OpConstantComposite %v3f %_c10 %_c10 %_c9
; Above references need correct constant IDs - declare explicit constants properly
; Re-declare scalar constants with explicit ids to ensure references
%c0 = OpConstant %f32 0.0
%c1 = OpConstant %f32 1.0
%c_pi = OpConstant %f32 3.14159
%c16 = OpConstant %f32 16.0
%c50 = OpConstant %f32 50.26544
%c04 = OpConstant %f32 0.4
%c06 = OpConstant %f32 0.6
%c09 = OpConstant %f32 0.9
%c01 = OpConstant %f32 0.1
; color vectors
%red = OpConstantComposite %v3f %c09 %c01 %c01
%blue = OpConstantComposite %v3f %c01 %c01 %c09
; pointer variables
%ptr_in_v3 = OpTypePointer Input %v3f
%ptr_in_v2 = OpTypePointer Input %v2f
%ptr_out_v4 = OpTypePointer Output %v4f
%ptr_uniform_v4 = OpTypePointer Uniform %v4f
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
; variables
%inWorldPos = OpVariable %ptr_in_v3 Input
%inNormal = OpVariable %ptr_in_v3 Input
%inUV = OpVariable %ptr_in_v2 Input
%outColor = OpVariable %ptr_out_v4 Output
%uboVar = OpVariable %ptr_uniform_UBO Uniform
; function main
%main = OpFunction %void None %void
%label = OpLabel
; load inputs
%uv = OpLoad %v2f %inUV
%uv_y = OpCompositeExtract %f32 %uv 1
; compute stripe value: sin(uv.y * pi * 16)
%mul1 = OpFMul %f32 %uv_y %c_pi
%mul2 = OpFMul %f32 %mul1 %c16
%stripe = OpExtInst %f32 %GLSL Sin %mul2
; compare stripe >= 0.0
%cond = OpFOrdGreaterThanEqual %bool %stripe %c0
; select color
%stripeColor = OpSelect %v3f %cond %red %blue
; prepare normal and worldPos
%normal = OpLoad %v3f %inNormal
%norm_n = OpExtInst %v3f %GLSL Normalize %normal
%wp = OpLoad %v3f %inWorldPos
; get lightPos from UBO (member 3)
%lightPtr = OpAccessChain %ptr_uniform_v4 %uboVar %int_3
%light4 = OpLoad %v4f %lightPtr
%lightPos = OpCompositeExtract %v3f %light4 0 1 2
; compute light direction
%ld = OpFSub %v3f %lightPos %wp
%ld_n = OpExtInst %v3f %GLSL Normalize %ld
; compute diffuse = max(dot(n,l), 0.0)
%dot = OpExtInst %f32 %GLSL Dot %norm_n %ld_n
%diffuse = OpExtInst %f32 %GLSL FMax %dot %c0
; factor = 0.4 + 0.6 * diffuse
%muldf = OpFMul %f32 %c06 %diffuse
%factor = OpFAdd %f32 %c04 %muldf
; multiply stripeColor by factor
%rgb = OpVectorTimesScalar %v3f %stripeColor %factor
; construct out color vec4(rgb, 1.0)
%outv = OpCompositeConstruct %v4f %rgb %c1
OpStore %outColor %outv
OpReturn
OpFunctionEnd

; Additional small constant ints
%int_0 = OpConstant %i32 0
%int_1 = OpConstant %i32 1
%int_2 = OpConstant %i32 2
%int_3 = OpConstant %i32 3
