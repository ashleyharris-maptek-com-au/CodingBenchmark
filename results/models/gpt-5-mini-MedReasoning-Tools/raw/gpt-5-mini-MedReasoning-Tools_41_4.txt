; SPIR-V

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPosVar %normalVar %uvVar %tangentVar %colorVar %outColorVar
OpExecutionMode %main OriginUpperLeft
OpExtInstImport %GLSL "GLSL.std.450"

; Decorations for inputs/outputs
OpDecorate %worldPosVar Location 0
OpDecorate %normalVar Location 1
OpDecorate %uvVar Location 2
OpDecorate %tangentVar Location 3
OpDecorate %colorVar Location 4
OpDecorate %outColorVar Location 0

; UBO struct and decorations (std140 offsets)
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224
OpDecorate %UBOType Block
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Names (optional for readability)
OpName %main "main"
OpName %worldPosVar "worldPos"
OpName %normalVar "normal"
OpName %uvVar "uv"
OpName %tangentVar "tangent"
OpName %colorVar "color"
OpName %outColorVar "outColor"
OpName %uboVar "ubo"
OpName %UBOType "UBO"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

; Pointer types
%ptr_input_v3 = OpTypePointer Input %v3f
%ptr_input_v2 = OpTypePointer Input %v2f
%ptr_output_v4 = OpTypePointer Output %v4f
%ptr_uniform_v4 = OpTypePointer Uniform %v4f
%ptr_uniform_UBO = OpTypePointer Uniform %UBOType
%ptr_function_v3 = OpTypePointer Function %v3f

; UBO struct type: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%UBOType = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f

; Variables
%worldPosVar = OpVariable %ptr_input_v3 Input
%normalVar = OpVariable %ptr_input_v3 Input
%uvVar = OpVariable %ptr_input_v2 Input
%tangentVar = OpVariable %ptr_input_v3 Input
%colorVar = OpVariable %ptr_input_v3 Input
%outColorVar = OpVariable %ptr_output_v4 Output

%uboVar = OpVariable %ptr_uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%wp = OpLoad %v3f %worldPosVar
%n_in = OpLoad %v3f %normalVar

; Access lightPos in UBO (member index 3)
%lightPtr = OpAccessChain %ptr_uniform_v4 %uboVar %int_3
%lightVec4 = OpLoad %v4f %lightPtr
; extract xyz from vec4
%light_xyz = OpVectorShuffle %v3f %lightVec4 %lightVec4 0 1 2

; compute L = normalize(lightPos.xyz - worldPos)
%Lsub = OpFSub %v3f %light_xyz %wp
%L = OpExtInst %v3f %GLSL Normalize %Lsub

; compute N = normalize(normal)
%N = OpExtInst %v3f %GLSL Normalize %n_in

; dot = max(dot(N,L), 0.0)
%dotNL = OpExtInst %float %GLSL Dot %N %L
%d_clamped = OpExtInst %float %GLSL FMax %dotNL %float_0

; build output color vec4(d,d,d,1.0)
%d_vec3 = OpCompositeConstruct %v3f %d_clamped %d_clamped %d_clamped
%outVec = OpCompositeConstruct %v4f %d_clamped %d_clamped %d_clamped %float_1
OpStore %outColorVar %outVec

OpReturn
OpFunctionEnd
