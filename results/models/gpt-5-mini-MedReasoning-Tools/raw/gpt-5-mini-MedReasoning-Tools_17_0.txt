Here's a complete, self-contained C# program that implements a fast, scalable heuristic for the 3D axis-aligned bin-packing described. It uses a layered approach: it stacks horizontal layers (along Y / height), and within each layer runs a fast 2D guillotine-style free-rectangle packer (a simplified MaxRects/Guillotine hybrid). Boxes may be rotated among axis-aligned permutations (w/h/d reordered). The code is written to be efficient for large inputs, prunes redundant free rectangles, and includes a runtime timeout safeguard so it returns a valid packing well before the 30s limit if needed.

Copy, compile and run with the input format you specified. The program prints the number of packed boxes followed by lines of "box_index x y z" (0-based indices and integer corner coordinates).

Code:

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Globalization;

class Program
{
    // Box with original index and dimensions
    struct Box
    {
        public int idx;
        public int w, h, d;
        public long vol;
        public Box(int i, int ww, int hh, int dd) { idx = i; w = ww; h = hh; d = dd; vol = (long)ww * hh * dd; }
    }

    // Placed box record
    struct Placed
    {
        public int idx;
        public int x, y, z;
        public int w, h, d;
        public Placed(int i, int X, int Y, int Z, int W, int H, int D) { idx = i; x = X; y = Y; z = Z; w = W; h = H; d = D; }
    }

    // 2D rectangle for free area in layer (coordinates: x,z) where z is depth-axis
    class Rect2D
    {
        public int x, z, w, d;
        public Rect2D(int x_, int z_, int w_, int d_) { x = x_; z = z_; w = w_; d = d_; }
    }

    // 2D packer (guillotine-like)
    class LayerPacker
    {
        public int W, D;
        public List<Rect2D> freeRects;

        public LayerPacker(int width, int depth)
        {
            W = width; D = depth;
            freeRects = new List<Rect2D> { new Rect2D(0, 0, W, D) };
        }

        // Attempt to place rectangle (wxd). Returns (success, x,z) and modifies freeRects.
        public bool TryPlace(int w, int d, out int outX, out int outZ)
        {
            outX = outZ = -1;
            int bestIndex = -1;
            int bestShort = int.MaxValue;
            long bestArea = long.MaxValue;

            for (int i = 0; i < freeRects.Count; i++)
            {
                var r = freeRects[i];
                if (w <= r.w && d <= r.d)
                {
                    int shortSide = Math.Min(r.w - w, r.d - d);
                    long areaLeft = (long)r.w * r.d - (long)w * d;
                    if (shortSide < bestShort || (shortSide == bestShort && areaLeft < bestArea))
                    {
                        bestShort = shortSide;
                        bestArea = areaLeft;
                        bestIndex = i;
                    }
                }
            }

            if (bestIndex == -1) return false;

            var chosen = freeRects[bestIndex];
            outX = chosen.x;
            outZ = chosen.z;

            // Guillotine-style split: remove chosen rect and create up to two rects:
            // right rect: area to the right of placed rect (x+w, z) size (r.w - w, r.d)
            // top rect: area below/behind placed rect in z direction (x, z+d) size (r.w, r.d - d)
            List<Rect2D> newRects = new List<Rect2D>();

            if (chosen.w - w > 0)
            {
                newRects.Add(new Rect2D(chosen.x + w, chosen.z, chosen.w - w, chosen.d));
            }
            if (chosen.d - d > 0)
            {
                newRects.Add(new Rect2D(chosen.x, chosen.z + d, chosen.w, chosen.d - d));
            }

            // Replace chosen with new rects
            freeRects.RemoveAt(bestIndex);
            foreach (var nr in newRects) freeRects.Add(nr);

            PruneFreeList();

            return true;
        }

        // Remove free rects that are contained in others (pruning)
        void PruneFreeList()
        {
            var list = freeRects;
            int n = list.Count;
            bool[] removed = new bool[n];

            for (int i = 0; i < n; i++)
            {
                if (removed[i]) continue;
                var a = list[i];
                for (int j = 0; j < n; j++)
                {
                    if (i == j || removed[j]) continue;
                    var b = list[j];
                    if (IsContained(a, b))
                    {
                        removed[i] = true;
                        break;
                    }
                }
            }

            var newList = new List<Rect2D>(n);
            for (int i = 0; i < n; i++) if (!removed[i]) newList.Add(list[i]);
            freeRects = newList;
        }

        static bool IsContained(Rect2D a, Rect2D b)
        {
            if (a.x >= b.x && a.z >= b.z && a.x + a.w <= b.x + b.w && a.z + a.d <= b.z + b.d) return true;
            return false;
        }
    }

    static void Main(string[] args)
    {
        // Read input
        var sw = Stopwatch.StartNew();
        string line = Console.ReadLine();
        if (line == null)
        {
            Console.WriteLine("0");
            return;
        }
        var parts = line.Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 4)
        {
            Console.WriteLine("0");
            return;
        }

        int N = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int W = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int H = int.Parse(parts[2], CultureInfo.InvariantCulture);
        int D = int.Parse(parts[3], CultureInfo.InvariantCulture);

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            string ln = Console.ReadLine();
            if (ln == null) break;
            var p = ln.Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (p.Length < 3) { i--; continue; }
            int ww = int.Parse(p[0], CultureInfo.InvariantCulture);
            int hh = int.Parse(p[1], CultureInfo.InvariantCulture);
            int dd = int.Parse(p[2], CultureInfo.InvariantCulture);
            boxes.Add(new Box(i, ww, hh, dd));
        }

        // Sort boxes by volume descending (large first) for better quality generally.
        boxes.Sort((a, b) =>
        {
            int c = b.vol.CompareTo(a.vol);
            if (c != 0) return c;
            int ma = Math.Max(a.w, Math.Max(a.h, a.d));
            int mb = Math.Max(b.w, Math.Max(b.h, b.d));
            return mb.CompareTo(ma);
        });

        var placed = new List<Placed>();
        var used = new bool[N];

        // layers: list of (y position, height, packer)
        var layerYs = new List<int>();
        var layerHs = new List<int>();
        var layerPackers = new List<LayerPacker>();

        int currentHeight = 0;

        // Pre-generate all 6 permutations for rotation to avoid duplicates
        Func<int, int, int, List<(int,int,int)>> GenPerms = (a,b,c) =>
        {
            var perms = new List<(int,int,int)>(6);
            perms.Add((a,b,c));
            perms.Add((a,c,b));
            perms.Add((b,a,c));
            perms.Add((b,c,a));
            perms.Add((c,a,b));
            perms.Add((c,b,a));
            // Remove duplicates if some dims equal
            var uniq = new List<(int,int,int)>();
            var seen = new HashSet<string>();
            foreach (var t in perms)
            {
                string key = $"{t.Item1}:{t.Item2}:{t.Item3}";
                if (!seen.Contains(key)) { seen.Add(key); uniq.Add(t); }
            }
            return uniq;
        };

        // Timeout guard
        TimeSpan timeLimit = TimeSpan.FromSeconds(28.0); // leave margin
        for (int bi = 0; bi < boxes.Count; bi++)
        {
            if (sw.Elapsed > timeLimit) break;

            var box = boxes[bi];

            // Skip boxes that are individually too large in all orientations
            bool canFitAny = false;
            var permsAll = GenPerms(box.w, box.h, box.d);
            foreach (var perm in permsAll)
            {
                if (perm.Item1 <= W && perm.Item2 <= H && perm.Item3 <= D) { canFitAny = true; break; }
            }
            if (!canFitAny) continue;

            bool placedThis = false;

            // Try to place in existing layers
            for (int li = 0; li < layerPackers.Count && !placedThis; li++)
            {
                if (sw.Elapsed > timeLimit) break;

                int layerH = layerHs[li];
                int layerY = layerYs[li];
                var packer = layerPackers[li];

                // Try any orientation that fits in layerHeight and layer's 2D plane
                foreach (var perm in permsAll)
                {
                    int w0 = perm.Item1, h0 = perm.Item2, d0 = perm.Item3;
                    if (h0 > layerH) continue; // must fit within this layer's height
                    if (w0 > W || d0 > D) continue;
                    if (packer.TryPlace(w0, d0, out int x0, out int z0))
                    {
                        placed.Add(new Placed(box.idx, x0, layerY, z0, w0, h0, d0));
                        used[box.idx] = true;
                        placedThis = true;
                        break;
                    }
                }
            }

            if (placedThis) continue;

            // Try to create a new layer for this box (choose orientation and place at start)
            // Choose orientation that minimizes layer height (to leave more room), but still fits
            (int ow, int oh, int od) bestOrient = (0,0,0);
            int bestLayerH = int.MaxValue;
            foreach (var perm in permsAll)
            {
                int w0 = perm.Item1, h0 = perm.Item2, d0 = perm.Item3;
                if (w0 <= W && d0 <= D && currentHeight + h0 <= H)
                {
                    if (h0 < bestLayerH) { bestLayerH = h0; bestOrient = (w0,h0,d0); }
                }
            }
            if (bestLayerH != int.MaxValue)
            {
                int newLayerY = currentHeight;
                int newLayerH = bestLayerH;
                if (newLayerY + newLayerH <= H)
                {
                    var newPacker = new LayerPacker(W, D);
                    // Place at origin
                    if (!newPacker.TryPlace(bestOrient.ow, bestOrient.od, out int putX, out int putZ))
                    {
                        // unexpected fail; skip creating layer
                    }
                    else
                    {
                        layerYs.Add(newLayerY);
                        layerHs.Add(newLayerH);
                        layerPackers.Add(newPacker);
                        placed.Add(new Placed(box.idx, putX, newLayerY, putZ, bestOrient.ow, bestOrient.oh, bestOrient.od));
                        used[box.idx] = true;
                        currentHeight += newLayerH;
                        placedThis = true;
                    }
                }
            }

            // continue to next box
        }

        // Output results
        Console.WriteLine(placed.Count);
        // Output placed boxes lines: box_index x y z
        foreach (var p in placed)
        {
            Console.WriteLine($"{p.idx} {p.x} {p.y} {p.z}");
        }
    }
}
```

Notes / design choices
- Layered strategy: We create horizontal layers (stacked along height). Each layer's height equals the tallest box assigned to that layer. This reduces the 3D problem into repeated 2D packing tasks, which scales well and is fast.
- 2D guillotine-like packer: fast placement with simple splitting and pruning of contained free rectangles. This is robust and fast for thousands of boxes.
- Box rotations: all axis-aligned permutations are tried (duplicates removed).
- Sorting: boxes processed by descending volume (large first), a common heuristic that tends to improve packing quality.
- Timeout safety: the solver checks elapsed time and stops adding boxes if it reaches ~28 seconds, ensuring the program finishes well inside 30s for very large inputs.
- The implementation focuses on a balance of speed and packing quality. For extremely tight optimal packing, more complex algorithms would be needed (and slower). This heuristic is practical and reliable across the requested scales.
