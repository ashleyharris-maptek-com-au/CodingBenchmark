#!/usr/bin/env python3
import sys
sys.setrecursionlimit(10000)

def main():
    data = sys.stdin.read().splitlines()
    if not data:
        print("NO SOLUTION")
        return

    # Parse first non-empty line for U and S
    first_line = None
    for ln in data:
        if ln.strip() != "":
            first_line = ln.strip()
            break
    if first_line is None:
        print("NO SOLUTION")
        return

    parts = first_line.split()
    if len(parts) < 2:
        print("NO SOLUTION")
        return
    try:
        U = int(parts[0])
        S = int(parts[1])
    except:
        print("NO SOLUTION")
        return

    # Read the next S lines (including possibly empty set lines).
    # Find index of first_line in data to locate subsequent lines reliably.
    start_idx = 0
    for i, ln in enumerate(data):
        if ln.strip() == first_line:
            start_idx = i
            break

    sets = []
    for i in range(S):
        line_idx = start_idx + 1 + i
        if line_idx < len(data):
            line = data[line_idx].strip()
            if line == "":
                sets.append([])
            else:
                try:
                    elems = list(map(int, line.split()))
                except:
                    elems = []
                # remove duplicates and keep deterministic order (sorted)
                elems = sorted(set(elems))
                sets.append(elems)
        else:
            sets.append([])

    # Quick edge cases
    if U == 0:
        # empty universe is trivially covered by empty selection
        print("SOLUTION")
        print()
        return
    if S == 0:
        print("NO SOLUTION")
        return

    # Build element -> list of sets mapping
    elem_to_sets = [[] for _ in range(U)]
    for si, elems in enumerate(sets):
        for e in elems:
            if 0 <= e < U:
                elem_to_sets[e].append(si)

    # If any element has no covering sets at all, no solution
    for e in range(U):
        if not elem_to_sets[e]:
            print("NO SOLUTION")
            return

    covered = [False] * U
    chosen = []
    chosen_set = [False] * S

    # Pre-sort lists of set indices for deterministic behavior
    for e in range(U):
        elem_to_sets[e].sort()

    # Helper: check if all elements covered
    def all_covered():
        for x in covered:
            if not x:
                return False
        return True

    # Recursive search using Algorithm X style heuristic:
    # choose uncovered element with fewest available sets (deterministic tie-breaking)
    def search():
        if all_covered():
            return True

        # For each uncovered element, compute available candidate sets (those whose all elems are currently uncovered)
        best_e = -1
        best_candidates = None
        best_count = None

        for e in range(U):
            if covered[e]:
                continue
            candidates = []
            for si in elem_to_sets[e]:
                if chosen_set[si]:
                    continue
                # check that this set does not conflict with already covered elements
                ok = True
                for ee in sets[si]:
                    if covered[ee]:
                        ok = False
                        break
                if ok:
                    candidates.append(si)
            cnt = len(candidates)
            if cnt == 0:
                return False  # uncovered element with no available sets -> fail
            if best_count is None or cnt < best_count or (cnt == best_count and e < best_e):
                best_count = cnt
                best_e = e
                best_candidates = candidates

        # Iterate candidate sets in deterministic order (they are already sorted)
        for si in best_candidates:
            # pick set si
            chosen_set[si] = True
            chosen.append(si)
            marked = []
            for ee in sets[si]:
                if not covered[ee]:
                    covered[ee] = True
                    marked.append(ee)

            if search():
                return True

            # undo
            for ee in marked:
                covered[ee] = False
            chosen.pop()
            chosen_set[si] = False

        return False

    found = search()
    if not found:
        print("NO SOLUTION")
    else:
        # Output indices in increasing order for determinism
        out = sorted(chosen)
        print("SOLUTION")
        if out:
            print(" ".join(map(str, out)))
        else:
            print()

if __name__ == "__main__":
    main()
