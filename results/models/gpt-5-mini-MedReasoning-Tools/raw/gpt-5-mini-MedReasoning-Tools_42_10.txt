; SPIR-V
; Geometry shader implementing a thick smooth curve using InputLinesAdjacency -> OutputTriangleStrip
; Emits 10 vertices (5 subdivided points * 2 vertices each) as a single triangle strip.
; Assembles with spirv-as (Vulkan 1.0)

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Entry point and interfaces
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Decorations for built-ins, locations, binding, sets
; gl_PerVertex in (array of 4)
; Define gl_PerVertex struct types
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_PerVertex out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; constants
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%bool = OpTypeBool

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%f_0 = OpConstant %float 0.0
%f_025 = OpConstant %float 0.25
%f_05 = OpConstant %float 0.5
%f_075 = OpConstant %float 0.75
%f_1 = OpConstant %float 1.0
%thickness = OpConstant %float 0.02

; vector types
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3

; arrays of per-vertex gl_PerVertex_in (InputLinesAdjacency => 4)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output per-vertex
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined input arrays: layout(location=N) in vec3 ...
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos Location 0
%normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %normal Location 1
%color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %color Location 2

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Uniform buffer object (set=0, binding=0) - declared but not used in shader logic here
%ubo_struct = OpTypeStruct %v4float %v4float %v4float %v4float %v4float ; placeholder members
OpDecorate %ubo_struct Block
%ptr_ubo = OpTypePointer UniformConstant %ubo_struct
%ubo = OpVariable %ptr_ubo UniformConstant
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Pointers for frequently used storage classes
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; function types
%void = OpTypeVoid
%func = OpTypeFunction %void

; helpers: composite types for temporaries
%ptr_fn_v3 = OpTypePointer Function %v3float
%ptr_fn_v2 = OpTypePointer Function %v2float
%ptr_fn_v4 = OpTypePointer Function %v4float
%ptr_fn_float = OpTypePointer Function %float
%ptr_fn_uint = OpTypePointer Function %uint

; Names (optional for readability)
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %ubo "ubo"

; ------------------------------------------------------------
; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; --- Load input gl_Position values from gl_in[0..3] ---
; We'll extract XYZ as v3float for P0..P3
%idx0 = OpConstantComposite %uint %uint_0
%idx1 = OpConstantComposite %uint %uint_1
%idx2 = OpConstantComposite %uint %uint_2
%idx3 = OpConstantComposite %uint %uint_3

; Access gl_in[0].gl_Position
%ptr_in_0 = OpAccessChain %ptr_in_v4 %gl_in %idx0 %int_0
%load_in_0 = OpLoad %v4float %ptr_in_0
%P0_xyz = OpVectorShuffle %v3float %load_in_0 %load_in_0 0 1 2

; gl_in[1]
%ptr_in_1 = OpAccessChain %ptr_in_v4 %gl_in %idx1 %int_0
%load_in_1 = OpLoad %v4float %ptr_in_1
%P1_xyz = OpVectorShuffle %v3float %load_in_1 %load_in_1 0 1 2

; gl_in[2]
%ptr_in_2 = OpAccessChain %ptr_in_v4 %gl_in %idx2 %int_0
%load_in_2 = OpLoad %v4float %ptr_in_2
%P2_xyz = OpVectorShuffle %v3float %load_in_2 %load_in_2 0 1 2

; gl_in[3]
%ptr_in_3 = OpAccessChain %ptr_in_v4 %gl_in %idx3 %int_0
%load_in_3 = OpLoad %v4float %ptr_in_3
%P3_xyz = OpVectorShuffle %v3float %load_in_3 %load_in_3 0 1 2

; --- Colors endpoints: color[1] and color[2] ---
%ptr_c1 = OpAccessChain %ptr_in_v3 %color %idx1
%col1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3 %color %idx2
%col2 = OpLoad %v3float %ptr_c2

; --- Tangents at start and end ---
%startT = OpFSub %v3float %P2_xyz %P0_xyz    ; P2 - P0
%endT = OpFSub %v3float %P3_xyz %P1_xyz      ; P3 - P1

; Prepare subdivision parameter vector: t = [0,0.25,0.5,0.75,1]
; We'll loop unrolled: compute for each t separately and emit two vertices

; ---- t = 0.0 ----
%t0 = %f_0
; pos = mix(P1,P2,t) => P1*(1-t) + P2*t, but for t=0 pos = P1
%one_minus_t0 = %f_1
%term1_0 = OpFMul %v3float %P1_xyz %one_minus_t0
%term2_0 = OpFMul %v3float %P2_xyz %t0
%pos0 = OpFAdd %v3float %term1_0 %term2_0
; tangent = normalize( (1-t)*startT + t*endT ) => for t=0 tangent = startT
%tan_mix_0 = %startT
%tan_len_0 = OpExtInst %float %GLSL.std.450 Length %tan_mix_0
%zero_check_0 = OpFOrdNotEqual %bool %tan_len_0 %f_0
; protect divide: if length==0 use (1,0,0)
%tan_norm_0 = OpSelect %v3float
    (OpCompositeConstruct %bool %zero_check_0)
    (OpFDiv %v3float %tan_mix_0 (OpCompositeConstruct %v3float %tan_len_0 %tan_len_0 %tan_len_0))
    (OpCompositeConstruct %v3float %f_1 %f_0 %f_0)
; take XY for perpendicular
%tan_xy_0 = OpVectorShuffle %v2float %tan_norm_0 %tan_norm_0 0 1
%perp2_0 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %tan_xy_0 1)) (OpCompositeExtract %float %tan_xy_0 0)
; normalize perp2
%perp_len_0 = OpExtInst %float %GLSL.std.450 Length %perp2_0
%perp_norm_0 = OpSelect %v2float
    (OpFOrdNotEqual %bool %perp_len_0 %f_0)
    (OpFDiv %v2float %perp2_0 (OpCompositeConstruct %v2float %perp_len_0 %perp_len_0))
    (OpCompositeConstruct %v2float %f_0 %f_0)
; apply thickness
%offset2_0 = OpFMul %v2float %perp_norm_0 (OpCompositeConstruct %v2float %thickness %thickness)
; build v4 offset (x,y,0,0)
%offset4_0 = OpCompositeConstruct %v4float (OpCompositeExtract %float %offset2_0 0) (OpCompositeExtract %float %offset2_0 1) %f_0 %f_0
; build pos4 = vec4(pos, 1.0)
%pos4_0 = OpCompositeConstruct %v4float (OpCompositeExtract %float %pos0 0) (OpCompositeExtract %float %pos0 1) (OpCompositeExtract %float %pos0 2) %f_1
; pos + offset
%emit_pos_a0 = OpFAdd %v4float %pos4_0 %offset4_0
%emit_pos_b0 = OpFSub %v4float %pos4_0 %offset4_0
; color interpolation col = mix(col1,col2,t)
%col_term1_0 = OpFMul %v3float %col1 %one_minus_t0
%col_term2_0 = OpFMul %v3float %col2 %t0
%col_out0 = OpFAdd %v3float %col_term1_0 %col_term2_0

; Emit first two vertices
; store gl_out.gl_Position = emit_pos_a0
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %emit_pos_a0
OpStore %outColor %col_out0
OpEmitVertex

; second vertex
OpStore %ptr_out_pos %emit_pos_b0
OpStore %outColor %col_out0
OpEmitVertex

; ---- t = 0.25 ----
%t1 = %f_025
%one_minus_t1 = OpFSub %float %f_1 %t1
%term1_1 = OpFMul %v3float %P1_xyz %one_minus_t1
%term2_1 = OpFMul %v3float %P2_xyz %t1
%pos1 = OpFAdd %v3float %term1_1 %term2_1
%tan_mix_1_pre1 = OpFMul %v3float %startT %one_minus_t1
%tan_mix_1_pre2 = OpFMul %v3float %endT %t1
%tan_mix_1 = OpFAdd %v3float %tan_mix_1_pre1 %tan_mix_1_pre2
%tan_len_1 = OpExtInst %float %GLSL.std.450 Length %tan_mix_1
%tan_norm_1 = OpSelect %v3float
    (OpFOrdNotEqual %bool %tan_len_1 %f_0)
    (OpFDiv %v3float %tan_mix_1 (OpCompositeConstruct %v3float %tan_len_1 %tan_len_1 %tan_len_1))
    (OpCompositeConstruct %v3float %f_1 %f_0 %f_0)
%tan_xy_1 = OpVectorShuffle %v2float %tan_norm_1 %tan_norm_1 0 1
%perp2_1 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %tan_xy_1 1)) (OpCompositeExtract %float %tan_xy_1 0)
%perp_len_1 = OpExtInst %float %GLSL.std.450 Length %perp2_1
%perp_norm_1 = OpSelect %v2float
    (OpFOrdNotEqual %bool %perp_len_1 %f_0)
    (OpFDiv %v2float %perp2_1 (OpCompositeConstruct %v2float %perp_len_1 %perp_len_1))
    (OpCompositeConstruct %v2float %f_0 %f_0)
%offset2_1 = OpFMul %v2float %perp_norm_1 (OpCompositeConstruct %v2float %thickness %thickness)
%offset4_1 = OpCompositeConstruct %v4float (OpCompositeExtract %float %offset2_1 0) (OpCompositeExtract %float %offset2_1 1) %f_0 %f_0
%pos4_1 = OpCompositeConstruct %v4float (OpCompositeExtract %float %pos1 0) (OpCompositeExtract %float %pos1 1) (OpCompositeExtract %float %pos1 2) %f_1
%emit_pos_a1 = OpFAdd %v4float %pos4_1 %offset4_1
%emit_pos_b1 = OpFSub %v4float %pos4_1 %offset4_1
%col_term1_1 = OpFMul %v3float %col1 %one_minus_t1
%col_term2_1 = OpFMul %v3float %col2 %t1
%col_out1 = OpFAdd %v3float %col_term1_1 %col_term2_1

OpStore %ptr_out_pos %emit_pos_a1
OpStore %outColor %col_out1
OpEmitVertex

OpStore %ptr_out_pos %emit_pos_b1
OpStore %outColor %col_out1
OpEmitVertex

; ---- t = 0.5 ----
%t2 = %f_05
%one_minus_t2 = OpFSub %float %f_1 %t2
%term1_2 = OpFMul %v3float %P1_xyz %one_minus_t2
%term2_2 = OpFMul %v3float %P2_xyz %t2
%pos2 = OpFAdd %v3float %term1_2 %term2_2
%tan_mix_2_pre1 = OpFMul %v3float %startT %one_minus_t2
%tan_mix_2_pre2 = OpFMul %v3float %endT %t2
%tan_mix_2 = OpFAdd %v3float %tan_mix_2_pre1 %tan_mix_2_pre2
%tan_len_2 = OpExtInst %float %GLSL.std.450 Length %tan_mix_2
%tan_norm_2 = OpSelect %v3float
    (OpFOrdNotEqual %bool %tan_len_2 %f_0)
    (OpFDiv %v3float %tan_mix_2 (OpCompositeConstruct %v3float %tan_len_2 %tan_len_2 %tan_len_2))
    (OpCompositeConstruct %v3float %f_1 %f_0 %f_0)
%tan_xy_2 = OpVectorShuffle %v2float %tan_norm_2 %tan_norm_2 0 1
%perp2_2 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %tan_xy_2 1)) (OpCompositeExtract %float %tan_xy_2 0)
%perp_len_2 = OpExtInst %float %GLSL.std.450 Length %perp2_2
%perp_norm_2 = OpSelect %v2float
    (OpFOrdNotEqual %bool %perp_len_2 %f_0)
    (OpFDiv %v2float %perp2_2 (OpCompositeConstruct %v2float %perp_len_2 %perp_len_2))
    (OpCompositeConstruct %v2float %f_0 %f_0)
%offset2_2 = OpFMul %v2float %perp_norm_2 (OpCompositeConstruct %v2float %thickness %thickness)
%offset4_2 = OpCompositeConstruct %v4float (OpCompositeExtract %float %offset2_2 0) (OpCompositeExtract %float %offset2_2 1) %f_0 %f_0
%pos4_2 = OpCompositeConstruct %v4float (OpCompositeExtract %float %pos2 0) (OpCompositeExtract %float %pos2 1) (OpCompositeExtract %float %pos2 2) %f_1
%emit_pos_a2 = OpFAdd %v4float %pos4_2 %offset4_2
%emit_pos_b2 = OpFSub %v4float %pos4_2 %offset4_2
%col_term1_2 = OpFMul %v3float %col1 %one_minus_t2
%col_term2_2 = OpFMul %v3float %col2 %t2
%col_out2 = OpFAdd %v3float %col_term1_2 %col_term2_2

OpStore %ptr_out_pos %emit_pos_a2
OpStore %outColor %col_out2
OpEmitVertex

OpStore %ptr_out_pos %emit_pos_b2
OpStore %outColor %col_out2
OpEmitVertex

; ---- t = 0.75 ----
%t3 = %f_075
%one_minus_t3 = OpFSub %float %f_1 %t3
%term1_3 = OpFMul %v3float %P1_xyz %one_minus_t3
%term2_3 = OpFMul %v3float %P2_xyz %t3
%pos3 = OpFAdd %v3float %term1_3 %term2_3
%tan_mix_3_pre1 = OpFMul %v3float %startT %one_minus_t3
%tan_mix_3_pre2 = OpFMul %v3float %endT %t3
%tan_mix_3 = OpFAdd %v3float %tan_mix_3_pre1 %tan_mix_3_pre2
%tan_len_3 = OpExtInst %float %GLSL.std.450 Length %tan_mix_3
%tan_norm_3 = OpSelect %v3float
    (OpFOrdNotEqual %bool %tan_len_3 %f_0)
    (OpFDiv %v3float %tan_mix_3 (OpCompositeConstruct %v3float %tan_len_3 %tan_len_3 %tan_len_3))
    (OpCompositeConstruct %v3float %f_1 %f_0 %f_0)
%tan_xy_3 = OpVectorShuffle %v2float %tan_norm_3 %tan_norm_3 0 1
%perp2_3 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %tan_xy_3 1)) (OpCompositeExtract %float %tan_xy_3 0)
%perp_len_3 = OpExtInst %float %GLSL.std.450 Length %perp2_3
%perp_norm_3 = OpSelect %v2float
    (OpFOrdNotEqual %bool %perp_len_3 %f_0)
    (OpFDiv %v2float %perp2_3 (OpCompositeConstruct %v2float %perp_len_3 %perp_len_3))
    (OpCompositeConstruct %v2float %f_0 %f_0)
%offset2_3 = OpFMul %v2float %perp_norm_3 (OpCompositeConstruct %v2float %thickness %thickness)
%offset4_3 = OpCompositeConstruct %v4float (OpCompositeExtract %float %offset2_3 0) (OpCompositeExtract %float %offset2_3 1) %f_0 %f_0
%pos4_3 = OpCompositeConstruct %v4float (OpCompositeExtract %float %pos3 0) (OpCompositeExtract %float %pos3 1) (OpCompositeExtract %float %pos3 2) %f_1
%emit_pos_a3 = OpFAdd %v4float %pos4_3 %offset4_3
%emit_pos_b3 = OpFSub %v4float %pos4_3 %offset4_3
%col_term1_3 = OpFMul %v3float %col1 %one_minus_t3
%col_term2_3 = OpFMul %v3float %col2 %t3
%col_out3 = OpFAdd %v3float %col_term1_3 %col_term2_3

OpStore %ptr_out_pos %emit_pos_a3
OpStore %outColor %col_out3
OpEmitVertex

OpStore %ptr_out_pos %emit_pos_b3
OpStore %outColor %col_out3
OpEmitVertex

; ---- t = 1.0 ----
%t4 = %f_1
%one_minus_t4 = OpFSub %float %f_1 %t4
%term1_4 = OpFMul %v3float %P1_xyz %one_minus_t4
%term2_4 = OpFMul %v3float %P2_xyz %t4
%pos4 = OpFAdd %v3float %term1_4 %term2_4
%tan_mix_4_pre1 = OpFMul %v3float %startT %one_minus_t4
%tan_mix_4_pre2 = OpFMul %v3float %endT %t4
%tan_mix_4 = OpFAdd %v3float %tan_mix_4_pre1 %tan_mix_4_pre2
%tan_len_4 = OpExtInst %float %GLSL.std.450 Length %tan_mix_4
%tan_norm_4 = OpSelect %v3float
    (OpFOrdNotEqual %bool %tan_len_4 %f_0)
    (OpFDiv %v3float %tan_mix_4 (OpCompositeConstruct %v3float %tan_len_4 %tan_len_4 %tan_len_4))
    (OpCompositeConstruct %v3float %f_1 %f_0 %f_0)
%tan_xy_4 = OpVectorShuffle %v2float %tan_norm_4 %tan_norm_4 0 1
%perp2_4 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %tan_xy_4 1)) (OpCompositeExtract %float %tan_xy_4 0)
%perp_len_4 = OpExtInst %float %GLSL.std.450 Length %perp2_4
%perp_norm_4 = OpSelect %v2float
    (OpFOrdNotEqual %bool %perp_len_4 %f_0)
    (OpFDiv %v2float %perp2_4 (OpCompositeConstruct %v2float %perp_len_4 %perp_len_4))
    (OpCompositeConstruct %v2float %f_0 %f_0)
%offset2_4 = OpFMul %v2float %perp_norm_4 (OpCompositeConstruct %v2float %thickness %thickness)
%offset4_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %offset2_4 0) (OpCompositeExtract %float %offset2_4 1) %f_0 %f_0
%pos4_4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %pos4 0) (OpCompositeExtract %float %pos4 1) (OpCompositeExtract %float %pos4 2) %f_1
%emit_pos_a4 = OpFAdd %v4float %pos4_4 %offset4_4
%emit_pos_b4 = OpFSub %v4float %pos4_4 %offset4_4
%col_term1_4 = OpFMul %v3float %col1 %one_minus_t4
%col_term2_4 = OpFMul %v3float %col2 %t4
%col_out4 = OpFAdd %v3float %col_term1_4 %col_term2_4

OpStore %ptr_out_pos %emit_pos_a4
OpStore %outColor %col_out4
OpEmitVertex

OpStore %ptr_out_pos %emit_pos_b4
OpStore %outColor %col_out4
OpEmitVertex

; End the strip (single primitive is now complete)
OpEndPrimitive

OpReturn
OpFunctionEnd

; GLSL.std.450 import for Length
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
