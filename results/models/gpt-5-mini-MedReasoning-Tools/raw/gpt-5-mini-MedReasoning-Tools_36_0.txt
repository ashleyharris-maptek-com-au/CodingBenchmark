Here's a single-file, complete C++ program (C++20-compatible) that computes an exact Maximum Independent Set (MIS). It uses an efficient exact maximum-clique branch-and-bound solver (Tomita-style with greedy coloring) on the complement graph, with bitset operations implemented using 64-bit words. The solver includes a simple preprocessing step to remove and always-include isolated vertices (which always belong to some MIS), and uses compact bit operations and pruning to scale well for modestly large graphs.

Copy, compile (MSVC /std:c++20), and run. It reads the graph from stdin in the format you specified and prints an exact MIS.

```cpp
#include <bits/stdc++.h>
using namespace std;

// Word type and helpers
using ULL = unsigned long long;

int n;
int words; // number of 64-bit words for bitsets
vector< vector<ULL> > adjComp; // adjacency bitsets for complement graph
vector<int> bestClique;        // best clique in complement -> MIS in original (on reduced graph)
vector<int> curClique;
vector<int> mapReducedToOriginal; // mapping from reduced vertex index to original index
vector<int> pos; // maps vertex id -> position in current candidate vs (or -1)

// fullMask for last word
vector<ULL> fullMask;

// Bitset helpers
inline void bitset_and(const vector<ULL>& a, const vector<ULL>& b, vector<ULL>& out) {
    for (int i = 0; i < words; ++i) out[i] = a[i] & b[i];
}
inline bool bitset_empty(const vector<ULL>& a) {
    for (int i = 0; i < words; ++i) if (a[i]) return false;
    return true;
}
inline void bitset_clear_bit(vector<ULL>& a, int idx) {
    a[idx >> 6] &= ~(1ULL << (idx & 63));
}
inline void bitset_set_bit(vector<ULL>& a, int idx) {
    a[idx >> 6] |= (1ULL << (idx & 63));
}
inline bool bitset_test(const vector<ULL>& a, int idx) {
    return (a[idx >> 6] >> (idx & 63)) & 1ULL;
}
inline int bitset_count(const vector<ULL>& a) {
    int c = 0;
    for (int i = 0; i < words; ++i) c += __builtin_popcountll(a[i]);
    return c;
}
inline void bitset_from_vertices(const vector<int>& vs, vector<ULL>& out) {
    for (int i = 0; i < words; ++i) out[i] = 0;
    for (int v : vs) out[v >> 6] |= (1ULL << (v & 63));
}
inline void bitset_to_vector(const vector<ULL>& a, vector<int>& out) {
    out.clear();
    for (int w = 0; w < words; ++w) {
        ULL x = a[w];
        while (x) {
            int b = __builtin_ctzll(x);
            out.push_back((w << 6) + b);
            x &= x - 1;
        }
    }
}

// Greedy coloring on candidate set 'vs' using adjComp (complement adjacency).
// Fills color[i] for vs[i]. Uses pos[] global to map vertex -> index in vs.
// Returns maximum color used.
int greedy_color(const vector<int>& vs, vector<int>& color) {
    int m = (int)vs.size();
    color.assign(m, 0);
    // We'll use a simple boolean vector for used colors; as colors are usually small, allocate (m+1).
    vector<char> used(m + 2, 0);
    int maxColor = 0;
    for (int i = 0; i < m; ++i) {
        int v = vs[i];
        // clear used marks
        // We only need to consider neighbors already colored (pos[u] < i)
        // mark colors used by earlier neighbors
        // For performance, we iterate neighbors of v intersected with candidate set, but check pos < i
        // We'll iterate bits in adjComp[v]
        // Reset used entries lazily by tracking which indices we set
        vector<int> touched;
        for (int w = 0; w < words; ++w) {
            ULL inter = adjComp[v][w];
            while (inter) {
                ULL lb = inter & -inter;
                int b = __builtin_ctzll(inter);
                int u = (w << 6) + b;
                int p = pos[u];
                if (p != -1 && p < i) {
                    int c = color[p];
                    if (!used[c]) {
                        used[c] = 1;
                        touched.push_back(c);
                    }
                }
                inter &= inter - 1;
            }
        }
        int c = 1;
        while (used[c]) ++c;
        color[i] = c;
        if (c > maxColor) maxColor = c;
        // reset used marks
        for (int tc : touched) used[tc] = 0;
    }
    return maxColor;
}

// The core branch-and-bound expansion on complement graph to find max clique
void expand(vector<ULL>& candBits, vector<int>& vs) {
    int m = (int)vs.size();
    if (m == 0) {
        if ((int)curClique.size() > (int)bestClique.size()) bestClique = curClique;
        return;
    }

    vector<int> color;
    // assign pos for vs
    for (int i = 0; i < m; ++i) pos[vs[i]] = i;
    // compute greedy coloring
    greedy_color(vs, color);

    // Process vertices in reverse order for better pruning (typical Tomita-style)
    for (int i = m - 1; i >= 0; --i) {
        int v = vs[i];
        // pruning: if current clique size + color[i] <= best found, no need to continue
        if ((int)curClique.size() + color[i] <= (int)bestClique.size()) {
            // reset pos for remaining vs and return
            for (int j = 0; j < i; ++j) pos[vs[j]] = -1;
            pos[v] = -1;
            return;
        }
        // choose v
        curClique.push_back(v);
        // new candidate = cand ∩ adjComp[v]
        vector<ULL> newCand(words);
        for (int w = 0; w < words; ++w) newCand[w] = candBits[w] & adjComp[v][w];

        if (bitset_empty(newCand)) {
            if ((int)curClique.size() > (int)bestClique.size()) {
                bestClique = curClique;
            }
        } else {
            // build new vs2 from newCand
            vector<int> vs2;
            bitset_to_vector(newCand, vs2);
            expand(newCand, vs2);
        }
        curClique.pop_back();
        // remove v from candBits and continue
        bitset_clear_bit(candBits, v);
        pos[v] = -1;
    }
}

// Entry: run max clique on entire complement graph (indices 0..n-1)
// Returns vector of chosen vertices (indices in reduced graph)
vector<int> maxCliqueAll() {
    // initial candidate set = all vertices 0..n-1
    vector<ULL> candBits(words);
    for (int i = 0; i < words; ++i) candBits[i] = fullMask[i];
    // clear bits beyond n
    if ((n & 63) != 0) {
        // last word already masked in fullMask
    }
    // ensure pos initialized
    pos.assign(n, -1);

    // prepare initial vs ordering: sort vertices by degree descending in complement graph
    vector<pair<int,int>> deg_v;
    deg_v.reserve(n);
    for (int v = 0; v < n; ++v) {
        int deg = 0;
        for (int w = 0; w < words; ++w) deg += __builtin_popcountll(adjComp[v][w]);
        deg_v.emplace_back(-deg, v);
    }
    sort(deg_v.begin(), deg_v.end());
    vector<int> vs;
    vs.reserve(n);
    for (auto &p : deg_v) vs.push_back(p.second);

    bestClique.clear();
    curClique.clear();
    expand(candBits, vs);
    return bestClique;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int m;
    if (!(cin >> n >> m)) return 0;

    vector<vector<int>> adjOrig(n, vector<int>());
    vector<vector<ULL>> adjOrigBits;

    // read edges
    vector<pair<int,int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        edges.emplace_back(u, v);
        adjOrig[u].push_back(v);
        adjOrig[v].push_back(u);
    }

    // handle trivial n==0
    if (n == 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    // Preprocessing: remove isolated vertices in original graph (they always belong to some MIS)
    vector<int> reducedIndex(n, -1);
    mapReducedToOriginal.clear();
    vector<char> isIsolated(n, 0);
    for (int v = 0; v < n; ++v) {
        if (adjOrig[v].empty()) isIsolated[v] = 1;
    }
    vector<int> forced; // isolated vertices (original indices) to include in final MIS
    for (int v = 0; v < n; ++v) {
        if (isIsolated[v]) forced.push_back(v);
    }

    // Build reduced graph excluding isolated vertices
    int nReduced = 0;
    for (int v = 0; v < n; ++v) {
        if (!isIsolated[v]) {
            reducedIndex[v] = nReduced++;
            mapReducedToOriginal.push_back(v);
        }
    }

    // Edge case: if all vertices isolated
    if (nReduced == 0) {
        cout << (int)forced.size() << "\n";
        for (size_t i = 0; i < forced.size(); ++i) {
            if (i) cout << ' ';
            cout << forced[i];
        }
        cout << "\n";
        return 0;
    }

    // Now set up complement adjacency on reduced graph (size nReduced)
    n = nReduced;
    words = (n + 63) >> 6;
    adjComp.assign(n, vector<ULL>(words, 0ULL));
    fullMask.assign(words, ~0ULL);
    if ((n & 63) != 0) {
        fullMask[words - 1] = ((1ULL << (n & 63)) - 1ULL);
    }

    // Build original adjacency bits for reduced graph
    vector<vector<ULL>> origBits(n, vector<ULL>(words, 0ULL));
    for (auto &e : edges) {
        int u0 = e.first, v0 = e.second;
        if (isIsolated[u0] || isIsolated[v0]) continue; // skip edges to isolated (they aren't in reduced graph)
        int u = reducedIndex[u0], v = reducedIndex[v0];
        origBits[u][v >> 6] |= (1ULL << (v & 63));
        origBits[v][u >> 6] |= (1ULL << (u & 63));
    }
    // complement adjacency: adjComp[v] = (~origBits[v]) & mask & bit_clear(v)
    for (int v = 0; v < n; ++v) {
        for (int w = 0; w < words; ++w) {
            adjComp[v][w] = (~origBits[v][w]) & fullMask[w];
        }
        // clear self-loop
        adjComp[v][v >> 6] &= ~(1ULL << (v & 63));
    }

    pos.assign(n, -1);

    // run max clique on complement graph (gives maximum independent set on reduced original)
    vector<int> clique = maxCliqueAll(); // indices in reduced graph

    // map clique vertices back to original, combine with forced isolated vertices
    vector<int> result;
    result.reserve(forced.size() + clique.size());
    for (int x : forced) result.push_back(x);
    for (int r : clique) result.push_back(mapReducedToOriginal[r]);

    // Output: k then list (any order)
    cout << result.size() << "\n";
    for (size_t i = 0; i < result.size(); ++i) {
        if (i) cout << ' ';
        cout << result[i];
    }
    cout << "\n";
    return 0;
}
```
