; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %ubo %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inUV "uv"
OpName %inTangent "tangent"
OpName %inColor "color"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpName %outColor "outColor"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%mat4 = OpTypeMatrix %v4f 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4 = OpTypePointer Uniform %v4f

%ptr_Input_v3 = OpTypePointer Input %v3f
%ptr_Input_v2 = OpTypePointer Input %v2f
%ptr_Input_v4 = OpTypePointer Input %v4f
%ptr_Output_v4 = OpTypePointer Output %v4f

; Inputs
%inWorldPos = OpVariable %ptr_Input_v3 Input
%inNormal = OpVariable %ptr_Input_v3 Input
%inUV = OpVariable %ptr_Input_v2 Input
%inTangent = OpVariable %ptr_Input_v3 Input
%inColor = OpVariable %ptr_Input_v3 Input

; Uniform buffer variable
%ubo = OpVariable %ptr_Uniform_UBO Uniform

; Output
%outColor = OpVariable %ptr_Output_v4 Output

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_05 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.4
%float_06 = OpConstant %float 0.6
%float_05m = OpConstant %float 0.05
%float_1m0 = OpConstant %float 0.1
%float_8 = OpConstant %float 8.0
%float_16 = OpConstant %float 16.0

%vec2_scale = OpConstantComposite %v2f %float_8 %float_16
%vec3_mortar = OpConstantComposite %v3f %float_1 %float_1 %float_1        ; placeholder, will set actual mortar color below
%vec3_brick = OpConstantComposite %v3f %float_1 %float_1 %float_1         ; placeholder, will set actual brick color below

; Because OpConstantComposite requires constants, define mortar and brick components
%mortar_r = OpConstant %float 0.7
%mortar_g = OpConstant %float 0.7
%mortar_b = OpConstant %float 0.7
%brick_r = OpConstant %float 0.7
%brick_g = OpConstant %float 0.2
%brick_b = OpConstant %float 0.1

%mortarColor = OpConstantComposite %v3f %mortar_r %mortar_g %mortar_b
%brickColor = OpConstantComposite %v3f %brick_r %brick_g %brick_b

; One minus mortar constants for comparisons
%one_f = OpConstant %float 1.0
%oneMinusMortarX = OpFSub %float %one_f %float_05m
%oneMinusMortarY = OpFSub %float %one_f %float_1m0

; GLSL.std.450 import
%glsl = OpExtInstImport "GLSL.std.450"

; Function main
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %inWorldPos
%normal = OpLoad %v3f %inNormal
%uv = OpLoad %v2f %inUV

; Normalize normal
%normal_n = OpExtInst %v3f %glsl Normalize %normal

; Load light position from UBO (member 3)
%light_ptr = OpAccessChain %ptr_Uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4f %light_ptr
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

; Compute light direction (lightPos - worldPos) and normalize
%ldir_pre = OpFSub %v3f %light_xyz %worldPos
%ldir = OpExtInst %v3f %glsl Normalize %ldir_pre

; Compute diffuse = max(dot(n, l), 0.0)
%dot_nl = OpExtInst %float %glsl Dot %normal_n %ldir
%diffuse = OpFMax %float %float_0 %dot_nl

; shading factor = 0.4 + 0.6 * diffuse
%tmp_mul = OpFMul %float %float_06 %diffuse
%shade = OpFAdd %float %float_04 %tmp_mul

; Scale UV by (8,16)
%uv_scaled = OpFMul %v2f %uv %vec2_scale

; Extract scaled components
%uvsx = OpCompositeExtract %float %uv_scaled 0
%uvsy = OpCompositeExtract %float %uv_scaled 1

; Compute row = floor(uv_scaled.y)
%rowf = OpExtInst %float %glsl Floor %uvsy
%rowi = OpConvertFToS %int %rowf

; row mod 2
%rowmod2 = OpSMod %int %rowi %int_2
%isOdd = OpIEqual %bool %rowmod2 %int_1

; x offset = isOdd ? 0.5 : 0.0
%xoffset = OpSelect %float %isOdd %float_05 %float_0

; apply x offset to uv.x
%uvsx_off = OpFAdd %float %uvsx %xoffset

; Reconstruct scaled UV with offset
%uv_off = OpCompositeConstruct %v2f %uvsx_off %uvsy

; fract to get cell local coords
%cell = OpExtInst %v2f %glsl Fract %uv_off

; Extract cell x,y
%cell_x = OpCompositeExtract %float %cell 0
%cell_y = OpCompositeExtract %float %cell 1

; Mortar checks: x < mortarX or x > 1-mortarX
%cond_x_low = OpFOrdLessThan %bool %cell_x %float_05m
%cond_x_high = OpFOrdGreaterThan %bool %cell_x %oneMinusMortarX
%cond_x = OpLogicalOr %bool %cond_x_low %cond_x_high

; y checks
%cond_y_low = OpFOrdLessThan %bool %cell_y %float_1m0
%cond_y_high = OpFOrdGreaterThan %bool %cell_y %oneMinusMortarY
%cond_y = OpLogicalOr %bool %cond_y_low %cond_y_high

; final mortar flag
%isMortar = OpLogicalOr %bool %cond_x %cond_y

; choose color
%baseColor = OpSelect %v3f %isMortar %mortarColor %brickColor

; apply shading factor (multiply vec3 by scalar)
%shadeVec = OpCompositeConstruct %v3f %shade %shade %shade
%litColor = OpFMul %v3f %baseColor %shadeVec

; build output vec4 with alpha = 1.0
%lit_x = OpCompositeExtract %float %litColor 0
%lit_y = OpCompositeExtract %float %litColor 1
%lit_z = OpCompositeExtract %float %litColor 2
%outVec = OpCompositeConstruct %v4f %lit_x %lit_y %lit_z %float_1

; store to output
OpStore %outColor %outVec

OpReturn
OpFunctionEnd
