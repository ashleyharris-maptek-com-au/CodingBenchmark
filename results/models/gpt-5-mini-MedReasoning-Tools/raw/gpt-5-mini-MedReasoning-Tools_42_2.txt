; SPIR-V
; Geometry shader: expand each input point into a hexagon made of 6 triangles (triangle strips of 3 verts each)
; Input primitive: InputPoints
; Output primitive: OutputTriangleStrip
; OutputVertices: 18  (6 triangles * 3 vertices)
; Invocations: 1

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
OpTypeVoid %void
OpTypeFunction %void %void

OpTypeBool %bool

OpTypeInt %int 32 1
OpTypeInt %uint 32 0
OpConstant %int %int_0 0
OpConstant %int %int_1 1

OpTypeFloat %float 32
OpConstant %float %f_0 0.0
OpConstant %float %f_1 1.0
OpConstant %float %f_0_5 0.5
OpConstant %float %f_0_8660254 0.8660254037844386
OpConstant %float %f_minus_0_5 -0.5
OpConstant %float %f_minus_0_8660254 -0.8660254037844386
OpConstant %float %f_minus1 -1.0
OpConstant %float %f_0_8 0.8
OpConstant %float %f_r 0.04

OpTypeVector %v2float %float 2
OpTypeVector %v3float %float 3
OpTypeVector %v4float %float 4

; vector constants for hexagon directions (unit circle samples)
OpConstantComposite %v2float %v2_1_0 %f_1 %f_0
OpConstantComposite %v2float %v2_0_5_0_866 %f_0_5 %f_0_8660254
OpConstantComposite %v2float %v2_m0_5_0_866 %f_minus_0_5 %f_0_8660254
OpConstantComposite %v2float %v2_m1_0 %f_minus1 %f_0
OpConstantComposite %v2float %v2_m0_5_m0_866 %f_minus_0_5 %f_minus_0_8660254
OpConstantComposite %v2float %v2_0_5_m0_866 %f_0_5 %f_minus_0_8660254

; Integer constants for array sizes and indices
OpConstant %uint %uint_0 0
OpConstant %uint %uint_1 1

; gl_PerVertex struct types (input array element and output single)
%v4p = %v4float
%gl_PerVertex_in = OpTypeStruct %v4p
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4p
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Array size for input arrays (InputPoints -> array size 1)
%arr_size_1 = %uint_1
%arr_glpv = OpTypeArray %gl_PerVertex_in %arr_size_1
%ptr_in_arr_glpv = OpTypePointer Input %arr_glpv
%gl_in = OpVariable %ptr_in_arr_glpv Input

; Input per-vertex arrays (location decorated)
%arr_v3 = OpTypeArray %v3float %arr_size_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3

%in_worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; Output per-vertex gl_PerVertex (single struct)
%ptr_out_glpv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_glpv Output

; Output varyings
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Function
%main = OpFunction %void None %functype
%func_label = OpLabel

; Local temporaries will be created as IDs by operations; no function-scope variables required.

; Load the input position and color (index 0 because InputPoints -> array size 1)
%ptr_world_elem = OpAccessChain %ptr_in_arr_v3 %in_worldPos %uint_0
%world_v3 = OpLoad %v3float %ptr_world_elem

%ptr_color_elem = OpAccessChain %ptr_in_arr_v3 %in_color %uint_0
%color_v3 = OpLoad %v3float %ptr_color_elem

; build center position as vec4(world, 1.0)
%cx = OpCompositeExtract %float %world_v3 0
%cy = OpCompositeExtract %float %world_v3 1
%cz = OpCompositeExtract %float %world_v3 2
%center_v4 = OpCompositeConstruct %v4float %cx %cy %cz %f_1

; compute darker edge color = color * 0.8
%edgeColor_v3 = OpVectorTimesScalar %v3float %color_v3 %f_0_8

; We will emit 6 triangle strips (each triangle as a 3-vertex strip): center, edge_i, edge_{i+1}
; Helper: pointer to gl_out Position member
%ptr_out_pos = OpAccessChain %ptr_out_glpv %gl_out %int_0

; --- Triangle 0: center, edge0 (0°), edge1 (60°) ---
; store center
OpStore %ptr_out_pos %center_v4
OpStore %outColor %color_v3
OpEmitVertex

; compute edge0
%center_xy = OpVectorShuffle %v2float %world_v3 %world_v3 0 1
%off0 = OpVectorTimesScalar %v2float %v2_1_0 %f_r
%edge0_xy = OpFAdd %v2float %center_xy %off0
%e0_x = OpCompositeExtract %float %edge0_xy 0
%e0_y = OpCompositeExtract %float %edge0_xy 1
%edge0_v4 = OpCompositeConstruct %v4float %e0_x %e0_y %cz %f_1
OpStore %ptr_out_pos %edge0_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

; compute edge1
%off1 = OpVectorTimesScalar %v2float %v2_0_5_0_866 %f_r
%edge1_xy = OpFAdd %v2float %center_xy %off1
%e1_x = OpCompositeExtract %float %edge1_xy 0
%e1_y = OpCompositeExtract %float %edge1_xy 1
%edge1_v4 = OpCompositeConstruct %v4float %e1_x %e1_y %cz %f_1
OpStore %ptr_out_pos %edge1_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

OpEndPrimitive

; --- Triangle 1: center, edge1, edge2 ---
; center
OpStore %ptr_out_pos %center_v4
OpStore %outColor %color_v3
OpEmitVertex

; edge1 (reuse previously computed edge1_v4)
OpStore %ptr_out_pos %edge1_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

; edge2
%off2 = OpVectorTimesScalar %v2float %v2_m0_5_0_866 %f_r
%edge2_xy = OpFAdd %v2float %center_xy %off2
%e2_x = OpCompositeExtract %float %edge2_xy 0
%e2_y = OpCompositeExtract %float %edge2_xy 1
%edge2_v4 = OpCompositeConstruct %v4float %e2_x %e2_y %cz %f_1
OpStore %ptr_out_pos %edge2_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

OpEndPrimitive

; --- Triangle 2: center, edge2, edge3 ---
OpStore %ptr_out_pos %center_v4
OpStore %outColor %color_v3
OpEmitVertex

OpStore %ptr_out_pos %edge2_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

%off3 = OpVectorTimesScalar %v2float %v2_m1_0 %f_r
%edge3_xy = OpFAdd %v2float %center_xy %off3
%e3_x = OpCompositeExtract %float %edge3_xy 0
%e3_y = OpCompositeExtract %float %edge3_xy 1
%edge3_v4 = OpCompositeConstruct %v4float %e3_x %e3_y %cz %f_1
OpStore %ptr_out_pos %edge3_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

OpEndPrimitive

; --- Triangle 3: center, edge3, edge4 ---
OpStore %ptr_out_pos %center_v4
OpStore %outColor %color_v3
OpEmitVertex

OpStore %ptr_out_pos %edge3_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

%off4 = OpVectorTimesScalar %v2float %v2_m0_5_m0_866 %f_r
%edge4_xy = OpFAdd %v2float %center_xy %off4
%e4_x = OpCompositeExtract %float %edge4_xy 0
%e4_y = OpCompositeExtract %float %edge4_xy 1
%edge4_v4 = OpCompositeConstruct %v4float %e4_x %e4_y %cz %f_1
OpStore %ptr_out_pos %edge4_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

OpEndPrimitive

; --- Triangle 4: center, edge4, edge5 ---
OpStore %ptr_out_pos %center_v4
OpStore %outColor %color_v3
OpEmitVertex

OpStore %ptr_out_pos %edge4_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

%off5 = OpVectorTimesScalar %v2float %v2_0_5_m0_866 %f_r
%edge5_xy = OpFAdd %v2float %center_xy %off5
%e5_x = OpCompositeExtract %float %edge5_xy 0
%e5_y = OpCompositeExtract %float %edge5_xy 1
%edge5_v4 = OpCompositeConstruct %v4float %e5_x %e5_y %cz %f_1
OpStore %ptr_out_pos %edge5_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

OpEndPrimitive

; --- Triangle 5: center, edge5, edge0 (close the hexagon) ---
OpStore %ptr_out_pos %center_v4
OpStore %outColor %color_v3
OpEmitVertex

OpStore %ptr_out_pos %edge5_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

; reuse edge0_v4 computed earlier
OpStore %ptr_out_pos %edge0_v4
OpStore %outColor %edgeColor_v3
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd

; Entry point and execution modes - list all interface variables
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; End of SPIR-V assembly
