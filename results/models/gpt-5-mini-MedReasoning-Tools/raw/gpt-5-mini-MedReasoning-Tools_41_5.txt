; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %ubo %outColor
OpExecutionMode %main OriginUpperLeft

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; Pointer types
%ptr_Input_v3 = OpTypePointer Input %v3f
%ptr_Input_v2 = OpTypePointer Input %v2f
%ptr_Input_v4 = OpTypePointer Input %v4f
%ptr_Output_v4 = OpTypePointer Output %v4f
%ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%ptr_Uniform_v4 = OpTypePointer Uniform %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %_arr_ubotype_dummy ; placeholder will be replaced below

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_32 = OpConstant %float 32.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

; Integer type (for indices)
%int = OpTypeInt 32 1

; Define correct pointer type for UBO (we need struct first)
; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%ubo_mat4 = %mat4
%ubo_v4 = %v4f
%UBO = OpTypeStruct %ubo_mat4 %ubo_mat4 %ubo_mat4 %ubo_v4 %ubo_v4 %ubo_v4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; Now override previous placeholder binding pointer (not strictly necessary but keep definitions consistent)
; Variables (Inputs / Outputs / Uniforms)
%inWorldPos = OpVariable %ptr_Input_v3 Input
%inNormal = OpVariable %ptr_Input_v3 Input
%inUV = OpVariable %ptr_Input_v2 Input
%inTangent = OpVariable %ptr_Input_v3 Input
%inColor = OpVariable %ptr_Input_v3 Input
%outColor = OpVariable %ptr_Output_v4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

; Decorations: Locations for inputs and output
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Names (optional but helpful)
OpName %main "main"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"

; --- Function main ---
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%wpos = OpLoad %v3f %inWorldPos
%n_in = OpLoad %v3f %inNormal

; Load lightPos (vec4) from UBO (member 3)
%lightPtr = OpAccessChain %ptr_Uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4f %lightPtr
%l_x = OpCompositeExtract %float %light_v4 0
%l_y = OpCompositeExtract %float %light_v4 1
%l_z = OpCompositeExtract %float %light_v4 2
%light_xyz = OpCompositeConstruct %v3f %l_x %l_y %l_z

; Load cameraPos (vec4) from UBO (member 4)
%camPtr = OpAccessChain %ptr_Uniform_v4 %ubo %int_4
%cam_v4 = OpLoad %v4f %camPtr
%c_x = OpCompositeExtract %float %cam_v4 0
%c_y = OpCompositeExtract %float %cam_v4 1
%c_z = OpCompositeExtract %float %cam_v4 2
%cam_xyz = OpCompositeConstruct %v3f %c_x %c_y %c_z

; Compute L = normalize(lightPos - worldPos)
%diffL = OpFSub %v3f %light_xyz %wpos
%L = OpExtInst %v3f %GLSLstd450 Normalize %diffL

; Compute N = normalize(normal)
%N = OpExtInst %v3f %GLSLstd450 Normalize %n_in

; Compute V = normalize(cameraPos - worldPos)
%diffV = OpFSub %v3f %cam_xyz %wpos
%V = OpExtInst %v3f %GLSLstd450 Normalize %diffV

; diffuse = max(dot(N,L), 0.0)
%dotNL = OpDot %float %N %L
%diffuse = OpFMax %float %dotNL %float_0

; R = reflect(-L, N)
%negL = OpFNegate %v3f %L
%R = OpExtInst %v3f %GLSLstd450 Reflect %negL %N

; specular = pow(max(dot(R,V),0), 32)
%dotRV = OpDot %float %R %V
%specBase = OpFMax %float %dotRV %float_0
%spec = OpExtInst %float %GLSLstd450 Pow %specBase %float_32

; intensity = 0.2 + 0.5*diffuse + 0.8*spec
%tmp1 = OpFMul %float %float_0_5 %diffuse
%tmp2 = OpFMul %float %float_0_8 %spec
%tmp3 = OpFAdd %float %tmp1 %tmp2
%intensity = OpFAdd %float %float_0_2 %tmp3

; build final color vec4(intensity, intensity, intensity, 1.0)
%col3 = OpCompositeConstruct %v3f %intensity %intensity %intensity
%finalColor = OpCompositeConstruct %v4f %intensity %intensity %intensity %float_1

; store to output
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd
