; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%1 = OpExtInstImport "GLSL.std.450"

OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%functype = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%bool = OpTypeBool

; Pointers
%ptr_input_v3 = OpTypePointer Input %v3f
%ptr_input_v2 = OpTypePointer Input %v2f
%ptr_output_v4 = OpTypePointer Output %v4f
%ptr_uniform_v4 = OpTypePointer Uniform %v4f

; UBO struct and pointer
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_uniform_UBO = OpTypePointer Uniform %UBO

; Variables
%inWorldPos = OpVariable %ptr_input_v3 Input
%inNormal = OpVariable %ptr_input_v3 Input
%inColor = OpVariable %ptr_input_v3 Input

%outColor = OpVariable %ptr_output_v4 Output

%ubo = OpVariable %ptr_uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_075 = OpConstant %float 0.75
%float_05 = OpConstant %float 0.5
%float_025 = OpConstant %float 0.25
%float_07 = OpConstant %float 0.7
%float_04 = OpConstant %float 0.4
%float_02 = OpConstant %float 0.2

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

; Entry point
%main = OpFunction %void None %functype
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %inWorldPos
%n = OpLoad %v3f %inNormal
%vc = OpLoad %v3f %inColor

; Normalize normal
%n_norm = OpExtInst %v3f %1 Normalize %n

; Load light position (vec4) from UBO member 3
%light_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4f %light_ptr

; Extract xyz of light
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

; Compute light direction L = normalize(light.xyz - worldPos)
%L_un = OpFSub %v3f %light_xyz %wp
%L = OpExtInst %v3f %1 Normalize %L_un

; Lambertian dot
%dot = OpDot %float %n_norm %L

; clamp to >= 0
%d = OpFMax %float %dot %float_0

; Compare against thresholds
%cond1 = OpFOrdGreaterThanEqual %bool %d %float_075
%cond2 = OpFOrdGreaterThanEqual %bool %d %float_05
%cond3 = OpFOrdGreaterThanEqual %bool %d %float_025

; Nested selects for bands:
;s3 = cond3 ? 0.4 : 0.2
%s3 = OpSelect %float %cond3 %float_04 %float_02
%s2 = cond2 ? 0.7 : s3
%s2 = OpSelect %float %cond2 %float_07 %s3
;band = cond1 ? 1.0 : s2
%s3
%s2
%_s3 = OpSelect %float %cond3 %float_04 %float_02
%_s2 = OpSelect %float %cond2 %float_07 %_s3
%band = OpSelect %float %cond1 %float_1 %_s2

; Multiply vertex color by band
%band_vec = OpCompositeConstruct %v3f %band %band %band
%rgb = OpFMul %v3f %vc %band_vec

; Build vec4 output (rgb, 1.0)
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%outv = OpCompositeConstruct %v4f %r %g %b %float_1

OpStore %outColor %outv

OpReturn
OpFunctionEnd
