struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4   lightPos;
    float4   cameraPos;
    float4   params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Hemisphere ambient blend factor: N.y * 0.5 + 0.5
    float hemiFactor = clamp(N.y * 0.5 + 0.5, 0.0, 1.0);

    // Colors
    float3 skyColor   = float3(0.4, 0.6, 1.0);
    float3 groundColor= float3(0.3, 0.15, 0.05);

    // Hemisphere ambient color
    float3 hemi = lerp(groundColor, skyColor, hemiFactor);

    // Small diffuse from point light
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float d = max(dot(N, L), 0.0);
    float3 diffuseTerm = 0.3 * d;

    // Apply to albedo (per-vertex color)
    float3 albedo = input.color;
    float3 finalColor = albedo * (hemi + diffuseTerm);

    finalColor = clamp(finalColor, 0.0, 1.0);
    return float4(finalColor, 1.0);
}
