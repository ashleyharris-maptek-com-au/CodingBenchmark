Here's a complete, self-contained C++20 solver that implements two packing strategies and automatically selects between them based on problem scale:

- For small-to-medium problems (N <= 2000) it uses a MaxRects (Best Short Side Fit) guillotine-free approach for better packing quality.
- For large problems (N > 2000) it uses a very fast shelf-first-fit-decreasing-height style heuristic (with rotation allowed), which is linear-ish and scales to 50k rectangles under the time limit.

It respects rotation, prevents overlap, and prints placements in the original rectangle order.

Copy, compile (MSVC /std:c++20) and run. The program reads stdin and writes stdout in the requested format.

```cpp
#include <bits/stdc++.h>
using namespace std;

struct RectReq {
    int w, h;
    int idx;
};

struct Placement {
    int board;
    int x, y;
    int rot; // 0 or 1
    Placement(): board(-1), x(0), y(0), rot(0) {}
};

//
// MAXRECTS BIN (for N <= threshold)
//
// A simplified MaxRects bin using Best Short Side Fit heuristic.
// Supports querying the best placement (without modifying),
// and placing (which updates free rects).
//
struct FreeRect {
    int x, y, w, h;
    FreeRect(int _x=0,int _y=0,int _w=0,int _h=0): x(_x),y(_y),w(_w),h(_h) {}
};

struct BestChoice {
    bool fit;
    int freeIndex;
    int x, y;
    int rw, rh;
    int rot; // 0 or 1
    int shortSide;
    int longSide;
    int areaFit;
    BestChoice(): fit(false), freeIndex(-1), x(0), y(0), rw(0), rh(0), rot(0), shortSide(INT_MAX), longSide(INT_MAX), areaFit(INT_MAX) {}
};

struct MaxRectsBin {
    int W, H;
    vector<FreeRect> freeRects;
    MaxRectsBin(int w=0,int h=0): W(w), H(h) {
        freeRects.clear();
        freeRects.emplace_back(0,0,W,H);
    }

    // find best position (without modifying)
    BestChoice findBest(int rw, int rh) const {
        BestChoice best;
        for (size_t i=0;i<freeRects.size();++i) {
            const FreeRect &fr = freeRects[i];
            if (rw <= fr.w && rh <= fr.h) {
                int leftoverHoriz = fr.w - rw;
                int leftoverVert = fr.h - rh;
                int shortSide = min(leftoverHoriz, leftoverVert);
                int longSide = max(leftoverHoriz, leftoverVert);
                int areaFit = fr.w * fr.h - rw * rh;
                bool better = false;
                if (!best.fit) better = true;
                else if (shortSide < best.shortSide) better = true;
                else if (shortSide == best.shortSide && longSide < best.longSide) better = true;
                else if (shortSide == best.shortSide && longSide == best.longSide && areaFit < best.areaFit) better = true;
                if (better) {
                    best.fit = true;
                    best.freeIndex = (int)i;
                    best.x = fr.x;
                    best.y = fr.y;
                    best.rw = rw; best.rh = rh;
                    best.rot = 0;
                    best.shortSide = shortSide;
                    best.longSide = longSide;
                    best.areaFit = areaFit;
                }
            }
            // try rotated
            if (rh <= fr.w && rw <= fr.h) {
                int rrw = rh, rrh = rw;
                int leftoverHoriz = fr.w - rrw;
                int leftoverVert = fr.h - rrh;
                int shortSide = min(leftoverHoriz, leftoverVert);
                int longSide = max(leftoverHoriz, leftoverVert);
                int areaFit = fr.w * fr.h - rrw * rrh;
                bool better = false;
                if (!best.fit) better = true;
                else if (shortSide < best.shortSide) better = true;
                else if (shortSide == best.shortSide && longSide < best.longSide) better = true;
                else if (shortSide == best.shortSide && longSide == best.longSide && areaFit < best.areaFit) better = true;
                if (better) {
                    best.fit = true;
                    best.freeIndex = (int)i;
                    best.x = fr.x;
                    best.y = fr.y;
                    best.rw = rrw; best.rh = rrh;
                    best.rot = 1;
                    best.shortSide = shortSide;
                    best.longSide = longSide;
                    best.areaFit = areaFit;
                }
            }
        }
        return best;
    }

    static bool rectContains(const FreeRect &a, const FreeRect &b) {
        return b.x >= a.x && b.y >= a.y && (b.x + b.w) <= (a.x + a.w) && (b.y + b.h) <= (a.y + a.h);
    }

    void splitFreeRectByPlaced(const FreeRect &freeR, const FreeRect &placed) {
        // Split freeR into up to 4 remaining rectangles around placed
        if (placed.x >= freeR.x + freeR.w || placed.x + placed.w <= freeR.x ||
            placed.y >= freeR.y + freeR.h || placed.y + placed.h <= freeR.y)
            return; // no overlap

        // left
        if (placed.x > freeR.x) {
            freeRects.emplace_back(freeR.x, freeR.y, placed.x - freeR.x, freeR.h);
        }
        // right
        if (placed.x + placed.w < freeR.x + freeR.w) {
            int nx = placed.x + placed.w;
            freeRects.emplace_back(nx, freeR.y, (freeR.x + freeR.w) - nx, freeR.h);
        }
        // top
        if (placed.y > freeR.y) {
            freeRects.emplace_back(freeR.x, freeR.y, freeR.w, placed.y - freeR.y);
        }
        // bottom
        if (placed.y + placed.h < freeR.y + freeR.h) {
            int ny = placed.y + placed.h;
            freeRects.emplace_back(freeR.x, ny, freeR.w, (freeR.y + freeR.h) - ny);
        }
    }

    void pruneFreeList() {
        // remove any free rect that is contained in another
        for (size_t i=0;i<freeRects.size();++i) {
            for (size_t j=i+1;j<freeRects.size();) {
                if (rectContains(freeRects[i], freeRects[j])) {
                    freeRects.erase(freeRects.begin()+j);
                } else if (rectContains(freeRects[j], freeRects[i])) {
                    freeRects.erase(freeRects.begin()+i);
                    --i;
                    break;
                } else ++j;
            }
        }
    }

    // Place (assumes findBest said fit exists). Performs splitting and pruning.
    bool placeRectAt(const BestChoice &b) {
        if (!b.fit) return false;
        FreeRect placed(b.x, b.y, b.rw, b.rh);
        // find free rect that we used (by index)
        // But the freeRects may have changed since we computed index; we trust we call this right away after find.
        if (b.freeIndex < 0 || b.freeIndex >= (int)freeRects.size()) {
            // fallback: find any freeRect that contains it
            int found = -1;
            for (size_t i=0;i<freeRects.size();++i) {
                const FreeRect &fr = freeRects[i];
                if (placed.x >= fr.x && placed.y >= fr.y &&
                    placed.x + placed.w <= fr.x + fr.w &&
                    placed.y + placed.h <= fr.y + fr.h) { found = (int)i; break; }
            }
            if (found == -1) return false;
            // split using found
            FreeRect fr = freeRects[found];
            freeRects.erase(freeRects.begin()+found);
            splitFreeRectByPlaced(fr, placed);
        } else {
            FreeRect fr = freeRects[b.freeIndex];
            // remove the free rect used
            freeRects.erase(freeRects.begin()+b.freeIndex);
            splitFreeRectByPlaced(fr, placed);
        }
        // prune contained
        pruneFreeList();
        return true;
    }
};

//
// SHELF-STYLE PACKER (for N > threshold)
// A fast heuristic: FFDH-like across multiple boards.
//
struct Shelf {
    int y; // top y
    int height;
    int xcur; // current x (next placement)
    int remainingW;
    Shelf(int _y=0,int _h=0,int _w=0): y(_y), height(_h), xcur(0), remainingW(_w) {}
};

struct ShelfBoard {
    int W, H;
    int usedH;
    vector<Shelf> shelves;
    ShelfBoard(int w=0,int h=0): W(w), H(h), usedH(0) { shelves.clear(); }

    // Try to place with shelf approach. Returns placement if success.
    bool tryPlace(int rw, int rh, int &outx, int &outy, int &outrot) {
        // try to fit in existing shelf
        for (size_t i=0;i<shelves.size();++i) {
            Shelf &s = shelves[i];
            // try orientation 0
            if (rh <= s.height && rw <= s.remainingW) {
                outx = s.xcur;
                outy = s.y;
                outrot = 0;
                s.xcur += rw;
                s.remainingW -= rw;
                return true;
            }
            // try rotated (swap)
            if (rw <= s.height && rh <= s.remainingW) {
                outx = s.xcur;
                outy = s.y;
                outrot = 1;
                s.xcur += rh;
                s.remainingW -= rh;
                return true;
            }
        }
        // need a new shelf
        int chooseH0 = rh;
        int chooseH1 = rw;
        int chooseH = min(chooseH0, chooseH1);
        if (usedH + chooseH > H) return false;
        // choose orientation that gives smaller shelf height, but ensure width fits
        int chosenRot = 0;
        int pw = rw, ph = rh;
        if (chooseH == chooseH1) { chosenRot = 1; pw = rh; ph = rw; }
        // If width doesn't fit in board width when oriented that way, try other orientation
        if (pw > W) {
            // try other
            chosenRot = 1 - chosenRot;
            if (chosenRot == 0) { pw = rw; ph = rh; } else { pw = rh; ph = rw; }
            if (pw > W || usedH + ph > H) return false;
        }
        // create shelf of height ph
        Shelf s(usedH, ph, W);
        outx = s.xcur;
        outy = s.y;
        outrot = chosenRot;
        s.xcur += pw;
        s.remainingW -= pw;
        shelves.push_back(s);
        usedH += ph;
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    int boardW, boardH;
    if (!(cin >> N >> boardW >> boardH)) return 0;
    vector<RectReq> reqs;
    reqs.reserve(N);
    for (int i=0;i<N;++i) {
        int w,h; cin >> w >> h;
        reqs.push_back({w,h,i});
    }

    // Validate feasibility: each rectangle must fit in board in at least one orientation
    for (const auto &r : reqs) {
        if (!((r.w <= boardW && r.h <= boardH) || (r.h <= boardW && r.w <= boardH))) {
            // Can't fit this rectangle in any orientation. Output each on its own board with coords 0,0 if possible else fail.
            // But since it's impossible to place, we'll exit with zero boards to indicate failure.
            // (Problem statement doesn't define impossible-case handling.)
            cerr << "A rectangle cannot fit in the board in any orientation. Exiting.\n";
            cout << 0 << "\n";
            for (int i=0;i<N;++i) cout << "0 0 0 0\n";
            return 0;
        }
    }

    vector<Placement> placements(N);

    const int MAXRECTS_THRESHOLD = 2000; // switch point, tunable

    if (N <= MAXRECTS_THRESHOLD) {
        // Use MaxRects bins with Best Short Side Fit
        vector<MaxRectsBin> bins;
        bins.reserve(64);
        for (const auto &r : reqs) {
            // We'll use a heuristic: try to find best candidate among existing bins (best short side).
            bool placed = false;
            int bestBin = -1;
            BestChoice bestChoiceGlobal;
            for (size_t b=0;b<bins.size();++b) {
                BestChoice bc = bins[b].findBest(r.w, r.h);
                if (!bc.fit) continue;
                // pick best across bins
                if (!bestChoiceGlobal.fit) {
                    bestChoiceGlobal = bc;
                    bestBin = (int)b;
                } else {
                    // lexicographic compare shortSide, longSide, areaFit
                    if (make_tuple(bc.shortSide, bc.longSide, bc.areaFit) <
                        make_tuple(bestChoiceGlobal.shortSide, bestChoiceGlobal.longSide, bestChoiceGlobal.areaFit)) {
                        bestChoiceGlobal = bc;
                        bestBin = (int)b;
                    }
                }
            }
            if (bestChoiceGlobal.fit) {
                // Place in bestBin
                bins[bestBin].placeRectAt(bestChoiceGlobal);
                placements[r.idx].board = bestBin;
                placements[r.idx].x = bestChoiceGlobal.x;
                placements[r.idx].y = bestChoiceGlobal.y;
                placements[r.idx].rot = bestChoiceGlobal.rot;
                placed = true;
            } else {
                // open new bin and place
                bins.emplace_back(boardW, boardH);
                MaxRectsBin &nb = bins.back();
                BestChoice bc = nb.findBest(r.w, r.h);
                if (!bc.fit) {
                    // Should not happen because we validated fit earlier
                    // Try brute force placement at 0,0 with rotation if needed
                    if (r.w <= nb.W && r.h <= nb.H) {
                        bc.fit = true; bc.x = 0; bc.y = 0; bc.rw = r.w; bc.rh = r.h; bc.rot = 0;
                    } else if (r.h <= nb.W && r.w <= nb.H) {
                        bc.fit = true; bc.x = 0; bc.y = 0; bc.rw = r.h; bc.rh = r.w; bc.rot = 1;
                    } else {
                        // impossible, but handle gracefully
                        cerr << "Unexpected placement failure.\n";
                        cout << 0 << "\n";
                        for (int i=0;i<N;++i) cout << "0 0 0 0\n";
                        return 0;
                    }
                }
                nb.placeRectAt(bc);
                int binIdx = (int)bins.size() - 1;
                placements[r.idx].board = binIdx;
                placements[r.idx].x = bc.x;
                placements[r.idx].y = bc.y;
                placements[r.idx].rot = bc.rot;
                placed = true;
            }
            if (!placed) {
                // fallback: place on new board at 0,0
                // (should not arrive)
                int binIdx = (int)bins.size();
                bins.emplace_back(boardW, boardH);
                placements[r.idx].board = binIdx;
                placements[r.idx].x = 0;
                placements[r.idx].y = 0;
                placements[r.idx].rot = 0;
            }
        }
        // done
        // number of boards is bins.size()
        cout << (int)bins.size() << "\n";
        for (int i=0;i<N;++i) {
            cout << placements[i].board << " " << placements[i].x << " " << placements[i].y << " " << placements[i].rot << "\n";
        }
    } else {
        // Large-scale: Shelf-based multi-board FFDH-like
        // Sort requests by max side (or area) descending for better pack
        vector<RectReq> sorted = reqs;
        sort(sorted.begin(), sorted.end(), [](const RectReq &a, const RectReq &b){
            long long va = 1LL * max(a.w, a.h) * max(a.w, a.h) + 1LL * a.w * a.h;
            long long vb = 1LL * max(b.w, b.h) * max(b.w, b.h) + 1LL * b.w * b.h;
            return va > vb;
        });

        vector<ShelfBoard> boards;
        boards.reserve(128);

        for (const auto &r : sorted) {
            bool placed = false;
            // First-fit across boards
            for (size_t b=0;b<boards.size();++b) {
                int ox, oy, orot;
                if (boards[b].tryPlace(r.w, r.h, ox, oy, orot)) {
                    placements[r.idx].board = (int)b;
                    placements[r.idx].x = ox;
                    placements[r.idx].y = oy;
                    placements[r.idx].rot = orot;
                    placed = true;
                    break;
                }
            }
            if (!placed) {
                // new board
                boards.emplace_back(boardW, boardH);
                int ox, oy, orot;
                bool ok = boards.back().tryPlace(r.w, r.h, ox, oy, orot);
                if (!ok) {
                    // should not happen due to earlier feasibility check, but handle
                    cerr << "Failed to place rectangle in fresh board (unexpected).\n";
                    cout << 0 << "\n";
                    for (int i=0;i<N;++i) cout << "0 0 0 0\n";
                    return 0;
                }
                int bi = (int)boards.size() - 1;
                placements[r.idx].board = bi;
                placements[r.idx].x = ox;
                placements[r.idx].y = oy;
                placements[r.idx].rot = orot;
            }
        }

        cout << (int)boards.size() << "\n";
        for (int i=0;i<N;++i) {
            cout << placements[i].board << " " << placements[i].x << " " << placements[i].y << " " << placements[i].rot << "\n";
        }
    }

    return 0;
}
```

Notes and rationale:
- The solver adapts automatically: MaxRects for quality on small/medium inputs, and a very fast shelf-first-fit for large inputs to respect the 30s runtime constraint.
- Outputs are board indices starting at 0 and integer x,y coordinates (top-left corner) with rotated = 1 when rectangle was rotated by 90° (width/height swapped).
- The MaxRects implementation uses Best Short Side Fit and prunes contained free rectangles for reasonable packing quality.
- The shelf algorithm sorts rectangles by a heuristic (dominant side + area) and uses First-Fit across boards; it's fast and memory-light for huge inputs.
- If you want different behavior (e.g., board indices starting at 1), tell me and I can change the output format.
