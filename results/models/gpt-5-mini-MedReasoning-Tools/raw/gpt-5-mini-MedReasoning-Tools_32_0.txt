Below is a complete, self-contained C++20 program that implements two approaches and chooses between them:

- Exact Dreyfus–Wagner dynamic programming (used when number of terminals is small, default threshold = 15).
- Fast heuristic (MST on metric closure + reconstruct shortest paths) for larger terminal counts.

The program reads the input format you specified, and prints a Steiner tree as:
total_weight edge_count
u v
... (0-indexed vertices)

The code compiles with MSVC (cl) and uses only standard headers.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <cstdint>
#include <unordered_map>
#include <unordered_set>
#include <tuple>

using namespace std;
using ll = long long;
const ll INF = (numeric_limits<ll>::max() / 4);

struct Edge { int to; int w; };

struct Prev {
    // type: -1 unset, 0 base (terminal shortest-path), 1 combine (p,q at same vertex), 2 relax (via neighbor)
    int type = -1;
    int p = -1;
    int q = -1;
    int prev_v = -1;
    int term = -1;
};

static inline long long edgeKey(int a, int b) {
    if (a > b) std::swap(a,b);
    return ( (long long) (uint32_t) a << 32 ) | (uint32_t) b;
}

void dijkstra(int src, const vector<vector<Edge>>& adj, vector<ll>& dist, vector<int>& parent) {
    int n = (int)adj.size();
    dist.assign(n, INF);
    parent.assign(n, -1);
    using pli = pair<ll,int>;
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    dist[src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d,u] = pq.top(); pq.pop();
        if (d != dist[u]) continue;
        for (auto &e : adj[u]) {
            int v = e.to; ll nd = d + e.w;
            if (nd < dist[v]) {
                dist[v] = nd;
                parent[v] = u;
                pq.push({nd, v});
            }
        }
    }
}

struct DSU {
    int n;
    vector<int> p;
    DSU(int n=0): n(n), p(n, -1) {}
    int find(int x){ return p[x]<0?x:p[x]=find(p[x]); }
    bool unite(int a,int b){
        a=find(a); b=find(b);
        if(a==b) return false;
        if(p[a]>p[b]) swap(a,b);
        p[a]+=p[b]; p[b]=a;
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;
    vector<vector<Edge>> adj(n);
    unordered_map<long long, ll> weightMap;
    for (int i = 0; i < m; ++i) {
        int u,v; ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, (int)w});
        adj[v].push_back({u, (int)w});
        long long key = edgeKey(u,v);
        auto it = weightMap.find(key);
        if (it == weightMap.end() || w < it->second) weightMap[key] = w;
    }
    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << "0 0\n";
        return 0;
    }
    if (k == 1) {
        // single terminal: no edges required
        cout << "0 0\n";
        return 0;
    }

    // Precompute single-source shortest paths from each terminal
    vector<vector<ll>> dist_term;
    vector<vector<int>> parent_term;
    dist_term.assign(k, vector<ll>());
    parent_term.assign(k, vector<int>());
    for (int i = 0; i < k; ++i) {
        dijkstra(terminals[i], adj, dist_term[i], parent_term[i]);
    }

    // If some terminals are unreachable pairwise, it's impossible -> print 0 0
    // Check connectivity among terminals via BFS/DSU using original edges
    {
        DSU dsu(n);
        for (auto &kv : weightMap) {
            long long key = kv.first;
            int u = (int)(key >> 32);
            int v = (int)(key & 0xffffffff);
            dsu.unite(u,v);
        }
        int root = dsu.find(terminals[0]);
        for (int t = 1; t < k; ++t) {
            if (dsu.find(terminals[t]) != root) {
                cout << "0 0\n";
                return 0;
            }
        }
    }

    // Choose algorithm: exact DP if small number of terminals
    const int DP_THRESHOLD = 15; // adjust for memory/time tradeoff
    if (k <= DP_THRESHOLD) {
        int T = k;
        int FULL = (1<<T);
        // dp[mask][v]
        vector<vector<ll>> dp(FULL, vector<ll>(n, INF));
        vector<vector<Prev>> prev(FULL, vector<Prev>(n));

        // Base: single terminals
        for (int i = 0; i < T; ++i) {
            int mask = 1<<i;
            for (int v = 0; v < n; ++v) {
                dp[mask][v] = dist_term[i][v];
                prev[mask][v].type = 0;
                prev[mask][v].term = i;
            }
        }

        // For all masks
        for (int mask = 1; mask < FULL; ++mask) {
            if ((mask & (mask - 1)) == 0) {
                // single bit, already initialized. Still run Dijkstra propagation to be safe (not necessary)
                // but skipping is fine.
            } else {
                // Combine smaller subsets
                // iterate proper non-empty submasks p of mask
                for (int p = (mask-1) & mask; p; p = (p-1) & mask) {
                    int q = mask ^ p;
                    if (q == 0) continue;
                    // To avoid duplicate symmetric partitions, allow all; cost update check will handle
                    for (int v = 0; v < n; ++v) {
                        ll cand = dp[p][v] + dp[q][v];
                        if (cand < dp[mask][v]) {
                            dp[mask][v] = cand;
                            Prev pr;
                            pr.type = 1;
                            pr.p = p;
                            pr.q = q;
                            prev[mask][v] = pr;
                        }
                    }
                }
            }
            // Run Dijkstra with initial distances dp[mask][*] to propagate along edges
            using pli = pair<ll,int>;
            priority_queue<pli, vector<pli>, greater<pli>> pq;
            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
            }
            while (!pq.empty()) {
                auto [d,u] = pq.top(); pq.pop();
                if (d != dp[mask][u]) continue;
                for (auto &e : adj[u]) {
                    int v = e.to; ll nd = d + e.w;
                    if (nd < dp[mask][v]) {
                        dp[mask][v] = nd;
                        Prev pr;
                        pr.type = 2;
                        pr.prev_v = u;
                        prev[mask][v] = pr;
                        pq.push({nd, v});
                    }
                }
            }
        }

        int full = FULL - 1;
        ll bestCost = INF;
        int bestV = -1;
        for (int v = 0; v < n; ++v) {
            if (dp[full][v] < bestCost) { bestCost = dp[full][v]; bestV = v; }
        }
        if (bestV == -1 || bestCost >= INF/2) {
            cout << "0 0\n";
            return 0;
        }

        // Reconstruct edges by tracing prev
        vector<vector<char>> visited(FULL, vector<char>(n, 0));
        unordered_set<long long> edgesSet;
        function<void(int,int)> collect = [&](int v, int mask) {
            if (visited[mask][v]) return;
            visited[mask][v] = 1;
            Prev pr = prev[mask][v];
            if (pr.type == -1) return;
            if (pr.type == 0) {
                int termIdx = pr.term;
                int cur = v;
                int src = terminals[termIdx];
                // Walk parent_term to src
                while (cur != src && cur != -1) {
                    int p = parent_term[termIdx][cur];
                    if (p == -1) break;
                    edgesSet.insert(edgeKey(cur, p));
                    cur = p;
                }
                return;
            } else if (pr.type == 1) {
                collect(v, pr.p);
                collect(v, pr.q);
                return;
            } else if (pr.type == 2) {
                int u = pr.prev_v;
                edgesSet.insert(edgeKey(u, v));
                collect(u, mask);
                return;
            }
        };

        collect(bestV, full);

        // Output edges
        vector<pair<int,int>> outEdges;
        outEdges.reserve(edgesSet.size());
        ll sumW = 0;
        for (auto key : edgesSet) {
            int u = (int)(key >> 32);
            int v = (int)(key & 0xffffffff);
            outEdges.push_back({u,v});
            auto it = weightMap.find(key);
            if (it != weightMap.end()) sumW += it->second;
            else {
                // theoretically shouldn't happen; but handle gracefully by computing weight from adj
                bool found = false;
                for (auto &e : adj[u]) if (e.to == v) { sumW += e.w; found = true; break; }
                if (!found) for (auto &e : adj[v]) if (e.to == u) { sumW += e.w; found = true; break; }
            }
        }
        cout << sumW << " " << outEdges.size() << "\n";
        for (auto &e : outEdges) cout << e.first << " " << e.second << "\n";
        return 0;
    } else {
        // Heuristic method for larger terminal sets:
        // 1) Build metric closure among terminals (weights = shortest path distances)
        // 2) Compute MST on these terminals (Kruskal)
        // 3) Reconstruct shortest paths for each selected terminal-edge and add their original edges to union set
        // 4) Run Kruskal/DSU on union edges and stop when all terminals connected

        // Build metric edges between terminals
        struct TEdge { int a,b; ll w; };
        vector<TEdge> metricEdges;
        for (int i = 0; i < k; ++i) {
            for (int j = i+1; j < k; ++j) {
                ll d = dist_term[i][terminals[j]];
                // Alternatively dist_term[i][terminals[j]] equals dist_term[j][terminals[i]]
                if (d < INF/2) metricEdges.push_back({i,j,d});
            }
        }
        // Kruskal on metric graph (terminals as nodes)
        sort(metricEdges.begin(), metricEdges.end(), [](const TEdge& a, const TEdge& b){ return a.w < b.w; });
        DSU dsuTerm(k);
        vector<pair<int,int>> chosenPairs; chosenPairs.reserve(k-1);
        for (auto &e : metricEdges) {
            if (dsuTerm.unite(e.a, e.b)) {
                chosenPairs.push_back({e.a, e.b});
                if ((int)chosenPairs.size() == k-1) break;
            }
        }
        // Reconstruct paths in original graph and collect union edges
        unordered_set<long long> unionEdgesSet;
        for (auto &pr : chosenPairs) {
            int i = pr.first;
            int j = pr.second;
            // reconstruct path between terminals[i] and terminals[j] using parent_term[i]
            int target = terminals[j];
            int cur = target;
            // if parent_term[i][target] == -1 then maybe we computed distances using different source because metric uses dist from i to j; but dist_term[i] stores distances from terminals[i]
            // We can use parent_term[i] to walk back to terminals[i]
            while (cur != terminals[i] && cur != -1) {
                int p = parent_term[i][cur];
                if (p == -1) break;
                unionEdgesSet.insert(edgeKey(cur, p));
                cur = p;
            }
            // If we didn't reach terminals[i] (maybe due to using parent arrays from different source), try the other way using parent_term[j]
            if (cur != terminals[i]) {
                cur = terminals[i];
                int target2 = terminals[j];
                vector<int> rev;
                int node = target2;
                while (node != -1 && node != cur) {
                    rev.push_back(node);
                    node = parent_term[j][node];
                }
                if (node == cur) {
                    // reconstruct path cur .. target2
                    for (size_t idx = 0; idx+1 < rev.size(); ++idx) {
                        unionEdgesSet.insert(edgeKey(rev[idx], rev[idx+1]));
                    }
                }
            }
        }

        // Build vector of union edges with weights to run Kruskal
        struct GEdge { int u,v; ll w; };
        vector<GEdge> unionEdges;
        unionEdges.reserve(unionEdgesSet.size());
        for (auto key : unionEdgesSet) {
            int u = (int)(key >> 32);
            int v = (int)(key & 0xffffffff);
            ll w = weightMap.count(key) ? weightMap[key] : INF;
            unionEdges.push_back({u,v,w});
        }
        // Kruskal but stop when terminals are connected
        sort(unionEdges.begin(), unionEdges.end(), [](const GEdge& a, const GEdge& b){ return a.w < b.w; });
        DSU dsu(n);
        int comps = k;
        unordered_map<int,int> compIdx; compIdx.reserve(k*2);
        for (int i = 0; i < k; ++i) compIdx[terminals[i]] = i;
        vector<pair<int,int>> finalEdges;
        ll totalW = 0;
        for (auto &e : unionEdges) {
            if (dsu.find(e.u) != dsu.find(e.v)) {
                // Check whether adding this edge reduces terminal component count
                int beforeRoots = 0;
                // We'll unite and then check terminal connectivity count more efficiently:
                // First unite if it connects different DSU components
                bool reduces = false;
                int ru = dsu.find(e.u), rv = dsu.find(e.v);
                // Count how many distinct terminal-representative components will merge
                // To make this efficient, we'll just unite and recalc terminal component sets lazily
                dsu.unite(ru, rv);
                finalEdges.push_back({e.u, e.v});
                totalW += e.w;
            }
        }
        // After Kruskal we must ensure terminals are connected; if not, connect using additional shortest-path edges
        // Simple fallback: run Dijkstra's from one terminal across the subgraph consisting of finalEdges to make sure it's connected, else add shortest-path edges by reconstructing from parent_term
        // To keep it simple and robust: create union of finalEdges and, if it's not connecting all terminals, add shortest paths between terminals incrementally
        // Build DSU on result
        DSU dsuRes(n);
        for (auto &e : finalEdges) dsuRes.unite(e.first, e.second);
        // Connect remaining terminals pairwise using direct shortest paths (greedy)
        for (int i = 1; i < k; ++i) {
            if (dsuRes.find(terminals[0]) != dsuRes.find(terminals[i])) {
                // add path from terminals[i] to terminals[0] using parent_term[i]
                int cur = terminals[0];
                int target = terminals[i];
                int node = target;
                // attempt using parent_term[i] to walk to terminals[i] is not helpful; instead walk back from target to source via parent_term[i]
                // We'll walk from target to source via parent_term[i]
                while (node != terminals[i] && node != -1) break; // noop
                // Use parent_term[i] to go from target back to source terminals[i]->...->target; but we need path between terminals[0] and terminals[i].
                // Instead reconstruct using parent_term[i] to walk from terminals[i] to terminals[0]:
                node = terminals[i];
                vector<int> path;
                while (node != -1 && node != terminals[0]) {
                    path.push_back(node);
                    node = parent_term[i][node];
                }
                if (node == terminals[0]) path.push_back(node);
                if (path.empty() || path.back() != terminals[0]) {
                    // fallback: try parent_term[0] walking from terminals[0] to terminals[i]
                    node = terminals[i];
                    vector<int> path2;
                    while (node != -1 && node != terminals[0]) {
                        path2.push_back(node);
                        node = parent_term[0][node];
                    }
                    if (node == terminals[0]) {
                        path2.push_back(node);
                        // add edges
                        for (size_t idx = 0; idx+1 < path2.size(); ++idx) {
                            int a = path2[idx], b = path2[idx+1];
                            long long key = edgeKey(a,b);
                            finalEdges.push_back({a,b});
                            if (weightMap.count(key)) totalW += weightMap[key];
                            dsuRes.unite(a,b);
                        }
                    }
                } else {
                    // add edges along path vector
                    for (size_t idx = 0; idx+1 < path.size(); ++idx) {
                        int a = path[idx], b = path[idx+1];
                        long long key = edgeKey(a,b);
                        finalEdges.push_back({a,b});
                        if (weightMap.count(key)) totalW += weightMap[key];
                        dsuRes.unite(a,b);
                    }
                }
            }
        }

        // Deduplicate finalEdges
        unordered_set<long long> finalSet;
        vector<pair<int,int>> outEdges;
        outEdges.reserve(finalEdges.size());
        ll sumW = 0;
        for (auto &e : finalEdges) {
            long long key = edgeKey(e.first, e.second);
            if (finalSet.insert(key).second) {
                outEdges.push_back({e.first, e.second});
                auto it = weightMap.find(key);
                if (it != weightMap.end()) sumW += it->second;
            }
        }

        cout << sumW << " " << outEdges.size() << "\n";
        for (auto &p : outEdges) cout << p.first << " " << p.second << "\n";
        return 0;
    }
}
```

Notes and remarks
- For exact solutions the code uses Dreyfus–Wagner DP with full path reconstruction. Threshold DP_THRESHOLD = 15 is conservative; you can raise it if you know terminal count is slightly larger and memory/time permit.
- For larger terminal sets the heuristic builds an MST over the metric closure and maps metric edges back to shortest paths; it then extracts a tree from the union of those paths.
- If the input places terminals in different disconnected components the program prints "0 0" (no feasible Steiner tree).
- The program uses 0-based vertex numbering as requested.
