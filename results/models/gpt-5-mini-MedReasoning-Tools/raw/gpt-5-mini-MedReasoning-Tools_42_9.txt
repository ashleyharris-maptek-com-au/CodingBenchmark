%uint = OpTypeInt 32 0

%ptr_Output_v4 = OpTypePointer Output %v4float

%ptr_Output_v3 = OpTypePointer Output %v3float

%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%float_thickness = OpConstant %float 0.025

; gl_PerVertex types for input and output
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Arrays of per-vertex input (size 4 because InputLinesAdjacency)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

; Array type for per-vertex user attributes (vec3) with 4 elements
%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_4
%ptr_Output_arr_v3 = OpTypePointer Output %arr_v3_4

; Interface variables (Inputs & Outputs)
%gl_in = OpVariable %ptr_in_arr_pv Input

; user-defined per-vertex inputs (arrays of 4)
%in_worldPos = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %in_worldPos Location 0
%in_normal = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %in_normal Location 1
%in_color = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %in_color Location 2

; Output per-vertex gl_PerVertex
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%outColor = OpVariable %ptr_Output_v3 Output
OpDecorate %outColor Location 0

; Entry point and execution modes
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; --- Load gl_Position v4s for indices 0..3 ---
; pointer type to struct member (access chain yields pointer to v4)
%ptr_in_elem = OpTypePointer Input %gl_PerVertex_in
%ptr_in_pos = OpTypePointer Input %v4float

%access_0 = OpAccessChain %ptr_in_pos %gl_in %int_0 %int_0
%pos0_v4 = OpLoad %v4float %access_0

%access_1 = OpAccessChain %ptr_in_pos %gl_in %int_1 %int_0
%pos1_v4 = OpLoad %v4float %access_1

%access_2 = OpAccessChain %ptr_in_pos %gl_in %int_2 %int_0
%pos2_v4 = OpLoad %v4float %access_2

%access_3 = OpAccessChain %ptr_in_pos %gl_in %int_3 %int_0
%pos3_v4 = OpLoad %v4float %access_3

; Extract XY as v2float using VectorShuffle
%pos0_xy = OpVectorShuffle %v2float %pos0_v4 %pos0_v4 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1_v4 %pos1_v4 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2_v4 %pos2_v4 0 1
%pos3_xy = OpVectorShuffle %v2float %pos3_v4 %pos3_v4 0 1

; --- Compute tangents ---
%diff_t0 = OpFSub %v2float %pos2_xy %pos0_xy
%t0 = OpExtInst %v2float %GLSL Normalize %diff_t0

%diff_t1 = OpFSub %v2float %pos3_xy %pos1_xy
%t1 = OpExtInst %v2float %GLSL Normalize %diff_t1

; --- Compute perpendiculars perp = (-t.y, t.x) ---
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; --- Scale perpendiculars by thickness ---
%off0 = OpVectorTimesScalar %v2float %perp0 %float_thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %float_thickness

; --- Convert offsets to vec4 (xy, 0, 0) for adding to gl_Position ---
%off0_x = OpCompositeExtract %float %off0 0
%off0_y = OpCompositeExtract %float %off0 1
%off0_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0

%off1_x = OpCompositeExtract %float %off1 0
%off1_y = OpCompositeExtract %float %off1 1
%off1_v4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0

; --- Prepare colors ---
%ptr_color1 = OpAccessChain %ptr_Input_v3 %in_color %int_1
%color1 = OpLoad %v3float %ptr_color1

%ptr_color2 = OpAccessChain %ptr_Input_v3 %in_color %int_2
%color2 = OpLoad %v3float %ptr_color2

; --- Emit vertices in order:
; start + perp0, start - perp0, end + perp1, end - perp1
; Vertex 0: start + off0
%start_plus = OpFAdd %v4float %pos1_v4 %off0_v4
%ptr_out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %ptr_out_pos %start_plus
OpStore %outColor %color1

; Vertex 1: start - off0
%start_minus = OpFSub %v4float %pos1_v4 %off0_v4
OpStore %ptr_out_pos %start_minus
OpStore %outColor %color1
OpEmitVertex

; Vertex 2: end + off1
%end_plus = OpFAdd %v4float %pos2_v4 %off1_v4
OpStore %ptr_out_pos %end_plus
OpStore %outColor %color2
OpEmitVertex

; Vertex 3: end - off1
%end_minus = OpFSub %v4float %pos2_v4 %off1_v4
OpStore %ptr_out_pos %end_minus
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive
