struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Checkerboard at 8x8 frequency
    float2 uv8 = input.uv * 8.0;
    int ix = (int)floor(uv8.x);
    int iy = (int)floor(uv8.y);
    int parity = (ix + iy) & 1;
    float baseVal = (parity == 0) ? 0.9 : 0.2;
    float3 baseColor = float3(baseVal, baseVal, baseVal);

    // Lambertian diffuse
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadeFactor = 0.5 + 0.5 * diffuse;

    float3 outColor = baseColor * shadeFactor;
    return float4(outColor, 1.0);
}
