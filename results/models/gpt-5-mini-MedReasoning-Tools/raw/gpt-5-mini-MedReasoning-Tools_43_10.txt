struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // Final shading factor as specified
    float shade = 0.4 + 0.6 * diffuse;

    // Stripes driven by sin(uv.y * pi * 16)
    float s = sin(input.uv.y * 3.14159 * 16.0);
    float stripe = s >= 0.0 ? 1.0 : 0.0;

    float3 red  = float3(0.9, 0.1, 0.1);
    float3 blue = float3(0.1, 0.1, 0.9);

    // Choose stripe color and apply Lambertian shading
    float3 baseColor = lerp(blue, red, stripe);
    float3 finalColor = baseColor * shade;

    return float4(finalColor, 1.0);
}
