struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Compute light direction (handle positional vs directional by w)
    float3 L;
    if (abs(lightPos.w) > 0.5) {
        // positional light
        L = normalize(lightPos.xyz - input.worldPos);
    } else {
        // directional light (assume direction is given in xyz)
        L = normalize(lightPos.xyz);
    }

    // Gooch parameters
    float3 cool = float3(0.2, 0.2, 0.75);
    float3 warm = float3(0.7, 0.7, 0.4);

    // t as specified
    float ndotl = dot(N, L);
    float t = ndotl * 0.5 + 0.5;

    // Interpolate between cool and warm
    float3 gooch = lerp(cool, warm, t);

    // Basic Lambertian diffuse
    float diffuse = max(ndotl, 0.0);

    // Small ambient so unlit/backfacing areas retain cool tone
    float3 ambient = cool * 0.1;

    // Combine
    float3 finalColor = gooch * diffuse + ambient;

    // Modulate by per-vertex color (optional variation)
    finalColor *= input.color;

    return float4(finalColor, 1.0);
}
