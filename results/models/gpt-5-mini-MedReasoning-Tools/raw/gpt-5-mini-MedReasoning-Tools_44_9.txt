#version 450

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Light direction (positional light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // Shading factor as requested
    float shadeFactor = 0.5 + 0.5 * diffuse;

    // 8x8 UV checkerboard: parity of integer cell coordinates
    vec2 uv8 = uv * 8.0;
    vec2 cell = floor(uv8);
    float parity = mod(cell.x + cell.y, 2.0); // 0.0 or 1.0

    // Alternate colors: 0.9 when parity==0, 0.2 when parity==1
    vec3 baseColor = mix(vec3(0.9), vec3(0.2), parity);

    // Apply Lambertian shading and output alpha = 1.0
    vec3 finalColor = baseColor * shadeFactor;

    outColor = vec4(finalColor, 1.0);
}
