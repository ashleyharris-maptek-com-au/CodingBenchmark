GLSL compute shader (save as prefix_scan.comp):
```
#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(set = 0, binding = 0) readonly buffer InBuffer {
    uint data[];
} inBuf;

layout(set = 0, binding = 1) buffer OutBuffer {
    uint data[];
} outBuf;

layout(set = 0, binding = 2) uniform Params {
    uint N;
    uint _pad1;
    uint _pad2;
    uint _pad3;
} params;

shared uint sdata[256u];

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint lid = gl_LocalInvocationID.x;
    uint group = gl_WorkGroupID.x;

    // Bounds check
    if (gid >= params.N) {
        // nothing to write
        return;
    }

    // If total N <= 256, perform a single-workgroup shared-memory inclusive scan.
    // Only the first workgroup participates (workgroup 0).
    if (params.N <= 256u) {
        if (group != 0u) {
            // other workgroups do nothing
            return;
        }

        // load element or 0
        uint v = 0u;
        if (lid < params.N) {
            v = inBuf.data[lid];
        }
        sdata[lid] = v;
        // ensure all writes visible
        memoryBarrierShared();
        barrier();

        // Hillis-Steele inclusive scan: for offsets 1,2,4,...,128
        for (uint offset = 1u; offset < 256u; offset <<= 1u) {
            uint addv = 0u;
            if (lid >= offset) {
                addv = sdata[lid - offset];
            }
            // ensure addv read before write
            memoryBarrierShared();
            barrier();
            sdata[lid] = sdata[lid] + addv;
            memoryBarrierShared();
            barrier();
        }

        // write output for indices < N
        if (lid < params.N) {
            outBuf.data[lid] = sdata[lid];
        }
        return;
    }

    // Fallback naive per-thread accumulation for general N > 256
    // each thread sums all previous elements up to its gid (inclusive)
    uint sum = 0u;
    // loop from 0 to gid inclusive
    // Note: This is O(N) per thread and is slow for large N but satisfies a simple fallback.
    for (uint i = 0u; i <= gid; ++i) {
        sum += inBuf.data[i];
    }
    outBuf.data[gid] = sum;
}
```
