Here is a complete, compilable C# program that implements a fast heuristic for the Quadratic Assignment Problem (QAP). It builds a sensible greedy initial solution, then performs local search using swap deltas computed in O(N) per candidate swap. It adapts its improvement strategy based on N (exhaustive for small N, randomized/targeted search for larger N) and uses a time budget so it will finish within reasonable limits for large N.

Copy, compile and run with the input format you specified.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

class Program
{
    // Fast integer scanner for large inputs
    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;
        public FastScanner(Stream s) { stream = s; }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            while ((c = Read()) <= 32)
            {
                if (c == -1) return int.MinValue;
            }
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    static int N;
    static int[][] flow;
    static int[][] dist;
    static long[] flowWeight;
    static long[] distWeight;
    static Random rng = new Random();

    static void Main(string[] args)
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        N = scanner.NextInt();
        if (N == int.MinValue) return;

        flow = new int[N][];
        dist = new int[N][];
        for (int i = 0; i < N; i++)
        {
            flow[i] = new int[N];
            for (int j = 0; j < N; j++) flow[i][j] = scanner.NextInt();
        }
        for (int i = 0; i < N; i++)
        {
            dist[i] = new int[N];
            for (int j = 0; j < N; j++) dist[i][j] = scanner.NextInt();
        }

        // Compute weights to build initial greedy perm
        flowWeight = new long[N];
        distWeight = new long[N];
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            for (int j = 0; j < N; j++) s += flow[i][j] + flow[j][i];
            flowWeight[i] = s;
        }
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            for (int j = 0; j < N; j++) s += dist[i][j] + dist[j][i];
            distWeight[i] = s;
        }

        int[] facilities = Enumerable.Range(0, N).ToArray();
        int[] locations = Enumerable.Range(0, N).ToArray();

        Array.Sort(facilities, (a, b) => flowWeight[b].CompareTo(flowWeight[a])); // descending
        Array.Sort(locations, (a, b) => distWeight[a].CompareTo(distWeight[b])); // ascending

        int[] p = new int[N]; // p[facility] = location
        for (int i = 0; i < N; i++) p[facilities[i]] = locations[i];

        long currentCost = ComputeCost(p);

        // Decide time budget based on N (ms)
        int timeBudgetMs;
        if (N <= 100) timeBudgetMs = 10000;
        else if (N <= 200) timeBudgetMs = 30000;
        else if (N <= 500) timeBudgetMs = 60000;
        else if (N <= 1000) timeBudgetMs = 120000;
        else timeBudgetMs = 240000; // up to 4 minutes for very large instances

        // For very small N allow exhaustive pairwise best-improvement
        if (N <= 200)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                long bestDelta = 0;
                int bestI = -1, bestJ = -1;
                for (int i = 0; i < N - 1; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        long d = SwapDelta(p, i, j);
                        if (d < bestDelta)
                        {
                            bestDelta = d;
                            bestI = i;
                            bestJ = j;
                        }
                    }
                }
                if (bestDelta < 0)
                {
                    ApplySwap(p, bestI, bestJ);
                    currentCost += bestDelta;
                    improved = true;
                }
            }
        }
        else
        {
            // Randomized & targeted local search within time budget
            Stopwatch sw = Stopwatch.StartNew();
            int topK = Math.Min(N, 100);
            int iterPerCycle = Math.Max(1000, N * 2);
            // Precompute an ordering by facility importance for targeted sampling
            int[] facByImportance = Enumerable.Range(0, N).ToArray();
            Array.Sort(facByImportance, (a, b) => flowWeight[b].CompareTo(flowWeight[a])); // descending

            while (sw.ElapsedMilliseconds < timeBudgetMs)
            {
                bool improvedInCycle = false;

                // 1) Targeted attempts: try swapping from top important facilities vs random other facilities
                int targetedTries = Math.Min(5000, iterPerCycle);
                for (int t = 0; t < targetedTries && sw.ElapsedMilliseconds < timeBudgetMs; t++)
                {
                    int i = facByImportance[rng.Next(Math.Min(topK, N))];
                    int j = rng.Next(N);
                    if (j == i) continue;
                    long delta = SwapDelta(p, i, j);
                    if (delta < 0)
                    {
                        ApplySwap(p, i, j);
                        currentCost += delta;
                        improvedInCycle = true;
                        break; // restart cycle to benefit from change
                    }
                }
                if (improvedInCycle) continue;

                // 2) Random sampling
                int randTries = Math.Min(iterPerCycle, 5000);
                for (int t = 0; t < randTries && sw.ElapsedMilliseconds < timeBudgetMs; t++)
                {
                    int i = rng.Next(N);
                    int j = rng.Next(N);
                    if (i == j) continue;
                    long delta = SwapDelta(p, i, j);
                    if (delta < 0)
                    {
                        ApplySwap(p, i, j);
                        currentCost += delta;
                        improvedInCycle = true;
                        break;
                    }
                }
                if (improvedInCycle) continue;

                // 3) Local refinement: try swapping among topK facilities (smaller search)
                if (topK >= 2)
                {
                    int tries = Math.Min(2000, topK * (topK - 1) / 2);
                    for (int t = 0; t < tries && sw.ElapsedMilliseconds < timeBudgetMs; t++)
                    {
                        int a = rng.Next(topK);
                        int b = rng.Next(topK);
                        if (a == b) continue;
                        int i = facByImportance[a];
                        int j = facByImportance[b];
                        long delta = SwapDelta(p, i, j);
                        if (delta < 0)
                        {
                            ApplySwap(p, i, j);
                            currentCost += delta;
                            improvedInCycle = true;
                            break;
                        }
                    }
                }

                // If nothing improved, perform a small perturbation to escape local minimum
                if (!improvedInCycle)
                {
                    // perform a few random swaps (accepting even non-improving with small probability)
                    for (int t = 0; t < 5 && sw.ElapsedMilliseconds < timeBudgetMs; t++)
                    {
                        int i = rng.Next(N);
                        int j = rng.Next(N);
                        if (i == j) continue;
                        long delta = SwapDelta(p, i, j);
                        if (delta < 0 || rng.NextDouble() < 0.01)
                        {
                            ApplySwap(p, i, j);
                            currentCost += delta;
                        }
                    }
                }
            }
        }

        // Output result
        var outBuilder = new StringBuilder();
        outBuilder.AppendLine(currentCost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < N; i++)
        {
            if (i > 0) outBuilder.Append(' ');
            outBuilder.Append(p[i]);
        }
        Console.WriteLine(outBuilder.ToString());
    }

    static long ComputeCost(int[] p)
    {
        long total = 0;
        int n = N;
        for (int i = 0; i < n; i++)
        {
            int pi = p[i];
            for (int j = 0; j < n; j++)
            {
                total += (long)flow[i][j] * dist[pi][p[j]];
            }
        }
        return total;
    }

    // Compute delta cost when swapping assignment of facilities i and j (p[i] <-> p[j])
    // Using formula: delta = sum_v (flow[i][v] - flow[j][v]) * (dist[pj][p[v]] - dist[pi][p[v]])
    //                         + (flow[v][i] - flow[v][j]) * (dist[p[v]][pj] - dist[p[v]][pi])
    static long SwapDelta(int[] p, int i, int j)
    {
        if (i == j) return 0;
        int pi = p[i];
        int pj = p[j];
        long delta = 0;
        for (int v = 0; v < N; v++)
        {
            int pv = p[v];
            int dp = dist[pj][pv] - dist[pi][pv];
            int dq = dist[pv][pj] - dist[pv][pi];
            delta += (long)(flow[i][v] - flow[j][v]) * dp + (long)(flow[v][i] - flow[v][j]) * dq;
        }
        return delta;
    }

    static void ApplySwap(int[] p, int i, int j)
    {
        int tmp = p[i];
        p[i] = p[j];
        p[j] = tmp;
    }
}
```

Notes and rationale
- The program computes swap deltas in O(N) and therefore a single delta test is O(N). For small N it does exhaustive best-improvement (good quality). For larger N it performs randomized and targeted local search within a configurable time budget so it will scale to large instances (N up to ~2000) while remaining reasonable in runtime.
- All arithmetic for costs uses 64-bit long to reduce overflow risk.
- The greedy initial solution sorts facilities by flow importance (row+col sums) and locations by distance centrality (sum of distances) and matches most important facilities to most central locations — a common simple heuristic that usually gives a decent starting point for local search.
- The strategy (time budget, sampling sizes) is conservative and tuned to provide improvement in reasonable time. You can adjust timeBudgetMs or sampling parameters if you want the program to run faster or spend more time improving solutions on large instances.

If you want stronger results for medium N, I can add more advanced heuristics (e.g., simulated annealing, tabu search, or more aggressive variable neighborhood search).
