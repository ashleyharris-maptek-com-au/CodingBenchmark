Below is a complete, compilable C# program that implements an adaptive Minesweeper solver. It combines fast local deterministic rules (zero-neighbor rule, full-neighbor rule), subset reasoning, and exact constraint enumeration on small frontier components. The solver never guesses — it only outputs cells that are provably safe. For huge or very complex components the solver avoids exponential enumeration and therefore scales to very large boards.

Save and compile with the provided environment (csc/.NET). The program reads stdin and writes stdout in the format you specified.

C# code:

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

class MinesweeperSolver
{
    static int H, W;
    static char[,] board;
    // Track deductions:
    static bool[,] deducedMine;
    static bool[,] deducedSafe;
    static bool[,] originalUnknown;

    // Directions
    static readonly int[] dx = new int[]{-1,0,1,-1,1,-1,0,1};
    static readonly int[] dy = new int[]{-1,-1,-1,0,0,1,1,1};

    // Configurable enumeration size limit (component variable count)
    const int MAX_ENUM_VARS = 24;

    static void Main()
    {
        // Read H W
        string line = Console.ReadLine();
        if (string.IsNullOrEmpty(line))
            return;
        var parts = line.Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        board = new char[H, W];
        deducedMine = new bool[H, W];
        deducedSafe = new bool[H, W];
        originalUnknown = new bool[H, W];

        for (int r = 0; r < H; r++)
        {
            string row = Console.ReadLine();
            if (row == null) row = "";
            // If row shorter than W, pad with spaces
            if (row.Length < W) row = row.PadRight(W, ' ');
            for (int c = 0; c < W; c++)
            {
                char ch = row[c];
                board[r, c] = ch;
                if (ch == ' ')
                {
                    originalUnknown[r, c] = true;
                }
                if (ch == '*')
                {
                    deducedMine[r, c] = true; // flagged as mine in input
                }
            }
        }

        // Apply an adaptive pipeline:
        // 1) Iterative local deterministic rules (0 rule, full-neighbor rule)
        // 2) Subset reasoning between constraints
        // 3) Exhaustive enumeration on small frontier components
        //
        // All deductions update deducedSafe and deducedMine. Only original unknown cells that are deducedSafe are output.

        // Step 1 & 2 repeated until no change
        bool changed = true;
        int iter = 0;
        while (changed && iter < 5) // iterate a few times to let deductions propagate
        {
            changed = false;
            if (ApplySimpleDeterministicRules()) changed = true;
            if (ApplySubsetReasoning()) changed = true;
            iter++;
        }

        // Step 3: Build frontier and enumerate small components
        var provablySafeFromEnumeration = ApplyEnumerationOnFrontier();

        // Combine deducedSafe and enumeration results
        HashSet<(int x,int y)> safeSet = new HashSet<(int x,int y)>();
        for (int r=0;r<H;r++)
            for (int c=0;c<W;c++)
                if (originalUnknown[r,c] && deducedSafe[r,c])
                    safeSet.Add((c,r));
        foreach(var p in provablySafeFromEnumeration)
            safeSet.Add(p);

        // Output
        var safeList = safeSet.ToList();
        Console.WriteLine(safeList.Count);
        foreach(var p in safeList)
        {
            Console.WriteLine($"{p.x} {p.y}");
        }
    }

    static IEnumerable<(int x,int y)> ApplyEnumerationOnFrontier()
    {
        // Build variable list: unknown cells adjacent to at least one number, and not already deduced safe/mine
        var varIndex = new Dictionary<(int r,int c),int>(); // maps (r,c) -> var idx
        var indexToPos = new List<(int r,int c)>();
        int nextVar = 0;

        // constraints: for each numbered cell produce a constraint referencing variables and target
        var constraints = new List<Constraint>();

        for (int r=0;r<H;r++)
        {
            for (int c=0;c<W;c++)
            {
                char ch = board[r,c];
                if (ch >= '0' && ch <= '8')
                {
                    int clue = ch - '0';
                    int flaggedCount = 0;
                    List<(int r,int c)> unkNeis = new List<(int r,int c)>();
                    for (int k=0;k<8;k++)
                    {
                        int nr = r + dy[k], nc = c + dx[k];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        if (board[nr, nc] == '*') flaggedCount++;
                        else if (deducedMine[nr,nc]) flaggedCount++;
                        else if (board[nr,nc] == ' ')
                        {
                            if (!deducedSafe[nr,nc] && !deducedMine[nr,nc])
                                unkNeis.Add((nr,nc));
                        }
                    }
                    int target = clue - flaggedCount;
                    if (unkNeis.Count > 0)
                    {
                        var varList = new List<int>();
                        foreach(var p in unkNeis)
                        {
                            if (!varIndex.TryGetValue(p, out int idx))
                            {
                                idx = nextVar++;
                                varIndex[p] = idx;
                                indexToPos.Add(p);
                            }
                            varList.Add(idx);
                        }
                        // note: target may be negative or > varList.Count if inconsistent board; clamp is not allowed - inconsistent board => no solutions for enumeration
                        constraints.Add(new Constraint { Vars = varList.ToArray(), Target = target, CellR = r, CellC = c });
                    }
                }
            }
        }

        if (nextVar == 0 || constraints.Count == 0) return new List<(int,int)>();

        // Build adjacency between variables: two variables adjacent if they appear in same constraint
        var adj = new List<int>[nextVar];
        for (int i=0;i<nextVar;i++) adj[i] = new List<int>();
        foreach(var con in constraints)
        {
            var v = con.Vars;
            for (int i=0;i<v.Length;i++)
                for (int j=i+1;j<v.Length;j++)
                {
                    adj[v[i]].Add(v[j]);
                    adj[v[j]].Add(v[i]);
                }
        }

        // Partition variables into components via BFS
        bool[] visited = new bool[nextVar];
        var provablySafe = new List<(int,int)>();

        for (int i=0;i<nextVar;i++)
        {
            if (visited[i]) continue;
            // BFS to collect component variables
            var queue = new Queue<int>();
            var compVars = new List<int>();
            visited[i] = true;
            queue.Enqueue(i);
            while (queue.Count>0)
            {
                int u = queue.Dequeue();
                compVars.Add(u);
                foreach(var nb in adj[u])
                {
                    if (!visited[nb]) { visited[nb]=true; queue.Enqueue(nb); }
                }
            }

            if (compVars.Count == 0) continue;

            if (compVars.Count > MAX_ENUM_VARS)
            {
                // Too big to enumerate exactly; skip enumeration for this component.
                // We already applied local rules and subset reasoning, so give up on this component.
                continue;
            }

            // Collect constraints that touch this component
            var compVarSet = new HashSet<int>(compVars);
            var compConstraints = new List<Constraint>();
            foreach(var con in constraints)
            {
                bool touches = false;
                foreach(var v in con.Vars) { if (compVarSet.Contains(v)) { touches=true; break; } }
                if (touches)
                {
                    // remap variable indices to component-local indices
                    var localVars = con.Vars.Where(v => compVarSet.Contains(v)).Distinct().ToArray();
                    // compute target as originally, but need to subtract any variables outside comp? Variables outside comp are not included and should be already known (they are not variables because they were either deduced or not in frontier).
                    // In our construction, constraints only included unknown neighbors as variables, so all unknown neighbors that are not deduced should be included in varIndex; thus localVars covers all unknowns in constraint that remained.
                    compConstraints.Add(new Constraint { Vars = localVars, Target = con.Target, CellR = con.CellR, CellC = con.CellC });
                }
            }

            // Map component global index -> local index 0..m-1
            var globalToLocal = new Dictionary<int,int>();
            for (int k=0;k<compVars.Count;k++) globalToLocal[compVars[k]] = k;

            // Build constraints in local indexing
            var localConstraints = new List<Constraint>();
            foreach(var con in compConstraints)
            {
                var localList = new List<int>();
                foreach(int gv in con.Vars) localList.Add(globalToLocal[gv]);
                localConstraints.Add(new Constraint { Vars = localList.ToArray(), Target = con.Target, CellR = con.CellR, CellC = con.CellC });
            }

            // Now enumerate all solutions for this component
            int m = compVars.Count;
            int conCount = localConstraints.Count;

            // Quick feasibility check: each target must be between 0 and Vars.Count
            bool feasible = true;
            foreach(var con in localConstraints)
            {
                if (con.Target < 0 || con.Target > con.Vars.Length) { feasible = false; break; }
            }
            if (!feasible) continue;

            var occurrences = new List<int>[m];
            for (int k=0;k<m;k++) occurrences[k] = new List<int>();
            for (int ci=0; ci<localConstraints.Count; ci++)
            {
                var vlist = localConstraints[ci].Vars;
                for (int j=0;j<vlist.Length;j++) occurrences[vlist[j]].Add(ci);
            }

            // Choose variable order: descending by occurrence count (heuristic)
            int[] order = Enumerable.Range(0,m).OrderByDescending(a => occurrences[a].Count).ToArray();
            int[] invOrder = new int[m];
            for (int p=0;p<m;p++) invOrder[order[p]] = p;

            // Remap constraints vars to the order indices to speed lookups
            var remappedConstraints = new List<Constraint>();
            foreach(var c in localConstraints)
            {
                var newVars = c.Vars.Select(v => invOrder[v]).ToArray();
                remappedConstraints.Add(new Constraint { Vars = newVars, Target = c.Target, CellR = c.CellR, CellC = c.CellC });
            }

            // State for backtracking
            int[] assigned = Enumerable.Repeat(-1, m).ToArray();
            int[] conSum = new int[remappedConstraints.Count];
            int[] conUnassigned = remappedConstraints.Select(c => c.Vars.Length).ToArray();

            long solutionCount = 0;
            long[] onesCount = new long[m]; // how many solutions had variable == 1

            // DFS backtracking
            bool aborted = false;
            void Dfs(int depth)
            {
                if (aborted) return;
                if (depth == m)
                {
                    // Check all constraints satisfied (should be by pruning)
                    solutionCount++;
                    for (int i = 0; i < m; i++) if (assigned[i] == 1) onesCount[i]++;
                    return;
                }
                int var = order[depth];

                // Try 0 then 1 (order doesn't matter)
                foreach (int val in new int[]{0,1})
                {
                    bool ok = true;
                    assigned[var] = val;
                    // update constraints involving var
                    foreach (int ci in occurrences[var])
                    {
                        conUnassigned[ci]--;
                        conSum[ci] += val;
                        // check feasibility: conSum <= target <= conSum + conUnassigned
                        if (conSum[ci] > remappedConstraints[ci].Target || conSum[ci] + conUnassigned[ci] < remappedConstraints[ci].Target)
                        {
                            ok = false;
                        }
                    }
                    if (ok)
                    {
                        Dfs(depth + 1);
                    }
                    // undo
                    foreach (int ci in occurrences[var])
                    {
                        conUnassigned[ci]++;
                        conSum[ci] -= val;
                    }
                    assigned[var] = -1;
                    if (aborted) break;
                }
            }

            Dfs(0);

            if (solutionCount == 0)
            {
                // No valid assignment found - inconsistent; skip
                continue;
            }

            // Variables that are always 0 across all solutions are provably safe
            for (int localIdx = 0; localIdx < m; localIdx++)
            {
                if (onesCount[localIdx] == 0)
                {
                    int globalVar = compVars[localIdx];
                    var pos = indexToPos[globalVar];
                    // only add if originally unknown and not deducedMine
                    if (originalUnknown[pos.r, pos.c] && !deducedMine[pos.r,pos.c])
                        provablySafe.Add((pos.c, pos.r));
                }
            }
        }

        return provablySafe;
    }

    class Constraint
    {
        public int[] Vars;
        public int Target;
        public int CellR, CellC;
    }

    static bool ApplySimpleDeterministicRules()
    {
        bool changed = false;
        var toMarkSafe = new List<(int r,int c)>();
        var toMarkMine = new List<(int r,int c)>();

        for (int r=0;r<H;r++)
        {
            for (int c=0;c<W;c++)
            {
                char ch = board[r,c];
                if (!(ch >= '0' && ch <= '8')) continue;
                int clue = ch - '0';
                int flaggedCount = 0;
                var unknowns = new List<(int r,int c)>();
                for (int k=0;k<8;k++)
                {
                    int nr = r + dy[k], nc = c + dx[k];
                    if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                    if (board[nr,nc] == '*') flaggedCount++;
                    else if (deducedMine[nr,nc]) flaggedCount++;
                    else if (board[nr,nc] == ' ' && !deducedSafe[nr,nc]) unknowns.Add((nr,nc));
                }
                int target = clue - flaggedCount;
                if (target <= 0 && unknowns.Count > 0)
                {
                    // all unknown neighbors safe
                    foreach(var p in unknowns)
                    {
                        if (!deducedSafe[p.r,p.c])
                        {
                            deducedSafe[p.r,p.c] = true;
                            changed = true;
                        }
                    }
                }
                else if (target == unknowns.Count && unknowns.Count > 0)
                {
                    // all unknown neighbors are mines
                    foreach(var p in unknowns)
                    {
                        if (!deducedMine[p.r,p.c])
                        {
                            deducedMine[p.r,p.c] = true;
                            changed = true;
                        }
                    }
                }
            }
        }
        return changed;
    }

    static bool ApplySubsetReasoning()
    {
        // Build list of constraints with their unknown neighbor sets (excluding deduced safe/mine)
        var cons = new List<(int r,int c, HashSet<(int,int)> set, int target)>();
        for (int r=0;r<H;r++)
        {
            for (int c=0;c<W;c++)
            {
                char ch = board[r,c];
                if (!(ch >= '0' && ch <= '8')) continue;
                int clue = ch - '0';
                int flaggedCount = 0;
                var unknowns = new HashSet<(int,int)>();
                for (int k=0;k<8;k++)
                {
                    int nr = r + dy[k], nc = c + dx[k];
                    if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                    if (board[nr,nc] == '*') flaggedCount++;
                    else if (deducedMine[nr,nc]) flaggedCount++;
                    else if (board[nr,nc] == ' ')
                    {
                        if (!deducedSafe[nr,nc] && !deducedMine[nr,nc])
                            unknowns.Add((nr,nc));
                    }
                }
                int target = clue - flaggedCount;
                if (unknowns.Count > 0)
                {
                    if (target < 0 || target > unknowns.Count)
                    {
                        // inconsistent, but we'll keep it and let other steps handle
                        // continue;
                    }
                    cons.Add((r,c,unknowns,target));
                }
            }
        }

        bool changed = false;

        // For each pair of constraints A and B, apply subset rules:
        // If A.set subset B.set:
        //  - if A.target == B.target -> B.set \ A.set are safe
        //  - if A.target + (|B.set|-|A.set|) == B.target -> B.set \ A.set are all mines
        // Also symmetric check.
        int n = cons.Count;
        for (int i=0;i<n;i++)
        {
            for (int j=i+1;j<n;j++)
            {
                var Ai = cons[i];
                var Aj = cons[j];
                var Aset = Ai.set;
                var Bset = Aj.set;
                int At = Ai.target;
                int Bt = Aj.target;

                // A subset B?
                if (IsSubset(Aset, Bset))
                {
                    int diff = Bset.Count - Aset.Count;
                    if (At == Bt)
                    {
                        // B \ A safe
                        foreach(var p in Bset.Except(Aset))
                        {
                            if (!deducedSafe[p.Item1,p.Item2])
                            {
                                deducedSafe[p.Item1,p.Item2] = true;
                                changed = true;
                            }
                        }
                    }
                    else if (At + diff == Bt)
                    {
                        // B \ A are mines
                        foreach(var p in Bset.Except(Aset))
                        {
                            if (!deducedMine[p.Item1,p.Item2])
                            {
                                deducedMine[p.Item1,p.Item2] = true;
                                changed = true;
                            }
                        }
                    }
                }

                // B subset A?
                if (IsSubset(Bset, Aset))
                {
                    int diff = Aset.Count - Bset.Count;
                    if (Bt == At)
                    {
                        foreach(var p in Aset.Except(Bset))
                        {
                            if (!deducedSafe[p.Item1,p.Item2])
                            {
                                deducedSafe[p.Item1,p.Item2] = true;
                                changed = true;
                            }
                        }
                    }
                    else if (Bt + diff == At)
                    {
                        foreach(var p in Aset.Except(Bset))
                        {
                            if (!deducedMine[p.Item1,p.Item2])
                            {
                                deducedMine[p.Item1,p.Item2] = true;
                                changed = true;
                            }
                        }
                    }
                }
            }
        }

        return changed;
    }

    static bool IsSubset(HashSet<(int,int)> a, HashSet<(int,int)> b)
    {
        if (a.Count > b.Count) return false;
        foreach(var x in a) if (!b.Contains(x)) return false;
        return true;
    }
}
```

Notes and design highlights
- Safety-first: the solver only outputs cells that are provably safe. It never guesses.
- Adaptive strategy:
  - Fast, linear-time local rules (0 neighbors => safe; full-neighbor => all mines).
  - Subset reasoning (standard logical rule used by many Minesweeper solvers).
  - Exact enumeration (complete solution search) limited to small frontier components (MAX_ENUM_VARS = 24). This limit avoids exponential blowups on enormous boards; you can raise it if you have time and memory budget.
- For massive boards the solver performs the cheap local reasoning and subset reasoning which are efficient and often deduce many safe cells. If a component remains too large for enumeration it is skipped (no guessing).
- The solver respects the input '*' flagged mines and further marks deduced mines internally (these affect reasoning but deduced mines are never output).
- Output coordinates are "x y" where x is column (0..W-1) and y is row (0..H-1), matching your spec.
