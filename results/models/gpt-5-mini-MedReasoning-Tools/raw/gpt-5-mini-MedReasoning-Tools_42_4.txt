; SPIR-V
; Geometry shader: expands each input line (2 verts) into a tapered ribbon triangle strip (4 verts)
; Emits: p0+perp*thick_start, p0-perp*thick_start, p1+perp*thick_end, p1-perp*thick_end
; Input locations: 0=worldPos[], 1=normal[], 2=color[]
; Output location 0 = outColor
; UBO (set=0,binding=0) declared but not used here (kept for compatibility)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %ubo "ubo"

; Decorations for built-in gl_PerVertex structs
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Locations for user I/O
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; UBO binding (set=0,binding=0)
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; GLSL.std.450 import for sqrt
%glsl = OpExtInstImport "GLSL.std.450"

; --- Types and constants ---
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%thick_start = OpConstant %float 0.04
%thick_end = OpConstant %float 0.005

; gl_PerVertex input struct (single vec4 position)
%gl_PerVertex_in = OpTypeStruct %v4float

; array of per-vertex gl_PerVertex (InputLines -> 2 entries)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; per-vertex input arrays: vec3[2]
%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; output color (per-emitted-vertex)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO (kept as a compatible layout). Not used in this shader logic but declared as available.
%mat4 = OpTypeMatrix %v4float 4
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_uniform_ubo Uniform

; --- Main function ---
%main = OpFunction %void None %void_fn
%entry = OpLabel

; --- Load endpoints and colors ---
; p0 = worldPos[0]
%ptr_world_p0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0_v3 = OpLoad %v3float %ptr_world_p0

; p1 = worldPos[1]
%ptr_world_p1 = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1_v3 = OpLoad %v3float %ptr_world_p1

; c0 = color[0]
%ptr_color_p0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0_v3 = OpLoad %v3float %ptr_color_p0

; c1 = color[1]
%ptr_color_p1 = OpAccessChain %ptr_in_v3 %color %int_1
%c1_v3 = OpLoad %v3float %ptr_color_p1

; --- Extract XY and Z components ---
%p0_x = OpCompositeExtract %float %p0_v3 0
%p0_y = OpCompositeExtract %float %p0_v3 1
%p0_z = OpCompositeExtract %float %p0_v3 2
%p1_x = OpCompositeExtract %float %p1_v3 0
%p1_y = OpCompositeExtract %float %p1_v3 1
%p1_z = OpCompositeExtract %float %p1_v3 2

%p0_xy = OpCompositeConstruct %v2float %p0_x %p0_y
%p1_xy = OpCompositeConstruct %v2float %p1_x %p1_y

; --- Compute direction and perpendicular in 2D ---
%dir = OpFSub %v2float %p1_xy %p0_xy       ; dir = p1.xy - p0.xy
%dot_dir = OpDot %float %dir %dir
%sqrt_len = OpExtInst %float %glsl Sqrt %dot_dir
%inv_len = OpFDiv %float %f_1 %sqrt_len
%dir_norm = OpVectorTimesScalar %v2float %dir %inv_len

; dir_norm.x and dir_norm.y
%dir_n_x = OpCompositeExtract %float %dir_norm 0
%dir_n_y = OpCompositeExtract %float %dir_norm 1

; perp = vec2(-dir.y, dir.x)
%neg_dir_n_y = OpFNegate %float %dir_n_y
%perp = OpCompositeConstruct %v2float %neg_dir_n_y %dir_n_x

; scale perp by start/end thickness
%perp_start = OpVectorTimesScalar %v2float %perp %thick_start
%perp_end = OpVectorTimesScalar %v2float %perp %thick_end

; --- Compute four corner positions in XY ---
%p0_plus_xy = OpFAdd %v2float %p0_xy %perp_start
%p0_minus_xy = OpFSub %v2float %p0_xy %perp_start
%p1_plus_xy = OpFAdd %v2float %p1_xy %perp_end
%p1_minus_xy = OpFSub %v2float %p1_xy %perp_end

; Extract XY components for assembly into vec4
%p0_plus_x = OpCompositeExtract %float %p0_plus_xy 0
%p0_plus_y = OpCompositeExtract %float %p0_plus_xy 1
%p0_minus_x = OpCompositeExtract %float %p0_minus_xy 0
%p0_minus_y = OpCompositeExtract %float %p0_minus_xy 1
%p1_plus_x = OpCompositeExtract %float %p1_plus_xy 0
%p1_plus_y = OpCompositeExtract %float %p1_plus_xy 1
%p1_minus_x = OpCompositeExtract %float %p1_minus_xy 0
%p1_minus_y = OpCompositeExtract %float %p1_minus_xy 1

; Build vec4 positions (z from original vertex, w=1.0)
%p0_plus_v4 = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0_z %f_1
%p0_minus_v4 = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0_z %f_1
%p1_plus_v4 = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %p1_z %f_1
%p1_minus_v4 = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %p1_z %f_1

; Pointer to gl_out.position member (index 0)
%ptr_out_pos = OpAccessChain %ptr_out_pv %gl_out %int_0

; --- Emit vertices in triangle strip order ---
; 1) p0 + perp*thick_start with color c0
OpStore %ptr_out_pos %p0_plus_v4
OpStore %outColor %c0_v3
OpEmitVertex

; 2) p0 - perp*thick_start with color c0
OpStore %ptr_out_pos %p0_minus_v4
OpStore %outColor %c0_v3
OpEmitVertex

; 3) p1 + perp*thick_end with color c1
OpStore %ptr_out_pos %p1_plus_v4
OpStore %outColor %c1_v3
OpEmitVertex

; 4) p1 - perp*thick_end with color c1
OpStore %ptr_out_pos %p1_minus_v4
OpStore %outColor %c1_v3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
