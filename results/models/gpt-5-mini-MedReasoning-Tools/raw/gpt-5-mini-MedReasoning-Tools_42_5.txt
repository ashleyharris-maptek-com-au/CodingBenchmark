; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 200
; Schema: 0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Imports
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%uint_2 = OpConstant %uint 2

%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%thickness = OpConstant %float 0.015
%fract75 = OpConstant %float 0.75
%arrowWidth = OpConstant %float 0.05

; Pointer types
%ptr_Input_v3 = OpTypePointer Input %v3f
%ptr_Input_v4 = OpTypePointer Input %v4f
%ptr_Output_v3 = OpTypePointer Output %v3f
%ptr_Output_v4 = OpTypePointer Output %v4f
%ptr_Uniform = OpTypePointer Uniform %v4f
%ptr_Input_arr_pv = OpTypePointer Input %v4f

; gl_PerVertex struct types for input (array element struct)
%gl_PerVertex_in = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_PerVertex_out struct
%gl_PerVertex_out = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; arrays and variables for per-vertex builtins (array size = 2 for InputLines)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output per-vertex variable
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined inputs (arrays)
%arr_v3_in = OpTypeArray %v3f %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output color
%ptr_out_v3 = OpTypePointer Output %v3f
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Uniform buffer object (set=0 binding=0) - declared but not used here
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
OpDecorate %ubo_struct Block
%ubo_ptr = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ubo_ptr Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Function main
%main = OpFunction %void None %void_fn
%main_entry = OpLabel

; Load p0 and p1 from worldPos array
%ptr_p0 = OpAccessChain %ptr_Input_v3 %worldPos %int_0
%ptr_p1 = OpAccessChain %ptr_Input_v3 %worldPos %int_1
%p0 = OpLoad %v3f %ptr_p0
%p1 = OpLoad %v3f %ptr_p1

; Load colors c0 and c1
%ptr_c0 = OpAccessChain %ptr_Input_v3 %color %int_0
%ptr_c1 = OpAccessChain %ptr_Input_v3 %color %int_1
%c0 = OpLoad %v3f %ptr_c0
%c1 = OpLoad %v3f %ptr_c1

; dir = p1 - p0
%dir = OpFSub %v3f %p1 %p0

; Extract dir.x and dir.y
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; perp2 = normalize(vec2(-dir.y, dir.x))
%neg_dir_y = OpFNegate %float %dir_y
%perp2_unnorm = OpCompositeConstruct %v2f %neg_dir_y %dir_x
%perp2 = OpExtInst %v2f %glsl Normalize %perp2_unnorm

; perp scaled by thickness (vec2)
%perp_thick = OpVectorTimesScalar %v2f %perp2 %thickness

; Make perp as vec3 with z = 0
%perp3_thick = OpCompositeConstruct %v3f
              (OpCompositeExtract %float %perp_thick 0)
              (OpCompositeExtract %float %perp_thick 1)
              %f0

; mid = p0 + 0.75 * dir
%dir_scaled = OpVectorTimesScalar %v3f %dir %fract75
%mid = OpFAdd %v3f %p0 %dir_scaled

; Precompute perp for arrowhead (0.05)
%perp_big2 = OpVectorTimesScalar %v2f %perp2 %arrowWidth
%perp3_big = OpCompositeConstruct %v3f
             (OpCompositeExtract %float %perp_big2 0)
             (OpCompositeExtract %float %perp_big2 1)
             %f0

; Create v4 positions when needed (vec3 + w=1.0)
; p0 - perp
%p0_minus = OpFSub %v3f %p0 %perp3_thick
%p0_minus_v4 = OpCompositeConstruct %v4f
               (OpCompositeExtract %float %p0_minus 0)
               (OpCompositeExtract %float %p0_minus 1)
               (OpCompositeExtract %float %p0_minus 2)
               %f1

; p0 + perp
%p0_plus = OpFAdd %v3f %p0 %perp3_thick
%p0_plus_v4 = OpCompositeConstruct %v4f
              (OpCompositeExtract %float %p0_plus 0)
              (OpCompositeExtract %float %p0_plus 1)
              (OpCompositeExtract %float %p0_plus 2)
              %f1

; mid - perp
%mid_minus = OpFSub %v3f %mid %perp3_thick
%mid_minus_v4 = OpCompositeConstruct %v4f
                (OpCompositeExtract %float %mid_minus 0)
                (OpCompositeExtract %float %mid_minus 1)
                (OpCompositeExtract %float %mid_minus 2)
                %f1

; mid + perp
%mid_plus = OpFAdd %v3f %mid %perp3_thick
%mid_plus_v4 = OpCompositeConstruct %v4f
               (OpCompositeExtract %float %mid_plus 0)
               (OpCompositeExtract %float %mid_plus 1)
               (OpCompositeExtract %float %mid_plus 2)
               %f1

; arrowhead vertices: mid + perp_big, mid - perp_big, p1
%mid_big_plus = OpFAdd %v3f %mid %perp3_big
%mid_big_plus_v4 = OpCompositeConstruct %v4f
                   (OpCompositeExtract %float %mid_big_plus 0)
                   (OpCompositeExtract %float %mid_big_plus 1)
                   (OpCompositeExtract %float %mid_big_plus 2)
                   %f1

%mid_big_minus = OpFSub %v3f %mid %perp3_big
%mid_big_minus_v4 = OpCompositeConstruct %v4f
                    (OpCompositeExtract %float %mid_big_minus 0)
                    (OpCompositeExtract %float %mid_big_minus 1)
                    (OpCompositeExtract %float %mid_big_minus 2)
                    %f1

%p1_v4 = OpCompositeConstruct %v4f
         (OpCompositeExtract %float %p1 0)
         (OpCompositeExtract %float %p1 1)
         (OpCompositeExtract %float %p1 2)
         %f1

; --- Emit shaft (triangle strip, 4 verts) with color c0 ---
; vertex 0: p0 - perp
%ptr_out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %ptr_out_pos %p0_minus_v4
OpStore %outColor %c0
OpEmitVertex

; vertex 1: p0 + perp
OpStore %ptr_out_pos %p0_plus_v4
OpStore %outColor %c0
OpEmitVertex

; vertex 2: mid - perp
OpStore %ptr_out_pos %mid_minus_v4
OpStore %outColor %c0
OpEmitVertex

; vertex 3: mid + perp
OpStore %ptr_out_pos %mid_plus_v4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; --- Emit arrowhead (triangle strip, 3 verts) with color c1 ---
; vertex 4: mid + perp_big
OpStore %ptr_out_pos %mid_big_plus_v4
OpStore %outColor %c1
OpEmitVertex

; vertex 5: mid - perp_big
OpStore %ptr_out_pos %mid_big_minus_v4
OpStore %outColor %c1
OpEmitVertex

; vertex 6: p1 (tip)
OpStore %ptr_out_pos %p1_v4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
