; SPIR-V
; Geometry shader: expands each point into a small colored quad (triangle strip of 4 vertices)
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Entry point must list ALL interface variables (inputs and outputs)
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; -------------------------------------------------------------------------
; Names (for readability)
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %outColor "outColor"

; -------------------------------------------------------------------------
; Types and constants
%void       = OpTypeVoid
%func_ty    = OpTypeFunction %void

%float      = OpTypeFloat 32
%v2float    = OpTypeVector %float 2
%v3float    = OpTypeVector %float 3
%v4float    = OpTypeVector %float 4

%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%uint_1    = OpConstant %uint 1

%float_0   = OpConstant %float 0.0
%float_1   = OpConstant %float 1.0
%h_const   = OpConstant %float 0.04

; Pointer types
%ptr_input_v4  = OpTypePointer Input %v4float
%ptr_input_v3  = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_output_v3 = OpTypePointer Output %v3float

; -------------------------------------------------------------------------
; gl_PerVertex (input) and gl_in array (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

; gl_PerVertex (output) and gl_out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_gl_pervertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_gl_pervertex_out Output

; -------------------------------------------------------------------------
; User-defined per-vertex input arrays (array size = 1 for InputPoints)
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_1

%in_worldPos = OpVariable %ptr_arr_v3_in Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_arr_v3_in Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_arr_v3_in Input
OpDecorate %in_color Location 2

; -------------------------------------------------------------------------
; Per-vertex output (to fragment shader)
%outColor = OpVariable %ptr_output_v3 Output
OpDecorate %outColor Location 0

; -------------------------------------------------------------------------
; Main function
%main = OpFunction %void None %func_ty
%entry = OpLabel

; Load gl_in[0].gl_Position
%ptr_pos_in_elem = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%pos_in = OpLoad %v4float %ptr_pos_in_elem

; Extract x,y from input position
%x_in = OpCompositeExtract %float %pos_in 0
%y_in = OpCompositeExtract %float %pos_in 1

; Load color[0]
%ptr_color_elem = OpAccessChain %ptr_input_v3 %in_color %int_0
%color_val = OpLoad %v3float %ptr_color_elem

; Compute corner offsets: h = 0.04
%x_minus_h = OpFSub %float %x_in %h_const
%x_plus_h  = OpFAdd %float %x_in %h_const
%y_minus_h = OpFSub %float %y_in %h_const
%y_plus_h  = OpFAdd %float %y_in %h_const

; Build four vec4 positions (z=0, w=1)
%p0 = OpCompositeConstruct %v4float %x_minus_h %y_minus_h %float_0 %float_1    ; (x-h, y-h)
%p1 = OpCompositeConstruct %v4float %x_plus_h  %y_minus_h %float_0 %float_1    ; (x+h, y-h)
%p2 = OpCompositeConstruct %v4float %x_minus_h %y_plus_h  %float_0 %float_1    ; (x-h, y+h)
%p3 = OpCompositeConstruct %v4float %x_plus_h  %y_plus_h  %float_0 %float_1    ; (x+h, y+h)

; Pointer to gl_out.gl_Position member
%ptr_gl_out_pos = OpAccessChain %ptr_output_v4 %gl_out %int_0

; Emit vertex 0
OpStore %ptr_gl_out_pos %p0
OpStore %outColor %color_val
OpEmitVertex

; Emit vertex 1
OpStore %ptr_gl_out_pos %p1
OpStore %outColor %color_val
OpEmitVertex

; Emit vertex 2
OpStore %ptr_gl_out_pos %p2
OpStore %outColor %color_val
OpEmitVertex

; Emit vertex 3
OpStore %ptr_gl_out_pos %p3
OpStore %outColor %color_val
OpEmitVertex

; Finish primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
