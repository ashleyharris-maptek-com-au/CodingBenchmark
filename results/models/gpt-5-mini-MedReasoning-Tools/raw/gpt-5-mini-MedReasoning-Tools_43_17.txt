struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Brick pattern parameters
    float2 uvScale = float2(8.0, 16.0);   // 8x16 bricks across UV
    float mortarX = 0.05;                 // mortar thickness in U (fraction of brick)
    float mortarY = 0.10;                 // mortar thickness in V (fraction of brick)
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);

    // Scale UVs to brick grid
    float2 uvS = input.uv * uvScale;

    // Determine row and staggering offset (half-brick on every other row)
    float row = floor(uvS.y);
    float rowMod2 = fmod(row, 2.0);
    float offset = (rowMod2 > 0.5) ? 0.5 : 0.0;

    // Apply offset and get local cell coordinates
    float2 uvOffset = uvS + float2(offset, 0.0);
    float2 local = frac(uvOffset); // [0,1) inside brick cell

    // Determine brick vs mortar (no smoothing)
    float maskX = step(mortarX, local.x) * step(local.x, 1.0 - mortarX);
    float maskY = step(mortarY, local.y) * step(local.y, 1.0 - mortarY);
    float brickMask = maskX * maskY; // 1.0 = brick, 0.0 = mortar

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lambertFactor = 0.4 + 0.6 * diffuse;

    // Final color
    float3 baseColor = lerp(mortarColor, brickColor, brickMask);
    float3 finalColor = baseColor * lambertFactor;

    return float4(finalColor, 1.0);
}
