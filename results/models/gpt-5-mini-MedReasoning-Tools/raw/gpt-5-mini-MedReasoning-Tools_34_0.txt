```python
#!/usr/bin/env python3
"""
CVRP heuristic solver.

Input (stdin):
N K CAPACITY
depot_x depot_y
x1 y1 demand1
...
xN yN demandN

Output (stdout):
total_distance
num_routes
route_1
route_2
...

Each route line lists 0-based customer indices in visit order.
"""

import sys
import math
from collections import namedtuple

Customer = namedtuple("Customer", ["x", "y", "demand"])

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return 0, 0, 0.0, (0.0, 0.0), []
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))
    depot_x = float(next(it)); depot_y = float(next(it))
    customers = []
    for i in range(N):
        x = float(next(it)); y = float(next(it)); d = float(next(it))
        customers.append(Customer(x, y, d))
    return N, K, CAP, (depot_x, depot_y), customers

def euclid(a, b):
    return math.hypot(a[0]-b[0], a[1]-b[1])

def build_distance_matrix(depot, customers):
    N = len(customers)
    coords = [depot] + [(c.x, c.y) for c in customers]
    D = [[0.0]*(N+1) for _ in range(N+1)]
    for i in range(N+1):
        for j in range(N+1):
            if i==j:
                D[i][j]=0.0
            else:
                D[i][j] = euclid(coords[i], coords[j])
    return D

def route_distance(route, D):
    # route: list of customer indices (0-based)
    if not route:
        return 0.0
    dist = 0.0
    # depot is index 0 in D; customers shifted by +1
    prev = 0  # depot
    for c in route:
        dist += D[prev][c+1]
        prev = c+1
    dist += D[prev][0]
    return dist

def total_distance(routes, D):
    return sum(route_distance(r, D) for r in routes)

def clarke_wright(N, K, CAP, depot, customers, D):
    demands = [c.demand for c in customers]
    # initial routes: one customer each
    routes = [[i] for i in range(N)]
    loads = [demands[i] for i in range(N)]

    # quick infeasibility checks
    infeasible_customers = [i for i,d in enumerate(demands) if d > CAP]
    if infeasible_customers:
        sys.stderr.write("Warning: some customer demand > capacity. Cannot respect capacity for these customers.\n")
        # We'll still proceed but it's infeasible to strictly respect capacity

    # savings list
    savings = []
    for i in range(N):
        for j in range(i+1, N):
            # d0i + d0j - dij
            s = D[0][i+1] + D[0][j+1] - D[i+1][j+1]
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])

    # mapping customer -> route index
    cust_route = list(range(N))

    for s, i, j in savings:
        ri = cust_route[i]
        rj = cust_route[j]
        if ri == rj:
            continue
        route_i = routes[ri]
        route_j = routes[rj]
        load_i = loads[ri]
        load_j = loads[rj]
        # Check feasible merge only if combined load <= CAP (allow if infeasible exists earlier)
        if (load_i + load_j) > CAP and not infeasible_customers:
            continue
        # Check if i is at an end of route_i and j is at an end of route_j in an orientation that allows merge without reversing internal order
        can_merge = False
        # try route_i end with route_j start
        if route_i[-1] == i and route_j[0] == j:
            new_route = route_i + route_j
            can_merge = True
        # try route_j end with route_i start => merge as route_j + route_i
        elif route_j[-1] == j and route_i[0] == i:
            new_route = route_j + route_i
            can_merge = True
        # try route_i start with route_j start: reverse one
        elif route_i[0] == i and route_j[0] == j:
            # reverse route_i and append
            new_route = list(reversed(route_i)) + route_j
            can_merge = True
        # try route_i end with route_j end
        elif route_i[-1] == i and route_j[-1] == j:
            # reverse route_j and append
            new_route = route_i + list(reversed(route_j))
            can_merge = True
        else:
            can_merge = False

        if not can_merge:
            continue

        # perform merge
        routes[ri] = new_route
        loads[ri] = load_i + load_j
        # remove route rj by moving last into rj pos
        # to maintain indices we replace rj with last and update cust_route mapping
        last_index = len(routes)-1
        if rj != last_index:
            # move last route into rj slot
            routes[rj] = routes[last_index]
            loads[rj] = loads[last_index]
            for c in routes[rj]:
                cust_route[c] = rj
        # shrink arrays
        routes.pop()
        loads.pop()
        # update cust_route for merged route
        for c in new_route:
            cust_route[c] = ri

        # stop early if we reached 1 route or satisfied K? We'll continue to greedily merge as possible
        if len(routes) <= K:
            # still try to improve further merges that don't exceed capacity to reduce distance,
            # but respecting limit on vehicle count is handled later (we won't force raising beyond K)
            pass

    # If after savings we still have more routes than K, try greedy merges minimizing distance increase
    def feasible_merge_pairs():
        pairs = []
        R = len(routes)
        for a in range(R):
            for b in range(a+1, R):
                if loads[a] + loads[b] > CAP and not infeasible_customers:
                    continue
                # try 4 orientations and pick orientation with minimal new distance
                best_new = None
                best_route = None
                ra = routes[a]; rb = routes[b]
                # orientations: ra + rb (maybe reverse ends)
                candidates = [
                    (ra + rb),
                    (ra + list(reversed(rb))),
                    (list(reversed(ra)) + rb),
                    (list(reversed(ra)) + list(reversed(rb))),
                ]
                for cand in candidates:
                    nd = route_distance(cand, D)
                    if best_new is None or nd < best_new:
                        best_new = nd
                        best_route = cand
                current = route_distance(ra, D) + route_distance(rb, D)
                increase = best_new - current
                pairs.append((increase, a, b, best_route))
        pairs.sort(key=lambda x: x[0])
        return pairs

    # Merge until <= K or no feasible merges
    while len(routes) > K:
        pairs = feasible_merge_pairs()
        if not pairs:
            break
        merged = False
        for increase, a, b, newr in pairs:
            # check loads
            if loads[a] + loads[b] > CAP and not infeasible_customers:
                continue
            # perform merge of a and b (we'll merge into index a and replace b with last)
            routes[a] = newr
            loads[a] = loads[a] + loads[b]
            last_index = len(routes)-1
            if b != last_index:
                routes[b] = routes[last_index]
                loads[b] = loads[last_index]
                for c in routes[b]:
                    # update mapping not strictly necessary for further loops here
                    pass
            routes.pop(); loads.pop()
            merged = True
            break
        if not merged:
            break

    if len(routes) > K:
        # Can't reduce to K without violating capacity; bump K up to achievable count
        sys.stderr.write(f"Note: required number of vehicles to respect capacity is {len(routes)} which is > given K={K}. Using {len(routes)} vehicles\n")
        K = len(routes)

    # Apply 2-opt on each route to improve order (intra-route)
    # Simple iterative 2-opt
    for idx in range(len(routes)):
        r = routes[idx]
        if len(r) <= 2:
            continue
        improved = True
        while improved:
            improved = False
            best_delta = 0.0
            best_i = best_j = None
            for i in range(0, len(r)-1):
                for j in range(i+1, len(r)):
                    if j - i == 1:
                        continue
                    # compute delta of reversing r[i+1:j+1] (classic 2-opt) on closed route
                    # nodes sequence: depot -> ... r[i] -> r[i+1]...r[j] -> r[j+1] ... -> depot
                    a = 0 if i < 0 else r[i]
                    b = r[i+1]
                    c = r[j]
                    d = 0 if j+1 >= len(r) else r[j+1]
                    # convert to indices in D (shift +1)
                    da_b = D[0 if i<0 else a+1][b+1]
                    bc = D[b+1][c+1]
                    cd = D[c+1][0 if j+1>=len(r) else d+1]
                    # after reversal: a->c and b->d
                    ac = D[0 if i<0 else a+1][c+1]
                    bd = D[b+1][0 if j+1>=len(r) else d+1]
                    delta = (ac + bd) - (da_b + cd)
                    if delta < best_delta - 1e-9:
                        best_delta = delta
                        best_i = i+1
                        best_j = j
            if best_i is not None and best_j is not None:
                # apply reversal r[best_i:best_j+1]
                r[best_i:best_j+1] = reversed(r[best_i:best_j+1])
                routes[idx] = r
                improved = True

    return routes, K

def main():
    N, K, CAP, depot, customers = read_input()
    if N == 0:
        print("0.0")
        print("0")
        return

    D = build_distance_matrix(depot, customers)
    total_demand = sum(c.demand for c in customers)
    min_veh = math.ceil(total_demand / CAP) if CAP > 0 else len(customers)

    if any(c.demand > CAP for c in customers):
        sys.stderr.write("Warning: At least one customer's demand exceeds vehicle capacity. Strict feasibility impossible.\n")

    if min_veh > K:
        sys.stderr.write(f"Warning: given K={K} is less than minimum required vehicles {min_veh} by capacity. Solver will use at least {min_veh} vehicles.\n")

    routes, usedK = clarke_wright(N, K, CAP, depot, customers, D)

    # Final sanity: ensure all customers covered exactly once
    covered = [c for r in routes for c in r]
    if len(covered) != N or set(covered) != set(range(N)):
        # fallback: simple greedy fill respecting capacity (makes sure coverage)
        sys.stderr.write("Warning: initial solution invalid. Falling back to greedy-first-fit.\n")
        demands = [c.demand for c in customers]
        remaining = set(range(N))
        routes = []
        # sort customers by descending demand to pack
        order = sorted(range(N), key=lambda i: -demands[i])
        for i in order:
            placed = False
            for r in routes:
                load = sum(customers[c].demand for c in r)
                if load + customers[i].demand <= CAP or any(customers[c].demand > CAP for c in [i]):
                    r.append(i); placed = True; break
            if not placed:
                routes.append([i])
        usedK = len(routes)

    # Final improvement: try to reassign small customers to reduce vehicles if >K allowed?
    # (We've already adjusted K to min feasible in solver.)

    dist = total_distance(routes, D)
    # print with 6 decimal places
    print(f"{dist:.6f}")
    print(len(routes))
    for r in routes:
        if r:
            print(" ".join(str(x) for x in r))
        else:
            print()

if __name__ == "__main__":
    main()
```
