#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main()
{
    uint N = counts.x;
    uint i = gl_GlobalInvocationID.x;

    // Guard against out-of-range invocations
    if (i >= N) {
        return;
    }

    // Read this body's position and velocity from input buffer
    uint base_i = i * 2u;
    vec3 pos_i = inData[int(base_i)].xyz;
    float mass_i = inData[int(base_i)].w;
    vec3 vel_i = inData[int(base_i + 1u)].xyz;

    // Simulation parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float soft2 = softening * softening;

    // Accumulate acceleration from all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue; // skip self-interaction

        uint base_j = j * 2u;
        vec3 pos_j = inData[int(base_j)].xyz;
        float mass_j = inData[int(base_j)].w;

        vec3 r = pos_j - pos_i;
        float dist2 = dot(r, r) + soft2; // |r|^2 + softening^2

        // Compute 1 / (|r|^2 + soft^2)^(3/2) safely using inversesqrt
        float invDist = inversesqrt(dist2);         // 1 / sqrt(dist2)
        float invDist3 = invDist * invDist * invDist; // 1 / dist2^(3/2)

        acc += G * mass_j * r * invDist3;
    }

    // Integration: semi-implicit (symplectic) Euler
    // Velocity advanced by acc * dt, then position advanced by (updated) velocity * dt
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_new * dt;

    // Write updated position+mass and velocity to output buffer
    outData[int(base_i)] = vec4(pos_new, mass_i);
    outData[int(base_i + 1u)] = vec4(vel_new, 0.0);
}
