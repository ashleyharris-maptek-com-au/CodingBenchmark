fn rotl32(x : u32, s : u32) -> u32 {
    return ((x << s) | (x >> (32u - s)));
}

fn ctz32(x : u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }
    var v : u32 = x;
    var cnt : u32 = 0u;
    loop {
        if ((v & 1u) == 1u) {
            break;
        }
        cnt = cnt + 1u;
        v = v >> 1u;
    }
    return cnt;
}

fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let gid_x : u32 = gid.x;

    // range guard: if gid.x >= params.z, return
    if (gid_x >= params.z) {
        return;
    }

    let nonce : u32 = params.y + gid_x;
    let hv : vec2<u32> = custom_hash_with_nonce(nonce);
    let h0 : u32 = hv.x;
    let h1 : u32 = hv.y;

    // Count trailing zeros in 64-bit value [h0 | (h1 << 32)]
    var tz : u32;
    if (h0 != 0u) {
        tz = ctz32(h0);
    } else {
        tz = 32u + ctz32(h1);

        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
