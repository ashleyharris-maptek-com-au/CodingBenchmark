```javascript
// Node.js program to plan delivery routes for a fleet of vans.
// Reads one JSON document from stdin with fields: depot, truckCount, packages
// Writes one JSON document to stdout: {"routes": [[packageId,...], ...]}
// Deterministic, scalable clustering + nearest-neighbor route construction.
// No logs printed to stdout.

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
let truckCount = Math.max(1, Math.floor(Number(data.truckCount) || 1));
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];

if (packages.length === 0) {
  // No packages: return empty routes array
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const n = packages.length;
const k = Math.min(truckCount, n);

// Precompute arrays for coordinates and keep original ids
const xs = new Array(n);
const ys = new Array(n);
const ids = new Array(n);
for (let i = 0; i < n; i++) {
  const p = packages[i];
  xs[i] = Number(p.x) || 0;
  ys[i] = Number(p.y) || 0;
  ids[i] = p.id;
}

// Utility: squared Euclidean distance
function sqDist(x1, y1, x2, y2) {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return dx * dx + dy * dy;
}

// Deterministic seed selection for k-means (farthest-first / spread)
// Choose first centroid as the point with smallest x, then smallest y (deterministic)
function initialCentroids() {
  const centroids = [];
  // find index with min x then min y
  let minIdx = 0;
  for (let i = 1; i < n; i++) {
    if (xs[i] < xs[minIdx] || (xs[i] === xs[minIdx] && ys[i] < ys[minIdx])) {
      minIdx = i;
    }
  }
  centroids.push({ x: xs[minIdx], y: ys[minIdx] });

  while (centroids.length < k) {
    // For each point, compute distance to nearest existing centroid
    let bestIdx = -1;
    let bestScore = -1;
    for (let i = 0; i < n; i++) {
      const px = xs[i], py = ys[i];
      let minD = Infinity;
      for (const c of centroids) {
        const d = sqDist(px, py, c.x, c.y);
        if (d < minD) minD = d;
      }
      // choose point with maximum minD; deterministic tie-break by index (lower index wins)
      if (minD > bestScore) {
        bestScore = minD;
        bestIdx = i;
      }
    }
    // If bestScore is 0 (all points coincide), break to avoid duplicates
    if (bestScore <= 0) break;
    centroids.push({ x: xs[bestIdx], y: ys[bestIdx] });
  }

  // If we have fewer than k centroids (all points identical), duplicate as needed
  while (centroids.length < k) {
    centroids.push({ x: centroids[0].x, y: centroids[0].y });
  }

  return centroids;
}

// k-means clustering (Euclidean) deterministic
function kMeans() {
  let centroids = initialCentroids();
  const assigns = new Array(n).fill(-1);
  const maxIter = 100;
  for (let iter = 0; iter < maxIter; iter++) {
    let changed = false;
    // assignment step: assign each point to nearest centroid
    for (let i = 0; i < n; i++) {
      let best = -1;
      let bestD = Infinity;
      for (let j = 0; j < centroids.length; j++) {
        const d = sqDist(xs[i], ys[i], centroids[j].x, centroids[j].y);
        if (d < bestD || (d === bestD && j < best)) {
          bestD = d;
          best = j;
        }
      }
      if (assigns[i] !== best) {
        assigns[i] = best;
        changed = true;
      }
    }

    // If no change, break
    if (!changed) break;

    // update centroids
    const sumsX = new Array(centroids.length).fill(0);
    const sumsY = new Array(centroids.length).fill(0);
    const counts = new Array(centroids.length).fill(0);
    for (let i = 0; i < n; i++) {
      const a = assigns[i];
      sumsX[a] += xs[i];
      sumsY[a] += ys[i];
      counts[a] += 1;
    }
    for (let j = 0; j < centroids.length; j++) {
      if (counts[j] === 0) {
        // reinitialize empty centroid deterministically: pick point farthest from all current centroids
        let bestIdx = -1;
        let bestScore = -1;
        for (let i = 0; i < n; i++) {
          // distance to nearest centroid
          let minD = Infinity;
          for (let c = 0; c < centroids.length; c++) {
            const d = sqDist(xs[i], ys[i], centroids[c].x, centroids[c].y);
            if (d < minD) minD = d;
          }
          if (minD > bestScore) {
            bestScore = minD;
            bestIdx = i;
          }
        }
        if (bestIdx >= 0) {
          centroids[j] = { x: xs[bestIdx], y: ys[bestIdx] };
        }
      } else {
        centroids[j] = { x: sumsX[j] / counts[j], y: sumsY[j] / counts[j] };
      }
    }
  }

  // Build clusters list of indices
  const clusters = Array.from({ length: centroids.length }, () => []);
  for (let i = 0; i < n; i++) {
    clusters[assigns[i]].push(i);
  }

  // Remove empty clusters (we will only return non-empty clusters, up to k)
  const nonEmpty = clusters.filter(c => c.length > 0);
  return nonEmpty;
}

// Nearest-neighbor TSP route construction starting and ending at depot
// Input: array of package indices (into xs, ys, ids). Returns array of package ids in visit order.
function buildRouteForCluster(clusterIndices) {
  // If clusterIndices is empty return empty
  if (!clusterIndices || clusterIndices.length === 0) return [];

  // We'll keep an array of unvisited indices; use simple nearest neighbor
  const unvisited = clusterIndices.slice(); // contains package indices
  // For deterministic tie-breaking when distances equal, we will compare by original package array index
  // Start from depot
  let cx = depot[0], cy = depot[1];
  const route = [];
  while (unvisited.length > 0) {
    let bestIdxInUnvisited = 0;
    let bestDist = sqDist(cx, cy, xs[unvisited[0]], ys[unvisited[0]]);
    for (let i = 1; i < unvisited.length; i++) {
      const idx = unvisited[i];
      const d = sqDist(cx, cy, xs[idx], ys[idx]);
      if (d < bestDist || (d === bestDist && idx < unvisited[bestIdxInUnvisited])) {
        bestDist = d;
        bestIdxInUnvisited = i;
      }
    }
    const chosen = unvisited.splice(bestIdxInUnvisited, 1)[0];
    route.push(ids[chosen]);
    cx = xs[chosen];
    cy = ys[chosen];
  }
  // route now is sequence from depot to last; return sequence (depot implicit)
  return route;
}

// If k === 1 special-case: build a single route with nearest-neighbor over all points
let clusters = kMeans();

// If there are more clusters than truckCount (shouldn't happen), merge smallest until count==k
if (clusters.length > k) {
  // sort clusters by size, keep largest k, merge the rest into nearest centroid cluster by average location
  clusters.sort((a, b) => b.length - a.length); // descending
  clusters = clusters.slice(0, k);
}

// If clusters fewer than k and truckCount > clusters, it's fine — we'll output fewer routes than truckCount.
const routes = [];
for (const cl of clusters) {
  // Build route order for each cluster
  const routeIds = buildRouteForCluster(cl);
  routes.push(routeIds);
}

// As a small improvement: if number of routes is less than k and there are routes with many points,
// optionally split the largest routes to increase route count up to k for better load balancing.
// We'll split by simple halving deterministically until routes.length == k or no route has >1 package.
if (routes.length < k) {
  // Work on indices referencing routes array
  while (routes.length < k) {
    // find largest route with more than 1 package
    let bestIdx = -1;
    let bestLen = 1;
    for (let i = 0; i < routes.length; i++) {
      if (routes[i].length > bestLen) {
        bestLen = routes[i].length;
        bestIdx = i;
      }
    }
    if (bestIdx === -1) break; // no splittable route
    const routeToSplit = routes[bestIdx];
    const half = Math.floor(routeToSplit.length / 2);
    const a = routeToSplit.slice(0, half);
    const b = routeToSplit.slice(half);
    // Replace routeToSplit with a and append b
    routes[bestIdx] = a;
    routes.push(b);
  }
}

// Ensure every package appears exactly once. If somehow some package missing or duplicated, fall back to deterministic assignment:
// Build a map of assigned ids
const assigned = new Map();
for (const r of routes) {
  for (const pid of r) assigned.set(pid, (assigned.get(pid) || 0) + 1);
}
if (assigned.size !== ids.length || Array.from(assigned.values()).some(v => v !== 1)) {
  // fallback deterministic simple partition: sort by angle around depot then chunk
  const pts = [];
  const dx0 = depot[0], dy0 = depot[1];
  for (let i = 0; i < n; i++) {
    const ang = Math.atan2(ys[i] - dy0, xs[i] - dx0);
    pts.push({ idx: i, ang });
  }
  pts.sort((a, b) => a.ang - b.ang || a.idx - b.idx);
  const newRoutes = Array.from({ length: k }, () => []);
  for (let i = 0; i < pts.length; i++) {
    const bucket = i % k;
    newRoutes[bucket].push(ids[pts[i].idx]);
  }
  // replace routes with newRoutes but remove empty ones at end
  const nonEmpty = newRoutes.filter(r => r.length > 0);
  console.log(JSON.stringify({ routes: nonEmpty }));
  process.exit(0);
}

// Output exactly the routes array
console.log(JSON.stringify({ routes }));
```
