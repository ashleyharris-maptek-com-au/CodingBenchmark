#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 colorIn;

layout(location = 0) out vec4 outColor;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;   // lightPos.xyz = position
    vec4 cameraPos;
    vec4 params;
} ubo;

void main()
{
    // Brick pattern parameters
    vec2 scale = vec2(8.0, 16.0);    // 8x16 UV scale
    float mortarX = 0.05;           // mortar thickness in X
    float mortarY = 0.10;           // mortar thickness in Y
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor  = vec3(0.7, 0.2, 0.1);

    // Scale UVs
    vec2 suv = uv * scale;

    // Compute row and offset every other row (stagger by half brick)
    float row = floor(suv.y);
    float rowMod2 = mod(row, 2.0);
    float xOffset = rowMod2 * 0.5; // 0 or 0.5
    vec2 cellUV = vec2(suv.x + xOffset, suv.y);

    // Local coordinates inside a brick cell [0,1)
    vec2 local = fract(cellUV);

    // Half mortar margin on each side
    float halfMortarX = mortarX * 0.5;
    float halfMortarY = mortarY * 0.5;

    // Brick mask (1.0 for brick, 0.0 for mortar)
    float insideX = step(halfMortarX, local.x) * step(local.x, 1.0 - halfMortarX);
    float insideY = step(halfMortarY, local.y) * step(local.y, 1.0 - halfMortarY);
    float brickMask = insideX * insideY;

    // Soft edges for anti-aliasing (small smoothing band)
    float edgeSmooth = 0.005;
    float leftEdge  = smoothstep(halfMortarX - edgeSmooth, halfMortarX + edgeSmooth, local.x);
    float rightEdge = 1.0 - smoothstep(1.0 - halfMortarX - edgeSmooth, 1.0 - halfMortarX + edgeSmooth, local.x);
    float bottomEdge= smoothstep(halfMortarY - edgeSmooth, halfMortarY + edgeSmooth, local.y);
    float topEdge   = 1.0 - smoothstep(1.0 - halfMortarY - edgeSmooth, 1.0 - halfMortarY + edgeSmooth, local.y);
    float smoothMask = leftEdge * rightEdge * bottomEdge * topEdge;

    float finalMask = mix(0.0, 1.0, smoothMask); // in [0,1]

    vec3 baseColor = mix(mortarColor, brickColor, finalMask);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float lambert = max(dot(N, L), 0.0);

    // Simple ambient + diffuse
    float ambient = 0.15;
    vec3 shaded = baseColor * (ambient + (1.0 - ambient) * lambert);

    outColor = vec4(shaded, 1.0);
}

To compile to SPIR‑V and get the raw hex string:

1. Install glslangValidator (part of Khronos glslang).
2. Compile to SPIR‑V binary:
   glslangValidator -V -S frag -o shader.spv shader.frag
3. Get a continuous lowercase hex string (no spaces/newlines):
   xxd -p -c 9999 shader.spv | tr -d '\n' > shader.spv.hex
   cat shader.spv.hex
