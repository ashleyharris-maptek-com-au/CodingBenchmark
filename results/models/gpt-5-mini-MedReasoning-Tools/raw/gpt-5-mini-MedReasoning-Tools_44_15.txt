#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // normalize interpolated normal
    vec3 N = normalize(normal);

    // compute light direction: support point (w==1) and directional (w==0) lights
    vec3 L = (ubo.lightPos.w == 0.0)
             ? normalize(ubo.lightPos.xyz)
             : normalize(ubo.lightPos.xyz - worldPos);

    // Gooch parameter t as specified
    float t = dot(N, L) * 0.5 + 0.5;
    t = clamp(t, 0.0, 1.0);

    // cool and warm colours
    vec3 cool = vec3(0.2, 0.2, 0.75);
    vec3 warm = vec3(0.7, 0.7, 0.4);

    // mix cool->warm by t
    vec3 colorGooch = mix(cool, warm, t);

    outColor = vec4(colorGooch, 1.0);
}
