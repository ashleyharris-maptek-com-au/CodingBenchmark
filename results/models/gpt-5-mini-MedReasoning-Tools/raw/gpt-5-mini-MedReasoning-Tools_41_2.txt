; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Names
OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %in_tangent "tangent"
OpName %in_color "color"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBOType "UBO"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4f 4

; UBO struct: mat4, mat4, mat4, vec4, vec4, vec4
%UBOType = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4f %v4f %v4f
%ptrUniform_UBO = OpTypePointer Uniform %UBOType

; Pointers for inputs/outputs
%ptrInput_v3 = OpTypePointer Input %v3f
%ptrInput_v2 = OpTypePointer Input %v2f
%ptrOutput_v4 = OpTypePointer Output %v4f

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Variables (interface and uniforms)
%in_worldPos = OpVariable %ptrInput_v3 Input
%in_normal = OpVariable %ptrInput_v3 Input
%in_uv = OpVariable %ptrInput_v2 Input
%in_tangent = OpVariable %ptrInput_v3 Input
%in_color = OpVariable %ptrInput_v3 Input

%outColor = OpVariable %ptrOutput_v4 Output

%ubo = OpVariable %ptrUniform_UBO Uniform

; Decorations: locations for inputs and output
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

OpDecorate %outColor Location 0

; UBO decorations: block and member offsets (std140 layout)
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 ColumnMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 ColumnMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 ColumnMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224

; Descriptor set and binding for ubo
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Function main
%main = OpFunction %void None %func
%main_entry = OpLabel

; Load uv, extract components, construct color, store to outColor
%uv_val = OpLoad %v2f %in_uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%frag_col = OpCompositeConstruct %v4f %uv_x %uv_y %float_0 %float_1
OpStore %outColor %frag_col

OpReturn
OpFunctionEnd
