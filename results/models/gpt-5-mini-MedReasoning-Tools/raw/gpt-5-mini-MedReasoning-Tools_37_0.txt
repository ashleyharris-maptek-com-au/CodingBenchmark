Here's a complete C# program that exactly solves the Directed Feedback Vertex Set problem. It finds a minimum-size set of vertices whose removal makes the directed graph acyclic. The solver works SCC-by-SCC (only SCCs with cycles need solving) and uses an exact exponential branching algorithm with reductions (removing vertices of zero in/out-degree) and memoization. It uses BigInteger bitmasks so it can handle SCCs of arbitrary size (within practical runtime limits).

Note: DFVS is NP-hard, so worst-case instances with large cyclic SCCs can be expensive. This solver is optimized for typical contest/test inputs and will be exact for any instance (but may be slow on very large dense cyclic SCCs).

Copy and compile with the provided .NET toolchain.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;
using System.Text;

class Program
{
    static int n, m;
    static List<int>[] g, gr;

    static void Main()
    {
        var tokens = ReadTokens();
        if (tokens.Count < 2) return;
        n = int.Parse(tokens[0]);
        m = int.Parse(tokens[1]);
        g = new List<int>[n];
        gr = new List<int>[n];
        for (int i = 0; i < n; i++) { g[i] = new List<int>(); gr[i] = new List<int>(); }
        int idx = 2;
        for (int i = 0; i < m; i++)
        {
            if (idx + 1 >= tokens.Count) break;
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            if (u >= 0 && u < n && v >= 0 && v < n)
            {
                g[u].Add(v);
                gr[v].Add(u);
            }
        }

        // Compute SCCs via Kosaraju
        var order = new List<int>();
        var used = new bool[n];
        for (int i = 0; i < n; i++) if (!used[i]) Dfs1(i, used, order);
        Array.Fill(used, false);
        var comp = new int[n];
        int compCnt = 0;
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!used[v])
            {
                var stack = new Stack<int>();
                stack.Push(v);
                used[v] = true;
                while (stack.Count > 0)
                {
                    int u = stack.Pop();
                    comp[u] = compCnt;
                    foreach (var w in gr[u])
                        if (!used[w]) { used[w] = true; stack.Push(w); }
                }
                compCnt++;
            }
        }

        // Collect vertices by component
        var compVerts = new List<int>[compCnt];
        for (int i = 0; i < compCnt; i++) compVerts[i] = new List<int>();
        for (int i = 0; i < n; i++) compVerts[comp[i]].Add(i);

        var resultSet = new List<int>();

        // Solve each component independently
        for (int c = 0; c < compCnt; c++)
        {
            var verts = compVerts[c];
            if (verts.Count == 0) continue;
            if (verts.Count == 1)
            {
                int v = verts[0];
                // single vertex, if self-loop then must remove
                if (g[v].Contains(v)) resultSet.Add(v);
                continue;
            }

            // Build local index mapping
            var map = new Dictionary<int, int>();
            for (int i = 0; i < verts.Count; i++) map[verts[i]] = i;
            int s = verts.Count;
            // Build adjacency inside component
            BigInteger[] outmask = new BigInteger[s];
            BigInteger[] inmask = new BigInteger[s];
            for (int i = 0; i < s; i++)
            {
                int u = verts[i];
                BigInteger om = BigInteger.Zero;
                foreach (var v in g[u])
                {
                    if (map.TryGetValue(v, out int vi))
                    {
                        om |= (BigInteger.One << vi);
                        inmask[vi] |= (BigInteger.One << i);
                    }
                }
                outmask[i] = om;
            }

            // If component already acyclic (shouldn't happen for SCC>1), skip
            BigInteger fullMask = (s >= 1 ? ((BigInteger.One << s) - BigInteger.One) : BigInteger.Zero);

            var solver = new SCCSolver(s, outmask, inmask, verts);
            var chosen = solver.Solve(fullMask);
            resultSet.AddRange(chosen);
        }

        // Output
        var distinct = resultSet.Distinct().ToList();
        Console.WriteLine(distinct.Count);
        if (distinct.Count > 0)
        {
            Console.WriteLine(string.Join(" ", distinct));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static void Dfs1(int v, bool[] used, List<int> order)
    {
        var stack = new Stack<(int, int)>(); // node, nextIndex
        stack.Push((v, 0));
        while (stack.Count > 0)
        {
            var top = stack.Pop();
            int node = top.Item1;
            int next = top.Item2;
            if (!used[node])
            {
                used[node] = true;
                stack.Push((node, 1)); // marker to push to order after processing neighbors
                foreach (var to in g[node])
                {
                    if (!used[to]) stack.Push((to, 0));
                }
            }
            else if (next == 1)
            {
                // second visit: add to order
                order.Add(node);
            }
        }
    }

    static List<string> ReadTokens()
    {
        var sb = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            if (line.Length == 0) continue;
            var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            foreach (var p in parts) sb.Add(p);
        }
        return sb;
    }

    class SCCSolver
    {
        int s;
        BigInteger[] outmask, inmask;
        List<int> globalVerts; // mapping local index -> original vertex id
        Dictionary<BigInteger, int> memo = new Dictionary<BigInteger, int>();
        Dictionary<BigInteger, int> choice = new Dictionary<BigInteger, int>();

        public SCCSolver(int s, BigInteger[] outmask, BigInteger[] inmask, List<int> globalVerts)
        {
            this.s = s;
            this.outmask = outmask;
            this.inmask = inmask;
            this.globalVerts = globalVerts;
        }

        public List<int> Solve(BigInteger startMask)
        {
            int best = SolveMask(startMask);
            // reconstruct chosen vertices
            var res = new List<int>();
            var mask = Reduce(startMask);
            while (mask != BigInteger.Zero)
            {
                if (!choice.TryGetValue(mask, out int ch) || ch < 0) break;
                // add global vertex id
                res.Add(globalVerts[ch]);
                mask &= ~(BigInteger.One << ch);
                mask = Reduce(mask);
            }
            return res;
        }

        int SolveMask(BigInteger mask)
        {
            mask = Reduce(mask);
            if (mask == BigInteger.Zero) return 0;
            if (memo.TryGetValue(mask, out int val)) return val;

            // find a directed cycle in mask
            var cycle = FindCycle(mask);
            if (cycle == null || cycle.Count == 0)
            {
                memo[mask] = 0;
                choice[mask] = -1;
                return 0;
            }

            int best = int.MaxValue;
            int bestChoice = -1;
            // Branch: remove one vertex from the cycle (one of the vertices must be removed)
            foreach (int v in cycle)
            {
                var newMask = mask & ~(BigInteger.One << v);
                int rec = SolveMask(newMask);
                if (rec != int.MaxValue && rec + 1 < best)
                {
                    best = rec + 1;
                    bestChoice = v;
                    if (best == 1) break; // cannot do better than 1
                }
            }
            memo[mask] = best;
            choice[mask] = bestChoice;
            return best;
        }

        BigInteger Reduce(BigInteger mask)
        {
            bool changed = true;
            while (changed)
            {
                changed = false;
                // find vertices with zero in-degree or zero out-degree inside mask
                for (int v = 0; v < s; v++)
                {
                    if (((mask >> v) & BigInteger.One) == BigInteger.Zero) continue;
                    bool hasIn = ((inmask[v] & mask) != BigInteger.Zero);
                    bool hasOut = ((outmask[v] & mask) != BigInteger.Zero);
                    if (!hasIn || !hasOut)
                    {
                        mask &= ~(BigInteger.One << v);
                        changed = true;
                    }
                }
            }
            return mask;
        }

        List<int> FindCycle(BigInteger mask)
        {
            int[] color = new int[s]; // 0 un, 1 visiting, 2 done
            int[] parent = new int[s];
            for (int i = 0; i < s; i++) parent[i] = -1;

            for (int start = 0; start < s; start++)
            {
                if (((mask >> start) & BigInteger.One) == BigInteger.Zero) continue;
                if (color[start] != 0) continue;
                var stack = new Stack<int>();
                var iter = new Stack<int>(); // next neighbor index to explore
                stack.Push(start);
                iter.Push(0);

                while (stack.Count > 0)
                {
                    int v = stack.Peek();
                    int ni = iter.Pop();
                    if (ni == 0 && color[v] == 0)
                    {
                        color[v] = 1;
                    }

                    bool pushed = false;
                    for (int j = ni; j < s; j++)
                    {
                        if (((mask >> j) & BigInteger.One) == BigInteger.Zero) continue;
                        if (((outmask[v] >> j) & BigInteger.One) == BigInteger.Zero) continue; // no edge v->j
                        if (color[j] == 0)
                        {
                            parent[j] = v;
                            // save next index for v
                            iter.Push(j + 1);
                            stack.Push(j);
                            iter.Push(0);
                            pushed = true;
                            break;
                        }
                        else if (color[j] == 1)
                        {
                            // found cycle: reconstruct from j to v
                            var cycle = new List<int>();
                            cycle.Add(j);
                            int cur = v;
                            while (cur != j && cur != -1)
                            {
                                cycle.Add(cur);
                                cur = parent[cur];
                            }
                            // cycle now contains vertices in cycle order
                            return cycle;
                        }
                    }
                    if (!pushed)
                    {
                        // finished v
                        stack.Pop();
                        color[v] = 2;
                    }
                }
            }
            return null;
        }
    }
}
```

Notes and explanation (brief):
- The program first decomposes the graph into strongly connected components (Kosaraju). Only SCCs with possible cycles need solving.
- For each SCC, it builds local bitmasks and runs an exact solver:
  - Repeatedly eliminates vertices that cannot belong to any cycle (in-degree 0 or out-degree 0 inside the current mask).
  - Finds one directed cycle (by DFS) in the remaining "core".
  - Branches: remove each vertex of that cycle (exact branching) and recurse, memoizing results.
- The solution reconstructs the removed vertices per SCC and prints the total set.
- The approach is exact (returns minimum) and practical for typical contest inputs. Because DFVS is NP-hard, very large dense cyclic SCCs may be slow.
