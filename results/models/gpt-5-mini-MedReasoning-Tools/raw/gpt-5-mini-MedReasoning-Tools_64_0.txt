/**
 * Drone grid path planner - TypeScript (Node.js)
 *
 * Reads one JSON document from stdin with fields:
 *  - probability: number[][]   (-1 = obstacle, >0 = search value)
 *  - drones: {id, x, y}[]
 *  - maxStepsPerDrone: number
 *
 * Writes one JSON document to stdout:
 *  { "paths": [ [[x,y], ...], ... ] }
 *
 * Behavior:
 *  - Each drone path starts at the given (x,y) (if that cell is an obstacle,
 *    the nearest free cell is chosen).
 *  - Each step a drone may move up/down/left/right or stay in place.
 *  - Paths are length = maxStepsPerDrone+1 (including the start cell).
 *  - Drones avoid obstacles and grid boundaries.
 *  - The planner uses a fast greedy heuristic:
 *      * At each time step, choose among neighbors (and stay) the cell with
 *        highest remaining probability (shared among drones to encourage unique coverage).
 *      * If all local choices have zero remaining value, BFS is used to find the
 *        nearest cell with remaining value > 0 and the drone takes one step towards it.
 *      * If no remaining valued cell exists, the drone stays put.
 *
 * This implementation is lightweight and scales: it avoids external deps,
 * uses O(drones * steps) local decisions and occasional BFS searches.
 *
 * Conforms to the input-reading pattern required by the task.
 */

declare const require: any;
const fs = require('fs');

type DroneInput = { id: any; x: number; y: number };
type Input = {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as Input;

const grid = data.probability;
const dronesIn = data.drones || [];
const maxSteps = Math.max(0, Math.floor(data.maxStepsPerDrone || 0));

const rows = grid.length;
const cols = rows > 0 ? grid[0].length : 0;

function inBounds(x: number, y: number) {
  return x >= 0 && x < cols && y >= 0 && y < rows;
}

function isFree(x: number, y: number) {
  return inBounds(x, y) && grid[y][x] !== -1;
}

// clone remaining values so we can zero out visited cells (shared across drones)
const remaining: number[][] = new Array(rows);
for (let y = 0; y < rows; y++) {
  remaining[y] = new Array(cols);
  for (let x = 0; x < cols; x++) {
    const v = grid[y][x];
    remaining[y][x] = (typeof v === 'number' && v > 0) ? v : 0;
  }
}

// visited boolean grid (for uniqueness)
const visited: boolean[][] = new Array(rows);
for (let y = 0; y < rows; y++) {
  visited[y] = new Array(cols).fill(false);
}

// Utility: find nearest free cell to (sx,sy) using BFS (4-neigh)
function findNearestFree(sx: number, sy: number): [number, number] | null {
  if (isFree(sx, sy)) return [sx, sy];
  const q: [number, number][] = [];
  const seen: boolean[][] = new Array(rows);
  for (let y = 0; y < rows; y++) {
    seen[y] = new Array(cols).fill(false);
  }
  if (!inBounds(sx, sy)) return null;
  q.push([sx, sy]);
  seen[sy][sx] = true;
  let qi = 0;
  while (qi < q.length) {
    const [x, y] = q[qi++];
    const neigh = [
      [x + 1, y],
      [x - 1, y],
      [x, y + 1],
      [x, y - 1],
    ];
    for (const [nx, ny] of neigh) {
      if (!inBounds(nx, ny) || seen[ny][nx]) continue;
      seen[ny][nx] = true;
      if (isFree(nx, ny)) return [nx, ny];
      q.push([nx, ny]);
    }
  }
  return null;
}

// BFS to nearest cell with remaining value > 0 (not necessarily unvisited? we zeroed visited)
function bfsNearestValued(sx: number, sy: number): { target: [number, number]; parent: Map<string, string> } | null {
  if (!inBounds(sx, sy) || !isFree(sx, sy)) return null;
  if (remaining[sy][sx] > 0) {
    // already at a valued cell
    return { target: [sx, sy], parent: new Map([[`${sx},${sy}`, '']]) };
  }
  const q: [number, number][] = [];
  const seen: boolean[][] = new Array(rows);
  for (let y = 0; y < rows; y++) seen[y] = new Array(cols).fill(false);
  const parent = new Map<string, string>();
  q.push([sx, sy]);
  seen[sy][sx] = true;
  parent.set(`${sx},${sy}`, '');
  let qi = 0;
  while (qi < q.length) {
    const [x, y] = q[qi++];
    const neigh = [
      [x + 1, y],
      [x - 1, y],
      [x, y + 1],
      [x, y - 1],
    ];
    for (const [nx, ny] of neigh) {
      if (!inBounds(nx, ny) || seen[ny][nx]) continue;
      if (!isFree(nx, ny)) {
        seen[ny][nx] = true;
        continue;
      }
      seen[ny][nx] = true;
      parent.set(`${nx},${ny}`, `${x},${y}`);
      if (remaining[ny][nx] > 0) {
        return { target: [nx, ny], parent };
      }
      q.push([nx, ny]);
    }
  }
  return null;
}

// reconstruct path from parent map: from target back to start
function reconstructPath(parent: Map<string, string>, startKey: string, targetKey: string): [number, number][] {
  const path: [number, number][] = [];
  let cur = targetKey;
  while (cur !== '' && cur !== undefined) {
    const [xs, ys] = cur.split(',').map(s => parseInt(s, 10));
    path.push([xs, ys]);
    const p = parent.get(cur);
    if (!p) break;
    cur = p;
  }
  path.reverse();
  // Ensure first is start. If not, attempt to prepend start.
  if (path.length === 0) return [];
  return path;
}

// Choose next cell for a drone at (x,y) using local greedy + small neighborhood tie-breaker.
// prefer unvisited/remaining high value. Avoid immediate back-and-forth by optionally favoring not previous cell.
function chooseLocalBest(x: number, y: number, prev: [number, number] | null): [number, number] {
  const candidates: [number, number][] = [
    [x, y], // stay
    [x + 1, y],
    [x - 1, y],
    [x, y + 1],
    [x, y - 1],
  ];
  let best: [number, number] = [x, y];
  let bestScore = -Infinity;
  for (const [nx, ny] of candidates) {
    if (!inBounds(nx, ny)) continue;
    if (!isFree(nx, ny)) continue;
    // score: remaining value at cell + 0.1 * sum of adjacent remaining values
    let score = remaining[ny][nx];
    // small neighborhood bonus
    const neigh = [
      [nx + 1, ny],
      [nx - 1, ny],
      [nx, ny + 1],
      [nx, ny - 1],
    ];
    let sumAdj = 0;
    for (const [ax, ay] of neigh) {
      if (inBounds(ax, ay) && isFree(ax, ay)) sumAdj += remaining[ay][ax];
    }
    score += 0.1 * sumAdj;
    // penalize going back to previous location to avoid oscillation
    if (prev && nx === prev[0] && ny === prev[1]) score -= 1e-6;
    if (score > bestScore) {
      bestScore = score;
      best = [nx, ny];
    }
  }
  return best;
}

// Plan paths
const paths: [number, number][][] = [];
const dronePositions: [number, number][] = [];
const dronePrev: ( [number, number] | null )[] = []; // previous position to discourage oscillation

// initialize drones: if starting cell is obstacle or out of bounds, move to nearest free cell
for (let i = 0; i < dronesIn.length; i++) {
  const d = dronesIn[i];
  let sx = Math.floor(d.x);
  let sy = Math.floor(d.y);
  if (!inBounds(sx, sy) || !isFree(sx, sy)) {
    const nearest = findNearestFree(sx, sy);
    if (nearest) {
      sx = nearest[0];
      sy = nearest[1];
    } else {
      // no free cell in grid - place at 0,0 if possible else skip
      sx = 0;
      sy = 0;
      if (!inBounds(sx, sy)) sx = 0, sy = 0;
    }
  }
  dronePositions.push([sx, sy]);
  dronePrev.push(null);
  // start path with starting cell
  paths.push([[sx, sy]]);
  // mark visited/covered
  if (!visited[sy][sx]) {
    visited[sy][sx] = true;
    remaining[sy][sx] = 0;
  }
}

// For each time step, pick move for each drone greedily
for (let step = 1; step <= maxSteps; step++) {
  for (let i = 0; i < dronePositions.length; i++) {
    const [cx, cy] = dronePositions[i];
    const prev = dronePrev[i];
    // choose local best
    let [nx, ny] = chooseLocalBest(cx, cy, prev);

    // if local best gives zero remaining and there's no valuable neighbor, try BFS to nearest valued cell
    if (remaining[ny][nx] <= 0) {
      const bfsRes = bfsNearestValued(cx, cy);
      if (bfsRes) {
        const startKey = `${cx},${cy}`;
        const targetKey = `${bfsRes.target[0]},${bfsRes.target[1]}`;
        const fullPath = reconstructPath(bfsRes.parent, startKey, targetKey);
        if (fullPath.length >= 2) {
          // take one step along path
          const next = fullPath[1];
          nx = next[0];
          ny = next[1];
        } else {
          // we are at target but it had zero? fallback to stay
          nx = cx;
          ny = cy;
        }
      } else {
        // no valued cell left, stay put (or we could wander; staying is simpler)
        nx = cx;
        ny = cy;
      }
    }

    // As a final safety, ensure chosen cell is free and in bounds; otherwise stay
    if (!inBounds(nx, ny) || !isFree(nx, ny)) {
      nx = cx;
      ny = cy;
    }

    // update drone position, path, visited and remaining
    dronePrev[i] = [cx, cy];
    dronePositions[i] = [nx, ny];
    paths[i].push([nx, ny]);
    if (!visited[ny][nx]) {
      visited[ny][nx] = true;
      remaining[ny][nx] = 0;
    }
  }
}

// Output JSON as required, no extra logs
const out = { paths: paths };
console.log(JSON.stringify(out));
