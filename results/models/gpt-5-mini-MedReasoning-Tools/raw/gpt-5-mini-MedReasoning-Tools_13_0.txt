Here's a single-file, complete C++20 program that adaptively solves the Longest Common Substring problem for the ranges you specified. It uses a binary-search-over-length strategy with a very-low-collision double 64-bit rolling-hash (two independent 64-bit polynomial hashes). The implementation is optimized for large inputs (memory reservations, sliding-window hashes, ordering strings by length to reduce intermediate set sizes). It reads strings line-by-line (including empty strings) and prints one longest common substring (empty line if none).

Compile with MSVC (/std:c++20) and run; it should handle trivial through extreme sizes within the resources you specified.

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <random>
#include <chrono>
#include <unordered_set>
#include <unordered_map>
#include <cstdint>

using namespace std;

// Pair key for double 64-bit hash
struct HKey {
    uint64_t a, b;
    bool operator==(HKey const& o) const noexcept { return a == o.a && b == o.b; }
};

// splitmix64 for hashing combination and unordered_map hashing
static inline uint64_t splitmix64(uint64_t x) noexcept {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct HKeyHash {
    size_t operator()(HKey const& k) const noexcept {
        // combine a and b with splitmix
        uint64_t z = k.a + 0x9e3779b97f4a7c15ULL;
        z = splitmix64(z) ^ (k.b + 0x9e3779b97f4a7c15ULL);
        return (size_t)z;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline

    vector<string> s;
    s.reserve(N);
    for (int i = 0; i < N; ++i) {
        string line;
        getline(cin, line);
        s.push_back(std::move(line));
    }

    if (N == 0) {
        cout << '\n';
        return 0;
    }

    // If any string is empty, longest common substring is empty
    for (int i = 0; i < N; ++i) {
        if (s[i].empty()) {
            cout << '\n';
            return 0;
        }
    }

    // sort indices by string length ascending (we will use shortest first to reduce set sizes)
    vector<int> idx(N);
    for (int i = 0; i < N; ++i) idx[i] = i;
    sort(idx.begin(), idx.end(), [&](int a, int b){ return s[a].size() < s[b].size(); });

    size_t min_len = s[idx[0]].size();
    // Precompute powers up to min_len for two bases (uint64 overflow arithmetic)
    // Choose random odd bases > 256 to reduce collisions
    std::mt19937_64 rng((unsigned)chrono::high_resolution_clock::now().time_since_epoch().count());
    uint64_t base1 = (uint64_t)(rng() | 1ULL) % 1000000007ULL;
    if (base1 < 131) base1 += 131;
    if ((base1 & 1) == 0) base1 |= 1;

    uint64_t base2 = (uint64_t)(rng() | 1ULL) % 1000000009ULL;
    if (base2 < 257) base2 += 257;
    if ((base2 & 1) == 0) base2 |= 1;

    // Ensure bases are distinct
    if (base1 == base2) base2 += 2;

    vector<uint64_t> pow1(min_len + 1), pow2(min_len + 1);
    pow1[0] = 1;
    pow2[0] = 1;
    for (size_t i = 1; i <= min_len; ++i) {
        pow1[i] = pow1[i-1] * base1;
        pow2[i] = pow2[i-1] * base2;
    }

    // Helper lambda: given length L, test if there's a common substring of length L.
    // If found, returns true and fills found_pos_in_shortest and found_key for retrieving substring.
    auto has_common_of_length = [&](size_t L, HKey &found_key, size_t &found_pos_in_shortest)->bool {
        if (L == 0) {
            found_pos_in_shortest = 0;
            found_key = {0,0};
            return true;
        }
        int first_idx = idx[0];
        const string &first = s[first_idx];
        size_t n0 = first.size();
        if (L > n0) return false;

        // Map from hash -> first occurrence position (in shortest string)
        unordered_map<HKey, size_t, HKeyHash> posMap;
        posMap.reserve((n0 - L + 1) * 2 + 10);

        // compute rolling hash for first string
        uint64_t h1 = 0, h2 = 0;
        for (size_t i = 0; i < L; ++i) {
            unsigned char c = (unsigned char)first[i];
            h1 = h1 * base1 + (uint64_t)(c + 1);
            h2 = h2 * base2 + (uint64_t)(c + 1);
        }
        posMap[{h1,h2}] = 0;
        for (size_t i = L; i < n0; ++i) {
            unsigned char outc = (unsigned char)first[i - L];
            unsigned char inc = (unsigned char)first[i];
            h1 = h1 * base1 - pow1[L] * (uint64_t)(outc + 1) + (uint64_t)(inc + 1);
            h2 = h2 * base2 - pow2[L] * (uint64_t)(outc + 1) + (uint64_t)(inc + 1);
            HKey k{h1,h2};
            // store only first occurrence (that's enough to retrieve substring later)
            if (posMap.find(k) == posMap.end()) posMap[k] = i - L + 1;
        }

        // current set of candidate hashes (start as keys of posMap)
        unordered_set<HKey, HKeyHash> curSet;
        curSet.reserve(posMap.size() * 2 + 10);
        for (auto &p : posMap) curSet.insert(p.first);

        // For each other string (ordered by increasing length), filter curSet
        for (int order = 1; order < N; ++order) {
            int sid = idx[order];
            const string &str = s[sid];
            size_t n = str.size();
            if (L > n) return false; // impossible if any string shorter than L

            unordered_set<HKey, HKeyHash> nextSet;
            nextSet.reserve(min(curSet.size(), (size_t)(n - L + 1)) * 2 + 10);

            uint64_t rh1 = 0, rh2 = 0;
            // sliding window
            for (size_t i = 0; i < L; ++i) {
                unsigned char c = (unsigned char)str[i];
                rh1 = rh1 * base1 + (uint64_t)(c + 1);
                rh2 = rh2 * base2 + (uint64_t)(c + 1);
            }
            HKey k{rh1,rh2};
            if (curSet.find(k) != curSet.end()) nextSet.insert(k);
            for (size_t i = L; i < n; ++i) {
                unsigned char outc = (unsigned char)str[i - L];
                unsigned char inc = (unsigned char)str[i];
                rh1 = rh1 * base1 - pow1[L] * (uint64_t)(outc + 1) + (uint64_t)(inc + 1);
                rh2 = rh2 * base2 - pow2[L] * (uint64_t)(outc + 1) + (uint64_t)(inc + 1);
                HKey kk{rh1,rh2};
                if (curSet.find(kk) != curSet.end()) nextSet.insert(kk);
            }

            // swap
            curSet.swap(nextSet);
            if (curSet.empty()) return false; // no common substring of length L
        }

        // If here, curSet contains hashes common to all strings. Pick one and return its pos from posMap.
        auto it = curSet.begin();
        found_key = *it;
        auto itpos = posMap.find(found_key);
        if (itpos == posMap.end()) {
            // extremely unlikely, but means the hash was seen in all other strings but not in first
            // (could happen due to hash collision). Return false to be safe.
            return false;
        }
        found_pos_in_shortest = itpos->second;
        return true;
    };

    // Binary search over length
    size_t lo = 0, hi = min_len;
    HKey bestKey{0,0};
    size_t bestPos = 0;
    while (lo <= hi) {
        size_t mid = (lo + hi) >> 1;
        HKey fk; size_t fpos;
        bool ok = has_common_of_length(mid, fk, fpos);
        if (ok) {
            // mid possible, try longer
            bestKey = fk;
            bestPos = fpos;
            lo = mid + 1;
        } else {
            if (mid == 0) break;
            hi = mid - 1;
        }
    }

    size_t resultLen = (lo == 0 ? 0 : lo - 1);
    if (resultLen == 0) {
        cout << '\n';
        return 0;
    } else {
        // retrieve substring from shortest string using bestPos and resultLen
        const string &shortest = s[idx[0]];
        if (bestPos + resultLen <= shortest.size()) {
            cout << shortest.substr(bestPos, resultLen) << '\n';
            return 0;
        } else {
            // Fallback: compute again for length resultLen and print first found
            HKey fk; size_t fpos;
            if (has_common_of_length(resultLen, fk, fpos)) {
                cout << shortest.substr(fpos, resultLen) << '\n';
            } else {
                cout << '\n';
            }
            return 0;
        }
    }
}
```

Notes and design choices
- Uses double 64-bit polynomial rolling-hash (two independent bases) to make collision probability astronomically small in practice.
- Orders strings by length ascending and always builds the initial hash-map from the shortest string to dramatically reduce intermediate set sizes.
- Uses sliding-window updates (no full prefix arrays) and reserves bucket sizes to reduce rehashing overhead.
- Binary-search over substring length reduces number of expensive scans to O(log(min_length)).
- Works for arbitrary bytes (reads whole lines), including spaces and empty strings.
- If you require 100% collision-free (mathematically guaranteed) behavior, replace the hashing approach with a suffix automaton / suffix array based algorithm — but for very large inputs (dozens of MBs per string) those can be heavier in memory/time. The double 64-bit hash with two independent bases is practical and extremely reliable for real-world use.
