struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (point light)
    float3 L = lightPos.xyz - input.worldPos;
    float3 Ldir = normalize(L);

    // View direction
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Ambient term
    float3 ambient = float3(0.1, 0.05, 0.05);

    // Diffuse term using vertex color
    float NdotL = max(dot(N, Ldir), 0.0);
    float3 diffuse = input.color * NdotL;

    // Specular term (white specular, shininess 32)
    float3 R = reflect(-Ldir, N);
    float specFactor = pow(max(dot(R, V), 0.0), 32.0);
    float3 specular = float3(1.0, 1.0, 1.0) * specFactor;

    // Combine and clamp
    float3 color = ambient + diffuse + specular;
    color = clamp(color, 0.0, 1.0);

    return float4(color, 1.0);
}
