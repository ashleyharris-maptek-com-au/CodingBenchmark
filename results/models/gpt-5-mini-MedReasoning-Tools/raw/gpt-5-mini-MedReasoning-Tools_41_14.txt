; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %ubo %outColor
OpExecutionMode %main OriginUpperLeft

; Imports
%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%_struct_UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
OpDecorate %_struct_UBO Block
OpMemberDecorate %_struct_UBO 0 Offset 0
OpMemberDecorate %_struct_UBO 1 Offset 64
OpMemberDecorate %_struct_UBO 2 Offset 128
OpMemberDecorate %_struct_UBO 3 Offset 192
OpMemberDecorate %_struct_UBO 4 Offset 208
OpMemberDecorate %_struct_UBO 5 Offset 224

%ptrUniform_UBO = OpTypePointer Uniform %_struct_UBO
%ptrUniform_v4 = OpTypePointer Uniform %v4f

%ptrInput_v3 = OpTypePointer Input %v3f
%ptrInput_v2 = OpTypePointer Input %v2f
%ptrOutput_v4 = OpTypePointer Output %v4f

; Variables (inputs/outputs/uniform)
%ubo = OpVariable %ptrUniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%inWorldPos = OpVariable %ptrInput_v3 Input
OpDecorate %inWorldPos Location 0
%inNormal = OpVariable %ptrInput_v3 Input
OpDecorate %inNormal Location 1
%inUV = OpVariable %ptrInput_v2 Input
OpDecorate %inUV Location 2
%inTangent = OpVariable %ptrInput_v3 Input
OpDecorate %inTangent Location 3
%inColor = OpVariable %ptrInput_v3 Input
OpDecorate %inColor Location 4

%outColor = OpVariable %ptrOutput_v4 Output
OpDecorate %outColor Location 0

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%f_0 = OpConstant %float 0.0
%f_0_2 = OpConstant %float 0.2
%f_0_3 = OpConstant %float 0.3
%f_0_7 = OpConstant %float 0.7
%f_0_8 = OpConstant %float 0.8
%f_1 = OpConstant %float 1.0
%f_1_5 = OpConstant %float 1.5
%f_3 = OpConstant %float 3.0

; Composite constants
%baseColor = OpConstantComposite %v3f %f_0_8 %f_0_2 %f_0_2
%fogColor = OpConstantComposite %v3f %f_0_7 %f_0_7 %f_0_8
%vec_zero3 = OpConstantComposite %v3f %f_0 %f_0 %f_0

; Function
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %inWorldPos
%n = OpLoad %v3f %inNormal

; Load cameraPos (member 4) and lightPos (member 3) from UBO
%ptr_cam4 = OpAccessChain %ptrUniform_v4 %ubo %int_4
%cam4 = OpLoad %v4f %ptr_cam4
%cam3 = OpVectorShuffle %v3f %cam4 %cam4 0 1 2

%ptr_light4 = OpAccessChain %ptrUniform_v4 %ubo %int_3
%light4 = OpLoad %v4f %ptr_light4
%light3 = OpVectorShuffle %v3f %light4 %light4 0 1 2

; Compute distance from camera to fragment
%deltaCam = OpFSub %v3f %cam3 %wp
%dist = OpExtInst %float %GLSL Length %deltaCam

; fog factor t = clamp((dist - 1.5) / 3.0, 0.0, 1.0)
%sub_dist = OpFSub %float %dist %f_1_5
%div_dist = OpFDiv %float %sub_dist %f_3
%clamped_t = OpExtInst %float %GLSL Clamp %div_dist %f_0 %f_1

; Compute Lambert diffuse: normal and light direction
%ln = OpFSub %v3f %light3 %wp
%nnorm = OpExtInst %v3f %GLSL Normalize %n
%lnorm = OpExtInst %v3f %GLSL Normalize %ln

%dotNL = OpDot %float %nnorm %lnorm
%diff = OpFMax %float %dotNL %f_0

; lambertFactor = 0.3 + 0.7 * diff
%mul07 = OpFMul %float %f_0_7 %diff
%lambert = OpFAdd %float %f_0_3 %mul07

; colorLit = baseColor * lambert
%lambertVec = OpCompositeConstruct %v3f %lambert %lambert %lambert
%colorLit = OpFMul %v3f %baseColor %lambertVec

; finalColor = colorLit * (1 - t) + fogColor * t
%one_minus_t = OpFSub %float %f_1 %clamped_t
%one_minus_t_vec = OpCompositeConstruct %v3f %one_minus_t %one_minus_t %one_minus_t
%t_vec = OpCompositeConstruct %v3f %clamped_t %clamped_t %clamped_t

%term1 = OpFMul %v3f %colorLit %one_minus_t_vec
%term2 = OpFMul %v3f %fogColor %t_vec
%final_rgb = OpFAdd %v3f %term1 %term2

; assemble vec4 with alpha = 1.0
%final_rgba = OpCompositeConstruct %v4f %final_rgb %f_1

; Store to output
OpStore %outColor %final_rgba

OpReturn
OpFunctionEnd
