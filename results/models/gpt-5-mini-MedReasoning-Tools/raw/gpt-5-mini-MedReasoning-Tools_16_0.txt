Below is a single-file, complete C++20 program that reads the job-shop instance from stdin and writes a feasible schedule to stdout. It adapts automatically to problem size:

- For very large instances it runs one fast list-scheduling pass (O(T log J) time, O(T) memory).
- For small/medium instances it runs multiple variations of list-scheduling (different tie-breaking heuristics, randomized tie-breaks) within a short time limit to improve makespan and returns the best found schedule.

The produced schedule always respects:
1) each task runs on its specified machine,
2) tasks in a job are executed in order,
3) machines run at most one task at a time,
4) tasks are non-preemptive.

Compile with MSVC (cl) /std:c++20.

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Task {
    int machine;
    ll dur;
};

struct HeapEntry {
    ll est;           // predicted earliest start
    ll tie;           // tie-break value
    int job;
    bool operator<(HeapEntry const& o) const {
        // priority_queue is max-heap by default, so invert comparison
        if (est != o.est) return est > o.est;
        if (tie != o.tie) return tie > o.tie;
        return job > o.job;
    }
};

struct ScheduleResult {
    vector<vector<ll>> starts; // starts[j][t]
    ll makespan;
};

ScheduleResult schedule_with_rule(
    const vector<vector<Task>>& jobs,
    int M,
    int tie_rule,
    mt19937_64 &rng)
{
    int J = (int)jobs.size();
    vector<int> idx(J, 0);            // next task index for each job
    vector<ll> jobReady(J, 0);        // when job is ready for next task
    vector<ll> machineAvail(M, 0);    // machine availability
    vector<ll> remaining(J, 0);       // remaining processing time per job

    int totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        ll s = 0;
        for (auto &tk : jobs[j]) s += tk.dur;
        remaining[j] = s;
        totalTasks += (int)jobs[j].size();
    }

    vector<vector<ll>> starts(J);
    for (int j = 0; j < J; ++j) starts[j].assign(jobs[j].size(), 0);

    priority_queue<HeapEntry> pq;

    auto compute_tie = [&](int j, int dur)->ll {
        switch (tie_rule) {
            case 0: // default: break ties by job id (stable)
                return j;
            case 1: // SPT (shortest processing time)
                return dur;
            case 2: // LPT (longest processing time) -> put negative so smaller is preferred
                return -dur;
            case 3: // LRPT (longest remaining processing time)
                return -remaining[j];
            case 4: // random tie-break
                return (ll)(rng() & 0x7fffffff);
            default:
                return j;
        }
    };

    // initialize heap with first operation of each job
    for (int j = 0; j < J; ++j) {
        if (jobs[j].empty()) continue;
        int m = jobs[j][0].machine;
        ll dur = jobs[j][0].dur;
        ll est = max(jobReady[j], machineAvail[m]);
        ll tie = compute_tie(j, (int)dur);
        pq.push({est, tie, j});
    }

    int scheduled = 0;
    while (!pq.empty()) {
        HeapEntry he = pq.top(); pq.pop();
        int j = he.job;
        if (idx[j] >= (int)jobs[j].size()) continue; // stale
        Task op = jobs[j][idx[j]];
        int m = op.machine;
        ll p = op.dur;

        // recompute actual earliest start
        ll est_actual = max(jobReady[j], machineAvail[m]);
        if (est_actual != he.est) {
            // reinsert with updated est and recomputed tie (remaining may have changed)
            ll tie = compute_tie(j, (int)p);
            pq.push({est_actual, tie, j});
            continue;
        }

        // schedule this operation at est_actual
        starts[j][idx[j]] = est_actual;
        ll finish = est_actual + p;
        jobReady[j] = finish;
        machineAvail[m] = finish;
        remaining[j] -= p;
        idx[j]++;
        scheduled++;

        // push next operation of this job (if any)
        if (idx[j] < (int)jobs[j].size()) {
            int m2 = jobs[j][idx[j]].machine;
            ll p2 = jobs[j][idx[j]].dur;
            ll est2 = max(jobReady[j], machineAvail[m2]);
            ll tie2 = compute_tie(j, (int)p2);
            pq.push({est2, tie2, j});
        }
    }

    ll makespan = 0;
    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            int last = (int)jobs[j].size() - 1;
            makespan = max(makespan, starts[j][last] + jobs[j][last].dur);
        }
    }

    return {starts, makespan};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
#if defined(_MSC_VER)
    // no special MSVC handling needed
#endif

    int J, M;
    if (!(cin >> J >> M)) return 0;
    vector<vector<Task>> jobs;
    jobs.resize(J);
    long long totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int m;
            long long d;
            cin >> m >> d;
            jobs[j][t].machine = m;
            jobs[j][t].dur = d;
            totalTasks++;
        }
    }

    // RNG
    mt19937_64 rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count() ^ (uint64_t)totalTasks);

    // Decide strategy based on totalTasks
    // small: many iterations & tie rules; medium: fewer; large: single pass
    int maxIterations = 1;
    double timeLimitSeconds = 0.0;
    if (totalTasks <= 2000) {
        maxIterations = 400;           // lots of variants for small instances
        timeLimitSeconds = 1.0;        // cap at ~1 second
    } else if (totalTasks <= 10000) {
        maxIterations = 60;
        timeLimitSeconds = 0.7;
    } else if (totalTasks <= 50000) {
        maxIterations = 8;
        timeLimitSeconds = 0.5;
    } else {
        maxIterations = 1;             // huge instances -> one fast pass
        timeLimitSeconds = 0.0;
    }

    // iterate over tie rules and randomized variations
    vector<int> tie_rules_to_try;
    // always include default EST
    tie_rules_to_try.push_back(0);
    // add some deterministic heuristics
    tie_rules_to_try.push_back(1); // SPT
    tie_rules_to_try.push_back(2); // LPT
    tie_rules_to_try.push_back(3); // LRPT
    tie_rules_to_try.push_back(4); // random

    ScheduleResult bestRes;
    bestRes.makespan = LLONG_MAX;

    auto tstart = chrono::steady_clock::now();
    int iterations = 0;
    // For reproducible coverage, we cycle through tie_rules and random seeds
    while (iterations < maxIterations) {
        for (int tr : tie_rules_to_try) {
            if (iterations >= maxIterations) break;
            if (timeLimitSeconds > 0.0) {
                auto now = chrono::steady_clock::now();
                double elapsed = chrono::duration<double>(now - tstart).count();
                if (elapsed > timeLimitSeconds) goto done_iter;
            }
            // vary RNG seed for randomized tie breaks occasionally
            if (tr == 4) {
                // reseed for random tiebreak each run
                rng.seed(rng() ^ (uint64_t)(iterations + 0x9e3779b97f4a7c15ULL));
            }
            ScheduleResult res = schedule_with_rule(jobs, M, tr, rng);
            if (res.makespan < bestRes.makespan) {
                bestRes = std::move(res);
            }
            iterations++;
        }
    }
done_iter:

    // If no schedule found (edge cases), run one default scheduling
    if (bestRes.makespan == LLONG_MAX) {
        mt19937_64 rng2(123456);
        bestRes = schedule_with_rule(jobs, M, 0, rng2);
    }

    // Output: for each job J lines: T pairs per line: start_time machine start_time machine ...
    // Output uses job order and tasks in order.
    // Use ll for times.
    for (int j = 0; j < J; ++j) {
        const auto &st = bestRes.starts[j];
        for (int t = 0; t < (int)jobs[j].size(); ++t) {
            if (t) cout << ' ';
            cout << st[t] << ' ' << jobs[j][t].machine;
        }
        cout << '\n';
    }

    return 0;
}
```

Notes and rationale:
- The core scheduling primitive is a very fast list-scheduler that always produces a feasible schedule. It uses a heap keyed by earliest feasible start time (est). When an entry is popped we recompute the actual est and reinsert if it changed — this keeps correctness while remaining efficient.
- Multiple tie-breaking heuristics (SPT, LPT, LRPT, random) are tried for small/medium instances to improve solution quality. This is light-weight and scales to large instances because the default single-pass option stays O(T log J).
- The solver is deterministic for deterministic tie rules and uses RNG only for the randomized tie break runs. Small instances get more iterations/time for quality; large instances use a single fast pass.
- All arithmetic for times uses 64-bit integers (long long) to avoid overflow on long schedules.
