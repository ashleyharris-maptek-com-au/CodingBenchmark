#version 450

// 3D Boids single-timestep compute shader (one step per dispatch).
// - Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// - Boid i is at inData[i*2], inData[i*2+1]
// - This shader performs one Euler integration step using Reynolds rules:
//   separation, alignment, cohesion + simple boundary force.
// - It reads uniforms from Params. See interface in the prompt.
//
// Notes:
// - Guard against gl_GlobalInvocationID.x >= N
// - Skip self and zero-distance neighbors
// - Average contributions by neighbor count (avoid divide-by-zero)
// - Clamp velocity to maxSpeed

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    // read current boid
    uint readIdx = i * 2u;
    vec3 pos = inData[readIdx + 0u].xyz;
    vec3 vel = inData[readIdx + 1u].xyz;

    // simulation parameters (from uniform)
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // accumulators
    vec3 sepAcc = vec3(0.0);
    uint sepCnt = 0u;

    vec3 alignAcc = vec3(0.0);
    uint alignCnt = 0u;

    vec3 cohAcc = vec3(0.0);
    uint cohCnt = 0u;

    // neighbor loop
    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;
        uint jIdx = j * 2u;
        vec3 posj = inData[jIdx + 0u].xyz;
        vec3 velj = inData[jIdx + 1u].xyz;

        vec3 r = posj - pos;
        float dist2 = dot(r, r);
        if (dist2 <= 0.0) {
            // skip zero-distance (avoid divide-by-zero) and identical positions
            continue;
        }

        // Separation: stronger when very close; use inverse-square falloff
        if (dist2 < sepRadius2) {
            // vector pointing away from neighbor, scaled by 1 / dist2 (repulsion)
            vec3 away = (pos - posj) / (dist2 + 1e-6);
            sepAcc += away;
            sepCnt += 1u;
        }

        // Alignment: average neighbor velocity
        if (dist2 < alignRadius2) {
            alignAcc += velj;
            alignCnt += 1u;
        }

        // Cohesion: average neighbor position
        if (dist2 < cohRadius2) {
            cohAcc += posj;
            cohCnt += 1u;
        }
    }

    // build acceleration from behaviors (averaging neighbor contributions)
    vec3 accel = vec3(0.0);

    if (sepCnt > 0u) {
        vec3 meanSep = sepAcc / float(sepCnt);
        // use meanSep direction as repulsion; scale by weight
        accel += meanSep * sepWeight;
    }

    if (alignCnt > 0u) {
        vec3 meanAlign = alignAcc / float(alignCnt);
        // steer toward average heading (desired - current)
        accel += (meanAlign - vel) * alignWeight;
    }

    if (cohCnt > 0u) {
        vec3 meanCohPos = cohAcc / float(cohCnt);
        // steer toward the center of mass
        accel += (meanCohPos - pos) * cohWeight;
    }

    // boundary force:
    // When a component exceeds boundSize*0.4 in magnitude, push back toward center.
    // Magnitude scaled by how far beyond threshold relative to boundSize*0.5 and by boundForce.
    float threshold = boundSize * 0.4;
    float scale = boundSize * 0.5;
    vec3 bforce = vec3(0.0);

    // X
    if (abs(pos.x) > threshold) {
        float overshoot = (abs(pos.x) - threshold) / max(scale, 1e-6);
        bforce.x = -sign(pos.x) * overshoot * boundForce;
    }
    // Y
    if (abs(pos.y) > threshold) {
        float overshoot = (abs(pos.y) - threshold) / max(scale, 1e-6);
        bforce.y = -sign(pos.y) * overshoot * boundForce;
    }
    // Z
    if (abs(pos.z) > threshold) {
        float overshoot = (abs(pos.z) - threshold) / max(scale, 1e-6);
        bforce.z = -sign(pos.z) * overshoot * boundForce;
    }

    accel += bforce;

    // integrate velocity and position (simple explicit Euler)
    vel += accel * dt;

    // clamp speed to maxSpeed
    float speed2 = dot(vel, vel);
    float maxSpeed2 = maxSpeed * maxSpeed;
    if (speed2 > maxSpeed2 && speed2 > 0.0) {
        vel = normalize(vel) * maxSpeed;
    }

    pos += vel * dt;

    // write out new state
    outData[readIdx + 0u] = vec4(pos, 0.0);
    outData[readIdx + 1u] = vec4(vel, 0.0);
}
