struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Map UV to complex plane c = (u*3-2, v*2-1)
    float2 uv = input.uv;
    float2 c = float2(uv.x * 3.0, uv.y * 2.0);
    c.x -= 2.0;
    c.y -= 1.0;

    float2 z = float2(0.0, 0.0);
    int iterations = 0;
    for (int i = 0; i < 20; ++i)
    {
        // z = z^2 + c  where z = (x,y), z^2 = (x^2 - y^2, 2xy)
        float x = z.x;
        float y = z.y;
        float xx = x*x - y*y + c.x;
        float yy = 2.0 * x * y + c.y;
        z = float2(xx, yy);

        if (dot(z, z) > 4.0)
        {
            iterations = i + 1; // count iterations up to escape
            break;
        }
    }

    if (iterations == 0)
    {
        // did not escape within 20 iterations
        iterations = 20;
    }

    float t = (float)iterations / 20.0;
    float3 color = float3(t, t * 0.5, 1.0 - t);
    return float4(color, 1.0);
}
