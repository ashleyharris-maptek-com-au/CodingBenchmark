Below is a single-file, portable C++20 solution that adapts between exact/backtracking for very small graphs, greedy DSATUR-like initialization, and a memory-backed fast min-conflicts repair (with a fallback scanning method when n*k would use too much memory). It tries to find a valid k-coloring within a long time budget (safe for the 5-minute environment you specified) and prints a color for every vertex (0..k-1).

Notes:
- It uses only standard headers and should compile with MSVC or g++ (C++17/C++20).
- For large graphs the algorithm uses a neighbor-color-count table (n*k integers). If n*k would exceed a configured threshold (to avoid excessive RAM), it uses on-the-fly scanning for color counts (slower but lower memory).
- If the graph contains impossible features (self-loop) the algorithm cannot satisfy the constraints (no valid coloring exists) — the program will still output something, but that case is logically unsatisfiable for any k.

Compile:
- g++ -O2 -std=c++17 graph_colorer.cpp
- Or compile with MSVC as specified.

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

// Adaptive graph colorer:
// - For tiny graphs (n <= 60), use backtracking with DSATUR-inspired ordering.
// - Otherwise: greedy initialization (largest-degree ordering) then min-conflicts repair.
// - Uses neighbor-color-count table (flat vector) when n * k is reasonable; otherwise fallback to scanning.

using chrono_clock = chrono::steady_clock;

int n, m, K;
vector<vector<int>> adj;
vector<int> degv;
mt19937_64 rng((uint64_t)chrono_clock::now().time_since_epoch().count());

// ----- Small exact solver (DSATUR-like backtracking) -----
bool backtrack_colorable(const vector<int> &order, vector<int> &color, int idx) {
    if (idx == (int)order.size()) return true;
    int v = order[idx];
    vector<char> used(K, 0);
    for (int u : adj[v]) if (color[u] != -1 && color[u] < K) used[color[u]] = 1;
    for (int c = 0; c < K; ++c) {
        if (!used[c]) {
            color[v] = c;
            if (backtrack_colorable(order, color, idx + 1)) return true;
            color[v] = -1;
        }
    }
    return false;
}

// ----- Main algorithm: greedy init + min-conflicts repair -----
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    if (!(cin >> n >> m >> K)) return 0;
    adj.assign(n, {});
    degv.assign(n, 0);
    bool self_loop = false;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        adj[u].push_back(v);
        if (u != v) adj[v].push_back(u);
        else self_loop = true;
    }
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
        degv[i] = (int)adj[i].size();
        // if self-loop exists (vertex adjacent to itself) -> impossible for any K
        if (find(adj[i].begin(), adj[i].end(), i) != adj[i].end()) self_loop = true;
    }

    // Trivial cases:
    if (n == 0) return 0;
    if (K <= 0) {
        // No valid colors; output zeros (will be invalid if any edge exists)
        for (int i = 0; i < n; ++i) cout << 0 << (i+1==n?'\n':' ');
        return 0;
    }
    if (self_loop) {
        // Graph has self-loop -> unsatisfiable. Still produce an output (judge will mark fail).
        for (int i = 0; i < n; ++i) cout << 0 << (i+1==n?'\n':' ');
        return 0;
    }

    // If very small, try exact backtracking (guarantees success if possible within K colors).
    if (n <= 60) {
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b){ return degv[a] > degv[b]; });
        vector<int> color(n, -1);
        bool ok = backtrack_colorable(order, color, 0);
        if (ok) {
            for (int i = 0; i < n; ++i) cout << color[i] << (i+1==n?'\n':' ');
            return 0;
        }
        // fallthrough to heuristic if exact fails
    }

    // Greedy initialization: largest-degree ordering, assign smallest available color [0..K-1]
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b){ return degv[a] > degv[b]; });

    vector<int> color(n, -1);
    vector<int> usedColors(K);
    for (int v : order) {
        if (degv[v] == 0) { color[v] = 0; continue; }
        fill(usedColors.begin(), usedColors.end(), 0);
        for (int u : adj[v]) if (color[u] != -1 && color[u] < K) usedColors[color[u]] = 1;
        int pick = -1;
        for (int c = 0; c < K; ++c) if (!usedColors[c]) { pick = c; break; }
        if (pick == -1) {
            // no free color: pick random color (introduces conflicts to be repaired)
            uniform_int_distribution<int> dis(0, K-1);
            pick = dis(rng);
        }
        color[v] = pick;
    }

    // Prepare for min-conflicts repair
    // Decide whether to allocate neighborColorCount table: size = (uint64_t)n * K
    const uint64_t MEM_THRESHOLD = 100000000ULL; // 100 million ints ~400MB
    bool use_table = (uint64_t)n * (uint64_t)K <= MEM_THRESHOLD;
    vector<int> neighCnt; // flat n*K table if used: neighCnt[v*K + c] = number neighbors of v colored c
    auto idx = [&](int v, int c)->size_t { return (size_t)v * (size_t)K + (size_t)c; };

    // conflict bookkeeping
    vector<int> conflicts(n, 0);
    vector<int> conflictPos(n, -1);
    vector<int> conflictList; conflictList.reserve(n);

    auto addConflict = [&](int v) {
        if (conflictPos[v] == -1) {
            conflictPos[v] = (int)conflictList.size();
            conflictList.push_back(v);
        }
    };
    auto removeConflict = [&](int v) {
        int p = conflictPos[v];
        if (p == -1) return;
        int last = conflictList.back();
        conflictList[p] = last;
        conflictPos[last] = p;
        conflictList.pop_back();
        conflictPos[v] = -1;
    };

    if (use_table) {
        neighCnt.assign((size_t)n * (size_t)K, 0);
        // build neighbor color counts
        for (int v = 0; v < n; ++v) {
            for (int u : adj[v]) {
                int cu = color[u];
                if (cu >= 0 && cu < K) neighCnt[idx(v, cu)]++;
            }
        }
        long long total_conflicts = 0;
        for (int v = 0; v < n; ++v) {
            int cv = color[v];
            if (cv < 0 || cv >= K) conflicts[v] = 0;
            else conflicts[v] = neighCnt[idx(v, cv)];
            if (conflicts[v] > 0) { addConflict(v); total_conflicts += conflicts[v]; }
        }
        // Each conflict counted twice (u and v), but we don't rely on that.
        // Min-conflicts iterative repair
        const double TIME_LIMIT = 280.0; // seconds
        auto start = chrono_clock::now();
        uniform_int_distribution<size_t> dis_index;
        size_t iterations = 0;
        while (!conflictList.empty()) {
            ++iterations;
            // time check
            if (iterations % 1024 == 0) {
                auto now = chrono_clock::now();
                double elapsed = chrono::duration_cast<chrono::duration<double>>(now - start).count();
                if (elapsed > TIME_LIMIT) break;
            }
            // pick random conflicting vertex
            dis_index = uniform_int_distribution<size_t>(0, conflictList.size() - 1);
            int v = conflictList[dis_index(rng)];
            int base = v * K;
            // choose color minimizing neighbor conflicts (neighCnt[v][c])
            int bestC = -1;
            int bestVal = INT_MAX;
            // small optimization: try current color first to potentially keep stable choices
            for (int c = 0; c < K; ++c) {
                int val = neighCnt[(size_t)base + (size_t)c];
                if (val < bestVal) { bestVal = val; bestC = c; }
            }
            if (bestC == color[v]) {
                // If current color already best, try random perturbation sometimes to escape plateau
                if (bestVal > 0 && (rng() & 15) == 0) {
                    uniform_int_distribution<int> disc(0, K-1);
                    bestC = disc(rng);
                } else {
                    // no improvement possible right now; pick random conflicting vertex next
                    // but do small perturbation: try next-best
                    // (we'll just continue to next loop iteration)
                }
            }
            int old = color[v];
            if (bestC == old) {
                // attempt to pick another color with equal cost but different
                bool found = false;
                for (int c = 0; c < K; ++c) {
                    if (c != old && neighCnt[(size_t)base + (size_t)c] == bestVal) { bestC = c; found = true; break; }
                }
                if (!found) { // leave as is, try different vertex next time
                    // small random jump
                    uniform_int_distribution<int> disc(0, K-1);
                    bestC = disc(rng);
                    if (bestC == old) continue;
                }
            }
            int newc = bestC;
            // perform recolor v: update neighbor counts and conflicts
            // For each neighbor u: neighCnt[u][old]-- ; neighCnt[u][newc]++
            for (int u : adj[v]) {
                // decrement count of color old at u
                neighCnt[idx(u, old)]--;
                // increment new color
                neighCnt[idx(u, newc)]++;
                // update conflicts[u]
                int prevConf = conflicts[u];
                int cu = color[u];
                if (cu >= 0 && cu < K) {
                    int newConf = neighCnt[idx(u, cu)];
                    conflicts[u] = newConf;
                    if (newConf > 0) addConflict(u);
                    else removeConflict(u);
                }
            }
            // update v's color and its conflicts
            color[v] = newc;
            conflicts[v] = neighCnt[idx(v, newc)];
            if (conflicts[v] > 0) addConflict(v); else removeConflict(v);
        }

        // Final check: if some conflicts remain, it's likely we timed out; still output
        // However ensure colors are within range 0..K-1
        for (int i = 0; i < n; ++i) {
            if (color[i] < 0 || color[i] >= K) color[i] = 0;
        }
        // Optional final repair: a pass of greedy recoloring of conflicting vertices
        for (int v = 0; v < n; ++v) {
            if (conflicts[v] == 0) continue;
            vector<char> used(K, 0);
            for (int u : adj[v]) if (color[u] >=0 && color[u] < K) used[color[u]] = 1;
            int pick = -1;
            for (int c = 0; c < K; ++c) if (!used[c]) { pick = c; break; }
            if (pick != -1) {
                color[v] = pick;
            }
        }

        for (int i = 0; i < n; ++i) cout << color[i] << (i+1==n?'\n':' ');
        return 0;
    } else {
        // Fallback: no table due to memory constraints -> use on-the-fly scans for counts
        // Build initial conflict counts by scanning neighbors
        long long total_conflicts = 0;
        for (int v = 0; v < n; ++v) {
            int cv = color[v];
            int cnt = 0;
            for (int u : adj[v]) if (color[u] == cv) ++cnt;
            conflicts[v] = cnt;
            if (cnt > 0) { addConflict(v); total_conflicts += cnt; }
        }

        const double TIME_LIMIT = 280.0; // seconds
        auto start = chrono_clock::now();
        uniform_int_distribution<size_t> dis_index;
        size_t iterations = 0;
        while (!conflictList.empty()) {
            ++iterations;
            if (iterations % 1024 == 0) {
                auto now = chrono_clock::now();
                double elapsed = chrono::duration_cast<chrono::duration<double>>(now - start).count();
                if (elapsed > TIME_LIMIT) break;
            }
            dis_index = uniform_int_distribution<size_t>(0, conflictList.size() - 1);
            int v = conflictList[dis_index(rng)];
            // For v evaluate for each color c how many neighbors have c (on the fly)
            int bestC = color[v];
            int bestVal = INT_MAX;
            // compute neighbor color counts by scanning neighbors and tabulating counts into local vector
            vector<int> cnts(K, 0);
            for (int u : adj[v]) {
                int cu = color[u];
                if (cu >= 0 && cu < K) cnts[cu]++;
            }
            for (int c = 0; c < K; ++c) {
                if (cnts[c] < bestVal) { bestVal = cnts[c]; bestC = c; }
            }
            if (bestC == color[v]) {
                // occasional random perturbation
                if (bestVal > 0 && (rng() & 15) == 0) {
                    uniform_int_distribution<int> disc(0, K-1);
                    bestC = disc(rng);
                } else {
                    // nothing to do right now
                    // pick another conflicting vertex next time
                }
            }
            int old = color[v];
            int newc = bestC;
            if (newc == old) continue;
            color[v] = newc;
            // Update conflicts for v and neighbors
            // Recompute conflicts for v and neighbors (on-the-fly)
            int newConfV = 0;
            for (int u : adj[v]) if (color[u] == color[v]) ++newConfV;
            conflicts[v] = newConfV;
            if (newConfV > 0) addConflict(v); else removeConflict(v);

            for (int u : adj[v]) {
                int cntu = 0;
                for (int w : adj[u]) if (color[w] == color[u]) ++cntu;
                conflicts[u] = cntu;
                if (cntu > 0) addConflict(u); else removeConflict(u);
            }
        }

        // Final greedy pass to try to fix remaining conflicts
        for (int v = 0; v < n; ++v) {
            if (conflicts[v] == 0) continue;
            vector<char> used(K, 0);
            for (int u : adj[v]) if (color[u] >= 0 && color[u] < K) used[color[u]] = 1;
            int pick = -1;
            for (int c = 0; c < K; ++c) if (!used[c]) { pick = c; break; }
            if (pick != -1) color[v] = pick;
        }

        for (int i = 0; i < n; ++i) {
            if (color[i] < 0 || color[i] >= K) color[i] = 0;
            cout << color[i] << (i+1==n?'\n':' ');
        }
        return 0;
    }
}
```
