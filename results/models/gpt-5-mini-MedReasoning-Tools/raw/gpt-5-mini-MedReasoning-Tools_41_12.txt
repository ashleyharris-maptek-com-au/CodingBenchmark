; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBOty "UBO"

; Decorations for input/output locations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOty Block
OpMemberDecorate %UBOty 0 Offset 0
OpMemberDecorate %UBOty 1 Offset 64
OpMemberDecorate %UBOty 2 Offset 128
OpMemberDecorate %UBOty 3 Offset 192
OpMemberDecorate %UBOty 4 Offset 208
OpMemberDecorate %UBOty 5 Offset 224

; Import GLSL std450 for helper ops
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Pointer types
%ptrInput_v3 = OpTypePointer Input %v3f
%ptrInput_v2 = OpTypePointer Input %v2f
%ptrOutput_v4 = OpTypePointer Output %v4f
%ptrUniform_UBO = OpTypePointer Uniform %UBOty
%ptrUniform_v4 = OpTypePointer Uniform %v4f

; UBO struct type
%UBOty = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f

; Variables (inputs, outputs, uniform)
%inWorldPos = OpVariable %ptrInput_v3 Input
%inNormal = OpVariable %ptrInput_v3 Input
%inUV = OpVariable %ptrInput_v2 Input
%inTangent = OpVariable %ptrInput_v3 Input
%inColor = OpVariable %ptrInput_v3 Input

%outColor = OpVariable %ptrOutput_v4 Output

%ubo = OpVariable %ptrUniform_UBO Uniform

; Constants
%const_f0 = OpConstant %float 0.0
%const_f0_05 = OpConstant %float 0.05
%const_f0_5 = OpConstant %float 0.5
%const_f1 = OpConstant %float 1.0
%const_f0_3 = OpConstant %float 0.3
%const_f0_4 = OpConstant %float 0.4
%const_f0_6 = OpConstant %float 0.6
%const_f0_15 = OpConstant %float 0.15
%const_f0_05b = OpConstant %float 0.05

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

; Composite constants for colors and vec3(1.0)
%skyColor = OpConstantComposite %v3f %const_f0_4 %const_f0_6 %const_f1
%groundColor = OpConstantComposite %v3f %const_f0_3 %const_f0_15 %const_f0_05b
%vec3_one = OpConstantComposite %v3f %const_f1 %const_f1 %const_f1

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %inWorldPos
%normal_in = OpLoad %v3f %inNormal

; Normalize the interpolated normal
%normal = OpExtInst %v3f %glsl Normalize %normal_in

; compute blend factor = normal.y * 0.5 + 0.5
%n_y = OpCompositeExtract %float %normal 1
%tmp_mul = OpFMul %float %n_y %const_f0_5
%factor = OpFAdd %float %tmp_mul %const_f0

; Note: user requested blend = N.y * 0.5 + 0.5 -> that yields range [0,1] if N.y in [-1,1]
; but above uses +0.0; correct to add 0.5, not 0.0. Fix: use const_f0_5 added
%factor_fixed = OpFAdd %float %tmp_mul %const_f0_5

; ambient = mix(groundColor, skyColor, factor) implemented as ground*(1-factor) + sky*factor
%one_minus_f = OpFSub %float %const_f1 %factor_fixed
%ground_scaled = OpVectorTimesScalar %v3f %groundColor %one_minus_f
%sky_scaled = OpVectorTimesScalar %v3f %skyColor %factor_fixed
%ambient = OpFAdd %v3f %ground_scaled %sky_scaled

; Fetch light position from UBO (member index 3)
%ptr_light = OpAccessChain %ptrUniform_v4 %ubo %int_3
%light_vec4 = OpLoad %v4f %ptr_light
; extract xyz
%light_xyz = OpVectorShuffle %v3f %v3f %v4f %light_vec4 %light_vec4 0 1 2

; compute lightDir = normalize(lightPos.xyz - worldPos)
%light_sub = OpFSub %v3f %light_xyz %worldPos
%lightDir = OpExtInst %v3f %glsl Normalize %light_sub

; compute dot = max(dot(normal, lightDir), 0.0)
%dot = OpExtInst %float %glsl Dot %normal %lightDir
%gt = OpFOrdGreaterThan %bool %dot %const_f0
%dot_clamped = OpSelect %float %gt %dot %const_f0

; diffuse term scalar = 0.3 * dot_clamped
%diff_scalar = OpFMul %float %const_f0_3 %dot_clamped
%diff_vec = OpVectorTimesScalar %v3f %vec3_one %diff_scalar

; final color = ambient + 0.3 * d
%final_rgb = OpFAdd %v3f %ambient %diff_vec

; output vec4(final_rgb, 1.0)
%out_vec4 = OpCompositeConstruct %v4f %final_rgb %const_f1
OpStore %outColor %out_vec4

OpReturn
OpFunctionEnd
