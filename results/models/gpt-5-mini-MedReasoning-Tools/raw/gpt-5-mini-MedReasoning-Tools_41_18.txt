; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %uboVar
OpExecutionMode %main OriginUpperLeft

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%bool = OpTypeBool
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; UBO struct: mat4 model, mat4 view, mat4 projection, vec4 lightPos, vec4 cameraPos, vec4 params
%_struct_UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer types
%ptrInput_v3 = OpTypePointer Input %v3float
%ptrInput_v2 = OpTypePointer Input %v2float
%ptrOutput_v4 = OpTypePointer Output %v4float
%ptrUniform_ubo = OpTypePointer Uniform %_struct_UBO
%ptrUniform_v4 = OpTypePointer Uniform %v4float

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0

; Variables (inputs/outputs/uniform)
%inWorldPos = OpVariable %ptrInput_v3 Input
%inNormal = OpVariable %ptrInput_v3 Input
%inUV = OpVariable %ptrInput_v2 Input
%inTangent = OpVariable %ptrInput_v3 Input
%inColor = OpVariable %ptrInput_v3 Input
%outColor = OpVariable %ptrOutput_v4 Output
%uboVar = OpVariable %ptrUniform_ubo Uniform

; Decorations: locations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; Decorations: UBO layout and binding
OpDecorate %_struct_UBO Block
OpMemberDecorate %_struct_UBO 0 Offset 0
OpMemberDecorate %_struct_UBO 1 Offset 64
OpMemberDecorate %_struct_UBO 2 Offset 128
OpMemberDecorate %_struct_UBO 3 Offset 192
OpMemberDecorate %_struct_UBO 4 Offset 208
OpMemberDecorate %_struct_UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Import GLSL extended instructions
%glsl = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load and normalize normal
%n_in = OpLoad %v3float %inNormal
%n_norm = OpExtInst %v3float %glsl Normalize %n_in

; Load world position
%wp = OpLoad %v3float %inWorldPos

; Load cameraPos (vec4) from UBO member 4 and extract xyz as vec3
%cam_ptr = OpAccessChain %ptrUniform_v4 %uboVar %int_4
%cam_v4 = OpLoad %v4float %cam_ptr
%cam_xyz = OpVectorShuffle %v3float %cam_v4 %cam_v4 0 1 2

; Compute view vector and normalize
%v_sub = OpFSub %v3float %cam_xyz %wp
%v_norm = OpExtInst %v3float %glsl Normalize %v_sub

; Dot product and clamp to >= 0
%dotNV = OpDot %float %n_norm %v_norm
%gt_zero = OpFOrdGreaterThan %bool %dotNV %float_0
%NdotV = OpSelect %float %gt_zero %dotNV %float_0

; Compare with 0.5
%lt_half = OpFOrdLessThan %bool %NdotV %float_0_5

OpSelectionMerge %merge None
OpBranchConditional %lt_half %then %else

%then = OpLabel
; if NdotV < 0.5: r=1, g=2*NdotV, b=0
%g_then = OpFMul %float %NdotV %float_2
%r_then = %float_1
%b_then = %float_0
%color_then = OpCompositeConstruct %v3float %r_then %g_then %b_then
OpBranch %merge

%else = OpLabel
; else: t = NdotV - 0.5; r=g=1 - t*2; b = t*2
%t = OpFSub %float %NdotV %float_0_5
%t2 = OpFMul %float %t %float_2
%one_minus = OpFSub %float %float_1 %t2
%r_else = %one_minus
%g_else = %one_minus
%b_else = %t2
%color_else = OpCompositeConstruct %v3float %r_else %g_else %b_else
OpBranch %merge

%merge = OpLabel
%resultColor = OpPhi %v3float %color_then %then %color_else %else

; Build vec4 with alpha = 1.0
%c_x = OpCompositeExtract %float %resultColor 0
%c_y = OpCompositeExtract %float %resultColor 1
%c_z = OpCompositeExtract %float %resultColor 2
%out_vec4 = OpCompositeConstruct %v4float %c_x %c_y %c_z %float_1
OpStore %outColor %out_vec4

OpReturn
OpFunctionEnd
