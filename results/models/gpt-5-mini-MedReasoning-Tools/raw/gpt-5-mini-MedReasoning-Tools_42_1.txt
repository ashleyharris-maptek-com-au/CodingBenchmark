OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpName %main "main"

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_h = OpConstant %float 0.05

; gl_PerVertex input block (array of size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float

; Per-vertex input arrays (array size 1 because InputPoints)
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Location decorations for inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; pointer to element types for access chains
%ptr_in_v3_elem = OpTypePointer Input %v3float

; Function (main)
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load the input position: gl_in[0].gl_Position
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr

; Load the input color: color[0]
%in_color_ptr = OpAccessChain %ptr_in_v3_elem %color %int_0
%in_color = OpLoad %v3float %in_color_ptr

; Extract components x,y,z,w
%in_x = OpCompositeExtract %float %in_pos 0
%in_y = OpCompositeExtract %float %in_pos 1
%in_z = OpCompositeExtract %float %in_pos 2
%in_w = OpCompositeExtract %float %in_pos 3

; compute offsets
%x_plus_h = OpFAdd %float %in_x %float_h
%x_minus_h = OpFSub %float %in_x %float_h
%y_plus_h = OpFAdd %float %in_y %float_h
%y_minus_h = OpFSub %float %in_y %float_h

; construct four output positions (top, right, left, bottom)
%pos_top = OpCompositeConstruct %v4float %in_x %y_plus_h %in_z %in_w
%pos_right = OpCompositeConstruct %v4float %x_plus_h %in_y %in_z %in_w
%pos_left = OpCompositeConstruct %v4float %x_minus_h %in_y %in_z %in_w
%pos_bottom = OpCompositeConstruct %v4float %in_x %y_minus_h %in_z %in_w

; pointer to gl_out.gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertices in triangle strip order: top, right, left, bottom
OpStore %out_pos_ptr %pos_top
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos_ptr %pos_right
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos_ptr %pos_left
OpStore %outColor %in_color
OpEmitVertex

OpStore %out_pos_ptr %pos_bottom
OpStore %outColor %in_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
