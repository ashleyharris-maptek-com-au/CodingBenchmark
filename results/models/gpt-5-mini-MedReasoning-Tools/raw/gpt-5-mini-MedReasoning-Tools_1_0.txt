Below is a single-file, complete C++20 program that adapts its strategy by problem size. It uses exact brute-force for tiny N, nearest-neighbor + 2-opt for small/medium N, and a very fast spatial ordering (Morton/Z-order) for large/epic N. The tour always starts at city 0 and the output is a single line with N integers (0-indexed), space-separated.

Notes:
- For extremely large N this implementation focuses on a memory- and time-efficient approximation (Morton ordering). That gives a valid permutation quickly and scales to millions of points; heavier local optimizations are only attempted for sizes where they're practical.
- The code attempts to respect a modest time budget for the more expensive optimization passes using chrono checks.
- Uses only standard headers supported by MSVC (C++20).

Compile (example):
cl /std:c++20 tsp.cpp

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

using uint64 = unsigned long long;
using uint32 = unsigned int;
using ll = long long;
struct Point { double x, y; };

static inline uint64 part1by1(uint64 x) {
    // spread lower 32 bits so there are zeros between bits (Morton)
    x &= 0x00000000ffffffffULL;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8))  & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2))  & 0x3333333333333333ULL;
    x = (x | (x << 1))  & 0x5555555555555555ULL;
    return x;
}

static inline uint64 morton2D(uint32 x, uint32 y) {
    uint64 xx = part1by1(x);
    uint64 yy = part1by1(y);
    return (xx << 1) | yy;
}

double dist2(const Point &a, const Point &b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx*dx + dy*dy;
}

vector<int> brute_force_tour(const vector<Point>& pts) {
    int N = (int)pts.size();
    if (N == 1) return {0};
    vector<int> perm(N);
    for (int i = 0; i < N; ++i) perm[i] = i;
    // fix 0 at start; permute 1..N-1
    vector<int> best = perm;
    double bestLen = numeric_limits<double>::infinity();
    vector<int> cur(N);
    cur[0] = 0;
    vector<int> rest;
    for (int i = 1; i < N; ++i) rest.push_back(i);
    sort(rest.begin(), rest.end());
    do {
        for (int i = 1; i < N; ++i) cur[i] = rest[i-1];
        double L = 0;
        for (int i = 0; i < N-1; ++i) L += sqrt(dist2(pts[cur[i]], pts[cur[i+1]]));
        if (L < bestLen) { bestLen = L; best = cur; }
    } while (next_permutation(rest.begin(), rest.end()));
    return best;
}

vector<int> nearest_neighbor(const vector<Point>& pts, int start=0) {
    int N = (int)pts.size();
    vector<int> tour;
    tour.reserve(N);
    vector<char> used(N, 0);
    int cur = start;
    tour.push_back(cur);
    used[cur] = 1;
    for (int k = 1; k < N; ++k) {
        int best = -1;
        double bestd = numeric_limits<double>::infinity();
        for (int i = 0; i < N; ++i) if (!used[i]) {
            double d = dist2(pts[cur], pts[i]);
            if (d < bestd) { bestd = d; best = i; }
        }
        cur = best;
        tour.push_back(cur);
        used[cur] = 1;
    }
    return tour;
}

bool try_2opt_once(vector<int>& tour, const vector<Point>& pts, int i, int k) {
    // perform 2-opt between i and k (i < k)
    // edges: (i-1,i) and (k,k+1) replaced by (i-1,k) and (i,k+1)
    int N = (int)tour.size();
    int a = tour[(i-1+N)%N];
    int b = tour[i];
    int c = tour[k];
    int d = tour[(k+1)%N];
    double oldd = sqrt(dist2(pts[a], pts[b])) + sqrt(dist2(pts[c], pts[d]));
    double newd = sqrt(dist2(pts[a], pts[c])) + sqrt(dist2(pts[b], pts[d]));
    if (newd + 1e-12 < oldd) {
        // reverse segment [i..k]
        reverse(tour.begin()+i, tour.begin()+k+1);
        return true;
    }
    return false;
}

void two_opt_full(vector<int>& tour, const vector<Point>& pts, chrono::steady_clock::time_point deadline) {
    int N = (int)tour.size();
    if (N <= 2) return;
    bool improved = true;
    while (improved && chrono::steady_clock::now() < deadline) {
        improved = false;
        for (int i = 1; i < N-1 && chrono::steady_clock::now() < deadline; ++i) {
            // consider k > i
            for (int k = i+1; k < N && chrono::steady_clock::now() < deadline; ++k) {
                if (try_2opt_once(tour, pts, i, k)) {
                    improved = true;
                    // after improvement restart inner loops (common strategy)
                }
            }
        }
    }
}

void windowed_2opt(vector<int>& tour, const vector<Point>& pts, int window, chrono::steady_clock::time_point deadline) {
    int N = (int)tour.size();
    if (N <= 2) return;
    bool improved = true;
    while (improved && chrono::steady_clock::now() < deadline) {
        improved = false;
        for (int i = 1; i < N-1 && chrono::steady_clock::now() < deadline; ++i) {
            int iend = min(N-1, i + window);
            for (int k = i+1; k <= iend && chrono::steady_clock::now() < deadline; ++k) {
                if (try_2opt_once(tour, pts, i, k)) {
                    improved = true;
                }
            }
        }
    }
}

vector<int> morton_order_tour(const vector<Point>& pts, int bits = 21) {
    int N = (int)pts.size();
    vector<int> idx(N);
    for (int i = 0; i < N; ++i) idx[i] = i;
    double xmin = numeric_limits<double>::infinity();
    double ymin = numeric_limits<double>::infinity();
    double xmax = -numeric_limits<double>::infinity();
    double ymax = -numeric_limits<double>::infinity();
    for (auto &p : pts) {
        xmin = min(xmin, p.x);
        ymin = min(ymin, p.y);
        xmax = max(xmax, p.x);
        ymax = max(ymax, p.y);
    }
    double dx = xmax - xmin;
    double dy = ymax - ymin;
    if (dx == 0) dx = 1.0;
    if (dy == 0) dy = 1.0;
    uint32 maxv = (bits >= 32) ? 0xFFFFFFFFu : ((1u << bits) - 1u);
    vector<uint64> keys(N);
    for (int i = 0; i < N; ++i) {
        uint32 xi = (uint32)floor((pts[i].x - xmin) / dx * double(maxv) + 0.5);
        uint32 yi = (uint32)floor((pts[i].y - ymin) / dy * double(maxv) + 0.5);
        keys[i] = morton2D(xi, yi);
    }
    sort(idx.begin(), idx.end(), [&](int a, int b){
        if (keys[a] != keys[b]) return keys[a] < keys[b];
        // tie-break by x then y
        if (pts[a].x != pts[b].x) return pts[a].x < pts[b].x;
        return pts[a].y < pts[b].y;
    });
    return idx;
}

void rotate_to_start(vector<int>& tour, int start_idx) {
    if (tour.empty()) return;
    int N = (int)tour.size();
    int pos = -1;
    for (int i = 0; i < N; ++i) if (tour[i] == start_idx) { pos = i; break; }
    if (pos <= 0) return;
    // rotate so that pos becomes 0
    vector<int> tmp;
    tmp.reserve(N);
    for (int i = 0; i < N; ++i) tmp.push_back(tour[(pos + i) % N]);
    tour.swap(tmp);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    vector<Point> pts;
    pts.reserve(max(1, N));
    for (int i = 0; i < N; ++i) {
        Point p;
        cin >> p.x >> p.y;
        pts.push_back(p);
    }

    vector<int> tour;

    if (N <= 1) {
        tour.push_back(0);
    } else if (N <= 10) {
        // exact brute force
        tour = brute_force_tour(pts);
    } else if (N <= 2000) {
        // nearest neighbor + (timed) 2-opt
        auto start = chrono::steady_clock::now();
        tour = nearest_neighbor(pts, 0);
        // ensure start is 0
        rotate_to_start(tour, 0);
        // give up to 4 seconds for 2-opt improvements (subject to overall limits)
        two_opt_full(tour, pts, start + chrono::seconds(4));
    } else if (N <= 100000) {
        // Morton ordering + windowed 2-opt
        auto start = chrono::steady_clock::now();
        int bits = 21; // ~2M grid resolution
        tour = morton_order_tour(pts, bits);
        // rotate to start at 0
        rotate_to_start(tour, 0);
        // run windowed 2-opt for some time proportional to N but capped
        int window = 500; // local window size
        auto deadline = start + chrono::seconds(5); // cap 5s
        if (N > 50000) deadline = start + chrono::seconds(3);
        windowed_2opt(tour, pts, window, deadline);
    } else {
        // Very large: fast Morton ordering only (good spatial locality)
        int bits = 21;
        // choose bits dynamically if huge N to increase resolution (but capped to 31)
        if (N > (1<<21)) bits = 24;
        if (N > (1<<24)) bits = 28;
        if (bits > 31) bits = 31;
        tour = morton_order_tour(pts, bits);
        rotate_to_start(tour, 0);
        // Optionally do a tiny local improvement on blocks
        // For extremely large N, skip heavy ops.
        if (N <= 200000) {
            auto start = chrono::steady_clock::now();
            windowed_2opt(tour, pts, 200, start + chrono::seconds(2));
        }
    }

    // final check: ensure it's a permutation starting with 0
    if (tour.size() != (size_t)N || tour[0] != 0) {
        // fallback: build simple permutation starting at 0
        vector<char> used(N, 0);
        vector<int> fallback;
        fallback.reserve(N);
        fallback.push_back(0);
        used[0] = 1;
        for (int i = 1; i < N; ++i) fallback.push_back(i), used[i]=1;
        tour.swap(fallback);
    } else {
        // check uniqueness, fallback if problem
        vector<char> seen(N, 0);
        bool ok = true;
        for (int v : tour) {
            if (v < 0 || v >= N || seen[v]) { ok = false; break; }
            seen[v] = 1;
        }
        if (!ok) {
            vector<int> fallback(N);
            iota(fallback.begin(), fallback.end(), 0);
            tour.swap(fallback);
        }
    }

    // Output one line with N integers, space-separated.
    // Use efficient integer->string conversion (std::to_chars).
    string out;
    out.reserve((size_t)N * 4 + 64);
    char buf[32];
    for (int i = 0; i < N; ++i) {
        auto v = tour[i];
        // use to_chars
        auto res = std::to_chars(buf, buf + sizeof(buf), v);
        out.append(buf, res.ptr);
        if (i + 1 < N) out.push_back(' ');
    }
    out.push_back('\n');
    cout << out;
    return 0;
}
```
