```rust
use std::io::{self, BufRead, Write};
use std::cmp::min;

/// Compact bitset for boolean flags (1 bit per index)
struct BitVec {
    data: Vec<u64>,
    len: usize,
}
impl BitVec {
    fn with_len(len: usize) -> Self {
        let words = (len + 63) / 64;
        BitVec { data: vec![0u64; words], len }
    }
    #[inline]
    fn set(&mut self, idx: usize) {
        let word = idx >> 6;
        let bit = idx & 63;
        self.data[word] |= 1u64 << bit;
    }
    #[inline]
    fn clear(&mut self, idx: usize) {
        let word = idx >> 6;
        let bit = idx & 63;
        self.data[word] &= !(1u64 << bit);
    }
    #[inline]
    fn get(&self, idx: usize) -> bool {
        let word = idx >> 6;
        let bit = idx & 63;
        (self.data[word] >> bit) & 1u64 == 1u64
    }
}

/// Packed storage for 2-bit values per index (values 0..3)
struct Packed2 {
    data: Vec<u64>,
    len: usize,
}
impl Packed2 {
    fn with_len(len: usize) -> Self {
        let total_bits = len.checked_mul(2).expect("too many bits");
        let words = (total_bits + 63) / 64;
        Packed2 { data: vec![0u64; words], len }
    }
    #[inline]
    fn set(&mut self, idx: usize, val: u8) {
        debug_assert!(val < 4);
        let pos = idx * 2;
        let word = pos >> 6;
        let off = pos & 63;
        // clear two bits
        self.data[word] &= !(3u64 << off);
        self.data[word] |= ((val as u64) << off);
        // if spans words
        if off > 62 {
            // bits split across words: high bits go to next word
            let carry = (val as u64) >> (64 - off);
            self.data[word + 1] &= !( (1u64 << (off - 62)) - 1 ); // clear only needed bits (safe)
            self.data[word + 1] |= carry;
        }
    }
    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let pos = idx * 2;
        let word = pos >> 6;
        let off = pos & 63;
        if off <= 62 {
            ((self.data[word] >> off) & 3u64) as u8
        } else {
            // split across words
            let low = (self.data[word] >> off) & ((1u64 << (64 - off)) - 1);
            let high = self.data[word + 1] & ((1u64 << (off - 62)) - 1);
            ((low | (high << (64 - off))) & 3u64) as u8
        }
    }
}

fn main() {
    // Fast line-based reading: read first line for H W, then process H subsequent lines streaming.
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    let mut first_line = String::new();
    if stdin_lock.read_line(&mut first_line).unwrap() == 0 {
        // no input
        return;
    }
    // parse H and W
    let mut parts = first_line.split_whitespace();
    let h_str = parts.next().unwrap_or("0");
    let w_str = parts.next().unwrap_or("0");
    let H: usize = match h_str.parse() {
        Ok(v) => v,
        Err(_) => {
            eprintln!("Invalid H");
            return;
        }
    };
    let W: usize = match w_str.parse() {
        Ok(v) => v,
        Err(_) => {
            eprintln!("Invalid W");
            return;
        }
    };

    // guard against absurd multiplication overflow
    let ncells = match (H as u128).checked_mul(W as u128) {
        Some(v) if v <= (usize::MAX as u128) => v as usize,
        _ => {
            // too big
            println!("0");
            return;
        }
    };

    // Allocate bitsets: walls, visited, and packed parent directions (2 bits each)
    let mut walls = BitVec::with_len(ncells);
    let mut visited = BitVec::with_len(ncells);
    let mut parent = Packed2::with_len(ncells);

    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    // read next H lines
    let mut line = String::new();
    for r in 0..H {
        line.clear();
        if stdin_lock.read_line(&mut line).unwrap() == 0 {
            // insufficient lines - treat missing as empty
            // fill remainder with walls
            for c in 0..W {
                let idx = r * W + c;
                walls.set(idx);
            }
            continue;
        }
        // remove trailing newline and possibly '\r'
        if line.ends_with('\n') {
            line.pop();
            if line.ends_with('\r') {
                line.pop();
            }
        } else if line.ends_with('\r') {
            line.pop();
        }
        // Ensure the line has at least W chars; if shorter, treat missing chars as walls
        let bytes = line.as_bytes();
        let len = bytes.len();
        let use_len = min(len, W);
        for c in 0..use_len {
            let b = bytes[c];
            let idx = r * W + c;
            match b {
                b'#' => walls.set(idx),
                b'A' => {
                    start_idx = Some(idx);
                    // open by default (no wall)
                }
                b'B' => {
                    goal_idx = Some(idx);
                }
                b' ' => { /* open */ }
                // treat any other char ('.', etc) as open unless it's '#'
                _ => { /* open */ }
            }
        }
        // If line shorter than W, mark remaining as walls
        for c in use_len..W {
            let idx = r * W + c;
            walls.set(idx);
        }
    }

    let start_idx = match start_idx {
        Some(v) => v,
        None => {
            // no start -> no path
            println!("0");
            return;
        }
    };
    let goal_idx = match goal_idx {
        Some(v) => v,
        None => {
            // no goal -> no path
            println!("0");
            return;
        }
    };

    if start_idx == goal_idx {
        // trivial
        println!("1");
        let r = start_idx / W;
        let c = start_idx % W;
        println!("{} {}", r, c);
        return;
    }

    // BFS
    // We'll use a simple Vec as a queue with a head index for very large workloads.
    let mut q: Vec<usize> = Vec::new();
    q.reserve(1024);
    let mut head: usize = 0;

    visited.set(start_idx);
    // parent for start left as 0 (unused)

    q.push(start_idx);

    // direction mapping: 0=up,1=right,2=down,3=left
    // dr = [-1,0,1,0], dc = [0,1,0,-1]
    let mut found = false;

    while head < q.len() {
        let cur = q[head];
        head += 1;

        let r = cur / W;
        let c = cur % W;

        // Up
        if r > 0 {
            let nid = cur - W;
            if !visited.get(nid) && !walls.get(nid) {
                parent.set(nid, 0); // moved up from parent -> parent -> child dir = up
                visited.set(nid);
                if nid == goal_idx { found = true; break; }
                q.push(nid);
            }
        }
        // Right
        if c + 1 < W {
            let nid = cur + 1;
            if !visited.get(nid) && !walls.get(nid) {
                parent.set(nid, 1);
                visited.set(nid);
                if nid == goal_idx { found = true; break; }
                q.push(nid);
            }
        }
        // Down
        if r + 1 < H {
            let nid = cur + W;
            if !visited.get(nid) && !walls.get(nid) {
                parent.set(nid, 2);
                visited.set(nid);
                if nid == goal_idx { found = true; break; }
                q.push(nid);
            }
        }
        // Left
        if c > 0 {
            let nid = cur - 1;
            if !visited.get(nid) && !walls.get(nid) {
                parent.set(nid, 3);
                visited.set(nid);
                if nid == goal_idx { found = true; break; }
                q.push(nid);
            }
        }
    }

    if !visited.get(goal_idx) {
        // no path
        println!("0");
        return;
    }

    // Reconstruct path from goal back to start using parent directions.
    // We'll collect coordinates in reverse then reverse them.
    let mut path_rev: Vec<(usize, usize)> = Vec::new();
    let mut cur = goal_idx;
    let mut steps = 0usize;
    path_rev.push((cur / W, cur % W));
    while cur != start_idx {
        let dir = parent.get(cur);
        // parent = cur - dr,dc where dr/dc correspond to parent's -> child movement
        // parent = child - (dr,dc)
        cur = match dir {
            0 => {
                // came from parent -> child was up (dr=-1), so parent = child - (-1,0) = child + (1,0)
                // parent is below: +W
                cur + W
            }
            1 => {
                // right: parent = child - (0,1) => -1
                cur - 1
            }
            2 => {
                // down: parent = child - (1,0) => -W
                cur - W
            }
            3 => {
                // left: parent = child - (0,-1) => +1
                cur + 1
            }
            _ => {
                // should not happen
                println!("0");
                return;
            }
        };
        path_rev.push((cur / W, cur % W));
        steps += 1;
        // safety: avoid infinite loop - but cur should eventually be start
        // we can skip extra checks for performance under assumption of valid BFS parent links
    }

    // reverse to get start -> goal
    path_rev.reverse();
    // Output as: Line1 N (number of steps in path)
    // Next N lines: x y  where x y are 0-indexed coordinates from top-left (we interpret as row col)
    let mut out = String::new();
    out.push_str(&format!("{}\n", path_rev.len()));
    for &(r, c) in path_rev.iter() {
        out.push_str(&format!("{} {}\n", r, c));
    }
    let stdout = io::stdout();
    let mut stdout_lock = stdout.lock();
    stdout_lock.write_all(out.as_bytes()).unwrap();
}
```
