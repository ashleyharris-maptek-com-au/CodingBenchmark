#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(location = 0) out vec4 outColor;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

void main() {
    // Polka-dot pattern in UV / cell space (6x6)
    vec2 scaledUV = uv * 6.0;
    vec2 cellCoord = fract(scaledUV) - 0.5; // center each cell at (0,0)
    float dist = length(cellCoord);

    // Dot radius in cell space
    float radius = 0.3;

    // Anti-aliasing using derivatives
    float edgeAA = fwidth(dist);
    // mask = 1.0 inside circle, smooth to 0.0 outside
    float mask = 1.0 - smoothstep(radius - edgeAA, radius + edgeAA, dist);

    // Colors
    vec3 dotColor = vec3(1.0, 0.8, 0.0);         // yellow
    vec3 bgColor  = vec3(0.1, 0.1, 0.3);         // dark blue

    vec3 baseColor = mix(bgColor, dotColor, mask);

    // Lambertian diffuse
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);

    float lambertFactor = 0.3 + 0.7 * diffuse;

    vec3 finalColor = baseColor * lambertFactor;

    outColor = vec4(finalColor, 1.0);
}
