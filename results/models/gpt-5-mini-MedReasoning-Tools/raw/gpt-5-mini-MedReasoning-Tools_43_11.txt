[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Polka-dot pattern parameters
    const float2 grid = float2(6.0, 6.0);      // 6x6 grid
    const float  radius = 0.3;                 // radius in cell space
    const float  aa = 0.008;                   // anti-alias width

    // Compute cell-space UV and distance from cell center
    float2 cellUV = input.uv * grid;
    float2 f = frac(cellUV);
    float2 centered = f - 0.5;
    float dist = length(centered);

    // Mask for dot (smooth edge)
    float mask = 1.0 - smoothstep(radius - aa, radius + aa, dist);

    // Colors
    float3 dotColor = float3(1.0, 0.8, 0.0);       // yellow
    float3 bgColor  = float3(0.1, 0.1, 0.3);       // dark blue
    float3 baseColor = lerp(bgColor, dotColor, mask);

    // Lambertian lighting
    float3 N = normalize(input.normal);
    float3 L = normalize((lightPos.xyz) - input.worldPos);
    float  diffuse = max(dot(N, L), 0.0);

    // Apply requested shading factor: 0.3 + 0.7 * diffuse
    float shade = 0.3 + 0.7 * diffuse;
    float3 shaded = baseColor * shade;

    return float4(shaded, 1.0);
}
