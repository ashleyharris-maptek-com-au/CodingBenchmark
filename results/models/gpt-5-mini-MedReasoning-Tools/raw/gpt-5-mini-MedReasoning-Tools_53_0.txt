# Autopilot for spacecraft docking (single-module, persistent across calls)
# Function signature:
# def autopilot_step(state: dict, dt: float) -> dict:
#
# Notes:
# - LVLH frame: x=radial (up), y=along-track (prograde), z=cross-track (north)
# - thrust_* outputs are normalized -1..1 (physical thruster ~220N each).
# - att_rate_* outputs are rad/s limited to ±0.035.

import math
import time

# ---------- Persistent module-level storage (survives across calls) ----------
_persist = {
    "t": 0.0,
    "last_update_time": None,
    # State estimate in LVLH (meters, m/s)
    "pos_est": [0.0, 0.0, 0.0],
    "vel_est": [0.0, 0.0, 0.0],
    # low-pass smoothing / sensor fusion weights history
    "pos_cov": [1000.0, 1000.0, 1000.0],
    "vel_cov": [1000.0, 1000.0, 1000.0],
    "last_valid_meas_time": 0.0,
    # integrators for simple I-term (anti-windup)
    "int_err": [0.0, 0.0, 0.0],
    # last commanded thrusts (for detecting uncommanded)
    "last_thrust_cmd": [0.0, 0.0, 0.0],
    # small timers for far-range burns
    "burn_until": 0.0,
    # low-pass previous values
    "smoothed_ranges": [None, None, None],
    # last measured angular rates (rad/s) average of gyros
    "ang_rate_meas": [0.0, 0.0, 0.0],
    # last time we saw a good position measurement
    "last_good_pos_time": 0.0,
}

# Assumed dry mass (kg) - used to convert throttle fraction. This is a conservative internal guess and
# autopilot will use the fuel_remaining_kg from state to compute mass.
_DRY_MASS_KG = 1500.0

# Thruster specs
_THRUST_PER_THRUSTER_N = 220.0  # N
# We'll assume a single axis equivalent thrust per throttle command; normalized outputs map to N via mass estimate.

# Attitude rate limits
_ATT_RATE_MAX = 0.035  # rad/s

# Safety distances and velocities
DIST_FAR = 200.0
DIST_MED = 50.0
DIST_FINAL = 10.0
TARGET_CLOSE_SPEED = 0.1  # m/s for final
DOCK_SUCCESS_RANGE = 2.0
DOCK_SUCCESS_SPEED = 0.3

# Controller gains (tuned conservatively)
# Gains produce desired acceleration (m/s^2) = Kp*pos_err + Kd*vel_err + Ki*int_err
_GAINS = {
    "close": {"Kp": 0.06, "Kd": 1.0, "Ki": 0.002, "max_acc": 0.15},
    "medium": {"Kp": 0.04, "Kd": 0.8, "Ki": 0.0, "max_acc": 0.12},
    "far": {"Kp": 0.01, "Kd": 0.6, "Ki": 0.0, "max_acc": 0.05},
}

# Helper utilities
def _clamp(v, lo, hi):
    return max(lo, min(hi, v))

def _deg2rad(d):
    return d * math.pi / 180.0

def _rad2deg(r):
    return r * 180.0 / math.pi

def _vec_add(a, b):
    return [a[i] + b[i] for i in range(3)]

def _vec_sub(a, b):
    return [a[i] - b[i] for i in range(3)]

def _vec_scale(a, s):
    return [a[i] * s for i in range(3)]

def _vec_len(a):
    return math.sqrt(a[0]**2 + a[1]**2 + a[2]**2)

def _safe_avg(vals, valids):
    total = 0.0
    count = 0
    for v, ok in zip(vals, valids):
        if ok:
            total += v
            count += 1
    if count == 0:
        return None
    return total / count

def _current_mass(state):
    fuel = state.get("fuel_remaining_kg", None)
    if fuel is None:
        fuel = 0.0
    return _DRY_MASS_KG + max(0.0, fuel)

# Sensor fusion for relative position & velocity (LVLH)
def _fuse_position_velocity(state, dt):
    """
    Produce best-estimate pos_est (m) and vel_est (m/s) in LVLH using available sensors.
    Priority: Lidar > Radar (vector pos+vel). If only scalar ranges, attempt to use camera bearing.
    Otherwise use last estimate (propagated).
    """
    pos_meas = [None, None, None]
    vel_meas = [None, None, None]
    valid_pos = [False, False, False]
    valid_vel = [False, False, False]

    # Lidar (preferred)
    if state.get("lidar_rel_x_valid") and state.get("lidar_rel_y_valid") and state.get("lidar_rel_z_valid"):
        pos_meas = [
            state.get("lidar_rel_pos_x_m", 0.0),
            state.get("lidar_rel_pos_y_m", 0.0),
            state.get("lidar_rel_pos_z_m", 0.0),
        ]
        valid_pos = [True, True, True]
        if state.get("lidar_relv_x_valid") and state.get("lidar_relv_y_valid") and state.get("lidar_relv_z_valid"):
            vel_meas = [
                state.get("lidar_rel_vel_x_ms", 0.0),
                state.get("lidar_rel_vel_y_ms", 0.0),
                state.get("lidar_rel_vel_z_ms", 0.0),
            ]
            valid_vel = [True, True, True]

    # Radar fallback (if lidar not all valid)
    elif state.get("radar_rel_x_valid") and state.get("radar_rel_y_valid") and state.get("radar_rel_z_valid"):
        pos_meas = [
            state.get("radar_rel_pos_x_m", 0.0),
            state.get("radar_rel_pos_y_m", 0.0),
            state.get("radar_rel_pos_z_m", 0.0),
        ]
        valid_pos = [True, True, True]
        if state.get("radar_relv_x_valid") and state.get("radar_relv_y_valid") and state.get("radar_relv_z_valid"):
            vel_meas = [
                state.get("radar_rel_vel_x_ms", 0.0),
                state.get("radar_rel_vel_y_ms", 0.0),
                state.get("radar_rel_vel_z_ms", 0.0),
            ]
            valid_vel = [True, True, True]

    # Docking camera fallback (active within ~500 m)
    elif state.get("dock_camera_active"):
        # bearing components are unit direction toward dock in camera frame aligned to LVLH here
        bx = state.get("dock_camera_bearing_x", 0.0)
        by = state.get("dock_camera_bearing_y", 0.0)
        bz = state.get("dock_camera_bearing_z", 0.0)
        br_valid = state.get("dock_camera_bearing_x_valid") and state.get("dock_camera_bearing_y_valid") and state.get("dock_camera_bearing_z_valid")
        rng = state.get("dock_camera_range_m", None)
        if br_valid and rng is not None:
            pos_meas = [bx * rng, by * rng, bz * rng]
            valid_pos = [True, True, True]

    # Scalar range triangulation: if 3 non-collinear range sensors valid, approximate position.
    # Here we do a simpler fallback: if a single range finds distance and there's a good last direction, trust radial distance along estimated LOS (i.e., use last pos unit vector).
    rvals = []
    rvalid = []
    for i in range(1, 4):
        r = state.get(f"range_{i}_m", None)
        v = bool(state.get(f"range_{i}_valid", False))
        rvals.append(r)
        rvalid.append(v)

    if not any(valid_pos) and any(rvalid):
        # If we have a prior estimated position direction, use that to set magnitude from the best range reading.
        best_idx = None
        for i, v in enumerate(rvalid):
            if v and rvals[i] is not None:
                best_idx = i
                break
        if best_idx is not None:
            rng = rvals[best_idx]
            # If we have a previous estimate, use its unit vector; otherwise assume along -y (V-bar behind)
            last_pos = _persist["pos_est"]
            if _vec_len(last_pos) > 1.0:
                dir_v = _vec_scale(last_pos, 1.0 / _vec_len(last_pos))
            else:
                # Assume approach primarily along -y (behind station)
                dir_v = [0.0, -1.0, 0.0]
            pos_meas = [dir_v[0] * rng, dir_v[1] * rng, dir_v[2] * rng]
            valid_pos = [True, True, True]

    # Fuse with simple complementary/KF-like update
    pos_est = _persist["pos_est"][:]
    vel_est = _persist["vel_est"][:]
    pos_cov = _persist["pos_cov"][:]
    vel_cov = _persist["vel_cov"][:]

    # Predict step: ballistic propagation (vel holds)
    pos_est = [pos_est[i] + vel_est[i] * dt for i in range(3)]
    # Slightly increase covariances
    for i in range(3):
        pos_cov[i] += 0.1 * dt
        vel_cov[i] += 0.1 * dt

    # Update step: blend measurements when available
    meas_used = False
    for i in range(3):
        if valid_pos[i]:
            meas_used = True
            z = pos_meas[i]
            # measurement covariance: lower for lidar, medium for radar/camera
            if state.get("lidar_rel_x_valid") and state.get("lidar_rel_y_valid") and state.get("lidar_rel_z_valid"):
                R = 0.5  # meters^2
            elif state.get("radar_rel_x_valid") and state.get("radar_rel_y_valid") and state.get("radar_rel_z_valid"):
                R = 2.0
            else:
                R = 5.0
            # Kalman-like gain
            K = pos_cov[i] / (pos_cov[i] + R)
            pos_est[i] = pos_est[i] + K * (z - pos_est[i])
            pos_cov[i] = (1.0 - K) * pos_cov[i]
            _persist["last_good_pos_time"] = _persist["t"]

        if valid_vel[i]:
            meas_used = True
            z = vel_meas[i]
            if state.get("lidar_relv_x_valid") and state.get("lidar_relv_y_valid") and state.get("lidar_relv_z_valid"):
                Rv = 0.05  # (m/s)^2
            elif state.get("radar_relv_x_valid") and state.get("radar_relv_y_valid") and state.get("radar_relv_z_valid"):
                Rv = 0.1
            else:
                Rv = 0.5
            K = vel_cov[i] / (vel_cov[i] + Rv)
            vel_est[i] = vel_est[i] + K * (z - vel_est[i])
            vel_cov[i] = (1.0 - K) * vel_cov[i]

    # If no measurement recently, try to damp velocity slowly (helps against uncommanded accelerations)
    if not meas_used:
        for i in range(3):
            vel_est[i] *= 0.999

    # Save back
    _persist["pos_est"] = pos_est
    _persist["vel_est"] = vel_est
    _persist["pos_cov"] = pos_cov
    _persist["vel_cov"] = vel_cov

    return pos_est, vel_est

def _compute_attitude_rates(state, dt):
    """
    Compute attitude rate commands to stabilize angular rates and align to star trackers if present.
    Returns (roll_rate, pitch_rate, yaw_rate) in rad/s limited to ±_ATT_RATE_MAX.
    Strategy:
      - If gyros available: damper => command to oppose measured body rates (PD damping).
      - If star tracker angles available: small proportional to angle error (deg->rad) to slowly align.
    """
    # Average available gyro rates (dps -> deg/s). Gyros keys have 3 sensors each with pitch/roll/yaw in dps.
    roll_rates_dps = []
    pitch_rates_dps = []
    yaw_rates_dps = []
    for g in (1, 2, 3):
        if state.get(f"gyro_{g}_roll_valid"):
            roll_rates_dps.append(state.get(f"gyro_{g}_roll_dps", 0.0))
        if state.get(f"gyro_{g}_pitch_valid"):
            pitch_rates_dps.append(state.get(f"gyro_{g}_pitch_dps", 0.0))
        if state.get(f"gyro_{g}_yaw_valid"):
            yaw_rates_dps.append(state.get(f"gyro_{g}_yaw_dps", 0.0))
    def _avg_degps(lst):
        return sum(lst) / len(lst) if len(lst) > 0 else 0.0
    roll_rate_meas = _avg_degps(roll_rates_dps) * (math.pi / 180.0)  # to rad/s
    pitch_rate_meas = _avg_degps(pitch_rates_dps) * (math.pi / 180.0)
    yaw_rate_meas = _avg_degps(yaw_rates_dps) * (math.pi / 180.0)

    _persist["ang_rate_meas"] = [roll_rate_meas, pitch_rate_meas, yaw_rate_meas]

    # Damping: oppose measured rates
    DampGain = 0.6
    roll_cmd = -DampGain * roll_rate_meas
    pitch_cmd = -DampGain * pitch_rate_meas
    yaw_cmd = -DampGain * yaw_rate_meas

    # If star tracker provides attitude, add small P term to align (angles in degrees)
    st_angles = [None, None, None]  # pitch, roll, yaw in deg
    st_valid_counts = 0
    for s in (1, 2):
        if state.get(f"star_tracker_{s}_pitch_valid"):
            if st_angles[0] is None:
                st_angles[0] = 0.0
            st_angles[0] += state.get(f"star_tracker_{s}_pitch_deg", 0.0)
            st_valid_counts += 1
        if state.get(f"star_tracker_{s}_roll_valid"):
            if st_angles[1] is None:
                st_angles[1] = 0.0
            st_angles[1] += state.get(f"star_tracker_{s}_roll_deg", 0.0)
        if state.get(f"star_tracker_{s}_yaw_valid"):
            if st_angles[2] is None:
                st_angles[2] = 0.0
            st_angles[2] += state.get(f"star_tracker_{s}_yaw_deg", 0.0)
    if st_valid_counts > 0:
        for i in range(3):
            if st_angles[i] is not None:
                st_angles[i] /= st_valid_counts
        # Convert deg->rad and apply small P
        AttAlignGain = 0.003
        # pitch -> pitch_cmd, roll -> roll_cmd, yaw -> yaw_cmd
        if st_angles[0] is not None:
            pitch_cmd += -AttAlignGain * _deg2rad(st_angles[0])
        if st_angles[1] is not None:
            roll_cmd += -AttAlignGain * _deg2rad(st_angles[1])
        if st_angles[2] is not None:
            yaw_cmd += -AttAlignGain * _deg2rad(st_angles[2])

    # Clip to rate limits
    roll_cmd = _clamp(roll_cmd, -_ATT_RATE_MAX, _ATT_RATE_MAX)
    pitch_cmd = _clamp(pitch_cmd, -_ATT_RATE_MAX, _ATT_RATE_MAX)
    yaw_cmd = _clamp(yaw_cmd, -_ATT_RATE_MAX, _ATT_RATE_MAX)

    return roll_cmd, pitch_cmd, yaw_cmd

def _control_law(pos_est, vel_est, state, dt):
    """
    Main guidance + control logic that outputs thrust fractions (-1..1) for radial(x), along-track(y), cross-track(z).
    Uses phased behavior based on distance and desired approach velocities.
    """

    # Distance and closing speed along y (along-track)
    r = _vec_len(pos_est)
    rel_y = pos_est[1]
    rel_vy = vel_est[1]

    mass = _current_mass(state)
    max_acc_per_throttle = _THRUST_PER_THRUSTER_N / mass  # m/s^2 per normalized thrust=1.0 (approx)
    # For safety, assume only one thruster per axis available effectively -> use this acceleration mapping.

    # Choose gains by range
    if r > DIST_FAR:
        mode = "far"
    elif r > DIST_MED:
        mode = "medium"
    else:
        mode = "close"

    gains = _GAINS[mode]

    # Desired approach velocities:
    if r > DIST_FAR:
        # Far: we want to close along y by using V-bar philosophy.
        # If we are behind (rel_y < 0), to catch up we must LOWER orbit => retrograde burn (negative thrust_y).
        # If ahead (rel_y > 0), to let station catch up, we may raise orbit (positive thrust_y) to slow relative approach.
        # Make desired along-track speed toward 0 with modest magnitude depending on distance.
        desired_vy = _clamp(-0.8 * math.copysign(1.0, rel_y) * min(1.0, abs(rel_y) / 1000.0), -1.5, 1.5)
    elif r > DIST_MED:
        # Medium: target gentle closing speed proportional to distance but capped (~0.5 m/s)
        desired_vy = _clamp(-0.5 * (rel_y / 150.0), -0.8, 0.8)
    else:
        # Close: linear regime target very small approach speeds
        desired_vy = _clamp(-0.3 * (rel_y / 20.0), -0.5, 0.5)
        # If within final approach zone reduce desired speed to <0.1 m/s
        if r < DIST_FINAL:
            desired_vy = _clamp(-TARGET_CLOSE_SPEED * math.copysign(1.0, rel_y), -TARGET_CLOSE_SPEED, TARGET_CLOSE_SPEED)

    # For radial and cross-track, aim to null position and velocity
    desired_vx = -0.5 * (pos_est[0] / max(1.0, r)) * min(1.0, r / 100.0)  # bring radial toward 0 slowly
    desired_vz = -0.6 * (pos_est[2] / max(1.0, r)) * min(1.0, r / 100.0)  # cross-track

    # Form velocity errors
    err_vx = vel_est[0] - desired_vx
    err_vy = vel_est[1] - desired_vy
    err_vz = vel_est[2] - desired_vz

    # Position errors for PD
    err_px = pos_est[0]
    err_py = pos_est[1]
    err_pz = pos_est[2]

    # Integrator update (only in close mode and small anti-windup)
    if mode == "close":
        Ki = gains["Ki"]
        # integrate only along axes when small control required
        _persist["int_err"][0] += err_px * dt * 0.1
        _persist["int_err"][1] += err_py * dt * 0.1
        _persist["int_err"][2] += err_pz * dt * 0.1
        # clamp integrators
        for i in range(3):
            _persist["int_err"][i] = _clamp(_persist["int_err"][i], -10.0, 10.0)
    else:
        # slowly decay integrator
        for i in range(3):
            _persist["int_err"][i] *= 0.995

    # Compute desired accelerations (m/s^2)
    ax_des = - (gains["Kp"] * err_px + gains["Kd"] * err_vx + gains["Ki"] * _persist["int_err"][0])
    ay_des = - (gains["Kp"] * err_py + gains["Kd"] * err_vy + gains["Ki"] * _persist["int_err"][1])
    az_des = - (gains["Kp"] * err_pz + gains["Kd"] * err_vz + gains["Ki"] * _persist["int_err"][2])

    # Clip accelerations to mode-specific max to avoid aggressive burns
    amax = gains["max_acc"]
    ax_des = _clamp(ax_des, -amax, amax)
    ay_des = _clamp(ay_des, -amax, amax)
    az_des = _clamp(az_des, -amax, amax)

    # Special logic: far-range V-bar Hohmann-like behavior: if behind and large distance, use a sustained retrograde burn to lower orbit
    thrust_cmd = [0.0, 0.0, 0.0]  # normalized

    if mode == "far" and abs(rel_y) > 100.0:
        # If behind (rel_y < 0) we want to lower orbit -> retrograde -> negative thrust_y
        if rel_y < -50.0:
            # schedule a burn for a short duration (proportional to distance)
            burn_duration = _clamp(abs(rel_y) / 50.0, 2.0, 12.0)
            # If not currently burning, set burn_until timer
            if _persist["burn_until"] < _persist["t"] + 1e-6:
                _persist["burn_until"] = _persist["t"] + burn_duration
            if _persist["t"] < _persist["burn_until"]:
                # apply retrograde proportional throttle
                thrust_cmd[1] = -0.6 * _clamp(abs(rel_y) / 500.0, 0.1, 1.0)
            else:
                # small corrective ay_des used
                thrust_cmd[1] = ay_des / max_acc_per_throttle
        elif rel_y > 50.0:
            # ahead: small prograde burn to raise orbit and slow relative approach
            thrust_cmd[1] = 0.25 * _clamp(abs(rel_y) / 500.0, 0.1, 1.0)
        else:
            thrust_cmd[1] = ay_des / max_acc_per_throttle
    else:
        thrust_cmd[1] = ay_des / max_acc_per_throttle

    # radial and cross-track direct PD controls
    thrust_cmd[0] = ax_des / max_acc_per_throttle
    thrust_cmd[2] = az_des / max_acc_per_throttle

    # If within close approach zone, scale down aggressively for smoothness
    if r < 20.0:
        scale = _clamp(r / 20.0, 0.05, 1.0)
        thrust_cmd = [_clamp(c * scale, -1.0, 1.0) for c in thrust_cmd]

    # Final micro approach: ensure approach speed along y is very small
    if r < DIST_FINAL:
        # reduce along-track command to get approach speed under target
        # aim to null relative velocity if closing too fast
        if abs(rel_vy) > TARGET_CLOSE_SPEED:
            # apply opposite thrust to reduce speed
            required_ay = -_clamp(rel_vy * 0.6, -0.05, 0.05)
            thrust_cmd[1] = required_ay / max_acc_per_throttle

    # Fuel-aware scaling: if fuel low, be conservative
    fuel_kg = state.get("fuel_remaining_kg", None)
    if fuel_kg is not None and fuel_kg < 60.0:
        # reduce maximum throttle to conserve fuel
        for i in range(3):
            thrust_cmd[i] *= 0.5

    # Clip thrust outputs -1..1
    for i in range(3):
        thrust_cmd[i] = _clamp(thrust_cmd[i], -1.0, 1.0)

    # Save last thrust
    _persist["last_thrust_cmd"] = thrust_cmd[:]

    return thrust_cmd

# Main entry point
def autopilot_step(state: dict, dt: float) -> dict:
    """
    Single-step autopilot. Call repeatedly at dt intervals. Persistent variables stored in module-level _persist.
    """

    # Sanity checks on dt
    if dt <= 0:
        dt = 0.1

    _persist["t"] = _persist.get("t", 0.0) + dt

    # 1) Estimate relative position/velocity using fusion
    pos_est, vel_est = _fuse_position_velocity(state, dt)

    # 2) Attitude control (rates)
    att_roll, att_pitch, att_yaw = _compute_attitude_rates(state, dt)

    # 3) Thrust control law
    thrust_x, thrust_y, thrust_z = _control_law(pos_est, vel_est, state, dt)

    # 4) Safety overrides and final clamps
    # If docking camera inactive but within close distance, be extra conservative
    r = _vec_len(pos_est)
    camera_active = bool(state.get("dock_camera_active", False))
    if r < 30.0 and not camera_active:
        # reduce thrust to prevent surprises
        thrust_x *= 0.6
        thrust_y *= 0.6
        thrust_z *= 0.6
        # also lower attitude commands
        att_roll *= 0.8
        att_pitch *= 0.8
        att_yaw *= 0.8

    # If we have a large relative speed, cap thrust magnitudes to avoid overcorrection
    rel_speed = _vec_len(vel_est)
    if rel_speed > 5.0:
        scale = _clamp(5.0 / rel_speed, 0.2, 1.0)
        thrust_x *= scale
        thrust_y *= scale
        thrust_z *= scale

    # If within final success window and speeds extremely small, zero commands to avoid bump
    if r < DOCK_SUCCESS_RANGE and rel_speed < DOCK_SUCCESS_SPEED:
        thrust_x = 0.0
        thrust_y = 0.0
        thrust_z = 0.0
        att_roll = 0.0
        att_pitch = 0.0
        att_yaw = 0.0

    # Clip final outputs strictly
    thrust_x = _clamp(thrust_x, -1.0, 1.0)
    thrust_y = _clamp(thrust_y, -1.0, 1.0)
    thrust_z = _clamp(thrust_z, -1.0, 1.0)
    att_roll = _clamp(att_roll, -_ATT_RATE_MAX, _ATT_RATE_MAX)
    att_pitch = _clamp(att_pitch, -_ATT_RATE_MAX, _ATT_RATE_MAX)
    att_yaw = _clamp(att_yaw, -_ATT_RATE_MAX, _ATT_RATE_MAX)

    # Record some telemetry into persistent store for future steps
    _persist["pos_est"] = pos_est
    _persist["vel_est"] = vel_est
    _persist["last_update_time"] = time.time()

    # Build output
    out = {
        "thrust_x": thrust_x,
        "thrust_y": thrust_y,
        "thrust_z": thrust_z,
        "att_rate_roll": att_roll,
        "att_rate_pitch": att_pitch,
        "att_rate_yaw": att_yaw,
    }

    return out
