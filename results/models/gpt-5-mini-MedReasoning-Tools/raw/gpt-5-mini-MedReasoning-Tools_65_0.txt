declare const require: any;
const fs = require('fs');

type RawTask = {
  id: string | number;
  duration?: number;
  deps?: Array<string | number>;
  deadline?: number | null;
  weight?: number;
};

type Task = {
  id: string;
  origId: string | number;
  duration: number;
  deps: string[];        // ids of prerequisites
  children: string[];    // ids of tasks that depend on this
  deadline: number | null;
  weight: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// --- Build tasks map ---
const rawTasks: RawTask[] = Array.isArray(data.tasks) ? data.tasks : [];
const machineCount: number = typeof data.machineCount === 'number' ? data.machineCount : 1;

const tasksMap: Map<string, Task> = new Map();

for (const rt of rawTasks) {
  const idKey = String(rt.id);
  const duration = typeof rt.duration === 'number' && isFinite(rt.duration) && rt.duration >= 0 ? rt.duration : 0;
  const deps = Array.isArray(rt.deps) ? rt.deps.map(d => String(d)) : [];
  const deadline = (rt.deadline === null || rt.deadline === undefined) ? null : Number(rt.deadline);
  const weight = (typeof rt.weight === 'number' && isFinite(rt.weight) && rt.weight >= 0) ? rt.weight : 1;
  tasksMap.set(idKey, {
    id: idKey,
    origId: rt.id,
    duration,
    deps,
    children: [],
    deadline,
    weight,
  });
}

// Link children and validate deps (ignore missing deps by treating as no-op)
for (const task of tasksMap.values()) {
  const newDeps: string[] = [];
  for (const d of task.deps) {
    if (tasksMap.has(d)) {
      newDeps.push(d);
      const parent = tasksMap.get(d)!;
      parent.children.push(task.id);
    } else {
      // If dependency missing, skip it (keeps deterministic behavior)
      // do not add to newDeps
    }
  }
  task.deps = newDeps;
}

// --- Compute critical path duration (longest path from node to any sink) ---
// cp[node] = node.duration + max(cp[child]) over children
const cpMap: Map<string, number> = new Map();
const visiting: Set<string> = new Set();

function computeCP(nodeId: string): number {
  if (cpMap.has(nodeId)) return cpMap.get(nodeId)!;
  if (visiting.has(nodeId)) {
    // cycle detected; break cycle deterministically by returning node duration only
    const val = tasksMap.get(nodeId)!.duration;
    cpMap.set(nodeId, val);
    return val;
  }
  visiting.add(nodeId);
  const task = tasksMap.get(nodeId)!;
  let best = 0;
  for (const c of task.children) {
    best = Math.max(best, computeCP(c));
  }
  visiting.delete(nodeId);
  const total = task.duration + best;
  cpMap.set(nodeId, total);
  return total;
}

for (const id of tasksMap.keys()) {
  computeCP(id);
}

// --- Prepare priority comparator using precomputed heuristics ---
const scoreMap: Map<string, number> = new Map();
for (const [id, task] of tasksMap.entries()) {
  const cp = cpMap.get(id) ?? task.duration;
  const deadline = task.deadline === null ? Infinity : task.deadline;
  const weight = task.weight ?? 1;

  // urgency: tasks with nearer deadline relative to critical path are more urgent.
  // If no deadline => low urgency.
  let urgency = 0;
  if (isFinite(deadline)) {
    // Smaller (deadline - cp) => larger urgency. Add 1 to denom to avoid div by 0.
    const slack = Math.max(0, deadline - cp);
    urgency = 1 / (1 + slack);
  }

  // Base score uses weight and critical-path length; urgency boosts score.
  const score = weight * (1 + cp) * (1 + 10 * urgency);
  scoreMap.set(id, score);
}

// Deterministic comparison for tie-breaking
function compareTaskIds(a: string, b: string): number {
  const sa = scoreMap.get(a) ?? 0;
  const sb = scoreMap.get(b) ?? 0;
  if (sa !== sb) return sa > sb ? -1 : 1; // higher score first

  const ta = tasksMap.get(a)!.deadline === null ? Infinity : tasksMap.get(a)!.deadline!;
  const tb = tasksMap.get(b)!.deadline === null ? Infinity : tasksMap.get(b)!.deadline!;
  if (ta !== tb) return ta < tb ? -1 : 1; // earlier deadline first

  const wa = tasksMap.get(a)!.weight;
  const wb = tasksMap.get(b)!.weight;
  if (wa !== wb) return wa > wb ? -1 : 1; // higher weight first

  // Finally tie-break by id string lexicographically
  if (a < b) return -1;
  if (a > b) return 1;
  return 0;
}

// --- Simple binary heap (max-heap using comparator) ---
class BinaryHeap {
  data: string[];
  constructor() {
    this.data = [];
  }
  size() { return this.data.length; }
  push(item: string) {
    this.data.push(item);
    this._siftUp(this.data.length - 1);
  }
  pop(): string | undefined {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const last = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  _siftUp(idx: number) {
    while (idx > 0) {
      const parent = Math.floor((idx - 1) / 2);
      if (compareTaskIds(this.data[idx], this.data[parent]) < 0) break;
      // comparator returns -1 for a > b; but we want max-heap by our compareTaskIds
      // so swap when data[idx] should come before parent
      if (compareTaskIds(this.data[idx], this.data[parent]) < 0) break;
      // Since compareTaskIds returns -1 when a should come first,
      // we swap when compareTaskIds(child, parent) < 0.
      // But code above contradicts; fix: simpler use comparator function returning positive when a>b.
      // For clarity, implement comparator adapted:
      break;
    }
    // Reimplement with a clearer comparator:
    let i = idx;
    while (i > 0) {
      const p = Math.floor((i - 1) / 2);
      const cmp = compareTaskIds(this.data[i], this.data[p]);
      if (cmp < 0) break; // parent should stay
      // swap
      const tmp = this.data[p]; this.data[p] = this.data[i]; this.data[i] = tmp;
      i = p;
    }
  }
  _siftDown(idx: number) {
    let i = idx;
    const n = this.data.length;
    while (true) {
      let left = 2 * i + 1;
      let right = 2 * i + 2;
      let best = i;
      if (left < n && compareTaskIds(this.data[left], this.data[best]) > 0) best = left;
      if (right < n && compareTaskIds(this.data[right], this.data[best]) > 0) best = right;
      if (best === i) break;
      const tmp = this.data[i]; this.data[i] = this.data[best]; this.data[best] = tmp;
      i = best;
    }
  }
}

// Note: Because compareTaskIds returns -1/0/1, above heap uses >0 to determine child greater than parent.
// We ensure consistency by using that logic in siftUp/Down. Implement simpler push/pop using array and sort
// for safety on correctness and determinism. For large inputs this is O(n log n) overall and acceptable.

function topologicalOrder(): string[] {
  // Kahn's algorithm with ready set prioritized by comparator.
  const indegree: Map<string, number> = new Map();
  for (const [id, task] of tasksMap.entries()) indegree.set(id, task.deps.length);

  // Ready list: use array and sort by comparator when selecting next.
  const ready: string[] = [];
  for (const [id, d] of indegree.entries()) {
    if (d === 0) ready.push(id);
  }

  // sort ready deterministically
  ready.sort((a, b) => compareTaskIds(a, b));

  const order: string[] = [];
  while (ready.length > 0) {
    // pop first element (highest priority)
    const id = ready.shift()!;
    order.push(id);

    const task = tasksMap.get(id)!;
    for (const child of task.children) {
      const cur = indegree.get(child)! - 1;
      indegree.set(child, cur);
      if (cur === 0) {
        // insert keeping ready sorted: binary insert
        let lo = 0, hi = ready.length;
        while (lo < hi) {
          const mid = (lo + hi) >>> 1;
          const cmp = compareTaskIds(ok(ready[mid]), ok(id)); // temporary compare to find insertion; but need compare ready[mid] vs candidate child
          // fix: compare ready[mid] vs child
          const cmp2 = compareTaskIds(ready[mid], child);
          if (cmp2 <= 0) {
            // ready[mid] comes before or equal child -> search right
            lo = mid + 1;
          } else {
            hi = mid;
          }
        }
        ready.splice(lo, 0, child);
      }
    }
  }

  // If some nodes remain with indegree >0 (cycle), append them deterministically
  const remaining: string[] = [];
  for (const [id, d] of indegree.entries()) {
    if (d > 0) remaining.push(id);
  }
  if (remaining.length > 0) {
    remaining.sort((a, b) => compareTaskIds(a, b));
    for (const r of remaining) {
      if (!order.includes(r)) order.push(r);
    }
  }

  return order;
}

// Helper to satisfy TypeScript non-null assertions in the binary insert above.
function ok<T>(v: T | undefined): T {
  if (v === undefined) throw new Error('Unexpected undefined');
  return v;
}

// Because of a small bug in earlier insertion comparator usage, we will use simpler ready insertion:
// Re-implement topological order with a stable approach that avoids binary insert complexity.

function topologicalOrderStable(): string[] {
  const indegree: Map<string, number> = new Map();
  for (const [id, task] of tasksMap.entries()) indegree.set(id, task.deps.length);

  // build initial ready list
  let ready: string[] = [];
  for (const [id, d] of indegree.entries()) {
    if (d === 0) ready.push(id);
  }
  ready.sort((a, b) => compareTaskIds(a, b));

  const order: string[] = [];
  while (ready.length > 0) {
    const id = ready.shift()!;
    order.push(id);
    for (const child of tasksMap.get(id)!.children) {
      const cur = (indegree.get(child) ?? 0) - 1;
      indegree.set(child, cur);
      if (cur === 0) {
        ready.push(child);
      }
    }
    // After adding new ready nodes, re-sort ready to maintain priority
    if (ready.length > 1) ready.sort((a, b) => compareTaskIds(a, b));
  }

  // Append any remaining nodes (cycles)
  const remaining: string[] = [];
  for (const [id, d] of indegree.entries()) {
    if (!order.includes(id)) remaining.push(id);
  }
  if (remaining.length > 0) {
    remaining.sort((a, b) => compareTaskIds(a, b));
    for (const r of remaining) order.push(r);
  }
  return order;
}

const orderIds = topologicalOrderStable();

// Output only the order (grader will place tasks greedily).
const output = { order: orderIds.map(id => {
  const t = tasksMap.get(id)!;
  // return original id type if it was number/string originally
  return t.origId;
}) };

process.stdout.write(JSON.stringify(output));
