```rust
use std::io::{self, Read};

fn main() {
    // Read entire stdin
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }

    // size of each partition
    let k = n / 2;
    if k == 0 {
        println!("0\n");
        return;
    }

    // We'll enumerate all subsets of size k but fix vertex 0 in partition A to cut symmetry.
    // That means we choose k-1 vertices out of {1..n-1}.
    let rest_n = n - 1;
    let r = k - 1;

    // Best result tracking
    let mut best_cut = std::usize::MAX;
    let mut best_selection: Vec<usize> = Vec::new();

    // Helper to compute cut given boolean selection array
    let compute_cut_bool = |sel: &Vec<bool>| -> usize {
        let mut cut = 0usize;
        for &(u, v) in &edges {
            if sel[u] ^ sel[v] {
                cut += 1;
            }
        }
        cut
    };

    // If n <= 128 we can use u128 bitmasks + Gosper to iterate combinations faster.
    if n <= 128 {
        // Precompute bit masks for edges as u128 for quick checks (safe because n <= 128)
        let mut edge_mask_u: Vec<u128> = Vec::with_capacity(m);
        let mut edge_mask_v: Vec<u128> = Vec::with_capacity(m);
        for &(u, v) in &edges {
            edge_mask_u.push(1u128 << u);
            edge_mask_v.push(1u128 << v);
        }

        if r > rest_n {
            // impossible, but handle gracefully
        } else if r == 0 {
            // Only selection is {0}
            let mask = 1u128; // only vertex 0 selected
            let mut cut = 0usize;
            for i in 0..m {
                let a = (mask & edge_mask_u[i]) != 0;
                let b = (mask & edge_mask_v[i]) != 0;
                if a ^ b { cut += 1; }
            }
            best_cut = cut;
            best_selection = vec![0usize];
        } else {
            // iterate all subsets of size r over the rest_n bits (positions 0..rest_n-1 represent original vertices 1..n-1)
            // Start mask with r ones.
            let mut mask: u128 = if r == 0 { 0 } else { ((1u128 << r) - 1) };
            let limit: u128 = 1u128 << rest_n;

            while mask < limit {
                // build real mask by shifting left 1 (to make room for fixed vertex 0) and set bit 0
                let real_mask = (mask << 1) | 1u128;

                // compute cut by checking each edge using precomputed masks
                let mut cut = 0usize;
                for i in 0..m {
                    let a = (real_mask & edge_mask_u[i]) != 0;
                    let b = (real_mask & edge_mask_v[i]) != 0;
                    if a ^ b { cut += 1; }
                    // small early exit if already worse than best
                    if cut >= best_cut {
                        break;
                    }
                }

                if cut < best_cut {
                    best_cut = cut;
                    // record selected vertices
                    best_selection.clear();
                    // bit 0 is always selected
                    best_selection.push(0usize);
                    // for bits 1..n-1 read from mask
                    let mut mm = mask;
                    let mut idx = 1usize;
                    while mm != 0 {
                        if (mm & 1u128) != 0 {
                            best_selection.push(idx);
                        }
                        mm >>= 1;
                        idx += 1;
                    }
                    // there may be trailing zeros: still need to increment idx to reach n-1 but not necessary
                }

                if best_cut == 0 {
                    break; // cannot get better than 0
                }

                // Gosper's hack to get next combination of same popcount for u128
                let c = mask & (!mask + 1);
                let rsum = mask + c;
                if rsum == 0 { break; } // overflow guard (shouldn't happen here)
                mask = (((rsum ^ mask) >> 2) / c) | rsum;
            }
        }
    } else {
        // Fallback: n > 128 (unlikely in practice). Use iterative combination of indices and boolean vector.
        // Generate all combinations of r picks from 1..=n-1
        if r > rest_n {
            // impossible
        } else if r == 0 {
            let mut sel = vec![false; n];
            sel[0] = true;
            let cut = compute_cut_bool(&sel);
            best_cut = cut;
            best_selection = vec![0usize];
        } else {
            // initial combination: [1,2,...,r]
            let mut comb: Vec<usize> = (1..=r).collect();
            loop {
                // build selection bool vector
                let mut sel = vec![false; n];
                sel[0] = true;
                for &x in &comb { sel[x] = true; }
                let cut = compute_cut_bool(&sel);
                if cut < best_cut {
                    best_cut = cut;
                    best_selection.clear();
                    best_selection.push(0usize);
                    best_selection.extend_from_slice(&comb);
                }
                if best_cut == 0 { break; }

                // next combination lexicographically
                let mut i = r;
                while i > 0 {
                    i -= 1;
                    if comb[i] != i + rest_n - r + 1 {
                        break;
                    }
                }
                if comb[0] == rest_n - r + 1 { break; }
                comb[i] += 1;
                for j in i+1..r {
                    comb[j] = comb[j-1] + 1;
                }
            }
        }
    }

    // As a final check (safety), ensure we have exactly k vertices in best_selection; if not, build from mask or error.
    if best_selection.len() != k {
        // try to reconstruct if possible (should not happen)
        // If not found, output a trivial valid partition: first k vertices
        best_selection.clear();
        for i in 0..k { best_selection.push(i); }
        // recompute cut
        let mut sel = vec![false; n];
        for &v in &best_selection { sel[v] = true; }
        best_cut = compute_cut_bool(&sel);
    }

    // Output
    println!("{}", best_cut);
    for (i, v) in best_selection.iter().enumerate() {
        if i + 1 == best_selection.len() {
            print!("{}", v);
        } else {
            print!("{} ", v);
        }
    }
    println!();
}
```
