#include <bits/stdc++.h>
using namespace std;

/*
  Exact solver for:
    maximize sum(c_i * x_i)
    subject to x_u + x_v <= 1 for given pairs (u,v)
    x_i in {0,1}

  Approach:
  - Drop all variables with non-positive weight (they never help the objective).
  - Decompose the original conflict graph (edges define conflicts) into connected components;
    the problem separates across components so we solve each component independently.
  - For each component we run an exact branch-and-bound search for the maximum-weight
    independent set in that component's induced subgraph.
    Representation: masks on up to 64 vertices per component (uint64_t).
  - Heuristics:
    * Initial greedy solution to get a good incumbent.
    * Branch on a vertex chosen as a pivot (max neighbors within current candidate set).
    * Prune when current + sum(weights of remaining candidates) <= best.
    * Memoization (dominance): if we've visited the same candidate mask with an equal
      or better current_weight before, skip exploring it.
*/

using int64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<int64> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    // Build original conflict adjacency lists
    vector<vector<int>> adj(n);
    vector<vector<char>> conflict(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (!conflict[u][v]) {
            conflict[u][v] = conflict[v][u] = 1;
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Keep track of final selection for original indices
    vector<int> solution(n, 0);

    // Collect indices with positive weight (non-positive weights cannot help)
    vector<int> posIdx;
    for (int i = 0; i < n; ++i) if (c[i] > 0) posIdx.push_back(i);

    // If no positive weights, optimal is 0 with all zeros
    if (posIdx.empty()) {
        cout << 0 << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << "\n";
        return 0;
    }

    // Build induced subgraph on positive vertices and find connected components (on original conflict graph)
    int pn = (int)posIdx.size();
    vector<int> posMap(n, -1);
    for (int i = 0; i < pn; ++i) posMap[posIdx[i]] = i;

    vector<vector<int>> posAdj(pn);
    for (int ii = 0; ii < pn; ++ii) {
        int orig = posIdx[ii];
        for (int nb : adj[orig]) {
            if (posMap[nb] != -1) posAdj[ii].push_back(posMap[nb]);
        }
    }

    vector<char> visited(pn, 0);
    vector<vector<int>> components;
    for (int i = 0; i < pn; ++i) {
        if (visited[i]) continue;
        // BFS/DFS to get component
        vector<int> comp;
        stack<int> st;
        st.push(i);
        visited[i] = 1;
        while (!st.empty()) {
            int v = st.top(); st.pop();
            comp.push_back(v);
            for (int nb : posAdj[v]) if (!visited[nb]) { visited[nb] = 1; st.push(nb); }
        }
        components.push_back(comp);
    }

    int64 totalBest = 0;

    // Solve each component independently
    for (const auto &comp : components) {
        int sz = (int)comp.size();
        // Map local index -> original posIdx index -> original index
        vector<int> locToPosIdx(sz), locToOrig(sz);
        for (int i = 0; i < sz; ++i) {
            locToPosIdx[i] = comp[i];
            locToOrig[i] = posIdx[comp[i]];
        }

        // If component size is 1, trivially take the vertex (positive weight)
        if (sz == 1) {
            solution[locToOrig[0]] = 1;
            totalBest += c[locToOrig[0]];
            continue;
        }

        // Build neighbor masks within the component for original conflict graph
        // neighborMask[v] has bits set for vertices in the component that conflict with v (original edges)
        vector<uint64_t> neighborMask(sz, 0ULL);
        vector<int64> w(sz);
        for (int i = 0; i < sz; ++i) {
            int orig = locToOrig[i];
            w[i] = c[orig];
            for (int nb : adj[orig]) {
                int mapped = posMap[nb];
                if (mapped != -1) {
                    // find local index of mapped
                    // create reverse map: posIdx index -> local index
                }
            }
        }
        // Build reverse mapping: posIdx index -> local index in this component
        unordered_map<int,int> posToLocal;
        posToLocal.reserve(sz*2);
        for (int i = 0; i < sz; ++i) posToLocal[locToPosIdx[i]] = i;
        for (int i = 0; i < sz; ++i) {
            int orig = locToOrig[i];
            uint64_t mask = 0ULL;
            for (int nb : adj[orig]) {
                int pidx = posMap[nb];
                if (pidx != -1) {
                    auto it = posToLocal.find(pidx);
                    if (it != posToLocal.end()) mask |= (1ULL << it->second);
                }
            }
            neighborMask[i] = mask;
        }

        // Initial greedy solution: pick vertex with largest weight, remove its neighbors, repeat
        {
            uint64_t S = 0;
            for (int i = 0; i < sz; ++i) S |= (1ULL << i);
            uint64_t chosenMask = 0;
            int64 greedySum = 0;
            while (S) {
                // pick vertex with maximum weight among S
                int bestv = -1;
                int64 bestw = LLONG_MIN;
                uint64_t tmp = S;
                while (tmp) {
                    uint64_t lb = tmp & -tmp;
                    int idx = __builtin_ctzll(lb);
                    tmp ^= lb;
                    if (w[idx] > bestw) { bestw = w[idx]; bestv = idx; }
                }
                if (bestv == -1) break;
                greedySum += w[bestv];
                chosenMask |= (1ULL << bestv);
                // remove bestv and its neighbors
                S &= ~(1ULL << bestv);
                S &= ~neighborMask[bestv];
            }
            totalBest += greedySum; // We'll add component best after search, so reset? Wait: we need to keep component best separate.
            // Oops above adds greedySum prematurely; fix by storing componentBest separately.
            totalBest -= greedySum; // undo; we will use greedySum to init incumbent in search
        }

        // Use DFS with branch-and-bound to find exact best for this component
        uint64_t fullMask = 0ULL;
        for (int i = 0; i < sz; ++i) fullMask |= (1ULL << i);

        // prepare initial incumbent via greedy (again properly)
        int64 incumbent = 0;
        uint64_t incumbentMask = 0;
        {
            uint64_t S = fullMask;
            uint64_t chosenMask = 0;
            int64 greedySum = 0;
            while (S) {
                int bestv = -1;
                int64 bestw = LLONG_MIN;
                uint64_t tmp = S;
                while (tmp) {
                    uint64_t lb = tmp & -tmp;
                    int idx = __builtin_ctzll(lb);
                    tmp ^= lb;
                    if (w[idx] > bestw) { bestw = w[idx]; bestv = idx; }
                }
                if (bestv == -1) break;
                greedySum += w[bestv];
                chosenMask |= (1ULL << bestv);
                S &= ~(1ULL << bestv);
                S &= ~neighborMask[bestv];
            }
            incumbent = greedySum;
            incumbentMask = chosenMask;
        }

        // We'll run DFS; use unordered_map for dominance pruning: store best seen current_weight for a given candidate mask
        unordered_map<uint64_t, int64> seen;
        seen.reserve(1 << min(20, sz)); // heuristic

        int64 bestWeight = incumbent;
        uint64_t bestMask = incumbentMask;

        // Precompute weight sums for single bits for speed
        vector<int64> weightByIndex(64, 0);
        for (int i = 0; i < sz; ++i) weightByIndex[i] = w[i];

        // DFS lambda
        function<void(uint64_t,int64,uint64_t)> dfs = [&](uint64_t S, int64 curW, uint64_t chosen) {
            if (S == 0) {
                if (curW > bestWeight) { bestWeight = curW; bestMask = chosen; }
                return;
            }
            // Dominance: if we've seen this candidate set with an equal or greater curW, prune
            auto it = seen.find(S);
            if (it != seen.end() && it->second >= curW) return;
            seen[S] = curW;

            // Compute sumRemaining
            int64 sumRem = 0;
            uint64_t tmp = S;
            while (tmp) {
                uint64_t lb = tmp & -tmp;
                int idx = __builtin_ctzll(lb);
                tmp ^= lb;
                sumRem += weightByIndex[idx];
            }
            if (curW + sumRem <= bestWeight) return;

            // Choose pivot: vertex with maximum neighbors inside S (to cut most in include branch)
            int pivot = -1;
            int maxdeg = -1;
            tmp = S;
            while (tmp) {
                uint64_t lb = tmp & -tmp;
                int idx = __builtin_ctzll(lb);
                tmp ^= lb;
                int deg = __builtin_popcountll(S & neighborMask[idx]);
                if (deg > maxdeg) { maxdeg = deg; pivot = idx; }
            }
            if (pivot == -1) {
                // shouldn't happen but treat as leaf
                if (curW > bestWeight) { bestWeight = curW; bestMask = chosen; }
                return;
            }

            // Branch 1: include pivot
            uint64_t includeS = S & ~(neighborMask[pivot] | (1ULL << pivot));
            dfs(includeS, curW + weightByIndex[pivot], chosen | (1ULL << pivot));

            // Branch 2: exclude pivot
            uint64_t excludeS = S & ~(1ULL << pivot);
            // quick check: if excluding pivot cannot improve, skip
            // (curW + sumRem - weight[pivot] <= bestWeight) -> prune
            if (curW + (sumRem - weightByIndex[pivot]) > bestWeight) {
                dfs(excludeS, curW, chosen);
            }
        };

        // run search
        dfs(fullMask, 0, 0);

        // bestMask/ bestWeight are component's solution
        totalBest += bestWeight;
        // set solution bits for original indices
        for (int i = 0; i < sz; ++i) {
            if (bestMask & (1ULL << i)) solution[locToOrig[i]] = 1;
        }
    }

    // Done, print result: objective value and full assignment x_0 ... x_n-1 (0/1).
    // Note: variables with non-positive weight were set to 0.
    int64 objective = 0;
    for (int i = 0; i < n; ++i) if (solution[i]) objective += c[i];
    cout << objective << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << solution[i];
    }
    cout << "\n";
    return 0;
}
