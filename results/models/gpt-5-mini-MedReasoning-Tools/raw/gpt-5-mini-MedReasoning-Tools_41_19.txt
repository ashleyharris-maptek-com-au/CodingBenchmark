; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %ubo %outColor
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %_struct_ubo "UBO"
OpName %lightPtr "lightPtr"
OpName %camPtr "camPtr"

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; Pointer types
%ptr_Input_v3f = OpTypePointer Input %v3f
%ptr_Input_v2f = OpTypePointer Input %v2f
%ptr_Output_v4f = OpTypePointer Output %v4f
%ptr_Uniform_ubo = OpTypePointer Uniform %_struct_ubo
%ptr_Uniform_v4f = OpTypePointer Uniform %v4f

; UBO struct
%_struct_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05000000074505806
%float_32 = OpConstant %float 32.0

%vec3_zero = OpConstantComposite %v3f %float_0 %float_0 %float_0
%vec3_one = OpConstantComposite %v3f %float_1 %float_1 %float_1
%ambient_c = OpConstantComposite %v3f %float_0_1 %float_0_05 %float_0_05
%specular_c = %vec3_one

; Variables (inputs/outputs/uniform)
%inWorldPos = OpVariable %ptr_Input_v3f Input
%inNormal = OpVariable %ptr_Input_v3f Input
%inUV = OpVariable %ptr_Input_v2f Input
%inTangent = OpVariable %ptr_Input_v3f Input
%inColor = OpVariable %ptr_Input_v3f Input

%outColor = OpVariable %ptr_Output_v4f Output

%ubo = OpVariable %ptr_Uniform_ubo Uniform

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %_struct_ubo Block
OpMemberDecorate %_struct_ubo 0 Offset 0
OpMemberDecorate %_struct_ubo 1 Offset 64
OpMemberDecorate %_struct_ubo 2 Offset 128
OpMemberDecorate %_struct_ubo 3 Offset 192
OpMemberDecorate %_struct_ubo 4 Offset 208
OpMemberDecorate %_struct_ubo 5 Offset 224

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Import GLSL.std.450 extended instructions
%glsl = OpExtInstImport "GLSL.std.450"

; Main
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %inWorldPos
%n_raw = OpLoad %v3f %inNormal
%vcol = OpLoad %v3f %inColor

; Normalize normal
%N = OpExtInst %v3f %glsl Normalize %n_raw

; Load light position (UBO member 3)
%lightPtr = OpAccessChain %ptr_Uniform_v4f %ubo %int_3
%light_v4 = OpLoad %v4f %lightPtr
%light_x = OpCompositeExtract %float %light_v4 0
%light_y = OpCompositeExtract %float %light_v4 1
%light_z = OpCompositeExtract %float %light_v4 2
%lightPos3 = OpCompositeConstruct %v3f %light_x %light_y %light_z

; Load camera position (UBO member 4)
%camPtr = OpAccessChain %ptr_Uniform_v4f %ubo %int_4
%cam_v4 = OpLoad %v4f %camPtr
%cam_x = OpCompositeExtract %float %cam_v4 0
%cam_y = OpCompositeExtract %float %cam_v4 1
%cam_z = OpCompositeExtract %float %cam_v4 2
%camPos3 = OpCompositeConstruct %v3f %cam_x %cam_y %cam_z

; Compute L and V directions (normalize)
%toLight = OpFSub %v3f %lightPos3 %wp
%L = OpExtInst %v3f %glsl Normalize %toLight

%toCam = OpFSub %v3f %camPos3 %wp
%V = OpExtInst %v3f %glsl Normalize %toCam

; Compute diffuse factor = max(dot(N,L), 0)
%dotNL = OpExtInst %float %glsl Dot %N %L
%gt0 = OpFOrdGreaterThan %bool %dotNL %float_0
%diffFactor = OpSelect %float %gt0 %dotNL %float_0

; diffuse color = vertexColor * diffFactor
%diffuse = OpVectorTimesScalar %v3f %vcol %diffFactor

; Compute reflection R = reflect(-L, N)
%negL = OpFSub %v3f %vec3_zero %L
%R = OpExtInst %v3f %glsl Reflect %negL %N

; Compute specular factor = pow(max(dot(R,V),0), shininess)
%dotRV = OpExtInst %float %glsl Dot %R %V
%gt0b = OpFOrdGreaterThan %bool %dotRV %float_0
%specBase = OpSelect %float %gt0b %dotRV %float_0
%specPow = OpExtInst %float %glsl Pow %specBase %float_32

%specular = OpVectorTimesScalar %v3f %specular_c %specPow

; Sum ambient + diffuse + specular
%sum1 = OpFAdd %v3f %ambient_c %diffuse
%color3 = OpFAdd %v3f %sum1 %specular

; Clamp to [0,1]
%clamped = OpExtInst %v3f %glsl FClamp %color3 %vec3_zero %vec3_one

; Construct out vec4 (alpha = 1.0)
%alpha = %float_1
%outv4 = OpCompositeConstruct %v4f %clamped %alpha

; Store to output
OpStore %outColor %outv4

OpReturn
OpFunctionEnd
