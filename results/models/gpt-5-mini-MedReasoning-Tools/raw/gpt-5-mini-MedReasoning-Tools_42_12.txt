; SPIR-V
; Version: 1.0
; Assembled for Vulkan GLSL450, Geometry shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; External instruction set for GLSL helper functions
%glsl = OpExtInstImport "GLSL.std.450"

; -----------------------
; Types and constants
; -----------------------
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_thickness = OpConstant %float 0.015
%f_half_thickness = OpConstant %float 0.0075
%f_dim = OpConstant %float 0.3

; -----------------------
; gl_PerVertex input (array of 6 elements for InputTrianglesAdjacency)
; struct { vec4 Position; }
; -----------------------
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; -----------------------
; User-defined per-vertex arrays (size 6)
; layout(location=0) in vec3 worldPos[];
; layout(location=1) in vec3 normal[];
; layout(location=2) in vec3 color[];
; -----------------------
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6

%worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos Location 0

%normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %normal Location 1

%color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %color Location 2

; -----------------------
; Output: gl_PerVertex_out and outColor
; -----------------------
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; -----------------------
; Uniform buffer object (set=0,binding=0) - declared but unused
; same layout as described in prompt (placeholder)
; -----------------------
%mat4 = OpTypeMatrix %v4float 4
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v4float
OpDecorate %ubo_t Block
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; -----------------------
; Helper pointers (for dynamic access)
; -----------------------
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; -----------------------
; Main function
; -----------------------
%main = OpFunction %void None %func
%entry = OpLabel

; Load the 6 worldPos entries (vec3)
%ap0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
%wp0 = OpLoad %v3float %ap0
%ap1 = OpAccessChain %ptr_in_v3 %worldPos %int_1
%wp1 = OpLoad %v3float %ap1
%ap2 = OpAccessChain %ptr_in_v3 %worldPos %int_2
%wp2 = OpLoad %v3float %ap2
%ap3 = OpAccessChain %ptr_in_v3 %worldPos %int_3
%wp3 = OpLoad %v3float %ap3
%ap4 = OpAccessChain %ptr_in_v3 %worldPos %int_4
%wp4 = OpLoad %v3float %ap4
%ap5 = OpAccessChain %ptr_in_v3 %worldPos %int_5
%wp5 = OpLoad %v3float %ap5

; Load colors for main triangle vertices: indices 0,2,4
%ac0 = OpAccessChain %ptr_in_v3 %color %int_0
%col0 = OpLoad %v3float %ac0
%ac2 = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %ac2
%ac4 = OpAccessChain %ptr_in_v3 %color %int_4
%col4 = OpLoad %v3float %ac4

; Helper: compute dimmed colors = color * 0.3
%col0_dim = OpVectorTimesScalar %v3float %col0 %f_dim
%col2_dim = OpVectorTimesScalar %v3float %col2 %f_dim
%col4_dim = OpVectorTimesScalar %v3float %col4 %f_dim

; Load gl_in positions (vec4) for indices used in triangle: 0,2,4
%ptr_pv0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pv0 = OpLoad %v4float %ptr_pv0
%ptr_pv2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pv2 = OpLoad %v4float %ptr_pv2
%ptr_pv4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pv4 = OpLoad %v4float %ptr_pv4

; -----------------------
; Emit filled triangle (triangle strip: emit 3 vertices -> one triangle)
; Using triangle vertices indices 0,2,4 with dimmed colors
; -----------------------
; Vertex 0 (index 0)
%store_pos = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %store_pos %pv0
OpStore %outColor %col0_dim
OpEmitVertex

; Vertex 1 (index 2)
OpStore %store_pos %pv2
OpStore %outColor %col2_dim
OpEmitVertex

; Vertex 2 (index 4)
OpStore %store_pos %pv4
OpStore %outColor %col4_dim
OpEmitVertex

OpEndPrimitive

; -----------------------
; Helper function: Check equality vec3(a) == vec3(b)
; We'll inline checks for each edge:
; -----------------------

; For each edge we will:
; - determine endpoints (A,B) indices in main triangle: (0,2), (2,4), (4,0)
; - corresponding adjacent indices: 1,3,5
; - compute whether adjacent equals A or B (boundary heuristic)
; - compute quad offsets in clip-space and emit quad as triangle strip with white color
; -----------------------

; Prepare constants for white color
%white = OpConstantComposite %v3float %f_1 %f_1 %f_1

; ----- EDGE 0: between main indices 0 and 2, adjacent index 1 -----
; Compare wp1 with wp0 or wp2
; component-wise comparisons for equality: (x==x && y==y && z==z)
%wp1_x = OpCompositeExtract %float %wp1 0
%wp1_y = OpCompositeExtract %float %wp1 1
%wp1_z = OpCompositeExtract %float %wp1 2
%wp0_x = OpCompositeExtract %float %wp0 0
%wp0_y = OpCompositeExtract %float %wp0 1
%wp0_z = OpCompositeExtract %float %wp0 2
%wp2_x = OpCompositeExtract %float %wp2 0
%wp2_y = OpCompositeExtract %float %wp2 1
%wp2_z = OpCompositeExtract %float %wp2 2

%eq1_x = OpFOrdEqual %bool %wp1_x %wp0_x
%eq1_y = OpFOrdEqual %bool %wp1_y %wp0_y
%eq1_z = OpFOrdEqual %bool %wp1_z %wp0_z
%eq1_xy = OpLogicalAnd %bool %eq1_x %eq1_y
%eq1_xyz = OpLogicalAnd %bool %eq1_xy %eq1_z

%eq2_x = OpFOrdEqual %bool %wp1_x %wp2_x
%eq2_y = OpFOrdEqual %bool %wp1_y %wp2_y
%eq2_z = OpFOrdEqual %bool %wp1_z %wp2_z
%eq2_xy = OpLogicalAnd %bool %eq2_x %eq2_y
%eq2_xyz = OpLogicalAnd %bool %eq2_xy %eq2_z

%edge0_is_boundary = OpLogicalOr %bool %eq1_xyz %eq2_xyz

; If boundary, emit quad for edge 0 between pv0 and pv2 with thickness
%edge0_label = OpLabel
OpSelectionMerge %edge0_merge None
OpBranchConditional %edge0_is_boundary %edge0_then %edge0_merge

%edge0_then = OpLabel
; load pv0 and pv2 (already loaded earlier as %pv0 and %pv2)
; compute ndc coords: ndc = (x/w, y/w)
%pv0_x = OpCompositeExtract %float %pv0 0
%pv0_y = OpCompositeExtract %float %pv0 1
%pv0_z = OpCompositeExtract %float %pv0 2
%pv0_w = OpCompositeExtract %float %pv0 3
%pv2_x = OpCompositeExtract %float %pv2 0
%pv2_y = OpCompositeExtract %float %pv2 1
%pv2_z = OpCompositeExtract %float %pv2 2
%pv2_w = OpCompositeExtract %float %pv2 3

%recip_w0 = OpFDiv %float %f_1 %pv0_w
%recip_w2 = OpFDiv %float %f_1 %pv2_w

%ndc0_x = OpFMul %float %pv0_x %recip_w0
%ndc0_y = OpFMul %float %pv0_y %recip_w0
%ndc2_x = OpFMul %float %pv2_x %recip_w2
%ndc2_y = OpFMul %float %pv2_y %recip_w2

%ndc0 = OpCompositeConstruct %v2float %ndc0_x %ndc0_y
%ndc2 = OpCompositeConstruct %v2float %ndc2_x %ndc2_y

; dir = ndc2 - ndc0
%dir0 = OpFSub %v2float %ndc2 %ndc0

; normalize dir, then form perpendicular (-y, x)
%dir0_n = OpExtInst %v2float %glsl Normalize %dir0
%dir0_n_x = OpCompositeExtract %float %dir0_n 0
%dir0_n_y = OpCompositeExtract %float %dir0_n 1
%neg_dir0_n_y = OpFNegate %float %dir0_n_y
%perp0 = OpCompositeConstruct %v2float %neg_dir0_n_y %dir0_n_x

; normalize perp and scale by half thickness (in NDC)
%perp0_n = OpExtInst %v2float %glsl Normalize %perp0
%perp0_scaled = OpVectorTimesScalar %v2float %perp0_n %f_half_thickness

; compute offsets in clip-space for pv0 and pv2:
; new_clip.xy = (ndc +/- perp_scaled) * w
%ndc0_plus = OpFAdd %v2float %ndc0 %perp0_scaled
%ndc0_minus = OpFSub %v2float %ndc0 %perp0_scaled
%ndc2_plus = OpFAdd %v2float %ndc2 %perp0_scaled
%ndc2_minus = OpFSub %v2float %ndc2 %perp0_scaled

%new0p_x = OpCompositeExtract %float %ndc0_plus 0
%new0p_y = OpCompositeExtract %float %ndc0_plus 1
%new0m_x = OpCompositeExtract %float %ndc0_minus 0
%new0m_y = OpCompositeExtract %float %ndc0_minus 1
%new2p_x = OpCompositeExtract %float %ndc2_plus 0
%new2p_y = OpCompositeExtract %float %ndc2_plus 1
%new2m_x = OpCompositeExtract %float %ndc2_minus 0
%new2m_y = OpCompositeExtract %float %ndc2_minus 1

%clip0p_x = OpFMul %float %new0p_x %pv0_w
%clip0p_y = OpFMul %float %new0p_y %pv0_w
%clip0m_x = OpFMul %float %new0m_x %pv0_w
%clip0m_y = OpFMul %float %new0m_y %pv0_w
%clip2p_x = OpFMul %float %new2p_x %pv2_w
%clip2p_y = OpFMul %float %new2p_y %pv2_w
%clip2m_x = OpFMul %float %new2m_x %pv2_w
%clip2m_y = OpFMul %float %new2m_y %pv2_w

; construct v4 for each quad corner: preserve original z and w
%v0p = OpCompositeConstruct %v4float %clip0p_x %clip0p_y %pv0_z %pv0_w
%v0m = OpCompositeConstruct %v4float %clip0m_x %clip0m_y %pv0_z %pv0_w
%v2p = OpCompositeConstruct %v4float %clip2p_x %clip2p_y %pv2_z %pv2_w
%v2m = OpCompositeConstruct %v4float %clip2m_x %clip2m_y %pv2_z %pv2_w

; Emit quad as triangle strip with white color: order (v0p, v0m, v2p, v2m)
OpStore %store_pos %v0p
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v0m
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v2p
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v2m
OpStore %outColor %white
OpEmitVertex

OpEndPrimitive
OpBranch %edge0_merge

%edge0_merge = OpLabel

; ----- EDGE 1: between main indices 2 and 4, adjacent index 3 -----
%wp3_x = OpCompositeExtract %float %wp3 0
%wp3_y = OpCompositeExtract %float %wp3 1
%wp3_z = OpCompositeExtract %float %wp3 2

%eq3a_x = OpFOrdEqual %bool %wp3_x %wp2_x
%eq3a_y = OpFOrdEqual %bool %wp3_y %wp2_y
%eq3a_z = OpFOrdEqual %bool %wp3_z %wp2_z
%eq3a_xy = OpLogicalAnd %bool %eq3a_x %eq3a_y
%eq3a_xyz = OpLogicalAnd %bool %eq3a_xy %eq3a_z

%eq3b_x = OpFOrdEqual %bool %wp3_x %wp4_x
%eq3b_y = OpFOrdEqual %bool %wp3_y %wp4_y
%eq3b_z = OpFOrdEqual %bool %wp3_z %wp4_z
%eq3b_xy = OpLogicalAnd %bool %eq3b_x %eq3b_y
%eq3b_xyz = OpLogicalAnd %bool %eq3b_xy %eq3b_z

%edge1_is_boundary = OpLogicalOr %bool %eq3a_xyz %eq3b_xyz

OpSelectionMerge %edge1_merge None
OpBranchConditional %edge1_is_boundary %edge1_then %edge1_merge

%edge1_then = OpLabel
; endpoints pv2 and pv4 (already loaded)
%pv2_x2 = OpCompositeExtract %float %pv2 0
%pv2_y2 = OpCompositeExtract %float %pv2 1
%pv2_z2 = OpCompositeExtract %float %pv2 2
%pv2_w2 = OpCompositeExtract %float %pv2 3
%pv4_x2 = OpCompositeExtract %float %pv4 0
%pv4_y2 = OpCompositeExtract %float %pv4 1
%pv4_z2 = OpCompositeExtract %float %pv4 2
%pv4_w2 = OpCompositeExtract %float %pv4 3

%recip_w2b = OpFDiv %float %f_1 %pv2_w2
%recip_w4b = OpFDiv %float %f_1 %pv4_w2

%ndc2_xb = OpFMul %float %pv2_x2 %recip_w2b
%ndc2_yb = OpFMul %float %pv2_y2 %recip_w2b
%ndc4_xb = OpFMul %float %pv4_x2 %recip_w4b
%ndc4_yb = OpFMul %float %pv4_y2 %recip_w4b

%ndc2b = OpCompositeConstruct %v2float %ndc2_xb %ndc2_yb
%ndc4b = OpCompositeConstruct %v2float %ndc4_xb %ndc4_yb

%dir1 = OpFSub %v2float %ndc4b %ndc2b
%dir1_n = OpExtInst %v2float %glsl Normalize %dir1
%dir1_n_x = OpCompositeExtract %float %dir1_n 0
%dir1_n_y = OpCompositeExtract %float %dir1_n 1
%neg_dir1_n_y = OpFNegate %float %dir1_n_y
%perp1 = OpCompositeConstruct %v2float %neg_dir1_n_y %dir1_n_x
%perp1_n = OpExtInst %v2float %glsl Normalize %perp1
%perp1_scaled = OpVectorTimesScalar %v2float %perp1_n %f_half_thickness

%ndc2_plus = OpFAdd %v2float %ndc2b %perp1_scaled
%ndc2_minus = OpFSub %v2float %ndc2b %perp1_scaled
%ndc4_plus = OpFAdd %v2float %ndc4b %perp1_scaled
%ndc4_minus = OpFSub %v2float %ndc4b %perp1_scaled

%n2px = OpCompositeExtract %float %ndc2_plus 0
%n2py = OpCompositeExtract %float %ndc2_plus 1
%n2mx = OpCompositeExtract %float %ndc2_minus 0
%n2my = OpCompositeExtract %float %ndc2_minus 1
%n4px = OpCompositeExtract %float %ndc4_plus 0
%n4py = OpCompositeExtract %float %ndc4_plus 1
%n4mx = OpCompositeExtract %float %ndc4_minus 0
%n4my = OpCompositeExtract %float %ndc4_minus 1

%c2px = OpFMul %float %n2px %pv2_w2
%c2py = OpFMul %float %n2py %pv2_w2
%c2mx = OpFMul %float %n2mx %pv2_w2
%c2my = OpFMul %float %n2my %pv2_w2
%c4px = OpFMul %float %n4px %pv4_w2
%c4py = OpFMul %float %n4py %pv4_w2
%c4mx = OpFMul %float %n4mx %pv4_w2
%c4my = OpFMul %float %n4my %pv4_w2

%v2p = OpCompositeConstruct %v4float %c2px %c2py %pv2_z2 %pv2_w2
%v2m = OpCompositeConstruct %v4float %c2mx %c2my %pv2_z2 %pv2_w2
%v4p = OpCompositeConstruct %v4float %c4px %c4py %pv4_z2 %pv4_w2
%v4m = OpCompositeConstruct %v4float %c4mx %c4my %pv4_z2 %pv4_w2

; Emit quad for edge1: (v2p, v2m, v4p, v4m)
OpStore %store_pos %v2p
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v2m
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v4p
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v4m
OpStore %outColor %white
OpEmitVertex

OpEndPrimitive
OpBranch %edge1_merge

%edge1_merge = OpLabel

; ----- EDGE 2: between main indices 4 and 0, adjacent index 5 -----
%wp5_x = OpCompositeExtract %float %wp5 0
%wp5_y = OpCompositeExtract %float %wp5 1
%wp5_z = OpCompositeExtract %float %wp5 2

%eq5a_x = OpFOrdEqual %bool %wp5_x %wp4_x
%eq5a_y = OpFOrdEqual %bool %wp5_y %wp4_y
%eq5a_z = OpFOrdEqual %bool %wp5_z %wp4_z
%eq5a_xy = OpLogicalAnd %bool %eq5a_x %eq5a_y
%eq5a_xyz = OpLogicalAnd %bool %eq5a_xy %eq5a_z

%eq5b_x = OpFOrdEqual %bool %wp5_x %wp0_x
%eq5b_y = OpFOrdEqual %bool %wp5_y %wp0_y
%eq5b_z = OpFOrdEqual %bool %wp5_z %wp0_z
%eq5b_xy = OpLogicalAnd %bool %eq5b_x %eq5b_y
%eq5b_xyz = OpLogicalAnd %bool %eq5b_xy %eq5b_z

%edge2_is_boundary = OpLogicalOr %bool %eq5a_xyz %eq5b_xyz

OpSelectionMerge %edge2_merge None
OpBranchConditional %edge2_is_boundary %edge2_then %edge2_merge

%edge2_then = OpLabel
; endpoints pv4 and pv0 (already loaded)
%pv4_x3 = OpCompositeExtract %float %pv4 0
%pv4_y3 = OpCompositeExtract %float %pv4 1
%pv4_z3 = OpCompositeExtract %float %pv4 2
%pv4_w3 = OpCompositeExtract %float %pv4 3
%pv0_x3 = OpCompositeExtract %float %pv0 0
%pv0_y3 = OpCompositeExtract %float %pv0 1
%pv0_z3 = OpCompositeExtract %float %pv0 2
%pv0_w3 = OpCompositeExtract %float %pv0 3

%recip_w4c = OpFDiv %float %f_1 %pv4_w3
%recip_w0c = OpFDiv %float %f_1 %pv0_w3

%ndc4_xc = OpFMul %float %pv4_x3 %recip_w4c
%ndc4_yc = OpFMul %float %pv4_y3 %recip_w4c
%ndc0_xc = OpFMul %float %pv0_x3 %recip_w0c
%ndc0_yc = OpFMul %float %pv0_y3 %recip_w0c

%ndc4c = OpCompositeConstruct %v2float %ndc4_xc %ndc4_yc
%ndc0c = OpCompositeConstruct %v2float %ndc0_xc %ndc0_yc

%dir2 = OpFSub %v2float %ndc0c %ndc4c
%dir2_n = OpExtInst %v2float %glsl Normalize %dir2
%dir2_n_x = OpCompositeExtract %float %dir2_n 0
%dir2_n_y = OpCompositeExtract %float %dir2_n 1
%neg_dir2_n_y = OpFNegate %float %dir2_n_y
%perp2 = OpCompositeConstruct %v2float %neg_dir2_n_y %dir2_n_x
%perp2_n = OpExtInst %v2float %glsl Normalize %perp2
%perp2_scaled = OpVectorTimesScalar %v2float %perp2_n %f_half_thickness

%ndc4_plus2 = OpFAdd %v2float %ndc4c %perp2_scaled
%ndc4_minus2 = OpFSub %v2float %ndc4c %perp2_scaled
%ndc0_plus2 = OpFAdd %v2float %ndc0c %perp2_scaled
%ndc0_minus2 = OpFSub %v2float %ndc0c %perp2_scaled

%n4p_x2 = OpCompositeExtract %float %ndc4_plus2 0
%n4p_y2 = OpCompositeExtract %float %ndc4_plus2 1
%n4m_x2 = OpCompositeExtract %float %ndc4_minus2 0
%n4m_y2 = OpCompositeExtract %float %ndc4_minus2 1
%n0p_x2 = OpCompositeExtract %float %ndc0_plus2 0
%n0p_y2 = OpCompositeExtract %float %ndc0_plus2 1
%n0m_x2 = OpCompositeExtract %float %ndc0_minus2 0
%n0m_y2 = OpCompositeExtract %float %ndc0_minus2 1

%c4p_x2 = OpFMul %float %n4p_x2 %pv4_w3
%c4p_y2 = OpFMul %float %n4p_y2 %pv4_w3
%c4m_x2 = OpFMul %float %n4m_x2 %pv4_w3
%c4m_y2 = OpFMul %float %n4m_y2 %pv4_w3
%c0p_x2 = OpFMul %float %n0p_x2 %pv0_w3
%c0p_y2 = OpFMul %float %n0p_y2 %pv0_w3
%c0m_x2 = OpFMul %float %n0m_x2 %pv0_w3
%c0m_y2 = OpFMul %float %n0m_y2 %pv0_w3

%v4p2 = OpCompositeConstruct %v4float %c4p_x2 %c4p_y2 %pv4_z3 %pv4_w3
%v4m2 = OpCompositeConstruct %v4float %c4m_x2 %c4m_y2 %pv4_z3 %pv4_w3
%v0p2 = OpCompositeConstruct %v4float %c0p_x2 %c0p_y2 %pv0_z3 %pv0_w3
%v0m2 = OpCompositeConstruct %v4float %c0m_x2 %c0m_y2 %pv0_z3 %pv0_w3

; Emit quad for edge2: (v4p2, v4m2, v0p2, v0m2)
OpStore %store_pos %v4p2
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v4m2
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v0p2
OpStore %outColor %white
OpEmitVertex

OpStore %store_pos %v0m2
OpStore %outColor %white
OpEmitVertex

OpEndPrimitive
OpBranch %edge2_merge

%edge2_merge = OpLabel

; End main
OpReturn
OpFunctionEnd
