static inline double orient2d(const Point& a, const Point& b, const Point& c) {
  const double abx = static_cast<double>(b.x) - static_cast<double>(a.x);
  const double aby = static_cast<double>(b.y) - static_cast<double>(a.y);
  const double acx = static_cast<double>(c.x) - static_cast<double>(a.x);
  const double acy = static_cast<double>(c.y) - static_cast<double>(a.y);
  return abx * acy - aby * acx;
}

static inline int sgn_eps(double v) {
  constexpr double eps = 1e-12;
  if (v > eps) return 1;
  if (v < -eps) return -1;
  return 0;
}

static inline bool segments_proper_intersect(const Point& a, const Point& b, const Point& c, const Point& d) {
  const int o1 = sgn_eps(orient2d(a, b, c));
  const int o2 = sgn_eps(orient2d(a, b, d));
  if (o1 == 0 || o2 == 0 || o1 == o2) return false;
  const int o3 = sgn_eps(orient2d(c, d, a));
  const int o4 = sgn_eps(orient2d(c, d, b));
  if (o3 == 0 || o4 == 0 || o3 == o4) return false;
  return true;
}

static inline std::uint64_t part1by1(std::uint32_t x) {
  std::uint64_t v = x;
  v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
  v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
  v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
  v = (v | (v << 2)) & 0x3333333333333333ULL;
  v = (v | (v << 1)) & 0x5555555555555555ULL;
  return v;
}

static inline std::uint64_t morton2d_u32(std::uint32_t x, std::uint32_t y) {
  return part1by1(x) | (part1by1(y) << 1);
}

struct MortonItem {
  std::uint64_t key;
  std::uint32_t id;
};

static void radix_sort_morton(std::vector<MortonItem>& a) {
  std::vector<MortonItem> b(a.size());
  constexpr int kRadix = 256;
  std::array<std::uint32_t, kRadix> cnt{};

  for (int pass = 0; pass < 8; ++pass) {
    cnt.fill(0);
    const int shift = pass * 8;
    for (const auto& it : a) ++cnt[static_cast<unsigned>((it.key >> shift) & 0xFFULL)];

    std::uint32_t sum = 0;
    for (int i = 0; i < kRadix; ++i) {
      const std::uint32_t c = cnt[i];
      cnt[i] = sum;
      sum += c;
    }

    for (const auto& it : a) {
      const unsigned bucket = static_cast<unsigned>((it.key >> shift) & 0xFFULL);
      b[cnt[bucket]++] = it;
    }
    a.swap(b);
  }
}

static std::vector<std::uint32_t> solve_exact_held_karp(const std::vector<Point>& pts) {
  const int n = static_cast<int>(pts.size());
  const int m = n - 1;
  const int full = (1 << m) - 1;
  const double INF = std::numeric_limits<double>::infinity();

  std::vector<double> dist(static_cast<std::size_t>(n) * static_cast<std::size_t>(n), 0.0);
  auto D = [&](int i, int j) -> double& { return dist[static_cast<std::size_t>(i) * n + j]; };
  for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n; ++j) {
      const double dx = static_cast<double>(pts[i].x) - static_cast<double>(pts[j].x);
      const double dy = static_cast<double>(pts[i].y) - static_cast<double>(pts[j].y);
      D(i, j) = std::sqrt(dx * dx + dy * dy);
    }
  }

  const std::size_t states = static_cast<std::size_t>(1u << m);
  std::vector<double> dp(states * static_cast<std::size_t>(m), INF);
  std::vector<int> parent(states * static_cast<std::size_t>(m), -1);
  auto idx = [&](int mask, int i) -> std::size_t { return static_cast<std::size_t>(mask) * m + i; };

  for (int i = 0; i < m; ++i) dp[idx(1 << i, i)] = D(0, i + 1);

  for (int mask = 1; mask <= full; ++mask) {
    for (int i = 0; i < m; ++i) {
      if (!(mask & (1 << i))) continue;
      const int pmask = mask ^ (1 << i);
      if (!pmask) continue;
      double best = dp[idx(mask, i)];
      int best_prev = parent[idx(mask, i)];
      for (int j = 0; j < m; ++j) {
        if (!(pmask & (1 << j))) continue;
        const double cand = dp[idx(pmask, j)] + D(j + 1, i + 1);
        if (cand < best) {
          best = cand;
          best_prev = j;
        }
      }
      dp[idx(mask, i)] = best;
      parent[idx(mask, i)] = best_prev;
    }
  }

  double best_cost = INF;
  int best_end = 0;
  for (int i = 0; i < m; ++i) {
    const double cand = dp[idx(full, i)] + D(i + 1, 0);
    if (cand < best_cost) {
      best_cost = cand;
      best_end = i;
    }
  }

  std::vector<std::uint32_t> rev;
  rev.reserve(static_cast<std::size_t>(m));
  int mask = full;
  int cur = best_end;
  while (cur != -1) {
    rev.push_back(static_cast<std::uint32_t>(cur + 1));
    const int prev = parent[idx(mask, cur)];
    mask ^= (1 << cur);
    cur = prev;
  }
  std::reverse(rev.begin(), rev.end());

  std::vector<std::uint32_t> tour;
  tour.reserve(static_cast<std::size_t>(n));
  tour.push_back(0);
  for (auto v : rev) tour.push_back(v);
  return tour;
}

    private static bool ShouldUseExactMatching(int n, int m, int oddCount)
    {
        // Exact bitmask DP on odd vertices, with Dijkstra from each odd vertex.
        // Safe for small odd-count and moderate graph sizes.

    private static List<int> BuildDuplicatesByTreeJoin(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] edgeId,
        List<int> oddVertices)
    {
        int[] parent = new int[n];
        int[] parentEdge = new int[n];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        int[] queue = new int[n];
        int[] order = new int[n];
        int qh = 0, qt = 0, oc = 0;
        parent[0] = 0;
        queue[qt++] = 0;

        while (qh < qt)
        {
            int vtx = queue[qh++];
            order[oc++] = vtx;
            for (int e = head[vtx]; e != -1; e = next[e])
            {
                int nb = to[e];
                if ((uint)nb >= (uint)n) continue;
                if (parent[nb] != -1) continue;
                parent[nb] = vtx;
                parentEdge[nb] = edgeId[e];
                queue[qt++] = nb;
            }
        }

        bool[] parity = new bool[n];
        for (int i = 0; i < oddVertices.Count; i++)
            parity[oddVertices[i]] = true;

        var duplicates = new List<int>(Math.Min(n, oddVertices.Count * 2 + 8));

        for (int i = oc - 1; i >= 1; i--) // skip root at order[0]
        {
            int vtx = order[i];
            if (parity[vtx])
            {
                int id = parentEdge[vtx];
                if (id >= 0) duplicates.Add(id);
                parity[parent[vtx]] = !parity[parent[vtx]];
            }
        }

        return duplicates;
    }

    private static List<int> BuildDuplicatesByExactMatching(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] weight,
        int[] edgeId,
        int[] u,
        int[] v,
        List<int> oddVertices)
    {
        int k = oddVertices.Count;
        long[,] distOdd = new long[k, k];
        int[][] parents = new int[k][];
        int[][] parentDirEdge = new int[k][];

        for (int i = 0; i < k; i++)
        {
            Dijkstra(
                n,
                head,
                to,
                next,
                weight,
                oddVertices[i],
                out long[] dist,
                out int[] parent,
                out int[] parentEdgeDir);

            parents[i] = parent;
            parentDirEdge[i] = parentEdgeDir;

            for (int j = 0; j < k; j++)
                distOdd[i, j] = dist[oddVertices[j]];
        }

        int fullMask = (1 << k) - 1;
        long[] memo = new long[1 << k];
        int[] choice = new int[1 << k];
        for (int i = 0; i < memo.Length; i++) memo[i] = long.MinValue;

        long Solve(int mask)
        {
            if (mask == 0) return 0;
            long cached = memo[mask];
            if (cached != long.MinValue) return cached;

            int i = BitOperations.TrailingZeroCount((uint)mask);
            int withoutI = mask & ~(1 << i);
            long best = long.MaxValue / 4;
            int bestJ = -1;

            for (int j = i + 1; j < k; j++)
            {
                int bit = 1 << j;
                if ((withoutI & bit) == 0) continue;
                int nextMask = withoutI & ~bit;
                long cost = distOdd[i, j] + Solve(nextMask);
                if (cost < best)
                {
                    best = cost;
                    bestJ = j;
                }
            }

            choice[mask] = bestJ;
            memo[mask] = best;
            return best;
        }

        _ = Solve(fullMask);

        var duplicates = new List<int>();
        int maskCur = fullMask;
        while (maskCur != 0)
        {
            int i = BitOperations.TrailingZeroCount((uint)maskCur);
            int j = choice[maskCur];
            if (j < 0) break;

            int src = oddVertices[i];
            int dst = oddVertices[j];

            int cur = dst;
            int[] parent = parents[i];
            int[] parentEdgeDir = parentDirEdge[i];
            while (cur != src)
            {
                int dir = parentEdgeDir[cur];
                if (dir < 0) break;
                duplicates.Add(edgeId[dir]);
                cur = parent[cur];
                if (cur < 0) break;
            }

            maskCur &= ~(1 << i);
            maskCur &= ~(1 << j);
        }

        return duplicates;
    }

    private static void Dijkstra(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] weight,
        int source,
        out long[] dist,
        out int[] parent,
        out int[] parentEdgeDir)
    {
        dist = new long[n];
        parent = new int[n];
        parentEdgeDir = new int[n];
        Array.Fill(dist, long.MaxValue / 4);
        Array.Fill(parent, -1);
        Array.Fill(parentEdgeDir, -1);

        dist[source] = 0;
        parent[source] = source;

        var heap = new MinHeap(Math.Min(1024, n));
        heap.Push(source, 0);

        while (heap.TryPop(out int vtx, out long dv))
        {
            if (dv != dist[vtx]) continue;

            for (int e = head[vtx]; e != -1; e = next[e])
            {
                int nb = to[e];
                long nd = dv + weight[e];
                if (nd < dist[nb])
                {
                    dist[nb] = nd;
                    parent[nb] = vtx;
                    parentEdgeDir[nb] = e;
                    heap.Push(nb, nd);
                }
            }
        }
    }

    private sealed class EulerGraph
    {
        private readonly int _n;
        private readonly int[] _head;
        private readonly int[] _to;
        private readonly int[] _next;
        private readonly int[] _rev;
        private readonly bool[] _used;
        private int _ptr;

        public EulerGraph(int n, int undirectedEdges)
        {
            _n = n;
            _head = new int[n];
            Array.Fill(_head, -1);
            int halfEdges = checked(undirectedEdges * 2);
            _to = new int[halfEdges];
            _next = new int[halfEdges];
            _rev = new int[halfEdges];
            _used = new bool[halfEdges];
            _ptr = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void AddEdge(int a, int b)
        {
            int i = _ptr++;
            int j = _ptr++;
            _to[i] = b;
            _to[j] = a;
            _rev[i] = j;
            _rev[j] = i;
            _next[i] = _head[a];
            _head[a] = i;
            _next[j] = _head[b];
            _head[b] = j;
        }

        public int[] EulerCircuit(int start)
        {
            int edges = _ptr / 2;
            int[] iter = new int[_n];
            Array.Copy(_head, iter, _n);

            int[] stack = new int[edges + 1];
            int sp = 0;
            stack[sp++] = start;

            int[] circuit = new int[edges + 1];
            int cp = 0;

            while (sp > 0)
            {
                int v = stack[sp - 1];
                int e = iter[v];
                while (e != -1 && _used[e]) e = _next[e];
                iter[v] = e;

                if (e == -1)
                {
                    sp--;
                    circuit[cp++] = v;
                    continue;
                }

                _used[e] = true;
                _used[_rev[e]] = true;
                iter[v] = _next[e];
                stack[sp++] = _to[e];
            }

            if (cp != circuit.Length)
            {
                // Graph not connected from start; truncate to what we found.
                Array.Resize(ref circuit, cp);
            }

            return circuit;
        }
    }

    private sealed class MinHeap
    {
        private int[] _node;
        private long[] _dist;
        private int _size;

        public MinHeap(int capacity)
        {
            _node = new int[Math.Max(4, capacity)];
            _dist = new long[_node.Length];
            _size = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Push(int node, long dist)
        {
            int i = _size++;
            if (i == _node.Length)
            {
                int newSize = _node.Length * 2;
                Array.Resize(ref _node, newSize);
                Array.Resize(ref _dist, newSize);
            }

            _node[i] = node;
            _dist[i] = dist;

            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_dist[p] <= dist) break;
                _node[i] = _node[p];
                _dist[i] = _dist[p];
                i = p;
            }

            _node[i] = node;
            _dist[i] = dist;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool TryPop(out int node, out long dist)
        {
            if (_size == 0)
            {
                node = 0;
                dist = 0;
                return false;
            }

            node = _node[0];
            dist = _dist[0];
            _size--;
            if (_size == 0) return true;

            int lastNode = _node[_size];
            long lastDist = _dist[_size];

            int i = 0;
            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= _size) break;
                int right = left + 1;
                int best = (right < _size && _dist[right] < _dist[left]) ? right : left;
                if (_dist[best] >= lastDist) break;
                _node[i] = _node[best];
                _dist[i] = _dist[best];
                i = best;
            }

            _node[i] = lastNode;
            _dist[i] = lastDist;
            return true;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len;
        private int _ptr;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryNextInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c == -1) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > 32 && c != -1)
            {
                x = (x * 10) + (c - '0');
                c = Read();
            }

            value = x * sign;
            return true;
        }
    }

    private sealed class FastOutput : IDisposable
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _pos;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void Flush()
        {
            if (_pos <= 0) return;
            _stream.Write(_buffer, 0, _pos);
            _pos = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            if (_pos == _buffer.Length) Flush();
            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace() => WriteByte((byte)' ');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine() => WriteByte((byte)'\n');

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[12];
            int len = 0;
            while (value > 0)
            {
                tmp[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }
            for (int i = len - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void Dispose() => Flush();
    }

fn quad_index(cx: f32, cy: f32, x: f32, y: f32) -> usize {

            return;

        nodes[node_idx].point = -1;

        }

        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {

            if c.is_ascii_digit() {
                val = val.saturating_mul(10).saturating_add((c - b'0') as usize);

    }

        if (i == ia || i == ib || i == ic || i == id) continue;
        order.push_back(i);
    }
    // Randomize insertion order for better expected behavior.
    std::mt19937_64 rng(0xC0D3C0D3C0D3C0D3ULL ^ static_cast<std::uint64_t>(n));
    std::shuffle(order.begin(), order.end(), rng);

    // Incremental hull.
    for (int idx : order) {
        const Vec3 p = P[idx];
        if (!isFiniteVec(p)) continue;

        std::vector<char> visible(faces.size(), 0);
        bool anyVisible = false;
        for (size_t fi = 0; fi < faces.size(); ++fi) {
            const Face& f = faces[fi];
            const double nd = dot(f.n, p) - f.d;
            // tolerance scaled by normal magnitude
            const double tol = epsLen * norm(f.n);

                anyVisible = true;
            }

        if (!anyVisible) continue;

        boundary.reserve(faces.size() * 3);

        auto addEdge = [&](int u, int v) {
            const std::uint64_t key = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(u)) << 32) |
                                      static_cast<std::uint32_t>(v);
            const std::uint64_t rev = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(v)) << 32) |
                                      static_cast<std::uint32_t>(u);
            auto it = boundary.find(rev);
            if (it != boundary.end()) {
                boundary.erase(it);
            } else {
                boundary.insert(key);
            }
        };

        for (size_t fi = 0; fi < faces.size(); ++fi) {
            if (!visible[fi]) continue;
            const Face& f = faces[fi];
            addEdge(f.a, f.b);
            addEdge(f.b, f.c);
            addEdge(f.c, f.a);
        }

        std::vector<Face> nextFaces;
        nextFaces.reserve(faces.size() - boundary.size() + boundary.size());
        for (size_t fi = 0; fi < faces.size(); ++fi) {
            if (!visible[fi]) nextFaces.push_back(faces[fi]);
        }

        for (const std::uint64_t key : boundary) {
            const int u = static_cast<int>(key >> 32);
            const int v = static_cast<int>(static_cast<std::uint32_t>(key));
            Face nf = makeFaceOriented(P, u, v, idx, interior, epsLen);
            if (nf.a >= 0) nextFaces.push_back(nf);
        }

        faces.swap(nextFaces);
        if (faces.size() < 4) return false;
    }

    facesOut.swap(faces);
    return true;
}

struct PlaneKey {
    std::int64_t nx{}, ny{}, nz{}, nt{};
    bool operator==(const PlaneKey& o) const { return nx == o.nx && ny == o.ny && nz == o.nz && nt == o.nt; }
};

struct PlaneKeyHash {
    size_t operator()(const PlaneKey& k) const noexcept {
        auto mix = [](std::uint64_t x) {
            x ^= x >> 33;
            x *= 0xff51afd7ed558ccdULL;
            x ^= x >> 33;
            x *= 0xc4ceb9fe1a85ec53ULL;
            x ^= x >> 33;
            return x;
        };
        std::uint64_t h = 0;
        h ^= mix(static_cast<std::uint64_t>(k.nx) + 0x9e3779b97f4a7c15ULL);
        h ^= mix(static_cast<std::uint64_t>(k.ny) + 0xD1B54A32D192ED03ULL);
        h ^= mix(static_cast<std::uint64_t>(k.nz) + 0x94D049BB133111EBULL);
        h ^= mix(static_cast<std::uint64_t>(k.nt) + 0x2545F4914F6CDD1DULL);
        return static_cast<size_t>(h);
    }
};

static std::vector<Plane> facesToUniquePlanes(const std::vector<Vec3>& P, const std::vector<Face>& faces, const Vec3& interior) {
    // Compute scale for tolerances.
    Vec3 mn = P[0], mx = P[0];
    for (const auto& v : P) {
        mn.x = std::min(mn.x, v.x);
        mn.y = std::min(mn.y, v.y);
        mn.z = std::min(mn.z, v.z);
        mx.x = std::max(mx.x, v.x);
        mx.y = std::max(mx.y, v.y);
        mx.z = std::max(mx.z, v.z);
    }
    const double scale = std::max(1.0, norm(mx - mn));
    const double epsN = 1e-6;
    const double epsT = 1e-8 * scale;

    std::unordered_map<PlaneKey, size_t, PlaneKeyHash> map;
    std::vector<Plane> planes;
    planes.reserve(faces.size());

    for (const Face& f : faces) {
        if (f.a < 0) continue;
        const Vec3 A = P[f.a];
        Vec3 n0 = f.n;
        const double len = norm(n0);
        if (!(len > 0)) continue;
        Vec3 n = n0 / len;
        double t = dot(n, A);

        // Safety: ensure interior is inside.
        if (dot(n, interior) > t + 1e-10 * scale) {
            n = n * -1.0;
            t = -t;
        }
        const double tRel = t - dot(n, interior); // offset relative to an interior point (more stable)

        PlaneKey key;
        key.nx = static_cast<std::int64_t>(std::llround(n.x / epsN));
        key.ny = static_cast<std::int64_t>(std::llround(n.y / epsN));
        key.nz = static_cast<std::int64_t>(std::llround(n.z / epsN));
        key.nt = static_cast<std::int64_t>(std::llround(tRel / epsT));

        auto it = map.find(key);
        if (it == map.end()) {
            Plane p;
            p.n = n;
            p.t = t;
            p.m = 0.0; // filled later
            map.emplace(key, planes.size());
            planes.push_back(p);
        } else {
            // Keep the tighter (smaller t) plane for safety.
            Plane& existing = planes[it->second];
            if (t < existing.t) {
                existing.n = n;
                existing.t = t;
            }
        }
    }

    return planes;
}

static double polyhedronVolumeFromPlanes(const std::vector<Vec3>& P, const std::vector<Face>& faces, const Vec3& interior) {
    // Decompose into tetrahedra (interior, A, B, C) for each triangular face.
    long double vol = 0.0L;
    for (const Face& f : faces) {
        if (f.a < 0) continue;
        const Vec3 A = P[f.a] - interior;
        const Vec3 B = P[f.b] - interior;
        const Vec3 C = P[f.c] - interior;
        const long double v6 = static_cast<long double>(dot(A, cross(B, C)));
        vol += std::fabs(v6) / 6.0L;
    }
    return static_cast<double>(vol);
}

static bool insideShrunk(const std::vector<Plane>& planes, const Vec3& center, double safetyMargin) {
    for (const Plane& pl : planes) {
        // Need n·center <= t - m - safetyMargin.
        if (dot(pl.n, center) > (pl.t - pl.m - safetyMargin)) return false;
    }
    return true;
}

        self.state = x;
        x
    }
}

fn parity_ok_for_path(grid: &Grid, require_cycle: bool) -> bool {
    let mut black = 0usize;
    let mut white = 0usize;
    for idx in 0..grid.free.len() {
        if !grid.free[idx] {
            continue;
        }
        let (x, y) = grid.xy(idx);
        if (x + y) % 2 == 0 {
            black += 1;
        } else {
            white += 1;
        }
    }
    let n = black + white;
    if n == 0 {
        return false;
    }
    if require_cycle {
        return n % 2 == 0 && black == white;
    }
    let diff = if black > white { black - white } else { white - black };
    if diff > 1 {
        return false;
    }
    if n % 2 == 0 {
        // endpoints have opposite colors => partitions must match
        black == white
    } else {
        // endpoints have same color; start at (0,0) is black
        black == white + 1
    }
}

fn degrees_ok(grid: &Grid, require_cycle: bool) -> bool {
    if require_cycle {
        for idx in 0..grid.free.len() {
            if grid.free[idx] && grid.degree(idx) < 2 {
                return false;
            }
        }
        return true;
    }

    // Necessary: at most 2 degree-1 vertices in the whole graph.
    let mut deg1 = 0usize;
    for idx in 0..grid.free.len() {
        if grid.free[idx] && grid.degree(idx) == 1 {
            deg1 += 1;
            if deg1 > 2 {
                return false;
            }
        }
    }
    true
}

fn is_unvisited_connected(grid: &Grid, visited: &[bool], unvisited_count: usize, stamp: &mut u32, seen: &mut [u32]) -> bool {
    if unvisited_count <= 1 {
        return true;
    }
    *stamp = stamp.wrapping_add(1).max(1);
    let s = *stamp;
    let mut start = None;
    for i in 0..grid.free.len() {
        if grid.free[i] && !visited[i] {
            start = Some(i);
            break;
        }
    }
    let Some(start) = start else { return true; };
    let mut q = VecDeque::new();
    q.push_back(start);
    seen[start] = s;
    let mut reached = 0usize;
    while let Some(v) = q.pop_front() {
        reached += 1;
        let len = grid.neigh_len[v] as usize;
        for k in 0..len {
            let nb = grid.neigh[v][k];
            if grid.free[nb] && !visited[nb] && seen[nb] != s {
                seen[nb] = s;
                q.push_back(nb);
            }
        }
    }
    reached == unvisited_count
}

fn validate_path(grid: &Grid, path: &[usize], require_cycle: bool) -> bool {
    if path.is_empty() {
        return false;
    }
    if path[0] != 0 {
        return false;
    }
    let mut seen = vec![false; grid.free.len()];
    for &idx in path {
        if idx >= grid.free.len() || !grid.free[idx] || seen[idx] {
            return false;
        }
        seen[idx] = true;
    }
    if path.len() != grid.free_count {
        return false;
    }
    for i in 1..path.len() {
        if !grid.is_adj(path[i - 1], path[i]) {
            return false;
        }
    }
    if require_cycle && !grid.is_adj(*path.last().unwrap(), 0) {
        return false;
    }
    true
}

fn greedy_attempt(
    grid: &Grid,
    require_cycle: bool,
    reserved_end: Option<usize>,
    rng: &mut XorShift64,
    deadline: Instant,
) -> Option<Vec<usize>> {
    let total = grid.free.len();
    let start = 0usize;
    if total == 0 || !grid.free[start] {
        return None;
    }
    let mut visited = vec![false; total];
    visited[start] = true;
    let mut path = Vec::with_capacity(grid.free_count);
    path.push(start);

    let mut stamp = 1u32;
    let mut seen = vec![0u32; total];
    let mut cur = start;
    while path.len() < grid.free_count {
        if Instant::now() >= deadline {
            return None;
        }
        let is_last_move = path.len() + 1 == grid.free_count;
        let mut cands: Vec<(usize, u8, u64)> = Vec::new();
        let len = grid.neigh_len[cur] as usize;
        for k in 0..len {
            let nb = grid.neigh[cur][k];
            if !grid.free[nb] || visited[nb] {
                continue;
            }
            if let Some(end) = reserved_end {
                if nb == end && !is_last_move {
                    continue;
                }
            }
            // Warnsdorff: count onward unvisited neighbors after moving to nb (exclude cur which is visited).
            let mut deg = 0u8;
            let nlen = grid.neigh_len[nb] as usize;
            for kk in 0..nlen {
                let nn = grid.neigh[nb][kk];
                if grid.free[nn] && !visited[nn] && nn != cur {
                    // if reserved_end exists and nn==reserved_end, it's still unvisited and counts.
                    deg += 1;
                }
            }
            cands.push((nb, deg, rng.next_u64()));
        }

        if cands.is_empty() {
            return None;
        }

        cands.sort_unstable_by(|a, b| match a.1.cmp(&b.1) {
            Ordering::Equal => a.2.cmp(&b.2),
            o => o,
        });

        let mut moved = false;
        for (nb, _, _) in cands {
            visited[nb] = true;
            let unvisited_left = grid.free_count - path.len() - 1;
            if is_unvisited_connected(grid, &visited, unvisited_left, &mut stamp, &mut seen) {
                path.push(nb);
                cur = nb;
                moved = true;
                break;
            }
            visited[nb] = false;
        }
        if !moved {
            return None;
        }
    }

    if let Some(end) = reserved_end {
        if *path.last().unwrap() != end {
            return None;
        }
    }
    if require_cycle && !grid.is_adj(*path.last().unwrap(), start) {
        return None;
    }
    if !validate_path(grid, &path, require_cycle) {
        return None;
    }
    Some(path)
}

fn dfs_exact(
    grid: &Grid,
    require_cycle: bool,
    reserved_end: Option<usize>,
    visited: &mut [bool],
    path: &mut Vec<usize>,
    cur: usize,
    stamp: &mut u32,
    seen: &mut [u32],
    deadline: Instant,
) -> bool {
    if Instant::now() >= deadline {
        return false;
    }
    if path.len() == grid.free_count {
        if let Some(end) = reserved_end {
            if cur != end {
                return false;
            }
        }
        return !require_cycle || grid.is_adj(cur, 0);
    }

    let is_last_move = path.len() + 1 == grid.free_count;
    let mut cands: Vec<(usize, u8)> = Vec::new();
    let len = grid.neigh_len[cur] as usize;
    for k in 0..len {
        let nb = grid.neigh[cur][k];
        if !grid.free[nb] || visited[nb] {
            continue;
        }
        if let Some(end) = reserved_end {
            if nb == end && !is_last_move {
                continue;
            }
        }
        let mut deg = 0u8;
        let nlen = grid.neigh_len[nb] as usize;
        for kk in 0..nlen {
            let nn = grid.neigh[nb][kk];
            if grid.free[nn] && !visited[nn] && nn != cur {
                deg += 1;
            }
        }
        cands.push((nb, deg));
    }

    cands.sort_unstable_by(|a, b| a.1.cmp(&b.1));

    for (nb, _) in cands {
        visited[nb] = true;
        path.push(nb);
        let unvisited_left = grid.free_count - path.len();
        let ok = is_unvisited_connected(grid, visited, unvisited_left, stamp, seen);
        if ok && dfs_exact(grid, require_cycle, reserved_end, visited, path, nb, stamp, seen, deadline) {
            return true;
        }
        path.pop();
        visited[nb] = false;
    }
    false
}

fn solve_with_obstacles(grid: &Grid, require_cycle: bool, global_deadline: Instant) -> Option<Vec<usize>> {
    let start = 0usize;
    if grid.free_count == 0 || grid.free.is_empty() {
        return None;
    }
    if !grid.free[start] {
        return None;
    }
    if grid.free_count == 1 {
        return if require_cycle { None } else { Some(vec![start]) };
    }

    if grid.bfs_reachable_from_start() != grid.free_count {
        return None;
    }
    if !parity_ok_for_path(grid, require_cycle) {
        return None;
    }
    if !degrees_ok(grid, require_cycle) {
        return None;
    }

    let mut end_candidates: Vec<Option<usize>> = Vec::new();
    if require_cycle {
        // End must be adjacent to start: try each free neighbor of start.
        let len = grid.neigh_len[start] as usize;
        for k in 0..len {
            let nb = grid.neigh[start][k];
            if grid.free[nb] {
                end_candidates.push(Some(nb));
            }
        }
        if end_candidates.is_empty() {
            return None;
        }
    } else {
        end_candidates.push(None);
    }

    // Heuristic-first for larger instances, exact search for small.
    let small = grid.free_count <= 180;

    for reserved_end in end_candidates {
        // Multiple greedy restarts with different tie-breaks.
        let mut seed = 0x9E37_79B9_7F4A_7C15u64;
        seed ^= (grid.w as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9);
        seed ^= (grid.h as u64).wrapping_mul(0x94D0_49BB_1331_11EB);
        seed ^= (grid.free_count as u64).wrapping_mul(0xD134_2543_DE82_EB97);
        if let Some(e) = reserved_end {
            seed ^= (e as u64).wrapping_mul(0x2545_F491_4F6C_DD1D);
        }
        let mut rng = XorShift64::new(seed);

        for _ in 0..200 {
            if Instant::now() >= global_deadline {
                return None;
            }
            let attempt_deadline = global_deadline.min(Instant::now() + Duration::from_millis(250));
            if let Some(p) = greedy_attempt(grid, require_cycle, reserved_end, &mut rng, attempt_deadline) {
                return Some(p);
            }
        }

        if small {
            // Exact backtracking within remaining time.
            let total = grid.free.len();
            let mut visited = vec![false; total];
            visited[start] = true;
            let mut path = Vec::with_capacity(grid.free_count);
            path.push(start);
            let mut stamp = 1u32;
            let mut seen = vec![0u32; total];
            if dfs_exact(
                grid,
                require_cycle,
                reserved_end,
                &mut visited,
                &mut path,
                start,
                &mut stamp,
                &mut seen,
                global_deadline,
            ) {
                if validate_path(grid, &path, require_cycle) {
                    return Some(path);
                }
            }
        }
    }

    None
}

fn stream_cycle_even_width(w_even: usize, h: usize, transpose: bool, out: &mut dyn Write) -> io::Result<()> {
    let total = w_even.saturating_mul(h);
    writeln!(out, "{total}")?;
    let emit = |x: usize, y: usize, out: &mut dyn Write| -> io::Result<()> {
        if transpose {
            writeln!(out, "{y} {x}")
        } else {
            writeln!(out, "{x} {y}")
        }
    };

    // Column 0 down.
    for y in 0..h {
        emit(0, y, out)?;
    }
    // Columns 1..w-1 zigzag on rows 1..h-1.
    for x in 1..w_even {
        if x % 2 == 1 {
            emit(x, h - 1, out)?;
            if h >= 3 {
                for y in (1..h - 1).rev() {
                    emit(x, y, out)?;
                }
            }
        } else {
            emit(x, 1, out)?;
            for y in 2..h {
                emit(x, y, out)?;
            }
        }
    }
    // Top row right-to-left (excluding x=0 which is already visited).
    emit(w_even - 1, 0, out)?;
    if w_even >= 3 {
        for x in (1..w_even - 1).rev() {
            emit(x, 0, out)?;
        }
    }
    Ok(())
}

fn stream_cycle_rect(w: usize, h: usize, out: &mut dyn Write) -> io::Result<bool> {
    if w <= 1 || h <= 1 {
        return Ok(false);
    }
    let total = match w.checked_mul(h) {
        Some(v) => v,
        None => return Ok(false),
    };
    if total % 2 == 1 {
        return Ok(false);
    }
    if w % 2 == 0 {
        stream_cycle_even_width(w, h, false, out)?;
        Ok(true)
    } else if h % 2 == 0 {
        // Build in transposed grid.
        stream_cycle_even_width(h, w, true, out)?;
        Ok(true)
    } else {
        Ok(false)
    }
}

                    int bit = remaining & -remaining;
                    int next = BitOperations.TrailingZeroCount((uint)bit);
                    int nmask = mask | bit;

                    double cand = cur + cost[last, next];
                    int idx = nmask * n + next;
                    if (cand < dp[idx])
                    {
                        dp[idx] = cand;
                        parent[idx] = last;
                    }
                    remaining ^= bit;
                }
            }
        }

        double best = double.PositiveInfinity;
        int bestLast = 0;
        for (int last = 0; last < n; last++)
        {
            double v = dp[all * n + last];
            if (v < best)
            {
                best = v;
                bestLast = last;
            }
        }

        int[] route = new int[n];
        int curMask = all;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int idx = curMask * n + curLast;
            int p = parent[idx];
            curMask ^= 1 << curLast;
            curLast = p < 0 ? 0 : p;
        }

        return route;
    }

    private static int[] SolveHeuristic(int n, double[] startCost, double[,] cost)
    {
        int multiStarts = n <= 30 ? 12 : (n <= 60 ? 8 : 5);
        int[] candidates = TopKIndices(startCost, Math.Min(n, multiStarts));

        double bestCost = double.PositiveInfinity;
        int[] bestRoute = new int[n];

        for (int s = 0; s < candidates.Length; s++)
        {
            int seed = candidates[s];
            int[] route = BuildBestEndRoute(n, seed, startCost, cost);
            TwoOptImprove(route, startCost, cost, maxPasses: n <= 60 ? 50 : 30);
            double c = RouteCost(route, startCost, cost);
            if (c < bestCost)
            {
                bestCost = c;
                Array.Copy(route, bestRoute, n);
            }
        }

        return bestRoute;
    }

    private static int[] BuildBestEndRoute(int n, int seed, double[] startCost, double[,] cost)
    {
        bool[] used = new bool[n];
        used[seed] = true;

        int[] deque = new int[n * 2 + 4];
        int head = n;
        int tail = n;
        deque[tail++] = seed;
        int count = 1;

        while (count < n)
        {
            int front = deque[head];
            int back = deque[tail - 1];
            double startToFront = startCost[front];

            double bestInc = double.PositiveInfinity;
            int bestNode = -1;
            bool bestToFront = false;

            for (int c = 0; c < n; c++)
            {
                if (used[c]) continue;

                double incFront = startCost[c] + cost[c, front] - startToFront;
                if (incFront < bestInc)
                {
                    bestInc = incFront;
                    bestNode = c;
                    bestToFront = true;
                }

                double incBack = cost[back, c];
                if (incBack < bestInc)
                {
                    bestInc = incBack;
                    bestNode = c;
                    bestToFront = false;
                }
            }

            used[bestNode] = true;
            if (bestToFront)
                deque[--head] = bestNode;
            else
                deque[tail++] = bestNode;
            count++;
        }

        int[] route = new int[n];
        for (int i = 0; i < n; i++)
            route[i] = deque[head + i];
        return route;
    }

    private static void TwoOptImprove(int[] route, double[] startCost, double[,] cost, int maxPasses)
    {
        int n = route.Length;
        if (n <= 3) return;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                int aPrev = i == 0 ? -1 : route[i - 1];
                int a = route[i];

                for (int j = i + 1; j < n; j++)
                {
                    int b = route[j];
                    int bNext = j == n - 1 ? -1 : route[j + 1];

                    double oldCost = EdgeCost(aPrev, a, startCost, cost) + EdgeCost(b, bNext, startCost, cost);
                    double newCost = EdgeCost(aPrev, b, startCost, cost) + EdgeCost(a, bNext, startCost, cost);

                    if (newCost + 1e-12 < oldCost)
                    {
                        Array.Reverse(route, i, j - i + 1);
                        improved = true;
                        break;
                    }
                }

                if (improved) break;
            }

            if (!improved) break;
        }
    }

    private static double EdgeCost(int from, int to, double[] startCost, double[,] cost)
    {
        if (to < 0) return 0.0;
        if (from < 0) return startCost[to];
        return cost[from, to];
    }

    private static double RouteCost(int[] route, double[] startCost, double[,] cost)
    {
        if (route.Length == 0) return 0.0;
        double total = startCost[route[0]];
        for (int i = 0; i < route.Length - 1; i++)
            total += cost[route[i], route[i + 1]];
        return total;
    }

    private static int[] TopKIndices(double[] arr, int k)
    {
        int n = arr.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;

        Array.Sort(idx, (a, b) => arr[a].CompareTo(arr[b]));

        int[] outIdx = new int[k];
        Array.Copy(idx, outIdx, k);
        return outIdx;

    private static double OrbitCost(Orbit a, Orbit b, double mu)
    {
        double r1 = a.P > 1e-9 ? a.P : a.RMag;
        double r2 = b.P > 1e-9 ? b.P : b.RMag;
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);

        double dvRad = HohmannDv(mu, r1, r2);

        double planeDot = Clamp(a.HHat.Dot(b.HHat), -1.0, 1.0);
        double angle = Math.Acos(planeDot);
        double rPlane = Math.Max(r1, r2);
        double vPlane = Math.Sqrt(mu / rPlane);
        double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * angle);

        double dvEcc = Math.Abs(a.E - b.E) * vPlane;

        return dvRad + dvPlane + 0.1 * dvEcc;
    }

    private static double HohmannDv(double mu, double r1, double r2)
    {
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);

        if (Math.Abs(r1 - r2) < 1e-12)
            return 0.0;

        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double aT = 0.5 * (r1 + r2);
        double vT1 = Math.Sqrt(mu * (2.0 / r1 - 1.0 / aT));
        double vT2 = Math.Sqrt(mu * (2.0 / r2 - 1.0 / aT));
        return Math.Abs(vT1 - v1) + Math.Abs(v2 - vT2);
    }

inline int Plane::classifyPolygon(const Polygon& polygon) const {
  // Bounding sphere fast path: if the whole sphere is strictly on one side, we can classify quickly.
  double dist = normal.dot(polygon.bsCenter) - w;
  if (dist > polygon.bsRadius + eps) return FRONT;
  if (dist < -polygon.bsRadius - eps) return BACK;

  int polygonType = 0;
  for (const auto& v : polygon.vertices) {
    double t = normal.dot(v.pos) - w;
    int type = (t < -eps) ? BACK : (t > eps) ? FRONT : COPLANAR;
    polygonType |= type;
    if (polygonType == SPANNING) break;
  }
  return polygonType;
}

inline void Plane::splitPolygon(const Polygon& polygon,
                                std::vector<Polygon>& coplanarFront,
                                std::vector<Polygon>& coplanarBack,
                                std::vector<Polygon>& front,
                                std::vector<Polygon>& back) const {
  // Fast path via bounding sphere.
  double dist = normal.dot(polygon.bsCenter) - w;
  if (dist > polygon.bsRadius + eps) {
    front.push_back(polygon);
    return;
  }
  if (dist < -polygon.bsRadius - eps) {
    back.push_back(polygon);
    return;
  }

  int polygonType = 0;
  std::vector<int> types;
  types.reserve(polygon.vertices.size());
  for (const auto& v : polygon.vertices) {
    double t = normal.dot(v.pos) - w;
    int type = (t < -eps) ? BACK : (t > eps) ? FRONT : COPLANAR;
    polygonType |= type;
    types.push_back(type);
  }

  switch (polygonType) {
    case COPLANAR: {
      // If polygon plane has same orientation as this plane, keep it in front list, else back.
      if (normal.dot(polygon.plane.normal) > 0) coplanarFront.push_back(polygon);
      else coplanarBack.push_back(polygon);
      break;
    }
    case FRONT:
      front.push_back(polygon);
      break;
    case BACK:
      back.push_back(polygon);
      break;
    case SPANNING: {
      std::vector<Vertex> f;
      std::vector<Vertex> b;
      f.reserve(polygon.vertices.size() + 2);
      b.reserve(polygon.vertices.size() + 2);

      const int n = static_cast<int>(polygon.vertices.size());
      for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        int ti = types[i], tj = types[j];
        const Vertex& vi = polygon.vertices[i];
        const Vertex& vj = polygon.vertices[j];

        if (ti != BACK) f.push_back(vi);
        if (ti != FRONT) b.push_back(vi);

        if ((ti | tj) == SPANNING) {
          Vec3 dir = vj.pos - vi.pos;
          double denom = normal.dot(dir);
          // denom should not be 0 here; guard anyway.
          double t = 0.0;
          if (std::abs(denom) > 0) {
            t = (w - normal.dot(vi.pos)) / denom;
          }
          // Clamp to reduce numeric drift.
          if (t < 0) t = 0;
          if (t > 1) t = 1;
          Vertex v = vi.interpolate(vj, t);
          f.push_back(v);
          b.push_back(v);
        }
      }

      if (f.size() >= 3) front.emplace_back(std::move(f), eps);
      if (b.size() >= 3) back.emplace_back(std::move(b), eps);
      break;
    }
  }

 public:
  Node() = default;
  explicit Node(std::vector<Polygon> polygons) { build(std::move(polygons)); }

  void invert() {
    // Iterative traversal to avoid deep recursion.
    std::vector<Node*> stack;
    stack.push_back(this);
    while (!stack.empty()) {
      Node* node = stack.back();
      stack.pop_back();
      for (auto& p : node->polygons_) p.flip();
      if (node->plane_) node->plane_ = std::make_unique<Plane>(node->plane_->flipped());
      std::swap(node->front_, node->back_);
      if (node->front_) stack.push_back(node->front_.get());
      if (node->back_) stack.push_back(node->back_.get());
    }
  }

  std::vector<Polygon> clipPolygons(const std::vector<Polygon>& polygons) const {
    if (!plane_) return polygons;
    std::vector<Polygon> front;
    std::vector<Polygon> back;
    front.reserve(polygons.size());
    back.reserve(polygons.size());
    for (const auto& p : polygons) {
      plane_->splitPolygon(p, front, back, front, back);
    }
    if (front_) front = front_->clipPolygons(front);
    if (back_) back = back_->clipPolygons(back);
    else back.clear();
    front.insert(front.end(), back.begin(), back.end());
    return front;
  }

  void clipTo(const Node& bsp) {
    // Iterative traversal.
    std::vector<Node*> stack;
    stack.push_back(this);
    while (!stack.empty()) {
      Node* node = stack.back();
      stack.pop_back();
      node->polygons_ = bsp.clipPolygons(node->polygons_);
      if (node->front_) stack.push_back(node->front_.get());
      if (node->back_) stack.push_back(node->back_.get());
    }
  }

  std::vector<Polygon> allPolygons() const {
    std::vector<Polygon> out;
    std::vector<const Node*> stack;
    stack.push_back(this);
    while (!stack.empty()) {
      const Node* node = stack.back();
      stack.pop_back();
      out.insert(out.end(), node->polygons_.begin(), node->polygons_.end());
      if (node->front_) stack.push_back(node->front_.get());
      if (node->back_) stack.push_back(node->back_.get());
    }
    return out;
  }

  void build(std::vector<Polygon> polygons) {
    if (polygons.empty()) return;
    if (!plane_) {
      // Heuristic plane selection: try a few candidates and pick the one that minimizes splits
      // while keeping the tree reasonably balanced.
      const int candidateCount = std::min<int>(static_cast<int>(polygons.size()), 12);
      const int testCount = std::min<int>(static_cast<int>(polygons.size()), 96);
      double bestScore = std::numeric_limits<double>::infinity();
      Plane bestPlane = polygons[0].plane;

      for (int c = 0; c < candidateCount; c++) {
        const Plane& candidate = polygons[c].plane;
        int numFront = 0, numBack = 0, numSplit = 0, numCoplanar = 0;
        for (int i = 0; i < testCount; i++) {
          int cls = candidate.classifyPolygon(polygons[i]);
          if (cls == Plane::FRONT) numFront++;
          else if (cls == Plane::BACK) numBack++;
          else if (cls == Plane::SPANNING) numSplit++;
          else numCoplanar++;
        }
        double balance = std::abs(numFront - numBack);
        double score = static_cast<double>(numSplit) * 8.0 + balance + static_cast<double>(numCoplanar) * 0.25;
        if (score < bestScore) {
          bestScore = score;
          bestPlane = candidate;
        }
      }
      plane_ = std::make_unique<Plane>(bestPlane);
    }

    std::vector<Polygon> front;
    std::vector<Polygon> back;
    front.reserve(polygons.size());
    back.reserve(polygons.size());

    for (auto& p : polygons) {
      plane_->splitPolygon(p, polygons_, polygons_, front, back);
    }

    if (!front.empty()) {
      if (!front_) front_ = std::make_unique<Node>();
      front_->build(std::move(front));
    }
    if (!back.empty()) {
      if (!back_) back_ = std::make_unique<Node>();
      back_->build(std::move(back));
    }
  }

 private:
  std::unique_ptr<Plane> plane_;
  std::unique_ptr<Node> front_;
  std::unique_ptr<Node> back_;
  std::vector<Polygon> polygons_;
};

inline std::vector<Polygon> unionPolygons(std::vector<Polygon> a, std::vector<Polygon> b) {
  Node A(std::move(a));
  Node B(std::move(b));
  A.clipTo(B);
  B.clipTo(A);
  B.invert();
  B.clipTo(A);
  B.invert();
  A.build(B.allPolygons());
  return A.allPolygons();
}

struct Mesh {
  std::vector<csg::Vec3> vertices;
  std::vector<std::array<int, 3>> faces;
};

static csg::Vec3 minVec(const csg::Vec3& a, const csg::Vec3& b) {
  return {std::min(a.x, b.x), std::min(a.y, b.y), std::min(a.z, b.z)};
}
static csg::Vec3 maxVec(const csg::Vec3& a, const csg::Vec3& b) {
  return {std::max(a.x, b.x), std::max(a.y, b.y), std::max(a.z, b.z)};
}

static void bbox(const Mesh& m, csg::Vec3& outMin, csg::Vec3& outMax) {
  outMin = { std::numeric_limits<double>::infinity(),
             std::numeric_limits<double>::infinity(),
             std::numeric_limits<double>::infinity() };
  outMax = { -std::numeric_limits<double>::infinity(),
             -std::numeric_limits<double>::infinity(),
             -std::numeric_limits<double>::infinity() };
  for (const auto& v : m.vertices) {
    outMin = minVec(outMin, v);
    outMax = maxVec(outMax, v);
  }
}

static bool aabbDisjoint(const csg::Vec3& aMin, const csg::Vec3& aMax,
                         const csg::Vec3& bMin, const csg::Vec3& bMax) {
  return (aMax.x < bMin.x) || (bMax.x < aMin.x) ||
         (aMax.y < bMin.y) || (bMax.y < aMin.y) ||
         (aMax.z < bMin.z) || (bMax.z < aMin.z);
}

struct QuantKey {
  std::int64_t x, y, z;
  bool operator==(const QuantKey& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct QuantKeyHash {
  std::size_t operator()(const QuantKey& k) const noexcept {
    // 64-bit mix (splitmix64-ish)
    auto mix = [](std::uint64_t v) {
      v += 0x9e3779b97f4a7c15ULL;
      v = (v ^ (v >> 30)) * 0xbf58476d1ce4e5b9ULL;
      v = (v ^ (v >> 27)) * 0x94d049bb133111ebULL;
      return v ^ (v >> 31);
    };
    std::uint64_t h = mix(static_cast<std::uint64_t>(k.x));
    h ^= mix(static_cast<std::uint64_t>(k.y) + 0x100000001b3ULL);
    h ^= mix(static_cast<std::uint64_t>(k.z) + 0x9e3779b97f4a7c15ULL);
    return static_cast<std::size_t>(h);
  }
};

static std::int64_t qround(double v, double invStep) {
  // round to nearest int64
  return static_cast<std::int64_t>(std::llround(v * invStep));
}

static void writeMeshAsUnion(const Mesh& a, const Mesh& b) {
  csg::Vec3 aMin, aMax, bMin, bMax;
  bbox(a, aMin, aMax);
  bbox(b, bMin, bMax);
  csg::Vec3 globalMin = minVec(aMin, bMin);
  csg::Vec3 globalMax = maxVec(aMax, bMax);
  double diag = (globalMax - globalMin).norm();
  if (!(diag > 0)) diag = 1.0;

  // Epsilon used for plane/side classification.
  double eps = 1e-5;
  // Try to adapt slightly to scale, but clamp to avoid over-blurring.
  eps = std::clamp(1e-8 * (diag + 1.0), 1e-7, 1e-5);

  // If AABBs are disjoint, the union is just both meshes concatenated.
  if (aabbDisjoint(aMin, aMax, bMin, bMax)) {
    std::vector<csg::Vec3> outV = a.vertices;
    outV.insert(outV.end(), b.vertices.begin(), b.vertices.end());
    std::vector<std::array<int,3>> outF = a.faces;
    int offset = static_cast<int>(a.vertices.size());
    outF.reserve(outF.size() + b.faces.size());
    for (auto f : b.faces) outF.push_back({f[0] + offset, f[1] + offset, f[2] + offset});

    std::cout.setf(std::ios::fixed);
    std::cout.precision(9);
    std::cout << outV.size() << " " << outF.size() << "\n";
    for (const auto& v : outV) std::cout << v.x << " " << v.y << " " << v.z << "\n";
    for (const auto& f : outF) std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    return;
  }

  // Convert input meshes to CSG polygons (triangles).
  std::vector<csg::Polygon> polysA;
  polysA.reserve(a.faces.size());
  for (const auto& f : a.faces) {
    const auto& v0 = a.vertices[f[0]];
    const auto& v1 = a.vertices[f[1]];
    const auto& v2 = a.vertices[f[2]];
    polysA.push_back(csg::Polygon::fromTriangle(v0, v1, v2, eps));
  }
  std::vector<csg::Polygon> polysB;
  polysB.reserve(b.faces.size());
  for (const auto& f : b.faces) {
    const auto& v0 = b.vertices[f[0]];
    const auto& v1 = b.vertices[f[1]];
    const auto& v2 = b.vertices[f[2]];
    polysB.push_back(csg::Polygon::fromTriangle(v0, v1, v2, eps));
  }

  std::vector<csg::Polygon> unionPolys = csg::unionPolygons(std::move(polysA), std::move(polysB));

  // Weld vertices by snapping to a grid (canonicalization-lite).
  const double weldStep = std::max(1e-7, eps * 0.5);
  const double invWeld = 1.0 / weldStep;

  std::unordered_map<QuantKey, int, QuantKeyHash> vertexMap;
  vertexMap.reserve(unionPolys.size() * 3);

  std::vector<csg::Vec3> outVertices;
  std::vector<std::array<int, 3>> outFaces;

  auto getIndex = [&](const csg::Vec3& p) -> int {
    QuantKey key{qround(p.x, invWeld), qround(p.y, invWeld), qround(p.z, invWeld)};
    auto it = vertexMap.find(key);
    if (it != vertexMap.end()) return it->second;
    int idx = static_cast<int>(outVertices.size());
    vertexMap.emplace(key, idx);
    // Store snapped position for consistency.
    outVertices.push_back({static_cast<double>(key.x) * weldStep,
                           static_cast<double>(key.y) * weldStep,
                           static_cast<double>(key.z) * weldStep});
    return idx;
  };

  // Triangulate each convex polygon by fan.
  for (const auto& poly : unionPolys) {
    if (poly.vertices.size() < 3) continue;
    int i0 = getIndex(poly.vertices[0].pos);
    for (size_t i = 1; i + 1 < poly.vertices.size(); i++) {
      int i1 = getIndex(poly.vertices[i].pos);
      int i2 = getIndex(poly.vertices[i + 1].pos);
      if (i0 == i1 || i1 == i2 || i2 == i0) continue;
      outFaces.push_back({i0, i1, i2});
    }
  }

  std::cout.setf(std::ios::fixed);
  std::cout.precision(9);
  std::cout << outVertices.size() << " " << outFaces.size() << "\n";
  for (const auto& v : outVertices) std::cout << v.x << " " << v.y << " " << v.z << "\n";
  for (const auto& f : outFaces) std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
}
