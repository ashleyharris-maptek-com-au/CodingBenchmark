const fs = require('fs');

const rawInput = fs.readFileSync(0, 'utf8');
const input = rawInput && rawInput.charCodeAt(0) === 0xfeff ? rawInput.slice(1) : rawInput;
const data = input && input.trim() ? JSON.parse(input) : {};

const width = Math.trunc(Number(data.width));
const height = Math.trunc(Number(data.height));
const grid = Array.isArray(data.grid) ? data.grid : [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
let cutBudget = Math.trunc(Number(data.cutBudget));
if (!Number.isFinite(cutBudget) || cutBudget < 0) cutBudget = 0;

const N = width * height;
if (width <= 0 || height <= 0 || !Number.isFinite(N) || N <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// 1 = burnable forest '.', 0 = water/rock '~' (or anything else).
const burnable = new Uint8Array(N);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  const base = y * width;
  for (let x = 0; x < width; x++) {
    burnable[base + x] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
  }
}

const isIgnition = new Uint8Array(N);
for (let i = 0; i < ignitions.length; i++) {
  const p = ignitions[i];
  if (!p || p.length < 2) continue;
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  isIgnition[y * width + x] = 1;
}

const isTown = new Uint8Array(N);
const townValueAt = new Float64Array(N);
const townCells = [];
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (!t) continue;
  const x = t.x | 0;
  const y = t.y | 0;
  const v = Number(t.value);
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  if (!Number.isFinite(v) || v <= 0) continue;
  const idx = y * width + x;
  if (!isTown[idx]) townCells.push(idx);
  isTown[idx] = 1;
  townValueAt[idx] += v;
}

// Multi-source BFS to find baseline burned cells (reachable from any ignition).
const reachable = new Uint8Array(N);
const parent = new Int32Array(N);
parent.fill(-1);
const q = new Int32Array(N);
let qh = 0;
let qt = 0;
for (let i = 0; i < ignitions.length; i++) {
  const p = ignitions[i];
  if (!p || p.length < 2) continue;
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const idx = y * width + x;
  if (!burnable[idx]) continue;
  if (reachable[idx]) continue;
  reachable[idx] = 1;
  q[qt++] = idx;
}

while (qh < qt) {
  const idx = q[qh++];
  const x = idx % width;
  // Left
  if (x > 0) {
    const n = idx - 1;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      parent[n] = idx;
      q[qt++] = n;
    }
  }
  // Right
  if (x + 1 < width) {
    const n = idx + 1;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      parent[n] = idx;
      q[qt++] = n;
    }
  }
  // Up
  if (idx >= width) {
    const n = idx - width;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      parent[n] = idx;
      q[qt++] = n;
    }
  }
  // Down
  if (idx + width < N) {
    const n = idx + width;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      parent[n] = idx;
      q[qt++] = n;
    }
  }
}

// Collect threatened towns (baseline-burned) and baseline burned value.
const threatened = [];
let baselineBurnedValue = 0;
for (let i = 0; i < townCells.length; i++) {
  const idx = townCells[i];
  const v = townValueAt[idx];
  if (v <= 0) continue;
  if (!burnable[idx]) continue; // non-burnable town cell is already safe
  if (!reachable[idx]) continue;
  threatened.push({ idx, value: v });
  baselineBurnedValue += v;
}

if (cutBudget <= 0 || threatened.length === 0 || baselineBurnedValue <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Plan cuts.
const plannedCut = new Uint8Array(N);
const firebreaks = [];
let usedBudget = 0;

function addCut(idx) {
  if (plannedCut[idx]) return false;
  plannedCut[idx] = 1;
  const x = idx % width;
  const y = (idx - x) / width;
  firebreaks.push([x, y]);
  usedBudget++;
  return true;
}

// --- Phase 1: protect adjacent-town clusters by cutting their boundary neighbors.
// Build DSU over threatened towns that are 4-neighbor adjacent.
const T = threatened.length;
const threatIdAt = new Int32Array(N);
threatIdAt.fill(-1);
for (let i = 0; i < T; i++) threatIdAt[threatened[i].idx] = i;

const ufParent = new Int32Array(T);
const ufRank = new Uint8Array(T);
for (let i = 0; i < T; i++) ufParent[i] = i;

function ufFind(a) {
  while (ufParent[a] !== a) {
    ufParent[a] = ufParent[ufParent[a]];
    a = ufParent[a];
  }
  return a;
}
function ufUnion(a, b) {
  let ra = ufFind(a);
  let rb = ufFind(b);
  if (ra === rb) return;
  const rka = ufRank[ra];
  const rkb = ufRank[rb];
  if (rka < rkb) {
    ufParent[ra] = rb;
  } else if (rka > rkb) {
    ufParent[rb] = ra;
  } else {
    ufParent[rb] = ra;
    ufRank[ra] = rka + 1;
  }
}

for (let i = 0; i < T; i++) {
  const idx = threatened[i].idx;
  const x = idx % width;
  if (x > 0) {
    const j = threatIdAt[idx - 1];
    if (j !== -1) ufUnion(i, j);
  }
  if (x + 1 < width) {
    const j = threatIdAt[idx + 1];
    if (j !== -1) ufUnion(i, j);
  }
  if (idx >= width) {
    const j = threatIdAt[idx - width];
    if (j !== -1) ufUnion(i, j);
  }
  if (idx + width < N) {
    const j = threatIdAt[idx + width];
    if (j !== -1) ufUnion(i, j);
  }
}

// Group towns by cluster root.
const rootToClusterId = Object.create(null);
const clusterTowns = [];
const clusterValue = [];

for (let i = 0; i < T; i++) {
  const r = ufFind(i);
  let cid = rootToClusterId[r];
  if (cid === undefined) {
    cid = clusterTowns.length;
    rootToClusterId[r] = cid;
    clusterTowns.push([]);
    clusterValue.push(0);
  }
  clusterTowns[cid].push(threatened[i].idx);
  clusterValue[cid] += threatened[i].value;
}

// Build boundary list per cluster, deduplicated using a marker array.
const boundaryMark = new Int32Array(N);
let boundaryStamp = 1;
const clusters = [];

for (let cid = 0; cid < clusterTowns.length; cid++) {
  const townsIdxs = clusterTowns[cid];
  const stamp = boundaryStamp++;
  const boundary = [];
  for (let k = 0; k < townsIdxs.length; k++) {
    const tIdx = townsIdxs[k];
    const x = tIdx % width;

    // For each neighbor cell outside the cluster, cut that neighbor if it is burnable.
    // Note: we don't cut town cells themselves here (only their neighbors), to avoid
    // relying on any assumptions about "cutting a town".
    if (x > 0) {
      const n = tIdx - 1;
      if (burnable[n] && !isTown[n] && threatIdAt[n] === -1 && boundaryMark[n] !== stamp) {
        boundaryMark[n] = stamp;
        boundary.push(n);
      }
    }
    if (x + 1 < width) {
      const n = tIdx + 1;
      if (burnable[n] && !isTown[n] && threatIdAt[n] === -1 && boundaryMark[n] !== stamp) {
        boundaryMark[n] = stamp;
        boundary.push(n);
      }
    }
    if (tIdx >= width) {
      const n = tIdx - width;
      if (burnable[n] && !isTown[n] && threatIdAt[n] === -1 && boundaryMark[n] !== stamp) {
        boundaryMark[n] = stamp;
        boundary.push(n);
      }
    }
    if (tIdx + width < N) {
      const n = tIdx + width;
      if (burnable[n] && !isTown[n] && threatIdAt[n] === -1 && boundaryMark[n] !== stamp) {
        boundaryMark[n] = stamp;
        boundary.push(n);
      }
    }
  }

  const cost = boundary.length;
  if (cost > 0) {
    clusters.push({
      value: clusterValue[cid],
      boundary,
      ratio: clusterValue[cid] / cost,
    });
  }
}

clusters.sort((a, b) => (b.ratio - a.ratio) || (b.value - a.value));

for (let i = 0; i < clusters.length && usedBudget < cutBudget; i++) {
  const cl = clusters[i];
  let addl = 0;
  const boundary = cl.boundary;
  for (let j = 0; j < boundary.length; j++) {
    const idx = boundary[j];
    if (!plannedCut[idx]) addl++;
  }
  if (addl <= 0) continue;
  if (usedBudget + addl > cutBudget) continue;

  for (let j = 0; j < boundary.length; j++) {
    const idx = boundary[j];
    if (!plannedCut[idx]) addCut(idx);
  }
}

// --- Phase 2: remaining budget goes to "chokepoint-like" cuts from a BFS-tree heuristic.
// Compute BFS-tree subtree threatened value.
const subtreeVal = new Float64Array(N);
for (let i = 0; i < threatened.length; i++) subtreeVal[threatened[i].idx] += threatened[i].value;

for (let i = qt - 1; i >= 0; i--) {
  const idx = q[i];
  const p = parent[idx];
  if (p >= 0) subtreeVal[p] += subtreeVal[idx];
}

function heapSwap(scores, indices, a, b) {
  const ts = scores[a];
  scores[a] = scores[b];
  scores[b] = ts;
  const ti = indices[a];
  indices[a] = indices[b];
  indices[b] = ti;
}

function heapSiftUp(scores, indices, i) {
  while (i > 0) {
    const p = (i - 1) >> 1;
    if (scores[p] <= scores[i]) break;
    heapSwap(scores, indices, p, i);
    i = p;
  }
}

function heapSiftDown(scores, indices, i) {
  const n = scores.length;
  while (true) {
    const l = i * 2 + 1;
    if (l >= n) break;
    let m = l;
    const r = l + 1;
    if (r < n && scores[r] < scores[l]) m = r;
    if (scores[i] <= scores[m]) break;
    heapSwap(scores, indices, i, m);
    i = m;
  }
}

function heapPushTopK(scores, indices, limit, score, idx) {
  if (limit <= 0) return;
  if (scores.length < limit) {
    scores.push(score);
    indices.push(idx);
    heapSiftUp(scores, indices, scores.length - 1);
    return;
  }
  if (score <= scores[0]) return;
  scores[0] = score;
  indices[0] = idx;
  heapSiftDown(scores, indices, 0);
}

if (usedBudget < cutBudget) {
  const remaining = cutBudget - usedBudget;
  const limit = Math.min(5000, remaining * 20);
  const hs = [];
  const hi = [];

  for (let i = 0; i < qt; i++) {
    const idx = q[i];
    if (plannedCut[idx]) continue;
    if (isIgnition[idx]) continue;
    if (isTown[idx]) continue;
    const val = subtreeVal[idx];
    if (val <= 0) continue;

    const x = idx % width;
    let deg = 0;
    if (x > 0 && reachable[idx - 1]) deg++;
    if (x + 1 < width && reachable[idx + 1]) deg++;
    if (idx >= width && reachable[idx - width]) deg++;
    if (idx + width < N && reachable[idx + width]) deg++;

    // Prefer lower-degree (corridor-like) cells, but still allow higher-degree cells
    // when they cover significant threatened value in the BFS tree.
    const score = val / (deg + 0.5);
    heapPushTopK(hs, hi, limit, score, idx);
  }

  const cands = new Array(hs.length);
  for (let i = 0; i < hs.length; i++) cands[i] = [hs[i], hi[i]];
  cands.sort((a, b) => b[0] - a[0]);

  for (let i = 0; i < cands.length && usedBudget < cutBudget; i++) {
    const idx = cands[i][1];
    if (plannedCut[idx]) continue;
    if (!burnable[idx]) continue;
    if (isIgnition[idx]) continue;
    if (isTown[idx]) continue;
    addCut(idx);
  }
}

// Clamp to cutBudget (defensive).
if (firebreaks.length > cutBudget) firebreaks.length = cutBudget;

process.stdout.write(JSON.stringify({ firebreaks }));
