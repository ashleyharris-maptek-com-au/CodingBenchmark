; SPIR-V
; Geometry shader: thick line with adjacency (miter joins)

OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%func_void = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%thickness = OpConstant %float 0.025

%main = OpFunction %void None %func_void
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr

%diff0_4 = OpFSub %v4float %p2 %p0
%diff0_2 = OpVectorShuffle %v2float %diff0_4 %diff0_4 0 1
%t0 = OpExtInst %v2float %glsl Normalize %diff0_2

%diff1_4 = OpFSub %v4float %p3 %p1
%diff1_2 = OpVectorShuffle %v2float %diff1_4 %diff1_4 0 1
%t1 = OpExtInst %v2float %glsl Normalize %diff1_2

%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t0_y_neg = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %t0_y_neg %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%t1_y_neg = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %t1_y_neg %t1_x

%off0 = OpVectorTimesScalar %v2float %perp0 %thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %thickness

%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1

%p1_xy_plus = OpFAdd %v2float %p1_xy %off0
%p1_xy_minus = OpFSub %v2float %p1_xy %off0

%p2_xy_plus = OpFAdd %v2float %p2_xy %off1
%p2_xy_minus = OpFSub %v2float %p2_xy %off1

%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3

%p1_plus_x = OpCompositeExtract %float %p1_xy_plus 0
%p1_plus_y = OpCompositeExtract %float %p1_xy_plus 1
%p1_minus_x = OpCompositeExtract %float %p1_xy_minus 0
%p1_minus_y = OpCompositeExtract %float %p1_xy_minus 1

%p2_plus_x = OpCompositeExtract %float %p2_xy_plus 0
%p2_plus_y = OpCompositeExtract %float %p2_xy_plus 1
%p2_minus_x = OpCompositeExtract %float %p2_xy_minus 0
%p2_minus_y = OpCompositeExtract %float %p2_xy_minus 1

%vtx0 = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %p1_z %p1_w
%vtx1 = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %p1_z %p1_w
%vtx2 = OpCompositeConstruct %v4float %p2_plus_x %p2_plus_y %p2_z %p2_w
%vtx3 = OpCompositeConstruct %v4float %p2_minus_x %p2_minus_y %p2_z %p2_w

%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %gl_out_pos %vtx0
OpStore %outColor %c1
OpEmitVertex

OpStore %gl_out_pos %vtx1
OpStore %outColor %c1
OpEmitVertex

OpStore %gl_out_pos %vtx2
OpStore %outColor %c2
OpEmitVertex

OpStore %gl_out_pos %vtx3
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
