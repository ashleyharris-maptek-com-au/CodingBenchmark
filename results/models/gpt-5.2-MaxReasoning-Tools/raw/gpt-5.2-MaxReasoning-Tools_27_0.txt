class FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char readChar() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

public:
    template <typename T>
    bool readInt(T& out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if constexpr (std::is_signed_v<T>) {
            if (c == '-') {
                neg = true;
                c = readChar();
            }
        }

        T val = 0;
        while (c > ' ') {
            val = static_cast<T>(val * 10 + (c - '0'));
            c = readChar();
        }

        out = neg ? -val : val;
        return true;
    }
};

class FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0;

public:
    ~FastOutput() { flush(); }

    inline void flush() {
        if (idx) {
            std::fwrite(buf, 1, idx, stdout);
            idx = 0;
        }
    }

    inline void writeChar(char c) {
        if (idx >= BUFSIZE) flush();
        buf[idx++] = c;
    }

    template <typename T>
    inline void writeInt(T x) {
        using U = std::make_unsigned_t<T>;
        U u = 0;
        if constexpr (std::is_signed_v<T>) {
            if (x < 0) {
                writeChar('-');
                // Avoid overflow on INT_MIN.
                u = static_cast<U>(-(x + 1));
                u += 1;
            } else {
                u = static_cast<U>(x);
            }
        } else {
            u = x;
        }

        if (u == 0) {
            writeChar('0');
            return;
        }
        char tmp[32];
        int len = 0;
        while (u > 0) {
            tmp[len++] = static_cast<char>('0' + (u % 10));
            u /= 10;
        }
        while (len--) writeChar(tmp[len]);
    }
};

struct GraphCSR {
    int n = 0;
    std::vector<int> offset; // n+1
    std::vector<int> adj;    // 2m

    inline int degree(int v) const { return offset[v + 1] - offset[v]; }
};

static bool verifyColoring(const GraphCSR& g, const std::vector<int>& color, int k) {
    if ((int)color.size() != g.n) return false;
    for (int v = 0; v < g.n; ++v) {
        if (color[v] < 0 || color[v] >= k) return false;
    }
    for (int v = 0; v < g.n; ++v) {
        const int cv = color[v];
        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            if (u == v) return false; // self-loop makes proper coloring impossible
            if (u > v && color[u] == cv) return false;
        }
    }
    return true;
}

static bool bipartiteColoring(const GraphCSR& g, std::vector<int>& outColor) {
    outColor.assign(g.n, -1);
    std::queue<int> q;
    for (int s = 0; s < g.n; ++s) {
        if (outColor[s] != -1) continue;
        outColor[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            int cv = outColor[v];
            for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
                int u = g.adj[ei];
                if (u == v) return false;
                if (outColor[u] == -1) {
                    outColor[u] = cv ^ 1;
                    q.push(u);
                } else if (outColor[u] == cv) {
                    return false;
                }
            }
        }
    }
    return true;
}

static int greedyUnlimitedColoring(const GraphCSR& g, const std::vector<int>& order, std::vector<int>& outColor) {
    outColor.assign(g.n, -1);
    std::vector<int> mark(g.n, -1);
    int stamp = 0;
    int maxColor = -1;

    for (int v : order) {
        ++stamp;
        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            int cu = outColor[u];
            if (cu >= 0) mark[cu] = stamp;
        }
        int c = 0;
        while (c < g.n && mark[c] == stamp) ++c;
        outColor[v] = c;
        if (c > maxColor) maxColor = c;
    }
    return maxColor + 1;
}

static std::vector<int> orderLargestFirst(const GraphCSR& g, const std::vector<int>& deg) {
    std::vector<int> order(g.n);
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (deg[a] != deg[b]) return deg[a] > deg[b];
        return a < b;
    });
    return order;
}

static std::vector<int> orderSmallestLast(const GraphCSR& g, const std::vector<int>& originalDeg) {
    std::vector<int> deg = originalDeg;
    int maxDeg = 0;
    for (int d : deg) maxDeg = std::max(maxDeg, d);

    std::vector<int> head(maxDeg + 1, -1), next(g.n, -1), prev(g.n, -1);
    auto bucketInsert = [&](int v, int d) {
        next[v] = head[d];
        prev[v] = -1;
        if (head[d] != -1) prev[head[d]] = v;
        head[d] = v;
    };
    auto bucketRemove = [&](int v, int d) {
        int pv = prev[v];
        int nv = next[v];
        if (pv != -1) next[pv] = nv;
        else head[d] = nv;
        if (nv != -1) prev[nv] = pv;
        prev[v] = next[v] = -1;
    };

    for (int v = 0; v < g.n; ++v) bucketInsert(v, deg[v]);

    std::vector<unsigned char> removed(g.n, 0);
    std::vector<int> order(g.n);
    int curMin = 0;

    for (int pos = g.n - 1; pos >= 0; --pos) {
        while (curMin <= maxDeg && head[curMin] == -1) ++curMin;
        if (curMin > maxDeg) break; // should not happen

        int v = head[curMin];
        // pop v from bucket curMin
        head[curMin] = next[v];
        if (head[curMin] != -1) prev[head[curMin]] = -1;
        next[v] = prev[v] = -1;

        removed[v] = 1;
        order[pos] = v;

        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            if (removed[u]) continue;
            int du = deg[u];
            if (du <= 0) continue;
            bucketRemove(u, du);
            --deg[u];
            bucketInsert(u, du - 1);
            if (du - 1 < curMin) curMin = du - 1;
        }
    }

    return order;
}

static void buildInitialKColoringGreedyMinConflict(
    const GraphCSR& g,
    int k,
    const std::vector<int>& order,
    std::vector<int>& color,
    std::mt19937& rng) {
    color.assign(g.n, -1);
    std::vector<int> freq(k, 0);

    for (int v : order) {
        std::fill(freq.begin(), freq.end(), 0);
        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            int cu = color[u];
            if (cu >= 0) ++freq[cu];
        }

        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (freq[c] == 0) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            int bestF = std::numeric_limits<int>::max();
            int bestCount = 0;
            for (int c = 0; c < k; ++c) {
                if (freq[c] < bestF) {
                    bestF = freq[c];
                    bestCount = 1;
                    chosen = c;
                } else if (freq[c] == bestF) {
                    ++bestCount;
                    if ((int)(rng() % (uint32_t)bestCount) == 0) chosen = c;
                }
            }
        }
        color[v] = chosen;
    }
}

static bool tabuSearchFixConflicts(
    const GraphCSR& g,
    int k,
    std::vector<int>& color,
    const std::chrono::steady_clock::time_point deadline,
    std::mt19937& rng) {
    const int n = g.n;
    std::vector<int> conflictCount(n, 0);
    std::vector<int> posInList(n, -1);
    std::vector<int> conflictList;
    conflictList.reserve(n / 16 + 16);

    auto addToList = [&](int v) {
        if (posInList[v] != -1) return;
        posInList[v] = (int)conflictList.size();
        conflictList.push_back(v);
    };
    auto removeFromList = [&](int v) {
        int p = posInList[v];
        if (p == -1) return;
        int last = conflictList.back();
        conflictList[p] = last;
        posInList[last] = p;
        conflictList.pop_back();
        posInList[v] = -1;
    };
    auto refreshMembership = [&](int v) {
        if (conflictCount[v] > 0) addToList(v);
        else removeFromList(v);
    };

    long long conflictsEdges2 = 0;
    for (int v = 0; v < n; ++v) {
        int cv = color[v];
        int cnt = 0;
        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            if (color[u] == cv) ++cnt;
        }
        conflictCount[v] = cnt;
        conflictsEdges2 += cnt;
    }
    long long conflictsEdges = conflictsEdges2 / 2;
    for (int v = 0; v < n; ++v) refreshMembership(v);
    if (conflictsEdges == 0) return true;

    std::vector<int> tabuColor(n, -1);
    std::vector<int> tabuUntil(n, 0);

    std::vector<int> freq(k, 0);
    std::vector<int> bestColors;
    bestColors.reserve(k);

    long long bestSeen = conflictsEdges;
    int iter = 0;
    int stagnant = 0;

    while (conflictsEdges > 0) {
        if (std::chrono::steady_clock::now() >= deadline) return false;
        if (conflictList.empty()) return false;

        ++iter;
        // pick a conflicted vertex (biased toward higher conflicts)
        int v = conflictList[(size_t)(rng() % (uint32_t)conflictList.size())];
        const int samples = (int)std::min<size_t>(16, conflictList.size());
        for (int s = 1; s < samples; ++s) {
            int cand = conflictList[(size_t)(rng() % (uint32_t)conflictList.size())];
            if (conflictCount[cand] > conflictCount[v]) v = cand;
        }

        std::fill(freq.begin(), freq.end(), 0);
        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            int cu = color[u];
            if ((unsigned)cu < (unsigned)k) ++freq[cu];
        }

        int curColor = color[v];
        int curConf = freq[curColor];

        long long bestAfter = std::numeric_limits<long long>::max();
        bestColors.clear();

        for (int c = 0; c < k; ++c) {
            if (c == curColor) continue;
            long long after = conflictsEdges + (long long)freq[c] - (long long)curConf;
            const bool isTabu = (tabuColor[v] == c) && (iter < tabuUntil[v]);
            if (isTabu && after >= bestSeen && after != 0) continue; // aspiration

            if (after < bestAfter) {
                bestAfter = after;
                bestColors.clear();
                bestColors.push_back(c);
            } else if (after == bestAfter) {
                bestColors.push_back(c);
            }
        }

        int newColor = curColor;
        if (!bestColors.empty()) {
            newColor = bestColors[(size_t)(rng() % (uint32_t)bestColors.size())];
        } else {
            // No admissible move (rare with our light tabu), pick random non-taboo color.
            int c = (curColor + 1 + (int)(rng() % (uint32_t)(k - 1))) % k;
            newColor = c;
        }

        if (newColor == curColor) {
            // Random kick
            newColor = (curColor + 1 + (int)(rng() % (uint32_t)(k - 1))) % k;
        }

        const int oldColor = curColor;
        const int oldConf = curConf;       // conflicts incident to v before
        const int newConf = freq[newColor]; // conflicts incident to v after
        const long long delta = (long long)newConf - (long long)oldConf;

        // apply move
        color[v] = newColor;
        conflictsEdges += delta;
        conflictCount[v] = newConf;

        // update neighbors
        for (int ei = g.offset[v]; ei < g.offset[v + 1]; ++ei) {
            int u = g.adj[ei];
            if (color[u] == oldColor) {
                --conflictCount[u];
                refreshMembership(u);
            } else if (color[u] == newColor) {
                ++conflictCount[u];
                refreshMembership(u);
            }
        }
        refreshMembership(v);

        // tabu: forbid returning to oldColor for a while
        tabuColor[v] = oldColor;
        tabuUntil[v] = iter + 7 + (int)(rng() % 8);

        if (conflictsEdges < bestSeen) {
            bestSeen = conflictsEdges;
            stagnant = 0;
            if (bestSeen == 0) return true;
        } else {
            ++stagnant;
            if (stagnant > 200000) return false; // trigger restart in caller
        }
    }

    return true;
}

static bool exactKColorBacktrackingSmall(const GraphCSR& g, int k, std::vector<int>& color) {
    if (k <= 0) return false;
    if (k > 64) return false;
    const int n = g.n;
    color.assign(n, -1);

    uint64_t allMask = (k == 64) ? ~0ULL : ((1ULL << k) - 1ULL);
    std::vector<uint64_t> avail(n, allMask);

    auto popcount64 = [](uint64_t x) -> int {
#if defined(_MSC_VER)
        return (int)__popcnt64(x);
#elif defined(__GNUG__) || defined(__clang__)
        return __builtin_popcountll(x);
#else
        int c = 0;
        while (x) {
            x &= (x - 1);
            ++c;
        }
        return c;
#endif
    };

    std::function<bool(int)> dfs = [&](int colored) -> bool {
        if (colored == n) return true;

        int vPick = -1;
        int bestOpts = std::numeric_limits<int>::max();
        int bestDeg = -1;
        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            int opts = popcount64(avail[v]);
            if (opts < bestOpts || (opts == bestOpts && g.degree(v) > bestDeg)) {
                bestOpts = opts;
                bestDeg = g.degree(v);
                vPick = v;
                if (bestOpts <= 1) break;
            }
        }
        if (vPick == -1) return true;
        uint64_t m = avail[vPick];
        if (m == 0) return false;

        // try colors in increasing order (bit index)
        while (m) {
            uint64_t bit = m & (~m + 1ULL);
            int c = 0;
#if defined(_MSC_VER)
            unsigned long idx;
            _BitScanForward64(&idx, bit);
            c = (int)idx;
#elif defined(__GNUG__) || defined(__clang__)
            c = (int)__builtin_ctzll(bit);
#else
            uint64_t t = bit;
            while ((t & 1ULL) == 0ULL) {
                t >>= 1ULL;
                ++c;
            }
#endif

            color[vPick] = c;
            std::vector<std::pair<int, uint64_t>> changed;
            changed.reserve(g.degree(vPick));
            bool ok = true;
            for (int ei = g.offset[vPick]; ei < g.offset[vPick + 1]; ++ei) {
                int u = g.adj[ei];
                if (color[u] != -1) {
                    if (color[u] == c) {
                        ok = false;
                        break;
                    }
                    continue;
                }
                uint64_t before = avail[u];
                uint64_t after = before & ~bit;
                if (after != before) {
                    if (after == 0) {
                        ok = false;
                        break;
                    }
                    avail[u] = after;
                    changed.emplace_back(u, before);
                }
            }
            if (ok && dfs(colored + 1)) return true;

            // undo
            for (auto& p : changed) avail[p.first] = p.second;
            color[vPick] = -1;

            m &= (m - 1ULL);
        }
        return false;
    };

    return dfs(0);
}

int main() {
    FastScanner fs;
    int n = 0;
    long long m = 0;
    int k = 0;
    if (!fs.readInt(n)) return 0;
    if (!fs.readInt(m)) return 0;
    if (!fs.readInt(k)) return 0;
    if (n <= 0) return 0;
    if (k <= 0) return 0;

    std::vector<int> deg(n, 0);
    std::vector<int> U;
    std::vector<int> V;
    U.resize((size_t)m);
    V.resize((size_t)m);

    bool hasSelfLoop = false;
    for (long long i = 0; i < m; ++i) {
        int u = 0, v = 0;
        fs.readInt(u);
        fs.readInt(v);
        if (u == v) hasSelfLoop = true;
        U[(size_t)i] = u;
        V[(size_t)i] = v;
        if ((unsigned)u < (unsigned)n) ++deg[u];
        if ((unsigned)v < (unsigned)n) ++deg[v];
    }

    GraphCSR g;
    g.n = n;
    g.offset.assign((size_t)n + 1, 0);
    for (int i = 0; i < n; ++i) g.offset[(size_t)i + 1] = g.offset[(size_t)i] + deg[i];
    g.adj.assign((size_t)g.offset[(size_t)n], 0);

    std::vector<int> cur = g.offset;
    for (long long i = 0; i < m; ++i) {
        int u = U[(size_t)i];
        int v = V[(size_t)i];
        if ((unsigned)u < (unsigned)n) g.adj[(size_t)cur[u]++] = v;
        if ((unsigned)v < (unsigned)n) g.adj[(size_t)cur[v]++] = u;
    }
    U.clear();
    V.clear();
    U.shrink_to_fit();
    V.shrink_to_fit();

    auto start = std::chrono::steady_clock::now();
    auto deadline = start + std::chrono::milliseconds(295000); // ~4m55s budget

    std::vector<int> color;

    if (m == 0) {
        color.assign(n, 0);
    } else if (k == 1) {
        color.assign(n, 0);
    } else if (!hasSelfLoop && k == 2) {
        if (!bipartiteColoring(g, color)) {
            // Fall back to heuristic (may still fail if not 2-colorable).
            std::mt19937 rng((uint32_t)std::chrono::high_resolution_clock::now().time_since_epoch().count());
            color.resize(n);
            for (int v = 0; v < n; ++v) color[v] = (int)(rng() % 2);
            tabuSearchFixConflicts(g, 2, color, deadline, rng);
        }
    } else if (n <= 300 && k <= 64 && !hasSelfLoop) {
        if (!exactKColorBacktrackingSmall(g, k, color)) {
            // If exact fails or times out (not timed), fall back to heuristic.
            std::mt19937 rng((uint32_t)std::chrono::high_resolution_clock::now().time_since_epoch().count());
            auto order = orderLargestFirst(g, deg);
            buildInitialKColoringGreedyMinConflict(g, k, order, color, rng);
            tabuSearchFixConflicts(g, k, color, deadline, rng);
        }
    } else {
        std::mt19937 rng((uint32_t)std::chrono::high_resolution_clock::now().time_since_epoch().count() ^
                         (uint32_t)(n * 2654435761u) ^ (uint32_t)(m));

        // Try a few fast proper colorings; if any uses <=k colors, we are done.
        std::vector<int> bestProper;
        bestProper.reserve(n);

        {
            auto order = orderSmallestLast(g, deg);
            std::vector<int> tmp;
            int used = greedyUnlimitedColoring(g, order, tmp);
            if (used <= k) color = std::move(tmp);
        }
        if (color.empty()) {
            auto order = orderLargestFirst(g, deg);
            std::vector<int> tmp;
            int used = greedyUnlimitedColoring(g, order, tmp);
            if (used <= k) color = std::move(tmp);
        }

        // If still not solved, run tabu/min-conflicts search with restarts.
        if (color.empty()) {
            auto orderDeg = orderLargestFirst(g, deg);
            auto orderSL = orderSmallestLast(g, deg);

            std::vector<int> work;
            bool solved = false;
            const int maxRestarts = 8;
            for (int r = 0; r < maxRestarts && std::chrono::steady_clock::now() < deadline; ++r) {
                if (r == 0) buildInitialKColoringGreedyMinConflict(g, k, orderDeg, work, rng);
                else if (r == 1) buildInitialKColoringGreedyMinConflict(g, k, orderSL, work, rng);
                else {
                    work.resize(n);
                    for (int v = 0; v < n; ++v) work[v] = (int)(rng() % (uint32_t)k);
                }
                if (tabuSearchFixConflicts(g, k, work, deadline, rng)) {
                    color = std::move(work);
                    solved = true;
                    break;
                }
            }
            if (!solved && color.empty()) {
                // Last resort: output a bounded greedy (may be invalid if k too small).
                color.resize(n);
                for (int v = 0; v < n; ++v) color[v] = v % k;
            }
        }
    }

    // Final safety net: if something went wrong, output a trivial in-range assignment.
    if (!verifyColoring(g, color, k)) {
        // Try one last repair if feasible
        std::mt19937 rng((uint32_t)std::chrono::high_resolution_clock::now().time_since_epoch().count());
        if ((int)color.size() != n) {
            color.assign(n, 0);
        } else {
            for (int v = 0; v < n; ++v) {
                if (color[v] < 0 || color[v] >= k) color[v] = (int)(rng() % (uint32_t)k);
            }
        }
        if (k >= 2) tabuSearchFixConflicts(g, k, color, deadline, rng);
        if (!verifyColoring(g, color, k)) {
            for (int v = 0; v < n; ++v) color[v] = 0;
        }
    }

    FastOutput out;
    for (int i = 0; i < n; ++i) {
        if (i) out.writeChar(' ');
        out.writeInt(color[i]);
    }
    out.writeChar('\n');
    out.flush();
    return 0;
}
