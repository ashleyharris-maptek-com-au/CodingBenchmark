use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};
use std::time::{Duration, Instant};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            x = x * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(x)
    }
}

#[derive(Clone, Copy)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let seed = if seed == 0 { 0x9e3779b97f4a7c15 } else { seed };
        Self { state: seed }
    }

    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }

    #[inline]
    fn next_u32(&mut self) -> u32 {
        (self.next_u64() >> 32) as u32
    }

    #[inline]
    fn gen_range_usize(&mut self, upper_exclusive: usize) -> usize {
        if upper_exclusive <= 1 {
            return 0;
        }
        // Slight modulo bias is fine for heuristics.
        (self.next_u64() as usize) % upper_exclusive
    }
}

fn shuffle_in_place<T>(arr: &mut [T], rng: &mut XorShift64) {
    for i in (1..arr.len()).rev() {
        let j = rng.gen_range_usize(i + 1);
        arr.swap(i, j);
    }
}

#[derive(Clone, Copy)]
struct HeapItem {
    score: i32,
    degree: usize,
    rand: u32,
    v: usize,
    stamp: u32,
}

impl PartialEq for HeapItem {
    fn eq(&self, other: &Self) -> bool {
        self.score == other.score
            && self.degree == other.degree
            && self.rand == other.rand
            && self.v == other.v
            && self.stamp == other.stamp
    }
}

impl Eq for HeapItem {}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        self.score
            .cmp(&other.score)
            .then_with(|| self.degree.cmp(&other.degree))
            .then_with(|| self.rand.cmp(&other.rand))
            .then_with(|| self.v.cmp(&other.v))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn add_vertex_update(
    v: usize,
    adj: &[Vec<usize>],
    degrees: &[usize],
    in_solution: &[bool],
    dominated: &mut [bool],
    dominated_count: &mut usize,
    score: &mut [i32],
    stamp: &mut [u32],
    heap: &mut BinaryHeap<HeapItem>,
    rng: &mut XorShift64,
    scratch_newly_dominated: &mut Vec<usize>,
) {
    scratch_newly_dominated.clear();

    let mut mark = |x: usize| {
        if !dominated[x] {
            dominated[x] = true;
            *dominated_count += 1;
            scratch_newly_dominated.push(x);
        }
    };

    mark(v);
    for &x in &adj[v] {
        mark(x);
    }

    let mut update_candidate = |a: usize| {
        if in_solution[a] {
            return;
        }
        let s = score[a];
        if s <= 0 {
            // Shouldn't happen with deduped adjacency, but keep it safe.
            return;
        }
        score[a] = s - 1;
        stamp[a] = stamp[a].wrapping_add(1);
        if score[a] > 0 {
            heap.push(HeapItem {
                score: score[a],
                degree: degrees[a],
                rand: rng.next_u32(),
                v: a,
                stamp: stamp[a],
            });
        }
    };

    for &x in scratch_newly_dominated.iter() {
        update_candidate(x);
        for &a in &adj[x] {
            update_candidate(a);
        }
    }
}

fn greedy_dominating_set(
    adj: &[Vec<usize>],
    degrees: &[usize],
    seed: u64,
    initial: Option<&[usize]>,
) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let mut rng = XorShift64::new(seed);
    let mut in_solution = vec![false; n];
    let mut solution = Vec::new();

    if let Some(init) = initial {
        for &v in init {
            if v < n && !in_solution[v] {
                in_solution[v] = true;
                solution.push(v);
            }
        }
    }

    let mut dominated = vec![false; n];
    let mut dominated_count = 0usize;
    let mut score: Vec<i32> = degrees.iter().map(|&d| (d as i32) + 1).collect();
    let mut stamp = vec![0u32; n];

    let mut heap = BinaryHeap::new();
    for v in 0..n {
        if !in_solution[v] {
            heap.push(HeapItem {
                score: score[v],
                degree: degrees[v],
                rand: rng.next_u32(),
                v,
                stamp: stamp[v],
            });
        }
    }

    let mut scratch_newly_dominated = Vec::new();
    for &v in solution.iter() {
        add_vertex_update(
            v,
            adj,
            degrees,
            &in_solution,
            &mut dominated,
            &mut dominated_count,
            &mut score,
            &mut stamp,
            &mut heap,
            &mut rng,
            &mut scratch_newly_dominated,
        );
    }

    while dominated_count < n {
        let mut chosen: Option<usize> = None;
        while let Some(it) = heap.pop() {
            if in_solution[it.v] {
                continue;
            }
            if it.stamp != stamp[it.v] {
                continue;
            }
            if it.score != score[it.v] {
                continue;
            }
            if it.score <= 0 {
                continue;
            }
            chosen = Some(it.v);
            break;
        }

        let v = match chosen {
            Some(v) => v,
            None => match dominated.iter().position(|&d| !d) {
                Some(u) => u,
                None => break,
            },
        };

        if in_solution[v] {
            continue;
        }
        in_solution[v] = true;
        solution.push(v);

        add_vertex_update(
            v,
            adj,
            degrees,
            &in_solution,
            &mut dominated,
            &mut dominated_count,
            &mut score,
            &mut stamp,
            &mut heap,
            &mut rng,
            &mut scratch_newly_dominated,
        );
    }

    // Final safety repair: ensure domination even if something went wrong.
    let mut dominated2 = vec![false; n];
    for &v in &solution {
        dominated2[v] = true;
        for &u in &adj[v] {
            dominated2[u] = true;
        }
    }
    for u in 0..n {
        if !dominated2[u] {
            in_solution[u] = true;
            solution.push(u);
            dominated2[u] = true;
            for &x in &adj[u] {
                dominated2[x] = true;
            }
        }
    }

    solution
}

fn purify_with_order(adj: &[Vec<usize>], solution: &[usize], order: &[usize]) -> Vec<usize> {
    let n = adj.len();
    let mut in_solution = vec![false; n];
    for &v in solution {
        if v < n {
            in_solution[v] = true;
        }
    }

    let mut dom_count = vec![0u32; n];
    for &v in solution {
        if v >= n {
            continue;
        }
        dom_count[v] += 1;
        for &u in &adj[v] {
            dom_count[u] += 1;
        }
    }

    let can_remove = |v: usize, in_solution: &[bool], dom_count: &[u32], adj: &[Vec<usize>]| {
        if !in_solution[v] {
            return false;
        }
        if dom_count[v] <= 1 {
            return false;
        }
        for &u in &adj[v] {
            if dom_count[u] == 1 {
                return false;
            }
        }
        true
    };

    for &v in order {
        if v >= n {
            continue;
        }
        if !can_remove(v, &in_solution, &dom_count, adj) {
            continue;
        }
        in_solution[v] = false;
        dom_count[v] -= 1;
        for &u in &adj[v] {
            dom_count[u] -= 1;
        }
    }

    let mut out = Vec::new();
    out.reserve(solution.len());
    for v in 0..n {
        if in_solution[v] {
            out.push(v);
        }
    }
    // Keep output order somewhat stable (not required).
    out.sort_unstable();
    // Avoid an empty dominating set on non-empty graphs (should not happen).
    if out.is_empty() && n > 0 {
        out.push(0);
    }
    // Optional: avoid leaving uncovered due to any bug; repair.
    ensure_dominating_set(adj, &mut out);
    out
}

fn ensure_dominating_set(adj: &[Vec<usize>], sol: &mut Vec<usize>) {
    let n = adj.len();
    if n == 0 {
        sol.clear();
        return;
    }
    let mut in_solution = vec![false; n];
    for &v in sol.iter() {
        if v < n {
            in_solution[v] = true;
        }
    }
    let mut dominated = vec![false; n];
    for v in 0..n {
        if in_solution[v] {
            dominated[v] = true;
            for &u in &adj[v] {
                dominated[u] = true;
            }
        }
    }
    for u in 0..n {
        if !dominated[u] {
            in_solution[u] = true;
            sol.push(u);
            dominated[u] = true;
            for &x in &adj[u] {
                dominated[x] = true;
            }
        }
    }
    sol.sort_unstable();
    sol.dedup();
}

fn refine_solution(adj: &[Vec<usize>], degrees: &[usize], sol: &[usize], seed: u64) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let mut rng = XorShift64::new(seed ^ 0x6a09e667f3bcc909);
    let mut best = sol.to_vec();
    ensure_dominating_set(adj, &mut best);

    // Degree ascending purification.
    let mut order = best.clone();
    order.sort_by_key(|&v| degrees[v]);
    let p1 = purify_with_order(adj, &best, &order);
    if p1.len() < best.len() {
        best = p1;
    }

    // Degree descending purification.
    let mut order = best.clone();
    order.sort_by_key(|&v| std::cmp::Reverse(degrees[v]));
    let p2 = purify_with_order(adj, &best, &order);
    if p2.len() < best.len() {
        best = p2;
    }

    // A couple randomized purification orders.
    for _ in 0..2 {
        let mut order = best.clone();
        shuffle_in_place(&mut order, &mut rng);
        let p = purify_with_order(adj, &best, &order);
        if p.len() < best.len() {
            best = p;
        }
    }

    best
}

fn high_degree_maximal_independent_set(adj: &[Vec<usize>], degrees: &[usize], seed: u64) -> Vec<usize> {
    let n = adj.len();
    let mut rng = XorShift64::new(seed ^ 0x3c6ef372fe94f82b);
    let mut order: Vec<usize> = (0..n).collect();
    shuffle_in_place(&mut order, &mut rng);
    // Stable sort: keep shuffled tie-breaking for equal degrees.
    order.sort_by_key(|&v| std::cmp::Reverse(degrees[v]));

    let mut forbidden = vec![false; n];
    let mut sol = Vec::new();
    for &v in &order {
        if !forbidden[v] {
            sol.push(v);
            forbidden[v] = true;
            for &u in &adj[v] {
                forbidden[u] = true;
            }
        }
    }
    sol.sort_unstable();
    sol.dedup();
    ensure_dominating_set(adj, &mut sol);
    sol
}

fn perturb_and_rebuild(
    adj: &[Vec<usize>],
    degrees: &[usize],
    base: &[usize],
    seed: u64,
) -> Vec<usize> {
    if base.is_empty() {
        return Vec::new();
    }
    let mut rng = XorShift64::new(seed ^ 0x510e527fade682d1);
    let mut tmp = base.to_vec();
    shuffle_in_place(&mut tmp, &mut rng);
    let k = (tmp.len() / 10).max(1).min(tmp.len());
    let initial: Vec<usize> = tmp[k..].to_vec();
    let sol = greedy_dominating_set(adj, degrees, seed ^ 0xa54ff53a5f1d36f1, Some(&initial));
    refine_solution(adj, degrees, &sol, seed ^ 0x1f83d9abfb41bd6b)
}

fn solve(adj: &[Vec<usize>]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }
    let degrees: Vec<usize> = adj.iter().map(|x| x.len()).collect();
    let m_undirected: usize = degrees.iter().sum::<usize>() / 2;

    let start = Instant::now();
    let time_limit = Duration::from_secs(290);

    let base_seed = (n as u64)
        .wrapping_mul(0x9e3779b97f4a7c15)
        ^ (m_undirected as u64).wrapping_mul(0xbf58476d1ce4e5b9)
        ^ 0x94d049bb133111eb;

    // Baseline greedy.
    let mut best = greedy_dominating_set(adj, &degrees, base_seed, None);
    best = refine_solution(adj, &degrees, &best, base_seed ^ 0x1234_5678_9abc_def0);

    // High-degree MIS candidate (often good for dense graphs).
    let mis = high_degree_maximal_independent_set(adj, &degrees, base_seed ^ 0xfeed_face_cafe_beef);
    let mis = refine_solution(adj, &degrees, &mis, base_seed ^ 0x0f0f_0f0f_0f0f_0f0f);
    if mis.len() < best.len() {
        best = mis;
    }

    // Multi-start / perturbation loop with a conservative iteration cap.
    let max_iter = if n <= 2_000 {
        200usize
    } else if n <= 20_000 {
        50usize
    } else {
        10usize
    };

    let mut iter = 0usize;
    while iter < max_iter && start.elapsed() < time_limit {
        let seed = base_seed ^ ((iter as u64 + 1).wrapping_mul(0x9e3779b97f4a7c15));
        let cand = greedy_dominating_set(adj, &degrees, seed, None);
        let cand = refine_solution(adj, &degrees, &cand, seed ^ 0xd6e8_feb8_6659_fd93);
        if cand.len() < best.len() {
            best = cand;
        }

        if start.elapsed() < time_limit && iter % 5 == 0 && best.len() >= 2 {
            let cand2 = perturb_and_rebuild(adj, &degrees, &best, seed ^ 0xa076_1d64_78bd_642f);
            if cand2.len() < best.len() {
                best = cand2;
            }
        }

        iter += 1;
    }

    ensure_dominating_set(adj, &mut best);
    best
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(x) => x,
        None => {
            println!("0\n");
            return;
        }
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u = match sc.next_usize() {
            Some(x) => x,
            None => break,
        };
        let v = match sc.next_usize() {
            Some(x) => x,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Deduplicate adjacency to avoid overcounting in score updates.
    for nbrs in adj.iter_mut() {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    let mut sol = solve(&adj);
    sol.sort_unstable();
    sol.dedup();

    let mut out = String::new();
    out.push_str(&format!("{}\n", sol.len()));
    for (i, v) in sol.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');
    print!("{out}");
}
