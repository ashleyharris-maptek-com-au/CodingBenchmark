using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var input = Console.OpenStandardInput();
        var scanner = new FastScanner(input);

        long xCountL = scanner.ReadInt64Token();
        long yCountL = scanner.ReadInt64Token();
        long zCountL = scanner.ReadInt64Token();
        _ = scanner.ReadDoubleToken(); // XBlockSize
        _ = scanner.ReadDoubleToken(); // YBlockSize
        _ = scanner.ReadDoubleToken(); // ZBlockSize

        scanner.ConsumeToEndOfLine();

        if (xCountL <= 0 || yCountL <= 0 || zCountL <= 0 || xCountL > int.MaxValue || yCountL > int.MaxValue || zCountL > int.MaxValue)
        {
            WriteNoOps();
            return;
        }

        int xCount = (int)xCountL;
        int yCount = (int)yCountL;
        int zCount = (int)zCountL;

        long xy = SaturatingMul(xCountL, yCountL);
        long total = SaturatingMul(xy, zCountL);

        // Adaptive strategy:
        // - Tiny worlds: read a couple of neighbor columns and mine any ore in (0,0,*) that is exposed from above OR by an empty neighbor.
        // - Large worlds: only mine the topmost contiguous ore run at (0,0,*) (always valid without knowing neighbors).
        bool tinyWorld = total <= 5_000_000;

        byte[] col00 = new byte[zCount];
        scanner.ReadBytes(col00);

        byte[]? col01 = null;
        byte[]? col10 = null;

        if (tinyWorld)
        {
            if (yCount > 1)
            {
                col01 = new byte[zCount];
                scanner.ReadBytes(col01);
            }

            if (xCount > 1)
            {
                int yAlreadyRead = 1 + (col01 is null ? 0 : 1);
                long toSkip = (long)Math.Max(0, yCount - yAlreadyRead) * zCount;
                scanner.SkipBytes(toSkip);

                col10 = new byte[zCount];
                scanner.ReadBytes(col10);
            }
        }

        bool[] dig = new bool[zCount];

        // Strategy A: side-exposed (requires tiny mode neighbor reads).
        if (tinyWorld && (col01 is not null || col10 is not null))
        {
            for (int z = 0; z < zCount; z++)
            {
                byte d = col00[z];
                int mass = d & 0x0F;
                if (mass == 0)
                    continue;

                int value = d >> 4;
                if (value <= 0)
                    continue;

                bool sideExposed =
                    (col01 is not null && ((col01[z] & 0x0F) == 0)) ||
                    (col10 is not null && ((col10[z] & 0x0F) == 0));

                if (sideExposed)
                    dig[z] = true;
            }
        }

        // Strategy B: top-down contiguous valuable ore run from the surface (always safe).
        bool hitFirstSolid = false;
        for (int z = zCount - 1; z >= 0; z--)
        {
            byte d = col00[z];
            int mass = d & 0x0F;
            if (mass == 0)
            {
                if (!hitFirstSolid)
                    continue; // still above the surface
                // Once we've hit the first solid, internal voids mean the next solid below is exposed-from-above.
                continue;
            }

            hitFirstSolid = true;
            int value = d >> 4;
            if (value <= 0)
                break;

            dig[z] = true;
        }

        // Emit operations in descending z so that any "from above" digs are always valid in sequence.
        int opCount = 0;
        for (int z = 0; z < zCount; z++)
        {
            if (dig[z])
                opCount++;
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(encoderShouldEmitUTF8Identifier: false), bufferSize: 1 << 16)
        {
            NewLine = "\n"
        };

        writer.WriteLine(opCount);
        for (int z = zCount - 1; z >= 0; z--)
        {
            if (!dig[z])
                continue;

            // N=0 steps; dump straight to the processing plant at (0,0).
            writer.Write("0 0 0 ");
            writer.Write(z);
            writer.WriteLine(" 0 0");
        }
    }

    private static void WriteNoOps()
    {
        Console.Out.NewLine = "\n";
        Console.WriteLine("0");
    }

    private static long SaturatingMul(long a, long b)
    {
        if (a == 0 || b == 0)
            return 0;
        if (a > long.MaxValue / b)
            return long.MaxValue;
        return a * b;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public long ReadInt64Token()
        {
            SkipWhitespace();
            int b = PeekRawByte();
            if (b < 0)
                return 0;

            long sign = 1;
            if (b == '-')
            {
                sign = -1;
                _pos++;
            }

            long value = 0;
            while (true)
            {
                b = PeekRawByte();
                if (b < 0 || b <= 32)
                    break;

                value = (value * 10) + (b - '0');
                _pos++;
            }

            return value * sign;
        }

        public double ReadDoubleToken()
        {
            Span<char> tmp = stackalloc char[64];
            int n = 0;

            SkipWhitespace();
            while (true)
            {
                int b = PeekRawByte();
                if (b < 0 || b <= 32)
                    break;
                if (n < tmp.Length)
                    tmp[n++] = (char)b;
                _pos++;
            }

            return double.Parse(tmp.Slice(0, n), CultureInfo.InvariantCulture);
        }

        public void ConsumeToEndOfLine()
        {
            int b;
            do
            {
                b = ReadRawByte();
                if (b < 0)
                    return;
            } while (b != '\n');
        }

        public void ReadBytes(Span<byte> dest)
        {
            int written = 0;
            while (written < dest.Length)
            {
                int avail = _len - _pos;
                if (avail <= 0)
                {
                    if (!Fill())
                        throw new EndOfStreamException();
                    avail = _len - _pos;
                }

                int toCopy = Math.Min(avail, dest.Length - written);
                new ReadOnlySpan<byte>(_buffer, _pos, toCopy).CopyTo(dest.Slice(written, toCopy));
                _pos += toCopy;
                written += toCopy;
            }
        }

        public void SkipBytes(long count)
        {
            while (count > 0)
            {
                int avail = _len - _pos;
                if (avail <= 0)
                {
                    if (!Fill())
                        return;
                    avail = _len - _pos;
                    if (avail <= 0)
                        return;
                }

                int step = (int)Math.Min((long)avail, count);
                _pos += step;
                count -= step;
            }
        }

        private void SkipWhitespace()
        {
            while (true)
            {
                int b = PeekRawByte();
                if (b < 0 || b > 32)
                    return;
                _pos++;
            }
        }

        private int PeekRawByte()
        {
            if (_pos >= _len && !Fill())
                return -1;
            return _buffer[_pos];
        }

        private int ReadRawByte()
        {
            if (_pos >= _len && !Fill())
                return -1;
            return _buffer[_pos++];
        }

        private bool Fill()
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _pos = 0;
            return _len > 0;
        }
    }
}
