'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
if (input.length === 0) {
  process.stdout.write('{"schedule":[]}');
  process.exit(0);
}

const cleanInput = input.charCodeAt(0) === 0xfeff ? input.slice(1) : input;
const data = JSON.parse(cleanInput);
const tasksIn = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksIn.length;

const machineCountRaw = data.machineCount;
const machineCount =
  typeof machineCountRaw === 'number' && isFinite(machineCountRaw) && machineCountRaw > 0
    ? Math.floor(machineCountRaw)
    : 1;

if (n === 0) {
  process.stdout.write('{"schedule":[]}');
  process.exit(0);
}

class BinaryHeap {
  constructor(less) {
    this.less = less;
    this.a = [];
  }
  size() {
    return this.a.length;
  }
  push(x) {
    const a = this.a;
    const less = this.less;
    let i = a.length;
    a.push(x);
    while (i > 0) {
      const p = (i - 1) >> 1;
      const ap = a[p];
      if (!less(x, ap)) break;
      a[i] = ap;
      i = p;
    }
    a[i] = x;
  }
  pop() {
    const a = this.a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const x = a.pop();
    if (n === 1) return top;

    const less = this.less;
    const last = x;
    let i = 0;
    const half = (n - 1) >> 1;
    while (i < half) {
      let left = (i << 1) + 1;
      let right = left + 1;
      let best = left;
      if (right < n - 1 && less(a[right], a[left])) best = right;
      if (!less(a[best], last)) break;
      a[i] = a[best];
      i = best;
    }
    a[i] = last;
    return top;
  }
}

// Parse tasks and build graph
const ids = new Array(n);
const duration = new Float64Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);
const succ = new Array(n);
for (let i = 0; i < n; i++) succ[i] = [];

const idToIndex = new Map();
for (let i = 0; i < n; i++) {
  const t = tasksIn[i] || {};
  ids[i] = t.id;
  idToIndex.set(t.id, i);

  const p = Number(t.duration);
  duration[i] = isFinite(p) && p > 0 ? p : 0;

  const d = Number(t.deadline);
  deadline[i] = isFinite(d) ? d : 0;

  const w = Number(t.weight);
  weight[i] = isFinite(w) && w > 0 ? w : 0;
}

const indegree = new Int32Array(n);
for (let i = 0; i < n; i++) {
  const t = tasksIn[i] || {};
  const deps = Array.isArray(t.deps) ? t.deps : [];
  for (let k = 0; k < deps.length; k++) {
    const depId = deps[k];
    const j = idToIndex.get(depId);
    if (j === undefined) continue;
    if (j === i) continue;
    indegree[i]++;
    succ[j].push(i);
  }
}

// Topological pass for earliest-start lower bound, and cycle detection
const es = new Float64Array(n);
const indegTmp = new Int32Array(indegree);
const q = new Int32Array(n);
let head = 0;
let tail = 0;
for (let i = 0; i < n; i++) if (indegTmp[i] === 0) q[tail++] = i;

const topo = new Int32Array(n);
let topoLen = 0;
while (head < tail) {
  const u = q[head++];
  topo[topoLen++] = u;
  const finish = es[u] + duration[u];
  const su = succ[u];
  for (let idx = 0; idx < su.length; idx++) {
    const v = su[idx];
    if (es[v] < finish) es[v] = finish;
    const nd = --indegTmp[v];
    if (nd === 0) q[tail++] = v;
  }
}

if (topoLen !== n) {
  // Cyclic/invalid dependency graph; fall back to a stable "order" output.
  let out = '{"order":[';
  for (let i = 0; i < n; i++) {
    if (i) out += ',';
    out += JSON.stringify(ids[i]);
  }
  out += ']}';
  process.stdout.write(out);
  process.exit(0);
}

// Lower-bound slack and critical-path style rank.
const slackLB = new Float64Array(n);
let sumDur = 0;
for (let i = 0; i < n; i++) {
  sumDur += duration[i];
  slackLB[i] = deadline[i] - (es[i] + duration[i]);
}

const rank = new Float64Array(n);
for (let ti = n - 1; ti >= 0; ti--) {
  const u = topo[ti];
  let best = 0;
  const su = succ[u];
  for (let idx = 0; idx < su.length; idx++) {
    const v = su[idx];
    const rv = rank[v];
    if (rv > best) best = rv;
  }
  rank[u] = duration[u] + best;
}

// Static ATC-like priority index using earliest-start lower bound.
const pBar = sumDur / n;
const atc = new Float64Array(n);
const atcDen = 2.0 * (pBar > 0 ? pBar : 1);
for (let i = 0; i < n; i++) {
  const p = duration[i] > 0 ? duration[i] : 1;
  const base = weight[i] > 0 ? weight[i] / p : 0;
  if (base === 0) {
    atc[i] = 0;
    continue;
  }
  const s = slackLB[i];
  const expo = s > 0 ? Math.exp(-s / atcDen) : 1;
  atc[i] = base * expo;
}

function scheduleWith(lessTask) {
  const remaining = new Int32Array(indegree);
  const ready = new BinaryHeap(lessTask);

  for (let i = 0; i < n; i++) if (remaining[i] === 0) ready.push(i);

  const freeMachines = [];
  for (let m = 0; m < machineCount; m++) freeMachines.push(m);

  const machineFinish = new Float64Array(machineCount);
  const machineTask = new Int32Array(machineCount);
  const running = new BinaryHeap((a, b) => {
    const fa = machineFinish[a];
    const fb = machineFinish[b];
    return fa < fb || (fa === fb && a < b);
  });

  const orderTask = new Array(n);
  const orderStart = new Float64Array(n);
  const orderMachine = new Int32Array(n);
  let scheduledCount = 0;

  let time = 0;
  let done = 0;
  let totalWeightedTardiness = 0;
  let makespan = 0;

  while (done < n) {
    while (freeMachines.length > 0 && ready.size() > 0) {
      const task = ready.pop();
      const machine = freeMachines.pop();

      orderTask[scheduledCount] = task;
      orderStart[scheduledCount] = time;
      orderMachine[scheduledCount] = machine;
      scheduledCount++;

      const finish = time + duration[task];
      machineFinish[machine] = finish;
      machineTask[machine] = task;
      running.push(machine);

      if (finish > deadline[task]) totalWeightedTardiness += weight[task] * (finish - deadline[task]);
      if (finish > makespan) makespan = finish;
    }

    if (done >= n) break;
    if (running.size() === 0) break; // Shouldn't happen for valid DAGs.

    const firstMachine = running.pop();
    time = machineFinish[firstMachine];

    const finishTime = time;
    let m = firstMachine;
    while (true) {
      const task = machineTask[m];
      done++;
      freeMachines.push(m);

      const su = succ[task];
      for (let idx = 0; idx < su.length; idx++) {
        const v = su[idx];
        const nd = --remaining[v];
        if (nd === 0) ready.push(v);
      }

      if (running.size() === 0) break;
      const nextMachine = running.a[0];
      if (machineFinish[nextMachine] !== finishTime) break;
      m = running.pop();
    }
  }

  // If we couldn't schedule everything (invalid deps), fall back to order output.
  if (scheduledCount !== n) {
    const fallback = new Array(n);
    for (let i = 0; i < n; i++) fallback[i] = i;
    return { kind: 'order', order: fallback, score: Number.POSITIVE_INFINITY };
  }

  return {
    kind: 'schedule',
    orderTask,
    orderStart,
    orderMachine,
    score: totalWeightedTardiness + makespan,
  };
}

// Candidate heuristics (all static orderings over the ready set)
let best = scheduleWith((a, b) => {
  const ra = rank[a];
  const rb = rank[b];
  if (ra !== rb) return ra > rb;
  const wa = weight[a];
  const wb = weight[b];
  if (wa !== wb) return wa > wb;
  const sa = slackLB[a];
  const sb = slackLB[b];
  if (sa !== sb) return sa < sb;
  const pa = duration[a];
  const pb = duration[b];
  if (pa !== pb) return pa > pb;
  return a < b;
});

{
  const cand = scheduleWith((a, b) => {
    const sa = slackLB[a];
    const sb = slackLB[b];
    if (sa !== sb) return sa < sb;
    const wa = weight[a];
    const wb = weight[b];
    if (wa !== wb) return wa > wb;
    const ra = rank[a];
    const rb = rank[b];
    if (ra !== rb) return ra > rb;
    const pa = duration[a];
    const pb = duration[b];
    if (pa !== pb) return pa > pb;
    return a < b;
  });
  if (cand.score < best.score) best = cand;
}

{
  const cand = scheduleWith((a, b) => {
    const ia = atc[a];
    const ib = atc[b];
    if (ia !== ib) return ia > ib;
    const sa = slackLB[a];
    const sb = slackLB[b];
    if (sa !== sb) return sa < sb;
    const ra = rank[a];
    const rb = rank[b];
    if (ra !== rb) return ra > rb;
    const pa = duration[a];
    const pb = duration[b];
    if (pa !== pb) return pa > pb;
    return a < b;
  });
  if (cand.score < best.score) best = cand;
}

if (best.kind !== 'schedule') {
  let out = '{"order":[';
  for (let i = 0; i < n; i++) {
    if (i) out += ',';
    out += JSON.stringify(ids[i]);
  }
  out += ']}';
  process.stdout.write(out);
  process.exit(0);
}

// Stream JSON output to avoid huge intermediate objects/strings.
let buffer = '{"schedule":[';
let first = true;
for (let i = 0; i < n; i++) {
  const task = best.orderTask[i];
  const start = best.orderStart[i];
  const machine = best.orderMachine[i];
  if (!first) buffer += ',';
  first = false;
  buffer += '{"id":' + JSON.stringify(ids[task]) + ',"start":' + String(start) + ',"machine":' + String(machine) + '}';
  if (buffer.length > (1 << 20)) {
    process.stdout.write(buffer);
    buffer = '';
  }
}
buffer += ']}';
process.stdout.write(buffer);
