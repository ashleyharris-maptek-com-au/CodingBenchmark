#include <algorithm>
#include <array>
#include <cstdint>
#include <cstdio>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

static inline void trim_cr(std::string& s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
}

static std::string lcs_dp(const std::string& a, const std::string& b) {
    if (a.empty() || b.empty()) return "";

    const std::string* pa = &a;
    const std::string* pb = &b;
    if (pb->size() > pa->size()) std::swap(pa, pb);
    const std::string& s = *pa;
    const std::string& t = *pb;

    std::vector<int> dp(t.size() + 1, 0);
    int best_len = 0;
    std::size_t best_end = 0;  // end position in s (exclusive)

    for (std::size_t i = 1; i <= s.size(); ++i) {
        int prev = 0;
        const char sc = s[i - 1];
        for (std::size_t j = 1; j <= t.size(); ++j) {
            const int saved = dp[j];
            if (sc == t[j - 1]) {
                dp[j] = prev + 1;
                if (dp[j] > best_len) {
                    best_len = dp[j];
                    best_end = i;
                }
            } else {
                dp[j] = 0;
            }
            prev = saved;
        }
    }

    if (best_len == 0) return "";
    return s.substr(best_end - static_cast<std::size_t>(best_len), static_cast<std::size_t>(best_len));
}

class SuffixAutomaton {
public:
    struct Edge {
        int32_t to;
        int32_t next;
        unsigned char ch;
    };

    explicit SuffixAutomaton(std::size_t expected_len = 0) {
        root_dense.fill(-1);
        const std::size_t max_size_t = (std::numeric_limits<std::size_t>::max)();
        if (expected_len > 0 && expected_len <= (max_size_t - 1) / 2) {
            const std::size_t st_reserve = expected_len * 2 + 1;
            len.reserve(st_reserve);
            link.reserve(st_reserve);
            head.reserve(st_reserve);
            firstpos.reserve(st_reserve);
        }
        if (expected_len > 0 && expected_len <= (max_size_t - 1) / 3) {
            edges.reserve(expected_len * 3 + 1);
        }
        new_state(0);
        last = 0;
    }

    void build(const std::string& s) {
        for (int32_t i = 0; i < static_cast<int32_t>(s.size()); ++i) {
            extend(static_cast<unsigned char>(s[static_cast<std::size_t>(i)]), i);
        }
        build_order(static_cast<int32_t>(s.size()));
        prepare_arrays();
    }

    void apply(const std::string& t) {
        int32_t v = 0;
        int32_t l = 0;
        for (unsigned char c : t) {
            int32_t nxt = transition(v, c);
            while (v != 0 && nxt == -1) {
                v = link[static_cast<std::size_t>(v)];
                l = len[static_cast<std::size_t>(v)];
                nxt = transition(v, c);
            }
            if (nxt != -1) {
                v = nxt;
                ++l;
                set_best(v, l);
            } else {
                v = 0;
                l = 0;
            }
        }

        for (int32_t i = static_cast<int32_t>(order.size()) - 1; i >= 0; --i) {
            const int32_t s = order[static_cast<std::size_t>(i)];
            const int32_t b = get_best(s);
            if (b < mn[static_cast<std::size_t>(s)]) mn[static_cast<std::size_t>(s)] = b;

            const int32_t p = link[static_cast<std::size_t>(s)];
            if (p >= 0 && b > 0) {
                int32_t propagated = b;
                if (propagated > len[static_cast<std::size_t>(p)]) propagated = len[static_cast<std::size_t>(p)];
                set_best(p, propagated);
            }
        }

        if (++iter == std::numeric_limits<int32_t>::max()) {
            std::fill(vis.begin(), vis.end(), 0);
            iter = 1;
        }
    }

    std::string extract_best_substring(const std::string& base) const {
        int32_t best_len = 0;
        int32_t best_state = 0;
        for (int32_t v = 1; v < static_cast<int32_t>(mn.size()); ++v) {
            if (mn[static_cast<std::size_t>(v)] > best_len) {
                best_len = mn[static_cast<std::size_t>(v)];
                best_state = v;
            }
        }
        if (best_len == 0) return "";

        int32_t v = best_state;
        while (link[static_cast<std::size_t>(v)] != -1 &&
               len[static_cast<std::size_t>(link[static_cast<std::size_t>(v)])] >= best_len) {
            v = link[static_cast<std::size_t>(v)];
        }

        const int32_t end_pos = firstpos[static_cast<std::size_t>(v)];
        const int64_t start64 = static_cast<int64_t>(end_pos) - static_cast<int64_t>(best_len) + 1;
        const std::size_t start = start64 < 0 ? 0 : static_cast<std::size_t>(start64);
        if (start >= base.size()) return "";
        const std::size_t max_take = base.size() - start;
        const std::size_t take = std::min<std::size_t>(static_cast<std::size_t>(best_len), max_take);
        return base.substr(start, take);
    }

private:
    std::array<int32_t, 256> root_dense{};
    std::vector<int32_t> len;
    std::vector<int32_t> link;
    std::vector<int32_t> head;
    std::vector<int32_t> firstpos;
    std::vector<Edge> edges;

    int32_t last = 0;

    std::vector<int32_t> order;  // states in ascending len
    std::vector<int32_t> mn;     // min over strings of best match length per state

    std::vector<int32_t> best;   // scratch: best match for current string
    std::vector<int32_t> vis;    // timestamp
    int32_t iter = 1;

    int32_t new_state(int32_t l) {
        len.push_back(l);
        link.push_back(-1);
        head.push_back(-1);
        firstpos.push_back(-1);
        return static_cast<int32_t>(len.size() - 1);
    }

    int32_t transition(int32_t v, unsigned char c) const {
        if (v == 0) return root_dense[c];
        for (int32_t e = head[static_cast<std::size_t>(v)]; e != -1; e = edges[static_cast<std::size_t>(e)].next) {
            const Edge& ed = edges[static_cast<std::size_t>(e)];
            if (ed.ch == c) return ed.to;
        }
        return -1;
    }

    void add_edge(int32_t v, unsigned char c, int32_t to) {
        edges.push_back(Edge{ to, head[static_cast<std::size_t>(v)], c });
        head[static_cast<std::size_t>(v)] = static_cast<int32_t>(edges.size() - 1);
        if (v == 0) root_dense[c] = to;
    }

    void set_transition(int32_t v, unsigned char c, int32_t to) {
        if (v == 0) root_dense[c] = to;
        for (int32_t e = head[static_cast<std::size_t>(v)]; e != -1; e = edges[static_cast<std::size_t>(e)].next) {
            Edge& ed = edges[static_cast<std::size_t>(e)];
            if (ed.ch == c) {
                ed.to = to;
                return;
            }
        }
        add_edge(v, c, to);
    }

    void copy_transitions(int32_t from, int32_t to) {
        for (int32_t e = head[static_cast<std::size_t>(from)]; e != -1; e = edges[static_cast<std::size_t>(e)].next) {
            const Edge& ed = edges[static_cast<std::size_t>(e)];
            add_edge(to, ed.ch, ed.to);
        }
    }

    void extend(unsigned char c, int32_t pos) {
        const int32_t cur = new_state(len[static_cast<std::size_t>(last)] + 1);
        firstpos[static_cast<std::size_t>(cur)] = pos;

        int32_t p = last;
        for (; p != -1 && transition(p, c) == -1; p = link[static_cast<std::size_t>(p)]) {
            set_transition(p, c, cur);
        }

        if (p == -1) {
            link[static_cast<std::size_t>(cur)] = 0;
            last = cur;
            return;
        }

        const int32_t q = transition(p, c);
        if (len[static_cast<std::size_t>(p)] + 1 == len[static_cast<std::size_t>(q)]) {
            link[static_cast<std::size_t>(cur)] = q;
            last = cur;
            return;
        }

        const int32_t clone = new_state(len[static_cast<std::size_t>(p)] + 1);
        copy_transitions(q, clone);
        link[static_cast<std::size_t>(clone)] = link[static_cast<std::size_t>(q)];
        firstpos[static_cast<std::size_t>(clone)] = firstpos[static_cast<std::size_t>(q)];

        while (p != -1 && transition(p, c) == q) {
            set_transition(p, c, clone);
            p = link[static_cast<std::size_t>(p)];
        }

        link[static_cast<std::size_t>(q)] = clone;
        link[static_cast<std::size_t>(cur)] = clone;
        last = cur;
    }

    void build_order(int32_t max_len) {
        const int32_t sz = static_cast<int32_t>(len.size());
        std::vector<int32_t> cnt(static_cast<std::size_t>(max_len) + 1, 0);
        for (int32_t i = 0; i < sz; ++i) cnt[static_cast<std::size_t>(len[static_cast<std::size_t>(i)])]++;
        for (int32_t i = 1; i <= max_len; ++i) cnt[static_cast<std::size_t>(i)] += cnt[static_cast<std::size_t>(i - 1)];

        order.assign(static_cast<std::size_t>(sz), 0);
        for (int32_t i = sz - 1; i >= 0; --i) {
            const int32_t l = len[static_cast<std::size_t>(i)];
            order[static_cast<std::size_t>(--cnt[static_cast<std::size_t>(l)])] = i;
        }
    }

    void prepare_arrays() {
        const std::size_t sz = len.size();
        mn.resize(sz);
        for (std::size_t i = 0; i < sz; ++i) mn[i] = len[i];
        best.assign(sz, 0);
        vis.assign(sz, 0);
        iter = 1;
    }

    void set_best(int32_t v, int32_t value) {
        if (value <= 0) return;
        const std::size_t idx = static_cast<std::size_t>(v);
        if (vis[idx] != iter) {
            vis[idx] = iter;
            best[idx] = value;
        } else if (value > best[idx]) {
            best[idx] = value;
        }
    }

    int32_t get_best(int32_t v) const {
        const std::size_t idx = static_cast<std::size_t>(v);
        return vis[idx] == iter ? best[idx] : 0;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) return 0;
    std::string line;
    std::getline(std::cin, line);  // consume remainder of line

    if (n <= 0) {
        std::cout << '\n';
        return 0;
    }

    if (n == 1) {
        std::getline(std::cin, line);
        trim_cr(line);
        std::cout << line << '\n';
        return 0;
    }

    if (n == 2) {
        std::string a, b;
        std::getline(std::cin, a);
        std::getline(std::cin, b);
        trim_cr(a);
        trim_cr(b);

        if (a.empty() || b.empty()) {
            std::cout << '\n';
            return 0;
        }

        constexpr std::uint64_t DP_LIMIT_CELLS = 80ULL * 1000ULL * 1000ULL;
        const std::uint64_t product = static_cast<std::uint64_t>(a.size()) * static_cast<std::uint64_t>(b.size());
        if (product <= DP_LIMIT_CELLS) {
            std::cout << lcs_dp(a, b) << '\n';
            return 0;
        }

        const std::string& base = (a.size() <= b.size()) ? a : b;
        const std::string& other = (a.size() <= b.size()) ? b : a;
        SuffixAutomaton sam(base.size());
        sam.build(base);
        sam.apply(other);
        std::cout << sam.extract_best_substring(base) << '\n';
        return 0;
    }

    const char* tmp_name = "__strings.bin";
    std::FILE* f = std::fopen(tmp_name, "wb+");

    if (!f) {
        std::vector<std::string> strings;
        strings.reserve(static_cast<std::size_t>(n));
        int base_idx = 0;
        for (int i = 0; i < n; ++i) {
            std::getline(std::cin, line);
            trim_cr(line);
            strings.push_back(line);
            if (strings.back().empty()) {
                std::cout << '\n';
                return 0;
            }
            if (strings.back().size() < strings[static_cast<std::size_t>(base_idx)].size()) base_idx = i;
        }

        const std::string& base = strings[static_cast<std::size_t>(base_idx)];
        SuffixAutomaton sam(base.size());
        sam.build(base);
        for (int i = 0; i < n; ++i) {
            if (i == base_idx) continue;
            sam.apply(strings[static_cast<std::size_t>(i)]);
        }
        std::cout << sam.extract_best_substring(base) << '\n';
        return 0;
    }

    int base_idx = -1;
    std::size_t base_len = (std::numeric_limits<std::size_t>::max)();
    std::string base;

    for (int i = 0; i < n; ++i) {
        std::getline(std::cin, line);
        if (!std::cin) line.clear();
        trim_cr(line);

        const std::uint64_t len64 = static_cast<std::uint64_t>(line.size());
        if (std::fwrite(&len64, sizeof(len64), 1, f) != 1) {
            std::fclose(f);
            std::remove(tmp_name);
            std::cout << '\n';
            return 0;
        }
        if (len64 && std::fwrite(line.data(), 1, static_cast<std::size_t>(len64), f) != static_cast<std::size_t>(len64)) {
            std::fclose(f);
            std::remove(tmp_name);
            std::cout << '\n';
            return 0;
        }

        if (line.empty()) {
            std::fclose(f);
            std::remove(tmp_name);
            std::cout << '\n';
            return 0;
        }

        if (line.size() < base_len) {
            base_len = line.size();
            base_idx = i;
            base = line;
        }
    }

    if (base_idx < 0) {
        std::fclose(f);
        std::remove(tmp_name);
        std::cout << '\n';
        return 0;
    }

    SuffixAutomaton sam(base.size());
    sam.build(base);

    std::rewind(f);
    std::string t;
    for (int i = 0; i < n; ++i) {
        std::uint64_t len64 = 0;
        if (std::fread(&len64, sizeof(len64), 1, f) != 1) break;
        t.assign(static_cast<std::size_t>(len64), '\0');
        if (len64 && std::fread(t.data(), 1, static_cast<std::size_t>(len64), f) != static_cast<std::size_t>(len64)) {
            t.clear();
        }
        if (i == base_idx) continue;
        sam.apply(t);
    }

    std::fclose(f);
    std::remove(tmp_name);

    std::cout << sam.extract_best_substring(base) << '\n';
    return 0;
}
