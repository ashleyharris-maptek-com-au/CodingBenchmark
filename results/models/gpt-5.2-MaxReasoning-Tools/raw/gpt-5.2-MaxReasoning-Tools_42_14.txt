; SPIR-V
; Version: 1.0
; Geometry shader: flat-shade main triangle of triangles-adjacency input

OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_0_33333334 = OpConstant %float 0.33333334
%v3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %fn
%entry = OpLabel

; p0/p2/p4 from worldPos[]
%p0_ptr = OpAccessChain %ptr_in_v3float %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v3float %worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v3float %worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

; cross(a,b) manually
%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2
%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2

%cy_mul1 = OpFMul %float %a_y %b_z
%cy_mul2 = OpFMul %float %a_z %b_y
%cross_x = OpFSub %float %cy_mul1 %cy_mul2

%cz_mul1 = OpFMul %float %a_z %b_x
%cz_mul2 = OpFMul %float %a_x %b_z
%cross_y = OpFSub %float %cz_mul1 %cz_mul2

%cx_mul1 = OpFMul %float %a_x %b_y
%cx_mul2 = OpFMul %float %a_y %b_x
%cross_z = OpFSub %float %cx_mul1 %cx_mul2

%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%N = OpExtInst %v3float %glsl Normalize %cross
%L = OpExtInst %v3float %glsl Normalize %v3_one
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0

; base = avg(color[0], color[2], color[4])
%c0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3float %color %int_4
%c4 = OpLoad %v3float %c4_ptr

%sum02 = OpFAdd %v3float %c0 %c2
%sum024 = OpFAdd %v3float %sum02 %c4
%base = OpVectorTimesScalar %v3float %sum024 %float_0_33333334

%scaledDiffuse = OpFMul %float %diffuse %float_0_8
%lightFactor = OpFAdd %float %float_0_2 %scaledDiffuse
%finalColor = OpVectorTimesScalar %v3float %base %lightFactor

; Emit vertex 0
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%outPos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %outPos_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 2
%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
OpStore %outPos_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 4
%pos4_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
OpStore %outPos_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
