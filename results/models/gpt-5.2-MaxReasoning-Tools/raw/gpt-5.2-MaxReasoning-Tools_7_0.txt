// CSG Union solver (triangle meshes) using BSP-based polygon clipping.
// Implements the classic algorithm from Evan Wallace's csg.js (MIT license),
// adapted to C++20 and this contest I/O format.
//
// Notes:
// - Assumes input meshes are closed (watertight), manifold, and free of self-intersections.
// - Uses floating-point epsilon for robustness; vertices are snapped to a grid on output
//   to weld coincident vertices and keep the result watertight.
//
// I/O:
//  stdin: VA FA VB FB, then vertices/faces for A then B
//  stdout: V F then vertices and triangle indices.

#include <algorithm>
#include <array>
#include <cstdint>
#include <cmath>
#include <iostream>
#include <limits>
#include <memory>
#include <unordered_map>
#include <utility>
#include <vector>

namespace csg {

struct Vec3 {
  double x = 0, y = 0, z = 0;

  Vec3() = default;
  Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

  Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
  Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
  Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
  Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }

  Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
  Vec3& operator-=(const Vec3& o) { x -= o.x; y -= o.y; z -= o.z; return *this; }

  double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
  Vec3 cross(const Vec3& o) const {
    return {y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x};
  }
  double norm2() const { return dot(*this); }
  double norm() const { return std::sqrt(norm2()); }
};

inline Vec3 lerp(const Vec3& a, const Vec3& b, double t) {
  return a + (b - a) * t;
}

inline Vec3 safeUnit(const Vec3& v) {
  double n = v.norm();
  if (n <= 0) return {0, 0, 0};
  return v / n;
}

struct Vertex {
  Vec3 pos;
  Vertex() = default;
  explicit Vertex(const Vec3& p) : pos(p) {}

  Vertex interpolate(const Vertex& other, double t) const {
    return Vertex{lerp(pos, other.pos, t)};
  }
};

struct Polygon;  // fwd

struct Plane {
  Vec3 normal;
  double w = 0;
  double eps = 1e-5;

  Plane() = default;
  Plane(const Vec3& n, double w_, double eps_) : normal(n), w(w_), eps(eps_) {}

  static Plane fromPoints(const Vec3& a, const Vec3& b, const Vec3& c, double eps) {
    Vec3 n = safeUnit((b - a).cross(c - a));
    return Plane{n, n.dot(a), eps};
  }

  Plane flipped() const { return Plane{normal * -1.0, -w, eps}; }

  enum PolyClass : int { COPLANAR = 0, FRONT = 1, BACK = 2, SPANNING = 3 };

  int classifyPolygon(const Polygon& polygon) const;

  void splitPolygon(const Polygon& polygon,
                    std::vector<Polygon>& coplanarFront,
                    std::vector<Polygon>& coplanarBack,
                    std::vector<Polygon>& front,
                    std::vector<Polygon>& back) const;
};

struct Polygon {
  std::vector<Vertex> vertices;
  Plane plane;
  Vec3 bsCenter;
  double bsRadius = 0.0;

  Polygon() = default;
  Polygon(std::vector<Vertex> verts, double eps) : vertices(std::move(verts)) {
    if (vertices.size() >= 3) {
      plane = Plane::fromPoints(vertices[0].pos, vertices[1].pos, vertices[2].pos, eps);
    } else {
      plane = Plane{{0, 0, 0}, 0.0, eps};
    }
    recomputeSphere();
  }

  static Polygon fromTriangle(const Vec3& a, const Vec3& b, const Vec3& c, double eps) {
    std::vector<Vertex> verts;
    verts.reserve(3);
    verts.emplace_back(a);
    verts.emplace_back(b);
    verts.emplace_back(c);
    return Polygon{std::move(verts), eps};
  }

  void flip() {
    std::reverse(vertices.begin(), vertices.end());
    plane = plane.flipped();
  }

  void recomputeSphere() {
    if (vertices.empty()) {
      bsCenter = {0, 0, 0};
      bsRadius = 0;
      return;
    }
    bsCenter = {0, 0, 0};
    for (const auto& v : vertices) bsCenter += v.pos;
    bsCenter = bsCenter / static_cast<double>(vertices.size());
    bsRadius = 0.0;
    for (const auto& v : vertices) {
      double d = (v.pos - bsCenter).norm();
      if (d > bsRadius) bsRadius = d;
    }
  }
};

inline int Plane::classifyPolygon(const Polygon& polygon) const {
  // Bounding sphere fast path: if the whole sphere is strictly on one side, we can classify quickly.
  double dist = normal.dot(polygon.bsCenter) - w;
  if (dist > polygon.bsRadius + eps) return FRONT;
  if (dist < -polygon.bsRadius - eps) return BACK;

  int polygonType = 0;
  for (const auto& v : polygon.vertices) {
    double t = normal.dot(v.pos) - w;
    int type = (t < -eps) ? BACK : (t > eps) ? FRONT : COPLANAR;
    polygonType |= type;
    if (polygonType == SPANNING) break;
  }
  return polygonType;
}

inline void Plane::splitPolygon(const Polygon& polygon,
                                std::vector<Polygon>& coplanarFront,
                                std::vector<Polygon>& coplanarBack,
                                std::vector<Polygon>& front,
                                std::vector<Polygon>& back) const {
  // Fast path via bounding sphere.
  double dist = normal.dot(polygon.bsCenter) - w;
  if (dist > polygon.bsRadius + eps) {
    front.push_back(polygon);
    return;
  }
  if (dist < -polygon.bsRadius - eps) {
    back.push_back(polygon);
    return;
  }

  int polygonType = 0;
  std::vector<int> types;
  types.reserve(polygon.vertices.size());
  for (const auto& v : polygon.vertices) {
    double t = normal.dot(v.pos) - w;
    int type = (t < -eps) ? BACK : (t > eps) ? FRONT : COPLANAR;
    polygonType |= type;
    types.push_back(type);
  }

  switch (polygonType) {
    case COPLANAR: {
      // If polygon plane has same orientation as this plane, keep it in front list, else back.
      if (normal.dot(polygon.plane.normal) > 0) coplanarFront.push_back(polygon);
      else coplanarBack.push_back(polygon);
      break;
    }
    case FRONT:
      front.push_back(polygon);
      break;
    case BACK:
      back.push_back(polygon);
      break;
    case SPANNING: {
      std::vector<Vertex> f;
      std::vector<Vertex> b;
      f.reserve(polygon.vertices.size() + 2);
      b.reserve(polygon.vertices.size() + 2);

      const int n = static_cast<int>(polygon.vertices.size());
      for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        int ti = types[i], tj = types[j];
        const Vertex& vi = polygon.vertices[i];
        const Vertex& vj = polygon.vertices[j];

        if (ti != BACK) f.push_back(vi);
        if (ti != FRONT) b.push_back(vi);

        if ((ti | tj) == SPANNING) {
          Vec3 dir = vj.pos - vi.pos;
          double denom = normal.dot(dir);
          // denom should not be 0 here; guard anyway.
          double t = 0.0;
          if (std::abs(denom) > 0) {
            t = (w - normal.dot(vi.pos)) / denom;
          }
          // Clamp to reduce numeric drift.
          if (t < 0) t = 0;
          if (t > 1) t = 1;
          Vertex v = vi.interpolate(vj, t);
          f.push_back(v);
          b.push_back(v);
        }
      }

      if (f.size() >= 3) front.emplace_back(std::move(f), eps);
      if (b.size() >= 3) back.emplace_back(std::move(b), eps);
      break;
    }
  }
}

class Node {
 public:
  Node() = default;
  explicit Node(std::vector<Polygon> polygons) { build(std::move(polygons)); }

  void invert() {
    // Iterative traversal to avoid deep recursion.
    std::vector<Node*> stack;
    stack.push_back(this);
    while (!stack.empty()) {
      Node* node = stack.back();
      stack.pop_back();
      for (auto& p : node->polygons_) p.flip();
      if (node->plane_) node->plane_ = std::make_unique<Plane>(node->plane_->flipped());
      std::swap(node->front_, node->back_);
      if (node->front_) stack.push_back(node->front_.get());
      if (node->back_) stack.push_back(node->back_.get());
    }
  }

  std::vector<Polygon> clipPolygons(const std::vector<Polygon>& polygons) const {
    if (!plane_) return polygons;
    std::vector<Polygon> front;
    std::vector<Polygon> back;
    front.reserve(polygons.size());
    back.reserve(polygons.size());
    for (const auto& p : polygons) {
      plane_->splitPolygon(p, front, back, front, back);
    }
    if (front_) front = front_->clipPolygons(front);
    if (back_) back = back_->clipPolygons(back);
    else back.clear();
    front.insert(front.end(), back.begin(), back.end());
    return front;
  }

  void clipTo(const Node& bsp) {
    // Iterative traversal.
    std::vector<Node*> stack;
    stack.push_back(this);
    while (!stack.empty()) {
      Node* node = stack.back();
      stack.pop_back();
      node->polygons_ = bsp.clipPolygons(node->polygons_);
      if (node->front_) stack.push_back(node->front_.get());
      if (node->back_) stack.push_back(node->back_.get());
    }
  }

  std::vector<Polygon> allPolygons() const {
    std::vector<Polygon> out;
    std::vector<const Node*> stack;
    stack.push_back(this);
    while (!stack.empty()) {
      const Node* node = stack.back();
      stack.pop_back();
      out.insert(out.end(), node->polygons_.begin(), node->polygons_.end());
      if (node->front_) stack.push_back(node->front_.get());
      if (node->back_) stack.push_back(node->back_.get());
    }
    return out;
  }

  void build(std::vector<Polygon> polygons) {
    if (polygons.empty()) return;
    if (!plane_) {
      // Heuristic plane selection: try a few candidates and pick the one that minimizes splits
      // while keeping the tree reasonably balanced.
      const int candidateCount = std::min<int>(static_cast<int>(polygons.size()), 12);
      const int testCount = std::min<int>(static_cast<int>(polygons.size()), 96);
      double bestScore = std::numeric_limits<double>::infinity();
      Plane bestPlane = polygons[0].plane;

      for (int c = 0; c < candidateCount; c++) {
        const Plane& candidate = polygons[c].plane;
        int numFront = 0, numBack = 0, numSplit = 0, numCoplanar = 0;
        for (int i = 0; i < testCount; i++) {
          int cls = candidate.classifyPolygon(polygons[i]);
          if (cls == Plane::FRONT) numFront++;
          else if (cls == Plane::BACK) numBack++;
          else if (cls == Plane::SPANNING) numSplit++;
          else numCoplanar++;
        }
        double balance = std::abs(numFront - numBack);
        double score = static_cast<double>(numSplit) * 8.0 + balance + static_cast<double>(numCoplanar) * 0.25;
        if (score < bestScore) {
          bestScore = score;
          bestPlane = candidate;
        }
      }
      plane_ = std::make_unique<Plane>(bestPlane);
    }

    std::vector<Polygon> front;
    std::vector<Polygon> back;
    front.reserve(polygons.size());
    back.reserve(polygons.size());

    for (auto& p : polygons) {
      plane_->splitPolygon(p, polygons_, polygons_, front, back);
    }

    if (!front.empty()) {
      if (!front_) front_ = std::make_unique<Node>();
      front_->build(std::move(front));
    }
    if (!back.empty()) {
      if (!back_) back_ = std::make_unique<Node>();
      back_->build(std::move(back));
    }
  }

 private:
  std::unique_ptr<Plane> plane_;
  std::unique_ptr<Node> front_;
  std::unique_ptr<Node> back_;
  std::vector<Polygon> polygons_;
};

inline std::vector<Polygon> unionPolygons(std::vector<Polygon> a, std::vector<Polygon> b) {
  Node A(std::move(a));
  Node B(std::move(b));
  A.clipTo(B);
  B.clipTo(A);
  B.invert();
  B.clipTo(A);
  B.invert();
  A.build(B.allPolygons());
  return A.allPolygons();
}

}  // namespace csg

struct Mesh {
  std::vector<csg::Vec3> vertices;
  std::vector<std::array<int, 3>> faces;
};

static csg::Vec3 minVec(const csg::Vec3& a, const csg::Vec3& b) {
  return {std::min(a.x, b.x), std::min(a.y, b.y), std::min(a.z, b.z)};
}
static csg::Vec3 maxVec(const csg::Vec3& a, const csg::Vec3& b) {
  return {std::max(a.x, b.x), std::max(a.y, b.y), std::max(a.z, b.z)};
}

static void bbox(const Mesh& m, csg::Vec3& outMin, csg::Vec3& outMax) {
  outMin = { std::numeric_limits<double>::infinity(),
             std::numeric_limits<double>::infinity(),
             std::numeric_limits<double>::infinity() };
  outMax = { -std::numeric_limits<double>::infinity(),
             -std::numeric_limits<double>::infinity(),
             -std::numeric_limits<double>::infinity() };
  for (const auto& v : m.vertices) {
    outMin = minVec(outMin, v);
    outMax = maxVec(outMax, v);
  }
}

static bool aabbDisjoint(const csg::Vec3& aMin, const csg::Vec3& aMax,
                         const csg::Vec3& bMin, const csg::Vec3& bMax) {
  return (aMax.x < bMin.x) || (bMax.x < aMin.x) ||
         (aMax.y < bMin.y) || (bMax.y < aMin.y) ||
         (aMax.z < bMin.z) || (bMax.z < aMin.z);
}

struct QuantKey {
  std::int64_t x, y, z;
  bool operator==(const QuantKey& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct QuantKeyHash {
  std::size_t operator()(const QuantKey& k) const noexcept {
    // 64-bit mix (splitmix64-ish)
    auto mix = [](std::uint64_t v) {
      v += 0x9e3779b97f4a7c15ULL;
      v = (v ^ (v >> 30)) * 0xbf58476d1ce4e5b9ULL;
      v = (v ^ (v >> 27)) * 0x94d049bb133111ebULL;
      return v ^ (v >> 31);
    };
    std::uint64_t h = mix(static_cast<std::uint64_t>(k.x));
    h ^= mix(static_cast<std::uint64_t>(k.y) + 0x100000001b3ULL);
    h ^= mix(static_cast<std::uint64_t>(k.z) + 0x9e3779b97f4a7c15ULL);
    return static_cast<std::size_t>(h);
  }
};

static std::int64_t qround(double v, double invStep) {
  // round to nearest int64
  return static_cast<std::int64_t>(std::llround(v * invStep));
}

static void writeMeshAsUnion(const Mesh& a, const Mesh& b) {
  csg::Vec3 aMin, aMax, bMin, bMax;
  bbox(a, aMin, aMax);
  bbox(b, bMin, bMax);
  csg::Vec3 globalMin = minVec(aMin, bMin);
  csg::Vec3 globalMax = maxVec(aMax, bMax);
  double diag = (globalMax - globalMin).norm();
  if (!(diag > 0)) diag = 1.0;

  // Epsilon used for plane/side classification.
  double eps = 1e-5;
  // Try to adapt slightly to scale, but clamp to avoid over-blurring.
  eps = std::clamp(1e-8 * (diag + 1.0), 1e-7, 1e-5);

  // If AABBs are disjoint, the union is just both meshes concatenated.
  if (aabbDisjoint(aMin, aMax, bMin, bMax)) {
    std::vector<csg::Vec3> outV = a.vertices;
    outV.insert(outV.end(), b.vertices.begin(), b.vertices.end());
    std::vector<std::array<int,3>> outF = a.faces;
    int offset = static_cast<int>(a.vertices.size());
    outF.reserve(outF.size() + b.faces.size());
    for (auto f : b.faces) outF.push_back({f[0] + offset, f[1] + offset, f[2] + offset});

    std::cout.setf(std::ios::fixed);
    std::cout.precision(9);
    std::cout << outV.size() << " " << outF.size() << "\n";
    for (const auto& v : outV) std::cout << v.x << " " << v.y << " " << v.z << "\n";
    for (const auto& f : outF) std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    return;
  }

  // Convert input meshes to CSG polygons (triangles).
  std::vector<csg::Polygon> polysA;
  polysA.reserve(a.faces.size());
  for (const auto& f : a.faces) {
    const auto& v0 = a.vertices[f[0]];
    const auto& v1 = a.vertices[f[1]];
    const auto& v2 = a.vertices[f[2]];
    polysA.push_back(csg::Polygon::fromTriangle(v0, v1, v2, eps));
  }
  std::vector<csg::Polygon> polysB;
  polysB.reserve(b.faces.size());
  for (const auto& f : b.faces) {
    const auto& v0 = b.vertices[f[0]];
    const auto& v1 = b.vertices[f[1]];
    const auto& v2 = b.vertices[f[2]];
    polysB.push_back(csg::Polygon::fromTriangle(v0, v1, v2, eps));
  }

  std::vector<csg::Polygon> unionPolys = csg::unionPolygons(std::move(polysA), std::move(polysB));

  // Weld vertices by snapping to a grid (canonicalization-lite).
  const double weldStep = std::max(1e-7, eps * 0.5);
  const double invWeld = 1.0 / weldStep;

  std::unordered_map<QuantKey, int, QuantKeyHash> vertexMap;
  vertexMap.reserve(unionPolys.size() * 3);

  std::vector<csg::Vec3> outVertices;
  std::vector<std::array<int, 3>> outFaces;

  auto getIndex = [&](const csg::Vec3& p) -> int {
    QuantKey key{qround(p.x, invWeld), qround(p.y, invWeld), qround(p.z, invWeld)};
    auto it = vertexMap.find(key);
    if (it != vertexMap.end()) return it->second;
    int idx = static_cast<int>(outVertices.size());
    vertexMap.emplace(key, idx);
    // Store snapped position for consistency.
    outVertices.push_back({static_cast<double>(key.x) * weldStep,
                           static_cast<double>(key.y) * weldStep,
                           static_cast<double>(key.z) * weldStep});
    return idx;
  };

  // Triangulate each convex polygon by fan.
  for (const auto& poly : unionPolys) {
    if (poly.vertices.size() < 3) continue;
    int i0 = getIndex(poly.vertices[0].pos);
    for (size_t i = 1; i + 1 < poly.vertices.size(); i++) {
      int i1 = getIndex(poly.vertices[i].pos);
      int i2 = getIndex(poly.vertices[i + 1].pos);
      if (i0 == i1 || i1 == i2 || i2 == i0) continue;
      outFaces.push_back({i0, i1, i2});
    }
  }

  std::cout.setf(std::ios::fixed);
  std::cout.precision(9);
  std::cout << outVertices.size() << " " << outFaces.size() << "\n";
  for (const auto& v : outVertices) std::cout << v.x << " " << v.y << " " << v.z << "\n";
  for (const auto& f : outFaces) std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int VA = 0, FA = 0, VB = 0, FB = 0;
  if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

  Mesh a, b;
  a.vertices.resize(VA);
  a.faces.resize(FA);
  b.vertices.resize(VB);
  b.faces.resize(FB);

  for (int i = 0; i < VA; i++) {
    double x, y, z;
    std::cin >> x >> y >> z;
    a.vertices[i] = {x, y, z};
  }
  for (int i = 0; i < FA; i++) {
    int x, y, z;
    std::cin >> x >> y >> z;
    a.faces[i] = {x, y, z};
  }
  for (int i = 0; i < VB; i++) {
    double x, y, z;
    std::cin >> x >> y >> z;
    b.vertices[i] = {x, y, z};
  }
  for (int i = 0; i < FB; i++) {
    int x, y, z;
    std::cin >> x >> y >> z;
    b.faces[i] = {x, y, z};
  }

  writeMeshAsUnion(a, b);
  return 0;
}
