; SPIR-V Geometry Shader (Vulkan 1.0)

OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %outColor "outColor"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%uint_3 = OpConstant %uint 3

%float_0 = OpConstant %float 0.0
%float_3 = OpConstant %float 3.0
%float_0_08 = OpConstant %float 0.08

%v3_three = OpConstantComposite %v3float %float_3 %float_3 %float_3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; p0/p1/p2 from gl_in[].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0_4 = OpLoad %v4float %p0_ptr
%p1_4 = OpLoad %v4float %p1_ptr
%p2_4 = OpLoad %v4float %p2_ptr

%p0 = OpVectorShuffle %v3float %p0_4 %p0_4 0 1 2
%p1 = OpVectorShuffle %v3float %p1_4 %p1_4 0 1 2
%p2 = OpVectorShuffle %v3float %p2_4 %p2_4 0 1 2

; center = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v3float %p0 %p1
%sum012 = OpFAdd %v3float %sum01 %p2
%center = OpFDiv %v3float %sum012 %v3_three

; face normal N = normalize(cross(p1-p0, p2-p0))
%e1 = OpFSub %v3float %p1 %p0
%e2 = OpFSub %v3float %p2 %p0
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%N = OpExtInst %v3float %glsl Normalize %cross

; offset = N * 0.08
%offset = OpVectorTimesScalar %v3float %N %float_0_08
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

%outPos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 0
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%p0_out = OpFAdd %v4float %p0_4 %offset4
OpStore %outPos_ptr %p0_out
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%p1_out = OpFAdd %v4float %p1_4 %offset4
OpStore %outPos_ptr %p1_out
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 2
%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr
%p2_out = OpFAdd %v4float %p2_4 %offset4
OpStore %outPos_ptr %p2_out
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
