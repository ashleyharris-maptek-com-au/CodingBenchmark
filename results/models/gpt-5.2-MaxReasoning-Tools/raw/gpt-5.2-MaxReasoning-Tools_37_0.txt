using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var graph = new Graph(n, m);
        var hasSelfLoop = new bool[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u < (uint)n && (uint)v < (uint)n && u == v) hasSelfLoop[u] = true;
            graph.AddEdge(u, v, i);
        }

        // SCC decomposition (Kosaraju)
        int[] comp = KosarajuScc(n, graph.Head, graph.To, graph.Next, graph.HeadR, graph.ToR, graph.NextR, out int compCount);

        // Group vertices by component
        int[] compSize = new int[compCount];
        for (int v = 0; v < n; v++) compSize[comp[v]]++;
        int[] compStart = new int[compCount + 1];
        for (int c = 0; c < compCount; c++) compStart[c + 1] = compStart[c] + compSize[c];
        int[] writePos = new int[compCount];
        Array.Copy(compStart, writePos, compCount);
        int[] vertsByComp = new int[n];
        for (int v = 0; v < n; v++) vertsByComp[writePos[comp[v]]++] = v;

        // Temporary mapping from global vertex id -> local index within current component.
        // Initialize to -1 once; we only touch indices of vertices in the component.
        int[] localOf = new int[n];
        Array.Fill(localOf, -1);

        var answer = new List<int>();

        for (int c = 0; c < compCount; c++)
        {
            int start = compStart[c];
            int size = compStart[c + 1] - start;
            if (size == 1)
            {
                int v = vertsByComp[start];
                if (hasSelfLoop[v]) answer.Add(v);
                continue;
            }

            // Extract component vertices
            int[] compVerts = new int[size];
            Array.Copy(vertsByComp, start, compVerts, 0, size);

            // Build local index mapping
            for (int i = 0; i < size; i++) localOf[compVerts[i]] = i;

            // Solve DFVS on this SCC exactly
            if (size <= 64)
            {
                var outBits = new ulong[size];
                var inBits = new ulong[size];

                for (int i = 0; i < size; i++)
                {
                    int gv = compVerts[i];
                    for (int e = graph.Head[gv]; e != -1; e = graph.Next[e])
                    {
                        int gw = graph.To[e];
                        if (comp[gw] != c) continue;
                        int j = localOf[gw];
                        ulong bit = 1UL << j;
                        outBits[i] |= bit;
                        inBits[j] |= 1UL << i;
                    }
                }

                var solver = new BitmaskDfvsSolver(outBits, inBits);
                ulong removedMask = solver.Solve();
                while (removedMask != 0)
                {
                    int idx = BitOperations.TrailingZeroCount(removedMask);
                    removedMask &= removedMask - 1;
                    answer.Add(compVerts[idx]);
                }
            }
            else
            {
                // Fallback for larger SCCs: branching with trimming on adjacency lists.
                // (Still exact; slower in the worst case.)
                int internalEdges = 0;
                for (int i = 0; i < size; i++)
                {
                    int gv = compVerts[i];
                    for (int e = graph.Head[gv]; e != -1; e = graph.Next[e])
                    {
                        int gw = graph.To[e];
                        if (comp[gw] == c) internalEdges++;
                    }
                }

                int[] headOut = new int[size];
                int[] headIn = new int[size];
                Array.Fill(headOut, -1);
                Array.Fill(headIn, -1);
                int[] toOut = new int[internalEdges];
                int[] nextOut = new int[internalEdges];
                int[] toIn = new int[internalEdges];
                int[] nextIn = new int[internalEdges];

                int ei = 0;
                for (int i = 0; i < size; i++)
                {
                    int gv = compVerts[i];
                    for (int e = graph.Head[gv]; e != -1; e = graph.Next[e])
                    {
                        int gw = graph.To[e];
                        if (comp[gw] != c) continue;
                        int j = localOf[gw];

                        toOut[ei] = j;
                        nextOut[ei] = headOut[i];
                        headOut[i] = ei;

                        toIn[ei] = i;
                        nextIn[ei] = headIn[j];
                        headIn[j] = ei;

                        ei++;
                    }
                }

                var solver = new ListDfvsSolver(size, headOut, toOut, nextOut, headIn, toIn, nextIn);
                List<int> removedLocal = solver.Solve();
                for (int i = 0; i < removedLocal.Count; i++) answer.Add(compVerts[removedLocal[i]]);
            }

            // Clear local mapping entries (restore to -1) for next component
            for (int i = 0; i < size; i++) localOf[compVerts[i]] = -1;
        }

        Console.WriteLine(answer.Count);
        if (answer.Count == 0)
        {
            Console.WriteLine();
        }
        else
        {
            Console.WriteLine(string.Join(' ', answer));
        }
    }

    private static int[] KosarajuScc(
        int n,
        int[] head, int[] to, int[] next,
        int[] headR, int[] toR, int[] nextR,
        out int compCount)
    {
        var visited = new bool[n];
        var order = new int[n];
        int orderLen = 0;

        int[] stackNode = new int[n];
        int[] stackEdge = new int[n];

        for (int s = 0; s < n; s++)
        {
            if (visited[s]) continue;

            int sp = 0;
            stackNode[0] = s;
            stackEdge[0] = head[s];
            visited[s] = true;

            while (sp >= 0)
            {
                int v = stackNode[sp];
                int e = stackEdge[sp];
                if (e != -1)
                {
                    stackEdge[sp] = next[e];
                    int w = to[e];
                    if (!visited[w])
                    {
                        visited[w] = true;
                        sp++;
                        stackNode[sp] = w;
                        stackEdge[sp] = head[w];
                    }
                }
                else
                {
                    order[orderLen++] = v;
                    sp--;
                }
            }
        }

        int[] comp = new int[n];
        Array.Fill(comp, -1);
        compCount = 0;

        int[] stack = new int[n];
        for (int i = n - 1; i >= 0; i--)
        {
            int v = order[i];
            if (comp[v] != -1) continue;

            int top = 0;
            stack[0] = v;
            comp[v] = compCount;
            while (top >= 0)
            {
                int x = stack[top--];
                for (int e = headR[x]; e != -1; e = nextR[e])
                {
                    int w = toR[e];
                    if (comp[w] == -1)
                    {
                        comp[w] = compCount;
                        stack[++top] = w;
                    }
                }
            }
            compCount++;
        }

        return comp;
    }

    private sealed class Graph
    {
        public int[] Head { get; }
        public int[] To { get; }
        public int[] Next { get; }
        public int[] HeadR { get; }
        public int[] ToR { get; }
        public int[] NextR { get; }

        public Graph(int n, int m)
        {
            Head = new int[n];
            HeadR = new int[n];
            Array.Fill(Head, -1);
            Array.Fill(HeadR, -1);

            To = new int[m];
            Next = new int[m];
            ToR = new int[m];
            NextR = new int[m];
        }

        public void AddEdge(int u, int v, int idx)
        {
            To[idx] = v;
            Next[idx] = Head[u];
            Head[u] = idx;

            ToR[idx] = u;
            NextR[idx] = HeadR[v];
            HeadR[v] = idx;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }

            return val * sign;
        }
    }

    private sealed class BitmaskDfvsSolver
    {
        private readonly int _n;
        private readonly ulong _fullMask;
        private readonly ulong[] _out;
        private readonly ulong[] _in;

        private int _bestCount;
        private ulong _bestMask;

        private readonly int[] _cycleBuf;
        private readonly int[] _pathBuf;
        private readonly int[] _seenAt;

        public BitmaskDfvsSolver(ulong[] outBits, ulong[] inBits)
        {
            _n = outBits.Length;
            _out = outBits;
            _in = inBits;
            _cycleBuf = new int[_n];
            _pathBuf = new int[_n];
            _seenAt = new int[_n];

            _fullMask = _n == 64 ? ulong.MaxValue : ((1UL << _n) - 1UL);
        }

        public ulong Solve()
        {
            ulong mandatory = 0;
            for (int i = 0; i < _n; i++)
            {
                if (((_out[i] >> i) & 1UL) != 0) mandatory |= 1UL << i;
            }

            // Greedy upper bound
            ulong greedyMask = mandatory;
            while (true)
            {
                ulong core = Core(_fullMask & ~greedyMask);
                if (core == 0) break;
                int len = GetBranchCycle(core, _cycleBuf);
                int bestV = _cycleBuf[0];
                int bestScore = -1;
                for (int i = 0; i < len; i++)
                {
                    int v = _cycleBuf[i];
                    int score = BitOperations.PopCount((_out[v] | _in[v]) & core);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestV = v;
                    }
                }
                greedyMask |= 1UL << bestV;
            }

            _bestMask = greedyMask;
            _bestCount = BitOperations.PopCount(greedyMask);

            Search(mandatory, BitOperations.PopCount(mandatory));
            return _bestMask;
        }

        private void Search(ulong removedMask, int removedCount)
        {
            if (removedCount >= _bestCount) return;

            ulong present = _fullMask & ~removedMask;
            ulong core = Core(present);
            if (core == 0)
            {
                _bestCount = removedCount;
                _bestMask = removedMask;
                return;
            }

            int lb = GreedyPackingLowerBound(core, _bestCount - removedCount);
            if (removedCount + lb >= _bestCount) return;

            int cycleLen = GetBranchCycle(core, _cycleBuf);
            SortCycleByDegreeDesc(_cycleBuf, cycleLen, core);

            for (int i = 0; i < cycleLen; i++)
            {
                int v = _cycleBuf[i];
                Search(removedMask | (1UL << v), removedCount + 1);
                if (removedCount + 1 >= _bestCount) return; // further branches won't improve
            }
        }

        private int GreedyPackingLowerBound(ulong core, int cap)
        {
            int lb = 0;
            ulong temp = core;
            while (temp != 0)
            {
                if (lb >= cap) return lb;
                temp = Core(temp);
                if (temp == 0) break;
                int len = GetSomeCycle(temp, _cycleBuf);
                ulong cycMask = 0;
                for (int i = 0; i < len; i++) cycMask |= 1UL << _cycleBuf[i];
                temp &= ~cycMask;
                lb++;
            }
            return lb;
        }

        private ulong Core(ulong present)
        {
            ulong alive = present;
            while (true)
            {
                ulong toRemove = 0;
                ulong tmp = alive;
                while (tmp != 0)
                {
                    int v = BitOperations.TrailingZeroCount(tmp);
                    tmp &= tmp - 1;
                    if (((_out[v] & alive) == 0) || ((_in[v] & alive) == 0))
                    {
                        toRemove |= 1UL << v;
                    }
                }
                if (toRemove == 0) return alive;
                alive &= ~toRemove;
                if (alive == 0) return 0;
            }
        }

        private int GetBranchCycle(ulong core, int[] buf)
        {
            // Prefer self-loops, then 2-cycles, else any directed cycle.
            ulong tmp = core;
            while (tmp != 0)
            {
                int v = BitOperations.TrailingZeroCount(tmp);
                tmp &= tmp - 1;
                if (((_out[v] >> v) & 1UL) != 0)
                {
                    buf[0] = v;
                    return 1;
                }
            }

            tmp = core;
            while (tmp != 0)
            {
                int u = BitOperations.TrailingZeroCount(tmp);
                tmp &= tmp - 1;
                ulong neigh = _out[u] & core;
                while (neigh != 0)
                {
                    int v = BitOperations.TrailingZeroCount(neigh);
                    neigh &= neigh - 1;
                    if (((_out[v] >> u) & 1UL) != 0)
                    {
                        buf[0] = u;
                        buf[1] = v;
                        return 2;
                    }
                }
            }

            return GetSomeCycle(core, buf);
        }

        private int GetSomeCycle(ulong core, int[] buf)
        {
            Array.Fill(_seenAt, -1);
            int pathLen = 0;

            int start = BitOperations.TrailingZeroCount(core);
            int cur = start;
            while (true)
            {
                int seen = _seenAt[cur];
                if (seen >= 0)
                {
                    int cycLen = pathLen - seen;
                    for (int i = 0; i < cycLen; i++) buf[i] = _pathBuf[seen + i];
                    return cycLen;
                }

                _seenAt[cur] = pathLen;
                _pathBuf[pathLen++] = cur;

                ulong nextMask = _out[cur] & core;
                cur = BitOperations.TrailingZeroCount(nextMask);
            }
        }

        private void SortCycleByDegreeDesc(int[] cycle, int len, ulong core)
        {
            // Simple insertion sort; len is typically small.
            for (int i = 1; i < len; i++)
            {
                int v = cycle[i];
                int dv = BitOperations.PopCount((_out[v] | _in[v]) & core);
                int j = i - 1;
                while (j >= 0)
                {
                    int u = cycle[j];
                    int du = BitOperations.PopCount((_out[u] | _in[u]) & core);
                    if (du >= dv) break;
                    cycle[j + 1] = u;
                    j--;
                }
                cycle[j + 1] = v;
            }
        }
    }

    private sealed class ListDfvsSolver
    {
        private readonly int _n;
        private readonly int[] _headOut;
        private readonly int[] _toOut;
        private readonly int[] _nextOut;
        private readonly int[] _headIn;
        private readonly int[] _toIn;
        private readonly int[] _nextIn;

        private readonly bool[] _removed;
        private readonly bool[] _alive; // for core computation
        private readonly int[] _indeg;
        private readonly int[] _outdeg;
        private readonly int[] _queue;
        private readonly int[] _coreVerts;

        private readonly int[] _cycleBuf;
        private readonly int[] _pathBuf;
        private readonly int[] _seenAt;

        private readonly List<int> _current;
        private List<int> _best;
        private int _bestCount;

        public ListDfvsSolver(
            int n,
            int[] headOut, int[] toOut, int[] nextOut,
            int[] headIn, int[] toIn, int[] nextIn)
        {
            _n = n;
            _headOut = headOut;
            _toOut = toOut;
            _nextOut = nextOut;
            _headIn = headIn;
            _toIn = toIn;
            _nextIn = nextIn;

            _removed = new bool[n];
            _alive = new bool[n];
            _indeg = new int[n];
            _outdeg = new int[n];
            _queue = new int[n];
            _coreVerts = new int[n];

            _cycleBuf = new int[n];
            _pathBuf = new int[n];
            _seenAt = new int[n];

            _current = new List<int>(Math.Min(64, n));
            _best = new List<int>();
            _bestCount = int.MaxValue;
        }

        public List<int> Solve()
        {
            // Mandatory self-loops
            for (int u = 0; u < _n; u++)
            {
                for (int e = _headOut[u]; e != -1; e = _nextOut[e])
                {
                    if (_toOut[e] == u)
                    {
                        _removed[u] = true;
                        _current.Add(u);
                        break;
                    }
                }
            }

            // Greedy upper bound
            var greedyRemoved = new List<int>(_current);
            var greedyMark = (bool[])_removed.Clone();
            while (true)
            {
                int coreCount = BuildCore(greedyMark);
                if (coreCount == 0) break;
                int len = GetSomeCycleInCore(coreCount, _cycleBuf);
                int bestV = _cycleBuf[0];
                int bestScore = -1;
                for (int i = 0; i < len; i++)
                {
                    int v = _cycleBuf[i];
                    int score = _indeg[v] + _outdeg[v];
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestV = v;
                    }
                }
                greedyMark[bestV] = true;
                greedyRemoved.Add(bestV);
            }
            _best = greedyRemoved;
            _bestCount = greedyRemoved.Count;

            Search(_current.Count);
            return _best;
        }

        private void Search(int removedCount)
        {
            if (removedCount >= _bestCount) return;

            int coreCount = BuildCore(_removed);
            if (coreCount == 0)
            {
                _bestCount = removedCount;
                _best = new List<int>(_current);
                return;
            }

            int cycleLen = GetSomeCycleInCore(coreCount, _cycleBuf);
            SortCycleByDegreeDesc(_cycleBuf, cycleLen);

            for (int i = 0; i < cycleLen; i++)
            {
                int v = _cycleBuf[i];
                if (_removed[v]) continue;
                _removed[v] = true;
                _current.Add(v);
                Search(removedCount + 1);
                _current.RemoveAt(_current.Count - 1);
                _removed[v] = false;

                if (removedCount + 1 >= _bestCount) return;
            }
        }

        private int BuildCore(bool[] removed)
        {
            // Initialize
            for (int v = 0; v < _n; v++)
            {
                _alive[v] = !removed[v];
                _indeg[v] = 0;
                _outdeg[v] = 0;
            }

            // Compute degrees in induced subgraph of alive vertices
            for (int u = 0; u < _n; u++)
            {
                if (!_alive[u]) continue;
                for (int e = _headOut[u]; e != -1; e = _nextOut[e])
                {
                    int v = _toOut[e];
                    if (_alive[v])
                    {
                        _outdeg[u]++;
                        _indeg[v]++;
                    }
                }
            }

            // Queue initial sources/sinks
            int qh = 0, qt = 0;
            for (int v = 0; v < _n; v++)
            {
                if (_alive[v] && (_indeg[v] == 0 || _outdeg[v] == 0))
                    _queue[qt++] = v;
            }

            while (qh < qt)
            {
                int x = _queue[qh++];
                if (!_alive[x]) continue;
                _alive[x] = false;

                // removing x affects indegree of its out-neighbors and outdegree of its in-neighbors
                for (int e = _headOut[x]; e != -1; e = _nextOut[e])
                {
                    int v = _toOut[e];
                    if (_alive[v])
                    {
                        _indeg[v]--;
                        if (_indeg[v] == 0 || _outdeg[v] == 0) _queue[qt++] = v;
                    }
                }
                for (int e = _headIn[x]; e != -1; e = _nextIn[e])
                {
                    int u = _toIn[e];
                    if (_alive[u])
                    {
                        _outdeg[u]--;
                        if (_indeg[u] == 0 || _outdeg[u] == 0) _queue[qt++] = u;
                    }
                }
            }

            int coreCount = 0;
            for (int v = 0; v < _n; v++)
                if (_alive[v]) _coreVerts[coreCount++] = v;
            return coreCount;
        }

        private int GetSomeCycleInCore(int coreCount, int[] buf)
        {
            Array.Fill(_seenAt, -1);
            int pathLen = 0;

            int start = _coreVerts[0];
            int cur = start;
            while (true)
            {
                int seen = _seenAt[cur];
                if (seen >= 0)
                {
                    int cycLen = pathLen - seen;
                    for (int i = 0; i < cycLen; i++) buf[i] = _pathBuf[seen + i];
                    return cycLen;
                }

                _seenAt[cur] = pathLen;
                _pathBuf[pathLen++] = cur;

                // pick any outgoing edge staying in core
                int next = -1;
                for (int e = _headOut[cur]; e != -1; e = _nextOut[e])
                {
                    int v = _toOut[e];
                    if (_alive[v])
                    {
                        next = v;
                        break;
                    }
                }
                cur = next;
            }
        }

        private void SortCycleByDegreeDesc(int[] cycle, int len)
        {
            for (int i = 1; i < len; i++)
            {
                int v = cycle[i];
                int dv = _indeg[v] + _outdeg[v];
                int j = i - 1;
                while (j >= 0)
                {
                    int u = cycle[j];
                    int du = _indeg[u] + _outdeg[u];
                    if (du >= dv) break;
                    cycle[j + 1] = u;
                    j--;
                }
                cycle[j + 1] = v;
            }
        }
    }
}
