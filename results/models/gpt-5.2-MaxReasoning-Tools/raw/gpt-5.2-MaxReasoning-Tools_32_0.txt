// Steiner Tree in undirected weighted graphs (exact for small terminal sets, otherwise heuristic).
// Reads:
//   num_vertices num_edges num_terminals
//   u v w  (0-indexed, undirected)
//   t1 t2 ... tk
// Writes:
//   total_weight edge_count
//   u v   (each on its own line; edges form a tree spanning all terminals)
//
// Compatible with MSVC /std:c++20 (no non-standard headers).

#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

using std::cin;
using std::cout;
using std::pair;
using std::priority_queue;
using std::uint32_t;
using std::uint64_t;
using std::vector;

static constexpr long long INF = (std::numeric_limits<long long>::max)() / 4;

struct Edge {
  int u = -1;
  int v = -1;
  long long w = 0;
};

struct AdjEdge {
  int to = -1;
  int id = -1;
  long long w = 0;
};

struct DSU {
  vector<int> parent;
  vector<int> size;

  explicit DSU(int n = 0) : parent(n), size(n, 1) { std::iota(parent.begin(), parent.end(), 0); }

  int find(int x) {
    while (parent[x] != x) {
      parent[x] = parent[parent[x]];
      x = parent[x];
    }
    return x;
  }

  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (size[a] < size[b]) std::swap(a, b);
    parent[b] = a;
    size[a] += size[b];
    return true;
  }
};

static uint64_t undirected_pair_key(int a, int b) {
  if (a > b) std::swap(a, b);
  return (uint64_t(uint32_t(a)) << 32) | uint32_t(b);
}

static void dijkstra_single_source(const vector<vector<AdjEdge>>& adj,
                                   int source,
                                   vector<long long>& dist,
                                   vector<int>& parent,
                                   vector<int>& parent_edge) {
  const int n = int(adj.size());
  dist.assign(n, INF);
  parent.assign(n, -1);
  parent_edge.assign(n, -1);

  using PQItem = pair<long long, int>;
  priority_queue<PQItem, vector<PQItem>, std::greater<PQItem>> pq;
  dist[source] = 0;
  pq.push({0, source});

  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != dist[v]) continue;
    for (const auto& e : adj[v]) {
      const long long nd = d + e.w;
      if (nd < dist[e.to]) {
        dist[e.to] = nd;
        parent[e.to] = v;
        parent_edge[e.to] = e.id;
        pq.push({nd, e.to});
      }
    }
  }
}

static void dijkstra_multi_source(const vector<vector<AdjEdge>>& adj,
                                  const vector<int>& sources,
                                  vector<long long>& dist,
                                  vector<int>& representative_terminal,
                                  vector<int>& parent,
                                  vector<int>& parent_edge) {
  const int n = int(adj.size());
  dist.assign(n, INF);
  representative_terminal.assign(n, -1);
  parent.assign(n, -1);
  parent_edge.assign(n, -1);

  using PQItem = pair<long long, int>;
  priority_queue<PQItem, vector<PQItem>, std::greater<PQItem>> pq;

  for (int s : sources) {
    if (s < 0 || s >= n) continue;
    dist[s] = 0;
    representative_terminal[s] = s;
    pq.push({0, s});
  }

  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != dist[v]) continue;
    const int rep = representative_terminal[v];
    for (const auto& e : adj[v]) {
      const long long nd = d + e.w;
      const int nrep = rep;
      if (nd < dist[e.to] || (nd == dist[e.to] && (representative_terminal[e.to] == -1 || nrep < representative_terminal[e.to]))) {
        dist[e.to] = nd;
        representative_terminal[e.to] = nrep;
        parent[e.to] = v;
        parent_edge[e.to] = e.id;
        pq.push({nd, e.to});
      }
    }
  }
}

static vector<int> kruskal_mst_from_edges(int n, const vector<Edge>& edges, const vector<int>& edge_ids) {
  vector<int> sorted = edge_ids;
  std::sort(sorted.begin(), sorted.end(), [&](int a, int b) {
    if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
    if (edges[a].u != edges[b].u) return edges[a].u < edges[b].u;
    return edges[a].v < edges[b].v;
  });

  DSU dsu(n);
  vector<int> picked;
  picked.reserve(sorted.size());
  for (int eid : sorted) {
    if (eid < 0 || eid >= int(edges.size())) continue;
    if (dsu.unite(edges[eid].u, edges[eid].v)) picked.push_back(eid);
  }
  return picked;
}

static vector<int> prune_nonterminal_leaves(int n,
                                            const vector<Edge>& edges,
                                            const vector<int>& tree_edge_ids,
                                            const vector<char>& is_terminal) {
  const int ecount = int(tree_edge_ids.size());
  if (ecount == 0) return {};

  vector<vector<pair<int, int>>> tadj(n);  // (to, localEdgeId)
  vector<int> degree(n, 0);
  vector<int> local_to_global(ecount, -1);
  vector<char> active(ecount, 1);
  for (int i = 0; i < ecount; i++) {
    const int gid = tree_edge_ids[i];
    local_to_global[i] = gid;
    const int u = edges[gid].u;
    const int v = edges[gid].v;
    tadj[u].push_back({v, i});
    tadj[v].push_back({u, i});
    degree[u]++;
    degree[v]++;
  }

  std::queue<int> q;
  for (int v = 0; v < n; v++) {
    if (degree[v] == 1 && !is_terminal[v]) q.push(v);
  }

  while (!q.empty()) {
    const int v = q.front();
    q.pop();
    if (degree[v] != 1 || is_terminal[v]) continue;
    int neighbor = -1;
    int local_eid = -1;
    for (const auto& [to, lid] : tadj[v]) {
      if (active[lid]) {
        neighbor = to;
        local_eid = lid;
        break;
      }
    }
    if (local_eid == -1) {
      degree[v] = 0;
      continue;
    }
    active[local_eid] = 0;
    degree[v]--;
    degree[neighbor]--;
    if (degree[neighbor] == 1 && !is_terminal[neighbor]) q.push(neighbor);
  }

  vector<int> kept;
  kept.reserve(ecount);
  for (int lid = 0; lid < ecount; lid++) {
    if (active[lid]) kept.push_back(local_to_global[lid]);
  }
  return kept;
}

static bool terminals_connected(int n, const vector<Edge>& edges, const vector<int>& edge_ids, const vector<int>& terminals) {
  if (terminals.empty()) return true;
  DSU dsu(n);
  for (int eid : edge_ids) dsu.unite(edges[eid].u, edges[eid].v);
  const int root = dsu.find(terminals[0]);
  for (int t : terminals) {
    if (dsu.find(t) != root) return false;
  }
  return true;
}

// Exact DP (Dreyfus-Wagner variant) with path reconstruction, feasible for small terminal sets.
static bool should_run_exact(int n, int m, int k) {
  if (k <= 1) return false;
  if (k > 10) return false;
  const uint64_t states = (1ULL << k);
  const uint64_t cells = states * uint64_t(n);
  // dp + pre roughly 16 bytes per cell -> keep it comfortably small.
  if (cells > 6'000'000ULL) return false;
  // Also avoid large graphs even if cells small (Dijkstra per mask).
  if (m > 300'000) return false;
  return true;
}

static vector<int> steiner_exact_dp(const vector<vector<AdjEdge>>& adj, const vector<Edge>& edges, const vector<int>& terminals) {
  const int n = int(adj.size());
  const int k = int(terminals.size());
  const int full = (1 << k) - 1;

  vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));
  vector<vector<pair<int, int>>> pre(1 << k, vector<pair<int, int>>(n, {-1, -1}));  // {subset,-1} or {from,eid}

  for (int i = 0; i < k; i++) dp[1 << i][terminals[i]] = 0;

  using PQItem = pair<long long, int>;
  for (int mask = 1; mask <= full; mask++) {
    for (int v = 0; v < n; v++) {
      for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
        const int other = mask ^ sub;
        if (sub > other) continue;
        const long long a = dp[sub][v];
        const long long b = dp[other][v];
        if (a >= INF || b >= INF) continue;
        const long long cand = a + b;
        if (cand < dp[mask][v]) {
          dp[mask][v] = cand;
          pre[mask][v] = {sub, -1};
        }
      }
    }
    if (mask == full) break;

    priority_queue<PQItem, vector<PQItem>, std::greater<PQItem>> pq;
    for (int v = 0; v < n; v++) {
      if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
    }
    while (!pq.empty()) {
      auto [d, v] = pq.top();
      pq.pop();
      if (d != dp[mask][v]) continue;
      for (const auto& e : adj[v]) {
        const long long nd = d + e.w;
        if (nd < dp[mask][e.to]) {
          dp[mask][e.to] = nd;
          pre[mask][e.to] = {v, e.id};
          pq.push({nd, e.to});
        }
      }
    }
  }

  const int root = terminals[0];
  if (dp[full][root] >= INF) return {};

  vector<int> ret;
  ret.reserve(n);
  std::queue<pair<int, int>> q;
  q.push({full, root});
  while (!q.empty()) {
    auto [mask, v] = q.front();
    q.pop();
    const auto [a, b] = pre[mask][v];
    if (a == -1) continue;
    if (b == -1) {
      const int sub = a;
      if (sub == 0 || sub == mask) continue;
      q.push({sub, v});
      q.push({mask ^ sub, v});
    } else {
      const int from = a;
      const int eid = b;
      ret.push_back(eid);
      q.push({mask, from});
    }
  }
  return ret;
}

static void collect_path_edges_to_terminal(int start,
                                           int terminal,
                                           const vector<int>& parent,
                                           const vector<int>& parent_edge,
                                           vector<int>& out) {
  int cur = start;
  while (cur != terminal && cur != -1) {
    const int eid = parent_edge[cur];
    if (eid < 0) break;
    out.push_back(eid);
    cur = parent[cur];
  }
}

// Mehlhorn distance-network heuristic (2-approx), expanded without extra shortest-path runs.
static vector<int> steiner_mehlhorn(const vector<vector<AdjEdge>>& adj, const vector<Edge>& edges, const vector<int>& terminals) {
  const int n = int(adj.size());
  const int k = int(terminals.size());
  if (k <= 1) return {};

  vector<long long> dist;
  vector<int> rep, parent, parent_edge;
  dijkstra_multi_source(adj, terminals, dist, rep, parent, parent_edge);

  vector<int> term_index(n, -1);
  for (int i = 0; i < k; i++) term_index[terminals[i]] = i;

  struct DNBest {
    int ta = -1, tb = -1;
    long long w = INF;
    int u = -1, v = -1;  // u in region ta, v in region tb
    int bridge_eid = -1;
  };

  std::unordered_map<uint64_t, DNBest> best;
  best.reserve(edges.size() * 2ULL);

  for (int eid = 0; eid < int(edges.size()); eid++) {
    const int a = edges[eid].u;
    const int b = edges[eid].v;
    const int ta = rep[a];
    const int tb = rep[b];
    if (ta < 0 || tb < 0 || ta == tb) continue;
    const long long cand = dist[a] + edges[eid].w + dist[b];
    const int tmin = std::min(ta, tb);
    const int tmax = std::max(ta, tb);
    const uint64_t key = undirected_pair_key(tmin, tmax);
    auto it = best.find(key);
    if (it == best.end() || cand < it->second.w) {
      DNBest info;
      info.ta = tmin;
      info.tb = tmax;
      info.w = cand;
      info.bridge_eid = eid;
      if (ta == tmin) {
        info.u = a;
        info.v = b;
      } else {
        info.u = b;
        info.v = a;
      }
      best[key] = info;
    }
  }

  vector<DNBest> dn_edges;
  dn_edges.reserve(best.size());
  for (const auto& kv : best) dn_edges.push_back(kv.second);

  std::sort(dn_edges.begin(), dn_edges.end(), [](const DNBest& x, const DNBest& y) {
    if (x.w != y.w) return x.w < y.w;
    if (x.ta != y.ta) return x.ta < y.ta;
    return x.tb < y.tb;
  });

  DSU dsu_terms(k);
  vector<DNBest> dn_mst;
  dn_mst.reserve(k - 1);
  for (const auto& e : dn_edges) {
    const int ia = term_index[e.ta];
    const int ib = term_index[e.tb];
    if (ia < 0 || ib < 0) continue;
    if (dsu_terms.unite(ia, ib)) {
      dn_mst.push_back(e);
      if (int(dn_mst.size()) == k - 1) break;
    }
  }
  if (int(dn_mst.size()) != k - 1) return {};

  vector<int> expanded;
  expanded.reserve(k * 8);
  for (const auto& e : dn_mst) {
    collect_path_edges_to_terminal(e.u, e.ta, parent, parent_edge, expanded);
    expanded.push_back(e.bridge_eid);
    collect_path_edges_to_terminal(e.v, e.tb, parent, parent_edge, expanded);
  }

  std::sort(expanded.begin(), expanded.end());
  expanded.erase(std::unique(expanded.begin(), expanded.end()), expanded.end());

  vector<int> tree = kruskal_mst_from_edges(n, edges, expanded);
  if (!terminals_connected(n, edges, tree, terminals)) return {};

  vector<char> is_terminal(n, 0);
  for (int t : terminals) is_terminal[t] = 1;
  tree = prune_nonterminal_leaves(n, edges, tree, is_terminal);
  if (!terminals_connected(n, edges, tree, terminals)) return {};
  return tree;
}

// Fallback: connect all terminals to a root using a shortest-path tree, then prune leaves.
static vector<int> steiner_shortest_path_tree(const vector<vector<AdjEdge>>& adj, const vector<Edge>& edges, const vector<int>& terminals) {
  const int n = int(adj.size());
  if (terminals.size() <= 1) return {};
  const int root = terminals[0];

  vector<long long> dist;
  vector<int> parent, parent_edge;
  dijkstra_single_source(adj, root, dist, parent, parent_edge);

  for (int t : terminals) {
    if (dist[t] >= INF) return {};
  }

  vector<char> in_tree_vertex(n, 0);
  in_tree_vertex[root] = 1;

  vector<int> used_edges;
  used_edges.reserve(terminals.size() * 4);

  for (int t : terminals) {
    int cur = t;
    while (!in_tree_vertex[cur]) {
      in_tree_vertex[cur] = 1;
      const int eid = parent_edge[cur];
      if (eid < 0) break;
      used_edges.push_back(eid);
      cur = parent[cur];
    }
  }

  std::sort(used_edges.begin(), used_edges.end());
  used_edges.erase(std::unique(used_edges.begin(), used_edges.end()), used_edges.end());

  vector<int> tree = used_edges;  // already a subtree of an SPT -> acyclic
  vector<char> is_terminal(n, 0);
  for (int t : terminals) is_terminal[t] = 1;
  tree = prune_nonterminal_leaves(n, edges, tree, is_terminal);
  if (!terminals_connected(n, edges, tree, terminals)) return {};
  return tree;
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n = 0, m_in = 0, k_in = 0;
  if (!(cin >> n >> m_in >> k_in)) return 0;

  // Collapse parallel edges by keeping only the minimum weight for each unordered pair.
  std::unordered_map<uint64_t, int> best_edge_id;
  best_edge_id.reserve(size_t(m_in) * 2ULL);
  vector<Edge> edges;
  edges.reserve(m_in);

  for (int i = 0; i < m_in; i++) {
    int u = 0, v = 0;
    long long w = 0;
    cin >> u >> v >> w;
    if (u == v) continue;
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    const int a = std::min(u, v);
    const int b = std::max(u, v);
    const uint64_t key = undirected_pair_key(a, b);
    auto it = best_edge_id.find(key);
    if (it == best_edge_id.end()) {
      const int id = int(edges.size());
      edges.push_back({a, b, w});
      best_edge_id.emplace(key, id);
    } else {
      const int id = it->second;
      if (w < edges[id].w) edges[id].w = w;
    }
  }

  vector<int> terminals;
  terminals.reserve(k_in);
  for (int i = 0; i < k_in; i++) {
    int t = 0;
    cin >> t;
    if (t >= 0 && t < n) terminals.push_back(t);
  }
  std::sort(terminals.begin(), terminals.end());
  terminals.erase(std::unique(terminals.begin(), terminals.end()), terminals.end());

  if (terminals.size() <= 1) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  vector<vector<AdjEdge>> adj(n);
  adj.reserve(n);
  for (int eid = 0; eid < int(edges.size()); eid++) {
    const auto& e = edges[eid];
    adj[e.u].push_back({e.v, eid, e.w});
    adj[e.v].push_back({e.u, eid, e.w});
  }

  vector<char> is_terminal(n, 0);
  for (int t : terminals) is_terminal[t] = 1;

  vector<int> tree_edges;

  if (should_run_exact(n, int(edges.size()), int(terminals.size()))) {
    vector<int> exact_edges = steiner_exact_dp(adj, edges, terminals);
    if (!exact_edges.empty()) {
      std::sort(exact_edges.begin(), exact_edges.end());
      exact_edges.erase(std::unique(exact_edges.begin(), exact_edges.end()), exact_edges.end());
      vector<int> mst = kruskal_mst_from_edges(n, edges, exact_edges);
      mst = prune_nonterminal_leaves(n, edges, mst, is_terminal);
      if (terminals_connected(n, edges, mst, terminals)) tree_edges = std::move(mst);
    }
  }

  if (tree_edges.empty()) {
    tree_edges = steiner_mehlhorn(adj, edges, terminals);
  }
  if (tree_edges.empty()) {
    tree_edges = steiner_shortest_path_tree(adj, edges, terminals);
  }
  if (tree_edges.empty()) {
    // As a last resort, output a trivial (invalid) solution is worse than exiting;
    // but instances should be feasible. Still, avoid UB.
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  long long total_weight = 0;
  for (int eid : tree_edges) total_weight += edges[eid].w;

  cout << total_weight << " " << tree_edges.size() << "\n";
  for (int eid : tree_edges) {
    cout << edges[eid].u << " " << edges[eid].v << "\n";
  }
  return 0;
}
