declare const require: any;
declare const process: any;

type AnyId = any;

interface InputTask {
  id: AnyId;
  duration: number;
  deps?: AnyId[];
  deadline?: number;
  weight?: number;
}

interface InputData {
  machineCount: number;
  tasks: InputTask[];
}

class BinaryHeap<T> {
  private data: T[] = [];
  constructor(private readonly compare: (a: T, b: T) => number) {}

  size(): number {
    return this.data.length;
  }

  peek(): T | undefined {
    return this.data[0];
  }

  push(item: T): void {
    const a = this.data;
    a.push(item);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(a[i], a[p]) < 0) {
        const tmp = a[i];
        a[i] = a[p];
        a[p] = tmp;
        i = p;
      } else {
        break;
      }
    }
  }

  pop(): T | undefined {
    const a = this.data;
    const n = a.length;
    if (n === 0) return undefined;
    const root = a[0];
    const last = a.pop() as T;
    if (n > 1) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        if (l >= a.length) break;
        const r = l + 1;
        let best = l;
        if (r < a.length && this.compare(a[r], a[l]) < 0) best = r;
        if (this.compare(a[best], a[i]) < 0) {
          const tmp = a[i];
          a[i] = a[best];
          a[best] = tmp;
          i = best;
        } else {
          break;
        }
      }
    }
    return root;
  }
}

function idKey(id: AnyId): string {
  const t = typeof id;
  if (t === "string") return "s:" + id;
  if (t === "number") return "n:" + String(id);
  if (t === "bigint") return "bi:" + String(id);
  if (id === null) return "null";
  return t + ":" + String(id);
}

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
let jsonText = input;
if (jsonText.length > 0 && jsonText.charCodeAt(0) === 0xfeff) jsonText = jsonText.slice(1);
const data = JSON.parse(jsonText) as InputData;
const tasks: InputTask[] = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;
const rawMachineCount = Number((data as any).machineCount);
const machineCountRaw = Number.isFinite(rawMachineCount) && rawMachineCount > 0 ? Math.trunc(rawMachineCount) : 1;

if (n === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const ids = new Array<AnyId>(n);
const dur = new Float64Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);
const idToIdx = new Map<string, number>();

for (let i = 0; i < n; i++) {
  const t = tasks[i] as any;
  ids[i] = t.id;
  idToIdx.set(idKey(ids[i]), i);

  const p = Number(t.duration);
  dur[i] = Number.isFinite(p) && p >= 0 ? Math.floor(p) : 0;

  const d = Number(t.deadline);
  deadline[i] = Number.isFinite(d) ? Math.floor(d) : Number.POSITIVE_INFINITY;

  const w = Number(t.weight);
  weight[i] = Number.isFinite(w) && w > 0 ? w : 1;
}

// Adjacency (successors): head[u] -> edge list in (to,next)
const head = new Int32Array(n);
head.fill(-1);
const to: number[] = [];
const next: number[] = [];
const indegree = new Int32Array(n);

for (let v = 0; v < n; v++) {
  const depsAny = (tasks[v] as any)?.deps;
  const deps = Array.isArray(depsAny) ? depsAny : [];
  if (deps.length === 0) continue;

  let seen: Set<number> | null = null;
  if (deps.length > 1) seen = new Set<number>();

  for (let k = 0; k < deps.length; k++) {
    const u = idToIdx.get(idKey(deps[k]));
    if (u === undefined) continue;
    if (u === v) continue;
    if (seen) {
      if (seen.has(u)) continue;
      seen.add(u);
    }
    const e = to.length;
    to.push(v);
    next.push(head[u]);
    head[u] = e;
    indegree[v]++;
  }
}

// Topological order (Kahn). If cyclic, append remaining nodes in index order.
const indegTmp = new Int32Array(indegree);
const queue: number[] = [];
for (let i = 0; i < n; i++) if (indegTmp[i] === 0) queue.push(i);
let qh = 0;
const topo: number[] = [];
while (qh < queue.length) {
  const u = queue[qh++];
  topo.push(u);
  for (let e = head[u]; e !== -1; e = next[e]) {
    const v = to[e];
    const d = indegTmp[v] - 1;
    indegTmp[v] = d;
    if (d === 0) queue.push(v);
  }
}
if (topo.length < n) {
  const seen = new Uint8Array(n);
  for (let i = 0; i < topo.length; i++) seen[topo[i]] = 1;
  for (let i = 0; i < n; i++) if (seen[i] === 0) topo.push(i);
}

// Bottom level (critical path length from node to exit, incl. self).
const bottom = new Float64Array(n);
for (let i = topo.length - 1; i >= 0; i--) {
  const u = topo[i];
  let maxSucc = 0;
  for (let e = head[u]; e !== -1; e = next[e]) {
    const v = to[e];
    const b = bottom[v];
    if (b > maxSucc) maxSucc = b;
  }
  bottom[u] = dur[u] + maxSucc;
}

// Effective deadlines: propagate successors' deadlines backward.
const effDeadline = new Float64Array(deadline);
for (let i = topo.length - 1; i >= 0; i--) {
  const u = topo[i];
  let best = effDeadline[u];
  for (let e = head[u]; e !== -1; e = next[e]) {
    const v = to[e];
    const cand = effDeadline[v] - dur[v];
    if (cand < best) best = cand;
  }
  effDeadline[u] = best;
}

const ratio = new Float64Array(n);
for (let i = 0; i < n; i++) ratio[i] = dur[i] / weight[i];

const machineCount = Math.min(machineCountRaw, n);
const beta = machineCount <= 1 ? 0 : 0.25;
const urgency = new Float64Array(n);
for (let i = 0; i < n; i++) urgency[i] = effDeadline[i] - beta * bottom[i];

const readyHeap = new BinaryHeap<number>((a, b) => {
  const ua = urgency[a],
    ub = urgency[b];
  if (ua < ub) return -1;
  if (ua > ub) return 1;

  const ra = ratio[a],
    rb = ratio[b];
  if (ra < rb) return -1;
  if (ra > rb) return 1;

  const ba = bottom[a],
    bb = bottom[b];
  if (ba > bb) return -1;
  if (ba < bb) return 1;

  return a - b;
});

const remainingDeps = new Int32Array(indegree);
for (let i = 0; i < n; i++) if (remainingDeps[i] === 0) readyHeap.push(i);

type MachineState = { time: number; machine: number; lastTask: number };
const machineHeap = new BinaryHeap<MachineState>((a, b) => {
  if (a.time < b.time) return -1;
  if (a.time > b.time) return 1;
  // Process completions before pure "idle wait" markers at the same timestamp.
  const aKind = a.lastTask === -1 ? 1 : 0;
  const bKind = b.lastTask === -1 ? 1 : 0;
  if (aKind !== bKind) return aKind - bKind;
  return a.machine - b.machine;
});

for (let m = 0; m < machineCount; m++) machineHeap.push({ time: 0, machine: m, lastTask: -1 });

const schedule: { id: AnyId; start: number; machine: number }[] = [];
let scheduledCount = 0;

while (scheduledCount < n) {
  const ms = machineHeap.pop();
  if (!ms) break;
  const t = ms.time;

  if (ms.lastTask !== -1) {
    const finished = ms.lastTask;
    for (let e = head[finished]; e !== -1; e = next[e]) {
      const v = to[e];
      const d = remainingDeps[v] - 1;
      remainingDeps[v] = d;
      if (d === 0) readyHeap.push(v);
    }
  }

  if (readyHeap.size() > 0) {
    const job = readyHeap.pop() as number;
    schedule.push({ id: ids[job], start: t, machine: ms.machine });
    const finish = t + dur[job];
    machineHeap.push({ time: finish, machine: ms.machine, lastTask: job });
    scheduledCount++;
  } else {
    const nextMs = machineHeap.peek();
    if (!nextMs) break;
    if (nextMs.time < t) break;
    if (nextMs.time === t && nextMs.lastTask === -1) break;
    machineHeap.push({ time: nextMs.time, machine: ms.machine, lastTask: -1 });
  }
}

if (scheduledCount === n) {
  process.stdout.write(JSON.stringify({ schedule }));
} else {
  const order = topo.map((i) => ids[i]);
  process.stdout.write(JSON.stringify({ order }));
}
