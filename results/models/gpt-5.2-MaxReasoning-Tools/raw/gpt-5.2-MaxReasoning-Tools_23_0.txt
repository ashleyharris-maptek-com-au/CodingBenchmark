use std::f64::consts::PI;
use std::io::{self, Read, Write};
use std::sync::mpsc;
use std::thread;
use std::time::{Duration, Instant};

struct Scanner<R: Read> {
    reader: R,
    buf: [u8; 1 << 16],
    len: usize,
    pos: usize,
    token: Vec<u8>,
}

impl<R: Read> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: [0; 1 << 16],
            len: 0,
            pos: 0,
            token: Vec::with_capacity(128),
        }
    }

    fn refill(&mut self) -> bool {
        self.pos = 0;
        match self.reader.read(&mut self.buf) {
            Ok(0) | Err(_) => {
                self.len = 0;
                false
            }
            Ok(n) => {
                self.len = n;
                true
            }
        }
    }

    fn next_token_bytes(&mut self) -> Option<&[u8]> {
        self.token.clear();

        loop {
            if self.pos >= self.len && !self.refill() {
                return None;
            }
            while self.pos < self.len && self.buf[self.pos].is_ascii_whitespace() {
                self.pos += 1;
            }
            if self.pos < self.len {
                break;
            }
        }

        loop {
            if self.pos >= self.len {
                if !self.refill() {
                    break;
                }
                continue;
            }
            let b = self.buf[self.pos];
            if b.is_ascii_whitespace() {
                break;
            }
            self.token.push(b);
            self.pos += 1;
        }

        Some(self.token.as_slice())
    }

    fn next_parse<T: std::str::FromStr>(&mut self) -> Option<T> {
        let tok = self.next_token_bytes()?;
        let s = unsafe { std::str::from_utf8_unchecked(tok) };
        s.parse::<T>().ok()
    }
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

#[derive(Clone, Copy, Debug)]
struct InitData {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    lz_min_x: f64, // inclusive edge in meters
    lz_max_x: f64, // inclusive edge in meters
    cruise_y: f64, // desired cruise y (smaller = higher)
}

#[derive(Clone, Copy, Debug)]
enum Mode {
    Transit,
    Descend,
}

struct Controller {
    init: InitData,
    mode: Mode,
    x_goal: f64,
    y_goal: f64,
}

impl Controller {
    fn new(init: InitData) -> Self {
        let mut x_goal = init.target_x;
        if init.lz_max_x > init.lz_min_x + 1.0 {
            x_goal = x_goal.clamp(init.lz_min_x + 0.5, init.lz_max_x - 0.5);
        }
        Self {
            init,
            mode: Mode::Transit,
            x_goal,
            y_goal: init.cruise_y,
        }
    }

    fn update_mode(&mut self, s: &State) {
        match self.mode {
            Mode::Transit => {
                let near_x = (s.x - self.x_goal).abs() < 2.0 && s.vx.abs() < 1.5;
                if near_x {
                    self.mode = Mode::Descend;
                }
            }
            Mode::Descend => {}
        }
    }

    fn compute(&mut self, s: &State) -> (f64, f64) {
        self.update_mode(s);

        let alt_above_pad = self.init.target_y - s.y; // >0 means above pad
        let in_zone = s.x >= self.init.lz_min_x && s.x <= self.init.lz_max_x;

        let (kx, kvx, vx_max, ky, kvy, vy_up_max, vy_down_max, tilt_max) = match self.mode {
            Mode::Transit => (0.25, 0.9, 20.0, 0.35, 0.9, 12.0, 18.0, 0.85),
            Mode::Descend => {
                let vy_down = (alt_above_pad * 0.5 + 1.5).clamp(1.5, 10.0);
                let tilt = if alt_above_pad < 8.0 { 0.35 } else { 0.65 };
                (0.45, 1.2, 7.0, 0.35, 1.2, 10.0, vy_down, tilt)
            }
        };

        self.y_goal = match self.mode {
            Mode::Transit => self.init.cruise_y,
            Mode::Descend => self.init.target_y,
        };

        let x_goal = if in_zone {
            self.x_goal
        } else {
            // When outside landing zone during descent, bias back to center to avoid sidewall clipping.
            if self.init.lz_max_x > self.init.lz_min_x + 1.0 {
                ((self.init.lz_min_x + self.init.lz_max_x) * 0.5).clamp(
                    self.init.lz_min_x + 0.5,
                    self.init.lz_max_x - 0.5,
                )
            } else {
                self.x_goal
            }
        };

        // Desired velocities from position errors.
        let x_err = x_goal - s.x;
        let vx_des = (x_err * kx).clamp(-vx_max, vx_max);
        let ax_des = (vx_des - s.vx) * kvx;

        let y_err = self.y_goal - s.y;
        let mut vy_des = (y_err * ky).clamp(-vy_up_max, vy_down_max);

        // Prevent aggressive downward commands close to ground.
        if matches!(self.mode, Mode::Descend) && alt_above_pad < 3.0 {
            vy_des = vy_des.min(1.8);
        }

        let mut ay_des = (vy_des - s.vy) * kvy;
        // Can't accelerate downward faster than free-fall in this model (no "downward thrust").
        ay_des = ay_des.min(self.init.gravity);

        // Convert desired net acceleration to desired thrust vector.
        let t_des_x = ax_des;
        let t_des_y = ay_des - self.init.gravity;

        let mut angle_des = if t_des_x.abs() < 1e-9 && t_des_y.abs() < 1e-9 {
            0.0
        } else {
            t_des_x.atan2(-t_des_y)
        };
        angle_des = angle_des.clamp(-tilt_max, tilt_max);

        // Turn controller (PD on angle).
        let angle_err = wrap_angle(angle_des - s.angle);
        let turn = (2.6 * angle_err - 0.8 * s.angular_vel).clamp(-1.0, 1.0);

        // Thrust magnitude: project desired thrust onto achievable direction.
        let dir_x = angle_des.sin();
        let dir_y = -angle_des.cos();
        let mut thrust_acc = t_des_x * dir_x + t_des_y * dir_y;
        if thrust_acc.is_nan() || !thrust_acc.is_finite() {
            thrust_acc = 0.0;
        }
        thrust_acc = thrust_acc.max(0.0);

        let mut thrust = (thrust_acc / self.init.max_thrust).clamp(0.0, 1.0);

        // Final landing: prefer upright, soften any residual sideways motion.
        if matches!(self.mode, Mode::Descend) && alt_above_pad < 5.0 {
            let upright_bias = (s.angle * 1.2).abs().clamp(0.0, 1.0);
            thrust = (thrust + 0.05 * upright_bias).clamp(0.0, 1.0);
        }

        (thrust, turn)
    }
}

fn wrap_angle(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn is_solid(c: u8) -> bool {
    matches!(c, b'#' | b'L' | b'T')
}

fn main() {
    let (init_tx, init_rx) = mpsc::channel::<InitData>();
    let (state_tx, state_rx) = mpsc::channel::<State>();

    thread::spawn(move || {
        let stdin = io::stdin();
        let lock = stdin.lock();
        let mut sc = Scanner::new(lock);

        let width: usize = match sc.next_parse() {
            Some(v) => v,
            None => return,
        };
        let height: usize = match sc.next_parse() {
            Some(v) => v,
            None => return,
        };
        let gravity: f64 = sc.next_parse().unwrap_or(1.62);
        let max_thrust: f64 = sc.next_parse().unwrap_or(10.0);
        let _fuel_total: f64 = sc.next_parse().unwrap_or(0.0);

        let start_x: f64 = sc.next_parse().unwrap_or(0.0);
        let start_y: f64 = sc.next_parse().unwrap_or(0.0);
        let target_x: f64 = sc.next_parse().unwrap_or(start_x);
        let target_y: f64 = sc.next_parse().unwrap_or(start_y);

        let margin_cols: isize = 200;
        let corridor_min_x = ((start_x.min(target_x).floor() as isize) - margin_cols)
            .clamp(0, width.saturating_sub(1) as isize) as usize;
        let corridor_max_x = ((start_x.max(target_x).ceil() as isize) + margin_cols)
            .clamp(0, width.saturating_sub(1) as isize) as usize;

        let mut lz_min_col: usize = usize::MAX;
        let mut lz_max_col: usize = 0;
        let mut seen_lz = false;

        let mut min_solid_y: usize = height.saturating_sub(1);
        let mut any_solid = false;

        for y in 0..height {
            let row = match sc.next_token_bytes() {
                Some(v) => v,
                None => break,
            };
            if row.len() < corridor_min_x + 1 {
                continue;
            }
            let end = (corridor_max_x + 1).min(row.len());
            for (dx, &c) in row[corridor_min_x..end].iter().enumerate() {
                let x = corridor_min_x + dx;
                if c == b'L' {
                    seen_lz = true;
                    lz_min_col = lz_min_col.min(x);
                    lz_max_col = lz_max_col.max(x);
                }
                if is_solid(c) {
                    any_solid = true;
                    if y < min_solid_y {
                        min_solid_y = y;
                    }
                }
            }
        }

        let (lz_min_x, lz_max_x) = if seen_lz && lz_min_col != usize::MAX && lz_max_col >= lz_min_col
        {
            (lz_min_col as f64, (lz_max_col + 1) as f64)
        } else {
            (target_x - 1.0, target_x + 1.0)
        };

        let cruise_margin = 20.0;
        let cruise_y = if any_solid {
            ((min_solid_y as f64) - cruise_margin).max(0.0)
        } else {
            // Default to "somewhat above the pad" if the corridor has no solids.
            (target_y - 60.0).max(0.0)
        };

        let init = InitData {
            width,
            height,
            gravity,
            max_thrust,
            start_x,
            start_y,
            target_x,
            target_y,
            lz_min_x,
            lz_max_x,
            cruise_y,
        };
        let _ = init_tx.send(init);

        loop {
            let x: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };
            let y: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };
            let vx: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };
            let vy: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };
            let angle: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };
            let angular_vel: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };
            let fuel_remaining: f64 = match sc.next_parse() {
                Some(v) => v,
                None => break,
            };

            let st = State {
                x,
                y,
                vx,
                vy,
                angle,
                angular_vel,
                fuel_remaining,
            };
            let _ = state_tx.send(st);
        }
    });

    let init = match init_rx.recv() {
        Ok(v) => v,
        Err(_) => return,
    };
    let mut controller = Controller::new(init);

    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(8 * 1024, stdout.lock());

    let tick = Duration::from_millis(50); // 20 Hz
    let mut next_tick = Instant::now();

    let mut latest_state: Option<State> = None;
    let mut disconnected_at: Option<Instant> = None;

    loop {
        // Drain any queued states; keep the newest.
        loop {
            match state_rx.try_recv() {
                Ok(s) => {
                    latest_state = Some(s);
                    disconnected_at = None;
                }
                Err(mpsc::TryRecvError::Empty) => break,
                Err(mpsc::TryRecvError::Disconnected) => {
                    if disconnected_at.is_none() {
                        disconnected_at = Some(Instant::now());
                    }
                    break;
                }
            }
        }

        let (thrust, turn) = if let Some(s) = latest_state {
            controller.compute(&s)
        } else {
            (0.0, 0.0)
        };

        // Output at >= 20Hz. Always flush to avoid buffering delays.
        let _ = writeln!(&mut out, "{:.3} {:.3}", thrust, turn);
        let _ = out.flush();

        // If input ended and we haven't received state for a while, exit.
        if let Some(t0) = disconnected_at {
            if Instant::now().duration_since(t0) > Duration::from_secs(2) {
                break;
            }
        }

        next_tick += tick;
        let now = Instant::now();
        if next_tick > now {
            let mut remaining = next_tick - now;
            if remaining > Duration::from_millis(2) {
                thread::sleep(remaining - Duration::from_millis(2));
                remaining = next_tick.saturating_duration_since(Instant::now());
            }
            while Instant::now() < next_tick {
                std::hint::spin_loop();
            }
        } else {
            next_tick = now;
        }
    }
}
