using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int vCount;
        int fCount;
        if (!scanner.TryNextInt(out vCount) || !scanner.TryNextInt(out fCount))
        {
            Console.WriteLine("0");
            return;
        }

        if (vCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        var verts = new Vec3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            double x = scanner.NextDouble();
            double y = scanner.NextDouble();
            double z = scanner.NextDouble();
            verts[i] = new Vec3(x, y, z);
        }

        // Faces are irrelevant for this conservative packing approach; still consume them.
        for (int i = 0; i < fCount; i++)
        {
            scanner.NextInt();
            scanner.NextInt();
            scanner.NextInt();
        }

        double W = scanner.NextDouble();
        double H = scanner.NextDouble();
        double D = scanner.NextDouble();

        if (!(W > 0.0 && H > 0.0 && D > 0.0))
        {
            Console.WriteLine("0");
            return;
        }

        // Adaptive candidate orientations:
        // - Always include the 24 axis-aligned cube rotations (fast, robust).
        // - Optionally add PCA-based orientation and a small set of deterministic random samples for better AABB fit.
        var candidates = new List<Orientation>(64);
        candidates.AddRange(OrientationLibrary.AxisAligned24());

        // PCA orientation can significantly help when the input mesh is not axis-aligned.
        if (vCount >= 3)
        {
            Orientation pca = OrientationLibrary.TryPcaOrientation(verts);
            if (pca.IsValid) candidates.Add(pca);
        }

        // Deterministic random sampling: bounded by vertex count for scalability.
        int randomSamples = vCount <= 20_000 ? 24 : (vCount <= 100_000 ? 8 : 0);
        if (randomSamples > 0)
            candidates.AddRange(OrientationLibrary.RandomSamples(verts, W, H, D, randomSamples));

        // Evaluate candidates by conservative "AABB brick" packing count.
        long bestTheoretical = 0;
        Orientation bestOrientation = default;
        Aabb bestAabb = default;
        (long nx, long ny, long nz) bestGrid = default;
        (double sx, double sy, double sz) bestStep = default;

        foreach (var ori in candidates)
        {
            Aabb aabb = Geometry.ComputeAabb(verts, ori.R);
            double dx = aabb.Dx, dy = aabb.Dy, dz = aabb.Dz;
            if (!(dx > 0.0 && dy > 0.0 && dz > 0.0)) continue;
            if (dx > W || dy > H || dz > D) continue;

            // Tight brick packing of conservative AABBs. Touching is typically allowed; this maximizes count.
            // We still guard against floating-point edge cases by validating the final fit in CountFitTight().
            double stepX = dx;
            double stepY = dy;
            double stepZ = dz;

            long nx = Packing.CountFitTight(W, dx);
            long ny = Packing.CountFitTight(H, dy);
            long nz = Packing.CountFitTight(D, dz);

            long total = Packing.SaturatingMul(Packing.SaturatingMul(nx, ny), nz);
            if (total > bestTheoretical)
            {
                bestTheoretical = total;
                bestOrientation = ori;
                bestAabb = aabb;
                bestGrid = (nx, ny, nz);
                bestStep = (stepX, stepY, stepZ);
            }
        }

        if (bestTheoretical <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        long maxOutput = Packing.DecideOutputCap(W, H, D, vCount);
        long outCount = bestTheoretical < maxOutput ? bestTheoretical : maxOutput;

        var writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 20);
        writer.NewLine = "\n";
        writer.WriteLine(outCount.ToString(CultureInfo.InvariantCulture));

        // Grid placement of conservative AABBs.
        // Translation is chosen so that the rotated mesh's AABB fits in [0,W]x[0,H]x[0,D].
        double startX = -bestAabb.MinX;
        double startY = -bestAabb.MinY;
        double startZ = -bestAabb.MinZ;

        long written = 0;
        for (long kz = 0; kz < bestGrid.nz && written < outCount; kz++)
        {
            double tz = startZ + kz * bestStep.sz;
            for (long ky = 0; ky < bestGrid.ny && written < outCount; ky++)
            {
                double ty = startY + ky * bestStep.sy;
                for (long kx = 0; kx < bestGrid.nx && written < outCount; kx++)
                {
                    double tx = startX + kx * bestStep.sx;

                    // Output: tx ty tz qw qx qy qz
                    // Keep formatting compact and culture-invariant.
                    writer.Write(Format.D(tx)); writer.Write(' ');
                    writer.Write(Format.D(ty)); writer.Write(' ');
                    writer.Write(Format.D(tz)); writer.Write(' ');
                    writer.Write(Format.D(bestOrientation.Qw)); writer.Write(' ');
                    writer.Write(Format.D(bestOrientation.Qx)); writer.Write(' ');
                    writer.Write(Format.D(bestOrientation.Qy)); writer.Write(' ');
                    writer.Write(Format.D(bestOrientation.Qz));
                    writer.Write('\n');

                    written++;
                }
            }
        }

        writer.Flush();
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public double Get(int axis) => axis == 0 ? X : (axis == 1 ? Y : Z);
    }

    private readonly struct Mat3
    {
        public readonly double M00, M01, M02;
        public readonly double M10, M11, M12;
        public readonly double M20, M21, M22;

        public Mat3(
            double m00, double m01, double m02,
            double m10, double m11, double m12,
            double m20, double m21, double m22)
        {
            M00 = m00; M01 = m01; M02 = m02;
            M10 = m10; M11 = m11; M12 = m12;
            M20 = m20; M21 = m21; M22 = m22;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public Vec3 Transform(in Vec3 v)
        {
            double x = M00 * v.X + M01 * v.Y + M02 * v.Z;
            double y = M10 * v.X + M11 * v.Y + M12 * v.Z;
            double z = M20 * v.X + M21 * v.Y + M22 * v.Z;
            return new Vec3(x, y, z);
        }

        public double Determinant()
        {
            return
                M00 * (M11 * M22 - M12 * M21)
              - M01 * (M10 * M22 - M12 * M20)
              + M02 * (M10 * M21 - M11 * M20);
        }

        public Mat3 Transpose()
        {
            return new Mat3(
                M00, M10, M20,
                M01, M11, M21,
                M02, M12, M22);
        }
    }

    private readonly struct Aabb
    {
        public readonly double MinX, MinY, MinZ;
        public readonly double MaxX, MaxY, MaxZ;
        public Aabb(double minX, double minY, double minZ, double maxX, double maxY, double maxZ)
        {
            MinX = minX; MinY = minY; MinZ = minZ;
            MaxX = maxX; MaxY = maxY; MaxZ = maxZ;
        }

        public double Dx => MaxX - MinX;
        public double Dy => MaxY - MinY;
        public double Dz => MaxZ - MinZ;
    }

    private readonly struct Orientation
    {
        public readonly Mat3 R;
        public readonly double Qw, Qx, Qy, Qz;
        public readonly bool IsValid;

        public Orientation(Mat3 r, double qw, double qx, double qy, double qz, bool valid = true)
        {
            R = r;
            Qw = qw; Qx = qx; Qy = qy; Qz = qz;
            IsValid = valid;
        }
    }

    private static class Geometry
    {
        public static Aabb ComputeAabb(Vec3[] verts, Mat3 r)
        {
            double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

            for (int i = 0; i < verts.Length; i++)
            {
                Vec3 p = r.Transform(verts[i]);
                if (p.X < minX) minX = p.X;
                if (p.Y < minY) minY = p.Y;
                if (p.Z < minZ) minZ = p.Z;
                if (p.X > maxX) maxX = p.X;
                if (p.Y > maxY) maxY = p.Y;
                if (p.Z > maxZ) maxZ = p.Z;
            }

            return new Aabb(minX, minY, minZ, maxX, maxY, maxZ);
        }
    }

    private static class Packing
    {
        public static long DecideOutputCap(double w, double h, double d, int vCount)
        {
            // Output size can dominate runtime on extreme instances. Keep bounded and scale slightly with problem size.
            // Conservative default: ~200k placements.
            double volume = w * h * d;
            long cap = volume <= 1e6 ? 1_000_000L : (volume <= 1e7 ? 400_000L : 200_000L);

            // If the polyhedron is extremely complex, avoid huge output even more.
            if (vCount > 200_000) cap = Math.Min(cap, 100_000L);
            return cap;
        }

        public static long CountFitTight(double container, double size)
        {
            // Number of boxes of length `size` that fit in a segment of length `container`, allowing touching.
            if (!(size > 0.0) || !(container > 0.0)) return 0;
            if (container < size) return 0;

            double ratio = container / size;
            if (double.IsInfinity(ratio) || double.IsNaN(ratio)) return 0;

            // Add a tiny epsilon to compensate for cases like ratio=9.999999999999998.
            long n = (long)Math.Floor(ratio + 1e-12);
            if (n < 1) n = 1;

            // Validate and clamp down if rounding pushed us too far.
            double tol = Math.Max(1e-12 * container, 1e-12 * size);
            while (n > 0 && n * size > container + tol) n--;
            return n;
        }

        public static long SaturatingMul(long a, long b)
        {
            if (a <= 0 || b <= 0) return 0;
            if (a > long.MaxValue / b) return long.MaxValue;
            return a * b;
        }
    }

    private static class OrientationLibrary
    {
        public static IEnumerable<Orientation> AxisAligned24()
        {
            // Proper rotations with matrix entries in {-1,0,1}. 24 total.
            int[][] perms =
            {
                new[] { 0, 1, 2 },
                new[] { 0, 2, 1 },
                new[] { 1, 0, 2 },
                new[] { 1, 2, 0 },
                new[] { 2, 0, 1 },
                new[] { 2, 1, 0 },
            };

            foreach (var perm in perms)
            {
                int parity = PermParity(perm); // +1 even, -1 odd

                for (int sx = -1; sx <= 1; sx += 2)
                for (int sy = -1; sy <= 1; sy += 2)
                for (int sz = -1; sz <= 1; sz += 2)
                {
                    if (sx * sy * sz != parity) continue; // enforce det +1

                    // Row i selects original axis perm[i] with sign s[i]
                    Mat3 r = new Mat3(
                        perm[0] == 0 ? sx : 0, perm[0] == 1 ? sx : 0, perm[0] == 2 ? sx : 0,
                        perm[1] == 0 ? sy : 0, perm[1] == 1 ? sy : 0, perm[1] == 2 ? sy : 0,
                        perm[2] == 0 ? sz : 0, perm[2] == 1 ? sz : 0, perm[2] == 2 ? sz : 0
                    );

                    (double qw, double qx, double qy, double qz) = MatrixToQuaternion(r);
                    yield return new Orientation(r, qw, qx, qy, qz);
                }
            }
        }

        public static Orientation TryPcaOrientation(Vec3[] verts)
        {
            // Compute covariance of vertices (mean-centered).
            int n = verts.Length;
            if (n < 3) return default;

            double mx = 0, my = 0, mz = 0;
            for (int i = 0; i < n; i++)
            {
                mx += verts[i].X;
                my += verts[i].Y;
                mz += verts[i].Z;
            }
            mx /= n; my /= n; mz /= n;

            double cxx = 0, cxy = 0, cxz = 0, cyy = 0, cyz = 0, czz = 0;
            for (int i = 0; i < n; i++)
            {
                double x = verts[i].X - mx;
                double y = verts[i].Y - my;
                double z = verts[i].Z - mz;
                cxx += x * x;
                cxy += x * y;
                cxz += x * z;
                cyy += y * y;
                cyz += y * z;
                czz += z * z;
            }
            double inv = 1.0 / n;
            cxx *= inv; cxy *= inv; cxz *= inv;
            cyy *= inv; cyz *= inv;
            czz *= inv;

            Mat3 cov = new Mat3(
                cxx, cxy, cxz,
                cxy, cyy, cyz,
                cxz, cyz, czz
            );

            // Jacobi eigen-decomposition for symmetric 3x3.
            if (!Jacobi3x3(cov, out Mat3 eigenVecs))
                return default;

            // Ensure right-handed basis; flip one axis if needed.
            if (eigenVecs.Determinant() < 0)
            {
                eigenVecs = new Mat3(
                    -eigenVecs.M00, -eigenVecs.M01, -eigenVecs.M02,
                    eigenVecs.M10, eigenVecs.M11, eigenVecs.M12,
                    eigenVecs.M20, eigenVecs.M21, eigenVecs.M22
                );
            }

            // Rotate into PCA frame: R = eigenVecs^T
            Mat3 r = eigenVecs.Transpose();
            (double qw, double qx, double qy, double qz) = MatrixToQuaternion(r);
            return new Orientation(r, qw, qx, qy, qz);
        }

        public static IEnumerable<Orientation> RandomSamples(Vec3[] verts, double w, double h, double d, int count)
        {
            ulong seed = 1469598103934665603UL; // FNV offset
            seed ^= (ulong)verts.Length * 1099511628211UL;
            if (verts.Length > 0)
            {
                seed ^= HashDouble(verts[0].X);
                seed ^= HashDouble(verts[verts.Length - 1].Y);
                seed ^= HashDouble(verts[verts.Length / 2].Z);
            }
            seed ^= HashDouble(w);
            seed ^= HashDouble(h);
            seed ^= HashDouble(d);

            var rng = new XorShift64Star(seed);
            for (int i = 0; i < count; i++)
            {
                Quat q = Quat.RandomUnit(ref rng);
                Mat3 r = q.ToMatrix();
                yield return new Orientation(r, q.W, q.X, q.Y, q.Z);
            }
        }

        private static int PermParity(int[] perm)
        {
            int inv = 0;
            for (int i = 0; i < 3; i++)
                for (int j = i + 1; j < 3; j++)
                    if (perm[i] > perm[j]) inv++;
            return (inv & 1) == 0 ? 1 : -1;
        }

        private static ulong HashDouble(double x)
        {
            ulong bits = (ulong)BitConverter.DoubleToInt64Bits(x);
            // Mix bits (splitmix-like).
            bits += 0x9E3779B97F4A7C15UL;
            bits = (bits ^ (bits >> 30)) * 0xBF58476D1CE4E5B9UL;
            bits = (bits ^ (bits >> 27)) * 0x94D049BB133111EBUL;
            return bits ^ (bits >> 31);
        }

        private static (double qw, double qx, double qy, double qz) MatrixToQuaternion(Mat3 m)
        {
            // Robust conversion (trace-based).
            double trace = m.M00 + m.M11 + m.M22;
            double qw, qx, qy, qz;
            if (trace > 0.0)
            {
                double s = Math.Sqrt(trace + 1.0) * 2.0;
                qw = 0.25 * s;
                qx = (m.M21 - m.M12) / s;
                qy = (m.M02 - m.M20) / s;
                qz = (m.M10 - m.M01) / s;
            }
            else if (m.M00 > m.M11 && m.M00 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
                qw = (m.M21 - m.M12) / s;
                qx = 0.25 * s;
                qy = (m.M01 + m.M10) / s;
                qz = (m.M02 + m.M20) / s;
            }
            else if (m.M11 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
                qw = (m.M02 - m.M20) / s;
                qx = (m.M01 + m.M10) / s;
                qy = 0.25 * s;
                qz = (m.M12 + m.M21) / s;
            }
            else
            {
                double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
                qw = (m.M10 - m.M01) / s;
                qx = (m.M02 + m.M20) / s;
                qy = (m.M12 + m.M21) / s;
                qz = 0.25 * s;
            }

            // Normalize to be safe.
            double norm = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
            if (norm <= 0.0) return (1.0, 0.0, 0.0, 0.0);
            double inv = 1.0 / norm;
            qw *= inv; qx *= inv; qy *= inv; qz *= inv;
            return (qw, qx, qy, qz);
        }

        private static bool Jacobi3x3(Mat3 a, out Mat3 v)
        {
            // Diagonalize symmetric matrix a using Jacobi rotations.
            // v accumulates eigenvectors as columns (start as identity).
            double a00 = a.M00, a01 = a.M01, a02 = a.M02;
            double a11 = a.M11, a12 = a.M12;
            double a22 = a.M22;

            double v00 = 1, v01 = 0, v02 = 0;
            double v10 = 0, v11 = 1, v12 = 0;
            double v20 = 0, v21 = 0, v22 = 1;

            const int maxIter = 32;
            for (int iter = 0; iter < maxIter; iter++)
            {
                // Find largest off-diagonal element (absolute).
                double p01 = Math.Abs(a01);
                double p02 = Math.Abs(a02);
                double p12 = Math.Abs(a12);

                int p = 0, q = 1;
                double apq = p01;
                if (p02 > apq) { apq = p02; p = 0; q = 2; }
                if (p12 > apq) { apq = p12; p = 1; q = 2; }

                if (apq < 1e-12)
                    break;

                // Get app, aqq, apq for selected p,q.
                double app, aqq, apqVal;
                if (p == 0 && q == 1) { app = a00; aqq = a11; apqVal = a01; }
                else if (p == 0 && q == 2) { app = a00; aqq = a22; apqVal = a02; }
                else { app = a11; aqq = a22; apqVal = a12; }

                double phi = 0.5 * Math.Atan2(2.0 * apqVal, (aqq - app));
                double c = Math.Cos(phi);
                double s = Math.Sin(phi);

                // Apply rotation to A (symmetric update).
                if (p == 0 && q == 1)
                {
                    // Update rows/cols 0 and 1.
                    double a00n = c * c * a00 - 2.0 * s * c * a01 + s * s * a11;
                    double a11n = s * s * a00 + 2.0 * s * c * a01 + c * c * a11;
                    double a01n = 0.0;
                    double a02n = c * a02 - s * a12;
                    double a12n = s * a02 + c * a12;
                    a00 = a00n; a11 = a11n; a01 = a01n; a02 = a02n; a12 = a12n;
                }
                else if (p == 0 && q == 2)
                {
                    double a00n = c * c * a00 - 2.0 * s * c * a02 + s * s * a22;
                    double a22n = s * s * a00 + 2.0 * s * c * a02 + c * c * a22;
                    double a02n = 0.0;
                    double a01n = c * a01 - s * a12;
                    double a12n = s * a01 + c * a12;
                    a00 = a00n; a22 = a22n; a02 = a02n; a01 = a01n; a12 = a12n;
                }
                else // p==1 && q==2
                {
                    double a11n = c * c * a11 - 2.0 * s * c * a12 + s * s * a22;
                    double a22n = s * s * a11 + 2.0 * s * c * a12 + c * c * a22;
                    double a12n = 0.0;
                    double a01n = c * a01 - s * a02;
                    double a02n = s * a01 + c * a02;
                    a11 = a11n; a22 = a22n; a12 = a12n; a01 = a01n; a02 = a02n;
                }

                // Accumulate eigenvectors V = V * G (rotate columns p,q).
                if (p == 0 && q == 1)
                {
                    RotateCols(ref v00, ref v01, c, s);
                    RotateCols(ref v10, ref v11, c, s);
                    RotateCols(ref v20, ref v21, c, s);
                }
                else if (p == 0 && q == 2)
                {
                    RotateCols(ref v00, ref v02, c, s);
                    RotateCols(ref v10, ref v12, c, s);
                    RotateCols(ref v20, ref v22, c, s);
                }
                else
                {
                    RotateCols(ref v01, ref v02, c, s);
                    RotateCols(ref v11, ref v12, c, s);
                    RotateCols(ref v21, ref v22, c, s);
                }
            }

            // Pack eigenvector columns into Mat3.
            v = new Mat3(
                v00, v01, v02,
                v10, v11, v12,
                v20, v21, v22
            );

            // Quick sanity: must be non-degenerate.
            if (double.IsNaN(v.M00) || double.IsNaN(v.M11) || double.IsNaN(v.M22))
                return false;

            return true;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static void RotateCols(ref double a, ref double b, double c, double s)
        {
            double an = c * a - s * b;
            double bn = s * a + c * b;
            a = an; b = bn;
        }

        private struct XorShift64Star
        {
            private ulong _x;
            public XorShift64Star(ulong seed) { _x = seed != 0 ? seed : 0x9E3779B97F4A7C15UL; }

            [MethodImpl(MethodImplOptions.AggressiveInlining)]
            public ulong NextU64()
            {
                ulong x = _x;
                x ^= x >> 12;
                x ^= x << 25;
                x ^= x >> 27;
                _x = x;
                return x * 2685821657736338717UL;
            }

            [MethodImpl(MethodImplOptions.AggressiveInlining)]
            public double NextDouble01()
            {
                // Use top 53 bits for IEEE-754 double fraction.
                ulong u = NextU64() >> 11;
                return u * (1.0 / 9007199254740992.0); // 2^53
            }
        }

        private readonly struct Quat
        {
            public readonly double W, X, Y, Z;
            public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }

            public Mat3 ToMatrix()
            {
                double w = W, x = X, y = Y, z = Z;
                double xx = x * x, yy = y * y, zz = z * z;
                double xy = x * y, xz = x * z, yz = y * z;
                double wx = w * x, wy = w * y, wz = w * z;

                return new Mat3(
                    1.0 - 2.0 * (yy + zz), 2.0 * (xy - wz), 2.0 * (xz + wy),
                    2.0 * (xy + wz), 1.0 - 2.0 * (xx + zz), 2.0 * (yz - wx),
                    2.0 * (xz - wy), 2.0 * (yz + wx), 1.0 - 2.0 * (xx + yy)
                );
            }

            public static Quat RandomUnit(ref XorShift64Star rng)
            {
                // Uniform random quaternion (LaValle / Graphics Gems).
                double u1 = rng.NextDouble01();
                double u2 = rng.NextDouble01();
                double u3 = rng.NextDouble01();
                double s1 = Math.Sqrt(1.0 - u1);
                double s2 = Math.Sqrt(u1);
                double t1 = 2.0 * Math.PI * u2;
                double t2 = 2.0 * Math.PI * u3;

                double x = s1 * Math.Sin(t1);
                double y = s1 * Math.Cos(t1);
                double z = s2 * Math.Sin(t2);
                double w = s2 * Math.Cos(t2);
                return new Quat(w, x, y, z);
            }
        }
    }

    private static class Format
    {
        private static readonly NumberFormatInfo Nfi = CultureInfo.InvariantCulture.NumberFormat;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public static string D(double x)
        {
            if (x == 0.0) return "0";
            // Compact, enough precision for placement, avoids "R" performance hit.
            return x.ToString("G17", Nfi);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;
        private bool _atStart = true;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int SkipBlanks()
        {
            int c;
            while (true)
            {
                c = ReadByte();
                if (c < 0) return -1;
                // Skip ASCII whitespace and a potential UTF-8 BOM (EF BB BF) at the very start of the stream.
                if (_atStart)
                {
                    if (c == 0xEF || c == 0xBB || c == 0xBF) continue;
                    if (c <= 32) continue;
                    _atStart = false;
                    return c;
                }
                if (c > 32) return c;
            }
        }

        public bool TryNextInt(out int value)
        {
            int c = SkipBlanks();
            if (c < 0) { value = 0; return false; }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int v = 0;
            while (c > 32 && c >= '0' && c <= '9')
            {
                v = v * 10 + (c - '0');
                c = ReadByte();
            }
            value = v * sign;
            return true;
        }

        public int NextInt()
        {
            if (!TryNextInt(out int v)) return 0;
            return v;
        }

        public double NextDouble()
        {
            int c = SkipBlanks();
            if (c < 0) return 0.0;

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            else if (c == '+')
            {
                c = ReadByte();
            }

            double intPart = 0.0;
            while (c >= '0' && c <= '9')
            {
                intPart = intPart * 10.0 + (c - '0');
                c = ReadByte();
            }

            double fracPart = 0.0;
            double fracScale = 1.0;
            if (c == '.')
            {
                c = ReadByte();
                while (c >= '0' && c <= '9')
                {
                    fracPart = fracPart * 10.0 + (c - '0');
                    fracScale *= 10.0;
                    c = ReadByte();
                }
            }

            double val = intPart + (fracPart / fracScale);

            // Optional exponent
            if (c == 'e' || c == 'E')
            {
                c = ReadByte();
                int expSign = 1;
                if (c == '-')
                {
                    expSign = -1;
                    c = ReadByte();
                }
                else if (c == '+')
                {
                    c = ReadByte();
                }

                int exp = 0;
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = ReadByte();
                }

                if (exp != 0)
                    val *= Math.Pow(10.0, expSign * exp);
            }

            return val * sign;
        }
    }
