; SPIR-V
; Version: 1.0

OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %outColor "outColor"
OpName %ubo "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_15 = OpConstant %float 0.15
%float_0_3 = OpConstant %float 0.3
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_1 = OpConstant %float 1.0

%int_3 = OpConstant %int 3

%skyColor = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05
%vec3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %fn_void
%entry = OpLabel
%worldPos = OpLoad %v3float %in_worldPos
%normal_in = OpLoad %v3float %in_normal
%N = OpExtInst %v3float %glsl Normalize %normal_in
%Ny = OpCompositeExtract %float %N 1
%Ny_half = OpFMul %float %Ny %float_0_5
%t = OpFAdd %float %Ny_half %float_0_5
%delta = OpFSub %v3float %skyColor %groundColor
%deltaScaled = OpVectorTimesScalar %v3float %delta %t
%hemi = OpFAdd %v3float %groundColor %deltaScaled

%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPos_ptr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%L_un = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %glsl Normalize %L_un
%ndotl = OpDot %float %N %L
%d = OpExtInst %float %glsl FMax %ndotl %float_0
%diffuse = OpFMul %float %d %float_0_3
%diffuseVec = OpVectorTimesScalar %v3float %vec3_1 %diffuse
%rgb = OpFAdd %v3float %hemi %diffuseVec

%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%out = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
