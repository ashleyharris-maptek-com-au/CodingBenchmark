'use strict';

const fs = require('fs');

let input = fs.readFileSync(0, 'utf8');
if (input.length > 0 && input.charCodeAt(0) === 0xfeff) input = input.slice(1);
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robotsIn = Array.isArray(data.robots) ? data.robots : [];
const jobsIn = Array.isArray(data.jobs) ? data.jobs : [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const N = W > 0 && H > 0 ? W * H : 0;

function toInt(v) {
  const n = Number(v);
  return Number.isFinite(n) ? Math.trunc(n) : 0;
}

function cmpId(a, b) {
  if (a === b) return 0;
  if (typeof a === 'number' && typeof b === 'number') return a - b;
  const sa = String(a);
  const sb = String(b);
  return sa < sb ? -1 : sa > sb ? 1 : 0;
}

// Build passability map
const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  const base = y * W;
  for (let x = 0; x < W; x++) open[base + x] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
}

// Normalize robots
const robots = new Array(robotsIn.length);
let maxRobotId = -1;
let minRobotId = 0x7fffffff;
const idSeen = new Set();
let idsAreContiguous = true;

for (let i = 0; i < robotsIn.length; i++) {
  const r = robotsIn[i] || {};
  const id = toInt(r.id);
  const x = toInt(r.x);
  const y = toInt(r.y);
  let idx = -1;
  if (x >= 0 && x < W && y >= 0 && y < H) idx = y * W + x;

  robots[i] = { id, x, y, idx };

  if (id < minRobotId) minRobotId = id;
  if (id > maxRobotId) maxRobotId = id;
  if (idSeen.has(id)) idsAreContiguous = false;
  idSeen.add(id);
}

// Determine output assignments length
let assignmentsLen;
if (robots.length === 0) {
  assignmentsLen = 0;
} else if (minRobotId === 0 && maxRobotId === robots.length - 1 && idsAreContiguous) {
  assignmentsLen = robots.length;
} else if (maxRobotId >= 0 && maxRobotId <= 200000) {
  assignmentsLen = maxRobotId + 1;
} else {
  // Fallback: treat sorted-by-id order as index (best-effort if ids are unexpectedly huge).
  assignmentsLen = robots.length;
  for (let i = 0; i < robots.length; i++) robots[i].id = i;
  maxRobotId = robots.length - 1;
  minRobotId = 0;
  idsAreContiguous = true;
}

// Precompute nearest robot (by grid distance) for every cell via multi-source BFS
const dist = new Int32Array(N);
dist.fill(-1);
const owner = new Int32Array(N);
owner.fill(0x7fffffff);

if (N > 0 && robots.length > 0) {
  // Multi-source BFS to compute distances, then a DP-style pass along distance layers
  // to pick the smallest robot id among all shortest paths.
  robots.sort((a, b) => a.id - b.id);

  const queue = new Int32Array(N);
  const order = new Int32Array(N);
  let qh = 0;
  let qt = 0;
  let orderLen = 0;

  // Seed all robot start cells (distance only), and store per-cell smallest robot id.
  for (let i = 0; i < robots.length; i++) {
    const r = robots[i];
    const idx = r.idx;
    if (idx < 0 || open[idx] === 0) continue;
    if (dist[idx] === -1) {
      dist[idx] = 0;
      queue[qt++] = idx;
    }
    if (r.id < owner[idx]) owner[idx] = r.id;
  }

  while (qh < qt) {
    const cur = queue[qh++];
    order[orderLen++] = cur;
    const nd = dist[cur] + 1;

    const x = cur % W;
    let n;

    if (x > 0) {
      n = cur - 1;
      if (open[n] && dist[n] === -1) {
        dist[n] = nd;
        queue[qt++] = n;
      }
    }

    if (x + 1 < W) {
      n = cur + 1;
      if (open[n] && dist[n] === -1) {
        dist[n] = nd;
        queue[qt++] = n;
      }
    }

    if (cur >= W) {
      n = cur - W;
      if (open[n] && dist[n] === -1) {
        dist[n] = nd;
        queue[qt++] = n;
      }
    }

    if (cur < N - W) {
      n = cur + W;
      if (open[n] && dist[n] === -1) {
        dist[n] = nd;
        queue[qt++] = n;
      }
    }
  }

  // Propagate smallest owner along shortest-distance edges: dist[v] == dist[u] + 1.
  for (let oi = 0; oi < orderLen; oi++) {
    const cur = order[oi];
    const curOwner = owner[cur];
    const nd = dist[cur] + 1;
    const x = cur % W;
    let n;

    if (x > 0) {
      n = cur - 1;
      if (open[n] && dist[n] === nd && curOwner < owner[n]) owner[n] = curOwner;
    }
    if (x + 1 < W) {
      n = cur + 1;
      if (open[n] && dist[n] === nd && curOwner < owner[n]) owner[n] = curOwner;
    }
    if (cur >= W) {
      n = cur - W;
      if (open[n] && dist[n] === nd && curOwner < owner[n]) owner[n] = curOwner;
    }
    if (cur < N - W) {
      n = cur + W;
      if (open[n] && dist[n] === nd && curOwner < owner[n]) owner[n] = curOwner;
    }
  }
}

// Helper for fallback assignment when unreachable or out-of-bounds
function assignByManhattan(jobX, jobY) {
  let bestId = 0;
  let bestD = 0x7fffffff;
  for (let i = 0; i < robots.length; i++) {
    const r = robots[i];
    const d = Math.abs(r.x - jobX) + Math.abs(r.y - jobY);
    if (d < bestD || (d === bestD && r.id < bestId)) {
      bestD = d;
      bestId = r.id;
    }
  }
  return [bestId, bestD];
}

// Spatial sort key (Morton when safe, else row-major)
const useMorton = W > 0 && H > 0 && W <= 65535 && H <= 65535;
function splitBy1(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n;
}
function morton2D(x, y) {
  return (splitBy1(x) | (splitBy1(y) << 1)) >>> 0;
}
function spatialKey(x, y) {
  return useMorton ? morton2D(x, y) : y * W + x;
}

// Build assignments
const jobsByRobot = new Array(assignmentsLen);
for (let i = 0; i < assignmentsLen; i++) jobsByRobot[i] = [];

for (let i = 0; i < jobsIn.length; i++) {
  const j = jobsIn[i] || {};
  const jobId = j.id;
  const x = toInt(j.x);
  const y = toInt(j.y);
  const priority = Number.isFinite(Number(j.priority)) ? Number(j.priority) : 0;

  let rid = 0;
  let dMetric = 0x7fffffff;

  if (robots.length > 0 && x >= 0 && x < W && y >= 0 && y < H) {
    const idx = y * W + x;
    if (idx >= 0 && idx < N && open[idx] && dist[idx] !== -1) {
      rid = owner[idx];
      dMetric = dist[idx];
    } else {
      const res = assignByManhattan(x, y);
      rid = res[0];
      dMetric = res[1];
    }
  } else if (robots.length > 0) {
    const res = assignByManhattan(x, y);
    rid = res[0];
    dMetric = res[1];
  } else {
    // No robots: can't validly assign; keep empty assignments.
    continue;
  }

  if (rid < 0 || rid >= assignmentsLen) {
    // Shouldn't happen unless ids are malformed; clamp deterministically.
    rid = 0;
  }

  jobsByRobot[rid].push({
    id: jobId,
    x,
    y,
    priority,
    dist: dMetric,
    key: spatialKey(x, y),
  });
}

const assignments = new Array(assignmentsLen);
for (let rid = 0; rid < assignmentsLen; rid++) {
  const list = jobsByRobot[rid];
  list.sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    if (a.dist !== b.dist) return a.dist - b.dist;
    if (a.key !== b.key) return a.key < b.key ? -1 : 1;
    return cmpId(a.id, b.id);
  });
  const out = new Array(list.length);
  for (let i = 0; i < list.length; i++) out[i] = list[i].id;
  assignments[rid] = out;
}

process.stdout.write(JSON.stringify({ assignments }));
