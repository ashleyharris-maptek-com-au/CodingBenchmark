namespace chessbot {

using U64 = std::uint64_t;

static constexpr int WHITE = 0;
static constexpr int BLACK = 1;

enum PieceType : int {
  PAWN = 0,
  KNIGHT = 1,
  BISHOP = 2,
  ROOK = 3,
  QUEEN = 4,
  KING = 5
};

static constexpr int PIECE_NB = 6;
static constexpr int MAX_PLY = 128;

static constexpr int INF = 32000;
static constexpr int MATE_SCORE = 32000;
static constexpr int MATE_THRESHOLD = 30000;

static constexpr int CASTLE_WK = 1 << 0;
static constexpr int CASTLE_WQ = 1 << 1;
static constexpr int CASTLE_BK = 1 << 2;
static constexpr int CASTLE_BQ = 1 << 3;

static constexpr U64 BB_EMPTY = 0ULL;

static constexpr U64 FILE_A = 0x0101010101010101ULL;
static constexpr U64 FILE_H = 0x8080808080808080ULL;
static constexpr U64 RANK_1 = 0x00000000000000FFULL;
static constexpr U64 RANK_4 = 0x00000000FF000000ULL;
static constexpr U64 RANK_5 = 0x000000FF00000000ULL;
static constexpr U64 RANK_8 = 0xFF00000000000000ULL;

inline int make_piece(int color, int type) { return color * 6 + type + 1; }
inline int piece_color(int p) { return (p - 1) / 6; }
inline int piece_type(int p) { return (p - 1) % 6; }

inline int pop_lsb(U64& bb) {
  int sq = static_cast<int>(std::countr_zero(bb));
  bb &= bb - 1;
  return sq;
}

inline int popcount(U64 bb) { return static_cast<int>(std::popcount(bb)); }

static std::array<U64, 64> KnightAtt{};
static std::array<U64, 64> KingAtt{};
static std::array<std::array<U64, 64>, 2> PawnAtt{};

struct Zobrist {
  std::array<std::array<std::array<U64, 64>, PIECE_NB>, 2> piece{};
  std::array<U64, 16> castle{};
  std::array<U64, 8> epFile{};
  U64 side = 0;
};

static Zobrist Z{};

static U64 splitmix64(U64& x) {
  U64 z = (x += 0x9e3779b97f4a7c15ULL);
  z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
  z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
  return z ^ (z >> 31);
}

static void init_tables() {
  for (int sq = 0; sq < 64; ++sq) {
    int r = sq / 8, f = sq % 8;

    U64 k = 0;
    for (int dr = -1; dr <= 1; ++dr) {
      for (int df = -1; df <= 1; ++df) {
        if (!dr && !df) continue;
        int rr = r + dr, ff = f + df;
        if (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) k |= 1ULL << (rr * 8 + ff);
      }
    }
    KingAtt[sq] = k;

    U64 n = 0;
    constexpr int kdr[8] = {+2, +2, +1, +1, -1, -1, -2, -2};
    constexpr int kdf[8] = {+1, -1, +2, -2, +2, -2, +1, -1};
    for (int i = 0; i < 8; ++i) {
      int rr = r + kdr[i], ff = f + kdf[i];
      if (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) n |= 1ULL << (rr * 8 + ff);
    }
    KnightAtt[sq] = n;

    U64 pw = 0, pb = 0;
    if (r + 1 < 8) {
      if (f - 1 >= 0) pw |= 1ULL << ((r + 1) * 8 + (f - 1));
      if (f + 1 < 8) pw |= 1ULL << ((r + 1) * 8 + (f + 1));
    }
    if (r - 1 >= 0) {
      if (f - 1 >= 0) pb |= 1ULL << ((r - 1) * 8 + (f - 1));
      if (f + 1 < 8) pb |= 1ULL << ((r - 1) * 8 + (f + 1));
    }
    PawnAtt[WHITE][sq] = pw;
    PawnAtt[BLACK][sq] = pb;
  }

  U64 seed = 0x243F6A8885A308D3ULL;
  for (int c = 0; c < 2; ++c)
    for (int pt = 0; pt < PIECE_NB; ++pt)
      for (int sq = 0; sq < 64; ++sq) Z.piece[c][pt][sq] = splitmix64(seed);
  for (int i = 0; i < 16; ++i) Z.castle[i] = splitmix64(seed);
  for (int f = 0; f < 8; ++f) Z.epFile[f] = splitmix64(seed);
  Z.side = splitmix64(seed);
}

inline U64 rook_attacks(int sq, U64 occ) {
  U64 attacks = 0;
  int r = sq / 8, f = sq % 8;
  for (int rr = r + 1; rr < 8; ++rr) {
    int s = rr * 8 + f;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  for (int rr = r - 1; rr >= 0; --rr) {
    int s = rr * 8 + f;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  for (int ff = f + 1; ff < 8; ++ff) {
    int s = r * 8 + ff;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  for (int ff = f - 1; ff >= 0; --ff) {
    int s = r * 8 + ff;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  return attacks;
}

inline U64 bishop_attacks(int sq, U64 occ) {
  U64 attacks = 0;
  int r = sq / 8, f = sq % 8;
  for (int rr = r + 1, ff = f + 1; rr < 8 && ff < 8; ++rr, ++ff) {
    int s = rr * 8 + ff;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  for (int rr = r + 1, ff = f - 1; rr < 8 && ff >= 0; ++rr, --ff) {
    int s = rr * 8 + ff;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  for (int rr = r - 1, ff = f + 1; rr >= 0 && ff < 8; --rr, ++ff) {
    int s = rr * 8 + ff;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  for (int rr = r - 1, ff = f - 1; rr >= 0 && ff >= 0; --rr, --ff) {
    int s = rr * 8 + ff;
    attacks |= 1ULL << s;
    if (occ & (1ULL << s)) break;
  }
  return attacks;
}

struct Move {
  std::uint8_t from = 0;
  std::uint8_t to = 0;
  std::uint8_t promo = 0; // PieceType, or 0 for none
  std::uint8_t flags = 0;
};

static constexpr std::uint8_t MF_CAPTURE = 1 << 0;
static constexpr std::uint8_t MF_ENPASSANT = 1 << 1;
static constexpr std::uint8_t MF_CASTLE = 1 << 2;
static constexpr std::uint8_t MF_DOUBLE = 1 << 3;

inline bool is_capture(const Move& m) { return (m.flags & MF_CAPTURE) != 0; }
inline bool is_ep(const Move& m) { return (m.flags & MF_ENPASSANT) != 0; }
inline bool is_castle(const Move& m) { return (m.flags & MF_CASTLE) != 0; }
inline bool is_promo(const Move& m) { return m.promo != 0; }

inline std::uint32_t encode_move(const Move& m) {
  return static_cast<std::uint32_t>(m.from) | (static_cast<std::uint32_t>(m.to) << 6) |
         (static_cast<std::uint32_t>(m.promo) << 12) | (static_cast<std::uint32_t>(m.flags) << 16);
}

inline Move decode_move(std::uint32_t x) {
  Move m;
  m.from = static_cast<std::uint8_t>(x & 63U);
  m.to = static_cast<std::uint8_t>((x >> 6) & 63U);
  m.promo = static_cast<std::uint8_t>((x >> 12) & 15U);
  m.flags = static_cast<std::uint8_t>((x >> 16) & 255U);
  return m;
}

struct Undo {
  int capturedPiece = 0;
  int capturedSquare = -1;
  int side = WHITE;
  int castling = 0;
  int ep = -1;
  int halfmove = 0;
  int fullmove = 1;
  std::array<int, 2> kingSq{4, 60};
  U64 key = 0;
};

struct Position {
  std::array<int, 64> board{};
  std::array<std::array<U64, PIECE_NB>, 2> pieces{};
  std::array<U64, 2> occ{};
  U64 allOcc = 0;

  int side = WHITE;
  int castling = 0;
  int ep = -1;
  int halfmove = 0;
  int fullmove = 1;
  std::array<int, 2> kingSq{4, 60};

  U64 key = 0;

  void clear() {
    board.fill(0);
    for (int c = 0; c < 2; ++c) pieces[c].fill(0);
    occ.fill(0);
    allOcc = 0;
    side = WHITE;
    castling = 0;
    ep = -1;
    halfmove = 0;
    fullmove = 1;
    kingSq = {4, 60};
    key = 0;
  }

  bool ep_hash_applicable() const {
    if (ep < 0) return false;
    int file = ep & 7;
    if (side == WHITE) {
      if (file > 0) {
        int from = ep - 9;
        if (from >= 0 && board[from] == make_piece(WHITE, PAWN)) return true;
      }
      if (file < 7) {
        int from = ep - 7;
        if (from >= 0 && board[from] == make_piece(WHITE, PAWN)) return true;
      }
    } else {
      if (file > 0) {
        int from = ep + 7;
        if (from < 64 && board[from] == make_piece(BLACK, PAWN)) return true;
      }
      if (file < 7) {
        int from = ep + 9;
        if (from < 64 && board[from] == make_piece(BLACK, PAWN)) return true;
      }
    }
    return false;
  }

  void recompute_bitboards() {
    for (int c = 0; c < 2; ++c) pieces[c].fill(0);
    occ.fill(0);
    allOcc = 0;
    for (int sq = 0; sq < 64; ++sq) {
      int p = board[sq];
      if (!p) continue;
      int c = piece_color(p), t = piece_type(p);
      pieces[c][t] |= 1ULL << sq;
      occ[c] |= 1ULL << sq;
      allOcc |= 1ULL << sq;
      if (t == KING) kingSq[c] = sq;
    }
  }

  void recompute_key() {
    key = 0;
    for (int sq = 0; sq < 64; ++sq) {
      int p = board[sq];
      if (!p) continue;
      int c = piece_color(p), t = piece_type(p);
      key ^= Z.piece[c][t][sq];
    }
    key ^= Z.castle[castling & 15];
    if (side == BLACK) key ^= Z.side;
    if (ep_hash_applicable()) key ^= Z.epFile[ep & 7];
  }

  void add_piece(int sq, int p) {
    board[sq] = p;
    int c = piece_color(p), t = piece_type(p);
    pieces[c][t] |= 1ULL << sq;
    occ[c] |= 1ULL << sq;
    allOcc |= 1ULL << sq;
    if (t == KING) kingSq[c] = sq;
    key ^= Z.piece[c][t][sq];
  }

  void remove_piece(int sq, int p) {
    board[sq] = 0;
    int c = piece_color(p), t = piece_type(p);
    pieces[c][t] &= ~(1ULL << sq);
    occ[c] &= ~(1ULL << sq);
    allOcc &= ~(1ULL << sq);
    key ^= Z.piece[c][t][sq];
  }

  bool square_attacked(int sq, int by) const {
    if (PawnAtt[by ^ 1][sq] & pieces[by][PAWN]) return true;
    if (KnightAtt[sq] & pieces[by][KNIGHT]) return true;
    if (KingAtt[sq] & pieces[by][KING]) return true;

    U64 bAtt = bishop_attacks(sq, allOcc);
    if (bAtt & (pieces[by][BISHOP] | pieces[by][QUEEN])) return true;

    U64 rAtt = rook_attacks(sq, allOcc);
    if (rAtt & (pieces[by][ROOK] | pieces[by][QUEEN])) return true;

    return false;
  }

  bool in_check(int c) const { return square_attacked(kingSq[c], c ^ 1); }

  int generate_moves(std::array<Move, 256>& out, bool capturesOnly) const {
    int n = 0;
    auto push = [&](const Move& m) {
      if (n < static_cast<int>(out.size())) out[static_cast<std::size_t>(n++)] = m;
    };
    int us = side;
    int them = us ^ 1;
    U64 ours = occ[us];
    U64 theirs = occ[them];
    U64 empty = ~allOcc;

    // Pawns
    U64 pawns = pieces[us][PAWN];
    while (pawns) {
      int from = pop_lsb(pawns);
      int r = from / 8;
      int f = from & 7;

      if (us == WHITE) {
        // captures
        if (f > 0) {
          int to = from + 7;
          if (to < 64) {
            if ((theirs & (1ULL << to)) || (ep == to)) {
              Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, MF_CAPTURE};
              if (ep == to && !(theirs & (1ULL << to))) m.flags |= MF_ENPASSANT;
              if (to >= 56) {
                for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                  Move mp = m;
                  mp.promo = static_cast<std::uint8_t>(promo);
                  push(mp);
                }
              } else {
                push(m);
              }
            }
          }
        }
        if (f < 7) {
          int to = from + 9;
          if (to < 64) {
            if ((theirs & (1ULL << to)) || (ep == to)) {
              Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, MF_CAPTURE};
              if (ep == to && !(theirs & (1ULL << to))) m.flags |= MF_ENPASSANT;
              if (to >= 56) {
                for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                  Move mp = m;
                  mp.promo = static_cast<std::uint8_t>(promo);
                  push(mp);
                }
              } else {
                push(m);
              }
            }
          }
        }

        if (!capturesOnly) {
          // single push
          int to = from + 8;
          if (to < 64 && (empty & (1ULL << to))) {
            if (to >= 56) {
              for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                push(Move{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to),
                          static_cast<std::uint8_t>(promo), 0});
              }
            } else {
              push(Move{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, 0});
              if (r == 1) {
                int to2 = from + 16;
                if (empty & (1ULL << to2)) {
                  push(Move{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to2), 0, MF_DOUBLE});
                }
              }
            }
          }
        }
      } else {
        // BLACK
        if (f > 0) {
          int to = from - 9;
          if (to >= 0) {
            if ((theirs & (1ULL << to)) || (ep == to)) {
              Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, MF_CAPTURE};
              if (ep == to && !(theirs & (1ULL << to))) m.flags |= MF_ENPASSANT;
              if (to < 8) {
                for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                  Move mp = m;
                  mp.promo = static_cast<std::uint8_t>(promo);
                  push(mp);
                }
              } else {
                push(m);
              }
            }
          }
        }
        if (f < 7) {
          int to = from - 7;
          if (to >= 0) {
            if ((theirs & (1ULL << to)) || (ep == to)) {
              Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, MF_CAPTURE};
              if (ep == to && !(theirs & (1ULL << to))) m.flags |= MF_ENPASSANT;
              if (to < 8) {
                for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                  Move mp = m;
                  mp.promo = static_cast<std::uint8_t>(promo);
                  push(mp);
                }
              } else {
                push(m);
              }
            }
          }
        }

        if (!capturesOnly) {
          int to = from - 8;
          if (to >= 0 && (empty & (1ULL << to))) {
            if (to < 8) {
              for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                push(Move{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to),
                          static_cast<std::uint8_t>(promo), 0});
              }
            } else {
              push(Move{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, 0});
              if (r == 6) {
                int to2 = from - 16;
                if (empty & (1ULL << to2)) {
                  push(Move{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to2), 0, MF_DOUBLE});
                }
              }
            }
          }
        }
      }
    }

    auto gen_from_bb = [&](int pt, auto attackFn, bool sliding) {
      U64 bb = pieces[us][pt];
      while (bb) {
        int from = pop_lsb(bb);
        U64 att = sliding ? attackFn(from, allOcc) : attackFn(from);
        att &= ~ours;
        if (capturesOnly) att &= theirs;
        U64 targets = att;
        while (targets) {
          int to = pop_lsb(targets);
          Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, 0};
          if (theirs & (1ULL << to)) m.flags |= MF_CAPTURE;
          push(m);
        }
      }
    };

    // Knights
    gen_from_bb(KNIGHT, [&](int sq) { return KnightAtt[sq]; }, false);
    // Bishops
    gen_from_bb(BISHOP, [&](int sq, U64 o) { return bishop_attacks(sq, o); }, true);
    // Rooks
    gen_from_bb(ROOK, [&](int sq, U64 o) { return rook_attacks(sq, o); }, true);
    // Queens
    {
      U64 bb = pieces[us][QUEEN];
      while (bb) {
        int from = pop_lsb(bb);
        U64 att = (rook_attacks(from, allOcc) | bishop_attacks(from, allOcc)) & ~ours;
        if (capturesOnly) att &= theirs;
        U64 targets = att;
        while (targets) {
          int to = pop_lsb(targets);
          Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, 0};
          if (theirs & (1ULL << to)) m.flags |= MF_CAPTURE;
          push(m);
        }
      }
    }
    // King moves
    {
      U64 bb = pieces[us][KING];
      if (bb) {
        int from = static_cast<int>(std::countr_zero(bb));
        U64 att = KingAtt[from] & ~ours;
        if (capturesOnly) att &= theirs;
        U64 targets = att;
        while (targets) {
          int to = pop_lsb(targets);
          Move m{static_cast<std::uint8_t>(from), static_cast<std::uint8_t>(to), 0, 0};
          if (theirs & (1ULL << to)) m.flags |= MF_CAPTURE;
          push(m);
        }

        if (!capturesOnly) {
          // Castling (standard chess only)
          if (us == WHITE) {
            if ((castling & CASTLE_WK) && board[4] == make_piece(WHITE, KING) && board[7] == make_piece(WHITE, ROOK) &&
              board[5] == 0 && board[6] == 0) {
              if (!square_attacked(4, them) && !square_attacked(5, them) && !square_attacked(6, them)) {
                push(Move{4, 6, 0, MF_CASTLE});
              }
            }
            if ((castling & CASTLE_WQ) && board[4] == make_piece(WHITE, KING) && board[0] == make_piece(WHITE, ROOK) &&
                board[1] == 0 && board[2] == 0 && board[3] == 0) {
              if (!square_attacked(4, them) && !square_attacked(3, them) && !square_attacked(2, them)) {
                push(Move{4, 2, 0, MF_CASTLE});
              }
            }
          } else {
            if ((castling & CASTLE_BK) && board[60] == make_piece(BLACK, KING) && board[63] == make_piece(BLACK, ROOK) &&
                board[61] == 0 && board[62] == 0) {
              if (!square_attacked(60, them) && !square_attacked(61, them) && !square_attacked(62, them)) {
                push(Move{60, 62, 0, MF_CASTLE});
              }
            }
            if ((castling & CASTLE_BQ) && board[60] == make_piece(BLACK, KING) && board[56] == make_piece(BLACK, ROOK) &&
                board[57] == 0 && board[58] == 0 && board[59] == 0) {
              if (!square_attacked(60, them) && !square_attacked(59, them) && !square_attacked(58, them)) {
                push(Move{60, 58, 0, MF_CASTLE});
              }
            }
          }
        }
      }
    }

    return n;
  }

  bool make_null(Undo& u) {
    u.side = side;
    u.castling = castling;
    u.ep = ep;
    u.halfmove = halfmove;
    u.fullmove = fullmove;
    u.kingSq = kingSq;
    u.key = key;

    if (ep_hash_applicable()) key ^= Z.epFile[ep & 7];
    ep = -1;
    halfmove++;
    side ^= 1;
    key ^= Z.side;
    return true;
  }

  void unmake_null(const Undo& u) {
    side = u.side;
    castling = u.castling;
    ep = u.ep;
    halfmove = u.halfmove;
    fullmove = u.fullmove;
    kingSq = u.kingSq;
    key = u.key;
  }

  bool make_move(const Move& m, Undo& u) {
    u.side = side;
    u.castling = castling;
    u.ep = ep;
    u.halfmove = halfmove;
    u.fullmove = fullmove;
    u.kingSq = kingSq;
    u.key = key;
    u.capturedPiece = 0;
    u.capturedSquare = -1;

    int us = side;
    int them = us ^ 1;

    // Update hash: remove castle and EP components for old position
    key ^= Z.castle[castling & 15];
    if (ep_hash_applicable()) key ^= Z.epFile[ep & 7];

    // Clear EP by default
    ep = -1;

    int from = m.from;
    int to = m.to;
    int moving = board[from];
    if (!moving) {
      side = u.side;
      castling = u.castling;
      ep = u.ep;
      halfmove = u.halfmove;
      fullmove = u.fullmove;
      kingSq = u.kingSq;
      key = u.key;
      return false;
    }
    int movingType = piece_type(moving);

    bool pawnMove = (movingType == PAWN);

    // Capture
    int capturedSq = to;
    int captured = 0;
    if (m.flags & MF_ENPASSANT) {
      capturedSq = (us == WHITE) ? (to - 8) : (to + 8);
      captured = board[capturedSq];
    } else {
      captured = board[to];
    }
    if (captured) {
      u.capturedPiece = captured;
      u.capturedSquare = capturedSq;
      remove_piece(capturedSq, captured);
    }

    // Move piece out of from
    remove_piece(from, moving);

    // Castling rook move
    if (m.flags & MF_CASTLE) {
      if (us == WHITE) {
        if (to == 6) { // O-O
          int rook = board[7];
          if (rook) {
            remove_piece(7, rook);
            add_piece(5, rook);
          }
        } else if (to == 2) { // O-O-O
          int rook = board[0];
          if (rook) {
            remove_piece(0, rook);
            add_piece(3, rook);
          }
        }
      } else {
        if (to == 62) {
          int rook = board[63];
          if (rook) {
            remove_piece(63, rook);
            add_piece(61, rook);
          }
        } else if (to == 58) {
          int rook = board[56];
          if (rook) {
            remove_piece(56, rook);
            add_piece(59, rook);
          }
        }
      }
    }

    // Promotion
    int placed = moving;
    if (m.promo) {
      placed = make_piece(us, static_cast<int>(m.promo));
    }
    add_piece(to, placed);

    // Update castling rights
    auto clear_castle = [&](int mask) { castling &= ~mask; };
    if (movingType == KING) {
      if (us == WHITE) clear_castle(CASTLE_WK | CASTLE_WQ);
      else clear_castle(CASTLE_BK | CASTLE_BQ);
    }
    if (movingType == ROOK) {
      if (from == 0) clear_castle(CASTLE_WQ);
      else if (from == 7) clear_castle(CASTLE_WK);
      else if (from == 56) clear_castle(CASTLE_BQ);
      else if (from == 63) clear_castle(CASTLE_BK);
    }
    if (captured && piece_type(captured) == ROOK) {
      if (capturedSq == 0) clear_castle(CASTLE_WQ);
      else if (capturedSq == 7) clear_castle(CASTLE_WK);
      else if (capturedSq == 56) clear_castle(CASTLE_BQ);
      else if (capturedSq == 63) clear_castle(CASTLE_BK);
    }

    // EP target after a double pawn push
    if (pawnMove && (m.flags & MF_DOUBLE)) {
      ep = (us == WHITE) ? (from + 8) : (from - 8);
    }

    // Halfmove clock
    if (pawnMove || captured) halfmove = 0;
    else halfmove++;

    // Side to move
    side ^= 1;
    if (us == BLACK) fullmove++;

    // Update hash: add new castle, side, and EP components
    key ^= Z.castle[castling & 15];
    key ^= Z.side;
    if (ep_hash_applicable()) key ^= Z.epFile[ep & 7];

    // Legality: king must not be in check
    if (square_attacked(kingSq[us], side)) {
      unmake_move(m, u);
      return false;
    }
    return true;
  }

  void unmake_move(const Move& m, const Undo& u) {
    // Undo piece movement/captures by reversing the transformation.
    // Then restore state variables exactly.
    int us_after = side;        // side after move was made
    int us_before = us_after ^ 1;

    int from = m.from;
    int to = m.to;

    int movedPlaced = board[to];
    if (movedPlaced) remove_piece(to, movedPlaced);

    // Undo castling rook move
    if (m.flags & MF_CASTLE) {
      if (us_before == WHITE) {
        if (to == 6) { // O-O
          int rook = board[5];
          if (rook) {
            remove_piece(5, rook);
            add_piece(7, rook);
          }
        } else if (to == 2) { // O-O-O
          int rook = board[3];
          if (rook) {
            remove_piece(3, rook);
            add_piece(0, rook);
          }
        }
      } else {
        if (to == 62) {
          int rook = board[61];
          if (rook) {
            remove_piece(61, rook);
            add_piece(63, rook);
          }
        } else if (to == 58) {
          int rook = board[59];
          if (rook) {
            remove_piece(59, rook);
            add_piece(56, rook);
          }
        }
      }
    }

    // Restore moved piece (handle promotion back to pawn).
    // We recover the original piece from the piece that was placed on `to` (stored in movedPlaced).
    // If promotion happened, restore pawn.
    int restorePiece = make_piece(u.side, PAWN);
    if (!(m.promo)) {
      // Determine original piece type from undo key is hard; use u.side and move source square on prior board.
      // But we can recover by looking at the piece that ended on 'to' before we removed it: it's in 'movedPlaced'.
      // For non-promotion, that piece is the original moving piece.
      restorePiece = movedPlaced;
    }
    add_piece(from, restorePiece);

    // Restore captured piece if any
    if (u.capturedPiece) {
      add_piece(u.capturedSquare, u.capturedPiece);
    }

    // Restore exact state
    side = u.side;
    castling = u.castling;
    ep = u.ep;
    halfmove = u.halfmove;
    fullmove = u.fullmove;
    kingSq = u.kingSq;
    key = u.key;
  }
};

enum TTFlag : std::uint8_t { TT_EXACT = 0, TT_LOWER = 1, TT_UPPER = 2 };

struct TTEntry {
  U64 key = 0;
  std::int16_t score = 0;
  std::int16_t depth = -1;
  std::uint8_t flag = TT_EXACT;
  std::uint8_t age = 0;
  std::uint32_t bestMove = 0;
};

struct TranspositionTable {
  std::vector<TTEntry> table;
  std::size_t mask = 0;
  std::uint8_t age = 0;

  explicit TranspositionTable(std::size_t pow2Entries) {
    table.resize(pow2Entries);
    mask = pow2Entries - 1;
  }

  TTEntry* probe(U64 key) {
    return &table[static_cast<std::size_t>(key) & mask];
  }

  void new_search() { age++; }
};

inline int to_tt_score(int score, int ply) {
  if (score > MATE_THRESHOLD) return score + ply;
  if (score < -MATE_THRESHOLD) return score - ply;
  return score;
}

inline int from_tt_score(int score, int ply) {
  if (score > MATE_THRESHOLD) return score - ply;
  if (score < -MATE_THRESHOLD) return score + ply;
  return score;
}

inline int mirror_sq(int sq) { return sq ^ 56; }

struct Search {
  Position pos;
  TranspositionTable tt;
  std::array<std::array<std::uint32_t, MAX_PLY>, 2> killers{};
  std::array<std::array<std::array<int, 64>, 64>, 2> history{}; // [side][from][to]

  std::chrono::steady_clock::time_point endTime{};
  std::uint64_t nodes = 0;
  bool stop = false;
  int selDepth = 0;

  std::vector<U64> keyHistory;

  explicit Search(std::size_t ttPow2Entries) : tt(ttPow2Entries) {
    for (auto& a : killers) a.fill(0);
    for (auto& s : history) for (auto& r : s) r.fill(0);
  }

  bool time_up() {
    if (stop) return true;
    if ((nodes & 2047ULL) == 0) {
      if (std::chrono::steady_clock::now() >= endTime) stop = true;
    }
    return stop;
  }

  bool is_repetition() const {
    // Check last positions with same side to move (step by 2).
    if (pos.halfmove < 2) return false;
    int limit = std::min<int>(static_cast<int>(keyHistory.size()), pos.halfmove);
    for (int i = static_cast<int>(keyHistory.size()) - 2; i >= static_cast<int>(keyHistory.size()) - limit; i -= 2) {
      if (i >= 0 && keyHistory[static_cast<std::size_t>(i)] == pos.key) return true;
    }
    return false;
  }

  int evaluate() const {
    int score = 0; // white perspective
    for (int sq = 0; sq < 64; ++sq) {
      int p = pos.board[sq];
      if (!p) continue;
      int c = piece_color(p);
      int t = piece_type(p);
      int psq = (c == WHITE) ? PST[t][sq] : PST[t][mirror_sq(sq)];
      int val = pieceValue[t] + psq;
      score += (c == WHITE) ? val : -val;
    }

    // Small mobility term for sliding pieces
    auto mobility = [&](int c) {
      int m = 0;
      U64 occAll = pos.allOcc;
      U64 bb = pos.pieces[c][BISHOP];
      while (bb) {
        int sq = pop_lsb(bb);
        m += popcount(bishop_attacks(sq, occAll) & ~pos.occ[c]);
      }
      bb = pos.pieces[c][ROOK];
      while (bb) {
        int sq = pop_lsb(bb);
        m += popcount(rook_attacks(sq, occAll) & ~pos.occ[c]);
      }
      bb = pos.pieces[c][QUEEN];
      while (bb) {
        int sq = pop_lsb(bb);
        m += popcount((rook_attacks(sq, occAll) | bishop_attacks(sq, occAll)) & ~pos.occ[c]);
      }
      return m;
    };
    score += 2 * (mobility(WHITE) - mobility(BLACK));

    return (pos.side == WHITE) ? score : -score;
  }

  int qsearch(int alpha, int beta, int ply) {
    nodes++;
    if (time_up()) return 0;
    selDepth = std::max(selDepth, ply);

    int stand = evaluate();
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;

    std::array<Move, 256> moves{};
    int moveCount = pos.generate_moves(moves, true);

    // Order captures by MVV-LVA
    std::sort(moves.begin(), moves.begin() + moveCount, [&](const Move& a, const Move& b) {
      auto capScore = [&](const Move& m) {
        int victim = 0;
        if (m.flags & MF_ENPASSANT) victim = pieceValue[PAWN];
        else {
          int cp = pos.board[m.to];
          victim = cp ? pieceValue[piece_type(cp)] : 0;
        }
        int attackerType = piece_type(pos.board[m.from]);
        return victim * 10 - pieceValue[attackerType];
      };
      return capScore(a) > capScore(b);
    });

    for (int i = 0; i < moveCount; ++i) {
      const Move& m = moves[static_cast<std::size_t>(i)];
      Undo u;
      if (!pos.make_move(m, u)) continue;
      keyHistory.push_back(pos.key);
      int score = -qsearch(-beta, -alpha, ply + 1);
      keyHistory.pop_back();
      pos.unmake_move(m, u);
      if (stop) return 0;
      if (score >= beta) return beta;
      if (score > alpha) alpha = score;
    }
    return alpha;
  }

  int negamax(int depth, int alpha, int beta, int ply, bool allowNull) {
    nodes++;
    if (time_up()) return 0;

    selDepth = std::max(selDepth, ply);

    if (pos.halfmove >= 100) return 0;
    if (is_repetition()) return 0;

    bool inCheck = pos.in_check(pos.side);
    if (inCheck) depth++;

    if (depth <= 0) return qsearch(alpha, beta, ply);

    int alphaOrig = alpha;

    // TT probe
    TTEntry* entry = tt.probe(pos.key);
    std::uint32_t ttMove = 0;
    if (entry->key == pos.key) {
      ttMove = entry->bestMove;
      if (entry->depth >= depth) {
        int ttScore = from_tt_score(entry->score, ply);
        if (entry->flag == TT_EXACT) return ttScore;
        if (entry->flag == TT_LOWER && ttScore >= beta) return ttScore;
        if (entry->flag == TT_UPPER && ttScore <= alpha) return ttScore;
      }
    }

    // Null move pruning
    if (allowNull && depth >= 3 && !inCheck) {
      U64 minorsMajors = pos.pieces[pos.side][KNIGHT] | pos.pieces[pos.side][BISHOP] | pos.pieces[pos.side][ROOK] |
                         pos.pieces[pos.side][QUEEN];
      if (minorsMajors) {
        Undo u;
        pos.make_null(u);
        keyHistory.push_back(pos.key);
        int score = -negamax(depth - 1 - 2, -beta, -beta + 1, ply + 1, false);
        keyHistory.pop_back();
        pos.unmake_null(u);
        if (stop) return 0;
        if (score >= beta) return beta;
      }
    }

    std::array<Move, 256> moves{};
    int movesCount = pos.generate_moves(moves, false);

    if (movesCount == 0) {
      if (inCheck) return -MATE_SCORE + ply;
      return 0;
    }

    auto move_order_score = [&](const Move& m) -> int {
      std::uint32_t enc = encode_move(m);
      if (ttMove && enc == ttMove) return 1000000000;
      int us = pos.side;
      if (is_capture(m)) {
        int victim = 0;
        if (m.flags & MF_ENPASSANT) victim = pieceValue[PAWN];
        else {
          int cp = pos.board[m.to];
          victim = cp ? pieceValue[piece_type(cp)] : 0;
        }
        int attacker = pieceValue[piece_type(pos.board[m.from])];
        int promo = m.promo ? pieceValue[m.promo] : 0;
        return 5000000 + victim * 10 - attacker + promo;
      }
      if (killers[0][ply] == enc) return 4000000;
      if (killers[1][ply] == enc) return 3990000;
      return history[us][m.from][m.to];
    };

    std::sort(moves.begin(), moves.begin() + movesCount,
              [&](const Move& a, const Move& b) { return move_order_score(a) > move_order_score(b); });

    int bestScore = -INF;
    std::uint32_t bestEnc = 0;

    int moveCount = 0;
    for (int mi = 0; mi < movesCount; ++mi) {
      const Move& m = moves[static_cast<std::size_t>(mi)];
      if (time_up()) break;
      Undo u;
      if (!pos.make_move(m, u)) continue;
      keyHistory.push_back(pos.key);

      int score;
      int newDepth = depth - 1;

      bool quiet = !is_capture(m) && !is_promo(m) && !(m.flags & MF_CASTLE);
      bool doLMR = (quiet && !inCheck && depth >= 3 && moveCount >= 3);

      if (moveCount == 0) {
        score = -negamax(newDepth, -beta, -alpha, ply + 1, true);
      } else {
        int reduced = doLMR ? (newDepth - 1) : newDepth;
        score = -negamax(reduced, -alpha - 1, -alpha, ply + 1, true);
        if (!stop && score > alpha && reduced != newDepth) {
          score = -negamax(newDepth, -alpha - 1, -alpha, ply + 1, true);
        }
        if (!stop && score > alpha && score < beta) {
          score = -negamax(newDepth, -beta, -alpha, ply + 1, true);
        }
      }

      keyHistory.pop_back();
      pos.unmake_move(m, u);
      if (stop) return 0;

      moveCount++;
      if (score > bestScore) {
        bestScore = score;
        bestEnc = encode_move(m);
      }
      if (score > alpha) {
        alpha = score;
        if (alpha >= beta) {
          // Beta cutoff
          if (!is_capture(m)) {
            std::uint32_t enc = encode_move(m);
            if (killers[0][ply] != enc) {
              killers[1][ply] = killers[0][ply];
              killers[0][ply] = enc;
            }
            history[pos.side][m.from][m.to] += depth * depth;
          }
          break;
        }
      }
    }

    // TT store
    TTFlag flag = TT_EXACT;
    if (bestScore <= alphaOrig) flag = TT_UPPER;
    else if (bestScore >= beta) flag = TT_LOWER;

    entry = tt.probe(pos.key);
    if (entry->key != pos.key || entry->depth <= depth) {
      entry->key = pos.key;
      entry->depth = static_cast<std::int16_t>(depth);
      entry->score = static_cast<std::int16_t>(to_tt_score(bestScore, ply));
      entry->flag = static_cast<std::uint8_t>(flag);
      entry->bestMove = bestEnc;
      entry->age = tt.age;
    }

    return bestScore;
  }

  std::pair<Move, int> search_best(double seconds) {
    stop = false;
    nodes = 0;
    selDepth = 0;
    tt.new_search();

    auto start = std::chrono::steady_clock::now();
    // Keep a small safety margin
    double margin = 0.06;
    double usable = std::max(0.05, seconds - margin);
    endTime = start + std::chrono::duration_cast<std::chrono::steady_clock::duration>(std::chrono::duration<double>(usable));

    keyHistory.clear();
    keyHistory.reserve(2048);
    keyHistory.push_back(pos.key);

    Move bestMove{};
    int bestScore = -INF;

    std::array<Move, 256> rootMoves{};
    int rootCount = pos.generate_moves(rootMoves, false);
    if (rootCount == 0) return {bestMove, 0};

    // Try TT best move first for root ordering.
    std::uint32_t ttMove = 0;
    TTEntry* e = tt.probe(pos.key);
    if (e->key == pos.key) ttMove = e->bestMove;
    if (ttMove) {
      for (int i = 0; i < rootCount; ++i) {
        if (encode_move(rootMoves[static_cast<std::size_t>(i)]) == ttMove) {
          std::swap(rootMoves[0], rootMoves[static_cast<std::size_t>(i)]);
          break;
        }
      }
    }

    int prevScore = 0;
    for (int depth = 1; depth <= 64; ++depth) {
      if (time_up()) break;
      selDepth = 0;

      std::uint32_t pvMove = encode_move(bestMove);

      auto root_order_score = [&](const Move& m) -> int {
        std::uint32_t enc = encode_move(m);
        if (pvMove && enc == pvMove) return 2000000000;
        if (ttMove && enc == ttMove) return 1990000000;
        if (is_capture(m)) {
          int victim = 0;
          if (m.flags & MF_ENPASSANT) victim = pieceValue[PAWN];
          else {
            int cp = pos.board[m.to];
            victim = cp ? pieceValue[piece_type(cp)] : 0;
          }
          int attacker = pieceValue[piece_type(pos.board[m.from])];
          int promo = m.promo ? pieceValue[m.promo] : 0;
          return 100000000 + victim * 10 - attacker + promo;
        }
        return history[pos.side][m.from][m.to];
      };

      std::sort(rootMoves.begin(), rootMoves.begin() + rootCount,
                [&](const Move& a, const Move& b) { return root_order_score(a) > root_order_score(b); });

      auto root_search = [&](int alpha, int beta, Move& outBest) -> int {
        int best = -INF;
        int a = alpha;
        Move bestLocal = rootMoves[0];
        for (int i = 0; i < rootCount; ++i) {
          const Move& m = rootMoves[static_cast<std::size_t>(i)];
          if (time_up()) break;
          Undo u;
          if (!pos.make_move(m, u)) continue;
          keyHistory.push_back(pos.key);
          int score = -negamax(depth - 1, -beta, -a, 1, true);
          keyHistory.pop_back();
          pos.unmake_move(m, u);
          if (stop) break;

          if (score > best) {
            best = score;
            bestLocal = m;
          }
          if (score > a) a = score;
          if (a >= beta) break;
        }
        outBest = bestLocal;
        return best;
      };

      Move depthBest = bestMove;
      int depthScore = -INF;

      if (depth < 3) {
        int alpha = -INF, beta = INF;
        depthScore = root_search(alpha, beta, depthBest);
      } else {
        int window = 60;
        int alpha = prevScore - window;
        int beta = prevScore + window;

        for (;;) {
          depthScore = root_search(alpha, beta, depthBest);
          if (stop) break;
          if (depthScore <= alpha) {
            alpha -= window;
            window *= 2;
            continue;
          }
          if (depthScore >= beta) {
            beta += window;
            window *= 2;
            continue;
          }
          break;
        }
      }

      if (stop) break;

      bestMove = depthBest;
      bestScore = depthScore;
      prevScore = bestScore;

      // Keep PV move first for the next iteration.
      for (int i = 0; i < rootCount; ++i) {
        if (encode_move(rootMoves[static_cast<std::size_t>(i)]) == encode_move(bestMove)) {
          std::swap(rootMoves[0], rootMoves[static_cast<std::size_t>(i)]);
          break;
        }
      }
    }

    return {bestMove, bestScore};
  }
};

static bool parse_fen(Position& pos, const std::string& fen) {
  pos.clear();
  std::istringstream iss(fen);
  std::string boardPart, sidePart, castlePart, epPart;
  int half = 0, full = 1;
  if (!(iss >> boardPart >> sidePart >> castlePart >> epPart >> half >> full)) return false;

  int rank = 7, file = 0;
  for (char c : boardPart) {
    if (c == '/') {
      rank--;
      file = 0;
      continue;
    }
    if (c >= '1' && c <= '8') {
      file += (c - '0');
      continue;
    }
    int color = (c >= 'a' && c <= 'z') ? BLACK : WHITE;
    char uc = (c >= 'a' && c <= 'z') ? static_cast<char>(c - 'a' + 'A') : c;
    int type = -1;
    switch (uc) {
      case 'P': type = PAWN; break;
      case 'N': type = KNIGHT; break;
      case 'B': type = BISHOP; break;
      case 'R': type = ROOK; break;
      case 'Q': type = QUEEN; break;
      case 'K': type = KING; break;
      default: return false;
    }
    if (file < 0 || file >= 8 || rank < 0 || rank >= 8) return false;
    int sq = rank * 8 + file;
    pos.board[sq] = make_piece(color, type);
    file++;
  }

  pos.side = (sidePart == "b") ? BLACK : WHITE;

  pos.castling = 0;
  if (castlePart != "-") {
    for (char c : castlePart) {
      if (c == 'K') pos.castling |= CASTLE_WK;
      else if (c == 'Q') pos.castling |= CASTLE_WQ;
      else if (c == 'k') pos.castling |= CASTLE_BK;
      else if (c == 'q') pos.castling |= CASTLE_BQ;
    }
  }

  pos.ep = -1;
  if (epPart != "-" && epPart.size() == 2) {
    int f = epPart[0] - 'a';
    int r = epPart[1] - '1';
    if (f >= 0 && f < 8 && r >= 0 && r < 8) pos.ep = r * 8 + f;
  }

  pos.halfmove = std::max(0, half);
  pos.fullmove = std::max(1, full);
  pos.recompute_bitboards();
  pos.recompute_key();
  return true;
}

static std::string square_to_str(int sq) {
  char file = static_cast<char>('a' + (sq & 7));
  char rank = static_cast<char>('1' + (sq / 8));
  std::string s;
  s.push_back(file);
  s.push_back(rank);
  return s;
}

static char piece_letter_for_output(int pieceType) {
  switch (pieceType) {
    case PAWN: return 'p';
    case KNIGHT: return 'N';
    case BISHOP: return 'B';
    case ROOK: return 'R';
    case QUEEN: return 'Q';
    case KING: return 'K';
    default: return '?';
  }
}

static char promo_letter(int promoType) {
  switch (promoType) {
    case QUEEN: return 'Q';
    case ROOK: return 'R';
    case BISHOP: return 'B';
    case KNIGHT: return 'N';
    default: return 'Q';
  }
}

static std::string move_to_string_with_check(Position& pos, const Move& m) {
  // Convert move according to the benchmark's "disambiguated algebraic notation".
  // Also appends '+' / '#' if the move gives check or checkmate.
  if (m.flags & MF_CASTLE) {
    std::string s = (m.to == 6 || m.to == 62) ? "O-O" : "O-O-O";
    Undo u;
    if (pos.make_move(m, u)) {
      bool check = pos.in_check(pos.side);
      std::array<Move, 256> replies{};
      int replyCount = pos.generate_moves(replies, false);
      bool hasLegal = false;
      for (int i = 0; i < replyCount; ++i) {
        const Move& r = replies[static_cast<std::size_t>(i)];
        Undo ur;
        if (pos.make_move(r, ur)) {
          hasLegal = true;
          pos.unmake_move(r, ur);
          break;
        }
      }
      if (check) s.push_back(hasLegal ? '+' : '#');
      pos.unmake_move(m, u);
    }
    return s;
  }

  int moving = pos.board[m.from];
  int t = moving ? piece_type(moving) : PAWN;
  char pl = piece_letter_for_output(t);
  std::string s;
  s.push_back(pl);
  s += square_to_str(m.from);
  if (m.flags & MF_CAPTURE) s.push_back('x');
  s += square_to_str(m.to);
  if (m.flags & MF_ENPASSANT) s += "ep";
  if (m.promo) {
    s.push_back('/');
    s.push_back(promo_letter(m.promo));
  }

  Undo u;
  if (pos.make_move(m, u)) {
    bool check = pos.in_check(pos.side);
    if (check) {
      std::array<Move, 256> replies{};
      int replyCount = pos.generate_moves(replies, false);
      bool hasLegal = false;
      for (int i = 0; i < replyCount; ++i) {
        const Move& r = replies[static_cast<std::size_t>(i)];
        Undo ur;
        if (pos.make_move(r, ur)) {
          hasLegal = true;
          pos.unmake_move(r, ur);
          break;
        }
      }
      s.push_back(hasLegal ? '+' : '#');
    }
    pos.unmake_move(m, u);
  }
  return s;
}

} // namespace chessbot

int main() {
  using namespace chessbot;
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  init_tables();

  std::string line1;
  std::string fen;
  if (!std::getline(std::cin, line1)) return 0;
  if (!std::getline(std::cin, fen)) return 0;

  double timeout = 5.0;
  try {
    timeout = std::stod(line1);
  } catch (...) {
    timeout = 5.0;
  }

  Search searcher(1ULL << 21); // ~2M entries
  if (!parse_fen(searcher.pos, fen)) {
    std::cout << "O-O\n";
    return 0;
  }

  auto [best, score] = searcher.search_best(timeout);
  (void)score;

  // Fallback: if best is default/invalid, output any legal move.
  std::array<Move, 256> legal{};
  int legalCount = searcher.pos.generate_moves(legal, false);
  Move picked = best;
  if (legalCount == 0) {
    std::cout << "O-O\n";
    return 0;
  }
  bool ok = false;
  bool haveFirstLegal = false;
  Move firstLegal = legal[0];
  for (int i = 0; i < legalCount; ++i) {
    const Move& m = legal[static_cast<std::size_t>(i)];
    Undo u;
    if (searcher.pos.make_move(m, u)) {
      searcher.pos.unmake_move(m, u);
      if (!haveFirstLegal) {
        firstLegal = m;
        haveFirstLegal = true;
      }
      if (encode_move(m) == encode_move(best)) ok = true;
    }
  }
  if (!ok) picked = haveFirstLegal ? firstLegal : legal[0];

  std::cout << move_to_string_with_check(searcher.pos, picked) << "\n";
  return 0;
}
