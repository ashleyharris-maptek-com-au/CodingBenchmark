use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read, Write};
use std::time::{Duration, Instant};

struct Scanner {
    buf: Vec<u8>,
    i: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, i: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        // Parse non-negative integers, ignoring any non-digit separators.
        // This is intentionally tolerant of UTF-16LE pipes (NUL bytes and BOMs).
        while self.i < self.buf.len() {
            let c = self.buf[self.i];
            if c == 0 {
                self.i += 1;
                continue;
            }
            if (b'0'..=b'9').contains(&c) {
                break;
            }
            self.i += 1;
        }
        if self.i >= self.buf.len() {
            return None;
        }
        let mut v: usize = 0;
        while self.i < self.buf.len() {
            let c = self.buf[self.i];
            if c == 0 {
                self.i += 1;
                continue;
            }
            if !(b'0'..=b'9').contains(&c) {
                break;
            }
            v = v.saturating_mul(10).saturating_add((c - b'0') as usize);
            self.i += 1;
        }
        Some(v)
    }
}

#[derive(Clone)]
struct Grid {
    w: usize,
    h: usize,
    free: Vec<bool>,
    neigh: Vec<[usize; 4]>,
    neigh_len: Vec<u8>,
    free_count: usize,
}

impl Grid {
    fn xy(&self, idx: usize) -> (usize, usize) {
        (idx % self.w, idx / self.w)
    }

    fn build(w: usize, h: usize, blocked: &[(usize, usize)]) -> Option<Self> {
        let total = w.checked_mul(h)?;
        let mut free = vec![true; total];
        for &(x, y) in blocked {
            if x >= w || y >= h {
                return None;
            }
            free[y * w + x] = false;
        }
        let mut free_count = 0usize;
        for &f in &free {
            if f {
                free_count += 1;
            }
        }

        let mut neigh = vec![[usize::MAX; 4]; total];
        let mut neigh_len = vec![0u8; total];
        for y in 0..h {
            for x in 0..w {
                let idx = y * w + x;
                let mut len = 0u8;
                let push = |nidx: usize, len: &mut u8, arr: &mut [usize; 4]| {
                    arr[*len as usize] = nidx;
                    *len += 1;
                };
                if x > 0 {
                    push(idx - 1, &mut len, &mut neigh[idx]);
                }
                if x + 1 < w {
                    push(idx + 1, &mut len, &mut neigh[idx]);
                }
                if y > 0 {
                    push(idx - w, &mut len, &mut neigh[idx]);
                }
                if y + 1 < h {
                    push(idx + w, &mut len, &mut neigh[idx]);
                }
                neigh_len[idx] = len;
            }
        }

        Some(Self {
            w,
            h,
            free,
            neigh,
            neigh_len,
            free_count,
        })
    }

    fn degree(&self, idx: usize) -> u8 {
        if !self.free[idx] {
            return 0;
        }
        let mut d = 0u8;
        let len = self.neigh_len[idx] as usize;
        for k in 0..len {
            let nb = self.neigh[idx][k];
            if self.free[nb] {
                d += 1;
            }
        }
        d
    }

    fn is_adj(&self, a: usize, b: usize) -> bool {
        let (ax, ay) = self.xy(a);
        let (bx, by) = self.xy(b);
        ax.abs_diff(bx) + ay.abs_diff(by) == 1
    }

    fn bfs_reachable_from_start(&self) -> usize {
        let start = 0usize;
        if self.free.is_empty() || !self.free[start] {
            return 0;
        }
        let mut q = VecDeque::new();
        let mut seen = vec![false; self.free.len()];
        seen[start] = true;
        q.push_back(start);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            let len = self.neigh_len[v] as usize;
            for k in 0..len {
                let nb = self.neigh[v][k];
                if self.free[nb] && !seen[nb] {
                    seen[nb] = true;
                    q.push_back(nb);
                }
            }
        }
        count
    }
}

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
}

fn parity_ok_for_path(grid: &Grid, require_cycle: bool) -> bool {
    let mut black = 0usize;
    let mut white = 0usize;
    for idx in 0..grid.free.len() {
        if !grid.free[idx] {
            continue;
        }
        let (x, y) = grid.xy(idx);
        if (x + y) % 2 == 0 {
            black += 1;
        } else {
            white += 1;
        }
    }
    let n = black + white;
    if n == 0 {
        return false;
    }
    if require_cycle {
        return n % 2 == 0 && black == white;
    }
    let diff = if black > white { black - white } else { white - black };
    if diff > 1 {
        return false;
    }
    if n % 2 == 0 {
        // endpoints have opposite colors => partitions must match
        black == white
    } else {
        // endpoints have same color; start at (0,0) is black
        black == white + 1
    }
}

fn degrees_ok(grid: &Grid, require_cycle: bool) -> bool {
    if require_cycle {
        for idx in 0..grid.free.len() {
            if grid.free[idx] && grid.degree(idx) < 2 {
                return false;
            }
        }
        return true;
    }

    // Necessary: at most 2 degree-1 vertices in the whole graph.
    let mut deg1 = 0usize;
    for idx in 0..grid.free.len() {
        if grid.free[idx] && grid.degree(idx) == 1 {
            deg1 += 1;
            if deg1 > 2 {
                return false;
            }
        }
    }
    true
}

fn is_unvisited_connected(grid: &Grid, visited: &[bool], unvisited_count: usize, stamp: &mut u32, seen: &mut [u32]) -> bool {
    if unvisited_count <= 1 {
        return true;
    }
    *stamp = stamp.wrapping_add(1).max(1);
    let s = *stamp;
    let mut start = None;
    for i in 0..grid.free.len() {
        if grid.free[i] && !visited[i] {
            start = Some(i);
            break;
        }
    }
    let Some(start) = start else { return true; };
    let mut q = VecDeque::new();
    q.push_back(start);
    seen[start] = s;
    let mut reached = 0usize;
    while let Some(v) = q.pop_front() {
        reached += 1;
        let len = grid.neigh_len[v] as usize;
        for k in 0..len {
            let nb = grid.neigh[v][k];
            if grid.free[nb] && !visited[nb] && seen[nb] != s {
                seen[nb] = s;
                q.push_back(nb);
            }
        }
    }
    reached == unvisited_count
}

fn validate_path(grid: &Grid, path: &[usize], require_cycle: bool) -> bool {
    if path.is_empty() {
        return false;
    }
    if path[0] != 0 {
        return false;
    }
    let mut seen = vec![false; grid.free.len()];
    for &idx in path {
        if idx >= grid.free.len() || !grid.free[idx] || seen[idx] {
            return false;
        }
        seen[idx] = true;
    }
    if path.len() != grid.free_count {
        return false;
    }
    for i in 1..path.len() {
        if !grid.is_adj(path[i - 1], path[i]) {
            return false;
        }
    }
    if require_cycle && !grid.is_adj(*path.last().unwrap(), 0) {
        return false;
    }
    true
}

fn greedy_attempt(
    grid: &Grid,
    require_cycle: bool,
    reserved_end: Option<usize>,
    rng: &mut XorShift64,
    deadline: Instant,
) -> Option<Vec<usize>> {
    let total = grid.free.len();
    let start = 0usize;
    if total == 0 || !grid.free[start] {
        return None;
    }
    let mut visited = vec![false; total];
    visited[start] = true;
    let mut path = Vec::with_capacity(grid.free_count);
    path.push(start);

    let mut stamp = 1u32;
    let mut seen = vec![0u32; total];
    let mut cur = start;
    while path.len() < grid.free_count {
        if Instant::now() >= deadline {
            return None;
        }
        let is_last_move = path.len() + 1 == grid.free_count;
        let mut cands: Vec<(usize, u8, u64)> = Vec::new();
        let len = grid.neigh_len[cur] as usize;
        for k in 0..len {
            let nb = grid.neigh[cur][k];
            if !grid.free[nb] || visited[nb] {
                continue;
            }
            if let Some(end) = reserved_end {
                if nb == end && !is_last_move {
                    continue;
                }
            }
            // Warnsdorff: count onward unvisited neighbors after moving to nb (exclude cur which is visited).
            let mut deg = 0u8;
            let nlen = grid.neigh_len[nb] as usize;
            for kk in 0..nlen {
                let nn = grid.neigh[nb][kk];
                if grid.free[nn] && !visited[nn] && nn != cur {
                    // if reserved_end exists and nn==reserved_end, it's still unvisited and counts.
                    deg += 1;
                }
            }
            cands.push((nb, deg, rng.next_u64()));
        }

        if cands.is_empty() {
            return None;
        }

        cands.sort_unstable_by(|a, b| match a.1.cmp(&b.1) {
            Ordering::Equal => a.2.cmp(&b.2),
            o => o,
        });

        let mut moved = false;
        for (nb, _, _) in cands {
            visited[nb] = true;
            let unvisited_left = grid.free_count - path.len() - 1;
            if is_unvisited_connected(grid, &visited, unvisited_left, &mut stamp, &mut seen) {
                path.push(nb);
                cur = nb;
                moved = true;
                break;
            }
            visited[nb] = false;
        }
        if !moved {
            return None;
        }
    }

    if let Some(end) = reserved_end {
        if *path.last().unwrap() != end {
            return None;
        }
    }
    if require_cycle && !grid.is_adj(*path.last().unwrap(), start) {
        return None;
    }
    if !validate_path(grid, &path, require_cycle) {
        return None;
    }
    Some(path)
}

fn dfs_exact(
    grid: &Grid,
    require_cycle: bool,
    reserved_end: Option<usize>,
    visited: &mut [bool],
    path: &mut Vec<usize>,
    cur: usize,
    stamp: &mut u32,
    seen: &mut [u32],
    deadline: Instant,
) -> bool {
    if Instant::now() >= deadline {
        return false;
    }
    if path.len() == grid.free_count {
        if let Some(end) = reserved_end {
            if cur != end {
                return false;
            }
        }
        return !require_cycle || grid.is_adj(cur, 0);
    }

    let is_last_move = path.len() + 1 == grid.free_count;
    let mut cands: Vec<(usize, u8)> = Vec::new();
    let len = grid.neigh_len[cur] as usize;
    for k in 0..len {
        let nb = grid.neigh[cur][k];
        if !grid.free[nb] || visited[nb] {
            continue;
        }
        if let Some(end) = reserved_end {
            if nb == end && !is_last_move {
                continue;
            }
        }
        let mut deg = 0u8;
        let nlen = grid.neigh_len[nb] as usize;
        for kk in 0..nlen {
            let nn = grid.neigh[nb][kk];
            if grid.free[nn] && !visited[nn] && nn != cur {
                deg += 1;
            }
        }
        cands.push((nb, deg));
    }

    cands.sort_unstable_by(|a, b| a.1.cmp(&b.1));

    for (nb, _) in cands {
        visited[nb] = true;
        path.push(nb);
        let unvisited_left = grid.free_count - path.len();
        let ok = is_unvisited_connected(grid, visited, unvisited_left, stamp, seen);
        if ok && dfs_exact(grid, require_cycle, reserved_end, visited, path, nb, stamp, seen, deadline) {
            return true;
        }
        path.pop();
        visited[nb] = false;
    }
    false
}

fn solve_with_obstacles(grid: &Grid, require_cycle: bool, global_deadline: Instant) -> Option<Vec<usize>> {
    let start = 0usize;
    if grid.free_count == 0 || grid.free.is_empty() {
        return None;
    }
    if !grid.free[start] {
        return None;
    }
    if grid.free_count == 1 {
        return if require_cycle { None } else { Some(vec![start]) };
    }

    if grid.bfs_reachable_from_start() != grid.free_count {
        return None;
    }
    if !parity_ok_for_path(grid, require_cycle) {
        return None;
    }
    if !degrees_ok(grid, require_cycle) {
        return None;
    }

    let mut end_candidates: Vec<Option<usize>> = Vec::new();
    if require_cycle {
        // End must be adjacent to start: try each free neighbor of start.
        let len = grid.neigh_len[start] as usize;
        for k in 0..len {
            let nb = grid.neigh[start][k];
            if grid.free[nb] {
                end_candidates.push(Some(nb));
            }
        }
        if end_candidates.is_empty() {
            return None;
        }
    } else {
        end_candidates.push(None);
    }

    // Heuristic-first for larger instances, exact search for small.
    let small = grid.free_count <= 180;

    for reserved_end in end_candidates {
        // Multiple greedy restarts with different tie-breaks.
        let mut seed = 0x9E37_79B9_7F4A_7C15u64;
        seed ^= (grid.w as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9);
        seed ^= (grid.h as u64).wrapping_mul(0x94D0_49BB_1331_11EB);
        seed ^= (grid.free_count as u64).wrapping_mul(0xD134_2543_DE82_EB97);
        if let Some(e) = reserved_end {
            seed ^= (e as u64).wrapping_mul(0x2545_F491_4F6C_DD1D);
        }
        let mut rng = XorShift64::new(seed);

        for _ in 0..200 {
            if Instant::now() >= global_deadline {
                return None;
            }
            let attempt_deadline = global_deadline.min(Instant::now() + Duration::from_millis(250));
            if let Some(p) = greedy_attempt(grid, require_cycle, reserved_end, &mut rng, attempt_deadline) {
                return Some(p);
            }
        }

        if small {
            // Exact backtracking within remaining time.
            let total = grid.free.len();
            let mut visited = vec![false; total];
            visited[start] = true;
            let mut path = Vec::with_capacity(grid.free_count);
            path.push(start);
            let mut stamp = 1u32;
            let mut seen = vec![0u32; total];
            if dfs_exact(
                grid,
                require_cycle,
                reserved_end,
                &mut visited,
                &mut path,
                start,
                &mut stamp,
                &mut seen,
                global_deadline,
            ) {
                if validate_path(grid, &path, require_cycle) {
                    return Some(path);
                }
            }
        }
    }

    None
}

fn stream_snake_path(w: usize, h: usize, out: &mut dyn Write) -> io::Result<()> {
    let total = w.saturating_mul(h);
    writeln!(out, "{total}")?;
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                writeln!(out, "{x} {y}")?;
            }
        } else {
            for x in (0..w).rev() {
                writeln!(out, "{x} {y}")?;
            }
        }
    }
    Ok(())
}

fn stream_cycle_even_width(w_even: usize, h: usize, transpose: bool, out: &mut dyn Write) -> io::Result<()> {
    let total = w_even.saturating_mul(h);
    writeln!(out, "{total}")?;
    let emit = |x: usize, y: usize, out: &mut dyn Write| -> io::Result<()> {
        if transpose {
            writeln!(out, "{y} {x}")
        } else {
            writeln!(out, "{x} {y}")
        }
    };

    // Column 0 down.
    for y in 0..h {
        emit(0, y, out)?;
    }
    // Columns 1..w-1 zigzag on rows 1..h-1.
    for x in 1..w_even {
        if x % 2 == 1 {
            emit(x, h - 1, out)?;
            if h >= 3 {
                for y in (1..h - 1).rev() {
                    emit(x, y, out)?;
                }
            }
        } else {
            emit(x, 1, out)?;
            for y in 2..h {
                emit(x, y, out)?;
            }
        }
    }
    // Top row right-to-left (excluding x=0 which is already visited).
    emit(w_even - 1, 0, out)?;
    if w_even >= 3 {
        for x in (1..w_even - 1).rev() {
            emit(x, 0, out)?;
        }
    }
    Ok(())
}

fn stream_cycle_rect(w: usize, h: usize, out: &mut dyn Write) -> io::Result<bool> {
    if w <= 1 || h <= 1 {
        return Ok(false);
    }
    let total = match w.checked_mul(h) {
        Some(v) => v,
        None => return Ok(false),
    };
    if total % 2 == 1 {
        return Ok(false);
    }
    if w % 2 == 0 {
        stream_cycle_even_width(w, h, false, out)?;
        Ok(true)
    } else if h % 2 == 0 {
        // Build in transposed grid.
        stream_cycle_even_width(h, w, true, out)?;
        Ok(true)
    } else {
        Ok(false)
    }
}

fn main() {
    let mut sc = Scanner::new();
    let w = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let h = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let obs = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let require_cycle = match sc.next_usize() {
        Some(v) => v != 0,
        None => return,
    };

    let mut blocked = Vec::with_capacity(obs);
    for _ in 0..obs {
        let x = match sc.next_usize() {
            Some(v) => v,
            None => break,
        };
        let y = match sc.next_usize() {
            Some(v) => v,
            None => break,
        };
        blocked.push((x, y));
    }

    let mut out = io::BufWriter::new(io::stdout());

    // Fast paths for obstacle-free rectangles: scalable to large sizes without building the grid.
    if obs == 0 {
        if !require_cycle {
            let _ = stream_snake_path(w, h, &mut out);
            return;
        }
        match stream_cycle_rect(w, h, &mut out) {
            Ok(true) => return,
            _ => {
                let _ = writeln!(out, "0");
                return;
            }
        }
    }

    // With obstacles: we need an explicit grid; cap memory for absurd sizes.
    let Some(total_cells) = w.checked_mul(h) else {
        let _ = writeln!(out, "0");
        return;
    };
    const MAX_CELLS: usize = 5_000_000;
    if total_cells > MAX_CELLS {
        let _ = writeln!(out, "0");
        return;
    }

    let Some(grid) = Grid::build(w, h, &blocked) else {
        let _ = writeln!(out, "0");
        return;
    };

    let deadline = Instant::now() + Duration::from_secs(290);
    if let Some(path) = solve_with_obstacles(&grid, require_cycle, deadline) {
        let _ = writeln!(out, "{}", path.len());
        for idx in path {
            let (x, y) = grid.xy(idx);
            let _ = writeln!(out, "{x} {y}");
        }
    } else {
        let _ = writeln!(out, "0");
    }
}
