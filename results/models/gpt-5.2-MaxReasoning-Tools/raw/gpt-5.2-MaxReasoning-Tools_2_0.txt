using System;
using System.IO;
using System.Numerics;
using System.Runtime.CompilerServices;
using System.Collections.Generic;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner();
        if (!fs.TryNextInt(out int n) || !fs.TryNextInt(out int m) || n <= 0)
            return;

        if (m <= 0)
        {
            using var fo = new FastOutput();
            fo.WriteInt(0);
            fo.WriteNewLine();
            return;
        }

        int[] u = new int[m];
        int[] v = new int[m];
        int[] w = new int[m];

        int[] degree = new int[n];

        int[] head = new int[n];
        Array.Fill(head, -1);

        int dirEdges = checked(m * 2);
        int[] to = new int[dirEdges];
        int[] next = new int[dirEdges];
        int[] weight = new int[dirEdges];
        int[] edgeId = new int[dirEdges];

        int ptr = 0;
        for (int i = 0; i < m; i++)
        {
            fs.TryNextInt(out int a);
            fs.TryNextInt(out int b);
            fs.TryNextInt(out int c);

            u[i] = a;
            v[i] = b;
            w[i] = c;

            if ((uint)a < (uint)n) degree[a]++;
            if ((uint)b < (uint)n) degree[b]++;

            // a -> b
            to[ptr] = b;
            weight[ptr] = c;
            edgeId[ptr] = i;
            next[ptr] = head[a];
            head[a] = ptr;
            ptr++;

            // b -> a
            to[ptr] = a;
            weight[ptr] = c;
            edgeId[ptr] = i;
            next[ptr] = head[b];
            head[b] = ptr;
            ptr++;
        }

        if (degree[0] == 0)
        {
            using var fo = new FastOutput();
            fo.WriteInt(0);
            fo.WriteNewLine();
            return;
        }

        var odd = new List<int>();
        odd.Capacity = Math.Min(n, 1024);
        for (int i = 0; i < n; i++)
        {
            if ((degree[i] & 1) != 0)
                odd.Add(i);
        }

        List<int> duplicates;
        if (odd.Count == 0)
        {
            duplicates = s_emptyDup;
        }
        else if (ShouldUseExactMatching(n, m, odd.Count))
        {
            duplicates = BuildDuplicatesByExactMatching(n, head, to, next, weight, edgeId, u, v, odd);
        }
        else
        {
            duplicates = BuildDuplicatesByTreeJoin(n, head, to, next, edgeId, odd);
        }

        int totalEdges = checked(m + duplicates.Count);
        var eg = new EulerGraph(n, totalEdges);
        for (int i = 0; i < m; i++)
            eg.AddEdge(u[i], v[i]);
        for (int i = 0; i < duplicates.Count; i++)
        {
            int id = duplicates[i];
            eg.AddEdge(u[id], v[id]);
        }

        int[] route = eg.EulerCircuit(0);

        using (var fo = new FastOutput())
        {
            for (int i = route.Length - 1; i >= 0; i--)
            {
                if (i != route.Length - 1) fo.WriteSpace();
                fo.WriteInt(route[i]);
            }
            fo.WriteNewLine();
        }
    }

    private static readonly List<int> s_emptyDup = new List<int>(0);

    private static bool ShouldUseExactMatching(int n, int m, int oddCount)
    {
        // Exact bitmask DP on odd vertices, with Dijkstra from each odd vertex.
        // Safe for small odd-count and moderate graph sizes.
        if ((oddCount & 1) != 0) return false;
        if (oddCount > 20) return false;
        if (n > 30000) return false;
        if (m > 200000) return false;
        return true;
    }

    private static List<int> BuildDuplicatesByTreeJoin(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] edgeId,
        List<int> oddVertices)
    {
        int[] parent = new int[n];
        int[] parentEdge = new int[n];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        int[] queue = new int[n];
        int[] order = new int[n];
        int qh = 0, qt = 0, oc = 0;
        parent[0] = 0;
        queue[qt++] = 0;

        while (qh < qt)
        {
            int vtx = queue[qh++];
            order[oc++] = vtx;
            for (int e = head[vtx]; e != -1; e = next[e])
            {
                int nb = to[e];
                if ((uint)nb >= (uint)n) continue;
                if (parent[nb] != -1) continue;
                parent[nb] = vtx;
                parentEdge[nb] = edgeId[e];
                queue[qt++] = nb;
            }
        }

        bool[] parity = new bool[n];
        for (int i = 0; i < oddVertices.Count; i++)
            parity[oddVertices[i]] = true;

        var duplicates = new List<int>(Math.Min(n, oddVertices.Count * 2 + 8));

        for (int i = oc - 1; i >= 1; i--) // skip root at order[0]
        {
            int vtx = order[i];
            if (parity[vtx])
            {
                int id = parentEdge[vtx];
                if (id >= 0) duplicates.Add(id);
                parity[parent[vtx]] = !parity[parent[vtx]];
            }
        }

        return duplicates;
    }

    private static List<int> BuildDuplicatesByExactMatching(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] weight,
        int[] edgeId,
        int[] u,
        int[] v,
        List<int> oddVertices)
    {
        int k = oddVertices.Count;
        long[,] distOdd = new long[k, k];
        int[][] parents = new int[k][];
        int[][] parentDirEdge = new int[k][];

        for (int i = 0; i < k; i++)
        {
            Dijkstra(
                n,
                head,
                to,
                next,
                weight,
                oddVertices[i],
                out long[] dist,
                out int[] parent,
                out int[] parentEdgeDir);

            parents[i] = parent;
            parentDirEdge[i] = parentEdgeDir;

            for (int j = 0; j < k; j++)
                distOdd[i, j] = dist[oddVertices[j]];
        }

        int fullMask = (1 << k) - 1;
        long[] memo = new long[1 << k];
        int[] choice = new int[1 << k];
        for (int i = 0; i < memo.Length; i++) memo[i] = long.MinValue;

        long Solve(int mask)
        {
            if (mask == 0) return 0;
            long cached = memo[mask];
            if (cached != long.MinValue) return cached;

            int i = BitOperations.TrailingZeroCount((uint)mask);
            int withoutI = mask & ~(1 << i);
            long best = long.MaxValue / 4;
            int bestJ = -1;

            for (int j = i + 1; j < k; j++)
            {
                int bit = 1 << j;
                if ((withoutI & bit) == 0) continue;
                int nextMask = withoutI & ~bit;
                long cost = distOdd[i, j] + Solve(nextMask);
                if (cost < best)
                {
                    best = cost;
                    bestJ = j;
                }
            }

            choice[mask] = bestJ;
            memo[mask] = best;
            return best;
        }

        _ = Solve(fullMask);

        var duplicates = new List<int>();
        int maskCur = fullMask;
        while (maskCur != 0)
        {
            int i = BitOperations.TrailingZeroCount((uint)maskCur);
            int j = choice[maskCur];
            if (j < 0) break;

            int src = oddVertices[i];
            int dst = oddVertices[j];

            int cur = dst;
            int[] parent = parents[i];
            int[] parentEdgeDir = parentDirEdge[i];
            while (cur != src)
            {
                int dir = parentEdgeDir[cur];
                if (dir < 0) break;
                duplicates.Add(edgeId[dir]);
                cur = parent[cur];
                if (cur < 0) break;
            }

            maskCur &= ~(1 << i);
            maskCur &= ~(1 << j);
        }

        return duplicates;
    }

    private static void Dijkstra(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] weight,
        int source,
        out long[] dist,
        out int[] parent,
        out int[] parentEdgeDir)
    {
        dist = new long[n];
        parent = new int[n];
        parentEdgeDir = new int[n];
        Array.Fill(dist, long.MaxValue / 4);
        Array.Fill(parent, -1);
        Array.Fill(parentEdgeDir, -1);

        dist[source] = 0;
        parent[source] = source;

        var heap = new MinHeap(Math.Min(1024, n));
        heap.Push(source, 0);

        while (heap.TryPop(out int vtx, out long dv))
        {
            if (dv != dist[vtx]) continue;

            for (int e = head[vtx]; e != -1; e = next[e])
            {
                int nb = to[e];
                long nd = dv + weight[e];
                if (nd < dist[nb])
                {
                    dist[nb] = nd;
                    parent[nb] = vtx;
                    parentEdgeDir[nb] = e;
                    heap.Push(nb, nd);
                }
            }
        }
    }

    private sealed class EulerGraph
    {
        private readonly int _n;
        private readonly int[] _head;
        private readonly int[] _to;
        private readonly int[] _next;
        private readonly int[] _rev;
        private readonly bool[] _used;
        private int _ptr;

        public EulerGraph(int n, int undirectedEdges)
        {
            _n = n;
            _head = new int[n];
            Array.Fill(_head, -1);
            int halfEdges = checked(undirectedEdges * 2);
            _to = new int[halfEdges];
            _next = new int[halfEdges];
            _rev = new int[halfEdges];
            _used = new bool[halfEdges];
            _ptr = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void AddEdge(int a, int b)
        {
            int i = _ptr++;
            int j = _ptr++;
            _to[i] = b;
            _to[j] = a;
            _rev[i] = j;
            _rev[j] = i;
            _next[i] = _head[a];
            _head[a] = i;
            _next[j] = _head[b];
            _head[b] = j;
        }

        public int[] EulerCircuit(int start)
        {
            int edges = _ptr / 2;
            int[] iter = new int[_n];
            Array.Copy(_head, iter, _n);

            int[] stack = new int[edges + 1];
            int sp = 0;
            stack[sp++] = start;

            int[] circuit = new int[edges + 1];
            int cp = 0;

            while (sp > 0)
            {
                int v = stack[sp - 1];
                int e = iter[v];
                while (e != -1 && _used[e]) e = _next[e];
                iter[v] = e;

                if (e == -1)
                {
                    sp--;
                    circuit[cp++] = v;
                    continue;
                }

                _used[e] = true;
                _used[_rev[e]] = true;
                iter[v] = _next[e];
                stack[sp++] = _to[e];
            }

            if (cp != circuit.Length)
            {
                // Graph not connected from start; truncate to what we found.
                Array.Resize(ref circuit, cp);
            }

            return circuit;
        }
    }

    private sealed class MinHeap
    {
        private int[] _node;
        private long[] _dist;
        private int _size;

        public MinHeap(int capacity)
        {
            _node = new int[Math.Max(4, capacity)];
            _dist = new long[_node.Length];
            _size = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Push(int node, long dist)
        {
            int i = _size++;
            if (i == _node.Length)
            {
                int newSize = _node.Length * 2;
                Array.Resize(ref _node, newSize);
                Array.Resize(ref _dist, newSize);
            }

            _node[i] = node;
            _dist[i] = dist;

            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_dist[p] <= dist) break;
                _node[i] = _node[p];
                _dist[i] = _dist[p];
                i = p;
            }

            _node[i] = node;
            _dist[i] = dist;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool TryPop(out int node, out long dist)
        {
            if (_size == 0)
            {
                node = 0;
                dist = 0;
                return false;
            }

            node = _node[0];
            dist = _dist[0];
            _size--;
            if (_size == 0) return true;

            int lastNode = _node[_size];
            long lastDist = _dist[_size];

            int i = 0;
            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= _size) break;
                int right = left + 1;
                int best = (right < _size && _dist[right] < _dist[left]) ? right : left;
                if (_dist[best] >= lastDist) break;
                _node[i] = _node[best];
                _dist[i] = _dist[best];
                i = best;
            }

            _node[i] = lastNode;
            _dist[i] = lastDist;
            return true;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len;
        private int _ptr;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryNextInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c == -1) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > 32 && c != -1)
            {
                x = (x * 10) + (c - '0');
                c = Read();
            }

            value = x * sign;
            return true;
        }
    }

    private sealed class FastOutput : IDisposable
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _pos;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void Flush()
        {
            if (_pos <= 0) return;
            _stream.Write(_buffer, 0, _pos);
            _pos = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            if (_pos == _buffer.Length) Flush();
            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace() => WriteByte((byte)' ');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine() => WriteByte((byte)'\n');

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[12];
            int len = 0;
            while (value > 0)
            {
                tmp[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }
            for (int i = len - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void Dispose() => Flush();
    }
}
