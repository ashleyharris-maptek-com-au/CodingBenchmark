declare const require: any;
declare const process: any;

const fs = require("fs");
const rawInput: string = fs.readFileSync(0, "utf8");
const input: string =
  rawInput.length && rawInput.charCodeAt(0) === 0xfeff ? rawInput.slice(1) : rawInput;
const trimmed = input.trim();
const data: any = trimmed.length ? JSON.parse(trimmed) : {};

type Pkg = { id: any; x: number; y: number };

const depotArr: any = data.depot;
const depotX = Number(Array.isArray(depotArr) ? depotArr[0] : 0);
const depotY = Number(Array.isArray(depotArr) ? depotArr[1] : 0);
const pkgs: Pkg[] = Array.isArray(data.packages) ? (data.packages as Pkg[]) : [];
const n = pkgs.length | 0;

if (n <= 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const depotNode = n; // nodes: [0..n-1]=packages, n=depot
const xs = new Float64Array(n + 1);
const ys = new Float64Array(n + 1);
const ids: any[] = new Array(n);

let minX = Infinity,
  maxX = -Infinity,
  minY = Infinity,
  maxY = -Infinity;

for (let i = 0; i < n; i++) {
  const p: any = pkgs[i];
  ids[i] = p.id;
  const x = Number(p.x);
  const y = Number(p.y);
  xs[i] = x;
  ys[i] = y;
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
}
xs[depotNode] = depotX;
ys[depotNode] = depotY;

function dist(a: number, b: number): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.hypot(dx, dy);
}

function dist2(a: number, b: number): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return dx * dx + dy * dy;
}

function part1By1(v: number): number {
  v &= 0xffff;
  v = (v | (v << 8)) & 0x00ff00ff;
  v = (v | (v << 4)) & 0x0f0f0f0f;
  v = (v | (v << 2)) & 0x33333333;
  v = (v | (v << 1)) & 0x55555555;
  return v >>> 0;
}

function morton2D(x16: number, y16: number): number {
  return (part1By1(x16) | (part1By1(y16) << 1)) >>> 0;
}

function clampU16(v: number): number {
  if (v <= 0) return 0;
  if (v >= 65535) return 65535;
  return v | 0;
}

function buildOrderByMorton(): number[] {
  const rangeX = maxX - minX || 1;
  const rangeY = maxY - minY || 1;
  const codes = new Uint32Array(n);
  for (let i = 0; i < n; i++) {
    const xn = (xs[i] - minX) / rangeX;
    const yn = (ys[i] - minY) / rangeY;
    const x16 = clampU16(Math.floor(xn * 65535 + 0.5));
    const y16 = clampU16(Math.floor(yn * 65535 + 0.5));
    codes[i] = morton2D(x16, y16);
  }
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => (codes[a] - codes[b]) || (a - b));
  return order;
}

function buildOrderByX(): number[] {
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => (xs[a] - xs[b]) || (ys[a] - ys[b]) || (a - b));
  return order;
}

function buildOrderByY(): number[] {
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => (ys[a] - ys[b]) || (xs[a] - xs[b]) || (a - b));
  return order;
}

const mortonOrder = buildOrderByMorton();
const xOrder = buildOrderByX();
const yOrder = buildOrderByY();

function buildPos(order: number[]): Int32Array {
  const pos = new Int32Array(n);
  for (let i = 0; i < n; i++) pos[order[i]] = i;
  return pos;
}

const posMorton = buildPos(mortonOrder);
const posX = buildPos(xOrder);
const posY = buildPos(yOrder);

function buildNeighborLists(window: number, kBest: number): number[][] {
  const marks = new Int32Array(n);
  const neighbors: number[][] = new Array(n);

  for (let u = 0; u < n; u++) {
    const stamp = u + 1;
    const cand: number[] = [];

    const add = (v: number) => {
      if (v === u) return;
      if (marks[v] === stamp) return;
      marks[v] = stamp;
      cand.push(v);
    };

    const puM = posMorton[u];
    for (let d = -window; d <= window; d++) {
      if (d === 0) continue;
      const q = puM + d;
      if (q >= 0 && q < n) add(mortonOrder[q]);
    }

    const puX = posX[u];
    for (let d = -window; d <= window; d++) {
      if (d === 0) continue;
      const q = puX + d;
      if (q >= 0 && q < n) add(xOrder[q]);
    }

    const puY = posY[u];
    for (let d = -window; d <= window; d++) {
      if (d === 0) continue;
      const q = puY + d;
      if (q >= 0 && q < n) add(yOrder[q]);
    }

    cand.sort((a, b) => dist2(u, a) - dist2(u, b));
    if (cand.length > kBest) cand.length = kBest;
    neighbors[u] = cand;
  }
  return neighbors;
}

const neighborWindow = n <= 2000 ? 10 : n <= 10000 ? 8 : 6;
const kBest = n <= 2000 ? 30 : n <= 10000 ? 24 : 18;
const neighborLists = buildNeighborLists(neighborWindow, kBest);

// Initial route: depot -> Morton order -> depot
const route = new Array<number>(n + 2);
route[0] = depotNode;
for (let i = 0; i < n; i++) route[i + 1] = mortonOrder[i];
route[n + 1] = depotNode;

function reverseInPlace(arr: number[], i: number, k: number): void {
  while (i < k) {
    const tmp = arr[i];
    arr[i] = arr[k];
    arr[k] = tmp;
    i++;
    k--;
  }
}

function twoOptImprovePath(arr: number[], timeLimitMs: number): void {
  if (n <= 2 || timeLimitMs <= 0) return;

  const pos = new Int32Array(n);
  for (let i = 1; i <= n; i++) pos[arr[i]] = i;

  const endTime = Date.now() + timeLimitMs;
  let passes = 0;

  while (Date.now() < endTime && passes < 50) {
    let improved = false;
    passes++;

    for (let iPos = 1; iPos <= n - 1 && Date.now() < endTime; iPos++) {
      const baseNode = arr[iPos];
      const cand = neighborLists[baseNode];
      for (let t = 0; t < cand.length; t++) {
        const cNode = cand[t];
        const kPos = pos[cNode];
        if (kPos === iPos) continue;

        let i = iPos;
        let k = kPos;
        if (k < i) {
          const tmp = i;
          i = k;
          k = tmp;
        }
        if (k <= i) continue;
        if (i < 1 || k > n) continue;

        const a = arr[i - 1];
        const d = arr[k + 1];
        const bi = arr[i];
        const ck = arr[k];

        const oldCost = dist(a, bi) + dist(ck, d);
        const newCost = dist(a, ck) + dist(bi, d);

        if (newCost + 1e-12 < oldCost) {
          reverseInPlace(arr, i, k);
          for (let p = i; p <= k; p++) pos[arr[p]] = p;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }

    if (!improved) break;
  }
}

// Time budget: keep bounded for scalability.
let timeBudget = 0;
if (n <= 500) timeBudget = 600;
else if (n <= 2000) timeBudget = 800;
else if (n <= 8000) timeBudget = 700;
else timeBudget = 250;

twoOptImprovePath(route, timeBudget);

const outRoute: any[] = new Array(n);
for (let i = 0; i < n; i++) outRoute[i] = ids[route[i + 1]];

process.stdout.write(JSON.stringify({ routes: [outRoute] }));
