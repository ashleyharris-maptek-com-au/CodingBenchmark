using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    private readonly struct Placement
    {
        public readonly int Index;
        public readonly int X;
        public readonly int Y;
        public readonly int Z;

        public Placement(int index, int x, int y, int z)
        {
            Index = index;
            X = x;
            Y = y;
            Z = z;
        }
    }

    private readonly struct Box
    {
        public readonly int Index;
        public readonly int W;
        public readonly int H;
        public readonly int D;
        public readonly long Volume;
        public readonly int MaxDim;
        public readonly int SumDims;
        public readonly long PotentialCount;

        public Box(int index, int w, int h, int d, int containerW, int containerH, int containerD)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
            Volume = (long)w * h * d;
            MaxDim = Math.Max(w, Math.Max(h, d));
            SumDims = w + h + d;

            long cw = w > 0 ? containerW / w : 0;
            long ch = h > 0 ? containerH / h : 0;
            long cd = d > 0 ? containerD / d : 0;
            long pc = cw * ch;
            pc = pc > 0 ? pc * cd : 0;
            PotentialCount = pc;
        }
    }

    private struct Row
    {
        public int Start;    // Start coordinate along the row axis (z when !swapXZ, x when swapXZ)
        public int Thickness; // Depth/width reserved for the row (d when !swapXZ, w when swapXZ)
        public int Cursor;   // Next coordinate along the packing axis within the row (x when !swapXZ, z when swapXZ)
    }

    private struct Level
    {
        public int Y;
        public int Height;
        public int RowStartIndex;
        public int RowCount;
        public int NextStart; // Next start coordinate for a new row (z when !swapXZ, x when swapXZ)
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _pos;

        public FastScanner(Stream stream) => _stream = stream;

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= ' '); // skip whitespace

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > ' ')
            {
                value = (value * 10) + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    private enum SortMode
    {
        VolumeAsc,
        PotentialDesc,
        MaxDimAsc,
        SumDimsAsc
    }

    private sealed class BoxComparer : IComparer<Box>
    {
        private readonly SortMode _mode;
        public BoxComparer(SortMode mode) => _mode = mode;

        public int Compare(Box a, Box b)
        {
            // Primary goal: maximize packed count. Favor "smaller/easier" items first in most modes.
            int c;
            switch (_mode)
            {
                case SortMode.PotentialDesc:
                    c = b.PotentialCount.CompareTo(a.PotentialCount);
                    if (c != 0) return c;
                    c = a.Volume.CompareTo(b.Volume);
                    if (c != 0) return c;
                    c = a.MaxDim.CompareTo(b.MaxDim);
                    if (c != 0) return c;
                    c = a.SumDims.CompareTo(b.SumDims);
                    if (c != 0) return c;
                    return a.Index.CompareTo(b.Index);

                case SortMode.MaxDimAsc:
                    c = a.MaxDim.CompareTo(b.MaxDim);
                    if (c != 0) return c;
                    c = a.SumDims.CompareTo(b.SumDims);
                    if (c != 0) return c;
                    c = a.Volume.CompareTo(b.Volume);
                    if (c != 0) return c;
                    return a.Index.CompareTo(b.Index);

                case SortMode.SumDimsAsc:
                    c = a.SumDims.CompareTo(b.SumDims);
                    if (c != 0) return c;
                    c = a.MaxDim.CompareTo(b.MaxDim);
                    if (c != 0) return c;
                    c = a.Volume.CompareTo(b.Volume);
                    if (c != 0) return c;
                    return a.Index.CompareTo(b.Index);

                case SortMode.VolumeAsc:
                default:
                    c = a.Volume.CompareTo(b.Volume);
                    if (c != 0) return c;
                    c = a.MaxDim.CompareTo(b.MaxDim);
                    if (c != 0) return c;
                    c = a.SumDims.CompareTo(b.SumDims);
                    if (c != 0) return c;
                    return a.Index.CompareTo(b.Index);
            }
        }
    }

    private static List<Placement> PackLimitedFirstFit(
        Box[] orderedBoxes,
        int containerW,
        int containerH,
        int containerD,
        bool swapXZ,
        int maxLevelLookback,
        int maxRowLookback,
        Stopwatch sw,
        long timeLimitMs)
    {
        int n = orderedBoxes.Length;
        var placements = new List<Placement>(Math.Min(n, 1 << 14));

        // Rows and levels are bounded by number of packed items.
        var levels = new Level[n];
        var rows = new Row[n];
        int levelCount = 0;
        int rowCount = 0;

        int nextLevelY = 0;

        int baseW = swapXZ ? containerD : containerW;
        int baseD = swapXZ ? containerW : containerD;

        for (int i = 0; i < n; i++)
        {
            if (sw.ElapsedMilliseconds >= timeLimitMs) break;

            Box box = orderedBoxes[i];
            int bw = swapXZ ? box.D : box.W;
            int bd = swapXZ ? box.W : box.D;
            int bh = box.H;

            if (bw <= 0 || bd <= 0 || bh <= 0) continue;
            if (bw > baseW || bd > baseD || bh > containerH) continue;

            bool placed = false;

            int liStart = levelCount - 1;
            int liStop = 0;
            if (maxLevelLookback != int.MaxValue)
                liStop = Math.Max(0, levelCount - maxLevelLookback);

            for (int li = liStart; li >= liStop && !placed; li--)
            {
                ref Level lvl = ref levels[li];
                if (bh > lvl.Height) continue;

                int rStart = lvl.RowStartIndex;
                int rCount = lvl.RowCount;

                int riStart = rStart + rCount - 1;
                int riStop = rStart;
                if (maxRowLookback != int.MaxValue)
                    riStop = Math.Max(rStart, rStart + rCount - maxRowLookback);

                // Try existing rows (recent-first).
                for (int ri = riStart; ri >= riStop; ri--)
                {
                    ref Row row = ref rows[ri];
                    if (bd > row.Thickness) continue;
                    if (row.Cursor + bw > baseW) continue;

                    int packX = row.Cursor;
                    int packZ = row.Start;
                    row.Cursor += bw;

                    int outX = swapXZ ? packZ : packX;
                    int outZ = swapXZ ? packX : packZ;
                    placements.Add(new Placement(box.Index, outX, lvl.Y, outZ));
                    placed = true;
                    break;
                }

                if (placed) break;

                // Try a new row in this level.
                if (lvl.NextStart + bd <= baseD && rowCount < rows.Length)
                {
                    int start = lvl.NextStart;
                    rows[rowCount] = new Row { Start = start, Thickness = bd, Cursor = bw };
                    lvl.RowCount++;
                    lvl.NextStart += bd;
                    rowCount++;

                    int packX = 0;
                    int packZ = start;
                    int outX = swapXZ ? packZ : packX;
                    int outZ = swapXZ ? packX : packZ;
                    placements.Add(new Placement(box.Index, outX, lvl.Y, outZ));
                    placed = true;
                    break;
                }
            }

            if (placed) continue;

            // New level.
            if (nextLevelY + bh > containerH || levelCount >= levels.Length) continue;

            int levelIndex = levelCount++;
            levels[levelIndex] = new Level
            {
                Y = nextLevelY,
                Height = bh,
                RowStartIndex = rowCount,
                RowCount = 0,
                NextStart = 0
            };

            nextLevelY += bh;

            ref Level newLvl = ref levels[levelIndex];
            if (rowCount >= rows.Length) continue;
            if (bd > baseD) continue;

            rows[rowCount] = new Row { Start = 0, Thickness = bd, Cursor = bw };
            newLvl.RowCount++;
            newLvl.NextStart = bd;
            rowCount++;

            int outX2 = swapXZ ? 0 : 0;
            int outZ2 = swapXZ ? 0 : 0;
            placements.Add(new Placement(box.Index, outX2, newLvl.Y, outZ2));
        }

        return placements;
    }

    public static void Main()
    {
        var sw = Stopwatch.StartNew();

        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue)
            return;

        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();

        if (n <= 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var boxesAll = new Box[n];
        int fitCount = 0;
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();

            // Keep all boxes; filtering happens in packer (dims <= 0 or too large).
            boxesAll[fitCount++] = new Box(i, w, h, d, W, H, D);
        }

        if (fitCount == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Parameters: bounded lookback for scalability; full first-fit for small instances.
        int maxLevelLookback = fitCount <= 200 ? int.MaxValue : (fitCount <= 2000 ? 64 : 16);
        int maxRowLookback = fitCount <= 200 ? int.MaxValue : (fitCount <= 2000 ? 128 : 32);

        // Run multiple fast heuristics and keep the best by packed count.
        // Leave a small buffer before the hard 30s limit.
        const long hardLimitMs = 29500;

        List<Placement>? best = null;
        int bestCount = -1;

        Span<(SortMode mode, bool swap)> strategies = stackalloc (SortMode, bool)[]
        {
            (SortMode.PotentialDesc, false),
            (SortMode.PotentialDesc, true),
            (SortMode.VolumeAsc, false),
            (SortMode.VolumeAsc, true),
            (SortMode.MaxDimAsc, false),
            (SortMode.SumDimsAsc, false),
        };

        for (int s = 0; s < strategies.Length; s++)
        {
            if (sw.ElapsedMilliseconds >= hardLimitMs) break;

            var (mode, swap) = strategies[s];
            var comparer = new BoxComparer(mode);

            var ordered = new Box[fitCount];
            Array.Copy(boxesAll, ordered, fitCount);
            Array.Sort(ordered, comparer);

            // Split time fairly; if small, allow more.
            long remaining = hardLimitMs - sw.ElapsedMilliseconds;
            long perTry = fitCount <= 500 ? remaining : Math.Min(remaining, 4000);
            long until = sw.ElapsedMilliseconds + Math.Max(500, perTry);

            var placements = PackLimitedFirstFit(
                ordered,
                W, H, D,
                swap,
                maxLevelLookback,
                maxRowLookback,
                sw,
                until);

            if (placements.Count > bestCount)
            {
                bestCount = placements.Count;
                best = placements;
            }
        }

        best ??= new List<Placement>(0);

        var sb = new StringBuilder(best.Count * 24 + 32);
        sb.Append(best.Count).Append('\n');
        for (int i = 0; i < best.Count; i++)
        {
            var p = best[i];
            sb.Append(p.Index).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }
}
