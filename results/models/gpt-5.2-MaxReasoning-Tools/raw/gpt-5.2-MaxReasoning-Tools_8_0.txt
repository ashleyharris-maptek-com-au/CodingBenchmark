use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, BufRead, BufReader, Write};

#[derive(Clone)]
struct BitSet {
    words: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        let words = (bits + 63) / 64;
        Self {
            words: vec![0u64; words],
        }
    }

    #[inline]
    fn get(&self, bit: usize) -> bool {
        let word = bit >> 6;
        let mask = 1u64 << (bit & 63);
        (self.words[word] & mask) != 0
    }

    #[inline]
    fn set(&mut self, bit: usize) {
        let word = bit >> 6;
        self.words[word] |= 1u64 << (bit & 63);
    }

    #[inline]
    fn test_and_set(&mut self, bit: usize) -> bool {
        let word = bit >> 6;
        let mask = 1u64 << (bit & 63);
        let prev = (self.words[word] & mask) != 0;
        self.words[word] |= mask;
        prev
    }
}

struct TwoBitVec {
    words: Vec<u64>,
}

impl TwoBitVec {
    fn new(len: usize) -> Self {
        // 2 bits per entry
        let bits = len.saturating_mul(2);
        let words = (bits + 63) / 64;
        Self {
            words: vec![0u64; words],
        }
    }

    #[inline]
    fn set(&mut self, idx: usize, val: u8) {
        let bit = idx * 2;
        let word = bit >> 6;
        let shift = bit & 63; // always even
        let mask = 0b11u64 << shift;
        self.words[word] = (self.words[word] & !mask) | ((val as u64) << shift);
    }

    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let bit = idx * 2;
        let word = bit >> 6;
        let shift = bit & 63;
        ((self.words[word] >> shift) & 0b11) as u8
    }
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapItem {
    f: u32,
    h: u32,
    g: u32,
    idx: u32,
    x: u16,
    y: u16,
}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        // Reverse ordering so BinaryHeap pops smallest f/h/g/idx first.
        other
            .f
            .cmp(&self.f)
            .then_with(|| other.h.cmp(&self.h))
            .then_with(|| other.g.cmp(&self.g))
            .then_with(|| other.idx.cmp(&self.idx))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

#[inline]
fn trim_newline(mut v: Vec<u8>) -> Vec<u8> {
    while matches!(v.last(), Some(b'\n' | b'\r')) {
        v.pop();
    }
    v
}

fn parse_hw(line: &[u8]) -> Option<(usize, usize)> {
    let mut out = [0usize; 2];
    let mut found = 0usize;
    let mut i = 0usize;
    while i < line.len() && found < 2 {
        while i < line.len() && line[i].is_ascii_whitespace() {
            i += 1;
        }
        if i >= line.len() {
            break;
        }
        let mut val = 0usize;
        let mut any = false;
        while i < line.len() && line[i].is_ascii_digit() {
            val = val * 10 + (line[i] - b'0') as usize;
            i += 1;
            any = true;
        }
        if any {
            out[found] = val;
            found += 1;
        } else {
            i += 1;
        }
    }
    if found == 2 {
        Some((out[0], out[1]))
    } else {
        None
    }
}

#[inline]
fn manhattan(x: u16, y: u16, gx: u16, gy: u16) -> u32 {
    x.abs_diff(gx) as u32 + y.abs_diff(gy) as u32
}

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());

    let mut first = Vec::<u8>::new();
    if reader.read_until(b'\n', &mut first).unwrap_or(0) == 0 {
        return;
    }
    let first = trim_newline(first);
    let (h, w) = match parse_hw(&first) {
        Some(v) => v,
        None => return,
    };
    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let cells = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut walls = BitSet::new(cells);
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    let mut line = Vec::<u8>::new();
    for y in 0..h {
        line.clear();
        if reader.read_until(b'\n', &mut line).unwrap_or(0) == 0 {
            break;
        }
        let mut row = trim_newline(std::mem::take(&mut line));
        if row.len() < w {
            row.resize(w, b' ');
        }
        // If longer, only first W are significant.
        let base = y * w;
        for x in 0..w {
            let idx = base + x;
            match row.get(x).copied().unwrap_or(b' ') {
                b'#' => walls.set(idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
        line = row; // reuse allocation
    }

    let Some(start_idx) = start else {
        println!("0");
        return;
    };
    let Some(goal_idx) = goal else {
        println!("0");
        return;
    };

    if start_idx == goal_idx {
        println!("1");
        println!("{} {}", start_idx % w, start_idx / w);
        return;
    }

    let mut visited = BitSet::new(cells);
    let mut parent = TwoBitVec::new(cells);
    visited.set(start_idx);

    const BFS_THRESHOLD: usize = 10_000_000; // cells

    let found = if cells <= BFS_THRESHOLD {
        let mut queue: Vec<u32> = Vec::new();
        queue.push(start_idx as u32);
        let mut head = 0usize;
        let mut ok = false;

        'bfs: while head < queue.len() {
            let cur = queue[head] as usize;
            head += 1;

            let x = cur % w;
            let y = cur / w;

            // Up (parent is Down)
            if y > 0 {
                let n = cur - w;
                if !walls.get(n) && !visited.test_and_set(n) {
                    parent.set(n, 2);
                    if n == goal_idx {
                        ok = true;
                        break 'bfs;
                    }
                    queue.push(n as u32);
                }
            }
            // Right (parent is Left)
            if x + 1 < w {
                let n = cur + 1;
                if !walls.get(n) && !visited.test_and_set(n) {
                    parent.set(n, 3);
                    if n == goal_idx {
                        ok = true;
                        break 'bfs;
                    }
                    queue.push(n as u32);
                }
            }
            // Down (parent is Up)
            if y + 1 < h {
                let n = cur + w;
                if !walls.get(n) && !visited.test_and_set(n) {
                    parent.set(n, 0);
                    if n == goal_idx {
                        ok = true;
                        break 'bfs;
                    }
                    queue.push(n as u32);
                }
            }
            // Left (parent is Right)
            if x > 0 {
                let n = cur - 1;
                if !walls.get(n) && !visited.test_and_set(n) {
                    parent.set(n, 1);
                    if n == goal_idx {
                        ok = true;
                        break 'bfs;
                    }
                    queue.push(n as u32);
                }
            }
        }
        ok
    } else {
        // A* (with early exit on goal discovery); parent set on discovery so any found path is valid.
        let gx = (goal_idx % w) as u16;
        let gy = (goal_idx / w) as u16;
        let sx = (start_idx % w) as u16;
        let sy = (start_idx / w) as u16;

        let mut heap = BinaryHeap::<HeapItem>::new();
        let h0 = manhattan(sx, sy, gx, gy);
        heap.push(HeapItem {
            f: h0,
            h: h0,
            g: 0,
            idx: start_idx as u32,
            x: sx,
            y: sy,
        });

        let mut ok = false;
        'astar: while let Some(cur) = heap.pop() {
            let idx = cur.idx as usize;
            if idx == goal_idx {
                ok = true;
                break 'astar;
            }
            let x = cur.x as usize;
            let y = cur.y as usize;
            let g = cur.g;

            // Expand neighbors in an order that tends to move toward the goal.
            // We'll compute preferred direction order based on the sign of (gx-x, gy-y).
            let dx = gx as i32 - x as i32;
            let dy = gy as i32 - y as i32;

            // 0=Up, 1=Right, 2=Down, 3=Left (movement from current to neighbor)
            let dirs: [u8; 4] = if dx.abs() >= dy.abs() {
                // Horizontal priority
                if dx >= 0 {
                    [1, 2, 0, 3]
                } else {
                    [3, 2, 0, 1]
                }
            } else {
                // Vertical priority
                if dy >= 0 {
                    [2, 1, 3, 0]
                } else {
                    [0, 1, 3, 2]
                }
            };

            for d in dirs {
                let (nidx, nx, ny, parent_dir): (usize, u16, u16, u8) = match d {
                    0 if y > 0 => (idx - w, cur.x, (cur.y - 1), 2),     // went Up; parent is Down
                    1 if x + 1 < w => (idx + 1, (cur.x + 1), cur.y, 3), // went Right; parent is Left
                    2 if y + 1 < h => (idx + w, cur.x, (cur.y + 1), 0), // went Down; parent is Up
                    3 if x > 0 => (idx - 1, (cur.x - 1), cur.y, 1),     // went Left; parent is Right
                    _ => continue,
                };

                if walls.get(nidx) || visited.test_and_set(nidx) {
                    continue;
                }
                parent.set(nidx, parent_dir);

                if nidx == goal_idx {
                    ok = true;
                    break 'astar;
                }

                let ng = g + 1;
                let nh = manhattan(nx, ny, gx, gy);
                let nf = ng + nh;
                heap.push(HeapItem {
                    f: nf,
                    h: nh,
                    g: ng,
                    idx: nidx as u32,
                    x: nx,
                    y: ny,
                });
            }
        }
        ok
    };

    if !found {
        println!("0");
        return;
    }

    // Reconstruct path from goal -> start using 2-bit parent directions.
    let mut path: Vec<u32> = Vec::new();
    let mut cur = goal_idx;
    path.push(cur as u32);
    while cur != start_idx {
        let dir = parent.get(cur);
        cur = match dir {
            0 => cur - w,     // parent is Up
            1 => cur + 1,     // parent is Right
            2 => cur + w,     // parent is Down
            3 => cur - 1,     // parent is Left
            _ => {
                println!("0");
                return;
            }
        };
        path.push(cur as u32);
    }
    path.reverse();

    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());
    writeln!(out, "{}", path.len()).ok();
    for idx in path {
        let idx = idx as usize;
        let x = idx % w;
        let y = idx / w;
        writeln!(out, "{} {}", x, y).ok();
    }
}
