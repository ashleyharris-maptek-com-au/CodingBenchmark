static inline int ctz64(uint64_t x) {
#ifdef _MSC_VER
  unsigned long idx = 0;
  _BitScanForward64(&idx, x);
  return static_cast<int>(idx);
#else
  return __builtin_ctzll(x);
#endif
}

static inline bool any_bits(const vector<uint64_t>& b) {
  for (uint64_t x : b) {
    if (x) return true;
  }
  return false;
}

static inline int first_bit(const vector<uint64_t>& b) {
  for (int i = 0; i < (int)b.size(); ++i) {
    uint64_t x = b[i];
    if (x) return i * 64 + ctz64(x);
  }
  return -1;
}

static inline void clear_bit(vector<uint64_t>& b, int idx) {
  b[idx >> 6] &= ~(1ULL << (idx & 63));
}

static inline vector<uint64_t> and_bits(const vector<uint64_t>& a, const vector<uint64_t>& b) {
  vector<uint64_t> r(a.size());
  for (int i = 0; i < (int)a.size(); ++i) r[i] = a[i] & b[i];
  return r;
}

struct MaxCliqueBBMC {
  int n = 0;
  int words = 0;
  vector<vector<uint64_t>> adj;
  vector<vector<uint64_t>> invAdj;
  vector<int> best;

  void colorOrder(const vector<uint64_t>& P, vector<int>& order, vector<int>& colors) const {
    order.clear();
    colors.clear();
    vector<uint64_t> U = P;
    int color = 0;
    while (any_bits(U)) {
      ++color;
      vector<uint64_t> Q = U;
      while (any_bits(Q)) {
        int v = first_bit(Q);
        order.push_back(v);
        colors.push_back(color);
        clear_bit(U, v);
        clear_bit(Q, v);
        const auto& invN = invAdj[v];
        for (int i = 0; i < words; ++i) Q[i] &= invN[i];
      }
    }
  }

  void expand(vector<uint64_t> P, vector<int>& C) {
    vector<int> order, colors;
    colorOrder(P, order, colors);
    for (int i = (int)order.size() - 1; i >= 0; --i) {
      if ((int)C.size() + colors[i] <= (int)best.size()) return;
      int v = order[i];
      C.push_back(v);

      vector<uint64_t> newP = and_bits(P, adj[v]);
      if (any_bits(newP)) {
        expand(std::move(newP), C);
      } else {
        if ((int)C.size() > (int)best.size()) best = C;
      }

      C.pop_back();
      clear_bit(P, v);
    }
  }

  // Returns a maximum clique as vertex indices [0..n-1].
  vector<int> solve(int n_, vector<vector<uint64_t>> adj_) {
    n = n_;
    adj = std::move(adj_);
    words = (n + 63) / 64;
    invAdj.assign(n, vector<uint64_t>(words, 0));
    uint64_t lastMask = (n % 64 == 0) ? ~0ULL : ((1ULL << (n % 64)) - 1ULL);
    for (int v = 0; v < n; ++v) {
      for (int i = 0; i < words; ++i) invAdj[v][i] = ~adj[v][i];
      invAdj[v][words - 1] &= lastMask;
    }

    best.clear();
    vector<int> C;

    vector<uint64_t> P(words, ~0ULL);
    P[words - 1] &= lastMask;
    expand(std::move(P), C);
    return best;
  }
};

static vector<int> solve_tree_mis(const vector<vector<int>>& g) {
  int n = (int)g.size();
  vector<int> parent(n, -1);
  vector<char> visited(n, 0);
  vector<int> order;
  order.reserve(n);

  // Build parent pointers and traversal order (iterative DFS).
  vector<int> st;
  st.reserve(n);
  st.push_back(0);
  visited[0] = 1;
  parent[0] = -1;
  while (!st.empty()) {
    int v = st.back();
    st.pop_back();
    order.push_back(v);
    for (int u : g[v]) {
      if (visited[u]) continue;
      visited[u] = 1;
      parent[u] = v;
      st.push_back(u);
    }
  }

  vector<int> dp0(n, 0), dp1(n, 0);
  for (int i = (int)order.size() - 1; i >= 0; --i) {
    int v = order[i];
    dp1[v] = 1;
    dp0[v] = 0;
    for (int u : g[v]) {
      if (u == parent[v]) continue;
      dp1[v] += dp0[u];
      dp0[v] += max(dp0[u], dp1[u]);
    }
  }

  vector<char> take(n, 0);
  vector<pair<int, char>> st2;
  st2.reserve(n);
  st2.push_back({0, 0});
  while (!st2.empty()) {
    auto [v, parentTaken] = st2.back();
    st2.pop_back();
    char takeV = 0;
    if (!parentTaken) takeV = (dp1[v] >= dp0[v]) ? 1 : 0;
    take[v] = takeV;
    for (int u : g[v]) {
      if (u == parent[v]) continue;
      st2.push_back({u, takeV});
    }
  }

  vector<int> res;
  for (int i = 0; i < n; ++i) {
    if (take[i]) res.push_back(i);
  }
  return res;
}

static bool bipartite_color(const vector<vector<int>>& g, vector<int>& color) {
  int n = (int)g.size();
  color.assign(n, -1);
  queue<int> q;
  for (int s = 0; s < n; ++s) {
    if (color[s] != -1) continue;
    color[s] = 0;
    q.push(s);
    while (!q.empty()) {
      int v = q.front();
      q.pop();
      for (int u : g[v]) {
        if (color[u] == -1) {
          color[u] = color[v] ^ 1;
          q.push(u);
        } else if (color[u] == color[v]) {
          return false;
        }
      }
    }
  }
  return true;
}

static vector<int> solve_bipartite_mis(const vector<vector<int>>& g, const vector<int>& color) {
  int n = (int)g.size();
  vector<int> leftVerts, rightVerts;
  leftVerts.reserve(n);
  rightVerts.reserve(n);

  vector<int> leftId(n, -1), rightId(n, -1);
  for (int v = 0; v < n; ++v) {
    if (color[v] == 0) {
      leftId[v] = (int)leftVerts.size();
      leftVerts.push_back(v);
    } else {
      rightId[v] = (int)rightVerts.size();
      rightVerts.push_back(v);
    }
  }

  int L = (int)leftVerts.size();
  int R = (int)rightVerts.size();
  vector<vector<int>> gL(L);
  for (int li = 0; li < L; ++li) {
    int v = leftVerts[li];
    auto& out = gL[li];
    out.reserve(g[v].size());
    for (int u : g[v]) {
      if (color[u] == 1) out.push_back(rightId[u]);
    }
  }

  vector<int> matchL(L, -1), matchR(R, -1), dist(L, -1);

  auto bfs = [&]() -> bool {
    queue<int> q;
    for (int i = 0; i < L; ++i) {
      if (matchL[i] == -1) {
        dist[i] = 0;
        q.push(i);
      } else {
        dist[i] = -1;
      }
    }
    bool found = false;
    while (!q.empty()) {
      int v = q.front();
      q.pop();
      for (int r : gL[v]) {
        int u = matchR[r];
        if (u == -1) {
          found = true;
        } else if (dist[u] == -1) {
          dist[u] = dist[v] + 1;
          q.push(u);
        }
      }
    }
    return found;
  };

  function<bool(int)> dfs = [&](int v) -> bool {
    for (int r : gL[v]) {
      int u = matchR[r];
      if (u == -1 || (dist[u] == dist[v] + 1 && dfs(u))) {
        matchL[v] = r;
        matchR[r] = v;
        return true;
      }
    }
    dist[v] = -1;
    return false;
  };

  while (bfs()) {
    for (int i = 0; i < L; ++i) {
      if (matchL[i] == -1) dfs(i);
    }
  }

  // Derive min vertex cover via alternating paths from free left vertices.
  vector<char> visL(L, 0), visR(R, 0);
  queue<int> q;
  for (int i = 0; i < L; ++i) {
    if (matchL[i] == -1) q.push(i);
  }
  while (!q.empty()) {
    int li = q.front();
    q.pop();
    if (visL[li]) continue;
    visL[li] = 1;
    for (int r : gL[li]) {
      if (matchL[li] == r) continue;  // only traverse non-matching edges L->R
      if (visR[r]) continue;
      visR[r] = 1;
      int li2 = matchR[r];  // traverse matching edge R->L
      if (li2 != -1 && !visL[li2]) q.push(li2);
    }
  }

  // Maximum independent set = (Left visited) ∪ (Right not visited).
  vector<int> indep;
  indep.reserve(n);
  for (int li = 0; li < L; ++li) {
    if (visL[li]) indep.push_back(leftVerts[li]);
  }
  for (int ri = 0; ri < R; ++ri) {
    if (!visR[ri]) indep.push_back(rightVerts[ri]);
  }
  return indep;
}

static vector<int> solve_general_mis_via_clique_complement(const vector<vector<int>>& g) {
  int n = (int)g.size();
  if (n == 0) return {};
  if (n == 1) return {0};

  // Order vertices by increasing degree in original graph (decreasing degree in complement).
  vector<int> perm(n);
  for (int i = 0; i < n; ++i) perm[i] = i;
  vector<int> deg(n);
  for (int i = 0; i < n; ++i) deg[i] = (int)g[i].size();
  stable_sort(perm.begin(), perm.end(), [&](int a, int b) {
    if (deg[a] != deg[b]) return deg[a] < deg[b];
    return a < b;
  });

  vector<int> oldToNew(n, -1), newToOld(n, -1);
  for (int i = 0; i < n; ++i) {
    oldToNew[perm[i]] = i;
    newToOld[i] = perm[i];
  }

  int words = (n + 63) / 64;
  uint64_t lastMask = (n % 64 == 0) ? ~0ULL : ((1ULL << (n % 64)) - 1ULL);
  vector<vector<uint64_t>> compAdj(n, vector<uint64_t>(words, ~0ULL));
  for (int v = 0; v < n; ++v) compAdj[v][words - 1] &= lastMask;

  for (int v = 0; v < n; ++v) {
    clear_bit(compAdj[v], v);
  }

  // Clear edges that exist in the original graph (in permuted indexing) to build complement.
  for (int ov = 0; ov < n; ++ov) {
    int v = oldToNew[ov];
    for (int ou : g[ov]) {
      int u = oldToNew[ou];
      clear_bit(compAdj[v], u);
    }
  }

  MaxCliqueBBMC solver;
  vector<int> clique = solver.solve(n, std::move(compAdj));

  vector<int> indep;
  indep.reserve(clique.size());
  for (int vNew : clique) indep.push_back(newToOld[vNew]);
  sort(indep.begin(), indep.end());
  indep.erase(unique(indep.begin(), indep.end()), indep.end());
  return indep;
}

static vector<int> solve_component_mis(const vector<int>& compVerts, const vector<vector<int>>& adjGlobal,
                                      vector<int>& tmpGlobalToLocal) {
  int s = (int)compVerts.size();
  if (s == 0) return {};
  if (s == 1) return {compVerts[0]};

  // Build local mapping and adjacency.
  for (int i = 0; i < s; ++i) tmpGlobalToLocal[compVerts[i]] = i;
  vector<vector<int>> g(s);
  long long twiceEdges = 0;
  for (int i = 0; i < s; ++i) {
    int gv = compVerts[i];
    g[i].reserve(adjGlobal[gv].size());
    for (int gu : adjGlobal[gv]) {
      int j = tmpGlobalToLocal[gu];
      if (j != -1) g[i].push_back(j);
    }
    twiceEdges += (long long)g[i].size();
  }
  long long edges = twiceEdges / 2;

  // Reset mapping for reuse before returning.
  auto resetMapping = [&]() {
    for (int v : compVerts) tmpGlobalToLocal[v] = -1;
  };

  vector<int> localResult;

  // Connected component with m = n - 1 is a tree (in simple graphs); treat as tree for speed.
  if (edges == s - 1) {
    vector<int> chosenLocal = solve_tree_mis(g);
    localResult = std::move(chosenLocal);
  } else {
    vector<int> color;
    if (bipartite_color(g, color)) {
      vector<int> chosenLocal = solve_bipartite_mis(g, color);
      localResult = std::move(chosenLocal);
    } else {
      vector<int> chosenLocal = solve_general_mis_via_clique_complement(g);
      localResult = std::move(chosenLocal);
    }
  }

  vector<int> globalResult;
  globalResult.reserve(localResult.size());
  for (int li : localResult) globalResult.push_back(compVerts[li]);
  resetMapping();
  return globalResult;
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n = 0, m = 0;
  if (!(cin >> n >> m)) return 0;

  vector<char> forbidden(n, 0);
  vector<pair<int, int>> edges;
  edges.reserve(m);
  for (int i = 0; i < m; ++i) {
    int u, v;
    cin >> u >> v;
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    if (u == v) {
      forbidden[u] = 1;
      continue;
    }
    edges.push_back({u, v});
  }

  vector<int> oldToNew(n, -1);
  vector<int> newToOld;
  newToOld.reserve(n);
  for (int i = 0; i < n; ++i) {
    if (!forbidden[i]) {
      oldToNew[i] = (int)newToOld.size();
      newToOld.push_back(i);
    }
  }

  int nA = (int)newToOld.size();
  if (nA == 0) {
    cout << 0 << "\n\n";
    return 0;
  }

  vector<vector<int>> adj(nA);
  for (auto [u0, v0] : edges) {
    int u = oldToNew[u0], v = oldToNew[v0];
    if (u == -1 || v == -1) continue;
    adj[u].push_back(v);
    adj[v].push_back(u);
  }

  // Find connected components in the active graph.
  vector<int> compId(nA, -1);
  vector<vector<int>> comps;
  comps.reserve(nA);
  queue<int> q;
  for (int i = 0; i < nA; ++i) {
    if (compId[i] != -1) continue;
    int cid = (int)comps.size();
    comps.push_back({});
    compId[i] = cid;
    q.push(i);
    while (!q.empty()) {
      int v = q.front();
      q.pop();
      comps[cid].push_back(v);
      for (int u : adj[v]) {
        if (compId[u] == -1) {
          compId[u] = cid;
          q.push(u);
        }
      }
    }
  }

  vector<int> tmpGlobalToLocal(nA, -1);
  vector<int> chosenActive;
  chosenActive.reserve(nA);
  for (const auto& comp : comps) {
    vector<int> part = solve_component_mis(comp, adj, tmpGlobalToLocal);
    chosenActive.insert(chosenActive.end(), part.begin(), part.end());
  }

  // Map back to original indices, excluding forbidden vertices.
  vector<int> answer;
  answer.reserve(chosenActive.size());
  for (int vA : chosenActive) answer.push_back(newToOld[vA]);
  sort(answer.begin(), answer.end());
  answer.erase(unique(answer.begin(), answer.end()), answer.end());

  cout << answer.size() << "\n";
  for (size_t i = 0; i < answer.size(); ++i) {
    if (i) cout << ' ';
    cout << answer[i];
  }
  cout << "\n";
  return 0;
}
