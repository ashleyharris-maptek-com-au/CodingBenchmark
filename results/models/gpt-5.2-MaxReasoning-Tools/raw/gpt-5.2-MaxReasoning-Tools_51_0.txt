import math
from typing import Dict, Tuple, Optional, List


# Persistent autopilot state across calls (allowed by prompt).
_AP = {
    "init": False,
    "last": {},  # fused sensor history + last commands
    "i": {  # integrators
        "alt_ft_s": 0.0,
        "spd_kt_s": 0.0,
        "hdg_deg_s": 0.0,
    },
}


def _clamp(x: float, lo: float, hi: float) -> float:
    if x < lo:
        return lo
    if x > hi:
        return hi
    return x


def _is_finite(x) -> bool:
    try:
        return math.isfinite(float(x))
    except Exception:
        return False


def _wrap180(deg: float) -> float:
    # returns in (-180, 180]
    return ((deg + 180.0) % 360.0) - 180.0


def _unwrap_to_ref(deg: float, ref_deg: float) -> float:
    return ref_deg + _wrap180(deg - ref_deg)


def _median(vals: List[float]) -> float:
    s = sorted(vals)
    n = len(s)
    if n == 0:
        return float("nan")
    mid = n // 2
    if n % 2 == 1:
        return s[mid]
    return 0.5 * (s[mid - 1] + s[mid])


def _rate_limit(new: float, old: float, rate_per_s: float, dt: float) -> float:
    if old is None or not _is_finite(old):
        return new
    max_step = abs(rate_per_s) * max(1e-3, dt)
    return old + _clamp(new - old, -max_step, max_step)


def _fuse_linear(
    cands: List[Tuple[float, bool]],
    last: Optional[float],
    med_scale: float,
    last_scale: float,
) -> Tuple[Optional[float], float, float]:
    """
    Returns (fused_value, quality[0..1], spread).
    - Uses a soft score based on agreement with median and last value.
    - Robust to one outlier; with two faults, it tends to follow temporal continuity.
    """
    vals: List[Tuple[float, bool]] = []
    for v, valid in cands:
        if v is None or not _is_finite(v):
            continue
        vals.append((float(v), bool(valid)))
    if not vals:
        return None, 0.0, float("inf")

    only_vals = [v for v, _ in vals]
    med = _median(only_vals)
    spread = max(only_vals) - min(only_vals) if len(only_vals) >= 2 else 0.0

    scored: List[Tuple[float, float]] = []
    for v, valid in vals:
        s = 1.0
        if not valid:
            s *= 0.25
        s *= 1.0 / (1.0 + abs(v - med) / max(1e-6, med_scale))
        if last is not None and _is_finite(last):
            s *= 1.0 / (1.0 + abs(v - float(last)) / max(1e-6, last_scale))
        scored.append((s, v))

    total = sum(s for s, _ in scored)
    if total <= 1e-9:
        # Fallback to median if everything scored poorly.
        return med, 0.05, spread
    fused = sum(s * v for s, v in scored) / total

    # Quality: prefer tight agreement and at least one high score.
    best = max(s for s, _ in scored)
    quality = _clamp(best, 0.0, 1.0)
    if len(vals) >= 2:
        quality *= 1.0 / (1.0 + (spread / max(1e-6, 3.0 * med_scale)))
    return fused, _clamp(quality, 0.0, 1.0), spread


def _fuse_angle_deg(
    cands: List[Tuple[float, bool]],
    last: Optional[float],
    med_scale: float,
    last_scale: float,
) -> Tuple[Optional[float], float, float]:
    # Unwrap around last (if available) then fuse as linear.
    unwrapped: List[Tuple[float, bool]] = []
    ref = None
    if last is not None and _is_finite(last):
        ref = float(last) % 360.0
    for v, valid in cands:
        if v is None or not _is_finite(v):
            continue
        a = float(v) % 360.0
        if ref is not None:
            a = _unwrap_to_ref(a, ref)
        unwrapped.append((a, bool(valid)))
    fused, q, spread = _fuse_linear(unwrapped, ref, med_scale=med_scale, last_scale=last_scale)
    if fused is None:
        return None, 0.0, float("inf")
    return float(fused) % 360.0, q, spread


def _triplet(state: dict, base: str, suffix: str, valid_suffix: str = "_valid") -> List[Tuple[float, bool]]:
    out: List[Tuple[float, bool]] = []
    for i in (1, 2, 3):
        v = state.get(f"{base}_{i}{suffix}")
        valid = state.get(f"{base}_{i}{valid_suffix}", True)
        out.append((v, bool(valid)))
    return out


def _est_on_ground(state: dict, radar_agl_ft: Optional[float]) -> bool:
    # Any WOW true or (radar alt very low + gear down) is treated as on-ground.
    if bool(state.get("weight_on_wheels_left")) or bool(state.get("weight_on_wheels_nose")) or bool(
        state.get("weight_on_wheels_right")
    ):
        return True
    if radar_agl_ft is not None and radar_agl_ft < 5.0 and bool(state.get("gear_handle_down", 1)):
        return True
    return False


def autopilot_step(state: dict, dt: float) -> dict:
    """
    Robust A320-class autopilot step for a simulator:
    - Triple-redundant sensor fusion with temporal continuity.
    - Cascaded control for heading/roll and altitude/vertical-speed/pitch.
    - Envelope protections: overspeed + stall/alpha protection.
    """
    global _AP

    # dt hygiene
    if not _is_finite(dt) or dt <= 0.0:
        dt = state.get("dt", 0.05) or 0.05
    try:
        dt = float(dt)
    except Exception:
        dt = 0.05
    dt = _clamp(dt, 0.01, 0.2)

    last = _AP["last"]

    # --- Fuse key sensors ---
    baro_alt, baro_q, baro_spread = _fuse_linear(
        _triplet(state, "baro_alt", "_ft"),
        last.get("baro_alt_ft"),
        med_scale=120.0,
        last_scale=max(30.0, 500.0 * dt),
    )
    gps_alt, gps_alt_q, gps_alt_spread = _fuse_linear(
        [
            (state.get("gps_1_alt_ft"), state.get("gps_1_valid", True)),
            (state.get("gps_2_alt_ft"), state.get("gps_2_valid", True)),
            (state.get("gps_3_alt_ft"), state.get("gps_3_valid", True)),
        ],
        last.get("gps_alt_ft"),
        med_scale=150.0,
        last_scale=max(50.0, 700.0 * dt),
    )

    # Radar altimeters are only meaningful near the ground.
    radar_alt, radar_q, radar_spread = _fuse_linear(
        _triplet(state, "radar_alt", "_ft"),
        last.get("radar_alt_ft"),
        med_scale=25.0,
        last_scale=max(15.0, 400.0 * dt),
    )
    if radar_alt is not None and radar_alt > 3000.0:
        radar_q *= 0.1

    ias, ias_q, ias_spread = _fuse_linear(
        _triplet(state, "ias", "_kt"),
        last.get("ias_kt"),
        med_scale=12.0,
        last_scale=max(8.0, 80.0 * dt),
    )
    mach, mach_q, _ = _fuse_linear(
        [(state.get("mach_1"), True), (state.get("mach_2"), True), (state.get("mach_3"), True)],
        last.get("mach"),
        med_scale=0.03,
        last_scale=max(0.02, 0.3 * dt),
    )
    aoa, aoa_q, aoa_spread = _fuse_linear(
        _triplet(state, "aoa", "_deg"),
        last.get("aoa_deg"),
        med_scale=1.8,
        last_scale=max(1.2, 10.0 * dt),
    )

    hdg, hdg_q, hdg_spread = _fuse_angle_deg(
        [(state.get("irs_1_heading_deg"), state.get("irs_1_valid", True)),
         (state.get("irs_2_heading_deg"), state.get("irs_2_valid", True)),
         (state.get("irs_3_heading_deg"), state.get("irs_3_valid", True))],
        last.get("hdg_deg"),
        med_scale=6.0,
        last_scale=max(4.0, 60.0 * dt),
    )
    # Fallback to nav-provided heading if IRS goes bad.
    nav_hdg = state.get("magnetic_heading_deg")
    if (hdg is None or hdg_q < 0.2) and _is_finite(nav_hdg):
        hdg = float(nav_hdg) % 360.0
        hdg_q = max(hdg_q, 0.25)

    roll, roll_q, _ = _fuse_linear(
        [(state.get("irs_1_roll_deg"), state.get("irs_1_valid", True)),
         (state.get("irs_2_roll_deg"), state.get("irs_2_valid", True)),
         (state.get("irs_3_roll_deg"), state.get("irs_3_valid", True))],
        last.get("roll_deg"),
        med_scale=3.0,
        last_scale=max(2.0, 60.0 * dt),
    )
    pitch, pitch_q, _ = _fuse_linear(
        [(state.get("irs_1_pitch_deg"), state.get("irs_1_valid", True)),
         (state.get("irs_2_pitch_deg"), state.get("irs_2_valid", True)),
         (state.get("irs_3_pitch_deg"), state.get("irs_3_valid", True))],
        last.get("pitch_deg"),
        med_scale=3.0,
        last_scale=max(2.0, 60.0 * dt),
    )

    vs, vs_q, _ = _fuse_linear(
        [(state.get("irs_1_vs_fpm"), state.get("irs_1_valid", True)),
         (state.get("irs_2_vs_fpm"), state.get("irs_2_valid", True)),
         (state.get("irs_3_vs_fpm"), state.get("irs_3_valid", True)),
         (state.get("vertical_speed_fpm"), True)],
        last.get("vs_fpm"),
        med_scale=250.0,
        last_scale=max(200.0, 8000.0 * dt),
    )

    yaw_rate, yaw_q, _ = _fuse_linear(
        [(state.get("gyro_1_yaw_rate_dps"), state.get("gyro_1_valid", True)),
         (state.get("gyro_2_yaw_rate_dps"), state.get("gyro_2_valid", True)),
         (state.get("gyro_3_yaw_rate_dps"), state.get("gyro_3_valid", True))],
        last.get("yaw_rate_dps"),
        med_scale=2.0,
        last_scale=max(1.0, 80.0 * dt),
    )
    roll_rate, rr_q, _ = _fuse_linear(
        [(state.get("gyro_1_roll_rate_dps"), state.get("gyro_1_valid", True)),
         (state.get("gyro_2_roll_rate_dps"), state.get("gyro_2_valid", True)),
         (state.get("gyro_3_roll_rate_dps"), state.get("gyro_3_valid", True))],
        last.get("roll_rate_dps"),
        med_scale=3.0,
        last_scale=max(2.0, 120.0 * dt),
    )
    pitch_rate, pr_q, _ = _fuse_linear(
        [(state.get("gyro_1_pitch_rate_dps"), state.get("gyro_1_valid", True)),
         (state.get("gyro_2_pitch_rate_dps"), state.get("gyro_2_valid", True)),
         (state.get("gyro_3_pitch_rate_dps"), state.get("gyro_3_valid", True))],
        last.get("pitch_rate_dps"),
        med_scale=3.0,
        last_scale=max(2.0, 120.0 * dt),
    )

    # Choose altitude source: prefer baro, but fall back to GPS if baro looks suspect.
    alt_ft = baro_alt
    alt_q = baro_q
    if alt_ft is None and gps_alt is not None:
        alt_ft, alt_q = gps_alt, gps_alt_q
    if alt_ft is not None and gps_alt is not None:
        # Large disagreement => downweight the one that is less temporally consistent.
        if abs(alt_ft - gps_alt) > 800.0:
            if baro_q < gps_alt_q:
                alt_ft, alt_q = gps_alt, gps_alt_q
            else:
                alt_ft, alt_q = baro_alt, baro_q

    # Estimate AGL when possible.
    agl_ft = radar_alt if (radar_alt is not None and radar_q > 0.2) else None
    on_ground = _est_on_ground(state, agl_ft)

    # Determine measurement reliability flags.
    airspeed_reliable = (ias is not None) and (ias_q > 0.25) and (ias_spread < 30.0)
    aoa_reliable = (aoa is not None) and (aoa_q > 0.25) and (aoa_spread < 6.0)
    hdg_reliable = (hdg is not None) and (hdg_q > 0.2) and (hdg_spread < 25.0)
    att_reliable = (roll is not None) and (pitch is not None) and (roll_q > 0.2) and (pitch_q > 0.2)
    alt_reliable = (alt_ft is not None) and (alt_q > 0.2)

    # --- Targets and mode flags ---
    ap_engaged = bool(state.get("ap_engaged", True))
    at_engaged = bool(state.get("at_engaged", True))

    target_alt = state.get("target_alt_ft", alt_ft if alt_ft is not None else 0.0)
    target_hdg = state.get("target_hdg_deg", hdg if hdg is not None else 0.0)
    target_ias = state.get("target_ias_kt", ias if ias is not None else 250.0)
    target_vs = state.get("target_vs_fpm", 0.0)

    target_alt = float(target_alt) if _is_finite(target_alt) else (alt_ft if alt_ft is not None else 0.0)
    target_hdg = float(target_hdg) if _is_finite(target_hdg) else (hdg if hdg is not None else 0.0)
    target_ias = float(target_ias) if _is_finite(target_ias) else (ias if ias is not None else 250.0)
    target_vs = float(target_vs) if _is_finite(target_vs) else 0.0

    # --- Basic flight envelope numbers (conservative) ---
    VNE_KIAS = 365.0
    VMO_KIAS = 350.0
    MMO = 0.82
    gross_wt = float(state.get("gross_weight_kg", 65000.0) or 65000.0)
    flaps_pos = float(state.get("flap_pos_left_deg", state.get("flap_pos_right_deg", 0.0)) or 0.0)
    flap_deg = max(flaps_pos, float(state.get("flap_pos_right_deg", flaps_pos) or flaps_pos))

    w_factor = _clamp((gross_wt - 60000.0) / 20000.0, -0.5, 0.8)
    min_safe_ias = 150.0 + 20.0 * w_factor
    if flap_deg >= 10.0:
        min_safe_ias -= 12.0
    if flap_deg >= 20.0:
        min_safe_ias -= 10.0
    if bool(state.get("gear_handle_down", False)):
        min_safe_ias -= 5.0
    min_safe_ias = _clamp(min_safe_ias, 110.0, 180.0)

    max_safe_ias = min(VMO_KIAS, VNE_KIAS - 10.0)
    if flap_deg >= 10.0:
        max_safe_ias = min(max_safe_ias, 230.0)
    if flap_deg >= 20.0:
        max_safe_ias = min(max_safe_ias, 200.0)

    # Clamp target IAS into a safe region.
    target_ias = _clamp(target_ias, min_safe_ias + 5.0, max_safe_ias)

    # --- Integrators ---
    # Altitude integrator (helps remove steady-state errors)
    alt_error = 0.0
    if alt_ft is not None:
        alt_error = float(target_alt) - float(alt_ft)
    _AP["i"]["alt_ft_s"] = _clamp(_AP["i"]["alt_ft_s"] + alt_error * dt, -20000.0, 20000.0)

    # Speed integrator
    spd_error = 0.0
    if airspeed_reliable:
        spd_error = float(target_ias) - float(ias)
    _AP["i"]["spd_kt_s"] = _clamp(_AP["i"]["spd_kt_s"] + spd_error * dt, -8000.0, 8000.0)

    # Heading integrator (small; mostly to counter steady winds)
    hdg_error = 0.0
    if hdg_reliable:
        hdg_error = _wrap180(float(target_hdg) - float(hdg))
    _AP["i"]["hdg_deg_s"] = _clamp(_AP["i"]["hdg_deg_s"] + hdg_error * dt, -500.0, 500.0)

    # --- Guidance (desired pitch/roll + throttle) ---
    # Vertical: altitude -> VS command (fpm), then VS -> pitch command.
    vs_cmd = 0.0
    if ap_engaged and alt_reliable and vs is not None and _is_finite(vs):
        vs_limit = _clamp(max(600.0, abs(target_vs) if abs(target_vs) > 200.0 else 2000.0), 600.0, 3200.0)
        # If target_vs points away from capturing altitude, ignore it.
        if abs(target_vs) > 50.0 and (target_vs * alt_error) > 0.0:
            vs_limit = max(vs_limit, abs(target_vs))
        kp_alt = 0.55  # fpm per ft
        ki_alt = 0.015  # fpm per (ft*s)
        vs_cmd = kp_alt * alt_error + ki_alt * _AP["i"]["alt_ft_s"]
        vs_cmd = _clamp(vs_cmd, -vs_limit, vs_limit)
        if abs(alt_error) < 80.0:
            vs_cmd = _clamp(vs_cmd, -400.0, 400.0)
        if abs(alt_error) < 25.0:
            vs_cmd = 0.0
    elif ap_engaged and abs(target_vs) > 50.0 and vs is not None and _is_finite(vs):
        vs_cmd = _clamp(target_vs, -2500.0, 2500.0)

    # Pitch command driven by VS error.
    pitch_cmd = 0.0
    if att_reliable and pitch is not None:
        pitch_cmd = float(pitch)
    if ap_engaged and att_reliable and pitch is not None and vs is not None and _is_finite(vs):
        vs_err = float(vs_cmd) - float(vs)
        pitch_cmd = float(pitch) + 0.0025 * vs_err  # deg per fpm
        # Conservative pitch limits (tighten near ground)
        pitch_up_lim = 15.0 if (agl_ft is None or agl_ft > 1000.0) else 12.0
        pitch_dn_lim = -10.0 if (agl_ft is None or agl_ft > 200.0) else -6.0
        pitch_cmd = _clamp(pitch_cmd, pitch_dn_lim, pitch_up_lim)

    # Lateral: heading -> roll command (deg), then roll -> aileron.
    roll_cmd = 0.0
    if att_reliable and roll is not None:
        roll_cmd = float(roll)
    if ap_engaged and hdg_reliable and att_reliable and roll is not None:
        hdg_err = _wrap180(float(target_hdg) - float(hdg))
        # Max bank lower on approach / in turbulence.
        turb = float(state.get("turbulence_intensity", 0.0) or 0.0)
        bank_max = 25.0
        if agl_ft is not None and agl_ft < 1500.0:
            bank_max = 18.0
        bank_max = _clamp(bank_max * (1.0 - 0.35 * _clamp(turb, 0.0, 1.0)), 12.0, 25.0)
        kp_hdg = 0.8
        ki_hdg = 0.002
        roll_cmd = kp_hdg * hdg_err + ki_hdg * _AP["i"]["hdg_deg_s"]
        roll_cmd = _clamp(roll_cmd, -bank_max, bank_max)

    # Throttle: speed hold when reliable, else alpha-based fallback.
    throttle_cmd = last.get("throttle_cmd", 0.5)
    if not _is_finite(throttle_cmd):
        throttle_cmd = 0.5

    if at_engaged:
        if airspeed_reliable:
            kp_spd = 0.010
            ki_spd = 0.0006
            # Small feedforward for climbs/descents to help energy management.
            ff = 0.0
            if vs_cmd > 300.0:
                ff += 0.10
            if vs_cmd < -800.0:
                ff -= 0.10
            throttle_cmd = 0.50 + ff + kp_spd * spd_error + ki_spd * _AP["i"]["spd_kt_s"]
        elif aoa_reliable and aoa is not None:
            # Maintain a moderate AoA when IAS is unreliable.
            aoa_tgt = 4.5 if flap_deg < 10.0 else 6.0
            aoa_err = float(aoa_tgt) - float(aoa)
            throttle_cmd = 0.60 - 0.08 * aoa_err
        else:
            # Last-resort pitch/thrust schedule (unreliable airspeed + AoA).
            if agl_ft is not None and agl_ft < 1500.0:
                throttle_cmd = 0.95
            elif alt_ft is not None and alt_ft < 10000.0:
                throttle_cmd = 0.85
            else:
                throttle_cmd = 0.78

    throttle_cmd = _clamp(float(throttle_cmd), 0.0, 1.0)

    # --- Envelope protections (override/shape commands) ---
    overspeed = False
    if airspeed_reliable and ias is not None and float(ias) > (VMO_KIAS - 3.0):
        overspeed = True
    if mach is not None and mach_q > 0.15 and float(mach) > (MMO - 0.015):
        overspeed = True

    aoa_warn = False
    aoa_stall = False
    if aoa_reliable and aoa is not None:
        aoa_warn = float(aoa) > (12.0 if flap_deg < 10.0 else 14.0)
        aoa_stall = float(aoa) > (15.0 if flap_deg < 10.0 else 17.0)
    low_speed = airspeed_reliable and ias is not None and float(ias) < (min_safe_ias + 2.0)

    # Pitch shaping for stall/overspeed protection.
    if att_reliable and pitch is not None:
        if overspeed:
            # Reduce speed: idle + speedbrake + gentle pitch-up (but not into stall).
            pitch_cmd = max(pitch_cmd, float(pitch) + 2.0)
            throttle_cmd = 0.0 if at_engaged else throttle_cmd
        if aoa_warn or low_speed:
            # Protect alpha: nudge nose down and demand thrust.
            pitch_cmd = min(pitch_cmd, float(pitch) - 2.0)
            if at_engaged:
                throttle_cmd = max(throttle_cmd, 0.85)
        if aoa_stall:
            # Stall recovery: push to reduce AoA, max thrust.
            pitch_cmd = min(pitch_cmd, float(pitch) - 6.0)
            if at_engaged:
                throttle_cmd = 1.0

    # --- Actuators ---
    # Elevator: pitch hold (PID-like with rate damping).
    elevator = 0.0
    if att_reliable and pitch is not None:
        pitch_err = float(pitch_cmd) - float(pitch)
        kp_pitch = 0.09
        kd_pr = 0.03
        pr = float(pitch_rate) if (pitch_rate is not None and _is_finite(pitch_rate)) else 0.0
        elevator = kp_pitch * pitch_err - kd_pr * pr
    elevator = _clamp(elevator, -1.0, 1.0)

    # Aileron: roll hold with roll-rate damping.
    aileron = 0.0
    if att_reliable and roll is not None:
        roll_err = float(roll_cmd) - float(roll)
        kp_roll = 0.055
        kd_rr = 0.022
        rr = float(roll_rate) if (roll_rate is not None and _is_finite(roll_rate)) else 0.0
        aileron = kp_roll * roll_err - kd_rr * rr
    aileron = _clamp(aileron, -1.0, 1.0)
    if on_ground:
        aileron = 0.0

    # Rudder: yaw damper + engine asymmetry compensation.
    rudder = 0.0
    drift = state.get("drift_angle_deg", 0.0) or 0.0
    if yaw_rate is not None and _is_finite(yaw_rate):
        k_yaw = 0.055
        rudder += -k_yaw * float(yaw_rate)
    if _is_finite(drift):
        rudder += -0.02 * float(drift)
    thrust1 = float(state.get("engine_1_thrust_pct", 0.0) or 0.0)
    thrust2 = float(state.get("engine_2_thrust_pct", 0.0) or 0.0)
    rudder += -0.006 * (thrust1 - thrust2)
    rudder = _clamp(rudder, -1.0, 1.0)
    if on_ground:
        rudder = 0.0

    # Speedbrake: use for overspeed or if too fast in descent near idle.
    speedbrake = 0.0
    if overspeed:
        speedbrake = 1.0
    elif at_engaged and airspeed_reliable and ias is not None:
        fast = float(ias) - float(target_ias)
        if fast > 12.0 and throttle_cmd < 0.15 and (agl_ft is None or agl_ft > 800.0):
            speedbrake = _clamp((fast - 12.0) / 30.0, 0.0, 1.0)
    if aoa_warn or low_speed:
        speedbrake = 0.0

    # Gear/flaps: conservative auto-config only when clearly on approach.
    gear = 1 if bool(state.get("gear_handle_down", False)) else 0
    flaps_cmd = flap_deg
    approach_mode = False
    if (agl_ft is not None and agl_ft < 7000.0) and (target_alt < 4000.0) and (target_ias < 210.0):
        approach_mode = True
    if approach_mode and (agl_ft is not None) and agl_ft < 2200.0 and target_ias < 210.0:
        gear = 1
    if not on_ground and (agl_ft is not None) and agl_ft > 2500.0 and target_alt > 6000.0:
        # likely enroute
        gear = 0

    if approach_mode and airspeed_reliable and ias is not None:
        spd = float(ias)
        desired_flaps = 0.0
        if spd < 210.0:
            desired_flaps = 10.0
        if spd < 190.0:
            desired_flaps = 15.0
        if spd < 175.0:
            desired_flaps = 20.0
        if spd < 160.0 and (agl_ft is not None and agl_ft < 1200.0):
            desired_flaps = 40.0
        flaps_cmd = max(flaps_cmd, desired_flaps)
    if on_ground:
        # Don't fight scenario configuration while on the ground.
        flaps_cmd = flap_deg
    elif (not approach_mode) and (agl_ft is None or agl_ft > 3000.0):
        # retract for climb/cruise unless already configured by scenario
        flaps_cmd = min(flaps_cmd, 0.0)

    flaps_cmd = _clamp(float(flaps_cmd), 0.0, 40.0)

    # Autotrim: drive sustained elevator deflection back toward zero.
    # Output is a trim *rate* command in [-1, 1].
    trim = 0.0
    if ap_engaged and att_reliable:
        if abs(elevator) > 0.08 and not aoa_stall:
            trim = _clamp(0.55 * elevator, -1.0, 1.0)
        else:
            trim = 0.0
    if aoa_stall:
        # Help unstick nose-up trim in deep stall.
        trim = _clamp(trim - 0.6, -1.0, 1.0)

    # --- Apply rate limits for smoothness ---
    elevator = _rate_limit(elevator, last.get("elevator"), rate_per_s=6.0, dt=dt)
    aileron = _rate_limit(aileron, last.get("aileron"), rate_per_s=6.0, dt=dt)
    rudder = _rate_limit(rudder, last.get("rudder"), rate_per_s=6.0, dt=dt)
    throttle_cmd = _rate_limit(throttle_cmd, last.get("throttle_cmd"), rate_per_s=0.8, dt=dt)
    speedbrake = _rate_limit(speedbrake, last.get("speedbrake"), rate_per_s=2.0, dt=dt)
    flaps_cmd = _rate_limit(flaps_cmd, last.get("flaps_cmd"), rate_per_s=12.0, dt=dt)  # deg/s

    # Engines: respect engine running flags (don't command thrust on a dead engine).
    eng1_run = bool(state.get("engine_1_running", True))
    eng2_run = bool(state.get("engine_2_running", True))
    throttle_1 = throttle_cmd if eng1_run else 0.0
    throttle_2 = throttle_cmd if eng2_run else 0.0

    out = {
        "elevator": _clamp(float(elevator), -1.0, 1.0),
        "aileron": _clamp(float(aileron), -1.0, 1.0),
        "rudder": _clamp(float(rudder), -1.0, 1.0),
        "throttle_1": _clamp(float(throttle_1), 0.0, 1.0),
        "throttle_2": _clamp(float(throttle_2), 0.0, 1.0),
        "elevator_trim": _clamp(float(trim), -1.0, 1.0),
        "flaps": float(_clamp(flaps_cmd, 0.0, 40.0)),
        "speedbrake": _clamp(float(speedbrake), 0.0, 1.0),
        "gear": 1 if gear else 0,
    }

    # --- Persist fused/command history for next step ---
    last.update(
        {
            "baro_alt_ft": baro_alt,
            "gps_alt_ft": gps_alt,
            "radar_alt_ft": radar_alt,
            "alt_ft": alt_ft,
            "ias_kt": ias,
            "mach": mach,
            "aoa_deg": aoa,
            "hdg_deg": hdg,
            "roll_deg": roll,
            "pitch_deg": pitch,
            "vs_fpm": vs,
            "yaw_rate_dps": yaw_rate,
            "roll_rate_dps": roll_rate,
            "pitch_rate_dps": pitch_rate,
            # commands
            "elevator": out["elevator"],
            "aileron": out["aileron"],
            "rudder": out["rudder"],
            "throttle_cmd": throttle_cmd,
            "speedbrake": out["speedbrake"],
            "flaps_cmd": out["flaps"],
        }
    )

    return out
