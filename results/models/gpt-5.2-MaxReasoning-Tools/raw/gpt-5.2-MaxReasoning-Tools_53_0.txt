from __future__ import annotations

from math import radians, sqrt

# Module-level persistence across calls within a scenario.
_AP = {
    "init": False,
    "t": 0.0,
    "p": [0.0, 0.0, 0.0],  # relative position estimate (LVLH): x(radial), y(along), z(cross)
    "v": [0.0, 0.0, 0.0],  # relative velocity estimate (LVLH)
    "i": [0.0, 0.0, 0.0],  # integral term for disturbance rejection
    "n": 0.0011,  # mean motion estimate (rad/s), used only for gentle tuning
    "last_valid_meas": 0.0,
}


def _clip(x: float, lo: float, hi: float) -> float:
    return lo if x < lo else hi if x > hi else x


def _norm3(v: list[float]) -> float:
    return sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2])


def _dot3(a: list[float], b: list[float]) -> float:
    return a[0] * b[0] + a[1] * b[1] + a[2] * b[2]


def _med(vals: list[float]) -> float | None:
    if not vals:
        return None
    s = sorted(vals)
    mid = len(s) // 2
    if len(s) % 2 == 1:
        return float(s[mid])
    return 0.5 * (float(s[mid - 1]) + float(s[mid]))


def _truthy(x) -> bool:
    # Validity flags may be bools, ints, or noisy floats; treat nonzero as True.
    try:
        return bool(int(x))
    except Exception:
        return bool(x)


def _read_radar(state: dict, kind: str) -> list[float | None]:
    if kind == "p":
        keys = ("radar_rel_pos_x_m", "radar_rel_pos_y_m", "radar_rel_pos_z_m")
        vkeys = ("radar_rel_x_valid", "radar_rel_y_valid", "radar_rel_z_valid")
    else:
        keys = ("radar_rel_vel_x_ms", "radar_rel_vel_y_ms", "radar_rel_vel_z_ms")
        vkeys = ("radar_relv_x_valid", "radar_relv_y_valid", "radar_relv_z_valid")
    out: list[float | None] = [None, None, None]
    for i in range(3):
        if _truthy(state.get(vkeys[i], 0)):
            try:
                out[i] = float(state[keys[i]])
            except Exception:
                out[i] = None
    return out


def _read_lidar(state: dict, kind: str) -> list[float | None]:
    if kind == "p":
        keys = ("lidar_rel_pos_x_m", "lidar_rel_pos_y_m", "lidar_rel_pos_z_m")
        vkeys = ("lidar_rel_x_valid", "lidar_rel_y_valid", "lidar_rel_z_valid")
    else:
        keys = ("lidar_rel_vel_x_ms", "lidar_rel_vel_y_ms", "lidar_rel_vel_z_ms")
        vkeys = ("lidar_relv_x_valid", "lidar_relv_y_valid", "lidar_relv_z_valid")
    out: list[float | None] = [None, None, None]
    for i in range(3):
        if _truthy(state.get(vkeys[i], 0)):
            try:
                out[i] = float(state[keys[i]])
            except Exception:
                out[i] = None
    return out


def _range_samples(state: dict) -> list[float]:
    vals: list[float] = []
    for i in (1, 2, 3):
        if _truthy(state.get(f"range_{i}_valid", 0)):
            try:
                r = float(state.get(f"range_{i}_m", 0.0))
            except Exception:
                continue
            if r > 0.0 and r < 1e7:
                vals.append(r)
    return vals


def _range_meas(state: dict) -> float | None:
    return _med(_range_samples(state))


def _camera_bearing_pos(state: dict) -> list[float] | None:
    if not _truthy(state.get("dock_camera_active", 0)):
        return None
    bxv = _truthy(state.get("dock_camera_bearing_x_valid", 0))
    byv = _truthy(state.get("dock_camera_bearing_y_valid", 0))
    bzv = _truthy(state.get("dock_camera_bearing_z_valid", 0))
    if not (bxv and byv and bzv):
        return None
    try:
        b = [
            float(state["dock_camera_bearing_x"]),
            float(state["dock_camera_bearing_y"]),
            float(state["dock_camera_bearing_z"]),
        ]
        r = float(state.get("dock_camera_range_m", 0.0))
    except Exception:
        return None
    bn = _norm3(b)
    if bn < 1e-6:
        return None
    if r <= 0.0:
        r = _range_meas(state) or 0.0
    if r <= 0.0:
        return None
    return [r * b[0] / bn, r * b[1] / bn, r * b[2] / bn]


def _fuse(a: list[float | None], b: list[float | None], wa: float, wb: float) -> list[float | None]:
    out: list[float | None] = [None, None, None]
    for i in range(3):
        ai, bi = a[i], b[i]
        if ai is None and bi is None:
            out[i] = None
        elif bi is None:
            out[i] = float(ai)
        elif ai is None:
            out[i] = float(bi)
        else:
            out[i] = (wa * float(ai) + wb * float(bi)) / max(1e-9, wa + wb)
    return out


def _speed_schedule(r: float) -> float:
    # Conservative closing speed schedule (m/s), tightened close-in.
    if r > 1200.0:
        return 2.5
    if r > 600.0:
        return 2.0
    if r > 250.0:
        return 1.0
    if r > 80.0:
        return 0.5
    if r > 20.0:
        return 0.2
    if r > 10.0:
        return 0.12
    if r > 3.0:
        return 0.08
    return 0.05


def _fuel_scale(state: dict) -> tuple[float, float]:
    # Returns (max_cmd_scale, deadband_scale).
    try:
        pct = float(state.get("fuel_remaining_pct", 1.0))
    except Exception:
        pct = 1.0
    try:
        kg = float(state.get("fuel_remaining_kg", 1e9))
    except Exception:
        kg = 1e9

    # Nominal.
    max_cmd = 1.0
    deadband = 1.0

    if pct < 0.20 or kg < 120.0:
        max_cmd = 0.55
        deadband = 1.3
    if pct < 0.12 or kg < 70.0:
        max_cmd = 0.38
        deadband = 1.7
    if pct < 0.08 or kg < 45.0:
        max_cmd = 0.28
        deadband = 2.2
    return max_cmd, deadband


def _attitude_rates(state: dict, dt: float) -> tuple[float, float, float]:
    # Rate-command a simple attitude hold: damp rates + (if available) drive star-tracker
    # Euler angles toward 0. (Assumes reported angles are relative to desired LVLH attitude.)
    gyro_roll: list[float] = []
    gyro_pitch: list[float] = []
    gyro_yaw: list[float] = []
    for i in (1, 2, 3):
        if _truthy(state.get(f"gyro_{i}_roll_valid", 0)):
            gyro_roll.append(float(state.get(f"gyro_{i}_roll_dps", 0.0)))
        if _truthy(state.get(f"gyro_{i}_pitch_valid", 0)):
            gyro_pitch.append(float(state.get(f"gyro_{i}_pitch_dps", 0.0)))
        if _truthy(state.get(f"gyro_{i}_yaw_valid", 0)):
            gyro_yaw.append(float(state.get(f"gyro_{i}_yaw_dps", 0.0)))
    w_roll = radians(_med(gyro_roll) or 0.0)
    w_pitch = radians(_med(gyro_pitch) or 0.0)
    w_yaw = radians(_med(gyro_yaw) or 0.0)

    st_roll: list[float] = []
    st_pitch: list[float] = []
    st_yaw: list[float] = []
    for i in (1, 2):
        if _truthy(state.get(f"star_tracker_{i}_roll_valid", 0)):
            st_roll.append(float(state.get(f"star_tracker_{i}_roll_deg", 0.0)))
        if _truthy(state.get(f"star_tracker_{i}_pitch_valid", 0)):
            st_pitch.append(float(state.get(f"star_tracker_{i}_pitch_deg", 0.0)))
        if _truthy(state.get(f"star_tracker_{i}_yaw_valid", 0)):
            st_yaw.append(float(state.get(f"star_tracker_{i}_yaw_deg", 0.0)))

    e_roll = radians(_med(st_roll) or 0.0)
    e_pitch = radians(_med(st_pitch) or 0.0)
    e_yaw = radians(_med(st_yaw) or 0.0)

    # PD: drive angles to 0 and damp gyro rates.
    k_ang = 0.7  # 1/s
    k_rate = 1.0
    cmd_roll = -k_ang * e_roll - k_rate * w_roll
    cmd_pitch = -k_ang * e_pitch - k_rate * w_pitch
    cmd_yaw = -k_ang * e_yaw - k_rate * w_yaw

    lim = 0.035
    return (
        _clip(cmd_roll, -lim, lim),
        _clip(cmd_pitch, -lim, lim),
        _clip(cmd_yaw, -lim, lim),
    )


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP

    # Sanitize dt.
    try:
        dt = float(dt)
    except Exception:
        dt = 0.1
    dt = _clip(dt, 0.02, 2.0)

    _AP["t"] += dt

    # Update mean-motion estimate (optional tuning knob).
    if _truthy(state.get("ground_semi_major_axis_m_valid", 0)) and _truthy(state.get("ground_speed_ms_valid", 0)):
        try:
            a = float(state.get("ground_semi_major_axis_m", 0.0))
            v = float(state.get("ground_speed_ms", 0.0))
        except Exception:
            a, v = 0.0, 0.0
        if a > 1e6 and v > 10.0:
            _AP["n"] = _clip(v / a, 0.0005, 0.0020)

    # --- Measurement fusion ---
    # Preference: radar for far range, lidar for near, camera bearing+range as backup within 500 m.
    p_radar = _read_radar(state, "p")
    v_radar = _read_radar(state, "v")
    p_lidar = _read_lidar(state, "p")
    v_lidar = _read_lidar(state, "v")

    # Coarse range guess for weighting + range sensor quality.
    range_samples = _range_samples(state)
    r_scalar = _med(range_samples)
    range_quality = 0.0
    if r_scalar is not None and range_samples:
        devs = [abs(x - r_scalar) for x in range_samples]
        mad = _med(devs) or 0.0
        rel_mad = mad / max(1e-6, r_scalar)
        # Heuristic: tight agreement => trustworthy; disagreement => treat as noisy.
        if len(range_samples) >= 2 and rel_mad <= 0.05:
            range_quality = 1.0
        elif len(range_samples) >= 2 and rel_mad <= 0.12:
            range_quality = 0.6
        elif len(range_samples) >= 1 and rel_mad <= 0.20:
            range_quality = 0.3
        else:
            range_quality = 0.0
    r_guess = r_scalar if r_scalar is not None else max(1.0, _norm3(_AP["p"]))

    # Weights: fade from radar->lidar as we get close.
    # (Keeps functioning during flicker/noise.)
    w_lidar = _clip(1.0 - (r_guess - 80.0) / 220.0, 0.2, 0.8)
    w_radar = 1.0 - w_lidar

    p_meas = _fuse(p_radar, p_lidar, w_radar, w_lidar)
    v_meas = _fuse(v_radar, v_lidar, w_radar, w_lidar)

    # If we have camera bearing and either camera range or rangefinder range, use it as a near-field position cue.
    p_cam = _camera_bearing_pos(state)
    if p_cam is not None and r_guess < 500.0:
        for i in range(3):
            if p_meas[i] is None:
                p_meas[i] = p_cam[i]
            else:
                p_meas[i] = 0.7 * float(p_meas[i]) + 0.3 * p_cam[i]

    # If scalar range is trustworthy, gently correct the magnitude (but not direction).
    if r_scalar is not None and range_quality > 0.0:
        pm = [p_meas[i] if p_meas[i] is not None else _AP["p"][i] for i in range(3)]
        rm = _norm3(pm)
        if rm > 1e-3:
            ratio = r_scalar / rm
            # Only apply if disagreement is large enough to matter.
            if abs(1.0 - ratio) > 0.25 and range_quality >= 0.6:
                ratio = _clip(ratio, 0.7, 1.3)
                ratio = 1.0 + range_quality * (ratio - 1.0)
                for i in range(3):
                    if p_meas[i] is not None:
                        p_meas[i] = float(p_meas[i]) * ratio

    have_pos = any(x is not None for x in p_meas)
    have_vel = any(x is not None for x in v_meas)

    # Initialize estimator if needed.
    if not _AP["init"]:
        if have_pos:
            _AP["p"] = [float(p_meas[i] if p_meas[i] is not None else 0.0) for i in range(3)]
        elif r_scalar is not None:
            # Default "most common start": behind station (negative y).
            _AP["p"] = [0.0, -float(r_scalar), 0.0]
        else:
            _AP["p"] = [0.0, -200.0, 0.0]
        if have_vel:
            _AP["v"] = [float(v_meas[i] if v_meas[i] is not None else 0.0) for i in range(3)]
        else:
            _AP["v"] = [0.0, 0.0, 0.0]
        _AP["i"] = [0.0, 0.0, 0.0]
        _AP["init"] = True

    # Alpha-beta(-vel) filter update.
    p_est = _AP["p"]
    v_est = _AP["v"]

    # Predict.
    p_pred = [p_est[0] + v_est[0] * dt, p_est[1] + v_est[1] * dt, p_est[2] + v_est[2] * dt]
    v_pred = [v_est[0], v_est[1], v_est[2]]

    # Gains: more trust close-in.
    r_for_gain = max(1.0, _norm3(p_pred))
    alpha = _clip(0.35 - 0.15 * (r_for_gain / 800.0), 0.12, 0.35)
    beta = _clip(0.10 - 0.05 * (r_for_gain / 800.0), 0.03, 0.10)
    gamma_v = 0.35  # direct velocity measurement blend

    if have_pos:
        pm = [float(p_meas[i] if p_meas[i] is not None else p_pred[i]) for i in range(3)]
        r = [pm[0] - p_pred[0], pm[1] - p_pred[1], pm[2] - p_pred[2]]
        p_pred = [p_pred[0] + alpha * r[0], p_pred[1] + alpha * r[1], p_pred[2] + alpha * r[2]]
        v_pred = [
            v_pred[0] + (beta / dt) * r[0],
            v_pred[1] + (beta / dt) * r[1],
            v_pred[2] + (beta / dt) * r[2],
        ]
        _AP["last_valid_meas"] = _AP["t"]

    if have_vel:
        vm = [float(v_meas[i] if v_meas[i] is not None else v_pred[i]) for i in range(3)]
        v_pred = [
            (1.0 - gamma_v) * v_pred[0] + gamma_v * vm[0],
            (1.0 - gamma_v) * v_pred[1] + gamma_v * vm[1],
            (1.0 - gamma_v) * v_pred[2] + gamma_v * vm[2],
        ]
        _AP["last_valid_meas"] = _AP["t"]

    _AP["p"] = p_pred
    _AP["v"] = v_pred

    p = _AP["p"]
    v = _AP["v"]
    r = max(1e-6, _norm3(p))

    # --- Guidance / control ---
    max_cmd, deadband_scale = _fuel_scale(state)
    n = float(_AP["n"])

    # Corridor logic: prefer to get onto V-bar corridor before aggressively closing.
    # This improves robustness for diagonal and cross-track-offset starts.
    corridor = max(6.0, 0.18 * abs(p[1]) + 2.0)
    corridor_ok = (abs(p[0]) <= corridor) and (abs(p[2]) <= corridor)
    close_weight = 1.0 if corridor_ok else 0.35

    # Blend between "orbital phasing intuition" far away and "linear/local" close-in.
    # far_w=1 => far regime; far_w=0 => close regime
    if r >= 220.0:
        far_w = 1.0
    elif r <= 70.0:
        far_w = 0.0
    else:
        far_w = (r - 70.0) / (220.0 - 70.0)

    # Axis gains (dimensionless -> thruster command); tuned to be conservative.
    # x,z are always "direct": thrust toward zero with damping.
    kpx_far, kdx_far = 0.0015, 0.05
    kpx_near, kdx_near = 0.020, 0.35
    kpz_far, kdz_far = 0.0015, 0.05
    kpz_near, kdz_near = 0.020, 0.35

    kpx = far_w * kpx_far + (1.0 - far_w) * kpx_near
    kdx = far_w * kdx_far + (1.0 - far_w) * kdx_near
    kpz = far_w * kpz_far + (1.0 - far_w) * kpz_near
    kdz = far_w * kdz_far + (1.0 - far_w) * kdz_near

    ux = -(kpx * p[0] + kdx * v[0])
    uz = -(kpz * p[2] + kdz * v[2])

    # y-axis: far away use "counter-intuitive" phasing (sign matches separation),
    # close-in use direct PD (sign opposes separation).
    kpy_far, kdy_far = 0.0010, 0.06
    kpy_near, kdy_near = 0.018, 0.40
    uy_far = +(kpy_far * p[1] + kdy_far * v[1])  # phasing
    uy_near = -(kpy_near * p[1] + kdy_near * v[1])  # direct
    uy = far_w * uy_far + (1.0 - far_w) * uy_near
    uy *= close_weight

    # Disturbance rejection (uncommanded tiny thrust): integrate small velocity error close-in.
    # Only integrate when we have recent measurements.
    fresh = (_AP["t"] - _AP["last_valid_meas"]) < 2.5
    if fresh and r < 120.0:
        # Desired velocity is a gentle glideslope toward the target in close range.
        vdes_mag = _speed_schedule(r)
        ph = [p[0] / r, p[1] / r, p[2] / r]
        v_des = [-vdes_mag * ph[0], -vdes_mag * ph[1], -vdes_mag * ph[2]]
        verr = [v[0] - v_des[0], v[1] - v_des[1], v[2] - v_des[2]]
        ki = 0.08
        i = _AP["i"]
        i[0] = _clip(i[0] + ki * verr[0] * dt, -0.25, 0.25)
        i[1] = _clip(i[1] + ki * verr[1] * dt, -0.25, 0.25)
        i[2] = _clip(i[2] + ki * verr[2] * dt, -0.25, 0.25)
        _AP["i"] = i
    else:
        # Leak integrator when far / no good data.
        i = _AP["i"]
        leak = 0.98
        _AP["i"] = [i[0] * leak, i[1] * leak, i[2] * leak]

    ux += -_AP["i"][0]
    uy += -_AP["i"][1]
    uz += -_AP["i"][2]

    # Closing speed governor (primarily in the near field).
    ph = [p[0] / r, p[1] / r, p[2] / r]
    closing = -_dot3(ph, v)  # >0 means approaching
    v_target = _speed_schedule(r)
    if r < 250.0 and closing > v_target:
        k_brake = 0.65
        excess = closing - v_target
        ux += k_brake * excess * ph[0]
        uy += k_brake * excess * ph[1]
        uz += k_brake * excess * ph[2]

    # Near-dock finesse: aggressively damp lateral components, keep y approach gentle.
    if r < 12.0:
        ux *= 1.15
        uz *= 1.15
        uy *= 0.75
    if r < 4.0:
        ux *= 0.9
        uz *= 0.9
        uy *= 0.65

    # Saturate and apply deadband (fuel-friendly).
    # Deadband increases with low fuel.
    db = (0.02 if r > 80.0 else 0.01) * deadband_scale
    ux = _clip(ux, -max_cmd, max_cmd)
    uy = _clip(uy, -max_cmd, max_cmd)
    uz = _clip(uz, -max_cmd, max_cmd)
    if abs(ux) < db:
        ux = 0.0
    if abs(uy) < db:
        uy = 0.0
    if abs(uz) < db:
        uz = 0.0

    # Attitude (rate commands in rad/s).
    att_roll, att_pitch, att_yaw = _attitude_rates(state, dt)

    return {
        "thrust_x": float(_clip(ux, -1.0, 1.0)),
        "thrust_y": float(_clip(uy, -1.0, 1.0)),
        "thrust_z": float(_clip(uz, -1.0, 1.0)),
        "att_rate_roll": float(att_roll),
        "att_rate_pitch": float(att_pitch),
        "att_rate_yaw": float(att_yaw),
    }
