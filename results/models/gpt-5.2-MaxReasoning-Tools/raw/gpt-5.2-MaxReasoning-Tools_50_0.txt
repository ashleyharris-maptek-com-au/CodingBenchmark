#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    float sepR2 = sepRadius * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2 = cohRadius * cohRadius;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint jbase = j * 2u;
        vec3 posJ = inData[jbase].xyz;
        vec3 velJ = inData[jbase + 1u].xyz;

        vec3 diff = pos - posJ;
        float dist2 = dot(diff, diff);
        if (dist2 == 0.0) {
            continue;
        }

        if (dist2 < sepR2) {
            sepSum += diff / dist2;
            sepCount++;
        }

        if (dist2 < alignR2) {
            alignSum += velJ;
            alignCount++;
        }

        if (dist2 < cohR2) {
            cohSum += posJ;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepSum / float(sepCount)) * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignSum / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 avgPos = cohSum / float(cohCount);
        accel += (avgPos - pos) * cohWeight;
    }

    float boundThresh = boundSize * 0.4;
    float boundScale = boundSize * 0.5;
    if (boundScale > 0.0) {
        if (pos.x > boundThresh || pos.x < -boundThresh) {
            accel.x += (-pos.x / boundScale) * boundForce;
        }
        if (pos.y > boundThresh || pos.y < -boundThresh) {
            accel.y += (-pos.y / boundScale) * boundForce;
        }
        if (pos.z > boundThresh || pos.z < -boundThresh) {
            accel.z += (-pos.z / boundScale) * boundForce;
        }
    }

    vel += accel * dt;

    float speed2 = dot(vel, vel);
    float maxSpeed2 = maxSpeed * maxSpeed;
    if (maxSpeed > 0.0 && speed2 > maxSpeed2) {
        vel *= maxSpeed * inversesqrt(speed2);
    }

    pos += vel * dt;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
