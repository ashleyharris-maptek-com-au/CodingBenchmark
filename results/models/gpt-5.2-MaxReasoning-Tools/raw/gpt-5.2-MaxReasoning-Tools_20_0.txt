use std::cmp::{Ordering, Reverse};
use std::collections::{BinaryHeap, HashMap};
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

#[derive(Default)]
struct FxHasher {
    hash: u64,
}

impl Hasher for FxHasher {
    fn write(&mut self, bytes: &[u8]) {
        // Fallback: rarely used by our keys (we mostly hit write_i32/write_u32).
        const K: u64 = 0x517cc1b727220a95;
        for &b in bytes {
            self.hash = self.hash.rotate_left(5) ^ (b as u64).wrapping_mul(K);
        }
    }
    fn write_u32(&mut self, i: u32) {
        const K: u64 = 0x517cc1b727220a95;
        self.hash = self.hash.rotate_left(5) ^ (i as u64).wrapping_mul(K);
    }
    fn write_i32(&mut self, i: i32) {
        self.write_u32(i as u32);
    }
    fn write_u64(&mut self, i: u64) {
        const K: u64 = 0x517cc1b727220a95;
        self.hash = self.hash.rotate_left(5) ^ i.wrapping_mul(K);
    }
    fn finish(&self) -> u64 {
        self.hash
    }
}

type FastBuildHasher = BuildHasherDefault<FxHasher>;
type FastMap<K, V> = HashMap<K, V, FastBuildHasher>;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn skip_ws(&mut self) {
        while let Some(b) = self.buf.get(self.idx).copied() {
            if b.is_ascii_whitespace() {
                self.idx += 1;
            } else {
                break;
            }
        }
    }

    fn next_string(&mut self) -> Option<String> {
        self.skip_ws();
        let start = self.idx;
        while let Some(b) = self.buf.get(self.idx).copied() {
            if b.is_ascii_whitespace() {
                break;
            }
            self.idx += 1;
        }
        if self.idx == start {
            return None;
        }
        Some(String::from_utf8_lossy(&self.buf[start..self.idx]).into_owned())
    }

    fn next_f32(&mut self) -> Option<f32> {
        let s = self.next_string()?;
        Some(s.parse::<f32>().unwrap_or(f32::NAN))
    }

    fn next_usize(&mut self) -> Option<usize> {
        let s = self.next_string()?;
        s.parse::<usize>().ok()
    }
}

#[derive(Copy, Clone, Eq, PartialEq, Hash)]
struct Cell {
    x: i32,
    y: i32,
    z: i32,
}

#[derive(Copy, Clone, Default)]
struct Stat {
    n: u32,
    mean: f32,
    m2: f32,
}

impl Stat {
    fn update(&mut self, x: f32) {
        self.n += 1;
        let n = self.n as f32;
        let delta = x - self.mean;
        self.mean += delta / n;
        let delta2 = x - self.mean;
        self.m2 += delta * delta2;
    }

    fn stddev(&self) -> f32 {
        if self.n <= 1 {
            0.0
        } else {
            (self.m2 / ((self.n - 1) as f32)).sqrt()
        }
    }
}

#[derive(Copy, Clone, Debug)]
struct Suspect {
    conf: f32,
    hole: u32,
    sample: u32,
    prop: u16,
}

impl PartialEq for Suspect {
    fn eq(&self, other: &Self) -> bool {
        self.conf.to_bits() == other.conf.to_bits()
            && self.hole == other.hole
            && self.sample == other.sample
            && self.prop == other.prop
    }
}
impl Eq for Suspect {}
impl PartialOrd for Suspect {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}
impl Ord for Suspect {
    fn cmp(&self, other: &Self) -> Ordering {
        self.conf
            .total_cmp(&other.conf)
            .then_with(|| self.hole.cmp(&other.hole))
            .then_with(|| self.sample.cmp(&other.sample))
            .then_with(|| self.prop.cmp(&other.prop))
    }
}

fn cell_of(x: f32, y: f32, z: f32, inv_cell: f32) -> Cell {
    Cell {
        x: (x * inv_cell).floor() as i32,
        y: (y * inv_cell).floor() as i32,
        z: (z * inv_cell).floor() as i32,
    }
}

fn median_sorted(sorted: &[f32]) -> f32 {
    let n = sorted.len();
    if n == 0 {
        return f32::NAN;
    }
    if n & 1 == 1 {
        sorted[n / 2]
    } else {
        0.5 * (sorted[n / 2 - 1] + sorted[n / 2])
    }
}

fn median_and_scale(values: &mut Vec<f32>, dev: &mut Vec<f32>) -> Option<(f32, f32)> {
    if values.len() < 3 {
        return None;
    }
    values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median_sorted(values);
    dev.clear();
    dev.extend(values.iter().map(|x| (x - med).abs()));
    dev.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median_sorted(dev);
    // Convert MAD to a robust std-like scale. Fallback avoids divide-by-zero explosions.
    let mut scale = 1.4826 * mad;
    if !scale.is_finite() {
        return None;
    }
    if scale < 1e-6 {
        scale = ((med.abs() + 1.0) * 0.01).max(1e-6);
    }
    Some((med, scale))
}

fn maybe_log_scale(sample: &mut [f32], min_v: f32, max_v: f32) -> bool {
    if !min_v.is_finite() || !max_v.is_finite() {
        return false;
    }
    if min_v < 0.0 {
        return false;
    }
    if sample.len() < 32 {
        return false;
    }
    sample.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median_sorted(sample);
    if !med.is_finite() || med <= 0.0 {
        return false;
    }
    let p90 = sample[(sample.len() * 9) / 10];
    let tail = if p90.is_finite() { p90 / med } else { 1.0 };
    let spread = max_v / med;
    tail > 50.0 || spread > 1.0e4
}

fn transform(v: f32, log_scale: bool) -> f32 {
    if log_scale {
        (v + 1.0).ln()
    } else {
        v
    }
}

fn hole_confidence(
    col: &[f32],
    i: usize,
    rel: u32,
    hole_len: u32,
    log_scale: bool,
    scratch: &mut Vec<f32>,
    dev: &mut Vec<f32>,
) -> f32 {
    scratch.clear();
    let rel_i = rel as i32;
    let len_i = hole_len as i32;
    for off in [-2, -1, 1, 2] {
        let rr = rel_i + off;
        if rr < 0 || rr >= len_i {
            continue;
        }
        let j = (i as i32 + off) as usize;
        let v = col[j];
        if v.is_nan() {
            continue;
        }
        scratch.push(transform(v, log_scale));
    }
    let (med, scale) = match median_and_scale(scratch, dev) {
        Some(x) => x,
        None => return 0.0,
    };
    let v0 = col[i];
    if v0.is_nan() {
        return 0.0;
    }
    let vt = transform(v0, log_scale);
    ((vt - med).abs() / scale).max(0.0)
}

fn main() {
    let mut sc = Scanner::new();
    let num_holes = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let num_props = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(s) = sc.next_string() {
            prop_names.push(s);
        } else {
            // Missing names; keep going with placeholders.
            prop_names.push("prop".to_string());
        }
    }

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut hole_counts: Vec<u32> = Vec::with_capacity(num_holes);

    let mut xs: Vec<f32> = Vec::new();
    let mut ys: Vec<f32> = Vec::new();
    let mut zs: Vec<f32> = Vec::new();
    let mut depths: Vec<f32> = Vec::new();
    let mut hole_of: Vec<u32> = Vec::new();
    let mut rel_idx: Vec<u32> = Vec::new();

    let mut cols: Vec<Vec<f32>> = (0..num_props).map(|_| Vec::new()).collect();

    let mut prop_min: Vec<f32> = vec![f32::INFINITY; num_props];
    let mut prop_max: Vec<f32> = vec![-f32::INFINITY; num_props];
    let mut prop_samples: Vec<Vec<f32>> = (0..num_props).map(|_| Vec::with_capacity(256)).collect();
    let sample_cap = 2048usize;

    let mut depth_intervals: Vec<f32> = Vec::with_capacity(10_000);

    for hole_i in 0..num_holes {
        let hole_id = match sc.next_string() {
            Some(s) => s,
            None => break,
        };
        let start_x = sc.next_f32().unwrap_or(f32::NAN);
        let start_y = sc.next_f32().unwrap_or(f32::NAN);
        let start_z = sc.next_f32().unwrap_or(f32::NAN);
        let dir_x = sc.next_f32().unwrap_or(0.0);
        let dir_y = sc.next_f32().unwrap_or(0.0);
        let dir_z = sc.next_f32().unwrap_or(1.0);
        let _length = sc.next_f32().unwrap_or(f32::NAN);
        let num_samples = match sc.next_usize() {
            Some(v) => v,
            None => 0,
        };

        let norm = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let inv_norm = if norm.is_finite() && norm > 1e-12 {
            1.0 / norm
        } else {
            1.0
        };
        let ux = dir_x * inv_norm;
        let uy = dir_y * inv_norm;
        let uz = dir_z * inv_norm;

        hole_ids.push(hole_id);
        hole_counts.push(0); // set after parsing (robust to truncated input)
        let hole_slot = hole_counts.len() - 1;

        let mut prev_depth: Option<f32> = None;
        let mut parsed_samples: u32 = 0;
        for s in 0..num_samples {
            let depth = match sc.next_f32() {
                Some(v) => v,
                None => break,
            };
            parsed_samples += 1;
            if let Some(pd) = prev_depth {
                let d = (depth - pd).abs();
                if d.is_finite() && d > 0.0 && depth_intervals.len() < 10_000 {
                    depth_intervals.push(d);
                }
            }
            prev_depth = Some(depth);

            let x = start_x + ux * depth;
            let y = start_y + uy * depth;
            let z = start_z + uz * depth;

            xs.push(x);
            ys.push(y);
            zs.push(z);
            depths.push(depth);
            hole_of.push(hole_i as u32);
            rel_idx.push(s as u32);

            for p in 0..num_props {
                let v = sc.next_f32().unwrap_or(f32::NAN);
                cols[p].push(v);
                if !v.is_nan() {
                    if v < prop_min[p] {
                        prop_min[p] = v;
                    }
                    if v > prop_max[p] {
                        prop_max[p] = v;
                    }
                    if prop_samples[p].len() < sample_cap {
                        prop_samples[p].push(v);
                    }
                }
            }
        }
        hole_counts[hole_slot] = parsed_samples;
    }

    let n_samples = xs.len();
    if n_samples == 0 || num_props == 0 || hole_ids.is_empty() {
        println!("0");
        return;
    }

    let mut spacing = 1.0f32;
    if !depth_intervals.is_empty() {
        depth_intervals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        spacing = median_sorted(&depth_intervals).max(1e-3);
    }

    let mut log_scale: Vec<bool> = vec![false; num_props];
    for p in 0..num_props {
        if prop_samples[p].len() >= 32 && prop_min[p].is_finite() && prop_min[p] >= 0.0 {
            log_scale[p] = maybe_log_scale(&mut prop_samples[p], prop_min[p], prop_max[p]);
        }
    }

    // Strategy selection: local kNN when it's cheap enough; otherwise use cell stats + along-hole check.
    let use_knn = n_samples <= 800_000 && num_props <= 48;

    let max_keep: usize = 50_000;
    let mut heap: BinaryHeap<Reverse<Suspect>> = BinaryHeap::new();

    if use_knn {
        let cell_size = (spacing * 4.0).clamp(0.5, 50.0);
        let inv_cell = 1.0 / cell_size;
        let radius2 = (cell_size * 1.8) * (cell_size * 1.8);
        let kmax: usize = 64;
        let min_neighbors: usize = 6;
        let threshold: f32 = 8.0;

        let mut cell_coords: Vec<Cell> = Vec::with_capacity(n_samples);
        let mut grid: FastMap<Cell, Vec<u32>> = FastMap::default();
        grid.reserve(n_samples / 8);

        for i in 0..n_samples {
            let c = cell_of(xs[i], ys[i], zs[i], inv_cell);
            cell_coords.push(c);
            grid.entry(c).or_insert_with(Vec::new).push(i as u32);
        }

        let mut neigh_indices: Vec<u32> = Vec::with_capacity(kmax);
        let mut scratch_vals: Vec<f32> = Vec::with_capacity(kmax);
        let mut dev: Vec<f32> = Vec::with_capacity(kmax);

        for i in 0..n_samples {
            neigh_indices.clear();
            let c = cell_coords[i];
            for dx in -1..=1 {
                for dy in -1..=1 {
                    for dz in -1..=1 {
                        let cc = Cell {
                            x: c.x + dx,
                            y: c.y + dy,
                            z: c.z + dz,
                        };
                        if let Some(list) = grid.get(&cc) {
                            for &j in list {
                                if j as usize == i {
                                    continue;
                                }
                                let jj = j as usize;
                                let dx = xs[jj] - xs[i];
                                let dy = ys[jj] - ys[i];
                                let dz = zs[jj] - zs[i];
                                let d2 = dx * dx + dy * dy + dz * dz;
                                if d2 <= radius2 {
                                    neigh_indices.push(j);
                                    if neigh_indices.len() >= kmax {
                                        break;
                                    }
                                }
                            }
                        }
                        if neigh_indices.len() >= kmax {
                            break;
                        }
                    }
                    if neigh_indices.len() >= kmax {
                        break;
                    }
                }
                if neigh_indices.len() >= kmax {
                    break;
                }
            }

            if neigh_indices.len() < min_neighbors {
                continue;
            }

            let hole = hole_of[i];
            let sample = rel_idx[i];

            for p in 0..num_props {
                let v0 = cols[p][i];
                if v0.is_nan() {
                    continue;
                }
                scratch_vals.clear();
                for &j in &neigh_indices {
                    let v = cols[p][j as usize];
                    if v.is_nan() {
                        continue;
                    }
                    scratch_vals.push(transform(v, log_scale[p]));
                }
                if scratch_vals.len() < min_neighbors {
                    continue;
                }
                let (med, scale) = match median_and_scale(&mut scratch_vals, &mut dev) {
                    Some(x) => x,
                    None => continue,
                };
                let vt = transform(v0, log_scale[p]);
                let conf = (vt - med).abs() / scale;
                if conf.is_finite() && conf >= threshold {
                    let s = Suspect {
                        conf,
                        hole,
                        sample,
                        prop: p as u16,
                    };
                    if heap.len() < max_keep {
                        heap.push(Reverse(s));
                    } else if let Some(peek) = heap.peek() {
                        if s.conf > peek.0.conf {
                            heap.pop();
                            heap.push(Reverse(s));
                        }
                    }
                }
            }
        }
    } else {
        // Cell stats + along-hole check.
        let scale_factor = ((n_samples as f32) / 2_000_000.0).max(1.0).cbrt();
        let cell_size = (spacing * 10.0 * scale_factor).clamp(1.0, 200.0);
        let inv_cell = 1.0 / cell_size;
        let min_cell: u32 = 10;
        let threshold: f32 = 7.0;

        // Build cell table: map cell -> index, contiguous stats storage (cell_count * num_props).
        let mut cell_index: FastMap<Cell, u32> = FastMap::default();
        cell_index.reserve(n_samples / 16);
        let mut sample_cell: Vec<u32> = Vec::with_capacity(n_samples);
        let mut cell_stats: Vec<Stat> = Vec::new();
        let mut global: Vec<Stat> = vec![Stat::default(); num_props];

        for i in 0..n_samples {
            let c = cell_of(xs[i], ys[i], zs[i], inv_cell);
            let idx = if let Some(&idx) = cell_index.get(&c) {
                idx
            } else {
                let idx = (cell_stats.len() / num_props) as u32;
                cell_index.insert(c, idx);
                cell_stats.extend(std::iter::repeat(Stat::default()).take(num_props));
                idx
            };
            sample_cell.push(idx);
            let base = (idx as usize) * num_props;
            for p in 0..num_props {
                let v = cols[p][i];
                if v.is_nan() {
                    continue;
                }
                let vt = transform(v, log_scale[p]);
                cell_stats[base + p].update(vt);
                global[p].update(vt);
            }
        }

        let mut scratch: Vec<f32> = Vec::with_capacity(8);
        let mut dev: Vec<f32> = Vec::with_capacity(8);

        for p in 0..num_props {
            let g = global[p];
            let g_std = g.stddev().max(1e-6);
            let col = &cols[p];
            for i in 0..n_samples {
                let v0 = col[i];
                if v0.is_nan() {
                    continue;
                }
                let vt = transform(v0, log_scale[p]);
                let cell_idx = sample_cell[i] as usize;
                let s = cell_stats[cell_idx * num_props + p];
                let conf_cell;
                let s_std = s.stddev();
                if s.n >= min_cell && s_std > 1e-6 {
                    conf_cell = (vt - s.mean).abs() / s_std;
                } else {
                    conf_cell = (vt - g.mean).abs() / g_std;
                }

                let hole = hole_of[i] as usize;
                let rel = rel_idx[i];
                let hole_len = hole_counts[hole];
                let conf_hole = hole_confidence(col, i, rel, hole_len, log_scale[p], &mut scratch, &mut dev);

                let conf = conf_cell.max(conf_hole);
                if conf.is_finite() && conf >= threshold {
                    let s = Suspect {
                        conf,
                        hole: hole as u32,
                        sample: rel,
                        prop: p as u16,
                    };
                    if heap.len() < max_keep {
                        heap.push(Reverse(s));
                    } else if let Some(peek) = heap.peek() {
                        if s.conf > peek.0.conf {
                            heap.pop();
                            heap.push(Reverse(s));
                        }
                    }
                }
            }
        }
    }

    let mut suspects: Vec<Suspect> = heap.into_iter().map(|r| r.0).collect();
    suspects.sort_by(|a, b| {
        b.conf
            .total_cmp(&a.conf)
            .then_with(|| a.hole.cmp(&b.hole))
            .then_with(|| a.sample.cmp(&b.sample))
            .then_with(|| a.prop.cmp(&b.prop))
    });

    let mut out = String::new();
    out.push_str(&format!("{}\n", suspects.len()));
    for s in suspects {
        let hole_id = &hole_ids[s.hole as usize];
        let prop = &prop_names[s.prop as usize];
        out.push_str(&format!(
            "{} {} {} {:.3}\n",
            hole_id, s.sample, prop, s.conf
        ));
    }
    print!("{out}");
}
