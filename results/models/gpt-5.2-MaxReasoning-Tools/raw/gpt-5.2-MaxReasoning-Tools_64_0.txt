declare const require: any;
declare const process: any;

const fs = require("fs");
let input: string = fs.readFileSync(0, "utf8");
if (input.length > 0 && input.charCodeAt(0) === 0xfeff) input = input.slice(1);

type DroneInput = { id?: unknown; x: number; y: number };
type InputData = {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
};

function clampInt(n: number, lo: number, hi: number): number {
  if (n < lo) return lo;
  if (n > hi) return hi;
  return n | 0;
}

function xorshift32(seed: number) {
  let x = seed | 0;
  if (x === 0) x = 0x6d2b79f5;
  return () => {
    x ^= x << 13;
    x ^= x >>> 17;
    x ^= x << 5;
    x |= 0;
    return x >>> 0;
  };
}

function main() {
  let data: InputData;
  try {
    data = JSON.parse(input);
  } catch {
    process.stdout.write(JSON.stringify({ paths: [] }));
    return;
  }

  const grid = data.probability;
  const drones = Array.isArray(data.drones) ? data.drones : [];
  const maxStepsPerDrone = Math.max(0, (data.maxStepsPerDrone as number) | 0);

  const H = Array.isArray(grid) ? grid.length : 0;
  const W = H > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;

  if (H <= 0 || W <= 0 || drones.length <= 0) {
    process.stdout.write(JSON.stringify({ paths: drones.map(() => []) }));
    return;
  }

  const nCells = H * W;
  const passable = new Uint8Array(nCells);
  const value = new Float64Array(nCells);

  for (let y = 0; y < H; y++) {
    const row = grid[y];
    for (let x = 0; x < W; x++) {
      const idx = y * W + x;
      const v = row[x];
      if (v === -1) {
        passable[idx] = 0;
        value[idx] = 0;
      } else {
        passable[idx] = 1;
        value[idx] = v > 0 ? v : 0;
      }
    }
  }

  // Box-filter "heat" to bias towards regions with dense probability mass.
  const HEAT_RADIUS = 4;
  const heat = new Float32Array(nCells);
  {
    const satW = W + 1;
    const satH = H + 1;
    const sat = new Float32Array(satW * satH);
    for (let y = 1; y <= H; y++) {
      let rowSum = 0;
      const base = (y - 1) * W;
      const satRow = y * satW;
      const satPrev = (y - 1) * satW;
      for (let x = 1; x <= W; x++) {
        rowSum += value[base + (x - 1)];
        sat[satRow + x] = sat[satPrev + x] + rowSum;
      }
    }

    for (let y = 0; y < H; y++) {
      const y0 = y - HEAT_RADIUS < 0 ? 0 : y - HEAT_RADIUS;
      const y1 = y + HEAT_RADIUS >= H ? H - 1 : y + HEAT_RADIUS;
      const satY0 = y0 * satW;
      const satY1 = (y1 + 1) * satW;
      for (let x = 0; x < W; x++) {
        const x0 = x - HEAT_RADIUS < 0 ? 0 : x - HEAT_RADIUS;
        const x1 = x + HEAT_RADIUS >= W ? W - 1 : x + HEAT_RADIUS;
        const a = sat[satY0 + x0];
        const b = sat[satY0 + (x1 + 1)];
        const c = sat[satY1 + x0];
        const d = sat[satY1 + (x1 + 1)];
        heat[y * W + x] = d - b - c + a;
      }
    }
  }

  // Geodesic Voronoi assignment (multi-source BFS) to spread drones out.
  const owner = new Int32Array(nCells);
  owner.fill(-1);
  {
    const q = new Int32Array(nCells);
    let head = 0;
    let tail = 0;

    for (let i = 0; i < drones.length; i++) {
      const d = drones[i] as DroneInput;
      const x = clampInt(d.x, 0, W - 1);
      const y = clampInt(d.y, 0, H - 1);
      const idx = y * W + x;
      if (passable[idx] !== 0 && owner[idx] === -1) {
        owner[idx] = i;
        q[tail++] = idx;
      }
    }

    while (head < tail) {
      const idx = q[head++];
      const own = owner[idx];

      const x = idx % W;
      const y = (idx / W) | 0;

      // Left
      if (x > 0) {
        const n = idx - 1;
        if (passable[n] !== 0 && owner[n] === -1) {
          owner[n] = own;
          q[tail++] = n;
        }
      }
      // Right
      if (x + 1 < W) {
        const n = idx + 1;
        if (passable[n] !== 0 && owner[n] === -1) {
          owner[n] = own;
          q[tail++] = n;
        }
      }
      // Up
      if (y > 0) {
        const n = idx - W;
        if (passable[n] !== 0 && owner[n] === -1) {
          owner[n] = own;
          q[tail++] = n;
        }
      }
      // Down
      if (y + 1 < H) {
        const n = idx + W;
        if (passable[n] !== 0 && owner[n] === -1) {
          owner[n] = own;
          q[tail++] = n;
        }
      }
    }
  }

  const covered = new Uint8Array(nCells); // 1 when any drone covers a cell

  // Reused BFS buffers for "best nearby target" search.
  const bfsStamp = new Int32Array(nCells);
  const bfsDist = new Int32Array(nCells);
  const bfsParent = new Int32Array(nCells);
  const bfsQueue = new Int32Array(nCells);
  let stamp = 1;

  const rng = xorshift32((nCells ^ (maxStepsPerDrone * 2654435761)) >>> 0);

  const HEAT_WEIGHT = 0.12;
  const TERRITORY_BONUS = 1.25;
  const OUT_OF_TERRITORY_PENALTY = 0.85;
  const DIST_EXP = 1.15;
  const MAX_EXPANSIONS_BASE = 24000;
  const MAX_DIST_CAP = 220;
  const COMMIT_MIN = 10;
  const COMMIT_MAX = 24;
  const RECENT_WINDOW = 18;

  const distFactor = new Float32Array(MAX_DIST_CAP + 2);
  for (let d = 0; d < distFactor.length; d++) distFactor[d] = 1 / Math.pow(d + 1, DIST_EXP);

  function scoreCandidate(idx: number, dist: number, homeOwner: number): number {
    if (covered[idx] !== 0) return 0;
    const v = value[idx];
    if (v <= 0) return 0;
    const h = heat[idx];
    const base = v + HEAT_WEIGHT * (h - v);
    const territoryMul = owner[idx] === homeOwner ? TERRITORY_BONUS : OUT_OF_TERRITORY_PENALTY;
    return base * territoryMul * distFactor[dist];
  }

  function boundedBestPath(
    startIdx: number,
    homeOwner: number,
    remainingMoves: number,
    recent: Int32Array,
  ): number[] {
    const maxDist = Math.min(remainingMoves, MAX_DIST_CAP);
    // Scale expansions a bit with remaining budget.
    const maxExpansions = Math.min(nCells, MAX_EXPANSIONS_BASE + remainingMoves * 40);

    const myStamp = ++stamp;
    if (myStamp === 0x7fffffff) {
      bfsStamp.fill(0);
      stamp = 1;
    }

    let head = 0;
    let tail = 0;
    bfsStamp[startIdx] = myStamp;
    bfsDist[startIdx] = 0;
    bfsParent[startIdx] = -1;
    bfsQueue[tail++] = startIdx;

    let bestIdx = -1;
    let bestScore = 0;
    let expansions = 0;

    while (head < tail && expansions < maxExpansions) {
      const idx = bfsQueue[head++];
      expansions++;
      const dist = bfsDist[idx];

      const s = scoreCandidate(idx, dist, homeOwner);
      if (s > bestScore) {
        bestScore = s;
        bestIdx = idx;
      }

      if (dist >= maxDist) continue;

      const x = idx % W;
      const y = (idx / W) | 0;

      // Explore in a deterministic but mildly randomized order to avoid directional bias.
      // Order is a small permutation of [L, R, U, D] based on RNG.
      const r = rng() & 3;
      for (let k = 0; k < 4; k++) {
        const dir = (r + k) & 3;
        let n = -1;
        if (dir === 0) {
          if (x > 0) n = idx - 1;
        } else if (dir === 1) {
          if (x + 1 < W) n = idx + 1;
        } else if (dir === 2) {
          if (y > 0) n = idx - W;
        } else {
          if (y + 1 < H) n = idx + W;
        }
        if (n < 0) continue;
        if (passable[n] === 0) continue;
        if (bfsStamp[n] === myStamp) continue;
        bfsStamp[n] = myStamp;
        bfsDist[n] = dist + 1;
        bfsParent[n] = idx;
        bfsQueue[tail++] = n;
      }
    }

    if (bestIdx === -1) return [];

    // Reconstruct path (excluding startIdx).
    const rev: number[] = [];
    let cur = bestIdx;
    while (cur !== startIdx && cur !== -1) {
      rev.push(cur);
      cur = bfsParent[cur];
    }
    rev.reverse();

    // If best cell is in the recent window, drop it and instead use an early prefix.
    if (rev.length > 0) {
      const last = rev[rev.length - 1];
      for (let i = 0; i < recent.length; i++) {
        if (recent[i] === last) {
          rev.pop();
          break;
        }
      }
    }

    if (rev.length === 0) return [];

    // Commit a limited segment (receding-horizon style).
    const commit = Math.max(
      COMMIT_MIN,
      Math.min(COMMIT_MAX, Math.floor(Math.sqrt(remainingMoves + 1)) + 8),
    );
    if (rev.length > commit) rev.length = commit;
    return rev;
  }

  function bestLocalMove(currIdx: number, homeOwner: number, recent: Int32Array): number {
    let best = currIdx;
    let bestScore = -1e30;

    const x = currIdx % W;
    const y = (currIdx / W) | 0;

    // Candidate moves: stay, L, R, U, D
    // We prefer uncovering new value, then heat, while avoiding cycling.
    function evalMove(n: number): number {
      if (passable[n] === 0) return -1e30;
      let s = 0;
      const v = covered[n] === 0 ? value[n] : 0;
      s += v * 1.0;
      s += heat[n] * 0.02;
      s *= owner[n] === homeOwner ? 1.05 : 0.92;
      for (let i = 0; i < recent.length; i++) {
        if (recent[i] === n) {
          s -= 0.8;
          break;
        }
      }
      return s;
    }

    const stayScore = evalMove(currIdx);
    if (stayScore > bestScore) {
      bestScore = stayScore;
      best = currIdx;
    }
    if (x > 0) {
      const n = currIdx - 1;
      const s = evalMove(n);
      if (s > bestScore) {
        bestScore = s;
        best = n;
      }
    }
    if (x + 1 < W) {
      const n = currIdx + 1;
      const s = evalMove(n);
      if (s > bestScore) {
        bestScore = s;
        best = n;
      }
    }
    if (y > 0) {
      const n = currIdx - W;
      const s = evalMove(n);
      if (s > bestScore) {
        bestScore = s;
        best = n;
      }
    }
    if (y + 1 < H) {
      const n = currIdx + W;
      const s = evalMove(n);
      if (s > bestScore) {
        bestScore = s;
        best = n;
      }
    }
    return best;
  }

  function markCovered(idx: number) {
    if (covered[idx] === 0 && value[idx] > 0) covered[idx] = 1;
  }

  const paths: number[][][] = [];

  for (let i = 0; i < drones.length; i++) {
    const d = drones[i] as DroneInput;
    let x = clampInt(d.x, 0, W - 1);
    let y = clampInt(d.y, 0, H - 1);
    let currIdx = y * W + x;

    // If starting position is invalid, find nearest passable cell (local BFS).
    if (passable[currIdx] === 0) {
      const myStamp = ++stamp;
      if (myStamp === 0x7fffffff) {
        bfsStamp.fill(0);
        stamp = 1;
      }
      let head = 0;
      let tail = 0;
      bfsStamp[currIdx] = myStamp;
      bfsQueue[tail++] = currIdx;
      bfsParent[currIdx] = -1;
      let found = -1;
      while (head < tail) {
        const idx = bfsQueue[head++];
        if (passable[idx] !== 0) {
          found = idx;
          break;
        }
        const cx = idx % W;
        const cy = (idx / W) | 0;
        if (cx > 0) {
          const n = idx - 1;
          if (bfsStamp[n] !== myStamp) {
            bfsStamp[n] = myStamp;
            bfsQueue[tail++] = n;
            bfsParent[n] = idx;
          }
        }
        if (cx + 1 < W) {
          const n = idx + 1;
          if (bfsStamp[n] !== myStamp) {
            bfsStamp[n] = myStamp;
            bfsQueue[tail++] = n;
            bfsParent[n] = idx;
          }
        }
        if (cy > 0) {
          const n = idx - W;
          if (bfsStamp[n] !== myStamp) {
            bfsStamp[n] = myStamp;
            bfsQueue[tail++] = n;
            bfsParent[n] = idx;
          }
        }
        if (cy + 1 < H) {
          const n = idx + W;
          if (bfsStamp[n] !== myStamp) {
            bfsStamp[n] = myStamp;
            bfsQueue[tail++] = n;
            bfsParent[n] = idx;
          }
        }
      }
      if (found !== -1) currIdx = found;
    }

    const homeOwner = owner[currIdx] >= 0 ? owner[currIdx] : i;

    const recent = new Int32Array(RECENT_WINDOW);
    recent.fill(-1);
    let recentPos = 0;

    const dronePath: number[][] = [];
    {
      const sx = currIdx % W;
      const sy = (currIdx / W) | 0;
      dronePath.push([sx, sy]);
    }
    markCovered(currIdx);
    recent[recentPos++ % RECENT_WINDOW] = currIdx;

    let segment: number[] = [];

    for (let step = 0; step < maxStepsPerDrone; step++) {
      if (segment.length === 0) {
        segment = boundedBestPath(currIdx, homeOwner, maxStepsPerDrone - step, recent);
      }

      let nextIdx = currIdx;
      if (segment.length > 0) {
        nextIdx = segment.shift() as number;
      } else {
        nextIdx = bestLocalMove(currIdx, homeOwner, recent);
      }

      if (passable[nextIdx] === 0) nextIdx = currIdx;
      currIdx = nextIdx;

      const px = currIdx % W;
      const py = (currIdx / W) | 0;
      dronePath.push([px, py]);

      markCovered(currIdx);
      recent[recentPos++ % RECENT_WINDOW] = currIdx;
    }

    paths.push(dronePath);
  }

  process.stdout.write(JSON.stringify({ paths }));
}

main();
