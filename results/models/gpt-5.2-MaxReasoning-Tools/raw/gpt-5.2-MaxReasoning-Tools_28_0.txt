use std::collections::VecDeque;
use std::io::{self, Read, Write};

#[derive(Clone, Copy)]
struct Edge {
    u: u32,
    v: u32,
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_digit() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            x = x * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        Some(x)
    }
}

fn push_u64(out: &mut Vec<u8>, mut x: u64) {
    if x == 0 {
        out.push(b'0');
        return;
    }
    let mut buf = [0u8; 20];
    let mut len = 0usize;
    while x > 0 {
        buf[len] = b'0' + (x % 10) as u8;
        len += 1;
        x /= 10;
    }
    for i in (0..len).rev() {
        out.push(buf[i]);
    }
}

fn mis_dfs(adj: &[u64], candidates: u64, current: u64, best: &mut u64) {
    let current_size = current.count_ones() as usize;
    let best_size = best.count_ones() as usize;
    let remaining = candidates.count_ones() as usize;
    if current_size + remaining <= best_size {
        return;
    }
    if candidates == 0 {
        if current_size > best_size {
            *best = current;
        }
        return;
    }

    let mut tmp = candidates;
    let mut chosen = tmp.trailing_zeros() as usize;
    let mut chosen_deg = 0u32;
    while tmp != 0 {
        let v = tmp.trailing_zeros() as usize;
        tmp &= tmp - 1;
        let deg = (adj[v] & candidates).count_ones();
        if deg > chosen_deg {
            chosen_deg = deg;
            chosen = v;
        }
    }

    if chosen_deg == 0 {
        let full = current | candidates;
        if full.count_ones() > best.count_ones() {
            *best = full;
        }
        return;
    }

    let bit = 1u64 << chosen;
    let include_candidates = candidates & !bit & !adj[chosen];
    mis_dfs(adj, include_candidates, current | bit, best);
    mis_dfs(adj, candidates & !bit, current, best);
}

fn solve_small_exact(n: usize, edges: &[Edge], self_loop: &[bool]) -> Vec<u32> {
    let mut adj = vec![0u64; n];
    for &e in edges {
        let u = e.u as usize;
        let v = e.v as usize;
        if u == v {
            continue;
        }
        adj[u] |= 1u64 << v;
        adj[v] |= 1u64 << u;
    }

    let all = (1u64 << n) - 1;
    let mut candidates = all;
    for i in 0..n {
        if self_loop[i] {
            candidates &= !(1u64 << i);
        }
    }

    let mut best_set: u64 = 0;
    mis_dfs(&adj, candidates, 0, &mut best_set);

    let cover_bits = all & !best_set;
    let mut cover = Vec::new();
    for i in 0..n {
        if (cover_bits >> i) & 1 == 1 {
            cover.push(i as u32);
        }
    }
    cover
}

fn solve_large_approx_minimal(n: usize, edges: &[Edge], self_loop: &[bool]) -> Vec<u8> {
    let mut in_cover = vec![0u8; n];

    // 2-approximation via greedy maximal matching: include both endpoints.
    for &e in edges {
        let u = e.u as usize;
        let v = e.v as usize;
        if u == v {
            in_cover[u] = 1;
            continue;
        }
        if in_cover[u] == 0 && in_cover[v] == 0 {
            in_cover[u] = 1;
            in_cover[v] = 1;
        }
    }
    for i in 0..n {
        if self_loop[i] {
            in_cover[i] = 1;
        }
    }

    // Build CSR adjacency for iterative redundancy removal.
    let mut deg = vec![0u32; n];
    for &e in edges {
        let u = e.u as usize;
        let v = e.v as usize;
        if u == v {
            continue;
        }
        deg[u] += 1;
        deg[v] += 1;
    }

    let mut offsets = vec![0u32; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i];
    }

    let mut adj = vec![0u32; offsets[n] as usize];
    let mut cur = offsets.clone();
    for &e in edges {
        let u = e.u as usize;
        let v = e.v as usize;
        if u == v {
            continue;
        }
        let iu = cur[u] as usize;
        adj[iu] = v as u32;
        cur[u] += 1;
        let iv = cur[v] as usize;
        adj[iv] = u as u32;
        cur[v] += 1;
    }

    // outside_count[v] = number of incident edges from v to a vertex not in the cover.
    let mut outside_count = vec![0u32; n];
    for v in 0..n {
        if in_cover[v] == 0 || self_loop[v] {
            continue;
        }
        let start = offsets[v] as usize;
        let end = offsets[v + 1] as usize;
        let mut cnt = 0u32;
        for i in start..end {
            let u = adj[i] as usize;
            if in_cover[u] == 0 {
                cnt += 1;
            }
        }
        outside_count[v] = cnt;
    }

    // Iteratively remove redundant vertices (those with all neighbors still in the cover).
    let mut q = VecDeque::new();
    for v in 0..n {
        if in_cover[v] == 1 && !self_loop[v] && outside_count[v] == 0 {
            q.push_back(v);
        }
    }

    while let Some(v) = q.pop_front() {
        if in_cover[v] == 0 || self_loop[v] || outside_count[v] != 0 {
            continue;
        }
        in_cover[v] = 0;
        let start = offsets[v] as usize;
        let end = offsets[v + 1] as usize;
        for i in start..end {
            let u = adj[i] as usize;
            if in_cover[u] == 1 {
                outside_count[u] += 1;
            }
        }
    }

    in_cover
}

fn main() {
    let mut input = FastInput::new();
    let Some(n) = input.next_usize() else {
        return;
    };
    let m = input.next_usize().unwrap_or(0);

    if n == 0 {
        let mut out = Vec::new();
        out.extend_from_slice(b"0\n\n");
        io::stdout().write_all(&out).unwrap();
        return;
    }

    let mut edges = Vec::with_capacity(m);
    let mut self_loop = vec![false; n];
    for _ in 0..m {
        let u = input.next_usize().unwrap();
        let v = input.next_usize().unwrap();
        edges.push(Edge {
            u: u as u32,
            v: v as u32,
        });
        if u == v {
            self_loop[u] = true;
        }
    }

    // Exact on very small graphs; otherwise a fast 2-approx with redundancy pruning.
    let cover: Vec<u32> = if n <= 60 {
        solve_small_exact(n, &edges, &self_loop)
    } else {
        let in_cover = solve_large_approx_minimal(n, &edges, &self_loop);
        let mut res = Vec::new();
        for (i, &b) in in_cover.iter().enumerate() {
            if b == 1 {
                res.push(i as u32);
            }
        }
        res
    };

    let mut out = Vec::new();
    push_u64(&mut out, cover.len() as u64);
    out.push(b'\n');
    for (i, v) in cover.iter().enumerate() {
        if i > 0 {
            out.push(b' ');
        }
        push_u64(&mut out, *v as u64);
    }
    out.push(b'\n');
    io::stdout().write_all(&out).unwrap();
}
