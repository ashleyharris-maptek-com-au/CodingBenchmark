using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const double G = 6.67430e-11;        // m^3 kg^-1 s^-2
    private const double SecondsPerDay = 86400.0;
    private const double EarthRadiusMeters = 6_371_000.0;

    private static readonly CultureInfo Invariant = CultureInfo.InvariantCulture;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        if (!fs.TryNext(out string first))
            return;

        int numBodies = int.Parse(first, Invariant);
        double warningDays = fs.NextDouble();
        double impactorMass = fs.NextDouble();
        double deltaVBudget = fs.NextDouble();
        double asteroidMassInput = fs.NextDouble(); // some sims repeat this; trust "Asteroid" body for mass if present
        _ = fs.NextDouble(); // t0_unix_utc (unused in this lightweight solver)

        var bodies = new Dictionary<string, Body>(StringComparer.OrdinalIgnoreCase);
        var list = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            string name = fs.Next();
            double mass = fs.NextDouble();
            var pos = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            var vel = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            var b = new Body(name, mass, pos, vel);
            bodies[name] = b;
            list.Add(b);
        }

        // Path to DE421 kernel (not used by this simplified solver).
        // Still consume the token if present.
        fs.TryNext(out _);

        if (!bodies.TryGetValue("Earth", out var earthBody) || !bodies.TryGetValue("Asteroid", out var asteroidBody))
        {
            // Output a valid, minimal plan.
            WritePlan(0.0, new Vec3(0, 0, 0), Math.Max(0.0, warningDays));
            return;
        }

        // Choose Sun as explicit "Sun" body if available, else most massive.
        Body sunBody;
        if (!bodies.TryGetValue("Sun", out sunBody))
        {
            sunBody = list[0];
            for (int i = 1; i < list.Count; i++)
                if (list[i].Mass > sunBody.Mass)
                    sunBody = list[i];
        }

        double asteroidMass = asteroidBody.Mass > 0 ? asteroidBody.Mass : asteroidMassInput;
        if (asteroidMass <= 0) asteroidMass = Math.Max(1.0, asteroidMassInput);

        // Shift to Sun-centered frame for 2-body dynamics around the Sun.
        Vec3 rE0 = earthBody.Pos - sunBody.Pos;
        Vec3 vE0 = earthBody.Vel - sunBody.Vel;
        Vec3 rA0 = asteroidBody.Pos - sunBody.Pos;
        Vec3 vA0 = asteroidBody.Vel - sunBody.Vel;

        double muSun = G * Math.Max(1.0, sunBody.Mass);

        // Coarse search over launch/impact times: choose a plan that fits delta-v budget and maximizes
        // the post-impact miss distance at t=warningDays in this simplified model.
        var best = Plan.None;

        // Adapt sampling density to warning time (bounded for speed).
        int launchCount = Clamp((int)Math.Round(6 + Math.Log10(1 + warningDays) * 4), 6, 18);
        int impactCount = Clamp((int)Math.Round(10 + Math.Log10(1 + warningDays) * 6), 10, 28);

        double minFlightDays = Math.Max(7.0, Math.Min(90.0, warningDays * 0.01));
        double maxLaunchDays = warningDays - minFlightDays;
        if (maxLaunchDays < 0.0) maxLaunchDays = 0.0;

        for (int li = 0; li < launchCount; li++)
        {
            double tLaunch = (launchCount == 1) ? 0.0 : (maxLaunchDays * li / (launchCount - 1));

            // Prefer earlier impact for more leverage, but still allow broad search.
            double minImpact = tLaunch + minFlightDays;
            if (minImpact > warningDays) continue;

            for (int ii = 0; ii < impactCount; ii++)
            {
                double frac = (impactCount == 1) ? 0.5 : (double)ii / (impactCount - 1);
                // Quadratic bias to earlier impacts
                frac = frac * frac;
                double tImpact = Lerp(minImpact, warningDays, frac);
                if (tImpact <= tLaunch) continue;

                double tofSec = (tImpact - tLaunch) * SecondsPerDay;
                if (tofSec <= 0.0) continue;

                PropagateKepler(rE0, vE0, muSun, tLaunch * SecondsPerDay, out Vec3 rE1, out Vec3 vE1);
                PropagateKepler(rA0, vA0, muSun, tImpact * SecondsPerDay, out Vec3 rA2, out Vec3 vA2);

                // Lambert from Earth@launch to asteroid-position@impact, 2-body about Sun.
                if (!LambertUniversal(rE1, rA2, tofSec, muSun, shortWay: true, out Vec3 vSc1s, out Vec3 vSc2s))
                    continue;

                // Try both short and long way; keep the better scoring one.
                EvaluateCandidate(ref best, warningDays, impactorMass, asteroidMass, deltaVBudget,
                    tLaunch, tImpact, rE0, vE0, rA0, vA0, muSun, rE1, vE1, rA2, vA2, vSc1s, vSc2s);

                if (LambertUniversal(rE1, rA2, tofSec, muSun, shortWay: false, out Vec3 vSc1l, out Vec3 vSc2l))
                {
                    EvaluateCandidate(ref best, warningDays, impactorMass, asteroidMass, deltaVBudget,
                        tLaunch, tImpact, rE0, vE0, rA0, vA0, muSun, rE1, vE1, rA2, vA2, vSc1l, vSc2l);
                }
            }
        }

        if (!best.IsValid)
        {
            // Fall back to a minimal valid output.
            WritePlan(0.0, new Vec3(0, 0, 0), Math.Max(0.0, warningDays));
            return;
        }

        WritePlan(best.LaunchDay, best.LaunchDV, best.ImpactDay);
    }

    private static void EvaluateCandidate(
        ref Plan best,
        double warningDays,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget,
        double tLaunch,
        double tImpact,
        Vec3 rE0,
        Vec3 vE0,
        Vec3 rA0,
        Vec3 vA0,
        double muSun,
        Vec3 rE1,
        Vec3 vE1,
        Vec3 rA2,
        Vec3 vA2,
        Vec3 vSc1,
        Vec3 vSc2)
    {
        Vec3 dvLaunch = vSc1 - vE1;
        double dvMag = dvLaunch.Norm();
        if (!double.IsFinite(dvMag) || dvMag > deltaVBudget)
            return;

        // Kinetic impact momentum transfer (perfectly inelastic, beta=1).
        Vec3 vRel = vSc2 - vA2;
        Vec3 dvAst = (impactorMass / asteroidMass) * vRel;
        Vec3 vA2New = vA2 + dvAst;

        double remainingDays = warningDays - tImpact;
        if (remainingDays < 0.0) remainingDays = 0.0;

        PropagateKepler(rE0, vE0, muSun, warningDays * SecondsPerDay, out Vec3 rEEnd, out _);
        PropagateKepler(rA2, vA2New, muSun, remainingDays * SecondsPerDay, out Vec3 rAEnd, out _);

        double dist = (rAEnd - rEEnd).Norm();
        double miss = dist - EarthRadiusMeters;

        // Prefer larger miss; tie-break by smaller delta-v.
        if (!best.IsValid ||
            miss > best.MissMeters + 1.0 ||
            (Math.Abs(miss - best.MissMeters) <= 1.0 && dvMag < best.LaunchDV.Norm()))
        {
            best = new Plan(true, tLaunch, tImpact, dvLaunch, miss);
        }
    }

    private static void WritePlan(double launchDay, Vec3 launchDV, double impactDay)
    {
        // Always emit at least launch + impact line.
        // Burn vector is delta-v in m/s in inertial barycentric coordinates (translation cancels between frames).
        var sb = new StringBuilder(256);
        sb.AppendLine("2");
        sb.Append(FormatDay(launchDay)).Append(' ')
          .Append(Format(launchDV.X)).Append(' ')
          .Append(Format(launchDV.Y)).Append(' ')
          .Append(Format(launchDV.Z)).AppendLine();
        sb.Append(FormatDay(impactDay)).Append(" 0 0 0").AppendLine();
        Console.Out.Write(sb.ToString());
    }

    private static string Format(double x) => x.ToString("R", Invariant);
    private static string FormatDay(double day)
    {
        if (day < 0) day = 0;
        return day.ToString("R", Invariant);
    }

    // --- Astrodynamics primitives (2-body) ---

    private static void PropagateKepler(in Vec3 r0, in Vec3 v0, double mu, double dtSec, out Vec3 r, out Vec3 v)
    {
        // Universal variable formulation (handles all conic sections).
        double r0mag = r0.Norm();
        if (r0mag <= 0 || !double.IsFinite(r0mag) || mu <= 0 || !double.IsFinite(mu))
        {
            r = r0;
            v = v0;
            return;
        }

        double v0sq = v0.Dot(v0);
        double alpha = 2.0 / r0mag - v0sq / mu;
        double sqrtMu = Math.Sqrt(mu);

        double chi;
        if (Math.Abs(alpha) > 1e-12)
            chi = sqrtMu * Math.Abs(alpha) * Math.Abs(dtSec);
        else
            chi = sqrtMu * Math.Abs(dtSec) / r0mag;

        if (dtSec < 0) chi = -chi;

        // Newton solve for universal anomaly chi.
        for (int iter = 0; iter < 50; iter++)
        {
            double z = alpha * chi * chi;
            Stumpff(z, out double c2, out double c3);

            double r0v0 = r0.Dot(v0);
            double chi2 = chi * chi;
            double chi3 = chi2 * chi;

            double F = (r0v0 / sqrtMu) * chi2 * c2
                       + (1.0 - alpha * r0mag) * chi3 * c3
                       + r0mag * chi
                       - sqrtMu * dtSec;

            double dF = (r0v0 / sqrtMu) * chi * (1.0 - z * c3)
                        + (1.0 - alpha * r0mag) * chi2 * c2
                        + r0mag * (1.0 - z * c2);

            if (Math.Abs(dF) < 1e-14) break;
            double delta = F / dF;
            chi -= delta;
            if (Math.Abs(delta) < 1e-9) break;
        }

        {
            double z = alpha * chi * chi;
            Stumpff(z, out double c2, out double c3);

            double chi2 = chi * chi;
            double chi3 = chi2 * chi;

            double f = 1.0 - (chi2 / r0mag) * c2;
            double g = dtSec - (chi3 / sqrtMu) * c3;

            r = f * r0 + g * v0;
            double rmag = r.Norm();
            if (rmag <= 0 || !double.IsFinite(rmag))
            {
                v = v0;
                return;
            }

            double fdot = (sqrtMu / (rmag * r0mag)) * (alpha * chi3 * c3 - chi);
            double gdot = 1.0 - (chi2 / rmag) * c2;
            v = fdot * r0 + gdot * v0;
        }
    }

    private static bool LambertUniversal(in Vec3 r1, in Vec3 r2, double dtSec, double mu, bool shortWay, out Vec3 v1, out Vec3 v2)
    {
        v1 = default;
        v2 = default;

        if (dtSec <= 0 || mu <= 0) return false;

        double r1mag = r1.Norm();
        double r2mag = r2.Norm();
        if (r1mag <= 0 || r2mag <= 0) return false;

        double cosDnu = Clamp(r1.Dot(r2) / (r1mag * r2mag), -1.0, 1.0);
        double A = (shortWay ? 1.0 : -1.0) * Math.Sqrt(Math.Max(0.0, r1mag * r2mag * (1.0 + cosDnu)));
        if (Math.Abs(A) < 1e-12) return false;

        double psiLow = -4.0 * Math.PI * Math.PI;
        double psiUp = 4.0 * Math.PI * Math.PI;
        double psi = 0.0;

        double sqrtMu = Math.Sqrt(mu);
        double dtTarget = dtSec;

        // Ensure y stays positive by nudging lower bound if needed.
        for (int guard = 0; guard < 8; guard++)
        {
            Stumpff(psiLow, out double c2l, out double c3l);
            double yLow = r1mag + r2mag + A * (psiLow * c3l - 1.0) / Math.Sqrt(Math.Max(1e-30, c2l));
            if (yLow > 0) break;
            psiLow *= 0.5;
        }

        double y = 0.0, chi = 0.0, dtNew = 0.0, c2 = 0.0, c3 = 0.0;
        for (int iter = 0; iter < 60; iter++)
        {
            Stumpff(psi, out c2, out c3);
            if (c2 <= 0.0) { psi = 0.5 * (psi + psiUp); continue; }

            y = r1mag + r2mag + A * (psi * c3 - 1.0) / Math.Sqrt(c2);
            if (y < 0.0)
            {
                // Increase psi until y becomes positive.
                psiLow = psi;
                psi = 0.5 * (psiLow + psiUp);
                continue;
            }

            chi = Math.Sqrt(y / c2);
            dtNew = (chi * chi * chi * c3 + A * Math.Sqrt(y)) / sqrtMu;

            if (Math.Abs(dtNew - dtTarget) < 1e-6) break;

            if (dtNew <= dtTarget)
                psiLow = psi;
            else
                psiUp = psi;

            psi = 0.5 * (psiLow + psiUp);
        }

        if (!(y > 0.0) || !(dtNew > 0.0)) return false;

        double f = 1.0 - y / r1mag;
        double g = A * Math.Sqrt(y / mu);
        double gdot = 1.0 - y / r2mag;

        if (Math.Abs(g) < 1e-12) return false;

        v1 = (r2 - f * r1) / g;
        v2 = (gdot * r2 - r1) / g;
        return v1.IsFinite && v2.IsFinite;
    }

    private static void Stumpff(double z, out double c2, out double c3)
    {
        if (Math.Abs(z) < 1e-8)
        {
            c2 = 0.5;
            c3 = 1.0 / 6.0;
            return;
        }

        if (z > 0.0)
        {
            double sz = Math.Sqrt(z);
            c2 = (1.0 - Math.Cos(sz)) / z;
            c3 = (sz - Math.Sin(sz)) / (sz * sz * sz);
            return;
        }
        else
        {
            double sz = Math.Sqrt(-z);
            c2 = (Math.Cosh(sz) - 1.0) / (-z);
            c3 = (Math.Sinh(sz) - sz) / (sz * sz * sz);
            return;
        }
    }

    private static int Clamp(int v, int lo, int hi) => v < lo ? lo : (v > hi ? hi : v);
    private static double Clamp(double v, double lo, double hi) => v < lo ? lo : (v > hi ? hi : v);
    private static double Lerp(double a, double b, double t) => a + (b - a) * t;

    private readonly record struct Plan(bool IsValid, double LaunchDay, double ImpactDay, Vec3 LaunchDV, double MissMeters)
    {
        public static readonly Plan None = new(false, 0, 0, default, double.NegativeInfinity);
    }

    private readonly record struct Body(string Name, double Mass, Vec3 Pos, Vec3 Vel);

    public readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public double Dot(in Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm() => Math.Sqrt(Dot(this));

        public bool IsFinite => double.IsFinite(X) && double.IsFinite(Y) && double.IsFinite(Z);

        public static Vec3 operator +(in Vec3 a, in Vec3 b) => new(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(in Vec3 a, in Vec3 b) => new(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(double s, in Vec3 a) => new(s * a.X, s * a.Y, s * a.Z);
        public static Vec3 operator *(in Vec3 a, double s) => new(s * a.X, s * a.Y, s * a.Z);
        public static Vec3 operator /(in Vec3 a, double s) => new(a.X / s, a.Y / s, a.Z / s);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryNext(out string token)
        {
            token = "";
            int c;
            do
            {
                c = Read();
                if (c == -1) return false;
            } while (c <= 32);

            var sb = new StringBuilder(32);
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
                if (c == -1) break;
            }
            token = sb.ToString();
            return true;
        }

        public string Next()
        {
            if (!TryNext(out var t))
                throw new EndOfStreamException();
            return t;
        }

        public double NextDouble() => double.Parse(Next(), Invariant);
    }
}
