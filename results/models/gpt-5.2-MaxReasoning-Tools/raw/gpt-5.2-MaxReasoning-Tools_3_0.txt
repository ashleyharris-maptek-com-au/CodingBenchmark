use std::io::{self, Read};

const GOLDEN_ANGLE: f32 = 2.39996323; // pi * (3 - sqrt(5))
const EPS: f32 = 1e-3;
const MIN_HALF: f32 = 1e-3;

#[derive(Clone)]
struct QuadNode {
    cx: f32,
    cy: f32,
    half: f32,
    mass: f32,
    sum_x: f32,
    sum_y: f32,
    point: i32,        // body index if leaf with single stored body, else -1
    child: [i32; 4],   // indices into nodes vec, -1 if absent
}

impl QuadNode {
    fn new(cx: f32, cy: f32, half: f32) -> Self {
        Self {
            cx,
            cy,
            half,
            mass: 0.0,
            sum_x: 0.0,
            sum_y: 0.0,
            point: -1,
            child: [-1, -1, -1, -1],
        }
    }

    #[inline]
    fn is_leaf(&self) -> bool {
        self.child[0] < 0
    }
}

#[inline]
fn quad_index(cx: f32, cy: f32, x: f32, y: f32) -> usize {
    let mut idx = 0usize;
    if x >= cx {
        idx |= 1;
    }
    if y >= cy {
        idx |= 2;
    }
    idx
}

fn insert_body(nodes: &mut Vec<QuadNode>, node_idx: usize, body_idx: usize, x: &[f32], y: &[f32]) {
    let px = x[body_idx];
    let py = y[body_idx];

    // Update aggregate at this node.
    {
        let n = &mut nodes[node_idx];
        n.mass += 1.0;
        n.sum_x += px;
        n.sum_y += py;
    }

    let is_leaf = nodes[node_idx].is_leaf();
    if is_leaf {
        let point = nodes[node_idx].point;
        if point < 0 {
            nodes[node_idx].point = body_idx as i32;
            return;
        }

        // Already contains a point; subdivide if possible.
        if nodes[node_idx].half <= MIN_HALF {
            // Too small to subdivide; keep as an aggregate leaf.
            return;
        }

        let existing = point as usize;
        nodes[node_idx].point = -1;

        let (cx, cy, half) = {
            let n = &nodes[node_idx];
            (n.cx, n.cy, n.half)
        };
        let child_half = half * 0.5;

        // Create 4 children.
        for q in 0..4 {
            let child_cx = cx + if (q & 1) != 0 { child_half } else { -child_half };
            let child_cy = cy + if (q & 2) != 0 { child_half } else { -child_half };
            let child_idx = nodes.len() as i32;
            nodes.push(QuadNode::new(child_cx, child_cy, child_half));
            nodes[node_idx].child[q] = child_idx;
        }

        // Insert existing and new points.
        let q1 = quad_index(cx, cy, x[existing], y[existing]);
        let c1 = nodes[node_idx].child[q1] as usize;
        insert_body(nodes, c1, existing, x, y);

        let q2 = quad_index(cx, cy, px, py);
        let c2 = nodes[node_idx].child[q2] as usize;
        insert_body(nodes, c2, body_idx, x, y);
        return;
    }

    // Internal node: insert into the appropriate child.
    let (cx, cy) = {
        let n = &nodes[node_idx];
        (n.cx, n.cy)
    };
    let q = quad_index(cx, cy, px, py);
    let child_idx = nodes[node_idx].child[q] as usize;
    insert_body(nodes, child_idx, body_idx, x, y);
}

fn build_quadtree(x: &[f32], y: &[f32]) -> Vec<QuadNode> {
    let n = x.len();
    let mut min_x = f32::INFINITY;
    let mut max_x = f32::NEG_INFINITY;
    let mut min_y = f32::INFINITY;
    let mut max_y = f32::NEG_INFINITY;
    for i in 0..n {
        let px = x[i];
        let py = y[i];
        if px < min_x {
            min_x = px;
        }
        if px > max_x {
            max_x = px;
        }
        if py < min_y {
            min_y = py;
        }
        if py > max_y {
            max_y = py;
        }
    }

    let w = (max_x - min_x).max(max_y - min_y).max(1.0);
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let half = w * 0.5 + 1e-2;

    let mut nodes = Vec::with_capacity((n * 2).max(16));
    nodes.push(QuadNode::new(cx, cy, half));
    for i in 0..n {
        insert_body(&mut nodes, 0, i, x, y);
    }
    nodes
}

#[inline]
fn choose_iterations(n: usize, m: usize) -> usize {
    // More iterations for smaller graphs; cap for very large graphs.
    if n <= 50 {
        400
    } else if n <= 200 {
        300
    } else if n <= 1_000 {
        200
    } else if n <= 5_000 {
        120
    } else if n <= 20_000 {
        60
    } else if n <= 100_000 {
        25
    } else {
        // If extremely sparse, a few extra steps help; otherwise keep it low.
        if m <= n.saturating_mul(3) {
            15
        } else {
            10
        }
    }
}

#[inline]
fn choose_theta(n: usize) -> f32 {
    if n <= 2_000 {
        0.7
    } else if n <= 20_000 {
        0.8
    } else {
        0.9
    }
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut val: usize = 0;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            let c = self.buf[self.idx];
            if c.is_ascii_digit() {
                val = val.saturating_mul(10).saturating_add((c - b'0') as usize);
            }
            self.idx += 1;
        }
        Some(val)
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m_in = sc.next_usize().unwrap_or(0);

    let mut eu: Vec<u32> = Vec::with_capacity(m_in);
    let mut ev: Vec<u32> = Vec::with_capacity(m_in);
    let mut deg: Vec<u32> = vec![0; n];

    for _ in 0..m_in {
        let u = sc.next_usize().unwrap_or(0);
        let v = sc.next_usize().unwrap_or(0);
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            continue; // ignore self-loops for layout forces
        }
        eu.push(u as u32);
        ev.push(v as u32);
        deg[u] = deg[u].saturating_add(1);
        deg[v] = deg[v].saturating_add(1);
    }

    // Build CSR adjacency for traversal/ordering.
    let mut offsets: Vec<usize> = vec![0; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i] as usize;
    }
    let mut adj: Vec<u32> = vec![0; offsets[n]];
    let mut next = offsets.clone();
    for e in 0..eu.len() {
        let u = eu[e] as usize;
        let v = ev[e] as usize;
        let iu = next[u];
        adj[iu] = v as u32;
        next[u] = iu + 1;
        let iv = next[v];
        adj[iv] = u as u32;
        next[v] = iv + 1;
    }

    // Reverse Cuthill-McKee (per component) to get a decent initial order.
    let mut visited: Vec<u8> = vec![0; n];
    let mut order: Vec<usize> = Vec::with_capacity(n);
    let mut comp_ranges: Vec<(usize, usize)> = Vec::new();
    let mut queue: Vec<usize> = Vec::new();
    let mut comp_order: Vec<usize> = Vec::new();
    let mut scratch: Vec<usize> = Vec::new();

    for start in 0..n {
        if visited[start] != 0 {
            continue;
        }
        visited[start] = 1;
        queue.clear();
        comp_order.clear();
        queue.push(start);
        let mut head = 0usize;
        while head < queue.len() {
            let u = queue[head];
            head += 1;
            comp_order.push(u);

            scratch.clear();
            let a = offsets[u];
            let b = offsets[u + 1];
            for idx in a..b {
                let v = adj[idx] as usize;
                if visited[v] == 0 {
                    visited[v] = 1;
                    scratch.push(v);
                }
            }
            scratch.sort_unstable_by_key(|&v| deg[v]);
            queue.extend_from_slice(&scratch);
        }
        comp_order.reverse();
        let s = order.len();
        order.extend_from_slice(&comp_order);
        let e = order.len();
        comp_ranges.push((s, e));
    }

    // Initial positions: sunflower (golden-angle) spiral per component, packed on a grid.
    let mut x: Vec<f32> = vec![0.0; n];
    let mut y: Vec<f32> = vec![0.0; n];
    let num_comps = comp_ranges.len().max(1);
    let mut max_rad = 1.0f32;
    for &(a, b) in &comp_ranges {
        let k = (b - a) as f32;
        max_rad = max_rad.max(k.sqrt().max(1.0));
    }
    let spacing = max_rad * 4.0 + 10.0;
    let grid_w = ((num_comps as f64).sqrt().ceil() as usize).max(1);
    let grid_h = (num_comps + grid_w - 1) / grid_w;

    for (ci, &(a, b)) in comp_ranges.iter().enumerate() {
        let gx = (ci % grid_w) as f32;
        let gy = (ci / grid_w) as f32;
        let cx = (gx - (grid_w as f32 - 1.0) * 0.5) * spacing;
        let cy = (gy - (grid_h as f32 - 1.0) * 0.5) * spacing;

        let k = b - a;
        for local in 0..k {
            let v = order[a + local];
            let t = (local as f32 + 0.5) * GOLDEN_ANGLE;
            let r = (local as f32 + 0.5).sqrt();
            x[v] = cx + r * t.cos();
            y[v] = cy + r * t.sin();
        }
    }

    let iters = choose_iterations(n, eu.len());
    let theta = choose_theta(n);
    let k: f32 = 1.0;
    let k2: f32 = k * k;
    let mut disp_x: Vec<f32> = vec![0.0; n];
    let mut disp_y: Vec<f32> = vec![0.0; n];
    let mut stack: Vec<usize> = Vec::new();

    // Edge sampling for very dense graphs.
    let m_eff = eu.len();
    let target_edges = if n <= 2_000 || m_eff <= n.saturating_mul(10) {
        m_eff
    } else {
        n.saturating_mul(10).max(1)
    };
    let stride = if target_edges == 0 {
        1usize
    } else {
        (m_eff + target_edges - 1) / target_edges
    };
    let edge_scale = (stride as f32).min(50.0);

    let t0 = (n as f32).sqrt() * 0.1 + 0.5;
    for iter in 0..iters {
        // Reset displacements.
        disp_x.fill(0.0);
        disp_y.fill(0.0);

        // Barnes-Hut repulsion.
        let nodes = build_quadtree(&x, &y);
        for i in 0..n {
            stack.clear();
            stack.push(0);
            let xi = x[i];
            let yi = y[i];
            let mut fx = 0.0f32;
            let mut fy = 0.0f32;
            while let Some(ni) = stack.pop() {
                let node = &nodes[ni];
                if node.mass == 0.0 {
                    continue;
                }
                let com_x = node.sum_x / node.mass;
                let com_y = node.sum_y / node.mass;
                let dx = xi - com_x;
                let dy = yi - com_y;
                let dist2 = dx * dx + dy * dy + EPS;

                if node.is_leaf() {
                    if node.mass == 1.0 && node.point == i as i32 {
                        continue;
                    }
                    let f = k2 * node.mass / dist2;
                    fx += dx * f;
                    fy += dy * f;
                } else {
                    let dist = dist2.sqrt();
                    let size = node.half * 2.0;
                    if size / dist < theta {
                        let f = k2 * node.mass / dist2;
                        fx += dx * f;
                        fy += dy * f;
                    } else {
                        for &c in &node.child {
                            if c >= 0 {
                                stack.push(c as usize);
                            }
                        }
                    }
                }
            }
            disp_x[i] += fx;
            disp_y[i] += fy;
        }

        // Edge attraction.
        if m_eff > 0 {
            let mut e = 0usize;
            while e < m_eff {
                let u = eu[e] as usize;
                let v = ev[e] as usize;
                let dx = x[u] - x[v];
                let dy = y[u] - y[v];
                let dist2 = dx * dx + dy * dy + EPS;
                let dist = dist2.sqrt();
                let factor = (dist / k) * edge_scale;
                disp_x[u] -= dx * factor;
                disp_y[u] -= dy * factor;
                disp_x[v] += dx * factor;
                disp_y[v] += dy * factor;
                e += stride;
            }
        }

        // Move with cooling schedule.
        let frac = 1.0 - (iter as f32) / (iters as f32);
        let t = (t0 * frac).max(0.01);
        let mut sum_x = 0.0f32;
        let mut sum_y = 0.0f32;
        for i in 0..n {
            let dx = disp_x[i];
            let dy = disp_y[i];
            let len2 = dx * dx + dy * dy;
            if len2 > 0.0 {
                let len = len2.sqrt();
                let step = t.min(len);
                x[i] += dx / len * step;
                y[i] += dy / len * step;
            }
            sum_x += x[i];
            sum_y += y[i];
        }

        // Recentre to avoid drift.
        let mx = sum_x / (n as f32);
        let my = sum_y / (n as f32);
        for i in 0..n {
            x[i] -= mx;
            y[i] -= my;
        }
    }

    // Normalize to a reasonable numeric range (helps avoid degeneracies).
    let mut min_x = f32::INFINITY;
    let mut max_x = f32::NEG_INFINITY;
    let mut min_y = f32::INFINITY;
    let mut max_y = f32::NEG_INFINITY;
    for i in 0..n {
        min_x = min_x.min(x[i]);
        max_x = max_x.max(x[i]);
        min_y = min_y.min(y[i]);
        max_y = max_y.max(y[i]);
    }
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let span = (max_x - min_x).max(max_y - min_y).max(1.0);
    let scale = 1000.0f32 / span;
    for i in 0..n {
        x[i] = (x[i] - cx) * scale;
        y[i] = (y[i] - cy) * scale;
    }

    // Output.
    use std::fmt::Write as _;
    let mut out = String::new();
    out.reserve(n.saturating_mul(24));
    for i in 0..n {
        let _ = writeln!(&mut out, "{:.6} {:.6}", x[i] as f64, y[i] as f64);
    }
    print!("{out}");
}
