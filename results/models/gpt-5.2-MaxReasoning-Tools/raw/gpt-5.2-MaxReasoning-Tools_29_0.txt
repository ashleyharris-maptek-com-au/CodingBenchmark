using System;
using System.Diagnostics;
using System.IO;
using System.Numerics;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    // Practical limits / switches.
    private const int DegeneracyLimitForPerVertex = 2048;   // Forward-neighbour subgraphs capped by degeneracy.
    private const int BitsetPrimaryNMax = 30000;            // If graph is dense enough, store as bitset matrix.
    private const int BitsetFallbackNMax = 20000;           // If sparse path degeneracy is large, build bitset and run BBMC.

    // Time management: stop a bit before 5 minutes.
    private const long TimeLimitMs = 295_000;

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        var fs = new FastScanner(Console.OpenStandardInput());

        int n;
        int m;
        try
        {
            n = fs.NextInt();
            m = fs.NextInt();
        }
        catch
        {
            return;
        }

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        // Don't infer completeness from m alone (inputs may contain duplicates).
        // If the graph is complete, the solvers below will quickly find the size-n clique.

        // Decide whether to store the graph as a dense bitset adjacency matrix directly.
        // Bitset memory ~ n^2/8 bytes. List memory ~ 16m bytes (two int arrays over 2m edges).
        bool denseBitset =
            n <= BitsetPrimaryNMax &&
            ((long)m * 128L > (long)n * (n - 1)) && // break-even heuristic (bitset smaller than adjacency list)
            ((long)n * (((long)n + 63) / 64) <= int.MaxValue); // array length safety

        int bestSize = 1;
        int[] bestClique = new[] { 0 };
        object bestLock = new object();
        if (denseBitset)
        {
            int words = (n + 63) >> 6;
            var adj = new ulong[n * words];
            var degree = new int[n];

            for (int i = 0; i < m; i++)
            {
                int u = fs.NextInt();
                int v = fs.NextInt();
                if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;

                // Set u-v and v-u
                int uw = u * words + (v >> 6);
                int vw = v * words + (u >> 6);
                ulong um = 1UL << (v & 63);
                ulong vm = 1UL << (u & 63);
                if ((adj[uw] & um) == 0)
                {
                    adj[uw] |= um;
                    adj[vw] |= vm;
                    degree[u]++;
                    degree[v]++;
                }
            }

            // Fast-path for complete graphs after de-duplication.
            bool complete = true;
            for (int i = 0; i < n; i++)
            {
                if (degree[i] != n - 1) { complete = false; break; }
            }
            if (complete)
            {
                Console.WriteLine(n);
                var sbAll = new StringBuilder(Math.Min(n * 2, 1_000_000));
                for (int i = 0; i < n; i++)
                {
                    if (i != 0) sbAll.Append(' ');
                    sbAll.Append(i);
                }
                Console.WriteLine(sbAll.ToString());
                return;
            }

            int upperBound = n; // trivial.
            var solver = new BitsetCliqueSolver(n, words, adj, sw, TimeLimitMs);
            int[] clique = solver.Solve(out int cliqueSize);

            if (cliqueSize > 0)
            {
                bestSize = cliqueSize;
                bestClique = clique;
            }
        }
        else
        {
            // Sparse adjacency lists (forward-star).
            // Memory: to/next arrays are sized to 2m (undirected).
            int[] head = new int[n];
            Array.Fill(head, -1);
            int[] to = new int[2 * m];
            int[] next = new int[2 * m];
            int[] degree = new int[n];

            int ei = 0;
            for (int i = 0; i < m; i++)
            {
                int u = fs.NextInt();
                int v = fs.NextInt();
                if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;

                to[ei] = v;
                next[ei] = head[u];
                head[u] = ei++;

                to[ei] = u;
                next[ei] = head[v];
                head[v] = ei++;

                degree[u]++;
                degree[v]++;
            }

            // Compute degeneracy ordering (removal order) + degeneracy value.
            var order = new int[n];
            var rank = new int[n];
            int degeneracy = ComputeDegeneracyOrdering(n, head, to, next, degree, order, rank);
            int globalUpperBound = degeneracy + 1; // clique number <= degeneracy+1

            // Quick win: if degeneracy is 0 then no edges => clique size is 1.
            if (degeneracy == 0)
            {
                bestSize = 1;
                bestClique = new[] { 0 };
            }
            else if (degeneracy <= DegeneracyLimitForPerVertex)
            {
                // Build a degeneracy orientation: direct each undirected edge from lower-rank to higher-rank.
                // Then out-neighbors correspond exactly to "later neighbors" (bounded by degeneracy).
                int undirectedEdges = ei / 2;
                int[] outHead = new int[n];
                Array.Fill(outHead, -1);
                int[] outTo = new int[undirectedEdges];
                int[] outNext = new int[undirectedEdges];
                int oe = 0;

                for (int v = 0; v < n; v++)
                {
                    int rv = rank[v];
                    for (int e = head[v]; e != -1; e = next[e])
                    {
                        int u = to[e];
                        if (rv < rank[u])
                        {
                            outTo[oe] = u;
                            outNext[oe] = outHead[v];
                            outHead[v] = oe;
                            oe++;
                        }
                    }
                }

                // Per-vertex exact search on induced subgraphs of forward neighbors.
                // Parallelize across vertices; per thread uses its own stamp arrays to avoid contention.
                int maxForward = degeneracy; // each vertex has <= degeneracy forward neighbors.

                var options = new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount };
                var ctxLocal = new ThreadLocal<PerThreadContext>(() => new PerThreadContext(n, maxForward, sw, TimeLimitMs));

                Parallel.For(0, n, options, (idx, loopState) =>
                {
                    if (sw.ElapsedMilliseconds >= TimeLimitMs)
                    {
                        loopState.Stop();
                        return;
                    }

                    int currentBest = Volatile.Read(ref bestSize);
                    if (currentBest >= globalUpperBound)
                    {
                        loopState.Stop();
                        return;
                    }

                    int v = order[idx];
                    var ctx = ctxLocal.Value!;

                    // Build forward neighbour list of v.
                    int fCount = 0;
                    for (int e = outHead[v]; e != -1; e = outNext[e])
                    {
                        int u = outTo[e];
                        if (fCount < ctx.Forward.Length) ctx.Forward[fCount++] = u;
                        else
                        {
                            // Should not happen unless input contains many duplicates.
                            Array.Resize(ref ctx.Forward, Math.Max(ctx.Forward.Length * 2, fCount + 1));
                            ctx.Forward[fCount++] = u;
                        }
                    }

                    // Upper bound for any clique with smallest vertex v is fCount + 1.
                    if (fCount + 1 <= currentBest) return;

                    // Early stop: if even the theoretical maximum meets global upper bound, and we find it, we can stop.
                    // Build local subgraph adjacency among forward neighbors.
                    int localN = fCount;
                    int words = (localN + 63) >> 6;
                    if (words == 0)
                    {
                        // Clique is just {v}.
                        if (currentBest < 1)
                        {
                            lock (bestLock)
                            {
                                if (bestSize < 1)
                                {
                                    bestSize = 1;
                                    bestClique = new[] { v };
                                }
                            }
                        }
                        return;
                    }

                    ctx.EnsureAdjCapacity(localN, words);

                    int stamp = ctx.NextStamp();
                    for (int i2 = 0; i2 < localN; i2++)
                    {
                        int u = ctx.Forward[i2];
                        ctx.Stamp[u] = stamp;
                        ctx.Map[u] = i2;
                    }

                    // Zero adjacency (only used portion).
                    Array.Clear(ctx.AdjFlat, 0, localN * words);

                    for (int i2 = 0; i2 < localN; i2++)
                    {
                        int x = ctx.Forward[i2];
                        int rowBase = i2 * words;
                        for (int e = outHead[x]; e != -1; e = outNext[e])
                        {
                            int y = outTo[e];
                            if (ctx.Stamp[y] == stamp)
                            {
                                int j = ctx.Map[y];
                                ctx.AdjFlat[rowBase + (j >> 6)] |= 1UL << (j & 63);
                                ctx.AdjFlat[j * words + (i2 >> 6)] |= 1UL << (i2 & 63);
                            }
                        }
                    }

                    // Solve maximum clique in this local graph, then add v.
                    ctx.Solver.Reset(localN, words, ctx.AdjFlat);
                    int[] localClique = ctx.Solver.Solve(out int localCliqueSize);
                    int candidateSize = localCliqueSize + 1;

                    if (candidateSize > currentBest)
                    {
                        int[] candidate = new int[candidateSize];
                        candidate[0] = v;
                        for (int i2 = 0; i2 < localCliqueSize; i2++)
                            candidate[i2 + 1] = ctx.Forward[localClique[i2]];

                        lock (bestLock)
                        {
                            if (candidateSize > bestSize)
                            {
                                bestSize = candidateSize;
                                bestClique = candidate;
                            }
                        }
                    }

                    if (Volatile.Read(ref bestSize) >= globalUpperBound)
                        loopState.Stop();
                });

                ctxLocal.Dispose();

                // If we ran out of time, bestClique is still a valid clique (it is built by construction).
            }
            else if (n <= BitsetFallbackNMax && sw.ElapsedMilliseconds < TimeLimitMs)
            {
                // Degeneracy is high; build a bitset adjacency matrix and run BBMC.
                int words = (n + 63) >> 6;
                var adj = new ulong[n * words];

                for (int v = 0; v < n; v++)
                {
                    for (int e = head[v]; e != -1; e = next[e])
                    {
                        int u = to[e];
                        adj[v * words + (u >> 6)] |= 1UL << (u & 63);
                    }
                }

                var solver = new BitsetCliqueSolver(n, words, adj, sw, TimeLimitMs);
                int[] clique = solver.Solve(out int cliqueSize);
                if (cliqueSize > bestSize)
                {
                    bestSize = cliqueSize;
                    bestClique = clique;
                }
            }
            else
            {
                // Fallback: greedy maximal clique (always valid) for very large / hard instances.
                int[] greedy = GreedyClique(n, head, to, next, degree, sw);
                bestClique = greedy.Length == 0 ? new[] { 0 } : greedy;
                bestSize = bestClique.Length;
            }
        }

        if (bestClique == null || bestClique.Length == 0)
        {
            bestSize = 1;
            bestClique = new[] { 0 };
        }

        Console.WriteLine(bestSize);
        var sb = new StringBuilder(bestSize * 2);
        for (int i = 0; i < bestClique.Length; i++)
        {
            if (i != 0) sb.Append(' ');
            sb.Append(bestClique[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static int[] GreedyClique(int n, int[] head, int[] to, int[] next, int[] degree, Stopwatch sw)
    {
        // Simple greedy: start from highest degree, iteratively intersect candidates.
        int start = 0;
        for (int i = 1; i < n; i++)
            if (degree[i] > degree[start]) start = i;

        int[] clique = new int[Math.Min(n, 1024)];
        int cliqueSize = 0;
        clique[cliqueSize++] = start;

        int[] stamp = new int[n];
        int tag = 1;

        // Initial candidates are neighbours of start.
        int[] cand = new int[Math.Min(degree[start], n)];
        int candCount = 0;
        for (int e = head[start]; e != -1; e = next[e])
            cand[candCount++] = to[e];

        while (candCount > 0 && sw.ElapsedMilliseconds < TimeLimitMs)
        {
            // Mark candidates.
            tag++;
            for (int i = 0; i < candCount; i++) stamp[cand[i]] = tag;

            // Pick candidate with highest degree (tie arbitrary).
            int bestIdx = 0;
            int bestV = cand[0];
            int bestDeg = degree[bestV];
            for (int i = 1; i < candCount; i++)
            {
                int v = cand[i];
                int d = degree[v];
                if (d > bestDeg)
                {
                    bestDeg = d;
                    bestV = v;
                    bestIdx = i;
                }
            }

            if (cliqueSize >= clique.Length)
                Array.Resize(ref clique, clique.Length * 2);
            clique[cliqueSize++] = bestV;

            // New candidates = cand ∩ N(bestV)
            int[] newCand = new int[Math.Min(degree[bestV], candCount)];
            int newCount = 0;
            for (int e = head[bestV]; e != -1; e = next[e])
            {
                int u = to[e];
                if (stamp[u] == tag) newCand[newCount++] = u;
            }

            cand = newCand;
            candCount = newCount;
        }

        Array.Resize(ref clique, cliqueSize);
        return clique;
    }

    private static int ComputeDegeneracyOrdering(
        int n,
        int[] head,
        int[] to,
        int[] next,
        int[] degreeInput,
        int[] order,
        int[] rank)
    {
        // Bucket-queue smallest-degree peeling. We return degeneracy and fill:
        // order[k] = k-th removed vertex (removal order), rank[v] = k.
        int[] deg = new int[n];
        Array.Copy(degreeInput, deg, n);

        int maxDeg = 0;
        for (int i = 0; i < n; i++) if (deg[i] > maxDeg) maxDeg = deg[i];

        int[] bucketHead = new int[maxDeg + 1];
        Array.Fill(bucketHead, -1);
        int[] prev = new int[n];
        int[] nxt = new int[n];
        Array.Fill(prev, -1);
        Array.Fill(nxt, -1);

        for (int v = 0; v < n; v++)
        {
            int d = deg[v];
            int h = bucketHead[d];
            bucketHead[d] = v;
            prev[v] = -1;
            nxt[v] = h;
            if (h != -1) prev[h] = v;
        }

        bool[] removed = new bool[n];
        int minDeg = 0;
        int degeneracy = 0;

        for (int k = 0; k < n; k++)
        {
            while (minDeg <= maxDeg && bucketHead[minDeg] == -1) minDeg++;
            int v = bucketHead[minDeg];
            // Remove v from its bucket.
            int vNext = nxt[v];
            bucketHead[minDeg] = vNext;
            if (vNext != -1) prev[vNext] = -1;
            nxt[v] = -1;
            prev[v] = -1;

            removed[v] = true;
            order[k] = v;
            rank[v] = k;
            if (minDeg > degeneracy) degeneracy = minDeg;

            for (int e = head[v]; e != -1; e = next[e])
            {
                int u = to[e];
                if (removed[u]) continue;

                int du = deg[u];
                // Remove u from bucket du
                int pu = prev[u];
                int nu = nxt[u];
                if (pu != -1) nxt[pu] = nu;
                else bucketHead[du] = nu;
                if (nu != -1) prev[nu] = pu;

                // Decrement and insert into bucket du-1
                du--;
                deg[u] = du;
                int head2 = bucketHead[du];
                bucketHead[du] = u;
                prev[u] = -1;
                nxt[u] = head2;
                if (head2 != -1) prev[head2] = u;

                if (du < minDeg) minDeg = du;
            }
        }

        return degeneracy;
    }
}

internal sealed class PerThreadContext
{
    public int[] Map;
    public int[] Stamp;
    public int[] Forward;
    public ulong[] AdjFlat;
    public BitsetCliqueSolver Solver;

    private int _stamp;
    private readonly Stopwatch _sw;
    private readonly long _timeLimitMs;

    public PerThreadContext(int n, int maxForward, Stopwatch sw, long timeLimitMs)
    {
        Map = new int[n];
        Stamp = new int[n];
        Forward = new int[Math.Max(4, maxForward)];
        AdjFlat = Array.Empty<ulong>();
        Solver = new BitsetCliqueSolver(0, 0, Array.Empty<ulong>(), sw, timeLimitMs);
        _stamp = 1;
        _sw = sw;
        _timeLimitMs = timeLimitMs;
    }

    public int NextStamp()
    {
        int s = ++_stamp;
        if (s == int.MaxValue)
        {
            // Reset stamps to avoid overflow wrap; O(n) but extremely unlikely in benchmarks.
            Array.Clear(Stamp, 0, Stamp.Length);
            _stamp = 1;
            s = 1;
        }
        return s;
    }

    public void EnsureAdjCapacity(int localN, int words)
    {
        int need = localN * words;
        if (AdjFlat.Length < need) AdjFlat = new ulong[need];
    }
}

internal sealed class BitsetCliqueSolver
{
    private int _n;
    private int _words;
    private ulong[] _adj;
    private readonly Stopwatch _sw;
    private readonly long _timeLimitMs;

    private int _bestSize;
    private int[] _bestClique;
    private int[] _clique; // current

    // Per-depth work buffers.
    private int[][] _orderByDepth;
    private int[][] _colorByDepth;
    private ulong[][] _pByDepth;
    private ulong[][] _copyPByDepth;
    private ulong[][] _qByDepth;

    public BitsetCliqueSolver(int n, int words, ulong[] adj, Stopwatch sw, long timeLimitMs)
    {
        _sw = sw;
        _timeLimitMs = timeLimitMs;
        _bestClique = Array.Empty<int>();
        _clique = Array.Empty<int>();
        _orderByDepth = Array.Empty<int[]>();
        _colorByDepth = Array.Empty<int[]>();
        _pByDepth = Array.Empty<ulong[]>();
        _copyPByDepth = Array.Empty<ulong[]>();
        _qByDepth = Array.Empty<ulong[]>();
        Reset(n, words, adj);
    }

    public void Reset(int n, int words, ulong[] adj)
    {
        _n = n;
        _words = words;
        _adj = adj;
        _bestSize = 0;
        _bestClique = Array.Empty<int>();
        if (_clique.Length < _n) _clique = new int[_n];

        // Ensure depth-0 buffers exist.
        EnsureDepth(0);
    }

    public int[] Solve(out int bestSize)
    {
        if (_n <= 0)
        {
            bestSize = 0;
            return Array.Empty<int>();
        }

        // Trivial fast-paths.
        if (_n == 1)
        {
            bestSize = 1;
            return new[] { 0 };
        }

        // Initial candidate set: all vertices.
        ulong[] p0 = GetP(0);
        Array.Clear(p0, 0, _words);
        for (int v = 0; v < _n; v++)
            p0[v >> 6] |= 1UL << (v & 63);

        Expand(0, p0);

        bestSize = _bestSize;
        return _bestClique;
    }

    private void Expand(int depth, ulong[] p)
    {
        if (_sw.ElapsedMilliseconds >= _timeLimitMs) return;

        int[] order = GetOrder(depth);
        int[] color = GetColor(depth);
        int m = ColourOrder(depth, p, order, color);

        for (int i = m - 1; i >= 0; i--)
        {
            if (depth + color[i] <= _bestSize) return;

            int v = order[i];
            _clique[depth] = v;

            ulong[] newP = GetP(depth + 1);
            IntersectWithNeighbourhood(p, v, newP);

            if (IsEmpty(newP))
            {
                int candidateSize = depth + 1;
                if (candidateSize > _bestSize)
                {
                    _bestSize = candidateSize;
                    _bestClique = new int[candidateSize];
                    Array.Copy(_clique, 0, _bestClique, 0, candidateSize);
                }
            }
            else
            {
                Expand(depth + 1, newP);
            }

            ClearBit(p, v);
        }
    }

    private int ColourOrder(int depth, ulong[] p, int[] order, int[] color)
    {
        ulong[] copyP = GetCopyP(depth);
        ulong[] q = GetQ(depth);
        Array.Copy(p, copyP, _words);

        int colour = 0;
        int idx = 0;

        while (!IsEmpty(copyP))
        {
            colour++;
            Array.Copy(copyP, q, _words);

            while (!IsEmpty(q))
            {
                int v = FirstBit(q);
                if (v < 0) break;

                ClearBit(q, v);
                ClearBit(copyP, v);

                order[idx] = v;
                color[idx] = colour;
                idx++;

                AndNotWithNeighbourhood(q, v);
            }
        }

        return idx;
    }

    private void IntersectWithNeighbourhood(ulong[] p, int v, ulong[] dest)
    {
        int rowBase = v * _words;
        for (int w = 0; w < _words; w++)
            dest[w] = p[w] & _adj[rowBase + w];
    }

    private void AndNotWithNeighbourhood(ulong[] set, int v)
    {
        int rowBase = v * _words;
        for (int w = 0; w < _words; w++)
            set[w] &= ~_adj[rowBase + w];
    }

    private static void ClearBit(ulong[] bits, int v)
    {
        bits[v >> 6] &= ~(1UL << (v & 63));
    }

    private static int FirstBit(ulong[] bits)
    {
        for (int wi = 0; wi < bits.Length; wi++)
        {
            ulong w = bits[wi];
            if (w != 0)
                return (wi << 6) + BitOperations.TrailingZeroCount(w);
        }
        return -1;
    }

    private static bool IsEmpty(ulong[] bits)
    {
        for (int i = 0; i < bits.Length; i++)
            if (bits[i] != 0) return false;
        return true;
    }

    private void EnsureDepth(int depth)
    {
        int need = depth + 1;
        if (_orderByDepth.Length >= need) return;

        int newLen = Math.Max(need, _orderByDepth.Length == 0 ? 4 : _orderByDepth.Length * 2);
        Array.Resize(ref _orderByDepth, newLen);
        Array.Resize(ref _colorByDepth, newLen);
        Array.Resize(ref _pByDepth, newLen);
        Array.Resize(ref _copyPByDepth, newLen);
        Array.Resize(ref _qByDepth, newLen);
    }

    private int[] GetOrder(int depth)
    {
        EnsureDepth(depth);
        int[] arr = _orderByDepth[depth];
        if (arr == null || arr.Length < _n)
        {
            arr = new int[_n];
            _orderByDepth[depth] = arr;
        }
        return arr;
    }

    private int[] GetColor(int depth)
    {
        EnsureDepth(depth);
        int[] arr = _colorByDepth[depth];
        if (arr == null || arr.Length < _n)
        {
            arr = new int[_n];
            _colorByDepth[depth] = arr;
        }
        return arr;
    }

    private ulong[] GetP(int depth)
    {
        EnsureDepth(depth);
        ulong[] arr = _pByDepth[depth];
        if (arr == null || arr.Length != _words)
        {
            arr = new ulong[_words];
            _pByDepth[depth] = arr;
        }
        return arr;
    }

    private ulong[] GetCopyP(int depth)
    {
        EnsureDepth(depth);
        ulong[] arr = _copyPByDepth[depth];
        if (arr == null || arr.Length != _words)
        {
            arr = new ulong[_words];
            _copyPByDepth[depth] = arr;
        }
        return arr;
    }

    private ulong[] GetQ(int depth)
    {
        EnsureDepth(depth);
        ulong[] arr = _qByDepth[depth];
        if (arr == null || arr.Length != _words)
        {
            arr = new ulong[_words];
            _qByDepth[depth] = arr;
        }
        return arr;
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _len;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 16];
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len <= 0) return -1;
        }
        return _buffer[_ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) throw new EndOfStreamException();
        } while (c <= 32);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > 32 && c != -1)
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return val * sign;
    }
}
