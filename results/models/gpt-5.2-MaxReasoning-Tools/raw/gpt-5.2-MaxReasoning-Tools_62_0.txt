declare const require: any;
declare const process: any;

const fs = require("fs");
let input: string = fs.readFileSync(0, "utf8");
if (input.length && input.charCodeAt(0) === 0xfeff) input = input.slice(1); // strip UTF-8 BOM if present
const data: any = input && input.trim().length ? JSON.parse(input) : {};

const width: number = (data.width ?? 0) | 0;
const height: number = (data.height ?? 0) | 0;
const grid: string[] = Array.isArray(data.grid) ? data.grid : [];
const cutBudgetRaw: number = (data.cutBudget ?? 0) | 0;
let budget: number = cutBudgetRaw > 0 ? cutBudgetRaw : 0;

const N: number = width * height;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}
function idx(x: number, y: number): number {
  return y * width + x;
}

if (width <= 0 || height <= 0 || N <= 0 || budget <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const passable = new Uint8Array(N);
for (let y = 0; y < height; y++) {
  const row = typeof grid[y] === "string" ? grid[y] : "";
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    passable[idx(x, y)] = c === 46 ? 1 : 0; // '.' => 46
  }
}

const isIgnition = new Uint8Array(N);
const ignitionCells: number[] = [];
if (Array.isArray(data.ignitions)) {
  for (const p of data.ignitions) {
    if (!Array.isArray(p) || p.length < 2) continue;
    const x = p[0] | 0;
    const y = p[1] | 0;
    if (!inBounds(x, y)) continue;
    const cell = idx(x, y);
    if (isIgnition[cell]) continue;
    isIgnition[cell] = 1;
    ignitionCells.push(cell);
  }
}

const isTown = new Uint8Array(N);
const townCellsAll: number[] = [];
const townValuesAll: number[] = [];
if (Array.isArray(data.towns)) {
  for (const t of data.towns) {
    if (!t) continue;
    const x = (t.x ?? 0) | 0;
    const y = (t.y ?? 0) | 0;
    if (!inBounds(x, y)) continue;
    const v = Number(t.value ?? 0);
    if (!Number.isFinite(v) || v <= 0) continue;
    const cell = idx(x, y);
    isTown[cell] = 1;
    townCellsAll.push(cell);
    townValuesAll.push(v);
  }
}

// Baseline burn region: multi-source BFS from ignitions through '.'.
const dist = new Int32Array(N);
dist.fill(-1);
const queue = new Int32Array(N);
let qh = 0;
let qt = 0;
for (let i = 0; i < ignitionCells.length; i++) {
  const s = ignitionCells[i];
  if (!passable[s]) continue;
  if (dist[s] !== -1) continue;
  dist[s] = 0;
  queue[qt++] = s;
}

while (qh < qt) {
  const u = queue[qh++];
  const x = u % width;
  // left
  if (x !== 0) {
    const v = u - 1;
    if (passable[v] && dist[v] === -1) {
      dist[v] = dist[u] + 1;
      queue[qt++] = v;
    }
  }
  // right
  if (x !== width - 1) {
    const v = u + 1;
    if (passable[v] && dist[v] === -1) {
      dist[v] = dist[u] + 1;
      queue[qt++] = v;
    }
  }
  // up
  if (u >= width) {
    const v = u - width;
    if (passable[v] && dist[v] === -1) {
      dist[v] = dist[u] + 1;
      queue[qt++] = v;
    }
  }
  // down
  const down = u + width;
  if (down < N) {
    const v = down;
    if (passable[v] && dist[v] === -1) {
      dist[v] = dist[u] + 1;
      queue[qt++] = v;
    }
  }
}

if (qt === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Town values that would otherwise burn.
const townValueAt = new Map<number, number>();
for (let i = 0; i < townCellsAll.length; i++) {
  const cell = townCellsAll[i];
  const v = townValuesAll[i];
  if (dist[cell] === -1) continue;
  const prev = townValueAt.get(cell) ?? 0;
  townValueAt.set(cell, prev + v);
}

if (townValueAt.size === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const burnedIgnitions: number[] = [];
for (let i = 0; i < ignitionCells.length; i++) {
  const c = ignitionCells[i];
  if (dist[c] !== -1) burnedIgnitions.push(c);
}
if (burnedIgnitions.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const cuttable = new Uint8Array(N);
for (let i = 0; i < qt; i++) {
  const cell = queue[i];
  cuttable[cell] = isTown[cell] || isIgnition[cell] ? 0 : 1;
}

// Iterative DFS low-link on the burned subgraph + a super-root connected to all ignitions.
const superRoot = N;
const totalNodes = N + 1;
const disc = new Int32Array(totalNodes);
disc.fill(-1);
const low = new Int32Array(totalNodes);
const dfsParent = new Int32Array(totalNodes);
dfsParent.fill(-1);
const finish = new Int32Array(totalNodes);

const head = new Int32Array(totalNodes);
head.fill(-1);
const maxIntervals = qt; // intervals cannot exceed tree edges in burned component
const intStart = new Int32Array(maxIntervals);
const intEnd = new Int32Array(maxIntervals);
const intNext = new Int32Array(maxIntervals);
let intCount = 0;
const candidates: number[] = [];

const maxStack = qt + 1;
const stackNode = new Int32Array(maxStack);
const stackNext = new Int32Array(maxStack);
let sp = 0;

let timer = 0;
disc[superRoot] = timer;
low[superRoot] = timer;
timer++;
stackNode[sp] = superRoot;
stackNext[sp] = 0;
sp++;

while (sp > 0) {
  const u = stackNode[sp - 1];
  const next = stackNext[sp - 1];

  let deg = 0;
  if (u === superRoot) {
    deg = burnedIgnitions.length;
  } else {
    deg = isIgnition[u] ? 5 : 4;
  }

  if (next < deg) {
    stackNext[sp - 1] = next + 1;

    let v = -1;
    if (u === superRoot) {
      v = burnedIgnitions[next];
    } else {
      let dir = next;
      if (isIgnition[u]) {
        if (next === 0) {
          v = superRoot;
        } else {
          dir = next - 1;
        }
      }
      if (v !== superRoot) {
        const x = u % width;
        if (dir === 0) {
          if (x !== 0) v = u - 1;
        } else if (dir === 1) {
          if (x !== width - 1) v = u + 1;
        } else if (dir === 2) {
          if (u >= width) v = u - width;
        } else {
          const d = u + width;
          if (d < N) v = d;
        }
      }
      if (v !== superRoot && v >= 0 && dist[v] === -1) v = -1;
    }

    if (v < 0) continue;

    if (disc[v] === -1) {
      dfsParent[v] = u;
      disc[v] = timer;
      low[v] = timer;
      timer++;
      stackNode[sp] = v;
      stackNext[sp] = 0;
      sp++;
    } else if (v !== dfsParent[u]) {
      const dv = disc[v];
      if (dv < low[u]) low[u] = dv;
    }
  } else {
    // finish u
    finish[u] = timer;
    sp--;
    const p = dfsParent[u];
    if (p !== -1) {
      if (low[u] < low[p]) low[p] = low[u];
      if (p !== superRoot && cuttable[p] && low[u] >= disc[p]) {
        if (intCount < maxIntervals) {
          const prev = head[p];
          intStart[intCount] = disc[u];
          intEnd[intCount] = finish[u];
          intNext[intCount] = prev;
          head[p] = intCount;
          if (prev === -1) candidates.push(p);
          intCount++;
        }
      }
    }
  }
}

// If no useful articulation candidates, skip straight to per-town shielding.
type TownEntry = { pos: number; value: number; cell: number };
const townEntries: TownEntry[] = [];
for (const [cell, value] of townValueAt) {
  const pos = disc[cell];
  if (pos >= 0) townEntries.push({ pos, value, cell });
}

townEntries.sort((a, b) => a.pos - b.pos);

const M = townEntries.length;
const townPos = new Int32Array(M);
const townVal = new Float64Array(M);
const townCell = new Int32Array(M);
for (let i = 0; i < M; i++) {
  const e = townEntries[i];
  townPos[i] = e.pos | 0;
  townVal[i] = e.value;
  townCell[i] = e.cell | 0;
}

function lowerBound(arr: Int32Array, target: number): number {
  let lo = 0;
  let hi = arr.length;
  while (lo < hi) {
    const mid = (lo + hi) >>> 1;
    if (arr[mid] < target) lo = mid + 1;
    else hi = mid;
  }
  return lo;
}

class Fenwick {
  n: number;
  bit: Float64Array;
  constructor(n: number) {
    this.n = n;
    this.bit = new Float64Array(n + 1);
  }
  buildFrom(values: Float64Array) {
    const n = this.n;
    const bit = this.bit;
    for (let i = 1; i <= n; i++) bit[i] = values[i - 1];
    for (let i = 1; i <= n; i++) {
      const j = i + (i & -i);
      if (j <= n) bit[j] += bit[i];
    }
  }
  add(i1: number, delta: number) {
    const n = this.n;
    const bit = this.bit;
    for (let i = i1; i <= n; i += i & -i) bit[i] += delta;
  }
  sum(i1: number): number {
    let res = 0;
    const bit = this.bit;
    for (let i = i1; i > 0; i -= i & -i) res += bit[i];
    return res;
  }
  rangeSum(l0: number, r0: number): number {
    // indices are 0-based, range [l0, r0)
    if (l0 >= r0) return 0;
    return this.sum(r0) - this.sum(l0);
  }
}

const fenwick = new Fenwick(M);
fenwick.buildFrom(townVal);

const dsuNext = new Int32Array(M + 1);
for (let i = 0; i <= M; i++) dsuNext[i] = i;
function findNext(i: number): number {
  while (dsuNext[i] !== i) {
    dsuNext[i] = dsuNext[dsuNext[i]];
    i = dsuNext[i];
  }
  return i;
}

function eraseIndex(i: number) {
  const v = townVal[i];
  if (v !== 0) {
    fenwick.add(i + 1, -v);
    townVal[i] = 0;
  }
  dsuNext[i] = i + 1;
}

function eraseRange(l: number, r: number) {
  let i = findNext(l);
  while (i < r) {
    eraseIndex(i);
    i = findNext(i);
  }
}

// Convert interval endpoints from DFS-timer space to indices in the sorted townPos array.
for (let e = 0; e < intCount; e++) {
  intStart[e] = lowerBound(townPos, intStart[e]);
  intEnd[e] = lowerBound(townPos, intEnd[e]);
}

class MaxHeap {
  nodes: Int32Array;
  keys: Float64Array;
  size: number;
  constructor(cap: number) {
    this.nodes = new Int32Array(cap > 0 ? cap : 1);
    this.keys = new Float64Array(cap > 0 ? cap : 1);
    this.size = 0;
  }
  push(node: number, key: number) {
    let i = this.size++;
    this.nodes[i] = node;
    this.keys[i] = key;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.keys[p] >= key) break;
      const tn = this.nodes[p];
      const tk = this.keys[p];
      this.nodes[p] = node;
      this.keys[p] = key;
      this.nodes[i] = tn;
      this.keys[i] = tk;
      i = p;
    }
  }
  pop(): { node: number; key: number } | null {
    const n = this.size;
    if (n <= 0) return null;
    const node = this.nodes[0];
    const key = this.keys[0];
    this.size = n - 1;
    if (this.size > 0) {
      const lastNode = this.nodes[this.size];
      const lastKey = this.keys[this.size];
      this.nodes[0] = lastNode;
      this.keys[0] = lastKey;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        if (l >= this.size) break;
        const r = l + 1;
        let best = l;
        if (r < this.size && this.keys[r] > this.keys[l]) best = r;
        if (this.keys[i] >= this.keys[best]) break;
        const tn = this.nodes[best];
        const tk = this.keys[best];
        this.nodes[best] = this.nodes[i];
        this.keys[best] = this.keys[i];
        this.nodes[i] = tn;
        this.keys[i] = tk;
        i = best;
      }
    }
    return { node, key };
  }
}

function gainForNode(u: number): number {
  let sum = 0;
  for (let e = head[u]; e !== -1; e = intNext[e]) {
    const l = intStart[e];
    const r = intEnd[e];
    if (l < r) sum += fenwick.rangeSum(l, r);
  }
  return sum;
}

const cutChosen = new Uint8Array(N);
const firebreaks: number[][] = [];

// Greedy selection of articulation points with dynamic (lazy) gains.
if (budget > 0 && candidates.length > 0 && M > 0) {
  const heap = new MaxHeap(candidates.length);
  for (let i = 0; i < candidates.length; i++) {
    const u = candidates[i];
    if (!cuttable[u]) continue;
    const g = gainForNode(u);
    if (g > 0) heap.push(u, g);
  }

  while (budget > 0) {
    const item = heap.pop();
    if (!item) break;
    const u = item.node;
    if (u < 0 || u >= N) continue;
    if (!cuttable[u]) continue;
    if (cutChosen[u]) continue;
    const actual = gainForNode(u);
    if (actual <= 0) continue;
    if (actual + 1e-9 < item.key) {
      heap.push(u, actual);
      continue;
    }
    cutChosen[u] = 1;
    firebreaks.push([u % width, (u / width) | 0]);
    budget--;
    for (let e = head[u]; e !== -1; e = intNext[e]) {
      const l = intStart[e];
      const r = intEnd[e];
      if (l < r) eraseRange(l, r);
    }
  }
}

function neighborCells(cell: number): number[] | null {
  // Returns cut cells needed to isolate `cell` from the burned region by cutting its burned neighbors.
  // Returns null if isolation is impossible (requires cutting a non-cuttable burned neighbor).
  const req: number[] = [];
  const x = cell % width;

  // left
  if (x !== 0) {
    const n = cell - 1;
    if (dist[n] !== -1 && !cutChosen[n]) {
      if (!passable[n] || isTown[n] || isIgnition[n]) return null;
      req.push(n);
    }
  }
  // right
  if (x !== width - 1) {
    const n = cell + 1;
    if (dist[n] !== -1 && !cutChosen[n]) {
      if (!passable[n] || isTown[n] || isIgnition[n]) return null;
      req.push(n);
    }
  }
  // up
  if (cell >= width) {
    const n = cell - width;
    if (dist[n] !== -1 && !cutChosen[n]) {
      if (!passable[n] || isTown[n] || isIgnition[n]) return null;
      req.push(n);
    }
  }
  // down
  const down = cell + width;
  if (down < N) {
    const n = down;
    if (dist[n] !== -1 && !cutChosen[n]) {
      if (!passable[n] || isTown[n] || isIgnition[n]) return null;
      req.push(n);
    }
  }
  return req;
}

// Fallback: shield high-value towns by cutting their (burned) adjacent cells.
if (budget > 0 && M > 0) {
  const shieldCandidates: { i: number; ratio: number; value: number }[] = [];
  for (let i = 0; i < M; i++) {
    const v = townVal[i];
    if (v <= 0) continue;
    const cell = townCell[i];
    const req = neighborCells(cell);
    if (req === null) continue;
    if (req.length === 0) {
      eraseIndex(i);
      continue;
    }
    const ratio = v / req.length;
    shieldCandidates.push({ i, ratio, value: v });
  }

  shieldCandidates.sort((a, b) => {
    if (b.ratio !== a.ratio) return b.ratio - a.ratio;
    return b.value - a.value;
  });

  for (let k = 0; k < shieldCandidates.length && budget > 0; k++) {
    const i = shieldCandidates[k].i;
    const v = townVal[i];
    if (v <= 0) continue;
    const cell = townCell[i];
    const req = neighborCells(cell);
    if (req === null) continue;
    // Remove already-cut cells from the required set (might have changed since candidate creation).
    let cost = 0;
    for (let j = 0; j < req.length; j++) if (!cutChosen[req[j]]) cost++;
    if (cost === 0) {
      eraseIndex(i);
      continue;
    }
    if (cost > budget) continue;

    for (let j = 0; j < req.length; j++) {
      const c = req[j];
      if (c < 0 || c >= N) continue;
      if (cutChosen[c]) continue;
      if (!cuttable[c]) continue;
      cutChosen[c] = 1;
      firebreaks.push([c % width, (c / width) | 0]);
      budget--;
      if (budget <= 0) break;
    }
    eraseIndex(i);
  }
}

process.stdout.write(JSON.stringify({ firebreaks }));
