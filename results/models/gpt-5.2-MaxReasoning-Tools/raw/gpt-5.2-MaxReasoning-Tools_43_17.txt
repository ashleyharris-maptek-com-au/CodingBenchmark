struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    column_major float4x4 model;
    column_major float4x4 view;
    column_major float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float3 ProceduralBrick(float2 uv)
{
    float2 p = uv * float2(8.0, 16.0);

    int row = (int)p.y;
    if ((row % 2) != 0)
        p.x += 0.5;

    int2 ip = (int2)p;
    float2 cell = p - (float2)ip;

    float mortarX = 0.05;
    float mortarY = 0.1;

    float mortarMask =
        ((cell.x < mortarX) || (cell.x > 1.0 - mortarX) || (cell.y < mortarY) || (cell.y > 1.0 - mortarY))
            ? 1.0
            : 0.0;

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);

    return lerp(brickColor, mortarColor, mortarMask);
}

float4 main(PSInput input) : SV_Target0
{
    float3 albedo = ProceduralBrick(input.uv);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);

    float shade = 0.4 + 0.6 * diffuse;
    return float4(albedo * shade, 1.0);
}
