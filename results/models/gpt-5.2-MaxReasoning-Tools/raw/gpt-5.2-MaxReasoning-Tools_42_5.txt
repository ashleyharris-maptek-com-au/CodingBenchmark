OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in = OpTypeArray %v3float %uint_2

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_0_75 = OpConstant %float 0.75
%float_thickness = OpConstant %float 0.015
%float_arrow = OpConstant %float 0.05

%main = OpFunction %void None %fn_void
%entry = OpLabel

%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%p0Ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0Ptr
%p1Ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1Ptr

%c0Ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0Ptr
%c1Ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1Ptr

%dir4 = OpFSub %v4float %p1 %p0
%dir4Scaled = OpVectorTimesScalar %v4float %dir4 %float_0_75
%mid = OpFAdd %v4float %p0 %dir4Scaled

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir2 = OpFSub %v2float %p1xy %p0xy
%dir2x = OpCompositeExtract %float %dir2 0
%dir2y = OpCompositeExtract %float %dir2 1
%negDir2y = OpFNegate %float %dir2y
%perpUn = OpCompositeConstruct %v2float %negDir2y %dir2x
%perp = OpExtInst %v2float %glsl Normalize %perpUn

%shaftOff2 = OpVectorTimesScalar %v2float %perp %float_thickness
%shaftOffX = OpCompositeExtract %float %shaftOff2 0
%shaftOffY = OpCompositeExtract %float %shaftOff2 1
%shaftOff4 = OpCompositeConstruct %v4float %shaftOffX %shaftOffY %float_0 %float_0

%arrowOff2 = OpVectorTimesScalar %v2float %perp %float_arrow
%arrowOffX = OpCompositeExtract %float %arrowOff2 0
%arrowOffY = OpCompositeExtract %float %arrowOff2 1
%arrowOff4 = OpCompositeConstruct %v4float %arrowOffX %arrowOffY %float_0 %float_0

; Shaft strip: p0±perp*thickness, mid±perp*thickness (color c0)
%p0Plus = OpFAdd %v4float %p0 %shaftOff4
OpStore %outPosPtr %p0Plus
OpStore %outColor %c0
OpEmitVertex

%p0Minus = OpFSub %v4float %p0 %shaftOff4
OpStore %outPosPtr %p0Minus
OpStore %outColor %c0
OpEmitVertex

%midPlus = OpFAdd %v4float %mid %shaftOff4
OpStore %outPosPtr %midPlus
OpStore %outColor %c0
OpEmitVertex

%midMinus = OpFSub %v4float %mid %shaftOff4
OpStore %outPosPtr %midMinus
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; Arrowhead strip: mid±perp*0.05, p1 (color c1)
%arrowPlus = OpFAdd %v4float %mid %arrowOff4
OpStore %outPosPtr %arrowPlus
OpStore %outColor %c1
OpEmitVertex

%arrowMinus = OpFSub %v4float %mid %arrowOff4
OpStore %outPosPtr %arrowMinus
OpStore %outColor %c1
OpEmitVertex

OpStore %outPosPtr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
