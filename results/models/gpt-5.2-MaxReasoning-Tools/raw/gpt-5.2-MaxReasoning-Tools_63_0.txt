declare const require: any;

type RobotIn = { id: number; x: number; y: number };
type JobIn = { id: any; x: number; y: number; priority: number };

type JobEx = { id: any; x: number; y: number; priority: number; morton: number };

function isNumber(x: any): x is number {
  return typeof x === 'number' && Number.isFinite(x);
}

function toFiniteNumber(x: any, fallback: number): number {
  const n = typeof x === 'number' ? x : Number(x);
  return Number.isFinite(n) ? n : fallback;
}

function cmpId(a: any, b: any): number {
  if (typeof a === 'number' && typeof b === 'number') return a - b;
  const sa = String(a);
  const sb = String(b);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  return 0;
}

function manhattan(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);
}

function part1By1(n: number): number {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n >>> 0;
}

function morton2D(x: number, y: number): number {
  return ((part1By1(y) << 1) | part1By1(x)) >>> 0;
}

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const cleanedInput = input.length > 0 && input.charCodeAt(0) === 0xfeff ? input.slice(1) : input;
const data: any = JSON.parse(cleanedInput);

const grid: string[] = Array.isArray(data.grid) ? data.grid : [];
const robotsIn: RobotIn[] = Array.isArray(data.robots) ? data.robots : [];
const jobsIn: JobIn[] = Array.isArray(data.jobs) ? data.jobs : [];

const H = grid.length;
const W = H > 0 && typeof grid[0] === 'string' ? grid[0].length : 0;
const N = W * H;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function idxOf(x: number, y: number): number {
  return y * W + x;
}

if (robotsIn.length === 0 || W === 0 || H === 0) {
  fs.writeFileSync(1, JSON.stringify({ assignments: [] }));
} else {
  // Build open/blocked grid.
  const open = new Uint8Array(N);
  let openCount = 0;
  for (let y = 0; y < H; y++) {
    const row = grid[y];
    for (let x = 0; x < W; x++) {
      const isOpen = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
      const idx = y * W + x;
      open[idx] = isOpen;
      openCount += isOpen;
    }
  }

  // Sort robots by id for deterministic multi-source BFS seeding / tie-breaking.
  const robots = robotsIn
    .filter((r) => isNumber(r.id) && isNumber(r.x) && isNumber(r.y))
    .slice()
    .sort((a, b) => a.id - b.id);

  let maxRobotId = -1;
  for (const r of robots) maxRobotId = r.id > maxRobotId ? r.id : maxRobotId;

  // Prepare output slots indexed by robot id.
  const jobsByRobot: JobEx[][] = Array.from({ length: maxRobotId + 1 }, () => []);

  // Multi-source BFS: owner[cell] = robotId that is closest by grid walking distance.
  const owner = new Int32Array(N);
  owner.fill(-1);
  const queue = new Int32Array(openCount > 0 ? openCount : 1);
  let qh = 0;
  let qt = 0;

  for (const r of robots) {
    const x = r.x | 0;
    const y = r.y | 0;
    if (!inBounds(x, y)) continue;
    const idx = idxOf(x, y);
    if (open[idx] !== 1) continue;
    const existing = owner[idx];
    if (existing === -1) {
      owner[idx] = r.id | 0;
      queue[qt++] = idx;
    } else if ((r.id | 0) < existing) {
      owner[idx] = r.id | 0;
    }
  }

  while (qh < qt) {
    const cur = queue[qh++];
    const curOwner = owner[cur];
    const x = cur % W;

    let n = cur - 1;
    if (x > 0 && open[n] === 1 && owner[n] === -1) {
      owner[n] = curOwner;
      queue[qt++] = n;
    }
    n = cur + 1;
    if (x + 1 < W && open[n] === 1 && owner[n] === -1) {
      owner[n] = curOwner;
      queue[qt++] = n;
    }
    n = cur - W;
    if (cur >= W && open[n] === 1 && owner[n] === -1) {
      owner[n] = curOwner;
      queue[qt++] = n;
    }
    n = cur + W;
    if (cur + W < N && open[n] === 1 && owner[n] === -1) {
      owner[n] = curOwner;
      queue[qt++] = n;
    }
  }

  // Assign each job to its closest robot (by BFS owner); fall back to Manhattan if unreachable.
  const robotsForFallback = robots.length > 0 ? robots : robotsIn;
  for (const j of jobsIn) {
    const x = toFiniteNumber(j.x, 0) | 0;
    const y = toFiniteNumber(j.y, 0) | 0;
    const priority = toFiniteNumber((j as any).priority, 0);
    const job: JobEx = { id: (j as any).id, x, y, priority, morton: morton2D(x, y) };

    let rid = -1;
    if (inBounds(x, y)) {
      const idx = idxOf(x, y);
      rid = owner[idx];
    }
    if (rid < 0 || rid > maxRobotId) {
      // Fallback: nearest robot by Manhattan distance from robot start.
      let bestId = robotsForFallback[0].id | 0;
      let best = Infinity;
      for (const r of robotsForFallback) {
        const d = manhattan(r.x | 0, r.y | 0, x, y);
        if (d < best || (d === best && (r.id | 0) < bestId)) {
          best = d;
          bestId = r.id | 0;
        }
      }
      rid = bestId;
    }
    if (rid < 0) rid = 0;
    if (rid > maxRobotId) continue;
    jobsByRobot[rid].push(job);
  }

  // Build per-robot routes: priority desc, then Morton order within each priority chunk (with direction chosen by current position).
  const assignments: any[][] = Array.from({ length: maxRobotId + 1 }, () => []);

  const robotById = new Map<number, RobotIn>();
  for (const r of robotsIn) if (isNumber(r.id)) robotById.set(r.id | 0, r);

  for (let rid = 0; rid <= maxRobotId; rid++) {
    const jobList = jobsByRobot[rid];
    if (!jobList || jobList.length === 0) continue;

    // Sort once globally; chunk by priority while preserving Morton order inside chunks.
    jobList.sort((a, b) => {
      if (a.priority !== b.priority) return b.priority - a.priority;
      if (a.morton !== b.morton) return a.morton < b.morton ? -1 : 1;
      return cmpId(a.id, b.id);
    });

    const startRobot = robotById.get(rid);
    let cx = startRobot ? (startRobot.x | 0) : 0;
    let cy = startRobot ? (startRobot.y | 0) : 0;

    const out: any[] = [];
    let i = 0;
    while (i < jobList.length) {
      const pr = jobList[i].priority;
      const start = i;
      while (i < jobList.length && jobList[i].priority === pr) i++;
      const end = i - 1;

      const first = jobList[start];
      const last = jobList[end];
      const dFirst = manhattan(cx, cy, first.x, first.y);
      const dLast = manhattan(cx, cy, last.x, last.y);

      if (dLast < dFirst) {
        for (let k = end; k >= start; k--) out.push(jobList[k].id);
        cx = first.x;
        cy = first.y;
      } else {
        for (let k = start; k <= end; k++) out.push(jobList[k].id);
        cx = last.x;
        cy = last.y;
      }
    }

    assignments[rid] = out;
  }

  fs.writeFileSync(1, JSON.stringify({ assignments }));
}
