use std::cmp::Ordering;
use std::io::{self, Read};

const EPS: f64 = 1e-9;
const PI_OVER_2: f64 = std::f64::consts::FRAC_PI_2;

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug, Default)]
struct Rect {
    x: f64,
    y: f64,
    w: f64,
    h: f64,
}

impl Rect {
    fn right(self) -> f64 {
        self.x + self.w
    }
    fn top(self) -> f64 {
        self.y + self.h
    }
}

#[derive(Clone, Debug)]
struct Piece {
    idx: usize,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
    width0: f64,
    height0: f64,
    bbox_area: f64,
    max_side: f64,
    verts: Option<Vec<Point>>, // Only needed if stock isn't a simple axis-aligned rectangle.
}

#[derive(Clone, Copy, Debug)]
struct Candidate {
    angle: f64,
    w: f64,
    h: f64,
    bbox_min_rot: Point, // bbox min AFTER rotation about origin, before translation
}

#[derive(Clone, Copy, Debug, Default)]
struct Placement {
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf: Vec<u8> = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        // Skip common BOMs to make tokenization robust on Windows.
        let mut idx = 0usize;
        if buf.starts_with(&[0xEF, 0xBB, 0xBF]) {
            idx = 3;
        } else if buf.starts_with(&[0xFF, 0xFE]) || buf.starts_with(&[0xFE, 0xFF]) {
            idx = 2;
        } else if buf.starts_with(&[0xFF, 0xFE, 0x00, 0x00]) || buf.starts_with(&[0x00, 0x00, 0xFE, 0xFF]) {
            idx = 4;
        }
        Self { buf, idx }
    }

    fn next<T: std::str::FromStr>(&mut self) -> T {
        // Be tolerant of UTF-16LE streams (PowerShell piping) which include interleaved NUL bytes.
        while self.idx < self.buf.len() && (self.buf[self.idx].is_ascii_whitespace() || self.buf[self.idx] == 0) {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() && self.buf[self.idx] != 0 {
            self.idx += 1;
        }
        // Tokens are numeric, so they should be ASCII/UTF-8 compatible even if the full stream isn't.
        let s = std::str::from_utf8(&self.buf[start..self.idx]).unwrap_or("");
        s.parse::<T>().ok().unwrap()
    }
}

fn feq(a: f64, b: f64) -> bool {
    (a - b).abs() <= EPS
}

fn bbox_min_max(poly: &[Point]) -> (Point, Point) {
    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in poly {
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    (Point { x: min_x, y: min_y }, Point { x: max_x, y: max_y })
}

fn is_axis_aligned_rectangle(poly: &[Point]) -> Option<(Point, Point)> {
    if poly.len() < 4 {
        return None;
    }
    let (mn, mx) = bbox_min_max(poly);
    if mx.x - mn.x <= EPS || mx.y - mn.y <= EPS {
        return None;
    }
    // All edges axis-aligned.
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let dx = (b.x - a.x).abs();
        let dy = (b.y - a.y).abs();
        if dx > EPS && dy > EPS {
            return None;
        }
    }
    // All vertices lie on the boundary of the AABB.
    for p in poly {
        let on_boundary = feq(p.x, mn.x) || feq(p.x, mx.x) || feq(p.y, mn.y) || feq(p.y, mx.y);
        if !on_boundary {
            return None;
        }
    }
    // Ensure all four sides are present.
    let mut has_left = false;
    let mut has_right = false;
    let mut has_bottom = false;
    let mut has_top = false;
    for p in poly {
        has_left |= feq(p.x, mn.x);
        has_right |= feq(p.x, mx.x);
        has_bottom |= feq(p.y, mn.y);
        has_top |= feq(p.y, mx.y);
    }
    if !(has_left && has_right && has_bottom && has_top) {
        return None;
    }
    Some((mn, mx))
}

fn cross(ax: f64, ay: f64, bx: f64, by: f64) -> f64 {
    ax * by - ay * bx
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    cross(b.x - a.x, b.y - a.y, c.x - a.x, c.y - a.y)
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let abx = b.x - a.x;
    let aby = b.y - a.y;
    let apx = p.x - a.x;
    let apy = p.y - a.y;
    let area2 = cross(abx, aby, apx, apy).abs();
    if area2 > EPS {
        return false;
    }
    let dot = (p.x - a.x) * (p.x - b.x) + (p.y - a.y) * (p.y - b.y);
    dot <= EPS
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    let n = poly.len();
    if n < 3 {
        return false;
    }
    // Boundary counts as inside.
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
    }

    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let ay = a.y;
        let by = b.y;
        let ax = a.x;
        let bx = b.x;
        let intersects = ((ay > p.y) != (by > p.y))
            && (p.x < (bx - ax) * (p.y - ay) / (by - ay) + ax);
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn sign_eps(v: f64) -> i32 {
    if v > EPS {
        1
    } else if v < -EPS {
        -1
    } else {
        0
    }
}

fn segments_proper_intersect(a: Point, b: Point, c: Point, d: Point) -> bool {
    let o1 = sign_eps(orient(a, b, c));
    let o2 = sign_eps(orient(a, b, d));
    let o3 = sign_eps(orient(c, d, a));
    let o4 = sign_eps(orient(c, d, b));
    (o1 * o2 < 0) && (o3 * o4 < 0)
}

fn rotate_point_0(p: Point) -> Point {
    p
}

fn rotate_point_90(p: Point) -> Point {
    Point { x: -p.y, y: p.x }
}

fn piece_fits_stock_general(piece: &Piece, cand: Candidate, stock_poly: &[Point], stock_min: Point, node: Rect) -> bool {
    let Some(verts) = piece.verts.as_deref() else {
        return true;
    };

    let t = Point {
        x: stock_min.x + node.x - cand.bbox_min_rot.x,
        y: stock_min.y + node.y - cand.bbox_min_rot.y,
    };

    // Vertex containment.
    for &v in verts {
        let vr = if cand.angle == 0.0 { rotate_point_0(v) } else { rotate_point_90(v) };
        let vt = Point { x: vr.x + t.x, y: vr.y + t.y };
        if !point_in_polygon(vt, stock_poly) {
            return false;
        }
    }

    // Edge-boundary intersection guard (handles concave stock polygons).
    let n_piece = verts.len();
    let n_stock = stock_poly.len();
    for i in 0..n_piece {
        let a0 = verts[i];
        let b0 = verts[(i + 1) % n_piece];
        let a = if cand.angle == 0.0 { rotate_point_0(a0) } else { rotate_point_90(a0) };
        let b = if cand.angle == 0.0 { rotate_point_0(b0) } else { rotate_point_90(b0) };
        let a = Point { x: a.x + t.x, y: a.y + t.y };
        let b = Point { x: b.x + t.x, y: b.y + t.y };
        for j in 0..n_stock {
            let c = stock_poly[j];
            let d = stock_poly[(j + 1) % n_stock];
            if segments_proper_intersect(a, b, c, d) {
                return false;
            }
        }
    }

    true
}

fn rect_contains(a: Rect, b: Rect) -> bool {
    b.x + EPS >= a.x
        && b.y + EPS >= a.y
        && b.right() <= a.right() + EPS
        && b.top() <= a.top() + EPS
}

fn rects_intersect(a: Rect, b: Rect) -> bool {
    !(a.x >= b.x + b.w - EPS
        || a.x + a.w <= b.x + EPS
        || a.y >= b.y + b.h - EPS
        || a.y + a.h <= b.y + EPS)
}

#[derive(Clone, Debug)]
struct MaxRectsBin {
    w: f64,
    h: f64,
    free: Vec<Rect>,
}

impl MaxRectsBin {
    fn new(w: f64, h: f64) -> Self {
        Self {
            w,
            h,
            free: vec![Rect { x: 0.0, y: 0.0, w, h }],
        }
    }

    fn find_best_short_side_fit<F>(&self, w: f64, h: f64, mut check: F) -> Option<(Rect, f64, f64)>
    where
        F: FnMut(Rect) -> bool,
    {
        let mut best_node: Option<Rect> = None;
        let mut best_short = f64::INFINITY;
        let mut best_long = f64::INFINITY;
        let mut best_y = f64::INFINITY;
        let mut best_x = f64::INFINITY;

        for &f in &self.free {
            if f.w + EPS < w || f.h + EPS < h {
                continue;
            }
            let node = Rect { x: f.x, y: f.y, w, h };
            if !check(node) {
                continue;
            }
            let leftover_w = (f.w - w).max(0.0);
            let leftover_h = (f.h - h).max(0.0);
            let short = leftover_w.min(leftover_h);
            let long = leftover_w.max(leftover_h);

            let better = (short + EPS < best_short)
                || (feq(short, best_short) && (long + EPS < best_long))
                || (feq(short, best_short) && feq(long, best_long) && (node.y + EPS < best_y))
                || (feq(short, best_short) && feq(long, best_long) && feq(node.y, best_y) && (node.x + EPS < best_x));

            if better {
                best_node = Some(node);
                best_short = short;
                best_long = long;
                best_y = node.y;
                best_x = node.x;
            }
        }

        best_node.map(|n| (n, best_short, best_long))
    }

    fn insert_best_short_side_fit<F>(&mut self, w: f64, h: f64, check: F) -> Option<(Rect, f64, f64)>
    where
        F: FnMut(Rect) -> bool,
    {
        let (node, best_short, best_long) = self.find_best_short_side_fit(w, h, check)?;
        self.place_rect(node);
        Some((node, best_short, best_long))
    }

    fn place_rect(&mut self, node: Rect) {
        let mut i = 0usize;
        while i < self.free.len() {
            let free_rect = self.free[i];
            if !rects_intersect(free_rect, node) {
                i += 1;
                continue;
            }

            // Remove current free rect and split into at most 4.
            self.free.swap_remove(i);

            // Left
            if node.x > free_rect.x + EPS {
                let r = Rect {
                    x: free_rect.x,
                    y: free_rect.y,
                    w: (node.x - free_rect.x).max(0.0),
                    h: free_rect.h,
                };
                if r.w > EPS && r.h > EPS {
                    self.free.push(r);
                }
            }
            // Right
            if node.right() < free_rect.right() - EPS {
                let r = Rect {
                    x: node.right(),
                    y: free_rect.y,
                    w: (free_rect.right() - node.right()).max(0.0),
                    h: free_rect.h,
                };
                if r.w > EPS && r.h > EPS {
                    self.free.push(r);
                }
            }
            // Bottom
            if node.y > free_rect.y + EPS {
                let r = Rect {
                    x: free_rect.x,
                    y: free_rect.y,
                    w: free_rect.w,
                    h: (node.y - free_rect.y).max(0.0),
                };
                if r.w > EPS && r.h > EPS {
                    self.free.push(r);
                }
            }
            // Top
            if node.top() < free_rect.top() - EPS {
                let r = Rect {
                    x: free_rect.x,
                    y: node.top(),
                    w: free_rect.w,
                    h: (free_rect.top() - node.top()).max(0.0),
                };
                if r.w > EPS && r.h > EPS {
                    self.free.push(r);
                }
            }
        }

        self.prune_free();
    }

    fn prune_free(&mut self) {
        let mut i = 0usize;
        while i < self.free.len() {
            let mut removed_i = false;
            let mut j = i + 1;
            while j < self.free.len() {
                let a = self.free[i];
                let b = self.free[j];
                if rect_contains(a, b) {
                    self.free.swap_remove(j);
                    continue;
                }
                if rect_contains(b, a) {
                    self.free.swap_remove(i);
                    removed_i = true;
                    break;
                }
                j += 1;
            }
            if !removed_i {
                i += 1;
            }
        }
    }
}

#[derive(Clone, Debug)]
struct ShelfBin {
    w: f64,
    h: f64,
    x: f64,
    y: f64,
    shelf_h: f64,
}

impl ShelfBin {
    fn new(w: f64, h: f64) -> Self {
        Self { w, h, x: 0.0, y: 0.0, shelf_h: 0.0 }
    }

    fn place_piece(&mut self, candidates: &[Candidate; 2]) -> Option<(f64, f64, Candidate)> {
        // Try in current shelf.
        if let Some((px, py, cand)) = self.try_place_in_current_shelf(candidates) {
            return Some((px, py, cand));
        }
        // Start new shelf and try again.
        self.x = 0.0;
        self.y += self.shelf_h;
        self.shelf_h = 0.0;
        self.try_place_in_current_shelf(candidates)
    }

    fn try_place_in_current_shelf(&mut self, candidates: &[Candidate; 2]) -> Option<(f64, f64, Candidate)> {
        let mut best: Option<(Candidate, f64, f64)> = None; // (cand, new_shelf_h, waste_w)
        for &cand in candidates {
            if cand.w > self.w + EPS || cand.h > self.h + EPS {
                continue;
            }
            if self.x + cand.w <= self.w + EPS && self.y + cand.h <= self.h + EPS {
                let new_shelf_h = self.shelf_h.max(cand.h);
                let waste_w = (self.w - (self.x + cand.w)).max(0.0);
                let better = match best {
                    None => true,
                    Some((_, best_h, best_waste)) => {
                        (new_shelf_h + EPS < best_h) || (feq(new_shelf_h, best_h) && (waste_w + EPS < best_waste))
                    }
                };
                if better {
                    best = Some((cand, new_shelf_h, waste_w));
                }
            }
        }
        let (cand, new_shelf_h, _) = best?;
        let px = self.x;
        let py = self.y;
        self.x += cand.w;
        self.shelf_h = new_shelf_h;
        Some((px, py, cand))
    }
}

fn candidates_for_piece(piece: &Piece) -> [Candidate; 2] {
    // Rotation 0: bbox min is (min_x, min_y).
    let c0 = Candidate {
        angle: 0.0,
        w: piece.width0,
        h: piece.height0,
        bbox_min_rot: Point { x: piece.min_x, y: piece.min_y },
    };
    // Rotation 90deg: x' = -y, y' = x. Min x' is -max_y; min y' is min_x.
    let c90 = Candidate {
        angle: PI_OVER_2,
        w: piece.height0,
        h: piece.width0,
        bbox_min_rot: Point { x: -piece.max_y, y: piece.min_x },
    };
    [c0, c90]
}

fn main() {
    let mut sc = Scanner::new();

    let sv: usize = sc.next();
    let np: usize = sc.next();

    let mut stock_poly: Vec<Point> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = sc.next();
        let y: f64 = sc.next();
        stock_poly.push(Point { x, y });
    }

    let (stock_bbox_min, stock_bbox_max) = bbox_min_max(&stock_poly);
    let stock_w = (stock_bbox_max.x - stock_bbox_min.x).max(0.0);
    let stock_h = (stock_bbox_max.y - stock_bbox_min.y).max(0.0);

    let stock_is_rect = is_axis_aligned_rectangle(&stock_poly).is_some();

    // Read pieces, keeping only what we need.
    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    let mut total_piece_vertices: usize = 0;
    for i in 0..np {
        let pv: usize = sc.next();
        total_piece_vertices += pv;
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        let mut verts: Option<Vec<Point>> = if stock_is_rect { None } else { Some(Vec::with_capacity(pv)) };
        for _ in 0..pv {
            let x: f64 = sc.next();
            let y: f64 = sc.next();
            min_x = min_x.min(x);
            min_y = min_y.min(y);
            max_x = max_x.max(x);
            max_y = max_y.max(y);
            if let Some(v) = verts.as_mut() {
                v.push(Point { x, y });
            }
        }
        let width0 = (max_x - min_x).max(0.0);
        let height0 = (max_y - min_y).max(0.0);
        let bbox_area = width0 * height0;
        let max_side = width0.max(height0);
        pieces.push(Piece {
            idx: i,
            min_x,
            max_x,
            min_y,
            max_y,
            width0,
            height0,
            bbox_area,
            max_side,
            verts,
        });
    }

    // Choose algorithm based on scale.
    // - MaxRects gives better packing on small/medium inputs.
    // - Shelf is faster and more predictable on huge inputs.
    let use_shelf = stock_is_rect && (np > 2500 || total_piece_vertices > 250_000);

    // Sort by decreasing bbox area (FFD-style).
    pieces.sort_by(|a, b| {
        b.bbox_area
            .partial_cmp(&a.bbox_area)
            .unwrap_or(Ordering::Equal)
            .then_with(|| b.max_side.partial_cmp(&a.max_side).unwrap_or(Ordering::Equal))
            .then_with(|| a.idx.cmp(&b.idx))
    });

    let mut out_place: Vec<Placement> = vec![Placement::default(); np];

    if use_shelf {
        let mut bins: Vec<ShelfBin> = Vec::new();
        for piece in &pieces {
            let cands = candidates_for_piece(piece);
            let mut placed: Option<(usize, f64, f64, Candidate)> = None;

            for (bin_idx, bin) in bins.iter_mut().enumerate() {
                if let Some((px, py, cand)) = bin.place_piece(&cands) {
                    placed = Some((bin_idx, px, py, cand));
                    break;
                }
            }

            if placed.is_none() {
                let mut new_bin = ShelfBin::new(stock_w, stock_h);
                let (px, py, cand) = new_bin
                    .place_piece(&cands)
                    .unwrap_or((0.0, 0.0, cands[0]));
                bins.push(new_bin);
                placed = Some((bins.len() - 1, px, py, cand));
            }

            let (stock_idx, px, py, cand) = placed.unwrap();
            let translation = Point {
                x: stock_bbox_min.x + px - cand.bbox_min_rot.x,
                y: stock_bbox_min.y + py - cand.bbox_min_rot.y,
            };

            out_place[piece.idx] = Placement {
                stock: stock_idx,
                x: translation.x,
                y: translation.y,
                rot: cand.angle,
            };
        }

        let mut output = String::new();
        output.push_str(&format!("{}\n", bins.len()));
        for p in out_place {
            output.push_str(&format!("{} {:.10} {:.10} {:.16}\n", p.stock, p.x, p.y, p.rot));
        }
        print!("{output}");
        return;
    }

    // MaxRects path.
    let mut bins: Vec<MaxRectsBin> = Vec::new();
    for piece in &pieces {
        let cands = candidates_for_piece(piece);
        let mut placed: Option<(usize, Rect, Candidate)> = None;

        for (bin_idx, bin) in bins.iter_mut().enumerate() {
            let mut best: Option<(Rect, Candidate, f64, f64)> = None; // node, cand, short, long
            for &cand in &cands {
                if cand.w > stock_w + EPS || cand.h > stock_h + EPS {
                    continue;
                }
                let res = bin.find_best_short_side_fit(cand.w, cand.h, |node| {
                    if stock_is_rect {
                        true
                    } else {
                        piece_fits_stock_general(piece, cand, &stock_poly, stock_bbox_min, node)
                    }
                });
                let Some((node, short, long)) = res else { continue };
                let better = match best {
                    None => true,
                    Some((best_node, _, best_short, best_long)) => {
                        (short + EPS < best_short)
                            || (feq(short, best_short) && (long + EPS < best_long))
                            || (feq(short, best_short) && feq(long, best_long) && (node.y + EPS < best_node.y))
                            || (feq(short, best_short) && feq(long, best_long) && feq(node.y, best_node.y) && (node.x + EPS < best_node.x))
                    }
                };
                if better {
                    best = Some((node, cand, short, long));
                }
            }

            if let Some((node, cand, _, _)) = best {
                bin.place_rect(node);
                placed = Some((bin_idx, node, cand));
                break;
            }
        }

        if placed.is_none() {
            let mut new_bin = MaxRectsBin::new(stock_w, stock_h);
            let mut best: Option<(Rect, Candidate, f64, f64)> = None;
            for &cand in &cands {
                if cand.w > stock_w + EPS || cand.h > stock_h + EPS {
                    continue;
                }
                let res = new_bin.find_best_short_side_fit(cand.w, cand.h, |node| {
                    if stock_is_rect {
                        true
                    } else {
                        piece_fits_stock_general(piece, cand, &stock_poly, stock_bbox_min, node)
                    }
                });
                let Some((node, short, long)) = res else { continue };
                let better = match best {
                    None => true,
                    Some((best_node, _, best_short, best_long)) => {
                        (short + EPS < best_short)
                            || (feq(short, best_short) && (long + EPS < best_long))
                            || (feq(short, best_short) && feq(long, best_long) && (node.y + EPS < best_node.y))
                            || (feq(short, best_short) && feq(long, best_long) && feq(node.y, best_node.y) && (node.x + EPS < best_node.x))
                    }
                };
                if better {
                    best = Some((node, cand, short, long));
                }
            }
            if let Some((node, cand, _, _)) = best {
                new_bin.place_rect(node);
                bins.push(new_bin);
                placed = Some((bins.len() - 1, node, cand));
            } else {
                // Oversized / no-feasible-placement fallback: reserve a stock index but don't
                // let anything else be placed in it (free list empty).
                bins.push(MaxRectsBin { w: stock_w, h: stock_h, free: Vec::new() });
                placed = Some((
                    bins.len() - 1,
                    Rect { x: 0.0, y: 0.0, w: cands[0].w, h: cands[0].h },
                    cands[0],
                ));
            }
        }

        let (stock_idx, node, cand) = placed.unwrap();
        let translation = Point {
            x: stock_bbox_min.x + node.x - cand.bbox_min_rot.x,
            y: stock_bbox_min.y + node.y - cand.bbox_min_rot.y,
        };

        out_place[piece.idx] = Placement {
            stock: stock_idx,
            x: translation.x,
            y: translation.y,
            rot: cand.angle,
        };
    }

    let mut output = String::new();
    output.push_str(&format!("{}\n", bins.len()));
    for p in out_place {
        output.push_str(&format!("{} {:.10} {:.10} {:.16}\n", p.stock, p.x, p.y, p.rot));
    }
    print!("{output}");
}
