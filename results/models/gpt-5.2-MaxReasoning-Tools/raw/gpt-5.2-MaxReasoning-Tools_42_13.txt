; SPIR-V Geometry Shader (wireframe edges as thin quads)
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_v3_in ArrayStride 16

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_half = OpConstant %float 0.5
%float_half_thickness = OpConstant %float 0.004
%float_eps = OpConstant %float 0.0000000001

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; ---------------- Edge 0 -> 2 ----------------
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr

%col0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col02_add = OpFAdd %v3float %col0 %col2
%col02_avg = OpVectorTimesScalar %v3float %col02_add %float_half

%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos0_w = OpCompositeExtract %float %pos0 3
%pos2_w = OpCompositeExtract %float %pos2 3
%invw0 = OpFDiv %float %float_1 %pos0_w
%invw2 = OpFDiv %float %float_1 %pos2_w
%ndc0 = OpVectorTimesScalar %v2float %pos0_xy %invw0
%ndc2 = OpVectorTimesScalar %v2float %pos2_xy %invw2
%delta02 = OpFSub %v2float %ndc2 %ndc0
%len2_02 = OpDot %float %delta02 %delta02
%len2_02_eps = OpFAdd %float %len2_02 %float_eps
%invlen_02 = OpExtInst %float %glsl InverseSqrt %len2_02_eps
%dir02 = OpVectorTimesScalar %v2float %delta02 %invlen_02
%dir02_x = OpCompositeExtract %float %dir02 0
%dir02_y = OpCompositeExtract %float %dir02 1
%neg_dir02_y = OpFNegate %float %dir02_y
%perp02 = OpCompositeConstruct %v2float %neg_dir02_y %dir02_x
%perp02_scaled = OpVectorTimesScalar %v2float %perp02 %float_half_thickness
%off0_02 = OpVectorTimesScalar %v2float %perp02_scaled %pos0_w
%off2_02 = OpVectorTimesScalar %v2float %perp02_scaled %pos2_w

%pos0_x = OpCompositeExtract %float %pos0 0
%pos0_y = OpCompositeExtract %float %pos0 1
%pos0_z = OpCompositeExtract %float %pos0 2
%off0_x = OpCompositeExtract %float %off0_02 0
%off0_y = OpCompositeExtract %float %off0_02 1
%pos0_px = OpFAdd %float %pos0_x %off0_x
%pos0_py = OpFAdd %float %pos0_y %off0_y
%pos0_mx = OpFSub %float %pos0_x %off0_x
%pos0_my = OpFSub %float %pos0_y %off0_y
%pos0_p = OpCompositeConstruct %v4float %pos0_px %pos0_py %pos0_z %pos0_w
%pos0_m = OpCompositeConstruct %v4float %pos0_mx %pos0_my %pos0_z %pos0_w

%pos2_x = OpCompositeExtract %float %pos2 0
%pos2_y = OpCompositeExtract %float %pos2 1
%pos2_z = OpCompositeExtract %float %pos2 2
%off2_x = OpCompositeExtract %float %off2_02 0
%off2_y = OpCompositeExtract %float %off2_02 1
%pos2_px = OpFAdd %float %pos2_x %off2_x
%pos2_py = OpFAdd %float %pos2_y %off2_y
%pos2_mx = OpFSub %float %pos2_x %off2_x
%pos2_my = OpFSub %float %pos2_y %off2_y
%pos2_p = OpCompositeConstruct %v4float %pos2_px %pos2_py %pos2_z %pos2_w
%pos2_m = OpCompositeConstruct %v4float %pos2_mx %pos2_my %pos2_z %pos2_w

%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outColor %col02_avg
OpStore %gl_out_pos_ptr %pos0_p
OpEmitVertex

OpStore %outColor %col02_avg
OpStore %gl_out_pos_ptr %pos0_m
OpEmitVertex

OpStore %outColor %col02_avg
OpStore %gl_out_pos_ptr %pos2_p
OpEmitVertex

OpStore %outColor %col02_avg
OpStore %gl_out_pos_ptr %pos2_m
OpEmitVertex
OpEndPrimitive

; ---------------- Edge 2 -> 4 ----------------
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

%col4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%col4 = OpLoad %v3float %col4_ptr
%col24_add = OpFAdd %v3float %col2 %col4
%col24_avg = OpVectorTimesScalar %v3float %col24_add %float_half

%pos2_xy_b = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos4_xy = OpVectorShuffle %v2float %pos4 %pos4 0 1
%pos4_w = OpCompositeExtract %float %pos4 3
%invw2_b = OpFDiv %float %float_1 %pos2_w
%invw4 = OpFDiv %float %float_1 %pos4_w
%ndc2_b = OpVectorTimesScalar %v2float %pos2_xy_b %invw2_b
%ndc4 = OpVectorTimesScalar %v2float %pos4_xy %invw4
%delta24 = OpFSub %v2float %ndc4 %ndc2_b
%len2_24 = OpDot %float %delta24 %delta24
%len2_24_eps = OpFAdd %float %len2_24 %float_eps
%invlen_24 = OpExtInst %float %glsl InverseSqrt %len2_24_eps
%dir24 = OpVectorTimesScalar %v2float %delta24 %invlen_24
%dir24_x = OpCompositeExtract %float %dir24 0
%dir24_y = OpCompositeExtract %float %dir24 1
%neg_dir24_y = OpFNegate %float %dir24_y
%perp24 = OpCompositeConstruct %v2float %neg_dir24_y %dir24_x
%perp24_scaled = OpVectorTimesScalar %v2float %perp24 %float_half_thickness
%off2_24 = OpVectorTimesScalar %v2float %perp24_scaled %pos2_w
%off4_24 = OpVectorTimesScalar %v2float %perp24_scaled %pos4_w

%off2_24_x = OpCompositeExtract %float %off2_24 0
%off2_24_y = OpCompositeExtract %float %off2_24 1
%pos2_px2 = OpFAdd %float %pos2_x %off2_24_x
%pos2_py2 = OpFAdd %float %pos2_y %off2_24_y
%pos2_mx2 = OpFSub %float %pos2_x %off2_24_x
%pos2_my2 = OpFSub %float %pos2_y %off2_24_y
%pos2_p2 = OpCompositeConstruct %v4float %pos2_px2 %pos2_py2 %pos2_z %pos2_w
%pos2_m2 = OpCompositeConstruct %v4float %pos2_mx2 %pos2_my2 %pos2_z %pos2_w

%pos4_x = OpCompositeExtract %float %pos4 0
%pos4_y = OpCompositeExtract %float %pos4 1
%pos4_z = OpCompositeExtract %float %pos4 2
%off4_x = OpCompositeExtract %float %off4_24 0
%off4_y = OpCompositeExtract %float %off4_24 1
%pos4_px = OpFAdd %float %pos4_x %off4_x
%pos4_py = OpFAdd %float %pos4_y %off4_y
%pos4_mx = OpFSub %float %pos4_x %off4_x
%pos4_my = OpFSub %float %pos4_y %off4_y
%pos4_p = OpCompositeConstruct %v4float %pos4_px %pos4_py %pos4_z %pos4_w
%pos4_m = OpCompositeConstruct %v4float %pos4_mx %pos4_my %pos4_z %pos4_w

OpStore %outColor %col24_avg
OpStore %gl_out_pos_ptr %pos2_p2
OpEmitVertex

OpStore %outColor %col24_avg
OpStore %gl_out_pos_ptr %pos2_m2
OpEmitVertex

OpStore %outColor %col24_avg
OpStore %gl_out_pos_ptr %pos4_p
OpEmitVertex

OpStore %outColor %col24_avg
OpStore %gl_out_pos_ptr %pos4_m
OpEmitVertex
OpEndPrimitive

; ---------------- Edge 4 -> 0 ----------------
%col40_add = OpFAdd %v3float %col4 %col0
%col40_avg = OpVectorTimesScalar %v3float %col40_add %float_half

%pos4_xy_b = OpVectorShuffle %v2float %pos4 %pos4 0 1
%pos0_xy_b = OpVectorShuffle %v2float %pos0 %pos0 0 1
%invw4_b = OpFDiv %float %float_1 %pos4_w
%invw0_b = OpFDiv %float %float_1 %pos0_w
%ndc4_b = OpVectorTimesScalar %v2float %pos4_xy_b %invw4_b
%ndc0_b = OpVectorTimesScalar %v2float %pos0_xy_b %invw0_b
%delta40 = OpFSub %v2float %ndc0_b %ndc4_b
%len2_40 = OpDot %float %delta40 %delta40
%len2_40_eps = OpFAdd %float %len2_40 %float_eps
%invlen_40 = OpExtInst %float %glsl InverseSqrt %len2_40_eps
%dir40 = OpVectorTimesScalar %v2float %delta40 %invlen_40
%dir40_x = OpCompositeExtract %float %dir40 0
%dir40_y = OpCompositeExtract %float %dir40 1
%neg_dir40_y = OpFNegate %float %dir40_y
%perp40 = OpCompositeConstruct %v2float %neg_dir40_y %dir40_x
%perp40_scaled = OpVectorTimesScalar %v2float %perp40 %float_half_thickness
%off4_40 = OpVectorTimesScalar %v2float %perp40_scaled %pos4_w
%off0_40 = OpVectorTimesScalar %v2float %perp40_scaled %pos0_w

%off4_40_x = OpCompositeExtract %float %off4_40 0
%off4_40_y = OpCompositeExtract %float %off4_40 1
%pos4_px2 = OpFAdd %float %pos4_x %off4_40_x
%pos4_py2 = OpFAdd %float %pos4_y %off4_40_y
%pos4_mx2 = OpFSub %float %pos4_x %off4_40_x
%pos4_my2 = OpFSub %float %pos4_y %off4_40_y
%pos4_p2 = OpCompositeConstruct %v4float %pos4_px2 %pos4_py2 %pos4_z %pos4_w
%pos4_m2 = OpCompositeConstruct %v4float %pos4_mx2 %pos4_my2 %pos4_z %pos4_w

%off0_40_x = OpCompositeExtract %float %off0_40 0
%off0_40_y = OpCompositeExtract %float %off0_40 1
%pos0_px2 = OpFAdd %float %pos0_x %off0_40_x
%pos0_py2 = OpFAdd %float %pos0_y %off0_40_y
%pos0_mx2 = OpFSub %float %pos0_x %off0_40_x
%pos0_my2 = OpFSub %float %pos0_y %off0_40_y
%pos0_p2 = OpCompositeConstruct %v4float %pos0_px2 %pos0_py2 %pos0_z %pos0_w
%pos0_m2 = OpCompositeConstruct %v4float %pos0_mx2 %pos0_my2 %pos0_z %pos0_w

OpStore %outColor %col40_avg
OpStore %gl_out_pos_ptr %pos4_p2
OpEmitVertex

OpStore %outColor %col40_avg
OpStore %gl_out_pos_ptr %pos4_m2
OpEmitVertex

OpStore %outColor %col40_avg
OpStore %gl_out_pos_ptr %pos0_p2
OpEmitVertex

OpStore %outColor %col40_avg
OpStore %gl_out_pos_ptr %pos0_m2
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
