OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3float_6 = OpTypeArray %v3float %uint_6

%_ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%_ptr_Input_arr_v3float_6 = OpTypePointer Input %arr_v3float_6
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float

%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %_ptr_Input_arr_pv_in Input
%worldPos = OpVariable %_ptr_Input_arr_v3float_6 Input
%normal = OpVariable %_ptr_Input_arr_v3float_6 Input
%color = OpVariable %_ptr_Input_arr_v3float_6 Input

%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_1 = OpConstant %float 1.0

%v3float_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %fn_void
%entry = OpLabel

%gl_out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0

; main triangle vertices: 0, 2, 4
%pos0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos4_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

%col0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %_ptr_Input_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %_ptr_Input_v3float %color %int_4
%col4 = OpLoad %v3float %col4_ptr

%col0_dim = OpVectorTimesScalar %v3float %col0 %float_0_3
%col2_dim = OpVectorTimesScalar %v3float %col2 %float_0_3
%col4_dim = OpVectorTimesScalar %v3float %col4 %float_0_3

; emit filled triangle (3 verts)
OpStore %gl_out_pos_ptr %pos0
OpStore %outColor %col0_dim
OpEmitVertex

OpStore %gl_out_pos_ptr %pos2
OpStore %outColor %col2_dim
OpEmitVertex

OpStore %gl_out_pos_ptr %pos4
OpStore %outColor %col4_dim
OpEmitVertex

OpEndPrimitive

; precompute xy, z, w and NDC xy for the 3 triangle verts
%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos4_xy = OpVectorShuffle %v2float %pos4 %pos4 0 1

%pos0_z = OpCompositeExtract %float %pos0 2
%pos2_z = OpCompositeExtract %float %pos2 2
%pos4_z = OpCompositeExtract %float %pos4 2

%pos0_w = OpCompositeExtract %float %pos0 3
%pos2_w = OpCompositeExtract %float %pos2 3
%pos4_w = OpCompositeExtract %float %pos4 3

%inv_pos0_w = OpFDiv %float %float_1 %pos0_w
%inv_pos2_w = OpFDiv %float %float_1 %pos2_w
%inv_pos4_w = OpFDiv %float %float_1 %pos4_w

%pos0_ndc = OpVectorTimesScalar %v2float %pos0_xy %inv_pos0_w
%pos2_ndc = OpVectorTimesScalar %v2float %pos2_xy %inv_pos2_w
%pos4_ndc = OpVectorTimesScalar %v2float %pos4_xy %inv_pos4_w

; ---------- edge 0->2 thick quad ----------
%diff_02 = OpFSub %v2float %pos2_ndc %pos0_ndc
%dir_02 = OpExtInst %v2float %glsl Normalize %diff_02
%dir_02_x = OpCompositeExtract %float %dir_02 0
%dir_02_y = OpCompositeExtract %float %dir_02 1
%neg_dir_02_y = OpFNegate %float %dir_02_y
%perp_02 = OpCompositeConstruct %v2float %neg_dir_02_y %dir_02_x
%offset_ndc_02 = OpVectorTimesScalar %v2float %perp_02 %float_0_015

%offset_02_0 = OpVectorTimesScalar %v2float %offset_ndc_02 %pos0_w
%offset_02_2 = OpVectorTimesScalar %v2float %offset_ndc_02 %pos2_w

%pos0_plus_xy_02 = OpFAdd %v2float %pos0_xy %offset_02_0
%pos0_minus_xy_02 = OpFSub %v2float %pos0_xy %offset_02_0
%pos2_plus_xy_02 = OpFAdd %v2float %pos2_xy %offset_02_2
%pos2_minus_xy_02 = OpFSub %v2float %pos2_xy %offset_02_2

%pos0_plus_x_02 = OpCompositeExtract %float %pos0_plus_xy_02 0
%pos0_plus_y_02 = OpCompositeExtract %float %pos0_plus_xy_02 1
%pos0_minus_x_02 = OpCompositeExtract %float %pos0_minus_xy_02 0
%pos0_minus_y_02 = OpCompositeExtract %float %pos0_minus_xy_02 1
%pos2_plus_x_02 = OpCompositeExtract %float %pos2_plus_xy_02 0
%pos2_plus_y_02 = OpCompositeExtract %float %pos2_plus_xy_02 1
%pos2_minus_x_02 = OpCompositeExtract %float %pos2_minus_xy_02 0
%pos2_minus_y_02 = OpCompositeExtract %float %pos2_minus_xy_02 1

%pos0_plus_02 = OpCompositeConstruct %v4float %pos0_plus_x_02 %pos0_plus_y_02 %pos0_z %pos0_w
%pos0_minus_02 = OpCompositeConstruct %v4float %pos0_minus_x_02 %pos0_minus_y_02 %pos0_z %pos0_w
%pos2_plus_02 = OpCompositeConstruct %v4float %pos2_plus_x_02 %pos2_plus_y_02 %pos2_z %pos2_w
%pos2_minus_02 = OpCompositeConstruct %v4float %pos2_minus_x_02 %pos2_minus_y_02 %pos2_z %pos2_w

OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos0_plus_02
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos0_minus_02
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos2_plus_02
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos2_minus_02
OpEmitVertex
OpEndPrimitive

; ---------- edge 2->4 thick quad ----------
%diff_24 = OpFSub %v2float %pos4_ndc %pos2_ndc
%dir_24 = OpExtInst %v2float %glsl Normalize %diff_24
%dir_24_x = OpCompositeExtract %float %dir_24 0
%dir_24_y = OpCompositeExtract %float %dir_24 1
%neg_dir_24_y = OpFNegate %float %dir_24_y
%perp_24 = OpCompositeConstruct %v2float %neg_dir_24_y %dir_24_x
%offset_ndc_24 = OpVectorTimesScalar %v2float %perp_24 %float_0_015

%offset_24_2 = OpVectorTimesScalar %v2float %offset_ndc_24 %pos2_w
%offset_24_4 = OpVectorTimesScalar %v2float %offset_ndc_24 %pos4_w

%pos2_plus_xy_24 = OpFAdd %v2float %pos2_xy %offset_24_2
%pos2_minus_xy_24 = OpFSub %v2float %pos2_xy %offset_24_2
%pos4_plus_xy_24 = OpFAdd %v2float %pos4_xy %offset_24_4
%pos4_minus_xy_24 = OpFSub %v2float %pos4_xy %offset_24_4

%pos2_plus_x_24 = OpCompositeExtract %float %pos2_plus_xy_24 0
%pos2_plus_y_24 = OpCompositeExtract %float %pos2_plus_xy_24 1
%pos2_minus_x_24 = OpCompositeExtract %float %pos2_minus_xy_24 0
%pos2_minus_y_24 = OpCompositeExtract %float %pos2_minus_xy_24 1
%pos4_plus_x_24 = OpCompositeExtract %float %pos4_plus_xy_24 0
%pos4_plus_y_24 = OpCompositeExtract %float %pos4_plus_xy_24 1
%pos4_minus_x_24 = OpCompositeExtract %float %pos4_minus_xy_24 0
%pos4_minus_y_24 = OpCompositeExtract %float %pos4_minus_xy_24 1

%pos2_plus_24 = OpCompositeConstruct %v4float %pos2_plus_x_24 %pos2_plus_y_24 %pos2_z %pos2_w
%pos2_minus_24 = OpCompositeConstruct %v4float %pos2_minus_x_24 %pos2_minus_y_24 %pos2_z %pos2_w
%pos4_plus_24 = OpCompositeConstruct %v4float %pos4_plus_x_24 %pos4_plus_y_24 %pos4_z %pos4_w
%pos4_minus_24 = OpCompositeConstruct %v4float %pos4_minus_x_24 %pos4_minus_y_24 %pos4_z %pos4_w

OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos2_plus_24
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos2_minus_24
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos4_plus_24
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos4_minus_24
OpEmitVertex
OpEndPrimitive

; ---------- edge 4->0 thick quad ----------
%diff_40 = OpFSub %v2float %pos0_ndc %pos4_ndc
%dir_40 = OpExtInst %v2float %glsl Normalize %diff_40
%dir_40_x = OpCompositeExtract %float %dir_40 0
%dir_40_y = OpCompositeExtract %float %dir_40 1
%neg_dir_40_y = OpFNegate %float %dir_40_y
%perp_40 = OpCompositeConstruct %v2float %neg_dir_40_y %dir_40_x
%offset_ndc_40 = OpVectorTimesScalar %v2float %perp_40 %float_0_015

%offset_40_4 = OpVectorTimesScalar %v2float %offset_ndc_40 %pos4_w
%offset_40_0 = OpVectorTimesScalar %v2float %offset_ndc_40 %pos0_w

%pos4_plus_xy_40 = OpFAdd %v2float %pos4_xy %offset_40_4
%pos4_minus_xy_40 = OpFSub %v2float %pos4_xy %offset_40_4
%pos0_plus_xy_40 = OpFAdd %v2float %pos0_xy %offset_40_0
%pos0_minus_xy_40 = OpFSub %v2float %pos0_xy %offset_40_0

%pos4_plus_x_40 = OpCompositeExtract %float %pos4_plus_xy_40 0
%pos4_plus_y_40 = OpCompositeExtract %float %pos4_plus_xy_40 1
%pos4_minus_x_40 = OpCompositeExtract %float %pos4_minus_xy_40 0
%pos4_minus_y_40 = OpCompositeExtract %float %pos4_minus_xy_40 1
%pos0_plus_x_40 = OpCompositeExtract %float %pos0_plus_xy_40 0
%pos0_plus_y_40 = OpCompositeExtract %float %pos0_plus_xy_40 1
%pos0_minus_x_40 = OpCompositeExtract %float %pos0_minus_xy_40 0
%pos0_minus_y_40 = OpCompositeExtract %float %pos0_minus_xy_40 1

%pos4_plus_40 = OpCompositeConstruct %v4float %pos4_plus_x_40 %pos4_plus_y_40 %pos4_z %pos4_w
%pos4_minus_40 = OpCompositeConstruct %v4float %pos4_minus_x_40 %pos4_minus_y_40 %pos4_z %pos4_w
%pos0_plus_40 = OpCompositeConstruct %v4float %pos0_plus_x_40 %pos0_plus_y_40 %pos0_z %pos0_w
%pos0_minus_40 = OpCompositeConstruct %v4float %pos0_minus_x_40 %pos0_minus_y_40 %pos0_z %pos0_w

OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos4_plus_40
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos4_minus_40
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos0_plus_40
OpEmitVertex
OpStore %outColor %v3float_1
OpStore %gl_out_pos_ptr %pos0_minus_40
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
