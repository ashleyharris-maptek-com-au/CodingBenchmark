OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %in_color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3_in ArrayStride 16

OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_n1 = OpConstant %float -1.0
%thickness = OpConstant %float 0.02
%one_third = OpConstant %float 0.333333343
%two_third = OpConstant %float 0.666666687
%min_denom = OpConstant %float 0.1

%main = OpFunction %void None %fn_void
%entry = OpLabel

%gl_out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p0_w = OpCompositeExtract %float %p0 3
%p0_ww = OpCompositeConstruct %v2float %p0_w %p0_w
%p0_ndc = OpFDiv %v2float %p0_xy %p0_ww

%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p1_w = OpCompositeExtract %float %p1 3
%p1_ww = OpCompositeConstruct %v2float %p1_w %p1_w
%p1_ndc = OpFDiv %v2float %p1_xy %p1_ww

%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p2_w = OpCompositeExtract %float %p2 3
%p2_ww = OpCompositeConstruct %v2float %p2_w %p2_w
%p2_ndc = OpFDiv %v2float %p2_xy %p2_ww

%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%p3_w = OpCompositeExtract %float %p3 3
%p3_ww = OpCompositeConstruct %v2float %p3_w %p3_w
%p3_ndc = OpFDiv %v2float %p3_xy %p3_ww

%dir0_un = OpFSub %v2float %p1_ndc %p0_ndc
%dir0 = OpExtInst %v2float %glsl Normalize %dir0_un
%dir1_un = OpFSub %v2float %p2_ndc %p1_ndc
%dir1 = OpExtInst %v2float %glsl Normalize %dir1_un
%dir2_un = OpFSub %v2float %p3_ndc %p2_ndc
%dir2 = OpExtInst %v2float %glsl Normalize %dir2_un

%dir0_x = OpCompositeExtract %float %dir0 0
%dir0_y = OpCompositeExtract %float %dir0 1
%ndir0_y = OpFNegate %float %dir0_y
%perp0 = OpCompositeConstruct %v2float %ndir0_y %dir0_x

%dir1_x = OpCompositeExtract %float %dir1 0
%dir1_y = OpCompositeExtract %float %dir1 1
%ndir1_y = OpFNegate %float %dir1_y
%perp1 = OpCompositeConstruct %v2float %ndir1_y %dir1_x

%dir2_x = OpCompositeExtract %float %dir2 0
%dir2_y = OpCompositeExtract %float %dir2 1
%ndir2_y = OpFNegate %float %dir2_y
%perp2 = OpCompositeConstruct %v2float %ndir2_y %dir2_x

%miterS_sum = OpFAdd %v2float %perp0 %perp1
%miterS_dir = OpExtInst %v2float %glsl Normalize %miterS_sum
%denomS = OpDot %float %miterS_dir %perp1
%denomS_neg = OpFOrdLessThan %bool %denomS %float_0
%denomS_sign = OpSelect %float %denomS_neg %float_n1 %float_1
%denomS_abs = OpExtInst %float %glsl FAbs %denomS
%denomS_abs_clamp = OpExtInst %float %glsl FMax %denomS_abs %min_denom
%denomS_safe = OpFMul %float %denomS_abs_clamp %denomS_sign
%scaleS = OpFDiv %float %thickness %denomS_safe
%miterS = OpVectorTimesScalar %v2float %miterS_dir %scaleS

%miterE_sum = OpFAdd %v2float %perp1 %perp2
%miterE_dir = OpExtInst %v2float %glsl Normalize %miterE_sum
%denomE = OpDot %float %miterE_dir %perp1
%denomE_neg = OpFOrdLessThan %bool %denomE %float_0
%denomE_sign = OpSelect %float %denomE_neg %float_n1 %float_1
%denomE_abs = OpExtInst %float %glsl FAbs %denomE
%denomE_abs_clamp = OpExtInst %float %glsl FMax %denomE_abs %min_denom
%denomE_safe = OpFMul %float %denomE_abs_clamp %denomE_sign
%scaleE = OpFDiv %float %thickness %denomE_safe
%miterE = OpVectorTimesScalar %v2float %miterE_dir %scaleE

%perp_thick = OpVectorTimesScalar %v2float %perp1 %thickness

%delta_clip = OpFSub %v4float %p2 %p1
%delta_clip_one_third = OpVectorTimesScalar %v4float %delta_clip %one_third
%pB = OpFAdd %v4float %p1 %delta_clip_one_third
%delta_clip_two_third = OpVectorTimesScalar %v4float %delta_clip %two_third
%pC = OpFAdd %v4float %p1 %delta_clip_two_third

%c1_ptr = OpAccessChain %ptr_in_v3float %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3float %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr
%delta_c = OpFSub %v3float %c2 %c1
%delta_c_one_third = OpVectorTimesScalar %v3float %delta_c %one_third
%cB = OpFAdd %v3float %c1 %delta_c_one_third
%delta_c_two_third = OpVectorTimesScalar %v3float %delta_c %two_third
%cC = OpFAdd %v3float %c1 %delta_c_two_third

; ---- Dash 1: [0, 1/3] ----
%offS_clip = OpVectorTimesScalar %v2float %miterS %p1_w
%pB_w = OpCompositeExtract %float %pB 3
%offB_clip = OpVectorTimesScalar %v2float %perp_thick %pB_w

%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%offS_x = OpCompositeExtract %float %offS_clip 0
%offS_y = OpCompositeExtract %float %offS_clip 1
%p1_lx = OpFAdd %float %p1_x %offS_x
%p1_ly = OpFAdd %float %p1_y %offS_y
%p1_rx = OpFSub %float %p1_x %offS_x
%p1_ry = OpFSub %float %p1_y %offS_y
%p1_left = OpCompositeConstruct %v4float %p1_lx %p1_ly %p1_z %p1_w
%p1_right = OpCompositeConstruct %v4float %p1_rx %p1_ry %p1_z %p1_w

%pB_x = OpCompositeExtract %float %pB 0
%pB_y = OpCompositeExtract %float %pB 1
%pB_z = OpCompositeExtract %float %pB 2
%offB_x = OpCompositeExtract %float %offB_clip 0
%offB_y = OpCompositeExtract %float %offB_clip 1
%pB_lx = OpFAdd %float %pB_x %offB_x
%pB_ly = OpFAdd %float %pB_y %offB_y
%pB_rx = OpFSub %float %pB_x %offB_x
%pB_ry = OpFSub %float %pB_y %offB_y
%pB_left = OpCompositeConstruct %v4float %pB_lx %pB_ly %pB_z %pB_w
%pB_right = OpCompositeConstruct %v4float %pB_rx %pB_ry %pB_z %pB_w

OpStore %gl_out_pos_ptr %p1_left
OpStore %outColor %c1
OpEmitVertex

OpStore %gl_out_pos_ptr %p1_right
OpStore %outColor %c1
OpEmitVertex

OpStore %gl_out_pos_ptr %pB_left
OpStore %outColor %cB
OpEmitVertex

OpStore %gl_out_pos_ptr %pB_right
OpStore %outColor %cB
OpEmitVertex

OpEndPrimitive

; ---- Dash 3: [2/3, 1] ----
%pC_w = OpCompositeExtract %float %pC 3
%offC_clip = OpVectorTimesScalar %v2float %perp_thick %pC_w
%offE_clip = OpVectorTimesScalar %v2float %miterE %p2_w

%pC_x = OpCompositeExtract %float %pC 0
%pC_y = OpCompositeExtract %float %pC 1
%pC_z = OpCompositeExtract %float %pC 2
%offC_x = OpCompositeExtract %float %offC_clip 0
%offC_y = OpCompositeExtract %float %offC_clip 1
%pC_lx = OpFAdd %float %pC_x %offC_x
%pC_ly = OpFAdd %float %pC_y %offC_y
%pC_rx = OpFSub %float %pC_x %offC_x
%pC_ry = OpFSub %float %pC_y %offC_y
%pC_left = OpCompositeConstruct %v4float %pC_lx %pC_ly %pC_z %pC_w
%pC_right = OpCompositeConstruct %v4float %pC_rx %pC_ry %pC_z %pC_w

%p2_x = OpCompositeExtract %float %p2 0
%p2_y = OpCompositeExtract %float %p2 1
%p2_z = OpCompositeExtract %float %p2 2
%offE_x = OpCompositeExtract %float %offE_clip 0
%offE_y = OpCompositeExtract %float %offE_clip 1
%p2_lx = OpFAdd %float %p2_x %offE_x
%p2_ly = OpFAdd %float %p2_y %offE_y
%p2_rx = OpFSub %float %p2_x %offE_x
%p2_ry = OpFSub %float %p2_y %offE_y
%p2_left = OpCompositeConstruct %v4float %p2_lx %p2_ly %p2_z %p2_w
%p2_right = OpCompositeConstruct %v4float %p2_rx %p2_ry %p2_z %p2_w

OpStore %gl_out_pos_ptr %pC_left
OpStore %outColor %cC
OpEmitVertex

OpStore %gl_out_pos_ptr %pC_right
OpStore %outColor %cC
OpEmitVertex

OpStore %gl_out_pos_ptr %p2_left
OpStore %outColor %c2
OpEmitVertex

OpStore %gl_out_pos_ptr %p2_right
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
