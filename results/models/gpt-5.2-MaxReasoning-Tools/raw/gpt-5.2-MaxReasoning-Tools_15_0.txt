use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, BufWriter, Read, Write};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

const REF_Z: f64 = 0.707_106_781_186_547_5; // sqrt(1/2)
const REF_VERT_RADIUS: f64 = 1.224_744_871_391_589; // sqrt(3/2)
const SQRT_3: f64 = 1.732_050_807_568_877_2;
const EPS: f64 = 1e-12;

const REF_VERTS: [Vec3; 4] = [
    Vec3 {
        x: 1.0,
        y: 0.0,
        z: -REF_Z,
    },
    Vec3 {
        x: -1.0,
        y: 0.0,
        z: -REF_Z,
    },
    Vec3 {
        x: 0.0,
        y: 1.0,
        z: REF_Z,
    },
    Vec3 {
        x: 0.0,
        y: -1.0,
        z: REF_Z,
    },
];

fn hypot2(x: f64, y: f64) -> f64 {
    (x * x + y * y).sqrt()
}

fn v2_add(a: Vec2, b: Vec2) -> Vec2 {
    Vec2 {
        x: a.x + b.x,
        y: a.y + b.y,
    }
}

fn v2_sub(a: Vec2, b: Vec2) -> Vec2 {
    Vec2 {
        x: a.x - b.x,
        y: a.y - b.y,
    }
}

fn v2_mul(a: Vec2, s: f64) -> Vec2 {
    Vec2 {
        x: a.x * s,
        y: a.y * s,
    }
}

fn v3_mul(a: Vec3, s: f64) -> Vec3 {
    Vec3 {
        x: a.x * s,
        y: a.y * s,
        z: a.z * s,
    }
}

fn cross2(a: Vec2, b: Vec2) -> f64 {
    a.x * b.y - a.y * b.x
}

fn project_to_z0(p: Vec3, kx: f64, ky: f64) -> Vec2 {
    Vec2 {
        x: p.x - p.z * kx,
        y: p.y - p.z * ky,
    }
}

fn quat_normalize(q: Quat) -> Quat {
    let n2 = q.w * q.w + q.x * q.x + q.y * q.y + q.z * q.z;
    if n2 <= 0.0 {
        return Quat {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        };
    }
    let inv = 1.0 / n2.sqrt();
    Quat {
        w: q.w * inv,
        x: q.x * inv,
        y: q.y * inv,
        z: q.z * inv,
    }
}

fn quat_to_mat3(q: Quat) -> [[f64; 3]; 3] {
    let q = quat_normalize(q);
    let w = q.w;
    let x = q.x;
    let y = q.y;
    let z = q.z;

    let xx = x * x;
    let yy = y * y;
    let zz = z * z;
    let xy = x * y;
    let xz = x * z;
    let yz = y * z;
    let wx = w * x;
    let wy = w * y;
    let wz = w * z;

    [
        [1.0 - 2.0 * (yy + zz), 2.0 * (xy - wz), 2.0 * (xz + wy)],
        [2.0 * (xy + wz), 1.0 - 2.0 * (xx + zz), 2.0 * (yz - wx)],
        [2.0 * (xz - wy), 2.0 * (yz + wx), 1.0 - 2.0 * (xx + yy)],
    ]
}

fn mat3_mul_vec3(m: &[[f64; 3]; 3], v: Vec3) -> Vec3 {
    Vec3 {
        x: m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z,
        y: m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z,
        z: m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z,
    }
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    if pts.len() <= 1 {
        return pts;
    }

    pts.sort_unstable_by(|a, b| {
        let cx = a.x.total_cmp(&b.x);
        if cx == Ordering::Equal {
            a.y.total_cmp(&b.y)
        } else {
            cx
        }
    });

    // Deduplicate nearly-equal points (only 4 points, so this is cheap).
    let mut uniq: Vec<Vec2> = Vec::with_capacity(pts.len());
    for p in pts {
        if let Some(last) = uniq.last() {
            let dx = p.x - last.x;
            let dy = p.y - last.y;
            if dx * dx + dy * dy <= 1e-24 {
                continue;
            }
        }
        uniq.push(p);
    }
    if uniq.len() <= 1 {
        return uniq;
    }

    let mut lower: Vec<Vec2> = Vec::with_capacity(uniq.len());
    for &p in &uniq {
        while lower.len() >= 2 {
            let a = lower[lower.len() - 2];
            let b = lower[lower.len() - 1];
            let ab = v2_sub(b, a);
            let ap = v2_sub(p, a);
            if cross2(ab, ap) <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }

    let mut upper: Vec<Vec2> = Vec::with_capacity(uniq.len());
    for &p in uniq.iter().rev() {
        while upper.len() >= 2 {
            let a = upper[upper.len() - 2];
            let b = upper[upper.len() - 1];
            let ab = v2_sub(b, a);
            let ap = v2_sub(p, a);
            if cross2(ab, ap) <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn clip_poly_halfplane(poly: &[Vec2], a: Vec2, b: Vec2, offset: f64) -> Vec<Vec2> {
    if poly.is_empty() {
        return Vec::new();
    }

    let e = v2_sub(b, a);
    let mut output: Vec<Vec2> = Vec::new();

    let mut s = poly[poly.len() - 1];
    let mut s_val = cross2(e, v2_sub(s, a)) - offset;
    for &p in poly {
        let p_val = cross2(e, v2_sub(p, a)) - offset;
        let s_in = s_val >= -EPS;
        let p_in = p_val >= -EPS;

        if p_in {
            if !s_in {
                let t = s_val / (s_val - p_val);
                output.push(v2_add(s, v2_mul(v2_sub(p, s), t)));
            }
            output.push(p);
        } else if s_in {
            let t = s_val / (s_val - p_val);
            output.push(v2_add(s, v2_mul(v2_sub(p, s), t)));
        }

        s = p;
        s_val = p_val;
    }
    output
}

fn chebyshev_center_and_radius(hull: &[Vec2]) -> (Vec2, f64) {
    if hull.len() < 3 {
        return (Vec2 { x: 0.0, y: 0.0 }, 0.0);
    }

    let mut mean = Vec2 { x: 0.0, y: 0.0 };
    for &p in hull {
        mean = v2_add(mean, p);
    }
    mean = v2_mul(mean, 1.0 / (hull.len() as f64));

    let mut hi: f64 = 0.0;
    for &p in hull {
        hi = hi.max(hypot2(p.x - mean.x, p.y - mean.y));
    }

    let mut lo: f64 = 0.0;
    let mut best_poly: Vec<Vec2> = hull.to_vec();

    for _ in 0..40 {
        let mid = (lo + hi) * 0.5;
        let mut poly = hull.to_vec();
        for i in 0..hull.len() {
            let a = hull[i];
            let b = hull[(i + 1) % hull.len()];
            let edge = v2_sub(b, a);
            let edge_len = hypot2(edge.x, edge.y);
            poly = clip_poly_halfplane(&poly, a, b, mid * edge_len);
            if poly.len() < 3 {
                break;
            }
        }

        if poly.len() >= 3 {
            lo = mid;
            best_poly = poly;
        } else {
            hi = mid;
        }
    }

    // Any point inside best_poly is a valid center for radius lo; use vertex average.
    let mut c = Vec2 { x: 0.0, y: 0.0 };
    for &p in &best_poly {
        c = v2_add(c, p);
    }
    c = v2_mul(c, 1.0 / (best_poly.len() as f64));
    (c, lo)
}

struct ShadowDisk {
    center: Vec2,
    radius: f64,
}

fn shadow_chebyshev_disk(q: Quat, scale: f64, kx: f64, ky: f64) -> ShadowDisk {
    let m = quat_to_mat3(q);
    let mut pts: Vec<Vec2> = Vec::with_capacity(4);
    for v in REF_VERTS {
        let vr = mat3_mul_vec3(&m, v);
        let wp = v3_mul(vr, scale);
        pts.push(project_to_z0(wp, kx, ky));
    }
    let hull = convex_hull(pts);
    let (center, radius) = chebyshev_center_and_radius(&hull);
    ShadowDisk { center, radius }
}

struct XorShift64Star {
    state: u64,
}

impl XorShift64Star {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9E37_79B9_7F4A_7C15 } else { seed };
        Self { state: s }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.state = x;
        x.wrapping_mul(2685821657736338717u64)
    }

    fn next_f64(&mut self) -> f64 {
        // Use the top 53 bits for a uniform [0,1) f64.
        const INV_2_POW_53: f64 = 1.0 / 9_007_199_254_740_992.0;
        let v = self.next_u64() >> 11;
        (v as f64) * INV_2_POW_53
    }
}

fn random_unit_quat(rng: &mut XorShift64Star) -> Quat {
    let u1 = rng.next_f64();
    let u2 = rng.next_f64();
    let u3 = rng.next_f64();

    let s1 = (1.0 - u1).sqrt();
    let s2 = u1.sqrt();
    let t1 = 2.0 * PI * u2;
    let t2 = 2.0 * PI * u3;

    // Shoemake (1992): uniform random rotation via unit quaternion.
    let x = s1 * t1.sin();
    let y = s1 * t1.cos();
    let z = s2 * t2.sin();
    let w = s2 * t2.cos();

    quat_normalize(Quat { w, x, y, z })
}

struct BestOrientation {
    q: Quat,
    disk: ShadowDisk,
}

fn pick_orientation(nv: usize, bbox_diag: f64, kx: f64, ky: f64) -> BestOrientation {
    let scale = 10.0;
    let mut iters = 128usize;

    // Adaptive effort: do more work when the polygon is small (where fewer tetrahedra are achievable).
    if nv <= 64 {
        iters = 2048;
    } else if nv <= 4096 && bbox_diag < 1.0e6 {
        iters = 512;
    } else if bbox_diag < 1.0e3 {
        iters = 1024;
    }

    let mut best_q = Quat {
        w: 1.0,
        x: 0.0,
        y: 0.0,
        z: 0.0,
    };
    let mut best_disk = shadow_chebyshev_disk(best_q, scale, kx, ky);

    // Seed chosen for deterministic behavior.
    let mut rng = XorShift64Star::new(0xD1B5_4A32_D192_ED03);
    for _ in 0..iters {
        let q = random_unit_quat(&mut rng);
        let d = shadow_chebyshev_disk(q, scale, kx, ky);
        if d.radius > best_disk.radius {
            best_q = q;
            best_disk = d;
        }
    }

    // Defensive fallback: if something went degenerate, use identity.
    if !(best_disk.radius > 0.0) {
        best_q = Quat {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        };
        best_disk = shadow_chebyshev_disk(best_q, scale, kx, ky);
    }

    BestOrientation { q: best_q, disk: best_disk }
}

fn hex_lattice_count(min_x: f64, max_x: f64, min_y: f64, max_y: f64, r: f64) -> usize {
    let dx = r * SQRT_3;
    let dy = 1.5 * r;

    let start_y = min_y - 2.0 * r;
    let end_y = max_y + 2.0 * r;
    let start_x = min_x - 2.0 * r - dx;
    let end_x = max_x + 2.0 * r + dx;

    let rows = ((end_y - start_y) / dy).ceil().max(0.0) as usize + 1;
    let mut count = 0usize;
    for row in 0..rows {
        let offset_x = if (row & 1) == 0 { 0.0 } else { 0.5 * dx };
        let x0 = start_x + offset_x;
        let cols = ((end_x - x0) / dx).ceil().max(0.0) as usize + 1;
        count += cols;
    }
    count
}

fn hex_lattice_for_each<F: FnMut(Vec2)>(
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
    r: f64,
    mut f: F,
) {
    let dx = r * SQRT_3;
    let dy = 1.5 * r;

    let start_y = min_y - 2.0 * r;
    let end_y = max_y + 2.0 * r;
    let start_x = min_x - 2.0 * r - dx;
    let end_x = max_x + 2.0 * r + dx;

    let rows = ((end_y - start_y) / dy).ceil().max(0.0) as usize + 1;
    for row in 0..rows {
        let y = start_y + (row as f64) * dy;
        let offset_x = if (row & 1) == 0 { 0.0 } else { 0.5 * dx };
        let x0 = start_x + offset_x;
        let cols = ((end_x - x0) / dx).ceil().max(0.0) as usize + 1;
        for col in 0..cols {
            let x = x0 + (col as f64) * dx;
            f(Vec2 { x, y });
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let input = input.trim_start_matches('\u{FEFF}');
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };

    if nv == 0 {
        println!("0");
        return;
    }

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        if x < min_x {
            min_x = x;
        }
        if x > max_x {
            max_x = x;
        }
        if y < min_y {
            min_y = y;
        }
        if y > max_y {
            max_y = y;
        }
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    // Oblique projection to z=0 along sun direction.
    let (kx, ky) = if sz.abs() < 1e-12 {
        (0.0, 0.0)
    } else {
        (sx / sz, sy / sz)
    };

    let width = (max_x - min_x).abs();
    let height = (max_y - min_y).abs();
    let diag = hypot2(width, height);

    let best = pick_orientation(nv, diag, kx, ky);
    let q = quat_normalize(best.q);
    let scale = 10.0f64;
    let disk_center0 = best.disk.center;
    let r = best.disk.radius;

    // 3D non-intersection: separate tetrahedra by z (bounding-sphere argument).
    let bound_r = scale * REF_VERT_RADIUS;
    let sep_z = 3.0 * bound_r + 1.0;
    let base_z = bound_r + 1.0;

    let mut out = BufWriter::new(io::stdout().lock());

    // If the AABB fits inside one guaranteed shadow disk, use a single tetrahedron.
    let req = 0.5 * diag;
    if req <= r {
        let target = Vec2 {
            x: 0.5 * (min_x + max_x),
            y: 0.5 * (min_y + max_y),
        };
        let tz = base_z;
        let offset = v2_sub(target, disk_center0);
        let tx = offset.x + tz * kx;
        let ty = offset.y + tz * ky;

        writeln!(out, "1").unwrap();
        writeln!(
            out,
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            tx, ty, tz, q.w, q.x, q.y, q.z, scale
        )
        .unwrap();
        out.flush().unwrap();
        return;
    }

    // Otherwise, cover the AABB with a hexagonal lattice of inscribed disks (disks lie inside the shadows).
    // This guarantees full coverage of the polygon (since polygon ⊆ AABB).
    let count = hex_lattice_count(min_x, max_x, min_y, max_y, r);
    writeln!(out, "{}", count).unwrap();

    let mut idx = 0usize;
    hex_lattice_for_each(min_x, max_x, min_y, max_y, r, |p| {
        let tz = base_z + (idx as f64) * sep_z;
        let offset = v2_sub(p, disk_center0);
        let tx = offset.x + tz * kx;
        let ty = offset.y + tz * ky;
        writeln!(
            out,
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            tx, ty, tz, q.w, q.x, q.y, q.z, scale
        )
        .unwrap();
        idx += 1;
    });

    out.flush().unwrap();
}
