import sys
import math
import random
import time


def read_instance():
    data = sys.stdin.buffer.read().split()
    if not data:
        return 0, 0, 0.0, 0.0, 0.0, [], [], []
    # Handle optional UTF-8 BOM on the first token.
    if data[0].startswith(b"\xef\xbb\xbf"):
        data[0] = data[0][3:]
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))
    xs = [0.0] * n
    ys = [0.0] * n
    demands = [0.0] * n
    for i in range(n):
        xs[i] = float(next(it))
        ys[i] = float(next(it))
        demands[i] = float(next(it))
    return n, k, capacity, depot_x, depot_y, xs, ys, demands


def build_distance_helpers(n, depot_x, depot_y, xs, ys):
    d0 = [0.0] * n
    for i in range(n):
        d0[i] = math.hypot(xs[i] - depot_x, ys[i] - depot_y)

    dist_mat = None
    # Full distance matrix is fast for local search but can be memory-heavy.
    if n <= 2000:
        dist_mat = [[0.0] * n for _ in range(n)]
        for i in range(n):
            xi, yi = xs[i], ys[i]
            row_i = dist_mat[i]
            for j in range(i + 1, n):
                d = math.hypot(xi - xs[j], yi - ys[j])
                row_i[j] = d
                dist_mat[j][i] = d

    if dist_mat is None:

        def dist_cust(i, j):
            return math.hypot(xs[i] - xs[j], ys[i] - ys[j])

    else:

        def dist_cust(i, j):
            return dist_mat[i][j]

    def dist_node(a, b):
        if a < 0:
            return d0[b]
        if b < 0:
            return d0[a]
        return dist_cust(a, b)

    return d0, dist_cust, dist_node


def route_load(route, demands):
    return sum(demands[i] for i in route)


def route_distance(route, dist_node):
    if not route:
        return 0.0
    total = dist_node(-1, route[0])
    for i in range(len(route) - 1):
        total += dist_node(route[i], route[i + 1])
    total += dist_node(route[-1], -1)
    return total


def total_distance(routes, dist_node):
    return sum(route_distance(r, dist_node) for r in routes)


def nearest_neighbor_order(customers, dist_node, d0):
    if not customers:
        return []
    remaining = customers[:]
    # Start at the closest customer to depot to keep early edges short.
    start_idx = min(range(len(remaining)), key=lambda t: d0[remaining[t]])
    cur = remaining.pop(start_idx)
    route = [cur]
    while remaining:
        next_pos = 0
        best_d = dist_node(cur, remaining[0])
        for i in range(1, len(remaining)):
            d = dist_node(cur, remaining[i])
            if d < best_d:
                best_d = d
                next_pos = i
        cur = remaining.pop(next_pos)
        route.append(cur)
    return route


def two_opt_first_improvement(route, dist_node, max_passes=50):
    m = len(route)
    if m < 4:
        return route
    passes = 0
    while passes < max_passes:
        passes += 1
        improved = False
        for i in range(m - 2):
            prev_node = -1 if i == 0 else route[i - 1]
            a = route[i]
            for j in range(i + 1, m):
                if j == i + 1:
                    continue
                b = route[j]
                next_node = -1 if j == m - 1 else route[j + 1]
                old = dist_node(prev_node, a) + dist_node(b, next_node)
                new = dist_node(prev_node, b) + dist_node(a, next_node)
                if new + 1e-12 < old:
                    route[i : j + 1] = reversed(route[i : j + 1])
                    improved = True
                    break
            if improved:
                break
        if not improved:
            break
    return route


def best_insertion_position(route, customer, dist_node):
    # Returns (delta_cost, position)
    if not route:
        return 2.0 * dist_node(-1, customer), 0

    best_delta = float("inf")
    best_pos = 0
    # Insert before first.
    delta = dist_node(-1, customer) + dist_node(customer, route[0]) - dist_node(-1, route[0])
    if delta < best_delta:
        best_delta = delta
        best_pos = 0
    # Insert between customers.
    for pos in range(1, len(route)):
        a = route[pos - 1]
        b = route[pos]
        delta = dist_node(a, customer) + dist_node(customer, b) - dist_node(a, b)
        if delta < best_delta:
            best_delta = delta
            best_pos = pos
    # Insert after last.
    delta = dist_node(route[-1], customer) + dist_node(customer, -1) - dist_node(route[-1], -1)
    if delta < best_delta:
        best_delta = delta
        best_pos = len(route)
    return best_delta, best_pos


def reduce_routes_to_k(routes, k, capacity, demands, dist_node, time_limit_s=1.0):
    # Attempts to remove routes by reinserting their customers into others.
    start = time.time()
    routes = [r[:] for r in routes if r]
    loads = [route_load(r, demands) for r in routes]
    blocked = set()

    while len(routes) > k and time.time() - start < time_limit_s:
        # Pick the smallest load route that isn't blocked.
        candidates = [(loads[i], len(routes[i]), i) for i in range(len(routes)) if i not in blocked]
        if not candidates:
            break
        _, _, ridx = min(candidates)
        to_remove = routes[ridx]
        to_remove_customers = to_remove[:]

        others = [i for i in range(len(routes)) if i != ridx]
        # Greedily insert each customer into the best place across all other routes.
        ok = True
        while to_remove_customers:
            best = None  # (delta, other_route_index, pos, customer_index_in_list)
            for ci, cust in enumerate(to_remove_customers):
                d = demands[cust]
                for oi in others:
                    if loads[oi] + d > capacity + 1e-9:
                        continue
                    delta, pos = best_insertion_position(routes[oi], cust, dist_node)
                    if best is None or delta < best[0]:
                        best = (delta, oi, pos, ci)
            if best is None:
                ok = False
                break
            _, oi, pos, ci = best
            cust = to_remove_customers.pop(ci)
            routes[oi].insert(pos, cust)
            loads[oi] += demands[cust]

        if ok:
            # Remove the emptied route.
            routes.pop(ridx)
            loads.pop(ridx)
        else:
            blocked.add(ridx)

    return routes


def sweep_construct(xs, ys, demands, capacity, depot_x, depot_y, offsets_to_try=12):
    n = len(xs)
    angles = []
    for i in range(n):
        angles.append((math.atan2(ys[i] - depot_y, xs[i] - depot_x), i))
    angles.sort()
    order = [i for _, i in angles]
    if n == 0:
        return []

    # Try several rotations; pick the one with fewest routes first, then by distance.
    step = max(1, n // max(1, offsets_to_try))
    offsets = {0}
    for t in range(1, offsets_to_try):
        offsets.add((t * step) % n)
    return order, sorted(offsets)


def build_routes_from_order(order, start_idx, demands, capacity):
    n = len(order)
    routes = []
    cur = []
    load = 0.0
    for t in range(n):
        c = order[(start_idx + t) % n]
        d = demands[c]
        if cur and load + d > capacity + 1e-9:
            routes.append(cur)
            cur = []
            load = 0.0
        cur.append(c)
        load += d
    if cur:
        routes.append(cur)
    return routes


def best_fit_decreasing_pack(customers, demands, capacity, k, attempts=60):
    # Returns list of routes (bins) or None if fails.
    # Multiple attempts with slight randomization to avoid heuristic failures.
    best = None
    rng = random.Random(0)
    for t in range(attempts):
        jitter = 1e-6 if t == 0 else 1e-3
        order = sorted(
            customers,
            key=lambda i: (-demands[i], rng.random() * jitter, i),
        )
        bins = [[] for _ in range(k)]
        rem = [capacity] * k
        ok = True
        for c in order:
            d = demands[c]
            best_bin = -1
            best_rem_after = None
            for bi in range(k):
                if rem[bi] + 1e-9 < d:
                    continue
                rem_after = rem[bi] - d
                if best_bin < 0 or rem_after < best_rem_after:
                    best_bin = bi
                    best_rem_after = rem_after
            if best_bin < 0:
                ok = False
                break
            bins[best_bin].append(c)
            rem[best_bin] -= d
        if ok:
            packed = [b for b in bins if b]
            # Prefer solutions with fewer non-empty bins.
            if best is None or len(packed) < len(best):
                best = packed
            if len(packed) <= k:
                return packed
    return best


def solve():
    n, k, capacity, depot_x, depot_y, xs, ys, demands = read_instance()
    if n == 0:
        sys.stdout.write("0\n0\n")
        return

    # Feasibility quick checks.
    if any(d > capacity + 1e-9 for d in demands):
        # Cannot produce a valid solution.
        # Still output each customer as a separate "route" to avoid crashing.
        routes = [[i] for i in range(n)]
        d0, _, dist_node = build_distance_helpers(n, depot_x, depot_y, xs, ys)
        tot = total_distance(routes, dist_node)
        out = [f"{tot:.6f}", str(len(routes))]
        out.extend(" ".join(map(str, r)) for r in routes)
        sys.stdout.write("\n".join(out) + "\n")
        return
    if sum(demands) > k * capacity + 1e-9:
        # Infeasible under given vehicle count; produce something but cannot satisfy constraints.
        routes = [[i] for i in range(n)]
        d0, _, dist_node = build_distance_helpers(n, depot_x, depot_y, xs, ys)
        tot = total_distance(routes, dist_node)
        out = [f"{tot:.6f}", str(len(routes))]
        out.extend(" ".join(map(str, r)) for r in routes)
        sys.stdout.write("\n".join(out) + "\n")
        return

    d0, _, dist_node = build_distance_helpers(n, depot_x, depot_y, xs, ys)

    order, offsets = sweep_construct(xs, ys, demands, capacity, depot_x, depot_y)
    best_routes = None
    best_key = None

    for off in offsets:
        routes = build_routes_from_order(order, off, demands, capacity)
        # Early prune: if routes already exceed k significantly, still consider (we may reduce).
        # Improve ordering within each route cheaply.
        routes2 = []
        for r in routes:
            r_nn = nearest_neighbor_order(r, dist_node, d0)
            routes2.append(r_nn)
        key = (len(routes2), total_distance(routes2, dist_node))
        if best_key is None or key < best_key:
            best_key = key
            best_routes = routes2

    routes = best_routes if best_routes is not None else [[i] for i in range(n)]

    # Attempt to reduce number of routes if needed.
    if len(routes) > k:
        routes = reduce_routes_to_k(routes, k, capacity, demands, dist_node, time_limit_s=1.5)

    # Hard fallback: pack into at most k bins (capacity only), then route-second.
    if len(routes) > k:
        customers = list(range(n))
        packed = best_fit_decreasing_pack(customers, demands, capacity, k, attempts=80)
        if packed is not None:
            routes = []
            for b in packed:
                r_nn = nearest_neighbor_order(b, dist_node, d0)
                routes.append(r_nn)

    # Local search: 2-opt within routes (cap-safe).
    # Skip very long routes to keep runtime bounded.
    for i in range(len(routes)):
        if len(routes[i]) <= 300:
            two_opt_first_improvement(routes[i], dist_node, max_passes=80)

    # Final sanity: ensure every customer visited exactly once.
    seen = [0] * n
    for ridx, r in enumerate(list(routes)):
        load = 0.0
        for c in r:
            if 0 <= c < n:
                seen[c] += 1
            load += demands[c]
        if load > capacity + 1e-6:
            # Capacity violation: last resort split greedily, then re-validate.
            new_routes = []
            cur = []
            cur_load = 0.0
            for c in r:
                d = demands[c]
                if cur and cur_load + d > capacity + 1e-9:
                    new_routes.append(cur)
                    cur = []
                    cur_load = 0.0
                cur.append(c)
                cur_load += d
            if cur:
                new_routes.append(cur)
            routes[ridx : ridx + 1] = new_routes
            # Force rebuild-from-scratch validation path below.
            seen = [0] * n
            for rr in routes:
                for c in rr:
                    if 0 <= c < n:
                        seen[c] += 1
            break

    missing = [i for i, cnt in enumerate(seen) if cnt == 0]
    dupes = [i for i, cnt in enumerate(seen) if cnt > 1]
    if missing or dupes:
        # Repair by rebuilding from scratch using bin packing.
        customers = list(range(n))
        packed = best_fit_decreasing_pack(customers, demands, capacity, k, attempts=160)
        if packed is None:
            packed = [[i] for i in customers]
        routes = []
        for b in packed:
            r_nn = nearest_neighbor_order(b, dist_node, d0)
            if len(r_nn) <= 300:
                two_opt_first_improvement(r_nn, dist_node, max_passes=60)
            routes.append(r_nn)

    routes = [r for r in routes if r]
    # If we ended up with fewer than k routes, that's fine (up to k vehicles).
    # If we ended up with more than k routes, try one more reduction pass.
    if len(routes) > k:
        routes = reduce_routes_to_k(routes, k, capacity, demands, dist_node, time_limit_s=2.0)
        routes = [r for r in routes if r]

    tot = total_distance(routes, dist_node)
    out_lines = [f"{tot:.6f}", str(len(routes))]
    out_lines.extend(" ".join(map(str, r)) for r in routes)
    sys.stdout.write("\n".join(out_lines) + "\n")


if __name__ == "__main__":
    solve()
