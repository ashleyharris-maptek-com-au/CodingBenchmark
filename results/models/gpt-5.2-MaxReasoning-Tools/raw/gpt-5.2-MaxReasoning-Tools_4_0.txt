// 3D Tetrahedron Packing Solver (C++20, MSVC)
// Packs congruent regular tetrahedra (fixed orientation) into the convex hull
// of the given container vertices using an FCC lattice of circumscribed spheres.
//
// Input:
//   V edge_length
//   V lines: x y z
// Output:
//   K
//   K lines: cx cy cz rx ry rz  (Euler angles in radians; we output 0 0 0)
//
// Notes:
// - Container is assumed to be the convex polyhedron defined by the vertices.
// - Uses a randomized incremental 3D convex hull to derive halfspace planes.
// - Non-overlap is guaranteed by keeping tetrahedra circumspheres disjoint.
// - Placement is conservative: verifies containment via halfspace "shrink" by
//   the tetrahedron's support function in each plane normal direction.

#include <algorithm>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <random>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

namespace {

struct Vec3 {
    double x = 0, y = 0, z = 0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return Vec3{x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return Vec3{x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return Vec3{x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return Vec3{x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3{a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return std::sqrt(norm2(v)); }

struct Face {
    int a = -1, b = -1, c = -1;
    Vec3 n;     // outward (not necessarily unit)
    double d{}; // plane constant: n·x = d (for points on the plane)
};

// Unit normal plane form: n is unit outward normal; n·x <= t for points inside.
struct Plane {
    Vec3 n;      // unit outward normal
    double t{};  // offset
    double m{};  // support margin for tetrahedron (max_i n·v_i)
};

static inline bool isFiniteVec(const Vec3& v) {
    return std::isfinite(v.x) && std::isfinite(v.y) && std::isfinite(v.z);
}

static Face makeFaceOriented(const std::vector<Vec3>& P, int a, int b, int c, const Vec3& interior, double epsLen) {
    Face f;
    f.a = a;
    f.b = b;
    f.c = c;
    const Vec3 A = P[a], B = P[b], C = P[c];
    Vec3 n0 = cross(B - A, C - A);
    const double len = norm(n0);
    if (!(len > epsLen)) {
        f.a = f.b = f.c = -1;
        return f;
    }
    // Ensure interior lies on the "inside" side: n·interior <= d
    double d0 = dot(n0, A);
    if (dot(n0, interior) > d0) {
        std::swap(f.b, f.c);
        n0 = n0 * -1.0;
        d0 = -d0;
    }
    f.n = n0;
    f.d = d0;
    return f;
}

static bool distinctIndices(int a, int b, int c, int d) {
    return a != b && a != c && a != d && b != c && b != d && c != d;
}

static bool buildInitialSimplex(const std::vector<Vec3>& P, int& ia, int& ib, int& ic, int& id, double epsLen) {
    const int n = static_cast<int>(P.size());
    if (n < 4) return false;

    auto argMinMax = [&](auto coordFn) -> std::pair<int, int> {
        int imin = 0, imax = 0;
        double vmin = coordFn(P[0]), vmax = coordFn(P[0]);
        for (int i = 1; i < n; ++i) {
            const double v = coordFn(P[i]);
            if (v < vmin) {
                vmin = v;
                imin = i;
            }
            if (v > vmax) {
                vmax = v;
                imax = i;
            }
        }
        return {imin, imax};
    };

    auto [minx, maxx] = argMinMax([](const Vec3& v) { return v.x; });
    ia = minx;
    ib = maxx;
    if (ia == ib) {
        auto [miny, maxy] = argMinMax([](const Vec3& v) { return v.y; });
        ia = miny;
        ib = maxy;
    }
    if (ia == ib) {
        auto [minz, maxz] = argMinMax([](const Vec3& v) { return v.z; });
        ia = minz;
        ib = maxz;
    }
    if (ia == ib) return false;

    // Pick point farthest from line (ia, ib)
    const Vec3 A = P[ia], B = P[ib];
    const Vec3 AB = B - A;
    const double ab2 = norm2(AB);
    if (!(ab2 > epsLen * epsLen)) return false;

    ic = -1;
    double best = -1.0;
    for (int i = 0; i < n; ++i) {
        if (i == ia || i == ib) continue;
        const Vec3 AP = P[i] - A;
        const Vec3 cr = cross(AB, AP);
        const double area2 = norm2(cr);
        if (area2 > best) {
            best = area2;
            ic = i;
        }
    }
    if (ic < 0) return false;
    if (!(best > (epsLen * epsLen) * ab2)) return false; // near-collinear

    // Pick point farthest from plane (ia, ib, ic)
    const Vec3 C = P[ic];
    Vec3 n0 = cross(B - A, C - A);
    const double nlen = norm(n0);
    if (!(nlen > epsLen)) return false;

    id = -1;
    double bestDist = -1.0;
    for (int i = 0; i < n; ++i) {
        if (i == ia || i == ib || i == ic) continue;
        const double dist = std::fabs(dot(n0, P[i] - A)) / nlen;
        if (dist > bestDist) {
            bestDist = dist;
            id = i;
        }
    }
    if (id < 0) return false;
    if (!(bestDist > epsLen)) return false; // near-coplanar

    if (!distinctIndices(ia, ib, ic, id)) return false;
    return true;
}

static bool computeConvexHullFaces(const std::vector<Vec3>& P, std::vector<Face>& facesOut, Vec3& interiorOut) {
    const int n = static_cast<int>(P.size());
    if (n < 4) return false;

    // Scale-dependent eps.
    Vec3 mn = P[0], mx = P[0];
    for (const auto& v : P) {
        mn.x = std::min(mn.x, v.x);
        mn.y = std::min(mn.y, v.y);
        mn.z = std::min(mn.z, v.z);
        mx.x = std::max(mx.x, v.x);
        mx.y = std::max(mx.y, v.y);
        mx.z = std::max(mx.z, v.z);
    }
    const double scale = norm(mx - mn);
    const double epsLen = std::max(1e-12, scale * 1e-10);

    int ia, ib, ic, id;
    if (!buildInitialSimplex(P, ia, ib, ic, id, epsLen)) return false;

    const Vec3 interior = (P[ia] + P[ib] + P[ic] + P[id]) / 4.0;
    interiorOut = interior;

    std::vector<Face> faces;
    faces.reserve(4);
    auto pushFace = [&](int a, int b, int c) {
        Face f = makeFaceOriented(P, a, b, c, interior, epsLen);
        if (f.a >= 0) faces.push_back(f);
    };
    // Tetrahedron faces
    pushFace(ia, ib, ic);
    pushFace(ia, ic, id);
    pushFace(ia, id, ib);
    pushFace(ib, id, ic);
    if (faces.size() < 4) return false;

    std::vector<int> order;
    order.reserve(n - 4);
    for (int i = 0; i < n; ++i) {
        if (i == ia || i == ib || i == ic || i == id) continue;
        order.push_back(i);
    }
    // Randomize insertion order for better expected behavior.
    std::mt19937_64 rng(0xC0D3C0D3C0D3C0D3ULL ^ static_cast<std::uint64_t>(n));
    std::shuffle(order.begin(), order.end(), rng);

    // Incremental hull.
    for (int idx : order) {
        const Vec3 p = P[idx];
        if (!isFiniteVec(p)) continue;

        std::vector<char> visible(faces.size(), 0);
        bool anyVisible = false;
        for (size_t fi = 0; fi < faces.size(); ++fi) {
            const Face& f = faces[fi];
            const double nd = dot(f.n, p) - f.d;
            // tolerance scaled by normal magnitude
            const double tol = epsLen * norm(f.n);
            if (nd > tol) {
                visible[fi] = 1;
                anyVisible = true;
            }
        }
        if (!anyVisible) continue;

        std::unordered_set<std::uint64_t> boundary;
        boundary.reserve(faces.size() * 3);

        auto addEdge = [&](int u, int v) {
            const std::uint64_t key = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(u)) << 32) |
                                      static_cast<std::uint32_t>(v);
            const std::uint64_t rev = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(v)) << 32) |
                                      static_cast<std::uint32_t>(u);
            auto it = boundary.find(rev);
            if (it != boundary.end()) {
                boundary.erase(it);
            } else {
                boundary.insert(key);
            }
        };

        for (size_t fi = 0; fi < faces.size(); ++fi) {
            if (!visible[fi]) continue;
            const Face& f = faces[fi];
            addEdge(f.a, f.b);
            addEdge(f.b, f.c);
            addEdge(f.c, f.a);
        }

        std::vector<Face> nextFaces;
        nextFaces.reserve(faces.size() - boundary.size() + boundary.size());
        for (size_t fi = 0; fi < faces.size(); ++fi) {
            if (!visible[fi]) nextFaces.push_back(faces[fi]);
        }

        for (const std::uint64_t key : boundary) {
            const int u = static_cast<int>(key >> 32);
            const int v = static_cast<int>(static_cast<std::uint32_t>(key));
            Face nf = makeFaceOriented(P, u, v, idx, interior, epsLen);
            if (nf.a >= 0) nextFaces.push_back(nf);
        }

        faces.swap(nextFaces);
        if (faces.size() < 4) return false;
    }

    facesOut.swap(faces);
    return true;
}

struct PlaneKey {
    std::int64_t nx{}, ny{}, nz{}, nt{};
    bool operator==(const PlaneKey& o) const { return nx == o.nx && ny == o.ny && nz == o.nz && nt == o.nt; }
};

struct PlaneKeyHash {
    size_t operator()(const PlaneKey& k) const noexcept {
        auto mix = [](std::uint64_t x) {
            x ^= x >> 33;
            x *= 0xff51afd7ed558ccdULL;
            x ^= x >> 33;
            x *= 0xc4ceb9fe1a85ec53ULL;
            x ^= x >> 33;
            return x;
        };
        std::uint64_t h = 0;
        h ^= mix(static_cast<std::uint64_t>(k.nx) + 0x9e3779b97f4a7c15ULL);
        h ^= mix(static_cast<std::uint64_t>(k.ny) + 0xD1B54A32D192ED03ULL);
        h ^= mix(static_cast<std::uint64_t>(k.nz) + 0x94D049BB133111EBULL);
        h ^= mix(static_cast<std::uint64_t>(k.nt) + 0x2545F4914F6CDD1DULL);
        return static_cast<size_t>(h);
    }
};

static std::vector<Plane> facesToUniquePlanes(const std::vector<Vec3>& P, const std::vector<Face>& faces, const Vec3& interior) {
    // Compute scale for tolerances.
    Vec3 mn = P[0], mx = P[0];
    for (const auto& v : P) {
        mn.x = std::min(mn.x, v.x);
        mn.y = std::min(mn.y, v.y);
        mn.z = std::min(mn.z, v.z);
        mx.x = std::max(mx.x, v.x);
        mx.y = std::max(mx.y, v.y);
        mx.z = std::max(mx.z, v.z);
    }
    const double scale = std::max(1.0, norm(mx - mn));
    const double epsN = 1e-6;
    const double epsT = 1e-8 * scale;

    std::unordered_map<PlaneKey, size_t, PlaneKeyHash> map;
    std::vector<Plane> planes;
    planes.reserve(faces.size());

    for (const Face& f : faces) {
        if (f.a < 0) continue;
        const Vec3 A = P[f.a];
        Vec3 n0 = f.n;
        const double len = norm(n0);
        if (!(len > 0)) continue;
        Vec3 n = n0 / len;
        double t = dot(n, A);

        // Safety: ensure interior is inside.
        if (dot(n, interior) > t + 1e-10 * scale) {
            n = n * -1.0;
            t = -t;
        }
        const double tRel = t - dot(n, interior); // offset relative to an interior point (more stable)

        PlaneKey key;
        key.nx = static_cast<std::int64_t>(std::llround(n.x / epsN));
        key.ny = static_cast<std::int64_t>(std::llround(n.y / epsN));
        key.nz = static_cast<std::int64_t>(std::llround(n.z / epsN));
        key.nt = static_cast<std::int64_t>(std::llround(tRel / epsT));

        auto it = map.find(key);
        if (it == map.end()) {
            Plane p;
            p.n = n;
            p.t = t;
            p.m = 0.0; // filled later
            map.emplace(key, planes.size());
            planes.push_back(p);
        } else {
            // Keep the tighter (smaller t) plane for safety.
            Plane& existing = planes[it->second];
            if (t < existing.t) {
                existing.n = n;
                existing.t = t;
            }
        }
    }

    return planes;
}

static double polyhedronVolumeFromPlanes(const std::vector<Vec3>& P, const std::vector<Face>& faces, const Vec3& interior) {
    // Decompose into tetrahedra (interior, A, B, C) for each triangular face.
    long double vol = 0.0L;
    for (const Face& f : faces) {
        if (f.a < 0) continue;
        const Vec3 A = P[f.a] - interior;
        const Vec3 B = P[f.b] - interior;
        const Vec3 C = P[f.c] - interior;
        const long double v6 = static_cast<long double>(dot(A, cross(B, C)));
        vol += std::fabs(v6) / 6.0L;
    }
    return static_cast<double>(vol);
}

static bool insideShrunk(const std::vector<Plane>& planes, const Vec3& center, double safetyMargin) {
    for (const Plane& pl : planes) {
        // Need n·center <= t - m - safetyMargin.
        if (dot(pl.n, center) > (pl.t - pl.m - safetyMargin)) return false;
    }
    return true;
}

} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double e;
    if (!(std::cin >> V >> e)) {
        return 0;
    }
    if (V <= 0 || !(e > 0) || !std::isfinite(e)) {
        std::cout << 0 << "\n";
        return 0;
    }

    std::vector<Vec3> P;
    P.reserve(static_cast<size_t>(std::max(0, V)));
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        if (!(std::cin >> x >> y >> z)) {
            std::cout << 0 << "\n";
            return 0;
        }
        P.emplace_back(x, y, z);
    }
    if (P.size() < 4) {
        std::cout << 0 << "\n";
        return 0;
    }

    // Build convex hull faces.
    std::vector<Face> faces;
    Vec3 interior;
    if (!computeConvexHullFaces(P, faces, interior) || faces.size() < 4) {
        std::cout << 0 << "\n";
        return 0;
    }

    // Compute a scale for tolerances and output safety.
    Vec3 mn = P[0], mx = P[0];
    for (const auto& v : P) {
        mn.x = std::min(mn.x, v.x);
        mn.y = std::min(mn.y, v.y);
        mn.z = std::min(mn.z, v.z);
        mx.x = std::max(mx.x, v.x);
        mx.y = std::max(mx.y, v.y);
        mx.z = std::max(mx.z, v.z);
    }
    const double scale = std::max(1.0, norm(mx - mn));

    // Unique planes for faster membership checks.
    std::vector<Plane> planes = facesToUniquePlanes(P, faces, interior);
    if (planes.empty()) {
        std::cout << 0 << "\n";
        return 0;
    }

    // Reference regular tetrahedron (edge length 2) centered at origin:
    // (±1,0,-1/sqrt2), (0,±1, +1/sqrt2). Scale by e/2.
    const double s = e / 2.0;
    const double invSqrt2 = 1.0 / std::sqrt(2.0);
    const std::vector<Vec3> tet = {
        Vec3{+s, 0.0, -s * invSqrt2},
        Vec3{-s, 0.0, -s * invSqrt2},
        Vec3{0.0, +s, +s * invSqrt2},
        Vec3{0.0, -s, +s * invSqrt2},
    };

    // Precompute per-plane support margin m = max_i n·v_i
    for (Plane& pl : planes) {
        double m = -std::numeric_limits<double>::infinity();
        for (const Vec3& v : tet) m = std::max(m, dot(pl.n, v));
        pl.m = m;
    }

    // Estimate container volume and tetrahedron volume (for adaptive behavior).
    const double containerVol = std::max(0.0, polyhedronVolumeFromPlanes(P, faces, interior));
    const double tetVol = (e * e * e) / (6.0 * std::sqrt(2.0));
    const double upperBound = (tetVol > 0) ? (containerVol / tetVol) : 0.0;

    // Packing parameters (circumsphere + FCC lattice).
    const double R = e * std::sqrt(3.0 / 8.0);      // tetrahedron circumsphere radius
    const double slack = 1.0 + 1e-9;                // avoid tiny FP overlaps on tight lattices
    const double A = 2.0 * std::sqrt(2.0) * R * slack; // FCC cube side (nearest-neighbor distance >= 2R)

    // Safety margin for containment checks (length units).
    const double safetyMargin = 5e-10 * scale + 1e-9 * e;

    // Cap output to keep runtime and output size reasonable on huge instances.
    std::size_t maxK = 250000;
    if (upperBound > 0 && upperBound < static_cast<double>(maxK)) {
        maxK = static_cast<std::size_t>(std::max(0.0, std::floor(upperBound + 5.0)));
    }
    maxK = std::max<std::size_t>(maxK, 0);

    // Candidate lattice offsets for one FCC cell.
    const Vec3 offsets[4] = {
        Vec3{0, 0, 0},
        Vec3{0, A * 0.5, A * 0.5},
        Vec3{A * 0.5, 0, A * 0.5},
        Vec3{A * 0.5, A * 0.5, 0},
    };

    // For small/medium instances, try a few lattice phase shifts and pick the best.
    // For huge instances, use a single shift and stream placements until maxK/time.
    const bool tryMultipleShifts = (upperBound > 0 && upperBound <= 20000.0 && planes.size() <= 3000);

    std::vector<Vec3> bestCenters;
    bestCenters.reserve(std::min<std::size_t>(maxK, 20000));

    auto packWithShift = [&](const Vec3& shift, std::size_t cap, double timeLimitSec) -> std::vector<Vec3> {
        std::vector<Vec3> centers;
        centers.reserve(std::min<std::size_t>(cap, 50000));

        // Lattice index ranges based on AABB; expanded by one cell.
        const double minx = mn.x - A;
        const double miny = mn.y - A;
        const double minz = mn.z - A;
        const double maxx = mx.x + A;
        const double maxy = mx.y + A;
        const double maxz = mx.z + A;

        const std::int64_t ix0 = static_cast<std::int64_t>(std::floor((minx - shift.x) / A));
        const std::int64_t iy0 = static_cast<std::int64_t>(std::floor((miny - shift.y) / A));
        const std::int64_t iz0 = static_cast<std::int64_t>(std::floor((minz - shift.z) / A));
        const std::int64_t ix1 = static_cast<std::int64_t>(std::ceil((maxx - shift.x) / A));
        const std::int64_t iy1 = static_cast<std::int64_t>(std::ceil((maxy - shift.y) / A));
        const std::int64_t iz1 = static_cast<std::int64_t>(std::ceil((maxz - shift.z) / A));

        const auto tStart = std::chrono::high_resolution_clock::now();
        std::uint64_t iterCount = 0;

        for (std::int64_t ix = ix0; ix <= ix1; ++ix) {
            for (std::int64_t iy = iy0; iy <= iy1; ++iy) {
                for (std::int64_t iz = iz0; iz <= iz1; ++iz) {
                    const Vec3 base{shift.x + ix * A, shift.y + iy * A, shift.z + iz * A};
                    for (const Vec3& off : offsets) {
                        const Vec3 c = base + off;
                        // quick AABB filter
                        if (c.x < mn.x || c.x > mx.x || c.y < mn.y || c.y > mx.y || c.z < mn.z || c.z > mx.z) {
                            continue;
                        }
                        if (!insideShrunk(planes, c, safetyMargin)) continue;
                        centers.push_back(c);
                        if (centers.size() >= cap) return centers;

                        if ((++iterCount & 8191ULL) == 0ULL) {
                            const auto now = std::chrono::high_resolution_clock::now();
                            const double sec =
                                std::chrono::duration<double>(now - tStart).count();
                            if (sec > timeLimitSec) return centers;
                        }
                    }
                }
            }
        }
        return centers;
    };

    if (tryMultipleShifts) {
        // Candidate phase shifts (in units of A/2).
        const Vec3 shifts[] = {
            Vec3{0, 0, 0},
            Vec3{A * 0.5, 0, 0},
            Vec3{0, A * 0.5, 0},
            Vec3{0, 0, A * 0.5},
            Vec3{A * 0.5, A * 0.5, 0},
            Vec3{A * 0.5, 0, A * 0.5},
            Vec3{0, A * 0.5, A * 0.5},
            Vec3{A * 0.5, A * 0.5, A * 0.5},
        };
        std::size_t best = 0;
        for (const Vec3& sh : shifts) {
            std::vector<Vec3> got = packWithShift(sh, maxK, 2.5); // short per-shift budget
            if (got.size() > best) {
                best = got.size();
                bestCenters.swap(got);
            }
        }
    } else {
        bestCenters = packWithShift(Vec3{0, 0, 0}, maxK, 28.0);
    }

    // Output.
    std::cout << bestCenters.size() << "\n";
    std::cout << std::setprecision(17);
    for (const Vec3& c : bestCenters) {
        std::cout << c.x << " " << c.y << " " << c.z << " 0 0 0\n";
    }

    return 0;
}
