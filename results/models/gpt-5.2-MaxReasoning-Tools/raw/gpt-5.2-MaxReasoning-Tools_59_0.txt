"use strict";

// Benchmark solution: multi-source BFS partition + greedy prize collection on BFS trees.
// Reads one JSON document from stdin and prints one JSON document to stdout.

const fs = require("fs");

let data;
try {
  let input = fs.readFileSync(0, "utf8");
  if (input.length > 0 && input.charCodeAt(0) === 0xfeff) input = input.slice(1);
  data = JSON.parse(input);
} catch {
  process.stdout.write('{"paths":[]}');
  process.exit(0);
}

const probability = Array.isArray(data?.probability) ? data.probability : [];
const drones = Array.isArray(data?.drones) ? data.drones : [];
const maxStepsPerDrone = Number.isFinite(data?.maxStepsPerDrone)
  ? Math.max(0, data.maxStepsPerDrone | 0)
  : 0;

const H = probability.length | 0;
const W = H > 0 && Array.isArray(probability[0]) ? probability[0].length | 0 : 0;
const D = drones.length | 0;

if (H <= 0 || W <= 0 || D <= 0 || maxStepsPerDrone < 0) {
  process.stdout.write(JSON.stringify({ paths: new Array(D).fill([]) }));
  process.exit(0);
}

const N = H * W;

const blocked = new Uint8Array(N);
const cellValue = new Float64Array(N);

for (let y = 0; y < H; y++) {
  const row = probability[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    const idx = y * W + x;
    if (v === -1 || v < 0) {
      blocked[idx] = 1;
      cellValue[idx] = 0;
    } else {
      blocked[idx] = 0;
      cellValue[idx] = v > 0 ? v : 0;
    }
  }
}

const startIdx = new Int32Array(D);
for (let i = 0; i < D; i++) {
  const dx = drones[i]?.x | 0;
  const dy = drones[i]?.y | 0;
  if (dx < 0 || dx >= W || dy < 0 || dy >= H) {
    startIdx[i] = -1;
    continue;
  }
  const s = dy * W + dx;
  startIdx[i] = s;
}

// Multi-source BFS: owner/dist/parent give a BFS forest rooted at each (unique) start cell.
const owner = new Int32Array(N);
const depth = new Int32Array(N);
const parent = new Int32Array(N);
owner.fill(-1);
depth.fill(-1);
parent.fill(-1);

const queue = new Int32Array(N);
let qh = 0;
let qt = 0;

for (let i = 0; i < D; i++) {
  const s = startIdx[i];
  if (s < 0 || s >= N) continue;
  if (blocked[s]) continue;
  if (depth[s] !== -1) continue; // duplicate start cell: first drone claims the region
  owner[s] = i;
  depth[s] = 0;
  parent[s] = -1;
  queue[qt++] = s;
}

while (qh < qt) {
  const idx = queue[qh++];
  const d = depth[idx] + 1;
  const o = owner[idx];
  const x = idx % W;
  const y = (idx / W) | 0;

  // Left
  if (x > 0) {
    const ni = idx - 1;
    if (depth[ni] === -1 && blocked[ni] === 0) {
      depth[ni] = d;
      owner[ni] = o;
      parent[ni] = idx;
      queue[qt++] = ni;
    }
  }
  // Right
  if (x + 1 < W) {
    const ni = idx + 1;
    if (depth[ni] === -1 && blocked[ni] === 0) {
      depth[ni] = d;
      owner[ni] = o;
      parent[ni] = idx;
      queue[qt++] = ni;
    }
  }
  // Up
  if (y > 0) {
    const ni = idx - W;
    if (depth[ni] === -1 && blocked[ni] === 0) {
      depth[ni] = d;
      owner[ni] = o;
      parent[ni] = idx;
      queue[qt++] = ni;
    }
  }
  // Down
  if (y + 1 < H) {
    const ni = idx + W;
    if (depth[ni] === -1 && blocked[ni] === 0) {
      depth[ni] = d;
      owner[ni] = o;
      parent[ni] = idx;
      queue[qt++] = ni;
    }
  }
}

// Map each drone to a region (owner id) based on the cell it starts on (robust to duplicates).
const regionOfDrone = new Int32Array(D);
for (let i = 0; i < D; i++) {
  const s = startIdx[i];
  const r = s >= 0 && s < N ? owner[s] : -1;
  regionOfDrone[i] = r >= 0 ? r : i;
}

class MinHeap {
  constructor(limit) {
    this.limit = limit | 0;
    this.val = [];
    this.idx = [];
  }
  push(value, index) {
    if (this.limit <= 0) return;
    const n = this.val.length;
    if (n < this.limit) {
      this.val.push(value);
      this.idx.push(index);
      this._siftUp(n);
      return;
    }
    if (value <= this.val[0]) return;
    this.val[0] = value;
    this.idx[0] = index;
    this._siftDown(0);
  }
  _siftUp(i) {
    const val = this.val;
    const idx = this.idx;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (val[p] <= val[i]) break;
      let t = val[p];
      val[p] = val[i];
      val[i] = t;
      t = idx[p];
      idx[p] = idx[i];
      idx[i] = t;
      i = p;
    }
  }
  _siftDown(i) {
    const val = this.val;
    const idx = this.idx;
    const n = val.length;
    while (true) {
      const l = i * 2 + 1;
      if (l >= n) break;
      const r = l + 1;
      let s = l;
      if (r < n && val[r] < val[l]) s = r;
      if (val[i] <= val[s]) break;
      let t = val[i];
      val[i] = val[s];
      val[s] = t;
      t = idx[i];
      idx[i] = idx[s];
      idx[s] = t;
      i = s;
    }
  }
}

function heapToSortedCandidates(heap) {
  const n = heap.val.length;
  if (n === 0) return { idx: [], val: [] };
  const order = new Array(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => heap.val[b] - heap.val[a]);
  const idx = new Array(n);
  const val = new Array(n);
  for (let i = 0; i < n; i++) {
    const j = order[i];
    idx[i] = heap.idx[j];
    val[i] = heap.val[j];
  }
  return { idx, val };
}

const K = Math.max(64, Math.min(3000, maxStepsPerDrone * 25));
const heaps = new Array(D);
for (let i = 0; i < D; i++) heaps[i] = new MinHeap(K);

for (let idx = 0; idx < N; idx++) {
  const v = cellValue[idx];
  if (v <= 0) continue;
  const o = owner[idx];
  if (o >= 0) heaps[o].push(v, idx);
}

const candidatesByRegionIdx = new Array(D);
const candidatesByRegionVal = new Array(D);
for (let i = 0; i < D; i++) {
  const { idx, val } = heapToSortedCandidates(heaps[i]);
  candidatesByRegionIdx[i] = idx;
  candidatesByRegionVal[i] = val;
}

// Timestamp marking for fast distance-to-many-targets queries in a rooted tree.
const mark = new Int32Array(N);
const markDist = new Int32Array(N);
let markToken = 1;

function prepareMarks(curr) {
  markToken++;
  if (markToken === 0x7fffffff) {
    mark.fill(0);
    markToken = 1;
  }
  let node = curr;
  let d = 0;
  while (node !== -1 && mark[node] !== markToken) {
    mark[node] = markToken;
    markDist[node] = d;
    node = parent[node];
    d++;
  }
}

function distUsingMarks(target) {
  let node = target;
  let d = 0;
  while (node !== -1 && mark[node] !== markToken) {
    node = parent[node];
    d++;
  }
  if (node === -1) return 1e9;
  return d + markDist[node];
}

function buildPathUsingMarks(curr, target) {
  if (curr === target) return [curr];
  const upTarget = [];
  let node = target;
  while (node !== -1 && mark[node] !== markToken) {
    upTarget.push(node);
    node = parent[node];
  }
  if (node === -1) return [curr];
  const lca = node;
  const path = [];
  let c = curr;
  while (c !== lca && c !== -1) {
    path.push(c);
    c = parent[c];
  }
  path.push(lca);
  for (let i = upTarget.length - 1; i >= 0; i--) path.push(upTarget[i]);
  return path;
}

function bestLocalStep(curr, prev, regionId, visitedPositive) {
  const x = curr % W;
  const y = (curr / W) | 0;

  let best = curr;
  let bestScore = -1e30;

  function consider(ni) {
    if (ni < 0 || ni >= N) return;
    if (blocked[ni]) return;
    if (owner[ni] !== regionId) return;

    let score = visitedPositive[ni] ? 0 : cellValue[ni];
    // Light lookahead: sum of unvisited neighbor values.
    const nx = ni % W;
    const ny = (ni / W) | 0;
    let nb = 0;
    if (nx > 0) {
      const t = ni - 1;
      if (owner[t] === regionId && !visitedPositive[t]) nb += cellValue[t];
    }
    if (nx + 1 < W) {
      const t = ni + 1;
      if (owner[t] === regionId && !visitedPositive[t]) nb += cellValue[t];
    }
    if (ny > 0) {
      const t = ni - W;
      if (owner[t] === regionId && !visitedPositive[t]) nb += cellValue[t];
    }
    if (ny + 1 < H) {
      const t = ni + W;
      if (owner[t] === regionId && !visitedPositive[t]) nb += cellValue[t];
    }
    score += 0.08 * nb;

    // Encourage exploration slightly outward.
    score += 0.0002 * depth[ni];

    // Penalties to avoid oscillation / idling.
    if (ni === prev) score -= 0.02;
    if (ni === curr) score -= 0.005;

    if (score > bestScore) {
      bestScore = score;
      best = ni;
    }
  }

  // Prefer movement directions deterministically.
  if (x + 1 < W) consider(curr + 1);
  if (x > 0) consider(curr - 1);
  if (y + 1 < H) consider(curr + W);
  if (y > 0) consider(curr - W);
  consider(curr); // stay

  return best;
}

function planDronePathString(droneIndex, visitedPositive) {
  const s = startIdx[droneIndex];
  const sx = drones[droneIndex]?.x | 0;
  const sy = drones[droneIndex]?.y | 0;

  // If invalid start, output a degenerate path staying at (0,0).
  if (s < 0 || s >= N || blocked[s]) {
    const parts = [];
    parts.push("[");
    for (let t = 0; t <= maxStepsPerDrone; t++) {
      if (t) parts.push(",");
      parts.push("[0,0]");
    }
    parts.push("]");
    return parts.join("");
  }

  const regionId = regionOfDrone[droneIndex];
  const candIdx = candidatesByRegionIdx[regionId] || [];
  const candVal = candidatesByRegionVal[regionId] || [];

  let curr = s;
  let prev = s;

  const parts = [];
  parts.push("[");
  parts.push(`[${sx},${sy}]`);
  if (cellValue[curr] > 0) visitedPositive[curr] = 1;

  let stepsDone = 0;
  const evalLimitBase = Math.max(64, Math.min(256, (maxStepsPerDrone / 2) | 0));
  const scanCap = 1024;

  while (stepsDone < maxStepsPerDrone) {
    const remaining = maxStepsPerDrone - stepsDone;

    let bestTarget = -1;
    let bestScore = -1;

    // Only chase targets if we have some remaining budget to get there.
    if (remaining >= 2 && candIdx.length > 0) {
      prepareMarks(curr);

      let scanned = 0;
      let evaluated = 0;
      const want = Math.min(evalLimitBase, candIdx.length);

      for (let j = 0; j < candIdx.length && scanned < scanCap && evaluated < want; j++) {
        const t = candIdx[j];
        scanned++;
        if (visitedPositive[t]) continue;
        evaluated++;
        const dist = distUsingMarks(t);
        if (dist <= 0 || dist > remaining) continue;
        const score = candVal[j] / (dist + 1);
        if (score > bestScore) {
          bestScore = score;
          bestTarget = t;
        }
      }
    }

    if (bestTarget >= 0 && bestTarget !== curr) {
      // Reuse the marks for current node to build the path.
      const segment = buildPathUsingMarks(curr, bestTarget);
      for (let k = 1; k < segment.length && stepsDone < maxStepsPerDrone; k++) {
        prev = curr;
        curr = segment[k];
        stepsDone++;
        parts.push(`,[${curr % W},${(curr / W) | 0}]`);
        if (cellValue[curr] > 0) visitedPositive[curr] = 1;
      }
      continue;
    }

    // Fallback: local greedy step within region.
    const next = bestLocalStep(curr, prev, regionId, visitedPositive);
    prev = curr;
    curr = next;
    stepsDone++;
    parts.push(`,[${curr % W},${(curr / W) | 0}]`);
    if (cellValue[curr] > 0) visitedPositive[curr] = 1;
  }

  parts.push("]");
  return parts.join("");
}

// Output streaming: build each path string separately to limit peak memory.
const visitedPositive = new Uint8Array(N);
process.stdout.write('{"paths":[');
for (let i = 0; i < D; i++) {
  if (i) process.stdout.write(",");
  process.stdout.write(planDronePathString(i, visitedPositive));
}
process.stdout.write("]}");
