use std::collections::HashMap;
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let target = match input.next_i64() {
        Some(v) => v,
        None => return,
    };

    let mut nums: Vec<(i64, usize)> = Vec::with_capacity(n);
    nums.reserve(n.saturating_sub(nums.capacity()));

    let mut gcd_all: u64 = 0;
    let mut sum_pos: i128 = 0;
    let mut sum_neg: i128 = 0;

    let mut direct_hit: Option<usize> = None;
    for idx in 0..n {
        let v = match input.next_i64() {
            Some(x) => x,
            None => break,
        };
        if direct_hit.is_none() && v == target {
            direct_hit = Some(idx);
        }
        gcd_all = gcd_u64(gcd_all, v.unsigned_abs());
        if v > 0 {
            sum_pos += v as i128;
        } else if v < 0 {
            sum_neg += v as i128;
        }
        nums.push((v, idx));
    }

    if target == 0 {
        print_yes(&[]);
        return;
    }

    if let Some(i) = direct_hit {
        print_yes(&[i]);
        return;
    }

    // If all numbers are zero, only sum 0 is possible (already handled).
    if gcd_all != 0 && (target.unsigned_abs() % gcd_all) != 0 {
        print_no();
        return;
    }

    let target_i128 = target as i128;
    if target_i128 < sum_neg || target_i128 > sum_pos {
        print_no();
        return;
    }

    if nums.len() <= 42 {
        if let Some(sol) = subset_sum_meet_in_middle(&nums, target) {
            print_yes(&sol);
        } else {
            print_no();
        }
        return;
    }

    let all_nonneg = nums.iter().all(|&(v, _)| v >= 0);
    if all_nonneg && target > 0 {
        let t_usize = match usize::try_from(target) {
            Ok(v) => v,
            Err(_) => {
                // target doesn't fit usize => can't do pseudo-polynomial DP.
                // Fall through to small-k heuristics.
                0
            }
        };
        if t_usize > 0 {
            if let Some(sol) = subset_sum_nonneg_bitset(t_usize, &nums) {
                print_yes(&sol);
                return;
            }
        }
    }

    // If signed range is small, do offset DP (exact).
    let range = sum_pos - sum_neg;
    const RANGE_LIMIT: i128 = 2_000_000;
    const RANGE_N_LIMIT: usize = 300_000;
    if range >= 0 && range <= RANGE_LIMIT && nums.len() <= RANGE_N_LIMIT {
        if let Some(sol) = subset_sum_signed_offset(&nums, target, sum_neg) {
            print_yes(&sol);
            return;
        }
    }

    // Exact 2-sum / 3-sum (useful on huge n if a small solution exists).
    if let Some(sol) = subset_sum_two_sum(&nums, target) {
        print_yes(&sol);
        return;
    }

    const THREE_SUM_LIMIT: usize = 25_000;
    if nums.len() <= THREE_SUM_LIMIT {
        let mut sorted = nums.clone();
        sorted.sort_unstable_by(|a, b| a.0.cmp(&b.0));
        if let Some(sol) = subset_sum_three_sum_sorted(&sorted, target) {
            print_yes(&sol);
            return;
        }
    }

    // Heuristic: try meet-in-the-middle on a small "important" subset (still exact for that subset).
    let picked = pick_top_k_by_abs(&nums, 42);
    if picked.len() >= 2 {
        if let Some(sol) = subset_sum_meet_in_middle(&picked, target) {
            print_yes(&sol);
            return;
        }
    }

    print_no();
}

fn print_yes(indices: &[usize]) {
    let mut out = String::new();
    out.push_str("YES\n[");
    if !indices.is_empty() {
        for (i, idx) in indices.iter().enumerate() {
            if i != 0 {
                out.push(' ');
            }
            out.push_str(&idx.to_string());
        }
    }
    out.push_str("]\n");
    print!("{out}");
}

fn print_no() {
    print!("NO\n");
}

struct FastInput {
    buf: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        let _ = io::stdin().read_to_end(&mut buf);
        Self { buf, pos: 0 }
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.buf.len() && is_space_like(self.buf[self.pos]) {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.buf[self.pos] == b'+' {
            self.pos += 1;
        }
        while self.pos < self.buf.len() && is_space_like(self.buf[self.pos]) {
            self.pos += 1;
        }
        let mut val: i64 = 0;
        let mut any = false;
        while self.pos < self.buf.len() {
            let c = self.buf[self.pos];
            if c == 0 {
                self.pos += 1;
                continue;
            }
            if !c.is_ascii_digit() {
                break;
            }
            any = true;
            val = val.saturating_mul(10).saturating_add((c - b'0') as i64);
            self.pos += 1;
        }
        if !any {
            return None;
        }
        Some(val.saturating_mul(sign))
    }

    fn next_usize(&mut self) -> Option<usize> {
        let v = self.next_i64()?;
        usize::try_from(v).ok()
    }
}

fn is_space_like(b: u8) -> bool {
    b == 0
        || b.is_ascii_whitespace()
        || b == 0xEF
        || b == 0xBB
        || b == 0xBF
        || b == 0xFE
        || b == 0xFF
}

fn gcd_u64(mut a: u64, mut b: u64) -> u64 {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

#[derive(Default)]
struct FxHasher {
    hash: u64,
}

impl Hasher for FxHasher {
    fn finish(&self) -> u64 {
        self.hash
    }

    fn write(&mut self, bytes: &[u8]) {
        let mut h = self.hash;
        for &b in bytes {
            h = h.rotate_left(5) ^ (b as u64);
            h = h.wrapping_mul(0x517cc1b727220a95);
        }
        self.hash = h;
    }

    fn write_i64(&mut self, i: i64) {
        self.write_u64(i as u64);
    }

    fn write_u64(&mut self, i: u64) {
        let mut x = i;
        x ^= x >> 33;
        x = x.wrapping_mul(0xff51afd7ed558ccd);
        x ^= x >> 33;
        x = x.wrapping_mul(0xc4ceb9fe1a85ec53);
        x ^= x >> 33;
        self.hash = x;
    }
}

type FxBuildHasher = BuildHasherDefault<FxHasher>;

fn subset_sum_two_sum(nums: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let mut map: HashMap<i64, usize, FxBuildHasher> = HashMap::default();
    map.reserve(nums.len().min(5_000_000));
    for &(v, idx) in nums {
        let need = match target.checked_sub(v) {
            Some(x) => x,
            None => continue,
        };
        if let Some(&j) = map.get(&need) {
            let mut res = vec![j, idx];
            res.sort_unstable();
            return Some(res);
        }
        map.entry(v).or_insert(idx);
    }
    None
}

fn subset_sum_three_sum_sorted(sorted: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    if sorted.len() < 3 {
        return None;
    }
    let target_i128 = target as i128;
    for i in 0..(sorted.len() - 2) {
        let a = sorted[i].0;
        let mut l = i + 1;
        let mut r = sorted.len() - 1;
        while l < r {
            let s = (a as i128) + (sorted[l].0 as i128) + (sorted[r].0 as i128);
            if s == target_i128 {
                let mut res = vec![sorted[i].1, sorted[l].1, sorted[r].1];
                res.sort_unstable();
                return Some(res);
            }
            if s < target_i128 {
                l += 1;
            } else {
                r -= 1;
            }
        }
    }
    None
}

fn pick_top_k_by_abs(nums: &[(i64, usize)], k: usize) -> Vec<(i64, usize)> {
    let mut best: Vec<(u64, i64, usize)> = Vec::with_capacity(k);
    for &(v, idx) in nums {
        let a = v.unsigned_abs();
        if best.len() < k {
            best.push((a, v, idx));
            continue;
        }
        let mut min_pos = 0usize;
        let mut min_abs = best[0].0;
        for (p, &(ab, _, _)) in best.iter().enumerate().skip(1) {
            if ab < min_abs {
                min_abs = ab;
                min_pos = p;
            }
        }
        if a > min_abs {
            best[min_pos] = (a, v, idx);
        }
    }
    best.into_iter().map(|(_, v, idx)| (v, idx)).collect()
}

fn subset_sum_meet_in_middle(nums: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return None;
    }
    if n == 1 {
        return if nums[0].0 == target {
            Some(vec![nums[0].1])
        } else {
            None
        };
    }
    let target_i128 = target as i128;
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    let left_sums = all_subset_sums(left);
    let mut right_sums = all_subset_sums(right);
    right_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    for (ls, lmask) in left_sums {
        let need = target_i128 - ls;
        if let Ok(pos) = right_sums.binary_search_by(|(rs, _)| rs.cmp(&need)) {
            let rmask = right_sums[pos].1;
            let mut res = Vec::new();
            for i in 0..left.len() {
                if (lmask >> i) & 1 != 0 {
                    res.push(left[i].1);
                }
            }
            for i in 0..right.len() {
                if (rmask >> i) & 1 != 0 {
                    res.push(right[i].1);
                }
            }
            res.sort_unstable();
            return Some(res);
        }
    }
    None
}

fn all_subset_sums(slice: &[(i64, usize)]) -> Vec<(i128, u64)> {
    let n = slice.len();
    let total = 1usize << n;
    let mut sums: Vec<(i128, u64)> = Vec::with_capacity(total);
    sums.push((0, 0));
    for i in 0..n {
        let (v, _) = slice[i];
        let current_len = sums.len();
        for j in 0..current_len {
            let (s, m) = sums[j];
            sums.push((s + (v as i128), m | (1u64 << i)));
        }
    }
    sums
}

#[derive(Clone, Copy)]
struct Piece {
    weight: usize,
    group: usize,
    start_in_group: usize,
    len: usize,
}

fn subset_sum_nonneg_bitset(target: usize, nums: &[(i64, usize)]) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    const TARGET_LIMIT: usize = 10_000_000;
    if target > TARGET_LIMIT {
        return None;
    }

    let mut items: Vec<(usize, usize)> = Vec::new();
    items.reserve(nums.len().min(5_000_000));
    for &(v, idx) in nums {
        if v <= 0 {
            continue;
        }
        let w = match usize::try_from(v) {
            Ok(x) => x,
            Err(_) => continue,
        };
        if w <= target {
            items.push((w, idx));
        }
    }
    if items.is_empty() {
        return None;
    }
    items.sort_unstable_by_key(|x| x.0);

    // Group indices by weight into one flat buffer.
    let mut group_weights: Vec<usize> = Vec::new();
    let mut group_starts: Vec<usize> = Vec::new();
    let mut group_lens: Vec<usize> = Vec::new();
    let mut group_indices: Vec<usize> = Vec::with_capacity(items.len());

    let mut i = 0usize;
    while i < items.len() {
        let w = items[i].0;
        group_weights.push(w);
        group_starts.push(group_indices.len());
        while i < items.len() && items[i].0 == w {
            group_indices.push(items[i].1);
            i += 1;
        }
        group_lens.push(group_indices.len() - *group_starts.last().unwrap());
    }

    let mut pieces: Vec<Piece> = Vec::new();
    pieces.reserve(group_weights.len() * 4);
    for g in 0..group_weights.len() {
        let w = group_weights[g];
        let cnt = group_lens[g];
        let cap = (target / w).min(cnt);
        if cap == 0 {
            continue;
        }
        let mut used = 0usize;
        let mut remaining = cap;
        let mut p = 1usize;
        while remaining > 0 {
            let take = p.min(remaining);
            pieces.push(Piece {
                weight: w * take,
                group: g,
                start_in_group: used,
                len: take,
            });
            used += take;
            remaining -= take;
            p = p.saturating_mul(2);
        }
    }
    if pieces.is_empty() {
        return None;
    }

    let words = (target / 64) + 1;
    let est_ops = (pieces.len() as u64) * (words as u64);
    const MAX_WORD_OPS: u64 = 650_000_000;
    if est_ops > MAX_WORD_OPS {
        return None;
    }

    let mut dp = vec![0u64; words];
    dp[0] = 1u64;
    let mut shifted = vec![0u64; words];
    let mut choice = vec![0i32; target + 1];

    for (pid, piece) in pieces.iter().enumerate() {
        shift_left_into(&dp, piece.weight, &mut shifted);
        for wi in 0..words {
            let old = dp[wi];
            let sh = shifted[wi];
            let new_bits = sh & !old;
            if new_bits != 0 {
                let base = wi * 64;
                let mut x = new_bits;
                while x != 0 {
                    let tz = x.trailing_zeros() as usize;
                    let bit = base + tz;
                    if bit <= target && choice[bit] == 0 {
                        choice[bit] = (pid as i32) + 1;
                    }
                    x &= x - 1;
                }
            }
            dp[wi] = old | sh;
        }
        if ((dp[target / 64] >> (target % 64)) & 1) != 0 {
            let mut res: Vec<usize> = Vec::new();
            let mut cur = target;
            while cur != 0 {
                let cid = choice[cur];
                if cid <= 0 {
                    break;
                }
                let pidx = (cid - 1) as usize;
                let pc = pieces[pidx];
                let start = group_starts[pc.group] + pc.start_in_group;
                let end = start + pc.len;
                res.extend_from_slice(&group_indices[start..end]);
                cur = cur.saturating_sub(pc.weight);
            }
            if cur == 0 {
                res.sort_unstable();
                return Some(res);
            }
        }
    }

    None
}

fn shift_left_into(src: &[u64], shift: usize, dst: &mut [u64]) {
    dst.fill(0);
    if shift == 0 {
        dst.copy_from_slice(src);
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    let n = src.len();
    for i in 0..n {
        let v = src[i];
        if v == 0 {
            continue;
        }
        let di = i + word_shift;
        if di >= dst.len() {
            continue;
        }
        dst[di] |= v << bit_shift;
        if bit_shift != 0 && di + 1 < dst.len() {
            dst[di + 1] |= v >> (64 - bit_shift);
        }
    }
}

fn subset_sum_signed_offset(
    nums: &[(i64, usize)],
    target: i64,
    min_sum: i128,
) -> Option<Vec<usize>> {
    let mut max_sum: i128 = 0;
    for &(v, _) in nums {
        if v > 0 {
            max_sum += v as i128;
        }
    }
    let range = max_sum - min_sum;
    if range < 0 {
        return None;
    }
    let range_usize = usize::try_from(range).ok()?;
    let offset = usize::try_from((-min_sum) as i128).ok()?;

    let target_pos_i128 = (target as i128) - min_sum;
    if target_pos_i128 < 0 || target_pos_i128 > range {
        return None;
    }
    let target_pos = target_pos_i128 as usize;

    let words = (range_usize / 64) + 1;
    let mut dp = vec![0u64; words];
    let mut shifted = vec![0u64; words];
    let mut choice = vec![0i32; range_usize + 1];
    dp[offset / 64] |= 1u64 << (offset % 64);

    for (eid, &(v, _idx)) in nums.iter().enumerate() {
        if v == 0 {
            continue;
        }
        if v > 0 {
            let w = usize::try_from(v).ok()?;
            shift_left_into(&dp, w, &mut shifted);
            for wi in 0..words {
                let old = dp[wi];
                let sh = shifted[wi];
                let new_bits = sh & !old;
                if new_bits != 0 {
                    let base = wi * 64;
                    let mut x = new_bits;
                    while x != 0 {
                        let tz = x.trailing_zeros() as usize;
                        let bit = base + tz;
                        if bit <= range_usize && choice[bit] == 0 {
                            choice[bit] = (eid as i32) + 1;
                        }
                        x &= x - 1;
                    }
                }
                dp[wi] = old | sh;
            }
        } else {
            let w = usize::try_from(v.unsigned_abs()).ok()?;
            shift_right_into(&dp, w, &mut shifted);
            for wi in 0..words {
                let old = dp[wi];
                let sh = shifted[wi];
                let new_bits = sh & !old;
                if new_bits != 0 {
                    let base = wi * 64;
                    let mut x = new_bits;
                    while x != 0 {
                        let tz = x.trailing_zeros() as usize;
                        let bit = base + tz;
                        if bit <= range_usize && choice[bit] == 0 {
                            choice[bit] = (eid as i32) + 1;
                        }
                        x &= x - 1;
                    }
                }
                dp[wi] = old | sh;
            }
        }

        if ((dp[target_pos / 64] >> (target_pos % 64)) & 1) != 0 {
            let mut res = Vec::new();
            let mut cur = target_pos;
            while cur != offset {
                let cid = choice[cur];
                if cid <= 0 {
                    break;
                }
                let ei = (cid - 1) as usize;
                let (val, orig_idx) = nums[ei];
                res.push(orig_idx);
                let prev_sum = (cur as i128) - (val as i128);
                if prev_sum < 0 || prev_sum > range as i128 {
                    break;
                }
                cur = prev_sum as usize;
            }
            if cur == offset {
                res.sort_unstable();
                return Some(res);
            }
        }
    }

    None
}

fn shift_right_into(src: &[u64], shift: usize, dst: &mut [u64]) {
    dst.fill(0);
    if shift == 0 {
        dst.copy_from_slice(src);
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for i in 0..src.len() {
        let v = src[i];
        if v == 0 {
            continue;
        }
        if i < word_shift {
            continue;
        }
        let di = i - word_shift;
        dst[di] |= v >> bit_shift;
        if bit_shift != 0 && di >= 1 {
            dst[di - 1] |= v << (64 - bit_shift);
        }
    }
}
