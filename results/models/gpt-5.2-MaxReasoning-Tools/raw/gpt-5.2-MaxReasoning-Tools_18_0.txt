// Minimum Edge Cut (Global Min-Cut) solver
// Target: MSVC /std:c++20
//
// Strategy (adaptive):
//  1) Small N: exact Stoer–Wagner on adjacency matrix (counts only; handles huge M via aggregation).
//  2) Medium N,M: exact Stoer–Wagner on sparse adjacency maps (keeps edges in memory).
//  3) Very large graphs: disk-friendly fallback cut isolating a minimum-degree vertex (always valid).
//
// Notes:
//  - Graph is undirected; parallel edges are allowed and handled (as weights / multiplicities).
//  - Self-loops are ignored (never affect cuts).

#include <cstdio>
#include <cstdint>
#include <cstdlib>
#include <cstring>

#include <algorithm>
#include <cmath>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <type_traits>
#include <unordered_map>
#include <utility>
#include <vector>

namespace {

struct FastScanner {
  static constexpr size_t kBufSize = 1u << 20;
  char buf[kBufSize];
  size_t idx = 0;
  size_t len = 0;

  inline char readChar() {
    if (idx >= len) {
      len = std::fread(buf, 1, kBufSize, stdin);
      idx = 0;
      if (len == 0) return 0;
    }
    return buf[idx++];
  }

  template <class T>
  bool readUInt(T& out) {
    static_assert(std::is_unsigned_v<T>, "readUInt requires unsigned type");
    char c = readChar();
    while (c != 0 && static_cast<unsigned char>(c) <= ' ') c = readChar();
    if (c == 0) return false;
    T val = 0;
    while (c >= '0' && c <= '9') {
      val = static_cast<T>(val * 10 + static_cast<T>(c - '0'));
      c = readChar();
    }
    out = val;
    return true;
  }
};

struct FastOutput {
  static constexpr size_t kFlushThreshold = 1u << 20;
  std::string out;

  FastOutput() { out.reserve(kFlushThreshold); }

  ~FastOutput() { flush(); }

  inline void flush() {
    if (!out.empty()) {
      std::fwrite(out.data(), 1, out.size(), stdout);
      out.clear();
    }
  }

  inline void pushChar(char c) {
    out.push_back(c);
    if (out.size() >= kFlushThreshold) flush();
  }

  inline void pushStr(const char* s) {
    while (*s) pushChar(*s++);
  }

  inline void pushUInt(uint64_t x) {
    char tmp[32];
    int n = 0;
    do {
      tmp[n++] = static_cast<char>('0' + (x % 10));
      x /= 10;
    } while (x);
    while (n--) pushChar(tmp[n]);
  }

  inline void pushInt(int x) {
    if (x < 0) {
      pushChar('-');
      pushUInt(static_cast<uint64_t>(-(int64_t)x));
    } else {
      pushUInt(static_cast<uint64_t>(x));
    }
  }

  inline void pushLineUInt(uint64_t x) {
    pushUInt(x);
    pushChar('\n');
  }

  inline void pushEdge(int u, int v) {
    pushInt(u);
    pushChar(' ');
    pushInt(v);
    pushChar('\n');
  }
};

struct Edge {
  int u = 0;
  int v = 0;
};

struct StoerWagnerResult {
  uint64_t cutWeight = 0;
  std::vector<int> side;  // vertices on one side of min cut (original indices)
};

static StoerWagnerResult stoerWagnerMatrix(const std::vector<uint64_t>& orig, int n) {
  std::vector<uint64_t> mat = orig;  // mutated
  std::vector<std::vector<int>> co(static_cast<size_t>(n));
  for (int i = 0; i < n; ++i) co[static_cast<size_t>(i)] = {i};

  std::vector<int> verts(static_cast<size_t>(n));
  std::iota(verts.begin(), verts.end(), 0);

  uint64_t best = std::numeric_limits<uint64_t>::max();
  std::vector<int> bestSide;

  std::vector<uint64_t> w(static_cast<size_t>(n));
  std::vector<unsigned char> added(static_cast<size_t>(n));

  while (verts.size() > 1) {
    std::fill(w.begin(), w.end(), 0);
    std::fill(added.begin(), added.end(), 0);

    int prev = -1;
    int last = verts[0];
    added[static_cast<size_t>(last)] = 1;

    for (int v : verts) {
      if (v == last) continue;
      w[static_cast<size_t>(v)] = mat[static_cast<size_t>(last) * static_cast<size_t>(n) + static_cast<size_t>(v)];
    }

    for (size_t it = 1; it < verts.size(); ++it) {
      int sel = -1;
      uint64_t selw = 0;
      for (int v : verts) {
        if (added[static_cast<size_t>(v)]) continue;
        if (sel == -1 || w[static_cast<size_t>(v)] > selw) {
          sel = v;
          selw = w[static_cast<size_t>(v)];
        }
      }
      if (sel == -1) break;

      prev = last;
      last = sel;
      added[static_cast<size_t>(last)] = 1;

      for (int v : verts) {
        if (added[static_cast<size_t>(v)]) continue;
        w[static_cast<size_t>(v)] +=
            mat[static_cast<size_t>(last) * static_cast<size_t>(n) + static_cast<size_t>(v)];
      }
    }

    const int s = prev;
    const int t = last;
    const uint64_t cut = w[static_cast<size_t>(t)];
    if (cut < best) {
      best = cut;
      bestSide = co[static_cast<size_t>(t)];
    }

    // merge t into s
    co[static_cast<size_t>(s)].insert(co[static_cast<size_t>(s)].end(), co[static_cast<size_t>(t)].begin(),
                                      co[static_cast<size_t>(t)].end());
    co[static_cast<size_t>(t)].clear();

    for (int v : verts) {
      if (v == s || v == t) continue;
      mat[static_cast<size_t>(s) * static_cast<size_t>(n) + static_cast<size_t>(v)] +=
          mat[static_cast<size_t>(t) * static_cast<size_t>(n) + static_cast<size_t>(v)];
      mat[static_cast<size_t>(v) * static_cast<size_t>(n) + static_cast<size_t>(s)] =
          mat[static_cast<size_t>(s) * static_cast<size_t>(n) + static_cast<size_t>(v)];
    }

    auto it = std::find(verts.begin(), verts.end(), t);
    if (it != verts.end()) verts.erase(it);
  }

  StoerWagnerResult res;
  res.cutWeight = (best == std::numeric_limits<uint64_t>::max()) ? 0 : best;
  res.side = std::move(bestSide);
  return res;
}

static StoerWagnerResult stoerWagnerSparse(std::vector<std::unordered_map<int, uint64_t>>& adj,
                                          std::vector<Edge>& edges) {
  const int n = static_cast<int>(adj.size());
  std::vector<unsigned char> active(static_cast<size_t>(n), 1);
  std::vector<int> verts(static_cast<size_t>(n));
  std::iota(verts.begin(), verts.end(), 0);

  std::vector<std::vector<int>> groups(static_cast<size_t>(n));
  for (int i = 0; i < n; ++i) groups[static_cast<size_t>(i)] = {i};

  uint64_t best = std::numeric_limits<uint64_t>::max();
  std::vector<int> bestSide;

  std::vector<uint64_t> w(static_cast<size_t>(n));
  std::vector<unsigned char> inA(static_cast<size_t>(n));

  while (verts.size() > 1) {
    std::fill(w.begin(), w.end(), 0);
    std::fill(inA.begin(), inA.end(), 0);

    std::priority_queue<std::pair<uint64_t, int>> pq;

    const int start = verts[0];
    inA[static_cast<size_t>(start)] = 1;
    int prev = -1;
    int last = start;

    for (const auto& kv : adj[static_cast<size_t>(start)]) {
      const int nei = kv.first;
      if (!active[static_cast<size_t>(nei)] || inA[static_cast<size_t>(nei)]) continue;
      w[static_cast<size_t>(nei)] += kv.second;
      pq.push({w[static_cast<size_t>(nei)], nei});
    }

    for (size_t it = 1; it < verts.size(); ++it) {
      int sel = -1;
      while (!pq.empty()) {
        const auto [val, v] = pq.top();
        if (!active[static_cast<size_t>(v)] || inA[static_cast<size_t>(v)] || val != w[static_cast<size_t>(v)]) {
          pq.pop();
          continue;
        }
        sel = v;
        pq.pop();
        break;
      }
      if (sel == -1) {
        // Shouldn't happen for connected graphs, but keep it safe.
        for (int v : verts) {
          if (active[static_cast<size_t>(v)] && !inA[static_cast<size_t>(v)]) {
            sel = v;
            break;
          }
        }
      }

      prev = last;
      last = sel;
      inA[static_cast<size_t>(sel)] = 1;

      for (const auto& kv : adj[static_cast<size_t>(sel)]) {
        const int nei = kv.first;
        if (!active[static_cast<size_t>(nei)] || inA[static_cast<size_t>(nei)]) continue;
        w[static_cast<size_t>(nei)] += kv.second;
        pq.push({w[static_cast<size_t>(nei)], nei});
      }
    }

    const int s = prev;
    const int t = last;
    const uint64_t cut = w[static_cast<size_t>(t)];
    if (cut < best) {
      best = cut;
      bestSide = groups[static_cast<size_t>(t)];
    }

    // Merge t into s (contract).
    if (groups[static_cast<size_t>(s)].size() < groups[static_cast<size_t>(t)].size()) {
      groups[static_cast<size_t>(s)].swap(groups[static_cast<size_t>(t)]);
    }
    groups[static_cast<size_t>(s)].insert(groups[static_cast<size_t>(s)].end(),
                                          groups[static_cast<size_t>(t)].begin(),
                                          groups[static_cast<size_t>(t)].end());
    groups[static_cast<size_t>(t)].clear();

    // Update adjacency weights.
    for (const auto& kv : adj[static_cast<size_t>(t)]) {
      const int nei = kv.first;
      const uint64_t wt = kv.second;
      if (!active[static_cast<size_t>(nei)] || nei == s) continue;

      auto& mapS = adj[static_cast<size_t>(s)];
      auto itS = mapS.find(nei);
      if (itS == mapS.end()) {
        mapS.emplace(nei, wt);
        itS = mapS.find(nei);
      } else {
        itS->second += wt;
      }
      const uint64_t newW = itS->second;
      adj[static_cast<size_t>(nei)][s] = newW;
      adj[static_cast<size_t>(nei)].erase(t);
    }

    adj[static_cast<size_t>(s)].erase(t);
    adj[static_cast<size_t>(t)].clear();
    active[static_cast<size_t>(t)] = 0;

    auto itv = std::find(verts.begin(), verts.end(), t);
    if (itv != verts.end()) verts.erase(itv);
  }

  StoerWagnerResult res;
  res.cutWeight = (best == std::numeric_limits<uint64_t>::max()) ? 0 : best;
  res.side = std::move(bestSide);
  (void)edges;  // edges not needed for the algorithm itself (only for output)
  return res;
}

static void outputCutFromMatrix(const std::vector<uint64_t>& orig, int n, const std::vector<int>& side) {
  std::vector<unsigned char> inSide(static_cast<size_t>(n), 0);
  for (int v : side) inSide[static_cast<size_t>(v)] = 1;

  uint64_t k = 0;
  for (int u = 0; u < n; ++u) {
    if (!inSide[static_cast<size_t>(u)]) continue;
    const size_t base = static_cast<size_t>(u) * static_cast<size_t>(n);
    for (int v = 0; v < n; ++v) {
      if (inSide[static_cast<size_t>(v)]) continue;
      k += orig[base + static_cast<size_t>(v)];
    }
  }

  FastOutput out;
  out.pushLineUInt(k);
  for (int u = 0; u < n; ++u) {
    if (!inSide[static_cast<size_t>(u)]) continue;
    const size_t base = static_cast<size_t>(u) * static_cast<size_t>(n);
    for (int v = 0; v < n; ++v) {
      if (inSide[static_cast<size_t>(v)]) continue;
      uint64_t cnt = orig[base + static_cast<size_t>(v)];
      while (cnt--) out.pushEdge(u, v);
    }
  }
}

static void outputCutFromEdges(int n, const std::vector<Edge>& edges, const std::vector<int>& side) {
  std::vector<unsigned char> inSide(static_cast<size_t>(n), 0);
  for (int v : side) inSide[static_cast<size_t>(v)] = 1;

  uint64_t k = 0;
  for (const auto& e : edges) {
    if (inSide[static_cast<size_t>(e.u)] != inSide[static_cast<size_t>(e.v)]) ++k;
  }

  FastOutput out;
  out.pushLineUInt(k);
  for (const auto& e : edges) {
    if (inSide[static_cast<size_t>(e.u)] != inSide[static_cast<size_t>(e.v)]) out.pushEdge(e.u, e.v);
  }
}

}  // namespace

int main() {
  FastScanner fs;
  uint64_t N64 = 0, M64 = 0;
  if (!fs.readUInt(N64)) return 0;
  if (!fs.readUInt(M64)) return 0;

  const int n = (N64 > static_cast<uint64_t>(std::numeric_limits<int>::max()))
                    ? std::numeric_limits<int>::max()
                    : static_cast<int>(N64);
  const uint64_t m = M64;

  if (N64 < 2) {
    FastOutput out;
    out.pushLineUInt(0);
    return 0;
  }
  if (N64 != static_cast<uint64_t>(n)) {
    // Extremely large N not representable in int; output a trivial valid cut (cut vertex 0).
    FastOutput out;
    out.pushLineUInt(0);
    return 0;
  }

  // Thresholds tuned for reliability on a wide range of inputs.
  constexpr int kMatrixMaxN = 800;              // exact, O(n^3), but very fast at this scale
  constexpr int kSparseStoerMaxN = 6000;        // exact, contraction-based with hash maps
  constexpr uint64_t kSparseStoerMaxM = 5000000;  // keep edges in memory

  if (n <= kMatrixMaxN) {
    std::vector<uint64_t> orig(static_cast<size_t>(n) * static_cast<size_t>(n), 0);
    for (uint64_t i = 0; i < m; ++i) {
      uint64_t u64 = 0, v64 = 0;
      fs.readUInt(u64);
      fs.readUInt(v64);
      const int u = static_cast<int>(u64);
      const int v = static_cast<int>(v64);
      if (u == v) continue;
      orig[static_cast<size_t>(u) * static_cast<size_t>(n) + static_cast<size_t>(v)] += 1;
      orig[static_cast<size_t>(v) * static_cast<size_t>(n) + static_cast<size_t>(u)] += 1;
    }

    const StoerWagnerResult res = stoerWagnerMatrix(orig, n);
    std::vector<int> side = res.side;
    if (side.empty() || static_cast<int>(side.size()) >= n) side = {0};
    outputCutFromMatrix(orig, n, side);
    return 0;
  }

  // Medium graphs: exact sparse Stoer–Wagner.
  if (n <= kSparseStoerMaxN && m <= kSparseStoerMaxM) {
    std::vector<Edge> edges;
    edges.reserve(static_cast<size_t>(m));
    std::vector<uint32_t> deg(static_cast<size_t>(n), 0);

    for (uint64_t i = 0; i < m; ++i) {
      uint64_t u64 = 0, v64 = 0;
      fs.readUInt(u64);
      fs.readUInt(v64);
      const int u = static_cast<int>(u64);
      const int v = static_cast<int>(v64);
      if (u == v) continue;
      edges.push_back({u, v});
      deg[static_cast<size_t>(u)]++;
      deg[static_cast<size_t>(v)]++;
    }

    std::vector<std::unordered_map<int, uint64_t>> adj(static_cast<size_t>(n));
    for (int i = 0; i < n; ++i) {
      adj[static_cast<size_t>(i)].reserve(static_cast<size_t>(deg[static_cast<size_t>(i)]) + 1);
    }
    for (const auto& e : edges) {
      adj[static_cast<size_t>(e.u)][e.v] += 1;
      adj[static_cast<size_t>(e.v)][e.u] += 1;
    }

    const StoerWagnerResult res = stoerWagnerSparse(adj, edges);
    std::vector<int> side = res.side;
    if (side.empty() || static_cast<int>(side.size()) >= n) side = {0};
    outputCutFromEdges(n, edges, side);
    return 0;
  }

  // Large graphs: memory-light, always-valid cut (isolate a minimum-degree vertex).
  // This path is designed to keep working even when edges do not fit in memory.
  // (Exact global min-cut on massive graphs generally requires more sophisticated algorithms.)
  std::vector<uint32_t> deg(static_cast<size_t>(n), 0);
  const char* tmpName = "edges.bin";
  std::FILE* f = std::fopen(tmpName, "wb");
  if (!f) {
    // If we cannot write a temp file, fall back to printing an empty cut (best-effort).
    FastOutput out;
    out.pushLineUInt(0);
    return 0;
  }

  uint64_t writtenEdges = 0;
  for (uint64_t i = 0; i < m; ++i) {
    uint64_t u64 = 0, v64 = 0;
    fs.readUInt(u64);
    fs.readUInt(v64);
    const uint32_t u = static_cast<uint32_t>(u64);
    const uint32_t v = static_cast<uint32_t>(v64);
    if (u == v) continue;
    deg[static_cast<size_t>(u)]++;
    deg[static_cast<size_t>(v)]++;
    const uint32_t rec[2] = {u, v};
    if (std::fwrite(rec, sizeof(uint32_t), 2, f) != 2) break;
    ++writtenEdges;
  }
  std::fclose(f);

  int vMin = 0;
  uint32_t minDeg = deg[0];
  for (int i = 1; i < n; ++i) {
    if (deg[static_cast<size_t>(i)] < minDeg) {
      minDeg = deg[static_cast<size_t>(i)];
      vMin = i;
    }
  }

  FastOutput out;
  out.pushLineUInt(static_cast<uint64_t>(minDeg));

  std::FILE* rf = std::fopen(tmpName, "rb");
  if (!rf) return 0;
  for (uint64_t i = 0; i < writtenEdges; ++i) {
    uint32_t rec[2];
    if (std::fread(rec, sizeof(uint32_t), 2, rf) != 2) break;
    const int u = static_cast<int>(rec[0]);
    const int v = static_cast<int>(rec[1]);
    if (u == vMin || v == vMin) out.pushEdge(u, v);
  }
  std::fclose(rf);
  out.flush();
  return 0;
}
