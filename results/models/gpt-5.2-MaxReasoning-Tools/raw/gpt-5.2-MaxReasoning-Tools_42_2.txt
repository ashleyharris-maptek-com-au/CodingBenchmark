OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_n1 = OpConstant %float -1
%float_0_5 = OpConstant %float 0.5
%float_n0_5 = OpConstant %float -0.5
%float_0_8660254 = OpConstant %float 0.8660254
%float_n0_8660254 = OpConstant %float -0.8660254
%float_r = OpConstant %float 0.04
%float_edgeMul = OpConstant %float 0.8

%dir0 = OpConstantComposite %v2float %float_1 %float_0
%dir1 = OpConstantComposite %v2float %float_0_5 %float_0_8660254
%dir2 = OpConstantComposite %v2float %float_n0_5 %float_0_8660254
%dir3 = OpConstantComposite %v2float %float_n1 %float_0
%dir4 = OpConstantComposite %v2float %float_n0_5 %float_n0_8660254
%dir5 = OpConstantComposite %v2float %float_0_5 %float_n0_8660254

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv_in Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

%outPosPtr = OpAccessChain %ptr_out_v4float %gl_out %int_0
%inPosPtr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%centerPos = OpLoad %v4float %inPosPtr
%inColorPtr = OpAccessChain %ptr_in_v3 %in_color %int_0
%centerColor = OpLoad %v3float %inColorPtr
%edgeColor = OpVectorTimesScalar %v3float %centerColor %float_edgeMul

%delta2_0 = OpVectorTimesScalar %v2float %dir0 %float_r
%delta4_0 = OpCompositeConstruct %v4float %delta2_0 %float_0 %float_0
%edgePos0 = OpFAdd %v4float %centerPos %delta4_0

%delta2_1 = OpVectorTimesScalar %v2float %dir1 %float_r
%delta4_1 = OpCompositeConstruct %v4float %delta2_1 %float_0 %float_0
%edgePos1 = OpFAdd %v4float %centerPos %delta4_1

%delta2_2 = OpVectorTimesScalar %v2float %dir2 %float_r
%delta4_2 = OpCompositeConstruct %v4float %delta2_2 %float_0 %float_0
%edgePos2 = OpFAdd %v4float %centerPos %delta4_2

%delta2_3 = OpVectorTimesScalar %v2float %dir3 %float_r
%delta4_3 = OpCompositeConstruct %v4float %delta2_3 %float_0 %float_0
%edgePos3 = OpFAdd %v4float %centerPos %delta4_3

%delta2_4 = OpVectorTimesScalar %v2float %dir4 %float_r
%delta4_4 = OpCompositeConstruct %v4float %delta2_4 %float_0 %float_0
%edgePos4 = OpFAdd %v4float %centerPos %delta4_4

%delta2_5 = OpVectorTimesScalar %v2float %dir5 %float_r
%delta4_5 = OpCompositeConstruct %v4float %delta2_5 %float_0 %float_0
%edgePos5 = OpFAdd %v4float %centerPos %delta4_5

; Triangle 0: center, e0, e1
OpStore %outPosPtr %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %outPosPtr %edgePos0
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %outPosPtr %edgePos1
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 1: center, e1, e2
OpStore %outPosPtr %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %outPosPtr %edgePos1
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %outPosPtr %edgePos2
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 2: center, e2, e3
OpStore %outPosPtr %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %outPosPtr %edgePos2
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %outPosPtr %edgePos3
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 3: center, e3, e4
OpStore %outPosPtr %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %outPosPtr %edgePos3
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %outPosPtr %edgePos4
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 4: center, e4, e5
OpStore %outPosPtr %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %outPosPtr %edgePos4
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %outPosPtr %edgePos5
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 5: center, e5, e0
OpStore %outPosPtr %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %outPosPtr %edgePos5
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %outPosPtr %edgePos0
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
