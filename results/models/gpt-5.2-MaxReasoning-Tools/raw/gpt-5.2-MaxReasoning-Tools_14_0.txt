using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const byte UNKNOWN = 9; // ' '
    private const byte MINE = 10;   // '*', or proven mine
    private const byte SAFE = 11;   // proven safe (was UNKNOWN in input)

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        int height = input.ReadInt();
        int width = input.ReadInt();
        if (height <= 0 || width <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        input.ConsumeToEol();

        byte[] board = new byte[height * width];
        for (int y = 0; y < height; y++)
        {
            input.ReadBoardRow(board, y * width, width);
        }

        var safeCells = new List<int>(1024);
        Solve(board, height, width, safeCells);

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(safeCells.Count);
        for (int i = 0; i < safeCells.Count; i++)
        {
            int idx = safeCells[i];
            int y = idx / width;
            int x = idx - y * width;
            writer.Write(x);
            writer.Write(' ');
            writer.WriteLine(y);
        }
    }

    private static void Solve(byte[] board, int h, int w, List<int> safeCells)
    {
        long totalCells = (long)h * w;

        var work = new IntQueue(1024);
        int frontierNumbers = EnqueueInitialFrontierNumbers(board, h, w, work);

        // Adaptive strategy:
        // - Always do single-point propagation.
        // - Add subset comparisons when the frontier is manageable.
        // - Add exact component enumeration only on smaller boards.
        bool useSubset = frontierNumbers <= 250_000;
        bool useEnumeration = totalCells <= 300_000 && frontierNumbers <= 80_000;

        var tmpA = new int[8];
        var tmpB = new int[8];
        var diff = new int[8];

        ProcessQueue(board, h, w, safeCells, work, useSubset, tmpA, tmpB, diff);

        if (!useEnumeration)
            return;

        // Enumeration can unlock deductions that pairwise logic can't.
        // Run a small bounded fixed-point loop: enumerate -> propagate -> repeat.
        for (int round = 0; round < 2; round++)
        {
            bool changed = EnumerateSmallFrontierComponents(board, h, w, safeCells, work);
            if (!changed)
                break;
            ProcessQueue(board, h, w, safeCells, work, useSubset: true, tmpA, tmpB, diff);
        }
    }

    private static int EnqueueInitialFrontierNumbers(byte[] board, int h, int w, IntQueue work)
    {
        int count = 0;
        for (int y = 0; y < h; y++)
        {
            int row = y * w;
            for (int x = 0; x < w; x++)
            {
                int idx = row + x;
                if (board[idx] > 8)
                    continue;
                if (HasAnyUnknownNeighbor(board, h, w, x, y))
                {
                    work.Enqueue(idx);
                    count++;
                }
            }
        }
        return count;
    }

    private static bool HasAnyUnknownNeighbor(byte[] board, int h, int w, int x, int y)
    {
        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)h)
                continue;
            int baseRow = ny * w;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0)
                    continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)w)
                    continue;
                if (board[baseRow + nx] == UNKNOWN)
                    return true;
            }
        }
        return false;
    }

    private static void ProcessQueue(
        byte[] board,
        int h,
        int w,
        List<int> safeCells,
        IntQueue work,
        bool useSubset,
        int[] tmpA,
        int[] tmpB,
        int[] diff)
    {
        while (work.Count > 0)
        {
            int idx = work.Dequeue();
            byte cell = board[idx];
            if (cell > 8)
                continue;

            int y = idx / w;
            int x = idx - y * w;

            int remainingA;
            int lenA = CollectUnknownNeighbors(board, h, w, x, y, tmpA, out remainingA);
            if (lenA == 0)
                continue;

            if ((uint)remainingA > (uint)lenA)
                continue; // inconsistent/no-op

            SortSmall(tmpA, lenA);

            if (remainingA == 0)
            {
                for (int i = 0; i < lenA; i++)
                    MarkSafe(board, h, w, tmpA[i], safeCells, work);
                continue;
            }

            if (remainingA == lenA)
            {
                for (int i = 0; i < lenA; i++)
                    MarkMine(board, h, w, tmpA[i], work);
                continue;
            }

            if (!useSubset)
                continue;

            // Subset logic: compare against nearby number constraints.
            // Any number whose unknown-neighbor set is a superset/subset can yield safe/mine deductions.
            for (int dy = -2; dy <= 2; dy++)
            {
                int by = y + dy;
                if ((uint)by >= (uint)h)
                    continue;
                for (int dx = -2; dx <= 2; dx++)
                {
                    int bx = x + dx;
                    if ((uint)bx >= (uint)w)
                        continue;
                    if (dx == 0 && dy == 0)
                        continue;

                    int bIdx = by * w + bx;
                    if (board[bIdx] > 8)
                        continue;

                    int remainingB;
                    int lenB = CollectUnknownNeighbors(board, h, w, bx, by, tmpB, out remainingB);
                    if (lenB == 0 || (uint)remainingB > (uint)lenB)
                        continue;

                    SortSmall(tmpB, lenB);

                    if (lenA <= lenB && TrySubsetDiff(tmpA, lenA, remainingA, tmpB, lenB, remainingB, diff, out int diffLen, out int diffMines))
                    {
                        if (diffMines == 0)
                        {
                            for (int i = 0; i < diffLen; i++)
                                MarkSafe(board, h, w, diff[i], safeCells, work);
                        }
                        else if (diffMines == diffLen)
                        {
                            for (int i = 0; i < diffLen; i++)
                                MarkMine(board, h, w, diff[i], work);
                        }
                    }

                    if (lenB <= lenA && TrySubsetDiff(tmpB, lenB, remainingB, tmpA, lenA, remainingA, diff, out diffLen, out diffMines))
                    {
                        if (diffMines == 0)
                        {
                            for (int i = 0; i < diffLen; i++)
                                MarkSafe(board, h, w, diff[i], safeCells, work);
                        }
                        else if (diffMines == diffLen)
                        {
                            for (int i = 0; i < diffLen; i++)
                                MarkMine(board, h, w, diff[i], work);
                        }
                    }
                }
            }
        }
    }

    private static int CollectUnknownNeighbors(byte[] board, int h, int w, int x, int y, int[] dst, out int remainingMines)
    {
        int idx = y * w + x;
        int number = board[idx];
        int flagged = 0;
        int len = 0;

        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)h)
                continue;
            int row = ny * w;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0)
                    continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)w)
                    continue;
                byte v = board[row + nx];
                if (v == MINE)
                    flagged++;
                else if (v == UNKNOWN)
                    dst[len++] = row + nx;
            }
        }

        remainingMines = number - flagged;
        return len;
    }

    private static void SortSmall(int[] a, int len)
    {
        for (int i = 1; i < len; i++)
        {
            int v = a[i];
            int j = i - 1;
            while (j >= 0 && a[j] > v)
            {
                a[j + 1] = a[j];
                j--;
            }
            a[j + 1] = v;
        }
    }

    private static bool TrySubsetDiff(
        int[] small,
        int smallLen,
        int smallMines,
        int[] large,
        int largeLen,
        int largeMines,
        int[] diff,
        out int diffLen,
        out int diffMines)
    {
        diffLen = 0;
        int i = 0, j = 0;
        while (j < largeLen)
        {
            int lv = large[j];
            if (i < smallLen && small[i] == lv)
            {
                i++;
                j++;
            }
            else
            {
                diff[diffLen++] = lv;
                j++;
            }
        }

        if (i != smallLen)
        {
            diffLen = 0;
            diffMines = 0;
            return false;
        }

        diffMines = largeMines - smallMines;
        return (uint)diffMines <= (uint)diffLen;
    }

    private static void MarkSafe(byte[] board, int h, int w, int idx, List<int> safeCells, IntQueue work)
    {
        if (board[idx] != UNKNOWN)
            return;
        board[idx] = SAFE;
        safeCells.Add(idx);
        EnqueueAdjacentNumbers(board, h, w, idx, work);
    }

    private static void MarkMine(byte[] board, int h, int w, int idx, IntQueue work)
    {
        if (board[idx] != UNKNOWN)
            return;
        board[idx] = MINE;
        EnqueueAdjacentNumbers(board, h, w, idx, work);
    }

    private static void EnqueueAdjacentNumbers(byte[] board, int h, int w, int idx, IntQueue work)
    {
        int y = idx / w;
        int x = idx - y * w;
        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)h)
                continue;
            int row = ny * w;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0)
                    continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)w)
                    continue;
                int nIdx = row + nx;
                if (board[nIdx] <= 8)
                    work.Enqueue(nIdx);
            }
        }
    }

    private static bool EnumerateSmallFrontierComponents(byte[] board, int h, int w, List<int> safeCells, IntQueue work)
    {
        // Build constraints from the current frontier.
        var varId = new Dictionary<int, int>(4096);
        var varBoardIndex = new List<int>(4096);
        var constraints = new List<Constraint>(4096);

        int[] tmp = new int[8];
        int[] tmpVars = new int[8];

        for (int y = 0; y < h; y++)
        {
            int row = y * w;
            for (int x = 0; x < w; x++)
            {
                int idx = row + x;
                byte v = board[idx];
                if (v > 8)
                    continue;

                int remaining;
                int len = CollectUnknownNeighbors(board, h, w, x, y, tmp, out remaining);
                if (len == 0 || (uint)remaining > (uint)len)
                    continue;

                if (remaining == 0 || remaining == len)
                    continue; // single-point would have handled these; keep frontier smaller

                int k = 0;
                for (int i = 0; i < len; i++)
                {
                    int bIdx = tmp[i];
                    if (!varId.TryGetValue(bIdx, out int id))
                    {
                        id = varBoardIndex.Count;
                        varId.Add(bIdx, id);
                        varBoardIndex.Add(bIdx);
                    }
                    tmpVars[k++] = id;
                }

                Array.Sort(tmpVars, 0, k);
                constraints.Add(new Constraint(CopyOf(tmpVars, k), remaining));
            }
        }

        int varsCount = varBoardIndex.Count;
        int constraintsCount = constraints.Count;
        if (varsCount == 0 || constraintsCount == 0)
            return false;

        var constraintsByVar = new List<int>[varsCount];
        for (int i = 0; i < varsCount; i++)
            constraintsByVar[i] = new List<int>(4);
        for (int ci = 0; ci < constraintsCount; ci++)
        {
            var c = constraints[ci];
            for (int j = 0; j < c.Vars.Length; j++)
                constraintsByVar[c.Vars[j]].Add(ci);
        }

        bool[] seenConstraint = new bool[constraintsCount];
        bool[] seenVar = new bool[varsCount];
        var compConstraints = new List<int>(64);
        var compVars = new List<int>(64);
        var queue = new Queue<int>(64);

        bool changed = false;

        for (int start = 0; start < constraintsCount; start++)
        {
            if (seenConstraint[start])
                continue;

            compConstraints.Clear();
            compVars.Clear();
            queue.Clear();

            seenConstraint[start] = true;
            queue.Enqueue(start);

            while (queue.Count > 0)
            {
                int ci = queue.Dequeue();
                compConstraints.Add(ci);
                var c = constraints[ci];
                for (int j = 0; j < c.Vars.Length; j++)
                {
                    int v = c.Vars[j];
                    if (!seenVar[v])
                    {
                        seenVar[v] = true;
                        compVars.Add(v);
                    }
                }
                for (int j = 0; j < c.Vars.Length; j++)
                {
                    int v = c.Vars[j];
                    var adj = constraintsByVar[v];
                    for (int k2 = 0; k2 < adj.Count; k2++)
                    {
                        int other = adj[k2];
                        if (!seenConstraint[other])
                        {
                            seenConstraint[other] = true;
                            queue.Enqueue(other);
                        }
                    }
                }
            }

            // Clear seenVar marks for next components.
            for (int i = 0; i < compVars.Count; i++)
                seenVar[compVars[i]] = false;

            if (compVars.Count > 24 || compConstraints.Count > 48)
                continue;

            if (SolveComponentByEnumeration(compVars, compConstraints, constraints, varBoardIndex, board, h, w, safeCells, work))
                changed = true;
        }

        return changed;
    }

    private static bool SolveComponentByEnumeration(
        List<int> compVarsGlobal,
        List<int> compConstraintsGlobal,
        List<Constraint> allConstraints,
        List<int> varBoardIndex,
        byte[] board,
        int h,
        int w,
        List<int> safeCells,
        IntQueue work)
    {
        int n = compVarsGlobal.Count;
        int m = compConstraintsGlobal.Count;

        var localOfVar = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++)
            localOfVar[compVarsGlobal[i]] = i;

        int[][] cVars = new int[m][];
        int[] cTarget = new int[m];
        int[] degree = new int[n];

        int totalOccurrences = 0;
        for (int i = 0; i < m; i++)
        {
            var c = allConstraints[compConstraintsGlobal[i]];
            int[] vars = new int[c.Vars.Length];
            for (int j = 0; j < vars.Length; j++)
            {
                int local = localOfVar[c.Vars[j]];
                vars[j] = local;
                degree[local]++;
            }
            cVars[i] = vars;
            cTarget[i] = c.Count;
            totalOccurrences += vars.Length;
        }

        // constraintsPerVar as arrays (2-pass allocation).
        int[][] constraintsPerVar = new int[n][];
        for (int v = 0; v < n; v++)
            constraintsPerVar[v] = new int[degree[v]];
        int[] fill = new int[n];
        for (int ci = 0; ci < m; ci++)
        {
            var vars = cVars[ci];
            for (int j = 0; j < vars.Length; j++)
            {
                int v = vars[j];
                constraintsPerVar[v][fill[v]++] = ci;
            }
        }

        // Variable order: highest degree first (insertion sort, n is tiny).
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        for (int i = 1; i < n; i++)
        {
            int v = order[i];
            int dv = constraintsPerVar[v].Length;
            int j = i - 1;
            while (j >= 0 && constraintsPerVar[order[j]].Length < dv)
            {
                order[j + 1] = order[j];
                j--;
            }
            order[j + 1] = v;
        }

        sbyte[] assign = new sbyte[n];
        Array.Fill(assign, (sbyte)-1);

        int[] remCount = new int[m];
        int[] remVars = new int[m];
        for (int i = 0; i < m; i++)
        {
            remCount[i] = cTarget[i];
            remVars[i] = cVars[i].Length;
        }

        bool[] seenMine = new bool[n];
        bool[] seenSafe = new bool[n];
        int solutions = 0;

        // Change stack for undo (allocation-free search).
        int[] chConstraint = new int[Math.Max(16, totalOccurrences)];
        int[] chPrevCount = new int[chConstraint.Length];
        int[] chPrevVars = new int[chConstraint.Length];
        int chTop = 0;

        bool AllDetermined()
        {
            for (int i = 0; i < n; i++)
            {
                if (!(seenMine[i] && seenSafe[i]))
                    return false;
            }
            return true;
        }

        bool ApplyAndCheck(int v, int value, int oldTop)
        {
            var cons = constraintsPerVar[v];
            for (int i = 0; i < cons.Length; i++)
            {
                int ci = cons[i];
                if (chTop >= chConstraint.Length)
                    return false; // should not happen with chosen bounds

                chConstraint[chTop] = ci;
                chPrevCount[chTop] = remCount[ci];
                chPrevVars[chTop] = remVars[ci];
                chTop++;

                remVars[ci]--;
                if (value == 1) remCount[ci]--;

                int rc = remCount[ci];
                int rv = remVars[ci];
                if (rc < 0 || rc > rv)
                {
                    // Undo partial updates for this assignment immediately.
                    while (chTop > oldTop)
                    {
                        chTop--;
                        int u = chConstraint[chTop];
                        remCount[u] = chPrevCount[chTop];
                        remVars[u] = chPrevVars[chTop];
                    }
                    return false;
                }
            }
            return true;
        }

        void UndoTo(int oldTop)
        {
            while (chTop > oldTop)
            {
                chTop--;
                int ci = chConstraint[chTop];
                remCount[ci] = chPrevCount[chTop];
                remVars[ci] = chPrevVars[chTop];
            }
        }

        bool Dfs(int depth)
        {
            if (depth == n)
            {
                solutions++;
                for (int i = 0; i < n; i++)
                {
                    if (assign[i] == 1) seenMine[i] = true;
                    else seenSafe[i] = true;
                }
                return AllDetermined();
            }

            int v = order[depth];
            int oldTop = chTop;

            // Try safe then mine.
            for (int value = 0; value <= 1; value++)
            {
                assign[v] = (sbyte)value;
                if (ApplyAndCheck(v, value, oldTop))
                {
                    if (Dfs(depth + 1))
                        return true;
                }
                UndoTo(oldTop);
                assign[v] = -1;
            }
            return false;
        }

        Dfs(0);

        if (solutions == 0)
            return false;

        bool changed = false;
        for (int i = 0; i < n; i++)
        {
            int boardIdx = varBoardIndex[compVarsGlobal[i]];
            if (board[boardIdx] != UNKNOWN)
                continue;

            if (!seenMine[i] && seenSafe[i])
            {
                MarkSafe(board, h, w, boardIdx, safeCells, work);
                changed = true;
            }
            else if (!seenSafe[i] && seenMine[i])
            {
                MarkMine(board, h, w, boardIdx, work);
                changed = true;
            }
        }

        return changed;
    }

    private static int[] CopyOf(int[] a, int len)
    {
        var r = new int[len];
        Array.Copy(a, 0, r, 0, len);
        return r;
    }

    private readonly struct Constraint
    {
        public readonly int[] Vars;
        public readonly int Count;

        public Constraint(int[] vars, int count)
        {
            Vars = vars;
            Count = count;
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;
        private int _peek; // -2: empty, -1: EOF, else byte

        public FastInput(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
            _peek = -2;
        }

        private int ReadRaw()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0)
                    return -1;
            }
            return _buffer[_pos++];
        }

        private int ReadByte()
        {
            if (_peek != -2)
            {
                int c = _peek;
                _peek = -2;
                return c;
            }
            return ReadRaw();
        }

        public int ReadInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int v = 0;
            while ((uint)(c - '0') <= 9)
            {
                v = (v * 10) + (c - '0');
                c = ReadByte();
            }

            _peek = c;
            return v * sign;
        }

        public void ConsumeToEol()
        {
            while (true)
            {
                int c = ReadByte();
                if (c == -1 || c == '\n')
                    return;
            }
        }

        public void ReadBoardRow(byte[] board, int offset, int width)
        {
            int x = 0;
            while (x < width)
            {
                int c = ReadByte();
                if (c == -1 || c == '\n')
                {
                    for (; x < width; x++)
                        board[offset + x] = UNKNOWN;
                    return;
                }
                if (c == '\r')
                    continue;

                board[offset + x] = DecodeCell(c);
                x++;
            }

            // Consume to EOL (if there are any extra characters).
            while (true)
            {
                int c = ReadByte();
                if (c == -1 || c == '\n')
                    return;
            }
        }

        private static byte DecodeCell(int c)
        {
            if (c == ' ') return UNKNOWN;
            if (c == '*') return MINE;
            int d = c - '0';
            if ((uint)d <= 8) return (byte)d;
            return UNKNOWN;
        }
    }

    private sealed class IntQueue
    {
        private int[] _data;
        private int _head;
        private int _tail;
        public int Count { get; private set; }

        public IntQueue(int capacity)
        {
            _data = new int[Math.Max(4, capacity)];
        }

        public void Enqueue(int value)
        {
            if (Count == _data.Length)
                Grow();
            _data[_tail] = value;
            _tail++;
            if (_tail == _data.Length)
                _tail = 0;
            Count++;
        }

        public int Dequeue()
        {
            int v = _data[_head];
            _head++;
            if (_head == _data.Length)
                _head = 0;
            Count--;
            return v;
        }

        private void Grow()
        {
            int newCap = _data.Length * 2;
            var nd = new int[newCap];
            if (Count > 0)
            {
                if (_head < _tail)
                {
                    Array.Copy(_data, _head, nd, 0, Count);
                }
                else
                {
                    int right = _data.Length - _head;
                    Array.Copy(_data, _head, nd, 0, right);
                    Array.Copy(_data, 0, nd, right, _tail);
                }
            }
            _data = nd;
            _head = 0;
            _tail = Count;
        }
    }
}
