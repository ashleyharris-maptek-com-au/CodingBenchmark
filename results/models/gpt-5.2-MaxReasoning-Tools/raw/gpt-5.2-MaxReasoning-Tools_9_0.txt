using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;

        public FastScanner(int bufferSize = 1 << 20)
        {
            _stream = Console.OpenStandardInput();
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            while (true)
            {
                do
                {
                    c = Read();
                    if (c < 0) return int.MinValue;
                } while (c <= 32);

                // Skip UTF-8 BOM if present at the very beginning of the stream.
                if (c == 0xEF)
                {
                    int c2 = Read();
                    int c3 = Read();
                    if (c2 == 0xBB && c3 == 0xBF) continue;
                    // Not a BOM; treat as invalid for numeric parsing and return sentinel.
                    return int.MinValue;
                }

                break;
            }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = (value * 10) + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    private sealed class FastOutput : IDisposable
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _pos;

        public FastOutput(int bufferSize = 1 << 20)
        {
            _stream = new BufferedStream(Console.OpenStandardOutput(), bufferSize);
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void Ensure(int needed)
        {
            if (_pos + needed <= _buffer.Length) return;
            _stream.Write(_buffer, 0, _pos);
            _pos = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            Ensure(1);
            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace() => WriteByte((byte)' ');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine() => WriteByte((byte)'\n');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[10];
            int len = 0;
            uint v = (uint)value;
            while (v != 0)
            {
                uint q = v / 10;
                tmp[len++] = (byte)('0' + (v - q * 10));
                v = q;
            }

            Ensure(len);
            for (int i = len - 1; i >= 0; i--)
            {
                _buffer[_pos++] = tmp[i];
            }
        }

        public void Flush()
        {
            if (_pos > 0)
            {
                _stream.Write(_buffer, 0, _pos);
                _pos = 0;
            }
            _stream.Flush();
        }

        public void Dispose()
        {
            Flush();
            _stream.Dispose();
        }
    }

    private struct XorShift32
    {
        private uint _state;
        public XorShift32(uint seed) => _state = seed != 0 ? seed : 2463534242u;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public uint NextUInt()
        {
            uint x = _state;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _state = x;
            return x;
        }
    }

    // Treap stored in parallel arrays. Node indices are 1..binCount (0 means null).
    private static long[] s_key = Array.Empty<long>();
    private static int[] s_pri = Array.Empty<int>();
    private static int[] s_left = Array.Empty<int>();
    private static int[] s_right = Array.Empty<int>();

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static int RotateRight(int y)
    {
        int x = s_left[y];
        int t2 = s_right[x];
        s_right[x] = y;
        s_left[y] = t2;
        return x;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static int RotateLeft(int x)
    {
        int y = s_right[x];
        int t2 = s_left[y];
        s_left[y] = x;
        s_right[x] = t2;
        return y;
    }

    private static int Insert(int root, int node)
    {
        if (root == 0) return node;
        if (s_key[node] < s_key[root])
        {
            int l = Insert(s_left[root], node);
            s_left[root] = l;
            if (l != 0 && s_pri[l] > s_pri[root]) root = RotateRight(root);
        }
        else
        {
            int r = Insert(s_right[root], node);
            s_right[root] = r;
            if (r != 0 && s_pri[r] > s_pri[root]) root = RotateLeft(root);
        }
        return root;
    }

    private static int Merge(int a, int b)
    {
        if (a == 0) return b;
        if (b == 0) return a;

        if (s_pri[a] > s_pri[b])
        {
            s_right[a] = Merge(s_right[a], b);
            return a;
        }
        else
        {
            s_left[b] = Merge(a, s_left[b]);
            return b;
        }
    }

    // Removes and returns the smallest node with key >= searchKey.
    private static int ExtractLowerBound(int root, long searchKey, out int extractedNode)
    {
        if (root == 0)
        {
            extractedNode = 0;
            return 0;
        }

        if (s_key[root] < searchKey)
        {
            s_right[root] = ExtractLowerBound(s_right[root], searchKey, out extractedNode);
            return root;
        }

        // s_key[root] >= searchKey; try to find a better (smaller) one in the left subtree first.
        int newLeft = ExtractLowerBound(s_left[root], searchKey, out extractedNode);
        if (extractedNode != 0)
        {
            s_left[root] = newLeft;
            return root;
        }

        // Nothing in the left subtree qualifies, so extract root.
        extractedNode = root;
        return Merge(s_left[root], s_right[root]);
    }

    private static void RadixSortInt32(int[] keys, int[] values)
    {
        int n = keys.Length;
        if (n <= 1) return;

        int[] tmpKeys = new int[n];
        int[] tmpValues = new int[n];
        const int R = 1 << 16;
        int[] count = new int[R];

        // Pass 0: low 16 bits
        Array.Clear(count, 0, R);
        for (int i = 0; i < n; i++) count[keys[i] & 0xFFFF]++;
        int sum = 0;
        for (int r = 0; r < R; r++)
        {
            int c = count[r];
            count[r] = sum;
            sum += c;
        }
        for (int i = 0; i < n; i++)
        {
            int k = keys[i];
            int p = count[k & 0xFFFF]++;
            tmpKeys[p] = k;
            tmpValues[p] = values[i];
        }

        // Pass 1: high 16 bits
        Array.Clear(count, 0, R);
        for (int i = 0; i < n; i++) count[(tmpKeys[i] >> 16) & 0xFFFF]++;
        sum = 0;
        for (int r = 0; r < R; r++)
        {
            int c = count[r];
            count[r] = sum;
            sum += c;
        }
        for (int i = 0; i < n; i++)
        {
            int k = tmpKeys[i];
            int p = count[(k >> 16) & 0xFFFF]++;
            keys[p] = k;
            values[p] = tmpValues[i];
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        int stockLength = fs.NextInt();

        int[] lengths = new int[n];
        int[] order = new int[n];
        for (int i = 0; i < n; i++)
        {
            lengths[i] = fs.NextInt();
            order[i] = i;
        }

        // Sort cuts by length (ascending); iterate from end for decreasing order.
        RadixSortInt32(lengths, order);

        int[] binForCut = new int[n];

        // Treap capacity: at most n bins, node index = binId + 1.
        s_key = new long[n + 1];
        s_pri = new int[n + 1];
        s_left = new int[n + 1];
        s_right = new int[n + 1];

        var rng = new XorShift32((uint)Environment.TickCount ^ 0x9E3779B9u);

        int root = 0;
        int binCount = 0;

        for (int pos = n - 1; pos >= 0; pos--)
        {
            int len = lengths[pos];
            int cutIndex = order[pos];

            long searchKey = ((long)len) << 32;
            int node;
            root = ExtractLowerBound(root, searchKey, out node);

            if (node != 0)
            {
                int binId = node - 1;
                int remaining = (int)(s_key[node] >> 32);
                int newRemaining = remaining - len;

                binForCut[cutIndex] = binId;

                // Detach before potential reinsert.
                s_left[node] = 0;
                s_right[node] = 0;

                if (newRemaining > 0)
                {
                    s_key[node] = (((long)newRemaining) << 32) | (uint)binId;
                    root = Insert(root, node);
                }
            }
            else
            {
                int binId = binCount++;
                binForCut[cutIndex] = binId;

                int remaining = stockLength - len;
                if (remaining > 0)
                {
                    int nodeId = binId + 1;
                    s_key[nodeId] = (((long)remaining) << 32) | (uint)binId;
                    s_pri[nodeId] = (int)rng.NextUInt();
                    s_left[nodeId] = 0;
                    s_right[nodeId] = 0;
                    root = Insert(root, nodeId);
                }
            }
        }

        using var fo = new FastOutput();
        fo.WriteInt(binCount);
        fo.WriteNewLine();

        if (binCount == 0)
        {
            fo.Flush();
            return;
        }

        int[] counts = new int[binCount];
        for (int i = 0; i < n; i++) counts[binForCut[i]]++;

        int[] start = new int[binCount + 1];
        for (int b = 0; b < binCount; b++) start[b + 1] = start[b] + counts[b];

        int[] writePos = new int[binCount];
        Array.Copy(start, writePos, binCount);

        int[] grouped = new int[n];
        for (int i = 0; i < n; i++)
        {
            int b = binForCut[i];
            grouped[writePos[b]++] = i;
        }

        for (int b = 0; b < binCount; b++)
        {
            int s = start[b];
            int e = start[b + 1];
            if (s < e)
            {
                fo.WriteInt(grouped[s]);
                for (int j = s + 1; j < e; j++)
                {
                    fo.WriteSpace();
                    fo.WriteInt(grouped[j]);
                }
            }
            fo.WriteNewLine();
        }

        fo.Flush();
    }
}
