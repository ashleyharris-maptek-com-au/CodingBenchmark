using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            while (c != '-' && (c < '0' || c > '9'))
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
                while (c <= 32)
                {
                    c = ReadByte();
                    if (c == -1) return int.MinValue;
                }
            }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
                if (c == -1) return int.MinValue;
            }

            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
                if (c == -1) break;
            }

            return val * sign;
        }
    }

    private static int N;
    private static int[] Flow = Array.Empty<int>();
    private static int[] Dist = Array.Empty<int>();

    private static long ComputeCost(int[] x)
    {
        int n = N;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int fi = x[i];
            int distRow = i * n;
            int flowRow = fi * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)Dist[distRow + j] * Flow[flowRow + x[j]];
            }
        }
        return cost;
    }

    private static long DeltaSwap(int[] x, int i, int j)
    {
        if (i == j) return 0;
        int n = N;
        int xi = x[i];
        int xj = x[j];
        int xiRow = xi * n;
        int xjRow = xj * n;

        long delta = 0;
        delta += ((long)Dist[i * n + i] - Dist[j * n + j]) * (Flow[xjRow + xj] - Flow[xiRow + xi]);
        delta += ((long)Dist[i * n + j] - Dist[j * n + i]) * (Flow[xjRow + xi] - Flow[xiRow + xj]);

        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            int xk = x[k];
            int xkRow = xk * n;

            delta += ((long)Dist[k * n + i] - Dist[k * n + j]) * (Flow[xkRow + xj] - Flow[xkRow + xi]);
            delta += ((long)Dist[i * n + k] - Dist[j * n + k]) * (Flow[xjRow + xk] - Flow[xiRow + xk]);
        }
        return delta;
    }

    private static void Swap(int[] x, int i, int j)
    {
        int tmp = x[i];
        x[i] = x[j];
        x[j] = tmp;
    }

    private static bool NextPermutation(int[] a)
    {
        int n = a.Length;
        int i = n - 2;
        while (i >= 0 && a[i] >= a[i + 1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (a[j] <= a[i]) j--;
        int tmp = a[i];
        a[i] = a[j];
        a[j] = tmp;
        Array.Reverse(a, i + 1, n - (i + 1));
        return true;
    }

    private static (int[] x, long cost) SolveExact()
    {
        int n = N;
        int[] perm = new int[n];
        for (int i = 0; i < n; i++) perm[i] = i;

        int[] best = (int[])perm.Clone();
        long bestCost = ComputeCost(perm);

        while (NextPermutation(perm))
        {
            long c = ComputeCost(perm);
            if (c < bestCost)
            {
                bestCost = c;
                Array.Copy(perm, best, n);
            }
        }

        return (best, bestCost);
    }

    private static void RandomPerturb(int[] x, Random rng, int swaps)
    {
        int n = x.Length;
        for (int t = 0; t < swaps; t++)
        {
            int i = rng.Next(n);
            int j = rng.Next(n - 1);
            if (j >= i) j++;
            Swap(x, i, j);
        }
    }

    private static void BuildWorstList(int[] x, long[] contrib, int[] idx, int[] worst)
    {
        int n = N;
        Array.Clear(contrib, 0, n);

        for (int i = 0; i < n; i++)
        {
            long ci = 0;
            int fi = x[i];
            int fiRow = fi * n;
            int distRowI = i * n;
            for (int j = 0; j < n; j++)
            {
                int fj = x[j];
                ci += (long)Dist[distRowI + j] * Flow[fiRow + fj];
                ci += (long)Dist[j * n + i] * Flow[fj * n + fi];
            }
            contrib[i] = ci;
            idx[i] = i;
        }

        Array.Sort(idx, (a, b) => contrib[b].CompareTo(contrib[a]));
        int kWorst = worst.Length;
        for (int t = 0; t < kWorst; t++) worst[t] = idx[t];
    }

    private static void LocalSearchAllPairs(int[] x, ref long cost, long deadlineTs, Random rng)
    {
        int n = N;
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        for (int i = n - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int tmp = order[i];
            order[i] = order[j];
            order[j] = tmp;
        }

        for (int pass = 0; pass < 3; pass++)
        {
            bool improved = false;
            for (int a = 0; a < n; a++)
            {
                int i = order[a];
                for (int b = a + 1; b < n; b++)
                {
                    if ((b & 63) == 0 && Stopwatch.GetTimestamp() >= deadlineTs) return;
                    int j = order[b];
                    long delta = DeltaSwap(x, i, j);
                    if (delta < 0)
                    {
                        Swap(x, i, j);
                        cost += delta;
                        improved = true;
                    }
                }
            }
            if (!improved) return;
        }
    }

    private static void LocalSearchStochastic(int[] x, ref long cost, long deadlineTs, Random rng)
    {
        int n = N;
        int kWorst = Math.Min(64, n);
        int[] worst = new int[kWorst];
        long[] contrib = new long[n];
        int[] idx = new int[n];
        BuildWorstList(x, contrib, idx, worst);

        int candidateJs = n <= 300 ? 16 : n <= 1000 ? 10 : 4;
        long updateAfterDeltas = n <= 500 ? 5000 : n <= 1000 ? 12000 : 20000;
        long nextUpdate = updateAfterDeltas;
        long deltaEvals = 0;

        int iter = 0;
        while (true)
        {
            if ((iter & 127) == 0 && Stopwatch.GetTimestamp() >= deadlineTs) return;

            int i = worst[rng.Next(kWorst)];
            int bestJ = -1;
            long bestDelta = 0;

            for (int c = 0; c < candidateJs; c++)
            {
                int j = rng.Next(n - 1);
                if (j >= i) j++;
                long delta = DeltaSwap(x, i, j);
                deltaEvals++;
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestJ = j;
                }
            }

            if (bestJ >= 0)
            {
                Swap(x, i, bestJ);
                cost += bestDelta;
            }

            if (deltaEvals >= nextUpdate)
            {
                BuildWorstList(x, contrib, idx, worst);
                nextUpdate += updateAfterDeltas;
            }

            iter++;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n <= 0 || n == int.MinValue) return;
        N = n;

        Flow = new int[n * n];
        Dist = new int[n * n];

        long[] flowRowSum = new long[n];
        long[] flowColSum = new long[n];
        for (int i = 0; i < n; i++)
        {
            int baseIdx = i * n;
            for (int j = 0; j < n; j++)
            {
                int v = fs.NextInt();
                Flow[baseIdx + j] = v;
                flowRowSum[i] += v;
                flowColSum[j] += v;
            }
        }

        long[] distRowSum = new long[n];
        long[] distColSum = new long[n];
        for (int i = 0; i < n; i++)
        {
            int baseIdx = i * n;
            for (int j = 0; j < n; j++)
            {
                int v = fs.NextInt();
                Dist[baseIdx + j] = v;
                distRowSum[i] += v;
                distColSum[j] += v;
            }
        }

        if (n <= 10)
        {
            var (bestXExact, _) = SolveExact();
            long finalCostExact = ComputeCost(bestXExact);
            WriteOutput(bestXExact, finalCostExact);
            return;
        }

        long[] facWeight = new long[n];
        long[] locWeight = new long[n];
        int[] facOrder = new int[n];
        int[] locOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            facWeight[i] = flowRowSum[i] + flowColSum[i];
            locWeight[i] = distRowSum[i] + distColSum[i];
            facOrder[i] = i;
            locOrder[i] = i;
        }

        Array.Sort(facOrder, (a, b) => facWeight[b].CompareTo(facWeight[a]));
        Array.Sort(locOrder, (a, b) => locWeight[a].CompareTo(locWeight[b]));

        int[] baseX = new int[n];
        for (int k = 0; k < n; k++) baseX[locOrder[k]] = facOrder[k];

        Random rng = new Random(unchecked(Environment.TickCount * 31 + n));
        long timeBudgetMs = Math.Min(285_000L, 2_000L + 150L * n);
        long startTs = Stopwatch.GetTimestamp();
        long deadlineTs = startTs + (timeBudgetMs * Stopwatch.Frequency) / 1000;

        int restarts;
        if (n <= 200) restarts = 5;
        else if (n <= 600) restarts = 3;
        else if (n <= 1200) restarts = 2;
        else restarts = 1;

        int[] bestX = (int[])baseX.Clone();
        long bestCost = ComputeCost(bestX);

        for (int r = 0; r < restarts; r++)
        {
            long nowTs = Stopwatch.GetTimestamp();
            if (nowTs >= deadlineTs) break;
            long restartDeadline = nowTs + (deadlineTs - nowTs) / (restarts - r);

            int[] x = (int[])baseX.Clone();
            if (r > 0)
            {
                int swaps = Math.Min(n, Math.Max(32, n / 10));
                RandomPerturb(x, rng, swaps);
            }

            long cost = ComputeCost(x);
            if (n <= 600) LocalSearchAllPairs(x, ref cost, restartDeadline, rng);
            else LocalSearchStochastic(x, ref cost, restartDeadline, rng);

            if (cost < bestCost)
            {
                bestCost = cost;
                bestX = x;
            }
        }

        long finalCost = ComputeCost(bestX);
        WriteOutput(bestX, finalCost);
    }

    private static void WriteOutput(int[] xLocToFac, long totalCost)
    {
        int n = xLocToFac.Length;
        int[] pFacToLoc = new int[n];
        for (int loc = 0; loc < n; loc++) pFacToLoc[xLocToFac[loc]] = loc;

        var sb = new StringBuilder(n * 6 + 64);
        sb.Append(totalCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i != 0) sb.Append(' ');
            sb.Append(pFacToLoc[i]);
        }
        sb.Append('\n');
        Console.Out.Write(sb.ToString());
    }
}
