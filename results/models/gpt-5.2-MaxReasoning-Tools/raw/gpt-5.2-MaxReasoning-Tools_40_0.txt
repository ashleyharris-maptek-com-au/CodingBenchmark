#include <algorithm>
#include <cassert>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <queue>
#include <string>
#include <utility>
#include <vector>

#ifdef _MSC_VER
#include <intrin.h>
#endif

using std::int64_t;

static inline int ctz64(uint64_t x) {
#ifdef _MSC_VER
  unsigned long idx = 0;
#if defined(_WIN64) || defined(_M_X64) || defined(_M_ARM64)
  _BitScanForward64(&idx, x);
  return static_cast<int>(idx);
#else
  uint32_t lo = static_cast<uint32_t>(x);
  if (lo) {
    _BitScanForward(&idx, lo);
    return static_cast<int>(idx);
  }
  uint32_t hi = static_cast<uint32_t>(x >> 32);
  _BitScanForward(&idx, hi);
  return static_cast<int>(idx + 32);
#endif
#else
  return __builtin_ctzll(x);
#endif
}

struct MaxWeightCliqueSolver {
  int n = 0;
  int blocks = 0;
  uint64_t last_mask = ~0ULL;

  std::vector<std::vector<uint64_t>> adj;
  std::vector<int64_t> w;

  int64_t best_weight = 0;
  std::vector<int> best_clique;

  static inline bool bitset_empty(const std::vector<uint64_t>& bs) {
    for (uint64_t word : bs) {
      if (word) return false;
    }
    return true;
  }

  inline void bitset_trim(std::vector<uint64_t>& bs) const {
    if (!bs.empty()) bs.back() &= last_mask;
  }

  inline void bitset_set(std::vector<uint64_t>& bs, int idx) const {
    bs[idx >> 6] |= (1ULL << (idx & 63));
  }

  inline void bitset_clear(std::vector<uint64_t>& bs, int idx) const {
    bs[idx >> 6] &= ~(1ULL << (idx & 63));
  }

  inline bool bitset_test(const std::vector<uint64_t>& bs, int idx) const {
    return (bs[idx >> 6] >> (idx & 63)) & 1ULL;
  }

  inline void bitset_and_inplace(std::vector<uint64_t>& a, const std::vector<uint64_t>& b) const {
    for (int i = 0; i < blocks; ++i) a[i] &= b[i];
  }

  inline void bitset_andnot_inplace(std::vector<uint64_t>& a, const std::vector<uint64_t>& b) const {
    for (int i = 0; i < blocks; ++i) a[i] &= ~b[i];
    bitset_trim(a);
  }

  inline int bitset_pop_first(std::vector<uint64_t>& bs) const {
    for (int bi = 0; bi < blocks; ++bi) {
      uint64_t word = bs[bi];
      if (!word) continue;
      int bit = ctz64(word);
      bs[bi] = word & (word - 1);
      return (bi << 6) + bit;
    }
    return -1;
  }

  int pick_max_weight_vertex(const std::vector<uint64_t>& bs) const {
    int best_v = -1;
    int64_t best_w = -1;
    for (int bi = 0; bi < blocks; ++bi) {
      uint64_t word = bs[bi];
      while (word) {
        int bit = ctz64(word);
        int v = (bi << 6) + bit;
        if (v < n && w[v] > best_w) {
          best_w = w[v];
          best_v = v;
        }
        word &= (word - 1);
      }
    }
    return best_v;
  }

  void greedy_initial_clique() {
    std::vector<uint64_t> cand(blocks, ~0ULL);
    bitset_trim(cand);
    int64_t total = 0;
    std::vector<int> clique;
    while (!bitset_empty(cand)) {
      int v = pick_max_weight_vertex(cand);
      if (v < 0) break;
      clique.push_back(v);
      total += w[v];
      bitset_and_inplace(cand, adj[v]);
    }
    if (total > best_weight) {
      best_weight = total;
      best_clique = std::move(clique);
    }
  }

  void color_bound(const std::vector<uint64_t>& cand,
                   std::vector<int>& order,
                   std::vector<int64_t>& bounds) const {
    order.clear();
    bounds.clear();

    std::vector<uint64_t> uncolored = cand;
    int64_t cumulative = 0;

    while (!bitset_empty(uncolored)) {
      std::vector<uint64_t> avail = uncolored;
      std::vector<int> color_class;
      int64_t class_max = 0;

      while (!bitset_empty(avail)) {
        int v = bitset_pop_first(avail);
        if (v < 0) break;

        color_class.push_back(v);
        class_max = std::max(class_max, w[v]);
        bitset_clear(uncolored, v);
        bitset_andnot_inplace(avail, adj[v]);
      }

      cumulative += class_max;
      std::sort(color_class.begin(), color_class.end(), [&](int a, int b) {
        if (w[a] != w[b]) return w[a] < w[b];
        return a < b;
      });
      for (int v : color_class) {
        order.push_back(v);
        bounds.push_back(cumulative);
      }
    }
  }

  void expand(std::vector<int>& clique, int64_t clique_weight, std::vector<uint64_t> cand) {
    std::vector<int> order;
    std::vector<int64_t> bounds;
    color_bound(cand, order, bounds);

    for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
      int v = order[i];
      if (!bitset_test(cand, v)) continue;
      if (clique_weight + bounds[i] <= best_weight) return;

      clique.push_back(v);
      int64_t new_weight = clique_weight + w[v];
      if (new_weight > best_weight) {
        best_weight = new_weight;
        best_clique = clique;
      }

      std::vector<uint64_t> new_cand = cand;
      bitset_and_inplace(new_cand, adj[v]);
      if (!bitset_empty(new_cand)) {
        expand(clique, new_weight, std::move(new_cand));
      }
      clique.pop_back();

      bitset_clear(cand, v);
    }
  }

  std::vector<int> solve() {
    best_weight = 0;
    best_clique.clear();

    greedy_initial_clique();

    std::vector<uint64_t> cand(blocks, ~0ULL);
    bitset_trim(cand);
    std::vector<int> clique;
    expand(clique, 0, std::move(cand));
    return best_clique;
  }
};

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n = 0, m = 0;
  if (!(std::cin >> n >> m)) return 0;
  std::vector<int64_t> c(n, 0);
  for (int i = 0; i < n; ++i) std::cin >> c[i];

  std::vector<uint8_t> forced_zero(n, 0);
  std::vector<std::pair<int, int>> edges;
  edges.reserve(std::max(0, m));
  for (int i = 0; i < m; ++i) {
    int u, v;
    std::cin >> u >> v;
    if (u == v) {
      if (0 <= u && u < n) forced_zero[u] = 1;
      continue;
    }
    if (u > v) std::swap(u, v);
    edges.emplace_back(u, v);
  }

  std::vector<int> old_to_active(n, -1);
  std::vector<int> active_to_old;
  active_to_old.reserve(n);
  for (int i = 0; i < n; ++i) {
    if (forced_zero[i]) continue;
    if (c[i] <= 0) continue;
    old_to_active[i] = static_cast<int>(active_to_old.size());
    active_to_old.push_back(i);
  }

  const int N = static_cast<int>(active_to_old.size());
  std::vector<int> x(n, 0);
  if (N == 0) {
    std::cout << 0 << "\n";
    for (int i = 0; i < n; ++i) {
      if (i) std::cout << ' ';
      std::cout << 0;
    }
    std::cout << "\n";
    return 0;
  }

  std::vector<std::vector<int>> conflict_adj(N);
  for (auto [u, v] : edges) {
    int au = (0 <= u && u < n) ? old_to_active[u] : -1;
    int av = (0 <= v && v < n) ? old_to_active[v] : -1;
    if (au < 0 || av < 0) continue;
    conflict_adj[au].push_back(av);
    conflict_adj[av].push_back(au);
  }
  for (auto& vec : conflict_adj) {
    std::sort(vec.begin(), vec.end());
    vec.erase(std::unique(vec.begin(), vec.end()), vec.end());
  }

  // Decompose by connected components in the conflict graph.
  std::vector<int> comp_id(N, -1);
  std::vector<std::vector<int>> comps;
  for (int i = 0; i < N; ++i) {
    if (comp_id[i] != -1) continue;
    int cid = static_cast<int>(comps.size());
    comps.push_back({});
    std::queue<int> q;
    q.push(i);
    comp_id[i] = cid;
    while (!q.empty()) {
      int v = q.front();
      q.pop();
      comps[cid].push_back(v);
      for (int to : conflict_adj[v]) {
        if (comp_id[to] == -1) {
          comp_id[to] = cid;
          q.push(to);
        }
      }
    }
  }

  std::vector<int> chosen_active;
  int64_t objective = 0;

  std::vector<int> g2l(N, -1);
  for (const auto& comp : comps) {
    const int k = static_cast<int>(comp.size());
    if (k == 0) continue;
    if (k == 1) {
      chosen_active.push_back(comp[0]);
      objective += c[active_to_old[comp[0]]];
      continue;
    }

    for (int i = 0; i < k; ++i) g2l[comp[i]] = i;

    MaxWeightCliqueSolver solver;
    solver.n = k;
    solver.blocks = (k + 63) / 64;
    solver.last_mask = (k % 64 == 0) ? ~0ULL : ((1ULL << (k % 64)) - 1ULL);
    solver.w.resize(k);
    for (int i = 0; i < k; ++i) solver.w[i] = c[active_to_old[comp[i]]];

    solver.adj.assign(k, std::vector<uint64_t>(solver.blocks, ~0ULL));
    for (int i = 0; i < k; ++i) {
      solver.adj[i].back() &= solver.last_mask;
      solver.adj[i][i >> 6] &= ~(1ULL << (i & 63));  // no self-loop
    }
    for (int gi : comp) {
      int li = g2l[gi];
      for (int gj : conflict_adj[gi]) {
        int lj = g2l[gj];
        if (lj >= 0) solver.adj[li][lj >> 6] &= ~(1ULL << (lj & 63));
      }
    }

    std::vector<int> clique = solver.solve();
    for (int li : clique) {
      int gi = comp[li];
      chosen_active.push_back(gi);
      objective += c[active_to_old[gi]];
    }

    for (int gi : comp) g2l[gi] = -1;
  }

  for (int gi : chosen_active) {
    int old = active_to_old[gi];
    x[old] = 1;
  }

  std::cout << objective << "\n";
  for (int i = 0; i < n; ++i) {
    if (i) std::cout << ' ';
    std::cout << x[i];
  }
  std::cout << "\n";
  return 0;
}
