use std::io::{self, Read};

#[derive(Clone)]
struct Edge {
    to: usize,
    rev: usize,
    cap: i32,
}

struct Dinic {
    g: Vec<Vec<Edge>>,
    level: Vec<i32>,
    it: Vec<usize>,
}

impl Dinic {
    fn new(n: usize) -> Self {
        Self {
            g: vec![Vec::new(); n],
            level: vec![0; n],
            it: vec![0; n],
        }
    }

    fn add_edge(&mut self, fr: usize, to: usize, cap: i32) {
        let rev_to = self.g[to].len();
        let rev_fr = self.g[fr].len();
        self.g[fr].push(Edge { to, rev: rev_to, cap });
        self.g[to].push(Edge {
            to: fr,
            rev: rev_fr,
            cap: 0,
        });
    }

    fn bfs(&mut self, s: usize, t: usize) -> bool {
        self.level.fill(-1);
        let mut q = std::collections::VecDeque::new();
        self.level[s] = 0;
        q.push_back(s);
        while let Some(v) = q.pop_front() {
            for e in &self.g[v] {
                if e.cap > 0 && self.level[e.to] < 0 {
                    self.level[e.to] = self.level[v] + 1;
                    q.push_back(e.to);
                }
            }
        }
        self.level[t] >= 0
    }

    fn dfs(&mut self, v: usize, t: usize, f: i32) -> i32 {
        if v == t {
            return f;
        }
        while self.it[v] < self.g[v].len() {
            let i = self.it[v];
            let to = self.g[v][i].to;
            let cap = self.g[v][i].cap;
            if cap > 0 && self.level[v] + 1 == self.level[to] {
                let pushed = self.dfs(to, t, f.min(cap));
                if pushed > 0 {
                    let rev = self.g[v][i].rev;
                    self.g[v][i].cap -= pushed;
                    self.g[to][rev].cap += pushed;
                    return pushed;
                }
            }
            self.it[v] += 1;
        }
        0
    }

    fn max_flow(&mut self, s: usize, t: usize) -> i32 {
        let mut flow = 0;
        while self.bfs(s, t) {
            self.it.fill(0);
            loop {
                let pushed = self.dfs(s, t, i32::MAX / 4);
                if pushed == 0 {
                    break;
                }
                flow += pushed;
            }
        }
        flow
    }
}

struct Solver {
    n: usize,
    k: usize,
    adj: Vec<Vec<usize>>,
    edges: Vec<(usize, usize)>,
    deg: Vec<usize>,
    best_cut: i32,
    best_assign: Vec<i8>, // 0 = A, 1 = B
}

impl Solver {
    fn new(n: usize, edges: Vec<(usize, usize)>) -> Self {
        let mut adj = vec![Vec::<usize>::new(); n];
        for &(u, v) in &edges {
            if u == v {
                continue;
            }
            adj[u].push(v);
            adj[v].push(u);
        }
        let deg = adj.iter().map(|v| v.len()).collect::<Vec<_>>();
        Self {
            n,
            k: n / 2,
            adj,
            edges,
            deg,
            best_cut: i32::MAX / 4,
            best_assign: vec![-1; n],
        }
    }

    fn compute_cut(&self, assign: &[i8]) -> i32 {
        let mut cut = 0i32;
        for &(u, v) in &self.edges {
            if u == v {
                continue;
            }
            if assign[u] != assign[v] {
                cut += 1;
            }
        }
        cut
    }

    fn greedy_initial(&self) -> Vec<i8> {
        let n = self.n;
        let k = self.k;
        let mut in_a = vec![false; n];
        in_a[0] = true;
        let mut score = vec![0i32; n];
        for &u in &self.adj[0] {
            score[u] += 1;
        }
        let mut a_size = 1usize;
        while a_size < k {
            let mut best_v: Option<usize> = None;
            let mut best_score: i32 = i32::MIN;
            for v in 0..n {
                if in_a[v] {
                    continue;
                }
                // Prefer more connections to current A, then higher degree.
                let s = score[v] * 1024 + self.deg[v] as i32;
                if s > best_score {
                    best_score = s;
                    best_v = Some(v);
                }
            }
            let v = best_v.expect("n is even and >= 2");
            in_a[v] = true;
            a_size += 1;
            for &u in &self.adj[v] {
                if !in_a[u] {
                    score[u] += 1;
                }
            }
        }
        let mut assign = vec![1i8; n];
        for v in 0..n {
            if in_a[v] {
                assign[v] = 0;
            }
        }
        assign
    }

    fn local_improve_swaps(&self, mut assign: Vec<i8>) -> Vec<i8> {
        // Simple greedy pair-swap improvement (keeps bisection size intact).
        let n = self.n;
        loop {
            let mut d = vec![0i32; n]; // external - internal
            for v in 0..n {
                let side = assign[v];
                let mut ext = 0i32;
                let mut int = 0i32;
                for &u in &self.adj[v] {
                    if assign[u] == side {
                        int += 1;
                    } else {
                        ext += 1;
                    }
                }
                d[v] = ext - int;
            }

            let mut best_gain = 0i32;
            let mut best_pair: Option<(usize, usize)> = None;

            for a in 0..n {
                if assign[a] != 0 || a == 0 {
                    continue; // keep vertex 0 fixed in A for symmetry breaking
                }
                for b in 0..n {
                    if assign[b] != 1 {
                        continue;
                    }
                    let mut gain = d[a] + d[b];
                    // Subtract 2 if edge (a,b) exists.
                    for &u in &self.adj[a] {
                        if u == b {
                            gain -= 2;
                            break;
                        }
                    }
                    if gain > best_gain {
                        best_gain = gain;
                        best_pair = Some((a, b));
                    }
                }
            }

            if best_gain <= 0 {
                break;
            }
            if let Some((a, b)) = best_pair {
                assign[a] = 1;
                assign[b] = 0;
            } else {
                break;
            }
        }
        assign
    }

    fn flow_lower_bound(&self, assign: &[i8]) -> i32 {
        // Min s-t cut between forced A and forced B sets (ignores balance constraint).
        let n = self.n;
        let s = n;
        let t = n + 1;
        let mut dinic = Dinic::new(n + 2);
        for &(u, v) in &self.edges {
            if u == v {
                continue;
            }
            dinic.add_edge(u, v, 1);
            dinic.add_edge(v, u, 1);
        }
        let inf = self.edges.len() as i32 + 1;
        for v in 0..n {
            if assign[v] == 0 {
                dinic.add_edge(s, v, inf);
            } else if assign[v] == 1 {
                dinic.add_edge(v, t, inf);
            }
        }
        dinic.max_flow(s, t)
    }

    fn dfs(
        &mut self,
        assign: &mut Vec<i8>,
        cost_to_a: &mut Vec<i32>,
        cost_to_b: &mut Vec<i32>,
        count_a: usize,
        count_b: usize,
        current_cut: i32,
    ) {
        let n = self.n;
        let k = self.k;

        if current_cut >= self.best_cut {
            return;
        }
        if count_a > k || count_b > k {
            return;
        }
        let assigned = count_a + count_b;
        let rem = n - assigned;
        if count_a + rem < k || count_b + rem < k {
            return;
        }

        // If one side is full, the rest are forced.
        if count_a == k {
            let mut add = 0i32;
            for v in 0..n {
                if assign[v] == -1 {
                    add += cost_to_a[v];
                }
            }
            let total = current_cut + add;
            if total < self.best_cut {
                self.best_cut = total;
                let mut sol = assign.clone();
                for v in 0..n {
                    if sol[v] == -1 {
                        sol[v] = 1;
                    }
                }
                self.best_assign = sol;
            }
            return;
        }
        if count_b == k {
            let mut add = 0i32;
            for v in 0..n {
                if assign[v] == -1 {
                    add += cost_to_b[v];
                }
            }
            let total = current_cut + add;
            if total < self.best_cut {
                self.best_cut = total;
                let mut sol = assign.clone();
                for v in 0..n {
                    if sol[v] == -1 {
                        sol[v] = 0;
                    }
                }
                self.best_assign = sol;
            }
            return;
        }

        // Edge-to-assigned lower bound with exact size constraint.
        let a_need = k - count_a;
        let mut base = 0i32;
        let mut deltas: Vec<i32> = Vec::new();
        deltas.reserve(rem);
        for v in 0..n {
            if assign[v] == -1 {
                let ca = cost_to_a[v];
                let cb = cost_to_b[v];
                base += ca; // cost if v goes to B
                deltas.push(cb - ca); // extra cost if v goes to A instead of B
            }
        }
        deltas.sort_unstable();
        let mut add = base;
        for i in 0..a_need {
            add += deltas[i];
        }
        let mut lb = current_cut + add;
        if lb >= self.best_cut {
            return;
        }

        // Optional: tighten with flow bound only when close to pruning.
        if lb + 2 >= self.best_cut && count_a > 0 && count_b > 0 {
            let f_lb = self.flow_lower_bound(assign);
            if f_lb > lb {
                lb = f_lb;
                if lb >= self.best_cut {
                    return;
                }
            }
        }

        // Choose next unassigned vertex: prioritize those most connected to already-assigned vertices.
        let mut next_v: Option<usize> = None;
        let mut best_score = i32::MIN;
        for v in 0..n {
            if assign[v] == -1 {
                let sc = (cost_to_a[v] + cost_to_b[v]) * 1024 + self.deg[v] as i32;
                if sc > best_score {
                    best_score = sc;
                    next_v = Some(v);
                }
            }
        }
        let v = match next_v {
            Some(v) => v,
            None => return, // should not happen
        };

        let inc_a = cost_to_b[v];
        let inc_b = cost_to_a[v];
        let first_side = if inc_a <= inc_b { 0i8 } else { 1i8 };
        let second_side = 1i8 - first_side;

        for &side in &[first_side, second_side] {
            if side == 0 && count_a == k {
                continue;
            }
            if side == 1 && count_b == k {
                continue;
            }

            assign[v] = side;
            let mut changed: Vec<usize> = Vec::new();
            let new_cut = if side == 0 {
                for &u in &self.adj[v] {
                    if assign[u] == -1 {
                        cost_to_a[u] += 1;
                        changed.push(u);
                    }
                }
                current_cut + inc_a
            } else {
                for &u in &self.adj[v] {
                    if assign[u] == -1 {
                        cost_to_b[u] += 1;
                        changed.push(u);
                    }
                }
                current_cut + inc_b
            };

            match side {
                0 => self.dfs(
                    assign,
                    cost_to_a,
                    cost_to_b,
                    count_a + 1,
                    count_b,
                    new_cut,
                ),
                1 => self.dfs(
                    assign,
                    cost_to_a,
                    cost_to_b,
                    count_a,
                    count_b + 1,
                    new_cut,
                ),
                _ => {}
            }

            if side == 0 {
                for u in changed {
                    cost_to_a[u] -= 1;
                }
            } else {
                for u in changed {
                    cost_to_b[u] -= 1;
                }
            }
            assign[v] = -1;
        }
    }

    fn solve(&mut self) {
        // Initial upper bound from a heuristic.
        let init = self.greedy_initial();
        let init = self.local_improve_swaps(init);
        let init_cut = self.compute_cut(&init);
        self.best_cut = init_cut;
        self.best_assign = init;

        // Exact search (branch and bound).
        let n = self.n;
        let mut assign = vec![-1i8; n];
        let mut cost_to_a = vec![0i32; n];
        let mut cost_to_b = vec![0i32; n];
        let mut count_a = 0usize;
        let count_b = 0usize;

        // Fix vertex 0 in A (symmetry breaking).
        assign[0] = 0;
        count_a += 1;
        for &u in &self.adj[0] {
            cost_to_a[u] += 1;
        }

        self.dfs(
            &mut assign,
            &mut cost_to_a,
            &mut cost_to_b,
            count_a,
            count_b,
            0,
        );
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u < n && v < n && u != v {
            edges.push((u, v));
        }
    }

    if n == 0 {
        return;
    }
    if n % 2 != 0 {
        // Problem statement guarantees even n, but keep output valid if not.
        println!("0\n");
        return;
    }

    let mut solver = Solver::new(n, edges);
    solver.solve();

    let cut = solver.compute_cut(&solver.best_assign);
    println!("{cut}");
    let mut a_vertices: Vec<usize> = (0..n).filter(|&v| solver.best_assign[v] == 0).collect();
    a_vertices.sort_unstable();
    for (i, v) in a_vertices.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{v}");
    }
    println!();
}
