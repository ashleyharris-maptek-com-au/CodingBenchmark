#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(std430, set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(std430, set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(std140, set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

shared vec4 shPosMass[256];

void main() {
    const uint TILE_SIZE = 256u;

    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float soft2 = softening * softening;

    bool active = (i < N);

    vec3 pos_i = vec3(0.0);
    vec3 vel_i = vec3(0.0);
    float mass_i = 0.0;

    if (active) {
        uint idx = i * 2u;
        vec4 posMass = inData[idx];
        vec4 vel4 = inData[idx + 1u];
        pos_i = posMass.xyz;
        mass_i = posMass.w;
        vel_i = vel4.xyz;
    }

    vec3 acc = vec3(0.0);

    for (uint tileBase = 0u; tileBase < N; tileBase += TILE_SIZE) {
        uint jGlobal = tileBase + gl_LocalInvocationID.x;
        shPosMass[gl_LocalInvocationID.x] = (jGlobal < N) ? inData[jGlobal * 2u] : vec4(0.0);

        barrier();

        if (active) {
            uint tileCount = min(TILE_SIZE, N - tileBase);
            for (uint j = 0u; j < tileCount; ++j) {
                uint jIndex = tileBase + j;
                if (jIndex == i) {
                    continue;
                }

                vec3 r = shPosMass[j].xyz - pos_i;
                float dist2 = dot(r, r) + soft2;
                float invDist = inversesqrt(dist2);
                float invDist3 = invDist * invDist * invDist;
                acc += (G * shPosMass[j].w) * (r * invDist3);
            }
        }

        barrier();
    }

    if (!active) {
        return;
    }

    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_new * dt;

    uint outIdx = i * 2u;
    outData[outIdx] = vec4(pos_new, mass_i);
    outData[outIdx + 1u] = vec4(vel_new, 0.0);
}
