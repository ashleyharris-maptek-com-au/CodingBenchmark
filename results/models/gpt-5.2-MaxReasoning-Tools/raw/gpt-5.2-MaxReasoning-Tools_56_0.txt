'use strict';

const fs = require('fs');

function decodeStdin(buf) {
  if (!buf || buf.length === 0) return '';

  // BOMs
  if (buf.length >= 3 && buf[0] === 0xef && buf[1] === 0xbb && buf[2] === 0xbf) {
    return buf.slice(3).toString('utf8');
  }
  if (buf.length >= 2 && buf[0] === 0xff && buf[1] === 0xfe) {
    return buf.slice(2).toString('utf16le');
  }
  if (buf.length >= 2 && buf[0] === 0xfe && buf[1] === 0xff) {
    const copy = Buffer.from(buf.slice(2));
    if (copy.length >= 2) copy.swap16();
    return copy.toString('utf16le');
  }

  // Heuristic UTF-16 detection (common when piping text from Windows PowerShell)
  const sample = Math.min(buf.length, 64);
  let zeroEven = 0;
  let zeroOdd = 0;
  for (let i = 0; i < sample; i++) {
    if (buf[i] === 0) {
      if (i & 1) zeroOdd++;
      else zeroEven++;
    }
  }

  const fracEven = zeroEven / sample;
  const fracOdd = zeroOdd / sample;
  if (fracOdd >= 0.25 && zeroOdd >= zeroEven * 2) {
    return buf.toString('utf16le');
  }
  if (fracEven >= 0.25 && zeroEven >= zeroOdd * 2) {
    const copy = Buffer.from(buf);
    if (copy.length >= 2) copy.swap16();
    return copy.toString('utf16le');
  }

  return buf.toString('utf8');
}

const stdinBuf = fs.readFileSync(0);
const input = decodeStdin(stdinBuf).replace(/^\uFEFF/, '');

let data;
try {
  data = JSON.parse(input);
} catch {
  process.stdout.write('{"routes":[]}');
  process.exit(0);
}

function dist(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.sqrt(dx * dx + dy * dy);
}

function part1by1(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n;
}

function morton16(x, y) {
  return (part1by1(x) | (part1by1(y) << 1)) >>> 0;
}

function clampInt(v, lo, hi) {
  v = Math.floor(v);
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

function chooseKCandidates(kMax, n) {
  const set = new Set();
  const add = (v) => set.add(clampInt(v, 1, kMax));
  add(kMax);

  if (n <= 5000) {
    add(Math.round(kMax * 0.75));
    add(Math.round(kMax * 0.5));
    add(Math.round(kMax * 0.33));
    add(Math.round(kMax * 0.25));
    add(Math.round(Math.sqrt(kMax)));
  } else if (n <= 20000) {
    add(Math.round(kMax * 0.67));
    add(Math.round(kMax * 0.4));
    add(Math.round(Math.sqrt(kMax)));
  } else {
    add(Math.round(kMax * 0.5));
    add(Math.round(Math.sqrt(kMax)));
  }

  add(1);
  return Array.from(set).sort((a, b) => a - b);
}

function estimateCost(order, xs, ys, depotX, depotY, k, shift) {
  const n = order.length;
  if (n === 0) return 0;
  const base = (n / k) | 0;
  const rem = n - base * k;

  let idx = shift % n;
  let total = 0;

  for (let r = 0; r < k; r++) {
    const size = base + (r < rem ? 1 : 0);
    if (size <= 0) continue;

    const firstIdx = order[idx];
    let prevX = xs[firstIdx];
    let prevY = ys[firstIdx];
    total += dist(depotX, depotY, prevX, prevY);

    for (let t = 1; t < size; t++) {
      const curIdx = order[(idx + t) % n];
      const curX = xs[curIdx];
      const curY = ys[curIdx];
      total += dist(prevX, prevY, curX, curY);
      prevX = curX;
      prevY = curY;
    }

    total += dist(prevX, prevY, depotX, depotY);
    idx = (idx + size) % n;
  }

  return total;
}

function findBestShift(order, xs, ys, depotX, depotY, k) {
  const n = order.length;
  if (n === 0) return 0;

  let samples = 32;
  if (n > 20000) samples = 8;
  else if (n > 5000) samples = 16;
  if (samples > n) samples = n;

  let bestShift = 0;
  let bestCost = Infinity;

  for (let s = 0; s < samples; s++) {
    const shift = ((s * n) / samples) | 0;
    const cost = estimateCost(order, xs, ys, depotX, depotY, k, shift);
    if (cost < bestCost) {
      bestCost = cost;
      bestShift = shift;
    }
  }

  return bestShift;
}

function partitionOrder(order, k, shift) {
  const n = order.length;
  const base = (n / k) | 0;
  const rem = n - base * k;

  const routes = new Array(k);
  let idx = shift % n;

  for (let r = 0; r < k; r++) {
    const size = base + (r < rem ? 1 : 0);
    const route = new Array(size);
    for (let t = 0; t < size; t++) {
      route[t] = order[(idx + t) % n];
    }
    routes[r] = route;
    idx = (idx + size) % n;
  }

  return routes;
}

function reverseSegment(arr, i, j) {
  while (i < j) {
    const tmp = arr[i];
    arr[i] = arr[j];
    arr[j] = tmp;
    i++;
    j--;
  }
}

function twoOpt(route, xs, ys, depotX, depotY, window, maxPasses) {
  const m = route.length;
  if (m < 4) return;

  const eps = 1e-12;
  const maxW = Math.max(2, Math.min(window, m - 1));

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < m - 1; i++) {
      const aIdx = i === 0 ? -1 : route[i - 1];
      const bIdx = route[i];
      const ax = aIdx === -1 ? depotX : xs[aIdx];
      const ay = aIdx === -1 ? depotY : ys[aIdx];
      const bx = xs[bIdx];
      const by = ys[bIdx];

      const maxJ = Math.min(m - 1, i + maxW);
      for (let j = i + 1; j <= maxJ; j++) {
        const cIdx = route[j];
        const dIdx = j === m - 1 ? -1 : route[j + 1];
        const cx = xs[cIdx];
        const cy = ys[cIdx];
        const dx = dIdx === -1 ? depotX : xs[dIdx];
        const dy = dIdx === -1 ? depotY : ys[dIdx];

        const cur = dist(ax, ay, bx, by) + dist(cx, cy, dx, dy);
        const neu = dist(ax, ay, cx, cy) + dist(bx, by, dx, dy);
        if (neu + eps < cur) {
          reverseSegment(route, i, j);
          improved = true;
        }
      }
    }

    if (!improved) break;
  }
}

function improveRoutes(routes, xs, ys, depotX, depotY) {
  for (let r = 0; r < routes.length; r++) {
    const route = routes[r];
    const m = route.length;
    if (m < 4) continue;

    let window;
    let passes;
    if (m <= 180) {
      window = m - 1;
      passes = 3;
    } else if (m <= 450) {
      window = m - 1;
      passes = 2;
    } else if (m <= 2000) {
      window = 60;
      passes = 2;
    } else if (m <= 8000) {
      window = 30;
      passes = 1;
    } else {
      continue;
    }

    twoOpt(route, xs, ys, depotX, depotY, window, passes);
  }
}

function buildOrderAngle(xs, ys, ids, depotX, depotY) {
  const n = ids.length;
  const angles = new Float64Array(n);
  const dist2 = new Float64Array(n);
  const typeCode = new Uint8Array(n);
  const idNum = new Float64Array(n);
  const idStr = new Array(n);

  for (let i = 0; i < n; i++) {
    const dx = xs[i] - depotX;
    const dy = ys[i] - depotY;
    angles[i] = Math.atan2(dy, dx);
    dist2[i] = dx * dx + dy * dy;

    const v = ids[i];
    const t = typeof v;
    if (t === 'number') {
      typeCode[i] = 0;
      idNum[i] = Number.isFinite(v) ? v : 0;
      idStr[i] = '';
    } else if (t === 'string') {
      typeCode[i] = 1;
      idNum[i] = 0;
      idStr[i] = v;
    } else {
      typeCode[i] = 2;
      idNum[i] = 0;
      idStr[i] = String(v);
    }
  }

  const order = new Array(n);
  for (let i = 0; i < n; i++) order[i] = i;

  order.sort((a, b) => {
    const aa = angles[a];
    const ab = angles[b];
    if (aa < ab) return -1;
    if (aa > ab) return 1;
    const da = dist2[a];
    const db = dist2[b];
    if (da < db) return -1;
    if (da > db) return 1;
    const ta = typeCode[a];
    const tb = typeCode[b];
    if (ta !== tb) return ta - tb;
    if (ta === 0) return idNum[a] - idNum[b];
    const sa = idStr[a];
    const sb = idStr[b];
    if (sa < sb) return -1;
    if (sa > sb) return 1;
    return 0;
  });

  return order;
}

function buildOrderMorton(xs, ys, ids, depotX, depotY) {
  const n = ids.length;
  let minX = xs[0],
    maxX = xs[0],
    minY = ys[0],
    maxY = ys[0];

  for (let i = 1; i < n; i++) {
    const x = xs[i];
    const y = ys[i];
    if (x < minX) minX = x;
    else if (x > maxX) maxX = x;
    if (y < minY) minY = y;
    else if (y > maxY) maxY = y;
  }

  const rangeX = maxX - minX;
  const rangeY = maxY - minY;
  const invX = rangeX > 0 ? 65535 / rangeX : 0;
  const invY = rangeY > 0 ? 65535 / rangeY : 0;

  const keys = new Uint32Array(n);
  const dist2 = new Float64Array(n);
  const typeCode = new Uint8Array(n);
  const idNum = new Float64Array(n);
  const idStr = new Array(n);

  for (let i = 0; i < n; i++) {
    const nx = rangeX > 0 ? clampInt((xs[i] - minX) * invX, 0, 65535) : 0;
    const ny = rangeY > 0 ? clampInt((ys[i] - minY) * invY, 0, 65535) : 0;
    keys[i] = morton16(nx, ny);

    const dx = xs[i] - depotX;
    const dy = ys[i] - depotY;
    dist2[i] = dx * dx + dy * dy;

    const v = ids[i];
    const t = typeof v;
    if (t === 'number') {
      typeCode[i] = 0;
      idNum[i] = Number.isFinite(v) ? v : 0;
      idStr[i] = '';
    } else if (t === 'string') {
      typeCode[i] = 1;
      idNum[i] = 0;
      idStr[i] = v;
    } else {
      typeCode[i] = 2;
      idNum[i] = 0;
      idStr[i] = String(v);
    }
  }

  const order = new Array(n);
  for (let i = 0; i < n; i++) order[i] = i;

  order.sort((a, b) => {
    const ka = keys[a];
    const kb = keys[b];
    if (ka !== kb) return ka - kb;
    const da = dist2[a];
    const db = dist2[b];
    if (da < db) return -1;
    if (da > db) return 1;
    const ta = typeCode[a];
    const tb = typeCode[b];
    if (ta !== tb) return ta - tb;
    if (ta === 0) return idNum[a] - idNum[b];
    const sa = idStr[a];
    const sb = idStr[b];
    if (sa < sb) return -1;
    if (sa > sb) return 1;
    return 0;
  });

  return order;
}

function solve(instance) {
  const depot = Array.isArray(instance.depot) ? instance.depot : [0, 0];
  const depotX = Number.isFinite(+depot[0]) ? +depot[0] : 0;
  const depotY = Number.isFinite(+depot[1]) ? +depot[1] : 0;

  const packages = Array.isArray(instance.packages) ? instance.packages : [];
  const n = packages.length;
  if (n === 0) return { routes: [] };

  let truckCount = Number.isFinite(+instance.truckCount) ? Math.floor(+instance.truckCount) : 1;
  if (truckCount <= 0) truckCount = 1;
  const kMax = Math.min(truckCount, n);

  const ids = new Array(n);
  const xs = new Float64Array(n);
  const ys = new Float64Array(n);
  for (let i = 0; i < n; i++) {
    const p = packages[i] || {};
    ids[i] = p.id;
    xs[i] = Number.isFinite(+p.x) ? +p.x : 0;
    ys[i] = Number.isFinite(+p.y) ? +p.y : 0;
  }

  const kCandidates = chooseKCandidates(kMax, n);
  const methods = [
    { name: 'angle', build: buildOrderAngle },
    { name: 'morton', build: buildOrderMorton },
  ];

  let best = null;

  for (let mi = 0; mi < methods.length; mi++) {
    const order = methods[mi].build(xs, ys, ids, depotX, depotY);
    for (let ki = 0; ki < kCandidates.length; ki++) {
      const k = kCandidates[ki];
      const shift = findBestShift(order, xs, ys, depotX, depotY, k);
      const cost = estimateCost(order, xs, ys, depotX, depotY, k, shift);
      if (!best || cost < best.cost) {
        best = { order, k, shift, cost };
      }
    }
  }

  const routesIdx = partitionOrder(best.order, best.k, best.shift);
  improveRoutes(routesIdx, xs, ys, depotX, depotY);

  const routes = new Array(routesIdx.length);
  for (let r = 0; r < routesIdx.length; r++) {
    const routeIdx = routesIdx[r];
    const out = new Array(routeIdx.length);
    for (let i = 0; i < routeIdx.length; i++) out[i] = ids[routeIdx[i]];
    routes[r] = out;
  }

  return { routes };
}

const result = solve(data);
process.stdout.write(JSON.stringify(result));
