// 2D Cutting Stock / Rectangle Bin Packing solver
// Heuristic, scalable: multi-start for small N; Next-k-fit + Skyline for large N.
// C++20, MSVC-friendly (no non-standard headers).

#include <algorithm>
#include <chrono>
#include <cstddef>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using std::int64_t;

struct Rect {
    int w = 0;
    int h = 0;
    int idx = 0;
    int64_t area = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct SkyPoint {
    int x = 0;
    int y = 0;
};

class SkylineBin {
public:
    SkylineBin(int width, int height) : W(width), H(height) {
        skyline.reserve(64);
        skyline.push_back({0, 0});
        skyline.push_back({W, 0}); // sentinel
    }

    struct Fit {
        bool ok = false;
        int x = 0;
        int y = 0;
        int w = 0;
        int h = 0;
        int rotated = 0;
        int score_y = 0;            // lower is better
        int64_t score_waste = 0;     // lower is better
    };

    Fit findBestForRect(int w, int h) const {
        Fit best;
        best.ok = false;

        Fit a = findBestFixed(w, h);
        if (a.ok) best = a;

        if (w != h) {
            Fit b = findBestFixed(h, w);
            if (b.ok) {
                b.rotated = 1;
                if (!best.ok || isBetter(b, best)) best = b;
            }
        }
        return best;
    }

    void place(const Fit& f) {
        // Precondition: f.ok and f within bounds.
        usedArea += int64_t(f.w) * int64_t(f.h);
        addSkyline(f.x, f.y, f.w, f.h);
    }

    int64_t getUsedArea() const { return usedArea; }

private:
    int W = 0, H = 0;
    std::vector<SkyPoint> skyline; // sorted by x, last is sentinel at x=W
    int64_t usedArea = 0;

    static bool isBetter(const Fit& a, const Fit& b) {
        if (a.score_y != b.score_y) return a.score_y < b.score_y;
        if (a.score_waste != b.score_waste) return a.score_waste < b.score_waste;
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    }

    Fit findBestFixed(int w, int h) const {
        Fit best;
        best.ok = false;

        if (w <= 0 || h <= 0) return best;
        if (w > W || h > H) return best;

        const int INF = std::numeric_limits<int>::max();
        best.score_y = INF;
        best.score_waste = std::numeric_limits<int64_t>::max();
        best.x = INF;
        best.y = INF;

        // Try placing at each skyline point as left edge.
        for (size_t i = 0; i + 1 < skyline.size(); ++i) {
            const int x0 = skyline[i].x;
            if (x0 + w > W) break; // increasing x0 => no further positions fit

            int y = 0;
            int64_t waste = 0;
            if (!computeYAndWaste(i, x0, w, h, y, waste)) continue;

            Fit cand;
            cand.ok = true;
            cand.x = x0;
            cand.y = y;
            cand.w = w;
            cand.h = h;
            cand.rotated = 0;
            cand.score_y = y;
            cand.score_waste = waste;

            if (!best.ok || isBetter(cand, best)) best = cand;
        }

        return best;
    }

    bool computeYAndWaste(size_t startIdx, int x0, int w, int h, int& outY, int64_t& outWaste) const {
        const int x1 = x0 + w;

        int maxY = skyline[startIdx].y;
        for (size_t j = startIdx + 1; j < skyline.size() && skyline[j].x < x1; ++j) {
            if (skyline[j].y > maxY) maxY = skyline[j].y;
            if (maxY + h > H) return false;
        }
        if (maxY + h > H) return false;

        int64_t waste = 0;
        for (size_t j = startIdx; j + 1 < skyline.size() && skyline[j].x < x1; ++j) {
            const int segStart = skyline[j].x;
            const int segEnd = std::min(skyline[j + 1].x, x1);
            const int segW = segEnd - segStart;
            if (segW > 0) {
                waste += int64_t(segW) * int64_t(maxY - skyline[j].y);
            }
        }

        outY = maxY;
        outWaste = waste;
        return true;
    }

    void addSkyline(int x0, int y0, int w, int h) {
        const int x1 = x0 + w;
        const int newY = y0 + h;

        auto lbX = [&](int x) {
            return std::lower_bound(
                skyline.begin(), skyline.end(), x,
                [](const SkyPoint& p, int v) { return p.x < v; });
        };

        auto itStart = lbX(x0);
        if (itStart == skyline.end() || itStart->x != x0) {
            // Should not happen (we only place at existing skyline points), but be robust.
            int prevY = (itStart == skyline.begin()) ? 0 : (itStart - 1)->y;
            itStart = skyline.insert(itStart, {x0, prevY});
        }

        auto itEnd = lbX(x1);
        if (itEnd == skyline.end() || itEnd->x != x1) {
            int prevY = (itEnd == skyline.begin()) ? 0 : (itEnd - 1)->y;
            itEnd = skyline.insert(itEnd, {x1, prevY});
        }

        // Recompute iterators after possible insertions.
        itStart = lbX(x0);
        itEnd = lbX(x1);

        // Raise segment starting at x0.
        itStart->y = newY;

        // Remove points strictly inside (x0, x1).
        if (itStart + 1 < itEnd) {
            skyline.erase(itStart + 1, itEnd);
        }

        // Merge consecutive points with same height.
        for (size_t i = 1; i < skyline.size();) {
            if (skyline[i].y == skyline[i - 1].y) {
                skyline.erase(skyline.begin() + static_cast<std::ptrdiff_t>(i));
            } else {
                ++i;
            }
        }

        // Ensure sentinel is present.
        if (skyline.empty() || skyline.back().x != W) {
            skyline.push_back({W, 0});
        }
    }
};

struct Solution {
    int boards = 0;
    std::vector<Placement> placements;
};

struct PackParams {
    int binW = 0;
    int binH = 0;
    int64_t binArea = 0;
    int n = 0;
    int windowK = 0;
    int extraSamples = 0;
    uint32_t sampleSeed = 123456789u;
};

static std::vector<int> buildOrder(const std::vector<Rect>& rects, int mode) {
    std::vector<int> order(rects.size());
    std::iota(order.begin(), order.end(), 0);

    auto cmp_area = [&](int a, int b) {
        const Rect& ra = rects[a];
        const Rect& rb = rects[b];
        if (ra.area != rb.area) return ra.area > rb.area;
        const int aMax = std::max(ra.w, ra.h), bMax = std::max(rb.w, rb.h);
        if (aMax != bMax) return aMax > bMax;
        const int aMin = std::min(ra.w, ra.h), bMin = std::min(rb.w, rb.h);
        if (aMin != bMin) return aMin > bMin;
        return ra.idx < rb.idx;
    };

    auto cmp_maxside = [&](int a, int b) {
        const Rect& ra = rects[a];
        const Rect& rb = rects[b];
        const int aMax = std::max(ra.w, ra.h), bMax = std::max(rb.w, rb.h);
        if (aMax != bMax) return aMax > bMax;
        const int aMin = std::min(ra.w, ra.h), bMin = std::min(rb.w, rb.h);
        if (aMin != bMin) return aMin > bMin;
        if (ra.area != rb.area) return ra.area > rb.area;
        return ra.idx < rb.idx;
    };

    auto cmp_perimeter = [&](int a, int b) {
        const Rect& ra = rects[a];
        const Rect& rb = rects[b];
        const int pa = ra.w + ra.h;
        const int pb = rb.w + rb.h;
        if (pa != pb) return pa > pb;
        if (ra.area != rb.area) return ra.area > rb.area;
        const int aMax = std::max(ra.w, ra.h), bMax = std::max(rb.w, rb.h);
        if (aMax != bMax) return aMax > bMax;
        return ra.idx < rb.idx;
    };

    if (mode == 0) {
        std::sort(order.begin(), order.end(), cmp_area);
    } else if (mode == 1) {
        std::sort(order.begin(), order.end(), cmp_maxside);
    } else {
        std::sort(order.begin(), order.end(), cmp_perimeter);
    }

    return order;
}

static Solution packWithOrder(const std::vector<Rect>& rects, const std::vector<int>& order, const PackParams& params) {
    Solution sol;
    sol.placements.assign(rects.size(), Placement{});

    std::vector<SkylineBin> bins;
    bins.reserve(64);

    auto ensureBin = [&]() -> int {
        bins.emplace_back(params.binW, params.binH);
        return static_cast<int>(bins.size() - 1);
    };

    ensureBin();

    std::vector<int> candBins;
    candBins.reserve(static_cast<size_t>(std::max(64, params.windowK + params.extraSamples + 8)));

    for (int ord = 0; ord < static_cast<int>(order.size()); ++ord) {
        const Rect& r = rects[order[ord]];

        SkylineBin::Fit bestFit;
        int bestBin = -1;

        // Build candidate bin list (reuses buffer for speed).
        candBins.clear();
        const int bsz = static_cast<int>(bins.size());
        if (params.n <= 5000 || bsz <= params.windowK) {
            candBins.resize(bsz);
            std::iota(candBins.begin(), candBins.end(), 0);
        } else {
            const int start = std::max(0, bsz - params.windowK);
            for (int i = bsz - 1; i >= start; --i) candBins.push_back(i);

            const int samples = std::min(params.extraSamples, bsz - static_cast<int>(candBins.size()));
            if (samples > 0 && start > 0) {
                uint32_t x = params.sampleSeed ^ (static_cast<uint32_t>(ord) * 2654435761u);
                for (int s = 0; s < samples; ++s) {
                    x ^= x << 13;
                    x ^= x >> 17;
                    x ^= x << 5;
                    int idx = static_cast<int>(x % static_cast<uint32_t>(start));
                    bool dup = false;
                    for (int v : candBins) {
                        if (v == idx) {
                            dup = true;
                            break;
                        }
                    }
                    if (!dup) candBins.push_back(idx);
                }
            }
        }

        for (int bi : candBins) {
            SkylineBin::Fit f = bins[bi].findBestForRect(r.w, r.h);
            if (!f.ok) continue;

            // Slight bias towards tighter bins when placements are otherwise identical.
            // This helps reduce board count without scanning every bin.
            int64_t freeAfter = params.binArea - (bins[bi].getUsedArea() + r.area);

            if (bestBin < 0) {
                bestFit = f;
                bestBin = bi;
            } else {
                bool better = false;
                if (f.score_y != bestFit.score_y) better = f.score_y < bestFit.score_y;
                else if (f.score_waste != bestFit.score_waste) better = f.score_waste < bestFit.score_waste;
                else if (f.x != bestFit.x) better = f.x < bestFit.x;
                else if (f.y != bestFit.y) better = f.y < bestFit.y;
                else {
                    int64_t bestFreeAfter = params.binArea - (bins[bestBin].getUsedArea() + r.area);
                    better = freeAfter < bestFreeAfter;
                }
                if (better) {
                    bestFit = f;
                    bestBin = bi;
                }
            }
        }

        if (bestBin < 0) {
            bestBin = ensureBin();
            bestFit = bins[bestBin].findBestForRect(r.w, r.h);
            if (!bestFit.ok) {
                // Impossible to place even in empty bin => no valid solution.
                // Emit a degenerate output (0 boards) later by setting boards=0.
                sol.boards = 0;
                return sol;
            }
        }

        bins[bestBin].place(bestFit);
        Placement p;
        p.board = bestBin;
        p.x = bestFit.x;
        p.y = bestFit.y;
        p.rotated = bestFit.rotated;
        sol.placements[r.idx] = p;
    }

    sol.boards = static_cast<int>(bins.size());
    return sol;
}

static PackParams makeParams(int n, int binW, int binH) {
    PackParams p;
    p.n = n;
    p.binW = binW;
    p.binH = binH;
    p.binArea = int64_t(binW) * int64_t(binH);

    if (n <= 5000) {
        // We still scan all bins for these sizes, but keep windowK bounded
        // so we don't accidentally reserve huge buffers.
        p.windowK = 64;
        p.extraSamples = 0;
    } else if (n <= 20000) {
        p.windowK = 32;
        p.extraSamples = 8;
    } else {
        p.windowK = 12;
        p.extraSamples = 6;
    }
    return p;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0, BW = 0, BH = 0;
    if (!(std::cin >> N >> BW >> BH)) return 0;

    std::vector<Rect> rects;
    rects.reserve(std::max(0, N));

    for (int i = 0; i < N; ++i) {
        int w = 0, h = 0;
        std::cin >> w >> h;
        Rect r;
        r.w = w;
        r.h = h;
        r.idx = i;
        r.area = int64_t(w) * int64_t(h);
        rects.push_back(r);
    }

    // Quick feasibility check: each rectangle must fit in the board in at least one orientation.
    for (const auto& r : rects) {
        const bool ok = (r.w <= BW && r.h <= BH) || (r.h <= BW && r.w <= BH);
        if (!ok) {
            std::cout << 0 << "\n";
            for (int i = 0; i < N; ++i) std::cout << -1 << " 0 0 0\n";
            return 0;
        }
    }

    const PackParams params = makeParams(N, BW, BH);

    // Multi-start ordering for smaller cases; single pass for large N.
    std::vector<int> modes;
    if (N <= 2000) modes = {0, 1, 2};
    else if (N <= 5000) modes = {0, 1};
    else modes = {0};

    Solution bestSol;
    bestSol.boards = std::numeric_limits<int>::max();

    for (int mode : modes) {
        std::vector<int> order = buildOrder(rects, mode);
        Solution sol = packWithOrder(rects, order, params);
        if (sol.boards == 0) continue; // failed pack

        if (sol.boards < bestSol.boards) {
            bestSol = std::move(sol);
        }
    }

    if (bestSol.boards == std::numeric_limits<int>::max()) {
        std::cout << 0 << "\n";
        for (int i = 0; i < N; ++i) std::cout << -1 << " 0 0 0\n";
        return 0;
    }

    std::cout << bestSol.boards << "\n";
    for (int i = 0; i < N; ++i) {
        const auto& p = bestSol.placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << "\n";
    }

    return 0;
}
