using System;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        if (!fs.TryNextInt(out int n))
            return;

        double mu = fs.NextDouble();

        var startState = ReadState(fs);
        Orbit startOrbit = Orbit.FromState(startState.R, startState.V, mu);

        Orbit[] stations = new Orbit[n];
        for (int i = 0; i < n; i++)
        {
            var st = ReadState(fs);
            stations[i] = Orbit.FromState(st.R, st.V, mu);
        }

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        double[] startCost = new double[n];
        double[,] cost = new double[n, n];

        for (int i = 0; i < n; i++)
            startCost[i] = OrbitCost(startOrbit, stations[i], mu);

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                double c = OrbitCost(stations[i], stations[j], mu);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] route = n <= 17
            ? SolveExactDp(n, startCost, cost)
            : SolveHeuristic(n, startCost, cost);

        var sb = new StringBuilder(capacity: n * 4);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private readonly struct State
    {
        public readonly Vec3 R;
        public readonly Vec3 V;
        public State(Vec3 r, Vec3 v) { R = r; V = v; }
    }

    private static State ReadState(FastScanner fs)
    {
        double x = fs.NextDouble();
        double y = fs.NextDouble();
        double z = fs.NextDouble();
        double vx = fs.NextDouble();
        double vy = fs.NextDouble();
        double vz = fs.NextDouble();
        return new State(new Vec3(x, y, z), new Vec3(vx, vy, vz));
    }

    private static int[] SolveExactDp(int n, double[] startCost, double[,] cost)
    {
        int all = (1 << n) - 1;
        int states = 1 << n;

        double[] dp = new double[states * n];
        int[] parent = new int[states * n];
        for (int i = 0; i < dp.Length; i++)
        {
            dp[i] = double.PositiveInfinity;
            parent[i] = -1;
        }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask * n + i] = startCost[i];
        }

        for (int mask = 1; mask <= all; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if (((mask >> last) & 1) == 0) continue;
                double cur = dp[mask * n + last];
                if (double.IsPositiveInfinity(cur)) continue;

                int remaining = all ^ mask;
                while (remaining != 0)
                {
                    int bit = remaining & -remaining;
                    int next = BitOperations.TrailingZeroCount((uint)bit);
                    int nmask = mask | bit;

                    double cand = cur + cost[last, next];
                    int idx = nmask * n + next;
                    if (cand < dp[idx])
                    {
                        dp[idx] = cand;
                        parent[idx] = last;
                    }
                    remaining ^= bit;
                }
            }
        }

        double best = double.PositiveInfinity;
        int bestLast = 0;
        for (int last = 0; last < n; last++)
        {
            double v = dp[all * n + last];
            if (v < best)
            {
                best = v;
                bestLast = last;
            }
        }

        int[] route = new int[n];
        int curMask = all;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int idx = curMask * n + curLast;
            int p = parent[idx];
            curMask ^= 1 << curLast;
            curLast = p < 0 ? 0 : p;
        }

        return route;
    }

    private static int[] SolveHeuristic(int n, double[] startCost, double[,] cost)
    {
        int multiStarts = n <= 30 ? 12 : (n <= 60 ? 8 : 5);
        int[] candidates = TopKIndices(startCost, Math.Min(n, multiStarts));

        double bestCost = double.PositiveInfinity;
        int[] bestRoute = new int[n];

        for (int s = 0; s < candidates.Length; s++)
        {
            int seed = candidates[s];
            int[] route = BuildBestEndRoute(n, seed, startCost, cost);
            TwoOptImprove(route, startCost, cost, maxPasses: n <= 60 ? 50 : 30);
            double c = RouteCost(route, startCost, cost);
            if (c < bestCost)
            {
                bestCost = c;
                Array.Copy(route, bestRoute, n);
            }
        }

        return bestRoute;
    }

    private static int[] BuildBestEndRoute(int n, int seed, double[] startCost, double[,] cost)
    {
        bool[] used = new bool[n];
        used[seed] = true;

        int[] deque = new int[n * 2 + 4];
        int head = n;
        int tail = n;
        deque[tail++] = seed;
        int count = 1;

        while (count < n)
        {
            int front = deque[head];
            int back = deque[tail - 1];
            double startToFront = startCost[front];

            double bestInc = double.PositiveInfinity;
            int bestNode = -1;
            bool bestToFront = false;

            for (int c = 0; c < n; c++)
            {
                if (used[c]) continue;

                double incFront = startCost[c] + cost[c, front] - startToFront;
                if (incFront < bestInc)
                {
                    bestInc = incFront;
                    bestNode = c;
                    bestToFront = true;
                }

                double incBack = cost[back, c];
                if (incBack < bestInc)
                {
                    bestInc = incBack;
                    bestNode = c;
                    bestToFront = false;
                }
            }

            used[bestNode] = true;
            if (bestToFront)
                deque[--head] = bestNode;
            else
                deque[tail++] = bestNode;
            count++;
        }

        int[] route = new int[n];
        for (int i = 0; i < n; i++)
            route[i] = deque[head + i];
        return route;
    }

    private static void TwoOptImprove(int[] route, double[] startCost, double[,] cost, int maxPasses)
    {
        int n = route.Length;
        if (n <= 3) return;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                int aPrev = i == 0 ? -1 : route[i - 1];
                int a = route[i];

                for (int j = i + 1; j < n; j++)
                {
                    int b = route[j];
                    int bNext = j == n - 1 ? -1 : route[j + 1];

                    double oldCost = EdgeCost(aPrev, a, startCost, cost) + EdgeCost(b, bNext, startCost, cost);
                    double newCost = EdgeCost(aPrev, b, startCost, cost) + EdgeCost(a, bNext, startCost, cost);

                    if (newCost + 1e-12 < oldCost)
                    {
                        Array.Reverse(route, i, j - i + 1);
                        improved = true;
                        break;
                    }
                }

                if (improved) break;
            }

            if (!improved) break;
        }
    }

    private static double EdgeCost(int from, int to, double[] startCost, double[,] cost)
    {
        if (to < 0) return 0.0;
        if (from < 0) return startCost[to];
        return cost[from, to];
    }

    private static double RouteCost(int[] route, double[] startCost, double[,] cost)
    {
        if (route.Length == 0) return 0.0;
        double total = startCost[route[0]];
        for (int i = 0; i < route.Length - 1; i++)
            total += cost[route[i], route[i + 1]];
        return total;
    }

    private static int[] TopKIndices(double[] arr, int k)
    {
        int n = arr.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;

        Array.Sort(idx, (a, b) => arr[a].CompareTo(arr[b]));

        int[] outIdx = new int[k];
        Array.Copy(idx, outIdx, k);
        return outIdx;
    }

    private static double OrbitCost(Orbit a, Orbit b, double mu)
    {
        double r1 = a.P > 1e-9 ? a.P : a.RMag;
        double r2 = b.P > 1e-9 ? b.P : b.RMag;
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);

        double dvRad = HohmannDv(mu, r1, r2);

        double planeDot = Clamp(a.HHat.Dot(b.HHat), -1.0, 1.0);
        double angle = Math.Acos(planeDot);
        double rPlane = Math.Max(r1, r2);
        double vPlane = Math.Sqrt(mu / rPlane);
        double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * angle);

        double dvEcc = Math.Abs(a.E - b.E) * vPlane;

        return dvRad + dvPlane + 0.1 * dvEcc;
    }

    private static double HohmannDv(double mu, double r1, double r2)
    {
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);

        if (Math.Abs(r1 - r2) < 1e-12)
            return 0.0;

        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double aT = 0.5 * (r1 + r2);
        double vT1 = Math.Sqrt(mu * (2.0 / r1 - 1.0 / aT));
        double vT2 = Math.Sqrt(mu * (2.0 / r2 - 1.0 / aT));
        return Math.Abs(vT1 - v1) + Math.Abs(v2 - vT2);
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private readonly struct Orbit
    {
        public readonly Vec3 R;
        public readonly Vec3 V;
        public readonly Vec3 HHat;
        public readonly double RMag;
        public readonly double P;
        public readonly double E;

        private Orbit(Vec3 r, Vec3 v, Vec3 hHat, double rMag, double p, double e)
        {
            R = r;
            V = v;
            HHat = hHat;
            RMag = rMag;
            P = p;
            E = e;
        }

        public static Orbit FromState(Vec3 r, Vec3 v, double mu)
        {
            double rMag = r.Norm();

            Vec3 h = r.Cross(v);
            double hMag = h.Norm();
            Vec3 hHat = hMag > 1e-12 ? h / hMag : new Vec3(0, 0, 1);

            double p = (mu > 0 && hMag > 1e-12) ? (hMag * hMag / mu) : rMag;

            double e = 0.0;
            if (mu > 0 && rMag > 1e-12 && hMag > 1e-12)
            {
                Vec3 eVec = (v.Cross(h) / mu) - (r / rMag);
                e = eVec.Norm();
                if (double.IsNaN(e) || double.IsInfinity(e)) e = 0.0;
            }

            return new Orbit(r, v, hHat, rMag, p, e);
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public double Dot(Vec3 o) => X * o.X + Y * o.Y + Z * o.Z;

        public Vec3 Cross(Vec3 o) =>
            new Vec3(
                Y * o.Z - Z * o.Y,
                Z * o.X - X * o.Z,
                X * o.Y - Y * o.X
            );

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        private int SkipWhite()
        {
            int c;
            while (true)
            {
                c = Read();
                if (c == -1) return -1;
                if (c > ' ') return c;
            }
        }

        public bool TryNextInt(out int value)
        {
            int c = SkipWhite();
            if (c == -1)
            {
                value = 0;
                return false;
            }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = Read();
            }

            value = x * sign;
            return true;
        }

        public double NextDouble()
        {
            int c = SkipWhite();
            if (c == -1) return 0.0;

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            double val = 0.0;
            while (c >= '0' && c <= '9')
            {
                val = (val * 10.0) + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double mul = 1.0;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    mul *= 0.1;
                    val += (c - '0') * mul;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int expSign = 1;
                int exp = 0;
                c = Read();
                if (c == '-')
                {
                    expSign = -1;
                    c = Read();
                }
                else if (c == '+')
                {
                    c = Read();
                }

                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }

                val *= Math.Pow(10.0, expSign * exp);
            }

            return val * sign;
        }
    }
}
