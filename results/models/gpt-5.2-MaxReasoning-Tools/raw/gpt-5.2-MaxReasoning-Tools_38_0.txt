import sys


def _select(X, Y, r):
    cols = []
    for j in Y[r]:
        for i in list(X[j]):
            for k in Y[i]:
                if k != j:
                    X[k].discard(i)
        cols.append(X.pop(j))
    return cols


def _deselect(X, Y, r, cols):
    for j in reversed(Y[r]):
        X[j] = cols.pop()
        for i in X[j]:
            for k in Y[i]:
                if k != j:
                    X[k].add(i)


def _solve(X, Y, solution):
    if not X:
        return list(solution)

    c = min(X, key=lambda j: (len(X[j]), j))
    if not X[c]:
        return None

    for r in sorted(X[c]):
        solution.append(r)
        cols = _select(X, Y, r)
        res = _solve(X, Y, solution)
        if res is not None:
            return res
        _deselect(X, Y, r, cols)
        solution.pop()
    return None


def main() -> None:
    sys.setrecursionlimit(1_000_000)

    first_line = sys.stdin.buffer.readline()
    if not first_line:
        return
    first = first_line.decode("utf-8-sig").split()
    if len(first) < 2:
        return
    U = int(first[0])
    S = int(first[1])

    Y = []
    X = {j: set() for j in range(U)}

    for i in range(S):
        elems = sys.stdin.buffer.readline().split()
        if not elems:
            row = ()
        else:
            # Deduplicate and keep deterministic ordering.
            s = set()
            for tok in elems:
                try:
                    e = int(tok)
                except ValueError:
                    continue
                if 0 <= e < U:
                    s.add(e)
            row = tuple(sorted(s))
        Y.append(row)
        for j in row:
            X[j].add(i)

    if any(len(rows) == 0 for rows in X.values()):
        sys.stdout.write("NO SOLUTION\n")
        return

    sol = _solve(X, Y, [])
    if sol is None:
        sys.stdout.write("NO SOLUTION\n")
        return

    sol_sorted = sorted(sol)
    sys.stdout.write("SOLUTION\n")
    sys.stdout.write(" ".join(map(str, sol_sorted)) + "\n")


if __name__ == "__main__":
    main()
