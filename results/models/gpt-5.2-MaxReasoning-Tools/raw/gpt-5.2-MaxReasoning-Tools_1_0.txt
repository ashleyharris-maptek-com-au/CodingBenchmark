// Adaptive Euclidean TSP heuristic/exact solver (C++20, MSVC-friendly).
// Input:  N then N lines of "x y" (floats)
// Output: permutation of 0..N-1 starting with 0 (implicit return to 0)

#include <algorithm>
#include <array>
#include <charconv>
#include <cmath>
#include <cstddef>
#include <cstdint>
#include <cstdio>
#include <cstring>
#include <limits>
#include <system_error>
#include <vector>

struct Point {
  float x;
  float y;
};

class FastInput {
 public:
  FastInput() = default;

  bool read_u64(std::uint64_t& out) { return read_number(out); }
  bool read_double(double& out) { return read_fp(out); }

 private:
  static constexpr std::size_t kBufSize = 1u << 20;  // 1 MiB
  std::array<char, kBufSize> buf_{};
  std::size_t pos_ = 0;
  std::size_t len_ = 0;

  bool refill() {
    len_ = std::fread(buf_.data(), 1, buf_.size(), stdin);
    pos_ = 0;
    return len_ != 0;
  }

  inline char peek() {
    if (pos_ >= len_) {
      if (!refill()) return 0;
    }
    return buf_[pos_];
  }

  inline void advance() { ++pos_; }

  inline void skip_ws() {
    for (;;) {
      char c = peek();
      if (!c || c > ' ') break;
      advance();
    }
  }

  template <class UInt>
  bool read_number(UInt& out) {
    skip_ws();
    char c = peek();
    if (!c) return false;

    const std::size_t start_pos = pos_;
    const char* start = buf_.data() + start_pos;
    while (pos_ < len_ && buf_[pos_] > ' ') ++pos_;
    if (pos_ < len_) {
      const char* end = buf_.data() + pos_;
      auto res = std::from_chars(start, end, out);
      if (res.ec != std::errc{}) return false;
      return true;
    }

    // Token crossed buffer boundary: copy into a small temp buffer.
    std::array<char, 128> tmp{};
    std::size_t tlen = 0;
    for (std::size_t i = start_pos; i < len_ && tlen + 1 < tmp.size(); ++i) tmp[tlen++] = buf_[i];
    while (true) {
      if (!refill()) break;
      while (pos_ < len_ && buf_[pos_] > ' ') {
        if (tlen + 1 < tmp.size()) tmp[tlen++] = buf_[pos_];
        ++pos_;
      }
      if (pos_ < len_) break;
    }
    auto res = std::from_chars(tmp.data(), tmp.data() + tlen, out);
    if (res.ec != std::errc{}) return false;
    return true;
  }

  bool read_fp(double& out) {
    skip_ws();
    char c = peek();
    if (!c) return false;

    const std::size_t start_pos = pos_;
    const char* start = buf_.data() + start_pos;
    while (pos_ < len_ && buf_[pos_] > ' ') ++pos_;
    if (pos_ < len_) {
      const char* end = buf_.data() + pos_;
      auto res = std::from_chars(start, end, out, std::chars_format::general);
      if (res.ec != std::errc{}) out = 0.0;
      return true;
    }

    // Token crossed buffer boundary: copy into a small temp buffer.
    std::array<char, 128> tmp{};
    std::size_t tlen = 0;

    // Copy the part we already have in the current buffer.
    for (std::size_t i = start_pos; i < len_ && tlen + 1 < tmp.size(); ++i) tmp[tlen++] = buf_[i];

    while (true) {
      if (!refill()) break;
      while (pos_ < len_ && buf_[pos_] > ' ') {
        if (tlen + 1 < tmp.size()) tmp[tlen++] = buf_[pos_];
        ++pos_;
      }
      if (pos_ < len_) break;  // found token terminator in this buffer
    }

    auto res = std::from_chars(tmp.data(), tmp.data() + tlen, out, std::chars_format::general);
    if (res.ec != std::errc{}) out = 0.0;
    return true;
  }
};

class FastOutput {
 public:
  FastOutput() = default;
  ~FastOutput() { flush(); }

  void flush() {
    if (pos_) {
      std::fwrite(buf_.data(), 1, pos_, stdout);
      pos_ = 0;
    }
  }

  void put_char(char c) {
    if (pos_ == buf_.size()) flush();
    buf_[pos_++] = c;
  }

  template <class UInt>
  void put_uint(UInt v) {
    char tmp[32];
    auto res = std::to_chars(tmp, tmp + 32, v);
    std::size_t n = static_cast<std::size_t>(res.ptr - tmp);
    if (pos_ + n >= buf_.size()) flush();
    std::memcpy(buf_.data() + pos_, tmp, n);
    pos_ += n;
  }

 private:
  static constexpr std::size_t kBufSize = 1u << 20;  // 1 MiB
  std::array<char, kBufSize> buf_{};
  std::size_t pos_ = 0;
};

static inline double orient2d(const Point& a, const Point& b, const Point& c) {
  const double abx = static_cast<double>(b.x) - static_cast<double>(a.x);
  const double aby = static_cast<double>(b.y) - static_cast<double>(a.y);
  const double acx = static_cast<double>(c.x) - static_cast<double>(a.x);
  const double acy = static_cast<double>(c.y) - static_cast<double>(a.y);
  return abx * acy - aby * acx;
}

static inline int sgn_eps(double v) {
  constexpr double eps = 1e-12;
  if (v > eps) return 1;
  if (v < -eps) return -1;
  return 0;
}

static inline bool segments_proper_intersect(const Point& a, const Point& b, const Point& c, const Point& d) {
  const int o1 = sgn_eps(orient2d(a, b, c));
  const int o2 = sgn_eps(orient2d(a, b, d));
  if (o1 == 0 || o2 == 0 || o1 == o2) return false;
  const int o3 = sgn_eps(orient2d(c, d, a));
  const int o4 = sgn_eps(orient2d(c, d, b));
  if (o3 == 0 || o4 == 0 || o3 == o4) return false;
  return true;
}

static inline std::uint64_t part1by1(std::uint32_t x) {
  std::uint64_t v = x;
  v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
  v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
  v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
  v = (v | (v << 2)) & 0x3333333333333333ULL;
  v = (v | (v << 1)) & 0x5555555555555555ULL;
  return v;
}

static inline std::uint64_t morton2d_u32(std::uint32_t x, std::uint32_t y) {
  return part1by1(x) | (part1by1(y) << 1);
}

struct MortonItem {
  std::uint64_t key;
  std::uint32_t id;
};

static void radix_sort_morton(std::vector<MortonItem>& a) {
  std::vector<MortonItem> b(a.size());
  constexpr int kRadix = 256;
  std::array<std::uint32_t, kRadix> cnt{};

  for (int pass = 0; pass < 8; ++pass) {
    cnt.fill(0);
    const int shift = pass * 8;
    for (const auto& it : a) ++cnt[static_cast<unsigned>((it.key >> shift) & 0xFFULL)];

    std::uint32_t sum = 0;
    for (int i = 0; i < kRadix; ++i) {
      const std::uint32_t c = cnt[i];
      cnt[i] = sum;
      sum += c;
    }

    for (const auto& it : a) {
      const unsigned bucket = static_cast<unsigned>((it.key >> shift) & 0xFFULL);
      b[cnt[bucket]++] = it;
    }
    a.swap(b);
  }
}

static void uncross_local(std::vector<std::uint32_t>& tour, const std::vector<Point>& pts, std::size_t window,
                          int max_passes) {
  const std::size_t n = tour.size();
  if (n < 4) return;

  for (int pass = 0; pass < max_passes; ++pass) {
    bool any = false;
    for (std::size_t i = 0; i + 2 < n; ++i) {
      const std::size_t j_end = std::min(n - 2, i + window);
      for (std::size_t j = i + 2; j <= j_end; ++j) {
        const std::uint32_t a = tour[i];
        const std::uint32_t b = tour[i + 1];
        const std::uint32_t c = tour[j];
        const std::uint32_t d = tour[j + 1];
        if (segments_proper_intersect(pts[a], pts[b], pts[c], pts[d])) {
          std::reverse(tour.begin() + static_cast<std::ptrdiff_t>(i + 1),
                       tour.begin() + static_cast<std::ptrdiff_t>(j + 1));
          any = true;
        }
      }
    }
    if (!any) break;
  }
}

static std::vector<std::uint32_t> solve_exact_held_karp(const std::vector<Point>& pts) {
  const int n = static_cast<int>(pts.size());
  const int m = n - 1;
  const int full = (1 << m) - 1;
  const double INF = std::numeric_limits<double>::infinity();

  std::vector<double> dist(static_cast<std::size_t>(n) * static_cast<std::size_t>(n), 0.0);
  auto D = [&](int i, int j) -> double& { return dist[static_cast<std::size_t>(i) * n + j]; };
  for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n; ++j) {
      const double dx = static_cast<double>(pts[i].x) - static_cast<double>(pts[j].x);
      const double dy = static_cast<double>(pts[i].y) - static_cast<double>(pts[j].y);
      D(i, j) = std::sqrt(dx * dx + dy * dy);
    }
  }

  const std::size_t states = static_cast<std::size_t>(1u << m);
  std::vector<double> dp(states * static_cast<std::size_t>(m), INF);
  std::vector<int> parent(states * static_cast<std::size_t>(m), -1);
  auto idx = [&](int mask, int i) -> std::size_t { return static_cast<std::size_t>(mask) * m + i; };

  for (int i = 0; i < m; ++i) dp[idx(1 << i, i)] = D(0, i + 1);

  for (int mask = 1; mask <= full; ++mask) {
    for (int i = 0; i < m; ++i) {
      if (!(mask & (1 << i))) continue;
      const int pmask = mask ^ (1 << i);
      if (!pmask) continue;
      double best = dp[idx(mask, i)];
      int best_prev = parent[idx(mask, i)];
      for (int j = 0; j < m; ++j) {
        if (!(pmask & (1 << j))) continue;
        const double cand = dp[idx(pmask, j)] + D(j + 1, i + 1);
        if (cand < best) {
          best = cand;
          best_prev = j;
        }
      }
      dp[idx(mask, i)] = best;
      parent[idx(mask, i)] = best_prev;
    }
  }

  double best_cost = INF;
  int best_end = 0;
  for (int i = 0; i < m; ++i) {
    const double cand = dp[idx(full, i)] + D(i + 1, 0);
    if (cand < best_cost) {
      best_cost = cand;
      best_end = i;
    }
  }

  std::vector<std::uint32_t> rev;
  rev.reserve(static_cast<std::size_t>(m));
  int mask = full;
  int cur = best_end;
  while (cur != -1) {
    rev.push_back(static_cast<std::uint32_t>(cur + 1));
    const int prev = parent[idx(mask, cur)];
    mask ^= (1 << cur);
    cur = prev;
  }
  std::reverse(rev.begin(), rev.end());

  std::vector<std::uint32_t> tour;
  tour.reserve(static_cast<std::size_t>(n));
  tour.push_back(0);
  for (auto v : rev) tour.push_back(v);
  return tour;
}

int main() {
  FastInput in;
  std::uint64_t N64 = 0;
  if (!in.read_u64(N64)) return 0;
  const std::size_t n = static_cast<std::size_t>(N64);

  if (n == 0) {
    std::fputc('\n', stdout);
    return 0;
  }
  if (n == 1) {
    std::fputs("0\n", stdout);
    return 0;
  }

  std::vector<Point> pts;
  pts.resize(n);

  double minx = std::numeric_limits<double>::infinity();
  double maxx = -std::numeric_limits<double>::infinity();
  double miny = std::numeric_limits<double>::infinity();
  double maxy = -std::numeric_limits<double>::infinity();

  for (std::size_t i = 0; i < n; ++i) {
    double x = 0.0, y = 0.0;
    in.read_double(x);
    in.read_double(y);
    if (!std::isfinite(x)) x = 0.0;
    if (!std::isfinite(y)) y = 0.0;
    pts[i] = Point{static_cast<float>(x), static_cast<float>(y)};
    minx = std::min(minx, x);
    maxx = std::max(maxx, x);
    miny = std::min(miny, y);
    maxy = std::max(maxy, y);
  }

  // Very small: exact optimal tour.
  if (n <= 20) {
    const auto tour = solve_exact_held_karp(pts);
    FastOutput out;
    for (std::size_t i = 0; i < tour.size(); ++i) {
      out.put_uint(tour[i]);
      out.put_char(i + 1 == tour.size() ? '\n' : ' ');
    }
    return 0;
  }

  const double rangex = (maxx > minx) ? (maxx - minx) : 1.0;
  const double rangey = (maxy > miny) ? (maxy - miny) : 1.0;
  constexpr double U32_MAX_D = 4294967295.0;

  std::vector<MortonItem> items;
  items.resize(n);
  for (std::size_t i = 0; i < n; ++i) {
    const double nx = (static_cast<double>(pts[i].x) - minx) / rangex;
    const double ny = (static_cast<double>(pts[i].y) - miny) / rangey;
    double qx = nx * U32_MAX_D;
    double qy = ny * U32_MAX_D;
    if (!(qx >= 0.0)) qx = 0.0;
    if (!(qy >= 0.0)) qy = 0.0;
    if (qx > U32_MAX_D) qx = U32_MAX_D;
    if (qy > U32_MAX_D) qy = U32_MAX_D;
    const auto ix = static_cast<std::uint32_t>(qx + 0.5);
    const auto iy = static_cast<std::uint32_t>(qy + 0.5);
    items[i] = MortonItem{morton2d_u32(ix, iy), static_cast<std::uint32_t>(i)};
  }

  // Medium/Large: Morton order. For big N, radix-sort is consistently faster than comparison sort.
  if (n <= 200000) {
    std::sort(items.begin(), items.end(), [](const MortonItem& a, const MortonItem& b) {
      return (a.key < b.key) || (a.key == b.key && a.id < b.id);
    });
  } else {
    radix_sort_morton(items);
  }

  std::vector<std::uint32_t> tour;
  tour.resize(n);
  for (std::size_t i = 0; i < n; ++i) tour[i] = items[i].id;

  // Local uncrossing for better tour quality on small/medium instances.
  if (n <= 8000) {
    uncross_local(tour, pts, n, 3);
  } else if (n <= 200000) {
    uncross_local(tour, pts, 64, 2);
  }

  // Rotate tour to start at city 0.
  std::size_t pos0 = 0;
  for (; pos0 < n; ++pos0) {
    if (tour[pos0] == 0) break;
  }
  if (pos0 != 0 && pos0 < n) {
    std::rotate(tour.begin(), tour.begin() + static_cast<std::ptrdiff_t>(pos0), tour.end());
  }

  FastOutput out;
  for (std::size_t i = 0; i < n; ++i) {
    out.put_uint(tour[i]);
    out.put_char(i + 1 == n ? '\n' : ' ');
  }
  return 0;
}
