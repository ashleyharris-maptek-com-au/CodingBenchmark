using System;
using System.Buffers;
using System.Diagnostics;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string NextToken()
        {
            int b;
            do
            {
                b = ReadByte();
                if (b == -1) return null;
            } while (b <= 32);

            var sb = new StringBuilder(8);
            while (b > 32)
            {
                sb.Append((char)b);
                b = ReadByte();
                if (b == -1) break;
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            string tok = NextToken();
            if (tok == null) throw new EndOfStreamException();
            return int.Parse(tok);
        }
    }

    private struct PieceRotation
    {
        public int Width;
        public int Height;
        public int[] Skirt;       // length = Width
        public ulong[] RowMasks;  // length = Height, bits relative to x=0

        public bool SameShape(PieceRotation other)
        {
            if (Width != other.Width || Height != other.Height) return false;
            if (RowMasks == null || other.RowMasks == null) return false;
            if (RowMasks.Length != other.RowMasks.Length) return false;
            for (int i = 0; i < RowMasks.Length; i++)
            {
                if (RowMasks[i] != other.RowMasks[i]) return false;
            }
            return true;
        }
    }

    private sealed class PieceDef
    {
        public readonly PieceRotation[] Rotations; // 0..3
        public readonly int[] UniqueRotationIndices;

        public PieceDef(PieceRotation[] rotations, int[] uniqueRotationIndices)
        {
            Rotations = rotations;
            UniqueRotationIndices = uniqueRotationIndices;
        }
    }

    private struct Move
    {
        public int Rotation;
        public int Column;
        public Move(int rotation, int column)
        {
            Rotation = rotation;
            Column = column;
        }
    }

    private static readonly char[] AllPieceTypes = new[] { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
    private static readonly PieceDef[] PieceByChar = new PieceDef[128];

    static Program()
    {
        // Canonical rotation-0 bodies (origin at lower-left, shifted as low/left as possible).
        // Each body is 4 blocks: x0,y0,x1,y1,x2,y2,x3,y3
        AddPiece('I', new[] { 0, 0, 1, 0, 2, 0, 3, 0 });
        AddPiece('O', new[] { 0, 0, 1, 0, 0, 1, 1, 1 });
        AddPiece('T', new[] { 0, 0, 1, 0, 2, 0, 1, 1 });
        AddPiece('S', new[] { 0, 0, 1, 0, 1, 1, 2, 1 }); // right-leaning
        AddPiece('Z', new[] { 1, 0, 2, 0, 0, 1, 1, 1 }); // left-leaning
        AddPiece('J', new[] { 0, 0, 1, 0, 2, 0, 0, 1 }); // left L
        AddPiece('L', new[] { 0, 0, 1, 0, 2, 0, 2, 1 }); // right L
    }

    private static void AddPiece(char ch, int[] basePoints)
    {
        var rotations = BuildRotations(basePoints);
        int[] unique = UniqueRotationIndices(rotations);
        PieceByChar[ch] = new PieceDef(rotations, unique);
    }

    private static PieceRotation[] BuildRotations(int[] basePoints)
    {
        var rots = new PieceRotation[4];
        int[] pts = (int[])basePoints.Clone();
        for (int r = 0; r < 4; r++)
        {
            rots[r] = MakeRotation(pts);
            pts = RotateClockwiseAndNormalize(pts);
        }
        return rots;
    }

    private static int[] UniqueRotationIndices(PieceRotation[] rots)
    {
        int[] tmp = new int[4];
        int count = 0;
        for (int r = 0; r < 4; r++)
        {
            bool dup = false;
            for (int i = 0; i < count; i++)
            {
                if (rots[r].SameShape(rots[tmp[i]]))
                {
                    dup = true;
                    break;
                }
            }
            if (!dup) tmp[count++] = r;
        }
        var unique = new int[count];
        Array.Copy(tmp, unique, count);
        return unique;
    }

    private static PieceRotation MakeRotation(int[] pts)
    {
        int maxX = 0, maxY = 0;
        for (int i = 0; i < pts.Length; i += 2)
        {
            int x = pts[i];
            int y = pts[i + 1];
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
        }

        int width = maxX + 1;
        int height = maxY + 1;
        var rowMasks = new ulong[height];
        var skirt = new int[width];
        for (int i = 0; i < width; i++) skirt[i] = int.MaxValue;

        for (int i = 0; i < pts.Length; i += 2)
        {
            int x = pts[i];
            int y = pts[i + 1];
            rowMasks[y] |= 1UL << x;
            if (y < skirt[x]) skirt[x] = y;
        }

        for (int x = 0; x < width; x++)
        {
            if (skirt[x] == int.MaxValue) skirt[x] = 0;
        }

        return new PieceRotation
        {
            Width = width,
            Height = height,
            Skirt = skirt,
            RowMasks = rowMasks
        };
    }

    private static int[] RotateClockwiseAndNormalize(int[] pts)
    {
        // Rotate about origin (x,y)->(y,-x), then translate so minX=minY=0.
        int[] outPts = new int[pts.Length];
        int minX = int.MaxValue, minY = int.MaxValue;
        for (int i = 0; i < pts.Length; i += 2)
        {
            int x = pts[i];
            int y = pts[i + 1];
            int nx = y;
            int ny = -x;
            outPts[i] = nx;
            outPts[i + 1] = ny;
            if (nx < minX) minX = nx;
            if (ny < minY) minY = ny;
        }

        for (int i = 0; i < outPts.Length; i += 2)
        {
            outPts[i] -= minX;
            outPts[i + 1] -= minY;
        }
        return outPts;
    }

    private static ulong FullMaskForWidth(int width)
    {
        if (width >= 64) return ulong.MaxValue;
        return (1UL << width) - 1UL;
    }

    private static int ComputeHeights(ulong[] rows, int width, int height, int[] heightsOut)
    {
        int maxHeight = 0;
        for (int x = 0; x < width; x++)
        {
            int h = 0;
            ulong bit = 1UL << x;
            for (int y = height - 1; y >= 0; y--)
            {
                if ((rows[y] & bit) != 0)
                {
                    h = y + 1;
                    break;
                }
            }
            heightsOut[x] = h;
            if (h > maxHeight) maxHeight = h;
        }
        return maxHeight;
    }

    private static bool TryPlaceAndClear(
        ulong[] srcRows,
        int[] srcHeights,
        int srcMaxHeight,
        int width,
        int height,
        ulong fullMask,
        PieceRotation rot,
        int x,
        ulong[] dstRows,
        int[] dstHeights,
        out int dstMaxHeight,
        out int linesCleared,
        out int landingY,
        out int pieceCellsClearedInLines)
    {
        // Copy base rows
        for (int i = 0; i < height; i++) dstRows[i] = srcRows[i];

        // Compute drop/landing Y using skirt + column heights.
        int y0 = 0;
        for (int dx = 0; dx < rot.Width; dx++)
        {
            int colH = srcHeights[x + dx];
            int y = colH - rot.Skirt[dx];
            if (y > y0) y0 = y;
        }
        landingY = y0;

        if (y0 + rot.Height > height)
        {
            linesCleared = 0;
            pieceCellsClearedInLines = 0;
            dstMaxHeight = srcMaxHeight;
            return false;
        }

        // Collision check + apply.
        for (int dy = 0; dy < rot.Height; dy++)
        {
            ulong mask = rot.RowMasks[dy] << x;
            if ((dstRows[y0 + dy] & mask) != 0)
            {
                linesCleared = 0;
                pieceCellsClearedInLines = 0;
                dstMaxHeight = srcMaxHeight;
                return false;
            }
        }
        for (int dy = 0; dy < rot.Height; dy++)
        {
            dstRows[y0 + dy] |= rot.RowMasks[dy] << x;
        }

        // Clear full rows and count piece cells removed.
        linesCleared = 0;
        pieceCellsClearedInLines = 0;
        int write = 0;
        for (int r = 0; r < height; r++)
        {
            if (dstRows[r] == fullMask)
            {
                linesCleared++;
                if (r >= y0 && r < y0 + rot.Height)
                {
                    ulong pm = rot.RowMasks[r - y0] << x;
                    pieceCellsClearedInLines += (int)BitOperations.PopCount(pm);
                }
                continue;
            }
            dstRows[write++] = dstRows[r];
        }
        for (int r = write; r < height; r++) dstRows[r] = 0;

        dstMaxHeight = ComputeHeights(dstRows, width, height, dstHeights);
        return true;
    }

    private static double EvaluateBoard(
        ulong[] rows,
        int[] heights,
        int maxHeight,
        int width,
        int height,
        int linesCleared,
        int landingY,
        int pieceCellsClearedInLines)
    {
        // Classic 4-feature baseline plus a few light additions.
        int aggregateHeight = 0;
        for (int i = 0; i < width; i++) aggregateHeight += heights[i];

        int bumpiness = 0;
        for (int i = 0; i < width - 1; i++)
            bumpiness += Math.Abs(heights[i] - heights[i + 1]);

        int holes = 0;
        int holeDepth = 0;
        for (int x = 0; x < width; x++)
        {
            int colH = heights[x];
            if (colH == 0) continue;
            bool seenFilled = false;
            ulong bit = 1UL << x;
            for (int y = colH - 1; y >= 0; y--)
            {
                bool filled = (rows[y] & bit) != 0;
                if (filled) seenFilled = true;
                else if (seenFilled)
                {
                    holes++;
                    holeDepth += (colH - y);
                }
            }
        }

        int wellSum = 0;
        for (int x = 0; x < width; x++)
        {
            int depth = 0;
            for (int y = 0; y < height; y++)
            {
                bool empty = ((rows[y] >> x) & 1UL) == 0UL;
                if (!empty)
                {
                    depth = 0;
                    continue;
                }
                bool leftFilled = x == 0 || (((rows[y] >> (x - 1)) & 1UL) != 0UL);
                bool rightFilled = x == width - 1 || (((rows[y] >> (x + 1)) & 1UL) != 0UL);
                if (leftFilled && rightFilled)
                {
                    depth++;
                    wellSum += depth;
                }
                else
                {
                    depth = 0;
                }
            }
        }

        double risk = (double)maxHeight / Math.Max(1, height);

        // Normalize to handle variable board sizes.
        double norm = (10.0 / Math.Max(1, width)) * (20.0 / Math.Max(1, height));
        double agg = aggregateHeight * norm;
        double hol = holes * norm;
        double bump = bumpiness * norm;

        // Adaptive re-weighting based on risk.
        double danger = risk * risk;
        double wLines = 0.760666 * (1.0 + 0.6 * danger);
        double wAgg = -0.510066 * (1.0 + 0.8 * danger);
        double wHoles = -0.35663 * (1.0 + 1.6 * danger);
        double wBump = -0.184483 * (1.0 + 0.5 * danger);

        // Extra terms (kept small for stability across rulesets).
        double wWell = -0.015 * (1.0 + 1.2 * danger);
        double wHoleDepth = -0.0025 * (1.0 + 2.0 * danger);
        double wMax = -1.2 * danger;
        double wLanding = -0.02 * (1.0 + 1.0 * danger);

        // Encourage multi-line clears a bit more than singles.
        double lineBonus = linesCleared == 4 ? 1.2 : (linesCleared == 3 ? 0.6 : (linesCleared == 2 ? 0.2 : 0.0));

        // Small "eroded piece cells" style reward encourages using the current piece to clear.
        int eroded = linesCleared * pieceCellsClearedInLines;

        return
            (wLines * linesCleared) + lineBonus +
            (wAgg * agg) +
            (wHoles * hol) +
            (wBump * bump) +
            (wWell * wellSum * norm) +
            (wHoleDepth * holeDepth * norm) +
            (wMax * maxHeight * norm) +
            (wLanding * landingY * norm) +
            (0.05 * eroded);
    }

    private static int ChooseSearchDepth(int width, int height, int maxHeight, int remainingPieces)
    {
        double risk = (double)maxHeight / Math.Max(1, height);
        // Adaptive strategy:
        // - When the board is calm and the branching factor is low, do depth-2 expectimax.
        // - When the board is tall or wide, stick to depth 1 for speed and stability.
        if (remainingPieces == 0) return 1;
        if (risk < 0.40 && width <= 14) return 2;
        return 1;
    }

    private static Move FindBestMove(
        char pieceType,
        ulong[] boardRows,
        int[] boardHeights,
        int boardMaxHeight,
        int width,
        int height,
        ulong fullMask,
        int remainingPieces)
    {
        PieceDef def = (pieceType < 128) ? PieceByChar[pieceType] : null;
        if (def == null) return new Move(0, 0);

        int depth = ChooseSearchDepth(width, height, boardMaxHeight, remainingPieces);
        var sw = Stopwatch.StartNew();
        int budgetMs = depth == 2 ? 200 : 60; // well under the 5s hard limit

        Move bestMove = new Move(0, 0);
        double bestScore = double.NegativeInfinity;
        bool foundAny = false;

        var poolU64 = ArrayPool<ulong>.Shared;
        var poolInt = ArrayPool<int>.Shared;

        ulong[] after1Rows = null;
        int[] after1Heights = null;

        try
        {
            foreach (int rIdx in def.UniqueRotationIndices)
            {
                PieceRotation rot = def.Rotations[rIdx];
                int maxX = width - rot.Width;
                if (maxX < 0) continue;

                for (int x = 0; x <= maxX; x++)
                {
                    if (sw.ElapsedMilliseconds > budgetMs) { depth = 1; }

                    after1Rows = poolU64.Rent(height);
                    after1Heights = poolInt.Rent(width);

                    int maxH1, cleared1, landingY1, pieceCellsCleared1;
                    bool ok1 = TryPlaceAndClear(
                        boardRows, boardHeights, boardMaxHeight,
                        width, height, fullMask,
                        rot, x,
                        after1Rows, after1Heights,
                        out maxH1,
                        out cleared1,
                        out landingY1,
                        out pieceCellsCleared1);

                    if (!ok1)
                    {
                        poolU64.Return(after1Rows, clearArray: false);
                        poolInt.Return(after1Heights, clearArray: false);
                        after1Rows = null;
                        after1Heights = null;
                        continue;
                    }

                    foundAny = true;
                    double score1 = EvaluateBoard(after1Rows, after1Heights, maxH1, width, height, cleared1, landingY1, pieceCellsCleared1);
                    double totalScore = score1;

                    if (depth == 2 && sw.ElapsedMilliseconds <= budgetMs)
                    {
                        // Expectimax with uniform next-piece distribution (no preview available).
                        double expectedBest = 0.0;
                        for (int p = 0; p < AllPieceTypes.Length; p++)
                        {
                            if (sw.ElapsedMilliseconds > budgetMs) { depth = 1; break; }

                            char nextType = AllPieceTypes[p];
                            PieceDef nextDef = PieceByChar[nextType];
                            double bestNext = double.NegativeInfinity;

                            foreach (int nr in nextDef.UniqueRotationIndices)
                            {
                                PieceRotation nrot = nextDef.Rotations[nr];
                                int nMaxX = width - nrot.Width;
                                if (nMaxX < 0) continue;

                                for (int nx = 0; nx <= nMaxX; nx++)
                                {
                                    ulong[] after2Rows = poolU64.Rent(height);
                                    int[] after2Heights = poolInt.Rent(width);

                                    int maxH2, cleared2, landingY2, pieceCellsCleared2;
                                    bool ok2 = TryPlaceAndClear(
                                        after1Rows, after1Heights, maxH1,
                                        width, height, fullMask,
                                        nrot, nx,
                                        after2Rows, after2Heights,
                                        out maxH2,
                                        out cleared2,
                                        out landingY2,
                                        out pieceCellsCleared2);

                                    if (ok2)
                                    {
                                        double s2 = EvaluateBoard(after2Rows, after2Heights, maxH2, width, height, cleared2, landingY2, pieceCellsCleared2);
                                        if (s2 > bestNext) bestNext = s2;
                                    }

                                    poolU64.Return(after2Rows, clearArray: false);
                                    poolInt.Return(after2Heights, clearArray: false);
                                }
                            }

                            if (double.IsNegativeInfinity(bestNext))
                            {
                                // If a next piece has no valid placements, that's catastrophic.
                                bestNext = -1e9;
                            }
                            expectedBest += bestNext;
                        }

                        expectedBest /= AllPieceTypes.Length;
                        double discount = 0.55; // planning but not overcommitting to expectations
                        totalScore = score1 + discount * expectedBest;
                    }

                    if (totalScore > bestScore + 1e-9 ||
                        (!double.IsInfinity(totalScore) && Math.Abs(totalScore - bestScore) <= 1e-9 &&
                         (rIdx < bestMove.Rotation || (rIdx == bestMove.Rotation && x < bestMove.Column))))
                    {
                        bestScore = totalScore;
                        bestMove = new Move(rIdx, x);
                    }

                    poolU64.Return(after1Rows, clearArray: false);
                    poolInt.Return(after1Heights, clearArray: false);
                    after1Rows = null;
                    after1Heights = null;
                }
            }
        }
        finally
        {
            if (after1Rows != null) poolU64.Return(after1Rows, clearArray: false);
            if (after1Heights != null) poolInt.Return(after1Heights, clearArray: false);
        }

        if (!foundAny) return new Move(0, 0);
        return bestMove;
    }

    private static bool ApplyMoveInPlace(
        ulong[] rows,
        int[] heights,
        ref int maxHeight,
        int width,
        int height,
        ulong fullMask,
        PieceRotation rot,
        int x)
    {
        int y0 = 0;
        for (int dx = 0; dx < rot.Width; dx++)
        {
            int colH = heights[x + dx];
            int y = colH - rot.Skirt[dx];
            if (y > y0) y0 = y;
        }

        if (y0 + rot.Height > height) return false;

        for (int dy = 0; dy < rot.Height; dy++)
        {
            ulong mask = rot.RowMasks[dy] << x;
            if ((rows[y0 + dy] & mask) != 0) return false;
        }
        for (int dy = 0; dy < rot.Height; dy++)
            rows[y0 + dy] |= rot.RowMasks[dy] << x;

        int write = 0;
        for (int r = 0; r < height; r++)
        {
            if (rows[r] == fullMask) continue;
            rows[write++] = rows[r];
        }
        for (int r = write; r < height; r++) rows[r] = 0;

        maxHeight = ComputeHeights(rows, width, height, heights);
        return true;
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        string wTok = scanner.NextToken();
        if (wTok == null) return;
        string hTok = scanner.NextToken();
        if (hTok == null) return;
        string nTok = scanner.NextToken();
        if (nTok == null) return;

        int width = int.Parse(wTok);
        int height = int.Parse(hTok);
        int numPieces = int.Parse(nTok);

        if (width <= 0 || height <= 0)
            return;

        ulong fullMask = FullMaskForWidth(width);
        var boardRows = new ulong[height];
        var boardHeights = new int[width];
        int boardMaxHeight = 0;

        int i = 0;
        while (numPieces < 0 || i < numPieces)
        {
            string pieceTok = scanner.NextToken();
            if (pieceTok == null) break;
            char piece = pieceTok.Length > 0 ? pieceTok[0] : 'I';

            int remaining = (numPieces < 0) ? -1 : (numPieces - i - 1);
            Move move = FindBestMove(piece, boardRows, boardHeights, boardMaxHeight, width, height, fullMask, remaining);

            // Ensure the move is actually placeable under our model; otherwise fall back to first legal.
            PieceDef def = (piece < 128) ? PieceByChar[piece] : null;
            bool applied = false;
            if (def != null)
            {
                PieceRotation rot = def.Rotations[move.Rotation];
                if (move.Column >= 0 && move.Column + rot.Width <= width)
                {
                    applied = ApplyMoveInPlace(boardRows, boardHeights, ref boardMaxHeight, width, height, fullMask, rot, move.Column);
                }
            }

            if (!applied && def != null)
            {
                // Brute-force a guaranteed legal move.
                bool found = false;
                foreach (int rIdx in def.UniqueRotationIndices)
                {
                    PieceRotation rot = def.Rotations[rIdx];
                    for (int x = 0; x <= width - rot.Width; x++)
                    {
                        // Try apply on a temp copy to avoid corrupting board on failure.
                        var tmp = ArrayPool<ulong>.Shared.Rent(height);
                        for (int rr = 0; rr < height; rr++) tmp[rr] = boardRows[rr];
                        var tmpHeights = ArrayPool<int>.Shared.Rent(width);
                        for (int cc = 0; cc < width; cc++) tmpHeights[cc] = boardHeights[cc];
                        int tmpMax = boardMaxHeight;
                        bool ok = ApplyMoveInPlace(tmp, tmpHeights, ref tmpMax, width, height, fullMask, rot, x);
                        if (ok)
                        {
                            // Commit temp.
                            for (int rr = 0; rr < height; rr++) boardRows[rr] = tmp[rr];
                            for (int cc = 0; cc < width; cc++) boardHeights[cc] = tmpHeights[cc];
                            boardMaxHeight = tmpMax;
                            move = new Move(rIdx, x);
                            found = true;
                        }
                        ArrayPool<ulong>.Shared.Return(tmp, clearArray: false);
                        ArrayPool<int>.Shared.Return(tmpHeights, clearArray: false);
                        if (found) break;
                    }
                    if (found) break;
                }
            }

            Console.WriteLine(move.Rotation + " " + move.Column);
            Console.Out.Flush();

            // Consume judge response token to stay in sync (not used for logic).
            string result = scanner.NextToken();
            if (result == null) break;
            if (result == "gameover") break;

            i++;
        }
    }
