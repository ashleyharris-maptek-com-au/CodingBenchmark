```rust
use std::cmp::Ordering;
use std::f64;
use std::io::{self, Read};

#[derive(Debug)]
struct Sample {
    depth: f64,
    props: Vec<f64>, // one per property
    // position along hole is implied by index
}

#[derive(Debug)]
struct Hole {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    samples: Vec<Sample>,
}

fn median(data: &mut [f64]) -> f64 {
    if data.is_empty() {
        return 0.0;
    }
    data.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = data.len();
    if n % 2 == 1 {
        data[n / 2]
    } else {
        (data[n / 2 - 1] + data[n / 2]) / 2.0
    }
}

fn mad(data: &mut [f64], med: f64) -> f64 {
    // Median absolute deviation
    let mut diffs: Vec<f64> = data.iter().map(|v| (v - med).abs()).collect();
    median(&mut diffs)
}

// A small helper to safely parse f64, default to NaN on failure
fn nf64(s: &str) -> f64 {
    s.parse::<f64>().unwrap_or(f64::NAN)
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("read stdin");
    let mut it = input.split_whitespace();

    // Read header
    let num_holes: usize = it.next().unwrap().parse().unwrap_or(0);
    let num_props: usize = it.next().unwrap().parse().unwrap_or(0);

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(p) = it.next() {
            prop_names.push(p.to_string());
        } else {
            prop_names.push("prop".to_string());
        }
    }

    // Read holes
    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    for _ in 0..num_holes {
        // hole header line
        let hole_id = it.next().unwrap_or("").to_string();
        let start_x: f64 = it.next().unwrap().parse().unwrap_or(0.0);
        let start_y: f64 = it.next().unwrap().parse().unwrap_or(0.0);
        let start_z: f64 = it.next().unwrap().parse().unwrap_or(0.0);
        let dir_x: f64 = it.next().unwrap().parse().unwrap_or(0.0);
        let dir_y: f64 = it.next().unwrap().parse().unwrap_or(0.0);
        let dir_z: f64 = it.next().unwrap().parse().unwrap_or(0.0);
        let length: f64 = it.next().unwrap().parse().unwrap_or(0.0);

        // number of samples for this hole
        let num_samples: usize = it.next().unwrap().parse().unwrap_or(0);

        let mut samples: Vec<Sample> = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            // depth line followed by properties
            let depth: f64 = it.next().unwrap().parse().unwrap_or(f64::NAN);
            let mut props: Vec<f64> = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                let v: f64 = it.next().unwrap().parse().unwrap_or(f64::NAN);
                props.push(v);
            }
            samples.push(Sample { depth, props });
        }

        holes.push(Hole {
            id: hole_id,
            start: [start_x, start_y, start_z],
            dir: [dir_x, dir_y, dir_z],
            length,
            samples,
        });
    }

    // Collect suspects
    // Heuristic strategy:
    // - If total_samples <= 10k: global robust outlier per property using MAD
    // - Else: per-hole local depth-neighborhood z-scores (|z|>3)
    let total_samples: usize = holes.iter().map(|h| h.samples.len()).sum();

    let mut suspects: Vec<(String, usize, String, f64)> = Vec::new();
    const Z_THRESH: f64 = 3.0;

    if total_samples == 0 {
        // nothing to do
    } else if total_samples <= 10_000 {
        // Global MAD per property
        // gather values per property
        let mut vals: Vec<Vec<f64>> = vec![Vec::new(); num_props];
        for h in &holes {
            for s in &h.samples {
                for (i, v) in s.props.iter().enumerate() {
                    if v.is_finite() {
                        vals[i].push(*v);
                    }
                }
            }
        }

        // For each property, compute median and MAD
        let mut medians: Vec<f64> = vec![0.0; num_props];
        let mut mads: Vec<f64> = vec![0.0; num_props];
        for p in 0..num_props {
            if vals[p].is_empty() {
                medians[p] = 0.0;
                mads[p] = 1.0; // avoid div by zero
                continue;
            }
            // clone to sort
            let mut tmp = vals[p].clone();
            let med = median(&mut tmp);
            medians[p] = med;
            // MAD
            let mut tmp2 = vals[p].clone();
            let madv = mad(&mut tmp2, med);
            // If MAD is 0 (no dispersion), set to small value to avoid div by zero
            mads[p] = if madv <= 0.0 { 1e-6 } else { madv };
        }

        // Now compute z-scores for each sample and collect outliers
        for h in &holes {
            for (idx, s) in h.samples.iter().enumerate() {
                for p in 0..num_props {
                    let v = s.props[p];
                    if !v.is_finite() {
                        continue;
                    }
                    let z = (v - medians[p]) / mads[p];
                    if z.abs() > Z_THRESH {
                        // confidence proportional to |z|/Z_THRESH
                        let conf = (z.abs() / Z_THRESH).min(1.0);
                        suspects.push((
                            h.id.clone(),
                            idx,
                            prop_names[p].clone(),
                            conf,
                        ));
                    }
                }
            }
        }
    } else {
        // Large dataset: per-hole local depth-based z-scores
        for h in &holes {
            let n = h.samples.len();
            if n == 0 {
                continue;
            }
            for i in 0..n {
                let s = &h.samples[i];
                for p in 0..num_props {
                    let v = s.props[p];
                    if !v.is_finite() {
                        continue;
                    }
                    // neighborhood: indices [i-2, i+2] within bounds
                    let start = if i >= 2 { i - 2 } else { 0 };
                    let end = if i + 2 < n { i + 2 } else { n - 1 };
                    // collect neighbors excluding i
                    let mut neigh: Vec<f64> = Vec::with_capacity((end - start) as usize);
                    for j in start..=end {
                        if j == i { continue; }
                        let w = h.samples[j].props[p];
                        if w.is_finite() {
                            neigh.push(w);
                        }
                    }
                    let mean = if neigh.is_empty() {
                        v // fallback
                    } else {
                        neigh.iter().sum::<f64>() / (neigh.len() as f64)
                    };
                    let std = {
                        if neigh.is_empty() {
                            0.0
                        } else {
                            let var = neigh.iter().map(|x| (x - mean).powi(2)).sum::<f64>() / (neigh.len() as f64);
                            var.sqrt()
                        }
                    };
                    let z = if std > 0.0 { (v - mean) / std } else { 0.0 };
                    if z.abs() > Z_THRESH {
                        let conf = (z.abs() / Z_THRESH).min(1.0);
                        suspects.push((
                            h.id.clone(),
                            i,
                            prop_names[p].clone(),
                            conf,
                        ));
                    }
                }
            }
        }
    }

    // Deduplicate possible duplicates (same hole, same sample, same property) by keeping max confidence
    use std::collections::HashMap;
    let mut map: HashMap<(String, usize, String), f64> = HashMap::new();
    for (hid, sidx, pname, conf) in suspects {
        let key = (hid.clone(), sidx, pname.clone());
        let e = map.entry(key).or_insert(0.0);
        if conf > *e {
            *e = conf;
        }
    }

    // Build final list with distinct entries
    let mut final_suspects: Vec<(String, usize, String, f64)> = map
        .into_iter()
        .map(|(k, conf)| (k.0, k.1, k.2, conf))
        .collect();

    // Sort by confidence desc
    final_suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap_or(Ordering::Equal));

    // Output
    println!("{}", final_suspects.len());
    for (hole_id, sample_idx, prop_name, conf) in final_suspects {
        // sample_idx is zero-based; if desired, could output 1-based. Spec didn't demand.
        println!("{} {} {} {:.6}", hole_id, sample_idx, prop_name, conf);
    }
}
``` 
