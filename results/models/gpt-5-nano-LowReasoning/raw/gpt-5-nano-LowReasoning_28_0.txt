use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { buf: input.into_bytes(), idx: 0 }
    }
    fn next_usize(&mut self) -> usize {
        // skip non-digit (and non-minus) characters
        let n = self.buf.len();
        while self.idx < n {
            let c = self.buf[self.idx];
            if c == b'-' || (c >= b'0' && c <= b'9') {
                break;
            }
            self.idx += 1;
        }
        // read sign (not needed for non-negative inputs here, but keep robust)
        let mut sign = 1i64;
        if self.idx < n && self.buf[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        }
        let mut val: i64 = 0;
        while self.idx < n {
            let c = self.buf[self.idx];
            if c < b'0' || c > b'9' {
                break;
            }
            val = val * 10 + (c - b'0') as i64;
            self.idx += 1;
        }
        (val * sign) as usize
    }
}

fn main() {
    // Fast IO setup
    let mut scanner = Scanner::new();
    let mut out = String::new();

    // Read number of vertices and edges
    let n = scanner.next_usize();
    let m = scanner.next_usize();

    // Edge list: (u, v)
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    // Adjacency: for each vertex, list of incident edge indices
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for i in 0..m {
        let u = scanner.next_usize();
        let v = scanner.next_usize();
        edges.push((u, v));
        adj[u].push(i);
        adj[v].push(i);
    }

    // State tracking
    let mut edge_removed = vec![false; m];
    let mut in_cover = vec![false; n];
    let mut cover: Vec<usize> = Vec::new();

    // Greedy 2-approximation via maximal matching:
    // Iterate edges, when finding an uncovered edge (u,v) with neither endpoint in cover,
    // add both endpoints to cover and remove all incident edges of u and v.
    for i in 0..m {
        if edge_removed[i] {
            continue;
        }
        let (u, v) = edges[i];
        if in_cover[u] || in_cover[v] {
            // This edge is already covered by an existing vertex in the cover.
            edge_removed[i] = true;
            continue;
        }
        // Add endpoints to the cover
        if !in_cover[u] {
            in_cover[u] = true;
            cover.push(u);
        }
        if !in_cover[v] {
            in_cover[v] = true;
            cover.push(v);
        }
        // Remove all edges incident to either u or v
        for &ei in &adj[u] {
            if !edge_removed[ei] {
                edge_removed[ei] = true;
            }
        }
        for &ei in &adj[v] {
            if !edge_removed[ei] {
                edge_removed[ei] = true;
            }
        }
    }

    // Output as specified:
    // vertex_count
    // vertex_1 vertex_2 ... vertex_k
    writeln!(io::BufWriter::new(io::stdout()), "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        for (idx, v) in cover.iter().enumerate() {
            if idx + 1 == cover.len() {
                // last without trailing space
                write!(out, "{}", v).unwrap();
            } else {
                write!(out, "{} ", v).unwrap();
            }
        }
        writeln!(io::BufWriter::new(io::stdout()), "{}", out).unwrap();
    } else {
        // If no vertex in cover, still print an empty line for the second line
        println!();
    }
}
