    x: f64,
    y: f64,
}

    fn add(&self, o: &Pt) -> Pt {
        Pt { x: self.x + o.x, y: self.y + o.y }
    }
    fn sub(&self, o: &Pt) -> Pt {
        Pt { x: self.x - o.x, y: self.y - o.y }
    }
}

#[derive(Clone, Debug)]

    v: Vec<Pt>,
}

impl Poly {
    fn new(points: Vec<Pt>) -> Self {
        Poly { v: points }
    }

    // Translate polygon by delta
    fn translated(&self, delta: &Pt) -> Poly {
        let mut w = Vec::with_capacity(self.v.len());
        for p in &self.v {
            w.push(Pt { x: p.x + delta.x, y: p.y + delta.y });
        }
        Poly::new(w)
    }

    // Check all vertices are inside (or on edge) the stock polygon
    fn all_points_in(&self, stock: &Poly) -> bool {
        for p in &self.v {
            if !point_in_polygon(p, stock) {
                return false;
            }
        }
        true
    }

    // Check polygon intersects with another polygon (edges cross or one inside)
    fn intersects(&self, other: &Poly) -> bool {
        // Edge-edge intersections
        for i in 0..self.v.len() {
            let a1 = self.v[i];
            let a2 = self.v[(i + 1) % self.v.len()];
            for j in 0..other.v.len() {
                let b1 = other.v[j];
                let b2 = other.v[(j + 1) % other.v.len()];

                    return true;
                }
            }
        }
        // One inside another?
        if point_in_polygon(self.v[0], other) { return true; }
        if point_in_polygon(other.v[0], self) { return true; }
        false
    }

    // Quick bbox
    fn bbox(&self) -> (Pt, Pt) {
        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for p in &self.v {
            if p.x < minx { minx = p.x; }
            if p.x > maxx { maxx = p.x; }
            if p.y < miny { miny = p.y; }
            if p.y > maxy { maxy = p.y; }
        }
        (Pt { x: minx, y: miny }, Pt { x: maxx, y: maxy })
    }
}

// Geometry helpers

fn cross(a: &Pt, b: &Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn orient(a: &Pt, b: &Pt, c: &Pt) -> f64 {
    // cross of AB x AC
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: &Pt, b: &Pt, p: &Pt) -> bool {
    // p on segment ab?
    let minx = a.x.min(b.x);
    let maxx = a.x.max(b.x);
    let miny = a.y.min(b.y);
    let maxy = a.y.max(b.y);
    (p.x - a.x).abs() < 1e-9 * 1.0 + (maxx - minx).abs() * 1e-9
        && (p.y - a.y).abs() < 1e-9 * 1.0 + (maxy - miny).abs() * 1e-9
        && orient(a, b, p).abs() < 1e-9
        && p.x >= minx - 1e-9
        && p.x <= maxx + 1e-9
        && p.y >= miny - 1e-9
        && p.y <= maxy + 1e-9
}

fn segments_intersect(a1: Pt, a2: Pt, b1: Pt, b2: Pt) -> bool {
    let o1 = orient(&a1, &a2, &b1);
    let o2 = orient(&a1, &a2, &b2);
    let o3 = orient(&b1, &b2, &a1);
    let o4 = orient(&b1, &b2, &a2);

    if (o1 * o2) < 0.0 && (o3 * o4) < 0.0 {
        return true;
    }
    // Colinear cases
    if o1.abs() < 1e-9 && on_segment(&a1, &a2, &b1) { return true; }
    if o2.abs() < 1e-9 && on_segment(&a1, &a2, &b2) { return true; }
    if o3.abs() < 1e-9 && on_segment(&b1, &b2, &a1) { return true; }
    if o4.abs() < 1e-9 && on_segment(&b1, &b2, &a2) { return true; }

    false
}

// Point in polygon (ray casting)
fn point_in_polygon(p: &Pt, poly: &Poly) -> bool {
    // Winding number / ray casting
    let mut inside = false;
    let n = poly.v.len();
    for i in 0..n {
        let a = poly.v[i];
        let b = poly.v[(i + 1) % n];
        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y).max(1e-18) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}
