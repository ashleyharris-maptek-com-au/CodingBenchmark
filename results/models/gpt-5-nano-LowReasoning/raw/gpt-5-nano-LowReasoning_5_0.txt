use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, PartialEq, Eq)]
struct Cell {
    x: i32,
    y: i32,
}

fn idx(x: i32, y: i32, w: i32, h: i32) -> usize {
    (y * w + x) as usize
}

fn in_bounds(x: i32, y: i32, w: i32, h: i32) -> bool {
    x >= 0 && y >= 0 && x < w && y < h
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let width: i32 = it.next().unwrap().parse().unwrap();
    let height: i32 = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = {
        let s = it.next().unwrap();
        s == "1" || s.eq_ignore_ascii_case("true")
    };

    let w = width;
    let h = height;

    // Grid: 0 = free, 1 = obstacle
    let mut grid = vec![0u8; (w * h) as usize];

    for _ in 0..obstacle_count {
        let ox: i32 = it.next().unwrap().parse().unwrap();
        let oy: i32 = it.next().unwrap().parse().unwrap();
        if in_bounds(ox, oy, w, h) {
            grid[idx(ox, oy, w, h)] = 1;
        }
    }

    // Quick checks
    if w <= 0 || h <= 0 {
        println!("0");
        return;
    }

    // Start cell must be free
    if grid[idx(0, 0, w, h)] == 1 {
        println!("0");
        return;
    }

    // Count non-obstacle cells
    let mut total_free = 0usize;
    for y in 0..h {
        for x in 0..w {
            if grid[idx(x, y, w, h)] == 0 {
                total_free += 1;
            }
        }
    }

    if total_free == 0 {
        // Nothing to visit; path of length 0? But constraints say start at (0,0) and visit every non-obstacle cell.
        // If there are no non-obstacle cells (should not happen since start is free), output 0.
        println!("0");
        return;
    }

    // If only one cell, trivially ok
    if total_free == 1 {
        println!("1");
        println!("0 0");
        return;
    }

    // Backtracking with heuristics
    let start = Cell { x: 0, y: 0 };
    let t0 = Instant::now();
    // Time budget: 4.5 minutes
    let time_budget = Duration::from_secs(270);
    // Visited mask
    let mut visited = vec![false; (w * h) as usize];
    let mut path: Vec<Cell> = Vec::with_capacity(total_free);

    // Initialize
    visited[idx(0, 0, w, h)] = true;
    path.push(start);

    // Precompute neighbor directions
    let dirs = [(1i32, 0i32), (-1, 0), (0, 1), (0, -1)];

    // Helper: count unvisited neighbors of a cell
    let mut neigh_count = vec![0u8; (w * h) as usize];
    // initialize counts for all free cells
    for y in 0..h {
        for x in 0..w {
            let i = idx(x, y, w, h);
            if grid[i] == 0 && !visited[i] {
                let mut c = 0u8;
                for (dx, dy) in &dirs {
                    let nx = x + dx;
                    let ny = y + dy;
                    if in_bounds(nx, ny, w, h) {
                        let ni = idx(nx, ny, w, h);
                        if grid[ni] == 0 && !visited[ni] {
                            c += 1;
                        }
                    }
                }
                neigh_count[i] = c;
            }
        }
    }

    // DFS with recursion-like stack
    fn next_candidates(
        cur: &Cell,
        w: i32,
        h: i32,
        grid: &Vec<u8>,
        visited: &Vec<bool>,
        dirs: &[(i32, i32)],
        w_tot: usize,
    ) -> Vec<Cell> {
        let mut cand = Vec::new();
        for (dx, dy) in dirs.iter() {
            let nx = cur.x + dx;
            let ny = cur.y + dy;
            if in_bounds(nx, ny, w, h) {
                let ni = idx(nx, ny, w, h);
                if grid[ni] == 0 && !visited[ni] {
                    cand.push(Cell { x: nx, y: ny });
                }
            }
        }
        // Warnsdorff: sort by number of onward moves (fewest first)
        cand.sort_by(|a, b| {
            let ca = onward_moves(*a, w, h, grid, visited, dirs);
            let cb = onward_moves(*b, w, h, grid, visited, dirs);
            ca.cmp(&cb)
        });
        cand
    }

    fn onward_moves(p: Cell, w: i32, h: i32, grid: &Vec<u8>, visited: &Vec<bool>, dirs: &[(i32, i32)]) -> u8 {
        let mut cnt = 0u8;
        for (dx, dy) in dirs.iter() {
            let nx = p.x + dx;
            let ny = p.y + dy;
            if in_bounds(nx, ny, w, h) {
                let ni = idx(nx, ny, w, h);
                if grid[ni] == 0 && !visited[ni] {
                    cnt += 1;
                }
            }
        }
        cnt
    }

    // Main iterative backtracking to avoid deep recursion (still uses call depth ~ N)
    // We'll implement a recursive-like function using explicit stack for clarity.
    // For simplicity, implement a recursive function with Rust stack; N up to ~2500 might overflow.
    // To be safe, implement explicit manual stack frames.

    #[derive(Clone)]
    struct Frame {
        cur: Cell,
        stage: usize, // 0 = first time, 1 = after trying one candidate, etc.
        cand: Vec<Cell>,
    }

    // We'll implement iterative DFS
    let mut stack: Vec<Frame> = Vec::new();
    // First frame is at start
    let start_frame = Frame {
        cur: start,
        stage: 0,
        cand: Vec::new(),
    };
    stack.push(start_frame);

    // We need access to mutable path and visited; we will mutate outside frames
    // To coordinate, we will implement a separate recursive-like function within main scope, but Rust borrow rules complicate.
    // Instead, implement a simple recursive closure using unsafe? Not possible easily.
    // For practicality, implement a straightforward recursive function with potential stack depth up to N.
    // Given Rust's default stack might handle ~ in the thousands; risk but acceptable for contest style.

    // We'll implement a real recursive function
    fn dfs_recursive(
        cur: Cell,
        w: i32,
        h: i32,
        grid: &Vec<u8>,
        visited: &mut Vec<bool>,
        path: &mut Vec<Cell>,
        total_free: usize,
        require_cycle: bool,
        dirs: &[(i32, i32)],
        start: Cell,
        time_limit: &Instant,
        start_time: Instant,
    ) -> bool {
        // Time check
        if start_time.elapsed() > *time_limit {
            return false;
        }

        // If path covers all
        if path.len() == total_free {
            if require_cycle {
                // last cell must be adjacent to start
                let mut ok = false;
                for (dx, dy) in dirs.iter() {
                    let nx = cur.x + dx;
                    let ny = cur.y + dy;
                    if nx == start.x && ny == start.y {
                        ok = true;
                        break;
                    }
                }
                if ok {
                    return true;
                } else {
                    return false;
                }
            } else {
                return true;
            }
        }

        // Generate candidates
        let mut cand = Vec::new();
        for (dx, dy) in dirs.iter() {
            let nx = cur.x + dx;
            let ny = cur.y + dy;
            if in_bounds(nx, ny, w, h) {
                let ni = idx(nx, ny, w, h);
                if grid[ni] == 0 && !visited[ni] {
                    cand.push(Cell { x: nx, y: ny });
                }
            }
        }

        // If no candidates, dead end
        if cand.is_empty() {
            return false;
        }

        // Sort by onward moves (Warnsdorff)
        cand.sort_by(|a, b| {
            let ca = onward_moves(*a, w, h, grid, visited, dirs);
            let cb = onward_moves(*b, w, h, grid, visited, dirs);
            ca.cmp(&cb)
        });

        // Try candidates
        for next in cand {
            // Pruning: simple isolation check: if we move to next, some remaining unvisited cell becomes isolated (no unvisited neighbors). We'll skip hard check for speed; rely on general pruning.

            // Choose
            let ni = idx(next.x, next.y, w, h);
            visited[ni] = true;
            path.push(next);
            if dfs_recursive(
                next,
                w,
                h,
                grid,
                visited,
                path,
                total_free,
                require_cycle,
                dirs,
                start,
                time_limit,
                start_time,
            ) {
                return true;
            }
            // Backtrack
            path.pop();
            visited[ni] = false;
        }

        false
    }

    // Run DFS
    let mut visited_mut = visited; // already marked start as visited
    // path currently has start
    let success = dfs_recursive(
        start,
        w,
        h,
        &grid,
        &mut visited_mut,
        &mut path,
        total_free,
        require_cycle,
        &dirs,
        start,
        &time_budget,
        t0,
    );

    if success {
        // Output path
        println!("{}", path.len());
        for cell in path.iter() {
            println!("{} {}", cell.x, cell.y);
        }
        return;
    } else {
        // If not found within time or dead end, output 0
        println!("0");
        return;
    }
}
