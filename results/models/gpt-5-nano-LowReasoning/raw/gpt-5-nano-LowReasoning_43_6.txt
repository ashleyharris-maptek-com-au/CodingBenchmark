float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal (assume world-space normal)
    float3 N = normalize(input.normal);

    // Light and view directions in world space
    float3 L = normalize( (float3)UBO.lightPos.xyz - input.worldPos );
    float3 V = normalize( (float3)UBO.cameraPos.xyz - input.worldPos );

    // Half-vector
    float3 H = normalize(L + V);

    // Blinn-Phong terms
    float diffuse = max(dot(N, L), 0.0);
    float specular = pow(max(dot(N, H), 0.0), 64.0);

    // Final intensity (grayscale), alpha = 1.0
    float intensity = 0.1 + 0.6 * diffuse + 0.8 * specular;

    return float4(intensity, intensity, intensity, 1.0);
}
