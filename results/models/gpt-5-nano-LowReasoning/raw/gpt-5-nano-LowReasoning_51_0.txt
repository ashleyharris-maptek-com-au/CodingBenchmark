```python
# autopilot_step.py

"""
A lightweight, educational autopilot step for an A320-class twin.
This is intentionally simplified but includes:
 - sensor fusion of triple redundant sensors
 - basic PI-style pitch (elevator/elevator_trim), roll (aileron), yaw (rudder)
 - speed/thrust scheduling toward target_ias_kt (throttle_1/throttle_2)
 - simple altitude/heading capture toward target_alt_ft and target_hdg_deg
 - basic safety constraints (Vne-ish limit, stall-ish behavior, trim handling)
 - persistence across steps via module-level integrators
"""

from math import copysign

# Persistence for integral terms, states across calls
_integral_pitch_error = 0.0
_prev_pitch_error = 0.0

_integral_roll_error = 0.0
_prev_roll_error = 0.0

_integral_yaw_error = 0.0
_prev_yaw_error = 0.0

# Some reasonable defaults / tunable constants
_MAX_ELEVATOR = 1.0
_MIN_ELEVATOR = -1.0

_MAX_AILERON = 1.0
_MIN_AILERON = -1.0

_MAX_RUDDER = 1.0
_MIN_RUDDER = -1.0

# Elevator trim baseline (persistent)
_elevator_trim = 0.0

# Simple Vne-ish safeguard (knots)
_VNE_KIAS = 365.0
# We'll assume target_ias_kt is provided in state as target_ias_kt
# and we can attempt to maintain a reasonable fraction of commanded speed.


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _avg_valid(vals, valid_flags):
    """
    Average values where corresponding valid_flags is truthy.
    """
    total = 0.0
    count = 0
    for v, ok in zip(vals, valid_flags):
        if ok:
            total += v
            count += 1
    if count > 0:
        return total / count
    return None


def _estimate_altitude(state):
    """
    Sensor fusion for altitude:
      - Use barometric alt (3 units) if at least two are valid
      - Else use radar alt (3 units) if at least two valid
      - Else use GPS altitude if valid
      - Fallback to 0.0
    Returns (alt_ft, alt_source) where source is a string.
    """
    # Barometric
    baros = [
        (state.get('baro_alt_1_ft'), state.get('baro_alt_1_valid', False)),
        (state.get('baro_alt_2_ft'), state.get('baro_alt_2_valid', False)),
        (state.get('baro_alt_3_ft'), state.get('baro_alt_3_valid', False)),
    ]
    baro_vals = [v for v, _ in baros]
    baro_vals_valid = [bool(v2) for _, v2 in baros]  # treat non-zero as valid flag; but we actually use *_valid flags below

    baro_valid_flags = [
        bool(state.get('baro_alt_1_valid', False)),
        bool(state.get('baro_alt_2_valid', False)),
        bool(state.get('baro_alt_3_valid', False)),
    ]
    baro_ok = sum(baro_valid_flags) >= 2
    baro_alt = _avg_valid(baro_vals, baro_valid_flags) if baro_ok else None

    if baro_alt is not None:
        return baro_alt, 'baro'

    # Radar
    radars = [
        (state.get('radar_alt_1_ft'), bool(state.get('radar_alt_1_valid', False))),
        (state.get('radar_alt_2_ft'), bool(state.get('radar_alt_2_valid', False))),
        (state.get('radar_alt_3_ft'), bool(state.get('radar_alt_3_valid', False))),
    ]
    radar_vals = [v for v, _ in radars]
    radar_valid_flags = [f for _, f in radars]
    radar_ok = sum(radar_valid_flags) >= 2
    radar_alt = _avg_valid(radar_vals, radar_valid_flags) if radar_ok else None

    if radar_alt is not None:
        return radar_alt, 'radar'

    # GPS altitude as last resort
    gps_alts = [
        state.get('gps_1_alt_ft', None),
        state.get('gps_2_alt_ft', None),
        state.get('gps_3_alt_ft', None)
    ]
    gps_valid = [
        bool(state.get('gps_1_valid', False)),
        bool(state.get('gps_2_valid', False)),
        bool(state.get('gps_3_valid', False)),
    ]
    gps_alt = _avg_valid(gps_alts, gps_valid) if sum(gps_valid) >= 2 else None
    if gps_alt is not None:
        return gps_alt, 'gps'

    # Fallback
    return 0.0, 'fallback'


def _current_ias(state):
    """
    Pick an indicative airspeed from available IAS/Mach data.
    We'll use the average of valid indicated airspeeds (ias_1_kt, ias_2_kt, ias_3_kt)
    or Mach to rough TAS if needed.
    """
    ias = [
        (state.get('ias_1_kt'), bool(state.get('ias_1_valid', False))),
        (state.get('ias_2_kt'), bool(state.get('ias_2_valid', False))),
        (state.get('ias_3_kt'), bool(state.get('ias_3_valid', False))),
    ]
    ias_vals = [v for v, _ in ias]
    ias_valid_flags = [f for _, f in ias]
    avg = _avg_valid(ias_vals, ias_valid_flags)
    if avg is not None:
        return avg

    # If IAS not available, try TAS
    tas = [
        state.get('tas_1_kt', 0.0),
        state.get('tas_2_kt', 0.0),
        state.get('tas_3_kt', 0.0),
    ]
    tas_valid = [True, True, True]  # assume TAS always present if IAS missing
    avg_tas = (_sum := sum(tas)) / 3.0
    return max(0.0, avg_tas)


def autopilot_step(state: dict, dt: float) -> dict:
    """
    Compute autopilot control surfaces given a state dict and timestep dt (seconds).
    Returns a dict with keys:
      elevator, aileron, rudder, throttle_1, throttle_2, elevator_trim, flaps, speedbrake, gear
    """
    global _integral_pitch_error, _prev_pitch_error
    global _integral_roll_error, _prev_roll_error
    global _integral_yaw_error, _prev_yaw_error
    global _elevator_trim

    # Defensive: ensure dt is sane
    if not isinstance(dt, (float, int)) or dt <= 0.0:
        dt = 0.05

    # Read target states (may be missing in tricky scenarios)
    target_alt = state.get('target_alt_ft', 0.0)
    target_hdg = state.get('target_hdg_deg', state.get('mag_heading_deg', 0.0))
    target_ias = state.get('target_ias_kt', 250.0)  # default cruise

    # Sensor fusion for altitude
    current_alt, source_alt = _estimate_altitude(state)

    # Altitude error
    alt_error = target_alt - current_alt

    # Simple PI for pitch to hold altitude
    # We'll map altitude error to elevator command, but also respect stall margin via airspeed check
    _integral_pitch_error += alt_error * dt
    # Proportional term
    Kp_pitch = 0.008  # tuned for small elevators
    Ki_pitch = 0.0006
    pitch_control = Kp_pitch * alt_error + Ki_pitch * _integral_pitch_error

    # Apply elevator trim baseline
    # Elevator_trim acts as a slow baseline adjustment
    _elevator_trim = _clamp(_elevator_trim, -1.0, 1.0)

    elevator_trim_cmd = _elevator_trim  # base trim
    # Total elevator command includes trim and pitch_control
    elevator_cmd = pitch_control + elevator_trim_cmd
    elevator_cmd = _clamp(elevator_cmd, _MIN_ELEVATOR, _MAX_ELEVATOR)

    # Update prev_error (for potential derivative, not used here heavily)
    _prev_pitch_error = alt_error

    # Lateral control: heading hold -> roll and small yaw
    # Heading error (wrap-around)
    head_err = (target_hdg - (state.get('mag_heading_deg', 0.0))) % 360.0
    if head_err > 180.0:
        head_err -= 360.0
    # Desired bank (roll) proportional to heading error
    Kp_roll = 0.02
    desired_roll = Kp_roll * head_err  # -0.5 .. +0.5 roughly for typical headings
    desired_roll = _clamp(desired_roll, -1.0, 1.0)

    # Roll/yaw dihedral damping: use rudder to help stabilize
    # We'll compute rudder as proportional to yaw error (which is the same as heading error here)
    yaw_error_deg = head_err
    Kp_yaw = 0.01
    rudder_cmd = Kp_yaw * yaw_error_deg
    # small additional damp to keep yaw stable
    rudder_cmd = _clamp(rudder_cmd, _MIN_RUDDER, _MAX_RUDDER)

    # Update integral yaw for potential future use (not heavily used here)
    _integral_yaw_error += yaw_error_deg * dt
    _prev_yaw_error = yaw_error_deg

    # Lateral inputs: steering with ailerons for roll
    aileron_cmd = desired_roll
    aileron_cmd = _clamp(aileron_cmd, _MIN_AILERON, _MAX_AILERON)

    # Speed control: simple throttle scheduling to reach target_ias
    current_ias = _current_ias(state)
    if current_ias is None:
        current_ias = 0.0

    speed_error = target_ias - current_ias
    # Proportional+Integral on throttle
    _integral_roll_error += speed_error * dt  # reuse as general throttle integral (not ideal but simple)
    Kp_throttle = 0.004
    Ki_throttle = 0.0002
    throttle_cmd = Kp_throttle * speed_error + Ki_throttle * _integral_roll_error

    # Clamp throttle commands to [0,1]
    throttle_cmd = _clamp(throttle_cmd, 0.0, 1.0)

    # Drive towards safe airspeed: basic stall/overspeed guard
    # If current_ias exceeds Vne, throttle down aggressively and trim nose up slightly to recover
    if current_ias > _VNE_KIAS * 1.0:
        throttle_cmd = 0.6 * throttle_cmd  # cut back
        elevator_cmd = max(elevator_cmd - 0.15, _MIN_ELEVATOR)

    # Simple gear/flap handling (no automation of flaps historically, but provide sane defaults)
    # We'll keep fixed defaults unless there is gear state in state
    gear_down = int(state.get('gear_handle_down', 0) > 0)

    # Output dictionary
    output = {
        # Controls (range [-1,1] for surfaces; 0 means neutral)
        'elevator': float(elevator_cmd),       # nose up / down relative to trim
        'aileron': float(aileron_cmd),         # roll left/right
        'rudder': float(rudder_cmd),           # yaw left/right

        # Throttle: per-engine (we keep a simple, identical throttle by default)
        # If we want to preserve current settings when not updating, we could omit.
        # Here we implement a simple two-engine throttle; both set by same value.
        'throttle_1': float(throttle_cmd),
        'throttle_2': float(throttle_cmd),

        # Elevator trim (persistent baseline)
        'elevator_trim': float(_elevator_trim),

        # Flaps / speedbrake (we keep defaults if not overridden)
        # A320 typical cruise uses flaps at 0; speedbrake at 0 unless needed
        'flaps': int(0),
        'speedbrake': float(0.0),

        # Gear (0 up, 1 down)
        'gear': int(gear_down),
    }

    # Optional: if state contains explicit overrides or to maintain current values,
    # the caller may override by omitting keys. Here we provide complete dict.

    return output
```
