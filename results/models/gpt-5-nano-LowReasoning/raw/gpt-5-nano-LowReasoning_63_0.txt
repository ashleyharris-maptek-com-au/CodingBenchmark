// Node.js TypeScript program to assign warehouse pick jobs to robots.
// Reads JSON from stdin and outputs assignments as described.

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid;
const robotsRaw: { id: number; x: number; y: number }[] = data.robots;
const jobs: Job[] = data.jobs;

// Normalize robots list
const robots: Robot[] = robotsRaw.map(r => ({ id: r.id, x: r.x, y: r.y }));

const height = grid.length;
const width = grid[0]?.length ?? 0;

// Helper: check if cell is in bounds and open
function isOpen(cx: number, cy: number): boolean {
  if (cx < 0 || cy < 0 || cx >= width || cy >= height) return false;
  const ch = grid[cy][cx];
  return ch === '.';
}

// BFS to compute distances from a robot position to all cells
function bfsFrom(rx: number, ry: number): number[][] {
  const dist: number[][] = Array.from({ length: height }, () => Array(width).fill(-1));
  if (!isOpen(rx, ry)) {
    // Robot starts on blocked cell; return all -1
    return dist;
  }

  const qx: number[] = [];
  const qy: number[] = [];
  let head = 0;

  dist[ry][rx] = 0;
  qx.push(rx);
  qy.push(ry);

  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  while (head < qx.length) {
    const x = qx[head];
    const y = qy[head];
    head++;

    const d = dist[y][x];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      if (!isOpen(nx, ny)) continue;
      if (dist[ny][nx] !== -1) continue;
      dist[ny][nx] = d + 1;
      qx.push(nx);
      qy.push(ny);
    }
  }

  return dist;
}

// Precompute distances from each robot
// Map robot id -> 2D distance grid
const robotDistances: Map<number, number[][]> = new Map();
for (const r of robots) {
  const d = bfsFrom(r.x, r.y);
  robotDistances.set(r.id, d);
}

// Prepare max robot id to shape output array accordingly
let maxRobotId = -1;
for (const r of robots) if (r.id > maxRobotId) maxRobotId = r.id;
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

// Determine best robot for each job (greedy: assign to robot with shortest distance)
for (const job of jobs) {
  let bestRobotId: number | null = null;
  let bestDist = Number.POSITIVE_INFINITY;

  for (const r of robots) {
    const dgrid = robotDistances.get(r.id);
    if (!dgrid) continue;
    const dd = dgrid[job.y]?.[job.x];
    if (dd !== undefined && dd >= 0) {
      if (dd < bestDist || (dd === bestDist && r.id < (bestRobotId ?? Infinity))) {
        bestDist = dd;
        bestRobotId = r.id;
      }
    }
  }

  // If no reachable robot found, fall back to any robot (shouldn't happen in valid inputs)
  if (bestRobotId == null) {
    // Choose first robot by id with any path (even if -1 would be invalid, but we try)
    for (const r of robots) {
      const dgrid = robotDistances.get(r.id);
      const dd = dgrid?.[job.y]?.[job.x];
      if (dd !== undefined) {
        bestRobotId = r.id;
        bestDist = dd;
        break;
      }
    }
  }

  // If still null (completely unreachable), skip (should not occur per problem statement)
  if (bestRobotId == null) continue;

  // Assign job to chosen robot
  assignments[bestRobotId].push(job.id);
}

// Output result as JSON
const output = { assignments };
process.stdout.write(JSON.stringify(output));
