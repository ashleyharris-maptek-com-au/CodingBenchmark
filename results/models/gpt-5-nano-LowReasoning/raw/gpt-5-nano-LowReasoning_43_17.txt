float4 main(PSInput input) : SV_Target0
{
    // Brick tiling in UV space: 8x16 scale
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Determine brick column/row
    float row = floor(scaledUV.y);
    bool isOddRow = (fmod(row, 2.0) > 0.5);

    // Apply half-brick offset on odd rows
    float localX = frac(scaledUV.x);
    if (isOddRow) {
        localX = frac(scaledUV.x - 0.5);
    }
    float localY = frac(scaledUV.y);

    // Mortar thickness
    const float mortarX = 0.05;
    const float mortarY = 0.1;

    // Determine if current point is in mortar
    bool inMortar = (localX < mortarX) || (localX > (1.0 - mortarX)) ||
                    (localY < mortarY) || (localY > (1.0 - mortarY));

    // Colors
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = inMortar ? mortarColor : brickColor;

    // Simple Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize((float3)lightPos.xyz - input.worldPos);
    float NdotL = max(dot(N, L), 0.0);
    float3 shaded = baseColor * (0.4 + 0.6 * NdotL);

    return float4(shaded, 1.0);
}
