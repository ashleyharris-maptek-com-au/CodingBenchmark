; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_color %outColor
OpExecutionMode %main OriginUpperLeft

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%one_f = OpConstant %float 1.0
%color4 = OpTypeVector %float 4  ; placeholder type for composite (will be same as vec4)
%_vec4 = %vec4

; Function
%main = OpFunction %void None %func
%entry = OpLabel

; Build vec4 from vec3 and 1.0
%col3 = OpLoad %vec3 %in_color
%out4 = OpCompositeConstruct %vec4 %col3 %one_f
OpStore %outColor %out4

OpReturn
OpFunctionEnd

; Decorations
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
