; SPIR-V assembly for a geometry shader that expands a single input point into a 4-vertex diamond
; Each emitted vertex uses the input point color.

; Note: This shader targets Vulkan 1.0 with a Geometry shader.
; It reads one input point (gl_Position from vertex shader) and emits a diamond
; composed of 4 vertices in a triangle strip: top, right, left, bottom.

; ---- Header ----
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main"
; Interface (input/output) declarations are listed below
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputPoints
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; ---- Types and constants ----
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeUInt
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0

; h = 0.05
%float_005 = OpConstant %float 0.05

; ---- Per-Vertex Input (gl_PerVertex_in) ----
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
; Member 0 is BuiltIn Position
%gl_PerVertex_in_member0 = OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Array of per-vertex inputs (size depends on primitive)
; Vertex input array: arr_pv_in[0]
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Access: gl_in[0].gl_Position
%ptr_glin_v4 = OpTypePointer Input %v4float
%ptr_in_struct = OpTypePointer Input %gl_PerVertex_in
%idx_0 = OpConstant %uint 0

; Access chain types (for nested access)
%ptr_in_struct_from_arr = OpAccessChain %ptr_in_arr_pv %gl_in %idx_0
%ptr_glin_v4 = OpAccessChain %ptr_glin_v4 %ptr_in_struct_from_arr %int_0

; ---- User inputs (arrays) ----
layout(location = 0) in vec3 worldPos[];   ; per-vertex positions in clip space
layout(location = 1) in vec3 normal[];     ; not used in this task but declared for completeness
layout(location = 2) in vec3 color[];      ; per-vertex colors

; ---- Output (per-emitted vertex) ----
; Output gl_Position block is modelled via gl_PerVertex_out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_glout_v4 = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_glout_v4 Output
; Output color per vertex
layout(location = 0) out vec3 outColor
%ptr_out_color = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_color Output

; ---- Main function ----
%void_fn = OpTypeFunction %void
%main_ty = OpTypeFunction %void

; Entry point
%main = OpFunction %void None %main_ty

; Local variables
; position for input vertex (x,y)
%pos_in = OpVariable %ptr_in_arr_pv Input
; current input vertex position (vec4)
%in_pos = OpVariable %ptr_glin_v4 Input
; input color (vec3)
%in_color = OpVariable %ptr_out_color Output ; reused storage for color per-vertex before EmitVertex

; ---- Function body begins ----
OpLabel

; Step 1: Read input gl_Position for the single input vertex
; Load gl_in[0].gl_Position
%gl_pos = OpLoad %v4float %ptr_glin_v4

; Extract x and y from gl_Position
%x = OpCompositeExtract %float %gl_pos 0
%y = OpCompositeExtract %float %gl_pos 1

; Read input color from color[] at index 0 (since InputPoints has 1 vertex)
%idx_pt = OpConstant %uint 0
%ptr_color_arr = OpAccessChain %ptr_in_arr_pv %gl_in %idx_pt
%color_ptr = OpAccessChain %ptr_out_color %ptr_color_arr  ; not exact; we will fetch color from array below

; Instead, fetch color directly from the input color array
%color_in_ptr = OpAccessChain %ptr_out_color %worldPos  ; placeholder to satisfy syntax

; Properly, fetch from color[]:
%ptr_color_vec = OpAccessChain %v3float %color %idx_pt
%col = OpLoad %v3float %ptr_color_vec
; For correctness in SPIR-V, we must replace the above with valid access; due to brevity, we assume %col is loaded

; Step 2: Compute diamond offsets
%hx = OpFAdd %float %x %float_005 ; x + h
%hy = OpFAdd %float %y %float_005
%hx_neg = OpFSub %float %x %float_005 ; x - h
%hy_neg = OpFSub %float %y %float_005

; Corner positions:
; Top: (x, y+h)
%top_x = %x
%top_y = %hy
; Right: (x+h, y)
%right_x = %hx
%right_y = %y
; Left: (x-h, y)
%left_x = %hx_neg
%left_y = %y
; Bottom: (x, y-h)
%bottom_x = %x
%bottom_y = %hy_neg

; Build four gl_Position vectors for the diamond:
; Top
%top = OpCompositeConstruct %v4float %top_x %top_y %gl_pos 0
; Right
%right = OpCompositeConstruct %v4float %right_x %right_y %gl_pos 2
; Left
%left = OpCompositeConstruct %v4float %left_x %left_y %gl_pos 2
; Bottom
%bottom = OpCompositeConstruct %v4float %bottom_x %bottom_y %gl_pos 2

; Since we must create vec4 with z,w from original gl_Position
%gl_pos_z = OpCompositeExtract %float %gl_pos 2
%gl_pos_w = OpCompositeExtract %float %gl_pos 3

%top4 = OpCompositeConstruct %v4float %top_x %top_y %gl_pos_z %gl_pos_w
%right4 = OpCompositeConstruct %v4float %right_x %right_y %gl_pos_z %gl_pos_w
%left4  = OpCompositeConstruct %v4float %left_x  %left_y  %gl_pos_z %gl_pos_w
%bottom4= OpCompositeConstruct %v4float %bottom_x %bottom_y %gl_pos_z %gl_pos_w

; Step 3: Emit four vertices in order: top, right, left, bottom
; Emit Vertex 1 (Top)
OpStore %gl_out %top4
OpStore %out_color %col
OpEmitVertex

; Emit Vertex 2 (Right)
OpStore %gl_out %right4
OpStore %out_color %col
OpEmitVertex

; Emit Vertex 3 (Left)
OpStore %gl_out %left4
OpStore %out_color %col
OpEmitVertex

; Emit Vertex 4 (Bottom)
OpStore %gl_out %bottom4
OpStore %out_color %col
OpEmitVertex

; End primitive (triangle strip)
OpEndPrimitive

; Function end
OpFunctionEnd

; End of file

; to the exact SPIR-V bindings and interface requirements (especially the per-vertex outputs). The example demonstrates the essential sequencing: read input position and color, compute offsets, construct four positions, emit sequential vertices, and end the primitive.
