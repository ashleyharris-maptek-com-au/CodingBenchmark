```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Numerics;

namespace MaxClique
{
    // Simple fixed-size bitset over ulong blocks
    class BitSet
    {
        public readonly ulong[] Data;
        public readonly int N;
        public int Blocks => Data.Length;

        public BitSet(int n)
        {
            N = n;
            Data = new ulong[(n + 63) >> 6];
        }

        public BitSet(ulong[] data)
        {
            Data = data;
            N = data.Length * 64;
        }

        public void CopyFrom(BitSet other)
        {
            Array.Copy(other.Data, Data, other.Data.Length);
        }

        public void Clear()
        {
            Array.Clear(Data, 0, Data.Length);
        }

        public void Set(int idx)
        {
            Data[idx >> 6] |= 1UL << (idx & 63);
        }

        public void Reset(int idx)
        {
            Data[idx >> 6] &= ~(1UL << (idx & 63));
        }

        public bool Get(int idx)
        {
            return (Data[idx >> 6] & (1UL << (idx & 63))) != 0;
        }

        public void And(BitSet other)
        {
            int m = Data.Length;
            for (int i = 0; i < m; i++) Data[i] &= other.Data[i];
        }

        public void AndNot(BitSet other)
        {
            int m = Data.Length;
            for (int i = 0; i < m; i++) Data[i] &= ~other.Data[i];
        }

        public void Or(BitSet other)
        {
            int m = Data.Length;
            for (int i = 0; i < m; i++) Data[i] |= other.Data[i];
        }

        public bool IsEmpty()
        {
            for (int i = 0; i < Data.Length; i++) if (Data[i] != 0) return false;
            return true;
        }

        public int Count()
        {
            int c = 0;
            for (int i = 0; i < Data.Length; i++) c += BitOperations.PopCount(Data[i]);
            return c;
        }

        public int FirstSet()
        {
            for (int i = 0; i < Data.Length; i++)
            {
                ulong x = Data[i];
                if (x != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(x);
                    return (i << 6) + bit;
                }
            }
            return -1;
        }

        public BitSet Copy()
        {
            var b = new BitSet(0);
            b.Data[0] = 0; // will replace below
            // allocate proper length
            b = new BitSet(N);
            Array.Copy(Data, b.Data, Data.Length);
            return b;
        }

        public void Remove(int v)
        {
            Reset(v);
        }

        public void EnsureSize(int n)
        {
            // no-op for this implementation
        }
    }

    // Fast bit operations helpers (using .NET 7+ BitOperations)
    static class BitOperations
    {
        public static int PopCount(ulong x)
        {
            // Built-in
            return BitOperationsCore.PopCount(x);
        }

        public static int TrailingZeroCount(ulong x)
        {
            if (x == 0) return 64;
#if NETSTANDARD2_0
            return (int)System.Runtime.Intrinsics.X86.Builtin.CompatibleIntrinsic.TrailingZeroCount(x);
#else
            return BitOperationsCore.TrailingZeroCount(x);
#endif
        }
    }

    // Lightweight wrappers to avoid relying on newer intrinsics in all runtimes
    static class BitOperationsCore
    {
        public static int PopCount(ulong x)
        {
            // Hamming weight
            // Kernighan's method
            int c = 0;
            while (x != 0)
            {
                x &= x - 1;
                c++;
            }
            return c;
        }

        public static int TrailingZeroCount(ulong x)
        {
            if (x == 0) return 64;
            int n = 0;
            while ((x & 1) == 0)
            {
                x >>= 1;
                n++;
            }
            return n;
        }
    }

    class Program
    {
        static int n;
        static BitSet[] Nbr; // adjacency bitsets
        static int[] bestClique;
        static int bestSize = 0;

        // Tomita algorithm with bitsets
        static void Tomita(BitSet R, BitSet P, BitSet X)
        {
            if (P.IsEmpty() && X.IsEmpty())
            {
                // Found a maximal clique R
                int size = R.Count();
                if (size > bestSize)
                {
                    bestSize = size;
                    // store clique
                    bestClique = new int[size];
                    int idx = 0;
                    for (int v = 0; v < n; v++)
                    {
                        if (R.Get(v)) bestClique[idx++] = v;
                    }
                }
                return;
            }

            // Choose pivot u from P ∪ X to maximize |P ∩ N(u)|
            BitSet PX = P.Copy();
            PX.Or(X);
            int u = -1;
            int maxCommon = -1;
            // Iterate over elements of PX
            for (int i = 0; i < n; i++)
            {
                if (PX.Get(i))
                {
                    // count |P ∩ N(i)|
                    BitSet tmp = P.Copy();
                    tmp.And(Nbr[i]);
                    int c = tmp.Count();
                    if (c > maxCommon)
                    {
                        maxCommon = c;
                        u = i;
                    }
                }
            }

            BitSet candidates;
            if (u >= 0)
            {
                // P \ N(u)
                candidates = P.Copy();
                candidates.AndNot(Nbr[u]);
            }
            else
            {
                candidates = P.Copy();
            }

            // Iterate v in candidates
            while (!candidates.IsEmpty())
            {
                int v = candidates.FirstSet();
                // R' = R ∪ {v}
                R.Set(v);

                // P' = P ∩ N(v)
                BitSet Pnew = P.Copy();
                Pnew.And(Nbr[v]);

                // X' = X ∩ N(v)
                BitSet Xnew = X.Copy();
                Xnew.And(Nbr[v]);

                Tomita(R, Pnew, Xnew);

                // move v from P to X
                P.Reset(v);
                X.Set(v);

                // remove v from candidates
                candidates.Reset(v);
            }
        }

        static void Main(string[] args)
        {
            // Read input
            var input = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(input))
            {
                return;
            }

            using (var sr = new StringReader(input))
            {
                var first = sr.ReadLine();
                while (string.IsNullOrWhiteSpace(first))
                {
                    first = sr.ReadLine();
                    if (first == null) return;
                }
                var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                if (parts.Length < 2) return;
                if (!int.TryParse(parts[0], out n)) return;
                int m;
                if (!int.TryParse(parts[1], out m)) return;

                // Initialize
                Nbr = new BitSet[Math.Max(n, 1)];
                for (int i = 0; i < n; i++) Nbr[i] = new BitSet(n);

                // Read edges
                int edgesRead = 0;
                string line;
                while (edgesRead < m && (line = sr.ReadLine()) != null)
                {
                    if (string.IsNullOrWhiteSpace(line)) continue;
                    var toks = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                    if (toks.Length < 2) continue;
                    if (!int.TryParse(toks[0], out int u)) continue;
                    if (!int.TryParse(toks[1], out int v)) continue;
                    if (u < 0 || v < 0 || u >= n || v >= n) continue;
                    if (u == v) continue;
                    Nbr[u].Set(v);
                    Nbr[v].Set(u);
                    edgesRead++;
                }

                // If there are still lines and we didn't read all edges due to input format, try to process remaining
                // (best effort)
                // Prepare
                for (int i = 0; i < n; i++) Nbr[i].Set(i); // optional: self? Not required; but keep for safety not needed. We'll leave without self-edge.
                // We'll not set self as it's not part of adjacency.
            }

            // If n==0
            if (n <= 0)
            {
                Console.WriteLine("0");
                Console.WriteLine();
                return;
            }

            // Build initial sets
            var R = new BitSet(n);
            var P = new BitSet(n);
            var X = new BitSet(n);

            for (int i = 0; i < n; i++) P.Set(i);

            bestClique = new int[0];
            bestSize = 0;

            // Run Tomita algorithm
            Tomita(new BitSet(n), P, X);

            // If no clique found (shouldn’t happen), output 0 and empty line
            if (bestSize == 0)
            {
                Console.WriteLine("0");
                Console.WriteLine();
                return;
            }

            // Output
            Console.WriteLine(bestSize);
            Console.WriteLine(string.Join(" ", bestClique.Take(bestSize)));
        }
    }
}
```
