```cpp
#include <bits/stdc++.h>
using namespace std;

/*
  2D Cutting Stock / Rectangle Bin Packing (Simple Shelf-based heuristic)
  - Boards: fixed size (boardW x boardH)
  - Rectangles: w x h, allowed to rotate 90 degrees
  - Output: number of boards used, then for each rectangle (in input order)
            the board index (0-based), x, y, rotated(0/1)

  Strategy:
  - Sort rectangles by a simple heuristic (descending by max(w,h)) to improve packing.
  - Pack using a simple shelf algorithm per board:
      - Each board is filled shelf by shelf from top to bottom.
      - Each shelf has a height equal to the tallest rectangle in that shelf.
      - Rectangles are placed left-to-right on the current shelf if they fit; otherwise, start a new shelf.
      - When starting a new shelf, push the previous shelf downward and create a new shelf at the current top.
  - Try to place each rectangle in the current shelf in two orientations (no rotation, then rotated). If it doesn't fit in the current shelf, attempt to start a new shelf.
  - If it doesn't fit on the current board, open a new board and place there.
  - If a rectangle cannot fit in a board in any orientation (larger than board in both dimensions), the problem is invalid for given board (we assume inputs are valid with at least one orientation fitting on some board).

  Complexity:
  - Sorting dominates: O(N log N)
  - Packing: linear in N with very small constants per item.
  - Scales to tens of thousands of rectangles within time limit.

  Note:
  - We preserve original input order when outputting placements by recording placements indexed by original id.
  - Board indices are 0-based in output as requested.
*/

struct Rect {
    int w, h;
    int id;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int boardW, boardH;
    if(!(cin >> boardW >> boardH)) return 0;

    // Read rectangles
    vector<Rect> rects;
    rects.reserve(1000);
    int idx = 0;
    while (true) {
        int w, h;
        if(!(cin >> w >> h)) break;
        rects.push_back({w, h, idx++});
    }
    int N = (int)rects.size();
    if (N == 0) {
        cout << 0 << "\n";
        return 0;
    }

    // Sort by a heuristic: by max(w,h) descending (to place big items first)
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    // Create a copy of input to sort with value
    vector<int> sortKey(N);
    for (int i = 0; i < N; ++i) {
        sortKey[i] = max(rects[i].w, rects[i].h);
    }
    sort(order.begin(), order.end(), [&](int a, int b){
        if (sortKey[a] != sortKey[b]) return sortKey[a] > sortKey[b];
        // tie-breaker: larger area first
        int va = rects[a].w * rects[a].h;
        int vb = rects[b].w * rects[b].h;
        if (va != vb) return va > vb;
        return a < b;
    });

    // Placement result arrays (per original id)
    struct Placement { int board; int x; int y; int rotated; };
    vector<Placement> placed(N, {-1,0,0,0});

    // Packer state per current board
    struct Shelf {
        int y;          // top position of this shelf
        int height;     // shelf height
        int usedX;      // next x position on this shelf
    };

    vector<Shelf> shelves; // shelves for current board
    int currentBoard = 0;
    int placedCount = 0;

    auto canPlaceOnShelf = [&](int rw, int rh, const Shelf& s)->bool {
        // fits in shelf horizontally?
        if (rw <= boardW - s.usedX && rh <= s.height) return true;
        // If shelf has height zero (first in board), allow placing with height rh and adjust shelf height
        // But we initialize shelf height to 0 only at first; handle separately in logic
        return false;
    };

    // We will implement the shelf algorithm with explicit handling:
    // Each board has shelves with a fixed vertical layout from top to bottom.
    // shelf.y is current top offset, shelves accumulate downward as we add shelves.
    // We'll track an overall used height on board via sum of shelf heights.

    // Start with a new board
    shelves.clear();
    int boardTopY = 0;      // cumulative height used so far (sum of shelf heights)
    int boardUsedHeight = 0; // total height used on board so far

    // Helper to start a new shelf on current board
    auto startNewShelf = [&](int shelfHeight, int &allocatedX) -> bool {
        // If starting a new shelf would exceed board height, fail
        if (boardUsedHeight + shelfHeight > boardH) return false;
        // New shelf is placed below existing shelves
        Shelf s;
        s.y = boardUsedHeight;
        s.height = shelfHeight;
        s.usedX = 0;
        shelves.push_back(s);
        boardUsedHeight += shelfHeight;
        allocatedX = 0;
        return true;
    };

    // Place a rectangle on the current board
    auto placeOnCurrentBoard = [&](int rw, int rh, int &outX, int &outY) -> bool {
        // Try to place on the last shelf if possible
        if (!shelves.empty()) {
            Shelf &s = shelves.back();
            if (rw <= boardW - s.usedX && rh <= s.height) {
                outX = s.usedX;
                outY = s.y;
                s.usedX += rw;
                return true;
            }
        }
        // Try to place on a new shelf (height is rh or maybe larger if we want to support bigger height)
        // Create a new shelf with height = rh (or possibly larger if needed to fit bigger rect)
        int neededShelfH = rh;
        int newUsedX;
        if (startNewShelf(neededShelfH, newUsedX)) {
            // Place at start of new shelf
            Shelf &s = shelves.back();
            outX = s.usedX;
            outY = s.y;
            s.usedX += rw;
            return true;
        }
        // Could not place on current board
        return false;
    };

    // We must also attempt rotation for each rectangle
    // Process in sorted order
    bool mustOpenNewBoard = false;
    for (int kIdx = 0; kIdx < N; ++kIdx) {
        int i = order[kIdx];
        int w = rects[i].w;
        int h = rects[i].h;
        bool rotated = false;
        int rw = w, rh = h;

        // Helper to try place an oriented rectangle on current board
        auto tryPlaceOrientation = [&](int rw, int rh, bool forceNewShelfIfNeeded) -> pair<bool, pair<int,int>> {
            int px, py;
            // First try on current board
            if (placeOnCurrentBoard(rw, rh, px, py)) {
                return {true, {px, py}};
            }
            // If not possible, try to start a new shelf explicitly if allowed
            if (forceNewShelfIfNeeded) {
                int newX;
                int newY;
                if (startNewShelf(rh, newX)) { // shelf height rh
                    Shelf &s = shelves.back();
                    // Place at start of this new shelf
                    newX = s.usedX;
                    newY = s.y;
                    s.usedX += rw;
                    return {true, {newX, newY}};
                }
            }
            return {false, {0,0}};
        };

        // Try no rotation first
        bool placedThisRect = false;
        int finalBoard = currentBoard;
        int finalX = -1, finalY = -1;
        // Ensure there is at least one board opened
        if (placedCount == 0 && shelves.empty()) {
            // Start first board with an initial shelf of height 0; we'll create on first placement
        }

        // Try without rotation
        {
            auto res = tryPlaceOrientation(rw, rh, true);
            if (res.first) {
                placedThisRect = true;
                finalX = res.second.first;
                finalY = res.second.second;
                rotated = false;
            }
        }

        // If not placed, try rotation
        if (!placedThisRect) {
            rw = h; rh = w;
            auto res = tryPlaceOrientation(rw, rh, true);
            if (res.first) {
                placedThisRect = true;
                finalX = res.second.first;
                finalY = res.second.second;
                rotated = true;
            }
        }

        if (!placedThisRect) {
            // Need to open a new board and place there (try both orientations)
            if (!shelves.empty()) {
                // Start a new board: reset all per-board state
            }
            // Reset board state
            shelves.clear();
            boardUsedHeight = 0;
            // Start by placing on new board with new shelf
            // Try no rotation on new board
            bool placedOnNewBoard = false;
            int newX, newY;
            // Try without rotation
            if (startNewShelf(h, newX)) {
                Shelf &s = shelves.back();
                newX = s.usedX;
                newY = s.y;
                s.usedX += w;
                placedOnNewBoard = true;
                finalX = newX;
                finalY = newY;
                rotated = false;
            }
            // If not yet placed, try rotation
            if (!placedOnNewBoard) {
                int rw2 = h, rh2 = w;
                int newX2, newY2;
                if (startNewShelf(rh2, newX2)) {
                    Shelf &s = shelves.back();
                    newX2 = s.usedX;
                    newY2 = s.y;
                    s.usedX += rw2;
                    finalX = newX2;
                    finalY = newY2;
                    rotated = true;
                    placedOnNewBoard = true;
                }
            }
            if (!placedOnNewBoard) {
                // If still cannot place on a fresh board, problem input too large for a single board
                // We'll abort gracefully
                cerr << "Rectangle too large for board on its own. Aborting.\n";
                return 0;
            } else {
                // Record placement for this rect
                placed[i] = {currentBoard, finalX, finalY, rotated ? 1 : 0};
                placedCount++;
                // Continue to next rectangle
                continue;
            }
        }

        // If we reached here, the rectangle was placed on current board (not a fresh board)
        if (finalX >= 0) {
            placed[i] = {currentBoard, finalX, finalY, rotated ? 1 : 0};
            placedCount++;
        } else {
            // Should not happen
            cerr << "Unexpected: placement coordinates not found.\n";
        }

        // If we used a board that had no initial board index yet, assign
        // Note: We track current boards implicitly by the index currentBoard. When we open a new board above, we increment currentBoard.
        // For simplicity, this implementation treats "opening a new board" by resetting board state and assigning to currentBoard+1.
        // But we didn't implement an explicit increment here except in the new-board branch. We'll handle that:

        // If we placed on a fresh board earlier (the code path above with new board), we would have updated currentBoard accordingly.
        // For the normal path (existing board), we keep currentBoard as is.
    }

    // The above implementation has a complexity in managing multiple boards and their shelf states robustly.
    // To ensure correctness and simplicity, we re-run a clean, clearer packing loop below that uses a straightforward per-board shelf structure.

    // Re-implement cleanly:

    // Prepare data structures anew
    struct BoardState {
        vector<Shelf> shelves;
        int usedHeight; // cumulative height used (sum of shelf heights)
        int width;
        int height;
        BoardState(int w, int h): usedHeight(0), width(w), height(h) {}
    };

    vector<BoardState> boards;
    boards.reserve(max(1, (int)ceil((double)N/4)));

    // Reset and do a fresh packing
    vector<Placement> finalPlaced(N, {-1,0,0,0});
    int boardsUsed = 0;
    auto placeRect = [&](int rw, int rh, int rectId, int &outBoard, int &outX, int &outY, int rotFlag) -> bool {
        // Try each existing board
        for (int b = 0; b < (int)boards.size(); ++b) {
            auto &bs = boards[b];
            // Try to place on existing shelves
            // Try on last shelf
            if (!bs.shelves.empty()) {
                Shelf &s = bs.shelves.back();
                if (rw <= bs.width - s.usedX && rh <= s.height) {
                    outBoard = b;
                    outX = s.usedX;
                    outY = s.y;
                    s.usedX += rw;
                    rotFlag = rotFlag;
                    return true;
                }
            }
            // Try to start a new shelf on this board
            int newShelfHeight = rh;
            int newUsedX;
            // Check if we can create a new shelf without exceeding board height
            if (bs.usedHeight + newShelfHeight <= bs.height) {
                Shelf s;
                s.y = bs.usedHeight;
                s.height = newShelfHeight;
                s.usedX = 0;
                bs.shelves.push_back(s);
                bs.usedHeight += newShelfHeight;
                // Place at start of new shelf
                Shelf &ns = bs.shelves.back();
                outBoard = b;
                outX = ns.usedX;
                outY = ns.y;
                ns.usedX += rw;
                return true;
            }
        }

        // Try to open a new board
        return false;
    };

    // Initialize first board
    if (N > 0) {
        boards.clear();
        boards.emplace_back(boardW, boardH);
        boards[0].shelves.clear();
        // We start with none shelves yet; we will create one on first placement
    }

    // We perform a clean two-pass packing: first pass with no rotation priority, then with rotation if needed.
    // But to keep it simple and fast, we will place using a very lightweight approach:
    // - For each rect in sorted order, try to place on existing boards by attempting:
    //     a) place on last shelf of a board
    //     b) start a new shelf on that board
    // - If none fit, open a new board and place as first shelf
    auto tryPlaceOnBoards = [&](int w, int h, int id, int &boardOut, int &xOut, int &yOut, int &rotOut)->bool {
        // Try existing rotation choice: first (w,h), then (h,w)
        int tryW[2] = {w, h};
        int tryH[2] = {h, w};
        for (int t = 0; t < 2; ++t) {
            int rw = tryW[t];
            int rh = tryH[t];
            for (int b = 0; b < (int)boards.size(); ++b) {
                auto &bs = boards[b];
                // Try on last shelf
                if (!bs.shelves.empty()) {
                    auto &s = bs.shelves.back();
                    if (rw <= bs.width - s.usedX && rh <= s.height) {
                        boardOut = b;
                        xOut = s.usedX;
                        yOut = s.y;
                        s.usedX += rw;
                        rotOut = (t==1);
                        return true;
                    }
                }
                // Try to start a new shelf on this board
                int newShelfH = rh;
                if (bs.usedHeight + newShelfH <= bs.height) {
                    Shelf ns;
                    ns.y = bs.usedHeight;
                    ns.height = newShelfH;
                    ns.usedX = 0;
                    bs.shelves.push_back(ns);
                    bs.usedHeight += newShelfH;
                    auto &s2 = bs.shelves.back();
                    boardOut = b;
                    xOut = s2.usedX;
                    yOut = s2.y;
                    s2.usedX += rw;
                    rotOut = (t==1);
                    return true;
                }
            }
        }
        return false;
    };

    // Now perform the actual placement in sorted order
    // Reinitialize boards
    boards.clear();
    boards.emplace_back(boardW, boardH);
    boards[0].shelves.clear();
    finalPlaced.assign(N, {-1,0,0,0});

    // We'll also keep a map from sorted index to original id for output
    for (int idxInOrder = 0; idxInOrder < N; ++idxInOrder) {
        int i = order[idxInOrder];
        int w = rects[i].w;
        int h = rects[i].h;
        int bestBoard=-1, bestX=-1, bestY=-1, rot=-1;

        // Try place without rotation first on existing boards
        // We'll attempt both orientations in one pass
        bool ok = false;
        // Try orientation (w,h) and existing boards
        for (int t = 0; t < 2 && !ok; ++t) {
            int rw = (t==0) ? w : h;
            int rh = (t==0) ? h : w;
            int boardOut, xOut, yOut;
            int rotOut;
            if (tryPlaceOnBoards(rw, rh, i, boardOut, xOut, yOut, rotOut)) {
                bestBoard = boardOut;
                bestX = xOut;
                bestY = yOut;
                rot = rotOut;
                ok = true;
            }
        }

        if (!ok) {
            // Open a new board
            boards.emplace_back(boardW, boardH);
            boards.back().shelves.clear();
            int boardOut, xOut, yOut, rotOut;
            // Try to place on new board (first shelf)
            // Try (w,h)
            if (w <= boardW && h <= boardH) {
                // create first shelf height = h
                Shelf s;
                s.y = 0;
                s.height = h;
                s.usedX = 0;
                boards.back().shelves.push_back(s);
                boards.back().usedHeight = h;
                boards.back().shelves.back().usedX = w;
                boardOut = (int)boards.size()-1;
                xOut = 0;
                yOut = 0;
                rotOut = 0;
                bestBoard = boardOut;
                bestX = xOut;
                bestY = yOut;
                rot = rotOut;
            } else if (h <= boardW && w <= boardH) {
                // try rotated on new board
                Shelf s;
                s.y = 0;
                s.height = w;
                s.usedX = 0;
                boards.back().shelves.push_back(s);
                boards.back().usedHeight = w;
                boards.back().shelves.back().usedX = h;
                boardOut = (int)boards.size()-1;
                xOut = 0;
                yOut = 0;
                rotOut = 1;
                bestBoard = boardOut;
                bestX = xOut;
                bestY = yOut;
                rot = rotOut;
            } else {
                // cannot place on a single board (shouldn't happen if board is large enough)
                cerr << "Rectangle too large for board even after rotation. Exiting.\n";
                return 0;
            }
        }

        // Record placement
        finalPlaced[i] = {bestBoard, bestX, bestY, rot};
    }

    // Compute total boards used (the maximum board index + 1 that has placements)
    int maxBoardIndex = -1;
    for (int i = 0; i < N; ++i) {
        maxBoardIndex = max(maxBoardIndex, finalPlaced[i].board);
    }
    int totalBoards = maxBoardIndex + 1;
    if (totalBoards < 1) totalBoards = 1;

    // Output
    cout << totalBoards << "\n";
    for (int i = 0; i < N; ++i) {
        cout << finalPlaced[i].board << " "
             << finalPlaced[i].x << " "
             << finalPlaced[i].y << " "
             << finalPlaced[i].rotated << "\n";
    }

    return 0;
}
```

Explanation of the approach:
- We implement a scalable, fast heuristic suitable for large inputs.
- Rectangles can be rotated by 90 degrees.
- Packing is done with a simple shelf-based algorithm per board:
  - Each board is filled shelf by shelf from top to bottom.
  - Each shelf has a height equal to the tallest rectangle placed on it.
  - Rectangles are placed left-to-right on the current shelf if they fit; otherwise, a new shelf is started (or a new board if needed).
- We sort rectangles by a simple heuristic (descending by max side) to improve packing density.
- The solution preserves the input order for the final output by tracking placements per original rectangle id.
