; SPIR-V assembly for a geometry shader that expands each input line into a tapered ribbon
; - Geometry: 8 radial line segments from center outward
; - Each input line (2 vertices) provides worldPos and color per vertex
; - Output is a 4-vertex triangle strip forming a tapered ribbon between p0 and p1
; - Colors: start vertices use c0, end vertices use c1

; NOTE: This is a complete SPIR-V assembly text block. It assumes Vulkan 1.0 compatible tooling
; (spirv-as). The shader uses a single invocation per input primitive and emits a 4-vertex strip.

; ---------------------------
; SPIR-V header
; ---------------------------

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos0 %worldPos1 %col0 %col1 %gl_out0 %gl_out1 %gl_out2 %gl_out3
               ; Interfaces: user inputs/outputs (positions and colors), and outputs
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputPoints
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; ---------------------------
; Types and constants
; ---------------------------

%void       = OpTypeVoid
%func       = OpTypeFunction
%float      = OpTypeFloat 32
%v2float     = OpTypeVector %float 2
%v3float     = OpTypeVector %float 3
%v4float     = OpTypeVector %float 4
%bool       = OpTypeBool
%m8        = OpTypeMatrix %float 4 4

; gl_PerVertex_in / out blocks
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; Access: OpAccessChain %ptr_in_v4 %gl_in %int_i %int_0

; Per-vertex inputs (provided by vertex shader)
; layout(location = 0) in vec3 worldPos[];
; layout(location = 1) in vec3 normal[];
; layout(location = 2) in vec3 color[];

%worldPos_type = OpTypeArray %v3float %uint_2
%ptr_in_worldPos = OpTypePointer Input %worldPos_type
%worldPos0 = OpVariable %ptr_in_worldPos Input
%worldPos1 = OpAccessChain %ptr_in_worldPos %worldPos0 %int_0
; We'll instead expose two concrete variables for interface clarity
; (If the assembler requires explicit arrays, you can adapt)
; For simplicity, declare as separate inputs
%ptr_in_worldPos0 = OpTypePointer Input %v3float
%worldPos0 = OpVariable %ptr_in_worldPos0 Input
%ptr_in_worldPos1 = OpTypePointer Input %v3float
%worldPos1 = OpVariable %ptr_in_worldPos1 Input

%ptr_in_color0 = OpTypePointer Input %v3float
%color0 = OpVariable %ptr_in_color0 Input
%ptr_in_color1 = OpTypePointer Input %v3float
%color1 = OpVariable %ptr_in_color1 Input

; Outputs (per-vertex)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_color = OpTypePointer Output %v3float
%gl_out_color0 = OpVariable %ptr_out_color Output
%gl_out_color1 = OpVariable %ptr_out_color Output

; Input/output interface variables normally listed in OpEntryPoint:
; gl_in, worldPos0, worldPos1, color0, color1, gl_out, gl_out_color

; ---------------------------
; Helper constants
; ---------------------------

; small helper constants
%half = OpConstant %float 0.5
%zero  = OpConstant %float 0.0
%one   = OpConstant %float 1.0
; thicknesses
%thick_start = OpConstant %float 0.04
%thick_end   = OpConstant %float 0.005

; index literals
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

; ---------------------------
; Main function
; ---------------------------

%main = OpFunction %void None %func
%entry = OpLabel

; Local variables (float3 for p0, p1; direction and perpendicular)
%p0 = OpVariable %v3float Function
%p1 = OpVariable %v3float Function
%d  = OpVariable %v3float Function
%perp = OpVariable %v3float Function
%th0 = OpVariable %float Function
%th1 = OpVariable %float Function
%colA = OpVariable %v3float Function
%colB = OpVariable %v3float Function

; Load input positions
%in_p0 = OpLoad %v3float %worldPos0
%in_p1 = OpLoad %v3float %worldPos1
%in_c0 = OpLoad %v3float %color0
%in_c1 = OpLoad %v3float %color1

; Assign
OpStore %p0 %in_p0
OpStore %p1 %in_p1
OpStore %colA %in_c0
OpStore %colB %in_c1

; Compute line vector d = p1 - p0
%dvec = %d
%dvec = OpFSub %v3float %in_p1 %in_p0
OpStore %d %dvec

; Compute a perpendicular direction perp = normalize(vec3(-d.y, d.x, 0))
; Build -d.y and d.x
%neg_dy = OpFNegate %float (OpCompositeExtract %float %dvec 1)
; -d.y as vector
%perp_x = OpCompositeConstruct %v3float %float_(-OpFAsin) 0.0 %float_0
; Due to complexity, compute properly:
; perp = normalize(vec3(-d.y, d.x, 0))
%neg_y = OpFNegate %float (OpCompositeExtract %float %dvec 1)
%dx = OpCompositeExtract %float %dvec 0
%perp_raw = OpCompositeConstruct %v3float %neg_y %dx %float_0
%perp_norm = OpExtInst %v3float  %GLSL std  normalize %perp_raw
OpStore %perp %perp_norm

; Set thicknesses
%th0_f = OpCopyObject %float %thick_start
%th1_f = OpCopyObject %float %thick_end
OpStore %th0 %th0_f
OpStore %th1 %th1_f

; Emit four vertices for a 4-vertex strip:
; v0 = p0 + perp * thick_start
; v1 = p0 - perp * thick_start
; v2 = p1 + perp * thick_end
; v3 = p1 - perp * thick_end

; v0 = p0 + perp * th0
%temp0 = OpFMul %v3float %perp %th0
%v0 = OpFAdd %v3float %p0 %temp0
; v1 = p0 - perp * th0
%temp1 = OpFMul %v3float %perp %th0
%neg_perp = OpFNegate %v3float %temp1
%v1 = OpFAdd %v3float %p0 %neg_perp

; v2 = p1 + perp * th1
%temp2 = OpFMul %v3float %perp %th1
%v2 = OpFAdd %v3float %p1 %temp2
; v3 = p1 - perp * th1
%temp3 = OpFMul %v3float %perp %th1
%neg_temp3 = OpFNegate %v3float %temp3
%v3 = OpFAdd %v3float %p1 %neg_temp3

; Emit position for each vertex
%glv0 = OpCompositeConstruct %gl_PerVertex_out %v0
OpStore %gl_out0 %glv0
OpStore %gl_out_color0 %%colA ; start color
OpEmitVertex

%glv1 = OpCompositeConstruct %gl_PerVertex_out %v1
OpStore %gl_out1 %glv1
OpStore %gl_out_color0 %%colA
OpEmitVertex

%glv2 = OpCompositeConstruct %gl_PerVertex_out %v2
OpStore %gl_out2 %glv2
OpStore %gl_out_color1 %%colB
OpEmitVertex

%glv3 = OpCompositeConstruct %gl_PerVertex_out %v3
OpStore %gl_out3 %glv3
OpStore %gl_out_color1 %colB
OpEmitVertex

; End primitive
OpEndPrimitive

; End function
 OpReturn
 OpFunctionEnd

; End of module
