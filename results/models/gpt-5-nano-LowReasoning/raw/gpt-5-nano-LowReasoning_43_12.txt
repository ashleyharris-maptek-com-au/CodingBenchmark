cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normal in world space
    float3 N = normalize(input.normal);

    // Light direction in world space (point light at lightPos)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Diffuse term
    float d = max(dot(N, L), 0.0);

    // Hemisphere ambient (blend ground to sky by N.y * 0.5 + 0.5)
    float t = N.y * 0.5 + 0.5;
    float3 groundColor = float3(0.3, 0.15, 0.05);
    float3 skyColor    = float3(0.4, 0.6, 1.0);
    float3 ambient = lerp(groundColor, skyColor, t);

    // Base color from vertex color (assumed in [0,1])
    float3 baseColor = input.color;

    // Final color: ambient + small diffuse term
    float3 finalColor = baseColor * ambient + 0.3 * d;

    // Clamp to [0,1]
    finalColor = clamp(finalColor, 0.0, 1.0);

    return float4(finalColor, 1.0);
}
