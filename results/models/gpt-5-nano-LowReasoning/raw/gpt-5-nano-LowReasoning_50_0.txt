Here is a complete GLSL compute shader implementing one timestep of a 3D boid flocking simulation using Reynolds rules. It reads from inData and writes to outData. It performs per-boid computation on a single shader invocation and uses a simple O(N^2) neighborhood check (skip self and zero-distance neighbors). The radii and weights follow the task specification, and a boundary pull toward the center is applied when a position component exceeds boundSize*0.4.

```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

// Helper to clamp a vector's length to maxLen
float clampToMaxLength(vec3 v, float maxLen) {
    float l = length(v);
    if (l > maxLen && l > 0.0) {
        return (v / l) * maxLen;
    }
    return v;
}

void main() {
    // Read N
    uint N = counts.x;
    int gid = int(gl_GlobalInvocationID.x);
    if (gid < 0 || uint(gid) >= N) {
        // Out of bounds, do nothing
        return;
    }

    // Indices in inData/outData
    int idxPos  = gid * 2 + 0;
    int idxVel  = gid * 2 + 1;

    // Read current boid
    vec3 pos_i = inData[idxPos].xyz;
    vec3 vel_i = inData[idxVel].xyz;

    // Accumulators
    vec3 sepAcc = vec3(0.0);
    vec3 alignAcc = vec3(0.0);
    vec3 cohAcc = vec3(0.0);

    int neighborCount = 0;

    // Radii and weights (from uniform)
    float dt           = radii.x;
    float sepRadius    = radii.y;
    float alignRadius  = radii.z;
    float cohRadius    = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    // Iterate over all boids to accumulate neighbor contributions
    for (uint j = 0u; j < N; ++j) {
        if (j == uint(gid)) continue; // skip self

        int jPos = int(j) * 2 + 0;
        int jVel = int(j) * 2 + 1;

        vec3 pos_j = inData[jPos].xyz;
        vec3 vel_j = inData[jVel].xyz;

        vec3 delta = pos_j - pos_i;
        float dist = length(delta);

        // Skip zero-distance (shouldn't happen often) and neighbors outside max radius
        if (dist < 1e-6) continue;

        bool isNeighbor = false;
        vec3 sepT = vec3(0.0);
        vec3 alignT = vec3(0.0);
        vec3 cohT = vec3(0.0);

        // Separation
        if (dist < sepRadius) {
            // Stronger repulsion when closer
            sepT = (pos_i - pos_j) / max(dist, 1e-6);
            // Optionally scale by proximity; here we keep it as direction
            sepAcc += sepT;
            isNeighbor = true;
        }

        // Alignment
        if (dist < alignRadius) {
            alignT = vel_j;
            alignAcc += alignT;
            isNeighbor = true;
        }

        // Cohesion
        if (dist < cohRadius) {
            cohT = pos_j;
            cohAcc += cohT;
            isNeighbor = true;
        }

        if (isNeighbor) {
            neighborCount += 1;
        }
    }

    vec3 accel = vec3(0.0);

    if (neighborCount > 0) {
        // Average contributions
        sepAcc  = sepAcc / float(neighborCount);
        alignAcc = alignAcc / float(neighborCount);
        cohAcc  = cohAcc / float(neighborCount);

        // Convert to steering terms
        // Separation uses current accumulated direction away from neighbors
        accel += sepAcc * sepWeight;

        // Alignment: steer toward average velocity minus current velocity
        vec3 alignVec = alignAcc - vel_i;
        accel += alignVec * alignWeight;

        // Cohesion: steer toward average position of neighbors
        vec3 cohesionVec = cohAcc - pos_i;
        accel += cohesionVec * cohWeight;
    }

    // Boundary penalty / pull toward center
    // If any component exceeds boundSize*0.4, apply a center pull
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    vec3 centerPullDir = -pos_i;
    float centerLen = length(centerPullDir);

    if (centerLen > 0.0) {
        // Normalized direction toward center
        vec3 centerDir = centerPullDir / centerLen;
        // Apply a constant force toward center, scaled by boundSize
        accel += centerDir * boundForce * (0.5 * boundSize);
    }

    // Integrate velocity with acceleration (dt in radii.x)
    vel_i = vel_i + accel * dt;

    // Clamp speed
    float speed = length(vel_i);
    if (speed > maxSpeed && speed > 0.0) {
        vel_i = (vel_i / speed) * maxSpeed;
    }

    // Integrate position
    pos_i = pos_i + vel_i * dt;

    // Write results to output buffers
    outData[idxPos] = vec4(pos_i, 0.0);
    outData[idxVel] = vec4(vel_i, 0.0);
}
```
