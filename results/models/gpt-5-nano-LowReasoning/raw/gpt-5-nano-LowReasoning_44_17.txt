#version 450

layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal (interpolated, may need normalize)
layout(location = 2) in vec2 uv;         // texture coordinates in [0,1]
layout(location = 3) in vec3 tangent;    // tangent vector
layout(location = 4) in vec3 color;      // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;          // offset 0   (column-major)
    mat4 view;           // offset 64
    mat4 projection;     // offset 128
    vec4 lightPos;       // value: (5.0, 5.0, 5.0, 1.0)
    vec4 cameraPos;      // value: (0.0, 0.0, 3.0, 1.0)
    vec4 params;         // value: (0.0, 0.5, 0.0, 0.0)
} ubo;

layout(location = 0) out vec4 outColor;  // RGBA float, stored as rgba8unorm

void main() {
    // Brick pattern: 8x16 UV scale with staggered rows
    // Determine if current row is staggered (half-brick offset in X)
    float bricksX = 8.0;
    float bricksY = 16.0;

    // Compute row index for Y to decide stagger
    int row = int(floor(uv.y * bricksY));
    bool stagger = (row & 1) != 0;

    // Compute local X within a brick column, applying stagger by shifting U by half-brick on staggered rows
    float shiftedU = uv.x * bricksX;
    if (stagger) {
        shiftedU -= 0.5;
    }
    float localX = fract(shiftedU);
    float localY = fract(uv.y * bricksY);

    // Mortar thickness in UV space per brick
    // Mortar in X: 0.05 (in local brick space), Mortar in Y: 0.1 (in local brick space)
    float mortarX = 0.05;
    float mortarY = 0.1;

    // Determine if this fragment is in mortar or brick
    bool isMortar = (localX < mortarX) || (localX > (1.0 - mortarX)) || (localY < mortarY) || (localY > (1.0 - mortarY));

    // Colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(vec3(ubo.lightPos) - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    vec3 baseColor = isMortar ? mortarColor : brickColor;
    vec3 finalColor = baseColor * shade;

    outColor = vec4(finalColor, 1.0);
}
