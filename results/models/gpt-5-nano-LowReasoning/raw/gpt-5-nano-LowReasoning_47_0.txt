// WGSL compute shader: PoW with custom ChaCha-like hash
// Searches for a nonce such that the 64-bit hash has at least `difficulty` trailing zero bits.
// Nonce range is [range_start, range_start + range_size)

struct Params {
  // x: difficulty (trailing zero bits required)
  // y: range_start
  // z: range_size
  // w: unused (keeps 4th field for alignment)
}

@group(0) @binding(0) var<storage, read>      baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl32(x : u32, n : u32) -> u32 {
  // rotate left by n (0 <= n < 32)
  return (x << n) | (x >> (32u - n));
}

// The hash function is exactly as specified:
// state[0..15] = data[0..15]
// for i in 0..15: state[i] ^= (nonce + i)
// 8 rounds, each with 4 quarter-rounds (groups of 4)
fn custom_hash(data : array<u32, 16>, nonce : u32) -> vec2<u32> {
  var state : array<u32, 16>;
  // Initialize state
  for (var i : u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = data[i];
  }
  // XOR with nonce+i
  for (var i : u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = state[i] ^ (nonce + i);
  }

  // 8 rounds of ChaCha-like quarter-rounds, unrolled per group (0..3, 4..7, 8..11, 12..15)
  // Each group: (a,b,c,d) = (state[i], state[i+1], state[i+2], state[i+3])
  // Apply:
  // a += b; d ^= a; d = rotl(d,16)
  // c += d; b ^= c; b = rotl(b,12)
  // a += b; d ^= a; d = rotl(d,8)
  // c += d; b ^= c; b = rotl(b,7)
  // Then write back to state[i..i+3]

  // Helper to perform one group transform
  // Group 0
  var a : u32; var b : u32; var c : u32; var d : u32;

  // Round 1 - group 0
  {
    let i = 0u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 2 - group 1 (i = 4)
  {
    let i = 4u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 3 - group 2 (i = 8)
  {
    let i = 8u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 4 - group 3 (i = 12)
  {
    let i = 12u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 5 - group 0 again
  {
    let i = 0u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 6 - group 1
  {
    let i = 4u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 7 - group 2
  {
    let i = 8u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Round 8 - group 3
  {
    let i = 12u;
    a = state[i]; b = state[i+1u]; c = state[i+2u]; d = state[i+3u];
    a = a + b; d = d ^ a; d = rotl32(d, 16u);
    c = c + d; b = b ^ c; b = rotl32(b, 12u);
    a = a + b; d = d ^ a; d = rotl32(d, 8u);
    c = c + d; b = b ^ c; b = rotl32(b, 7u);
    state[i]   = a; state[i+1u] = b; state[i+2u] = c; state[i+3u] = d;
  }

  // Fold 16 words to 2 words
  var h0 : u32 = state[0] ^ state[2]  ^ state[4]  ^ state[6]  ^ state[8]  ^ state[10] ^ state[12] ^ state[14];
  var h1 : u32 = state[1] ^ state[3]  ^ state[5]  ^ state[7]  ^ state[9]  ^ state[11] ^ state[13] ^ state[15];

  return vec2<u32>(h0, h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  let diff      : u32 = params.x;
  let rangeStart: u32 = params.y;
  let rangeSize : u32 = params.z;

  // Guard against out-of-range threads
  let idx : u32 = gid.x;
  if (idx >= rangeSize) {
    return;
  }

  let nonce : u32 = rangeStart + idx;

  // Compute hash
  var hpair = custom_hash(baseData, nonce);
  var h0 : u32 = hpair.x;
  var h1 : u32 = hpair.y;

  // Count trailing zeros of 64-bit value (h0 | (h1<<32)), counting from LSB
  // If h0 != 0, tz = trailing zeros in h0
  // Else tz = 32 + trailing zeros in h1
  var tz : u32 = 0u;
  var found : bool = false;

  // Trailing zeros in h0
  if (h0 != 0u) {
    // find first set bit position in h0
    for (var i : u32 = 0u; i < 32u; i = i + 1u) {
      if ((h0 & (1u << i)) != 0u) {
        tz = i;
        found = true;
        break;
      }
    }
  } else {
    // h0 == 0, need to inspect h1
    tz = 32u; // baseline
    if (h1 != 0u) {
      for (var j : u32 = 0u; j < 32u; j = j + 1u) {
        if ((h1 & (1u << j)) != 0u) {
          tz = 32u + j;
          found = true;
          break;
        }
      }
    } else {
      // both zero
      tz = 64u;
      found = true;
    }
  }

  // Check if meets difficulty
  if (tz >= diff) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h0;
    result[3] = h1;
  }
}
