; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo_set DescriptorSet 0
OpDecorate %ubo_binding Binding 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec4 = OpTypePointer Input %vec4
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_float = OpTypePointer Uniform %float
%ptr_Input_float = OpTypePointer Input %float
%ptr_Input_void = OpTypePointer Input %void

; Globals/Variables
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec4 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform Uniform

; Constant / helpers
%ZERO = OpConstant %float 0.0
%ONE = OpConstant %float 1.0
%TWO = OpConstant %float 2.0
%HALF = OpConstant %float 0.5
%64 = OpConstant %float 64.0

; Entry point
%main = OpFunction %void None %func
%label = OpLabel

; Load uniforms
%u_ptr = OpAccessChain %ptr_Uniform %ubo
%model_ptr = OpPtrAccessChain %ptr_Uniform 0
%view_ptr = OpPtrAccessChain %ptr_Uniform 1
%proj_ptr = OpPtrAccessChain %ptr_Uniform 2
%light_ptr = OpPtrAccessChain %ptr_Uniform 3
%cam_ptr = OpPtrAccessChain %ptr_Uniform 4
%param_ptr = OpPtrAccessChain %ptr_Uniform 5

%model = OpLoad %mat4 %model_ptr
%view = OpLoad %mat4 %view_ptr
%proj = OpLoad %mat4 %proj_ptr
%lightPos = OpLoad %vec4 %light_ptr
%camPos = OpLoad %vec4 %cam_ptr
%params = OpLoad %vec4 %param_ptr

; Input values
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%LUT = OpLoad %vec3 %in_tangent
%dummy_uv = OpLoad %vec2 %in_uv
%col = OpLoad %vec4 %in_color

; Normalize normal
%N = OpExtInst %float 0; ; placeholder to avoid empty lines (not valid, but keeps structure)

; Compute vectors L, V in world space
%L = OpVector3Normalize %vec3
%V = OpVector3Normalize %vec3
H = OpVector3Normalize %vec3

; Diffuse and specular
%NdotL = OpExtInst %float 0 0 ; placeholder
%NdotH = OpExtInst %float 0 0 ; placeholder
%diffuse = OpFMax %float %NdotL %ZERO
%specular = OpPow %float %NdotH %64
%intensity = OpFAdd %float %HALF (OpFAdd %float (OpFMul %float %diffuse %float 0.6) (OpFMul %float %specular %float 0.8))

; Build vec4 result grayscale and alpha = 1
%col_out = OpCompositeConstruct %vec4 %intensity %intensity %intensity %ONE
OpStore %outColor %col_out

OpReturn
OpFunctionEnd
