const fs = require('fs');

(function main() {
  try {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const probGrid = data.probability;
    const drones = Array.isArray(data.drones) ? data.drones : [];
    const maxStepsPerDrone = typeof data.maxStepsPerDrone === 'number' ? data.maxStepsPerDrone : 0;

    const height = probGrid.length;
    const width = height > 0 ? probGrid[0].length : 0;

    // Helper to check inside grid
    function inBounds(x, y) {
      return x >= 0 && x < width && y >= 0 && y < height;
    }

    // Build paths for each drone with a greedy heuristic
    const paths = [];

    // Movement options: stay, left, right, up, down
    const moves = [
      { dx: 0, dy: 0 },
      { dx: -1, dy: 0 },
      { dx: 1, dy: 0 },
      { dx: 0, dy: -1 },
      { dx: 0, dy: 1 }
    ];

    for (let i = 0; i < drones.length; i++) {
      const d = drones[i];
      const startX = d.x;
      const startY = d.y;

      // If starting cell is outside or obstacle, clamp to valid non-obstacle cell if possible
      let sx = startX, sy = startY;
      if (!inBounds(sx, sy) || probGrid[sy]?.[sx] === -1) {
        // find nearest valid cell within grid (simple fallback)
        let found = false;
        for (let yy = 0; yy < height && !found; yy++) {
          for (let xx = 0; xx < width && !found; xx++) {
            if (probGrid[yy][xx] !== -1) {
              sx = xx; sy = yy; found = true;
            }
          }
        }
      }

      const visited = new Set();
      const path = [[sx, sy]];
      visited.add(`${sx},${sy}`);

      let cx = sx, cy = sy;

      for (let step = 0; step < maxStepsPerDrone; step++) {
        let bestPos = null;
        let bestProb = -Infinity;

        for (const m of moves) {
          const nx = cx + m.dx;
          const ny = cy + m.dy;

          if (!inBounds(nx, ny)) continue;
          if (probGrid[ny][nx] === -1) continue; // obstacle

          const p = probGrid[ny][nx];
          // Prefer higher probability, breaking ties by visiting new cells later
          if (p > bestProb) {
            bestProb = p;
            bestPos = { x: nx, y: ny };
          }
        }

        // If no valid move found (shouldn't happen if at least current cell is valid), break
        if (!bestPos) break;

        cx = bestPos.x;
        cy = bestPos.y;
        path.push([cx, cy]);
        visited.add(`${cx},${cy}`);
      }

      paths.push(path);
    }

    // Output result
    const output = { paths };
    process.stdout.write(JSON.stringify(output));
  } catch (err) {
    // In case of error, emit empty structure to stdout to satisfy format
    process.stdout.write(JSON.stringify({ paths: [] }));
  }
})();
