struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normal in world space (normalize to be safe)
    float3 N = normalize(input.normal);

    // Light and view directions in world space
    float3 L = normalize((float3)lightPos - input.worldPos);
    float3 V = normalize((float3)cameraPos - input.worldPos);

    // Reflection of L about N
    float3 R = reflect(-L, N);

    // Phong components
    float diffuse = max(dot(N, L), 0.0);
    float specular = pow(max(dot(R, V), 0.0), 32.0);

    // Final intensity (grayscale)
    float intensity = 0.2 + 0.5 * diffuse + 0.8 * specular;

    float3 color = float3(intensity, intensity, intensity);
    return float4(color, 1.0);
}
