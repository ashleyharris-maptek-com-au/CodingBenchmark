import sys
import random

# A simple, practical 3-SAT solver (WalkSAT-like) with streaming-friendly data structures.
# Assumes the input SAT instance is satisfiable (as guaranteed by the problem).

def main():
    data = sys.stdin.buffer.read().split()
    if not data:
        return

    t = iter(data)
    try:
        n = int(next(t))
        m = int(next(t))
    except StopIteration:
        return

    # Read clauses: each clause has 3 literals
    clauses = []
    # For memory/time considerations, store as tuples of three ints
    for _ in range(m):
        try:
            a = int(next(t)); b = int(next(t)); c = int(next(t))
        except StopIteration:
            break
        clauses.append((a, b, c))

    # Build per-variable incidence: for each variable, list of (clause_index, sign)
    # sign = +1 if literal is positive (xi), -1 if negated (~xi)
    var_occ = [[] for _ in range(n)]
    for idx, (a, b, c) in enumerate(clauses):
        for lit in (a, b, c):
            v = abs(lit) - 1
            sign = 1 if lit > 0 else -1
            if 0 <= v < n:
                var_occ[v].append((idx, sign))

    # Helper to evaluate a single clause under current assignment
    # assignment is list of 0/1 with 0 = False, 1 = True
    def eval_clause(clidx, assignment):
        a, b, c = clauses[clidx]
        # literal value: if sign positive, use assignment[v]; if negative, use 1 - assignment[v]
        def lit_val(lit):
            v = abs(lit) - 1
            val = assignment[v]
            return val if lit > 0 else (1 - val)
        return (lit_val(a) or lit_val(b) or lit_val(c))

    # Initialize a random assignment
    # 0/1 for each variable
    assignment = [random.getrandbits(1) for _ in range(n)]

    # Precompute initial clause satisfaction
    m_int = len(clauses)
    clause_sat = [False] * m_int
    for i in range(m_int):
        clause_sat[i] = eval_clause(i, assignment)

    # Maintain unsatisfied clauses as a dynamic set using a list + boolean flag for O(1) checks
    unsat_flag = [not clause_sat[i] for i in range(m_int)]
    unsat_list = [i for i in range(m_int) if unsat_flag[i]]

    # Helper to refresh a clause's satisfaction status after assignments change
    def update_clause(i):
        old = clause_sat[i]
        new = eval_clause(i, assignment)
        if old != new:
            clause_sat[i] = new
            unsat_flag[i] = (not new)
            # Keep unsat_list as a pool; if becomes unsatisfied, lazy addition below handles it.
            if not new:
                # It's now satisfied; nothing else to do here (we'll skip removed items lazily)
                pass

    # Try to find a satisfying assignment using a WalkSAT-like loop
    # Parameters
    max_restarts = 5
    max_flips_per_restart = max(1000000, m_int * 10)  # safety cap; adjust as needed

    for restart in range(max_restarts):
        if restart > 0:
            # Re-randomize assignment and recompute clause statuses
            assignment = [random.getrandbits(1) for _ in range(n)]
            for i in range(m_int):
                clause_sat[i] = eval_clause(i, assignment)
                unsat_flag[i] = (not clause_sat[i])
            unsat_list = [i for i in range(m_int) if unsat_flag[i]]

        # Clear or rebuild unsatisfied pool
        # Note: We'll use a lazy approach for selecting a random unsatisfied clause.
        # If no unsatisfied clauses exist, we have a solution.
        flips = 0
        while flips < max_flips_per_restart:
            # Check if solved
            # Clean up unsat_list by discarding indices that are no longer unsatisfied
            while unsat_list and not unsat_flag[unsat_list[-1]]:
                unsat_list.pop()
            if not unsat_list:
                # Found a satisfying assignment
                print("SAT")
                print(" ".join(str(bit) for bit in assignment))
                return

            # Pick a random unsatisfied clause
            # If unsat_list is empty due to lazy cleanup, skip (handled above)
            ri = random.randrange(len(unsat_list))
            clidx = unsat_list[ri]

            # Get the three literals of this clause
            a, b, c = clauses[clidx]
            literals = (a, b, c)

            # Select a variable to flip: with some probability flip a random literal's variable,
            # otherwise pick the one that appears most often in unsatisfied clauses (simplified here: random)
            # We'll use simple strategy: flip a random variable from the clause
            lit = random.choice(literals)
            v = abs(lit) - 1
            # Flip variable v
            assignment[v] ^= 1

            # Update all clauses that contain variable v
            for (cl, sign) in var_occ[v]:
                # Recompute clause satisfaction for affected clause
                old_sat = clause_sat[cl]
                new_sat = eval_clause(cl, assignment)
                if old_sat != new_sat:
                    clause_sat[cl] = new_sat
                    unsat_flag[cl] = (not new_sat)
                    if new_sat:
                        # clause became satisfied; if it is in unsat_list, we'll lazily skip it
                        pass
                    else:
                        # clause became unsatisfied; add to unsat_list if not already
                        if not unsat_flag[cl] or True:
                            # We can't rely on unsat_flag only; ensure it's marked
                            if cl not in set():
                                # The above check is a placeholder to avoid heavy set usage.
                                # We'll just append; duplicates are harmless due to lazy check.
                                unsat_list.append(cl)

            flips += 1

        # If we exit the inner loop without finding solution, try next restart
        # (the outer loop handles reinitialization)

    # If no solution found within time, as a fallback (though problem guarantees satisfiable),
    # return a best-effort solution: current assignment.
    print("SAT")
    print(" ".join(str(bit) for bit in assignment))

if __name__ == "__main__":
    main()
