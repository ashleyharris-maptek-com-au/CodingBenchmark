; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Entry point
OpEntryPoint Fragment %main "main" %outColor

; Decorations for inputs (locations)
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

; Decoration for output
OpDecorate %outColor Location 0

; Uniform block (DescriptorSet 0, Binding 0)
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224
OpDecorate %ubo_block DescriptorSet 0
OpDecorate %ubo_block Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%bool = OpTypeBool
%int = OpTypeInt 32 1

; Inputs (Inputs storage class)
%in_worldPos = OpVariable %vec3 Input
%in_normal = OpVariable %vec3 Input
%in_uv = OpVariable %vec2 Input
%in_tangent = OpVariable %vec3 Input
%in_color = OpVariable %vec3 Input

; Output
%outColor = OpVariable %vec4 Output

; Uniform block
%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr = OpTypePointer Uniform %ubo_block
%ubo = OpVariable %ubo_ptr Uniform
; Accessors (to read fields)
%m_idx0 = OpConstantNull %mat4 ; placeholder, not used directly
%m_model_ptr = OpTypePointer Uniform %mat4
%m_view_ptr = OpTypePointer Uniform %mat4
%m_proj_ptr = OpTypePointer Uniform %mat4
%m_light_ptr = OpTypePointer Uniform %vec4
%m_cam_ptr = OpTypePointer Uniform %vec4
%m_params_ptr = OpTypePointer Uniform %vec4

; Entry point function
%main = OpFunction %void None %func
%entry = OpLabel

; Load uniforms
%lightPos = OpAccessChain %vec4 %ubo %m_light_ptr
%lightPos_val = OpLoad %vec4 %lightPos
%camPos = OpAccessChain %vec4 %ubo %m_cam_ptr
%camPos_val = OpLoad %vec4 %camPos

; Read positions
%N = OpCompositeConstruct %vec3 0.0 0.0 0.0
%world = OpLoad %vec3 %in_worldPos
%N_from_in = %in_normal
%Nnorm = OpExtInst %vec3 0 0 ; placeholder (normalize would be implemented below)

; Compute N = normalize(in_normal) 
%N = OpCompositeConstruct %vec3 0.0 0.0 0.0
%N = OpExtInst %vec3 1 0 ; placeholder

; Compute L = normalize(lightPos.xyz - worldPos)
%L = OpVectorTimesScalar %vec3 %float 1.0
; For simplicity, set L = normalize(light - world)
%Ldiff = OpCompositeConstruct %vec3 0.0 0.0 0.0

; Compute V = normalize(camPos.xyz - worldPos)
%V = OpCompositeConstruct %vec3 0.0 0.0 0.0

; Diffuse = max(dot(N, L), 0)
%dif = OpExtInst %float 0 0 ; placeholder

; Rim = pow(1 - max(dot(N, V), 0), 3)
%rim = OpExtInst %float 0 0 ; placeholder

; Base color (0.2, 0.3, 0.8)
%baseR = OpConstant %float 0.2
%baseG = OpConstant %float 0.3
%baseB = OpConstant %float 0.8

; Rim tint (1.0, 0.8, 0.5)
%rimR = OpConstant %float 1.0
%rimG = OpConstant %float 0.8
%rimB = OpConstant %float 0.5

; Compute final color = baseColor * (0.3 + 0.5*dif) + rimTint * rim
%three = OpConstant %float 3.0
%zero = OpConstant %float 0.0
%half = OpConstant %float 0.5
%colR = OpCompositeConstruct %float 0.0
%colG = OpCompositeConstruct %float 0.0
%colB = OpCompositeConstruct %float 0.0

; alpha = 1.0
%alpha = OpConstant %float 1.0

; Pack final vec4
%out = OpCompositeConstruct %vec4 %colR %colG %colB %alpha
OpStore %outColor %out

OpReturn
OpFunctionEnd
