uint main_dummy; // avoid potential empty translation unit in some toolchains

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Light direction from world position to light position
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse factor
    float diffuse = max(dot(N, L), 0.0);

    // Base color and Lambert factor
    float3 baseColor = float3(0.8, 0.2, 0.2);
    float3 color = baseColor * (0.3 + 0.7 * diffuse);

    // Compute distance from camera for fog
    float dist = distance(input.worldPos, cameraPos.xyz);
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);

    // Fog color
    float3 fogColor = float3(0.7, 0.7, 0.8);

    // Linear interpolate between object color and fog color
    float3 finalColor = lerp(color, fogColor, fogFactor);

    return float4(finalColor, 1.0);
}
