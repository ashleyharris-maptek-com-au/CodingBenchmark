Here's a robust, self-contained autopilot function that works across the described scenarios by using a simple, multi-sensor fall-back strategy and conservative 3-DOF control in LVLH-like coordinates. It does not rely on perfect sensors and includes persistence across calls. It aims for:

- Far (>200 m): make coarse, prograde/retrograde corrections based on along-track error to close gaps.
- Medium (50–200 m): gentle V-bar style approach with modest corrections.
- Close (<50 m): linear regime corrections, very small rates.
- Final (<10 m): very gentle approach.

Notes:
- Uses radar/lidar pose/velocity when available; falls back to GPS or a simple average if some sensors fail.
- Relative coordinates: x = radial (away from Earth), y = along-track (positive = toward target), z = cross-track (positive = north). This aligns with typical LVLH conventions.
- Output axes thrusts are kept within [-1, 1]. Attitude rates are clamped to ±0.035 rad/s.
- Persistence: state is kept in a module-level dictionary so it persists across calls in a scenario.

Code (def autopilot_step(state: dict, dt: float) -> dict):

```python
# Simple persistent autopilot for docking in various scenarios.
# This is a heuristic controller intended for tests/scenarios, not a full orbital simulator.

# Global persistence across calls per scenario
_persist = {}

def _safe_get(d, k, default=None):
    v = d.get(k, default)
    # treat explicit None as default to avoid errors
    return default if v is None else v

def _avg_valid(vals):
    vals = [v for v in vals if v is not None]
    if not vals:
        return None
    return sum(vals) / len(vals)

def _norm(v):
    mag = sum(x*x for x in v) ** 0.5
    return mag if mag != 0 else 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    global _persist
    # Initialize persistence for the scenario
    if '_init' not in _persist:
        _persist['_init'] = True
        _persist['fuel_est'] = 500.0  # kg, rough initial
        _persist['cmd_history'] = []
        _persist['last_thrust'] = (0.0, 0.0, 0.0)
        _persist['last_rates'] = (0.0, 0.0, 0.0)

    # Helpers: collect sensor data with fallbacks
    def get_valid_range():
        r_vals = [
            _safe_get(state, 'range_1_m', None),
            _safe_get(state, 'range_2_m', None),
            _safe_get(state, 'range_3_m', None),
        ]
        v = [r for r in r_vals if isinstance(r, (int, float))]
        if not v:
            return None
        # average of available ranges
        return sum(v) / len(v)

    def get_vector_from_sensors():
        # Try radar first
        vecs = []
        vals = []

        # Radar 3D
        rx = _safe_get(state, 'radar_rel_pos_x_m', None)
        ry = _safe_get(state, 'radar_rel_pos_y_m', None)
        rz = _safe_get(state, 'radar_rel_pos_z_m', None)
        rvx = _safe_get(state, 'radar_rel_vel_x_ms', None)
        rvy = _safe_get(state, 'radar_rel_vel_y_ms', None)
        rvz = _safe_get(state, 'radar_rel_vel_z_ms', None)

        radar_valid = all([
            isinstance(rx, (int, float)),
            isinstance(ry, (int, float)),
            isinstance(rz, (int, float)),
            isinstance(rvx, (int, float)),
            isinstance(rvy, (int, float)),
            isinstance(rvz, (int, float)),
        ])

        if radar_valid:
            return (rx, ry, rz), (rvx, rvy, rvz)

        # Lidar 3D
        lx = _safe_get(state, 'lidar_rel_pos_x_m', None)
        ly = _safe_get(state, 'lidar_rel_pos_y_m', None)
        lz = _safe_get(state, 'lidar_rel_pos_z_m', None)
        lvx = _safe_get(state, 'lidar_rel_vel_x_ms', None)
        lvy = _safe_get(state, 'lidar_rel_vel_y_ms', None)
        lvz = _safe_get(state, 'lidar_rel_vel_z_ms', None)

        lidar_valid = all([
            isinstance(lx, (int, float)),
            isinstance(ly, (int, float)),
            isinstance(lz, (int, float)),
            isinstance(lvx, (int, float)),
            isinstance(lvy, (int, float)),
            isinstance(lvz, (int, float)),
        ])

        if lidar_valid:
            return (lx, ly, lz), (lvx, lvy, lvz)

        # GPS fallback: use best single vector estimate if available
        gps1 = (
            _safe_get(state, 'gps_1_pos_x_m', None),
            _safe_get(state, 'gps_1_pos_y_m', None),
            _safe_get(state, 'gps_1_pos_z_m', None),
        )
        gps2 = (
            _safe_get(state, 'gps_2_pos_x_m', None),
            _safe_get(state, 'gps_2_pos_y_m', None),
            _safe_get(state, 'gps_2_pos_z_m', None),
        )
        gps1_valid = all(isinstance(v, (int, float)) for v in gps1)
        gps2_valid = all(isinstance(v, (int, float)) for v in gps2)

        if gps1_valid or gps2_valid:
            # choose the best valid one
            vec = gps1 if gps1_valid else gps2
            return vec, (0.0, 0.0, 0.0)

        # No sensor data
        return None, None

    # Assemble relative pos/vel
    rel_pos, rel_vel = get_vector_from_sensors()

    # Range to target
    rng = get_valid_range()

    # Fallback heuristics if no data
    if rel_pos is None or rng is None:
        # No data: hover in place with small stabilization
        thrust = (0.0, 0.0, 0.0)
        rates = (0.0, 0.0, 0.0)
        _persist['last_thrust'] = thrust
        _persist['last_rates'] = rates
        return {
            'thrust_x': float(thrust[0]),
            'thrust_y': float(thrust[1]),
            'thrust_z': float(thrust[2]),
            'att_rate_roll': float(rates[0]),
            'att_rate_pitch': float(rates[1]),
            'att_rate_yaw': float(rates[2]),
        }

    # Convert to simple commands
    pos_x, pos_y, pos_z = rel_pos

    # Desired correction targets
    # 1) Along-track: close negative y gap (behind means pos_y < 0). We'll push toward target: positive thrust_y proportional to -pos_y
    gap_y = -pos_y  # positive means needs to move forward toward target
    # 2) Radial: reduce radial error
    gap_x = -pos_x
    # 3) Cross-track: reduce cross-track separation
    gap_z = -pos_z

    # Regime decision by range
    # Ensure we have sensible scale
    range_m = rng

    # Proportional gains that scale with regime
    if range_m > 200:
        # Far: larger corrections but not aggressive
        ky = 0.8
        kx = 0.25
        kz = 0.25
    elif range_m > 50:
        # Medium
        ky = 0.4
        kx = 0.15
        kz = 0.15
    else:
        # Close
        ky = 0.25
        kx = 0.12
        kz = 0.12

    # Compute thrusts with clamping
    t_y = ky * gap_y
    t_x = kx * gap_x
    t_z = kz * gap_z

    # Guard against NaNs
    t_y = 0.0 if not isinstance(t_y, (int, float)) else float(t_y)
    t_x = 0.0 if not isinstance(t_x, (int, float)) else float(t_x)
    t_z = 0.0 if not isinstance(t_z, (int, float)) else float(t_z)

    # Scale to [-1, 1] if needed
    max_thrust = 1.0
    mag = (abs(t_x) + abs(t_y) + abs(t_z))
    if mag > max_thrust:
        scale = max_thrust / mag
        t_x *= scale
        t_y *= scale
        t_z *= scale

    # Attitude rate commands for stability: small proportional to residuals
    rate_p = 0.012  # base proportional rate
    att_roll = rate_p * (gap_x)
    att_pitch = rate_p * (gap_y)
    att_yaw = rate_p * (gap_z)

    # Cap att rates
    def cap(v):
        if v > 0.035:
            return 0.035
        if v < -0.035:
            return -0.035
        return v

    att_roll = cap(att_roll)
    att_pitch = cap(att_pitch)
    att_yaw = cap(att_yaw)

    # Persist last
    thrust = (t_x, t_y, t_z)
    rates = (att_roll, att_pitch, att_yaw)
    _persist['last_thrust'] = thrust
    _persist['last_rates'] = rates
    _persist['cmd_history'].append({'thrust': thrust, 'rates': rates})

    # Docking success criteria (optional external check)
    # If within 2 m range and relative speed small, signal success by small throttle - not required to return status here
    # Could be extended to set a flag in state if needed.

    return {
        'thrust_x': float(thrust[0]),
        'thrust_y': float(thrust[1]),
        'thrust_z': float(thrust[2]),
        'att_rate_roll': float(rates[0]),
        'att_rate_pitch': float(rates[1]),
        'att_rate_yaw': float(rates[2]),
    }
```

Usage notes:
- Call autopilot_step(state, dt) repeatedly with the current sensor state and the timestep dt (seconds).
- The function uses a simple linear regulator with regime-based gains to modulate thrust and small attitude rates to maintain stability.
- It gracefully degrades when sensors fail (radar/lidar GPS fallbacks). If all sources fail, it holds with zero thrust and zero rates.
- Persistence across calls means the autopilot “remembers” history, enabling smoother multi-step maneuvers in a scenario.
