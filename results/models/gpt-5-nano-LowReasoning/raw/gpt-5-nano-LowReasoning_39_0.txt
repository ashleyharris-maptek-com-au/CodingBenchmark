```rust
use std::io::{self, Read};

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // n vertices and m edges
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => 0,
    };

    // n must be even per problem statement
    // Build adjacency as u128 bitmasks (works for n <= 128)
    let mut adj: Vec<u128> = vec![0; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u < n && v < n {
            adj[u] |= 1u128 << v;
            adj[v] |= 1u128 << u;
        }
    }

    let k = n / 2; // size of partition A
    // We will force vertex 0 to be in A to cut the search space roughly in half.
    // So we choose k-1 additional vertices among {1, 2, ..., n-1}
    // Represent the subset of {1..n-1} as a bitmask over w = n-1 bits.
    if k == 0 {
        // Degenerate (n=0) not expected, but handle gracefully
        println!("0\n\n");
        return;
    }

    let w = if n > 0 { n - 1 } else { 0 };
    if w == 0 {
        // n == 1 which shouldn't happen since n is even; handle gracefully
        println!("0\n0\n");
        return;
    }

    // Helper: mask with w bits
    let mask_w: u128 = if w >= 128 {
        // safety for extremely large w; but n <= 128 in our approach
        u128::MAX
    } else {
        (1u128 << w) - 1
    };

    // If k-1 == 0, there is only one combination: sub = 0
    // Start with the smallest combination: the lowest k-1 bits among positions [1..n-1]
    let mut solutions_searched = 0u64;
    let mut best_cut: Option<u128> = None;
    let mut best_S: u128 = 0;

    // Function to compute cut size for a given full mask S (over n bits)
    // S must have bit 0 set and exactly k bits set in total
    let compute_cut = |s: u128, adj: &Vec<u128>| -> u128 {
        // cut = sum_{v in S} popcount(adj[v] & ~S)
        let mut t = s;
        let mut cut: u128 = 0;
        while t != 0 {
            // extract least significant set bit
            let lsb = t & (!t + 1);
            let v = lsb.trailing_zeros() as usize;
            let neigh_in_other_side = adj[v] & !s;
            cut += neigh_in_other_side.count_ones() as u128;
            t -= lsb;
        }
        cut
    };

    // If k-1 > w, impossible, but that would mean impossible to place; handle gracefully.
    if k - 1 > w {
        // No valid partition with vertex 0 forced in A
        // As a fallback, try the simplest valid partition: take first k vertices as A (ignore the 0-forcing).
        // But to keep correctness, just output a trivial valid partition (not optimal). However problem expects optimal.
        // We'll attempt to generate without forcing vertex 0 if needed.
    }

    // Start with initial submask over bits [0..w-1] corresponding to vertices [1..n-1].
    // We want exactly k-1 bits set in this submask.
    // Use Gosper's hack for fixed width w.
    // Initial submask with (k-1) ones: place ones in the lowest (k-1) positions, then shift by 1
    // into positions [1..n-1].
    if k - 1 == 0 {
        // Only one candidate: S = 1 (vertex 0) and no others in A
        let S: u128 = 1u128;
        let cut = compute_cut(S, &adj);
        best_cut = Some(cut);
        best_S = S;
        solutions_searched = 1;
    } else {
        // initial submask over w bits: ones in lowest (k-1) positions, then mapped to positions [1..]
        let mut sub: u128 = ((1u128 << (k - 1)) - 1) << 1; // bits 1.. up
        // Ensure we only keep within w bits (though shifted 1 already keeps in range)
        sub &= mask_w;

        // Iterate all combinations of size k-1 among w bits
        loop {
            // Build full S: include vertex 0, plus vertices corresponding to set bits in sub
            let S: u128 = sub | 1u128; // ensure bit 0 is set
            // Optional quick skip: verify population count is k
            // (sub has popcount k-1 by construction)
            let cut = compute_cut(S, &adj);
            match best_cut {
                None => {
                    best_cut = Some(cut);
                    best_S = S;
                }
                Some(cur) => {
                    if cut < cur {
                        best_cut = Some(cut);
                        best_S = S;
                    }
                }
            }
            solutions_searched += 1;

            // Next combination within width w
            // Gosper's hack for fixed width w on x
            // x must have at least one 1
            let x = sub;
            // if x has no 1s or is the last combination, break
            // compute next
            let c = x & (!x + 1);
            if c == 0 {
                break;
            }
            let r = x + c;
            let mut next = (((r ^ x) >> 2) / c) | r;
            // Mask to width w
            next &= mask_w;
            if next <= x {
                // wrapped around, finished
                break;
            }
            sub = next;
            // If we ran out (i.e., all combinations done), loop will exit when next can't be formed
            // However, we need to detect when we've exhausted: if sub after update has all ones in top positions?
            // The width check above suffices: when next becomes zero after mask or unchanged, break
            if sub == 0 {
                break;
            }
        }
    }

    // If we didn't find any solution (shouldn't happen), fallback
    let (min_cut, a_mask) = match best_cut {
        Some(c) => (c, best_S),
        None => {
            // Fallback: first k vertices in A
            let mut S: u128 = 0;
            for i in 0..k {
                S |= 1u128 << i;
            }
            (compute_cut(S, &adj), S)
        }
    };

    // Output
    // First line: cut size
    // Second line: vertices in partition A (indices in ascending order)
    println!("{}", min_cut);

    // Collect vertices in A from a_mask
    let mut first = true;
    let mut out = Vec::new();
    for i in 0..n {
        if (a_mask >> i) & 1 == 1 {
            out.push(i);
        }
    }
    // Ensure exactly n/2 vertices
    // If not, it's an error state; but we attempt to print whatever we have
    for (idx, v) in out.iter().enumerate() {
        if idx > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
```
