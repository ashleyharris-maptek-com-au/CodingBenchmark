%fragInWorldPos = variable %_ptr_Input_vec3 Input
%fragInNormal = variable %_ptr_Input_vec3 Input
%fragInUV = variable %_ptr_Input_vec2 Input
%fragOutColor = variable %_ptr_Output_vec4 Output
%UBO_ptr = variable %_ptr_Uniform_UBO Uniform

%main = function %void None %0
%0 = label
; load inputs
%worldPos = load %vec3Ptr %fragInWorldPos
%normal = load %vec3Ptr %fragInNormal
%uv = load %vec2Ptr %fragInUV

; load UBO
%ub = load %UBO_ptr
%modelMat = accessChain %mat4 ub 0
%viewMat = accessChain %mat4 ub 0 1
%projMat = accessChain %mat4 ub 0 2
%lightPos = accessChain %vec4 ub 0 3
%camPos = accessChain %vec4 ub 0 4
%params = accessChain %vec4 ub 0 5

; compute normal normalize
%n = normalize %vec3 %normal
; compute light dir
%lightWPos = accessChain %vec4 ub 0 3
%lightXYZ = swizzle %vec3 %lightWPos 0 1 2
%lightDir = normalize %vec3
; lightDir = lightPos.xyz - worldPos
%ldx = fsub %float %lightXYZ.x %worldPos.x
%ldy = fsub %float %lightXYZ.y %worldPos.y
%ldz = fsub %float %lightXYZ.z %worldPos.z
%lightDirXYZ = ( %vec3 %ldx %ldy %ldz )
%ld = normalize %vec3 %lightDirXYZ

; dot product
%d = dot %float %n %ld
%diff = max %float %d %float_0
%diffMul = fmul %float %diff %float_0
%diffClamp = max %float %diff %float_0
%diffLambert = fmul %float %diffClamp %float 1.0
; shading = 0.4 + 0.6*diff
%sh1 = fmul %float %diffClamp %float_0
%sh2 = fadd %float 0.4 %fmul
%sh = fadd %float 0.4 % ( fmul %float 0.6 % %diffClamp )

; stripes
%pi = float 3.1415927
%scale = fmul %float %uv.y %float_pi
%scale2 = fmul %float %scale %float_16
%sinv = call %float %sin %scale2
%cond = fgt %bool %sinv %float_0
; if sinv>0 red else blue
%red = vec3 0.9 0.1 0.1
%blue = vec3 0.1 0.1 0.9
;color = select(colorBlue, colorRed, cond)
%col = select %vec3 %blue %red %cond

; final color = color * shading
%colR = fmul %vec3 %col %vec3_1.1
; alpha = 1.0
%alpha = float 1.0
; final color rgba
%out = fadd %float %colR ???

; end
