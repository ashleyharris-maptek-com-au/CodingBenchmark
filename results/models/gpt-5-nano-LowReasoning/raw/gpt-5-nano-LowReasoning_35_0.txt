use std::cmp::Ordering;
use std::io::{self, Read};

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Read number of vertices and edges
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => 0,
    };

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let a: usize = it.next().unwrap().parse().unwrap();
        let b: usize = it.next().unwrap().parse().unwrap();
        if a >= n || b >= n { continue; }
        adj[a].push(b);
        adj[b].push(a);
    }

    // Remove potential duplicates in adjacency for efficiency
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    // Helper: given a vertex v, the set it dominates (including itself)
    let mut dominates = vec![Vec::<usize>::new(); n];
    for v in 0..n {
        dominates[v].push(v);
        for &to in &adj[v] {
            dominates[v].push(to);
        }
        // Ensure unique just in case
        dominates[v].sort_unstable();
        dominates[v].dedup();
    }

    // Track dominated vertices
    let mut dominated = vec![false; n];
    let mut dominating_set: Vec<usize> = Vec::new();

    // Greedy: repeatedly pick vertex that dominates the most yet-undominated vertices
    let mut remaining = n;
    while remaining > 0 {
        // For each vertex not yet in the dominating set, compute how many currently undominated vertices it would cover
        let mut best_v = None::<usize>;
        let mut best_gain = 0usize;

        for v in 0..n {
            // If already in the dominating set, skip
            if dominating_set.contains(&v) {
                continue;
            }
            // Count how many vertices in dominates[v] are not yet dominated
            let gain = dominates[v].iter().filter(|&&u| !dominated[u]).count();

            // Prefer larger gain; early exit if max possible attainable
            match gain.cmp(&best_gain) {
                Ordering::Greater => {
                    best_gain = gain;
                    best_v = Some(v);
                }
                _ => {}
            }
        }

        // If nothing gains any new coverage (shouldn't happen unless graph empty), break
        let v = match best_v {
            Some(x) => x,
            None => break,
        };

        // Add to dominating set
        dominating_set.push(v);
        // Mark dominated vertices
        for &u in &dominates[v] {
            if !dominated[u] {
                dominated[u] = true;
                remaining = remaining.saturating_sub(1);
            }
        }
    }

    // Post-process: try to remove any vertex that is not essential
    // A vertex is essential if removing it would leave some vertex undominated.
    // We'll attempt a simple prune: try removing in reverse order.
    // Recompute dominated after removal to test viability.
    if !dominating_set.is_empty() {
        // Create a set for quick membership test
        let mut in_set = vec![false; n];
        for &v in &dominating_set {
            in_set[v] = true;
        }

        // Compute initial domination status with current set
        let mut dominated2 = vec![false; n];
        for &v in &dominating_set {
            for &u in &dominates[v] {
                dominated2[u] = true;
            }
        }

        // If there are any undominated vertices, this would be a logic error, but guard anyway
        let mut i = dominating_set.len();
        while i > 0 {
            i -= 1;
            let v = dominating_set[i];
            // Try removing v and see if all vertices remain dominated
            // Simulate removal: mark v as not in set temporarily
            let mut still_dom = dominated2.clone();

            // Remove effects of v
            for &u in &dominates[v] {
                // Recompute since multiple dominators may cover u
                // A simple way: check if u is dominated by any other vertex in the current set (excluding v)
                still_dom[u] = false;
            }

            // Re-derive domination without v
            for &w in &dominating_set {
                if w == v { continue; }
                for &u in &dominates[w] {
                    still_dom[u] = true;
                }
            }

            // Check if all vertices are dominated
            let all_dom = still_dom.iter().all(|&b| b);
            if all_dom {
                // removal is safe
                in_set[v] = false;
                dominating_set.remove(i);
                dominated2 = still_dom;
            } // else keep v
        }
    }

    // Final output
    println!("{}", dominating_set.len());
    if !dominating_set.is_empty() {
        // Print space-separated vertex indices
        for (idx, v) in dominating_set.iter().enumerate() {
            if idx > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
