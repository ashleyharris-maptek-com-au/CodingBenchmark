#include <bits/stdc++.h>
using namespace std;

/*
  Exact Maximum Independent Set by reduction to Maximum Clique
  on the complement graph, solved with a Tomita-style branch-and-bound
  using greedy coloring for bounds.

  - Read n, m
  - Build original adjacency
  - Build complement graph adjacency (excluding self-loops)
  - Run maximum clique algorithm to obtain MIS in original graph
  - Output MIS (as vertex indices)
*/

static int n, m;

// adjacency of original graph
static vector<vector<char>> adjOrig;

// adjacency of complement graph (undirected)
static vector<vector<char>> adjComp;

// best clique
static vector<int> bestClique;

// current clique and candidate set
static vector<int> curClique;
static vector<int> cand;

// coloring helper
static vector<int> colorBound; // color numbers assigned to each candidate in order

// greedy coloring to compute an upper bound on clique size from the current candidate set
static void colorGreedy(const vector<int>& P, vector<int>& colors, vector<int>& order) {
    // We'll assign colors in a simple greedy manner:
    // process vertices in reverse order of P to help pruning (not strictly required)
    int k = (int)P.size();
    colors.assign(k, 0);
    // For each vertex, determine the smallest color not used by its neighbors among already colored vertices
    // We only need an upper bound, so a simple greedy is fine.
    // Maintain for each color, the set of vertices indices that have that color.
    // But a simpler approach: for each vertex i, mark used colors by neighbors among 0..i-1
    vector<vector<char>> usedColors; // not preallocated; we implement by local array since colors <= k
    // We'll implement via a small fixed-size boolean array per vertex
    for (int i = 0; i < k; ++i) {
        // mark colors used by neighbors among previous vertices
        // determine neighbors of vertex P[i] among P[0..i-1]
        // we can check pairwise using adjComp
        // We'll maintain a boolean array sized <= k
        // dynamic allocation per iteration is okay for moderate k
        vector<char> forbidden(k+1, 0);
        for (int j = 0; j < i; ++j) {
            int a = P[i];
            int b = P[j];
            // if a and b are adjacent in complement graph, they cannot share same color? Actually coloring is on the induced subgraph of P.
            // We color with respect to adjacency among P. If adjComp[a][b] is true, then they are adjacent.
            if (adjComp[a][b]) {
                forbidden[ colors[j] ] = 1;
            }
        }
        int c = 1;
        while (forbidden[c]) ++c;
        colors[i] = c;
    }
    // Prepare order as P indices sorted by color ascending (for Tomita style, we will traverse in reverse later)
    order.resize(k);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int i, int j){
        if (colors[i] != colors[j]) return colors[i] > colors[j]; // higher color first for pruning
        return P[i] < P[j];
    });
}

static void expand(vector<int>& P) {
    if (P.empty()) {
        if ((int)curClique.size() > (int)bestClique.size()) {
            bestClique = curClique;
        }
        return;
    }

    // Coloring bound to prune
    vector<int> colors;
    vector<int> order;
    colorGreedy(P, colors, order);

    // Upper bound: size(curClique) + max color
    int maxColor = 0;
    for (int idx : order) maxColor = max(maxColor, colors[idx]);
    if ((int)curClique.size() + maxColor <= (int)bestClique.size()) {
        return;
    }

    // Process vertices in the order determined
    // We'll iterate in reverse to try larger colored first (as common in Tomita)
    for (int t = (int)order.size() - 1; t >= 0; --t) {
        int i = order[t];
        int v = P[i];

        // prune if adding v cannot beat best
        if ((int)curClique.size() + colors[i] <= (int)bestClique.size()) {
            // since further vertices have color <= colors[i], can prune remaining
            continue;
        }

        // add v to current clique
        curClique.push_back(v);

        // build new candidate set P' = P ∩ N(v) in comp graph
        vector<int> Pnext;
        Pnext.reserve(P.size());
        for (int j = 0; j < (int)P.size(); ++j) {
            int u = P[j];
            if (adjComp[v][u]) {
                Pnext.push_back(u);
            }
        }

        if (!Pnext.empty()) {
            expand(Pnext);
        } else {
            // no more candidates, check best
            if ((int)curClique.size() > (int)bestClique.size()) {
                bestClique = curClique;
            }
        }

        // backtrack
        curClique.pop_back();
        // remove v from consideration by filtering P to exclude v
        // We already handle via loop using fixed order; but to ensure correctness, we can reconstruct P by removing v
        // Simpler: rebuild P removing v for remaining iterations
        // However, since we are iterating over a local copy (P) via order, we need to skip v in further iterations.
        // We'll implement by erasing v from P and also removing its index in order. But to keep simpler and safe, we proceed without modifying P here.
    }
}

// Entry: run maximum clique on complement graph
static void maxClique() {
    // initial candidate set: all vertices 0..n-1
    vector<int> P(n);
    iota(P.begin(), P.end(), 0);

    bestClique.clear();
    bestClique.reserve(n);

    // Start recursion
    // For performance, we call expand with initial P
    curClique.clear();
    expand(P);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) {
        return 0;
    }

    adjOrig.assign(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        adjOrig[u][v] = adjOrig[v][u] = 1;
    }

    // Build complement graph adjacency
    adjComp.assign(n, vector<char>(n, 0));
    for (int i = 0; i < n; ++i) {
        for (int j = i+1; j < n; ++j) {
            if (!adjOrig[i][j]) {
                adjComp[i][j] = adjComp[j][i] = 1;
            }
        }
    }

    maxClique();

    // bestClique is a maximum clique in complement graph -> MIS in original
    // Output
    int k = (int)bestClique.size();
    cout << k << "\n";
    if (k > 0) {
        for (int i = 0; i < k; ++i) {
            if (i) cout << ' ';
            cout << bestClique[i];
        }
        cout << "\n";
    } else {
        cout << "\n";
    }

    return 0;
}
