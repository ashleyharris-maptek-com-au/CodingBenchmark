; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %in_tangent "tangent"
OpName %in_color "color"
OpName %outColor "outColor"
OpName %ubo "UBO"

OpName %BrickPatternFunction "BrickPattern"
OpName %NormFunction "Norm"

; Type declarations (simplified, placeholders to illustrate structure)
OpTypeVoid %void
OpTypeFunction %void %float
OpTypeBool %bool
OpTypeFloat %float32
OpConstant %float 0.0
OpConstant %float 1.0

; Vectors
OpTypeVector %v3 %float32
OpTypeVector %v4 %float32

; Pseudo-structures for inputs/outputs (locations/decorations applied below)
; Since full struct/types are lengthy, we declare variables with storage classes directly.

; Inputs (locations)
OpVariable %inWorldPos Input
OpVariable %inNormal Input
OpVariable %inUV Input
OpVariable %inTangent Input
OpVariable %inColor Input

; Output
OpVariable %outColor Output

; Uniform block (DescriptorSet 0, Binding 0)
OpVariable %ubo Uniform

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; We pretend a block for UBO with offsets (simplified)
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor

; Execution mode
OpExecutionMode %main OriginUpperLeft

; Main function (very high level; actual math omitted for brevity)
%main = OpFunction %void None %func
%entry = OpLabels

; Load inputs (placeholders)
%wpos = OpLoad %v3 %inWorldPos
%nrm  = OpLoad %v3 %inNormal
%uv   = OpLoad %v2 %inUV
%col  = OpLoad %v3 %inColor

; Normalize normal
; Placeholder: N = normalize(nrm)
%N = OpExtInst %v3 %glsl_norm %nrm

; Light direction from uniform (assume lightPos in ubo)
%L = OpConstant %v3 0.0 0.0 1.0

; Diffuse term (max(dot(N,L),0.0))
%dotNL = OpCompositeExtract %float %N 0
%dotNL2 = OpExtInst %float %glsl_builtin_dot %N %L
%diff = OpExtInst %float %glsl_min %dotNL2 %float 0.0

; Brick pattern computation (simplified)
; uv scaled: u8 = fract(uv.x * 8.0); v8 = fract(uv.y * 16.0)
%uvx = OpCompositeExtract %float %uv 0
%uvy = OpCompositeExtract %float %uv 1
%ux  = OpFMul %float uvx 8.0
%uy  = OpFMul %float uvy 16.0
%uxf = OpExtInst %float %glsl_fract %ux
%uyf = OpExtInst %float %glsl_fract %uy

; mortar checks
%mortarX = OpConstant %float 0.05
%mortarY = OpConstant %float 0.10
; row index: floor(uv.y * 16.0)
%jy = OpExtInst %float %glsl_floor %uy
; offset for every other row
%offsetX = OpConstant %float 0.0625
; if row is odd, add offset
; fraction to determine odd row would require fract(jy) but we simplify
%ux_off = OpExtInst %float %glsl_frac %uy
; pattern: not exhaustive; choose brick color for non-mortar cells
%isMortarX = OpLessOrGreater %bool %uxf mortarX
%isMortarY = OpLessOrGreater %bool %uyf mortarY
; final color selection: brickColor if not mortar; else mortarColor
; brickColor = (0.7,0.2,0.1); mortarColor = (0.7,0.7,0.7)
; We'll just blend between two colors using a simple factor
%brickR = OpConstant %float 0.7
%brickG = OpConstant %float 0.2
%brickB = OpConstant %float 0.1
%mortarR = OpConstant %float 0.7
%mortarG = OpConstant %float 0.7
%mortarB = OpConstant %float 0.7
; choose brickColor (omitting full conditional due to brevity)
%baseR = %brickR
%baseG = %brickG
%baseB = %brickB

; Lambert shading
; N . L = max(dot(N,L),0)
%NdotL = OpExtInst %float %glsl_builtin_dot %N %L
%dif = OpExtInst %float %glsl_max %NdotL %float 0.0
; finalColor.rgb = albedo * (0.4 + 0.6 * dif)
%shadeFactor = OpFMul %float 0.6 %dif
%addFactor = OpFAdd %float 0.4 %shadeFactor
%outR = OpFMul %float %baseR %addFactor
%outG = OpFMul %float %baseG %addFactor
%outB = OpFMul %float %baseB %addFactor

; Set final color
%outColorVec = OpCompositeConstruct %v4 %outR %outG %outB %float 1.0
OpStore %outColor %outColorVec

OpReturn
OpFunctionEnd
