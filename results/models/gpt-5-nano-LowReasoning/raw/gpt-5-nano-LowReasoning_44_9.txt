#version 450
layout(location = 0) out vec4 outColor;  // RGBA float, stored as rgba8unorm

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

void main() {
    // 8x8 UV checkerboard
    vec2 uvScaled = uv * 8.0;
    vec2 f = floor(uvScaled);
    int ix = int(f.x);
    int iy = int(f.y);
    int parity = (ix + iy) & 1;

    vec3 baseColor = (parity == 0) ? vec3(0.9) : vec3(0.2);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diff = max(dot(N, L), 0.0);
    float shading = 0.5 + 0.5 * diff;

    vec3 finalColor = baseColor * shading;

    outColor = vec4(finalColor, 1.0);
}
