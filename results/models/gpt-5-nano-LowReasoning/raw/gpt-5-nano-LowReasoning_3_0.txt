```rust
use std::io::{self, Read};
use std::f64;
use std::cmp::min;

/// A simple, robust, scalable graph layout (force-directed) solver.
///
/// Notes:
/// - Works for small to very large graphs (6 up to ~200k nodes as per problem statement).
/// - Uses Fruchterman-Reingold style forces.
/// - For very large graphs, the repulsive force is approximated to avoid O(N^2).
/// - Handles disconnected components, self-loops, and parallel edges gracefully (self-loops ignored in layout, parallel edges simply add attraction).
/// - Outputs one line per node with x y coordinates (floats). Coordinates roughly in [0, 1], scaled by the caller if desired.
///
/// This is a self-contained Rust 1.56+ compatible program (Rust 1.92 in the prompt is older, but this code targets stable features available around that era).
fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // N M
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    // Build edge list
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut max_id = 0usize;
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            // Self-loop ignored for layout purposes
            continue;
        }
        if u < n && v < n {
            edges.push((u, v));
        }
        max_id = max(max_id, max(u, v));
    }
    // Create adjacency (for potential optimization, though not strictly required)
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for &(u, v) in &edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    // Random number generator (simple xorshift)
    struct RNG { a: u64 }
    impl RNG {
        fn new(seed: u64) -> Self { Self { a: seed ^ 0x9e3779b97f4a7c15 } }
        fn next_f64(&mut self) -> f64 {
            // xorshift64*
            let mut x = self.a;
            x ^= x >> 12;
            x ^= x << 25;
            x ^= x >> 27;
            self.a = x;
            // map to (0,1)
            let val = (x as f64) / (std::u64::MAX as f64);
            // avoid exactly 0 or 1
            if val <= 0.0 { 1e-12 } else if val >= 1.0 { 1.0 - 1e-12 } else { val }
        }
        fn next_usize(&mut self, bound: usize) -> usize {
            // modulo-based, but reduce bias a bit
            if bound <= 0 { return 0; }
            (self.next_f64() * bound as f64) as usize
        }
    }

    // Layout state per node
    #[derive(Clone, Copy)]
    struct Attr {
        x: f64, y: f64, vx: f64, vy: f64,
    }

    // Initialize positions randomly in [0, 1)
    let mut rnd = RNG::new(123456789);
    let mut nodes: Vec<Attr> = Vec::with_capacity(n);
    for _ in 0..n {
        nodes.push(Attr {
            x: rnd.next_f64(),
            y: rnd.next_f64(),
            vx: 0.0,
            vy: 0.0,
        });
    }

    // Parameters (tweakable)
    // We adapt the iteration strategy by graph size.
    // Small graphs get a more thorough layout; very large graphs get fewer expensive passes.
    let (iterations, area_scale, use_approx_repulsion) = if n <= 500 {
        // small
        (1600, 1.0, false)
    } else if n <= 5000 {
        (900, 1.0, true)  // medium
    } else if n <= 100000 {
        (600, 1.2, true) // large
    } else {
        (400, 1.2, true) // extreme
    };

    // Fruchterman-Reingold constants
    // k is the ideal distance between nodes: k ~ sqrt(area / n)
    // We'll scale area to keep coordinates reasonable
    // We adapt k by area_scale and current n
    let area: f64 = (n as f64).max(1.0);
    let k_base = area.sqrt(); // base ideal distance
    let k = k_base * area_scale;

    // Helpers for forces
    // For small graphs we compute all-pairs repulsion.
    // For large graphs we do a light, approximate repulsion (grid-based locality + random sampling).
    // We implement a simple grid-based approximation with cell size ~ k
    // and accumulate repulsive forces from neighboring cells.
    // To avoid complex data structures, we do a hybrid approach per iteration.

    // Simulation loop
    // We perform in-place updates. To keep things stable, we use a step size/drag.
    let mut step = 0usize;
    while step < iterations {
        step += 1;

        // Reset forces
        // We'll accumulate forces into temporary arrays
        let mut fx = vec![0.0f64; n];
        let mut fy = vec![0.0f64; n];

        // Repulsive forces
        if n <= 2500 || !use_approx_repulsion {
            // All-pairs repulsion (O(n^2)) for small graphs
            for i in 0..n {
                let xi = nodes[i].x;
                let yi = nodes[i].y;
                for j in (i+1)..n {
                    let dx = xi - nodes[j].x;
                    let dy = yi - nodes[j].y;
                    let dist = (dx*dx + dy*dy).sqrt() + 1e-9;
                    let f = (k * k) / dist;
                    let (dxn, dyn) = if dist > 0.0 { (dx / dist, dy / dist) } else { (0.0, 0.0) };
                    fx[i] += f * dxn;
                    fy[i] += f * dyn;
                    fx[j] -= f * dxn;
                    fy[j] -= f * dyn;
                }
            }
        } else {
            // Approximate repulsion: grid-based neighborhood + random sampling
            // Build a simple hash grid
            let cell = (k.max(0.1)).max(0.001);
            let mut grid_min_x = f64::MAX;
            let mut grid_min_y = f64::MAX;
            let mut grid_max_x = f64::MIN;
            let mut grid_max_y = f64::MIN;
            for i in 0..n {
                if nodes[i].x < grid_min_x { grid_min_x = nodes[i].x; }
                if nodes[i].y < grid_min_y { grid_min_y = nodes[i].y; }
                if nodes[i].x > grid_max_x { grid_max_x = nodes[i].x; }
                if nodes[i].y > grid_max_y { grid_max_y = nodes[i].y; }
            }
            // normalize to [0,1] area
            // build grid counts
            let gsize = 32; // grid dimension
            let mut buckets: Vec<Vec<usize>> = vec![Vec::new(); gsize * gsize];
            let mut gx = vec![0usize; n];
            let mut gy = vec![0usize; n];
            for i in 0..n {
                // compute grid coord
                let mut cx = ((nodes[i].x * (gsize as f64 - 1.0)).floor() as isize).clamp(0, gsize as isize - 1) as usize;
                let mut cy = ((nodes[i].y * (gsize as f64 - 1.0)).floor() as isize).clamp(0, gsize as isize - 1) as usize;
                gx[i] = cx;
                gy[i] = cy;
                let idx = cy * gsize + cx;
                buckets[idx].push(i);
            }
            // For each node, accumulate repulsion from neighboring cells
            for i in 0..n {
                let mut acc_fx = 0.0;
                let mut acc_fy = 0.0;
                let cx = gx[i];
                let cy = gy[i];
                // consider a 3x3 neighborhood
                for oy in cy.saturating_sub(1)..=min(cy + 1, gsize - 1) {
                    for ox in cx.saturating_sub(1)..=min(cx + 1, gsize - 1) {
                        let idx = oy * gsize + ox;
                        for &j in &buckets[idx] {
                            if j == i { continue; }
                            let dx = nodes[i].x - nodes[j].x;
                            let dy = nodes[i].y - nodes[j].y;
                            let dist = (dx*dx + dy*dy).sqrt() + 1e-9;
                            let f = (k * k) / dist;
                            let (dxn, dyn) = if dist > 0.0 { (dx / dist, dy / dist) } else { (0.0, 0.0) };
                            acc_fx += f * dxn;
                            acc_fy += f * dyn;
                        }
                    }
                }
                fx[i] += acc_fx;
                fy[i] += acc_fy;
            }
            // Additionally, sample a random set of other nodes to add some global repulsion
            let samples = 8;
            for i in 0..n {
                for _s in 0..samples {
                    let j = (i + 1 + rnd.next_usize(min(n - i - 1, 1000))) % n;
                    if j == i { continue; }
                    let dx = nodes[i].x - nodes[j].x;
                    let dy = nodes[i].y - nodes[j].y;
                    let dist = (dx*dx + dy*dy).sqrt() + 1e-9;
                    let f = (k * k) / dist;
                    let (dxn, dyn) = if dist > 0.0 { (dx / dist, dy / dist) } else { (0.0, 0.0) };
                    fx[i] += f * dxn;
                    fy[i] += f * dyn;
                }
            }
        }

        // Attractive forces along edges
        // For each edge (u,v), apply attraction: f = dist^2 / k
        for &(u, v) in &edges {
            let dx = nodes[u].x - nodes[v].x;
            let dy = nodes[u].y - nodes[v].y;
            let dist = (dx*dx + dy*dy).sqrt() + 1e-9;
            // use average k for attraction
            let f = (dist * dist) / k;
            let (dxn, dyn) = if dist > 0.0 { (dx / dist, dy / dist) } else { (0.0, 0.0) };
            fx[u] -= f * dxn;
            fy[u] -= f * dyn;
            fx[v] += f * dxn;
            fy[v] += f * dyn;
        }

        // Integrate velocities and positions
        // Parameters: temperature-like decay to gradually freeze layout
        // We'll implement a simple cooling: variable 't' decreases over iterations
        // t in [t0, t1]
        let t = 0.1 + (iterations - step) as f64 * 0.00008; // simple cooling
        for i in 0..n {
            // update velocity
            nodes[i].vx = (nodes[i].vx + fx[i]) * 0.82;
            nodes[i].vy = (nodes[i].vy + fy[i]) * 0.82;

            // update position
            nodes[i].x += nodes[i].vx * t;
            nodes[i].y += nodes[i].vy * t;
        }

        // Optional: bounds wraparound to keep within [0,1]
        for i in 0..n {
            if nodes[i].x < 0.0 { nodes[i].x = 0.0; nodes[i].vx = 0.0; }
            if nodes[i].x > 1.0 { nodes[i].x = 1.0; nodes[i].vx = 0.0; }
            if nodes[i].y < 0.0 { nodes[i].y = 0.0; nodes[i].vy = 0.0; }
            if nodes[i].y > 1.0 { nodes[i].y = 1.0; nodes[i].vy = 0.0; }
        }

        // Optional: early stop if energy is small (not strictly computed here)
        // Skipped for simplicity and robustness
    }

    // Output: one line per node with x y coordinates
    // The problem allows an optional first line with N; we provide only coordinates.
    // Print with sufficient precision.
    let mut out = String::new();
    for i in 0..n {
        // scale / shift if desired by caller; we keep [0,1]
        // Ensure pretty formatting
        out.push_str(&format!("{:.6} {:.6}\n", nodes[i].x, nodes[i].y));
    }

    print!("{}", out);
}
```
