```cpp
#include <bits/stdc++.h>
using namespace std;

/*
  N-Dimensional Snake Player (simple heuristic solver)
  - Works for arbitrary dimension D (D >= 1)
  - Maintains a snake body as a deque of D-dimensional integer points
  - At each turn, considers 2*D possible moves along each axis (+1 / -1)
  - Avoids obstacles and body collisions (allows moving into the current tail unless growing)
  - Tries to move toward the nearest food (Manhattan distance)
  - If food eaten, the snake grows (tail does not move)
  - If no safe move available, the game is lost
  - Initialization: the initial body is laid out along the first feasible axis from the head
*/

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        // FNV-1a like simple combine
        size_t h = 1469598103934665603ull;
        for (int x : v) {
            h ^= static_cast<size_t>(x);
            h *= 1099511628211ull;
        }
        return h;
    }
};

static inline int manhattan_dist(const vector<int>& a, const vector<int>& b) {
    int d = 0;
    for (size_t i = 0; i < a.size(); ++i) d += abs(a[i] - b[i]);
    return d;
}

static inline bool vec_eq(const vector<int>& a, const vector<int>& b) {
    if (a.size() != b.size()) return false;
    for (size_t i = 0; i < a.size(); ++i) if (a[i] != b[i]) return false;
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    long long max_turns;
    if (!(cin >> D >> max_turns)) {
        return 0;
    }
    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_len;
    vector<int> head(D);
    cin >> snake_len;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> foods;
    foods.reserve(F);
    for (int f = 0; f < F; ++f) {
        vector<int> pos(D);
        for (int i = 0; i < D; ++i) cin >> pos[i];
        foods.push_back(pos);
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash> obstacles;
    obstacles.reserve(O * 2 + 8);
    for (int o = 0; o < O; ++o) {
        vector<int> pos(D);
        for (int i = 0; i < D; ++i) cin >> pos[i];
        obstacles.insert(pos);
    }

    // Normalize: ensure foods unique and/or in bounds (not strictly required)
    // Build initial snake body along the first axis that can accommodate length
    deque<vector<int>> body;
    bool placed = false;

    // Try to place along axis 0, 1, ..., D-1
    for (int axis = 0; axis < D && !placed; ++axis) {
        // attempt to lay head, head - i along axis
        int tail_index = snake_len - 1;
        int head_coord = head[axis];
        int min_coord = head_coord - tail_index;
        if (min_coord >= 0 && head_coord <= bounds[axis]) {
            // feasible along this axis
            body.clear();
            for (int i = 0; i < snake_len; ++i) {
                vector<int> cell = head;
                cell[axis] = head_coord - i;
                // If any coordinate goes out of bounds (other dims unchanged), check
                bool ok = true;
                // Check bounds for all dims
                for (int d = 0; d < D; ++d) {
                    if (cell[d] < 0 || cell[d] > bounds[d]) { ok = false; break; }
                }
                if (!ok) { ok = false; }
                if (ok) {
                    body.push_back(cell);
                } else {
                    ok = false;
                    break;
                }
            }
            if ((int)body.size() == snake_len) {
                placed = true;
            } else {
                body.clear();
            }
        }
    }

    if (!placed) {
        // Fallback: place along axis 0 starting from head as far as fits
        body.clear();
        int axis = 0;
        int max_len = min(snake_len, bounds[axis] + 1);
        int head_coord = head[axis];
        int min_coord = max(0, head_coord - (snake_len - 1));
        if (min_coord + snake_len - 1 <= bounds[axis]) {
            for (int i = 0; i < snake_len; ++i) {
                vector<int> cell = head;
                cell[axis] = head_coord - i;
                body.push_back(cell);
            }
            placed = true;
        } else {
            // last resort: construct a straight line within bounds by clamping
            body.clear();
            for (int i = 0; i < snake_len; ++i) {
                vector<int> cell = head;
                int c = max(0, min(bounds[axis], head_coord - i));
                cell[axis] = c;
                body.push_back(cell);
            }
            placed = true;
        }
    }

    // If still not valid, exit gracefully
    if ((int)body.size() != snake_len) {
        cerr << "Error: could not initialize snake body with given head and length.\n";
        return 0;
    }

    // Build a quick lookup for current body points
    auto in_body = [&](const vector<int>& p) -> bool {
        for (const auto &q : body) if (vec_eq(p, q)) return true;
        return false;
    };

    // Print per-turn moves as required
    long long turn = 0;
    // Main loop
    while (turn < max_turns) {
        // Check win condition: no foods remaining
        if (foods.empty()) {
            // Won
            break;
        }

        // Determine closest food (Manhattan)
        int best_food_idx = -1;
        int best_dist = INT_MAX;
        for (size_t i = 0; i < foods.size(); ++i) {
            int d = manhattan_dist(head, foods[i]);
            if (d < best_dist) {
                best_dist = d;
                best_food_idx = (int)i;
            }
        }

        // Generate candidate moves
        struct Move { int axis; int dir; vector<int> new_head; bool safe; int dist_after; bool will_grow; };
        vector<Move> candidates;
        candidates.reserve(2 * D);

        // Current tail (for safe move check)
        const vector<int>& tail = body.back();

        for (int axis = 0; axis < D; ++axis) {
            for (int sgn = -1; sgn <= 1; sgn += 2) {
                vector<int> nh = head;
                nh[axis] += sgn;

                // bounds check
                if (nh[axis] < 0 || nh[axis] > bounds[axis]) continue;

                // collision with obstacles?
                if (obstacles.find(nh) != obstacles.end()) continue;

                // determine if growing on this move (head lands on a food)
                bool growing = false;
                if (!foods.empty()) {
                    if (vec_eq(nh, foods[best_food_idx])) growing = true;
                    else {
                        // if some other food is at nh
                        // (still consider if any food exactly at nh)
                        for (size_t fi = 0; fi < foods.size(); ++fi) {
                            if (vec_eq(nh, foods[fi])) { growing = true; break; }
                        }
                    }
                }

                // collision with body: allowed to move into current tail if not growing
                bool coll = false;
                for (size_t i = 0; i + 1 < body.size(); ++i) {
                    if (vec_eq(nh, body[i])) { coll = true; break; }
                }
                if (coll) continue;
                // Allow tail collision only if not growing (tail will move)
                if (vec_eq(nh, tail) && growing) {
                    // moving into tail but tail won't vacate since we grow
                    continue;
                }

                int dist = INT_MAX;
                // if we know nearest food index, compute distance after move to that food
                if (!foods.empty()) {
                    dist = manhattan_dist(nh, foods[best_food_idx]);
                }

                candidates.push_back({axis, sgn, nh, true, dist, growing});
            }
        }

        // Choose best candidate
        Move chosen;
        bool found = false;
        if (!candidates.empty()) {
            // Prefer those that reduce distance to chosen food
            int cur_dist = (best_food_idx >= 0) ? manhattan_dist(head, foods[best_food_idx]) : INT_MAX;
            int best_score = INT_MAX;
            for (auto &m : candidates) {
                int score = (m.dist_after == INT_MAX ? cur_dist + 1000 : m.dist_after);
                // slight bias for smaller axis
                score = score * 1000 + (1000 - m.axis);
                if (score < best_score) {
                    best_score = score;
                    chosen = m;
                    found = true;
                }
            }
        }

        if (!found) {
            // No safe move
            break; // lose
        }

        // Apply move
        int axis = chosen.axis;
        int dir = chosen.dir;
        vector<int> new_head = chosen.new_head;

        bool eating = false;
        // If new_head equals a food, eat it
        int eat_index = -1;
        for (size_t fi = 0; fi < foods.size(); ++fi) {
            if (vec_eq(new_head, foods[fi])) { eating = true; eat_index = (int)fi; break; }
        }
        if (eating) {
            // remove that food
            foods.erase(foods.begin() + eat_index);
        }

        // Update body
        body.push_front(new_head);
        if (!eating) {
            // move tail
            body.pop_back();
        }

        // Update head reference
        head = new_head;

        // Output move
        cout << axis << ' ' << dir << "\n";
        cout.flush();

        ++turn;
        if (foods.empty()) {
            // Reached win condition
            break;
        }
    }

    // Program ends when either win/lose or max turns reached.
    return 0;
}
```
