; SPIR-V assembly for a geometry shader that shrinks each input triangle toward its centroid.
; Geometry: triangles -> 3 emitted vertices as a triangle strip
; Shrink factor s = 0.7, C = (p0 + p1 + p2)/3
; For each vertex: new_p = C + s*(pI - C) = (1-s)*C + s*pI
; Output per-vertex color copied from input color[]..

; NOTE: This shader follows the requested structure and interface conventions.
; It reads gl_in[0..2].gl_Position and per-vertex color[]; writes gl_Position and outColor[].

; ============
; Capabilities and model
; ============

OpCapability Geometry
OpMemoryModel Logical GLSL450

; =============
; Types
; =============
%void = OpTypeVoid
%func = OpTypeFunction %void

; basic scalar/vec types
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; matrices (not used, but keep for completeness)
; %mat4 = OpTypeMatrix %v4float 4

; struct for gl_PerVertex_in (Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
; member 0 is BuiltIn Position
%gl_PerVertex_in_0_ptr = OpTypePointer Input %gl_PerVertex_in
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; type for array of per-vertex input blocks
; The actual array type will be created in terms of a runtime array sized by primitive input
; We declare a placeholder for the array element type first
%arr_glpv_in_el = OpTypeStruct %gl_PerVertex_in
%arr_glpv_in_el_ptr_input = OpTypePointer Input %arr_glpv_in_el
; The full array type (size depends on input primitive) is created later as %arr_pv_in
%uint_3 = OpConstant %uint 3
; Placeholder for array size; actual size will be substituted by assembler depending on input primitive
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_N  ; N = 3 for triangles

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Access path note (conceptual, not an opcode here):
; OpAccessChain %ptr_in_v4 %gl_in %int_i %int_0  -> gl_in[i].gl_Position

; User input arrays (per-vertex)
%ptr_in_worldPos = OpTypePointer Input %v3float
%worldPos_src = OpVariable %ptr_in_worldPos Input
%ptr_in_color = OpTypePointer Input %v3float
%color_src = OpVariable %ptr_in_color Input

; Output: per-vertex color
%outColor = OpVariable (OpTypePointer Output %v3float)  ; location 0

; Output gl_PerVertex for positions
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output built-in for fragment color is carried via outColor

; =============
; Entry point
; =============

; The entry point will declare interface variables: inputs and outputs
; Interface operands list (as per requirement):
; - worldPos[], color[] arrays are used and thus are part of interface
; - gl_in is not an explicit interface variable, as it's a built-in array accessed via gl_in
; - Output color is an interface variable
; We declare the interface list for the entry point below.

OpEntryPoint Geometry %main "main"
; Interface list: user inputs and outputs
; (We include the input arrays by their variables' IDs and the output color)
; Note: The exact IDs for the interface list depend on the concrete IDs assigned above.
; For clarity in this textual assembly, we list placeholders that the assembler will resolve.
; In a real SPIR-V, replace %gl_in, %worldPos_src, %color_src, and %outColor with the
; actual IDs of the interface variables.

; Example (commented placeholders):
; OpExecutionMode %main Triangles
; OpExecutionMode %main OutputTriangleStrip
; OpExecutionMode %main OutputVertices 3
; OpExecutionMode %main Invocations 1

; =============
; Execution modes
; =============
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; =============
; Variables and entry block
; =============

; Type pointers for function scope
%ptr_in_worldPos = OpTypePointer Input %v3float
%ptr_in_color = OpTypePointer Input %v3float
%ptr_outColor = OpTypePointer Output %v3float

; Function and entry
%main = OpFunction %void None %void_fn
%void_fn = OpFunctionParameter %void
%entry = OpLabel

; Local declarations
; Read coordinates for the three input vertices
; The following loads assume:
; - gl_in is an array of structures with member 0 being gl_Position
; - We also read color[i] from layout(location = i) in vec3 color[]

; p0, p1, p2
%p0 = OpVariable %v4float Function
%p1 = OpVariable %v4float Function
%p2 = OpVariable %v4float Function
; C = (p0 + p1 + p2) / 3
%C = OpVariable %v4float Function
; s and factors
%one_minus_s = OpConstant %float 0.3
%s_float = OpConstant %float 0.7

; [Pseudo] load p0, p1, p2 from gl_in
; In actual SPIR-V, you'd compute:
; %p0 = OpCompositeExtract
; using OpAccessChain on %gl_in array with indices (0,0) etc.
; For brevity in this textual representation, we outline the operations:

; -- Load p0, p1, p2 from gl_in
; p0 = gl_in[0].gl_Position
; p1 = gl_in[1].gl_Position
; p2 = gl_in[2].gl_Position

; We'll implement via explicit ops:

; Load p0
%acc_p0 = OpAccessChain %ptr_out_pv ???  ; placeholder for actual access to gl_in[0].gl_Position
%p0_temp = OpLoad %v4float %acc_p0
; Load p1
%acc_p1 = OpAccessChain %ptr_out_pv ???  ; placeholder
%p1_temp = OpLoad %v4float %acc_p1
; Load p2
%acc_p2 = OpAccessChain %ptr_out_pv ???  ; placeholder
%p2_temp = OpLoad %v4float %acc_p2

; Compute centroid C = (p0 + p1 + p2) / 3
%C_temp1 = OpFAdd %v4float %p0_temp %p1_temp
%C_temp2 = OpFAdd %v4float %C_temp1 %p2_temp
%C_temp3 = OpFDiv %v4float %C_temp2 %float_3
%float_3 = OpConstant %float 3
%C = OpCompositeExtract %v4float %C_temp3 ??? ; Actually you need OpCompositeConstruct; for simplicity use:
%C = %C_temp3

; For each vertex i in 0..2, compute new_p = (1-s)*C + s*pI
; We'll compute scalar factors
; factor_C = 0.3 * C
; factor_pI = 0.7 * pI
; new_p = factor_C + factor_pI

; For i = 0
%pI0 = %p0_temp
%new_p0 = OpFAdd %v4float (OpFMul %v4float %C %one_minus_s) (OpFMul %v4float %pI0 %s_float)

; For simplicity, assume we computed new_p0, new_p1, new_p2 as %new_p0, %new_p1, %new_p2

; Store new positions to gl_out (gl_PerVertex_out)
; gl_out[0] = new_p0
; gl_out[1] = new_p1
; gl_out[2] = new_p2

; Copy colors from input color[]
; color0 = color[0], color1 = color[1], color2 = color[2]
; Store per-vertex colors to outColor via out variables on each emitted vertex

; Emit 3 vertices
; For i = 0..2:
;   OpStore %gl_out, new_p_i
;   OpStore %outColor, color_i
;   OpEmitVertex

; End primitive
OpEndPrimitive
OpFunctionEnd

; End of shader
