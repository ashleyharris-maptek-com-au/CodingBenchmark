```csharp
using System;
using System.Linq;
using System.Collections.Generic;

namespace OrbitalTSP
{
    class Program
    {
        // Vector3-like struct for simple 3D vector math
        struct Vec3
        {
            public double x, y, z;
            public Vec3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
            public static double Distance(Vec3 a, Vec3 b)
            {
                double dx = a.x - b.x;
                double dy = a.y - b.y;
                double dz = a.z - b.z;
                return Math.Sqrt(dx * dx + dy * dy + dz * dz);
            }
        }

        // Read all inputs and build data
        static void Main(string[] args)
        {
            try
            {
                // Read first line: N mu (mu is provided but not used directly here)
                string line1 = Console.ReadLine();
                if (string.IsNullOrWhiteSpace(line1)) return;
                string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                int N = int.Parse(parts1[0]);
                // double mu = double.Parse(parts1[1]); // mu is not used directly in this simple heuristic

                // Second line: start orbit: x y z vx vy vz
                string lineStart = Console.ReadLine();
                Vec3 startPos = new Vec3();
                {
                    string[] p = lineStart.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                    startPos = new Vec3(
                        double.Parse(p[0]),
                        double.Parse(p[1]),
                        double.Parse(p[2])
                    );
                    // Velocity ignored for the heuristic
                }

                // Remaining N lines: station orbits
                Vec3[] stations = new Vec3[N];
                for (int i = 0; i < N; i++)
                {
                    string line = Console.ReadLine();
                    string[] p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                    stations[i] = new Vec3(
                        double.Parse(p[0]),
                        double.Parse(p[1]),
                        double.Parse(p[2])
                    );
                }

                // Precompute distances
                double[,] dist = new double[N, N];
                for (int i = 0; i < N; i++)
                {
                    dist[i, i] = 0;
                    for (int j = i + 1; j < N; j++)
                    {
                        double d = Vec3.Distance(stations[i], stations[j]);
                        dist[i, j] = dist[j, i] = d;
                    }
                }

                double[] distFromStart = new double[N];
                for (int i = 0; i < N; i++)
                    distFromStart[i] = Vec3.Distance(startPos, stations[i]);

                // Choose strategy based on N
                List<int> bestOrder;

                if (N <= 8)
                {
                    // Trivial: brute-force permutation search
                    bestOrder = SolveBruteForce(N, distFromStart, dist);
                }
                else
                {
                    // Medium to Large: heuristic with improvement
                    bestOrder = SolveHeuristic(N, distFromStart, dist);
                }

                // Output: one line with N integers: visitation order (0..N-1)
                Console.WriteLine(string.Join(" ", bestOrder.Select(i => i.ToString())));
            }
            catch (Exception ex)
            {
                // In case of input issues, fail gracefully
                Console.WriteLine("0");
            }
        }

        // Brute-force solution for small N (<= 8)
        static List<int> SolveBruteForce(int N, double[] distFromStart, double[,] dist)
        {
            int[] idx = Enumerable.Range(0, N).ToArray();
            double bestCost = double.PositiveInfinity;
            int[] bestPerm = null;

            // We'll generate permutations iteratively to avoid recursion depth
            // Use simple next_permutation implementation
            Array.Sort(idx);
            bool done = false;
            int[] perm = new int[N];
            while (!done)
            {
                for (int i = 0; i < N; i++) perm[i] = idx[i];

                // Compute cost: start -> perm[0] -> perm[1] -> ... -> perm[N-1]
                double cost = distFromStart[perm[0]];
                for (int i = 0; i < N - 1; i++)
                    cost += dist[perm[i], perm[i + 1]];

                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestPerm = new int[N];
                    Array.Copy(perm, bestPerm, N);
                }

                // Next permutation
                done = !NextPermutation(idx);
            }

            return bestPerm.ToList();
        }

        // Generate next lexicographic permutation. Returns false if there is no next.
        static bool NextPermutation(int[] a)
        {
            int n = a.Length;
            int i = n - 2;
            while (i >= 0 && a[i] >= a[i + 1]) i--;
            if (i < 0) return false;
            int j = n - 1;
            while (a[j] <= a[i]) j--;
            Swap(ref a[i], ref a[j]);
            Reverse(a, i + 1, n - 1);
            return true;
        }

        static void Swap(ref int a, ref int b)
        {
            int t = a; a = b; b = t;
        }

        static void Reverse(int[] a, int l, int r)
        {
            while (l < r)
            {
                Swap(ref a[l], ref a[r]);
                l++; r--;
            }
        }

        // Heuristic solution for larger N
        static List<int> SolveHeuristic(int N, double[] distFromStart, double[,] dist)
        {
            // 1) Initial tour via greedy nearest-neighbor starting from the start position
            bool[] used = new bool[N];
            List<int> tour = new List<int>(N);

            // Start by picking the nearest unvisited station from the start
            int current = -1;
            double bestD = double.PositiveInfinity;
            for (int i = 0; i < N; i++)
            {
                if (distFromStart[i] < bestD)
                {
                    bestD = distFromStart[i];
                    current = i;
                }
            }

            if (current == -1) current = 0;
            tour.Add(current);
            used[current] = true;

            for (int k = 1; k < N; k++)
            {
                int nearest = -1;
                double best = double.PositiveInfinity;
                for (int i = 0; i < N; i++)
                {
                    if (used[i]) continue;
                    double d = dist[tour[k - 1], i];
                    if (d < best)
                    {
                        best = d;
                        nearest = i;
                    }
                }
                if (nearest == -1) break;
                tour.Add(nearest);
                used[nearest] = true;
            }

            // If somehow we didn't visit all, fill the rest (shouldn't happen often)
            for (int i = 0; i < N; i++)
                if (!used[i]) tour.Add(i);

            // 2) Improve with 2-opt until no improvement or limit reached
            bool improved = true;
            int maxIter = Math.Min(2000, N * N * 4);
            int iter = 0;

            while (improved && iter < maxIter)
            {
                improved = false;
                iter++;

                for (int i = 0; i < N - 1; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        double delta = DeltaTwoOpt(tour, dist, i, j);
                        if (delta < -1e-12) // improvement
                        {
                            TwoOptSwap(tour, i, j);
                            improved = true;
                        }
                    }
                }
            }

            return tour;
        }

        // Compute the change in total distance if we reverse the segment [i..j]
        static double DeltaTwoOpt(List<int> tour, double[,] dist, int i, int j)
        {
            int N = tour.Count;
            int a = tour[(i - 1 + N) % N];
            int b = tour[i];
            int c = tour[j];
            int d = tour[(j + 1) % N];

            // Current edges: (a-b) and (c-d)
            // New edges after reversal: (a-c) and (b-d)
            double current = dist[a, b] + dist[c, d];
            double next = dist[a, c] + dist[b, d];
            return next - current;
        }

        // Perform reversal of segment [i..j] in the tour
        static void TwoOptSwap(List<int> tour, int i, int j)
        {
            while (i < j)
            {
                int tmp = tour[i];
                tour[i] = tour[j];
                tour[j] = tmp;
                i++; j--;
            }
        }
    }
}
```
