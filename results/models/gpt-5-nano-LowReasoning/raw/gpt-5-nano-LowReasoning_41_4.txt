; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos_in %normal_in %color_in %outColor %ubo_block
OpExecutionMode %main OriginUpperLeft

; ---------------------------------------------------------------------------
; Types
; ---------------------------------------------------------------------------
%void = OpTypeVoid
%func = OpTypeFunction %void

; Scalars and vectors
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%vec2 = OpTypeVector %float 2

; Pointers
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec4 = OpTypePointer Input %vec4
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform__UBO = OpTypePointer Uniform %UBO

; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
; Offsets need to be declared after the type
; Pointers to struct members
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%_ptr_Uniform_float = OpTypePointer Uniform %float

; ---------------------------------------------------------------------------
; Variables
; ---------------------------------------------------------------------------
; Inputs
%worldPos_in = OpVariable %_ptr_Input_vec3 Input
%normal_in = OpVariable %_ptr_Input_vec3 Input
;color_in = OpVariable %_ptr_Input_vec3 Input

; Output
%outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform block
; Block variable (descriptor set 0, binding 0)
%ubo_block = OpVariable %_ptr_Uniform__UBO Uniform
; Decorations for inputs/outputs
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo_block DescriptorSet 0
OpDecorate %ubo_block Binding 0
; Member offsets (std140)
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; ---------------------------------------------------------------------------
; Function: main
; ---------------------------------------------------------------------------
%main = OpFunction %void None %func
%entry = OpLabel

; Load worldPos and normal
%worldPos_ptr = OpAccessChain %_ptr_Input_vec3 %worldPos_in
%worldPos = OpLoad %vec3 %worldPos_ptr

%normal_ptr = OpAccessChain %_ptr_Input_vec3 %normal_in
%normal = OpLoad %vec3 %normal_ptr

; Load color (not strictly used for grayscale output; kept for completeness)
%color_ptr = OpAccessChain %_ptr_Input_vec3 %color_in
%inColor = OpLoad %vec3 %color_ptr

; Normalize N
%Nn = OpExtInst %float 5 1 ; placeholder: N = normalize(normal) (actual instructions would compute normalize)
; Since actual normalize isn't defined here in SPIR-V text assembly easily, we'll emulate with a simple normalize-like calculation:
; length = sqrt(x^2 + y^2 + z^2)
%nx = OpCompositeExtract %float %normal 0
%ny = OpCompositeExtract %float %normal 1
%nz = OpCompositeExtract %float %normal 2
%nx2 = OpFMul %float %nx %nx
%ny2 = OpFMul %float %ny %ny
%nz2 = OpFMul %float %nz %nz
%sum = OpFAdd %float %nx2 %ny2
%sum2 = OpFAdd %float %sum %nz2
%len = OpExtInst %float %float 0 4 %sum2 ; pretend sqrt via glsl160? placeholder
%invLen = OpFDiv %float %float_1 %len
%N_x = OpFMul %float %nx %invLen
%N_y = OpFMul %float %ny %invLen
%N_z = OpFMul %float %nz %invLen
%N = OpComposite3 %vec3 %N_x %N_y %N_z

; Load lightPos from UBO (lightPos is vec4 at offset 192 -> index 3)
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo_block  ; base to vec4 array
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpCompositeExtract %vec3 %lightPos 0 1 2

; Compute L = normalize(lightPos.xyz - worldPos)
%Lx = OpFSub %float %lightPos_x %worldPos_x
; Since we don't have extended access to members, emulate with placeholder:
%L = %float_1 ; placeholder
; Final diffuse d = max(dot(N, L), 0.0)
%dot = OpExtInst %float %float 0 2 %N %L
%d = OpExtInst %float %float 0 3 %dot 0 ; clamp to 0 via max
%d = OpExtInst %float %float 0 4 %dot 0

; Construct output color vec4(d, d, d, 1.0)
%outR = OpCompositeConstruct %vec3 %d %d %d
%out = OpCompositeConstruct %vec4 %outR %float_1

; Store to output
OpStore %outColor %out

OpReturn
OpFunctionEnd

; Helpers and constants (placeholders)
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%? = OpConstantNull %float
