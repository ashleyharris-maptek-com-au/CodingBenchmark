; SPIR-V Geometry Shader (GLSL450 flavor via SPIR-V assembly)
; Purpose: explode each input triangle outward along its face normal by offset 0.08
; Geometry: OutputTriangleStrip with 3 vertices (a triangle expanded to a strip of 3)

 OpCapability Geometry
 OpMemoryModel Logical GLSL450
; GLSL standard include for helper ops
 %GLSL450 = OpExtInstImport "GLSL.std.450"

; Type definitions
 %void = OpTypeVoid
 %func = OpTypeFunction %void
 %float = OpTypeFloat 32
 %v2float = OpTypeVector %float 2
 %v3float = OpTypeVector %float 3
 %v4float = OpTypeVector %float 4
 %v4float_padded = OpTypeVector %float 4
 %v3float_ptr_input = OpTypePointer Input %v3float
 %v4float_ptr_input = OpTypePointer Input %v4float
 %v3float_ptr_output = OpTypePointer Output %v3float
 %v4float_ptr_output = OpTypePointer Output %v4float
 %v4float_ptr_out = OpTypePointer Output %v4float
 %ptr_input_arr_pv = OpTypePointer Input %gl_PerVertex_in
 %ptr_out_arr_pv = OpTypePointer Output %gl_PerVertex_out

; Floating constants
 %float_0 = OpConstant %float 0.0
 %float_08 = OpConstant %float 0.08

; Forward declarations of the per-vertex in/out blocks (as described)
 ; gl_PerVertex_in structurally contains a vec4 gl_Position at member 0
 ; gl_PerVertex_out structurally contains a vec4 gl_Position at member 0

 ; Original GL_PER_VERTEX type and accessors (as provided in prompt)
 %gl_PerVertex_in = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_in Block
 OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_N       ; N = 1/2/3/4/6 depending on input primitive
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
 %gl_in = OpVariable %ptr_in_arr_pv Input

 ; Our per-vertex inputs from vertex shader (arrays)
 %worldPos_arr_in = OpTypeArray %v3float %uint_N
 %ptr_worldPos_in = OpTypePointer Input %worldPos_arr_in
 %worldPos_in = OpVariable %ptr_worldPos_in Input

 %normal_arr_in = OpTypeArray %v3float %uint_N
 %ptr_normal_in = OpTypePointer Input %normal_arr_in
 %normal_in = OpVariable %ptr_normal_in Input

 %color_arr_in = OpTypeArray %v3float %uint_N
 %ptr_color_in = OpTypePointer Input %color_arr_in
 %color_in = OpVariable %ptr_color_in Input

 ; Output gl_PerVertex
 %gl_PerVertex_out = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_out Block
 OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
 %gl_out = OpVariable %ptr_out_pv Output

 ; Output color
 layout(location = 0) out vec3 outColor
 ; No separate storage class pointer for outColor is required beyond the declared variable above

 ; Input/output interface and entry point
 OpName %main "main"
 OpEntryPoint Geometry %main "main" ; interface variables listed below

 ; Execution modes
 OpExecutionMode %main Triangles
 OpExecutionMode %main OutputTriangleStrip
 OpExecutionMode %main OutputVertices 3
 OpExecutionMode %main Invocations 1

 ; Interface: we declare the input/output and UBO for completeness
;  Input interface variables (worldPos, normal, color) are used by the GS
 ; Declare memory for gl_in access to each vertex (as described)
 ; Nothing more required here for interface; they are part of the operation as inputs

; Uniforms (UBO) - set=0, binding=0 (as per prompt). Declared here but not strictly used in this minimal explode pass.
; Access to UBO fields would be via OpAccessChain with a pointer to a block inside the uniform.
 %ubo_type = OpTypeStruct
 %ptr_ubo = OpTypePointer Uniform %ubo_type
 %ubo = OpVariable %ptr_ubo Uniform

 ; We also need a few constants for indexing
 %int_0 = OpConstant %int 0
 %int_1 = OpConstant %int 1
 %int_2 = OpConstant %int 2
 %int_3 = OpConstant %int 3
 %int_4 = OpConstant %int 4
 %int_5 = OpConstant %int 5
 %int_6 = OpConstant %int 6

 ; Helper: define function
 %main = OpFunction %void None %func
 %entry = OpLabel

 ; Step 1: Load the three positions from gl_in
 ; p0 = gl_in[0].gl_Position
 %p0_ptr = OpAccessChain %v4float_ptr_input %gl_in %int_0 %int_0
 %p0 = OpLoad %v4float %p0_ptr

 ; p1 = gl_in[1].gl_Position
 %p1_ptr = OpAccessChain %v4float_ptr_input %gl_in %int_1 %int_0
 %p1 = OpLoad %v4float %p1_ptr

 ; p2 = gl_in[2].gl_Position
 %p2_ptr = OpAccessChain %v4float_ptr_input %gl_in %int_2 %int_0
 %p2 = OpLoad %v4float %p2_ptr

 ; Extract xyz for normal computation
 %p0_xyz = OpCompositeExtract %v3float %p0 0 1 2
 %p1_xyz = OpCompositeExtract %v3float %p1 0 1 2
 %p2_xyz = OpCompositeExtract %v3float %p2 0 1 2

 ; e1 = p1.xyz - p0.xyz
 %e1 = OpFSub %v3float %p1_xyz %p0_xyz
 ; e2 = p2.xyz - p0.xyz
 %e2 = OpFSub %v3float %p2_xyz %p0_xyz

 ; cross = cross(e1, e2) using GLSL std450 cross
 %cross = OpExtInst %v3float %GLSL450 Cross %e1 %e2

 ; N = normalize(cross)
 %N = OpExtInst %v3float %GLSL450 Normalize %cross

 ; offset = N * 0.08
 %offset_vec3 = OpFMul %v3float %N %float_08

 ; Build offset as vec4 (x,y,z, w=0) for addition with a vec4 gl_Position
 %offset_vec4 = OpCompositeConstruct %v4float %offset_vec3 0.0

 ; We now compute new positions for the 3 vertices by adding offset to each original gl_Position
 ; newPos_i = gl_in[i].gl_Position + vec4(offset.xyz, 0.0)

 ; Vertex 0
 %gl_in0_ptr = OpAccessChain %v4float_ptr_input %gl_in %int_0 %int_0
 %gl_in0 = OpLoad %v4float %gl_in0_ptr
 %gl_in0_shifted = OpFAdd %v4float %gl_in0 %offset_vec4

 ; Color for vertex 0
 %col0_ptr = OpAccessChain %v3float_ptr_input %color_in %int_0
 %col0 = OpLoad %v3float %col0_ptr
 ; Store gl_Position
 %gl_out0_ptr = OpAccessChain %gl_PerVertex_out %gl_out %int_0
 %gl_out0 = OpCompositeConstruct %v4float %gl_in0_shifted  ; but need to place Position inside gl_PerVertex_out
 ; Since gl_out is a struct with gl_Position as first member, we store directly:
 OpStore %gl_out %gl_in0_shifted

 ; Write outColor for vertex 0
 OpStore %outColor %col0
 ; Emit vertex 0
 OpEmitVertex

 ; Vertex 1
 %gl_in1_ptr = OpAccessChain %v4float_ptr_input %gl_in %int_1 %int_0
 %gl_in1 = OpLoad %v4float %gl_in1_ptr
 %gl_in1_shifted = OpFAdd %v4float %gl_in1 %offset_vec4

 %col1_ptr = OpAccessChain %v3float_ptr_input %color_in %int_1
 %col1 = OpLoad %v3float %col1_ptr

 OpStore %gl_out %gl_in1_shifted
 OpStore %outColor %col1
 OpEmitVertex

 ; Vertex 2
 %gl_in2_ptr = OpAccessChain %v4float_ptr_input %gl_in %int_2 %int_0
 %gl_in2 = OpLoad %v4float %gl_in2_ptr
 %gl_in2_shifted = OpFAdd %v4float %gl_in2 %offset_vec4

 %col2_ptr = OpAccessChain %v3float_ptr_input %color_in %int_2
 %col2 = OpLoad %v3float %col2_ptr

 OpStore %gl_out %gl_in2_shifted
 OpStore %outColor %col2
 OpEmitVertex

 ; End the primitive (triangle strip with 3 vertices)
 OpEndPrimitive

 ; Function end
 OpReturn
 OpFunctionEnd
