```csharp
using System;
using System.Collections.Generic;

class MinesweeperSolver
{
    // State encoding:
    // -1 : Unknown (' ')
    // -2 : Flagged ('*')
    // -3 : Proven Safe (provably not a mine) (temporary/derived)
    //  0..8 : Revealed with that adjacent mine count

    static void Main()
    {
        // Read board dimensions
        string first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
            return;
        string[] parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);

        int[,] cell = new int[H, W]; // encoding as described
        bool[,] originallyUnknown = new bool[H, W];

        // Initialize
        for (int r = 0; r < H; r++)
        {
            string line = Console.ReadLine();
            if (line == null) line = "";
            // Pad if line too short
            if (line.Length < W)
                line = line.PadRight(W, ' ');

            for (int c = 0; c < W; c++)
            {
                char ch = line[c];
                if (ch == ' ')
                {
                    cell[r, c] = -1; // Unknown
                    originallyUnknown[r, c] = true;
                }
                else if (ch == '*')
                {
                    cell[r, c] = -2; // Flagged mine
                }
                else if (ch >= '0' && ch <= '8')
                {
                    cell[r, c] = ch - '0'; // Revealed with count
                }
                else
                {
                    // Treat any other as Unknown for safety
                    cell[r, c] = -1;
                    originallyUnknown[r, c] = true;
                }
            }
        }

        // Adaptive solving: constraint propagation with simple Minesweeper rules
        // Rule: If a revealed cell shows n, among its neighbors the number of flagged mines is m,
        // and the number of unknown neighbors is k, then r = n - m.
        // If r == 0, all unknown neighbors are safe.

        // Queue of revealed cells to check
        Queue<(int r, int c)> q = new Queue<(int, int)>();
        // Enqueue all revealed cells initially
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (cell[r, c] >= 0) q.Enqueue((r, c));

        // Directions for 8 neighbors
        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        // Helper to safely mark a cell as Safe if currently Unknown
        Action<int,int> markSafe = (rr, cc) =>
        {
            if (rr < 0 || cc < 0 || rr >= H || cc >= W) return;
            if (cell[rr, cc] == -1) // Unknown
            {
                cell[rr, cc] = -3; // Mark as Safe (provably without mine)
                // When a cell becomes Safe, any adjacent revealed cells constraints
                // may be affected; enqueue neighboring revealed cells to re-check.
                for (int d = 0; d < 8; d++)
                {
                    int ar = rr + dr[d];
                    int ac = cc + dc[d];
                    if (ar < 0 || ac < 0 || ar >= H || ac >= W) continue;
                    if (cell[ar, ac] >= 0) // revealed
                        q.Enqueue((ar, ac));
                }
            }
        };

        // Also a quick helper to know if a cell is Unknown
        Func<int,int,bool> isUnknown = (rr, cc) => (rr >= 0 && cc >= 0 && rr < H && cc < W && cell[rr,cc] == -1);

        while (q.Count > 0)
        {
            var cur = q.Dequeue();
            int r = cur.r, c = cur.c;

            // Only process if this is a revealed cell
            int val = cell[r, c];
            if (val < 0) continue; // not a revealed numeric cell

            int m = 0; // flagged neighbors
            List<(int, int)> unknowns = new List<(int,int)>();

            for (int k = 0; k < 8; k++)
            {
                int nr = r + dr[k];
                int nc = c + dc[k];
                if (nr < 0 || nc < 0 || nr >= H || nc >= W) continue;
                int st = cell[nr, nc];
                if (st == -2) m++;
                else if (st == -1) unknowns.Add((nr, nc));
                // Safe (-3) or Revealed (>=0) or Flagged (-2) are not counted as unknown
            }

            int remaining = val - m;
            if (remaining < 0) continue; // inconsistent state; skip

            if (unknowns.Count > 0 && remaining == 0)
            {
                // All unknown neighbors are safe
                foreach (var (rr, cc) in unknowns)
                {
                    markSafe(rr, cc);
                }
            }
            // We could implement more advanced deductions (e.g., remaining == unknowns.Count),
            // but for the purpose of provably safe cells, the r==0 rule suffices in many cases.
        }

        // Collect provably safe cells that were originally unknown
        List<(int x, int y)> result = new List<(int, int)>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (originallyUnknown[r, c] && cell[r, c] == -3)
                {
                    // Output format: x y with 0-based indices
                    result.Add((c, r));
                }
            }
        }

        // Output
        Console.WriteLine(result.Count);
        foreach (var p in result)
        {
            Console.WriteLine($"{p.x} {p.y}");
        }
    }
}
``` 
