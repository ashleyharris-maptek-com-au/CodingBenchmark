```javascript
// Node.js 24+ compatible
// Simple heuristic planner for delivery routes.
// Input: JSON with { depot: [x, y], truckCount: N, packages: [{id, x, y}, ...]}
// Output: { "routes": [[packageId, packageId, ...], ...] }
// Each route starts/ends at the depot (depot is not included in the route array).

'use strict';

const fs = require('fs');

// Read input from stdin as a single JSON document
const input = fs.readFileSync(0, 'utf8');
let data;
try {
  data = JSON.parse(input);
} catch (e) {
  // If parsing fails, fail gracefully
  console.error('Invalid JSON input');
  process.exit(1);
}

// Basic input validation / normalization
const depot = Array.isArray(data.depot) && data.depot.length >= 2 ? [Number(data.depot[0]), Number(data.depot[1])] : [0, 0];
const truckCount = Number.isFinite(data.truckCount) && data.truckCount > 0 ? Math.floor(data.truckCount) : 1;
const packages = Array.isArray(data.packages) ? data.packages.map(p => ({
  id: p && p.id !== undefined ? p.id : null,
  x: Number(p && p.x),
  y: Number(p && p.y)
})) : [];

// Filter out invalid packages
const validPackages = packages.filter(p => p && p.id !== null && Number.isFinite(p.x) && Number.isFinite(p.y));

// If there are no packages, return empty routes
if (validPackages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// Helper: Euclidean distance between two points
function dist(a, b) {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.hypot(dx, dy);
}

// Prepare structures
// We'll keep a list of unassigned package objects {id, x, y}
let unassigned = validPackages.map(p => ({ id: p.id, x: p.x, y: p.y, distToDepot: dist([p.x, p.y], depot) }));

// Routes output: array of routes, each a list of package IDs in visitation order
const routes = [];

// If there are more trucks than packages, we still won't create empty routes.
// We'll create up to truckCount routes, distributing packages as evenly as possible.
for (let r = 0; r < truckCount; r++) {
  if (unassigned.length === 0) break;

  // Remaining routes and packages
  const remainingRoutes = truckCount - r;
  const remainingPackages = unassigned.length;

  // Basic load balancing: max packages this route should handle
  // Ensure at least 1 per remaining route, but cap by remaining packages
  const maxPerRoute = Math.max(1, Math.ceil(remainingPackages / remainingRoutes));

  // Build a route by greedy nearest-neighbor from the depot
  // Start from depot
  let currentPoint = depot.slice(); // [x, y]
  const route = [];

  // We will greedily pick up to maxPerRoute packages for this route
  for (let taken = 0; taken < maxPerRoute; ) {
    if (unassigned.length === 0) break;

    // Find the nearest unassigned package to currentPoint
    let bestIdx = -1;
    let bestD = Infinity;

    for (let i = 0; i < unassigned.length; i++) {
      const p = unassigned[i];
      const d = dist(currentPoint, [p.x, p.y]);
      if (d < bestD) {
        bestD = d;
        bestIdx = i;
      }
    }

    if (bestIdx === -1) break; // should not happen

    const chosen = unassigned.splice(bestIdx, 1)[0];
    route.push(chosen.id);
    currentPoint = [chosen.x, chosen.y];
    taken++;
  }

  // If a route ended up empty (shouldn't happen due to maxPerRoute >= 1), skip
  if (route.length > 0) {
    routes.push(route);
  }
}

// If there are still unassigned packages (due to rounding or edge cases), assign them to the last route
if (unassigned.length > 0) {
  // If there was at least one route already, append remaining to the last route
  // Otherwise, create a new route with all remaining packages
  const remainingIds = unassigned.map(p => p.id);
  if (routes.length > 0) {
    routes[routes.length - 1] = routes[routes.length - 1].concat(remainingIds);
  } else {
    routes.push(remainingIds);
  }
  unassigned = [];
}

// Final output: ensure every package appears exactly once
// (We can optionally validate, but per spec, we just output routes)
console.log(JSON.stringify({ routes }));
```
