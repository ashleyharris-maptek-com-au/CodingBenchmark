; SPIR-V Assembly for a passthrough Geometry Shader (Triangles -> TriangleStrip with 3 vertices)
; Vulkan 1.0 compatible
; Identity / passthrough of input triangle

               OpCapability Geometry
               OpMemoryModel Logical GLSL450

; Types
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%vec3 = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2

; gl_PerVertex_in: input per-vertex gl_Position
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input vertex shader user inputs (arrays)
; layout(location = 0) in vec3 worldPos[];
; layout(location = 1) in vec3 normal[];
; layout(location = 2) in vec3 color[];
%vec3 = OpTypeVector %float 3

%arr_worldPos = OpTypeArray %vec3 %uint_3
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos_in = OpVariable %ptr_in_arr_worldPos Input

%arr_normal = OpTypeArray %vec3 %uint_3
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal_in = OpVariable %ptr_in_arr_normal Input

%arr_color = OpTypeArray %vec3 %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color_in = OpVariable %ptr_in_arr_color Input

; gl_PerVertex_out: output per-vertex gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_color = OpTypePointer Output %vec3
%outColor = OpVariable %ptr_out_color Output

; Access helpers for per-vertex gl_Position in gl_in
%ptr_in_v4 = OpTypePointer Input %v4float
; Access to gl_in[i].gl_Position: chain as gl_in (index i) then member 0
; We'll use OpAccessChain with indices: i, 0

; Accessors for color[i]
%ptr_in_color = OpTypePointer Input %vec3
; Access color_in[i]

; Entry point and interface
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor
; Declare interface: all input/output variables used by the shader
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Function
%void = OpTypeVoid
%func = OpTypeFunction %void

; We will implement a simple passthrough for the 3 input vertices of the input triangle.
; Helper constants for indices
%idx0 = OpConstant %uint 0
%idx1 = OpConstant %uint 1
%idx2 = OpConstant %uint 2

;---------------------------------------------------------------------------
; Function %main
%main = OpFunction %void None %voidfn
%entry = OpLabel

; Vertex 0
; gl_in[0].gl_Position
%gl_in_0_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %idx0 %%zero? ; placeholder
; The above placeholder will be replaced with concrete indices in the final code
; However, SPIR-V assembly requires exact chain assembly. We'll construct correctly:

; For clarity, define explicit access chains using correct indices:

; gl_in[0].gl_Position
%gl_in_0_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %idx0 %%zero?
; The precise chain uses member index 0 for Position, which is the first (and only) member
; Since gl_in is array of gl_PerVertex_in, index 0 selects the vertex, then 0 selects Position.

; But we cannot include placeholders. We'll implement explicit chains with numeric literals via OpConstant.

; Re-define proper chains:

%gl_in_0_pos_ptr0 = OpAccessChain %ptr_in_v4 %gl_in %idx0 %idx0
%gl_in_0_pos = OpLoad %v4float %gl_in_0_pos_ptr0

; worldPos[0] for potential use (not required, but available)
; color[0]
%gl_color_0_ptr = OpAccessChain %ptr_in_color %color_in %idx0
%gl_color_0 = OpLoad %vec3 %gl_color_0_ptr

; Write gl_Position for vertex 0
%gl_out_0_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %idx0
%gl_out_0_pos = OpAccessChain %ptr_out_v4 %gl_out_0_pos_ptr
OpStore %gl_out_0_pos_ptr %gl_in_0_pos

; Write color for vertex 0
%gl_out_0_color_ptr = OpAccessChain %ptr_out_color %gl_out %idx0
OpStore %gl_out_0_color_ptr %gl_color_0

; Emit vertex 0
OpEmitVertex

; Vertex 1
%gl_in_1_pos_ptr0 = OpAccessChain %ptr_in_v4 %gl_in %idx1 %idx0
%gl_in_1_pos = OpLoad %v4float %gl_in_1_pos_ptr0

%gl_color_1_ptr = OpAccessChain %ptr_in_color %color_in %idx1
%gl_color_1 = OpLoad %vec3 %gl_color_1_ptr

%gl_out_1_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %idx1
; We need to store to gl_Position
OpStore %gl_out_1_pos_ptr %gl_in_1_pos

%gl_out_1_color_ptr = OpAccessChain %ptr_out_color %gl_out %idx1
OpStore %gl_out_1_color_ptr %gl_color_1

OpEmitVertex

; Vertex 2
%gl_in_2_pos_ptr0 = OpAccessChain %ptr_in_v4 %gl_in %idx2 %idx0
%gl_in_2_pos = OpLoad %v4float %gl_in_2_pos_ptr0

%gl_color_2_ptr = OpAccessChain %ptr_in_color %color_in %idx2
%gl_color_2 = OpLoad %vec3 %gl_color_2_ptr

%gl_out_2_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %idx2
OpStore %gl_out_2_pos_ptr %gl_in_2_pos

%gl_out_2_color_ptr = OpAccessChain %ptr_out_color %gl_out %idx2
OpStore %gl_out_2_color_ptr %gl_color_2

OpEmitVertex

; End primitive
OpEndPrimitive

OpLabel %voidfn
OpReturn

OpFunctionEnd

; Note: The above is a schematic outline. In actual SPIR-V assembly, exact OpAccessChain chains must be constructed with defined pointers and proper indices. The key behavior is:
; - For i in {0,1,2}: read gl_in[i].gl_Position, read color[i], write to gl_out.gl_Position and outColor, then EmitVertex.
; - After three vertices, EndPrimitive. The geometry shader outputs 3 vertices, producing a 3-vertex triangle strip identical to input.
