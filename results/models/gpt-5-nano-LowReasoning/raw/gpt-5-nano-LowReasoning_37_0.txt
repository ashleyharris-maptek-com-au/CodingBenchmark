```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    // Global graph data
    static int n;
    static List<int>[] adj;

    // For cycle detection with a given removed set
    static bool[] removed;
    static int[] color;     // 0 = unvisited, 1 = in stack, 2 = done
    static int[] parent;

    // Cycle found helper
    static List<int> cycleBuffer;

    // Solution collection
    static List<int> solution;  // removed vertices (order is not important)

    // Detect a cycle in the graph after removing vertices marked in 'removed'
    // If a cycle exists, fills 'cycleBuffer' with vertices on one cycle and returns true
    // Otherwise returns false
    static bool FindCycle(out List<int> foundCycle)
    {
        color = new int[n];
        parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;
        cycleBuffer = null;
        for (int u = 0; u < n; u++)
        {
            if (removed[u]) continue;
            if (color[u] != 0) continue;
            if (DFSVisit(u))
            {
                foundCycle = new List<int>(cycleBuffer);
                return true;
            }
        }
        foundCycle = null;
        return false;
    }

    // DFS helper for cycle detection
    static bool DFSVisit(int u)
    {
        color[u] = 1;
        foreach (var v in adj[u])
        {
            if (removed[v]) continue;
            if (color[v] == 0)
            {
                parent[v] = u;
                if (DFSVisit(v)) return true;
            }
            else if (color[v] == 1)
            {
                // Found a cycle: backtrack from u to v using parent[]
                cycleBuffer = new List<int>();
                int w = u;
                cycleBuffer.Add(w);
                while (w != v)
                {
                    w = parent[w];
                    cycleBuffer.Add(w);
                }
                // We have a cycle in reverse-ish order; that's fine
                return true;
            }
        }
        color[u] = 2;
        return false;
    }

    // Depth-limited search to find a feedback vertex set of size <= limit
    // Returns true if a removal set of size exactly used <= limit exists
    // On success, 'solution' contains the removed vertices (in any order; will be reversed in output)
    static bool DfsWithLimit(int limit, int used)
    {
        // Quick check: if current graph (with 'removed') is acyclic, success
        if (!FindCycle(out List<int> cyc))
        {
            return true;
        }
        // If no removals left, fail
        if (used == limit) return false;

        // Branch on each vertex in the found cycle
        foreach (int v in cyc)
        {
            if (removed[v]) continue;
            removed[v] = true;
            if (DfsWithLimit(limit, used + 1))
            {
                // Record this vertex as part of solution
                solution.Add(v);
                return true;
            }
            removed[v] = false;
        }
        return false;
    }

    // Fast input reader
    struct FastScanner
    {
        private readonly TextReader inStream;
        private readonly char[] buffer;
        private int idx, len;
        public FastScanner(TextReader input)
        {
            inStream = input;
            buffer = new char[1 << 16];
            idx = 0; len = 0;
        }
        private int ReadChar()
        {
            if (idx >= len)
            {
                len = inStream.Read(buffer, 0, buffer.Length);
                idx = 0;
                if (len <= 0) return -1;
            }
            return buffer[idx++];
        }
        public int NextInt()
        {
            int c, sign = 1, val = 0;
            do { c = ReadChar(); } while (c <= ' ' && c != -1);
            if (c == '-')
            {
                sign = -1;
                c = ReadChar();
            }
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadChar();
            }
            return val * sign;
        }
    }

    static void Main()
    {
        FastScanner fs = new FastScanner(Console.In);

        n = fs.NextInt();
        int m = fs.NextInt();

        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u < 0 || u >= n || v < 0 || v >= n) continue; // guard
            adj[u].Add(v);
        }

        // Special case: no vertices
        if (n == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        // Try increasing k from 0 to n
        removed = new bool[n];
        solution = new List<int>();

        int answerK = -1;
        for (int k = 0; k <= n; k++)
        {
            // reset removal state
            for (int i = 0; i < n; i++) removed[i] = false;
            solution.Clear();

            if (DfsWithLimit(k, 0))
            {
                // solution now contains the vertices used in successful path, in reverse order of addition
                // It may contain only k elements; ensure size is k
                // If needed, trim or pad
                // It should be exactly k
                answerK = k;
                break;
            }
        }

        if (answerK == -1)
        {
            // Fallback (shouldn't happen): remove all vertices
            answerK = n;
            Console.WriteLine(answerK);
            StringBuilder sbAll = new StringBuilder();
            for (int i = 0; i < n; i++) sbAll.Append(i).Append(' ');
            Console.WriteLine(sbAll.ToString().TrimEnd());
            return;
        }

        // Prepare output: solution contains the removed vertices for the found k
        // It may be shorter/longer due to recursion order; ensure exactly k distinct indices
        // If for some reason size != k, adjust by adding remaining not-in-removed
        // But we know during success we added exactly k vertices to 'solution'
        // and that is our set (order arbitrary)
        // Print
        Console.WriteLine(answerK);
        // The 'solution' contains the vertices in reverse order of addition along the successful path
        // We can output them in any order. Let's output in the order they appear, but ensure exactly k items.
        // If the solution size is not exactly k due to edge cases, construct a set to satisfy size.
        int[] outSet;
        if (solution.Count == answerK)
        {
            outSet = solution.ToArray();
        }
        else
        {
            // Fill from solution plus any remaining not removed
            HashSet<int> seen = new HashSet<int>(solution);
            List<int> tmp = new List<int>(solution);
            for (int i = 0; i < n && tmp.Count < answerK; i++)
            {
                if (!seen.Contains(i))
                {
                    tmp.Add(i);
                    seen.Add(i);
                }
            }
            outSet = tmp.ToArray();
        }

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < outSet.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(outSet[i]);
        }
        if (outSet.Length > 0)
            Console.WriteLine(sb.ToString());
        else
            Console.WriteLine();
    }
}
```
