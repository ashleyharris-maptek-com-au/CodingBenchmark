```cpp
#include <bits/stdc++.h>
using namespace std;

/*
  Job-Shop Scheduling Problem (JSSP) solver
  - Reads:
      J M
      For each job j in [0..J-1]:
        T_j
        For t in [0..T_j-1]:
          machine (int, 0..M-1)  duration (int)
  - Outputs per job j (one line):
      start_0 machine_0 start_1 machine_1 ... start_{T_j-1} machine_{T_j-1}
      (each pair corresponds to a task in order)
  - Constraints:
      - Each task runs on its specified machine
      - Tasks within a job execute in order (i before i+1)
      - A machine can handle at most one task at a time
      - Tasks are non-preemptive
  - Strategy:
      - If total number of tasks is small (<= 12), use exact Branch-and-Bound (DFS) to optimality.
      - Otherwise, use a greedy, adaptive heuristic (list scheduling) for larger instances.

  Notes:
  - This is a self-contained C++17/20 compatible solution.
  - The exact solver uses simple pruning by current makespan bound.
  - The heuristic assigns ready tasks in a greedy fashion prioritizing earliest possible start.
*/

struct Task {
    int machine;
    long long duration;
};

struct Candidate {
    int job;      // which job this task belongs to
    int index;    // task index within job
    long long ready_time; // last finish time of this job
    int machine;
    long long duration;
};

int J, M;
vector<vector<Task>> jobs;      // per job: list of tasks
vector<int> next_idx;             // next task index to schedule per job
vector<long long> avail;            // machine availability times
vector<long long> last_finish;      // last finish time per job
vector<vector<long long>> start_times; // per job, per task start time (to be filled)
int total_tasks = 0;

// Best solution found (for exact solver)
long long best_makespan = LLONG_MAX;
vector<vector<long long>> best_start_times;

// Utility: compute current makespan (lower bound for pruning)
long long current_makespan(const vector<long long>& avail_vec, const vector<long long>& last_fin) {
    long long mb = 0;
    for (long long t : avail_vec) mb = max(mb, t);
    for (long long t : last_fin) mb = max(mb, t);
    return mb;
}

// Exact solver: Branch-and-Bound DFS
void dfs_exact(int scheduled_count) {
    if (scheduled_count == total_tasks) {
        // All tasks scheduled; compute makespan
        long long makespan = current_makespan(avail, last_finish);
        if (makespan < best_makespan) {
            best_makespan = makespan;
            best_start_times = start_times;
        }
        return;
    }

    // Lower bound pruning
    long long lb = current_makespan(avail, last_finish);
    if (lb >= best_makespan) return;

    // Build list of ready tasks (one per job that still has tasks)
    vector<Candidate> ready;
    ready.reserve(J);
    for (int j = 0; j < J; ++j) {
        int idx = next_idx[j];
        if (idx < (int)jobs[j].size()) {
            const Task& tk = jobs[j][idx];
            Candidate c;
            c.job = j;
            c.index = idx;
            c.ready_time = last_finish[j];
            c.machine = tk.machine;
            c.duration = tk.duration;
            ready.push_back(c);
        }
    }

    if (ready.empty()) return; // should not happen if total_tasks tracked correctly

    // Heuristic: try candidates in order of increasing earliest start
    sort(ready.begin(), ready.end(), [&](const Candidate& a, const Candidate& b){
        long long sa = max(avail[a.machine], a.ready_time);
        long long sb = max(avail[b.machine], b.ready_time);
        if (sa != sb) return sa < sb;
        // tie-breaker: shorter duration first
        if (a.duration != b.duration) return a.duration < b.duration;
        return a.job < b.job;
    });

    // Branch on each ready candidate
    for (const auto& c : ready) {
        int j = c.job;
        int idx = c.index;
        int m = c.machine;
        long long dur = c.duration;
        long long start = max(avail[m], last_finish[j]);
        long long finish = start + dur;

        // Save state
        long long prev_avail = avail[m];
        long long prev_last = last_finish[j];
        auto& st = start_times[j];
        long long prev_start = st[idx];

        // Apply
        avail[m] = finish;
        last_finish[j] = finish;
        st[idx] = start;
        next_idx[j] = idx + 1;

        dfs_exact(scheduled_count + 1);

        // Restore
        avail[m] = prev_avail;
        last_finish[j] = prev_last;
        st[idx] = prev_start;
        next_idx[j] = idx;
    }
}

// Heuristic solver (large instances)
void solve_heuristic(vector<vector<long long>>& out_start) {
    // Reset
    avail.assign(M, 0);
    last_finish.assign(J, 0);
    start_times.assign(J, vector<long long>());

    for (int j = 0; j < J; ++j) {
        start_times[j].assign(jobs[j].size(), -1);
    }

    // Keep track of how many tasks scheduled per job
    vector<int> idx(J, 0);

    int tasks_left = total_tasks;
    while (tasks_left > 0) {
        // Build ready candidates
        vector<Candidate> ready;
        for (int j = 0; j < J; ++j) {
            if (idx[j] < (int)jobs[j].size()) {
                const Task& t = jobs[j][idx[j]];
                Candidate c{j, idx[j], last_finish[j], t.machine, t.duration};
                ready.push_back(c);
            }
        }
        if (ready.empty()) break; // should not happen

        // Pick candidate with earliest possible start
        int best_i = 0;
        long long best_start = LLONG_MAX;
        for (int i = 0; i < (int)ready.size(); ++i) {
            long long s = max(avail[ready[i].machine], ready[i].ready_time);
            if (s < best_start) {
                best_start = s;
                best_i = i;
            }
        }

        auto c = ready[best_i];
        int j = c.job;
        int m = c.machine;
        long long start = max(avail[m], last_finish[j]);
        long long finish = start + c.duration;

        // Apply
        avail[m] = finish;
        last_finish[j] = finish;
        start_times[j][idx[j]] = start;
        idx[j]++;

        tasks_left--;
    }

    out_start = start_times;
}

// Entry
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> J >> M)) {
        return 0;
    }

    jobs.assign(J, {});
    total_tasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int machine;
            long long duration;
            cin >> machine >> duration;
            jobs[j][t] = {machine, duration};
        }
        total_tasks += T;
    }

    // Adaptive strategy
    if (total_tasks <= 12) {
        // Prepare state
        next_idx.assign(J, 0);
        avail.assign(M, 0);
        last_finish.assign(J, 0);
        start_times.assign(J, {});
        for (int j = 0; j < J; ++j) {
            start_times[j].assign(jobs[j].size(), -1);
        }
        best_makespan = LLONG_MAX;
        best_start_times.clear();
        best_start_times.resize(J);
        for (int j = 0; j < J; ++j) best_start_times[j].resize(jobs[j].size(), -1);

        dfs_exact(0);

        // If something went wrong, fall back to heuristic
        if (best_start_times.empty()) {
            vector<vector<long long>> tmp;
            solve_heuristic(tmp);
            best_start_times = tmp;
        }

        // Output
        for (int j = 0; j < J; ++j) {
            for (size_t t = 0; t < best_start_times[j].size(); ++t) {
                cout << best_start_times[j][t] << ' ' << jobs[j][t].machine;
                if (t + 1 != best_start_times[j].size()) cout << ' ';
            }
            if (j + 1 != J) cout << "\n";
        }
    } else {
        // Large instance: heuristic only
        vector<vector<long long>> out;
        solve_heuristic(out);
        // Output
        for (int j = 0; j < J; ++j) {
            for (size_t t = 0; t < out[j].size(); ++t) {
                cout << out[j][t] << ' ' << jobs[j][t].machine;
                if (t + 1 != out[j].size()) cout << ' ';
            }
            if (j + 1 != J) cout << "\n";
        }
    }

    return 0;
}
```

Explanation of key parts:
- Data structures store the JSSP instance: per-job task list with machine and duration.
- Exact solver (Branch-and-Bound) is implemented via dfs_exact:
  - Maintains per-machine availability (avail), per-job last finish time (last_finish), and per-task start times (start_times).
  - At each step, collects all ready tasks (the next task of each job) and branches by scheduling any of them next.
  - Prunes branches whose current lower bound (max of machine availabilities and job finish times) is already not better than the best found so far.
  - Tracks the best full schedule (start times) when all tasks are scheduled.
- Large instances use a simple greedy heuristic (solve_heuristic):
  - Repeatedly schedules any ready task that can start earliest given machine availability and job progress.
  - Keeps track of start times per task and outputs them.
- Adaptive logic:
  - If total number of tasks is small (<= 12), try to find optimal schedule with exact solver.
  - Otherwise, use the heuristic for scalability.

Compilation:
- This code is standard C++17/20. Compile with MSVC or any modern C++ compiler:
  - cl /EHsc / std:c++20 yourfile.cpp
  - Or g++ -std=c++20 yourfile.cpp

Usage:
- Read from stdin as described.
- Outputs for each job a line with T pairs: start_time machine for each task in order.
