float4 main(PSInput input) : SV_Target0
{
    // Normalize normal (in world space)
    float3 N = normalize(input.normal);

    // View vector from fragment to camera in world space
    float3 V = normalize((cameraPos.xyz) - input.worldPos);

    // N dot V clamped to [0, 1]
    float NdotV = max(dot(N, V), 0.0);

    float3 color;

    if (NdotV < 0.5)
    {
        // blue-green heatmap ramp from red to yellow
        color = float3(1.0, 2.0 * NdotV, 0.0);
    }
    else
    {
        // red-green ramp with blue rising towards white as NdotV approaches 1
        float t = NdotV - 0.5;
        color = float3(1.0 - t * 2.0, 1.0 - t * 2.0, t * 2.0);
    }

    return float4(color, 1.0);
}
