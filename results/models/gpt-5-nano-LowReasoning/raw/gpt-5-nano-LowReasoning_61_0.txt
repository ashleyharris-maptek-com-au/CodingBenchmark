// Node.js TypeScript program for multi-vehicle routing benchmark.
// Strategy: simple deterministic heuristic suitable for large inputs without external deps.
//  - Distribute packages to trucks in round-robin fashion
//  - For each truck, order its assigned packages using a greedy nearest-neighbor starting from the depot
//  - Each route starts and ends at the depot
//  - Output: {"routes": [[packageId, packageId, ...], ...]}

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Point = [number, number];
type Package = { id: string | number; x: number; y: number };
type Depot = { x: number; y: number };

function euclidean(a: Point, b: Point): number {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.hypot(dx, dy);
}

// Safely extract fields with basic validation (keeps deterministic behavior)
function getDepot(obj: any): Depot {
  const d = obj.depot;
  if (!Array.isArray(d) || d.length !== 2) {
    // Fallback to origin if invalid
    return { x: 0, y: 0 };
  }
  return { x: Number(d[0]), y: Number(d[1]) };
}

function toPackages(arr: any[]): Package[] {
  const pkgs: Package[] = [];
  if (!Array.isArray(arr)) return pkgs;
  for (const p of arr) {
    if (p && (typeof p.id !== 'undefined') && typeof p.x === 'number' && typeof p.y === 'number') {
      pkgs.push({ id: p.id, x: p.x, y: p.y });
    }
  }
  return pkgs;
}

function planRoutes(inputData: any): string[] {
  const depot = getDepot(inputData);
  const truckCount = Number(inputData.truckCount) || 1;
  const packages: Package[] = toPackages(inputData.packages || []);

  // If there are fewer trucks than packages, still create up to truckCount routes.
  // Ensure truckCount is at least 1.
  const k = Math.max(1, truckCount);

  // Deterministic ordering: sort by id string representation to ensure reproducibility
  const sortedPkgs = packages.sort((a, b) => {
    const sa = String(a.id);
    const sb = String(b.id);
    if (sa < sb) return -1;
    if (sa > sb) return 1;
    // tie-break by coordinates
    if (a.x !== b.x) return a.x - b.x;
    if (a.y !== b.y) return a.y - b.y;
    return 0;
  });

  // Distribute packages to trucks in round-robin fashion
  const routesPkgLists: Package[][] = Array.from({ length: k }, () => []);
  for (let i = 0; i < sortedPkgs.length; i++) {
    routesPkgLists[i % k].push(sortedPkgs[i]);
  }

  // For each route, order its packages with a greedy nearest-neighbor heuristic
  // Route format to output: array of package ids in visiting order
  const routes: (Array<string | number>)[] = [];

  for (let r = 0; r < k; r++) {
    const assigned = routesPkgLists[r];
    const ordered: string[] = [];

    // If no packages assigned, keep empty route
    if (assigned.length === 0) {
      routes.push(ordered);
      continue;
    }

    // Greedy NN
    let current: Point = [depot.x, depot.y];
    // Work on a copy to mutate
    const remaining: Package[] = assigned.slice();
    while (remaining.length > 0) {
      let bestIdx = 0;
      let bestDist = Number.POSITIVE_INFINITY;
      for (let i = 0; i < remaining.length; i++) {
        const p = remaining[i];
        const d = Math.hypot(current[0] - p.x, current[1] - p.y);
        if (d < bestDist) {
          bestDist = d;
          bestIdx = i;
        }
      }
      const chosen = remaining.splice(bestIdx, 1)[0];
      ordered.push(chosen.id);
      current = [chosen.x, chosen.y];
    }

    routes.push(ordered);
  }

  // Output the JSON as per required format
  // Note: The problem expects a single JSON document on stdout.
  const out = { routes };
  // Serialize without indentation to keep output compact
  return JSON.stringify(out);
}

// Run planning and print result
const outputJson = planRoutes(data);
process.stdout.write(outputJson);
