using System;
using System.IO;
using System.Collections.Generic;

class Program
{
    // Simple fast token reader using splitting (sufficient for typical contest inputs)
    static class FastInput
    {
        public static long NextLong(ref string[] toks, ref int idx, ref string[] pool)
        {
            while (idx >= toks.Length)
            {
                var s = Console.In.ReadToEnd();
                if (string.IsNullOrEmpty(s)) return 0;
                toks = s.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                idx = 0;
            }
            long v = long.Parse(toks[idx++]);
            return v;
        }

        // However, to keep things straightforward and robust for various environments,
        // we'll implement a broader NextInt using a one-shot read.
        public static int NextInt(ref string[] toks, ref int idx)
        {
            while (idx >= toks.Length)
            {
                var s = Console.In.ReadToEnd();
                if (string.IsNullOrEmpty(s)) return 0;
                toks = s.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                idx = 0;
            }
            int v = int.Parse(toks[idx++]);
            return v;
        }
    }

    // Simple integer box with index and dimensions
    struct Box
    {
        public int index;
        public int w, h, d;
        public long volume => (long)w * h * d;
    }

    static void Main()
    {
        // Read all input from stdin robustly
        string all;
        using (var r = Console.OpenStandardInput())
        {
            using (var ms = new MemoryStream())
            {
                r.CopyTo(ms);
                ms.Position = 0;
                using (var sr = new StreamReader(ms))
                {
                    all = sr.ReadToEnd();
                }
            }
        }

        if (string.IsNullOrWhiteSpace(all))
            return;

        var toks = all.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int pos = 0;

        int N = int.Parse(toks[pos++]);
        int W = int.Parse(toks[pos++]);
        int H = int.Parse(toks[pos++]);
        int D = int.Parse(toks[pos++]);

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            int w = int.Parse(toks[pos++]);
            int h = int.Parse(toks[pos++]);
            int d = int.Parse(toks[pos++]);
            boxes.Add(new Box { index = i, w = w, h = h, d = d });
        }

        // Prepare rotations (6 permutations)
        int[][] perms = new int[][]
        {
            new int[]{0,1,2},
            new int[]{0,2,1},
            new int[]{1,0,2},
            new int[]{1,2,0},
            new int[]{2,0,1},
            new int[]{2,1,0}
        };

        // Sort boxes by volume descending to attempt packing larger first
        boxes.Sort((a,b) => b.volume.CompareTo(a.volume));

        // Occupancy grid: W x H x D
        // Flatten index: x + W*(y + H*z)
        long totalVolume = (long)W * H * D;
        // If total cells exceed int, ensure cast safety
        int totalCells = (int)totalVolume;

        var occupancy = new byte[totalCells]; // 0 = empty, 1 = filled

        // List of placed boxes: (index, x, y, z)
        var placed = new List<(int idx, int x, int y, int z)>();

        // Helper to check fit for a given position and rotated dims
        bool CanPlace(int x, int y, int z, int rw, int rh, int rd)
        {
            if (x < 0 || y < 0 || z < 0) return false;
            if (x + rw > W || y + rh > H || z + rd > D) return false;

            // Iterate cells inside the box volume
            // Flattened index: fx + W*(fy + H*fz)
            for (int zz = 0; zz < rd; zz++)
            {
                int baseZ = z + zz;
                for (int yy = 0; yy < rh; yy++)
                {
                    int baseY = y + yy;
                    int planeBase = baseZ * (W * H) + baseY * W + x;
                    // Check contiguous rw cells
                    for (int xx = 0; xx < rw; xx++)
                    {
                        int idx = planeBase + xx;
                        if (occupancy[idx] != 0)
                            return false;
                    }
                }
            }
            return true;
        }

        void Place(int x, int y, int z, int rw, int rh, int rd)
        {
            for (int zz = 0; zz < rd; zz++)
            {
                int baseZ = z + zz;
                for (int yy = 0; yy < rh; yy++)
                {
                    int baseY = y + yy;
                    int planeBase = baseZ * (W * H) + baseY * W + x;
                    for (int xx = 0; xx < rw; xx++)
                    {
                        int idx = planeBase + xx;
                        occupancy[idx] = 1;
                    }
                }
            }
        }

        // For performance in large scenarios, a simple but scalable heuristic:
        // Try to place each box in first-fit position scanning y from 0..H-1, z from 0..D-1, x from 0..W-1
        // with all 6 rotations, choose first fit maximizing y (low gravity) or simply first found.
        foreach (var b in boxes)
        {
            bool placedAny = false;

            // Try 6 rotations
            foreach (var p in perms)
            {
                int rw = GetDim(b, p[0]);
                int rh = GetDim(b, p[1]);
                int rd = GetDim(b, p[2]);

                // Quick skip if doesn't fit in container in any dimension
                if (rw > W || rh > H || rd > D) continue;

                // Brute-force search for a position
                // We try to place as low as possible in y, then x, then z
                for (int y = 0; y <= H - rh; y++)
                {
                    for (int z = 0; z <= D - rd; z++)
                    {
                        // We can optimize by scanning x inner
                        for (int x = 0; x <= W - rw; x++)
                        {
                            if (CanPlace(x, y, z, rw, rh, rd))
                            {
                                Place(x, y, z, rw, rh, rd);
                                placed.Add((b.index, x, y, z));
                                placedAny = true;
                                break;
                            }
                        }
                        if (placedAny) break;
                    }
                    if (placedAny) break;
                }

                if (placedAny) break;
            }
        }

        // Output
        var outLines = new List<string>();
        outLines.Add(placed.Count.ToString());
        foreach (var item in placed)
        {
            outLines.Add($"{item.idx} {item.x} {item.y} {item.z}");
        }

        Console.WriteLine(string.Join(Environment.NewLine, outLines));
    }

    // Helper to map rotation index to dimension
    static int GetDim(Box b, int which)
    {
        // which: 0 -> w, 1 -> h, 2 -> d, but after rotation we use indices:
        // We encoded permutations by indices into [w,h,d] fields
        // So mapping:
        // 0 => w
        // 1 => h
        // 2 => d
        // However, since we pass only the dimension by using b.w / b.h / b.d, we translate:
        switch (which)
        {
            case 0: return b.w;
            case 1: return b.h;
            case 2: return b.d;
            default: return b.w;
        }
    }
}
