    class Edge
    {
        public int u, v;
        public long w;
        public Edge(int u, int v, long w)
        {
            this.u = u;
            this.v = v;
            this.w = w;
        }
    }

    // For Dijkstra adjacency: neighbor and edge id
    class AdjItem
    {
        public int to;
        public long w;
        public int id;
        public AdjItem(int to, long w, int id)
        {
            this.to = to;
            this.w = w;
            this.id = id;
        }
    }

    static void Main()

        var ri = new System.IO.StreamReader(Console.OpenStandardInput());
        string line = ri.ReadLine();
        if (string.IsNullOrWhiteSpace(line))
            return;
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        // Graph storage
        List<Edge> edges = new List<Edge>();
        List<List<AdjItem>> adj = new List<List<AdjItem>>();
        for (int i = 0; i < n; i++) adj.Add(new List<AdjItem>());

        for (int i = 0; i < m; i++)
        {
            line = ri.ReadLine();
            while (string.IsNullOrWhiteSpace(line)) line = ri.ReadLine();
            var p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(p[0]);
            int v = int.Parse(p[1]);
            long w = long.Parse(p[2]);
            Edge e = new Edge(u, v, w);
            edges.Add(e);
            adj[u].Add(new AdjItem(v, w, i));
            adj[v].Add(new AdjItem(u, w, i));
        }

        // Compute degree parity
        List<int> oddNodes = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (adj[i].Count % 2 == 1)
                oddNodes.Add(i);
        }

        // If no odd nodes, just Euler tour on original graph
        // Else need to pair odd nodes by duplicating shortest paths
        // Precompute all-pairs shortest paths among odd nodes using Dijkstra
        int k = oddNodes.Count;
        long[,] distOdd = new long[Math.Max(0, k), Math.Max(0, k)];

        Dictionary<(int, int), List<int>> pathEdgesBetween = new Dictionary<(int,int), List<int>>();

        // A function to run Dijkstra from source s, returning dist[] and parentEdge[] (edge ids)
        long[] RunDijkstra(int src, out int[] parentEdge, out int[] parentNode)
        {
            long[] dist = new long[n];
            const long INF = long.MaxValue / 4;
            for (int i = 0; i < n; i++) dist[i] = INF;
            parentEdge = new int[n];
            parentNode = new int[n];
            for (int i = 0; i < n; i++) parentEdge[i] = -1, parentNode[i] = -1;

            dist[src] = 0;
            // min-heap: (dist, node)
            var pq = new SimplePriorityQueue<int>();
            pq.Enqueue(src, 0);

            while (pq.Count > 0)
            {
                int u = pq.Dequeue();
                long du = dist[u];
                foreach (var e in adj[u])
                {
                    int v = e.to;
                    long w = e.w;
                    if (dist[v] > du + w)
                    {
                        dist[v] = du + w;
                        parentNode[v] = u;
                        parentEdge[v] = e.id;
                        pq.Enqueue(v, (int)dist[v]);
                    }
                }
            }

            return dist;
        }

        if (k > 0)

            // Precompute from each odd node
            for (int i = 0; i < k; i++)
            {
                int s = oddNodes[i];
                int[] parEdge, parNode;
                long[] dist = RunDijkstra(s, out parEdge, out parNode);
                for (int j = i + 1; j < k; j++)
                {
                    int t = oddNodes[j];
                    distOdd[i, j] = dist[t];
                    distOdd[j, i] = dist[t];
                    // Reconstruct path edges from s to t via parents
                    List<int> path = ReconstructPathEdges(s, t, parEdgeForTarget(parEdge, parNode, t), parNode);
                    // However we don't have direct access to parEdge/parentNode arrays here.
                    // We'll implement a local reconstruction using a helper by re-running Dijkstra with parent tracking.
                }
            }
        }

        // The above attempt to reconstruct paths inside the loop is flawed due to lacking access.
        // We'll implement a robust approach: store, for each source s, the parentEdge and parentNode arrays,
        // and after Dijkstra, for each target t we reconstruct path using those arrays.

        // Redo with proper storage
        pathEdgesBetween.Clear();
        distOdd = new long[Math.Max(0, k), Math.Max(0, k)];
        if (k > 0)
        {
            // For each source, store its parents
            List<int[]> allParentEdge = new List<int[]>();
            List<int[]> allParentNode = new List<int[]>();
            List<long[]> allDist = new List<long[]>();

            for (int i = 0; i < k; i++)
            {
                int s = oddNodes[i];
                int[] pEdge, pNode;
                long[] dist = RunDijkstra(s, out pEdge, out pNode);
                allParentEdge.Add(pEdge);
                allParentNode.Add(pNode);
                allDist.Add(dist.Select(x => x).ToArray());
            }

            for (int i = 0; i < k; i++)
            {
                for (int j = i + 1; j < k; j++)
                {
                    int s = oddNodes[i], t = oddNodes[j];
                    long d = allDist[i][t];
                    distOdd[i, j] = d;
                    distOdd[j, i] = d;

                    // Reconstruct path edges from s to t using parents of source i
                    int[] pEdge = allParentEdge[i];
                    int[] pNode = allParentNode[i];
                    List<int> path = new List<int>();
                    int cur = t;
                    while (cur != s && cur != -1)
                    {
                        int eId = pEdge[cur];
                        if (eId >= 0)

                        cur = pNode[cur];
                    }
                    path.Reverse();
                    pathEdgesBetween[(i, j)] = path;
                    pathEdgesBetween[(j, i)] = new List<int>(path); // symmetric
                }
            }

        // Build the augmented graph after duplications
        List<Edge> augmentedEdges = new List<Edge>(edges);
        // For each original edge, we will have an adjacency; we also need to keep track of duplicates
        // We'll maintain a used array for all edges in augmented graph.
        // We'll collect all adjacency for traversal based on augmentedEdges
        List<List<(int to, int id)>> augAdj = new List<List<(int to, int id)>>();
        for (int i = 0; i < n; i++) augAdj.Add(new List<(int to, int id)>());

            augmentedEdges[i] = new Edge(e.u, e.v, e.w); // copy

        for (int i = 0; i < m; i++)

            augAdj[e.u].Add((e.v, i));
            augAdj[e.v].Add((e.u, i));

        List<(int a, int b)> pairs = new List<(int a, int b)>();
        if (k == 0)

            // nothing
        }
        else if (k <= 16)
        {
            // Exact minimum weight perfect matching via DP
            int full = (1 << k) - 1;
            long[] dp = new long[1 << k];
            int[] choice = new int[1 << k];
            for (int mask = 0; mask <= full; mask++) dp[mask] = long.MaxValue / 4;
            dp[0] = 0;
            for (int mask = 1; mask <= full; mask++)
            {
                int bits = CountBits(mask);
                if ((bits & 1) == 1) continue;
                // find first set bit
                int iBit = FirstBit(mask);
                int iNode = iBit;
                int maskWithoutI = mask ^ (1 << iNode);
                // pair i with j
                int rest = maskWithoutI;
                for (int j = iNode + 1; j < k; j++)
                {
                    if ((mask & (1 << j)) != 0)
                    {
                        int newMask = maskWithoutI ^ (1 << j);
                        long cost = distOdd[iNode, j] + dp[newMask];
                        if (cost < dp[mask])
                        {
                            dp[mask] = cost;
                            choice[mask] = j;
                        }
                    }
                }
            }

            // Reconstruct
            int curMask = full;
            while (curMask != 0)
            {
                int iBit = FirstBit(curMask);
                int iNode = iBit;
                int withoutI = curMask ^ (1 << iNode);
                int j = -1;
                // find partner from choice by checking which j produced dp[curMask]
                // We'll simply recompute by trying options
                long best = long.MaxValue / 4;
                for (int t = iNode + 1; t < k; t++)
                {
                    if ((withoutI & (1 << t)) != 0)
                    {
                        int nm = withoutI ^ (1 << t);
                        long cost = distOdd[iNode, t] + dp[nm];
                        if (cost < best && Math.Abs(cost - dp[curMask]) < 1e-9)
                        {
                            best = cost;
                            j = t;
                        }
                    }
                }
                if (j == -1)
                {
                    // Fallback: find any j that matches
                    for (int t = iNode + 1; t < k; t++)
                    {
                        if ((withoutI & (1 << t)) != 0)
                        {
                            j = t;
                            break;
                        }
                    }
                }
                if (j == -1) break; // should not happen
                // record pair (iNode, j)
                pairs.Add((oddNodes[iNode], oddNodes[j]));
                curMask = withoutI ^ (1 << j);
            }
        }
        else
        {
            // Greedy pairing by nearest distance
            bool[] usedOdd = new bool[k];
            while (true)
            {
                int a = -1, b = -1;
                long best = long.MaxValue;
                for (int i = 0; i < k; i++) if (!usedOdd[i])
                    for (int j = i + 1; j < k; j++) if (!usedOdd[j])
                    {
                        if (distOdd[i, j] < best)
                        {
                            best = distOdd[i, j];
                            a = i; b = j;
                        }
                    }
                if (a == -1) break;
                usedOdd[a] = usedOdd[b] = true;
                pairs.Add((oddNodes[a], oddNodes[b]));
            }
        }

        // If we used both indexing from dp/greedy, ensure we have pathEdgesBetween for (i,j)
        // For safety, if not present, we can recompute a simple shortest path between the two odd nodes using Dijkstra again.
        foreach (var pr in pairs)

            int a = oddNodes.IndexOf(pr.a);
            int b = oddNodes.IndexOf(pr.b);
            List<int> path = null;
            if (a >= 0 && b >= 0 && pathEdgesBetween.TryGetValue((a, b), out path))
            {
                // duplicate along this path
            }
            else
            {
                // compute on demand
                // Run Dijkstra from pr.a to pr.b
                int[] pE, pN;
                long[] dist = RunSimpleDijkstra(pr.a, pr.b, out pE, out pN);
                path = new List<int>();
                int cur = pr.b;
                while (cur != pr.a && cur != -1)
                {
                    int eId = pE[cur];
                    if (eId >= 0) path.Add(eId);
                    cur = pN[cur];
                }
                path.Reverse();
            }

            // Duplicate each edge along path
            foreach (int eid in path)
            {
                var e = edges[eid];
                // Create duplicate edge
                Edge de = new Edge(e.u, e.v, e.w);
                augmentedEdges.Add(de);
                int newId = augmentedEdges.Count - 1;
                augAdj[e.u].Add((e.v, newId));
                augAdj[e.v].Add((e.u, newId));
            }
        }

        // Run Hierholzer to get Euler tour starting at 0
        int totalEdges = augmentedEdges.Count;
        bool[] used = new bool[totalEdges];
        // Build adjacency from augmentedEdges is already prepared in augAdj
        // We need per-node an index pointer for efficient traversal
        int[] iters = new int[n];
        Stack<int> st = new Stack<int>();
        List<int> eulerNodes = new List<int>();

        st.Push(0);
        while (st.Count > 0)
        {
            int v = st.Peek();
            // advance to next unused edge
            while (iters[v] < augAdj[v].Count && used[augAdj[v][iters[v]].id])
                iters[v]++;
            if (iters[v] == augAdj[v].Count)
            {
                // no more edges
                eulerNodes.Add(v);
                st.Pop();
            }
            else
            {
                var info = augAdj[v][iters[v]];
                iters[v]++;
                int to = info.to;
                int id = info.id;
                if (used[id]) continue;
                used[id] = true;
                st.Push(to);
            }
        }

        // The produced route is in reverse order in eulerNodes
        eulerNodes.Reverse();

        // Output: nodes sequence, space-separated
        // Optional: prefix with length L equal to number of nodes
        // We'll print only the route for simplicity
        Console.WriteLine(string.Join(" ", eulerNodes));
    }

    // Simple priority queue for Dijkstra (min-heap)
    class SimplePriorityQueue<T> where T : notnull
    {
        private List<(int key, T item)> heap = new List<(int, T)>();
        public int Count => heap.Count;
        public void Enqueue(T item, int priority)
        {
            heap.Add((priority, item));
            SiftUp(heap.Count - 1);
        }
        public T Dequeue()
        {
            var res = heap[0].item;
            var last = heap[heap.Count - 1];
            heap[0] = last;
            heap.RemoveAt(heap.Count - 1);
            SiftDown(0);
            return res;
        }
        private void SiftUp(int idx)
        {
            while (idx > 0)
            {
                int parent = (idx - 1) >> 1;
                if (heap[parent].key <= heap[idx].key) break;
                var tmp = heap[parent];
                heap[parent] = heap[idx];
                heap[idx] = tmp;
                idx = parent;
            }
        }
        private void SiftDown(int idx)
        {
            int n = heap.Count;
            while (true)
            {
                int l = idx * 2 + 1;
                if (l >= n) break;
                int r = l + 1;
                int smallest = (r < n && heap[r].key < heap[l].key) ? r : l;
                if (heap[idx].key <= heap[smallest].key) break;
                var tmp = heap[idx];
                heap[idx] = heap[smallest];
                heap[smallest] = tmp;
                idx = smallest;
            }
        }
    }

    static int CountBits(int x) => System.Numerics.BitOperations.PopCount((uint)x);
    static int FirstBit(int x)
    {
        // return index of least significant 1-bit
        int idx = 0;
        while (((x >> idx) & 1) == 0) idx++;
        return idx;
    }

    // Reconstruct path edges given parent arrays
    static List<int> ReconstructPathEdges(int s, int t, int[] parEdge, int[] parNode)
    {
        List<int> path = new List<int>();
        int cur = t;
        while (cur != s && cur != -1)
        {
            int eId = parEdge[cur];
            if (eId >= 0) path.Add(eId);
            cur = parNode[cur];
        }
        path.Reverse();
        return path;
    }
