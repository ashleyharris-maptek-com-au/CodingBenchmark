; SPIR-V
; Version: 1.0
; Generator: manual
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4

; UBO block
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%UBOPtr = OpTypePointer Uniform %UBO

; Inputs
%inWorldPos = OpVariable %vec3 Input
%inNormal  = OpVariable %vec3 Input
%inUV      = OpVariable %vec2 Input
%inTangent = OpVariable %vec3 Input
%inColor   = OpVariable %vec3 Input

; Output
%outColor = OpVariable %vec4 Output

; Uniform block instance
%ubo = OpVariable %UBOPtr Uniform

; Function types and main
%funcType = OpTypeFunction %void

; Function declaration
%main = OpFunction %void None %funcType
%entryLabel = OpLabel

; Build color from uv: vec4(uv.x, uv.y, 0.0, 1.0)
%uvX = OpCompositeExtract %float %inUV 0
%uvY = OpCompositeExtract %float %inUV 1
%zero  = OpConstant %float 0.0
%one   = OpConstant %float 1.0
%rgb   = OpCompositeConstruct %vec3 %uvX %uvY %zero
%rgba  = OpCompositeConstruct %vec4 %uvX %uvY %zero %one

; Write to output
OpStore %outColor %rgba

OpReturn
OpFunctionEnd

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal     Location 1
OpDecorate %inUV         Location 2
OpDecorate %inTangent    Location 3
OpDecorate %inColor      Location 4

OpDecorate %UBO Block

OpDecorate %ubo Binding 0

OpMemberDecorate %UBO 0 Offset 0

OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192

OpMemberDecorate %UBO 5 Offset 224

; Entry point declaration
OpEntryPoint Fragment %main "main" %inUV %outColor
