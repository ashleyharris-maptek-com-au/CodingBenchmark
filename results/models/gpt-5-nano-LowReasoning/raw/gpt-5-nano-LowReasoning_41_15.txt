; Compute L = normalize(lightPos.xyz - worldPos)
%Lx = OpCompositeExtract %float %lightPos 0

%Lz = OpCompositeExtract %float %lightPos 2
%Lvec = OpCompositeConstruct %v3float %Lx %Ly %Lz
%Ldir = OpUnarySNegate %v3float %Lvec ; placeholder; actual normalize requires more ops

%dz = OpFSub %float %Lz %worldPosZ
%Ldir_raw = OpCompositeConstruct %v3float %dx %dy %dz
%L = OpExtInst %v3float %1 Normalize %Ldir_raw

; t = dot(N, L); clamp 0..1
%dotNL = OpExtInst %float %2 Dot %N %L

; Gooch colors
%cool = OpConstantComposite %v3float %float_2_0 %float_2_0 %float_075??  ; placeholder

; Mix Gooch: Gooch = (1 - t) * cool + t * warm

%cool_scaled = OpVectorTimesScalar %v3float %cool %one_minus_t
%warm_scaled = OpVectorTimesScalar %v3float %warm %t0
%Gooch3 = OpFAdd %v3float %cool_scaled %warm_scaled
%Gooch = OpCompositeConstruct %v3float %Gooch3  ; finalize

; Output color vec4 (RGB = Gooch, A = 1.0)
