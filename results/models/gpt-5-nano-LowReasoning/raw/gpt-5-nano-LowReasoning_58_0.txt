const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
let data;
try {
  data = JSON.parse(input);
} catch (e) {
  console.error('Invalid JSON input');
  process.exit(1);
}

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = grid[0] ? grid[0].length : 0;

// Build helper structures
// Map of robotId -> {x, y}
const robotPos = {};
let maxRobotId = -1;
for (const r of robots) {
  if (typeof r.id === 'number' && typeof r.x === 'number' && typeof r.y === 'number') {
    robotPos[r.id] = { x: r.x, y: r.y };
    if (r.id > maxRobotId) maxRobotId = r.id;
  }
}

// Jobs: map id -> {x, y, priority}
const jobMap = {};
let unassignedJobs = new Set();
for (const j of jobs) {
  if (typeof j.id === 'number' && typeof j.x === 'number' && typeof j.y === 'number') {
    jobMap[j.id] = { x: j.x, y: j.y, priority: j.priority };
    unassignedJobs.add(j.id);
  }
}

// Prepare output structure: an array indexed by robot id. Each element is an array of job ids assigned to that robot.
const assignments = [];
for (let i = 0; i <= maxRobotId; i++) {
  assignments[i] = [];
}

// Helper: BFS from (sx, sy) to compute distance to all cells.
// Returns a flat Int32Array of size W*H with distance, -1 for unreachable or blocked.
function bfsDistances(sx, sy) {
  const dist = new Int32Array(W * H);
  for (let i = 0; i < dist.length; i++) dist[i] = -1;
  const idx = sy * W + sx;
  dist[idx] = 0;

  const qx = new Int16Array(W * H);
  const qy = new Int16Array(W * H);
  let head = 0, tail = 0;
  qx[tail] = sx;
  qy[tail] = sy;
  tail++;

  const inBounds = (x, y) => x >= 0 && x < W && y >= 0 && y < H;
  const isOpen = (x, y) => grid[y][x] !== '#';

  const dirs = [
    [1, 0], [-1, 0], [0, 1], [0, -1]
  ];

  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;
    const d = dist[y * W + x];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (inBounds(nx, ny) && isOpen(nx, ny)) {
        const nidx = ny * W + nx;
        if (dist[nidx] === -1) {
          dist[nidx] = d + 1;
          qx[tail] = nx;
          qy[tail] = ny;
          tail++;
        }
      }
    }
  }

  return dist;
}

// Greedy assignment loop
while (unassignedJobs.size > 0) {
  let bestRobot = null;
  let bestJob = null;
  let bestDist = Number.POSITIVE_INFINITY;

  // For each robot, compute distances and evaluate all unassigned jobs
  for (const [ridStr, pos] of Object.entries(robotPos)) {
    const rid = Number(ridStr);
    // If robot position is invalid or there are no more unassigned jobs, skip
    if (!pos || typeof pos.x !== 'number' || typeof pos.y !== 'number') continue;

    const distGrid = bfsDistances(pos.x, pos.y);

    // Iterate unassigned jobs to find the best
    for (const jobId of unassignedJobs) {
      const j = jobMap[jobId];
      if (!j) continue;
      const d = distGrid[j.y * W + j.x];
      if (d >= 0) {
        // tie-break: prefer smaller distance, then smaller robot id, then smaller job id
        if (
          d < bestDist ||
          (d === bestDist && (rid < bestRobot || bestRobot === null)) ||
          (d === bestDist && rid === bestRobot && jobId < bestJob)
        ) {
          bestDist = d;
          bestRobot = rid;
          bestJob = jobId;
        }
      }
    }
  }

  // If no assignment possible (shouldn't happen in valid inputs), break to avoid infinite loop
  if (bestRobot === null || bestJob === null) {
    // As a fallback, assign any remaining job to the first robot with a valid position
    const anyRobotId = Object.keys(robotPos)[0];
    const anyJobId = unassignedJobs.values().next().value;
    if (anyRobotId !== undefined && anyJobId !== undefined) {
      bestRobot = Number(anyRobotId);
      bestJob = anyJobId;
      bestDist = 0; // unknown, but proceed
    } else {
      // Nothing to do
      break;
    }
  }

  // Perform assignment
  assignments[bestRobot].push(bestJob);

  // Update robot position to job location
  const jb = jobMap[bestJob];
  robotPos[bestRobot] = { x: jb.x, y: jb.y };

  // Remove job from unassigned
  unassignedJobs.delete(bestJob);
}

// Output result
const output = { assignments };
process.stdout.write(JSON.stringify(output));
