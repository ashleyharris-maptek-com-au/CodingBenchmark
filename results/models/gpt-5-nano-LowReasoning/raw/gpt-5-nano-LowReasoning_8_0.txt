use std::cmp::min;
use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.lines();

    // Read first line: H W
    let first = iter.next().unwrap_or("").trim();
    if first.is_empty() {
        println!("0");
        return;
    }
    let mut hw = first.split_whitespace();
    let h: usize = match hw.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => 0,
    };
    let w: usize = match hw.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => 0,
    };

    // Read maze lines
    let mut grid: Vec<u8> = Vec::with_capacity(h * w);
    grid.resize(h * w, b' ');

    let mut start: Option<usize> = None;
    let mut end: Option<usize> = None;

    for r in 0..h {
        let line = iter.next().unwrap_or("");
        // Ensure line has at least w characters; if shorter, pad with spaces
        let mut bytes = line.as_bytes().to_vec();
        if bytes.len() < w {
            bytes.resize(w, b' ');
        }
        for c in 0..w {
            let ch = bytes[c] as char;
            let idx = r * w + c;
            grid[idx] = bytes[c];
            if ch == 'A' {
                start = Some(idx);
            } else if ch == 'B' {
                end = Some(idx);
            }
        }
    }

    // If start or end missing, nothing to do
    let start_idx = match start {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };
    let end_idx = match end {
        Some(e) => e,
        None => {
            println!("0");
            return;
        }
    };

    // BFS to find a path from start to end
    let n = h * w;
    let mut visited = vec![0u8; n];
    // parent stores previous index + 1; 0 means "no parent yet / none"
    let mut parent = vec![0i32; n];

    let mut q = VecDeque::new();
    visited[start_idx] = 1;
    q.push_back(start_idx);

    // Directions: up, down, left, right
    let get_neighbors = |idx: usize, h: usize, w: usize| -> [Option<usize>; 4] {
        let r = idx / w;
        let c = idx % w;
        let mut neigh = [None; 4];
        // up
        if r > 0 {
            neigh[0] = Some((r - 1) * w + c);
        }
        // down
        if r + 1 < h {
            neigh[1] = Some((r + 1) * w + c);
        }
        // left
        if c > 0 {
            neigh[2] = Some(r * w + (c - 1));
        }
        // right
        if c + 1 < w {
            neigh[3] = Some(r * w + (c + 1));
        }
        neigh
    };

    let mut found = false;
    while let Some(cur) = q.pop_front() {
        if cur == end_idx {
            found = true;
            break;
        }
        for nb_opt in get_neighbors(cur, h, w).iter() {
            if let Some(nb) = nb_opt {
                // Skip walls
                let ch = grid[*nb] as char;
                if ch == '#' {
                    continue;
                }
                if visited[*nb] == 0 {
                    visited[*nb] = 1;
                    parent[*nb] = (cur as i32) + 1; // encode parent
                    q.push_back(*nb);
                }
            }
        }
    }

    if !found && start_idx != end_idx {
        // If end wasn't reached, there is no path
        println!("0");
        return;
    }

    // Reconstruct path from end to start using parent
    let mut path: Vec<(usize, usize)> = Vec::new();
    // If start == end, path is just that single cell
    let mut cur = end_idx;
    path.push((cur / w, cur % w));
    while cur != start_idx {
        let p = parent[cur] - 1;
        if p < 0 {
            // No path (shouldn't happen if found)
            println!("0");
            return;
        }
        cur = p as usize;
        path.push((cur / w, cur % w));
        if path.len() > n {
            // safety check to avoid potential infinite loop
            break;
        }
    }
    path.reverse();

    // Output
    println!("{}", path.len());
    for (r, c) in path {
        println!("{} {}", r, c);
    }
}
