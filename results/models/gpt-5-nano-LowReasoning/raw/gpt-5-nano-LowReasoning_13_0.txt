```cpp
#include <bits/stdc++.h>
using namespace std;

/*
  Longest Common Substring among N strings using Suffix Automaton (SAM).

  Approach:
  - Build SAM on the first string S0.
  - For each subsequent string Ti:
      - Walk the automaton to collect, for every state v, the maximum length
        of a substring of Ti that ends at some position and corresponds to v.
      - Propagate those values along suffix links in decreasing order of state length
        to ensure proper maximal matches for all states.
      - Maintain g[v] = min over processed strings of the maximum match length for state v.
  - The answer length L is max_v g[v].
  - Recover the substring from S0 using the state that attains L and the stored first_pos.

  Notes:
  - This solution adapts to very large inputs by using a memory-friendly SAM
    representation (unordered_map per state for transitions).
  - Time is roughly linear in the total length of all strings plus a factor
    for propagation over the number of automaton states.
  - If no common substring exists across all strings, the output is an empty line.
*/

struct SuffixAutomaton {
    struct State {
        int len;                     // length of the longest string represented by this state
        int link;                    // suffix link
        int first_pos;               // position of the end of the first string reaching this state
        unordered_map<char,int> next; // transitions
        State() : len(0), link(-1), first_pos(-1) {}
    };

    vector<State> st;
    int last;

    SuffixAutomaton() { st.reserve(1); st.push_back(State()); last = 0; }

    void reset() {
        st.clear();
        st.push_back(State());
        last = 0;
    }

    void extend(char c, int pos) {
        int cur = (int)st.size();
        st.push_back(State());
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && !hasTransition(p, c)) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]); // copy
                st[clone].len = st[p].len + 1;
                // first_pos can stay as in q
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    bool hasTransition(int v, char c) const {
        auto it = st[v].next.find(c);
        return it != st[v].next.end();
    }

    int nextState(int v, char c) const {
        auto it = st[v].next.find(c);
        if (it == st[v].next.end()) return -1;
        return it->second;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline

    vector<string> S(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, S[i]);
    }

    if (N == 0) {
        cout << "\n";
        return 0;
    }

    // Build SAM on the first string
    const string &S0 = S[0];
    SuffixAutomaton sam;
    for (size_t i = 0; i < S0.size(); ++i) {
        sam.extend(S0[i], (int)i);
    }

    int states = (int)sam.st.size();
    // g[v] = minimal over processed strings of the maximum match length for state v
    vector<int> g(states, 0);
    for (int v = 0; v < states; ++v) g[v] = sam.st[v].len;

    // Precompute order of states by len (descending) for propagation
    vector<int> order(states);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b){
        return sam.st[a].len > sam.st[b].len;
    });

    // Process each subsequent string
    for (int idx = 1; idx < N; ++idx) {
        const string &T = S[idx];
        vector<int> f(states, 0);

        int v = 0;
        int l = 0;

        for (char c : T) {
            if (sam.nextState(v, c) != -1) {
                int u = sam.nextState(v, c);
                v = u;
                ++l;
            } else {
                while (v != -1 && sam.nextState(v, c) == -1) {
                    v = sam.st[v].link;
                }
                if (v == -1) {
                    v = 0;
                    l = 0;
                } else {
                    l = sam.st[v].len + 1;
                    v = sam.nextState(v, c);
                }
            }
            f[v] = max(f[v], l);
        }

        // Propagate along suffix links: ensure parents accumulate from children
        for (int i = 0; i < states; ++i) {
            // nothing
        }
        for (int vi : order) {
            int par = sam.st[vi].link;
            if (par != -1) {
                int propagated = min(f[vi], sam.st[par].len);
                if (propagated > f[par])
                    f[par] = propagated;
            }
        }

        // Update g
        for (int i = 0; i < states; ++i) {
            g[i] = min(g[i], f[i]);
        }
    }

    // Find the maximum length common to all strings
    int bestLen = 0;
    int bestState = -1;
    for (int v = 0; v < states; ++v) {
        if (g[v] > bestLen) {
            bestLen = g[v];
            bestState = v;
        }
    }

    if (bestLen <= 0 || bestState == -1) {
        cout << "\n";
        return 0;
    }

    int endPos = sam.st[bestState].first_pos;
    int startPos = endPos - bestLen + 1;
    if (startPos < 0 || endPos < 0) {
        cout << "\n";
        return 0;
    }

    cout << S0.substr(startPos, bestLen) << "\n";
    return 0;
}
```
