import sys
import math

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None

    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))

    depot_x = float(next(it))
    depot_y = float(next(it))

    xs = []
    ys = []
    demands = []
    for _ in range(N):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        xs.append(x)
        ys.append(y)
        demands.append(d)

    return {
        'N': N,
        'K': K,
        'CAP': CAP,
        'depot': (depot_x, depot_y),
        'coords': list(zip(xs, ys)),
        'demands': demands
    }

def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])

def solve_vrp(cfg):
    N = cfg['N']
    K = cfg['K']
    CAP = cfg['CAP']
    depot = cfg['depot']
    coords = cfg['coords']
    demands = cfg['demands']

    unvisited = set(range(N))
    routes = []

    # Precompute distances from depot to each customer and between customers
    depot_dists = [dist(depot, coords[i]) for i in range(N)]
    between = [[0.0]*N for _ in range(N)]
    for i in range(N):
        for j in range(N):
            if i == j:
                between[i][j] = 0.0
            else:
                between[i][j] = dist(coords[i], coords[j])

    total_distance = 0.0

    # Construct up to K routes using a greedy nearest-feasible heuristic
    for _ in range(K):
        if not unvisited:
            break

        route = []
        load = 0.0
        # If there is nothing we can load, break to avoid infinite loop
        # Start from depot
        current_idx = -1  # -1 means depot
        added_any = False

        while True:
            best_candidate = None
            best_dist = float('inf')

            # Try to pick the nearest feasible unvisited customer
            for i in list(unvisited):
                if load + demands[i] > CAP + 1e-9:
                    continue
                d = depot_dists[i] if current_idx == -1 else between[current_idx][i]
                if d < best_dist - 1e-12:
                    best_dist = d
                    best_candidate = i

            if best_candidate is None:
                break

            # Add to route
            route.append(best_candidate)
            unvisited.remove(best_candidate)
            load += demands[best_candidate]
            current_idx = best_candidate
            added_any = True

            # If full capacity, stop extending this route
            if load > CAP - 1e-9:
                break

        if not added_any:
            # Cannot add any customer to this route (e.g., all remaining exceed capacity from depot)
            break

        routes.append(route)

    # If there are still unvisited customers, try to assign them to the last route if possible
    if unvisited:
        # Try to append remaining customers to the last route while respecting capacity
        # This is a best-effort repair to ensure all customers are served if feasible
        if routes:
            last_route = routes[-1]
            load = sum(demands[i] for i in last_route)
            current_idx = last_route[-1] if last_route else -1
            # try to add remaining by nearest-feasible from last stop or depot if route empty
            while unvisited:
                best_candidate = None
                best_dist = float('inf')
                for i in list(unvisited):
                    if load + demands[i] > CAP + 1e-9:
                        continue
                    if current_idx == -1:
                        d = depot_dists[i]
                    else:
                        d = between[current_idx][i]
                    if d < best_dist - 1e-12:
                        best_dist = d
                        best_candidate = i
                if best_candidate is None:
                    break
                last_route.append(best_candidate)
                unvisited.remove(best_candidate)
                load += demands[best_candidate]
                current_idx = best_candidate
        # If still unvisited, problem instance is not solvable with given K and heuristic.
        # We'll leave remaining unvisited as is (the specification expects all customers visited).
        # To be strict, one could raise an error or try more sophisticated methods.

    # Compute total distance of the produced routes (depot -> first -> ... -> last -> depot)
    total_distance = 0.0
    for route in routes:
        if not route:
            continue
        # from depot to first
        total_distance += depot_dists[route[0]]
        # between consecutive customers
        for a, b in zip(route, route[1:]):
            total_distance += between[a][b]
        # back to depot from last
        total_distance += depot_dists[route[-1]]

    # Output
    out_lines = []
    out_lines.append(f"{total_distance:.6f}")
    out_lines.append(str(len(routes)))
    for route in routes:
        if route:
            out_lines.append(" ".join(str(i) for i in route))
        else:
            out_lines.append("")
    return "\n".join(out_lines)

def main():
    cfg = read_input()
    if cfg is None:
        return
    result = solve_vrp(cfg)
    print(result)

if __name__ == "__main__":
    main()
