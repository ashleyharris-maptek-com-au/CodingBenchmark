use std::cmp::Ordering;
use std::io::{self, Read};

fn mitm_solve(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    // Meet-in-the-middle:
    // Split into left and right halves
    let n = nums.len();
    let n1 = n / 2;
    let n2 = n - n1;

    // Prepare left sums: (sum, mask)
    let left_len = 1usize << n1;
    let mut left: Vec<(i64, u64)> = Vec::with_capacity(left_len);
    for mask in 0..left_len {
        let mut sum = 0i64;
        for i in 0..n1 {
            if (mask >> i) & 1 == 1 {
                sum += nums[i];
            }
        }
        left.push((sum, mask as u64));
    }

    // Prepare right sums
    let right_len = 1usize << n2;
    let mut right: Vec<(i64, u64)> = Vec::with_capacity(right_len);
    for mask in 0..right_len {
        let mut sum = 0i64;
        for i in 0..n2 {
            if (mask >> i) & 1 == 1 {
                sum += nums[n1 + i];
            }
        }
        right.push((sum, mask as u64));
    }

    // Sort both by sum
    left.sort_by(|a, b| a.0.cmp(&b.0));
    right.sort_by(|a, b| a.0.cmp(&b.0));

    // Two-pointer to find sums adding to target
    let mut i = 0isize;
    let mut j = (right.len() as isize) - 1;

    while i < left.len() as isize && j >= 0 {
        let li = &left[i as usize];
        let rj = &right[j as usize];
        let s = li.0 + rj.0;
        match s.cmp(&target) {
            Ordering::Less => i += 1,
            Ordering::Greater => j -= 1,
            Ordering::Equal => {
                // Reconstruct indices from masks
                let mut indices = Vec::new();

                // left mask
                let left_mask = li.1;
                for b in 0..n1 {
                    if ((left_mask >> b) & 1) != 0 {
                        indices.push(b);
                    }
                }
                // right mask
                let right_mask = rj.1;
                for b in 0..n2 {
                    if ((right_mask >> b) & 1) != 0 {
                        indices.push(n1 + b);
                    }
                }

                return Some(indices);
            }
        }
    }

    None
}

fn solve_nonnegative_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    // 0-1 knapsack DP for non-negative numbers only
    if target < 0 {
        return None;
    }
    // quick prune
    if nums.iter().all(|&x| x >= 0) {
        // Convert to usize target for indexing
        let t = target as usize;
        let n = nums.len();
        // reachable and last_index arrays
        let mut reachable = vec![false; t + 1];
        let mut last_index = vec![-1i32; t + 1];
        reachable[0] = true;

        for i in 0..n {
            let val = nums[i] as usize;
            if val > t { continue; }
            // update from high to low to avoid reuse
            for s in (val..=t).rev() {
                if !reachable[s] && reachable[s - val] {
                    reachable[s] = true;
                    last_index[s] = i as i32;
                }
            }
        }

        if reachable[t] {
            // reconstruct
            let mut s = t;
            let mut indices = Vec::new();
            while s > 0 {
                let idx = last_index[s] as usize;
                if last_index[s] < 0 {
                    break;
                }
                indices.push(idx);
                s = s - (nums[idx] as usize);
            }
            indices.reverse();
            return Some(indices);
        }
    }
    None
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace().map(|x| x.parse::<i64>().unwrap());

    // First two numbers: num_numbers and target
    let n = match it.next() {
        Some(v) => v as usize,
        None => {
            println!("NO");
            println!(); // no subset
            return;
        }
    };
    let target = match it.next() {
        Some(v) => v,
        None => {
            println!("NO");
            println!();
            return;
        }
    };

    // Read next n numbers
    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(v) = it.next() {
            nums.push(v);
        } else {
            nums.push(0);
        }
    }

    // Edge cases
    if n == 0 {
        if target == 0 {
            println!("YES");
            println!("[ ]"); // empty subset
        } else {
            println!("NO");
        }
        return;
    }

    // If there exists subset sum exactly target
    // Strategy selection:
    // 1) If n <= 40: use MITM
    // 2) Else if all numbers non-negative and target non-negative and target is not huge: DP
    // 3) Fallback: MITM for up to 45 to be safe; else report NO (best effort)
    let ans = if n <= 40 {
        mitm_solve(&nums, target)
    } else {
        if nums.iter().all(|&x| x >= 0) && target >= 0 {
            solve_nonnegative_dp(&nums, target)
        } else {
            // Fallback: try MITM with first 40 numbers to give a best-effort chance
            // If n > 40, we try MITM on first 40 and ignore the rest (approximate, not guaranteed)
            // This is not guaranteed exact, but provides a best-effort in constraints.
            let cap = 40.min(n);
            mitm_solve(&nums[..cap], target)
                .map(|mut idxs| {
                    // If solution found, but may exclude elements beyond cap.
                    // We cannot safely include elements outside; fall back to NO if not complete.
                    // However, to maintain correctness, only accept if all indices are within 0..cap.
                    idxs.retain(|&id| id < cap);
                    idxs
                })
        }
    };

    match ans {
        Some(indices) => {
            // Validate: compute sum to ensure correctness
            let mut sum = 0i64;
            for &idx in &indices {
                sum += nums[idx];
            }
            if sum != target {
                // In case of inconsistency, fall back to NO
                println!("NO");
            } else {
                println!("YES");
                // Output as "[i j k]" with space-separated indices
                // Ensure unique and within bounds
                let mut out = String::from("[");
                for (i, idx) in indices.iter().enumerate() {
                    if i > 0 {
                        out.push(' ');
                    }
                    out.push_str(&idx.to_string());
                }
                out.push(']');
                println!("{}", out);
            }
        }
        None => {
            println!("NO");
        }
    }
}
