// plan-grid-paths.ts
// Node.js + TypeScript solution for a grid path planning benchmark.
// Reads JSON from stdin and outputs a JSON with planned paths for each drone.
//
// Input JSON shape:
// {
//   "probability": number[][],      // -1 indicates obstacle, positive values indicate search value
//   "drones": [{ "id": string | number, "x": number, "y": number }, ...],
//   "maxStepsPerDrone": number
// }
//
// Output JSON shape:
// { "paths": [ [ [x,y], [x,y], ... ], ... ] }
//
// Each path corresponds to the drone at the same index in the input `drones` array.
// Moves allowed: up, down, left, right, or stay in place (no diagonal moves).
// All moves must stay within the grid bounds and avoid obstacles (-1 values in probability).
// The path starts at the drone's starting position.

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Pt = { x: number; y: number };
type Drone = { id: any; x: number; y: number };

function inBounds(x: number, y: number, grid: number[][]): boolean {
  return y >= 0 && y < grid.length && x >= 0 && x < grid[0].length;
}

function isObstacle(x: number, y: number, grid: number[][]): boolean {
  if (!inBounds(x, y, grid)) return true;
  return grid[y][x] === -1;
}

function bestNextMove(pos: Pt, grid: number[][]): Pt | null {
  // Options: up, down, left, right, stay
  const moves: Pt[] = [
    { x: pos.x,     y: pos.y - 1 }, // up
    { x: pos.x,     y: pos.y + 1 }, // down
    { x: pos.x - 1, y: pos.y     }, // left
    { x: pos.x + 1, y: pos.y     }, // right
    { x: pos.x,     y: pos.y     }  // stay
  ];

  // Filter valid and pick the one with maximum probability value at the target cell
  // If cell is obstacle (-1) or out of bounds, discard
  let best: Pt | null = null;
  let bestVal = -Infinity;

  for (const m of moves) {
    if (!inBounds(m.x, m.y, grid)) continue;
    if (grid[m.y][m.x] === -1) continue;
    const val = grid[m.y][m.x];
    // Prefer higher probability value; if equal, keep the first encountered (deterministic)
    if (val > bestVal) {
      bestVal = val;
      best = m;
    }
  }

  // If no valid moves (shouldn't happen since stay is allowed unless current is obstacle),
  // return null to signal termination.
  return best;
}

function main() {
  const grid: number[][] = data.probability;
  const drones: Drone[] = data.drones;
  const maxSteps: number = data.maxStepsPerDrone;

  const paths: number[][][] = [];

  // For each drone, generate a path up to maxSteps moves (inclusive of starting position)
  for (let i = 0; i < drones.length; i++) {
    const d = drones[i];
    // Start position must be inside grid and not an obstacle; if invalid, place as is but clamp?
    // We'll clamp within bounds and avoid obstacles by adjusting to nearest valid cell.
    let startX = d.x;
    let startY = d.y;

    // Clamp to grid and avoid obstacles for starting point
    if (!inBounds(startX, startY, grid) || grid[startY][startX] === -1) {
      // If invalid, attempt to find the nearest non-obstacle cell within a small radius.
      let found = false;
      const maxRadius = Math.max(grid[0].length, grid.length);
      for (let r = 1; r <= maxRadius && !found; r++) {
        for (let dy = -r; dy <= r; dy++) {
          for (let dx = -r; dx <= r; dx++) {
            const nx = startX + dx;
            const ny = startY + dy;
            if (inBounds(nx, ny, grid) && grid[ny][nx] !== -1) {
              startX = nx;
              startY = ny;
              found = true;
              break;
            }
          }
          if (found) break;
        }
      }
      // If still not found, keep original (likely obstacle) and let moves handle it.
    }

    const path: number[][] = [];
    path.push([startX, startY]);
    let pos: Pt = { x: startX, y: startY };

    for (let step = 0; step < maxSteps; step++) {
      // If current position is obstacle, we cannot move; break
      if (!inBounds(pos.x, pos.y, grid) || grid[pos.y][pos.x] === -1) {
        break;
      }
      const nxt = bestNextMove(pos, grid);
      if (nxt == null) {
        // No valid moves; stop
        break;
      }
      pos = { x: nxt.x, y: nxt.y };
      path.push([pos.x, pos.y]);
    }

    paths.push(path);
  }

  const output = { paths };
  console.log(JSON.stringify(output));
}

main();
