cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Mandelbrot rendering based on UV coordinates
    float2 c = float2(input.uv.x * 3.0 - 2.0, input.uv.y * 2.0 - 1.0);
    float2 z = float2(0.0, 0.0);

    int iterations = 20;
    int i;
    for (i = 0; i < iterations; ++i)
    {
        // z = z^2 + c
        float x = z.x;
        float y = z.y;
        float x2 = x * x;
        float y2 = y * y;

        float2 znew;
        znew.x = x2 - y2 + c.x;
        znew.y = 2.0 * x * y + c.y;
        z = znew;

        if (dot(z, z) > 4.0)
        {
            break;
        }
    }

    float t = float(i) / float(iterations);
    float3 color = float3(t, t * 0.5, 1.0 - t);
    return float4(color, 1.0);
}
