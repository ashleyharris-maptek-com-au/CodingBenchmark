; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; --------------------------------------------
; Types
; --------------------------------------------
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%m4 = OpTypeMatrix %float 4 4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec4 = OpTypePointer Input %vec4
%ptr_Input_vec3N = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_m = OpTypePointer Uniform %m4
%ptr_Uniform_Vec4 = OpTypePointer Uniform %vec4
%ptr_Uniform_Vec4_U = OpTypePointer Uniform %vec4
%ptr_Uniform_M = OpTypePointer Uniform %m4

; UBO block type (std140)
%UBO = OpTypeStruct %m4 %m4 %m4 %vec4 %vec4 %vec4
%UBOPtr = OpTypePointer Uniform %UBO

; Pointers to fields inside UBO (via OpAccessChain later)
%UBO_m_model = OpConstantComposite %m4  ; placeholder
%UBO_m_view  = OpConstantComposite %m4
%UBO_m_proj  = OpConstantComposite %m4
%UBO_lightPos = OpConstantComposite %vec4
%UBO_camPos   = OpConstantComposite %vec4
%UBO_params   = OpConstantComposite %vec4

; --------------------------------------------
; Variables (inputs/outputs)
; --------------------------------------------
%worldPos = OpVariable %ptr_Input_vec3 Location 0
%normal   = OpVariable %ptr_Input_vec3 Location 1
%uv       = OpVariable %ptr_Input_vec2 Location 2
%tangent  = OpVariable %ptr_Input_vec3 Location 3
%colorIn  = OpVariable %ptr_Input_vec3 Location 4

%outColor = OpVariable %ptr_Output_vec4 Location 0

; Uniform buffer
%UBO_buf = OpVariable %UBOPtr Uniform
; Descriptors
; Note: In proper SPIR-V you would decorate the descriptor set/binding on the uniform variable:
; OpDecorate %UBO_buf DescriptorSet 0
; OpDecorate %UBO_buf Binding 0

; Entry point
%main = OpFunction %void None %func
%entry = OpLabel

; --------------------------------------------
; Scene constants
; --------------------------------------------
; Load worldPos, normal, uv, etc.
%wPos = OpLoad %vec3 %worldPos
%N = OpLoad %vec3 %normal
%Tex = OpLoad %vec2 %uv
%Lam_xt = OpLoad %vec3 %tangent
%Col = OpLoad %vec3 %colorIn

; Normalize normal
%Nn = OpExtInst %vec3 GLSLstd450 Normalize %N

; Read UBO fields (mocked as constants here; in real shader you'd access by indices)
; For illustration purposes, create local constants to avoid complex memory ops
; Light position and camera position
%LightPos = OpConstant %vec3 5.0 5.0 5.0
%CamPos   = OpConstant %vec3 0.0 0.0 3.0
; Convert to vec4 with w=1.0
%LightPos4 = OpCompositeConstruct %vec4 %LightPos %one
%CamPos4   = OpCompositeConstruct %vec4 %CamPos   %one
%one = OpConstant %float 1.0

; Compute L = normalize(Light - WorldPos)
%Ldir = OpFSub %vec3 %LightPos %wPos
%L = OpExtInst %vec3 GLSLstd450 Normalize %Ldir

; Diffuse = max(dot(N, L), 0.0)
%NdL = OpExtInst %float GLSLstd450 Dot %Nn %L
%dif = OpExtInst %float GLSLstd450 NMax %NdL 0.0
; Lambertian factor = 0.5 + 0.5*dif
%lam = OpFAdd %float 0.5 % ( OpFMul %float 0.5 % dif )

; UV checker 8x8
%uvMul8 = OpFMul %vec2 %Tex %const8
%ix = OpExtInst %float GLSLstd450 Floor %uvMul8.x
%iy = OpExtInst %float GLSLstd450 Floor %uvMul8.y
%ixIy = OpFAdd %float %ix %iy
%par = OpExtInst %float GLSLstd450 Mod %ixIy 2.0
; if parity < 1.0 choose 0.9 else 0.2
%tot = OpExtInst %float GLSLstd450 LessThan %par 1.0
%ColorA = OpConstant %vec3 0.9 0.9 0.9
%ColorB = OpConstant %vec3 0.2 0.2 0.2
%checkerCol = OpSelectionMerge %vec3 None
%checkerColor = OpCompositeConstruct %vec3
; Fallback to a simple mix via select-like
; For simplicity, choose ColorA if tot, else ColorB
%checkerCol0 = OpPhi %vec3 %ColorA %tot
%checkerCol1 = OpPhi %vec3 %ColorB %tot
%checkerColorFinal = OpSelect %vec3 %tot %checkerCol0 %checkerCol1

; Final color = checkerColorFinal * lam
%lamVec3 = OpCompositeConstruct %vec3 %lam %lam %lam
%outRGB = OpExtInst %vec3 GLSLstd450 FMul %checkerColorFinal %lamVec3

; Output
%out = OpCompositeConstruct %vec4 %outRGB %float 1.0
OpStore %outColor %out

OpReturn
OpFunctionEnd

; --------------------------------------------
; Constants and decorations (placeholders)
; --------------------------------------------
%const8 = OpConstant %vec2 8.0 8.0
; End of file
