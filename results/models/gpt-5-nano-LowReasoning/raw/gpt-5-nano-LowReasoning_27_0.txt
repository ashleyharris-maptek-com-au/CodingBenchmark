```cpp
#include <bits/stdc++.h>
using namespace std;

/*
    Graph Coloring: exact backtracking with DSATUR heuristic + adaptive fallback.
    - Input: n m k followed by m edges (0-indexed vertices)
    - Output: a line with n color numbers (0..k-1) for each vertex

    Strategy:
    1) Try an exact DSATUR-based backtracking to color with at most k colors.
       - The backtracking prefers vertices with highest saturation (number of
         different colors seen among its neighbors) and breaks ties by degree.
       - It allows using a new color (up to k-1 index) if needed, to attempt
         to reach exactly k colors.
    2) If the exact search exceeds a time budget or fails to find a k-coloring,
       fall back to a greedy coloring (largest-degree first) with colors 0..k-1.
       Then, attempt a post-processing step to introduce any unused colors by
       re-coloring individual vertices to an unused color that is not used by
       its neighbors.

    Notes:
    - The code is designed to compile with MSVC (C++17 / C++20) or g++.
    - Time budget is implemented to keep the solver responsive. If time runs out,
      we return the best coloring found (which should still be valid, though it
      may not use all k colors; a post-processing step tries to enforce that).
*/

struct GraphColorer {
    int n;
    vector<vector<int>> adj;

    // Backtracking state
    vector<int> color;            // color[v] = assigned color or -1
    int k;                          // target number of colors
    uint64_t timeBudgetMs;          // time budget for the exact search
    bool timedOut = false;

    // Timing
    chrono::time_point<chrono::steady_clock> startTime;

    GraphColorer(int n_, const vector<vector<int>>& adj_, int k_, uint64_t ms)
        : n(n_), adj(adj_), k(k_), timeBudgetMs(ms) {
        color.assign(n, -1);
    }

    // Helpers
    inline uint64_t elapsedMs() const {
        auto now = chrono::steady_clock::now();
        auto d = chrono::duration_cast<chrono::milliseconds>(now - startTime);
        return (uint64_t)d.count();
    }

    // Compute neighbor colors mask for a vertex (colors that appear among its colored neighbors)
    uint64_t neighborMask(int v) const {
        uint64_t mask = 0;
        for (int u : adj[v]) {
            int c = color[u];
            if (c >= 0) mask |= (1ULL << c);
        }
        return mask;
    }

    // Degree helper
    int degree(int v) const {
        return (int)adj[v].size();
    }

    // DSATUR backtracking
    bool dfs(int colored, int usedColors) {
        if (timedOut) return false;
        if (colored == n) {
            // Successfully colored all vertices
            return true;
        }

        // Time check
        if (timeBudgetMs > 0 && elapsedMs() > timeBudgetMs) {
            timedOut = true;
            return false;
        }

        // Select uncolored vertex with max saturation, break ties by degree
        int bestV = -1;
        int bestSat = -1;
        int bestDeg = -1;
        uint64_t bestMask = 0;

        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            uint64_t neighMask = neighborMask(v);
            int sat = (int)std::popcount(neighMask);
            int deg = degree(v);
            if (sat > bestSat || (sat == bestSat && deg > bestDeg)) {
                bestSat = sat;
                bestDeg = deg;
                bestV = v;
                bestMask = neighMask;
            }
        }

        if (bestV == -1) return false; // should not happen

        int v = bestV;
        uint64_t neighMask = bestMask;

        // Try existing colors first (that do not appear in neighborMask)
        // We try colors 0..min(usedColors-1, k-1)
        for (int c = 0; c < usedColors; ++c) {
            if ((neighMask & (1ULL << c)) == 0) {
                color[v] = c;
                if (dfs(colored + 1, usedColors)) return true;
                color[v] = -1;
            }
        }

        // Try a new color if we still can use up to k colors
        if (usedColors < k) {
            int c = usedColors; // new color index
            if ((neighMask & (1ULL << c)) == 0) {
                color[v] = c;
                if (dfs(colored + 1, usedColors + 1)) return true;
                color[v] = -1;
            }
        }

        return false;
    }

    // Public interface to run exact search
    bool solveExact() {
        startTime = chrono::steady_clock::now();
        timedOut = false;
        int colored = 0;
        int usedColors = 0;
        // Start with no color assignments, the DFS will handle choices
        bool res = dfs(0, 0);
        return res && !timedOut;
    }

    // Fallback: greedy coloring with k colors (largest-degree first)
    vector<int> greedyColoring() {
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b){
            int da = degree(a), db = degree(b);
            if (da != db) return da > db;
            return a < b;
        });
        vector<int> col(n, -1);
        for (int v : order) {
            uint64_t neighMask = 0;
            for (int u : adj[v]) {
                if (col[u] >= 0) neighMask |= (1ULL << col[u]);
            }
            int c = 0;
            while (c < k && (neighMask & (1ULL << c)) != 0) ++c;
            if (c >= k) {
                // cannot color with k colors in this greedy pass
                // leave as -1 to indicate failure
                col[v] = -1;
            } else {
                col[v] = c;
            }
        }
        return col;
    }

    // Post-processing: try to introduce unused colors by recoloring single vertices
    void tryIntroduceUnusedColors(vector<int>& col) {
        // Count used colors
        vector<int> used(k, 0);
        for (int c : col) if (c >= 0 && c < k) used[c] = 1;
        int usedCount = accumulate(used.begin(), used.end(), 0);

        // If already using all colors, nothing to do
        if (usedCount == k) return;

        bool progress = true;
        // Repeat attempts to assign some vertex to an unused color
        while (progress && usedCount < k) {
            progress = false;
            for (int v = 0; v < n; ++v) {
                if (col[v] < 0) continue;
                // If we can assign a new color to this vertex without conflict
                // try colors 0..k-1 that are not used by its neighbors
                uint64_t neighMask = 0;
                for (int u : adj[v]) {
                    int cu = col[u];
                    if (cu >= 0) neighMask |= (1ULL << cu);
                }
                // Try to move to any color that is not in neighMask and not already used by v
                for (int c = 0; c < k; ++c) {
                    if ((neighMask & (1ULL << c)) == 0 && col[v] != c) {
                        // Move v to color c
                        col[v] = c;
                        // Update used tracking
                        // recompute used
                        vector<int> used2(k, 0);
                        int cnt2 = 0;
                        for (int w = 0; w < n; ++w) {
                            if (col[w] >= 0) {
                                if (!used2[col[w]]) {
                                    used2[col[w]] = 1;
                                    ++cnt2;
                                }
                            }
                        }
                        if (cnt2 > usedCount) {
                            usedCount = cnt2;
                            progress = true;
                        }
                        break;
                    }
                }
                if (usedCount == k) break;
            }
        }
        // If still not all k colors used, try a simple re-coloring pass:
        if (usedCount < k) {
            // For each uncolored or color-movable vertex, try to assign an unused color
            for (int v = 0; v < n && usedCount < k; ++v) {
                uint64_t neighMask = 0;
                for (int u : adj[v]) {
                    if (col[u] >= 0) neighMask |= (1ULL << col[u]);
                }
                // find a color not in neighMask
                for (int c = 0; c < k; ++c) {
                    if ((neighMask & (1ULL << c)) == 0) {
                        if (col[v] != c) {
                            col[v] = c;
                            // update usedCount
                            vector<int> used2(k, 0);
                            int cnt2 = 0;
                            for (int w = 0; w < n; ++w) {
                                if (col[w] >= 0) {
                                    if (!used2[col[w]]) {
                                        used2[col[w]] = 1;
                                        ++cnt2;
                                    }
                                }
                            }
                            usedCount = cnt2;
                        }
                        break;
                    }
                }
            }
        }
        // Final: ensure no conflicts (basic assertion for safety)
        for (int v = 0; v < n; ++v) {
            if (col[v] < 0) col[v] = 0; // fallback to color 0 if somehow uncolored
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Normalize: remove multi-edges (optional)
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    // Strategy selection: adaptive
    // Conservative time budget for exact search (in milliseconds)
    uint64_t exactTimeBudgetMs = 4ull * 60ull * 1000ull; // 4 minutes

    GraphColorer solver(n, adj, k, exactTimeBudgetMs);

    // Try exact DSATUR-based coloring
    bool success = solver.solveExact();

    vector<int> finalColors(n, -1);
    if (success && !solver.timedOut) {
        // We found a coloring using up to k colors.
        // Ensure all vertices colored
        bool allColored = true;
        for (int c : solver.color) if (c < 0) { allColored = false; break; }
        if (allColored) {
            finalColors = solver.color;
        }
    }

    if (finalColors[0] == -1) {
        // Fallback: greedy coloring with up to k colors
        finalColors = solver.greedyColoring();
        // If greedy failed to color some vertex (=-1), attempt to fix by simple reassignment
        bool haveUncolored = false;
        for (int c : finalColors) if (c < 0) { haveUncolored = true; break; }
        if (haveUncolored) {
            // Try to recolor uncolored vertices with a simple local search
            for (int v = 0; v < n; ++v) {
                if (finalColors[v] != -1) continue;
                uint64_t neighMask = 0;
                for (int u : adj[v]) {
                    int cu = finalColors[u];
                    if (cu >= 0) neighMask |= (1ULL << cu);
                }
                int c = 0;
                while (c < k && (neighMask & (1ULL << c)) != 0) ++c;
                if (c < k) finalColors[v] = c;
                else finalColors[v] = 0; // last resort
            }
        }
        // Post-processing to attempt using all k colors
        solver.tryIntroduceUnusedColors(finalColors);
    }

    // Final sanity: ensure all edges have different colors
    bool valid = true;
    for (int v = 0; v < n && valid; ++v) {
        if (finalColors[v] < 0 || finalColors[v] >= k) {
            valid = false;
            break;
        }
        for (int u : adj[v]) {
            if (u < v && finalColors[u] == finalColors[v]) {
                valid = false;
                break;
            }
        }
    }

    if (!valid) {
        // As a last resort, try a robust deterministic greedy with recoloring passes
        // Simple approach: try to assign colors sequentially ensuring validity
        vector<int> col(n, -1);
        for (int v = 0; v < n; ++v) {
            uint64_t neighMask = 0;
            for (int u : adj[v]) if (col[u] >= 0) neighMask |= (1ULL << col[u]);
            int c = 0;
            for (; c < k; ++c) {
                if ((neighMask & (1ULL << c)) == 0) break;
            }
            if (c < k) col[v] = c;
            else col[v] = 0; // fallback
        }
        finalColors = col;
    }

    // Output: colors for vertices 0..n-1
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        int c = finalColors[i];
        if (c < 0) c = 0;
        cout << c;
    }
    cout << "\n";

    return 0;
}
```
