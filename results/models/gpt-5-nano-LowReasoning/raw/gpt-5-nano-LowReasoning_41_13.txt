; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%bool = OpTypeBool
%mat4 = OpTypeMatrix %float 4

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_05 = OpConstant %float 0.5

; Inputs / Outputs
%in_worldPos = OpVariable %_ptr_Input_v3 %vec3
%outColor = OpVariable %_ptr_Output_v4 %vec4

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %outColor Location 0

; Type pointers
%_ptr_Input_v3 = OpTypePointer Input %vec3
%_ptr_Output_v4 = OpTypePointer Output %vec4

; Function
%main = OpFunction %void None %function_type
%_func_type = OpTypeFunction %void
; No parameters for main
OpLabel

; Access worldPos
%worldPos = OpVariable %_ptr_Input_v3 Input
OpStore %in_worldPos %vec3_value_undef  ; placeholder to ensure variable exists (will be overwritten by real input)

; Load worldPos
%wp = OpLoad %vec3 %in_worldPos
; Extract y component
%y = OpCompositeExtract %float %wp 1

; Compute t = clamp(y * 0.5 + 0.5, 0, 1)
%half_y = OpFMul %float %y %float_05
%t = OpFAdd %float %half_y %float_05

; Clamp to [0,1]
%t_min = OpFMin %float %t %float_1
%t_clamped = OpFMax %float %float_0 %t_min

; Colors
%color_bottom = OpConstantComposite %vec3 %float_0_1  ; placeholder
; Since we cannot define direct named constants, construct vec3 literals
%bottom_x = OpConstant %float 0.1
%bottom_y = OpConstant %float 0.4
%bottom_z = OpConstant %float 0.1
%vec3_bottom = OpConstantComposite %vec3 %bottom_x %bottom_y %bottom_z

%top_x = OpConstant %float 1.0
%top_y = OpConstant %float 1.0
%top_z = OpConstant %float 1.0
%vec3_top = OpConstantComposite %vec3 %top_x %top_y %top_z

; Compute (1 - t)
%one_minus_t = OpFSub %float %float_1 %t_clamped

; bottom * (1 - t)
%bottom_term = OpVectorTimesScalar %vec3 %vec3_bottom %one_minus_t

; top * t
%top_term = OpVectorTimesScalar %vec3 %vec3_top %t_clamped

; color_vec3 = bottom_term + top_term
%color_vec3 = OpVectorAdd %vec3 %bottom_term %top_term

; vec4(color, 1.0)
%color_vec4 = OpCompositeConstruct %vec4 %color_vec3 %float_1

; Store to output
OpStore %outColor %color_vec4

; Return
OpReturn
OpFunctionEnd

; Note: The above SPIR-V blob is a simplified assembly representation focusing on the color computation logic as requested.
