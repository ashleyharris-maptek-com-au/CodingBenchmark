// Node.js + TypeScript program for a simple firebreak placement benchmark.
// Reads JSON from stdin, computes a greedy set of cut cells along shortest paths
// from ignitions to towns, and outputs the cut coordinates as JSON.
//
// Input format (example):
// {
//   "width": 5,
//   "height": 5,
//   "grid": [ ".....", ".~...", ".....", "...~.", "....." ],
//   "ignitions": [[0,0]],
//   "towns": [ {"id":"A","x":4,"y":4,"value":10} ],
//   "cutBudget": 10
// }
//
// Output format:
// {"firebreaks": [[x, y], [x, y], ...]}

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Town = { id: string; x: number; y: number; value: number };

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid; // array of strings, each length width
const ignitions: Array<[number, number]> = data.ignitions;
const towns: Town[] = data.towns || [];
const cutBudget: number = typeof data.cutBudget === 'number' ? data.cutBudget : 0;

// Helpers
function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

// BFS from all ignitions through burnable cells ('.')
//
// We'll treat '.' as burnable forest.
// '~' is water/rock (blocked).
// We compute dist[y][x] = distance from nearest ignition, or -1 if unreachable.
// We also store prev[y][x] = previous cell along the shortest path (as a string "x,y")
// for path reconstruction later.
const dist: number[][] = Array.from({ length: height }, () => Array(width).fill(-1));
const prev: Array<Array<string | null>> = Array.from({ length: height }, () => Array(width).fill(null));

// Initialize queue with all ignitions
const qx: number[] = [];
const qy: number[] = [];
for (const [ix, iy] of ignitions) {
  if (inBounds(ix, iy) && grid[iy][ix] === '.') {
    dist[iy][ix] = 0;
    qx.push(ix);
    qy.push(iy);
  }
}

// Simple BFS in 4 directions
for (let head = 0; head < qx.length; head++) {
  const x = qx[head];
  const y = qy[head];
  const d = dist[y][x];
  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];
  for (const [dx, dy] of dirs) {
    const nx = x + dx;
    const ny = y + dy;
    if (!inBounds(nx, ny)) continue;
    // Fire can spread through burnable forest ('.')
    if (grid[ny][nx] !== '.') continue;
    if (dist[ny][nx] !== -1) continue;
    dist[ny][nx] = d + 1;
    prev[ny][nx] = `${x},${y}`;
    qx.push(nx);
    qy.push(ny);
  }
}

// Reconstruct paths from towns back to ignitions along shortest paths.
// Collect candidate cut cells along these paths (excluding ignition cells).
// We'll take unique coordinates, up to cutBudget.
const cutSet: Set<string> = new Set<string>();

for (const t of towns) {
  const tx = t.x;
  const ty = t.y;
  // Town must be on burnable forest and reachable
  if (!inBounds(tx, ty)) continue;
  if (grid[ty][tx] !== '.') continue;
  if (dist[ty][tx] === -1) continue; // unreachable town

  // Walk back from town to ignition along prev pointers
  let cx = tx;
  let cy = ty;
  // While current cell has a predecessor, i.e., not an ignition
  while (true) {
    const d = dist[cy][cx];
    // do not cut ignition cells (dist == 0)
    if (d > 0) {
      const key = `${cx},${cy}`;
      cutSet.add(key);
      // If we already reached budget, break early
      if (cutSet.size >= cutBudget) break;
    }
    const p = prev[cy][cx];
    if (!p) break; // reached ignition or no predecessor
    const [px, py] = p.split(',').map((s) => parseInt(s, 10)) as [number, number];
    cx = px;
    cy = py;
  }
  if (cutSet.size >= cutBudget) break;
}

// Convert cutSet to array of [x,y]
const firebreaks: Array<[number, number]> = Array.from(cutSet).map((s) => {
  const [x, y] = s.split(',').map((v) => parseInt(v, 10)) as [number, number];
  return [x, y];
});

// If budget allows more cuts but we had duplicates or grid constraints, we simply stop at what we've chosen.
// Ensure we don't attempt to cut water/rock or out-of-bounds (we already avoided by construction).
// Return JSON
const output = {
  firebreaks: firebreaks
};

console.log(JSON.stringify(output));
