// Gooch shading in HLSL for Vulkan (GLSL-like HLSL with Vulkan bindings)
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;        // (5,5,5,1)
    float4 cameraPos;        // (0,0,3,1)
    float4 params;           // (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Light direction (towards the light) in world space
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Gooch shading parameter t in [0,1]
    float t = dot(N, L) * 0.5 + 0.5;
    t = clamp(t, 0.0, 1.0);

    // Gooch cool and warm colors
    float3 coolColor = float3(0.2, 0.2, 0.75);
    float3 warmColor = float3(0.7, 0.7, 0.4);

    // Interpolate between cool and warm based on t
    float3 goochColor = lerp(coolColor, warmColor, t);

    // Basic diffuse shading
    float NdotL = max(dot(N, L), 0.0);

    // Apply diffuse term
    float3 litColor = goochColor * NdotL;

    // Modulate with per-vertex color
    float3 finalColor = litColor * input.color;

    // Alpha = 1.0
    return float4(finalColor, 1.0);
}
