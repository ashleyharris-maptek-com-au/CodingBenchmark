```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint N = counts.x;
    uint i = gl_GlobalInvocationID.x;

    // Guard against out-of-bounds work-items
    if (i >= N) {
        return;
    }

    // Read i-th body's position and velocity from input
    vec3 pos_i = inData[i * 2].xyz;
    float mass_i = inData[i * 2].w; // mass is stored in w of the position row (per spec)

    vec3 vel_i = inData[i * 2 + 1].xyz;

    // Parameters
    float dt       = physics.x;
    float softening = physics.y;
    float G        = physics.z;

    // Compute acceleration due to all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue; // skip self-interaction

        vec3 pos_j = inData[j * 2].xyz;
        float mass_j = inData[j * 2].w;

        vec3 dx = pos_j - pos_i;
        float dist2 = dot(dx, dx);

        // Add softening to avoid singularities
        float dist2_plus_soft = dist2 + softening * softening;

        // invDist^3 = 1 / (dist^2 + soft^2)^(3/2)
        float invDist = inversesqrt(dist2_plus_soft);
        float invDist3 = invDist * invDist * invDist;

        acc += (dx * (G * mass_j * invDist3));
    }

    // Update velocity and position
    vec3 vel_old = vel_i;
    vec3 vel_new = vel_old + acc * dt;
    vec3 pos_new = pos_i + vel_old * dt; // position updated by old velocity (explicit Euler for position)

    // Write updated state to output buffers
    outData[i * 2]     = vec4(pos_new, mass_i); // keep mass unchanged in the first vec4's w
    outData[i * 2 + 1] = vec4(vel_new, 0.0);
}
```
