#version 450

layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal
layout(location = 2) in vec2 uv;         // texture coordinates (unused)
layout(location = 3) in vec3 tangent;    // tangent (unused)
layout(location = 4) in vec3 color;      // per-vertex color (albedo)

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;       // (5,5,5,1)
    vec4 cameraPos;      // (0,0,3,1)
    vec4 params;         // (unused in this shader)
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(vec3(ubo.lightPos) - worldPos);
    float d = max(dot(normalize(N), L), 0.0);

    // Hemisphere ambient: blend ground and sky by N.y
    vec3 skyColor = vec3(0.4, 0.6, 1.0);
    vec3 groundColor = vec3(0.3, 0.15, 0.05);
    float t = clamp(N.y * 0.5 + 0.5, 0.0, 1.0);
    vec3 ambientColor = mix(groundColor, skyColor, t);

    vec3 albedo = color;

    vec3 ambientTerm = albedo * ambientColor;
    vec3 diffuseTerm = albedo * (0.3 * d);

    vec3 finalColor = ambientTerm + diffuseTerm;

    outColor = vec4(finalColor, 1.0);
}
