#version 450

layout(location = 0) out vec4 outColor;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

void main() {
    // Normal in world space (assumed provided)
    vec3 N = normalize(normal);

    // Light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    // View direction
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

    // Diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // Rim term: rim = pow(1 - max(dot(N,V), 0), 3)
    float rim = pow( max(1.0 - max(dot(N, V), 0.0), 0.0), 3.0 );

    // Base color modulation
    vec3 baseColor = vec3(0.2, 0.3, 0.8) * (0.3 + 0.5 * diffuse);

    // Rim tint
    vec3 rimTint = vec3(1.0, 0.8, 0.5) * rim;

    vec3 color = baseColor + rimTint;

    outColor = vec4(color, 1.0);
}
