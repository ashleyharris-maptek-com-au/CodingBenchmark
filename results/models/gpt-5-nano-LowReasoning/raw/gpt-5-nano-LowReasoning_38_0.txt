import sys

def main():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        print("NO SOLUTION")
        return

    # Parse first line: U S
    first = data[0].strip().split()
    if len(first) < 2:
        print("NO SOLUTION")
        return
    try:
        U = int(first[0])
        S = int(first[1])
    except ValueError:
        print("NO SOLUTION")
        return

    # Read S lines of sets
    sets_bits = []
    for i in range(1, min(len(data), S+1)):
        line = data[i].strip()
        if line == "":
            elems = []
        else:
            elems = line.split()
        bit = 0
        for e in elems:
            try:
                v = int(e)
            except ValueError:
                continue
            if 0 <= v < U:
                bit |= 1 << v
        sets_bits.append(bit)

    # If fewer lines than S provided, pad with empty sets
    while len(sets_bits) < S:
        sets_bits.append(0)

    full_mask = (1 << U) - 1

    # Precompute element -> list of sets containing it
    contains = [[] for _ in range(U)]
    for idx, bitset in enumerate(sets_bits):
        for v in range(U):
            if (bitset >> v) & 1:
                contains[v].append(idx)

    # Backtracking search for an exact cover
    sys.setrecursionlimit(10000)
    solution = []
    seen = set()  # memoization on (covered_mask, tuple(sorted(solution))) optional

    # We will frequently need to pick an uncovered element with fewest available sets
    def search(covered_mask, start_index=None):
        if covered_mask == full_mask:
            return True
        # Prune by seen states
        key = (covered_mask, tuple(sorted(solution)))
        if key in seen:
            return False
        seen.add(key)

        # Choose an uncovered element with minimal candidates given current covered
        # Find uncovered elements
        # Heuristic: for each element not yet covered, count available sets that can be used without conflicting with current cover
        best_e = -1
        best_opts = None

        for e in range(U):
            if ((covered_mask >> e) & 1) == 0:
                # candidate sets that contain e
                cand = []
                for idx in contains[e]:
                    bitset = sets_bits[idx]
                    # If this set covers any already-covered element, it's invalid (would duplicate)
                    if bitset & covered_mask:
                        continue
                    cand.append(idx)
                if not cand:
                    # No available set can cover this element -> fail fast
                    return False
                if best_opts is None or len(cand) < len(best_opts):
                    best_opts = cand
                    best_e = e
                    if len(best_opts) == 1:
                        break

        if best_opts is None:
            return False

        # Try each candidate set
        for idx in best_opts:
            bitset = sets_bits[idx]
            # If this set tries to cover an already covered element, skip (redundant check)
            if bitset & covered_mask:
                continue
            solution.append(idx)
            if search(covered_mask | bitset, start_index=idx+1):
                return True
            solution.pop()

        return False

    ok = search(0)
    if not ok:
        print("NO SOLUTION")
        return

    # Print solution
    print("SOLUTION")
    print(" ".join(str(i) for i in solution))

if __name__ == "__main__":
    main()
