```cpp
#include <bits/stdc++.h>
using namespace std;

/*
  Solve the 0/1 ILP: maximize sum c[i] * x[i] subject to
    x_u + x_v <= 1 for each given edge (u, v)
  This is equivalent to Maximum Weight Independent Set on the graph with
  vertices 0..n-1 and edges given by (u, v).

  We implement a branch-and-bound algorithm:
    - At each step, maintain a set of candidate vertices (cand) not yet decided.
    - Maintain a current set of included vertices (inc) and current_weight.
    - Upper bound = current_weight + sum(weights of vertices in cand).
      If this <= best_found, prune this branch.
    - Branch on choosing the last vertex v in cand:
        1) Include v: remove v and all its neighbors from cand, add v to inc, increase current_weight.
        2) Exclude v: remove v from cand, inc unchanged.
    - When cand becomes empty, update best solution if current_weight is better.

  This approach yields an exact optimum for reasonable input sizes.
*/

static int n, m;
static vector<long long> w; // weights c[i]
static vector<vector<int>> adj; // adjacency list

static long long best_weight = LLONG_MIN;
static vector<char> best_inc; // best inclusion set (0/1)
static vector<char> current_inc; // current inclusion set during search

// Depth-first search with branch-and-bound
void dfs(vector<int> cand, long long current_weight) {
    if (cand.empty()) {
        if (current_weight > best_weight) {
            best_weight = current_weight;
            best_inc = current_inc;
        }
        return;
    }

    // Compute upper bound: current_weight + sum of weights in cand
    long long sumCand = 0;
    for (int v : cand) sumCand += w[v];
    if (current_weight + sumCand <= best_weight) {
        // prune
        return;
    }

    // Choose a vertex v to branch on. Use the last element for simplicity.
    int v = cand.back();
    // Prepare cand without v
    vector<int> cand_without_v = cand;
    cand_without_v.pop_back();

    // Branch 1: Include v
    // New candidate set: remove v and all neighbors of v from cand_without_v
    vector<char> banned(n, false);
    banned[v] = true;
    for (int u : adj[v]) banned[u] = true;

    vector<int> cand_incl;
    cand_incl.reserve(cand_without_v.size());
    for (int u : cand_without_v) {
        if (!banned[u]) cand_incl.push_back(u);
    }
    // Update current inclusion and weight
    current_inc[v] = 1;
    dfs(move(cand_incl), current_weight + w[v]);
    current_inc[v] = 0;

    // Branch 2: Exclude v
    // cand_without_v remains as is, current_inc unchanged
    dfs(move(cand_without_v), current_weight);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) return 0;
    w.assign(n, 0);
    for (int i = 0; i < n; ++i) cin >> w[i];

    adj.assign(n, {});
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        // assuming 0-based indices as per problem statement
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Initialize best solution (all zeros)
    best_weight = LLONG_MIN;
    best_inc.assign(n, 0);
    current_inc.assign(n, 0);

    // Initial candidate set: all vertices
    vector<int> cand(n);
    iota(cand.begin(), cand.end(), 0);

    dfs(move(cand), 0LL);

    if (best_weight == LLONG_MIN) best_weight = 0;

    // Output: objective value then x_0 ... x_(n-1)
    cout << best_weight << "\n";
    for (int i = 0; i < n; ++i) {
        cout << (best_inc[i] ? 1 : 0) << (i + 1 == n ? '\n' : ' ');
    }

    return 0;
}
```
