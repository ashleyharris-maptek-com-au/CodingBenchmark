```csharp
using System;
using System.IO;
using System.Text;
using System.Linq;
using System.Collections.Generic;

class Program
{
    // Fast integer reader (reads long)
    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0;
        private int _ ptr = 0;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool EndOfStream
        {
            get
            {
                int c;
                while ((c = Read()) != -1)
                {
                    if (!char.IsWhiteSpace((char)c))
                    {
                        _ptr--; // step back
                        return false;
                    }
                }
                return true;
            }
        }

        public long NextLong()
        {
            int c;
            // skip whitespace
            do
            {
                c = Read();
            } while (c != -1 && char.IsWhiteSpace((char)c));

            if (c == -1) throw new EndOfStreamException();

            int sign = 1;
            if (c == '-' )
            {
                sign = -1;
                c = Read();
            }

            long val = 0;
            while (c != -1 && !char.IsWhiteSpace((char)c))
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public int NextInt()
        {
            return (int)NextLong();
        }
    }

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        // Read N
        int N;
        try
        {
            N = (int)scanner.NextLong();
        }
        catch
        {
            return;
        }

        // We'll store matrices as flattened 1D arrays to minimize overhead
        long[] flow = new long[N * N];
        long[] dist = new long[N * N];

        // Read flow matrix: N lines each with N numbers
        for (int i = 0; i < N * N; i++)
        {
            flow[i] = scanner.NextLong();
        }

        // Read dist matrix: N lines each with N numbers
        for (int i = 0; i < N * N; i++)
        {
            dist[i] = scanner.NextLong();
        }

        // Heuristic: assign facilities to locations by sorting by row-sums and column-sums
        // Compute row sums for flow
        long[] rowSumFlow = new long[N];
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            int base = i * N;
            for (int j = 0; j < N; j++)
            {
                s += flow[base + j];
            }
            rowSumFlow[i] = s;
        }

        // Compute column sums for dist (sum over i of dist[i][j])
        long[] colSumDist = new long[N];
        for (int j = 0; j < N; j++)
        {
            long s = 0;
            for (int i = 0; i < N; i++)
            {
                s += dist[i * N + j];
            }
            colSumDist[j] = s;
        }

        // Create ordered indices
        int[] facilities = Enumerable.Range(0, N).ToArray();
        int[] locations = Enumerable.Range(0, N).ToArray();

        // Sort facilities by increasing rowSumFlow
        Array.Sort(facilities, (a, b) => rowSumFlow[a].CompareTo(rowSumFlow[b]));
        // Sort locations by increasing colSumDist
        Array.Sort(locations, (a, b) => colSumDist[a].CompareTo(colSumDist[b]));

        // Build permutation p[i] = location assigned to facility i
        int[] p = new int[N];
        for (int t = 0; t < N; t++)
        {
            int f = facilities[t];
            int loc = locations[t];
            p[f] = loc;
        }

        // Compute total cost for this permutation
        long totalCost = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            int baseFi = i * N;
            for (int j = 0; j < N; j++)
            {
                int pj = p[j];
                totalCost += flow[baseFi + j] * dist[pi * N + pj];
            }
        }

        // Output: total_cost and permutation p[i] for i from 0..N-1
        var sb = new StringBuilder();
        sb.Append(totalCost.ToString());
        sb.Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        sb.Append('\n');

        Console.Write(sb.ToString());
    }
}
```
