using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static void Main()
    {
        // Read all input tokens
        var tokens = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            if (line.Length == 0) continue;
            var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            foreach (var p in parts) tokens.Add(p);
        }

        if (tokens.Count < 2)
        {
            Console.WriteLine("0");
            return;
        }

        int pos = 0;
        int n = int.Parse(tokens[pos++]);
        int stockLength = int.Parse(tokens[pos++]);

        int[] cuts = new int[n];
        for (int i = 0; i < n && pos < tokens.Count; i++)
        {
            cuts[i] = int.Parse(tokens[pos++]);
        }

        // Build cuts with original indices
        var items = new List<(int Length, int Index)>(n);
        for (int i = 0; i < n; i++)
        {
            int len = cuts[i];
            // If a cut is longer than stock, to be safe, still include (will be placed on its own stock)
            items.Add((len, i));
        }

        // Sort cuts by length descending to fit efficiently
        items.Sort((a, b) =>
        {
            int c = b.Length.CompareTo(a.Length);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        });

        // Each stock keeps its assigned cut indices
        var stocks = new List<List<int>>();
        // Remaining length on each stock
        var remaining = new List<int>();

        foreach (var it in items)
        {
            int len = it.Length;
            int idx = it.Index;

            // Try to place into existing stock that can fit
            int placedStock = -1;
            for (int s = 0; s < remaining.Count; s++)
            {
                if (len <= remaining[s])
                {
                    placedStock = s;
                    break;
                }
            }

            if (placedStock >= 0)
            {
                stocks[placedStock].Add(idx);
                remaining[placedStock] -= len;
            }
            else
            {
                // Create new stock
                stocks.Add(new List<int> { idx });
                remaining.Add(stockLength - len);
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.AppendLine(stocks.Count.ToString());
        foreach (var st in stocks)
        {
            if (st.Count == 0)
            {
                sb.AppendLine();
            }
            else
            {
                for (int i = 0; i < st.Count; i++)
                {
                    if (i > 0) sb.Append(' ');
                    sb.Append(st[i]);
                }
                sb.AppendLine();
            }
        }

        Console.Write(sb.ToString());
    }
}
