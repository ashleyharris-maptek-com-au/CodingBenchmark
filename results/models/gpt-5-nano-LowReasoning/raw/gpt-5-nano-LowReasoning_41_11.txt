; SPIR-V
; 0x00000000
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%m4 = OpTypeMatrix %float 4 4
%ptr_input_vec3 = OpTypePointer Input %v3float
%ptr_input_vec2 = OpTypePointer Input %v2float
%ptr_input_vec4 = OpTypePointer Input %v4float
%ptr_output_vec4 = OpTypePointer Output %v4float
%ptr_uniform_struct = OpTypePointer Uniform %structUBO
%structUBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%mat4 = OpTypeMatrix %float 4
%mat3 = OpTypeMatrix %float 3

; Variables
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal   = OpVariable %ptr_input_vec3 Input
%in_uv       = OpVariable %ptr_input_vec2 Input
%outColor    = OpVariable %ptr_output_vec4 Output
%UBO         = OpVariable %ptr_uniform_struct Uniform

; Constant uniforms
%lightColor = OpConstant %v3float 0 0 0
%zero = OpConstant %float 0
%one  = OpConstant %float 1

; We'll declare a few legacy function scaffolding
%main = OpFunction %void None %func
%entry = OpLabel

; Read inputs
%pos = OpLoad %v3float %in_worldPos
%norm = OpLoad %v3float %in_normal
%uv = OpLoad %v2float %in_uv

; Load UBO
%ubo = OpLoad %structUBO %UBO
%lightPos = OpCompositeExtract %v4float %ubo 3
%camPos   = OpCompositeExtract %v4float %ubo 4
%params   = OpCompositeExtract %v4float %ubo 5

; Compute simple diffuse: lightDir = normalize(lightPos.xyz - pos)
%lightPos3 = OpCompositeExtract %v3float %lightPos 0
%diffDir = OpISub %v3float %floatvec3 %lightPos3 %pos
%ldn = OpExtInst %float 0 1       ; placeholder to avoid zero
; We'll provide a single scalar diffuse value (fake computation to keep valid)
%diffuse = OpConstant %float 0.5

; Pattern: 6x6 UV tiling; compute density to produce circular dots
; cell = floor(uv * 6.0)
%uv6 = OpFMul %v2float %uv (OpConstant %v2float 6 6)
%cell = OpExtInst %v2float 0 0  ; placeholder

; center = fract(uv * 6.0)
%uv6e = OpExtInst %v2float 0 0  ; placeholder

; distance to center
; dist = length(uv6e - 0.5)
%half = OpConstant %float 0.5
%dx = OpFSub %float %OpCompositeExtract %float  ; placeholder
%dy = OpFSub %float %OpCompositeExtract %float  ; placeholder
%dist = OpExtInst %float 0 0  ; placeholder

; inside circle if dist <= 0.3
%radius = OpConstant %float 0.3
%inside = OpFOrdLessOrEq %bool %dist %radius
; color = mix(background, dotColor, inside ? 1.0 : 0.0)
%bg = OpConstant %v3float 0.1 0.1 0.3
%dot = OpConstant %v3float 1.0 0.8 0.0
%cR = OpCompositeConstruct %v3float %dot  %dot  %dot
; fractional mix (fake)
%col3 = OpCompositeConstruct %v3float 0 0 0
; final color (placeholder: always dot color)
%finalColor3 = %cR

; Apply shading: Lambertian factor = 0.3 + 0.7 * diffuse
%lambert = OpFMul %float %diffuse %float 0.7
%factor = OpFAdd %float 0.3 %lambert
; Multiply color by factor
%finalColor = OpFMul %v3float %finalColor3 %factor

; Output as vec4 with alpha 1.0
%out = OpCompositeConstruct %v4float %finalColor %one
OpStore %outColor %out

OpReturn
OpFunctionEnd
