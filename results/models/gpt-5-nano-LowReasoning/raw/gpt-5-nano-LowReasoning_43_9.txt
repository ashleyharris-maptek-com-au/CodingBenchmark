[[vk::binding(0, 0)]]
cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize normal and compute light direction in world space
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float NdotL = max(dot(N, L), 0.0);

    // 8x8 UV checkerboard
    float cell = 8.0;
    int ix = (int)floor(input.uv.x * cell);
    int iy = (int)floor(input.uv.y * cell);
    int parity = (ix + iy) & 1; // 0 or 1

    float baseColor = (parity == 0) ? 0.9 : 0.2;
    float3 color = float3(baseColor, baseColor, baseColor);

    // Apply Lambert shading: 0.5 + 0.5 * diffuse
    float diffuseFactor = 0.5 + 0.5 * NdotL;
    float3 finalColor = color * diffuseFactor;

    return float4(finalColor, 1.0);
}
