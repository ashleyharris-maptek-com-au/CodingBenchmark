    [[vk::location(0)]] float3 worldPos : TEXCOORD0;

    [[vk::location(3)]] float3 tangent  : TEXCOORD3;

};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0
    float4x4 view;           // offset 64

    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)

};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(UBO.lightPos.xyz - input.worldPos);
    
    // Calculate view direction (from surface to camera)
    float3 V = normalize(UBO.cameraPos.xyz - input.worldPos);
    
    // Calculate half-vector
    float3 H = normalize(L + V);
    
    // Calculate diffuse component
    float diffuse = max(dot(N, L), 0.0);
    
    // Calculate specular component
    float specular = pow(max(dot(N, H), 0.0), 64.0);
    
    // Calculate final color (grayscale intensity)
    float intensity = 0.1 + 0.6 * diffuse + 0.8 * specular;
    
    return float4(intensity, intensity, intensity, 1.0);
}
