%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

; Input variables
%worldPos = OpVariable %ptr_uniform_v3float Uniform
%normal = OpVariable %ptr_uniform_v3float Uniform
%uv = OpVariable %ptr_uniform_v2float Uniform
%tangent = OpVariable %ptr_uniform_v3float Uniform
%color = OpVariable %ptr_uniform_v3float Uniform

; Function
%main = OpFunction %void None %void
%entry = OpLabel

; Get UV coordinates
%uv_val = OpLoad %v2float %uv

; Calculate checkerboard
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%uv_x_scaled = OpFMul %float %uv_x (8.0)
%uv_y_scaled = OpFMul %float %uv_y (8.0)
%uv_x_floor = OpFNegate %float %uv_x_scaled
%uv_y_floor = OpFNegate %float %uv_y_scaled
%sum = OpFAdd %float %uv_x_floor %uv_y_floor
%checker = OpFAdd %float %sum (1.0)  ; This is a hack to get floor working
%parity = OpFAdd %float %checker (0.5)
%color_val = OpFAdd %float %parity (0.5)
%final_color = OpSelect %float %color_val 0.9 0.2

; Get normal
%normal_val = OpLoad %v3float %normal

; Get light position from UBO
%light_pos_ptr = OpAccessChain %ptr_uniform_v4float %ubo 0
%light_pos = OpLoad %v4float %light_pos_ptr
%light_dir = OpCompositeExtract %v3float %light_pos 0

%lambert = OpFAdd %float %dot_product (1.0)  ; Add 1 to get range 0-2, then divide by 2
%lambert = OpFMul %float %lambert 0.5

; Combine colors
%shaded_color = OpFMul %float %final_color %lambert

; Output color

OpReturn
OpFunctionEnd
