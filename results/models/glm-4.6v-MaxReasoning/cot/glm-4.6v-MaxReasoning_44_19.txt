```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);
    
    // Calculate light direction (from fragment to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate view direction (from fragment to camera)
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Calculate reflection vector
    vec3 R = reflect(-L, N);
    
    // Ambient component
    vec3 ambient = vec3(0.1, 0.05, 0.05);
    
    // Diffuse component - use vertex color and max(dot(N,L), 0)
    vec3 diffuse = color * max(dot(N, L), 0.0);
    
    // Specular component - use (1,1,1) with shininess 32
    float specularIntensity = pow(max(dot(R, V), 0.0), 32.0);
    vec3 specular = vec3(1.0, 1.0, 1.0) * specularIntensity;
    
    // Sum all components
    vec3 result = ambient + diffuse + specular;
    
    // Clamp to [0,1]
    result = clamp(result, 0.0, 1.0);
    
    // Set output color with alpha 1.0
    outColor = vec4(result, 1.0);
}
```
