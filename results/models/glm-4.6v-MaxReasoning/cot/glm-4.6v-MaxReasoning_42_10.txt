void main() {
    // Get input vertices
    vec4 P0 = gl_in[0].gl_Position;
    vec4 P1 = gl_in[1].gl_Position;
    vec4 P2 = gl_in[2].gl_Position;
    vec4 P3 = gl_in[3].gl_Position;

    vec3 C1 = color[1];
    vec3 C2 = color[2];
    vec3 C3 = color[3];

    vec3 N1 = normal[1];
    vec3 N2 = normal[2];

    // Constants
    const int NUM_SEGMENTS = 4;
    const float THICKNESS = 0.02;
    const float INV_NUM_SEGMENTS = 1.0 / float(NUM_SEGMENTS);

    // Loop to emit the strip
    for (int i = 0; i <= NUM_SEGMENTS; ++i) {
        float t = float(i) * INV_NUM_SEGMENTS;

        // Interpolate position (simplified Catmull-Rom)
        vec4 pos = mix(P1, P2, t);

        // Calculate varying tangent
        vec3 tangent_start = normalize(P2.xyz - P0.xyz);
        vec3 tangent_end = normalize(P3.xyz - P1.xyz);
        vec3 tangent = normalize(mix(tangent_start, tangent_end, t));

        // Calculate perpendicular (using interpolated normal for up vector)
        vec3 perp_start = normalize(cross(tangent_start, N1));
        vec3 perp_end = normalize(cross(tangent_end, N2));
        vec3 perp = normalize(mix(perp_start, perp_end, t)) * THICKNESS;

        // Interpolate color
        vec3 c = mix(C1, C2, t);

        // Emit the two vertices for this segment point
        gl_Position = pos + vec4(perp, 0.0);
        outColor = c;
        EmitVertex();

        gl_Position = pos - vec4(perp, 0.0);
        outColor = c;
        EmitVertex();
    }

    EndPrimitive();
}
