Gooch shading is a non-photorealistic rendering technique that uses warm colors for surfaces facing toward the light and cool colors for surfaces facing away from the light. The formula typically involves:
- t = dot(N, L) * 0.5 + 0.5 (where N is the normal and L is the light direction)
- Warm color (for t close to 1)
- Cool color (for t close to 0)
- The final color is a mix between warm and cool based on t

Given:
- Cool color: (0.2, 0.2, 0.75)
- Warm color: (0.7, 0.7, 0.4)
- t = dot(N, L) * 0.5 + 0.5

    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64

    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal vector
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(UBO.lightPos.xyz - input.worldPos);
    
    // Calculate t = dot(N, L) * 0.5 + 0.5
    float t = dot(N, L) * 0.5 + 0.5;
    
    // Define warm and cool colors
    float3 warmColor = float3(0.7, 0.7, 0.4);
    float3 coolColor = float3(0.2, 0.2, 0.75);
    
    // Mix between warm and cool colors based on t
    float3 finalColor = lerp(coolColor, warmColor, t);
    
    // Return with alpha = 1.0
    return float4(finalColor, 1.0);
}
