```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal (it might not be normalized after interpolation)
    float3 N = normalize(input.normal);
    
    // Calculate light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate diffuse factor
    float diffuse = max(0.0, dot(N, L));
    
    // Apply Lambertian shading factor
    float shadingFactor = 0.5 + 0.5 * diffuse;
    
    // Calculate checkerboard pattern
    float2 uv = input.uv * 8.0; // 8x8 frequency
    float2 checker = floor(uv);
    float parity = (checker.x + checker.y) % 2;
    float checkerColor = parity < 0.5 ? 0.9 : 0.2;
    
    // Combine checkerboard color with shading
    float3 finalColor = checkerColor * shadingFactor;
    
    // Return with alpha = 1.0
    return float4(finalColor, 1.0);
}
```
