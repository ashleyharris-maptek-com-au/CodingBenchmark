    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;

[[vk::binding(0, 0)]]

    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal (it might be interpolated and not unit length)
    float3 N = normalize(input.normal);
    
    // Calculate view direction (from fragment to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    
    // Calculate NdotV
    float NdotV = max(dot(N, V), 0.0);
    
    // Calculate color based on NdotV
    float4 outputColor;
    if (NdotV < 0.5) {
        outputColor.r = 1.0;
        outputColor.g = 2.0 * NdotV;
        outputColor.b = 0.0;
    } else {
        outputColor.r = 1.0 - (NdotV - 0.5) * 2.0;
        outputColor.g = 1.0 - (NdotV - 0.5) * 2.0;
        outputColor.b = (NdotV - 0.5) * 2.0;
    }
    outputColor.a = 1.0;
    
    return outputColor;
}
