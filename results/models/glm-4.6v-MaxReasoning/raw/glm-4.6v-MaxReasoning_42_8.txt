
```spirv
; SPIR-V assembly for a geometry shader that shrinks triangles toward their centroid.
; Input primitive: Triangles (3 vertices per invocation)
; Output primitive: TriangleStrip (3 vertices per primitive)
; OutputVertices: 3

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Define types
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%vec3 = OpTypeVector %float 3
%mat4 = OpTypeMatrix %v4float 4
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%worldPos_in = OpTypeArray %vec3 %uint_3
%ptr_in_worldPos = OpTypePointer Input %worldPos_in
%worldPos = OpVariable %ptr_in_worldPos Input

%normal_in = OpTypeArray %vec3 %uint_3
%ptr_in_normal = OpTypePointer Input %normal_in
%normal = OpVariable %ptr_in_normal Input

%color_in = OpTypeArray %vec3 %uint_3
%ptr_in_color = OpTypePointer Input %color_in
%color = OpVariable %ptr_in_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%outColor = OpTypePointer Output %vec3
%outColor_var = OpVariable %outColor Output

; UBO type (same as test 41)
%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %vec3 %vec3 %vec2
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0

; Function type for main
%void = OpTypeVoid
%func_type = OpTypeFunction %void

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Read input positions (gl_in[0].gl_Position, gl_in[1].gl_Position, gl_in[2].gl_Position)
%pos0_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%pos0 = OpLoad %gl_PerVertex_in %pos0_ptr
%pos0_vec = OpCompositeExtract %v4float %pos0 0

%pos1_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_1 %int_0
%pos1 = OpLoad %gl_PerVertex_in %pos1_ptr
%pos1_vec = OpCompositeExtract %v4float %pos1 0

%pos2_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
%pos2 = OpLoad %gl_PerVertex_in %pos2_ptr
%pos2_vec = OpCompositeExtract %v4float %pos2 0

; Compute centroid: C = (p0 + p1 + p2) / 3
%sum = OpFAdd %v4float %pos0_vec %pos1_vec
%sum = OpFAdd %v4float %sum %pos2_vec
%centroid = OpVectorTimesScalar %v4float %sum %float_0.3333333333333333

; Compute new positions: new_pos_i = C + 0.7 * (p_i - C)
%diff0 = OpFSub %v4float %pos0_vec %centroid
%new_pos0 = OpFAdd %v4float %centroid (OpVectorTimesScalar %v4float %diff0 %float_0.7)

%diff1 = OpFSub %v4float %pos1_vec %centroid
%new_pos1 = OpFAdd %v4float %centroid (OpVectorTimesScalar %v4float %diff1 %float_0.7)

%diff2 = OpFSub %v4float %pos2_vec %centroid
%new_pos2 = OpFAdd %v4float %centroid (OpVectorTimesScalar %v4float %diff2 %float_0.7)

; Emit vertex 0
%out_pos0_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
%out_pos0_struct = OpLoad %gl_PerVertex_out %out_pos0_ptr
%out_pos0_vec = OpCompositeInsert %gl_PerVertex_out %new_pos0 %out_pos0_struct 0
OpStore %out_pos0_ptr %out_pos0_vec

%color0_ptr = OpAccessChain %ptr_in_color %color %int_0
%color0 = OpLoad %vec3 %color0_ptr
OpStore %outColor_var %color0
OpEmitVertex

; Emit vertex 1
%out_pos1_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
%out_pos1_struct = OpLoad %gl_PerVertex_out %out_pos1_ptr
%out_pos1_vec = OpCompositeInsert %gl_PerVertex_out %new_pos1 %out_pos1_struct 0
OpStore %out_pos1_ptr %out_pos1_vec

%color1_ptr = OpAccessChain %ptr_in_color %color %int_1
%color1 = OpLoad %vec3 %color1_ptr
OpStore %outColor_var %color1
OpEmitVertex

; Emit vertex 2
%out_pos2_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
%out_pos2_struct = OpLoad %gl_PerVertex_out %out_pos2_ptr
%out_pos2_vec = OpCompositeInsert %gl_PerVertex_out %new_pos2 %out_pos2_struct 0
OpStore %out_pos2_ptr %out_pos2_vec

%color2_ptr = OpAccessChain %ptr_in_color %color %int_2
%color2 = OpLoad %vec3 %color2_ptr
OpStore %outColor_var %color2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd

; Entry point and execution modes
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor_var %ubo
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
```
