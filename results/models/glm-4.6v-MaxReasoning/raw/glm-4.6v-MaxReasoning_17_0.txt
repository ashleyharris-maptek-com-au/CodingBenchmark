using System;
using System.Collections.Generic;
using System.Linq;

public class BinPacking3D
{
    public static void Main()
    {
        // Read input
        var input = Console.ReadLine().Split();
        int N = int.Parse(input[0]);
        int W = int.Parse(input[1]);
        int H = int.Parse(input[2]);
        int D = int.Parse(input[3]);
        
        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            var boxInput = Console.ReadLine().Split();
            int w = int.Parse(boxInput[0]);
            int h = int.Parse(boxInput[1]);
            int d = int.Parse(boxInput[2]);
            boxes.Add(new Box(i, w, h, d));
        }
        
        // Sort boxes by volume (largest first)
        boxes.Sort((a, b) => b.Volume.CompareTo(a.Volume));
        
        // Initialize container
        var container = new Container(W, H, D);
        var packedBoxes = new List<PackedBox>();
        
        // Try to pack each box
        foreach (var box in boxes)
        {
            var bestPosition = FindBestPosition(container, box);
            if (bestPosition != null)
            {
                container.PlaceBox(box, bestPosition.Value);
                packedBoxes.Add(new PackedBox(box.Index, bestPosition.Value));
            }
        }
        
        // Output results
        Console.WriteLine(packedBoxes.Count);
        foreach (var packedBox in packedBoxes.OrderBy(pb => pb.Index))
        {
            Console.WriteLine($"{packedBox.Index} {packedBox.X} {packedBox.Y} {packedBox.Z}");
        }
    }
    
    private static Position? FindBestPosition(Container container, Box box)
    {
        // Try all orientations
        var orientations = new[] {
            new Box(box.Index, box.W, box.H, box.D),
            new Box(box.Index, box.W, box.D, box.H),
            new Box(box.Index, box.H, box.W, box.D),
            new Box(box.Index, box.H, box.D, box.W),
            new Box(box.Index, box.D, box.W, box.H),
            new Box(box.Index, box.D, box.H, box.W)
        };
        
        Position? bestPosition = null;
        double bestScore = double.MinValue;
        
        foreach (var orientedBox in orientations)
        {
            var position = container.FindPosition(orientedBox);
            if (position != null)
            {
                // Score based on how well it fits and position preference
                double score = CalculateScore(container, orientedBox, position.Value);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestPosition = position;
                }
            }
        }
        
        return bestPosition;
    }
    
    private static double CalculateScore(Container container, Box box, Position position)
    {
        // Simple scoring: prefer bottom-left-back corner
        double score = -position.X - position.Y - position.Z;
        
        // Bonus for fitting well
        double wastedSpace = container.GetWastedSpace(box, position);
        score -= wastedSpace * 0.1; // Small penalty for wasted space
        
        return score;
    }
}

public class Box
{
    public int Index { get; }
    public int W { get; }
    public int H { get; }
    public int D { get; }
    public int Volume => W * H * D;
    
    public Box(int index, int w, int h, int d)
    {
        Index = index;
        W = w;
        H = h;
        D = d;
    }
}

public class PackedBox
{
    public int Index { get; }
    public int X { get; }
    public int Y { get; }
    public int Z { get; }
    
    public PackedBox(int index, Position position)
    {
        Index = index;
        X = position.X;
        Y = position.Y;
        Z = position.Z;
    }
}

public class Position
{
    public int X { get; }
    public int Y { get; }
    public int Z { get; }
    
    public Position(int x, int y, int z)
    {
        X = x;
        Y = y;
        Z = z;
    }
}

public class Container
{
    public int W { get; }
    public int H { get; }
    public int D { get; }
    public List<FreeSpace> FreeSpaces { get; }
    
    public Container(int w, int h, int d)
    {
        W = w;
        H = h;
        D = d;
        FreeSpaces = new List<FreeSpace> { new FreeSpace(0, 0, 0, w, h, d) };
    }
    
    public void PlaceBox(Box box, Position position)
    {
        // Remove the space occupied by the box and add new free spaces
        var newFreeSpaces = new List<FreeSpace>();
        
        foreach (var space in FreeSpaces)
        {
            if (space.Overlaps(position, box.W, box.H, box.D))
            {
                // Split the space into smaller spaces
                newFreeSpaces.AddRange(space.Split(position, box.W, box.H, box.D));
            }
            else
            {
                newFreeSpaces.Add(space);
            }
        }
        
        FreeSpaces = newFreeSpaces;
    }
    
    public Position? FindPosition(Box box)
    {
        // Simple: find the first space that can fit the box
        foreach (var space in FreeSpaces)
        {
            if (space.CanFit(box.W, box.H, box.D))
            {
                return new Position(space.X, space.Y, space.Z);
            }
        }
        return null;
    }
    
    public double GetWastedSpace(Box box, Position position)
    {
        // Calculate wasted space for this placement
        double boxVolume = box.W * box.H * box.D;
        return 0; // Simplified for now
    }
}

public class FreeSpace
{
    public int X { get; }
    public int Y { get; }
    public int Z { get; }
    public int W { get; }
    public int H { get; }
    public int D { get; }
    
    public FreeSpace(int x, int y, int z, int w, int h, int d)
    {
        X = x;
        Y = y;
        Z = z;
        W = w;
        H = h;
        D = d;
    }
    
    public bool CanFit(int boxW, int boxH, int boxD)
    {
        return W >= boxW && H >= boxH && D >= boxD;
    }
    
    public bool Overlaps(Position position, int boxW, int boxH, int boxD)
    {
        return position.X < X + W && position.X + boxW > X &&
               position.Y < Y + H && position.Y + boxH > Y &&
               position.Z < Z + D && position.Z + boxD > Z;
    }
    
    public List<FreeSpace> Split(Position position, int boxW, int boxH, int boxD)
    {
        var newSpaces = new List<FreeSpace>();
        
        // Space to the right
        if (position.X + boxW < X + W)
        {
            newSpaces.Add(new FreeSpace(position.X + boxW, Y, Z, X + W - (position.X + boxW), H, D));
        }
        
        // Space above
        if (position.Y + boxH < Y + H)
        {
            newSpaces.Add(new FreeSpace(X, position.Y + boxH, Z, boxW, Y + H - (position.Y + boxH), D));
        }
        
        // Space in front
        if (position.Z + boxD < Z + D)
        {
            newSpaces.Add(new FreeSpace(X, Y, position.Z + boxD, W, H, Z + D - (position.Z + boxD)));
        }
        
        return newSpaces;
    }
}
