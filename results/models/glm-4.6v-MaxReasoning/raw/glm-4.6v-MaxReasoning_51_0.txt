# PID Controller class
class PIDController:
    def __init__(self, kp, ki, kd, integral_limit):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.integral = 0
        self.last_error = 0
        self.integral_limit = integral_limit
        
    def update(self, error, dt):
        self.integral += error * dt
        self.integral = max(-self.integral_limit, min(self.integral_limit, self.integral))
        
        derivative = (error - self.last_error) / dt if dt > 0 else 0
        self.last_error = error
        
        return self.kp * error + self.ki * self.integral + self.kd * derivative

def get_reliable_altitude(state):
    # Get all available altitude measurements
    baro_alts = []
    radar_alts = []
    gps_alts = []
    irs_alts = []
    
    # Barometric altitudes
    for i in range(1, 4):
        if state.get(f'baro_alt_{i}_valid', False):
            baro_alts.append(state.get(f'baro_alt_{i}_ft', 0))
    
    # Radar altitudes
    for i in range(1, 4):
        if state.get(f'radar_alt_{i}_valid', False):
            radar_alts.append(state.get(f'radar_alt_{i}_ft', 0))
    
    # GPS altitudes
    for i in range(1, 4):
        if state.get(f'gps_{i}_valid', False):
            gps_alts.append(state.get(f'gps_{i}_alt_ft', 0))
    
    # IRS altitudes (derived from pressure)
    for i in range(1, 4):
        if state.get(f'irs_{i}_valid', False):
            # IRS doesn't directly provide altitude, but we can use vertical speed and time
            # For simplicity, we'll assume it's similar to barometric
            irs_alts.append(state.get(f'irs_{i}_vs_fpm', 0) * 0.00328)  # Convert fpm to ft/s
    
    # Determine the most reliable altitude
    altitudes = []
    valid_flags = []
    
    # Add barometric altitudes
    for alt in baro_alts:
        altitudes.append(alt)
        valid_flags.append(True)
    
    # Add radar altitudes (more reliable at low altitude)
    for alt in radar_alts:
        altitudes.append(alt)
        valid_flags.append(True)
    
    # Add GPS altitudes
    for alt in gps_alts:
        altitudes.append(alt)
        valid_flags.append(True)
    
    # Add IRS altitudes
    for alt in irs_alts:
        altitudes.append(alt)
        valid_flags.append(True)
    
    if not altitudes:
        return 0  # Default if no altitude data
    
    # Simple voting: choose the median of valid altitudes
    valid_altitudes = [alt for alt, valid in zip(altitudes, valid_flags) if valid]
    if not valid_altitudes:
        return 0
    
    return np.median(valid_altitudes)

def autopilot_step(state: dict, dt: float) -> dict:
    # Initialize autopilot state if not exists
    if not hasattr(autopilot_step, "autopilot_state"):
        autopilot_step.autopilot_state = AutopilotState()
    
    ap_state = autopilot_step.autopilot_state
    
    # Get reliable sensor data through fusion
    alt = get_reliable_altitude(state)
    ias = get_reliable_ias(state)
    hdg = get_reliable_hdg(state)
    roll = get_reliable_roll(state)
    pitch = get_reliable_pitch(state)
    yaw_rate = get_reliable_yaw_rate(state)
    vs = get_reliable_vs(state)
    aoa = get_reliable_aoa(state)
    
    # Check for special situations
    ap_state.stall_warning = aoa > 15  # Example threshold
    ap_state.asymmetric_thrust = (state.get('engine_1_running', False) != 
                               state.get('engine_2_running', False))
    
    # Check for Vne (Velocity Never Exceed)
    vne_ias = 365  # knots
    if ias > vne_ias * 0.9:  # Warning at 90% of Vne
        ap_state.vne_warning = True
    else:
        ap_state.vne_warning = False
    
    # Determine flight mode based on autopilot targets
    if state.get('ap_engaged', False):
        if state.get('at_engaged', False):
            ap_state.mode = "SPEED_HOLD"
        elif abs(state.get('target_vs_fpm', 0)) > 100:
            ap_state.mode = "VS_HOLD"
        elif abs(state.get('target_hdg_deg', 0)) > 1:
            ap_state.mode = "HDG_HOLD"
        else:
            ap_state.mode = "ALT_HOLD"
    
    # Calculate control outputs based on mode
    controls = {}
    
    if ap_state.mode == "ALT_HOLD":
        # Altitude hold logic
        alt_error = state.get('target_alt_ft', alt) - alt
        vs_command = ap_state.alt_pid.update(alt_error, dt)
        controls['elevator'] = vs_command * 0.5  # Convert vertical speed to elevator
        
    elif ap_state.mode == "VS_HOLD":
        # Vertical speed hold logic
        vs_error = state.get('target_vs_fpm', vs) - vs
        controls['elevator'] = ap_state.vs_pid.update(vs_error, dt)
        
    elif ap_state.mode == "HDG_HOLD":
        # Heading hold logic
        hdg_error = normalize_angle(state.get('target_hdg_deg', hdg) - hdg)
        roll_command = ap_state.hdg_pid.update(hdg_error, dt)
        controls['aileron'] = roll_command
        
    elif ap_state.mode == "SPEED_HOLD":
        # Speed hold logic with auto-throttle
        ias_error = state.get('target_ias_kt', ias) - ias
        throttle_command = ap_state.ias_pid.update(ias_error, dt)
        
        # Apply throttle to both engines
        controls['throttle_1'] = max(0, min(1, throttle_command))
        controls['throttle_2'] = max(0, min(1, throttle_command))
        
        # Adjust elevator for speed
        elevator_command = ap_state.ias_pid.update(ias_error, dt) * 0.3
        controls['elevator'] = elevator_command
    
    # Yaw damper
    yaw_error = -yaw_rate  # Want to reduce yaw rate
    yaw_command = ap_state.yaw_damper.update(yaw_error, dt)
    controls['rudder'] = yaw_command * 0.3  # Scale down for stability
    
    # Auto-trim
    if abs(controls.get('elevator', 0)) < 0.1:  # If elevator is near neutral
        # Gradually adjust trim to reduce control input
        trim_adjustment = controls.get('elevator', 0) * ap_state.trim_rate * dt
        ap_state.elevator_trim += trim_adjustment
        controls['elevator_trim'] = ap_state.elevator_trim
    
    # Handle takeoff and landing
    if state.get('gear_handle_down', False) and state.get('weight_on_wheels_nose', False):
        # Landing mode
        ap_state.landing_mode = True
        # Deploy flaps
        controls['flaps'] = 15  # Example flap setting for landing
    else:
        ap_state.landing_mode = False
    
    # Stall prevention
    if ap_state.stall_warning:
        # Reduce angle of attack by pitching down
        controls['elevator'] = max(controls.get('elevator', 0) - 0.1, -1.0)
    
    # Asymmetric thrust compensation
    if ap_state.asymmetric_thrust:
        # Apply rudder to counter asymmetric thrust
        if state.get('engine_1_running', False):
            # Engine 1 is running, engine 2 is not
            controls['rudder'] = 0.2  # Rudder right
        else:
            # Engine 2 is running, engine 1 is not
            controls['rudder'] = -0.2  # Rudder left
    
    # Vne warning - reduce thrust
    if ap_state.vne_warning:
        if 'throttle_1' in controls:
            controls['throttle_1'] *= 0.8
        if 'throttle_2' in controls:
            controls['throttle_2'] *= 0.8
    
    # Ensure all control outputs are within limits
    for key in ['elevator', 'aileron', 'rudder', 'throttle_1', 'throttle_2', 'elevator_trim', 'flaps', 'speedbrake', 'gear']:
        if key in controls:
            if key == 'flaps':
                controls[key] = max(0, min(40, controls[key]))
            elif key == 'gear':
                controls[key] = 1 if controls[key] > 0.5 else 0
            else:
                controls[key] = max(-1.0, min(1.0, controls[key]))
    
    return controls

# Helper function to normalize angles
def normalize_angle(angle):
    while angle > 180:
        angle -= 360
    while angle < -180:
        angle += 360
    return angle
