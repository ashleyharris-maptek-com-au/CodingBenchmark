
```spirv
; SPIR-V Assembly for a Geometry Shader
; Target: Vulkan 1.0
; Input: InputLinesAdjacency (4 vertices)
; Output: OutputTriangleStrip (10 vertices max)

; Capabilities
OpCapability Geometry
OpCapability Shader

; Memory Model
OpMemoryModel Logical GLSL450

; Debug information (optional but good practice)
OpString %str_0 "main"
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %ubo "UBO"

; Types
%v4float = OpTypeFloat 32
%v3float = OpTypeVector %v4float 3
%v2float = OpTypeVector %v4float 2
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%bool = OpTypeBool
%void = OpTypeVoid
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_worldPos = OpTypePointer Input %v3float
%ptr_in_normal = OpTypePointer Input %v3float
%ptr_in_color = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_color = OpTypePointer Output %v3float
%ptr_ubo = OpTypePointer Uniform %UBOStruct
%mat4 = OpTypeMatrix %v4float 4
%UBOStruct = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %float
%ptr_ubo = OpTypePointer Uniform %UBOStruct

; Constants
%zero_int = OpConstant %int 0
%one_int = OpConstant %int 1
%two_int = OpConstant %int 2
%three_int = OpConstant %int 3
%four_int = OpConstant %int 4
%zero_float = OpConstant %float 0
%one_float = OpConstant %float 1
%two_float = OpConstant %float 2
%four_float = OpConstant %float 4
%thickness = OpConstant %float 0.02
%inv_num_segments = OpConstant %float 0.25

; gl_PerVertex input block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input arrays (size 4 for InputLinesAdjacency)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %four_int
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_worldPos = OpTypeArray %v3float %four_int
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal = OpTypeArray %v3float %four_int
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_arr_normal Input

%arr_color = OpTypeArray %v3float %four_int
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_arr_color Input

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; UBO
OpDecorate %UBOStruct Block
OpMemberDecorate %UBOStruct 0 Offset 0
OpMemberDecorate %UBOStruct 1 Offset 64
OpMemberDecorate %UBOStruct 2 Offset 128
OpMemberDecorate %UBOStruct 3 Offset 192
OpMemberDecorate %UBOStruct 4 Offset 208
OpMemberDecorate %UBOStruct 5 Offset 224
%ubo = OpVariable %ptr_ubo Uniform

; Entry Point
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo

; Execution Modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Function: main
%main = OpFunction %void None %void_fn
%void_fn = OpTypeFunction %void
%L_main = OpLabel
%num_segments = OpVariable %ptr_in_int Input
%thickness_c = OpVariable %ptr_in_float Input

; Load UBO (not used in this specific task, but accessible as requested)
%model_ptr = OpAccessChain %ptr_mat4 %ubo %zero_int %zero_int
%view_ptr = OpAccessChain %ptr_mat4 %ubo %zero_int %one_int
%proj_ptr = OpAccessChain %ptr_mat4 %ubo %zero_int %two_int
%lightPos_ptr = OpAccessChain %ptr_v3float %ubo %zero_int %three_int
%cameraPos_ptr = OpAccessChain %ptr_v3float %ubo %zero_int %four_int
%params_ptr = OpAccessChain %ptr_float %ubo %zero_int %five_int

; Get input vertices P0, P1, P2, P3
%ptr_P0_pos = OpAccessChain %ptr_in_pv %gl_in %zero_int %zero_int
%P0_pos = OpLoad %v4float %ptr_P0_pos
%ptr_P1_pos = OpAccessChain %ptr_in_pv %gl_in %one_int %zero_int
%P1_pos = OpLoad %v4float %ptr_P1_pos
%ptr_P2_pos = OpAccessChain %ptr_in_pv %gl_in %two_int %zero_int
%P2_pos = OpLoad %v4float %ptr_P2_pos
%ptr_P3_pos = OpAccessChain %ptr_in_pv %gl_in %three_int %zero_int
%P3_pos = OpLoad %v4float %ptr_P3_pos

; Get input colors C1, C2
%ptr_C1 = OpAccessChain %ptr_in_color %color %one_int
%C1 = OpLoad %v3float %ptr_C1
%ptr_C2 = OpAccessChain %ptr_in_color %color %two_int
%C2 = OpLoad %v3float %ptr_C2

; Get input normals N1, N2
%ptr_N1 = OpAccessChain %ptr_in_normal %normal %one_int
%N1 = OpLoad %v3float %ptr_N1
%ptr_N2 = OpAccessChain %ptr_in_normal %normal %two_int
%N2 = OpLoad %v3float %ptr_N2

; Loop setup
OpBranch %L_loop_start
OpLabel %L_loop_start
%loop_i = OpPhi %int %zero_int %L_loop_start %i_next %L_loop_increment

; Calculate interpolation factor t
%t = OpFMul %float %loop_i %inv_num_segments

; Interpolate position (simplified Catmull-Rom)
%pos = OpFMix %v4float %P1_pos %P2_pos %t

; Calculate varying tangent
%P0_xyz = OpVectorShuffle %v3float %P0_pos %P0_pos 0 1 2 -1
%P1_xyz = OpVectorShuffle %v3float %P1_pos %P1_pos 0 1 2 -1
%P2_xyz = OpVectorShuffle %v3float %P2_pos %P2_pos 0 1 2 -1
%P3_xyz = OpVectorShuffle %v3float %P3_pos %P3_pos 0 1 2 -1

%tangent_start = OpFSub %v3float %P2_xyz %P0_xyz
%tangent_end = OpFSub %v3float %P3_xyz %P1_xyz
%tangent = OpFMix %v3float %tangent_start %tangent_end %t
%tangent = OpNormalize %v3float %tangent

; Calculate perpendicular (using interpolated normal for up vector)
%perp_start = OpCross %v3float %tangent_start %N1
%perp_end = OpCross %v3float %tangent_end %N2
%perp = OpFMix %v3float %perp_start %perp_end %t
%perp = OpNormalize %v3float %perp
%perp = OpFMul %v3float %perp %thickness

; Interpolate color
%c = OpFMix %v3float %C1 %C2 %t

; Emit first vertex of the pair (pos + perp)
%pos_plus_perp = OpFAdd %v4float %pos (OpCompositeInsert %v4float %perp %zero_float 3)
OpStore %gl_out %pos_plus_perp
OpStore %outColor %c
OpEmitVertex()

; Emit second vertex of the pair (pos - perp)
%pos_minus_perp = OpFSub %v4float %pos (OpCompositeInsert %v4float %perp %zero_float 3)
OpStore %gl_out %pos_minus_perp
OpStore %outColor %c
OpEmitVertex()

; Loop increment and condition
OpBranch %L_loop_continue
OpLabel %L_loop_continue
%i_next = OpIAdd %int %loop_i %one_int
%i_less_than_num = OpSLessThan %bool %i_next %four_int
OpBranchConditional %i_less_than_num %L_loop_start %L_loop_merge

OpLabel %L_loop_merge
OpBranch %L_loop_end

OpLabel %L_loop_end
OpEndPrimitive
OpReturn
OpFunctionEnd
```
