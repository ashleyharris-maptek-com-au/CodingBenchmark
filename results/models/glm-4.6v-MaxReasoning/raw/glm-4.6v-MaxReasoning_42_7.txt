
OpCapability Geometry
OpMemoryModel Logical GLSL450

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%v3float = OpTypeVector %float 3
%arr_worldPos = OpTypeArray %v3float %uint_3
%ptr_in_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_worldPos Input
OpDecorate %worldPos Location 0

%arr_normal = OpTypeArray %v3float %uint_3
%ptr_in_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_normal Input
OpDecorate %normal Location 1

%arr_color = OpTypeArray %v3float %uint_3
%ptr_in_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_color Input
OpDecorate %color Location 2

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0.0
%float_1_3 = OpConstant %v4float (0.33333334, 0.33333334, 0.33333334, 0.33333334)
%float_0_08 = OpConstant %v3float (0.08, 0.08, 0.08)

%void = OpTypeVoid
%void_func = OpTypeFunction %void
%main = OpFunction %void %void_func
%entry = OpLabel

%ptr_p0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%center = OpFMul %v4float %sum012 %float_1_3

%e1 = OpFSub %v4float %p1 %p0
%e2 = OpFSub %v4float %p2 %p0

%e1_xyz = OpVectorShuffle %v3float %e1 %e1 0 1 2
%e2_xyz = OpVectorShuffle %v3float %e2 %e2 0 1 2

%e1_b = OpVectorShuffle %float %e1_xyz %e1_xyz 1
%e1_c = OpVectorShuffle %float %e1_xyz %e1_xyz 2
%e2_f = OpVectorShuffle %float %e2_xyz %e2_xyz 2
%e2_e = OpVectorShuffle %float %e2_xyz %e2_xyz 1
%cross0 = OpFSub %float (OpFMul %float %e1_b %e2_f) (OpFMul %float %e1_c %e2_e)

%e1_c = OpVectorShuffle %float %e1_xyz %e1_xyz 2
%e1_a = OpVectorShuffle %float %e1_xyz %e1_xyz 0
%e2_d = OpVectorShuffle %float %e2_xyz %e2_xyz 0
%e2_f = OpVectorShuffle %float %e2_xyz %e2_xyz 2
%cross1 = OpFSub %float (OpFMul %float %e1_c %e2_d) (OpFMul %float %e1_a %e2_f)

%e1_a = OpVectorShuffle %float %e1_xyz %e1_xyz 0
%e1_b = OpVectorShuffle %float %e1_xyz %e1_xyz 1
%e2_e = OpVectorShuffle %float %e2_xyz %e2_xyz 1
%e2_d = OpVectorShuffle %float %e2_xyz %e2_xyz 0
%cross2 = OpFSub %float (OpFMul %float %e1_a %e2_e) (OpFMul %float %e1_b %e2_d)

%cross = OpCompositeConstruct %v3float %cross0 %cross1 %cross2
%N = OpNormalize %v3float %cross
%offset = OpFMul %v3float %N %float_0_08

%ptr_c0 = OpAccessChain %ptr_in_color %color %int_0
%c0 = OpLoad %v3float %ptr_c0
%offset_vec4 = OpCompositeConstruct %v4float %offset %offset %offset %float_0
%shifted_pos0 = OpFAdd %v4float %p0 %offset_vec4
%ptr_out_pos0 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %ptr_out_pos0 %shifted_pos0
%ptr_out_color0 = OpAccessChain %ptr_out_color %outColor
OpStore %ptr_out_color0 %c0
OpEmitVertex

%ptr_c1 = OpAccessChain %ptr_in_color %color %int_1
%c1 = OpLoad %v3float %ptr_c1
%shifted_pos1 = OpFAdd %v4float %p1 %offset_vec4
%ptr_out_pos1 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %ptr_out_pos1 %shifted_pos1
%ptr_out_color1 = OpAccessChain %ptr_out_color %outColor
OpStore %ptr_out_color1 %c1
OpEmitVertex

%ptr_c2 = OpAccessChain %ptr_in_color %color %int_2
%c2 = OpLoad %v3float %ptr_c2
%shifted_pos2 = OpFAdd %v4float %p2 %offset_vec4
%ptr_out_pos2 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %ptr_out_pos2 %shifted_pos2
%ptr_out_color2 = OpAccessChain %ptr_out_color %outColor
OpStore %ptr_out_color2 %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
