
```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds access
    if (gid >= N) {
        return;
    }
    
    // Read current body data
    vec3 pos_i = inData[gid * 2].xyz;
    float mass_i = inData[gid * 2].w;
    vec3 vel_i = inData[gid * 2 + 1].xyz;
    
    // Initialize acceleration
    vec3 acc = vec3(0.0);
    
    // Extract simulation parameters
    float G = physics.z;
    float softening = physics.y;
    
    // Compute acceleration from all other bodies
    for (uint j = 0; j < N; ++j) {
        if (j == gid) {
            continue; // Skip self-interaction
        }
        
        vec3 pos_j = inData[j * 2].xyz;
        float mass_j = inData[j * 2].w;
        
        vec3 r = pos_j - pos_i;
        float r2 = dot(r, r) + softening * softening;
        
        // Compute 1 / (r^3) with softening: 1.0 / (r2 * sqrt(r2))
        float inv_r3 = 1.0 / (r2 * sqrt(r2));
        
        // Accumulate gravitational acceleration
        acc += G * mass_j * r * inv_r3;
    }
    
    // Update velocity and position using Euler integration
    float dt = physics.x;
    vel_i += acc * dt;
    pos_i += vel_i * dt;
    
    // Write updated data to output buffer
    outData[gid * 2] = vec4(pos_i, mass_i);       // position + mass
    outData[gid * 2 + 1] = vec4(vel_i, 0.0);      // velocity
}
```
