
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal (it might not be normalized after interpolation)
    vec3 N = normalize(normal);
    
    // Calculate light direction (from fragment to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse component
    float diffuse = max(dot(N, L), 0.0);
    
    // Scale UV coordinates for brick pattern
    vec2 scaledUV = uv * vec2(8.0, 16.0);
    
    // Calculate brick pattern
    float brickU = fract(scaledUV.x);
    float brickV = fract(scaledUV.y);
    
    // Determine if we're in a staggered row
    float rowOffset = floor(scaledUV.y) % 2.0;
    float adjustedU = brickU + rowOffset * 0.5;
    adjustedU = fract(adjustedU); // Keep it in [0,1]
    
    // Define brick and mortar regions
    float brickWidth = 0.95; // 1.0 - 0.05 mortar thickness
    float brickHeight = 0.9;  // 1.0 - 0.1 mortar thickness
    
    // Check if we're in mortar or brick
    bool inMortar = (adjustedU < 0.05) || (adjustedU > brickWidth) || 
                   (brickV < 0.1) || (brickV > brickHeight);
    
    // Base color based on brick or mortar
    vec3 baseColor = inMortar ? vec3(0.7, 0.7, 0.7) : vec3(0.7, 0.2, 0.1);
    
    // Apply Lambertian shading
    float shading = 0.4 + 0.6 * diffuse;
    vec3 finalColor = baseColor * shading;
    
    // Output the final color with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
