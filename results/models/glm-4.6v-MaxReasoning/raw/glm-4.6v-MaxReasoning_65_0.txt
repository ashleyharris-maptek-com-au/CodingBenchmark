declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
    id: number;
    duration: number;
    deps: number[];
    deadline: number;
    weight: number;
}

interface ScheduledTask {
    id: number;
    start: number;
    machine: number;
}

function scheduleMissionControl(data: any): any {
    const machineCount = data.machineCount;
    const tasks: Task[] = data.tasks;
    
    // Build task map and dependency graph
    const taskMap = new Map<number, Task>();
    const graph = new Map<number, number[]>(); // task id -> list of dependent tasks
    const inDegree = new Map<number, number>(); // task id -> number of dependencies
    
    for (const task of tasks) {
        taskMap.set(task.id, task);
        inDegree.set(task.id, task.deps.length);
        graph.set(task.id, []);
    }
    
    for (const task of tasks) {
        for (const depId of task.deps) {
            graph.get(depId)!.push(task.id);
        }
    }
    
    // Topological sort using Kahn's algorithm
    const queue: number[] = [];
    const order: number[] = [];
    
    // Initialize queue with tasks that have no dependencies
    for (const [id, degree] of inDegree) {
        if (degree === 0) {
            queue.push(id);
        }
    }
    
    while (queue.length > 0) {
        // Sort by deadline and weight for better scheduling
        queue.sort((a, b) => {
            const taskA = taskMap.get(a)!;
            const taskB = taskMap.get(b)!;
            if (taskA.deadline !== taskB.deadline) {
                return taskA.deadline - taskB.deadline;
            }
            return taskB.weight - taskA.weight;
        });
        
        const current = queue.shift()!;
        order.push(current);
        
        for (const neighbor of graph.get(current)!) {
            const newDegree = inDegree.get(neighbor)! - 1;
            inDegree.set(neighbor, newDegree);
            if (newDegree === 0) {
                queue.push(neighbor);
            }
        }
    }
    
    // If there are cycles (shouldn't happen with valid input), we'll just use the order we have
    if (order.length !== tasks.length) {
        // Fallback: just use the original order
        return { order: tasks.map(t => t.id) };
    }
    
    // List scheduling with priority based on deadline and weight
    const machineAvailability: number[] = new Array(machineCount).fill(0);
    const schedule: ScheduledTask[] = [];
    
    // Priority queue for ready tasks
    const readyQueue: number[] = [];
    
    // Track task completion times
    const completionTimes = new Map<number, number>();
    
    let currentTime = 0;
    
    // Process tasks in topological order
    for (const taskId of order) {
        const task = taskMap.get(taskId)!;
        
        // Check if dependencies are completed
        let canStart = true;
        for (const depId of task.deps) {
            if (!completionTimes.has(depId)) {
                canStart = false;
                break;
            }
        }
        
        if (canStart) {
            readyQueue.push(taskId);
        } else {
            // This shouldn't happen with a valid topological order
            continue;
        }
    }
    
    // Process ready tasks
    while (readyQueue.length > 0 || schedule.length < tasks.length) {
        // Add tasks that are ready to the queue
        for (const taskId of order) {
            if (readyQueue.includes(taskId)) continue;
            
            const task = taskMap.get(taskId)!;
            let canStart = true;
            for (const depId of task.deps) {
                if (!completionTimes.has(depId)) {
                    canStart = false;
                    break;
                }
            }
            
            if (canStart) {
                readyQueue.push(taskId);
            }
        }
        
        if (readyQueue.length === 0) break;
        
        // Sort ready tasks by priority (deadline, then weight)
        readyQueue.sort((a, b) => {
            const taskA = taskMap.get(a)!;
            const taskB = taskMap.get(b)!;
            if (taskA.deadline !== taskB.deadline) {
                return taskA.deadline - taskB.deadline;
            }
            return taskB.weight - taskA.weight;
        });
        
        const taskId = readyQueue.shift()!;
        const task = taskMap.get(taskId)!;
        
        // Find the earliest available machine
        let minMachine = 0;
        let minTime = machineAvailability[0];
        for (let i = 1; i < machineCount; i++) {
            if (machineAvailability[i] < minTime) {
                minTime = machineAvailability[i];
                minMachine = i;
            }
        }
        
        // Calculate start time (max of machine availability and completion of dependencies)
        let startTime = minTime;
        for (const depId of task.deps) {
            const depCompletion = completionTimes.get(depId)!;
            if (depCompletion > startTime) {
                startTime = depCompletion;
            }
        }
        
        // Schedule the task
        const endTime = startTime + task.duration;
        schedule.push({
            id: taskId,
            start: startTime,
            machine: minMachine
        });
        
        // Update machine availability and completion times
        machineAvailability[minMachine] = endTime;
        completionTimes.set(taskId, endTime);
    }
    
    // Calculate makespan and weighted lateness
    let makespan = 0;
    let weightedLateness = 0;
    
    for (const scheduled of schedule) {
        const task = taskMap.get(scheduled.id)!;
        const completionTime = scheduled.start + task.duration;
        makespan = Math.max(makespan, completionTime);
        const lateness = Math.max(0, completionTime - task.deadline);
        weightedLateness += lateness * task.weight;
    }
    
    // Return the schedule
    return { schedule };
}

const result = scheduleMissionControl(data);
console.log(JSON.stringify(result));
