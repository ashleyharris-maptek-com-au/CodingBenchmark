interface Drone {
  id: number;
  x: number;
  y: number;
}

interface Input {
  probability: number[][];
  drones: Drone[];
  maxStepsPerDrone: number;
}

interface Output {
  paths: number[][][];
}

class PathFinder {
  private grid: number[][];
  private rows: number;
  private cols: number;
  private drones: Drone[];
  private maxSteps: number;
  private visited: Set<string>;
  private paths: number[][][];
  private gridScore: number[][];

  constructor(grid: number[][], drones: Drone[], maxSteps: number) {
    this.grid = grid;
    this.rows = grid.length;
    this.cols = grid[0].length;
    this.drones = drones;
    this.maxSteps = maxSteps;
    this.visited = new Set();
    this.paths = [];
    this.gridScore = Array(this.rows).fill(null).map(() => Array(this.cols).fill(0));
    
    // Initialize grid score (higher values are more valuable)
    for (let i = 0; i < this.rows; i++) {
      for (let j = 0; j < this.cols; j++) {
        if (this.grid[i][j] > 0) {
          this.gridScore[i][j] = this.grid[i][j];
        }
      }
    }
  }

  public findPaths(): number[][][] {
    // Sort drones by starting position (optional optimization)
    this.drones.sort((a, b) => {
      if (a.x !== b.x) return a.x - b.x;
      return a.y - b.y;
    });

    for (const drone of this.drones) {
      const path = this.findPathForDrone(drone);
      this.paths.push(path);
    }

    return this.paths;
  }

  private findPathForDrone(drone: Drone): number[][] {
    const path: number[][] = [];
    let currentX = drone.x;
    let currentY = drone.y;
    let stepsRemaining = this.maxSteps;

    path.push([currentX, currentY]);

    while (stepsRemaining > 0) {
      const nextMove = this.findBestMove(currentX, currentY);
      
      if (!nextMove) break; // No valid moves available

      currentX = nextMove.x;
      currentY = nextMove.y;
      path.push([currentX, currentY]);
      stepsRemaining--;

      // Mark this cell as visited
      this.visited.add(`${currentX},${currentY}`);
    }

    return path;
  }

  private findBestMove(x: number, y: number): { x: number; y: number } | null {
    const moves = [
      { x: x - 1, y }, // up
      { x: x + 1, y }, // down
      { x, y: y - 1 }, // left
      { x, y: y + 1 }, // right
      { x, y }        // stay
    ];

    let bestMove: { x: number; y: number } | null = null;
    let bestScore = -Infinity;

    for (const move of moves) {
      // Check if move is valid
      if (!this.isValidMove(move.x, move.y)) continue;

      // Calculate score for this move
      const score = this.calculateMoveScore(move.x, move.y);

      if (score > bestScore) {
        bestScore = score;
        bestMove = move;
      }
    }

    return bestMove;
  }

  private isValidMove(x: number, y: number): boolean {
    // Check grid boundaries
    if (x < 0 || x >= this.rows || y < 0 || y >= this.cols) return false;
    
    // Check for obstacles
    if (this.grid[x][y] === -1) return false;
    
    // Check if already visited (optional - can be relaxed for better coverage)
    // if (this.visited.has(`${x},${y}`)) return false;

    return true;
  }

  private calculateMoveScore(x: number, y: number): number {
    let score = this.gridScore[x][y];

    // Add bonus for unvisited cells
    if (!this.visited.has(`${x},${y}`)) {
      score += this.gridScore[x][y] * 0.5;
    }

    // Add distance penalty (prefer closer moves)
    const distanceFromStart = Math.abs(x - this.drones[0].x) + Math.abs(y - this.drones[0].y);
    score -= distanceFromStart * 0.1;

    return score;
  }
}

function main(): void {
  // Read input from stdin
  const fs = require('fs');
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Input = JSON.parse(input);

  const { probability, drones, maxStepsPerDrone } = data;

  // Create path finder and find paths
  const pathFinder = new PathFinder(probability, drones, maxStepsPerDrone);
  const paths = pathFinder.findPaths();

  // Prepare output
  const output: Output = { paths };

  // Write output to stdout
  console.log(JSON.stringify(output));
}

// Run the main function
main();
