import { createRequire } from 'module';
const require = createRequire(import.meta.url);
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
    id: string;
    x: number;
    y: number;
}

interface Depot {
    x: number;
    y: number;
}

interface InputData {
    depot: Depot;
    truckCount: number;
    packages: Package[];
}

interface OutputData {
    routes: string[][];
}

// Calculate Euclidean distance between two points
function distance(p1: {x: number, y: number}, p2: {x: number, y: number}): number {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
}

// K-means clustering to group packages
function kMeansClustering(packages: Package[], k: number, depot: Depot): Package[][] {
    // Initialize centroids with random packages
    let centroids = packages.slice(0, k).map(p => ({x: p.x, y: p.y}));
    
    let maxIterations = 100;
    let iterations = 0;
    
    while (iterations < maxIterations) {
        // Assign packages to nearest centroid
        let clusters: Package[][] = Array(k).fill(null).map(() => []);
        
        for (const pkg of packages) {
            let minDist = Infinity;
            let clusterIndex = 0;
            
            for (let i = 0; i < k; i++) {
                const dist = distance(pkg, centroids[i]);
                if (dist < minDist) {
                    minDist = dist;
                    clusterIndex = i;
                }
            }
            
            clusters[clusterIndex].push(pkg);
        }
        
        // Update centroids
        let newCentroids = centroids.map((centroid, i) => {
            if (clusters[i].length === 0) return centroid;
            
            let sumX = 0, sumY = 0;
            for (const pkg of clusters[i]) {
                sumX += pkg.x;
                sumY += pkg.y;
            }
            
            return {
                x: sumX / clusters[i].length,
                y: sumY / clusters[i].length
            };
        });
        
        // Check for convergence
        let converged = true;
        for (let i = 0; i < k; i++) {
            if (Math.abs(newCentroids[i].x - centroids[i].x) > 0.001 || 
                Math.abs(newCentroids[i].y - centroids[i].y) > 0.001) {
                converged = false;
                break;
            }
        }
        
        centroids = newCentroids;
        iterations++;
        
        if (converged) break;
    }
    
    return clusters;
}

// Solve TSP using nearest neighbor heuristic
function solveTSP(packages: Package[], depot: Depot): string[] {
    if (packages.length === 0) return [];
    
    let unvisited = [...packages];
    let route: string[] = [];
    let current = depot;
    
    while (unvisited.length > 0) {
        let nearestIndex = 0;
        let minDist = Infinity;
        
        for (let i = 0; i < unvisited.length; i++) {
            const dist = distance(current, unvisited[i]);
            if (dist < minDist) {
                minDist = dist;
                nearestIndex = i;
            }
        }
        
        const next = unvisited.splice(nearestIndex, 1)[0];
        route.push(next.id);
        current = next;
    }
    
    return route;
}

// Main function to plan routes
function planRoutes(input: InputData): OutputData {
    const { depot, truckCount, packages } = input;
    
    // If there are fewer packages than trucks, we can assign one package per truck
    if (packages.length <= truckCount) {
        const routes: string[][] = packages.map(pkg => [pkg.id]);
        // Add empty routes if needed
        while (routes.length < truckCount) {
            routes.push([]);
        }
        return { routes };
    }
    
    // Cluster packages
    const clusters = kMeansClustering(packages, truckCount, depot);
    
    // Solve TSP for each cluster
    const routes: string[][] = clusters.map(cluster => {
        if (cluster.length === 0) return [];
        return solveTSP(cluster, depot);
    });
    
    return { routes };
}

// Run the planner
const output = planRoutes(data);
console.log(JSON.stringify(output));
