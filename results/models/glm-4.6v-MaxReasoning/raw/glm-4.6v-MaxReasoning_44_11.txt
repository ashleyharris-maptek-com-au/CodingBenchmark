#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Calculate cell coordinates for 6x6 grid
    vec2 cellSize = 1.0 / 6.0;
    vec2 cellCoord = floor(uv / cellSize);
    vec2 cellCenter = (cellCoord + 0.5) * cellSize;
    
    // Calculate distance from UV to cell center
    float dist = distance(uv, cellCenter);
    float radius = 0.3 * cellSize.x; // 0.3 in cell space
    
    // Determine base color (yellow for dots, dark blue for background)
    vec3 baseColor = dist < radius ? vec3(1.0, 0.8, 0.0) : vec3(0.1, 0.1, 0.3);
    
    // Calculate lighting
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    vec3 viewDir = normalize(ubo.cameraPos.xyz - worldPos);
    vec3 norm = normalize(normal);
    
    // Lambertian diffuse
    float diffuse = max(dot(norm, lightDir), 0.0);
    
    // Apply shading factor
    vec3 shadedColor = baseColor * (0.3 + 0.7 * diffuse);
    
    // Output final color with alpha 1.0
    outColor = vec4(shadedColor, 1.0);
}
