#include <iostream>

// Configuration
constexpr double GRAVITY = 1.62; // m/s^2
constexpr double MAX_THRUST = 1.0; // normalized
constexpr double FUEL_RATE = 1.0; // units per second at full thrust
constexpr double ANGULAR_ACCEL = 2.0; // rad/s^2 per command unit
constexpr double ANGULAR_DAMPING = 0.95;
constexpr double MAX_LANDING_SPEED = 3.0; // m/s
constexpr double LANDING_RADIUS_FACTOR = 0.02;
constexpr double TARGET_RADIUS = 5.0; // minimum radius

// PID Controller parameters
struct PIDParams {
    double kp, ki, kd;
    double integral;
    double prev_error;
    double output;
};

class PIDController {
private:
    PIDParams params;
    double min_output, max_output;
    bool integral_clamp;

public:
    PIDController(double kp, double ki, double kd, double min_out, double max_out, bool clamp = true)
        : params{kp, ki, kd, 0.0, 0.0, 0.0}, min_output(min_out), max_output(max_out), integral_clamp(clamp) {}

    double update(double error, double dt) {
        double proportional = params.kp * error;
        params.integral += params.ki * error * dt;
        
        if (integral_clamp) {
            params.integral = std::clamp(params.integral, min_output, max_output);
        }
        
        double derivative = params.kd * (error - params.prev_error) / dt;
        params.prev_error = error;
        
        params.output = proportional + params.integral + derivative;
        params.output = std::clamp(params.output, min_output, max_output);
        
        return params.output;
    }

    void reset() {
        params.integral = 0.0;
        params.prev_error = 0.0;
        params.output = 0.0;
    }
};

// Terrain data structure
struct TerrainCell {
    uint16_t ground_rock_top;
    uint16_t floating_rock_bottom;
    uint16_t floating_rock_top;
};

class Terrain {
private:
    std::vector<TerrainCell> cells;
    int width, depth, height;
    double cell_size_x, cell_size_y;

public:
    Terrain(int w, int d, int h) : width(w), depth(d), height(h), cells(w * d) {}

    void load_from_file(const std::string& file_path) {
        std::ifstream file(file_path, std::ios::binary);
        if (!file) {
            throw std::runtime_error("Failed to open terrain file");
        }

        file.seekg(0, std::ios::end);
        size_t file_size = file.tellg();
        file.seekg(0, std::ios::beg);

        if (file_size != width * depth * 6) {
            throw std::runtime_error("Terrain file size mismatch");
        }

        for (int i = 0; i < width * depth; ++i) {
            uint16_t a, b, c;
            file.read(reinterpret_cast<char*>(&a), sizeof(uint16_t));
            file.read(reinterpret_cast<char*>(&b), sizeof(uint16_t));
            file.read(reinterpret_cast<char*>(&c), sizeof(uint16_t));
            
            cells[i].ground_rock_top = a;
            cells[i].floating_rock_bottom = b;
            cells[i].floating_rock_top = c;
        }
    }

    bool is_solid(double x, double y, double z) const {
        if (x < 0 || x >= width || y < 0 || y >= depth || z < 0 || z >= height) {
            return true;
        }

        int ix = static_cast<int>(x);
        int iy = static_cast<int>(y);
        int idx = iy * width + ix;
        
        if (idx < 0 || idx >= cells.size()) {
            return true;
        }

        const TerrainCell& cell = cells[idx];
        
        if (z < cell.ground_rock_top) {
            return true;
        }
        if (z >= cell.floating_rock_bottom && z < cell.floating_rock_top) {
            return true;
        }
        
        return false;
    }

    double get_ground_height(double x, double y) const {
        if (x < 0 || x >= width || y < 0 || y >= depth) {
            return 0.0;
        }

        int ix = static_cast<int>(x);
        int iy = static_cast<int>(y);
        int idx = iy * width + ix;
        
        if (idx < 0 || idx >= cells.size()) {
            return 0.0;
        }

        return cells[idx].ground_rock_top;
    }
};

// Lander state
struct LanderState {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel_remaining;
    double time;

    void update(double dt, double thrust, double pitch_cmd, double yaw_cmd, double roll_cmd, double max_thrust) {
        // Update angular rates
        pitch_rate += pitch_cmd * ANGULAR_ACCEL * dt;
        yaw_rate += yaw_cmd * ANGULAR_ACCEL * dt;
        roll_rate += roll_cmd * ANGULAR_ACCEL * dt;
        
        // Apply damping
        pitch_rate *= ANGULAR_DAMPING;
        yaw_rate *= ANGULAR_DAMPING;
        roll_rate *= ANGULAR_DAMPING;
        
        // Update angles
        pitch += pitch_rate * dt;
        yaw += yaw_rate * dt;
        roll += roll_rate * dt;
        
        // Clamp pitch
        pitch = std::clamp(pitch, -M_PI/2, M_PI/2);
        
        // Calculate thrust direction in world coordinates
        double thrust_z = thrust * max_thrust * std::cos(pitch);
        double thrust_x = -thrust * max_thrust * std::sin(pitch) * std::cos(yaw);
        double thrust_y = -thrust * max_thrust * std::sin(pitch) * std::sin(yaw);
        
        // Update velocity
        vx += thrust_x * dt;
        vy += thrust_y * dt;
        vz += (thrust_z - GRAVITY) * dt;
        
        // Update position
        x += vx * dt;
        y += vy * dt;
        z += vz * dt;
        
        // Update fuel
        fuel_remaining -= thrust * max_thrust * FUEL_RATE * dt;
        fuel_remaining = std::max(0.0, fuel_remaining);
        
        time += dt;
    }
};

class LanderController {
private:
    // PID controllers
    PIDController pos_x_ctrl, pos_y_ctrl, pos_z_ctrl;
    PIDController vel_x_ctrl, vel_y_ctrl, vel_z_ctrl;
    PIDController pitch_ctrl, yaw_ctrl, roll_ctrl;
    
    // State tracking
    LanderState current_state;
    LanderState target_state;
    Terrain terrain;
    double max_thrust;
    double fuel;
    double target_radius;
    bool landing_mode;
    double approach_speed;
    
    // Timing
    std::chrono::steady_clock::time_point last_update;
    double dt;

public:
    LanderController(int width, int depth, int height, double gravity, double max_thrust, double fuel,
                   double start_x, double start_y, double start_z,
                   double target_x, double target_y, double target_z,
                   const std::string& terrain_file)
        : pos_x_ctrl(0.5, 0.1, 0.2, -1.0, 1.0),
          pos_y_ctrl(0.5, 0.1, 0.2, -1.0, 1.0),
          pos_z_ctrl(1.0, 0.2, 0.3, -1.0, 1.0),
          vel_x_ctrl(1.0, 0.0, 0.5, -1.0, 1.0),
          vel_y_ctrl(1.0, 0.0, 0.5, -1.0, 1.0),
          vel_z_ctrl(2.0, 0.0, 1.0, -1.0, 1.0),
          pitch_ctrl(1.0, 0.1, 0.5, -1.0, 1.0),
          yaw_ctrl(1.0, 0.1, 0.5, -1.0, 1.0),
          roll_ctrl(0.5, 0.0, 0.2, -1.0, 1.0),
          terrain(width, depth, height),
          max_thrust(max_thrust),
          fuel(fuel),
          target_radius(std::max(TARGET_RADIUS, std::min(width, depth) * LANDING_RADIUS_FACTOR)),
          landing_mode(false),
          approach_speed(5.0) {
        
        // Initialize state
        current_state = {start_x, start_y, start_z, 0, 0, 0, 0, 0, 0, 0, 0, 0, fuel, 0};
        target_state = {target_x, target_y, target_z, 0, 0, 0, 0, 0, 0, 0, 0, 0, fuel, 0};
        
        // Load terrain in background thread
        std::thread terrain_loader([this, terrain_file]() {
            try {
                terrain.load_from_file(terrain_file);
            } catch (const std::exception& e) {
                std::cerr << "Terrain loading error: " << e.what() << std::endl;
            }
        });
        terrain_loader.detach();
        
        last_update = std::chrono::steady_clock::now();
    }

    void update_state(double x, double y, double z, double vx, double vy, double vz,
                    double pitch, double yaw, double roll,
                    double pitch_rate, double yaw_rate, double roll_rate,
                    double fuel_remaining) {
        current_state = {x, y, z, vx, vy, vz, pitch, yaw, roll, 
                       pitch_rate, yaw_rate, roll_rate, fuel_remaining, current_state.time};
        
        dt = std::chrono::duration<double>(std::chrono::steady_clock::now() - last_update).count();
        last_update = std::chrono::steady_clock::now();
    }

    bool is_landed() const {
        double dist_to_target = std::sqrt(
            std::pow(current_state.x - target_state.x, 2) +
            std::pow(current_state.y - target_state.y, 2)
        );
        
        double speed = std::sqrt(
            std::pow(current_state.vx, 2) +
            std::pow(current_state.vy, 2) +
            std::pow(current_state.vz, 2)
        );
        
        return dist_to_target <= target_radius && speed < MAX_LANDING_SPEED && 
               current_state.z <= target_state.z + 2.0;
    }

    bool is_crashed() const {
        return current_state.fuel <= 0 || 
               terrain.is_solid(current_state.x, current_state.y, current_state.z);
    }

    std::tuple<double, double, double, double> get_commands() {
        if (is_landed()) {
            return {0.0, 0.0, 0.0, 0.0};
        }
        
        if (is_crashed()) {
            return {0.0, 0.0, 0.0, 0.0};
        }
        
        // Check if we should enter landing mode
        double dist_to_target = std::sqrt(
            std::pow(current_state.x - target_state.x, 2) +
            std::pow(current_state.y - target_state.y, 2)
        );
        
        double height = current_state.z - target_state.z;
        
        if (height < 100.0 && dist_to_target < target_radius * 2) {
            landing_mode = true;
        }
        
        double thrust = 0.0;
        double pitch_cmd = 0.0;
        double yaw_cmd = 0.0;
        double roll_cmd = 0.0;
        
        if (landing_mode) {
            // Landing mode: controlled descent
            double target_height = 2.0;
            double target_speed = 1.0;
            
            // Position control
            double pos_x_error = target_state.x - current_state.x;
            double pos_y_error = target_state.y - current_state.y;
            double pos_z_error = target_height - height;
            
            // Velocity control
            double vel_x_error = -current_state.vx;
            double vel_y_error = -current_state.vy;
            double vel_z_error = target_speed - current_state.vz;
            
            // PID outputs
            double pos_x_out = pos_x_ctrl.update(pos_x_error, dt);
            double pos_y_out = pos_y_ctrl.update(pos_y_error, dt);
            double pos_z_out = pos_z_ctrl.update(pos_z_error, dt);
            
            double vel_x_out = vel_x_ctrl.update(vel_x_error, dt);
            double vel_y_out = vel_y_ctrl.update(vel_y_error, dt);
            double vel_z_out = vel_z_ctrl.update(vel_z_error, dt);
            
            // Combine position and velocity control
            pitch_cmd = pos_x_out + vel_x_out;
            yaw_cmd = pos_y_out + vel_y_out;
            thrust = pos_z_out + vel_z_out;
            
            // Normalize thrust
            thrust = std::clamp(thrust, 0.0, 1.0);
            
            // Adjust for fuel conservation
            if (current_state.fuel < 10.0) {
                thrust *= 0.5;
            }
            
        } else {
            // Approach mode: fast approach to target
            double target_height = 100.0;
            double target_speed = approach_speed;
            
            // Position control
            double pos_x_error = target_state.x - current_state.x;
            double pos_y_error = target_state.y - current_state.y;
            double pos_z_error = target_height - height;
            
            // Velocity control
            double vel_x_error = -current_state.vx;
            double vel_y_error = -current_state.vy;
            double vel_z_error = target_speed - current_state.vz;
            
            // PID outputs
            double pos_x_out = pos_x_ctrl.update(pos_x_error, dt);
            double pos_y_out = pos_y_ctrl.update(pos_y_error, dt);
            double pos_z_out = pos_z_ctrl.update(pos_z_error, dt);
            
            double vel_x_out = vel_x_ctrl.update(vel_x_error, dt);
            double vel_y_out = vel_y_ctrl.update(vel_y_error, dt);
            double vel_z_out = vel_z_ctrl.update(vel_z_error, dt);
            
            // Combine position and velocity control
            pitch_cmd = pos_x_out + vel_x_out;
            yaw_cmd = pos_y_out + vel_y_out;
            thrust = pos_z_out + vel_z_out;
            
            // Normalize thrust
            thrust = std::clamp(thrust, 0.0, 1.0);
        }
        
        // Anti-crash: avoid terrain
        if (height < 20.0 && current_state.vz < -5.0) {
            thrust = std::min(thrust, 0.5);
        }
        
        // Anti-spin: stabilize roll
        double roll_error = -current_state.roll;
        roll_cmd = roll_ctrl.update(roll_error, dt);
        
        return {thrust, pitch_cmd, yaw_cmd, roll_cmd};
    }
};

int main() {
    // Read initial header
    int width, depth, height;
    double gravity, max_thrust, fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    std::string terrain_file;
    
    std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    
    // Read terrain file path (may contain spaces)
    std::getline(std::cin, terrain_file); // consume newline
    std::getline(std::cin, terrain_file);
    
    // Initialize controller
    LanderController controller(width, depth, height, gravity, max_thrust, fuel,
                             start_x, start_y, start_z,
                             target_x, target_y, target_z,
                             terrain_file);
    
    // Skip "STATE" line
    std::string state_line;
    std::getline(std::cin, state_line);
    
    // Main control loop
    while (true) {
        // Read state update
        double x, y, z, vx, vy, vz;
        double pitch, yaw, roll;
        double pitch_rate, yaw_rate, roll_rate;
        double fuel_remaining;
        
        if (!(std::cin >> x >> y >> z >> vx >> vy >> vz >> 
              pitch >> yaw >> roll >> 
              pitch_rate >> yaw_rate >> roll_rate >> 
              fuel_remaining)) {
            break; // End of input
        }
        
        // Update controller state
        controller.update_state(x, y, z, vx, vy, vz,
                              pitch, yaw, roll,
                              pitch_rate, yaw_rate, roll_rate,
                              fuel_remaining);
        
        // Get commands
        auto [thrust, pitch_cmd, yaw_cmd, roll_cmd] = controller.get_commands();
        
        // Output commands
        std::cout << std::fixed << std::setprecision(6);
        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << std::endl;
        std::cout.flush();
        
        // Check for landing or crash
        if (controller.is_landed()) {
            std::cout << "0 0 0 0" << std::endl;
            std::cout.flush();
            break;
        }
        
        if (controller.is_crashed()) {
            std::cout << "0 0 0 0" << std::endl;
            std::cout.flush();
            break;
        }
    }
    
    return 0;
}
