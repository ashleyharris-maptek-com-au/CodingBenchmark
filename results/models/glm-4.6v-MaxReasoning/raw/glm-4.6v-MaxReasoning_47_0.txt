fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    // 1. Compute my_nonce from the global invocation ID and range_start
    let my_nonce = params.y + gid.x;

    // 2. Guard: if the thread ID is out of the specified search range, exit.
    if (gid.x >= params.z) {
        return;
    }

    // 3. Initialize the hash state with the base data
    var state: array<u32, 16>;
    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i];
    }

    // 4. Apply the nonce to the state
    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] ^= (my_nonce + i);
    }

    // 5. Perform 8 rounds of quarter-round mixing (ChaCha-like)
    // The inner loop is unrolled as requested.
    for (var round: u32 = 0u; round < 8u; round = round + 1u) {
        // Group 0: state[0], state[1], state[2], state[3]
        {
            var a = state[0];
            var b = state[1];
            var c = state[2];
            var d = state[3];

            a += b;
            d ^= a;
            d = (d << 16u) | (d >> (32u - 16u));

            c += d;
            b ^= c;
            b = (b << 12u) | (b >> (32u - 12u));

            a += b;
            d ^= a;
            d = (d << 8u) | (d >> (32u - 8u));

            c += d;
            b ^= c;
            b = (b << 7u) | (b >> (32u - 7u));

            state[0] = a;
            state[1] = b;
            state[2] = c;
            state[3] = d;
        }

        {
            var a = state[4];
            var b = state[5];
            var c = state[6];
            var d = state[7];

            a += b;
            d ^= a;
            d = (d << 16u) | (d >> (32u - 16u));

            c += d;
            b ^= c;
            b = (b << 12u) | (b >> (32u - 12u));

            b ^= c;
            b = (b << 7u) | (b >> (32u - 7u));

            state[4] = a;
            state[5] = b;
            state[6] = c;
            state[7] = d;
        }

            var b = state[9];
            var c = state[10];
            var d = state[11];

            a += b;
            d ^= a;
            d = (d << 16u) | (d >> (32u - 16u));

            c += d;
            b ^= c;
            b = (b << 12u) | (b >> (32u - 12u));

            a += b;
            d ^= a;
            d = (d << 8u) | (d >> (32u - 8u));

            c += d;
            b ^= c;
            b = (b << 7u) | (b >> (32u - 7u));

            state[8] = a;
            state[9] = b;
            state[10] = c;
            state[11] = d;
        }

        // Group 3: state[12], state[13], state[14], state[15]
        {
            var a = state[12];
            var b = state[13];
            var c = state[14];
            var d = state[15];

            a += b;
            d ^= a;
            d = (d << 16u) | (d >> (32u - 16u));

            c += d;
            b ^= c;
            b = (b << 12u) | (b >> (32u - 12u));

            a += b;
            d ^= a;
            d = (d << 8u) | (d >> (32u - 8u));

            c += d;
            b ^= c;
            b = (b << 7u) | (b >> (32u - 7u));

            state[12] = a;
            state[13] = b;
            state[14] = c;
            state[15] = d;
        }
    }

    // 6. Fold the 16-word state into a 64-bit hash (h0, h1)
    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    // 7. Count trailing zero bits of the 64-bit hash value (h0 | (h1 << 32))
    fn count_trailing_zeros_64(h0: u32, h1: u32) -> u32 {
        if (h0 == 0u) {
            // If h0 is zero, the trailing zeros come from h1, plus the 32 bits of h0.
            if (h1 == 0u) {
                return 64u; // The entire 64-bit value is zero
            }
            return 32u + firstBitLow(h1);
        } else {
            // If h0 is non-zero, count its trailing zeros directly.
            return firstBitLow(h0);
        }
    }

    let trailing_zeros = count_trailing_zeros_64(h0, h1);

    // 8. Check if the difficulty requirement is met and store the result
    if (trailing_zeros >= params.x) {
        result[0] = 1u; // Solution found flag
        result[1] = my_nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
