#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <cstddef>

using ull = unsigned long long;

const ull base1 = 911382323ULL;
const ull base2 = 972663749ULL;

// Precompute powers of base1 and base2 up to max_len
void precompute_powers(std::vector<ull>& pow1, std::vector<ull>& pow2, size_t max_len) {
    pow1.resize(max_len + 1);
    pow2.resize(max_len + 1);
    pow1[0] = 1;
    pow2[0] = 1;
    for (size_t i = 1; i <= max_len; ++i) {
        pow1[i] = pow1[i - 1] * base1;
        pow2[i] = pow2[i - 1] * base2;
    }
}

// Compute prefix hashes for a string
void compute_prefix_hashes(const std::string& s, std::vector<ull>& pref1, std::vector<ull>& pref2) {
    size_t n = s.size();
    pref1.resize(n + 1);
    pref2.resize(n + 1);
    pref1[0] = 0;
    pref2[0] = 0;
    for (size_t i = 0; i < n; ++i) {
        pref1[i + 1] = pref1[i] * base1 + s[i];
        pref2[i + 1] = pref2[i] * base2 + s[i];
    }
}

// Get hash of substring s[l..r] (inclusive)
std::pair<ull, ull> get_hash(const std::vector<ull>& pref1, const std::vector<ull>& pref2,
                           const std::vector<ull>& pow1, const std::vector<ull>& pow2,
                           size_t l, size_t r) {
    size_t len = r - l + 1;
    ull h1 = pref1[r + 1] - pref1[l] * pow1[len];
    ull h2 = pref2[r + 1] - pref2[l] * pow2[len];
    return {h1, h2};
}

// Check if there exists a common substring of length L
bool exists_common_substring(const std::vector<std::string>& strings, size_t L,
                           const std::vector<ull>& pow1, const std::vector<ull>& pow2) {
    if (L == 0) return true;
    const std::string& first = strings[0];
    size_t n = first.size();
    if (L > n) return false;

    std::unordered_set<std::pair<ull, ull>> hashes;
    std::vector<ull> pref1, pref2;
    compute_prefix_hashes(first, pref1, pref2);

    for (size_t i = 0; i <= n - L; ++i) {
        auto h = get_hash(pref1, pref2, pow1, pow2, i, i + L - 1);
        hashes.insert(h);
    }

    for (size_t idx = 1; idx < strings.size(); ++idx) {
        const std::string& s = strings[idx];
        size_t m = s.size();
        if (L > m) return false;

        std::vector<ull> pref1_s, pref2_s;
        compute_prefix_hashes(s, pref1_s, pref2_s);

        for (size_t i = 0; i <= m - L; ++i) {
            auto h = get_hash(pref1_s, pref2_s, pow1, pow2, i, i + L - 1);
            if (hashes.count(h)) {
                return true;
            }
        }
    }
    return false;
}

// Find a common substring of length L
std::string find_common_substring(const std::vector<std::string>& strings, size_t L,
                                const std::vector<ull>& pow1, const std::vector<ull>& pow2) {
    if (L == 0) return "";
    const std::string& first = strings[0];
    size_t n = first.size();
    if (L > n) return "";

    std::unordered_map<std::pair<ull, ull>, size_t> hash_to_index;
    std::vector<ull> pref1, pref2;
    compute_prefix_hashes(first, pref1, pref2);

    for (size_t i = 0; i <= n - L; ++i) {
        auto h = get_hash(pref1, pref2, pow1, pow2, i, i + L - 1);
        hash_to_index[h] = i;
    }

    for (size_t idx = 1; idx < strings.size(); ++idx) {
        const std::string& s = strings[idx];
        size_t m = s.size();
        if (L > m) return "";

        std::vector<ull> pref1_s, pref2_s;
        compute_prefix_hashes(s, pref1_s, pref2_s);

        for (size_t i = 0; i <= m - L; ++i) {
            auto h = get_hash(pref1_s, pref2_s, pow1, pow2, i, i + L - 1);
            auto it = hash_to_index.find(h);
            if (it != hash_to_index.end()) {
                // Verify to avoid collision (very rare)
                size_t start = it->second;
                bool match = true;
                for (size_t j = 0; j < L; ++j) {
                    if (first[start + j] != s[i + j]) {
                        match = false;
                        break;
                    }
                }
                if (match) {
                    return first.substr(start, L);
                }
            }
        }
    }
    return "";
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;
    std::vector<std::string> strings(N);
    size_t max_len = 0;
    for (int i = 0; i < N; ++i) {
        std::cin >> strings[i];
        max_len = std::max(max_len, strings[i].size());
    }

    if (N == 0) {
        std::cout << "
";
        return 0;
    }
    if (N == 1) {
        std::cout << strings[0] << "
";
        return 0;
    }

    // Precompute powers
    std::vector<ull> pow1, pow2;
    precompute_powers(pow1, pow2, max_len);

    // Find minimum length
    size_t min_len = strings[0].size();
    for (const auto& s : strings) {
        min_len = std::min(min_len, s.size());
    }

    // Binary search for the maximum length
    size_t low = 0, high = min_len;
    while (low < high) {
        size_t mid = (low + high + 1) / 2;
        if (exists_common_substring(strings, mid, pow1, pow2)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }

    if (low == 0) {
        std::cout << "
";
        return 0;
    }

    std::string result = find_common_substring(strings, low, pow1, pow2);
    std::cout << result << "
";

    return 0;
}
