using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static void Main()
    {
        // Read input
        var dimensions = Console.ReadLine().Split(' ');
        int H = int.Parse(dimensions[0]);
        int W = int.Parse(dimensions[1]);
        
        char[,] board = new char[H, W];
        for (int i = 0; i < H; i++)
        {
            string row = Console.ReadLine();
            for (int j = 0; j < W; j++)
            {
                board[i, j] = row[j];
            }
        }
        
        // Solve the board
        var safeCells = Solve(board);
        
        // Output results
        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.Item1} {cell.Item2}");
        }
    }
    
    static List<Tuple<int, int>> Solve(char[,] board)
    {
        int H = board.GetLength(0);
        int W = board.GetLength(1);
        var safeCells = new List<Tuple<int, int>>();
        var queue = new Queue<Tuple<int, int>>();
        
        // Initialize queue with all numbered cells
        for (int i = 0; i < H; i++)
        {
            for (int j = 0; j < W; j++)
            {
                if (char.IsDigit(board[i, j]))
                {
                    queue.Enqueue(new Tuple<int, int>(j, i));
                }
            }
        }
        
        while (queue.Count > 0)
        {
            var current = queue.Dequeue();
            int col = current.Item1;
            int row = current.Item2;
            int number = board[row, col] - '0';
            
            var neighbors = GetNeighbors(row, col, H, W);
            int unknownCount = 0;
            int flaggedCount = 0;
            
            foreach (var neighbor in neighbors)
            {
                if (board[neighbor.Item2, neighbor.Item1] == ' ')
                    unknownCount++;
                else if (board[neighbor.Item2, neighbor.Item1] == '*')
                    flaggedCount++;
            }
            
            // If all mines are found, remaining unknowns are safe
            if (number == flaggedCount && unknownCount > 0)
            {
                foreach (var neighbor in neighbors)
                {
                    if (board[neighbor.Item2, neighbor.Item1] == ' ')
                    {
                        safeCells.Add(neighbor);
                        board[neighbor.Item2, neighbor.Item1] = 'S'; // Mark as safe
                        queue.Enqueue(neighbor); // Add to queue for further processing
                    }
                }
            }
            // If all unknowns must be mines
            else if (number == unknownCount + flaggedCount && unknownCount > 0)
            {
                foreach (var neighbor in neighbors)
                {
                    if (board[neighbor.Item2, neighbor.Item1] == ' ')
                    {
                        board[neighbor.Item2, neighbor.Item1] = '*'; // Mark as mine
                        queue.Enqueue(neighbor); // Add to queue for further processing
                    }
                }
            }
        }
        
        // Remove duplicates and cells that are no longer unknown
        safeCells = safeCells.Distinct().Where(cell => board[cell.Item2, cell.Item1] == 'S').ToList();
        
        return safeCells;
    }
    
    static List<Tuple<int, int>> GetNeighbors(int row, int col, int H, int W)
    {
        var neighbors = new List<Tuple<int, int>>();
        for (int i = -1; i <= 1; i++)
        {
            for (int j = -1; j <= 1; j++)
            {
                if (i == 0 && j == 0) continue;
                int newRow = row + i;
                int newCol = col + j;
                if (newRow >= 0 && newRow < H && newCol >= 0 && newCol < W)
                {
                    neighbors.Add(new Tuple<int, int>(newCol, newRow));
                }
            }
        }
        return neighbors;
    }
}
