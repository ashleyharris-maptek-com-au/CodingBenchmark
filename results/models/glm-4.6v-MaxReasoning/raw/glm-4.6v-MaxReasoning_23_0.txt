use std::io::{self, BufRead, BufReader};
use std::time::{Duration, Instant};
use std::f64::consts::PI;

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut line = String::new();

    // Read initial map parameters
    reader.read_line(&mut line)?;
    let mut params = line.split_whitespace();
    let width: usize = params.next().unwrap().parse().unwrap();
    let height: usize = params.next().unwrap().parse().unwrap();
    let gravity: f64 = params.next().unwrap().parse().unwrap();
    let max_thrust: f64 = params.next().unwrap().parse().unwrap();
    let fuel: f64 = params.next().unwrap().parse().unwrap();

    // Read start and target positions
    line.clear();
    reader.read_line(&mut line)?;
    let mut pos_params = line.split_whitespace();
    let start_x: f64 = pos_params.next().unwrap().parse().unwrap();
    let start_y: f64 = pos_params.next().unwrap().parse().unwrap();
    let target_x: f64 = pos_params.next().unwrap().parse().unwrap();
    let target_y: f64 = pos_params.next().unwrap().parse().unwrap();

    // Read map
    let mut map = vec![vec![' '; width]; height];
    for y in 0..height {
        line.clear();
        reader.read_line(&mut line)?;
        let row = line.trim();
        for (x, ch) in row.chars().enumerate() {
            if x < width {
                map[y][x] = ch;
            }
        }
    }

    // Initialize state
    let mut x = start_x;
    let mut y = start_y;
    let mut vx = 0.0;
    let mut vy = 0.0;
    let mut angle = -PI / 2.0; // Start facing up
    let mut angular_vel = 0.0;
    let mut fuel_remaining = fuel;

    // PID controller parameters
    let kp_pos = 0.5;
    let ki_pos = 0.1;
    let kd_pos = 0.3;
    let kp_angle = 2.0;
    let ki_angle = 0.5;
    let kd_angle = 0.8;

    let mut pos_error_integral = 0.0;
    let mut angle_error_integral = 0.0;
    let mut last_pos_error = 0.0;
    let mut last_angle_error = 0.0;

    // Timing
    let mut last_update = Instant::now();
    let target_update_interval = Duration::from_millis(50); // 20Hz

    // Main control loop
    loop {
        // Read state update
        line.clear();
        if reader.read_line(&mut line).unwrap() == 0 {
            break; // End of input
        }
        
        let mut state_params = line.split_whitespace();
        x = state_params.next().unwrap().parse().unwrap();
        y = state_params.next().unwrap().parse().unwrap();
        vx = state_params.next().unwrap().parse().unwrap();
        vy = state_params.next().unwrap().parse().unwrap();
        angle = state_params.next().unwrap().parse().unwrap();
        angular_vel = state_params.next().unwrap().parse().unwrap();
        fuel_remaining = state_params.next().unwrap().parse().unwrap();

        // Calculate time delta
        let now = Instant::now();
        let dt = (now - last_update).as_secs_f64();
        last_update = now;

        // Calculate position error (distance to target)
        let dx = target_x - x;
        let dy = target_y - y;
        let distance = (dx * dx + dy * dy).sqrt();
        
        // Calculate desired angle (pointing to target)
        let desired_angle = dy.atan2(dx);
        let angle_error = desired_angle - angle;
        
        // Normalize angle error to [-PI, PI]
        let mut normalized_angle_error = angle_error;
        while normalized_angle_error > PI {
            normalized_angle_error -= 2.0 * PI;
        }
        while normalized_angle_error < -PI {
            normalized_angle_error += 2.0 * PI;
        }

        // PID for position
        pos_error_integral += distance * dt;
        let pos_error_derivative = distance - last_pos_error;
        last_pos_error = distance;
        
        // PID for angle
        angle_error_integral += normalized_angle_error * dt;
        let angle_error_derivative = normalized_angle_error - last_angle_error;
        last_angle_error = normalized_angle_error;

        // Calculate control outputs
        let pos_control = kp_pos * distance + ki_pos * pos_error_integral + kd_pos * pos_error_derivative;
        let angle_control = kp_angle * normalized_angle_error + ki_angle * angle_error_integral + kd_angle * angle_error_derivative;

        // Calculate thrust (clamped to 0-1)
        let mut thrust = pos_control / max_thrust;
        thrust = thrust.clamp(0.0, 1.0);

        // Calculate turn (clamped to -1 to 1)
        let mut turn = angle_control;
        turn = turn.clamp(-1.0, 1.0);

        // Check for obstacles and adjust if needed
        let lookahead_distance = 5.0;
        let lookahead_x = x + vx * dt * lookahead_distance;
        let lookahead_y = y + vy * dt * lookahead_distance;
        
        if lookahead_x >= 0.0 && lookahead_x < width as f64 && 
           lookahead_y >= 0.0 && lookahead_y < height as f64 {
            let map_x = lookahead_x.floor() as usize;
            let map_y = lookahead_y.floor() as usize;
            if map[map_y][map_x] == '#' {
                // Obstacle ahead, reduce thrust and adjust angle
                thrust *= 0.3;
                turn = if angle_control > 0.0 { 1.0 } else { -1.0 };
            }
        }

        // Check if we're close to target and should reduce thrust for soft landing
        if distance < 5.0 {
            let speed = (vx * vx + vy * vy).sqrt();
            if speed < 2.0 {
                // Very close, minimal thrust for landing
                thrust = 0.1;
            } else {
                // Reduce thrust based on distance and speed
                thrust = thrust * (1.0 - distance / 10.0).max(0.1);
            }
        }

        // Ensure we have enough fuel
        if fuel_remaining < 1.0 {
            thrust = 0.0;
        }

        // Output control commands
        println!("{} {}", thrust, turn);

        // Throttle to maintain 20Hz
        let elapsed = now.elapsed();
        if elapsed < target_update_interval {
            std::thread::sleep(target_update_interval - elapsed);
        }
    }

    Ok(())
}
