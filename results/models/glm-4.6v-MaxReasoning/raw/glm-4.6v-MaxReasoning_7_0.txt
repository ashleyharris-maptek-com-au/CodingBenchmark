#include <vector>
#include <iostream>
#include <unordered_map>

#include <mutex>

// Vector 3D structure with basic operations
struct Vec3 {
    double x, y, z;

    Vec3 operator-(const Vec3& other) const {
        return {x - other.x, y - other.y, z - other.z};
    }
    Vec3 operator+(const Vec3& other) const {
        return {x + other.x, y + other.y, z + other.z};
    }
    Vec3 operator*(double scalar) const {
        return {x * scalar, y * scalar, z * scalar};
    }
    Vec3 operator/(double scalar) const {
        return {x / scalar, y / scalar, z / scalar};
    }
    double dot(const Vec3& other) const {
        return x * other.x + y * other.y + z * other.z;
    }
    Vec3 cross(const Vec3& other) const {
        return {
            y * other.z - z * other.y,
            z * other.x - x * other.z,
            x * other.y - y * other.x
        };
    }
    double length() const {
        return std::sqrt(x * x + y * y + z * z);
    }
    Vec3 normalized() const {
        double len = length();
        if (len == 0) return *this;
        return *this / len;
    }
};

// Plane structure: normal . (x, y, z) + d = 0
struct Plane {
    Vec3 normal;
    double d;
};

// Face structure (triangle)
struct Face {
    int v0, v1, v2;
};

// Mesh structure
struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<Face> faces;
};

// Enumeration for point classification against a plane
enum class Side {
    FRONT,
    BACK,
    ON
};

// Classify a point against a plane
Side classify_point(const Vec3& p, const Plane& plane) {
    double dist = plane.normal.dot(p) + plane.d;
    const double epsilon = 1e-9;
    if (dist > epsilon) return Side::FRONT;
    if (dist < -epsilon) return Side::BACK;
    return Side::ON;
}

// Create a plane from a triangular face
Plane from_face(const Face& f, const Mesh& mesh) {
    Vec3 v0 = mesh.vertices[f.v0];
    Vec3 v1 = mesh.vertices[f.v1];
    Vec3 v2 = mesh.vertices[f.v2];
    Vec3 normal = (v1 - v0).cross(v2 - v0).normalized();
    double d = -normal.dot(v0);
    return {normal, d};
}

// Sutherland-Hodgman polygon clipping for a single plane
std::vector<Vec3> clip_polygon(const std::vector<Vec3>& poly, const Plane& plane) {
    std::vector<Vec3> output;
    if (poly.empty()) return output;

    const Vec3& start = poly.back();
    Side start_side = classify_point(start, plane);

    for (const Vec3& end : poly) {
        Side end_side = classify_point(end, plane);

        if (end_side != Side::BACK) {
            if (start_side == Side::BACK) {
                // Edge straddles, add intersection point
                Vec3 dir = end - start;
                double t = -(plane.normal.dot(start) + plane.d) / plane.normal.dot(dir);
                output.push_back(start + dir * t);
            }
            output.push_back(end);
        } else if (start_side != Side::BACK) {
            // Edge straddles, add intersection point
            Vec3 dir = end - start;
            double t = -(plane.normal.dot(start) + plane.d) / plane.normal.dot(dir);
            output.push_back(start + dir * t);
        }
        start_side = end_side;
    }
    return output;
}

// BSP Tree Node
struct BSPNode {
    Plane plane;
    std::unique_ptr<BSPNode> front;
    std::unique_ptr<BSPNode> back;
    BSPNode(const Plane& p) : plane(p) {}
};

// Recursive function to build the BSP tree
BSPNode* build_bsp_tree(const Mesh& mesh, const std::vector<int>& face_indices, int depth = 0) {
    if (face_indices.empty() || depth > 30) {
        return nullptr;
    }

    // Heuristic: find the plane closest to the average centroid of remaining faces
    Vec3 avg_centroid(0, 0, 0);
    for (int i : face_indices) {
        const Face& f = mesh.faces[i];
        Vec3 centroid = (mesh.vertices[f.v0] + mesh.vertices[f.v1] + mesh.vertices[f.v2]) / 3.0;
        avg_centroid += centroid;
    }
    avg_centroid /= face_indices.size();

    double best_dist_sq = std::numeric_limits<double>::max();
    int best_split_index = face_indices[0];
    for (int i : face_indices) {
        const Face& f = mesh.faces[i];
        Plane p = from_face(f, mesh);
        double dist_sq = (p.normal.dot(avg_centroid) + p.d) * (p.normal.dot(avg_centroid) + p.d);
        if (dist_sq < best_dist_sq) {
            best_dist_sq = dist_sq;
            best_split_index = i;
        }
    }

    Plane split_plane = from_face(mesh.faces[best_split_index], mesh);

    std::vector<int> front_indices, back_indices;
    for (int i : face_indices) {
        const Face& f = mesh.faces[i];
        Side v0_side = classify_point(mesh.vertices[f.v0], split_plane);
        Side v1_side = classify_point(mesh.vertices[f.v1], split_plane);
        Side v2_side = classify_point(mesh.vertices[f.v2], split_plane);

        // Simple heuristic: if all on one side, put in front/back
        if (v0_side == v1_side && v1_side == v2_side) {
            if (v0_side == Side::FRONT) front_indices.push_back(i);
            else if (v0_side == Side::BACK) back_indices.push_back(i);
        } else {
            // Face straddles, put in front for now (will be split later)
            front_indices.push_back(i);
        }
    }

    BSPNode* node = new BSPNode(split_plane);
    node->front.reset(build_bsp_tree(mesh, front_indices, depth + 1));
    node->back.reset(build_bsp_tree(mesh, back_indices, depth + 1));
    return node;
}

// Helper to quantize a vertex for the map to avoid floating-point issues
std::tuple<int, int, int> quantize_vertex(const Vec3& v, double precision = 1000.0) {
    return std::make_tuple(
        static_cast<int>(std::round(v.x * precision)),
        static_cast<int>(std::round(v.y * precision)),
        static_cast<int>(std::round(v.z * precision))
    );
}

// Helper to get or add a vertex to a mesh and map
int get_or_add_vertex(const Vec3& v, std::unordered_map<std::tuple<int, int, int>, int>& vertex_map, std::vector<Vec3>& vertices) {
    auto key = quantize_vertex(v);
    auto it = vertex_map.find(key);
    if (it != vertex_map.end()) {
        return it->second;
    }
    int index = vertices.size();
    vertices.push_back(v);
    vertex_map[key] = index;
    return index;
}

// Recursive function to clip a face against the BSP tree
std::vector<Vec3> clip_face_against_bsp(const std::vector<Vec3>& poly, const BSPNode* node, std::unordered_map<std::tuple<int, int, int>, int>& vertex_map) {
    if (!node) {
        return poly;
    }

    std::vector<Vec3> clipped_poly = clip_polygon(poly, node->plane);
    if (clipped_poly.empty()) {
        return {};
    }

    // Classify the clipped polygon
    bool has_front = false, has_back = false, has_on = false;
    for (const auto& v : clipped_poly) {
        Side s = classify_point(v, node->plane);
        if (s == Side::FRONT) has_front = true;
        else if (s == Side::BACK) has_back = true;
        else has_on = true;
    }

    std::vector<Vec3> result;
    if (has_front) {
        auto front_part = clip_face_against_bsp(clipped_poly, node->front.get(), vertex_map);
        result.insert(result.end(), front_part.begin(), front_part.end());
    }
    if (has_back) {
        auto back_part = clip_face_against_bsp(clipped_poly, node->back.get(), vertex_map);
        result.insert(result.end(), back_part.begin(), back_part.end());
    }
    return result;
}

// Function to clip a mesh against a BSP tree (parallelized)
Mesh clip_mesh(const Mesh& mesh, const BSPNode* root) {
    Mesh result;
    std::unordered_map<std::tuple<int, int, int>, int> vertex_map;
    std::mutex vertex_map_mutex;

    const int num_faces = mesh.faces.size();
    const int num_threads = std::thread::hardware_concurrency();
    int faces_per_thread = (num_faces + num_threads - 1) / num_threads;

    std::vector<std::future<void>> futures;
    for (int i = 0; i < num_threads; ++i) {
        int start = i * faces_per_thread;
        int end = std::min(start + faces_per_thread, num_faces);
        if (start >= num_faces) break;

        futures.emplace_back(std::async(std::launch::async, [&, start, end]() {
            for (int j = start; j < end; ++j) {
                const Face& f = mesh.faces[j];
                std::vector<Vec3> poly = {mesh.vertices[f.v0], mesh.vertices[f.v1], mesh.vertices[f.v2]};
                std::vector<Vec3> clipped_poly = clip_face_against_bsp(poly, root, vertex_map);

                // Triangulate the clipped polygon (which can be a triangle or quad)
                for (size_t k = 1; k < clipped_poly.size() - 1; ++k) {
                    std::lock_guard<std::mutex> lock(vertex_map_mutex);
                    result.faces.push_back({
                        get_or_add_vertex(clipped_poly[0], vertex_map, result.vertices),
                        get_or_add_vertex(clipped_poly[k], vertex_map, result.vertices),
                        get_or_add_vertex(clipped_poly[k+1], vertex_map, result.vertices)
                    });
                }
            }
        }));
    }

    for (auto& future : futures) {
        future.get();
    }

    return result;
}

int main() {
    // Read input
    int va, fa, vb, fb;
    std::cin >> va >> fa >> vb >> fb;

    Mesh mesh_a, mesh_b;

    // Read Mesh A vertices
    for (int i = 0; i < va; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        mesh_a.vertices.emplace_back(x, y, z);
    }

    // Read Mesh A faces
    for (int i = 0; i < fa; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        mesh_a.faces.emplace_back(a, b, c);
    }

    // Read Mesh B vertices
    for (int i = 0; i < vb; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        mesh_b.vertices.emplace_back(x, y, z);
    }

    // Read Mesh B faces
    for (int i = 0; i < fb; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        mesh_b.faces.emplace_back(a, b, c);
    }

    // Build BSP tree for Mesh A
    std::vector<int> face_indices_a(fa);
    std::iota(face_indices_a.begin(), face_indices_a.end(), 0);
    std::unique_ptr<BSPNode> bsp_root(build_bsp_tree(mesh_a, face_indices_a));

    // Clip Mesh B against the BSP tree of Mesh A
    Mesh clipped_b = clip_mesh(mesh_b, bsp_root.get());

    // The union is Mesh A plus the clipped part of Mesh B
    Mesh union_mesh;
    union_mesh.vertices = mesh_a.vertices;
    union_mesh.faces = mesh_a.faces;

    // Add clipped B's vertices and faces
    int base_vertex_index = union_mesh.vertices.size();
    for (const auto& v : clipped_b.vertices) {
        union_mesh.vertices.push_back(v);
    }
    for (const auto& f : clipped_b.faces) {
        union_mesh.faces.push_back({f.v0 + base_vertex_index, f.v1 + base_vertex_index, f.v2 + base_vertex_index});
    }

    // Output the result
    std::cout << union_mesh.vertices.size() << " " << union_mesh.faces.size() << std::endl;
    for (const auto& v : union_mesh.vertices) {
        std::cout << v.x << " " << v.y << " " << v.z << std::endl;
    }
    for (const auto& f : union_mesh.faces) {
        std::cout << f.v0 << " " << f.v1 << " " << f.v2 << std::endl;
    }

    return 0;
}
