; SPIR-V assembly for a geometry shader that renders a thick smooth curve using adjacency
; Input: InputLinesAdjacency (4 vertices: [prev, start, end, next])
; Output: Triangle strip with 4 vertices for the thick line segment

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Input types and variables
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%v3float = OpTypeVector %float 3
%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output types and variables
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%v3float_out = OpTypeVector %float 3
%ptr_out_v3 = OpTypePointer Output %v3float_out
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%float_0 = OpConstant %float 0
%float_025 = OpConstant %float 0.025
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Main function
%main = OpFunction %void None %voidfn
%entry = OpLabel
%pos_prev = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos_start = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos_end = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos_next = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%pos_prev_val = OpLoad %v4float %pos_prev
%pos_start_val = OpLoad %v4float %pos_start
%pos_end_val = OpLoad %v4float %pos_end
%pos_next_val = OpLoad %v4float %pos_next

; Compute tangents
%vec_start_end = OpFSub %v4float %pos_end_val %pos_prev_val
%vec_end_next = OpFSub %v4float %pos_next_val %pos_start_val

%t0 = OpVectorTimesScalar %v4float %vec_start_end %float_025
%t1 = OpVectorTimesScalar %v4float %vec_end_next %float_025

%t0_norm = OpNormalize %v4float %t0
%t1_norm = OpNormalize %v4float %t1

; Compute perpendiculars (2D projection)
%t0_xy = OpVectorShuffle %v2float %t0_norm 1 2
%t1_xy = OpVectorShuffle %v2float %t1_norm 1 2

%perp0 = OpVectorShuffle %v2float %t0_xy 1 0
%perp1 = OpVectorShuffle %v2float %t1_xy 1 0

%perp0_neg = OpFNegate %v2float %perp0
%perp1_neg = OpFNegate %v2float %perp1

; Thickness
%thickness = OpConstant %float 0.025

; Compute vertex positions
%start_perp0 = OpVectorTimesScalar %v2float %perp0_neg %thickness
%start_perp0_4d = OpCompositeConstruct %v4float %start_perp0 %float_0 %float_0
%start_perp0_pos = OpFAdd %v4float %pos_start_val %start_perp0_4d

%start_perp0_neg = OpVectorTimesScalar %v2float %perp0 %thickness
%start_perp0_neg_4d = OpCompositeConstruct %v4float %start_perp0_neg %float_0 %float_0
%start_perp0_neg_pos = OpFSub %v4float %pos_start_val %start_perp0_neg_4d

%end_perp1 = OpVectorTimesScalar %v2float %perp1_neg %thickness
%end_perp1_4d = OpCompositeConstruct %v4float %end_perp1 %float_0 %float_0
%end_perp1_pos = OpFAdd %v4float %pos_end_val %end_perp1_4d

%end_perp1_neg = OpVectorTimesScalar %v2float %perp1 %thickness
%end_perp1_neg_4d = OpCompositeConstruct %v4float %end_perp1_neg %float_0 %float_0
%end_perp1_neg_pos = OpFSub %v4float %pos_end_val %end_perp1_neg_4d

; Output colors
%color_start = OpAccessChain %ptr_in_v3 %color %int_1
%color_end = OpAccessChain %ptr_in_v3 %color %int_2
%color_start_val = OpLoad %v3float %color_start
%color_end_val = OpLoad %v3float %color_end

; Emit vertices
; Vertex 1: start + perp0*thickness
OpStore %outColor %color_start_val
%out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1 %start_perp0_pos
OpEmitVertex

; Vertex 2: start - perp0*thickness
OpStore %outColor %color_start_val
%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %start_perp0_neg_pos
OpEmitVertex

; Vertex 3: end + perp1*thickness
OpStore %outColor %color_end_val
%out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos3 %end_perp1_pos
OpEmitVertex

; Vertex 4: end - perp1*thickness
OpStore %outColor %color_end_val
%out_pos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos4 %end_perp1_neg_pos
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
