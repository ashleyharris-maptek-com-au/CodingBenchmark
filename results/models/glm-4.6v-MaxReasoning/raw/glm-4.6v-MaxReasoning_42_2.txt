; SPIR-V assembly for a geometry shader that expands points into hexagons
; Input: InputPoints (1 vertex per invocation)
; Output: OutputTriangleStrip (8 vertices per point: center + 6 edges + close)
; Capabilities, Memory Model, and Entrypoint
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Types and decorations for input gl_PerVertex block
%v4float = OpTypeFloat 32
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Types and decorations for user-defined inputs
%v3float = OpTypeVector %v4float 3
%worldPos = OpTypeArray %v3float %uint_1
%normal = OpTypeArray %v3float %uint_1
%color = OpTypeArray %v3float %uint_1
%ptr_in_worldPos = OpTypePointer Input %worldPos
%ptr_in_normal = OpTypePointer Input %normal
%ptr_in_color = OpTypePointer Input %color
%worldPos_in = OpVariable %ptr_in_worldPos Input
%normal_in = OpVariable %ptr_in_normal Input
%color_in = OpVariable %ptr_in_color Input
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2

; Types and decorations for output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Type and decoration for user-defined output
%outColor = OpTypePointer Output %v3float
%outColor_var = OpVariable %outColor Output
OpDecorate %outColor_var Location 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %v4float 0.0
%float_1 = OpConstant %v4float 1.0
%float_04 = OpConstant %v4float 0.04
%float_60 = OpConstant %v4float 1.0472  ; 60 degrees in radians
%float_120 = OpConstant %v4float 2.0944  ; 120 degrees
%float_180 = OpConstant %v4float 3.1416  ; 180 degrees
%float_240 = OpConstant %v4float 4.1888  ; 240 degrees
%float_300 = OpConstant %v4float 5.2360  ; 300 degrees
%float_360 = OpConstant %v4float 6.2832  ; 360 degrees
%v3float_0 = OpConstant %v3float 0.0 0.0 0.0
%v3float_1 = OpConstant %v3float 1.0 1.0 1.0

; Function definition
%main = OpFunction %void None %void_type
%void_type = OpTypeVoid
%main_fn = OpFunction %void None %void_type
%entry = OpLabel
OpBranch %main_body

%main_body = OpLabel
; Load input position (gl_in[0].gl_Position)
%gl_in_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
%gl_PerVertex_in_ptr = OpAccessChain %gl_PerVertex_in %gl_in_ptr %int_0
%gl_Position_in = OpLoad %gl_PerVertex_in %gl_in_ptr
%pos = OpAccessChain %v4float %gl_PerVertex_in_ptr %int_0
%pos_val = OpLoad %v4float %pos

; Load input color (color[0])
%color_ptr = OpAccessChain %ptr_in_color %color_in %int_0
%color_val = OpLoad %v3float %color_ptr

; Define hexagon vertices (center + 6 edges)
; Center vertex (V0)
%center_pos = OpAccessChain %v4float %gl_out %int_0
OpStore %center_pos %pos_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %color_val
OpEmitVertex

; Edge vertex 0 (0°)
%edge0_pos = OpAccessChain %v4float %gl_out %int_0
%edge0_val = OpFAdd %v4float %pos_val %float_04
OpStore %edge0_pos %edge0_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
%dark_color = OpFAdd %v3float %color_val %v3float_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; Edge vertex 1 (60°)
%edge1_pos = OpAccessChain %v4float %gl_out %int_0
%edge1_val = OpFAdd %v4float %pos_val %float_60
OpStore %edge1_pos %edge1_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; Edge vertex 2 (120°)
%edge2_pos = OpAccessChain %v4float %gl_out %int_0
%edge2_val = OpFAdd %v4float %pos_val %float_120
OpStore %edge2_pos %edge2_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; Edge vertex 3 (180°)
%edge3_pos = OpAccessChain %v4float %gl_out %int_0
%edge3_val = OpFAdd %v4float %pos_val %float_180
OpStore %edge3_pos %edge3_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; Edge vertex 4 (240°)
%edge4_pos = OpAccessChain %v4float %gl_out %int_0
%edge4_val = OpFAdd %v4float %pos_val %float_240
OpStore %edge4_pos %edge4_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; Edge vertex 5 (300°)
%edge5_pos = OpAccessChain %v4float %gl_out %int_0
%edge5_val = OpFAdd %v4float %pos_val %float_300
OpStore %edge5_pos %edge5_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; Close the hexagon (edge 0 again)
%close_pos = OpAccessChain %v4float %gl_out %int_0
%close_val = OpFAdd %v4float %pos_val %float_04
OpStore %close_pos %close_val
%outColor_ptr = OpAccessChain %v3float %outColor_var %int_0
OpStore %outColor_ptr %dark_color
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
