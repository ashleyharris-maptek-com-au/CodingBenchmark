#include <iostream>
#include <vector>
#include <queue>
#include <tuple>
#include <algorithm>

using namespace std;

struct Task {
    int machine;
    long long duration;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<Task>> tasks(J);
    long long total_tasks = 0;

    for (int i = 0; i < J; ++i) {
        int T;
        cin >> T;
        total_tasks += T;
        tasks[i].resize(T);
        for (int j = 0; j < T; ++j) {
            cin >> tasks[i][j].machine >> tasks[i][j].duration;
        }
    }

    vector<vector<pair<long long, int>>> schedule(J);

    // Adaptive strategy: use EST rule for small instances, simple list scheduling for large ones.
    const long long THRESHOLD = 5000; // Adjust based on empirical performance.

    if (total_tasks <= THRESHOLD) {
        // Earliest Start Time (EST) heuristic – interleaves tasks from different jobs.
        vector<long long> job_last_finish(J, 0); // Time when the last task of each job finished.
        vector<int> next_task_index(J, 0);      // Index of the next task to schedule for each job.
        vector<long long> machine_free(M, 0);   // When each machine becomes free.

        // Priority queue: (start_time, machine, job_id, task_id)
        using TaskEvent = tuple<long long, int, int, int>;
        priority_queue<TaskEvent, vector<TaskEvent>, greater<TaskEvent>> pq;

        // Initialise: push the first ready task of each job.
        for (int i = 0; i < J; ++i) {
            if (next_task_index[i] < static_cast<int>(tasks[i].size())) {
                int machine = tasks[i][next_task_index[i]].machine;
                long long duration = tasks[i][next_task_index[i]].duration;
                long long start = max(job_last_finish[i], machine_free[machine]);
                pq.push({start, machine, i, next_task_index[i]});
            }
        }

        while (!pq.empty()) {
            auto [start, machine, job_id, task_id] = pq.top();
            pq.pop();

            long long duration = tasks[job_id][task_id].duration;
            schedule[job_id].push_back({start, machine});

            // Update machine and job state.
            machine_free[machine] = start + duration;
            job_last_finish[job_id] = start + duration;

            // Advance to the next task of the same job.
            next_task_index[job_id]++;
            if (next_task_index[job_id] < static_cast<int>(tasks[job_id].size())) {
                int next_machine = tasks[job_id][next_task_index[job_id]].machine;
                long long next_duration = tasks[job_id][next_task_index[job_id]].duration;
                long long next_start = max(job_last_finish[job_id], machine_free[next_machine]);
                pq.push({next_start, next_machine, job_id, next_task_index[job_id]});
            }
        }
    } else {
        // Simple list scheduling – processes jobs in input order.
        vector<long long> machine_free(M, 0);
        for (int i = 0; i < J; ++i) {
            long long job_start = 0; // Start time of the first task of the current job.
            for (const auto& task : tasks[i]) {
                int machine = task.machine;
                long long duration = task.duration;
                long long start = max(job_start, machine_free[machine]);
                schedule[i].push_back({start, machine});
                machine_free[machine] = start + duration;
                job_start = start + duration;
            }
        }
    }

    // Output the schedule.
    for (int i = 0; i < J; ++i) {
        for (size_t j = 0; j < schedule[i].size(); ++j) {
            cout << schedule[i][j].first << " " << schedule[i][j].second;
            if (j + 1 < schedule[i].size()) {
                cout << " ";
            }
        }
        cout << "
";
    }

    return 0;
}
