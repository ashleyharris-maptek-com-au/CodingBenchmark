using namespace std;
using namespace std::chrono;

// Configuration
constexpr double TETRAHEDRON_EDGE_LENGTH = 1.0;
constexpr double TETRAHEDRON_HEIGHT = sqrt(2.0 / 3.0);
constexpr double TETRAHEDRON_CIRCUMRADIUS = sqrt(3.0 / 8.0);
constexpr double TETRAHEDRON_INRADIUS = 1.0 / sqrt(24.0);
constexpr double OVERLAP_MARGIN = 0.001; // Small margin for numerical stability
constexpr int MAX_PLACEMENT_ATTEMPTS = 100;
constexpr int GRID_CELL_SIZE = 2; // Size of grid cells for spatial partitioning

// 3D vector class
struct Vec3 {
    double x, y, z;
    
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}
    
    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double scalar) const { return {x * scalar, y * scalar, z * scalar}; }
    Vec3 operator/(double scalar) const { return {x / scalar, y / scalar, z / scalar}; }
    
    double dot(const Vec3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vec3 cross(const Vec3& other) const { 
        return {y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x}; 
    }
    
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalized() const { double len = length(); return len > 0 ? (*this) / len : Vec3(); }
    
    static double distance(const Vec3& a, const Vec3& b) { return (a - b).length(); }
};

// Tetrahedron vertices (regular tetrahedron with edge length 1)
const vector<Vec3> TETRAHEDRON_VERTICES = {
    {0, 0, 0},
    {1, 0, 0},
    {0.5, sqrt(3.0) / 2.0, 0},
    {0.5, sqrt(3.0) / 6.0, sqrt(2.0 / 3.0)}
};

// Euler angles to rotation matrix
struct EulerAngles {
    double rx, ry, rz;
};

// Apply rotation to a point
Vec3 rotatePoint(const Vec3& point, const EulerAngles& angles) {
    // Simplified rotation using Rodrigues' rotation formula
    // This is a simplified approach - for production code, use proper rotation matrices
    
    // Convert angles to radians (they're already in radians)
    double cx = cos(angles.rx), sx = sin(angles.rx);
    double cy = cos(angles.ry), sy = sin(angles.ry);
    double cz = cos(angles.rz), sz = sin(angles.rz);
    
    // Rotation matrices
    double Rz[3][3] = {{cz, -sz, 0}, {sz, cz, 0}, {0, 0, 1}};
    double Ry[3][3] = {{cy, 0, sy}, {0, 1, 0}, {-sy, 0, cy}};
    double Rx[3][3] = {{1, 0, 0}, {0, cx, -sx}, {0, sx, cx}};
    
    // Combined rotation (Rz * Ry * Rx)
    double R[3][3] = {
        {cy * cz, cy * sz, sy},
        {sx * sy * cz - cx * sz, sx * sy * sz + cx * cz, -sx * cy},
        {cx * sy * cz + sx * sz, cx * sy * sz - sx * cz, cx * cy}
    };
    
    // Apply rotation
    Vec3 rotated;
    rotated.x = R[0][0] * point.x + R[0][1] * point.y + R[0][2] * point.z;
    rotated.y = R[1][0] * point.x + R[1][1] * point.y + R[1][2] * point.z;
    rotated.z = R[2][0] * point.x + R[2][1] * point.y + R[2][2] * point.z;
    
    return rotated;
}

// Get tetrahedron vertices in world space
vector<Vec3> getTetrahedronVertices(const Vec3& center, const EulerAngles& angles) {
    vector<Vec3> worldVertices;
    for (const auto& vertex : TETRAHEDRON_VERTICES) {
        worldVertices.push_back(center + rotatePoint(vertex, angles));
    }
    return worldVertices;
}

// Bounding sphere for a tetrahedron
struct BoundingSphere {
    Vec3 center;
    double radius;
    
    BoundingSphere(const Vec3& center, double radius) : center(center), radius(radius) {}
    
    bool overlaps(const BoundingSphere& other) const {
        double dist = Vec3::distance(center, other.center);
        return dist < radius + other.radius + OVERLAP_MARGIN;
    }
};

// Container polyhedron
struct Container {
    vector<Vec3> vertices;
    Vec3 minBound, maxBound;
    
    Container(const vector<Vec3>& vertices) : vertices(vertices) {
        computeBounds();
    }
    
    void computeBounds() {
        if (vertices.empty()) return;
        
        minBound = maxBound = vertices[0];
        for (const auto& v : vertices) {
            minBound.x = min(minBound.x, v.x);
            minBound.y = min(minBound.y, v.y);
            minBound.z = min(minBound.z, v.z);
            maxBound.x = max(maxBound.x, v.x);
            maxBound.y = max(maxBound.y, v.y);
            maxBound.z = max(maxBound.z, v.z);
        }
    }
    
    bool contains(const Vec3& point) const {
        // Simplified containment check - for convex polyhedra, we could use point-in-polyhedron test
        // Here we use a bounding box approximation for speed
        return point.x >= minBound.x && point.x <= maxBound.x &&
               point.y >= minBound.y && point.y <= maxBound.y &&
               point.z >= minBound.z && point.z <= maxBound.z;
    }
    
    bool containsTetrahedron(const Vec3& center, const EulerAngles& angles) const {
        auto vertices = getTetrahedronVertices(center, angles);
        for (const auto& v : vertices) {
            if (!contains(v)) return false;
        }
        return true;
    }
};

// Spatial grid for efficient overlap detection
class SpatialGrid {
private:
    struct GridCell {
        vector<int> tetrahedronIndices;
    };
    
    vector<GridCell> cells;
    Vec3 gridMin, gridMax;
    int gridX, gridY, gridZ;
    double cellSize;
    
public:
    SpatialGrid(const Vec3& minBound, const Vec3& maxBound, double cellSize) 
        : gridMin(minBound), gridMax(maxBound), cellSize(cellSize) {
        
        gridX = max(1, static_cast<int>((gridMax.x - gridMin.x) / cellSize) + 1);
        gridY = max(1, static_cast<int>((gridMax.y - gridMin.y) / cellSize) + 1);
        gridZ = max(1, static_cast<int>((gridMax.z - gridMin.z) / cellSize) + 1);
        
        cells.resize(gridX * gridY * gridZ);
    }
    
    int getCellIndex(int x, int y, int z) const {
        return z * gridY * gridX + y * gridX + x;
    }
    
    tuple<int, int, int> getCellCoords(const Vec3& point) const {
        int x = max(0, min(gridX - 1, static_cast<int>((point.x - gridMin.x) / cellSize)));
        int y = max(0, min(gridY - 1, static_cast<int>((point.y - gridMin.y) / cellSize)));
        int z = max(0, min(gridZ - 1, static_cast<int>((point.z - gridMin.z) / cellSize)));
        return {x, y, z};
    }
    
    void addTetrahedron(int index, const BoundingSphere& sphere) {
        auto [x, y, z] = getCellCoords(sphere.center);
        
        // Add to neighboring cells for better overlap detection
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    int nx = x + dx, ny = y + dy, nz = z + dz;
                    if (nx >= 0 && nx < gridX && ny >= 0 && ny < gridY && nz >= 0 && nz < gridZ) {
                        cells[getCellIndex(nx, ny, nz)].tetrahedronIndices.push_back(index);
                    }
                }
            }
        }
    }
    
    vector<int> getPotentialOverlaps(const BoundingSphere& sphere) const {
        vector<int> overlaps;
        auto [x, y, z] = getCellCoords(sphere.center);
        
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    int nx = x + dx, ny = y + dy, nz = z + dz;
                    if (nx >= 0 && nx < gridX && ny >= 0 && ny < gridY && nz >= 0 && nz < gridZ) {
                        const auto& cell = cells[getCellIndex(nx, ny, nz)];
                        overlaps.insert(overlaps.end(), cell.tetrahedronIndices.begin(), cell.tetrahedronIndices.end());
                    }
                }
            }
        }
        
        // Remove duplicates
        sort(overlaps.begin(), overlaps.end());
        overlaps.erase(unique(overlaps.begin(), overlaps.end()), overlaps.end());
        
        return overlaps;
    }
};

// Tetrahedron placement
struct TetrahedronPlacement {
    Vec3 center;
    EulerAngles angles;
    BoundingSphere boundingSphere;
    
    TetrahedronPlacement(const Vec3& center, const EulerAngles& angles)
        : center(center), angles(angles), 
          boundingSphere(center, TETRAHEDRON_CIRCUMRADIUS) {}
};

// Main solver class
class TetrahedronPacker {
private:
    Container container;
    vector<TetrahedronPlacement> placements;
    SpatialGrid spatialGrid;
    mt19937 rng;
    uniform_real_distribution<double> dist;
    
public:
    TetrahedronPacker(const Container& container) 
        : container(container), 
          spatialGrid(container.minBound, container.maxBound, GRID_CELL_SIZE),
          rng(random_device{}()),
          dist(0.0, 1.0) {}
    
    // Check if two tetrahedrons overlap
    bool tetrahedronsOverlap(const TetrahedronPlacement& a, const TetrahedronPlacement& b) {
        // First check bounding spheres
        if (!a.boundingSphere.overlaps(b.boundingSphere)) {
            return false;
        }
        
        // If spheres overlap, check actual tetrahedron intersection
        auto verticesA = getTetrahedronVertices(a.center, a.angles);
        auto verticesB = getTetrahedronVertices(b.center, b.angles);
        
        // Simple overlap check - check if any vertex of A is inside B or vice versa
        // This is a simplified approach; for production, use proper tetrahedron intersection
        
        for (const auto& vA : verticesA) {
            if (pointInTetrahedron(vA, verticesB)) return true;
        }
        for (const auto& vB : verticesB) {
            if (pointInTetrahedron(vB, verticesA)) return true;
        }
        
        return false;
    }
    
    // Check if a point is inside a tetrahedron (using barycentric coordinates)
    bool pointInTetrahedron(const Vec3& point, const vector<Vec3>& tetrahedron) {
        // Compute vectors
        Vec3 v0 = tetrahedron[1] - tetrahedron[0];
        Vec3 v1 = tetrahedron[2] - tetrahedron[0];
        Vec3 v2 = tetrahedron[3] - tetrahedron[0];
        Vec3 vP = point - tetrahedron[0];
        
        // Compute dot products
        double dot00 = v0.dot(v0);
        double dot01 = v0.dot(v1);
        double dot02 = v0.dot(v2);
        double dot11 = v1.dot(v1);
        double dot12 = v1.dot(v2);
        double dot22 = v2.dot(v2);
        double dot0P = v0.dot(vP);
        double dot1P = v1.dot(vP);
        double dot2P = v2.dot(vP);
        
        // Compute barycentric coordinates
        double invDenom = 1.0 / (dot00 * dot11 * dot22 + 2.0 * dot01 * dot12 * dot02 - 
                               dot00 * dot12 * dot12 - dot01 * dot01 * dot22 - 
                               dot02 * dot02 * dot11);
        
        double u = (dot11 * dot22 - dot12 * dot12) * dot0P + 
                  (dot02 * dot12 - dot01 * dot22) * dot1P + 
                  (dot01 * dot12 - dot02 * dot11) * dot2P;
        u *= invDenom;
        
        double v = (dot02 * dot12 - dot01 * dot22) * dot0P + 
                  (dot00 * dot22 - dot02 * dot02) * dot1P + 
                  (dot01 * dot02 - dot00 * dot12) * dot2P;
        v *= invDenom;
        
        double w = (dot01 * dot12 - dot02 * dot11) * dot0P + 
                  (dot01 * dot02 - dot00 * dot12) * dot1P + 
                  (dot00 * dot11 - dot01 * dot01) * dot2P;
        w *= invDenom;
        
        double t = 1.0 - u - v - w;
        
        return u >= 0 && v >= 0 && w >= 0 && t >= 0;
    }
    
    // Try to place a tetrahedron at a given position and orientation
    bool tryPlaceTetrahedron(const Vec3& center, const EulerAngles& angles) {
        TetrahedronPlacement newPlacement(center, angles);
        
        // Check if inside container
        if (!container.containsTetrahedron(center, angles)) {
            return false;
        }
        
        // Check for overlaps with existing placements
        auto potentialOverlaps = spatialGrid.getPotentialOverlaps(newPlacement.boundingSphere);
        for (int index : potentialOverlaps) {
            if (tetrahedronsOverlap(newPlacement, placements[index])) {
                return false;
            }
        }
        
        // Place the tetrahedron
        placements.push_back(newPlacement);
        spatialGrid.addTetrahedron(placements.size() - 1, newPlacement.boundingSphere);
        return true;
    }
    
    // Generate random orientation
    EulerAngles randomOrientation() {
        return {dist(rng) * 2 * M_PI, dist(rng) * 2 * M_PI, dist(rng) * 2 * M_PI};
    }
    
    // Pack tetrahedrons using a greedy approach
    void pack() {
        // Estimate container volume
        double containerVolume = (container.maxBound.x - container.minBound.x) * 
                               (container.maxBound.y - container.minBound.y) * 
                               (container.maxBound.z - container.minBound.z);
        
        // Estimate number of tetrahedrons we can fit (very rough estimate)
        int targetCount = static_cast<int>(containerVolume / (TETRAHEDRON_CIRCUMRADIUS * TETRAHEDRON_CIRCUMRADIUS * TETRAHEDRON_CIRCUMRADIUS * 4 * M_PI / 3));
        targetCount = min(targetCount, 1000000); // Cap for very large containers
        
        // Try to place tetrahedrons
        for (int i = 0; i < targetCount; ++i) {
            // Try multiple positions for each tetrahedron
            bool placed = false;
            for (int attempt = 0; attempt < MAX_PLACEMENT_ATTEMPTS; ++attempt) {
                // Generate random position within container bounds
                Vec3 center(
                    container.minBound.x + dist(rng) * (container.maxBound.x - container.minBound.x),
                    container.minBound.y + dist(rng) * (container.maxBound.y - container.minBound.y),
                    container.minBound.z + dist(rng) * (container.maxBound.z - container.minBound.z)
                );
                
                // Try random orientation
                EulerAngles angles = randomOrientation();
                
                if (tryPlaceTetrahedron(center, angles)) {
                    placed = true;
                    break;
                }
            }
            
            // Early exit if we can't place more tetrahedrons
            if (!placed && i > targetCount / 2) {
                break;
            }
        }
    }
    
    // Get results
    vector<TetrahedronPlacement> getPlacements() const {
        return placements;
    }
};

// Read input
Container readContainer() {
    int V;
    cin >> V;
    
    vector<Vec3> vertices;
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        cin >> x >> y >> z;
        vertices.emplace_back(x, y, z);
    }
    
    return Container(vertices);
}

// Write output
void writeOutput(const vector<TetrahedronPlacement>& placements) {
    cout << placements.size() << endl;
    for (const auto& placement : placements) {
        cout << placement.center.x << " " << placement.center.y << " " << placement.center.z << " "
             << placement.angles.rx << " " << placement.angles.ry << " " << placement.angles.rz << endl;
    }
}

int main() {
    // Read input
    Container container = readContainer();
    
    // Create and run packer
    TetrahedronPacker packer(container);
    
    // Time the packing process
    auto start = high_resolution_clock::now();
    packer.pack();
    auto end = high_resolution_clock::now();
    
    // Write output
    writeOutput(packer.getPlacements());
    
    // Optional: print timing info
    // auto duration = duration_cast<milliseconds>(end - start).count();
    // cerr << "Packing took " << duration << " ms" << endl;
    
    return 0;
}
