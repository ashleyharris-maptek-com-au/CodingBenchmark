
```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate light direction (from fragment to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Calculate view direction (from fragment to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Calculate reflection vector
    float3 R = reflect(-L, N);

    // Ambient component
    float3 ambient = float3(0.1, 0.05, 0.05);

    // Diffuse component (using vertex color)
    float diffuseIntensity = max(dot(N, L), 0.0);
    float3 diffuse = input.color * diffuseIntensity;

    // Specular component (white color)
    float shininess = 32.0;
    float specularIntensity = pow(max(dot(R, V), 0.0), shininess);
    float3 specular = float3(1.0, 1.0, 1.0) * specularIntensity;

    // Combine components
    float3 finalColor = ambient + diffuse + specular;

    // Clamp the final color to [0, 1]
    finalColor = clamp(finalColor, 0.0, 1.0);

    // Set alpha to 1.0
    return float4(finalColor, 1.0);
}
```
