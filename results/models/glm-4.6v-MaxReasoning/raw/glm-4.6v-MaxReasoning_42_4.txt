
```spirv
; SPIR-V assembly for geometry shader: Expand lines into tapered ribbons
; Input: InputLines (2 vertices per invocation)
; Output: OutputTriangleStrip (4 vertices per primitive)

OpCapability Geometry
OpMemoryModel Logical GLSL450

%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%mat4 = OpTypeMatrix %v4float 4

; Input gl_PerVertex block (position only)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input arrays (size 2 for InputLines)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_pos = OpTypeArray %v3float %uint_2
%ptr_in_pos = OpTypePointer Input %arr_pos
%worldPos = OpVariable %ptr_in_pos Input
OpDecorate %worldPos Location 0

%arr_normal = OpTypeArray %v3float %uint_2
%ptr_in_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_normal Input
OpDecorate %normal Location 1

%arr_color = OpTypeArray %v3float %uint_2
%ptr_in_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_color Input
OpDecorate %color Location 2

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%vec3_out = OpTypePointer Output %v3float
%outColor = OpVariable %vec3_out Output
OpDecorate %outColor Location 0

; UBO (set=0, binding=0)
%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input vertices
%pos0_ptr = OpAccessChain %ptr_in_pos %worldPos %int_0
%pos0 = OpLoad %v3float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_pos %worldPos %int_1
%pos1 = OpLoad %v3float %pos1_ptr

%color0_ptr = OpAccessChain %ptr_in_color %color %int_0
%color0 = OpLoad %v3float %color0_ptr
%color1_ptr = OpAccessChain %ptr_in_color %color %int_1
%color1 = OpLoad %v3float %color1_ptr

%normal0_ptr = OpAccessChain %ptr_in_normal %normal %int_0
%normal0 = OpLoad %v3float %normal0_ptr

; Compute line direction and perpendicular vector
%dir = OpFSub %v3float %pos1 %pos0
%perp = OpCross %v3float %dir %normal0

; Normalize perpendicular vector
%perp_len = OpCompositeExtract %float %perp 0
%perp_len = OpFAdd %float %perp_len (OpCompositeExtract %float %perp 1)
%perp_len = OpFAdd %float %perp_len (OpCompositeExtract %float %perp 2)
%perp_len = OpFSqrt %float %perp_len
%perp = OpCompositeInsert %v3float %perp (OpFDiv %float (OpCompositeExtract %v3float %perp 0) %perp_len) 0
%perp = OpCompositeInsert %v3float %perp (OpFDiv %float (OpCompositeExtract %v3float %perp 1) %perp_len) 1
%perp = OpCompositeInsert %v3float %perp (OpFDiv %float (OpCompositeExtract %v3float %perp 2) %perp_len) 2

; Thickness values
%thick_start = OpConstant %float 0.04
%thick_end = OpConstant %float 0.005

; Vertex 0: p0 + perp*thick_start
%v0_pos = OpFAdd %v3float %pos0 (OpFMul %v3float %perp %thick_start)
%v0_pos = OpCompositeConstruct %v4float (OpCompositeExtract %v3float %v0_pos 0) (OpCompositeExtract %v3float %v0_pos 1) (OpCompositeExtract %v3float %v0_pos 2) %float_1.0
%v0_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %v0_pos_ptr %v0_pos
%v0_color = OpLoad %v3float %color0
%v0_color_ptr = OpAccessChain %vec3_out %outColor
OpStore %v0_color_ptr %v0_color
OpEmitVertex

; Vertex 1: p0 - perp*thick_start
%v1_pos = OpFSub %v3float %pos0 (OpFMul %v3float %perp %thick_start)
%v1_pos = OpCompositeConstruct %v4float (OpCompositeExtract %v3float %v1_pos 0) (OpCompositeExtract %v3float %v1_pos 1) (OpCompositeExtract %v3float %v1_pos 2) %float_1.0
%v1_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %v1_pos_ptr %v1_pos
%v1_color = OpLoad %v3float %color0
%v1_color_ptr = OpAccessChain %vec3_out %outColor
OpStore %v1_color_ptr %v1_color
OpEmitVertex

; Vertex 2: p1 + perp*thick_end
%v2_pos = OpFAdd %v3float %pos1 (OpFMul %v3float %perp %thick_end)
%v2_pos = OpCompositeConstruct %v4float (OpCompositeExtract %v3float %v2_pos 0) (OpCompositeExtract %v3float %v2_pos 1) (OpCompositeExtract %v3float %v2_pos 2) %float_1.0
%v2_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %v2_pos_ptr %v2_pos
%v2_color = OpLoad %v3float %color1
%v2_color_ptr = OpAccessChain %vec3_out %outColor
OpStore %v2_color_ptr %v2_color
OpEmitVertex

; Vertex 3: p1 - perp*thick_end
%v3_pos = OpFSub %v3float %pos1 (OpFMul %v3float %perp %thick_end)
%v3_pos = OpCompositeConstruct %v4float (OpCompositeExtract %v3float %v3_pos 0) (OpCompositeExtract %v3float %v3_pos 1) (OpCompositeExtract %v3float %v3_pos 2) %float_1.0
%v3_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %v3_pos_ptr %v3_pos
%v3_color = OpLoad %v3float %color1
%v3_color_ptr = OpAccessChain %vec3_out %outColor
OpStore %v3_color_ptr %v3_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
```
