
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);
    
    // Calculate checkerboard pattern (8x8 frequency)
    float checkerX = floor(uv.x * 8.0);
    float checkerY = floor(uv.y * 8.0);
    float parity = mod(checkerX + checkerY, 2.0);
    vec3 checkerColor = parity < 0.5 ? vec3(0.9) : vec3(0.2);
    
    // Calculate light direction (world space)
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse component using Lambert's cosine law
    float diffuse = max(dot(N, lightDir), 0.0);
    
    // Apply Lambertian shading factor
    float shadingFactor = 0.5 + 0.5 * diffuse;
    
    // Combine checkerboard color with shading
    vec3 finalColor = checkerColor * shadingFactor;
    
    // Output final color with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
