use std::io::{self, BufRead, BufWriter, Write};
use std::time::Instant;
use std::collections::{HashSet, HashMap};
use std::sync::Arc;
use rayon::prelude::*;
use rand::Rng;

fn main() -> io::Result<()> {
    let start = Instant::now();
    
    // Read input
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap()?;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n = parts[0].parse::<usize>().unwrap();
    let target = parts[1].parse::<i64>().unwrap();
    
    let numbers: Vec<i64> = lines.next().unwrap()?
        .split_whitespace()
        .take(n)
        .map(|x| x.parse::<i64>().unwrap())
        .collect();
    
    let result = solve_subset_sum(&numbers, target);
    
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    match result {
        Some(indices) => {
            writeln!(out, "YES")?;
            let indices_str: Vec<String> = indices.iter().map(|&i| i.to_string()).collect();
            writeln!(out, "[{}]", indices_str.join(" "))?;
        }
        None => {
            writeln!(out, "NO")?;
        }
    }
    
    writeln!(out, "Time: {:?}", start.elapsed())?;
    out.flush()?;
    
    Ok(())
}

fn solve_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    
    // Handle edge cases
    if target == 0 {
        return Some(vec![]); // Empty subset sums to 0
    }
    
    // If all numbers are positive and sum is less than target, no solution
    if numbers.iter().all(|&x| x > 0) && numbers.iter().sum::<i64>() < target {
        return None;
    }
    
    // If all numbers are negative and sum is greater than target, no solution
    if numbers.iter().all(|&x| x < 0) && numbers.iter().sum::<i64>() > target {
        return None;
    }
    
    // Small input: use DP with bitset
    if n <= 100 {
        return dp_bitset_solution(numbers, target);
    }
    
    // Medium input: use meet-in-the-middle
    if n <= 10000 {
        return meet_in_the_middle(numbers, target);
    }
    
    // Large input: use Karmarkar-Karp heuristic
    karmarkar_karp(numbers, target)
}

fn dp_bitset_solution(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mut dp = vec![HashSet::new(); n + 1];
    dp[0].insert(0);
    
    for i in 0..n {
        let num = numbers[i];
        for j in (0..=i).rev() {
            for &sum in &dp[j] {
                dp[j + 1].insert(sum + num);
            }
        }
    }
    
    if dp[n].contains(&target) {
        // Reconstruct the subset
        let mut result = Vec::new();
        let mut current_sum = target;
        let mut remaining = n;
        
        while remaining > 0 {
            for i in 0..remaining {
                if dp[remaining - 1].contains(&(current_sum - numbers[i])) {
                    result.push(i);
                    current_sum -= numbers[i];
                    remaining -= 1;
                    break;
                }
            }
        }
        
        result.reverse();
        Some(result)
    } else {
        None
    }
}

fn meet_in_the_middle(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let half = n / 2;
    
    let left_part = &numbers[..half];
    let right_part = &numbers[half..];
    
    // Generate all subset sums for left part
    let left_sums: Vec<(i64, Vec<usize>)> = generate_subset_sums(left_part);
    
    // Generate all subset sums for right part
    let right_sums: Vec<(i64, Vec<usize>)> = generate_subset_sums(right_part);
    
    // Create a hash map for right sums
    let mut right_map: HashMap<i64, Vec<Vec<usize>>> = HashMap::new();
    for (_, indices) in &right_sums {
        let sum = indices.iter().map(|&i| right_part[i]).sum::<i64>();
        right_map.entry(sum).or_default().push(indices.clone());
    }
    
    // Check for each left sum if there's a matching right sum
    for (left_sum, left_indices) in &left_sums {
        let needed = target - left_sum;
        if let Some(right_indices_list) = right_map.get(&needed) {
            for right_indices in right_indices_list {
                let mut result = Vec::new();
                result.extend(left_indices.iter().map(|&i| i));
                result.extend(right_indices.iter().map(|&i| i + half));
                return Some(result);
            }
        }
    }
    
    None
}

fn generate_subset_sums(nums: &[i64]) -> Vec<(i64, Vec<usize>)> {
    let n = nums.len();
    let mut result = Vec::new();
    
    for mask in 0..(1 << n) {
        let mut sum = 0;
        let mut indices = Vec::new();
        
        for i in 0..n {
            if mask & (1 << i) != 0 {
                sum += nums[i];
                indices.push(i);
            }
        }
        
        result.push((sum, indices));
    }
    
    result
}

fn karmarkar_karp(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mut rng = rand::thread_rng();
    
    // Try multiple random partitions
    for _ in 0..1000 {
        let mut subset = Vec::new();
        let mut sum = 0;
        
        for i in 0..n {
            if rng.gen::<bool>() {
                subset.push(i);
                sum += numbers[i];
            }
        }
        
        if sum == target {
            return Some(subset);
        }
    }
    
    None
}
