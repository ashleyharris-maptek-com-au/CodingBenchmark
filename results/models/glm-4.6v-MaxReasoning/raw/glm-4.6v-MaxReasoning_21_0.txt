#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <deque>
#include <algorithm>
#include <cmath>
#include <limits>

using namespace std;

class NDSnakeGame {
private:
    int D; // number of dimensions
    vector<int> bounds; // bounds for each dimension
    deque<vector<int>> snake; // snake body, head at front
    unordered_set<long long> occupied; // occupied positions (snake + obstacles)
    vector<vector<int>> food; // food positions
    int foodIndex = 0; // index of next food to eat
    int maxTurns;
    int currentTurn = 0;
    
    // Helper function to convert vector<int> to a unique long long identifier
    long long posToKey(const vector<int>& pos) {
        long long key = 0;
        long long multiplier = 1;
        for (int i = 0; i < D; ++i) {
            key += pos[i] * multiplier;
            multiplier *= (bounds[i] + 1); // +1 to handle 0-based indexing
        }
        return key;
    }
    
    // Helper function to convert long long key back to vector<int>
    vector<int> keyToPos(long long key) {
        vector<int> pos(D);
        for (int i = D - 1; i >= 0; --i) {
            pos[i] = key % (bounds[i] + 1);
            key /= (bounds[i] + 1);
        }
        return pos;
    }
    
    // Check if a position is valid (within bounds and not occupied)
    bool isValid(const vector<int>& pos) {
        for (int i = 0; i < D; ++i) {
            if (pos[i] < 0 || pos[i] >= bounds[i]) {
                return false;
            }
        }
        return occupied.find(posToKey(pos)) == occupied.end();
    }
    
    // Manhattan distance in N dimensions
    int manhattanDistance(const vector<int>& a, const vector<int>& b) {
        int dist = 0;
        for (int i = 0; i < D; ++i) {
            dist += abs(a[i] - b[i]);
        }
        return dist;
    }
    
    // A* pathfinding with limited search depth and early termination
    vector<int> findPathToFood(int maxDepth = 1000) {
        if (foodIndex >= food.size()) return {};
        
        vector<int> target = food[foodIndex];
        vector<int> head = snake.front();
        
        // Priority queue: (f_score, g_score, position)
        priority_queue<tuple<int, int, vector<int>>, vector<tuple<int, int, vector<int>>>, greater<>> pq;
        unordered_map<long long, int> gScore; // cost from start to current
        unordered_map<long long, vector<int>> cameFrom; // for path reconstruction
        
        long long startKey = posToKey(head);
        gScore[startKey] = 0;
        pq.push({manhattanDistance(head, target), 0, head});
        
        vector<int> directions(D * 2);
        for (int i = 0; i < D; ++i) {
            directions[i * 2] = i;
            directions[i * 2 + 1] = 1;
            directions[i * 2 + 2] = i;
            directions[i * 2 + 3] = -1;
        }
        
        int nodesExplored = 0;
        while (!pq.empty() && nodesExplored < maxDepth) {
            auto [fScore, gScoreVal, current] = pq.top();
            pq.pop();
            nodesExplored++;
            
            if (current == target) {
                // Reconstruct path
                vector<int> path;
                long long currentKey = posToKey(current);
                while (currentKey != startKey) {
                    vector<int> prev = keyToPos(cameFrom[currentKey]);
                    for (int i = 0; i < D; ++i) {
                        if (current[i] != prev[i]) {
                            path.push_back(i);
                            path.push_back(current[i] > prev[i] ? 1 : -1);
                            break;
                        }
                    }
                    current = prev;
                    currentKey = posToKey(current);
                }
                reverse(path.begin(), path.end());
                return path;
            }
            
            for (int i = 0; i < D * 2; i += 2) {
                vector<int> next = current;
                next[directions[i]] += directions[i + 1];
                
                if (!isValid(next)) continue;
                
                long long nextKey = posToKey(next);
                int tentativeGScore = gScoreVal + 1;
                
                if (gScore.find(nextKey) == gScore.end() || tentativeGScore < gScore[nextKey]) {
                    cameFrom[nextKey] = current;
                    gScore[nextKey] = tentativeGScore;
                    int fScore = tentativeGScore + manhattanDistance(next, target);
                    pq.push({fScore, tentativeGScore, next});
                }
            }
        }
        
        return {}; // No path found within depth limit
    }
    
    // Find any valid move (for survival)
    vector<int> findAnyValidMove() {
        vector<int> head = snake.front();
        
        // Try to find any valid move
        for (int i = 0; i < D; ++i) {
            for (int dir : {1, -1}) {
                vector<int> next = head;
                next[i] += dir;
                if (isValid(next)) {
                    return {i, dir};
                }
            }
        }
        
        return {}; // No valid moves
    }
    
public:
    NDSnakeGame(int D, const vector<int>& bounds, int snakeLength, const vector<int>& startPos, 
               const vector<vector<int>>& food, const vector<vector<int>>& obstacles, int maxTurns)
        : D(D), bounds(bounds), food(food), maxTurns(maxTurns) {
        // Initialize snake
        snake.push_back(startPos);
        occupied.insert(posToKey(startPos));
        
        // Add initial snake body (simplified - assumes snake is initially straight)
        vector<int> current = startPos;
        for (int i = 1; i < snakeLength; ++i) {
            vector<int> tail = current;
            // Move one step back from head (simplified)
            tail[0] -= 1; // Assuming movement along first dimension for simplicity
            snake.push_back(tail);
            occupied.insert(posToKey(tail));
            current = tail;
        }
        
        // Add obstacles
        for (const auto& obstacle : obstacles) {
            occupied.insert(posToKey(obstacle));
        }
    }
    
    void play() {
        while (currentTurn < maxTurns) {
            // Find path to next food
            vector<int> path = findPathToFood();
            
            if (path.empty()) {
                // No path to food, try to survive
                vector<int> move = findAnyValidMove();
                
                if (move.empty()) {
                    // No valid moves, game over
                    break;
                }
                
                int axis = move[0];
                int dir = move[1];
                
                // Move snake
                vector<int> head = snake.front();
                vector<int> next = head;
                next[axis] += dir;
                
                snake.push_front(next);
                occupied.insert(posToKey(next));
                
                // Remove tail (since no food was eaten)
                vector<int> tail = snake.back();
                occupied.erase(posToKey(tail));
                snake.pop_back();
                
                // Output move
                cout << axis << " " << dir << endl;
                currentTurn++;
                
                continue;
            }
            
            // Follow the path
            if (!path.empty()) {
                int axis = path[0];
                int dir = path[1];
                
                // Move snake
                vector<int> head = snake.front();
                vector<int> next = head;
                next[axis] += dir;
                
                snake.push_front(next);
                occupied.insert(posToKey(next));
                
                // Check if we ate food
                if (next == food[foodIndex]) {
                    foodIndex++;
                } else {
                    // Remove tail
                    vector<int> tail = snake.back();
                    occupied.erase(posToKey(tail));
                    snake.pop_back();
                }
                
                // Output move
                cout << axis << " " << dir << endl;
                currentTurn++;
                
                // Check win condition
                if (foodIndex >= food.size()) {
                    break; // All food eaten
                }
            }
        }
    }
};

int main() {
    int D, maxTurns;
    cin >> D >> maxTurns;
    
    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        cin >> bounds[i];
    }
    
    int snakeLength;
    vector<int> startPos(D);
    cin >> snakeLength;
    for (int i = 0; i < D; ++i) {
        cin >> startPos[i];
    }
    
    int F;
    cin >> F;
    vector<vector<int>> food(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) {
            cin >> food[i][j];
        }
    }
    
    int O;
    cin >> O;
    vector<vector<int>> obstacles(O, vector<int>(D));
    for (int i = 0; i < O; ++i) {
        for (int j = 0; j < D; ++j) {
            cin >> obstacles[i][j];
        }
    }
    
    NDSnakeGame game(D, bounds, snakeLength, startPos, food, obstacles, maxTurns);
    game.play();
    
    return 0;
}
