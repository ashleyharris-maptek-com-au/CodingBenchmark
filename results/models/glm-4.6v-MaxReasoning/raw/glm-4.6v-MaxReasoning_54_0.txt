import math
import numpy as np

class Autopilot:
    def __init__(self):
        # Initialize persistent state variables
        self.phase = "intercept"  # intercept, approach, flare, rollout
        self.go_around_active = False
        self.go_around_count = 0
        self.flaps_configured = False
        self.gear_down = False
        self.spoilers_deployed = False
        self.reverse_deployed = False
        self.autobrake_active = False
        self.last_radio_alt = 0
        self.touchdown_vertical_speed = 0
        self.touchdown_altitude = 0
        self.runway_centerline = 0
        self.runway_threshold_distance = 0
        self.last_ias = 0
        self.target_speed = 0
        self.elevator_trim = 0
        self.aileron_trim = 0
        self.rudder_trim = 0
        self.last_gs_dots = 0
        self.last_loc_dots = 0
        self.last_pitch = 0
        self.last_roll = 0
        self.last_heading = 0
        self.last_track = 0
        self.last_gs = 0
        self.last_irs_gs = 0
        self.last_irs_track = 0
        self.last_dme = 0
        self.crosswind_compensation = 0
        self.wind_shear_detected = False
        self.wind_shear_recovery = False
        self.engine_failure_detected = False
        self.single_engine_failure = False
        self.elevator_jammed = False
        self.aileron_jammed = False
        self.ils_failure = False
        self.radar_alt_failed = False
        self.flaps_asymmetric = False
        self.turbulence_active = False
        self.ils_flickering = False
        self.go_around_initiated = False
        self.go_around_altitude = 0
        self.go_around_heading = 0
        
    def get_valid_sensor(self, sensors, key_prefix, count=3):
        """Get the most valid sensor reading from redundant sensors"""
        valid_readings = []
        for i in range(1, count + 1):
            key = f"{key_prefix}_{i}"
            valid_key = f"{key_prefix}_{i}_valid"
            if state.get(valid_key, False):
                value = state.get(key, 0)
                valid_readings.append(value)
        
        if not valid_readings:
            return 0, False
        
        # Use median to reject outliers
        valid_readings.sort()
        median = valid_readings[len(valid_readings) // 2]
        return median, True
    
    def get_ils_data(self, state):
        """Get valid ILS data with failure detection"""
        # Get glideslope data
        gs_dots, gs_valid = self.get_valid_sensor(state, "glideslope")
        if not gs_valid:
            gs_dots = 0
        
        # Get localizer data
        loc_dots, loc_valid = self.get_valid_sensor(state, "localizer")
        if not loc_valid:
            loc_dots = 0
        
        # Check for ILS failures
        gs_valid_count = sum(1 for i in range(1, 4) if state.get(f"glideslope_{i}_valid", False))
        loc_valid_count = sum(1 for i in range(1, 4) if state.get(f"localizer_{i}_valid", False))
        
        self.ils_failure = (gs_valid_count < 2 or loc_valid_count < 2)
        
        return gs_dots, loc_dots, gs_valid_count, loc_valid_count
    
    def get_airspeed(self, state):
        """Get valid airspeed with failure detection"""
        ias, ias_valid = self.get_valid_sensor(state, "ias")
        if not ias_valid:
            ias = state.get("target_ias_kt", 137)
        
        self.last_ias = ias
        return ias
    
    def get_altitude(self, state):
        """Get valid altitude with failure detection"""
        # Prefer radio altimeter for low altitude
        radio_alt = state.get("radio_alt_ft", 1000)
        if radio_alt < 500:
            return radio_alt
        
        # Use baro altimeters
        baro_alt, baro_valid = self.get_valid_sensor(state, "baro_alt")
        if baro_valid:
            return baro_alt
        
        # Fallback to GPS
        gps_alt, gps_valid = self.get_valid_sensor(state, "gps", 3)
        if gps_valid:
            return gps_alt
        
        return radio_alt
    
    def get_attitude(self, state):
        """Get valid attitude data"""
        pitch, pitch_valid = self.get_valid_sensor(state, "irs", "pitch_deg")
        roll, roll_valid = self.get_valid_sensor(state, "irs", "roll_deg")
        heading, heading_valid = self.get_valid_sensor(state, "irs", "heading_deg")
        track, track_valid = self.get_valid_sensor(state, "irs", "track_deg")
        gs, gs_valid = self.get_valid_sensor(state, "irs", "gs_kt")
        
        if not pitch_valid:
            pitch = state.get("target_gs_dots", 0) * 0.5  # Rough estimate
        if not roll_valid:
            roll = 0
        if not heading_valid:
            heading = 0
        if not track_valid:
            track = heading
        if not gs_valid:
            gs = state.get("groundspeed_kt", 150)
        
        self.last_pitch = pitch
        self.last_roll = roll
        self.last_heading = heading
        self.last_track = track
        self.last_irs_gs = gs
        
        return pitch, roll, heading, track, gs
    
    def detect_failures(self, state):
        """Detect various failure conditions"""
        # Engine failures
        engine1_running = state.get("engine_1_running", True)
        engine2_running = state.get("engine_2_running", True)
        self.single_engine_failure = (not engine1_running or not engine2_running)
        self.engine_failure_detected = (not engine1_running and not engine2_running)
        
        # Control surface failures
        self.elevator_jammed = state.get("elevator_jammed", False)
        self.aileron_jammed = state.get("aileron_jammed", False)
        
        # Sensor failures
        self.radar_alt_failed = not state.get("radio_alt_ft", False)
        
        # Flaps asymmetric
        self.flaps_asymmetric = state.get("flaps_asymmetric", False)
        
        # Turbulence
        self.turbulence_active = state.get("turbulence_active", False)
        
        # ILS flickering
        self.ils_flickering = state.get("ils_flickering", False)
    
    def configure_landing(self, state):
        """Configure aircraft for landing"""
        if not self.flaps_configured:
            # Deploy flaps based on altitude and speed
            radio_alt = state.get("radio_alt_ft", 1000)
            ias = self.get_airspeed(state)
            
            if radio_alt > 1000:
                flaps = 0
            elif radio_alt > 500:
                flaps = 10
            elif ias > 160:
                flaps = 15
            elif ias > 140:
                flaps = 25
            else:
                flaps = 40
            
            self.flaps_configured = True
            return flaps
        
        return 40  # Full flaps for landing
    
    def intercept_phase(self, state):
        """Intercept phase (>1000ft): Track ILS and configure"""
        gs_dots, loc_dots, _, _ = self.get_ils_data(state)
        pitch, roll, heading, track, gs = self.get_attitude(state)
        ias = self.get_airspeed(state)
        radio_alt = state.get("radio_alt_ft", 1000)
        
        # Calculate intercept angles
        intercept_pitch = 2.0  # degrees
        intercept_roll = 0
        
        # Track localizer
        if abs(loc_dots) > 0.5:
            intercept_roll = -loc_dots * 0.3
        
        # Track glideslope
        if abs(gs_dots) > 0.5:
            intercept_pitch = -gs_dots * 0.2
        
        # Speed control
        target_speed = state.get("target_ias_kt", 160)
        speed_error = ias - target_speed
        pitch_correction = speed_error * 0.01
        
        # Combine controls
        elevator = intercept_pitch + pitch_correction
        aileron = intercept_roll
        rudder = 0
        
        # Throttle for intercept
        throttle = 0.7
        
        # Configure for landing
        flaps = self.configure_landing(state)
        gear = 1.0  # Down
        speedbrake = 0.0
        
        return {
            "elevator": elevator,
            "aileron": aileron,
            "rudder": rudder,
            "throttle_1": throttle,
            "throttle_2": throttle,
            "flaps": flaps,
            "gear": gear,
            "speedbrake": speedbrake,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
            "elevator_trim": 0.0
        }
    
    def approach_phase(self, state):
        """Approach phase (1000ft-200ft): Stabilize on ILS"""
        gs_dots, loc_dots, _, _ = self.get_ils_data(state)
        pitch, roll, heading, track, gs = self.get_attitude(state)
        ias = self.get_airspeed(state)
        radio_alt = state.get("radio_alt_ft", 1000)
        
        # ILS tracking with gain scheduling
        loc_gain = 0.4 if radio_alt > 500 else 0.6
        gs_gain = 0.3 if radio_alt > 500 else 0.5
        
        # Localizer correction
        loc_correction = -loc_dots * loc_gain
        
        # Glideslope correction
        gs_correction = -gs_dots * gs_gain
        
        # Speed control
        target_speed = state.get("target_ias_kt", 137)
        speed_error = ias - target_speed
        speed_correction = speed_error * 0.02
        
        # Combine controls
        elevator = gs_correction + speed_correction
        aileron = loc_correction
        rudder = 0
        
        # Throttle for approach
        throttle = 0.5
        
        # Check for approach instability
        if abs(gs_dots) > 2.0 or abs(loc_dots) > 2.0 or abs(speed_error) > 20:
            self.go_around()
        
        return {
            "elevator": elevator,
            "aileron": aileron,
            "rudder": rudder,
            "throttle_1": throttle,
            "throttle_2": throttle,
            "flaps": 40,
            "gear": 1.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
            "elevator_trim": 0.0
        }
    
    def flare_phase(self, state):
        """Flare phase (<50ft): Reduce descent rate"""
        radio_alt = state.get("radio_alt_ft", 0)
        pitch, roll, heading, track, gs = self.get_attitude(state)
        ias = self.get_airspeed(state)
        
        # flare at 50ft
        if radio_alt > 50:
            # Continue approach
            return self.approach_phase(state)
        
        # Calculate flare rate
        descent_rate = state.get("irs_1_vs_fpm", 0)
        if descent_rate > 0:
            descent_rate = -descent_rate  # Make positive for descent
        
        # Flare control
        flare_pitch = 5.0  # degrees nose up for flare
        elevator = flare_pitch
        
        # Reduce thrust for flare
        throttle = 0.1
        
        # Check for hard landing
        if descent_rate > 600:
            self.go_around()
        
        return {
            "elevator": elevator,
            "aileron": 0,
            "rudder": 0,
            "throttle_1": throttle,
            "throttle_2": throttle,
            "flaps": 40,
            "gear": 1.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
            "elevator_trim": 0.0
        }
    
    def rollout_phase(self, state):
        """Touchdown and rollout: Deploy spoilers, reverse thrust"""
        radio_alt = state.get("radio_alt_ft", 0)
        on_ground = state.get("on_ground", False)
        weight_on_wheels = state.get("weight_on_wheels", False)
        groundspeed = state.get("groundspeed_kt", 0)
        
        # Deploy spoilers on touchdown
        if not self.spoilers_deployed and (on_ground or weight_on_wheels):
            self.spoilers_deployed = True
        
        # Deploy reverse thrust
        if groundspeed > 60 and not self.reverse_deployed:
            self.reverse_deployed = True
            self.autobrake_active = True
        
        # Autobrake control
        if self.autobrake_active:
            autobrake = 0.8
        else:
            autobrake = 0.0
        
        # Stop reverse thrust at low speed
        if groundspeed < 30:
            reverse = 0.0
        else:
            reverse = 1.0 if self.reverse_deployed else 0.0
        
        return {
            "elevator": 0,
            "aileron": 0,
            "rudder": 0,
            "throttle_1": reverse,
            "throttle_2": reverse,
            "flaps": 40,
            "gear": 1.0,
            "speedbrake": 0.0,
            "spoiler": 1.0 if self.spoilers_deployed else 0.0,
            "autobrake": autobrake,
            "reverse_1": reverse,
            "reverse_2": reverse,
            "elevator_trim": 0.0
        }
    
    def go_around(self):
        """Initiate go-around procedure"""
        if not self.go_around_active:
            self.go_around_active = True
            self.go_around_initiated = True
            self.go_around_count += 1
            self.phase = "go_around"
            self.go_around_altitude = state.get("baro_alt_1_ft", 200)
            self.go_around_heading = state.get("irs_1_heading_deg", 0)
    
    def go_around_phase(self, state):
        """Execute go-around maneuver"""
        # TOGA thrust
        throttle = 1.0
        
        # Pitch up
        elevator = 10.0
        
        # Retract speedbrake
        speedbrake = 0.0
        
        # Climb to 1500ft AGL
        baro_alt = self.get_altitude(state)
        if baro_alt > 1500:
            self.go_around_active = False
            self.phase = "intercept"
            self.go_around_initiated = False
        
        return {
            "elevator": elevator,
            "aileron": 0,
            "rudder": 0,
            "throttle_1": throttle,
            "throttle_2": throttle,
            "flaps": 15,  # Retract flaps partially
            "gear": 1.0,  # Keep gear down
            "speedbrake": speedbrake,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
            "elevator_trim": 0.0
        }
    
    def handle_failures(self, state, controls):
        """Apply failure corrections to controls"""
        # Engine failure compensation
        if self.single_engine_failure:
            # Apply rudder to counter asymmetric thrust
            controls["rudder"] = 0.3 if not state.get("engine_1_running", True) else -0.3
        
        # Elevator jammed
        if self.elevator_jammed:
            controls["elevator"] = 0.0  # Can't control elevator
        
        # Aileron jammed
        if self.aileron_jammed:
            controls["aileron"] = 0.0  # Can't control aileron
        
        # Crosswind compensation
        if abs(self.crosswind_compensation) > 0.1:
            controls["rudder"] = self.crosswind_compensation
        
        return controls
    
    def determine_phase(self, state):
        """Determine current flight phase"""
        radio_alt = state.get("radio_alt_ft", 1000)
        on_ground = state.get("on_ground", False)
        weight_on_wheels = state.get("weight_on_wheels", False)
        
        if self.go_around_active:
            return "go_around"
        
        if on_ground or weight_on_wheels:
            return "rollout"
        
        if radio_alt < 50:
            return "flare"
        
        if radio_alt < 1000:
            return "approach"
        
        return "intercept"
    
    def autopilot_step(self, state: dict, dt: float) -> dict:
        """Main autopilot function"""
        # Reset between scenarios
        if not hasattr(self, 'phase'):
            self.__init__()
        
        # Detect failures
        self.detect_failures(state)
        
        # Determine flight phase
        self.phase = self.determine_phase(state)
        
        # Execute phase-specific logic
        if self.phase == "intercept":
            controls = self.intercept_phase(state)
        elif self.phase == "approach":
            controls = self.approach_phase(state)
        elif self.phase == "flare":
            controls = self.flare_phase(state)
        elif self.phase == "rollout":
            controls = self.rollout_phase(state)
        elif self.phase == "go_around":
            controls = self.go_around_phase(state)
        else:
            controls = self.intercept_phase(state)
        
        # Apply failure corrections
        controls = self.handle_failures(state, controls)
        
        # Limit control outputs
        for key in controls:
            if key in ["elevator", "aileron", "rudder", "elevator_trim"]:
                controls[key] = max(-1.0, min(1.0, controls[key]))
            elif key in ["throttle_1", "throttle_2", "flaps", "gear", "speedbrake", "spoiler", "autobrake", "reverse_1", "reverse_2"]:
                controls[key] = max(0.0, min(1.0, controls[key]))
        
        return controls

# Create global autopilot instance
autopilot = Autopilot()

def autopilot_step(state: dict, dt: float) -> dict:
    return autopilot.autopilot_step(state, dt)
