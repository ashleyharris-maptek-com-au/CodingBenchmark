using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;
using System.IO;
using System.Threading.Tasks;

public class BinPacking3D
{
    // Main entry point
    public static void Main(string[] args)
    {
        var solver = new BinPackingSolver();
        solver.Solve();
    }
}

public class BinPackingSolver
{
    private Polyhedron polyhedron;
    private Vector3 containerSize;
    private List<PackedObject> packedObjects = new List<PackedObject>();
    private List<Quaternion> possibleRotations;
    
    // Performance metrics
    private int vertexCount;
    private int faceCount;
    private double containerVolume;
    private double polyhedronVolume;
    
    public void Solve()
    {
        // Parse input
        ParseInput();
        
        // Determine problem scale and set strategy
        DetermineStrategy();
        
        // Generate possible rotations
        GeneratePossibleRotations();
        
        // Pack objects
        PackObjects();
        
        // Output results
        OutputResults();
    }
    
    private void ParseInput()
    {
        using (var reader = new StreamReader(Console.OpenStandardInput()))
        {
            // Read V and F
            var firstLine = reader.ReadLine().Split(' ');
            vertexCount = int.Parse(firstLine[0]);
            faceCount = int.Parse(firstLine[1]);
            
            // Read vertices
            var vertices = new List<Vector3>();
            for (int i = 0; i < vertexCount; i++)
            {
                var coords = reader.ReadLine().Split(' ').Select(float.Parse).ToArray();
                vertices.Add(new Vector3(coords[0], coords[1], coords[2]));
            }
            
            // Read faces
            var faces = new List<int[]>();
            for (int i = 0; i < faceCount; i++)
            {
                var indices = reader.ReadLine().Split(' ').Select(int.Parse).ToArray();
                faces.Add(indices);
            }
            
            // Read container size
            var size = reader.ReadLine().Split(' ').Select(float.Parse).ToArray();
            containerSize = new Vector3(size[0], size[1], size[2]);
            
            // Create polyhedron
            polyhedron = new Polyhedron(vertices, faces);
            
            // Calculate volumes
            polyhedronVolume = CalculateVolume(polyhedron);
            containerVolume = containerSize.X * containerSize.Y * containerSize.Z;
        }
    }
    
    private void DetermineStrategy()
    {
        // Simple heuristic based on container size and polyhedron complexity
        double scale = containerSize.X * containerSize.Y * containerSize.Z;
        int complexity = vertexCount + faceCount;
        
        // Determine strategy based on scale and complexity
        if (scale < 1000 || complexity < 50) // Trivial
        {
            // Use exhaustive or detailed search
            Console.Error.WriteLine("Using detailed packing strategy for trivial problem");
        }
        else if (scale < 8000 || complexity < 200) // Medium
        {
            // Use greedy with rotation optimization
            Console.Error.WriteLine("Using greedy packing with rotation optimization for medium problem");
        }
        else if (scale < 125000 || complexity < 500) // Large
        {
            // Use simplified greedy with limited rotations
            Console.Error.WriteLine("Using simplified greedy packing for large problem");
        }
        else // Extreme
        {
            // Use very simplified approach
            Console.Error.WriteLine("Using simplified approach for extreme problem");
        }
    }
    
    private void GeneratePossibleRotations()
    {
        // Generate a set of possible rotations (quaternions)
        // For performance, limit the number based on problem scale
        possibleRotations = new List<Quaternion>();
        
        // Base rotations (identity and 90-degree rotations around axes)
        possibleRotations.Add(Quaternion.Identity);
        
        // 90-degree rotations around axes
        possibleRotations.Add(new Quaternion(0, 0, 0.7071f, 0.7071f)); // 90° around Z
        possibleRotations.Add(new Quaternion(0, 0.7071f, 0, 0.7071f)); // 90° around Y
        possibleRotations.Add(new Quaternion(0.7071f, 0, 0, 0.7071f)); // 90° around X
        
        // 180-degree rotations
        possibleRotations.Add(new Quaternion(0, 0, 1, 0)); // 180° around Z
        possibleRotations.Add(new Quaternion(0, 1, 0, 0)); // 180° around Y
        possibleRotations.Add(new Quaternion(1, 0, 0, 0)); // 180° around X
        
        // Additional rotations for medium and large problems
        if (vertexCount > 50 || containerSize.X > 20)
        {
            // Add more rotations (45-degree increments)
            possibleRotations.Add(new Quaternion(0.9239f, 0, 0, 0.3827f)); // 45° around Z
            possibleRotations.Add(new Quaternion(0.9239f, 0, 0.3827f, 0)); // 45° around Y
            possibleRotations.Add(new Quaternion(0.9239f, 0.3827f, 0, 0)); // 45° around X
        }
    }
    
    private void PackObjects()
    {
        // Simple greedy packing algorithm
        // For each object, try to place it in the best position
        int maxAttempts = 1000;
        int attempts = 0;
        
        while (attempts < maxAttempts)
        {
            var bestPosition = FindBestPosition();
            
            if (bestPosition != null)
            {
                packedObjects.Add(bestPosition);
                attempts = 0; // Reset attempts when successful placement
            }
            else
            {
                attempts++;
            }
        }
    }
    
    private PackedObject FindBestPosition()
    {
        PackedObject bestPosition = null;
        float bestScore = float.MinValue;
        
        // Try different rotations
        foreach (var rotation in possibleRotations)
        {
            // Try different positions
            for (float x = 0; x <= containerSize.X; x += 0.5f)
            {
                for (float y = 0; y <= containerSize.Y; y += 0.5f)
                {
                    for (float z = 0; z <= containerSize.Z; z += 0.5f)
                    {
                        var translation = new Vector3(x, y, z);
                        var testObject = new PackedObject(translation, rotation);
                        
                        if (IsValidPlacement(testObject))
                        {
                            float score = CalculatePlacementScore(testObject);
                            if (score > bestScore)
                            {
                                bestScore = score;
                                bestPosition = testObject;
                            }
                        }
                    }
                }
            }
        }
        
        return bestPosition;
    }
    
    private bool IsValidPlacement(PackedObject obj)
    {
        // Check if object is within container bounds
        var bounds = obj.GetBounds();
        if (bounds.Max.X > containerSize.X || bounds.Max.Y > containerSize.Y || bounds.Max.Z > containerSize.Z)
            return false;
        if (bounds.Min.X < 0 || bounds.Min.Y < 0 || bounds.Min.Z < 0)
            return false;
        
        // Check collision with other packed objects
        foreach (var packed in packedObjects)
        {
            if (CheckCollision(obj, packed))
                return false;
        }
        
        return true;
    }
    
    private bool CheckCollision(PackedObject obj1, PackedObject obj2)
    {
        // Simple bounding box collision check (can be improved with more detailed collision detection)
        var bounds1 = obj1.GetBounds();
        var bounds2 = obj2.GetBounds();
        
        return !(bounds1.Max.X < bounds2.Min.X || bounds1.Min.X > bounds2.Max.X ||
                 bounds1.Max.Y < bounds2.Min.Y || bounds1.Min.Y > bounds2.Max.Y ||
                 bounds1.Max.Z < bounds2.Min.Z || bounds1.Min.Z > bounds2.Max.Z);
    }
    
    private float CalculatePlacementScore(PackedObject obj)
    {
        // Simple scoring: prefer positions that are lower and more compact
        float score = -obj.Translation.Y; // Prefer lower positions
        
        // Add penalty for wasted space
        var bounds = obj.GetBounds();
        float wastedSpace = (bounds.Max.X - bounds.Min.X) * (bounds.Max.Y - bounds.Min.Y) * (bounds.Max.Z - bounds.Min.Z) - polyhedronVolume;
        score -= wastedSpace * 0.01f;
        
        return score;
    }
    
    private void OutputResults()
    {
        Console.WriteLine(packedObjects.Count);
        foreach (var obj in packedObjects)
        {
            Console.WriteLine($"{obj.Translation.X} {obj.Translation.Y} {obj.Translation.Z} " +
                            $"{obj.Rotation.W} {obj.Rotation.X} {obj.Rotation.Y} {obj.Rotation.Z}");
        }
    }
    
    private double CalculateVolume(Polyhedron polyhedron)
    {
        // Simple volume approximation using bounding box
        var bounds = polyhedron.GetBounds();
        return (bounds.Max.X - bounds.Min.X) * (bounds.Max.Y - bounds.Min.Y) * (bounds.Max.Z - bounds.Min.Z);
    }
}

public class Polyhedron
{
    public List<Vector3> Vertices { get; }
    public List<int[]> Faces { get; }
    
    public Polyhedron(List<Vector3> vertices, List<int[]> faces)
    {
        Vertices = vertices;
        Faces = faces;
    }
    
    public (Vector3 Min, Vector3 Max) GetBounds()
    {
        if (Vertices.Count == 0)
            return (new Vector3(0, 0, 0), new Vector3(0, 0, 0));
        
        var min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
        var max = new Vector3(float.MinValue, float.MinValue, float.MinValue);
        
        foreach (var vertex in Vertices)
        {
            min = Vector3.Min(min, vertex);
            max = Vector3.Max(max, vertex);
        }
        
        return (min, max);
    }
}

public class PackedObject
{
    public Vector3 Translation { get; }
    public Quaternion Rotation { get; }
    
    public PackedObject(Vector3 translation, Quaternion rotation)
    {
        Translation = translation;
        Rotation = rotation;
    }
    
    public (Vector3 Min, Vector3 Max) GetBounds()
    {
        // This is a simplified bounding box calculation
        // In a real implementation, you would transform all vertices and find min/max
        var bounds = new Polyhedron(new List<Vector3>(), new List<int[]>()).GetBounds();
        
        // Apply rotation and translation
        var rotatedBounds = new Vector3(
            bounds.Max.X * Math.Abs(Rotation.W) + bounds.Max.Y * Math.Abs(Rotation.X) + bounds.Max.Z * Math.Abs(Rotation.Y),
            bounds.Max.X * Math.Abs(Rotation.X) + bounds.Max.Y * Math.Abs(Rotation.W) + bounds.Max.Z * Math.Abs(Rotation.Z),
            bounds.Max.X * Math.Abs(Rotation.Y) + bounds.Max.Y * Math.Abs(Rotation.Z) + bounds.Max.Z * Math.Abs(Rotation.W)
        );
        
        return (Translation - rotatedBounds, Translation + rotatedBounds);
    }
}
