
```spirv
; Capabilities
OpCapability Geometry
OpCapability Shader

; Memory model
OpMemoryModel Logical GLSL450

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input arrays
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs
%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3float = OpTypePointer Input %arr_v3float_6
%worldPos = OpVariable %ptr_in_arr_v3float Input
%normal = OpVariable %ptr_in_arr_v3float Input
%color = OpVariable %ptr_in_arr_v3float Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Output arrays
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined output
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; UBO
%mat4 = OpTypeMatrix %v4float 4
%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %float
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform

; Constants
%thickness = OpConstant %float 0.008
%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5

; Function
%main = OpFunction %void None %void
%entry = OpLabel

; Get positions of vertices 0 and 2
%pos0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%pos0_val = OpLoad %v4float %pos0
%pos2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
%pos2_val = OpLoad %v4float %pos2

; Get colors of vertices 0 and 2
%color0 = OpAccessChain %ptr_in_arr_v3float %color %int_0
%color0_val = OpLoad %v3float %color0
%color2 = OpAccessChain %ptr_in_arr_v3float %color %int_2
%color2_val = OpLoad %v3float %color2

; Compute edge direction and perpendicular
%edge_dir = OpFSub %v3float %pos2_val %pos0_val
%edge_dir_xy = OpVectorShuffle %v2float %edge_dir %edge_dir 0 1
%perp = OpVectorShuffle %v2float %edge_dir_xy %edge_dir_xy 1 0
%perp = OpFNeg %v2float %perp
%perp = OpCompositeConstruct %v3float %perp %float_0
%perp = OpNormalize %v3float %perp

; Scale perpendicular by thickness
%perp_scaled = OpFMul %v3float %perp %thickness

; Compute the 4 vertices of the quad
%v1 = OpFAdd %v3float %pos0_val %perp_scaled
%v2 = OpFSub %v3float %pos0_val %perp_scaled
%v3 = OpFAdd %v3float %pos2_val %perp_scaled
%v4 = OpFSub %v3float %pos2_val %perp_scaled

; Compute average color
%avg_color = OpFAdd %v3float %color0_val %color2_val
%avg_color = OpFMul %v3float %avg_color %float_0_5

; Emit vertices
%gl_out_pos = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %gl_out_pos %v1
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v2
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v3
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v4
OpStore %outColor %avg_color
OpEmitVertex

OpEndPrimitive

; Edge 2→4
%pos4 = OpAccessChain %ptr_in_arr_pv %gl_in %int_4 %int_0
%pos4_val = OpLoad %v4float %pos4
%color4 = OpAccessChain %ptr_in_arr_v3float %color %int_4
%color4_val = OpLoad %v3float %color4

%edge_dir = OpFSub %v3float %pos4_val %pos2_val
%edge_dir_xy = OpVectorShuffle %v2float %edge_dir %edge_dir 0 1
%perp = OpVectorShuffle %v2float %edge_dir_xy %edge_dir_xy 1 0
%perp = OpFNeg %v2float %perp
%perp = OpCompositeConstruct %v3float %perp %float_0
%perp = OpNormalize %v3float %perp
%perp_scaled = OpFMul %v3float %perp %thickness

%v1 = OpFAdd %v3float %pos2_val %perp_scaled
%v2 = OpFSub %v3float %pos2_val %perp_scaled
%v3 = OpFAdd %v3float %pos4_val %perp_scaled
%v4 = OpFSub %v3float %pos4_val %perp_scaled

%avg_color = OpFAdd %v3float %color2_val %color4_val
%avg_color = OpFMul %v3float %avg_color %float_0_5

OpStore %gl_out_pos %v1
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v2
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v3
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v4
OpStore %outColor %avg_color
OpEmitVertex

OpEndPrimitive

; Edge 4→0
%edge_dir = OpFSub %v3float %pos0_val %pos4_val
%edge_dir_xy = OpVectorShuffle %v2float %edge_dir %edge_dir 0 1
%perp = OpVectorShuffle %v2float %edge_dir_xy %edge_dir_xy 1 0
%perp = OpFNeg %v2float %perp
%perp = OpCompositeConstruct %v3float %perp %float_0
%perp = OpNormalize %v3float %perp
%perp_scaled = OpFMul %v3float %perp %thickness

%v1 = OpFAdd %v3float %pos4_val %perp_scaled
%v2 = OpFSub %v3float %pos4_val %perp_scaled
%v3 = OpFAdd %v3float %pos0_val %perp_scaled
%v4 = OpFSub %v3float %pos0_val %perp_scaled

%avg_color = OpFAdd %v3float %color4_val %color0_val
%avg_color = OpFMul %v3float %avg_color %float_0_5

OpStore %gl_out_pos %v1
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v2
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v3
OpStore %outColor %avg_color
OpEmitVertex

OpStore %gl_out_pos %v4
OpStore %outColor %avg_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
