
```spirv
; SPIR-V Geometry Shader: Line to Ribbon
; Input: InputLines (2 vertices)
; Output: OutputTriangleStrip (4 vertices per line)

; Capabilities and Memory Model
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Basic Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 0

; gl_PerVertex Input Struct
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_PerVertex Output Struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input/Output Pointer Types
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

; UBO (declared but not used in this specific logic)
%ubo_data_type = OpTypeStruct ... ; Placeholder for the actual UBO struct
%ptr_ubo = OpTypePointer Uniform %ubo_data_type
%ubo = OpVariable %ptr_ubo Uniform

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_03 = OpConstant %float 0.03
%float_0_0 = OpConstant %float 0.0
%float_1_0 = OpConstant %float 1.0

; Input Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_v3 Input
%normal_in = OpVariable %ptr_in_v3 Input
%color_in = OpVariable %ptr_in_v3 Input

; Output Variables
%gl_out = OpVariable %ptr_out_pv Output
%outColor_out = OpVariable %ptr_out_v3 Output

; Main Function
%main_fn = OpFunction %void None
%entry_point = OpLabel
    ; Load input positions: p0 = gl_in[0].gl_Position, p1 = gl_in[1].gl_Position
    %pv0_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
    %pv1_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_1
    %p0_ptr = OpAccessChain %ptr_in_v4 %pv0_ptr %int_0
    %p1_ptr = OpAccessChain %ptr_in_v4 %pv1_ptr %int_0
    %p0 = OpLoad %v4float %p0_ptr
    %p1 = OpLoad %v4float %p1_ptr

    ; Load input colors: c0 = color[0], c1 = color[1]
    %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
    %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
    %c0 = OpLoad %v3float %c0_ptr
    %c1 = OpLoad %v3float %c1_ptr

    ; Calculate direction vector: dir = p1 - p0
    %dir = OpFSub %v4float %p1 %p0

    ; Calculate perpendicular vector: perp = normalize(-dir.y, dir.x, 0, 0) * thickness
    ; Extract components
    %dir_x = OpCompositeExtract %float %dir 0
    %dir_y = OpCompositeExtract %float %dir 1
    ; Negate y and multiply by thickness
    %neg_dir_y = OpFNegate %float %dir_y
    %perp_x = OpFMul %float %neg_dir_y %float_0_03
    ; Multiply x by thickness
    %perp_y = OpFMul %float %dir_x %float_0_03
    ; Construct perpendicular vector
    %perp = OpCompositeConstruct %v4float %perp_x %perp_y %float_0_0 %float_0_0

    ; Calculate quad vertices
    %p0_plus_perp = OpFAdd %v4float %p0 %perp
    %p0_minus_perp = OpFSub %v4float %p0 %perp
    %p1_plus_perp = OpFAdd %v4float %p1 %perp
    %p1_minus_perp = OpFSub %v4float %p1 %perp

    ; Emit Vertex 0: p0 + perp, color c0
    %out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
    OpStore %out_pos_ptr %p0_plus_perp
    OpStore %outColor_out %c0
    OpEmitVertex

    ; Emit Vertex 1: p0 - perp, color c0
    OpStore %out_pos_ptr %p0_minus_perp
    OpStore %outColor_out %c0
    OpEmitVertex

    ; Emit Vertex 2: p1 + perp, color c1
    OpStore %out_pos_ptr %p1_plus_perp
    OpStore %outColor_out %c1
    OpEmitVertex

    ; Emit Vertex 3: p1 - perp, color c1
    OpStore %out_pos_ptr %p1_minus_perp
    OpStore %outColor_out %c1
    OpEmitVertex

    ; End the primitive (the triangle strip)
    OpEndPrimitive
OpReturn
OpFunctionEnd

; Execution Modes
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Entry Point
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor_out
```
