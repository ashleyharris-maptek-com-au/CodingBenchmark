#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <thread>
#include <mutex>
#include <functional>

struct City {
    float x, y;
};

float distance(const City& a, const City& b) {
    return std::hypotf(a.x - b.x, a.y - b.y);
}

// Hilbert curve implementation to order cities
unsigned long long hilbert_index(float x, float y, float min_x, float max_x, float min_y, float max_y, int order) {
    // Scale the coordinates to the range [0, 2^order - 1]
    int scaled_x = static_cast<int>((x - min_x) / (max_x - min_x) * ((1ULL << order) - 1));
    int scaled_y = static_cast<int>((y - min_y) / (max_y - min_y) * ((1ULL << order) - 1));
    // Ensure the scaled coordinates are within the range
    scaled_x = std::max(0, std::min(scaled_x, (1ULL << order) - 1));
    scaled_y = std::max(0, std::min(scaled_y, (1ULL << order) - 1));
    // Compute the Hilbert index
    unsigned long long index = 0;
    for (int i = order - 1; i >= 0; --i) {
        unsigned long long quad_x = (scaled_x >> i) & 1;
        unsigned long long quad_y = (scaled_y >> i) & 1;
        index = (index << 2) | (quad_x << 1) | quad_y;
        if (quad_y == 0) {
            if (quad_x == 1) {
                scaled_x = ~scaled_x;
                scaled_y = ~scaled_y;
            }
            std::swap(scaled_x, scaled_y);
        }
    }
    return index;
}

// 2-opt heuristic to improve the tour
void two_opt(std::vector<int>& tour, const std::vector<City>& cities) {
    bool improved = true;
    int n = tour.size();
    while (improved) {
        improved = false;
        for (int i = 0; i < n - 2; ++i) {
            for (int j = i + 2; j < n; ++j) {
                // Check if reversing the segment [i+1, j] improves the tour
                float delta = distance(cities[tour[i]], cities[tour[j]]) + distance(cities[tour[i + 1]], cities[tour[j + 1]])
                           - distance(cities[tour[i]], cities[tour[i + 1]]) - distance(cities[tour[j]], cities[tour[j + 1]]);
                if (delta < 0) {
                    // Reverse the segment [i+1, j]
                    std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Nearest Neighbor heuristic to generate an initial tour
std::vector<int> nearest_neighbor_tour(const std::vector<City>& cities) {
    int n = cities.size();
    std::vector<int> tour;
    std::vector<bool> visited(n, false);
    tour.push_back(0); // Start at city 0
    visited[0] = true;
    for (int i = 1; i < n; ++i) {
        int current_city = tour.back();
        int nearest_city = -1;
        float min_distance = std::numeric_limits<float>::max();
        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                float d = distance(cities[current_city], cities[j]);
                if (d < min_distance) {
                    min_distance = d;
                    nearest_city = j;
                }
            }
        }
        tour.push_back(nearest_city);
        visited[nearest_city] = true;
    }
    return tour;
}

// Farthest Insertion heuristic to generate an initial tour
std::vector<int> farthest_insertion_tour(const std::vector<City>& cities) {
    int n = cities.size();
    std::vector<int> tour;
    std::vector<bool> visited(n, false);
    // Start with two farthest cities
    int farthest_city1 = 0;
    int farthest_city2 = 1;
    float max_distance = distance(cities[0], cities[1]);
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            float d = distance(cities[i], cities[j]);
            if (d > max_distance) {
                max_distance = d;
                farthest_city1 = i;
                farthest_city2 = j;
            }
        }
    }
    tour.push_back(farthest_city1);
    tour.push_back(farthest_city2);
    visited[farthest_city1] = true;
    visited[farthest_city2] = true;
    // Insert the remaining cities
    for (int i = 0; i < n - 2; ++i) {
        int farthest_city = -1;
        int best_position = -1;
        float max_distance = -1;
        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                // Find the best position to insert city j
                float min_increase = std::numeric_limits<float>::max();
                int best_pos = -1;
                for (int k = 0; k < tour.size(); ++k) {
                    int next_city = (k + 1) % tour.size();
                    float increase = distance(cities[tour[k]], cities[j]) + distance(cities[j], cities[tour[next_city]])
                                  - distance(cities[tour[k]], cities[tour[next_city]]);
                    if (increase < min_increase) {
                        min_increase = increase;
                        best_pos = k;
                    }
                }
                if (min_increase > max_distance) {
                    max_distance = min_increase;
                    farthest_city = j;
                    best_position = best_pos;
                }
            }
        }
        // Insert the farthest city at the best position
        tour.insert(tour.begin() + best_position + 1, farthest_city);
        visited[farthest_city] = true;
    }
    return tour;
}

// Held-Karp DP to solve TSP exactly for small N
std::vector<int> held_karp_tsp(const std::vector<City>& cities) {
    int n = cities.size();
    if (n == 0) return {};
    if (n == 1) return {0};
    // dp[mask][i] = minimum cost to visit all cities in mask, ending at city i
    std::vector<std::vector<float>> dp(1 << n, std::vector<float>(n, std::numeric_limits<float>::max()));
    // Base case: dp[1 << 0][0] = 0
    dp[1 << 0][0] = 0;
    // Iterate over all masks
    for (int mask = 1; mask < (1 << n); ++mask) {
        if ((mask & (1 << 0)) == 0) continue; // Start city must be in the mask
        for (int u = 0; u < n; ++u) {
            if ((mask & (1 << u)) == 0) continue; // City u must be in the mask
            if (u == 0 && mask != (1 << 0)) continue; // City 0 is the start city
            for (int v = 0; v < n; ++v) {
                if ((mask & (1 << v)) == 0) {
                    // Try to add city v to the tour
                    int new_mask = mask | (1 << v);
                    if (dp[mask][u] + distance(cities[u], cities[v]) < dp[new_mask][v]) {
                        dp[new_mask][v] = dp[mask][u] + distance(cities[u], cities[v]);
                    }
                }
            }
        }
    }
    // Find the minimum cost tour that visits all cities and ends at city 0
    float min_cost = std::numeric_limits<float>::max();
    int last_city = -1;
    for (int u = 1; u < n; ++u) {
        if (dp[(1 << n) - 1][u] + distance(cities[u], cities[0]) < min_cost) {
            min_cost = dp[(1 << n) - 1][u] + distance(cities[u], cities[0]);
            last_city = u;
        }
    }
    // Reconstruct the tour
    std::vector<int> tour;
    int mask = (1 << n) - 1;
    tour.push_back(0);
    while (last_city != 0) {
        tour.push_back(last_city);
        int prev_city = -1;
        for (int u = 0; u < n; ++u) {
            if ((mask & (1 << u)) && u != last_city) {
                if (dp[mask][last_city] == dp[mask ^ (1 << last_city)][u] + distance(cities[u], cities[last_city])) {
                    prev_city = u;
                    break;
                }
            }
        }
        mask ^= (1 << last_city);
        last_city = prev_city;
    }
    std::reverse(tour.begin() + 1, tour.end());
    return tour;
}

int main() {
    int n;
    std::cin >> n;
    std::vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }
    std::vector<int> tour;
    if (n <= 20) {
        tour = held_karp_tsp(cities);
    } else if (n <= 100) {
        tour = nearest_neighbor_tour(cities);
        two_opt(tour, cities);
    } else if (n <= 1000) {
        tour = farthest_insertion_tour(cities);
        two_opt(tour, cities);
    } else if (n <= 10000) {
        // Use Hilbert curve + 2-opt
        // Compute the bounding box
        float min_x = std::numeric_limits<float>::max();
        float max_x = std::numeric_limits<float>::min();
        float min_y = std::numeric_limits<float>::max();
        float max_y = std::numeric_limits<float>::min();
        for (const auto& city : cities) {
            min_x = std::min(min_x, city.x);
            max_x = std::max(max_x, city.x);
            min_y = std::min(min_y, city.y);
            max_y = std::max(max_y, city.y);
        }
        // Choose the order of the Hilbert curve
        int order = std::ceil(std::log2(n));
        // Compute the Hilbert index for each city
        std::vector<std::pair<unsigned long long, int>> indexed_cities;
        for (int i = 0; i < n; ++i) {
            unsigned long long index = hilbert_index(cities[i].x, cities[i].y, min_x, max_x, min_y, max_y, order);
            indexed_cities.emplace_back(index, i);
        }
        // Sort the cities by their Hilbert index
        std::sort(indexed_cities.begin(), indexed_cities.end());
        // Extract the tour
        tour.resize(n);
        for (int i = 0; i < n; ++i) {
            tour[i] = indexed_cities[i].second;
        }
        // Run 2-opt for a limited number of iterations
        two_opt(tour, cities);
    } else {
        // Use Hilbert curve only
        // Compute the bounding box
        float min_x = std::numeric_limits<float>::max();
        float max_x = std::numeric_limits<float>::min();
        float min_y = std::numeric_limits<float>::max();
        float max_y = std::numeric_limits<float>::min();
        for (const auto& city : cities) {
            min_x = std::min(min_x, city.x);
            max_x = std::max(max_x, city.x);
            min_y = std::min(min_y, city.y);
            max_y = std::max(max_y, city.y);
        }
        // Choose the order of the Hilbert curve
        int order = std::ceil(std::log2(n));
        // Compute the Hilbert index for each city
        std::vector<std::pair<unsigned long long, int>> indexed_cities;
        for (int i = 0; i < n; ++i) {
            unsigned long long index = hilbert_index(cities[i].x, cities[i].y, min_x, max_x, min_y, max_y, order);
            indexed_cities.emplace_back(index, i);
        }
        // Sort the cities by their Hilbert index
        std::sort(indexed_cities.begin(), indexed_cities.end());
        // Extract the tour
        tour.resize(n);
        for (int i = 0; i < n; ++i) {
            tour[i] = indexed_cities[i].second;
        }
    }
    // Print the tour
    for (int i = 0; i < n; ++i) {
        std::cout << tour[i] << " ";
    }
    std::cout << std::endl;
    return 0;
}
