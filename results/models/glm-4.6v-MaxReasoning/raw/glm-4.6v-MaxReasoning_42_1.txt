
```spirv
; SPIR-V Geometry Shader for expanding points into diamonds
; Input: InputPoints (1 vertex per invocation)
; Output: OutputTriangleStrip (4 vertices per point)

; Capabilities and Memory Model
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_arr = OpTypeArray %v3float %uint_1
%ptr_in_worldPos = OpTypePointer Input %worldPos_arr
%worldPos = OpVariable %ptr_in_worldPos Input
OpDecorate %worldPos Location 0
%normal_arr = OpTypeArray %v3float %uint_1
%ptr_in_normal = OpTypePointer Input %normal_arr
%normal = OpVariable %ptr_in_normal Input
OpDecorate %normal Location 1
%color_arr = OpTypeArray %v3float %uint_1
%ptr_in_color = OpTypePointer Input %color_arr
%color = OpVariable %ptr_in_color Input
OpDecorate %color Location 2
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%outColor_ptr = OpTypePointer Output %v3float
%outColor = OpVariable %outColor_ptr Output
OpDecorate %outColor Location 0

; Entry Point and Execution Modes
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Main Function
%main = OpFunction %void None %void
%entry = OpLabel
%idx0 = OpConstant %uint 0
%h = OpConstant %float 0.05

; Read input position (gl_in[0].gl_Position)
%gl_in_elem = OpAccessChain %ptr_in_arr_pv %gl_in %idx0
%gl_in_struct = OpLoad %arr_pv_in %gl_in_elem
%gl_in_pos_ptr = OpAccessChain %gl_PerVertex_in %gl_in_struct %uint_0
%pos_in = OpLoad %v4float %gl_in_pos_ptr

; Read input color (color[0])
%color_elem = OpAccessChain %ptr_in_color %color %idx0
%color_in = OpLoad %v3float %color_elem

; Create top vertex: (x, y+h, z, w)
%top_y = OpFAdd %float %pos_in 1 %h
%top_pos = OpCompositeConstruct %v4float %pos_in 0 %top_y %pos_in 2 %pos_in 3

; Create right vertex: (x+h, y, z, w)
%right_x = OpFAdd %float %pos_in 0 %h
%right_pos = OpCompositeConstruct %v4float %right_x %pos_in 1 %pos_in 2 %pos_in 3

; Create left vertex: (x-h, y, z, w)
%left_x = OpFSub %float %pos_in 0 %h
%left_pos = OpCompositeConstruct %v4float %left_x %pos_in 1 %pos_in 2 %pos_in 3

; Create bottom vertex: (x, y-h, z, w)
%bottom_y = OpFSub %float %pos_in 1 %h
%bottom_pos = OpCompositeConstruct %v4float %pos_in 0 %bottom_y %pos_in 2 %pos_in 3

; Emit top vertex
%gl_out_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %uint_0
OpStore %gl_out_pos_ptr %top_pos
OpStore %outColor %color_in
OpEmitVertex

; Emit right vertex
OpStore %gl_out_pos_ptr %right_pos
OpStore %outColor %color_in
OpEmitVertex

; Emit left vertex
OpStore %gl_out_pos_ptr %left_pos
OpStore %outColor %color_in
OpEmitVertex

; Emit bottom vertex
OpStore %gl_out_pos_ptr %bottom_pos
OpStore %outColor %color_in
OpEmitVertex

; End primitive
OpEndPrimitive
OpReturn
OpFunctionEnd
```
