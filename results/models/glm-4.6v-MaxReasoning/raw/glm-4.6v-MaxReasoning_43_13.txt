
```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float t = clamp(input.worldPos.y * 0.5 + 0.5, 0.0, 1.0);
    float3 bottomColor = float3(0.1, 0.4, 0.1);
    float3 topColor = float3(1.0, 1.0, 1.0);
    float3 color = lerp(bottomColor, topColor, t);
    return float4(color, 1.0);
}
```
