using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

class MaximumClique
{
    static int n; // number of vertices
    static List<int>[] adj; // adjacency list
    static int[] degree; // degree of each vertex
    static int bestSize; // size of the best clique found
    static List<int> bestClique; // the best clique found
    static DateTime startTime;
    static TimeSpan timeLimit = TimeSpan.FromMinutes(4); // 4 minutes for exact search

    static void Main()
    {
        ReadInput();
        startTime = DateTime.Now;

        // Greedy heuristic to get an initial upper bound
        List<int> greedyClique = GreedyClique();
        bestSize = greedyClique.Count;
        bestClique = greedyClique;

        // Bron-Kerbosch with pivot and pruning
        BronKerbosch();

        // Output the result
        Console.WriteLine(bestSize);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(v => v)));
    }

    static void ReadInput()
    {
        var parts = Console.ReadLine().Split();
        n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        adj = new List<int>[n];
        degree = new int[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            adj[u].Add(v);
            adj[v].Add(u);
            degree[u]++;
            degree[v]++;
        }
    }

    static List<int> GreedyClique()
    {
        // Greedy algorithm: iteratively add the vertex with the highest degree in the current graph
        List<int> clique = new List<int>();
        bool[] inClique = new bool[n];
        bool[] removed = new bool[n];

        for (int i = 0; i < n; i++)
        {
            int bestVertex = -1;
            int maxDegree = -1;

            for (int v = 0; v < n; v++)
            {
                if (!inClique[v] && !removed[v])
                {
                    int deg = 0;
                    foreach (int neighbor in adj[v])
                    {
                        if (!removed[neighbor])
                        {
                            deg++;
                        }
                    }
                    if (deg > maxDegree)
                    {
                        maxDegree = deg;
                        bestVertex = v;
                    }
                }
            }

            if (bestVertex == -1) break;

            // Check if it can be added to the clique
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!adj[bestVertex].Contains(u))
                {
                    canAdd = false;
                    break;
                }
            }

            if (canAdd)
            {
                clique.Add(bestVertex);
                inClique[bestVertex] = true;
            }
            else
            {
                removed[bestVertex] = true;
            }
        }

        return clique;
    }

    static void BronKerbosch()
    {
        // We will use a bitset representation for sets
        // Each bitset is an array of ulongs, where each ulong represents 64 bits
        int words = (n + 63) / 64;
        ulong[] all = new ulong[words];
        for (int i = 0; i < words; i++)
        {
            all[i] = ulong.MaxValue;
        }

        // We will use a stack-based approach to avoid recursion depth issues
        // R: current clique being built
        // P: potential vertices to add
        // X: vertices already processed
        Stack<(List<int> R, ulong[] P, ulong[] X)> stack = new Stack<(List<int>, ulong[], ulong[])>();
        stack.Push((new List<int>(), all, new ulong[words]));

        while (stack.Count > 0)
        {
            var (R, P, X) = stack.Pop();

            // Check time limit
            if (DateTime.Now - startTime > timeLimit)
            {
                return; // Time is up, stop the search
            }

            // If P and X are empty, we have a maximal clique
            if (IsEmpty(P) && IsEmpty(X))
            {
                if (R.Count > bestSize)
                {
                    bestSize = R.Count;
                    bestClique = new List<int>(R);
                }
                continue;
            }

            // Choose a pivot u from P union X
            int u = ChoosePivot(P, X);

            // The set of vertices to consider: P \\ N(u)
            ulong[] candidates = Difference(P, GetNeighbors(u));

            // Iterate over each vertex v in candidates
            for (int i = 0; i < words; i++)
            {
                ulong word = candidates[i];
                while (word != 0)
                {
                    int v = (i * 64) + BitOperations.TrailingZeroCount(word);
                    word &= word - 1; // Remove the lowest set bit

                    // Check if v is within bounds
                    if (v >= n) continue;

                    // Add v to R
                    R.Add(v);

                    // P intersect N(v)
                    ulong[] newP = Intersection(P, GetNeighbors(v));

                    // X intersect N(v)
                    ulong[] newX = Intersection(X, GetNeighbors(v));

                    // Prune if the size of newP plus R is less than bestSize
                    if (R.Count + CountBits(newP) > bestSize)
                    {
                        stack.Push((new List<int>(R), newP, newX));
                    }

                    // Backtrack: remove v from R
                    R.RemoveAt(R.Count - 1);
                }
            }
        }
    }

    static bool IsEmpty(ulong[] set)
    {
        foreach (var word in set)
        {
            if (word != 0) return false;
        }
        return true;
    }

    static int ChoosePivot(ulong[] P, ulong[] X)
    {
        // Choose the vertex with the highest degree in P union X
        int bestVertex = -1;
        int maxDegree = -1;

        for (int v = 0; v < n; v++)
        {
            if (IsInSet(P, v) || IsInSet(X, v))
            {
                int deg = 0;
                foreach (int neighbor in adj[v])
                {
                    if (IsInSet(P, neighbor) || IsInSet(X, neighbor))
                    {
                        deg++;
                    }
                }
                if (deg > maxDegree)
                {
                    maxDegree = deg;
                    bestVertex = v;
                }
            }
        }

        return bestVertex;
    }

    static ulong[] GetNeighbors(int v)
    {
        // Return a bitset of the neighbors of v
        int words = (n + 63) / 64;
        ulong[] neighbors = new ulong[words];
        foreach (int u in adj[v])
        {
            SetBit(neighbors, u);
        }
        return neighbors;
    }

    static void SetBit(ulong[] set, int v)
    {
        int wordIndex = v / 64;
        int bitIndex = v % 64;
        set[wordIndex] |= 1UL << bitIndex;
    }

    static bool IsInSet(ulong[] set, int v)
    {
        int wordIndex = v / 64;
        int bitIndex = v % 64;
        return (set[wordIndex] & (1UL << bitIndex)) != 0;
    }

    static ulong[] Intersection(ulong[] a, ulong[] b)
    {
        int words = a.Length;
        ulong[] result = new ulong[words];
        for (int i = 0; i < words; i++)
        {
            result[i] = a[i] & b[i];
        }
        return result;
    }

    static ulong[] Difference(ulong[] a, ulong[] b)
    {
        int words = a.Length;
        ulong[] result = new ulong[words];
        for (int i = 0; i < words; i++)
        {
            result[i] = a[i] & ~b[i];
        }
        return result;
    }

    static int CountBits(ulong[] set)
    {
        int count = 0;
        foreach (var word in set)
        {
            count += BitOperations.PopCount(word);
        }
        return count;
    }
}
