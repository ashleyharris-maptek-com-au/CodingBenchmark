
```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Calculate cell coordinates for 6x6 pattern
    float2 cellUV = input.uv * 6.0;
    float2 cellCenter = floor(cellUV) + 0.5; // Center of each cell
    float2 distFromCenter = abs(cellUV - cellCenter);
    
    // Check if we're in a dot (circular)
    float dist = length(distFromCenter);
    float inDot = step(dist, 0.3); // 1.0 if in dot, 0.0 otherwise
    
    // Base colors
    float3 dotColor = float3(1.0, 0.8, 0.0); // Yellow
    float3 bgColor = float3(0.1, 0.1, 0.3); // Dark blue
    
    // Mix colors based on dot pattern
    float3 baseColor = lerp(bgColor, dotColor, inDot);
    
    // Calculate lighting
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float3 normal = normalize(input.normal);
    
    // Lambertian diffuse
    float diffuse = max(dot(normal, lightDir), 0.0);
    
    // Apply Lambertian shading factor
    float3 shadedColor = baseColor * (0.3 + 0.7 * diffuse);
    
    return float4(shadedColor, 1.0);
}
```
