OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%mat4 = OpTypeMatrix %float 4 4
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Uniform_ubo = OpTypePointer Uniform %ubo
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float

; Variables
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%tangent = OpVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v4float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_ubo Uniform

; Constants
%skyColor = OpConstantComposite %v3float %float 0.4 %float 0.6 %float 1.0
%groundColor = OpConstantComposite %v3float %float 0.3 %float 0.15 %float 0.05
%one = OpConstant %float 1.0
%zero = OpConstant %float 0.0
%half = OpConstant %float 0.5
%three_tenths = OpConstant %float 0.3

; Main function
%main = OpFunction %void None %void
%entry = OpLabel
  ; Get normal and normalize it
  %norm = OpLoad %v3float %normal
  %norm_norm = OpNormalize %v3float %norm
  
  ; Calculate blend factor: N.y * 0.5 + 0.5
  %ny = OpVectorExtractDynamic %float %norm_norm %zero
  %blend = OpFMul %float %ny %half
  %blend_final = OpFAdd %float %blend %half
  
  ; Calculate ambient color: lerp(groundColor, skyColor, blend)
  %skyColor_x_blend = OpVectorTimesScalar %v3float %skyColor %blend_final
  %groundColor_x_one_minus_blend = OpVectorTimesScalar %v3float %groundColor %OpFSub %float %one %blend_final
  %ambient = OpFAdd %v3float %skyColor_x_blend %groundColor_x_one_minus_blend
  
  ; Add diffuse term (0.3 * d)
  ; For a unit sphere, the diffuse term can be approximated as 0.3 * max(0, N.y)
  %diffuse_factor = OpFMax %float %ny %zero
  %diffuse = OpFMul %float %three_tenths %diffuse_factor
  %diffuse_color = OpCompositeConstruct %v3float %diffuse %diffuse %diffuse
  
  ; Combine ambient and diffuse
  %final_color = OpFAdd %v3float %ambient %diffuse_color
  
  ; Set alpha to 1.0
  %alpha = OpCompositeConstruct %v4float %final_color %one
  
  ; Output the color
  OpStore %outColor %alpha
  
  OpReturn
OpFunctionEnd
