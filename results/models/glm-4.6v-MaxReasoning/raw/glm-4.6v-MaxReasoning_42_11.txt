
```spirv
; Capabilities
OpCapability Geometry
OpCapability Shader

; Memory model
OpMemoryModel Logical GLSL450

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6
%int_7 = OpConstant %int 7
%float_0 = OpConstant %float 0
%float_0_02 = OpConstant %float 0.02
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3

; Input gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs
%worldPos = OpVariable %ptr_in_arr_pv Input
%normal = OpVariable %ptr_in_arr_pv Input
%color = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined output
%outColor = OpVariable %ptr_out_pv Output

; Main function
%main = OpFunction %void None %void
%entry = OpLabel

; Get start position (gl_in[1].gl_Position)
%idx1 = OpConstant %int 1
%gl_in_1 = OpAccessChain %ptr_in_arr_pv %gl_in %idx1
%gl_PerVertex_in_1 = OpLoad %arr_pv_in %gl_in_1
%gl_PerVertex_in_1_0 = OpAccessChain %gl_PerVertex_in %gl_PerVertex_in_1 %int_0
%start_pos = OpLoad %v4float %gl_PerVertex_in_1_0

; Get end position (gl_in[2].gl_Position)
%idx2 = OpConstant %int 2
%gl_in_2 = OpAccessChain %ptr_in_arr_pv %gl_in %idx2
%gl_PerVertex_in_2 = OpLoad %arr_pv_in %gl_in_2
%gl_PerVertex_in_2_0 = OpAccessChain %gl_PerVertex_in %gl_PerVertex_in_2 %int_0
%end_pos = OpLoad %v4float %gl_PerVertex_in_2_0

; Get color[1]
%color_idx1 = OpAccessChain %ptr_in_arr_pv %color %idx1
%color_1 = OpLoad %v3float %color_idx1

; Get color[2]
%color_idx2 = OpAccessChain %ptr_in_arr_pv %color %idx2
%color_2 = OpLoad %v3float %color_idx2

; Get previous vertex (gl_in[0].gl_Position)
%idx0 = OpConstant %int 0
%gl_in_0 = OpAccessChain %ptr_in_arr_pv %gl_in %idx0
%gl_PerVertex_in_0 = OpLoad %arr_pv_in %gl_in_0
%gl_PerVertex_in_0_0 = OpAccessChain %gl_PerVertex_in %gl_PerVertex_in_0 %int_0
%prev_pos = OpLoad %v4float %gl_PerVertex_in_0_0

; Get next vertex (gl_in[3].gl_Position)
%idx3 = OpConstant %int 3
%gl_in_3 = OpAccessChain %ptr_in_arr_pv %gl_in %idx3
%gl_PerVertex_in_3 = OpLoad %arr_pv_in %gl_in_3
%gl_PerVertex_in_3_0 = OpAccessChain %gl_PerVertex_in %gl_PerVertex_in_3 %int_0
%next_pos = OpLoad %v4float %gl_PerVertex_in_3_0

; Compute tangent at start (from prev to start)
%tangent_start = OpFSub %start_pos %prev_pos

; Compute tangent at end (from end to next)
%tangent_end = OpFSub %next_pos %end_pos

; Average the tangents for a smooth curve
%tangent = OpFAdd %tangent_start %tangent_end
%tangent = OpFMul %tangent %float_0_5

; Normalize tangent
%tangent_x = OpCompositeExtract %float %tangent 0
%tangent_y = OpCompositeExtract %float %tangent 1
%tangent_z = OpCompositeExtract %float %tangent 2
%tangent_w = OpCompositeExtract %float %tangent 3

%tangent_len_sq = OpFMul %tangent_x %tangent_x
%tangent_len_sq = OpFAdd %tangent_len_sq (OpFMul %tangent_y %tangent_y)
%tangent_len_sq = OpFAdd %tangent_len_sq (OpFMul %tangent_z %tangent_z)
%tangent_len_sq = OpFAdd %tangent_len_sq (OpFMul %tangent_w %tangent_w)

%tangent_len = OpSqrt %tangent_len_sq

%one_over_len = OpFDiv %float_1 %tangent_len
%tangent_norm = OpVectorTimesScalar %v4float %tangent %one_over_len

; Compute perpendicular (using cross product with up vector)
%up = OpConstantComposite %v4float %float_0 %float_1 %float_0 %float_0
%perp = OpCross %v4float %tangent_norm %up

; Normalize perpendicular
%perp_x = OpCompositeExtract %float %perp 0
%perp_y = OpCompositeExtract %float %perp 1
%perp_z = OpCompositeExtract %float %perp 2
%perp_w = OpCompositeExtract %float %perp 3

%perp_len_sq = OpFMul %perp_x %perp_x
%perp_len_sq = OpFAdd %perp_len_sq (OpFMul %perp_y %perp_y)
%perp_len_sq = OpFAdd %perp_len_sq (OpFMul %perp_z %perp_z)
%perp_len_sq = OpFAdd %perp_len_sq (OpFMul %perp_w %perp_w)

%perp_len = OpSqrt %perp_len_sq

%one_over_perp_len = OpFDiv %float_1 %perp_len
%perp_norm = OpVectorTimesScalar %v4float %perp %one_over_perp_len

; Compute segment length
%segment = OpFSub %end_pos %start_pos
%segment_x = OpCompositeExtract %float %segment 0
%segment_y = OpCompositeExtract %float %segment 1
%segment_z = OpCompositeExtract %float %segment 2
%segment_w = OpCompositeExtract %float %segment 3

%segment_len_sq = OpFMul %segment_x %segment_x
%segment_len_sq = OpFAdd %segment_len_sq (OpFMul %segment_y %segment_y)
%segment_len_sq = OpFAdd %segment_len_sq (OpFMul %segment_z %segment_z)
%segment_len_sq = OpFAdd %segment_len_sq (OpFMul %segment_w %segment_w)

%segment_len = OpSqrt %segment_len_sq

; Length of each part
%part_len = OpFDiv %segment_len %float_3

; Position for 1st part (start + part_len * tangent_norm)
%part1_pos = OpFAdd %start_pos (OpVectorTimesScalar %v4float %tangent_norm %part_len)

; Position for 3rd part (start + 2 * part_len * tangent_norm)
%part3_pos = OpFAdd %start_pos (OpVectorTimesScalar %v4float %tangent_norm %float_2)
%part3_pos = OpFAdd %part3_pos (OpVectorTimesScalar %v4float %tangent_norm %part_len)

; First part: 4 vertices
; Vertex 0: part1_pos - perp_norm * thickness
%v0_pos = OpFSub %part1_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)

; Vertex 1: part1_pos + perp_norm * thickness
%v1_pos = OpFAdd %part1_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)

; Vertex 2: part1_pos + perp_norm * thickness + tangent_norm * part_len
%v2_pos = OpFAdd %part1_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)
%v2_pos = OpFAdd %v2_pos (OpVectorTimesScalar %v4float %tangent_norm %part_len)

; Vertex 3: part1_pos - perp_norm * thickness + tangent_norm * part_len
%v3_pos = OpFSub %part1_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)
%v3_pos = OpFAdd %v3_pos (OpVectorTimesScalar %v4float %tangent_norm %part_len)

; Color for first part (interpolated between color_1 and color_2)
%color_part1 = OpFAdd %color_1 %color_2
%color_part1 = OpFMul %color_part1 %float_0_5

; Emit vertex 0
%gl_out_0 = OpAccessChain %ptr_out_pv %gl_out %int_0
%gl_PerVertex_out_0 = OpLoad %gl_PerVertex_out %gl_out_0
%gl_PerVertex_out_0_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_0 %int_0
OpStore %gl_PerVertex_out_0_0 %v0_pos
%outColor_0 = OpAccessChain %ptr_out_pv %outColor %int_0
OpStore %outColor_0 %color_part1
OpEmitVertex

; Emit vertex 1
%gl_out_1 = OpAccessChain %ptr_out_pv %gl_out %int_1
%gl_PerVertex_out_1 = OpLoad %gl_PerVertex_out %gl_out_1
%gl_PerVertex_out_1_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_1 %int_0
OpStore %gl_PerVertex_out_1_0 %v1_pos
%outColor_1 = OpAccessChain %ptr_out_pv %outColor %int_1
OpStore %outColor_1 %color_part1
OpEmitVertex

; Emit vertex 2
%gl_out_2 = OpAccessChain %ptr_out_pv %gl_out %int_2
%gl_PerVertex_out_2 = OpLoad %gl_PerVertex_out %gl_out_2
%gl_PerVertex_out_2_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_2 %int_0
OpStore %gl_PerVertex_out_2_0 %v2_pos
%outColor_2 = OpAccessChain %ptr_out_pv %outColor %int_2
OpStore %outColor_2 %color_part1
OpEmitVertex

; Emit vertex 3
%gl_out_3 = OpAccessChain %ptr_out_pv %gl_out %int_3
%gl_PerVertex_out_3 = OpLoad %gl_PerVertex_out %gl_out_3
%gl_PerVertex_out_3_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_3 %int_0
OpStore %gl_PerVertex_out_3_0 %v3_pos
%outColor_3 = OpAccessChain %ptr_out_pv %outColor %int_3
OpStore %outColor_3 %color_part1
OpEmitVertex

; End primitive
OpEndPrimitive

; Third part: 4 vertices
; Vertex 4: part3_pos - perp_norm * thickness
%v4_pos = OpFSub %part3_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)

; Vertex 5: part3_pos + perp_norm * thickness
%v5_pos = OpFAdd %part3_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)

; Vertex 6: part3_pos + perp_norm * thickness + tangent_norm * part_len
%v6_pos = OpFAdd %part3_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)
%v6_pos = OpFAdd %v6_pos (OpVectorTimesScalar %v4float %tangent_norm %part_len)

; Vertex 7: part3_pos - perp_norm * thickness + tangent_norm * part_len
%v7_pos = OpFSub %part3_pos (OpVectorTimesScalar %v4float %perp_norm %float_0_02)
%v7_pos = OpFAdd %v7_pos (OpVectorTimesScalar %v4float %tangent_norm %part_len)

; Color for third part (interpolated between color_1 and color_2)
%color_part3 = OpFAdd %color_1 %color_2
%color_part3 = OpFMul %color_part3 %float_0_5

; Emit vertex 4
%gl_out_4 = OpAccessChain %ptr_out_pv %gl_out %int_4
%gl_PerVertex_out_4 = OpLoad %gl_PerVertex_out %gl_out_4
%gl_PerVertex_out_4_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_4 %int_0
OpStore %gl_PerVertex_out_4_0 %v4_pos
%outColor_4 = OpAccessChain %ptr_out_pv %outColor %int_4
OpStore %outColor_4 %color_part3
OpEmitVertex

; Emit vertex 5
%gl_out_5 = OpAccessChain %ptr_out_pv %gl_out %int_5
%gl_PerVertex_out_5 = OpLoad %gl_PerVertex_out %gl_out_5
%gl_PerVertex_out_5_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_5 %int_0
OpStore %gl_PerVertex_out_5_0 %v5_pos
%outColor_5 = OpAccessChain %ptr_out_pv %outColor %int_5
OpStore %outColor_5 %color_part3
OpEmitVertex

; Emit vertex 6
%gl_out_6 = OpAccessChain %ptr_out_pv %gl_out %int_6
%gl_PerVertex_out_6 = OpLoad %gl_PerVertex_out %gl_out_6
%gl_PerVertex_out_6_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_6 %int_0
OpStore %gl_PerVertex_out_6_0 %v6_pos
%outColor_6 = OpAccessChain %ptr_out_pv %outColor %int_6
OpStore %outColor_6 %color_part3
OpEmitVertex

; Emit vertex 7
%gl_out_7 = OpAccessChain %ptr_out_pv %gl_out %int_7
%gl_PerVertex_out_7 = OpLoad %gl_PerVertex_out %gl_out_7
%gl_PerVertex_out_7_0 = OpAccessChain %gl_PerVertex_out %gl_PerVertex_out_7 %int_0
OpStore %gl_PerVertex_out_7_0 %v7_pos
%outColor_7 = OpAccessChain %ptr_out_pv %outColor %int_7
OpStore %outColor_7 %color_part3
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
```
