
```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Extract input data
const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;

// Function to compute Euclidean distance between two points
function distance(p1, p2) {
    const dx = p1[0] - p2[0];
    const dy = p1[1] - p2[1];
    return Math.sqrt(dx * dx + dy * dy);
}

// Compute angle from depot for each package (for sweep heuristic)
for (const pkg of packages) {
    const angle = Math.atan2(pkg.y - depot[1], pkg.x - depot[0]);
    pkg.angle = angle;
    // Also compute distance from depot for tie-breaking
    pkg.distFromDepot = distance(depot, [pkg.x, pkg.y]);
}

// Sort all packages by angle (ascending) and then by id for deterministic tie-breaking
packages.sort((a, b) => {
    if (a.angle !== b.angle) {
        return a.angle - b.angle;
    }
    return a.id - b.id;
});

// Assign packages to trucks in round-robin fashion
const routes = new Array(truckCount).fill(null).map(() => []);
for (let i = 0; i < packages.length; i++) {
    const truckIndex = i % truckCount;
    routes[truckIndex].push(packages[i]);
}

// For each truck, sort its assigned packages by angle (and then by id) and build the route
for (let truckIndex = 0; truckIndex < truckCount; truckIndex++) {
    const truckPackages = routes[truckIndex];
    if (truckPackages.length === 0) {
        // Empty route for trucks with no packages
        continue;
    }
    // Sort by angle and then by id for deterministic order
    truckPackages.sort((a, b) => {
        if (a.angle !== b.angle) {
            return a.angle - b.angle;
        }
        return a.id - b.id;
    });
    // Build the route: the order is the sorted package IDs
    routes[truckIndex] = truckPackages.map(pkg => pkg.id);
}

// Output the result
const output = { routes };
console.log(JSON.stringify(output));
```
