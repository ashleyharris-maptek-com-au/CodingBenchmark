
```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    if (i >= counts.x) return; // Guard against out-of-bounds

    // Read current boid data
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    // Initialize forces
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);

    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;

    // Iterate over all boids to compute forces
    for (uint j = 0; j < counts.x; ++j) {
        if (i == j) continue; // Skip self

        vec3 neighborPos = inData[j * 2].xyz;
        vec3 neighborVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - neighborPos;
        float dist = length(diff);

        if (dist < 1e-6) continue; // Skip zero-distance neighbors

        // Separation: repel if too close
        if (dist < radii.y) {
            sepForce += normalize(diff) / dist;
            sepCount++;
        }

        // Alignment: match velocity if within radius
        if (dist < radii.z) {
            alignForce += neighborVel;
            alignCount++;
        }

        // Cohesion: move towards center of mass if within radius
        if (dist < radii.w) {
            cohForce += neighborPos;
            cohCount++;
        }
    }

    // Average alignment and cohesion forces
    if (alignCount > 0) {
        alignForce = (alignForce / float(alignCount)) - vel;
    }
    if (cohCount > 0) {
        cohForce = (cohForce / float(cohCount)) - pos;
    }

    // Apply weights
    sepForce *= weights.x;
    alignForce *= weights.y;
    cohForce *= weights.z;

    // Total force
    vec3 force = sepForce + alignForce + cohForce;

    // Boundary force: push towards center if outside bounds
    vec3 boundForce = vec3(0.0);
    float boundSize = bounds.x;
    float boundForceMag = bounds.y;
    if (abs(pos.x) > boundSize * 0.4) {
        boundForce.x = -sign(pos.x) * boundForceMag;
    }
    if (abs(pos.y) > boundSize * 0.4) {
        boundForce.y = -sign(pos.y) * boundForceMag;
    }
    if (abs(pos.z) > boundSize * 0.4) {
        boundForce.z = -sign(pos.z) * boundForceMag;
    }
    force += boundForce;

    // Update velocity
    vel += force * radii.x; // dt

    // Clamp speed
    float maxSpeed = weights.w;
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position
    pos += vel * radii.x; // dt

    // Write output
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
