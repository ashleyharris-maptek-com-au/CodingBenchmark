const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Parse input
const width = data.width;
const height = data.height;
const grid = data.grid;
const ignitions = data.ignitions;
const towns = data.towns;
const cutBudget = data.cutBudget;

// Create a 2D array for the grid
const grid2D = [];
for (let y = 0; y < height; y++) {
    grid2D.push(grid[y].split(''));
}

// Function to check if a cell is valid
function isValid(x, y) {
    return x >= 0 && x < width && y >= 0 && y < height;
}

// Function to check if a cell is burnable
function isBurnable(x, y) {
    return isValid(x, y) && grid2D[y][x] === '.';
}

// Function to calculate distances from a set of points
function calculateDistances(points) {
    const distances = new Array(height).fill().map(() => new Array(width).fill(Infinity));
    const queue = [];
    const visited = new Set();
    
    // Initialize with the given points
    for (const [x, y] of points) {
        distances[y][x] = 0;
        queue.push([x, y, 0]);
        visited.add(`${x},${y}`);
    }
    
    // BFS to calculate distances
    while (queue.length > 0) {
        const [x, y, dist] = queue.shift();
        
        // Check 4 directions
        const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        for (const [dx, dy] of directions) {
            const nx = x + dx;
            const ny = y + dy;
            
            if (isValid(nx, ny) && !visited.has(`${nx},${ny}`)) {
                if (grid2D[ny][nx] === '.' || grid2D[ny][nx] === '~') {
                    distances[ny][nx] = dist + 1;
                    visited.add(`${nx},${ny}`);
                    queue.push([nx, ny, dist + 1]);
                }
            }
        }
    }
    
    return distances;
}

// Calculate distances from ignitions
const ignitionDistances = calculateDistances(ignitions);

// Calculate distances from towns
const townPoints = towns.map(town => [town.x, town.y]);
const townDistances = calculateDistances(townPoints);

// Function to simulate fire spread and see which towns would burn
function simulateFireSpread(cutCells) {
    // Create a copy of the grid
    const tempGrid = grid2D.map(row => [...row]);
    
    // Mark cut cells as non-burnable
    for (const [x, y] of cutCells) {
        if (isValid(x, y)) {
            tempGrid[y][x] = '#'; // Use # to represent cut cells
        }
    }
    
    // Initialize fire spread
    const fireQueue = [];
    const visited = new Set();
    
    // Start fires at ignition points
    for (const [x, y] of ignitions) {
        fireQueue.push([x, y]);
        visited.add(`${x},${y}`);
    }
    
    // BFS fire spread
    while (fireQueue.length > 0) {
        const [x, y] = fireQueue.shift();
        
        // Check 4 directions
        const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        for (const [dx, dy] of directions) {
            const nx = x + dx;
            const ny = y + dy;
            
            if (isValid(nx, ny) && !visited.has(`${nx},${ny}`)) {
                if (tempGrid[ny][nx] === '.' || tempGrid[ny][nx] === '#') {
                    visited.add(`${nx},${ny}`);
                    fireQueue.push([nx, ny]);
                }
            }
        }
    }
    
    // Calculate which towns are burned
    let totalValue = 0;
    const burnedTowns = new Set();
    
    for (const town of towns) {
        if (visited.has(`${town.x},${town.y}`)) {
            burnedTowns.add(town.id);
            totalValue += town.value;
        }
    }
    
    return { burnedTowns, totalValue };
}

// Function to find critical cells for protecting towns
function findCriticalCells() {
    const criticalCells = new Map();
    
    // For each town, find cells that would block fire
    for (const town of towns) {
        const queue = [[town.x, town.y, []]];
        const visited = new Set([`${town.x},${town.y}`]);
        const townCriticalCells = new Set();
        
        while (queue.length > 0) {
            const [x, y, path] = queue.shift();
            
            // Check if we reached an ignition point
            for (const [ix, iy] of ignitions) {
                if (x === ix && y === iy) {
                    // We found a path, add the first cell to critical cells
                    if (path.length > 0) {
                        townCriticalCells.add(`${path[0][0]},${path[0][1]}`);
                    }
                    break;
                }
            }
            
            // Check 4 directions
            const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
            for (const [dx, dy] of directions) {
                const nx = x + dx;
                const ny = y + dy;
                
                if (isValid(nx, ny) && !visited.has(`${nx},${ny}`)) {
                    if (grid2D[ny][nx] === '.') {
                        visited.add(`${nx},${ny}`);
                        queue.push([nx, ny, [...path, [nx, ny]]]);
                    }
                }
            }
        }
        
        criticalCells.set(town.id, townCriticalCells);
    }
    
    return criticalCells;
}

// Greedy approach to select firebreaks
function selectFirebreaks() {
    const firebreaks = [];
    const criticalCells = findCriticalCells();
    
    // Sort towns by value (higher value first)
    const sortedTowns = [...towns].sort((a, b) => b.value - a.value);
    
    // Select firebreaks
    for (const town of sortedTowns) {
        if (firebreaks.length >= cutBudget) break;
        
        const townCriticalCells = criticalCells.get(town.id);
        
        // Find the most valuable critical cell that hasn't been cut yet
        let bestCell = null;
        let bestValue = 0;
        
        for (const cellStr of townCriticalCells) {
            const [x, y] = cellStr.split(',').map(Number);
            
            if (isBurnable(x, y) && !firebreaks.some(([fx, fy]) => fx === x && fy === y)) {
                // Calculate the value of this cell (how many towns it protects)
                let cellValue = 0;
                for (const otherTown of towns) {
                    if (criticalCells.get(otherTown.id).has(cellStr)) {
                        cellValue += otherTown.value;
                    }
                }
                
                if (cellValue > bestValue) {
                    bestValue = cellValue;
                    bestCell = [x, y];
                }
            }
        }
        
        if (bestCell) {
            firebreaks.push(bestCell);
        }
    }
    
    return firebreaks;
}

// Select firebreaks
const firebreaks = selectFirebreaks();

// Output result
console.log(JSON.stringify({ firebreaks }));
