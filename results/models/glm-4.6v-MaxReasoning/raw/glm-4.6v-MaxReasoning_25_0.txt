    private static TrajectorySolution FindOptimalTrajectory(List<Body> bodies, Body earth, Body asteroid,

    private static TrajectoryResult CalculateInterceptTrajectory(List<Body> bodies, Body earth, Body asteroid,

        {
            TimeDays = launchTime,
            DeltaV = deltaV
        });
        
        result.ImpactDV = impactVel;
