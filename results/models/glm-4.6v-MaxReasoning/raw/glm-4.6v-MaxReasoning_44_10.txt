
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse component
    float diffuse = max(dot(N, lightDir), 0.0);
    
    // Calculate stripe pattern and choose color
    float stripe = sin(uv.y * 3.14159 * 16.0);
    vec3 stripeColor = mix(vec3(0.9, 0.1, 0.1), vec3(0.1, 0.1, 0.9), stripe <= 0.0);
    
    // Apply Lambertian shading
    float shading = 0.4 + 0.6 * diffuse;
    vec3 finalColor = stripeColor * shading;
    
    // Output color with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
