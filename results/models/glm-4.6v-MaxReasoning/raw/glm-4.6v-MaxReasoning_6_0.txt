using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

public class OrbitalTSPSolver
{
    // Constants
    private const double Mu = 398600.4418; // Earth gravitational parameter in km³/s²
    private const double EarthRadius = 6371.0; // Earth radius in km
    
    // Input data
    private int N;
    private double mu;
    private Vector3 startOrbit;
    private List<Vector3> stations;
    
    // Distance matrix (delta-V between stations)
    private double[,] distanceMatrix;
    
    public static void Main(string[] args)
    {
        var solver = new OrbitalTSPSolver();
        solver.ReadInput();
        solver.CalculateDistances();
        var solution = solver.SolveTSP();
        solver.PrintSolution(solution);
    }
    
    private void ReadInput()
    {
        // Read N and mu
        var firstLine = Console.ReadLine().Split(' ');
        N = int.Parse(firstLine[0]);
        mu = double.Parse(firstLine[1]);
        
        // Read start orbit
        var startLine = Console.ReadLine().Split(' ');
        startOrbit = new Vector3(
            double.Parse(startLine[0]),
            double.Parse(startLine[1]),
            double.Parse(startLine[2])
        );
        
        // Read station orbits
        stations = new List<Vector3>();
        for (int i = 0; i < N; i++)
        {
            var stationLine = Console.ReadLine().Split(' ');
            stations.Add(new Vector3(
                double.Parse(stationLine[0]),
                double.Parse(stationLine[1]),
                double.Parse(stationLine[2])
            ));
        }
    }
    
    private void CalculateDistances()
    {
        distanceMatrix = new double[N + 1, N + 1];
        
        // Calculate distances from start to all stations
        for (int i = 0; i < N; i++)
        {
            distanceMatrix[0, i + 1] = CalculateDeltaV(startOrbit, stations[i]);
            distanceMatrix[i + 1, 0] = distanceMatrix[0, i + 1];
        }
        
        // Calculate distances between stations
        for (int i = 0; i < N; i++)
        {
            for (int j = i + 1; j < N; j++)
            {
                double deltaV = CalculateDeltaV(stations[i], stations[j]);
                distanceMatrix[i + 1, j + 1] = deltaV;
                distanceMatrix[j + 1, i + 1] = deltaV;
            }
        }
    }
    
    private double CalculateDeltaV(Vector3 from, Vector3 to)
    {
        // Simplified orbital mechanics calculation
        // In a real implementation, this would solve Lambert's problem
        double distance = (from - to).Magnitude;
        
        // Simple approximation: delta-V proportional to distance
        // This is a placeholder - real implementation would be much more complex
        return distance / 1000.0; // Simplified delta-V in km/s
    }
    
    private List<int> SolveTSP()
    {
        // Choose algorithm based on number of stations
        if (N <= 5)
        {
            return SolveBruteForce();
        }
        else if (N <= 10)
        {
            return SolveNearestNeighborWith2Opt();
        }
        else if (N <= 20)
        {
            return SolveGeneticAlgorithm();
        }
        else
        {
            return SolveChristofides();
        }
    }
    
    private List<int> SolveBruteForce()
    {
        // For small N, we can try all permutations
        var bestTour = new List<int>();
        double bestDistance = double.MaxValue;
        
        var stationsToVisit = Enumerable.Range(1, N).ToList();
        
        foreach (var permutation in Permutations(stationsToVisit))
        {
            double distance = CalculateTourDistance(permutation);
            if (distance < bestDistance)
            {
                bestDistance = distance;
                bestTour = permutation;
            }
        }
        
        return bestTour;
    }
    
    private List<int> SolveNearestNeighborWith2Opt()
    {
        // Start with nearest neighbor
        var tour = new List<int>();
        var unvisited = new HashSet<int>(Enumerable.Range(1, N));
        int current = 0;
        
        while (unvisited.Count > 0)
        {
            int nearest = -1;
            double minDistance = double.MaxValue;
            
            foreach (int station in unvisited)
            {
                double distance = distanceMatrix[current, station];
                if (distance < minDistance)
                {
                    minDistance = distance;
                    nearest = station;
                }
            }
            
            tour.Add(nearest);
            unvisited.Remove(nearest);
            current = nearest;
        }
        
        // Apply 2-opt optimization
        bool improved;
        do
        {
            improved = false;
            for (int i = 0; i < tour.Count - 1; i++)
            {
                for (int j = i + 1; j < tour.Count; j++)
                {
                    if (2OptImproves(tour, i, j))
                    {
                        TwoOptSwap(tour, i, j);
                        improved = true;
                    }
                }
            }
        } while (improved);
        
        return tour;
    }
    
    private List<int> SolveGeneticAlgorithm()
    {
        // Genetic algorithm for medium to large problems
        int populationSize = 50;
        int generations = 100;
        double mutationRate = 0.01;
        
        var population = InitializePopulation(populationSize);
        
        for (int gen = 0; gen < generations; gen++)
        {
            var newPopulation = new List<List<int>>();
            
            // Elitism: keep best individuals
            var sorted = population.OrderBy(CalculateTourDistance).ToList();
            newPopulation.AddRange(sorted.Take(5));
            
            // Crossover and mutation
            while (newPopulation.Count < populationSize)
            {
                var parent1 = TournamentSelection(population);
                var parent2 = TournamentSelection(population);
                var child = Crossover(parent1, parent2);
                Mutate(child, mutationRate);
                newPopulation.Add(child);
            }
            
            population = newPopulation;
        }
        
        return population.OrderBy(CalculateTourDistance).First();
    }
    
    private List<int> SolveChristofides()
    {
        // Christofides algorithm for large problems
        // This is a 1.5-approximation algorithm for TSP
        var mst = BuildMST();
        var oddDegreeNodes = FindOddDegreeNodes(mst);
        var matching = FindMinimumWeightPerfectMatching(oddDegreeNodes);
        var eulerianMultigraph = CombineMSTAndMatching(mst, matching);
        var tour = FindEulerianTour(eulerianMultigraph);
        var hamiltonianTour = ConvertToHamiltonianTour(tour);
        
        return hamiltonianTour;
    }
    
    private double CalculateTourDistance(List<int> tour)
    {
        double distance = 0;
        int prev = 0; // Start from the start orbit (index 0)
        
        foreach (int station in tour)
        {
            distance += distanceMatrix[prev, station];
            prev = station;
        }
        
        // Return to start
        distance += distanceMatrix[prev, 0];
        
        return distance;
    }
    
    private bool TwoOptImproves(List<int> tour, int i, int j)
    {
        // Check if swapping edges (i, i+1) and (j, j+1) improves the tour
        int i1 = tour[i];
        int i2 = tour[(i + 1) % tour.Count];
        int j1 = tour[j];
        int j2 = tour[(j + 1) % tour.Count];
        
        double currentDistance = distanceMatrix[i1, i2] + distanceMatrix[j1, j2];
        double newDistance = distanceMatrix[i1, j1] + distanceMatrix[i2, j2];
        
        return newDistance < currentDistance;
    }
    
    private void TwoOptSwap(List<int> tour, int i, int j)
    {
        // Reverse the segment between i and j
        var segment = tour.GetRange(i, j - i + 1);
        segment.Reverse();
        for (int k = 0; k < segment.Count; k++)
        {
            tour[i + k] = segment[k];
        }
    }
    
    private List<List<int>> InitializePopulation(int size)
    {
        var population = new List<List<int>>();
        var stationsToVisit = Enumerable.Range(1, N).ToList();
        
        for (int i = 0; i < size; i++)
        {
            var permutation = new List<int>(stationsToVisit);
            permutation.Shuffle();
            population.Add(permutation);
        }
        
        return population;
    }
    
    private List<int> TournamentSelection(List<List<int>> population)
    {
        int tournamentSize = 5;
        var tournament = new List<int>();
        
        for (int i = 0; i < tournamentSize; i++)
        {
            tournament.Add(new Random().Next(population.Count));
        }
        
        return population[tournament.OrderBy(x => CalculateTourDistance(population[x])).First()];
    }
    
    private List<int> Crossover(List<int> parent1, List<int> parent2)
    {
        // Order crossover (OX)
        int size = parent1.Count;
        int start = new Random().Next(size);
        int end = new Random().Next(start, size);
        
        var child = new List<int>(new int[size]);
        var used = new HashSet<int>();
        
        // Copy segment from parent1
        for (int i = start; i <= end; i++)
        {
            child[i] = parent1[i];
            used.Add(parent1[i]);
        }
        
        // Fill remaining positions with elements from parent2
        int current = (end + 1) % size;
        int parent2Index = (end + 1) % size;
        
        while (current != start)
        {
            if (!used.Contains(parent2[parent2Index]))
            {
                child[current] = parent2[parent2Index];
                used.Add(parent2[parent2Index]);
                current = (current + 1) % size;
            }
            parent2Index = (parent2Index + 1) % size;
        }
        
        return child;
    }
    
    private void Mutate(List<int> tour, double mutationRate)
    {
        for (int i = 0; i < tour.Count; i++)
        {
            if (new Random().NextDouble() < mutationRate)
            {
                int j = new Random().Next(tour.Count);
                var temp = tour[i];
                tour[i] = tour[j];
                tour[j] = temp;
            }
        }
    }
    
    private List<(int, int)> BuildMST()
    {
        // Prim's algorithm for MST
        var mst = new List<(int, int)>();
        var visited = new HashSet<int>();
        var minEdge = new Dictionary<int, (int, double)>();
        
        visited.Add(0); // Start from the start orbit
        
        // Initialize min edges
        for (int i = 1; i <= N; i++)
        {
            minEdge[i] = (0, distanceMatrix[0, i]);
        }
        
        while (visited.Count < N + 1)
        {
            // Find minimum edge to an unvisited node
            int nextNode = -1;
            double minDistance = double.MaxValue;
            
            foreach (var node in minEdge)
            {
                if (!visited.Contains(node.Key) && node.Value.Item2 < minDistance)
                {
                    minDistance = node.Value.Item2;
                    nextNode = node.Key;
                }
            }
            
            if (nextNode == -1) break; // Should not happen in a connected graph
            
            mst.Add((minEdge[nextNode].Item1, nextNode));
            visited.Add(nextNode);
            
            // Update min edges
            for (int i = 1; i <= N; i++)
            {
                if (!visited.Contains(i) && distanceMatrix[nextNode, i] < minEdge[i].Item2)
                {
                    minEdge[i] = (nextNode, distanceMatrix[nextNode, i]);
                }
            }
        }
        
        return mst;
    }
    
    private List<int> FindOddDegreeNodes(List<(int, int)> mst)
    {
        var degree = new Dictionary<int, int>();
        
        foreach (var edge in mst)
        {
            degree[edge.Item1] = degree.GetValueOrDefault(edge.Item1, 0) + 1;
            degree[edge.Item2] = degree.GetValueOrDefault(edge.Item2, 0) + 1;
        }
        
        return degree.Where(kvp => kvp.Value % 2 == 1).Select(kvp => kvp.Key).ToList();
    }
    
    private List<(int, int)> FindMinimumWeightPerfectMatching(List<int> oddNodes)
    {
        // Brute force for small odd node sets, otherwise use greedy
        if (oddNodes.Count <= 10)
        {
            return FindMinimumWeightPerfectMatchingBruteForce(oddNodes);
        }
        else
        {
            return FindMinimumWeightPerfectMatchingGreedy(oddNodes);
        }
    }
    
    private List<(int, int)> FindMinimumWeightPerfectMatchingBruteForce(List<int> oddNodes)
    {
        var bestMatching = new List<(int, int)>();
        double bestWeight = double.MaxValue;
        
        foreach (var matching in Permutations(oddNodes, oddNodes.Count / 2))
        {
            var currentMatching = new List<(int, int)>();
            double currentWeight = 0;
            
            for (int i = 0; i < matching.Count; i += 2)
            {
                currentMatching.Add((matching[i], matching[i + 1]));
                currentWeight += distanceMatrix[matching[i], matching[i + 1]];
            }
            
            if (currentWeight < bestWeight)
            {
                bestWeight = currentWeight;
                bestMatching = currentMatching;
            }
        }
        
        return bestMatching;
    }
    
    private List<(int, int)> FindMinimumWeightPerfectMatchingGreedy(List<int> oddNodes)
    {
        var matching = new List<(int, int)>();
        var remaining = new List<int>(oddNodes);
        
        while (remaining.Count > 0)
        {
            int node1 = remaining[0];
            double minDistance = double.MaxValue;
            int node2 = -1;
            
            for (int i = 1; i < remaining.Count; i++)
            {
                double distance = distanceMatrix[node1, remaining[i]];
                if (distance < minDistance)
                {
                    minDistance = distance;
                    node2 = remaining[i];
                }
            }
            
            matching.Add((node1, node2));
            remaining.Remove(node1);
            remaining.Remove(node2);
        }
        
        return matching;
    }
    
    private List<(int, int)> CombineMSTAndMatching(List<(int, int)> mst, List<(int, int)> matching)
    {
        var multigraph = new List<(int, int)>();
        multigraph.AddRange(mst);
        multigraph.AddRange(matching);
        return multigraph;
    }
    
    private List<int> FindEulerianTour(List<(int, int)> multigraph)
    {
        // Hierholzer's algorithm
        var graph = new Dictionary<int, List<int>>();
        
        foreach (var edge in multigraph)
        {
            if (!graph.ContainsKey(edge.Item1)) graph[edge.Item1] = new List<int>();
            if (!graph.ContainsKey(edge.Item2)) graph[edge.Item2] = new List<int>();
            graph[edge.Item1].Add(edge.Item2);
            graph[edge.Item2].Add(edge.Item1);
        }
        
        var tour = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0); // Start from the start orbit
        
        while (stack.Count > 0)
        {
            int current = stack.Peek();
            if (graph[current].Count > 0)
            {
                int next = graph[current][0];
                graph[current].RemoveAt(0);
                graph[next].RemoveAt(graph[next].IndexOf(current));
                stack.Push(next);
            }
            else
            {
                tour.Add(stack.Pop());
            }
        }
        
        tour.Reverse();
        return tour;
    }
    
    private List<int> ConvertToHamiltonianTour(List<int> eulerianTour)
    {
        var tour = new List<int>();
        var visited = new HashSet<int>();
        
        foreach (int node in eulerianTour)
        {
            if (!visited.Contains(node) && node != 0)
            {
                tour.Add(node);
                visited.Add(node);
            }
        }
        
        return tour;
    }
    
    // Helper method to generate permutations
    private IEnumerable<List<int>> Permutations(List<int> list)
    {
        if (list.Count == 1)
        {
            yield return list;
        }
        else
        {
            for (int i = 0; i < list.Count; i++)
            {
                var first = list[i];
                var rest = new List<int>(list.Where((_, index) => index != i));
                foreach (var perm in Permutations(rest))
                {
                    var result = new List<int> { first };
                    result.AddRange(perm);
                    yield return result;
                }
            }
        }
    }
    
    private IEnumerable<List<int>> Permutations(List<int> list, int k)
    {
        if (k == 1)
        {
            foreach (var item in list)
            {
                yield return new List<int> { item };
            }
        }
        else
        {
            for (int i = 0; i < list.Count; i++)
            {
                var first = list[i];
                var rest = new List<int>(list.Where((_, index) => index != i));
                foreach (var perm in Permutations(rest, k - 1))
                {
                    var result = new List<int> { first };
                    result.AddRange(perm);
                    yield return result;
                }
            }
        }
    }
    
    private void PrintSolution(List<int> solution)
    {
        Console.WriteLine(string.Join(" ", solution));
    }
}

// Vector3 class for 3D vector operations
public class Vector3
{
    public double X { get; set; }
    public double Y { get; set; }
    public double Z { get; set; }
    
    public Vector3(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }
    
    public static Vector3 operator -(Vector3 a, Vector3 b)
    {
        return new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }
    
    public double Magnitude
    {
        get
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }
    }
}
