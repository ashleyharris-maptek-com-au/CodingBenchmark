using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class MiningProblem {
    // Define the voxel structure
    public struct Voxel {
        public byte mass;
        public byte valuePerTonne;
        public bool isDug;
        public bool isExposed;
    }

    // Grid dimensions
    public int XSize, YSize, ZSize;
    public double XBlockSize, YBlockSize, ZBlockSize;
    
    // The voxel grid
    public Voxel[,,] grid;
    
    // Processing plant location
    public const int PROCESSING_X = 0;
    public const int PROCESSING_Y = 0;
    
    public void Solve() {
        // Read input
        ReadInput();
        
        // Identify exposed voxels
        IdentifyExposedVoxels();
        
        // Plan mining operations
        List<Operation> operations = PlanOperations();
        
        // Output results
        WriteOutput(operations);
    }
    
    private void ReadInput() {
        // Read the first line
        string[] header = Console.ReadLine().Split();
        XSize = int.Parse(header[0]);
        YSize = int.Parse(header[1]);
        ZSize = int.Parse(header[2]);
        XBlockSize = double.Parse(header[3]);
        YBlockSize = double.Parse(header[4]);
        ZBlockSize = double.Parse(header[5]);
        
        // Initialize the grid
        grid = new Voxel[XSize, YSize, ZSize];
        
        // Read voxel data
        for (int x = 0; x < XSize; x++) {
            for (int y = 0; y < YSize; y++) {
                for (int z = 0; z < ZSize; z++) {
                    int d = Console.Read(); // Read a byte
                    grid[x, y, z].mass = (byte)(d % 16);
                    grid[x, y, z].valuePerTonne = (byte)(d / 16);
                    grid[x, y, z].isDug = false;
                }
            }
        }
    }
    
    private void IdentifyExposedVoxels() {
        // A voxel is exposed if it has empty space above it or on at least one side
        for (int x = 0; x < XSize; x++) {
            for (int y = 0; y < YSize; y++) {
                for (int z = 0; z < ZSize; z++) {
                    // Check if voxel is empty (mass = 0)
                    if (grid[x, y, z].mass == 0) {
                        grid[x, y, z].isExposed = true;
                        continue;
                    }
                    
                    // Check if voxel is on the top layer
                    if (z == ZSize - 1) {
                        grid[x, y, z].isExposed = true;
                        continue;
                    }
                    
                    // Check if there's empty space above
                    if (grid[x, y, z + 1].mass == 0) {
                        grid[x, y, z].isExposed = true;
                        continue;
                    }
                    
                    // Check if any side is exposed
                    bool sideExposed = false;
                    if (x == 0 || grid[x - 1, y, z].mass == 0) sideExposed = true;
                    if (x == XSize - 1 || grid[x + 1, y, z].mass == 0) sideExposed = true;
                    if (y == 0 || grid[x, y - 1, z].mass == 0) sideExposed = true;
                    if (y == YSize - 1 || grid[x, y + 1, z].mass == 0) sideExposed = true;
                    
                    grid[x, y, z].isExposed = sideExposed;
                }
            }
        }
    }
    
    private List<Operation> PlanOperations() {
        List<Operation> operations = new List<Operation>();
        
        // For large worlds, we need a more efficient approach
        if (XSize * YSize * ZSize > 1000000) {
            operations = PlanOperationsLarge();
        } else {
            operations = PlanOperationsSmall();
        }
        
        return operations;
    }
    
    private List<Operation> PlanOperationsSmall() {
        List<Operation> operations = new List<Operation>();
        
        // Simple strategy: dig all exposed valuable ore first
        // Then dig other exposed voxels
        
        // First, find all exposed valuable ore
        List<(int x, int y, int z)> valuableOre = new List<(int, int, int)>();
        for (int x = 0; x < XSize; x++) {
            for (int y = 0; y < YSize; y++) {
                for (int z = 0; z < ZSize; z++) {
                    if (grid[x, y, z].isExposed && grid[x, y, z].valuePerTonne > 0) {
                        valuableOre.Add((x, y, z));
                    }
                }
            }
        }
        
        // Sort by value (descending)
        valuableOre.Sort((a, b) => grid[b.x, b.y, b.z].valuePerTonne.CompareTo(grid[a.x, a.y, a.z].valuePerTonne));
        
        // Plan operations for valuable ore
        foreach (var (x, y, z) in valuableOre) {
            if (grid[x, y, z].isDug) continue;
            
            // Find a path to processing plant
            var path = FindPath(x, y, z, PROCESSING_X, PROCESSING_Y);
            if (path != null) {
                operations.Add(new Operation {
                    digX = x,
                    digY = y,
                    digZ = z,
                    steps = path,
                    destX = PROCESSING_X,
                    destY = PROCESSING_Y
                });
                
                // Mark voxel as dug
                grid[x, y, z].isDug = true;
            }
        }
        
        // Then, find other exposed voxels
        List<(int x, int y, int z)> otherVoxels = new List<(int, int, int)>();
        for (int x = 0; x < XSize; x++) {
            for (int y = 0; y < YSize; y++) {
                for (int z = 0; z < ZSize; z++) {
                    if (grid[x, y, z].isExposed && grid[x, y, z].valuePerTonne == 0 && !grid[x, y, z].isDug) {
                        otherVoxels.Add((x, y, z));
                    }
                }
            }
        }
        
        // Sort by mass (descending)
        otherVoxels.Sort((a, b) => grid[b.x, b.y, b.z].mass.CompareTo(grid[a.x, a.y, a.z].mass));
        
        // Plan operations for other voxels
        foreach (var (x, y, z) in otherVoxels) {
            if (grid[x, y, z].isDug) continue;
            
            // Find a path to a dump location
            var path = FindPath(x, y, z, x, y); // Dump at same location (simplified)
            if (path != null) {
                operations.Add(new Operation {
                    digX = x,
                    digY = y,
                    digZ = z,
                    steps = path,
                    destX = x,
                    destY = y
                });
                
                // Mark voxel as dug
                grid[x, y, z].isDug = true;
            }
        }
        
        return operations;
    }
    
    private List<Operation> PlanOperationsLarge() {
        // For large worlds, we need a more efficient approach
        // We'll use a greedy algorithm that prioritizes high-value ore near the processing plant
        
        List<Operation> operations = new List<Operation>();
        
        // Use a priority queue to prioritize voxels by value and distance to processing plant
        var priorityQueue = new SimplePriorityQueue<(int x, int y, int z), double>();
        
        // Add all exposed valuable ore to the queue
        for (int x = 0; x < XSize; x++) {
            for (int y = 0; y < YSize; y++) {
                for (int z = 0; z < ZSize; z++) {
                    if (grid[x, y, z].isExposed && grid[x, y, z].valuePerTonne > 0) {
                        double priority = -grid[x, y, z].valuePerTonne / (Math.Abs(x - PROCESSING_X) + Math.Abs(y - PROCESSING_Y) + 1);
                        priorityQueue.Enqueue((x, y, z), priority);
                    }
                }
            }
        }
        
        // Process the queue
        while (priorityQueue.Count > 0) {
            var (x, y, z) = priorityQueue.Dequeue();
            
            if (grid[x, y, z].isDug) continue;
            
            // Find a path to processing plant
            var path = FindPath(x, y, z, PROCESSING_X, PROCESSING_Y);
            if (path != null) {
                operations.Add(new Operation {
                    digX = x,
                    digY = y,
                    digZ = z,
                    steps = path,
                    destX = PROCESSING_X,
                    destY = PROCESSING_Y
                });
                
                // Mark voxel as dug
                grid[x, y, z].isDug = true;
            }
        }
        
        return operations;
    }
    
    private List<(int x, int y)> FindPath(int startX, int startY, int startZ, int endX, int endY) {
        // Use A* algorithm to find a path that respects the slope constraint
        // The path is in X and Y only (4-connected)
        
        // Priority queue for A*
        var openSet = new SimplePriorityQueue<(int x, int y), double>();
        openSet.Enqueue((startX, startY), 0);
        
        // Dictionary to store cameFrom
        var cameFrom = new Dictionary<(int x, int y), (int x, int y)>();
        
        // Dictionary to store gScore
        var gScore = new Dictionary<(int x, int y), double>();
        gScore[(startX, startY)] = 0;
        
        // Dictionary to store fScore
        var fScore = new Dictionary<(int x, int y), double>();
        fScore[(startX, startY)] = Heuristic(startX, startY, endX, endY);
        
        while (openSet.Count > 0) {
            var current = openSet.Dequeue();
            
            if (current.x == endX && current.y == endY) {
                // Reconstruct path
                var path = new List<(int x, int y)>();
                var node = current;
                while (cameFrom.ContainsKey(node)) {
                    path.Add(node);
                    node = cameFrom[node];
                }
                path.Reverse();
                return path;
            }
            
            // Check neighbors
            var neighbors = GetNeighbors(current.x, current.y, startZ);
            foreach (var neighbor in neighbors) {
                // Check slope constraint
                if (Math.Abs(grid[neighbor.x, neighbor.y, startZ].mass - grid[current.x, current.y, startZ].mass) > 1) {
                    continue;
                }
                
                double tentativeGScore = gScore[current] + 1; // Cost of moving one step
                
                if (!gScore.ContainsKey(neighbor) || tentativeGScore < gScore[neighbor]) {
                    cameFrom[neighbor] = current;
                    gScore[neighbor] = tentativeGScore;
                    fScore[neighbor] = tentativeGScore + Heuristic(neighbor.x, neighbor.y, endX, endY);
                    openSet.Enqueue(neighbor, fScore[neighbor]);
                }
            }
        }
        
        return null; // No path found
    }
    
    private List<(int x, int y)> GetNeighbors(int x, int y, int z) {
        var neighbors = new List<(int x, int y)>();
        
        // 4-connected neighbors
        if (x > 0) neighbors.Add((x - 1, y));
        if (x < XSize - 1) neighbors.Add((x + 1, y));
        if (y > 0) neighbors.Add((x, y - 1));
        if (y < YSize - 1) neighbors.Add((x, y + 1));
        
        return neighbors;
    }
    
    private double Heuristic(int x1, int y1, int x2, int y2) {
        // Manhattan distance
        return Math.Abs(x1 - x2) + Math.Abs(y1 - y2);
    }
    
    private void WriteOutput(List<Operation> operations) {
        Console.WriteLine(operations.Count);
        
        foreach (var op in operations) {
            Console.Write(op.steps.Count + " " + op.digX + " " + op.digY + " " + op.digZ);
            
            foreach (var step in op.steps) {
                Console.Write(" " + step.x + " " + step.y);
            }
            
            Console.WriteLine(" " + op.destX + " " + op.destY);
        }
    }
    
    public struct Operation {
        public int digX, digY, digZ;
        public List<(int x, int y)> steps;
        public int destX, destY;
    }
    
    // Simple priority queue implementation
    public class SimplePriorityQueue<TElement, TPriority> where TPriority : IComparable<TPriority> {
        private List<(TElement element, TPriority priority)> elements = new List<(TElement, TPriority)>();
        
        public void Enqueue(TElement element, TPriority priority) {
            elements.Add((element, priority));
            int i = elements.Count - 1;
            
            while (i > 0) {
                int parent = (i - 1) / 2;
                if (elements[parent].priority.CompareTo(elements[i].priority) <= 0) break;
                
                var temp = elements[i];
                elements[i] = elements[parent];
                elements[parent] = temp;
                i = parent;
            }
        }
        
        public TElement Dequeue() {
            var result = elements[0].element;
            elements[0] = elements[elements.Count - 1];
            elements.RemoveAt(elements.Count - 1);
            
            int i = 0;
            while (true) {
                int left = 2 * i + 1;
                int right = 2 * i + 2;
                int smallest = i;
                
                if (left < elements.Count && elements[left].priority.CompareTo(elements[smallest].priority) < 0) {
                    smallest = left;
                }
                
                if (right < elements.Count && elements[right].priority.CompareTo(elements[smallest].priority) < 0) {
                    smallest = right;
                }
                
                if (smallest == i) break;
                
                var temp = elements[i];
                elements[i] = elements[smallest];
                elements[smallest] = temp;
                i = smallest;
            }
            
            return result;
        }
        
        public int Count => elements.Count;
    }
    
    public static void Main(string[] args) {
        var solver = new MiningProblem();
        solver.Solve();
    }
}
