const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Parse input
const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

// Priority Queue implementation
class PriorityQueue {
    constructor() {
        this.items = [];
    }
    
    enqueue(element, priority) {
        const queueElement = {element, priority};
        let added = false;
        
        for (let i = 0; i < this.items.length; i++) {
            if (priority < this.items[i].priority) {
                this.items.splice(i, 0, queueElement);
                added = true;
                break;
            }
        }
        
        if (!added) {
            this.items.push(queueElement);
        }
    }
    
    dequeue() {
        return this.items.shift().element;
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
}

// BFS for shortest path
function bfsDistance(start, end, grid) {
    const rows = grid.length;
    const cols = grid[0].length;
    
    // Check if start or end is blocked
    if (grid[start.y][start.x] === '#' || grid[end.y][end.x] === '#') {
        return Infinity;
    }
    
    // If start and end are the same
    if (start.x === end.x && start.y === end.y) {
        return 0;
    }
    
    const queue = [{x: start.x, y: start.y, dist: 0}];
    const visited = new Set([`${start.x},${start.y}`]);
    
    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]];
    
    while (queue.length > 0) {
        const current = queue.shift();
        
        for (const [dx, dy] of directions) {
            const nx = current.x + dx;
            const ny = current.y + dy;
            
            if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && 
                grid[ny][nx] === '.' && !visited.has(`${nx},${ny}`)) {
                
                if (nx === end.x && ny === end.y) {
                    return current.dist + 1;
                }
                
                visited.add(`${nx},${ny}`);
                queue.push({x: nx, y: ny, dist: current.dist + 1});
            }
        }
    }
    
    return Infinity; // No path found
}

// Main function to assign jobs
function assignJobs() {
    const assignments = Array(robots.length).fill().map(() => []);
    
    // Initialize robot positions
    const robotPositions = robots.map(r => ({x: r.x, y: r.y}));
    
    // Sort jobs by priority (higher priority first)
    const sortedJobs = [...jobs].sort((a, b) => b.priority - a.priority);
    
    // Use a priority queue to always pick the best next assignment
    const pq = new PriorityQueue();
    
    // Initialize with all possible first assignments
    for (let i = 0; i < robots.length; i++) {
        for (let j = 0; j < sortedJobs.length; j++) {
            const job = sortedJobs[j];
            const distance = bfsDistance(robotPositions[i], job, grid);
            pq.enqueue({robot: i, job, distance, totalDistance: distance}, distance);
        }
    }
    
    // Track which jobs have been assigned
    const assignedJobs = new Set();
    
    // Process the priority queue
    while (!pq.isEmpty() && assignedJobs.size < sortedJobs.length) {
        const {robot, job, distance, totalDistance} = pq.dequeue();
        
        if (!assignedJobs.has(job.id)) {
            // Assign this job to the robot
            assignments[robot].push(job.id);
            assignedJobs.add(job.id);
            
            // Update robot position
            robotPositions[robot] = {x: job.x, y: job.y};
            
            // Add new possible assignments for this robot
            for (let j = 0; j < sortedJobs.length; j++) {
                const nextJob = sortedJobs[j];
                if (!assignedJobs.has(nextJob.id)) {
                    const nextDistance = bfsDistance(robotPositions[robot], nextJob, grid);
                    const nextTotalDistance = totalDistance + nextDistance;
                    pq.enqueue({
                        robot, 
                        job: nextJob, 
                        distance: nextDistance, 
                        totalDistance: nextTotalDistance
                    }, nextTotalDistance);
                }
            }
        }
    }
    
    return {assignments};
}

// Run the assignment and output the result
console.log(JSON.stringify(assignJobs()));
