
```hlsl
[[vk::location(0)]] float3 worldPos : TEXCOORD0;
[[vk::location(1)]] float3 normal   : TEXCOORD1;
[[vk::location(2)]] float2 uv       : TEXCOORD2;
[[vk::location(3)]] float3 tangent  : TEXCOORD3;
[[vk::location(4)]] float3 color    : TEXCOORD4;

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate view direction (from surface to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    
    // Calculate diffuse component
    float diffuse = max(dot(N, L), 0.0);
    
    // Procedural brick pattern
    float2 scaledUV = input.uv * float2(8.0, 16.0);
    
    // Determine if we're on an odd row for staggering
    bool isOddRow = floor(scaledUV.y) % 2.0 > 0.5;
    
    // Apply staggered offset for odd rows
    float2 brickUV = scaledUV;
    if (isOddRow) {
        brickUV.x += 0.5;
    }
    
    // Get fractional parts to determine position within brick
    float2 fracUV = frac(brickUV);
    
    // Calculate mortar thickness in UV space
    float mortarX = 0.05 / 8.0;  // 0.05 in X, scaled by 8
    float mortarY = 0.1 / 16.0;  // 0.1 in Y, scaled by 16
    
    // Determine if we're in mortar or brick
    bool isInMortar = (fracUV.x < mortarX) || (fracUV.y < mortarY) || 
                     (fracUV.x > 1.0 - mortarX) || (fracUV.y > 1.0 - mortarY);
    
    // Set base color based on brick or mortar
    float3 baseColor = isInMortar ? float3(0.7, 0.7, 0.7) : float3(0.7, 0.2, 0.1);
    
    // Apply Lambertian shading factor
    float shading = 0.4 + 0.6 * diffuse;
    
    // Apply color and shading
    float3 finalColor = baseColor * shading;
    
    return float4(finalColor, 1.0);
}
```
