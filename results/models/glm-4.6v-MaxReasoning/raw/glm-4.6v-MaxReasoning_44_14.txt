
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);
    
    // Calculate light direction (from fragment to light)
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate distance from camera to fragment
    float dist = length(worldPos - ubo.cameraPos.xyz);
    
    // Calculate fog factor
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);
    
    // Calculate diffuse component using Lambertian lighting
    float diffuse = max(dot(N, lightDir), 0.0);
    
    // Calculate base color with Lambertian factor
    vec3 baseColor = vec3(0.8, 0.2, 0.2) * (0.3 + 0.7 * diffuse);
    
    // Fog color
    vec3 fogColor = vec3(0.7, 0.7, 0.8);
    
    // Mix base color with fog color
    vec3 finalColor = mix(baseColor, fogColor, fogFactor);
    
    // Output color with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
