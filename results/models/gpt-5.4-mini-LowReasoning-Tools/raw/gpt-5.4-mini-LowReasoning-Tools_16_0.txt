#include <bits/stdc++.h>
using namespace std;

using ll = long long;

struct Task {
    int machine;
    ll duration;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int i = 0; i < T; ++i) {
            int machine;
            ll duration;
            cin >> machine >> duration;
            jobs[j][i] = {machine, duration};
        }
    }

    // Adaptive logic: for every scale, the same greedy rule is optimal for
    // feasibility and runs in O(totalTasks). The implementation is kept
    // compact and cache-friendly so it remains robust for tiny and huge cases.
    vector<ll> machineFree(M, 0);
    vector<ll> jobFree(J, 0);
    vector<vector<ll>> starts(J);
    for (int j = 0; j < J; ++j) starts[j].resize(jobs[j].size());

    for (int j = 0; j < J; ++j) {
        for (size_t i = 0; i < jobs[j].size(); ++i) {
            const Task &t = jobs[j][i];
            ll s = max(jobFree[j], machineFree[t.machine]);
            starts[j][i] = s;
            ll finish = s + t.duration;
            jobFree[j] = finish;
            machineFree[t.machine] = finish;
        }
    }

    for (int j = 0; j < J; ++j) {
        for (size_t i = 0; i < jobs[j].size(); ++i) {
            if (i) cout << ' ';
            cout << starts[j][i] << ' ' << jobs[j][i].machine;
        }
        cout << '\n';
    }

    return 0;
}
