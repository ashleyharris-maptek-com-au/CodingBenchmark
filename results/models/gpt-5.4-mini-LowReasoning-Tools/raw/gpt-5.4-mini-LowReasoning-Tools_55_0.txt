using U64 = uint64_t;

enum Piece { WP, WN, WB, WR, WQ, WK, BP, BN, BB, BR, BQ, BK, PIECE_NONE };
enum Color { WHITE, BLACK };

constexpr int INF = 1e9;
constexpr int MATE = 100000;

inline int poplsb(U64 &b) {
    int s = std::countr_zero(b);
    b &= b - 1;
    return s;
}

inline U64 bit(int s) { return 1ULL << s; }
inline int file_of(int s) { return s & 7; }
inline int rank_of(int s) { return s >> 3; }
inline bool on_board(int s) { return s >= 0 && s < 64; }
inline string sq(int s) {
    string r;
    r.push_back(char('a' + file_of(s)));
    r.push_back(char('1' + rank_of(s)));
    return r;
}

U64 knight_attacks[64], king_attacks[64];
U64 pawn_attacks[2][64];

U64 slide_attacks(int s, U64 occ, const int *dirs, int ndirs) {
    U64 a = 0;
    int f0 = file_of(s), r0 = rank_of(s);
    for (int i = 0; i < ndirs; ++i) {
        int d = dirs[i], x = s;
        while (true) {
            int nx = x + d;
            if (!on_board(nx)) break;
            int fx = file_of(nx), rx = rank_of(nx);
            int df = abs(fx - file_of(x)), dr = abs(rx - rank_of(x));
            if (d == 1 || d == -1) {
                if (rx != rank_of(x)) break;
            } else if (d == 8 || d == -8) {
                if (fx != file_of(x)) break;
            } else {
                if (df != 1 || dr != 1) break;
            }
            a |= bit(nx);
            if (occ & bit(nx)) break;
            x = nx;
        }
    }
    return a;
}

struct Move {
    int from, to;
    int piece;
    int promo;
    bool capture;
    bool ep;
    bool castle;
};

struct Position {
    array<int, 64> board{};
    U64 bb[12]{};
    U64 occ[2]{};
    U64 all = 0;
    int side = WHITE;
    int castle = 0;
    int ep = -1;
    int halfmove = 0;
    int fullmove = 1;
    uint64_t hash = 0;
};

uint64_t splitmix64(uint64_t &x) {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
}

uint64_t ZPiece[12][64], ZCastle[16], ZEp[8], ZSide;

uint64_t compute_hash(const Position &p) {
    uint64_t h = 0;
    for (int s = 0; s < 64; ++s) {
        int pc = p.board[s];
        if (pc != PIECE_NONE) h ^= ZPiece[pc][s];
    }
    h ^= ZCastle[p.castle];
    if (p.ep != -1) h ^= ZEp[file_of(p.ep)];
    if (p.side == BLACK) h ^= ZSide;
    return h;
}

int piece_color(int p) { return p < 6 ? WHITE : BLACK; }
int piece_type(int p) { return p % 6; }

void add_piece(Position &p, int sqr, int pc) {
    p.board[sqr] = pc;
    p.bb[pc] |= bit(sqr);
    p.occ[piece_color(pc)] |= bit(sqr);
    p.all |= bit(sqr);
}

void remove_piece(Position &p, int sqr, int pc) {
    p.board[sqr] = PIECE_NONE;
    p.bb[pc] &= ~bit(sqr);
    p.occ[piece_color(pc)] &= ~bit(sqr);
    p.all &= ~bit(sqr);
}

void move_piece(Position &p, int from, int to, int pc) {
    remove_piece(p, from, pc);
    add_piece(p, to, pc);
}

int king_square(const Position &p, int c) {
    U64 b = p.bb[c == WHITE ? WK : BK];
    return b ? std::countr_zero(b) : -1;
}

bool attacked(const Position &p, int sqr, int by) {
    if (by == WHITE) {
        U64 paw = p.bb[WP];
        U64 att = pawn_attacks[BLACK][sqr] & paw;
        if (att) return true;
    } else {
        U64 paw = p.bb[BP];
        U64 att = pawn_attacks[WHITE][sqr] & paw;
        if (att) return true;
    }
    if (knight_attacks[sqr] & (by == WHITE ? p.bb[WN] : p.bb[BN])) return true;
    if (king_attacks[sqr] & (by == WHITE ? p.bb[WK] : p.bb[BK])) return true;
    static const int bishop_dirs[4] = {9, 7, -7, -9};
    static const int rook_dirs[4] = {1, -1, 8, -8};
    U64 occ = p.all;
    for (int d : bishop_dirs) {
        int x = sqr;
        while (true) {
            int y = x + d;
            if (!on_board(y)) break;
            if (abs(file_of(y) - file_of(x)) != 1) break;
            int pc = p.board[y];
            if (pc != PIECE_NONE) {
                if (piece_color(pc) == by && (piece_type(pc) == 2 || piece_type(pc) == 4)) return true;
                break;
            }
            x = y;
        }
    }
    for (int d : rook_dirs) {
        int x = sqr;
        while (true) {
            int y = x + d;
            if (!on_board(y)) break;
            if ((d == 1 || d == -1) && rank_of(y) != rank_of(x)) break;
            int pc = p.board[y];
            if (pc != PIECE_NONE) {
                if (piece_color(pc) == by && (piece_type(pc) == 3 || piece_type(pc) == 4)) return true;
                break;
            }
            x = y;
        }
    }
    return false;
}

bool in_check(const Position &p, int c) {
    return attacked(p, king_square(p, c), c ^ 1);
}

void make_move(Position &p, const Move &m, int &captured, int &oldCastle, int &oldEp, int &oldHalf, int &oldFull, int &oldSide) {
    oldCastle = p.castle; oldEp = p.ep; oldHalf = p.halfmove; oldFull = p.fullmove; oldSide = p.side;
    captured = PIECE_NONE;
    int mover = p.board[m.from];
    if (m.ep) {
        int capSq = p.side == WHITE ? m.to - 8 : m.to + 8;
        captured = p.board[capSq];
        remove_piece(p, capSq, captured);
    } else if (m.capture) {
        captured = p.board[m.to];
        remove_piece(p, m.to, captured);
    }
    remove_piece(p, m.from, mover);
    int placed = mover;
    if (m.promo != PIECE_NONE) placed = m.promo;
    add_piece(p, m.to, placed);
    if (m.castle) {
        if (m.to == 6) { remove_piece(p, 7, WR); add_piece(p, 5, WR); }
        else if (m.to == 2) { remove_piece(p, 0, WR); add_piece(p, 3, WR); }
        else if (m.to == 62) { remove_piece(p, 63, BR); add_piece(p, 61, BR); }
        else if (m.to == 58) { remove_piece(p, 56, BR); add_piece(p, 59, BR); }
    }
    p.ep = -1;
    if (mover == WP && m.to - m.from == 16) p.ep = m.from + 8;
    if (mover == BP && m.from - m.to == 16) p.ep = m.from - 8;
    if (mover == WK) p.castle &= ~3;
    if (mover == BK) p.castle &= ~12;
    if (m.from == 0 || m.to == 0) p.castle &= ~2;
    if (m.from == 7 || m.to == 7) p.castle &= ~1;
    if (m.from == 56 || m.to == 56) p.castle &= ~8;
    if (m.from == 63 || m.to == 63) p.castle &= ~4;
    if (captured != PIECE_NONE) p.halfmove = 0;
    else if (mover == WP || mover == BP) p.halfmove = 0;
    else ++p.halfmove;
    if (p.side == BLACK) ++p.fullmove;
    p.side ^= 1;
    p.hash = compute_hash(p);
}

void unmake_move(Position &p, const Move &m, int captured, int oldCastle, int oldEp, int oldHalf, int oldFull, int oldSide) {
    p.side = oldSide;
    p.castle = oldCastle; p.ep = oldEp; p.halfmove = oldHalf; p.fullmove = oldFull;
    int mover = p.board[m.to];
    if (m.castle) {
        if (m.to == 6) { remove_piece(p, 5, WR); add_piece(p, 7, WR); }
        else if (m.to == 2) { remove_piece(p, 3, WR); add_piece(p, 0, WR); }
        else if (m.to == 62) { remove_piece(p, 61, BR); add_piece(p, 63, BR); }
        else if (m.to == 58) { remove_piece(p, 59, BR); add_piece(p, 56, BR); }
    }
    remove_piece(p, m.to, mover);
    int placed = m.promo != PIECE_NONE ? m.promo : p.board[m.from];
    int original = (m.promo != PIECE_NONE) ? (oldSide == WHITE ? WP : BP) : mover;
    add_piece(p, m.from, original);
    if (m.ep) {
        int capSq = oldSide == WHITE ? m.to - 8 : m.to + 8;
        add_piece(p, capSq, captured);
    } else if (captured != PIECE_NONE) {
        add_piece(p, m.to, captured);
    }
    p.hash = compute_hash(p);
}

vector<Move> gen_moves(const Position &p, bool capturesOnly = false) {
    vector<Move> mv;
    int us = p.side, them = us ^ 1;
    U64 occ = p.all, own = p.occ[us], opp = p.occ[them];
    auto add = [&](int f, int t, int piece, int promo = PIECE_NONE, bool cap = false, bool ep = false, bool cs = false) {
        mv.push_back({f, t, piece, promo, cap, ep, cs});
    };
    if (us == WHITE) {
        U64 paw = p.bb[WP];
        while (paw) {
            int s = poplsb(paw);
            int r = rank_of(s), f = file_of(s);
            int one = s + 8;
            if (!capturesOnly && on_board(one) && p.board[one] == PIECE_NONE) {
                if (r == 6) { add(s, one, WP, WQ); add(s, one, WP, WR); add(s, one, WP, WB); add(s, one, WP, WN); }
                else {
                    add(s, one, WP);
                    if (r == 1 && p.board[s + 16] == PIECE_NONE) add(s, s + 16, WP);
                }
            }
            if (f > 0) {
                int t = s + 7;
                if (on_board(t) && (opp & bit(t))) {
                    if (r == 6) { add(s, t, WP, WQ, true); add(s, t, WP, WR, true); add(s, t, WP, WB, true); add(s, t, WP, WN, true); }
                    else add(s, t, WP, PIECE_NONE, true);
                }
                if (t == p.ep) add(s, t, WP, PIECE_NONE, true, true);
            }
            if (f < 7) {
                int t = s + 9;
                if (on_board(t) && (opp & bit(t))) {
                    if (r == 6) { add(s, t, WP, WQ, true); add(s, t, WP, WR, true); add(s, t, WP, WB, true); add(s, t, WP, WN, true); }
                    else add(s, t, WP, PIECE_NONE, true);
                }
                if (t == p.ep) add(s, t, WP, PIECE_NONE, true, true);
            }
        }
    } else {
        U64 paw = p.bb[BP];
        while (paw) {
            int s = poplsb(paw);
            int r = rank_of(s), f = file_of(s);
            int one = s - 8;
            if (!capturesOnly && on_board(one) && p.board[one] == PIECE_NONE) {
                if (r == 1) { add(s, one, BP, BQ); add(s, one, BP, BR); add(s, one, BP, BB); add(s, one, BP, BN); }
                else {
                    add(s, one, BP);
                    if (r == 6 && p.board[s - 16] == PIECE_NONE) add(s, s - 16, BP);
                }
            }
            if (f > 0) {
                int t = s - 9;
                if (on_board(t) && (opp & bit(t))) {
                    if (r == 1) { add(s, t, BP, BQ, true); add(s, t, BP, BR, true); add(s, t, BP, BB, true); add(s, t, BP, BN, true); }
                    else add(s, t, BP, PIECE_NONE, true);
                }
                if (t == p.ep) add(s, t, BP, PIECE_NONE, true, true);
            }
            if (f < 7) {
                int t = s - 7;
                if (on_board(t) && (opp & bit(t))) {
                    if (r == 1) { add(s, t, BP, BQ, true); add(s, t, BP, BR, true); add(s, t, BP, BB, true); add(s, t, BP, BN, true); }
                    else add(s, t, BP, PIECE_NONE, true);
                }
                if (t == p.ep) add(s, t, BP, PIECE_NONE, true, true);
            }
        }
    }
    U64 knights = p.bb[us == WHITE ? WN : BN];
    while (knights) {
        int s = poplsb(knights);
        U64 a = knight_attacks[s] & ~own;
        while (a) {
            int t = poplsb(a);
            add(s, t, us == WHITE ? WN : BN, PIECE_NONE, (opp & bit(t)));
        }
    }
    U64 bishops = p.bb[us == WHITE ? WB : BB];
    static const int bdirs[4] = {9, 7, -7, -9};
    while (bishops) {
        int s = poplsb(bishops);
        for (int d : bdirs) {
            int x = s;
            while (true) {
                int y = x + d;
                if (!on_board(y) || abs(file_of(y) - file_of(x)) != 1) break;
                if (own & bit(y)) break;
                add(s, y, us == WHITE ? WB : BB, PIECE_NONE, (opp & bit(y)));
                if (occ & bit(y)) break;
                x = y;
            }
        }
    }
    U64 rooks = p.bb[us == WHITE ? WR : BR];
    static const int rdirs[4] = {1, -1, 8, -8};
    while (rooks) {
        int s = poplsb(rooks);
        for (int d : rdirs) {
            int x = s;
            while (true) {
                int y = x + d;
                if (!on_board(y)) break;
                if ((d == 1 || d == -1) && rank_of(y) != rank_of(x)) break;
                if (own & bit(y)) break;
                add(s, y, us == WHITE ? WR : BR, PIECE_NONE, (opp & bit(y)));
                if (occ & bit(y)) break;
                x = y;
            }
        }
    }
    U64 queens = p.bb[us == WHITE ? WQ : BQ];
    while (queens) {
        int s = poplsb(queens);
        for (int d : bdirs) {
            int x = s;
            while (true) {
                int y = x + d;
                if (!on_board(y) || abs(file_of(y) - file_of(x)) != 1) break;
                if (own & bit(y)) break;
                add(s, y, us == WHITE ? WQ : BQ, PIECE_NONE, (opp & bit(y)));
                if (occ & bit(y)) break;
                x = y;
            }
        }
        for (int d : rdirs) {
            int x = s;
            while (true) {
                int y = x + d;
                if (!on_board(y)) break;
                if ((d == 1 || d == -1) && rank_of(y) != rank_of(x)) break;
                if (own & bit(y)) break;
                add(s, y, us == WHITE ? WQ : BQ, PIECE_NONE, (opp & bit(y)));
                if (occ & bit(y)) break;
                x = y;
            }
        }
    }
    int ks = king_square(p, us);
    U64 km = king_attacks[ks] & ~own;
    while (km) {
        int t = poplsb(km);
        add(ks, t, us == WHITE ? WK : BK, PIECE_NONE, (opp & bit(t)));
    }
    if (!capturesOnly) {
        if (us == WHITE) {
            if ((p.castle & 1) && p.board[5] == PIECE_NONE && p.board[6] == PIECE_NONE && !attacked(p, 4, BLACK) && !attacked(p, 5, BLACK) && !attacked(p, 6, BLACK))
                add(4, 6, WK, PIECE_NONE, false, false, true);
            if ((p.castle & 2) && p.board[1] == PIECE_NONE && p.board[2] == PIECE_NONE && p.board[3] == PIECE_NONE && !attacked(p, 4, BLACK) && !attacked(p, 3, BLACK) && !attacked(p, 2, BLACK))
                add(4, 2, WK, PIECE_NONE, false, false, true);
        } else {
            if ((p.castle & 4) && p.board[61] == PIECE_NONE && p.board[62] == PIECE_NONE && !attacked(p, 60, WHITE) && !attacked(p, 61, WHITE) && !attacked(p, 62, WHITE))
                add(60, 62, BK, PIECE_NONE, false, false, true);
            if ((p.castle & 8) && p.board[57] == PIECE_NONE && p.board[58] == PIECE_NONE && p.board[59] == PIECE_NONE && !attacked(p, 60, WHITE) && !attacked(p, 59, WHITE) && !attacked(p, 58, WHITE))
                add(60, 58, BK, PIECE_NONE, false, false, true);
        }
    }
    vector<Move> legal;
    legal.reserve(mv.size());
    for (auto &m : mv) {
        int cap, oc, oe, oh, of, os;
        make_move(const_cast<Position&>(p), m, cap, oc, oe, oh, of, os);
        bool ok = !in_check(p, us);
        unmake_move(const_cast<Position&>(p), m, cap, oc, oe, oh, of, os);
        if (ok) legal.push_back(m);
    }
    return legal;
}

int pst[6][64];

int eval(const Position &p) {
    static const int val[6] = {100, 320, 330, 500, 900, 0};
    int s = 0;
    for (int sqr = 0; sqr < 64; ++sqr) {
        int pc = p.board[sqr];
        if (pc == PIECE_NONE) continue;
        int t = piece_type(pc), c = piece_color(pc);
        int idx = c == WHITE ? sqr : (63 - sqr);
        int v = val[t] + pst[t][idx];
        s += c == WHITE ? v : -v;
    }
    return p.side == WHITE ? s : -s;
}

struct TTEntry { uint64_t key; int depth; int flag; int score; Move best; };
unordered_map<uint64_t, TTEntry> tt;

int quiesce(Position &p, int alpha, int beta, chrono::steady_clock::time_point end, bool &stopped);

int negamax(Position &p, int depth, int alpha, int beta, chrono::steady_clock::time_point end, bool &stopped) {
    if (chrono::steady_clock::now() >= end) { stopped = true; return 0; }
    auto it = tt.find(p.hash);
    if (it != tt.end() && it->second.depth >= depth) {
        int sc = it->second.score;
        if (it->second.flag == 0) return sc;
        if (it->second.flag == 1 && sc <= alpha) return alpha;
        if (it->second.flag == 2 && sc >= beta) return beta;
    }
    if (depth == 0) return quiesce(p, alpha, beta, end, stopped);
    vector<Move> moves = gen_moves(p, false);
    if (moves.empty()) return in_check(p, p.side) ? -MATE + (10 - depth) : 0;
    Move ttbest{};
    if (it != tt.end()) ttbest = it->second.best;
    auto score_move = [&](const Move &m) {
        int s = 0;
        if (ttbest.from == m.from && ttbest.to == m.to && ttbest.promo == m.promo && ttbest.castle == m.castle) s += 1000000;
        if (m.capture) s += 10000 + (m.promo != PIECE_NONE ? 5000 : 0);
        if (m.promo != PIECE_NONE) s += 8000;
        if (m.castle) s += 1000;
        return s;
    };
    sort(moves.begin(), moves.end(), [&](const Move &a, const Move &b){ return score_move(a) > score_move(b); });
    int bestScore = -INF, origAlpha = alpha;
    Move bestMove{};
    for (auto &m : moves) {
        int cap, oc, oe, oh, of, os;
        make_move(p, m, cap, oc, oe, oh, of, os);
        int sc = -negamax(p, depth - 1, -beta, -alpha, end, stopped);
        unmake_move(p, m, cap, oc, oe, oh, of, os);
        if (stopped) return 0;
        if (sc > bestScore) { bestScore = sc; bestMove = m; }
        if (sc > alpha) alpha = sc;
        if (alpha >= beta) break;
    }
    TTEntry e{p.hash, depth, bestScore <= origAlpha ? 1 : (bestScore >= beta ? 2 : 0), bestScore, bestMove};
    tt[p.hash] = e;
    return bestScore;
}

int quiesce(Position &p, int alpha, int beta, chrono::steady_clock::time_point end, bool &stopped) {
    if (chrono::steady_clock::now() >= end) { stopped = true; return 0; }
    int stand = eval(p);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;
    vector<Move> moves = gen_moves(p, true);
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b) {
        if (a.promo != b.promo) return a.promo > b.promo;
        return a.capture > b.capture;
    });
    for (auto &m : moves) {
        int cap, oc, oe, oh, of, os;
        make_move(p, m, cap, oc, oe, oh, of, os);
        int sc = -quiesce(p, -beta, -alpha, end, stopped);
        unmake_move(p, m, cap, oc, oe, oh, of, os);
        if (stopped) return 0;
        if (sc >= beta) return beta;
        if (sc > alpha) alpha = sc;
    }
    return alpha;
}

string move_to_string(const Move &m, const Position &before, const Position &after) {
    if (m.castle) return m.to == 6 || m.to == 62 ? "O-O" : "O-O-O";
    auto piece_char = [&](int pc) {
        static const char* names = "pNBRQK";
        char c = names[piece_type(pc)];
        return c;
    };
    int pc = before.board[m.from];
    string s;
    s.push_back(piece_char(pc));
    s += sq(m.from);
    s += m.capture ? "x" : "";
    s += sq(m.to);
    if (m.ep) s += "ep";
    if (m.promo != PIECE_NONE) {
        s.push_back('/');
        s.push_back(piece_char(m.promo));
    }
    if (in_check(after, after.side)) {
        auto replies = gen_moves(after, false);
        s += replies.empty() ? "#" : "+";
    }
    return s;
}

Position parse_fen(const string &fen) {
    Position p;
    stringstream ss(fen);
    string board, stm, cast, ep;
    ss >> board >> stm >> cast >> ep >> p.halfmove >> p.fullmove;
    int sqr = 56;
    for (char c : board) {
        if (c == '/') { sqr -= 16; continue; }
        if (isdigit((unsigned char)c)) { sqr += c - '0'; continue; }
        int pc = PIECE_NONE;
        switch (c) {
            case 'P': pc = WP; break; case 'N': pc = WN; break; case 'B': pc = WB; break;
            case 'R': pc = WR; break; case 'Q': pc = WQ; break; case 'K': pc = WK; break;
            case 'p': pc = BP; break; case 'n': pc = BN; break; case 'b': pc = BB; break;
            case 'r': pc = BR; break; case 'q': pc = BQ; break; case 'k': pc = BK; break;
        }
        add_piece(p, sqr++, pc);
    }
    p.side = stm == "w" ? WHITE : BLACK;
    p.castle = 0;
    if (cast.find('K') != string::npos) p.castle |= 1;
    if (cast.find('Q') != string::npos) p.castle |= 2;
    if (cast.find('k') != string::npos) p.castle |= 4;
    if (cast.find('q') != string::npos) p.castle |= 8;
    if (ep != "-") {
        int f = ep[0] - 'a', r = ep[1] - '1';
        p.ep = r * 8 + f;
    }
    p.hash = compute_hash(p);
    return p;
}

void init_tables() {
    for (int s = 0; s < 64; ++s) {
        int f = file_of(s), r = rank_of(s);
        U64 k = 0;
        const int kn[8][2] = {{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
        for (auto &d : kn) {
            int nf = f + d[0], nr = r + d[1];
            if (0 <= nf && nf < 8 && 0 <= nr && nr < 8) k |= bit(nr * 8 + nf);
        }
        knight_attacks[s] = k;
        U64 g = 0;
        for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) {
            if (!df && !dr) continue;
            int nf = f + df, nr = r + dr;
            if (0 <= nf && nf < 8 && 0 <= nr && nr < 8) g |= bit(nr * 8 + nf);
        }
        king_attacks[s] = g;
        pawn_attacks[WHITE][s] = 0;
        pawn_attacks[BLACK][s] = 0;
        if (f > 0 && r < 7) pawn_attacks[WHITE][s] |= bit((r + 1) * 8 + (f - 1));
        if (f < 7 && r < 7) pawn_attacks[WHITE][s] |= bit((r + 1) * 8 + (f + 1));
        if (f > 0 && r > 0) pawn_attacks[BLACK][s] |= bit((r - 1) * 8 + (f - 1));
        if (f < 7 && r > 0) pawn_attacks[BLACK][s] |= bit((r - 1) * 8 + (f + 1));
    }
    uint64_t seed = 20260515;
    for (int p = 0; p < 12; ++p) for (int s = 0; s < 64; ++s) ZPiece[p][s] = splitmix64(seed);
    for (int i = 0; i < 16; ++i) ZCastle[i] = splitmix64(seed);
    for (int i = 0; i < 8; ++i) ZEp[i] = splitmix64(seed);
    ZSide = splitmix64(seed);
    memset(pst, 0, sizeof(pst));
    for (int r = 0; r < 8; ++r) for (int f = 0; f < 8; ++f) {
        int s = r * 8 + f;
        int center = 3 - abs(f - 3) + 3 - abs(r - 3);
        pst[WP][s] = r * 6 + center * 2;
        pst[WN][s] = center * 4;
        pst[WB][s] = center * 3;
        pst[WR][s] = r * 2;
        pst[WQ][s] = center * 2;
        pst[WK][s] = -(abs(f - 3) + abs(r - 0)) * 2;
    }
}
} // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    init_tables();
    double timeout;
    string fen;
    if (!(cin >> timeout)) return 0;
    getline(cin, fen);
    if (fen.empty()) getline(cin, fen);
    Position pos = parse_fen(fen);
    auto end = chrono::steady_clock::now() + chrono::milliseconds((int)(timeout * 1000.0) - 50);
    bool stopped = false;
    vector<Move> root = gen_moves(pos, false);
    if (root.empty()) {
        cout << "O-O\n";
        return 0;
    }
    Move best = root[0];
    int bestScore = -INF;
    for (int depth = 1; depth <= 12 && !stopped; ++depth) {
        int localBest = -INF;
        Move localMove = best;
        auto ordered = root;
        sort(ordered.begin(), ordered.end(), [](const Move &a, const Move &b) {
            if (a.capture != b.capture) return a.capture > b.capture;
            if (a.promo != b.promo) return a.promo > b.promo;
            return a.castle > b.castle;
        });
        for (auto &m : ordered) {
            int cap, oc, oe, oh, of, os;
            make_move(pos, m, cap, oc, oe, oh, of, os);
            int sc = -negamax(pos, depth - 1, -INF, INF, end, stopped);
            unmake_move(pos, m, cap, oc, oe, oh, of, os);
            if (stopped) break;
            if (sc > localBest) { localBest = sc; localMove = m; }
        }
        if (!stopped) { best = localMove; bestScore = localBest; }
    }
    int cap, oc, oe, oh, of, os;
    Position after = pos;
    make_move(after, best, cap, oc, oe, oh, of, os);
    cout << move_to_string(best, pos, after) << '\n';
    return 0;
}
