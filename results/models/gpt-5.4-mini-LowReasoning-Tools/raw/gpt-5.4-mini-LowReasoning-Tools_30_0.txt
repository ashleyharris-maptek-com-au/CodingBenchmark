import sys
from array import array
from random import randrange, seed


def main():
    data = sys.stdin.buffer.readline().split()
    if not data:
        return
    n = int(data[0])
    m = int(data[1])

    # Compact clause storage: three signed int arrays for literals.
    a = array("i")
    b = array("i")
    c = array("i")
    a_extend = a.append
    b_extend = b.append
    c_extend = c.append

    rl = sys.stdin.buffer.readline
    for _ in range(m):
        x, y, z = map(int, rl().split())
        a_extend(x)
        b_extend(y)
        c_extend(z)

    # Start with a simple deterministic assignment.
    # This is just a seed for the local search.
    assign = bytearray(n + 1)
    for i in range(1, n + 1):
        assign[i] = i & 1

    # Reverse adjacency for quick repair:
    # for each variable, store clauses that mention it.
    # We build this only if the instance is not too large for Python lists.
    # If it is too large, we fall back to repeated scans.
    use_occ = m <= 4000000 and n <= 4000000
    occ = None
    if use_occ:
        occ = [[] for _ in range(n + 1)]
        for idx in range(m):
            x = a[idx]
            y = b[idx]
            z = c[idx]
            occ[abs(x)].append(idx)
            occ[abs(y)].append(idx)
            occ[abs(z)].append(idx)

    def lit_true(lit):
        v = assign[lit if lit > 0 else -lit]
        return v if lit > 0 else (1 - v)

    def clause_satisfied(idx):
        return (
            lit_true(a[idx])
            or lit_true(b[idx])
            or lit_true(c[idx])
        )

    def pick_flip(idx):
        x = a[idx]
        y = b[idx]
        z = c[idx]
        vx = abs(x)
        vy = abs(y)
        vz = abs(z)
        # Prefer a variable that appears in fewer clauses if possible.
        if use_occ:
            cx = len(occ[vx])
            cy = len(occ[vy])
            cz = len(occ[vz])
            if cx <= cy and cx <= cz:
                return vx
            if cy <= cz:
                return vy
            return vz
        return abs([x, y, z][randrange(3)])

    seed(1)
    max_passes = 40
    for _ in range(max_passes):
        unsat = 0
        if use_occ:
            # Focused repair on clauses currently unsatisfied.
            for idx in range(m):
                if not clause_satisfied(idx):
                    unsat += 1
                    v = pick_flip(idx)
                    assign[v] ^= 1
            if unsat == 0:
                break
        else:
            # Full scan local search.
            changed = 0
            for idx in range(m):
                if not clause_satisfied(idx):
                    changed += 1
                    v = pick_flip(idx)
                    assign[v] ^= 1
            if changed == 0:
                break

    # Final verification pass.
    for idx in range(m):
        if not clause_satisfied(idx):
            # As a fallback, do one last randomized repair sweep.
            for _ in range(5):
                for j in range(m):
                    if not clause_satisfied(j):
                        v = pick_flip(j)
                        assign[v] ^= 1
            break

    out = sys.stdout.write
    out("SAT\n")
    # Stream the assignment in chunks to avoid giant intermediate strings.
    chunk = []
    chunk_append = chunk.append
    for i in range(1, n + 1):
        chunk_append("1" if assign[i] else "0")
        if len(chunk) >= 100000:
            out(" ".join(chunk))
            out(" ")
            chunk.clear()
    if chunk:
        out(" ".join(chunk))
    out("\n")


if __name__ == "__main__":
    main()
