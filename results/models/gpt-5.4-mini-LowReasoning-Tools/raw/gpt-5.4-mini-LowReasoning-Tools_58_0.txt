const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = H ? grid[0].length : 0;
const N = H * W;

function idx(x, y) {
  return y * W + x;
}

function inBounds(x, y) {
  return x >= 0 && x < W && y >= 0 && y < H;
}

const passable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    passable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const robotPos = new Int32Array(robots.length);
const robotCell = new Int32Array(N);
robotCell.fill(-1);
for (let i = 0; i < robots.length; i++) {
  const r = robots[i];
  const p = idx(r.x, r.y);
  robotPos[i] = p;
  robotCell[p] = i;
}

const assignments = Array.from({ length: robots.length }, () => []);

const jobOrder = jobs.map((job, i) => ({ job, i }));
jobOrder.sort((a, b) => {
  if (b.job.priority !== a.job.priority) return b.job.priority - a.job.priority;
  if (a.job.y !== b.job.y) return a.job.y - b.job.y;
  if (a.job.x !== b.job.x) return a.job.x - b.job.x;
  return a.i - b.i;
});

const dist = new Int32Array(N);
const seen = new Int32Array(N);
let stamp = 1;
const queue = new Int32Array(N);

function nearestRobotFromJob(start) {
  stamp++;
  let qh = 0;
  let qt = 0;
  queue[qt++] = start;
  seen[start] = stamp;
  dist[start] = 0;

  let bestRobot = -1;
  let bestDist = -1;

  while (qh < qt) {
    const v = queue[qh++];
    const d = dist[v];
    if (bestDist !== -1 && d > bestDist) break;

    const robot = robotCell[v];
    if (robot !== -1) {
      if (bestDist === -1) {
        bestDist = d;
        bestRobot = robot;
      } else if (d === bestDist && robot < bestRobot) {
        bestRobot = robot;
      }
      continue;
    }

    const x = v % W;
    const y = (v / W) | 0;

    let nv;

    if (x > 0) {
      nv = v - 1;
      if (passable[nv] && seen[nv] !== stamp) {
        seen[nv] = stamp;
        dist[nv] = d + 1;
        queue[qt++] = nv;
      }
    }
    if (x + 1 < W) {
      nv = v + 1;
      if (passable[nv] && seen[nv] !== stamp) {
        seen[nv] = stamp;
        dist[nv] = d + 1;
        queue[qt++] = nv;
      }
    }
    if (y > 0) {
      nv = v - W;
      if (passable[nv] && seen[nv] !== stamp) {
        seen[nv] = stamp;
        dist[nv] = d + 1;
        queue[qt++] = nv;
      }
    }
    if (y + 1 < H) {
      nv = v + W;
      if (passable[nv] && seen[nv] !== stamp) {
        seen[nv] = stamp;
        dist[nv] = d + 1;
        queue[qt++] = nv;
      }
    }
  }

  if (bestRobot !== -1) return bestRobot;

  let fallbackRobot = 0;
  let bestManhattan = Number.POSITIVE_INFINITY;
  const sx = start % W;
  const sy = (start / W) | 0;
  for (let i = 0; i < robots.length; i++) {
    const p = robotPos[i];
    const x = p % W;
    const y = (p / W) | 0;
    const m = Math.abs(x - sx) + Math.abs(y - sy);
    if (m < bestManhattan || (m === bestManhattan && i < fallbackRobot)) {
      bestManhattan = m;
      fallbackRobot = i;
    }
  }
  return fallbackRobot;
}

for (const entry of jobOrder) {
  const job = entry.job;
  const start = idx(job.x, job.y);
  const robotId = nearestRobotFromJob(start);
  assignments[robotId].push(job.id);
  const oldPos = robotPos[robotId];
  robotCell[oldPos] = -1;
  robotPos[robotId] = start;
  robotCell[start] = robotId;
}

process.stdout.write(JSON.stringify({ assignments }));
