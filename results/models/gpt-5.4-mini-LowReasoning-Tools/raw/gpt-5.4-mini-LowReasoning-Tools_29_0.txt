using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    private struct Edge
    {
        public int U, V;
        public Edge(int u, int v) { U = u; V = v; }
    }

    private sealed class AdjGraph
    {
        public readonly int N;
        public readonly int[][] Adj;
        public AdjGraph(int n, int[][] adj) { N = n; Adj = adj; }
    }

    private static int[] CurrentBest = Array.Empty<int>();
    private static int[] BestOrder = Array.Empty<int>();
    private static int[][]? BitAdj;
    private static int BitWordCount;
    private static int[] Degrees = Array.Empty<int>();
    private static int N;

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();

        N = n;
        var edges = new Edge[m];
        var deg = new int[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            edges[i] = new Edge(u, v);
            deg[u]++;
            deg[v]++;
        }
        Degrees = deg;

        if (n == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        if (n <= 2200)
        {
            SolveExactSmall(n, edges);
        }
        else
        {
            SolveHeuristicLarge(n, edges);
        }
    }

    private static void SolveExactSmall(int n, Edge[] edges)
    {
        BitWordCount = (n + 63) >> 6;
        var adj = new ulong[n][];
        for (int i = 0; i < n; i++) adj[i] = new ulong[BitWordCount];
        foreach (var e in edges)
        {
            if (e.U == e.V) continue;
            adj[e.U][e.V >> 6] |= 1UL << (e.V & 63);
            adj[e.V][e.U >> 6] |= 1UL << (e.U & 63);
        }
        BitAdj = new int[n][];
        var all = new int[n];
        for (int i = 0; i < n; i++) all[i] = i;
        var order = new int[n];
        Array.Copy(all, order, n);
        Array.Sort(order, (a, b) => Degrees[b].CompareTo(Degrees[a]));
        CurrentBest = new[] { order[0] };
        BestOrder = CurrentBest;
        var cand = new int[n];
        Array.Copy(order, cand, n);
        BronKerboschBit(new ulong[BitWordCount], ToUlongArray(cand, n), new ulong[BitWordCount], adj);
        OutputClique(CurrentBest);
    }

    private static ulong[] ToUlongArray(int[] verts, int len)
    {
        var arr = new ulong[BitWordCount];
        for (int i = 0; i < len; i++)
        {
            int v = verts[i];
            arr[v >> 6] |= 1UL << (v & 63);
        }
        return arr;
    }

    private static void BronKerboschBit(ulong[] r, ulong[] p, ulong[] x, ulong[][] adj)
    {
        if (PopCount(p) == 0 && PopCount(x) == 0)
        {
            int size = PopCount(r);
            if (size > CurrentBest.Length)
            {
                var clique = new int[size];
                int idx = 0;
                for (int i = 0; i < N; i++)
                    if (((r[i >> 6] >> (i & 63)) & 1UL) != 0)
                        clique[idx++] = i;
                CurrentBest = clique;
            }
            return;
        }

        int ub = PopCount(r) + PopCount(p);
        if (ub <= CurrentBest.Length) return;

        int pivot = -1;
        int bestDeg = -1;
        for (int i = 0; i < N; i++)
        {
            if (((p[i >> 6] >> (i & 63)) & 1UL) == 0) continue;
            int d = 0;
            for (int w = 0; w < BitWordCount; w++)
                d += PopCount64(p[w] & adj[i][w]);
            if (d > bestDeg) { bestDeg = d; pivot = i; }
        }

        ulong[] ext = new ulong[BitWordCount];
        if (pivot >= 0)
        {
            for (int w = 0; w < BitWordCount; w++)
                ext[w] = p[w] & ~adj[pivot][w];
        }
        else
        {
            Array.Copy(p, ext, BitWordCount);
        }

        for (int v = 0; v < N; v++)
        {
            if (((ext[v >> 6] >> (v & 63)) & 1UL) == 0) continue;
            var nr = (ulong[])r.Clone();
            nr[v >> 6] |= 1UL << (v & 63);
            var np = new ulong[BitWordCount];
            var nx = new ulong[BitWordCount];
            for (int w = 0; w < BitWordCount; w++)
            {
                np[w] = p[w] & adj[v][w];
                nx[w] = x[w] & adj[v][w];
            }
            if (PopCount(nr) + PopCount(np) > CurrentBest.Length)
                BronKerboschBit(nr, np, nx, adj);
            p[v >> 6] &= ~(1UL << (v & 63));
            x[v >> 6] |= 1UL << (v & 63);
        }
    }

    private static void SolveHeuristicLarge(int n, Edge[] edges)
    {
        var adj = BuildAdjacency(n, edges);
        int[] best = GreedyClique(adj);

        int limit = Math.Min(n, 4000);
        var seed = new int[limit];
        for (int i = 0; i < limit; i++) seed[i] = i;
        Array.Sort(seed, (a, b) => Degrees[b].CompareTo(Degrees[a]));
        var top = new List<int>(limit);
        for (int i = 0; i < limit; i++) top.Add(seed[i]);
        var refined = ExactOnSubset(adj, top);
        if (refined.Length > best.Length) best = refined;

        OutputClique(best);
    }

    private static AdjGraph BuildAdjacency(int n, Edge[] edges)
    {
        var lists = new List<int>[n];
        for (int i = 0; i < n; i++) lists[i] = new List<int>();
        foreach (var e in edges)
        {
            if (e.U == e.V) continue;
            lists[e.U].Add(e.V);
            lists[e.V].Add(e.U);
        }
        var adj = new int[n][];
        for (int i = 0; i < n; i++)
        {
            lists[i].Sort();
            adj[i] = lists[i].ToArray();
        }
        return new AdjGraph(n, adj);
    }

    private static int[] GreedyClique(AdjGraph g)
    {
        int n = g.N;
        var order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => Degrees[b].CompareTo(Degrees[a]));
        var clique = new List<int>();
        foreach (int v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!HasEdge(g.Adj, v, clique[i])) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }
        return clique.ToArray();
    }

    private static int[] ExactOnSubset(AdjGraph g, List<int> subset)
    {
        int k = subset.Count;
        if (k == 0) return Array.Empty<int>();
        var map = new Dictionary<int, int>(k);
        for (int i = 0; i < k; i++) map[subset[i]] = i;
        var sub = new int[k][];
        for (int i = 0; i < k; i++)
        {
            var list = new List<int>();
            int v = subset[i];
            foreach (int to in g.Adj[v])
                if (map.ContainsKey(to))
                    list.Add(map[to]);
            sub[i] = list.ToArray();
        }
        int[] best = Array.Empty<int>();
        var cand = new List<int>(k);
        for (int i = 0; i < k; i++) cand.Add(i);
        BronKerboschList(new List<int>(), cand, new List<int>(), sub, ref best, subset);
        return best;
    }

    private static void BronKerboschList(List<int> r, List<int> p, List<int> x, int[][] adj, ref int[] best, List<int> orig)
    {
        if (p.Count == 0 && x.Count == 0)
        {
            if (r.Count > best.Length)
            {
                var arr = new int[r.Count];
                for (int i = 0; i < r.Count; i++) arr[i] = orig[r[i]];
                best = arr;
            }
            return;
        }
        if (r.Count + p.Count <= best.Length) return;
        int pivot = -1, bestDeg = -1;
        var union = new HashSet<int>(p);
        union.UnionWith(x);
        foreach (int u in union)
        {
            int d = 0;
            foreach (int v in adj[u]) if (union.Contains(v)) d++;
            if (d > bestDeg) { bestDeg = d; pivot = u; }
        }
        var ext = new List<int>();
        if (pivot >= 0)
        {
            var neigh = new HashSet<int>(adj[pivot]);
            foreach (int v in p)
                if (!neigh.Contains(v))
                    ext.Add(v);
        }
        else ext.AddRange(p);

        foreach (int v in ext)
        {
            var nr = new List<int>(r) { v };
            var np = Intersect(p, adj[v]);
            var nx = Intersect(x, adj[v]);
            BronKerboschList(nr, np, nx, adj, ref best, orig);
            p.Remove(v);
            x.Add(v);
        }
    }

    private static List<int> Intersect(List<int> a, int[] b)
    {
        var set = new HashSet<int>(b);
        var res = new List<int>();
        foreach (int x in a) if (set.Contains(x)) res.Add(x);
        return res;
    }

    private static bool HasEdge(int[][] adj, int a, int b)
    {
        var arr = adj[a];
        int lo = 0, hi = arr.Length - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            if (arr[mid] == b) return true;
            if (arr[mid] < b) lo = mid + 1; else hi = mid - 1;
        }
        return false;
    }

    private static void OutputClique(int[] clique)
    {
        Array.Sort(clique);
        var sb = new StringBuilder();
        sb.AppendLine(clique.Length.ToString());
        if (clique.Length > 0)
        {
            for (int i = 0; i < clique.Length; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(clique[i]);
            }
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }

    private static int PopCount(ulong[] arr)
    {
        int s = 0;
        for (int i = 0; i < arr.Length; i++) s += PopCount64(arr[i]);
        return s;
    }

    private static int PopCount64(ulong x)
    {
        x -= (x >> 1) & 0x5555555555555555UL;
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        return (int)((((x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FUL) * 0x0101010101010101UL) >> 56);
    }
}
