OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%1 = OpExtInstImport "GLSL.std.450"

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f2 = OpConstant %float 2.0
%f3 = OpConstant %float 3.0
%f_half = OpConstant %float 0.5
%f_third = OpConstant %float 0.33333334
%f_two_third = OpConstant %float 0.66666669
%f_thickness = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%cdiff = OpFSub %v3float %c2 %c1

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

%d0 = OpFSub %v2float %p2xy %p0xy
%d1 = OpFSub %v2float %p3xy %p1xy
%t0 = OpExtInst %v2float %1 Normalize %d0
%t1 = OpExtInst %v2float %1 Normalize %d1

%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1

%n0x = OpFNegate %float %t0y
%n1x = OpFNegate %float %t1y
%n0 = OpCompositeConstruct %v2float %n0x %t0x
%n1 = OpCompositeConstruct %v2float %n1x %t1x

; segment 1: 0 -> 1/3
%seg1_s = OpCompositeConstruct %v4float %f0 %f0 %f0 %f1
%seg1_e = OpCompositeConstruct %v4float %f_third %f_third %f_third %f1

%s1a = OpFSub %v4float %p2 %p1
%s1b = OpVectorTimesScalar %v4float %s1a %f_third
%q1s = OpCopyObject %v4float %p1
%q1e = OpFAdd %v4float %p1 %s1b

%c1s = OpVectorTimesScalar %v3float %cdiff %f0
%c1e = OpVectorTimesScalar %v3float %cdiff %f_third
%col1s = OpFAdd %v3float %c1 %c1s
%col1e = OpFAdd %v3float %c1 %c1e

%tq1s = OpExtInst %v2float %1 Normalize %n0
%tq1e = OpExtInst %v2float %1 Normalize %n0
%off1s = OpVectorTimesScalar %v2float %tq1s %f_half
%off1e = OpVectorTimesScalar %v2float %tq1e %f_half

%off1sp = OpCompositeExtract %float %off1s 0
%off1sq = OpCompositeExtract %float %off1s 1
%off1ep = OpCompositeExtract %float %off1e 0
%off1eq = OpCompositeExtract %float %off1e 1

%off1s4 = OpCompositeConstruct %v4float %off1sp %off1sq %f0 %f0
%off1e4 = OpCompositeConstruct %v4float %off1ep %off1eq %f0 %f0
%off1s4n = OpFNegate %v4float %off1s4
%off1e4n = OpFNegate %v4float %off1e4

%v1_0 = OpFAdd %v4float %q1s %off1s4
%v1_1 = OpFAdd %v4float %q1s %off1s4n
%v1_2 = OpFAdd %v4float %q1e %off1e4
%v1_3 = OpFAdd %v4float %q1e %off1e4n

%p1_0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p1_0_ptr %v1_0
OpStore %outColor %col1s
OpEmitVertex
%p1_1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p1_1_ptr %v1_1
OpStore %outColor %col1s
OpEmitVertex
%p1_2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p1_2_ptr %v1_2
OpStore %outColor %col1e
OpEmitVertex
%p1_3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p1_3_ptr %v1_3
OpStore %outColor %col1e
OpEmitVertex
OpEndPrimitive

; segment 2: 2/3 -> 1
%s2a = OpFSub %v4float %p2 %p1
%s2b = OpVectorTimesScalar %v4float %s2a %f_two_third
%q2s = OpFAdd %v4float %p1 %s2b
%s2c = OpVectorTimesScalar %v4float %s2a %f1
%q2e = OpFAdd %v4float %p1 %s2c

%c2s = OpVectorTimesScalar %v3float %cdiff %f_two_third
%c2e = OpVectorTimesScalar %v3float %cdiff %f1
%col2s = OpFAdd %v3float %c1 %c2s
%col2e = OpFAdd %v3float %c1 %c2e

%tq2s = OpExtInst %v2float %1 Normalize %n1
%tq2e = OpExtInst %v2float %1 Normalize %n1
%off2s = OpVectorTimesScalar %v2float %tq2s %f_half
%off2e = OpVectorTimesScalar %v2float %tq2e %f_half

%off2sp = OpCompositeExtract %float %off2s 0
%off2sq = OpCompositeExtract %float %off2s 1
%off2ep = OpCompositeExtract %float %off2e 0
%off2eq = OpCompositeExtract %float %off2e 1

%off2s4 = OpCompositeConstruct %v4float %off2sp %off2sq %f0 %f0
%off2e4 = OpCompositeConstruct %v4float %off2ep %off2eq %f0 %f0
%off2s4n = OpFNegate %v4float %off2s4
%off2e4n = OpFNegate %v4float %off2e4

%v2_0 = OpFAdd %v4float %q2s %off2s4
%v2_1 = OpFAdd %v4float %q2s %off2s4n
%v2_2 = OpFAdd %v4float %q2e %off2e4
%v2_3 = OpFAdd %v4float %q2e %off2e4n

%p2_0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p2_0_ptr %v2_0
OpStore %outColor %col2s
OpEmitVertex
%p2_1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p2_1_ptr %v2_1
OpStore %outColor %col2s
OpEmitVertex
%p2_2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p2_2_ptr %v2_2
OpStore %outColor %col2e
OpEmitVertex
%p2_3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p2_3_ptr %v2_3
OpStore %outColor %col2e
OpEmitVertex
OpEndPrimitive
