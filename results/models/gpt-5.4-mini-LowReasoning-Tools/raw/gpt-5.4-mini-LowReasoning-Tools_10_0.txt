#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <vector>

using namespace std;

struct Rect {
    int w = 0;
    int h = 0;
    int id = 0;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Shelf {
    int y = 0;
    int h = 0;
    int x = 0;
};

struct Board {
    vector<Shelf> shelves;
    int usedHeight = 0;
};

static inline bool fitsOrientation(int rw, int rh, int bw, int bh) {
    return rw <= bw && rh <= bh;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) {
        return 0;
    }

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
        long long areaA = 1LL * a.w * a.h;
        long long areaB = 1LL * b.w * b.h;
        int ma = max(a.w, a.h);
        int mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        if (areaA != areaB) return areaA > areaB;
        int mina = min(a.w, a.h);
        int minb = min(b.w, b.h);
        if (mina != minb) return mina > minb;
        if (a.w != b.w) return a.w > b.w;
        return a.h > b.h;
    });

    vector<Board> boards;
    boards.reserve(max(1, N / 8));
    boards.push_back(Board{});

    vector<Placement> ans(N);

    auto placeInBoard = [&](Board& board, const Rect& r, int boardIndex, Placement& out) -> bool {
        int bestShelf = -1;
        int bestWaste = numeric_limits<int>::max();
        int bestRot = 0;
        int bestW = 0, bestH = 0;

        for (int s = 0; s < (int)board.shelves.size(); ++s) {
            const Shelf& shelf = board.shelves[s];
            if (r.h <= shelf.h && r.w <= BW - shelf.x) {
                int waste = (BW - shelf.x) - r.w;
                if (waste < bestWaste) {
                    bestWaste = waste;
                    bestShelf = s;
                    bestRot = 0;
                    bestW = r.w;
                    bestH = r.h;
                }
            }
            if (r.w != r.h && r.w <= shelf.h && r.h <= BW - shelf.x) {
                int waste = (BW - shelf.x) - r.h;
                if (waste < bestWaste) {
                    bestWaste = waste;
                    bestShelf = s;
                    bestRot = 1;
                    bestW = r.h;
                    bestH = r.w;
                }
            }
        }

        if (bestShelf != -1) {
            Shelf& shelf = board.shelves[bestShelf];
            out.board = boardIndex;
            out.x = shelf.x;
            out.y = shelf.y;
            out.rotated = bestRot;
            shelf.x += bestW;
            return true;
        }

        // Start a new shelf if possible.
        int orient = -1;
        int pw = 0, ph = 0;
        bool ok0 = fitsOrientation(r.w, r.h, BW, BH);
        bool ok1 = fitsOrientation(r.h, r.w, BW, BH);
        if (ok0 && ok1) {
            if (r.h <= r.w) {
                orient = 0;
                pw = r.w;
                ph = r.h;
            } else {
                orient = 1;
                pw = r.h;
                ph = r.w;
            }
        } else if (ok0) {
            orient = 0;
            pw = r.w;
            ph = r.h;
        } else if (ok1) {
            orient = 1;
            pw = r.h;
            ph = r.w;
        }
        if (orient == -1) return false;

        if (board.usedHeight + ph > BH) return false;

        Shelf shelf;
        shelf.y = board.usedHeight;
        shelf.h = ph;
        shelf.x = pw;
        board.shelves.push_back(shelf);
        board.usedHeight += ph;

        out.board = boardIndex;
        out.x = 0;
        out.y = shelf.y;
        out.rotated = orient;
        return true;
    };

    for (const Rect& r : rects) {
        bool placed = false;

        // Try to fit in an existing board before opening a new one.
        for (int b = 0; b < (int)boards.size(); ++b) {
            if (placeInBoard(boards[b], r, b, ans[r.id])) {
                placed = true;
                break;
            }
        }

        if (!placed) {
            boards.push_back(Board{});
            int b = (int)boards.size() - 1;
            if (!placeInBoard(boards[b], r, b, ans[r.id])) {
                cout << 0 << '\n';
                for (int i = 0; i < N; ++i) {
                    cout << 0 << ' ' << 0 << ' ' << 0 << ' ' << 0 << '\n';
                }
                return 0;
            }
        }
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }

    return 0;
}
