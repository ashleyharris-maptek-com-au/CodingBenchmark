const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const px = +depot[0] || 0;
const py = +depot[1] || 0;
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];

const n = packages.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

function dist2Point(x1, y1, x2, y2) {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return dx * dx + dy * dy;
}

for (const p of packages) {
  p.dx = p.x - px;
  p.dy = p.y - py;
  p.ang = Math.atan2(p.dy, p.dx);
  p.rad = p.dx * p.dx + p.dy * p.dy;
}

packages.sort((a, b) => {
  if (a.ang !== b.ang) return a.ang - b.ang;
  if (a.rad !== b.rad) return a.rad - b.rad;
  return String(a.id).localeCompare(String(b.id));
});

function buildSingleRoute() {
  const route = packages.slice();
  if (route.length <= 1) return route;

  if (route.length <= 600) {
    const used = new Array(route.length).fill(false);
    const ordered = [];
    let cx = px;
    let cy = py;
    for (let step = 0; step < route.length; step++) {
      let best = -1;
      let bestD = Infinity;
      for (let i = 0; i < route.length; i++) {
        if (used[i]) continue;
        const p = route[i];
        const d = dist2Point(cx, cy, p.x, p.y);
        if (d < bestD) {
          bestD = d;
          best = i;
        }
      }
      used[best] = true;
      const p = route[best];
      ordered.push(p);
      cx = p.x;
      cy = p.y;
    }
    return ordered;
  }

  return route;
}

const route = buildSingleRoute();
const routes = route.length ? [route.map(p => p.id)] : [];

process.stdout.write(JSON.stringify({ routes }));
