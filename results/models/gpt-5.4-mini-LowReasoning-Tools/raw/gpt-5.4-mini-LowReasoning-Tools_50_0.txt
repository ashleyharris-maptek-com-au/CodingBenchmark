#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohSum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint jbase = j * 2u;
        vec3 otherPos = inData[jbase].xyz;
        vec3 otherVel = inData[jbase + 1u].xyz;

        vec3 delta = otherPos - pos;
        float dist2 = dot(delta, delta);
        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            // Repel from neighbors within the separation radius.
            sepForce -= delta / sqrt(dist2);
            ++sepCount;
        }
        if (dist2 < alignRadius2) {
            // Match neighboring velocities within the alignment radius.
            alignForce += otherVel;
            ++alignCount;
        }
        if (dist2 < cohRadius2) {
            // Move toward the average position of nearby boids.
            cohSum += otherPos;
            ++cohCount;
        }
    }

    if (sepCount > 0u) {
        sepForce /= float(sepCount);
    }

    vec3 alignForceAvg = vec3(0.0);
    if (alignCount > 0u) {
        alignForceAvg = (alignForce / float(alignCount)) - vel;
    }

    vec3 cohForce = vec3(0.0);
    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        cohForce = center - pos;
    }

    vec3 accel = sepForce * weights.x + alignForceAvg * weights.y + cohForce * weights.z;

    float boundSize = bounds.x;
    float boundForce = bounds.y;
    float boundaryLimit = boundSize * 0.4;
    float boundaryScale = boundForce / max(boundSize * 0.5, 1e-6);

    // Apply a restoring force when outside the inner boundary region.
    if (pos.x > boundaryLimit) accel.x -= (pos.x - boundaryLimit) * boundaryScale;
    if (pos.x < -boundaryLimit) accel.x += (-boundaryLimit - pos.x) * boundaryScale;
    if (pos.y > boundaryLimit) accel.y -= (pos.y - boundaryLimit) * boundaryScale;
    if (pos.y < -boundaryLimit) accel.y += (-boundaryLimit - pos.y) * boundaryScale;
    if (pos.z > boundaryLimit) accel.z -= (pos.z - boundaryLimit) * boundaryScale;
    if (pos.z < -boundaryLimit) accel.z += (-boundaryLimit - pos.z) * boundaryScale;

    vel += accel * dt;

    float maxSpeed = weights.w;
    float speed2 = dot(vel, vel);
    float maxSpeed2 = maxSpeed * maxSpeed;
    if (speed2 > maxSpeed2 && speed2 > 0.0) {
        vel *= maxSpeed / sqrt(speed2);
    }

    pos += vel * dt;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
