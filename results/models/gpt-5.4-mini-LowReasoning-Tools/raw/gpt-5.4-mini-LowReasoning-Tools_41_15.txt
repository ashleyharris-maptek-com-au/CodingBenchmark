; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
%1 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%c0 = OpConstant %int 0
%c1 = OpConstant %int 1
%c3 = OpConstant %int 3

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0

%cool = OpConstantComposite %v3float %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_4
%alpha = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
%ptr_ubo_UBO = OpTypePointer Uniform %UBO
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_v4float = OpTypePointer Uniform %v4float
%ptr_uniform_v3float = OpTypePointer Uniform %v3float

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%worldPos = OpVariable %ptr_input_v3float Input
%normal = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo = OpVariable %ptr_ubo_UBO Uniform

%main = OpFunction %void None %func
%label = OpLabel
%worldPos_val = OpLoad %v3float %worldPos
%normal_val = OpLoad %v3float %normal

%ubo_light_ptr = OpAccessChain %ptr_uniform_v4float %ubo %c3
%lightPos = OpLoad %v4float %ubo_light_ptr

%n = OpExtInst %v3float %1 Normalize %normal_val
%light_xyz = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%toLight = OpFSub %v3float %light_xyz %worldPos_val
%l = OpExtInst %v3float %1 Normalize %toLight

%ndotl = OpDot %float %n %l
%scaled = OpFMul %float %ndotl %float_0_5
%t = OpFAdd %float %scaled %float_0_5

%one_minus_t = OpFSub %float %float_1 %t
%cool_scaled = OpVectorTimesScalar %v3float %cool %one_minus_t
%warm_scaled = OpVectorTimesScalar %v3float %warm %t
%rgb = OpFAdd %v3float %cool_scaled %warm_scaled

%out = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
