OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%c0 = OpConstant %int 0
%c1 = OpConstant %int 1
%c2 = OpConstant %int 2
%c3 = OpConstant %int 3
%c4 = OpConstant %int 4
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%fh = OpConstant %float 0.04
%negfh = OpConstantFNegate %float %fh

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %c1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %c1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %funct
%entry = OpLabel

%p_ptr = OpAccessChain %ptr_in_v4 %gl_in %c0 %c0
%p = OpLoad %v4float %p_ptr
%c_ptr = OpAccessChain %ptr_in_v3 %color %c0
%cval = OpLoad %v3float %c_ptr

%px = OpCompositeExtract %float %p 0
%py = OpCompositeExtract %float %p 1

%xmh = OpFAdd %float %px %negfh
%xph = OpFAdd %float %px %fh
%ymh = OpFAdd %float %py %negfh
%yph = OpFAdd %float %py %fh

%p0 = OpCompositeConstruct %v4float %xmh %ymh %f0 %f1
%p1 = OpCompositeConstruct %v4float %xph %ymh %f0 %f1
%p2 = OpCompositeConstruct %v4float %xmh %yph %f0 %f1
%p3 = OpCompositeConstruct %v4float %xph %yph %f0 %f1

%outp0 = OpAccessChain %ptr_out_v4 %gl_out %c0
OpStore %outp0 %p0
OpStore %outColor %cval
OpEmitVertex

%outp1 = OpAccessChain %ptr_out_v4 %gl_out %c0
OpStore %outp1 %p1
OpStore %outColor %cval
OpEmitVertex

%outp2 = OpAccessChain %ptr_out_v4 %gl_out %c0
OpStore %outp2 %p2
OpStore %outColor %cval
OpEmitVertex

%outp3 = OpAccessChain %ptr_out_v4 %gl_out %c0
OpStore %outp3 %p3
OpStore %outColor %cval
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
