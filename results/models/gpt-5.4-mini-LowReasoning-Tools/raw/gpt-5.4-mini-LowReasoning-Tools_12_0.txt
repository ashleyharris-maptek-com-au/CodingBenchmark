using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    private struct Mat3
    {
        public double M00, M01, M02;
        public double M10, M11, M12;
        public double M20, M21, M22;

        public static Vec3 Mul(in Mat3 m, in Vec3 v)
        {
            return new Vec3(
                m.M00 * v.X + m.M01 * v.Y + m.M02 * v.Z,
                m.M10 * v.X + m.M11 * v.Y + m.M12 * v.Z,
                m.M20 * v.X + m.M21 * v.Y + m.M22 * v.Z);
        }
    }

    private struct Quat
    {
        public double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
    }

    private struct Candidate
    {
        public Mat3 R;
        public Quat Q;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int vCount = fs.NextInt();
        int fCount = fs.NextInt();
        var vertices = new Vec3[vCount];
        for (int i = 0; i < vCount; i++)
            vertices[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        for (int i = 0; i < fCount; i++)
        {
            fs.NextInt();
            fs.NextInt();
            fs.NextInt();
        }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        if (vCount == 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var candidates = BuildCandidates();
        double containerVolume = W * H * D;
        bool small = containerVolume <= 1000.0 || Math.Min(W, Math.Min(H, D)) <= 12.0;

        Candidate best = candidates[0];
        double bestScore = double.NegativeInfinity;
        double bestDx = 0, bestDy = 0, bestDz = 0;
        int bestCount = 0;

        foreach (var cand in candidates)
        {
            GetBounds(vertices, cand.R, out double minX, out double minY, out double minZ, out double maxX, out double maxY, out double maxZ);
            double dx = maxX - minX;
            double dy = maxY - minY;
            double dz = maxZ - minZ;
            if (dx <= 1e-12 || dy <= 1e-12 || dz <= 1e-12) continue;

            int nx = (int)Math.Floor((W + 1e-9) / dx);
            int ny = (int)Math.Floor((H + 1e-9) / dy);
            int nz = (int)Math.Floor((D + 1e-9) / dz);
            int count = nx * ny * nz;
            if (count <= 0) continue;

            double fill = (count * dx * dy * dz) / containerVolume;
            double score = count * 1_000_000.0 + fill;
            if (small)
                score += 1.0 / (dx + dy + dz + 1e-9);

            if (score > bestScore)
            {
                bestScore = score;
                best = cand;
                bestDx = dx;
                bestDy = dy;
                bestDz = dz;
                bestCount = count;
            }
        }

        if (bestCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        GetBounds(vertices, best.R, out double minBx, out _, out double minBz, out _, out _, out _);
        // Recompute exact bounds for the chosen orientation so translations are stable.
        GetBounds(vertices, best.R, out minBx, out double minBy, out minBz, out double maxBx, out double maxBy, out double maxBz);

        int finalNx = (int)Math.Floor((W + 1e-9) / bestDx);
        int finalNy = (int)Math.Floor((H + 1e-9) / bestDy);
        int finalNz = (int)Math.Floor((D + 1e-9) / bestDz);
        int total = finalNx * finalNy * finalNz;

        Console.WriteLine(total);
        string qw = best.Q.W.ToString("R", CultureInfo.InvariantCulture);
        string qx = best.Q.X.ToString("R", CultureInfo.InvariantCulture);
        string qy = best.Q.Y.ToString("R", CultureInfo.InvariantCulture);
        string qz = best.Q.Z.ToString("R", CultureInfo.InvariantCulture);

        for (int z = 0; z < finalNz; z++)
        {
            for (int y = 0; y < finalNy; y++)
            {
                for (int x = 0; x < finalNx; x++)
                {
                    double tx = x * bestDx - minBx;
                    double ty = y * bestDy - minBy;
                    double tz = z * bestDz - minBz;
                    Console.WriteLine(
                        tx.ToString("R", CultureInfo.InvariantCulture) + " " +
                        ty.ToString("R", CultureInfo.InvariantCulture) + " " +
                        tz.ToString("R", CultureInfo.InvariantCulture) + " " +
                        qw + " " + qx + " " + qy + " " + qz);
                }
            }
        }
    }

    private static void GetBounds(Vec3[] vertices, in Mat3 r, out double minX, out double minY, out double minZ, out double maxX, out double maxY, out double maxZ)
    {
        minX = minY = minZ = double.PositiveInfinity;
        maxX = maxY = maxZ = double.NegativeInfinity;
        for (int i = 0; i < vertices.Length; i++)
        {
            Vec3 p = Mat3.Mul(in r, in vertices[i]);
            if (p.X < minX) minX = p.X;
            if (p.Y < minY) minY = p.Y;
            if (p.Z < minZ) minZ = p.Z;
            if (p.X > maxX) maxX = p.X;
            if (p.Y > maxY) maxY = p.Y;
            if (p.Z > maxZ) maxZ = p.Z;
        }
    }

    private static List<Candidate> BuildCandidates()
    {
        var result = new List<Candidate>(24);
        int[][] perms =
        {
            new[] { 0, 1, 2 },
            new[] { 0, 2, 1 },
            new[] { 1, 0, 2 },
            new[] { 1, 2, 0 },
            new[] { 2, 0, 1 },
            new[] { 2, 1, 0 }
        };

        foreach (var p in perms)
        {
            for (int sx = -1; sx <= 1; sx += 2)
            {
                for (int sy = -1; sy <= 1; sy += 2)
                {
                    int sz = DeterminantSign(p) * sx * sy;
                    var m = new Mat3();
                    SetColumn(ref m, 0, p[0], sx);
                    SetColumn(ref m, 1, p[1], sy);
                    SetColumn(ref m, 2, p[2], sz);
                    result.Add(new Candidate { R = m, Q = MatrixToQuat(in m) });
                }
            }
        }

        return result;
    }

    private static int DeterminantSign(int[] p)
    {
        int inversions = 0;
        for (int i = 0; i < 3; i++)
            for (int j = i + 1; j < 3; j++)
                if (p[i] > p[j]) inversions++;
        return (inversions & 1) == 0 ? 1 : -1;
    }

    private static void SetColumn(ref Mat3 m, int col, int axis, int sign)
    {
        double v = sign;
        if (axis == 0)
        {
            if (col == 0) m.M00 = v;
            else if (col == 1) m.M10 = v;
            else m.M20 = v;
        }
        else if (axis == 1)
        {
            if (col == 0) m.M01 = v;
            else if (col == 1) m.M11 = v;
            else m.M21 = v;
        }
        else
        {
            if (col == 0) m.M02 = v;
            else if (col == 1) m.M12 = v;
            else m.M22 = v;
        }
    }

    private static Quat MatrixToQuat(in Mat3 m)
    {
        double trace = m.M00 + m.M11 + m.M22;
        double w, x, y, z;
        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            w = 0.25 * s;
            x = (m.M21 - m.M12) / s;
            y = (m.M02 - m.M20) / s;
            z = (m.M10 - m.M01) / s;
        }
        else if (m.M00 > m.M11 && m.M00 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
            w = (m.M21 - m.M12) / s;
            x = 0.25 * s;
            y = (m.M01 + m.M10) / s;
            z = (m.M02 + m.M20) / s;
        }
        else if (m.M11 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
            w = (m.M02 - m.M20) / s;
            x = (m.M01 + m.M10) / s;
            y = 0.25 * s;
            z = (m.M12 + m.M21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
            w = (m.M10 - m.M01) / s;
            x = (m.M02 + m.M20) / s;
            y = (m.M12 + m.M21) / s;
            z = 0.25 * s;
        }

        double norm = Math.Sqrt(w * w + x * x + y * y + z * z);
        if (norm > 0)
        {
            w /= norm;
            x /= norm;
            y /= norm;
            z /= norm;
        }
        if (w < 0)
        {
            w = -w; x = -x; y = -y; z = -z;
        }
        return new Quat(w, x, y, z);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string Next()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            if (c < 0) return null;
            var chars = new List<char>();
            while (c > 32)
            {
                chars.Add((char)c);
                c = Read();
            }
            return new string(chars.ToArray());
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }
}
