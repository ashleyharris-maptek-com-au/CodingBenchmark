OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0_7 = OpConstant %float 0.7
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p1_v4 = OpLoad %v4float %p1_ptr
%p2_v4 = OpLoad %v4float %p2_ptr
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

%sum01 = OpFAdd %v3float %p0 %p1
%sum012 = OpFAdd %v3float %sum01 %p2
%centroid = OpFDiv %v3float %sum012 %float_3

%d0 = OpFSub %v3float %p0 %centroid
%d1 = OpFSub %v3float %p1 %centroid
%d2 = OpFSub %v3float %p2 %centroid

%sd0 = OpFMul %v3float %d0 %float_0_7
%sd1 = OpFMul %v3float %d1 %float_0_7
%sd2 = OpFMul %v3float %d2 %float_0_7

%n0 = OpFAdd %v3float %centroid %sd0
%n1 = OpFAdd %v3float %centroid %sd1
%n2 = OpFAdd %v3float %centroid %sd2

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%n0x = OpCompositeExtract %float %n0 0
%n0y = OpCompositeExtract %float %n0 1
%n0z = OpCompositeExtract %float %n0 2
%n1x = OpCompositeExtract %float %n1 0
%n1y = OpCompositeExtract %float %n1 1
%n1z = OpCompositeExtract %float %n1 2
%n2x = OpCompositeExtract %float %n2 0
%n2y = OpCompositeExtract %float %n2 1
%n2z = OpCompositeExtract %float %n2 2

%pos0 = OpCompositeConstruct %v4float %n0x %n0y %n0z %float_1
%pos1 = OpCompositeConstruct %v4float %n1x %n1y %n1z %float_1
%pos2 = OpCompositeConstruct %v4float %n2x %n2y %n2z %float_1

%outpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr %pos0
OpStore %outColor %c0
OpEmitVertex

OpStore %outpos_ptr %pos1
OpStore %outColor %c1
OpEmitVertex

OpStore %outpos_ptr %pos2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
