static constexpr double EPS = 1e-9;

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}
static inline double norm2(const Vec3& a) { return dot(a, a); }

struct Plane {
    Vec3 n;
    double w{};
    Plane() = default;
    Plane(const Vec3& a, const Vec3& b, const Vec3& c) {
        n = cross(b - a, c - a);
        double len = std::sqrt(norm2(n));
        if (len > 0) n = n * (1.0 / len);
        w = dot(n, a);
    }
    int classifyPoint(const Vec3& p) const {
        double t = dot(n, p) - w;
        if (t < -EPS) return -1;
        if (t > EPS) return 1;
        return 0;
    }
    void flip() {
        n = n * -1.0;
        w = -w;
    }
};

struct Vertex {
    Vec3 p;
    Vertex() = default;
    explicit Vertex(const Vec3& p_) : p(p_) {}
    Vertex lerp(const Vertex& other, double t) const {
        return Vertex(p + (other.p - p) * t);
    }
};

struct Polygon {
    vector<Vertex> verts;
    Plane plane;
    Polygon() = default;
    explicit Polygon(vector<Vertex> v) : verts(std::move(v)), plane(verts[0].p, verts[1].p, verts[2].p) {}
    void flip() {
        std::reverse(verts.begin(), verts.end());
        plane.flip();
    }
};

struct Node {
    Plane plane;
    bool hasPlane = false;
    vector<Polygon> polygons;
    Node* front = nullptr;
    Node* back = nullptr;

    Node() = default;
    explicit Node(const vector<Polygon>& polys) { build(polys); }
    ~Node() { delete front; delete back; }

    void splitPolygon(const Polygon& poly, vector<Polygon>& coplanarFront, vector<Polygon>& coplanarBack, vector<Polygon>& frontList, vector<Polygon>& backList) const {
        if (!hasPlane) {
            frontList.push_back(poly);
            return;
        }
        int polygonType = 0;
        vector<int> types(poly.verts.size());
        for (size_t i = 0; i < poly.verts.size(); ++i) {
            int t = plane.classifyPoint(poly.verts[i].p);
            types[i] = t;
            polygonType |= (t > 0 ? 1 : 0);
            polygonType |= (t < 0 ? 2 : 0);
        }

        if (polygonType == 0) {
            if (dot(plane.n, poly.plane.n) > 0) coplanarFront.push_back(poly);
            else coplanarBack.push_back(poly);
        } else if (polygonType == 1) {
            frontList.push_back(poly);
        } else if (polygonType == 2) {
            backList.push_back(poly);
        } else {
            vector<Vertex> f, b;
            for (size_t i = 0; i < poly.verts.size(); ++i) {
                size_t j = (i + 1) % poly.verts.size();
                int ti = types[i], tj = types[j];
                const Vertex& vi = poly.verts[i];
                const Vertex& vj = poly.verts[j];
                if (ti >= 0) f.push_back(vi);
                if (ti <= 0) b.push_back(vi);
                if ((ti | tj) == 0 || ti == tj) continue;
                double denom = dot(plane.n, vj.p - vi.p);
                double t = 0.0;
                if (std::fabs(denom) > EPS) t = (plane.w - dot(plane.n, vi.p)) / denom;
                Vertex v = vi.lerp(vj, t);
                f.push_back(v);
                b.push_back(v);
            }
            if (f.size() >= 3) frontList.emplace_back(f);
            if (b.size() >= 3) backList.emplace_back(b);
        }
    }

    void clipPolygons(const vector<Polygon>& polys, vector<Polygon>& out) const {
        if (!hasPlane) {
            out.insert(out.end(), polys.begin(), polys.end());
            return;
        }
        vector<Polygon> frontList, backList;
        frontList.reserve(polys.size());
        backList.reserve(polys.size());
        vector<Polygon> coplanarFront, coplanarBack;
        for (const auto& p : polys) splitPolygon(p, coplanarFront, coplanarBack, frontList, backList);
        out.insert(out.end(), coplanarFront.begin(), coplanarFront.end());
        if (front) front->clipPolygons(frontList, out); else out.insert(out.end(), frontList.begin(), frontList.end());
        if (back) back->clipPolygons(backList, out);
    }

    void clipTo(const Node& other) {
        vector<Polygon> clipped;
        other.clipPolygons(polygons, clipped);
        polygons.swap(clipped);
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }

    vector<Polygon> allPolygons() const {
        vector<Polygon> res = polygons;
        if (front) {
            auto t = front->allPolygons();
            res.insert(res.end(), t.begin(), t.end());
        }
        if (back) {
            auto t = back->allPolygons();
            res.insert(res.end(), t.begin(), t.end());
        }
        return res;
    }

    void invert() {
        for (auto& p : polygons) p.flip();
        if (hasPlane) plane.flip();
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    void build(const vector<Polygon>& polys) {
        if (polys.empty()) return;
        if (!hasPlane) {
            plane = polys[0].plane;
            hasPlane = true;
        }
        vector<Polygon> frontList, backList;
        vector<Polygon> coplanarFront, coplanarBack;
        for (const auto& p : polys) splitPolygon(p, coplanarFront, coplanarBack, frontList, backList);
        polygons.insert(polygons.end(), coplanarFront.begin(), coplanarFront.end());
        polygons.insert(polygons.end(), coplanarBack.begin(), coplanarBack.end());
        if (!frontList.empty()) {
            if (!front) front = new Node();
            front->build(frontList);
        }
        if (!backList.empty()) {
            if (!back) back = new Node();
            back->build(backList);
        }
    }
};

static vector<Polygon> makePolygons(const vector<Vec3>& verts, const vector<array<int,3>>& faces) {
    vector<Polygon> polys;
    polys.reserve(faces.size());
    for (const auto& f : faces) {
        vector<Vertex> v;
        v.reserve(3);
        v.emplace_back(verts[f[0]]);
        v.emplace_back(verts[f[1]]);
        v.emplace_back(verts[f[2]]);
        polys.emplace_back(std::move(v));
    }
    return polys;
}

struct Key {
    long long x, y, z;
    bool operator==(const Key& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct KeyHash {
    size_t operator()(const Key& k) const noexcept {
        size_t h = std::hash<long long>{}(k.x);
        h ^= std::hash<long long>{}(k.y + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2));
        h ^= std::hash<long long>{}(k.z + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2));
        return h;
    }
};

static Key quantize(const Vec3& p) {
    const double s = 1e9;
    return { (long long)std::llround(p.x * s), (long long)std::llround(p.y * s), (long long)std::llround(p.z * s) };
}

static bool bboxDisjoint(const vector<Vec3>& a, const vector<Vec3>& b) {
    if (a.empty() || b.empty()) return true;
    Vec3 amin{std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity()};
    Vec3 amax{-amin.x, -amin.y, -amin.z};
    Vec3 bmin = amin, bmax = amax;
    for (auto& p : a) {
        amin.x = std::min(amin.x, p.x); amin.y = std::min(amin.y, p.y); amin.z = std::min(amin.z, p.z);
        amax.x = std::max(amax.x, p.x); amax.y = std::max(amax.y, p.y); amax.z = std::max(amax.z, p.z);
    }
    for (auto& p : b) {
        bmin.x = std::min(bmin.x, p.x); bmin.y = std::min(bmin.y, p.y); bmin.z = std::min(bmin.z, p.z);
        bmax.x = std::max(bmax.x, p.x); bmax.y = std::max(bmax.y, p.y); bmax.z = std::max(bmax.z, p.z);
    }
    return amax.x < bmin.x - EPS || bmax.x < amin.x - EPS ||
           amax.y < bmin.y - EPS || bmax.y < amin.y - EPS ||
           amax.z < bmin.z - EPS || bmax.z < amin.z - EPS;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    vector<Vec3> A(VA), B(VB);
    for (int i = 0; i < VA; ++i) cin >> A[i].x >> A[i].y >> A[i].z;
    vector<array<int,3>> FAi(FA);
    for (int i = 0; i < FA; ++i) cin >> FAi[i][0] >> FAi[i][1] >> FAi[i][2];
    for (int i = 0; i < VB; ++i) cin >> B[i].x >> B[i].y >> B[i].z;
    vector<array<int,3>> FBi(FB);
    for (int i = 0; i < FB; ++i) cin >> FBi[i][0] >> FBi[i][1] >> FBi[i][2];

    if (bboxDisjoint(A, B)) {
        vector<Vec3> outV = A;
        outV.insert(outV.end(), B.begin(), B.end());
        vector<array<int,3>> outF = FAi;
        for (auto f : FBi) outF.push_back({f[0] + VA, f[1] + VA, f[2] + VA});
        cout << outV.size() << ' ' << outF.size() << '\n';
        for (auto& p : outV) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (auto& f : outF) cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
        return 0;
    }

    auto polysA = makePolygons(A, FAi);
    auto polysB = makePolygons(B, FBi);
    Node a(polysA), b(polysB);
    a.clipTo(b);
    b.clipTo(a);
    b.invert();
    b.clipTo(a);
    b.invert();
    a.build(b.allPolygons());

    auto polys = a.allPolygons();
    vector<Vec3> outV;
    vector<array<int,3>> outF;
    std::unordered_map<Key, int, KeyHash> idx;
    idx.reserve(polys.size() * 3);

    auto getIndex = [&](const Vec3& p) {
        Key k = quantize(p);
        auto it = idx.find(k);
        if (it != idx.end()) return it->second;
        int id = (int)outV.size();
        idx.emplace(k, id);
        outV.push_back(p);
        return id;
    };

    for (const auto& poly : polys) {
        if (poly.verts.size() < 3) continue;
        int v0 = getIndex(poly.verts[0].p);
        for (size_t i = 1; i + 1 < poly.verts.size(); ++i) {
            int v1 = getIndex(poly.verts[i].p);
            int v2 = getIndex(poly.verts[i + 1].p);
            if (v0 != v1 && v1 != v2 && v2 != v0) outF.push_back({v0, v1, v2});
        }
    }

    cout << outV.size() << ' ' << outF.size() << '\n';
    cout.setf(std::ios::fixed); cout.precision(12);
    for (auto& p : outV) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (auto& f : outF) cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    return 0;
}
