; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %ubo %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Uniform_v3float = OpTypePointer Uniform %v3float
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_02 = OpConstant %float 0.2
%float_05 = OpConstant %float 0.5
%float_08 = OpConstant %float 0.8
%float_32 = OpConstant %float 32.0
%v3float_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v4float_1 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1
%v4float_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%v3float_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%glsl = OpExtInstImport "GLSL.std.450"
%main = OpFunction %void None %funct
%label = OpLabel
%wp = OpLoad %v3float %worldPos
%n_in = OpLoad %v3float %normal
%n = OpExtInst %v3float %glsl Normalize %n_in
%light_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %light_ptr
%cam_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cam4 = OpLoad %v4float %cam_ptr
%lightx = OpCompositeExtract %float %light4 0
%lighty = OpCompositeExtract %float %light4 1
%lightz = OpCompositeExtract %float %light4 2
%camx = OpCompositeExtract %float %cam4 0
%camy = OpCompositeExtract %float %cam4 1
%camz = OpCompositeExtract %float %cam4 2
%light3 = OpCompositeConstruct %v3float %lightx %lighty %lightz
%cam3 = OpCompositeConstruct %v3float %camx %camy %camz
%lvec = OpFSub %v3float %light3 %wp
%vvec = OpFSub %v3float %cam3 %wp
%L = OpExtInst %v3float %glsl Normalize %lvec
%V = OpExtInst %v3float %glsl Normalize %vvec
%ndotl = OpDot %float %n %L
%diffuse = OpExtInst %float %glsl FMax %ndotl %float_0
%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl Reflect %negL %n
%rdotv = OpDot %float %R %V
%spec_base = OpExtInst %float %glsl FMax %rdotv %float_0
%specular = OpExtInst %float %glsl Pow %spec_base %float_32
%diff_term = OpFMul %float %float_05 %diffuse
%spec_term = OpFMul %float %float_08 %specular
%sum1 = OpFAdd %float %float_02 %diff_term
%intensity = OpFAdd %float %sum1 %spec_term
%rgb = OpCompositeConstruct %v3float %intensity %intensity %intensity
%color = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
