use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: usize,
    y: usize,
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: i32 = it.next().unwrap().parse().unwrap();

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[idx(x, y, w)] = true;
        }
    }

    if w == 0 || h == 0 || blocked[idx(0, 0, w)] {
        println!("0");
        return;
    }

    let total_free = blocked.iter().filter(|&&b| !b).count();
    if total_free == 0 {
        println!("0");
        return;
    }
    if require_cycle == 1 && total_free % 2 == 1 {
        println!("0");
        return;
    }

    if let Some(path) = solve(w, h, &blocked, require_cycle == 1) {
        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }
}

fn solve(w: usize, h: usize, blocked: &[bool], want_cycle: bool) -> Option<Vec<Pt>> {
    let n = w * h;
    let start = idx(0, 0, w);
    let free_count = blocked.iter().filter(|&&b| !b).count();
    let mut visited = vec![false; n];
    let mut path = Vec::with_capacity(free_count);
    visited[start] = true;
    path.push(Pt { x: 0, y: 0 });

    // Fast path for obstacle-free grids.
    if blocked.iter().all(|&b| !b) {
        return construct_empty_grid(w, h, want_cycle);
    }

    let mut neigh = vec![Vec::<usize>::new(); n];
    for y in 0..h {
        for x in 0..w {
            let i = idx(x, y, w);
            if blocked[i] {
                continue;
            }
            if x > 0 && !blocked[idx(x - 1, y, w)] {
                neigh[i].push(idx(x - 1, y, w));
            }
            if x + 1 < w && !blocked[idx(x + 1, y, w)] {
                neigh[i].push(idx(x + 1, y, w));
            }
            if y > 0 && !blocked[idx(x, y - 1, w)] {
                neigh[i].push(idx(x, y - 1, w));
            }
            if y + 1 < h && !blocked[idx(x, y + 1, w)] {
                neigh[i].push(idx(x, y + 1, w));
            }
        }
    }

    if !connectivity_ok(start, blocked, &visited, &neigh) {
        return None;
    }

    if dfs(w, h, blocked, &neigh, &mut visited, &mut path, start, want_cycle, free_count) {
        Some(path)
    } else {
        None
    }
}

fn construct_empty_grid(w: usize, h: usize, want_cycle: bool) -> Option<Vec<Pt>> {
    let n = w * h;
    let mut path = Vec::with_capacity(n);
    if !want_cycle {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push(Pt { x, y });
                }
            } else {
                for x in (0..w).rev() {
                    path.push(Pt { x, y });
                }
            }
        }
        return Some(path);
    }

    if n % 2 == 1 {
        return None;
    }
    if w == 1 || h == 1 {
        return None;
    }

    // Simple Hamiltonian cycle for rectangles with at least 2x2 and even area.
    if w % 2 == 0 {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push(Pt { x, y });
                }
            } else {
                for x in (0..w).rev() {
                    path.push(Pt { x, y });
                }
            }
        }
    } else {
        for x in 0..w {
            if x % 2 == 0 {
                for y in 0..h {
                    path.push(Pt { x, y });
                }
            } else {
                for y in (0..h).rev() {
                    path.push(Pt { x, y });
                }
            }
        }
    }
    Some(path)
}

fn dfs(
    w: usize,
    h: usize,
    blocked: &[bool],
    neigh: &[Vec<usize>],
    visited: &mut [bool],
    path: &mut Vec<Pt>,
    start: usize,
    want_cycle: bool,
    free_count: usize,
) -> bool {
    if path.len() == free_count {
        if !want_cycle {
            return true;
        }
        let cur = idx(path.last().unwrap().x, path.last().unwrap().y, w);
        return neigh[cur].iter().any(|&u| u == start);
    }

    if path.len() == 1 {
        if !connectivity_ok(start, blocked, visited, neigh) {
            return false;
        }
    }

    let cur = idx(path.last().unwrap().x, path.last().unwrap().y, w);
    let mut cand = Vec::new();
    for &nxt in &neigh[cur] {
        if !visited[nxt] {
            cand.push(nxt);
        }
    }
    if cand.is_empty() {
        return false;
    }

    cand.sort_by(|&a, &b| {
        let da = onward_degree(a, visited, neigh);
        let db = onward_degree(b, visited, neigh);
        match da.cmp(&db) {
            Ordering::Equal => a.cmp(&b),
            other => other,
        }
    });

    for nxt in cand {
        visited[nxt] = true;
        path.push(Pt { x: nxt % w, y: nxt / w });

        let ok = local_prune(nxt, neigh, visited)
            && connectivity_ok(nxt, blocked, visited, neigh);

        if ok && dfs(w, h, blocked, neigh, visited, path, start, want_cycle, free_count) {
            return true;
        }

        path.pop();
        visited[nxt] = false;
    }
    false
}

fn onward_degree(v: usize, visited: &[bool], neigh: &[Vec<usize>]) -> usize {
    neigh[v].iter().filter(|&&u| !visited[u]).count()
}

fn local_prune(cur: usize, neigh: &[Vec<usize>], visited: &[bool]) -> bool {
    for i in 0..neigh.len() {
        if visited[i] {
            continue;
        }
        let d = neigh[i].iter().filter(|&&u| !visited[u] || u == cur).count();
        if d == 0 {
            return false;
        }
    }
    true
}

fn connectivity_ok(start: usize, blocked: &[bool], visited: &[bool], neigh: &[Vec<usize>]) -> bool {
    let mut q = VecDeque::new();
    let mut seen = vec![false; visited.len()];
    let mut seen_count = 0usize;

    q.push_back(start);
    seen[start] = true;

    while let Some(v) = q.pop_front() {
        if !blocked[v] && !visited[v] {
            seen_count += 1;
        }
        for &u in &neigh[v] {
            if !seen[u] && !visited[u] {
                seen[u] = true;
                q.push_back(u);
            }
        }
    }

    let remaining = visited
        .iter()
        .enumerate()
        .filter(|&(i, v)| !blocked[i] && !*v)
        .count();
    seen_count == remaining
}
