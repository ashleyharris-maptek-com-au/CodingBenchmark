use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Eq, PartialEq)]
struct Item {
    score: usize,
    degree: usize,
    v: usize,
}

impl Ord for Item {
    fn cmp(&self, other: &Self) -> Ordering {
        self.score
            .cmp(&other.score)
            .then_with(|| self.degree.cmp(&other.degree))
            .then_with(|| other.v.cmp(&self.v))
    }
}

impl PartialOrd for Item {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    let mut dominated = vec![false; n];
    let mut chosen = vec![false; n];
    let mut ans = Vec::<usize>::new();
    let mut heap = BinaryHeap::<Item>::new();

    for v in 0..n {
        heap.push(Item {
            score: adj[v].len() + 1,
            degree: adj[v].len(),
            v,
        });
    }

    let mut dominated_cnt = 0usize;

    let mut mark = |start: usize,
                    dominated: &mut Vec<bool>,
                    dominated_cnt: &mut usize,
                    adj: &Vec<Vec<usize>>| {
        if !dominated[start] {
            dominated[start] = true;
            *dominated_cnt += 1;
        }
        for &to in &adj[start] {
            if !dominated[to] {
                dominated[to] = true;
                *dominated_cnt += 1;
            }
        }
    };

    while dominated_cnt < n {
        let mut best = None;
        while let Some(item) = heap.pop() {
            if chosen[item.v] {
                continue;
            }
            let mut score = if dominated[item.v] { 0 } else { 1 };
            for &to in &adj[item.v] {
                if !dominated[to] {
                    score += 1;
                }
            }
            if score == 0 {
                continue;
            }
            best = Some(Item {
                score,
                degree: item.degree,
                v: item.v,
            });
            break;
        }

        let item = if let Some(x) = best {
            x
        } else {
            let mut v = 0usize;
            while v < n && dominated[v] {
                v += 1;
            }
            if v == n {
                break;
            }
            Item {
                score: 1,
                degree: adj[v].len(),
                v,
            }
        };

        chosen[item.v] = true;
        ans.push(item.v);
        mark(item.v, &mut dominated, &mut dominated_cnt, &adj);
    }

    if dominated_cnt < n {
        for v in 0..n {
            if !dominated[v] {
                chosen[v] = true;
                ans.push(v);
                mark(v, &mut dominated, &mut dominated_cnt, &adj);
            }
        }
    }

    let mut cover = vec![0usize; n];
    for &v in &ans {
        cover[v] += 1;
        for &to in &adj[v] {
            cover[to] += 1;
        }
    }

    let mut kept = vec![true; n];
    for &v in ans.iter().rev() {
        let mut removable = true;
        if cover[v] <= 1 {
            removable = false;
        } else {
            for &to in &adj[v] {
                if cover[to] <= 1 {
                    removable = false;
                    break;
                }
            }
        }
        if removable {
            kept[v] = false;
            cover[v] -= 1;
            for &to in &adj[v] {
                cover[to] -= 1;
            }
        }
    }

    let mut out = Vec::new();
    let mut final_set = Vec::new();
    for &v in &ans {
        if kept[v] {
            final_set.push(v);
        }
    }
    out.push(final_set.len().to_string());
    if final_set.is_empty() {
        out.push(String::new());
    } else {
        out.push(final_set.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" "));
    }
    println!("{}", out.join("\n"));
}
