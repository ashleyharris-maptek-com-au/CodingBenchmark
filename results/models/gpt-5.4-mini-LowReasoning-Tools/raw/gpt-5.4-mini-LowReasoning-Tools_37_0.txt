using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    private struct Edge
    {
        public int U;
        public int V;
        public Edge(int u, int v) { U = u; V = v; }
    }

    private static int N, M;
    private static Edge[] Edges = Array.Empty<Edge>();
    private static List<int>[] Out = Array.Empty<List<int>>();
    private static List<int>[] In = Array.Empty<List<int>>();

    private static bool[] BestRemoved = Array.Empty<bool>();
    private static int BestSize = int.MaxValue;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = fs.NextInt();
        M = fs.NextInt();
        if (N < 0) return;

        Edges = new Edge[M];
        Out = new List<int>[N];
        In = new List<int>[N];
        for (int i = 0; i < N; i++)
        {
            Out[i] = new List<int>();
            In[i] = new List<int>();
        }

        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            Edges[i] = new Edge(u, v);
            Out[u].Add(v);
            In[v].Add(u);
        }

        bool[] removed = new bool[N];
        BestRemoved = new bool[N];
        BestSize = N + 1;

        // Greedy initial upper bound to prune the exact search.
        var greedy = new bool[N];
        int greedySize = GreedyFvs(greedy);
        if (greedySize < BestSize)
        {
            BestSize = greedySize;
            Array.Copy(greedy, BestRemoved, N);
        }

        Search(removed, 0);

        var answer = new List<int>();
        for (int i = 0; i < N; i++) if (BestRemoved[i]) answer.Add(i);

        using var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(answer.Count);
        if (answer.Count > 0)
        {
            for (int i = 0; i < answer.Count; i++)
            {
                if (i > 0) sw.Write(' ');
                sw.Write(answer[i]);
            }
            sw.WriteLine();
        }
        else
        {
            sw.WriteLine();
        }
    }

    private static void Search(bool[] removed, int removedCount)
    {
        if (removedCount >= BestSize) return;

        if (IsAcyclic(removed))
        {
            BestSize = removedCount;
            Array.Copy(removed, BestRemoved, N);
            return;
        }

        if (!FindCycle(removed, out var cycle))
        {
            BestSize = removedCount;
            Array.Copy(removed, BestRemoved, N);
            return;
        }

        // Branch on vertices from a cycle. Try high-impact vertices first.
        cycle.Sort((a, b) => DegreeScore(b).CompareTo(DegreeScore(a)));
        foreach (int v in cycle)
        {
            if (removed[v]) continue;
            removed[v] = true;
            Search(removed, removedCount + 1);
            removed[v] = false;
            if (removedCount + 1 >= BestSize) return;
        }
    }

    private static int DegreeScore(int v) => Out[v].Count + In[v].Count;

    private static int LowerBound(bool[] removed)
    {
        return 0;
    }

    private static bool IsAcyclic(bool[] removed)
    {
        int[] indeg = new int[N];
        for (int i = 0; i < M; i++)
        {
            var e = Edges[i];
            if (removed[e.U] || removed[e.V]) continue;
            indeg[e.V]++;
        }

        int[] q = new int[N];
        int head = 0, tail = 0;
        int alive = 0;
        for (int i = 0; i < N; i++)
        {
            if (removed[i]) continue;
            alive++;
            if (indeg[i] == 0) q[tail++] = i;
        }

        int processed = 0;
        while (head < tail)
        {
            int u = q[head++];
            processed++;
            foreach (int v in Out[u])
            {
                if (removed[v]) continue;
                if (--indeg[v] == 0) q[tail++] = v;
            }
        }
        return processed == alive;
    }

    private static bool FindCycle(bool[] removed, out List<int> cycle)
    {
        cycle = new List<int>();
        byte[] state = new byte[N]; // 0 unvisited, 1 on stack, 2 done
        int[] parent = new int[N];
        Array.Fill(parent, -1);
        var stack = new Stack<(int U, int NextIdx)>();

        for (int s = 0; s < N; s++)
        {
            if (removed[s] || state[s] != 0) continue;
            stack.Push((s, 0));
            state[s] = 1;
            parent[s] = -1;

            while (stack.Count > 0)
            {
                var (u, idx) = stack.Pop();
                if (idx < Out[u].Count)
                {
                    int v = Out[u][idx];
                    stack.Push((u, idx + 1));
                    if (removed[v]) continue;
                    if (state[v] == 0)
                    {
                        state[v] = 1;
                        parent[v] = u;
                        stack.Push((v, 0));
                    }
                    else if (state[v] == 1)
                    {
                        var path = new List<int>();
                        path.Add(v);
                        int x = u;
                        while (x != v && x != -1)
                        {
                            path.Add(x);
                            x = parent[x];
                        }
                        if (x != v) continue;
                        cycle = path;
                        return true;
                    }
                }
                else
                {
                    state[u] = 2;
                }
            }
        }
        return false;
    }

    private static int GreedyFvs(bool[] removed)
    {
        int count = 0;
        while (!IsAcyclic(removed))
        {
            if (!FindCycle(removed, out var cycle)) break;
            int best = cycle[0];
            int bestScore = DegreeScore(best);
            for (int i = 1; i < cycle.Count; i++)
            {
                int v = cycle[i];
                int score = DegreeScore(v);
                if (score > bestScore)
                {
                    bestScore = score;
                    best = v;
                }
            }
            if (removed[best]) break;
            removed[best] = true;
            count++;
        }
        return count;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
