using System;
using System.IO;
using System.Text;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - (byte)'0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        int arcCount = m * 2;
        int[] head = new int[n];
        Array.Fill(head, -1);
        int[] to = new int[arcCount];
        int[] next = new int[arcCount];
        bool[] used = new bool[arcCount];

        int arc = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            fs.NextInt(); // weight is irrelevant for a valid route

            to[arc] = v;
            next[arc] = head[u];
            head[u] = arc++;

            to[arc] = u;
            next[arc] = head[v];
            head[v] = arc++;
        }

        int[] stack = new int[arcCount + 1];
        int[] iter = new int[n];
        Array.Copy(head, iter, n);

        int sp = 0;
        stack[sp++] = 0;

        int[] route = new int[arcCount + 1];
        int rp = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = iter[v];
            while (e != -1 && used[e])
                e = next[e];
            iter[v] = e;

            if (e == -1)
            {
                route[rp++] = v;
                sp--;
            }
            else
            {
                used[e] = true;
                stack[sp++] = to[e];
                iter[v] = next[e];
            }
        }

        var sb = new StringBuilder(route.Length * 2);
        for (int i = rp - 1; i >= 0; i--)
        {
            if (sb.Length > 0) sb.Append(' ');
            sb.Append(route[i]);
        }

        Console.WriteLine(sb.ToString());
    }
}
