#include <algorithm>
#include <cstdint>
#include <iostream>
#include <string>
#include <unordered_set>
#include <vector>

using namespace std;

struct HashPair {
    uint64_t a;
    uint64_t b;
    bool operator==(const HashPair& other) const noexcept {
        return a == other.a && b == other.b;
    }
};

struct HashPairHasher {
    size_t operator()(const HashPair& h) const noexcept {
        uint64_t x = h.a ^ (h.b + 0x9e3779b97f4a7c15ULL + (h.a << 6) + (h.a >> 2));
        x ^= x >> 33;
        x *= 0xff51afd7ed558ccdULL;
        x ^= x >> 33;
        x *= 0xc4ceb9fe1a85ec53ULL;
        x ^= x >> 33;
        return static_cast<size_t>(x);
    }
};

static constexpr uint64_t BASE1 = 1315423911ULL;
static constexpr uint64_t BASE2 = 11400714819323198485ULL;

static inline HashPair roll_hash_window(const string& s, size_t len) {
    HashPair h{0, 0};
    for (size_t i = 0; i < len; ++i) {
        uint64_t c = static_cast<unsigned char>(s[i]) + 1;
        h.a = h.a * BASE1 + c;
        h.b = h.b * BASE2 + c;
    }
    return h;
}

static inline void roll_extend(HashPair& h, unsigned char add, unsigned char rem, uint64_t pow1, uint64_t pow2) {
    h.a = h.a * BASE1 + (static_cast<uint64_t>(add) + 1);
    h.a -= pow1 * (static_cast<uint64_t>(rem) + 1);
    h.b = h.b * BASE2 + (static_cast<uint64_t>(add) + 1);
    h.b -= pow2 * (static_cast<uint64_t>(rem) + 1);
}

static bool verify_substring(const vector<string>& strings, const string& cand) {
    for (const auto& s : strings) {
        if (s.find(cand) == string::npos) return false;
    }
    return true;
}

static bool locate_and_verify(const vector<string>& strings, const string& base, size_t len,
                              const unordered_set<HashPair, HashPairHasher>& current, string& found) {
    if (len == 0) {
        found.clear();
        return true;
    }
    if (base.size() < len) return false;

    uint64_t pow1 = 1, pow2 = 1;
    for (size_t i = 0; i < len; ++i) {
        pow1 *= BASE1;
        pow2 *= BASE2;
    }

    HashPair h = roll_hash_window(base, len);
    if (current.find(h) != current.end()) {
        string cand = base.substr(0, len);
        if (verify_substring(strings, cand)) {
            found = move(cand);
            return true;
        }
    }
    for (size_t pos = len; pos < base.size(); ++pos) {
        roll_extend(h, static_cast<unsigned char>(base[pos]), static_cast<unsigned char>(base[pos - len]), pow1, pow2);
        if (current.find(h) != current.end()) {
            string cand = base.substr(pos - len + 1, len);
            if (verify_substring(strings, cand)) {
                found = move(cand);
                return true;
            }
        }
    }
    return false;
}

static bool check_length(const vector<string>& strings, const string& base, size_t len, string& found) {
    if (len == 0) {
        found.clear();
        return true;
    }
    if (base.size() < len) return false;

    unordered_set<HashPair, HashPairHasher> current;
    current.reserve(base.size() - len + 1);

    uint64_t pow1 = 1, pow2 = 1;
    for (size_t i = 0; i < len; ++i) {
        pow1 *= BASE1;
        pow2 *= BASE2;
    }

    HashPair h = roll_hash_window(base, len);
    current.insert(h);
    for (size_t i = len; i < base.size(); ++i) {
        roll_extend(h, static_cast<unsigned char>(base[i]), static_cast<unsigned char>(base[i - len]), pow1, pow2);
        current.insert(h);
    }

    for (const auto& s : strings) {
        if (&s == &base) continue;
        if (current.empty()) return false;
        if (s.size() < len) return false;

        unordered_set<HashPair, HashPairHasher> seen;
        seen.reserve(min(current.size() * 2 + 1, s.size() - len + 1));

        uint64_t pow1 = 1, pow2 = 1;
        for (size_t i = 0; i < len; ++i) {
            pow1 *= BASE1;
            pow2 *= BASE2;
        }

        HashPair h = roll_hash_window(s, len);
        if (current.find(h) != current.end()) seen.insert(h);
        for (size_t i = len; i < s.size(); ++i) {
            roll_extend(h, static_cast<unsigned char>(s[i]), static_cast<unsigned char>(s[i - len]), pow1, pow2);
            if (current.find(h) != current.end()) seen.insert(h);
        }

        if (seen.empty()) return false;

        unordered_set<HashPair, HashPairHasher> next;
        next.reserve(seen.size() * 2 + 1);
        for (const auto& h2 : current) {
            if (seen.find(h2) != seen.end()) next.insert(h2);
        }
        current.swap(next);
    }

    return locate_and_verify(strings, base, len, current, found);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> strings;
    strings.reserve(max(0, N));
    for (int i = 0; i < N; ++i) {
        string s;
        getline(cin, s);
        strings.push_back(move(s));
    }

    if (strings.empty()) {
        cout << '\n';
        return 0;
    }

    size_t min_idx = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].size() < strings[min_idx].size()) min_idx = i;
    }
    if (strings[min_idx].empty()) {
        cout << '\n';
        return 0;
    }
    if (min_idx != 0) swap(strings[0], strings[min_idx]);

    const string& base = strings[0];
    size_t lo = 0, hi = base.size();
    string best;

    while (lo < hi) {
        size_t mid = lo + (hi - lo + 1) / 2;
        string candidate;
        if (check_length(strings, base, mid, candidate)) {
            lo = mid;
            best = move(candidate);
        } else {
            hi = mid - 1;
        }
    }

    if (lo == 0) {
        cout << '\n';
        return 0;
    }

    if (best.empty()) {
        string candidate;
        check_length(strings, base, lo, candidate);
        best = move(candidate);
    }

    cout << best << '\n';
    return 0;
}
