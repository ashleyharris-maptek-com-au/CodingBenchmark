using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private sealed class Constraint
    {
        public int MineCount;
        public List<int> Cells = new List<int>();
    }

    private sealed class Component
    {
        public List<int> Cells = new List<int>();
        public List<Constraint> Constraints = new List<Constraint>();
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        int h = input.NextInt();
        int w = input.NextInt();
        if (h <= 0 || w <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        string[] rows = new string[h];
        for (int i = 0; i < h; i++)
        {
            rows[i] = input.NextBoardLine(w);
        }

        var safe = Solve(rows, h, w);
        var ordered = safe.OrderBy(p => p.row).ThenBy(p => p.col).ToList();

        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(ordered.Count);
        foreach (var p in ordered)
        {
            writer.Write(p.col);
            writer.Write(' ');
            writer.WriteLine(p.row);
        }
        writer.Flush();
    }

    private static List<(int col, int row)> Solve(string[] rows, int h, int w)
    {
        var safe = new bool[h * w];
        var isUnknown = new bool[h * w];
        var isNumber = new bool[h * w];
        var isFlag = new bool[h * w];

        for (int r = 0; r < h; r++)
        {
            string row = rows[r];
            for (int c = 0; c < w; c++)
            {
                int idx = r * w + c;
                char ch = row[c];
                if (ch == ' ')
                {
                    isUnknown[idx] = true;
                }
                else if (ch == '*')
                {
                    isFlag[idx] = true;
                }
                else if (ch >= '0' && ch <= '8')
                {
                    isNumber[idx] = true;
                }
            }
        }

        var numberToConstraint = new Dictionary<int, Constraint>();
        var cellToConstraints = new List<int>[h * w];
        for (int i = 0; i < cellToConstraints.Length; i++)
            cellToConstraints[i] = new List<int>();

        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                int idx = r * w + c;
                if (!isNumber[idx]) continue;

                int need = rows[r][c] - '0';
                var cells = new List<int>(8);
                for (int k = 0; k < 8; k++)
                {
                    int nr = r + dr[k];
                    int nc = c + dc[k];
                    if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                    int nidx = nr * w + nc;
                    if (isFlag[nidx]) need--;
                    else if (isUnknown[nidx]) cells.Add(nidx);
                }
                if (need < 0 || need > cells.Count)
                {
                    continue;
                }
                var constraint = new Constraint { MineCount = need, Cells = cells };
                numberToConstraint[idx] = constraint;
            }
        }

        var activeConstraints = numberToConstraint.Values.Where(c => c.Cells.Count > 0).ToList();
        PropagateSubset(activeConstraints, isUnknown, isFlag, safe);

        // Build connected components over the frontier.
        var cellIndex = new Dictionary<int, int>();
        var frontierCells = new List<int>();
        foreach (var c in activeConstraints)
        {
            foreach (var cell in c.Cells)
            {
                if (!cellIndex.ContainsKey(cell))
                {
                    cellIndex[cell] = frontierCells.Count;
                    frontierCells.Add(cell);
                }
            }
        }

        if (frontierCells.Count == 0)
        {
            return CollectSafe(safe, isUnknown, w);
        }

        var cellAdj = new List<int>[frontierCells.Count];
        for (int i = 0; i < cellAdj.Length; i++) cellAdj[i] = new List<int>();
        for (int ci = 0; ci < activeConstraints.Count; ci++)
        {
            foreach (var cell in activeConstraints[ci].Cells)
            {
                cellAdj[cellIndex[cell]].Add(ci);
            }
        }

        var visitedCell = new bool[frontierCells.Count];
        var visitedCon = new bool[activeConstraints.Count];
        const int ExactCellThreshold = 24;
        const int ExactConstraintThreshold = 28;

        for (int s = 0; s < frontierCells.Count; s++)
        {
            if (visitedCell[s]) continue;

            var compCells = new List<int>();
            var compCons = new List<int>();
            var q = new Queue<int>();
            q.Enqueue(s);
            visitedCell[s] = true;

            while (q.Count > 0)
            {
                int ci = q.Dequeue();
                int cellIdx = frontierCells[ci];
                compCells.Add(cellIdx);
                foreach (var constrIdx in cellAdj[ci])
                {
                    if (!visitedCon[constrIdx])
                    {
                        visitedCon[constrIdx] = true;
                        compCons.Add(constrIdx);
                        foreach (var otherCell in activeConstraints[constrIdx].Cells)
                        {
                            int oi = cellIndex[otherCell];
                            if (!visitedCell[oi])
                            {
                                visitedCell[oi] = true;
                                q.Enqueue(oi);
                            }
                        }
                    }
                }
            }

            if (compCells.Count <= ExactCellThreshold && compCons.Count <= ExactConstraintThreshold)
            {
                ExactSolveComponent(compCells, compCons, activeConstraints, safe);
            }
            else
            {
                CheapDeduceComponent(compCells, compCons, activeConstraints, isUnknown, safe);
            }
        }

        return CollectSafe(safe, isUnknown, w);
    }

    private static void PropagateSubset(List<Constraint> constraints, bool[] isUnknown, bool[] isFlag, bool[] safe)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int i = 0; i < constraints.Count; i++)
            {
                var a = constraints[i];
                if (a.Cells.Count == 0) continue;
                if (a.MineCount == 0)
                {
                    foreach (var cell in a.Cells)
                    {
                        if (isUnknown[cell] && !safe[cell])
                        {
                            safe[cell] = true;
                            changed = true;
                        }
                    }
                }
                if (a.MineCount == a.Cells.Count)
                {
                    // Known mines already flagged in input, but this helps if deduction ever needs it.
                    continue;
                }

                for (int j = i + 1; j < constraints.Count; j++)
                {
                    var b = constraints[j];
                    if (b.Cells.Count == 0) continue;
                    if (a.Cells.Count > b.Cells.Count)
                    {
                        var tmp = a; a = b; b = tmp;
                    }

                    if (IsSubset(a.Cells, b.Cells))
                    {
                        int diffCount = b.Cells.Count - a.Cells.Count;
                        int diffMines = b.MineCount - a.MineCount;
                        if (diffMines == 0)
                        {
                            foreach (var cell in Diff(b.Cells, a.Cells))
                            {
                                if (isUnknown[cell] && !safe[cell])
                                {
                                    safe[cell] = true;
                                    changed = true;
                                }
                            }
                        }
                        else if (diffMines == diffCount)
                        {
                            // All cells in B\A are mines.
                        }
                    }
                }
            }
        }
    }

    private static bool IsSubset(List<int> a, List<int> b)
    {
        if (a.Count > b.Count) return false;
        var set = new HashSet<int>(b);
        foreach (var x in a)
            if (!set.Contains(x)) return false;
        return true;
    }

    private static IEnumerable<int> Diff(List<int> a, List<int> b)
    {
        var set = new HashSet<int>(b);
        foreach (var x in a)
            if (!set.Contains(x)) yield return x;
    }

    private static void ExactSolveComponent(List<int> compCells, List<int> compCons, List<Constraint> constraints, bool[] safe)
    {
        int n = compCells.Count;
        var pos = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) pos[compCells[i]] = i;

        var consCells = new int[compCons.Count][];
        var consNeed = new int[compCons.Count];
        for (int i = 0; i < compCons.Count; i++)
        {
            var c = constraints[compCons[i]];
            consNeed[i] = c.MineCount;
            consCells[i] = c.Cells.Select(cell => pos[cell]).ToArray();
        }

        var cellToCons = new List<int>[n];
        for (int i = 0; i < n; i++) cellToCons[i] = new List<int>();
        for (int i = 0; i < consCells.Length; i++)
            foreach (var cell in consCells[i])
                cellToCons[cell].Add(i);

        int[] assigned = new int[n];
        Array.Fill(assigned, -1);
        int[] remain = new int[compCons.Count];
        int[] unassigned = new int[compCons.Count];
        for (int i = 0; i < compCons.Count; i++) unassigned[i] = consCells[i].Length;

        bool[] everMine = new bool[n];
        bool[] everSafe = new bool[n];
        int solutionCount = 0;

        int ChooseCell()
        {
            int best = -1;
            int bestDeg = -1;
            for (int i = 0; i < n; i++)
            {
                if (assigned[i] != -1) continue;
                int deg = cellToCons[i].Count;
                if (deg > bestDeg)
                {
                    bestDeg = deg;
                    best = i;
                }
            }
            return best;
        }

        bool Apply(int cell, int val, Stack<(int cons, int dr, int du)> delta)
        {
            assigned[cell] = val;
            if (val == 1) everMine[cell] = true; else everSafe[cell] = true;

            foreach (var ci in cellToCons[cell])
            {
                delta.Push((ci, 0, 0));
                unassigned[ci]--;
                if (val == 1) remain[ci]++;
                if (remain[ci] < 0 || remain[ci] > unassigned[ci]) return false;
            }
            return true;
        }

        void Undo(int cell, Stack<(int cons, int dr, int du)> delta)
        {
            assigned[cell] = -1;
            while (delta.Count > 0)
            {
                var t = delta.Pop();
                unassigned[t.cons]++;
                // remain updates are symmetric via cell value, so recompute from branches not needed here.
                // Instead, the pushed items are markers to know how many constraints were touched.
            }
        }

        void Restore(int cell, int val, List<int> touchedCons)
        {
            assigned[cell] = -1;
            foreach (var ci in touchedCons)
            {
                unassigned[ci]++;
                if (val == 1) remain[ci]--;
            }
        }

        bool Dfs()
        {
            int cell = ChooseCell();
            if (cell == -1)
            {
                for (int i = 0; i < consNeed.Length; i++)
                    if (remain[i] != consNeed[i]) return false;
                solutionCount++;
                return true;
            }

            bool any = false;
            var touched = cellToCons[cell];

            // Try safe first.
            {
                var backupRemain = new int[touched.Count];
                var backupUnassigned = new int[touched.Count];
                for (int i = 0; i < touched.Count; i++)
                {
                    int ci = touched[i];
                    backupRemain[i] = remain[ci];
                    backupUnassigned[i] = unassigned[ci];
                    unassigned[ci]--;
                }
                assigned[cell] = 0;
                for (int i = 0; i < touched.Count; i++)
                {
                    int ci = touched[i];
                    if (remain[ci] < 0 || remain[ci] > unassigned[ci]) goto SafeFail;
                }
                any |= Dfs();
            SafeFail:
                assigned[cell] = -1;
                for (int i = 0; i < touched.Count; i++)
                {
                    int ci = touched[i];
                    remain[ci] = backupRemain[i];
                    unassigned[ci] = backupUnassigned[i];
                }
            }

            // Try mine.
            {
                var backupRemain = new int[touched.Count];
                var backupUnassigned = new int[touched.Count];
                for (int i = 0; i < touched.Count; i++)
                {
                    int ci = touched[i];
                    backupRemain[i] = remain[ci];
                    backupUnassigned[i] = unassigned[ci];
                    remain[ci]++;
                    unassigned[ci]--;
                }
                assigned[cell] = 1;
                for (int i = 0; i < touched.Count; i++)
                {
                    int ci = touched[i];
                    if (remain[ci] < 0 || remain[ci] > unassigned[ci]) goto MineFail;
                }
                any |= Dfs();
            MineFail:
                assigned[cell] = -1;
                for (int i = 0; i < touched.Count; i++)
                {
                    int ci = touched[i];
                    remain[ci] = backupRemain[i];
                    unassigned[ci] = backupUnassigned[i];
                }
            }

            return any;
        }

        // Initialize constraint counts.
        for (int i = 0; i < consCells.Length; i++)
        {
            remain[i] = 0;
        }

        Dfs();

        for (int i = 0; i < n; i++)
        {
            if (everSafe[i] && !everMine[i])
            {
                safe[compCells[i]] = true;
            }
        }
    }

    private static void CheapDeduceComponent(List<int> compCells, List<int> compCons, List<Constraint> constraints, bool[] isUnknown, bool[] safe)
    {
        foreach (var ci in compCons)
        {
            var c = constraints[ci];
            if (c.MineCount == 0)
            {
                foreach (var cell in c.Cells)
                    if (isUnknown[cell]) safe[cell] = true;
            }
        }
    }

    private static List<(int col, int row)> CollectSafe(bool[] safe, bool[] isUnknown, int w)
    {
        var result = new List<(int col, int row)>();
        for (int i = 0; i < safe.Length; i++)
        {
            if (safe[i] && isUnknown[i])
            {
                result.Add((i % w, i / w));
            }
        }
        return result;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastInput(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= ' ' && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public string NextBoardLine(int expectedLength)
        {
            var chars = new char[expectedLength];
            int idx = 0;
            while (idx < expectedLength)
            {
                int c = Read();
                if (c < 0) break;
                if (c == '\r') continue;
                if (c == '\n')
                {
                    if (idx == 0) continue;
                    break;
                }
                chars[idx++] = (char)c;
            }
            return new string(chars, 0, idx);
        }
    }
}
