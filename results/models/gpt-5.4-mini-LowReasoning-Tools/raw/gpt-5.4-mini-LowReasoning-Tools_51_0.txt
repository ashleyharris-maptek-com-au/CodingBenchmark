from __future__ import annotations

import math
from statistics import median


_STATE = {
    "alt_i": 0.0,
    "spd_i": 0.0,
    "hdg_i": 0.0,
    "vs_i": 0.0,
    "trim": 0.0,
    "last_time": None,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _wrap180(a):
    return ((a + 180.0) % 360.0) - 180.0


def _wrap360(a):
    return a % 360.0


def _is_num(x):
    return isinstance(x, (int, float)) and math.isfinite(x)


def _pick(values, valids=None, lo=None, hi=None):
    cand = []
    for i, v in enumerate(values):
        ok = _is_num(v)
        if valids is not None:
            ok = ok and bool(valids[i])
        if ok:
            if lo is not None and v < lo:
                continue
            if hi is not None and v > hi:
                continue
            cand.append(float(v))
    if not cand:
        cand = [float(v) for v in values if _is_num(v)]
    if not cand:
        return None
    cand.sort()
    n = len(cand)
    if n == 1:
        return cand[0]
    if n == 2:
        return 0.5 * (cand[0] + cand[1])
    # Robust against a silent outlier.
    return median(cand)


def _circular_pick(values, valids=None):
    pts = []
    for i, v in enumerate(values):
        ok = _is_num(v)
        if valids is not None:
            ok = ok and bool(valids[i])
        if ok:
            pts.append(float(v) % 360.0)
    if not pts:
        pts = [float(v) % 360.0 for v in values if _is_num(v)]
    if not pts:
        return None
    sx = sum(math.cos(math.radians(a)) for a in pts)
    sy = sum(math.sin(math.radians(a)) for a in pts)
    if sx == 0.0 and sy == 0.0:
        return pts[0]
    return math.degrees(math.atan2(sy, sx)) % 360.0


def _rate_limit(prev, target, up_rate, down_rate, dt):
    delta = target - prev
    lim = up_rate * dt if delta > 0 else down_rate * dt
    return prev + _clamp(delta, -lim, lim)


def autopilot_step(state: dict, dt: float) -> dict:
    # Time base
    sim_dt = state.get("dt", dt)
    if not _is_num(sim_dt) or sim_dt <= 0:
        sim_dt = dt if _is_num(dt) and dt > 0 else 0.05
    dt = float(sim_dt)

    # Read redundant sources.
    ias = _pick([state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")],
                [state.get("ias_1_valid"), state.get("ias_2_valid"), state.get("ias_3_valid")],
                lo=0.0, hi=700.0)
    mach = _pick([state.get("mach_1"), state.get("mach_2"), state.get("mach_3")], lo=0.0, hi=1.5)
    tas = _pick([state.get("tas_1_kt"), state.get("tas_2_kt"), state.get("tas_3_kt")], lo=0.0, hi=800.0)
    aoa = _pick([state.get("aoa_1_deg"), state.get("aoa_2_deg"), state.get("aoa_3_deg")],
                [state.get("aoa_1_valid"), state.get("aoa_2_valid"), state.get("aoa_3_valid")],
                lo=-10.0, hi=45.0)

    baro_alt = _pick([state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")],
                     [state.get("baro_alt_1_valid"), state.get("baro_alt_2_valid"), state.get("baro_alt_3_valid")],
                     lo=-2000.0, hi=60000.0)
    radar_alt = _pick([state.get("radar_alt_1_ft"), state.get("radar_alt_2_ft"), state.get("radar_alt_3_ft")],
                      [state.get("radar_alt_1_valid"), state.get("radar_alt_2_valid"), state.get("radar_alt_3_valid")],
                      lo=-50.0, hi=20000.0)
    gps_alt = _pick([state.get("gps_1_alt_ft"), state.get("gps_2_alt_ft"), state.get("gps_3_alt_ft")],
                    [state.get("gps_1_valid"), state.get("gps_2_valid"), state.get("gps_3_valid")],
                    lo=-2000.0, hi=60000.0)
    gps_gs = _pick([state.get("gps_1_gs_kt"), state.get("gps_2_gs_kt"), state.get("gps_3_gs_kt")],
                   [state.get("gps_1_valid"), state.get("gps_2_valid"), state.get("gps_3_valid")],
                   lo=0.0, hi=800.0)

    gps_track = _circular_pick([state.get("gps_1_track_deg"), state.get("gps_2_track_deg"), state.get("gps_3_track_deg")],
                               [state.get("gps_1_valid"), state.get("gps_2_valid"), state.get("gps_3_valid")])
    irs_track = _circular_pick([state.get("irs_1_track_deg"), state.get("irs_2_track_deg"), state.get("irs_3_track_deg")],
                               [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")])
    heading = _circular_pick([state.get("irs_1_heading_deg"), state.get("irs_2_heading_deg"), state.get("irs_3_heading_deg")],
                             [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")])
    mag_hdg = _pick([state.get("magnetic_heading_deg")], lo=0.0, hi=360.0)
    track = gps_track if gps_track is not None else irs_track if irs_track is not None else mag_hdg
    if track is None:
        track = heading if heading is not None else 0.0

    vs = _pick([state.get("vertical_speed_fpm"), state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm")],
               [True, state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")],
               lo=-10000.0, hi=10000.0)

    # Targets.
    target_alt = float(state.get("target_alt_ft", baro_alt if baro_alt is not None else 0.0) or 0.0)
    target_hdg = float(state.get("target_hdg_deg", track))
    target_ias = float(state.get("target_ias_kt", ias if ias is not None else 220.0))
    target_vs = state.get("target_vs_fpm")
    target_vs = float(target_vs) if _is_num(target_vs) else None
    ap = bool(state.get("ap_engaged", True))
    at = bool(state.get("at_engaged", True))

    # Health / fallback.
    if baro_alt is None:
        baro_alt = gps_alt if gps_alt is not None else radar_alt if radar_alt is not None else 0.0
    if gps_alt is None:
        gps_alt = baro_alt
    if radar_alt is None:
        radar_alt = max(0.0, baro_alt - 1000.0)
    if ias is None:
        ias = gps_gs if gps_gs is not None else 220.0
    if mach is None:
        mach = 0.0
    if heading is None:
        heading = track

    # Basic regime detection.
    on_ground = any(bool(state.get(k)) for k in ("weight_on_wheels_left", "weight_on_wheels_nose", "weight_on_wheels_right"))
    low_alt = radar_alt is not None and radar_alt < 2500.0

    # Speed protection.
    stall_margin = 18.0
    vne_ias = 365.0
    vne_mach = 0.82
    if aoa is not None and aoa > 13.5:
        stall_margin += min(10.0, (aoa - 13.5) * 1.5)

    # Select altitude source with simple consistency logic.
    alt_for_control = baro_alt
    if radar_alt is not None and gps_alt is not None:
        # If one source is clearly inconsistent near the ground, favor radar.
        if low_alt and abs((gps_alt or radar_alt) - radar_alt) > 800.0 and abs(baro_alt - radar_alt) > 800.0:
            alt_for_control = radar_alt
        elif abs(baro_alt - gps_alt) > 1500.0 and abs(radar_alt - gps_alt) < abs(baro_alt - gps_alt):
            alt_for_control = gps_alt
    if alt_for_control is None:
        alt_for_control = 0.0

    # Heading control uses track when in coordinated flight; otherwise heading.
    hdg_source = track if track is not None else heading
    hdg_err = _wrap180(target_hdg - hdg_source)
    hdg_hold = _clamp(hdg_err / 45.0, -1.0, 1.0)
    if target_vs is None and abs(hdg_err) > 60.0:
        # Reduce aggressiveness in large heading changes at speed.
        hdg_hold *= 0.7

    # Altitude / vertical speed control.
    alt_err = target_alt - alt_for_control
    if target_vs is not None:
        vs_cmd = _clamp((target_vs - (vs or 0.0)) / 1200.0, -1.0, 1.0)
    else:
        # Capture profile: bias to stronger corrections when far away.
        vs_target = _clamp(alt_err * 4.0, -1800.0, 1800.0)
        vs_cmd = _clamp((vs_target - (vs or 0.0)) / 1200.0, -1.0, 1.0)
    _STATE["alt_i"] = _clamp(_STATE["alt_i"] + alt_err * dt / 3000.0, -0.4, 0.4)

    # Speed control with pitch/throttle split.
    spd_err = target_ias - ias
    if mach > vne_mach or ias > vne_ias:
        spd_bias = -1.0
    elif ias < 1.15 * stall_margin:
        spd_bias = 1.0
    else:
        spd_bias = _clamp(spd_err / 35.0, -1.0, 1.0)
    _STATE["spd_i"] = _clamp(_STATE["spd_i"] + spd_err * dt / 250.0, -0.25, 0.25)

    # Longitudinal control law.
    pitch_cmd = 0.0
    pitch_cmd += -0.00035 * alt_err
    pitch_cmd += 0.45 * vs_cmd
    pitch_cmd += -0.03 * spd_bias
    pitch_cmd += _STATE["alt_i"] * 0.6
    pitch_cmd += _STATE["spd_i"] * -0.4
    if aoa is not None:
        if aoa > 12.0:
            pitch_cmd = min(pitch_cmd, 0.1)
            if ias < 1.2 * stall_margin:
                pitch_cmd = min(pitch_cmd, -0.2)
        elif aoa < 4.0 and ias > target_ias + 25:
            pitch_cmd = min(pitch_cmd, -0.15)
    if on_ground and target_alt < 1000.0 and ias < 80.0:
        pitch_cmd = _clamp((target_ias - ias) / 50.0, -0.1, 0.25)

    # Lateral control law.
    roll_cmd = _clamp(hdg_err / 30.0, -1.0, 1.0)
    if gps_gs is not None and ias is not None and gps_gs > 30.0:
        drift = _wrap180((track if track is not None else heading) - (heading if heading is not None else track))
        roll_cmd += _clamp(drift / 40.0, -0.4, 0.4)
    roll_cmd = _clamp(roll_cmd, -1.0, 1.0)

    # Rudder for coordination and asymmetry.
    rudder_cmd = 0.0
    if heading is not None and track is not None:
        rudder_cmd += _clamp(_wrap180(heading - track) / 25.0, -0.5, 0.5)
    eng1 = float(state.get("engine_1_thrust_pct", 0.0) or 0.0)
    eng2 = float(state.get("engine_2_thrust_pct", 0.0) or 0.0)
    rudder_cmd += _clamp((eng2 - eng1) / 100.0, -0.5, 0.5)
    rudder_cmd += _clamp(roll_cmd * 0.15, -0.2, 0.2)
    rudder_cmd = _clamp(rudder_cmd, -1.0, 1.0)

    # Thrust management.
    throttle_base = 0.55
    throttle_base += _clamp(spd_err / 80.0, -0.35, 0.35)
    throttle_base += _clamp(-alt_err / 30000.0, -0.15, 0.15)
    throttle_base += _STATE["spd_i"] * 0.5
    throttle_base = _clamp(throttle_base, 0.0, 1.0)
    if not at:
        throttle_base = max(throttle_base, 0.65 if target_ias > 180 else 0.35)
    if mach > 0.78 or ias > 330:
        throttle_base = min(throttle_base, 0.85)
    if ias < 130 and alt_for_control < 5000:
        throttle_base = max(throttle_base, 0.55)

    # Engine asymmetry and failure handling.
    eng1_run = bool(state.get("engine_1_running", True))
    eng2_run = bool(state.get("engine_2_running", True))
    fire1 = bool(state.get("engine_1_fire", False))
    fire2 = bool(state.get("engine_2_fire", False))
    if fire1:
        throttle_1 = 0.0
        throttle_2 = throttle_base
    elif fire2:
        throttle_1 = throttle_base
        throttle_2 = 0.0
    else:
        diff = _clamp(rudder_cmd * 0.08, -0.12, 0.12)
        throttle_1 = _clamp(throttle_base - diff, 0.0, 1.0)
        throttle_2 = _clamp(throttle_base + diff, 0.0, 1.0)
        if not eng1_run:
            throttle_1 = 0.0
            throttle_2 = max(throttle_2, throttle_base + 0.1)
        if not eng2_run:
            throttle_2 = 0.0
            throttle_1 = max(throttle_1, throttle_base + 0.1)

    # High-speed drag devices / landing configuration.
    flaps = 0
    speedbrake = 0.0
    gear = 0
    if on_ground:
        gear = 1
        flaps = 15 if target_ias < 160 else 0
    else:
        if alt_for_control < 3000 and target_ias < 180:
            flaps = 15
        elif alt_for_control < 1000 and target_ias < 155:
            flaps = 20
        else:
            flaps = 0
        if ias > 300 or mach > 0.78:
            speedbrake = _clamp((max(0.0, ias - 300.0) / 60.0) + max(0.0, mach - 0.78) * 4.0, 0.0, 1.0)
        if alt_for_control < 2500 and target_ias < 170 and abs(alt_err) < 600 and (vs or 0.0) < -500:
            gear = 1

    # Trim as a slowly moving baseline.
    _STATE["trim"] = _clamp(_STATE["trim"] + pitch_cmd * dt * 0.08, -0.8, 0.8)
    elevator_trim = _clamp(_STATE["trim"] - 0.25 * _STATE["alt_i"], -1.0, 1.0)
    elevator_cmd = _clamp(pitch_cmd - elevator_trim * 0.25, -1.0, 1.0)
    aileron_cmd = _clamp(roll_cmd, -1.0, 1.0)
    rudder_cmd = _clamp(rudder_cmd, -1.0, 1.0)

    # Gentle smoothing / rate limiting against actuator saturation.
    out = {
        "elevator": elevator_cmd,
        "aileron": aileron_cmd,
        "rudder": rudder_cmd,
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": elevator_trim,
        "flaps": int(_clamp(round(flaps / 5.0) * 5.0, 0, 40)),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "gear": int(gear),
    }

    # If close to target and stable, relax the system slightly to avoid hunting.
    if abs(alt_err) < 75 and abs(spd_err) < 8 and abs(hdg_err) < 5:
        out["elevator"] *= 0.5
        out["aileron"] *= 0.5
        out["rudder"] *= 0.5
        out["speedbrake"] = 0.0
        if not on_ground and alt_for_control > 2000:
            out["flaps"] = 0

    # Keep outputs bounded exactly as requested.
    out["elevator"] = _clamp(out["elevator"], -1.0, 1.0)
    out["aileron"] = _clamp(out["aileron"], -1.0, 1.0)
    out["rudder"] = _clamp(out["rudder"], -1.0, 1.0)
    out["throttle_1"] = _clamp(out["throttle_1"], 0.0, 1.0)
    out["throttle_2"] = _clamp(out["throttle_2"], 0.0, 1.0)
    out["elevator_trim"] = _clamp(out["elevator_trim"], -1.0, 1.0)
    out["speedbrake"] = _clamp(out["speedbrake"], 0.0, 1.0)
    out["gear"] = 1 if out["gear"] else 0
    return out
