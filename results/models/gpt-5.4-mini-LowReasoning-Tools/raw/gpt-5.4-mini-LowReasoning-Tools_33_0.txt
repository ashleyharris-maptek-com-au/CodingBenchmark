using System;
using System.IO;
using System.Linq;
using System.Text;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n <= 0)
            return;

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextInt();
        }
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextInt();
        }

        long[] flowScore = new long[n];
        long[] distScore = new long[n];
        for (int i = 0; i < n; i++)
        {
            long fsum = 0, dsum = 0;
            for (int j = 0; j < n; j++)
            {
                fsum += Math.Abs((long)flow[i][j]) + Math.Abs((long)flow[j][i]);
                dsum += Math.Abs((long)dist[i][j]) + Math.Abs((long)dist[j][i]);
            }
            flowScore[i] = fsum;
            distScore[i] = dsum;
        }

        int[] facOrder = Enumerable.Range(0, n).OrderByDescending(i => flowScore[i]).ToArray();
        int[] locOrder = Enumerable.Range(0, n).OrderBy(i => distScore[i]).ToArray();

        int[] perm = new int[n];
        int[] posOfFacility = new int[n];
        for (int idx = 0; idx < n; idx++)
        {
            int f = facOrder[idx];
            int loc = locOrder[idx];
            perm[f] = loc;
            posOfFacility[f] = idx;
        }

        long currentCost = EvaluateCost(flow, dist, perm);

        int limit = Math.Min(n, 250);
        int window = n <= 350 ? n : 40;
        bool improved = true;
        int rounds = 0;
        while (improved && rounds < 4)
        {
            improved = false;
            rounds++;
            for (int a = 0; a < limit; a++)
            {
                int fa = facOrder[a];
                int pa = posOfFacility[fa];
                int start = Math.Max(0, a - window);
                int end = Math.Min(n - 1, a + window);
                for (int b = start; b <= end; b++)
                {
                    if (a == b) continue;
                    int fb = facOrder[b];
                    if (fa == fb) continue;
                    long delta = SwapDelta(flow, dist, perm, fa, fb);
                    if (delta < 0)
                    {
                        int la = perm[fa];
                        int lb = perm[fb];
                        perm[fa] = lb;
                        perm[fb] = la;
                        posOfFacility[fa] = b;
                        posOfFacility[fb] = a;
                        currentCost += delta;
                        improved = true;
                    }
                }
            }
        }

        if (n <= 80)
        {
            // Small instances can afford a few extra exhaustive swap passes.
            bool changed = true;
            int extra = 0;
            while (changed && extra < 3)
            {
                changed = false;
                extra++;
                for (int i = 0; i < n; i++)
                {
                    for (int j = i + 1; j < n; j++)
                    {
                        long delta = SwapDelta(flow, dist, perm, i, j);
                        if (delta < 0)
                        {
                            int tmp = perm[i];
                            perm[i] = perm[j];
                            perm[j] = tmp;
                            currentCost += delta;
                            changed = true;
                        }
                    }
                }
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(currentCost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    static long EvaluateCost(int[][] flow, int[][] dist, int[] perm)
    {
        int n = perm.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int[] flowRow = flow[i];
            for (int j = 0; j < n; j++)
                cost += (long)flowRow[j] * dist[pi][perm[j]];
        }
        return cost;
    }

    static long SwapDelta(int[][] flow, int[][] dist, int[] perm, int a, int b)
    {
        if (a == b) return 0;
        int n = perm.Length;
        int la = perm[a];
        int lb = perm[b];
        long delta = 0;
        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int lk = perm[k];
            delta += ((long)flow[a][k] - flow[b][k]) * ((long)dist[lb][lk] - dist[la][lk]);
            delta += ((long)flow[k][a] - flow[k][b]) * ((long)dist[lk][lb] - dist[lk][la]);
        }
        delta += ((long)flow[a][a] - flow[b][b]) * ((long)dist[lb][lb] - dist[la][la]);
        delta += ((long)flow[a][b] - flow[b][a]) * ((long)dist[lb][la] - dist[la][lb]);
        return delta;
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = ReadByte(); while (c <= ' ' && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }
}
