use std::io::{self, Read};

fn popcount(x: u64) -> usize {
    x.count_ones() as usize
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let k = n / 2;

    let mut adj = vec![0u64; n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u] |= 1u64 << v;
        adj[v] |= 1u64 << u;
        edges.push((u, v));
    }

    if n <= 26 {
        solve_bruteforce(n, k, &adj);
    } else {
        solve_branch_and_bound(n, k, &adj, &edges);
    }
}

fn cut_of_mask(mask: u64, n: usize, adj: &[u64]) -> usize {
    let mut cut = 0usize;
    for i in 0..n {
        let inside = ((mask >> i) & 1) != 0;
        let mut neigh = adj[i];
        while neigh != 0 {
            let j = neigh.trailing_zeros() as usize;
            neigh &= neigh - 1;
            if j > i {
                let other_inside = ((mask >> j) & 1) != 0;
                if inside != other_inside {
                    cut += 1;
                }
            }
        }
    }
    cut
}

fn solve_bruteforce(n: usize, k: usize, adj: &[u64]) {
    let mut best_cut = usize::MAX;
    let mut best_mask = 0u64;

    if k == 0 {
        println!("0\n");
        return;
    }

    let mut mask: u64 = (1u64 << k) - 1;
    let limit = 1u64 << n;

    while mask < limit {
        if (mask & 1) != 0 {
            let cut = cut_of_mask(mask, n, adj);
            if cut < best_cut {
                best_cut = cut;
                best_mask = mask;
            }
        }

        let c = mask & (!mask + 1);
        let r = mask + c;
        if r >= limit {
            break;
        }
        mask = (((r ^ mask) >> 2) / c) | r;
    }

    output_solution(best_cut, best_mask, n);
}

fn output_solution(best_cut: usize, mask: u64, n: usize) {
    let mut part = Vec::new();
    for i in 0..n {
        if ((mask >> i) & 1) != 0 {
            part.push(i.to_string());
        }
    }
    println!("{}", best_cut);
    println!("{}", part.join(" "));
}

fn solve_branch_and_bound(n: usize, k: usize, adj: &[u64], edges: &[(usize, usize)]) {
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by_key(|&v| std::cmp::Reverse(popcount(adj[v])));

    let mut pos = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        pos[v] = i;
    }

    let mut sorted_adj = vec![0u64; n];
    for v in 0..n {
        let mut bits = 0u64;
        let mut neigh = adj[v];
        while neigh != 0 {
            let j = neigh.trailing_zeros() as usize;
            neigh &= neigh - 1;
            bits |= 1u64 << pos[j];
        }
        sorted_adj[pos[v]] = bits;
    }

    let mut assigned = vec![None; n];
    assigned[0] = Some(true);
    let mut count_a = 1usize;
    let mut count_b = 0usize;
    let mut cur_cut = 0usize;
    let mut best_cut = usize::MAX;
    let mut best_mask = 0u64;

    let mut order2 = order.clone();
    order2[0] = order[0];

    fn dfs(
        idx: usize,
        n: usize,
        k: usize,
        order: &[usize],
        adj: &[u64],
        assigned: &mut [Option<bool>],
        count_a: usize,
        count_b: usize,
        cur_cut: usize,
        best_cut: &mut usize,
        best_mask: &mut u64,
    ) {
        if cur_cut >= *best_cut {
            return;
        }
        if idx == n {
            if count_a == k && count_b == n - k && cur_cut < *best_cut {
                *best_cut = cur_cut;
                let mut mask = 0u64;
                for i in 0..n {
                    if assigned[i] == Some(true) {
                        mask |= 1u64 << order[i];
                    }
                }
                *best_mask = mask;
            }
            return;
        }

        if assigned[idx].is_some() {
            dfs(
                idx + 1,
                n,
                k,
                order,
                adj,
                assigned,
                count_a,
                count_b,
                cur_cut,
                best_cut,
                best_mask,
            );
            return;
        }

        let remaining = n - idx;
        if count_a > k || count_b > k || count_a + remaining < k || count_b + remaining < k {
            return;
        }

        let mut try_side = |side_a: bool, assigned: &mut [Option<bool>], cur_cut: usize| -> Option<usize> {
            let mut add = 0usize;
            for j in 0..idx {
                if ((adj[idx] >> j) & 1) != 0 {
                    if let Some(other) = assigned[j] {
                        if other != side_a {
                            add += 1;
                        }
                    }
                }
            }
            assigned[idx] = Some(side_a);
            Some(cur_cut + add)
        };

        if count_a < k {
            if let Some(nc) = try_side(true, assigned, cur_cut) {
                dfs(
                    idx + 1,
                    n,
                    k,
                    order,
                    adj,
                    assigned,
                    count_a + 1,
                    count_b,
                    nc,
                    best_cut,
                    best_mask,
                );
            }
            assigned[idx] = None;
        }
        if count_b < k {
            if let Some(nc) = try_side(false, assigned, cur_cut) {
                dfs(
                    idx + 1,
                    n,
                    k,
                    order,
                    adj,
                    assigned,
                    count_a,
                    count_b + 1,
                    nc,
                    best_cut,
                    best_mask,
                );
            }
            assigned[idx] = None;
        }
    }

    let mut assigned_by_pos = vec![None; n];
    assigned_by_pos[0] = Some(true);
    dfs(
        1,
        n,
        k,
        &order,
        &sorted_adj,
        &mut assigned_by_pos,
        count_a,
        count_b,
        cur_cut,
        &mut best_cut,
        &mut best_mask,
    );

    if best_cut == usize::MAX {
        best_cut = 0;
        best_mask = 0;
        for i in 0..k {
            best_mask |= 1u64 << order[i];
        }
    }

    output_solution(best_cut, best_mask, n);
}
