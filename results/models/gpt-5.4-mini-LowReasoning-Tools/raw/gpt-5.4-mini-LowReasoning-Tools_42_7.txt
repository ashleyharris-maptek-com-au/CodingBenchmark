OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %v3float "v3float"
OpName %v4float "v4float"
OpName %float_0_08 "float_0_08"
OpName %float_3 "float_3"
OpName %uint_0 "uint_0"
OpName %int_0 "int_0"
OpName %int_1 "int_1"
OpName %int_2 "int_2"
OpName %int_3 "int_3"
OpName %face_cross "cross"
OpName %face_normal "normalize"
OpName %center "center"

%glsl = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_0 = OpConstant %uint 0
%uint_3 = OpConstant %uint 3
%float_0_08 = OpConstant %float 0.08
%float_third = OpConstant %float 0.333333343
%float_3 = OpConstant %float 3.0

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
OpDecorate %outColor Location 0
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %funct
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2

%e1 = OpCompositeConstruct %v3float %p1x %p1y %p1z
%e1b = OpCompositeConstruct %v3float %p0x %p0y %p0z
%e1v = OpFSub %v3float %e1 %e1b
%e2 = OpCompositeConstruct %v3float %p2x %p2y %p2z
%e2v = OpFSub %v3float %e2 %e1b

%face_cross = OpExtInst %v3float %glsl Cross %e1v %e2v
%face_normal = OpExtInst %v3float %glsl Normalize %face_cross
%offset = OpVectorTimesScalar %v3float %face_normal %float_0_08

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%center = OpVectorTimesScalar %v4float %sum012 %float_third

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%emit0 = OpFAdd %v4float %p0 %offset
%emit1 = OpFAdd %v4float %p1 %offset
%emit2 = OpFAdd %v4float %p2 %offset

%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %emit0
OpStore %outColor %c0
OpEmitVertex

%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %emit1
OpStore %outColor %c1
OpEmitVertex

%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %emit2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
