#include <algorithm>
#include <cstdint>
#include <iostream>
#include <queue>
#include <unordered_map>
#include <vector>

#ifdef _MSC_VER
#include <intrin.h>
#endif

using namespace std;

struct Bitset {
    int n = 0;
    vector<uint64_t> a;

    Bitset() = default;
    explicit Bitset(int n_) { init(n_); }

    void init(int n_) {
        n = n_;
        a.assign((n + 63) >> 6, 0);
    }

    void set(int i) {
        a[i >> 6] |= 1ULL << (i & 63);
    }

    void reset(int i) {
        a[i >> 6] &= ~(1ULL << (i & 63));
    }

    bool test(int i) const {
        return (a[i >> 6] >> (i & 63)) & 1ULL;
    }

    bool any() const {
        for (uint64_t x : a) if (x) return true;
        return false;
    }

    int popcount() const {
        int s = 0;
        for (uint64_t x : a) {
            while (x) {
                x &= (x - 1);
                ++s;
            }
        }
        return s;
    }
};

static inline int lsb_index(uint64_t x) {
#ifdef _MSC_VER
    unsigned long idx;
    _BitScanForward64(&idx, x);
    return (int)idx;
#else
    return __builtin_ctzll(x);
#endif
}

struct Solver {
    int n = 0;
    vector<Bitset> adj;
    vector<int> best_global;
    vector<int> cur;

    vector<int> solve_component(const vector<int>& verts) {
        int k = (int)verts.size();
        if (k == 0) return {};

        unordered_map<int, int> idx;
        idx.reserve(k * 2 + 1);
        for (int i = 0; i < k; ++i) idx[verts[i]] = i;

        vector<Bitset> g(k, Bitset(k));
        vector<int> deg(k, 0);
        for (int i = 0; i < k; ++i) {
            int u = verts[i];
            for (uint64_t w : adj[u].a) {
                (void)w;
            }
            for (int j = 0; j < k; ++j) {
                if (i == j) continue;
                int v = verts[j];
                if (adj[u].test(v)) {
                    g[i].set(j);
                    deg[i]++;
                }
            }
        }

        vector<int> answer_local;
        vector<int> current_local;

        vector<Bitset> compAdj = g;
        Bitset all(k);
        for (int i = 0; i < k; ++i) all.set(i);

        function<void(Bitset&, vector<int>&)> expand = [&](Bitset& P, vector<int>& path) {
            if (!P.any()) {
                if (path.size() > answer_local.size()) answer_local = path;
                return;
            }

            vector<int> order;
            vector<int> bounds;
            order.reserve(P.popcount());
            bounds.reserve(P.popcount());

            Bitset rem = P;
            int color = 0;
            while (rem.any()) {
                ++color;
                Bitset avail = rem;
                while (avail.any()) {
                    int v = -1;
                    for (int wi = 0; wi < (int)avail.a.size(); ++wi) {
                        if (avail.a[wi]) {
                            v = (wi << 6) + lsb_index(avail.a[wi]);
                            break;
                        }
                    }
                    order.push_back(v);
                    bounds.push_back(color);
                    rem.reset(v);
                    avail.reset(v);
                    for (int wi = 0; wi < (int)avail.a.size(); ++wi) {
                        avail.a[wi] &= g[v].a[wi];
                    }
                }
            }

            for (int i = (int)order.size() - 1; i >= 0; --i) {
                if ((int)path.size() + bounds[i] <= (int)answer_local.size()) return;
                int v = order[i];
                Bitset P2 = P;
                for (int wi = 0; wi < (int)P2.a.size(); ++wi) {
                    P2.a[wi] &= g[v].a[wi];
                }
                path.push_back(v);
                expand(P2, path);
                path.pop_back();
                P.reset(v);
            }
        };

        Bitset P(k);
        for (int i = 0; i < k; ++i) P.set(i);
        vector<int> path;
        expand(P, path);

        vector<int> res;
        for (int x : answer_local) res.push_back(verts[x]);
        return res;
    }

    vector<int> solve() {
        vector<char> vis(n, 0);
        vector<int> ans;
        for (int s = 0; s < n; ++s) {
            if (vis[s]) continue;
            queue<int> q;
            q.push(s);
            vis[s] = 1;
            vector<int> comp;
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                comp.push_back(u);
                for (int v = 0; v < n; ++v) {
                    if (!vis[v] && adj[u].test(v)) {
                        vis[v] = 1;
                        q.push(v);
                    }
                }
            }

            bool independent_component = true;
            for (int u : comp) {
                for (int v : comp) {
                    if (u < v && adj[u].test(v)) {
                        independent_component = false;
                        break;
                    }
                }
                if (!independent_component) break;
            }
            if (independent_component) {
                ans.insert(ans.end(), comp.begin(), comp.end());
                continue;
            }

            vector<int> part = solve_component(comp);
            ans.insert(ans.end(), part.begin(), part.end());
        }
        sort(ans.begin(), ans.end());
        return ans;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    Solver solver;
    solver.n = n;
    solver.adj.assign(n, Bitset(n));

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        solver.adj[u].set(v);
        solver.adj[v].set(u);
    }

    vector<int> ans = solver.solve();
    cout << ans.size() << '\n';
    for (size_t i = 0; i < ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
