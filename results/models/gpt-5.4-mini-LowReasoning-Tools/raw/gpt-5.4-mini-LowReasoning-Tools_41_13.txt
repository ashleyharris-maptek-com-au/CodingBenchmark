; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 35
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %outColor "outColor"
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_input_float = OpTypePointer Input %float
%c_int_1 = OpConstant %int 1
%c_float_0_0 = OpConstant %float 0.0
%c_float_0_1 = OpConstant %float 0.1
%c_float_0_4 = OpConstant %float 0.4
%c_float_0_5 = OpConstant %float 0.5
%c_float_1_0 = OpConstant %float 1.0
%glsl = OpExtInstImport "GLSL.std.450"
%worldPos = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%main = OpFunction %void None %func
%label = OpLabel
%y_ptr = OpAccessChain %ptr_input_float %worldPos %c_int_1
%y = OpLoad %float %y_ptr
%half_y = OpFMul %float %y %c_float_0_5
%t_unclamped = OpFAdd %float %half_y %c_float_0_5
%t = OpExtInst %float %glsl FClamp %t_unclamped %c_float_0_0 %c_float_1_0
%bottom = OpCompositeConstruct %v3float %c_float_0_1 %c_float_0_4 %c_float_0_1
%top = OpCompositeConstruct %v3float %c_float_1_0 %c_float_1_0 %c_float_1_0
%rgb = OpExtInst %v3float %glsl FMix %bottom %top %t
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%color = OpCompositeConstruct %v4float %r %g %b %c_float_1_0
OpStore %outColor %color
OpReturn
OpFunctionEnd
