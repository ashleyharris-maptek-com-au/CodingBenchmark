import math


# Persistent state across calls within a scenario.
_prev_t = {
    "x": 0.0,
    "y": 0.0,
    "z": 0.0,
    "vx": 0.0,
    "vy": 0.0,
    "vz": 0.0,
    "init": False,
}


def _clamp(v, lo, hi):
    return lo if v < lo else hi if v > hi else v


def _valid(state, key):
    return bool(state.get(key, False))


def _fnum(v):
    try:
        return float(v)
    except Exception:
        return None


def _weighted_average(samples):
    num = [0.0, 0.0, 0.0]
    den = [0.0, 0.0, 0.0]
    for value, valid, w in samples:
        if not valid or value is None:
            continue
        for i in range(3):
            if value[i] is None or math.isnan(value[i]) or math.isinf(value[i]):
                continue
            num[i] += w * value[i]
            den[i] += w
    return [
        num[i] / den[i] if den[i] > 0 else None
        for i in range(3)
    ]


def _best_scalar(samples):
    num = 0.0
    den = 0.0
    for value, valid, w in samples:
        if not valid or value is None or math.isnan(value) or math.isinf(value):
            continue
        num += w * value
        den += w
    return num / den if den > 0 else None


def _estimate_attitude(state):
    pitch_vals = []
    roll_vals = []
    yaw_vals = []
    for i, w in ((1, 1.0), (2, 0.9), (3, 0.8)):
        pv = _fnum(state.get(f"gyro_{i}_pitch_dps"))
        rv = _fnum(state.get(f"gyro_{i}_roll_dps"))
        yv = _fnum(state.get(f"gyro_{i}_yaw_dps"))
        if _valid(state, f"gyro_{i}_pitch_valid") and pv is not None:
            pitch_vals.append((pv, True, w))
        if _valid(state, f"gyro_{i}_roll_valid") and rv is not None:
            roll_vals.append((rv, True, w))
        if _valid(state, f"gyro_{i}_yaw_valid") and yv is not None:
            yaw_vals.append((yv, True, w))

    # Star trackers are absolute attitude references.
    for i, w in ((1, 1.2), (2, 1.1)):
        pv = _fnum(state.get(f"star_tracker_{i}_pitch_deg"))
        rv = _fnum(state.get(f"star_tracker_{i}_roll_deg"))
        yv = _fnum(state.get(f"star_tracker_{i}_yaw_deg"))
        if _valid(state, f"star_tracker_{i}_pitch_valid") and pv is not None:
            pitch_vals.append((0.0, True, 0.0))  # placeholder to keep shape
        if _valid(state, f"star_tracker_{i}_roll_valid") and rv is not None:
            roll_vals.append((0.0, True, 0.0))
        if _valid(state, f"star_tracker_{i}_yaw_valid") and yv is not None:
            yaw_vals.append((0.0, True, 0.0))

    # Return only rate estimates from gyros; star trackers mainly confirm availability.
    pitch = _best_scalar(pitch_vals) if pitch_vals else 0.0
    roll = _best_scalar(roll_vals) if roll_vals else 0.0
    yaw = _best_scalar(yaw_vals) if yaw_vals else 0.0
    return pitch or 0.0, roll or 0.0, yaw or 0.0


def _fuse_relative_state(state, dt):
    pos_samples = []
    vel_samples = []

    radar_pos = [
        _fnum(state.get("radar_rel_pos_x_m")),
        _fnum(state.get("radar_rel_pos_y_m")),
        _fnum(state.get("radar_rel_pos_z_m")),
    ]
    radar_vel = [
        _fnum(state.get("radar_rel_vel_x_ms")),
        _fnum(state.get("radar_rel_vel_y_ms")),
        _fnum(state.get("radar_rel_vel_z_ms")),
    ]
    if all(_valid(state, k) for k in (
        "radar_rel_x_valid", "radar_rel_y_valid", "radar_rel_z_valid"
    )):
        pos_samples.append((radar_pos, True, 1.0))
    if all(_valid(state, k) for k in (
        "radar_relv_x_valid", "radar_relv_y_valid", "radar_relv_z_valid"
    )):
        vel_samples.append((radar_vel, True, 1.0))

    lidar_pos = [
        _fnum(state.get("lidar_rel_pos_x_m")),
        _fnum(state.get("lidar_rel_pos_y_m")),
        _fnum(state.get("lidar_rel_pos_z_m")),
    ]
    lidar_vel = [
        _fnum(state.get("lidar_rel_vel_x_ms")),
        _fnum(state.get("lidar_rel_vel_y_ms")),
        _fnum(state.get("lidar_rel_vel_z_ms")),
    ]
    if all(_valid(state, k) for k in (
        "lidar_rel_x_valid", "lidar_rel_y_valid", "lidar_rel_z_valid"
    )):
        pos_samples.append((lidar_pos, True, 1.2))
    if all(_valid(state, k) for k in (
        "lidar_relv_x_valid", "lidar_relv_y_valid", "lidar_relv_z_valid"
    )):
        vel_samples.append((lidar_vel, True, 1.2))

    # Range finders: use as magnitude only, bias along-track if camera is unavailable.
    ranges = []
    for i, w in ((1, 0.5), (2, 0.7), (3, 0.6)):
        rv = _fnum(state.get(f"range_{i}_m"))
        if _valid(state, f"range_{i}_valid") and rv is not None and rv > 0:
            ranges.append((rv, True, w))

    pos = _weighted_average(pos_samples)
    vel = _weighted_average(vel_samples)
    rng = _best_scalar(ranges)

    # Fallback / propagation when one or more sensors fail.
    if pos[0] is None or pos[1] is None or pos[2] is None:
        if _prev_t["init"]:
            pos = [_prev_t["x"], _prev_t["y"], _prev_t["z"]]
        else:
            pos = [0.0, 0.0, 0.0]
    if vel[0] is None or vel[1] is None or vel[2] is None:
        if _prev_t["init"]:
            vel = [_prev_t["vx"], _prev_t["vy"], _prev_t["vz"]]
        else:
            vel = [0.0, 0.0, 0.0]

    # Use range to improve norm estimate when only noisy position is available.
    if rng is not None:
        norm = math.sqrt(pos[0] * pos[0] + pos[1] * pos[1] + pos[2] * pos[2])
        if norm > 1e-6:
            scale = rng / norm
            pos = [pos[0] * scale, pos[1] * scale, pos[2] * scale]

    # Camera provides final bearing if active.
    cam_active = bool(state.get("dock_camera_active", False))
    if cam_active:
        bx = _fnum(state.get("dock_camera_bearing_x"))
        by = _fnum(state.get("dock_camera_bearing_y"))
        bz = _fnum(state.get("dock_camera_bearing_z"))
        br = _fnum(state.get("dock_camera_range_m"))
        if bx is not None and by is not None and bz is not None:
            n = math.sqrt(bx * bx + by * by + bz * bz)
            if n > 1e-6:
                bx, by, bz = bx / n, by / n, bz / n
                if br is not None and br > 0:
                    pos = [bx * br, by * br, bz * br]
                else:
                    if rng is not None:
                        pos = [bx * rng, by * rng, bz * rng]

    _prev_t.update({"x": pos[0], "y": pos[1], "z": pos[2], "vx": vel[0], "vy": vel[1], "vz": vel[2], "init": True})
    return pos, vel, rng


def autopilot_step(state: dict, dt: float) -> dict:
    pos, vel, rng = _fuse_relative_state(state, dt)
    x, y, z = pos
    vx, vy, vz = vel

    dist = math.sqrt(x * x + y * y + z * z)
    if dist < 1e-6:
        dist = 0.0
    speed = math.sqrt(vx * vx + vy * vy + vz * vz)

    # Control law selection.
    if dist > 200.0:
        # Far range: exploit orbital mechanics. If behind, go retrograde to drop orbit;
        # if ahead, prograde to climb. Use radial/cross-track only for gross alignment.
        thrust_y = -0.9 if y < 0 else 0.9 if y > 0 else 0.0
        thrust_x = _clamp(-0.003 * x - 0.12 * vx, -0.35, 0.35)
        thrust_z = _clamp(-0.003 * z - 0.12 * vz, -0.35, 0.35)
    elif dist > 50.0:
        # Medium range: V-bar approach with damped guidance, avoid aggressive radial motion.
        # Bias along-track to close the y gap, hold x/z near zero.
        thrust_y = _clamp(-0.006 * y - 0.25 * vy, -0.8, 0.8)
        thrust_x = _clamp(-0.010 * x - 0.28 * vx, -0.45, 0.45)
        thrust_z = _clamp(-0.010 * z - 0.28 * vz, -0.45, 0.45)
        # If ahead, prefer braking rather than chasing.
        if y > 30.0:
            thrust_y = min(thrust_y, -0.15)
        elif y < -30.0:
            thrust_y = max(thrust_y, 0.15)
    elif dist > 10.0:
        # Close range: transition to linear regime.
        thrust_y = _clamp(-0.012 * y - 0.35 * vy, -0.5, 0.5)
        thrust_x = _clamp(-0.018 * x - 0.38 * vx, -0.5, 0.5)
        thrust_z = _clamp(-0.018 * z - 0.38 * vz, -0.5, 0.5)
    else:
        # Final approach: very gentle, prioritize speed reduction.
        thrust_y = _clamp(-0.020 * y - 0.55 * vy, -0.22, 0.22)
        thrust_x = _clamp(-0.025 * x - 0.60 * vx, -0.22, 0.22)
        thrust_z = _clamp(-0.025 * z - 0.60 * vz, -0.22, 0.22)

    # Special handling for dangerous closing speed.
    closing = 0.0
    if dist > 1e-6:
        closing = (x * vx + y * vy + z * vz) / dist
    if closing < -0.3:
        brake = _clamp((-closing - 0.3) * 1.5, 0.0, 1.0)
        thrust_x -= brake * (vx / (speed + 1e-6))
        thrust_y -= brake * (vy / (speed + 1e-6))
        thrust_z -= brake * (vz / (speed + 1e-6))

    # Cross-track is especially important; suppress if near dock unless actually offset.
    if abs(z) < 1.0 and abs(vz) < 0.05:
        thrust_z *= 0.5

    # If camera has bearing, use it to align the final vector more precisely.
    if bool(state.get("dock_camera_active", False)):
        bx = _fnum(state.get("dock_camera_bearing_x"))
        by = _fnum(state.get("dock_camera_bearing_y"))
        bz = _fnum(state.get("dock_camera_bearing_z"))
        if bx is not None and by is not None and bz is not None:
            n = math.sqrt(bx * bx + by * by + bz * bz)
            if n > 1e-6:
                bx, by, bz = bx / n, by / n, bz / n
                # Convert bearing error into a gentle corrective push.
                thrust_x += _clamp(-0.15 * bx, -0.2, 0.2)
                thrust_y += _clamp(-0.15 * by, -0.2, 0.2)
                thrust_z += _clamp(-0.15 * bz, -0.2, 0.2)

    # Ground truth is stale but reliable; use as long-term bias if we have almost no relative data.
    if rng is None and dist == 0.0:
        gsa = _fnum(state.get("ground_semi_major_axis_m"))
        if gsa is not None and gsa > 0:
            # Slightly prefer conservative braking in uncertain orbit geometry.
            thrust_y *= 0.9
            thrust_x *= 0.9
            thrust_z *= 0.9

    # Fuel management: scale back when low fuel or already nearly docked.
    fuel_pct = _fnum(state.get("fuel_remaining_pct"))
    fuel_kg = _fnum(state.get("fuel_remaining_kg"))
    fuel_scale = 1.0
    if fuel_pct is not None:
        fuel_scale = min(fuel_scale, max(0.25, fuel_pct / 100.0))
    if fuel_kg is not None and fuel_kg < 100.0:
        fuel_scale = min(fuel_scale, 0.55)
    thrust_x *= fuel_scale
    thrust_y *= fuel_scale
    thrust_z *= fuel_scale

    # Small deadband to prevent chatter from sensor flicker and uncommanded thrusts.
    def deadband(v, d=0.02):
        return 0.0 if abs(v) < d else v

    thrust_x = deadband(_clamp(thrust_x, -1.0, 1.0))
    thrust_y = deadband(_clamp(thrust_y, -1.0, 1.0))
    thrust_z = deadband(_clamp(thrust_z, -1.0, 1.0))

    # Attitude rates: keep near-zero rates, but allow gentle reorientation when far away.
    # We do not know the spacecraft's axis mapping, so use small coordinated rates.
    att_rate_roll = _clamp(-0.002 * z - 0.010 * vz, -0.035, 0.035)
    att_rate_pitch = _clamp(-0.002 * x - 0.010 * vx, -0.035, 0.035)
    att_rate_yaw = _clamp(-0.003 * y - 0.012 * vy, -0.035, 0.035)

    # If we are very close, prioritize stable docking attitude.
    if dist < 15.0:
        att_rate_roll *= 0.5
        att_rate_pitch *= 0.5
        att_rate_yaw *= 0.5

    return {
        "thrust_x": thrust_x,
        "thrust_y": thrust_y,
        "thrust_z": thrust_z,
        "att_rate_roll": att_rate_roll,
        "att_rate_pitch": att_rate_pitch,
        "att_rate_yaw": att_rate_yaw,
    }
