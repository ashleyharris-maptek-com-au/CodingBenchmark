#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

struct Solver {
    int n, m;
    vector<int64> w;
    vector<vector<int>> adj;
    vector<vector<uint64_t>> nb;
    vector<int> bestSol, curSol;
    int64 bestWeight = 0;

    static bool testBit(const vector<uint64_t>& bs, int v) {
        return (bs[v >> 6] >> (v & 63)) & 1ULL;
    }

    static void setBit(vector<uint64_t>& bs, int v) {
        bs[v >> 6] |= 1ULL << (v & 63);
    }

    static bool any(const vector<uint64_t>& bs) {
        for (uint64_t x : bs) if (x) return true;
        return false;
    }

    vector<uint64_t> removeVertex(const vector<uint64_t>& cand, int v) {
        vector<uint64_t> out = cand;
        out[v >> 6] &= ~(1ULL << (v & 63));
        return out;
    }

    int pickVertex(const vector<uint64_t>& cand) {
        int best = -1, bestDeg = -1;
        for (int v = 0; v < n; ++v) {
            if (!testBit(cand, v)) continue;
            int deg = 0;
            for (int u : adj[v]) if (testBit(cand, u)) ++deg;
            if (deg > bestDeg || (deg == bestDeg && (best == -1 || w[v] > w[best]))) {
                bestDeg = deg;
                best = v;
            }
        }
        return best;
    }

    int64 greedyUpperBound(vector<uint64_t> cand) {
        int64 ub = 0;
        while (any(cand)) {
            int best = -1;
            for (int v = 0; v < n; ++v) {
                if (!testBit(cand, v)) continue;
                if (best == -1 || w[v] > w[best]) best = v;
            }
            ub += w[best];
            cand = removeVertex(cand, best);
            for (int u : adj[best]) if (testBit(cand, u)) {
                cand[u >> 6] &= ~(1ULL << (u & 63));
            }
        }
        return ub;
    }

    void dfs(const vector<uint64_t>& cand, int64 curWeight) {
        if (curWeight > bestWeight) {
            bestWeight = curWeight;
            bestSol = curSol;
        }
        if (!any(cand)) return;
        int64 ub = curWeight + greedyUpperBound(cand);
        if (ub <= bestWeight) return;

        int v = pickVertex(cand);
        if (v == -1) return;

        // Take v.
        {
            curSol[v] = 1;
            vector<uint64_t> next = removeVertex(cand, v);
            for (int u : adj[v]) if (testBit(next, u)) next[u >> 6] &= ~(1ULL << (u & 63));
            dfs(next, curWeight + w[v]);
            curSol[v] = 0;
        }

        // Skip v.
        {
            vector<uint64_t> next = removeVertex(cand, v);
            dfs(next, curWeight);
        }
    }

    void solve() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        cin >> n >> m;
        w.resize(n);
        for (int i = 0; i < n; ++i) cin >> w[i];
        adj.assign(n, {});
        for (int i = 0; i < m; ++i) {
            int u, v;
            cin >> u >> v;
            if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
            adj[u].push_back(v);
            adj[v].push_back(u);
        }

        int words = (n + 63) / 64;
        nb.assign(n, vector<uint64_t>(words, 0));
        for (int i = 0; i < n; ++i) {
            for (int j : adj[i]) setBit(nb[i], j);
        }

        vector<uint64_t> cand(words, ~0ULL);
        if (n % 64) cand.back() &= ((1ULL << (n % 64)) - 1ULL);
        curSol.assign(n, 0);
        bestSol.assign(n, 0);
        bestWeight = 0;

        dfs(cand, 0);

        cout << bestWeight << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << bestSol[i];
        }
        cout << "\n";
    }
};

int main() {
    Solver s;
    s.solve();
    return 0;
}
