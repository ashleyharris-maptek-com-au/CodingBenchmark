OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %color "color"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex"
OpName %gl_PerVertex_out "gl_PerVertex"

OpDecorate %worldPos Location 0
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %gl_PerVertex_out Block

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_6 = OpConstant %uint 6
%uint_1v = OpConstant %uint 1
%uint_6v = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_n1 = OpConstant %float -1
%float_p05 = OpConstant %float 0.5
%float_n05 = OpConstant %float -0.5
%float_s3_2 = OpConstant %float 0.866025388
%float_ns3_2 = OpConstant %float -0.866025388
%float_r = OpConstant %float 0.04
%v4float_1 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_v3arr = OpTypePointer Input %arr_v3_1
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_v3arr Input
%color = OpVariable %ptr_in_v3arr Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel
%wp0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
%c0 = OpAccessChain %ptr_in_v3 %color %int_0
%pos3 = OpLoad %v3float %wp0
%col = OpLoad %v3float %c0

%x = OpCompositeExtract %float %pos3 0
%y = OpCompositeExtract %float %pos3 1
%z = OpCompositeExtract %float %pos3 2

%r0 = OpFMul %float %float_r %float_1
%r1 = OpFMul %float %float_r %float_p05
%r2 = OpFMul %float %float_r %float_n05
%r3 = OpFMul %float %float_r %float_n1
%r4 = OpFMul %float %float_r %float_n05
%r5 = OpFMul %float %float_r %float_p05
%s0 = OpFMul %float %float_r %float_0
%s1 = OpFMul %float %float_r %float_s3_2
%s2 = OpFMul %float %float_r %float_s3_2
%s3 = OpFMul %float %float_r %float_0
%s4 = OpFMul %float %float_r %float_ns3_2
%s5 = OpFMul %float %float_r %float_ns3_2

%c0x = OpFAdd %float %x %r0
%c0y = OpFAdd %float %y %s0
%c1x = OpFAdd %float %x %r1
%c1y = OpFAdd %float %y %s1
%c2x = OpFAdd %float %x %r2
%c2y = OpFAdd %float %y %s2
%c3x = OpFAdd %float %x %r3
%c3y = OpFAdd %float %y %s3
%c4x = OpFAdd %float %x %r4
%c4y = OpFAdd %float %y %s4
%c5x = OpFAdd %float %x %r5
%c5y = OpFAdd %float %y %s5

%p_center = OpCompositeConstruct %v4float %x %y %z %float_1
%e0 = OpCompositeConstruct %v4float %c0x %c0y %z %float_1
%e1 = OpCompositeConstruct %v4float %c1x %c1y %z %float_1
%e2 = OpCompositeConstruct %v4float %c2x %c2y %z %float_1
%e3 = OpCompositeConstruct %v4float %c3x %c3y %z %float_1
%e4 = OpCompositeConstruct %v4float %c4x %c4y %z %float_1
%e5 = OpCompositeConstruct %v4float %c5x %c5y %z %float_1

OpStore %outColor %col
%outp = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp %p_center
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e0
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e1
OpEmitVertex
OpEndPrimitive

OpStore %outColor %col
OpStore %outp %p_center
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e1
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e2
OpEmitVertex
OpEndPrimitive

OpStore %outColor %col
OpStore %outp %p_center
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e2
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e3
OpEmitVertex
OpEndPrimitive

OpStore %outColor %col
OpStore %outp %p_center
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e3
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e4
OpEmitVertex
OpEndPrimitive

OpStore %outColor %col
OpStore %outp %p_center
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e4
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e5
OpEmitVertex
OpEndPrimitive

OpStore %outColor %col
OpStore %outp %p_center
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e5
OpEmitVertex
OpStore %outColor %col
OpStore %outp %e0
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
