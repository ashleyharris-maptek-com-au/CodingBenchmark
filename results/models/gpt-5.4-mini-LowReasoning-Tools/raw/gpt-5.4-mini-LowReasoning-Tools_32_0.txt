#include <algorithm>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    int id;
    long long w;
};

struct InputEdge {
    int u, v;
    long long w;
};

static const long long INF = (1LL << 60);

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<InputEdge> edges(m);
    vector<vector<Edge>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        g[u].push_back({v, i, w});
        g[v].push_back({u, i, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    auto dijkstra = [&](int src) {
        vector<long long> dist(n, INF);
        vector<int> pv(n, -1), pe(n, -1);
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        dist[src] = 0;
        pq.push({0, src});
        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dist[v]) continue;
            for (const auto &e : g[v]) {
                if (dist[e.to] > d + e.w) {
                    dist[e.to] = d + e.w;
                    pv[e.to] = v;
                    pe[e.to] = e.id;
                    pq.push({dist[e.to], e.to});
                }
            }
        }
        return tuple<vector<long long>, vector<int>, vector<int>>(move(dist), move(pv), move(pe));
    };

    auto emit_tree = [&](const vector<char> &used) {
        vector<int> parent(n), rnk(n, 0);
        iota(parent.begin(), parent.end(), 0);
        function<int(int)> findp = [&](int x) {
            return parent[x] == x ? x : parent[x] = findp(parent[x]);
        };
        auto unite = [&](int a, int b) {
            a = findp(a);
            b = findp(b);
            if (a == b) return false;
            if (rnk[a] < rnk[b]) swap(a, b);
            parent[b] = a;
            if (rnk[a] == rnk[b]) ++rnk[a];
            return true;
        };

        vector<int> out;
        long long total = 0;
        for (int i = 0; i < m; ++i) {
            if (!used[i]) continue;
            if (unite(edges[i].u, edges[i].v)) {
                out.push_back(i);
                total += edges[i].w;
            }
        }

        cout << total << ' ' << out.size() << '\n';
        for (int id : out) {
            cout << edges[id].u << ' ' << edges[id].v << '\n';
        }
    };

    if (k == 0) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    if (k <= 18) {
        int S = 1 << k;
        vector<vector<long long>> dp(S, vector<long long>(n, INF));
        vector<vector<int>> splitA(S, vector<int>(n, -1)), splitB(S, vector<int>(n, -1));
        vector<vector<int>> prevV(S, vector<int>(n, -1)), prevE(S, vector<int>(n, -1));
        vector<vector<char>> kind(S, vector<char>(n, 0)); // 1 split, 2 path

        for (int i = 0; i < k; ++i) dp[1 << i][terminals[i]] = 0;

        for (int mask = 1; mask < S; ++mask) {
            if ((mask & (mask - 1)) != 0) {
                for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                    int other = mask ^ sub;
                    if (sub > other) continue;
                    for (int v = 0; v < n; ++v) {
                        long long cand = dp[sub][v] + dp[other][v];
                        if (cand < dp[mask][v]) {
                            dp[mask][v] = cand;
                            splitA[mask][v] = sub;
                            splitB[mask][v] = other;
                            kind[mask][v] = 1;
                        }
                    }
                }
            }

            priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
            }
            while (!pq.empty()) {
                auto [d, v] = pq.top();
                pq.pop();
                if (d != dp[mask][v]) continue;
                for (const auto &e : g[v]) {
                    if (dp[mask][e.to] > d + e.w) {
                        dp[mask][e.to] = d + e.w;
                        prevV[mask][e.to] = v;
                        prevE[mask][e.to] = e.id;
                        kind[mask][e.to] = 2;
                        pq.push({dp[mask][e.to], e.to});
                    }
                }
            }
        }

        int full = S - 1;
        long long best = INF;
        int bestv = -1;
        for (int v = 0; v < n; ++v) {
            if (dp[full][v] < best) {
                best = dp[full][v];
                bestv = v;
            }
        }

        vector<char> used(m, 0);
        function<void(int, int)> build = [&](int mask, int v) {
            if ((mask & (mask - 1)) == 0) return;
            if (kind[mask][v] == 1) {
                build(splitA[mask][v], v);
                build(splitB[mask][v], v);
            } else if (kind[mask][v] == 2) {
                int p = prevV[mask][v];
                if (p != -1) {
                    build(mask, p);
                    used[prevE[mask][v]] = 1;
                }
            }
        };

        if (bestv != -1) build(full, bestv);
        emit_tree(used);
        return 0;
    }

    // Heuristic fallback: terminal MST on metric closure, then prune cycles.
    vector<vector<long long>> td(k, vector<long long>(n, INF));
    vector<vector<int>> tpv(k, vector<int>(n, -1)), tpe(k, vector<int>(n, -1));
    for (int i = 0; i < k; ++i) {
        auto [dist, pv, pe] = dijkstra(terminals[i]);
        td[i] = move(dist);
        tpv[i] = move(pv);
        tpe[i] = move(pe);
    }

    struct KEdge {
        long long w;
        int a, b;
    };
    vector<KEdge> kt;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            kt.push_back({td[i][terminals[j]], i, j});
        }
    }
    sort(kt.begin(), kt.end(), [](const KEdge &x, const KEdge &y) { return x.w < y.w; });

    vector<int> p(k), r(k, 0);
    iota(p.begin(), p.end(), 0);
    function<int(int)> f = [&](int x) { return p[x] == x ? x : p[x] = f(p[x]); };
    auto u = [&](int a, int b) {
        a = f(a);
        b = f(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    };

    vector<char> used(m, 0);
    for (const auto &e : kt) {
        if (!u(e.a, e.b)) continue;
        int src = terminals[e.a];
        int cur = terminals[e.b];
        while (cur != src && cur != -1) {
            int id = tpe[e.a][cur];
            used[id] = 1;
            cur = tpv[e.a][cur];
        }
    }

    emit_tree(used);
    return 0;
}
