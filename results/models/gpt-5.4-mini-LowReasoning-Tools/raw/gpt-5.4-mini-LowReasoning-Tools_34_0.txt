from math import hypot
import sys
from heapq import nlargest


def dist(a, b):
    return hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, pts, depot):
    if not route:
        return 0.0
    total = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        total += dist(pts[route[i]], pts[route[i + 1]])
    total += dist(pts[route[-1]], depot)
    return total


def two_opt(route, pts, depot):
    n = len(route)
    if n < 4:
        return route
    improved = True
    best = route[:]
    best_len = route_distance(best, pts, depot)
    while improved:
        improved = False
        for i in range(n - 2):
            a = depot if i == 0 else pts[best[i - 1]]
            b = pts[best[i]]
            for j in range(i + 2, n):
                c = pts[best[j - 1]]
                d = depot if j == n else pts[best[j]]
                old = dist(a, b) + dist(c, d)
                new = dist(a, c) + dist(b, d)
                if new + 1e-12 < old:
                    best[i:j] = reversed(best[i:j])
                    best_len += new - old
                    improved = True
                    break
            if improved:
                break
    return best


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))
    depot = (float(next(it)), float(next(it)))
    pts = []
    demands = []
    for _ in range(N):
        x = float(next(it)); y = float(next(it)); d = float(next(it))
        pts.append((x, y))
        demands.append(d)

    # Initial singleton routes.
    routes = [[i] for i in range(N)]
    route_load = demands[:]
    route_of = list(range(N))

    # Clarke-Wright savings.
    savings = []
    for i in range(N):
        di = dist(depot, pts[i])
        for j in range(i + 1, N):
            s = di + dist(depot, pts[j]) - dist(pts[i], pts[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    def route_valid(rid):
        return rid is not None and 0 <= rid < len(routes) and routes[rid] is not None

    for s, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj or not route_valid(ri) or not route_valid(rj):
            continue
        r1 = routes[ri]
        r2 = routes[rj]
        if route_load[ri] + route_load[rj] > CAP + 1e-9:
            continue
        # Only merge if i and j are at route endpoints.
        if r1[-1] == i and r2[0] == j:
            new_route = r1 + r2
        elif r1[0] == i and r2[-1] == j:
            new_route = r2 + r1
        elif r1[0] == i and r2[0] == j:
            new_route = list(reversed(r1)) + r2
        elif r1[-1] == i and r2[-1] == j:
            new_route = r1 + list(reversed(r2))
        else:
            continue
        # Perform merge.
        routes[ri] = None
        routes[rj] = None
        merged_load = route_load[ri] + route_load[rj]
        route_load[ri] = route_load[rj] = 0.0
        new_rid = len(routes)
        routes.append(new_route)
        route_load.append(merged_load)
        for c in new_route:
            route_of[c] = new_rid

    final_routes = [r for r in routes if r]

    # If too many routes remain, greedily insert small routes into others when possible.
    # Since each customer initially has its own route, we may still exceed K if merging is limited.
    # Merge by cheapest feasible concatenation at route ends.
    changed = True
    while changed and len(final_routes) > K:
        changed = False
        best_move = None
        best_gain = -1e18
        m = len(final_routes)
        loads = [sum(demands[c] for c in r) for r in final_routes]
        for a in range(m):
            for b in range(a + 1, m):
                if loads[a] + loads[b] > CAP + 1e-9:
                    continue
                ra = final_routes[a]
                rb = final_routes[b]
                candidates = [
                    (ra + rb, route_distance(ra, pts, depot) + route_distance(rb, pts, depot) - route_distance(ra + rb, pts, depot)),
                    (rb + ra, route_distance(ra, pts, depot) + route_distance(rb, pts, depot) - route_distance(rb + ra, pts, depot)),
                    (list(reversed(ra)) + rb, route_distance(ra, pts, depot) + route_distance(rb, pts, depot) - route_distance(list(reversed(ra)) + rb, pts, depot)),
                    (ra + list(reversed(rb)), route_distance(ra, pts, depot) + route_distance(rb, pts, depot) - route_distance(ra + list(reversed(rb)), pts, depot)),
                    (list(reversed(ra)) + list(reversed(rb)), route_distance(ra, pts, depot) + route_distance(rb, pts, depot) - route_distance(list(reversed(ra)) + list(reversed(rb)), pts, depot)),
                ]
                for nr, gain in candidates:
                    if gain > best_gain:
                        best_gain = gain
                        best_move = (a, b, nr)
        if best_move is not None:
            a, b, nr = best_move
            if a > b:
                a, b = b, a
            new_list = []
            for idx, r in enumerate(final_routes):
                if idx not in (a, b):
                    new_list.append(r)
            new_list.append(nr)
            final_routes = new_list
            changed = True

    # Route polishing.
    final_routes = [two_opt(r, pts, depot) for r in final_routes]

    total = sum(route_distance(r, pts, depot) for r in final_routes)
    print(f"{total:.6f}")
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(map(str, r)))


if __name__ == "__main__":
    main()
