using System;
using System.Collections.Generic;
using System.Globalization;

public class Program
{
    private const double PosScale = 1.0 / 5000.0;
    private const double VelScale = 1.0 / 10.0;

    private struct State
    {
        public double X, Y, Z, VX, VY, VZ;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        double mu = fs.NextDouble();

        State start = ReadState(fs);
        var stations = new State[n];
        for (int i = 0; i < n; i++) stations[i] = ReadState(fs);

        int[] order = Solve(start, stations);
        var sb = new System.Text.StringBuilder();
        for (int i = 0; i < order.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(order[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static State ReadState(FastScanner fs)
    {
        return new State
        {
            X = fs.NextDouble(),
            Y = fs.NextDouble(),
            Z = fs.NextDouble(),
            VX = fs.NextDouble(),
            VY = fs.NextDouble(),
            VZ = fs.NextDouble()
        };
    }

    private static int[] Solve(State start, State[] stations)
    {
        int n = stations.Length;
        if (n == 0) return Array.Empty<int>();

        if (n <= 10)
            return ExactSolve(start, stations);

        double[,] cost = BuildCostMatrix(start, stations);

        int[] order = NearestNeighbor(cost, n);
        ImproveOrder(order, cost, 4);
        return order;
    }

    private static int[] ExactSolve(State start, State[] stations)
    {
        int n = stations.Length;
        int m = 1 << n;
        double[,] dist = BuildCostMatrix(start, stations);
        double[,] dp = new double[m, n];
        int[,] parent = new int[m, n];
        for (int mask = 0; mask < m; mask++)
            for (int i = 0; i < n; i++)
                dp[mask, i] = double.PositiveInfinity;

        for (int i = 0; i < n; i++)
        {
            int bit = 1 << i;
            dp[bit, i] = dist[0, i + 1];
            parent[bit, i] = -1;
        }

        for (int mask = 1; mask < m; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (double.IsPositiveInfinity(cur)) continue;
                for (int nxt = 0; nxt < n; nxt++)
                {
                    int bit = 1 << nxt;
                    if ((mask & bit) != 0) continue;
                    int nmask = mask | bit;
                    double cand = cur + dist[last + 1, nxt + 1];
                    if (cand < dp[nmask, nxt])
                    {
                        dp[nmask, nxt] = cand;
                        parent[nmask, nxt] = last;
                    }
                }
            }
        }

        int full = m - 1;
        int bestLast = 0;
        double best = double.PositiveInfinity;
        for (int i = 0; i < n; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                bestLast = i;
            }
        }

        int[] order = new int[n];
        int mask2 = full;
        int idx = n - 1;
        while (idx >= 0)
        {
            order[idx] = bestLast;
            int prev = parent[mask2, bestLast];
            mask2 ^= 1 << bestLast;
            bestLast = prev;
            idx--;
        }
        return order;
    }

    private static int[] NearestNeighbor(double[,] cost, int n)
    {
        bool[] used = new bool[n];
        int[] order = new int[n];
        int cur = -1;
        for (int k = 0; k < n; k++)
        {
            int best = -1;
            double bestCost = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double c = cur < 0 ? cost[0, i + 1] : cost[cur + 1, i + 1];
                if (c < bestCost)
                {
                    bestCost = c;
                    best = i;
                }
            }
            used[best] = true;
            order[k] = best;
            cur = best;
        }
        return order;
    }

    private static void ImproveOrder(int[] order, double[,] cost, int passes)
    {
        int n = order.Length;
        if (n < 3) return;

        for (int pass = 0; pass < passes; pass++)
        {
            bool changed = false;

            for (int i = 0; i < n - 1; i++)
            {
                int a = i == 0 ? -1 : order[i - 1];
                int b = order[i];
                for (int j = i + 1; j < n; j++)
                {
                    int c = order[j];
                    int d = j == n - 1 ? -1 : order[j + 1];

                    double before = (a < 0 ? cost[0, b + 1] : cost[a + 1, b + 1]) + (d < 0 ? 0 : cost[c + 1, d + 1]);
                    double after = (a < 0 ? cost[0, c + 1] : cost[a + 1, c + 1]) + (d < 0 ? 0 : cost[b + 1, d + 1]);

                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(order, i, j - i + 1);
                        changed = true;
                        goto NextPass;
                    }
                }
            }

            int bestFrom = -1, bestTo = -1;
            double bestGain = 0.0;
            for (int i = 0; i < n; i++)
            {
                int node = order[i];
                int prev = i == 0 ? -1 : order[i - 1];
                int next = i == n - 1 ? -1 : order[i + 1];
                double remove = (prev < 0 ? cost[0, node + 1] : cost[prev + 1, node + 1]) + (next < 0 ? 0 : cost[node + 1, next + 1]);
                for (int j = 0; j < n; j++)
                {
                    if (j == i) continue;
                    int left = j < i ? (j == 0 ? -1 : order[j - 1]) : order[j];
                    int right = j < i ? order[j] : (j == n - 1 ? -1 : order[j + 1]);
                    if (j == i - 1) continue;
                    if (j == i) continue;
                    double insert = (left < 0 ? cost[0, node + 1] : cost[left + 1, node + 1]) + (right < 0 ? 0 : cost[node + 1, right + 1]);
                    double old = (left < 0 ? (j < i ? cost[0, right + 1] : cost[0, node + 1]) : cost[left + 1, right + 1]);
                    double gain = remove - insert;
                    if (gain > bestGain)
                    {
                        bestGain = gain;
                        bestFrom = i;
                        bestTo = j;
                    }
                }
            }

            if (bestFrom >= 0 && bestGain > 1e-12)
            {
                int node = order[bestFrom];
                if (bestFrom < bestTo) Array.Copy(order, bestFrom + 1, order, bestFrom, bestTo - bestFrom);
                else Array.Copy(order, bestTo, order, bestTo + 1, bestFrom - bestTo);
                order[bestTo] = node;
                changed = true;
            }

        NextPass:
            if (!changed) break;
        }

        TwoOptFine(order, cost);
    }

    private static void TwoOptFine(int[] order, double[,] cost)
    {
        int n = order.Length;
        if (n < 4) return;
        int limit = Math.Min(n * 3, 250);
        int tries = 0;
        bool improved = true;
        while (improved && tries++ < limit)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                int a = i == 0 ? -1 : order[i - 1];
                int b = order[i];
                for (int j = i + 1; j < n; j++)
                {
                    int c = order[j];
                    int d = j == n - 1 ? -1 : order[j + 1];
                    double before = (a < 0 ? cost[0, b + 1] : cost[a + 1, b + 1]) + (d < 0 ? 0 : cost[c + 1, d + 1]);
                    double after = (a < 0 ? cost[0, c + 1] : cost[a + 1, c + 1]) + (d < 0 ? 0 : cost[b + 1, d + 1]);
                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(order, i, j - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    private static double[,] BuildCostMatrix(State start, State[] stations)
    {
        int n = stations.Length;
        double[,] cost = new double[n + 1, n + 1];
        for (int i = 0; i <= n; i++)
        {
            for (int j = i + 1; j <= n; j++)
            {
                double c = Dist(i == 0 ? start : stations[i - 1], j == 0 ? start : stations[j - 1]);
                cost[i, j] = cost[j, i] = c;
            }
        }
        return cost;
    }

    private static double Dist(State a, State b)
    {
        double dx = (a.X - b.X) * PosScale;
        double dy = (a.Y - b.Y) * PosScale;
        double dz = (a.Z - b.Z) * PosScale;
        double dvx = (a.VX - b.VX) * VelScale;
        double dvy = (a.VY - b.VY) * VelScale;
        double dvz = (a.VZ - b.VZ) * VelScale;
        return Math.Sqrt(dx * dx + dy * dy + dz * dz + dvx * dvx + dvy * dvy + dvz * dvz);
    }

    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= ' ' && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public double NextDouble()
        {
            int c;
            do c = Read(); while (c <= ' ' && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            double val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            if (c == '.')
            {
                double div = 1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    div *= 10;
                    val += (c - '0') / div;
                    c = Read();
                }
            }
            if (c == 'e' || c == 'E')
            {
                int esign = 1;
                int exp = 0;
                c = Read();
                if (c == '-')
                {
                    esign = -1;
                    c = Read();
                }
                else if (c == '+')
                {
                    c = Read();
                }
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                val *= Math.Pow(10, esign * exp);
            }
            return sign * val;
        }
    }
}
