@group(0) @binding(0) var<storage, read> baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result : array<u32, 4>;
@group(0) @binding(2) var<uniform> params : vec4<u32>;

fn rotl(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce : u32 = params.y + gid.x;

    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        var a0 : u32 = state[0u];
        var b0 : u32 = state[1u];
        var c0 : u32 = state[2u];
        var d0 : u32 = state[3u];
        a0 = a0 + b0;
        d0 = rotl(d0 ^ a0, 16u);
        c0 = c0 + d0;
        b0 = rotl(b0 ^ c0, 12u);
        a0 = a0 + b0;
        d0 = rotl(d0 ^ a0, 8u);
        c0 = c0 + d0;
        b0 = rotl(b0 ^ c0, 7u);
        state[0u] = a0;
        state[1u] = b0;
        state[2u] = c0;
        state[3u] = d0;

        var a1 : u32 = state[4u];
        var b1 : u32 = state[5u];
        var c1 : u32 = state[6u];
        var d1 : u32 = state[7u];
        a1 = a1 + b1;
        d1 = rotl(d1 ^ a1, 16u);
        c1 = c1 + d1;
        b1 = rotl(b1 ^ c1, 12u);
        a1 = a1 + b1;
        d1 = rotl(d1 ^ a1, 8u);
        c1 = c1 + d1;
        b1 = rotl(b1 ^ c1, 7u);
        state[4u] = a1;
        state[5u] = b1;
        state[6u] = c1;
        state[7u] = d1;

        var a2 : u32 = state[8u];
        var b2 : u32 = state[9u];
        var c2 : u32 = state[10u];
        var d2 : u32 = state[11u];
        a2 = a2 + b2;
        d2 = rotl(d2 ^ a2, 16u);
        c2 = c2 + d2;
        b2 = rotl(b2 ^ c2, 12u);
        a2 = a2 + b2;
        d2 = rotl(d2 ^ a2, 8u);
        c2 = c2 + d2;
        b2 = rotl(b2 ^ c2, 7u);
        state[8u] = a2;
        state[9u] = b2;
        state[10u] = c2;
        state[11u] = d2;

        var a3 : u32 = state[12u];
        var b3 : u32 = state[13u];
        var c3 : u32 = state[14u];
        var d3 : u32 = state[15u];
        a3 = a3 + b3;
        d3 = rotl(d3 ^ a3, 16u);
        c3 = c3 + d3;
        b3 = rotl(b3 ^ c3, 12u);
        a3 = a3 + b3;
        d3 = rotl(d3 ^ a3, 8u);
        c3 = c3 + d3;
        b3 = rotl(b3 ^ c3, 7u);
        state[12u] = a3;
        state[13u] = b3;
        state[14u] = c3;
        state[15u] = d3;
    }

    let h0 : u32 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^ state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    let h1 : u32 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^ state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

    var tz : u32 = 0u;
    if (h0 != 0u) {
        var x : u32 = h0;
        while ((x & 1u) == 0u) {
            tz = tz + 1u;
            x = x >> 1u;
        }
    } else if (h1 != 0u) {
        tz = 32u;
        var x1 : u32 = h1;
        while ((x1 & 1u) == 0u) {
            tz = tz + 1u;
            x1 = x1 >> 1u;
        }
    } else {
        tz = 64u;
    }

    if (tz >= params.x) {
        result[0u] = 1u;
        result[1u] = nonce;
        result[2u] = h0;
        result[3u] = h1;
    }
}
