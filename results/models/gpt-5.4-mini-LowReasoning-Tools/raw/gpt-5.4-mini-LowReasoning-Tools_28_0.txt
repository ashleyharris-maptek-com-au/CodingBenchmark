use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).expect("failed to read stdin");
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut num = 0usize;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            num = num
                .saturating_mul(10)
                .saturating_add((self.buf[self.idx] - b'0') as usize);
            self.idx += 1;
        }
        num
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        edges.push((u as u32, v as u32));
    }

    let mut matched = vec![false; n];
    let mut cover = vec![false; n];

    for &(u32_u, u32_v) in &edges {
        let u = u32_u as usize;
        let v = u32_v as usize;
        if !matched[u] && !matched[v] {
            matched[u] = true;
            matched[v] = true;
            cover[u] = true;
            cover[v] = true;
        }
    }

    let mut out = String::new();
    let mut count = 0usize;
    for &is_in_cover in &cover {
        if is_in_cover {
            count += 1;
        }
    }
    out.push_str(&count.to_string());
    out.push('\n');

    let mut first = true;
    for i in 0..n {
        if cover[i] {
            if !first {
                out.push(' ');
            }
            first = false;
            out.push_str(&i.to_string());
        }
    }
    out.push('\n');

    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(out.as_bytes()).expect("failed to write stdout");
}
