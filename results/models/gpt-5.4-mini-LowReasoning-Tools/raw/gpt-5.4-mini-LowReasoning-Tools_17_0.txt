using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private struct Box
    {
        public int Index;
        public int W;
        public int H;
        public int D;
        public long Volume;
        public int MaxSide;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
        public long Volume => (long)W * H * D;
    }

    private struct Placement
    {
        public int Index;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
        public long Score;
    }

    private static readonly (int a, int b, int c)[] Rotations =
    {
        (0, 1, 2), (0, 2, 1), (1, 0, 2), (1, 2, 0), (2, 0, 1), (2, 1, 0)
    };

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int cw = fs.NextInt();
        int ch = fs.NextInt();
        int cd = fs.NextInt();

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            long vol = (long)w * h * d;
            boxes.Add(new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Volume = vol,
                MaxSide = Math.Max(w, Math.Max(h, d))
            });
        }

        boxes.Sort((a, b) =>
        {
            int c = a.Volume.CompareTo(b.Volume);
            if (c != 0) return c;
            c = a.MaxSide.CompareTo(b.MaxSide);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        });

        var spaces = new List<Space>(Math.Max(4, n * 2));
        spaces.Add(new Space { X = 0, Y = 0, Z = 0, W = cw, H = ch, D = cd });

        var placements = new List<Placement>(n);

        for (int bi = 0; bi < boxes.Count; bi++)
        {
            var box = boxes[bi];
            int bestSpace = -1;
            Placement bestPlacement = default;
            long bestScore = long.MaxValue;

            for (int si = 0; si < spaces.Count; si++)
            {
                var s = spaces[si];
                for (int r = 0; r < 6; r++)
                {
                    var rot = Rotations[r];
                    int[] dims = { box.W, box.H, box.D };
                    int bw = dims[rot.a];
                    int bh = dims[rot.b];
                    int bd = dims[rot.c];
                    if (bw > s.W || bh > s.H || bd > s.D) continue;

                    long leftover = (long)s.W * s.H * s.D - (long)bw * bh * bd;
                    long tightness = ((long)(s.W - bw) << 42) + ((long)(s.H - bh) << 21) + (s.D - bd);
                    long score = (leftover << 16) + tightness;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestSpace = si;
                        bestPlacement = new Placement
                        {
                            Index = box.Index,
                            X = s.X,
                            Y = s.Y,
                            Z = s.Z,
                            W = bw,
                            H = bh,
                            D = bd,
                            Score = score
                        };
                    }
                }
            }

            if (bestSpace < 0)
                continue;

            var chosen = spaces[bestSpace];
            spaces.RemoveAt(bestSpace);
            placements.Add(bestPlacement);

            int px = chosen.X;
            int py = chosen.Y;
            int pz = chosen.Z;
            int bw2 = bestPlacement.W;
            int bh2 = bestPlacement.H;
            int bd2 = bestPlacement.D;

            int rightW = chosen.W - bw2;
            if (rightW > 0)
            {
                spaces.Add(new Space
                {
                    X = px + bw2,
                    Y = py,
                    Z = pz,
                    W = rightW,
                    H = chosen.H,
                    D = chosen.D
                });
            }

            int frontH = chosen.H - bh2;
            if (frontH > 0)
            {
                spaces.Add(new Space
                {
                    X = px,
                    Y = py + bh2,
                    Z = pz,
                    W = bw2,
                    H = frontH,
                    D = chosen.D
                });
            }

            int topD = chosen.D - bd2;
            if (topD > 0)
            {
                spaces.Add(new Space
                {
                    X = px,
                    Y = py,
                    Z = pz + bd2,
                    W = bw2,
                    H = bh2,
                    D = topD
                });
            }

            if (spaces.Count > 1)
                PruneContainedSpaces(spaces);
        }

        var output = new StringWriter();
        output.WriteLine(placements.Count);
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            output.Write(p.Index);
            output.Write(' ');
            output.Write(p.X);
            output.Write(' ');
            output.Write(p.Y);
            output.Write(' ');
            output.Write(p.Z);
            output.WriteLine();
        }

        Console.Write(output.ToString());
    }

    private static void PruneContainedSpaces(List<Space> spaces)
    {
        for (int i = 0; i < spaces.Count; i++)
        {
            var a = spaces[i];
            for (int j = spaces.Count - 1; j > i; j--)
            {
                var b = spaces[j];
                if (Contains(a, b))
                {
                    spaces.RemoveAt(j);
                }
                else if (Contains(b, a))
                {
                    spaces.RemoveAt(i);
                    i--;
                    break;
                }
            }
        }
    }

    private static bool Contains(in Space a, in Space b)
    {
        return b.X >= a.X && b.Y >= a.Y && b.Z >= a.Z
            && b.X + b.W <= a.X + a.W
            && b.Y + b.H <= a.Y + a.H
            && b.Z + b.D <= a.Z + a.D;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = c - '0';
            while (true)
            {
                c = Read();
                if (c < '0' || c > '9') break;
                val = val * 10 + (c - '0');
            }
            return val * sign;
        }
    }
}
