using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public string? Next()
        {
            while (true)
            {
                if (_ptr >= _len)
                {
                    _len = _stream.Read(_buffer, 0, _buffer.Length);
                    _ptr = 0;
                    if (_len <= 0) return null;
                }

                while (_ptr < _len && char.IsWhiteSpace((char)_buffer[_ptr])) _ptr++;
                if (_ptr < _len) break;
            }

            int start = _ptr;
            while (true)
            {
                if (_ptr >= _len) break;
                if (char.IsWhiteSpace((char)_buffer[_ptr])) break;
                _ptr++;
            }

            return System.Text.Encoding.ASCII.GetString(_buffer, start, _ptr - start);
        }

        public int NextInt()
        {
            var s = Next();
            return s == null ? 0 : int.Parse(s);
        }
    }

    private readonly struct Rotation
    {
        public readonly int Width;
        public readonly int Height;
        public readonly (int X, int Y)[] Cells;

        public Rotation(int width, int height, (int X, int Y)[] cells)
        {
            Width = width;
            Height = height;
            Cells = cells;
        }
    }

    private sealed class Board
    {
        private readonly int _width;
        private readonly List<int> _rows = new();

        public Board(int width) => _width = width;

        public int Width => _width;
        public int Height => _rows.Count;

        public int GetCell(int x, int y)
        {
            if (y < 0 || y >= _rows.Count) return 0;
            return (_rows[y] & (1 << x)) != 0 ? 1 : 0;
        }

        public int ColumnHeight(int x)
        {
            for (int y = _rows.Count - 1; y >= 0; y--)
            {
                if ((_rows[y] & (1 << x)) != 0) return y + 1;
            }
            return 0;
        }

        public int[] GetHeights()
        {
            var h = new int[_width];
            for (int x = 0; x < _width; x++) h[x] = ColumnHeight(x);
            return h;
        }

        public int CountHoles()
        {
            int holes = 0;
            for (int x = 0; x < _width; x++)
            {
                bool seenBlock = false;
                for (int y = _rows.Count - 1; y >= 0; y--)
                {
                    bool filled = (_rows[y] & (1 << x)) != 0;
                    if (filled) seenBlock = true;
                    else if (seenBlock) holes++;
                }
            }
            return holes;
        }

        public int SurfaceRoughness()
        {
            var h = GetHeights();
            int rough = 0;
            for (int i = 1; i < h.Length; i++) rough += Math.Abs(h[i] - h[i - 1]);
            return rough;
        }

        public int MaxHeight()
        {
            int mh = 0;
            for (int x = 0; x < _width; x++) mh = Math.Max(mh, ColumnHeight(x));
            return mh;
        }

        public bool CanPlace(Rotation rot, int left, int bottom)
        {
            if (left < 0 || left + rot.Width > _width) return false;
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int x = left + rot.Cells[i].X;
                int y = bottom + rot.Cells[i].Y;
                if (y < 0) return false;
                if (GetCell(x, y) != 0) return false;
            }
            return true;
        }

        public int DropY(Rotation rot, int left)
        {
            int y = 0;
            while (true)
            {
                if (!CanPlace(rot, left, y)) return y - 1;
                y++;
            }
        }

        public int SimulateAndScore(Rotation rot, int left, HeuristicWeights w, out int cleared)
        {
            var snapshot = new List<int>(_rows);
            return SimulateOnRows(snapshot, rot, left, w, out cleared);
        }

        private int SimulateOnRows(List<int> rows, Rotation rot, int left, HeuristicWeights w, out int cleared)
        {
            int y = DropYOnRows(rows, rot, left);
            if (y < 0)
            {
                cleared = -1;
                return int.MinValue / 4;
            }

            EnsureRows(rows, y + rot.Height);
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int x = left + rot.Cells[i].X;
                int row = y + rot.Cells[i].Y;
                rows[row] |= 1 << x;
            }

            int clearedNow = ClearFullRows(rows);
            cleared = clearedNow;

            int[] heights = GetHeights(rows);
            int agg = 0, maxH = 0;
            for (int i = 0; i < heights.Length; i++)
            {
                agg += heights[i];
                if (heights[i] > maxH) maxH = heights[i];
            }
            int holes = CountHoles(rows);
            int rough = SurfaceRoughness(rows);
            int well = WellDepth(heights);

            int score = 0;
            score += w.LineClear * clearedNow;
            score -= w.AggregateHeight * agg;
            score -= w.Holes * holes;
            score -= w.Roughness * rough;
            score -= w.MaxHeight * maxH;
            score -= w.WellDepth * well;

            return score;
        }

        public void ApplyPlacement(Rotation rot, int left)
        {
            int y = DropY(rot, left);
            if (y < 0) return;
            EnsureRows(_rows, y + rot.Height);
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int x = left + rot.Cells[i].X;
                int row = y + rot.Cells[i].Y;
                _rows[row] |= 1 << x;
            }
            ClearFullRows(_rows);
        }

        private void EnsureRows(List<int> rows, int count)
        {
            while (rows.Count < count) rows.Add(0);
        }

        private int ClearFullRows(List<int> rows)
        {
            int full = (1 << _width) - 1;
            int cleared = 0;
            for (int i = rows.Count - 1; i >= 0; i--)
            {
                if (rows[i] == full)
                {
                    rows.RemoveAt(i);
                    cleared++;
                }
            }
            return cleared;
        }

        private int WellDepth(int[] heights)
        {
            int sum = 0;
            for (int x = 0; x < _width; x++)
            {
                int left = x == 0 ? heights[x] + 2 : heights[x - 1];
                int right = x == _width - 1 ? heights[x] + 2 : heights[x + 1];
                int minNeighbor = Math.Min(left, right);
                if (minNeighbor > heights[x]) sum += minNeighbor - heights[x];
            }
            return sum;
        }

        private int DropYOnRows(List<int> rows, Rotation rot, int left)
        {
            int y = 0;
            while (true)
            {
                if (!CanPlaceOnRows(rows, rot, left, y)) return y - 1;
                y++;
            }
        }

        private bool CanPlaceOnRows(List<int> rows, Rotation rot, int left, int bottom)
        {
            if (left < 0 || left + rot.Width > _width) return false;
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int x = left + rot.Cells[i].X;
                int y = bottom + rot.Cells[i].Y;
                if (y < 0) return false;
                if (y < rows.Count && (rows[y] & (1 << x)) != 0) return false;
            }
            return true;
        }

        private int[] GetHeights(List<int> rows)
        {
            var h = new int[_width];
            for (int x = 0; x < _width; x++)
            {
                for (int y = rows.Count - 1; y >= 0; y--)
                {
                    if ((rows[y] & (1 << x)) != 0) { h[x] = y + 1; break; }
                }
            }
            return h;
        }

        private int CountHoles(List<int> rows)
        {
            int holes = 0;
            for (int x = 0; x < _width; x++)
            {
                bool seenBlock = false;
                for (int y = rows.Count - 1; y >= 0; y--)
                {
                    bool filled = (rows[y] & (1 << x)) != 0;
                    if (filled) seenBlock = true;
                    else if (seenBlock) holes++;
                }
            }
            return holes;
        }

        private int SurfaceRoughness(List<int> rows)
        {
            var h = GetHeights(rows);
            int rough = 0;
            for (int i = 1; i < h.Length; i++) rough += Math.Abs(h[i] - h[i - 1]);
            return rough;
        }
    }

    private readonly struct HeuristicWeights
    {
        public readonly int LineClear;
        public readonly int AggregateHeight;
        public readonly int Holes;
        public readonly int Roughness;
        public readonly int MaxHeight;
        public readonly int WellDepth;

        public HeuristicWeights(int lineClear, int aggregateHeight, int holes, int roughness, int maxHeight, int wellDepth)
        {
            LineClear = lineClear;
            AggregateHeight = aggregateHeight;
            Holes = holes;
            Roughness = roughness;
            MaxHeight = maxHeight;
            WellDepth = wellDepth;
        }
    }

    private static readonly Dictionary<char, Rotation[]> Pieces = BuildPieces();

    public static void Main()
    {
        var fs = new FastScanner();
        string? first = fs.Next();
        if (first == null) return;

        int width = int.Parse(first);
        int height = fs.NextInt();
        int numPieces = fs.NextInt();
        var board = new Board(width);
        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        for (int i = 0; i < numPieces; i++)
        {
            string? token = fs.Next();
            if (token == null) break;
            char piece = token[0];
            var rotations = Pieces[piece];

            bool complexBoard = width >= 14 || board.MaxHeight() >= height / 2 || board.CountHoles() > width;
            var weights = complexBoard
                ? new HeuristicWeights(9000, 14, 120, 6, 20, 8)
                : new HeuristicWeights(12000, 10, 140, 8, 24, 10);

            int bestScore = int.MinValue;
            int bestRot = 0;
            int bestCol = 0;
            int legalCount = 0;

            for (int r = 0; r < rotations.Length; r++)
            {
                var rot = rotations[r];
                for (int col = 0; col + rot.Width <= width; col++)
                {
                    int score = board.SimulateAndScore(rot, col, weights, out int cleared);
                    if (cleared >= 0)
                    {
                        legalCount++;
                        if (score > bestScore)
                        {
                            bestScore = score;
                            bestRot = r;
                            bestCol = col;
                        }
                    }
                }
            }

            if (legalCount == 0)
            {
                writer.WriteLine("0 0");
                string? result = fs.Next();
                if (result == null || result == "gameover") break;
                continue;
            }

            writer.WriteLine($"{bestRot} {bestCol}");

            string? outcome = fs.Next();
            if (outcome == null || outcome == "gameover") break;
            board.ApplyPlacement(rotations[bestRot], bestCol);
        }
    }

    private static Dictionary<char, Rotation[]> BuildPieces()
    {
        var dict = new Dictionary<char, Rotation[]>();
        dict['I'] = Unique(new[]
        {
            new Rotation(4, 1, new[] { (0,0), (1,0), (2,0), (3,0) }),
            new Rotation(1, 4, new[] { (0,0), (0,1), (0,2), (0,3) }),
        });
        dict['O'] = Unique(new[]
        {
            new Rotation(2, 2, new[] { (0,0), (1,0), (0,1), (1,1) }),
        });
        dict['T'] = Unique(new[]
        {
            new Rotation(3, 2, new[] { (0,0), (1,0), (2,0), (1,1) }),
            new Rotation(2, 3, new[] { (0,0), (0,1), (0,2), (1,1) }),
            new Rotation(3, 2, new[] { (1,0), (0,1), (1,1), (2,1) }),
            new Rotation(2, 3, new[] { (1,0), (0,1), (1,1), (1,2) }),
        });
        dict['S'] = Unique(new[]
        {
            new Rotation(3, 2, new[] { (1,0), (2,0), (0,1), (1,1) }),
            new Rotation(2, 3, new[] { (0,0), (0,1), (1,1), (1,2) }),
        });
        dict['Z'] = Unique(new[]
        {
            new Rotation(3, 2, new[] { (0,0), (1,0), (1,1), (2,1) }),
            new Rotation(2, 3, new[] { (1,0), (0,1), (1,1), (0,2) }),
        });
        dict['J'] = Unique(new[]
        {
            new Rotation(3, 2, new[] { (0,0), (0,1), (1,1), (2,1) }),
            new Rotation(2, 3, new[] { (0,0), (1,0), (0,1), (0,2) }),
            new Rotation(3, 2, new[] { (0,0), (1,0), (2,0), (2,1) }),
            new Rotation(2, 3, new[] { (1,0), (1,1), (0,2), (1,2) }),
        });
        dict['L'] = Unique(new[]
        {
            new Rotation(3, 2, new[] { (2,0), (0,1), (1,1), (2,1) }),
            new Rotation(2, 3, new[] { (0,0), (0,1), (0,2), (1,2) }),
            new Rotation(3, 2, new[] { (0,0), (1,0), (2,0), (0,1) }),
            new Rotation(2, 3, new[] { (0,0), (1,0), (1,1), (1,2) }),
        });
        return dict;
    }

    private static Rotation[] Unique(Rotation[] rots)
    {
        var list = new List<Rotation>();
        var seen = new HashSet<string>();
        foreach (var r in rots)
        {
            var key = Canon(r);
            if (seen.Add(key)) list.Add(r);
        }
        return list.ToArray();
    }

    private static string Canon(Rotation r)
    {
        var arr = new List<string>(r.Cells.Length);
        foreach (var c in r.Cells) arr.Add($"{c.X}:{c.Y}");
        arr.Sort(StringComparer.Ordinal);
        return r.Width + "x" + r.Height + "|" + string.Join(",", arr);
    }
}
