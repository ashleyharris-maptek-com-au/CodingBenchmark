#include <vector>
using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        size_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x) ^ 0x9e3779b97f4a7c15ull);
            h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return h;
    }
};

struct Node {
    int f, g;
    vector<int> pos;
    bool operator<(const Node& other) const {
        if (f != other.f) return f > other.f;
        return g > other.g;
    }
};

static inline int manhattan(const vector<int>& a, const vector<int>& b) {
    int d = 0;
    for (size_t i = 0; i < a.size(); ++i) d += abs(a[i] - b[i]);
    return d;
}

static inline bool inBounds(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static vector<pair<int,int>> reconstructMoves(
    const vector<int>& start,
    const vector<int>& goal,
    const unordered_map<vector<int>, pair<vector<int>, pair<int,int>>, VecHash>& parent)
{
    vector<pair<int,int>> moves;
    vector<int> cur = goal;
    while (cur != start) {
        auto it = parent.find(cur);
        if (it == parent.end()) break;
        moves.push_back(it->second.second);
        cur = it->second.first;
    }
    reverse(moves.begin(), moves.end());
    return moves;
}

static vector<pair<int,int>> astarPath(
    const vector<int>& start,
    const vector<int>& goal,
    const vector<int>& bounds,
    const unordered_set<vector<int>, VecHash>& blocked)
{
    if (start == goal) return {};
    if (blocked.count(goal)) return {};

    priority_queue<Node> pq;
    unordered_map<vector<int>, int, VecHash> bestG;
    unordered_map<vector<int>, pair<vector<int>, pair<int,int>>, VecHash> parent;

    bestG[start] = 0;
    pq.push({manhattan(start, goal), 0, start});

    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();

        auto itG = bestG.find(cur.pos);
        if (itG == bestG.end() || cur.g != itG->second) continue;
        if (cur.pos == goal) return reconstructMoves(start, goal, parent);

        for (int axis = 0; axis < (int)cur.pos.size(); ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> nxt = cur.pos;
                nxt[axis] += dir;
                if (!inBounds(nxt, bounds)) continue;
                if (blocked.count(nxt) && nxt != goal) continue;
                int ng = cur.g + 1;
                auto it = bestG.find(nxt);
                if (it == bestG.end() || ng < it->second) {
                    bestG[nxt] = ng;
                    parent[nxt] = {cur.pos, {axis, dir}};
                    pq.push({ng + manhattan(nxt, goal), ng, std::move(nxt)});
                }
            }
        }
    }
    return {};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> foods(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) cin >> foods[i][j];
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash> blocked;
    blocked.reserve((size_t)O * 2 + 16);
    for (int i = 0; i < O; ++i) {
        vector<int> p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        blocked.insert(std::move(p));
    }

    vector<pair<int,int>> plan;
    vector<int> cur = head;
    vector<char> used(F, 0);

    while ((int)plan.size() < max_turns) {
        int bestIdx = -1;
        vector<pair<int,int>> bestPath;
        int bestLen = INT_MAX;

        for (int i = 0; i < F; ++i) {
            if (used[i]) continue;
            auto path = astarPath(cur, foods[i], bounds, blocked);
            if (!path.empty() || cur == foods[i]) {
                if ((int)path.size() < bestLen) {
                    bestLen = (int)path.size();
                    bestIdx = i;
                    bestPath = std::move(path);
                }
            }
        }

        if (bestIdx == -1) break;
        used[bestIdx] = 1;
        for (auto mv : bestPath) {
            if ((int)plan.size() >= max_turns) break;
            plan.push_back(mv);
            cur[mv.first] += mv.second;
        }
        if (cur != foods[bestIdx]) {
            // If a food is already at the current position, consume it without moving.
            cur = foods[bestIdx];
        }
    }

    for (auto [axis, dir] : plan) {
        cout << axis << ' ' << dir << '\n';
    }
    return 0;
}
