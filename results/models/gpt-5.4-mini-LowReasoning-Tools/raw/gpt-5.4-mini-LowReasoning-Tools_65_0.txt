declare const require: any;
const fs = require('fs');

type Task = {
  id: string | number;
  duration: number;
  deps: Array<string | number>;
  deadline: number;
  weight: number;
};

type InputData = {
  machineCount: number;
  tasks: Task[];
};

type ReadyItem = {
  idx: number;
  priority1: number;
  priority2: number;
  priority3: number;
};

type RunningItem = {
  finish: number;
  machine: number;
  idx: number;
};

class MaxHeap<T> {
  private a: T[] = [];
  constructor(private cmp: (x: T, y: T) => number) {}
  size(): number { return this.a.length; }
  peek(): T | undefined { return this.a[0]; }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) <= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1;
        let r = l + 1;
        let best = i;
        if (l < a.length && this.cmp(a[l], a[best]) > 0) best = l;
        if (r < a.length && this.cmp(a[r], a[best]) > 0) best = r;
        if (best === i) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
}

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as InputData;

const machineCount = Math.max(1, Math.floor(data.machineCount || 1));
const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

const idToIdx = new Map<string, number>();
for (let i = 0; i < n; i++) idToIdx.set(String(tasks[i].id), i);

const duration = new Int32Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);
const indeg = new Int32Array(n);
const succ: number[][] = Array.from({ length: n }, () => []);
const pred: number[][] = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  duration[i] = Math.max(0, Math.floor(Number(t.duration) || 0));
  deadline[i] = Number.isFinite(Number(t.deadline)) ? Number(t.deadline) : 0;
  weight[i] = Number.isFinite(Number(t.weight)) ? Number(t.weight) : 0;
}

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (const dep of deps) {
    const j = idToIdx.get(String(dep));
    if (j === undefined) continue;
    pred[i].push(j);
    succ[j].push(i);
    indeg[i]++;
  }
}

// Topological pass for critical-path style priority from each task to the end.
const topo: number[] = [];
{
  const q: number[] = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  for (let head = 0; head < q.length; head++) {
    const u = q[head];
    topo.push(u);
    for (const v of succ[u]) {
      indeg[v]--;
      if (indeg[v] === 0) q.push(v);
    }
  }
}

const indeg2 = new Int32Array(n);
for (let i = 0; i < n; i++) indeg2[i] = pred[i].length;
const tail = new Float64Array(n);
for (let k = topo.length - 1; k >= 0; k--) {
  const u = topo[k];
  let best = 0;
  for (const v of succ[u]) {
    const cand = tail[v];
    if (cand > best) best = cand;
  }
  tail[u] = duration[u] + best;
}

const readyCmp = (a: ReadyItem, b: ReadyItem): number => {
  if (a.priority1 !== b.priority1) return a.priority1 - b.priority1;
  if (a.priority2 !== b.priority2) return a.priority2 - b.priority2;
  if (a.priority3 !== b.priority3) return a.priority3 - b.priority3;
  return b.idx - a.idx;
};

const runningCmp = (a: RunningItem, b: RunningItem): number => {
  if (a.finish !== b.finish) return b.finish - a.finish;
  return b.machine - a.machine;
};

const ready = new MaxHeap<ReadyItem>(readyCmp);
const running = new MaxHeap<RunningItem>(runningCmp);
const machineFree: number[] = [];
for (let m = machineCount - 1; m >= 0; m--) machineFree.push(m);

const scheduledStart = new Int32Array(n);
scheduledStart.fill(-1);
const scheduledMachine = new Int32Array(n);
scheduledMachine.fill(-1);
const done = new Uint8Array(n);

for (let i = 0; i < n; i++) {
  if (pred[i].length === 0) {
    const sl = deadline[i] - tail[i];
    ready.push({
      idx: i,
      priority1: Math.round(-sl * 1000),
      priority2: Math.round(-weight[i] * 1000),
      priority3: Math.round(-duration[i] * 1000),
    });
  }
}

let now = 0;
let finishedCount = 0;

function computePriority(idx: number): ReadyItem {
  const slack = deadline[idx] - tail[idx];
  return {
    idx,
    priority1: Math.round(-slack * 1000),
    priority2: Math.round(-weight[idx] * 1000),
    priority3: Math.round(-duration[idx] * 1000),
  };
}

while (finishedCount < n) {
  while (running.size() > 0 && (running.peek()!.finish <= now || machineFree.length === 0 && ready.size() === 0)) {
    const item = running.pop()!;
    now = Math.max(now, item.finish);
    done[item.idx] = 1;
    finishedCount++;
    machineFree.push(item.machine);
    for (const v of succ[item.idx]) {
      indeg2[v]--;
      if (indeg2[v] === 0) ready.push(computePriority(v));
    }
  }

  while (machineFree.length > 0 && ready.size() > 0) {
    const task = ready.pop()!;
    const machine = machineFree.pop()!;
    scheduledStart[task.idx] = now;
    scheduledMachine[task.idx] = machine;
    running.push({
      finish: now + duration[task.idx],
      machine,
      idx: task.idx,
    });
  }

  if (finishedCount >= n) break;
  if (running.size() > 0) {
    const next = running.peek()!.finish;
    if (next > now) now = next;
    continue;
  }
  if (ready.size() > 0 && machineFree.length === 0) {
    continue;
  }
  // Fallback for disconnected/cyclic inputs: advance time so we can still emit a partial schedule.
  break;
}

const schedule = [];
for (let i = 0; i < n; i++) {
  if (scheduledStart[i] >= 0 && scheduledMachine[i] >= 0) {
    schedule.push({
      id: tasks[i].id,
      start: scheduledStart[i],
      machine: scheduledMachine[i],
    });
  }
}

process.stdout.write(JSON.stringify({ schedule }));
