declare const require: any;
declare const process: any;
const fs = require('fs');

type Point = { id: string; x: number; y: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depotX: number = data.depot[0];
const depotY: number = data.depot[1];
const truckCount: number = Math.max(1, data.truckCount | 0);
const packages: Point[] = Array.isArray(data.packages) ? data.packages.map((p: any) => ({
  id: String(p.id),
  x: Number(p.x),
  y: Number(p.y),
})) : [];

const n = packages.length;

function dist2(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return dx * dx + dy * dy;
}

function dist(ax: number, ay: number, bx: number, by: number): number {
  return Math.sqrt(dist2(ax, ay, bx, by));
}

function routeCost(route: Point[]): number {
  let cost = 0;
  let px = depotX, py = depotY;
  for (const p of route) {
    cost += dist(px, py, p.x, p.y);
    px = p.x; py = p.y;
  }
  cost += dist(px, py, depotX, depotY);
  return cost;
}

function improve2Opt(route: Point[]): Point[] {
  if (route.length < 4) return route;
  let improved = true;
  while (improved) {
    improved = false;
    for (let i = 0; i < route.length - 1; i++) {
      const a = i === 0 ? null : route[i - 1];
      const b = route[i];
      for (let k = i + 1; k < route.length; k++) {
        const c = route[k];
        const d = k + 1 < route.length ? route[k + 1] : null;
        const before =
          (a ? dist(a.x, a.y, b.x, b.y) : dist(depotX, depotY, b.x, b.y)) +
          (d ? dist(c.x, c.y, d.x, d.y) : dist(c.x, c.y, depotX, depotY));
        const after =
          (a ? dist(a.x, a.y, c.x, c.y) : dist(depotX, depotY, c.x, c.y)) +
          (d ? dist(b.x, b.y, d.x, d.y) : dist(b.x, b.y, depotX, depotY));
        if (after + 1e-12 < before) {
          const segment = route.slice(i, k + 1).reverse();
          route.splice(i, k - i + 1, ...segment);
          improved = true;
        }
      }
      if (improved) break;
    }
  }
  return route;
}

function sweepSeed(): Point[] {
  return [...packages].sort((a, b) => {
    const angA = Math.atan2(a.y - depotY, a.x - depotX);
    const angB = Math.atan2(b.y - depotY, b.x - depotX);
    if (angA !== angB) return angA - angB;
    const da = dist2(depotX, depotY, a.x, a.y);
    const db = dist2(depotX, depotY, b.x, b.y);
    return da - db;
  });
}

function seedByFarthest(): Point[] {
  if (n === 0) return [];
  const chosen: Point[] = [];
  const used = new Set<number>();
  let first = 0;
  let best = -1;
  for (let i = 0; i < n; i++) {
    const d = dist2(depotX, depotY, packages[i].x, packages[i].y);
    if (d > best) {
      best = d;
      first = i;
    }
  }
  chosen.push(packages[first]);
  used.add(first);
  while (chosen.length < Math.min(truckCount, n)) {
    let pick = -1;
    let pickScore = -1;
    for (let i = 0; i < n; i++) {
      if (used.has(i)) continue;
      let nearest = Infinity;
      for (const c of chosen) {
        const d = dist2(packages[i].x, packages[i].y, c.x, c.y);
        if (d < nearest) nearest = d;
      }
      if (nearest > pickScore) {
        pickScore = nearest;
        pick = i;
      }
    }
    if (pick < 0) break;
    chosen.push(packages[pick]);
    used.add(pick);
  }
  return chosen;
}

function buildRoutes(order: Point[]): Point[][] {
  if (n === 0) return [];
  const k = Math.min(truckCount, n);
  const routes: Point[][] = Array.from({ length: k }, () => []);
  const routeLoad: number[] = new Array(k).fill(0);

  const seeds = seedByFarthest();
  const assigned = new Set<string>();

  for (let i = 0; i < seeds.length; i++) {
    routes[i].push(seeds[i]);
    routeLoad[i] += 1;
    assigned.add(seeds[i].id);
  }

  for (const p of order) {
    if (assigned.has(p.id)) continue;
    let bestRoute = 0;
    let bestDelta = Infinity;
    for (let r = 0; r < k; r++) {
      const route = routes[r];
      let delta = 0;
      if (route.length === 0) {
        delta = 2 * dist(depotX, depotY, p.x, p.y);
      } else {
        let insertionBest = Infinity;
        for (let i = 0; i <= route.length; i++) {
          const prev = i === 0 ? null : route[i - 1];
          const next = i === route.length ? null : route[i];
          const removed = (prev ? dist(prev.x, prev.y, next ? next.x : depotX, next ? next.y : depotY) : 0);
          const added = (prev ? dist(prev.x, prev.y, p.x, p.y) : dist(depotX, depotY, p.x, p.y)) +
            (next ? dist(p.x, p.y, next.x, next.y) : dist(p.x, p.y, depotX, depotY));
          const d = added - removed;
          if (d < insertionBest) insertionBest = d;
        }
        delta = insertionBest;
      }
      const loadPenalty = routeLoad[r] === 0 ? 0.25 : 0;
      const score = delta + loadPenalty;
      if (score < bestDelta) {
        bestDelta = score;
        bestRoute = r;
      }
    }
    routes[bestRoute].push(p);
    routeLoad[bestRoute] += 1;
  }

  for (const route of routes) {
    route.sort((a, b) => {
      const aa = Math.atan2(a.y - depotY, a.x - depotX);
      const ab = Math.atan2(b.y - depotY, b.x - depotX);
      if (aa !== ab) return aa - ab;
      return dist2(depotX, depotY, a.x, a.y) - dist2(depotX, depotY, b.x, b.y);
    });
    improve2Opt(route);
  }

  return routes.filter(r => r.length > 0);
}

const routes = buildRoutes(sweepSeed());
const output = { routes: routes.map(route => route.map(p => p.id)) };
process.stdout.write(JSON.stringify(output));
