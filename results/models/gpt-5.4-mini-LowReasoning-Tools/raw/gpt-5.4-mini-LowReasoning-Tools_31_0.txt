use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

fn read_input() -> Vec<i64> {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut nums = Vec::new();
    let mut i = 0usize;
    while i < input.len() {
        while i < input.len() && input[i].is_ascii_whitespace() {
            i += 1;
        }
        if i >= input.len() {
            break;
        }
        let mut sign = 1i64;
        if input[i] == b'-' {
            sign = -1;
            i += 1;
        } else if input[i] == b'+' {
            i += 1;
        }
        let mut val = 0i64;
        let mut has_digit = false;
        while i < input.len() && input[i].is_ascii_digit() {
            has_digit = true;
            val = val
                .saturating_mul(10)
                .saturating_add((input[i] - b'0') as i64);
            i += 1;
        }
        if has_digit {
            nums.push(val * sign);
        } else {
            i += 1;
        }
    }
    nums
}

fn reconstruct(parent: &[i32], prev: &[i32], mut sum: usize) -> Vec<usize> {
    let mut out = Vec::new();
    while parent[sum] >= 0 {
        let idx = parent[sum] as usize;
        out.push(idx);
        sum = prev[sum] as usize;
    }
    out.reverse();
    out
}

fn solve_range_dp(items: &[Item], target: i64, min_sum: i64, max_sum: i64) -> Option<Vec<usize>> {
    let offset = -min_sum;
    let width_i64 = max_sum - min_sum;
    let width = usize::try_from(width_i64).ok()?;
    if width > 2_500_000 {
        return None;
    }
    let target_idx_i64 = target + offset;
    if target_idx_i64 < 0 || target_idx_i64 > width_i64 {
        return Some(Vec::new()).filter(|_| false);
    }
    let target_idx = usize::try_from(target_idx_i64).ok()?;

    let mut reachable = vec![false; width + 1];
    let mut parent = vec![-1i32; width + 1];
    let mut prev = vec![-1i32; width + 1];
    let zero = usize::try_from(offset).ok()?;
    reachable[zero] = true;
    parent[zero] = -2;
    prev[zero] = -2;

    for item in items {
        let x = item.value;
        let idx = item.index as i32;
        if x >= 0 {
            let xu = usize::try_from(x).ok()?;
            if xu > width {
                continue;
            }
            for s in (0..=width - xu).rev() {
                if reachable[s] {
                    let ns = s + xu;
                    if !reachable[ns] {
                        reachable[ns] = true;
                        parent[ns] = idx;
                        prev[ns] = s as i32;
                    }
                }
            }
        } else {
            let xu = usize::try_from(-x).ok()?;
            if xu > width {
                continue;
            }
            for s in xu..=width {
                if reachable[s] {
                    let ns = s - xu;
                    if !reachable[ns] {
                        reachable[ns] = true;
                        parent[ns] = idx;
                        prev[ns] = s as i32;
                    }
                }
            }
        }
    }

    if !reachable[target_idx] {
        return Some(Vec::new()).filter(|_| false);
    }
    Some(reconstruct(&parent, &prev, target_idx))
}

fn subset_sum_mitm(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    let mid = n / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let mut left_sums: Vec<(i64, u64)> = Vec::with_capacity(1usize << left.len().min(22));
    let lcount = left.len();
    if lcount > 24 {
        return None;
    }
    for mask in 0u64..(1u64 << lcount) {
        let mut sum = 0i64;
        let mut m = mask;
        let mut idx = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                sum += left[idx].value;
            }
            idx += 1;
            m >>= 1;
        }
        left_sums.push((sum, mask));
    }
    left_sums.sort_unstable_by_key(|&(s, _)| s);

    let rcount = right.len();
    if rcount > 24 {
        return None;
    }
    for mask in 0u64..(1u64 << rcount) {
        let mut sum = 0i64;
        let mut m = mask;
        let mut idx = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                sum += right[idx].value;
            }
            idx += 1;
            m >>= 1;
        }
        let need = target - sum;
        if let Ok(pos) = left_sums.binary_search_by_key(&need, |&(s, _)| s) {
            let mut ans = Vec::new();
            let lmask = left_sums[pos].1;
            for i in 0..lcount {
                if ((lmask >> i) & 1) != 0 {
                    ans.push(left[i].index);
                }
            }
            for i in 0..rcount {
                if ((mask >> i) & 1) != 0 {
                    ans.push(right[i].index);
                }
            }
            ans.sort_unstable();
            return Some(ans);
        }
    }
    None
}

fn main() {
    let nums = read_input();
    if nums.len() < 2 {
        println!("NO");
        return;
    }

    let n = nums[0].max(0) as usize;
    let target = nums[1];
    if n == 0 {
        if target == 0 {
            println!("YES");
            println!();
        } else {
            println!("NO");
        }
        return;
    }
    if nums.len() < 2 + n {
        println!("NO");
        return;
    }

    let mut items = Vec::with_capacity(n);
    let mut min_sum = 0i64;
    let mut max_sum = 0i64;
    for i in 0..n {
        let v = nums[2 + i];
        items.push(Item { value: v, index: i });
        if v >= 0 {
            max_sum = max_sum.saturating_add(v);
        } else {
            min_sum = min_sum.saturating_add(v);
        }
    }

    if target == 0 {
        println!("YES");
        println!();
        return;
    }

    for item in &items {
        if item.value == target {
            println!("YES");
            println!("{}", item.index);
            return;
        }
    }

    if n <= 40 {
        if let Some(mut ans) = subset_sum_mitm(&items, target) {
            println!("YES");
            ans.sort_unstable();
            if ans.is_empty() {
                println!();
            } else {
                for (i, idx) in ans.iter().enumerate() {
                    if i > 0 {
                        print!(" ");
                    }
                    print!("{}", idx);
                }
                println!();
            }
            return;
        }
    }

    if let Some(mut ans) = solve_range_dp(&items, target, min_sum, max_sum) {
        println!("YES");
        ans.sort_unstable();
        if ans.is_empty() {
            println!();
        } else {
            for (i, idx) in ans.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!();
        }
        return;
    }

    let mut positive = 0i64;
    let mut negative = 0i64;
    let mut best: HashMap<i64, usize> = HashMap::new();
    best.insert(0, usize::MAX);
    let mut parents: HashMap<i64, (i64, usize)> = HashMap::new();
    for item in &items {
        let mut next = best.clone();
        for (&sum, _) in &best {
            let ns = sum.saturating_add(item.value);
            if !next.contains_key(&ns) {
                next.insert(ns, item.index);
                parents.insert(ns, (sum, item.index));
            }
        }
        best = next;
        if best.contains_key(&target) {
            let mut cur = target;
            let mut ans = Vec::new();
            while cur != 0 {
                if let Some(&(prev, idx)) = parents.get(&cur) {
                    ans.push(idx);
                    cur = prev;
                } else {
                    break;
                }
            }
            ans.sort_unstable();
            println!("YES");
            if ans.is_empty() {
                println!();
            } else {
                for (i, idx) in ans.iter().enumerate() {
                    if i > 0 {
                        print!(" ");
                    }
                    print!("{}", idx);
                }
                println!();
            }
            return;
        }
        if item.value >= 0 {
            positive = positive.saturating_add(item.value);
        } else {
            negative = negative.saturating_add(item.value);
        }
        if (best.len() > 200_000 && item.index > 1000) || positive - negative > 5_000_000 {
            break;
        }
    }

    if best.contains_key(&target) {
        println!("YES");
        let mut cur = target;
        let mut ans = Vec::new();
        while cur != 0 {
            if let Some(&(prev, idx)) = parents.get(&cur) {
                ans.push(idx);
                cur = prev;
            } else {
                break;
            }
        }
        ans.sort_unstable();
        if ans.is_empty() {
            println!();
        } else {
            for (i, idx) in ans.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!();
        }
    } else {
        println!("NO");
    }
}
