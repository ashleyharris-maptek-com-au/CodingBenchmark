#include <algorithm>
#include <atomic>
#include <cmath>
#include <cstdint>
#include <fstream>
#include <iostream>
#include <limits>
#include <mutex>
#include <sstream>
#include <string>
#include <thread>

using std::clamp;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static double norm(const Vec3& v) {
    return std::sqrt(dot(v, v));
}

static Vec3 normalized(const Vec3& v) {
    double n = norm(v);
    if (n <= 1e-12) return {0.0, 0.0, 1.0};
    return v * (1.0 / n);
}

static double wrap_pi(double a) {
    while (a > 3.14159265358979323846) a -= 2.0 * 3.14159265358979323846;
    while (a < -3.14159265358979323846) a += 2.0 * 3.14159265358979323846;
    return a;
}

struct State {
    Vec3 pos;
    Vec3 vel;
    double pitch = 0.0, yaw = 0.0, roll = 0.0;
    double pitch_rate = 0.0, yaw_rate = 0.0, roll_rate = 0.0;
    double fuel = 0.0;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int width = 0, depth = 0, height = 0;
    double gravity = 1.62;
    double max_thrust = 1.0;
    double fuel_start = 0.0;
    double start_x = 0.0, start_y = 0.0, start_z = 0.0;
    double target_x = 0.0, target_y = 0.0, target_z = 0.0;

    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel_start)) {
        return 0;
    }
    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 0;
    }
    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    std::string state_marker;
    std::cin >> state_marker; // "STATE"

    std::atomic<bool> terrain_ready{false};
    std::thread terrain_thread([terrain_path, &terrain_ready]() {
        std::ifstream file(terrain_path, std::ios::binary);
        if (file) {
            char buffer[1 << 16];
            while (file.read(buffer, sizeof(buffer)) || file.gcount() > 0) {
                // Discard terrain contents; the controller does not need them
                // for the baseline guidance strategy.
            }
        }
        terrain_ready.store(true, std::memory_order_release);
    });
    terrain_thread.detach();

    const Vec3 target{target_x, target_y, target_z};
    const double g = gravity;
    const double maxA = std::max(0.1, max_thrust);

    while (true) {
        State s;
        if (!(std::cin >> s.pos.x >> s.pos.y >> s.pos.z
                      >> s.vel.x >> s.vel.y >> s.vel.z
                      >> s.pitch >> s.yaw >> s.roll
                      >> s.pitch_rate >> s.yaw_rate >> s.roll_rate
                      >> s.fuel)) {
            break;
        }

        Vec3 to_target = target - s.pos;
        double horiz_dist = std::sqrt(to_target.x * to_target.x + to_target.y * to_target.y);

        // Horizontal guidance: brake hard when close, otherwise move toward the pad.
        Vec3 desired_horiz_vel{0.0, 0.0, 0.0};
        if (horiz_dist > 1e-6) {
            Vec3 dir = {to_target.x / horiz_dist, to_target.y / horiz_dist, 0.0};
            double max_speed = clamp(horiz_dist * 0.35, 0.0, 18.0);
            if (horiz_dist < 40.0) max_speed = clamp(horiz_dist * 0.20, 0.0, 8.0);
            if (horiz_dist < 15.0) max_speed = clamp(horiz_dist * 0.10, 0.0, 3.0);
            desired_horiz_vel = dir * max_speed;
        }

        // Vertical guidance: descend aggressively far away, then slow down for touchdown.
        double desired_vz = 0.0;
        if (to_target.z > 120.0) desired_vz = -12.0;
        else if (to_target.z > 60.0) desired_vz = -7.0;
        else if (to_target.z > 25.0) desired_vz = -4.0;
        else if (to_target.z > 10.0) desired_vz = -2.0;
        else if (to_target.z > 4.0) desired_vz = -0.8;
        else desired_vz = -0.25;

        // Simple velocity controller translated into required thrust acceleration.
        const double k_h = 0.9;
        const double k_z = 1.3;
        Vec3 accel_cmd{
            k_h * (desired_horiz_vel.x - s.vel.x),
            k_h * (desired_horiz_vel.y - s.vel.y),
            k_z * (desired_vz - s.vel.z)
        };

        // Compensate lunar gravity.
        Vec3 required_thrust_accel{accel_cmd.x, accel_cmd.y, accel_cmd.z + g};
        double thrust_mag = norm(required_thrust_accel);

        Vec3 thrust_dir = normalized(required_thrust_accel);
        double thrust = clamp(thrust_mag / maxA, 0.0, 1.0);

        // If we are nearly on target and slow, keep a gentle hover instead of overshooting.
        if (horiz_dist < 2.0 && std::abs(to_target.z) < 2.0 && norm(s.vel) < 2.5) {
            thrust_dir = {0.0, 0.0, 1.0};
            thrust = clamp(g / maxA, 0.0, 1.0);
        }

        // Convert desired thrust direction to pitch/yaw targets.
        double horiz_len = std::sqrt(thrust_dir.x * thrust_dir.x + thrust_dir.y * thrust_dir.y);
        double yaw_des = std::atan2(thrust_dir.y, -thrust_dir.x); // yaw=0 means forward is -X
        double pitch_des = std::atan2(horiz_len, thrust_dir.z);

        double yaw_err = wrap_pi(yaw_des - s.yaw);
        double pitch_err = wrap_pi(pitch_des - s.pitch);

        // PD attitude control. Roll is kept level.
        double pitch_cmd = clamp(2.2 * pitch_err - 0.7 * s.pitch_rate, -1.0, 1.0);
        double yaw_cmd = clamp(2.2 * yaw_err - 0.7 * s.yaw_rate, -1.0, 1.0);
        double roll_cmd = clamp(-1.8 * s.roll - 0.6 * s.roll_rate, -1.0, 1.0);

        // When almost aligned and landing, reduce oscillation by softening attitude inputs.
        if (horiz_dist < 8.0 && to_target.z < 15.0) {
            pitch_cmd = clamp(pitch_cmd, -0.6, 0.6);
            yaw_cmd = clamp(yaw_cmd, -0.6, 0.6);
        }

        std::cout.setf(std::ios::fixed);
        std::cout.precision(6);
        std::cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n';
        std::cout.flush();
    }

    return 0;
}
