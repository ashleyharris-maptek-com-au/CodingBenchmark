float4 main(PSInput input) : SV_Target0
{
    float2 brickUV = input.uv * float2(8.0, 16.0);
    float row = floor(brickUV.y);
    float rowOffset = fmod(row, 2.0) * 0.5;

    float2 cellUV = float2(frac(brickUV.x + rowOffset), frac(brickUV.y));

    float mortarX = 0.05;
    float mortarY = 0.10;

    bool isMortar = (cellUV.x < mortarX) || (cellUV.x > (1.0 - mortarX)) ||
                    (cellUV.y < mortarY) || (cellUV.y > (1.0 - mortarY));

    float3 baseColor = isMortar ? float3(0.7, 0.7, 0.7) : float3(0.7, 0.2, 0.1);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    return float4(baseColor * shade, 1.0);
}
