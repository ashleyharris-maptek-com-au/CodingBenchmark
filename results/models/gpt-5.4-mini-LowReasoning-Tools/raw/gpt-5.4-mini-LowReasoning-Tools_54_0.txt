from __future__ import annotations

from typing import Dict, Any, List, Tuple


_phase = "approach"
_go_around = False
_flare_timer = 0.0
_touchdown_timer = 0.0
_last_radio_alt = None
_last_gs_error = 0.0
_last_loc_error = 0.0
_unstable_time = 0.0


def _valid(state: Dict[str, Any], key: str, default=None):
    v = state.get(key, default)
    return v


def _truthy(v) -> bool:
    return bool(v)


def _mean(vals: List[float], default: float = 0.0) -> float:
    vals = [v for v in vals if v is not None]
    return sum(vals) / len(vals) if vals else default


def _median(vals: List[float], default: float = 0.0) -> float:
    vals = sorted(v for v in vals if v is not None)
    if not vals:
        return default
    n = len(vals)
    m = n // 2
    if n % 2:
        return vals[m]
    return 0.5 * (vals[m - 1] + vals[m])


def _pick_valid_triplet(state: Dict[str, Any], base: str) -> List[float]:
    out = []
    for i in (1, 2, 3):
        if _truthy(state.get(f"{base}_{i}_valid", True)):
            out.append(state.get(f"{base}_{i}"))
    return [v for v in out if isinstance(v, (int, float))]


def _wrap_angle_deg(a: float) -> float:
    while a > 180:
        a -= 360
    while a < -180:
        a += 360
    return a


def _sensor_vote(state: Dict[str, Any], prefix: str, suffix: str = "") -> float:
    vals = []
    for i in (1, 2, 3):
        key = f"{prefix}_{i}{suffix}"
        valid_key = f"{prefix}_{i}_valid"
        if suffix and not suffix.startswith("_"):
            key = f"{prefix}_{i}_{suffix}"
        if _truthy(state.get(valid_key, True)):
            v = state.get(key)
            if isinstance(v, (int, float)):
                vals.append(float(v))
    return _median(vals, default=0.0)


def _clamp(x: float, lo: float, hi: float) -> float:
    return max(lo, min(hi, x))


def _command(**kwargs):
    return {k: v for k, v in kwargs.items() if v is not None}


def autopilot_step(state: dict, dt: float) -> dict:
    global _phase, _go_around, _flare_timer, _touchdown_timer
    global _last_radio_alt, _last_gs_error, _last_loc_error, _unstable_time

    # Fused inputs
    gs = _sensor_vote(state, "glideslope", "_dots")
    loc = _sensor_vote(state, "localizer", "_dots")
    ias = _median([
        state.get("ias_1_kt") if state.get("ias_1_valid", True) else None,
        state.get("ias_2_kt") if state.get("ias_2_valid", True) else None,
        state.get("ias_3_kt") if state.get("ias_3_valid", True) else None,
    ], default=state.get("target_ias_kt", 140.0))
    aoa = _median([
        state.get("aoa_1_deg") if state.get("aoa_1_valid", True) else None,
        state.get("aoa_2_deg") if state.get("aoa_2_valid", True) else None,
        state.get("aoa_3_deg") if state.get("aoa_3_valid", True) else None,
    ], default=0.0)
    pitch = _median([
        state.get("irs_1_pitch_deg") if state.get("irs_1_valid", True) else None,
        state.get("irs_2_pitch_deg") if state.get("irs_2_valid", True) else None,
        state.get("irs_3_pitch_deg") if state.get("irs_3_valid", True) else None,
    ], default=0.0)
    roll = _median([
        state.get("irs_1_roll_deg") if state.get("irs_1_valid", True) else None,
        state.get("irs_2_roll_deg") if state.get("irs_2_valid", True) else None,
        state.get("irs_3_roll_deg") if state.get("irs_3_valid", True) else None,
    ], default=0.0)
    heading = _median([
        state.get("irs_1_heading_deg") if state.get("irs_1_valid", True) else None,
        state.get("irs_2_heading_deg") if state.get("irs_2_valid", True) else None,
        state.get("irs_3_heading_deg") if state.get("irs_3_valid", True) else None,
    ], default=state.get("target_hdg_deg", 0.0))
    track = _median([
        state.get("irs_1_track_deg") if state.get("irs_1_valid", True) else None,
        state.get("irs_2_track_deg") if state.get("irs_2_valid", True) else None,
        state.get("irs_3_track_deg") if state.get("irs_3_valid", True) else None,
    ], default=heading)
    vs = _median([
        state.get("irs_1_vs_fpm") if state.get("irs_1_valid", True) else None,
        state.get("irs_2_vs_fpm") if state.get("irs_2_valid", True) else None,
        state.get("irs_3_vs_fpm") if state.get("irs_3_valid", True) else None,
    ], default=0.0)
    radio_alt = state.get("radio_alt_ft")
    if not isinstance(radio_alt, (int, float)):
        radio_alt = _median([state.get("gps_1_alt_ft"), state.get("gps_2_alt_ft"), state.get("gps_3_alt_ft")], default=state.get("target_alt_ft", 0.0))
    target_ias = float(state.get("target_ias_kt", state.get("target_vref_kt", 137.0)))
    target_vref = float(state.get("target_vref_kt", 137.0))
    target_gs = float(state.get("target_gs_dots", 0.0))
    target_loc = float(state.get("target_loc_dots", 0.0))
    target_alt = float(state.get("target_alt_ft", 0.0))
    on_ground = _truthy(state.get("on_ground", False)) or _truthy(state.get("weight_on_wheels", False))
    wow = _truthy(state.get("weight_on_wheels", False))
    below_dh = _truthy(state.get("below_decision_height", False))

    # Detect unstable or unsafe conditions.
    gs_err = gs - target_gs
    loc_err = loc - target_loc
    speed_err = ias - target_ias
    unstable = (
        abs(loc_err) > 1.75 and radio_alt > 200
        or abs(gs_err) > 1.5 and radio_alt > 200
        or speed_err > 25 and radio_alt > 200
        or speed_err < -20 and radio_alt > 50
        or abs(roll) > 20 and radio_alt > 100
        or abs(pitch) > 12 and radio_alt > 100
        or (below_dh and radio_alt > 30 and abs(loc_err) > 0.8)
    )
    _unstable_time = _unstable_time + dt if unstable else max(0.0, _unstable_time - 2.0 * dt)

    if _unstable_time > 4.0 and radio_alt > 50 and not on_ground:
        _go_around = True

    if _go_around and not on_ground:
        _phase = "goaround"

    # Phase logic
    if on_ground:
        _phase = "rollout"
    elif radio_alt is not None and radio_alt < 50:
        _phase = "flare"
    elif radio_alt is not None and radio_alt < 200:
        _phase = "approach"
    else:
        _phase = "intercept"

    # Default config
    cmd = {
        "elevator": 0.0,
        "aileron": 0.0,
        "rudder": 0.0,
        "throttle_1": 0.55,
        "throttle_2": 0.55,
        "elevator_trim": 0.0,
        "flaps": 30.0,
        "gear": 1.0,
        "speedbrake": 0.0,
        "spoiler": 0.0,
        "autobrake": 0.0,
        "reverse_1": 0.0,
        "reverse_2": 0.0,
    }

    # Energy management and handling asymmetries.
    flap_target = 30.0 if target_ias < 170 else 20.0
    if radio_alt < 1500:
        flap_target = max(flap_target, 20.0)
    if radio_alt < 1000:
        flap_target = max(flap_target, 25.0)
    if radio_alt < 500:
        flap_target = max(flap_target, 30.0)
    if radio_alt < 100:
        flap_target = 40.0
    cmd["flaps"] = flap_target
    cmd["gear"] = 1.0

    # Crosswind / localizer correction: aileron for bank, rudder to align track with heading.
    track_heading_err = _wrap_angle_deg(track - heading)
    cmd["rudder"] = _clamp(-track_heading_err / 20.0, -1.0, 1.0)
    cmd["aileron"] = _clamp(-loc_err * 0.35 - roll / 25.0, -1.0, 1.0)
    if abs(loc_err) > 1.0:
        cmd["aileron"] = _clamp(cmd["aileron"] - 0.15 * (1 if loc_err > 0 else -1), -1.0, 1.0)

    # Thrust schedule.
    if _phase == "goaround":
        cmd["throttle_1"] = 1.0
        cmd["throttle_2"] = 1.0
        cmd["speedbrake"] = 0.0
        cmd["elevator"] = 0.35
        cmd["elevator_trim"] = 0.15
        cmd["gear"] = 1.0
        cmd["flaps"] = max(cmd["flaps"], 20.0)
        return _command(**cmd)

    if _phase == "rollout":
        cmd["throttle_1"] = 0.0
        cmd["throttle_2"] = 0.0
        cmd["spoiler"] = 1.0
        cmd["autobrake"] = 1.0 if ias > 60 else 0.7
        cmd["reverse_1"] = 1.0 if ias > 40 else 0.5
        cmd["reverse_2"] = 1.0 if ias > 40 else 0.5
        cmd["speedbrake"] = 1.0
        cmd["elevator"] = 0.15
        cmd["elevator_trim"] = 0.05
        return _command(**cmd)

    if _phase == "flare":
        _flare_timer += dt
        cmd["throttle_1"] = 0.12
        cmd["throttle_2"] = 0.12
        cmd["speedbrake"] = 0.0
        flare_strength = 0.18 + max(0.0, (50.0 - radio_alt) / 50.0) * 0.22
        cmd["elevator"] = _clamp(flare_strength + max(0.0, -vs) / 5000.0, 0.0, 0.55)
        cmd["elevator_trim"] = _clamp(0.08 + (target_vref - ias) / 200.0, -0.2, 0.2)
        if radio_alt < 15:
            cmd["throttle_1"] = 0.0
            cmd["throttle_2"] = 0.0
        return _command(**cmd)

    if _phase == "approach":
        cmd["throttle_1"] = _clamp(0.55 + speed_err / 120.0, 0.18, 0.82)
        cmd["throttle_2"] = _clamp(0.55 + speed_err / 120.0, 0.18, 0.82)
        # Glideslope control: below path -> pitch up, above path -> pitch down.
        cmd["elevator"] = _clamp(-gs_err * 0.45 + (target_vref - ias) / 140.0 - pitch / 60.0, -0.35, 0.35)
        cmd["elevator_trim"] = _clamp((target_vref - ias) / 250.0 - gs_err * 0.08, -0.25, 0.25)
        if radio_alt < 300:
            cmd["flaps"] = max(cmd["flaps"], 30.0)
        if radio_alt < 80 and ias < target_vref + 15:
            cmd["speedbrake"] = 0.0
        return _command(**cmd)

    # Intercept / high altitude capture.
    cmd["throttle_1"] = _clamp(0.65 + speed_err / 150.0, 0.25, 0.9)
    cmd["throttle_2"] = _clamp(0.65 + speed_err / 150.0, 0.25, 0.9)
    cmd["elevator"] = _clamp(-gs_err * 0.25 + (target_vref + 10.0 - ias) / 180.0, -0.3, 0.3)
    cmd["elevator_trim"] = _clamp((target_vref - ias) / 220.0, -0.2, 0.2)
    cmd["speedbrake"] = 0.0
    if radio_alt > 2500:
        cmd["flaps"] = 10.0
    elif radio_alt > 1800:
        cmd["flaps"] = 15.0
    else:
        cmd["flaps"] = 20.0

    # Add a small correction for excessive sink or speed excursions.
    if vs < -1800:
        cmd["elevator"] = max(cmd["elevator"], 0.25)
        cmd["throttle_1"] = max(cmd["throttle_1"], 0.8)
        cmd["throttle_2"] = max(cmd["throttle_2"], 0.8)
    if ias > target_ias + 20:
        cmd["speedbrake"] = 0.5
    if radio_alt < 3000 and abs(loc_err) > 0.5:
        cmd["rudder"] = _clamp(cmd["rudder"] - 0.1 * (1 if loc_err > 0 else -1), -1.0, 1.0)

    _last_radio_alt = radio_alt
    _last_gs_error = gs_err
    _last_loc_error = loc_err
    return _command(**cmd)
