use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Pt3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

fn add3(a: Pt3, b: Pt3) -> Pt3 {
    Pt3 { x: a.x + b.x, y: a.y + b.y, z: a.z + b.z }
}

fn mul3(a: Pt3, s: f64) -> Pt3 {
    Pt3 { x: a.x * s, y: a.y * s, z: a.z * s }
}

fn dot3(a: Pt3, b: Pt3) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

fn norm3(a: Pt3) -> f64 {
    dot3(a, a).sqrt()
}

fn normalize3(a: Pt3) -> Pt3 {
    let n = norm3(a);
    if n == 0.0 {
        Pt3 { x: 0.0, y: 0.0, z: -1.0 }
    } else {
        mul3(a, 1.0 / n)
    }
}

fn sub2(a: Pt2, b: Pt2) -> Pt2 {
    Pt2 { x: a.x - b.x, y: a.y - b.y }
}

fn cross2(a: Pt2, b: Pt2) -> f64 {
    a.x * b.y - a.y * b.x
}

fn rotate_vec(q: Quat, p: Pt3) -> Pt3 {
    let u = Pt3 { x: q.x, y: q.y, z: q.z };
    let s = q.w;
    let uv = cross3(u, p);
    let uuv = cross3(u, uv);
    add3(p, add3(mul3(uv, 2.0 * s), mul3(uuv, 2.0)))
}

fn cross3(a: Pt3, b: Pt3) -> Pt3 {
    Pt3 {
        x: a.y * b.z - a.z * b.y,
        y: a.z * b.x - a.x * b.z,
        z: a.x * b.y - a.y * b.x,
    }
}

fn quat_axis_angle(axis: Pt3, angle: f64) -> Quat {
    let a = normalize3(axis);
    let h = angle * 0.5;
    let s = h.sin();
    Quat { w: h.cos(), x: a.x * s, y: a.y * s, z: a.z * s }
}

fn quat_mul(a: Quat, b: Quat) -> Quat {
    Quat {
        w: a.w * b.w - a.x * b.x - a.y * b.y - a.z * b.z,
        x: a.w * b.x + a.x * b.w + a.y * b.z - a.z * b.y,
        y: a.w * b.y - a.x * b.z + a.y * b.w + a.z * b.x,
        z: a.w * b.z + a.x * b.y - a.y * b.x + a.z * b.w,
    }
}

fn project_shadow(p: Pt3, sun: Pt3) -> Pt2 {
    let t = p.z / sun.z;
    Pt2 { x: p.x - t * sun.x, y: p.y - t * sun.y }
}

fn convex_hull(mut pts: Vec<Pt2>) -> Vec<Pt2> {
    pts.sort_by(|a, b| a.x.total_cmp(&b.x).then(a.y.total_cmp(&b.y)));
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1e-12 && (a.y - b.y).abs() < 1e-12);
    if pts.len() <= 1 {
        return pts;
    }
    let mut lower: Vec<Pt2> = Vec::new();
    for p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            let a = lower[l - 2];
            let b = lower[l - 1];
            if cross2(sub2(b, a), sub2(*p, b)) <= 1e-15 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(*p);
    }
    let mut upper: Vec<Pt2> = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            let a = upper[l - 2];
            let b = upper[l - 1];
            if cross2(sub2(b, a), sub2(*p, b)) <= 1e-15 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(*p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn polygon_center(poly: &[Pt2]) -> Pt2 {
    let mut sx = 0.0;
    let mut sy = 0.0;
    for p in poly {
        sx += p.x;
        sy += p.y;
    }
    let n = poly.len() as f64;
    Pt2 { x: sx / n, y: sy / n }
}

fn dist_point_edge(p: Pt2, a: Pt2, b: Pt2) -> f64 {
    let ab = sub2(b, a);
    let ap = sub2(p, a);
    cross2(ab, ap).abs() / (ab.x * ab.x + ab.y * ab.y).sqrt()
}

fn inradius(poly: &[Pt2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let c = polygon_center(poly);
    let mut r = f64::INFINITY;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        r = r.min(dist_point_edge(c, a, b));
    }
    r
}

fn candidate_quaternions() -> Vec<Quat> {
    let mut qs = Vec::new();
    qs.push(Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 });
    let axes = [
        Pt3 { x: 1.0, y: 0.0, z: 0.0 },
        Pt3 { x: 0.0, y: 1.0, z: 0.0 },
        Pt3 { x: 0.0, y: 0.0, z: 1.0 },
    ];
    for axis in axes {
        for &deg in &[PI * 0.5, PI, PI * 1.5] {
            qs.push(quat_axis_angle(axis, deg));
        }
    }
    let r1 = quat_mul(quat_axis_angle(axes[2], PI * 0.25), quat_axis_angle(axes[0], PI * 0.25));
    let r2 = quat_mul(quat_axis_angle(axes[1], PI * 0.25), quat_axis_angle(axes[2], PI * 0.25));
    let r3 = quat_mul(quat_axis_angle(axes[0], PI * 0.25), quat_axis_angle(axes[1], PI * 0.25));
    qs.push(r1);
    qs.push(r2);
    qs.push(r3);
    qs
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = it.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(nv);
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        minx = minx.min(x);
        maxx = maxx.max(x);
        miny = miny.min(y);
        maxy = maxy.max(y);
        poly.push(Pt2 { x, y });
    }
    let sun = normalize3(Pt3 {
        x: it.next().unwrap().parse().unwrap(),
        y: it.next().unwrap().parse().unwrap(),
        z: it.next().unwrap().parse().unwrap(),
    });
    let base = [
        Pt3 { x: 1.0, y: 0.0, z: -0.707 },
        Pt3 { x: -1.0, y: 0.0, z: -0.707 },
        Pt3 { x: 0.0, y: 1.0, z: 0.707 },
        Pt3 { x: 0.0, y: -1.0, z: 0.707 },
    ];

    let mut best_q = Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 };
    let mut _best_poly: Vec<Pt2> = Vec::new();
    let mut best_r = -1.0;
    for q in candidate_quaternions() {
        let mut pts = Vec::with_capacity(4);
        for v in base {
            let p = rotate_vec(q, v);
            pts.push(project_shadow(p, sun));
        }
        let hull = convex_hull(pts);
        let r = inradius(&hull);
        if r > best_r {
            best_r = r;
            best_q = q;
            _best_poly = hull;
        }
    }
    if best_r <= 1e-9 {
        best_r = 1.0;
    }
    let scale = 10.0;
    let shadow_r = best_r * scale;
    let step = (shadow_r * 1.25).max(1e-3);
    let margin = shadow_r * 1.1;
    let mut out: Vec<(Pt3, Quat, f64)> = Vec::new();
    let mut k: i64 = 0;
    let mut y = miny - margin;
    while y <= maxy + margin {
        let mut x = minx - margin;
        while x <= maxx + margin {
            let shift = mul3(sun, 1000.0 * k as f64);
            let pos = add3(Pt3 { x, y, z: 0.0 }, shift);
            out.push((pos, best_q, scale));
            k += 1;
            x += step;
        }
        y += step;
    }

    if out.is_empty() {
        let c = polygon_center(&poly);
        out.push((
            Pt3 { x: c.x, y: c.y, z: 0.0 },
            best_q,
            scale,
        ));
    }

    println!("{}", out.len());
    for (p, q, s) in out {
        println!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            p.x, p.y, p.z, q.w, q.x, q.y, q.z, s
        );
    }
}
