const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget | 0;

const n = width * height;
const passable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    passable[y * width + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const ignitionSet = new Uint8Array(n);
const ignitionIds = [];
for (const p of ignitions) {
  const x = p[0] | 0, y = p[1] | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const id = y * width + x;
  if (!passable[id]) continue;
  if (!ignitionSet[id]) {
    ignitionSet[id] = 1;
    ignitionIds.push(id);
  }
}

const townValue = new Float64Array(n);
for (const t of towns) {
  const x = t.x | 0, y = t.y | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const id = y * width + x;
  townValue[id] += Number(t.value) || 0;
}

const reachable = new Uint8Array(n);
const q = new Int32Array(n);
let qh = 0, qt = 0;
for (const s of ignitionIds) {
  if (!reachable[s]) {
    reachable[s] = 1;
    q[qt++] = s;
  }
}
while (qh < qt) {
  const u = q[qh++];
  const x = u % width;
  const y = (u / width) | 0;
  if (x > 0) {
    const v = u - 1;
    if (passable[v] && !reachable[v]) { reachable[v] = 1; q[qt++] = v; }
  }
  if (x + 1 < width) {
    const v = u + 1;
    if (passable[v] && !reachable[v]) { reachable[v] = 1; q[qt++] = v; }
  }
  if (y > 0) {
    const v = u - width;
    if (passable[v] && !reachable[v]) { reachable[v] = 1; q[qt++] = v; }
  }
  if (y + 1 < height) {
    const v = u + width;
    if (passable[v] && !reachable[v]) { reachable[v] = 1; q[qt++] = v; }
  }
}

const visited = new Uint8Array(n + 1); // extra slot for virtual source
const disc = new Int32Array(n + 1);
const low = new Int32Array(n + 1);
const parent = new Int32Array(n + 1);
const tin = new Int32Array(n + 1);
const tout = new Int32Array(n + 1);
const artChildCounts = new Int32Array(n + 1);
const artFlags = new Uint8Array(n + 1);
const eulerToCell = [];
const stackNode = [];
const stackNext = [];
const stackEntered = [];
const stackRootChildCount = [];
let time = 0;

function pushFrame(u, nextIdx, entered) {
  stackNode.push(u);
  stackNext.push(nextIdx);
  stackEntered.push(entered);
  stackRootChildCount.push(0);
}

// Virtual source is node n. It connects to all ignition cells.
visited[n] = 1;
disc[n] = low[n] = ++time;
tin[n] = time;
eulerToCell[time] = n;
pushFrame(n, 0, true);

while (stackNode.length) {
  const top = stackNode.length - 1;
  const u = stackNode[top];
  let nextIdx = stackNext[top];
  let entered = stackEntered[top];

  if (!entered) {
    visited[u] = 1;
    disc[u] = low[u] = ++time;
    tin[u] = time;
    eulerToCell[time] = u;
    stackEntered[top] = true;
  }

  let advanced = false;
  while (true) {
    let v = -1;
    if (u === n) {
      if (nextIdx < ignitionIds.length) v = ignitionIds[nextIdx++];
    } else {
      const x = u % width;
      const y = (u / width) | 0;
      switch (nextIdx) {
        case 0:
          if (x > 0) v = u - 1;
          break;
        case 1:
          if (x + 1 < width) v = u + 1;
          break;
        case 2:
          if (y > 0) v = u - width;
          break;
        case 3:
          if (y + 1 < height) v = u + width;
          break;
      }
      nextIdx++;
    }

    stackNext[top] = nextIdx;
    if (v < 0) break;
    if (u !== n && !reachable[v]) continue;
    if (v === parent[u]) continue;

    if (!visited[v]) {
      parent[v] = u;
      if (u === n) stackRootChildCount[top]++;
      pushFrame(v, 0, false);
      advanced = true;
      break;
    } else {
      if (disc[v] < disc[u]) {
        if (disc[v] < low[u]) low[u] = disc[v];
      }
    }
  }

  if (advanced) continue;

  // Finish node.
  stackNode.pop();
  stackNext.pop();
  stackEntered.pop();
  const rootChildren = stackRootChildCount.pop();

  tout[u] = time;
  if (u !== n) {
    const p = parent[u];
    if (p !== 0 || u === ignitionIds[0]) {
      if (low[u] < low[p]) low[p] = low[u];
      if (p !== n && low[u] >= disc[p]) {
        artFlags[p] = 1;
      }
    }
    if (p === n) {
      artChildCounts[n] = rootChildren;
    }
  }
}

// Gather candidate cuts from articulation points.
const candidates = [];
for (let u = 0; u < n; u++) {
  if (!visited[u] || !reachable[u]) continue;
  if (!artFlags[u] && artChildCounts[u] === 0) continue;
  if (!passable[u] || ignitionSet[u] || townValue[u] > 0) continue; // cut only plain forest cells

  let total = 0;
  const intervals = [];
  const ux = u % width;
  const uy = (u / width) | 0;
  const checkNeighbor = (v) => {
    if (!visited[v] || parent[v] !== u) return;
    if (u !== n && low[v] >= disc[u]) {
      intervals.push([tin[v], tout[v]]);
      let sum = 0;
      for (let t = tin[v]; t <= tout[v]; t++) {
        const cell = eulerToCell[t];
        if (cell < n) sum += townValue[cell];
      }
      total += sum;
    }
  };
  if (ux > 0) checkNeighbor(u - 1);
  if (ux + 1 < width) checkNeighbor(u + 1);
  if (uy > 0) checkNeighbor(u - width);
  if (uy + 1 < height) checkNeighbor(u + width);
  if (total > 0) {
    candidates.push({ u, total, intervals });
  }
}

// Fallback: if no articulation candidates, consider cells adjacent to the
// reachable frontier toward the highest-value towns.
if (candidates.length === 0) {
  const frontier = [];
  for (let i = 0; i < n; i++) {
    if (!reachable[i] || !passable[i] || ignitionSet[i] || townValue[i] > 0) continue;
    let score = 0;
    const x = i % width;
    const y = (i / width) | 0;
    const add = (v) => {
      if (v >= 0 && v < n && reachable[v] && townValue[v] > 0) score += townValue[v];
    };
    if (x > 0) add(i - 1);
    if (x + 1 < width) add(i + 1);
    if (y > 0) add(i - width);
    if (y + 1 < height) add(i + width);
    if (score > 0) frontier.push({ u: i, total: score, intervals: [[i, i]] });
  }
  frontier.sort((a, b) => b.total - a.total);
  candidates.push(...frontier);
}

candidates.sort((a, b) => b.total - a.total);

// Fenwick tree over Euler positions containing town values that are not yet saved.
const size = time + 2;
const bit = new Float64Array(size + 1);
function bitAdd(i, delta) {
  for (let x = i; x <= size; x += x & -x) bit[x] += delta;
}
function bitSum(i) {
  let s = 0;
  for (let x = i; x > 0; x -= x & -x) s += bit[x];
  return s;
}
function rangeSum(l, r) {
  if (r < l) return 0;
  return bitSum(r) - bitSum(l - 1);
}

const eulerTownPos = [];
for (let pos = 1; pos <= time; pos++) {
  const cell = eulerToCell[pos];
  if (cell < n && townValue[cell] > 0) {
    eulerTownPos.push(pos);
    bitAdd(pos, townValue[cell]);
  }
}

// Disjoint-set next-pointer over town positions for fast deletion.
const posIndex = new Int32Array(size + 2);
for (let i = 0; i < eulerTownPos.length; i++) posIndex[eulerTownPos[i]] = i + 1;
const nextPos = new Int32Array(eulerTownPos.length + 2);
for (let i = 0; i <= eulerTownPos.length + 1; i++) nextPos[i] = i;
function findPos(i) {
  let x = i;
  while (nextPos[x] !== x) {
    nextPos[x] = nextPos[nextPos[x]];
    x = nextPos[x];
  }
  return x;
}
function eraseTownPositions(l, r) {
  let idx = findPos(lowerBound(eulerTownPos, l));
  const end = upperBound(eulerTownPos, r);
  while (idx < end) {
    const pos = eulerTownPos[idx];
    const cell = eulerToCell[pos];
    if (cell < n && townValue[cell] > 0) {
      bitAdd(pos, -townValue[cell]);
      townValue[cell] = 0;
    }
    nextPos[idx] = idx + 1;
    idx = findPos(idx);
  }
}
function lowerBound(arr, target) {
  let l = 0, r = arr.length;
  while (l < r) {
    const m = (l + r) >> 1;
    if (arr[m] < target) l = m + 1; else r = m;
  }
  return l;
}
function upperBound(arr, target) {
  let l = 0, r = arr.length;
  while (l < r) {
    const m = (l + r) >> 1;
    if (arr[m] <= target) l = m + 1; else r = m;
  }
  return l;
}

const chosen = [];
const used = new Uint8Array(n);
for (const cand of candidates) {
  if (chosen.length >= cutBudget) break;
  if (used[cand.u]) continue;
  let marginal = 0;
  for (const [l, r] of cand.intervals) marginal += rangeSum(l, r);
  if (marginal <= 0) continue;
  chosen.push(cand.u);
  used[cand.u] = 1;
  for (const [l, r] of cand.intervals) {
    eraseTownPositions(l, r);
  }
}

// If budget remains, use a simple greedy frontier around still-reachable towns.
if (chosen.length < cutBudget) {
  const extras = [];
  for (let i = 0; i < n; i++) {
    if (!reachable[i] || !passable[i] || ignitionSet[i] || townValue[i] > 0 || used[i]) continue;
    let score = 0;
    const x = i % width;
    const y = (i / width) | 0;
    if (x > 0) score += townValue[i - 1] || 0;
    if (x + 1 < width) score += townValue[i + 1] || 0;
    if (y > 0) score += townValue[i - width] || 0;
    if (y + 1 < height) score += townValue[i + width] || 0;
    if (score > 0) extras.push({ i, score });
  }
  extras.sort((a, b) => b.score - a.score);
  for (const e of extras) {
    if (chosen.length >= cutBudget) break;
    chosen.push(e.i);
  }
}

const firebreaks = chosen.map(id => [id % width, (id / width) | 0]);
process.stdout.write(JSON.stringify({ firebreaks }));
