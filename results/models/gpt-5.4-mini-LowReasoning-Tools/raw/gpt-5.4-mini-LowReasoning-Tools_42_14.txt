OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_3 = OpConstant %uint 3
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_02 = OpConstant %float 0.2
%float_08 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_33333334 = OpConstant %float 0.33333334

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_3
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_arr_pv = OpTypePointer Output %arr_pv_out
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_in_arr_v3 = OpTypePointer Input %v3float
%arr_v3_in = OpTypeArray %v3float %uint_6
%arr_v3_out = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3_t = OpTypePointer Input %arr_v3_in
%ptr_out_arr_v3_t = OpTypePointer Output %arr_v3_out

%worldPos = OpVariable %ptr_in_arr_v3_t Input
%normal = OpVariable %ptr_in_arr_v3_t Input
%color = OpVariable %ptr_in_arr_v3_t Input
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_arr_pv Output
%outColor = OpVariable %ptr_out_arr_v3_t Output

%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %funct
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2 = OpLoad %v4float %p2_ptr
%p4 = OpLoad %v4float %p4_ptr

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2
%p4x = OpCompositeExtract %float %p4 0
%p4y = OpCompositeExtract %float %p4 1
%p4z = OpCompositeExtract %float %p4 2

%a_x = OpFSub %float %p2x %p0x
%a_y = OpFSub %float %p2y %p0y
%a_z = OpFSub %float %p2z %p0z
%b_x = OpFSub %float %p4x %p0x
%b_y = OpFSub %float %p4y %p0y
%b_z = OpFSub %float %p4z %p0z

%nx1 = OpFMul %float %a_y %b_z
%nx2 = OpFMul %float %a_z %b_y
%nx = OpFSub %float %nx1 %nx2
%ny1 = OpFMul %float %a_z %b_x
%ny2 = OpFMul %float %a_x %b_z
%ny = OpFSub %float %ny1 %ny2
%nz1 = OpFMul %float %a_x %b_y
%nz2 = OpFMul %float %a_y %b_x
%nz = OpFSub %float %nz1 %nz2
%face_vec = OpCompositeConstruct %v3float %nx %ny %nz
%face_n = OpExtInst %v3float %glsl Normalize %face_vec

%l_raw = OpConstantComposite %v3float %float_1 %float_1 %float_1
%light_dir = OpExtInst %v3float %glsl Normalize %l_raw
%diff0 = OpDot %float %face_n %light_dir
%diffuse = OpExtInst %float %glsl FMax %diff0 %float_0
%lit_scale_1 = OpFMul %float %float_08 %diffuse
%lit_scale = OpFAdd %float %float_02 %lit_scale_1

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%c_sum1 = OpFAdd %v3float %c0 %c2
%c_sum2 = OpFAdd %v3float %c_sum1 %c4
%base = OpVectorTimesScalar %v3float %c_sum2 %float_33333334
%final_color = OpVectorTimesScalar %v3float %base %lit_scale

%o0_p = OpAccessChain %ptr_out_v4 %gl_out %int_0 %int_0
%o0_c = OpAccessChain %ptr_out_v3 %outColor %int_0
OpStore %o0_p %p0
OpStore %o0_c %final_color
OpEmitVertex

%o1_p = OpAccessChain %ptr_out_v4 %gl_out %int_1 %int_0
%o1_c = OpAccessChain %ptr_out_v3 %outColor %int_1
OpStore %o1_p %p2
OpStore %o1_c %final_color
OpEmitVertex

%o2_p = OpAccessChain %ptr_out_v4 %gl_out %int_2 %int_0
%o2_c = OpAccessChain %ptr_out_v3 %outColor %int_2
OpStore %o2_p %p4
OpStore %o2_c %final_color
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
