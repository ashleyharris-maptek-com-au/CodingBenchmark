declare const require: any;
const fs = require('fs');

type Town = { id: string | number; x: number; y: number; value: number };
type InputData = {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const W = data.width | 0;
const H = data.height | 0;
const grid = data.grid;
const N = W * H;
const cutBudget = Math.max(0, data.cutBudget | 0);

const burnable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < W; x++) {
    burnable[y * W + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const ignitionSet = new Uint8Array(N);
const ignitionCells: number[] = [];
for (const [x, y] of data.ignitions) {
  if (x >= 0 && x < W && y >= 0 && y < H) {
    const id = y * W + x;
    if (burnable[id] && !ignitionSet[id]) {
      ignitionSet[id] = 1;
      ignitionCells.push(id);
    }
  }
}

const townValueAt = new Float64Array(N);
for (const t of data.towns) {
  if (t.x >= 0 && t.x < W && t.y >= 0 && t.y < H) {
    const id = t.y * W + t.x;
    townValueAt[id] += t.value;
  }
}

const dist = new Int32Array(N);
dist.fill(-1);
const parent = new Int32Array(N);
parent.fill(-1);
const q = new Int32Array(N);
let qh = 0, qt = 0;
for (const s of ignitionCells) {
  dist[s] = 0;
  parent[s] = s;
  q[qt++] = s;
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];
while (qh < qt) {
  const v = q[qh++];
  const vx = v % W;
  const vy = (v / W) | 0;
  const nd = dist[v] + 1;
  for (let k = 0; k < 4; k++) {
    const nx = vx + dx[k];
    const ny = vy + dy[k];
    if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
    const u = ny * W + nx;
    if (!burnable[u] || dist[u] !== -1) continue;
    dist[u] = nd;
    parent[u] = v;
    q[qt++] = u;
  }
}

const score = new Float64Array(N);
const touched = new Uint8Array(N);
const candidateList: number[] = [];

for (const t of data.towns) {
  if (t.x < 0 || t.x >= W || t.y < 0 || t.y >= H) continue;
  const tid = t.y * W + t.x;
  if (dist[tid] < 0) continue;

  const value = t.value;
  const path: number[] = [];
  let cur = tid;
  while (true) {
    path.push(cur);
    const p = parent[cur];
    if (p === cur || p < 0) break;
    cur = p;
  }

  const len = path.length;
  if (len <= 1) continue;
  for (let i = 1; i < len; i++) {
    const cell = path[i];
    if (ignitionSet[cell]) continue;
    const weight = value / (i + 1);
    score[cell] += weight;
    if (!touched[cell]) {
      touched[cell] = 1;
      candidateList.push(cell);
    }
  }
}

// Add some global pressure on cells close to many burnable towns, even if the
// chosen shortest paths are not perfect.
for (let i = 0; i < N; i++) {
  if (!burnable[i] || ignitionSet[i]) continue;
  const tv = townValueAt[i];
  if (tv > 0) {
    score[i] += tv * 2;
    if (!touched[i]) {
      touched[i] = 1;
      candidateList.push(i);
    }
  }
}

candidateList.sort((a, b) => score[b] - score[a] || a - b);

const chosen: number[] = [];
const chosenSet = new Uint8Array(N);
for (const cell of candidateList) {
  if (chosen.length >= cutBudget) break;
  if (score[cell] <= 0) break;
  if (!burnable[cell] || ignitionSet[cell] || chosenSet[cell]) continue;
  chosenSet[cell] = 1;
  chosen.push(cell);
}

// If we still have budget, place cuts on the hottest frontier cells near fire.
if (chosen.length < cutBudget) {
  const frontier: { cell: number; s: number }[] = [];
  for (let i = 0; i < N; i++) {
    if (!burnable[i] || ignitionSet[i] || chosenSet[i]) continue;
    if (dist[i] < 0) continue;
    let nearFire = false;
    const x = i % W;
    const y = (i / W) | 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dx[k];
      const ny = y + dy[k];
      if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
      const u = ny * W + nx;
      if (dist[u] === 0) {
        nearFire = true;
        break;
      }
    }
    if (nearFire) frontier.push({ cell: i, s: score[i] });
  }
  frontier.sort((a, b) => b.s - a.s || a.cell - b.cell);
  for (const f of frontier) {
    if (chosen.length >= cutBudget) break;
    if (chosenSet[f.cell] || f.s <= 0) continue;
    chosenSet[f.cell] = 1;
    chosen.push(f.cell);
  }
}

const firebreaks = chosen.map((id) => [id % W, (id / W) | 0]);
process.stdout.write(JSON.stringify({ firebreaks }));
