%void = OpTypeVoid

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Function_v3float = OpTypePointer Function %v3float
%ptr_Function_v2float = OpTypePointer Function %v2float
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_int = OpTypePointer Function %int
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9

%main = OpFunction %void None %functype
%label = OpLabel
%worldPosVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%uvVal = OpLoad %v2float %uv
%lightPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %light4 %light4 0 1 2
%nrm = OpExtInst %v3float %1 Normalize %normalVal
%toLight = OpFSub %v3float %lightPos %worldPosVal
%lightDir = OpExtInst %v3float %1 Normalize %toLight
%diffuseDot = OpDot %float %nrm %lightDir
%diffuse = OpExtInst %float %1 FMax %diffuseDot %float_0
%shadeMul = OpFMul %float %float_0_5 %diffuse
%shade = OpFAdd %float %float_0_5 %shadeMul
%uvScaled = OpVectorTimesScalar %v2float %uvVal %float_8
%uvFloor = OpExtInst %v2float %1 Floor %uvScaled
%x = OpCompositeExtract %float %uvFloor 0
%y = OpCompositeExtract %float %uvFloor 1
%sumx = OpConvertFToS %int %x
%sumy = OpConvertFToS %int %y
%sum = OpIAdd %int %sumx %sumy
%parity = OpBitwiseAnd %int %sum %int_1
%isEven = OpIEqual %bool %parity %int_0
%checker = OpSelect %float %isEven %float_0_9 %float_0_2
%base = OpFMul %float %checker %shade
%rgb = OpCompositeConstruct %v3float %base %base %base
%out = OpCompositeConstruct %v4float %base %base %base %float_1
OpStore %outColor %out
OpReturn
