OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "proj"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %_arr_gl_PerVertex_in_uint_2 ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16

OpMemberDecorate %UBO 2 MatrixStride 16

OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %_arr_v3float_uint_2 ArrayStride 16

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_015 = OpConstant %float 0.015
%float_05 = OpConstant %float 0.05
%float_075 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%_arr_gl_PerVertex_in_uint_2 = OpTypeArray %gl_PerVertex_in %uint_2
%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %_arr_gl_PerVertex_in_uint_2
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%_arr_v3float_uint_2 = OpTypeArray %v3float %uint_2
%_ptr_Input_arr_v3float = OpTypePointer Input %_arr_v3float_uint_2
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO

%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input
%worldPos = OpVariable %_ptr_Input_arr_v3float Input
%normal = OpVariable %_ptr_Input_arr_v3float Input
%color = OpVariable %_ptr_Input_arr_v3float Input
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %functype
%entry = OpLabel

%p0_ptr = OpAccessChain %_ptr_Input_v3float %worldPos %int_0
%p1_ptr = OpAccessChain %_ptr_Input_v3float %worldPos %int_1
%c0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%c1_ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%dir = OpFSub %v3float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0
%perp = OpExtInst %v3float %glsl Normalize %perp_raw
%dir75 = OpVectorTimesScalar %v3float %dir %float_075
%mid = OpFAdd %v3float %p0 %dir75

%shaft_off = OpVectorTimesScalar %v3float %perp %float_015
%arrow_off = OpVectorTimesScalar %v3float %perp %float_05

%shaft_v0 = OpFAdd %v3float %p0 %shaft_off
%shaft_v1 = OpFSub %v3float %p0 %shaft_off
%shaft_v2 = OpFAdd %v3float %mid %shaft_off
%shaft_v3 = OpFSub %v3float %mid %shaft_off
%arrow_v0 = OpFAdd %v3float %mid %arrow_off
%arrow_v1 = OpFSub %v3float %mid %arrow_off

%pos0 = OpCompositeConstruct %v4float %shaft_v0 %float_1
OpStore %outColor %c0
%gl_pos0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos0 %pos0
OpEmitVertex

%pos1 = OpCompositeConstruct %v4float %shaft_v1 %float_1
OpStore %outColor %c0
%gl_pos1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos1 %pos1
OpEmitVertex

%pos2 = OpCompositeConstruct %v4float %shaft_v2 %float_1
OpStore %outColor %c0
%gl_pos2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos2 %pos2
OpEmitVertex

%pos3 = OpCompositeConstruct %v4float %shaft_v3 %float_1
OpStore %outColor %c0
%gl_pos3 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos3 %pos3
OpEmitVertex
OpEndPrimitive

%apos0 = OpCompositeConstruct %v4float %arrow_v0 %float_1
OpStore %outColor %c1
%gl_pos4 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos4 %apos0
OpEmitVertex

%apos1 = OpCompositeConstruct %v4float %arrow_v1 %float_1
OpStore %outColor %c1
%gl_pos5 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos5 %apos1
OpEmitVertex

%apos2 = OpCompositeConstruct %v4float %p1 %float_1
OpStore %outColor %c1
%gl_pos6 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_pos6 %apos2
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
