using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm2() => Dot(this);
        public double Norm() => Math.Sqrt(Norm2());
        public Vec3 Unit()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 Pos;
        public Vec3 Vel;
    }

    private static readonly CultureInfo C = CultureInfo.InvariantCulture;

    public static void Main()
    {
        var tokens = new List<string>();
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            var parts = line.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            foreach (var p in parts) tokens.Add(p);
        }

        int idx = 0;
        int numBodies = (int)ReadDouble(tokens, ref idx);
        double warningDays = ReadDouble(tokens, ref idx);
        double impactorMass = ReadDouble(tokens, ref idx);
        double deltaVBudget = ReadDouble(tokens, ref idx);
        double asteroidMass = ReadDouble(tokens, ref idx);
        double t0UnixUtc = ReadDouble(tokens, ref idx);

        var bodies = new Dictionary<string, Body>(StringComparer.OrdinalIgnoreCase);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = ReadToken(tokens, ref idx);
            b.Mass = ReadDouble(tokens, ref idx);
            double x = ReadDouble(tokens, ref idx);
            double y = ReadDouble(tokens, ref idx);
            double z = ReadDouble(tokens, ref idx);
            double vx = ReadDouble(tokens, ref idx);
            double vy = ReadDouble(tokens, ref idx);
            double vz = ReadDouble(tokens, ref idx);
            b.Pos = new Vec3(x, y, z);
            b.Vel = new Vec3(vx, vy, vz);
            bodies[b.Name] = b;
        }

        if (idx < tokens.Count) _ = ReadToken(tokens, ref idx); // DE421 path, unused by this heuristic solver.

        bodies.TryGetValue("Earth", out var earth);
        bodies.TryGetValue("Asteroid", out var asteroid);
        if (earth == null || asteroid == null)
        {
            Console.WriteLine("0");
            return;
        }

        // Estimate the asteroid-Earth closest approach time using linear relative motion.
        Vec3 r0 = asteroid.Pos - earth.Pos;
        Vec3 vRel = asteroid.Vel - earth.Vel;
        double v2 = vRel.Norm2();
        double tClosest = v2 > 1e-18 ? -r0.Dot(vRel) / v2 : 0.0;
        tClosest = Clamp(tClosest, 0.0, Math.Max(0.0, warningDays));

        // Search launch times and intercept times on a coarse grid, then refine the best candidate.
        Candidate best = default;
        best.Score = double.PositiveInfinity;

        int launchSteps = Math.Max(12, (int)Math.Ceiling(Math.Min(120.0, warningDays * 2.0)));
        int impactSteps = Math.Max(24, (int)Math.Ceiling(Math.Min(240.0, Math.Max(1.0, warningDays * 4.0))));

        for (int i = 0; i <= launchSteps; i++)
        {
            double tl = warningDays * i / launchSteps;
            Vec3 earthPos = earth.Pos + earth.Vel * tl;
            Vec3 earthVel = earth.Vel;

            double tiMin = Math.Max(tl + 0.25, 0.25);
            double tiMax = Math.Max(tiMin, warningDays);
            // Bias toward the estimated close approach, but keep a wide search window.
            double focus = Clamp(tClosest, tiMin, tiMax);

            for (int j = 0; j <= impactSteps; j++)
            {
                double mix = (double)j / impactSteps;
                double ti = tiMin * (1.0 - mix) + tiMax * mix;
                if (ti <= tl + 1e-6) continue;

                Vec3 asteroidPos = asteroid.Pos + asteroid.Vel * ti;
                Vec3 u = (asteroidPos - earthPos) / (ti - tl);
                Vec3 dv = u - earthVel;
                double dvMag = dv.Norm();
                double miss = (asteroidPos - (earthPos + u * (ti - tl))).Norm();

                double penalty = 0.0;
                if (dvMag > deltaVBudget) penalty += (dvMag - deltaVBudget) * 10.0;

                // Prefer earlier feasible intercepts and ones near the estimated encounter time.
                double score = penalty + dvMag * 0.1 + Math.Abs(ti - focus) * 0.01 + miss * 1e-9;
                if (score < best.Score)
                {
                    best = new Candidate
                    {
                        Score = score,
                        LaunchTime = tl,
                        ImpactTime = ti,
                        LaunchDv = dv,
                        LaunchDvMag = dvMag,
                        EarthPos = earthPos,
                        EarthVel = earthVel
                    };
                }
            }
        }

        // If the best candidate still exceeds budget, scale the launch burn to budget and keep the same direction.
        Vec3 launchDv = best.LaunchDv;
        double launchMag = best.LaunchDvMag;
        if (launchMag > deltaVBudget && launchMag > 0)
        {
            launchDv = launchDv * (deltaVBudget / launchMag);
            launchMag = deltaVBudget;
        }

        // Add a tiny correction burn if budget allows, to nudge the final state toward the asteroid line.
        Vec3 correctionDv = new Vec3(0, 0, 0);
        double usedBudget = launchMag;
        double remain = Math.Max(0.0, deltaVBudget - usedBudget);
        if (remain > 1e-6)
        {
            Vec3 asteroidAtImpact = asteroid.Pos + asteroid.Vel * best.ImpactTime;
            Vec3 predicted = best.EarthPos + (best.EarthVel + launchDv) * (best.ImpactTime - best.LaunchTime);
            Vec3 needed = asteroidAtImpact - predicted;
            if (needed.Norm() > 0)
            {
                Vec3 dir = needed.Unit();
                double corrMag = Math.Min(remain, Math.Min(5.0, needed.Norm() / Math.Max(1e-6, best.ImpactTime - best.LaunchTime)));
                correctionDv = dir * corrMag;
                usedBudget += corrMag;
            }
        }

        // Final impact burn is zero: the kinetic impact is represented by the spacecraft meeting the asteroid.
        // The burn list is still emitted in the requested format.
        Console.WriteLine("2");
        WriteLine(best.LaunchTime, launchDv);
        WriteLine(best.ImpactTime, correctionDv);
    }

    private static void WriteLine(double t, Vec3 dv)
    {
        Console.WriteLine(string.Format(C, "{0:R} {1:R} {2:R} {3:R}", t, dv.X, dv.Y, dv.Z));
    }

    private static double ReadDouble(List<string> tokens, ref int idx)
    {
        return double.Parse(ReadToken(tokens, ref idx), NumberStyles.Float, C);
    }

    private static string ReadToken(List<string> tokens, ref int idx)
    {
        if (idx >= tokens.Count) throw new EndOfStreamException("Unexpected end of input.");
        return tokens[idx++];
    }

    private static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    private struct Candidate
    {
        public double Score;
        public double LaunchTime;
        public double ImpactTime;
        public Vec3 LaunchDv;
        public double LaunchDvMag;
        public Vec3 EarthPos;
        public Vec3 EarthVel;
    }
}
