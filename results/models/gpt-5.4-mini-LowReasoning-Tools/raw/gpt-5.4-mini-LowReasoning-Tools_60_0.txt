const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ order: [] }));
  process.exit(0);
}

const idToIndex = new Map();
for (let i = 0; i < n; i++) {
  idToIndex.set(tasks[i].id, i);
}

const indeg = new Int32Array(n);
const successors = Array.from({ length: n }, () => []);
const duration = new Float64Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  duration[i] = Number.isFinite(t.duration) ? t.duration : 0;
  deadline[i] = Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY;
  weight[i] = Number.isFinite(t.weight) ? t.weight : 1;
}

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  indeg[i] = deps.length;
  for (let k = 0; k < deps.length; k++) {
    const j = idToIndex.get(deps[k]);
    if (j !== undefined) successors[j].push(i);
  }
}

// Reverse critical-path estimate: task duration plus the longest downstream path.
const remaining = new Float64Array(n);
const outdeg = new Int32Array(n);
const revQueue = [];
for (let i = 0; i < n; i++) {
  outdeg[i] = successors[i].length;
  if (outdeg[i] === 0) revQueue.push(i);
}

for (let qh = 0; qh < revQueue.length; qh++) {
  const v = revQueue[qh];
  let best = 0;
  const succ = successors[v];
  for (let i = 0; i < succ.length; i++) {
    const u = succ[i];
    if (remaining[u] > best) best = remaining[u];
  }
  remaining[v] = duration[v] + best;
  const deps = Array.isArray(tasks[v].deps) ? tasks[v].deps : [];
  for (let i = 0; i < deps.length; i++) {
    const p = idToIndex.get(deps[i]);
    if (p !== undefined) {
      outdeg[p]--;
      if (outdeg[p] === 0) revQueue.push(p);
    }
  }
}

class MaxHeap {
  constructor() {
    this.a = [];
  }
  size() {
    return this.a.length;
  }
  better(x, y) {
    if (x.score !== y.score) return x.score > y.score;
    if (x.deadline !== y.deadline) return x.deadline < y.deadline;
    if (x.remaining !== y.remaining) return x.remaining > y.remaining;
    if (x.weight !== y.weight) return x.weight > y.weight;
    if (x.duration !== y.duration) return x.duration > y.duration;
    return x.index < y.index;
  }
  push(v) {
    const a = this.a;
    let i = a.length;
    a.push(v);
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.better(a[p], v)) break;
      a[i] = a[p];
      i = p;
    }
    a[i] = v;
  }
  pop() {
    const a = this.a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const v = a.pop();
    if (n > 1) {
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let c = l;
        if (r < a.length && this.better(a[r], a[l])) c = r;
        if (this.better(v, a[c])) break;
        a[i] = a[c];
        i = c;
      }
      a[i] = v;
    }
    return top;
  }
}

const heap = new MaxHeap();
const totalDuration = duration.reduce((s, x) => s + x, 0) || 1;

for (let i = 0; i < n; i++) {
  if (indeg[i] === 0) {
    const d = deadline[i];
    const urg = Number.isFinite(d) ? Math.max(0, d - duration[i]) : totalDuration * 2;
    const score = remaining[i] * 1e6 + weight[i] * 1e3 - urg;
    heap.push({
      index: i,
      score,
      deadline: d,
      remaining: remaining[i],
      weight: weight[i],
      duration: duration[i],
    });
  }
}

const order = [];
while (heap.size() > 0) {
  const cur = heap.pop();
  const i = cur.index;
  order.push(tasks[i].id);
  const succ = successors[i];
  for (let k = 0; k < succ.length; k++) {
    const j = succ[k];
    indeg[j]--;
    if (indeg[j] === 0) {
      const d = deadline[j];
      const urg = Number.isFinite(d) ? Math.max(0, d - duration[j]) : totalDuration * 2;
      const score = remaining[j] * 1e6 + weight[j] * 1e3 - urg;
      heap.push({
        index: j,
        score,
        deadline: d,
        remaining: remaining[j],
        weight: weight[j],
        duration: duration[j],
      });
    }
  }
}

process.stdout.write(JSON.stringify({ order }));
