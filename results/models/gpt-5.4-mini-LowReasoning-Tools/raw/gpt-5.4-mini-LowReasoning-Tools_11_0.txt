use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn add(self, o: Pt) -> Pt {
        Pt::new(self.x + o.x, self.y + o.y)
    }
    fn sub(self, o: Pt) -> Pt {
        Pt::new(self.x - o.x, self.y - o.y)
    }
    fn rot(self, ang: f64) -> Pt {
        let (s, c) = ang.sin_cos();
        Pt::new(self.x * c - self.y * s, self.x * s + self.y * c)
    }
}

#[derive(Clone)]
struct Poly {
    pts: Vec<Pt>,
    bbox_min: Pt,
    bbox_max: Pt,
    area: f64,
}

impl Poly {
    fn new(mut pts: Vec<Pt>) -> Self {
        if pts.len() >= 2 && pts[0].x == pts[pts.len() - 1].x && pts[0].y == pts[pts.len() - 1].y {
            pts.pop();
        }
        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for p in &pts {
            minx = minx.min(p.x);
            miny = miny.min(p.y);
            maxx = maxx.max(p.x);
            maxy = maxy.max(p.y);
        }
        let mut area = 0.0;
        for i in 0..pts.len() {
            let a = pts[i];
            let b = pts[(i + 1) % pts.len()];
            area += a.x * b.y - a.y * b.x;
        }
        Self {
            pts,
            bbox_min: Pt::new(minx, miny),
            bbox_max: Pt::new(maxx, maxy),
            area: area.abs() * 0.5,
        }
    }
    fn width(&self) -> f64 {
        self.bbox_max.x - self.bbox_min.x
    }
    fn height(&self) -> f64 {
        self.bbox_max.y - self.bbox_min.y
    }
}

fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(b.sub(a), c.sub(a))
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let eps = 1e-9;
    orient(a, b, p).abs() <= eps
        && p.x >= a.x.min(b.x) - eps
        && p.x <= a.x.max(b.x) + eps
        && p.y >= a.y.min(b.y) - eps
        && p.y <= a.y.max(b.y) + eps
}

fn seg_intersect(a1: Pt, a2: Pt, b1: Pt, b2: Pt) -> bool {
    let eps = 1e-9;
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);
    if (o1 > eps && o2 < -eps || o1 < -eps && o2 > eps)
        && (o3 > eps && o4 < -eps || o3 < -eps && o4 > eps)
    {
        return true;
    }
    on_segment(a1, a2, b1) || on_segment(a1, a2, b2) || on_segment(b1, b2, a1) || on_segment(b1, b2, a2)
}

fn point_in_poly(pt: Pt, poly: &Poly) -> bool {
    let mut inside = false;
    let n = poly.pts.len();
    for i in 0..n {
        let a = poly.pts[i];
        let b = poly.pts[(i + 1) % n];
        if on_segment(a, b, pt) {
            return true;
        }
        let intersect = ((a.y > pt.y) != (b.y > pt.y))
            && (pt.x < (b.x - a.x) * (pt.y - a.y) / (b.y - a.y + 0.0) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn transform(poly: &Poly, rot: f64, shift: Pt) -> Vec<Pt> {
    poly.pts.iter().map(|&p| p.rot(rot).add(shift)).collect()
}

fn bbox_of(pts: &[Pt]) -> (Pt, Pt) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in pts {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (Pt::new(minx, miny), Pt::new(maxx, maxy))
}

struct StockInstance {
    cursor_x: f64,
    cursor_y: f64,
    row_h: f64,
}

fn fit_in_stock(stock: &Poly, pts: &[Pt]) -> bool {
    for &p in pts {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    let n = pts.len();
    for i in 0..n {
        let a = pts[i];
        let b = pts[(i + 1) % n];
        for j in 0..stock.pts.len() {
            let c = stock.pts[j];
            let d = stock.pts[(j + 1) % stock.pts.len()];
            if seg_intersect(a, b, c, d) {
                let shared = on_segment(a, b, c) || on_segment(a, b, d) || on_segment(c, d, a) || on_segment(c, d, b);
                if !shared {
                    return false;
                }
            }
        }
    }
    true
}

fn parse_input() -> (Poly, Vec<Poly>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();
    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_pts.push(Pt::new(x, y));
    }
    let stock = Poly::new(stock_pts);
    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Pt::new(x, y));
        }
        pieces.push(Poly::new(pts));
    }
    (stock, pieces)
}

fn candidate_angles(n: usize) -> Vec<f64> {
    if n > 1000 {
        vec![0.0]
    } else if n > 200 {
        vec![0.0, PI / 2.0]
    } else {
        vec![0.0, PI / 6.0, PI / 4.0, PI / 3.0, PI / 2.0]
    }
}

fn main() {
    let (stock, pieces) = parse_input();
    let n = pieces.len();
    let angles = candidate_angles(n);
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&i, &j| pieces[j].area.partial_cmp(&pieces[i].area).unwrap());

    let stock_w = stock.width();
    let stock_h = stock.height();
    let mut stock_instances: Vec<StockInstance> = Vec::new();
    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); n];

    for &idx in &order {
        let piece = &pieces[idx];
        let mut local_angles = angles.clone();
        local_angles.sort_by(|a, b| a.partial_cmp(b).unwrap());
        let pad = if n > 1000 { 1e-6 } else { 1e-4 };
        let mut chosen: Option<(usize, f64, f64, f64, f64, f64)> = None;

        for (sidx, inst) in stock_instances.iter().enumerate() {
            for &ang in &local_angles {
                let rotated = transform(piece, ang, Pt::new(0.0, 0.0));
                let (rmin, rmax) = bbox_of(&rotated);
                let rw = rmax.x - rmin.x;
                let rh = rmax.y - rmin.y;
                if rw > stock_w + 1e-9 || rh > stock_h + 1e-9 {
                    continue;
                }
                let mut x = stock.bbox_min.x - rmin.x + inst.cursor_x;
                let mut y = stock.bbox_min.y - rmin.y + inst.cursor_y;
                if x + rw > stock.bbox_max.x + 1e-9 {
                    x = stock.bbox_min.x - rmin.x;
                    y = stock.bbox_min.y - rmin.y + inst.cursor_y + inst.row_h + pad;
                }
                if y + rh > stock.bbox_max.y + 1e-9 {
                    continue;
                }
                let shifted: Vec<Pt> = rotated.iter().map(|&p| p.add(Pt::new(x, y))).collect();
                if fit_in_stock(&stock, &shifted) {
                    chosen = Some((sidx, x, y, ang, rw, rh));
                    break;
                }
            }
            if chosen.is_some() {
                break;
            }
        }

        let (sidx, x, y, ang, rw, rh) = if let Some(v) = chosen {
            v
        } else {
            let sidx = stock_instances.len();
            stock_instances.push(StockInstance { cursor_x: 0.0, cursor_y: 0.0, row_h: 0.0 });
            let ang = local_angles[0];
            let rotated = transform(piece, ang, Pt::new(0.0, 0.0));
            let (rmin, rmax) = bbox_of(&rotated);
            let rw = rmax.x - rmin.x;
            let rh = rmax.y - rmin.y;
            let x = stock.bbox_min.x - rmin.x;
            let y = stock.bbox_min.y - rmin.y;
            (sidx, x, y, ang, rw, rh)
        };

        placements[idx] = (sidx, x, y, ang);
        if stock_instances.len() <= sidx {
            stock_instances.push(StockInstance { cursor_x: 0.0, cursor_y: 0.0, row_h: 0.0 });
        }
        let inst = &mut stock_instances[sidx];
        if inst.cursor_x + rw + pad > stock_w {
            inst.cursor_x = 0.0;
            inst.cursor_y += inst.row_h + pad;
            inst.row_h = rh;
        } else {
            inst.cursor_x += rw + pad;
            inst.row_h = inst.row_h.max(rh);
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", stock_instances.len().max(1)));
    for (sidx, x, y, ang) in placements {
        out.push_str(&format!("{} {:.10} {:.10} {:.10}\n", sidx, x, y, ang));
    }
    print!("{}", out);
}
