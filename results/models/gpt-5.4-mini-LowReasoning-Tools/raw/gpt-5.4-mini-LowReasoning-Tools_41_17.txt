; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_input_v2 = OpTypePointer Input %v2float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%cfloat0 = OpConstant %float 0
%cfloat1 = OpConstant %float 1
%cfloat05 = OpConstant %float 0.5
%cfloat04 = OpConstant %float 0.4
%cfloat06 = OpConstant %float 0.6
%cfloat07 = OpConstant %float 0.7
%cfloat02 = OpConstant %float 0.2
%cfloat01 = OpConstant %float 0.1
%cfloat005 = OpConstant %float 0.05
%cfloat09 = OpConstant %float 0.9
%cfloat095 = OpConstant %float 0.95
%cfloat08 = OpConstant %float 8
%cfloat16 = OpConstant %float 16
%cint0 = OpConstant %int 0
%cint1 = OpConstant %int 1
%scale = OpConstantComposite %v2float %cfloat08 %cfloat16
%mortarColor = OpConstantComposite %v3float %cfloat07 %cfloat07 %cfloat07
%brickColor = OpConstantComposite %v3float %cfloat07 %cfloat02 %cfloat01

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%uv = OpVariable %ptr_input_v2 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %func
%entry = OpLabel
%uvVal = OpLoad %v2float %uv
%scaledUV = OpFMul %v2float %uvVal %scale

%rowY = OpCompositeExtract %float %scaledUV 1
%rowFloor = OpExtInst %float %glsl Floor %rowY
%rowI = OpConvertFToS %int %rowFloor
%odd = OpBitwiseAnd %int %rowI %cint1
%oddBool = OpINotEqual %bool %odd %cint0
%xShift = OpSelect %float %oddBool %cfloat05 %cfloat0

%xCoord = OpCompositeExtract %float %scaledUV 0
%xShifted = OpFAdd %float %xCoord %xShift
%yShifted = OpFSub %float %rowY %rowFloor
%xFrac = OpExtInst %float %glsl Fract %xShifted
%yFrac = OpExtInst %float %glsl Fract %yShifted

%xLow = OpFOrdLessThan %bool %xFrac %cfloat005
%xHigh = OpFOrdGreaterThan %bool %xFrac %cfloat095
%yLow = OpFOrdLessThan %bool %yFrac %cfloat01
%yHigh = OpFOrdGreaterThan %bool %yFrac %cfloat09
%xMortar = OpLogicalOr %bool %xLow %xHigh
%yMortar = OpLogicalOr %bool %yLow %yHigh
%mortar = OpLogicalOr %bool %xMortar %yMortar
%baseColor = OpSelect %v3float %mortar %mortarColor %brickColor

%normalVal = OpLoad %v3float %normal
%normalN = OpExtInst %v3float %glsl Normalize %normalVal
%uboVal = OpLoad %UBO %ubo
%lightPos = OpCompositeExtract %v4float %uboVal 3
%lightX = OpCompositeExtract %float %lightPos 0
%lightY = OpCompositeExtract %float %lightPos 1
%lightZ = OpCompositeExtract %float %lightPos 2
%light3 = OpCompositeConstruct %v3float %lightX %lightY %lightZ
%worldVal = OpLoad %v3float %worldPos
%lightDirRaw = OpFSub %v3float %light3 %worldVal
%lightDir = OpExtInst %v3float %glsl Normalize %lightDirRaw
%diffDot = OpDot %float %normalN %lightDir
%diffuse = OpExtInst %float %glsl FMax %diffDot %cfloat0
%shadeMul = OpFMul %float %diffuse %cfloat06
%shade = OpFAdd %float %cfloat04 %shadeMul
%litColor = OpVectorTimesScalar %v3float %baseColor %shade
%outv = OpCompositeConstruct %v4float %litColor %cfloat1
OpStore %outColor %outv
OpReturn
OpFunctionEnd
