use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    let mut edges = Vec::with_capacity(m);
    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push((u, v));
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let mut seen = vec![false; n];
    let mut pos = vec![Pt { x: 0.0, y: 0.0 }; n];
    let mut component_id = 0usize;

    for start in 0..n {
        if seen[start] {
            continue;
        }
        let comp = collect_component(start, &adj, &mut seen);
        if comp.is_empty() {
            continue;
        }
        let comp_pos = if comp.len() <= 1500 {
            force_layout(&comp, &adj, &edges)
        } else {
            layered_layout(&comp, &adj)
        };

        let offset_x = (component_id as f64).sqrt() * 2000.0;
        let offset_y = (component_id as f64 / 8.0).floor() * 2000.0;
        for (v, p) in comp.into_iter().zip(comp_pos.into_iter()) {
            pos[v] = Pt {
                x: p.x + offset_x,
                y: p.y + offset_y,
            };
        }
        component_id += 1;
    }

    let mut out = String::new();
    for p in pos {
        out.push_str(&format!("{:.6} {:.6}\n", p.x, p.y));
    }
    print!("{}", out);
}

fn collect_component(start: usize, adj: &[Vec<usize>], seen: &mut [bool]) -> Vec<usize> {
    let mut q = VecDeque::new();
    let mut comp = Vec::new();
    seen[start] = true;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        comp.push(u);
        for &v in &adj[u] {
            if !seen[v] {
                seen[v] = true;
                q.push_back(v);
            }
        }
    }
    comp
}

fn force_layout(comp: &[usize], adj: &[Vec<usize>], edges: &[(usize, usize)]) -> Vec<Pt> {
    let n = comp.len();
    let mut index = vec![usize::MAX; adj.len()];
    for (i, &v) in comp.iter().enumerate() {
        index[v] = i;
    }

    let mut p = vec![Pt { x: 0.0, y: 0.0 }; n];
    let r = (n as f64).sqrt() * 10.0 + 10.0;
    for (i, pt) in p.iter_mut().enumerate() {
        let a = 2.0 * PI * (i as f64) / (n as f64).max(1.0);
        pt.x = r * a.cos();
        pt.y = r * a.sin();
    }

    let comp_set = {
        let mut s = vec![false; adj.len()];
        for &v in comp {
            s[v] = true;
        }
        s
    };

    let iters = if n <= 150 { 60 } else { 28 };
    let area = (n as f64).max(1.0) * 1000.0;
    let k = (area / n as f64).sqrt().max(1.0);

    for t in 0..iters {
        let temp = k * (1.0 - t as f64 / iters as f64).max(0.05);
        let mut disp = vec![(0.0f64, 0.0f64); n];

        for i in 0..n {
            for j in (i + 1)..n {
                let dx = p[i].x - p[j].x;
                let dy = p[i].y - p[j].y;
                let mut d2 = dx * dx + dy * dy;
                if d2 < 1e-6 {
                    d2 = 1e-6;
                }
                let d = d2.sqrt();
                let force = k * k / d;
                let fx = dx / d * force;
                let fy = dy / d * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        for &(u, v) in edges {
            if !(comp_set[u] && comp_set[v]) || u == v {
                continue;
            }
            let iu = index[u];
            let iv = index[v];
            let dx = p[iu].x - p[iv].x;
            let dy = p[iu].y - p[iv].y;
            let mut d2 = dx * dx + dy * dy;
            if d2 < 1e-6 {
                d2 = 1e-6;
            }
            let d = d2.sqrt();
            let force = d2 / k;
            let fx = dx / d * force;
            let fy = dy / d * force;
            disp[iu].0 -= fx;
            disp[iu].1 -= fy;
            disp[iv].0 += fx;
            disp[iv].1 += fy;
        }

        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let len = (dx * dx + dy * dy).sqrt();
            if len > 0.0 {
                let step = len.min(temp) / len;
                p[i].x += dx * step;
                p[i].y += dy * step;
            }
        }

        if n > 50 && t % 4 == 3 {
            let mut cx = 0.0;
            let mut cy = 0.0;
            for pt in &p {
                cx += pt.x;
                cy += pt.y;
            }
            cx /= n as f64;
            cy /= n as f64;
            for pt in &mut p {
                pt.x = (pt.x - cx) * 0.98;
                pt.y = (pt.y - cy) * 0.98;
            }
        }
    }

    normalize(&mut p);
    p
}

fn layered_layout(comp: &[usize], adj: &[Vec<usize>]) -> Vec<Pt> {
    let n = comp.len();
    let mut index = vec![usize::MAX; adj.len()];
    for (i, &v) in comp.iter().enumerate() {
        index[v] = i;
    }

    let mut local_adj = vec![Vec::<usize>::new(); n];
    for (i, &v) in comp.iter().enumerate() {
        for &to in &adj[v] {
            let j = index[to];
            if j != usize::MAX {
                local_adj[i].push(j);
            }
        }
        local_adj[i].sort_unstable();
        local_adj[i].dedup();
    }

    let root = comp
        .iter()
        .copied()
        .max_by_key(|&v| adj[v].len())
        .unwrap_or(comp[0]);
    let root_i = index[root];
    let far1 = bfs_farthest(root_i, &local_adj).0;
    let (far2, parent) = bfs_farthest_with_parent(far1, &local_adj);

    let mut depth = vec![usize::MAX; n];
    let mut q = VecDeque::new();
    depth[far2] = 0;
    q.push_back(far2);
    while let Some(u) = q.pop_front() {
        for &v in &local_adj[u] {
            if depth[v] == usize::MAX {
                depth[v] = depth[u] + 1;
                q.push_back(v);
            }
        }
    }

    for i in 0..n {
        if depth[i] == usize::MAX {
            depth[i] = parent_depth(i, &local_adj);
        }
    }

    let max_depth = *depth.iter().max().unwrap_or(&0);
    let mut layers: Vec<Vec<usize>> = vec![Vec::new(); max_depth + 1];
    for i in 0..n {
        layers[depth[i]].push(i);
    }

    let mut order: Vec<usize> = (0..n).collect();
    let mut pos_in_order = vec![0usize; n];
    for sweep in 0..6 {
        let top_down = sweep % 2 == 0;
        let layer_indices: Vec<usize> = if top_down {
            (0..=max_depth).collect()
        } else {
            (0..=max_depth).rev().collect()
        };
        for &d in &layer_indices {
            let layer = &mut layers[d];
            layer.sort_by(|&a, &b| {
                let ca = barycenter(a, &local_adj, &pos_in_order, top_down);
                let cb = barycenter(b, &local_adj, &pos_in_order, top_down);
                ca.partial_cmp(&cb).unwrap_or(Ordering::Equal)
            });
            for (i, &v) in layer.iter().enumerate() {
                pos_in_order[v] = i;
            }
        }
        order.clear();
        for layer in &layers {
            for &v in layer {
                order.push(v);
            }
        }
        for (i, &v) in order.iter().enumerate() {
            pos_in_order[v] = i;
        }
        let _ = &parent;
    }

    let mut p = vec![Pt { x: 0.0, y: 0.0 }; n];
    let x_gap = 18.0;
    let y_gap = 42.0;
    for d in 0..=max_depth {
        let layer = &layers[d];
        let width = layer.len() as f64;
        for (i, &v) in layer.iter().enumerate() {
            let x = (i as f64 - (width - 1.0) / 2.0) * x_gap;
            let y = d as f64 * y_gap;
            p[v] = Pt { x, y };
        }
    }

    normalize(&mut p);
    p
}

fn bfs_farthest(start: usize, adj: &[Vec<usize>]) -> (usize, Vec<usize>) {
    let mut dist = vec![usize::MAX; adj.len()];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    let mut far = start;
    for i in 0..adj.len() {
        if dist[i] != usize::MAX && dist[i] > dist[far] {
            far = i;
        }
    }
    (far, dist)
}

fn bfs_farthest_with_parent(start: usize, adj: &[Vec<usize>]) -> (usize, Vec<usize>) {
    let mut dist = vec![usize::MAX; adj.len()];
    let mut parent = vec![usize::MAX; adj.len()];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                parent[v] = u;
                q.push_back(v);
            }
        }
    }
    let mut far = start;
    for i in 0..adj.len() {
        if dist[i] != usize::MAX && dist[i] > dist[far] {
            far = i;
        }
    }
    (far, parent)
}

fn parent_depth(start: usize, adj: &[Vec<usize>]) -> usize {
    let mut dist = vec![usize::MAX; adj.len()];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    dist.into_iter().filter(|&d| d != usize::MAX).max().unwrap_or(0)
}

fn barycenter(v: usize, adj: &[Vec<usize>], pos: &[usize], top_down: bool) -> f64 {
    let mut sum = 0.0;
    let mut cnt = 0.0;
    for &to in &adj[v] {
        let p = pos[to] as f64;
        sum += p;
        cnt += 1.0;
    }
    if cnt == 0.0 {
        pos[v] as f64
    } else {
        let bias = if top_down { 0.0 } else { 0.1 };
        sum / cnt + bias
    }
}

fn normalize(p: &mut [Pt]) {
    if p.is_empty() {
        return;
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for pt in p.iter() {
        cx += pt.x;
        cy += pt.y;
    }
    cx /= p.len() as f64;
    cy /= p.len() as f64;
    let mut scale: f64 = 0.0;
    for pt in p.iter() {
        scale = scale.max((pt.x - cx).abs());
        scale = scale.max((pt.y - cy).abs());
    }
    let scale = if scale < 1e-9 { 1.0 } else { 1000.0 / scale };
    for pt in p.iter_mut() {
        pt.x = (pt.x - cx) * scale;
        pt.y = (pt.y - cy) * scale;
    }
}
