use std::collections::VecDeque;
use std::io::{self, Read, Write};

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();

    let mut idx = 0usize;
    let h = read_usize(&input, &mut idx);
    let w = read_usize(&input, &mut idx);

    while idx < input.len() && (input[idx] == b'\n' || input[idx] == b'\r') {
        idx += 1;
    }

    let mut grid = vec![b'#'; h * w];
    let mut start = None;
    let mut goal = None;

    for y in 0..h {
        let line_start = idx;
        while idx < input.len() && input[idx] != b'\n' {
            idx += 1;
        }
        let mut line_end = idx;
        if line_end > line_start && input[line_end - 1] == b'\r' {
            line_end -= 1;
        }
        let row = &input[line_start..line_end];
        for x in 0..w {
            let ch = if x < row.len() { row[x] } else { b'#' };
            let pos = y * w + x;
            grid[pos] = ch;
            if ch == b'A' {
                start = Some(pos);
            } else if ch == b'B' {
                goal = Some(pos);
            }
        }
        if idx < input.len() && input[idx] == b'\n' {
            idx += 1;
        }
    }

    let (start, goal) = match (start, goal) {
        (Some(s), Some(g)) => (s, g),
        _ => {
            let mut out = io::BufWriter::new(io::stdout());
            writeln!(out, "0").unwrap();
            return;
        }
    };

    let mut parent = vec![u32::MAX; h * w];
    let mut q = VecDeque::new();
    parent[start] = start as u32;
    q.push_back(start);

    let mut found = false;
    while let Some(pos) = q.pop_front() {
        if pos == goal {
            found = true;
            break;
        }
        let x = pos % w;
        let y = pos / w;

        if x > 0 {
            let npos = pos - 1;
            if parent[npos] == u32::MAX && is_open(grid[npos]) {
                parent[npos] = pos as u32;
                q.push_back(npos);
            }
        }
        if x + 1 < w {
            let npos = pos + 1;
            if parent[npos] == u32::MAX && is_open(grid[npos]) {
                parent[npos] = pos as u32;
                q.push_back(npos);
            }
        }
        if y > 0 {
            let npos = pos - w;
            if parent[npos] == u32::MAX && is_open(grid[npos]) {
                parent[npos] = pos as u32;
                q.push_back(npos);
            }
        }
        if y + 1 < h {
            let npos = pos + w;
            if parent[npos] == u32::MAX && is_open(grid[npos]) {
                parent[npos] = pos as u32;
                q.push_back(npos);
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout());
    if !found && start != goal {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path = Vec::new();
    let mut cur = goal;
    path.push(cur);
    while cur != start {
        cur = parent[cur] as usize;
        path.push(cur);
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for pos in path {
        writeln!(out, "{} {}", pos % w, pos / w).unwrap();
    }
}

fn is_open(ch: u8) -> bool {
    ch != b'#'
}

fn read_usize(input: &[u8], idx: &mut usize) -> usize {
    while *idx < input.len() && input[*idx].is_ascii_whitespace() {
        *idx += 1;
    }
    let mut n = 0usize;
    while *idx < input.len() && input[*idx].is_ascii_digit() {
        n = n * 10 + (input[*idx] - b'0') as usize;
        *idx += 1;
    }
    n
}
