use std::f64::consts::PI;
use std::io::{self, BufRead, BufWriter, Write};

#[derive(Clone, Copy, Debug)]
struct Target {
    x1: f64,
    x2: f64,
    y: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn parse_f64s(line: &str) -> Vec<f64> {
    line.split_whitespace()
        .filter_map(|s| s.parse::<f64>().ok())
        .collect()
}

fn desired_vertical_speed(alt: f64) -> f64 {
    if alt > 120.0 {
        22.0
    } else if alt > 70.0 {
        16.0
    } else if alt > 35.0 {
        10.0
    } else if alt > 15.0 {
        6.0
    } else if alt > 7.0 {
        3.0
    } else {
        1.0
    }
}

fn main() {
    let stdin = io::stdin();
    let mut input = stdin.lock();
    let mut stdout = BufWriter::new(io::stdout());

    let mut line = String::new();

    line.clear();
    if input.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
        return;
    }
    let first = parse_f64s(&line);
    if first.len() < 5 {
        return;
    }
    let width = first[0] as usize;
    let height = first[1] as usize;
    let gravity = first[2];
    let max_thrust = first[3];
    let fuel_capacity = first[4];

    line.clear();
    if input.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
        return;
    }
    let second = parse_f64s(&line);
    if second.len() < 4 {
        return;
    }
    let start_x = second[0];
    let start_y = second[1];
    let target_x = second[2];
    let target_y = second[3];

    let mut landing: Option<Target> = None;
    let mut pad_row = target_y;
    let mut pad_center = target_x;

    for y in 0..height {
        line.clear();
        if input.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
            return;
        }
        let row = line.trim_end_matches(['\r', '\n']);
        let bytes = row.as_bytes();
        let mut start: Option<usize> = None;
        let mut end: Option<usize> = None;
        for (x, &b) in bytes.iter().enumerate().take(width) {
            if b == b'T' || b == b'L' {
                if start.is_none() {
                    start = Some(x);
                }
                end = Some(x);
                pad_row = y as f64;
            }
        }
        if let (Some(s), Some(e)) = (start, end) {
            landing = Some(Target {
                x1: s as f64,
                x2: (e + 1) as f64,
                y: y as f64,
            });
            pad_center = (s as f64 + (e as f64 + 1.0)) * 0.5;
        }
    }

    let pad = landing.unwrap_or(Target {
        x1: target_x - 2.0,
        x2: target_x + 2.0,
        y: pad_row,
    });

    let mut last_angle_cmd = 0.0;

    loop {
        line.clear();
        if input.read_line(&mut line).ok().filter(|&n| n > 0).is_none() {
            break;
        }
        let vals = parse_f64s(&line);
        if vals.len() < 7 {
            continue;
        }

        let x = vals[0];
        let y = vals[1];
        let vx = vals[2];
        let vy = vals[3];
        let angle = vals[4];
        let angular_vel = vals[5];
        let fuel_remaining = vals[6].min(fuel_capacity);

        let dx = pad_center - x;
        let alt = (pad.y - y).max(0.0);
        let in_pad_band = x >= pad.x1 - 1.0 && x <= pad.x2 + 1.0;
        let horizontal_close = dx.abs() < 6.0;

        let mut desired_angle = clamp(0.14 * dx - 0.55 * vx, -0.9, 0.9);
        if alt < 35.0 {
            desired_angle = clamp(0.08 * dx - 0.35 * vx, -0.45, 0.45);
        }
        if alt < 12.0 {
            desired_angle = clamp(0.03 * dx - 0.20 * vx, -0.20, 0.20);
        }
        if in_pad_band && alt < 25.0 {
            desired_angle = clamp(-0.20 * vx, -0.18, 0.18);
        }

        let angle_error = wrap_pi(desired_angle - angle);
        let turn = clamp(2.2 * angle_error - 0.45 * angular_vel, -1.0, 1.0);

        let mut thrust = 0.0;
        if fuel_remaining > 0.01 {
            let tilt_penalty = angle.cos().abs().max(0.25);
            let upward_accel_needed = gravity + 0.45 * vy.max(0.0) + 0.18 * vx.abs();
            let mut required = upward_accel_needed / (max_thrust * tilt_penalty);

            let target_vy = desired_vertical_speed(alt);
            let vy_error = vy - target_vy;
            required += clamp(vy_error * 0.03, -0.25, 0.35);

            if alt < 18.0 {
                required = required.max(0.65);
            }
            if alt < 8.0 {
                required = required.max(0.82);
            }
            if alt < 4.0 {
                required = required.max(0.95);
            }
            if !horizontal_close && alt < 30.0 {
                required = required.max(0.72);
            }
            thrust = clamp(required, 0.0, 1.0);

            if vy > 2.5 && alt < 12.0 {
                thrust = thrust.max(0.90);
            }
            if vy > 1.2 && alt < 5.0 {
                thrust = thrust.max(1.0);
            }
        }

        if fuel_remaining <= 0.01 {
            thrust = 0.0;
        }

        if alt < 2.0 && in_pad_band {
            thrust = thrust.max(0.95);
        }

        let _ = writeln!(stdout, "{:.3} {:.3}", thrust, turn);
        let _ = stdout.flush();

        last_angle_cmd = turn;
    }

    let _ = start_x;
    let _ = start_y;
    let _ = last_angle_cmd;
}
