using namespace std;

struct Point {
    double x, y;
};

static inline uint64_t splitBy1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t mortonKey(double x, double y, double minx, double miny, double sx, double sy) {
    constexpr uint32_t SCALE = (1u << 16) - 1u;
    uint32_t ix = (uint32_t)max(0.0, min((x - minx) * sx, (double)SCALE));
    uint32_t iy = (uint32_t)max(0.0, min((y - miny) * sy, (double)SCALE));
    return (splitBy1(ix) << 1) | splitBy1(iy);
}

static double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static vector<int> bruteForceTSP(const vector<Point>& p) {
    int n = (int)p.size();
    vector<int> perm;
    for (int i = 1; i < n; ++i) perm.push_back(i);
    vector<int> best = perm;
    double bestLen = numeric_limits<double>::infinity();
    do {
        double cur = 0.0;
        int prev = 0;
        for (int v : perm) {
            cur += dist2(p[prev], p[v]);
            prev = v;
        }
        cur += dist2(p[prev], p[0]);
        if (cur < bestLen) {
            bestLen = cur;
            best = perm;
        }
    } while (next_permutation(perm.begin(), perm.end()));

    vector<int> ans;
    ans.reserve(n);
    ans.push_back(0);
    for (int v : best) ans.push_back(v);
    return ans;
}

static vector<int> nearestNeighborOrder(const vector<Point>& p) {
    int n = (int)p.size();
    if (n == 0) return {};
    if (n == 1) return {0};

    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (auto& pt : p) {
        minx = min(minx, pt.x);
        maxx = max(maxx, pt.x);
        miny = min(miny, pt.y);
        maxy = max(maxy, pt.y);
    }

    double spanx = maxx - minx, spany = maxy - miny;
    if (spanx == 0) spanx = 1;
    if (spany == 0) spany = 1;

    int g = max(1, (int)sqrt((double)n / 2.0));
    double cellW = spanx / g, cellH = spany / g;
    if (cellW == 0) cellW = 1;
    if (cellH == 0) cellH = 1;

    vector<vector<int>> grid((size_t)g * g);
    grid.shrink_to_fit();
    vector<int> cx(n), cy(n);
    for (int i = 0; i < n; ++i) {
        int ix = (int)((p[i].x - minx) / cellW);
        int iy = (int)((p[i].y - miny) / cellH);
        if (ix < 0) ix = 0; if (ix >= g) ix = g - 1;
        if (iy < 0) iy = 0; if (iy >= g) iy = g - 1;
        cx[i] = ix; cy[i] = iy;
        grid[(size_t)iy * g + ix].push_back(i);
    }

    vector<char> used(n, 0);
    vector<int> ans;
    ans.reserve(n);
    int cur = 0;
    used[cur] = 1;
    ans.push_back(cur);

    auto removeFromCell = [&](int id) {
        int idx = (size_t)cy[id] * g + cx[id];
        auto& v = grid[idx];
        for (size_t k = 0; k < v.size(); ++k) {
            if (v[k] == id) {
                v[k] = v.back();
                v.pop_back();
                return;
            }
        }
    };

    removeFromCell(cur);
    int remaining = n - 1;
    while (remaining > 0) {
        int bx = cx[cur], by = cy[cur];
        int best = -1;
        double bestD = numeric_limits<double>::infinity();
        for (int rad = 0; rad < g; ++rad) {
            bool found = false;
            int x0 = max(0, bx - rad), x1 = min(g - 1, bx + rad);
            int y0 = max(0, by - rad), y1 = min(g - 1, by + rad);
            for (int y = y0; y <= y1; ++y) {
                for (int x = x0; x <= x1; ++x) {
                    if (x != x0 && x != x1 && y != y0 && y != y1) continue;
                    auto& bucket = grid[(size_t)y * g + x];
                    for (int v : bucket) {
                        if (used[v]) continue;
                        double d = dist2(p[cur], p[v]);
                        if (d < bestD) {
                            bestD = d;
                            best = v;
                            found = true;
                        }
                    }
                }
            }
            if (found) break;
        }
        if (best == -1) {
            for (int v = 0; v < n; ++v) if (!used[v]) { best = v; break; }
        }
        used[best] = 1;
        removeFromCell(best);
        ans.push_back(best);
        cur = best;
        --remaining;
    }
    return ans;
}

static void twoOptImprove(const vector<Point>& p, vector<int>& tour, int passesLimit = 2) {
    int n = (int)tour.size();
    if (n < 4) return;
    for (int pass = 0; pass < passesLimit; ++pass) {
        bool improved = false;
        for (int i = 1; i < n - 2; ++i) {
            int a = tour[i - 1], b = tour[i];
            for (int k = i + 1; k < n - 1; ++k) {
                int c = tour[k], d = tour[k + 1];
                double before = dist2(p[a], p[b]) + dist2(p[c], p[d]);
                double after = dist2(p[a], p[c]) + dist2(p[b], p[d]);
                if (after + 1e-12 < before) {
                    reverse(tour.begin() + i, tour.begin() + k + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

static vector<int> adaptiveSolve(const vector<Point>& p) {
    int n = (int)p.size();
    if (n <= 12) return bruteForceTSP(p);
    if (n <= 2500) {
        vector<int> ans = nearestNeighborOrder(p);
        twoOptImprove(p, ans, 2);
        if (ans.empty() || ans[0] != 0) {
            auto it = find(ans.begin(), ans.end(), 0);
            if (it != ans.end()) rotate(ans.begin(), it, ans.end());
        }
        return ans;
    }

    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (auto& pt : p) {
        minx = min(minx, pt.x);
        maxx = max(maxx, pt.x);
        miny = min(miny, pt.y);
        maxy = max(maxy, pt.y);
    }
    double sx = (maxx > minx) ? ((double)((1u << 16) - 1u) / (maxx - minx)) : 1.0;
    double sy = (maxy > miny) ? ((double)((1u << 16) - 1u) / (maxy - miny)) : 1.0;

    vector<int> order;
    order.reserve(n);
    for (int i = 1; i < n; ++i) order.push_back(i);
    sort(order.begin(), order.end(), [&](int a, int b) {
        uint64_t ka = mortonKey(p[a].x, p[a].y, minx, miny, sx, sy);
        uint64_t kb = mortonKey(p[b].x, p[b].y, minx, miny, sx, sy);
        if (ka != kb) return ka < kb;
        if (p[a].x != p[b].x) return p[a].x < p[b].x;
        return p[a].y < p[b].y;
    });

    vector<int> ans;
    ans.reserve(n);
    ans.push_back(0);
    for (int v : order) ans.push_back(v);
    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    vector<Point> p(N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        cin >> x >> y;
        p[i] = {x, y};
    }

    vector<int> ans = adaptiveSolve(p);
    if ((int)ans.size() != N) {
        vector<char> seen(N, 0);
        vector<int> fixed;
        fixed.reserve(N);
        fixed.push_back(0);
        seen[0] = 1;
        for (int v : ans) if (0 <= v && v < N && !seen[v]) { seen[v] = 1; fixed.push_back(v); }
        for (int i = 0; i < N; ++i) if (!seen[i]) fixed.push_back(i);
        ans.swap(fixed);
    }

    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
