declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid;
const robots: Robot[] = data.robots.slice().sort((a: Robot, b: Robot) => a.id - b.id);
const jobs: Job[] = data.jobs.slice();

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const n = h * w;

function idx(x: number, y: number): number {
  return x * w + y;
}
function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < h && y >= 0 && y < w;
}
function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && grid[x].charCodeAt(y) !== 35;
}

let maxRobotId = 0;
for (const r of robots) maxRobotId = Math.max(maxRobotId, r.id);
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

const owner = new Int32Array(n);
const dist = new Int32Array(n);
for (let i = 0; i < n; i++) {
  owner[i] = -1;
  dist[i] = -1;
}

const q = new Int32Array(n);
let head = 0;
let tail = 0;

for (const r of robots) {
  if (!isOpen(r.x, r.y)) continue;
  const p = idx(r.x, r.y);
  if (owner[p] === -1) {
    owner[p] = r.id;
    dist[p] = 0;
    q[tail++] = p;
  }
}

const dx = [-1, 1, 0, 0];
const dy = [0, 0, -1, 1];
while (head < tail) {
  const p = q[head++];
  const x = Math.floor(p / w);
  const y = p % w;
  const nd = dist[p] + 1;
  const curOwner = owner[p];
  for (let k = 0; k < 4; k++) {
    const nx = x + dx[k];
    const ny = y + dy[k];
    if (!isOpen(nx, ny)) continue;
    const np = idx(nx, ny);
    if (dist[np] === -1) {
      dist[np] = nd;
      owner[np] = curOwner;
      q[tail++] = np;
    } else if (dist[np] === nd && curOwner < owner[np]) {
      owner[np] = curOwner;
    }
  }
}

for (let i = 0; i < jobs.length; i++) {
  const j = jobs[i];
  let rid = -1;
  if (isOpen(j.x, j.y)) {
    rid = owner[idx(j.x, j.y)];
  }
  if (rid < 0 || rid > maxRobotId) {
    let best = 0;
    let bestD = Number.POSITIVE_INFINITY;
    for (let r = 0; r < robots.length; r++) {
      const rr = robots[r];
      const d = Math.abs(rr.x - j.x) + Math.abs(rr.y - j.y);
      if (d < bestD || (d === bestD && rr.id < robots[best].id)) {
        bestD = d;
        best = r;
      }
    }
    rid = robots[best].id;
  }
  assignments[rid].push(i);
}

const robotById = new Map<number, Robot>();
for (const r of robots) robotById.set(r.id, r);
for (const r of robots) {
  const rr = r;
  assignments[r.id].sort((ia, ib) => {
    const a = jobs[ia];
    const b = jobs[ib];
    if (b.priority !== a.priority) return b.priority - a.priority;
    const da = Math.abs(a.x - rr.x) + Math.abs(a.y - rr.y);
    const db = Math.abs(b.x - rr.x) + Math.abs(b.y - rr.y);
    if (da !== db) return da - db;
    if (a.x !== b.x) return a.x - b.x;
    if (a.y !== b.y) return a.y - b.y;
    return a.id - b.id;
  });
}

const out = { assignments: assignments.map(list => list.map(i => jobs[i].id)) };
process.stdout.write(JSON.stringify(out));
