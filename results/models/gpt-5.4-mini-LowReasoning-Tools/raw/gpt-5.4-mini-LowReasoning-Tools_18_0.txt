struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char read() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        char c;
        T sign = 1;
        T val = 0;
        c = read();
        if (!c) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = read();
            if (!c) return false;
        }
        if (c == '-') {
            sign = -1;
            c = read();
        }
        for (; c >= '0' && c <= '9'; c = read()) val = val * 10 + (c - '0');
        out = val * sign;
        return true;
    }
};

struct FastOutput {
    string out;
    void push_back(char c) { out.push_back(c); }
    void writeInt(long long x) {
        if (x == 0) {
            out.push_back('0');
            return;
        }
        if (x < 0) {
            out.push_back('-');
            x = -x;
        }
        char s[24];
        int n = 0;
        while (x) {
            s[n++] = char('0' + x % 10);
            x /= 10;
        }
        while (n--) out.push_back(s[n]);
    }
    void writeEdge(int u, int v) {
        writeInt(u);
        out.push_back(' ');
        writeInt(v);
        out.push_back('\n');
    }
};

struct DSU {
    vector<int> p, r;
    DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        iota(p.begin(), p.end(), 0);
    }
    int find(int x) { return p[x] == x ? x : p[x] = find(p[x]); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

struct BridgeFinder {
    int n;
    const vector<Edge> &edges;
    vector<vector<pair<int,int>>> g;
    vector<int> tin, low, it, parentV, parentE;
    int timer = 0;

    BridgeFinder(int n_, const vector<Edge> &e) : n(n_), edges(e) {
        g.assign(n, {});
        for (int i = 0; i < (int)edges.size(); ++i) {
            auto &ed = edges[i];
            g[ed.u].push_back({ed.v, i});
            g[ed.v].push_back({ed.u, i});
        }
        tin.assign(n, -1);
        low.assign(n, -1);
        it.assign(n, 0);
        parentV.assign(n, -1);
        parentE.assign(n, -1);
    }

    bool findBridge(int &eid) {
        for (int s = 0; s < n; ++s) {
            if (tin[s] != -1) continue;
            stack<int> st;
            st.push(s);
            parentV[s] = -1;
            parentE[s] = -1;
            while (!st.empty()) {
                int v = st.top();
                if (tin[v] == -1) {
                    tin[v] = low[v] = timer++;
                }
                if (it[v] < (int)g[v].size()) {
                    auto [to, id] = g[v][it[v]++];
                    if (id == parentE[v]) continue;
                    if (tin[to] == -1) {
                        parentV[to] = v;
                        parentE[to] = id;
                        st.push(to);
                    } else {
                        low[v] = min(low[v], tin[to]);
                    }
                } else {
                    st.pop();
                    if (parentV[v] != -1) {
                        low[parentV[v]] = min(low[parentV[v]], low[v]);
                        if (low[v] > tin[parentV[v]]) {
                            eid = parentE[v];
                            return true;
                        }
                    }
                }
            }
        }
        return false;
    }
};

struct StoerWagner {
    int n;
    const vector<Edge> &edges;
    vector<vector<long long>> w;
    vector<vector<int>> members;

    StoerWagner(int n_, const vector<Edge> &e) : n(n_), edges(e) {
        w.assign(n, vector<long long>(n, 0));
        for (auto &ed : edges) {
            if (ed.u == ed.v) continue;
            w[ed.u][ed.v] += 1;
            w[ed.v][ed.u] += 1;
        }
        members.resize(n);
        for (int i = 0; i < n; ++i) members[i].push_back(i);
    }

    pair<long long, vector<int>> run() {
        vector<int> verts(n);
        iota(verts.begin(), verts.end(), 0);
        long long best = numeric_limits<long long>::max();
        vector<int> bestSide;

        while (verts.size() > 1) {
            vector<long long> dist(n, 0);
            vector<char> added(n, 0);
            int prev = -1, sel = -1;
            for (size_t it = 0; it < verts.size(); ++it) {
                sel = -1;
                for (int v : verts) if (!added[v] && (sel == -1 || dist[v] > dist[sel])) sel = v;
                added[sel] = 1;
                if (it + 1 == verts.size()) {
                    if (dist[sel] < best) {
                        best = dist[sel];
                        bestSide = members[sel];
                    }
                    if (prev != -1) {
                        for (int v : verts) {
                            if (v == sel || added[v]) continue;
                            w[prev][v] += w[sel][v];
                            w[v][prev] = w[prev][v];
                        }
                        members[prev].insert(members[prev].end(), members[sel].begin(), members[sel].end());
                        verts.erase(find(verts.begin(), verts.end(), sel));
                    }
                    break;
                }
                prev = sel;
                for (int v : verts) if (!added[v]) dist[v] += w[sel][v];
            }
        }
        return {best, bestSide};
    }
};

struct RandomCut {
    int n;
    const vector<Edge> &edges;
    mt19937_64 rng;
    RandomCut(int n_, const vector<Edge> &e) : n(n_), edges(e) {
        rng.seed((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count());
    }

    pair<int, vector<int>> runOnce() {
        DSU dsu(n);
        int comps = n;
        uniform_int_distribution<size_t> dist(0, edges.size() - 1);
        while (comps > 2) {
            const Edge &ed = edges[dist(rng)];
            if (dsu.unite(ed.u, ed.v)) --comps;
        }
        vector<int> roots;
        for (int i = 0; i < n; ++i) {
            int r = dsu.find(i);
            if (find(roots.begin(), roots.end(), r) == roots.end()) roots.push_back(r);
        }
        if (roots.size() < 2) return {numeric_limits<int>::max(), {}};
        vector<int> side(n, 0);
        for (int i = 0; i < n; ++i) if (dsu.find(i) == roots[0]) side[i] = 1;
        int cut = 0;
        for (auto &ed : edges) if (side[ed.u] != side[ed.v]) ++cut;
        vector<int> verts;
        for (int i = 0; i < n; ++i) if (side[i]) verts.push_back(i);
        return {cut, verts};
    }
};

        for (auto &e : cutEdges) out.writeEdge(e.u, e.v);
        cout << out.out;

    }

    RandomCut rc(n, edges);
    int best = numeric_limits<int>::max();
    vector<int> bestSide;
    int trials = max(20, min(200, (int)(2 * log((double)max(2, n)) * log((double)max(2, n)))));
    for (int i = 0; i < trials; ++i) {
        auto [cut, sideVerts] = rc.runOnce();
        if (cut < best) {
            best = cut;
            bestSide = move(sideVerts);
        }
    }
    vector<char> side(n, 0);
    for (int v : bestSide) side[v] = 1;
    vector<Edge> cutEdges;
    for (auto &e : edges) if (side[e.u] != side[e.v]) cutEdges.push_back(e);
    if (cutEdges.empty() && !edges.empty()) cutEdges.push_back(edges[0]);
    FastOutput out;
    out.writeInt((int)cutEdges.size());
    out.push_back('\n');
    for (auto &e : cutEdges) out.writeEdge(e.u, e.v);
    cout << out.out;
