const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);
const H = grid.length;
const W = H ? (grid[0] ? grid[0].length : 0) : 0;

function inBounds(x, y) {
  return y >= 0 && y < H && x >= 0 && x < W;
}
function isOpen(x, y) {
  return inBounds(x, y) && grid[y][x] !== -1;
}
function cellValue(x, y) {
  const v = grid[y][x];
  return v > 0 ? v : 0;
}

const dirs = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

// Build connected components of traversable cells to avoid planning across impossible barriers.
const compId = Array.from({ length: H }, () => Array(W).fill(-1));
const comps = [];
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    if (!isOpen(x, y) || compId[y][x] !== -1) continue;
    const id = comps.length;
    const qx = [x], qy = [y];
    compId[y][x] = id;
    const cells = [];
    for (let qi = 0; qi < qx.length; qi++) {
      const cx = qx[qi], cy = qy[qi];
      cells.push([cx, cy]);
      for (const [dx, dy] of dirs.slice(1)) {
        const nx = cx + dx, ny = cy + dy;
        if (isOpen(nx, ny) && compId[ny][nx] === -1) {
          compId[ny][nx] = id;
          qx.push(nx); qy.push(ny);
        }
      }
    }
    let totalValue = 0;
    for (const [cx, cy] of cells) totalValue += cellValue(cx, cy);
    comps.push({ cells, totalValue });
  }
}

function makePath(startX, startY) {
  const path = [[startX, startY]];
  let x = startX, y = startY;
  const visited = new Set();
  if (isOpen(x, y)) visited.add(`${x},${y}`);
  for (let step = 0; step < maxSteps; step++) {
    let best = null;
    let bestScore = -Infinity;

    for (const [dx, dy] of dirs) {
      const nx = x + dx, ny = y + dy;
      if (!isOpen(nx, ny)) continue;
      const key = `${nx},${ny}`;
      const newGain = visited.has(key) ? 0 : cellValue(nx, ny);

      // Prefer new value, but also keep some mobility and prefer cells near additional value.
      let frontier = 0;
      for (const [adx, ady] of dirs.slice(1)) {
        const fx = nx + adx, fy = ny + ady;
        if (isOpen(fx, fy) && !visited.has(`${fx},${fy}`)) frontier += cellValue(fx, fy);
      }
      const edgeBonus = (nx === 0 || ny === 0 || nx === W - 1 || ny === H - 1) ? -0.1 : 0;
      const score = newGain * 1000 + frontier * 10 + edgeBonus - (dx === 0 && dy === 0 ? 0.5 : 0);

      if (score > bestScore) {
        bestScore = score;
        best = [nx, ny];
      }
    }

    if (!best) {
      path.push([x, y]);
      continue;
    }

    [x, y] = best;
    visited.add(`${x},${y}`);
    path.push([x, y]);
  }
  return path;
}

// If there are multiple drones, try to diversify by giving each a different target bias.
// Cells are sorted by value; each drone greedily heads toward its own slice while still exploring locally.
const positiveCells = [];
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    if (grid[y][x] > 0) positiveCells.push([x, y, grid[y][x]]);
  }
}
positiveCells.sort((a, b) => b[2] - a[2]);

const droneCount = drones.length;
const sliceTargets = Array.from({ length: droneCount }, () => []);
for (let i = 0; i < positiveCells.length; i++) {
  sliceTargets[i % Math.max(1, droneCount)].push(positiveCells[i]);
}

function bfsDistances(sx, sy) {
  const dist = Array.from({ length: H }, () => Array(W).fill(-1));
  if (!isOpen(sx, sy)) return dist;
  const qx = [sx], qy = [sy];
  dist[sy][sx] = 0;
  for (let qi = 0; qi < qx.length; qi++) {
    const x = qx[qi], y = qy[qi];
    const d = dist[y][x];
    for (const [dx, dy] of dirs.slice(1)) {
      const nx = x + dx, ny = y + dy;
      if (isOpen(nx, ny) && dist[ny][nx] === -1) {
        dist[ny][nx] = d + 1;
        qx.push(nx); qy.push(ny);
      }
    }
  }
  return dist;
}

function planDrone(drone, targetList) {
  const { x: sx, y: sy } = drone;
  if (!isOpen(sx, sy)) {
    return Array.from({ length: maxSteps + 1 }, () => [sx, sy]);
  }

  const dist = bfsDistances(sx, sy);
  let bestTarget = null;
  let bestTargetScore = -Infinity;
  const limit = Math.min(targetList.length, 150);
  for (let i = 0; i < limit; i++) {
    const [tx, ty, val] = targetList[i];
    const d = dist[ty][tx];
    if (d < 0) continue;
    const score = val * 100 - d * 3;
    if (score > bestTargetScore) {
      bestTargetScore = score;
      bestTarget = [tx, ty];
    }
  }

  // If no target is reachable, fall back to local greedy walk.
  if (!bestTarget) return makePath(sx, sy);

  const [tx, ty] = bestTarget;
  const path = [[sx, sy]];
  let x = sx, y = sy;
  const seen = new Set([`${x},${y}`]);

  // Move toward target using BFS distance gradient, then continue with local greedy collection.
  for (let step = 0; step < maxSteps; step++) {
    let next = null;
    let best = Infinity;
    const curD = dist[y][x];
    if (curD === 0 && x === tx && y === ty) break;

    for (const [dx, dy] of dirs) {
      const nx = x + dx, ny = y + dy;
      if (!isOpen(nx, ny)) continue;
      const nd = dist[ny][nx];
      if (nd >= 0 && nd < best && nd <= curD) {
        best = nd;
        next = [nx, ny];
      }
    }

    if (!next || (next[0] === x && next[1] === y && curD !== 0)) {
      // Small local expansion: choose the best adjacent unvisited value.
      let localBest = null;
      let localScore = -Infinity;
      for (const [dx, dy] of dirs) {
        const nx = x + dx, ny = y + dy;
        if (!isOpen(nx, ny)) continue;
        const gain = seen.has(`${nx},${ny}`) ? 0 : cellValue(nx, ny);
        const score = gain * 100 + (nx === tx && ny === ty ? 50 : 0) - (dx === 0 && dy === 0 ? 1 : 0);
        if (score > localScore) {
          localScore = score;
          localBest = [nx, ny];
        }
      }
      next = localBest || [x, y];
    }

    x = next[0];
    y = next[1];
    seen.add(`${x},${y}`);
    path.push([x, y]);
  }

  while (path.length < maxSteps + 1) path.push([x, y]);
  return path;
}

const paths = [];
for (let i = 0; i < drones.length; i++) {
  paths.push(planDrone(drones[i], sliceTargets[i]));
}

process.stdout.write(JSON.stringify({ paths }));
