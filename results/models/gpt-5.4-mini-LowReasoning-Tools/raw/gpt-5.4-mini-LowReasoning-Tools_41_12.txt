; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 60
OpCapability Shader
OpMemoryModel Logical GLSL450
%1 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fnvoid = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_03 = OpConstant %float 0.3
%float_05 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.4
%float_06 = OpConstant %float 0.6
%float_015 = OpConstant %float 0.15
%float_005 = OpConstant %float 0.05
%skyColor = OpConstantComposite %v3float %float_04 %float_06 %float_1
%groundColor = OpConstantComposite %v3float %float_03 %float_015 %float_005
%v3zero = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v3one = OpConstantComposite %v3float %float_1 %float_1 %float_1

%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %fnvoid
%label = OpLabel
%wp = OpLoad %v3float %in_worldPos
%n_raw = OpLoad %v3float %in_normal
%norm = OpExtInst %v3float %1 Normalize %n_raw
%light_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %light_ptr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%Lvec = OpFSub %v3float %light3 %wp
%L = OpExtInst %v3float %1 Normalize %Lvec
%dotNL = OpDot %float %norm %L
%d = OpExtInst %float %1 FMax %dotNL %float_0
%ny = OpCompositeExtract %float %norm 1
%t0 = OpFMul %float %ny %float_05
%t = OpFAdd %float %t0 %float_05
%diff = OpFSub %v3float %skyColor %groundColor
%ambient_scaled = OpVectorTimesScalar %v3float %diff %t
%ambient = OpFAdd %v3float %groundColor %ambient_scaled
%diffuse_scalar = OpFMul %float %d %float_03
%diffuse = OpCompositeConstruct %v3float %diffuse_scalar %diffuse_scalar %diffuse_scalar
%rgb = OpFAdd %v3float %ambient %diffuse
%color = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
