use std::cmp::Ordering;
use std::io::{self, Read, Write};

#[derive(Clone)]
struct Sample {
    depth: f64,
    values: Vec<Option<f64>>,
    orig_idx: usize,
}

#[derive(Clone)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

#[derive(Clone)]
struct Stat {
    median: f64,
    mad: f64,
    p05: f64,
    p95: f64,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn parse_f64(token: Option<&str>) -> Option<f64> {
    token.and_then(|t| t.parse::<f64>().ok())
}

fn median(sorted: &[f64]) -> f64 {
    let n = sorted.len();
    if n == 0 {
        return 0.0;
    }
    if n % 2 == 1 {
        sorted[n / 2]
    } else {
        (sorted[n / 2 - 1] + sorted[n / 2]) * 0.5
    }
}

fn percentile(sorted: &[f64], p: f64) -> f64 {
    if sorted.is_empty() {
        return 0.0;
    }
    let pos = p.clamp(0.0, 1.0) * (sorted.len() as f64 - 1.0);
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        sorted[lo]
    } else {
        let t = pos - lo as f64;
        sorted[lo] * (1.0 - t) + sorted[hi] * t
    }
}

fn robust_stats(values: &[f64]) -> Option<Stat> {
    if values.is_empty() {
        return None;
    }
    let mut v = values.to_vec();
    v.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median(&v);
    let mut devs: Vec<f64> = v.iter().map(|x| (x - med).abs()).collect();
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median(&devs);
    let p05 = percentile(&v, 0.05);
    let p95 = percentile(&v, 0.95);
    Some(Stat { median: med, mad, p05, p95 })
}

fn robust_z(value: f64, stat: &Stat) -> f64 {
    let scale = if stat.mad > 1e-12 { 1.4826 * stat.mad } else { ((stat.p95 - stat.p05).abs() / 3.2897).max(1e-12) };
    (value - stat.median).abs() / scale
}

fn local_score(samples: &[Sample], idx: usize, prop: usize, window: usize) -> Option<f64> {
    let n = samples.len();
    if idx >= n {
        return None;
    }
    let value = samples[idx].values.get(prop).and_then(|v| *v)?;
    let start = idx.saturating_sub(window);
    let end = (idx + window + 1).min(n);
    let mut neigh = Vec::with_capacity(end - start);
    for j in start..end {
        if j == idx { continue; }
        if let Some(v) = samples[j].values.get(prop).and_then(|x| *x) {
            neigh.push(v);
        }
    }
    if neigh.len() < 3 {
        return None;
    }
    neigh.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median(&neigh);
    let mut devs: Vec<f64> = neigh.iter().map(|x| (x - med).abs()).collect();
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median(&devs);
    let scale = if mad > 1e-12 { 1.4826 * mad } else { ((percentile(&neigh, 0.9) - percentile(&neigh, 0.1)).abs() / 2.563).max(1e-12) };
    Some((value - med).abs() / scale)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let num_holes: usize = match it.next().and_then(|t| t.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let num_props: usize = match it.next().and_then(|t| t.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(it.next().unwrap_or("").to_string());
    }

    let mut holes = Vec::with_capacity(num_holes);
    let mut global_values: Vec<Vec<f64>> = vec![Vec::new(); num_props];
    let mut total_samples = 0usize;

    for _ in 0..num_holes {
        let hole_id = match it.next() { Some(s) => s.to_string(), None => break };
        let _start_x = parse_f64(it.next());
        let _start_y = parse_f64(it.next());
        let _start_z = parse_f64(it.next());
        let _dir_x = parse_f64(it.next());
        let _dir_y = parse_f64(it.next());
        let _dir_z = parse_f64(it.next());
        let _length = parse_f64(it.next());
        let num_samples: usize = it.next().and_then(|t| t.parse().ok()).unwrap_or(0);

        let mut samples = Vec::with_capacity(num_samples);
        for sample_idx in 0..num_samples {
            let depth = parse_f64(it.next()).unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_props);
            for p in 0..num_props {
                let v = parse_f64(it.next());
                if let Some(x) = v {
                    if x.is_finite() {
                        global_values[p].push(x);
                    }
                }
                values.push(v.filter(|x| x.is_finite()));
            }
            samples.push(Sample { depth, values, orig_idx: sample_idx });
        }
        samples.sort_by(|a, b| {
            a.depth.partial_cmp(&b.depth)
                .unwrap_or(Ordering::Equal)
                .then_with(|| a.orig_idx.cmp(&b.orig_idx))
        });
        total_samples += samples.len();
        holes.push(Hole { id: hole_id, samples });
    }

    let globals: Vec<Option<Stat>> = global_values.iter().map(|v| robust_stats(v)).collect();

    let window = if total_samples > 100_000 { 2 } else if total_samples > 20_000 { 3 } else { 5 };
    let local_weight = if total_samples > 100_000 { 0.65 } else { 0.8 };
    let mut suspects = Vec::<Suspect>::new();

    for hole in &holes {
        let n = hole.samples.len();
        if n == 0 { continue; }
        for i in 0..n {
            for p in 0..num_props {
                let value = match hole.samples[i].values.get(p).and_then(|v| *v) {
                    Some(v) => v,
                    None => continue,
                };
                let mut scores = Vec::new();
                if let Some(s) = local_score(&hole.samples, i, p, window) {
                    scores.push(s);
                }
                if let Some(stat) = &globals[p] {
                    scores.push(robust_z(value, stat));
                }
                if scores.is_empty() { continue; }
                let mut best = scores[0];
                for s in scores.into_iter().skip(1) {
                    if s > best { best = s; }
                }
                let conf = best * local_weight + (best * best).sqrt() * (1.0 - local_weight);
                if conf >= 3.5 {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: hole.samples[i].orig_idx + 1,
                        property_name: prop_names[p].clone(),
                        confidence: conf,
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    let mut out = String::new();
    out.push_str(&format!("{}\n", suspects.len()));
    for s in suspects {
        out.push_str(&format!("{} {} {} {:.6}\n", s.hole_id, s.sample_idx, s.property_name, s.confidence));
    }
    io::stdout().write_all(out.as_bytes()).unwrap();
}
