using namespace std;

struct Graph {
    int n = 0, m = 0, k = 0;
    vector<int> head, to, next, deg;
    int edge_it = 0;

    Graph(int n_, int m_) : n(n_), m(m_), head(n_, -1), deg(n_, 0) {
        to.resize(2 * m_);
        next.resize(2 * m_);
    }

    void add_edge(int u, int v) {
        to[edge_it] = v;
        next[edge_it] = head[u];
        head[u] = edge_it++;
        deg[u]++;
    }
};

static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct Solver {
    Graph g;
    vector<int> color;
    vector<int> order;
    vector<int> mark;
    int stamp = 1;
    mt19937_64 rng;

    Solver(int n, int m, int k) : g(n, m), color(n, -1), mark(k + 1, 0) {
        rng.seed(chrono::high_resolution_clock::now().time_since_epoch().count());
    }

    bool greedy_order(const vector<int>& ord) {
        fill(color.begin(), color.end(), -1);
        vector<int> used(g.k, 0);
        for (int u : ord) {
            int banned = 0;
            for (int e = g.head[u]; e != -1; e = g.next[e]) {
                int v = g.to[e];
                int c = color[v];
                if (c >= 0 && c < g.k) used[c] = 1;
            }
            int c = -1;
            for (int i = 0; i < g.k; ++i) {
                if (!used[i]) { c = i; break; }
            }
            for (int e = g.head[u]; e != -1; e = g.next[e]) {
                int v = g.to[e];
                int c2 = color[v];
                if (c2 >= 0 && c2 < g.k) used[c2] = 0;
            }
            if (c == -1) return false;
            color[u] = c;
        }
        return true;
    }

    vector<int> build_degree_order() {
        vector<int> ord(g.n);
        for (int i = 0; i < g.n; ++i) ord[i] = i;
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
            return a < b;
        });
        return ord;
    }

    vector<int> build_randomized_order() {
        vector<int> ord(g.n);
        for (int i = 0; i < g.n; ++i) ord[i] = i;
        shuffle(ord.begin(), ord.end(), rng);
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
            return a < b;
        });
        return ord;
    }

    int choose_dsatur_vertex(const vector<int>& sat, const vector<int>& rem, const vector<char>& alive) {
        int best = -1;
        for (int u = 0; u < g.n; ++u) {
            if (!alive[u]) continue;
            if (best == -1 || sat[u] > sat[best] || (sat[u] == sat[best] && g.deg[u] > g.deg[best])) best = u;
        }
        return best;
    }

    bool dsatur_small() {
        fill(color.begin(), color.end(), -1);
        vector<int> sat(g.n, 0);
        vector<char> alive(g.n, 1);
        vector<int> neigh_colors(g.k + 1, 0);

        for (int step = 0; step < g.n; ++step) {
            int u = choose_dsatur_vertex(sat, sat, alive);
            if (u == -1) break;
            fill(neigh_colors.begin(), neigh_colors.end(), 0);
            for (int e = g.head[u]; e != -1; e = g.next[e]) {
                int v = g.to[e];
                int c = color[v];
                if (c >= 0 && c < g.k) neigh_colors[c] = 1;
            }
            int c = -1;
            for (int i = 0; i < g.k; ++i) {
                if (!neigh_colors[i]) { c = i; break; }
            }
            if (c == -1) return false;
            color[u] = c;
            alive[u] = 0;
            for (int e = g.head[u]; e != -1; e = g.next[e]) {
                int v = g.to[e];
                if (alive[v]) {
                    int before = sat[v];
                    int t = color[u];
                    if (before == 0) sat[v] = 1;
                    else {
                        // Approximate saturation updates using local neighborhood scans.
                        bool seen = false;
                        for (int ee = g.head[v]; ee != -1; ee = g.next[ee]) {
                            int w = g.to[ee];
                            if (color[w] == t) { seen = true; break; }
                        }
                        if (seen) sat[v] = before + 1;
                    }
                }
            }
        }
        return true;
    }

    bool color_with_restarts() {
        if (g.k <= 0) return false;
        if (g.n == 0) return true;

        vector<int> degree_order = build_degree_order();
        if (greedy_order(degree_order)) return true;

        for (int attempt = 0; attempt < 30; ++attempt) {
            vector<int> ord = build_randomized_order();
            if (greedy_order(ord)) return true;
        }

        if (g.n <= 200000) {
            for (int attempt = 0; attempt < 8; ++attempt) {
                if (dsatur_small()) return true;
                shuffle(degree_order.begin(), degree_order.end(), rng);
                sort(degree_order.begin(), degree_order.end(), [&](int a, int b) {
                    if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
                    return a < b;
                });
                if (greedy_order(degree_order)) return true;
            }
        }

        return false;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;
    Solver solver(n, m, k);
    solver.g.k = k;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
        solver.g.add_edge(u, v);
        solver.g.add_edge(v, u);
    }

    bool ok = false;
    if (k >= n) {
        for (int i = 0; i < n; ++i) solver.color[i] = i % k;
        ok = true;
    } else if (k == 1) {
        ok = (m == 0);
        if (ok) fill(solver.color.begin(), solver.color.end(), 0);
    } else {
        ok = solver.color_with_restarts();
    }

    if (!ok) {
        // Fallback: deterministic greedy by degree order, even if it may fail for the instance.
        vector<int> ord(n);
        for (int i = 0; i < n; ++i) ord[i] = i;
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            if (solver.g.deg[a] != solver.g.deg[b]) return solver.g.deg[a] > solver.g.deg[b];
            return a < b;
        });
        vector<int> col(n, -1), used(k, 0);
        for (int u : ord) {
            fill(used.begin(), used.end(), 0);
            for (int e = solver.g.head[u]; e != -1; e = solver.g.next[e]) {
                int v = solver.g.to[e];
                if (col[v] >= 0) used[col[v]] = 1;
            }
            int c = 0;
            while (c < k && used[c]) ++c;
            if (c == k) c = 0;
            col[u] = c;
        }
        solver.color.swap(col);
    }

    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << solver.color[i];
    }
    cout << '\n';
    return 0;
}
