import sys


def main():
    data = sys.stdin.buffer.read().splitlines()
    if not data:
        return

    first = data[0].split()
    if len(first) < 2:
        return
    U = int(first[0])
    S = int(first[1])

    sets = []
    elem_to_sets = [[] for _ in range(U)]
    for i in range(S):
        line = data[i + 1].split() if i + 1 < len(data) else []
        elems = []
        seen = set()
        for tok in line:
            x = int(tok)
            if 0 <= x < U and x not in seen:
                seen.add(x)
                elems.append(x)
        sets.append(elems)
        for e in elems:
            elem_to_sets[e].append(i)

    uncovered = set(range(U))
    chosen = []

    def backtrack():
        if not uncovered:
            return True

        # Choose the uncovered element with the fewest candidate sets.
        best_elem = None
        best_candidates = None
        for e in uncovered:
            candidates = [idx for idx in elem_to_sets[e] if all(x in uncovered for x in sets[idx])]
            if not candidates:
                return False
            if best_candidates is None or len(candidates) < len(best_candidates):
                best_elem = e
                best_candidates = candidates
                if len(best_candidates) == 1:
                    break

        for idx in best_candidates:
            elems = sets[idx]
            if any(e not in uncovered for e in elems):
                continue
            chosen.append(idx)
            removed = []
            for e in elems:
                uncovered.remove(e)
                removed.append(e)
            if backtrack():
                return True
            for e in removed:
                uncovered.add(e)
            chosen.pop()
        return False

    if backtrack():
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write(" ".join(map(str, chosen)) + "\n")
    else:
        sys.stdout.write("NO SOLUTION\n")


if __name__ == "__main__":
    main()
