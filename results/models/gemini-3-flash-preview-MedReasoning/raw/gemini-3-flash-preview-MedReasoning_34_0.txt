I have implemented a solution for the Capacitated Vehicle Routing Problem (CVRP) using the **Clarke-Wright Savings Algorithm**. This is a powerful and efficient construction heuristic that aims to minimize total travel distance while respecting vehicle capacity constraints.

### Key Features of the Implementation:
- **Heuristic Quality:** Uses the Clarke-Wright savings method, which consistently provides high-quality solutions for CVRP by greedily merging routes that offer the greatest distance reduction.
- **Robustness:** Handles various edge cases, including instances with no customers or single-customer routes. It uses an efficient input-reading strategy suitable for large datasets.
- **Efficiency:** The algorithm runs in approximately $O(N^2 \log N)$ time, making it suitable for instances with hundreds to thousands of customers.
- **Compliance:** Strictly follows the specified input/output formats and respects the vehicle capacity and customer coverage constraints.

The solution is contained in `solution.py` and is ready to be executed in a standard Python environment.

```python
import sys
import math

def solve():
    # Read all input from stdin
    try:
        data = sys.stdin.read().split()
    except EOFError:
        return
    if not data:
        return
    
    it = iter(data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAPACITY = int(next(it))
        
        depot_x = float(next(it))
        depot_y = float(next(it))
        
        cust_x = []
        cust_y = []
        demands = []
        for _ in range(N):
            cust_x.append(float(next(it)))
            cust_y.append(float(next(it)))
            demands.append(int(next(it)))
    except StopIteration:
        pass 

    if N == 0:
        print("0.0")
        print("0")
        return

    def dist(x1, y1, x2, y2):
        return math.sqrt((x1-x2)**2 + (y1-y2)**2)

    # Distances from depot to each customer
    d0 = [dist(depot_x, depot_y, cust_x[i], cust_y[i]) for i in range(N)]
    
    # Calculate savings for all pairs of customers: S_ij = d0_i + d0_j - d_ij
    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            d_ij = dist(cust_x[i], cust_y[i], cust_x[j], cust_y[j])
            s = d0[i] + d0[j] - d_ij
            if s > 0:
                savings.append((s, i, j))
    
    # Sort savings in descending order to prioritize the best merges
    savings.sort(key=lambda x: x[0], reverse=True)
    
    # Initial state: Each customer in their own route (Depot -> i -> Depot)
    routes = [[i] for i in range(N)]
    route_demands = [demands[i] for i in range(N)]
    customer_to_route = list(range(N))
    
    for s, i, j in savings:
        ri = customer_to_route[i]
        rj = customer_to_route[j]
        
        if ri == rj:
            continue
            
        if routes[ri] is None or routes[rj] is None:
            continue
            
        # Check if the combined demand fits in a single vehicle
        if route_demands[ri] + route_demands[rj] <= CAPACITY:
            route_i = routes[ri]
            route_j = routes[rj]
            
            # Merging is only possible if both customers are currently connected to the depot
            i_first = (route_i[0] == i)
            i_last = (route_i[-1] == i)
            j_first = (route_j[0] == j)
            j_last = (route_j[-1] == j)
            
            if (i_first or i_last) and (j_first or j_last):
                # Perform the merge based on which ends are connected
                if i_last and j_first:
                    res_route = route_i + route_j
                elif i_last and j_last:
                    res_route = route_i + route_j[::-1]
                elif i_first and j_first:
                    res_route = route_i[::-1] + route_j
                elif i_first and j_last:
                    res_route = route_j + route_i
                else:
                    continue
                
                # Update route data structures
                routes[ri] = res_route
                routes[rj] = None
                route_demands[ri] += route_demands[rj]
                for node in res_route:
                    customer_to_route[node] = ri

    # Filter to get only active routes
    final_routes = [r for r in routes if r is not None]
    
    # Calculate the final total distance (including returns to depot)
    total_distance = 0
    for r in final_routes:
        total_distance += d0[r[0]]
        for k in range(len(r) - 1):
            total_distance += dist(cust_x[r[k]], cust_y[r[k]], cust_x[r[k+1]], cust_y[r[k+1]])
        total_distance += d0[r[-1]]
        
    # Output the total distance, number of routes, and the routes themselves
    print(total_distance)
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    solve()
```
