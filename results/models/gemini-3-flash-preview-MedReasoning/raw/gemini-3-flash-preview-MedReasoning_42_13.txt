OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color_in %worldPos_in %normal_in
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputVertices 12

; Import GLSL instructions for InverseSqrt
%std = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%main_type = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%int = OpTypeInt 32 1

; Constants for indices
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

; Interface blocks/variables
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%color_in = OpVariable %ptr_in_arr_v3 Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointers for access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Math constants
%float_0_5 = OpConstant %float 0.5
%float_0_004 = OpConstant %float 0.004

; Main
%main = OpFunction %void None %main_type
%label_main = OpLabel

; Access gl_out.gl_Position pointer
%gl_out_p = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Edge (0, 2) ---
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

; avg color
%csum02 = OpFAdd %v3float %c0 %c2
%cavg02 = OpVectorTimesScalar %v3float %csum02 %float_0_5
OpStore %outColor %cavg02

; edge vector xy
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%edge02 = OpFSub %v2float %p2_xy %p0_xy

; normalize
%lenSq02 = OpDot %float %edge02 %edge02
%invLen02 = OpExtInst %float %std InverseSqrt %lenSq02
%dir02 = OpVectorTimesScalar %v2float %edge02 %invLen02

; perpendicular (-dy, dx)
%dir02_x = OpCompositeExtract %float %dir02 0
%dir02_y = OpCompositeExtract %float %dir02 1
%negDir02_y = OpFNegate %float %dir02_y
%perp02 = OpCompositeConstruct %v2float %negDir02_y %dir02_x

; offset
%offset02_base = OpVectorTimesScalar %v2float %perp02 %float_0_004
%w0 = OpCompositeExtract %float %p0 3
%w2 = OpCompositeExtract %float %p2 3
%offset02_0 = OpVectorTimesScalar %v2float %offset02_base %w0
%offset02_2 = OpVectorTimesScalar %v2float %offset02_base %w2

; Extract components for manual addition
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p2_x = OpCompositeExtract %float %p2 0
%p2_y = OpCompositeExtract %float %p2 1
%p2_z = OpCompositeExtract %float %p2 2

%off02_0x = OpCompositeExtract %float %offset02_0 0
%off02_0y = OpCompositeExtract %float %offset02_0 1
%off02_2x = OpCompositeExtract %float %offset02_2 0
%off02_2y = OpCompositeExtract %float %offset02_2 1

; V1: P0 + offset
%v1_x = OpFAdd %float %p0_x %off02_0x
%v1_y = OpFAdd %float %p0_y %off02_0y
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %w0
OpStore %gl_out_p %v1
OpEmitVertex

; V2: P0 - offset
%v2_x = OpFSub %float %p0_x %off02_0x
%v2_y = OpFSub %float %p0_y %off02_0y
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p0_z %w0
OpStore %gl_out_p %v2
OpEmitVertex

; V3: P2 + offset
%v3_x = OpFAdd %float %p2_x %off02_2x
%v3_y = OpFAdd %float %p2_y %off02_2y
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p2_z %w2
OpStore %gl_out_p %v3
OpEmitVertex

; V4: P2 - offset
%v4_x = OpFSub %float %p2_x %off02_2x
%v4_y = OpFSub %float %p2_y %off02_2y
%v4 = OpCompositeConstruct %v4float %v4_x %v4_y %p2_z %w2
OpStore %gl_out_p %v4
OpEmitVertex

OpEndPrimitive

; --- Edge (2, 4) ---
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color_in %int_4
%c4 = OpLoad %v3float %c4_ptr

; avg color
%csum24 = OpFAdd %v3float %c2 %c4
%cavg24 = OpVectorTimesScalar %v3float %csum24 %float_0_5
OpStore %outColor %cavg24

; edge vector xy
%p4_xy = OpVectorShuffle %v2float %p4 %p4 0 1
%edge24 = OpFSub %v2float %p4_xy %p2_xy

; normalize
%lenSq24 = OpDot %float %edge24 %edge24
%invLen24 = OpExtInst %float %std InverseSqrt %lenSq24
%dir24 = OpVectorTimesScalar %v2float %edge24 %invLen24

; perpendicular (-dy, dx)
%dir24_x = OpCompositeExtract %float %dir24 0
%dir24_y = OpCompositeExtract %float %dir24 1
%negDir24_y = OpFNegate %float %dir24_y
%perp24 = OpCompositeConstruct %v2float %negDir24_y %dir24_x

; offset
%offset24_base = OpVectorTimesScalar %v2float %perp24 %float_0_004
%w4 = OpCompositeExtract %float %p4 3
%offset24_2 = OpVectorTimesScalar %v2float %offset24_base %w2
%offset24_4 = OpVectorTimesScalar %v2float %offset24_base %w4

%p4_x = OpCompositeExtract %float %p4 0
%p4_y = OpCompositeExtract %float %p4 1
%p4_z = OpCompositeExtract %float %p4 2

%off24_2x = OpCompositeExtract %float %offset24_2 0
%off24_2y = OpCompositeExtract %float %offset24_2 1
%off24_4x = OpCompositeExtract %float %offset24_4 0
%off24_4y = OpCompositeExtract %float %offset24_4 1

; V1: P2 + offset
%v24_1_x = OpFAdd %float %p2_x %off24_2x
%v24_1_y = OpFAdd %float %p2_y %off24_2y
%v24_1 = OpCompositeConstruct %v4float %v24_1_x %v24_1_y %p2_z %w2
OpStore %gl_out_p %v24_1
OpEmitVertex

; V2: P2 - offset
%v24_2_x = OpFSub %float %p2_x %off24_2x
%v24_2_y = OpFSub %float %p2_y %off24_2y
%v24_2 = OpCompositeConstruct %v4float %v24_2_x %v24_2_y %p2_z %w2
OpStore %gl_out_p %v24_2
OpEmitVertex

; V3: P4 + offset
%v24_3_x = OpFAdd %float %p4_x %off24_4x
%v24_3_y = OpFAdd %float %p4_y %off24_4y
%v24_3 = OpCompositeConstruct %v4float %v24_3_x %v24_3_y %p4_z %w4
OpStore %gl_out_p %v24_3
OpEmitVertex

; V4: P4 - offset
%v24_4_x = OpFSub %float %p4_x %off24_4x
%v24_4_y = OpFSub %float %p4_y %off24_4y
%v24_4 = OpCompositeConstruct %v4float %v24_4_x %v24_4_y %p4_z %w4
OpStore %gl_out_p %v24_4
OpEmitVertex

OpEndPrimitive

; --- Edge (4, 0) ---
; avg color
%csum40 = OpFAdd %v3float %c4 %c0
%cavg40 = OpVectorTimesScalar %v3float %csum40 %float_0_5
OpStore %outColor %cavg40

; edge vector xy
%edge40 = OpFSub %v2float %p0_xy %p4_xy

; normalize
%lenSq40 = OpDot %float %edge40 %edge40
%invLen40 = OpExtInst %float %std InverseSqrt %lenSq40
%dir40 = OpVectorTimesScalar %v2float %edge40 %invLen40

; perpendicular (-dy, dx)
%dir40_x = OpCompositeExtract %float %dir40 0
%dir40_y = OpCompositeExtract %float %dir40 1
%negDir40_y = OpFNegate %float %dir40_y
%perp40 = OpCompositeConstruct %v2float %negDir40_y %dir40_x

; offset
%offset40_base = OpVectorTimesScalar %v2float %perp40 %float_0_004
%offset40_4 = OpVectorTimesScalar %v2float %offset40_base %w4
%offset40_0 = OpVectorTimesScalar %v2float %offset40_base %w0

%off40_4x = OpCompositeExtract %float %offset40_4 0
%off40_4y = OpCompositeExtract %float %offset40_4 1
%off40_0x = OpCompositeExtract %float %offset40_0 0
%off40_0y = OpCompositeExtract %float %offset40_0 1

; V1: P4 + offset
%v40_1_x = OpFAdd %float %p4_x %off40_4x
%v40_1_y = OpFAdd %float %p4_y %off40_4y
%v40_1 = OpCompositeConstruct %v4float %v40_1_x %v40_1_y %p4_z %w4
OpStore %gl_out_p %v40_1
OpEmitVertex

; V2: P4 - offset
%v40_2_x = OpFSub %float %p4_x %off40_4x
%v40_2_y = OpFSub %float %p4_y %off40_4y
%v40_2 = OpCompositeConstruct %v4float %v40_2_x %v40_2_y %p4_z %w4
OpStore %gl_out_p %v40_2
OpEmitVertex

; V3: P0 + offset
%v40_3_x = OpFAdd %float %p0_x %off40_0x
%v40_3_y = OpFAdd %float %p0_y %off40_0y
%v40_3 = OpCompositeConstruct %v4float %v40_3_x %v40_3_y %p0_z %w0
OpStore %gl_out_p %v40_3
OpEmitVertex

; V4: P0 - offset
%v40_4_x = OpFSub %float %p0_x %off40_0x
%v40_4_y = OpFSub %float %p0_y %off40_0y
%v40_4 = OpCompositeConstruct %v4float %v40_4_x %v40_4_y %p0_z %w0
OpStore %gl_out_p %v40_4
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
