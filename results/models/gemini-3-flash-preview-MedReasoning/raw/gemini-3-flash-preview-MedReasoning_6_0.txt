I have implemented a scalable and efficient Orbital Travelling Salesman Problem (OTSP) solver in C#. The solution uses a Greedy initial path followed by multiple restarts of a 2-opt refinement algorithm, ensuring high-quality solutions across all scales (2 to 100 stations).

### Key Features:
- **Scalability**: The algorithm handles small and large datasets efficiently, using up to 2000 restarts of the 2-opt heuristic within the allotted 5-minute time limit.
- **Physics-Based Distance Metric**: Instead of simple Euclidean distance, the solver uses a weighted distance metric in state space: $d = \sqrt{(n \Delta r)^2 + (\Delta v)^2}$, where $n$ is the average orbital frequency ($\sqrt{\mu/r^3}$). This metric serves as an excellent proxy for the $\Delta v$ required for impulsive orbital transfers, accounting for both position and velocity (including plane change costs).
- **Hamiltonian Path Optimization**: The 2-opt implementation is specifically tailored for a path starting at a fixed point and visiting all stations without returning to the start.
- **Performance**: The solver includes a safety check against a 5-minute execution limit and uses `InvariantCulture` to ensure robust handling of numerical data across different system locales.

The complete code is provided in `Program.cs`. To use it, compile with `csc Program.cs` and run the executable, providing the problem definition via standard input.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Globalization;

public class Program
{
    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    public struct State
    {
        public Vector3 Pos;
        public Vector3 Vel;
        public int Id;
    }

    public static void Main()
    {
        CultureInfo.DefaultThreadCurrentCulture = CultureInfo.InvariantCulture;

        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        int n = int.Parse(parts[0]);
        double mu = double.Parse(parts[1]);

        line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        State start = new State
        {
            Pos = new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])),
            Vel = new Vector3(double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5])),
            Id = -1
        };

        State[] stations = new State[n];
        for (int i = 0; i < n; i++)
        {
            line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) break;
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 6) break;
            stations[i] = new State
            {
                Pos = new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])),
                Vel = new Vector3(double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5])),
                Id = i
            };
        }

        int[] order = Solve(start, stations, mu);
        Console.WriteLine(string.Join(" ", order));
    }

    static int[] Solve(State start, State[] stations, double mu)
    {
        int n = stations.Length;
        if (n == 0) return new int[0];
        if (n == 1) return new[] { 0 };

        double sumR = start.Pos.Length();
        for (int i = 0; i < n; i++) sumR += stations[i].Pos.Length();
        double avgR = sumR / (n + 1);
        double orbital_n = Math.Sqrt(mu / (avgR * avgR * avgR));

        double[] distFromStart = new double[n];
        double[,] distMatrix = new double[n, n];

        for (int i = 0; i < n; i++)
        {
            distFromStart[i] = GetDistance(start, stations[i], orbital_n);
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                distMatrix[i, j] = GetDistance(stations[i], stations[j], orbital_n);
            }
        }

        int[] bestPath = new int[n];
        double bestCost = double.MaxValue;
        Random rng = new Random(42);
        var watch = System.Diagnostics.Stopwatch.StartNew();

        int numRestarts = n > 50 ? 1000 : 2000;
        if (n < 10) numRestarts = 100;

        for (int iter = 0; iter < numRestarts; iter++)
        {
            if (iter > 0 && iter % 10 == 0 && watch.ElapsedMilliseconds > 280000) break;

            int[] currentPath = new int[n];
            if (iter == 0)
            {
                bool[] visited = new bool[n];
                int last = -1;
                double minDist = double.MaxValue;
                for (int i = 0; i < n; i++)
                    if (distFromStart[i] < minDist) { minDist = distFromStart[i]; last = i; }
                currentPath[0] = last;
                visited[last] = true;
                for (int i = 1; i < n; i++)
                {
                    int next = -1;
                    minDist = double.MaxValue;
                    for (int j = 0; j < n; j++)
                        if (!visited[j] && distMatrix[last, j] < minDist) { minDist = distMatrix[last, j]; next = j; }
                    currentPath[i] = next;
                    visited[next] = true;
                    last = next;
                }
            }
            else
            {
                for (int i = 0; i < n; i++) currentPath[i] = i;
                for (int i = 0; i < n; i++)
                {
                    int swapIdx = rng.Next(i, n);
                    int tmp = currentPath[i];
                    currentPath[i] = currentPath[swapIdx];
                    currentPath[swapIdx] = tmp;
                }
            }

            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < n; i++)
                {
                    for (int j = i + 1; j < n; j++)
                    {
                        double d1_old = (i == 0) ? distFromStart[currentPath[i]] : distMatrix[currentPath[i - 1], currentPath[i]];
                        double d2_old = (j == n - 1) ? 0 : distMatrix[currentPath[j], currentPath[j + 1]];
                        double d1_new = (i == 0) ? distFromStart[currentPath[j]] : distMatrix[currentPath[i - 1], currentPath[j]];
                        double d2_new = (j == n - 1) ? 0 : distMatrix[currentPath[i], currentPath[j + 1]];

                        if (d1_new + d2_new < d1_old + d2_old - 1e-9)
                        {
                            Array.Reverse(currentPath, i, j - i + 1);
                            improved = true;
                        }
                    }
                }
            }

            double currentCost = distFromStart[currentPath[0]];
            for (int i = 0; i < n - 1; i++) currentCost += distMatrix[currentPath[i], currentPath[i + 1]];

            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                Array.Copy(currentPath, bestPath, n);
            }
        }
        return bestPath;
    }

    static double GetDistance(State a, State b, double orbital_n)
    {
        double dx = a.Pos.X - b.Pos.X, dy = a.Pos.Y - b.Pos.Y, dz = a.Pos.Z - b.Pos.Z;
        double dvx = a.Vel.X - b.Vel.X, dvy = a.Vel.Y - b.Vel.Y, dvz = a.Vel.Z - b.Vel.Z;
        return Math.Sqrt((dx * dx + dy * dy + dz * dz) * orbital_n * orbital_n + (dvx * dvx + dvy * dvy + dvz * dvz));
    }
}
```
