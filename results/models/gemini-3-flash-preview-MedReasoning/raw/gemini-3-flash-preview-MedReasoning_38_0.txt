The Exact Cover problem is solved using a backtracking approach based on Algorithm X. The implementation prioritizes efficiency and determinism by always selecting the element with the fewest covering options and processing them in a consistent order.

### Implementation Details:
- **Heuristic:** The solver employs the "Minimum Remaining Values" (MRV) heuristic, selecting the universe element covered by the fewest remaining sets at each step. This significantly prunes the search space.
- **Determinism:** To ensure the "planted" solution is found (if multiple exist), the solver uses deterministic tie-breaking for element selection and processes sets in their input order.
- **Robust Input Handling:** The script reads input line-by-line to correctly handle sets that may be empty or contain multiple elements, as specified in the problem format.
- **Validation:** Sets are pre-validated to ensure they only contain elements within the universe and do not contain internal duplicates, both of which would violate exact cover constraints.

```python
import sys

# Increase recursion depth for deep backtracking trees
sys.setrecursionlimit(10000)

def solve():
    # Read all lines from stdin
    lines = sys.stdin.readlines()
    if not lines:
        return
    
    # Find the first line with U and S
    line_idx = 0
    while line_idx < len(lines) and not lines[line_idx].strip():
        line_idx += 1
    
    if line_idx >= len(lines):
        return
    
    try:
        parts = lines[line_idx].split()
        U = int(parts[0])
        S = int(parts[1])
        line_idx += 1
    except (ValueError, IndexError):
        return

    # Read S lines, each being a set
    sets = []
    for _ in range(S):
        if line_idx < len(lines):
            try:
                sets.append(list(map(int, lines[line_idx].split())))
            except ValueError:
                sets.append([])
            line_idx += 1
        else:
            sets.append([])

    # Pre-validate sets: an exact cover set cannot have duplicates,
    # and all its elements must be within the universe [0, U-1].
    valid_set_flags = [True] * S
    for i, s in enumerate(sets):
        seen = set()
        for e in s:
            if e < 0 or e >= U or e in seen:
                valid_set_flags[i] = False
                break
            seen.add(e)

    # element_to_sets[e] stores indices of valid sets that contain element e
    element_to_sets = [[] for _ in range(U)]
    for i, s in enumerate(sets):
        if valid_set_flags[i]:
            for e in s:
                element_to_sets[e].append(i)

    # Backtracking state
    remaining_elements = set(range(U))
    excluded_sets = [not v for v in valid_set_flags]
    solution = []

    def backtrack():
        if not remaining_elements:
            return True
        
        # Heuristic: Pick the element covered by the fewest remaining sets
        # Sort remaining_elements to ensure deterministic behavior
        min_options = float('inf')
        best_e = -1
        
        for e in sorted(remaining_elements):
            options_count = 0
            for s_idx in element_to_sets[e]:
                if not excluded_sets[s_idx]:
                    options_count += 1
            
            if options_count < min_options:
                min_options = options_count
                best_e = e
                # Optimization: if we find an element that can't be covered, prioritize pruning
                if options_count == 0:
                    return False
                if options_count == 1:
                    break 
        
        if best_e == -1:
            return False
            
        # Try all sets that can cover the chosen element
        options = [s_idx for s_idx in element_to_sets[best_e] if not excluded_sets[s_idx]]
        
        for s_idx in options:
            solution.append(s_idx)
            
            # Elements covered by this set
            covered_in_this_step = sets[s_idx]
            for e in covered_in_this_step:
                remaining_elements.remove(e)
            
            # Sets that are now excluded because they overlap with this set
            excluded_in_this_step = []
            for e in covered_in_this_step:
                for other_s_idx in element_to_sets[e]:
                    if not excluded_sets[other_s_idx]:
                        excluded_sets[other_s_idx] = True
                        excluded_in_this_step.append(other_s_idx)
            
            if backtrack():
                return True
            
            # Backtrack state changes
            for other_s_idx in excluded_in_this_step:
                excluded_sets[other_s_idx] = False
            for e in covered_in_this_step:
                remaining_elements.add(e)
            solution.pop()
            
        return False

    if backtrack():
        print("SOLUTION")
        print(*(sorted(solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
```
