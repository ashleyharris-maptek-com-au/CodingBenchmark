using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPacking
{
    class Program
    {
        struct Box
        {
            public int Id;
            public int W, H, D;
            public long Vol;
            public int MinDim, MidDim, MaxDim;
        }

        struct Space
        {
            public int X, Y, Z;
            public int W, H, D;
            public long Vol;
        }

        struct Placement
        {
            public int Id;
            public int X, Y, Z;
        }

        static void Main()
        {
            string firstLine = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(firstLine)) return;

            string[] parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 4) return;

            if (!int.TryParse(parts[0], out int N)) return;
            if (!int.TryParse(parts[1], out int CW)) return;
            if (!int.TryParse(parts[2], out int CH)) return;
            if (!int.TryParse(parts[3], out int CD)) return;

            Box[] boxes = new Box[N];
            for (int i = 0; i < N; i++)
            {
                string line = Console.ReadLine();
                if (line == null) break;
                string[] dims = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                if (dims.Length < 3) continue;
                
                if (!int.TryParse(dims[0], out int w)) continue;
                if (!int.TryParse(dims[1], out int h)) continue;
                if (!int.TryParse(dims[2], out int d)) continue;
                
                int bMin = Math.Min(w, Math.Min(h, d));
                int bMax = Math.Max(w, Math.Max(h, d));
                int bMid = w + h + d - bMin - bMax;

                boxes[i] = new Box
                {
                    Id = i,
                    W = w,
                    H = h,
                    D = d,
                    Vol = (long)w * h * d,
                    MinDim = bMin,
                    MidDim = bMid,
                    MaxDim = bMax
                };
            }

            // Sort by volume ascending to maximize the number of boxes packed
            var sortedBoxes = boxes.OrderBy(b => b.Vol).ToArray();

            List<Space> freeSpaces = new List<Space>();
            freeSpaces.Add(new Space 
            { 
                X = 0, Y = 0, Z = 0, 
                W = CW, H = CH, D = CD, 
                Vol = (long)CW * CH * CD 
            });

            List<Placement> results = new List<Placement>(N);

            foreach (var box in sortedBoxes)
            {
                int bestSpaceIdx = -1;
                int bestW = 0, bestH = 0, bestD = 0;
                long minWaste = long.MaxValue;

                for (int i = 0; i < freeSpaces.Count; i++)
                {
                    Space s = freeSpaces[i];
                    if (box.Vol > s.Vol) continue;

                    // Quick check if any orientation fits in this space
                    int sMin = Math.Min(s.W, Math.Min(s.H, s.D));
                    int sMax = Math.Max(s.W, Math.Max(s.H, s.D));
                    int sMid = s.W + s.H + s.D - sMin - sMax;

                    if (box.MinDim > sMin || box.MidDim > sMid || box.MaxDim > sMax)
                        continue;

                    // Check all 6 orientations
                    Check(s, box.W, box.H, box.D, i, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref minWaste);
                    if (minWaste == 0) break;
                    Check(s, box.W, box.D, box.H, i, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref minWaste);
                    if (minWaste == 0) break;
                    Check(s, box.H, box.W, box.D, i, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref minWaste);
                    if (minWaste == 0) break;
                    Check(s, box.H, box.D, box.W, i, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref minWaste);
                    if (minWaste == 0) break;
                    Check(s, box.D, box.W, box.H, i, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref minWaste);
                    if (minWaste == 0) break;
                    Check(s, box.D, box.H, box.W, i, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref minWaste);
                    if (minWaste == 0) break;
                }

                if (bestSpaceIdx != -1)
                {
                    Space s = freeSpaces[bestSpaceIdx];
                    results.Add(new Placement { Id = box.Id, X = s.X, Y = s.Y, Z = s.Z });

                    // Remove used space efficiently
                    int lastIdx = freeSpaces.Count - 1;
                    Space chosenSpace = s;
                    freeSpaces[bestSpaceIdx] = freeSpaces[lastIdx];
                    freeSpaces.RemoveAt(lastIdx);

                    // Split chosenSpace into 3 disjoint boxes (Guillotine)
                    if (chosenSpace.W - bestW > 0)
                    {
                        freeSpaces.Add(new Space 
                        { 
                            X = chosenSpace.X + bestW, Y = chosenSpace.Y, Z = chosenSpace.Z, 
                            W = chosenSpace.W - bestW, H = chosenSpace.H, D = chosenSpace.D, 
                            Vol = (long)(chosenSpace.W - bestW) * chosenSpace.H * chosenSpace.D 
                        });
                    }
                    if (chosenSpace.H - bestH > 0)
                    {
                        freeSpaces.Add(new Space 
                        { 
                            X = chosenSpace.X, Y = chosenSpace.Y + bestH, Z = chosenSpace.Z, 
                            W = bestW, H = chosenSpace.H - bestH, D = chosenSpace.D, 
                            Vol = (long)bestW * (chosenSpace.H - bestH) * chosenSpace.D 
                        });
                    }
                    if (chosenSpace.D - bestD > 0)
                    {
                        freeSpaces.Add(new Space 
                        { 
                            X = chosenSpace.X, Y = chosenSpace.Y, Z = chosenSpace.Z + bestD, 
                            W = bestW, H = bestH, D = chosenSpace.D - bestD, 
                            Vol = (long)bestW * bestH * (chosenSpace.D - bestD) 
                        });
                    }
                }
            }

            Console.WriteLine(results.Count);
            foreach (var r in results)
            {
                Console.WriteLine($"{r.Id} {r.X} {r.Y} {r.Z}");
            }
        }

        static void Check(Space s, int bw, int bh, int bd, int idx, ref int bestIdx, ref int bestW, ref int bestH, ref int bestD, ref long minWaste)
        {
            if (bw <= s.W && bh <= s.H && bd <= s.D)
            {
                long waste = s.Vol - (long)bw * bh * bd;
                if (waste < minWaste)
                {
                    minWaste = waste;
                    bestIdx = idx;
                    bestW = bw;
                    bestH = bh;
                    bestD = bd;
                }
            }
        }
    }
}
