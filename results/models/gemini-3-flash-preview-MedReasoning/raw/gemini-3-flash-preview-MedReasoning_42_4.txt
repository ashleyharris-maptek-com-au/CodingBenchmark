OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
%main_type = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_2 = OpConstant %uint 2
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
 %normal_in = OpVariable %ptr_in_arr_v3 Input
  %color_in = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

 %float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

       %main = OpFunction %void None %main_type
 %main_label = OpLabel

    %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr

    %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr
    %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr

      %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
    %diff_xy = OpFSub %v2float %p1_xy %p0_xy
     %dir_xy = OpExtInst %v2float %1 Normalize %diff_xy

      %dir_x = OpCompositeExtract %float %dir_xy 0
      %dir_y = OpCompositeExtract %float %dir_xy 1
 %neg_dir_y = OpFNegate %float %dir_y
       %perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0: p0 + perp*0.04
   %offset0 = OpVectorTimesScalar %v2float %perp %float_0_04
      %v0_xy = OpFAdd %v2float %p0_xy %offset0
       %v0_x = OpCompositeExtract %float %v0_xy 0
       %v0_y = OpCompositeExtract %float %v0_xy 1
       %p0_z = OpCompositeExtract %float %p0 2
       %p0_w = OpCompositeExtract %float %p0 3
     %v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
               OpStore %out_pos_ptr %v0_pos
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 1: p0 - perp*0.04
      %v1_xy = OpFSub %v2float %p0_xy %offset0
       %v1_x = OpCompositeExtract %float %v1_xy 0
       %v1_y = OpCompositeExtract %float %v1_xy 1
     %v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
               OpStore %out_pos_ptr %v1_pos
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 2: p1 + perp*0.005
   %offset1 = OpVectorTimesScalar %v2float %perp %float_0_005
      %v2_xy = OpFAdd %v2float %p1_xy %offset1
       %v2_x = OpCompositeExtract %float %v2_xy 0
       %v2_y = OpCompositeExtract %float %v2_xy 1
       %p1_z = OpCompositeExtract %float %p1 2
       %p1_w = OpCompositeExtract %float %p1 3
     %v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
               OpStore %out_pos_ptr %v2_pos
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 3: p1 - perp*0.005
      %v3_xy = OpFSub %v2float %p1_xy %offset1
       %v3_x = OpCompositeExtract %float %v3_xy 0
       %v3_y = OpCompositeExtract %float %v3_xy 1
     %v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
               OpStore %out_pos_ptr %v3_pos
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
