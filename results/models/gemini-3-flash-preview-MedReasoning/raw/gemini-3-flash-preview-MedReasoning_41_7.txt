; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %normal %worldPos %outColor
               OpExecutionMode %main OriginUpperLeft

               ; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %u_ubo DescriptorSet 0
               OpDecorate %u_ubo Binding 0

               ; Types and Constants
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4

    %float_0 = OpConstant %float 0
  %float_0_2 = OpConstant %float 0.2
  %float_0_3 = OpConstant %float 0.3
  %float_0_5 = OpConstant %float 0.5
  %float_0_8 = OpConstant %float 0.8
  %float_1_0 = OpConstant %float 1.0
  %float_3_0 = OpConstant %float 3.0

%v3_base_color = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
%v3_rim_tint   = OpConstantComposite %v3float %float_1_0 %float_0_8 %float_0_5

               ; Variables
      %u_ubo = OpVariable %_ptr_Uniform_UBO Uniform
   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output

               ; Main
       %main = OpFunction %void None %3
      %label = OpLabel

; Normalize Normal
      %load_N = OpLoad %v3float %normal
          %N = OpExtInst %v3float %1 Normalize %load_N

; Get worldPos
   %load_pos = OpLoad %v3float %worldPos

; Get LightPos.xyz
      %ptr_LP = OpAccessChain %_ptr_Uniform_v4float %u_ubo %int_3
      %load_LP = OpLoad %v4float %ptr_LP
         %LP = OpVectorShuffle %v3float %load_LP %load_LP 0 1 2

; Get CameraPos.xyz
      %ptr_CP = OpAccessChain %_ptr_Uniform_v4float %u_ubo %int_4
      %load_CP = OpLoad %v4float %ptr_CP
         %CP = OpVectorShuffle %v3float %load_CP %load_CP 0 1 2

; L = normalize(LP - worldPos)
      %L_un   = OpFSub %v3float %LP %load_pos
      %L      = OpExtInst %v3float %1 Normalize %L_un

; V = normalize(CP - worldPos)
      %V_un   = OpFSub %v3float %CP %load_pos
      %V      = OpExtInst %v3float %1 Normalize %V_un

; diffuse = max(dot(N, L), 0.0)
     %dotNL   = OpDot %float %N %L
    %diffuse  = OpExtInst %float %1 FMax %dotNL %float_0

; rim = pow(1.0 - max(dot(N, V), 0.0), 3.0)
     %dotNV   = OpDot %float %N %V
   %maxNV     = OpExtInst %float %1 FMax %dotNV %float_0
   %rim_base  = OpFSub %float %float_1_0 %maxNV
   %rim       = OpExtInst %float %1 Pow %rim_base %float_3_0

; base_color * (0.3 + 0.5 * diffuse)
  %diff_scale = OpFMul %float %float_0_5 %diffuse
  %diff_term  = OpFAdd %float %float_0_3 %diff_scale
  %vec_diff   = OpVectorTimesScalar %v3float %v3_base_color %diff_term

; rim_tint * rim
  %vec_rim    = OpVectorTimesScalar %v3float %v3_rim_tint %rim

; final_color = vec_diff + vec_rim
   %final_v3  = OpFAdd %v3float %vec_diff %vec_rim

; outColor = vec4(final_color, 1.0)
   %out_v4    = OpCompositeConstruct %v4float %final_v3 %float_1_0
                OpStore %outColor %out_v4

                OpReturn
                OpFunctionEnd
