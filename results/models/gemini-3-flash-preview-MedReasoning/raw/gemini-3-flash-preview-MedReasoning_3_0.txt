use std::io::{self, Read, Write, BufWriter};
use std::thread;

/// A simple force-directed graph layout solver (Fruchterman-Reingold)
/// optimized for performance and scalability.
fn main() {
    // Fast I/O: read all input at once
    let mut input = String::new();
    if let Err(_) = io::stdin().read_to_string(&mut input) {
        return;
    }
    let mut tokens = input.split_whitespace();

    // Parse N and M
    let n: usize = match tokens.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = match tokens.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => 0,
    };

    // Build adjacency list
    // Handles parallel edges and disconnected components naturally.
    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match tokens.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        let v: usize = match tokens.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue; // Skip out of bounds and self-loops
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    if n == 0 {
        return;
    }

    // Positions: Initialized randomly in a 1000x1000 box
    let mut x = vec![0.0f32; n];
    let mut y = vec![0.0f32; n];

    let mut rng = 12345u64;
    for i in 0..n {
        // LCG for fast pseudo-randomness
        rng = rng.wrapping_mul(6364136223846793005).wrapping_add(1);
        x[i] = ((rng >> 32) as f32 / 4294967296.0) * 1000.0;
        rng = rng.wrapping_mul(6364136223846793005).wrapping_add(1);
        y[i] = ((rng >> 32) as f32 / 4294967296.0) * 1000.0;
    }

    // Parameters for Fruchterman-Reingold
    // Iterations decrease as N increases to stay within time limits.
    let iterations = if n < 100 {
        500
    } else if n < 1000 {
        200
    } else if n < 10000 {
        100
    } else if n < 100000 {
        70
    } else {
        50
    };

    let area = 1000.0 * 1000.0f32;
    let k = (area / n as f32).sqrt(); // Optimal distance between nodes
    let k_sq = k * k;

    // Initial temperature: proportional to scale or a base value
    let mut temp = 100.0f32.max(k);
    let dt = temp / iterations as f32;

    let num_threads = 12; // Environment has 12 cores

    for _iter in 0..iterations {
        let mut dx = vec![0.0f32; n];
        let mut dy = vec![0.0f32; n];

        // 1. Calculate bounding box for grid-based repulsion
        let mut min_x = x[0];
        let mut max_x = x[0];
        let mut min_y = y[0];
        let mut max_y = y[0];
        for i in 1..n {
            if x[i] < min_x { min_x = x[i]; }
            if x[i] > max_x { max_x = x[i]; }
            if y[i] < min_y { min_y = y[i]; }
            if y[i] > max_y { max_y = y[i]; }
        }

        // 2. Build Grid for Spatial Partitioning
        // Grid size is roughly sqrt(N) / 2 to keep O(1) nodes per cell.
        let grid_res = ((n as f32).sqrt() / 2.0).max(1.0) as usize;
        let mut head = vec![-1isize; grid_res * grid_res];
        let mut next = vec![-1isize; n];
        let cell_w = (max_x - min_x + 0.001) / grid_res as f32;
        let cell_h = (max_y - min_y + 0.001) / grid_res as f32;

        for i in 0..n {
            let gx = (((x[i] - min_x) / cell_w) as usize).min(grid_res - 1);
            let gy = (((y[i] - min_y) / cell_h) as usize).min(grid_res - 1);
            let idx = gy * grid_res + gx;
            next[i] = head[idx];
            head[idx] = i as isize;
        }

        // 3. Parallel Force Calculation
        thread::scope(|s| {
            let chunk_size = (n + num_threads - 1) / num_threads;
            let mut dx_chunks = dx.chunks_mut(chunk_size);
            let mut dy_chunks = dy.chunks_mut(chunk_size);
            
            for t in 0..num_threads {
                let start_idx = t * chunk_size;
                if start_idx >= n { break; }
                
                let dx_chunk = dx_chunks.next().unwrap();
                let dy_chunk = dy_chunks.next().unwrap();
                
                // Borrow context for threads
                let x_ref = &x;
                let y_ref = &y;
                let head_ref = &head;
                let next_ref = &next;
                let adj_ref = &adj;

                s.spawn(move || {
                    for i_in_chunk in 0..dx_chunk.len() {
                        let i = start_idx + i_in_chunk;
                        let xi = x_ref[i];
                        let yi = y_ref[i];
                        let mut dxi = 0.0f32;
                        let mut dyi = 0.0f32;
                        
                        // Repulsive forces: Only from nearby cells in the grid
                        let gx = (((xi - min_x) / cell_w) as usize).min(grid_res - 1);
                        let gy = (((yi - min_y) / cell_h) as usize).min(grid_res - 1);
                        
                        for ny in (gy as isize - 1)..=(gy as isize + 1) {
                            if ny < 0 || ny >= grid_res as isize { continue; }
                            for nx in (gx as isize - 1)..=(gx as isize + 1) {
                                if nx < 0 || nx >= grid_res as isize { continue; }
                                let n_idx = (ny as usize) * grid_res + (nx as usize);
                                let mut curr = head_ref[n_idx];
                                while curr != -1 {
                                    let j = curr as usize;
                                    if i != j {
                                        let dxx = xi - x_ref[j];
                                        let dyy = yi - y_ref[j];
                                        let dist_sq = dxx * dxx + dyy * dyy;
                                        if dist_sq < 0.0001 {
                                            // Break overlap with a small nudge
                                            dxi += k * 0.1;
                                        } else if dist_sq < 100.0 * k_sq {
                                            // Force falls off, we limit range for performance
                                            let dist = dist_sq.sqrt();
                                            let force = k_sq / dist;
                                            dxi += (dxx / dist) * force;
                                            dyi += (dyy / dist) * force;
                                        }
                                    }
                                    curr = next_ref[j];
                                }
                            }
                        }
                        
                        // Attractive forces: Only between connected nodes
                        for &v in &adj_ref[i] {
                            let dxx = xi - x_ref[v];
                            let dyy = yi - y_ref[v];
                            let dist_sq = dxx * dxx + dyy * dyy;
                            if dist_sq > 0.0001 {
                                let dist = dist_sq.sqrt();
                                let force = dist_sq / k;
                                dxi -= (dxx / dist) * force;
                                dyi -= (dyy / dist) * force;
                            }
                        }
                        
                        // Gravity: Slight pull towards the center to keep components together
                        let dxx = xi - 500.0;
                        let dyy = yi - 500.0;
                        let dist_sq = dxx * dxx + dyy * dyy;
                        if dist_sq > 1.0 {
                            let dist = dist_sq.sqrt();
                            let force = dist / 100.0;
                            dxi -= (dxx / dist) * force;
                            dyi -= (dyy / dist) * force;
                        }
                        
                        dx_chunk[i_in_chunk] = dxi;
                        dy_chunk[i_in_chunk] = dyi;
                    }
                });
            }
        });

        // 4. Update positions with cooling
        for i in 0..n {
            let d_sq = dx[i] * dx[i] + dy[i] * dy[i];
            if d_sq > 0.0001 {
                let d = d_sq.sqrt();
                let limited = d.min(temp);
                x[i] += (dx[i] / d) * limited;
                y[i] += (dy[i] / d) * limited;
            }
        }
        
        // Cool down
        temp -= dt;
        if temp < 0.0 { temp = 0.0; }
    }

    // Output results
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    // Prefix with N as allowed by the format
    if let Err(_) = writeln!(writer, "{}", n) { return; }
    for i in 0..n {
        if let Err(_) = writeln!(writer, "{:.6} {:.6}", x[i], y[i]) { break; }
    }
}
