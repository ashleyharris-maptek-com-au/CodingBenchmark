; SPIR-V
; Version: 1.0
; Generator: Manual
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%uboVar = OpVariable %_ptr_Uniform_UBO Uniform

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

; Function type
%void_func = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32

%vec3_ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05
%vec3_specular_color = OpConstantComposite %v3float %float_1 %float_1 %float_1
%vec3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%vec3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

; Function main
%main = OpFunction %void None %void_func
%label = OpLabel

; Load inputs
%in_worldPos = OpLoad %v3float %worldPos
%in_normal = OpLoad %v3float %normal
%in_color = OpLoad %v3float %color

; Get lightPos and cameraPos from UBO
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_3
%in_lightPos_v4 = OpLoad %v4float %lightPos_ptr
%in_lightPos = OpVectorShuffle %v3float %in_lightPos_v4 %in_lightPos_v4 0 1 2

%cameraPos_ptr = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_4
%in_cameraPos_v4 = OpLoad %v4float %cameraPos_ptr
%in_cameraPos = OpVectorShuffle %v3float %in_cameraPos_v4 %in_cameraPos_v4 0 1 2

; Normalize normal
%N = OpExtInst %v3float %1 Normalize %in_normal

; Light direction L
%L_unnorm = OpFSub %v3float %in_lightPos %in_worldPos
%L = OpExtInst %v3float %1 Normalize %L_unnorm

; View direction V
%V_unnorm = OpFSub %v3float %in_cameraPos %in_worldPos
%V = OpExtInst %v3float %1 Normalize %V_unnorm

; Diffuse: max(dot(N, L), 0) * color
%dotNL = OpDot %float %N %L
%diffuse_factor = OpExtInst %float %1 FMax %dotNL %float_0
%diffuse = OpVectorTimesScalar %v3float %in_color %diffuse_factor

; Specular: R = reflect(-L, N), factor = pow(max(dot(V, R), 0), 32)
%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %1 Reflect %negL %N
%dotVR = OpDot %float %V %R
%spec_base = OpExtInst %float %1 FMax %dotVR %float_0
%spec_factor = OpExtInst %float %1 Pow %spec_base %float_32
%specular = OpVectorTimesScalar %v3float %vec3_specular_color %spec_factor

; Combine
%sum1 = OpFAdd %v3float %vec3_ambient %diffuse
%sum2 = OpFAdd %v3float %sum1 %specular

; Clamp and Alpha
%final_color_v3 = OpExtInst %v3float %1 FClamp %sum2 %vec3_0 %vec3_1
%final_color = OpCompositeConstruct %v4float %final_color_v3 %float_1

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
