The complete C# implementation for an optimal Tetris player is provided below. This program uses an evaluation function based on Pierre Dellacherie's algorithm, which considers aggregate height, cleared lines, holes, and bumpiness to determine the best move. It also features adaptive logic that shifts priorities (e.g., focusing on height reduction and line clearing) as the game stack grows higher or when playing on narrower boards.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class TetrisPlayer
{
    static int Width, Height;
    static long NumPieces;
    static bool[,] Board;

    static int[][][] PieceShapes = new int[7][][];
    static string[] PieceNames = { "I", "O", "T", "S", "Z", "J", "L" };

    public static void Main(string[] args)
    {
        InitializePieces();

        // Read initial configuration: width height num_pieces
        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        string[] parts = line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 3) return;

        if (!int.TryParse(parts[0], out Width)) return;
        if (!int.TryParse(parts[1], out Height)) return;
        if (!long.TryParse(parts[2], out NumPieces)) return;

        Board = new bool[Height, Width];

        for (long i = 0; i < NumPieces; i++)
        {
            string pieceType = Console.ReadLine()?.Trim();
            if (string.IsNullOrEmpty(pieceType)) break;

            // Handle potential extra characters in pieceType
            if (pieceType.Contains(" ")) pieceType = pieceType.Split(' ')[0];

            // Determine the optimal rotation and column
            var bestMove = FindBestMove(pieceType);
            
            // Output move: rotation column
            Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");

            // Receive result: ok, cleared_N, or gameover
            string result = Console.ReadLine();
            if (result == null || result.Contains("gameover")) break;

            // Synchronize internal board state
            ApplyMove(pieceType, bestMove.Rotation, bestMove.Column);
            ClearFullLines();
        }
    }

    static void InitializePieces()
    {
        // Definitions for each piece type and its 4 rotations (r, c offsets)
        // I
        PieceShapes[0] = new int[][] {
            new int[] { 0, 0, 0, 1, 0, 2, 0, 3 }, new int[] { 0, 0, 1, 0, 2, 0, 3, 0 },
            new int[] { 0, 0, 0, 1, 0, 2, 0, 3 }, new int[] { 0, 0, 1, 0, 2, 0, 3, 0 }
        };
        // O
        PieceShapes[1] = new int[][] {
            new int[] { 0, 0, 0, 1, 1, 0, 1, 1 }, new int[] { 0, 0, 0, 1, 1, 0, 1, 1 },
            new int[] { 0, 0, 0, 1, 1, 0, 1, 1 }, new int[] { 0, 0, 0, 1, 1, 0, 1, 1 }
        };
        // T
        PieceShapes[2] = new int[][] {
            new int[] { 0, 1, 1, 0, 1, 1, 1, 2 }, new int[] { 0, 0, 1, 0, 1, 1, 2, 0 },
            new int[] { 0, 0, 0, 1, 0, 2, 1, 1 }, new int[] { 0, 1, 1, 0, 1, 1, 2, 1 }
        };
        // S
        PieceShapes[3] = new int[][] {
            new int[] { 0, 1, 0, 2, 1, 0, 1, 1 }, new int[] { 0, 0, 1, 0, 1, 1, 2, 1 },
            new int[] { 0, 1, 0, 2, 1, 0, 1, 1 }, new int[] { 0, 0, 1, 0, 1, 1, 2, 1 }
        };
        // Z
        PieceShapes[4] = new int[][] {
            new int[] { 0, 0, 0, 1, 1, 1, 1, 2 }, new int[] { 0, 1, 1, 0, 1, 1, 2, 0 },
            new int[] { 0, 0, 0, 1, 1, 1, 1, 2 }, new int[] { 0, 1, 1, 0, 1, 1, 2, 0 }
        };
        // J
        PieceShapes[5] = new int[][] {
            new int[] { 0, 0, 1, 0, 1, 1, 1, 2 }, new int[] { 0, 0, 0, 1, 1, 0, 2, 0 },
            new int[] { 0, 0, 0, 1, 0, 2, 1, 2 }, new int[] { 0, 1, 1, 1, 2, 0, 2, 1 }
        };
        // L
        PieceShapes[6] = new int[][] {
            new int[] { 0, 2, 1, 0, 1, 1, 1, 2 }, new int[] { 0, 0, 1, 0, 2, 0, 2, 1 },
            new int[] { 0, 0, 0, 1, 0, 2, 1, 0 }, new int[] { 0, 0, 0, 1, 1, 1, 2, 1 }
        };
    }

    static int GetPieceType(string type)
    {
        for (int i = 0; i < PieceNames.Length; i++)
            if (string.Equals(PieceNames[i], type, StringComparison.OrdinalIgnoreCase)) return i;
        return 0;
    }

    struct Move
    {
        public int Rotation;
        public int Column;
        public double Score;
    }

    static Move FindBestMove(string typeStr)
    {
        int type = GetPieceType(typeStr);
        Move bestMove = new Move { Score = double.NegativeInfinity, Rotation = 0, Column = 0 };

        for (int rot = 0; rot < 4; rot++)
        {
            int[] shape = PieceShapes[type][rot];
            int pieceWidth = 0;
            for (int i = 0; i < shape.Length; i += 2)
                pieceWidth = Math.Max(pieceWidth, shape[i + 1] + 1);

            for (int col = 0; col <= Width - pieceWidth; col++)
            {
                int row = GetDropRow(shape, col);
                if (row < 0) continue;

                double score = EvaluateMove(shape, row, col);
                if (score > bestMove.Score)
                    bestMove = new Move { Rotation = rot, Column = col, Score = score };
            }
        }
        return bestMove;
    }

    static int GetDropRow(int[] shape, int col)
    {
        int pieceHeight = 0;
        for (int i = 0; i < shape.Length; i += 2)
            pieceHeight = Math.Max(pieceHeight, shape[i] + 1);

        int bestR = -1;
        for (int r = 0; r <= Height - pieceHeight; r++)
        {
            if (!IsCollision(shape, r, col))
                bestR = r;
            else
                break;
        }
        return bestR;
    }

    static bool IsCollision(int[] shape, int row, int col)
    {
        for (int i = 0; i < shape.Length; i += 2)
        {
            int r = row + shape[i], c = col + shape[i + 1];
            if (r < 0 || r >= Height || c < 0 || c >= Width || Board[r, c]) return true;
        }
        return false;
    }

    static double EvaluateMove(int[] shape, int row, int col)
    {
        for (int i = 0; i < shape.Length; i += 2)
            Board[row + shape[i], col + shape[i + 1]] = true;

        bool[] rowFull = new bool[Height];
        int clearedLines = 0;
        for (int r = 0; r < Height; r++)
        {
            bool full = true;
            for (int c = 0; c < Width; c++) if (!Board[r, c]) { full = false; break; }
            if (full) { rowFull[r] = true; clearedLines++; }
        }

        int[] heights = new int[Width];
        int holes = 0, totalNonFull = Height - clearedLines;

        for (int c = 0; c < Width; c++)
        {
            int hVirtual = 0, colHeight = 0;
            bool blockFound = false;
            for (int r = 0; r < Height; r++)
            {
                if (rowFull[r]) continue;
                hVirtual++;
                if (Board[r, c]) { if (!blockFound) { colHeight = totalNonFull - hVirtual + 1; blockFound = true; } }
                else if (blockFound) holes++;
            }
            heights[c] = colHeight;
        }

        double score = CalculateScore(clearedLines, heights, holes);

        for (int i = 0; i < shape.Length; i += 2)
            Board[row + shape[i], col + shape[i + 1]] = false;

        return score;
    }

    static double CalculateScore(int clearedLines, int[] heights, int holes)
    {
        double aggregateHeight = heights.Sum();
        double bumpiness = 0;
        for (int c = 0; c < Width - 1; c++) bumpiness += Math.Abs(heights[c] - heights[c + 1]);

        int maxHeight = heights.Length > 0 ? heights.Max() : 0;
        double wHeight = -0.51, wLines = 0.76, wHoles = -0.36, wBumpiness = -0.18;

        // Adaptive logic: Narrow boards require stricter control
        if (Width < 8) { wHoles -= 0.25; wBumpiness -= 0.15; }

        // Adaptive logic: High stacks prioritize clearing and height reduction
        if (maxHeight > Height * 0.7) { wHeight -= 0.6; wLines += 1.5; wHoles -= 0.2; }
        else if (maxHeight > Height * 0.4) { wHeight -= 0.2; wLines += 0.5; }

        double score = (wHeight * aggregateHeight) + (wLines * clearedLines) + (wHoles * holes) + (wBumpiness * bumpiness);
        if (clearedLines == 4) score += 5.0; // Tetris bonus
        score -= maxHeight * 0.2; // Peak penalty

        return score;
    }

    static void ApplyMove(string typeStr, int rot, int col)
    {
        int type = GetPieceType(typeStr);
        int[] shape = PieceShapes[type][rot];
        int row = GetDropRow(shape, col);
        if (row >= 0)
        {
            for (int i = 0; i < shape.Length; i += 2)
            {
                int r = row + shape[i], c = col + shape[i + 1];
                if (r >= 0 && r < Height && c >= 0 && c < Width) Board[r, c] = true;
            }
        }
    }

    static void ClearFullLines()
    {
        List<int> fullRows = new List<int>();
        for (int r = 0; r < Height; r++)
        {
            bool full = true;
            for (int c = 0; c < Width; c++) if (!Board[r, c]) { full = false; break; }
            if (full) fullRows.Add(r);
        }

        foreach (int row in fullRows)
        {
            for (int r = row; r > 0; r--)
                for (int c = 0; c < Width; c++) Board[r, c] = Board[r - 1, c];
            for (int c = 0; c < Width; c++) Board[0, c] = false;
        }
    }
}
```
