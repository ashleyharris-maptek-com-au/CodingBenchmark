OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 8

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
  %void_func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

  %arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Constants
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
 %float_0_01 = OpConstant %float 0.01
%float_0_333333 = OpConstant %float 0.333333
%float_0_666667 = OpConstant %float 0.666667

       %main = OpFunction %void None %void_func
      %label = OpLabel

; Load positions
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr
     %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr
     %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %p3_ptr

; Compute 2D positions (xy / w)
      %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
       %p0_w = OpCompositeExtract %float %p0 3
    %p0_w_v2 = OpCompositeConstruct %v2float %p0_w %p0_w
      %p0_2d = OpFDiv %v2float %p0_xy %p0_w_v2

      %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
       %p1_w = OpCompositeExtract %float %p1 3
    %p1_w_v2 = OpCompositeConstruct %v2float %p1_w %p1_w
      %p1_2d = OpFDiv %v2float %p1_xy %p1_w_v2

      %p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
       %p2_w = OpCompositeExtract %float %p2 3
    %p2_w_v2 = OpCompositeConstruct %v2float %p2_w %p2_w
      %p2_2d = OpFDiv %v2float %p2_xy %p2_w_v2

      %p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
       %p3_w = OpCompositeExtract %float %p3 3
    %p3_w_v2 = OpCompositeConstruct %v2float %p3_w %p3_w
      %p3_2d = OpFDiv %v2float %p3_xy %p3_w_v2

; dir1 = normalize(p2_2d - p0_2d)
      %diff1 = OpFSub %v2float %p2_2d %p0_2d
       %dir1 = OpExtInst %v2float %1 Normalize %diff1

; dir2 = normalize(p3_2d - p1_2d)
      %diff2 = OpFSub %v2float %p3_2d %p1_2d
       %dir2 = OpExtInst %v2float %1 Normalize %diff2

; norm1 = vec2(-dir1.y, dir1.x) * 0.01
        %d1y = OpCompositeExtract %float %dir1 1
        %d1x = OpCompositeExtract %float %dir1 0
       %nd1y = OpFNegate %float %d1y
        %n1r = OpCompositeConstruct %v2float %nd1y %d1x
      %norm1 = OpVectorTimesScalar %v2float %n1r %float_0_01

; norm2 = vec2(-dir2.y, dir2.x) * 0.01
        %d2y = OpCompositeExtract %float %dir2 1
        %d2x = OpCompositeExtract %float %dir2 0
       %nd2y = OpFNegate %float %d2y
        %n2r = OpCompositeConstruct %v2float %nd2y %d2x
      %norm2 = OpVectorTimesScalar %v2float %n2r %float_0_01

; Colors
     %c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %c1_ptr
     %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %c2_ptr

%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Segment 1 (t=0 to t=1/3)
; t=0
  %n1_off_v2 = OpVectorTimesScalar %v2float %norm1 %p1_w
   %n1_off_x = OpCompositeExtract %float %n1_off_v2 0
   %n1_off_y = OpCompositeExtract %float %n1_off_v2 1
     %n1_off = OpCompositeConstruct %v4float %n1_off_x %n1_off_y %float_0 %float_0

       %v0_p = OpFAdd %v4float %p1 %n1_off
               OpStore %pos_out_ptr %v0_p
               OpStore %outColor %c1
               OpEmitVertex
       %v1_p = OpFSub %v4float %p1 %n1_off
               OpStore %pos_out_ptr %v1_p
               OpStore %outColor %c1
               OpEmitVertex

; t=1/3
        %p13 = OpExtInst %v4float %1 FMix %p1 %p2 %float_0_333333
       %n13_2 = OpExtInst %v2float %1 FMix %norm1 %norm2 %float_0_333333
       %p13w = OpCompositeExtract %float %p13 3
 %n13_off_v2 = OpVectorTimesScalar %v2float %n13_2 %p13w
  %n13_off_x = OpCompositeExtract %float %n13_off_v2 0
  %n13_off_y = OpCompositeExtract %float %n13_off_v2 1
    %n13_off = OpCompositeConstruct %v4float %n13_off_x %n13_off_y %float_0 %float_0
        %c13 = OpExtInst %v3float %1 FMix %c1 %c2 %float_0_333333

       %v2_p = OpFAdd %v4float %p13 %n13_off
               OpStore %pos_out_ptr %v2_p
               OpStore %outColor %c13
               OpEmitVertex
       %v3_p = OpFSub %v4float %p13 %n13_off
               OpStore %pos_out_ptr %v3_p
               OpStore %outColor %c13
               OpEmitVertex
               OpEndPrimitive

; Segment 2 (t=2/3 to t=1)
; t=2/3
        %p23 = OpExtInst %v4float %1 FMix %p1 %p2 %float_0_666667
       %n23_2 = OpExtInst %v2float %1 FMix %norm1 %norm2 %float_0_666667
       %p23w = OpCompositeExtract %float %p23 3
 %n23_off_v2 = OpVectorTimesScalar %v2float %n23_2 %p23w
  %n23_off_x = OpCompositeExtract %float %n23_off_v2 0
  %n23_off_y = OpCompositeExtract %float %n23_off_v2 1
    %n23_off = OpCompositeConstruct %v4float %n23_off_x %n23_off_y %float_0 %float_0
        %c23 = OpExtInst %v3float %1 FMix %c1 %c2 %float_0_666667

       %v4_p = OpFAdd %v4float %p23 %n23_off
               OpStore %pos_out_ptr %v4_p
               OpStore %outColor %c23
               OpEmitVertex
       %v5_p = OpFSub %v4float %p23 %n23_off
               OpStore %pos_out_ptr %v5_p
               OpStore %outColor %c23
               OpEmitVertex

; t=1
  %n2_off_v2 = OpVectorTimesScalar %v2float %norm2 %p2_w
   %n2_off_x = OpCompositeExtract %float %n2_off_v2 0
   %n2_off_y = OpCompositeExtract %float %n2_off_v2 1
     %n2_off = OpCompositeConstruct %v4float %n2_off_x %n2_off_y %float_0 %float_0

       %v6_p = OpFAdd %v4float %p2 %n2_off
               OpStore %pos_out_ptr %v6_p
               OpStore %outColor %c2
               OpEmitVertex
       %v7_p = OpFSub %v4float %p2 %n2_off
               OpStore %pos_out_ptr %v7_p
               OpStore %outColor %c2
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
