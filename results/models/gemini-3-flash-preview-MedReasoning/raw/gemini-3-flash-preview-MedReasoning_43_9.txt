struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalization of interpolated normal
    float3 N = normalize(input.normal);
    
    // Light direction for a point light in world space
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Lambertian diffuse component
    float diffuse = max(dot(N, L), 0.0);
    
    // Custom shading factor: (0.5 + 0.5 * diffuse)
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // 8x8 UV checkerboard logic
    float2 uv8 = floor(input.uv * 8.0);
    float parity = fmod(uv8.x + uv8.y, 2.0);
    float checker = (abs(parity) < 0.5) ? 0.9 : 0.2;

    // Final color calculation with alpha 1.0
    float colorVal = checker * shadingFactor;
    return float4(colorVal, colorVal, colorVal, 1.0);
}
