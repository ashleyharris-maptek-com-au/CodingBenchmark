#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against gl_GlobalInvocationID.x >= N
    if (i >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Load current body state
    vec4 posMass_i = inData[i * 2];
    vec3 pos_i = posMass_i.xyz;
    float mass_i = posMass_i.w;
    vec3 vel_i = inData[i * 2 + 1].xyz;

    vec3 acc = vec3(0.0);

    // Compute gravitational acceleration from all other bodies
    for (uint j = 0; j < N; j++) {
        // Skip self-interaction
        if (i == j) continue;

        vec4 posMass_j = inData[j * 2];
        vec3 pos_j = posMass_j.xyz;
        float mass_j = posMass_j.w;

        vec3 r = pos_j - pos_i;
        float distSqr = dot(r, r);
        
        // Use the softening parameter to avoid singularities
        // Denominator: (|pos_j - pos_i|^2 + softening^2)^(3/2)
        float invDist = inversesqrt(distSqr + softening * softening);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    // Integration step (Semi-implicit Euler / Leapfrog-like)
    // Velocity is advanced by acc * dt
    vel_i += acc * dt;
    // Position is advanced by updated velocity * dt
    pos_i += vel_i * dt;

    // Write updated position+mass and velocity back to output buffer
    outData[i * 2] = vec4(pos_i, mass_i);
    outData[i * 2 + 1] = vec4(vel_i, 0.0);
}
