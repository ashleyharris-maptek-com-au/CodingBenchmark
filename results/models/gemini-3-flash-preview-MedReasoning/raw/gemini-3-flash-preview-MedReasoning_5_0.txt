use std::io::{self, BufRead};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let parts: Vec<i32> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    if parts.len() < 4 {
        return;
    }
    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2] as usize;
    let require_cycle = parts[3] == 1;

    let mut obstacles = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        if let Some(Ok(line)) = lines.next() {
            let coords: Vec<i32> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            if coords.len() == 2 {
                obstacles.push((coords[0], coords[1]));
            }
        }
    }

    // Use a separate thread with a large stack to handle deep recursion on large grids.
    let builder = std::thread::Builder::new().stack_size(256 * 1024 * 1024);
    let handler = builder.spawn(move || {
        if let Some(mut solver) = Solver::new(width, height, obstacles, require_cycle) {
            if solver.solve() {
                println!("{}", solver.path.len());
                for (x, y) in &solver.path {
                    println!("{} {}", x, y);
                }
            } else {
                println!("0");
            }
        } else {
            println!("0");
        }
    }).unwrap();
    handler.join().unwrap();
}

struct Solver {
    width: i32,
    height: i32,
    blocked: Vec<bool>,
    visited: Vec<bool>,
    degrees: Vec<i8>,
    deg_counts: [usize; 5],
    total_to_visit: usize,
    require_cycle: bool,
    path: Vec<(i32, i32)>,
}

impl Solver {
    fn new(width: i32, height: i32, obstacles: Vec<(i32, i32)>, require_cycle: bool) -> Option<Self> {
        if width <= 0 || height <= 0 {
            return None;
        }
        let size = (width * height) as usize;
        let mut blocked = vec![false; size];
        for (x, y) in obstacles {
            if x >= 0 && x < width && y >= 0 && y < height {
                blocked[(y * width + x) as usize] = true;
            }
        }

        if blocked[0] {
            return None;
        }

        let mut total_non_obstacles = 0;
        let mut even_count = 0;
        let mut odd_count = 0;
        for y in 0..height {
            for x in 0..width {
                if !blocked[(y * width + x) as usize] {
                    total_non_obstacles += 1;
                    if (x + y) % 2 == 0 {
                        even_count += 1;
                    } else {
                        odd_count += 1;
                    }
                }
            }
        }

        // Bipartition check: necessary condition for Hamiltonian path/cycle in grid graphs.
        if require_cycle {
            if total_non_obstacles % 2 != 0 || even_count != odd_count {
                return None;
            }
        } else {
            if total_non_obstacles % 2 == 0 {
                if even_count != odd_count {
                    return None;
                }
            } else {
                if even_count != odd_count + 1 {
                    return None;
                }
            }
        }

        // Reachability check: all non-obstacle cells must be reachable from (0,0).
        let mut q = VecDeque::new();
        let mut reachable = vec![false; size];
        q.push_back((0, 0));
        reachable[0] = true;
        let mut reachable_count = 0;
        while let Some((cx, cy)) = q.pop_front() {
            reachable_count += 1;
            for (dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
                let nx = cx + dx;
                let ny = cy + dy;
                if nx >= 0 && nx < width && ny >= 0 && ny < height {
                    let idx = (ny * width + nx) as usize;
                    if !blocked[idx] && !reachable[idx] {
                        reachable[idx] = true;
                        q.push_back((nx, ny));
                    }
                }
            }
        }
        if reachable_count < total_non_obstacles {
            return None;
        }

        let mut degrees = vec![0i8; size];
        let mut deg_counts = [0usize; 5];
        for y in 0..height {
            for x in 0..width {
                let idx = (y * width + x) as usize;
                if !blocked[idx] {
                    let mut d = 0;
                    for (dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
                        let nx = x + dx;
                        let ny = y + dy;
                        if nx >= 0 && nx < width && ny >= 0 && ny < height {
                            if !blocked[(ny * width + nx) as usize] {
                                d += 1;
                            }
                        }
                    }
                    degrees[idx] = d;
                    deg_counts[d as usize] += 1;
                }
            }
        }

        Some(Solver {
            width,
            height,
            blocked,
            visited: vec![false; size],
            degrees,
            deg_counts,
            total_to_visit: total_non_obstacles,
            require_cycle,
            path: Vec::with_capacity(total_non_obstacles),
        })
    }

    fn solve(&mut self) -> bool {
        self.set_visited(0, true);
        self.path.push((0, 0));
        for (dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
            let nx = 0 + dx;
            let ny = 0 + dy;
            if self.is_in_bounds(nx, ny) {
                self.update_degree((ny * self.width + nx) as usize, -1);
            }
        }

        self.backtrack(0, 0)
    }

    fn is_in_bounds(&self, x: i32, y: i32) -> bool {
        x >= 0 && x < self.width && y >= 0 && y < self.height
    }

    fn set_visited(&mut self, idx: usize, v: bool) {
        let deg = self.degrees[idx] as usize;
        if v {
            self.visited[idx] = true;
            self.deg_counts[deg] -= 1;
        } else {
            self.visited[idx] = false;
            self.deg_counts[deg] += 1;
        }
    }

    fn update_degree(&mut self, idx: usize, delta: i8) {
        if self.visited[idx] || self.blocked[idx] {
            self.degrees[idx] += delta;
            return;
        }
        let old_deg = self.degrees[idx] as usize;
        self.degrees[idx] += delta;
        let new_deg = self.degrees[idx] as usize;
        self.deg_counts[old_deg] -= 1;
        self.deg_counts[new_deg] += 1;
    }

    fn backtrack(&mut self, cx: i32, cy: i32) -> bool {
        if self.path.len() == self.total_to_visit {
            if self.require_cycle {
                return cx.abs() + cy.abs() == 1;
            }
            return true;
        }

        // Pruning: if any unvisited cell becomes unreachable (degree 0)
        // and it's not the final cell we need to visit.
        if self.deg_counts[0] > 0 {
            if self.deg_counts[0] > 1 || self.path.len() + 1 < self.total_to_visit {
                return false;
            }
        }

        // Pruning: for Hamiltonian Path, at most two cells can have degree 1
        // (the current neighbor we move to and the ultimate end).
        if self.deg_counts[1] > 2 {
            return false;
        }

        let mut neighbors = [(0, 0, 0i8); 4];
        let mut count = 0;
        let mut forced = None;

        for &(dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
            let nx = cx + dx;
            let ny = cy + dy;
            if self.is_in_bounds(nx, ny) {
                let idx = (ny * self.width + nx) as usize;
                if !self.blocked[idx] && !self.visited[idx] {
                    let d = self.degrees[idx];
                    if d == 0 {
                        // This neighbor has no other unvisited neighbors.
                        // We must move to it now, and it must be the last cell.
                        if self.path.len() + 1 == self.total_to_visit {
                            forced = Some((nx, ny, d));
                        } else {
                            // If we move here, we'll be stuck before visiting everyone.
                            return false;
                        }
                    }
                    neighbors[count] = (nx, ny, d);
                    count += 1;
                }
            }
        }

        if let Some(f) = forced {
            // Only keep the forced neighbor
            let mut i = 0;
            while i < count {
                if neighbors[i].0 == f.0 && neighbors[i].1 == f.1 {
                    neighbors[0] = neighbors[i];
                    count = 1;
                    break;
                }
                i += 1;
            }
        }

        if count == 0 {
            return false;
        }

        // Warnsdorff's Rule: always move to the neighbor with the minimum degree.
        if count > 1 {
            neighbors[0..count].sort_by_key(|n| n.2);
        }

        for i in 0..count {
            let (nx, ny, _) = neighbors[i];
            let idx = (ny * self.width + nx) as usize;
            self.set_visited(idx, true);
            self.path.push((nx, ny));
            for &(ddx, ddy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
                let nnx = nx + ddx;
                let nny = ny + ddy;
                if self.is_in_bounds(nnx, nny) {
                    self.update_degree((nny * self.width + nnx) as usize, -1);
                }
            }

            if self.backtrack(nx, ny) {
                return true;
            }

            // Backtrack: restore state
            for &(ddx, ddy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
                let nnx = nx + ddx;
                let nny = ny + ddy;
                if self.is_in_bounds(nnx, nny) {
                    self.update_degree((nny * self.width + nnx) as usize, 1);
                }
            }
            self.path.pop();
            self.set_visited(idx, false);
        }

        false
    }
}
