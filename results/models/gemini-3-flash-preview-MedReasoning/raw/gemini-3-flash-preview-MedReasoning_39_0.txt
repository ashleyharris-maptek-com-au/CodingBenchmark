use std::io::{self, BufRead};

#[derive(Clone)]
struct BitSet {
    data: Vec<u64>,
}

impl BitSet {
    fn new(n: usize) -> Self {
        BitSet {
            data: vec![0; (n + 63) / 64],
        }
    }
    fn set(&mut self, i: usize) {
        self.data[i / 64] |= 1 << (i % 64);
    }
    fn unset(&mut self, i: usize) {
        self.data[i / 64] &= !(1 << (i % 64));
    }
    fn get(&self, i: usize) -> bool {
        (self.data[i / 64] >> (i % 64)) & 1 == 1
    }
    fn count_ones_and_not(&self, other: &BitSet) -> usize {
        self.data
            .iter()
            .zip(&other.data)
            .map(|(a, b)| (a & !b).count_ones() as usize)
            .sum()
    }
    fn count_ones_and_and(&self, mask: &BitSet, other: &BitSet) -> usize {
        self.data
            .iter()
            .zip(&mask.data)
            .zip(&other.data)
            .map(|((a, m), b)| (a & m & b).count_ones() as usize)
            .sum()
    }
    fn count_ones_and_and_not(&self, mask: &BitSet, other: &BitSet) -> usize {
        self.data
            .iter()
            .zip(&mask.data)
            .zip(&other.data)
            .map(|((a, m), b)| (a & m & !b).count_ones() as usize)
            .sum()
    }
}

fn calculate_cut(n: usize, adj: &[BitSet], partition: &BitSet) -> usize {
    let mut cut = 0;
    for i in 0..n {
        if partition.get(i) {
            cut += adj[i].count_ones_and_not(partition);
        }
    }
    cut
}

fn solve(
    idx: usize,
    count_a: usize,
    current_cut: usize,
    n: usize,
    adj: &[BitSet],
    assigned_masks: &[BitSet],
    partition: &mut BitSet,
    best_cut: &mut usize,
    best_partition: &mut Vec<bool>,
) {
    let rem = n - idx;
    let rem_a = n / 2 - count_a;
    let rem_b = rem - rem_a;

    if idx == n {
        if current_cut < *best_cut {
            *best_cut = current_cut;
            for i in 0..n {
                best_partition[i] = partition.get(i);
            }
        }
        return;
    }

    // Lower bound pruning
    let mut sum_na = 0;
    let mut diffs = Vec::with_capacity(rem);
    let mask = &assigned_masks[idx];
    for i in idx..n {
        let na = adj[i].count_ones_and_and(mask, partition);
        let nb = adj[i].count_ones_and_and_not(mask, partition);
        sum_na += na;
        diffs.push(nb as i32 - na as i32);
    }
    
    let mut min_additional = sum_na as i32;
    if rem_a > 0 {
        if rem_a < diffs.len() {
            let (less, median, _greater) = diffs.select_nth_unstable(rem_a - 1);
            for d in less {
                min_additional += *d;
            }
            min_additional += *median;
        } else {
            for d in diffs {
                min_additional += d;
            }
        }
    }
    if current_cut + (min_additional.max(0) as usize) >= *best_cut {
        return;
    }

    // Choice order
    let na = adj[idx].count_ones_and_and(mask, partition);
    let nb = adj[idx].count_ones_and_and_not(mask, partition);
    
    if nb < na {
        if count_a < n / 2 {
            partition.set(idx);
            solve(idx + 1, count_a + 1, current_cut + nb, n, adj, assigned_masks, partition, best_cut, best_partition);
            partition.unset(idx);
        }
        if idx - count_a < n / 2 {
            solve(idx + 1, count_a, current_cut + na, n, adj, assigned_masks, partition, best_cut, best_partition);
        }
    } else {
        if idx - count_a < n / 2 {
            solve(idx + 1, count_a, current_cut + na, n, adj, assigned_masks, partition, best_cut, best_partition);
        }
        if count_a < n / 2 {
            partition.set(idx);
            solve(idx + 1, count_a + 1, current_cut + nb, n, adj, assigned_masks, partition, best_cut, best_partition);
            partition.unset(idx);
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let mut parts = first_line.split_whitespace();
    let n: usize = match parts.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = match parts.next() {
        Some(s) => s.parse().unwrap(),
        None => 0,
    };

    if n == 0 {
        println!("0\n");
        return;
    }

    let mut adj = vec![BitSet::new(n); n];
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut edge = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (edge.next(), edge.next()) {
                let u: usize = u_str.parse().unwrap();
                let v: usize = v_str.parse().unwrap();
                if u < n && v < n && u != v {
                    adj[u].set(v);
                    adj[v].set(u);
                }
            }
        }
    }

    // Initial heuristic
    let mut partition = BitSet::new(n);
    let mut count_a = 0;
    let mut count_b = 0;
    partition.set(0);
    count_a = 1;
    let mut assigned = BitSet::new(n);
    assigned.set(0);

    for i in 1..n {
        let na = adj[i].count_ones_and_and(&assigned, &partition);
        let nb = adj[i].count_ones_and_and_not(&assigned, &partition);
        if count_a < n / 2 && count_b < n / 2 {
            if nb <= na { partition.set(i); count_a += 1; }
            else { count_b += 1; }
        } else if count_a < n / 2 { partition.set(i); count_a += 1; }
        else { count_b += 1; }
        assigned.set(i);
    }
    
    let mut best_cut = calculate_cut(n, &adj, &partition);
    let mut best_partition = vec![false; n];
    for i in 0..n { best_partition[i] = partition.get(i); }

    let mut assigned_masks = Vec::with_capacity(n);
    let mut current_mask = BitSet::new(n);
    for i in 0..n {
        assigned_masks.push(current_mask.clone());
        current_mask.set(i);
    }

    let mut partition = BitSet::new(n);
    partition.set(0);
    solve(1, 1, 0, n, &adj, &assigned_masks, &mut partition, &mut best_cut, &mut best_partition);

    println!("{}", best_cut);
    let mut res = Vec::new();
    for i in 0..n {
        if best_partition[i] {
            res.push(i.to_string());
        }
    }
    println!("{}", res.join(" "));
}
