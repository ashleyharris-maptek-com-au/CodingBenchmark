#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <cmath>

using namespace std;

typedef vector<int> Point;

struct PointHash {
    size_t operator()(const Point& v) const {
        size_t seed = v.size();
        for (int x : v) {
            seed ^= hash<int>{}(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

int D;
int max_turns;
Point bounds;
deque<Point> snake;
unordered_map<Point, int, PointHash> snake_map;
unordered_set<Point, PointHash> obstacles;
vector<Point> foods;

bool is_valid(const Point& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    if (obstacles.count(p)) return false;
    if (snake_map.count(p)) return false;
    return true;
}

int manhattan(const Point& a, const Point& b) {
    int dist = 0;
    for (int i = 0; i < D; ++i) {
        dist += abs(a[i] - b[i]);
    }
    return dist;
}

struct Node {
    Point p;
    int g;
    int h;
    bool operator>(const Node& other) const {
        return (g + h) > (other.g + other.h);
    }
};

vector<Point> find_path(const Point& start, const Point& target) {
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    unordered_map<Point, int, PointHash> g_score;
    unordered_map<Point, Point, PointHash> parent;

    pq.push({start, 0, manhattan(start, target)});
    g_score[start] = 0;

    int visited_count = 0;
    const int MAX_VISITED = 20000;

    while (!pq.empty()) {
        Node current = pq.top();
        pq.pop();

        if (current.p == target) {
            vector<Point> path;
            Point curr = target;
            while (curr != start) {
                path.push_back(curr);
                curr = parent[curr];
            }
            reverse(path.begin(), path.end());
            return path;
        }

        if (current.g > g_score[current.p]) continue;
        if (++visited_count > MAX_VISITED) break;

        for (int i = 0; i < D; ++i) {
            for (int dir : {-1, 1}) {
                Point next = current.p;
                next[i] += dir;

                if (next == target || is_valid(next)) {
                    int next_g = current.g + 1;
                    if (g_score.find(next) == g_score.end() || next_g < g_score[next]) {
                        g_score[next] = next_g;
                        parent[next] = current.p;
                        pq.push({next, next_g, manhattan(next, target)});
                    }
                }
            }
        }
    }
    return {};
}

Point get_survival_move() {
    Point head = snake.front();
    Point best_neighbor;
    int max_free = -1;

    for (int i = 0; i < D; ++i) {
        for (int dir : {-1, 1}) {
            Point next = head;
            next[i] += dir;
            if (is_valid(next)) {
                int free_count = 0;
                for (int j = 0; j < D; ++j) {
                    for (int d2 : {-1, 1}) {
                        Point n2 = next;
                        n2[j] += d2;
                        if (is_valid(n2)) free_count++;
                    }
                }
                if (free_count > max_free) {
                    max_free = free_count;
                    best_neighbor = next;
                }
            }
        }
    }
    
    if (best_neighbor.empty()) {
        for (int i = 0; i < D; ++i) {
            for (int dir : {-1, 1}) {
                Point next = head;
                next[i] += dir;
                bool ok = true;
                for (int k = 0; k < D; ++k) if (next[k] < 0 || next[k] >= bounds[k]) ok = false;
                if (ok && !obstacles.count(next) && !snake_map.count(next)) return next;
            }
        }
        for (int i = 0; i < D; ++i) {
            for (int dir : {-1, 1}) {
                Point next = head;
                next[i] += dir;
                bool ok = true;
                for (int k = 0; k < D; ++k) if (next[k] < 0 || next[k] >= bounds[k]) ok = false;
                if (ok && !obstacles.count(next)) return next;
            }
        }
    }
    return best_neighbor;
}

void move_snake(const Point& next_head) {
    snake.push_front(next_head);
    snake_map[next_head]++;
    
    bool ate = false;
    for (auto it = foods.begin(); it != foods.end(); ++it) {
        if (*it == next_head) {
            foods.erase(it);
            ate = true;
            break;
        }
    }
    
    if (!ate) {
        Point tail = snake.back();
        snake.pop_back();
        if (--(snake_map[tail]) == 0) {
            snake_map.erase(tail);
        }
    }
}

void output_move(const Point& from, const Point& to) {
    if (to.empty()) {
        cout << "0 1" << endl;
        return;
    }
    for (int i = 0; i < D; ++i) {
        if (to[i] != from[i]) {
            cout << i << " " << (to[i] > from[i] ? 1 : -1) << endl;
            return;
        }
    }
    cout << "0 1" << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> D >> max_turns)) return 0;
    bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_len;
    cin >> snake_len;
    Point start_pos(D);
    for (int i = 0; i < D; ++i) cin >> start_pos[i];
    for (int i = 0; i < snake_len; ++i) {
        snake.push_back(start_pos);
    }
    snake_map[start_pos] = snake_len;

    int F;
    if (!(cin >> F)) F = 0;
    foods.clear();
    for (int i = 0; i < F; ++i) {
        Point f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        foods.push_back(f);
    }

    int O;
    if (!(cin >> O)) O = 0;
    obstacles.clear();
    for (int i = 0; i < O; ++i) {
        Point obs(D);
        for (int j = 0; j < D; ++j) cin >> obs[j];
        obstacles.insert(obs);
    }

    for (int turn = 0; turn < max_turns; ++turn) {
        if (foods.empty()) break;

        Point head = snake.front();
        sort(foods.begin(), foods.end(), [&](const Point& a, const Point& b) {
            return manhattan(head, a) < manhattan(head, b);
        });

        vector<Point> path;
        int check_count = min((int)foods.size(), 5);
        for (int i = 0; i < check_count; ++i) {
            path = find_path(head, foods[i]);
            if (!path.empty()) break;
        }

        Point next_move;
        if (!path.empty()) {
            next_move = path[0];
        } else {
            next_move = get_survival_move();
        }

        output_move(head, next_move);
        if (next_move.empty()) break;
        move_snake(next_move);
    }

    return 0;
}
