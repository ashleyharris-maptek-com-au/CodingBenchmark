The following Rust solution implements an adaptive clustering algorithm designed to handle 3D point datasets ranging from millions to billions of points.

### Key Features:
- **Scalability**: It uses a sampling-based approach for large datasets. It samples up to 2 million points (adjustable) to train centroids and then performs a single streaming pass for the entire dataset.
- **Performance**: 
    - Uses k-means++ for high-quality centroid initialization.
    - Employs a **KD-tree** for fast nearest-neighbor lookups during both training (when K is large) and the final assignment phase.
    - Single-pass assignment ensures minimal I/O overhead.
- **Memory Efficiency**: The algorithm is fully streaming for the assignment phase, allowing it to handle datasets much larger than available RAM.
- **Binary I/O**: Optimized with large buffers (`BufReader`/`BufWriter`) for high-speed binary processing.

```rust
use std::io::{self, Read, Write, BufReader, BufWriter, BufRead};

struct Xorshift {
    state: u64,
}

impl Xorshift {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 1 } else { seed } }
    }
    fn next_u64(&mut self) -> u64 {
        self.state ^= self.state << 13;
        self.state ^= self.state >> 7;
        self.state ^= self.state << 17;
        self.state
    }
    fn next_f32(&mut self) -> f32 {
        (self.next_u64() & 0xFFFFFF) as f32 / 16777216.0
    }
}

fn dist_sq(a: [f32; 3], b: [f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn find_nearest_brute_force(p: [f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best_idx = 0;
    let mut best_dist_sq = dist_sq(p, centroids[0]);
    for i in 1..centroids.len() {
        let d2 = dist_sq(p, centroids[i]);
        if d2 < best_dist_sq {
            best_dist_sq = d2;
            best_idx = i;
        }
    }
    best_idx
}

struct KdNode {
    centroid_idx: usize,
    left: i32,
    right: i32,
    split_dim: usize,
}

struct KdTree {
    nodes: Vec<KdNode>,
    centroids: Vec<[f32; 3]>,
}

impl KdTree {
    fn build(centroids: &Vec<[f32; 3]>) -> Self {
        let mut indices: Vec<usize> = (0..centroids.len()).collect();
        let mut nodes = Vec::new();
        if !indices.is_empty() {
            Self::build_recursive(centroids, &mut indices, 0, &mut nodes);
        }
        KdTree { nodes, centroids: centroids.clone() }
    }

    fn build_recursive(centroids: &[[f32; 3]], indices: &mut [usize], depth: usize, nodes: &mut Vec<KdNode>) -> i32 {
        if indices.is_empty() { return -1; }
        let split_dim = depth % 3;
        indices.sort_unstable_by(|&a, &b| centroids[a][split_dim].partial_cmp(&centroids[b][split_dim]).unwrap());
        let mid = indices.len() / 2;
        let centroid_idx = indices[mid];
        let node_idx = nodes.len();
        nodes.push(KdNode { centroid_idx, left: -1, right: -1, split_dim });
        let left = Self::build_recursive(centroids, &mut indices[..mid], depth + 1, nodes);
        let right = Self::build_recursive(centroids, &mut indices[mid+1..], depth + 1, nodes);
        nodes[node_idx].left = left;
        nodes[node_idx].right = right;
        node_idx as i32
    }

    fn find_nearest(&self, p: [f32; 3]) -> usize {
        if self.nodes.is_empty() { return 0; }
        let mut best_idx = self.nodes[0].centroid_idx;
        let mut best_dist_sq = dist_sq(p, self.centroids[best_idx]);
        self.search(0, p, &mut best_idx, &mut best_dist_sq);
        best_idx
    }

    fn search(&self, node_idx: i32, p: [f32; 3], best_idx: &mut usize, best_dist_sq: &mut f32) {
        if node_idx == -1 { return; }
        let node = &self.nodes[node_idx as usize];
        let c = self.centroids[node.centroid_idx];
        let d2 = dist_sq(p, c);
        if d2 < *best_dist_sq {
            *best_dist_sq = d2;
            *best_idx = node.centroid_idx;
        }
        let diff = p[node.split_dim] - c[node.split_dim];
        let (near, far) = if diff <= 0.0 { (node.left, node.right) } else { (node.right, node.left) };
        self.search(near, p, best_idx, best_dist_sq);
        if diff * diff < *best_dist_sq {
            self.search(far, p, best_idx, best_dist_sq);
        }
    }
}

fn main() -> io::Result<()> {
    let stdin_raw = io::stdin().lock();
    let mut stdin = BufReader::with_capacity(1 << 20, stdin_raw);
    let stdout_raw = io::stdout().lock();
    let mut out_writer = BufWriter::with_capacity(1 << 20, stdout_raw);

    let mut num_points = 0u64;
    let mut num_clusters = 0usize;
    let mut found = 0;
    while found < 2 {
        let mut line = Vec::new();
        if stdin.read_until(b'\n', &mut line)? == 0 { break; }
        let s = String::from_utf8_lossy(&line);
        for word in s.split_whitespace() {
            if found == 0 {
                num_points = word.parse().unwrap_or(0);
                found = 1;
            } else if found == 1 {
                num_clusters = word.parse().unwrap_or(0);
                found = 2;
            }
        }
    }

    if num_points == 0 || num_clusters == 0 { return Ok(()); }

    let sample_limit = 2_000_000;
    let sample_size = if num_points < sample_limit as u64 { num_points as usize } else { sample_limit };
    let mut points = Vec::with_capacity(sample_size);

    let mut p_buf = [0u8; 12];
    for _ in 0..sample_size {
        if stdin.read_exact(&mut p_buf).is_err() { break; }
        points.push([
            f32::from_le_bytes([p_buf[0], p_buf[1], p_buf[2], p_buf[3]]),
            f32::from_le_bytes([p_buf[4], p_buf[5], p_buf[6], p_buf[7]]),
            f32::from_le_bytes([p_buf[8], p_buf[9], p_buf[10], p_buf[11]]),
        ]);
    }

    let mut centroids = Vec::with_capacity(num_clusters);
    let mut rng = Xorshift::new(42);
    if num_clusters >= points.len() {
        for i in 0..num_clusters {
            centroids.push(if i < points.len() { points[i] } else { [0.0, 0.0, 0.0] });
        }
    } else {
        centroids.push(points[rng.next_u64() as usize % points.len()]);
        let mut min_d2 = vec![f32::MAX; points.len()];
        let k_pp_limit = if num_points > 100_000_000 { 500 } else { 2000 };
        let k_pp = num_clusters.min(k_pp_limit);
        for _ in 1..k_pp {
            let last = centroids.last().unwrap();
            let mut sum_d2 = 0.0;
            for i in 0..points.len() {
                let d2 = dist_sq(points[i], *last);
                if d2 < min_d2[i] { min_d2[i] = d2; }
                sum_d2 += min_d2[i];
            }
            let target = rng.next_f32() * sum_d2;
            let mut cur = 0.0;
            let mut idx = points.len() - 1;
            for i in 0..points.len() {
                cur += min_d2[i];
                if cur >= target { idx = i; break; }
            }
            centroids.push(points[idx]);
        }
        for _ in k_pp..num_clusters {
            centroids.push(points[rng.next_u64() as usize % points.len()]);
        }
    }

    let iterations = if num_points > 100_000_000 { 3 } else if num_points > 10_000_000 { 7 } else { 15 };
    for _ in 0..iterations {
        let mut new_centroids = vec![[0.0, 0.0, 0.0]; num_clusters];
        let mut counts = vec![0usize; num_clusters];
        let tree = if num_clusters > 64 { Some(KdTree::build(&centroids)) } else { None };
        for p in &points {
            let idx = if let Some(ref t) = tree { t.find_nearest(*p) } else { find_nearest_brute_force(*p, &centroids) };
            new_centroids[idx][0] += p[0];
            new_centroids[idx][1] += p[1];
            new_centroids[idx][2] += p[2];
            counts[idx] += 1;
        }
        for i in 0..num_clusters {
            if counts[i] > 0 {
                centroids[i][0] = new_centroids[i][0] / counts[i] as f32;
                centroids[i][1] = new_centroids[i][1] / counts[i] as f32;
                centroids[i][2] = new_centroids[i][2] / counts[i] as f32;
            } else {
                centroids[i] = points[rng.next_u64() as usize % points.len()];
            }
        }
    }

    let final_tree = if num_clusters > 64 { Some(KdTree::build(&centroids)) } else { None };
    for p in &points {
        let idx = if let Some(ref t) = final_tree { t.find_nearest(*p) } else { find_nearest_brute_force(*p, &centroids) };
        out_writer.write_all(&(idx as u32).to_le_bytes())?;
    }
    for _ in (points.len() as u64)..num_points {
        if stdin.read_exact(&mut p_buf).is_err() { break; }
        let p = [
            f32::from_le_bytes([p_buf[0], p_buf[1], p_buf[2], p_buf[3]]),
            f32::from_le_bytes([p_buf[4], p_buf[5], p_buf[6], p_buf[7]]),
            f32::from_le_bytes([p_buf[8], p_buf[9], p_buf[10], p_buf[11]]),
        ];
        let idx = if let Some(ref t) = final_tree { t.find_nearest(p) } else { find_nearest_brute_force(p, &centroids) };
        out_writer.write_all(&(idx as u32).to_le_bytes())?;
    }
    out_writer.flush()?;
    Ok(())
}
```
