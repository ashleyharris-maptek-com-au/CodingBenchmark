#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <chrono>
#include <random>
#include <intrin.h>
#include <stdint.h>

using namespace std;

/**
 * Longest Common Substring (LCS) for N strings.
 * 
 * Strategy:
 * 1. Adaptive selection: Choose the shortest string (S0) to limit the search space.
 * 2. Binary search on the length of the LCS (L).
 * 3. Rolling Hash verification:
 *    - Use a 122-bit hash (two 2^61-1 hashes) to minimize collisions to near-zero.
 *    - Build a custom hash table for hashes of all length-L substrings in S0.
 *    - For each subsequent string, check its length-L substrings against the table.
 *    - Track which hashes appear in ALL strings using a "last_seen" field in the table.
 * 4. Complexity: O(Sum(length(Si)) * log(min_length)).
 *    - For 1GB of total string data and 24 binary search steps, this is roughly 2.4e10 operations.
 *    - Using optimized 2^61-1 modular arithmetic, this easily fits within the 300s limit.
 */

typedef unsigned __int64 uint64;

// Mersenne prime 2^61 - 1
const uint64 MOD = (1ULL << 61) - 1;

// Optimized multiplication for 2^61 - 1 using the property: 2^64 = 8 (mod 2^61-1)
inline uint64 mod_mul(uint64 a, uint64 b) {
    uint64 high;
    uint64 low = _umul128(a, b, &high);
    uint64 res = (high << 3) | (low >> 61);
    res += low & MOD;
    if (res >= MOD) res -= MOD;
    return res;
}

inline uint64 mod_add(uint64 a, uint64 b) {
    a += b;
    if (a >= MOD) a -= MOD;
    return a;
}

inline uint64 mod_sub(uint64 a, uint64 b) {
    a += MOD - b;
    if (a >= MOD) a -= MOD;
    return a;
}

struct TableEntry {
    uint64 h1, h2;
    long long first_idx;
    int last_seen;
};

// Optimized Open-Addressing Hash Table
struct HashTable {
    vector<TableEntry> table;
    uint64 mask;
    
    HashTable() : mask(0) {}
    
    void resize(size_t n) {
        uint64 sz = 1;
        // Aim for a load factor < 0.5. sz must be power of 2.
        while (sz < (uint64)n * 2) {
            sz <<= 1;
            if (sz == 0) break; // Should not happen given RAM limits
        }
        if (sz < 16) sz = 16;
        table.assign(sz, {0, 0, -1, -1});
        mask = sz - 1;
    }
    
    void clear() {
        // Fast clear: only need to reset first_idx
        for (auto& entry : table) entry.first_idx = -1;
    }
    
    void insert(uint64 h1, uint64 h2, long long idx) {
        uint64 pos = h1 & mask;
        while (table[pos].first_idx != -1) {
            if (table[pos].h1 == h1 && table[pos].h2 == h2) return;
            pos = (pos + 1) & mask;
        }
        table[pos] = {h1, h2, idx, 0};
    }
    
    void find_and_update(uint64 h1, uint64 h2, int string_idx) {
        uint64 pos = h1 & mask;
        while (table[pos].first_idx != -1) {
            if (table[pos].h1 == h1 && table[pos].h2 == h2) {
                // If this hash was seen in the previous string, mark it for the current one
                if (table[pos].last_seen == string_idx - 1) {
                    table[pos].last_seen = string_idx;
                }
                return;
            }
            pos = (pos + 1) & mask;
        }
    }

    bool has_any(int target_last_seen) {
        for (const auto& entry : table) {
            if (entry.first_idx != -1 && entry.last_seen == target_last_seen) return true;
        }
        return false;
    }

    long long get_first_valid(int target_last_seen) {
        for (const auto& entry : table) {
            if (entry.first_idx != -1 && entry.last_seen == target_last_seen) return entry.first_idx;
        }
        return -1;
    }
};

uint64 base1, base2;
vector<uint64> pow1, pow2;

void ensure_powers(size_t max_len) {
    if (pow1.size() > max_len) return;
    size_t old_size = pow1.size();
    pow1.resize(max_len + 1);
    pow2.resize(max_len + 1);
    if (old_size == 0) {
        pow1[0] = 1;
        pow2[0] = 1;
        old_size = 1;
    }
    for (size_t i = old_size; i <= max_len; ++i) {
        pow1[i] = mod_mul(pow1[i - 1], base1);
        pow2[i] = mod_mul(pow2[i - 1], base2);
    }
}

HashTable ht;

bool check(size_t k, int N, const vector<string>& strings, long long& result_idx) {
    if (k == 0) return true;
    ht.clear();
    
    const string& s0 = strings[0];
    uint64 h1 = 0, h2 = 0;
    for (size_t i = 0; i < k; ++i) {
        h1 = mod_add(mod_mul(h1, base1), (unsigned char)s0[i]);
        h2 = mod_add(mod_mul(h2, base2), (unsigned char)s0[i]);
    }
    ht.insert(h1, h2, 0);
    
    uint64 pk1 = pow1[k];
    uint64 pk2 = pow2[k];
    
    size_t s0_len = s0.length();
    for (size_t i = 1; i <= s0_len - k; ++i) {
        h1 = mod_add(mod_mul(h1, base1), (unsigned char)s0[i + k - 1]);
        h1 = mod_sub(h1, mod_mul((unsigned char)s0[i - 1], pk1));
        h2 = mod_add(mod_mul(h2, base2), (unsigned char)s0[i + k - 1]);
        h2 = mod_sub(h2, mod_mul((unsigned char)s0[i - 1], pk2));
        ht.insert(h1, h2, (long long)i);
    }
    
    for (int j = 1; j < N; ++j) {
        const string& sj = strings[j];
        size_t sj_len = sj.length();
        if (sj_len < k) return false;
        
        uint64 ch1 = 0, ch2 = 0;
        for (size_t i = 0; i < k; ++i) {
            ch1 = mod_add(mod_mul(ch1, base1), (unsigned char)sj[i]);
            ch2 = mod_add(mod_mul(ch2, base2), (unsigned char)sj[i]);
        }
        ht.find_and_update(ch1, ch2, j);
        
        for (size_t i = 1; i <= sj_len - k; ++i) {
            ch1 = mod_add(mod_mul(ch1, base1), (unsigned char)sj[i + k - 1]);
            ch1 = mod_sub(ch1, mod_mul((unsigned char)sj[i - 1], pk1));
            ch2 = mod_add(mod_mul(ch2, base2), (unsigned char)sj[i + k - 1]);
            ch2 = mod_sub(ch2, mod_mul((unsigned char)sj[i - 1], pk2));
            ht.find_and_update(ch1, ch2, j);
        }
        
        // If no common hash survives after this string, LCS of length k doesn't exist
        if (!ht.has_any(j)) return false;
    }
    
    result_idx = ht.get_first_valid(N - 1);
    return result_idx != -1;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N_input;
    if (!(cin >> N_input)) return 0;
    string dummy;
    getline(cin, dummy); // Consume trailing newline
    
    if (N_input <= 0) {
        return 0;
    }
    
    vector<string> strings;
    strings.reserve(N_input);
    int min_idx = -1;
    size_t min_len = 0;
    size_t global_max_len = 0;
    
    for (int i = 0; i < N_input; ++i) {
        string s;
        if (!getline(cin, s)) break;
        // Clean up Windows carriage returns if present
        if (!s.empty() && s.back() == '\r') s.pop_back();
        
        if (min_idx == -1 || s.length() < min_len) {
            min_len = s.length();
            min_idx = (int)strings.size();
        }
        if (s.length() > global_max_len) global_max_len = s.length();
        strings.push_back(std::move(s));
    }
    
    int N = (int)strings.size();
    if (N == 0) return 0;
    if (min_len == 0) {
        cout << endl;
        return 0;
    }
    
    // Adaptive logic:
    // 1. Pick the shortest string as the reference (reduces hash table size)
    // 2. Sort other strings by length (faster pruning during rolling hash intersection)
    swap(strings[0], strings[min_idx]);
    if (N > 2) {
        sort(strings.begin() + 1, strings.end(), [](const string& a, const string& b) {
            return a.length() < b.length();
        });
    }
    
    if (N == 1) {
        cout << strings[0] << endl;
        return 0;
    }
    
    // Setup randomized bases to prevent anti-hash attacks
    mt19937_64 rng((uint64)chrono::steady_clock::now().time_since_epoch().count());
    uniform_int_distribution<uint64> dist(300, MOD - 1);
    base1 = dist(rng);
    base2 = dist(rng);
    
    ensure_powers(global_max_len);
    ht.resize(min_len);
    
    size_t low = 1, high = min_len;
    size_t ans_len = 0;
    long long ans_idx = -1;
    
    // Binary search for the longest possible length
    while (low <= high) {
        size_t mid = low + (high - low) / 2;
        long long idx = -1;
        if (check(mid, N, strings, idx)) {
            ans_len = mid;
            ans_idx = idx;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    if (ans_idx != -1) {
        cout << strings[0].substr((size_t)ans_idx, ans_len) << endl;
    } else {
        cout << endl;
    }
    
    return 0;
}
