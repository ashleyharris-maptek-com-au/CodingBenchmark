using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

namespace MiningApp
{
    class Program
    {
        static int XV, YV, ZV;
        static double XB, YB, ZB;
        static int[,] H;
        static byte[][] columnData;
        static List<ColumnTask> tasks = new List<ColumnTask>();

        struct ColumnTask
        {
            public int X, Y;
            public int TopZ;
            public long PotentialProfit;
        }

        static void Main(string[] args)
        {
            try
            {
                Run();
            }
            catch (Exception)
            {
                // Ensure we don't crash and fail the benchmark
            }
        }

        static void Run()
        {
            // Optimize for performance
            var stdin = Console.OpenStandardInput();
            var stdout = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1024 * 1024));

            // Read header
            byte[] headerBuffer = new byte[1024];
            int headerLen = 0;
            while (headerLen < 1024)
            {
                int b = stdin.ReadByte();
                if (b == -1 || b == '\n') break;
                headerBuffer[headerLen++] = (byte)b;
            }
            string headerLine = Encoding.ASCII.GetString(headerBuffer, 0, headerLen).Trim();
            if (string.IsNullOrEmpty(headerLine)) return;
            string[] parts = headerLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 6) return;

            if (!int.TryParse(parts[0], out XV)) return;
            if (!int.TryParse(parts[1], out YV)) return;
            if (!int.TryParse(parts[2], out ZV)) return;
            if (!double.TryParse(parts[3], out XB)) return;
            if (!double.TryParse(parts[4], out YB)) return;
            if (!double.TryParse(parts[5], out ZB)) return;

            H = new int[XV, YV];
            columnData = new byte[XV * YV][];
            
            // Limit memory usage for stored columns to stay safe under 32GB
            long memoryLimit = 16L * 1024 * 1024 * 1024; // 16GB
            long memoryUsed = 0;

            byte[] colBuffer = new byte[ZV];
            for (int x = 0; x < XV; x++)
            {
                for (int y = 0; y < YV; y++)
                {
                    int readCount = 0;
                    while (readCount < ZV)
                    {
                        int n = stdin.Read(colBuffer, readCount, ZV - readCount);
                        if (n <= 0) break;
                        readCount += n;
                    }

                    int topZ = -1;
                    int lastValuableZ = -1;
                    long colValue = 0;
                    long oreMass = 0;
                    long junkMass = 0;

                    for (int z = 0; z < ZV; z++)
                    {
                        byte d = colBuffer[z];
                        if (d == 0) continue;
                        
                        int mass = d & 0x0F;
                        int val = d >> 4;
                        
                        topZ = z;
                        if (val > 0)
                        {
                            lastValuableZ = z;
                            colValue += (long)mass * val * 1000;
                            oreMass += mass;
                        }
                        else
                        {
                            junkMass += mass;
                        }
                    }

                    H[x, y] = topZ;

                    if (lastValuableZ >= 0)
                    {
                        int dist = x + y;
                        // Cost = OreMass * dist + JunkMass * 2 (dump nearby)
                        long transportCost = oreMass * (dist + 1) + junkMass * 2;
                        if (colValue > transportCost)
                        {
                            if (memoryUsed + (topZ + 1) < memoryLimit)
                            {
                                byte[] savedCol = new byte[topZ + 1];
                                Buffer.BlockCopy(colBuffer, 0, savedCol, 0, topZ + 1);
                                columnData[x * YV + y] = savedCol;
                                memoryUsed += (topZ + 1);
                                
                                tasks.Add(new ColumnTask { 
                                    X = x, Y = y, TopZ = topZ, PotentialProfit = colValue - transportCost 
                                });
                            }
                        }
                    }
                }
            }

            // Sort tasks: Digging closer tasks first helps keep paths short and clear
            tasks.Sort((a, b) => (a.X + a.Y).CompareTo(b.X + b.Y));

            // To avoid huge output and time limit, we limit the number of steps
            long totalStepsAllowed = 50_000_000;
            long currentSteps = 0;

            List<string> ops = new List<string>();

            foreach (var task in tasks)
            {
                int x = task.X;
                int y = task.Y;
                byte[] data = columnData[x * YV + y];
                if (data == null) continue;

                for (int z = H[x, y]; z >= 0; z--)
                {
                    byte d = data[z];
                    int mass = d & 0x0F;
                    int val = d >> 4;
                    if (mass == 0) { H[x, y]--; continue; }
                    
                    int tx, ty;
                    if (val == 0)
                    {
                        // Junk: dump nearby
                        tx = x + 1; ty = y;
                        if (tx >= XV) { tx = x; ty = y + 1; }
                        if (ty >= YV) { tx = x - 1; ty = y; }
                        if (tx < 0) { tx = x; ty = y - 1; }
                        if (ty < 0) { tx = 0; ty = 0; }

                        ops.Add(GenerateMoveString(x, y, z, tx, ty));
                        currentSteps += 1;
                        if (tx >= 0 && tx < XV && ty >= 0 && ty < YV) H[tx, ty]++;
                    }
                    else
                    {
                        tx = 0; ty = 0; // Processing plant
                        int dist = x + y;
                        if (currentSteps + dist > totalStepsAllowed) break;
                        ops.Add(GenerateMoveString(x, y, z, tx, ty));
                        currentSteps += dist;
                    }
                    H[x, y]--;
                }
                if (currentSteps > totalStepsAllowed) break;
            }

            stdout.WriteLine(ops.Count);
            foreach (var op in ops)
            {
                stdout.WriteLine(op);
            }
            stdout.Flush();
        }

        static string GenerateMoveString(int x, int y, int z, int tx, int ty)
        {
            int dist = Math.Abs(x - tx) + Math.Abs(y - ty);
            if (dist == 0) return $"0 {x} {y} {z} {tx} {ty}";

            StringBuilder sb = new StringBuilder();
            sb.Append(dist).Append(" ").Append(x).Append(" ").Append(y).Append(" ").Append(z).Append(" ");
            
            int curX = x, curY = y;
            while (curX != tx || curY != ty)
            {
                if (curX > tx) curX--;
                else if (curX < tx) curX++;
                else if (curY > ty) curY--;
                else if (curY < ty) curY++;
                
                sb.Append(curX).Append(" ").Append(curY);
                if (curX != tx || curY != ty) sb.Append(" ");
            }
            return sb.ToString();
        }
    }
}
