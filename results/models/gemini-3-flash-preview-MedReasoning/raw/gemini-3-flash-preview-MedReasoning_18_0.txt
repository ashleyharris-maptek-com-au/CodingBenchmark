The Minimum Edge Cut problem is solved by finding the minimum number of edges whose removal disconnects an undirected connected graph. For small graphs, this is straightforward, but for massive, multi-GB graphs, a memory-efficient and scalable approach is required.

The implemented C++ solver uses a **three-pass streaming algorithm** designed to operate within a 32GB RAM limit while handling graphs of any size:

1.  **Pass 1: Analysis & Streaming**
    The program reads edges from `stdin` and immediately streams them to a temporary binary file on disk. Simultaneously, it maintains a count of node degrees to identify the minimum degree ($\delta$), which serves as an initial upper bound for the global minimum cut.
2.  **Pass 2: Sparse Certificate Construction**
    Using the edges stored on disk, the solver builds a **k-sparse certificate**. This certificate is a subgraph (the union of $k$ spanning forests) that preserves all cuts of size less than $k$. The value of $k$ is adaptively chosen based on the number of nodes ($N$) and available RAM, typically targeting between 40 and 150. This reduces the problem complexity from the original edge count ($M$) to $O(kN)$.
3.  **Pass 3: Global Min Cut via Max-Flow**
    The solver runs **Dinic's max-flow algorithm** on the sparse certificate. By calculating the max-flow from a fixed node (of minimum degree) to a set of random target nodes, the global minimum cut is identified with high probability. This is significantly faster than deterministic algorithms for massive graphs.
4.  **Final Pass: Output**
    Once the minimum cut is determined, the solver performs a final pass over the disk-stored edges to identify and output every original edge that crosses the partition.

This architecture ensures the solver is robust against memory exhaustion and executes efficiently by focusing heavy computation only on the sparse certificate.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <random>
#include <chrono>
#include <cstdio>
#include <cstring>
#include <numeric>

using namespace std;

typedef uint32_t Node;
typedef long long Flow;

// Fast I/O for efficient processing of large inputs
class FastReader {
    static const int BUF_SIZE = 1024 * 1024;
    char* buf;
    int pos, len;
    bool eof_flag;

public:
    FastReader() : pos(0), len(0), eof_flag(false) {
        buf = new char[BUF_SIZE];
    }
    ~FastReader() { delete[] buf; }

    char readChar() {
        if (pos == len) {
            pos = 0;
            len = (int)fread(buf, 1, BUF_SIZE, stdin);
            if (len == 0) {
                eof_flag = true;
                return EOF;
            }
        }
        return buf[pos++];
    }

    bool readUint(uint32_t& res) {
        char c = readChar();
        while (c != EOF && (c < '0' || c > '9')) c = readChar();
        if (c == EOF) return false;
        res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = readChar();
        }
        return true;
    }

    bool readUlong(unsigned long long& res) {
        char c = readChar();
        while (c != EOF && (c < '0' || c > '9')) c = readChar();
        if (c == EOF) return false;
        res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = readChar();
        }
        return true;
    }
};

// Dinic's algorithm for max flow, optimized for undirected graphs
struct Dinic {
    struct Edge {
        Node to;
        Flow cap;
        Flow flow;
        uint32_t next;
    };
    vector<Edge> edges;
    vector<uint32_t> head;
    vector<int> level;
    vector<uint32_t> ptr;

    Dinic(Node n) : head(n, uint32_t(-1)), level(n), ptr(n) {}

    // Adds an undirected edge by adding two directed edges
    void addEdge(Node from, Node to, Flow cap) {
        edges.push_back({to, cap, 0, head[from]});
        head[from] = (uint32_t)edges.size() - 1;
        edges.push_back({from, cap, 0, head[to]});
        head[to] = (uint32_t)edges.size() - 1;
    }

    bool bfs(Node s, Node t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<Node> q;
        q.push(s);
        while (!q.empty()) {
            Node v = q.front();
            q.pop();
            for (uint32_t i = head[v]; i != uint32_t(-1); i = edges[i].next) {
                if (edges[i].cap - edges[i].flow > 0 && level[edges[i].to] == -1) {
                    level[edges[i].to] = level[v] + 1;
                    q.push(edges[i].to);
                }
            }
        }
        return level[t] != -1;
    }

    Flow dfs(Node v, Node t, Flow pushed) {
        if (pushed == 0) return 0;
        if (v == t) return pushed;
        for (uint32_t& cid = ptr[v]; cid != uint32_t(-1); cid = edges[cid].next) {
            auto& edge = edges[cid];
            Node tr = edge.to;
            if (level[v] + 1 != level[tr] || edge.cap - edge.flow == 0) continue;
            Flow tr_pushed = dfs(tr, t, min(pushed, edge.cap - edge.flow));
            if (tr_pushed == 0) continue;
            edge.flow += tr_pushed;
            edges[cid ^ 1].flow -= tr_pushed;
            return tr_pushed;
        }
        return 0;
    }

    Flow maxFlow(Node s, Node t) {
        Flow flow = 0;
        while (bfs(s, t)) {
            ptr = head;
            while (Flow pushed = dfs(s, t, (Flow)1e18)) {
                flow += pushed;
            }
        }
        return flow;
    }
};

// Disjoint Set Union for sparse certificate construction
struct DSU {
    uint32_t* parent;
    DSU() : parent(nullptr) {}
    void init(uint32_t n, uint32_t* storage) {
        parent = storage;
        for (uint32_t i = 0; i < n; ++i) parent[i] = i;
    }
    uint32_t find(uint32_t i) {
        if (parent[i] == i) return i;
        return parent[i] = find(parent[i]);
    }
    bool unite(uint32_t i, uint32_t j) {
        uint32_t root_i = find(i);
        uint32_t root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
            return true;
        }
        return false;
    }
};

int main() {
    FastReader fr;
    uint32_t N;
    unsigned long long M;
    if (!fr.readUint(N) || !fr.readUlong(M)) return 0;

    // Use a temporary file to store edges if they don't fit in memory
    FILE* tmp = tmpfile();
    bool tmp_is_local = false;
    if (!tmp) {
        tmp = fopen("temp_edges.bin", "wb+");
        tmp_is_local = true;
    }

    // Pass 1: Stream edges to disk and count degrees
    vector<unsigned long long> degrees(N, 0);
    for (unsigned long long i = 0; i < M; ++i) {
        uint32_t u, v;
        if (!fr.readUint(u) || !fr.readUint(v)) break;
        if (u >= N || v >= N || u == v) {
            uint32_t sentinel = 0xFFFFFFFF;
            fwrite(&sentinel, 4, 1, tmp);
            fwrite(&sentinel, 4, 1, tmp);
            continue;
        }
        fwrite(&u, 4, 1, tmp);
        fwrite(&v, 4, 1, tmp);
        degrees[u]++;
        degrees[v]++;
    }

    // Find a node with minimum degree as a starting point for global min cut
    unsigned long long min_deg = (unsigned long long)-1;
    uint32_t s = 0;
    for (uint32_t i = 0; i < N; ++i) {
        if (degrees[i] < min_deg) {
            min_deg = degrees[i];
            s = i;
        }
    }

    if (min_deg == (unsigned long long)-1) {
        if (tmp_is_local) { fclose(tmp); remove("temp_edges.bin"); }
        else { fclose(tmp); }
        return 0;
    }

    // Adaptively choose k for the sparse certificate to fit in 32GB RAM
    int max_k = 100;
    if (N > 0) {
        long long available = 28000000000LL; // target 28GB to be safe
        max_k = (int)((available / N - 40) / 52);
    }
    if (max_k < 1) max_k = 1;
    if (max_k > 150) max_k = 150;

    int k = min((int)min_deg, max_k);

    // Pass 2: Build k-sparse certificate from disk
    uint32_t* dsu_storage = new uint32_t[(size_t)k * N];
    vector<DSU> dsus(k);
    for (int i = 0; i < k; ++i) dsus[i].init(N, dsu_storage + (size_t)i * N);

    struct CertEdge { uint32_t u, v; };
    vector<CertEdge> cert_edges;

    fseek(tmp, 0, SEEK_SET);
    for (unsigned long long i = 0; i < M; ++i) {
        uint32_t u, v;
        if (fread(&u, 4, 1, tmp) != 1) break;
        if (fread(&v, 4, 1, tmp) != 1) break;
        if (u == 0xFFFFFFFF) continue;
        for (int j = 0; j < k; ++j) {
            if (dsus[j].unite(u, v)) {
                cert_edges.push_back({u, v});
                break;
            }
        }
    }

    delete[] dsu_storage;
    dsus.clear();

    // Prepare weighted graph for Dinic from the sparse certificate
    for (auto& e : cert_edges) if (e.u > e.v) swap(e.u, e.v);
    sort(cert_edges.begin(), cert_edges.end(), [](const CertEdge& a, const CertEdge& b) {
        if (a.u != b.u) return a.u < b.u;
        return a.v < b.v;
    });

    Dinic dinic(N);
    for (size_t i = 0; i < cert_edges.size(); ) {
        size_t j = i;
        while (j < cert_edges.size() && cert_edges[j].u == cert_edges[i].u && cert_edges[j].v == cert_edges[i].v) {
            j++;
        }
        dinic.addEdge(cert_edges[i].u, cert_edges[i].v, (int)(j - i));
        i = j;
    }
    cert_edges.clear();
    cert_edges.shrink_to_fit();

    // Initial min cut is the edges incident to the min-degree node
    Flow min_cut = (Flow)min_deg;
    vector<bool> in_S(N, false);
    in_S[s] = true;

    // Probabilistic global min cut: try max flow to several random nodes
    mt19937 rng(1337);
    vector<uint32_t> targets;
    if (N > 1) {
        vector<uint32_t> candidates;
        int num_cand = min(N, 1000U);
        for(int i=0; i<num_cand; ++i) candidates.push_back((uint32_t)(i * ( (unsigned long long)N / num_cand )));
        shuffle(candidates.begin(), candidates.end(), rng);
        for(uint32_t t : candidates) {
            if (t != s) {
                targets.push_back(t);
                if (targets.size() >= 20) break;
            }
        }
    }

    for (Node t : targets) {
        for (auto& e : dinic.edges) e.flow = 0;
        Flow flow = dinic.maxFlow(s, t);
        if (flow < min_cut) {
            min_cut = flow;
            fill(in_S.begin(), in_S.end(), false);
            queue<Node> q_s;
            q_s.push(s);
            in_S[s] = true;
            while (!q_s.empty()) {
                Node v = q_s.front();
                q_s.pop();
                for (uint32_t i = dinic.head[v]; i != uint32_t(-1); i = dinic.edges[i].next) {
                    if (dinic.edges[i].cap - dinic.edges[i].flow > 0 && !in_S[dinic.edges[i].to]) {
                        in_S[dinic.edges[i].to] = true;
                        q_s.push(dinic.edges[i].to);
                    }
                }
            }
        }
    }

    // Output results
    printf("%lld\n", min_cut);
    fseek(tmp, 0, SEEK_SET);
    for (unsigned long long i = 0; i < M; ++i) {
        uint32_t u, v;
        if (fread(&u, 4, 1, tmp) != 1) break;
        if (fread(&v, 4, 1, tmp) != 1) break;
        if (u == 0xFFFFFFFF) continue;
        if (in_S[u] != in_S[v]) {
            printf("%u %u\n", u, v);
        }
    }

    if (tmp_is_local) { fclose(tmp); remove("temp_edges.bin"); }
    else { fclose(tmp); }
    return 0;
}
```
