OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 15

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
      %void = OpTypeVoid
         %3 = OpTypeFunction %void
     %float = OpTypeFloat 32
   %v2float = OpTypeVector %float 2
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
      %uint = OpTypeInt 32 0
       %int = OpTypeInt 32 1
    %uint_6 = OpConstant %uint 6

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_in_pv = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_in_pv
     %gl_in = OpVariable %ptr_in_arr_pv Input

  %arr_in_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_in_v3
  %worldPos = OpVariable %ptr_in_arr_v3 Input
    %normal = OpVariable %ptr_in_arr_v3 Input
     %color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Constants
     %int_0 = OpConstant %int 0
     %int_2 = OpConstant %int 2
     %int_4 = OpConstant %int 4
   %float_0_3 = OpConstant %float 0.3
   %float_0_0075 = OpConstant %float 0.0075
   %float_1 = OpConstant %float 1
   %v3float_1_1_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

      %main = OpFunction %void None %3
%label_main = OpLabel

; Load positions 0, 2, 4
    %ptr_0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %pos0 = OpLoad %v4float %ptr_0_pos
    %ptr_2_pos = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %pos2 = OpLoad %v4float %ptr_2_pos
    %ptr_4_pos = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
       %pos4 = OpLoad %v4float %ptr_4_pos

; --- Emit Triangle (0, 2, 4) ---
    %ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0
    %ptr_0_col = OpAccessChain %ptr_in_v3 %color %int_0
       %col0 = OpLoad %v3float %ptr_0_col
      %col0_d = OpVectorTimesScalar %v3float %col0 %float_0_3
               OpStore %ptr_out_pos %pos0
               OpStore %outColor %col0_d
               OpEmitVertex

; Vertex 2
    %ptr_2_col = OpAccessChain %ptr_in_v3 %color %int_2
       %col2 = OpLoad %v3float %ptr_2_col
      %col2_d = OpVectorTimesScalar %v3float %col2 %float_0_3
               OpStore %ptr_out_pos %pos2
               OpStore %outColor %col2_d
               OpEmitVertex

; Vertex 4
    %ptr_4_col = OpAccessChain %ptr_in_v3 %color %int_4
       %col4 = OpLoad %v3float %ptr_4_col
      %col4_d = OpVectorTimesScalar %v3float %col4 %float_0_3
               OpStore %ptr_out_pos %pos4
               OpStore %outColor %col4_d
               OpEmitVertex
               OpEndPrimitive

; --- Common calculations ---
    %w0 = OpCompositeExtract %float %pos0 3
    %xy0 = OpVectorShuffle %v2float %pos0 %pos0 0 1
    %ndc0 = OpFDiv %v2float %xy0 %w0
    %z0 = OpCompositeExtract %float %pos0 2

    %w2 = OpCompositeExtract %float %pos2 3
    %xy2 = OpVectorShuffle %v2float %pos2 %pos2 0 1

; --- Edge 0-2 ---
    %d02 = OpFSub %v2float %ndc2 %ndc0
    %l02 = OpExtInst %float %1 Length %d02
    %dir02 = OpFDiv %v2float %d02 %l02

    %dir02_y = OpCompositeExtract %float %dir02 1
    %neg_dir02_y = OpFNegate %float %dir02_y

    %v02_1_x = OpCompositeExtract %float %v02_1_xy 0
    %v02_1_y = OpCompositeExtract %float %v02_1_xy 1
    %v02_1 = OpCompositeConstruct %v4float %v02_1_x %v02_1_y %z0 %w0
               OpStore %ptr_out_pos %v02_1
               OpStore %outColor %v3float_1_1_1
               OpEmitVertex
