The 1D Cutting Stock Problem is solved using the **First Fit Decreasing (FFD)** heuristic, which is highly efficient and provides solutions very close to optimal. For $N$ cuts, the algorithm achieves $O(N \log N)$ complexity by using a **Segment Tree** to find the first stock with sufficient remaining capacity in logarithmic time. This ensures that even millions of cuts can be processed well within the 5-minute time limit.

The implementation includes:
1.  **Fast I/O:** A custom `FastReader` and a `BufferedStream` for `StreamWriter` to handle massive inputs and outputs efficiently.
2.  **$O(N \log N)$ Algorithm:** Sorting the cuts in descending order and using a Segment Tree for placement.
3.  **Memory Efficiency:** Utilizing primitive arrays and structs to keep memory overhead low, well within the 32 GB limit.

### C# Solution (`Solution.cs`)

```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class Solution {
    class FastReader {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int pos, limit;

        public FastReader(Stream s) {
            stream = s;
        }

        private byte Read() {
            if (pos >= limit) {
                pos = 0;
                limit = stream.Read(buffer, 0, buffer.Length);
                if (limit <= 0) return 0;
            }
            return buffer[pos++];
        }

        public long NextLong() {
            long val = 0;
            byte b = Read();
            while (b > 0 && b <= 32) b = Read();
            if (b == 0) return -1;
            while (b >= '0' && b <= '9') {
                val = val * 10 + (b - '0');
                b = Read();
            }
            return val;
        }

        public int NextInt() {
            return (int)NextLong();
        }
    }

    public static void Main() {
        var input = Console.OpenStandardInput();
        var reader = new FastReader(input);
        
        int n = reader.NextInt();
        if (n < 0) return;
        if (n == 0) {
            Console.WriteLine(0);
            return;
        }

        long stockLength = reader.NextLong();

        long[] cutLengths = new long[n];
        int[] originalIndices = new int[n];
        for (int i = 0; i < n; i++) {
            cutLengths[i] = reader.NextLong();
            originalIndices[i] = i;
        }

        // First Fit Decreasing heuristic: Sort cuts descending
        Array.Sort(cutLengths, originalIndices);
        Array.Reverse(cutLengths);
        Array.Reverse(originalIndices);

        // Segment Tree to find the first stock with enough space in O(log N)
        // tree[i] stores the max remaining space in the range of stocks it covers.
        int treeSize = 1;
        while (treeSize < n) treeSize *= 2;
        long[] tree = new long[2 * treeSize];
        
        // Initialize all stocks with full capacity
        for (int i = 0; i < treeSize; i++) {
            tree[treeSize + i] = stockLength;
        }
        for (int i = treeSize - 1; i >= 1; i--) {
            tree[i] = stockLength;
        }

        int[] cutToStock = new int[n];
        int maxStockUsed = 0;

        for (int i = 0; i < n; i++) {
            long len = cutLengths[i];
            
            // Find first stock index such that tree[stockIdx] >= len
            int stockIdx = FindFirst(tree, treeSize, len);
            cutToStock[i] = stockIdx;
            if (stockIdx > maxStockUsed) maxStockUsed = stockIdx;
            
            // Update tree with new remaining space
            long currentRemaining = tree[treeSize + stockIdx] - len;
            Update(tree, treeSize, stockIdx, currentRemaining);
        }

        int numStocks = maxStockUsed + 1;
        
        // Group cuts by stock efficiently
        int[] stockCount = new int[numStocks];
        for (int i = 0; i < n; i++) stockCount[cutToStock[i]]++;

        int[] stockStart = new int[numStocks + 1];
        for (int i = 0; i < numStocks; i++) {
            stockStart[i + 1] = stockStart[i] + stockCount[i];
        }

        int[] sortedIndices = new int[n];
        int[] currentStockPos = new int[numStocks];
        for (int i = 0; i < numStocks; i++) {
            currentStockPos[i] = stockStart[i];
        }

        for (int i = 0; i < n; i++) {
            int sIdx = cutToStock[i];
            sortedIndices[currentStockPos[sIdx]++] = originalIndices[i];
        }

        // Fast output
        using (var writer = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1024 * 1024))) {
            writer.WriteLine(numStocks);
            for (int i = 0; i < numStocks; i++) {
                int start = stockStart[i];
                int end = stockStart[i + 1];
                for (int j = start; j < end; j++) {
                    writer.Write(sortedIndices[j]);
                    if (j < end - 1) writer.Write(' ');
                }
                writer.WriteLine();
            }
        }
    }

    private static int FindFirst(long[] tree, int treeSize, long minVal) {
        int idx = 1;
        while (idx < treeSize) {
            int left = idx << 1;
            if (tree[left] >= minVal) {
                idx = left;
            } else {
                idx = left | 1;
            }
        }
        return idx - treeSize;
    }

    private static void Update(long[] tree, int treeSize, int idx, long val) {
        idx += treeSize;
        tree[idx] = val;
        while (idx > 1) {
            idx >>= 1;
            tree[idx] = Math.Max(tree[idx << 1], tree[(idx << 1) | 1]);
        }
    }
}
```
