; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %outColor
               OpExecutionMode %main OriginUpperLeft
               OpDecorate %worldPos Location 0
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
  %float_0_5 = OpConstant %float 0.5
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
  %float_0_1 = OpConstant %float 0.1
  %float_0_4 = OpConstant %float 0.4
%vec3_bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
   %vec3_top = OpConstantComposite %v3float %float_1 %float_1 %float_1
   %worldPos = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output
       %main = OpFunction %void None %3
          %5 = OpLabel
    %pos_val = OpLoad %v3float %worldPos
      %pos_y = OpCompositeExtract %float %pos_val 1
     %tmp_mul = OpFMul %float %pos_y %float_0_5
     %tmp_add = OpFAdd %float %tmp_mul %float_0_5
          %t = OpExtInst %float %1 FClamp %tmp_add %float_0 %float_1
      %color = OpExtInst %v3float %1 FMix %vec3_bottom %vec3_top %t
  %out_color = OpCompositeConstruct %v4float %color %float_1
               OpStore %outColor %out_color
               OpReturn
               OpFunctionEnd
