"use strict";

const fs = require('fs');

function solve() {
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }
    
    const { machineCount, tasks } = data;
    if (!tasks || tasks.length === 0) {
        process.stdout.write(JSON.stringify({ order: [] }));
        return;
    }

    const taskMap = new Map();
    const adj = new Map();
    const revAdj = new Map();
    const inDegree = new Map();
    const outDegree = new Map();

    for (const task of tasks) {
        taskMap.set(task.id, task);
        adj.set(task.id, []);
        revAdj.set(task.id, []);
        inDegree.set(task.id, 0);
        outDegree.set(task.id, 0);
    }

    for (const task of tasks) {
        if (task.deps && Array.isArray(task.deps)) {
            for (const depId of task.deps) {
                if (taskMap.has(depId)) {
                    adj.get(depId).push(task.id);
                    revAdj.get(task.id).push(depId);
                    inDegree.set(task.id, (inDegree.get(task.id) || 0) + 1);
                    outDegree.set(depId, (outDegree.get(depId) || 0) + 1);
                }
            }
        }
    }

    const modDeadlines = new Map();
    for (const task of tasks) {
        modDeadlines.set(task.id, task.deadline);
    }

    const outDegCopy = new Map(outDegree);
    const leafQueue = [];
    for (const task of tasks) {
        if (outDegCopy.get(task.id) === 0) {
            leafQueue.push(task.id);
        }
    }

    let head = 0;
    while (head < leafQueue.length) {
        const taskId = leafQueue[head++];
        const currentModD = modDeadlines.get(taskId);
        const task = taskMap.get(taskId);
        
        const predecessors = revAdj.get(taskId);
        for (const predId of predecessors) {
            const predModD = modDeadlines.get(predId);
            const newModD = currentModD - task.duration;
            if (newModD < predModD) {
                modDeadlines.set(predId, newModD);
            }
            
            const newOutDeg = outDegCopy.get(predId) - 1;
            outDegCopy.set(predId, newOutDeg);
            if (newOutDeg === 0) {
                leafQueue.push(predId);
            }
        }
    }

    class PriorityQueue {
        constructor(comparator) {
            this.heap = [];
            this.comparator = comparator;
        }
        push(val) {
            this.heap.push(val);
            let idx = this.heap.length - 1;
            while (idx > 0) {
                let parent = (idx - 1) >> 1;
                if (this.comparator(this.heap[idx], this.heap[parent]) < 0) {
                    let tmp = this.heap[idx];
                    this.heap[idx] = this.heap[parent];
                    this.heap[parent] = tmp;
                    idx = parent;
                } else break;
            }
        }
        pop() {
            if (this.heap.length === 0) return null;
            const top = this.heap[0];
            const bottom = this.heap.pop();
            if (this.heap.length > 0) {
                this.heap[0] = bottom;
                let idx = 0;
                while (true) {
                    let left = (idx << 1) + 1;
                    let right = (idx << 1) + 2;
                    let smallest = idx;
                    if (left < this.heap.length && this.comparator(this.heap[left], this.heap[smallest]) < 0) {
                        smallest = left;
                    }
                    if (right < this.heap.length && this.comparator(this.heap[right], this.heap[smallest]) < 0) {
                        smallest = right;
                    }
                    if (smallest !== idx) {
                        let tmp = this.heap[idx];
                        this.heap[idx] = this.heap[smallest];
                        this.heap[smallest] = tmp;
                        idx = smallest;
                    } else break;
                }
            }
            return top;
        }
        size() {
            return this.heap.length;
        }
    }

    const pq = new PriorityQueue((a, b) => {
        const da = modDeadlines.get(a);
        const db = modDeadlines.get(b);
        if (da !== db) return da - db;
        const wa = taskMap.get(a).weight;
        const wb = taskMap.get(b).weight;
        if (wa !== wb) return wb - wa;
        const dura = taskMap.get(a).duration;
        const durb = taskMap.get(b).duration;
        return dura - durb;
    });

    for (const task of tasks) {
        if (inDegree.get(task.id) === 0) {
            pq.push(task.id);
        }
    }

    const order = [];
    while (pq.size() > 0) {
        const taskId = pq.pop();
        order.push(taskId);

        const successors = adj.get(taskId);
        for (const succId of successors) {
            const newInDeg = inDegree.get(succId) - 1;
            inDegree.set(succId, newInDeg);
            if (newInDeg === 0) {
                pq.push(succId);
            }
        }
    }

    process.stdout.write(JSON.stringify({ order }));
}

solve();
