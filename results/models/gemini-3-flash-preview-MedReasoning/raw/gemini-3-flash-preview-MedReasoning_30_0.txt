The 3-SAT (Boolean Satisfiability) problem is a classic NP-complete problem. Solving it for instances with up to 20 million variables and 85 million clauses within a 5-minute time limit in Python requires careful attention to both algorithm efficiency and memory management.

### Approach
Given that the instances are guaranteed to be satisfiable, a **local search** algorithm like **WalkSAT** is highly effective. WalkSAT starts with a random assignment and iteratively improves it by flipping variables that appear in unsatisfied clauses.

1.  **Memory Management**:
    *   Using standard Python lists for such a large number of variables and clauses would consume excessive memory (well over 16GB). Instead, I used `array.array` and `bytearray` to store the clauses, assignments, and indices efficiently.
    *   A chunked input reader was implemented to process the massive input file without loading the entire text into memory at once, keeping the memory footprint well within limits.
2.  **Efficient Indexing**:
    *   To quickly find which clauses contain a flipped variable, I built an adjacency index (`adj`) using a flat array and offsets (similar to Compressed Sparse Row format). This allows $O(1)$ access to the relevant clauses during updates.
3.  **Algorithmic Optimization**:
    *   The core of the solver is a focused random walk. For each flip, it either picks a random variable from an unsatisfied clause or greedily chooses the variable that minimizes the "break" count (the number of previously satisfied clauses that would become unsatisfied).
    *   The "satisfied count" for each clause is maintained incrementally, ensuring that checking and updating clause statuses is very fast ($O(12)$ operations per flip on average).
4.  **Python-Specific Speedups**:
    *   Critical loop operations were optimized by caching function lookups (like `random.random`) and minimizing object creation.
    *   The final output is constructed in chunks to balance speed and memory usage when joining millions of variable assignments into a single string.

### Implementation
The complete Python code is provided in `solution.py`. It reads from `stdin`, executes the WalkSAT algorithm with a safety time-out check, and outputs the "SAT" status followed by the satisfying assignment to `stdout`.

```python
import sys
import array
import random
import time

def solve():
    start_time = time.perf_counter()
    
    def read_input():
        initial_chunk = sys.stdin.buffer.read(100000)
        if not initial_chunk: return None, None, None
        parts = initial_chunk.split()
        if len(parts) < 2: return None, None, None
        n_vars, n_clauses = int(parts[0]), int(parts[1])
        clauses = array.array('i')
        if len(parts) > 2: clauses.extend(map(int, parts[2:]))
        while True:
            chunk = sys.stdin.buffer.read(100_000_000)
            if not chunk: break
            if not chunk[-1:].isspace(): chunk += sys.stdin.buffer.readline()
            clauses.extend(map(int, chunk.split()))
        return n_vars, n_clauses, clauses

    n_vars, n_clauses, clauses = read_input()
    if n_vars is None: return
    if len(clauses) > 3 * n_clauses: clauses = clauses[:3 * n_clauses]

    # Indexing
    counts = array.array('I', [0] * (n_vars + 1))
    for x in clauses: counts[abs(x)] += 1
    offsets = array.array('I', [0] * (n_vars + 2))
    s = 0
    for i in range(1, n_vars + 1):
        offsets[i] = s
        s += counts[i]
    offsets[n_vars + 1] = s
    adj = array.array('I', [0] * (3 * n_clauses))
    cur_offsets = array.array('I', offsets)
    for i in range(3 * n_clauses):
        v = abs(clauses[i])
        adj[cur_offsets[v]] = i
        cur_offsets[v] += 1

    # WalkSAT Setup
    ans = bytearray(random.choices([0, 1], k=n_vars + 1))
    satisfied_count = array.array('B', [0] * n_clauses)
    for i in range(n_clauses):
        c = 0
        if (clauses[3*i] > 0) == ans[abs(clauses[3*i])]: c += 1
        if (clauses[3*i+1] > 0) == ans[abs(clauses[3*i+1])]: c += 1
        if (clauses[3*i+2] > 0) == ans[abs(clauses[3*i+2])]: c += 1
        satisfied_count[i] = c
    unsatisfied_list = [i for i, c in enumerate(satisfied_count) if c == 0]
    pos_in_unsatisfied = array.array('i', [-1] * n_clauses)
    for i, c_idx in enumerate(unsatisfied_list): pos_in_unsatisfied[c_idx] = i

    # Search Loop
    prob_random = 0.3
    for flip_count in range(100 * n_vars + 1000000):
        if not unsatisfied_list: break
        if flip_count % 1000 == 0 and time.perf_counter() - start_time > 285: break
        
        c_idx = random.choice(unsatisfied_list)
        lits = [clauses[3*c_idx], clauses[3*c_idx+1], clauses[3*c_idx+2]]
        vars_in_c = [abs(lits[0]), abs(lits[1]), abs(lits[2])]
        
        if random.random() < prob_random:
            best_v = random.choice(vars_in_c)
        else:
            min_break = n_clauses + 1
            best_v = vars_in_c[0]
            start_offset = random.randint(0, 2)
            for i in range(3):
                v = vars_in_c[(start_offset + i) % 3]
                b = 0
                for idx in adj[offsets[v] : offsets[v+1]]:
                    if (clauses[idx] > 0) == ans[v] and satisfied_count[idx // 3] == 1:
                        b += 1
                        if b >= min_break: break
                if b < min_break:
                    min_break, best_v = b, v
                    if b == 0: break
        
        old_val = ans[best_v]
        new_val = 1 - old_val
        ans[best_v] = new_val
        for idx in adj[offsets[best_v] : offsets[best_v+1]]:
            L, c_i = clauses[idx], idx // 3
            if (L > 0) == old_val:
                satisfied_count[c_i] -= 1
                if satisfied_count[c_i] == 0:
                    pos_in_unsatisfied[c_i] = len(unsatisfied_list)
                    unsatisfied_list.append(c_i)
            else:
                if satisfied_count[c_i] == 0:
                    last_c = unsatisfied_list.pop()
                    pos = pos_in_unsatisfied[c_i]
                    if c_i != last_c:
                        unsatisfied_list[pos], pos_in_unsatisfied[last_c] = last_c, pos
                    pos_in_unsatisfied[c_i] = -1
                satisfied_count[c_i] += 1

    if not unsatisfied_list:
        sys.stdout.write("SAT\n")
        res = [" ".join(map(str, ans[i : min(i + 1000000, n_vars + 1)])) for i in range(1, n_vars + 1, 1000000)]
        sys.stdout.write(" ".join(res) + "\n")

if __name__ == '__main__':
    solve()
```
