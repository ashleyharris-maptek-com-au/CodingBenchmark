using System;
using System.Collections.Generic;
using System.Linq;

namespace DFVS
{
    class Program
    {
        static DateTime startTime;

        static void Main()
        {
            startTime = DateTime.Now;
            var line = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            if (line == null) return;
            if (!int.TryParse(line[0], out int n) || !int.TryParse(line[1], out int m)) return;

            var adj = new HashSet<int>[n];
            var revAdj = new HashSet<int>[n];
            for (int i = 0; i < n; i++)
            {
                adj[i] = new HashSet<int>();
                revAdj[i] = new HashSet<int>();
            }

            for (int i = 0; i < m; i++)
            {
                var edge = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
                if (edge == null) break;
                if (edge.Length < 2) continue;
                if (int.TryParse(edge[0], out int u) && int.TryParse(edge[1], out int v))
                {
                    if (u >= 0 && u < n && v >= 0 && v < n)
                    {
                        adj[u].Add(v);
                        revAdj[v].Add(u);
                    }
                }
            }

            var result = Solve(n, adj, revAdj);
            Console.WriteLine(result.Count);
            Console.WriteLine(string.Join(" ", result.OrderBy(x => x)));
        }

        static List<int> Solve(int n, HashSet<int>[ ] adj, HashSet<int>[ ] revAdj)
        {
            var fvs = new List<int>();
            var active = new HashSet<int>(Enumerable.Range(0, n));
            ApplyReductions(adj, revAdj, active, fvs);

            if (active.Count > 0)
            {
                var sccs = GetSCCs(adj, active, n);
                foreach (var scc in sccs)
                {
                    if (scc.Count == 0) continue;
                    var sccFVS = SolveSCC(adj, revAdj, scc);
                    fvs.AddRange(sccFVS);
                }
            }
            return fvs;
        }

        static bool ApplyReductions(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active, List<int> fvs)
        {
            bool totalChanged = false;
            bool changed = true;
            while (changed)
            {
                changed = false;
                var nodes = active.ToList();
                foreach (var v in nodes)
                {
                    if (!active.Contains(v)) continue;
                    if (adj[v].Contains(v)) { fvs.Add(v); RemoveNode(adj, revAdj, active, v); changed = true; totalChanged = true; continue; }
                    if (adj[v].Count == 0 || revAdj[v].Count == 0) { RemoveNode(adj, revAdj, active, v); changed = true; totalChanged = true; continue; }
                    if (revAdj[v].Count == 1) { MergeIn(adj, revAdj, active, v, revAdj[v].First()); changed = true; totalChanged = true; continue; }
                    if (adj[v].Count == 1) { MergeOut(adj, revAdj, active, v, adj[v].First()); changed = true; totalChanged = true; continue; }
                }
            }
            return totalChanged;
        }

        static void RemoveNode(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active, int v)
        {
            foreach (var u in revAdj[v]) if (u != v) adj[u].Remove(v);
            foreach (var w in adj[v]) if (w != v) revAdj[w].Remove(v);
            adj[v].Clear(); revAdj[v].Clear(); active.Remove(v);
        }

        static void MergeIn(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active, int v, int u)
        {
            foreach (var w in adj[v])
            {
                if (w == v) continue;
                if (w == u) { adj[u].Add(u); revAdj[u].Add(u); }
                else { adj[u].Add(w); revAdj[w].Add(u); }
            }
            RemoveNode(adj, revAdj, active, v);
        }

        static void MergeOut(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active, int v, int u)
        {
            foreach (var w in revAdj[v])
            {
                if (w == v) continue;
                if (w == u) { adj[u].Add(u); revAdj[u].Add(u); }
                else { revAdj[u].Add(w); adj[w].Add(u); }
            }
            RemoveNode(adj, revAdj, active, v);
        }

        static List<int> SolveSCC(HashSet<int>[ ] globalAdj, HashSet<int>[ ] globalRevAdj, List<int> vertices)
        {
            var active = new HashSet<int>(vertices);
            var adj = new HashSet<int>[globalAdj.Length];
            var revAdj = new HashSet<int>[globalAdj.Length];
            foreach (var v in vertices)
            {
                adj[v] = new HashSet<int>();
                foreach (var w in globalAdj[v]) if (active.Contains(w)) adj[v].Add(w);
                revAdj[v] = new HashSet<int>();
                foreach (var w in globalRevAdj[v]) if (active.Contains(w)) revAdj[v].Add(w);
            }

            int bestSize = int.MaxValue;
            List<int> bestFVS = null;

            void Search(HashSet<int>[ ] curAdj, HashSet<int>[ ] curRevAdj, HashSet<int> curActive, List<int> currentPicked)
            {
                if (currentPicked.Count >= bestSize) return;
                if ((DateTime.Now - startTime).TotalSeconds > 28) return;

                var localActive = new HashSet<int>(curActive);
                var localAdj = CloneEdges(curAdj, localActive);
                var localRevAdj = CloneEdges(curRevAdj, localActive);
                var localPicked = new List<int>(currentPicked);

                ApplyReductions(localAdj, localRevAdj, localActive, localPicked);
                if (localPicked.Count >= bestSize) return;
                if (localActive.Count == 0) { if (localPicked.Count < bestSize) { bestSize = localPicked.Count; bestFVS = localPicked; } return; }
                if (localPicked.Count + GetLowerBound(localAdj, localActive) >= bestSize) return;

                int v = PickBranchVertex(localAdj, localRevAdj, localActive);
                
                var nextActive1 = new HashSet<int>(localActive);
                var nextAdj1 = CloneEdges(localAdj, nextActive1);
                var nextRevAdj1 = CloneEdges(localRevAdj, nextActive1);
                RemoveNode(nextAdj1, nextRevAdj1, nextActive1, v);
                var nextP1 = new List<int>(localPicked); nextP1.Add(v);
                Search(nextAdj1, nextRevAdj1, nextActive1, nextP1);

                if (localPicked.Count < bestSize)
                {
                    var nextActive2 = new HashSet<int>(localActive);
                    var nextAdj2 = CloneEdges(localAdj, nextActive2);
                    var nextRevAdj2 = CloneEdges(localRevAdj, nextActive2);
                    Bypass(nextAdj2, nextRevAdj2, nextActive2, v);
                    Search(nextAdj2, nextRevAdj2, nextActive2, localPicked);
                }
            }

            bestFVS = Greedy(CloneEdges(adj, active), CloneEdges(revAdj, active), new HashSet<int>(active));
            bestSize = bestFVS.Count;
            Search(adj, revAdj, active, new List<int>());
            return bestFVS;
        }

        static HashSet<int>[ ] CloneEdges(HashSet<int>[ ] edges, HashSet<int> active)
        {
            var clone = new HashSet<int>[edges.Length];
            foreach (int v in active) clone[v] = new HashSet<int>(edges[v]);
            return clone;
        }

        static int PickBranchVertex(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active)
        {
            int bestV = -1; long minVal = long.MaxValue;
            foreach (int v in active)
            {
                long val = (long)adj[v].Count * revAdj[v].Count;
                if (val < minVal) { minVal = val; bestV = v; }
            }
            return bestV;
        }

        static void Bypass(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active, int v)
        {
            foreach (var u in revAdj[v])
            {
                if (u == v) continue;
                foreach (var w in adj[v])
                {
                    if (w == v) continue;
                    if (u == w) { adj[u].Add(u); revAdj[u].Add(u); }
                    else { adj[u].Add(w); revAdj[w].Add(u); }
                }
            }
            RemoveNode(adj, revAdj, active, v);
        }

        static int GetLowerBound(HashSet<int>[ ] adj, HashSet<int> active)
        {
            int lb = 0; var seen = new HashSet<int>();
            foreach (int v in active)
            {
                if (seen.Contains(v)) continue;
                foreach (int w in adj[v]) if (w != v && active.Contains(w) && adj[w].Contains(v)) { lb++; seen.Add(v); seen.Add(w); break; }
            }
            return lb;
        }

        static List<int> Greedy(HashSet<int>[ ] adj, HashSet<int>[ ] revAdj, HashSet<int> active)
        {
            var fvs = new List<int>();
            while (active.Count > 0)
            {
                ApplyReductions(adj, revAdj, active, fvs);
                if (active.Count == 0) break;
                int bestV = -1; long maxVal = -1;
                foreach (int v in active)
                {
                    long val = (long)adj[v].Count * revAdj[v].Count;
                    if (val > maxVal) { maxVal = val; bestV = v; }
                }
                fvs.Add(bestV); RemoveNode(adj, revAdj, active, bestV);
            }
            return fvs;
        }

        static List<List<int>> GetSCCs(HashSet<int>[ ] adj, HashSet<int> active, int n)
        {
            int[ ] index = new int[n]; int[ ] lowlink = new int[n]; bool[ ] onStack = new bool[n];
            var stack = new Stack<int>(); int currentIndex = 0; var sccs = new List<List<int>>();
            foreach (var i in active) if (index[i] == 0) StrongConnectIterative(i, adj, active, index, lowlink, onStack, stack, ref currentIndex, sccs);
            return sccs;
        }

        static void StrongConnectIterative(int startNode, HashSet<int>[ ] adj, HashSet<int> active, int[ ] index, int[ ] lowlink, bool[ ] onStack, Stack<int> stack, ref int currentIndex, List<List<int>> sccs)
        {
            var callStack = new Stack<(int v, IEnumerator<int> it)>();
            callStack.Push((startNode, adj[startNode].GetEnumerator()));
            index[startNode] = lowlink[startNode] = ++currentIndex;
            stack.Push(startNode); onStack[startNode] = true;
            while (callStack.Count > 0)
            {
                var top = callStack.Peek(); int v = top.v; var it = top.it;
                if (it.MoveNext())
                {
                    int w = it.Current;
                    if (!active.Contains(w)) continue;
                    if (index[w] == 0)
                    {
                        index[w] = lowlink[w] = ++currentIndex;
                        stack.Push(w); onStack[w] = true;
                        callStack.Push((w, adj[w].GetEnumerator()));
                    }
                    else if (onStack[w]) lowlink[v] = Math.Min(lowlink[v], index[w]);
                }
                else
                {
                    callStack.Pop();
                    if (lowlink[v] == index[v])
                    {
                        var scc = new List<int>(); int w;
                        do { w = stack.Pop(); onStack[w] = false; scc.Add(w); } while (w != v);
                        if (scc.Count > 1 || adj[v].Contains(v)) sccs.Add(scc);
                    }
                    if (callStack.Count > 0)
                    {
                        var parent = callStack.Peek().v;
                        lowlink[parent] = Math.Min(lowlink[parent], lowlink[v]);
                    }
                }
            }
        }
    }
}
