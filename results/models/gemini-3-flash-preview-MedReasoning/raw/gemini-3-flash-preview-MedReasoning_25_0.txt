The C# solution implements a robust orbital mechanics solver designed to handle varying warning times and mission complexities. It uses a **KDK (Kick-Drift-Kick) Leapfrog integrator** to perform a high-precision N-body simulation of the solar system, including the Sun, Earth, Moon, and major planets, matching the simulation's physical model.

Key features of the implementation:
1.  **N-Body Integration:** The solver pre-integrates the trajectories of all celestial bodies from the initial state provided in the input, identifying the precise time and state of the potential collision between the asteroid and Earth.
2.  **Trajectory Optimization:** It performs a grid search over the launch window and impact times. For each candidate window, it employs a **shooting method** (differential correction) to find the precise launch velocity required to intercept the asteroid in the full N-body gravity field.
3.  **Kinetic Impact Maximization:** The solver accounts for the spacecraft's mass and the available delta-v budget. Any budget remaining after the initial transfer is used for a final burn at impact to maximize the relative velocity and, consequently, the momentum transfer to the asteroid, maximizing its deflection away from Earth.
4.  **Robustness:** The code handles both short-term (1-2 years) and long-term (30-50 years) scenarios by adaptively adjusting the simulation step size and searching a wide range of trajectory options.

The output provides the number of burns and a precise schedule of maneuvers (launch and impact) in SI units, formatted as required for the simulation environment.

using System;

namespace AsteroidInterception

    public struct Vector3d
    {
        public double X, Y, Z;
        public Vector3d(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3d operator +(Vector3d a, Vector3d b) => new Vector3d(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3d operator -(Vector3d a, Vector3d b) => new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3d operator *(Vector3d a, double b) => new Vector3d(a.X * b, a.Y * b, a.Z * b);
        public static Vector3d operator *(double a, Vector3d b) => new Vector3d(a * b.X, a * b.Y, a * b.Z);
        public static Vector3d operator /(Vector3d a, double b) => new Vector3d(a.X / b, a.Y / b, a.Z / b);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double LengthSquared() => X * X + Y * Y + Z * Z;
        public static double Dot(Vector3d a, Vector3d b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3d Cross(Vector3d a, Vector3d b) => new Vector3d(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    public class Body
    {
        public string Name;
        public double Mass;
    }

    public class Solution
    {
        public double LaunchTime;
        public Vector3d LaunchDV;
        public double ImpactTime;
        public Vector3d FinalBurn;
    }

    class Program
    {
        static double G = 6.67430e-11;

        static void Main(string[] args)
        {

            TextReader reader = Console.In;
            string line = reader.ReadLine();
            if (string.IsNullOrEmpty(line)) return;
            string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int numBodies = int.Parse(parts[0]);
            double warningDays = double.Parse(parts[1]);
            double impactorMass = double.Parse(parts[2]);
            double deltaVBudget = double.Parse(parts[3]);
            double asteroidMass = double.Parse(parts[4]);
            double t0_unix = double.Parse(parts[5]);

            List<Body> bodies = new List<Body>();
            Vector3d[] initialPos = new Vector3d[numBodies];
            Vector3d[] initialVel = new Vector3d[numBodies];
            double[] masses = new double[numBodies];

            int earthIdx = -1, astIdx = -1;

            for (int i = 0; i < numBodies; i++)
            {
                line = reader.ReadLine();
                parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                bodies.Add(new Body { Name = parts[0], Mass = double.Parse(parts[1]) });
                masses[i] = bodies[i].Mass;
                initialPos[i] = new Vector3d(double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4]));
                initialVel[i] = new Vector3d(double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]));

                if (bodies[i].Name == "Earth") earthIdx = i;
                if (bodies[i].Name == "Asteroid") astIdx = i;
            }

            string bspPath = reader.ReadLine();

            double dt = 3600.0;
            if (warningDays < 730) dt = 600.0;
            if (warningDays > 10000) dt = 7200.0;
            
            double totalDuration = (warningDays + 10) * 86400.0;
            int numSteps = (int)(totalDuration / dt) + 1;

            double[][] posTable = new double[numBodies][];
            double[][] velTable = new double[numBodies][];
            for (int i = 0; i < numBodies; i++) {
                posTable[i] = new double[numSteps * 3];
                velTable[i] = new double[numSteps * 3];
            }

            Vector3d[] currentPos = (Vector3d[])initialPos.Clone();
            Vector3d[] currentVel = (Vector3d[])initialVel.Clone();
            Vector3d[] acc = new Vector3d[numBodies];

            ComputeAcc(currentPos, masses, acc);
            for (int s = 0; s < numSteps; s++)
            {
                for (int i = 0; i < numBodies; i++) {
                    posTable[i][s * 3] = currentPos[i].X;
                    posTable[i][s * 3 + 1] = currentPos[i].Y;
                    posTable[i][s * 3 + 2] = currentPos[i].Z;
                    velTable[i][s * 3] = currentVel[i].X;
                    velTable[i][s * 3 + 1] = currentVel[i].Y;
                    velTable[i][s * 3 + 2] = currentVel[i].Z;

                for (int i = 0; i < numBodies; i++) currentPos[i] += currentVel[i] * dt;
                ComputeAcc(currentPos, masses, acc);
                for (int i = 0; i < numBodies; i++) currentVel[i] += acc[i] * (dt * 0.5);

                        Vector3d vRel = vFinalSC - vAI;
                        Vector3d finalBurn = vRel * (leftover / (vRel.Length() + 1e-9));

                        if (miss > bestMiss)

                            bestMiss = miss;
                            bestSol = new Solution {

        static void ComputeAcc(Vector3d[] pos, double[] masses, Vector3d[] acc)
        {
            for (int i = 0; i < pos.Length; i++) {
                double ax = 0, ay = 0, az = 0;
                for (int j = 0; j < pos.Length; j++) {
                    if (i == j) continue;
                    double dx = pos[j].X - pos[i].X;
                    double dy = pos[j].Y - pos[i].Y;
                    double dz = pos[j].Z - pos[i].Z;
                    double r2 = dx * dx + dy * dy + dz * dz;
                    double f = G * masses[j] / (r2 * Math.Sqrt(r2));
                    ax += dx * f; ay += dy * f; az += dz * f;
                }
                acc[i] = new Vector3d(ax, ay, az);
            }
        }

        static Vector3d GetStateFromTable(double[][] table, int idx, int step) => new Vector3d(table[idx][step * 3], table[idx][step * 3 + 1], table[idx][step * 3 + 2]);

        static Vector3d GetInterpolated(double[][] table, int idx, double t, double dt)
        {
            double s = t / dt;
            int i = (int)s;
            if (i < 0) i = 0;
            int n = table[idx].Length / 3;
            if (i >= n - 1) return GetStateFromTable(table, idx, n - 1);
            double f = s - i;
            return GetStateFromTable(table, idx, i) * (1 - f) + GetStateFromTable(table, idx, i + 1) * f;
        }

        static Vector3d PropagateSpacecraft(Vector3d p0, Vector3d v0, double tStart, double tEnd, double dtTable, double[][] posTable, double[] masses, out Vector3d vFinal)
        {
            Vector3d p = p0, v = v0;
            double t = tStart;
            double totalT = tEnd - tStart;
            int nSteps = 1000;
            double step = totalT / nSteps;
            Vector3d a = ComputeSpacecraftAcc(p, t, dtTable, posTable, masses);
            for (int i = 0; i < nSteps; i++)
            {
                v += a * (step * 0.5);
                p += v * step;
                t += step;
                a = ComputeSpacecraftAcc(p, t, dtTable, posTable, masses);
                v += a * (step * 0.5);
            }
            vFinal = v;
            return p;
        }

        static Vector3d ComputeSpacecraftAcc(Vector3d p, double t, double dtTable, double[][] posTable, double[] masses)
        {
            double ax = 0, ay = 0, az = 0;
            for (int i = 0; i < masses.Length; i++)
            {
                Vector3d pi = GetInterpolated(posTable, i, t, dtTable);
                double dx = pi.X - p.X, dy = pi.Y - p.Y, dz = pi.Z - p.Z;
                double r2 = dx * dx + dy * dy + dz * dz;
                double r2s = r2 + 1e6; 
                double f = G * masses[i] / (r2s * Math.Sqrt(r2s));
                ax += dx * f; ay += dy * f; az += dz * f;
            }
            return new Vector3d(ax, ay, az);
        }
    }
