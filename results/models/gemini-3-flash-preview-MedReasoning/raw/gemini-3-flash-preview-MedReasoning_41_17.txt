; Get UV and scale it
     %uv_val = OpLoad %v2float %in_uv
  %scaled_uv = OpFMul %v2float %uv_val %v2_8_16

; Staggered rows (half-brick offset every other row)
   %scaled_y = OpCompositeExtract %float %scaled_uv 1
    %floor_y = OpExtInst %float %1 Floor %scaled_y
    %mod_y_2 = OpFMod %float %floor_y %float_2
  %x_offset  = OpFMul %float %mod_y_2 %float_0_5
   %scaled_x = OpCompositeExtract %float %scaled_uv 0
%new_scaled_x = OpFAdd %float %scaled_x %x_offset

; Fractional part for cell coordinates
%fract_x_val = OpExtInst %float %1 Fract %new_scaled_x
%fract_y_val = OpExtInst %float %1 Fract %scaled_y

; Determine if current fragment is part of a brick or mortar
    %step_x = OpFOrdGreaterThanEqual %bool %fract_x_val %float_0_05
    %step_y = OpFOrdGreaterThanEqual %bool %fract_y_val %float_0_1
    
    %v1_f = OpSelect %float %step_x %float_1 %float_0
    %v2_f = OpSelect %float %step_y %float_1 %float_0
 %is_brick_f = OpFMul %float %v1_f %v2_f

; Blend mortar and brick colors
%pattern_color = OpExtInst %v3float %1 Mix %mortar_color %brick_color %is_brick_f

; Lambertian shading: (0.4 + 0.6 * max(dot(N, L), 0.0))
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos_v4 = OpLoad %v4float %lightPos_ptr
%lightPos_v3 = OpVectorShuffle %v3float %lightPos_v4 %lightPos_v4 0 1 2
%worldPos_val = OpLoad %v3float %in_worldPos
    %L_unnorm = OpFSub %v3float %lightPos_v3 %worldPos_val
          %L = OpExtInst %v3float %1 Normalize %L_unnorm
%normal_val = OpLoad %v3float %in_normal
          %N = OpExtInst %v3float %1 Normalize %normal_val
      %dotNL = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %dotNL %float_0
   %shading_1 = OpFMul %float %float_0_6 %diffuse
    %shading = OpFAdd %float %float_0_4 %shading_1

; Combine pattern color with shading and output final color
 %final_rgb = OpVectorTimesScalar %v3float %pattern_color %shading
      %out_r = OpCompositeExtract %float %final_rgb 0
      %out_g = OpCompositeExtract %float %final_rgb 1
      %out_b = OpCompositeExtract %float %final_rgb 2
     %out_v4 = OpCompositeConstruct %v4float %out_r %out_g %out_b %float_1
