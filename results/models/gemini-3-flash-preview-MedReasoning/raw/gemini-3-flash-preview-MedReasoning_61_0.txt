The following TypeScript program implements a Vehicle Routing Problem (VRP) solver designed for scalability and performance. It utilizes the **Sweep Algorithm** to partition packages into angular slices, ensuring that each van covers a spatially coherent area. Within each slice, it applies a **Nearest Neighbor** heuristic to order the deliveries, with a windowed search optimization to maintain efficiency ($O(N \log N)$ or $O(N \cdot \text{window})$) for large datasets.

```typescript
"use strict";

declare const require: any;
const fs = require('fs');

interface Package {
    id: string | number;
    x: number;
    y: number;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

interface MetaPackage extends Package {
    angle: number;
    distSq: number;
}

/**
 * Standard O(N^2) Nearest Neighbor solver for smaller chunks.
 */
function solveNN(chunk: MetaPackage[], dx: number, dy: number): (string | number)[] {
    const n = chunk.length;
    const route: (string | number)[] = [];
    const visited = new Uint8Array(n);
    let currX = dx;
    let currY = dy;

    for (let i = 0; i < n; i++) {
        let bestIdx = -1;
        let minDistSq = Infinity;
        for (let j = 0; j < n; j++) {
            if (visited[j]) continue;
            const p = chunk[j];
            const dSq = (p.x - currX) ** 2 + (p.y - currY) ** 2;
            if (dSq < minDistSq) {
                minDistSq = dSq;
                bestIdx = j;
            }
        }
        if (bestIdx !== -1) {
            visited[bestIdx] = 1;
            const p = chunk[bestIdx];
            route.push(p.id);
            currX = p.x;
            currY = p.y;
        }
    }
    return route;
}

/**
 * Optimized Windowed Nearest Neighbor solver for large chunks.
 * Searches within a local window of the angle-sorted list to find neighbors efficiently.
 */
function solveWindowedNN(chunk: MetaPackage[], dx: number, dy: number): (string | number)[] {
    const n = chunk.length;
    if (n === 0) return [];
    
    // Start with the point nearest to the depot
    let startIdx = 0;
    let minDistSq = chunk[0].distSq;
    for (let i = 1; i < n; i++) {
        if (chunk[i].distSq < minDistSq) {
            minDistSq = chunk[i].distSq;
            startIdx = i;
        }
    }

    const route: (string | number)[] = [];
    const visited = new Uint8Array(n);
    let currIdx = startIdx;
    visited[currIdx] = 1;
    route.push(chunk[currIdx].id);
    let currX = chunk[currIdx].x;
    let currY = chunk[currIdx].y;

    const windowSize = 500;

    for (let i = 1; i < n; i++) {
        let bestIdx = -1;
        let bestDistSq = Infinity;

        const maxStep = Math.min(windowSize, n);
        for (let step = 1; step <= maxStep; step++) {
            // Check right in sorted list
            const rIdx = currIdx + step;
            if (rIdx < n && !visited[rIdx]) {
                const dSq = (chunk[rIdx].x - currX) ** 2 + (chunk[rIdx].y - currY) ** 2;
                if (dSq < bestDistSq) {
                    bestDistSq = dSq;
                    bestIdx = rIdx;
                }
            }
            // Check left in sorted list
            const lIdx = currIdx - step;
            if (lIdx >= 0 && !visited[lIdx]) {
                const dSq = (chunk[lIdx].x - currX) ** 2 + (chunk[lIdx].y - currY) ** 2;
                if (dSq < bestDistSq) {
                    bestDistSq = dSq;
                    bestIdx = lIdx;
                }
            }
        }

        // Fallback: jump to the first available unvisited point if window is exhausted
        if (bestIdx === -1) {
            for (let j = 0; j < n; j++) {
                if (!visited[j]) {
                    bestIdx = j;
                    break;
                }
            }
        }

        if (bestIdx !== -1) {
            visited[bestIdx] = 1;
            route.push(chunk[bestIdx].id);
            currX = chunk[bestIdx].x;
            currY = chunk[bestIdx].y;
            currIdx = bestIdx;
        }
    }
    return route;
}

function solve() {
    let input: string = '';
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data: InputData;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }
    const { depot, truckCount, packages } = data;

    if (!packages || packages.length === 0) {
        process.stdout.write(JSON.stringify({ routes: [] }) + "\n");
        return;
    }

    const [dx, dy] = depot;
    const pkgs: MetaPackage[] = packages.map(p => ({
        ...p,
        angle: Math.atan2(p.y - dy, p.x - dx),
        distSq: (p.x - dx) ** 2 + (p.y - dy) ** 2
    }));

    // Sweep Algorithm: Sort all packages by angle relative to depot
    pkgs.sort((a, b) => a.angle - b.angle);

    const n = pkgs.length;
    let actualTruckCount = Math.min(truckCount, n);
    if (actualTruckCount <= 0 && n > 0) actualTruckCount = 1;
    
    const routes: (string | number)[][] = [];

    // Partition sorted packages into slices for each truck
    for (let i = 0; i < actualTruckCount; i++) {
        const start = Math.floor((i * n) / actualTruckCount);
        const end = Math.floor(((i + 1) * n) / actualTruckCount);
        const chunk = pkgs.slice(start, end);
        
        if (chunk.length > 0) {
            // Use O(N^2) for small chunks, optimized windowed search for large chunks
            if (chunk.length < 4000) {
                routes.push(solveNN(chunk, dx, dy));
            } else {
                routes.push(solveWindowedNN(chunk, dx, dy));
            }
        }
    }

    process.stdout.write(JSON.stringify({ routes }) + "\n");
}

solve();
```
