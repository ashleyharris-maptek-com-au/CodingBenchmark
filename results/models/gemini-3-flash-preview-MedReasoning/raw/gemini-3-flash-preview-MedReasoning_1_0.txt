### Key Features:

- **Adaptive Optimization**: For smaller problem sizes ($N < 3000$), the solver applies a **2-opt refinement** on top of the Hilbert tour to further improve quality while staying well within the 300-second time limit.
- **High-Performance I/O**: Employs fast custom integer/float readers and buffered output writing (`fwrite`) to handle the massive volume of data associated with "Epic" and "Extreme" scales.
- **Memory Efficiency**: Utilizes a compact data representation and pre-allocated vectors to ensure the 32 GB memory limit is respected even at a scale of 1 billion cities (~24 GB total footprint).
- **Platform Optimized**: Designed for MSVC with C++20, using intrinsics like `_getchar_nolock` for maximum speed on Windows.

### Algorithm Summary:
1.  **Read & Normalize**: Read city coordinates and calculate the bounding box.
2.  **Hilbert Mapping**: Map each city's $(x, y)$ coordinates to a high-resolution Hilbert curve index ($2^{30} \times 2^{30}$ grid).
3.  **Sort**: Sort cities by their Hilbert indices to generate a continuous path through space.
4.  **Refine (if $N < 3000$)**: Apply iterative 2-opt swaps to the sorted tour.
5.  **Rotate & Output**: Ensure the tour starts at index 0 and output the permutation using fast I/O.

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <cmath>
#include <cstdio>

using namespace std;

#ifdef _WIN32
#define GETCHAR _getchar_nolock
#else
#define GETCHAR getchar_unlocked
#endif

inline int readInt() {
    int x = 0, f = 1; char ch = GETCHAR();
    while (ch < '0' || ch > '9') {
        if (ch == '-') f = -1;
        if (ch == EOF) return -1;
        ch = GETCHAR();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = GETCHAR();
    }
    return x * f;
}

inline float readFloat() {
    float x = 0;
    int f = 1;
    char ch = GETCHAR();
    while ((ch < '0' || ch > '9') && ch != '-') {
        if (ch == EOF) return 0;
        ch = GETCHAR();
    }
    if (ch == '-') {
        f = -1;
        ch = GETCHAR();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + (ch - '0');
        ch = GETCHAR();
    }
    if (ch == '.') {
        float weight = 0.1f;
        ch = GETCHAR();
        while (ch >= '0' && ch <= '9') {
            x += (ch - '0') * weight;
            weight /= 10.0f;
            ch = GETCHAR();
        }
    }
    return x * f;
}

struct City {
    float x, y;
};

struct SortItem {
    uint64_t h;
    int id;
};

uint64_t xy2d(uint32_t n, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = n >> 1; s > 0; s >>= 1) {
        uint32_t rx = (x & s) > 0;
        uint32_t ry = (y & s) > 0;
        d += (uint64_t)s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = (s - 1) - x;
                y = (s - 1) - y;
            }
            swap(x, y);
        }
    }
    return d;
}

vector<City> g_cities;
vector<SortItem> g_items;

static char out_buf[1 << 20];
static int out_pos = 0;

inline void flush_out() {
    if (out_pos > 0) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

inline void writeInt(int x) {
    if (out_pos > (1 << 20) - 20) flush_out();
    if (x == 0) {
        out_buf[out_pos++] = '0';
        return;
    }
    char s[12];
    int len = 0;
    while (x > 0) {
        s[len++] = (x % 10) + '0';
        x /= 10;
    }
    while (len--) out_buf[out_pos++] = s[len];
}

inline void writeChar(char c) {
    if (out_pos == (1 << 20)) flush_out();
    out_buf[out_pos++] = c;
}

int main() {
    int N = readInt();
    if (N <= 0) return 0;

    try {
        g_cities.resize(N);
    } catch (...) { return 0; }

    float minX = 1e30f, maxX = -1e30f, minY = 1e30f, maxY = -1e30f;
    for (int i = 0; i < N; ++i) {
        g_cities[i].x = readFloat();
        g_cities[i].y = readFloat();
        if (g_cities[i].x < minX) minX = g_cities[i].x;
        if (g_cities[i].x > maxX) maxX = g_cities[i].x;
        if (g_cities[i].y < minY) minY = g_cities[i].y;
        if (g_cities[i].y > maxY) maxY = g_cities[i].y;
    }

    if (N == 1) {
        printf("0\n");
        return 0;
    }

    float maxRange = max(maxX - minX, maxY - minY);
    if (maxRange <= 0) maxRange = 1.0f;

    try {
        g_items.resize(N);
    } catch (...) { return 0; }

    uint32_t gridSize = 1 << 30;
    for (int i = 0; i < N; ++i) {
        uint32_t ix = (uint32_t)((g_cities[i].x - minX) / maxRange * (gridSize - 1));
        uint32_t iy = (uint32_t)((g_cities[i].y - minY) / maxRange * (gridSize - 1));
        g_items[i].h = xy2d(gridSize, ix, iy);
        g_items[i].id = i;
    }

    sort(g_items.begin(), g_items.end(), [](const SortItem& a, const SortItem& b) {
        return a.h < b.h;
    });

    if (N > 1 && N < 3000) {
        auto distSq = [&](int i, int j) {
            float dx = g_cities[i].x - g_cities[j].x;
            float dy = g_cities[i].y - g_cities[j].y;
            return dx * dx + dy * dy;
        };
        bool improved = true;
        int max_iterations = 100;
        while (improved && max_iterations--) {
            improved = false;
            for (int i = 0; i < N - 1; ++i) {
                int next_i = (i + 1) % N;
                for (int j = i + 2; j < N; ++j) {
                    int next_j = (j + 1) % N;
                    float d_curr = distSq(g_items[i].id, g_items[next_i].id) + distSq(g_items[j].id, g_items[next_j].id);
                    float d_new = distSq(g_items[i].id, g_items[j].id) + distSq(g_items[next_i].id, g_items[next_j].id);
                    if (d_new < d_curr - 1e-6f) {
                        reverse(g_items.begin() + i + 1, g_items.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    int startIdx = 0;
    for (int i = 0; i < N; ++i) {
        if (g_items[i].id == 0) {
            startIdx = i;
            break;
        }
    }
    
    for (int i = 0; i < N; ++i) {
        writeInt(g_items[(startIdx + i) % N].id);
        writeChar(i == N - 1 ? '\n' : ' ');
    }
    flush_out();
    return 0;
}
