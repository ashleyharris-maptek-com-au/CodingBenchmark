The 2D Polygon Cutting Stock Problem is a complex NP-hard optimization challenge. For a solution to scale from a few simple shapes to over 5000 complex polygons within 5 minutes, it must balance geometric accuracy with computational efficiency.

The following Rust implementation uses a **Greedy Placement strategy** enhanced by:
1.  **Candidate-Point Heuristic**: Instead of a dense grid search, it identifies "corners" generated by the stock boundary and already-placed pieces, which are the most likely locations for a tight fit.
2.  **Spatial Grid Partitioning**: A spatial grid significantly accelerates collision detection by limiting overlap checks to nearby polygons, transforming an $O(N^2)$ problem into a near-linear one.
3.  **Adaptive Precision**: It adjusts the number of rotations and candidate points based on the problem scale, ensuring performance for massive inputs while maintaining quality for smaller ones.
4.  **Robust Geometry**: Full polygon-in-polygon and polygon-overlap tests are implemented using ray-casting and edge-intersection algorithms, handling concave stocks and complex pieces.

```rust
use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn rotate(&self, angle: f64) -> Point {
        let s = angle.sin();
        let c = angle.cos();
        Point {
            x: self.x * c - self.y * s,
            y: self.x * s + self.y * c,
        }
    }
    fn add(&self, other: Point) -> Point {
        Point { x: self.x + other.x, y: self.y + other.y }
    }
}

fn polygon_area(vertices: &[Point]) -> f64 {
    let mut area = 0.0;
    for i in 0..vertices.len() {
        let j = (i + 1) % vertices.len();
        area += vertices[i].x * vertices[j].y;
        area -= vertices[j].x * vertices[i].y;
    }
    area.abs() * 0.5
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

impl BBox {
    fn from_vertices(vertices: &[Point]) -> Self {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for p in vertices {
            if p.x < min_x { min_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.x > max_x { max_x = p.x; }
            if p.y > max_y { max_y = p.y; }
        }
        BBox { min_x, min_y, max_x, max_y }
    }
    fn intersects(&self, other: &BBox) -> bool {
        self.min_x < other.max_x && self.max_x > other.min_x &&
        self.min_y < other.max_y && self.max_y > other.min_y
    }
}

#[inline(always)]
fn edges_intersect(a: Point, b: Point, c: Point, d: Point) -> bool {
    let det = (b.x - a.x) * (d.y - c.y) - (b.y - a.y) * (d.x - c.x);
    if det.abs() < 1e-10 { return false; }
    let t = ((c.x - a.x) * (d.y - c.y) - (c.y - a.y) * (d.x - c.x)) / det;
    let u = ((c.x - a.x) * (b.y - a.y) - (c.y - a.y) * (b.x - a.x)) / det;
    t > 1e-10 && t < 1.0 - 1e-10 && u > 1e-10 && u < 1.0 - 1e-10
}

fn is_point_in_poly(p: Point, poly: &[Point]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        if ((poly[i].y > p.y) != (poly[j].y > p.y)) &&
           (p.x < (poly[j].x - poly[i].x) * (p.y - poly[i].y) / (poly[j].y - poly[i].y) + poly[i].x) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn polygons_overlap(p1: &[Point], bbox1: &BBox, p2: &[Point], bbox2: &BBox) -> bool {
    if !bbox1.intersects(bbox2) { return false; }
    for i in 0..p1.len() {
        let a = p1[i];
        let b = p1[(i + 1) % p1.len()];
        for j in 0..p2.len() {
            if edges_intersect(a, b, p2[j], p2[(j + 1) % p2.len()]) { return true; }
        }
    }
    if is_point_in_poly(p1[0], p2) { return true; }
    if is_point_in_poly(p2[0], p1) { return true; }
    false
}

fn is_poly_in_poly(small: &[Point], small_bbox: &BBox, large: &[Point], large_bbox: &BBox) -> bool {
    if small_bbox.min_x < large_bbox.min_x - 1e-7 || small_bbox.max_x > large_bbox.max_x + 1e-7 ||
       small_bbox.min_y < large_bbox.min_y - 1e-7 || small_bbox.max_y > large_bbox.max_y + 1e-7 {
        return false;
    }
    for &p in small {
        if !is_point_in_poly(p, large) { return false; }
    }
    for i in 0..small.len() {
        let a = small[i];
        let b = small[(i + 1) % small.len()];
        for j in 0..large.len() {
            if edges_intersect(a, b, large[j], large[(j + 1) % large.len()]) { return false; }
        }
    }
    true
}

#[derive(Clone, Debug)]
struct RotatedPiece {
    angle: f64,
    vertices: Vec<Point>,
    bbox: BBox,
}

#[derive(Clone, Debug)]
struct Piece {
    id: usize,
    rotations: Vec<RotatedPiece>,
    area: f64,
}

struct PlacedPiece {
    vertices: Vec<Point>,
    bbox: BBox,
}

struct SpatialGrid {
    cell_size: f64,
    min_x: f64,
    min_y: f64,
    nx: usize,
    ny: usize,
    cells: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(bbox: BBox, count: usize) -> Self {
        let count = count.max(100);
        let area = (bbox.max_x - bbox.min_x) * (bbox.max_y - bbox.min_y);
        let cell_size = (area / count as f64).sqrt().max(1.0);
        let nx = (((bbox.max_x - bbox.min_x) / cell_size).ceil() as usize).max(1) + 1;
        let ny = (((bbox.max_y - bbox.min_y) / cell_size).ceil() as usize).max(1) + 1;
        SpatialGrid {
            cell_size, min_x: bbox.min_x, min_y: bbox.min_y, nx, ny,
            cells: vec![Vec::new(); nx * ny],
        }
    }
    fn add(&mut self, bbox: BBox, idx: usize) {
        let x1 = (((bbox.min_x - self.min_x) / self.cell_size).floor() as i32).max(0);
        let x2 = (((bbox.max_x - self.min_x) / self.cell_size).floor() as i32).min(self.nx as i32 - 1);
        let y1 = (((bbox.min_y - self.min_y) / self.cell_size).floor() as i32).max(0);
        let y2 = (((bbox.max_y - self.min_y) / self.cell_size).floor() as i32).min(self.ny as i32 - 1);
        for ix in x1..=x2 {
            for iy in y1..=y2 {
                self.cells[(ix as usize) * self.ny + (iy as usize)].push(idx);
            }
        }
    }
    fn query(&self, bbox: BBox) -> Vec<usize> {
        let mut results = Vec::new();
        let x1 = (((bbox.min_x - self.min_x) / self.cell_size).floor() as i32).max(0);
        let x2 = (((bbox.max_x - self.min_x) / self.cell_size).floor() as i32).min(self.nx as i32 - 1);
        let y1 = (((bbox.min_y - self.min_y) / self.cell_size).floor() as i32).max(0);
        let y2 = (((bbox.max_y - self.min_y) / self.cell_size).floor() as i32).min(self.ny as i32 - 1);
        for ix in x1..=x2 {
            for iy in y1..=y2 {
                for &idx in &self.cells[(ix as usize) * self.ny + (iy as usize)] {
                    results.push(idx);
                }
            }
        }
        results.sort_unstable();
        results.dedup();
        results
    }
}

struct Stock {
    vertices: Vec<Point>,
    bbox: BBox,
    placed: Vec<PlacedPiece>,
    grid: SpatialGrid,
    candidate_points: Vec<Point>,
}

impl Stock {
    fn new(vertices: Vec<Point>, bbox: BBox, np: usize) -> Self {
        let mut candidate_points = vec![Point { x: bbox.min_x, y: bbox.min_y }];
        for &v in &vertices { candidate_points.push(v); }
        Stock {
            vertices, bbox, placed: Vec::new(),
            grid: SpatialGrid::new(bbox, np),
            candidate_points,
        }
    }
    fn check_collision(&self, piece_verts: &[Point], piece_bbox: &BBox) -> bool {
        if !is_poly_in_poly(piece_verts, piece_bbox, &self.vertices, &self.bbox) { return true; }
        let nearby = self.grid.query(*piece_bbox);
        for &idx in &nearby {
            let p = &self.placed[idx];
            if polygons_overlap(piece_verts, piece_bbox, &p.vertices, &p.bbox) { return true; }
        }
        false
    }
    fn place(&mut self, piece_verts: Vec<Point>, piece_bbox: BBox) {
        let idx = self.placed.len();
        self.placed.push(PlacedPiece { vertices: piece_verts, bbox: piece_bbox });
        self.grid.add(piece_bbox, idx);
        self.candidate_points.push(Point { x: piece_bbox.max_x, y: piece_bbox.min_y });
        self.candidate_points.push(Point { x: piece_bbox.min_x, y: piece_bbox.max_y });
        self.candidate_points.retain(|p| {
            p.x < piece_bbox.min_x - 1e-7 || p.x > piece_bbox.max_x + 1e-7 ||
            p.y < piece_bbox.min_y - 1e-7 || p.y > piece_bbox.max_y + 1e-7
        });
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = match lines.next() { Some(Ok(l)) => l, _ => return };
    let mut parts = first_line.split_whitespace();
    let sv: usize = parts.next().unwrap_or("0").parse().unwrap_or(0);
    let np: usize = parts.next().unwrap_or("0").parse().unwrap_or(0);
    if sv == 0 || np == 0 { return; }
    
    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        if let Some(Ok(line)) = lines.next() {
            let mut p = line.split_whitespace();
            stock_vertices.push(Point { x: p.next().unwrap().parse().unwrap(), y: p.next().unwrap().parse().unwrap() });
        }
    }
    let stock_bbox = BBox::from_vertices(&stock_vertices);
    
    let mut pieces = Vec::new();
    for i in 0..np {
        if let Some(Ok(pv_line)) = lines.next() {
            let pv: usize = pv_line.trim().parse().unwrap();
            let mut vertices = Vec::new();
            for _ in 0..pv {
                if let Some(Ok(line)) = lines.next() {
                    let mut p = line.split_whitespace();
                    vertices.push(Point { x: p.next().unwrap().parse().unwrap(), y: p.next().unwrap().parse().unwrap() });
                }
            }
            let area = polygon_area(&vertices);
            let mut rotations = Vec::new();
            let num_rots = if np > 2000 { 2 } else if np > 500 { 4 } else { 8 };
            for r in 0..num_rots {
                let angle = (r as f64) * (2.0 * PI / (num_rots as f64));
                let rotated_verts: Vec<Point> = vertices.iter().map(|v| v.rotate(angle)).collect();
                rotations.push(RotatedPiece { angle, vertices: rotated_verts.clone(), bbox: BBox::from_vertices(&rotated_verts) });
            }
            pieces.push(Piece { id: i, rotations, area });
        }
    }
    
    pieces.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap());
    let mut stocks: Vec<Stock> = Vec::new();
    let mut results = vec![(0, 0.0, 0.0, 0.0); np];
    
    for piece in pieces {
        let mut placed = false;
        'try_stocks: for (s_idx, stock) in stocks.iter_mut().enumerate() {
            stock.candidate_points.sort_by(|a, b| a.y.partial_cmp(&b.y).unwrap().then(a.x.partial_cmp(&b.x).unwrap()));
            stock.candidate_points.dedup_by(|a, b| (a.x - b.x).abs() < 1e-7 && (a.y - b.y).abs() < 1e-7);
            let cp_limit = if np > 1000 { 50 } else { 200 };
            for rot in &piece.rotations {
                for cp in stock.candidate_points.iter().take(cp_limit) {
                    let translation = Point { x: cp.x - rot.bbox.min_x, y: cp.y - rot.bbox.min_y };
                    let piece_bbox = BBox { min_x: translation.x + rot.bbox.min_x, min_y: translation.y + rot.bbox.min_y, max_x: translation.x + rot.bbox.max_x, max_y: translation.y + rot.bbox.max_y };
                    let moved_verts: Vec<Point> = rot.vertices.iter().map(|v| v.add(translation)).collect();
                    if !stock.check_collision(&moved_verts, &piece_bbox) {
                        results[piece.id] = (s_idx, translation.x, translation.y, rot.angle);
                        stock.place(moved_verts, piece_bbox);
                        placed = true; break 'try_stocks;
                    }
                }
            }
        }
        if !placed {
            let mut new_stock = Stock::new(stock_vertices.clone(), stock_bbox, np);
            let s_idx = stocks.len();
            let mut found = false;
            'new_stock_try: for rot in &piece.rotations {
                for cp in &new_stock.candidate_points {
                    let translation = Point { x: cp.x - rot.bbox.min_x, y: cp.y - rot.bbox.min_y };
                    let piece_bbox = BBox { min_x: translation.x + rot.bbox.min_x, min_y: translation.y + rot.bbox.min_y, max_x: translation.x + rot.bbox.max_x, max_y: translation.y + rot.bbox.max_y };
                    let moved_verts: Vec<Point> = rot.vertices.iter().map(|v| v.add(translation)).collect();
                    if !new_stock.check_collision(&moved_verts, &piece_bbox) {
                        results[piece.id] = (s_idx, translation.x, translation.y, rot.angle);
                        new_stock.place(moved_verts, piece_bbox);
                        stocks.push(new_stock); found = true; break 'new_stock_try;
                    }
                }
            }
            if !found { results[piece.id] = (s_idx, 0.0, 0.0, 0.0); stocks.push(new_stock); }
        }
    }
    println!("{}", stocks.len());
    for res in results { println!("{} {:.8} {:.8} {:.8}", res.0, res.1, res.2, res.3); }
}
```
