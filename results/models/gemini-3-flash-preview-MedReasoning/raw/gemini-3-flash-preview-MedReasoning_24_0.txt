#include <iostream>
#include <vector>
#include <string>
#include <cmath>
#include <thread>
#include <mutex>
#include <atomic>
#include <chrono>
#include <algorithm>
#include <fstream>
#include <cstdio>
#include <iomanip>

using namespace std;

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

#pragma pack(push, 1)
struct TerrainCell {
    uint16_t a, b, c;
};
#pragma pack(pop)

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool updated = false;
};

atomic<bool> g_running{true};
State g_state;
mutex g_state_mutex;

double g_width, g_depth, g_height, g_gravity, g_max_thrust, g_fuel_initial;
double g_start_x, g_start_y, g_start_z;
double g_target_x, g_target_y, g_target_z;
string g_terrain_path;

TerrainCell* g_terrain = nullptr;
atomic<bool> g_terrain_loaded{false};

TerrainCell get_terrain(double x, double y) {
    if (!g_terrain_loaded || !g_terrain) return {0, 0, 0};
    long long ix = (long long)floor(x);
    long long iy = (long long)floor(y);
    if (ix < 0) ix = 0; if (ix >= (long long)g_width) ix = (long long)g_width - 1;
    if (iy < 0) iy = 0; if (iy >= (long long)g_depth) iy = (long long)g_depth - 1;
    return g_terrain[ix * (long long)g_depth + iy];
}

void read_stdin_thread() {
    while (g_running) {
        double x, y, z, vx, vy, vz, pitch, yaw, roll, pr, yr, rr, fuel;
        if (!(cin >> x >> y >> z >> vx >> vy >> vz >> pitch >> yaw >> roll >> pr >> yr >> rr >> fuel)) {
            g_running = false;
            break;
        }
        lock_guard<mutex> lock(g_state_mutex);
        g_state.x = x; g_state.y = y; g_state.z = z;
        g_state.vx = vx; g_state.vy = vy; g_state.vz = vz;
        g_state.pitch = pitch; g_state.yaw = yaw; g_state.roll = roll;
        g_state.pitch_rate = pr; g_state.yaw_rate = yr; g_state.roll_rate = rr;
        g_state.fuel = fuel;
        g_state.updated = true;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> g_width >> g_depth >> g_height >> g_gravity >> g_max_thrust >> g_fuel_initial)) return 0;
    cin >> g_start_x >> g_start_y >> g_start_z >> g_target_x >> g_target_y >> g_target_z;

    string dummy;
    getline(cin, dummy); 
    if (!getline(cin, g_terrain_path)) return 0;

    string marker;
    if (!(cin >> marker)) return 0; // "STATE"

    long long num_cells = (long long)g_width * (long long)g_depth;
    g_terrain = new TerrainCell[num_cells];

    thread terrain_loader([&]() {
        FILE* f = fopen(g_terrain_path.c_str(), "rb");
        if (f) {
            fread(g_terrain, sizeof(TerrainCell), (size_t)num_cells, f);
            fclose(f);
        }
        g_terrain_loaded = true;
    });

    thread reader(read_stdin_thread);

    cout << fixed << setprecision(6);

    while (g_running) {
        State s;
        bool has_update = false;
        {
            lock_guard<mutex> lock(g_state_mutex);
            if (g_state.updated) {
                s = g_state;
                g_state.updated = false;
                has_update = true;
            }
        }

        if (!has_update) {
            this_thread::sleep_for(chrono::milliseconds(1));
            continue;
        }

        if (!g_terrain_loaded) {
            // Hover neutrally while terrain is loading
            cout << "0.3 0.0 0.0 0.0" << endl;
            continue;
        }

        static bool target_gap_identified = false;
        static bool target_in_upper = false;
        if (!target_gap_identified) {
            TerrainCell target_cell = get_terrain(g_target_x, g_target_y);
            target_in_upper = (g_target_z >= target_cell.c);
            target_gap_identified = true;
        }

        double dx = g_target_x - s.x;
        double dy = g_target_y - s.y;
        double dist_h = sqrt(dx*dx + dy*dy);

        // Horizontal velocity control
        double max_v_h = 150.0;
        if (dist_h < 1000) max_v_h = 40.0;
        if (dist_h < 100) max_v_h = 10.0;
        if (dist_h < 20) max_v_h = 2.0;

        double tvx = dx * 0.5;
        double tvy = dy * 0.5;
        double tv_mag = sqrt(tvx*tvx + tvy*tvy);
        if (tv_mag > max_v_h) {
            tvx *= max_v_h / tv_mag;
            tvy *= max_v_h / tv_mag;
        }

        double ax = (tvx - s.vx) * 1.5;
        double ay = (tvy - s.vy) * 1.5;

        // Gap identification and terrain lookahead
        auto get_gap = [&](double x, double y) {
            TerrainCell tc = get_terrain(x, y);
            double mn, mx;
            if (target_in_upper) {
                mn = tc.c + 10.0; mx = g_height - 10.0;
            } else {
                mn = tc.a + 10.0; mx = (tc.b == tc.c) ? g_height - 10.0 : tc.b - 10.0;
            }
            if (mn > mx) mn = mx = (mn + mx) / 2.0;
            return make_pair(mn, mx);
        };

        auto current_gap = get_gap(s.x, s.y);
        double check_min_z = current_gap.first;
        double check_max_z = current_gap.second;
        
        // Lookahead for future terrain
        double lookahead_dist = min(dist_h, 100.0);
        if (dist_h > 1e-3) {
            for (int i = 1; i <= 5; ++i) {
                double px = s.x + dx / dist_h * lookahead_dist * (i / 5.0);
                double py = s.y + dy / dist_h * lookahead_dist * (i / 5.0);
                auto gap = get_gap(px, py);
                check_min_z = max(check_min_z, gap.first);
                check_max_z = min(check_max_z, gap.second);
            }
        }

        double desired_z;
        if (dist_h > 150.0) {
            // Fly near the top of the safe gap for extra clearance
            desired_z = check_max_z - 5.0;
        } else {
            desired_z = g_target_z;
        }

        // Clamp desired altitude to current safe gap
        if (desired_z < check_min_z) desired_z = check_min_z;
        if (desired_z > check_max_z) desired_z = check_max_z;

        double dz = desired_z - s.z;
        double tvz = dz * 0.5;
        if (tvz > 30.0) tvz = 30.0;
        if (tvz < -30.0) tvz = -30.0;
        
        // Soft landing descent rate
        if (dist_h < 20.0 && abs(dz) < 30.0) {
            if (tvz < -1.5) tvz = -1.5;
            if (tvz > 1.5) tvz = 1.5;
        }

        double az_pure = (tvz - s.vz) * 1.0 + g_gravity;
        if (az_pure < 0) az_pure = 0;

        // Calculate total thrust and desired orientation
        double D_mag = sqrt(ax*ax + ay*ay + az_pure*az_pure);
        double thrust_cmd = D_mag / g_max_thrust;
        if (thrust_cmd > 1.0) thrust_cmd = 1.0;

        double d_pitch = 0;
        double d_yaw = s.yaw;
        if (D_mag > 1e-6) {
            d_pitch = acos(max(-1.0, min(1.0, az_pure / D_mag)));
            // Limit tilt to 60 degrees
            if (d_pitch > M_PI / 3.0) d_pitch = M_PI / 3.0;
            
            if (sqrt(ax*ax + ay*ay) > 1e-6) {
                d_yaw = atan2(-ay, -ax);
            }
        }

        // PD control for attitude commands
        double p_err = d_pitch - s.pitch;
        double y_err = d_yaw - s.yaw;
        while (y_err > M_PI) y_err -= 2*M_PI;
        while (y_err < -M_PI) y_err += 2*M_PI;

        double p_cmd = p_err * 12.0 - s.pitch_rate * 3.0;
        double y_cmd = y_err * 12.0 - s.yaw_rate * 3.0;
        double r_cmd = -s.roll * 5.0 - s.roll_rate * 2.0;

        cout << thrust_cmd << " " << p_cmd << " " << y_cmd << " " << r_cmd << endl;
    }

    if (terrain_loader.joinable()) terrain_loader.join();
    if (reader.joinable()) reader.join();
    delete[] g_terrain;
    return 0;
}
