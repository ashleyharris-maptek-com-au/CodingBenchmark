use std::io::{self, BufRead};

#[derive(Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quat {
    fn rotate(&self, v: &Vec3) -> Vec3 {
        let q_vec = Vec3 { x: self.x, y: self.y, z: self.z };
        let cross1 = Vec3 {
            x: q_vec.y * v.z - q_vec.z * v.y,
            y: q_vec.z * v.x - q_vec.x * v.z,
            z: q_vec.x * v.y - q_vec.y * v.x,
        };
        let term1 = Vec3 {
            x: cross1.x + self.w * v.x,
            y: cross1.y + self.w * v.y,
            z: cross1.z + self.w * v.z,
        };
        let cross2 = Vec3 {
            x: q_vec.y * term1.z - q_vec.z * term1.y,
            y: q_vec.z * term1.x - q_vec.x * term1.z,
            z: q_vec.x * term1.y - q_vec.y * term1.x,
        };
        Vec3 {
            x: v.x + 2.0 * cross2.x,
            y: v.y + 2.0 * cross2.y,
            z: v.z + 2.0 * cross2.z,
        }
    }
}

fn cross_product(a: &Vec3, b: &Vec3) -> Vec3 {
    Vec3 {
        x: a.y * b.z - a.z * b.y,
        y: a.z * b.x - a.x * b.z,
        z: a.x * b.y - a.y * b.x,
    }
}

fn dot_product(a: &Vec3, b: &Vec3) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

fn is_point_in_polygon(p: &Point, poly: &[Point]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        if ((poly[i].y > p.y) != (poly[j].y > p.y)) &&
           (p.x < (poly[j].x - poly[i].x) * (p.y - poly[i].y) / (poly[j].y - poly[i].y) + poly[i].x) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn intersect_segments(a: &Point, b: &Point, c: &Point, d: &Point) -> bool {
    fn ccw(p1: &Point, p2: &Point, p3: &Point) -> bool {
        (p3.y - p1.y) * (p2.x - p1.x) > (p2.y - p1.y) * (p3.x - p1.x)
    }
    ccw(a, c, d) != ccw(b, c, d) && ccw(a, b, c) != ccw(a, b, d)
}

fn square_intersects_polygon(sq_min: &Point, sq_max: &Point, poly: &[Point]) -> bool {
    for p in poly {
        if p.x >= sq_min.x && p.x <= sq_max.x && p.y >= sq_min.y && p.y <= sq_max.y {
            return true;
        }
    }
    let sq_pts = [
        Point { x: sq_min.x, y: sq_min.y },
        Point { x: sq_max.x, y: sq_min.y },
        Point { x: sq_max.x, y: sq_max.y },
        Point { x: sq_min.x, y: sq_max.y },
    ];
    for p in &sq_pts {
        if is_point_in_polygon(p, poly) {
            return true;
        }
    }
    let sq_edges = [
        (&sq_pts[0], &sq_pts[1]),
        (&sq_pts[1], &sq_pts[2]),
        (&sq_pts[2], &sq_pts[3]),
        (&sq_pts[3], &sq_pts[0]),
    ];
    for i in 0..poly.len() {
        let p1 = &poly[i];
        let p2 = &poly[(i + 1) % poly.len()];
        let min_px = p1.x.min(p2.x);
        let max_px = p1.x.max(p2.x);
        let min_py = p1.y.min(p2.y);
        let max_py = p1.y.max(p2.y);
        if max_px < sq_min.x || min_px > sq_max.x || max_py < sq_min.y || min_py > sq_max.y {
            continue;
        }
        for sq_edge in &sq_edges {
            if intersect_segments(p1, p2, sq_edge.0, sq_edge.1) {
                return true;
            }
        }
    }
    false
}

fn get_convex_hull_4(pts: [Point; 4]) -> Vec<Point> {
    let mut sorted = pts.to_vec();
    sorted.sort_by(|a, b| a.x.partial_cmp(&b.x).unwrap().then(a.y.partial_cmp(&b.y).unwrap()));
    fn cross(o: &Point, a: &Point, b: &Point) -> f64 {
        (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
    }
    let mut lower = Vec::new();
    for p in &sorted {
        while lower.len() >= 2 && cross(&lower[lower.len()-2], &lower[lower.len()-1], p) <= 0.0 {
            lower.pop();
        }
        lower.push(*p);
    }
    let mut upper = Vec::new();
    for p in sorted.iter().rev() {
        while upper.len() >= 2 && cross(&upper[upper.len()-2], &upper[upper.len()-1], p) <= 0.0 {
            upper.pop();
        }
        upper.push(*p);
    }
    if lower.len() > 0 { lower.pop(); }
    if upper.len() > 0 { upper.pop(); }
    lower.extend(upper);
    lower
}

fn is_point_in_convex_poly(p: &Point, poly: &[Point]) -> bool {
    if poly.len() < 3 { return false; }
    for i in 0..poly.len() {
        let p1 = &poly[i];
        let p2 = &poly[(i + 1) % poly.len()];
        if (p2.x - p1.x) * (p.y - p1.y) - (p2.y - p1.y) * (p.x - p1.x) < -1e-9 {
            return false;
        }
    }
    true
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let mut nv_line = String::new();
    while let Some(line) = lines.next() {
        nv_line = line.unwrap().trim().to_string();
        if !nv_line.is_empty() { break; }
    }
    if nv_line.is_empty() { return; }
    let nv: usize = nv_line.parse().unwrap();
    let mut poly = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let pts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        poly.push(Point { x: pts[0], y: pts[1] });
    }
    let sun_line = lines.next().unwrap().unwrap();
    let sun_vec: Vec<f64> = sun_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let sx = sun_vec[0];
    let sy = sun_vec[1];
    let sz = sun_vec[2];
    let d_len = (sx*sx + sy*sy + sz*sz).sqrt();
    let dx = sx / d_len;
    let dy = sy / d_len;
    let dz = sz / d_len;
    let target = Vec3 { x: dx, y: dy, z: dz };
    let start = Vec3 { x: 0.0, y: 0.0, z: -1.0 };
    let mut quat = if dot_product(&start, &target) > 0.999999 {
        Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 }
    } else if dot_product(&start, &target) < -0.999999 {
        Quat { w: 0.0, x: 1.0, y: 0.0, z: 0.0 }
    } else {
        let axis = cross_product(&start, &target);
        let dot = dot_product(&start, &target);
        let s = ((1.0 + dot) * 2.0).sqrt();
        Quat { w: s / 2.0, x: axis.x / s, y: axis.y / s, z: axis.z / s }
    };
    let q_len = (quat.w*quat.w + quat.x*quat.x + quat.y*quat.y + quat.z*quat.z).sqrt();
    quat.w /= q_len; quat.x /= q_len; quat.y /= q_len; quat.z /= q_len;
    let sz_safe = if sz.abs() < 1e-12 { if sz >= 0.0 { 1e-12 } else { -1e-12 } } else { sz };
    let u = sx / sz_safe;
    let v = sy / sz_safe;
    let v_ref = [
        Vec3 { x: 1.0, y: 0.0, z: -0.707 },
        Vec3 { x: -1.0, y: 0.0, z: -0.707 },
        Vec3 { x: 0.0, y: 1.0, z: 0.707 },
        Vec3 { x: 0.0, y: -1.0, z: 0.707 },
    ];
    let mut w_pts = [Point{x:0.0, y:0.0}; 4];
    for i in 0..4 {
        let v_rot = quat.rotate(&v_ref[i]);
        w_pts[i] = Point { x: v_rot.x - v_rot.z * u, y: v_rot.y - v_rot.z * v };
    }
    let scale = 10.0;
    let sh_hull = get_convex_hull_4([
        Point { x: w_pts[0].x * scale, y: w_pts[0].y * scale },
        Point { x: w_pts[1].x * scale, y: w_pts[1].y * scale },
        Point { x: w_pts[2].x * scale, y: w_pts[2].y * scale },
        Point { x: w_pts[3].x * scale, y: w_pts[3].y * scale },
    ]);
    let mut low = 0.0;
    let mut high = 20.0;
    for _ in 0..30 {
        let mid = (low + high) / 2.0;
        let delta = mid / 2.0;
        let test_pts = [
            Point { x: delta, y: delta }, Point { x: -delta, y: delta },
            Point { x: -delta, y: -delta }, Point { x: delta, y: -delta },
        ];
        let mut ok = true;
        for tp in &test_pts { if !is_point_in_convex_poly(tp, &sh_hull) { ok = false; break; } }
        if ok { low = mid; } else { high = mid; }
    }
    let g = (low * 0.98).max(0.1);
    let mut min_x = poly[0].x; let mut max_x = poly[0].x;
    let mut min_y = poly[0].y; let mut max_y = poly[0].y;
    for p in &poly {
        if p.x < min_x { min_x = p.x; } if p.x > max_x { max_x = p.x; }
        if p.y < min_y { min_y = p.y; } if p.y > max_y { max_y = p.y; }
    }
    let nx = ((max_x - min_x) / g).ceil() as i32 + 1;
    let ny = ((max_y - min_y) / g).ceil() as i32 + 1;
    let mut grid_to_cover = Vec::new();
    for i in -1..nx+1 {
        for j in -1..ny+1 {
            let x_start = min_x + i as f64 * g;
            let y_start = min_y + j as f64 * g;
            let sq_min = Point { x: x_start, y: y_start };
            let sq_max = Point { x: x_start + g, y: y_start + g };
            if square_intersects_polygon(&sq_min, &sq_max, &poly) {
                grid_to_cover.push((i, j));
            }
        }
    }
    println!("{}", grid_to_cover.len());
    for (i, j) in grid_to_cover {
        let x_start = min_x + i as f64 * g;
        let y_start = min_y + j as f64 * g;
        let xc = x_start + g/2.0;
        let yc = y_start + g/2.0;
        let zt = 30.0 * (i.rem_euclid(4) + 4 * j.rem_euclid(4)) as f64 + 25.0;
        let xt = xc + zt * u;
        let yt = yc + zt * v;
        println!("{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}", xt, yt, zt, quat.w, quat.x, quat.y, quat.z, scale);
    }
}
