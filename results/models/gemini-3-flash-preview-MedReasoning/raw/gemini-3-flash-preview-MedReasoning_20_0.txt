use std::io::{self, BufRead, BufWriter, Write};
use std::collections::{HashMap, BinaryHeap};
use std::cmp::Ordering;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
struct GridKey {
    ix: i32,
    iy: i32,
    iz: i32,
}

struct Suspect {
    hole_idx: u32,
    sample_idx: u32,
    prop_idx: u32,
    confidence: f32,
}

impl PartialEq for Suspect {
    fn eq(&self, other: &Self) -> bool {
        self.confidence == other.confidence
    }
}

impl Eq for Suspect {}

impl Ord for Suspect {
    fn cmp(&self, other: &Self) -> Ordering {
        // Min-heap: the smallest confidence value should be at the top of the BinaryHeap
        // so that we can pop it when we find a larger one.
        other.confidence.partial_cmp(&self.confidence).unwrap_or(Ordering::Equal)
    }
}

impl PartialOrd for Suspect {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn next_token(reader: &mut impl BufRead, token: &mut Vec<u8>) -> bool {
    token.clear();
    let mut found = false;
    loop {
        let (consumed, done, f) = {
            let buf = match reader.fill_buf() {
                Ok(b) if b.is_empty() => return found,
                Ok(b) => b,
                Err(_) => return found,
            };
            let mut c = 0;
            let mut d = false;
            let mut local_found = found;
            for &b in buf {
                c += 1;
                if b <= b' ' {
                    if local_found {
                        d = true;
                        break;
                    }
                } else {
                    token.push(b);
                    local_found = true;
                }
            }
            (c, d, local_found)
        };
        found = f;
        reader.consume(consumed);
        if done || consumed == 0 { break; }
    }
    found
}

fn parse_f64(token: &[u8]) -> f64 {
    std::str::from_utf8(token).ok().and_then(|s| s.parse().ok()).unwrap_or(0.0)
}

fn parse_f32(token: &[u8]) -> f32 {
    std::str::from_utf8(token).ok().and_then(|s| s.parse().ok()).unwrap_or(0.0)
}

fn parse_usize(token: &[u8]) -> usize {
    std::str::from_utf8(token).ok().and_then(|s| s.parse().ok()).unwrap_or(0)
}

fn read_f64(reader: &mut impl BufRead, token: &mut Vec<u8>) -> f64 {
    if next_token(reader, token) { parse_f64(token) } else { 0.0 }
}

fn read_f32(reader: &mut impl BufRead, token: &mut Vec<u8>) -> f32 {
    if next_token(reader, token) { parse_f32(token) } else { 0.0 }
}

fn read_usize(reader: &mut impl BufRead, token: &mut Vec<u8>) -> usize {
    if next_token(reader, token) { parse_usize(token) } else { 0 }
}

fn solve() -> io::Result<()> {
    let stdin = io::stdin();
    // Using a large buffer for efficient I/O
    let mut reader = io::BufReader::with_capacity(1024 * 1024, stdin.lock());
    let mut token = Vec::with_capacity(64);
    
    // Read header
    if !next_token(&mut reader, &mut token) { return Ok(()); }
    let num_holes = parse_usize(&token);
    
    if !next_token(&mut reader, &mut token) { return Ok(()); }
    let num_properties = parse_usize(&token);
    
    if num_holes == 0 || num_properties == 0 {
        println!("0");
        return Ok(());
    }

    // Read property names
    let mut property_names = Vec::with_capacity(num_properties);
    for _ in 0..num_properties {
        if next_token(&mut reader, &mut token) {
            property_names.push(String::from_utf8_lossy(&token).into_owned());
        }
    }
    
    let mut hole_ids = Vec::with_capacity(num_holes);
    let mut samples_pos = Vec::new();
    let mut samples_props = Vec::new();
    let mut samples_info = Vec::new(); // (hole_idx, sample_idx)
    
    // Read hole data
    for h_idx in 0..num_holes {
        if !next_token(&mut reader, &mut token) { break; }
        hole_ids.push(String::from_utf8_lossy(&token).into_owned());
        
        let sx = read_f64(&mut reader, &mut token);
        let sy = read_f64(&mut reader, &mut token);
        let sz = read_f64(&mut reader, &mut token);
        let dx = read_f64(&mut reader, &mut token);
        let dy = read_f64(&mut reader, &mut token);
        let dz = read_f64(&mut reader, &mut token);
        let _length = read_f64(&mut reader, &mut token);
        let n_samples = read_usize(&mut reader, &mut token);
        
        let mag = (dx*dx + dy*dy + dz*dz).sqrt();
        let (ux, uy, uz) = if mag > 1e-9 { (dx/mag, dy/mag, dz/mag) } else { (0.0, 0.0, 0.0) };
        
        for s_idx in 0..n_samples {
            let depth = read_f64(&mut reader, &mut token);
            samples_pos.push((sx + ux * depth) as f32);
            samples_pos.push((sy + uy * depth) as f32);
            samples_pos.push((sz + uz * depth) as f32);
            
            samples_info.push((h_idx as u32, s_idx as u32));
            
            for _ in 0..num_properties {
                samples_props.push(read_f32(&mut reader, &mut token));
            }
        }
    }
    
    let n_total = samples_info.len();
    if n_total == 0 {
        println!("0");
        return Ok(());
    }
    
    // Calculate Bounding Box
    let mut min_p = [f32::MAX; 3];
    let mut max_p = [f32::MIN; 3];
    for i in 0..n_total {
        for j in 0..3 {
            let p = samples_pos[i * 3 + j];
            if p < min_p[j] { min_p[j] = p; }
            if p > max_p[j] { max_p[j] = p; }
        }
    }
    
    // Determine Adaptive Grid Size
    let vol = ((max_p[0]-min_p[0]) * (max_p[1]-min_p[1]) * (max_p[2]-min_p[2])).abs();
    // Target ~50 samples per cell for efficient local aggregation
    let mut cell_size = (vol / (n_total as f32 / 50.0 + 1.0)).powf(1.0/3.0);
    if cell_size < 0.1 { cell_size = 0.1; }
    
    // Assign samples to grid cells
    let mut sample_grid = Vec::with_capacity(n_total);
    for i in 0..n_total {
        let ix = ((samples_pos[i*3+0] - min_p[0]) / cell_size).floor() as i32;
        let iy = ((samples_pos[i*3+1] - min_p[1]) / cell_size).floor() as i32;
        let iz = ((samples_pos[i*3+2] - min_p[2]) / cell_size).floor() as i32;
        sample_grid.push((GridKey { ix, iy, iz }, i as u32));
    }
    
    // Sort samples by grid cell
    sample_grid.sort_unstable_by_key(|x| x.0);
    
    // Identify unique occupied cells and their sample ranges
    let mut unique_cells = Vec::new();
    {
        let mut start = 0;
        let mut current_key = sample_grid[0].0;
        for i in 1..n_total {
            if sample_grid[i].0 != current_key {
                unique_cells.push((current_key, start, i));
                start = i;
                current_key = sample_grid[i].0;
            }
        }
        unique_cells.push((current_key, start, n_total));
    }
    
    let mut cell_lookup = HashMap::with_capacity(unique_cells.len());
    for (i, (key, _, _)) in unique_cells.iter().enumerate() {
        cell_lookup.insert(*key, i);
    }
    
    // Pre-calculate sums and squared sums for each cell
    let mut cell_sums = vec![0.0f32; unique_cells.len() * num_properties];
    let mut cell_sum_sqs = vec![0.0f32; unique_cells.len() * num_properties];
    let mut cell_counts = vec![0u32; unique_cells.len()];
    
    for (c_idx, (_, start, end)) in unique_cells.iter().enumerate() {
        cell_counts[c_idx] = (end - start) as u32;
        for i in *start..*end {
            let s_idx = sample_grid[i].1 as usize;
            for p in 0..num_properties {
                let v = samples_props[s_idx * num_properties + p];
                if v.is_finite() {
                    cell_sums[c_idx * num_properties + p] += v;
                    cell_sum_sqs[c_idx * num_properties + p] += v * v;
                }
            }
        }
    }
    
    // Calculate global epsilons for numerical stability
    let mut epsilons = vec![0.0f32; num_properties];
    for p in 0..num_properties {
        let mut g_sum = 0.0;
        let mut g_sum_sq = 0.0;
        let mut g_count = 0;
        for c in 0..unique_cells.len() {
            g_sum += cell_sums[c * num_properties + p];
            g_sum_sq += cell_sum_sqs[c * num_properties + p];
            g_count += cell_counts[c];
        }
        if g_count > 0 {
            let mean = g_sum / g_count as f32;
            let var = (g_sum_sq / g_count as f32 - mean * mean).max(0.0);
            epsilons[p] = (var.sqrt() * 1e-4).max(1e-6);
        } else {
            epsilons[p] = 1e-6;
        }
    }
    
    // Outlier Detection
    let mut suspects = BinaryHeap::new();
    let max_suspects = 100000; // Limit output size
    
    let mut n_sums = vec![0.0f32; num_properties];
    let mut n_sum_sqs = vec![0.0f32; num_properties];
    
    for (c_idx, (key, start, end)) in unique_cells.iter().enumerate() {
        n_sums.fill(0.0);
        n_sum_sqs.fill(0.0);
        let mut n_count = 0;
        
        // Sum up neighbors (27-cell neighborhood)
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    let n_key = GridKey { ix: key.ix + dx, iy: key.iy + dy, iz: key.iz + dz };
                    if let Some(&other_c_idx) = cell_lookup.get(&n_key) {
                        for p in 0..num_properties {
                            n_sums[p] += cell_sums[other_c_idx * num_properties + p];
                            n_sum_sqs[p] += cell_sum_sqs[other_c_idx * num_properties + p];
                        }
                        n_count += cell_counts[other_c_idx];
                    }
                }
            }
        }
        
        if n_count < 2 { continue; }
        let n_inv = 1.0 / n_count as f32;
        
        // Compute Z-score for each sample in the current cell
        for i in *start..*end {
            let s_idx = sample_grid[i].1 as usize;
            for p in 0..num_properties {
                let v = samples_props[s_idx * num_properties + p];
                if !v.is_finite() { continue; }
                
                let mean = n_sums[p] * n_inv;
                let var = (n_sum_sqs[p] * n_inv - mean * mean).max(0.0);
                let std_dev = var.sqrt();
                let confidence = (v - mean).abs() / (std_dev + epsilons[p]);
                
                if confidence > 4.0 { // Significant outlier threshold
                    let (h_idx, s_local) = samples_info[s_idx];
                    let suspect = Suspect {
                        hole_idx: h_idx,
                        sample_idx: s_local,
                        prop_idx: p as u32,
                        confidence,
                    };
                    if suspects.len() < max_suspects {
                        suspects.push(suspect);
                    } else if let Some(mut min_s) = suspects.peek_mut() {
                        if suspect.confidence > min_s.confidence {
                            *min_s = suspect;
                        }
                    }
                }
            }
        }
    }
    
    // Sort and output results
    let mut final_suspects: Vec<_> = suspects.into_vec();
    final_suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal));
    
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    writeln!(out, "{}", final_suspects.len())?;
    for s in final_suspects {
        writeln!(out, "{} {} {} {:.4}", 
            hole_ids[s.hole_idx as usize], 
            s.sample_idx, 
            property_names[s.prop_idx as usize], 
            s.confidence)?;
    }
    
    Ok(())
}

fn main() {
    // Silence errors to match competitive programming style, or use eprintln
    if let Err(_) = solve() {}
}
