; Version: 1.0
; Generator: Khronos
; Bound: 100

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

               ; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

 %arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
      %color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%float_0_08 = OpConstant %float 0.08
    %float_0 = OpConstant %float 0
  %float_1_3 = OpConstant %float 0.33333334

       %main = OpFunction %void None %3
          %5 = OpLabel

               ; 1. Read the 3 vertex positions
         %20 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %20
         %21 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %21
         %22 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %22

               ; 2. Compute face center
        %sum = OpFAdd %v4float %p0 %p1
       %sum2 = OpFAdd %v4float %sum %p2
     %center = OpVectorTimesScalar %v4float %sum2 %float_1_3

               ; 3. Compute face normal using cross product
        %p0_3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
        %p1_3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
        %p2_3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
         %e1 = OpFSub %v3float %p1_3 %p0_3
         %e2 = OpFSub %v3float %p2_3 %p0_3
      %cross = OpExtInst %v3float %1 Cross %e1 %e2
          %N = OpExtInst %v3float %1 Normalize %cross

               ; 4. Offset = N * 0.08
     %offset = OpVectorTimesScalar %v3float %N %float_0_08
    %offset4 = OpCompositeConstruct %v4float %offset %float_0

               ; 5. Emit 3 vertices shifted by the offset
               ; 6. Color each vertex with its original vertex color
               ; Vertex 0
         %c0 = OpAccessChain %ptr_in_v3 %color %int_0
       %clr0 = OpLoad %v3float %c0
               OpStore %outColor %clr0
        %pos0 = OpFAdd %v4float %p0 %offset4
        %og0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %og0 %pos0
               OpEmitVertex

               ; Vertex 1
         %c1 = OpAccessChain %ptr_in_v3 %color %int_1
       %clr1 = OpLoad %v3float %c1
               OpStore %outColor %clr1
        %pos1 = OpFAdd %v4float %p1 %offset4
        %og1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %og1 %pos1
               OpEmitVertex

               ; Vertex 2
         %c2 = OpAccessChain %ptr_in_v3 %color %int_2
       %clr2 = OpLoad %v3float %c2
               OpStore %outColor %clr2
        %pos2 = OpFAdd %v4float %p2 %offset4
        %og2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %og2 %pos2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
