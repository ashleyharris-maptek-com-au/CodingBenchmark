#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

using namespace std;

// Structure to represent a single task in a job
struct Task {
    int machine;
    int duration;
};

// Structure to represent a job, containing multiple tasks
struct Job {
    vector<Task> tasks;
    long long remaining_work;
};

// Structure to represent the result of a scheduled task
struct ResultTask {
    long long start_time;
    int machine;
};

typedef vector<vector<ResultTask>> Schedule;

// Priority rules for the dispatching heuristic
enum class PriorityRule {
    MWKR, // Most Work Remaining - generally effective for JSSP
    SPT,  // Shortest Processing Time - can help reduce average flow time
    FIFO  // First In First Out - simple and robust
};

/**
 * Calculates the makespan (total completion time) of a given schedule.
 */
long long calculate_makespan(const Schedule& schedule, const vector<Job>& jobs) {
    long long max_time = 0;
    for (int i = 0; i < (int)jobs.size(); ++i) {
        if (!schedule[i].empty()) {
            int last_idx = (int)schedule[i].size() - 1;
            long long finish_time = schedule[i][last_idx].start_time + jobs[i].tasks[last_idx].duration;
            if (finish_time > max_time) max_time = finish_time;
        }
    }
    return max_time;
}

/**
 * Solves the Job-Shop Scheduling problem using a Non-Delay priority-rule based heuristic.
 * Complexity: O(TotalTasks * NumActiveJobs)
 * This approach is efficient enough to handle the "Extreme" scale within typical time limits.
 */
Schedule solve(int J, int M, const vector<Job>& initial_jobs, PriorityRule rule) {
    int total_tasks = 0;
    for (int i = 0; i < J; ++i) total_tasks += (int)initial_jobs[i].tasks.size();

    vector<int> next_task_idx(J, 0);
    vector<long long> job_ready_time(J, 0);
    vector<long long> machine_ready_time(M, 0);
    vector<long long> remaining_work(J);
    for (int i = 0; i < J; ++i) remaining_work[i] = initial_jobs[i].remaining_work;

    Schedule results(J);
    for (int i = 0; i < J; ++i) results[i].resize(initial_jobs[i].tasks.size());

    // Keep track of jobs that still have tasks to be scheduled
    vector<int> active_jobs;
    active_jobs.reserve(J);
    for (int i = 0; i < J; ++i) {
        if (!initial_jobs[i].tasks.empty()) active_jobs.push_back(i);
    }

    int tasks_done = 0;
    while (tasks_done < total_tasks) {
        long long t_min = numeric_limits<long long>::max();
        
        // Phase 1: Identify the earliest possible start time among all ready tasks.
        for (int j : active_jobs) {
            int m = initial_jobs[j].tasks[next_task_idx[j]].machine;
            long long earliest_start = (job_ready_time[j] > machine_ready_time[m]) ? job_ready_time[j] : machine_ready_time[m];
            if (earliest_start < t_min) t_min = earliest_start;
        }

        // Phase 2: Identify the best task to schedule among those that can start at t_min.
        int best_j_in_active = -1;
        long long best_priority_val = numeric_limits<long long>::min();

        for (int i = 0; i < (int)active_jobs.size(); ++i) {
            int j = active_jobs[i];
            int m = initial_jobs[j].tasks[next_task_idx[j]].machine;
            long long earliest_start = (job_ready_time[j] > machine_ready_time[m]) ? job_ready_time[j] : machine_ready_time[m];

            if (earliest_start == t_min) {
                long long current_priority = 0;
                switch (rule) {
                    case PriorityRule::MWKR:
                        current_priority = remaining_work[j];
                        break;
                    case PriorityRule::SPT:
                        current_priority = -initial_jobs[j].tasks[next_task_idx[j]].duration;
                        break;
                    case PriorityRule::FIFO:
                        current_priority = -j;
                        break;
                }
                if (best_j_in_active == -1 || current_priority > best_priority_val) {
                    best_priority_val = current_priority;
                    best_j_in_active = i;
                }
            }
        }

        // Phase 3: Schedule the chosen task and update states.
        int j = active_jobs[best_j_in_active];
        int m = initial_jobs[j].tasks[next_task_idx[j]].machine;
        int dur = initial_jobs[j].tasks[next_task_idx[j]].duration;
        
        results[j][next_task_idx[j]] = {t_min, m};
        job_ready_time[j] = t_min + dur;
        machine_ready_time[m] = t_min + dur;
        remaining_work[j] -= dur;
        next_task_idx[j]++;
        tasks_done++;

        // If the job has no more tasks, remove it from active set.
        if (next_task_idx[j] == (int)initial_jobs[j].tasks.size()) {
            active_jobs[best_j_in_active] = active_jobs.back();
            active_jobs.pop_back();
        }
    }

    return results;
}

int main() {
    // Optimize I/O performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<Job> jobs(J);
    int total_tasks = 0;
    for (int i = 0; i < J; ++i) {
        int T;
        if (!(cin >> T)) break;
        jobs[i].tasks.resize(T);
        long long job_total_work = 0;
        for (int k = 0; k < T; ++k) {
            cin >> jobs[i].tasks[k].machine >> jobs[i].tasks[k].duration;
            job_total_work += jobs[i].tasks[k].duration;
        }
        jobs[i].remaining_work = job_total_work;
        total_tasks += T;
    }

    Schedule best_schedule;
    long long min_makespan = numeric_limits<long long>::max();
    PriorityRule rules[] = {PriorityRule::MWKR, PriorityRule::SPT, PriorityRule::FIFO};
    
    // Adaptive logic:
    // Scale estimation: N * J iterations in the worst case per rule.
    // 5*10^7 is a safe threshold for multiple rules on modern hardware within typical time limits.
    long long complexity_estimate = (long long)total_tasks * J;
    int num_rules_to_try = (complexity_estimate < 50000000) ? 3 : 1;

    for (int r = 0; r < num_rules_to_try; ++r) {
        Schedule s = solve(J, M, jobs, rules[r]);
        long long makespan = calculate_makespan(s, jobs);
        if (makespan < min_makespan) {
            min_makespan = makespan;
            best_schedule = std::move(s);
        }
    }

    // Output results in the specified format: start_time machine ... for each task in the job
    for (int i = 0; i < J; ++i) {
        for (size_t k = 0; k < best_schedule[i].size(); ++k) {
            cout << best_schedule[i][k].start_time << " " << best_schedule[i][k].machine << (k == best_schedule[i].size() - 1 ? "" : " ");
        }
        cout << "\n";
    }

    return 0;
}
