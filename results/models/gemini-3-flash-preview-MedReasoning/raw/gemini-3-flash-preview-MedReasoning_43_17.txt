struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Procedural brick pattern at 8x16 UV scale
    float2 uvScale = float2(8.0, 16.0);
    float2 scaledUV = input.uv * uvScale;
    
    // Staggered rows: half-brick offset every other row
    if (fmod(floor(scaledUV.y), 2.0) >= 1.0) {
        scaledUV.x += 0.5;
    }
    
    float2 brickUV = frac(scaledUV);
    
    // Mortar thickness: 0.05 in X, 0.1 in Y (relative to brick cell)
    bool isMortar = (brickUV.x < 0.05) || (brickUV.y < 0.1);
    
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 patternColor = isMortar ? mortarColor : brickColor;
    
    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;
    
    return float4(patternColor * shadingFactor, 1.0);
}
