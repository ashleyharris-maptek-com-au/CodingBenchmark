use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let width: i32 = parts.next().unwrap().parse().unwrap();
    let height: i32 = parts.next().unwrap().parse().unwrap();
    let obstacle_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle: bool = parts.next().unwrap().parse::<i32>().unwrap() == 1;

    let mut grid = vec![vec![false; width as usize]; height as usize];
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let x: i32 = parts.next().unwrap().parse().unwrap();
        let y: i32 = parts.next().unwrap().parse().unwrap();
        if x >= 0 && x < width && y >= 0 && y < height {
            grid[y as usize][x as usize] = true;
        }
    }

    if grid[0][0] {
        println!("0");
        return;
    }

    let mut non_obstacle_cells = 0;
    for r in 0..height {
        for c in 0..width {
            if !grid[r as usize][c as usize] {
                non_obstacle_cells += 1;
            }
        }
    }
    
    // Connectivity check
    let mut reachable_cells = 0;
    let mut q = std::collections::VecDeque::new();
    let mut visited_conn = vec![vec![false; width as usize]; height as usize];
    
    q.push_back((0,0));
    visited_conn[0][0] = true;
    reachable_cells += 1;

    while let Some((y, x)) = q.pop_front() {
        let moves = [(0, 1), (0, -1), (1, 0), (-1, 0)];
        for (dy, dx) in moves.iter() {
            let ny = y + dy;
            let nx = x + dx;

            if ny >= 0 && ny < height && nx >= 0 && nx < width &&
               !grid[ny as usize][nx as usize] && !visited_conn[ny as usize][nx as usize] {
                visited_conn[ny as usize][nx as usize] = true;
                reachable_cells += 1;
                q.push_back((ny, nx));
            }
        }
    }

    if reachable_cells != non_obstacle_cells {
        println!("0");
        return;
    }


    let mut path = Vec::new();
    let mut visited = vec![vec![false; width as usize]; height as usize];

    if solve(0, 0, &mut path, &mut visited, width, height, &grid, non_obstacle_cells, require_cycle) {
        println!("{}", path.len());
        for (y, x) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn get_neighbors(y: i32, x: i32, width: i32, height: i32, grid: &Vec<Vec<bool>>, visited: &Vec<Vec<bool>>) -> Vec<(i32, i32)> {
    let mut neighbors = Vec::new();
    let moves = [(0, 1), (0, -1), (1, 0), (-1, 0)];
    for (dy, dx) in moves.iter() {
        let ny = y + dy;
        let nx = x + dx;
        if ny >= 0 && ny < height && nx >= 0 && nx < width && !grid[ny as usize][nx as usize] && !visited[ny as usize][nx as usize] {
            neighbors.push((ny, nx));
        }
    }
    neighbors
}

fn solve(y: i32, x: i32, path: &mut Vec<(i32, i32)>, visited: &mut Vec<Vec<bool>>, width: i32, height: i32, grid: &Vec<Vec<bool>>, non_obstacle_cells: i32, require_cycle: bool) -> bool {
    visited[y as usize][x as usize] = true;
    path.push((y, x));

    if path.len() as i32 == non_obstacle_cells {
        if require_cycle {
            let (last_y, last_x) = path.last().unwrap();
            if (*last_y - 0).abs() + (*last_x - 0).abs() == 1 {
                return true;
            } else {
                path.pop();
                visited[y as usize][x as usize] = false;
                return false;
            }
        }
        return true;
    }

    let mut neighbors = get_neighbors(y, x, width, height, grid, visited);
    
    // Warnsdorff's rule heuristic
    neighbors.sort_by_key(|(ny, nx)| {
        get_neighbors(*ny, *nx, width, height, grid, visited).len()
    });

    for (ny, nx) in neighbors {
        if solve(ny, nx, path, visited, width, height, grid, non_obstacle_cells, require_cycle) {
            return true;
        }
    }

    path.pop();
    visited[y as usize][x as usize] = false;
    false
}
