use std::io::{self, BufRead};
use rand::Rng;

const WIDTH: f32 = 1000.0;
const HEIGHT: f32 = 1000.0;

#[derive(Clone, Debug)]
struct Node {
    x: f32,
    y: f32,
    dx: f32,
    dy: f32,
}

impl Node {
    fn new() -> Self {
        let mut rng = rand::thread_rng();
        Node {
            x: rng.gen::<f32>() * WIDTH,
            y: rng.gen::<f32>() * HEIGHT,
            dx: 0.0,
            dy: 0.0,
        }
    }
}

struct Edge {
    u: usize,
    v: usize,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut nodes = (0..n).map(|_| Node::new()).collect::<Vec<_>>();
    let mut edges = Vec::with_capacity(m);

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        if u < n && v < n { // Basic validation
            edges.push(Edge { u, v });
        }
    }

    if n > 2000 {
        layout_large(&mut nodes, &edges, n);
    } else {
        layout_small_medium(&mut nodes, &edges, n);
    }

    println!("{}", n);
    for node in nodes {
        println!("{} {}", node.x, node.y);
    }
}

fn layout_small_medium(nodes: &mut Vec<Node>, edges: &Vec<Edge>, n: usize) {
    let area = WIDTH * HEIGHT;
    let k = (area / n as f32).sqrt();
    let mut temp = WIDTH / 10.0;
    let iterations = 100;

    for _ in 0..iterations {
        // Repulsive forces
        for i in 0..n {
            nodes[i].dx = 0.0;
            nodes[i].dy = 0.0;
            for j in 0..n {
                if i == j { continue; }
                let dx = nodes[i].x - nodes[j].x;
                let dy = nodes[i].y - nodes[j].y;
                let dist_sq = dx * dx + dy * dy;
                if dist_sq > 0.0 {
                    let dist = dist_sq.sqrt();
                    let repulsive_force = k * k / dist;
                    nodes[i].dx += dx / dist * repulsive_force;
                    nodes[i].dy += dy / dist * repulsive_force;
                }
            }
        }

        // Attractive forces
        for edge in edges {
            let dx = nodes[edge.v].x - nodes[edge.u].x;
            let dy = nodes[edge.v].y - nodes[edge.u].y;
            let dist = (dx * dx + dy * dy).sqrt();
            let attractive_force = dist * dist / k;

            let force_x = dx / dist * attractive_force;
            let force_y = dy / dist * attractive_force;

            nodes[edge.u].dx += force_x;
            nodes[edge.u].dy += force_y;
            nodes[edge.v].dx -= force_x;
            nodes[edge.v].dy -= force_y;
        }

        // Update positions
        for i in 0..n {
            let disp_mag = (nodes[i].dx * nodes[i].dx + nodes[i].dy * nodes[i].dy).sqrt();
            if disp_mag > 0.0 {
                let scale = (temp / disp_mag).min(1.0);
                nodes[i].x += nodes[i].dx * scale;
                nodes[i].y += nodes[i].dy * scale;
            }

            nodes[i].x = nodes[i].x.min(WIDTH).max(0.0);
            nodes[i].y = nodes[i].y.min(HEIGHT).max(0.0);
        }

        // Cool down
        temp *= 0.95;
    }
}


fn layout_large(nodes: &mut Vec<Node>, edges: &Vec<Edge>, n: usize) {
    let k = (WIDTH * HEIGHT / n as f32).sqrt();
    let iterations = 50; 
    let spring_const = 0.001;
    let repulsion_const = 1.0;

    for _ in 0..iterations {
        // Simplified forces
        for edge in edges {
            let dx = nodes[edge.v].x - nodes[edge.u].x;
            let dy = nodes[edge.v].y - nodes[edge.u].y;
            let dist_sq = dx*dx + dy*dy;

            if dist_sq > 0.0 {
                let dist = dist_sq.sqrt();
                let force = spring_const * (dist - k);
                let fx = force * dx / dist;
                let fy = force * dy / dist;

                nodes[edge.u].dx += fx;
                nodes[edge.u].dy += fy;
                nodes[edge.v].dx -= fx;
                nodes[edge.v].dy -= fy;
            }
        }
        
        // Simplified repulsion (grid-based approximation would be better, but is more complex)
        // For this solution, we do a limited random sampling of repulsion
        let mut rng = rand::thread_rng();
        for i in 0..n {
            for _ in 0..5 { // Check against 5 random nodes
                let j = rng.gen_range(0..n);
                if i == j { continue; }
                 let dx = nodes[i].x - nodes[j].x;
                let dy = nodes[i].y - nodes[j].y;
                let dist_sq = dx * dx + dy * dy;
                if dist_sq > 1.0 { // Avoid division by zero
                    let dist = dist_sq.sqrt();
                    let force = repulsion_const / dist_sq;
                    nodes[i].dx += force * dx / dist;
                    nodes[i].dy += force * dy / dist;
                }
            }
        }

        // Update positions
        for i in 0..n {
            nodes[i].x += nodes[i].dx;
            nodes[i].y += nodes[i].dy;
            nodes[i].x = nodes[i].x.min(WIDTH).max(0.0);
            nodes[i].y = nodes[i].y.min(HEIGHT).max(0.0);
            
            // Dampening
            nodes[i].dx *= 0.8;
            nodes[i].dy *= 0.8;
        }
    }
}
