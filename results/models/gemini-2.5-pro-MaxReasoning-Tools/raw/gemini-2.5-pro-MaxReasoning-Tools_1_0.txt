#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <limits>

// Structure to represent a city's coordinates
struct City {
    double x, y;
};

// Function to calculate the squared Euclidean distance between two cities
double distanceSquared(const City& c1, const City& c2) {
    double dx = c1.x - c2.x;
    double dy = c1.y - c2.y;
    return dx * dx + dy * dy;
}

// Greedy algorithm (Nearest Neighbor)
std::vector<int> nearestNeighbor(int n, const std::vector<City>& cities) {
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current_city_idx = 0;

    for (int i = 1; i < n; ++i) {
        int nearest_city_idx = -1;
        double min_dist_sq = std::numeric_limits<double>::max();

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d_sq = distanceSquared(cities[current_city_idx], cities[j]);
                if (d_sq < min_dist_sq) {
                    min_dist_sq = d_sq;
                    nearest_city_idx = j;
                }
            }
        }
        tour.push_back(nearest_city_idx);
        visited[nearest_city_idx] = true;
        current_city_idx = nearest_city_idx;
    }

    return tour;
}

// Function to calculate the total distance of a tour
double totalDistance(const std::vector<int>& tour, const std::vector<City>& cities) {
    double total_dist = 0;
    for (size_t i = 0; i < tour.size() - 1; ++i) {
        total_dist += std::sqrt(distanceSquared(cities[tour[i]], cities[tour[i + 1]]));
    }
    // Add distance from last city back to the first
    total_dist += std::sqrt(distanceSquared(cities[tour.back()], cities[tour.front()]));
    return total_dist;
}

// 2-opt optimization
void twoOpt(std::vector<int>& tour, const std::vector<City>& cities) {
    bool improved = true;
    int n = tour.size();
    while (improved) {
        improved = false;
        for (int i = 1; i < n - 2; ++i) {
            for (int j = i + 1; j < n -1; ++j) {
                // Current edges: (i-1, i) and (j, j+1)
                // New edges: (i-1, j) and (i, j+1)
                
                int A = tour[i - 1];
                int B = tour[i];
                int C = tour[j];
                int D = tour[j + 1];

                double current_dist_sq = distanceSquared(cities[A], cities[B]) + distanceSquared(cities[C], cities[D]);
                double new_dist_sq = distanceSquared(cities[A], cities[C]) + distanceSquared(cities[B], cities[D]);

                if (new_dist_sq < current_dist_sq) {
                    std::reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// For Morton curve
// Function to interleave bits
uint64_t interleave(uint32_t x, uint32_t y) {
    uint64_t z = 0;
    for (int i = 0; i < 32; ++i) {
        z |= (uint64_t)(x & 1) << (2 * i + 1);
        z |= (uint64_t)(y & 1) << (2 * i);
        x >>= 1;
        y >>= 1;
    }
    return z;
}

int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std.cin.tie(NULL);

    int n;
    std::cin >> n;

    std::vector<City> cities(n);
    double min_x = std::numeric_limits<double>::max();
    double max_x = std::numeric_limits<double>::min();
    double min_y = std::numeric_limits<double>::max();
    double max_y = std::numeric_limits<double>::min();

    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        min_x = std::min(min_x, cities[i].x);
        max_x = std::max(max_x, cities[i].x);
        min_y = std::min(min_y, cities[i].y);
        max_y = std::max(max_y, cities[i].y);
    }

    std::vector<int> tour;

    if (n <= 200) {
        tour = nearestNeighbor(n, cities);
        if (n > 2) {
             twoOpt(tour, cities);
        }
    } else {
        // Morton curve (Z-order) sorting
        std::vector<int> p(n);
        std::iota(p.begin(), p.end(), 0);

        double range_x = max_x - min_x;
        double range_y = max_y - min_y;

        std::sort(p.begin(), p.end(), [&](int a, int b) {
            uint32_t ax = (cities[a].x - min_x) / range_x * (std::numeric_limits<uint32_t>::max() -1);
            uint32_t ay = (cities[a].y - min_y) / range_y * (std::numeric_limits<uint32_t>::max() -1);
            uint64_t a_morton = interleave(ax, ay);

            uint32_t bx = (cities[b].x - min_x) / range_x * (std::numeric_limits<uint32_t>::max()-1);
            uint32_t by = (cities[b].y - min_y) / range_y * (std::numeric_limits<uint32_t>::max()-1);
            uint64_t b_morton = interleave(bx, by);

            return a_morton < b_morton;
        });

        auto it_zero = std::find(p.begin(), p.end(), 0);
        std::rotate(p.begin(), it_zero, it_zero + 1);
        
        tour = p;
    }


    for (int i = 0; i < n; ++i) {
        std::cout << tour[i] << (i == n - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
