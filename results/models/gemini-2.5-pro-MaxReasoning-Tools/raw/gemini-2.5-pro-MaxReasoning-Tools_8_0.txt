use std::collections::{BinaryHeap, HashMap};
use std::io::{self, BufRead};
use std::cmp::Ordering;

#[derive(Clone, Eq, PartialEq, Hash, Debug)]
struct Point {
    x: usize,
    y: usize,
}

#[derive(Clone, Eq, PartialEq)]
struct State {
    cost: usize,
    position: Point,
}

impl Ord for State {
    fn cmp(&self, other: &Self) -> Ordering {
        other.cost.cmp(&self.cost)
    }
}

impl PartialOrd for State {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn heuristic(a: &Point, b: &Point) -> usize {
    (a.x as isize - b.x as isize).abs() as usize + (a.y as isize - b.y as isize).abs() as usize
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let dimensions = lines.next().unwrap().unwrap();
    let mut parts = dimensions.split_whitespace();
    let height: usize = parts.next().unwrap().parse().unwrap();
    let width: usize = parts.next().unwrap().parse().unwrap();

    let mut maze = Vec::new();
    let mut start = Point { x: 0, y: 0 };
    let mut end = Point { x: 0, y: 0 };

    for y in 0..height {
        let row = lines.next().unwrap().unwrap();
        maze.push(row.chars().collect::<Vec<char>>());
        for x in 0..width {
            if maze[y][x] == 'A' {
                start = Point { x, y };
            } else if maze[y][x] == 'B' {
                end = Point { x, y };
            }
        }
    }

    let mut open_set = BinaryHeap::new();
    let mut came_from: HashMap<Point, Point> = HashMap::new();
    let mut g_score: HashMap<Point, usize> = HashMap::new();

    g_score.insert(start.clone(), 0);
    open_set.push(State { cost: heuristic(&start, &end), position: start.clone() });

    let mut path_found = false;

    while let Some(State { position, .. }) = open_set.pop() {
        if position == end {
            path_found = true;
            break;
        }

        for (dx, dy) in &[(0, 1), (0, -1), (1, 0), (-1, 0)] {
            let next_x = position.x as isize + dx;
            let next_y = position.y as isize + dy;

            if next_x >= 0 && next_x < width as isize && next_y >= 0 && next_y < height as isize {
                let neighbor = Point { x: next_x as usize, y: next_y as usize };
                if maze[neighbor.y][neighbor.x] == '#' {
                    continue;
                }

                let tentative_g_score = g_score.get(&position).unwrap_or(&usize::MAX) + 1;

                if tentative_g_score < *g_score.get(&neighbor).unwrap_or(&usize::MAX) {
                    came_from.insert(neighbor.clone(), position.clone());
                    g_score.insert(neighbor.clone(), tentative_g_score);
                    let f_score = tentative_g_score + heuristic(&neighbor, &end);
                    open_set.push(State { cost: f_score, position: neighbor });
                }
            }
        }
    }
    
    if path_found {
        let mut path = Vec::new();
        let mut current = end.clone();
        path.push(current.clone());
        while current != start {
            current = came_from.get(&current).unwrap().clone();
            path.push(current.clone());
        }
        path.reverse();

        println!("{}", path.len());
        for point in path {
            println!("{} {}", point.x, point.y);
        }
    } else {
        println!("0");
    }
}
