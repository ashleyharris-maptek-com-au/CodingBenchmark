// Structure to represent a city's coordinates
struct City {

// Function to calculate the squared Euclidean distance between two cities
double distanceSquared(const City& c1, const City& c2) {
    double dx = c1.x - c2.x;
    double dy = c1.y - c2.y;
    return dx * dx + dy * dy;
}

// Greedy algorithm (Nearest Neighbor)
std::vector<int> nearestNeighbor(int n, const std::vector<City>& cities) {
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current_city_idx = 0;

    for (int i = 1; i < n; ++i) {
        int nearest_city_idx = -1;
        double min_dist_sq = std::numeric_limits<double>::max();

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d_sq = distanceSquared(cities[current_city_idx], cities[j]);
                if (d_sq < min_dist_sq) {
                    min_dist_sq = d_sq;
                    nearest_city_idx = j;
                }
            }
        }
        tour.push_back(nearest_city_idx);
        visited[nearest_city_idx] = true;
        current_city_idx = nearest_city_idx;
    }

    return tour;
}

// Function to calculate the total distance of a tour
double totalDistance(const std::vector<int>& tour, const std::vector<City>& cities) {
    double total_dist = 0;
    for (size_t i = 0; i < tour.size() - 1; ++i) {
        total_dist += std::sqrt(distanceSquared(cities[tour[i]], cities[tour[i + 1]]));
    }
    // Add distance from last city back to the first
    total_dist += std::sqrt(distanceSquared(cities[tour.back()], cities[tour.front()]));
    return total_dist;
}

// 2-opt optimization
void twoOpt(std::vector<int>& tour, const std::vector<City>& cities) {
    bool improved = true;
    int n = tour.size();
    while (improved) {
        improved = false;
        for (int i = 1; i < n - 2; ++i) {
            for (int j = i + 1; j < n -1; ++j) {
                // Current edges: (i-1, i) and (j, j+1)
                // New edges: (i-1, j) and (i, j+1)
                
                int A = tour[i - 1];
                int B = tour[i];
                int C = tour[j];
                int D = tour[j + 1];

                double current_dist_sq = distanceSquared(cities[A], cities[B]) + distanceSquared(cities[C], cities[D]);
                double new_dist_sq = distanceSquared(cities[A], cities[C]) + distanceSquared(cities[B], cities[D]);

                if (new_dist_sq < current_dist_sq) {
                    std::reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Function to interleave bits
uint64_t interleave(uint32_t x, uint32_t y) {
    uint64_t z = 0;
    for (int i = 0; i < 32; ++i) {
        z |= (uint64_t)(x & 1) << (2 * i + 1);

#[derive(Clone, Eq, PartialEq, Hash, Debug)]
struct Point {
    x: usize,
    y: usize,
}

#[derive(Clone, Eq, PartialEq)]
struct State {
    cost: usize,
    position: Point,
}

impl Ord for State {
    fn cmp(&self, other: &Self) -> Ordering {
        other.cost.cmp(&self.cost)
    }
}

impl PartialOrd for State {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn heuristic(a: &Point, b: &Point) -> usize {
    (a.x as isize - b.x as isize).abs() as usize + (a.y as isize - b.y as isize).abs() as usize
}

My solution implements the First-Fit-Decreasing (FFD) algorithm to solve the 1D Cutting Stock Problem. This algorithm is a widely-used heuristic that provides a good approximation for this NP-hard problem, and is generally more effective at minimizing the number of stock pieces ('bins') than a simple First-Fit approach.

The core logic is as follows:
1.  **Sorting:** The cuts are first sorted in descending order of their length. This is the 'Decreasing' part of the algorithm. It prioritizes placing the largest cuts first, which tends to leave more usable space for smaller cuts later.
2.  **Optimized Placement:** For each cut, instead of linearly scanning through all existing stock pieces to find one with enough capacity, I've implemented a more efficient method. I use a `SortedList` data structure to maintain records of the available stock pieces, bucketed by their remaining capacity.
3.  **Efficient Search:** The `SortedList` keeps the capacity values sorted. I can then use a binary search on the keys of this `SortedList` to find the smallest capacity that is greater than or equal to the length of the current cut. This reduces the search time for a suitable stock from O(N) to O(log K), where K is the number of distinct capacity values (K <= N). This optimization is crucial for handling the large number of cuts mentioned in the problem description.

This approach results in an overall time complexity of approximately O(N log N) (dominated by the initial sort), which is efficient enough to handle millions of cuts within the given time constraints. The solution is contained within a single `Program.cs` file as requested.
