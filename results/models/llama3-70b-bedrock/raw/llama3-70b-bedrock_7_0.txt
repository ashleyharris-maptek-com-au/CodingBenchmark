```cpp
#include <iostream>
#include <vector>
#include <CGAL/Exact_predicates_inexact_constructions_kernel.h>
#include <CGAL/Polyhedron_3.h>
#include <CGAL/boost/graph/graph_traits_Polyhedron_3.h>
#include <CGAL/boost/graph/properties_Polyhedron_3.h>
#include <CGAL/Polyhedron_incremental_builder_3.h>

typedef CGAL::Exact_predicates_inexact_constructions_kernel Kernel;
typedef CGAL::Polyhedron_3<Kernel> Polyhedron;
typedef Polyhedron::Vertex_iterator Vertex_iterator;
typedef Polyhedron::Face_iterator Face_iterator;

void read_mesh(Polyhedron& mesh, int num_vertices, int num_faces) {
    mesh.clear();
    meshreserve(num_vertices, num_faces);

    for (int i = 0; i < num_vertices; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        mesh.make_vertex(Kernel::Point_3(x, y, z));
    }

    for (int i = 0; i < num_faces; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        mesh.make_face(mesh.vertex(a), mesh.vertex(b), mesh.vertex(c));
    }
}

void write_mesh(const Polyhedron& mesh) {
    std::cout << mesh.size_of_vertices() << " " << mesh.size_of_facets() << std::endl;

    for (Vertex_iterator vi = mesh.vertices_begin(); vi != mesh.vertices_end(); ++vi) {
        std::cout << vi->point() << std::endl;
    }

    for (Face_iterator fi = mesh.facets_begin(); fi != mesh.facets_end(); ++fi) {
        std::cout << fi->halfedge()->vertex()->index() << " ";
        std::cout << fi->halfedge()->next()->vertex()->index() << " ";
        std::cout << fi->halfedge()->next()->next()->vertex()->index() << std::endl;
    }
}

int main() {
    int va, fa, vb, fb;
    std::cin >> va >> fa >> vb >> fb;

    Polyhedron mesh_a, mesh_b, mesh_union;

    read_mesh(mesh_a, va, fa);
    read_mesh(mesh_b, vb, fb);

    mesh_union = mesh_a;
    mesh_union += mesh_b;

    write_mesh(mesh_union);

    return 0;
}
```

This code reads two meshes from the standard input, computes their union using the `Polyhedron` class from CGAL, and writes the resulting mesh to the standard output.

The `read_mesh` function reads a mesh from the standard input and constructs a `Polyhedron` object from it. The `write_mesh` function writes a mesh to the standard output.

The `main` function reads the input, computes the union of the two meshes, and writes the result.

Note that this code uses the `Exact_predicates_inexact_constructions_kernel` kernel, which provides a good trade-off between accuracy and performance. You can change the kernel to `Exact_predicates_exact_constructions_kernel` for exact constructions, but this may be slower for large meshes.

Also, make sure to install the CGAL library and link against it when compiling the code.
