using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;

namespace MiningProblem
{
    class Program
    {
        static void Main(string[] args)
        {
            // Read input header
            string[] header = Console.ReadLine().Split(' ');
            int xVoxelCount = int.Parse(header[0]);
            int yVoxelCount = int.Parse(header[1]);
            int zVoxelCount = int.Parse(header[2]);
            float xBlockSize = float.Parse(header[3]);
            float yBlockSize = float.Parse(header[4]);
            float zBlockSize = float.Parse(header[5]);

            // Initialize voxel grid
            byte[,,] voxels = new byte[xVoxelCount, yVoxelCount, zVoxelCount];

            // Read voxel data
            for (int x = 0; x < xVoxelCount; x++)
            {
                for (int y = 0; y < yVoxelCount; y++)
                {
                    for (int z = 0; z < zVoxelCount; z++)
                    {
                        byte d = (byte)Console.Read();
                        voxels[x, y, z] = d;
                    }
                }
            }

            // Choose strategy based on problem scale
            Strategy strategy;
            if (xVoxelCount * yVoxelCount * zVoxelCount < 1000000)
            {
                strategy = new BruteForceStrategy(voxels, xBlockSize, yBlockSize, zBlockSize);
            }
            else
            {
                strategy = new HeuristicStrategy(voxels, xBlockSize, yBlockSize, zBlockSize);
            }

            // Solve the problem
            List<Operation> operations = strategy.Solve();

            // Output solution
            Console.WriteLine(operations.Count);
            foreach (Operation op in operations)
            {
                Console.WriteLine($"{op.N} {op.DigX} {op.DigY} {op.DigZ} {string.Join(" ", op.Steps)} {op.DestX} {op.DestY}");
            }
        }
    }

    class Operation
    {
        public int N { get; set; }
        public int DigX { get; set; }
        public int DigY { get; set; }
        public int DigZ { get; set; }
        public List<(int, int)> Steps { get; set; }
        public int DestX { get; set; }
        public int DestY { get; set; }
    }

    abstract class Strategy
    {
        protected byte[,,] voxels;
        protected float xBlockSize;
        protected float yBlockSize;
        protected float zBlockSize;

        public Strategy(byte[,,] voxels, float xBlockSize, float yBlockSize, float zBlockSize)
        {
            this.voxels = voxels;
            this.xBlockSize = xBlockSize;
            this.yBlockSize = yBlockSize;
            this.zBlockSize = zBlockSize;
        }

        public abstract List<Operation> Solve();
    }

    class BruteForceStrategy : Strategy
    {
        public BruteForceStrategy(byte[,,] voxels, float xBlockSize, float yBlockSize, float zBlockSize)
            : base(voxels, xBlockSize, yBlockSize, zBlockSize)
        {
        }

        public override List<Operation> Solve()
        {
            // Brute force strategy: try all possible operations and choose the best one
            List<Operation> operations = new List<Operation>();
            for (int x = 0; x < voxels.GetLength(0); x++)
            {
                for (int y = 0; y < voxels.GetLength(1); y++)
                {
                    for (int z = 0; z < voxels.GetLength(2); z++)
                    {
                        byte voxel = voxels[x, y, z];
                        if (voxel != 0)
                        {
                            int mass = voxel % 16;
                            int value = voxel / 16;
                            if (value > 0)
                            {
                                // Try to dig and dump the voxel
                                Operation op = new Operation();
                                op.N = 1;
                                op.DigX = x;
                                op.DigY = y;
                                op.DigZ = z;
                                op.Steps = new List<(int, int)>();
                                int destX = 0;
                                int destY = 0;
                                int dx = x - destX;
                                int dy = y - destY;
                                while (dx != 0 || dy != 0)
                                {
                                    int stepX = dx > 0 ? -1 : 1;
                                    int stepY = dy > 0 ? -1 : 1;
                                    op.Steps.Add((stepX, stepY));
                                    dx += stepX;
                                    dy += stepY;
                                }
                                op.DestX = destX;
                                op.DestY = destY;
                                operations.Add(op);
                            }
                        }
                    }
                }
            }
            return operations;
        }
    }

    class HeuristicStrategy : Strategy
    {
        public HeuristicStrategy(byte[,,] voxels, float xBlockSize, float yBlockSize, float zBlockSize)
            : base(voxels, xBlockSize, yBlockSize, zBlockSize)
        {
        }

        public override List<Operation> Solve()
        {
            // Heuristic strategy: use a priority queue to focus on the most valuable voxels
            List<Operation> operations = new List<Operation>();
            PriorityQueue<Voxel> queue = new PriorityQueue<Voxel>();
            for (int x = 0; x < voxels.GetLength(0); x++)
            {
                for (int y = 0; y < voxels.GetLength(1); y++)
                {
                    for (int z = 0; z < voxels.GetLength(2); z++)
                    {
                        byte voxel = voxels[x, y, z];
                        if (voxel != 0)
                        {
                            int mass = voxel % 16;
                            int value = voxel / 16;
                            if (value > 0)
                            {
                                Voxel v = new Voxel(x, y, z, mass, value);
                                queue.Enqueue(v);
                            }
                        }
                    }
                }
            }
            while (queue.Count > 0)
            {
                Voxel v = queue.Dequeue();
                Operation op = new Operation();
                op.N = 1;
                op.DigX = v.X;
                op.DigY = v.Y;
                op.DigZ = v.Z;
                op.Steps = new List<(int, int)>();
                int destX = 0;
                int destY = 0;
                int dx = v.X - destX;
                int dy = v.Y - destY;
                while (dx != 0 || dy != 0)
                {
                    int stepX = dx > 0 ? -1 : 1;
                    int stepY = dy > 0 ? -1 : 1;
                    op.Steps.Add((stepX, stepY));
                    dx += stepX;
                    dy += stepY;
                }
                op.DestX = destX;
                op.DestY = destY;
                operations.Add(op);
            }
            return operations;
        }
    }

    class Voxel : IComparable<Voxel>
    {
        public int X { get; set; }
        public int Y { get; set; }
        public int Z { get; set; }
        public int Mass { get; set; }
        public int Value { get; set; }

        public Voxel(int x, int y, int z, int mass, int value)
        {
            X = x;
            Y = y;
            Z = z;
            Mass = mass;
            Value = value;
        }

        public int CompareTo(Voxel other)
        {
            return other.Value.CompareTo(Value);
        }
    }

    class PriorityQueue<T> where T : IComparable<T>
    {
        private List<T> list = new List<T>();

        public void Enqueue(T item)
        {
            list.Add(item);
            list.Sort();
        }

        public T Dequeue()
        {
            T item = list[0];
            list.RemoveAt(0);
            return item;
        }

        public int Count
        {
            get { return list.Count; }
        }
    }

The `PriorityQueue<T>` class is a simple priority queue implementation that uses a list to store items and sorts the list when an item is enqueued.
