#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <vector>

using namespace std;

struct Point {
    double x, y;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x, dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFull;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFull;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0Full;
    v = (v | (v << 2))  & 0x3333333333333333ull;
    v = (v | (v << 1))  & 0x5555555555555555ull;
    return v;
}

static inline uint64_t morton(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

static vector<int> nearest_neighbor(const vector<Point>& p) {
    int n = (int)p.size();
    vector<int> tour;
    vector<unsigned char> used(n, 0);
    tour.reserve(n);
    int cur = 0;
    tour.push_back(0);
    used[0] = 1;
    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bd = numeric_limits<double>::infinity();
        for (int i = 0; i < n; ++i) {
            if (!used[i]) {
                double d = dist2(p[cur], p[i]);
                if (d < bd) bd = d, best = i;
            }
        }
        cur = best;
        used[cur] = 1;
        tour.push_back(cur);
    }
    return tour;
}

static void two_opt_limited(vector<int>& tour, const vector<Point>& p, int rounds) {
    int n = (int)tour.size();
    if (n < 4) return;
    for (int r = 0; r < rounds; ++r) {
        bool changed = false;
        for (int i = 1; i + 2 < n; ++i) {
            int a = tour[i - 1], b = tour[i];
            int limit = min(n - 1, i + 80);
            for (int k = i + 1; k < limit; ++k) {
                int c = tour[k], d = tour[(k + 1) % n];
                double oldv = sqrt(dist2(p[a], p[b])) + sqrt(dist2(p[c], p[d]));
                double newv = sqrt(dist2(p[a], p[c])) + sqrt(dist2(p[b], p[d]));
                if (newv + 1e-12 < oldv) {
                    reverse(tour.begin() + i, tour.begin() + k + 1);
                    changed = true;
                    break;
                }
            }
        }
        if (!changed) break;
    }
    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());
}

static vector<int> morton_tour(const vector<Point>& p) {
    int n = (int)p.size();
    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (const auto& q : p) {
        minx = min(minx, q.x); maxx = max(maxx, q.x);
        miny = min(miny, q.y); maxy = max(maxy, q.y);
    }
    double sx = (maxx > minx) ? (4294967295.0 / (maxx - minx)) : 0.0;
    double sy = (maxy > miny) ? (4294967295.0 / (maxy - miny)) : 0.0;
    vector<pair<uint64_t, int>> key;
    key.reserve(n);
    for (int i = 0; i < n; ++i) {
        uint32_t xi = (uint32_t)min(4294967295.0, max(0.0, (p[i].x - minx) * sx));
        uint32_t yi = (uint32_t)min(4294967295.0, max(0.0, (p[i].y - miny) * sy));
        key.push_back({morton(xi, yi), i});
    }
    sort(key.begin(), key.end(), [](const auto& a, const auto& b) {
        return a.first < b.first || (a.first == b.first && a.second < b.second);
    });
    vector<int> tour;
    tour.reserve(n);
    for (auto& e : key) tour.push_back(e.second);
    auto it = find(tour.begin(), tour.end(), 0);
    if (it != tour.end()) rotate(tour.begin(), it, tour.end());
    return tour;
}

static vector<int> grid_snake_tour(const vector<Point>& p) {
    int n = (int)p.size();
    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (const auto& q : p) {
        minx = min(minx, q.x); maxx = max(maxx, q.x);
        miny = min(miny, q.y); maxy = max(maxy, q.y);
    }
    int bins = max(1, (int)sqrt((double)n / 32.0));
    double h = (maxy > miny) ? ((maxy - miny) / bins) : 1.0;
    vector<pair<uint64_t, int>> key;
    key.reserve(n);
    for (int i = 0; i < n; ++i) {
        int row = min(bins - 1, max(0, (int)((p[i].y - miny) / h)));
        uint32_t xf = (uint32_t)(max(0.0, min(4294967295.0,
                           (p[i].x - minx) * (4294967295.0 / max(1e-300, maxx - minx)))));
        uint64_t xkey = (row & 1) ? (4294967295u - xf) : xf;
        key.push_back({((uint64_t)(uint32_t)row << 32) | xkey, i});
    }
    sort(key.begin(), key.end());
    vector<int> tour;
    tour.reserve(n);
    for (auto& e : key) tour.push_back(e.second);
    auto it = find(tour.begin(), tour.end(), 0);
    if (it != tour.end()) rotate(tour.begin(), it, tour.end());
    return tour;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;
    vector<Point> p(n);
    for (int i = 0; i < n; ++i) cin >> p[i].x >> p[i].y;

    vector<int> tour;
    if (n <= 1) {
        tour.resize(max(0, n));
        if (n == 1) tour[0] = 0;
    } else if (n <= 2500) {
        tour = nearest_neighbor(p);
        two_opt_limited(tour, p, n <= 250 ? 4 : 2);
    } else if (n <= 1000000) {
        tour = morton_tour(p);
        if (n <= 50000) two_opt_limited(tour, p, 1);
    } else {
        tour = grid_snake_tour(p);
    }

    if (!tour.empty() && tour[0] != 0) {
        auto it = find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) rotate(tour.begin(), it, tour.end());
    }

    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
