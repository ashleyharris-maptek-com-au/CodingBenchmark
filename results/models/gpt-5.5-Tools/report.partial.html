uint64_t hashState(const State& s){

State parseFen(const string& fen){
    State s; s.b.fill(0);
    string board, stm, castle, ep; stringstream ss(fen);
    ss >> board >> stm >> castle >> ep >> s.halfmove >> s.fullmove;
    int r=7,f=0;
    for(char c: board){
        if(c=='/'){ r--; f=0; }
        else if(c>='1'&&c<='8') f += c-'0';
        else if(on(f,r)) s.b[r*8+f++] = pieceFromChar(c);
    }
    s.white = stm!="b";
    s.castling=0;
    if(castle.find('K')!=string::npos) s.castling|=1;
    if(castle.find('Q')!=string::npos) s.castling|=2;
    if(castle.find('k')!=string::npos) s.castling|=4;
    if(castle.find('q')!=string::npos) s.castling|=8;
    s.ep = -1;
    if(ep.size()==2 && ep[0]!='-') s.ep = (ep[1]-'1')*8 + (ep[0]-'a');
    return s;
}

State makeMove(const State& s, const Move& m){

    }
    if(abs(p)==1 && abs(m.to-m.from)==16) n.ep=(m.from+m.to)/2;
    if(p==WK) n.castling&=~3; if(p==BK) n.castling&=~12;

    return n;
}

int quiesce(State& s, int alpha, int beta){
    if(chrono::steady_clock::now()>=deadline){ stopped=true; return alpha; }
    int stand=eval(s); if(stand>=beta) return beta; if(stand>alpha) alpha=stand;
    vector<Move> moves=legalMoves(s);
    moves.erase(remove_if(moves.begin(),moves.end(),[](const Move&m){return !(m.flags&CAPTURE) && !(m.flags&PROMO);}),moves.end());
    orderMoves(s,moves);
    for(auto& m:moves){ State n=makeMove(s,m); int sc=-quiesce(n,-beta,-alpha); if(stopped) return alpha; if(sc>=beta) return beta; if(sc>alpha) alpha=sc; }
    return alpha;
}

int search(State& s, int depth, int alpha, int beta, int ply){
    if(chrono::steady_clock::now()>=deadline){ stopped=true; return alpha; }
    if(depth<=0) return quiesce(s,alpha,beta);
    uint64_t h=hashState(s); auto it=tt.find(h);
    if(it!=tt.end() && it->second.first>=depth) return it->second.second;
    vector<Move> moves=legalMoves(s); orderMoves(s,moves);
    if(moves.empty()) return inCheck(s,s.white) ? -MATE+ply : 0;
    int best=-INF, orig=alpha;
    for(auto& m:moves){ State n=makeMove(s,m); int sc=-search(n,depth-1,-beta,-alpha,ply+1); if(stopped) return alpha; if(sc>best) best=sc; if(sc>alpha) alpha=sc; if(alpha>=beta) break; }
    if(!stopped && best>orig) tt[h]={depth,best};
    return best;
}

Move findBest(State root, int seconds){
    vector<Move> moves=legalMoves(root); if(moves.empty()) return Move();
    orderMoves(root,moves); Move best=moves[0];
    auto now=chrono::steady_clock::now();
    deadline = now + chrono::milliseconds(max(1000, seconds*1000-150));
    int maxDepth=64;
    for(int depth=1; depth<=maxDepth; depth++){
        stopped=false; int bestScore=-INF; Move iterBest=best; orderMoves(root,moves);
        for(auto& m:moves){
            State n=makeMove(root,m);
            int sc=-search(n,depth-1,-INF,INF,1);
            if(stopped) break;
            m.score=sc;
            if(sc>bestScore){ bestScore=sc; iterBest=m; }
        }
        if(stopped) break;
        best=iterBest;
        sort(moves.begin(),moves.end(),[](const Move&a,const Move&b){return a.score>b.score;});
        if(abs(bestScore)>MATE/2) break;
    }
    return best;
}
