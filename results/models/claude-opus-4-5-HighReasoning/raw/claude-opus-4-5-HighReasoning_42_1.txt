; SPIR-V Geometry Shader: Expand points to diamonds
; Assembles with: spirv-as --target-env vulkan1.0

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Input/output location decorations
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
   %void_fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
    %uint_1 = OpConstant %uint 1
  %float_0 = OpConstant %float 0.0
  %float_h = OpConstant %float 0.05
  %float_nh = OpConstant %float -0.05

; Input gl_PerVertex block (array of 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_1
%ptr_in_v4 = OpTypePointer Input %v4float
      %gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
     %gl_out = OpVariable %ptr_out_pv Output

; Input color array (location 2)
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
%ptr_in_v3 = OpTypePointer Input %v3float
    %inColor = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %void_fn
      %entry = OpLabel

; Load input position: gl_in[0].gl_Position
  %pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %pos = OpLoad %v4float %pos_ptr

; Extract x, y, z, w
        %x = OpCompositeExtract %float %pos 0
        %y = OpCompositeExtract %float %pos 1
        %z = OpCompositeExtract %float %pos 2
        %w = OpCompositeExtract %float %pos 3

; Load input color: inColor[0]
%color_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
    %color = OpLoad %v3float %color_ptr

; Calculate vertex positions
  %y_plus_h = OpFAdd %float %y %float_h
 %y_minus_h = OpFAdd %float %y %float_nh
  %x_plus_h = OpFAdd %float %x %float_h
 %x_minus_h = OpFAdd %float %x %float_nh

; Vertex 1: Top (x, y+h, z, w)
   %pos_top = OpCompositeConstruct %v4float %x %y_plus_h %z %w
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr %pos_top
               OpStore %outColor %color
               OpEmitVertex

; Vertex 2: Right (x+h, y, z, w)
 %pos_right = OpCompositeConstruct %v4float %x_plus_h %y %z %w
               OpStore %out_pos_ptr %pos_right
               OpStore %outColor %color
               OpEmitVertex

; Vertex 3: Left (x-h, y, z, w)
  %pos_left = OpCompositeConstruct %v4float %x_minus_h %y %z %w
               OpStore %out_pos_ptr %pos_left
               OpStore %outColor %color
               OpEmitVertex

; Vertex 4: Bottom (x, y-h, z, w)
%pos_bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w
               OpStore %out_pos_ptr %pos_bottom
               OpStore %outColor %color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
