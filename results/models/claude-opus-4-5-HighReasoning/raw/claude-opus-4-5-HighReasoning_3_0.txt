use std::io::{self, BufRead, Write, BufWriter};
use std::collections::{HashMap, VecDeque};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut p = line.split_whitespace();
            let u: usize = p.next().unwrap().parse().unwrap();
            let v: usize = p.next().unwrap().parse().unwrap();
            if u < n && v < n && u != v {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    let positions = layout_graph(n, &adj);

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    for i in 0..n {
        writeln!(out, "{} {}", positions[i].0, positions[i].1).unwrap();
    }
}

fn layout_graph(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    if n == 0 {
        return vec![];
    }
    if n == 1 {
        return vec![(0.0, 0.0)];
    }

    let components = find_components(n, adj);
    let num_components = *components.iter().max().unwrap_or(&0) + 1;

    if num_components == 1 {
        return layout_single_component(n, adj);
    }

    let mut comp_nodes: Vec<Vec<usize>> = vec![Vec::new(); num_components];
    for i in 0..n {
        comp_nodes[components[i]].push(i);
    }

    let mut positions = vec![(0.0, 0.0); n];
    let mut offset_x = 0.0;

    for comp_idx in 0..num_components {
        let nodes = &comp_nodes[comp_idx];
        if nodes.is_empty() {
            continue;
        }

        let cn = nodes.len();
        let mut local_adj: Vec<Vec<usize>> = vec![Vec::new(); cn];
        let mut global_to_local: HashMap<usize, usize> = HashMap::new();

        for (local_idx, &global_idx) in nodes.iter().enumerate() {
            global_to_local.insert(global_idx, local_idx);
        }

        for (local_idx, &global_idx) in nodes.iter().enumerate() {
            for &neighbor in &adj[global_idx] {
                if let Some(&local_neighbor) = global_to_local.get(&neighbor) {
                    local_adj[local_idx].push(local_neighbor);
                }
            }
        }

        let local_pos = layout_single_component(cn, &local_adj);

        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        for &(x, _) in &local_pos {
            min_x = min_x.min(x);
            max_x = max_x.max(x);
        }

        let width = max_x - min_x + 1.0;

        for (local_idx, &global_idx) in nodes.iter().enumerate() {
            positions[global_idx] = (local_pos[local_idx].0 - min_x + offset_x, local_pos[local_idx].1);
        }

        offset_x += width + 2.0;
    }

    positions
}

fn find_components(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    let mut component = vec![usize::MAX; n];
    let mut current_comp = 0;

    for start in 0..n {
        if component[start] != usize::MAX {
            continue;
        }

        let mut queue = VecDeque::new();
        queue.push_back(start);
        component[start] = current_comp;

        while let Some(node) = queue.pop_front() {
            for &neighbor in &adj[node] {
                if component[neighbor] == usize::MAX {
                    component[neighbor] = current_comp;
                    queue.push_back(neighbor);
                }
            }
        }
        current_comp += 1;
    }

    component
}

fn layout_single_component(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    if n <= 2 {
        let mut pos = Vec::with_capacity(n);
        for i in 0..n {
            pos.push((i as f64, 0.0));
        }
        return pos;
    }

    let total_edges: usize = adj.iter().map(|v| v.len()).sum::<usize>() / 2;

    if is_tree(n, adj) {
        return layout_tree(n, adj);
    }

    if n <= 50 && total_edges <= 100 {
        return force_directed_small(n, adj, 500);
    } else if n <= 500 {
        return force_directed_small(n, adj, 200);
    } else if n <= 2000 {
        return force_directed_large(n, adj, 100);
    } else {
        return force_directed_large(n, adj, 50);
    }
}

fn is_tree(n: usize, adj: &[Vec<usize>]) -> bool {
    let total_edges: usize = adj.iter().map(|v| v.len()).sum::<usize>() / 2;
    total_edges == n - 1
}

fn layout_tree(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let degree: Vec<usize> = adj.iter().map(|v| v.len()).collect();
    let root = (0..n).max_by_key(|&i| degree[i]).unwrap_or(0);

    let mut pos = vec![(0.0, 0.0); n];
    let mut visited = vec![false; n];
    let mut queue = VecDeque::new();

    visited[root] = true;
    queue.push_back((root, 0));

    let mut layers: Vec<Vec<usize>> = Vec::new();

    while let Some((node, depth)) = queue.pop_front() {
        while layers.len() <= depth {
            layers.push(Vec::new());
        }
        layers[depth].push(node);

        for &neighbor in &adj[node] {
            if !visited[neighbor] {
                visited[neighbor] = true;
                queue.push_back((neighbor, depth + 1));
            }
        }
    }

    for (depth, layer) in layers.iter().enumerate() {
        let layer_width = layer.len() as f64;
        for (i, &node) in layer.iter().enumerate() {
            pos[node] = ((i as f64 + 0.5) / layer_width * n as f64, depth as f64 * 2.0);
        }
    }

    pos
}

fn force_directed_small(n: usize, adj: &[Vec<usize>], iterations: usize) -> Vec<(f64, f64)> {
    let area = (n as f64).sqrt() * 10.0;
    let k = area / (n as f64).sqrt();

    let mut pos: Vec<(f64, f64)> = Vec::with_capacity(n);
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        let r = area * 0.4;
        pos.push((area / 2.0 + r * angle.cos(), area / 2.0 + r * angle.sin()));
    }

    let mut temp = area / 10.0;
    let cooling = 0.95;

    for _ in 0..iterations {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

        for i in 0..n {
            for j in (i + 1)..n {
                let dx = pos[i].0 - pos[j].0;
                let dy = pos[i].1 - pos[j].1;
                let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                let force = k * k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        for i in 0..n {
            for &j in &adj[i] {
                if i < j {
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                    let force = dist * dist / k;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[i].0 -= fx;
                    disp[i].1 -= fy;
                    disp[j].0 += fx;
                    disp[j].1 += fy;
                }
            }
        }

        for i in 0..n {
            let disp_len = (disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1).sqrt().max(0.01);
            let capped = disp_len.min(temp);
            pos[i].0 += disp[i].0 / disp_len * capped;
            pos[i].1 += disp[i].1 / disp_len * capped;
        }

        temp *= cooling;
    }

    pos
}

fn force_directed_large(n: usize, adj: &[Vec<usize>], iterations: usize) -> Vec<(f64, f64)> {
    let area = (n as f64).sqrt() * 10.0;
    let k = area / (n as f64).sqrt();
    let cell_size = k * 2.0;

    let mut pos: Vec<(f64, f64)> = Vec::with_capacity(n);
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        let r = area * 0.4;
        pos.push((area / 2.0 + r * angle.cos(), area / 2.0 + r * angle.sin()));
    }

    let mut temp = area / 10.0;
    let cooling = 0.9;

    for _ in 0..iterations {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

        let mut grid: HashMap<(i32, i32), Vec<usize>> = HashMap::new();
        for i in 0..n {
            let cx = (pos[i].0 / cell_size).floor() as i32;
            let cy = (pos[i].1 / cell_size).floor() as i32;
            grid.entry((cx, cy)).or_default().push(i);
        }

        for i in 0..n {
            let cx = (pos[i].0 / cell_size).floor() as i32;
            let cy = (pos[i].1 / cell_size).floor() as i32;

            for dcx in -1..=1 {
                for dcy in -1..=1 {
                    if let Some(neighbors) = grid.get(&(cx + dcx, cy + dcy)) {
                        for &j in neighbors {
                            if i >= j {
                                continue;
                            }
                            let dx = pos[i].0 - pos[j].0;
                            let dy = pos[i].1 - pos[j].1;
                            let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                            if dist > k * 4.0 {
                                continue;
                            }
                            let force = k * k / dist;
                            let fx = dx / dist * force;
                            let fy = dy / dist * force;
                            disp[i].0 += fx;
                            disp[i].1 += fy;
                            disp[j].0 -= fx;
                            disp[j].1 -= fy;
                        }
                    }
                }
            }
        }

        for i in 0..n {
            for &j in &adj[i] {
                if i < j {
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                    let force = dist * dist / k;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[i].0 -= fx;
                    disp[i].1 -= fy;
                    disp[j].0 += fx;
                    disp[j].1 += fy;
                }
            }
        }

        for i in 0..n {
            let disp_len = (disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1).sqrt().max(0.01);
            let capped = disp_len.min(temp);
            pos[i].0 += disp[i].0 / disp_len * capped;
            pos[i].1 += disp[i].1 / disp_len * capped;
        }

        temp *= cooling;
    }

    pos
}
