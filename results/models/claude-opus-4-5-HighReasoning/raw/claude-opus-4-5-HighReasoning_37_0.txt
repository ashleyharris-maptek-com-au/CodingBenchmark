using System;
using System.Collections.Generic;
using System.Linq;

class Solution
{
    static int n;
    static List<int>[] adj, radj;
    static List<int> bestFVS;

    static void Main()
    {
        var parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        adj = new List<int>[n];
        radj = new List<int>[n];
        for (int i = 0; i < n; i++) { adj[i] = new List<int>(); radj[i] = new List<int>(); }

        var selfLoops = new List<int>();
        for (int i = 0; i < m; i++)
        {
            parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]), v = int.Parse(parts[1]);
            if (u == v) { if (!selfLoops.Contains(u)) selfLoops.Add(u); }
            else { adj[u].Add(v); radj[v].Add(u); }
        }

        var removed = new HashSet<int>(selfLoops);
        ApplyReductions(removed);

        var active = new List<int>();
        for (int i = 0; i < n; i++) if (!removed.Contains(i)) active.Add(i);

        if (!HasCycle(active, removed))
        {
            Console.WriteLine(selfLoops.Count);
            Console.WriteLine(selfLoops.Count > 0 ? string.Join(" ", selfLoops) : "");
            return;
        }

        bestFVS = new List<int>(active);
        var sccs = GetSCCs(active, removed);
        var sccVertices = new HashSet<int>();
        foreach (var scc in sccs) if (scc.Count > 1) foreach (var v in scc) sccVertices.Add(v);

        var candidates = active.Where(v => sccVertices.Contains(v)).ToList();

        for (int k = 1; k <= candidates.Count && k < bestFVS.Count; k++)
        {
            if (Search(candidates, removed, new List<int>(), 0, k)) break;
        }

        var result = selfLoops.Concat(bestFVS).Distinct().ToList();
        Console.WriteLine(result.Count);
        Console.WriteLine(result.Count > 0 ? string.Join(" ", result) : "");
    }

    static void ApplyReductions(HashSet<int> removed)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int i = 0; i < n; i++)
            {
                if (removed.Contains(i)) continue;
                int inD = 0, outD = 0;
                foreach (var v in radj[i]) if (!removed.Contains(v)) inD++;
                foreach (var v in adj[i]) if (!removed.Contains(v)) outD++;
                if (inD == 0 || outD == 0) { removed.Add(i); changed = true; }
            }
        }
    }

    static bool HasCycle(List<int> verts, HashSet<int> removed)
    {
        var color = new int[n];
        foreach (var v in verts) if (!removed.Contains(v) && color[v] == 0 && DFSCycle(v, removed, color)) return true;
        return false;
    }

    static bool DFSCycle(int u, HashSet<int> removed, int[] color)
    {
        color[u] = 1;
        foreach (var v in adj[u])
        {
            if (removed.Contains(v)) continue;
            if (color[v] == 1) return true;
            if (color[v] == 0 && DFSCycle(v, removed, color)) return true;
        }
        color[u] = 2;
        return false;
    }

    static List<int> FindCycle(List<int> verts, HashSet<int> removed)
    {
        var parent = new int[n];
        var color = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        foreach (var start in verts)
        {
            if (removed.Contains(start) || color[start] != 0) continue;
            var stack = new Stack<(int, int)>();
            stack.Push((start, 0));
            while (stack.Count > 0)
            {
                var (u, idx) = stack.Pop();
                if (idx == 0) color[u] = 1;

                bool found = false;
                for (int i = idx; i < adj[u].Count; i++)
                {
                    int v = adj[u][i];
                    if (removed.Contains(v)) continue;
                    if (color[v] == 1)
                    {
                        var cycle = new List<int> { v };
                        int cur = u;
                        while (cur != v) { cycle.Add(cur); cur = parent[cur]; }
                        return cycle;
                    }
                    if (color[v] == 0)
                    {
                        parent[v] = u;
                        stack.Push((u, i + 1));
                        stack.Push((v, 0));
                        found = true;
                        break;
                    }
                }
                if (!found) color[u] = 2;
            }
        }
        return null;
    }

    static List<List<int>> GetSCCs(List<int> verts, HashSet<int> removed)
    {
        var order = new List<int>();
        var visited = new bool[n];
        foreach (var v in verts) if (!removed.Contains(v) && !visited[v]) DFS1(v, removed, visited, order);

        var comp = new int[n];
        for (int i = 0; i < n; i++) comp[i] = -1;
        var sccs = new List<List<int>>();
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (comp[v] == -1)
            {
                var scc = new List<int>();
                DFS2(v, removed, comp, sccs.Count, scc);
                sccs.Add(scc);
            }
        }
        return sccs;
    }

    static void DFS1(int u, HashSet<int> removed, bool[] visited, List<int> order)
    {
        visited[u] = true;
        foreach (var v in adj[u]) if (!removed.Contains(v) && !visited[v]) DFS1(v, removed, visited, order);
        order.Add(u);
    }

    static void DFS2(int u, HashSet<int> removed, int[] comp, int c, List<int> scc)
    {
        comp[u] = c;
        scc.Add(u);
        foreach (var v in radj[u]) if (!removed.Contains(v) && comp[v] == -1) DFS2(v, removed, comp, c, scc);
    }

    static bool Search(List<int> cands, HashSet<int> baseRemoved, List<int> current, int start, int maxK)
    {
        if (current.Count == maxK)
        {
            var rem = new HashSet<int>(baseRemoved);
            foreach (var v in current) rem.Add(v);
            if (!HasCycleQuick(rem))
            {
                bestFVS = new List<int>(current);
                return true;
            }
            return false;
        }

        var removed = new HashSet<int>(baseRemoved);
        foreach (var v in current) removed.Add(v);

        var cycle = FindCycle(cands, removed);
        if (cycle == null)
        {
            bestFVS = new List<int>(current);
            return true;
        }

        foreach (var v in cycle)
        {
            if (removed.Contains(v)) continue;
            current.Add(v);
            if (Search(cands, baseRemoved, current, 0, maxK)) return true;
            current.RemoveAt(current.Count - 1);
        }
        return false;
    }

    static bool HasCycleQuick(HashSet<int> removed)
    {
        var color = new int[n];
        for (int i = 0; i < n; i++)
        {
            if (!removed.Contains(i) && color[i] == 0 && DFSCycle(i, removed, color)) return true;
        }
        return false;
    }
}
