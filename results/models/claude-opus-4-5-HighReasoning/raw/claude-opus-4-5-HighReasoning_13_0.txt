#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <unordered_map>

using namespace std;
using ull = unsigned long long;

const ull MOD1 = 1000000007ULL;
const ull MOD2 = 998244353ULL;
const ull BASE1 = 31ULL;
const ull BASE2 = 37ULL;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    if (n == 0) {
        cout << "\n";
        return 0;
    }

    vector<string> strings(n);
    int minLen = INT_MAX;
    int minIdx = 0;
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
        if ((int)strings[i].size() < minLen) {
            minLen = (int)strings[i].size();
            minIdx = i;
        }
    }

    if (minLen == 0) {
        cout << "\n";
        return 0;
    }

    if (n == 1) {
        cout << strings[0] << "\n";
        return 0;
    }

    vector<ull> pow1(minLen + 1), pow2(minLen + 1);
    pow1[0] = pow2[0] = 1;
    for (int i = 1; i <= minLen; i++) {
        pow1[i] = pow1[i-1] * BASE1 % MOD1;
        pow2[i] = pow2[i-1] * BASE2 % MOD2;
    }

    const string& base = strings[minIdx];
    string answer;

    int lo = 1, hi = minLen;
    while (lo <= hi) {
        int len = (lo + hi) / 2;

        unordered_map<ull, int> candidates;
        candidates.reserve(minLen - len + 1);

        ull h1 = 0, h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE1 + (unsigned char)base[i]) % MOD1;
            h2 = (h2 * BASE2 + (unsigned char)base[i]) % MOD2;
        }
        candidates[(h1 << 32) | h2] = 0;

        for (int i = len; i < (int)base.size(); i++) {
            h1 = (h1 + MOD1 - pow1[len-1] * (unsigned char)base[i-len] % MOD1) % MOD1;
            h1 = (h1 * BASE1 + (unsigned char)base[i]) % MOD1;
            h2 = (h2 + MOD2 - pow2[len-1] * (unsigned char)base[i-len] % MOD2) % MOD2;
            h2 = (h2 * BASE2 + (unsigned char)base[i]) % MOD2;
            ull combined = (h1 << 32) | h2;
            if (!candidates.count(combined)) {
                candidates[combined] = i - len + 1;
            }
        }

        for (int s = 0; s < n && !candidates.empty(); s++) {
            if (s == minIdx) continue;

            const string& str = strings[s];
            if ((int)str.size() < len) {
                candidates.clear();
                break;
            }

            unordered_set<ull> found;
            found.reserve(candidates.size());

            h1 = h2 = 0;
            for (int i = 0; i < len; i++) {
                h1 = (h1 * BASE1 + (unsigned char)str[i]) % MOD1;
                h2 = (h2 * BASE2 + (unsigned char)str[i]) % MOD2;
            }
            ull combined = (h1 << 32) | h2;
            if (candidates.count(combined)) found.insert(combined);

            for (int i = len; i < (int)str.size(); i++) {
                h1 = (h1 + MOD1 - pow1[len-1] * (unsigned char)str[i-len] % MOD1) % MOD1;
                h1 = (h1 * BASE1 + (unsigned char)str[i]) % MOD1;
                h2 = (h2 + MOD2 - pow2[len-1] * (unsigned char)str[i-len] % MOD2) % MOD2;
                h2 = (h2 * BASE2 + (unsigned char)str[i]) % MOD2;
                combined = (h1 << 32) | h2;
                if (candidates.count(combined)) found.insert(combined);
            }

            for (auto it = candidates.begin(); it != candidates.end(); ) {
                if (!found.count(it->first)) {
                    it = candidates.erase(it);
                } else {
                    ++it;
                }
            }
        }

        if (!candidates.empty()) {
            int pos = candidates.begin()->second;
            answer = base.substr(pos, len);
            lo = len + 1;
        } else {
            hi = len - 1;
        }
    }

    cout << answer << "\n";
    return 0;
}
