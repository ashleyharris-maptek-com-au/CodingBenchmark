#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
using namespace std;

typedef long long ll;
typedef pair<ll, int> pli;
const ll INF = 1e18;

int n, m, k;
vector<pair<int, int>> adj[5005];
int terminals[5005];
bool isTerminal[5005];
ll dist[5005][5005];
int parent[5005][5005];

void dijkstra(int src) {
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    for (int i = 0; i < n; i++) {
        dist[src][i] = INF;
        parent[src][i] = -1;
    }
    dist[src][src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[src][u]) continue;
        for (auto [v, w] : adj[u]) {
            if (dist[src][u] + w < dist[src][v]) {
                dist[src][v] = dist[src][u] + w;
                parent[src][v] = u;
                pq.push({dist[src][v], v});
            }
        }
    }
}

vector<pair<int,int>> resultEdges;
ll resultWeight = 0;

struct DSU {
    int p[5005];
    void init(int n) { for (int i = 0; i < n; i++) p[i] = i; }
    int find(int x) { return p[x] == x ? x : p[x] = find(p[x]); }
    bool unite(int x, int y) {
        x = find(x); y = find(y);
        if (x == y) return false;
        p[x] = y;
        return true;
    }
} dsu;

bool inTree[5005];
vector<pair<int,int>> treeAdj[5005];

void addPath(int src, int dst) {
    int cur = dst;
    while (cur != src) {
        int prev = parent[src][cur];
        int w = 0;
        for (auto [v, wt] : adj[prev]) {
            if (v == cur && dist[src][prev] + wt == dist[src][cur]) {
                w = wt;
                break;
            }
        }
        if (dsu.unite(prev, cur)) {
            resultEdges.push_back({prev, cur});
            resultWeight += w;
            treeAdj[prev].push_back({cur, w});
            treeAdj[cur].push_back({prev, w});
        }
        inTree[prev] = true;
        inTree[cur] = true;
        cur = prev;
    }
}

void pruneTree() {
    bool changed = true;
    while (changed) {
        changed = false;
        for (int i = 0; i < n; i++) {
            if (inTree[i] && !isTerminal[i] && treeAdj[i].size() == 1) {
                int nei = treeAdj[i][0].first;
                int w = treeAdj[i][0].second;
                for (auto it = treeAdj[nei].begin(); it != treeAdj[nei].end(); ++it) {
                    if (it->first == i) {
                        treeAdj[nei].erase(it);
                        break;
                    }
                }
                treeAdj[i].clear();
                inTree[i] = false;
                resultWeight -= w;
                for (auto it = resultEdges.begin(); it != resultEdges.end(); ++it) {
                    if ((it->first == i && it->second == nei) || (it->first == nei && it->second == i)) {
                        resultEdges.erase(it);
                        break;
                    }
                }
                changed = true;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;

    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
        isTerminal[terminals[i]] = true;
    }

    if (k <= 1) {
        cout << "0 0\n";
        return 0;
    }

    for (int i = 0; i < k; i++) {
        dijkstra(terminals[i]);
    }

    vector<tuple<ll, int, int>> edges;
    for (int i = 0; i < k; i++) {
        for (int j = i + 1; j < k; j++) {
            edges.push_back({dist[terminals[i]][terminals[j]], terminals[i], terminals[j]});
        }
    }
    sort(edges.begin(), edges.end());

    dsu.init(n);

    for (auto [w, u, v] : edges) {
        if (dsu.find(u) != dsu.find(v)) {
            addPath(u, v);
        }
    }

    pruneTree();

    cout << resultWeight << " " << resultEdges.size() << "\n";
    for (auto [u, v] : resultEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
