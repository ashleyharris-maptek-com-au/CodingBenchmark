using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static void Main()
    {
        var firstLine = Console.ReadLine().Split(' ');
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        if (n == 0)
        {
            Console.WriteLine(0);
            return;
        }

        var cutLine = Console.ReadLine().Split(' ');
        var cuts = new int[n];
        var indices = new int[n];

        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(cutLine[i]);
            indices[i] = i;
        }

        // Sort indices by cut length descending (First Fit Decreasing)
        Array.Sort(indices, (a, b) => cuts[b].CompareTo(cuts[a]));

        var bins = new List<int>();           // remaining space in each bin
        var binContents = new List<List<int>>(); // original indices in each bin

        for (int i = 0; i < n; i++)
        {
            int idx = indices[i];
            int cutLen = cuts[idx];

            // Find first bin that fits
            int binIdx = -1;
            for (int j = 0; j < bins.Count; j++)
            {
                if (bins[j] >= cutLen)
                {
                    binIdx = j;
                    break;
                }
            }

            if (binIdx == -1)
            {
                // Create new bin
                bins.Add(stockLength - cutLen);
                binContents.Add(new List<int> { idx });
            }
            else
            {
                bins[binIdx] -= cutLen;
                binContents[binIdx].Add(idx);
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(bins.Count.ToString());

        foreach (var bin in binContents)
        {
            sb.AppendLine(string.Join(" ", bin));
        }

        Console.Write(sb);
    }
}
