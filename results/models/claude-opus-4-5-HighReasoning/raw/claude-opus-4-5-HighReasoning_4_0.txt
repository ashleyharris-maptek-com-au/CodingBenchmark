#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>

using namespace std;

const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3(double x_=0, double y_=0, double z_=0) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x+v.x, y+v.y, z+v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x-v.x, y-v.y, z-v.z); }
    Vec3 operator*(double s) const { return Vec3(x*s, y*s, z*s); }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y*v.z - z*v.y, z*v.x - x*v.z, x*v.y - y*v.x);
    }
    double norm() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const {
        double n = norm();
        return n > EPS ? (*this) * (1.0/n) : Vec3();
    }
};

int V;
double edgeLen;
vector<Vec3> verts;
vector<pair<Vec3, double>> planes;

vector<Vec3> baseTet;

void initBaseTet() {
    double s = sqrt(2.0) / 4.0;
    baseTet = {
        Vec3(s, s, s),
        Vec3(s, -s, -s),
        Vec3(-s, s, -s),
        Vec3(-s, -s, s)
    };
}

void computePlanes() {
    int n = (int)verts.size();
    int maxCheck = min(n, 200);

    for (int i = 0; i < maxCheck; i++) {
        for (int j = i+1; j < maxCheck; j++) {
            for (int k = j+1; k < maxCheck; k++) {
                Vec3 a = verts[i], b = verts[j], c = verts[k];
                Vec3 normal = (b - a).cross(c - a);
                double len = normal.norm();
                if (len < EPS) continue;
                normal = normal * (1.0 / len);

                double d = normal.dot(a);

                bool allSameSide = true;
                int sign = 0;
                for (int m = 0; m < n && allSameSide; m++) {
                    double dist = normal.dot(verts[m]) - d;
                    if (abs(dist) < EPS) continue;
                    int s = dist > 0 ? 1 : -1;
                    if (sign == 0) sign = s;
                    else if (sign != s) allSameSide = false;
                }

                if (allSameSide && sign != 0) {
                    if (sign > 0) {
                        normal = normal * -1;
                        d = -d;
                    }

                    bool dup = false;
                    for (auto& p : planes) {
                        if ((p.first - normal).norm() < 0.001 && abs(p.second - d) < 0.001) {
                            dup = true; break;
                        }
                    }
                    if (!dup) planes.push_back({normal, d});
                }
            }
        }
    }
}

bool isInsideContainer(const Vec3& p) {
    for (auto& plane : planes) {
        if (plane.first.dot(p) > plane.second + EPS) return false;
    }
    return true;
}

bool tetFitsInside(const Vec3& center, double e) {
    for (auto& bv : baseTet) {
        Vec3 v = center + bv * e;
        if (!isInsideContainer(v)) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    auto startTime = chrono::steady_clock::now();

    cin >> V >> edgeLen;
    verts.resize(V);
    for (int i = 0; i < V; i++) {
        cin >> verts[i].x >> verts[i].y >> verts[i].z;
    }

    initBaseTet();
    computePlanes();

    Vec3 lo = verts[0], hi = verts[0];
    for (auto& v : verts) {
        lo.x = min(lo.x, v.x); lo.y = min(lo.y, v.y); lo.z = min(lo.z, v.z);
        hi.x = max(hi.x, v.x); hi.y = max(hi.y, v.y); hi.z = max(hi.z, v.z);
    }

    double R = edgeLen * sqrt(3.0/8.0);
    double spacing = edgeLen * 1.0;

    double volume = (hi.x - lo.x) * (hi.y - lo.y) * (hi.z - lo.z);
    double tetVol = (edgeLen * edgeLen * edgeLen) / (6.0 * sqrt(2.0));
    double estCount = volume / tetVol * 0.3;

    if (estCount > 1e7) {
        spacing = edgeLen * 1.5;
    }

    vector<Vec3> placed;

    int layer = 0;
    for (double z = lo.z + R; z <= hi.z - R; z += spacing * 0.8) {
        int row = 0;
        double yOffset = (layer % 2) * spacing * 0.25;
        for (double y = lo.y + R + yOffset; y <= hi.y - R; y += spacing * 0.866) {
            double xOffset = (row % 2) * spacing * 0.5;
            for (double x = lo.x + R + xOffset; x <= hi.x - R; x += spacing) {
                Vec3 c(x, y, z);

                if (tetFitsInside(c, edgeLen)) {
                    placed.push_back(c);
                }

                auto now = chrono::steady_clock::now();
                if (chrono::duration_cast<chrono::seconds>(now - startTime).count() > 25) {
                    goto done;
                }
            }
            row++;
        }
        layer++;
    }

done:
    cout << placed.size() << "\n";
    cout << fixed;
    cout.precision(10);
    for (auto& p : placed) {
        cout << p.x << " " << p.y << " " << p.z << " 0 0 0\n";
    }

    return 0;
}
