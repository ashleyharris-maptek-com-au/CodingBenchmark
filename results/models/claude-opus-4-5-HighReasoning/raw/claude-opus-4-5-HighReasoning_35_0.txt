use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    // Greedy construction: pick vertex that dominates most uncovered vertices
    let mut in_set = vec![false; n];
    let mut dominated = vec![false; n];
    let mut uncovered_count = n;

    while uncovered_count > 0 {
        let mut best_v = 0;
        let mut best_gain = 0;

        for v in 0..n {
            if in_set[v] {
                continue;
            }
            let mut gain = 0;
            if !dominated[v] {
                gain += 1;
            }
            for &u in &adj[v] {
                if !dominated[u] {
                    gain += 1;
                }
            }
            if gain > best_gain {
                best_gain = gain;
                best_v = v;
            }
        }

        in_set[best_v] = true;
        if !dominated[best_v] {
            dominated[best_v] = true;
            uncovered_count -= 1;
        }
        for &u in &adj[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                uncovered_count -= 1;
            }
        }
    }

    // Local search: try to remove redundant vertices
    let mut dom_set: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();

    // Sort by degree (ascending) to try removing low-degree vertices first
    dom_set.sort_by_key(|&v| adj[v].len());

    let mut final_set: Vec<bool> = in_set.clone();

    for &v in &dom_set {
        // Try removing v
        final_set[v] = false;

        // Check if still dominating
        let mut valid = true;
        'check: for u in 0..n {
            if final_set[u] {
                continue;
            }
            let mut is_dominated = false;
            for &w in &adj[u] {
                if final_set[w] {
                    is_dominated = true;
                    break;
                }
            }
            if !is_dominated {
                valid = false;
                break 'check;
            }
        }

        if !valid {
            final_set[v] = true; // Put it back
        }
    }

    // Additional local search: swap improvements
    let current_set: Vec<usize> = (0..n).filter(|&v| final_set[v]).collect();

    // Try swapping each vertex in set with vertices outside
    'outer: for _ in 0..3 {
        let set_now: Vec<usize> = (0..n).filter(|&v| final_set[v]).collect();
        for &v in &set_now {
            // Try removing v and adding some other vertex
            final_set[v] = false;

            // Find vertices that become uncovered
            let mut uncovered: Vec<usize> = Vec::new();
            for u in 0..n {
                if final_set[u] {
                    continue;
                }
                let mut is_dom = false;
                for &w in &adj[u] {
                    if final_set[w] {
                        is_dom = true;
                        break;
                    }
                }
                if !is_dom {
                    uncovered.push(u);
                }
            }

            if uncovered.is_empty() {
                continue 'outer; // Found improvement
            }

            // Put it back
            final_set[v] = true;
        }
    }

    let result: Vec<usize> = (0..n).filter(|&v| final_set[v]).collect();

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", result.len()).unwrap();
    let s: Vec<String> = result.iter().map(|v| v.to_string()).collect();
    writeln!(out, "{}", s.join(" ")).unwrap();
}
