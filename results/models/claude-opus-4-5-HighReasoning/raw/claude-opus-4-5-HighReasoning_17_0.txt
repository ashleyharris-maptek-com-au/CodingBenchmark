using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int W, H, D;
    static List<(int x, int y, int z, int w, int h, int d, int idx)> placed = new();
    static Dictionary<(int, int, int), List<int>> grid = new();
    static int cellSize = 1;

    static void Main()
    {
        var firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        W = int.Parse(firstLine[1]);
        H = int.Parse(firstLine[2]);
        D = int.Parse(firstLine[3]);

        var boxes = new List<(int w, int h, int d, int idx, long vol)>();
        int maxDim = 1;
        for (int i = 0; i < n; i++)
        {
            var parts = Console.ReadLine().Split();
            int bw = int.Parse(parts[0]);
            int bh = int.Parse(parts[1]);
            int bd = int.Parse(parts[2]);
            boxes.Add((bw, bh, bd, i, (long)bw * bh * bd));
            maxDim = Math.Max(maxDim, Math.Max(bw, Math.Max(bh, bd)));
        }

        cellSize = Math.Max(1, maxDim);
        boxes.Sort((a, b) => b.vol.CompareTo(a.vol));

        var eps = new SortedSet<(int y, int x, int z)> { (0, 0, 0) };

        foreach (var box in boxes)
        {
            (int px, int py, int pz, int pw, int ph, int pd) best = (-1, -1, -1, 0, 0, 0);
            (int y, int x, int z) bestEp = (0, 0, 0);

            var orientations = GetOrientations(box.w, box.h, box.d);

            foreach (var ep in eps)
            {
                int x = ep.x, y = ep.y, z = ep.z;
                foreach (var (ow, oh, od) in orientations)
                {
                    if (x + ow <= W && y + oh <= H && z + od <= D)
                    {
                        if (!Overlaps(x, y, z, ow, oh, od))
                        {
                            if (best.px < 0 || y < best.py || (y == best.py && x < best.px) || (y == best.py && x == best.px && z < best.pz))
                            {
                                best = (x, y, z, ow, oh, od);
                                bestEp = ep;
                            }
                            break;
                        }
                    }
                }
                if (best.px >= 0 && best.py == 0) break;
            }

            if (best.px >= 0)
            {
                int idx = placed.Count;
                placed.Add((best.px, best.py, best.pz, best.pw, best.ph, best.pd, box.idx));
                AddToGrid(best.px, best.py, best.pz, best.pw, best.ph, best.pd, idx);
                eps.Remove(bestEp);

                int x2 = best.px + best.pw;
                int y2 = best.py + best.ph;
                int z2 = best.pz + best.pd;

                if (x2 < W) eps.Add((best.py, x2, best.pz));
                if (y2 < H) eps.Add((y2, best.px, best.pz));
                if (z2 < D) eps.Add((best.py, best.px, z2));
            }
        }

        Console.WriteLine(placed.Count);
        foreach (var p in placed)
        {
            Console.WriteLine($"{p.idx} {p.x} {p.y} {p.z}");
        }
    }

    static List<(int, int, int)> GetOrientations(int w, int h, int d)
    {
        var set = new HashSet<(int, int, int)>
        {
            (w, h, d), (w, d, h), (h, w, d), (h, d, w), (d, w, h), (d, h, w)
        };
        return set.OrderBy(o => o.Item2).ThenBy(o => o.Item1).ThenBy(o => o.Item3).ToList();
    }

    static void AddToGrid(int x, int y, int z, int w, int h, int d, int idx)
    {
        int cx1 = x / cellSize, cx2 = (x + w - 1) / cellSize;
        int cy1 = y / cellSize, cy2 = (y + h - 1) / cellSize;
        int cz1 = z / cellSize, cz2 = (z + d - 1) / cellSize;
        for (int cx = cx1; cx <= cx2; cx++)
            for (int cy = cy1; cy <= cy2; cy++)
                for (int cz = cz1; cz <= cz2; cz++)
                {
                    var key = (cx, cy, cz);
                    if (!grid.ContainsKey(key)) grid[key] = new List<int>();
                    grid[key].Add(idx);
                }
    }

    static bool Overlaps(int x, int y, int z, int w, int h, int d)
    {
        int cx1 = x / cellSize, cx2 = (x + w - 1) / cellSize;
        int cy1 = y / cellSize, cy2 = (y + h - 1) / cellSize;
        int cz1 = z / cellSize, cz2 = (z + d - 1) / cellSize;
        var checked_ = new HashSet<int>();
        for (int cx = cx1; cx <= cx2; cx++)
            for (int cy = cy1; cy <= cy2; cy++)
                for (int cz = cz1; cz <= cz2; cz++)
                {
                    var key = (cx, cy, cz);
                    if (grid.TryGetValue(key, out var list))
                    {
                        foreach (var i in list)
                        {
                            if (checked_.Contains(i)) continue;
                            checked_.Add(i);
                            var p = placed[i];
                            if (x < p.x + p.w && x + w > p.x &&
                                y < p.y + p.h && y + h > p.y &&
                                z < p.z + p.d && z + d > p.z)
                                return true;
                        }
                    }
                }
        return false;
    }
}
