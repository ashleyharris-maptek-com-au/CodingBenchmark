using System;
using System.Collections.Generic;
using System.Diagnostics;

class QAP
{
    static int N;
    static int[,] flow;
    static int[,] dist;
    static int[] bestPerm;
    static long bestCost;
    static Random rand = new Random(42);

    static void Main()
    {
        ReadInput();

        if (N <= 10)
        {
            // For very small N, try all permutations
            bestCost = long.MaxValue;
            bestPerm = new int[N];
            int[] perm = new int[N];
            bool[] used = new bool[N];
            Backtrack(perm, used, 0);
        }
        else
        {
            // For larger N, use heuristics
            bestPerm = GreedyConstruction();
            bestCost = ComputeCost(bestPerm);

            // Local search with time limit
            var sw = Stopwatch.StartNew();
            int maxMs = 280000; // ~4.5 minutes

            // Multiple restarts with local search
            int restarts = 0;
            while (sw.ElapsedMilliseconds < maxMs)
            {
                int[] current;
                if (restarts == 0)
                    current = (int[])bestPerm.Clone();
                else
                    current = RandomPermutation();

                long currentCost = ComputeCost(current);

                // 2-opt local search
                bool improved = true;
                while (improved && sw.ElapsedMilliseconds < maxMs)
                {
                    improved = false;
                    for (int i = 0; i < N - 1 && sw.ElapsedMilliseconds < maxMs; i++)
                    {
                        for (int j = i + 1; j < N; j++)
                        {
                            long delta = SwapDelta(current, i, j);
                            if (delta < 0)
                            {
                                Swap(current, i, j);
                                currentCost += delta;
                                improved = true;

                                if (currentCost < bestCost)
                                {
                                    bestCost = currentCost;
                                    Array.Copy(current, bestPerm, N);
                                }
                            }
                        }
                    }
                }

                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    Array.Copy(current, bestPerm, N);
                }

                restarts++;
                if (N > 500 && restarts > 3) break; // Limit restarts for large N
                if (N > 100 && restarts > 20) break;
            }
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static void ReadInput()
    {
        N = int.Parse(Console.ReadLine().Trim());
        flow = new int[N, N];
        dist = new int[N, N];

        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N; j++)
                flow[i, j] = int.Parse(parts[j]);
        }

        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N; j++)
                dist[i, j] = int.Parse(parts[j]);
        }
    }

    static long ComputeCost(int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                cost += (long)flow[i, j] * dist[perm[i], perm[j]];
        return cost;
    }

    static long SwapDelta(int[] perm, int r, int s)
    {
        // Compute change in cost when swapping positions r and s
        long delta = 0;
        int pr = perm[r], ps = perm[s];

        for (int k = 0; k < N; k++)
        {
            if (k == r || k == s) continue;
            int pk = perm[k];

            // Contribution from (r, k) and (k, r)
            delta -= (long)flow[r, k] * dist[pr, pk];
            delta -= (long)flow[k, r] * dist[pk, pr];
            delta += (long)flow[r, k] * dist[ps, pk];
            delta += (long)flow[k, r] * dist[pk, ps];

            // Contribution from (s, k) and (k, s)
            delta -= (long)flow[s, k] * dist[ps, pk];
            delta -= (long)flow[k, s] * dist[pk, ps];
            delta += (long)flow[s, k] * dist[pr, pk];
            delta += (long)flow[k, s] * dist[pk, pr];
        }

        // Contribution from (r, s) and (s, r)
        delta -= (long)flow[r, s] * dist[pr, ps];
        delta -= (long)flow[s, r] * dist[ps, pr];
        delta += (long)flow[r, s] * dist[ps, pr];
        delta += (long)flow[s, r] * dist[pr, ps];

        return delta;
    }

    static void Swap(int[] perm, int i, int j)
    {
        int tmp = perm[i];
        perm[i] = perm[j];
        perm[j] = tmp;
    }

    static int[] GreedyConstruction()
    {
        // Compute flow and distance sums for each facility/location
        long[] flowSum = new long[N];
        long[] distSum = new long[N];

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += flow[i, j] + flow[j, i];
                distSum[i] += dist[i, j] + dist[j, i];
            }
        }

        // Sort facilities by flow sum (descending)
        int[] facilities = new int[N];
        for (int i = 0; i < N; i++) facilities[i] = i;
        Array.Sort(facilities, (a, b) => flowSum[b].CompareTo(flowSum[a]));

        // Sort locations by distance sum (ascending)
        int[] locations = new int[N];
        for (int i = 0; i < N; i++) locations[i] = i;
        Array.Sort(locations, (a, b) => distSum[a].CompareTo(distSum[b]));

        // Assign high-flow facilities to low-distance locations
        int[] perm = new int[N];
        for (int i = 0; i < N; i++)
            perm[facilities[i]] = locations[i];

        return perm;
    }

    static int[] RandomPermutation()
    {
        int[] perm = new int[N];
        for (int i = 0; i < N; i++) perm[i] = i;
        for (int i = N - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            int tmp = perm[i];
            perm[i] = perm[j];
            perm[j] = tmp;
        }
        return perm;
    }

    static void Backtrack(int[] perm, bool[] used, int pos)
    {
        if (pos == N)
        {
            long cost = ComputeCost(perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                Array.Copy(perm, bestPerm, N);
            }
            return;
        }

        for (int loc = 0; loc < N; loc++)
        {
            if (!used[loc])
            {
                used[loc] = true;
                perm[pos] = loc;
                Backtrack(perm, used, pos + 1);
                used[loc] = false;
            }
        }
    }
}
