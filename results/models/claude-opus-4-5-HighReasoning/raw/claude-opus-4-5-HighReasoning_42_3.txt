; SPIR-V Geometry Shader - Line to Thick Ribbon
; Expands thin lines into thick quads

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%thickness = OpConstant %float 0.03

; Input gl_PerVertex block (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_2
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Input color array (location 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load gl_in[0].gl_Position (p0)
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0

; Load gl_in[1].gl_Position (p1)
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1

; Load colors
%ptr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %ptr_c1

; Compute direction: dir = p1 - p0
%dir4 = OpFSub %v4float %p1 %p0

; Extract x and y components
%dir_x = OpCompositeExtract %float %dir4 0
%dir_y = OpCompositeExtract %float %dir4 1

; Compute perpendicular: perp_raw = (-dir.y, dir.x)
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize perpendicular
%perp_len = OpExtInst %float %1 Length %perp_raw
%inv_len = OpFDiv %float %float_1 %perp_len
%perp_unit = OpVectorTimesScalar %v2float %perp_raw %inv_len
%perp = OpVectorTimesScalar %v2float %perp_unit %thickness

; Extract perp components
%perp_x = OpCompositeExtract %float %perp 0
%perp_y = OpCompositeExtract %float %perp 1

; Get p0 and p1 components
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

; Vertex 1: p0 + perp
%v1_x = OpFAdd %float %p0_x %perp_x
%v1_y = OpFAdd %float %p0_y %perp_y
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
%ptr_gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_gl_out_pos %v1
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p0 - perp
%v2_x = OpFSub %float %p0_x %perp_x
%v2_y = OpFSub %float %p0_y %perp_y
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p0_z %p0_w
OpStore %ptr_gl_out_pos %v2
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: p1 + perp
%v3_x = OpFAdd %float %p1_x %perp_x
%v3_y = OpFAdd %float %p1_y %perp_y
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
OpStore %ptr_gl_out_pos %v3
OpStore %outColor %c1
OpEmitVertex

; Vertex 4: p1 - perp
%v4_x = OpFSub %float %p1_x %perp_x
%v4_y = OpFSub %float %p1_y %perp_y
%v4 = OpCompositeConstruct %v4float %v4_x %v4_y %p1_z %p1_w
OpStore %ptr_gl_out_pos %v4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
