; SPIR-V Geometry Shader - Wireframe from Triangle Adjacency
; Emits each triangle edge as a thin quad

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%thickness = OpConstant %float 0.008

; Input types (array size 6 for InputTrianglesAdjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_6
%ptr_in_v4float = OpTypePointer Input %v4float
%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_6
%ptr_in_v3float = OpTypePointer Input %v3float

; Output types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3float Output

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load positions for main triangle vertices (indices 0, 2, 4)
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos4_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

; Load colors for main triangle vertices
%col0_ptr = OpAccessChain %ptr_in_v3float %inColor %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3float %inColor %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3float %inColor %int_4
%col4 = OpLoad %v3float %col4_ptr

; Output position pointer
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

; ========== Edge 0->2 ==========
; Compute edge direction
%edge02 = OpFSub %v4float %pos2 %pos0
%edge02_x = OpCompositeExtract %float %edge02 0
%edge02_y = OpCompositeExtract %float %edge02 1
; Perpendicular in 2D: (-y, x)
%perp02_x = OpFNegate %float %edge02_y
%perp02_y = %edge02_x
; Normalize perpendicular
%perp02_x_sq = OpFMul %float %perp02_x %perp02_x
%perp02_y_sq = OpFMul %float %perp02_y %perp02_y
%perp02_len_sq = OpFAdd %float %perp02_x_sq %perp02_y_sq
%perp02_len = OpExtInst %float %1 Sqrt %perp02_len_sq
%perp02_x_n = OpFDiv %float %perp02_x %perp02_len
%perp02_y_n = OpFDiv %float %perp02_y %perp02_len
; Scale by thickness
%off02_x = OpFMul %float %perp02_x_n %thickness
%off02_y = OpFMul %float %perp02_y_n %thickness
%offset02 = OpCompositeConstruct %v4float %off02_x %off02_y %float_0 %float_0
; Average color
%col02_sum = OpFAdd %v3float %col0 %col2
%col02 = OpVectorTimesScalar %v3float %col02_sum %float_0_5

; Emit 4 vertices for edge quad
%v0_minus = OpFSub %v4float %pos0 %offset02
OpStore %out_pos_ptr %v0_minus
OpStore %outColor %col02
OpEmitVertex

%v0_plus = OpFAdd %v4float %pos0 %offset02
OpStore %out_pos_ptr %v0_plus
OpStore %outColor %col02
OpEmitVertex

%v2_minus = OpFSub %v4float %pos2 %offset02
OpStore %out_pos_ptr %v2_minus
OpStore %outColor %col02
OpEmitVertex

%v2_plus = OpFAdd %v4float %pos2 %offset02
OpStore %out_pos_ptr %v2_plus
OpStore %outColor %col02
OpEmitVertex

OpEndPrimitive

; ========== Edge 2->4 ==========
%edge24 = OpFSub %v4float %pos4 %pos2
%edge24_x = OpCompositeExtract %float %edge24 0
%edge24_y = OpCompositeExtract %float %edge24 1
%perp24_x = OpFNegate %float %edge24_y
%perp24_y = %edge24_x
%perp24_x_sq = OpFMul %float %perp24_x %perp24_x
%perp24_y_sq = OpFMul %float %perp24_y %perp24_y
%perp24_len_sq = OpFAdd %float %perp24_x_sq %perp24_y_sq
%perp24_len = OpExtInst %float %1 Sqrt %perp24_len_sq
%perp24_x_n = OpFDiv %float %perp24_x %perp24_len
%perp24_y_n = OpFDiv %float %perp24_y %perp24_len
%off24_x = OpFMul %float %perp24_x_n %thickness
%off24_y = OpFMul %float %perp24_y_n %thickness
%offset24 = OpCompositeConstruct %v4float %off24_x %off24_y %float_0 %float_0
%col24_sum = OpFAdd %v3float %col2 %col4
%col24 = OpVectorTimesScalar %v3float %col24_sum %float_0_5

%v2b_minus = OpFSub %v4float %pos2 %offset24
OpStore %out_pos_ptr %v2b_minus
OpStore %outColor %col24
OpEmitVertex

%v2b_plus = OpFAdd %v4float %pos2 %offset24
OpStore %out_pos_ptr %v2b_plus
OpStore %outColor %col24
OpEmitVertex

%v4_minus = OpFSub %v4float %pos4 %offset24
OpStore %out_pos_ptr %v4_minus
OpStore %outColor %col24
OpEmitVertex

%v4_plus = OpFAdd %v4float %pos4 %offset24
OpStore %out_pos_ptr %v4_plus
OpStore %outColor %col24
OpEmitVertex

OpEndPrimitive

; ========== Edge 4->0 ==========
%edge40 = OpFSub %v4float %pos0 %pos4
%edge40_x = OpCompositeExtract %float %edge40 0
%edge40_y = OpCompositeExtract %float %edge40 1
%perp40_x = OpFNegate %float %edge40_y
%perp40_y = %edge40_x
%perp40_x_sq = OpFMul %float %perp40_x %perp40_x
%perp40_y_sq = OpFMul %float %perp40_y %perp40_y
%perp40_len_sq = OpFAdd %float %perp40_x_sq %perp40_y_sq
%perp40_len = OpExtInst %float %1 Sqrt %perp40_len_sq
%perp40_x_n = OpFDiv %float %perp40_x %perp40_len
%perp40_y_n = OpFDiv %float %perp40_y %perp40_len
%off40_x = OpFMul %float %perp40_x_n %thickness
%off40_y = OpFMul %float %perp40_y_n %thickness
%offset40 = OpCompositeConstruct %v4float %off40_x %off40_y %float_0 %float_0
%col40_sum = OpFAdd %v3float %col4 %col0
%col40 = OpVectorTimesScalar %v3float %col40_sum %float_0_5

%v4b_minus = OpFSub %v4float %pos4 %offset40
OpStore %out_pos_ptr %v4b_minus
OpStore %outColor %col40
OpEmitVertex

%v4b_plus = OpFAdd %v4float %pos4 %offset40
OpStore %out_pos_ptr %v4b_plus
OpStore %outColor %col40
OpEmitVertex

%v0b_minus = OpFSub %v4float %pos0 %offset40
OpStore %out_pos_ptr %v0b_minus
OpStore %outColor %col40
OpEmitVertex

%v0b_plus = OpFAdd %v4float %pos0 %offset40
OpStore %out_pos_ptr %v0b_plus
OpStore %outColor %col40
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
