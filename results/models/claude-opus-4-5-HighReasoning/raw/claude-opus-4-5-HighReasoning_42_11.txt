; SPIR-V Geometry Shader - Dashed Thick Curve with Lines Adjacency
; Capability and memory model
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
               OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input/output location decorations
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
      %void = OpTypeVoid
  %void_fn = OpTypeFunction %void
     %float = OpTypeFloat 32
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0

; Constants
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_2 = OpConstant %int 2
    %int_3 = OpConstant %int 3
   %uint_4 = OpConstant %uint 4
  %float_0 = OpConstant %float 0
  %float_1 = OpConstant %float 1
%float_0_02 = OpConstant %float 0.02
%float_0_333 = OpConstant %float 0.333333343
%float_0_667 = OpConstant %float 0.666666687
%float_0_5 = OpConstant %float 0.5

; Input gl_PerVertex struct and array
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_4
     %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

; Input color array
%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
   %inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3float = OpTypePointer Input %v3float

; Output color
%ptr_out_v3float = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3float Output

; Main function
      %main = OpFunction %void None %void_fn
     %entry = OpLabel

; Load gl_in[0].gl_Position (previous adjacency vertex)
  %ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
      %p0 = OpLoad %v4float %ptr_p0

; Load gl_in[1].gl_Position (start of segment)
  %ptr_p1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
      %p1 = OpLoad %v4float %ptr_p1

; Load gl_in[2].gl_Position (end of segment)
  %ptr_p2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
      %p2 = OpLoad %v4float %ptr_p2

; Load gl_in[3].gl_Position (next adjacency vertex)
  %ptr_p3 = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
      %p3 = OpLoad %v4float %ptr_p3

; Load colors
%ptr_c1 = OpAccessChain %ptr_in_v3float %inColor %int_1
    %c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3float %inColor %int_2
    %c2 = OpLoad %v3float %ptr_c2

; Extract xy components for 2D calculations
  %p0_xy = OpVectorShuffle %v3float %p0 %p0 0 1 4
  %p1_xy = OpVectorShuffle %v3float %p1 %p1 0 1 4
  %p2_xy = OpVectorShuffle %v3float %p2 %p2 0 1 4
  %p3_xy = OpVectorShuffle %v3float %p3 %p3 0 1 4

; Compute tangent at p1: normalize(p2 - p0)
 %tan1_raw = OpFSub %v3float %p2_xy %p0_xy
     %tan1 = OpExtInst %v3float %1 Normalize %tan1_raw

; Compute tangent at p2: normalize(p3 - p1)
 %tan2_raw = OpFSub %v3float %p3_xy %p1_xy
     %tan2 = OpExtInst %v3float %1 Normalize %tan2_raw

; Compute perpendiculars (rotate 90 degrees): perp = (-tan.y, tan.x, 0)
 %tan1_x = OpCompositeExtract %float %tan1 0
 %tan1_y = OpCompositeExtract %float %tan1 1
%neg_tan1_y = OpFNegate %float %tan1_y
   %perp1 = OpCompositeConstruct %v3float %neg_tan1_y %tan1_x %float_0

 %tan2_x = OpCompositeExtract %float %tan2 0
 %tan2_y = OpCompositeExtract %float %tan2 1
%neg_tan2_y = OpFNegate %float %tan2_y
   %perp2 = OpCompositeConstruct %v3float %neg_tan2_y %tan2_x %float_0

; Scale perpendiculars by thickness
%offset1 = OpVectorTimesScalar %v3float %perp1 %float_0_02
%offset2 = OpVectorTimesScalar %v3float %perp2 %float_0_02

; Compute segment direction for interpolating offsets along the segment
  %segDir = OpFSub %v3float %p2_xy %p1_xy

; Compute points at 1/3 and 2/3 along segment
%seg_third = OpVectorTimesScalar %v3float %segDir %float_0_333
%seg_two_third = OpVectorTimesScalar %v3float %segDir %float_0_667
%pt_third = OpFAdd %v3float %p1_xy %seg_third
%pt_two_third = OpFAdd %v3float %p1_xy %seg_two_third

; Interpolate perpendicular offsets at 1/3 and 2/3
%lerp_third = OpExtInst %v3float %1 FMix %offset1 %offset2 %float_0_333
%lerp_two_third = OpExtInst %v3float %1 FMix %offset1 %offset2 %float_0_667

; Interpolate colors at 1/3 and 2/3
%color_third = OpExtInst %v3float %1 FMix %c1 %c2 %float_0_333
%color_two_third = OpExtInst %v3float %1 FMix %c1 %c2 %float_0_667

; Extract z and w from original positions for depth/perspective
  %p1_z = OpCompositeExtract %float %p1 2
  %p1_w = OpCompositeExtract %float %p1 3
  %p2_z = OpCompositeExtract %float %p2 2
  %p2_w = OpCompositeExtract %float %p2 3

; Interpolate z and w at 1/3 and 2/3
%z_third = OpExtInst %float %1 FMix %p1_z %p2_z %float_0_333
%w_third = OpExtInst %float %1 FMix %p1_w %p2_w %float_0_333
%z_two_third = OpExtInst %float %1 FMix %p1_z %p2_z %float_0_667
%w_two_third = OpExtInst %float %1 FMix %p1_w %p2_w %float_0_667

; ============ FIRST DASH (0 to 1/3) ============

; Vertex 0: p1 - offset1 (bottom-left of first dash)
%v0_xy = OpFSub %v3float %p1_xy %offset1
%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %p1_z %p1_w
%out_ptr_0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
       OpStore %out_ptr_0 %v0_pos
       OpStore %outColor %c1
       OpEmitVertex

; Vertex 1: p1 + offset1 (top-left of first dash)
%v1_xy = OpFAdd %v3float %p1_xy %offset1
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %p1_w
       OpStore %out_ptr_0 %v1_pos
       OpStore %outColor %c1
       OpEmitVertex

; Vertex 2: pt_third - lerp_third (bottom-right of first dash)
%v2_xy = OpFSub %v3float %pt_third %lerp_third
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %z_third %w_third
       OpStore %out_ptr_0 %v2_pos
       OpStore %outColor %color_third
       OpEmitVertex

; Vertex 3: pt_third + lerp_third (top-right of first dash)
%v3_xy = OpFAdd %v3float %pt_third %lerp_third
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %z_third %w_third
       OpStore %out_ptr_0 %v3_pos
       OpStore %outColor %color_third
       OpEmitVertex

       OpEndPrimitive

; ============ SECOND DASH (2/3 to 1) ============

; Vertex 4: pt_two_third - lerp_two_third (bottom-left of second dash)
%v4_xy = OpFSub %v3float %pt_two_third %lerp_two_third
%v4_x = OpCompositeExtract %float %v4_xy 0
%v4_y = OpCompositeExtract %float %v4_xy 1
%v4_pos = OpCompositeConstruct %v4float %v4_x %v4_y %z_two_third %w_two_third
       OpStore %out_ptr_0 %v4_pos
       OpStore %outColor %color_two_third
       OpEmitVertex

; Vertex 5: pt_two_third + lerp_two_third (top-left of second dash)
%v5_xy = OpFAdd %v3float %pt_two_third %lerp_two_third
%v5_x = OpCompositeExtract %float %v5_xy 0
%v5_y = OpCompositeExtract %float %v5_xy 1
%v5_pos = OpCompositeConstruct %v4float %v5_x %v5_y %z_two_third %w_two_third
       OpStore %out_ptr_0 %v5_pos
       OpStore %outColor %color_two_third
       OpEmitVertex

; Vertex 6: p2 - offset2 (bottom-right of second dash)
%v6_xy = OpFSub %v3float %p2_xy %offset2
%v6_x = OpCompositeExtract %float %v6_xy 0
%v6_y = OpCompositeExtract %float %v6_xy 1
%v6_pos = OpCompositeConstruct %v4float %v6_x %v6_y %p2_z %p2_w
       OpStore %out_ptr_0 %v6_pos
       OpStore %outColor %c2
       OpEmitVertex

; Vertex 7: p2 + offset2 (top-right of second dash)
%v7_xy = OpFAdd %v3float %p2_xy %offset2
%v7_x = OpCompositeExtract %float %v7_xy 0
%v7_y = OpCompositeExtract %float %v7_xy 1
%v7_pos = OpCompositeConstruct %v4float %v7_x %v7_y %p2_z %p2_w
       OpStore %out_ptr_0 %v7_pos
       OpStore %outColor %c2
       OpEmitVertex

       OpEndPrimitive

       OpReturn
       OpFunctionEnd
