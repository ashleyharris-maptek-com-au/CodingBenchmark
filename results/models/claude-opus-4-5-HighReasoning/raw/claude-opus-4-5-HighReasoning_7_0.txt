const double EPS = 1e-10;
const double EPS_COPLANAR = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x+v.x, y+v.y, z+v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x-v.x, y-v.y, z-v.z); }
    Vec3 operator*(double s) const { return Vec3(x*s, y*s, z*s); }
    Vec3 operator/(double s) const { return Vec3(x/s, y/s, z/s); }
    bool operator<(const Vec3& v) const {
        if (abs(x - v.x) > EPS) return x < v.x;
        if (abs(y - v.y) > EPS) return y < v.y;
        return z < v.z - EPS;
    }
    bool operator==(const Vec3& v) const {
        return abs(x-v.x) < EPS && abs(y-v.y) < EPS && abs(z-v.z) < EPS;
    }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y*v.z - z*v.y, z*v.x - x*v.z, x*v.y - y*v.x);
    }
    double length() const { return sqrt(x*x + y*y + z*z); }
    double lengthSq() const { return x*x + y*y + z*z; }
    Vec3 normalized() const {
        double l = length();
        return l > EPS ? Vec3(x/l, y/l, z/l) : Vec3(0,0,0);
    }
};

struct Triangle {
    int v[3];
    int meshId;
};

struct AABB {
    Vec3 mn, mx;
    AABB() : mn(1e18,1e18,1e18), mx(-1e18,-1e18,-1e18) {}
    void expand(const Vec3& p) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    void expand(const AABB& b) {
        mn.x = min(mn.x, b.mn.x); mn.y = min(mn.y, b.mn.y); mn.z = min(mn.z, b.mn.z);
        mx.x = max(mx.x, b.mx.x); mx.y = max(mx.y, b.mx.y); mx.z = max(mx.z, b.mx.z);
    }
    bool intersects(const AABB& b) const {
        return mn.x <= b.mx.x + EPS && mx.x >= b.mn.x - EPS &&
               mn.y <= b.mx.y + EPS && mx.y >= b.mn.y - EPS &&
               mn.z <= b.mx.z + EPS && mx.z >= b.mn.z - EPS;
    }
    Vec3 center() const { return (mn + mx) * 0.5; }
};

class Mesh {
public:
    vector<Vec3> verts;
    vector<Triangle> tris;
    int meshId;

    AABB getTriAABB(int ti) const {
        AABB b;
        b.expand(verts[tris[ti].v[0]]);
        b.expand(verts[tris[ti].v[1]]);
        b.expand(verts[tris[ti].v[2]]);
        return b;
    }

    Vec3 getTriNormal(int ti) const {
        Vec3 a = verts[tris[ti].v[0]];
        Vec3 b = verts[tris[ti].v[1]];
        Vec3 c = verts[tris[ti].v[2]];
        return (b-a).cross(c-a).normalized();
    }
};

struct BVHNode {
    AABB box;
    int left, right;
    int triStart, triCount;
    bool isLeaf() const { return triCount > 0; }
};

class BVH {
public:
    vector<BVHNode> nodes;
    vector<int> triIndices;
    const Mesh* mesh;

    void build(const Mesh* m) {
        mesh = m;
        if (m->tris.empty()) return;
        triIndices.resize(m->tris.size());
        iota(triIndices.begin(), triIndices.end(), 0);
        nodes.reserve(m->tris.size() * 2);
        buildNode(0, (int)triIndices.size(), 0);
    }

    int buildNode(int start, int end, int depth) {
        BVHNode node;
        node.triStart = start;
        node.triCount = 0;

        for (int i = start; i < end; i++) {
            node.box.expand(mesh->getTriAABB(triIndices[i]));
        }

        int count = end - start;
        if (count <= 4 || depth > 25) {
            node.triCount = count;
            node.left = node.right = -1;
            nodes.push_back(node);
            return (int)nodes.size() - 1;
        }

        int axis = 0;
        Vec3 ext = node.box.mx - node.box.mn;
        if (ext.y > ext.x && ext.y > ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z > ext.y) axis = 2;

        auto getAxis = [axis](const Vec3& v) {
            return axis == 0 ? v.x : (axis == 1 ? v.y : v.z);
        };

        int mid = (start + end) / 2;
        nth_element(triIndices.begin() + start, triIndices.begin() + mid,
                    triIndices.begin() + end, [&](int a, int b) {
            return getAxis(mesh->getTriAABB(a).center()) < getAxis(mesh->getTriAABB(b).center());
        });

        int nodeIdx = (int)nodes.size();
        nodes.push_back(node);

        nodes[nodeIdx].left = buildNode(start, mid, depth + 1);
        nodes[nodeIdx].right = buildNode(mid, end, depth + 1);

        return nodeIdx;
    }

    template<typename F>
    void queryBox(const AABB& box, F&& callback) const {
        if (nodes.empty()) return;
        queryBoxNode(0, box, callback);
    }

    template<typename F>
    void queryBoxNode(int nodeIdx, const AABB& box, F&& callback) const {
        const BVHNode& node = nodes[nodeIdx];
        if (!node.box.intersects(box)) return;

        if (node.isLeaf()) {
            for (int i = 0; i < node.triCount; i++) {
                callback(triIndices[node.triStart + i]);
            }
        } else {
            queryBoxNode(node.left, box, callback);
            queryBoxNode(node.right, box, callback);
        }
    }
};

double trianglePointDistance(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 ab = b - a, ac = c - a, ap = p - a;
    double d1 = ab.dot(ap), d2 = ac.dot(ap);
    if (d1 <= 0 && d2 <= 0) return (p - a).length();

    Vec3 bp = p - b;
    double d3 = ab.dot(bp), d4 = ac.dot(bp);
    if (d3 >= 0 && d4 <= d3) return (p - b).length();

    double vc = d1*d4 - d3*d2;
    if (vc <= 0 && d1 >= 0 && d3 <= 0) {
        double v = d1 / (d1 - d3);
        return (p - (a + ab * v)).length();
    }

    Vec3 cp = p - c;
    double d5 = ab.dot(cp), d6 = ac.dot(cp);
    if (d6 >= 0 && d5 <= d6) return (p - c).length();

    double vb = d5*d2 - d1*d6;
    if (vb <= 0 && d2 >= 0 && d6 <= 0) {
        double w = d2 / (d2 - d6);
        return (p - (a + ac * w)).length();
    }

    double va = d3*d6 - d5*d4;
    if (va <= 0 && (d4-d3) >= 0 && (d5-d6) >= 0) {
        double w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
        return (p - (b + (c - b) * w)).length();
    }

    double denom = 1.0 / (va + vb + vc);
    double v = vb * denom, w = vc * denom;
    return (p - (a + ab * v + ac * w)).length();
}

bool rayTriangle(const Vec3& orig, const Vec3& dir, const Vec3& a, const Vec3& b, const Vec3& c, double& t) {
    Vec3 e1 = b - a, e2 = c - a;
    Vec3 pvec = dir.cross(e2);
    double det = e1.dot(pvec);
    if (abs(det) < EPS) return false;

    double invDet = 1.0 / det;
    Vec3 tvec = orig - a;
    double u = tvec.dot(pvec) * invDet;
    if (u < -EPS || u > 1.0 + EPS) return false;

    Vec3 qvec = tvec.cross(e1);
    double v = dir.dot(qvec) * invDet;
    if (v < -EPS || u + v > 1.0 + EPS) return false;

    t = e2.dot(qvec) * invDet;
    return t > EPS;
}

int countRayHits(const Vec3& point, const Vec3& dir, const Mesh& mesh, const BVH& bvh) {
    int count = 0;
    AABB rayBox;
    rayBox.expand(point);
    rayBox.expand(point + dir * 1e6);

    bvh.queryBox(rayBox, [&](int ti) {
        const Triangle& tri = mesh.tris[ti];
        double t;
        if (rayTriangle(point, dir, mesh.verts[tri.v[0]], mesh.verts[tri.v[1]], mesh.verts[tri.v[2]], t)) {
            count++;
        }
    });
    return count;
}

bool isPointInMesh(const Vec3& point, const Mesh& mesh, const BVH& bvh) {
    Vec3 dirs[] = {
        Vec3(1, 0, 0),
        Vec3(0, 1, 0),
        Vec3(0, 0, 1),
        Vec3(0.577, 0.577, 0.577)
    };

    int votes = 0;
    for (auto& d : dirs) {
        if (countRayHits(point, d, mesh, bvh) % 2 == 1) votes++;
    }
    return votes >= 2;
}

int classifyTriangle(int ti, const Mesh& fromMesh, const Mesh& otherMesh, const BVH& otherBVH) {
    const Triangle& tri = fromMesh.tris[ti];
    Vec3 centroid = (fromMesh.verts[tri.v[0]] + fromMesh.verts[tri.v[1]] + fromMesh.verts[tri.v[2]]) / 3.0;
    Vec3 normal = fromMesh.getTriNormal(ti);

    Vec3 testPoint = centroid + normal * 1e-7;
    Vec3 testPoint2 = centroid - normal * 1e-7;

    bool inside1 = isPointInMesh(testPoint, otherMesh, otherBVH);
    bool inside2 = isPointInMesh(testPoint2, otherMesh, otherBVH);

    if (inside1 && inside2) return -1;
    if (!inside1 && !inside2) return 1;
    return 0;
}

double signedVolume(const Vec3& a, const Vec3& b, const Vec3& c, const Vec3& d) {
    return (b-a).cross(c-a).dot(d-a) / 6.0;
}

bool trianglesIntersect(const Vec3& a0, const Vec3& a1, const Vec3& a2,
                        const Vec3& b0, const Vec3& b1, const Vec3& b2) {
    auto sign = [](double x) { return x > EPS ? 1 : (x < -EPS ? -1 : 0); };

    int sa0 = sign(signedVolume(b0, b1, b2, a0));
    int sa1 = sign(signedVolume(b0, b1, b2, a1));
    int sa2 = sign(signedVolume(b0, b1, b2, a2));

    if (sa0 == sa1 && sa1 == sa2 && sa0 != 0) return false;

    int sb0 = sign(signedVolume(a0, a1, a2, b0));
    int sb1 = sign(signedVolume(a0, a1, a2, b1));
    int sb2 = sign(signedVolume(a0, a1, a2, b2));

    if (sb0 == sb1 && sb1 == sb2 && sb0 != 0) return false;

    if (sa0 == 0 && sa1 == 0 && sa2 == 0) {
        return true;
    }

    return true;
}

bool segmentIntersectsTriangle(const Vec3& p, const Vec3& q, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 n = (b-a).cross(c-a);
    double denom = n.dot(q - p);
    if (abs(denom) < EPS) return false;

    double t = n.dot(a - p) / denom;
    if (t < -EPS || t > 1.0 + EPS) return false;

    Vec3 hit = p + (q - p) * t;

    Vec3 v0 = c - a, v1 = b - a, v2 = hit - a;
    double d00 = v0.dot(v0), d01 = v0.dot(v1), d11 = v1.dot(v1);
    double d20 = v2.dot(v0), d21 = v2.dot(v1);
    double inv = d00*d11 - d01*d01;
    if (abs(inv) < EPS) return false;
    inv = 1.0 / inv;
    double u = (d11*d20 - d01*d21) * inv;
    double v = (d00*d21 - d01*d20) * inv;

    return u >= -EPS && v >= -EPS && u + v <= 1.0 + EPS;
}

bool trianglesIntersectProper(const Vec3& a0, const Vec3& a1, const Vec3& a2,
                              const Vec3& b0, const Vec3& b1, const Vec3& b2) {
    if (segmentIntersectsTriangle(a0, a1, b0, b1, b2)) return true;
    if (segmentIntersectsTriangle(a1, a2, b0, b1, b2)) return true;
    if (segmentIntersectsTriangle(a2, a0, b0, b1, b2)) return true;
    if (segmentIntersectsTriangle(b0, b1, a0, a1, a2)) return true;
    if (segmentIntersectsTriangle(b1, b2, a0, a1, a2)) return true;
    if (segmentIntersectsTriangle(b2, b0, a0, a1, a2)) return true;
    return false;
}

struct VertexKey {
    long long x, y, z;
    bool operator==(const VertexKey& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct VertexKeyHash {
    size_t operator()(const VertexKey& k) const {
        return hash<long long>()(k.x) ^ (hash<long long>()(k.y) << 1) ^ (hash<long long>()(k.z) << 2);
    }
};

VertexKey makeKey(const Vec3& v, double scale = 1e8) {
    return {(long long)round(v.x * scale), (long long)round(v.y * scale), (long long)round(v.z * scale)};
}

        meshA.tris[i].meshId = 0;

    }

    meshB.tris.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> meshB.tris[i].v[0] >> meshB.tris[i].v[1] >> meshB.tris[i].v[2];
        meshB.tris[i].meshId = 1;
    }

    BVH bvhA, bvhB;
    bvhA.build(&meshA);
    bvhB.build(&meshB);

    AABB boxA, boxB;
    for (auto& v : meshA.verts) boxA.expand(v);
    for (auto& v : meshB.verts) boxB.expand(v);

    if (!boxA.intersects(boxB)) {
        unordered_map<VertexKey, int, VertexKeyHash> vertMap;
        vector<Vec3> outVerts;
        vector<array<int,3>> outFaces;

        auto addVert = [&](const Vec3& v) -> int {
            VertexKey key = makeKey(v);
            auto it = vertMap.find(key);
            if (it != vertMap.end()) return it->second;
            int idx = (int)outVerts.size();
            outVerts.push_back(v);
            vertMap[key] = idx;
            return idx;
        };

        for (int i = 0; i < FA; i++) {
            int a = addVert(meshA.verts[meshA.tris[i].v[0]]);
            int b = addVert(meshA.verts[meshA.tris[i].v[1]]);
            int c = addVert(meshA.verts[meshA.tris[i].v[2]]);
            outFaces.push_back({a, b, c});
        }

        for (int i = 0; i < FB; i++) {
            int a = addVert(meshB.verts[meshB.tris[i].v[0]]);
            int b = addVert(meshB.verts[meshB.tris[i].v[1]]);
            int c = addVert(meshB.verts[meshB.tris[i].v[2]]);
            outFaces.push_back({a, b, c});
        }

        cout << outVerts.size() << " " << outFaces.size() << "\n";
        cout << fixed << setprecision(9);
        for (auto& v : outVerts) {
            cout << v.x << " " << v.y << " " << v.z << "\n";
        }
        for (auto& f : outFaces) {
            cout << f[0] << " " << f[1] << " " << f[2] << "\n";
        }
        return 0;
    }

    vector<bool> triAIntersects(FA, false);
    vector<bool> triBIntersects(FB, false);

    for (int ai = 0; ai < FA; ai++) {
        AABB boxTri = meshA.getTriAABB(ai);
        const Triangle& triA = meshA.tris[ai];
        const Vec3& a0 = meshA.verts[triA.v[0]];
        const Vec3& a1 = meshA.verts[triA.v[1]];
        const Vec3& a2 = meshA.verts[triA.v[2]];

        bvhB.queryBox(boxTri, [&](int bi) {
            const Triangle& triB = meshB.tris[bi];
            const Vec3& b0 = meshB.verts[triB.v[0]];
            const Vec3& b1 = meshB.verts[triB.v[1]];
            const Vec3& b2 = meshB.verts[triB.v[2]];

            if (trianglesIntersectProper(a0, a1, a2, b0, b1, b2)) {
                triAIntersects[ai] = true;
                triBIntersects[bi] = true;
            }
        });
    }

    unordered_map<VertexKey, int, VertexKeyHash> vertMap;
    vector<Vec3> outVerts;
    vector<array<int,3>> outFaces;

    auto addVert = [&](const Vec3& v) -> int {
        VertexKey key = makeKey(v);
        auto it = vertMap.find(key);
        if (it != vertMap.end()) return it->second;
        int idx = (int)outVerts.size();
        outVerts.push_back(v);
        vertMap[key] = idx;
        return idx;
    };

    for (int ai = 0; ai < FA; ai++) {
        int cl = classifyTriangle(ai, meshA, meshB, bvhB);
        if (cl >= 0) {
            const Triangle& tri = meshA.tris[ai];
            int a = addVert(meshA.verts[tri.v[0]]);
            int b = addVert(meshA.verts[tri.v[1]]);
            int c = addVert(meshA.verts[tri.v[2]]);
            outFaces.push_back({a, b, c});
        }
    }

    for (int bi = 0; bi < FB; bi++) {
        int cl = classifyTriangle(bi, meshB, meshA, bvhA);
        if (cl >= 0) {
            const Triangle& tri = meshB.tris[bi];
            int a = addVert(meshB.verts[tri.v[0]]);
            int b = addVert(meshB.verts[tri.v[1]]);
            int c = addVert(meshB.verts[tri.v[2]]);
            outFaces.push_back({a, b, c});
        }
    }

    if (outFaces.empty()) {
        bool AinsideB = true;
        for (int ai = 0; ai < FA && AinsideB; ai++) {
            Vec3 centroid = (meshA.verts[meshA.tris[ai].v[0]] +
                           meshA.verts[meshA.tris[ai].v[1]] +
                           meshA.verts[meshA.tris[ai].v[2]]) / 3.0;
            if (!isPointInMesh(centroid, meshB, bvhB)) {
                AinsideB = false;
            }
        }

        if (AinsideB) {
            for (int bi = 0; bi < FB; bi++) {
                const Triangle& tri = meshB.tris[bi];
                int a = addVert(meshB.verts[tri.v[0]]);
                int b = addVert(meshB.verts[tri.v[1]]);
                int c = addVert(meshB.verts[tri.v[2]]);
                outFaces.push_back({a, b, c});
            }
        } else {
            for (int ai = 0; ai < FA; ai++) {
                const Triangle& tri = meshA.tris[ai];
                int a = addVert(meshA.verts[tri.v[0]]);
                int b = addVert(meshA.verts[tri.v[1]]);
                int c = addVert(meshA.verts[tri.v[2]]);
                outFaces.push_back({a, b, c});
            }
        }
    }

    cout << outVerts.size() << " " << outFaces.size() << "\n";
    cout << fixed << setprecision(9);
    for (auto& v : outVerts) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (auto& f : outFaces) {
        cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }

    return 0;
}
