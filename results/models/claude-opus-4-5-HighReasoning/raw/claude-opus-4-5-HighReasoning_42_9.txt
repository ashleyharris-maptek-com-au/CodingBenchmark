; SPIR-V Geometry Shader - Thick Smooth Curve with Miter Joins
; Input: LinesAdjacency (4 vertices), Output: TriangleStrip (4 vertices)

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input/output color decorations
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Type declarations
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%thickness = OpConstant %float 0.025

; Input gl_PerVertex struct and array
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr = OpTypePointer Input %arr_gl_in
%ptr_in_v4float = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr Input

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_struct = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_struct Output

; Input color array (location 2)
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_4
%ptr_in_v3float = OpTypePointer Input %v3float
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load all 4 input positions
%ptr_pos0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %ptr_pos0
%ptr_pos1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %ptr_pos1
%ptr_pos2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %ptr_pos2
%ptr_pos3 = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %ptr_pos3

; Load colors for start (index 1) and end (index 2)
%ptr_col1 = OpAccessChain %ptr_in_v3float %inColor %int_1
%col1 = OpLoad %v3float %ptr_col1
%ptr_col2 = OpAccessChain %ptr_in_v3float %inColor %int_2
%col2 = OpLoad %v3float %ptr_col2

; Extract xy components for 2D calculations
%p0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%p1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%p2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%p3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

; Compute tangent at start: t0 = normalize(pos2 - pos0)
%dir0 = OpFSub %v2float %p2_xy %p0_xy
%len0_sq = OpDot %float %dir0 %dir0
%len0 = OpExtInst %float %1 Sqrt %len0_sq
%t0_x = OpCompositeExtract %float %dir0 0
%t0_y = OpCompositeExtract %float %dir0 1
%t0_len_inv = OpFDiv %float %float_1 %len0
%t0_norm_x = OpFMul %float %t0_x %t0_len_inv
%t0_norm_y = OpFMul %float %t0_y %t0_len_inv

; Compute tangent at end: t1 = normalize(pos3 - pos1)
%dir1 = OpFSub %v2float %p3_xy %p1_xy
%len1_sq = OpDot %float %dir1 %dir1
%len1 = OpExtInst %float %1 Sqrt %len1_sq
%t1_x = OpCompositeExtract %float %dir1 0
%t1_y = OpCompositeExtract %float %dir1 1
%t1_len_inv = OpFDiv %float %float_1 %len1
%t1_norm_x = OpFMul %float %t1_x %t1_len_inv
%t1_norm_y = OpFMul %float %t1_y %t1_len_inv

; Compute perpendiculars: perp = (-ty, tx)
%neg_t0_y = OpFNegate %float %t0_norm_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_norm_x
%neg_t1_y = OpFNegate %float %t1_norm_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_norm_x

; Scale perpendiculars by thickness
%offset0 = OpVectorTimesScalar %v2float %perp0 %thickness
%offset1 = OpVectorTimesScalar %v2float %perp1 %thickness

; Get z and w from start and end positions
%p1_z = OpCompositeExtract %float %pos1 2
%p1_w = OpCompositeExtract %float %pos1 3
%p2_z = OpCompositeExtract %float %pos2 2
%p2_w = OpCompositeExtract %float %pos2 3

; Output pointer
%out_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

; Vertex 1: start + perp0 * thickness
%v1_xy = OpFAdd %v2float %p1_xy %offset0
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %p1_w
OpStore %out_ptr %v1_pos
OpStore %outColor %col1
OpEmitVertex

; Vertex 2: start - perp0 * thickness
%v2_xy = OpFSub %v2float %p1_xy %offset0
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
OpStore %out_ptr %v2_pos
OpStore %outColor %col1
OpEmitVertex

; Vertex 3: end + perp1 * thickness
%v3_xy = OpFAdd %v2float %p2_xy %offset1
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %p2_z %p2_w
OpStore %out_ptr %v3_pos
OpStore %outColor %col2
OpEmitVertex

; Vertex 4: end - perp1 * thickness
%v4_xy = OpFSub %v2float %p2_xy %offset1
%v4_x = OpCompositeExtract %float %v4_xy 0
%v4_y = OpCompositeExtract %float %v4_xy 1
%v4_pos = OpCompositeConstruct %v4float %v4_x %v4_y %p2_z %p2_w
OpStore %out_ptr %v4_pos
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
