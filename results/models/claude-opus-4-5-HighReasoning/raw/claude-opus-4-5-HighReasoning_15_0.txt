use std::io::{self, BufRead, Write, BufWriter};

const TETRA_VERTS: [[f64; 3]; 4] = [
    [1.0, 0.0, -0.707107],
    [-1.0, 0.0, -0.707107],
    [0.0, 1.0, 0.707107],
    [0.0, -1.0, 0.707107],
];

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut polygon: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let mut iter = line.split_whitespace();
        let x: f64 = iter.next().unwrap().parse().unwrap();
        let y: f64 = iter.next().unwrap().parse().unwrap();
        polygon.push((x, y));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let mut sun_iter = sun_line.split_whitespace();
    let sx: f64 = sun_iter.next().unwrap().parse().unwrap();
    let sy: f64 = sun_iter.next().unwrap().parse().unwrap();
    let sz: f64 = sun_iter.next().unwrap().parse().unwrap();

    let results = solve(&polygon, sx, sy, sz);

    writeln!(out, "{}", results.len()).unwrap();
    for (px, py, pz, qw, qx, qy, qz, s) in &results {
        writeln!(out, "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
                 px, py, pz, qw, qx, qy, qz, s).unwrap();
    }
}

fn solve(polygon: &[(f64, f64)], sx: f64, sy: f64, sz: f64) -> Vec<(f64, f64, f64, f64, f64, f64, f64, f64)> {
    if polygon.len() < 3 {
        return vec![];
    }

    let (min_x, max_x, min_y, max_y) = polygon.iter().fold(
        (f64::INFINITY, f64::NEG_INFINITY, f64::INFINITY, f64::NEG_INFINITY),
        |(minx, maxx, miny, maxy), &(x, y)|
            (minx.min(x), maxx.max(x), miny.min(y), maxy.max(y))
    );

    let width = max_x - min_x;
    let height = max_y - min_y;
    let diag = (width.powi(2) + height.powi(2)).sqrt();

    let shadow_unit: Vec<(f64, f64)> = TETRA_VERTS.iter()
        .map(|v| (v[0] - v[2] * sx / sz, v[1] - v[2] * sy / sz))
        .collect();

    let inscribed_r = compute_inscribed_radius(&shadow_unit);
    let coverage_diameter = 2.0 * inscribed_r;

    if diag <= coverage_diameter * 10.0 * 0.9 {
        let scale = (diag / (coverage_diameter * 0.9)).max(0.5).min(10.0);
        let pz = scale * 2.0;
        let center_x = (min_x + max_x) / 2.0;
        let center_y = (min_y + max_y) / 2.0;
        let px = center_x + pz * sx / sz;
        let py = center_y + pz * sy / sz;
        return vec![(px, py, pz, 1.0, 0.0, 0.0, 0.0, scale)];
    }

    let scale = 10.0;
    let step = inscribed_r * scale * 1.8;

    let base_h = scale * 2.0;
    let h_step = scale * 3.0;
    let heights = [base_h, base_h + h_step, base_h + 2.0 * h_step, base_h + 3.0 * h_step];

    let mut results = Vec::new();
    let margin = step;
    let mut ix = 0i32;
    let mut x = min_x - margin;

    while x <= max_x + margin {
        let mut iy = 0i32;
        let mut y = min_y - margin;

        while y <= max_y + margin {
            let h_idx = ((ix.rem_euclid(2)) + 2 * (iy.rem_euclid(2))) as usize;
            let pz = heights[h_idx];
            let px = x + pz * sx / sz;
            let py = y + pz * sy / sz;

            let shadow_r = scale * 1.5;
            if x + shadow_r >= min_x && x - shadow_r <= max_x &&
               y + shadow_r >= min_y && y - shadow_r <= max_y {
                results.push((px, py, pz, 1.0, 0.0, 0.0, 0.0, scale));
            }

            y += step;
            iy += 1;
        }
        x += step;
        ix += 1;
    }

    results
}

fn compute_inscribed_radius(quad: &[(f64, f64)]) -> f64 {
    let n = quad.len();
    let mut min_dist = f64::INFINITY;

    for i in 0..n {
        let j = (i + 1) % n;
        let (ax, ay) = quad[i];
        let (bx, by) = quad[j];
        let len = ((bx - ax).powi(2) + (by - ay).powi(2)).sqrt();
        if len < 1e-10 { continue; }
        let dist = ((by - ay) * 0.0 - (bx - ax) * 0.0 + bx * ay - by * ax).abs() / len;
        min_dist = min_dist.min(dist);
    }

    min_dist.max(0.1)
}
