; SPIR-V Geometry Shader - Flat shading with triangles adjacency
; Capability and memory model
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
   %void_fun = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

; Constants
    %int_0 = OpConstant %int 0
    %int_2 = OpConstant %int 2
    %int_4 = OpConstant %int 4
   %uint_6 = OpConstant %uint 6
  %float_0 = OpConstant %float 0
  %float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_333 = OpConstant %float 0.333333343
%float_0_577 = OpConstant %float 0.577350259

; Light direction normalized (1,1,1)/sqrt(3)
%light_dir = OpConstantComposite %v3float %float_0_577 %float_0_577 %float_0_577

; Input gl_PerVertex block (array of 6 for triangles adjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_6
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Input color array (location 2)
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main function
       %main = OpFunction %void None %void_fun
      %entry = OpLabel

; Load positions p0, p2, p4 (main triangle vertices)
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %ptr_p0
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2_v4 = OpLoad %v4float %ptr_p2
%ptr_p4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4_v4 = OpLoad %v4float %ptr_p4

; Extract xyz for normal calculation
%p0_xyz = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
%p4_xyz = OpVectorShuffle %v3float %p4_v4 %p4_v4 0 1 2

; Compute edge vectors: e1 = p2 - p0, e2 = p4 - p0
%e1 = OpFSub %v3float %p2_xyz %p0_xyz
%e2 = OpFSub %v3float %p4_xyz %p0_xyz

; Cross product: N = e1 × e2
; (a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x)
%e1_x = OpCompositeExtract %float %e1 0
%e1_y = OpCompositeExtract %float %e1 1
%e1_z = OpCompositeExtract %float %e1 2
%e2_x = OpCompositeExtract %float %e2 0
%e2_y = OpCompositeExtract %float %e2 1
%e2_z = OpCompositeExtract %float %e2 2

%cross_x_a = OpFMul %float %e1_y %e2_z
%cross_x_b = OpFMul %float %e1_z %e2_y
%cross_x = OpFSub %float %cross_x_a %cross_x_b

%cross_y_a = OpFMul %float %e1_z %e2_x
%cross_y_b = OpFMul %float %e1_x %e2_z
%cross_y = OpFSub %float %cross_y_a %cross_y_b

%cross_z_a = OpFMul %float %e1_x %e2_y
%cross_z_b = OpFMul %float %e1_y %e2_x
%cross_z = OpFSub %float %cross_z_a %cross_z_b

%cross_vec = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

; Normalize the normal
%normal = OpExtInst %v3float %1 Normalize %cross_vec

; Compute diffuse: max(dot(N, L), 0)
%dot_nl = OpDot %float %normal %light_dir
%diffuse = OpExtInst %float %1 FMax %dot_nl %float_0

; Load colors c0, c2, c4
%ptr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c2 = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %ptr_c2
%ptr_c4 = OpAccessChain %ptr_in_v3 %inColor %int_4
%c4 = OpLoad %v3float %ptr_c4

; Base color = average(c0, c2, c4)
%c_sum_02 = OpFAdd %v3float %c0 %c2
%c_sum_024 = OpFAdd %v3float %c_sum_02 %c4
%scale_vec = OpCompositeConstruct %v3float %float_0_333 %float_0_333 %float_0_333
%base_color = OpFMul %v3float %c_sum_024 %scale_vec

; Final color = base * (0.2 + 0.8 * diffuse)
%scaled_diff = OpFMul %float %float_0_8 %diffuse
%light_factor = OpFAdd %float %float_0_2 %scaled_diff
%light_vec = OpCompositeConstruct %v3float %light_factor %light_factor %light_factor
%final_color = OpFMul %v3float %base_color %light_vec

; Emit vertex 0
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr %p0_v4
               OpStore %outColor %final_color
               OpEmitVertex

; Emit vertex 2
               OpStore %out_pos_ptr %p2_v4
               OpStore %outColor %final_color
               OpEmitVertex

; Emit vertex 4
               OpStore %out_pos_ptr %p4_v4
               OpStore %outColor %final_color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
