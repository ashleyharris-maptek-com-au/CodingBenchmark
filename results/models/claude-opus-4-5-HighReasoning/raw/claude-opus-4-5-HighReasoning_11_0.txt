use std::io::{self, BufRead, Write, BufWriter};

type Point = (f64, f64);

fn parse_line<T: std::str::FromStr>(line: &str) -> Vec<T>
where T::Err: std::fmt::Debug {
    line.split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect()
}

fn read_polygon(lines: &mut impl Iterator<Item = String>, n: usize) -> Vec<Point> {
    (0..n).map(|_| {
        let line = lines.next().unwrap();
        let coords: Vec<f64> = parse_line(&line);
        (coords[0], coords[1])
    }).collect()
}

fn polygon_area(poly: &[Point]) -> f64 {
    let n = poly.len();
    let mut area = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        area += poly[i].0 * poly[j].1;
        area -= poly[j].0 * poly[i].1;
    }
    area.abs() / 2.0
}

fn bounding_box(poly: &[Point]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut min_y = f64::MAX;
    let mut max_x = f64::MIN;
    let mut max_y = f64::MIN;
    for &(x, y) in poly {
        min_x = min_x.min(x);
        min_y = min_y.min(y);
        max_x = max_x.max(x);
        max_y = max_y.max(y);
    }
    (min_x, min_y, max_x, max_y)
}

fn rotate_polygon(poly: &[Point], angle: f64) -> Vec<Point> {
    let cos_a = angle.cos();
    let sin_a = angle.sin();
    poly.iter().map(|&(x, y)| {
        (x * cos_a - y * sin_a, x * sin_a + y * cos_a)
    }).collect()
}

fn translate_polygon(poly: &[Point], dx: f64, dy: f64) -> Vec<Point> {
    poly.iter().map(|&(x, y)| (x + dx, y + dy)).collect()
}

fn point_in_polygon(point: Point, poly: &[Point]) -> bool {
    let (px, py) = point;
    let n = poly.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn point_on_edge(point: Point, poly: &[Point], eps: f64) -> bool {
    let (px, py) = point;
    let n = poly.len();
    for i in 0..n {
        let (x1, y1) = poly[i];
        let (x2, y2) = poly[(i + 1) % n];
        let dx = x2 - x1;
        let dy = y2 - y1;
        let len_sq = dx * dx + dy * dy;
        if len_sq < 1e-12 { continue; }
        let t = ((px - x1) * dx + (py - y1) * dy) / len_sq;
        if t >= -eps && t <= 1.0 + eps {
            let closest_x = x1 + t.clamp(0.0, 1.0) * dx;
            let closest_y = y1 + t.clamp(0.0, 1.0) * dy;
            let dist_sq = (px - closest_x).powi(2) + (py - closest_y).powi(2);
            if dist_sq < eps * eps {
                return true;
            }
        }
    }
    false
}

fn polygon_inside_stock(piece: &[Point], stock: &[Point]) -> bool {
    let eps = 1e-6;
    for &p in piece {
        if !point_in_polygon(p, stock) && !point_on_edge(p, stock, eps) {
            return false;
        }
    }
    true
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    fn cross(o: Point, a: Point, b: Point) -> f64 {
        (a.0 - o.0) * (b.1 - o.1) - (a.1 - o.1) * (b.0 - o.0)
    }
    fn on_segment(p: Point, q: Point, r: Point) -> bool {
        q.0 <= p.0.max(r.0) && q.0 >= p.0.min(r.0) &&
        q.1 <= p.1.max(r.1) && q.1 >= p.1.min(r.1)
    }
    let d1 = cross(b1, b2, a1);
    let d2 = cross(b1, b2, a2);
    let d3 = cross(a1, a2, b1);
    let d4 = cross(a1, a2, b2);

    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }
    let eps = 1e-9;
    if d1.abs() < eps && on_segment(b1, a1, b2) { return true; }
    if d2.abs() < eps && on_segment(b1, a2, b2) { return true; }
    if d3.abs() < eps && on_segment(a1, b1, a2) { return true; }
    if d4.abs() < eps && on_segment(a1, b2, a2) { return true; }
    false
}

fn polygons_overlap(p1: &[Point], p2: &[Point]) -> bool {
    let n1 = p1.len();
    let n2 = p2.len();
    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(p1[i], p1[(i+1)%n1], p2[j], p2[(j+1)%n2]) {
                return true;
            }
        }
    }
    let centroid1 = (
        p1.iter().map(|p| p.0).sum::<f64>() / n1 as f64,
        p1.iter().map(|p| p.1).sum::<f64>() / n1 as f64
    );
    if point_in_polygon(centroid1, p2) {
        return true;
    }
    let centroid2 = (
        p2.iter().map(|p| p.0).sum::<f64>() / n2 as f64,
        p2.iter().map(|p| p.1).sum::<f64>() / n2 as f64
    );
    if point_in_polygon(centroid2, p1) {
        return true;
    }
    false
}

struct Stock {
    placed: Vec<Vec<Point>>,
}

impl Stock {
    fn new() -> Self {
        Stock { placed: Vec::new() }
    }

    fn can_place(&self, piece: &[Point], stock_poly: &[Point]) -> bool {
        if !polygon_inside_stock(piece, stock_poly) {
            return false;
        }
        for existing in &self.placed {
            if polygons_overlap(piece, existing) {
                return false;
            }
        }
        true
    }

    fn place(&mut self, piece: Vec<Point>) {
        self.placed.push(piece);
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    let first_line = lines.next().unwrap();
    let params: Vec<usize> = parse_line(&first_line);
    let sv = params[0];
    let np = params[1];

    let stock_poly = read_polygon(&mut lines, sv);
    let stock_bb = bounding_box(&stock_poly);
    let stock_w = stock_bb.2 - stock_bb.0;
    let stock_h = stock_bb.3 - stock_bb.1;

    let mut pieces: Vec<(usize, Vec<Point>)> = Vec::with_capacity(np);
    for i in 0..np {
        let pv_line = lines.next().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        let poly = read_polygon(&mut lines, pv);
        pieces.push((i, poly));
    }

    pieces.sort_by(|a, b| {
        let area_a = polygon_area(&a.1);
        let area_b = polygon_area(&b.1);
        area_b.partial_cmp(&area_a).unwrap()
    });

    let mut stocks: Vec<Stock> = Vec::new();
    let mut placements: Vec<(usize, usize, f64, f64, f64)> = Vec::with_capacity(np);

    let rotations: &[f64] = if np > 1000 {
        &[0.0, std::f64::consts::FRAC_PI_2]
    } else if np > 100 {
        &[0.0, std::f64::consts::FRAC_PI_4, std::f64::consts::FRAC_PI_2, 3.0 * std::f64::consts::FRAC_PI_4]
    } else {
        &[0.0, std::f64::consts::FRAC_PI_6, std::f64::consts::FRAC_PI_4, std::f64::consts::FRAC_PI_3,
          std::f64::consts::FRAC_PI_2, 2.0 * std::f64::consts::FRAC_PI_3, 3.0 * std::f64::consts::FRAC_PI_4,
          5.0 * std::f64::consts::FRAC_PI_6]
    };

    let grid_step = if np > 1000 {
        (stock_w.max(stock_h) / 10.0).max(1.0)
    } else if np > 100 {
        (stock_w.max(stock_h) / 20.0).max(0.5)
    } else {
        (stock_w.max(stock_h) / 40.0).max(0.25)
    };

    for (orig_idx, piece) in pieces {
        let mut placed = false;

        'stock_loop: for (stock_idx, stock) in stocks.iter_mut().enumerate() {
            for &rot in rotations {
                let rotated = rotate_polygon(&piece, rot);
                let bb = bounding_box(&rotated);

                let mut y = stock_bb.1;
                while y <= stock_bb.3 - (bb.3 - bb.1) + 0.01 {
                    let mut x = stock_bb.0;
                    while x <= stock_bb.2 - (bb.2 - bb.0) + 0.01 {
                        let dx = x - bb.0;
                        let dy = y - bb.1;
                        let translated = translate_polygon(&rotated, dx, dy);

                        if stock.can_place(&translated, &stock_poly) {
                            stock.place(translated);
                            placements.push((orig_idx, stock_idx, dx, dy, rot));
                            placed = true;
                            break 'stock_loop;
                        }
                        x += grid_step;
                    }
                    y += grid_step;
                }
            }
        }

        if !placed {
            let stock_idx = stocks.len();
            let mut new_stock = Stock::new();

            'rot_loop: for &rot in rotations {
                let rotated = rotate_polygon(&piece, rot);
                let bb = bounding_box(&rotated);

                let mut y = stock_bb.1;
                while y <= stock_bb.3 - (bb.3 - bb.1) + 0.01 {
                    let mut x = stock_bb.0;
                    while x <= stock_bb.2 - (bb.2 - bb.0) + 0.01 {
                        let dx = x - bb.0;
                        let dy = y - bb.1;
                        let translated = translate_polygon(&rotated, dx, dy);

                        if new_stock.can_place(&translated, &stock_poly) {
                            new_stock.place(translated);
                            placements.push((orig_idx, stock_idx, dx, dy, rot));
                            placed = true;
                            break 'rot_loop;
                        }
                        x += grid_step;
                    }
                    y += grid_step;
                }
            }

            if !placed {
                let bb = bounding_box(&piece);
                let dx = stock_bb.0 - bb.0;
                let dy = stock_bb.1 - bb.1;
                let translated = translate_polygon(&piece, dx, dy);
                new_stock.place(translated);
                placements.push((orig_idx, stock_idx, dx, dy, 0.0));
            }

            stocks.push(new_stock);
        }
    }

    placements.sort_by_key(|p| p.0);

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", stocks.len()).unwrap();
    for (_, stock_idx, dx, dy, rot) in &placements {
        writeln!(out, "{} {} {} {}", stock_idx, dx, dy, rot).unwrap();
    }
}
