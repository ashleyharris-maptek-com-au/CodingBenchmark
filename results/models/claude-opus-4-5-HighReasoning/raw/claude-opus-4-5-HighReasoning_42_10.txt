; SPIR-V Geometry Shader - Thick Smooth Curve with Catmull-Rom Interpolation
; Input: LinesAdjacency (4 vertices), Output: TriangleStrip (10 vertices)

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void = OpTypeVoid
  %voidfunc = OpTypeFunction %void
     %float = OpTypeFloat 32
   %v4float = OpTypeVector %float 4
   %v3float = OpTypeVector %float 3
   %v2float = OpTypeVector %float 2
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0

               ; Constants
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_2 = OpConstant %int 2
    %int_3 = OpConstant %int 3
   %uint_4 = OpConstant %uint 4
  %float_0 = OpConstant %float 0.0
  %float_1 = OpConstant %float 1.0
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_0_02 = OpConstant %float 0.02
%float_n0_02 = OpConstant %float -0.02

               ; Struct types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_4
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

               ; Color array type
%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

               ; Variables
    %gl_in = OpVariable %ptr_in_arr_pv Input
   %gl_out = OpVariable %ptr_out_pv Output
  %inColor = OpVariable %ptr_in_arr_v3 Input
 %outColor = OpVariable %ptr_out_v3 Output

               ; Main function
      %main = OpFunction %void None %voidfunc
     %entry = OpLabel

               ; Load positions P0, P1, P2, P3
   %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %p0 = OpLoad %v4float %p0_ptr
   %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %p1 = OpLoad %v4float %p1_ptr
   %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %p2 = OpLoad %v4float %p2_ptr
   %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
       %p3 = OpLoad %v4float %p3_ptr

               ; Load colors at indices 1 and 2
  %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
      %c1 = OpLoad %v3float %c1_ptr
  %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
      %c2 = OpLoad %v3float %c2_ptr

               ; Compute tangent directions
               ; tan_start = P2 - P0, tan_end = P3 - P1
%tan_start = OpFSub %v4float %p2 %p0
  %tan_end = OpFSub %v4float %p3 %p1

               ; Output pointer for gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; ========== Point 0 (t=0.0) at P1 ==========
               ; Perpendicular from tan_start
 %ts_x_0 = OpCompositeExtract %float %tan_start 0
 %ts_y_0 = OpCompositeExtract %float %tan_start 1
%perp_x_0 = OpFNegate %float %ts_y_0
%perp_y_0 = OpCopyObject %float %ts_x_0
               ; Normalize perpendicular
%len2_0 = OpFMul %float %perp_x_0 %perp_x_0
%tmp_0 = OpFMul %float %perp_y_0 %perp_y_0
%len2_0b = OpFAdd %float %len2_0 %tmp_0
%len_0 = OpExtInst %float %1 Sqrt %len2_0b
%inv_len_0 = OpFDiv %float %float_1 %len_0
%norm_px_0 = OpFMul %float %perp_x_0 %inv_len_0
%norm_py_0 = OpFMul %float %perp_y_0 %inv_len_0
               ; Scale by thickness
%off_x_0 = OpFMul %float %norm_px_0 %float_0_02
%off_y_0 = OpFMul %float %norm_py_0 %float_0_02
               ; pos + perp
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%pos0_px = OpFAdd %float %p1_x %off_x_0
%pos0_py = OpFAdd %float %p1_y %off_y_0
%pos0_p = OpCompositeConstruct %v4float %pos0_px %pos0_py %p1_z %p1_w
               OpStore %out_pos_ptr %pos0_p
               OpStore %outColor %c1
               OpEmitVertex
               ; pos - perp
%pos0_mx = OpFSub %float %p1_x %off_x_0
%pos0_my = OpFSub %float %p1_y %off_y_0
%pos0_m = OpCompositeConstruct %v4float %pos0_mx %pos0_my %p1_z %p1_w
               OpStore %out_pos_ptr %pos0_m
               OpStore %outColor %c1
               OpEmitVertex

               ; ========== Point 1 (t=0.25) ==========
               ; Lerp position: pos = P1 + 0.25*(P2-P1)
%diff_1 = OpFSub %v4float %p2 %p1
%scaled_1 = OpVectorTimesScalar %v4float %diff_1 %float_0_25
%pos_1 = OpFAdd %v4float %p1 %scaled_1
               ; Lerp tangent: tan = tan_start + 0.25*(tan_end - tan_start)
%tan_diff_1 = OpFSub %v4float %tan_end %tan_start
%tan_sc_1 = OpVectorTimesScalar %v4float %tan_diff_1 %float_0_25
%tan_1 = OpFAdd %v4float %tan_start %tan_sc_1
               ; Perpendicular
%t1_x = OpCompositeExtract %float %tan_1 0
%t1_y = OpCompositeExtract %float %tan_1 1
%perp_x_1 = OpFNegate %float %t1_y
%perp_y_1 = OpCopyObject %float %t1_x
%len2_1 = OpFMul %float %perp_x_1 %perp_x_1
%tmp_1 = OpFMul %float %perp_y_1 %perp_y_1
%len2_1b = OpFAdd %float %len2_1 %tmp_1
%len_1 = OpExtInst %float %1 Sqrt %len2_1b
%inv_len_1 = OpFDiv %float %float_1 %len_1
%norm_px_1 = OpFMul %float %perp_x_1 %inv_len_1
%norm_py_1 = OpFMul %float %perp_y_1 %inv_len_1
%off_x_1 = OpFMul %float %norm_px_1 %float_0_02
%off_y_1 = OpFMul %float %norm_py_1 %float_0_02
               ; Lerp color
%col_diff_1 = OpFSub %v3float %c2 %c1
%col_sc_1 = OpVectorTimesScalar %v3float %col_diff_1 %float_0_25
%col_1 = OpFAdd %v3float %c1 %col_sc_1
               ; Extract position components
%p1_x_1 = OpCompositeExtract %float %pos_1 0
%p1_y_1 = OpCompositeExtract %float %pos_1 1
%p1_z_1 = OpCompositeExtract %float %pos_1 2
%p1_w_1 = OpCompositeExtract %float %pos_1 3
               ; pos + perp
%pos1_px = OpFAdd %float %p1_x_1 %off_x_1
%pos1_py = OpFAdd %float %p1_y_1 %off_y_1
%pos1_p = OpCompositeConstruct %v4float %pos1_px %pos1_py %p1_z_1 %p1_w_1
               OpStore %out_pos_ptr %pos1_p
               OpStore %outColor %col_1
               OpEmitVertex
               ; pos - perp
%pos1_mx = OpFSub %float %p1_x_1 %off_x_1
%pos1_my = OpFSub %float %p1_y_1 %off_y_1
%pos1_m = OpCompositeConstruct %v4float %pos1_mx %pos1_my %p1_z_1 %p1_w_1
               OpStore %out_pos_ptr %pos1_m
               OpStore %outColor %col_1
               OpEmitVertex

               ; ========== Point 2 (t=0.5) ==========
%scaled_2 = OpVectorTimesScalar %v4float %diff_1 %float_0_5
%pos_2 = OpFAdd %v4float %p1 %scaled_2
%tan_sc_2 = OpVectorTimesScalar %v4float %tan_diff_1 %float_0_5
%tan_2 = OpFAdd %v4float %tan_start %tan_sc_2
%t2_x = OpCompositeExtract %float %tan_2 0
%t2_y = OpCompositeExtract %float %tan_2 1
%perp_x_2 = OpFNegate %float %t2_y
%perp_y_2 = OpCopyObject %float %t2_x
%len2_2 = OpFMul %float %perp_x_2 %perp_x_2
%tmp_2 = OpFMul %float %perp_y_2 %perp_y_2
%len2_2b = OpFAdd %float %len2_2 %tmp_2
%len_2 = OpExtInst %float %1 Sqrt %len2_2b
%inv_len_2 = OpFDiv %float %float_1 %len_2
%norm_px_2 = OpFMul %float %perp_x_2 %inv_len_2
%norm_py_2 = OpFMul %float %perp_y_2 %inv_len_2
%off_x_2 = OpFMul %float %norm_px_2 %float_0_02
%off_y_2 = OpFMul %float %norm_py_2 %float_0_02
%col_sc_2 = OpVectorTimesScalar %v3float %col_diff_1 %float_0_5
%col_2 = OpFAdd %v3float %c1 %col_sc_2
%p2_x_2 = OpCompositeExtract %float %pos_2 0
%p2_y_2 = OpCompositeExtract %float %pos_2 1
%p2_z_2 = OpCompositeExtract %float %pos_2 2
%p2_w_2 = OpCompositeExtract %float %pos_2 3
%pos2_px = OpFAdd %float %p2_x_2 %off_x_2
%pos2_py = OpFAdd %float %p2_y_2 %off_y_2
%pos2_p = OpCompositeConstruct %v4float %pos2_px %pos2_py %p2_z_2 %p2_w_2
               OpStore %out_pos_ptr %pos2_p
               OpStore %outColor %col_2
               OpEmitVertex
%pos2_mx = OpFSub %float %p2_x_2 %off_x_2
%pos2_my = OpFSub %float %p2_y_2 %off_y_2
%pos2_m = OpCompositeConstruct %v4float %pos2_mx %pos2_my %p2_z_2 %p2_w_2
               OpStore %out_pos_ptr %pos2_m
               OpStore %outColor %col_2
               OpEmitVertex

               ; ========== Point 3 (t=0.75) ==========
%scaled_3 = OpVectorTimesScalar %v4float %diff_1 %float_0_75
%pos_3 = OpFAdd %v4float %p1 %scaled_3
%tan_sc_3 = OpVectorTimesScalar %v4float %tan_diff_1 %float_0_75
%tan_3 = OpFAdd %v4float %tan_start %tan_sc_3
%t3_x = OpCompositeExtract %float %tan_3 0
%t3_y = OpCompositeExtract %float %tan_3 1
%perp_x_3 = OpFNegate %float %t3_y
%perp_y_3 = OpCopyObject %float %t3_x
%len2_3 = OpFMul %float %perp_x_3 %perp_x_3
%tmp_3 = OpFMul %float %perp_y_3 %perp_y_3
%len2_3b = OpFAdd %float %len2_3 %tmp_3
%len_3 = OpExtInst %float %1 Sqrt %len2_3b
%inv_len_3 = OpFDiv %float %float_1 %len_3
%norm_px_3 = OpFMul %float %perp_x_3 %inv_len_3
%norm_py_3 = OpFMul %float %perp_y_3 %inv_len_3
%off_x_3 = OpFMul %float %norm_px_3 %float_0_02
%off_y_3 = OpFMul %float %norm_py_3 %float_0_02
%col_sc_3 = OpVectorTimesScalar %v3float %col_diff_1 %float_0_75
%col_3 = OpFAdd %v3float %c1 %col_sc_3
%p3_x_3 = OpCompositeExtract %float %pos_3 0
%p3_y_3 = OpCompositeExtract %float %pos_3 1
%p3_z_3 = OpCompositeExtract %float %pos_3 2
%p3_w_3 = OpCompositeExtract %float %pos_3 3
%pos3_px = OpFAdd %float %p3_x_3 %off_x_3
%pos3_py = OpFAdd %float %p3_y_3 %off_y_3
%pos3_p = OpCompositeConstruct %v4float %pos3_px %pos3_py %p3_z_3 %p3_w_3
               OpStore %out_pos_ptr %pos3_p
               OpStore %outColor %col_3
               OpEmitVertex
%pos3_mx = OpFSub %float %p3_x_3 %off_x_3
%pos3_my = OpFSub %float %p3_y_3 %off_y_3
%pos3_m = OpCompositeConstruct %v4float %pos3_mx %pos3_my %p3_z_3 %p3_w_3
               OpStore %out_pos_ptr %pos3_m
               OpStore %outColor %col_3
               OpEmitVertex

               ; ========== Point 4 (t=1.0) at P2 ==========
               ; Perpendicular from tan_end
%te_x = OpCompositeExtract %float %tan_end 0
%te_y = OpCompositeExtract %float %tan_end 1
%perp_x_4 = OpFNegate %float %te_y
%perp_y_4 = OpCopyObject %float %te_x
%len2_4 = OpFMul %float %perp_x_4 %perp_x_4
%tmp_4 = OpFMul %float %perp_y_4 %perp_y_4
%len2_4b = OpFAdd %float %len2_4 %tmp_4
%len_4 = OpExtInst %float %1 Sqrt %len2_4b
%inv_len_4 = OpFDiv %float %float_1 %len_4
%norm_px_4 = OpFMul %float %perp_x_4 %inv_len_4
%norm_py_4 = OpFMul %float %perp_y_4 %inv_len_4
%off_x_4 = OpFMul %float %norm_px_4 %float_0_02
%off_y_4 = OpFMul %float %norm_py_4 %float_0_02
%p2_x = OpCompositeExtract %float %p2 0
%p2_y = OpCompositeExtract %float %p2 1
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3
%pos4_px = OpFAdd %float %p2_x %off_x_4
%pos4_py = OpFAdd %float %p2_y %off_y_4
%pos4_p = OpCompositeConstruct %v4float %pos4_px %pos4_py %p2_z %p2_w
               OpStore %out_pos_ptr %pos4_p
               OpStore %outColor %c2
               OpEmitVertex
%pos4_mx = OpFSub %float %p2_x %off_x_4
%pos4_my = OpFSub %float %p2_y %off_y_4
%pos4_m = OpCompositeConstruct %v4float %pos4_mx %pos4_my %p2_z %p2_w
               OpStore %out_pos_ptr %pos4_m
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
