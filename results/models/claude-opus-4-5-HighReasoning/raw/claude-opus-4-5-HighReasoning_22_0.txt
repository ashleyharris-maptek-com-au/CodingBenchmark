using System;
using System.Collections.Generic;

class TetrisPlayer
{
    static int width, height;
    static int[,] board;

    static readonly Dictionary<char, int[][][]> Pieces = new Dictionary<char, int[][][]>
    {
        ['I'] = new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {3,0} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {0,3} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {3,0} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {0,3} }
        },
        ['O'] = new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {0,1}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {0,1}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {0,1}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {0,1}, new int[] {1,1} }
        },
        ['T'] = new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {0,2} },
            new int[][] { new int[] {1,0}, new int[] {0,1}, new int[] {1,1}, new int[] {2,1} },
            new int[][] { new int[] {1,0}, new int[] {0,1}, new int[] {1,1}, new int[] {1,2} }
        },
        ['S'] = new int[][][] {
            new int[][] { new int[] {1,0}, new int[] {2,0}, new int[] {0,1}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {1,2} },
            new int[][] { new int[] {1,0}, new int[] {2,0}, new int[] {0,1}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {1,2} }
        },
        ['Z'] = new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {1,1}, new int[] {2,1} },
            new int[][] { new int[] {1,0}, new int[] {0,1}, new int[] {1,1}, new int[] {0,2} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {1,1}, new int[] {2,1} },
            new int[][] { new int[] {1,0}, new int[] {0,1}, new int[] {1,1}, new int[] {0,2} }
        },
        ['J'] = new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {2,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {0,1}, new int[] {0,2} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {2,1} },
            new int[][] { new int[] {1,0}, new int[] {1,1}, new int[] {0,2}, new int[] {1,2} }
        },
        ['L'] = new int[][][] {
            new int[][] { new int[] {2,0}, new int[] {0,1}, new int[] {1,1}, new int[] {2,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {1,2} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {0,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {1,1}, new int[] {1,2} }
        }
    };

    static void Main()
    {
        string[] header = Console.ReadLine().Split(new char[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        width = int.Parse(header[0]);
        height = int.Parse(header[1]);
        int numPieces = int.Parse(header[2]);

        board = new int[width, height];

        for (int i = 0; i < numPieces; i++)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) { i--; continue; }
            char pieceType = line.Trim()[0];

            var (bestRot, bestCol) = FindBestPlacement(pieceType);
            Console.WriteLine($"{bestRot} {bestCol}");
            Console.Out.Flush();

            string result = Console.ReadLine();
            if (result != null && !result.Contains("gameover"))
            {
                PlacePiece(pieceType, bestRot, bestCol);
                if (result.Contains("cleared"))
                {
                    ClearLines();
                }
            }
            else if (result != null && result.Contains("gameover"))
            {
                break;
            }
        }
    }

    static (int rot, int col) FindBestPlacement(char pieceType)
    {
        int bestRot = 0, bestCol = 0;
        double bestScore = double.MinValue;

        int[][] shape = Pieces[pieceType][0];
        int uniqueRotations = GetUniqueRotations(pieceType);

        for (int rot = 0; rot < uniqueRotations; rot++)
        {
            shape = Pieces[pieceType][rot];
            int pieceWidth = GetPieceWidth(shape);
            int pieceHeight = GetPieceHeight(shape);

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                int dropRow = GetDropRow(shape, col);
                if (dropRow < 0) continue;

                double score = EvaluatePlacement(shape, col, dropRow);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }

        return (bestRot, bestCol);
    }

    static int GetUniqueRotations(char pieceType)
    {
        return pieceType switch
        {
            'O' => 1,
            'I' or 'S' or 'Z' => 2,
            _ => 4
        };
    }

    static int GetPieceWidth(int[][] shape)
    {
        int maxX = 0;
        foreach (var cell in shape) maxX = Math.Max(maxX, cell[0]);
        return maxX + 1;
    }

    static int GetPieceHeight(int[][] shape)
    {
        int maxY = 0;
        foreach (var cell in shape) maxY = Math.Max(maxY, cell[1]);
        return maxY + 1;
    }

    static int GetDropRow(int[][] shape, int col)
    {
        for (int row = height - GetPieceHeight(shape); row >= 0; row--)
        {
            if (CanPlace(shape, col, row))
            {
                return row;
            }
        }
        return -1;
    }

    static bool CanPlace(int[][] shape, int col, int row)
    {
        foreach (var cell in shape)
        {
            int x = col + cell[0];
            int y = row + cell[1];
            if (x < 0 || x >= width || y < 0 || y >= height) return false;
            if (board[x, y] != 0) return false;
        }
        return true;
    }

    static double EvaluatePlacement(int[][] shape, int col, int row)
    {
        int[,] testBoard = (int[,])board.Clone();
        foreach (var cell in shape)
        {
            testBoard[col + cell[0], row + cell[1]] = 1;
        }

        int linesCleared = CountClearedLines(testBoard);
        int holes = CountHoles(testBoard);
        int aggregateHeight = GetAggregateHeight(testBoard);
        int bumpiness = GetBumpiness(testBoard);
        int maxHeight = GetMaxHeight(testBoard);

        double score = linesCleared * 1000.0
                     - holes * 500.0
                     - aggregateHeight * 5.0
                     - bumpiness * 10.0
                     - maxHeight * 20.0;

        if (maxHeight > height - 4) score -= (maxHeight - (height - 4)) * 200;

        return score;
    }

    static int CountClearedLines(int[,] b)
    {
        int cleared = 0;
        for (int y = 0; y < height; y++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (b[x, y] == 0) { full = false; break; }
            }
            if (full) cleared++;
        }
        return cleared;
    }

    static int CountHoles(int[,] b)
    {
        int holes = 0;
        for (int x = 0; x < width; x++)
        {
            bool foundBlock = false;
            for (int y = 0; y < height; y++)
            {
                if (b[x, y] != 0) foundBlock = true;
                else if (foundBlock) holes++;
            }
        }
        return holes;
    }

    static int GetAggregateHeight(int[,] b)
    {
        int total = 0;
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (b[x, y] != 0)
                {
                    total += height - y;
                    break;
                }
            }
        }
        return total;
    }

    static int GetBumpiness(int[,] b)
    {
        int[] heights = new int[width];
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (b[x, y] != 0)
                {
                    heights[x] = height - y;
                    break;
                }
            }
        }
        int bumpiness = 0;
        for (int x = 0; x < width - 1; x++)
        {
            bumpiness += Math.Abs(heights[x] - heights[x + 1]);
        }
        return bumpiness;
    }

    static int GetMaxHeight(int[,] b)
    {
        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                if (b[x, y] != 0) return height - y;
            }
        }
        return 0;
    }

    static void PlacePiece(char pieceType, int rot, int col)
    {
        int[][] shape = Pieces[pieceType][rot];
        int row = GetDropRow(shape, col);
        if (row >= 0)
        {
            foreach (var cell in shape)
            {
                board[col + cell[0], row + cell[1]] = 1;
            }
        }
    }

    static void ClearLines()
    {
        for (int y = height - 1; y >= 0; y--)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (board[x, y] == 0) { full = false; break; }
            }
            if (full)
            {
                for (int y2 = y; y2 > 0; y2--)
                {
                    for (int x = 0; x < width; x++)
                    {
                        board[x, y2] = board[x, y2 - 1];
                    }
                }
                for (int x = 0; x < width; x++) board[x, 0] = 0;
                y++;
            }
        }
    }
}
