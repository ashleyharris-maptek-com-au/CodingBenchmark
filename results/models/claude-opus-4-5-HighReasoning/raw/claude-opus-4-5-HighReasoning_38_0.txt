import sys

def solve():
    lines = sys.stdin.read().strip().split('\n')
    U, S = map(int, lines[0].split())

    sets = []
    for i in range(1, S + 1):
        if i < len(lines) and lines[i].strip():
            sets.append(set(map(int, lines[i].split())))
        else:
            sets.append(set())

    universe = set(range(U))

    def backtrack(covered, chosen):
        if covered == universe:
            return chosen[:]

        # Find the smallest uncovered element
        uncovered = universe - covered
        if not uncovered:
            return chosen[:]

        target = min(uncovered)

        # Try each set that contains target and doesn't overlap with covered
        for i in range(S):
            if i not in chosen and target in sets[i] and not (sets[i] & covered):
                chosen.append(i)
                result = backtrack(covered | sets[i], chosen)
                if result is not None:
                    return result
                chosen.pop()

        return None

    result = backtrack(set(), [])

    if result is not None:
        print("SOLUTION")
        print(' '.join(map(str, sorted(result))))
    else:
        print("NO SOLUTION")

solve()
