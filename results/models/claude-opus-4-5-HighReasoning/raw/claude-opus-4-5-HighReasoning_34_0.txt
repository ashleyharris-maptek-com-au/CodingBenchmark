import sys
import math

def dist(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve_cvrp():
    data = sys.stdin.read().split()
    idx = 0
    N, K, capacity = int(data[idx]), int(data[idx+1]), int(data[idx+2])
    idx += 3
    depot = (float(data[idx]), float(data[idx+1]))
    idx += 2
    customers = []
    demands = []
    for _ in range(N):
        x, y, d = float(data[idx]), float(data[idx+1]), int(data[idx+2])
        customers.append((x, y))
        demands.append(d)
        idx += 3

    if N == 0:
        print(0.0)
        print(0)
        return

    # Clarke-Wright Savings Algorithm
    # Calculate savings: s(i,j) = d(depot,i) + d(depot,j) - d(i,j)
    savings = []
    for i in range(N):
        for j in range(i+1, N):
            s = dist(depot, customers[i]) + dist(depot, customers[j]) - dist(customers[i], customers[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    # Initialize: each customer in its own route
    route_of = list(range(N))  # route_of[i] = route index for customer i
    routes = [[i] for i in range(N)]
    route_demands = demands[:]

    def is_interior(c):
        r = routes[route_of[c]]
        if len(r) <= 2:
            return False
        return r[0] != c and r[-1] != c

    for s, i, j in savings:
        ri, rj = route_of[i], route_of[j]
        if ri == rj:
            continue
        if is_interior(i) or is_interior(j):
            continue
        if route_demands[ri] + route_demands[rj] > capacity:
            continue

        # Merge routes
        r_i = routes[ri]
        r_j = routes[rj]

        # Determine how to merge based on endpoints
        if r_i[-1] == i and r_j[0] == j:
            new_route = r_i + r_j
        elif r_i[-1] == i and r_j[-1] == j:
            new_route = r_i + r_j[::-1]
        elif r_i[0] == i and r_j[0] == j:
            new_route = r_i[::-1] + r_j
        elif r_i[0] == i and r_j[-1] == j:
            new_route = r_j + r_i
        else:
            continue

        new_demand = route_demands[ri] + route_demands[rj]
        routes[ri] = new_route
        routes[rj] = []
        route_demands[ri] = new_demand
        route_demands[rj] = 0
        for c in new_route:
            route_of[c] = ri

    # Collect non-empty routes
    final_routes = [r for r in routes if r]

    # Limit to K routes if needed - merge smallest routes
    while len(final_routes) > K:
        merged = False
        final_routes.sort(key=lambda r: sum(demands[c] for c in r))
        for i in range(len(final_routes)):
            for j in range(i+1, len(final_routes)):
                d1 = sum(demands[c] for c in final_routes[i])
                d2 = sum(demands[c] for c in final_routes[j])
                if d1 + d2 <= capacity:
                    final_routes[i] = final_routes[i] + final_routes[j]
                    final_routes.pop(j)
                    merged = True
                    break
            if merged:
                break
        if not merged:
            final_routes[0] = final_routes[0] + final_routes[-1]
            final_routes.pop()

    # 2-opt improvement for each route
    def route_dist(route):
        if not route:
            return 0
        d = dist(depot, customers[route[0]])
        for k in range(len(route)-1):
            d += dist(customers[route[k]], customers[route[k+1]])
        d += dist(customers[route[-1]], depot)
        return d

    def two_opt(route):
        if len(route) < 2:
            return route
        improved = True
        while improved:
            improved = False
            for i in range(len(route)-1):
                for j in range(i+2, len(route)):
                    new_route = route[:i+1] + route[i+1:j+1][::-1] + route[j+1:]
                    if route_dist(new_route) < route_dist(route) - 1e-9:
                        route = new_route
                        improved = True
                        break
                if improved:
                    break
        return route

    final_routes = [two_opt(r) for r in final_routes]

    # Calculate total distance
    total_dist = sum(route_dist(r) for r in final_routes)

    print(f"{total_dist:.6f}")
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    solve_cvrp()
