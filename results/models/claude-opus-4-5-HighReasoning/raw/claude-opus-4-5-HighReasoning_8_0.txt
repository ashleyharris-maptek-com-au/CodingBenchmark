use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let stdin = stdin.lock();
    let mut lines = stdin.lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut dims = first_line.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();

    let mut maze: Vec<u8> = Vec::with_capacity(h * w);
    let mut start = 0usize;
    let mut end = 0usize;

    for y in 0..h {
        let row = lines.next().unwrap().unwrap();
        let bytes = row.as_bytes();
        for x in 0..w {
            let c = if x < bytes.len() { bytes[x] } else { b' ' };
            if c == b'A' {
                start = y * w + x;
            } else if c == b'B' {
                end = y * w + x;
            }
            maze.push(c);
        }
    }

    let mut parent: Vec<u8> = vec![0; h * w];
    let mut queue = VecDeque::new();

    queue.push_back(start);
    parent[start] = 1;

    while let Some(pos) = queue.pop_front() {
        if pos == end {
            break;
        }

        let x = pos % w;
        let y = pos / w;

        if y > 0 {
            let npos = pos - w;
            if parent[npos] == 0 && maze[npos] != b'#' {
                parent[npos] = 2;
                queue.push_back(npos);
            }
        }
        if y + 1 < h {
            let npos = pos + w;
            if parent[npos] == 0 && maze[npos] != b'#' {
                parent[npos] = 3;
                queue.push_back(npos);
            }
        }
        if x > 0 {
            let npos = pos - 1;
            if parent[npos] == 0 && maze[npos] != b'#' {
                parent[npos] = 4;
                queue.push_back(npos);
            }
        }
        if x + 1 < w {
            let npos = pos + 1;
            if parent[npos] == 0 && maze[npos] != b'#' {
                parent[npos] = 5;
                queue.push_back(npos);
            }
        }
    }

    if parent[end] == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path = Vec::new();
    let mut pos = end;

    while parent[pos] != 1 {
        path.push(pos);
        pos = match parent[pos] {
            2 => pos + w,
            3 => pos - w,
            4 => pos + 1,
            5 => pos - 1,
            _ => unreachable!(),
        };
    }
    path.push(start);
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for pos in path {
        writeln!(out, "{} {}", pos % w, pos / w).unwrap();
    }
}
