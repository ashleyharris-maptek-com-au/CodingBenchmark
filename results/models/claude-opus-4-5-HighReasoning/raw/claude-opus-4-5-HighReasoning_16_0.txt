#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

struct Task {
    int machine;
    int duration;
};

struct ScheduledTask {
    int start;
    int machine;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<Task>> jobs(J);
    int totalTasks = 0;

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        totalTasks += T;
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].machine >> jobs[j][t].duration;
        }
    }

    vector<vector<ScheduledTask>> schedule(J);
    for (int j = 0; j < J; j++) {
        schedule[j].resize(jobs[j].size());
    }

    vector<int> machineAvail(M, 0);
    vector<int> jobNextTask(J, 0);
    vector<int> jobAvail(J, 0);

    int scheduled = 0;
    while (scheduled < totalTasks) {
        int bestJob = -1;
        int bestStart = INT_MAX;
        int bestDuration = INT_MAX;

        for (int j = 0; j < J; j++) {
            if (jobNextTask[j] < (int)jobs[j].size()) {
                int taskIdx = jobNextTask[j];
                int m = jobs[j][taskIdx].machine;
                int earliest = max(jobAvail[j], machineAvail[m]);
                int dur = jobs[j][taskIdx].duration;

                if (earliest < bestStart ||
                    (earliest == bestStart && dur < bestDuration)) {
                    bestStart = earliest;
                    bestJob = j;
                    bestDuration = dur;
                }
            }
        }

        if (bestJob == -1) break;

        int j = bestJob;
        int taskIdx = jobNextTask[j];
        int m = jobs[j][taskIdx].machine;
        int dur = jobs[j][taskIdx].duration;
        int startTime = max(jobAvail[j], machineAvail[m]);

        schedule[j][taskIdx].start = startTime;
        schedule[j][taskIdx].machine = m;

        machineAvail[m] = startTime + dur;
        jobAvail[j] = startTime + dur;
        jobNextTask[j]++;
        scheduled++;
    }

    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)jobs[j].size(); t++) {
            if (t > 0) cout << ' ';
            cout << schedule[j][t].start << ' ' << schedule[j][t].machine;
        }
        cout << '\n';
    }

    return 0;
}
