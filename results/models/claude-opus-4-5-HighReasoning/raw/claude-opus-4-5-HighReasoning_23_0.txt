use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    timestamp: Instant,
}

impl State {
    fn new() -> Self {
        State {
            x: 0.0, y: 0.0, vx: 0.0, vy: 0.0,
            angle: 0.0, angular_vel: 0.0, fuel: 1.0,
            timestamp: Instant::now(),
        }
    }

    fn extrapolate(&self, gravity: f64) -> (f64, f64, f64, f64) {
        let dt = self.timestamp.elapsed().as_secs_f64();
        let x = self.x + self.vx * dt;
        let y = self.y + self.vy * dt + 0.5 * gravity * dt * dt;
        let vx = self.vx;
        let vy = self.vy + gravity * dt;
        (x, y, vx, vy)
    }
}

struct Config {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

fn compute_control(
    x: f64, y: f64, vx: f64, vy: f64, angle: f64, angular_vel: f64,
    config: &Config,
) -> (f64, f64) {
    let dx = config.target_x - x;
    let dy = config.target_y - y;
    let dist = (dx * dx + dy * dy).sqrt().max(0.1);

    let approach_factor = (dist / 20.0).min(1.0);
    let max_approach_vx = 4.0 * approach_factor;
    let max_approach_vy = 3.0 * approach_factor;

    let kp = 0.25;
    let desired_vx = (dx * kp).clamp(-max_approach_vx, max_approach_vx);
    let desired_vy = (dy * kp).clamp(-0.5, max_approach_vy);

    if dist < 5.0 {
        let landing_vy = 1.0_f64.min(dy.max(0.0) * 0.3);
        let vx_err = -vx;
        let vy_err = landing_vy - vy;

        let kv = 2.0;
        let ax = vx_err * kv;
        let ay = vy_err * kv - config.gravity;

        let desired_angle = ax.atan2(-ay);
        let angle_err = normalize_angle(desired_angle - angle);

        let ka = 3.0;
        let kd = 1.5;
        let turn = (angle_err * ka - angular_vel * kd).clamp(-1.0, 1.0);

        let thrust_mag = (ax * ax + ay * ay).sqrt();
        let mut thrust = (thrust_mag / config.max_thrust).clamp(0.0, 1.0);

        if angle_err.abs() > 0.3 {
            thrust *= 0.3;
        }

        return (thrust, turn);
    }

    let vx_err = desired_vx - vx;
    let vy_err = desired_vy - vy;

    let kv = 1.8;
    let ax = vx_err * kv;
    let ay = vy_err * kv - config.gravity;

    let desired_angle = ax.atan2(-ay);
    let angle_err = normalize_angle(desired_angle - angle);

    let ka = 2.5;
    let kd = 1.2;
    let turn = (angle_err * ka - angular_vel * kd).clamp(-1.0, 1.0);

    let thrust_mag = (ax * ax + ay * ay).sqrt();
    let mut thrust = (thrust_mag / config.max_thrust).clamp(0.0, 1.0);

    if angle_err.abs() > 0.5 {
        thrust *= 0.4;
    } else if angle_err.abs() > 0.2 {
        thrust *= 0.7;
    }

    (thrust, turn)
}

fn normalize_angle(mut a: f64) -> f64 {
    use std::f64::consts::PI;
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }
    a
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();

    line.clear();
    if reader.read_line(&mut line).unwrap() == 0 { return; }
    let nums: Vec<f64> = line.split_whitespace()
        .filter_map(|s| s.parse().ok()).collect();
    let height = nums[1] as usize;
    let gravity = nums[2];
    let max_thrust = nums[3];

    line.clear();
    if reader.read_line(&mut line).unwrap() == 0 { return; }
    let nums: Vec<f64> = line.split_whitespace()
        .filter_map(|s| s.parse().ok()).collect();
    let target_x = nums[2];
    let target_y = nums[3];

    let config = Arc::new(Config { gravity, max_thrust, target_x, target_y });

    for _ in 0..height {
        line.clear();
        let _ = reader.read_line(&mut line);
    }

    drop(reader);

    let state = Arc::new(Mutex::new(State::new()));
    let state_clone = Arc::clone(&state);
    let config_clone = Arc::clone(&config);

    thread::spawn(move || {
        let stdin = io::stdin();
        for line in stdin.lock().lines() {
            let line = match line { Ok(l) => l, Err(_) => break };
            let nums: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok()).collect();
            if nums.len() >= 7 {
                let mut s = state_clone.lock().unwrap();
                s.x = nums[0];
                s.y = nums[1];
                s.vx = nums[2];
                s.vy = nums[3];
                s.angle = nums[4];
                s.angular_vel = nums[5];
                s.fuel = nums[6];
                s.timestamp = Instant::now();
            }
        }
        let _ = &config_clone;
    });

    let stdout = io::stdout();
    let mut out = stdout.lock();
    let interval = Duration::from_millis(20);

    loop {
        let start = Instant::now();

        let (thrust, turn) = {
            let s = state.lock().unwrap();
            let (x, y, vx, vy) = s.extrapolate(config.gravity);
            compute_control(x, y, vx, vy, s.angle, s.angular_vel, &config)
        };

        if writeln!(out, "{:.4} {:.4}", thrust, turn).is_err() { break; }
        if out.flush().is_err() { break; }

        let elapsed = start.elapsed();
        if elapsed < interval {
            thread::sleep(interval - elapsed);
        }
    }
}
