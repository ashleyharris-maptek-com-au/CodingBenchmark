using System;
using System.Collections.Generic;

class Program
{
    static int H, W;
    static char[][] board;
    static HashSet<(int, int)> safe = new HashSet<(int, int)>();
    static HashSet<(int, int)> mines = new HashSet<(int, int)>();
    static int[] dx = {-1, 0, 1, -1, 1, -1, 0, 1};
    static int[] dy = {-1, -1, -1, 0, 0, 1, 1, 1};

    static void Main()
    {
        ReadInput();
        SimpleSolve();
        if ((long)H * W <= 2000000) ConstraintSolve();
        Output();
    }

    static void ReadInput()
    {
        var dims = Console.ReadLine().Split(new[]{' '}, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(dims[0]);
        W = int.Parse(dims[1]);
        board = new char[H][];
        for (int y = 0; y < H; y++)
        {
            var row = Console.ReadLine() ?? "";
            board[y] = new char[W];
            for (int x = 0; x < W; x++)
            {
                board[y][x] = x < row.Length ? row[x] : ' ';
                if (board[y][x] == '*') mines.Add((x, y));
            }
        }
    }

    static (int mineCount, List<(int,int)> unknowns) GetNeighborInfo(int x, int y)
    {
        int mineCount = 0;
        var unknowns = new List<(int, int)>();
        for (int d = 0; d < 8; d++)
        {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
            if (mines.Contains((nx, ny))) mineCount++;
            else if (board[ny][nx] == ' ' && !safe.Contains((nx, ny)))
                unknowns.Add((nx, ny));
        }
        return (mineCount, unknowns);
    }

    static void SimpleSolve()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    char c = board[y][x];
                    if (c < '0' || c > '8') continue;

                    int num = c - '0';
                    var (mineCount, unknowns) = GetNeighborInfo(x, y);
                    int remaining = num - mineCount;

                    if (remaining == 0 && unknowns.Count > 0)
                    {
                        foreach (var p in unknowns) if (safe.Add(p)) changed = true;
                    }
                    else if (remaining > 0 && remaining == unknowns.Count)
                    {
                        foreach (var p in unknowns) if (mines.Add(p)) changed = true;
                    }
                }
            }
        }
    }

    static void ConstraintSolve()
    {
        var constraints = new List<(HashSet<(int,int)> cells, int mineCount)>();

        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c < '0' || c > '8') continue;

                int num = c - '0';
                var (mineCount, unknownList) = GetNeighborInfo(x, y);
                var unknowns = new HashSet<(int, int)>(unknownList);
                int remaining = num - mineCount;

                if (unknowns.Count > 0 && remaining >= 0 && remaining <= unknowns.Count)
                    constraints.Add((unknowns, remaining));
            }
        }

        bool changed = true;
        int iter = 0;
        while (changed && iter++ < 100 && constraints.Count < 50000)
        {
            changed = false;
            var newConstraints = new List<(HashSet<(int,int)>, int)>();

            for (int i = 0; i < constraints.Count; i++)
            {
                var (cellsA, minesA) = constraints[i];
                if (cellsA.Count == 0) continue;

                for (int j = i + 1; j < constraints.Count; j++)
                {
                    var (cellsB, minesB) = constraints[j];
                    if (cellsB.Count == 0) continue;

                    if (cellsA.IsSubsetOf(cellsB) && cellsA.Count < cellsB.Count)
                    {
                        var diff = new HashSet<(int,int)>(cellsB);
                        diff.ExceptWith(cellsA);
                        int diffMines = minesB - minesA;

                        if (diffMines == 0)
                        {
                            foreach (var p in diff) if (safe.Add(p)) changed = true;
                        }
                        else if (diffMines == diff.Count)
                        {
                            foreach (var p in diff) if (mines.Add(p)) changed = true;
                        }
                        else if (diff.Count > 1 && diffMines > 0)
                        {
                            newConstraints.Add((diff, diffMines));
                        }
                    }
                    else if (cellsB.IsSubsetOf(cellsA) && cellsB.Count < cellsA.Count)
                    {
                        var diff = new HashSet<(int,int)>(cellsA);
                        diff.ExceptWith(cellsB);
                        int diffMines = minesA - minesB;

                        if (diffMines == 0)
                        {
                            foreach (var p in diff) if (safe.Add(p)) changed = true;
                        }
                        else if (diffMines == diff.Count)
                        {
                            foreach (var p in diff) if (mines.Add(p)) changed = true;
                        }
                        else if (diff.Count > 1 && diffMines > 0)
                        {
                            newConstraints.Add((diff, diffMines));
                        }
                    }
                }
            }

            constraints.AddRange(newConstraints);

            foreach (var c in constraints)
            {
                c.cells.ExceptWith(safe);
                c.cells.ExceptWith(mines);
            }
            constraints.RemoveAll(c => c.cells.Count == 0);
        }
    }

    static void Output()
    {
        var results = new List<(int, int)>();
        foreach (var (x, y) in safe)
            if (board[y][x] == ' ')
                results.Add((x, y));

        Console.WriteLine(results.Count);
        foreach (var (x, y) in results)
            Console.WriteLine($"{x} {y}");
    }
}
