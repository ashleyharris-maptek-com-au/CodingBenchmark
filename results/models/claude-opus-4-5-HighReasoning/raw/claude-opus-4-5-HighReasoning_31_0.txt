use std::collections::HashMap;
use std::io::{self, BufRead, BufWriter, Write};
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;
use std::thread;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let target: i64 = parts.next().unwrap().parse().unwrap();

    let mut numbers: Vec<i64> = Vec::with_capacity(n);
    for line in lines {
        let line = line.unwrap();
        for num_str in line.split_whitespace() {
            if numbers.len() >= n {
                break;
            }
            numbers.push(num_str.parse().unwrap());
        }
        if numbers.len() >= n {
            break;
        }
    }

    if target == 0 {
        writeln!(out, "YES").unwrap();
        writeln!(out, "[]").unwrap();
        return;
    }

    if n == 0 {
        writeln!(out, "NO").unwrap();
        return;
    }

    for (i, &num) in numbers.iter().enumerate() {
        if num == target {
            writeln!(out, "YES").unwrap();
            writeln!(out, "[{}]", i).unwrap();
            return;
        }
    }

    let total_sum: i64 = numbers.iter().sum();
    if total_sum == target {
        writeln!(out, "YES").unwrap();
        let indices: Vec<String> = (0..n).map(|i| i.to_string()).collect();
        writeln!(out, "[{}]", indices.join(" ")).unwrap();
        return;
    }

    let pos_sum: i64 = numbers.iter().filter(|&&x| x > 0).sum();
    let neg_sum: i64 = numbers.iter().filter(|&&x| x < 0).sum();

    if target > pos_sum || target < neg_sum {
        writeln!(out, "NO").unwrap();
        return;
    }

    if n <= 25 {
        if let Some(indices) = brute_force(&numbers, target) {
            writeln!(out, "YES").unwrap();
            let s: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            writeln!(out, "[{}]", s.join(" ")).unwrap();
        } else {
            writeln!(out, "NO").unwrap();
        }
        return;
    }

    if n <= 40 {
        if let Some(indices) = meet_in_middle(&numbers, target) {
            writeln!(out, "YES").unwrap();
            let s: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            writeln!(out, "[{}]", s.join(" ")).unwrap();
        } else {
            writeln!(out, "NO").unwrap();
        }
        return;
    }

    let abs_target = target.unsigned_abs() as usize;
    let max_abs = numbers.iter().map(|x| x.unsigned_abs()).max().unwrap_or(0) as usize;

    if abs_target <= 10_000_000 && max_abs <= 1_000_000 && n <= 100_000 {
        if let Some(indices) = dp_hashmap(&numbers, target) {
            writeln!(out, "YES").unwrap();
            let s: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            writeln!(out, "[{}]", s.join(" ")).unwrap();
        } else {
            writeln!(out, "NO").unwrap();
        }
        return;
    }

    if let Some(indices) = greedy_heuristic(&numbers, target) {
        writeln!(out, "YES").unwrap();
        let s: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
        writeln!(out, "[{}]", s.join(" ")).unwrap();
        return;
    }

    if let Some(indices) = parallel_random_search(&numbers, target, 12) {
        writeln!(out, "YES").unwrap();
        let s: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
        writeln!(out, "[{}]", s.join(" ")).unwrap();
        return;
    }

    writeln!(out, "NO").unwrap();
}

fn brute_force(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    for mask in 1u32..(1u32 << n) {
        let mut sum: i64 = 0;
        for i in 0..n {
            if mask & (1 << i) != 0 {
                sum += numbers[i];
            }
        }
        if sum == target {
            let indices: Vec<usize> = (0..n).filter(|&i| mask & (1 << i) != 0).collect();
            return Some(indices);
        }
    }
    None
}

fn meet_in_middle(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mid = n / 2;

    let mut left_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    left_sums.insert(0, vec![]);

    for i in 0..mid {
        let current: Vec<(i64, Vec<usize>)> = left_sums.iter()
            .map(|(s, v)| (*s, v.clone())).collect();
        for (sum, indices) in current {
            let new_sum = sum + numbers[i];
            let mut new_indices = indices.clone();
            new_indices.push(i);
            left_sums.entry(new_sum).or_insert(new_indices);
        }
    }

    let mut right_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    right_sums.insert(0, vec![]);

    for i in mid..n {
        let current: Vec<(i64, Vec<usize>)> = right_sums.iter()
            .map(|(s, v)| (*s, v.clone())).collect();
        for (sum, indices) in current {
            let new_sum = sum + numbers[i];
            let mut new_indices = indices.clone();
            new_indices.push(i);
            right_sums.entry(new_sum).or_insert(new_indices);
        }
    }

    for (left_sum, left_indices) in &left_sums {
        let needed = target - left_sum;
        if let Some(right_indices) = right_sums.get(&needed) {
            if left_indices.is_empty() && right_indices.is_empty() {
                continue;
            }
            let mut result = left_indices.clone();
            result.extend(right_indices);
            result.sort();
            return Some(result);
        }
    }
    None
}

fn dp_hashmap(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for (i, &num) in numbers.iter().enumerate() {
        let current: Vec<(i64, Vec<usize>)> = dp.iter()
            .map(|(s, v)| (*s, v.clone())).collect();

        for (sum, indices) in current {
            let new_sum = sum + num;
            if !dp.contains_key(&new_sum) {
                let mut new_indices = indices.clone();
                new_indices.push(i);
                if new_sum == target {
                    return Some(new_indices);
                }
                dp.insert(new_sum, new_indices);
            }
        }

        if dp.len() > 5_000_000 {
            break;
        }
    }

    dp.get(&target).cloned().filter(|v| !v.is_empty())
}

fn greedy_heuristic(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut indexed: Vec<(usize, i64)> = numbers.iter().enumerate()
        .map(|(i, &x)| (i, x)).collect();

    indexed.sort_by(|a, b| b.1.abs().cmp(&a.1.abs()));

    let mut current_sum: i64 = 0;
    let mut selected: Vec<usize> = Vec::new();

    for (idx, val) in &indexed {
        let new_sum = current_sum + val;
        if (new_sum - target).abs() < (current_sum - target).abs() {
            current_sum = new_sum;
            selected.push(*idx);
            if current_sum == target {
                selected.sort();
                return Some(selected);
            }
        }
    }

    let mut current_sum: i64 = 0;
    let mut selected: Vec<usize> = Vec::new();

    indexed.sort_by_key(|(_, x)| *x);

    if target > 0 {
        indexed.reverse();
    }

    for (idx, val) in &indexed {
        if target > 0 && *val <= 0 {
            continue;
        }
        if target < 0 && *val >= 0 {
            continue;
        }

        let new_sum = current_sum + val;
        if (target > 0 && new_sum <= target) || (target < 0 && new_sum >= target) {
            current_sum = new_sum;
            selected.push(*idx);
            if current_sum == target {
                selected.sort();
                return Some(selected);
            }
        }
    }

    None
}

fn parallel_random_search(numbers: &[i64], target: i64, num_threads: usize) -> Option<Vec<usize>> {
    let found = Arc::new(AtomicBool::new(false));
    let numbers = Arc::new(numbers.to_vec());
    let result: Arc<std::sync::Mutex<Option<Vec<usize>>>> = Arc::new(std::sync::Mutex::new(None));

    let mut handles = vec![];

    for thread_id in 0..num_threads {
        let found = Arc::clone(&found);
        let numbers = Arc::clone(&numbers);
        let result = Arc::clone(&result);

        let handle = thread::spawn(move || {
            let mut rng_state: u64 = (thread_id as u64 + 1) * 0x9E3779B97F4A7C15;

            for _ in 0..1_000_000 {
                if found.load(Ordering::Relaxed) {
                    return;
                }

                let mut sum: i64 = 0;
                let mut indices: Vec<usize> = Vec::new();

                let n = numbers.len();
                for i in 0..n {
                    rng_state = rng_state.wrapping_mul(0x5DEECE66D).wrapping_add(0xB);
                    if (rng_state >> 32) & 1 == 1 {
                        sum += numbers[i];
                        indices.push(i);
                    }

                    if sum == target && !indices.is_empty() {
                        if !found.swap(true, Ordering::SeqCst) {
                            let mut res = result.lock().unwrap();
                            *res = Some(indices);
                        }
                        return;
                    }
                }

                if sum == target && !indices.is_empty() {
                    if !found.swap(true, Ordering::SeqCst) {
                        let mut res = result.lock().unwrap();
                        *res = Some(indices);
                    }
                    return;
                }
            }
        });

        handles.push(handle);
    }

    for handle in handles {
        let _ = handle.join();
    }

    let res = result.lock().unwrap();
    res.clone()
}
