use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut in_cover = vec![false; n];
    let mut uncovered_deg: Vec<usize> = adj.iter().map(|a| a.len()).collect();

    let mut cover = Vec::new();

    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new();
    for i in 0..n {
        if uncovered_deg[i] > 0 {
            heap.push((uncovered_deg[i], i));
        }
    }

    while let Some((d, v)) = heap.pop() {
        if in_cover[v] || uncovered_deg[v] == 0 {
            continue;
        }
        if uncovered_deg[v] != d {
            heap.push((uncovered_deg[v], v));
            continue;
        }

        in_cover[v] = true;
        cover.push(v);

        for &u in &adj[v] {
            if !in_cover[u] {
                uncovered_deg[u] -= 1;
            }
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        write!(out, "{}", cover[0]).unwrap();
        for &v in &cover[1..] {
            write!(out, " {}", v).unwrap();
        }
    }
    writeln!(out).unwrap();
}
