use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let width: i32 = parts.next().unwrap().parse().unwrap();
    let height: i32 = parts.next().unwrap().parse().unwrap();
    let obstacle_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle: bool = parts.next().unwrap() == "1";

    let w = width as usize;
    let h = height as usize;
    let total_cells = w * h;

    let mut blocked = vec![false; total_cells];

    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let mut coords = line.split_whitespace();
        let x: usize = coords.next().unwrap().parse().unwrap();
        let y: usize = coords.next().unwrap().parse().unwrap();
        blocked[y * w + x] = true;
    }

    if blocked[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let target_len = total_cells - obstacle_count;

    if target_len == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let dirs: [(i32, i32); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    let neighbors_of = |idx: usize| -> Vec<usize> {
        let x = (idx % w) as i32;
        let y = (idx / w) as i32;
        let mut result = Vec::with_capacity(4);
        for &(dx, dy) in &dirs {
            let nx = x + dx;
            let ny = y + dy;
            if nx >= 0 && nx < width && ny >= 0 && ny < height {
                let nidx = (ny as usize) * w + (nx as usize);
                if !blocked[nidx] {
                    result.push(nidx);
                }
            }
        }
        result
    };

    let adj: Vec<Vec<usize>> = (0..total_cells).map(|i| neighbors_of(i)).collect();

    if require_cycle && adj[0].is_empty() {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut visited = vec![false; total_cells];
    let mut path = Vec::with_capacity(target_len);

    fn is_connected(adj: &[Vec<usize>], visited: &[bool], blocked: &[bool], start: usize, expected: usize) -> bool {
        if expected == 0 {
            return true;
        }
        let mut seen = vec![false; visited.len()];
        let mut queue = VecDeque::new();
        queue.push_back(start);
        seen[start] = true;
        let mut count = 1;
        while let Some(u) = queue.pop_front() {
            for &v in &adj[u] {
                if !seen[v] && !visited[v] && !blocked[v] {
                    seen[v] = true;
                    count += 1;
                    queue.push_back(v);
                }
            }
        }
        count == expected
    }

    fn count_unvisited_neighbors(adj: &[Vec<usize>], visited: &[bool], node: usize) -> usize {
        adj[node].iter().filter(|&&n| !visited[n]).count()
    }

    fn solve(
        adj: &[Vec<usize>],
        visited: &mut Vec<bool>,
        path: &mut Vec<usize>,
        blocked: &[bool],
        target_len: usize,
        require_cycle: bool,
        start_neighbors: &[usize],
        w: usize,
    ) -> bool {
        if path.len() == target_len {
            if require_cycle {
                let last = *path.last().unwrap();
                return start_neighbors.contains(&last);
            }
            return true;
        }

        let current = *path.last().unwrap();
        let remaining = target_len - path.len();

        let mut candidates: Vec<(usize, usize)> = adj[current]
            .iter()
            .filter(|&&n| !visited[n])
            .map(|&n| (n, count_unvisited_neighbors(adj, visited, n)))
            .collect();

        candidates.sort_by_key(|&(_, deg)| deg);

        for (next, _) in candidates {
            visited[next] = true;
            path.push(next);

            let should_check = remaining > 10 && path.len() % 5 == 0;
            let connected = if should_check {
                let unvisited_neighbor = adj[next].iter().find(|&&n| !visited[n]);
                match unvisited_neighbor {
                    Some(&start) => is_connected(adj, visited, blocked, start, remaining - 1),
                    None => remaining == 1,
                }
            } else {
                true
            };

            if connected && solve(adj, visited, path, blocked, target_len, require_cycle, start_neighbors, w) {
                return true;
            }

            path.pop();
            visited[next] = false;
        }

        false
    }

    visited[0] = true;
    path.push(0);

    let start_neighbors = adj[0].clone();

    if solve(&adj, &mut visited, &mut path, &blocked, target_len, require_cycle, &start_neighbors, w) {
        writeln!(out, "{}", path.len()).unwrap();
        for &idx in &path {
            let x = idx % w;
            let y = idx / w;
            writeln!(out, "{} {}", x, y).unwrap();
        }
    } else {
        writeln!(out, "0").unwrap();
    }
}
