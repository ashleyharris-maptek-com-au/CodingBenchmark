using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class AsteroidInterceptor
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        int numBodies = int.Parse(input[idx++]);
        double warningDays = double.Parse(input[idx++]);
        double impactorMass = double.Parse(input[idx++]);
        double deltaVBudget = double.Parse(input[idx++]);
        double asteroidMass = double.Parse(input[idx++]);
        double t0Unix = double.Parse(input[idx++]);

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            string name = input[idx++];
            double mass = double.Parse(input[idx++]);
            double x = double.Parse(input[idx++]);
            double y = double.Parse(input[idx++]);
            double z = double.Parse(input[idx++]);
            double vx = double.Parse(input[idx++]);
            double vy = double.Parse(input[idx++]);
            double vz = double.Parse(input[idx++]);
            bodies.Add(new Body { Name = name, Mass = mass, Pos = new Vec3(x, y, z), Vel = new Vec3(vx, vy, vz) });
        }

        string bspPath = idx < input.Length ? input[idx++] : "";

        Body earth = bodies.First(b => b.Name == "Earth");
        Body asteroid = bodies.First(b => b.Name == "Asteroid");
        Body sun = bodies.FirstOrDefault(b => b.Name == "Sun") ?? bodies.OrderByDescending(b => b.Mass).First();

        double mu = G * sun.Mass;

        double impactTime = FindImpactTime(bodies, warningDays, mu);

        double bestLaunch = 0, bestIntercept = impactTime * 0.8;
        Vec3 bestDv1 = new Vec3(), bestDv2 = new Vec3();
        double bestTotalDv = double.MaxValue;

        int launchSteps = Math.Min(60, (int)(warningDays / 5) + 15);
        int interceptSteps = Math.Min(50, (int)(warningDays / 10) + 12);

        for (int li = 0; li < launchSteps; li++)
        {
            double launchDay = warningDays * 0.35 * li / Math.Max(1, launchSteps - 1);

            var statesAtLaunch = PropagateAll(bodies, launchDay * DAY, mu);
            Vec3 earthPosLaunch = statesAtLaunch.First(b => b.Name == "Earth").Pos;
            Vec3 earthVelLaunch = statesAtLaunch.First(b => b.Name == "Earth").Vel;

            for (int ii = 0; ii < interceptSteps; ii++)
            {
                double interceptDay = launchDay + (impactTime - launchDay) * 0.25 +
                                      (impactTime - launchDay) * 0.70 * ii / Math.Max(1, interceptSteps - 1);

                if (interceptDay <= launchDay + 5) continue;
                if (interceptDay >= impactTime - 1) continue;

                var statesAtIntercept = PropagateAll(bodies, interceptDay * DAY, mu);
                Vec3 asteroidPosIntercept = statesAtIntercept.First(b => b.Name == "Asteroid").Pos;
                Vec3 asteroidVelIntercept = statesAtIntercept.First(b => b.Name == "Asteroid").Vel;

                double tof = (interceptDay - launchDay) * DAY;

                var lambert = SolveLambert(earthPosLaunch, asteroidPosIntercept, tof, mu, false);
                if (lambert == null)
                    lambert = SolveLambert(earthPosLaunch, asteroidPosIntercept, tof, mu, true);
                if (lambert == null) continue;

                Vec3 v1 = lambert.Value.v1;
                Vec3 v2 = lambert.Value.v2;

                Vec3 dv1 = v1 - earthVelLaunch;
                Vec3 dv2 = v2 - asteroidVelIntercept;

                double launchDv = dv1.Magnitude();
                double impactVel = dv2.Magnitude();

                double momentumTransfer = impactorMass * impactVel;
                double asteroidDeltaV = momentumTransfer / asteroidMass;
                double timeToImpact = (impactTime - interceptDay) * DAY;
                double deflection = asteroidDeltaV * timeToImpact;

                if (launchDv < bestTotalDv && launchDv <= deltaVBudget * 1.2 && deflection > 1e6)
                {
                    bestTotalDv = launchDv;
                    bestLaunch = launchDay;
                    bestIntercept = interceptDay;
                    bestDv1 = dv1;
                    bestDv2 = dv2;
                }
            }
        }

        if (bestTotalDv == double.MaxValue)
        {
            bestLaunch = 1.0;
            bestIntercept = impactTime * 0.7;
            var stL = PropagateAll(bodies, bestLaunch * DAY, mu);
            var stI = PropagateAll(bodies, bestIntercept * DAY, mu);
            Vec3 dir = (stI.First(b => b.Name == "Asteroid").Pos - stL.First(b => b.Name == "Earth").Pos).Normalize();
            bestDv1 = dir * Math.Min(deltaVBudget * 0.9, 5000);
            bestDv2 = dir * (-100);
        }

        if (bestTotalDv > deltaVBudget)
        {
            double scale = deltaVBudget * 0.95 / bestTotalDv;
            bestDv1 = bestDv1 * scale;
        }

        var burns = new List<Burn>();
        burns.Add(new Burn { Time = bestLaunch, Dv = bestDv1 });

        double mcTime = bestLaunch + (bestIntercept - bestLaunch) * 0.55;
        double remainingDv = deltaVBudget - bestDv1.Magnitude();
        if (remainingDv > 10 && (bestIntercept - bestLaunch) > 30)
        {
            double mcDvMag = Math.Min(remainingDv * 0.1, 50);
            Vec3 mcDv = bestDv1.Normalize() * mcDvMag;
            burns.Add(new Burn { Time = mcTime, Dv = mcDv });
        }

        burns.Add(new Burn { Time = bestIntercept, Dv = bestDv2 });

        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.Time:F6} {burn.Dv.X:E8} {burn.Dv.Y:E8} {burn.Dv.Z:E8}");
        }
    }

    static double FindImpactTime(List<Body> bodies, double warningDays, double mu)
    {
        double minDist = double.MaxValue;
        double impactTime = warningDays * 0.95;

        int steps = Math.Min(200, (int)warningDays + 50);
        for (int i = 0; i <= steps; i++)
        {
            double t = warningDays * 0.4 + warningDays * 0.6 * i / steps;
            var states = PropagateAll(bodies, t * DAY, mu);
            Vec3 ePos = states.First(b => b.Name == "Earth").Pos;
            Vec3 aPos = states.First(b => b.Name == "Asteroid").Pos;
            double dist = (ePos - aPos).Magnitude();
            if (dist < minDist)
            {
                minDist = dist;
                impactTime = t;
            }
        }
        return impactTime;
    }

    static List<Body> PropagateAll(List<Body> bodies, double dt, double mu)
    {
        var result = new List<Body>();
        foreach (var b in bodies)
        {
            var (pos, vel) = PropagateKepler(b.Pos, b.Vel, dt, mu);
            result.Add(new Body { Name = b.Name, Mass = b.Mass, Pos = pos, Vel = vel });
        }
        return result;
    }

    static (Vec3 pos, Vec3 vel) PropagateKepler(Vec3 r0, Vec3 v0, double dt, double mu)
    {
        if (Math.Abs(dt) < 1e-10) return (r0, v0);

        double r0Mag = r0.Magnitude();
        if (r0Mag < 1e-10) return (r0, v0);

        double v0Mag = v0.Magnitude();
        double vr0 = Vec3.Dot(r0, v0) / r0Mag;
        double alpha = 2.0 / r0Mag - v0Mag * v0Mag / mu;

        double xi;
        if (Math.Abs(alpha) > 1e-10)
            xi = Math.Sqrt(mu) * Math.Abs(alpha) * dt;
        else
            xi = Math.Sqrt(mu) * dt / r0Mag;

        xi = Math.Max(-1e12, Math.Min(1e12, xi));

        for (int iter = 0; iter < 60; iter++)
        {
            double xi2 = xi * xi;
            double psi = alpha * xi2;
            psi = Math.Max(-50, Math.Min(50, psi));

            double c2 = StumpffC2(psi);
            double c3 = StumpffC3(psi);

            double r = xi2 * c2 + vr0 * r0Mag / Math.Sqrt(mu) * xi * (1.0 - psi * c3) + r0Mag * (1.0 - psi * c2);
            if (Math.Abs(r) < 1e-10) r = 1e-10;

            double F = Math.Sqrt(mu) * dt - xi2 * xi * c3 - vr0 * r0Mag / Math.Sqrt(mu) * xi2 * c2 - r0Mag * xi * (1.0 - psi * c3);
            double dxi = F / r;

            xi += dxi;
            if (Math.Abs(dxi) < 1e-12 * Math.Abs(xi) + 1e-14) break;
        }

        double xi2f = xi * xi;
        double psif = alpha * xi2f;
        psif = Math.Max(-50, Math.Min(50, psif));

        double c2f = StumpffC2(psif);
        double c3f = StumpffC3(psif);

        double f = 1.0 - xi2f * c2f / r0Mag;
        double g = dt - xi2f * xi * c3f / Math.Sqrt(mu);

        Vec3 r1 = r0 * f + v0 * g;
        double r1Mag = r1.Magnitude();
        if (r1Mag < 1e-10) return (r0, v0);

        double df = Math.Sqrt(mu) * xi * (psif * c3f - 1.0) / (r0Mag * r1Mag);
        double dg = 1.0 - xi2f * c2f / r1Mag;

        Vec3 v1 = r0 * df + v0 * dg;

        return (r1, v1);
    }

    static double StumpffC2(double psi)
    {
        if (psi > 1e-6) return (1.0 - Math.Cos(Math.Sqrt(psi))) / psi;
        if (psi < -1e-6) { double sp = Math.Sqrt(-psi); return (1.0 - Math.Cosh(sp)) / psi; }
        return 0.5 - psi / 24.0 + psi * psi / 720.0;
    }

    static double StumpffC3(double psi)
    {
        if (psi > 1e-6) { double sq = Math.Sqrt(psi); return (sq - Math.Sin(sq)) / (psi * sq); }
        if (psi < -1e-6) { double sq = Math.Sqrt(-psi); return (Math.Sinh(sq) - sq) / (-psi * sq); }
        return 1.0 / 6.0 - psi / 120.0 + psi * psi / 5040.0;
    }

    static (Vec3 v1, Vec3 v2)? SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu, bool retrograde)
    {
        double r1Mag = r1.Magnitude();
        double r2Mag = r2.Magnitude();
        if (r1Mag < 1e-10 || r2Mag < 1e-10 || tof < 1e-10) return null;

        double cosNu = Vec3.Dot(r1, r2) / (r1Mag * r2Mag);
        cosNu = Math.Max(-0.9999, Math.Min(0.9999, cosNu));

        Vec3 h = Vec3.Cross(r1, r2);
        double dm = retrograde ? -1 : 1;
        if (h.Z < 0) dm = -dm;

        double A = dm * Math.Sqrt(r1Mag * r2Mag * (1 + cosNu));
        if (Math.Abs(A) < 1e-10) return null;

        double z = 0.0;
        double zLow = -4 * Math.PI * Math.PI;
        double zHigh = 4 * Math.PI * Math.PI;

        for (int iter = 0; iter < 100; iter++)
        {
            double c2 = StumpffC2(z);
            double c3 = StumpffC3(z);

            double sqrtC2 = Math.Sqrt(Math.Abs(c2));
            if (sqrtC2 < 1e-14) sqrtC2 = 1e-14;

            double y = r1Mag + r2Mag + A * (z * c3 - 1) / sqrtC2;
            if (y < 0)
            {
                zLow = z;
                z = (zLow + zHigh) / 2;
                continue;
            }

            double x = Math.Sqrt(y / c2);
            double t = (x * x * x * c3 + A * Math.Sqrt(y)) / Math.Sqrt(mu);

            if (Math.Abs(t - tof) < 1e-8 * tof) break;

            if (t < tof) zHigh = z;
            else zLow = z;
            z = (zLow + zHigh) / 2;
        }

        double c2f = StumpffC2(z);
        double c3f = StumpffC3(z);
        double sqrtC2f = Math.Sqrt(Math.Abs(c2f));
        if (sqrtC2f < 1e-14) sqrtC2f = 1e-14;

        double yf = r1Mag + r2Mag + A * (z * c3f - 1) / sqrtC2f;
        if (yf < 0) return null;

        double f = 1 - yf / r1Mag;
        double g = A * Math.Sqrt(yf / mu);
        if (Math.Abs(g) < 1e-14) return null;

        double dg = 1 - yf / r2Mag;

        Vec3 v1 = (r2 - r1 * f) * (1.0 / g);
        Vec3 v2 = (r2 * dg - r1) * (1.0 / g);

        if (double.IsNaN(v1.X) || double.IsInfinity(v1.X)) return null;
        if (double.IsNaN(v2.X) || double.IsInfinity(v2.X)) return null;

        return (v1, v2);
    }

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalize() { double m = Magnitude(); return m > 1e-14 ? this * (1.0 / m) : new Vec3(1, 0, 0); }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    class Body { public string Name; public double Mass; public Vec3 Pos, Vel; }
    class Burn { public double Time; public Vec3 Dv; }
}
