#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <functional>
#include <cstdint>
using namespace std;

int n, m;
vector<set<int>> adj;

inline int pop64(uint64_t x) {
    int c = 0;
    while (x) { c++; x &= x - 1; }
    return c;
}

inline int ctz64(uint64_t x) {
    int c = 0;
    while (!(x & 1)) { c++; x >>= 1; }
    return c;
}

vector<int> parent, rnk;
int find(int x) {
    if (parent[x] != x) parent[x] = find(parent[x]);
    return parent[x];
}
void unite(int x, int y) {
    int px = find(x), py = find(y);
    if (px == py) return;
    if (rnk[px] < rnk[py]) swap(px, py);
    parent[py] = px;
    if (rnk[px] == rnk[py]) rnk[px]++;
}

vector<int> solve_component(vector<int>& nodes) {
    int k = nodes.size();
    if (k == 0) return {};
    if (k == 1) return {nodes[0]};

    map<int, int> g2l;
    for (int i = 0; i < k; i++) g2l[nodes[i]] = i;

    if (k <= 64) {
        vector<uint64_t> adj_local(k, 0);
        for (int i = 0; i < k; i++) {
            int u = nodes[i];
            for (int v : adj[u]) {
                if (g2l.count(v)) {
                    adj_local[i] |= 1ULL << g2l[v];
                }
            }
        }

        uint64_t best = 0;
        int best_size = 0;

        function<void(uint64_t, uint64_t)> solve = [&](uint64_t cur, uint64_t cand) {
            int cs = pop64(cur), cd = pop64(cand);
            if (cs + cd <= best_size) return;
            if (!cand) {
                if (cs > best_size) { best_size = cs; best = cur; }
                return;
            }

            int bv = ctz64(cand), md = pop64(adj_local[bv] & cand);
            for (uint64_t t = cand & (cand - 1); t; t &= t - 1) {
                int v = ctz64(t);
                int d = pop64(adj_local[v] & cand);
                if (d > md) { md = d; bv = v; }
            }

            uint64_t vb = 1ULL << bv, nc = cand ^ vb;
            solve(cur, nc);
            solve(cur | vb, nc & ~adj_local[bv]);
        };

        uint64_t all = (k == 64) ? ~0ULL : (1ULL << k) - 1;
        solve(0, all);

        vector<int> result;
        for (int i = 0; i < k; i++) {
            if (best & (1ULL << i)) result.push_back(nodes[i]);
        }
        return result;
    } else {
        vector<int> best_sol;
        vector<int> cur;
        int best_size = 0;

        function<void(set<int>&)> solve = [&](set<int>& cand) {
            if (cur.size() + cand.size() <= (size_t)best_size) return;
            if (cand.empty()) {
                if ((int)cur.size() > best_size) {
                    best_size = cur.size();
                    best_sol = cur;
                }
                return;
            }

            int bv = *cand.begin(), md = 0;
            for (int v : cand) {
                int d = 0;
                for (int u : adj[v]) if (cand.count(u)) d++;
                if (d > md) { md = d; bv = v; }
            }

            cand.erase(bv);
            solve(cand);

            vector<int> removed;
            for (int u : adj[bv]) {
                if (cand.count(u)) {
                    removed.push_back(u);
                    cand.erase(u);
                }
            }
            cur.push_back(bv);
            solve(cand);
            cur.pop_back();

            for (int u : removed) cand.insert(u);
            cand.insert(bv);
        };

        set<int> cand(nodes.begin(), nodes.end());
        solve(cand);
        return best_sol;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    if (n == 0) {
        cout << "0\n\n";
        return 0;
    }

    adj.resize(n);
    parent.resize(n);
    rnk.resize(n, 0);
    for (int i = 0; i < n; i++) parent[i] = i;

    for (int i = 0; i < m; i++) {
        int u, v; cin >> u >> v;
        adj[u].insert(v);
        adj[v].insert(u);
        unite(u, v);
    }

    map<int, vector<int>> components;
    for (int i = 0; i < n; i++) {
        components[find(i)].push_back(i);
    }

    vector<int> result;
    for (auto& [root, nodes] : components) {
        auto sol = solve_component(nodes);
        for (int v : sol) result.push_back(v);
    }

    cout << result.size() << "\n";
    for (size_t i = 0; i < result.size(); i++) {
        if (i) cout << " ";
        cout << result[i];
    }
    cout << "\n";

    return 0;
}
