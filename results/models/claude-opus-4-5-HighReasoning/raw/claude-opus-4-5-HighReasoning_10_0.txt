#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, W, H;
    std::cin >> n >> W >> H;

    struct Rect { int w, h, idx; };
    std::vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    // Sort by max dimension descending, then by area
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        int aMax = std::max(a.w, a.h), bMax = std::max(b.w, b.h);
        return aMax != bMax ? aMax > bMax : a.w * a.h > b.w * b.h;
    });

    struct Shelf { int y, height, used; };
    struct Board { std::vector<Shelf> shelves; int freeY = 0; };

    std::vector<Board> boards;
    struct Placement { int board, x, y; bool rotated; };
    std::vector<Placement> placements(n);

    for (const auto& rect : rects) {
        int bestBoard = -1, bestWaste = INT_MAX;
        int bestX = 0, bestY = 0, bestShelf = -1;
        bool bestRot = false;

        int searchStart = std::max(0, (int)boards.size() - 50);

        for (int b = searchStart; b < (int)boards.size() && bestWaste > 0; b++) {
            Board& bd = boards[b];

            for (int rot = 0; rot < 2; rot++) {
                int rw = rot ? rect.h : rect.w;
                int rh = rot ? rect.w : rect.h;
                if (rw > W || rh > H) continue;

                for (int s = 0; s < (int)bd.shelves.size(); s++) {
                    auto& sh = bd.shelves[s];
                    if (sh.used + rw <= W && sh.height >= rh) {
                        int waste = sh.height - rh;
                        if (waste < bestWaste) {
                            bestWaste = waste;
                            bestBoard = b;
                            bestX = sh.used;
                            bestY = sh.y;
                            bestShelf = s;
                            bestRot = rot;
                        }
                    }
                }

                if (bd.freeY + rh <= H && 0 < bestWaste) {
                    bestWaste = 0;
                    bestBoard = b;
                    bestX = 0;
                    bestY = bd.freeY;
                    bestShelf = -1;
                    bestRot = rot;
                }
            }
        }

        if (bestBoard < 0) {
            boards.push_back(Board{});
            bestBoard = (int)boards.size() - 1;

            bool canNorm = rect.w <= W && rect.h <= H;
            bool canRot = rect.h <= W && rect.w <= H;
            bestRot = !canNorm && canRot;
            bestX = 0;
            bestY = 0;
            bestShelf = -1;
        }

        Board& bd = boards[bestBoard];
        int rw = bestRot ? rect.h : rect.w;
        int rh = bestRot ? rect.w : rect.h;

        if (bestShelf >= 0) {
            bd.shelves[bestShelf].used += rw;
        } else {
            bd.shelves.push_back({bd.freeY, rh, rw});
            bd.freeY += rh;
        }

        placements[rect.idx] = {bestBoard, bestX, bestY, bestRot};
    }

    std::cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        auto& p = placements[i];
        std::cout << p.board << " " << p.x << " " << p.y << " " << (p.rotated ? 1 : 0) << "\n";
    }

    return 0;
}
