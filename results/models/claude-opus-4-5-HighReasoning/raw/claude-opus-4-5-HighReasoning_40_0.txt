#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;
long long bestObj;
vector<int> bestSol;
vector<long long> suffix;

void dfs(int idx, vector<int>& x, long long curObj) {
    if (curObj + suffix[idx] <= bestObj) return;

    if (idx == n) {
        bestObj = curObj;
        bestSol = x;
        return;
    }

    bool canTake = true;
    for (int v : adj[idx]) {
        if (v < idx && x[v] == 1) {
            canTake = false;
            break;
        }
    }

    if (canTake && c[idx] > 0) {
        x[idx] = 1;
        dfs(idx + 1, x, curObj + c[idx]);
        x[idx] = 0;
        dfs(idx + 1, x, curObj);
    } else if (canTake) {
        dfs(idx + 1, x, curObj);
        x[idx] = 1;
        dfs(idx + 1, x, curObj + c[idx]);
        x[idx] = 0;
    } else {
        dfs(idx + 1, x, curObj);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    c.resize(n);
    adj.resize(n);

    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    suffix.resize(n + 1);
    suffix[n] = 0;
    for (int i = n - 1; i >= 0; i--) {
        suffix[i] = suffix[i + 1] + max(0LL, c[i]);
    }

    bestObj = 0;
    bestSol.assign(n, 0);

    vector<int> x(n, 0);
    dfs(0, x, 0);

    cout << bestObj << "\n";
    for (int i = 0; i < n; i++) {
        cout << bestSol[i];
        if (i < n - 1) cout << " ";
    }
    cout << "\n";

    return 0;
}
