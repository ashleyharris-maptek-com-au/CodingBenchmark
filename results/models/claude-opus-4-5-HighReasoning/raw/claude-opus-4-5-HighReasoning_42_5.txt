; SPIR-V Geometry Shader - Arrow Lines
; Assembles with: spirv-as --target-env vulkan1.0

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
       %void = OpTypeVoid
   %void_fun = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
    %v2float = OpTypeVector %float 2
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

               ; Constants
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
 %float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
 %float_0_05 = OpConstant %float 0.05

               ; Struct types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_2
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

               ; Input color array
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%ptr_in_v3 = OpTypePointer Input %v3float

               ; Output color
%ptr_out_v3 = OpTypePointer Output %v3float

               ; Variables
      %gl_in = OpVariable %ptr_in_arr_pv Input
     %gl_out = OpVariable %ptr_out_pv Output
    %inColor = OpVariable %ptr_in_arr_v3 Input
   %outColor = OpVariable %ptr_out_v3 Output

               ; Main function
       %main = OpFunction %void None %void_fun
      %entry = OpLabel

               ; Load p0 (gl_in[0].gl_Position)
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0_v4 = OpLoad %v4float %p0_ptr
        %p0 = OpVectorShuffle %v2float %p0_v4 %p0_v4 0 1

               ; Load p1 (gl_in[1].gl_Position)
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1_v4 = OpLoad %v4float %p1_ptr
        %p1 = OpVectorShuffle %v2float %p1_v4 %p1_v4 0 1

               ; Load colors
     %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
         %c0 = OpLoad %v3float %c0_ptr
     %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
         %c1 = OpLoad %v3float %c1_ptr

               ; Get z and w from p0 for output
       %p0_z = OpCompositeExtract %float %p0_v4 2
       %p0_w = OpCompositeExtract %float %p0_v4 3

               ; Compute direction: dir = p1 - p0
        %dir = OpFSub %v2float %p1 %p0

               ; Compute perpendicular: perp = normalize(-dir.y, dir.x)
      %dir_x = OpCompositeExtract %float %dir 0
      %dir_y = OpCompositeExtract %float %dir 1
  %neg_dir_y = OpFNegate %float %dir_y
   %perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x

               ; Normalize perpendicular
   %perp_dot = OpDot %float %perp_raw %perp_raw
   %perp_len = OpExtInst %float %1 Sqrt %perp_dot
  %perp_len2 = OpCompositeConstruct %v2float %perp_len %perp_len
       %perp = OpFDiv %v2float %perp_raw %perp_len2

               ; mid = p0 + 0.75 * dir
  %scale_075 = OpCompositeConstruct %v2float %float_0_75 %float_0_75
 %dir_scaled = OpFMul %v2float %dir %scale_075
        %mid = OpFAdd %v2float %p0 %dir_scaled

               ; Shaft thickness offset: perp * 0.015
 %scale_0015 = OpCompositeConstruct %v2float %float_0_015 %float_0_015
  %shaft_off = OpFMul %v2float %perp %scale_0015

               ; Arrow width offset: perp * 0.05
  %scale_005 = OpCompositeConstruct %v2float %float_0_05 %float_0_05
  %arrow_off = OpFMul %v2float %perp %scale_005

               ; Get output position pointer
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; === SHAFT (4 vertices) ===
               ; Vertex 1: p0 + perp * 0.015
   %shaft_v1 = OpFAdd %v2float %p0 %shaft_off
 %shaft_v1_x = OpCompositeExtract %float %shaft_v1 0
 %shaft_v1_y = OpCompositeExtract %float %shaft_v1 1
 %shaft_v1_4 = OpCompositeConstruct %v4float %shaft_v1_x %shaft_v1_y %p0_z %p0_w
               OpStore %out_pos_ptr %shaft_v1_4
               OpStore %outColor %c0
               OpEmitVertex

               ; Vertex 2: p0 - perp * 0.015
   %shaft_v2 = OpFSub %v2float %p0 %shaft_off
 %shaft_v2_x = OpCompositeExtract %float %shaft_v2 0
 %shaft_v2_y = OpCompositeExtract %float %shaft_v2 1
 %shaft_v2_4 = OpCompositeConstruct %v4float %shaft_v2_x %shaft_v2_y %p0_z %p0_w
               OpStore %out_pos_ptr %shaft_v2_4
               OpStore %outColor %c0
               OpEmitVertex

               ; Vertex 3: mid + perp * 0.015
   %shaft_v3 = OpFAdd %v2float %mid %shaft_off
 %shaft_v3_x = OpCompositeExtract %float %shaft_v3 0
 %shaft_v3_y = OpCompositeExtract %float %shaft_v3 1
 %shaft_v3_4 = OpCompositeConstruct %v4float %shaft_v3_x %shaft_v3_y %p0_z %p0_w
               OpStore %out_pos_ptr %shaft_v3_4
               OpStore %outColor %c0
               OpEmitVertex

               ; Vertex 4: mid - perp * 0.015
   %shaft_v4 = OpFSub %v2float %mid %shaft_off
 %shaft_v4_x = OpCompositeExtract %float %shaft_v4 0
 %shaft_v4_y = OpCompositeExtract %float %shaft_v4 1
 %shaft_v4_4 = OpCompositeConstruct %v4float %shaft_v4_x %shaft_v4_y %p0_z %p0_w
               OpStore %out_pos_ptr %shaft_v4_4
               OpStore %outColor %c0
               OpEmitVertex

               OpEndPrimitive

               ; === ARROWHEAD (3 vertices) ===
               ; Vertex 1: mid + perp * 0.05
   %arrow_v1 = OpFAdd %v2float %mid %arrow_off
 %arrow_v1_x = OpCompositeExtract %float %arrow_v1 0
 %arrow_v1_y = OpCompositeExtract %float %arrow_v1 1
 %arrow_v1_4 = OpCompositeConstruct %v4float %arrow_v1_x %arrow_v1_y %p0_z %p0_w
               OpStore %out_pos_ptr %arrow_v1_4
               OpStore %outColor %c1
               OpEmitVertex

               ; Vertex 2: mid - perp * 0.05
   %arrow_v2 = OpFSub %v2float %mid %arrow_off
 %arrow_v2_x = OpCompositeExtract %float %arrow_v2 0
 %arrow_v2_y = OpCompositeExtract %float %arrow_v2 1
 %arrow_v2_4 = OpCompositeConstruct %v4float %arrow_v2_x %arrow_v2_y %p0_z %p0_w
               OpStore %out_pos_ptr %arrow_v2_4
               OpStore %outColor %c1
               OpEmitVertex

               ; Vertex 3: p1 (tip)
               OpStore %out_pos_ptr %p1_v4
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
