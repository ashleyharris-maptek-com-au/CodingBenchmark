```cpp
#include <iostream>
#include <fstream>
#include <sstream>
#include <cmath>
#include <thread>
#include <atomic>
#include <mutex>
#include <string>
#include <chrono>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool valid = false;
};

std::mutex state_mutex;
State current_state;
std::atomic<bool> running{true};
std::atomic<bool> state_ready{false};

double width, depth, height, gravity, max_thrust, init_fuel;
double start_x, start_y, start_z;
double target_x, target_y, target_z;

constexpr double PI = 3.14159265358979323846;

double clamp(double v, double lo, double hi) {
    return v < lo ? lo : (v > hi ? hi : v);
}

void reader_thread() {
    std::string line;
    while (running && std::getline(std::cin, line)) {
        if (line.empty()) continue;

        std::istringstream iss(line);
        State s;
        if (iss >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                >> s.pitch >> s.yaw >> s.roll
                >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel) {
            s.valid = true;
            std::lock_guard<std::mutex> lock(state_mutex);
            current_state = s;
            state_ready = true;
        }
    }
    running = false;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);

    std::cin >> width >> depth >> height >> gravity >> max_thrust >> init_fuel;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    std::cin.ignore();

    std::string terrain_path;
    std::getline(std::cin, terrain_path);

    std::string state_marker;
    std::getline(std::cin, state_marker);

    std::thread reader(reader_thread);

    const double Kp_pos = 0.12;
    const double Kd_pos = 0.7;
    const double Kp_alt = 0.25;
    const double Kd_alt = 1.0;
    const double Kp_angle = 2.5;
    const double Kd_angle = 1.8;

    while (running) {
        State s;
        {
            std::lock_guard<std::mutex> lock(state_mutex);
            s = current_state;
        }

        if (!state_ready || !s.valid) {
            std::cout << "0.5 0 0 0\n";
            std::cout.flush();
            std::this_thread::sleep_for(std::chrono::milliseconds(50));
            continue;
        }

        double dx = target_x - s.x;
        double dy = target_y - s.y;
        double dz = target_z - s.z;
        double dist_horiz = std::sqrt(dx*dx + dy*dy);
        double dist_total = std::sqrt(dx*dx + dy*dy + dz*dz);

        double ax_desired = Kp_pos * dx - Kd_pos * s.vx;
        double ay_desired = Kp_pos * dy - Kd_pos * s.vy;

        double speed_horiz = std::sqrt(s.vx*s.vx + s.vy*s.vy);

        double target_vz;
        if (dz < -1.0) {
            double safe_descent = -std::sqrt(std::max(0.5, 2.0 * 0.4 * gravity * std::abs(dz)));
            safe_descent = std::max(safe_descent, -2.5);
            if (dist_horiz > 10.0 || speed_horiz > 2.0) {
                safe_descent = std::max(safe_descent, -1.0);
            }
            target_vz = safe_descent;
        } else if (dz < 0) {
            target_vz = -0.5;
        } else {
            target_vz = std::min(2.0, Kp_alt * dz);
        }

        if (dist_horiz > 5.0 && dz < 5.0) {
            target_vz = std::max(target_vz, 0.5);
        }

        double az_desired = Kd_alt * (target_vz - s.vz) + gravity;

        if (dist_total < 15.0 && dist_horiz < 3.0) {
            ax_desired *= 0.5;
            ay_desired *= 0.5;
        }

        double a_horiz = std::sqrt(ax_desired*ax_desired + ay_desired*ay_desired);
        double a_total = std::sqrt(a_horiz*a_horiz + az_desired*az_desired);

        double thrust = clamp(a_total / max_thrust, 0.0, 1.0);

        double target_pitch = 0.0;
        double target_yaw = s.yaw;

        if (az_desired > 0.01) {
            target_pitch = std::atan2(a_horiz, az_desired);
            target_pitch = clamp(target_pitch, -1.0, 1.0);

            if (a_horiz > 0.01) {
                target_yaw = std::atan2(-ax_desired, ay_desired);
            }
        }

        double pitch_error = target_pitch - s.pitch;
        double pitch_cmd = Kp_angle * pitch_error - Kd_angle * s.pitch_rate;
        pitch_cmd = clamp(pitch_cmd, -1.0, 1.0);

        double yaw_error = target_yaw - s.yaw;
        while (yaw_error > PI) yaw_error -= 2.0 * PI;
        while (yaw_error < -PI) yaw_error += 2.0 * PI;
        double yaw_cmd = Kp_angle * yaw_error - Kd_angle * s.yaw_rate;
        yaw_cmd = clamp(yaw_cmd, -1.0, 1.0);

        double roll_cmd = -Kd_angle * s.roll_rate - 0.5 * s.roll;
        roll_cmd = clamp(roll_cmd, -1.0, 1.0);

        if (dist_total < 8.0 && dist_horiz < 2.0 && std::abs(s.vz) < 2.0) {
            pitch_cmd *= 0.3;
            yaw_cmd *= 0.2;
            thrust = clamp((gravity + 0.5 * (target_vz - s.vz)) / max_thrust, 0.0, 1.0);
        }

        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();

        std::this_thread::sleep_for(std::chrono::milliseconds(50));
    }

    if (reader.joinable()) {
        reader.join();
    }
    return 0;
}
```
