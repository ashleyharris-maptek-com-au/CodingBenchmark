struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Polka dot pattern: 6x6 grid
    float2 cellUV = input.uv * 6.0;
    float2 cellCenter = frac(cellUV) - 0.5;
    float dist = length(cellCenter);

    // Dot radius 0.3 in cell space
    float dotRadius = 0.3;

    // Colors
    float3 dotColor = float3(1.0, 0.8, 0.0);
    float3 bgColor = float3(0.1, 0.1, 0.3);

    // Select color based on distance from cell center
    float3 baseColor = (dist < dotRadius) ? dotColor : bgColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.3 + 0.7 * diffuse;

    float3 finalColor = baseColor * shading;

    return float4(finalColor, 1.0);
}
