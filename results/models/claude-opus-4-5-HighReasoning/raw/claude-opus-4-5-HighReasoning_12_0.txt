using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;

class Program
{
    static void Main()
    {
        var firstLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(firstLine[0]);
        int F = int.Parse(firstLine[1]);

        Vector3[] vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            vertices[i] = new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2]));
        }

        for (int i = 0; i < F; i++) Console.ReadLine();

        var lastLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        float W = float.Parse(lastLine[0]);
        float H = float.Parse(lastLine[1]);
        float D = float.Parse(lastLine[2]);

        var rotations = GetRotations();
        var placements = new List<(Vector3 trans, Quaternion rot)>();

        Vector3 bestSize = Vector3.Zero;
        Quaternion bestRot = Quaternion.Identity;
        float bestVolume = float.MaxValue;

        foreach (var rot in rotations)
        {
            var (min, max) = GetBoundingBox(vertices, rot);
            var size = max - min;
            float vol = size.X * size.Y * size.Z;
            if (vol < bestVolume)
            {
                bestVolume = vol;
                bestSize = size;
                bestRot = rot;
            }
        }

        var (bmin, bmax) = GetBoundingBox(vertices, bestRot);
        float sizeX = bmax.X - bmin.X;
        float sizeY = bmax.Y - bmin.Y;
        float sizeZ = bmax.Z - bmin.Z;

        if (sizeX <= 0) sizeX = 0.001f;
        if (sizeY <= 0) sizeY = 0.001f;
        if (sizeZ <= 0) sizeZ = 0.001f;

        float margin = 0.001f;
        float stepX = sizeX + margin;
        float stepY = sizeY + margin;
        float stepZ = sizeZ + margin;

        int countX = (int)Math.Floor(W / stepX);
        int countY = (int)Math.Floor(H / stepY);
        int countZ = (int)Math.Floor(D / stepZ);

        if (countX < 1) countX = 1;
        if (countY < 1) countY = 1;
        if (countZ < 1) countZ = 1;

        float offsetX = -bmin.X;
        float offsetY = -bmin.Y;
        float offsetZ = -bmin.Z;

        for (int ix = 0; ix < countX; ix++)
        {
            for (int iy = 0; iy < countY; iy++)
            {
                for (int iz = 0; iz < countZ; iz++)
                {
                    float tx = ix * stepX + offsetX;
                    float ty = iy * stepY + offsetY;
                    float tz = iz * stepZ + offsetZ;

                    if (IsValid(vertices, bestRot, new Vector3(tx, ty, tz), W, H, D))
                    {
                        placements.Add((new Vector3(tx, ty, tz), bestRot));
                    }
                }
            }
        }

        Console.WriteLine(placements.Count);
        foreach (var (trans, rot) in placements)
        {
            Console.WriteLine($"{trans.X:F6} {trans.Y:F6} {trans.Z:F6} {rot.W:F6} {rot.X:F6} {rot.Y:F6} {rot.Z:F6}");
        }
    }

    static (Vector3 min, Vector3 max) GetBoundingBox(Vector3[] vertices, Quaternion rot)
    {
        Vector3 min = new Vector3(float.MaxValue);
        Vector3 max = new Vector3(float.MinValue);

        foreach (var v in vertices)
        {
            var rv = Vector3.Transform(v, rot);
            min = Vector3.Min(min, rv);
            max = Vector3.Max(max, rv);
        }
        return (min, max);
    }

    static bool IsValid(Vector3[] vertices, Quaternion rot, Vector3 trans, float W, float H, float D)
    {
        foreach (var v in vertices)
        {
            var rv = Vector3.Transform(v, rot) + trans;
            if (rv.X < -0.0001f || rv.X > W + 0.0001f ||
                rv.Y < -0.0001f || rv.Y > H + 0.0001f ||
                rv.Z < -0.0001f || rv.Z > D + 0.0001f)
                return false;
        }
        return true;
    }

    static List<Quaternion> GetRotations()
    {
        var rots = new List<Quaternion>();
        float[] angles = { 0, MathF.PI / 2, MathF.PI, 3 * MathF.PI / 2 };

        foreach (float ax in angles)
        {
            foreach (float ay in angles)
            {
                foreach (float az in angles)
                {
                    var qx = Quaternion.CreateFromAxisAngle(Vector3.UnitX, ax);
                    var qy = Quaternion.CreateFromAxisAngle(Vector3.UnitY, ay);
                    var qz = Quaternion.CreateFromAxisAngle(Vector3.UnitZ, az);
                    rots.Add(Quaternion.Normalize(qx * qy * qz));
                }
            }
        }

        var unique = new List<Quaternion>();
        foreach (var r in rots)
        {
            bool found = false;
            foreach (var u in unique)
            {
                if (Math.Abs(Quaternion.Dot(r, u)) > 0.999f)
                {
                    found = true;
                    break;
                }
            }
            if (!found) unique.Add(r);
        }
        return unique;
    }
}
