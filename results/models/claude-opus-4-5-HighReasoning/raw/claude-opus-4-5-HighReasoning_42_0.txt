; SPIR-V Geometry Shader - Expand points to quads
; Vulkan 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Decorations for user-defined input/output
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void = OpTypeVoid
  %voidfunc = OpTypeFunction %void
     %float = OpTypeFloat 32
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0

               ; Constants
     %int_0 = OpConstant %int 0
    %uint_1 = OpConstant %uint 1
   %float_0 = OpConstant %float 0
   %float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04

               ; Input gl_PerVertex struct and array (size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

               ; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

               ; Input color array (size 1 for InputPoints)
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
   %inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

               ; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3 Output

               ; Main function
      %main = OpFunction %void None %voidfunc
     %entry = OpLabel

               ; Load input position: gl_in[0].gl_Position
   %pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %pos4 = OpLoad %v4float %pos_ptr
         %x = OpCompositeExtract %float %pos4 0
         %y = OpCompositeExtract %float %pos4 1

               ; Load input color: color[0]
 %color_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
     %color = OpLoad %v3float %color_ptr

               ; Calculate quad corner positions
               ; h = 0.04
      %x_mh = OpFSub %float %x %float_0_04
      %x_ph = OpFAdd %float %x %float_0_04
      %y_mh = OpFSub %float %y %float_0_04
      %y_ph = OpFAdd %float %y %float_0_04

               ; Output position pointer
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; Vertex 1: (x-h, y-h, 0, 1)
     %pos_1 = OpCompositeConstruct %v4float %x_mh %y_mh %float_0 %float_1
               OpStore %out_pos_ptr %pos_1
               OpStore %outColor %color
               OpEmitVertex

               ; Vertex 2: (x+h, y-h, 0, 1)
     %pos_2 = OpCompositeConstruct %v4float %x_ph %y_mh %float_0 %float_1
               OpStore %out_pos_ptr %pos_2
               OpStore %outColor %color
               OpEmitVertex

               ; Vertex 3: (x-h, y+h, 0, 1)
     %pos_3 = OpCompositeConstruct %v4float %x_mh %y_ph %float_0 %float_1
               OpStore %out_pos_ptr %pos_3
               OpStore %outColor %color
               OpEmitVertex

               ; Vertex 4: (x+h, y+h, 0, 1)
     %pos_4 = OpCompositeConstruct %v4float %x_ph %y_ph %float_0 %float_1
               OpStore %out_pos_ptr %pos_4
               OpStore %outColor %color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
