import sys
import random

def main():
    data = sys.stdin.buffer.read()
    tokens = data.split()
    n = int(tokens[0])
    m = int(tokens[1])

    clauses = [(int(tokens[2 + 3*i]), int(tokens[3 + 3*i]), int(tokens[4 + 3*i])) for i in range(m)]

    pos_list = [[] for _ in range(n + 1)]
    neg_list = [[] for _ in range(n + 1)]
    for i, (a, b, c) in enumerate(clauses):
        for lit in (a, b, c):
            if lit > 0:
                pos_list[lit].append(i)
            else:
                neg_list[-lit].append(i)

    assign = [False] * (n + 1)
    for v in range(1, n + 1):
        assign[v] = len(pos_list[v]) >= len(neg_list[v])

    true_count = [0] * m
    for i, (a, b, c) in enumerate(clauses):
        for lit in (a, b, c):
            v = abs(lit)
            if (lit > 0) == assign[v]:
                true_count[i] += 1

    unsat = []
    in_unsat = [False] * m
    for i in range(m):
        if true_count[i] == 0:
            in_unsat[i] = True
            unsat.append(i)

    noise = 0.57
    max_flips = 20 * m + 1000000

    for _ in range(max_flips):
        if not unsat:
            break

        idx = random.randint(0, len(unsat) - 1)
        ci = unsat[idx]

        if not in_unsat[ci]:
            unsat[idx] = unsat[-1]
            unsat.pop()
            continue

        a, b, c = clauses[ci]
        lits = (a, b, c)

        if random.random() < noise:
            var = abs(random.choice(lits))
        else:
            best_var = abs(a)
            best_brk = float('inf')
            for lit in lits:
                v = abs(lit)
                brk = 0
                check_list = pos_list[v] if assign[v] else neg_list[v]
                for cj in check_list:
                    if true_count[cj] == 1:
                        brk += 1
                if brk < best_brk:
                    best_brk = brk
                    best_var = v
                    if brk == 0:
                        break
            var = best_var

        old = assign[var]
        assign[var] = not old

        if old:
            for cj in pos_list[var]:
                true_count[cj] -= 1
                if true_count[cj] == 0 and not in_unsat[cj]:
                    in_unsat[cj] = True
                    unsat.append(cj)
            for cj in neg_list[var]:
                true_count[cj] += 1
                if true_count[cj] == 1 and in_unsat[cj]:
                    in_unsat[cj] = False
        else:
            for cj in neg_list[var]:
                true_count[cj] -= 1
                if true_count[cj] == 0 and not in_unsat[cj]:
                    in_unsat[cj] = True
                    unsat.append(cj)
            for cj in pos_list[var]:
                true_count[cj] += 1
                if true_count[cj] == 1 and in_unsat[cj]:
                    in_unsat[cj] = False

    sys.stdout.write("SAT\n")
    sys.stdout.write(" ".join("1" if assign[i] else "0" for i in range(1, n + 1)))
    sys.stdout.write("\n")

if __name__ == "__main__":
    main()
