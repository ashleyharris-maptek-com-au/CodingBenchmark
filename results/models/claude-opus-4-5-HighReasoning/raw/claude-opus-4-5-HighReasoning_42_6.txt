; SPIR-V Geometry Shader - Identity/Passthrough for Triangles
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

; Input gl_PerVertex block (array of 3 for triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in_3
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

; Input color array (location 2)
%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_3
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3float = OpTypePointer Input %v3float

; Output color (location 0)
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; Main function
%main = OpFunction %void None %void_func
%entry = OpLabel

; Vertex 0
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%out_pos0_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos0_ptr %pos0
%col0_ptr = OpAccessChain %ptr_in_v3float %inColor %int_0
%col0 = OpLoad %v3float %col0_ptr
OpStore %outColor %col0
OpEmitVertex

; Vertex 1
%pos1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%out_pos1_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos1_ptr %pos1
%col1_ptr = OpAccessChain %ptr_in_v3float %inColor %int_1
%col1 = OpLoad %v3float %col1_ptr
OpStore %outColor %col1
OpEmitVertex

; Vertex 2
%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%out_pos2_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos2_ptr %pos2
%col2_ptr = OpAccessChain %ptr_in_v3float %inColor %int_2
%col2 = OpLoad %v3float %col2_ptr
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
