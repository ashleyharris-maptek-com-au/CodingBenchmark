; SPIR-V Geometry Shader - Silhouette Edge Detection
; Input: TrianglesAdjacency (6 vertices), Output: TriangleStrip
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%bool = OpTypeBool

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_n0_015 = OpConstant %float -0.015

%white = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Input gl_PerVertex block (array of 6 for triangles adjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_6
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

; Input color array (location 2)
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%ptr_in_v3 = OpTypePointer Input %v3float
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load main triangle positions (indices 0, 2, 4)
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

; Load colors for main triangle
%col0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%col4 = OpLoad %v3float %col4_ptr

; Dim colors for triangle fill (color * 0.3)
%dim0 = OpVectorTimesScalar %v3float %col0 %float_0_3
%dim2 = OpVectorTimesScalar %v3float %col2 %float_0_3
%dim4 = OpVectorTimesScalar %v3float %col4 %float_0_3

; === Emit edge 0->2 as thick line (quad) ===
; Compute perpendicular offset
%e02 = OpFSub %v4float %pos2 %pos0
%e02_x = OpCompositeExtract %float %e02 0
%e02_y = OpCompositeExtract %float %e02 1
%perp02_x = OpFNegate %float %e02_y
%perp02_y = OpCopyObject %float %e02_x
; Normalize perpendicular (simplified - just scale by thickness)
%off02 = OpCompositeConstruct %v4float %perp02_x %perp02_y %float_0 %float_0
%off02_scaled = OpVectorTimesScalar %v4float %off02 %float_0_015

; Emit 4 vertices for edge quad
%q0_a = OpFAdd %v4float %pos0 %off02_scaled
%q0_b = OpFSub %v4float %pos0 %off02_scaled
%q2_a = OpFAdd %v4float %pos2 %off02_scaled
%q2_b = OpFSub %v4float %pos2 %off02_scaled

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr %q0_a
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q0_b
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q2_a
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q2_b
OpStore %outColor %white
OpEmitVertex

OpEndPrimitive

; === Emit edge 2->4 as thick line (quad) ===
%e24 = OpFSub %v4float %pos4 %pos2
%e24_x = OpCompositeExtract %float %e24 0
%e24_y = OpCompositeExtract %float %e24 1
%perp24_x = OpFNegate %float %e24_y
%perp24_y = OpCopyObject %float %e24_x
%off24 = OpCompositeConstruct %v4float %perp24_x %perp24_y %float_0 %float_0
%off24_scaled = OpVectorTimesScalar %v4float %off24 %float_0_015

%q2_c = OpFAdd %v4float %pos2 %off24_scaled
%q2_d = OpFSub %v4float %pos2 %off24_scaled
%q4_a = OpFAdd %v4float %pos4 %off24_scaled
%q4_b = OpFSub %v4float %pos4 %off24_scaled

OpStore %out_pos_ptr %q2_c
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q2_d
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q4_a
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q4_b
OpStore %outColor %white
OpEmitVertex

OpEndPrimitive

; === Emit edge 4->0 as thick line (quad) ===
%e40 = OpFSub %v4float %pos0 %pos4
%e40_x = OpCompositeExtract %float %e40 0
%e40_y = OpCompositeExtract %float %e40 1
%perp40_x = OpFNegate %float %e40_y
%perp40_y = OpCopyObject %float %e40_x
%off40 = OpCompositeConstruct %v4float %perp40_x %perp40_y %float_0 %float_0
%off40_scaled = OpVectorTimesScalar %v4float %off40 %float_0_015

%q4_c = OpFAdd %v4float %pos4 %off40_scaled
%q4_d = OpFSub %v4float %pos4 %off40_scaled
%q0_c = OpFAdd %v4float %pos0 %off40_scaled
%q0_d = OpFSub %v4float %pos0 %off40_scaled

OpStore %out_pos_ptr %q4_c
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q4_d
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q0_c
OpStore %outColor %white
OpEmitVertex

OpStore %out_pos_ptr %q0_d
OpStore %outColor %white
OpEmitVertex

OpEndPrimitive

; === Emit the original triangle with dimmed colors ===
OpStore %out_pos_ptr %pos0
OpStore %outColor %dim0
OpEmitVertex

OpStore %out_pos_ptr %pos2
OpStore %outColor %dim2
OpEmitVertex

OpStore %out_pos_ptr %pos4
OpStore %outColor %dim4
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
