use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj = vec![vec![]; n];

    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let half = n / 2;
    let mut best_cut = usize::MAX;
    let mut best_partition: Vec<usize> = vec![];
    let mut partition_a = Vec::with_capacity(half);
    let mut in_a = vec![false; n];

    backtrack(0, n, half, 0, &mut partition_a, &mut in_a, &adj, 0, &mut best_cut, &mut best_partition);

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", best_cut).unwrap();
    writeln!(out, "{}", best_partition.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ")).unwrap();
}

fn backtrack(
    v: usize,
    n: usize,
    half: usize,
    count_a: usize,
    partition_a: &mut Vec<usize>,
    in_a: &mut [bool],
    adj: &[Vec<usize>],
    cut: usize,
    best_cut: &mut usize,
    best_partition: &mut Vec<usize>,
) {
    if cut >= *best_cut {
        return;
    }

    if count_a == half {
        let mut final_cut = cut;
        for u in v..n {
            for &neighbor in &adj[u] {
                if neighbor < u && in_a[neighbor] {
                    final_cut += 1;
                }
            }
        }
        if final_cut < *best_cut {
            *best_cut = final_cut;
            *best_partition = partition_a.clone();
        }
        return;
    }

    if v >= n {
        return;
    }

    let remaining = n - v;
    let need_a = half - count_a;
    let need_b = half - (v - count_a);

    if need_a > remaining || need_b > remaining {
        return;
    }

    // Put v in A
    partition_a.push(v);
    in_a[v] = true;
    let mut added = 0;
    for &neighbor in &adj[v] {
        if neighbor < v && !in_a[neighbor] {
            added += 1;
        }
    }
    backtrack(v + 1, n, half, count_a + 1, partition_a, in_a, adj, cut + added, best_cut, best_partition);
    partition_a.pop();
    in_a[v] = false;

    // Put v in B
    let mut added_b = 0;
    for &neighbor in &adj[v] {
        if neighbor < v && in_a[neighbor] {
            added_b += 1;
        }
    }
    backtrack(v + 1, n, half, count_a, partition_a, in_a, adj, cut + added_b, best_cut, best_partition);
}
