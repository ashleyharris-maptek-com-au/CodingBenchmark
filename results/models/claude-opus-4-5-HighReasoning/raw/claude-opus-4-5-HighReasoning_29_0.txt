using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class MaxClique
{
    static int n;
    static HashSet<int>[] adj;
    static int[] bestClique;
    static int bestSize = 0;
    static object lockObj = new object();
    static Stopwatch sw;
    static readonly long TimeLimit = 270000; // 4.5 minutes in ms
    static volatile bool timeExpired = false;

    static void Main()
    {
        sw = Stopwatch.StartNew();

        var firstLine = Console.ReadLine().Split(' ');
        n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split(' ');
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        bestClique = new int[0];

        // Start timeout monitor
        Task.Run(() => {
            while (sw.ElapsedMilliseconds < TimeLimit) Thread.Sleep(100);
            timeExpired = true;
        });

        // Calculate density
        double density = n > 1 ? (2.0 * m) / (n * (n - 1.0)) : 0;

        // For very large sparse graphs, use greedy only
        if (n > 10000)
        {
            RunParallelGreedy();
        }
        else if (n > 500 || (n > 200 && density > 0.5))
        {
            // Run greedy first to get a good lower bound
            var greedyTask = Task.Run(() => RunParallelGreedy());

            // Then try exact with time limit
            try { BranchAndBound(); } catch { }
        }
        else
        {
            // Small graph - exact algorithm
            BronKerbosch();
        }

        // Output result
        Console.WriteLine(bestSize);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }

    static void RunParallelGreedy()
    {
        int[] degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adj[i].Count;

        // Multiple greedy strategies in parallel
        Parallel.For(0, Math.Min(Environment.ProcessorCount, 12), pid =>
        {
            if (timeExpired) return;

            List<int> vertices = Enumerable.Range(0, n).ToList();
            Random rnd = new Random(pid * 1000);

            // Different orderings
            if (pid % 4 == 0)
                vertices = vertices.OrderByDescending(v => degrees[v]).ToList();
            else if (pid % 4 == 1)
                vertices = vertices.OrderBy(v => degrees[v]).ToList();
            else if (pid % 4 == 2)
                vertices = vertices.OrderByDescending(v => degrees[v] + rnd.NextDouble() * 10).ToList();
            else
                vertices = vertices.OrderBy(_ => rnd.Next()).ToList();

            // Run greedy multiple times with different starts
            for (int iter = 0; iter < (n > 5000 ? 100 : 500) && !timeExpired; iter++)
            {
                var clique = GreedyClique(vertices, rnd);
                LocalSearch(clique, rnd);

                lock (lockObj)
                {
                    if (clique.Count > bestSize)
                    {
                        bestSize = clique.Count;
                        bestClique = clique.ToArray();
                    }
                }

                // Shuffle for next iteration
                for (int i = vertices.Count - 1; i > 0; i--)
                {
                    int j = rnd.Next(i + 1);
                    (vertices[i], vertices[j]) = (vertices[j], vertices[i]);
                }
            }
        });
    }

    static List<int> GreedyClique(List<int> vertices, Random rnd)
    {
        List<int> clique = new List<int>();
        HashSet<int> candidates = new HashSet<int>(vertices);

        while (candidates.Count > 0)
        {
            // Pick vertex with most connections to candidates
            int best = -1;
            int bestConn = -1;

            foreach (int v in candidates)
            {
                int conn = 0;
                foreach (int u in candidates)
                    if (u != v && adj[v].Contains(u)) conn++;

                if (conn > bestConn || (conn == bestConn && rnd.Next(2) == 0))
                {
                    bestConn = conn;
                    best = v;
                }
            }

            if (best == -1) break;

            clique.Add(best);
            candidates.RemoveWhere(v => v == best || !adj[best].Contains(v));
        }

        return clique;
    }

    static void LocalSearch(List<int> clique, Random rnd)
    {
        HashSet<int> cliqueSet = new HashSet<int>(clique);

        for (int iter = 0; iter < 100 && !timeExpired; iter++)
        {
            bool improved = false;

            // Try adding vertices
            for (int v = 0; v < n && !timeExpired; v++)
            {
                if (cliqueSet.Contains(v)) continue;

                bool canAdd = true;
                foreach (int u in cliqueSet)
                {
                    if (!adj[v].Contains(u)) { canAdd = false; break; }
                }

                if (canAdd)
                {
                    cliqueSet.Add(v);
                    clique.Add(v);
                    improved = true;
                }
            }

            if (!improved)
            {
                // Try swap: remove one, add two
                foreach (int rem in cliqueSet.ToList())
                {
                    if (timeExpired) break;

                    var tempClique = new HashSet<int>(cliqueSet);
                    tempClique.Remove(rem);

                    // Find vertices that connect to all except rem
                    var candidates = new List<int>();
                    for (int v = 0; v < n; v++)
                    {
                        if (cliqueSet.Contains(v)) continue;
                        bool ok = true;
                        foreach (int u in tempClique)
                            if (!adj[v].Contains(u)) { ok = false; break; }
                        if (ok) candidates.Add(v);
                    }

                    // Try to add two from candidates
                    for (int i = 0; i < candidates.Count && !improved; i++)
                    {
                        for (int j = i + 1; j < candidates.Count; j++)
                        {
                            if (adj[candidates[i]].Contains(candidates[j]))
                            {
                                cliqueSet.Remove(rem);
                                cliqueSet.Add(candidates[i]);
                                cliqueSet.Add(candidates[j]);
                                clique.Clear();
                                clique.AddRange(cliqueSet);
                                improved = true;
                                break;
                            }
                        }
                    }
                    if (improved) break;
                }
            }

            if (!improved) break;
        }
    }

    static void BronKerbosch()
    {
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => adj[v].Count).ToArray();
        int[] pos = new int[n];
        for (int i = 0; i < n; i++) pos[order[i]] = i;

        // Run with different starting points in parallel
        Parallel.For(0, n, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, i =>
        {
            if (timeExpired) return;

            int v = order[i];
            List<int> R = new List<int> { v };
            List<int> P = new List<int>();
            List<int> X = new List<int>();

            foreach (int u in adj[v])
            {
                if (pos[u] > i) P.Add(u);
                else X.Add(u);
            }

            BK(R, P, X);
        });
    }

    static void BK(List<int> R, List<int> P, List<int> X)
    {
        if (timeExpired) return;

        if (P.Count == 0 && X.Count == 0)
        {
            lock (lockObj)
            {
                if (R.Count > bestSize)
                {
                    bestSize = R.Count;
                    bestClique = R.ToArray();
                }
            }
            return;
        }

        // Pruning
        lock (lockObj) { if (R.Count + P.Count <= bestSize) return; }

        // Pivot selection
        int pivot = -1;
        int maxConn = -1;
        foreach (int u in P.Concat(X))
        {
            int conn = P.Count(v => adj[u].Contains(v));
            if (conn > maxConn) { maxConn = conn; pivot = u; }
        }

        var candidates = pivot >= 0 ? P.Where(v => !adj[pivot].Contains(v)).ToList() : P.ToList();

        foreach (int v in candidates)
        {
            if (timeExpired) return;

            var newR = new List<int>(R) { v };
            var newP = P.Where(u => adj[v].Contains(u)).ToList();
            var newX = X.Where(u => adj[v].Contains(u)).ToList();

            BK(newR, newP, newX);

            P.Remove(v);
            X.Add(v);
        }
    }

    static void BranchAndBound()
    {
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => adj[v].Count).ToArray();

        // Greedy coloring for upper bound
        int[] color = new int[n];
        int numColors = 0;
        for (int i = 0; i < n; i++)
        {
            int v = order[i];
            bool[] used = new bool[n + 1];
            foreach (int u in adj[v])
                if (color[u] > 0) used[color[u]] = true;

            for (int c = 1; ; c++)
            {
                if (!used[c]) { color[v] = c; numColors = Math.Max(numColors, c); break; }
            }
        }

        // Run greedy first
        RunParallelGreedy();

        // Branch and bound with coloring
        int[] current = new int[n];
        int size = 0;
        BBSearch(order, 0, current, ref size, color);
    }

    static void BBSearch(int[] order, int idx, int[] current, ref int size, int[] color)
    {
        if (timeExpired) return;

        lock (lockObj)
        {
            if (size > bestSize)
            {
                bestSize = size;
                bestClique = current.Take(size).ToArray();
            }
        }

        if (idx >= n) return;

        // Build candidate set
        List<int> candidates = new List<int>();
        for (int i = idx; i < n; i++)
        {
            int v = order[i];
            bool ok = true;
            for (int j = 0; j < size; j++)
                if (!adj[current[j]].Contains(v)) { ok = false; break; }
            if (ok) candidates.Add(v);
        }

        // Upper bound pruning
        lock (lockObj) { if (size + candidates.Count <= bestSize) return; }

        // Color-based upper bound
        int[] colorCount = new int[n + 1];
        foreach (int v in candidates) colorCount[color[v]]++;
        int ub = size + colorCount.Count(c => c > 0);
        lock (lockObj) { if (ub <= bestSize) return; }

        foreach (int v in candidates.OrderByDescending(x => adj[x].Count))
        {
            if (timeExpired) return;

            current[size] = v;
            size++;
            BBSearch(order, Array.IndexOf(order, v) + 1, current, ref size, color);
            size--;

            lock (lockObj) { if (size + candidates.Count <= bestSize) return; }
        }
    }
}
