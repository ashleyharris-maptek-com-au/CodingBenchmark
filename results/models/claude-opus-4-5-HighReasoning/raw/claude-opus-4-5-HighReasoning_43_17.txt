struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Scale UV for brick pattern: 8 bricks horizontally, 16 vertically
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Determine row and apply half-brick offset for odd rows
    float row = floor(scaledUV.y);
    float2 brickUV = scaledUV;
    if (fmod(row, 2.0) >= 1.0) {
        brickUV.x += 0.5;
    }

    // Get position within current brick cell
    float2 cellUV = frac(brickUV);

    // Mortar thickness: 0.05 in X, 0.1 in Y
    float mortarX = 0.05;
    float mortarY = 0.1;

    // Check if in mortar region
    bool inMortar = (cellUV.x < mortarX) || (cellUV.y < mortarY);

    // Choose color based on mortar or brick
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 baseColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diffuse;

    float3 finalColor = baseColor * lighting;

    return float4(finalColor, 1.0);
}
