using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    const double MU = 398600.4418; // km³/s²
    const double EARTH_RADIUS = 6371.0; // km

    static double[] startPos, startVel;
    static double[][] stationPos, stationVel;
    static double[][] costMatrix;
    static int N;
    static Random rng = new Random(42);

    static void Main()
    {
        ReadInput();
        BuildCostMatrix();

        int[] order;
        if (N <= 8)
            order = BruteForce();
        else if (N <= 20)
            order = NearestNeighborWith2Opt();
        else
            order = SimulatedAnnealing();

        Console.WriteLine(string.Join(" ", order));
    }

    static void ReadInput()
    {
        var line1 = Console.ReadLine().Split();
        N = int.Parse(line1[0]);

        var startLine = Console.ReadLine().Split();
        startPos = new double[] { double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2]) };
        startVel = new double[] { double.Parse(startLine[3]), double.Parse(startLine[4]), double.Parse(startLine[5]) };

        stationPos = new double[N][];
        stationVel = new double[N][];

        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Split();
            stationPos[i] = new double[] { double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2]) };
            stationVel[i] = new double[] { double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5]) };
        }
    }

    static double Magnitude(double[] v) => Math.Sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2]);

    static double[] Cross(double[] a, double[] b) => new double[] {
        a[1]*b[2] - a[2]*b[1],
        a[2]*b[0] - a[0]*b[2],
        a[0]*b[1] - a[1]*b[0]
    };

    static double Dot(double[] a, double[] b) => a[0]*b[0] + a[1]*b[1] + a[2]*b[2];

    static (double a, double e, double i, double[] h) GetOrbitalElements(double[] r, double[] v)
    {
        double rMag = Magnitude(r);
        double vMag = Magnitude(v);
        double[] h = Cross(r, v);
        double hMag = Magnitude(h);

        double energy = vMag*vMag/2 - MU/rMag;
        double a = -MU / (2 * energy);

        double[] eVec = new double[3];
        for (int i = 0; i < 3; i++)
            eVec[i] = (vMag*vMag/MU - 1/rMag)*r[i] - Dot(r,v)/MU*v[i];
        double e = Magnitude(eVec);

        double inc = Math.Acos(h[2] / hMag);

        return (a, e, inc, h);
    }

    static double EstimateDeltaV(double[] r1, double[] v1, double[] r2, double[] v2)
    {
        var (a1, e1, i1, h1) = GetOrbitalElements(r1, v1);
        var (a2, e2, i2, h2) = GetOrbitalElements(r2, v2);

        // Handle edge cases
        if (double.IsNaN(a1) || double.IsNaN(a2) || a1 <= 0 || a2 <= 0)
        {
            double dist = Math.Sqrt(
                Math.Pow(r1[0]-r2[0], 2) +
                Math.Pow(r1[1]-r2[1], 2) +
                Math.Pow(r1[2]-r2[2], 2));
            return dist * 0.001; // Fallback approximation
        }

        // Hohmann-like transfer approximation
        double r1Mag = Magnitude(r1);
        double r2Mag = Magnitude(r2);

        double v1Circ = Math.Sqrt(MU / a1);
        double v2Circ = Math.Sqrt(MU / a2);

        // Coplanar Hohmann
        double aTransfer = (a1 + a2) / 2;
        double vPerigee = Math.Sqrt(MU * (2/a1 - 1/aTransfer));
        double vApogee = Math.Sqrt(MU * (2/a2 - 1/aTransfer));

        double dv1 = Math.Abs(vPerigee - v1Circ);
        double dv2 = Math.Abs(v2Circ - vApogee);
        double hohmannDV = dv1 + dv2;

        // Plane change cost
        double incDiff = Math.Abs(i1 - i2);
        double planeChangeDV = 2 * v1Circ * Math.Sin(incDiff / 2);

        return hohmannDV + planeChangeDV;
    }

    static void BuildCostMatrix()
    {
        // N+1 nodes: 0 = start, 1..N = stations
        costMatrix = new double[N + 1][];
        for (int i = 0; i <= N; i++)
            costMatrix[i] = new double[N + 1];

        // Start to each station
        for (int j = 0; j < N; j++)
            costMatrix[0][j + 1] = EstimateDeltaV(startPos, startVel, stationPos[j], stationVel[j]);

        // Station to station
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                if (i != j)
                    costMatrix[i + 1][j + 1] = EstimateDeltaV(stationPos[i], stationVel[i], stationPos[j], stationVel[j]);
    }

    static double TourCost(int[] order)
    {
        double cost = costMatrix[0][order[0] + 1]; // Start to first
        for (int i = 0; i < order.Length - 1; i++)
            cost += costMatrix[order[i] + 1][order[i + 1] + 1];
        return cost;
    }

    static int[] BruteForce()
    {
        var indices = Enumerable.Range(0, N).ToArray();
        var best = (int[])indices.Clone();
        double bestCost = TourCost(best);

        foreach (var perm in Permutations(indices))
        {
            double cost = TourCost(perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                best = (int[])perm.Clone();
            }
        }
        return best;
    }

    static IEnumerable<int[]> Permutations(int[] arr)
    {
        if (arr.Length <= 1) { yield return arr; yield break; }

        var c = new int[arr.Length];
        yield return arr;

        int i = 0;
        while (i < arr.Length)
        {
            if (c[i] < i)
            {
                if (i % 2 == 0)
                    (arr[0], arr[i]) = (arr[i], arr[0]);
                else
                    (arr[c[i]], arr[i]) = (arr[i], arr[c[i]]);
                yield return arr;
                c[i]++;
                i = 0;
            }
            else
            {
                c[i] = 0;
                i++;
            }
        }
    }

    static int[] NearestNeighbor()
    {
        var visited = new bool[N];
        var order = new List<int>();
        int current = -1; // Start position

        for (int step = 0; step < N; step++)
        {
            double bestDist = double.MaxValue;
            int bestNext = -1;

            for (int j = 0; j < N; j++)
            {
                if (visited[j]) continue;
                double dist = current == -1 ? costMatrix[0][j + 1] : costMatrix[current + 1][j + 1];
                if (dist < bestDist)
                {
                    bestDist = dist;
                    bestNext = j;
                }
            }

            visited[bestNext] = true;
            order.Add(bestNext);
            current = bestNext;
        }

        return order.ToArray();
    }

    static int[] NearestNeighborWith2Opt()
    {
        var order = NearestNeighbor();

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    if (TwoOptImproves(order, i, j))
                    {
                        TwoOptSwap(order, i, j);
                        improved = true;
                    }
                }
            }
        }

        return order;
    }

    static bool TwoOptImproves(int[] order, int i, int j)
    {
        int a = i == 0 ? -1 : order[i - 1];
        int b = order[i];
        int c = order[j];
        int d = j == N - 1 ? -1 : order[j + 1];

        double oldCost = (a == -1 ? costMatrix[0][b + 1] : costMatrix[a + 1][b + 1])
                       + (d == -1 ? 0 : costMatrix[c + 1][d + 1]);
        double newCost = (a == -1 ? costMatrix[0][c + 1] : costMatrix[a + 1][c + 1])
                       + (d == -1 ? 0 : costMatrix[b + 1][d + 1]);

        return newCost < oldCost - 1e-9;
    }

    static void TwoOptSwap(int[] order, int i, int j)
    {
        while (i < j)
        {
            (order[i], order[j]) = (order[j], order[i]);
            i++; j--;
        }
    }

    static int[] SimulatedAnnealing()
    {
        var current = NearestNeighbor();
        var best = (int[])current.Clone();
        double currentCost = TourCost(current);
        double bestCost = currentCost;

        double temp = 100.0;
        double cooling = 0.9999;
        int iterations = Math.Min(1000000, N * N * 1000);

        for (int iter = 0; iter < iterations; iter++)
        {
            int i = rng.Next(N);
            int j = rng.Next(N);
            if (i == j) continue;
            if (i > j) (i, j) = (j, i);

            // Try 2-opt move
            var candidate = (int[])current.Clone();
            TwoOptSwap(candidate, i, j);
            double candidateCost = TourCost(candidate);

            double delta = candidateCost - currentCost;
            if (delta < 0 || rng.NextDouble() < Math.Exp(-delta / temp))
            {
                current = candidate;
                currentCost = candidateCost;

                if (currentCost < bestCost)
                {
                    best = (int[])current.Clone();
                    bestCost = currentCost;
                }
            }

            temp *= cooling;
        }

        // Final 2-opt polish
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    if (TwoOptImproves(best, i, j))
                    {
                        TwoOptSwap(best, i, j);
                        improved = true;
                    }
                }
            }
        }

        return best;
    }
}
