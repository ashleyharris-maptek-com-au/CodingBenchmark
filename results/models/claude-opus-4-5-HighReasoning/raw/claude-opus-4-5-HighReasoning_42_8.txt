; SPIR-V Geometry Shader - Shrink triangles toward centroid
; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 200
; Schema: 0

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
       %void = OpTypeVoid
   %void_func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

               ; Constants
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
     %uint_3 = OpConstant %uint 3
    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
  %float_0_7 = OpConstant %float 0.7
  %float_inv3 = OpConstant %float 0.333333343

               ; Input gl_PerVertex block (array of 3 for triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_3
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

               ; Input color array (location 2)
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
    %inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3float = OpTypePointer Input %v3float

               ; Output color (location 0)
%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output

               ; Main function
       %main = OpFunction %void None %void_func
      %entry = OpLabel

               ; Load positions from gl_in[0], gl_in[1], gl_in[2]
   %ptr_pos0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
   %ptr_pos1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
   %ptr_pos2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
       %pos0 = OpLoad %v4float %ptr_pos0
       %pos1 = OpLoad %v4float %ptr_pos1
       %pos2 = OpLoad %v4float %ptr_pos2

               ; Load colors from inColor[0], inColor[1], inColor[2]
%ptr_color0 = OpAccessChain %ptr_in_v3float %inColor %int_0
%ptr_color1 = OpAccessChain %ptr_in_v3float %inColor %int_1
%ptr_color2 = OpAccessChain %ptr_in_v3float %inColor %int_2
    %color0 = OpLoad %v3float %ptr_color0
    %color1 = OpLoad %v3float %ptr_color1
    %color2 = OpLoad %v3float %ptr_color2

               ; Compute centroid: C = (p0 + p1 + p2) / 3
    %sum_01 = OpFAdd %v4float %pos0 %pos1
   %sum_012 = OpFAdd %v4float %sum_01 %pos2
  %inv3_vec = OpCompositeConstruct %v4float %float_inv3 %float_inv3 %float_inv3 %float_inv3
  %centroid = OpFMul %v4float %sum_012 %inv3_vec

               ; Shrink factor vector
  %shrink_vec = OpCompositeConstruct %v4float %float_0_7 %float_0_7 %float_0_7 %float_0_7

               ; Vertex 0: new_pos = C + s * (p0 - C)
    %diff0 = OpFSub %v4float %pos0 %centroid
  %scaled0 = OpFMul %v4float %diff0 %shrink_vec
 %newpos0 = OpFAdd %v4float %centroid %scaled0
%ptr_out_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %ptr_out_pos %newpos0
               OpStore %outColor %color0
               OpEmitVertex

               ; Vertex 1: new_pos = C + s * (p1 - C)
    %diff1 = OpFSub %v4float %pos1 %centroid
  %scaled1 = OpFMul %v4float %diff1 %shrink_vec
 %newpos1 = OpFAdd %v4float %centroid %scaled1
               OpStore %ptr_out_pos %newpos1
               OpStore %outColor %color1
               OpEmitVertex

               ; Vertex 2: new_pos = C + s * (p2 - C)
    %diff2 = OpFSub %v4float %pos2 %centroid
  %scaled2 = OpFMul %v4float %diff2 %shrink_vec
 %newpos2 = OpFAdd %v4float %centroid %scaled2
               OpStore %ptr_out_pos %newpos2
               OpStore %outColor %color2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
