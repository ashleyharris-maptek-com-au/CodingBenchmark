```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <set>

using namespace std;

class GraphColorer {
    int n, m, k;
    vector<vector<int>> adj;
    vector<int> color;
    vector<int> degree;
    mt19937 rng;
    chrono::steady_clock::time_point startTime;

    bool timeExceeded() {
        auto now = chrono::steady_clock::now();
        return chrono::duration_cast<chrono::seconds>(now - startTime).count() > 280;
    }

    bool dsatur() {
        fill(color.begin(), color.end(), -1);
        vector<int> saturation(n, 0);

        auto cmp = [&](int a, int b) {
            if (saturation[a] != saturation[b]) return saturation[a] < saturation[b];
            return degree[a] < degree[b];
        };

        set<int> uncolored;
        for (int i = 0; i < n; i++) uncolored.insert(i);

        while (!uncolored.empty()) {
            int v = -1;
            for (int u : uncolored) {
                if (v == -1 || cmp(v, u)) v = u;
            }
            uncolored.erase(v);

            vector<bool> used(k, false);
            for (int u : adj[v]) {
                if (color[u] >= 0 && color[u] < k) used[color[u]] = true;
            }

            int c = -1;
            for (int i = 0; i < k; i++) {
                if (!used[i]) { c = i; break; }
            }

            if (c == -1) return false;
            color[v] = c;

            for (int u : adj[v]) {
                if (color[u] == -1) saturation[u]++;
            }
        }
        return true;
    }

    bool greedyLDF() {
        fill(color.begin(), color.end(), -1);
        vector<int> order(n);
        for (int i = 0; i < n; i++) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            return degree[a] > degree[b];
        });

        for (int v : order) {
            vector<bool> used(k, false);
            for (int u : adj[v]) {
                if (color[u] >= 0) used[color[u]] = true;
            }
            int c = -1;
            for (int i = 0; i < k; i++) {
                if (!used[i]) { c = i; break; }
            }
            if (c == -1) return false;
            color[v] = c;
        }
        return true;
    }

    bool backtrack(int idx, const vector<int>& order) {
        if (timeExceeded()) return false;
        if (idx == n) return true;

        int v = order[idx];
        vector<bool> used(k, false);
        for (int u : adj[v]) {
            if (color[u] >= 0) used[color[u]] = true;
        }

        vector<int> colors;
        for (int c = 0; c < k; c++) {
            if (!used[c]) colors.push_back(c);
        }
        shuffle(colors.begin(), colors.end(), rng);

        for (int c : colors) {
            color[v] = c;
            if (backtrack(idx + 1, order)) return true;
        }
        color[v] = -1;
        return false;
    }

    bool tryBacktracking() {
        fill(color.begin(), color.end(), -1);
        vector<int> order(n);
        for (int i = 0; i < n; i++) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            return degree[a] > degree[b];
        });
        return backtrack(0, order);
    }

    bool localSearch(int maxIter) {
        for (int i = 0; i < n; i++) {
            color[i] = rng() % k;
        }

        for (int iter = 0; iter < maxIter && !timeExceeded(); iter++) {
            vector<int> conflicts;
            for (int v = 0; v < n; v++) {
                for (int u : adj[v]) {
                    if (color[v] == color[u]) {
                        conflicts.push_back(v);
                        break;
                    }
                }
            }

            if (conflicts.empty()) return true;

            int v = conflicts[rng() % conflicts.size()];

            vector<int> conflictCount(k, 0);
            for (int u : adj[v]) {
                conflictCount[color[u]]++;
            }

            int bestColor = 0;
            int bestConflicts = conflictCount[0];
            for (int c = 1; c < k; c++) {
                if (conflictCount[c] < bestConflicts ||
                    (conflictCount[c] == bestConflicts && rng() % 2)) {
                    bestColor = c;
                    bestConflicts = conflictCount[c];
                }
            }
            color[v] = bestColor;
        }
        return false;
    }

public:
    GraphColorer() : rng(chrono::steady_clock::now().time_since_epoch().count()) {}

    void solve() {
        startTime = chrono::steady_clock::now();

        cin >> n >> m >> k;
        adj.resize(n);
        color.resize(n, -1);
        degree.resize(n, 0);

        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            adj[u].push_back(v);
            adj[v].push_back(u);
            degree[u]++;
            degree[v]++;
        }

        bool solved = false;

        if (!solved) solved = dsatur();
        if (!solved) solved = greedyLDF();
        if (!solved && n <= 500) solved = tryBacktracking();
        if (!solved) {
            int iterations = min(10000000, max(100000, n * 100));
            solved = localSearch(iterations);
        }
        while (!solved && !timeExceeded()) {
            solved = localSearch(1000000);
        }

        for (int i = 0; i < n; i++) {
            if (i > 0) cout << " ";
            cout << color[i];
        }
        cout << "\n";
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    GraphColorer colorer;
    colorer.solve();

    return 0;
}
```
