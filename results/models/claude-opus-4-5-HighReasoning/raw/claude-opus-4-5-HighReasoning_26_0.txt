use std::io::{self, BufReader, BufWriter, Read, Write};

const BATCH_SIZE: usize = 100_000;
const MAX_ITERATIONS: usize = 50;
const CONVERGENCE_THRESHOLD: f64 = 1e-6;

#[derive(Clone, Copy)]
struct Point3D {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3D {
    fn distance_sq(&self, other: &Point3D) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

struct Centroid {
    sum_x: f64,
    sum_y: f64,
    sum_z: f64,
    count: u64,
}

impl Centroid {
    fn new() -> Self {
        Centroid { sum_x: 0.0, sum_y: 0.0, sum_z: 0.0, count: 0 }
    }

    fn add(&mut self, p: &Point3D) {
        self.sum_x += p.x as f64;
        self.sum_y += p.y as f64;
        self.sum_z += p.z as f64;
        self.count += 1;
    }

    fn to_point(&self) -> Point3D {
        if self.count == 0 {
            Point3D { x: 0.0, y: 0.0, z: 0.0 }
        } else {
            Point3D {
                x: (self.sum_x / self.count as f64) as f32,
                y: (self.sum_y / self.count as f64) as f32,
                z: (self.sum_z / self.count as f64) as f32,
            }
        }
    }

    fn reset(&mut self) {
        self.sum_x = 0.0;
        self.sum_y = 0.0;
        self.sum_z = 0.0;
        self.count = 0;
    }
}

fn find_nearest_centroid(p: &Point3D, centroids: &[Point3D]) -> u32 {
    let mut best = 0u32;
    let mut best_dist = f32::MAX;
    for (i, c) in centroids.iter().enumerate() {
        let d = p.distance_sq(c);
        if d < best_dist {
            best_dist = d;
            best = i as u32;
        }
    }
    best
}

fn read_points_binary(reader: &mut impl Read, count: usize) -> io::Result<Vec<Point3D>> {
    let mut points = Vec::with_capacity(count);
    let mut buf = [0u8; 12];
    for _ in 0..count {
        reader.read_exact(&mut buf)?;
        let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
        points.push(Point3D { x, y, z });
    }
    Ok(points)
}

fn init_centroids_plusplus(points: &[Point3D], k: usize) -> Vec<Point3D> {
    use std::collections::hash_map::DefaultHasher;
    use std::hash::{Hash, Hasher};

    let mut hasher = DefaultHasher::new();
    points.len().hash(&mut hasher);
    let mut seed = hasher.finish();

    let mut rng = || {
        seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
        seed
    };

    let mut centroids = Vec::with_capacity(k);
    if points.is_empty() || k == 0 {
        return centroids;
    }

    let first_idx = (rng() as usize) % points.len();
    centroids.push(points[first_idx]);

    let mut distances: Vec<f32> = points.iter().map(|p| p.distance_sq(&centroids[0])).collect();

    for _ in 1..k {
        let total: f64 = distances.iter().map(|&d| d as f64).sum();
        if total <= 0.0 {
            let idx = (rng() as usize) % points.len();
            centroids.push(points[idx]);
        } else {
            let threshold = (rng() as f64 / u64::MAX as f64) * total;
            let mut cumulative = 0.0;
            let mut chosen = points.len() - 1;
            for (i, &d) in distances.iter().enumerate() {
                cumulative += d as f64;
                if cumulative >= threshold {
                    chosen = i;
                    break;
                }
            }
            centroids.push(points[chosen]);
        }

        let new_centroid = centroids.last().unwrap();
        for (i, p) in points.iter().enumerate() {
            let d = p.distance_sq(new_centroid);
            if d < distances[i] {
                distances[i] = d;
            }
        }
    }

    centroids
}

fn kmeans_full(points: &[Point3D], k: usize) -> Vec<u32> {
    if points.is_empty() || k == 0 {
        return vec![];
    }

    let mut centroids = init_centroids_plusplus(points, k);
    let mut assignments: Vec<u32> = vec![0; points.len()];
    let mut accumulators: Vec<Centroid> = (0..k).map(|_| Centroid::new()).collect();

    for _ in 0..MAX_ITERATIONS {
        for acc in accumulators.iter_mut() {
            acc.reset();
        }

        for (i, p) in points.iter().enumerate() {
            let cluster = find_nearest_centroid(p, &centroids);
            assignments[i] = cluster;
            accumulators[cluster as usize].add(p);
        }

        let mut max_shift: f64 = 0.0;
        for (i, acc) in accumulators.iter().enumerate() {
            let new_c = acc.to_point();
            let shift = centroids[i].distance_sq(&new_c) as f64;
            if shift > max_shift {
                max_shift = shift;
            }
            centroids[i] = new_c;
        }

        if max_shift < CONVERGENCE_THRESHOLD {
            break;
        }
    }

    for (i, acc) in accumulators.iter().enumerate() {
        if acc.count == 0 && !points.is_empty() {
            centroids[i] = points[0];
        }
    }

    for (i, p) in points.iter().enumerate() {
        assignments[i] = find_nearest_centroid(p, &centroids);
    }

    assignments
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::with_capacity(1024 * 1024, stdin.lock());
    let mut writer = BufWriter::with_capacity(1024 * 1024, stdout.lock());

    let mut header = String::new();
    let mut byte = [0u8; 1];
    loop {
        reader.read_exact(&mut byte)?;
        if byte[0] == b'\n' {
            break;
        }
        header.push(byte[0] as char);
    }

    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    let num_points: u64 = parts[0].parse().unwrap();
    let num_clusters: usize = parts[1].parse().unwrap();

    let memory_threshold = 50_000_000u64;

    if num_points <= memory_threshold {
        let points = read_points_binary(&mut reader, num_points as usize)?;
        let assignments = kmeans_full(&points, num_clusters);
        for a in assignments {
            writer.write_all(&a.to_le_bytes())?;
        }
    } else {
        let sample_size = (num_clusters * 100).min(1_000_000);
        let sample_interval = (num_points as usize / sample_size).max(1);

        let mut samples = Vec::with_capacity(sample_size);
        let mut buf = [0u8; 12];
        let mut all_points: Vec<Point3D> = Vec::with_capacity(num_points as usize);

        let mut point_idx = 0u64;
        while point_idx < num_points {
            reader.read_exact(&mut buf)?;
            let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
            let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
            let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
            let p = Point3D { x, y, z };

            if point_idx as usize % sample_interval == 0 && samples.len() < sample_size {
                samples.push(p);
            }
            all_points.push(p);
            point_idx += 1;
        }

        let mut centroids = init_centroids_plusplus(&samples, num_clusters);
        let mut accumulators: Vec<Centroid> = (0..num_clusters).map(|_| Centroid::new()).collect();

        for _ in 0..MAX_ITERATIONS {
            for acc in accumulators.iter_mut() {
                acc.reset();
            }

            for p in &samples {
                let cluster = find_nearest_centroid(p, &centroids);
                accumulators[cluster as usize].add(p);
            }

            let mut max_shift: f64 = 0.0;
            for (i, acc) in accumulators.iter().enumerate() {
                if acc.count > 0 {
                    let new_c = acc.to_point();
                    let shift = centroids[i].distance_sq(&new_c) as f64;
                    if shift > max_shift {
                        max_shift = shift;
                    }
                    centroids[i] = new_c;
                }
            }

            if max_shift < CONVERGENCE_THRESHOLD {
                break;
            }
        }

        for p in &all_points {
            let cluster = find_nearest_centroid(p, &centroids);
            writer.write_all(&cluster.to_le_bytes())?;
        }
    }

    writer.flush()?;
    Ok(())
}
