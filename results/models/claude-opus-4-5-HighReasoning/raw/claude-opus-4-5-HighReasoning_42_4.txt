; SPIR-V Geometry Shader - Tapered Ribbon from Lines
; Assembles with: spirv-as --target-env vulkan1.0

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Input/output locations
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
   %void_fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
    %uint_2 = OpConstant %uint 2
  %float_0 = OpConstant %float 0.0
  %float_1 = OpConstant %float 1.0
  %thick_start = OpConstant %float 0.04
  %thick_end = OpConstant %float 0.005

; Input gl_PerVertex block (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_2
     %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

; Input color array (location 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
   %inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3float = OpTypePointer Input %v3float

; Output color (location 0)
%ptr_out_v3float = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3float Output

       %main = OpFunction %void None %void_fn
      %entry = OpLabel

; Load input positions gl_in[0].gl_Position and gl_in[1].gl_Position
   %ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
       %p0 = OpLoad %v4float %ptr_p0
   %ptr_p1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
       %p1 = OpLoad %v4float %ptr_p1

; Load input colors
  %ptr_c0 = OpAccessChain %ptr_in_v3float %inColor %int_0
       %c0 = OpLoad %v3float %ptr_c0
  %ptr_c1 = OpAccessChain %ptr_in_v3float %inColor %int_1
       %c1 = OpLoad %v3float %ptr_c1

; Extract components from positions
    %p0_x = OpCompositeExtract %float %p0 0
    %p0_y = OpCompositeExtract %float %p0 1
    %p0_z = OpCompositeExtract %float %p0 2
    %p0_w = OpCompositeExtract %float %p0 3
    %p1_x = OpCompositeExtract %float %p1 0
    %p1_y = OpCompositeExtract %float %p1 1
    %p1_z = OpCompositeExtract %float %p1 2
    %p1_w = OpCompositeExtract %float %p1 3

; Compute direction vector (p1 - p0) in xy
    %dx = OpFSub %float %p1_x %p0_x
    %dy = OpFSub %float %p1_y %p0_y

; Compute perpendicular direction (normalized): perp = normalize(-dy, dx)
  %neg_dy = OpFNegate %float %dy
  %len_sq = OpFMul %float %dx %dx
  %dy_sq = OpFMul %float %dy %dy
  %len_sq2 = OpFAdd %float %len_sq %dy_sq
     %len = OpExtInst %float %1 Sqrt %len_sq2
  %inv_len = OpFDiv %float %float_1 %len
  %perp_x = OpFMul %float %neg_dy %inv_len
  %perp_y = OpFMul %float %dx %inv_len

; Compute offsets for start (thick_start) and end (thick_end)
%off_start_x = OpFMul %float %perp_x %thick_start
%off_start_y = OpFMul %float %perp_y %thick_start
%off_end_x = OpFMul %float %perp_x %thick_end
%off_end_y = OpFMul %float %perp_y %thick_end

; Get output position pointer
%ptr_gl_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0

; Vertex 1: p0 + perp*thick_start
  %v1_x = OpFAdd %float %p0_x %off_start_x
  %v1_y = OpFAdd %float %p0_y %off_start_y
    %v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
          OpStore %ptr_gl_pos %v1
          OpStore %outColor %c0
          OpEmitVertex

; Vertex 2: p0 - perp*thick_start
  %v2_x = OpFSub %float %p0_x %off_start_x
  %v2_y = OpFSub %float %p0_y %off_start_y
    %v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p0_z %p0_w
          OpStore %ptr_gl_pos %v2
          OpStore %outColor %c0
          OpEmitVertex

; Vertex 3: p1 + perp*thick_end
  %v3_x = OpFAdd %float %p1_x %off_end_x
  %v3_y = OpFAdd %float %p1_y %off_end_y
    %v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
          OpStore %ptr_gl_pos %v3
          OpStore %outColor %c1
          OpEmitVertex

; Vertex 4: p1 - perp*thick_end
  %v4_x = OpFSub %float %p1_x %off_end_x
  %v4_y = OpFSub %float %p1_y %off_end_y
    %v4 = OpCompositeConstruct %v4float %v4_x %v4_y %p1_z %p1_w
          OpStore %ptr_gl_pos %v4
          OpStore %outColor %c1
          OpEmitVertex

          OpEndPrimitive
          OpReturn
          OpFunctionEnd
