; SPIR-V Geometry Shader - Point to Hexagon
; Expands each input point into a hexagon (6 triangles)

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void = OpTypeVoid
  %voidfunc = OpTypeFunction %void
     %float = OpTypeFloat 32
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0

               ; Constants
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
   %uint_1 = OpConstant %uint 1
  %float_0 = OpConstant %float 0.0
  %float_1 = OpConstant %float 1.0
%float_0_04 = OpConstant %float 0.04
%float_0_7 = OpConstant %float 0.7
%float_0_5 = OpConstant %float 0.5
%float_0_866 = OpConstant %float 0.866025
%float_n0_5 = OpConstant %float -0.5
%float_n1 = OpConstant %float -1.0
%float_n0_866 = OpConstant %float -0.866025

               ; Input gl_PerVertex block (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr = OpTypePointer Input %arr_pv_in_1
     %gl_in = OpVariable %ptr_in_arr Input
%ptr_in_v4 = OpTypePointer Input %v4float

               ; Input color array
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
   %inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

               ; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3 Output

               ; Main function
      %main = OpFunction %void None %voidfunc
     %entry = OpLabel

               ; Load input position: gl_in[0].gl_Position
   %pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
   %pos_in = OpLoad %v4float %pos_ptr

               ; Load input color: inColor[0]
 %color_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
 %color_in = OpLoad %v3float %color_ptr

               ; Darker color for edges (multiply by 0.7)
%dark_color = OpVectorTimesScalar %v3float %color_in %float_0_7

               ; Extract position components
   %pos_x = OpCompositeExtract %float %pos_in 0
   %pos_y = OpCompositeExtract %float %pos_in 1
   %pos_z = OpCompositeExtract %float %pos_in 2
   %pos_w = OpCompositeExtract %float %pos_in 3

               ; Precompute edge positions (6 vertices at 60 degree intervals)
               ; Edge 0: angle 0 (cos=1, sin=0)
   %e0_x = OpFAdd %float %pos_x %float_0_04
   %e0_y = OpFAdd %float %pos_y %float_0
   %e0_pos = OpCompositeConstruct %v4float %e0_x %e0_y %pos_z %pos_w

               ; Edge 1: angle 60 (cos=0.5, sin=0.866)
 %e1_dx = OpFMul %float %float_0_04 %float_0_5
 %e1_dy = OpFMul %float %float_0_04 %float_0_866
   %e1_x = OpFAdd %float %pos_x %e1_dx
   %e1_y = OpFAdd %float %pos_y %e1_dy
   %e1_pos = OpCompositeConstruct %v4float %e1_x %e1_y %pos_z %pos_w

               ; Edge 2: angle 120 (cos=-0.5, sin=0.866)
 %e2_dx = OpFMul %float %float_0_04 %float_n0_5
 %e2_dy = OpFMul %float %float_0_04 %float_0_866
   %e2_x = OpFAdd %float %pos_x %e2_dx
   %e2_y = OpFAdd %float %pos_y %e2_dy
   %e2_pos = OpCompositeConstruct %v4float %e2_x %e2_y %pos_z %pos_w

               ; Edge 3: angle 180 (cos=-1, sin=0)
 %e3_dx = OpFMul %float %float_0_04 %float_n1
   %e3_x = OpFAdd %float %pos_x %e3_dx
   %e3_y = OpFAdd %float %pos_y %float_0
   %e3_pos = OpCompositeConstruct %v4float %e3_x %e3_y %pos_z %pos_w

               ; Edge 4: angle 240 (cos=-0.5, sin=-0.866)
 %e4_dx = OpFMul %float %float_0_04 %float_n0_5
 %e4_dy = OpFMul %float %float_0_04 %float_n0_866
   %e4_x = OpFAdd %float %pos_x %e4_dx
   %e4_y = OpFAdd %float %pos_y %e4_dy
   %e4_pos = OpCompositeConstruct %v4float %e4_x %e4_y %pos_z %pos_w

               ; Edge 5: angle 300 (cos=0.5, sin=-0.866)
 %e5_dx = OpFMul %float %float_0_04 %float_0_5
 %e5_dy = OpFMul %float %float_0_04 %float_n0_866
   %e5_x = OpFAdd %float %pos_x %e5_dx
   %e5_y = OpFAdd %float %pos_y %e5_dy
   %e5_pos = OpCompositeConstruct %v4float %e5_x %e5_y %pos_z %pos_w

               ; Output pointer for gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; Triangle 0: center, e0, e1
               OpStore %out_pos_ptr %pos_in
               OpStore %outColor %color_in
               OpEmitVertex
               OpStore %out_pos_ptr %e0_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpStore %out_pos_ptr %e1_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpEndPrimitive

               ; Triangle 1: center, e1, e2
               OpStore %out_pos_ptr %pos_in
               OpStore %outColor %color_in
               OpEmitVertex
               OpStore %out_pos_ptr %e1_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpStore %out_pos_ptr %e2_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpEndPrimitive

               ; Triangle 2: center, e2, e3
               OpStore %out_pos_ptr %pos_in
               OpStore %outColor %color_in
               OpEmitVertex
               OpStore %out_pos_ptr %e2_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpStore %out_pos_ptr %e3_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpEndPrimitive

               ; Triangle 3: center, e3, e4
               OpStore %out_pos_ptr %pos_in
               OpStore %outColor %color_in
               OpEmitVertex
               OpStore %out_pos_ptr %e3_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpStore %out_pos_ptr %e4_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpEndPrimitive

               ; Triangle 4: center, e4, e5
               OpStore %out_pos_ptr %pos_in
               OpStore %outColor %color_in
               OpEmitVertex
               OpStore %out_pos_ptr %e4_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpStore %out_pos_ptr %e5_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpEndPrimitive

               ; Triangle 5: center, e5, e0 (closing the hexagon)
               OpStore %out_pos_ptr %pos_in
               OpStore %outColor %color_in
               OpEmitVertex
               OpStore %out_pos_ptr %e5_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpStore %out_pos_ptr %e0_pos
               OpStore %outColor %dark_color
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
