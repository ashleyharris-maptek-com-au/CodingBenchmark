using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class MiningOptimizer
{
    static int XCount, YCount, ZCount;
    static double XBlockSize, YBlockSize, ZBlockSize;
    static byte[,,] voxels;
    static int[,] groundHeight;
    static List<string> operations = new List<string>();

    static void Main()
    {
        using var stdin = Console.OpenStandardInput();
        using var reader = new StreamReader(stdin);

        string header = reader.ReadLine();
        var parts = header.Split(' ');
        XCount = int.Parse(parts[0]);
        YCount = int.Parse(parts[1]);
        ZCount = int.Parse(parts[2]);
        XBlockSize = double.Parse(parts[3]);
        YBlockSize = double.Parse(parts[4]);
        ZBlockSize = double.Parse(parts[5]);

        voxels = new byte[XCount, YCount, ZCount];
        groundHeight = new int[XCount, YCount];

        using var binReader = new BinaryReader(stdin);
        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                groundHeight[x, y] = -1;
                for (int z = 0; z < ZCount; z++)
                {
                    byte d = binReader.ReadByte();
                    voxels[x, y, z] = d;
                    if (d > 0 && groundHeight[x, y] < z)
                        groundHeight[x, y] = z;
                }
            }
        }

        if (XCount <= 50 && YCount <= 50 && ZCount <= 500)
            SolveSmall();
        else if (XCount <= 500 && YCount <= 500)
            SolveMedium();
        else
            SolveLarge();

        Console.WriteLine(operations.Count);
        foreach (var op in operations)
            Console.WriteLine(op);
    }

    static int GetMass(byte d) => d & 0x0F;
    static int GetValue(byte d) => d >> 4;

    static bool IsExposed(int x, int y, int z)
    {
        if (voxels[x, y, z] == 0) return false;
        if (z == ZCount - 1 || voxels[x, y, z + 1] == 0) return true;
        if (x > 0 && voxels[x - 1, y, z] == 0) return true;
        if (x < XCount - 1 && voxels[x + 1, y, z] == 0) return true;
        if (y > 0 && voxels[x, y - 1, z] == 0) return true;
        if (y < YCount - 1 && voxels[x, y + 1, z] == 0) return true;
        return false;
    }

    static double CalculatePathCost(int sx, int sy, int sz, int dx, int dy, int mass)
    {
        int cx = sx, cy = sy;
        int cz = sz;
        double cost = 0;

        while (cx != dx || cy != dy)
        {
            int nx = cx, ny = cy;
            if (cx < dx) nx++;
            else if (cx > dx) nx--;
            else if (cy < dy) ny++;
            else if (cy > dy) ny--;

            int nz = (nx == dx && ny == dy) ? GetGroundAt(dx, dy) + 1 : GetGroundAt(nx, ny);

            if (nz > cz)
            {
                int climb = nz - cz;
                if (climb > 1) return double.MaxValue;
                cost += 5.0 * mass;
            }
            else
            {
                cost += 1.0 * mass;
            }

            cx = nx;
            cy = ny;
            cz = nz;
        }
        return cost;
    }

    static int GetGroundAt(int x, int y)
    {
        if (x < 0 || x >= XCount || y < 0 || y >= YCount) return 0;
        return groundHeight[x, y];
    }

    static List<(int, int)> FindPath(int sx, int sy, int sz, int dx, int dy)
    {
        var path = new List<(int, int)>();
        int cx = sx, cy = sy;

        while (cx != dx)
        {
            cx += (dx > cx) ? 1 : -1;
            path.Add((cx, cy));
        }
        while (cy != dy)
        {
            cy += (dy > cy) ? 1 : -1;
            path.Add((cx, cy));
        }
        return path;
    }

    static void DigAndMove(int x, int y, int z, int destX, int destY)
    {
        byte d = voxels[x, y, z];
        if (d == 0) return;

        var path = FindPath(x, y, z, destX, destY);
        string pathStr = string.Join(" ", path.Select(p => $"{p.Item1} {p.Item2}"));
        operations.Add($"{path.Count} {x} {y} {z} {pathStr} {destX} {destY}");

        voxels[x, y, z] = 0;

        if (z == groundHeight[x, y])
        {
            while (groundHeight[x, y] >= 0 && voxels[x, y, groundHeight[x, y]] == 0)
                groundHeight[x, y]--;
        }

        if (!(destX == 0 && destY == 0 && GetValue(d) > 0))
        {
            groundHeight[destX, destY]++;
            int newZ = groundHeight[destX, destY];
            if (newZ < ZCount)
                voxels[destX, destY, newZ] = d;
        }
    }

    static (int, int) FindDumpSite(int fromX, int fromY, int fromZ)
    {
        int bestX = XCount - 1, bestY = YCount - 1;
        double bestCost = double.MaxValue;

        for (int x = Math.Max(0, fromX - 20); x < Math.Min(XCount, fromX + 20); x++)
        {
            for (int y = Math.Max(0, fromY - 20); y < Math.Min(YCount, fromY + 20); y++)
            {
                if (x == 0 && y == 0) continue;
                int gh = GetGroundAt(x, y);
                if (gh >= ZCount - 1) continue;

                double cost = Math.Abs(x - fromX) + Math.Abs(y - fromY);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestX = x;
                    bestY = y;
                }
            }
        }
        return (bestX, bestY);
    }

    static void SolveSmall()
    {
        var valuable = new List<(int x, int y, int z, int mass, int value, double profit)>();

        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                for (int z = 0; z <= groundHeight[x, y]; z++)
                {
                    byte d = voxels[x, y, z];
                    int mass = GetMass(d);
                    int value = GetValue(d);
                    if (value > 0 && mass > 0)
                    {
                        double revenue = 1000.0 * mass * value;
                        double estCost = (x + y) * mass;
                        valuable.Add((x, y, z, mass, value, revenue - estCost));
                    }
                }
            }
        }

        valuable = valuable.OrderByDescending(v => v.profit).ThenByDescending(v => v.z).ToList();

        foreach (var v in valuable)
        {
            if (voxels[v.x, v.y, v.z] == 0) continue;

            while (groundHeight[v.x, v.y] > v.z)
            {
                int topZ = groundHeight[v.x, v.y];
                if (!IsExposed(v.x, v.y, topZ)) break;

                var dump = FindDumpSite(v.x, v.y, topZ);
                DigAndMove(v.x, v.y, topZ, dump.Item1, dump.Item2);
            }

            if (voxels[v.x, v.y, v.z] != 0 && IsExposed(v.x, v.y, v.z))
            {
                byte d = voxels[v.x, v.y, v.z];
                int mass = GetMass(d);
                int value = GetValue(d);
                double revenue = 1000.0 * mass * value;
                double cost = CalculatePathCost(v.x, v.y, v.z, 0, 0, mass);

                if (revenue > cost && cost < double.MaxValue)
                {
                    DigAndMove(v.x, v.y, v.z, 0, 0);
                }
            }
        }
    }

    static void SolveMedium()
    {
        var candidates = new List<(int x, int y, int z, double profit)>();

        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                int z = groundHeight[x, y];
                if (z < 0) continue;

                byte d = voxels[x, y, z];
                int mass = GetMass(d);
                int value = GetValue(d);

                if (value > 0)
                {
                    double revenue = 1000.0 * mass * value;
                    double cost = (x + y) * mass;
                    if (revenue > cost * 2)
                        candidates.Add((x, y, z, revenue - cost));
                }
            }
        }

        candidates = candidates.OrderByDescending(c => c.profit).Take(10000).ToList();

        foreach (var c in candidates)
        {
            if (groundHeight[c.x, c.y] != c.z) continue;
            if (voxels[c.x, c.y, c.z] == 0) continue;
            if (!IsExposed(c.x, c.y, c.z)) continue;

            byte d = voxels[c.x, c.y, c.z];
            int mass = GetMass(d);
            int value = GetValue(d);
            double revenue = 1000.0 * mass * value;
            double cost = CalculatePathCost(c.x, c.y, c.z, 0, 0, mass);

            if (revenue > cost && cost < double.MaxValue)
            {
                DigAndMove(c.x, c.y, c.z, 0, 0);
            }
        }
    }

    static void SolveLarge()
    {
        int sampleStep = Math.Max(1, Math.Min(XCount, YCount) / 100);
        var candidates = new List<(int x, int y, int z, double profit)>();

        for (int x = 0; x < XCount; x += sampleStep)
        {
            for (int y = 0; y < YCount; y += sampleStep)
            {
                int z = groundHeight[x, y];
                if (z < 0) continue;

                byte d = voxels[x, y, z];
                int mass = GetMass(d);
                int value = GetValue(d);

                if (value >= 10)
                {
                    double revenue = 1000.0 * mass * value;
                    double cost = (x + y) * mass;
                    if (revenue > cost * 3)
                        candidates.Add((x, y, z, revenue - cost));
                }
            }
        }

        candidates = candidates.OrderByDescending(c => c.profit).Take(1000).ToList();

        foreach (var c in candidates)
        {
            if (groundHeight[c.x, c.y] != c.z) continue;
            if (voxels[c.x, c.y, c.z] == 0) continue;

            byte d = voxels[c.x, c.y, c.z];
            int mass = GetMass(d);
            int value = GetValue(d);
            double revenue = 1000.0 * mass * value;
            double cost = CalculatePathCost(c.x, c.y, c.z, 0, 0, mass);

            if (revenue > cost && cost < double.MaxValue)
            {
                DigAndMove(c.x, c.y, c.z, 0, 0);
            }
        }
    }
}
