using System;
using System.Collections.Generic;

class Program
{
    static int n, m;
    static List<(int to, int w, int idx)>[] adj;
    static (int u, int v, int w)[] edges;

    static void Main()
    {
        string[] nm = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(nm[0]);
        m = int.Parse(nm[1]);

        adj = new List<(int, int, int)>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<(int, int, int)>();
        edges = new (int, int, int)[m];

        for (int i = 0; i < m; i++)
        {
            string[] parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            edges[i] = (u, v, w);
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
        }

        int[] deg = new int[n];
        foreach (var (u, v, _) in edges)
        {
            deg[u]++;
            deg[v]++;
        }

        List<int> odd = new List<int>();
        for (int i = 0; i < n; i++)
            if (deg[i] % 2 == 1)
                odd.Add(i);

        int[] mult = new int[m];
        Array.Fill(mult, 1);

        if (odd.Count > 0)
        {
            if (odd.Count <= 18 && m <= 50000)
                OptimalMatch(odd, mult);
            else if (odd.Count <= 200 || m <= 100000)
                GreedyMatchDijkstra(odd, mult);
            else
                GreedyMatchBFS(odd, mult);
        }

        var circuit = Hierholzer(mult);
        Console.WriteLine(string.Join(" ", circuit));
    }

    static void OptimalMatch(List<int> odd, int[] mult)
    {
        int k = odd.Count;
        long[][] dist = new long[k][];
        int[][] par = new int[k][];
        int[][] parE = new int[k][];

        for (int i = 0; i < k; i++)
            (dist[i], par[i], parE[i]) = Dijkstra(odd[i]);

        long[] dp = new long[1 << k];
        int[] ch = new int[1 << k];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int f = -1;
            for (int i = 0; i < k; i++)
                if ((mask & (1 << i)) == 0) { f = i; break; }
            if (f == -1) continue;

            for (int s = f + 1; s < k; s++)
            {
                if ((mask & (1 << s)) != 0) continue;
                int nm = mask | (1 << f) | (1 << s);
                long c = dist[f][odd[s]];
                if (dp[mask] + c < dp[nm])
                {
                    dp[nm] = dp[mask] + c;
                    ch[nm] = (f << 8) | s;
                }
            }
        }

        int fm = (1 << k) - 1;
        while (fm != 0)
        {
            int c = ch[fm];
            int f = c >> 8, s = c & 0xFF;
            int cur = odd[s];
            while (cur != odd[f])
            {
                mult[parE[f][cur]]++;
                cur = par[f][cur];
            }
            fm ^= (1 << f) | (1 << s);
        }
    }

    static void GreedyMatchDijkstra(List<int> odd, int[] mult)
    {
        bool[] used = new bool[odd.Count];

        for (int i = 0; i < odd.Count; i++)
        {
            if (used[i]) continue;
            var (dist, par, parE) = Dijkstra(odd[i]);

            long best = long.MaxValue;
            int bj = -1;
            for (int j = i + 1; j < odd.Count; j++)
            {
                if (used[j]) continue;
                if (dist[odd[j]] < best)
                {
                    best = dist[odd[j]];
                    bj = j;
                }
            }

            if (bj != -1)
            {
                used[i] = used[bj] = true;
                int cur = odd[bj];
                while (cur != odd[i])
                {
                    mult[parE[cur]]++;
                    cur = par[cur];
                }
            }
        }
    }

    static void GreedyMatchBFS(List<int> odd, int[] mult)
    {
        var bfsDist = new int[n];
        Array.Fill(bfsDist, -1);
        var q = new Queue<int>();
        q.Enqueue(0);
        bfsDist[0] = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            foreach (var (v, _, _) in adj[u])
            {
                if (bfsDist[v] == -1)
                {
                    bfsDist[v] = bfsDist[u] + 1;
                    q.Enqueue(v);
                }
            }
        }
        odd.Sort((a, b) => bfsDist[a].CompareTo(bfsDist[b]));

        for (int i = 0; i < odd.Count; i += 2)
        {
            int a = odd[i], b = odd[i + 1];
            int[] par = new int[n];
            int[] parE = new int[n];
            Array.Fill(par, -1);
            par[a] = a;

            var bfsQ = new Queue<int>();
            bfsQ.Enqueue(a);

            while (bfsQ.Count > 0)
            {
                int u = bfsQ.Dequeue();
                if (u == b) break;
                foreach (var (v, _, idx) in adj[u])
                {
                    if (par[v] == -1)
                    {
                        par[v] = u;
                        parE[v] = idx;
                        bfsQ.Enqueue(v);
                    }
                }
            }

            int cur = b;
            while (cur != a)
            {
                mult[parE[cur]]++;
                cur = par[cur];
            }
        }
    }

    static (long[] dist, int[] par, int[] parE) Dijkstra(int src)
    {
        long[] dist = new long[n];
        int[] par = new int[n];
        int[] parE = new int[n];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(par, -1);
        dist[src] = 0;

        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            foreach (var (v, w, idx) in adj[u])
            {
                if (dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    par[v] = u;
                    parE[v] = idx;
                    pq.Enqueue(v, dist[v]);
                }
            }
        }

        return (dist, par, parE);
    }

    static List<int> Hierholzer(int[] mult)
    {
        int tot = 0;
        for (int i = 0; i < m; i++) tot += mult[i];

        var al = new List<(int to, int eid)>[n];
        for (int i = 0; i < n; i++) al[i] = new List<(int, int)>();

        int eid = 0;
        for (int i = 0; i < m; i++)
        {
            var (u, v, _) = edges[i];
            for (int j = 0; j < mult[i]; j++)
            {
                al[u].Add((v, eid));
                al[v].Add((u, eid));
                eid++;
            }
        }

        bool[] used = new bool[tot];
        int[] ptr = new int[n];
        var st = new Stack<int>();
        var res = new List<int>();

        st.Push(0);
        while (st.Count > 0)
        {
            int u = st.Peek();
            bool found = false;
            while (ptr[u] < al[u].Count)
            {
                var (v, e) = al[u][ptr[u]++];
                if (!used[e])
                {
                    used[e] = true;
                    st.Push(v);
                    found = true;
                    break;
                }
            }
            if (!found) res.Add(st.Pop());
        }

        res.Reverse();
        return res;
    }
}
