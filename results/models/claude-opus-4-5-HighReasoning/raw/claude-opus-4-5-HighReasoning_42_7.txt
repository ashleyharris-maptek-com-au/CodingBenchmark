; SPIR-V Geometry Shader - Explode triangles along face normal
OpCapability Geometry
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_3 = OpConstant %float 3.0
%float_0_08 = OpConstant %float 0.08

; Input gl_PerVertex block (array of 3 for triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_3
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Input color array (location 2)
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load positions from gl_in[0..2].gl_Position
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

; Extract xyz from positions
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute edge vectors: e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; Compute face normal: N = normalize(cross(e1, e2))
%cross_n = OpExtInst %v3float %ext Cross %e1 %e2
%normal = OpExtInst %v3float %ext Normalize %cross_n

; Compute offset = N * 0.08
%offset_vec = OpVectorTimesScalar %v3float %normal %float_0_08
%offset = OpCompositeConstruct %v4float %offset_vec %float_0

; Load colors
%ptr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %ptr_c2

; Emit vertex 0
%new_p0 = OpFAdd %v4float %p0 %offset
%out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %new_p0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1
%new_p1 = OpFAdd %v4float %p1 %offset
OpStore %out_pos_ptr0 %new_p1
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 2
%new_p2 = OpFAdd %v4float %p2 %offset
OpStore %out_pos_ptr0 %new_p2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
