use std::collections::HashMap;
use std::io::{self, BufRead, Write};

#[derive(Clone)]
struct Sample {
    hole_id: String,
    sample_idx: usize,
    x: f64,
    y: f64,
    z: f64,
    values: Vec<f64>,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    let prop_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = prop_line.split_whitespace().map(|s| s.to_string()).collect();

    let mut all_samples: Vec<Sample> = Vec::new();

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = hole_line.split_whitespace().collect();
        let hole_id = parts[0].to_string();
        let start_x: f64 = parts[1].parse().unwrap_or(0.0);
        let start_y: f64 = parts[2].parse().unwrap_or(0.0);
        let start_z: f64 = parts[3].parse().unwrap_or(0.0);
        let dir_x: f64 = parts[4].parse().unwrap_or(0.0);
        let dir_y: f64 = parts[5].parse().unwrap_or(0.0);
        let dir_z: f64 = parts[6].parse().unwrap_or(-1.0);
        let num_samples: usize = parts[8].parse().unwrap_or(0);

        let dir_len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (dx, dy, dz) = if dir_len > 1e-9 {
            (dir_x / dir_len, dir_y / dir_len, dir_z / dir_len)
        } else {
            (0.0, 0.0, -1.0)
        };

        for sample_idx in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let parts: Vec<&str> = sample_line.split_whitespace().collect();
            let depth: f64 = parts[0].parse().unwrap_or(0.0);

            let mut values = Vec::with_capacity(num_properties);
            for i in 0..num_properties {
                let val = if i + 1 < parts.len() {
                    parts[i + 1].parse().unwrap_or(f64::NAN)
                } else {
                    f64::NAN
                };
                values.push(val);
            }

            let x = start_x + depth * dx;
            let y = start_y + depth * dy;
            let z = start_z + depth * dz;

            all_samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx,
                x,
                y,
                z,
                values,
            });
        }
    }

    if all_samples.is_empty() || num_properties == 0 {
        println!("0");
        return;
    }

    let total_samples = all_samples.len();
    let k_neighbors = if total_samples < 50 { total_samples.saturating_sub(1).max(1) }
                      else if total_samples < 500 { 20 }
                      else if total_samples < 5000 { 15 }
                      else { 10 };

    let cell_size = estimate_cell_size(&all_samples, k_neighbors);
    let grid = build_spatial_grid(&all_samples, cell_size);

    let mut suspects: Vec<Suspect> = Vec::new();

    for (idx, sample) in all_samples.iter().enumerate() {
        let neighbors = find_neighbors(&all_samples, &grid, cell_size, idx, k_neighbors);

        if neighbors.len() < 3 {
            continue;
        }

        for prop_idx in 0..num_properties {
            let val = sample.values[prop_idx];
            if val.is_nan() {
                continue;
            }

            let neighbor_vals: Vec<f64> = neighbors.iter()
                .filter_map(|&n| {
                    let v = all_samples[n].values[prop_idx];
                    if v.is_nan() { None } else { Some(v) }
                })
                .collect();

            if neighbor_vals.len() < 3 {
                continue;
            }

            let confidence = compute_outlier_score(val, &neighbor_vals);

            if confidence > 2.5 {
                suspects.push(Suspect {
                    hole_id: sample.hole_id.clone(),
                    sample_idx: sample.sample_idx,
                    property_name: property_names[prop_idx].clone(),
                    confidence,
                });
            }
        }
    }

    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    let max_output = if total_samples < 100 { 50 }
                     else if total_samples < 1000 { 100 }
                     else if total_samples < 10000 { 500 }
                     else { 1000 };
    let suspects: Vec<_> = suspects.into_iter().take(max_output).collect();

    let stdout = io::stdout();
    let mut out = stdout.lock();
    writeln!(out, "{}", suspects.len()).unwrap();
    for s in suspects {
        writeln!(out, "{} {} {} {:.4}", s.hole_id, s.sample_idx, s.property_name, s.confidence).unwrap();
    }
}

fn estimate_cell_size(samples: &[Sample], k: usize) -> f64 {
    if samples.len() < 10 {
        return 100.0;
    }

    let n = samples.len().min(1000);
    let step = samples.len() / n;

    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    let mut min_z = f64::MAX;
    let mut max_z = f64::MIN;

    for i in (0..samples.len()).step_by(step.max(1)) {
        let s = &samples[i];
        min_x = min_x.min(s.x);
        max_x = max_x.max(s.x);
        min_y = min_y.min(s.y);
        max_y = max_y.max(s.y);
        min_z = min_z.min(s.z);
        max_z = max_z.max(s.z);
    }

    let volume = (max_x - min_x + 1.0) * (max_y - min_y + 1.0) * (max_z - min_z + 1.0);
    let density = samples.len() as f64 / volume;
    let avg_spacing = (1.0 / density).powf(1.0 / 3.0);

    (avg_spacing * (k as f64).powf(1.0 / 3.0) * 2.0).max(1.0)
}

fn build_spatial_grid(samples: &[Sample], cell_size: f64) -> HashMap<(i64, i64, i64), Vec<usize>> {
    let mut grid: HashMap<(i64, i64, i64), Vec<usize>> = HashMap::new();

    for (idx, s) in samples.iter().enumerate() {
        let cx = (s.x / cell_size).floor() as i64;
        let cy = (s.y / cell_size).floor() as i64;
        let cz = (s.z / cell_size).floor() as i64;
        grid.entry((cx, cy, cz)).or_default().push(idx);
    }

    grid
}

fn find_neighbors(
    samples: &[Sample],
    grid: &HashMap<(i64, i64, i64), Vec<usize>>,
    cell_size: f64,
    target_idx: usize,
    k: usize,
) -> Vec<usize> {
    let target = &samples[target_idx];
    let cx = (target.x / cell_size).floor() as i64;
    let cy = (target.y / cell_size).floor() as i64;
    let cz = (target.z / cell_size).floor() as i64;

    let mut candidates: Vec<(usize, f64)> = Vec::new();

    for dx in -2..=2 {
        for dy in -2..=2 {
            for dz in -2..=2 {
                if let Some(indices) = grid.get(&(cx + dx, cy + dy, cz + dz)) {
                    for &idx in indices {
                        if idx != target_idx {
                            let s = &samples[idx];
                            let dist_sq = (s.x - target.x).powi(2)
                                        + (s.y - target.y).powi(2)
                                        + (s.z - target.z).powi(2);
                            candidates.push((idx, dist_sq));
                        }
                    }
                }
            }
        }
    }

    candidates.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
    candidates.into_iter().take(k).map(|(idx, _)| idx).collect()
}

fn compute_outlier_score(value: f64, neighbors: &[f64]) -> f64 {
    let n = neighbors.len() as f64;
    let mean: f64 = neighbors.iter().sum::<f64>() / n;
    let variance: f64 = neighbors.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / n;
    let std_dev = variance.sqrt();

    if std_dev < 1e-10 {
        if (value - mean).abs() < 1e-10 {
            return 0.0;
        }
        return 10.0;
    }

    let z_score = (value - mean).abs() / std_dev;

    let mut sorted = neighbors.to_vec();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let median = if sorted.len() % 2 == 0 {
        (sorted[sorted.len() / 2 - 1] + sorted[sorted.len() / 2]) / 2.0
    } else {
        sorted[sorted.len() / 2]
    };

    let mut abs_devs: Vec<f64> = sorted.iter().map(|v| (v - median).abs()).collect();
    abs_devs.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let mad = if abs_devs.len() % 2 == 0 {
        (abs_devs[abs_devs.len() / 2 - 1] + abs_devs[abs_devs.len() / 2]) / 2.0
    } else {
        abs_devs[abs_devs.len() / 2]
    };

    let mad_score = if mad < 1e-10 {
        if (value - median).abs() < 1e-10 { 0.0 } else { 10.0 }
    } else {
        (value - median).abs() / (mad * 1.4826)
    };

    z_score.max(mad_score)
}
