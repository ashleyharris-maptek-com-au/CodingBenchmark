#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

class Dinic {
    struct Edge { int to, rev, cap, flow; };
    vector<vector<Edge>> g;
    vector<int> level, iter;

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (const auto& e : g[v]) {
                if (level[e.to] < 0 && e.cap > e.flow) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t) return pushed;
        for (int& i = iter[v]; i < (int)g[v].size(); i++) {
            Edge& e = g[v][i];
            if (level[v] < level[e.to] && e.cap > e.flow) {
                int d = dfs(e.to, t, min(pushed, e.cap - e.flow));
                if (d > 0) {
                    e.flow += d;
                    g[e.to][e.rev].flow -= d;
                    return d;
                }
            }
        }
        return 0;
    }

public:
    int n;
    Dinic(int n_) : n(n_), g(n_), level(n_), iter(n_) {}

    void addEdge(int from, int to) {
        g[from].push_back({to, (int)g[to].size(), 1, 0});
        g[to].push_back({from, (int)g[from].size() - 1, 1, 0});
    }

    int maxflow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, n + 1)) > 0) flow += f;
        }
        return flow;
    }

    void reset() {
        for (auto& adj : g) for (auto& e : adj) e.flow = 0;
    }

    vector<bool> reachable(int s) {
        vector<bool> vis(n, false);
        queue<int> q;
        q.push(s);
        vis[s] = true;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (const auto& e : g[v]) {
                if (!vis[e.to] && e.cap > e.flow) {
                    vis[e.to] = true;
                    q.push(e.to);
                }
            }
        }
        return vis;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    if (n <= 1) { cout << 0 << '\n'; return 0; }

    vector<pair<int,int>> edges(m);
    vector<int> deg(n, 0);
    Dinic dinic(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        edges[i] = {u, v};
        dinic.addEdge(u, v);
        deg[u]++;
        deg[v]++;
    }

    if (m == 0) { cout << 0 << '\n'; return 0; }

    int minDeg = n + 1;
    for (int i = 0; i < n; i++) {
        if (deg[i] > 0 && deg[i] < minDeg) {
            minDeg = deg[i];
        }
    }

    if (minDeg == 1) {
        for (const auto& e : edges) {
            if (deg[e.first] == 1 || deg[e.second] == 1) {
                cout << 1 << '\n' << e.first << ' ' << e.second << '\n';
                return 0;
            }
        }
    }

    int minCut = n + 1, bestT = 1;

    vector<int> order;
    for (int i = 1; i < n; i++) order.push_back(i);
    sort(order.begin(), order.end(), [&](int a, int b) { return deg[a] < deg[b]; });

    for (int t : order) {
        if (minCut <= 1) break;
        dinic.reset();
        int cut = dinic.maxflow(0, t);
        if (cut < minCut) {
            minCut = cut;
            bestT = t;
        }
        if (minCut <= min(deg[0], deg[t])) break;
    }

    dinic.reset();
    dinic.maxflow(0, bestT);
    vector<bool> reach = dinic.reachable(0);

    vector<pair<int,int>> cutEdges;
    for (const auto& e : edges) {
        if (reach[e.first] != reach[e.second]) {
            cutEdges.push_back(e);
        }
    }

    cout << cutEdges.size() << '\n';
    for (const auto& e : cutEdges) {
        cout << e.first << ' ' << e.second << '\n';
    }

    return 0;
}
