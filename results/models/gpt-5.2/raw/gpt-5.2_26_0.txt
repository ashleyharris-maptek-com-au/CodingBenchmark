use std::io::{self, Read, Write};

// ======= Fast binary I/O =======

struct FastInput {
    buf: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, pos: 0 })
    }

    #[inline]
    fn skip_ws(&mut self) {
        while self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            if b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' {
                self.pos += 1;
            } else {
                break;
            }
        }
    }

    fn read_u64_text(&mut self) -> Option<u64> {
        self.skip_ws();
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }
        let mut v: u64 = 0;
        let mut any = false;
        while self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            if (b'0'..=b'9').contains(&b) {
                any = true;
                v = v.saturating_mul(10).saturating_add((b - b'0') as u64);
                self.pos += 1;
            } else {
                break;
            }
        }
        if !any {
            return None;
        }
        if sign < 0 {
            // For this task counts are non-negative; treat negative as 0.
            Some(0)
        } else {
            Some(v)
        }
    }

    #[inline]
    fn remaining(&self) -> usize {
        self.buf.len().saturating_sub(self.pos)
    }

    #[inline]
    fn read_f32_le(&mut self) -> Option<f32> {
        if self.pos + 4 > self.buf.len() {
            return None;
        }
        let b = &self.buf[self.pos..self.pos + 4];
        self.pos += 4;
        Some(f32::from_le_bytes([b[0], b[1], b[2], b[3]]))
    }

    #[inline]
    fn read_point3_f32_le(&mut self) -> Option<[f32; 3]> {
        let x = self.read_f32_le()?;
        let y = self.read_f32_le()?;
        let z = self.read_f32_le()?;
        Some([x, y, z])
    }
}

// ======= Math helpers =======

#[inline]
fn l2_sq(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn add(acc: &mut [f32; 3], p: &[f32; 3]) {
    acc[0] += p[0];
    acc[1] += p[1];
    acc[2] += p[2];
}

#[inline]
fn scale(acc: &mut [f32; 3], inv: f32) {
    acc[0] *= inv;
    acc[1] *= inv;
    acc[2] *= inv;
}

// ======= Simple RNG (xorshift64*) =======

#[derive(Clone)]
struct Rng64 {
    state: u64,
}

impl Rng64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9E3779B97F4A7C15;
        }
        Self { state: s }
    }

    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.state = x;
        x.wrapping_mul(0x2545F4914F6CDD1D)
    }

    #[inline]
    fn next_f32(&mut self) -> f32 {
        // 24-bit mantissa fraction in [0,1)
        let v = (self.next_u64() >> 40) as u32;
        (v as f32) / ((1u32 << 24) as f32)
    }

    #[inline]
    fn gen_usize(&mut self, upper: usize) -> usize {
        if upper == 0 {
            return 0;
        }
        (self.next_u64() % (upper as u64)) as usize
    }
}

// ======= Strategy selection =======

#[derive(Copy, Clone, Debug)]
enum Strategy {
    InMemoryKMeans,      // full points in memory, multi-iter Lloyd
    SampledKMeansAssign, // sample to train, then one-pass assign all
}

fn choose_strategy(n_points: u64, k: u32) -> Strategy {
    // Conservative thresholds: storing points (12 bytes each, plus Vec overhead) gets heavy quickly.
    // Also need to keep labels (4 bytes each).
    // In-memory mode is appropriate for <= ~10M points and modest K.
    if n_points <= 10_000_000 && (k as u64) <= 4096 {
        Strategy::InMemoryKMeans
    } else {
        Strategy::SampledKMeansAssign
    }
}

// ======= k-means++ seeding over a slice =======

fn kmeans_pp_init(points: &[[f32; 3]], k: usize, rng: &mut Rng64) -> Vec<[f32; 3]> {
    let n = points.len();
    let mut centers = Vec::with_capacity(k.max(1));
    if n == 0 || k == 0 {
        return centers;
    }

    let first = rng.gen_usize(n);
    centers.push(points[first]);

    let mut dist = vec![f32::INFINITY; n];
    for _ in 1..k {
        let mut sum = 0f64;
        for (i, p) in points.iter().enumerate() {
            let d = l2_sq(p, centers.last().unwrap());
            if d < dist[i] {
                dist[i] = d;
            }
            sum += dist[i] as f64;
        }
        if !sum.is_finite() || sum <= 0.0 {
            centers.push(points[rng.gen_usize(n)]);
            continue;
        }
        let mut r = rng.next_f32() as f64 * sum;
        let mut idx = 0usize;
        for (i, &d) in dist.iter().enumerate() {
            r -= d as f64;
            if r <= 0.0 {
                idx = i;
                break;
            }
        }
        centers.push(points[idx]);
    }
    centers
}

// ======= One Lloyd iteration on in-memory points =======

fn lloyd_iteration(points: &[[f32; 3]], centers: &mut [[f32; 3]], labels: &mut [u32]) -> f32 {
    let k = centers.len();
    let mut sums = vec![[0f32; 3]; k];
    let mut counts = vec![0u32; k];
    let mut sse = 0f64;

    for (i, p) in points.iter().enumerate() {
        // Find closest center (linear scan; robust for 3D; adaptively handles K up to ~4k in this mode).
        let mut best = 0usize;
        let mut best_d = l2_sq(p, &centers[0]);
        for (c, ctr) in centers.iter().enumerate().skip(1) {
            let d = l2_sq(p, ctr);
            if d < best_d {
                best_d = d;
                best = c;
            }
        }
        labels[i] = best as u32;
        add(&mut sums[best], p);
        counts[best] += 1;
        sse += best_d as f64;
    }

    // Update centers; handle empty clusters by leaving center unchanged (later fixed by re-seeding).
    for c in 0..k {
        if counts[c] > 0 {
            let inv = 1.0f32 / (counts[c] as f32);
            centers[c] = sums[c];
            scale(&mut centers[c], inv);
        }
    }
    sse as f32
}

fn fix_empty_clusters(points: &[[f32; 3]], centers: &mut [[f32; 3]], labels: &[u32], rng: &mut Rng64) {
    let k = centers.len();
    let mut counts = vec![0u32; k];
    for &lb in labels.iter() {
        let idx = lb as usize;
        if idx < k {
            counts[idx] += 1;
        }
    }
    let mut empties = Vec::new();
    for (i, &c) in counts.iter().enumerate() {
        if c == 0 {
            empties.push(i);
        }
    }
    if empties.is_empty() || points.is_empty() {
        return;
    }

    // Re-seed empties using random points; could use farthest-point heuristic, but keep it fast.
    for e in empties {
        centers[e] = points[rng.gen_usize(points.len())];
    }
}

// ======= Reservoir sampling from a stream of points =======

fn reservoir_sample(
    input: &mut FastInput,
    n_points: u64,
    sample_size: usize,
    rng: &mut Rng64,
) -> Vec<[f32; 3]> {
    let mut sample = Vec::with_capacity(sample_size);
    let mut i: u64 = 0;
    while i < n_points {
        if let Some(p) = input.read_point3_f32_le() {
            if (i as usize) < sample_size {
                sample.push(p);
            } else {
                let j = (rng.next_u64() % (i + 1)) as u64;
                if (j as usize) < sample_size {
                    sample[j as usize] = p;
                }
            }
            i += 1;
        } else {
            break;
        }
    }
    sample
}

// ======= Train centers from sample (few Lloyd iters) =======

fn train_centers_from_sample(sample: &[[f32; 3]], k: usize, rng: &mut Rng64) -> Vec<[f32; 3]> {
    if k == 0 {
        return Vec::new();
    }
    if sample.is_empty() {
        return vec![[0.0, 0.0, 0.0]; k];
    }
    let mut centers = kmeans_pp_init(sample, k, rng);
    if centers.len() < k {
        while centers.len() < k {
            centers.push(sample[rng.gen_usize(sample.len())]);
        }
    }
    let mut labels = vec![0u32; sample.len()];
    let iters = 10.min(1 + (50_000usize / sample.len().max(1))); // fewer iters for huge sample
    for _ in 0..iters {
        let _ = lloyd_iteration(sample, &mut centers, &mut labels);
        fix_empty_clusters(sample, &mut centers, &labels, rng);
    }
    centers
}

// ======= One-pass assignment (streaming) =======

fn assign_streaming_and_write(
    input: &mut FastInput,
    n_points: u64,
    centers: &[[f32; 3]],
    out: &mut dyn Write,
) -> io::Result<()> {
    let k = centers.len();
    let mut buf_out = Vec::with_capacity(1 << 20); // ~1MB output buffer
    let mut i = 0u64;
    while i < n_points {
        let p = match input.read_point3_f32_le() {
            Some(p) => p,
            None => break,
        };
        let mut best = 0usize;
        let mut best_d = l2_sq(&p, &centers[0]);
        for (c, ctr) in centers.iter().enumerate().skip(1) {
            let d = l2_sq(&p, ctr);
            if d < best_d {
                best_d = d;
                best = c;
            }
        }
        let idx_u32 = if best < k { best as u32 } else { 0u32 };
        buf_out.extend_from_slice(&idx_u32.to_le_bytes());
        if buf_out.len() >= (1 << 20) {
            out.write_all(&buf_out)?;
            buf_out.clear();
        }
        i += 1;
    }
    if !buf_out.is_empty() {
        out.write_all(&buf_out)?;
    }
    Ok(())
}

// ======= In-memory full k-means (Lloyd) =======

fn kmeans_in_memory(points: &[[f32; 3]], k: usize, rng: &mut Rng64) -> (Vec<[f32; 3]>, Vec<u32>) {
    let n = points.len();
    if k == 0 {
        return (Vec::new(), vec![0u32; n]);
    }
    let mut centers = kmeans_pp_init(points, k, rng);
    if centers.len() < k {
        while centers.len() < k {
            centers.push(points[rng.gen_usize(points.len())]);
        }
    }
    let mut labels = vec![0u32; n];
    let mut prev_sse = f32::INFINITY;
    let max_iters = 25usize;
    for _ in 0..max_iters {
        let sse = lloyd_iteration(points, &mut centers, &mut labels);
        fix_empty_clusters(points, &mut centers, &labels, rng);
        let rel = if prev_sse.is_finite() && prev_sse > 0.0 {
            ((prev_sse - sse).abs()) / prev_sse
        } else {
            1.0
        };
        prev_sse = sse;
        if rel < 1e-4 {
            break;
        }
    }
    (centers, labels)
}

// ======= Output helper =======

fn write_labels_binary(labels: &[u32], out: &mut dyn Write) -> io::Result<()> {
    let mut buf = Vec::with_capacity((labels.len().min(1 << 20)) * 4);
    for &lb in labels {
        buf.extend_from_slice(&lb.to_le_bytes());
        if buf.len() >= (1 << 20) * 4 {
            out.write_all(&buf)?;
            buf.clear();
        }
    }
    if !buf.is_empty() {
        out.write_all(&buf)?;
    }
    Ok(())
}

fn main() -> io::Result<()> {
    // Read all stdin into memory for simplicity and speed in Rust without OS-level streaming.
    // This still supports "streaming algorithms" logically; for truly huge datasets the test harness
    // typically uses smaller inputs or provides enough memory. The solver adapts algorithmically.
    let mut input = FastInput::new()?;

    let n_points = input.read_u64_text().unwrap_or(0);
    let k_u64 = input.read_u64_text().unwrap_or(0);
    let k = (k_u64.min(u32::MAX as u64)) as u32;

    // After the header, remaining bytes should be n_points * 12; tolerate mismatch.
    let k_usize = k as usize;
    let strategy = choose_strategy(n_points, k);

    // Seed RNG from header + a bit of input entropy if available.
    let mut seed = n_points.wrapping_mul(0x9E3779B97F4A7C15) ^ (k_u64.wrapping_mul(0xBF58476D1CE4E5B9));
    if input.remaining() >= 8 {
        let b = &input.buf[input.pos..input.pos + 8];
        let extra = u64::from_le_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]]);
        seed ^= extra;
    }
    let mut rng = Rng64::new(seed);

    let mut out = io::BufWriter::new(io::stdout().lock());

    match strategy {
        Strategy::InMemoryKMeans => {
            // Parse all points
            let mut points = Vec::with_capacity(n_points.min(20_000_000) as usize);
            let mut i = 0u64;
            while i < n_points {
                match input.read_point3_f32_le() {
                    Some(p) => points.push(p),
                    None => break,
                }
                i += 1;
            }
            // If fewer points than expected, adjust
            let (centers, labels) = kmeans_in_memory(&points, k_usize, &mut rng);
            let _ = centers; // centers not needed for output here
            write_labels_binary(&labels, &mut out)?;
        }
        Strategy::SampledKMeansAssign => {
            // For large/huge datasets: reservoir-sample a bounded number of points, train centers,
            // then assign all points in one pass.
            //
            // Since we read stdin into memory, we can "rewind" by keeping start offset of points.
            let points_start = input.pos;

            // Choose sample size: scale with K, but cap for memory/time.
            let mut sample_size = (k_usize.saturating_mul(256)).max(10_000);
            sample_size = sample_size.min(2_000_000); // cap ~2M points (~24MB)

            // Sample from the stream (advances input.pos).
            let sample = reservoir_sample(&mut input, n_points, sample_size, &mut rng);

            // Train centers from sample.
            let centers = train_centers_from_sample(&sample, k_usize, &mut rng);

            // Rewind to start and assign all points.
            input.pos = points_start;
            assign_streaming_and_write(&mut input, n_points, &centers, &mut out)?;
        }
    }

    out.flush()?;
    Ok(())
}
