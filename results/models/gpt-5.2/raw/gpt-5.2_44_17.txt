#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Procedural brick pattern parameters
    const vec2 scale = vec2(8.0, 16.0);
    const float mortarX = 0.05;
    const float mortarY = 0.10;
    const vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    const vec3 brickColor  = vec3(0.7, 0.2, 0.1);

    // Scale UVs into brick grid space
    vec2 st = uv * scale;

    // Stagger rows: half-brick offset every other row
    float row = floor(st.y);
    if (mod(row, 2.0) >= 1.0) {
        st.x += 0.5;
    }

    vec2 f = fract(st);

    // Mortar bands near edges of each brick cell
    float mx = step(f.x, mortarX) + step(1.0 - mortarX, f.x);
    float my = step(f.y, mortarY) + step(1.0 - mortarY, f.y);
    float mortar = clamp(mx + my, 0.0, 1.0);

    vec3 baseColor = mix(brickColor, mortarColor, mortar);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    outColor = vec4(baseColor * shade, 1.0);
}
