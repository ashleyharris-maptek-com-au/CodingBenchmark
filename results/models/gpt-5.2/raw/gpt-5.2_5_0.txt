use std::cmp::Reverse;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone)]
struct Grid {
    w: usize,
    h: usize,
    blocked: Vec<bool>,
    open: Vec<bool>,
    n_open: usize,
}

impl Grid {
    fn idx(&self, x: usize, y: usize) -> usize {
        y * self.w + x
    }
    fn inb(&self, x: i32, y: i32) -> bool {
        x >= 0 && y >= 0 && (x as usize) < self.w && (y as usize) < self.h
    }
    fn is_open_idx(&self, i: usize) -> bool {
        self.open[i]
    }
    fn xy(&self, i: usize) -> (usize, usize) {
        (i % self.w, i / self.w)
    }
    fn neigh_indices(&self, i: usize, out: &mut [usize; 4]) -> usize {
        let (x, y) = self.xy(i);
        let mut k = 0usize;
        if x > 0 {
            let j = i - 1;
            if self.open[j] {
                out[k] = j;
                k += 1;
            }
        }
        if x + 1 < self.w {
            let j = i + 1;
            if self.open[j] {
                out[k] = j;
                k += 1;
            }
        }
        if y > 0 {
            let j = i - self.w;
            if self.open[j] {
                out[k] = j;
                k += 1;
            }
        }
        if y + 1 < self.h {
            let j = i + self.w;
            if self.open[j] {
                out[k] = j;
                k += 1;
            }
        }
        k
    }
}

fn parse_input() -> (Grid, bool) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            // No input.
            std::process::exit(0);
        }
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: i32 = it.next().unwrap().parse().unwrap();
    let mut blocked = vec![false; w * h];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }
    let mut open = vec![true; w * h];
    for i in 0..w * h {
        if blocked[i] {
            open[i] = false;
        }
    }
    let n_open = open.iter().filter(|&&b| b).count();
    (
        Grid {
            w,
            h,
            blocked,
            open,
            n_open,
        },
        require_cycle != 0,
    )
}

fn start_and_sanity(g: &Grid, require_cycle: bool) -> Option<usize> {
    if g.w == 0 || g.h == 0 {
        return None;
    }
    let s = g.idx(0, 0);
    if !g.is_open_idx(s) {
        return None;
    }
    if g.n_open == 0 {
        return None;
    }
    if require_cycle && g.n_open == 1 {
        return None;
    }
    Some(s)
}

fn parity_necessary(g: &Grid, require_cycle: bool) -> bool {
    // Checkerboard parity necessary condition.
    // For a Hamiltonian path on a bipartite graph:
    // - If it's a path (not required cycle): |A|-|B| must be 0 or 1, and endpoints are on:
    //     - same part if diff==1 (larger part)
    //     - opposite parts if diff==0
    // Here start is fixed at (0,0) which is even.
    // - If it's a cycle: |A|==|B|.
    let mut even = 0usize;
    let mut odd = 0usize;
    for y in 0..g.h {
        for x in 0..g.w {
            let i = g.idx(x, y);
            if !g.open[i] {
                continue;
            }
            if ((x + y) & 1) == 0 {
                even += 1;
            } else {
                odd += 1;
            }
        }
    }
    if require_cycle {
        even == odd
    } else {
        let diff = if even >= odd { even - odd } else { odd - even };
        if diff > 1 {
            return false;
        }
        // Start at even (0,0). If even==odd+1, endpoints must be even; possible.
        // If odd==even+1, endpoints must be odd; impossible because start is even.
        if odd == even + 1 {
            return false;
        }
        true
    }
}

fn connected_necessary(g: &Grid, start: usize) -> bool {
    // BFS over open cells to ensure single component.
    let n = g.w * g.h;
    let mut vis = vec![false; n];
    let mut q = VecDeque::new();
    vis[start] = true;
    q.push_back(start);
    let mut seen = 1usize;
    let mut neigh = [0usize; 4];
    while let Some(u) = q.pop_front() {
        let k = g.neigh_indices(u, &mut neigh);
        for t in 0..k {
            let v = neigh[t];
            if !vis[v] {
                vis[v] = true;
                seen += 1;
                q.push_back(v);
            }
        }
    }
    seen == g.n_open
}

fn degrees(g: &Grid) -> Vec<u8> {
    let n = g.w * g.h;
    let mut deg = vec![0u8; n];
    let mut neigh = [0usize; 4];
    for i in 0..n {
        if !g.open[i] {
            continue;
        }
        let k = g.neigh_indices(i, &mut neigh);
        deg[i] = k as u8;
    }
    deg
}

fn degree_necessary(g: &Grid, start: usize, require_cycle: bool, deg: &[u8]) -> bool {
    // Any open cell (except possibly start/end in path case) must have degree >= 2
    // in the induced graph; degree 0 impossible; in cycle case, all need degree>=2.
    // For fixed start path: start needs degree>=1, end needs degree>=1.
    if require_cycle {
        for i in 0..g.w * g.h {
            if g.open[i] && deg[i] < 2 {
                return false;
            }
        }
        return true;
    }
    // Path case:
    if deg[start] == 0 {
        return false;
    }
    for i in 0..g.w * g.h {
        if !g.open[i] || i == start {
            continue;
        }
        if deg[i] == 0 {
            return false;
        }
    }
    true
}

fn is_neighbor(g: &Grid, a: usize, b: usize) -> bool {
    if a == b {
        return false;
    }
    let ax = a % g.w;
    let ay = a / g.w;
    let bx = b % g.w;
    let by = b / g.w;
    let dx = if ax > bx { ax - bx } else { bx - ax };
    let dy = if ay > by { ay - by } else { by - ay };
    dx + dy == 1
}

fn deterministic_fill_order(g: &Grid) -> Vec<usize> {
    // Simple fallback traversal order: row-major over open cells.
    let mut v = Vec::with_capacity(g.n_open);
    for y in 0..g.h {
        for x in 0..g.w {
            let i = g.idx(x, y);
            if g.open[i] {
                v.push(i);
            }
        }
    }
    v
}

fn snake_path_rectangle(w: usize, h: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(w * h);
    for y in 0..h {
        if (y & 1) == 0 {
            for x in 0..w {
                path.push((x, y));
            }
        } else {
            for x in (0..w).rev() {
                path.push((x, y));
            }
        }
    }
    path
}

fn try_fast_perfect_rectangle(g: &Grid, require_cycle: bool) -> Option<Vec<usize>> {
    // If no obstacles, produce a constructive solution immediately.
    if g.n_open != g.w * g.h {
        return None;
    }
    let mut coords = snake_path_rectangle(g.w, g.h);
    if coords.is_empty() || coords[0] != (0, 0) {
        // Our snake always starts at (0,0).
        return None;
    }
    if require_cycle {
        // Need last adjacent to start.
        let (lx, ly) = *coords.last().unwrap();
        let ok = (lx == 1 && ly == 0) || (lx == 0 && ly == 1);
        if !ok {
            // Simple rectangular cycle exists iff at least one dimension even.
            // Our snake doesn't necessarily end adjacent. Build a known cycle ordering when possible:
            // If w is even, traverse rows in a cycle-friendly pattern.
            if g.w % 2 == 0 && g.w >= 2 && g.h >= 2 {
                coords.clear();
                // Cycle for even width: go along top row left->right,
                // then snake back and finish at (1,0) adjacent to start.
                for x in 0..g.w {
                    coords.push((x, 0));
                }
                for y in 1..g.h {
                    if (y & 1) == 1 {
                        for x in (0..g.w).rev() {
                            coords.push((x, y));
                        }
                    } else {
                        for x in 0..g.w {
                            coords.push((x, y));
                        }
                    }
                }
                // This ends at (0, h-1) if h odd else (w-1,h-1); not guaranteed adjacent.
                // Instead, use a simple standard cycle construction for even width:
                coords.clear();
                // Construct by making a "ladder": go down first column, then up next, etc.
                // This yields a Hamiltonian cycle for even width (grid graph) by ordering the cycle,
                // but here we need a Hamiltonian path that closes to start. We output cycle as a path
                // starting at (0,0) and ending adjacent to it.
                // Use: walk down column 0, up column 1, down column 2, ...; this ends at (1,0) when w even.
                for x in 0..g.w {
                    if (x & 1) == 0 {
                        for y in 0..g.h {
                            coords.push((x, y));
                        }
                    } else {
                        for y in (0..g.h).rev() {
                            coords.push((x, y));
                        }
                    }
                }
                let (lx2, ly2) = *coords.last().unwrap();
                if !((lx2 == 1 && ly2 == 0) || (lx2 == 0 && ly2 == 1)) {
                    // If not adjacent, fall back to search.
                    return None;
                }
            } else if g.h % 2 == 0 && g.w >= 2 && g.h >= 2 {
                // Similar ladder by rows ends at (0,1) when h even.
                coords.clear();
                for y in 0..g.h {
                    if (y & 1) == 0 {
                        for x in 0..g.w {
                            coords.push((x, y));
                        }
                    } else {
                        for x in (0..g.w).rev() {
                            coords.push((x, y));
                        }
                    }
                }
                let (lx2, ly2) = *coords.last().unwrap();
                if !((lx2 == 1 && ly2 == 0) || (lx2 == 0 && ly2 == 1)) {
                    // Try transposed ladder for even height:
                    coords.clear();
                    for y in 0..g.h {
                        if (y & 1) == 0 {
                            for x in 0..g.w {
                                coords.push((x, y));
                            }
                        } else {
                            for x in (0..g.w).rev() {
                                coords.push((x, y));
                            }
                        }
                    }
                    let (lx3, ly3) = *coords.last().unwrap();
                    if !((lx3 == 1 && ly3 == 0) || (lx3 == 0 && ly3 == 1)) {
                        return None;
                    }
                }
            } else {
                // Odd x odd rectangle cannot have Hamiltonian cycle.
                return None;
            }
        }
    }
    let mut out = Vec::with_capacity(coords.len());
    for (x, y) in coords {
        out.push(g.idx(x, y));
    }
    Some(out)
}

fn greedy_path_with_repairs(g: &Grid, start: usize, require_cycle: bool) -> Option<Vec<usize>> {
    // Heuristic DFS with Warnsdorff-style choice + connectivity pruning.
    // Designed to scale to ~2500 open cells with sparse obstacles.
    let n = g.w * g.h;
    let deg = degrees(g);
    if !degree_necessary(g, start, require_cycle, &deg) {
        return None;
    }

    let mut used = vec![false; n];
    used[start] = true;
    let mut path = Vec::with_capacity(g.n_open);
    path.push(start);

    let mut neigh = [0usize; 4];

    // Stack for iterative DFS: (current, next_choice_index, candidate_list)
    // Candidate list stored in a Vec<usize> per depth for backtracking.
    let mut choices: Vec<Vec<usize>> = Vec::with_capacity(g.n_open);
    choices.push(Vec::new());
    let mut choice_pos: Vec<usize> = Vec::with_capacity(g.n_open);
    choice_pos.push(0);

    // For speed, maintain dynamic remaining degree counts to pick low-degree moves.
    // rem_deg[v] = number of open unused neighbors.
    let mut rem_deg = vec![0u8; n];
    for i in 0..n {
        if !g.open[i] {
            continue;
        }
        let k = g.neigh_indices(i, &mut neigh);
        rem_deg[i] = k as u8;
    }

    // Helper: update rem_deg around a vertex when marking/unmarking used.
    fn mark_used(g: &Grid, v: usize, used: &mut [bool], rem_deg: &mut [u8]) {
        let mut neigh = [0usize; 4];
        used[v] = true;
        let k = g.neigh_indices(v, &mut neigh);
        for t in 0..k {
            let u = neigh[t];
            if !used[u] && rem_deg[u] > 0 {
                rem_deg[u] -= 1;
            }
        }
    }
    fn unmark_used(g: &Grid, v: usize, used: &mut [bool], rem_deg: &mut [u8]) {
        let mut neigh = [0usize; 4];
        used[v] = false;
        let k = g.neigh_indices(v, &mut neigh);
        for t in 0..k {
            let u = neigh[t];
            if !used[u] {
                rem_deg[u] += 1;
            }
        }
    }

    // Seed for start already used: decrease rem_deg of its unused neighbors by 1.
    {
        let k = g.neigh_indices(start, &mut neigh);
        for t in 0..k {
            let u = neigh[t];
            if !used[u] && rem_deg[u] > 0 {
                rem_deg[u] -= 1;
            }
        }
    }

    // Connectivity / cut pruning: if there exists an unused vertex (not current end)
    // with rem_deg == 0, impossible. If too many vertices with rem_deg==1, prune lightly.
    let mut dead_count = 0usize;
    for i in 0..n {
        if g.open[i] && !used[i] {
            if rem_deg[i] == 0 {
                dead_count += 1;
            }
        }
    }
    if dead_count > 0 {
        return None;
    }

    // Limit for backtracking effort based on size; keep within 5 minutes in worst cases.
    // This is a heuristic guardrail; it may return None even if a path exists.
    let max_nodes = g.n_open;
    let mut budget: u64 = if max_nodes <= 200 {
        50_000_000
    } else if max_nodes <= 800 {
        120_000_000
    } else if max_nodes <= 2500 {
        250_000_000
    } else {
        80_000_000
    };

    while path.len() < g.n_open {
        if budget == 0 {
            return None;
        }
        budget -= 1;

        let cur = *path.last().unwrap();

        if choices.len() < path.len() {
            choices.push(Vec::new());
            choice_pos.push(0);
        }

        if choices[path.len() - 1].is_empty() {
            // Build candidate list for this depth.
            let k = g.neigh_indices(cur, &mut neigh);
            let mut cand: Vec<(u8, Reverse<u8>, usize)> = Vec::with_capacity(k);
            for t in 0..k {
                let v = neigh[t];
                if used[v] {
                    continue;
                }
                // Primary key: rem_deg[v] (fewest onward moves)
                // Secondary: prefer moves that don't create early dead ends (count of neighbors with rem_deg==1 after move approx)
                let onward = rem_deg[v];
                // Rough tie-break: static degree (higher can be safer)
                let static_deg = deg[v];
                cand.push((onward, Reverse(static_deg), v));
            }
            // Sort by onward ascending, then static_deg descending.
            cand.sort_by(|a, b| {
                if a.0 != b.0 {
                    a.0.cmp(&b.0)
                } else {
                    a.1.cmp(&b.1)
                }
            });
            let mut list = Vec::with_capacity(cand.len());
            for (_, _, v) in cand {
                list.push(v);
            }
            choices[path.len() - 1] = list;
            choice_pos[path.len() - 1] = 0;
        }

        let pos = choice_pos[path.len() - 1];
        if pos >= choices[path.len() - 1].len() {
            // Backtrack.
            choices[path.len() - 1].clear();
            choice_pos[path.len() - 1] = 0;
            if path.len() == 1 {
                return None;
            }
            let last = path.pop().unwrap();
            unmark_used(g, last, &mut used, &mut rem_deg);
            // Recompute dead/ones locally is expensive; do a quick global check occasionally.
            let prev_depth = path.len() - 1;
            choice_pos[prev_depth] += 1;
            continue;
        }

        let nxt = choices[path.len() - 1][pos];

        // Tentatively move.
        mark_used(g, nxt, &mut used, &mut rem_deg);
        path.push(nxt);
        // Reset child choice list
        if choices.len() >= path.len() {
            choices[path.len() - 1].clear();
            choice_pos[path.len() - 1] = 0;
        }

        // Prune: no unused vertex can have rem_deg==0 unless it's allowed endpoint at final step.
        // Also ensure we haven't isolated start if cycle required (end must connect back at end).
        let mut has_dead = false;
        // Quick local check: neighbors of nxt changed, plus nxt itself.
        // But for safety on obstacles, do a small scan every few steps.
        if (path.len() & 31) == 0 {
            for i in 0..n {
                if g.open[i] && !used[i] && rem_deg[i] == 0 {
                    has_dead = true;
                    break;
                }
            }
        } else {
            // Check nxt neighbors and their neighbors for newly dead vertices.
            let kk = g.neigh_indices(nxt, &mut neigh);
            for t in 0..kk {
                let u = neigh[t];
                if g.open[u] && !used[u] && rem_deg[u] == 0 {
                    has_dead = true;
                    break;
                }
            }
        }
        if has_dead {
            // Undo move, advance choice.
            let v = path.pop().unwrap();
            unmark_used(g, v, &mut used, &mut rem_deg);
            choice_pos[path.len() - 1] += 1;
            continue;
        }

        // Forced move propagation: if any unused vertex has rem_deg==1, prioritize its only neighbor.
        // Implement lightly: if current has exactly one candidate, we'll naturally take it due to sorting.
        // Additional propagation omitted for simplicity.
    }

    // Cycle constraint check.
    if require_cycle {
        let end = *path.last().unwrap();
        if !is_neighbor(g, end, start) {
            return None;
        }
    }
    // Validate adjacency and uniqueness quickly.
    for i in 1..path.len() {
        if !is_neighbor(g, path[i - 1], path[i]) {
            return None;
        }
    }
    Some(path)
}

fn solve(g: &Grid, require_cycle: bool) -> Option<Vec<usize>> {
    let start = start_and_sanity(g, require_cycle)?;
    if !parity_necessary(g, require_cycle) {
        return None;
    }
    if !connected_necessary(g, start) {
        return None;
    }

    if let Some(p) = try_fast_perfect_rectangle(g, require_cycle) {
        return Some(p);
    }

    // For very large or very sparse graphs, greedy path is the best shot.
    // Try heuristic backtracking once.
    if let Some(p) = greedy_path_with_repairs(g, start, require_cycle) {
        return Some(p);
    }

    // Last-ditch: attempt a deterministic path order when obstacles are few and grid is small (may fail).
    // We only accept if it is a valid Hamiltonian path.
    if g.n_open <= 200 {
        let order = deterministic_fill_order(g);
        // Check if order is a valid path starting at start and adjacent steps.
        if !order.is_empty() && order[0] == start {
            let mut ok = true;
            for i in 1..order.len() {
                if !is_neighbor(g, order[i - 1], order[i]) {
                    ok = false;
                    break;
                }
            }
            if ok && (!require_cycle || is_neighbor(g, *order.last().unwrap(), start)) {
                return Some(order);
            }
        }
    }
    None
}

fn main() {
    let (g, require_cycle) = parse_input();
    let Some(start) = start_and_sanity(&g, require_cycle) else {
        println!("0");
        return;
    };
    // Keep start used for lint silence
    let _ = start;
    match solve(&g, require_cycle) {
        None => {
            println!("0");
        }
        Some(path) => {
            println!("{}", path.len());
            for i in path {
                let (x, y) = g.xy(i);
                println!("{} {}", x, y);
            }
        }
    }
}
