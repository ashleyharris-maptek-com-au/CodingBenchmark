```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint j = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (j >= N) return;

    float dt         = radii.x;
    float sepRadius  = radii.y;
    float alnRadius  = radii.z;
    float cohRadius  = radii.w;

    float sepWeight  = weights.x;
    float alnWeight  = weights.y;
    float cohWeight  = weights.z;
    float maxSpeed   = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    vec3 pos = inData[j * 2u + 0u].xyz;
    vec3 vel = inData[j * 2u + 1u].xyz;

    vec3 sepAcc = vec3(0.0);
    vec3 alnAcc = vec3(0.0);
    vec3 cohAcc = vec3(0.0);

    uint sepCount = 0u;
    uint alnCount = 0u;
    uint cohCount = 0u;

    float sepR2 = sepRadius * sepRadius;
    float alnR2 = alnRadius * alnRadius;
    float cohR2 = cohRadius * cohRadius;

    for (uint i = 0u; i < N; ++i) {
        if (i == j) continue;

        vec3 otherPos = inData[i * 2u + 0u].xyz;
        vec3 d = pos - otherPos;
        float dist2 = dot(d, d);
        if (dist2 <= 0.0) continue;

        if (dist2 < sepR2) {
            // Separation: steer away; weighted more strongly when close.
            // Use inverse distance (avoid sqrt by using dist2).
            sepAcc += d * inversesqrt(dist2);
            sepCount++;
        }

        if (dist2 < alnR2) {
            vec3 otherVel = inData[i * 2u + 1u].xyz;
            alnAcc += otherVel;
            alnCount++;
        }

        if (dist2 < cohR2) {
            cohAcc += otherPos;
            cohCount++;
        }
    }

    if (sepCount > 0u) sepAcc /= float(sepCount);
    if (alnCount > 0u) alnAcc = (alnAcc / float(alnCount)) - vel; // steer toward avg velocity
    if (cohCount > 0u) cohAcc = (cohAcc / float(cohCount)) - pos; // steer toward center of mass

    vec3 accel = sepAcc * sepWeight + alnAcc * alnWeight + cohAcc * cohWeight;

    // Boundary force: if beyond boundSize*0.4 in any axis, push toward center.
    // Use boundSize*0.5 scaling as requested.
    float threshold = boundSize * 0.4;
    float halfSize  = boundSize * 0.5;
    if (abs(pos.x) > threshold) accel.x += (-pos.x / halfSize) * boundForce;
    if (abs(pos.y) > threshold) accel.y += (-pos.y / halfSize) * boundForce;
    if (abs(pos.z) > threshold) accel.z += (-pos.z / halfSize) * boundForce;

    vel += accel * dt;

    float spd2 = dot(vel, vel);
    float maxSpd2 = maxSpeed * maxSpeed;
    if (spd2 > maxSpd2 && spd2 > 0.0) {
        vel *= maxSpeed * inversesqrt(spd2);
    }

    pos += vel * dt;

    outData[j * 2u + 0u] = vec4(pos, 0.0);
    outData[j * 2u + 1u] = vec4(vel, 0.0);
}
```
