#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

// Steiner Tree solver:
// - Exact Dreyfus-Wagner DP when number of terminals K <= 20 (configurable).
// - Otherwise, heuristic: metric-closure MST over terminals using multi-source Dijkstra,
//   then expand MST edges to original shortest paths and take MST again to remove cycles.
//
// Output: total_weight edge_count then edges (u v), 0-indexed.

using ll = long long;
static constexpr ll INFLL = (std::numeric_limits<ll>::max() / 4);

struct Edge {
  int to;
  int w;
  int id;
};

struct InputEdge {
  int u, v;
  int w;
};

struct DSU {
  std::vector<int> p, r;
  explicit DSU(int n = 0) : p(n), r(n, 0) { std::iota(p.begin(), p.end(), 0); }
  int find(int a) {
    while (p[a] != a) {
      p[a] = p[p[a]];
      a = p[a];
    }
    return a;
  }
  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (r[a] < r[b]) std::swap(a, b);
    p[b] = a;
    if (r[a] == r[b]) r[a]++;
    return true;
  }
};

struct DijkstraResult {
  std::vector<ll> dist;
  std::vector<int> parent_v;
  std::vector<int> parent_eid;
};

static DijkstraResult dijkstra(int n, const std::vector<std::vector<Edge>>& g, int src) {
  DijkstraResult res;
  res.dist.assign(n, INFLL);
  res.parent_v.assign(n, -1);
  res.parent_eid.assign(n, -1);
  using P = std::pair<ll, int>;
  std::priority_queue<P, std::vector<P>, std::greater<P>> pq;
  res.dist[src] = 0;
  pq.push({0, src});
  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != res.dist[v]) continue;
    for (const auto& e : g[v]) {
      ll nd = d + (ll)e.w;
      if (nd < res.dist[e.to]) {
        res.dist[e.to] = nd;
        res.parent_v[e.to] = v;
        res.parent_eid[e.to] = e.id;
        pq.push({nd, e.to});
      }
    }
  }
  return res;
}

// Reconstruct edges (by edge id) along shortest path from src to dst using parent pointers.
static void collect_path_edges(int src, int dst, const std::vector<int>& parent_v,
                               const std::vector<int>& parent_eid, std::vector<int>& out_edge_ids) {
  int cur = dst;
  while (cur != src && cur != -1) {
    int pe = parent_eid[cur];
    int pv = parent_v[cur];
    if (pe == -1 || pv == -1) break;
    out_edge_ids.push_back(pe);
    cur = pv;
  }
}

static std::vector<std::pair<int, int>> edges_from_ids(const std::vector<InputEdge>& edges,
                                                       const std::vector<int>& eids) {
  std::vector<std::pair<int, int>> out;
  out.reserve(eids.size());
  for (int id : eids) {
    int a = edges[id].u, b = edges[id].v;
    if (a > b) std::swap(a, b);
    out.push_back({a, b});
  }
  return out;
}

static void sort_unique_pairs(std::vector<std::pair<int, int>>& v) {
  std::sort(v.begin(), v.end());
  v.erase(std::unique(v.begin(), v.end()), v.end());
}

// Compute MST on a set of edges (u,v,w) over n vertices; returns chosen edge ids (indices into edges list).
struct WeightedEdge {
  int u, v;
  ll w;
  int id;  // arbitrary identifier
};

static std::vector<int> kruskal_ids(int n, std::vector<WeightedEdge>& we) {
  std::sort(we.begin(), we.end(), [](const WeightedEdge& a, const WeightedEdge& b) {
    if (a.w != b.w) return a.w < b.w;
    if (a.u != b.u) return a.u < b.u;
    return a.v < b.v;
  });
  DSU dsu(n);
  std::vector<int> chosen;
  for (const auto& e : we) {
    if (dsu.unite(e.u, e.v)) chosen.push_back(e.id);
  }
  return chosen;
}

// Exact Steiner tree for K <= 20 using Dreyfus-Wagner style DP with shortest paths.
// Returns edge ids in original graph that form the optimal Steiner tree.
static std::vector<int> steiner_exact(int n, const std::vector<std::vector<Edge>>& g,
                                      const std::vector<InputEdge>& edges,
                                      const std::vector<int>& terminals) {
  const int K = (int)terminals.size();
  const int FULL = 1 << K;

  // All-pairs shortest paths from every vertex (for transition and reconstruction).
  // For reconstruction we need parent pointers per source.
  std::vector<std::vector<ll>> dist_from(n, std::vector<ll>(n, INFLL));
  std::vector<std::vector<int>> parent_v_from(n), parent_e_from(n);
  for (int s = 0; s < n; s++) {
    auto dj = dijkstra(n, g, s);
    dist_from[s] = std::move(dj.dist);
    parent_v_from[s] = std::move(dj.parent_v);
    parent_e_from[s] = std::move(dj.parent_eid);
  }

  // dp[mask][v] = min cost of Steiner tree connecting terminals in mask and ending at v.
  std::vector<std::vector<ll>> dp(FULL, std::vector<ll>(n, INFLL));
  // For backtracking: store either split (mask1) at v, or "move" from u to v via shortest path.
  struct Prev {
    int type;      // 0 = none, 1 = split, 2 = move
    int a, b;      // for split: a=mask1, b=mask2 ; for move: a=from_vertex
  };
  std::vector<std::vector<Prev>> prev(FULL, std::vector<Prev>(n, Prev{0, 0, 0}));

  for (int i = 0; i < K; i++) {
    int t = terminals[i];
    int m = 1 << i;
    for (int v = 0; v < n; v++) {
      dp[m][v] = dist_from[t][v];
      prev[m][v] = Prev{2, t, 0};  // move from t to v
    }
  }

  // Subset DP: combine at same vertex, then run shortest-path relaxation (SPFA-like via Dijkstra on metric).
  // We'll do: for each mask, for each v: dp[mask][v] = min over submask: dp[sub][v]+dp[mask^sub][v]
  // Then "move" to allow ending at any other vertex via shortest paths: dp[mask][v] = min_u dp[mask][u]+dist(u,v)
  for (int mask = 1; mask < FULL; mask++) {
    if ((mask & (mask - 1)) == 0) continue;  // singletons already set
    for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
      int other = mask ^ sub;
      if (other == 0) continue;
      // avoid duplicates by requiring sub < other
      if (sub > other) continue;
      for (int v = 0; v < n; v++) {
        ll cand = dp[sub][v] + dp[other][v];
        if (cand < dp[mask][v]) {
          dp[mask][v] = cand;
          prev[mask][v] = Prev{1, sub, other};
        }
      }
    }

    // Now relax through all vertices using all-pairs dist_from.
    // O(n^2) per mask; ok for K<=20 and moderate n (benchmark-style).
    std::vector<ll> best = dp[mask];
    std::vector<int> best_from(n, -1);
    for (int u = 0; u < n; u++) best_from[u] = u;

    for (int u = 0; u < n; u++) {
      if (best[u] >= INFLL) continue;
      for (int v = 0; v < n; v++) {
        ll cand = best[u] + dist_from[u][v];
        if (cand < dp[mask][v]) {
          dp[mask][v] = cand;
          prev[mask][v] = Prev{2, u, 0};  // move from u to v
        }
      }
    }
  }

  // Choose best root vertex.
  int fullmask = FULL - 1;
  int best_v = 0;
  ll best_cost = dp[fullmask][0];
  for (int v = 1; v < n; v++) {
    if (dp[fullmask][v] < best_cost) {
      best_cost = dp[fullmask][v];
      best_v = v;
    }
  }

  // Backtrack to collect path edges.
  std::vector<int> chosen_edge_ids;
  chosen_edge_ids.reserve(n);

  std::function<void(int, int)> rec = [&](int mask, int v) {
    Prev p = prev[mask][v];
    if (p.type == 0) return;
    if (p.type == 1) {
      rec(p.a, v);
      rec(p.b, v);
      return;
    }
    if (p.type == 2) {
      int from = p.a;
      if (from == v) return;
      // add shortest path edges from 'from' to v using parent pointers from source 'from'
      collect_path_edges(from, v, parent_v_from[from], parent_e_from[from], chosen_edge_ids);
      // then continue recursion at 'from' with same mask, if this wasn't a base terminal move
      // but to know whether this is base, we can stop when mask is singleton and from is that terminal.
      if ((mask & (mask - 1)) == 0) {
        // singleton: base path from terminal to v, done
        return;
      }
      rec(mask, from);
    }
  };

  rec(fullmask, best_v);

  // Deduplicate edge ids (paths can overlap).
  std::sort(chosen_edge_ids.begin(), chosen_edge_ids.end());
  chosen_edge_ids.erase(std::unique(chosen_edge_ids.begin(), chosen_edge_ids.end()),
                        chosen_edge_ids.end());

  // The union of shortest paths can contain cycles; take MST over the induced subgraph edges
  // using original weights, restricted to vertices that appear in these edges.
  // Simpler: run Kruskal over all n vertices but with only these edges.
  std::vector<WeightedEdge> we;
  we.reserve(chosen_edge_ids.size());
  for (int id : chosen_edge_ids) {
    we.push_back({edges[id].u, edges[id].v, (ll)edges[id].w, id});
  }
  auto final_ids = kruskal_ids(n, we);
  return final_ids;
}

// Heuristic for large K:
// 1) For each terminal, run Dijkstra to get distances + parents.
// 2) Build complete graph on terminals with shortest path distances.
// 3) MST on terminals in metric closure.
// 4) Expand MST edges to original shortest paths (edge ids), union them.
// 5) Take MST again on union edges to ensure tree.
static std::vector<int> steiner_heuristic(int n, const std::vector<std::vector<Edge>>& g,
                                         const std::vector<InputEdge>& edges,
                                         const std::vector<int>& terminals) {
  const int K = (int)terminals.size();
  std::vector<DijkstraResult> dj(K);
  for (int i = 0; i < K; i++) dj[i] = dijkstra(n, g, terminals[i]);

  std::vector<WeightedEdge> metric_edges;
  metric_edges.reserve((size_t)K * (K - 1) / 2);
  int mid = 0;
  for (int i = 0; i < K; i++) {
    for (int j = i + 1; j < K; j++) {
      ll d = dj[i].dist[terminals[j]];
      if (d >= INFLL) continue;
      metric_edges.push_back({i, j, d, mid++});
    }
  }

  // MST over K nodes
  std::vector<WeightedEdge> metric_edges_copy = metric_edges;
  auto chosen_metric_ids = kruskal_ids(K, metric_edges_copy);

  // Map metric edge id -> (i,j)
  std::vector<std::pair<int, int>> metric_id_to_pair(mid, {-1, -1});
  for (const auto& me : metric_edges) metric_id_to_pair[me.id] = {me.u, me.v};

  std::vector<int> union_edge_ids;
  union_edge_ids.reserve((size_t)K * 10);
  for (int cid : chosen_metric_ids) {
    auto [i, j] = metric_id_to_pair[cid];
    if (i < 0) continue;
    int src_term = terminals[i];
    int dst_term = terminals[j];
    collect_path_edges(src_term, dst_term, dj[i].parent_v, dj[i].parent_eid, union_edge_ids);
  }

  std::sort(union_edge_ids.begin(), union_edge_ids.end());
  union_edge_ids.erase(std::unique(union_edge_ids.begin(), union_edge_ids.end()),
                       union_edge_ids.end());

  std::vector<WeightedEdge> we;
  we.reserve(union_edge_ids.size());
  for (int id : union_edge_ids) {
    we.push_back({edges[id].u, edges[id].v, (ll)edges[id].w, id});
  }
  auto final_ids = kruskal_ids(n, we);
  return final_ids;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n, m, k;
  if (!(std::cin >> n >> m >> k)) return 0;
  std::vector<InputEdge> edges;
  edges.reserve(m);
  std::vector<std::vector<Edge>> g(n);
  for (int i = 0; i < m; i++) {
    int u, v;
    int w;
    std::cin >> u >> v >> w;
    edges.push_back({u, v, w});
    g[u].push_back({v, w, i});
    g[v].push_back({u, w, i});
  }
  std::vector<int> terminals(k);
  for (int i = 0; i < k; i++) std::cin >> terminals[i];
  std::sort(terminals.begin(), terminals.end());
  terminals.erase(std::unique(terminals.begin(), terminals.end()), terminals.end());
  k = (int)terminals.size();

  // Trivial cases
  if (k <= 1) {
    std::cout << 0 << " " << 0 << "\n";
    return 0;
  }

  // Choose method.
  const int EXACT_K_MAX = 20;
  std::vector<int> chosen_edge_ids;
  if (k <= EXACT_K_MAX) {
    chosen_edge_ids = steiner_exact(n, g, edges, terminals);
  } else {
    chosen_edge_ids = steiner_heuristic(n, g, edges, terminals);
  }

  // Build adjacency from chosen edges and verify connectivity among terminals (best-effort).
  // Also compute total weight.
  ll total = 0;
  for (int id : chosen_edge_ids) total += edges[id].w;

  std::cout << total << " " << (int)chosen_edge_ids.size() << "\n";
  for (int id : chosen_edge_ids) {
    std::cout << edges[id].u << " " << edges[id].v << "\n";
  }
  return 0;
}
