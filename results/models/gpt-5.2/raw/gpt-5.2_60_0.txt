const fs = require('fs');

function solve(data) {
  const machineCount = Math.max(1, data.machineCount | 0);
  const tasksArr = Array.isArray(data.tasks) ? data.tasks : [];

  const n = tasksArr.length;
  if (n === 0) return { order: [] };

  // Map task id -> index
  const idToIdx = new Map();
  for (let i = 0; i < n; i++) idToIdx.set(tasksArr[i].id, i);

  const duration = new Array(n);
  const deadline = new Array(n);
  const weight = new Array(n);
  const deps = new Array(n);
  const succ = Array.from({ length: n }, () => []);
  const indeg = new Int32Array(n);

  for (let i = 0; i < n; i++) {
    const t = tasksArr[i] || {};
    duration[i] = (t.duration | 0) > 0 ? (t.duration | 0) : 0;
    // deadlines/weights may be missing; default to 0/1
    deadline[i] = Number.isFinite(t.deadline) ? t.deadline : 0;
    weight[i] = Number.isFinite(t.weight) ? t.weight : 1;
    const d = Array.isArray(t.deps) ? t.deps : [];
    deps[i] = d;
  }

  // Build graph; ignore unknown deps gracefully.
  for (let i = 0; i < n; i++) {
    const d = deps[i];
    for (let k = 0; k < d.length; k++) {
      const depId = d[k];
      const j = idToIdx.get(depId);
      if (j === undefined) continue;
      succ[j].push(i);
      indeg[i]++;
    }
  }

  // Topological order with a priority that tends to reduce weighted tardiness:
  // prioritize higher weight, tighter slack (deadline - duration), then longer duration.
  // Use a binary heap for scalability.
  class MaxHeap {
    constructor(cmp) {
      this.a = [];
      this.cmp = cmp;
    }
    push(x) {
      const a = this.a;
      a.push(x);
      let i = a.length - 1;
      while (i > 0) {
        const p = (i - 1) >> 1;
        if (this.cmp(a[i], a[p]) <= 0) break;
        const tmp = a[i];
        a[i] = a[p];
        a[p] = tmp;
        i = p;
      }
    }
    pop() {
      const a = this.a;
      if (a.length === 0) return undefined;
      const top = a[0];
      const last = a.pop();
      if (a.length) {
        a[0] = last;
        let i = 0;
        for (;;) {
          let l = i * 2 + 1;
          if (l >= a.length) break;
          let r = l + 1;
          let m = r < a.length && this.cmp(a[r], a[l]) > 0 ? r : l;
          if (this.cmp(a[m], a[i]) <= 0) break;
          const tmp = a[i];
          a[i] = a[m];
          a[m] = tmp;
          i = m;
        }
      }
      return top;
    }
    get size() {
      return this.a.length;
    }
  }

  // Heuristic score for selection among currently-available nodes.
  // Higher is better.
  // Add small tie-break on id string to keep determinism.
  const ids = tasksArr.map(t => t.id);
  const slack = new Array(n);
  for (let i = 0; i < n; i++) slack[i] = deadline[i] - duration[i];

  function cmpIdx(i, j) {
    // Weight dominant.
    const wi = weight[i], wj = weight[j];
    if (wi !== wj) return wi > wj ? 1 : -1;
    // Smaller slack first => higher priority.
    const si = slack[i], sj = slack[j];
    if (si !== sj) return si < sj ? 1 : -1;
    // Longer duration next.
    const di = duration[i], dj = duration[j];
    if (di !== dj) return di > dj ? 1 : -1;
    // Deadline earlier next.
    const dli = deadline[i], dlj = deadline[j];
    if (dli !== dlj) return dli < dlj ? 1 : -1;
    // Deterministic fallback: lexicographic by JSON string form.
    const ai = String(ids[i]);
    const aj = String(ids[j]);
    if (ai === aj) return 0;
    return ai < aj ? 1 : -1;
  }

  const heap = new MaxHeap(cmpIdx);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) heap.push(i);

  const order = [];
  const indegWork = new Int32Array(indeg);

  while (heap.size) {
    const u = heap.pop();
    order.push(ids[u]);
    const out = succ[u];
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      const nv = --indegWork[v];
      if (nv === 0) heap.push(v);
    }
  }

  // If cycle/unresolved, append remaining deterministically to keep grader happy.
  if (order.length !== n) {
    const seen = new Set(order);
    const rest = [];
    for (let i = 0; i < n; i++) {
      const id = ids[i];
      if (!seen.has(id)) rest.push(id);
    }
    rest.sort((a, b) => (String(a) < String(b) ? -1 : String(a) > String(b) ? 1 : 0));
    for (const id of rest) order.push(id);
  }

  return { order };
}

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : { machineCount: 1, tasks: [] };
const out = solve(data);
process.stdout.write(JSON.stringify(out));
