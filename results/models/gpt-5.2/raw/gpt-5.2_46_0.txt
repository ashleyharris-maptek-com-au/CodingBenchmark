#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    uint baseI = i * 2u;
    vec4 posMassI = inData[baseI + 0u];
    vec3 posI = posMassI.xyz;
    float massI = posMassI.w;
    vec3 velI = inData[baseI + 1u].xyz;

    vec3 acc = vec3(0.0);
    float eps2 = softening * softening;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint baseJ = j * 2u;
        vec4 posMassJ = inData[baseJ + 0u];
        vec3 r = posMassJ.xyz - posI;
        float mj = posMassJ.w;

        float r2 = dot(r, r) + eps2;
        float invR = inversesqrt(r2);
        float invR3 = invR * invR * invR;

        acc += (G * mj) * r * invR3;
    }

    // Semi-implicit Euler (often referred to as Euler/leapfrog in simple real-time N-body demos):
    // v(t+dt) = v(t) + a(t) * dt
    // x(t+dt) = x(t) + v(t+dt) * dt
    velI += acc * dt;
    posI += velI * dt;

    outData[baseI + 0u] = vec4(posI, massI);
    outData[baseI + 1u] = vec4(velI, 0.0);
}
