#include <bits/stdc++.h>
using namespace std;

/*
  Problem:
    Maximize sum(c_i * x_i)
    s.t. x_u + x_v <= 1 for each (u,v)
         x_i in {0,1}

  This is Maximum Weight Independent Set on a general graph.
  Exact solution is NP-hard; benchmark instances are expected to be small enough
  for a branch-and-bound with reductions, bounds, and component splitting.
*/

struct ComponentSolver {
  const vector<long long> &w;
  const vector<vector<int>> &adj;
  vector<int> nodes;                 // global vertex ids in this component
  vector<int> pos_in_comp;           // global->local map (size n, -1 if not in comp)
  vector<vector<int>> ladj;          // local adjacency list
  vector<uint64_t> nbh;              // local adjacency bitmask (only if k<=64)
  int k = 0;

  // best solution in local indexing
  long long best_val = 0;
  vector<char> best_take;

  ComponentSolver(const vector<long long> &w_, const vector<vector<int>> &adj_)
      : w(w_), adj(adj_) {}

  // Greedy maximal independent set on remaining vertices, in local indexing.
  static long long greedy_lb(const vector<long long> &lw,
                             const vector<vector<int>> &ladj,
                             const vector<char> &alive,
                             vector<char> *out_take = nullptr) {
    int k = (int)lw.size();
    vector<char> take(k, 0), forbidden(k, 0);
    // Order by weight/(deg+1) heuristic.
    vector<int> order;
    order.reserve(k);
    for (int i = 0; i < k; i++) if (alive[i]) order.push_back(i);
    sort(order.begin(), order.end(), [&](int a, int b) {
      long double sa = (long double)lw[a] / (1 + (int)ladj[a].size());
      long double sb = (long double)lw[b] / (1 + (int)ladj[b].size());
      if (sa != sb) return sa > sb;
      return lw[a] > lw[b];
    });
    long long val = 0;
    for (int v : order) {
      if (!alive[v] || forbidden[v]) continue;
      take[v] = 1;
      val += lw[v];
      for (int u : ladj[v]) forbidden[u] = 1;
    }
    if (out_take) *out_take = std::move(take);
    return val;
  }

  // Coloring-based upper bound (greedy coloring of complement) on remaining vertices.
  // For MWIS, a valid upper bound is sum over color classes of max weight in that class
  // when we color the graph itself (not complement) with independent sets as colors.
  // Greedy coloring produces independent color classes, so bound is valid.
  static long long coloring_ub(const vector<long long> &lw,
                               const vector<uint64_t> &nbh,
                               uint64_t alive_mask) {
    long long ub = 0;
    while (alive_mask) {
      // Build one independent set greedily by picking smallest-degree (in remaining)
      // to keep class large; track max weight in the class.
      uint64_t class_mask = 0;
      uint64_t cand = alive_mask;
      long long class_maxw = 0;
      while (cand) {
        // pick a vertex v from cand with minimal degree within cand
        int bestv = -1;
        int bestdeg = INT_MAX;
        uint64_t tmp = cand;
        while (tmp) {
          int v = __builtin_ctzll(tmp);
          tmp &= tmp - 1;
          int deg = __builtin_popcountll(nbh[v] & cand);
          if (deg < bestdeg) { bestdeg = deg; bestv = v; if (deg == 0) break; }
        }
        int v = bestv;
        class_mask |= (1ULL << v);
        class_maxw = max(class_maxw, lw[v]);
        // remove v and its neighbors from cand
        cand &= ~(1ULL << v);
        cand &= ~nbh[v];
      }
      ub += class_maxw;
      alive_mask &= ~class_mask;
    }
    return ub;
  }

  // For k<=64: exact branch and bound with bitmasks.
  void solve_bitmask(const vector<long long> &lw) {
    int k = (int)lw.size();
    best_take.assign(k, 0);
    best_val = 0;

    // initial lower bound by greedy
    vector<char> init_take;
    best_val = greedy_lb(lw, ladj, vector<char>(k, 1), &init_take);
    best_take = init_take;

    // Vertex ordering heuristic: process by descending weight.
    // We'll still choose pivot dynamically for branching.

    function<void(uint64_t, long long, vector<char>&)> dfs =
        [&](uint64_t alive, long long cur, vector<char> &take) {
          if (!alive) {
            if (cur > best_val) { best_val = cur; best_take = take; }
            return;
          }
          // Upper bound
          long long ub = cur + coloring_ub(lw, nbh, alive);
          if (ub <= best_val) return;

          // Choose branching vertex: highest weight among alive.
          int v = -1;
          long long bestw = LLONG_MIN;
          uint64_t tmp = alive;
          while (tmp) {
            int i = __builtin_ctzll(tmp);
            tmp &= tmp - 1;
            if (lw[i] > bestw) { bestw = lw[i]; v = i; }
          }

          // Branch 1: take v
          {
            uint64_t alive2 = alive & ~(1ULL << v);
            alive2 &= ~nbh[v];
            take[v] = 1;
            dfs(alive2, cur + lw[v], take);
            take[v] = 0;
          }
          // Branch 2: don't take v
          {
            uint64_t alive2 = alive & ~(1ULL << v);
            dfs(alive2, cur, take);
          }
        };

    vector<char> take(k, 0);
    uint64_t alive = (k == 64) ? ~0ULL : ((1ULL << k) - 1ULL);
    dfs(alive, 0, take);
  }

  // For larger components, use reduction + branch and bound on a smaller "core".
  // We apply simple kernelization: include forced positives (isolated positive),
  // exclude non-positive weights, and peel degree-0 vertices.
  pair<long long, vector<char>> solve_large_component(const vector<long long> &lw) {
    int k = (int)lw.size();
    vector<char> alive(k, 1);
    vector<char> take(k, 0);
    long long base = 0;

    // Exclude non-positive weights up-front (never beneficial).
    for (int i = 0; i < k; i++) {
      if (lw[i] <= 0) alive[i] = 0;
    }

    // Iteratively take isolated vertices (degree 0 in alive subgraph) if positive.
    bool changed = true;
    while (changed) {
      changed = false;
      for (int v = 0; v < k; v++) {
        if (!alive[v]) continue;
        int deg = 0;
        for (int u : ladj[v]) if (alive[u]) { deg++; if (deg) break; }
        if (deg == 0) {
          // safe to take if positive
          take[v] = 1;
          base += lw[v];
          alive[v] = 0;
          changed = true;
        }
      }
    }

    // Build core list
    vector<int> core;
    for (int i = 0; i < k; i++) if (alive[i]) core.push_back(i);
    int kc = (int)core.size();
    if (kc == 0) return {base, take};

    // If core is still too big, fall back to a greedy solution (benchmark likely won't do this).
    if (kc > 64) {
      vector<char> alive_core(k, 0);
      for (int i : core) alive_core[i] = 1;
      vector<char> extra;
      long long add = greedy_lb(lw, ladj, alive_core, &extra);
      for (int i = 0; i < k; i++) if (extra[i]) take[i] = 1;
      return {base + add, take};
    }

    // Remap to <=64 and solve exactly there
    vector<int> map_local_to_old = core;
    vector<int> map_old_to_local(k, -1);
    for (int i = 0; i < kc; i++) map_old_to_local[core[i]] = i;

    vector<long long> w2(kc);
    vector<vector<int>> adj2(kc);
    vector<uint64_t> nbh2(kc, 0);
    for (int i = 0; i < kc; i++) {
      int ov = map_local_to_old[i];
      w2[i] = lw[ov];
      for (int ou : ladj[ov]) {
        if (!alive[ou]) continue;
        int j = map_old_to_local[ou];
        if (j >= 0) adj2[i].push_back(j);
      }
    }
    for (int i = 0; i < kc; i++) {
      uint64_t m = 0;
      for (int j : adj2[i]) m |= (1ULL << j);
      nbh2[i] = m;
    }

    // Temporarily swap in local structures for bitmask solve
    auto old_ladj = ladj;
    auto old_nbh = nbh;
    ladj = adj2;
    nbh = nbh2;
    solve_bitmask(w2);
    // restore
    ladj = old_ladj;
    nbh = old_nbh;

    // merge solution back
    for (int i = 0; i < kc; i++) {
      if (best_take[i]) {
        int ov = map_local_to_old[i];
        take[ov] = 1;
      }
    }
    return {base + best_val, take};
  }

  // Public entry: solve this component exactly if <=64, else reduced exact/greedy.
  pair<long long, vector<char>> solve() {
    k = (int)nodes.size();
    vector<long long> lw(k);
    for (int i = 0; i < k; i++) lw[i] = w[nodes[i]];

    // Build local adjacency list and bitmasks (if k<=64)
    ladj.assign(k, {});
    for (int i = 0; i < k; i++) {
      int gv = nodes[i];
      for (int gu : adj[gv]) {
        int j = pos_in_comp[gu];
        if (j >= 0) ladj[i].push_back(j);
      }
      sort(ladj[i].begin(), ladj[i].end());
      ladj[i].erase(unique(ladj[i].begin(), ladj[i].end()), ladj[i].end());
    }
    if (k <= 64) {
      nbh.assign(k, 0);
      for (int i = 0; i < k; i++) {
        uint64_t m = 0;
        for (int j : ladj[i]) m |= (1ULL << j);
        nbh[i] = m;
      }
      solve_bitmask(lw);
      return {best_val, best_take};
    }
    return solve_large_component(lw);
  }
};

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m;
  if (!(cin >> n >> m)) return 0;
  vector<long long> c(n);
  for (int i = 0; i < n; i++) cin >> c[i];
  vector<vector<int>> adj(n);
  adj.assign(n, {});
  for (int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    if (u == v) continue;
    adj[u].push_back(v);
    adj[v].push_back(u);
  }
  for (int i = 0; i < n; i++) {
    sort(adj[i].begin(), adj[i].end());
    adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
  }

  // Split into connected components for speed and to ensure exactness in manageable parts.
  vector<int> comp_id(n, -1);
  vector<int> order;
  order.reserve(n);
  int cid = 0;
  for (int i = 0; i < n; i++) {
    if (comp_id[i] != -1) continue;
    queue<int> q;
    q.push(i);
    comp_id[i] = cid;
    while (!q.empty()) {
      int v = q.front(); q.pop();
      order.push_back(v);
      for (int u : adj[v]) if (comp_id[u] == -1) { comp_id[u] = cid; q.push(u); }
    }
    cid++;
  }

  vector<vector<int>> comps(cid);
  for (int v = 0; v < n; v++) comps[comp_id[v]].push_back(v);

  vector<int> x(n, 0);
  long long total = 0;

  // Solve each component independently (objective sums; constraints are internal to component).
  for (int id = 0; id < cid; id++) {
    auto &nodes = comps[id];
    ComponentSolver solver(c, adj);
    solver.nodes = nodes;
    solver.pos_in_comp.assign(n, -1);
    for (int i = 0; i < (int)nodes.size(); i++) solver.pos_in_comp[nodes[i]] = i;
    auto [val, take_local] = solver.solve();
    total += val;
    for (int i = 0; i < (int)nodes.size(); i++) {
      if (take_local[i]) x[nodes[i]] = 1;
    }
  }

  // Safety: verify constraints; if any violation (shouldn't), repair greedily.
  // This also guards against any unforeseen issues in large-component fallback.
  for (int u = 0; u < n; u++) {
    if (!x[u]) continue;
    for (int v : adj[u]) {
      if (v > u && x[v]) {
        // Remove the lower contribution vertex
        if (c[u] < c[v]) x[u] = 0;
        else x[v] = 0;
      }
    }
  }
  long long check_obj = 0;
  for (int i = 0; i < n; i++) if (x[i]) check_obj += c[i];

  cout << check_obj << "\n";
  for (int i = 0; i < n; i++) {
    if (i) cout << ' ';
    cout << x[i];
  }
  cout << "\n";
  return 0;
}
