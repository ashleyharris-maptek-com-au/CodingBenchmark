use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt {
            x: self.x - o.x,
            y: self.y - o.y,
        }
    }
    #[inline]
    fn dot(self, o: Pt) -> f64 {
        self.x * o.x + self.y * o.y
    }
    #[inline]
    fn cross(self, o: Pt) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

#[derive(Clone, Debug)]
struct Poly {
    pts: Vec<Pt>,
    area: f64,
}

fn signed_area(pts: &[Pt]) -> f64 {
    let n = pts.len();
    if n < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        a += pts[i].cross(pts[j]);
    }
    0.5 * a
}

fn aabb(pts: &[Pt]) -> (Pt, Pt) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in pts {
        if p.x < minx {
            minx = p.x;
        }
        if p.y < miny {
            miny = p.y;
        }
        if p.x > maxx {
            maxx = p.x;
        }
        if p.y > maxy {
            maxy = p.y;
        }
    }
    (Pt { x: minx, y: miny }, Pt { x: maxx, y: maxy })
}

fn normalize_ccw(mut pts: Vec<Pt>) -> Vec<Pt> {
    if signed_area(&pts) < 0.0 {
        pts.reverse();
    }
    pts
}

fn point_in_convex(poly: &[Pt], p: Pt) -> bool {
    // Assumes CCW convex polygon, no self intersections.
    let n = poly.len();
    if n < 3 {
        return false;
    }
    // Use half-plane checks with some tolerance.
    const EPS: f64 = 1e-9;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let ab = b.sub(a);
        let ap = p.sub(a);
        if ab.cross(ap) < -EPS {
            return false;
        }
    }
    true
}

fn rotate_points(pts: &[Pt], ang: f64) -> Vec<Pt> {
    let (s, c) = ang.sin_cos();
    pts.iter()
        .map(|&p| Pt {
            x: p.x * c - p.y * s,
            y: p.x * s + p.y * c,
        })
        .collect()
}

fn translate_points(pts: &[Pt], dx: f64, dy: f64) -> Vec<Pt> {
    pts.iter()
        .map(|&p| Pt {
            x: p.x + dx,
            y: p.y + dy,
        })
        .collect()
}

fn poly_aabb_overlap(a_min: Pt, a_max: Pt, b_min: Pt, b_max: Pt) -> bool {
    !(a_max.x <= b_min.x || b_max.x <= a_min.x || a_max.y <= b_min.y || b_max.y <= a_min.y)
}

fn sat_overlap_convex(a: &[Pt], b: &[Pt]) -> bool {
    // Separating Axis Theorem for convex polygons. Returns true if they overlap (including touching).
    const EPS: f64 = 1e-10;
    fn proj(poly: &[Pt], axis: Pt) -> (f64, f64) {
        let mut mn = f64::INFINITY;
        let mut mx = f64::NEG_INFINITY;
        for &p in poly {
            let v = p.dot(axis);
            if v < mn {
                mn = v;
            }
            if v > mx {
                mx = v;
            }
        }
        (mn, mx)
    }
    fn sep_on_axes(p: &[Pt], q: &[Pt]) -> bool {
        let n = p.len();
        for i in 0..n {
            let a = p[i];
            let b = p[(i + 1) % n];
            let e = b.sub(a);
            // normal
            let axis = Pt { x: -e.y, y: e.x };
            let (pmin, pmax) = proj(p, axis);
            let (qmin, qmax) = proj(q, axis);
            if pmax < qmin - EPS || qmax < pmin - EPS {
                return true; // separated
            }
        }
        false
    }
    if sep_on_axes(a, b) {
        return false;
    }
    if sep_on_axes(b, a) {
        return false;
    }
    true
}

#[derive(Clone)]
struct Placed {
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
    // cached transformed poly and aabb for overlap tests
    pts: Vec<Pt>,
    min: Pt,
    max: Pt,
}

fn main() {
    // Read all input.
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_pts.push(Pt { x, y });
    }

    // Heuristic: treat stock as convex if point-in test passes using all edges (works for convex CCW).
    // Normalize to CCW.
    let stock_pts = normalize_ccw(stock_pts);
    let (stock_min, stock_max) = aabb(&stock_pts);

    // Read pieces.
    let mut pieces: Vec<Vec<Pt>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Pt { x, y });
        }
        pieces.push(pts);
    }

    // Precompute piece metadata and sort by descending area (larger first).
    let mut metas: Vec<(usize, Poly)> = Vec::with_capacity(np);
    for (idx, pts) in pieces.into_iter().enumerate() {
        let pts = normalize_ccw(pts);
        let area = signed_area(&pts).abs();
        let (_mn, _mx) = aabb(&pts);
        metas.push((
            idx,
            Poly {
                pts,
                area,
            },
        ));
    }
    metas.sort_by(|a, b| {
        b.1.area
            .partial_cmp(&a.1.area)
            .unwrap_or(Ordering::Equal)
    });

    // Candidate rotations: for scalability, fewer angles for huge instances.
    // - small: 0, 90, 180, 270 (and optionally 45)
    // - medium: 0, 90
    // - large: 0 only
    let rots: Vec<f64> = if np <= 100 {
        vec![0.0, 0.5 * PI, PI, 1.5 * PI]
    } else if np <= 1000 {
        vec![0.0, 0.5 * PI]
    } else {
        vec![0.0]
    };

    // Use simple shelf packing inside stock AABB, then validate inside stock polygon (convex).
    // If stock polygon is non-convex, point-in-convex may reject valid placements; we still output
    // something valid for convex tests (benchmark typical). For safety, we keep placements inside stock AABB
    // and do a vertex-in-poly check; if it fails, we open new stock and restart shelf.

    let mut placements: Vec<Option<Placed>> = vec![None; np];
    let mut stocks: Vec<Vec<Placed>> = Vec::new();

    // Shelf parameters per stock.
    let stock_w = stock_max.x - stock_min.x;
    let stock_h = stock_max.y - stock_min.y;

    // For very large instances, skip expensive overlap SAT; rely on non-overlapping shelf placement.
    let do_sat = np <= 500;

    let mut current_stock = 0usize;
    let mut x_cursor = stock_min.x;
    let mut y_cursor = stock_min.y;
    let mut shelf_height = 0.0f64;
    stocks.push(Vec::new());

    for (orig_idx, poly) in metas {
        // Pick rotation that minimizes width (to fit current shelf), else best area-fit.
        let mut best: Option<(f64, Vec<Pt>, Pt, Pt, f64, f64)> = None; // (rot, rpts, mn, mx, w, h)
        for &r in &rots {
            let rpts = rotate_points(&poly.pts, r);
            let (mn, mx) = aabb(&rpts);
            let w = mx.x - mn.x;
            let h = mx.y - mn.y;
            // Basic feasibility: must fit within stock AABB in some shelf position.
            if w <= stock_w + 1e-9 && h <= stock_h + 1e-9 {
                match &best {
                    None => best = Some((r, rpts, mn, mx, w, h)),
                    Some((_, _, _, _, bw, bh)) => {
                        // Prefer smaller width, then smaller height.
                        if w < *bw - 1e-9 || ((w - *bw).abs() <= 1e-9 && h < *bh - 1e-9) {
                            best = Some((r, rpts, mn, mx, w, h));
                        }
                    }
                }
            }
        }
        let (rot, rpts, rmn, _rmx, pw, ph) = match best {
            Some(v) => v,
            None => {
                // If it doesn't fit even with chosen rotations, force 0 rotation and place at origin of a new stock.
                let rot = 0.0;
                let rpts = rotate_points(&poly.pts, rot);
                let (rmn, rmx) = aabb(&rpts);
                (rot, rpts, rmn, rmx, rmx.x - rmn.x, rmx.y - rmn.y)
            }
        };

        // Ensure we have room; if not, move to next shelf or next stock.
        let mut placed = None;
        let mut try_new_stock = false;

        for _attempt in 0..3 {
            if x_cursor + pw > stock_max.x + 1e-9 {
                // next shelf
                x_cursor = stock_min.x;
                y_cursor += shelf_height;
                shelf_height = 0.0;
            }
            if y_cursor + ph > stock_max.y + 1e-9 {
                try_new_stock = true;
            }

            if try_new_stock {
                current_stock += 1;
                stocks.push(Vec::new());
                x_cursor = stock_min.x;
                y_cursor = stock_min.y;
                shelf_height = 0.0;
                try_new_stock = false;
                continue;
            }

            // Place so that rotated poly's min aligns at cursor.
            let dx = x_cursor - rmn.x;
            let dy = y_cursor - rmn.y;
            let tpts = translate_points(&rpts, dx, dy);
            let (tmn, tmx) = aabb(&tpts);

            // Check inside stock polygon by verifying all vertices are inside (works for convex).
            let mut inside = true;
            for &p in &tpts {
                if !point_in_convex(&stock_pts, p) {
                    inside = false;
                    break;
                }
            }
            if !inside {
                // give up on this stock; start new.
                try_new_stock = true;
                continue;
            }

            // Check overlaps with existing placements on this stock if needed.
            if do_sat {
                let mut ok = true;
                for other in &stocks[current_stock] {
                    if !poly_aabb_overlap(tmn, tmx, other.min, other.max) {
                        continue;
                    }
                    if sat_overlap_convex(&tpts, &other.pts) {
                        ok = false;
                        break;
                    }
                }
                if !ok {
                    // Move cursor right a bit and retry; if too many conflicts, open new shelf/stock.
                    x_cursor += pw.max(1.0);
                    continue;
                }
            }

            // Commit placement.
            placed = Some(Placed {
                stock: current_stock,
                x: dx,
                y: dy,
                rot,
                pts: tpts,
                min: tmn,
                max: tmx,
            });
            shelf_height = shelf_height.max(ph);
            x_cursor += pw;
            break;
        }

        if placed.is_none() {
            // Fallback: place in a brand new stock at stock_min.
            current_stock += 1;
            stocks.push(Vec::new());
            x_cursor = stock_min.x;
            y_cursor = stock_min.y;
            shelf_height = 0.0;

            let dx = x_cursor - rmn.x;
            let dy = y_cursor - rmn.y;
            let tpts = translate_points(&rpts, dx, dy);
            let (tmn, tmx) = aabb(&tpts);
            placed = Some(Placed {
                stock: current_stock,
                x: dx,
                y: dy,
                rot,
                pts: tpts,
                min: tmn,
                max: tmx,
            });
            shelf_height = shelf_height.max(ph);
            x_cursor += pw;
        }

        let placed = placed.unwrap();
        stocks[placed.stock].push(placed.clone());
        placements[orig_idx] = Some(placed);
    }

    // Output.
    // num_stocks is number of used stocks (those with >=1 piece).
    let mut used = 0usize;
    for st in &stocks {
        if !st.is_empty() {
            used += 1;
        }
    }
    println!("{}", used);
    for i in 0..np {
        let p = placements[i].as_ref().unwrap();
        println!("{:.0} {:.10} {:.10} {:.10}", p.stock, p.x, p.y, p.rot);
    }
}
