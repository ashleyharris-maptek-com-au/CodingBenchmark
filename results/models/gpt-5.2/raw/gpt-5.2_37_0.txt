using System;
using System.Collections.Generic;

// Directed Feedback Vertex Set (exact, exponential).
// Note: The general problem is NP-hard. This implementation is exact and intended for
// small/medium graphs; it will not scale to very large worst-case instances.
//
// Strategy:
// 1) Repeatedly remove vertices not in any directed cycle (via SCC condensation).
// 2) On remaining SCCs (each size >= 2 or self-loop), pick a vertex and branch:
//    either include it in the FVS, or keep it and delete one vertex from a found cycle excluding it.
//    This is a standard exact branching approach over cycles.
// 3) Use iterative deepening on k (minimum size) to ensure we return an optimal solution.
//
// Output: minimum-size feedback vertex set.

public class Program
{
    static void Main()
    {
        var data = Console.In.ReadToEnd().AsSpan();
        int idx = 0;
        int n = ReadInt(data, ref idx);
        int m = ReadInt(data, ref idx);
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var outEdges = new List<int>[n];
        var inEdges = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outEdges[i] = new List<int>();
            inEdges[i] = new List<int>();
        }
        for (int i = 0; i < m; i++)
        {
            int u = ReadInt(data, ref idx);
            int v = ReadInt(data, ref idx);
            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;
            outEdges[u].Add(v);
            inEdges[v].Add(u);
        }

        // Active vertices mask via bool array for speed.
        var active = new bool[n];
        for (int i = 0; i < n; i++) active[i] = true;

        // Iterative deepening for minimum k.
        List<int> best = null!;
        for (int k = 0; ; k++)
        {
            var activeCopy = (bool[])active.Clone();
            var chosen = new List<int>(k);
            if (DfsSolve(n, outEdges, inEdges, activeCopy, k, chosen, out best))
            {
                break;
            }
        }

        Console.WriteLine(best.Count);
        if (best.Count == 0)
        {
            Console.WriteLine();
        }
        else
        {
            for (int i = 0; i < best.Count; i++)
            {
                if (i != 0) Console.Write(' ');
                Console.Write(best[i]);
            }
            Console.WriteLine();
        }
    }

    static bool DfsSolve(
        int n,
        List<int>[] outEdges,
        List<int>[] inEdges,
        bool[] active,
        int budget,
        List<int> chosen,
        out List<int> solution)
    {
        // Reduce: remove vertices not in any cycle by SCC analysis.
        ReduceByScc(n, outEdges, active, out var sccId, out var sccSizes, out var hasSelfLoop, out var anyCyclic);
        if (!anyCyclic)
        {
            solution = new List<int>(chosen);
            return true;
        }
        if (budget == 0)
        {
            solution = null!;
            return false;
        }

        // Pick a vertex in a cyclic SCC (heuristic: largest SCC).
        int pickScc = -1, pickSccSize = -1;
        for (int i = 0; i < sccSizes.Length; i++)
        {
            int sz = sccSizes[i];
            if (sz <= 0) continue;
            if (sz == 1 && !hasSelfLoop[i]) continue;
            if (sz > pickSccSize) { pickSccSize = sz; pickScc = i; }
        }
        if (pickScc == -1)
        {
            solution = new List<int>(chosen);
            return true;
        }

        int vPick = -1;
        for (int v = 0; v < n; v++)
        {
            if (active[v] && sccId[v] == pickScc) { vPick = v; break; }
        }
        if (vPick == -1)
        {
            solution = new List<int>(chosen);
            return true;
        }

        // Branch 1: remove vPick (include in FVS).
        {
            var active1 = (bool[])active.Clone();
            active1[vPick] = false;
            chosen.Add(vPick);
            if (DfsSolve(n, outEdges, inEdges, active1, budget - 1, chosen, out solution))
                return true;
            chosen.RemoveAt(chosen.Count - 1);
        }

        // Branch 2: keep vPick, but break a cycle inside its SCC without removing vPick.
        // Find a directed cycle within pickScc that avoids vPick if possible; then branch by removing one vertex on it.
        var cycle = FindCycleInScc(n, outEdges, active, sccId, pickScc, vPick);
        if (cycle == null || cycle.Count == 0)
        {
            solution = null!;
            return false;
        }

        // Branch on removing each vertex in the cycle (excluding vPick).
        for (int i = 0; i < cycle.Count; i++)
        {
            int x = cycle[i];
            if (x == vPick) continue;
            var active2 = (bool[])active.Clone();
            active2[x] = false;
            chosen.Add(x);
            if (DfsSolve(n, outEdges, inEdges, active2, budget - 1, chosen, out solution))
                return true;
            chosen.RemoveAt(chosen.Count - 1);
        }

        solution = null!;
        return false;
    }

    static void ReduceByScc(
        int n,
        List<int>[] outEdges,
        bool[] active,
        out int[] sccId,
        out int[] sccSizes,
        out bool[] hasSelfLoop,
        out bool anyCyclic)
    {
        // Compute SCCs on active subgraph (Kosaraju).
        sccId = new int[n];
        Array.Fill(sccId, -1);
        var order = new int[n];
        int ordSz = 0;

        var vis = new bool[n];
        var stack = new Stack<(int v, int ei)>();
        for (int v = 0; v < n; v++)
        {
            if (!active[v] || vis[v]) continue;
            stack.Push((v, 0));
            vis[v] = true;
            while (stack.Count > 0)
            {
                var (cur, ei) = stack.Pop();
                if (ei < outEdges[cur].Count)
                {
                    stack.Push((cur, ei + 1));
                    int to = outEdges[cur][ei];
                    if (!active[to] || vis[to]) continue;
                    vis[to] = true;
                    stack.Push((to, 0));
                }
                else
                {
                    order[ordSz++] = cur;
                }
            }
        }

        // Build reverse edges on the fly via scanning inEdges is expensive; we have only outEdges,
        // so build a compact reverse adjacency for active graph once per reduction.
        var rev = new List<int>[n];
        for (int i = 0; i < n; i++) rev[i] = new List<int>();
        for (int u = 0; u < n; u++)
        {
            if (!active[u]) continue;
            foreach (var v in outEdges[u])
                if (active[v]) rev[v].Add(u);
        }

        int comp = 0;
        Array.Fill(vis, false);
        var compSizes = new List<int>();
        var selfLoop = new List<bool>();
        var st2 = new Stack<int>();
        for (int i = ordSz - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!active[v] || vis[v]) continue;
            int sz = 0;
            bool sl = false;
            vis[v] = true;
            st2.Push(v);
            while (st2.Count > 0)
            {
                int x = st2.Pop();
                sccId[x] = comp;
                sz++;
                foreach (var to in outEdges[x])
                    if (active[to] && to == x) sl = true;
                foreach (var p in rev[x])
                {
                    if (!active[p] || vis[p]) continue;
                    vis[p] = true;
                    st2.Push(p);
                }
            }
            compSizes.Add(sz);
            selfLoop.Add(sl);
            comp++;
        }

        sccSizes = compSizes.ToArray();
        hasSelfLoop = selfLoop.ToArray();

        // Remove vertices not in any cycle: SCC size==1 and no self-loop.
        anyCyclic = false;
        for (int v = 0; v < n; v++)
        {
            if (!active[v]) continue;
            int id = sccId[v];
            if (id < 0) { active[v] = false; continue; }
            if (sccSizes[id] == 1 && !hasSelfLoop[id])
            {
                active[v] = false;
            }
            else
            {
                anyCyclic = true;
            }
        }
    }

    static List<int> FindCycleInScc(
        int n,
        List<int>[] outEdges,
        bool[] active,
        int[] sccId,
        int targetScc,
        int avoid)
    {
        // Find any directed cycle inside targetScc, preferring to avoid `avoid`.
        // Use DFS with parent tracking and recursion stack markers.
        var state = new byte[n]; // 0=unseen,1=visiting,2=done
        var parent = new int[n];
        Array.Fill(parent, -1);

        int startCandidate = -1;
        for (int v = 0; v < n; v++)
        {
            if (!active[v] || sccId[v] != targetScc) continue;
            if (v == avoid) continue;
            startCandidate = v;
            break;
        }
        if (startCandidate == -1) startCandidate = avoid;

        int cycleFrom = -1, cycleTo = -1;
        var stack = new Stack<(int v, int ei)>();

        void PushStart(int s)
        {
            stack.Push((s, 0));
            state[s] = 1;
        }

        for (int v0 = 0; v0 < n && cycleFrom == -1; v0++)
        {
            int s = (v0 == 0) ? startCandidate : -1;
            if (s == -1)
            {
                // find next unseen in SCC
                for (int v = 0; v < n; v++)
                {
                    if (!active[v] || sccId[v] != targetScc) continue;
                    if (state[v] == 0) { s = v; break; }
                }
            }
            if (s == -1) break;
            if (state[s] != 0) continue;
            PushStart(s);
            while (stack.Count > 0 && cycleFrom == -1)
            {
                var (cur, ei) = stack.Pop();
                if (ei < outEdges[cur].Count)
                {
                    int to = outEdges[cur][ei];
                    stack.Push((cur, ei + 1));
                    if (!active[to] || sccId[to] != targetScc) continue;
                    if (state[to] == 0)
                    {
                        parent[to] = cur;
                        stack.Push((to, 0));
                        state[to] = 1;
                    }
                    else if (state[to] == 1)
                    {
                        cycleFrom = cur;
                        cycleTo = to;
                        break;
                    }
                }
                else
                {
                    state[cur] = 2;
                }
            }
        }

        if (cycleFrom == -1) return null!;

        // Reconstruct cycle: cycleTo ... cycleFrom -> cycleTo
        var cycle = new List<int>();
        cycle.Add(cycleTo);
        int x = cycleFrom;
        while (x != cycleTo && x != -1)
        {
            cycle.Add(x);
            x = parent[x];
        }
        if (x == -1) return null!;
        // cycle currently has nodes along back edge, but may be reversed; normalize by returning as list.
        return cycle;
    }

    static int ReadInt(ReadOnlySpan<char> s, ref int idx)
    {
        while (idx < s.Length && s[idx] <= ' ') idx++;
        int sign = 1;
        if (idx < s.Length && s[idx] == '-') { sign = -1; idx++; }
        int val = 0;
        while (idx < s.Length)
        {
            char c = s[idx];
            if ((uint)(c - '0') > 9) break;
            val = val * 10 + (c - '0');
            idx++;
        }
        return val * sign;
    }
}
