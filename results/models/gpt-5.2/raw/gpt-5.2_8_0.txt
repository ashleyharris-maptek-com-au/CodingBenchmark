use std::cmp::min;
use std::collections::VecDeque;
use std::io::{self, Read};

#[inline(always)]
fn idx(r: usize, c: usize, w: usize) -> usize {
    r * w + c
}

fn main() {
    // Read all stdin at once (fast for large inputs).
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }

    // Parse H W then H lines. Lines may contain spaces; so we parse first line via split_whitespace,
    // then consume the next H raw lines.
    let mut lines = input.lines();
    let first = match lines.next() {
        Some(l) => l,
        None => return,
    };
    let mut it = first.split_whitespace();
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    if h == 0 || w == 0 {
        return;
    }

    // Read grid rows; keep exactly W bytes per row (pad with spaces if line shorter).
    let mut grid: Vec<u8> = vec![b'#'; h * w];
    let mut start: Option<(usize, usize)> = None;
    let mut goal: Option<(usize, usize)> = None;

    for r in 0..h {
        let line = lines.next().unwrap_or("");
        // Using bytes is faster and memory lighter than chars.
        let bytes = line.as_bytes();
        let take = min(bytes.len(), w);
        let row_off = r * w;
        if take > 0 {
            grid[row_off..row_off + take].copy_from_slice(&bytes[..take]);
        }
        if take < w {
            // Remaining already '#'; but missing characters in input should be treated as spaces,
            // not walls. Fill with spaces.
            for c in take..w {
                grid[row_off + c] = b' ';
            }
        }
        // Scan row for A/B (only within width).
        for c in 0..w {
            match grid[row_off + c] {
                b'A' => start = Some((r, c)),
                b'B' => goal = Some((r, c)),
                _ => {}
            }
        }
    }

    let (sr, sc) = match start {
        Some(v) => v,
        None => {
            // No start => no path
            return;
        }
    };
    let (gr, gc) = match goal {
        Some(v) => v,
        None => {
            // No goal => no path
            return;
        }
    };

    let n = h * w;

    // Special-case: start == goal
    if sr == gr && sc == gc {
        println!("1");
        println!("{} {}", sc, sr);
        return;
    }

    // BFS from A to find any shortest path to B.
    // For very large grids, parent + visited are the main memory consumers.
    // parent_dir: 0..4 encodes direction from cell to its parent, 255 = unvisited.
    // Using a u8 array minimizes memory.
    let mut parent_dir: Vec<u8> = vec![255u8; n];
    let mut q: VecDeque<usize> = VecDeque::new();

    let sidx = idx(sr, sc, w);
    let gidx = idx(gr, gc, w);
    parent_dir[sidx] = 254; // sentinel for root
    q.push_back(sidx);

    // Neighbor deltas: (dr, dc, dir_code)
    // dir_code indicates how to go from current cell to its parent when reconstructing:
    // if we move from parent -> child by (dr,dc), then child's parent_dir is opposite.
    const DIRS: [(isize, isize, u8); 4] = [
        (-1, 0, 2), // moved up to reach child; to go back to parent, move down (code 2)
        (1, 0, 0),  // moved down; back up (code 0)
        (0, -1, 3), // moved left; back right (code 3)
        (0, 1, 1),  // moved right; back left (code 1)
    ];
    // Codes map to inverse deltas during reconstruction:
    // 0: up (-1,0), 1: left (0,-1), 2: down (1,0), 3: right (0,1)
    const BACK: [(isize, isize); 4] = [(-1, 0), (0, -1), (1, 0), (0, 1)];

    let mut found = false;
    while let Some(cur) = q.pop_front() {
        if cur == gidx {
            found = true;
            break;
        }
        let r = cur / w;
        let c = cur - r * w;
        for (dr, dc, back_code) in DIRS {
            let nr = r as isize + dr;
            let nc = c as isize + dc;
            if nr < 0 || nc < 0 {
                continue;
            }
            let nr = nr as usize;
            let nc = nc as usize;
            if nr >= h || nc >= w {
                continue;
            }
            let ni = idx(nr, nc, w);
            if parent_dir[ni] != 255 {
                continue;
            }
            let cell = grid[ni];
            if cell == b'#' {
                continue;
            }
            parent_dir[ni] = back_code;
            q.push_back(ni);
        }
    }

    if !found {
        // Output nothing (or 0) if no path exists.
        // We'll output 0 for clarity/robustness.
        println!("0");
        return;
    }

    // Reconstruct path from goal back to start using parent_dir.
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut cur = gidx;
    loop {
        let r = cur / w;
        let c = cur - r * w;
        path.push((c, r)); // output x y (col row)
        if cur == sidx {
            break;
        }
        let code = parent_dir[cur];
        if code == 254 || code == 255 {
            // Shouldn't happen unless corrupted; treat as no path.
            println!("0");
            return;
        }
        let (dr, dc) = BACK[code as usize];
        let pr = r as isize + dr;
        let pc = c as isize + dc;
        cur = idx(pr as usize, pc as usize, w);
    }
    path.reverse();

    println!("{}", path.len());
    for (x, y) in path {
        println!("{} {}", x, y);
    }
}
