// HLSL pixel shader for Vulkan (glslangValidator HLSL mode)

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float shade = 0.3 + 0.7 * diffuse;

    // 6x6 polka-dot pattern in UV space
    float2 gridUV = input.uv * 6.0;
    float2 cell = frac(gridUV);
    float2 d = cell - float2(0.5, 0.5);
    float dist = length(d);

    float radius = 0.3;
    float dotMask = 1.0 - step(radius, dist); // 1 inside circle, 0 outside

    float3 dotColor = float3(1.0, 0.8, 0.0);
    float3 bgColor  = float3(0.1, 0.1, 0.3);
    float3 base = lerp(bgColor, dotColor, dotMask);

    float3 rgb = base * shade;
    return float4(rgb, 1.0);
}
