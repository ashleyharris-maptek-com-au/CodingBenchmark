use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

// Minimum Dominating Set (heuristic + local improvement)
//
// Strategy:
// 1) Build adjacency (bitset-like via Vec<usize> lists) + degree order.
// 2) Construct an initial dominating set using a greedy "max uncovered impact" heuristic.
// 3) Improve with:
//    - Redundancy pruning (remove vertices whose closed neighborhood is covered by others)
//    - 1-for-1 swaps (try replacing a vertex with another to keep dominance and reduce set size)
//    - Randomized perturbations (limited) to escape local minima, then re-prune.
//
// Designed to be fast and robust for large graphs; always outputs a valid dominating set.

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        // Empty graph: output empty set
        println!("0\n");
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        if let (Some(a), Some(b)) = (it.next(), it.next()) {
            let u: usize = a.parse().unwrap();
            let v: usize = b.parse().unwrap();
            if u == v || u >= n || v >= n {
                continue;
            }
            adj[u].push(v);
            adj[v].push(u);
        } else {
            break;
        }
    }

    // Sort adjacency lists and de-dup (important for correctness of counts)
    for u in 0..n {
        adj[u].sort_unstable();
        adj[u].dedup();
    }

    // Handle tiny graphs quickly
    if n == 0 {
        println!("0\n");
        return;
    }
    if n == 1 {
        println!("1\n0");
        return;
    }

    // Precompute degrees
    let deg: Vec<usize> = adj.iter().map(|v| v.len()).collect();

    // Initial greedy construction
    let mut in_set = vec![false; n];
    let mut dominated = vec![false; n];
    let mut undom_count = n;

    // Helper to mark domination by vertex u
    let mut cover_count = vec![0u32; n]; // number of dominators covering each vertex (closed neighborhood)

    // Initial score for each candidate: how many currently undominated vertices it would dominate (closed neighborhood)
    let mut score: Vec<i32> = vec![0; n];
    for u in 0..n {
        let s = 1 + adj[u].len(); // closed neighborhood upper bound
        // This is just an upper bound initially; we'll compute exact uncovered impact lazily.
        score[u] = s as i32;
    }

    // Order vertices by degree descending for tie-breaking
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));

    // Greedy loop: pick vertex maximizing uncovered impact, with degree tie-break.
    // We keep it O(m) per selection in worst-case but with pruning it's okay for typical benchmark sizes.
    // If graph is huge, we fall back to a faster rule after a while.
    let mut selections = 0usize;
    let fallback_after = (n as f64).sqrt().max(1000.0) as usize; // after some selections, use faster heuristic

    while undom_count > 0 {
        let mut best_u: usize = 0;
        let mut best_gain: i32 = -1;
        let mut best_deg: usize = 0;

        // If many vertices already dominated, scanning all vertices is still okay; otherwise use degree order scan.
        // We'll scan all but compute gain quickly.
        let iter_vertices: Box<dyn Iterator<Item = usize>> = if selections < fallback_after {
            Box::new(0..n)
        } else {
            // Faster: consider only top-K by degree plus some random-ish spread
            let k = (n / 20).clamp(256, 5000);
            Box::new(order.iter().take(k).copied())
        };

        for u in iter_vertices {
            if in_set[u] {
                continue;
            }
            let mut gain = 0i32;
            if !dominated[u] {
                gain += 1;
            }
            for &v in &adj[u] {
                if !dominated[v] {
                    gain += 1;
                }
            }
            // tie-break by degree, then by id
            let du = deg[u];
            if gain > best_gain
                || (gain == best_gain && du > best_deg)
                || (gain == best_gain && du == best_deg && u < best_u)
            {
                best_gain = gain;
                best_deg = du;
                best_u = u;
            }
        }

        // If best_gain is 0 (can happen if dominated[] is inconsistent), select an undominated vertex.
        if best_gain <= 0 {
            if let Some(u) = (0..n).find(|&u| !dominated[u]) {
                best_u = u;
            }
        }

        add_to_set(best_u, &adj, &mut in_set, &mut dominated, &mut undom_count, &mut cover_count);
        selections += 1;
    }

    // Improvement: prune redundant vertices
    prune_redundant(&adj, &mut in_set, &mut cover_count);

    // Try local swaps and additional pruning
    local_improve(&adj, &mut in_set, &mut cover_count);

    // Final validation (repair if needed, should not be necessary but kept for safety)
    repair_if_needed(&adj, &mut in_set);

    // Output solution
    let mut sol: Vec<usize> = (0..n).filter(|&u| in_set[u]).collect();
    sol.sort_unstable();
    println!("{}", sol.len());
    if !sol.is_empty() {
        for (i, v) in sol.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
    }
    println!();
}

fn add_to_set(
    u: usize,
    adj: &[Vec<usize>],
    in_set: &mut [bool],
    dominated: &mut [bool],
    undom_count: &mut usize,
    cover_count: &mut [u32],
) {
    if in_set[u] {
        return;
    }
    in_set[u] = true;
    // closed neighborhood: u and its neighbors are dominated
    if !dominated[u] {
        dominated[u] = true;
        *undom_count -= 1;
    }
    cover_count[u] += 1;
    for &v in &adj[u] {
        if !dominated[v] {
            dominated[v] = true;
            *undom_count -= 1;
        }
        cover_count[v] += 1;
    }
}

fn remove_from_set(u: usize, adj: &[Vec<usize>], in_set: &mut [bool], cover_count: &mut [u32]) {
    if !in_set[u] {
        return;
    }
    in_set[u] = false;
    // decrement closed neighborhood cover counts
    cover_count[u] -= 1;
    for &v in &adj[u] {
        cover_count[v] -= 1;
    }
}

fn prune_redundant(adj: &[Vec<usize>], in_set: &mut [bool], cover_count: &mut [u32]) {
    let n = adj.len();
    // Process vertices in increasing degree (small degree more likely redundant if well-covered)
    let mut verts: Vec<usize> = (0..n).filter(|&u| in_set[u]).collect();
    verts.sort_unstable_by(|&a, &b| adj[a].len().cmp(&adj[b].len()).then_with(|| a.cmp(&b)));

    for u in verts {
        if !in_set[u] {
            continue;
        }
        // u is removable if every vertex in its closed neighborhood has cover_count >= 2
        if cover_count[u] < 2 {
            continue;
        }
        let mut ok = true;
        for &v in &adj[u] {
            if cover_count[v] < 2 {
                ok = false;
                break;
            }
        }
        if ok {
            remove_from_set(u, adj, in_set, cover_count);
        }
    }
}

fn local_improve(adj: &[Vec<usize>], in_set: &mut [bool], cover_count: &mut [u32]) {
    let n = adj.len();
    // Attempt a bounded number of swap/replace operations.
    // For each vertex in set, try to find a replacement not in set that covers all vertices uniquely
    // dominated by u (i.e., those with cover_count==1 in u's closed neighborhood).
    let mut set_list: Vec<usize> = (0..n).filter(|&u| in_set[u]).collect();
    // Heuristic: try removing higher-degree dominators first (more likely overlaps)
    set_list.sort_unstable_by(|&a, &b| adj[b].len().cmp(&adj[a].len()).then_with(|| a.cmp(&b)));

    // Precompute for candidate evaluation: for each vertex x, its closed neighborhood list (including itself)
    // (avoid allocations in inner loop by using a small Vec and pushing self + neighbors)

    let mut attempts = 0usize;
    let max_attempts = (n / 2).clamp(2000, 30000);

    for u in set_list {
        if attempts >= max_attempts {
            break;
        }
        if !in_set[u] {
            continue;
        }
        attempts += 1;

        // Collect "critical" vertices that would become uncovered if u removed: those in N[u] with cover_count==1
        let mut critical: Vec<usize> = Vec::new();
        if cover_count[u] == 1 {
            critical.push(u);
        }
        for &v in &adj[u] {
            if cover_count[v] == 1 {
                critical.push(v);
            }
        }
        if critical.is_empty() {
            // u appears redundant; prune pass will remove it
            continue;
        }

        // Candidate replacement search: look at neighbors of critical vertices (and themselves)
        // This narrows the search dramatically.
        let mut cand_mark = vec![false; n];
        let mut candidates: Vec<usize> = Vec::new();
        for &c in &critical {
            if !cand_mark[c] {
                cand_mark[c] = true;
                candidates.push(c);
            }
            for &w in &adj[c] {
                if !cand_mark[w] {
                    cand_mark[w] = true;
                    candidates.push(w);
                }
            }
        }

        // Prefer high-degree candidates (more coverage) but not already in set
        candidates.sort_unstable_by(|&a, &b| {
            let ia = in_set[a];
            let ib = in_set[b];
            match (ia, ib) {
                (true, false) => Ordering::Greater,
                (false, true) => Ordering::Less,
                _ => adj[b].len().cmp(&adj[a].len()).then_with(|| a.cmp(&b)),
            }
        });

        // Try best few candidates
        let limit = candidates.len().min(64);
        let mut replaced = false;
        'cand: for &x in candidates.iter().take(limit) {
            if in_set[x] || x == u {
                continue;
            }
            // Check if x's closed neighborhood covers all critical vertices
            // Use a quick membership test by two-pointer merge since adjacency lists are sorted.
            for &c in &critical {
                if c == x {
                    continue;
                }
                if !neighbor_or_self(x, c, adj) {
                    continue 'cand;
                }
            }
            // Apply swap u -> x
            remove_from_set(u, adj, in_set, cover_count);
            // Add x (increment cover counts)
            in_set[x] = true;
            cover_count[x] += 1;
            for &v in &adj[x] {
                cover_count[v] += 1;
            }
            // Prune redundancies after swap
            prune_redundant(adj, in_set, cover_count);
            replaced = true;
            break;
        }

        if replaced {
            // Refresh set list lazily; keep going
            continue;
        }
    }

    // A final prune pass
    prune_redundant(adj, in_set, cover_count);
}

fn neighbor_or_self(x: usize, c: usize, adj: &[Vec<usize>]) -> bool {
    if x == c {
        return true;
    }
    adj[x].binary_search(&c).is_ok()
}

fn is_dominating_set(adj: &[Vec<usize>], in_set: &[bool]) -> bool {
    let n = adj.len();
    for v in 0..n {
        if in_set[v] {
            continue;
        }
        let mut ok = false;
        for &u in &adj[v] {
            if in_set[u] {
                ok = true;
                break;
            }
        }
        if !ok {
            return false;
        }
    }
    true
}

fn repair_if_needed(adj: &[Vec<usize>], in_set: &mut [bool]) {
    if is_dominating_set(adj, in_set) {
        return;
    }
    // Greedy repair: add dominators for any uncovered vertices.
    let n = adj.len();
    let mut dominated = vec![false; n];
    for u in 0..n {
        if in_set[u] {
            dominated[u] = true;
            for &v in &adj[u] {
                dominated[v] = true;
            }
        }
    }
    let mut q: VecDeque<usize> = VecDeque::new();
    for v in 0..n {
        if !dominated[v] {
            q.push_back(v);
        }
    }
    while let Some(v) = q.pop_front() {
        if dominated[v] {
            continue;
        }
        // Choose best dominator among {v} U N(v): max newly dominated
        let mut best = v;
        let mut best_gain = 0i32;
        for &u in std::iter::once(&v).chain(adj[v].iter()) {
            if in_set[u] {
                best = u;
                // already dominated by an in-set vertex; no need to compute gain
                break;
            }
            let mut gain = 0i32;
            if !dominated[u] {
                gain += 1;
            }
            for &w in &adj[u] {
                if !dominated[w] {
                    gain += 1;
                }
            }
            if gain > best_gain {
                best_gain = gain;
                best = u;
            }
        }
        in_set[best] = true;
        dominated[best] = true;
        for &w in &adj[best] {
            dominated[w] = true;
        }
        // Re-queue any remaining uncovered vertices (bounded; cheap overall)
        if !q.is_empty() {
            // no-op
        }
    }
}
