const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
let data = {};
if (input.length) {
  try {
    data = JSON.parse(input);
  } catch {
    data = {};
  }
}

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const N = width * height;

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

// Passability: '.' burnable; '~' blocks. Anything else treat as blocked.
const burnable = new Uint8Array(N);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    burnable[idx(x, y)] = c === 46 /* '.' */ ? 1 : 0;
  }
}

// Precompute towns-at-cell total value to weight protection.
const townValueAt = new Float64Array(N);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  if (!inBounds(x, y)) continue;
  townValueAt[idx(x, y)] += +t.value || 0;
}

// Multi-source BFS to compute earliest burn times ignoring cuts.
const INF = 1e9;
const dist = new Int32Array(N);
for (let i = 0; i < N; i++) dist[i] = INF;

// Simple array queue for BFS.
const q = new Int32Array(N);
let qh = 0, qt = 0;

for (let k = 0; k < ignitions.length; k++) {
  const p = ignitions[k];
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  if (!burnable[id]) continue;
  if (dist[id] !== 0) {
    dist[id] = 0;
    q[qt++] = id;
  }
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];

while (qh < qt) {
  const v = q[qh++];
  const x = v % width;
  const y = (v / width) | 0;
  const dv = dist[v] + 1;
  for (let d = 0; d < 4; d++) {
    const nx = x + dx[d];
    const ny = y + dy[d];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!burnable[ni]) continue;
    if (dist[ni] > dv) {
      dist[ni] = dv;
      q[qt++] = ni;
    }
  }
}

// If no budget or no reachable fire, output empty.
if (cutBudget === 0 || qt === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Compute an "influence" score for each cell: how much town value flows through it
// along shortest paths from ignitions. This approximates which cuts block valuable areas.
//
// Method:
// 1) Sort reachable cells by dist descending.
// 2) DP backwards: valueAtCell += townValueAt + sum(childContribution / numParents(child))
//    where edges follow dist+1. This distributes each cell's "value" to its predecessors.
//
// Also compute numParents for each cell in shortest-path DAG.
const reachable = new Int32Array(qt);
for (let i = 0; i < qt; i++) reachable[i] = q[i]; // q currently contains BFS order (increasing dist)

// We need reachable sorted by dist descending; we can bucket by distance up to maxDist.
let maxDist = 0;
for (let i = 0; i < qt; i++) {
  const d = dist[reachable[i]];
  if (d < INF && d > maxDist) maxDist = d;
}

// Buckets using counting sort by dist.
const counts = new Int32Array(maxDist + 2);
for (let i = 0; i < qt; i++) {
  const d = dist[reachable[i]];
  if (d < INF) counts[d]++;
}
for (let d = 1; d <= maxDist; d++) counts[d] += counts[d - 1];
const orderAsc = new Int32Array(qt);
for (let i = qt - 1; i >= 0; i--) {
  const v = reachable[i];
  const d = dist[v];
  if (d >= INF) continue;
  orderAsc[--counts[d]] = v;
}

const numParents = new Int16Array(N);
for (let ii = 0; ii < qt; ii++) {
  const v = orderAsc[ii];
  const x = v % width;
  const y = (v / width) | 0;
  const dv = dist[v];
  // Count neighbors with dist = dv - 1
  let c = 0;
  if (dv > 0) {
    for (let d = 0; d < 4; d++) {
      const nx = x + dx[d];
      const ny = y + dy[d];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (dist[ni] === dv - 1) c++;
    }
  }
  numParents[v] = c;
}

const flowValue = new Float64Array(N);
for (let ii = qt - 1; ii >= 0; ii--) {
  const v = orderAsc[ii];
  let vval = townValueAt[v];
  // Add from successors.
  const x = v % width;
  const y = (v / width) | 0;
  const dv = dist[v];
  for (let d = 0; d < 4; d++) {
    const nx = x + dx[d];
    const ny = y + dy[d];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (dist[ni] === dv + 1) {
      const parents = numParents[ni];
      if (parents > 0) vval += flowValue[ni] / parents;
    }
  }
  flowValue[v] = vval;
}

// Candidate cuts: focus on a frontier band near where fire is expanding.
// Prefer cells with high flowValue and that are not too close to ignition (to avoid trivial cuts)
// and not too late (so it can matter). Choose up to ~K candidates and then greedily pick.
const K = Math.min(N, Math.max(2000, cutBudget * 50));
// Estimate an effective time horizon based on town distances (median-ish).
let townDistSum = 0, townCount = 0;
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0, y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  const d = dist[id];
  if (d < INF) {
    townDistSum += d;
    townCount++;
  }
}
const avgTownDist = townCount ? (townDistSum / townCount) : maxDist;
const minBand = Math.max(1, (avgTownDist * 0.25) | 0);
const maxBand = Math.max(minBand + 1, (avgTownDist * 1.1) | 0);

// Build candidate list with a simple top-K selection by score using partial heap.
class MinHeap {
  constructor(cap) { this.a = []; this.cap = cap; }
  push(item) {
    const a = this.a;
    if (a.length < this.cap) { a.push(item); this._up(a.length - 1); return; }
    if (item.score <= a[0].score) return;
    a[0] = item;
    this._down(0);
  }
  _up(i) {
    const a = this.a;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (a[p].score <= a[i].score) break;
      const t = a[p]; a[p] = a[i]; a[i] = t;
      i = p;
    }
  }
  _down(i) {
    const a = this.a;
    const n = a.length;
    while (true) {
      let l = i * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let m = r < n && a[r].score < a[l].score ? r : l;
      if (a[i].score <= a[m].score) break;
      const t = a[i]; a[i] = a[m]; a[m] = t;
      i = m;
    }
  }
}

const heap = new MinHeap(K);
for (let ii = 0; ii < qt; ii++) {
  const v = orderAsc[ii];
  const d = dist[v];
  if (d < minBand || d > maxBand) continue;
  // Don't cut where a town exists (could be allowed, but usually worse); still allow if no choice.
  const tv = townValueAt[v];
  const score = flowValue[v] - tv * 0.5;
  if (score <= 0) continue;
  heap.push({ v, score });
}

let candidates = heap.a;
if (candidates.length === 0) {
  // Fallback: pick highest flowValue anywhere reachable excluding ignitions.
  const heap2 = new MinHeap(K);
  for (let ii = 0; ii < qt; ii++) {
    const v = orderAsc[ii];
    const d = dist[v];
    if (d <= 0) continue;
    const score = flowValue[v];
    if (score <= 0) continue;
    heap2.push({ v, score });
  }
  candidates = heap2.a;
}

// Sort candidates by score descending for greedy selection.
candidates.sort((a, b) => b.score - a.score);

// Greedy pick: ensure cuts are on/near a bottleneck by favoring cells with low alternate bypass.
// We approximate bypass by local degree within reachable burnable set.
function degree(v) {
  const x = v % width;
  const y = (v / width) | 0;
  let deg = 0;
  for (let d = 0; d < 4; d++) {
    const nx = x + dx[d], ny = y + dy[d];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (burnable[ni] && dist[ni] < INF) deg++;
  }
  return deg;
}

const picked = [];
const pickedSet = new Uint8Array(N);
for (let i = 0; i < candidates.length && picked.length < cutBudget; i++) {
  const v = candidates[i].v;
  if (!burnable[v]) continue;
  if (pickedSet[v]) continue;
  // Avoid picking adjacent cuts too densely unless budget is large.
  const x = v % width;
  const y = (v / width) | 0;
  let near = 0;
  for (let d = 0; d < 4; d++) {
    const nx = x + dx[d], ny = y + dy[d];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (pickedSet[ni]) near++;
  }
  if (near >= 2 && picked.length < (cutBudget * 0.7)) continue;

  // Prefer lower degree (more chokepoint-like) but don't over-filter.
  const deg = degree(v);
  if (deg >= 4 && picked.length < (cutBudget * 0.5)) continue;

  pickedSet[v] = 1;
  picked.push([x, y]);
}

// If we still have budget, fill with next best regardless of spacing/degree.
if (picked.length < cutBudget) {
  for (let i = 0; i < candidates.length && picked.length < cutBudget; i++) {
    const v = candidates[i].v;
    if (!burnable[v] || pickedSet[v]) continue;
    pickedSet[v] = 1;
    picked.push([v % width, (v / width) | 0]);
  }
}

process.stdout.write(JSON.stringify({ firebreaks: picked }));
