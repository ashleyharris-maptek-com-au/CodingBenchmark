// Wildfire firebreak placement heuristic solver
// Reads one JSON from stdin and outputs {"firebreaks":[[x,y],...]}.

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Town = { id: string | number; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const gridStrs: string[] = data.grid;
const ignitions: [number, number][] = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const N = width * height;
const idx = (x: number, y: number) => y * width + x;
const inb = (x: number, y: number) => x >= 0 && y >= 0 && x < width && y < height;

const passable = new Uint8Array(N); // 1 for forest '.', 0 otherwise
for (let y = 0; y < height; y++) {
  const row = gridStrs[y];
  for (let x = 0; x < width; x++) {
    passable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
  }
}

const townAt = new Int32Array(N);
townAt.fill(-1);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (inb(t.x, t.y)) townAt[idx(t.x, t.y)] = i;
}

// Multi-source BFS on forest to compute earliest burn time ignoring cuts.
const INF = 1e9;
const burnTime = new Int32Array(N);
for (let i = 0; i < N; i++) burnTime[i] = INF;

const q = new Int32Array(N);
let qh = 0,
  qt = 0;
for (const [x, y] of ignitions) {
  if (!inb(x, y)) continue;
  const p = idx(x, y);
  if (passable[p] === 0) continue;
  if (burnTime[p] > 0) {
    burnTime[p] = 0;
    q[qt++] = p;
  }
}

const dirs = [1, 0, -1, 0, 0, 1, 0, -1];
while (qh < qt) {
  const p = q[qh++];
  const t = burnTime[p] + 1;
  const x = p % width;
  const y = (p / width) | 0;
  for (let k = 0; k < 8; k += 2) {
    const nx = x + dirs[k],
      ny = y + dirs[k + 1];
    if (!inb(nx, ny)) continue;
    const np = idx(nx, ny);
    if (passable[np] === 0) continue;
    if (burnTime[np] > t) {
      burnTime[np] = t;
      q[qt++] = np;
    }
  }
}

// Quick exit if no budget or no burning.
if (cutBudget === 0 || qt === 0 || towns.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Compute connected components of forest cells, and which components burn under no cuts.
const compId = new Int32Array(N);
compId.fill(-1);
const compSize: number[] = [];
const compBurns: number[] = []; // 0/1
const compTownValue: number[] = [];

const stack = new Int32Array(N);
let compCount = 0;
for (let i = 0; i < N; i++) {
  if (passable[i] === 0 || compId[i] !== -1) continue;
  let sp = 0;
  stack[sp++] = i;
  compId[i] = compCount;
  let size = 0;
  let burns = 0;
  let tval = 0;
  while (sp > 0) {
    const p = stack[--sp];
    size++;
    if (burnTime[p] < INF) burns = 1;
    const ti = townAt[p];
    if (ti !== -1) tval += towns[ti].value || 0;
    const x = p % width;
    const y = (p / width) | 0;
    for (let k = 0; k < 8; k += 2) {
      const nx = x + dirs[k],
        ny = y + dirs[k + 1];
      if (!inb(nx, ny)) continue;
      const np = idx(nx, ny);
      if (passable[np] === 0 || compId[np] !== -1) continue;
      compId[np] = compCount;
      stack[sp++] = np;
    }
  }
  compSize.push(size);
  compBurns.push(burns);
  compTownValue.push(tval);
  compCount++;
}

// Identify "frontier" edges: for each cell that burns, its predecessor closer to ignition.
// We'll score candidate cut cells by how much "burning component town value" they could potentially protect,
// discounted by distance-to-ignition (earlier cuts are more effective).

// Precompute a coarse "downstream value" from each cell: sum of town values in its component (only if burns).
// This is a crude upper bound: cutting a single cell won't necessarily save whole component, but guides choices.
const compDownValue = new Float64Array(compCount);
for (let c = 0; c < compCount; c++) compDownValue[c] = compBurns[c] ? compTownValue[c] : 0;

// Candidate scoring:
// score(cell) = sum over neighbors with burnTime = burnTime[cell]+1 of compDownValue[comp(nei)] / (1 + burnTime[nei])
// This prioritizes chokepoints that many burning paths pass through.
const candScore = new Float64Array(N);
const candMark = new Uint8Array(N);

// To avoid scanning whole grid for large cases, only consider cells that are on or near the fire's shortest-path tree:
// cells that have burnTime finite.
for (let qi = 0; qi < qt; qi++) {
  const p = q[qi];
  const bt = burnTime[p];
  const x = p % width;
  const y = (p / width) | 0;
  // consider cutting this cell (but not if ignition itself at time 0, still can cut but often pointless)
  if (passable[p] === 1) candMark[p] = 1;
  for (let k = 0; k < 8; k += 2) {
    const nx = x + dirs[k],
      ny = y + dirs[k + 1];
    if (!inb(nx, ny)) continue;
    const np = idx(nx, ny);
    if (passable[np] === 0) continue;
    if (burnTime[np] === bt + 1) {
      const c = compId[np];
      const val = c >= 0 ? compDownValue[c] : 0;
      if (val > 0) candScore[p] += val / (1 + (bt + 1));
    }
  }
}

// Add extra weight for proximity to towns that would burn early.
// For each town, walk back along burnTime gradient for up to K steps and add value.
const K = 40;
for (let ti = 0; ti < towns.length; ti++) {
  const t = towns[ti];
  if (!inb(t.x, t.y)) continue;
  const p0 = idx(t.x, t.y);
  const bt0 = burnTime[p0];
  if (bt0 >= INF) continue; // already safe
  const val = +t.value || 0;
  let p = p0;
  let bt = bt0;
  const seenLocal = new Int32Array(Math.min(K, bt0 + 1));
  let seenN = 0;
  for (let step = 0; step < K && bt > 0; step++) {
    // pick a predecessor with burnTime = bt-1 (any).
    const x = p % width;
    const y = (p / width) | 0;
    let best = -1;
    for (let k = 0; k < 8; k += 2) {
      const nx = x + dirs[k],
        ny = y + dirs[k + 1];
      if (!inb(nx, ny)) continue;
      const np = idx(nx, ny);
      if (passable[np] === 0) continue;
      if (burnTime[np] === bt - 1) {
        best = np;
        break;
      }
    }
    if (best === -1) break;
    p = best;
    bt = burnTime[p];
    // avoid double-counting within this town trace
    let dup = false;
    for (let s = 0; s < seenN; s++) if (seenLocal[s] === p) dup = true;
    if (dup) break;
    if (seenN < seenLocal.length) seenLocal[seenN++] = p;
    candMark[p] = 1;
    candScore[p] += val / (1 + bt);
  }
}

// Build list of candidates and pick top M by score.
type Cand = { p: number; score: number };
const cands: Cand[] = [];
for (let p = 0; p < N; p++) {
  if (candMark[p] && passable[p] === 1) {
    const s = candScore[p];
    if (s > 0) cands.push({ p, score: s });
  }
}
// If we somehow have no scored candidates, fall back to cutting nearest-to-ignition cells (excluding ignitions).
if (cands.length === 0) {
  const fb: number[][] = [];
  // gather up to budget cells with smallest burnTime
  // do one pass over burned cells in BFS order (already increasing burnTime)
  const cut = new Uint8Array(N);
  let used = 0;
  for (let qi = 0; qi < qt && used < cutBudget; qi++) {
    const p = q[qi];
    if (burnTime[p] === 0) continue;
    if (passable[p] === 1 && cut[p] === 0) {
      cut[p] = 1;
      fb.push([p % width, (p / width) | 0]);
      used++;
    }
  }
  process.stdout.write(JSON.stringify({ firebreaks: fb }));
  process.exit(0);
}

// Sort descending by score; cap to manageable set for simulation refinement.
cands.sort((a, b) => b.score - a.score);
const M = Math.min(cands.length, Math.max(200, cutBudget * 50));
const top = cands.slice(0, M);

// Greedy with fast re-simulation on restricted candidates:
// Maintain a set of cuts; for each step, evaluate a subset of remaining candidates via incremental simulation:
// We approximate marginal gain by running BFS with current cuts plus this candidate, but limit BFS to region that burns.

const cutMask = new Uint8Array(N);

function simulateSavedValue(extraCutP: number): number {
  // Returns total value of towns that do NOT burn under cuts = cutMask U {extraCutP}.
  const visited = new Uint8Array(N);
  const qq = new Int32Array(qt + 16);
  let h = 0,
    t = 0;
  // Seed ignitions if not cut
  for (const [x, y] of ignitions) {
    if (!inb(x, y)) continue;
    const p = idx(x, y);
    if (passable[p] === 0) continue;
    if (cutMask[p] || p === extraCutP) continue;
    if (visited[p] === 0) {
      visited[p] = 1;
      qq[t++] = p;
    }
  }
  // burn set; then compute burned towns
  const townBurned = new Uint8Array(towns.length);
  while (h < t) {
    const p = qq[h++];
    const ti = townAt[p];
    if (ti !== -1) townBurned[ti] = 1;
    const x = p % width;
    const y = (p / width) | 0;
    for (let k = 0; k < 8; k += 2) {
      const nx = x + dirs[k],
        ny = y + dirs[k + 1];
      if (!inb(nx, ny)) continue;
      const np = idx(nx, ny);
      if (passable[np] === 0) continue;
      if (cutMask[np] || np === extraCutP) continue;
      if (visited[np] === 0) {
        visited[np] = 1;
        qq[t++] = np;
      }
    }
  }
  let saved = 0;
  for (let i = 0; i < towns.length; i++) {
    if (townBurned[i] === 0) saved += towns[i].value || 0;
  }
  return saved;
}

// Precompute baseline saved value with current cuts (none initially).
let currentSaved = simulateSavedValue(-1);

// For speed, each iteration only evaluates a window of the best remaining candidates.
const firebreaks: number[][] = [];
let remaining = top.map((c) => c.p);

for (let step = 0; step < cutBudget && remaining.length > 0; step++) {
  const window = Math.min(remaining.length, Math.max(50, 5 * (cutBudget - step)));
  let bestP = -1;
  let bestGain = -1;
  // evaluate first `window` candidates
  for (let i = 0; i < window; i++) {
    const p = remaining[i];
    if (cutMask[p]) continue;
    // avoid cutting ignitions (rarely helpful)
    if (burnTime[p] === 0) continue;
    const saved = simulateSavedValue(p);
    const gain = saved - currentSaved;
    if (gain > bestGain) {
      bestGain = gain;
      bestP = p;
    }
  }
  if (bestP === -1 || bestGain <= 0) break;
  cutMask[bestP] = 1;
  currentSaved += bestGain;
  firebreaks.push([bestP % width, (bestP / width) | 0]);
  // remove chosen and also drop any candidates now cut
  const next: number[] = [];
  for (let i = 0; i < remaining.length; i++) {
    const p = remaining[i];
    if (p !== bestP && !cutMask[p]) next.push(p);
  }
  remaining = next;
}

process.stdout.write(JSON.stringify({ firebreaks }));
