import sys


def main() -> None:
    data = sys.stdin.buffer.read().splitlines()
    if not data:
        return
    try:
        u_s = data[0].decode("utf-8", "ignore").strip().split()
    except Exception:
        u_s = []
    if len(u_s) < 2:
        return

    U = int(u_s[0])
    S = int(u_s[1])

    sets = []
    element_to_sets = [[] for _ in range(U)]
    for i in range(S):
        line = data[i + 1].decode("utf-8", "ignore").strip()
        if not line:
            elems = []
        else:
            elems = [int(x) for x in line.split()]
        # normalize: unique elements, ignore out-of-range
        seen = set()
        norm = []
        for e in elems:
            if 0 <= e < U and e not in seen:
                seen.add(e)
                norm.append(e)
        sets.append(norm)
        for e in norm:
            element_to_sets[e].append(i)

    # If any element has zero candidate sets, impossible.
    for e in range(U):
        if not element_to_sets[e]:
            sys.stdout.write("NO SOLUTION\n")
            return

    chosen = [False] * S
    covered = [False] * U
    solution = []

    # Precompute bitmasks for speed when U <= 63 is not guaranteed; use lists.

    def select_set(si: int, undo_stack: list) -> bool:
        """Try to select set si; returns False if conflict (covers covered element)."""
        for e in sets[si]:
            if covered[e]:
                return False
        chosen[si] = True
        solution.append(si)
        undo_stack.append(("choose", si))
        for e in sets[si]:
            covered[e] = True
            undo_stack.append(("cover", e))
        return True

    def unselect_until(undo_stack: list, marker_len: int) -> None:
        while len(undo_stack) > marker_len:
            typ, val = undo_stack.pop()
            if typ == "cover":
                covered[val] = False
            elif typ == "choose":
                chosen[val] = False
                solution.pop()

    # For each element, track which sets are still "available" (i.e., don't conflict with current coverage).
    # We compute on the fly using covered[]; also include pruning via forced selections.

    def pick_uncovered_element():
        best_e = -1
        best_count = 10**9
        for e in range(U):
            if covered[e]:
                continue
            cnt = 0
            for si in element_to_sets[e]:
                if chosen[si]:
                    continue
                ok = True
                for ee in sets[si]:
                    if covered[ee]:
                        ok = False
                        break
                if ok:
                    cnt += 1
                    if cnt >= best_count:
                        break
            if cnt == 0:
                return e, 0
            if cnt < best_count:
                best_count = cnt
                best_e = e
                if best_count == 1:
                    # can't beat 1
                    break
        return best_e, best_count

    sys.setrecursionlimit(1_000_000)

    def dfs() -> bool:
        # All covered?
        for e in range(U):
            if not covered[e]:
                break
        else:
            return True

        e, cnt = pick_uncovered_element()
        if cnt == 0:
            return False

        # Build candidate sets for e
        candidates = []
        for si in element_to_sets[e]:
            if chosen[si]:
                continue
            ok = True
            for ee in sets[si]:
                if covered[ee]:
                    ok = False
                    break
            if ok:
                candidates.append(si)

        # Heuristic: try smaller sets first
        candidates.sort(key=lambda si: len(sets[si]))

        for si in candidates:
            undo_stack = []
            marker = len(undo_stack)
            if not select_set(si, undo_stack):
                continue
            if dfs():
                return True
            unselect_until(undo_stack, marker)
        return False

    if not dfs():
        sys.stdout.write("NO SOLUTION\n")
        return

    solution.sort()
    sys.stdout.write("SOLUTION\n")
    sys.stdout.write(" ".join(map(str, solution)) + ("\n" if solution else "\n"))


if __name__ == "__main__":
    main()
