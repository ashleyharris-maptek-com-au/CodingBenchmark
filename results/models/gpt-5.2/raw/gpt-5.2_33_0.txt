using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    // Fast integer scanner
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c != -1);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    private static ulong SplitMix64(ref ulong x)
    {
        x += 0x9E3779B97F4A7C15UL;
        ulong z = x;
        z = (z ^ (z >> 30)) * 0xBF58476D1CE4E5B9UL;
        z = (z ^ (z >> 27)) * 0x94D049BB133111EBUL;
        return z ^ (z >> 31);
    }

    private static int NextInt(ref ulong state, int bound)
    {
        // bound > 0
        return (int)(SplitMix64(ref state) % (uint)bound);
    }

    // Greedy "start solution": assign facilities in descending flow strength
    // to locations with ascending distance centrality (approx).
    private static int[] BuildInitial(int n, int[] flow, int[] dist)
    {
        long[] flowRowSum = new long[n];
        long[] distRowSum = new long[n];
        for (int i = 0; i < n; i++)
        {
            long sf = 0, sd = 0;
            int fi = i * n;
            int di = i * n;
            for (int j = 0; j < n; j++)
            {
                sf += flow[fi + j];
                sd += dist[di + j];
            }
            flowRowSum[i] = sf;
            distRowSum[i] = sd;
        }

        int[] fac = new int[n];
        int[] loc = new int[n];
        for (int i = 0; i < n; i++) { fac[i] = i; loc[i] = i; }
        Array.Sort(fac, (a, b) => flowRowSum[b].CompareTo(flowRowSum[a]));
        Array.Sort(loc, (a, b) => distRowSum[a].CompareTo(distRowSum[b]));

        int[] p = new int[n];
        for (int k = 0; k < n; k++)
            p[fac[k]] = loc[k];
        return p;
    }

    private static long ComputeCost(int n, int[] flow, int[] dist, int[] p)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = p[i];
            int fi = i * n;
            int dpi = pi * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[fi + j] * dist[dpi + p[j]];
            }
        }
        return cost;
    }

    // Try to improve via randomized pair swaps with fast delta evaluation.
    // Delta formula:
    //   swap facilities a and b (locations pa, pb), change in cost is:
    //   sum_k (flow[a][k]-flow[b][k])*(dist[pb][p[k]]-dist[pa][p[k]])
    // + sum_k (flow[k][a]-flow[k][b])*(dist[p[k]][pb]-dist[p[k]][pa])
    // + (flow[a][a]-flow[b][b])*(dist[pb][pb]-dist[pa][pa])
    // + (flow[a][b]-flow[b][a])*(dist[pb][pa]-dist[pa][pb])
    // + (flow[b][a]-flow[a][b])*(dist[pa][pb]-dist[pb][pa])  [covered by above term]
    // We compute carefully without double counting by using full directed matrices.
    private static long SwapDelta(int n, int[] flow, int[] dist, int[] p, int a, int b)
    {
        int pa = p[a], pb = p[b];
        if (pa == pb) return 0;

        long delta = 0;
        int aRow = a * n;
        int bRow = b * n;
        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = p[k];
            int kRow = k * n;

            // outgoing from a,b to k
            int fa_k = flow[aRow + k];
            int fb_k = flow[bRow + k];
            delta += (long)(fa_k - fb_k) * (dist[pb * n + pk] - dist[pa * n + pk]);

            // incoming from k to a,b
            int fk_a = flow[kRow + a];
            int fk_b = flow[kRow + b];
            delta += (long)(fk_a - fk_b) * (dist[pk * n + pb] - dist[pk * n + pa]);
        }

        // terms involving a and b themselves (directed, include diagonals)
        // (a,a) and (b,b)
        delta += (long)(flow[aRow + a] - flow[bRow + b]) * (dist[pb * n + pb] - dist[pa * n + pa]);

        // (a,b) and (b,a)
        int fab = flow[aRow + b];
        int fba = flow[bRow + a];
        delta += (long)(fab - fba) * (dist[pb * n + pa] - dist[pa * n + pb]);
        delta += (long)(fba - fab) * (dist[pa * n + pb] - dist[pb * n + pa]); // symmetric directed counterpart

        return delta;
    }

    private static void Improve(int n, int[] flow, int[] dist, int[] p, ref long bestCost)
    {
        // For N very small, do steepest local search over all swaps.
        // For larger N, do randomized first-improvement.
        var sw = System.Diagnostics.Stopwatch.StartNew();
        long timeLimitMs = 295_000; // ~5 min safety margin

        if (n <= 400)
        {
            bool improved = true;
            while (improved && sw.ElapsedMilliseconds < timeLimitMs)
            {
                improved = false;
                int bestA = -1, bestB = -1;
                long bestDelta = 0;
                for (int a = 0; a < n; a++)
                {
                    for (int b = a + 1; b < n; b++)
                    {
                        long d = SwapDelta(n, flow, dist, p, a, b);
                        if (d < bestDelta)
                        {
                            bestDelta = d;
                            bestA = a; bestB = b;
                        }
                    }
                }
                if (bestDelta < 0)
                {
                    int tmp = p[bestA]; p[bestA] = p[bestB]; p[bestB] = tmp;
                    bestCost += bestDelta;
                    improved = true;
                }
            }
            return;
        }

        // Randomized search
        ulong rng = (ulong)DateTime.UtcNow.Ticks ^ 0xD1B54A32D192ED03UL;
        int stall = 0;
        int maxStall = Math.Max(200_000, n * 500);

        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            int a = NextInt(ref rng, n);
            int b = NextInt(ref rng, n - 1);
            if (b >= a) b++;
            if (a > b) { int t = a; a = b; b = t; }

            long d = SwapDelta(n, flow, dist, p, a, b);
            if (d < 0)
            {
                int tmp = p[a]; p[a] = p[b]; p[b] = tmp;
                bestCost += d;
                stall = 0;
            }
            else
            {
                stall++;
                if (stall > maxStall)
                {
                    // small random perturbation to escape local minima
                    int kicks = 3 + NextInt(ref rng, 8);
                    for (int k = 0; k < kicks; k++)
                    {
                        int i = NextInt(ref rng, n);
                        int j = NextInt(ref rng, n - 1);
                        if (j >= i) j++;
                        int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
                    }
                    bestCost = ComputeCost(n, flow, dist, p);
                    stall = 0;
                }
            }
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n;
        try { n = fs.NextInt(); }
        catch { return; }

        int nn = checked(n * n);
        int[] flow = new int[nn];
        int[] dist = new int[nn];

        for (int i = 0; i < nn; i++) flow[i] = fs.NextInt();
        for (int i = 0; i < nn; i++) dist[i] = fs.NextInt();

        int[] p = BuildInitial(n, flow, dist);
        long cost = ComputeCost(n, flow, dist, p);
        Improve(n, flow, dist, p, ref cost);

        var w = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        w.WriteLine(cost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < n; i++)
        {
            if (i > 0) w.Write(' ');
            w.Write(p[i].ToString(CultureInfo.InvariantCulture));
        }
        w.WriteLine();
        w.Flush();
    }
}
