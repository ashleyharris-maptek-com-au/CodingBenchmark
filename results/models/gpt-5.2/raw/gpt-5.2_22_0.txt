using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const int MaxRot = 4;

    private sealed class Piece
    {
        public readonly string Type;
        public readonly Shape[] Rot;
        public Piece(string type, Shape[] rot) { Type = type; Rot = rot; }
    }

    private sealed class Shape
    {
        public readonly int[] Rows; // bitmask per local row, LSB is local x=0
        public readonly int W;
        public readonly int H;
        public Shape(int[] rows, int w, int h) { Rows = rows; W = w; H = h; }
    }

    private struct Move
    {
        public int Rot;
        public int Col;
        public double Score;
        public Move(int rot, int col, double score) { Rot = rot; Col = col; Score = score; }
    }

    private static readonly Dictionary<string, Piece> Pieces = BuildPieces();

    public static void Main()
    {
        var input = Console.In;
        string? tok = NextToken(input);
        if (tok == null) return;
        int width = int.Parse(tok);
        int height = int.Parse(NextToken(input) ?? "20");
        int numPieces = int.Parse(NextToken(input) ?? "0");

        var board = new Board(width, height);

        // Adaptive: widen the search (lookahead) for small boards / short sequences,
        // and keep it fast for larger boards / marathon runs.
        int baseLookahead = (width <= 10 && height <= 22) ? 1 : 0;
        int deeperLookahead = (width <= 12 && height <= 24) ? 2 : 1;

        for (int i = 0; i < numPieces; i++)
        {
            string? pType = NextToken(input);
            if (pType == null) break;
            if (!Pieces.TryGetValue(pType, out var piece))
            {
                // Unknown piece: fall back
                Console.WriteLine("0 0");
                Console.Out.Flush();
                string? res = NextToken(input);
                if (res == null) break;
                continue;
            }

            int filled = board.FilledCells;
            // Adaptive logic:
            // - early game: prioritize building flat surface (more lookahead helps)
            // - late game / cluttered: prioritize survival (fast decisions)
            int lookahead = baseLookahead;
            if (filled > width * height / 6) lookahead = Math.Min(deeperLookahead, 1);
            if (filled < width * 2) lookahead = deeperLookahead; // very early

            Move best = FindBestMove(board, piece, lookahead);
            Console.WriteLine($"{best.Rot} {best.Col}");
            Console.Out.Flush();

            // Read result token (ok, cleared_N, gameover)
            string? result = NextToken(input);
            if (result == null) break;
            if (result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
                break;

            // Apply the move locally so we can keep state.
            // If our move was illegal (shouldn't happen), degrade gracefully.
            if (!board.TryPlace(piece.Rot[best.Rot], best.Col, out _))
            {
                // Attempt any legal move quickly
                if (!TryAnyLegal(board, piece, out best))
                {
                    // Nothing fits; game should end
                    break;
                }
                board.TryPlace(piece.Rot[best.Rot], best.Col, out _);
            }
        }
    }

    private static bool TryAnyLegal(Board b, Piece p, out Move mv)
    {
        mv = new Move(0, 0, double.NegativeInfinity);
        for (int r = 0; r < p.Rot.Length; r++)
        {
            var s = p.Rot[r];
            for (int c = 0; c <= b.Width - s.W; c++)
            {
                if (b.CanPlace(s, c, out _))
                {
                    mv = new Move(r, c, 0);
                    return true;
                }
            }
        }
        return false;
    }

    private static Move FindBestMove(Board board, Piece piece, int lookahead)
    {
        Move best = new Move(0, 0, double.NegativeInfinity);
        int maxR = piece.Rot.Length;

        // Generate all legal moves; choose best by heuristic (+ optional shallow lookahead).
        for (int r = 0; r < maxR; r++)
        {
            var shape = piece.Rot[r];
            for (int c = 0; c <= board.Width - shape.W; c++)
            {
                if (!board.CanPlace(shape, c, out int landingY)) continue;

                var sim = board.Clone();
                sim.PlaceAt(shape, c, landingY, out int cleared);
                double score = Evaluate(sim, cleared, board, landingY);

                if (lookahead > 0)
                {
                    // Shallow expectimax-ish: approximate by best continuation over piece set
                    // without modeling random distribution precisely.
                    score += 0.35 * ContinuationScore(sim, lookahead - 1);
                }

                // Tie-breakers: prefer central columns and lower placements for stability
                score += 0.0005 * (board.Width / 2.0 - Math.Abs(c - (board.Width - shape.W) / 2.0));
                score += 0.0001 * (board.Height - landingY);

                if (score > best.Score)
                    best = new Move(r, c, score);
            }
        }

        return best;
    }

    private static double ContinuationScore(Board board, int depth)
    {
        // If depth==0, just evaluate current.
        if (depth <= 0) return Evaluate(board, 0, null, 0);

        // For performance: sample a subset deterministically.
        // We iterate all 7 pieces but cap evaluated moves per piece based on width.
        double acc = 0.0;
        int count = 0;
        foreach (var kv in Pieces)
        {
            var p = kv.Value;
            Move best = new Move(0, 0, double.NegativeInfinity);

            int maxR = p.Rot.Length;
            for (int r = 0; r < maxR; r++)
            {
                var s = p.Rot[r];
                if (s.W > board.Width) continue;
                // stride to reduce cost on large boards
                int stride = board.Width <= 10 ? 1 : (board.Width <= 14 ? 2 : 3);
                for (int c = 0; c <= board.Width - s.W; c += stride)
                {
                    if (!board.CanPlace(s, c, out int y)) continue;
                    var sim = board.Clone();
                    sim.PlaceAt(s, c, y, out int cleared);
                    double sc = Evaluate(sim, cleared, board, y);
                    if (depth > 1) sc += 0.25 * ContinuationScore(sim, depth - 1);
                    if (sc > best.Score) best = new Move(r, c, sc);
                }
            }

            if (best.Score > double.NegativeInfinity / 2)
            {
                acc += best.Score;
                count++;
            }
        }
        if (count == 0) return -1e9;
        return acc / count;
    }

    private static double Evaluate(Board b, int linesCleared, Board? prev, int landingY)
    {
        // Heuristic features:
        // - maximize line clears (with increasing reward)
        // - minimize aggregate height, holes, and bumpiness
        // - penalize deep wells / overhangs
        // - prefer keeping top low (survival)
        int w = b.Width;

        int[] heights = b.ColumnHeights;
        int aggHeight = 0;
        int maxHeight = 0;
        for (int i = 0; i < w; i++)
        {
            int h = heights[i];
            aggHeight += h;
            if (h > maxHeight) maxHeight = h;
        }

        int holes = b.Holes;
        int bump = 0;
        for (int i = 0; i < w - 1; i++)
            bump += Math.Abs(heights[i] - heights[i + 1]);

        int wells = b.WellSums;
        int transitions = b.RowTransitions + b.ColTransitions;

        // Late game scaling: punish height and holes more when near top
        double danger = Math.Max(0, maxHeight - (b.Height * 0.6));
        double heightWeight = 0.45 + 0.10 * (danger / Math.Max(1.0, b.Height * 0.4));
        double holeWeight = 1.20 + 0.45 * (danger / Math.Max(1.0, b.Height * 0.4));

        // Line clear reward: strong for tetrises, decent for singles, etc.
        double clearReward = linesCleared switch
        {
            0 => 0.0,
            1 => 1.5,
            2 => 4.2,
            3 => 8.5,
            _ => 15.5
        };

        // Small reward for placement lower (stability), avoid stacking high
        double landingReward = 0.0;
        if (prev != null) landingReward = 0.02 * (prev.Height - landingY);

        double score =
            8.0 * clearReward
            - heightWeight * aggHeight
            - 2.8 * maxHeight
            - holeWeight * holes
            - 0.35 * bump
            - 0.20 * wells
            - 0.06 * transitions
            + landingReward;

        return score;
    }

    private sealed class Board
    {
        public int Width { get; }
        public int Height { get; }

        // Bottom row is y=0.
        private readonly uint[] rows;

        private int[]? colHeightsCache;
        private int? holesCache;
        private int? filledCache;
        private int? wellCache;
        private int? rowTransCache;
        private int? colTransCache;

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            rows = new uint[height];
        }

        private Board(int width, int height, uint[] rowsCopy)
        {
            Width = width;
            Height = height;
            rows = rowsCopy;
        }

        public Board Clone()
        {
            var copy = new uint[rows.Length];
            Array.Copy(rows, copy, rows.Length);
            return new Board(Width, Height, copy);
        }

        public int FilledCells
        {
            get
            {
                if (filledCache.HasValue) return filledCache.Value;
                int sum = 0;
                for (int y = 0; y < Height; y++)
                    sum += PopCount(rows[y]);
                filledCache = sum;
                return sum;
            }
        }

        public int[] ColumnHeights
        {
            get
            {
                if (colHeightsCache != null) return colHeightsCache;
                var h = new int[Width];
                for (int x = 0; x < Width; x++)
                {
                    int y = Height - 1;
                    while (y >= 0)
                    {
                        if (((rows[y] >> x) & 1U) != 0) break;
                        y--;
                    }
                    h[x] = y + 1;
                }
                colHeightsCache = h;
                return h;
            }
        }

        public int Holes
        {
            get
            {
                if (holesCache.HasValue) return holesCache.Value;
                int holes = 0;
                var h = ColumnHeights;
                for (int x = 0; x < Width; x++)
                {
                    for (int y = 0; y < h[x]; y++)
                    {
                        if (((rows[y] >> x) & 1U) == 0) holes++;
                    }
                }
                holesCache = holes;
                return holes;
            }
        }

        public int WellSums
        {
            get
            {
                if (wellCache.HasValue) return wellCache.Value;
                var h = ColumnHeights;
                int sum = 0;
                for (int x = 0; x < Width; x++)
                {
                    int left = (x == 0) ? int.MaxValue : h[x - 1];
                    int right = (x == Width - 1) ? int.MaxValue : h[x + 1];
                    int minSide = Math.Min(left, right);
                    if (minSide > h[x])
                    {
                        int depth = minSide - h[x];
                        sum += depth * (depth + 1) / 2;
                    }
                }
                wellCache = sum;
                return sum;
            }
        }

        public int RowTransitions
        {
            get
            {
                if (rowTransCache.HasValue) return rowTransCache.Value;
                uint fullMask = Width >= 32 ? 0xFFFFFFFFU : ((1U << Width) - 1U);
                int trans = 0;
                for (int y = 0; y < Height; y++)
                {
                    uint r = rows[y] & fullMask;
                    uint prev = 1U; // left wall treated as filled
                    for (int x = 0; x < Width; x++)
                    {
                        uint cur = (r >> x) & 1U;
                        if (cur != prev) trans++;
                        prev = cur;
                    }
                    if (prev == 0U) trans++; // right wall filled
                }
                rowTransCache = trans;
                return trans;
            }
        }

        public int ColTransitions
        {
            get
            {
                if (colTransCache.HasValue) return colTransCache.Value;
                int trans = 0;
                for (int x = 0; x < Width; x++)
                {
                    uint prev = 1U; // bottom wall filled
                    for (int y = 0; y < Height; y++)
                    {
                        uint cur = (rows[y] >> x) & 1U;
                        if (cur != prev) trans++;
                        prev = cur;
                    }
                    if (prev == 0U) trans++; // top wall filled
                }
                colTransCache = trans;
                return trans;
            }
        }

        private void Invalidate()
        {
            colHeightsCache = null;
            holesCache = null;
            filledCache = null;
            wellCache = null;
            rowTransCache = null;
            colTransCache = null;
        }

        public bool CanPlace(Shape s, int col, out int landingY)
        {
            landingY = -1;
            if (col < 0 || col + s.W > Width) return false;

            // Drop from top: find the highest y such that shape fits without collision.
            // We represent rows bottom-up, so start at y = Height - s.H and go down.
            for (int y = Height - s.H; y >= 0; y--)
            {
                if (!CollidesAt(s, col, y))
                {
                    // Ensure it cannot go further down
                    if (y == 0 || CollidesAt(s, col, y - 1))
                    {
                        landingY = y;
                        return true;
                    }
                }
            }
            return false;
        }

        public bool TryPlace(Shape s, int col, out int cleared)
        {
            cleared = 0;
            if (!CanPlace(s, col, out int y)) return false;
            PlaceAt(s, col, y, out cleared);
            return true;
        }

        public void PlaceAt(Shape s, int col, int y, out int cleared)
        {
            cleared = 0;
            for (int sy = 0; sy < s.H; sy++)
            {
                uint mask = (uint)(s.Rows[sy] << col);
                rows[y + sy] |= mask;
            }

            uint full = Width >= 32 ? 0xFFFFFFFFU : ((1U << Width) - 1U);
            int write = 0;
            for (int read = 0; read < Height; read++)
            {
                if ((rows[read] & full) == full)
                {
                    cleared++;
                    continue;
                }
                if (write != read) rows[write] = rows[read];
                write++;
            }
            for (int k = write; k < Height; k++) rows[k] = 0;
            Invalidate();
        }

        private bool CollidesAt(Shape s, int col, int y)
        {
            for (int sy = 0; sy < s.H; sy++)
            {
                uint mask = (uint)(s.Rows[sy] << col);
                if ((rows[y + sy] & mask) != 0) return true;
            }
            return false;
        }

        private static int PopCount(uint v)
        {
            // Built-in popcount is available in .NET, but keep it simple and fast.
            v = v - ((v >> 1) & 0x55555555u);
            v = (v & 0x33333333u) + ((v >> 2) & 0x33333333u);
            return (int)((((v + (v >> 4)) & 0x0F0F0F0Fu) * 0x01010101u) >> 24);
        }
    }

    private static Dictionary<string, Piece> BuildPieces()
    {
        // Shapes are defined in local coordinates with bottom row first.
        // Each row is a bitmask; width/height are tight bounding boxes.
        var d = new Dictionary<string, Piece>(StringComparer.OrdinalIgnoreCase);

        d["I"] = new Piece("I", new[]
        {
            new Shape(new[]{ 0b1111 }, 4, 1),
            new Shape(new[]{ 0b1,0b1,0b1,0b1 }, 1, 4),
            new Shape(new[]{ 0b1111 }, 4, 1),
            new Shape(new[]{ 0b1,0b1,0b1,0b1 }, 1, 4),
        });

        d["O"] = new Piece("O", new[]
        {
            new Shape(new[]{ 0b11, 0b11 }, 2, 2),
            new Shape(new[]{ 0b11, 0b11 }, 2, 2),
            new Shape(new[]{ 0b11, 0b11 }, 2, 2),
            new Shape(new[]{ 0b11, 0b11 }, 2, 2),
        });

        d["T"] = new Piece("T", new[]
        {
            new Shape(new[]{ 0b111, 0b010 }, 3, 2), // up
            new Shape(new[]{ 0b01, 0b11, 0b01 }, 2, 3), // right
            new Shape(new[]{ 0b010, 0b111 }, 3, 2), // down
            new Shape(new[]{ 0b10, 0b11, 0b10 }, 2, 3), // left
        });

        d["S"] = new Piece("S", new[]
        {
            new Shape(new[]{ 0b011, 0b110 }, 3, 2),
            new Shape(new[]{ 0b1, 0b11, 0b10 }, 2, 3),
            new Shape(new[]{ 0b011, 0b110 }, 3, 2),
            new Shape(new[]{ 0b1, 0b11, 0b10 }, 2, 3),
        });

        d["Z"] = new Piece("Z", new[]
        {
            new Shape(new[]{ 0b110, 0b011 }, 3, 2),
            new Shape(new[]{ 0b10, 0b11, 0b01 }, 2, 3),
            new Shape(new[]{ 0b110, 0b011 }, 3, 2),
            new Shape(new[]{ 0b10, 0b11, 0b01 }, 2, 3),
        });

        d["J"] = new Piece("J", new[]
        {
            new Shape(new[]{ 0b111, 0b100 }, 3, 2),
            new Shape(new[]{ 0b11, 0b01, 0b01 }, 2, 3),
            new Shape(new[]{ 0b001, 0b111 }, 3, 2),
            new Shape(new[]{ 0b10, 0b10, 0b11 }, 2, 3),
        });

        d["L"] = new Piece("L", new[]
        {
            new Shape(new[]{ 0b111, 0b001 }, 3, 2),
            new Shape(new[]{ 0b01, 0b01, 0b11 }, 2, 3),
            new Shape(new[]{ 0b100, 0b111 }, 3, 2),
            new Shape(new[]{ 0b11, 0b10, 0b10 }, 2, 3),
        });

        return d;
    }

    private static string? NextToken(TextReader tr)
    {
        int c;
        do
        {
            c = tr.Read();
            if (c < 0) return null;
        } while (c <= 32);

        var sb = new StringBuilder();
        while (c > 32)
        {
            sb.Append((char)c);
            c = tr.Read();
            if (c < 0) break;
        }
        return sb.ToString();
    }
}
