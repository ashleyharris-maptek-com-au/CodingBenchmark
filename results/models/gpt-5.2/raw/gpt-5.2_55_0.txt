using namespace std;

static inline int file_of(int sq) { return sq & 7; }
static inline int rank_of(int sq) { return sq >> 3; }
static inline int sq_of(int f, int r) { return (r << 3) | f; }

static inline string sq_to_str(int sq) {
    string s;
    s.push_back(char('a' + file_of(sq)));
    s.push_back(char('1' + rank_of(sq)));
    return s;
}

enum Piece : int {
    EMPTY = 0,
    WP=1, WN, WB, WR, WQ, WK,
    BP=7, BN, BB, BR, BQ, BK
};

static inline bool is_white(int p){ return p>=WP && p<=WK; }
static inline bool is_black(int p){ return p>=BP && p<=BK; }
static inline int  color_of(int p){ return is_white(p)?0:(is_black(p)?1:-1); }
static inline int  type_of(int p){
    if(p==EMPTY) return 0;
    if(p>=BP) return p-6;
    return p;
}

struct Move {
    uint8_t from=0, to=0;
    uint8_t promo=0;        // 0 none else WN..WQ etc type in 1..6
    uint8_t flags=0;        // bits: 1 capture,2 ep,4 castle,8 pawnDouble
};

struct Position {
    array<int,64> b{};
    int side=0; // 0 white, 1 black
    bool wk=true,wq=true,bk=true,bq=true;
    int ep=-1; // square index or -1
    int halfmove=0, fullmove=1;
};

static inline int piece_value(int t){
    switch(t){
        case 1: return 100;
        case 2: return 320;
        case 3: return 330;
        case 4: return 500;
        case 5: return 900;
        case 6: return 20000;
        default: return 0;
    }
}

static inline int pst_value(int type, int sq, int color){
    int idx = (color==0)? sq : (sq_of(file_of(sq), 7-rank_of(sq)));
    switch(type){
        case 1: return PST_P[idx];
        case 2: return PST_N[idx];
        case 3: return PST_B[idx];
        case 4: return PST_R[idx];
        case 5: return PST_Q[idx];
        case 6: return PST_K[idx];
        default: return 0;
    }
}

static bool parse_fen(const string& fen, Position& pos){
    pos = Position{};
    for(int i=0;i<64;i++) pos.b[i]=EMPTY;
    string s=fen;
    size_t i=0;
    int r=7,f=0;
    while(i<s.size() && s[i]!=' '){
        char c=s[i++];
        if(c=='/'){ r--; f=0; continue; }
        if(c>='1'&&c<='8'){ f += (c-'0'); continue; }
        int p=EMPTY;
        switch(c){
            case 'P': p=WP; break; case 'N': p=WN; break; case 'B': p=WB; break;
            case 'R': p=WR; break; case 'Q': p=WQ; break; case 'K': p=WK; break;
            case 'p': p=BP; break; case 'n': p=BN; break; case 'b': p=BB; break;
            case 'r': p=BR; break; case 'q': p=BQ; break; case 'k': p=BK; break;
            default: return false;
        }
        if(f<0||f>7||r<0||r>7) return false;
        pos.b[sq_of(f,r)]=p;
        f++;
    }
    if(i>=s.size()||s[i]!=' ') return false;
    i++;
    if(i>=s.size()) return false;
    pos.side = (s[i]=='w')?0:1;
    while(i<s.size() && s[i]!=' ') i++;
    if(i>=s.size()) return false;
    i++;
    pos.wk=pos.wq=pos.bk=pos.bq=false;
    if(i>=s.size()) return false;
    if(s[i]=='-'){ i++; }
    else{
        while(i<s.size() && s[i]!=' '){
            switch(s[i]){
                case 'K': pos.wk=true; break;
                case 'Q': pos.wq=true; break;
                case 'k': pos.bk=true; break;
                case 'q': pos.bq=true; break;
            }
            i++;
        }
    }
    if(i>=s.size()||s[i]!=' ') return false;
    i++;
    if(i>=s.size()) return false;
    if(s[i]=='-'){ pos.ep=-1; i++; }
    else{
        if(i+1>=s.size()) return false;
        char fc=s[i], rc=s[i+1];
        if(fc<'a'||fc>'h'||rc<'1'||rc>'8') return false;
        pos.ep=sq_of(fc-'a', rc-'1');
        i+=2;
    }
    while(i<s.size() && s[i]==' ') i++;
    if(i<s.size()){
        pos.halfmove = atoi(s.c_str()+i);
        while(i<s.size() && s[i]!=' ') i++;
        if(i<s.size()){ i++; pos.fullmove = atoi(s.c_str()+i); }
    }
    return true;
}

static inline bool on_board(int f,int r){ return f>=0&&f<8&&r>=0&&r<8; }

static bool is_attacked(const Position& pos, int sq, int byColor){
    // pawns
    int f=file_of(sq), r=rank_of(sq);
    if(byColor==0){
        int r2=r-1;
        if(r2>=0){
            if(f-1>=0 && pos.b[sq_of(f-1,r2)]==WP) return true;
            if(f+1<8 && pos.b[sq_of(f+1,r2)]==WP) return true;
        }
    }else{
        int r2=r+1;
        if(r2<8){
            if(f-1>=0 && pos.b[sq_of(f-1,r2)]==BP) return true;
            if(f+1<8 && pos.b[sq_of(f+1,r2)]==BP) return true;
        }
    }
    // knights
    static const int kdf[8]={1,2,2,1,-1,-2,-2,-1};
    static const int kdr[8]={2,1,-1,-2,-2,-1,1,2};
    int kn = (byColor==0)?WN:BN;
    for(int i=0;i<8;i++){
        int nf=f+kdf[i], nr=r+kdr[i];
        if(on_board(nf,nr) && pos.b[sq_of(nf,nr)]==kn) return true;
    }
    // sliders
    auto ray = [&](int df,int dr,int bishopLike)->bool{
        int nf=f+df, nr=r+dr;
        while(on_board(nf,nr)){
            int p=pos.b[sq_of(nf,nr)];
            if(p!=EMPTY){
                int t=type_of(p);
                if(color_of(p)==byColor){
                    if(bishopLike){
                        if(t==3||t==5) return true;
                    }else{
                        if(t==4||t==5) return true;
                    }
                }
                return false;
            }
            nf+=df; nr+=dr;
        }
        return false;
    };
    // bishops/queens
    if(ray(1,1,1)||ray(1,-1,1)||ray(-1,1,1)||ray(-1,-1,1)) return true;
    // rooks/queens
    if(ray(1,0,0)||ray(-1,0,0)||ray(0,1,0)||ray(0,-1,0)) return true;
    // king
    int king = (byColor==0)?WK:BK;
    for(int df=-1;df<=1;df++) for(int dr=-1;dr<=1;dr++){
        if(df==0&&dr==0) continue;
        int nf=f+df,nr=r+dr;
        if(on_board(nf,nr) && pos.b[sq_of(nf,nr)]==king) return true;
    }
    return false;
}

static int find_king(const Position& pos, int color){
    int k = (color==0)?WK:BK;
    for(int i=0;i<64;i++) if(pos.b[i]==k) return i;
    return -1;
}

static bool in_check(const Position& pos, int color){
    int ksq=find_king(pos,color);
    if(ksq<0) return false;
    return is_attacked(pos, ksq, color^1);
}

struct Undo {
    int captured=EMPTY;
    bool wk,wq,bk,bq;
    int ep;
};

static void do_move(Position& pos, const Move& m, Undo& u){
    u.captured = pos.b[m.to];
    u.wk=pos.wk; u.wq=pos.wq; u.bk=pos.bk; u.bq=pos.bq;
    u.ep=pos.ep;
    int p = pos.b[m.from];
    int color=color_of(p);
    pos.ep=-1;
    // castling rights updates on king/rook move/capture
    if(p==WK){ pos.wk=pos.wq=false; }
    if(p==BK){ pos.bk=pos.bq=false; }
    if(p==WR && m.from==sq_of(0,0)) pos.wq=false;
    if(p==WR && m.from==sq_of(7,0)) pos.wk=false;
    if(p==BR && m.from==sq_of(0,7)) pos.bq=false;
    if(p==BR && m.from==sq_of(7,7)) pos.bk=false;
    if(u.captured==WR && m.to==sq_of(0,0)) pos.wq=false;
    if(u.captured==WR && m.to==sq_of(7,0)) pos.wk=false;
    if(u.captured==BR && m.to==sq_of(0,7)) pos.bq=false;
    if(u.captured==BR && m.to==sq_of(7,7)) pos.bk=false;

    // special moves
    if(m.flags & 2){ // en passant capture
        int capSq = (color==0)? (m.to-8) : (m.to+8);
        u.captured = pos.b[capSq];
        pos.b[capSq]=EMPTY;
    }
    pos.b[m.to]=p;
    pos.b[m.from]=EMPTY;
    if(m.flags & 4){ // castle: move rook too
        if(p==WK){
            if(m.to==sq_of(6,0)){ // O-O
                pos.b[sq_of(5,0)]=WR; pos.b[sq_of(7,0)]=EMPTY;
            }else if(m.to==sq_of(2,0)){ // O-O-O
                pos.b[sq_of(3,0)]=WR; pos.b[sq_of(0,0)]=EMPTY;
            }
        }else if(p==BK){
            if(m.to==sq_of(6,7)){
                pos.b[sq_of(5,7)]=BR; pos.b[sq_of(7,7)]=EMPTY;
            }else if(m.to==sq_of(2,7)){
                pos.b[sq_of(3,7)]=BR; pos.b[sq_of(0,7)]=EMPTY;
            }
        }
    }
    if(m.flags & 8){ // pawn double
        pos.ep = (color==0)? (m.from+8) : (m.from-8);
    }
    // promotion
    if(m.promo){
        int t=m.promo;
        pos.b[m.to] = (color==0)? t : (t+6);
    }
    pos.side ^= 1;
}

static void undo_move(Position& pos, const Move& m, const Undo& u){
    pos.side ^= 1;
    pos.wk=u.wk; pos.wq=u.wq; pos.bk=u.bk; pos.bq=u.bq;
    pos.ep=u.ep;
    int p = pos.b[m.to];
    int color = pos.side;
    // undo promotion -> pawn
    if(m.promo){
        p = (color==0)? WP : BP;
    }
    // undo castling rook move
    if(m.flags & 4){
        if(p==WK){
            if(m.to==sq_of(6,0)){
                pos.b[sq_of(7,0)]=WR; pos.b[sq_of(5,0)]=EMPTY;
            }else if(m.to==sq_of(2,0)){
                pos.b[sq_of(0,0)]=WR; pos.b[sq_of(3,0)]=EMPTY;
            }
        }else if(p==BK){
            if(m.to==sq_of(6,7)){
                pos.b[sq_of(7,7)]=BR; pos.b[sq_of(5,7)]=EMPTY;
            }else if(m.to==sq_of(2,7)){
                pos.b[sq_of(0,7)]=BR; pos.b[sq_of(3,7)]=EMPTY;
            }
        }
    }
    pos.b[m.from]=p;
    pos.b[m.to]=u.captured;
    if(m.flags & 2){
        int capSq = (color==0)? (m.to-8) : (m.to+8);
        pos.b[m.to]=EMPTY;
        pos.b[capSq]=u.captured;
    }
}

static void gen_moves(const Position& pos, vector<Move>& out){
    out.clear();
    int us=pos.side;
    int pawn = (us==0)? WP:BP;
    int knight=(us==0)? WN:BN;
    int bishop=(us==0)? WB:BB;
    int rook=(us==0)? WR:BR;
    int queen=(us==0)? WQ:BQ;
    int king=(us==0)? WK:BK;
    for(int sq=0;sq<64;sq++){
        int p=pos.b[sq];
        if(p==EMPTY || color_of(p)!=us) continue;
        int f=file_of(sq), r=rank_of(sq);
        int t=type_of(p);
        if(p==pawn){
            int dir = (us==0)? 1 : -1;
            int startRank = (us==0)? 1 : 6;
            int promoRank = (us==0)? 7 : 0;
            int r1=r+dir;
            if(r1>=0 && r1<8){
                int to = sq_of(f,r1);
                if(pos.b[to]==EMPTY){
                    if(r1==promoRank){
                        for(int pr : {5,4,3,2}){ // Q,R,B,N
                            Move m; m.from=sq; m.to=to; m.promo=pr; out.push_back(m);
                        }
                    }else{
                        Move m; m.from=sq; m.to=to; out.push_back(m);
                        if(r==startRank){
                            int r2=r+2*dir;
                            int to2=sq_of(f,r2);
                            if(pos.b[to2]==EMPTY){
                                Move dm; dm.from=sq; dm.to=to2; dm.flags|=8; out.push_back(dm);
                            }
                        }
                    }
                }
                // captures
                for(int df : {-1,1}){
                    int nf=f+df;
                    if(nf<0||nf>=8) continue;
                    int cap=sq_of(nf,r1);
                    int cp=pos.b[cap];
                    if(cp!=EMPTY && color_of(cp)!=(int)us){
                        if(r1==promoRank){
                            for(int pr : {5,4,3,2}){
                                Move m; m.from=sq; m.to=cap; m.promo=pr; m.flags|=1; out.push_back(m);
                            }
                        }else{
                            Move m; m.from=sq; m.to=cap; m.flags|=1; out.push_back(m);
                        }
                    }
                    // en passant
                    if(pos.ep==cap){
                        Move m; m.from=sq; m.to=cap; m.flags|=1; m.flags|=2; out.push_back(m);
                    }
                }
            }
        } else if(p==knight){
            static const int kdf[8]={1,2,2,1,-1,-2,-2,-1};
            static const int kdr[8]={2,1,-1,-2,-2,-1,1,2};
            for(int i=0;i<8;i++){
                int nf=f+kdf[i], nr=r+kdr[i];
                if(!on_board(nf,nr)) continue;
                int to=sq_of(nf,nr);
                int tp=pos.b[to];
                if(tp==EMPTY || color_of(tp)!=(int)us){
                    Move m; m.from=sq; m.to=to;
                    if(tp!=EMPTY) m.flags|=1;
                    out.push_back(m);
                }
            }
        } else if(p==bishop || p==rook || p==queen){
            vector<pair<int,int>> dirs;
            if(p==bishop || p==queen){
                dirs.push_back({1,1}); dirs.push_back({1,-1}); dirs.push_back({-1,1}); dirs.push_back({-1,-1});
            }
            if(p==rook || p==queen){
                dirs.push_back({1,0}); dirs.push_back({-1,0}); dirs.push_back({0,1}); dirs.push_back({0,-1});
            }
            for(auto [df,dr]:dirs){
                int nf=f+df,nr=r+dr;
                while(on_board(nf,nr)){
                    int to=sq_of(nf,nr);
                    int tp=pos.b[to];
                    if(tp==EMPTY){
                        Move m; m.from=sq; m.to=to; out.push_back(m);
                    }else{
                        if(color_of(tp)!=(int)us){
                            Move m; m.from=sq; m.to=to; m.flags|=1; out.push_back(m);
                        }
                        break;
                    }
                    nf+=df; nr+=dr;
                }
            }
        } else if(p==king){
            for(int df=-1;df<=1;df++) for(int dr=-1;dr<=1;dr++){
                if(df==0&&dr==0) continue;
                int nf=f+df,nr=r+dr;
                if(!on_board(nf,nr)) continue;
                int to=sq_of(nf,nr);
                int tp=pos.b[to];
                if(tp==EMPTY || color_of(tp)!=(int)us){
                    Move m; m.from=sq; m.to=to;
                    if(tp!=EMPTY) m.flags|=1;
                    out.push_back(m);
                }
            }
            // castling
            if(us==0){
                if(pos.wk && pos.b[sq_of(5,0)]==EMPTY && pos.b[sq_of(6,0)]==EMPTY){
                    if(!is_attacked(pos, sq_of(4,0), 1) && !is_attacked(pos, sq_of(5,0), 1) && !is_attacked(pos, sq_of(6,0), 1)){
                        Move m; m.from=sq; m.to=sq_of(6,0); m.flags|=4; out.push_back(m);
                    }
                }
                if(pos.wq && pos.b[sq_of(3,0)]==EMPTY && pos.b[sq_of(2,0)]==EMPTY && pos.b[sq_of(1,0)]==EMPTY){
                    if(!is_attacked(pos, sq_of(4,0), 1) && !is_attacked(pos, sq_of(3,0), 1) && !is_attacked(pos, sq_of(2,0), 1)){
                        Move m; m.from=sq; m.to=sq_of(2,0); m.flags|=4; out.push_back(m);
                    }
                }
            }else{
                if(pos.bk && pos.b[sq_of(5,7)]==EMPTY && pos.b[sq_of(6,7)]==EMPTY){
                    if(!is_attacked(pos, sq_of(4,7), 0) && !is_attacked(pos, sq_of(5,7), 0) && !is_attacked(pos, sq_of(6,7), 0)){
                        Move m; m.from=sq; m.to=sq_of(6,7); m.flags|=4; out.push_back(m);
                    }
                }
                if(pos.bq && pos.b[sq_of(3,7)]==EMPTY && pos.b[sq_of(2,7)]==EMPTY && pos.b[sq_of(1,7)]==EMPTY){
                    if(!is_attacked(pos, sq_of(4,7), 0) && !is_attacked(pos, sq_of(3,7), 0) && !is_attacked(pos, sq_of(2,7), 0)){
                        Move m; m.from=sq; m.to=sq_of(2,7); m.flags|=4; out.push_back(m);
                    }
                }
            }
        }
    }
}

static void gen_legal_moves(const Position& pos, vector<Move>& out){
    vector<Move> pseudo;
    gen_moves(pos,pseudo);
    out.clear();
    Position tmp=pos;
    for(const auto& m : pseudo){
        Undo u;
        tmp=pos;
        do_move(tmp,m,u);
        if(!in_check(tmp, pos.side)){
            out.push_back(m);
        }
    }
}

static int evaluate(const Position& pos){
    int score=0;
    for(int sq=0;sq<64;sq++){
        int p=pos.b[sq];
        if(p==EMPTY) continue;
        int c=color_of(p);
        int t=type_of(p);
        int v=piece_value(t) + pst_value(t,sq,c);
        score += (c==0)? v : -v;
    }
    // small mobility term
    vector<Move> moves;
    gen_legal_moves(pos,moves);
    int mob = (int)moves.size();
    score += (pos.side==0)? (mob*2) : -(mob*2);
    return score;
}

struct TTEntry{
    uint64_t key=0;
    int depth=0;
    int score=0;
    uint8_t flag=0; //0 exact 1 alpha 2 beta
    Move best{};
};

static uint64_t zobEp[9]; // file 0..7, 8 none

static uint64_t hash_pos(const Position& pos){
    uint64_t h=0;
    for(int sq=0;sq<64;sq++){
        int p=pos.b[sq];
        if(p!=EMPTY) h ^= zobPiece[p][sq];
    }
    if(pos.side) h ^= zobSide;
    int c=0;
    if(pos.wk) c|=1;
    if(pos.wq) c|=2;
    if(pos.bk) c|=4;
    if(pos.bq) c|=8;
    h ^= zobCastle[c];
    int epFile = (pos.ep==-1)?8:file_of(pos.ep);
    h ^= zobEp[epFile];
    return h;
}

static void init_zob(){
    std::mt19937_64 rng(20260515ULL);
    auto r64=[&](){ return rng(); };
    for(int p=0;p<13;p++) for(int sq=0;sq<64;sq++) zobPiece[p][sq]=r64();
    zobSide=r64();
    for(int i=0;i<16;i++) zobCastle[i]=r64();
    for(int i=0;i<9;i++) zobEp[i]=r64();
}

struct Searcher {
    vector<TTEntry> tt;
    chrono::steady_clock::time_point start;
    chrono::steady_clock::time_point deadline;
    atomic<bool> stop{false};
    int nodes=0;
    Move bestMove{};
    int bestScore=0;
    int rootDepth=0;

    explicit Searcher(size_t ttSize=1<<20){
        tt.resize(ttSize);
    }

    inline TTEntry* probe(uint64_t key){
        return &tt[key % tt.size()];
    }

    int qsearch(Position& pos, int alpha, int beta){
        if(stop.load(memory_order_relaxed)) return 0;
        if(chrono::steady_clock::now() >= deadline){
            stop.store(true, memory_order_relaxed);
            return 0;
        }
        nodes++;
        int stand = evaluate(pos);
        if(stand >= beta) return beta;
        if(stand > alpha) alpha = stand;

        vector<Move> moves;
        gen_legal_moves(pos,moves);
        // only captures in qsearch (and promotions)
        for(const auto& m : moves){
            if(!(m.flags&1) && m.promo==0) continue;
            Undo u;
            do_move(pos,m,u);
            int score = -qsearch(pos, -beta, -alpha);
            undo_move(pos,m,u);
            if(stop.load(memory_order_relaxed)) return 0;
            if(score >= beta) return beta;
            if(score > alpha) alpha = score;
        }
        return alpha;
    }

    int negamax(Position& pos, int depth, int alpha, int beta){
        if(stop.load(memory_order_relaxed)) return 0;
        if(chrono::steady_clock::now() >= deadline){
            stop.store(true, memory_order_relaxed);
            return 0;
        }
        nodes++;
        uint64_t key=hash_pos(pos);
        TTEntry* e=probe(key);
        if(e->key==key && e->depth>=depth){
            if(e->flag==0) return e->score;
            if(e->flag==1 && e->score<=alpha) return e->score;
            if(e->flag==2 && e->score>=beta) return e->score;
        }

        vector<Move> moves;
        gen_legal_moves(pos,moves);
        if(moves.empty()){
            if(in_check(pos,pos.side)){
                return -30000 + (rootDepth - depth);
            }
            return 0; // stalemate
        }

        // move ordering: TT move, captures/promotions first, then others
        Move ttMove{};
        bool hasTT=false;
        if(e->key==key){ ttMove=e->best; hasTT=true; }
        auto score_move = [&](const Move& m)->int{
            int s=0;
            if(hasTT && m.from==ttMove.from && m.to==ttMove.to && m.promo==ttMove.promo && m.flags==ttMove.flags) s+=1000000;
            if(m.promo) s+=900000;
            if(m.flags&1){
                int cap=pos.b[m.to];
                if(m.flags&2){
                    cap = (pos.side==0)? BP:WP;
                }
                int mv=piece_value(type_of(pos.b[m.from]));
                int cv=piece_value(type_of(cap));
                s += 500000 + (cv*10 - mv);
            }
            // slight centralization
            int tf=file_of(m.to), tr=rank_of(m.to);
            s += 10*(3-abs(tf-3)) + 10*(3-abs(tr-3));
            return s;
        };
        std::stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b){
            return score_move(a) > score_move(b);
        });

        int origAlpha=alpha;
        Move bestLocal{};
        int best=-32000;
        for(const auto& m : moves){
            Undo u;
            do_move(pos,m,u);
            int score;
            if(depth<=1){
                score = -qsearch(pos, -beta, -alpha);
            }else{
                score = -negamax(pos, depth-1, -beta, -alpha);
            }
            undo_move(pos,m,u);
            if(stop.load(memory_order_relaxed)) return 0;
            if(score>best){
                best=score;
                bestLocal=m;
            }
            if(score>alpha) alpha=score;
            if(alpha>=beta) break;
        }

        // store TT
        e->key=key;
        e->depth=depth;
        e->score=best;
        e->best=bestLocal;
        if(best<=origAlpha) e->flag=1;
        else if(best>=beta) e->flag=2;
        else e->flag=0;
        return best;
    }

    Move search(Position root, double timeoutSeconds){
        start = chrono::steady_clock::now();
        // keep a safety margin
        auto ms = (int)(timeoutSeconds * 1000.0);
        int safety = 60;
        if(ms > 200) safety = 120;
        deadline = start + chrono::milliseconds(max(1, ms - safety));
        stop.store(false, memory_order_relaxed);
        nodes=0;
        bestScore=-32000;
        bestMove = Move{};
        vector<Move> legal;
        gen_legal_moves(root, legal);
        if(legal.empty()) return Move{};
        bestMove=legal[0];

        for(int d=1; ; d++){
            if(stop.load(memory_order_relaxed)) break;
            if(chrono::steady_clock::now() >= deadline) break;
            rootDepth=d;
            Position pos=root;
            int score = negamax(pos, d, -32000, 32000);
            if(stop.load(memory_order_relaxed)) break;
            bestScore=score;
            // TT at root should have best move
            uint64_t key=hash_pos(root);
            TTEntry* e=probe(key);
            if(e->key==key){
                bestMove=e->best;
            }
        }
        return bestMove;
    }
};

static char piece_letter_for_move(int pieceType){
    switch(pieceType){
        case 1: return 'p';
        case 2: return 'N';
        case 3: return 'B';
        case 4: return 'R';
        case 5: return 'Q';
        case 6: return 'K';
        default: return '?';
    }
}

static string format_move(const Position& pos, const Move& m){
    int p = pos.b[m.from];
    int t = type_of(p);
    // castling
    if((m.flags&4) && t==6){
        if(m.to==sq_of(6,0) || m.to==sq_of(6,7)) return "O-O";
        return "O-O-O";
    }
    string s;
    s.push_back(piece_letter_for_move(t));
    s += sq_to_str(m.from);
    if(m.flags & 1) s.push_back('x');
    s += sq_to_str(m.to);
    if(m.flags & 2) s += "ep";
    if(m.promo){
        s.push_back('/');
        char pl = piece_letter_for_move(m.promo);
        // promotion wants piece letter, use uppercase as given (N/B/R/Q/K)
        s.push_back(pl);
    }
    // append check/mate if needed
    Position tmp=pos;
    Undo u;
    do_move(tmp,m,u);
    vector<Move> opp;
    gen_legal_moves(tmp, opp);
    bool givesCheck = in_check(tmp, tmp.side);
    if(givesCheck){
        if(opp.empty()) s.push_back('#');
        else s.push_back('+');
    }
    return s;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeout=5.0;
    string fen;
    if(!(cin>>timeout)) return 0;
    string rest;
    getline(cin, rest); // endline
    getline(cin, fen);
    if(fen.empty()) return 0;

    init_zob();
    Position pos;
    if(!parse_fen(fen, pos)){
        cout << "pe2e4\n";
        return 0;
    }

    Searcher s(1<<20);
    Move best = s.search(pos, timeout);
    vector<Move> legal;
    gen_legal_moves(pos, legal);
    if(legal.empty()){
        cout << "pe2e4\n";
        return 0;
    }
    // If search returned default/illegal, fall back to first legal
    bool ok=false;
    for(const auto& m: legal){
        if(m.from==best.from && m.to==best.to && m.promo==best.promo && m.flags==best.flags){ ok=true; break; }
    }
    if(!ok) best=legal[0];
    cout << format_move(pos, best) << "\n";
    return 0;
}
