using namespace std;

static inline int file_of(int sq) { return sq & 7; }
static inline int rank_of(int sq) { return sq >> 3; }
static inline int sq_of(int f, int r) { return r * 8 + f; }

static inline string sq_to_str(int sq) {
    string s;
    s.push_back(char('a' + file_of(sq)));
    s.push_back(char('1' + rank_of(sq)));
    return s;
}

static inline bool is_white_piece(char p) { return p >= 'A' && p <= 'Z'; }
static inline bool is_black_piece(char p) { return p >= 'a' && p <= 'z'; }
static inline bool is_empty(char p) { return p == '.'; }

enum Piece {
    EMPTY = 0,
    WPAWN, WKNIGHT, WBISHOP, WROOK, WQUEEN, WKING,
    BPAWN, BKNIGHT, BBISHOP, BROOK, BQUEEN, BKING
};

static inline Piece char_to_piece(char c) {
    switch (c) {
    case 'P': return WPAWN;
    case 'N': return WKNIGHT;
    case 'B': return WBISHOP;
    case 'R': return WROOK;
    case 'Q': return WQUEEN;
    case 'K': return WKING;
    case 'p': return BPAWN;
    case 'n': return BKNIGHT;
    case 'b': return BBISHOP;
    case 'r': return BROOK;
    case 'q': return BQUEEN;
    case 'k': return BKING;
    default: return EMPTY;
    }
}

static inline char piece_to_char(Piece p) {
    switch (p) {
    case WPAWN: return 'P';
    case WKNIGHT: return 'N';
    case WBISHOP: return 'B';
    case WROOK: return 'R';
    case WQUEEN: return 'Q';
    case WKING: return 'K';
    case BPAWN: return 'p';
    case BKNIGHT: return 'n';
    case BBISHOP: return 'b';
    case BROOK: return 'r';
    case BQUEEN: return 'q';
    case BKING: return 'k';
    default: return '.';
    }
}

static inline bool is_white(Piece p) { return p >= WPAWN && p <= WKING; }
static inline bool is_black(Piece p) { return p >= BPAWN && p <= BKING; }
static inline bool same_color(Piece a, Piece b) {
    if (a == EMPTY || b == EMPTY) return false;
    return (is_white(a) && is_white(b)) || (is_black(a) && is_black(b));
}

struct Move {
    int from = -1;
    int to = -1;
    Piece promo = EMPTY; // promotion piece (color-specific)
    bool capture = false;
    bool ep = false;
    bool castleK = false;
    bool castleQ = false;
};

struct Undo {
    Piece captured = EMPTY;
    int epSquare = -1;
    uint8_t castling = 0;
    int halfmove = 0;
    int fullmove = 1;
};

struct Position {
    array<Piece, 64> board{};
    bool whiteToMove = true;
    uint8_t castling = 0; // bits: 1=K,2=Q,4=k,8=q
    int epSquare = -1; // target square for en passant capture (the square behind moved pawn)
    int halfmove = 0;
    int fullmove = 1;

    int kingSqW = -1;
    int kingSqB = -1;

    void recompute_kings() {
        kingSqW = kingSqB = -1;
        for (int i = 0; i < 64; i++) {
            if (board[i] == WKING) kingSqW = i;
            else if (board[i] == BKING) kingSqB = i;
        }
    }
};

static Position parse_fen(const string& fen) {
    Position pos;
    pos.board.fill(EMPTY);

    // Split by spaces
    vector<string> parts;
    {
        string cur;
        for (char c : fen) {
            if (c == ' ') {
                if (!cur.empty()) { parts.push_back(cur); cur.clear(); }
            } else cur.push_back(c);
        }
        if (!cur.empty()) parts.push_back(cur);
    }
    if (parts.size() < 4) {
        // minimal fallback
        pos.whiteToMove = true;
        pos.castling = 0;
        pos.epSquare = -1;
        pos.halfmove = 0;
        pos.fullmove = 1;
        return pos;
    }

    // Board
    const string& b = parts[0];
    int r = 7, f = 0;
    for (char c : b) {
        if (c == '/') {
            r--;
            f = 0;
            continue;
        }
        if (isdigit((unsigned char)c)) {
            int n = c - '0';
            for (int i = 0; i < n; i++) {
                if (f < 8 && r >= 0) pos.board[sq_of(f, r)] = EMPTY;
                f++;
            }
        } else {
            Piece p = char_to_piece(c);
            if (f < 8 && r >= 0) pos.board[sq_of(f, r)] = p;
            f++;
        }
    }

    // Side to move
    pos.whiteToMove = (parts[1] == "w");

    // Castling
    pos.castling = 0;
    if (parts[2] != "-") {
        for (char c : parts[2]) {
            if (c == 'K') pos.castling |= 1;
            else if (c == 'Q') pos.castling |= 2;
            else if (c == 'k') pos.castling |= 4;
            else if (c == 'q') pos.castling |= 8;
        }
    }

    // En passant
    pos.epSquare = -1;
    if (parts[3] != "-") {
        if (parts[3].size() == 2) {
            int ff = parts[3][0] - 'a';
            int rr = parts[3][1] - '1';
            if (ff >= 0 && ff < 8 && rr >= 0 && rr < 8) pos.epSquare = sq_of(ff, rr);
        }
    }

    // Halfmove / fullmove
    pos.halfmove = 0;
    pos.fullmove = 1;
    if (parts.size() >= 5) pos.halfmove = max(0, stoi(parts[4]));
    if (parts.size() >= 6) pos.fullmove = max(1, stoi(parts[5]));

    pos.recompute_kings();
    return pos;
}

static inline bool on_board(int sq) { return sq >= 0 && sq < 64; }

static bool is_square_attacked(const Position& pos, int sq, bool byWhite) {
    // Pawns
    int f = file_of(sq), r = rank_of(sq);
    if (byWhite) {
        // white pawns attack (f-1,r+1) and (f+1,r+1) => from those squares to sq
        int rfrom = r - 1;
        if (rfrom >= 0) {
            if (f - 1 >= 0) {
                int s = sq_of(f - 1, rfrom);
                if (pos.board[s] == WPAWN) return true;
            }
            if (f + 1 < 8) {
                int s = sq_of(f + 1, rfrom);
                if (pos.board[s] == WPAWN) return true;
            }
        }
    } else {
        int rfrom = r + 1;
        if (rfrom < 8) {
            if (f - 1 >= 0) {
                int s = sq_of(f - 1, rfrom);
                if (pos.board[s] == BPAWN) return true;
            }
            if (f + 1 < 8) {
                int s = sq_of(f + 1, rfrom);
                if (pos.board[s] == BPAWN) return true;
            }
        }
    }

    // Knights
    static const int kD[8][2] = {
        {1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}
    };
    for (auto& d : kD) {
        int nf = f + d[0], nr = r + d[1];
        if (nf>=0 && nf<8 && nr>=0 && nr<8) {
            Piece p = pos.board[sq_of(nf,nr)];
            if (byWhite && p == WKNIGHT) return true;
            if (!byWhite && p == BKNIGHT) return true;
        }
    }

    // King
    for (int df=-1; df<=1; df++) for (int dr=-1; dr<=1; dr++) if (df||dr) {
        int nf=f+df, nr=r+dr;
        if (nf>=0 && nf<8 && nr>=0 && nr<8) {
            Piece p = pos.board[sq_of(nf,nr)];
            if (byWhite && p == WKING) return true;
            if (!byWhite && p == BKING) return true;
        }
    }

    // Sliders
    auto ray = [&](int df, int dr, Piece bishop, Piece rook, Piece queen) -> bool {
        int nf=f+df, nr=r+dr;
        while (nf>=0 && nf<8 && nr>=0 && nr<8) {
            Piece p = pos.board[sq_of(nf,nr)];
            if (p != EMPTY) {
                if (byWhite) {
                    if (p == bishop || p == rook || p == queen) return true;
                } else {
                    if (p == bishop || p == rook || p == queen) return true;
                }
                return false;
            }
            nf += df; nr += dr;
        }
        return false;
    };

    if (byWhite) {
        // bishop/queen diagonals
        if (ray(1,1, WBISHOP, WROOK, WQUEEN)) return true;
        if (ray(1,-1, WBISHOP, WROOK, WQUEEN)) return true;
        if (ray(-1,1, WBISHOP, WROOK, WQUEEN)) return true;
        if (ray(-1,-1, WBISHOP, WROOK, WQUEEN)) return true;
        // rook/queen orthogonals
        if (ray(1,0, WBISHOP, WROOK, WQUEEN)) return true;
        if (ray(-1,0, WBISHOP, WROOK, WQUEEN)) return true;
        if (ray(0,1, WBISHOP, WROOK, WQUEEN)) return true;
        if (ray(0,-1, WBISHOP, WROOK, WQUEEN)) return true;
    } else {
        if (ray(1,1, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(1,-1, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(-1,1, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(-1,-1, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(1,0, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(-1,0, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(0,1, BBISHOP, BROOK, BQUEEN)) return true;
        if (ray(0,-1, BBISHOP, BROOK, BQUEEN)) return true;
    }

    return false;
}

static bool in_check(const Position& pos, bool white) {
    int ksq = white ? pos.kingSqW : pos.kingSqB;
    if (ksq < 0) return false;
    return is_square_attacked(pos, ksq, !white);
}

static void add_move(vector<Move>& mv, int from, int to, bool cap=false, Piece promo=EMPTY, bool ep=false) {
    Move m;
    m.from = from; m.to = to; m.capture = cap; m.promo = promo; m.ep = ep;
    mv.push_back(m);
}

static void generate_pseudo_legal(const Position& pos, vector<Move>& mv) {
    mv.clear();
    bool usWhite = pos.whiteToMove;

    for (int sq = 0; sq < 64; sq++) {
        Piece p = pos.board[sq];
        if (p == EMPTY) continue;
        if (usWhite && !is_white(p)) continue;
        if (!usWhite && !is_black(p)) continue;

        int f = file_of(sq), r = rank_of(sq);

        if (p == WPAWN || p == BPAWN) {
            int dir = (p == WPAWN) ? 1 : -1;
            int startRank = (p == WPAWN) ? 1 : 6;
            int promoRank = (p == WPAWN) ? 7 : 0;

            int oneR = r + dir;
            if (oneR >= 0 && oneR < 8) {
                int to = sq_of(f, oneR);
                if (pos.board[to] == EMPTY) {
                    if (oneR == promoRank) {
                        // promote to Q/R/B/N (best-first later)
                        add_move(mv, sq, to, false, (p==WPAWN)?WQUEEN:BQUEEN, false);
                        add_move(mv, sq, to, false, (p==WPAWN)?WROOK:BROOK, false);
                        add_move(mv, sq, to, false, (p==WPAWN)?WBISHOP:BBISHOP, false);
                        add_move(mv, sq, to, false, (p==WPAWN)?WKNIGHT:BKNIGHT, false);
                    } else {
                        add_move(mv, sq, to, false, EMPTY, false);
                        if (r == startRank) {
                            int twoR = r + 2*dir;
                            int to2 = sq_of(f, twoR);
                            if (pos.board[to2] == EMPTY) {
                                add_move(mv, sq, to2, false, EMPTY, false);
                            }
                        }
                    }
                }
            }

            // captures and en passant
            for (int df : {-1, 1}) {
                int nf = f + df;
                int nr = r + dir;
                if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
                int to = sq_of(nf, nr);
                if (to == pos.epSquare) {
                    add_move(mv, sq, to, true, EMPTY, true);
                } else {
                    Piece tp = pos.board[to];
                    if (tp != EMPTY && !same_color(p, tp)) {
                        if (nr == promoRank) {
                            add_move(mv, sq, to, true, (p==WPAWN)?WQUEEN:BQUEEN, false);
                            add_move(mv, sq, to, true, (p==WPAWN)?WROOK:BROOK, false);
                            add_move(mv, sq, to, true, (p==WPAWN)?WBISHOP:BBISHOP, false);
                            add_move(mv, sq, to, true, (p==WPAWN)?WKNIGHT:BKNIGHT, false);
                        } else {
                            add_move(mv, sq, to, true, EMPTY, false);
                        }
                    }
                }
            }
        } else if (p == WKNIGHT || p == BKNIGHT) {
            static const int d[8][2] = {
                {1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}
            };
            for (auto& dd : d) {
                int nf = f + dd[0], nr = r + dd[1];
                if (nf<0||nf>=8||nr<0||nr>=8) continue;
                int to = sq_of(nf,nr);
                Piece tp = pos.board[to];
                if (tp == EMPTY) add_move(mv, sq, to, false);
                else if (!same_color(p,tp)) add_move(mv, sq, to, true);
            }
        } else if (p == WBISHOP || p == BBISHOP || p == WROOK || p == BROOK || p == WQUEEN || p == BQUEEN) {
            vector<pair<int,int>> dirs;
            bool bishopLike = (p == WBISHOP || p == BBISHOP || p == WQUEEN || p == BQUEEN);
            bool rookLike = (p == WROOK || p == BROOK || p == WQUEEN || p == BQUEEN);
            if (bishopLike) {
                dirs.push_back({1,1}); dirs.push_back({1,-1}); dirs.push_back({-1,1}); dirs.push_back({-1,-1});
            }
            if (rookLike) {
                dirs.push_back({1,0}); dirs.push_back({-1,0}); dirs.push_back({0,1}); dirs.push_back({0,-1});
            }
            for (auto [df,dr] : dirs) {
                int nf=f+df, nr=r+dr;
                while (nf>=0&&nf<8&&nr>=0&&nr<8) {
                    int to=sq_of(nf,nr);
                    Piece tp=pos.board[to];
                    if (tp==EMPTY) {
                        add_move(mv,sq,to,false);
                    } else {
                        if (!same_color(p,tp)) add_move(mv,sq,to,true);
                        break;
                    }
                    nf+=df; nr+=dr;
                }
            }
        } else if (p == WKING || p == BKING) {
            for (int df=-1; df<=1; df++) for (int dr=-1; dr<=1; dr++) if (df||dr) {
                int nf=f+df, nr=r+dr;
                if (nf<0||nf>=8||nr<0||nr>=8) continue;
                int to=sq_of(nf,nr);
                Piece tp=pos.board[to];
                if (tp==EMPTY) add_move(mv,sq,to,false);
                else if (!same_color(p,tp)) add_move(mv,sq,to,true);
            }
            // castling
            if (usWhite) {
                if ((pos.castling & 1) && pos.board[sq_of(5,0)]==EMPTY && pos.board[sq_of(6,0)]==EMPTY) {
                    Move m; m.castleK=true; mv.push_back(m);
                }
                if ((pos.castling & 2) && pos.board[sq_of(3,0)]==EMPTY && pos.board[sq_of(2,0)]==EMPTY && pos.board[sq_of(1,0)]==EMPTY) {
                    Move m; m.castleQ=true; mv.push_back(m);
                }
            } else {
                if ((pos.castling & 4) && pos.board[sq_of(5,7)]==EMPTY && pos.board[sq_of(6,7)]==EMPTY) {
                    Move m; m.castleK=true; mv.push_back(m);
                }
                if ((pos.castling & 8) && pos.board[sq_of(3,7)]==EMPTY && pos.board[sq_of(2,7)]==EMPTY && pos.board[sq_of(1,7)]==EMPTY) {
                    Move m; m.castleQ=true; mv.push_back(m);
                }
            }
        }
    }
}

static bool make_move(Position& pos, const Move& m, Undo& u) {
    u.captured = EMPTY;
    u.epSquare = pos.epSquare;
    u.castling = pos.castling;
    u.halfmove = pos.halfmove;
    u.fullmove = pos.fullmove;

    bool usWhite = pos.whiteToMove;

    pos.epSquare = -1;

    if (m.castleK || m.castleQ) {
        int r = usWhite ? 0 : 7;
        int kFrom = sq_of(4,r);
        int kTo   = sq_of(m.castleK ? 6 : 2, r);
        int rookFrom = sq_of(m.castleK ? 7 : 0, r);
        int rookTo   = sq_of(m.castleK ? 5 : 3, r);

        // legality: cannot castle out of/through/into check
        if (in_check(pos, usWhite)) return false;
        // squares crossed by king
        int step1 = sq_of(m.castleK ? 5 : 3, r);
        Position tmp = pos;
        // move king to step1, test
        tmp.board[step1] = tmp.board[kFrom];
        tmp.board[kFrom] = EMPTY;
        tmp.recompute_kings();
        if (in_check(tmp, usWhite)) return false;
        // move king to final, test
        tmp.board[kTo] = tmp.board[step1];
        tmp.board[step1] = EMPTY;
        tmp.recompute_kings();
        if (in_check(tmp, usWhite)) return false;

        // execute
        pos.board[kTo] = pos.board[kFrom];
        pos.board[kFrom] = EMPTY;
        pos.board[rookTo] = pos.board[rookFrom];
        pos.board[rookFrom] = EMPTY;

        // update castling rights
        if (usWhite) pos.castling &= ~uint8_t(1|2);
        else pos.castling &= ~uint8_t(4|8);

        pos.recompute_kings();
        pos.whiteToMove = !pos.whiteToMove;
        pos.halfmove++;
        if (!pos.whiteToMove) pos.fullmove++; // after black move
        return true;
    }

    int from = m.from, to = m.to;
    Piece moving = pos.board[from];
    if (moving == EMPTY) return false;

    // halfmove reset on pawn move or capture
    bool pawnMove = (moving == WPAWN || moving == BPAWN);
    pos.halfmove = (pawnMove || m.capture) ? 0 : (pos.halfmove + 1);

    // capture
    if (m.ep) {
        // capture pawn behind target square
        int capSq = to + (usWhite ? -8 : 8);
        u.captured = pos.board[capSq];
        pos.board[capSq] = EMPTY;
    } else {
        u.captured = pos.board[to];
    }

    // move piece
    pos.board[to] = moving;
    pos.board[from] = EMPTY;

    // promotion
    if (m.promo != EMPTY) {
        pos.board[to] = m.promo;
    }

    // set en passant square if double pawn push
    if (pawnMove) {
        int fr = rank_of(from), tr = rank_of(to);
        if (abs(tr - fr) == 2) {
            pos.epSquare = (from + to) / 2;
        }
    }

    // update castling rights if king/rook moves or rook captured
    if (moving == WKING) pos.castling &= ~uint8_t(1|2);
    if (moving == BKING) pos.castling &= ~uint8_t(4|8);
    if (moving == WROOK) {
        if (from == sq_of(0,0)) pos.castling &= ~uint8_t(2);
        if (from == sq_of(7,0)) pos.castling &= ~uint8_t(1);
    }
    if (moving == BROOK) {
        if (from == sq_of(0,7)) pos.castling &= ~uint8_t(8);
        if (from == sq_of(7,7)) pos.castling &= ~uint8_t(4);
    }
    if (u.captured == WROOK) {
        if (to == sq_of(0,0)) pos.castling &= ~uint8_t(2);
        if (to == sq_of(7,0)) pos.castling &= ~uint8_t(1);
    }
    if (u.captured == BROOK) {
        if (to == sq_of(0,7)) pos.castling &= ~uint8_t(8);
        if (to == sq_of(7,7)) pos.castling &= ~uint8_t(4);
    }

    pos.recompute_kings();

    // illegal if leaves own king in check
    if (in_check(pos, usWhite)) {
        // undo quickly
        // (call unmake_move logic inline)
        pos.whiteToMove = usWhite;
        pos.board[from] = moving;
        pos.board[to] = (m.capture && !m.ep) ? u.captured : EMPTY;
        if (m.promo != EMPTY) pos.board[from] = moving; // already
        if (m.ep) {
            int capSq = to + (usWhite ? -8 : 8);
            pos.board[capSq] = u.captured;
            pos.board[to] = EMPTY;
        }
        pos.epSquare = u.epSquare;
        pos.castling = u.castling;
        pos.halfmove = u.halfmove;
        pos.fullmove = u.fullmove;
        pos.recompute_kings();
        return false;
    }

    pos.whiteToMove = !pos.whiteToMove;
    if (!pos.whiteToMove) pos.fullmove++; // increment after black move
    return true;
}

static void unmake_move(Position& pos, const Move& m, const Undo& u, bool prevWhiteToMove) {
    pos.whiteToMove = prevWhiteToMove;
    pos.epSquare = u.epSquare;
    pos.castling = u.castling;
    pos.halfmove = u.halfmove;
    pos.fullmove = u.fullmove;

    bool usWhite = prevWhiteToMove;

    if (m.castleK || m.castleQ) {
        int r = usWhite ? 0 : 7;
        int kFrom = sq_of(4,r);
        int kTo   = sq_of(m.castleK ? 6 : 2, r);
        int rookFrom = sq_of(m.castleK ? 7 : 0, r);
        int rookTo   = sq_of(m.castleK ? 5 : 3, r);

        pos.board[kFrom] = pos.board[kTo];
        pos.board[kTo] = EMPTY;
        pos.board[rookFrom] = pos.board[rookTo];
        pos.board[rookTo] = EMPTY;
        pos.recompute_kings();
        return;
    }

    Piece moved = pos.board[m.to];
    // if promotion, restore pawn
    if (m.promo != EMPTY) {
        moved = usWhite ? WPAWN : BPAWN;
    }

    pos.board[m.from] = moved;

    if (m.ep) {
        pos.board[m.to] = EMPTY;
        int capSq = m.to + (usWhite ? -8 : 8);
        pos.board[capSq] = u.captured;
    } else {
        pos.board[m.to] = u.captured;
    }

    pos.recompute_kings();
}

static int piece_value(Piece p) {
    switch (p) {
    case WPAWN: case BPAWN: return 100;
    case WKNIGHT: case BKNIGHT: return 320;
    case WBISHOP: case BBISHOP: return 330;
    case WROOK: case BROOK: return 500;
    case WQUEEN: case BQUEEN: return 900;
    case WKING: case BKING: return 0;
    default: return 0;
    }
}

static int evaluate(const Position& pos) {
    // simple material + small mobility, from side-to-move perspective later
    int score = 0;
    for (int i = 0; i < 64; i++) {
        Piece p = pos.board[i];
        if (p == EMPTY) continue;
        int v = piece_value(p);
        score += is_white(p) ? v : -v;
    }

    // bonus: king safety (avoid being in check)
    if (in_check(pos, true)) score -= 30;
    if (in_check(pos, false)) score += 30;

    return score;
}

static uint64_t splitmix64(uint64_t& x) {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
}

struct Zobrist {
    uint64_t piece[13][64]{};
    uint64_t side{};
    uint64_t castling[16]{};
    uint64_t epFile[9]{}; // 0..7 files, 8 = no ep
    Zobrist() {
        uint64_t seed = 0x123456789abcdef0ULL;
        for (int p = 0; p < 13; p++)
            for (int s = 0; s < 64; s++)
                piece[p][s] = splitmix64(seed);
        side = splitmix64(seed);
        for (int i = 0; i < 16; i++) castling[i] = splitmix64(seed);
        for (int i = 0; i < 9; i++) epFile[i] = splitmix64(seed);
    }
    uint64_t hash(const Position& pos) const {
        uint64_t h = 0;
        for (int s = 0; s < 64; s++) {
            Piece p = pos.board[s];
            if (p != EMPTY) h ^= piece[(int)p][s];
        }
        if (pos.whiteToMove) h ^= side;
        h ^= castling[pos.castling & 15];
        int ef = 8;
        if (pos.epSquare != -1) ef = file_of(pos.epSquare);
        h ^= epFile[ef];
        return h;
    }
};

struct TTEntry {
    uint64_t key = 0;
    int depth = -1;
    int score = 0;
    uint8_t flag = 0; // 0=exact,1=lower,2=upper
    Move best{};
};

static constexpr int INF = 1'000'000;
static constexpr int MATE = 900'000;

static inline int score_mate_in(int ply) { return MATE - ply; }
static inline int score_mated_in(int ply) { return -MATE + ply; }

static int mvv_lva(Piece attacker, Piece victim) {
    return 10 * piece_value(victim) - piece_value(attacker);
}

static string move_to_notation(Position& pos, const Move& m) {
    if (m.castleK) return "O-O";
    if (m.castleQ) return "O-O-O";

    Piece moving = pos.board[m.from];
    char letter;
    if (moving == WPAWN || moving == BPAWN) letter = 'p';
    else {
        // Use uppercase piece letter per spec (N/B/R/Q/K)
        char c = piece_to_char(moving);
        letter = (char)toupper((unsigned char)c);
    }

    string s;
    s.push_back(letter);
    s += sq_to_str(m.from);

    if (m.capture) s.push_back('x');

    s += sq_to_str(m.to);

    if (m.ep) s += "ep";

    if (m.promo != EMPTY) {
        s.push_back('/');
        char pc = piece_to_char(m.promo);
        s.push_back((char)toupper((unsigned char)pc));
    }

    // append check/mate by making move temporarily
    Undo u;
    bool prev = pos.whiteToMove;
    Position tmp = pos;
    if (make_move(tmp, m, u)) {
        bool oppWhite = tmp.whiteToMove;
        vector<Move> replies, legalReplies;
        generate_pseudo_legal(tmp, replies);
        for (auto& rm : replies) {
            Undo u2;
            Position t2 = tmp;
            if (make_move(t2, rm, u2)) legalReplies.push_back(rm);
        }
        bool oppInCheck = in_check(tmp, oppWhite);
        if (oppInCheck) {
            if (legalReplies.empty()) s.push_back('#');
            else s.push_back('+');
        }
    }

    return s;
}

struct Searcher {
    Zobrist zob;
    vector<TTEntry> tt;
    atomic<bool>* stop = nullptr;
    chrono::steady_clock::time_point endTime;
    int nodes = 0;

    Searcher(size_t ttSize = (1u << 22)) { // ~4 million entries
        tt.resize(ttSize);
    }

    TTEntry* probe(uint64_t key) {
        TTEntry& e = tt[(size_t)key % tt.size()];
        if (e.key == key) return &e;
        return nullptr;
    }
    void store(uint64_t key, int depth, int score, uint8_t flag, const Move& best) {
        TTEntry& e = tt[(size_t)key % tt.size()];
        if (e.key != key || depth >= e.depth) {
            e.key = key; e.depth = depth; e.score = score; e.flag = flag; e.best = best;
        }
    }

    bool time_up() const {
        if (stop && stop->load(memory_order_relaxed)) return true;
        return chrono::steady_clock::now() >= endTime;
    }

    int qsearch(Position& pos, int alpha, int beta, int ply) {
        if (time_up()) return 0;
        nodes++;

        int stand = evaluate(pos);
        int perspective = pos.whiteToMove ? stand : -stand;
        if (perspective >= beta) return beta;
        alpha = max(alpha, perspective);

        vector<Move> moves;
        generate_pseudo_legal(pos, moves);

        // captures/promotions only
        vector<pair<int, Move>> ordered;
        ordered.reserve(moves.size());
        for (auto& m : moves) {
            if (m.castleK || m.castleQ) continue;
            if (!m.capture && m.promo == EMPTY) continue;
            Piece attacker = pos.board[m.from];
            Piece victim = m.ep ? (pos.whiteToMove ? BPAWN : WPAWN) : pos.board[m.to];
            int sc = 0;
            if (m.promo != EMPTY) sc += 800;
            if (m.capture) sc += mvv_lva(attacker, victim);
            ordered.push_back({-sc, m});
        }
        sort(ordered.begin(), ordered.end(), [](auto& a, auto& b){ return a.first < b.first; });

        for (auto& it : ordered) {
            Move m = it.second;
            Undo u;
            bool prev = pos.whiteToMove;
            if (!make_move(pos, m, u)) continue;
            int score = -qsearch(pos, -beta, -alpha, ply+1);
            unmake_move(pos, m, u, prev);

            if (score >= beta) return beta;
            alpha = max(alpha, score);
        }
        return alpha;
    }

    int negamax(Position& pos, int depth, int alpha, int beta, int ply) {
        if (time_up()) return 0;
        nodes++;

        uint64_t key = zob.hash(pos);
        if (auto* e = probe(key)) {
            if (e->depth >= depth) {
                if (e->flag == 0) return e->score;
                if (e->flag == 1) alpha = max(alpha, e->score);
                else if (e->flag == 2) beta = min(beta, e->score);
                if (alpha >= beta) return e->score;
            }
        }

        bool usWhite = pos.whiteToMove;

        if (depth == 0) {
            return qsearch(pos, alpha, beta, ply);
        }

        vector<Move> moves;
        generate_pseudo_legal(pos, moves);

        // create legal moves list with ordering (TT move first, then captures, promotions, checks-ish)
        Move ttBest{};
        bool hasTT = false;
        if (auto* e = probe(key)) { ttBest = e->best; hasTT = true; }

        vector<pair<int, Move>> ordered;
        ordered.reserve(moves.size());
        for (auto& m : moves) {
            int sc = 0;
            if (hasTT && ((m.castleK==ttBest.castleK && m.castleQ==ttBest.castleQ && m.from==ttBest.from && m.to==ttBest.to && m.promo==ttBest.promo && m.ep==ttBest.ep)))
                sc += 1'000'000;
            if (m.promo != EMPTY) sc += 200'000;
            if (m.capture) {
                Piece attacker = (m.castleK||m.castleQ) ? EMPTY : pos.board[m.from];
                Piece victim = m.ep ? (usWhite ? BPAWN : WPAWN) : ((m.castleK||m.castleQ)?EMPTY:pos.board[m.to]);
                sc += 50'000 + mvv_lva(attacker, victim);
            }
            if (m.castleK || m.castleQ) sc += 10'000;
            ordered.push_back({-sc, m});
        }
        sort(ordered.begin(), ordered.end(), [](auto& a, auto& b){ return a.first < b.first; });

        int legalCount = 0;
        int bestScore = -INF;
        Move bestMove{};
        int alpha0 = alpha;

        for (auto& it : ordered) {
            Move m = it.second;
            Undo u;
            bool prev = pos.whiteToMove;
            if (!make_move(pos, m, u)) continue;
            legalCount++;

            int score = -negamax(pos, depth-1, -beta, -alpha, ply+1);
            unmake_move(pos, m, u, prev);

            if (time_up()) break;

            if (score > bestScore) { bestScore = score; bestMove = m; }
            alpha = max(alpha, score);
            if (alpha >= beta) break;
        }

        if (legalCount == 0) {
            // checkmate or stalemate from side to move
            if (in_check(pos, usWhite)) return score_mated_in(ply);
            return 0; // stalemate
        }

        uint8_t flag = 0;
        if (bestScore <= alpha0) flag = 2; // upper
        else if (bestScore >= beta) flag = 1; // lower
        store(key, depth, bestScore, flag, bestMove);
        return bestScore;
    }

    Move search_best(Position root, double seconds, atomic<bool>& stopFlag) {
        stop = &stopFlag;
        endTime = chrono::steady_clock::now() + chrono::milliseconds((int)(seconds * 1000.0));

        vector<Move> moves;
        generate_pseudo_legal(root, moves);

        // collect legal moves
        vector<Move> legal;
        legal.reserve(moves.size());
        for (auto& m : moves) {
            Position t = root;
            Undo u;
            if (make_move(t, m, u)) legal.push_back(m);
        }
        if (legal.empty()) {
            // no legal move: output something (shouldn't happen in valid tasks)
            return Move{};
        }

        // iterative deepening
        Move best = legal[0];
        int bestScore = -INF;

        for (int depth = 1; ; depth++) {
            if (time_up()) break;

            // Aspiration window
            int alpha = -INF, beta = INF;
            if (depth >= 3 && bestScore > -INF/2) {
                alpha = bestScore - 50;
                beta  = bestScore + 50;
            }

            // root move ordering: prefer previous best
            vector<pair<int, Move>> ordered;
            for (auto& m : legal) {
                int sc = 0;
                if (m.from == best.from && m.to == best.to && m.promo == best.promo && m.castleK==best.castleK && m.castleQ==best.castleQ && m.ep==best.ep)
                    sc += 1'000'000;
                if (m.promo != EMPTY) sc += 200'000;
                if (m.capture) {
                    Piece attacker = root.board[m.from];
                    Piece victim = m.ep ? (root.whiteToMove ? BPAWN : WPAWN) : root.board[m.to];
                    sc += 50'000 + mvv_lva(attacker, victim);
                }
                if (m.castleK || m.castleQ) sc += 10'000;
                ordered.push_back({-sc, m});
            }
            sort(ordered.begin(), ordered.end(), [](auto& a, auto& b){ return a.first < b.first; });

            Move iterBest = best;
            int iterBestScore = -INF;

            auto search_with_window = [&](int a, int b)->pair<int,Move> {
                int localBestScore = -INF;
                Move localBest = ordered[0].second;

                for (auto& it : ordered) {
                    Move m = it.second;
                    Position t = root;
                    Undo u;
                    if (!make_move(t, m, u)) continue;

                    int score = -negamax(t, depth-1, -b, -a, 1);

                    if (time_up()) break;

                    if (score > localBestScore) { localBestScore = score; localBest = m; }
                    a = max(a, score);
                    if (a >= b) break;
                }
                return {localBestScore, localBest};
            };

            auto [sc1, mv1] = search_with_window(alpha, beta);
            if (!time_up() && (sc1 <= alpha || sc1 >= beta)) {
                // re-search with full window
                auto [sc2, mv2] = search_with_window(-INF, INF);
                sc1 = sc2; mv1 = mv2;
            }

            if (time_up()) break;

            iterBestScore = sc1;
            iterBest = mv1;

            best = iterBest;
            bestScore = iterBestScore;
        }

        return best;
    }
};

static string format_move(Position pos, const Move& m) {
    // pos passed by value so we can use it in check/mate suffix generation safely
    if (m.castleK || m.castleQ) return m.castleK ? "O-O" : "O-O-O";
    return move_to_notation(pos, m);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSeconds;
    string fen;
    if (!(cin >> timeoutSeconds)) return 0;
    getline(cin, fen); // consume endline
    getline(cin, fen);

    Position pos = parse_fen(fen);

    // Use ~90% of time budget for safety
    double think = max(0.1, timeoutSeconds * 0.90);

    atomic<bool> stopFlag{false};

    // Optional stopper thread to guarantee stop near timeout
    auto start = chrono::steady_clock::now();
    thread stopper([&](){
        auto endt = start + chrono::milliseconds((int)(think * 1000.0));
        while (chrono::steady_clock::now() < endt) {
            this_thread::sleep_for(chrono::milliseconds(2));
        }
        stopFlag.store(true, memory_order_relaxed);
    });

    Searcher searcher;
    Move best = searcher.search_best(pos, think, stopFlag);

    stopFlag.store(true, memory_order_relaxed);
    if (stopper.joinable()) stopper.join();

    string out = format_move(pos, best);
    if (out.empty()) out = "O-O"; // fallback (unlikely)
    cout << out << "\n";
    return 0;
}
