#include <bits/stdc++.h>
using namespace std;

// Exact Maximum Independent Set (MIS) for general graphs.
//
// Approach:
// - If n <= 64: solve MIS directly with branch & bound on 64-bit masks.
// - Else: MIS in G equals maximum clique in complement graph Gc.
//         Solve maximum clique in Gc with Tomita-style branch & bound using
//         greedy coloring upper bounds and dynamic bitsets.

// ---------------- small n (<=64) MIS ----------------
struct MIS64 {
    int n;
    vector<uint64_t> adj; // adjacency in original graph, within n bits
    uint64_t bestMask = 0;
    int bestSize = 0;

    explicit MIS64(int n_) : n(n_), adj(n_, 0) {}

    void addEdge(int u, int v) {
        if (u == v) return;
        adj[u] |= (1ULL << v);
        adj[v] |= (1ULL << u);
    }

    static inline int popcount64(uint64_t x) {
#ifdef _MSC_VER
        return (int)__popcnt64(x);
#else
        return __builtin_popcountll(x);
#endif
    }

    void dfs(uint64_t cand, uint64_t chosen, int chosenSize) {
        if (chosenSize + popcount64(cand) <= bestSize) return;
        if (!cand) {
            if (chosenSize > bestSize) {
                bestSize = chosenSize;
                bestMask = chosen;
            }
            return;
        }
        // pick vertex with maximum degree inside cand
        int pick = -1, bestDeg = -1;
        uint64_t tmp = cand;
        while (tmp) {
            uint64_t lsb = tmp & -tmp;
#ifdef _MSC_VER
            unsigned long idx;
            _BitScanForward64(&idx, lsb);
            int v = (int)idx;
#else
            int v = __builtin_ctzll(lsb);
#endif
            int deg = popcount64(adj[v] & cand);
            if (deg > bestDeg) { bestDeg = deg; pick = v; }
            tmp ^= lsb;
        }
        // include pick
        dfs(cand & ~((1ULL << pick) | adj[pick]), chosen | (1ULL << pick), chosenSize + 1);
        // exclude pick
        dfs(cand & ~(1ULL << pick), chosen, chosenSize);
    }

    vector<int> solve() {
        uint64_t cand = (n == 64) ? ~0ULL : ((1ULL << n) - 1ULL);
        dfs(cand, 0, 0);
        vector<int> res;
        for (int i = 0; i < n; i++) if ((bestMask >> i) & 1ULL) res.push_back(i);
        return res;
    }
};

// ---------------- dynamic bitset helpers ----------------
struct Bitset {
    int n = 0;
    int blocks = 0;
    vector<uint64_t> w;

    Bitset() = default;
    explicit Bitset(int n_) { init(n_); }
    void init(int n_) {
        n = n_;
        blocks = (n + 63) >> 6;
        w.assign(blocks, 0);
    }
    inline void set1(int i) { w[i >> 6] |= (1ULL << (i & 63)); }
    inline void reset1(int i) { w[i >> 6] &= ~(1ULL << (i & 63)); }
    inline bool any() const { for (uint64_t x : w) if (x) return true; return false; }
};

static inline Bitset band(const Bitset& a, const Bitset& b) {
    Bitset r;
    r.n = a.n; r.blocks = a.blocks;
    r.w.resize(a.blocks);
    for (int i = 0; i < a.blocks; i++) r.w[i] = a.w[i] & b.w[i];
    return r;
}
static inline void bsub_inplace(Bitset& a, const Bitset& b) {
    for (int i = 0; i < a.blocks; i++) a.w[i] &= ~b.w[i];
}

static inline int pop_lsb(Bitset& b) {
    for (int i = 0; i < b.blocks; i++) {
        uint64_t x = b.w[i];
        if (!x) continue;
#ifdef _MSC_VER
        unsigned long idx;
        _BitScanForward64(&idx, x);
        b.w[i] = x & (x - 1);
        return (i << 6) + (int)idx;
#else
        int idx = __builtin_ctzll(x);
        b.w[i] = x & (x - 1);
        return (i << 6) + idx;
#endif
    }
    return -1;
}

static inline int popcount_bitset(const Bitset& b) {
    int c = 0;
    for (uint64_t x : b.w) {
#ifdef _MSC_VER
        c += (int)__popcnt64(x);
#else
        c += __builtin_popcountll(x);
#endif
    }
    return c;
}

static inline vector<int> to_vertices(const Bitset& b, int n) {
    vector<int> vs;
    vs.reserve((size_t)popcount_bitset(b));
    for (int bi = 0; bi < b.blocks; bi++) {
        uint64_t x = b.w[bi];
        while (x) {
            uint64_t lsb = x & -x;
#ifdef _MSC_VER
            unsigned long idx;
            _BitScanForward64(&idx, lsb);
            int bit = (int)idx;
#else
            int bit = __builtin_ctzll(lsb);
#endif
            int v = (bi << 6) + bit;
            if (v < n) vs.push_back(v);
            x ^= lsb;
        }
    }
    return vs;
}

// ---------------- maximum clique on complement ----------------
struct MaxClique {
    int n;
    vector<Bitset> adj; // adjacency in complement graph
    vector<int> best;

    explicit MaxClique(int n_) : n(n_), adj(n_, Bitset(n_)) {}

    void color_sort(const Bitset& P, vector<int>& order, vector<int>& bounds) {
        // Greedy coloring of the subgraph induced by P.
        // bounds[i] = number of colors used up to i (upper bound for clique size in suffix).
        Bitset U = P;
        vector<vector<int>> classes;
        classes.reserve(64);

        while (U.any()) {
            Bitset Q = U;
            vector<int> cls;
            while (Q.any()) {
                int v = pop_lsb(Q);
                cls.push_back(v);
                // remove neighbors of v from Q to keep cls independent (same color)
                bsub_inplace(Q, adj[v]);
            }
            for (int v : cls) U.reset1(v);
            classes.push_back(move(cls));
        }

        order.clear();
        bounds.clear();
        int color = 0;
        for (auto& cls : classes) {
            ++color;
            for (int v : cls) {
                order.push_back(v);
                bounds.push_back(color);
            }
        }
    }

    void expand(Bitset P, vector<int>& C) {
        if (!P.any()) {
            if (C.size() > best.size()) best = C;
            return;
        }

        vector<int> order, bounds;
        color_sort(P, order, bounds);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            int v = order[i];
            if (C.size() + (size_t)bounds[i] <= best.size()) return;
            C.push_back(v);
            Bitset P2 = band(P, adj[v]);
            expand(P2, C);
            C.pop_back();
            P.reset1(v);
        }
    }

    vector<int> solve() {
        best.clear();
        Bitset P(n);
        for (int i = 0; i < n; i++) P.set1(i);
        vector<int> C;
        expand(P, C);
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<pair<int,int>> edges;
    edges.reserve((size_t)m);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        if (u == v) continue;
        edges.push_back({u, v});
    }

    vector<int> indep;
    if (n <= 64) {
        MIS64 s(n);
        for (auto [u, v] : edges) s.addEdge(u, v);
        indep = s.solve();
    } else {
        // Build complement adjacency without O(n^2) initialization:
        // Start with all-ones rows, then clear self and original edges.
        vector<Bitset> comp(n, Bitset(n));
        int blocks = (n + 63) >> 6;
        for (int i = 0; i < n; i++) {
            comp[i].w.assign(blocks, ~0ULL);
            // clear bits beyond n in last block
            int rem = n & 63;
            if (rem) comp[i].w[blocks - 1] &= ((1ULL << rem) - 1ULL);
            comp[i].reset1(i);
        }
        for (auto [u, v] : edges) {
            comp[u].reset1(v);
            comp[v].reset1(u);
        }

        MaxClique mc(n);
        mc.adj = move(comp);
        vector<int> clique = mc.solve();
        vector<char> inClique(n, 0);
        for (int v : clique) inClique[v] = 1;
        indep.reserve((size_t)(n - (int)clique.size()));
        for (int i = 0; i < n; i++) if (!inClique[i]) indep.push_back(i);
    }

    cout << indep.size() << "\n";
    for (size_t i = 0; i < indep.size(); i++) {
        if (i) cout << ' ';
        cout << indep[i];
    }
    cout << "\n";
    return 0;
}
