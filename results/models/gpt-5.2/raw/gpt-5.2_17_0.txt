using System;using System.Collections.Generic;using System.Globalization;using System.Runtime.CompilerServices;// 3D Bin Packing (axis-aligned) heuristic solver.// Outputs a valid (non-overlapping, in-bounds) packing; aims to maximize packed count.//// Heuristic:// - Maintain a bounded list of "empty maximal spaces" (EMS) axis-aligned boxes.// - Place boxes greedily (large-first) into best EMS among a candidate prefix.// - After placement, split used EMS into a small set of new EMS.// - Periodically prune contained/degenerate spaces and cap list size.//// Boxes can be rotated by permuting (w,h,d).public static class Program{    private struct Space    {        public int X, Y, Z;        public int W, H, D;        public Space(int x, int y, int z, int w, int h, int d) { X = x; Y = y; Z = z; W = w; H = h; D = d; }        public int Volume => W * H * D;    }    private struct Box    {        public int Index;        public int W, H, D;        public int Volume;    }    private struct Placement    {        public int BoxIndex;        public int X, Y, Z;    }    private sealed class FastScanner    {        private readonly byte[] _data;        private int _p;        public FastScanner() { _data = System.Text.Encoding.ASCII.GetBytes(Console.In.ReadToEnd()); _p = 0; }        [MethodImpl(MethodImplOptions.AggressiveInlining)]        private void Skip()        {            while (_p < _data.Length)            {                byte c = _data[_p];                if (c > 32) break;                _p++;            }        }        [MethodImpl(MethodImplOptions.AggressiveInlining)]        public int NextInt()        {            Skip();            int sign = 1;            if (_p < _data.Length && _data[_p] == (byte)'-') { sign = -1; _p++; }            int val = 0;            while (_p < _data.Length)            {                byte c = _data[_p];                if (c <= 32) break;                val = val * 10 + (c - (byte)'0');                _p++;            }            return val * sign;        }    }    public static void Main()    {        var fs = new FastScanner();        int n;        try { n = fs.NextInt(); } catch { return; }        int containerW = fs.NextInt();        int containerH = fs.NextInt();        int containerD = fs.NextInt();        var boxes = new Box[n];        for (int i = 0; i < n; i++)        {            int w = fs.NextInt();            int h = fs.NextInt();            int d = fs.NextInt();            boxes[i] = new Box { Index = i, W = w, H = h, D = d, Volume = w * h * d };        }        if (containerW <= 0 || containerH <= 0 || containerD <= 0 || n <= 0)        {            Console.WriteLine(0);            return;        }        int maxSpaces = n >= 20000 ? 800 : (n >= 5000 ? 1400 : (n >= 1000 ? 2600 : 6000));        int maxCandidateSpacesPerBox = n >= 20000 ? 24 : (n >= 5000 ? 40 : (n >= 1000 ? 64 : 128));        Array.Sort(boxes, (a, b) =>        {            int c = b.Volume.CompareTo(a.Volume);            if (c != 0) return c;            int am = a.W; if (a.H > am) am = a.H; if (a.D > am) am = a.D;            int bm = b.W; if (b.H > bm) bm = b.H; if (b.D > bm) bm = b.D;            c = bm.CompareTo(am);            if (c != 0) return c;            c = b.W.CompareTo(a.W);            if (c != 0) return c;            c = b.H.CompareTo(a.H);            if (c != 0) return c;            return b.D.CompareTo(a.D);        });        var spaces = new List<Space>(maxSpaces + 64) { new Space(0, 0, 0, containerW, containerH, containerD) };        var placements = new List<Placement>(Math.Min(n, 1 << 16));        Pack(boxes, spaces, placements, maxSpaces, maxCandidateSpacesPerBox);        if (n <= 5000)        {            var packed = new bool[n];            for (int i = 0; i < placements.Count; i++) packed[placements[i].BoxIndex] = true;            var leftover = new List<Box>();            for (int i = 0; i < boxes.Length; i++)                if (!packed[boxes[i].Index]) leftover.Add(boxes[i]);            if (leftover.Count > 0 && spaces.Count > 0)            {                leftover.Sort((a, b) =>                {                    int c = a.Volume.CompareTo(b.Volume);                    if (c != 0) return c;                    int am = a.W; if (a.H > am) am = a.H; if (a.D > am) am = a.D;                    int bm = b.W; if (b.H > bm) bm = b.H; if (b.D > bm) bm = b.D;                    c = am.CompareTo(bm);                    if (c != 0) return c;                    c = a.W.CompareTo(b.W);                    if (c != 0) return c;                    c = a.H.CompareTo(b.H);                    if (c != 0) return c;                    return a.D.CompareTo(b.D);                });                Pack(leftover.ToArray(), spaces, placements, maxSpaces, Math.Max(16, maxCandidateSpacesPerBox / 2));            }        }        Console.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));        for (int i = 0; i < placements.Count; i++)        {            var p = placements[i];            Console.Write(p.BoxIndex);            Console.Write(' ');            Console.Write(p.X);            Console.Write(' ');            Console.Write(p.Y);            Console.Write(' ');            Console.WriteLine(p.Z);        }    }    private static void Pack(Box[] boxes, List<Space> spaces, List<Placement> placements, int maxSpaces, int maxCandidateSpacesPerBox)    {        int opsSinceClean = 0;        int cleanEvery = 256;        for (int bi = 0; bi < boxes.Length; bi++)        {            if (spaces.Count == 0) break;            var box = boxes[bi];            int bestSpaceIdx = -1;            int bestW = 0, bestH = 0, bestD = 0;            long bestScore = long.MaxValue;            int limit = spaces.Count;            if (limit > maxCandidateSpacesPerBox) limit = maxCandidateSpacesPerBox;            for (int si = 0; si < limit; si++)            {                var sp = spaces[si];                TryOrientation(box.W, box.H, box.D, sp, si, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore);                TryOrientation(box.W, box.D, box.H, sp, si, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore);                TryOrientation(box.H, box.W, box.D, sp, si, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore);                TryOrientation(box.H, box.D, box.W, sp, si, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore);                TryOrientation(box.D, box.W, box.H, sp, si, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore);                TryOrientation(box.D, box.H, box.W, sp, si, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore);            }            if (bestSpaceIdx < 0) continue;            var target = spaces[bestSpaceIdx];            placements.Add(new Placement { BoxIndex = box.Index, X = target.X, Y = target.Y, Z = target.Z });            int last = spaces.Count - 1;            spaces[bestSpaceIdx] = spaces[last];            spaces.RemoveAt(last);            int rw = target.W - bestW;            int th = target.H - bestH;            int fd = target.D - bestD;            if (rw > 0) spaces.Add(new Space(target.X + bestW, target.Y, target.Z, rw, bestH, bestD));            if (th > 0) spaces.Add(new Space(target.X, target.Y + bestH, target.Z, target.W, th, bestD));            if (fd > 0) spaces.Add(new Space(target.X, target.Y, target.Z + bestD, target.W, target.H, fd));            if (rw > 0 && target.H - bestH > 0)                spaces.Add(new Space(target.X + bestW, target.Y + bestH, target.Z, rw, target.H - bestH, bestD));            if (rw > 0 && target.D - bestD > 0)                spaces.Add(new Space(target.X + bestW, target.Y, target.Z + bestD, rw, target.H, target.D - bestD));            if (th > 0 && target.D - bestD > 0)                spaces.Add(new Space(target.X, target.Y + bestH, target.Z + bestD, target.W, th, target.D - bestD));            opsSinceClean++;            if (opsSinceClean >= cleanEvery)            {                opsSinceClean = 0;                CleanSpaces(spaces, maxSpaces);            }            else if (spaces.Count > maxSpaces * 3 / 2)            {                CleanSpaces(spaces, maxSpaces);            }        }        if (spaces.Count > maxSpaces) CleanSpaces(spaces, maxSpaces);    }    [MethodImpl(MethodImplOptions.AggressiveInlining)]    private static void TryOrientation(int bw, int bh, int bd, Space sp, int spIdx,        ref int bestSpaceIdx, ref int bestW, ref int bestH, ref int bestD, ref long bestScore)    {        if (bw <= 0 || bh <= 0 || bd <= 0) return;        if (bw > sp.W || bh > sp.H || bd > sp.D) return;        int lw = sp.W - bw;        int lh = sp.H - bh;        int ld = sp.D - bd;        long leftoverVol = (long)lw * sp.H * sp.D + (long)bw * lh * sp.D + (long)bw * bh * ld;        int maxEdge = lw;        if (lh > maxEdge) maxEdge = lh;        if (ld > maxEdge) maxEdge = ld;        long score =            (leftoverVol << 12) +            ((long)maxEdge << 6) +            ((long)sp.Z << 4) +            ((long)sp.Y << 2) +            (long)sp.X;        if (score < bestScore)        {            bestScore = score;            bestSpaceIdx = spIdx;            bestW = bw; bestH = bh; bestD = bd;        }    }    private static void CleanSpaces(List<Space> spaces, int maxSpaces)    {        if (spaces.Count <= 1) return;        for (int i = spaces.Count - 1; i >= 0; i--)        {            var s = spaces[i];            if (s.W <= 0 || s.H <= 0 || s.D <= 0)            {                int last = spaces.Count - 1;                spaces[i] = spaces[last];                spaces.RemoveAt(last);            }        }        spaces.Sort((a, b) =>        {            int c = a.Z.CompareTo(b.Z);            if (c != 0) return c;            c = a.Y.CompareTo(b.Y);            if (c != 0) return c;            c = a.X.CompareTo(b.X);            if (c != 0) return c;            return b.Volume.CompareTo(a.Volume);        });        int m = spaces.Count;        int window = m > 2500 ? 2500 : m;        var keep = new bool[m];        for (int i = 0; i < m; i++) keep[i] = true;        for (int i = 0; i < window; i++)        {            if (!keep[i]) continue;            var a = spaces[i];            int ax2 = a.X + a.W;            int ay2 = a.Y + a.H;            int az2 = a.Z + a.D;            for (int j = i + 1; j < window; j++)            {                if (!keep[j]) continue;                var b = spaces[j];                if (b.X >= a.X && b.Y >= a.Y && b.Z >= a.Z &&                    b.X + b.W <= ax2 && b.Y + b.H <= ay2 && b.Z + b.D <= az2)                {                    keep[j] = false;                    continue;                }                if (a.X >= b.X && a.Y >= b.Y && a.Z >= b.Z &&                    ax2 <= b.X + b.W && ay2 <= b.Y + b.H && az2 <= b.Z + b.D)                {                    keep[i] = false;                    break;                }            }        }        int write = 0;        for (int i = 0; i < m; i++)            if (keep[i]) spaces[write++] = spaces[i];        if (write < m) spaces.RemoveRange(write, m - write);        if (spaces.Count > maxSpaces)            spaces.RemoveRange(maxSpaces, spaces.Count - maxSpaces);    }}
