declare const require: any;

type Grid = string[];

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

type Input = { grid: Grid; robots: Robot[]; jobs: Job[] };
type Output = { assignments: number[][] };

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8").trim();
const data: Input = input.length ? JSON.parse(input) : ({} as any);

function computeAssignments(grid: Grid, robots: Robot[], jobs: Job[]): number[][] {
  const height = grid.length;
  const width = height ? grid[0].length : 0;
  const open = new Uint8Array(width * height);
  for (let y = 0; y < height; y++) {
    const row = grid[y];
    for (let x = 0; x < width; x++) open[y * width + x] = row.charCodeAt(x) === 35 ? 0 : 1; // '#'
  }

  const robotCount = robots.length;
  const assignments: number[][] = Array.from({ length: robotCount }, () => []);

  // Group jobs by priority (descending). Within a priority, deterministic order by id.
  const jobsByPriority = new Map<number, Job[]>();
  for (const j of jobs) {
    let arr = jobsByPriority.get(j.priority);
    if (!arr) {
      arr = [];
      jobsByPriority.set(j.priority, arr);
    }
    arr.push(j);
  }
  const priorities = Array.from(jobsByPriority.keys()).sort((a, b) => b - a);
  for (const p of priorities) jobsByPriority.get(p)!.sort((a, b) => a.id - b.id);

  // Per-robot position as it accumulates assigned jobs (start at robot location).
  const curX = new Int32Array(robotCount);
  const curY = new Int32Array(robotCount);
  for (let i = 0; i < robotCount; i++) {
    curX[i] = robots[i].x | 0;
    curY[i] = robots[i].y | 0;
  }

  // Multi-source BFS for each group of jobs (same priority) to get nearest robot
  // from each job cell. This approximates a greedy that minimizes incremental travel,
  // while being much cheaper than running BFS per job/robot.
  const dist = new Int32Array(width * height);
  const owner = new Int32Array(width * height);
  const q = new Int32Array(width * height);

  const inBounds = (x: number, y: number) => x >= 0 && y >= 0 && x < width && y < height;
  const idx = (x: number, y: number) => y * width + x;

  for (const p of priorities) {
    const group = jobsByPriority.get(p)!;
    if (!group.length) continue;

    // init BFS arrays
    dist.fill(-1);
    owner.fill(-1);
    let head = 0;
    let tail = 0;

    // seed with robots at their current positions (only if on open cell)
    for (let r = 0; r < robotCount; r++) {
      const rx = curX[r];
      const ry = curY[r];
      if (!inBounds(rx, ry)) continue;
      const pos = idx(rx, ry);
      if (!open[pos]) continue;
      if (dist[pos] === -1) {
        dist[pos] = 0;
        owner[pos] = r;
        q[tail++] = pos;
      } else {
        // tie: smaller robot index wins for determinism
        if (owner[pos] > r) owner[pos] = r;
      }
    }

    // BFS
    while (head < tail) {
      const v = q[head++];
      const d = dist[v] + 1;
      const o = owner[v];
      const x = v % width;
      const y = (v / width) | 0;

      // 4-neighbors
      // left
      if (x > 0) {
        const u = v - 1;
        if (open[u]) {
          if (dist[u] === -1) {
            dist[u] = d;
            owner[u] = o;
            q[tail++] = u;
          } else if (dist[u] === d && owner[u] > o) {
            owner[u] = o;
          }
        }
      }
      // right
      if (x + 1 < width) {
        const u = v + 1;
        if (open[u]) {
          if (dist[u] === -1) {
            dist[u] = d;
            owner[u] = o;
            q[tail++] = u;
          } else if (dist[u] === d && owner[u] > o) {
            owner[u] = o;
          }
        }
      }
      // up
      if (y > 0) {
        const u = v - width;
        if (open[u]) {
          if (dist[u] === -1) {
            dist[u] = d;
            owner[u] = o;
            q[tail++] = u;
          } else if (dist[u] === d && owner[u] > o) {
            owner[u] = o;
          }
        }
      }
      // down
      if (y + 1 < height) {
        const u = v + width;
        if (open[u]) {
          if (dist[u] === -1) {
            dist[u] = d;
            owner[u] = o;
            q[tail++] = u;
          } else if (dist[u] === d && owner[u] > o) {
            owner[u] = o;
          }
        }
      }
    }

    // Assign jobs in this priority group in deterministic order, using nearest-owner,
    // with fallback to greedy BFS from the best robot when job cell unreachable.
    for (const job of group) {
      const jx = job.x | 0;
      const jy = job.y | 0;
      let chosen = 0;
      let d = -1;
      if (inBounds(jx, jy)) {
        const pos = idx(jx, jy);
        d = dist[pos];
        chosen = owner[pos];
      }
      if (d < 0 || chosen < 0) {
        // Fallback: run BFS from job until hitting any robot start, choose min dist then robot index.
        // This is rare (disconnected components or invalid positions).
        // Reverse BFS from job is generally faster than per-robot BFS.
        chosen = 0;
        let bestDist = 1e9;
        dist.fill(-1);
        head = 0;
        tail = 0;
        if (inBounds(jx, jy)) {
          const start = idx(jx, jy);
          if (open[start]) {
            dist[start] = 0;
            q[tail++] = start;
          }
        }
        while (head < tail) {
          const v = q[head++];
          const curD = dist[v];
          // Early exit if we've already exceeded best found.
          if (curD >= bestDist) continue;
          const x = v % width;
          const y = (v / width) | 0;
          // Check if this cell matches any robot current position.
          for (let r = 0; r < robotCount; r++) {
            if (curX[r] === x && curY[r] === y) {
              if (curD < bestDist || (curD === bestDist && r < chosen)) {
                bestDist = curD;
                chosen = r;
              }
            }
          }
          const nd = curD + 1;
          if (x > 0) {
            const u = v - 1;
            if (open[u] && dist[u] === -1) {
              dist[u] = nd;
              q[tail++] = u;
            }
          }
          if (x + 1 < width) {
            const u = v + 1;
            if (open[u] && dist[u] === -1) {
              dist[u] = nd;
              q[tail++] = u;
            }
          }
          if (y > 0) {
            const u = v - width;
            if (open[u] && dist[u] === -1) {
              dist[u] = nd;
              q[tail++] = u;
            }
          }
          if (y + 1 < height) {
            const u = v + width;
            if (open[u] && dist[u] === -1) {
              dist[u] = nd;
              q[tail++] = u;
            }
          }
        }
        // If still unreachable, keep chosen=0 (deterministic).
      }

      assignments[chosen].push(job.id);
      curX[chosen] = jx;
      curY[chosen] = jy;
    }
  }

  return assignments;
}

const assignments = computeAssignments(data.grid || [], data.robots || [], data.jobs || []);
const out: Output = { assignments };
process.stdout.write(JSON.stringify(out));
