use std::cmp::Reverse;
use std::collections::{BinaryHeap, VecDeque};
use std::io::{self, Read};

fn main() {
    // Read entire stdin fast.
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        // Empty input: output empty cover.
        println!("0\n");
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            println!("0\n");
            return;
        }
    };

    // Adjacency as edge list per vertex, plus global edge arrays.
    let mut eu: Vec<u32> = Vec::with_capacity(m);
    let mut ev: Vec<u32> = Vec::with_capacity(m);
    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for i in 0..m {
        let u: u32 = it.next().unwrap().parse().unwrap();
        let v: u32 = it.next().unwrap().parse().unwrap();
        eu.push(u);
        ev.push(v);
        adj[u as usize].push(i as u32);
        adj[v as usize].push(i as u32);
    }

    // Covered edges bitmap.
    let mut covered: Vec<u8> = vec![0; m];
    let mut uncovered_cnt: i64 = m as i64;

    // Current uncovered degree per vertex.
    let mut deg: Vec<u32> = adj
        .iter()
        .map(|lst| lst.len() as u32)
        .collect();

    // Greedy: repeatedly pick vertex with maximum uncovered degree.
    // Use lazy max-heap (degree, vertex).
    let mut heap: BinaryHeap<(u32, u32)> = BinaryHeap::new();
    for v in 0..n {
        let d = deg[v];
        if d > 0 {
            heap.push((d, v as u32));
        }
    }

    let mut chosen: Vec<u32> = Vec::new();
    let mut in_cover: Vec<u8> = vec![0; n];

    while uncovered_cnt > 0 {
        // Pop until consistent.
        let (d, v) = loop {
            match heap.pop() {
                Some((d, v)) => {
                    if in_cover[v as usize] != 0 {
                        continue;
                    }
                    if d == deg[v as usize] {
                        break (d, v);
                    }
                }
                None => {
                    // Fallback: pick any uncovered edge endpoint.
                    // Find an uncovered edge.
                    let mut picked: Option<u32> = None;
                    for e in 0..m {
                        if covered[e] == 0 {
                            picked = Some(eu[e]);
                            break;
                        }
                    }
                    if let Some(vp) = picked {
                        break (deg[vp as usize], vp);
                    } else {
                        break (0, 0);
                    }
                }
            }
        };
        if d == 0 {
            // No uncovered edges.
            break;
        }
        // Select v.
        in_cover[v as usize] = 1;
        chosen.push(v);

        // Cover incident edges and update degrees of neighbors.
        for &eid_u32 in &adj[v as usize] {
            let eid = eid_u32 as usize;
            if covered[eid] != 0 {
                continue;
            }
            covered[eid] = 1;
            uncovered_cnt -= 1;
            let a = eu[eid];
            let b = ev[eid];
            // decrement uncovered degree for both endpoints if not already in cover
            if in_cover[a as usize] == 0 && deg[a as usize] > 0 {
                deg[a as usize] -= 1;
                heap.push((deg[a as usize], a));
            }
            if in_cover[b as usize] == 0 && deg[b as usize] > 0 {
                deg[b as usize] -= 1;
                heap.push((deg[b as usize], b));
            }
        }
        // After selecting v, its degree is effectively 0 for uncovered edges.
        deg[v as usize] = 0;
    }

    // Post-processing: remove redundant vertices.
    // Recompute edge coverage counts by chosen vertices.
    let k = chosen.len();
    let mut edge_cov_count: Vec<u16> = vec![0; m];
    let mut chosen_pos: Vec<i32> = vec![-1; n];
    for (idx, &v) in chosen.iter().enumerate() {
        chosen_pos[v as usize] = idx as i32;
        for &eid_u32 in &adj[v as usize] {
            let eid = eid_u32 as usize;
            edge_cov_count[eid] = edge_cov_count[eid].saturating_add(1);
        }
    }

    // Attempt to remove vertices with smallest contribution first (heuristic).
    // Compute "unique edges" count for each chosen v: edges where cov_count==1.
    let mut unique_edges: Vec<u32> = vec![0; k];
    for (idx, &v) in chosen.iter().enumerate() {
        let mut uniq = 0u32;
        for &eid_u32 in &adj[v as usize] {
            let eid = eid_u32 as usize;
            if edge_cov_count[eid] == 1 {
                uniq += 1;
            }
        }
        unique_edges[idx] = uniq;
    }

    // Min-heap by unique_edges (try remove low-uniq first).
    let mut rem_heap: BinaryHeap<(Reverse<u32>, u32)> = BinaryHeap::new();
    for (idx, &u) in unique_edges.iter().enumerate() {
        rem_heap.push((Reverse(u), idx as u32));
    }

    let mut alive: Vec<u8> = vec![1; k];
    while let Some((Reverse(_u), idx_u32)) = rem_heap.pop() {
        let idx = idx_u32 as usize;
        if alive[idx] == 0 {
            continue;
        }
        let v = chosen[idx] as usize;

        // Check if v is removable: all incident edges have cov_count>=2.
        let mut can_remove = true;
        for &eid_u32 in &adj[v] {
            let eid = eid_u32 as usize;
            if edge_cov_count[eid] == 1 {
                can_remove = false;
                break;
            }
        }
        if !can_remove {
            continue;
        }
        // Remove v.
        alive[idx] = 0;
        for &eid_u32 in &adj[v] {
            let eid = eid_u32 as usize;
            if edge_cov_count[eid] > 0 {
                edge_cov_count[eid] -= 1;
            }
        }
    }

    let mut final_cover: Vec<u32> = Vec::with_capacity(chosen.len());
    for (idx, &v) in chosen.iter().enumerate() {
        if alive[idx] != 0 {
            final_cover.push(v);
        }
    }

    // Safety check pass: ensure all edges covered; if not, patch by adding one endpoint.
    // (Should not happen, but keep output valid.)
    let mut in_final: Vec<u8> = vec![0; n];
    for &v in &final_cover {
        in_final[v as usize] = 1;
    }
    let mut add_queue: VecDeque<u32> = VecDeque::new();
    for e in 0..m {
        let a = eu[e] as usize;
        let b = ev[e] as usize;
        if in_final[a] == 0 && in_final[b] == 0 {
            // Add the endpoint with higher degree in original adjacency.
            let pick = if adj[a].len() >= adj[b].len() { a as u32 } else { b as u32 };
            if in_final[pick as usize] == 0 {
                in_final[pick as usize] = 1;
                add_queue.push_back(pick);
            }
        }
    }
    while let Some(v) = add_queue.pop_front() {
        final_cover.push(v);
    }

    // Output.
    println!("{}", final_cover.len());
    if !final_cover.is_empty() {
        // Keep stable, sorted output (optional).
        final_cover.sort_unstable();
        for (i, v) in final_cover.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
    }
    println!();
}
