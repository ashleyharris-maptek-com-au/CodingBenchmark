using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private readonly struct Pos : IEquatable<Pos>
    {
        public readonly int R;
        public readonly int C;
        public Pos(int r, int c) { R = r; C = c; }
        public bool Equals(Pos other) => R == other.R && C == other.C;
        public override bool Equals(object? obj) => obj is Pos p && Equals(p);
        public override int GetHashCode() => (R * 397) ^ C;
        public override string ToString() => $"{C} {R}";
    }

    private sealed class FastScanner
    {
        private readonly TextReader _tr;
        private string? _line;
        private int _idx;
        public FastScanner(TextReader tr) { _tr = tr; }

        public bool TryReadInt(out int value)
        {
            value = 0;
            while (true)
            {
                if (_line == null || _idx >= _line.Length)
                {
                    _line = _tr.ReadLine();
                    _idx = 0;
                    if (_line == null) return false;
                }
                while (_idx < _line.Length && _line[_idx] == ' ') _idx++;
                if (_idx < _line.Length) break;
                _line = null;
            }
            int sign = 1;
            if (_line![_idx] == '-') { sign = -1; _idx++; }
            int v = 0;
            while (true)
            {
                if (_line == null || _idx >= _line.Length) break;
                char ch = _line[_idx];
                if (ch < '0' || ch > '9') break;
                v = v * 10 + (ch - '0');
                _idx++;
            }
            value = v * sign;
            return true;
        }
    }

    private sealed class BitGrid
    {
        // Stores flags for cells in a memory-efficient way: 1 bit per cell.
        private readonly ulong[] _data;
        private readonly int _w;
        public readonly int H;
        public readonly int W;
        public BitGrid(int h, int w)
        {
            H = h; W = w; _w = w;
            long n = (long)h * w;
            _data = new ulong[(int)((n + 63) >> 6)];
        }
        public bool Get(int r, int c)
        {
            long idx = (long)r * _w + c;
            int word = (int)(idx >> 6);
            int bit = (int)(idx & 63);
            return ((_data[word] >> bit) & 1UL) != 0UL;
        }
        public void Set(int r, int c)
        {
            long idx = (long)r * _w + c;
            int word = (int)(idx >> 6);
            int bit = (int)(idx & 63);
            _data[word] |= 1UL << bit;
        }
    }

    private sealed class Constraint
    {
        public int Remaining; // mines remaining in this constraint
        public int ActiveCount; // number of currently-unknown vars in it
        public readonly List<int> Vars; // variable ids
        public Constraint(int remaining, List<int> vars)
        {
            Remaining = remaining;
            Vars = vars;
            ActiveCount = vars.Count;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.In);
        if (!fs.TryReadInt(out int H) || !fs.TryReadInt(out int W))
        {
            Console.WriteLine(0);
            return;
        }

        // Read H board rows as raw lines (can include leading/trailing spaces).
        string[] rows = new string[H];
        for (int r = 0; r < H; r++)
        {
            string? line = Console.In.ReadLine();
            if (line == null) line = "";
            // Ensure length W: if shorter, pad with spaces; if longer, take first W.
            if (line.Length < W) line = line + new string(' ', W - line.Length);
            else if (line.Length > W) line = line.Substring(0, W);
            rows[r] = line;
        }

        // Decide strategy based on frontier size and board area.
        // We always do fast local rules + set-propagation.
        // If the connected frontier components are small, we do exact enumeration per component.

        char[,] board = new char[H, W];
        for (int r = 0; r < H; r++)
        {
            var line = rows[r];
            for (int c = 0; c < W; c++) board[r, c] = line[c];
        }

        // Mine known flags from '*'
        var knownMine = new BitGrid(H, W);
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == '*') knownMine.Set(r, c);

        var safeOut = new BitGrid(H, W); // proven safe unknown cells
        var isUnknown = new BitGrid(H, W);
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == ' ') isUnknown.Set(r, c);

        // If no revealed numbers, we can't prove anything.
        bool anyNumber = false;
        for (int r = 0; r < H && !anyNumber; r++)
            for (int c = 0; c < W && !anyNumber; c++)
                if (board[r, c] >= '0' && board[r, c] <= '8') anyNumber = true;
        if (!anyNumber)
        {
            Console.WriteLine(0);
            return;
        }

        // Build constraints for frontier unknown cells adjacent to a number.
        // Unknown cells not adjacent to any number can't be proven safe via pure logic here.
        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        // Variable mapping only for frontier unknown cells.
        // For massive boards, avoid Dictionary allocations per cell: only index frontier.
        var varId = new Dictionary<long, int>(capacity: 1 << 16);
        var varPos = new List<Pos>(capacity: 1 << 16);

        // First pass: identify number cells that touch unknowns (frontier constraints).
        var numberCells = new List<Pos>(capacity: 1 << 16);
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                char ch = board[r, c];
                if (ch < '0' || ch > '8') continue;
                bool touchesUnknown = false;
                for (int k = 0; k < 8; k++)
                {
                    int rr = r + dr[k], cc = c + dc[k];
                    if ((uint)rr >= (uint)H || (uint)cc >= (uint)W) continue;
                    if (board[rr, cc] == ' ') { touchesUnknown = true; break; }
                }
                if (touchesUnknown) numberCells.Add(new Pos(r, c));
            }
        }

        // If no constraints (numbers adjacent to unknowns), nothing provably safe.
        if (numberCells.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Helper to get or create variable id for unknown cell.
        int GetVar(int r, int c)
        {
            long key = ((long)r << 32) ^ (uint)c;
            if (varId.TryGetValue(key, out int id)) return id;
            id = varPos.Count;
            varId[key] = id;
            varPos.Add(new Pos(r, c));
            return id;
        }

        // Constraints list and incidence lists (var -> constraints containing it).
        var constraints = new List<Constraint>(numberCells.Count);
        var varToCons = new List<List<int>>(capacity: 1 << 16);

        for (int i = 0; i < numberCells.Count; i++)
        {
            var p = numberCells[i];
            int r = p.R, c = p.C;
            int clue = board[r, c] - '0';
            int knownMinesAdj = 0;
            var vars = new List<int>(8);
            for (int k = 0; k < 8; k++)
            {
                int rr = r + dr[k], cc = c + dc[k];
                if ((uint)rr >= (uint)H || (uint)cc >= (uint)W) continue;
                char nb = board[rr, cc];
                if (nb == '*') knownMinesAdj++;
                else if (nb == ' ') vars.Add(GetVar(rr, cc));
            }
            int remaining = clue - knownMinesAdj;
            if (remaining < 0) remaining = 0; // inconsistent board; be conservative
            if (vars.Count == 0) continue;
            int idx = constraints.Count;
            constraints.Add(new Constraint(remaining, vars));
            foreach (int v in vars)
            {
                while (varToCons.Count <= v) varToCons.Add(new List<int>(4));
                varToCons[v].Add(idx);
            }
        }

        if (constraints.Count == 0 || varPos.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        int V = varPos.Count;

        // State per variable:
        // 0 = unassigned (unknown), 1 = proven safe, 2 = proven mine.
        byte[] varState = new byte[V];

        // Queue for propagation when constraints update.
        var q = new Queue<int>(constraints.Count);
        for (int i = 0; i < constraints.Count; i++) q.Enqueue(i);

        // Track active vars in constraints (we update Remaining and ActiveCount).
        // Propagation rules:
        // - Remaining == 0 => all vars safe
        // - Remaining == ActiveCount => all vars mines
        bool changed = true;
        while (changed)
        {
            changed = false;
            int iterations = 0;
            while (q.Count > 0)
            {
                int ci = q.Dequeue();
                if ((uint)ci >= (uint)constraints.Count) continue;
                var con = constraints[ci];

                // Recompute ActiveCount lazily if needed (cheap due to max 8 vars typical).
                int active = 0;
                int remaining = con.Remaining;
                for (int j = 0; j < con.Vars.Count; j++)
                {
                    int v = con.Vars[j];
                    byte st = varState[v];
                    if (st == 0) active++;
                    else if (st == 2) remaining--; // assigned mine satisfies one mine
                }
                if (remaining < 0) remaining = 0; // conservative
                con.ActiveCount = active;
                con.Remaining = remaining;

                if (active == 0) continue;
                if (remaining == 0)
                {
                    for (int j = 0; j < con.Vars.Count; j++)
                    {
                        int v = con.Vars[j];
                        if (varState[v] != 0) continue;
                        varState[v] = 1;
                        var p = varPos[v];
                        safeOut.Set(p.R, p.C);
                        changed = true;
                        foreach (int otherCi in varToCons[v]) q.Enqueue(otherCi);
                    }
                }
                else if (remaining == active)
                {
                    for (int j = 0; j < con.Vars.Count; j++)
                    {
                        int v = con.Vars[j];
                        if (varState[v] != 0) continue;
                        varState[v] = 2;
                        var p = varPos[v];
                        knownMine.Set(p.R, p.C);
                        changed = true;
                        foreach (int otherCi in varToCons[v]) q.Enqueue(otherCi);
                    }
                }

                // Hard stop guard against pathological inputs; keep it linear-time.
                if (++iterations > 5_000_000) { q.Clear(); break; }
            }

            // Subset rule pass (still local, but more powerful than pure singles):
            // For two constraints A,B:
            // - If vars(A) ⊆ vars(B): then (vars(B)\vars(A)) has mines = rem(B)-rem(A).
            // This can infer safes/mines when the difference equals 0 or size.
            // Do this only when constraints count is manageable.
            if (!changed)
            {
                int C = constraints.Count;
                if (C <= 200_000)
                {
                    // Index constraints by a small signature: first var id (min) to find candidates.
                    var bucket = new Dictionary<int, List<int>>(capacity: Math.Min(C, 1 << 16));
                    for (int i = 0; i < C; i++)
                    {
                        var con = constraints[i];
                        int minVar = int.MaxValue;
                        for (int j = 0; j < con.Vars.Count; j++)
                        {
                            int v = con.Vars[j];
                            if (varState[v] != 0) continue;
                            if (v < minVar) minVar = v;
                        }
                        if (minVar == int.MaxValue) continue;
                        if (!bucket.TryGetValue(minVar, out var list))
                        {
                            list = new List<int>(4);
                            bucket[minVar] = list;
                        }
                        list.Add(i);
                    }

                    foreach (var kv in bucket)
                    {
                        var list = kv.Value;
                        if (list.Count < 2) continue;
                        for (int a = 0; a < list.Count; a++)
                        {
                            var A = constraints[list[a]];
                            // Build active var set for A
                            Span<int> aVars = stackalloc int[8];
                            int aCount = 0;
                            int aRem = A.Remaining;
                            for (int j = 0; j < A.Vars.Count; j++)
                            {
                                int v = A.Vars[j];
                                if (varState[v] == 2) aRem--;
                                else if (varState[v] == 0) aVars[aCount++] = v;
                            }
                            if (aCount == 0) continue;
                            for (int b = 0; b < list.Count; b++)
                            {
                                if (a == b) continue;
                                var B = constraints[list[b]];
                                Span<int> bVars = stackalloc int[8];
                                int bCount = 0;
                                int bRem = B.Remaining;
                                for (int j = 0; j < B.Vars.Count; j++)
                                {
                                    int v = B.Vars[j];
                                    if (varState[v] == 2) bRem--;
                                    else if (varState[v] == 0) bVars[bCount++] = v;
                                }
                                if (bCount == 0) continue;

                                // Check if A subset of B (both small, do O(64) comparisons).
                                bool subset = true;
                                for (int iav = 0; iav < aCount; iav++)
                                {
                                    int v = aVars[iav];
                                    bool found = false;
                                    for (int ibv = 0; ibv < bCount; ibv++)
                                        if (bVars[ibv] == v) { found = true; break; }
                                    if (!found) { subset = false; break; }
                                }
                                if (!subset) continue;

                                // Compute difference vars = B \ A
                                Span<int> diff = stackalloc int[8];
                                int diffCount = 0;
                                for (int ibv = 0; ibv < bCount; ibv++)
                                {
                                    int v = bVars[ibv];
                                    bool inA = false;
                                    for (int iav = 0; iav < aCount; iav++)
                                        if (aVars[iav] == v) { inA = true; break; }
                                    if (!inA) diff[diffCount++] = v;
                                }
                                if (diffCount == 0) continue;
                                int diffMines = bRem - aRem;
                                if (diffMines < 0) continue;
                                if (diffMines == 0)
                                {
                                    for (int iDiff = 0; iDiff < diffCount; iDiff++)
                                    {
                                        int v = diff[iDiff];
                                        if (varState[v] != 0) continue;
                                        varState[v] = 1;
                                        var p = varPos[v];
                                        safeOut.Set(p.R, p.C);
                                        changed = true;
                                        foreach (int otherCi in varToCons[v]) q.Enqueue(otherCi);
                                    }
                                }
                                else if (diffMines == diffCount)
                                {
                                    for (int iDiff = 0; iDiff < diffCount; iDiff++)
                                    {
                                        int v = diff[iDiff];
                                        if (varState[v] != 0) continue;
                                        varState[v] = 2;
                                        var p = varPos[v];
                                        knownMine.Set(p.R, p.C);
                                        changed = true;
                                        foreach (int otherCi in varToCons[v]) q.Enqueue(otherCi);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        // If frontier is small enough, perform exact enumeration per connected component
        // to prove additional safe cells (cells that are safe in ALL satisfying assignments).
        // We only do this when V is moderate and constraints are not huge.
        int V_limit = 2000; // overall cap for enumeration attempts
        int component_limit = 28; // variable cap per component for full enumeration
        if (V <= V_limit && constraints.Count <= 200_000)
        {
            // Build adjacency among variables via shared constraints.
            var adj = new List<int>[V];
            for (int v = 0; v < V; v++) adj[v] = new List<int>(8);
            for (int ci = 0; ci < constraints.Count; ci++)
            {
                var con = constraints[ci];
                // collect currently unassigned vars
                int[] tmp = new int[con.Vars.Count];
                int t = 0;
                for (int j = 0; j < con.Vars.Count; j++)
                {
                    int v = con.Vars[j];
                    if (varState[v] == 0) tmp[t++] = v;
                }
                for (int i = 0; i < t; i++)
                {
                    int a = tmp[i];
                    for (int j = i + 1; j < t; j++)
                    {
                        int b = tmp[j];
                        adj[a].Add(b);
                        adj[b].Add(a);
                    }
                }
            }

            var seen = new bool[V];
            var comp = new List<int>(64);
            for (int v0 = 0; v0 < V; v0++)
            {
                if (varState[v0] != 0 || seen[v0]) continue;
                comp.Clear();
                var stack = new Stack<int>();
                stack.Push(v0);
                seen[v0] = true;
                while (stack.Count > 0)
                {
                    int v = stack.Pop();
                    comp.Add(v);
                    var list = adj[v];
                    for (int i = 0; i < list.Count; i++)
                    {
                        int u = list[i];
                        if (varState[u] != 0 || seen[u]) continue;
                        seen[u] = true;
                        stack.Push(u);
                    }
                }

                if (comp.Count == 0) continue;
                if (comp.Count > component_limit) continue;

                // Gather constraints involving this component (only those with at least one var in comp).
                var compSet = new HashSet<int>(comp);
                var compConstraints = new List<(int rem, int[] vars)>();
                for (int ci = 0; ci < constraints.Count; ci++)
                {
                    var con = constraints[ci];
                    // compute remaining mines and vars restricted to component, accounting for already-known mines
                    int rem = con.Remaining;
                    var vars = new List<int>(8);
                    for (int j = 0; j < con.Vars.Count; j++)
                    {
                        int v = con.Vars[j];
                        if (varState[v] == 2) rem--;
                        else if (varState[v] == 0 && compSet.Contains(v)) vars.Add(v);
                    }
                    if (vars.Count == 0) continue;
                    if (rem < 0) rem = 0;
                    // Note: vars outside component are treated as "unknown external";
                    // To keep proofs sound, we only enumerate if all vars of the constraint are in component.
                    bool allInside = true;
                    for (int j = 0; j < con.Vars.Count; j++)
                    {
                        int v = con.Vars[j];
                        if (varState[v] == 0 && !compSet.Contains(v)) { allInside = false; break; }
                    }
                    if (!allInside) continue;
                    compConstraints.Add((rem, vars.ToArray()));
                }
                if (compConstraints.Count == 0) continue;

                // Map component vars to local 0..n-1
                int n = comp.Count;
                var localIndex = new Dictionary<int, int>(n);
                for (int i = 0; i < n; i++) localIndex[comp[i]] = i;
                // Use arrays of local indices (not bitmask) for n up to 28.
                var consVars = new int[compConstraints.Count][];
                var consRem = new int[compConstraints.Count];
                for (int i = 0; i < compConstraints.Count; i++)
                {
                    consRem[i] = compConstraints[i].rem;
                    var vs = compConstraints[i].vars;
                    int[] loc = new int[vs.Length];
                    for (int j = 0; j < vs.Length; j++) loc[j] = localIndex[vs[j]];
                    consVars[i] = loc;
                }

                // Backtracking enumeration:
                // Track for each constraint: assignedMines, assignedCount.
                int m = compConstraints.Count;
                int[] assignedCount = new int[m];
                int[] assignedMines = new int[m];
                int[] totalCount = new int[m];
                for (int i = 0; i < m; i++) totalCount[i] = consVars[i].Length;

                // Build var -> constraints incidence in local indices
                var v2c = new List<int>[n];
                for (int i = 0; i < n; i++) v2c[i] = new List<int>(4);
                for (int ci = 0; ci < m; ci++)
                {
                    var vs = consVars[ci];
                    for (int j = 0; j < vs.Length; j++) v2c[vs[j]].Add(ci);
                }

                // Order variables by degree (fail-first).
                int[] order = Enumerable.Range(0, n).ToArray();
                Array.Sort(order, (a, b) => v2c[b].Count.CompareTo(v2c[a].Count));
                int[] assign = new int[n]; // -1 unassigned, 0 safe, 1 mine
                for (int i = 0; i < n; i++) assign[i] = -1;

                // For proof: count in how many satisfying assignments each var is mine.
                long totalSolutions = 0;
                long[] mineCount = new long[n];

                bool Dfs(int idx)
                {
                    if (idx == n)
                    {
                        // verify all constraints satisfied
                        for (int ci = 0; ci < m; ci++)
                        {
                            if (assignedMines[ci] != consRem[ci]) return true; // invalid leaf
                        }
                        totalSolutions++;
                        for (int i = 0; i < n; i++) if (assign[i] == 1) mineCount[i]++;
                        return true;
                    }

                    int v = order[idx];

                    // Try safe (0) then mine (1). Either is allowed; we only use proofs when consistent.
                    for (int val = 0; val <= 1; val++)
                    {
                        // Apply
                        assign[v] = val;
                        bool ok = true;
                        var consList = v2c[v];
                        for (int i = 0; i < consList.Count; i++)
                        {
                            int ci = consList[i];
                            assignedCount[ci]++;
                            if (val == 1) assignedMines[ci]++;

                            int mines = assignedMines[ci];
                            int cnt = assignedCount[ci];
                            int rem = consRem[ci];
                            int left = totalCount[ci] - cnt;
                            if (mines > rem) ok = false;
                            else if (mines + left < rem) ok = false;
                        }

                        if (ok) Dfs(idx + 1);

                        // Undo
                        for (int i = 0; i < consList.Count; i++)
                        {
                            int ci = consList[i];
                            if (val == 1) assignedMines[ci]--;
                            assignedCount[ci]--;
                        }
                        assign[v] = -1;
                    }
                    return true;
                }

                Dfs(0);

                if (totalSolutions <= 0) continue;
                for (int i = 0; i < n; i++)
                {
                    if (mineCount[i] == 0)
                    {
                        int globalVar = comp[i];
                        if (varState[globalVar] == 0)
                        {
                            varState[globalVar] = 1;
                            var p = varPos[globalVar];
                            safeOut.Set(p.R, p.C);
                        }
                    }
                }
            }
        }

        // Output all cells that are currently unknown and proven safe.
        // Note: only output cells that were ' ' in input.
        var output = new List<Pos>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (isUnknown.Get(r, c) && safeOut.Get(r, c)) output.Add(new Pos(r, c));
            }
        }

        Console.WriteLine(output.Count);
        for (int i = 0; i < output.Count; i++) Console.WriteLine(output[i].ToString());
    }
}
