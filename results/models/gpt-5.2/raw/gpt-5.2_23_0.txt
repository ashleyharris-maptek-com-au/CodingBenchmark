use std::f64::consts::PI;
use std::io::{self, Read, Write};
use std::sync::mpsc::{self, Receiver, SyncSender};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug, Default)]
struct MapInfo {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn parse_f64(s: &str) -> Option<f64> {
    s.parse::<f64>().ok()
}

fn parse_usize(s: &str) -> Option<usize> {
    s.parse::<usize>().ok()
}

fn reader_thread(tx: SyncSender<State>) {
    // Read stdin in a tight loop and push parsed states through a bounded channel.
    // We intentionally keep parsing simple and allocation-light.
    let mut input = String::new();
    let mut stdin = io::stdin();
    // Read all available first; but stdin is streaming. Use Read::read_to_string blocks.
    // Instead, read in chunks and split whitespace manually.
    let mut buf = [0u8; 1 << 16];
    let mut carry = String::new();

    loop {
        match stdin.read(&mut buf) {
            Ok(0) => {
                // EOF
                break;
            }
            Ok(n) => {
                let chunk = String::from_utf8_lossy(&buf[..n]);
                input.clear();
                input.push_str(&carry);
                input.push_str(&chunk);

                // Split by whitespace; keep tail if it ends mid-token.
                let mut last_is_ws = false;
                if let Some(ch) = input.chars().last() {
                    last_is_ws = ch.is_whitespace();
                }

                let mut it = input.split_whitespace();
                // We don't know where initial map parsing ended; the controller thread does that.
                // Here, we only parse state updates: 7 floats at a time.
                // If initial map lines are still in stream, controller will have already consumed them
                // before starting this thread in main.
                let mut vals: [f64; 7] = [0.0; 7];
                loop {
                    let mut ok = true;
                    for i in 0..7 {
                        if let Some(tok) = it.next() {
                            if let Some(v) = parse_f64(tok) {
                                vals[i] = v;
                            } else {
                                ok = false;
                                break;
                            }
                        } else {
                            ok = false;
                            break;
                        }
                    }
                    if !ok {
                        break;
                    }
                    let s = State {
                        x: vals[0],
                        y: vals[1],
                        vx: vals[2],
                        vy: vals[3],
                        angle: vals[4],
                        angular_vel: vals[5],
                        fuel: vals[6],
                    };
                    // If channel is full, drop oldest by trying a send; if it blocks, we can
                    // just skip (best-effort). Use try_send behavior by bounded + blocking send
                    // with short timeout is not available; so we use send which may block.
                    // To avoid reader stalling, we try a non-blocking pattern by first draining
                    // receiver is not possible here. So we accept occasional blocking but keep
                    // buffer small and controller fast.
                    if tx.send(s).is_err() {
                        return;
                    }
                }

                if last_is_ws {
                    carry.clear();
                } else {
                    // Save the last partial token: find last whitespace boundary.
                    if let Some(pos) = input.rfind(|c: char| c.is_whitespace()) {
                        carry.clear();
                        carry.push_str(&input[pos + 1..]);
                    } else {
                        carry = input.clone();
                    }
                }
            }
            Err(_) => break,
        }
    }
}

fn parse_initial_map(mut raw: String) -> Option<(MapInfo, Vec<(i32, i32)>, (i32, i32, i32, i32))> {
    // Returns (info, landing_zone_cells, bbox)
    // bbox = (min_x, max_x, min_y, max_y) of solid/ground (# or L or T) for basic height inference.

    // We cannot assume map is small; but the benchmark says it can be huge.
    // We therefore only parse header tokens from the beginning and then scan rows line-by-line,
    // keeping only L cells near the target and tracking ground height by column around target.

    // However, we were given the entire raw initial map string here, so use it directly.
    // In the real judge, initial map will likely be moderate; but still keep memory modest.

    let mut it = raw.split_whitespace();

    let width = parse_usize(it.next()?)?;
    let height = parse_usize(it.next()?)?;
    let gravity = parse_f64(it.next()?)?;
    let max_thrust = parse_f64(it.next()?)?;
    let fuel = parse_f64(it.next()?)?;

    let start_x = parse_f64(it.next()?)?;
    let start_y = parse_f64(it.next()?)?;
    let target_x = parse_f64(it.next()?)?;
    let target_y = parse_f64(it.next()?)?;

    let mut landing_cells: Vec<(i32, i32)> = Vec::new();

    // Basic bbox for solids
    let mut min_x = i32::MAX;
    let mut max_x = i32::MIN;
    let mut min_y = i32::MAX;
    let mut max_y = i32::MIN;

    // Remaining tokens are row strings (may contain no whitespace). Because split_whitespace will
    // treat each row as a token, we can iterate height rows.
    for y in 0..height {
        let row = it.next().unwrap_or("");
        // If row is shorter, treat missing as '.'
        for (x, ch) in row.chars().take(width).enumerate() {
            if ch == 'L' {
                landing_cells.push((x as i32, y as i32));
            }
            if ch == '#' || ch == 'L' || ch == 'T' {
                let xi = x as i32;
                let yi = y as i32;
                if xi < min_x {
                    min_x = xi;
                }
                if xi > max_x {
                    max_x = xi;
                }
                if yi < min_y {
                    min_y = yi;
                }
                if yi > max_y {
                    max_y = yi;
                }
            }
        }
    }

    if min_x == i32::MAX {
        min_x = 0;
        max_x = (width as i32).saturating_sub(1);
        min_y = 0;
        max_y = (height as i32).saturating_sub(1);
    }

    let info = MapInfo {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    Some((info, landing_cells, (min_x, max_x, min_y, max_y)))
}

fn compute_landing_zone(landing_cells: &[(i32, i32)], info: &MapInfo) -> (f64, f64, f64) {
    // Returns (zone_left, zone_right, zone_y) in meters. If missing, fallback around target.
    if landing_cells.is_empty() {
        let half = 2.0;
        return (info.target_x - half, info.target_x + half, info.target_y);
    }
    let mut min_x = i32::MAX;
    let mut max_x = i32::MIN;
    let mut sum_y = 0i64;
    for &(x, y) in landing_cells {
        if x < min_x {
            min_x = x;
        }
        if x > max_x {
            max_x = x;
        }
        sum_y += y as i64;
    }
    let zone_y = (sum_y as f64) / (landing_cells.len() as f64);
    (min_x as f64, (max_x as f64) + 1.0, zone_y)
}

fn controller(info: MapInfo, zone_left: f64, zone_right: f64, zone_y: f64, rx: Receiver<State>) {
    let mut out = io::BufWriter::new(io::stdout());

    // Control gains / heuristics
    let g = info.gravity;
    let amax = info.max_thrust;

    let dt_out = Duration::from_millis(20); // 50 Hz output
    let mut last_out = Instant::now();

    let mut cur = State::default();
    let mut have = false;

    // Keep a short-term estimate of time since last state update to avoid stalling.
    let mut last_state_time = Instant::now();

    loop {
        // Drain latest state if available
        while let Ok(s) = rx.try_recv() {
            cur = s;
            have = true;
            last_state_time = Instant::now();
        }

        if !have {
            // Before first state, output safe neutral
            writeln!(out, "0 0").ok();
            out.flush().ok();
            thread::sleep(dt_out);
            continue;
        }

        // If state stream stalls, still output something to keep 20Hz.
        let now = Instant::now();
        if now.duration_since(last_out) < dt_out {
            // Yield a bit
            thread::sleep(Duration::from_millis(1));
            continue;
        }
        last_out = now;

        // Position errors
        let pad_x = info.target_x;
        let pad_y = zone_y.max(info.target_y);
        let ex = pad_x - cur.x;
        let ey = pad_y - cur.y;

        // Desired horizontal speed: proportional, but limited. If outside zone, aim to enter.
        let zone_center = 0.5 * (zone_left + zone_right);
        let zone_half = 0.5 * (zone_right - zone_left).max(1.0);

        let x_goal = if cur.x < zone_left {
            zone_left + 0.25 * zone_half
        } else if cur.x > zone_right {
            zone_right - 0.25 * zone_half
        } else {
            zone_center
        };
        let ex2 = x_goal - cur.x;

        let mut vx_des = clamp(0.7 * ex2, -8.0, 8.0);
        // As we get close to ground, reduce lateral motion
        let altitude = (pad_y - cur.y).max(0.0);
        if altitude < 20.0 {
            vx_des = clamp(vx_des, -2.0, 2.0);
        }
        if altitude < 8.0 {
            vx_des = clamp(vx_des, -0.8, 0.8);
        }

        // Desired vertical speed: descend faster when high, slow near pad.
        let mut vy_des = if altitude > 60.0 {
            18.0
        } else if altitude > 25.0 {
            10.0
        } else if altitude > 12.0 {
            6.0
        } else if altitude > 6.0 {
            3.0
        } else {
            1.0
        };
        // If we are above target (negative ey because y increases down), keep descending.
        // vy positive is downward.
        if ey < 0.0 {
            vy_des = vy_des.max(2.0);
        }

        // Acceleration commands (world frame)
        let ax_cmd = clamp(0.9 * (vx_des - cur.vx), -amax * 0.6, amax * 0.6);
        let ay_cmd = clamp(0.9 * (vy_des - cur.vy), -amax * 0.8, amax * 0.8);

        // We want thrust to achieve (-ax_cmd, -ay_cmd - g)?? Careful with axes:
        // World y positive down. Gravity adds +g to vy.
        // Thrust accel components:
        // a_tx = thrust_acc * sin(angle)
        // a_ty = -thrust_acc * cos(angle)  (since angle=0 facing up => thrust upward = negative y)
        // Total accel: ax = a_tx; ay = g + a_ty
        // We want ax -> ax_cmd, ay -> ay_cmd.
        // So desired thrust vector in world:
        // ax_des = ax_cmd
        // ay_des = ay_cmd - g  (because thrust must provide ay - g)
        let ax_des = ax_cmd;
        let ay_des = ay_cmd - g;

        // Convert desired vector to desired angle and thrust magnitude.
        // thrust_acc * [sin(a), -cos(a)] = [ax_des, ay_des]
        // => sin(a)=ax_des/t, -cos(a)=ay_des/t
        // Let t = sqrt(ax_des^2 + ay_des^2)
        let t_acc = (ax_des * ax_des + ay_des * ay_des).sqrt();
        let t_acc_clamped = clamp(t_acc, 0.0, amax);
        let thrust = if amax > 1e-9 {
            clamp(t_acc_clamped / amax, 0.0, 1.0)
        } else {
            0.0
        };

        // Desired angle from components: a = atan2(ax_des, -ay_des)
        // because sin(a)=ax/t, cos(a)=? from -cos(a)=ay/t => cos(a)=-ay/t
        let angle_des = if t_acc > 1e-9 {
            (ax_des).atan2(-ay_des)
        } else {
            0.0
        };

        // Limit aggressive tilts near landing
        let mut angle_des_limited = angle_des;
        if altitude < 15.0 {
            angle_des_limited = clamp(angle_des_limited, -0.35, 0.35);
        } else if altitude < 40.0 {
            angle_des_limited = clamp(angle_des_limited, -0.7, 0.7);
        } else {
            angle_des_limited = clamp(angle_des_limited, -1.1, 1.1);
        }

        // PD on angle to produce turn rate command [-1,1]
        let ang_err = wrap_pi(angle_des_limited - cur.angle);
        let ang_rate_des = 6.0 * ang_err - 1.5 * cur.angular_vel;
        let turn = clamp(ang_rate_des / 6.0, -1.0, 1.0);

        // Fuel-aware: if low altitude and still fast downward, prioritize vertical braking.
        let mut thrust_adj = thrust;
        if altitude < 8.0 {
            // Ensure we don't smash: target vy near 1, if vy too high, boost.
            let vy_err = cur.vy - 1.0;
            if vy_err > 0.0 {
                thrust_adj = clamp(thrust_adj + 0.25 + 0.05 * vy_err, 0.0, 1.0);
            }
        }

        // If nearly on pad and lateral speed small, force upright.
        if altitude < 6.0 && cur.vx.abs() < 1.0 {
            let ang_err2 = wrap_pi(0.0 - cur.angle);
            let ang_rate_des2 = 7.0 * ang_err2 - 2.0 * cur.angular_vel;
            let turn2 = clamp(ang_rate_des2 / 6.0, -1.0, 1.0);
            writeln!(out, "{:.6} {:.6}", thrust_adj, turn2).ok();
        } else {
            writeln!(out, "{:.6} {:.6}", thrust_adj, turn).ok();
        }
        out.flush().ok();

        // If we haven't received a state update in a while, back off CPU.
        if Instant::now().duration_since(last_state_time) > Duration::from_millis(200) {
            thread::sleep(Duration::from_millis(5));
        }
    }
}

fn main() {
    // Read initial map synchronously first (it can be large, but the judge provides it once).
    // We read enough to parse the map and then spawn the streaming reader.
    // Since stdin is a stream, the simplest approach: read the entire stdin into memory would be
    // unsafe; instead we read header and map rows line-wise.

    let mut stdin = io::stdin().lock();

    // Read first 2 lines of numbers + then height rows.
    // But input format is whitespace-separated, so rows may be provided with newlines.
    // We'll read tokens using a simple scanner that can read lines but avoid gigabyte buffering.

    struct Scanner<R: Read> {
        r: R,
        buf: Vec<u8>,
        idx: usize,
    }
    impl<R: Read> Scanner<R> {
        fn new(r: R) -> Self {
            Self {
                r,
                buf: vec![0u8; 1 << 16],
                idx: 0,
            }
        }
        fn next_token(&mut self) -> Option<String> {
            let mut out = Vec::new();
            loop {
                if self.idx >= self.buf.len() {
                    self.buf.resize(1 << 16, 0);
                    self.idx = 0;
                }
                // Fill buffer by reading into it, but keep a separate temp.
                // Simpler: read byte-by-byte with internal refill.
                break;
            }
            drop(out);
            None
        }
    }

    // Implement a robust streaming whitespace scanner.
    struct FastScanner<R: Read> {
        r: R,
        buf: [u8; 1 << 16],
        len: usize,
        pos: usize,
    }
    impl<R: Read> FastScanner<R> {
        fn new(r: R) -> Self {
            Self {
                r,
                buf: [0u8; 1 << 16],
                len: 0,
                pos: 0,
            }
        }
        fn refill(&mut self) -> bool {
            match self.r.read(&mut self.buf) {
                Ok(0) => false,
                Ok(n) => {
                    self.len = n;
                    self.pos = 0;
                    true
                }
                Err(_) => false,
            }
        }
        fn next(&mut self) -> Option<String> {
            let mut s = Vec::new();
            loop {
                if self.pos >= self.len {
                    if !self.refill() {
                        break;
                    }
                }
                while self.pos < self.len && self.buf[self.pos].is_ascii_whitespace() {
                    self.pos += 1;
                }
                if self.pos < self.len {
                    break;
                }
            }
            loop {
                if self.pos >= self.len {
                    if !self.refill() {
                        break;
                    }
                }
                if self.pos >= self.len {
                    break;
                }
                let b = self.buf[self.pos];
                if b.is_ascii_whitespace() {
                    break;
                }
                s.push(b);
                self.pos += 1;
            }
            if s.is_empty() {
                None
            } else {
                Some(String::from_utf8_lossy(&s).to_string())
            }
        }
    }

    let mut fs = FastScanner::new(&mut stdin);

    let width: usize = fs.next().and_then(|t| parse_usize(&t)).unwrap_or(0);
    let height: usize = fs.next().and_then(|t| parse_usize(&t)).unwrap_or(0);
    let gravity: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(1.62);
    let max_thrust: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(20.0);
    let fuel: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(10.0);

    let start_x: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(0.0);
    let start_y: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(0.0);
    let target_x: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(0.0);
    let target_y: f64 = fs.next().and_then(|t| parse_f64(&t)).unwrap_or(0.0);

    let info = MapInfo {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    // Parse map rows, but keep only landing zone extents (L cells). If missing, fallback.
    let mut landing_cells: Vec<(i32, i32)> = Vec::new();
    for y in 0..height {
        if let Some(row) = fs.next() {
            for (x, ch) in row.chars().take(width).enumerate() {
                if ch == 'L' {
                    landing_cells.push((x as i32, y as i32));
                }
            }
        } else {
            break;
        }
    }

    let (zone_left, zone_right, zone_y) = compute_landing_zone(&landing_cells, &info);

    // Spawn reader thread for state updates using stdin handle.
    // We already borrowed stdin; release borrow by dropping fs and stdin lock.
    drop(fs);
    drop(stdin);

    let (tx, rx) = mpsc::sync_channel::<State>(64);

    // Reader uses raw stdin (unlocked) now.
    thread::spawn(move || {
        reader_thread(tx);
    });

    controller(info, zone_left, zone_right, zone_y, rx);
}
