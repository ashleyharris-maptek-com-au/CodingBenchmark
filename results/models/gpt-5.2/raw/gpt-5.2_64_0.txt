declare const require: any;
const fs = require("fs");

type Cell = [number, number]; // [x,y]

function clampInt(v: any, def: number): number {
  const n = Number(v);
  return Number.isFinite(n) ? (n | 0) : def;
}

function keyOf(x: number, y: number): number {
  return (y << 16) ^ x;
}

function parseInput(input: string): {
  prob: number[][];
  drones: { id: any; x: number; y: number }[];
  maxSteps: number;
} {
  const data = JSON.parse(input);
  const prob = (data.probability ?? []) as number[][];
  const dronesIn = (data.drones ?? []) as any[];
  const drones = dronesIn.map((d) => ({
    id: d?.id,
    x: clampInt(d?.x, 0),
    y: clampInt(d?.y, 0),
  }));
  const maxSteps = clampInt(data.maxStepsPerDrone, 0);
  return { prob, drones, maxSteps };
}

function inBounds(x: number, y: number, w: number, h: number): boolean {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function isObstacle(prob: number[][], x: number, y: number): boolean {
  return prob[y][x] === -1;
}

function cellValue(prob: number[][], x: number, y: number): number {
  const v = prob[y][x];
  return v > 0 ? v : 0;
}

function planPaths(
  prob: number[][],
  drones: { x: number; y: number }[],
  maxStepsPerDrone: number
): Cell[][] {
  const h = prob.length | 0;
  const w = h > 0 ? (prob[0].length | 0) : 0;

  const paths: Cell[][] = [];
  if (w <= 0 || h <= 0 || drones.length === 0 || maxStepsPerDrone <= 0) {
    for (let i = 0; i < drones.length; i++) paths.push([]);
    return paths;
  }

  // Global "covered" map for uniqueness across all drones.
  // Only matters for positive-valued cells.
  const covered = new Set<number>();

  // Greedy, scalable: at each step, each drone targets the best nearby uncovered cell.
  // Use an adaptive radius search capped to keep per-step work bounded.
  const DIRS: number[] = [0, 0, 1, 0, -1, 0, 0, 1, 0, -1]; // (dx,dy) pairs: stay, right, left, down, up

  // Precompute "walkable" quick check.
  function walkable(x: number, y: number): boolean {
    return inBounds(x, y, w, h) && !isObstacle(prob, x, y);
  }

  // If a starting cell is invalid/obstacle, try to nudge to nearest walkable via small BFS.
  function normalizeStart(x0: number, y0: number): { x: number; y: number } {
    if (walkable(x0, y0)) return { x: x0, y: y0 };
    const maxDepth = 64;
    const qx = new Int32Array(w * h);
    const qy = new Int32Array(w * h);
    const dist = new Int32Array(w * h);
    dist.fill(-1);
    const startIdx = y0 * w + x0;
    if (inBounds(x0, y0, w, h)) dist[startIdx] = 0;
    let qh = 0,
      qt = 0;
    if (inBounds(x0, y0, w, h)) {
      qx[qt] = x0;
      qy[qt] = y0;
      qt++;
    } else {
      // Out of bounds: clamp to grid then proceed
      const cx = Math.max(0, Math.min(w - 1, x0));
      const cy = Math.max(0, Math.min(h - 1, y0));
      dist[cy * w + cx] = 0;
      qx[qt] = cx;
      qy[qt] = cy;
      qt++;
    }
    while (qh < qt) {
      const x = qx[qh];
      const y = qy[qh];
      const d = dist[y * w + x];
      qh++;
      if (d > maxDepth) break;
      if (walkable(x, y)) return { x, y };
      // 4-neighborhood
      const nx1 = x + 1,
        nx2 = x - 1,
        ny1 = y + 1,
        ny2 = y - 1;
      if (inBounds(nx1, y, w, h) && dist[y * w + nx1] === -1) {
        dist[y * w + nx1] = d + 1;
        qx[qt] = nx1;
        qy[qt] = y;
        qt++;
      }
      if (inBounds(nx2, y, w, h) && dist[y * w + nx2] === -1) {
        dist[y * w + nx2] = d + 1;
        qx[qt] = nx2;
        qy[qt] = y;
        qt++;
      }
      if (inBounds(x, ny1, w, h) && dist[ny1 * w + x] === -1) {
        dist[ny1 * w + x] = d + 1;
        qx[qt] = x;
        qy[qt] = ny1;
        qt++;
      }
      if (inBounds(x, ny2, w, h) && dist[ny2 * w + x] === -1) {
        dist[ny2 * w + x] = d + 1;
        qx[qt] = x;
        qy[qt] = ny2;
        qt++;
      }
    }
    // Fallback: find any walkable by scan (should be rare).
    for (let y = 0; y < h; y++) for (let x = 0; x < w; x++) if (walkable(x, y)) return { x, y };
    return { x: 0, y: 0 };
  }

  const states = drones.map((d) => normalizeStart(d.x, d.y));
  for (let i = 0; i < drones.length; i++) paths.push([]);

  function considerCover(x: number, y: number) {
    const v = cellValue(prob, x, y);
    if (v <= 0) return;
    covered.add(keyOf(x, y));
  }

  // Cover initial positions if valuable.
  for (let i = 0; i < states.length; i++) considerCover(states[i].x, states[i].y);

  // Bounded neighborhood search:
  // For each drone, find the best uncovered value within radius R (Manhattan),
  // where R grows slowly (up to Rmax) when no good target is nearby.
  const R_START = 6;
  const R_MAX = 22;

  // Per drone adaptive radius.
  const radii = new Int32Array(states.length);
  radii.fill(R_START);

  for (let step = 0; step < maxStepsPerDrone; step++) {
    for (let i = 0; i < states.length; i++) {
      let { x, y } = states[i];

      // Pick best target in neighborhood.
      let bestTx = x;
      let bestTy = y;
      let bestScore = -1;

      const R = radii[i];
      // Scan diamond. Complexity O(R^2) with small R.
      for (let dy = -R; dy <= R; dy++) {
        const ay = y + dy;
        if (ay < 0 || ay >= h) continue;
        const rem = R - (dy < 0 ? -dy : dy);
        const xMin = x - rem;
        const xMax = x + rem;
        const row = prob[ay];
        for (let ax = xMin; ax <= xMax; ax++) {
          if (ax < 0 || ax >= w) continue;
          const v = row[ax];
          if (v <= 0) continue;
          if (v === -1) continue;
          const k = keyOf(ax, ay);
          if (covered.has(k)) continue;
          // Heuristic: value discounted by distance.
          const dist = (ax - x < 0 ? x - ax : ax - x) + (dy < 0 ? -dy : dy);
          const score = v / (1 + dist);
          if (score > bestScore) {
            bestScore = score;
            bestTx = ax;
            bestTy = ay;
          }
        }
      }

      if (bestScore < 0) {
        // No uncovered target nearby: increase radius and do a local greedy step
        // towards any positive cell (covered or not), else wander minimally.
        radii[i] = Math.min(R_MAX, R + 2);
      } else {
        // Found target: tighten radius a bit for efficiency.
        radii[i] = Math.max(R_START, R - 1);
      }

      // Choose next move (including stay) that best improves "distance to target"
      // while also preferring to step onto uncovered value.
      let nextX = x;
      let nextY = y;
      let bestMoveScore = -1e18;

      // If we didn't find an uncovered target, set target to current (so we don't oscillate).
      const tx = bestScore < 0 ? x : bestTx;
      const ty = bestScore < 0 ? y : bestTy;

      for (let d = 0; d < DIRS.length; d += 2) {
        const nx = x + DIRS[d];
        const ny = y + DIRS[d + 1];
        if (!walkable(nx, ny)) continue;
        const dist = (nx - tx < 0 ? tx - nx : nx - tx) + (ny - ty < 0 ? ty - ny : ny - ty);
        const k = keyOf(nx, ny);
        const v = cellValue(prob, nx, ny);
        const uncoveredBonus = v > 0 && !covered.has(k) ? v : 0;
        // Score: prioritize uncovering value, then getting closer, then mild inertia.
        const moveScore = uncoveredBonus * 1000 - dist * 10 - (nx === x && ny === y ? 1 : 0);
        if (moveScore > bestMoveScore) {
          bestMoveScore = moveScore;
          nextX = nx;
          nextY = ny;
        }
      }

      states[i].x = nextX;
      states[i].y = nextY;
      paths[i].push([nextX, nextY]);
      considerCover(nextX, nextY);
    }
  }

  return paths;
}

function main() {
  const input: string = fs.readFileSync(0, "utf8");
  const { prob, drones, maxSteps } = parseInput(input);
  const paths = planPaths(
    prob,
    drones.map((d) => ({ x: d.x, y: d.y })),
    maxSteps
  );
  process.stdout.write(JSON.stringify({ paths }));
}

main();
