using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

// Maximum Clique in practice across graph densities:
// - Exact for small graphs using Bron–Kerbosch with pivoting + degeneracy ordering (bitset-based).
// - For large graphs: fast greedy coloring bound + local search + multi-start (heuristic),
//   with optional exact refinement on an induced subgraph around the best solution.
//
// Notes:
// - Maximum Clique is NP-hard; no single algorithm is guaranteed fast for 1e6 vertices in worst case.
// - This implementation aims to be robust: always outputs a valid clique; tries hard within time.
public static class Program
{
    // Hard time budget (seconds). Keep some slack for IO and printing.
    private const double TimeLimitSeconds = 295.0;
    private static long _startTicks;
    private static readonly double TickToSeconds = 1.0 / TimeSpan.TicksPerSecond;

    private static bool TimeUp(double slackSeconds = 0.0)
        => (DateTime.UtcNow.Ticks - _startTicks) * TickToSeconds >= (TimeLimitSeconds - slackSeconds);

    public static void Main()
    {
        _startTicks = DateTime.UtcNow.Ticks;

        var fs = Console.OpenStandardInput();
        var reader = new FastScanner(fs);
        int n = reader.NextInt();
        int m = reader.NextInt();
        if (n <= 0)
        {
            Console.WriteLine("0\n");
            return;
        }

        // Build adjacency list (CSR-like): store edges in vectors then compact.
        // For very large n, allocate degree array first.
        var deg = new int[n];
        int[] eu = new int[m];
        int[] ev = new int[m];
        for (int i = 0; i < m; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            eu[i] = u;
            ev[i] = v;
            if ((uint)u < (uint)n && (uint)v < (uint)n && u != v)
            {
                deg[u]++;
                deg[v]++;
            }
        }

        int[] offsets = new int[n + 1];
        for (int i = 0; i < n; i++) offsets[i + 1] = offsets[i] + deg[i];
        int[] adj = new int[offsets[n]];
        Array.Copy(offsets, 0, deg, 0, n); // reuse deg as write cursor
        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adj[deg[u]++] = v;
            adj[deg[v]++] = u;
        }

        // Sort adjacency lists for binary search intersections.
        // Parallel sort for large graphs.
        int proc = Math.Max(1, Environment.ProcessorCount);
        if (n >= 20000)
        {
            Parallel.For(0, proc, worker =>
            {
                int from = (int)((long)n * worker / proc);
                int to = (int)((long)n * (worker + 1) / proc);
                for (int v = from; v < to; v++)
                {
                    int s = offsets[v], e = offsets[v + 1];
                    if (e - s > 1) Array.Sort(adj, s, e - s);
                }
            });
        }
        else
        {
            for (int v = 0; v < n; v++)
            {
                int s = offsets[v], e = offsets[v + 1];
                if (e - s > 1) Array.Sort(adj, s, e - s);
            }
        }

        // Remove potential duplicates in adjacency lists (rare unless input has duplicates).
        // Only do if m is large enough; keep linear-time dedup.
        // Build new CSR if duplicates are found; otherwise keep.
        if (!TimeUp(10))
        {
            bool likelyDup = m > n * 2;
            if (likelyDup)
            {
                bool anyDup = false;
                for (int v = 0; v < n && !anyDup; v++)
                {
                    int s = offsets[v], e = offsets[v + 1];
                    for (int i = s + 1; i < e; i++)
                        if (adj[i] == adj[i - 1]) { anyDup = true; break; }
                }
                if (anyDup)
                {
                    int[] newOffsets = new int[n + 1];
                    for (int v = 0; v < n; v++)
                    {
                        int s = offsets[v], e = offsets[v + 1];
                        int cnt = 0;
                        int prev = int.MinValue;
                        for (int i = s; i < e; i++)
                        {
                            int x = adj[i];
                            if (x != prev) { cnt++; prev = x; }
                        }
                        newOffsets[v + 1] = newOffsets[v] + cnt;
                    }
                    int[] newAdj = new int[newOffsets[n]];
                    for (int v = 0; v < n; v++)
                    {
                        int s = offsets[v], e = offsets[v + 1];
                        int w = newOffsets[v];
                        int prev = int.MinValue;
                        for (int i = s; i < e; i++)
                        {
                            int x = adj[i];
                            if (x != prev) { newAdj[w++] = x; prev = x; }
                        }
                    }
                    offsets = newOffsets;
                    adj = newAdj;
                }
            }
        }

        // Quick lower bound: greedy on degeneracy (or degree) ordering.
        var bestClique = new List<int>();
        if (!TimeUp(2))
        {
            var seedClique = GreedyClique(n, offsets, adj);
            bestClique = seedClique;
        }

        // Decide approach based on size.
        // Exact BK feasible up to maybe 5000-20000 depending on density and time; we use cautious threshold.
        if (n <= 12000 && !TimeUp(20))
        {
            // Exact search on full graph with bitsets (needs O(n^2/word) memory if dense, but we build neighborhood bitsets on the fly).
            // For n ~ 12000, bitset length ~ 188 ulongs, per vertex would be huge; so we do induced-subgraph exact by degeneracy core.
            // Compute k-core-like degeneracy ordering and take a reduced candidate set of highest core numbers.
            int targetSize = Math.Max(bestClique.Count + 1, 2);
            var reduced = SelectReducedSubgraph(n, offsets, adj, maxVertices: Math.Min(5000, n), minCore: targetSize - 1);
            if (reduced.Count > 0 && reduced.Count <= 6000)
            {
                var exact = ExactCliqueOnInduced(reduced, offsets, adj, bestClique.Count);
                if (exact.Count > bestClique.Count) bestClique = exact;
            }
            // If still small, try exact on all vertices for very small n.
            if (n <= 2500 && !TimeUp(10))
            {
                var all = Enumerable.Range(0, n).ToList();
                var exactAll = ExactCliqueOnInduced(all, offsets, adj, bestClique.Count);
                if (exactAll.Count > bestClique.Count) bestClique = exactAll;
            }
        }

        // Heuristic phase for large graphs (and also useful for medium).
        if (!TimeUp(2))
        {
            var improved = HeuristicSearch(n, offsets, adj, bestClique);
            if (improved.Count > bestClique.Count) bestClique = improved;
        }

        // Final exact refinement on a neighborhood-induced subgraph around best clique (safe size cap).
        if (!TimeUp(8) && bestClique.Count >= 2)
        {
            var refineVertices = BuildNeighborhoodInduced(bestClique, offsets, adj, cap: 7000);
            if (refineVertices.Count >= bestClique.Count && refineVertices.Count <= 7000)
            {
                var exact = ExactCliqueOnInduced(refineVertices, offsets, adj, bestClique.Count);
                if (exact.Count > bestClique.Count) bestClique = exact;
            }
        }

        // Ensure output is a clique (defensive).
        bestClique = MakeCliqueValid(bestClique, offsets, adj);

        Console.WriteLine(bestClique.Count);
        if (bestClique.Count == 0) { Console.WriteLine(); return; }
        bestClique.Sort();
        Console.WriteLine(string.Join(" ", bestClique));
    }

    private static List<int> GreedyClique(int n, int[] offsets, int[] adj)
    {
        // Simple greedy: start from high degree vertices, extend by intersecting candidate set.
        // Build ordering by degree descending.
        int[] order = new int[n];
        int[] degree = new int[n];
        for (int i = 0; i < n; i++)
        {
            order[i] = i;
            degree[i] = offsets[i + 1] - offsets[i];
        }
        Array.Sort(order, (a, b) => degree[b].CompareTo(degree[a]));

        var best = new List<int>();
        // Try multiple starts (limited) for robustness.
        int tries = Math.Min(n, n <= 200000 ? 256 : 64);
        for (int t = 0; t < tries && !TimeUp(290); t++)
        {
            int start = order[t];
            var clique = new List<int>(best.Count + 4) { start };
            // candidate set is neighbors of start.
            var cand = new List<int>();
            int s = offsets[start], e = offsets[start + 1];
            for (int i = s; i < e; i++) cand.Add(adj[i]);
            // Sort candidates by degree descending for faster growth.
            cand.Sort((a, b) => degree[b].CompareTo(degree[a]));
            while (cand.Count > 0)
            {
                int v = cand[0];
                clique.Add(v);
                // Intersect cand with N(v)
                cand = IntersectWithNeighborhood(cand, v, offsets, adj);
            }
            if (clique.Count > best.Count) best = clique;
        }
        return best;
    }

    private static List<int> IntersectWithNeighborhood(List<int> candidates, int v, int[] offsets, int[] adj)
    {
        int s = offsets[v], e = offsets[v + 1];
        var res = new List<int>(candidates.Count);
        int i = 0, j = s;
        // candidates is not necessarily sorted by id; so use hash-based filter if big, else binary search.
        // For speed, if candidates is large, build a boolean hashset via sorting candidates by id.
        // We'll do sort-by-id and two-pointer intersection (good enough).
        candidates.Sort();
        while (i < candidates.Count && j < e)
        {
            int a = candidates[i], b = adj[j];
            if (a == b) { res.Add(a); i++; j++; }
            else if (a < b) i++;
            else j++;
        }
        return res;
    }

    private static List<int> MakeCliqueValid(List<int> vertices, int[] offsets, int[] adj)
    {
        // Remove any vertex that breaks clique property (shouldn't happen, but keeps correctness).
        var clique = new List<int>(vertices);
        clique.Sort();
        for (int i = 0; i < clique.Count; i++)
        {
            int vi = clique[i];
            for (int j = i + 1; j < clique.Count; j++)
            {
                int vj = clique[j];
                if (!HasEdge(vi, vj, offsets, adj))
                {
                    // Drop vj (arbitrary) and restart scan.
                    clique.RemoveAt(j);
                    j--;
                }
            }
        }
        return clique;
    }

    private static bool HasEdge(int u, int v, int[] offsets, int[] adj)
    {
        int s = offsets[u], e = offsets[u + 1];
        // Binary search in sorted adjacency list.
        int lo = s, hi = e - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = adj[mid];
            if (x == v) return true;
            if (x < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    private static List<int> HeuristicSearch(int n, int[] offsets, int[] adj, List<int> currentBest)
    {
        // Multi-start greedy + local improvement by swap/add.
        // For dense graphs, greedy finds near-optimal quickly.
        // For sparse graphs, focus on high-core vertices.
        int bestSize = currentBest.Count;
        var best = new List<int>(currentBest);

        int[] degree = new int[n];
        for (int i = 0; i < n; i++) degree[i] = offsets[i + 1] - offsets[i];

        // Sample starting vertices: top degrees + random.
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        if (n <= 500000) Array.Sort(order, (a, b) => degree[b].CompareTo(degree[a]));
        else
        {
            // partial selection for huge n: pick top K via heap-like scan
            Array.Sort(order, 0, Math.Min(n, 200000), Comparer<int>.Create((a, b) => degree[b].CompareTo(degree[a])));
        }

        int starts = n <= 200000 ? 256 : 96;
        starts = Math.Min(starts, n);
        var rng = new XorShift64((ulong)(DateTime.UtcNow.Ticks * 6364136223846793005L + n));

        // Use tasks for parallel starts (bounded).
        int parallel = Math.Min(Environment.ProcessorCount, 8);
        var lockObj = new object();
        int idx = 0;

        Parallel.For(0, parallel, worker =>
        {
            var localBest = new List<int>();
            while (true)
            {
                int t;
                lock (lockObj)
                {
                    if (idx >= starts) break;
                    t = idx++;
                }
                if (TimeUp(3)) break;
                int start = (t < starts / 2) ? order[t] : (int)(rng.NextUInt() % (uint)n);
                var clique = GreedyFromSeed(start, degree, offsets, adj);
                LocalImprove(clique, degree, offsets, adj, maxIters: 200);
                if (clique.Count > localBest.Count) localBest = clique;
                if (localBest.Count >= bestSize + 2 && !TimeUp(3))
                {
                    // quick attempt to expand from this clique by trying common neighbors
                    ExpandByCommonNeighbors(localBest, offsets, adj, degree);
                }
            }
            lock (lockObj)
            {
                if (localBest.Count > best.Count) best = localBest;
            }
        });

        return best;
    }

    private static List<int> GreedyFromSeed(int start, int[] degree, int[] offsets, int[] adj)
    {
        var clique = new List<int>(8) { start };
        // Candidate list: neighbors of start sorted by degree desc.
        var cand = new List<int>();
        int s = offsets[start], e = offsets[start + 1];
        for (int i = s; i < e; i++) cand.Add(adj[i]);
        cand.Sort((a, b) => degree[b].CompareTo(degree[a]));
        while (cand.Count > 0 && !TimeUp(2))
        {
            int v = cand[0];
            clique.Add(v);
            cand = IntersectWithNeighborhood(cand, v, offsets, adj);
        }
        return clique;
    }

    private static void LocalImprove(List<int> clique, int[] degree, int[] offsets, int[] adj, int maxIters)
    {
        // 1-opt and 2-opt like improvements:
        // Try to add a vertex that connects to all in clique; if none, try swap (remove one, add one).
        if (clique.Count == 0) return;
        var inClique = new HashSet<int>(clique);
        for (int iter = 0; iter < maxIters && !TimeUp(2); iter++)
        {
            bool changed = false;

            // Try add: scan neighbors of a high-degree clique vertex.
            int pivot = clique[0];
            for (int i = 1; i < clique.Count; i++)
                if (degree[clique[i]] > degree[pivot]) pivot = clique[i];

            int s = offsets[pivot], e = offsets[pivot + 1];
            for (int ai = s; ai < e && !TimeUp(2); ai++)
            {
                int v = adj[ai];
                if (inClique.Contains(v)) continue;
                bool ok = true;
                for (int j = 0; j < clique.Count; j++)
                {
                    if (!HasEdge(v, clique[j], offsets, adj)) { ok = false; break; }
                }
                if (ok)
                {
                    clique.Add(v);
                    inClique.Add(v);
                    changed = true;
                    break;
                }
            }
            if (changed) continue;

            // Try swap: find v that connects to all but one clique vertex.
            // Choose candidate vertices from neighbors of pivot (cheap).
            for (int ai = s; ai < e && !TimeUp(2); ai++)
            {
                int v = adj[ai];
                if (inClique.Contains(v)) continue;
                int missing = -1;
                for (int j = 0; j < clique.Count; j++)
                {
                    if (!HasEdge(v, clique[j], offsets, adj))
                    {
                        if (missing != -1) { missing = -2; break; }
                        missing = clique[j];
                    }
                }
                if (missing >= 0)
                {
                    // Swap out missing for v.
                    inClique.Remove(missing);
                    clique.Remove(missing);
                    clique.Add(v);
                    inClique.Add(v);
                    changed = true;
                    break;
                }
            }
            if (!changed) break;
        }
    }

    private static void ExpandByCommonNeighbors(List<int> clique, int[] offsets, int[] adj, int[] degree)
    {
        // Compute common neighbors of current clique via iterative intersection of adjacency lists (sorted by id).
        clique.Sort();
        if (clique.Count == 0) return;
        List<int> common = null!;
        for (int i = 0; i < clique.Count; i++)
        {
            int v = clique[i];
            var neigh = new List<int>();
            int s = offsets[v], e = offsets[v + 1];
            for (int j = s; j < e; j++) neigh.Add(adj[j]);
            if (common == null) common = neigh;
            else
            {
                common.Sort();
                neigh.Sort();
                common = IntersectSorted(common, neigh);
            }
            if (common.Count == 0) break;
        }
        if (common == null || common.Count == 0) return;
        // Add greedily highest degree common neighbors.
        common.Sort((a, b) => degree[b].CompareTo(degree[a]));
        var inClique = new HashSet<int>(clique);
        foreach (var v in common)
        {
            if (inClique.Contains(v)) continue;
            bool ok = true;
            for (int j = 0; j < clique.Count; j++)
                if (!HasEdge(v, clique[j], offsets, adj)) { ok = false; break; }
            if (ok) { clique.Add(v); inClique.Add(v); }
            if (TimeUp(2)) break;
        }
    }

    private static List<int> IntersectSorted(List<int> a, List<int> b)
    {
        int i = 0, j = 0;
        var res = new List<int>(Math.Min(a.Count, b.Count));
        while (i < a.Count && j < b.Count)
        {
            int x = a[i], y = b[j];
            if (x == y) { res.Add(x); i++; j++; }
            else if (x < y) i++;
            else j++;
        }
        return res;
    }

    private static List<int> BuildNeighborhoodInduced(List<int> clique, int[] offsets, int[] adj, int cap)
    {
        // Induced subgraph on clique plus union of their neighbors (capped).
        var set = new HashSet<int>(clique);
        foreach (var v in clique)
        {
            int s = offsets[v], e = offsets[v + 1];
            for (int i = s; i < e; i++)
            {
                set.Add(adj[i]);
                if (set.Count >= cap) break;
            }
            if (set.Count >= cap) break;
        }
        return set.ToList();
    }

    private static List<int> SelectReducedSubgraph(int n, int[] offsets, int[] adj, int maxVertices, int minCore)
    {
        // Approx degeneracy ordering and pick vertices with high core numbers (k-core pruning).
        // Linear-time core decomposition with bucket queue.
        int[] deg = new int[n];
        int maxDeg = 0;
        for (int v = 0; v < n; v++)
        {
            int d = offsets[v + 1] - offsets[v];
            deg[v] = d;
            if (d > maxDeg) maxDeg = d;
        }
        int[] bin = new int[maxDeg + 1];
        for (int v = 0; v < n; v++) bin[deg[v]]++;
        int start = 0;
        for (int d = 0; d <= maxDeg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }
        int[] vert = new int[n];
        int[] pos = new int[n];
        for (int v = 0; v < n; v++)
        {
            pos[v] = bin[deg[v]];
            vert[pos[v]] = v;
            bin[deg[v]]++;
        }
        for (int d = maxDeg; d >= 1; d--) bin[d] = bin[d - 1];
        bin[0] = 0;

        int[] core = new int[n];
        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            core[v] = deg[v];
            int s = offsets[v], e = offsets[v + 1];
            for (int j = s; j < e; j++)
            {
                int u = adj[j];
                if (deg[u] > deg[v])
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];
                    if (u != w)
                    {
                        vert[pu] = w; pos[w] = pu;
                        vert[pw] = u; pos[u] = pw;
                    }
                    bin[du]++;
                    deg[u]--;
                }
            }
            if (TimeUp(20)) break;
        }

        // Collect vertices with core >= minCore, then take top by core and degree.
        var cand = new List<int>();
        for (int v = 0; v < n; v++)
            if (core[v] >= minCore) cand.Add(v);
        if (cand.Count == 0) return cand;
        cand.Sort((a, b) =>
        {
            int ca = core[a], cb = core[b];
            if (cb != ca) return cb.CompareTo(ca);
            int da = offsets[a + 1] - offsets[a];
            int db = offsets[b + 1] - offsets[b];
            return db.CompareTo(da);
        });
        if (cand.Count > maxVertices) cand.RemoveRange(maxVertices, cand.Count - maxVertices);
        return cand;
    }

    private static List<int> ExactCliqueOnInduced(List<int> vertices, int[] offsets, int[] adj, int incumbent)
    {
        // Exact maximum clique on induced subgraph using Bron–Kerbosch with pivoting and coloring bound.
        // Map original vertex ids -> 0..k-1.
        int k = vertices.Count;
        var map = new Dictionary<int, int>(k);
        for (int i = 0; i < k; i++) map[vertices[i]] = i;

        // Build adjacency bitsets for induced graph.
        int words = (k + 63) >> 6;
        ulong[][] nb = new ulong[k][];
        for (int i = 0; i < k; i++) nb[i] = new ulong[words];

        for (int i = 0; i < k; i++)
        {
            int v = vertices[i];
            int s = offsets[v], e = offsets[v + 1];
            for (int idx = s; idx < e; idx++)
            {
                int u = adj[idx];
                if (map.TryGetValue(u, out int j))
                {
                    nb[i][j >> 6] |= 1UL << (j & 63);
                }
            }
        }

        // Initial candidate set: all vertices.
        var P = new ulong[words];
        for (int w = 0; w < words; w++) P[w] = ulong.MaxValue;
        int extraBits = (words << 6) - k;
        if (extraBits > 0) P[words - 1] &= ulong.MaxValue >> extraBits;

        var R = new List<int>(64);
        var best = new List<int>();

        void BK(ulong[] pset, List<int> current)
        {
            if (TimeUp(1.5)) return;
            int pCount = PopCount(pset);
            if (pCount == 0)
            {
                if (current.Count > best.Count)
                    best = new List<int>(current);
                return;
            }
            // Upper bound via greedy coloring.
            int[] colors = new int[pCount];
            int[] verts = new int[pCount];
            int used = ColorSort(pset, nb, colors, verts);
            // Explore in reverse color order (largest bound first).
            for (int idx = used - 1; idx >= 0; idx--)
            {
                if (current.Count + colors[idx] <= best.Count) return;
                int v = verts[idx];
                // Include v
                current.Add(v);
                var newP = And(pset, nb[v]);
                BK(newP, current);
                current.RemoveAt(current.Count - 1);
                // Exclude v
                ClearBit(pset, v);
                if (TimeUp(1.5)) return;
            }
        }

        // incumbent pruning
        if (incumbent > 0 && k <= incumbent) return new List<int>();

        BK(P, R);
        if (best.Count <= incumbent) return new List<int>();

        // Map back to original ids.
        var res = new List<int>(best.Count);
        for (int i = 0; i < best.Count; i++) res.Add(vertices[best[i]]);
        return res;
    }

    private static int ColorSort(ulong[] pset, ulong[][] nb, int[] colors, int[] verts)
    {
        // Greedy coloring producing an ordering and bounds:
        // verts[i] is the i-th selected vertex, colors[i] is the number of colors used up to i (upper bound).
        // Uses repeated maximal independent sets on the complement within pset approximation.
        // Simple DSATUR-ish greedy would be slower with bitsets; we do classical greedy coloring:
        // For each color, take a maximal set of mutually non-adjacent vertices by iteratively picking any vertex
        // and removing its neighbors.
        int words = pset.Length;
        var remaining = (ulong[])pset.Clone();
        int count = 0;
        int color = 0;
        while (Any(remaining))
        {
            color++;
            var avail = (ulong[])remaining.Clone();
            // build one color class
            while (Any(avail))
            {
                int v = FirstBit(avail);
                verts[count] = v;
                colors[count] = color;
                count++;
                // remove v from remaining
                ClearBit(remaining, v);
                // remove v and its neighbors from avail
                ClearBit(avail, v);
                var neigh = nb[v];
                for (int w = 0; w < words; w++) avail[w] &= ~neigh[w];
            }
        }
        return count;
    }

    private static bool Any(ulong[] a)
    {
        for (int i = 0; i < a.Length; i++) if (a[i] != 0) return true;
        return false;
    }

    private static ulong[] And(ulong[] a, ulong[] b)
    {
        int n = a.Length;
        var r = new ulong[n];
        for (int i = 0; i < n; i++) r[i] = a[i] & b[i];
        return r;
    }

    private static void ClearBit(ulong[] a, int bit)
    {
        a[bit >> 6] &= ~(1UL << (bit & 63));
    }

    private static int FirstBit(ulong[] a)
    {
        for (int w = 0; w < a.Length; w++)
        {
            ulong x = a[w];
            if (x == 0) continue;
            int t = BitOperations.TrailingZeroCount(x);
            return (w << 6) + t;
        }
        return -1;
    }

    private static int PopCount(ulong[] a)
    {
        int c = 0;
        for (int i = 0; i < a.Length; i++) c += BitOperations.PopCount(a[i]);
        return c;
    }

    private struct XorShift64
    {
        private ulong _x;
        public XorShift64(ulong seed) { _x = seed == 0 ? 88172645463393265UL : seed; }
        public uint NextUInt()
        {
            ulong x = _x;
            x ^= x << 7;
            x ^= x >> 9;
            _x = x;
            return (uint)(x & 0xFFFFFFFF);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c != -1);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private static class BitOperations
    {
        public static int PopCount(ulong value)
        {
            // Use hardware intrinsic if available via built-in, else software.
            // .NET provides System.Numerics.BitOperations but to stay compatible with older csc setups we implement.
            value -= (value >> 1) & 0x5555555555555555UL;
            value = (value & 0x3333333333333333UL) + ((value >> 2) & 0x3333333333333333UL);
            return (int)((((value + (value >> 4)) & 0x0F0F0F0F0F0F0F0FUL) * 0x0101010101010101UL) >> 56);
        }

        public static int TrailingZeroCount(ulong value)
        {
            // De Bruijn sequence method
            if (value == 0) return 64;
            ulong isolated = value & (~value + 1);
            int idx = (int)((isolated * 0x03F79D71B4CB0A89UL) >> 58);
            return _tzLookup[idx];
        }

        private static readonly int[] _tzLookup = new int[64]
        {
            0, 1, 2, 53, 3, 7, 54, 27,
            4, 38, 41, 8, 34, 55, 48, 28,
            62, 5, 39, 46, 44, 42, 22, 9,
            24, 35, 59, 56, 49, 18, 29, 11,
            63, 52, 6, 26, 37, 40, 33, 47,
            61, 45, 43, 21, 23, 58, 17, 10,
            51, 25, 36, 32, 60, 20, 57, 16,
            50, 31, 19, 15, 30, 14, 13, 12
        };
    }
