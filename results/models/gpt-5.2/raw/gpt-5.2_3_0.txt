use std::collections::VecDeque;
use std::io::{self, Read};

// Fast RNG (SplitMix64)
#[derive(Clone)]
struct Rng {
    state: u64,
}
impl Rng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        let mut z = {
            self.state = self.state.wrapping_add(0x9E3779B97F4A7C15);
            self.state
        };
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
    fn next_f64(&mut self) -> f64 {
        // [0,1)
        let x = self.next_u64() >> 11;
        (x as f64) * (1.0 / ((1u64 << 53) as f64))
    }
    fn gen_range_usize(&mut self, lo: usize, hi: usize) -> usize {
        if hi <= lo + 1 {
            return lo;
        }
        let span = (hi - lo) as u64;
        (lo as u64 + (self.next_u64() % span)) as usize
    }
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    // Adjacency (store undirected, keep self loops but ignore for forces)
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut deg: Vec<usize> = vec![0; n];
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
        if u != v {
            deg[u] += 1;
            deg[v] += 1;
        }
    }

    // Determine density/scale and choose strategy.
    // Heuristics:
    // - Very large N: ultra-fast: per-component BFS order on circle / grid.
    // - Medium N: multi-component circle with degree-based order.
    // - Small N: light force-directed refinement with Barnes-Hut-ish sampling (random repulsion).
    let avg_deg = if n == 0 { 0.0 } else { (2.0 * (m as f64)) / (n as f64) };

    let mut x: Vec<f64> = vec![0.0; n];
    let mut y: Vec<f64> = vec![0.0; n];

    // Components via BFS
    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut q = VecDeque::new();
    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        comp_id[s] = comps.len();
        q.clear();
        q.push_back(s);
        let mut nodes = Vec::new();
        while let Some(u) = q.pop_front() {
            nodes.push(u);
            for &v in &adj[u] {
                if v >= n {
                    continue;
                }
                if comp_id[v] == usize::MAX {
                    comp_id[v] = comp_id[s];
                    q.push_back(v);
                }
            }
        }
        comps.push(nodes);
    }

    // Sort components by size descending
    let mut comp_order: Vec<usize> = (0..comps.len()).collect();
    comp_order.sort_by(|&a, &b| comps[b].len().cmp(&comps[a].len()));

    // Layout components on a grid of "tiles"
    let k = comps.len().max(1);
    let grid_cols = (k as f64).sqrt().ceil() as usize;
    let grid_cols = grid_cols.max(1);
    let tile_gap = 10.0f64;
    let mut tile_x = vec![0.0; k];
    let mut tile_y = vec![0.0; k];
    for (idx, &cid) in comp_order.iter().enumerate() {
        let r = idx / grid_cols;
        let c = idx % grid_cols;
        tile_x[cid] = (c as f64) * tile_gap;
        tile_y[cid] = (r as f64) * tile_gap;
    }

    // Helper: BFS-based ordering within component (ties by degree)
    fn bfs_order(nodes: &[usize], adj: &Vec<Vec<usize>>, deg: &Vec<usize>) -> Vec<usize> {
        let mut in_comp = vec![false; nodes.len()];
        // Map global node -> local index via hash-like sort mapping:
        // For performance, do O(k log k) sort and binary_search mapping.
        let mut sorted = nodes.to_vec();
        sorted.sort_unstable();
        for (i, &u) in sorted.iter().enumerate() {
            in_comp[i] = true;
            let _ = u;
        }
        let mut seen = vec![false; nodes.len()];
        let mut order = Vec::with_capacity(nodes.len());
        let mut q = VecDeque::new();
        // Start from max degree node
        let mut start = nodes[0];
        for &u in nodes {
            if deg[u] > deg[start] {
                start = u;
            }
        }
        // local index lookup via binary_search on sorted
        let local = |sorted: &Vec<usize>, u: usize| -> Option<usize> {
            sorted.binary_search(&u).ok()
        };
        if let Some(li) = local(&sorted, start) {
            seen[li] = true;
        }
        q.push_back(start);
        while let Some(u) = q.pop_front() {
            order.push(u);
            // neighbors in component, prioritize higher degree
            let mut nbrs: Vec<usize> = adj[u]
                .iter()
                .copied()
                .filter(|&v| local(&sorted, v).is_some())
                .collect();
            nbrs.sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));
            for v in nbrs {
                if let Some(li) = local(&sorted, v) {
                    if !seen[li] {
                        seen[li] = true;
                        q.push_back(v);
                    }
                }
            }
        }
        // Add any isolated/unreached nodes (due to self-loops only etc.)
        if order.len() != nodes.len() {
            // remaining in degree order
            let mut rem: Vec<usize> = nodes
                .iter()
                .copied()
                .filter(|&u| {
                    // check seen
                    match local(&sorted, u) {
                        Some(li) => !seen[li],
                        None => true,
                    }
                })
                .collect();
            rem.sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));
            order.extend(rem);
        }
        order
    }

    // Place nodes per component
    for (cid, nodes) in comps.iter().enumerate() {
        let sz = nodes.len().max(1);
        let bx = tile_x[cid];
        let by = tile_y[cid];

        // Large components: circle layout with BFS/degree ordering
        let order = bfs_order(nodes, &adj, &deg);
        let radius = (sz as f64).sqrt().max(1.0) * 0.7 + 1.0;
        for (i, &u) in order.iter().enumerate() {
            let t = (i as f64) * (std::f64::consts::TAU / (sz as f64));
            x[u] = bx + radius * t.cos();
            y[u] = by + radius * t.sin();
        }
    }

    // Optional refinement: only for smaller graphs
    let do_refine = n <= 5000 && m <= 200_000;
    if do_refine {
        let mut rng = Rng::new((n as u64).wrapping_mul(0xA24BAED4963EE407) ^ (m as u64));
        // Parameters depend on size/density
        let iterations: usize = match n {
            0..=80 => 400,
            81..=300 => 250,
            301..=1200 => 120,
            _ => 60,
        };
        let samples_per_node: usize = match n {
            0..=200 => 30,
            201..=800 => 12,
            801..=2000 => 6,
            _ => 3,
        };
        let temperature = 1.5 / (1.0 + avg_deg * 0.05);
        let k_attr = 0.02;
        let k_rep = 0.05;
        let eps = 1e-9;

        let mut dx = vec![0.0f64; n];
        let mut dy = vec![0.0f64; n];

        for itn in 0..iterations {
            dx.fill(0.0);
            dy.fill(0.0);

            // Attractive forces along edges
            // (Skip self-loops; parallel edges simply add more attraction, OK)
            for u in 0..n {
                let xu = x[u];
                let yu = y[u];
                for &v in &adj[u] {
                    if v <= u || v >= n || v == u {
                        continue;
                    }
                    let mut vx = x[v] - xu;
                    let mut vy = y[v] - yu;
                    let dist2 = vx * vx + vy * vy + eps;
                    let dist = dist2.sqrt();
                    // spring: pull to length ~1.0
                    let force = k_attr * (dist - 1.0);
                    vx /= dist;
                    vy /= dist;
                    let fx = force * vx;
                    let fy = force * vy;
                    dx[u] += fx;
                    dy[u] += fy;
                    dx[v] -= fx;
                    dy[v] -= fy;
                }
            }

            // Repulsive forces by random sampling (cheap approximation)
            for u in 0..n {
                let xu = x[u];
                let yu = y[u];
                for _ in 0..samples_per_node {
                    let v = rng.gen_range_usize(0, n);
                    if v == u {
                        continue;
                    }
                    let vx = xu - x[v];
                    let vy = yu - y[v];
                    let dist2 = vx * vx + vy * vy + 0.01; // soften
                    let inv = 1.0 / dist2;
                    let force = k_rep * inv;
                    dx[u] += force * vx;
                    dy[u] += force * vy;
                }
            }

            // Apply displacements with cooling
            let cool = 1.0 - (itn as f64) / (iterations as f64);
            let step = temperature * cool.max(0.05);
            for u in 0..n {
                let mut fx = dx[u];
                let mut fy = dy[u];
                let norm = (fx * fx + fy * fy).sqrt();
                if norm > 0.0 {
                    let scale = step / norm.min(10.0);
                    fx *= scale;
                    fy *= scale;
                }
                x[u] += fx;
                y[u] += fy;
            }

            // Mild jitter reduction / bounding
            if itn % 40 == 39 {
                // recentre each component tile area slightly to avoid drift
                let mut cx = vec![0.0f64; comps.len()];
                let mut cy = vec![0.0f64; comps.len()];
                let mut cnt = vec![0usize; comps.len()];
                for u in 0..n {
                    let cid = comp_id[u];
                    cx[cid] += x[u];
                    cy[cid] += y[u];
                    cnt[cid] += 1;
                }
                for cid in 0..comps.len() {
                    if cnt[cid] == 0 {
                        continue;
                    }
                    cx[cid] /= cnt[cid] as f64;
                    cy[cid] /= cnt[cid] as f64;
                    let tx = tile_x[cid];
                    let ty = tile_y[cid];
                    let shift_x = (tx - cx[cid]) * 0.15;
                    let shift_y = (ty - cy[cid]) * 0.15;
                    for &u in &comps[cid] {
                        x[u] += shift_x;
                        y[u] += shift_y;
                    }
                }
            }
        }
    } else {
        // For very large graphs, add tiny deterministic jitter to break symmetries
        let mut rng = Rng::new((n as u64).wrapping_mul(0xD6E8FEB86659FD93) ^ 0x9C3D_1A2B_3C4D_5E6F);
        let jitter = 0.01;
        for u in 0..n {
            let jx = (rng.next_f64() - 0.5) * jitter;
            let jy = (rng.next_f64() - 0.5) * jitter;
            x[u] += jx;
            y[u] += jy;
        }
    }

    // Final normalization: scale to reasonable range to avoid huge numbers for extreme N.
    // Compute bounding box and scale to ~[-1000,1000].
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for u in 0..n {
        minx = minx.min(x[u]);
        maxx = maxx.max(x[u]);
        miny = miny.min(y[u]);
        maxy = maxy.max(y[u]);
    }
    let spanx = (maxx - minx).max(1e-9);
    let spany = (maxy - miny).max(1e-9);
    let scale = 2000.0 / spanx.max(spany);
    let cx = (minx + maxx) * 0.5;
    let cy = (miny + maxy) * 0.5;
    for u in 0..n {
        x[u] = (x[u] - cx) * scale;
        y[u] = (y[u] - cy) * scale;
    }

    // Output N lines (no prefix to keep simple)
    // Use fixed precision for stability
    let mut out = String::with_capacity(n * 24);
    for u in 0..n {
        // Avoid -0.000000
        let mut xx = x[u];
        let mut yy = y[u];
        if xx.abs() < 1e-12 {
            xx = 0.0;
        }
        if yy.abs() < 1e-12 {
            yy = 0.0;
        }
        out.push_str(&format!("{:.6} {:.6}\n", xx, yy));
    }
    print!("{out}");
}
