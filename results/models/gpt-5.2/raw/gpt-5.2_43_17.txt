struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    // Brick pattern: 8x16 UV scale with staggered rows (half-brick offset every other row)
    float2 scale = float2(8.0, 16.0);
    float2 uvS = input.uv * scale;

    // Determine row parity in Y and apply half-brick offset in X on odd rows
    float row = floor(uvS.y);
    float xOffset = (fmod(row, 2.0) >= 1.0) ? 0.5 : 0.0;
    uvS.x += xOffset;

    float2 cell = frac(uvS);

    // Mortar thickness in local cell units
    const float mortarX = 0.05;
    const float mortarY = 0.10;

    float mortarMask = (cell.x < mortarX) || (cell.x > (1.0 - mortarX)) ||
                       (cell.y < mortarY) || (cell.y > (1.0 - mortarY));

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);
    float3 baseColor = mortarMask ? mortarColor : brickColor;

    return float4(baseColor * shade, 1.0);
}
