const fs = require('fs');

// Fleet routing heuristic (VRP-like) for benchmark scoring.
// Input: { depot:[x,y], truckCount:number, packages:[{id,x,y},...] }
// Output: {"routes":[[id,...], ...]}

'use strict';

function dist(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

function makeSeededRng(seed) {
  let x = (seed | 0) || 123456789;
  return function next() {
    x ^= x << 13;
    x ^= x >>> 17;
    x ^= x << 5;
    return (x >>> 0) / 4294967296;
  };
}

function hashIntFromData(depot, truckCount, n) {
  const sx = (Math.floor(depot[0] * 1000) | 0);
  const sy = (Math.floor(depot[1] * 1000) | 0);
  let h = 2166136261 >>> 0;
  h ^= sx; h = Math.imul(h, 16777619) >>> 0;
  h ^= sy; h = Math.imul(h, 16777619) >>> 0;
  h ^= (truckCount | 0); h = Math.imul(h, 16777619) >>> 0;
  h ^= (n | 0); h = Math.imul(h, 16777619) >>> 0;
  return h | 0;
}

function clamp(v, lo, hi) {
  return v < lo ? lo : (v > hi ? hi : v);
}

function angleFromDepot(pkg, depotX, depotY) {
  return Math.atan2(pkg.y - depotY, pkg.x - depotX);
}

function farthestFirstSeeds(pkgs, depotX, depotY, k, rng) {
  const n = pkgs.length;
  if (k <= 1) return [0];
  let first = 0;
  let bestD = -1;
  for (let i = 0; i < n; i++) {
    const p = pkgs[i];
    const d = (p.x - depotX) * (p.x - depotX) + (p.y - depotY) * (p.y - depotY);
    if (d > bestD) { bestD = d; first = i; }
  }
  const seeds = [first];
  const minDist2 = new Float64Array(n);
  for (let i = 0; i < n; i++) minDist2[i] = Infinity;
  for (let i = 0; i < n; i++) {
    const p = pkgs[i];
    const s = pkgs[first];
    const dx = p.x - s.x, dy = p.y - s.y;
    minDist2[i] = dx * dx + dy * dy;
  }
  const maxExact = 64;
  const sampleSize = 4096;
  const chooseFrom = (k > maxExact && n > sampleSize) ? sampleSize : n;
  while (seeds.length < k) {
    let idx = 0;
    let best = -1;
    if (chooseFrom === n) {
      for (let i = 0; i < n; i++) {
        const d2 = minDist2[i];
        if (d2 > best) { best = d2; idx = i; }
      }
    } else {
      for (let t = 0; t < chooseFrom; t++) {
        const i = (rng() * n) | 0;
        const d2 = minDist2[i];
        if (d2 > best) { best = d2; idx = i; }
      }
    }
    seeds.push(idx);
    const s = pkgs[idx];
    for (let i = 0; i < n; i++) {
      const p = pkgs[i];
      const dx = p.x - s.x, dy = p.y - s.y;
      const d2 = dx * dx + dy * dy;
      if (d2 < minDist2[i]) minDist2[i] = d2;
    }
  }
  return seeds;
}

function kmeansCluster(pkgs, depotX, depotY, k, seed) {
  const n = pkgs.length;
  if (k <= 1) return { assignment: new Int32Array(n) };
  const rng = makeSeededRng(seed);
  const seedIdx = farthestFirstSeeds(pkgs, depotX, depotY, k, rng);
  const centersX = new Float64Array(k);
  const centersY = new Float64Array(k);
  for (let c = 0; c < k; c++) {
    const p = pkgs[seedIdx[c]];
    centersX[c] = p.x;
    centersY[c] = p.y;
  }
  const assignment = new Int32Array(n);
  const iters = 6;
  for (let iter = 0; iter < iters; iter++) {
    let changed = 0;
    for (let i = 0; i < n; i++) {
      const p = pkgs[i];
      let bestC = 0;
      let bestD2 = Infinity;
      for (let c = 0; c < k; c++) {
        const dx = p.x - centersX[c];
        const dy = p.y - centersY[c];
        const d2 = dx * dx + dy * dy;
        if (d2 < bestD2) { bestD2 = d2; bestC = c; }
      }
      if (assignment[i] !== bestC) { assignment[i] = bestC; changed++; }
    }
    if (changed === 0 && iter > 0) break;
    const sumX = new Float64Array(k);
    const sumY = new Float64Array(k);
    const count = new Int32Array(k);
    for (let i = 0; i < n; i++) {
      const c = assignment[i];
      sumX[c] += pkgs[i].x;
      sumY[c] += pkgs[i].y;
      count[c]++;
    }
    for (let c = 0; c < k; c++) {
      if (count[c] > 0) {
        centersX[c] = sumX[c] / count[c];
        centersY[c] = sumY[c] / count[c];
      } else {
        const p = pkgs[(rng() * n) | 0];
        centersX[c] = p.x;
        centersY[c] = p.y;
      }
    }
  }
  return { assignment };
}

function angleBinCluster(pkgs, depotX, depotY, k) {
  const n = pkgs.length;
  const idx = new Array(n);
  for (let i = 0; i < n; i++) idx[i] = i;
  idx.sort((a, b) => angleFromDepot(pkgs[a], depotX, depotY) - angleFromDepot(pkgs[b], depotX, depotY));
  const assignment = new Int32Array(n);
  for (let rank = 0; rank < n; rank++) {
    assignment[idx[rank]] = Math.floor((rank * k) / n);
  }
  return { assignment };
}

function orderRoute(pkgsIdx, pkgs, depotX, depotY) {
  const m = pkgsIdx.length;
  if (m <= 1) return pkgsIdx.slice();
  pkgsIdx.sort((ia, ib) => angleFromDepot(pkgs[ia], depotX, depotY) - angleFromDepot(pkgs[ib], depotX, depotY));
  const ordered = new Array(m);
  const used = new Uint8Array(m);
  ordered[0] = pkgsIdx[0];
  used[0] = 1;
  for (let pos = 1; pos < m; pos++) {
    const lastIdx = ordered[pos - 1];
    const last = pkgs[lastIdx];
    let bestJ = -1;
    let bestD2 = Infinity;
    const base = pos;
    const window = clamp(32, 8, m);
    const start = clamp(base - window, 0, m - 1);
    const end = clamp(base + window, 0, m - 1);
    for (let j = start; j <= end; j++) {
      if (used[j]) continue;
      const cand = pkgs[pkgsIdx[j]];
      const dx = cand.x - last.x, dy = cand.y - last.y;
      const d2 = dx * dx + dy * dy;
      if (d2 < bestD2) { bestD2 = d2; bestJ = j; }
    }
    if (bestJ === -1) {
      for (let j = 0; j < m; j++) if (!used[j]) { bestJ = j; break; }
    }
    used[bestJ] = 1;
    ordered[pos] = pkgsIdx[bestJ];
  }
  const iters = 2;
  const maxSeg = 32;
  for (let it = 0; it < iters; it++) {
    let improved = false;
    for (let i = 0; i < m - 3; i++) {
      const aIdx = (i === 0) ? -1 : ordered[i - 1];
      const bIdx = ordered[i];
      const ax = (aIdx === -1) ? depotX : pkgs[aIdx].x;
      const ay = (aIdx === -1) ? depotY : pkgs[aIdx].y;
      const bx = pkgs[bIdx].x, by = pkgs[bIdx].y;
      const jMax = Math.min(m - 1, i + maxSeg);
      for (let j = i + 2; j < jMax; j++) {
        const cIdx = ordered[j];
        const dIdx = (j === m - 1) ? -1 : ordered[j + 1];
        const cx = pkgs[cIdx].x, cy = pkgs[cIdx].y;
        const dx = (dIdx === -1) ? depotX : pkgs[dIdx].x;
        const dy = (dIdx === -1) ? depotY : pkgs[dIdx].y;
        const before = dist(ax, ay, bx, by) + dist(cx, cy, dx, dy);
        const after = dist(ax, ay, cx, cy) + dist(bx, by, dx, dy);
        if (after + 1e-9 < before) {
          for (let l = i, r = j; l < r; l++, r--) {
            const tmp = ordered[l];
            ordered[l] = ordered[r];
            ordered[r] = tmp;
          }
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
  return ordered;
}

function solve(data) {
  const depotX = data.depot[0];
  const depotY = data.depot[1];
  const pkgs = data.packages || [];
  const n = pkgs.length;
  const truckCount = Math.max(1, (data.truckCount | 0) || 1);
  if (n === 0) return { routes: [] };
  const k = Math.min(truckCount, n);
  const seed = hashIntFromData(data.depot, k, n);
  let assignment;
  if (k <= 64 && n <= 200000) assignment = kmeansCluster(pkgs, depotX, depotY, k, seed).assignment;
  else assignment = angleBinCluster(pkgs, depotX, depotY, k).assignment;
  const buckets = Array.from({ length: k }, () => []);
  for (let i = 0; i < n; i++) buckets[assignment[i]].push(i);
  const routes = [];
  for (let r = 0; r < k; r++) {
    const idxs = buckets[r];
    if (!idxs.length) continue;
    const orderedIdxs = orderRoute(idxs, pkgs, depotX, depotY);
    const route = new Array(orderedIdxs.length);
    for (let j = 0; j < orderedIdxs.length; j++) route[j] = pkgs[orderedIdxs[j]].id;
    routes.push(route);
  }
  return { routes };
}

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
process.stdout.write(JSON.stringify(solve(data)));
