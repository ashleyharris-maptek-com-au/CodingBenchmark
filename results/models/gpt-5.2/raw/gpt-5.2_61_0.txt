// Node.js TypeScript solution: scalable heuristic VRP (multiple routes, Euclidean metric)
// Reads one JSON document from stdin and prints {"routes":[[id,...],...]} to stdout.

// Use CommonJS require pattern as requested by the prompt.
// (The benchmark runner compiles TypeScript; this also runs as plain JS if needed.)
// eslint-disable-next-line @typescript-eslint/no-var-requires
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input || '{}');

type PackageIn = { id: string | number; x: number; y: number };
type Node = { id: string; x: number; y: number };
type Route = { nodes: number[]; dist: number };

const depotArr: [number, number] = Array.isArray(data.depot) ? data.depot : [0, 0];
const depotX = +depotArr[0] || 0;
const depotY = +depotArr[1] || 0;
let truckCount: number = Math.max(1, (data.truckCount | 0) || 1);
const pkgs: PackageIn[] = Array.isArray(data.packages) ? data.packages : [];

const n = pkgs.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}
if (truckCount > n) truckCount = n;

// Normalize ids to strings for output stability.
const nodes: Node[] = new Array(n);
for (let i = 0; i < n; i++) {
  const p = pkgs[i];
  nodes[i] = { id: String(p.id), x: +p.x, y: +p.y };
}

function distXY(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

// Precompute depot distances and coordinate arrays for speed.
const xs = new Float64Array(n);
const ys = new Float64Array(n);
const depDist = new Float64Array(n);
for (let i = 0; i < n; i++) {
  xs[i] = nodes[i].x;
  ys[i] = nodes[i].y;
  depDist[i] = distXY(depotX, depotY, xs[i], ys[i]);
}

// Seed routes with farthest points spread by angle.
function buildInitialRoutes(): Route[] {
  // Compute centroid (including depot) to angle-sort.
  let cx = depotX;
  let cy = depotY;
  for (let i = 0; i < n; i++) {
    cx += xs[i];
    cy += ys[i];
  }
  cx /= (n + 1);
  cy /= (n + 1);

  const idxs = new Int32Array(n);
  for (let i = 0; i < n; i++) idxs[i] = i;

  // Angle sort (O(n log n)), then split into truckCount slices.
  const angles = new Float64Array(n);
  for (let i = 0; i < n; i++) angles[i] = Math.atan2(ys[i] - cy, xs[i] - cx);
  const tmp = Array.from(idxs);
  tmp.sort((a, b) => angles[a] - angles[b]);

  const routes: Route[] = new Array(truckCount);
  for (let r = 0; r < truckCount; r++) routes[r] = { nodes: [], dist: 0 };

  // Round-robin assignment by angle gives compact radial sectors.
  for (let i = 0; i < n; i++) {
    routes[i % truckCount].nodes.push(tmp[i]);
  }

  // Order each route by nearest-neighbor then 2-opt.
  for (let r = 0; r < truckCount; r++) {
    const ord = nearestNeighborOrder(routes[r].nodes);
    routes[r].nodes = ord;
    routes[r].dist = routeDistance(ord);
    twoOptInPlace(routes[r]);
  }
  return routes;
}

function routeDistance(order: number[]): number {
  const m = order.length;
  if (m === 0) return 0;
  let d = depDist[order[0]];
  for (let i = 1; i < m; i++) {
    const a = order[i - 1];
    const b = order[i];
    d += distXY(xs[a], ys[a], xs[b], ys[b]);
  }
  d += depDist[order[m - 1]];
  return d;
}

function nearestNeighborOrder(list: number[]): number[] {
  const m = list.length;
  if (m <= 2) return list.slice();
  const remaining = list.slice();

  // Start from farthest-from-depot point for better spread.
  let bestStart = 0;
  let bestD = -1;
  for (let i = 0; i < m; i++) {
    const id = remaining[i];
    const d = depDist[id];
    if (d > bestD) {
      bestD = d;
      bestStart = i;
    }
  }
  const order: number[] = [];
  order.push(remaining[bestStart]);
  remaining[bestStart] = remaining[remaining.length - 1];
  remaining.pop();

  while (remaining.length) {
    const last = order[order.length - 1];
    let bestIdx = 0;
    let best = Infinity;
    const lx = xs[last];
    const ly = ys[last];
    for (let i = 0; i < remaining.length; i++) {
      const cand = remaining[i];
      const d = distXY(lx, ly, xs[cand], ys[cand]);
      if (d < best) {
        best = d;
        bestIdx = i;
      }
    }
    order.push(remaining[bestIdx]);
    remaining[bestIdx] = remaining[remaining.length - 1];
    remaining.pop();
  }
  return order;
}

// 2-opt optimization per route; capped for scalability.
function twoOptInPlace(route: Route): void {
  const order = route.nodes;
  const m = order.length;
  if (m < 4) {
    route.dist = routeDistance(order);
    return;
  }

  // Limit iterations based on size (keeps runtime bounded).
  const maxPasses = m <= 200 ? 20 : m <= 1000 ? 8 : 3;
  let improved = true;
  let passes = 0;

  function getX(i: number): number {
    return i < 0 ? depotX : xs[order[i]];
  }
  function getY(i: number): number {
    return i < 0 ? depotY : ys[order[i]];
  }

  while (improved && passes < maxPasses) {
    improved = false;
    passes++;
    for (let i = 0; i < m - 2; i++) {
      const ax = i === 0 ? depotX : xs[order[i - 1]];
      const ay = i === 0 ? depotY : ys[order[i - 1]];
      const bx = xs[order[i]];
      const by = ys[order[i]];

      // Candidate j starts at i+1, but avoid adjacent edges.
      // Also include edge to depot at end by treating j==m-1 with next being depot.
      for (let j = i + 1; j < m - 1; j++) {
        const cx = xs[order[j]];
        const cy = ys[order[j]];
        const dx = xs[order[j + 1]];
        const dy = ys[order[j + 1]];

        const before = distXY(ax, ay, bx, by) + distXY(cx, cy, dx, dy);
        const after = distXY(ax, ay, cx, cy) + distXY(bx, by, dx, dy);
        if (after + 1e-9 < before) {
          // Reverse segment [i..j]
          for (let l = i, r = j; l < r; l++, r--) {
            const t = order[l];
            order[l] = order[r];
            order[r] = t;
          }
          improved = true;
        }
      }

      // Check swap involving end-depot edge occasionally (only when i is small).
      if (i < 64) {
        const j = m - 1;
        const cx = xs[order[j]];
        const cy = ys[order[j]];
        const dx = depotX;
        const dy = depotY;
        // pick k = i+1..m-2 as cut point; limit scans for big routes
        const step = m > 2000 ? 8 : 1;
        for (let k = i + 1; k < m - 1; k += step) {
          const ex = xs[order[k]];
          const ey = ys[order[k]];
          const fx = k + 1 < m ? xs[order[k + 1]] : depotX;
          const fy = k + 1 < m ? ys[order[k + 1]] : depotY;
          const before = distXY(ax, ay, bx, by) + distXY(ex, ey, fx, fy) + distXY(cx, cy, dx, dy);
          const after = distXY(ax, ay, ex, ey) + distXY(bx, by, fx, fy) + distXY(cx, cy, dx, dy);
          if (after + 1e-9 < before) {
            for (let l = i, r = k; l < r; l++, r--) {
              const t = order[l];
              order[l] = order[r];
              order[r] = t;
            }
            improved = true;
          }
        }
      }
    }
  }
  route.dist = routeDistance(order);
}

// Improve solution by relocating points between routes using a regret insertion heuristic.
function improveInterRoute(routes: Route): void {}

// Simple inter-route local search: move single node from one route to another if improves total.
function interRouteLocalSearch(routes: Route[]): void {
  if (routes.length <= 1) return;

  // Build membership arrays for quick selection.
  let totalNodes = 0;
  for (const r of routes) totalNodes += r.nodes.length;
  if (totalNodes !== n) return;

  const maxRounds = n <= 500 ? 4 : n <= 2000 ? 2 : 1;
  for (let round = 0; round < maxRounds; round++) {
    let moved = 0;

    // Process donor routes in descending length to free clustering.
    const orderRoutes = routes.map((r, idx) => ({ idx, len: r.nodes.length, dist: r.dist }));
    orderRoutes.sort((a, b) => b.len - a.len);

    for (const donorInfo of orderRoutes) {
      const di = donorInfo.idx;
      const donor = routes[di];
      if (donor.nodes.length <= 1) continue;

      // Limit candidate nodes from donor: sample evenly.
      const m = donor.nodes.length;
      const sample = m <= 60 ? m : 60;
      for (let s = 0; s < sample; s++) {
        const pos = (s * m / sample) | 0;
        const nodeId = donor.nodes[pos];

        // Compute donor cost change if removed.
        const donorDelta = deltaRemove(donor.nodes, pos);

        // Try insert into best receiver (sample routes).
        let bestGain = 0;
        let bestR = -1;
        let bestPos = -1;

        // Prefer shorter routes as receivers.
        const recvOrder = routes.map((r, idx) => ({ idx, len: r.nodes.length }));
        recvOrder.sort((a, b) => a.len - b.len);
        const recvSample = Math.min(routes.length, routes.length <= 12 ? routes.length : 12);

        for (let t = 0; t < recvSample; t++) {
          const ri = recvOrder[t].idx;
          if (ri === di) continue;
          const recv = routes[ri];
          const ins = bestInsert(recv.nodes, nodeId);
          const gain = -(donorDelta + ins.delta); // negative delta means improvement; gain > 0 is good
          if (gain > bestGain + 1e-9) {
            bestGain = gain;
            bestR = ri;
            bestPos = ins.pos;
          }
        }

        if (bestR !== -1) {
          // Apply move.
          donor.nodes.splice(pos, 1);
          routes[bestR].nodes.splice(bestPos, 0, nodeId);
          donor.dist += donorDelta;
          routes[bestR].dist += bestInsertCostAt(routes[bestR].nodes, bestPos, nodeId); // recompute local delta precisely

          // Repair distances with quick recompute for touched routes and 2-opt light pass.
          donor.dist = routeDistance(donor.nodes);
          routes[bestR].dist = routeDistance(routes[bestR].nodes);
          if (donor.nodes.length >= 4) twoOptInPlace(donor);
          if (routes[bestR].nodes.length >= 4) twoOptInPlace(routes[bestR]);
          moved++;
          // Refresh m after mutation; break to avoid stale indices.
          break;
        }
      }
    }
    if (moved === 0) break;
  }
}

function deltaRemove(order: number[], pos: number): number {
  const m = order.length;
  const removed = order[pos];
  const prev = pos === 0 ? -1 : order[pos - 1];
  const next = pos === m - 1 ? -1 : order[pos + 1];

  const prevX = prev === -1 ? depotX : xs[prev];
  const prevY = prev === -1 ? depotY : ys[prev];
  const remX = xs[removed];
  const remY = ys[removed];
  const nextX = next === -1 ? depotX : xs[next];
  const nextY = next === -1 ? depotY : ys[next];

  const before = distXY(prevX, prevY, remX, remY) + distXY(remX, remY, nextX, nextY);
  const after = distXY(prevX, prevY, nextX, nextY);
  return after - before;
}

function bestInsert(order: number[], nodeId: number): { pos: number; delta: number } {
  const m = order.length;
  const nx = xs[nodeId];
  const ny = ys[nodeId];
  if (m === 0) {
    const d = 2 * distXY(depotX, depotY, nx, ny);
    return { pos: 0, delta: d };
  }
  let bestPos = 0;
  let bestDelta = Infinity;

  // Consider inserting between prev and next (including depot at ends).
  // For large routes, sample positions to reduce O(m) worst-case.
  const step = m <= 400 ? 1 : m <= 2000 ? 2 : 4;
  for (let i = 0; i <= m; i += step) {
    const prev = i === 0 ? -1 : order[i - 1];
    const next = i === m ? -1 : order[i];
    const prevX = prev === -1 ? depotX : xs[prev];
    const prevY = prev === -1 ? depotY : ys[prev];
    const nextX = next === -1 ? depotX : xs[next];
    const nextY = next === -1 ? depotY : ys[next];

    const before = distXY(prevX, prevY, nextX, nextY);
    const after = distXY(prevX, prevY, nx, ny) + distXY(nx, ny, nextX, nextY);
    const delta = after - before;
    if (delta < bestDelta) {
      bestDelta = delta;
      bestPos = Math.min(i, m);
    }
  }
  return { pos: bestPos, delta: bestDelta };
}

function bestInsertCostAt(order: number[], pos: number, nodeId: number): number {
  // Compute delta if nodeId is inserted at pos in current order (where it already is).
  // Use neighbors excluding the inserted node itself.
  const m = order.length;
  const prev = pos === 0 ? -1 : order[pos - 1];
  const next = pos === m - 1 ? -1 : order[pos + 1];
  const prevX = prev === -1 ? depotX : xs[prev];
  const prevY = prev === -1 ? depotY : ys[prev];
  const nextX = next === -1 ? depotX : xs[next];
  const nextY = next === -1 ? depotY : ys[next];
  const nx = xs[nodeId];
  const ny = ys[nodeId];
  const before = distXY(prevX, prevY, nextX, nextY);
  const after = distXY(prevX, prevY, nx, ny) + distXY(nx, ny, nextX, nextY);
  return after - before;
}

// If too many trucks, allow some empty routes, but output only non-empty routes.
let routes = buildInitialRoutes();

// Inter-route improvement.
interRouteLocalSearch(routes);

// Final: ensure every package exactly once; output as routes of package ids.
// Also drop empty routes for compactness.
const used = new Int32Array(n);
for (const r of routes) for (const v of r.nodes) used[v]++;
// Repair if any duplicates/missing (should not happen, but be safe).
let missing: number[] = [];
for (let i = 0; i < n; i++) if (used[i] === 0) missing.push(i);
if (missing.length) {
  // Insert missing into best positions across routes.
  for (const mid of missing) {
    let bestR = 0;
    let bestPos = 0;
    let bestDelta = Infinity;
    for (let ri = 0; ri < routes.length; ri++) {
      const ins = bestInsert(routes[ri].nodes, mid);
      if (ins.delta < bestDelta) {
        bestDelta = ins.delta;
        bestR = ri;
        bestPos = ins.pos;
      }
    }
    routes[bestR].nodes.splice(bestPos, 0, mid);
  }
}

// Remove duplicates if any (keep first occurrence).
const seen = new Int32Array(n);
for (const r of routes) {
  const cleaned: number[] = [];
  for (const v of r.nodes) {
    if (seen[v] === 0) {
      seen[v] = 1;
      cleaned.push(v);
    }
  }
  r.nodes = cleaned;
  r.dist = routeDistance(cleaned);
}

// Any still-unseen? assign.
missing = [];
for (let i = 0; i < n; i++) if (seen[i] === 0) missing.push(i);
for (const mid of missing) {
  const ins = bestInsert(routes[0].nodes, mid);
  routes[0].nodes.splice(ins.pos, 0, mid);
}

// Re-optimize touched routes lightly.
for (const r of routes) {
  if (r.nodes.length >= 4) twoOptInPlace(r);
}

const outRoutes: string[][] = [];
for (const r of routes) {
  if (r.nodes.length === 0) continue;
  const arr = new Array(r.nodes.length);
  for (let i = 0; i < r.nodes.length; i++) arr[i] = nodes[r.nodes[i]].id;
  outRoutes.push(arr);
}

process.stdout.write(JSON.stringify({ routes: outRoutes }));
