; === Emit thick lines for all 3 edges (simplified approach) ===
; For each edge (a,b): emit quad as triangle strip: a+perp, a-perp, b+perp, b-perp
; Using clip-space XY for direction; perpendicular in XY plane; thickness = 0.015

; -------- Edge 0 -> 2 --------
; edgeDir = normalize((p2.xy - p0.xy, 0))
%p0v = OpLoad %v4float %p0
%p2v = OpLoad %v4float %p2
%p0xy = OpVectorShuffle %v3float %p0v %p0v 0 1 2
%p2xy = OpVectorShuffle %v3float %p2v %p2v 0 1 2
%d02 = OpFSub %v3float %p2xy %p0xy
; len = sqrt(d.x*d.x + d.y*d.y)
%dx = OpCompositeExtract %float %d02 0
%dy = OpCompositeExtract %float %d02 1
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%sum = OpFAdd %float %dx2 %dy2
%l = OpExtInst %float %1 Sqrt %sum
OpStore %len %l
%inv = OpFDiv %float %float_1 %l
%dir = OpVectorTimesScalar %v3float %d02 %inv
OpStore %edgeDir %dir
; perp = (-dir.y, dir.x, 0) * thickness
%diry = OpCompositeExtract %float %dir 1
%dirx = OpCompositeExtract %float %dir 0
%ndiry = OpFNegate %float %diry
%perp0 = OpCompositeConstruct %v3float %ndiry %dirx %float_0
%perpS = OpVectorTimesScalar %v3float %perp0 %float_0_015
OpStore %perp %perpS

; emit quad
%perpV = OpLoad %v3float %perp
%perpV4 = OpCompositeConstruct %v4float
             %perpV %float_0
; (constructing v4 from v3 + 0 is awkward; do manually)
%px = OpCompositeExtract %float %perpV 0
%py = OpCompositeExtract %float %perpV 1
%pz = OpCompositeExtract %float %perpV 2
%perp4 = OpCompositeConstruct %v4float %px %py %pz %float_0

%p0p = OpFAdd %v4float %p0v %perp4
%p0m = OpFSub %v4float %p0v %perp4
%p2p = OpFAdd %v4float %p2v %perp4
%p2m = OpFSub %v4float %p2v %perp4

OpStore %outpos0 %p0p
OpStore %outColor %v3_1
OpEmitVertex
OpStore %outpos0 %p0m
OpStore %outColor %v3_1

; -------- Edge 4 -> 0 --------
