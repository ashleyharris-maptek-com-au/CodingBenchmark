// Minimum Edge Cut in an undirected connected graph.
//
// Practical adaptive strategy:
// - Always compute a valid upper bound quickly (bridge test + small randomized trials).
// - For moderate graphs: run deterministic Stoer-Wagner (exact global min-cut).
// - For large graphs: run randomized Karger-Stein style contraction (high probability exact)
//   with time/iteration budget; always keep best found cut (valid) and print it.
//
// Note on "graphs that don't fit in memory":
// Exact global min-cut algorithms require repeated access to adjacency.
// In practice, any solver must at least iterate edges; here we stream input and store
// compact adjacency (int32 endpoints + next pointers) and can fall back to printing a
// valid cut using only bridge-finding when memory becomes tight.

struct EdgeIn {
  int u, v;
};

static inline uint64_t splitmix64(uint64_t &x) {
  uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
  z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
  z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
  return z ^ (z >> 31);
}

// ---------- Bridge finding (Tarjan) ----------
// If there is a bridge, min cut = 1 and that edge is the answer.
struct BridgeFinder {
  int n;
  std::vector<int> head, to, nxt, eu, ev;
  int ec = 0;
  std::vector<int> tin, low;
  int timer = 0;
  int bridge_eid = -1;

  explicit BridgeFinder(int n_) : n(n_) {
    head.assign(n, -1);
    tin.assign(n, -1);
    low.assign(n, -1);
  }

  void add_edge(int u, int v, int eid) {
    if ((int)to.size() < 2 * (eid + 1)) {
      // ensure capacity for this eid in eu/ev too
    }
    // store original endpoints for eid
    if ((int)eu.size() <= eid) {
      eu.resize(eid + 1);
      ev.resize(eid + 1);
    }
    eu[eid] = u;
    ev[eid] = v;
    // add directed arcs
    to.push_back(v);
    nxt.push_back(head[u]);
    head[u] = ec++;
    to.push_back(u);
    nxt.push_back(head[v]);
    head[v] = ec++;
  }

  void dfs(int v, int pe_arc) {
    tin[v] = low[v] = timer++;
    for (int a = head[v]; a != -1; a = nxt[a]) {
      if (a == (pe_arc ^ 1)) continue;
      int u = to[a];
      if (tin[u] != -1) {
        low[v] = std::min(low[v], tin[u]);
      } else {
        dfs(u, a);
        low[v] = std::min(low[v], low[u]);
        if (low[u] > tin[v]) {
          bridge_eid = a / 2;
        }
      }
    }
  }

  int find_any_bridge() {
    timer = 0;
    std::fill(tin.begin(), tin.end(), -1);
    std::fill(low.begin(), low.end(), -1);
    bridge_eid = -1;
    dfs(0, -1);
    return bridge_eid;
  }
};

// ---------- Stoer-Wagner (exact) ----------
// O(n^3) with adjacency weights; for unweighted multigraph we store weights in matrix-like sparse rows
// via vector<int> and dynamic merging using arrays.
struct StoerWagner {
  int n;
  std::vector<std::vector<int>> w; // weights (int) between supernodes
  std::vector<std::vector<int>> members; // original vertices in each supernode

  explicit StoerWagner(int n_) : n(n_), w(n_, std::vector<int>(n_, 0)), members(n_) {
    for (int i = 0; i < n; i++) members[i] = {i};
  }

  void add_edge(int u, int v) {
    if (u == v) return;
    w[u][v] += 1;
    w[v][u] += 1;
  }

  // returns {mincut_value, partition_vertices_on_one_side}
  std::pair<int, std::vector<int>> solve() {
    int N = n;
    std::vector<int> v(N);
    for (int i = 0; i < N; i++) v[i] = i;
    int best = std::numeric_limits<int>::max();
    std::vector<int> best_side;
    std::vector<int> added(N), weights(N), prev(N);

    while (N > 1) {
      std::fill(added.begin(), added.end(), 0);
      std::fill(weights.begin(), weights.end(), 0);
      int prev_v = -1;
      int last_v = -1;

      for (int it = 0; it < N; it++) {
        int sel = -1;
        for (int i = 0; i < N; i++) {
          int node = v[i];
          if (!added[node] && (sel == -1 || weights[node] > weights[sel])) sel = node;
        }
        added[sel] = 1;
        prev[sel] = prev_v;
        prev_v = sel;
        last_v = sel;
        for (int i = 0; i < N; i++) {
          int node = v[i];
          if (!added[node]) weights[node] += w[sel][node];
        }
      }

      int s = prev[last_v]; // second to last added
      int t = last_v;       // last added
      int cut_value = weights[t];
      if (cut_value < best) {
        best = cut_value;
        // side is members of t (the last vertex added) in this phase
        best_side = members[t];
      }

      // merge t into s
      if (s == -1) break;
      for (int i = 0; i < (int)w.size(); i++) {
        if (i == s) continue;
        w[s][i] += w[t][i];
        w[i][s] += w[i][t];
      }
      members[s].insert(members[s].end(), members[t].begin(), members[t].end());
      // remove t from v list (swap with end)
      int pos_t = -1;
      for (int i = 0; i < N; i++) if (v[i] == t) { pos_t = i; break; }
      if (pos_t != -1) v[pos_t] = v[N - 1];
      N--;
    }

    return {best, best_side};
  }
};

// ---------- Randomized contraction (Karger-style) ----------
// For big graphs we run repeated contractions to get small cuts quickly.
// We track a union-find of contracted vertices and sample random edges.
struct DSU {
  std::vector<int> p, r, sz;
  explicit DSU(int n = 0) { init(n); }
  void init(int n) {
    p.resize(n);
    r.assign(n, 0);
    sz.assign(n, 1);
    for (int i = 0; i < n; i++) p[i] = i;
  }
  int find(int a) { return p[a] == a ? a : (p[a] = find(p[a])); }
  bool unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return false;
    if (r[a] < r[b]) std::swap(a, b);
    p[b] = a;
    sz[a] += sz[b];
    if (r[a] == r[b]) r[a]++;
    return true;
  }
  int size(int a) { return sz[find(a)]; }
};

static std::vector<EdgeIn> extract_cut_edges(const std::vector<EdgeIn> &edges, const std::vector<char> &side) {
  std::vector<EdgeIn> cut;
  cut.reserve(128);
  for (const auto &e : edges) {
    if (side[e.u] != side[e.v]) cut.push_back(e);
  }
  return cut;
}

static std::vector<EdgeIn> karger_one(const std::vector<EdgeIn> &edges, int n, uint64_t seed) {
  DSU dsu(n);
  int comps = n;
  uint64_t x = seed;
  // contract until 2 supernodes remain
  while (comps > 2) {
    uint64_t r = splitmix64(x);
    const EdgeIn &e = edges[(size_t)(r % edges.size())];
    int a = dsu.find(e.u), b = dsu.find(e.v);
    if (a == b) continue;
    dsu.unite(a, b);
    comps--;
  }
  // build side array from one representative
  int repA = -1, repB = -1;
  for (int i = 0; i < n; i++) {
    int f = dsu.find(i);
    if (repA == -1) repA = f;
    else if (f != repA) { repB = f; break; }
  }
  std::vector<char> side(n, 0);
  for (int i = 0; i < n; i++) side[i] = (dsu.find(i) == repA) ? 0 : 1;
  return extract_cut_edges(edges, side);
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int N;
  long long Mll;
  if (!(std::cin >> N >> Mll)) return 0;
  long long M = Mll;
  if (N <= 1) {
    std::cout << 0 << "\n";
    return 0;
  }

  // Read edges; store in a compact vector (fits most practical cases).
  // If M is huge, this may fail due to memory; in that case we still try to proceed with what we have.
  std::vector<EdgeIn> edges;
  edges.reserve((size_t)std::min<long long>(M, 50'000'000LL));
  for (long long i = 0; i < M; i++) {
    int u, v;
    std::cin >> u >> v;
    if (!std::cin) break;
    if (u < 0 || v < 0 || u >= N || v >= N) continue;
    edges.push_back({u, v});
  }
  if (edges.empty()) {
    // Connected graph with N>1 but no edges can't happen per statement, but be safe.
    std::cout << 0 << "\n";
    return 0;
  }

  // 1) Fast bridge test
  int bestK = std::numeric_limits<int>::max();
  std::vector<EdgeIn> bestCut;
  {
    BridgeFinder bf(N);
    bf.to.reserve(edges.size() * 2);
    bf.nxt.reserve(edges.size() * 2);
    for (int i = 0; i < (int)edges.size(); i++) bf.add_edge(edges[i].u, edges[i].v, i);
    int be = bf.find_any_bridge();
    if (be != -1) {
      std::cout << 1 << "\n" << edges[be].u << " " << edges[be].v << "\n";
      return 0;
    }
    // no bridge => min cut >= 2
  }

  // 2) Decide algorithm by scale
  // Stoer-Wagner is exact but O(N^3) memory O(N^2) due to weights matrix.
  // Use it only when N is small enough.
  const int STOER_MAX_N = 1500; // 1500^2 ints ~ 9MB, OK.
  bool can_stoer = (N <= STOER_MAX_N);

  if (can_stoer) {
    StoerWagner sw(N);
    for (const auto &e : edges) sw.add_edge(e.u, e.v);
    auto res = sw.solve();
    int mincut = res.first;
    std::vector<char> side(N, 0);
    for (int v : res.second) side[v] = 1;
    bestCut = extract_cut_edges(edges, side);
    bestK = (int)bestCut.size();
    // For connected graphs, mincut >=1; if sw gave 0 (shouldn't), fall back.
    if (bestK == 0) bestK = std::max(1, mincut);
  }

  // 3) For larger graphs (or as extra safety), run randomized contractions with a time budget.
  // Keep best valid cut found.
  {
    using clock = std::chrono::steady_clock;
    auto start = clock::now();
    // Budget: up to ~1.5s or a number of iterations, whichever first.
    const double TIME_BUDGET_SEC = can_stoer ? 0.5 : 1.5;
    // Iterations heuristic: more for smaller N; capped for huge graphs.
    int iters = 0;
    int maxIters = 200;
    if (N <= 2000) maxIters = 400;
    if (N <= 500) maxIters = 800;
    if (edges.size() > 50'000'000ULL) maxIters = 50;
    if (bestK == std::numeric_limits<int>::max()) bestK = std::numeric_limits<int>::max();

    uint64_t seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
    while (iters < maxIters) {
      auto now = clock::now();
      double elapsed = std::chrono::duration<double>(now - start).count();
      if (elapsed > TIME_BUDGET_SEC) break;
      auto cut = karger_one(edges, N, seed + (uint64_t)iters * 0x9e3779b97f4a7c15ULL);
      int k = (int)cut.size();
      if (k > 0 && k < bestK) {
        bestK = k;
        bestCut.swap(cut);
        if (bestK == 2) {
          // Can't be 1 (no bridges). 2 is common minimum; early stop.
          // Still allow a couple more attempts when not using Stoer.
          if (!can_stoer) {
            // continue briefly for chance of 2 anyway already
          } else {
            // exact already
          }
        }
      }
      iters++;
    }
  }

  if (bestCut.empty()) {
    // As a last resort: output any single edge (valid cut only if bridge; but we already checked no bridge).
    // Instead, output 2 edges incident to node 0 (disconnects? not guaranteed).
    // We'll construct a guaranteed cut by separating a vertex using all its incident edges.
    // This always disconnects: remove all edges adjacent to vertex v => v becomes isolated.
    int v = 0;
    std::vector<EdgeIn> cut;
    for (const auto &e : edges) if (e.u == v || e.v == v) cut.push_back(e);
    if (!cut.empty()) {
      bestCut = std::move(cut);
      bestK = (int)bestCut.size();
    } else {
      // pick any vertex that has incident edges
      for (int vv = 1; vv < N && bestCut.empty(); vv++) {
        std::vector<EdgeIn> c2;
        for (const auto &e : edges) if (e.u == vv || e.v == vv) c2.push_back(e);
        if (!c2.empty()) { bestCut = std::move(c2); bestK = (int)bestCut.size(); }
      }
      if (bestCut.empty()) {
        std::cout << 0 << "\n";
        return 0;
      }
    }
  }

  std::cout << bestK << "\n";
  for (const auto &e : bestCut) {
    std::cout << e.u << " " << e.v << "\n";
  }
  return 0;
}
