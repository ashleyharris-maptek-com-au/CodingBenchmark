OpCapability Shader
OpCapability Matrix
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %in_uv "uv"
OpName %in_normal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpDecorate %in_uv Location 2
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fn = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4

%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_3_14159 = OpConstant %float 3.14159
%float_16 = OpConstant %float 16.0

%v3_red = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
%v3_blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9

%in_uv = OpVariable %ptr_Input_v2float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%int_3 = OpConstant %int 3

%main = OpFunction %void None %fn
%entry = OpLabel

; stripe = sin(uv.y * pi * 16.0)
%uv_val = OpLoad %v2float %in_uv
%uv_y = OpCompositeExtract %float %uv_val 1
%t1 = OpFMul %float %uv_y %float_3_14159
%t2 = OpFMul %float %t1 %float_16
%stripe = OpExtInst %float %1 Sin %t2

; choose red when stripe > 0 else blue
%is_pos = OpFOrdGreaterThan %bool %stripe %float_0
%base = OpSelect %v3float %is_pos %v3_red %v3_blue

; lambert: diffuse = max(dot(normalize(normal), normalize(lightPos.xyz)), 0)
%n = OpLoad %v3float %in_normal
%n_norm = OpExtInst %v3float %1 Normalize %n

%lightpos = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%lp4 = OpLoad %v4float %lightpos
%lp3 = OpVectorShuffle %v3float %lp4 %lp4 0 1 2
%l_norm = OpExtInst %v3float %1 Normalize %lp3

%dotnl = OpDot %float %n_norm %l_norm
%diffuse = OpExtInst %float %1 FMax %dotnl %float_0

; shade = 0.4 + 0.6*diffuse
%s1 = OpFMul %float %diffuse %float_0_6
%shade = OpFAdd %float %float_0_4 %s1

%rgb = OpVectorTimesScalar %v3float %base %shade
%outv = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %outv

OpReturn
OpFunctionEnd
