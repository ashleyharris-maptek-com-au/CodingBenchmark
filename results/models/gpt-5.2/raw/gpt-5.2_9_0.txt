using System;
using System.Buffers;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;

// 1D Cutting Stock (bin packing) heuristic solver optimized for very large N.
// Uses First-Fit Decreasing with a balanced-tree of remaining capacities.
// Outputs a feasible assignment (not guaranteed optimal) quickly.
public static class Program
{
    public static void Main()
    {
        var fs = Console.OpenStandardInput();
        var reader = new FastReader(fs);

        int n = reader.NextInt();
        int stockLen = reader.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Read cuts
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            int v = reader.NextInt();
            // Problem states no cut can exceed stock_length; still guard.
            if (v > stockLen) v = stockLen;
            cuts[i] = v;
        }

        // Sort indices by decreasing cut length
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) => cuts[b].CompareTo(cuts[a]));

        // Each stock: remaining capacity.
        var rem = new List<int>(Math.Min(n, 1_000_000));
        // Per stock: list of assigned original indices.
        var stocks = new List<List<int>>(Math.Min(n, 1_000_000));

        // Map remaining capacity -> list of stock ids with that remaining.
        // Sorted by remaining ascending so we can find minimal remaining >= cut (best-fit).
        var tree = new SortedDictionary<int, List<int>>();

        for (int order = 0; order < n; order++)
        {
            int i = idx[order];
            int cut = cuts[i];

            // Find smallest remaining capacity >= cut.
            int chosenStock = -1;
            int chosenRemKey = 0;
            if (tree.Count != 0)
            {
                foreach (var kv in tree)
                {
                    if (kv.Key >= cut)
                    {
                        var list = kv.Value;
                        int last = list.Count - 1;
                        chosenStock = list[last];
                        list.RemoveAt(last);
                        chosenRemKey = kv.Key;
                        if (list.Count == 0) tree.Remove(kv.Key);
                        break;
                    }
                }
            }

            if (chosenStock < 0)
            {
                // New stock
                chosenStock = rem.Count;
                int r = stockLen - cut;
                rem.Add(r);
                var lst = new List<int>(8) { i };
                stocks.Add(lst);
                AddToTree(tree, r, chosenStock);
            }
            else
            {
                // Assign to existing stock and update remaining.
                stocks[chosenStock].Add(i);
                int newRem = chosenRemKey - cut;
                rem[chosenStock] = newRem;
                AddToTree(tree, newRem, chosenStock);
            }
        }

        // Output
        var sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 1 << 20);
        sw.WriteLine(stocks.Count);
        for (int s = 0; s < stocks.Count; s++)
        {
            var lst = stocks[s];
            for (int k = 0; k < lst.Count; k++)
            {
                if (k != 0) sw.Write(' ');
                sw.Write(lst[k]);
            }
            sw.WriteLine();
        }
        sw.Flush();
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void AddToTree(SortedDictionary<int, List<int>> tree, int rem, int stockId)
    {
        if (!tree.TryGetValue(rem, out var list))
        {
            list = new List<int>(4);
            tree.Add(rem, list);
        }
        list.Add(stockId);
    }

    // Fast integer reader for large inputs.
    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;

        public FastReader(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = ArrayPool<byte>.Shared.Rent(bufferSize);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int val = 0;
            while (c > 32)
            {
                val = (val * 10) + (c - '0');
                c = ReadByte();
                if (c == -1) break;
            }
            return val * sign;
        }
    }
}
