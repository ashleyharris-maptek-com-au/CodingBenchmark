// Node.js JavaScript program for planning grid paths for search drones.
// Reads one JSON document from stdin and prints one JSON document to stdout.

const fs = require('fs');

function inBounds(x, y, w, h) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function clamp(n, lo, hi) {
  return n < lo ? lo : n > hi ? hi : n;
}

function buildNeighborTable(w, h) {
  const n = w * h;
  const neigh = new Array(n);
  for (let y = 0; y < h; y++) {
    for (let x = 0; x < w; x++) {
      const idx = y * w + x;
      const arr = [];
      // stay
      arr.push(idx);
      if (x > 0) arr.push(idx - 1);
      if (x + 1 < w) arr.push(idx + 1);
      if (y > 0) arr.push(idx - w);
      if (y + 1 < h) arr.push(idx + w);
      neigh[idx] = arr;
    }
  }
  return neigh;
}

// Lightweight xorshift RNG for deterministic tie-breaking.
class XorShift32 {
  constructor(seed) {
    this.x = seed >>> 0 || 2463534242;
  }
  nextU32() {
    let x = this.x;
    x ^= (x << 13) >>> 0;
    x ^= (x >>> 17) >>> 0;
    x ^= (x << 5) >>> 0;
    this.x = x >>> 0;
    return this.x;
  }
  nextFloat() {
    return (this.nextU32() >>> 0) / 4294967296;
  }
}

function planPaths(prob, drones, maxStepsPerDrone) {
  const h = prob.length | 0;
  const w = h ? (prob[0].length | 0) : 0;
  const n = w * h;
  const neigh = buildNeighborTable(w, h);

  const isObstacle = new Uint8Array(n);
  const value = new Float64Array(n);
  for (let y = 0; y < h; y++) {
    const row = prob[y];
    for (let x = 0; x < w; x++) {
      const idx = y * w + x;
      const v = row[x];
      if (v === -1) {
        isObstacle[idx] = 1;
        value[idx] = 0;
      } else {
        value[idx] = typeof v === 'number' && v > 0 ? v : 0;
      }
    }
  }

  const covered = new Uint8Array(n);

  const D = drones.length | 0;
  const pos = new Int32Array(D);
  const stepsLeft = new Int32Array(D);
  const paths = new Array(D);

  for (let i = 0; i < D; i++) {
    const d = drones[i] || {};
    const x = clamp(d.x | 0, 0, w - 1);
    const y = clamp(d.y | 0, 0, h - 1);
    let p = y * w + x;
    // If start is an obstacle, nudge to a nearby free cell with a bounded BFS.
    if (isObstacle[p]) {
      const q = [p];
      const seen = new Uint8Array(n);
      seen[p] = 1;
      let found = -1;
      for (let qi = 0; qi < q.length && qi < 1024; qi++) {
        const cur = q[qi];
        if (!isObstacle[cur]) {
          found = cur;
          break;
        }
        const ns = neigh[cur];
        for (let k = 0; k < ns.length; k++) {
          const nx = ns[k];
          if (!seen[nx]) {
            seen[nx] = 1;
            q.push(nx);
          }
        }
      }
      if (found !== -1) p = found;
    }
    pos[i] = p;
    stepsLeft[i] = maxStepsPerDrone | 0;
    paths[i] = [[p % w, (p / w) | 0]];
  }

  for (let i = 0; i < D; i++) {
    const p = pos[i];
    if (!isObstacle[p] && value[p] > 0) covered[p] = 1;
  }

  // Seed RNG from scenario shape and starts (deterministic).
  let seed = 2166136261 >>> 0;
  seed ^= (w + 0x9e3779b9) >>> 0;
  seed = Math.imul(seed, 16777619) >>> 0;
  seed ^= (h + 0x9e3779b9) >>> 0;
  seed = Math.imul(seed, 16777619) >>> 0;
  seed ^= (D + 0x9e3779b9) >>> 0;
  seed = Math.imul(seed, 16777619) >>> 0;
  for (let i = 0; i < D; i++) {
    seed ^= (pos[i] + 0x9e3779b9) >>> 0;
    seed = Math.imul(seed, 16777619) >>> 0;
  }
  const rng = new XorShift32(seed);

  const depth = maxStepsPerDrone <= 2 ? 1 : maxStepsPerDrone <= 20 ? 2 : 3;

  const lastPosA = new Int32Array(D);
  const lastPosB = new Int32Array(D);
  lastPosA.fill(-1);
  lastPosB.fill(-1);

  function scoreMove(droneIndex, nextCell) {
    if (isObstacle[nextCell]) return -1e30;
    let score = 0;
    if (!covered[nextCell]) score += value[nextCell];
    if (nextCell === pos[droneIndex]) score -= 0.02;
    if (nextCell === lastPosA[droneIndex]) score -= 0.05;
    if (nextCell === lastPosB[droneIndex]) score -= 0.02;

    if (depth > 1) {
      let cur = nextCell;
      let discount = 0.6;
      const localVisited = [pos[droneIndex], nextCell];
      for (let d = 1; d < depth; d++) {
        const ns = neigh[cur];
        let best = -1e30;
        let bestCell = cur;
        for (let k = 0; k < ns.length; k++) {
          const c = ns[k];
          if (isObstacle[c]) continue;
          let s = 0;
          if (!covered[c]) s += value[c];
          for (let t = 0; t < localVisited.length; t++) {
            if (localVisited[t] === c) {
              s -= 0.03;
              break;
            }
          }
          if (s > best) {
            best = s;
            bestCell = c;
          }
        }
        if (best <= 0) break;
        score += best * discount;
        discount *= 0.6;
        cur = bestCell;
        if (localVisited.length < 6) localVisited.push(cur);
      }
    }
    return score;
  }

  let totalMoves = (maxStepsPerDrone | 0) * D;
  while (totalMoves > 0) {
    let bestDrone = -1;
    let bestNext = -1;
    let bestScore = -1e30;

    for (let i = 0; i < D; i++) {
      if (stepsLeft[i] <= 0) continue;
      const p = pos[i];
      const ns = neigh[p];
      let localBestScore = -1e30;
      let localBestCell = p;
      for (let k = 0; k < ns.length; k++) {
        const c = ns[k];
        const s = scoreMove(i, c) + (rng.nextFloat() - 0.5) * 1e-6;
        if (s > localBestScore) {
          localBestScore = s;
          localBestCell = c;
        }
      }
      if (localBestScore > bestScore) {
        bestScore = localBestScore;
        bestDrone = i;
        bestNext = localBestCell;
      }
    }

    if (bestDrone === -1) break;

    const i = bestDrone;
    const prev = pos[i];
    const next = bestNext;
    lastPosB[i] = lastPosA[i];
    lastPosA[i] = prev;
    pos[i] = next;
    stepsLeft[i]--;
    totalMoves--;

    paths[i].push([next % w, (next / w) | 0]);
    if (!isObstacle[next] && value[next] > 0) covered[next] = 1;
  }

  const targetLen = (maxStepsPerDrone | 0) + 1;
  for (let i = 0; i < D; i++) {
    const p = paths[i];
    if (p.length < targetLen) {
      const last = p[p.length - 1];
      while (p.length < targetLen) p.push([last[0], last[1]]);
    } else if (p.length > targetLen) {
      p.length = targetLen;
    }
  }

  return { paths };
}

function main() {
  const input = fs.readFileSync(0, 'utf8').trim();
  const data = input.length ? JSON.parse(input) : {};
  const prob = Array.isArray(data.probability) ? data.probability : [];
  const drones = Array.isArray(data.drones) ? data.drones : [];
  const maxStepsPerDrone = data.maxStepsPerDrone | 0;

  const out = planPaths(prob, drones, Math.max(0, maxStepsPerDrone));
  process.stdout.write(JSON.stringify(out));
}

main();
