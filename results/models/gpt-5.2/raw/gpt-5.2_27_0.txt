// C++20
#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <utility>
#include <vector>

// Practical k-coloring heuristic:
// - Builds adjacency lists
// - Greedy coloring (largest-degree first) for a fast initial solution
// - If conflicts remain or k is tight, uses a time-bounded min-conflicts local search
//   (random restarts).
//
// Designed to scale to very large graphs: O(n+m) memory and mostly linear passes.
// (No non-standard headers; MSVC friendly.)

struct FastRng {
  // splitmix64
  uint64_t x;
  explicit FastRng(uint64_t seed) : x(seed) {}
  uint64_t next_u64() {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
  }
  uint32_t next_u32() { return static_cast<uint32_t>(next_u64()); }
  uint32_t uniform_u32(uint32_t bound) {
    // Lemire method
    uint64_t r = next_u64();
    return static_cast<uint32_t>((r * bound) >> 32);
  }
};

static inline uint64_t now_ticks() {
  return static_cast<uint64_t>(
      std::chrono::high_resolution_clock::now().time_since_epoch().count());
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n = 0;
  int64_t m = 0;
  int k = 0;
  if (!(std::cin >> n >> m >> k)) return 0;
  if (n <= 0 || k <= 0) return 0;

  std::vector<int> deg(n, 0);
  std::vector<std::pair<int, int>> edges;
  edges.reserve(static_cast<size_t>(m));

  for (int64_t i = 0; i < m; ++i) {
    int u, v;
    std::cin >> u >> v;
    if (u < 0 || v < 0 || u >= n || v >= n || u == v) continue;
    edges.emplace_back(u, v);
    ++deg[u];
    ++deg[v];
  }

  // Build CSR-like adjacency for memory efficiency.
  std::vector<int> off(n + 1, 0);
  for (int i = 0; i < n; ++i) off[i + 1] = off[i] + deg[i];
  std::vector<int> adj(off.back());
  std::vector<int> cur = off;
  for (auto [u, v] : edges) {
    adj[cur[u]++] = v;
    adj[cur[v]++] = u;
  }
  edges.clear();
  edges.shrink_to_fit();

  // Order vertices by descending degree (for initial greedy).
  std::vector<int> order(n);
  for (int i = 0; i < n; ++i) order[i] = i;
  std::sort(order.begin(), order.end(), [&](int a, int b) {
    if (deg[a] != deg[b]) return deg[a] > deg[b];
    return a < b;
  });

  std::vector<int> color(n, -1);
  std::vector<int> seen(k, -1);
  int stamp = 0;

  auto greedy_color = [&]() {
    std::fill(color.begin(), color.end(), -1);
    stamp = 0;
    for (int idx = 0; idx < n; ++idx) {
      int v = order[idx];
      ++stamp;
      for (int ei = off[v]; ei < off[v + 1]; ++ei) {
        int u = adj[ei];
        int c = color[u];
        if (c >= 0) seen[c] = stamp;
      }
      int chosen = -1;
      for (int c = 0; c < k; ++c) {
        if (seen[c] != stamp) {
          chosen = c;
          break;
        }
      }
      if (chosen < 0) chosen = 0; // fallback; will be fixed by local search if possible
      color[v] = chosen;
    }
  };

  greedy_color();

  // Conflict accounting for min-conflicts search.
  std::vector<int> conflicts(n, 0);
  int64_t total_conflicts = 0;
  auto recompute_conflicts = [&]() {
    std::fill(conflicts.begin(), conflicts.end(), 0);
    total_conflicts = 0;
    for (int v = 0; v < n; ++v) {
      int cv = color[v];
      int cnt = 0;
      for (int ei = off[v]; ei < off[v + 1]; ++ei) {
        int u = adj[ei];
        if (color[u] == cv) ++cnt;
      }
      conflicts[v] = cnt;
      total_conflicts += cnt;
    }
    total_conflicts /= 2; // each edge counted twice
  };

  recompute_conflicts();

  // If greedy already valid, output.
  if (total_conflicts == 0) {
    for (int i = 0; i < n; ++i) {
      if (i) std::cout << ' ';
      int c = color[i];
      if (c < 0) c = 0;
      if (c >= k) c = c % k;
      std::cout << c;
    }
    std::cout << "\n";
    return 0;
  }

  // Adaptive local search parameters.
  // Keep memory small: for each move evaluate neighbor colors via small hash using stamps.
  // Limit time by iteration cap scaled to graph size.
  uint64_t seed = now_ticks() ^ (uint64_t)n * 0x9e3779b97f4a7c15ULL ^ (uint64_t)k;
  FastRng rng(seed);

  // Helper: compute conflicts if v had color newc; also counts how many neighbors have each color.
  std::vector<int> neigh_count(k, 0);
  std::vector<int> used_stamp(k, 0);
  int used_clock = 1;

  auto best_color_for_vertex = [&](int v) -> int {
    ++used_clock;
    int cv = color[v];
    int bestc = cv;
    int bestscore = std::numeric_limits<int>::max();

    // Count neighbor colors.
    for (int ei = off[v]; ei < off[v + 1]; ++ei) {
      int u = adj[ei];
      int cu = color[u];
      if (cu < 0) continue;
      if (used_stamp[cu] != used_clock) {
        used_stamp[cu] = used_clock;
        neigh_count[cu] = 1;
      } else {
        neigh_count[cu]++;
      }
    }

    // Try all colors if k small; otherwise sample a subset including current and a few random.
    if (k <= 64) {
      for (int c = 0; c < k; ++c) {
        int score = (used_stamp[c] == used_clock) ? neigh_count[c] : 0;
        if (score < bestscore || (score == bestscore && (uint32_t)c == rng.uniform_u32((uint32_t)k))) {
          bestscore = score;
          bestc = c;
        }
      }
      return bestc;
    }

    // For large k, evaluate a mixed candidate set.
    auto eval = [&](int c) {
      int score = (used_stamp[c] == used_clock) ? neigh_count[c] : 0;
      if (score < bestscore) {
        bestscore = score;
        bestc = c;
      } else if (score == bestscore) {
        // tie-break randomly
        if ((rng.next_u32() & 7u) == 0u) bestc = c;
      }
    };

    eval(cv);
    for (int t = 0; t < 32; ++t) eval((int)rng.uniform_u32((uint32_t)k));
    // Also check colors seen among neighbors (often relevant)
    for (int c = 0; c < k; ++c) {
      if (used_stamp[c] == used_clock) eval(c);
    }
    return bestc;
  };

  // Maintain a list of currently conflicting vertices for fast sampling.
  std::vector<int> bad;
  bad.reserve(std::min<int64_t>(n, 1'000'000));
  std::vector<int> pos_in_bad(n, -1);

  auto rebuild_bad = [&]() {
    bad.clear();
    std::fill(pos_in_bad.begin(), pos_in_bad.end(), -1);
    for (int v = 0; v < n; ++v) {
      if (conflicts[v] > 0) {
        pos_in_bad[v] = (int)bad.size();
        bad.push_back(v);
      }
    }
  };
  rebuild_bad();

  auto bad_add = [&](int v) {
    if (pos_in_bad[v] != -1) return;
    pos_in_bad[v] = (int)bad.size();
    bad.push_back(v);
  };
  auto bad_remove = [&](int v) {
    int p = pos_in_bad[v];
    if (p == -1) return;
    int last = bad.back();
    bad[p] = last;
    pos_in_bad[last] = p;
    bad.pop_back();
    pos_in_bad[v] = -1;
  };

  // Incremental update when recoloring v from oldc to newc.
  auto apply_move = [&](int v, int newc) {
    int oldc = color[v];
    if (oldc == newc) return;

    // Update total_conflicts by considering edges (v,u) that were/are conflicts.
    for (int ei = off[v]; ei < off[v + 1]; ++ei) {
      int u = adj[ei];
      int cu = color[u];
      if (cu < 0) continue;
      bool was = (cu == oldc);
      bool now = (cu == newc);
      if (was && !now) {
        // resolve conflict edge
        total_conflicts--;
        conflicts[u]--;
      } else if (!was && now) {
        // create conflict edge
        total_conflicts++;
        conflicts[u]++;
      }
    }

    color[v] = newc;

    // Recompute conflicts[v] exactly (degree(v) work).
    int cnt = 0;
    for (int ei = off[v]; ei < off[v + 1]; ++ei) {
      int u = adj[ei];
      if (color[u] == newc) ++cnt;
    }
    conflicts[v] = cnt;

    // Maintain bad set for v and its neighbors (neighbors updated above).
    if (conflicts[v] > 0) bad_add(v);
    else bad_remove(v);
    for (int ei = off[v]; ei < off[v + 1]; ++ei) {
      int u = adj[ei];
      if (conflicts[u] > 0) bad_add(u);
      else bad_remove(u);
    }
  };

  // Iteration budget: scaled; uses restarts to escape local minima.
  // For huge graphs, keep per-vertex work low.
  const int64_t base_iters = 200000;
  int64_t iters = base_iters;
  if (n > 200000) iters = base_iters + (int64_t)(n / 2);
  if (n > 1000000) iters = base_iters + (int64_t)(n / 3);
  if (m > 5000000) iters += (int64_t)(m / 10);
  // Cap to avoid runaway.
  const int64_t max_iters = 50'000'000;
  if (iters > max_iters) iters = max_iters;

  int restarts = 0;
  int64_t no_improve = 0;
  int64_t best_conf = total_conflicts;
  std::vector<int> best_color = color;

  auto random_init = [&]() {
    for (int i = 0; i < n; ++i) color[i] = (int)rng.uniform_u32((uint32_t)k);
    recompute_conflicts();
    rebuild_bad();
  };

  // If greedy is very conflicted relative to m, try random init too.
  if (total_conflicts > std::max<int64_t>(1, m / 50)) {
    random_init();
    best_conf = total_conflicts;
    best_color = color;
  }

  for (int64_t it = 0; it < iters; ++it) {
    if (total_conflicts == 0) break;

    if (total_conflicts < best_conf) {
      best_conf = total_conflicts;
      best_color = color;
      no_improve = 0;
    } else {
      ++no_improve;
    }

    // Restart if stuck.
    if (no_improve > (int64_t)std::max<int>(10000, n / 10)) {
      ++restarts;
      no_improve = 0;
      if (restarts % 2 == 1) greedy_color();
      else random_init();
      recompute_conflicts();
      rebuild_bad();
      continue;
    }

    if (bad.empty()) {
      // Should imply zero conflicts, but keep safe.
      recompute_conflicts();
      rebuild_bad();
      continue;
    }

    // Pick a conflicting vertex.
    int v = bad[(int)rng.uniform_u32((uint32_t)bad.size())];
    int newc = best_color_for_vertex(v);
    apply_move(v, newc);
  }

  if (total_conflicts != 0) {
    // Fall back to best found (often helps if last steps got worse).
    color.swap(best_color);
    recompute_conflicts();
  }

  // Output a complete assignment (even if imperfect); ensures well-formed output.
  for (int i = 0; i < n; ++i) {
    if (i) std::cout << ' ';
    int c = color[i];
    if (c < 0) c = 0;
    if (c >= k) c = c % k;
    std::cout << c;
  }
  std::cout << "\n";
  return 0;
}
