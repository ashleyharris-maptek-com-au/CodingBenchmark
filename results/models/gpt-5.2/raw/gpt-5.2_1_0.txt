#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <utility>
#include <vector>

// Adaptive TSP tour constructor for Euclidean points.
// Goals: always output a valid permutation starting with 0; run fast on huge N.
// Strategy (by N):
// - N <= 12: exact (Held-Karp DP) with path reconstruction (starting at 0).
// - N <= 200000: build space-filling-ish order via 2D Morton code (Z-order) after scaling.
// - Otherwise: even lighter: angle sort around centroid (O(N log N), low constant).
// Both approximations produce a reasonable route quickly and with low memory overhead.

struct Pt {
    float x;
    float y;
};

static inline uint32_t splitBy1_16(uint32_t a) {
    // Interleave lower 16 bits with zeros: abcdefghijklmnop -> a0b0c0...p0
    a &= 0x0000ffffu;
    a = (a | (a << 8)) & 0x00ff00ffu;
    a = (a | (a << 4)) & 0x0f0f0f0fu;
    a = (a | (a << 2)) & 0x33333333u;
    a = (a | (a << 1)) & 0x55555555u;
    return a;
}

static inline uint32_t morton2D_16(uint32_t x, uint32_t y) {
    return (splitBy1_16(x) | (splitBy1_16(y) << 1));
}

static inline double dist2(const Pt& a, const Pt& b) {
    double dx = (double)a.x - (double)b.x;
    double dy = (double)a.y - (double)b.y;
    return dx * dx + dy * dy;
}

// Exact TSP for N<=12 using Held-Karp DP, returning a Hamiltonian path starting at 0.
// Output order is the visit order; return to 0 is implicit.
static std::vector<int> solve_exact_small(const std::vector<Pt>& pts) {
    int n = (int)pts.size();
    if (n == 1) return {0};
    const int FULL = 1 << n;
    const double INF = std::numeric_limits<double>::infinity();

    // dp[mask][j] = best cost to start at 0, visit mask, end at j.
    // Only consider masks that include 0.
    std::vector<double> dp((size_t)FULL * (size_t)n, INF);
    std::vector<int16_t> parent((size_t)FULL * (size_t)n, -1);
    auto idx = [n](int mask, int j) -> size_t { return (size_t)mask * (size_t)n + (size_t)j; };

    dp[idx(1 << 0, 0)] = 0.0;

    // Precompute distances
    double d[12][12];
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            d[i][j] = std::sqrt(dist2(pts[i], pts[j]));

    for (int mask = 0; mask < FULL; ++mask) {
        if ((mask & 1) == 0) continue;
        for (int j = 0; j < n; ++j) {
            double cur = dp[idx(mask, j)];
            if (!(cur < INF)) continue;
            for (int k = 0; k < n; ++k) {
                if (mask & (1 << k)) continue;
                int nmask = mask | (1 << k);
                double cand = cur + d[j][k];
                size_t id = idx(nmask, k);
                if (cand < dp[id]) {
                    dp[id] = cand;
                    parent[id] = (int16_t)j;
                }
            }
        }
    }

    // Pick best endpoint for full visit (path, not cycle)
    int fullMask = FULL - 1;
    double best = INF;
    int bestEnd = 0;
    for (int j = 0; j < n; ++j) {
        double v = dp[idx(fullMask, j)];
        if (v < best) {
            best = v;
            bestEnd = j;
        }
    }

    // Reconstruct path backwards
    std::vector<int> order;
    order.reserve(n);
    int mask = fullMask;
    int cur = bestEnd;
    while (cur != -1) {
        order.push_back(cur);
        int prev = parent[idx(mask, cur)];
        mask ^= (1 << cur);
        cur = prev;
    }
    std::reverse(order.begin(), order.end());

    // Ensure it starts with 0 (it should). If not, rotate (fallback safety).
    auto it0 = std::find(order.begin(), order.end(), 0);
    if (it0 != order.begin() && it0 != order.end()) {
        std::rotate(order.begin(), it0, order.end());
    }
    if (order.empty() || order[0] != 0) {
        // Ultimate safety: output 0..n-1
        order.resize(n);
        std::iota(order.begin(), order.end(), 0);
    }
    return order;
}

static std::vector<int> solve_morton_order(const std::vector<Pt>& pts) {
    const int n = (int)pts.size();
    std::vector<int> order(n);
    if (n == 0) return order;
    if (n == 1) return {0};

    float minx = pts[0].x, maxx = pts[0].x;
    float miny = pts[0].y, maxy = pts[0].y;
    for (int i = 1; i < n; ++i) {
        minx = std::min(minx, pts[i].x);
        maxx = std::max(maxx, pts[i].x);
        miny = std::min(miny, pts[i].y);
        maxy = std::max(maxy, pts[i].y);
    }
    double dx = (double)maxx - (double)minx;
    double dy = (double)maxy - (double)miny;
    if (dx <= 0) dx = 1.0;
    if (dy <= 0) dy = 1.0;

    // Use 16-bit quantization per axis => 32-bit morton.
    std::vector<std::pair<uint32_t, int>> keyed;
    keyed.reserve((size_t)n);
    for (int i = 0; i < n; ++i) {
        double nx = ((double)pts[i].x - (double)minx) / dx;
        double ny = ((double)pts[i].y - (double)miny) / dy;
        uint32_t qx = (uint32_t)std::min(65535.0, std::max(0.0, nx * 65535.0));
        uint32_t qy = (uint32_t)std::min(65535.0, std::max(0.0, ny * 65535.0));
        uint32_t code = morton2D_16(qx, qy);
        keyed.emplace_back(code, i);
    }
    std::sort(keyed.begin(), keyed.end(), [](const auto& a, const auto& b) {
        if (a.first != b.first) return a.first < b.first;
        return a.second < b.second;
    });

    // Build order in morton order
    for (int i = 0; i < n; ++i) order[i] = keyed[i].second;

    // Rotate to start at 0
    auto it0 = std::find(order.begin(), order.end(), 0);
    if (it0 != order.begin() && it0 != order.end()) std::rotate(order.begin(), it0, order.end());

    // If duplicates or missing (shouldn't), fallback to identity.
    if (order.empty() || order[0] != 0) {
        order.resize(n);
        std::iota(order.begin(), order.end(), 0);
        return order;
    }
    return order;
}

static std::vector<int> solve_angle_order(const std::vector<Pt>& pts) {
    const int n = (int)pts.size();
    if (n <= 1) return {0};

    double cx = 0.0, cy = 0.0;
    for (int i = 0; i < n; ++i) {
        cx += (double)pts[i].x;
        cy += (double)pts[i].y;
    }
    cx /= (double)n;
    cy /= (double)n;

    std::vector<std::pair<float, int>> keyed;
    keyed.reserve((size_t)n);
    for (int i = 0; i < n; ++i) {
        float ang = std::atan2((float)((double)pts[i].y - cy), (float)((double)pts[i].x - cx));
        keyed.emplace_back(ang, i);
    }
    std::sort(keyed.begin(), keyed.end(), [](const auto& a, const auto& b) {
        if (a.first != b.first) return a.first < b.first;
        return a.second < b.second;
    });

    std::vector<int> order(n);
    for (int i = 0; i < n; ++i) order[i] = keyed[i].second;
    auto it0 = std::find(order.begin(), order.end(), 0);
    if (it0 != order.begin() && it0 != order.end()) std::rotate(order.begin(), it0, order.end());
    if (order.empty() || order[0] != 0) {
        order.resize(n);
        std::iota(order.begin(), order.end(), 0);
    }
    return order;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int64_t N64;
    if (!(std::cin >> N64)) return 0;
    if (N64 <= 0) return 0;
    if (N64 > (int64_t)std::numeric_limits<int>::max()) {
        // Can't store indices in int safely; output trivial sequence.
        // Still must start at 0 and be a permutation; but we cannot represent full range.
        // Given typical judges, N fits in int. Best effort.
        return 0;
    }
    int N = (int)N64;

    std::vector<Pt> pts;
    pts.resize((size_t)N);
    for (int i = 0; i < N; ++i) {
        float x, y;
        std::cin >> x >> y;
        pts[i] = {x, y};
    }

    std::vector<int> order;
    if (N <= 12) {
        order = solve_exact_small(pts);
    } else if (N <= 200000) {
        order = solve_morton_order(pts);
    } else {
        order = solve_angle_order(pts);
    }

    // Final validity hardening: ensure permutation of 0..N-1 and starts with 0.
    // (This is O(N) and safe even for large N.)
    if ((int)order.size() != N) {
        order.resize(N);
        std::iota(order.begin(), order.end(), 0);
    }
    if (order.empty() || order[0] != 0) {
        // rotate if 0 exists, else reset
        auto it0 = std::find(order.begin(), order.end(), 0);
        if (it0 != order.end()) std::rotate(order.begin(), it0, order.end());
        if (order.empty() || order[0] != 0) {
            std::iota(order.begin(), order.end(), 0);
        }
    }
    {
        std::vector<uint8_t> seen((size_t)N, 0);
        bool ok = true;
        for (int v : order) {
            if (v < 0 || v >= N) { ok = false; break; }
            if (seen[(size_t)v]) { ok = false; break; }
            seen[(size_t)v] = 1;
        }
        if (!ok) {
            std::iota(order.begin(), order.end(), 0);
        }
    }

    for (int i = 0; i < N; ++i) {
        if (i) std::cout << ' ';
        std::cout << order[i];
    }
    std::cout << "\n";
    return 0;
}
