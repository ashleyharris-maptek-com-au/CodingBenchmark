#include <bits/stdc++.h>
using namespace std;

/*
Job-Shop Scheduling (JSSP) constructive solver.

Input:
  J M
  For each job j:
    Tj
    then Tj lines: machine duration

Output:
  For each job j (J lines):
    Tj pairs: start_time machine start_time machine ...

This produces a feasible schedule (not necessarily optimal). It uses adaptive
logic:
  - Small/medium instances: refreshable priority queue for better dispatching.
  - Large instances: fast list scheduling with stale-key skipping.
*/

static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct RNG {
    uint64_t s;
    explicit RNG(uint64_t seed) : s(seed) {}
    uint32_t next_u32() { s = splitmix64(s); return (uint32_t)s; }
};

struct Problem {
    int J = 0, M = 0;
    vector<int> T;
    vector<vector<int>> mach;
    vector<vector<int>> dur;
};

// Fast list scheduling for extreme sizes (O(N log J)).
static void schedule_fast(const Problem& P, vector<vector<int>>& starts) {
    const int J = P.J, M = P.M;
    starts.assign(J, {});
    for (int j = 0; j < J; j++) starts[j].assign(P.T[j], 0);

    vector<int> next_idx(J, 0);
    vector<long long> job_ready(J, 0);
    vector<long long> mach_ready(M, 0);

    struct Node {
        long long key_time;
        int dur;
        int job;
        int idx;
        int machine;
    };
    struct Cmp {
        bool operator()(const Node& a, const Node& b) const {
            if (a.key_time != b.key_time) return a.key_time > b.key_time;
            if (a.dur != b.dur) return a.dur > b.dur;
            if (a.job != b.job) return a.job > b.job;
            return a.idx > b.idx;
        }
    };
    priority_queue<Node, vector<Node>, Cmp> pq;

    auto push_job = [&](int j) {
        int i = next_idx[j];
        if (i >= P.T[j]) return;
        int m = P.mach[j][i];
        int d = P.dur[j][i];
        long long est = max(job_ready[j], mach_ready[m]);
        pq.push(Node{est, d, j, i, m});
    };

    for (int j = 0; j < J; j++) push_job(j);

    while (!pq.empty()) {
        Node n = pq.top(); pq.pop();
        int j = n.job;
        int i = n.idx;
        if (i != next_idx[j]) continue; // stale
        int m = P.mach[j][i];
        int d = P.dur[j][i];
        long long st = max(job_ready[j], mach_ready[m]);
        starts[j][i] = (int)st;
        long long ft = st + d;
        job_ready[j] = ft;
        mach_ready[m] = ft;
        next_idx[j]++;
        push_job(j);
    }
}

// More careful for small/medium: refresh keys to better approximate global earliest-start-first.
static void schedule_small_medium(const Problem& P, vector<vector<int>>& starts) {
    const int J = P.J, M = P.M;
    starts.assign(J, {});
    for (int j = 0; j < J; j++) starts[j].assign(P.T[j], 0);

    vector<int> next_idx(J, 0);
    vector<long long> job_ready(J, 0);
    vector<long long> mach_ready(M, 0);

    struct Cand {
        int job;
        int idx;
        int machine;
        int dur;
        long long est_snapshot;
        uint32_t tie;
    };
    struct Cmp {
        bool operator()(const Cand& a, const Cand& b) const {
            if (a.est_snapshot != b.est_snapshot) return a.est_snapshot > b.est_snapshot;
            if (a.dur != b.dur) return a.dur > b.dur;
            return a.tie > b.tie;
        }
    };
    priority_queue<Cand, vector<Cand>, Cmp> pq;
    RNG rng(0xC0FFEE123456789ULL ^ (uint64_t)P.J * 1315423911ULL ^ (uint64_t)P.M * 2654435761ULL);

    auto push = [&](int j) {
        int i = next_idx[j];
        if (i >= P.T[j]) return;
        int m = P.mach[j][i];
        int d = P.dur[j][i];
        long long est = max(job_ready[j], mach_ready[m]);
        pq.push(Cand{j, i, m, d, est, rng.next_u32()});
    };
    for (int j = 0; j < J; j++) push(j);

    while (!pq.empty()) {
        auto c = pq.top(); pq.pop();
        int j = c.job;
        int i = c.idx;
        if (i != next_idx[j]) continue;
        int m = P.mach[j][i];
        int d = P.dur[j][i];
        long long est = max(job_ready[j], mach_ready[m]);
        if (est != c.est_snapshot) {
            pq.push(Cand{j, i, m, d, est, c.tie});
            continue;
        }
        starts[j][i] = (int)est;
        long long ft = est + d;
        job_ready[j] = ft;
        mach_ready[m] = ft;
        next_idx[j]++;
        push(j);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Problem P;
    if (!(cin >> P.J >> P.M)) return 0;
    P.T.assign(P.J, 0);
    P.mach.assign(P.J, {});
    P.dur.assign(P.J, {});

    long long total_tasks = 0;
    for (int j = 0; j < P.J; j++) {
        int Tj;
        cin >> Tj;
        P.T[j] = Tj;
        P.mach[j].resize(Tj);
        P.dur[j].resize(Tj);
        for (int i = 0; i < Tj; i++) {
            int m, d;
            cin >> m >> d;
            P.mach[j][i] = m;
            P.dur[j][i] = d;
            total_tasks++;
        }
    }

    vector<vector<int>> starts;
    if (total_tasks <= 5000) schedule_small_medium(P, starts);
    else schedule_fast(P, starts);

    for (int j = 0; j < P.J; j++) {
        for (int i = 0; i < P.T[j]; i++) {
            if (i) cout << ' ';
            cout << starts[j][i] << ' ' << P.mach[j][i];
        }
        cout << "\n";
    }
    return 0;
}
