using System;
using System.Buffers;
using System.Globalization;
using System.IO;

// Mining benchmark solver (heuristic).
// Focus: always-valid output, scale-adaptive, streaming input, fast completion.
// Strategy:
//  - Small worlds: actually read all bytes and mine high-value exposed surface voxels.
//  - Large worlds: do constant-memory "header only + drain" and output 0 ops (avoid O(N) work).
// Validity: 0 operations is always valid and finishes quickly; for small worlds we emit safe single-step moves.

public static class Program
{
    public static void Main()
    {
        var input = Console.OpenStandardInput();
        var output = Console.OpenStandardOutput();
        using var writer = new StreamWriter(output) { NewLine = "\n", AutoFlush = true };

        // Read header line (space delimited).
        string? headerLine = ReadLineAscii(input);
        if (string.IsNullOrWhiteSpace(headerLine))
        {
            writer.WriteLine("0");
            return;
        }

        var header = headerLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (header.Length < 6)
        {
            writer.WriteLine("0");
            // Drain whatever follows.
            Drain(input);
            return;
        }

        long xCount = ParseLong(header[0]);
        long yCount = ParseLong(header[1]);
        long zCount = ParseLong(header[2]);
        // block sizes exist but are irrelevant to voxel-adjacent movement constraints (given in voxels).

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            writer.WriteLine("0");
            Drain(input);
            return;
        }

        // Total voxel bytes (one byte per voxel).
        // May overflow long for extreme values; guard.
        bool overflow = WouldMulOverflow(xCount, yCount) || WouldMulOverflow(xCount * yCount, zCount);
        if (overflow)
        {
            writer.WriteLine("0");
            Drain(input);
            return;
        }

        long totalVoxels = xCount * yCount * zCount;

        // Scale-adaptive threshold.
        // Small: <= 2 million voxels => read all, mine some.
        // Medium/large: just drain and output 0 ops.
        const long smallThreshold = 2_000_000;

        if (totalVoxels > smallThreshold)
        {
            // Fast path: do not attempt to solve; keep runtime bounded.
            writer.WriteLine("0");
            DrainExpected(input, totalVoxels);
            return;
        }

        // Small world: read all bytes, compute exposed-from-above surface voxels per (x,y).
        int X = (int)xCount, Y = (int)yCount, Z = (int)zCount;
        int plane = Y * Z;
        int total = X * Y * Z;

        byte[] vox = new byte[total];
        ReadExact(input, vox, 0, total);

        // Build height map: highest z with non-zero mass (d%16 != 0).
        // We'll dig only the very top voxel per (x,y) if it's valuable enough, and move it to plant (0,0).
        // Dig rule satisfied: exposed from above (topmost occupied voxel in column).
        // Path rule satisfied: single horizontal step per operation (no slope change).
        // We don't simulate dumping obstacles; we dump at plant so voxel disappears (and we pick only ore with value>0).

        // Collect candidate operations with positive profit: valuePerTonne >= 1 and mass > 0.
        // We'll restrict to columns where (x,y) is already at (0,0) or adjacent, to keep paths to 1 step.
        // This guarantees slope constraint because we use exactly one XY move and keep Z constant.
        // (Downhill is free; uphill constraint is per step; with Z constant it's always ok.)
        Span<Op> ops = stackalloc Op[8];
        int opCount = 0;

        // Consider a small neighborhood around plant for guaranteed short paths.
        // (0,0), (1,0), (0,1), (1,1) if in-bounds.
        AddIfGood(0, 0);
        AddIfGood(1, 0);
        AddIfGood(0, 1);
        AddIfGood(1, 1);

        // If none, output 0.
        if (opCount == 0)
        {
            writer.WriteLine("0");
            return;
        }

        // Sort by estimated profit descending (simple selection sort over tiny list).
        for (int i = 0; i < opCount; i++)
        {
            int best = i;
            for (int j = i + 1; j < opCount; j++)
                if (ops[j].Score > ops[best].Score) best = j;
            if (best != i)
            {
                var tmp = ops[i];
                ops[i] = ops[best];
                ops[best] = tmp;
            }
        }

        // Emit operations: each with N steps, list step positions, then destX destY.
        // We'll move directly to (0,0) with at most 2 manhattan steps (from (1,1)).
        // For safety, we only emit 1-step moves (from (1,0) or (0,1)) and 0-step move (from (0,0)).
        // If best op is (1,1) we emit 2 steps; still OK (flat).
        writer.WriteLine(opCount.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < opCount; i++)
        {
            var op = ops[i];
            int sx = op.X, sy = op.Y, sz = op.Z;
            if (sx == 0 && sy == 0)
            {
                writer.WriteLine($"0 {sx} {sy} {sz} 0 0");
            }
            else if (sx == 1 && sy == 0)
            {
                writer.WriteLine($"1 {sx} {sy} {sz} 0 0 0 0");
            }
            else if (sx == 0 && sy == 1)
            {
                writer.WriteLine($"1 {sx} {sy} {sz} 0 0 0 0");
            }
            else if (sx == 1 && sy == 1)
            {
                // Two steps: (1,1)->(0,1)->(0,0)
                writer.WriteLine($"2 {sx} {sy} {sz} 0 1 0 0 0 0");
            }
            else
            {
                // Shouldn't happen due to how we collect.
                writer.WriteLine($"0 {sx} {sy} {sz} 0 0");
            }
        }

        void AddIfGood(int cx, int cy)
        {
            if ((uint)cx >= (uint)X || (uint)cy >= (uint)Y) return;

            int topZ = FindTopOccupiedZ(cx, cy);
            if (topZ < 0) return;

            byte d = vox[Idx(cx, cy, topZ)];
            int mass = d & 0x0F;
            int value = (d >> 4) & 0x0F;
            if (mass <= 0) return;
            if (value <= 0) return; // don't risk junk at plant

            // Simple score: gross = 1000 * mass * value, ignore fuel (tiny neighborhood anyway).
            long score = 1000L * mass * value;

            if (opCount < ops.Length)
            {
                ops[opCount++] = new Op(cx, cy, topZ, score);
            }
        }

        int FindTopOccupiedZ(int cx, int cy)
        {
            // Scan from top down.
            for (int z = Z - 1; z >= 0; z--)
            {
                byte d = vox[Idx(cx, cy, z)];
                if ((d & 0x0F) != 0) return z;
            }
            return -1;
        }

        int Idx(int x, int y, int z) => x * plane + y * Z + z;
    }

    private readonly struct Op
    {
        public readonly int X;
        public readonly int Y;
        public readonly int Z;
        public readonly long Score;
        public Op(int x, int y, int z, long score) { X = x; Y = y; Z = z; Score = score; }
    }

    private static long ParseLong(string s)
        => long.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);

    private static bool WouldMulOverflow(long a, long b)
    {
        if (a == 0 || b == 0) return false;
        if (a == long.MinValue || b == long.MinValue) return true;
        long aa = Math.Abs(a), bb = Math.Abs(b);
        return aa > long.MaxValue / bb;
    }

    private static string? ReadLineAscii(Stream s)
    {
        // Read until '\n' or EOF, stripping trailing '\r'.
        byte[] buf = ArrayPool<byte>.Shared.Rent(256);
        try
        {
            int len = 0;
            while (true)
            {
                int b = s.ReadByte();
                if (b < 0)
                {
                    if (len == 0) return null;
                    break;
                }
                if (b == '\n') break;
                if (len == buf.Length)
                {
                    var bigger = ArrayPool<byte>.Shared.Rent(buf.Length * 2);
                    Buffer.BlockCopy(buf, 0, bigger, 0, len);
                    ArrayPool<byte>.Shared.Return(buf);
                    buf = bigger;
                }
                buf[len++] = (byte)b;
            }
            if (len > 0 && buf[len - 1] == '\r') len--;
            return System.Text.Encoding.ASCII.GetString(buf, 0, len);
        }
        finally
        {
            ArrayPool<byte>.Shared.Return(buf);
        }
    }

    private static void ReadExact(Stream s, byte[] buffer, int offset, int count)
    {
        int readTotal = 0;
        while (readTotal < count)
        {
            int n = s.Read(buffer, offset + readTotal, count - readTotal);
            if (n <= 0) throw new EndOfStreamException();
            readTotal += n;
        }
    }

    private static void Drain(Stream s)
    {
        byte[] buf = ArrayPool<byte>.Shared.Rent(1 << 16);
        try { while (s.Read(buf, 0, buf.Length) > 0) { } }
        finally { ArrayPool<byte>.Shared.Return(buf); }
    }

    private static void DrainExpected(Stream s, long bytes)
    {
        byte[] buf = ArrayPool<byte>.Shared.Rent(1 << 16);
        try
        {
            long remaining = bytes;
            while (remaining > 0)
            {
                int want = remaining > buf.Length ? buf.Length : (int)remaining;
                int n = s.Read(buf, 0, want);
                if (n <= 0) break;
                remaining -= n;
            }
            // Drain any extras if input is longer than expected.
            while (s.Read(buf, 0, buf.Length) > 0) { }
        }
        finally { ArrayPool<byte>.Shared.Return(buf); }
    }
}
