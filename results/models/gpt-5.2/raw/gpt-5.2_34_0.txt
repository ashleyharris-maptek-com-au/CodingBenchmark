```python
import sys
import math
import random


def dist(a, b):
    dx = a[0] - b[0]
    dy = a[1] - b[1]
    return math.hypot(dx, dy)


def route_distance(route, pts, depot):
    if not route:
        return 0.0
    d = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        d += dist(pts[route[i]], pts[route[i + 1]])
    d += dist(pts[route[-1]], depot)
    return d


def total_distance(routes, pts, depot):
    return sum(route_distance(r, pts, depot) for r in routes)


def compute_demands(routes, demands):
    return [sum(demands[i] for i in r) for r in routes]


def two_opt(route, pts, depot):
    n = len(route)
    if n < 4:
        return route
    improved = True
    best = route
    best_d = route_distance(best, pts, depot)
    while improved:
        improved = False
        for i in range(0, n - 2):
            for j in range(i + 2, n):
                if j - i == 1:
                    continue
                cand = best[:i + 1] + list(reversed(best[i + 1:j + 1])) + best[j + 1:]
                cd = route_distance(cand, pts, depot)
                if cd + 1e-12 < best_d:
                    best_d = cd
                    best = cand
                    improved = True
        n = len(best)
    return best


def build_initial_routes(N, K, CAP, depot, pts, demands):
    customers = list(range(N))
    customers.sort(key=lambda i: dist(depot, pts[i]), reverse=True)

    routes = [[] for _ in range(min(K, N))]
    loads = [0] * len(routes)

    for c in customers:
        best_r = -1
        best_inc = None
        for r_idx in range(len(routes)):
            if loads[r_idx] + demands[c] > CAP:
                continue
            r = routes[r_idx]
            if not r:
                inc = 2.0 * dist(depot, pts[c])
            else:
                inc = (
                    dist(depot, pts[r[0]])
                    + dist(pts[r[-1]], depot)
                    - (dist(depot, pts[c]) + dist(pts[c], depot))
                )
                inc = route_distance(r + [c], pts, depot) - route_distance(r, pts, depot)
            if best_inc is None or inc < best_inc:
                best_inc = inc
                best_r = r_idx
        if best_r == -1:
            routes.append([])
            loads.append(0)
            best_r = len(routes) - 1
        routes[best_r].append(c)
        loads[best_r] += demands[c]

    routes = [r for r in routes if r]
    return routes


def greedy_insert_order(route, pts, depot):
    if not route:
        return route
    unvisited = set(route)
    current = None
    best_start = min(unvisited, key=lambda i: dist(depot, pts[i]))
    current = best_start
    unvisited.remove(current)
    ordered = [current]
    while unvisited:
        nxt = min(unvisited, key=lambda j: dist(pts[current], pts[j]))
        unvisited.remove(nxt)
        ordered.append(nxt)
        current = nxt
    return ordered


def improve_routes(routes, pts, depot):
    improved = []
    for r in routes:
        r2 = greedy_insert_order(r, pts, depot)
        r2 = two_opt(r2, pts, depot)
        improved.append(r2)
    return improved


def relocate_move(routes, loads, demands, pts, depot, CAP):
    best_delta = 0.0
    best = None
    for ri, r in enumerate(routes):
        if len(r) <= 1:
            continue
        for pos, c in enumerate(r):
            for rj, r2 in enumerate(routes):
                if ri == rj:
                    continue
                if loads[rj] + demands[c] > CAP:
                    continue
                for ins in range(len(r2) + 1):
                    old = route_distance(r, pts, depot) + route_distance(r2, pts, depot)
                    nr = r[:pos] + r[pos + 1:]
                    nr2 = r2[:ins] + [c] + r2[ins:]
                    new = route_distance(nr, pts, depot) + route_distance(nr2, pts, depot)
                    delta = new - old
                    if delta < best_delta - 1e-12:
                        best_delta = delta
                        best = (ri, pos, rj, ins, c)
    if best is None:
        return False
    ri, pos, rj, ins, c = best
    routes[ri].pop(pos)
    routes[rj].insert(ins, c)
    loads[ri] -= demands[c]
    loads[rj] += demands[c]
    if not routes[ri]:
        routes.pop(ri)
        loads.pop(ri)
    return True


def swap_move(routes, loads, demands, pts, depot, CAP):
    best_delta = 0.0
    best = None
    for ri in range(len(routes)):
        for rj in range(ri + 1, len(routes)):
            r1, r2 = routes[ri], routes[rj]
            for i, a in enumerate(r1):
                for j, b in enumerate(r2):
                    l1 = loads[ri] - demands[a] + demands[b]
                    l2 = loads[rj] - demands[b] + demands[a]
                    if l1 > CAP or l2 > CAP:
                        continue
                    old = route_distance(r1, pts, depot) + route_distance(r2, pts, depot)
                    nr1 = r1[:]
                    nr2 = r2[:]
                    nr1[i], nr2[j] = nr2[j], nr1[i]
                    new = route_distance(nr1, pts, depot) + route_distance(nr2, pts, depot)
                    delta = new - old
                    if delta < best_delta - 1e-12:
                        best_delta = delta
                        best = (ri, i, rj, j)
    if best is None:
        return False
    ri, i, rj, j = best
    a = routes[ri][i]
    b = routes[rj][j]
    routes[ri][i], routes[rj][j] = b, a
    loads[ri] = loads[ri] - demands[a] + demands[b]
    loads[rj] = loads[rj] - demands[b] + demands[a]
    return True


def local_search(routes, pts, depot, demands, CAP, iters=200):
    routes = improve_routes(routes, pts, depot)
    loads = compute_demands(routes, demands)
    for r in range(len(routes)):
        routes[r] = two_opt(routes[r], pts, depot)
    for _ in range(iters):
        moved = relocate_move(routes, loads, demands, pts, depot, CAP)
        if moved:
            continue
        moved = swap_move(routes, loads, demands, pts, depot, CAP)
        if moved:
            continue
        break
    for r in range(len(routes)):
        routes[r] = two_opt(routes[r], pts, depot)
    return routes


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = int(next(it))
    depot = (float(next(it)), float(next(it)))
    pts = []
    demands = []
    for _ in range(N):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        pts.append((x, y))
        demands.append(d)

    routes = build_initial_routes(N, K, CAP, depot, pts, demands)
    routes = local_search(routes, pts, depot, demands, CAP, iters=150)

    # Ensure feasibility: if any route exceeds capacity (shouldn't), split.
    feasible_routes = []
    for r in routes:
        cur = []
        load = 0
        for c in r:
            if load + demands[c] <= CAP:
                cur.append(c)
                load += demands[c]
            else:
                if cur:
                    feasible_routes.append(cur)
                cur = [c]
                load = demands[c]
        if cur:
            feasible_routes.append(cur)

    # If we have more than K routes, keep as is (problem says "up to K", but feasibility is paramount).
    routes = feasible_routes

    td = total_distance(routes, pts, depot)
    sys.stdout.write(f"{td:.6f}\n")
    sys.stdout.write(f"{len(routes)}\n")
    for r in routes:
        sys.stdout.write(" ".join(map(str, r)) + "\n")


if __name__ == "__main__":
    random.seed(0)
    main()
```
