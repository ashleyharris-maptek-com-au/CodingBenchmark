@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, s: u32) -> u32 {
  return (x << s) | (x >> (32u - s));
}

fn quarter_round(a: u32, b: u32, c: u32, d: u32) -> vec4<u32> {
  var aa: u32 = a;
  var bb: u32 = b;
  var cc: u32 = c;
  var dd: u32 = d;

  aa = aa + bb; dd = dd ^ aa; dd = rotl32(dd, 16u);
  cc = cc + dd; bb = bb ^ cc; bb = rotl32(bb, 12u);
  aa = aa + bb; dd = dd ^ aa; dd = rotl32(dd, 8u);
  cc = cc + dd; bb = bb ^ cc; bb = rotl32(bb, 7u);

  return vec4<u32>(aa, bb, cc, dd);
}

fn custom_hash(data: array<u32, 16>, nonce: u32) -> vec2<u32> {
  var state: array<u32, 16>;

  // state[i] = data[i] ^ (nonce + i)
  state[0]  = data[0]  ^ (nonce + 0u);
  state[1]  = data[1]  ^ (nonce + 1u);
  state[2]  = data[2]  ^ (nonce + 2u);
  state[3]  = data[3]  ^ (nonce + 3u);
  state[4]  = data[4]  ^ (nonce + 4u);
  state[5]  = data[5]  ^ (nonce + 5u);
  state[6]  = data[6]  ^ (nonce + 6u);
  state[7]  = data[7]  ^ (nonce + 7u);
  state[8]  = data[8]  ^ (nonce + 8u);
  state[9]  = data[9]  ^ (nonce + 9u);
  state[10] = data[10] ^ (nonce + 10u);
  state[11] = data[11] ^ (nonce + 11u);
  state[12] = data[12] ^ (nonce + 12u);
  state[13] = data[13] ^ (nonce + 13u);
  state[14] = data[14] ^ (nonce + 14u);
  state[15] = data[15] ^ (nonce + 15u);

  // 8 rounds; inner quarter-round groups unrolled explicitly.
  for (var r: u32 = 0u; r < 8u; r = r + 1u) {
    var q0: vec4<u32> = quarter_round(state[0], state[1], state[2], state[3]);
    state[0] = q0.x; state[1] = q0.y; state[2] = q0.z; state[3] = q0.w;

    var q1: vec4<u32> = quarter_round(state[4], state[5], state[6], state[7]);
    state[4] = q1.x; state[5] = q1.y; state[6] = q1.z; state[7] = q1.w;

    var q2: vec4<u32> = quarter_round(state[8], state[9], state[10], state[11]);
    state[8] = q2.x; state[9] = q2.y; state[10] = q2.z; state[11] = q2.w;

    var q3: vec4<u32> = quarter_round(state[12], state[13], state[14], state[15]);
    state[12] = q3.x; state[13] = q3.y; state[14] = q3.z; state[15] = q3.w;
  }

  let h0: u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
  let h1: u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];
  return vec2<u32>(h0, h1);
}

fn ctz32(x: u32) -> u32 {
  // Returns 0..32
  if (x == 0u) {
    return 32u;
  }
  var n: u32 = 0u;
  var v: u32 = x;
  for (var i: u32 = 0u; i < 32u; i = i + 1u) {
    if ((v & 1u) == 1u) {
      break;
    }
    n = n + 1u;
    v = v >> 1u;
  }
  return n;
}

fn trailing_zeros_64(h0: u32, h1: u32) -> u32 {
  if (h0 != 0u) {
    return ctz32(h0);
  }
  return 32u + ctz32(h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  let tid: u32 = gid.x;
  if (tid >= params.z) {
    return;
  }

  let nonce: u32 = params.y + tid;
  let h: vec2<u32> = custom_hash(baseData, nonce);
  let tz: u32 = trailing_zeros_64(h.x, h.y);

  if (tz >= params.x) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h.x;
    result[3] = h.y;
  }
}
