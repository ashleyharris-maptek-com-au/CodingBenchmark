#include <algorithm>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <random>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

// N-D snake solver.
// Strategy:
// 1) If we have a path to current target food under conservative collision rules, follow it.
// 2) Otherwise, take a safe step that maximizes distance from hazards and (weakly) approaches food.
// Designed to work for D up to 20 and varying hyper-rectangle sizes.

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        // 64-bit FNV-1a-ish mixing
        uint64_t h = 1469598103934665603ULL;
        for (int x : v) {
            uint64_t z = (uint64_t)(uint32_t)x + 0x9e3779b97f4a7c15ULL;
            h ^= z + (h << 6) + (h >> 2);
        }
        return (size_t)h;
    }
};

static inline uint64_t mix64(uint64_t x) {
    x ^= x >> 33;
    x *= 0xff51afd7ed558ccdULL;
    x ^= x >> 33;
    x *= 0xc4ceb9fe1a85ec53ULL;
    x ^= x >> 33;
    return x;
}

struct KeyHash {
    size_t operator()(uint64_t k) const noexcept { return (size_t)mix64(k); }
};

static uint64_t packKey(const vector<int>& p, const vector<int>& bounds) {
    // Mixed-radix packing into 64-bit. If overflow risk, fall back to hashed mixing.
    __int128 acc = 0;
    __int128 mul = 1;
    bool overflow = false;
    for (int i = 0; i < (int)p.size(); i++) {
        __int128 b = bounds[i];
        if (b <= 0) { overflow = true; break; }
        acc += mul * (__int128)p[i];
        mul *= b;
        if (mul > (__int128)numeric_limits<uint64_t>::max() ||
            acc > (__int128)numeric_limits<uint64_t>::max()) {
            overflow = true;
            break;
        }
    }
    if (!overflow) return (uint64_t)acc;
    // Hash-mix coordinates if packing doesn't fit.
    uint64_t h = 0x84222325a2b77a5dULL;
    for (int x : p) {
        h ^= mix64((uint64_t)(uint32_t)x + 0x9e3779b97f4a7c15ULL);
        h = mix64(h + 0x27d4eb2f165667c5ULL);
    }
    return h;
}

static inline int manhattan(const vector<int>& a, const vector<int>& b) {
    int s = 0;
    for (size_t i = 0; i < a.size(); i++) s += abs(a[i] - b[i]);
    return s;
}

static inline bool inBounds(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); i++) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static inline bool eqVec(const vector<int>& a, const vector<int>& b) {
    return a.size() == b.size() && equal(a.begin(), a.end(), b.begin());
}

static vector<pair<int,int>> neighborMoves(int D) {
    vector<pair<int,int>> mv;
    mv.reserve((size_t)D * 2);
    for (int ax = 0; ax < D; ax++) {
        mv.push_back({ax, +1});
        mv.push_back({ax, -1});
    }
    return mv;
}

// Limited A* search on grid with obstacles and (conservative) snake body treated as blocked.
// Returns sequence of moves (axis, dir) from start (head) to goal, excluding start.
static bool astar_path(
    const vector<int>& start,
    const vector<int>& goal,
    const vector<int>& bounds,
    const unordered_set<uint64_t, KeyHash>& blocked,
    int node_budget,
    vector<pair<int,int>>& out_moves
) {
    if (eqVec(start, goal)) { out_moves.clear(); return true; }

    const int D = (int)start.size();
    auto moves = neighborMoves(D);

    struct Node {
        int f, g;
        uint64_t key;
        vector<int> pos;
    };
    struct Cmp { bool operator()(const Node& a, const Node& b) const { return a.f > b.f; } };

    priority_queue<Node, vector<Node>, Cmp> pq;
    unordered_map<uint64_t, int, KeyHash> bestg;
    unordered_map<uint64_t, pair<uint64_t, pair<int,int>>, KeyHash> parent; // child -> (parKey, move)

    uint64_t sKey = packKey(start, bounds);
    uint64_t gKey = packKey(goal, bounds);
    pq.push(Node{manhattan(start, goal), 0, sKey, start});
    bestg[sKey] = 0;

    int expanded = 0;
    while (!pq.empty() && expanded < node_budget) {
        Node cur = pq.top(); pq.pop();
        auto itg = bestg.find(cur.key);
        if (itg == bestg.end() || cur.g != itg->second) continue;
        expanded++;
        if (cur.key == gKey && eqVec(cur.pos, goal)) {
            // reconstruct
            vector<pair<int,int>> rev;
            uint64_t k = cur.key;
            while (k != sKey) {
                auto itp = parent.find(k);
                if (itp == parent.end()) break;
                rev.push_back(itp->second.second);
                k = itp->second.first;
            }
            reverse(rev.begin(), rev.end());
            out_moves = std::move(rev);
            return true;
        }

        for (auto [ax, dir] : moves) {
            vector<int> nxt = cur.pos;
            nxt[ax] += dir;
            if (!inBounds(nxt, bounds)) continue;
            uint64_t nk = packKey(nxt, bounds);
            if (blocked.find(nk) != blocked.end()) continue;
            int ng = cur.g + 1;
            auto bg = bestg.find(nk);
            if (bg == bestg.end() || ng < bg->second) {
                bestg[nk] = ng;
                int h = manhattan(nxt, goal);
                pq.push(Node{ng + h, ng, nk, std::move(nxt)});
                parent[nk] = {cur.key, {ax, dir}};
            }
        }
    }
    return false;
}

static int score_step(
    const vector<int>& head,
    const vector<int>& nxt,
    const vector<int>& food,
    const vector<int>& bounds,
    const unordered_set<uint64_t, KeyHash>& obstacles,
    const unordered_set<uint64_t, KeyHash>& body_blocked,
    uint64_t tailKey,
    bool can_use_tail_cell
) {
    // Higher is better.
    // - Strongly prefer not stepping into immediate traps near boundaries/obstacles/body.
    // - Prefer getting closer to food.
    // - Prefer larger "free degree" (neighbors available).

    if (!inBounds(nxt, bounds)) return numeric_limits<int>::min() / 4;
    uint64_t nk = packKey(nxt, bounds);
    if (obstacles.find(nk) != obstacles.end()) return numeric_limits<int>::min() / 4;

    // Body collision: allow stepping into current tail position iff we are not growing this turn.
    if (body_blocked.find(nk) != body_blocked.end()) {
        if (!(can_use_tail_cell && nk == tailKey)) return numeric_limits<int>::min() / 4;
    }

    int s = 0;
    int df = manhattan(nxt, food);
    s -= df * 10;

    // Boundary penalty (avoid corners/edges in high-D too)
    int boundary = 0;
    for (size_t i = 0; i < nxt.size(); i++) {
        boundary += min(nxt[i], bounds[i] - 1 - nxt[i]);
    }
    s += boundary * 2;

    // Local freedom
    int D = (int)nxt.size();
    int free_deg = 0;
    for (int ax = 0; ax < D; ax++) {
        for (int dir : {+1, -1}) {
            vector<int> nn = nxt;
            nn[ax] += dir;
            if (!inBounds(nn, bounds)) continue;
            uint64_t kk = packKey(nn, bounds);
            if (obstacles.find(kk) != obstacles.end()) continue;
            if (body_blocked.find(kk) != body_blocked.end()) {
                if (!(can_use_tail_cell && kk == tailKey)) continue;
            }
            free_deg++;
        }
    }
    s += free_deg * 15;

    // Slight preference for not reversing into tight area: reward moving away from head boundary
    (void)head;
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    long long max_turns;
    if (!(cin >> D >> max_turns)) return 0;
    vector<int> bounds(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    int snake_len;
    vector<int> head(D);
    cin >> snake_len;
    for (int i = 0; i < D; i++) cin >> head[i];

    // We are not given the rest of the body; assume initial length occupies only head.
    deque<vector<int>> snake;
    snake.push_front(head);
    if (snake_len < 1) snake_len = 1;

    int F;
    cin >> F;
    vector<vector<int>> foods(F, vector<int>(D));
    for (int i = 0; i < F; i++) {
        for (int j = 0; j < D; j++) cin >> foods[i][j];
    }
    int O;
    cin >> O;
    unordered_set<uint64_t, KeyHash> obstacles;
    obstacles.reserve((size_t)O * 2 + 8);
    for (int i = 0; i < O; i++) {
        vector<int> p(D);
        for (int j = 0; j < D; j++) cin >> p[j];
        obstacles.insert(packKey(p, bounds));
    }

    // RNG for tie-breaking
    uint64_t seed = 0x1234567890abcdefULL;
    for (int x : bounds) seed = mix64(seed ^ (uint64_t)(uint32_t)x);
    for (int x : head) seed = mix64(seed ^ (uint64_t)(uint32_t)x + 0x9e3779b97f4a7c15ULL);
    std::mt19937_64 rng(seed);

    // Food index pointer: always chase nearest remaining.
    vector<char> eaten(F, 0);
    int remaining = F;

    auto choose_food = [&]() -> int {
        int best = -1;
        int bestd = numeric_limits<int>::max();
        for (int i = 0; i < F; i++) {
            if (eaten[i]) continue;
            int d = manhattan(snake.front(), foods[i]);
            if (d < bestd) { bestd = d; best = i; }
        }
        return best;
    };

    for (long long turn = 0; turn < max_turns; turn++) {
        if (remaining == 0) break;

        int fi = choose_food();
        if (fi < 0) break;
        const vector<int>& food = foods[fi];

        // Build blocked set for pathfinding.
        // Conservative: treat all snake cells except tail as blocked.
        unordered_set<uint64_t, KeyHash> body_blocked;
        body_blocked.reserve(snake.size() * 2 + 8);
        for (size_t i = 0; i < snake.size(); i++) body_blocked.insert(packKey(snake[i], bounds));
        uint64_t tailKey = packKey(snake.back(), bounds);

        bool will_grow_if_step_on_food = true; // depends on chosen move

        // Attempt A* with a budget depending on dimension/space.
        // Larger D often means smaller bounds; still keep budget bounded.
        int base_budget = 5000;
        if (D <= 2) base_budget = 40000;
        else if (D <= 4) base_budget = 20000;
        else if (D <= 8) base_budget = 12000;
        else base_budget = 8000;

        vector<pair<int,int>> planned;
        // Allow stepping onto tail cell for path search only if we are not growing on that step,
        // but A* doesn't simulate growth; keep conservative: block entire body.
        unordered_set<uint64_t, KeyHash> blocked = obstacles;
        for (auto k : body_blocked) blocked.insert(k);

        bool have_path = astar_path(snake.front(), food, bounds, blocked, base_budget, planned);

        pair<int,int> move = {0, +1};
        bool picked = false;

        if (have_path && !planned.empty()) {
            move = planned[0];
            picked = true;
        }

        // If no path (or empty), pick best safe step by heuristic, allowing tail reuse when not growing.
        if (!picked) {
            vector<pair<int,int>> mvs = neighborMoves(D);
            // Shuffle for tie-breaking
            shuffle(mvs.begin(), mvs.end(), rng);
            int bestScore = numeric_limits<int>::min() / 4;
            pair<int,int> bestMv = {0, +1};
            vector<int> bestNxt;

            for (auto [ax, dir] : mvs) {
                vector<int> nxt = snake.front();
                nxt[ax] += dir;
                uint64_t nk = packKey(nxt, bounds);
                bool grows = eqVec(nxt, food);
                bool can_use_tail = !grows;
                int sc = score_step(snake.front(), nxt, food, bounds, obstacles, body_blocked, tailKey, can_use_tail);
                if (sc > bestScore) {
                    bestScore = sc;
                    bestMv = {ax, dir};
                    bestNxt = std::move(nxt);
                }
            }

            // Fallback: if all moves are terrible (shouldn't happen often), just pick first in-bounds non-obstacle.
            if (bestScore <= numeric_limits<int>::min() / 8) {
                bool found = false;
                for (auto [ax, dir] : mvs) {
                    vector<int> nxt = snake.front();
                    nxt[ax] += dir;
                    if (!inBounds(nxt, bounds)) continue;
                    uint64_t nk = packKey(nxt, bounds);
                    if (obstacles.find(nk) != obstacles.end()) continue;
                    // allow tail reuse if not growing
                    bool grows = eqVec(nxt, food);
                    if (body_blocked.find(nk) != body_blocked.end()) {
                        if (!( !grows && nk == tailKey)) continue;
                    }
                    bestMv = {ax, dir};
                    found = true;
                    break;
                }
                if (!found) {
                    // Truly boxed in: output something valid.
                    bestMv = {0, +1};
                }
            }

            move = bestMv;
        }

        // Emit chosen move
        cout << move.first << " " << move.second << "\n";

        // Update local snake simulation so future turns are coherent.
        vector<int> newHead = snake.front();
        newHead[move.first] += move.second;
        bool grows = (inBounds(newHead, bounds) && eqVec(newHead, food));

        // If we collide, we still output moves; judge will end game. Keep simulation minimal.
        snake.push_front(newHead);
        if (grows) {
            eaten[fi] = 1;
            remaining--;
            snake_len++;
        } else {
            // maintain length
            while ((int)snake.size() > snake_len) snake.pop_back();
        }
    }

    return 0;
}
