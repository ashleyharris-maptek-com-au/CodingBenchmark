use std::cmp::Ordering;
use std::collections::{BTreeMap, HashMap};
use std::io::{self, Read};

fn gcd_i64(mut a: i64, mut b: i64) -> i64 {
    if a < 0 {
        a = -a;
    }
    if b < 0 {
        b = -b;
    }
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

fn solve_small_mitm(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    let mid = n / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let left_n = left.len();
    let right_n = right.len();

    let left_size: usize = 1usize << left_n;
    let right_size: usize = 1usize << right_n;

    let mut left_sums: Vec<(i64, u64)> = Vec::with_capacity(left_size);
    for mask in 0u64..(left_size as u64) {
        let mut s: i64 = 0;
        let mut i = 0usize;
        while i < left_n {
            if (mask >> i) & 1 != 0 {
                s = s.wrapping_add(left[i].value);
            }
            i += 1;
        }
        left_sums.push((s, mask));
    }
    left_sums.sort_by(|a, b| a.0.cmp(&b.0));

    for mask in 0u64..(right_size as u64) {
        let mut s: i64 = 0;
        let mut i = 0usize;
        while i < right_n {
            if (mask >> i) & 1 != 0 {
                s = s.wrapping_add(right[i].value);
            }
            i += 1;
        }
        let need = target.wrapping_sub(s);
        if let Ok(pos) = left_sums.binary_search_by(|x| x.0.cmp(&need)) {
            let left_mask = left_sums[pos].1;
            let mut res: Vec<usize> = Vec::new();
            for i in 0..left_n {
                if (left_mask >> i) & 1 != 0 {
                    res.push(left[i].index);
                }
            }
            for i in 0..right_n {
                if (mask >> i) & 1 != 0 {
                    res.push(right[i].index);
                }
            }
            return Some(res);
        }
    }
    None
}

fn solve_nonneg_bitset(items: &[Item], target: usize) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    let word_bits: usize = 64;
    let words = (target / word_bits) + 1;
    let mut bits = vec![0u64; words];
    bits[0] = 1u64;

    let mut prev_sum = vec![usize::MAX; target + 1];
    let mut prev_item = vec![usize::MAX; target + 1];

    for (pos, it) in items.iter().enumerate() {
        let v_i64 = it.value;
        if v_i64 <= 0 {
            continue;
        }
        let v = v_i64 as usize;
        if v > target {
            continue;
        }
        let shift_words = v / word_bits;
        let shift_bits = v % word_bits;

        let mut shifted = vec![0u64; words];
        for i in 0..words {
            let src = bits[i];
            if src == 0 {
                continue;
            }
            let di = i + shift_words;
            if di >= words {
                continue;
            }
            shifted[di] |= src << shift_bits;
            if shift_bits != 0 && di + 1 < words {
                shifted[di + 1] |= src >> (word_bits - shift_bits);
            }
        }

        for i in 0..words {
            let newly = shifted[i] & !bits[i];
            if newly != 0 {
                let mut w = newly;
                while w != 0 {
                    let tz = w.trailing_zeros() as usize;
                    let sum = i * word_bits + tz;
                    if sum <= target {
                        prev_sum[sum] = sum - v;
                        prev_item[sum] = pos;
                    }
                    w &= w - 1;
                }
            }
            bits[i] |= shifted[i];
        }

        let word = bits[target / word_bits];
        if ((word >> (target % word_bits)) & 1) != 0 {
            let mut res = Vec::new();
            let mut s = target;
            while s != 0 {
                let pi = prev_item[s];
                if pi == usize::MAX {
                    break;
                }
                res.push(items[pi].index);
                s = prev_sum[s];
            }
            res.sort_unstable();
            return Some(res);
        }
    }
    None
}

fn solve_hash_dp(items: &[Item], target: i64, cap_states: usize) -> Option<Vec<usize>> {
    let mut states: HashMap<i64, (i64, usize)> = HashMap::new();
    states.insert(0, (0, usize::MAX));

    for (pos, it) in items.iter().enumerate() {
        let v = it.value;
        if states.contains_key(&target) {
            break;
        }
        let mut updates: Vec<(i64, (i64, usize))> = Vec::new();
        updates.reserve(states.len().min(4096));
        for (&s, _) in states.iter() {
            let ns = s.wrapping_add(v);
            if !states.contains_key(&ns) {
                updates.push((ns, (s, pos)));
            }
        }
        for (ns, prev) in updates {
            states.insert(ns, prev);
        }
        if states.len() > cap_states {
            let mut vec: Vec<(i64, (i64, usize))> = states.drain().collect();
            vec.sort_by(|a, b| {
                let da = (a.0 - target).abs();
                let db = (b.0 - target).abs();
                da.cmp(&db).then_with(|| a.0.cmp(&b.0))
            });
            vec.truncate(cap_states);
            states.extend(vec.into_iter());
        }
    }

    if !states.contains_key(&target) {
        return None;
    }
    let mut res: Vec<usize> = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev_sum, item_pos) = states[&cur];
        if item_pos == usize::MAX {
            break;
        }
        res.push(items[item_pos].index);
        cur = prev_sum;
    }
    res.sort_unstable();
    Some(res)
}

fn solve_window_bounded(
    items: &[Item],
    target: i64,
    window: i64,
    cap_states: usize,
) -> Option<Vec<usize>> {
    let mut map: BTreeMap<i64, (i64, usize)> = BTreeMap::new();
    map.insert(0, (0, usize::MAX));

    for (pos, it) in items.iter().enumerate() {
        if map.contains_key(&target) {
            break;
        }
        let v = it.value;
        let keys: Vec<i64> = map.keys().copied().collect();
        for s in keys {
            let ns = s.wrapping_add(v);
            if ns < target - window || ns > target + window {
                continue;
            }
            if !map.contains_key(&ns) {
                map.insert(ns, (s, pos));
            }
        }
        if map.len() > cap_states {
            let mut vec: Vec<(i64, (i64, usize))> = map.into_iter().collect();
            vec.sort_by(|a, b| {
                let da = (a.0 - target).abs();
                let db = (b.0 - target).abs();
                da.cmp(&db).then_with(|| a.0.cmp(&b.0))
            });
            vec.truncate(cap_states);
            map = vec.into_iter().collect();
        }
    }

    if !map.contains_key(&target) {
        return None;
    }
    let mut res = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev_sum, item_pos) = map[&cur];
        if item_pos == usize::MAX {
            break;
        }
        res.push(items[item_pos].index);
        cur = prev_sum;
    }
    res.sort_unstable();
    Some(res)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n_opt = it.next();
    if n_opt.is_none() {
        return;
    }
    let n: usize = n_opt.unwrap().parse().unwrap();
    let target: i64 = it.next().unwrap().parse().unwrap();

    let mut items: Vec<Item> = Vec::with_capacity(n);
    for idx in 0..n {
        if let Some(s) = it.next() {
            let v: i64 = s.parse().unwrap();
            items.push(Item { value: v, index: idx });
        } else {
            break;
        }
    }

    if target == 0 {
        println!("YES");
        println!();
        return;
    }

    for it in &items {
        if it.value == target {
            println!("YES");
            println!("{}", it.index);
            return;
        }
    }

    let mut g: i64 = 0;
    for it in &items {
        g = gcd_i64(g, it.value);
        if g == 1 {
            break;
        }
    }
    if g != 0 && (target % g) != 0 {
        println!("NO");
        return;
    }

    if items.len() <= 44 {
        if let Some(sol) = solve_small_mitm(&items, target) {
            println!("YES");
            if sol.is_empty() {
                println!();
            } else {
                let mut out = String::new();
                for (i, idx) in sol.iter().enumerate() {
                    if i != 0 {
                        out.push(' ');
                    }
                    out.push_str(&idx.to_string());
                }
                println!("{}", out);
            }
            return;
        }
        println!("NO");
        return;
    }

    let all_nonneg = items.iter().all(|x| x.value >= 0);
    if all_nonneg && target > 0 {
        let t_usize = match usize::try_from(target) {
            Ok(v) => v,
            Err(_) => {
                println!("NO");
                return;
            }
        };
        let max_bits = 50_000_000usize;
        if t_usize <= max_bits {
            let mut positives: Vec<Item> = items.iter().copied().filter(|x| x.value > 0).collect();
            positives.sort_by(|a, b| b.value.cmp(&a.value));
            if let Some(sol) = solve_nonneg_bitset(&positives, t_usize) {
                println!("YES");
                if sol.is_empty() {
                    println!();
                } else {
                    let mut out = String::new();
                    for (i, idx) in sol.iter().enumerate() {
                        if i != 0 {
                            out.push(' ');
                        }
                        out.push_str(&idx.to_string());
                    }
                    println!("{}", out);
                }
                return;
            }
            println!("NO");
            return;
        }
    }

    let mut by_abs: Vec<Item> = items.clone();
    by_abs.sort_by(|a, b| b.value.abs().cmp(&a.value.abs()));

    let cap_states: usize = 250_000;
    if items.len() <= 6000 {
        if let Some(sol) = solve_hash_dp(&by_abs, target, cap_states) {
            println!("YES");
            if sol.is_empty() {
                println!();
            } else {
                let mut out = String::new();
                for (i, idx) in sol.iter().enumerate() {
                    if i != 0 {
                        out.push(' ');
                    }
                    out.push_str(&idx.to_string());
                }
                println!("{}", out);
            }
            return;
        }
    }

    let window = (target.abs() / 50).max(1_000_000);
    let max_use = 200_000usize.min(by_abs.len());
    let slice = &by_abs[..max_use];
    if let Some(sol) = solve_window_bounded(slice, target, window, cap_states) {
        println!("YES");
        if sol.is_empty() {
            println!();
        } else {
            let mut out = String::new();
            for (i, idx) in sol.iter().enumerate() {
                if i != 0 {
                    out.push(' ');
                }
                out.push_str(&idx.to_string());
            }
            println!("{}", out);
        }
        return;
    }

    if all_nonneg && target > 0 {
        let mut positives: Vec<Item> = items.iter().copied().filter(|x| x.value > 0).collect();
        positives.sort_by(|a, b| b.value.cmp(&a.value));
        let mut sum: i64 = 0;
        let mut res: Vec<usize> = Vec::new();
        for it in &positives {
            if sum < target {
                let ns = sum + it.value;
                if ns <= target {
                    sum = ns;
                    res.push(it.index);
                }
            }
            if sum == target {
                break;
            }
        }
        if sum == target {
            res.sort_unstable();
            println!("YES");
            let mut out = String::new();
            for (i, idx) in res.iter().enumerate() {
                if i != 0 {
                    out.push(' ');
                }
                out.push_str(&idx.to_string());
            }
            println!("{}", out);
            return;
        }
    } else {
        let max = 250_000usize.min(by_abs.len());
        let mut sum: i64 = 0;
        let mut res: Vec<usize> = Vec::new();
        for it in by_abs.iter().take(max) {
            let cur_dist = (sum - target).abs();
            let ns = sum + it.value;
            let new_dist = (ns - target).abs();
            match new_dist.cmp(&cur_dist) {
                Ordering::Less => {
                    sum = ns;
                    res.push(it.index);
                }
                Ordering::Equal => {
                    if (ns <= target && sum > target) || (ns >= target && sum < target) {
                        sum = ns;
                        res.push(it.index);
                    }
                }
                Ordering::Greater => {}
            }
            if sum == target {
                break;
            }
        }
        if sum == target {
            res.sort_unstable();
            println!("YES");
            let mut out = String::new();
            for (i, idx) in res.iter().enumerate() {
                if i != 0 {
                    out.push(' ');
                }
                out.push_str(&idx.to_string());
            }
            println!("{}", out);
            return;
        }
    }

    println!("NO");
}
