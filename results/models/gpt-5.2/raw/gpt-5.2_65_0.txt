// Mission-control scheduler benchmark solution (Node.js; TS-compatible style, no deps)
// Reads one JSON document from stdin and prints {"schedule":[...]} as JSON.

const fs = require("fs");

function toIdString(id) {
  return typeof id === "string" ? id : String(id);
}

// Binary heap (min-heap) with comparator: less(x,y) true iff x has higher priority (comes first)
class Heap {
  constructor(less) {
    this.less = less;
    this.a = [];
  }
  get size() {
    return this.a.length;
  }
  peek() {
    return this.a[0];
  }
  push(v) {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop() {
    const a = this.a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (n > 1) {
      a[0] = last;
      this._down(0);
    }
    return top;
  }
  _down(i) {
    const a = this.a;
    const n = a.length;
    while (true) {
      const l = i * 2 + 1;
      if (l >= n) break;
      const r = l + 1;
      let m = l;
      if (r < n && this.less(a[r], a[l])) m = r;
      if (!this.less(a[m], a[i])) break;
      const tmp = a[i];
      a[i] = a[m];
      a[m] = tmp;
      i = m;
    }
  }
}

function buildTasks(doc) {
  const raw = doc.tasks || [];
  const idToIndex = new Map();
  for (let i = 0; i < raw.length; i++) idToIndex.set(toIdString(raw[i].id), i);

  const tasks = new Array(raw.length);
  for (let i = 0; i < raw.length; i++) {
    const t = raw[i];
    const depsRaw = t.deps || [];
    const depsIdx = [];
    for (let j = 0; j < depsRaw.length; j++) {
      const k = idToIndex.get(toIdString(depsRaw[j]));
      if (k !== undefined) depsIdx.push(k);
    }
    const deadline = Number.isFinite(t.deadline) ? t.deadline : Infinity;
    const weight = Number.isFinite(t.weight) ? Math.max(0, t.weight) : 1;
    const duration = (t.duration | 0) < 0 ? 0 : (t.duration | 0);
    tasks[i] = { id: toIdString(t.id), duration, deps: depsIdx, deadline, weight };
  }
  return { tasks };
}

function topoOrder(tasks) {
  const n = tasks.length;
  const indeg = new Array(n).fill(0);
  const out = Array.from({ length: n }, () => []);
  for (let v = 0; v < n; v++) {
    for (const u of tasks[v].deps) {
      indeg[v]++;
      out[u].push(v);
    }
  }

  const pq = new Heap((a, b) => {
    const ta = tasks[a];
    const tb = tasks[b];
    if (ta.deadline !== tb.deadline) return ta.deadline < tb.deadline;
    const ra = ta.weight / Math.max(1, ta.duration);
    const rb = tb.weight / Math.max(1, tb.duration);
    if (ra !== rb) return ra > rb;
    if (ta.weight !== tb.weight) return ta.weight > tb.weight;
    if (ta.duration !== tb.duration) return ta.duration > tb.duration;
    return ta.id < tb.id;
  });

  for (let i = 0; i < n; i++) if (indeg[i] === 0) pq.push(i);

  const order = [];
  while (pq.size) {
    const u = pq.pop();
    order.push(u);
    for (const v of out[u]) {
      indeg[v]--;
      if (indeg[v] === 0) pq.push(v);
    }
  }

  if (order.length !== n) {
    const seen = new Array(n).fill(false);
    for (const x of order) seen[x] = true;
    for (let i = 0; i < n; i++) if (!seen[i]) order.push(i);
  }
  return order;
}

function schedule(tasks, machineCount) {
  const n = tasks.length;
  const m = Math.max(1, machineCount | 0);

  const out = Array.from({ length: n }, () => []);
  const indeg = new Array(n).fill(0);
  for (let v = 0; v < n; v++) {
    for (const u of tasks[v].deps) {
      out[u].push(v);
      indeg[v]++;
    }
  }

  const remainingDeps = indeg.slice();
  const readyAt = new Array(n).fill(0);
  const finish = new Array(n).fill(0);
  const scheduled = new Array(n).fill(false);

  const machHeap = new Heap((a, b) => (a.t !== b.t ? a.t < b.t : a.i < b.i));
  for (let i = 0; i < m; i++) machHeap.push({ t: 0, i });

  const available = new Heap((a, b) => {
    const ta = tasks[a];
    const tb = tasks[b];
    if (ta.deadline !== tb.deadline) return ta.deadline < tb.deadline;
    const ra = ta.weight / Math.max(1, ta.duration);
    const rb = tb.weight / Math.max(1, tb.duration);
    if (ra !== rb) return ra > rb;
    if (ta.weight !== tb.weight) return ta.weight > tb.weight;
    if (ta.duration !== tb.duration) return ta.duration > tb.duration;
    return ta.id < tb.id;
  });

  for (let v = 0; v < n; v++) if (remainingDeps[v] === 0) available.push(v);

  const order = topoOrder(tasks);
  const scheduleItems = [];
  let count = 0;

  while (count < n) {
    if (available.size === 0) {
      // Break cycles / missing deps: schedule remaining in topo fallback order.
      for (const v of order) {
        if (scheduled[v]) continue;
        const mach = machHeap.pop();
        const start = mach.t;
        const end = start + tasks[v].duration;
        scheduleItems.push({ id: tasks[v].id, start, machine: mach.i });
        scheduled[v] = true;
        finish[v] = end;
        machHeap.push({ t: end, i: mach.i });
      }
      break;
    }

    const mach = machHeap.pop();
    const now = mach.t;

    const top = available.peek();
    if (top === undefined) break;
    if (readyAt[top] > now) {
      machHeap.push({ t: readyAt[top], i: mach.i });
      continue;
    }

    const v = available.pop();
    if (scheduled[v]) continue;
    if (readyAt[v] > now) {
      available.push(v);
      machHeap.push({ t: readyAt[v], i: mach.i });
      continue;
    }

    const start = now;
    const end = start + tasks[v].duration;
    scheduleItems.push({ id: tasks[v].id, start, machine: mach.i });
    scheduled[v] = true;
    finish[v] = end;
    machHeap.push({ t: end, i: mach.i });
    count++;

    for (const w of out[v]) {
      if (scheduled[w]) continue;
      if (finish[v] > readyAt[w]) readyAt[w] = finish[v];
      remainingDeps[w]--;
      if (remainingDeps[w] === 0) available.push(w);
    }
  }

  scheduleItems.sort((a, b) => (a.start !== b.start ? a.start - b.start : a.machine - b.machine));
  return scheduleItems;
}

function main() {
  const input = fs.readFileSync(0, "utf8");
  const doc = JSON.parse(input);
  const { tasks } = buildTasks(doc);
  const sched = schedule(tasks, doc.machineCount);
  process.stdout.write(JSON.stringify({ schedule: sched }));
}

main();
