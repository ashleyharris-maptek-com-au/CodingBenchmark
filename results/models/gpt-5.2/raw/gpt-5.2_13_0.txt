using std::size_t;

static inline uint64_t splitmix64(uint64_t x) {
  x += 0x9e3779b97f4a7c15ULL;
  x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
  x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
  return x ^ (x >> 31);
}

struct SAResult {
  std::vector<int> sa;
  std::vector<int> lcp; // lcp[i] = LCP(sa[i], sa[i-1]), lcp[0]=0
  std::vector<int> owner; // which original string each suffix belongs to, -1 for separators
  std::string text;
};

// Build suffix array using prefix-doubling (O(n log n))
static std::vector<int> build_sa(const std::string& s) {
  const int n = (int)s.size();
  std::vector<int> sa(n), rnk(n), tmp(n);
  for (int i = 0; i < n; ++i) {
    sa[i] = i;
    rnk[i] = (unsigned char)s[i];
  }
  for (int k = 1;; k <<= 1) {
    auto cmp = [&](int i, int j) {
      if (rnk[i] != rnk[j]) return rnk[i] < rnk[j];
      int ri = (i + k < n) ? rnk[i + k] : -1;
      int rj = (j + k < n) ? rnk[j + k] : -1;
      return ri < rj;
    };
    std::sort(sa.begin(), sa.end(), cmp);
    tmp[sa[0]] = 0;
    for (int i = 1; i < n; ++i) tmp[sa[i]] = tmp[sa[i - 1]] + (cmp(sa[i - 1], sa[i]) ? 1 : 0);
    for (int i = 0; i < n; ++i) rnk[i] = tmp[i];
    if (rnk[sa[n - 1]] == n - 1) break;
  }
  return sa;
}

// Kasai LCP (O(n))
static std::vector<int> build_lcp(const std::string& s, const std::vector<int>& sa) {
  const int n = (int)s.size();
  std::vector<int> rank(n), lcp(n);
  for (int i = 0; i < n; ++i) rank[sa[i]] = i;
  int h = 0;
  for (int i = 0; i < n; ++i) {
    int r = rank[i];
    if (r == 0) {
      lcp[r] = 0;
      continue;
    }
    int j = sa[r - 1];
    while (i + h < n && j + h < n && s[i + h] == s[j + h]) ++h;
    lcp[r] = h;
    if (h > 0) --h;
  }
  return lcp;
}

static SAResult build_sa_multi(const std::vector<std::string>& strings) {
  SAResult res;
  // concatenate with unique separators not present in input (use bytes 1..N, shift if conflicts)
  // For safety with arbitrary bytes in input (but lines likely ASCII), use separators in range 1..31 and then 127..255.
  // Here strings are read as text lines; still, we avoid collisions by scanning.
  std::array<bool, 256> used{};
  used.fill(false);
  for (const auto& t : strings) {
    for (unsigned char c : t) used[c] = true;
  }
  std::vector<unsigned char> seps;
  seps.reserve(strings.size());
  for (int c = 1; c < 256 && (int)seps.size() < (int)strings.size(); ++c) {
    if (!used[(unsigned char)c]) seps.push_back((unsigned char)c);
  }
  // Fallback: if all bytes used (extremely unlikely for line-based), still proceed by using 0 which can't appear in std::string from getline? It can, but rare.
  while ((int)seps.size() < (int)strings.size()) seps.push_back(0);

  res.text.clear();
  res.owner.clear();
  res.text.reserve(std::accumulate(strings.begin(), strings.end(), size_t(0),
                                   [](size_t a, const std::string& b) { return a + b.size() + 1; }));
  for (int i = 0; i < (int)strings.size(); ++i) {
    for (char ch : strings[i]) {
      res.text.push_back(ch);
      res.owner.push_back(i);
    }
    res.text.push_back((char)seps[i]);
    res.owner.push_back(-1);
  }

  res.sa = build_sa(res.text);
  res.lcp = build_lcp(res.text, res.sa);
  // Map suffix owners by suffix start position
  // owner[pos] already stores owner for that position; for suffix starting at pos, owner is owner[pos].
  // Ensure separators are -1.
  return res;
}

static std::string lcs_via_suffix_array_all(const std::vector<std::string>& strings) {
  const int nstr = (int)strings.size();
  if (nstr == 0) return std::string();
  if (nstr == 1) return strings[0];

  SAResult sares = build_sa_multi(strings);
  const int n = (int)sares.text.size();
  const auto& sa = sares.sa;
  const auto& lcp = sares.lcp;
  const auto& ownerAtPos = sares.owner;

  // Sliding window over SA that includes all owners; window score is minimum LCP in window (between adjacent suffixes).
  std::vector<int> cnt(nstr, 0);
  int have = 0;
  int bestLen = 0;
  int bestPos = -1; // position in concatenated text

  // Maintain deque of indices for minimum lcp in window [L+1..R] (since lcp[i] compares sa[i] and sa[i-1])
  std::vector<int> dq;
  dq.reserve(n);

  auto push_lcp = [&](int idx) {
    while (!dq.empty() && lcp[dq.back()] >= lcp[idx]) dq.pop_back();
    dq.push_back(idx);
  };
  auto pop_lcp_before = [&](int idx) {
    while (!dq.empty() && dq.front() < idx) dq.erase(dq.begin());
  };

  int L = 0;
  for (int R = 0; R < n; ++R) {
    int ownR = ownerAtPos[sa[R]];
    if (ownR >= 0) {
      if (cnt[ownR]++ == 0) ++have;
    }
    if (R > 0) push_lcp(R);

    while (have == nstr && L <= R) {
      // window [L..R] valid, min LCP is min over lcp[L+1..R]
      int minL = (L == R) ? 0 : lcp[dq.front()];
      if (minL > bestLen) {
        bestLen = minL;
        bestPos = sa[R]; // any suffix in window works; choose sa[R], substring length bestLen
      }
      // move L forward
      int ownL = ownerAtPos[sa[L]];
      if (L + 1 <= R) {
        pop_lcp_before(L + 2); // because lcp index corresponds to right endpoint; removing L means lcp[L+1] goes away
      }
      if (ownL >= 0) {
        if (--cnt[ownL] == 0) --have;
      }
      ++L;
    }
  }

  if (bestLen <= 0) return std::string();
  if (bestPos < 0) return std::string();
  return sares.text.substr((size_t)bestPos, (size_t)bestLen);
}

// Rolling hash for binary-search L on shortest string, verifying across all strings.
// Uses double 64-bit (mod 2^64) with independent seeds; collision probability negligible.
struct RH {
  std::vector<uint64_t> p1, p2, h1, h2;
  uint64_t b1, b2;
  RH() = default;
  RH(std::string_view s, uint64_t base1, uint64_t base2) : b1(base1), b2(base2) {
    size_t n = s.size();
    p1.resize(n + 1);
    p2.resize(n + 1);
    h1.resize(n + 1);
    h2.resize(n + 1);
    p1[0] = 1;
    p2[0] = 1;
    h1[0] = 0;
    h2[0] = 0;
    for (size_t i = 0; i < n; ++i) {
      uint64_t x = (uint64_t)(unsigned char)s[i] + 1;
      p1[i + 1] = p1[i] * b1;
      p2[i + 1] = p2[i] * b2;
      h1[i + 1] = h1[i] * b1 + x;
      h2[i + 1] = h2[i] * b2 + x;
    }
  }
  inline std::pair<uint64_t, uint64_t> get(size_t l, size_t r) const {
    uint64_t x1 = h1[r] - h1[l] * p1[r - l];
    uint64_t x2 = h2[r] - h2[l] * p2[r - l];
    return {x1, x2};
  }
};

struct PairHash {
  size_t operator()(const std::pair<uint64_t, uint64_t>& p) const noexcept {
    uint64_t x = p.first ^ splitmix64(p.second);
    return (size_t)splitmix64(x);
  }
};

// Intersect sets of hashes across strings; also keep one witness position in the chosen base string.
static bool exists_common_len(
    const std::vector<std::string_view>& sv,
    int baseIdx,
    size_t L,
    std::pair<uint64_t, uint64_t>& outHash,
    size_t& outPos,
    const std::vector<RH>& rhs) {
  if (L == 0) {
    outHash = {0, 0};
    outPos = 0;
    return true;
  }
  const size_t baseN = sv[baseIdx].size();
  if (baseN < L) return false;

  // Start with all hashes from base string.
  std::unordered_map<std::pair<uint64_t, uint64_t>, size_t, PairHash> cur;
  cur.reserve(baseN - L + 1);
  for (size_t i = 0; i + L <= baseN; ++i) {
    auto hv = rhs[baseIdx].get(i, i + L);
    // Keep earliest witness
    if (cur.find(hv) == cur.end()) cur.emplace(hv, i);
  }
  if (cur.empty()) return false;

  // For each other string, keep only those hashes that appear there.
  std::unordered_map<std::pair<uint64_t, uint64_t>, size_t, PairHash> nxt;
  for (int si = 0; si < (int)sv.size(); ++si) {
    if (si == baseIdx) continue;
    const size_t n = sv[si].size();
    if (n < L) return false;
    nxt.clear();
    nxt.reserve(std::min(cur.size(), n - L + 1));
    for (size_t i = 0; i + L <= n; ++i) {
      auto hv = rhs[si].get(i, i + L);
      auto it = cur.find(hv);
      if (it != cur.end()) {
        // Preserve base witness position
        nxt.emplace(hv, it->second);
      }
    }
    cur.swap(nxt);
    if (cur.empty()) return false;
  }
  // Return any remaining candidate
  auto it = cur.begin();
  outHash = it->first;
  outPos = it->second;
  return true;
}

static std::string lcs_via_hash_bs(const std::vector<std::string>& strings) {
  const int n = (int)strings.size();
  if (n == 0) return std::string();
  if (n == 1) return strings[0];

  std::vector<std::string_view> sv;
  sv.reserve(n);
  for (const auto& s : strings) sv.emplace_back(s);

  // Choose shortest string as base to minimize candidates.
  int baseIdx = 0;
  for (int i = 1; i < n; ++i)
    if (sv[i].size() < sv[baseIdx].size()) baseIdx = i;

  // Deterministic seeds
  uint64_t seed = 0x1234567890abcdefULL;
  uint64_t b1 = (splitmix64(seed) | 1ULL);
  uint64_t b2 = (splitmix64(seed ^ 0x9e3779b97f4a7c15ULL) | 1ULL);

  std::vector<RH> rhs;
  rhs.reserve(n);
  for (int i = 0; i < n; ++i) rhs.emplace_back(sv[i], b1, b2);

  size_t lo = 0, hi = sv[baseIdx].size();
  size_t bestL = 0;
  size_t bestPos = 0;
  while (lo <= hi) {
    size_t mid = lo + (hi - lo) / 2;
    std::pair<uint64_t, uint64_t> hv;
    size_t pos = 0;
    if (exists_common_len(sv, baseIdx, mid, hv, pos, rhs)) {
      bestL = mid;
      bestPos = pos;
      lo = mid + 1;
    } else {
      if (mid == 0) break;
      hi = mid - 1;
    }
  }

  if (bestL == 0) return std::string();
  // Verify by direct check to eliminate any chance of collision: the witness substring must appear in all strings.
  std::string candidate = strings[baseIdx].substr(bestPos, bestL);
  for (int i = 0; i < n; ++i) {
    if (i == baseIdx) continue;
    if (strings[i].find(candidate) == std::string::npos) {
      // Rare collision; fall back to suffix-array exact method on manageable sizes.
      // If sizes are too large, try to shrink by re-running with slightly smaller length via decremental search.
      for (size_t L = bestL; L > 0; --L) {
        bool ok = false;
        // Scan base for some substring of length L and test via find (expensive, but only if collision occurred).
        for (size_t p = 0; p + L <= strings[baseIdx].size(); ++p) {
          std::string sub = strings[baseIdx].substr(p, L);
          bool all = true;
          for (int j = 0; j < n; ++j) {
            if (j == baseIdx) continue;
            if (strings[j].find(sub) == std::string::npos) {
              all = false;
              break;
            }
          }
          if (all) {
            return sub;
          }
        }
        if (ok) break;
      }
      return std::string();
    }
  }
  return candidate;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int N;
  if (!(std::cin >> N)) return 0;
  std::string dummy;
  std::getline(std::cin, dummy); // consume rest of line

  std::vector<std::string> strings;
  strings.reserve(std::max(0, N));
  for (int i = 0; i < N; ++i) {
    std::string s;
    std::getline(std::cin, s);
    // Keep exact line (may be empty).
    strings.push_back(std::move(s));
  }

  if (N <= 0) {
    std::cout << "\n";
    return 0;
  }

  // Heuristic adaptive strategy:
  // - Small total size: exact generalized suffix array (fast and exact).
  // - Very large: binary search + rolling hash intersection (near-linear, then verify).
  //
  // Note: For "extreme" inputs with 100 strings of 10MB+, building a suffix array over concatenation is infeasible.
  size_t total = 0;
  size_t minLen = std::numeric_limits<size_t>::max();
  for (const auto& s : strings) {
    total += s.size();
    minLen = std::min(minLen, s.size());
  }

  std::string ans;
  const bool smallN = (N <= 10);
  const bool moderateTotal = (total <= 5'000'000ULL); // 5MB
  const bool moderateMin = (minLen <= 200'000ULL);    // cap SA feasibility a bit

  if (smallN && moderateTotal && moderateMin) {
    ans = lcs_via_suffix_array_all(strings); // exact
  } else {
    ans = lcs_via_hash_bs(strings); // scalable; verified on output
    // If verification failed inside (collision fallback), ans may be empty; that's acceptable.
  }

  std::cout << ans << "\n";
  return 0;
}
