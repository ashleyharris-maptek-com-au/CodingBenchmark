#!/usr/bin/env python3
import sys
from collections import deque


# Practical 3-SAT solver for huge guaranteed-satisfiable instances.
# Uses a randomized local-search algorithm (Papadimitriou/Schöning-style):
# repeatedly flip a variable from an unsatisfied clause.
#
# Notes:
# - Input sizes can be massive; store clauses in compact arrays and stream read fast.
# - Guaranteed satisfiable => local search typically finds a solution quickly on such benchmarks.


def _fast_ints_from_stdin():
    data = sys.stdin.buffer.read()
    n = len(data)
    i = 0
    while i < n:
        b = data[i]
        while b <= 32:
            i += 1
            if i >= n:
                return
            b = data[i]
        sign = 1
        if b == 45:  # '-'
            sign = -1
            i += 1
        x = 0
        while i < n:
            b = data[i]
            if b <= 32:
                break
            x = x * 10 + (b - 48)
            i += 1
        yield sign * x


class XorShift64:
    __slots__ = ("s",)

    def __init__(self, seed):
        if seed == 0:
            seed = 0x9E3779B97F4A7C15
        self.s = seed & ((1 << 64) - 1)

    def next_u64(self):
        x = self.s
        x ^= (x << 13) & ((1 << 64) - 1)
        x ^= (x >> 7) & ((1 << 64) - 1)
        x ^= (x << 17) & ((1 << 64) - 1)
        self.s = x
        return x

    def randbelow(self, m):
        return self.next_u64() % m


def solve(num_vars, num_clauses, a, b, c):
    rng = XorShift64(0xD1B54A32D192ED03 ^ (num_vars << 1) ^ num_clauses)

    # Initial random assignment.
    # Use bytearray for compactness and fast flipping.
    assign = bytearray(num_vars + 1)
    for i in range(1, num_vars + 1):
        assign[i] = rng.next_u64() & 1

    # Helper to evaluate a literal under current assignment.
    def lit_true(lit):
        v = lit if lit > 0 else -lit
        val = assign[v]
        return val == 1 if lit > 0 else val == 0

    # Build initial list of unsatisfied clauses indices.
    unsat = []
    for i in range(num_clauses):
        if not (lit_true(a[i]) or lit_true(b[i]) or lit_true(c[i])):
            unsat.append(i)

    if not unsat:
        return assign

    # Local search parameters:
    # - restart when stuck
    # - flips per restart scales with vars
    # Keep conservative to fit 5 minutes on big data; guaranteed SAT helps.
    max_restarts = 50
    flips_per_restart = max(10000, min(5_000_000, 20 * num_vars))

    # To avoid O(M) rescans each flip, we do occasional full recomputes.
    # Between recomputes, track unsat via a small sample queue of candidates.
    # This keeps memory low and is robust enough for typical satisfiable benchmarks.
    recompute_every = 20000

    for _ in range(max_restarts):
        # Full recompute of unsatisfied clauses.
        unsat = []
        for i in range(num_clauses):
            if not (lit_true(a[i]) or lit_true(b[i]) or lit_true(c[i])):
                unsat.append(i)
        if not unsat:
            return assign

        q = deque(unsat[:1024])  # worklist of currently known unsatisfied clauses

        for t in range(flips_per_restart):
            if not q:
                # Refill by sampling; if still empty, do a full recompute.
                if t % recompute_every == 0:
                    unsat = []
                    for i in range(num_clauses):
                        if not (lit_true(a[i]) or lit_true(b[i]) or lit_true(c[i])):
                            unsat.append(i)
                    if not unsat:
                        return assign
                    q.extend(unsat[:1024])
                else:
                    # random clause probe
                    idx = rng.randbelow(num_clauses)
                    if not (lit_true(a[idx]) or lit_true(b[idx]) or lit_true(c[idx])):
                        q.append(idx)
                    continue

            ci = q.popleft()
            if lit_true(a[ci]) or lit_true(b[ci]) or lit_true(c[ci]):
                continue  # no longer unsatisfied

            # Pick one variable from the unsatisfied clause and flip it.
            r = rng.randbelow(3)
            lit = a[ci] if r == 0 else (b[ci] if r == 1 else c[ci])
            v = lit if lit > 0 else -lit
            assign[v] ^= 1

            # After a flip, push a few random probes to keep queue fresh.
            # Also re-check the same clause index quickly.
            q.append(ci)
            for _k in range(2):
                idx = rng.randbelow(num_clauses)
                if not (lit_true(a[idx]) or lit_true(b[idx]) or lit_true(c[idx])):
                    q.append(idx)

            if (t + 1) % recompute_every == 0:
                unsat = []
                for i in range(num_clauses):
                    if not (lit_true(a[i]) or lit_true(b[i]) or lit_true(c[i])):
                        unsat.append(i)
                if not unsat:
                    return assign
                q.clear()
                q.extend(unsat[:1024])

        # Restart with a new random assignment.
        for i in range(1, num_vars + 1):
            assign[i] = rng.next_u64() & 1

    # Given guarantee of satisfiable instances, we should virtually always succeed.
    # As a fallback, output current assignment (may fail if extremely adversarial).
    return assign


def main():
    it = _fast_ints_from_stdin()
    try:
        n = next(it)
        m = next(it)
    except StopIteration:
        return

    # Store clauses in Python lists of ints.
    # Memory note: large instances can exceed Python memory; this solver targets
    # realistic large-but-fit-in-RAM cases used in benchmarks.
    a = [0] * m
    b = [0] * m
    c = [0] * m
    for i in range(m):
        try:
            a[i] = next(it)
            b[i] = next(it)
            c[i] = next(it)
        except StopIteration:
            break

    assign = solve(n, m, a, b, c)

    out = sys.stdout.write
    out("SAT\n")
    # Print 1..N as 0/1
    # Use join for speed; chunk to avoid huge temporary for extreme N.
    CHUNK = 1_000_000
    parts = []
    start = 1
    while start <= n:
        end = min(n + 1, start + CHUNK)
        parts.append(" ".join("1" if assign[i] else "0" for i in range(start, end)))
        start = end
    out("\n".join([parts[0]] + parts[1:]))
    out("\n")


if __name__ == "__main__":
    main()
