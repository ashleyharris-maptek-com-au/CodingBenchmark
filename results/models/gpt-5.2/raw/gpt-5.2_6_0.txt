using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm2() => Dot(this);
        public double Norm() => Math.Sqrt(Norm2());
    }

    private struct Orbit
    {
        public Vec3 R;
        public Vec3 V;
        public Orbit(Vec3 r, Vec3 v) { R = r; V = v; }
    }

    // For TSP-like routing, we need a fast, robust edge "cost" between two orbits.
    // We approximate rendezvous delta-V with a blend of:
    // - instantaneous velocity mismatch at departure/arrival (|v1-v2|)
    // - plane mismatch proxy via angular momentum direction difference
    // - radial mismatch proxy via |r1-r2| scaled
    // This is NOT full Lambert + phasing, but is stable and cheap for N up to 100.
    private static double EdgeCost(in Orbit a, in Orbit b, double mu)
    {
        Vec3 dv = a.V - b.V;
        double vel = dv.Norm();

        Vec3 ha = Cross(a.R, a.V);
        Vec3 hb = Cross(b.R, b.V);
        double haN = ha.Norm();
        double hbN = hb.Norm();
        double plane = 0.0;
        if (haN > 1e-9 && hbN > 1e-9)
        {
            double c = Clamp(ha.Dot(hb) / (haN * hbN), -1.0, 1.0);
            plane = Math.Acos(c); // radians
        }

        double dr = (a.R - b.R).Norm(); // km

        // Typical LEO speeds ~7.5 km/s; keep terms comparable.
        // Tune weights for a smooth metric across scales.
        double cost = vel
                      + 2.0 * plane // rad ~ small; 2 km/s per rad proxy
                      + 0.0015 * dr; // 1000 km -> 1.5 km/s proxy

        // Lightly bias by orbital energy difference (semi-major axis proxy).
        double ea = a.V.Norm2() / 2.0 - mu / Math.Max(1e-9, a.R.Norm());
        double eb = b.V.Norm2() / 2.0 - mu / Math.Max(1e-9, b.R.Norm());
        cost += 0.15 * Math.Abs(ea - eb); // km^2/s^2 proxy

        return cost;
    }

    private static Vec3 Cross(Vec3 a, Vec3 b)
        => new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);

    private static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    private static void ReadAll(out int n, out double mu, out Orbit start, out Orbit[] stations)
    {
        var tokens = new List<string>(1024);
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            if (line.Length == 0) continue;
            var parts = line.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            for (int i = 0; i < parts.Length; i++) tokens.Add(parts[i]);
        }
        int idx = 0;
        n = int.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        mu = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        start = new Orbit(
            new Vec3(
                double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                double.Parse(tokens[idx++], CultureInfo.InvariantCulture)),
            new Vec3(
                double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                double.Parse(tokens[idx++], CultureInfo.InvariantCulture))
        );
        stations = new Orbit[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = new Orbit(
                new Vec3(
                    double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                    double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                    double.Parse(tokens[idx++], CultureInfo.InvariantCulture)),
                new Vec3(
                    double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                    double.Parse(tokens[idx++], CultureInfo.InvariantCulture),
                    double.Parse(tokens[idx++], CultureInfo.InvariantCulture))
            );
        }
    }

    // Build full directed cost matrix from start + stations.
    // Node 0 is start, nodes 1..N are stations (station i => node i+1).
    private static double[][] BuildCostMatrix(in Orbit start, Orbit[] stations, double mu)
    {
        int n = stations.Length;
        int m = n + 1;
        var all = new Orbit[m];
        all[0] = start;
        Array.Copy(stations, 0, all, 1, n);

        var cost = new double[m][];
        for (int i = 0; i < m; i++)
        {
            cost[i] = new double[m];
        }
        for (int i = 0; i < m; i++)
        {
            cost[i][i] = 0;
            for (int j = 0; j < m; j++)
            {
                if (i == j) continue;
                cost[i][j] = EdgeCost(all[i], all[j], mu);
            }
        }
        return cost;
    }

    // Nearest neighbor seed from start (node 0).
    private static int[] NearestNeighbor(double[][] cost)
    {
        int m = cost.Length;
        int n = m - 1;
        var used = new bool[m];
        used[0] = true;
        var tour = new int[m]; // includes start at index 0
        tour[0] = 0;
        int cur = 0;
        for (int k = 1; k < m; k++)
        {
            int best = -1;
            double bestC = double.PositiveInfinity;
            for (int j = 1; j < m; j++)
            {
                if (used[j]) continue;
                double c = cost[cur][j];
                if (c < bestC)
                {
                    bestC = c;
                    best = j;
                }
            }
            used[best] = true;
            tour[k] = best;
            cur = best;
        }
        // return station indices order (0..n-1), excluding start node
        var order = new int[n];
        for (int i = 0; i < n; i++) order[i] = tour[i + 1] - 1;
        return order;
    }

    private static double TourCostFromStart(double[][] cost, int[] order)
    {
        // cost start->first + sum station->station (no return)
        int n = order.Length;
        if (n == 0) return 0;
        int prevNode = 0;
        double sum = 0;
        for (int i = 0; i < n; i++)
        {
            int node = order[i] + 1;
            sum += cost[prevNode][node];
            prevNode = node;
        }
        return sum;
    }

    private static void TwoOptImprove(double[][] cost, int[] order, int timeBudgetMs)
    {
        int n = order.Length;
        if (n < 4) return;
        long deadline = Environment.TickCount64 + timeBudgetMs;
        bool improved = true;
        while (improved && Environment.TickCount64 < deadline)
        {
            improved = false;
            // 2-opt over path (open tour) from start node (fixed) to end
            // We reverse segment [i..k] (0-based on stations), affecting edges:
            // (prev->i) + (k->next) vs (prev->k) + (i->next)
            for (int i = 0; i < n - 2 && Environment.TickCount64 < deadline; i++)
            {
                int prevNode = (i == 0) ? 0 : (order[i - 1] + 1);
                int iNode = order[i] + 1;
                for (int k = i + 1; k < n - 1 && Environment.TickCount64 < deadline; k++)
                {
                    int kNode = order[k] + 1;
                    int nextNode = order[k + 1] + 1;

                    double before = cost[prevNode][iNode] + cost[kNode][nextNode];
                    double after = cost[prevNode][kNode] + cost[iNode][nextNode];
                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(order, i, k - i + 1);
                        improved = true;
                        iNode = order[i] + 1; // refresh (micro-opt)
                    }
                }
            }
        }
    }

    // For small N, do exact DP (Held-Karp) for path starting at start and visiting all stations once (no return).
    private static int[] SolveExactDP(double[][] cost)
    {
        int m = cost.Length;
        int n = m - 1;
        if (n == 0) return Array.Empty<int>();

        int full = 1 << n;
        var dp = new double[full, n];
        var parent = new int[full, n];
        double INF = 1e100;
        for (int mask = 0; mask < full; mask++)
            for (int j = 0; j < n; j++)
            {
                dp[mask, j] = INF;
                parent[mask, j] = -1;
            }

        for (int j = 0; j < n; j++)
        {
            int mask = 1 << j;
            dp[mask, j] = cost[0][j + 1];
        }

        for (int mask = 1; mask < full; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double cur = dp[mask, last];
                if (cur >= INF / 2) continue;
                int prevMask = mask ^ (1 << last);
                if (prevMask == 0) continue;
                // relax from prev -> last
                for (int prev = 0; prev < n; prev++)
                {
                    if ((prevMask & (1 << prev)) == 0) continue;
                    double cand = dp[prevMask, prev] + cost[prev + 1][last + 1];
                    if (cand < cur)
                    {
                        cur = cand;
                        dp[mask, last] = cand;
                        parent[mask, last] = prev;
                    }
                }
            }
        }

        // best end
        int bestLast = 0;
        double best = INF;
        int allMask = full - 1;
        for (int last = 0; last < n; last++)
        {
            double v = dp[allMask, last];
            if (v < best)
            {
                best = v;
                bestLast = last;
            }
        }

        var order = new int[n];
        int mask2 = allMask;
        int last2 = bestLast;
        for (int i = n - 1; i >= 0; i--)
        {
            order[i] = last2;
            int p = parent[mask2, last2];
            mask2 ^= 1 << last2;
            last2 = p;
            if (last2 < 0 && i > 0) break;
        }
        return order; // already station indices 0..n-1
    }

    // Multi-start NN + 2-opt for larger N; also include greedy insertion variant for diversification.
    private static int[] SolveHeuristic(double[][] cost, int timeBudgetMs)
    {
        int n = cost.Length - 1;
        if (n == 0) return Array.Empty<int>();

        long deadline = Environment.TickCount64 + timeBudgetMs;
        int[] bestOrder = NearestNeighbor(cost);
        double best = TourCostFromStart(cost, bestOrder);

        // Diversify by choosing different first station candidates from start.
        // Try top-K nearest as first, then NN completion, then 2-opt.
        int K = Math.Min(8, n);
        int[] firstCandidates = TopKFromStart(cost, K);

        foreach (var first in firstCandidates)
        {
            if (Environment.TickCount64 >= deadline) break;
            int[] order = BuildNNWithFixedFirst(cost, first);
            TwoOptImprove(cost, order, Math.Max(10, (int)(timeBudgetMs * 0.35)));
            double c = TourCostFromStart(cost, order);
            if (c < best)
            {
                best = c;
                bestOrder = order;
            }
        }

        // Greedy insertion seed (cheaper than full k-regret, good on 50-100).
        if (Environment.TickCount64 < deadline && n >= 8)
        {
            int[] ins = GreedyInsertion(cost, deadline);
            TwoOptImprove(cost, ins, Math.Max(10, (int)(timeBudgetMs * 0.25)));
            double c = TourCostFromStart(cost, ins);
            if (c < best)
            {
                best = c;
                bestOrder = ins;
            }
        }

        // Final quick 2-opt polish on best within remaining time.
        int remaining = (int)Math.Max(0, deadline - Environment.TickCount64);
        if (remaining > 0) TwoOptImprove(cost, bestOrder, remaining);
        return bestOrder;
    }

    private static int[] TopKFromStart(double[][] cost, int k)
    {
        int n = cost.Length - 1;
        // Keep a simple sorted list of (station, cost) of size <= k.
        int[] stations = new int[k];
        double[] costs = new double[k];
        int size = 0;

        for (int s = 0; s < n; s++)
        {
            double c = cost[0][s + 1];
            int pos = size;
            if (size < k)
            {
                stations[size] = s;
                costs[size] = c;
                size++;
            }
            else if (c >= costs[size - 1])
            {
                continue;
            }
            else
            {
                pos = size - 1;
                stations[pos] = s;
                costs[pos] = c;
            }

            // insertion sort step to keep [0..size) sorted by costs asc
            for (int i = Math.Min(size - 1, k - 1); i > 0; i--)
            {
                if (costs[i] >= costs[i - 1]) break;
                double tc = costs[i]; costs[i] = costs[i - 1]; costs[i - 1] = tc;
                int ts = stations[i]; stations[i] = stations[i - 1]; stations[i - 1] = ts;
            }
        }

        if (size == k) return stations;
        var res = new int[size];
        Array.Copy(stations, res, size);
        return res;
    }

    private static int[] BuildNNWithFixedFirst(double[][] cost, int firstStation)
    {
        int n = cost.Length - 1;
        var used = new bool[n];
        var order = new int[n];
        order[0] = firstStation;
        used[firstStation] = true;
        int curNode = firstStation + 1;
        for (int k = 1; k < n; k++)
        {
            int best = -1;
            double bestC = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[curNode][j + 1];
                if (c < bestC)
                {
                    bestC = c;
                    best = j;
                }
            }
            used[best] = true;
            order[k] = best;
            curNode = best + 1;
        }
        return order;
    }

    private static int[] GreedyInsertion(double[][] cost, long deadline)
    {
        int n = cost.Length - 1;
        // Start with the closest station to start, then the closest to that.
        int first = 0;
        double bestC = double.PositiveInfinity;
        for (int i = 0; i < n; i++)
        {
            double c = cost[0][i + 1];
            if (c < bestC) { bestC = c; first = i; }
        }

        int second = -1;
        bestC = double.PositiveInfinity;
        for (int i = 0; i < n; i++)
        {
            if (i == first) continue;
            double c = cost[first + 1][i + 1];
            if (c < bestC) { bestC = c; second = i; }
        }
        if (second < 0) return new[] { first };

        var inPath = new bool[n];
        inPath[first] = true;
        inPath[second] = true;
        var path = new List<int>(n) { first, second };

        // Insert remaining stations where they cause least additional cost.
        while (path.Count < n && Environment.TickCount64 < deadline)
        {
            int bestStation = -1;
            int bestPos = -1;
            double bestIncrease = double.PositiveInfinity;

            for (int s = 0; s < n; s++)
            {
                if (inPath[s]) continue;
                // positions between nodes: start->path[0], between path[i-1]->path[i], and path[last] end (no next)
                // For open path, insertion affects one edge (a->b) replaced by (a->s)+(s->b), with a possibly start.
                // Also allow insertion at end: increase = cost[last][s]
                // and at beginning: increase = cost[start][s] + cost[s][first] - cost[start][first]
                // Evaluate all.
                int firstNode = path[0] + 1;
                double incBegin = cost[0][s + 1] + cost[s + 1][firstNode] - cost[0][firstNode];
                if (incBegin < bestIncrease)
                {
                    bestIncrease = incBegin;
                    bestStation = s;
                    bestPos = 0;
                }

                for (int i = 1; i < path.Count; i++)
                {
                    int a = path[i - 1] + 1;
                    int b = path[i] + 1;
                    double inc = cost[a][s + 1] + cost[s + 1][b] - cost[a][b];
                    if (inc < bestIncrease)
                    {
                        bestIncrease = inc;
                        bestStation = s;
                        bestPos = i;
                    }
                }

                int lastNode = path[path.Count - 1] + 1;
                double incEnd = cost[lastNode][s + 1];
                if (incEnd < bestIncrease)
                {
                    bestIncrease = incEnd;
                    bestStation = s;
                    bestPos = path.Count;
                }
            }

            if (bestStation < 0) break;
            inPath[bestStation] = true;
            path.Insert(bestPos, bestStation);
        }

        return path.ToArray();
    }

    public static void Main()
    {
        ReadAll(out int n, out double mu, out Orbit start, out Orbit[] stations);
        var cost = BuildCostMatrix(start, stations, mu);

        int[] order;
        if (n <= 12)
        {
            // Exact for small cases.
            order = SolveExactDP(cost);
        }
        else
        {
            // Heuristic with time budget scaling; keep under 5 minutes worst-case.
            // Use ~2 seconds for 13-20, ~10 seconds for 21-50, ~25 seconds for 51-100.
            int budgetMs = n <= 20 ? 2000 : (n <= 50 ? 10000 : 25000);
            order = SolveHeuristic(cost, budgetMs);
        }

        // Output station indices.
        // If any missing due to unexpected parsing issues, append remaining deterministically.
        if (order.Length != n)
        {
            var seen = new bool[n];
            for (int i = 0; i < order.Length; i++) if ((uint)order[i] < (uint)n) seen[order[i]] = true;
            var list = new List<int>(n);
            for (int i = 0; i < order.Length; i++) if ((uint)order[i] < (uint)n) list.Add(order[i]);
            for (int i = 0; i < n; i++) if (!seen[i]) list.Add(i);
            order = list.ToArray();
        }

        var sw = new System.Text.StringBuilder();
        for (int i = 0; i < order.Length; i++)
        {
            if (i != 0) sw.Append(' ');
            sw.Append(order[i]);
        }
        Console.WriteLine(sw.ToString());
    }
}
