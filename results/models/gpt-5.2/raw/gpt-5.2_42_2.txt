; SPIR-V
; Version: 1.0
; Bound: 79
; Schema: 0
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpSource GLSL 450

; ---- Decorations ----
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; ---- Types ----
%void = OpTypeVoid
%fn = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0
%float_r = OpConstant %float 0.04
%float_edge = OpConstant %float 0.85

%float_cos0 = OpConstant %float 1
%float_sin0 = OpConstant %float 0
%float_cos60 = OpConstant %float 0.5
%float_sin60 = OpConstant %float 0.8660254
%float_cos120 = OpConstant %float -0.5
%float_sin120 = OpConstant %float 0.8660254
%float_cos180 = OpConstant %float -1
%float_sin180 = OpConstant %float 0
%float_cos240 = OpConstant %float -0.5
%float_sin240 = OpConstant %float -0.8660254
%float_cos300 = OpConstant %float 0.5
%float_sin300 = OpConstant %float -0.8660254

%main = OpFunction %void None %fn
%entry = OpLabel

; load center position and w
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%center_x = OpCompositeExtract %float %in_pos 0
%center_y = OpCompositeExtract %float %in_pos 1
%center_z = OpCompositeExtract %float %in_pos 2
%center_w = OpCompositeExtract %float %in_pos 3

; load input color and compute edge color
%in_col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%in_col = OpLoad %v3float %in_col_ptr
%edge_scale = OpCompositeConstruct %v3float %float_edge %float_edge %float_edge
%edge_col = OpFMul %v3float %in_col %edge_scale

; compute edge positions (xyz)
%rx0 = OpFMul %float %float_r %float_cos0
%ry0 = OpFMul %float %float_r %float_sin0
%e0x = OpFAdd %float %center_x %rx0
%e0y = OpFAdd %float %center_y %ry0

%rx60 = OpFMul %float %float_r %float_cos60
%ry60 = OpFMul %float %float_r %float_sin60
%e60x = OpFAdd %float %center_x %rx60
%e60y = OpFAdd %float %center_y %ry60

%rx120 = OpFMul %float %float_r %float_cos120
%ry120 = OpFMul %float %float_r %float_sin120
%e120x = OpFAdd %float %center_x %rx120
%e120y = OpFAdd %float %center_y %ry120

%rx180 = OpFMul %float %float_r %float_cos180
%ry180 = OpFMul %float %float_r %float_sin180
%e180x = OpFAdd %float %center_x %rx180
%e180y = OpFAdd %float %center_y %ry180

%rx240 = OpFMul %float %float_r %float_cos240
%ry240 = OpFMul %float %float_r %float_sin240
%e240x = OpFAdd %float %center_x %rx240
%e240y = OpFAdd %float %center_y %ry240

%rx300 = OpFMul %float %float_r %float_cos300
%ry300 = OpFMul %float %float_r %float_sin300
%e300x = OpFAdd %float %center_x %rx300
%e300y = OpFAdd %float %center_y %ry300

; reusable ptr to gl_out.gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- tri 0: center, e0, e60
OpStore %out_pos_ptr %in_pos
OpStore %outColor %in_col
OpEmitVertex
%e0pos = OpCompositeConstruct %v4float %e0x %e0y %center_z %center_w
OpStore %out_pos_ptr %e0pos
OpStore %outColor %edge_col
OpEmitVertex
%e60pos = OpCompositeConstruct %v4float %e60x %e60y %center_z %center_w
OpStore %out_pos_ptr %e60pos
OpStore %outColor %edge_col
OpEmitVertex
OpEndPrimitive

; ---- tri 1: center, e60, e120
OpStore %out_pos_ptr %in_pos
OpStore %outColor %in_col
OpEmitVertex
OpStore %out_pos_ptr %e60pos
OpStore %outColor %edge_col
OpEmitVertex
%e120pos = OpCompositeConstruct %v4float %e120x %e120y %center_z %center_w
OpStore %out_pos_ptr %e120pos
OpStore %outColor %edge_col
OpEmitVertex
OpEndPrimitive

; ---- tri 2: center, e120, e180
OpStore %out_pos_ptr %in_pos
OpStore %outColor %in_col
OpEmitVertex
OpStore %out_pos_ptr %e120pos
OpStore %outColor %edge_col
OpEmitVertex
%e180pos = OpCompositeConstruct %v4float %e180x %e180y %center_z %center_w
OpStore %out_pos_ptr %e180pos
OpStore %outColor %edge_col
OpEmitVertex
OpEndPrimitive

; ---- tri 3: center, e180, e240
OpStore %out_pos_ptr %in_pos
OpStore %outColor %in_col
OpEmitVertex
OpStore %out_pos_ptr %e180pos
OpStore %outColor %edge_col
OpEmitVertex
%e240pos = OpCompositeConstruct %v4float %e240x %e240y %center_z %center_w
OpStore %out_pos_ptr %e240pos
OpStore %outColor %edge_col
OpEmitVertex
OpEndPrimitive

; ---- tri 4: center, e240, e300
OpStore %out_pos_ptr %in_pos
OpStore %outColor %in_col
OpEmitVertex
OpStore %out_pos_ptr %e240pos
OpStore %outColor %edge_col
OpEmitVertex
%e300pos = OpCompositeConstruct %v4float %e300x %e300y %center_z %center_w
OpStore %out_pos_ptr %e300pos
OpStore %outColor %edge_col
OpEmitVertex
OpEndPrimitive

; ---- tri 5: center, e300, e0
OpStore %out_pos_ptr %in_pos
OpStore %outColor %in_col
OpEmitVertex
OpStore %out_pos_ptr %e300pos
OpStore %outColor %edge_col
OpEmitVertex
OpStore %out_pos_ptr %e0pos
OpStore %outColor %edge_col
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
