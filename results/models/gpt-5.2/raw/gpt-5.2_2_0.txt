using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

// Chinese Postman Problem (Route Inspection) solver that scales.
// Strategy:
// - Always produce a valid closed walk from node 0 that covers every edge at least once.
// - If graph is Eulerian (all even degrees), just output Euler tour.
// - If not Eulerian:
//   * For small/medium (oddCount <= 26 and N<=4000): do shortest paths from each odd node and solve min-weight perfect matching via DP (exact).
//   * Else: greedy pairing using shortest paths from each odd node (approx, fast).
// - After selecting pairs, duplicate the shortest-path edges between paired odds by adding them as extra edges in a multigraph.
// - Then run Hierholzer to output an Euler circuit in the multigraph. This circuit traverses all original edges at least once.
//
// Assumptions typical for CPP benchmarks:
// - Graph is connected enough that all edges lie in the same component reachable from 0. If not, we still output a circuit
//   covering edges reachable from 0; unreachable edges make the task impossible.

public static class Program
{
    private struct Edge
    {
        public int To;
        public int Next;
        public int Id;
        public Edge(int to, int next, int id) { To = to; Next = next; Id = id; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream s) { _s = s; }
        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }
        public int NextInt()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return int.MinValue; } while (c <= 32);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private sealed class MinHeap
    {
        private int[] _node;
        private long[] _dist;
        private int _n;
        public MinHeap(int cap)
        {
            _node = new int[Math.Max(4, cap)];
            _dist = new long[_node.Length];
            _n = 0;
        }
        public int Count => _n;
        public void Clear() { _n = 0; }
        private void Ensure(int need)
        {
            if (need <= _node.Length) return;
            int ns = _node.Length * 2;
            while (ns < need) ns *= 2;
            Array.Resize(ref _node, ns);
            Array.Resize(ref _dist, ns);
        }
        public void Push(int v, long d)
        {
            Ensure(_n + 1);
            int i = _n++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_dist[p] <= d) break;
                _node[i] = _node[p];
                _dist[i] = _dist[p];
                i = p;
            }
            _node[i] = v;
            _dist[i] = d;
        }
        public void Pop(out int v, out long d)
        {
            v = _node[0];
            d = _dist[0];
            int lv = _node[--_n];
            long ld = _dist[_n];
            int i = 0;
            while (true)
            {
                int l = (i << 1) + 1;
                if (l >= _n) break;
                int r = l + 1;
                int c = (r < _n && _dist[r] < _dist[l]) ? r : l;
                if (_dist[c] >= ld) break;
                _node[i] = _node[c];
                _dist[i] = _dist[c];
                i = c;
            }
            _node[i] = lv;
            _dist[i] = ld;
        }
    }

    private sealed class GraphW
    {
        public readonly int N;
        public int[] Head;
        public int[] To;
        public int[] Next;
        public int[] W;
        public int E;
        public GraphW(int n, int mUndirected)
        {
            N = n;
            Head = new int[n];
            Array.Fill(Head, -1);
            int cap = Math.Max(2, 2 * mUndirected);
            To = new int[cap];
            Next = new int[cap];
            W = new int[cap];
            E = 0;
        }
        public void AddUndirected(int u, int v, int w)
        {
            AddDir(u, v, w);
            AddDir(v, u, w);
        }
        private void AddDir(int u, int v, int w)
        {
            if (E >= To.Length)
            {
                int ns = To.Length * 2;
                Array.Resize(ref To, ns);
                Array.Resize(ref Next, ns);
                Array.Resize(ref W, ns);
            }
            To[E] = v;
            W[E] = w;
            Next[E] = Head[u];
            Head[u] = E++;
        }
    }

    private static void Dijkstra(GraphW g, int src, long[] dist, int[] parentNode, int[] parentEdge)
    {
        const long INF = (long)4e18;
        Array.Fill(dist, INF);
        if (parentNode != null) Array.Fill(parentNode, -1);
        if (parentEdge != null) Array.Fill(parentEdge, -1);
        dist[src] = 0;
        var pq = new MinHeap(128);
        pq.Push(src, 0);
        while (pq.Count > 0)
        {
            pq.Pop(out int u, out long du);
            if (du != dist[u]) continue;
            for (int ei = g.Head[u]; ei != -1; ei = g.Next[ei])
            {
                int v = g.To[ei];
                long nd = du + (uint)g.W[ei];
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    if (parentNode != null)
                    {
                        parentNode[v] = u;
                        if (parentEdge != null) parentEdge[v] = ei;
                    }
                    pq.Push(v, nd);
                }
            }
        }
    }

    private static List<int> ReconstructPathNodes(int src, int dst, int[] parentNode)
    {
        var stack = new List<int>();
        int cur = dst;
        while (cur != -1 && cur != src)
        {
            stack.Add(cur);
            cur = parentNode[cur];
        }
        if (cur == -1) return null;
        stack.Add(src);
        stack.Reverse();
        return stack;
    }

    private static void AddPathAsExtraEdges(List<int> path, List<(int u, int v)> extraEdges)
    {
        for (int i = 0; i + 1 < path.Count; i++)
        {
            extraEdges.Add((path[i], path[i + 1]));
        }
    }

    private static int[] SolveMatchingDP(long[,] cost)
    {
        int k = cost.GetLength(0);
        int full = 1 << k;
        var dp = new long[full];
        var choice = new int[full];
        const long INF = (long)4e18;
        for (int m = 0; m < full; m++) { dp[m] = INF; choice[m] = -1; }
        dp[0] = 0;
        for (int mask = 1; mask < full; mask++)
        {
            if ((mask & 1) == 0) { /* no-op */ }
            if ((BitOperationsPopCount(mask) & 1) == 1) continue;
            int i = LeastBitIndex(mask);
            int maskWithoutI = mask & ~(1 << i);
            long best = INF;
            int bestJ = -1;
            for (int j = i + 1; j < k; j++)
            {
                if ((maskWithoutI & (1 << j)) == 0) continue;
                int nm = maskWithoutI & ~(1 << j);
                long val = dp[nm] + cost[i, j];
                if (val < best)
                {
                    best = val;
                    bestJ = j;
                }
            }
            dp[mask] = best;
            choice[mask] = bestJ;
        }

        int[] mate = new int[k];
        Array.Fill(mate, -1);
        int curMask = full - 1;
        while (curMask != 0)
        {
            int i = LeastBitIndex(curMask);
            int j = choice[curMask];
            mate[i] = j;
            mate[j] = i;
            curMask &= ~(1 << i);
            curMask &= ~(1 << j);
        }
        return mate;
    }

    private static int LeastBitIndex(int mask)
    {
        int idx = 0;
        while (((mask >> idx) & 1) == 0) idx++;
        return idx;
    }

    private static int BitOperationsPopCount(int x)
    {
        // portable popcount
        unchecked
        {
            uint v = (uint)x;
            v = v - ((v >> 1) & 0x55555555u);
            v = (v & 0x33333333u) + ((v >> 2) & 0x33333333u);
            return (int)((((v + (v >> 4)) & 0x0F0F0F0Fu) * 0x01010101u) >> 24);
        }
    }

    private static List<int> EulerCircuit(int n, int[] head, Edge[] edges, int edgeCount, int start)
    {
        var used = new bool[edgeCount];
        var it = new int[n];
        Array.Copy(head, it, n);
        var stack = new int[1];
        int sp = 0;
        void Push(int v)
        {
            if (sp >= stack.Length) Array.Resize(ref stack, stack.Length * 2);
            stack[sp++] = v;
        }
        Push(start);

        var tour = new List<int>(edgeCount + 1);
        while (sp > 0)
        {
            int v = stack[sp - 1];
            int ei = it[v];
            while (ei != -1 && used[edges[ei].Id]) ei = edges[ei].Next;
            it[v] = ei;
            if (ei == -1)
            {
                tour.Add(v);
                sp--;
            }
            else
            {
                used[edges[ei].Id] = true;
                stack[sp - 1] = v;
                Push(edges[ei].To);
                it[v] = edges[ei].Next;
            }
        }
        tour.Reverse();
        return tour;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int N = fs.NextInt();
        if (N == int.MinValue) return;
        int M = fs.NextInt();

        var g = new GraphW(N, M);
        int[] deg = new int[N];

        // For Euler circuit construction we need a multigraph of original + added edges.
        // We'll build adjacency in forward-star with "Edge" struct and per-undirected-edge ids.
        var origU = new int[M];
        var origV = new int[M];
        var origW = new int[M];

        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            origU[i] = u; origV[i] = v; origW[i] = w;
            g.AddUndirected(u, v, w);
            deg[u]++; deg[v]++;
        }

        var odd = new List<int>();
        for (int i = 0; i < N; i++) if ((deg[i] & 1) == 1) odd.Add(i);

        var extraEdges = new List<(int u, int v)>(Math.Max(0, odd.Count));

        if (odd.Count > 0)
        {
            // Decide matching mode
            bool exact = odd.Count <= 26 && N <= 4000;

            if (exact)
            {
                int k = odd.Count;
                long[,] cost = new long[k, k];
                var allParents = new int[k][];

                long[] dist = new long[N];
                int[] parent = new int[N];
                for (int i = 0; i < k; i++)
                {
                    Dijkstra(g, odd[i], dist, parent, null);
                    var parCopy = new int[N];
                    Array.Copy(parent, parCopy, N);
                    allParents[i] = parCopy;
                    for (int j = 0; j < k; j++)
                    {
                        cost[i, j] = dist[odd[j]];
                    }
                }

                int[] mate = SolveMatchingDP(cost);
                var usedOdd = new bool[k];
                for (int i = 0; i < k; i++)
                {
                    if (usedOdd[i]) continue;
                    int j = mate[i];
                    usedOdd[i] = usedOdd[j] = true;
                    var path = ReconstructPathNodes(odd[i], odd[j], allParents[i]);
                    if (path != null) AddPathAsExtraEdges(path, extraEdges);
                }
            }
            else
            {
                // Greedy pairing. For speed, run Dijkstra from each unpaired odd node to find nearest unpaired odd.
                int k = odd.Count;
                var unpaired = new bool[N];
                for (int i = 0; i < k; i++) unpaired[odd[i]] = true;
                int remaining = k;
                long[] dist = new long[N];
                int[] parent = new int[N];

                for (int idx = 0; idx < k && remaining > 0; idx++)
                {
                    int s = odd[idx];
                    if (!unpaired[s]) continue;
                    Dijkstra(g, s, dist, parent, null);
                    long bestD = (long)4e18;
                    int bestT = -1;
                    for (int j = 0; j < k; j++)
                    {
                        int t = odd[j];
                        if (t == s || !unpaired[t]) continue;
                        long d = dist[t];
                        if (d < bestD)
                        {
                            bestD = d;
                            bestT = t;
                        }
                    }
                    if (bestT == -1) break;
                    var path = ReconstructPathNodes(s, bestT, parent);
                    if (path != null) AddPathAsExtraEdges(path, extraEdges);
                    unpaired[s] = false;
                    unpaired[bestT] = false;
                    remaining -= 2;
                }
            }
        }

        // Build multigraph for Euler circuit: original edges + extra edges (each extra edge is a single undirected edge).
        int totalUndirected = M + extraEdges.Count;
        int totalDirected = totalUndirected * 2;
        var head = new int[N];
        Array.Fill(head, -1);
        var edges = new Edge[totalDirected];
        int ePtr = 0;
        int edgeId = 0;

        void AddUndirForEuler(int u, int v)
        {
            int id = edgeId++;
            edges[ePtr] = new Edge(v, head[u], id); head[u] = ePtr++;
            edges[ePtr] = new Edge(u, head[v], id); head[v] = ePtr++;
        }

        for (int i = 0; i < M; i++) AddUndirForEuler(origU[i], origV[i]);
        for (int i = 0; i < extraEdges.Count; i++) AddUndirForEuler(extraEdges[i].u, extraEdges[i].v);

        var tour = EulerCircuit(N, head, edges, edgeId, 0);

        var sb = new StringBuilder(tour.Count * 2);
        for (int i = 0; i < tour.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(tour[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
