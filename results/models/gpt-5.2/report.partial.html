fn wrap_pi(mut a: f64) -> f64 {
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }

/// Parse initial header and map from a whitespace stream.
/// Map rows are read as tokens (strings without whitespace). We do not store the map.
fn read_initial_from_tokens(tokens: &mut dyn Iterator<Item = String>) -> Init {
    let width: usize = tokens.next().unwrap().parse().unwrap();
    let height: usize = tokens.next().unwrap().parse().unwrap();
    let g: f64 = tokens.next().unwrap().parse().unwrap();
    let max_thrust: f64 = tokens.next().unwrap().parse().unwrap();
    let fuel0: f64 = tokens.next().unwrap().parse().unwrap();

    let start_x: f64 = tokens.next().unwrap().parse().unwrap();
    let start_y: f64 = tokens.next().unwrap().parse().unwrap();
    let target_x: f64 = tokens.next().unwrap().parse().unwrap();
    let target_y: f64 = tokens.next().unwrap().parse().unwrap();

    // Determine landing zone extent by scanning for 'L' and 'T' characters.
    // We assume landing zone is contiguous on some row including 'T'.
    // We pick the row that contains 'T' as the landing row.
    let mut lz_min: Option<usize> = None;
    let mut lz_max: Option<usize> = None;

    for _ in 0..height {
        let row = tokens.next().unwrap_or_default();
        if row.is_empty() { continue; }
        if row.contains('T') {
            let bytes = row.as_bytes();
            for (i, &b) in bytes.iter().enumerate() {
                if b == b'L' || b == b'T' {
                    lz_min = Some(lz_min.map(|m| m.min(i)).unwrap_or(i));
                    lz_max = Some(lz_max.map(|m| m.max(i)).unwrap_or(i));
                }
            }
        }
    }

    let (lz_xmin, lz_xmax) = if let (Some(a), Some(b)) = (lz_min, lz_max) {
        (a as f64, b as f64)
    } else {
        // Fallback if map lacks explicit zone: use a small pad around target.
        (target_x - 2.0, target_x + 2.0)
    };

    Init {
        width,
        height,
        g,
        max_thrust,
        fuel0,
        start_x,
        start_y,
        target_x,
        target_y,
        lz_xmin,
        lz_xmax,
    }
}

fn compute_control(init: &Init, s: &State) -> (f64, f64) {
    // Phases based on distance to target and altitude above target.
    let dx = init.target_x - s.x;
    let dy = init.target_y - s.y; // Note: y increases downward; target_y likely ground. If s.y < target_y, dy>0 means below? Actually dy = target - current.

    let alt_above_target = (init.target_y - s.y); // positive if we are above the pad.

    // Desired hover height above target for approach.
    let hover_h = 25.0;
    let desired_y = if alt_above_target > hover_h { init.target_y - hover_h } else { init.target_y - 2.0 };

    // Tighten lateral control near ground.
    let near_ground = alt_above_target < 18.0;
    let very_near = alt_above_target < 6.0;

    // Keep within landing zone x when close.
    let target_x = if near_ground {
        clamp(init.target_x, init.lz_xmin + 0.5, init.lz_xmax - 0.5)
    } else {
        init.target_x
    };

    let ex = target_x - s.x;
    let ey = desired_y - s.y;

    // PD gains (tuned to be stable across many g/thrust settings).
    let kp_x = if near_ground { 0.12 } else { 0.06 };
    let kd_x = if near_ground { 0.35 } else { 0.25 };

    let kp_y = if near_ground { 0.20 } else { 0.10 };
    let kd_y = if near_ground { 0.55 } else { 0.35 };

    // Desired accelerations (world frame). y down.
    let mut ax = kp_x * ex - kd_x * s.vx;

    // Vertical: aim for gentle descent near pad.
    // If very near, directly regulate vy to about +0.5..+1.0 m/s down, and kill vx.
    let desired_vy = if very_near {
        0.8
    } else if near_ground {
        1.5
    } else {
        // allow faster descent when high
        6.0
    };

    let mut ay = kp_y * ey - kd_y * (s.vy - desired_vy);

    // Kill lateral harder when very near.
    if very_near {
        ax = 0.0 * ex - 0.8 * s.vx;
    }

    // Convert desired accel to thrust vector requirement.
    // Net acceleration = thrust_accel + gravity (down +g).
    // So thrust_accel_y should be (ay - g) ???
    // If we want net ay (down positive), and gravity contributes +g,
    // then thrust contribution in y is (ay - g).
    let thrust_ay = ay - init.g;
    let thrust_ax = ax;

    let desired_thrust_acc = (thrust_ax * thrust_ax + thrust_ay * thrust_ay).sqrt();

    // Desired angle points in direction of thrust acceleration.
    // angle=0 => thrust along +x, angle=PI/2 => along +y (down).
    // We generally want thrust upward => negative y => angle around -PI/2.
    let desired_angle = thrust_ay.atan2(thrust_ax);

    // Convert thrust accel to throttle.
    let mut throttle = if init.max_thrust > 1e-6 {
        desired_thrust_acc / init.max_thrust
    } else {
        0.0
    };

    // Fuel safety: if out of fuel, throttle = 0.
    if s.fuel <= 0.0 {
        throttle = 0.0;
    }

    // Clamp throttle.
    throttle = clamp(throttle, 0.0, 1.0);

    // Angle control (PD) to turn rate.
    let ang_err = wrap_pi(desired_angle - s.angle);
    let kp_ang = 3.2;
    let kd_ang = 0.6;
    let mut turn = kp_ang * ang_err - kd_ang * s.ang_vel;
    turn = clamp(turn, -1.0, 1.0);

    // Touchdown guard: if extremely close to target height, reduce thrust to avoid bouncing,
    // but keep enough to keep vy small.
    if alt_above_target < 1.2 {
        // If descending too fast, burn hard.
        if s.vy > 2.0 {
            throttle = throttle.max(0.6);
        } else {
            throttle = throttle.min(0.25);
        }
        // Prefer upright-ish (thrust up) by limiting angle magnitude.
        // We can only request turn rate, but bias desired to -PI/2.
        let desired = -PI / 2.0;
        let e = wrap_pi(desired - s.angle);
        turn = clamp(2.5 * e - 0.5 * s.ang_vel, -1.0, 1.0);
    }

    (throttle, turn)
}

                    // skip whitespace

                        self.i += 1;
                    }
                    if self.i < self.buf.len() {
                        break;
                    }

                        return None;

}

/// Parse initial header and map from a whitespace stream.
/// Map rows are read as tokens (strings without whitespace). We do not store the map.
fn read_initial_from_tokens(tokens: &mut dyn Iterator<Item = String>) -> Init {
    let width: usize = tokens.next().unwrap().parse().unwrap();
    let height: usize = tokens.next().unwrap().parse().unwrap();
    let g: f64 = tokens.next().unwrap().parse().unwrap();
    let max_thrust: f64 = tokens.next().unwrap().parse().unwrap();
    let fuel0: f64 = tokens.next().unwrap().parse().unwrap();

    let start_x: f64 = tokens.next().unwrap().parse().unwrap();
    let start_y: f64 = tokens.next().unwrap().parse().unwrap();
    let target_x: f64 = tokens.next().unwrap().parse().unwrap();
    let target_y: f64 = tokens.next().unwrap().parse().unwrap();

    // Determine landing zone extent by scanning for 'L' and 'T' characters.
    // We assume landing zone is contiguous on some row including 'T'.
    // We pick the row that contains 'T' as the landing row.
    let mut lz_min: Option<usize> = None;
    let mut lz_max: Option<usize> = None;

    for _ in 0..height {
        let row = tokens.next().unwrap_or_default();
        if row.is_empty() { continue; }
        if row.contains('T') {
            let bytes = row.as_bytes();
            for (i, &b) in bytes.iter().enumerate() {
                if b == b'L' || b == b'T' {
                    lz_min = Some(lz_min.map(|m| m.min(i)).unwrap_or(i));
                    lz_max = Some(lz_max.map(|m| m.max(i)).unwrap_or(i));
                }
            }
        }
    }

    let (lz_xmin, lz_xmax) = if let (Some(a), Some(b)) = (lz_min, lz_max) {
        (a as f64, b as f64)
    } else {
        // Fallback if map lacks explicit zone: use a small pad around target.
        (target_x - 2.0, target_x + 2.0)
    };

    Init {
        width,
        height,
        g,
        max_thrust,
        fuel0,
        start_x,
        start_y,
        target_x,
        target_y,
        lz_xmin,
        lz_xmax,
    }
}

fn compute_control(init: &Init, s: &State) -> (f64, f64) {
    // Phases based on distance to target and altitude above target.
    let dx = init.target_x - s.x;
    let dy = init.target_y - s.y; // Note: y increases downward; target_y likely ground. If s.y < target_y, dy>0 means below? Actually dy = target - current.

    let alt_above_target = (init.target_y - s.y); // positive if we are above the pad.

    // Desired hover height above target for approach.
    let hover_h = 25.0;
    let desired_y = if alt_above_target > hover_h { init.target_y - hover_h } else { init.target_y - 2.0 };

    // Tighten lateral control near ground.
    let near_ground = alt_above_target < 18.0;
    let very_near = alt_above_target < 6.0;

    // Keep within landing zone x when close.
    let target_x = if near_ground {
        clamp(init.target_x, init.lz_xmin + 0.5, init.lz_xmax - 0.5)
    } else {
        init.target_x
    };

    let ex = target_x - s.x;
    let ey = desired_y - s.y;

    // PD gains (tuned to be stable across many g/thrust settings).
    let kp_x = if near_ground { 0.12 } else { 0.06 };
    let kd_x = if near_ground { 0.35 } else { 0.25 };

    let kp_y = if near_ground { 0.20 } else { 0.10 };
    let kd_y = if near_ground { 0.55 } else { 0.35 };

    // Desired accelerations (world frame). y down.
    let mut ax = kp_x * ex - kd_x * s.vx;

    // Vertical: aim for gentle descent near pad.
    // If very near, directly regulate vy to about +0.5..+1.0 m/s down, and kill vx.
    let desired_vy = if very_near {
        0.8
    } else if near_ground {
        1.5
    } else {
        // allow faster descent when high
        6.0
    };

    let mut ay = kp_y * ey - kd_y * (s.vy - desired_vy);

    // Kill lateral harder when very near.
    if very_near {
        ax = 0.0 * ex - 0.8 * s.vx;
    }

    // Convert desired accel to thrust vector requirement.
    // Net acceleration = thrust_accel + gravity (down +g).
    // So thrust_accel_y should be (ay - g) ???
    // If we want net ay (down positive), and gravity contributes +g,
    // then thrust contribution in y is (ay - g).
    let thrust_ay = ay - init.g;
    let thrust_ax = ax;

    let desired_thrust_acc = (thrust_ax * thrust_ax + thrust_ay * thrust_ay).sqrt();

    // Desired angle points in direction of thrust acceleration.
    // angle=0 => thrust along +x, angle=PI/2 => along +y (down).
    // We generally want thrust upward => negative y => angle around -PI/2.
    let desired_angle = thrust_ay.atan2(thrust_ax);

    // Convert thrust accel to throttle.
    let mut throttle = if init.max_thrust > 1e-6 {
        desired_thrust_acc / init.max_thrust
    } else {
        0.0
    };

    // Fuel safety: if out of fuel, throttle = 0.
    if s.fuel <= 0.0 {
        throttle = 0.0;
    }

    // Clamp throttle.
    throttle = clamp(throttle, 0.0, 1.0);

    // Angle control (PD) to turn rate.
    let ang_err = wrap_pi(desired_angle - s.angle);
    let kp_ang = 3.2;
    let kd_ang = 0.6;
    let mut turn = kp_ang * ang_err - kd_ang * s.ang_vel;
    turn = clamp(turn, -1.0, 1.0);

    // Touchdown guard: if extremely close to target height, reduce thrust to avoid bouncing,
    // but keep enough to keep vy small.
    if alt_above_target < 1.2 {
        // If descending too fast, burn hard.
        if s.vy > 2.0 {
            throttle = throttle.max(0.6);
        } else {
            throttle = throttle.min(0.25);
        }
        // Prefer upright-ish (thrust up) by limiting angle magnitude.
        // We can only request turn rate, but bias desired to -PI/2.
        let desired = -PI / 2.0;
        let e = wrap_pi(desired - s.angle);
        turn = clamp(2.5 * e - 0.5 * s.ang_vel, -1.0, 1.0);
    }

    (throttle, turn)
}

// --- Full working version ---

fn main_working() {
    // Single stdin reader thread; main thread becomes controller.
    // To parse init in controller thread without risking long stalls, we do it quickly:
    //  - read first few lines and the map rows; then hand the locked stdin to reader thread is not possible.
    // So we instead parse init+map in the reader thread, and send Init to controller via a channel.

    let (init_tx, init_rx) = mpsc::sync_channel::<Init>(1);
    let (state_tx, state_rx) = mpsc::sync_channel::<State>(1);

    let reader = thread::spawn(move || {
        // Use BufRead for streaming tokens.
        let stdin = io::stdin();
        let mut reader = io::BufReader::new(stdin.lock());
        let mut bytes = Vec::<u8>::with_capacity(1 << 16);
        let mut tmp = [0u8; 8192];

        // Tokenizer: reads bytes and yields tokens without allocating per-char.
        // We implement a simple pull tokenizer with an internal buffer.
        struct Tok {
            buf: Vec<u8>,
            i: usize,
        }
        impl Tok {
            fn new() -> Self { Self { buf: Vec::new(), i: 0 } }
            fn refill<R: Read>(&mut self, r: &mut R, tmp: &mut [u8]) -> bool {
                let n = match r.read(tmp) { Ok(0) | Err(_) => 0, Ok(n) => n };
                if n == 0 { return false; }
                if self.i > 0 {
                    self.buf.drain(0..self.i);
                    self.i = 0;
                }
                self.buf.extend_from_slice(&tmp[..n]);
                true
            }
            fn next_token<R: Read>(&mut self, r: &mut R, tmp: &mut [u8]) -> Option<Vec<u8>> {
                loop {
                    // skip whitespace
                    while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
                        self.i += 1;
                    }
                    if self.i < self.buf.len() {
                        break;
                    }
                    if !self.refill(r, tmp) {
                        return None;
                    }
                }
                let start = self.i;
                while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
                    self.i += 1;
                }
                Some(self.buf[start..self.i].to_vec())
            }
            fn next_f64<R: Read>(&mut self, r: &mut R, tmp: &mut [u8]) -> Option<f64> {
                let t = self.next_token(r, tmp)?;
                let s = unsafe { std::str::from_utf8_unchecked(&t) };
                s.parse::<f64>().ok()
            }
            fn next_usize<R: Read>(&mut self, r: &mut R, tmp: &mut [u8]) -> Option<usize> {
                let t = self.next_token(r, tmp)?;
                let s = unsafe { std::str::from_utf8_unchecked(&t) };
                s.parse::<usize>().ok()
            }
        }

        let mut tok = Tok::new();

        // Parse init
        let width = tok.next_usize(&mut reader, &mut tmp).unwrap();
        let height = tok.next_usize(&mut reader, &mut tmp).unwrap();
        let g = tok.next_f64(&mut reader, &mut tmp).unwrap();
        let max_thrust = tok.next_f64(&mut reader, &mut tmp).unwrap();
        let fuel0 = tok.next_f64(&mut reader, &mut tmp).unwrap();

        let start_x = tok.next_f64(&mut reader, &mut tmp).unwrap();
        let start_y = tok.next_f64(&mut reader, &mut tmp).unwrap();
        let target_x = tok.next_f64(&mut reader, &mut tmp).unwrap();
        let target_y = tok.next_f64(&mut reader, &mut tmp).unwrap();

        let mut lz_min: Option<usize> = None;
        let mut lz_max: Option<usize> = None;

        for _ in 0..height {
            let row_bytes = tok.next_token(&mut reader, &mut tmp).unwrap_or_default();
            if row_bytes.is_empty() { continue; }
            let row = row_bytes;
            let has_t = row.iter().any(|&b| b == b'T');
            if has_t {
                for (i, &b) in row.iter().enumerate() {
                    if b == b'L' || b == b'T' {
                        lz_min = Some(lz_min.map(|m| m.min(i)).unwrap_or(i));
                        lz_max = Some(lz_max.map(|m| m.max(i)).unwrap_or(i));
                    }
                }
            }
        }

        let (lz_xmin, lz_xmax) = if let (Some(a), Some(b)) = (lz_min, lz_max) {
            (a as f64, b as f64)
        } else {
            (target_x - 2.0, target_x + 2.0)
        };

        let init = Init {
            width,
            height,
            g,
            max_thrust,
            fuel0,
            start_x,
            start_y,
            target_x,
            target_y,
            lz_xmin,
            lz_xmax,
        };

        let _ = init_tx.send(init);

        // Stream states
        loop {
            let x = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };
            let y = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };
            let vx = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };
            let vy = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };
            let angle = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };
            let ang_vel = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };
            let fuel = match tok.next_f64(&mut reader, &mut tmp) { Some(v) => v, None => break };

            let s = State { x, y, vx, vy, angle, ang_vel, fuel };

            match state_tx.try_send(s) {
                Ok(_) => {}
                Err(TrySendError::Full(_)) => {
                    // Drop oldest by receiving once, then send.
                    let _ = state_rx.recv_timeout(Duration::from_millis(0));
                    let _ = state_tx.try_send(s);
                }
                Err(TrySendError::Disconnected(_)) => break,
            }
        }
    });

    let init = init_rx.recv().unwrap();

    let mut out = BufWriter::new(io::stdout().lock());

    // Control loop at 50 Hz to satisfy >=20 Hz output even if inputs come at 10 Hz.
    let dt = Duration::from_millis(20);
    let mut next_tick = Instant::now();
    let mut last_state: State = State { x: init.start_x, y: init.start_y, ..State::default() };

    loop {
        // Drain newest state if available.
        while let Ok(s) = state_rx.try_recv() {
            last_state = s;
        }

        let (thrust, turn) = compute_control(&init, &last_state);
        let _ = writeln!(out, "{:.6} {:.6}", thrust, turn);
        let _ = out.flush();

        next_tick += dt;
        let now = Instant::now();
        if next_tick > now {
            thread::sleep(next_tick - now);
        } else {
            // If we fell behind, resync quickly.
            next_tick = now;
        }
    }

    let _ = reader.join();
}
