vector<int> nnTour(){

void twoOptKNN(vector<int>& tour, const vector<vector<int>>& knn){
    int n=(int)tour.size();
    vector<int> pos(N); for(int i=0;i<n;i++) pos[tour[i]]=i;
    bool imp=true;
    while(imp && elapsed()<TL){
        imp=false;
        for(int i=0;i<n && elapsed()<TL;i++){

            float dab=D(a,b);
            for(int c : knn[a]){
                if(D(a,c) >= dab) break; // knn sorted ascending; no improvement possible
                int j=pos[c];
                if(j<=i+1) continue;
                int d=tour[(j+1)%n];

                if(gain>1e-6f){
                    reverse(tour.begin()+i+1, tour.begin()+j+1);

                    b=tour[(i+1)%n]; dab=D(a,b);
                    imp=true; break;
                }
            }
        }
    }
}

// ---- Held-Karp DP for 12 <= N <= 20 ----
vector<int> heldKarp(){
    int n=N, S=1<<n;
    vector<float> dp((long long)S*n, 1e30f);
    vector<int> par((long long)S*n, -1);
    dp[1*n+0]=0.0f;
    for(int mask=1;mask<S;mask++){
        if(!(mask&1)) continue;
        for(int u=0;u<n;u++){
            if(!(mask>>u&1)) continue;
            float du=dp[(long long)mask*n+u];
            if(du>=1e29f) continue;
            for(int v=0;v<n;v++){
                if(mask>>v&1) continue;
                int nm=mask|(1<<v);
                float nd=du+D(u,v);
                long long idx=(long long)nm*n+v;
                if(nd<dp[idx]){ dp[idx]=nd; par[idx]=u; }
            }
        }
    }
    int full=S-1; float best=1e30f; int last=-1;
    for(int i=1;i<n;i++){
        float d=dp[(long long)full*n+i]+D(i,0);
        if(d<best){ best=d; last=i; }
    }
    vector<int> t; int mask=full, cur=last;
    while(cur!=-1){
        t.push_back(cur);
        int p2=par[(long long)mask*n+cur];
        mask^=(1<<cur); cur=p2;
    }
    reverse(t.begin(),t.end());
    return t;
}

            }
            else
            {
                SpanningTreeParity(deg);
            }
        }

        var circuit = EulerianCircuit();

        for (int i = 0; i < circuit.Count; i++)

    static (long[] dist, int[] prevEdge) Dijkstra(int src)
    {
        var dist = new long[N];
        var prevEdge = new int[N];
        Array.Fill(dist, long.MaxValue / 2);
        Array.Fill(prevEdge, -1);
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.TryDequeue(out int u, out long d))
        {
            if (d > dist[u]) continue;
            foreach (var (v, w, eidx) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    prevEdge[v] = eidx;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, prevEdge);
    }

    static void AddPath(int dest, int[] prevEdge)
    {
        int v = dest;
        while (prevEdge[v] >= 0)
        {
            int e = prevEdge[v];
            mult[e]++;
            v = (ev[e] == v) ? eu[e] : ev[e];
        }
    }

    static void SpanningTreeParity(int[] deg)
    {
        int[] parent = new int[N];
        int[] parentEdge = new int[N];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        var bfsOrder = new List<int>(N);
        var visited = new bool[N];
        var queue = new Queue<int>();
        queue.Enqueue(0);
        visited[0] = true;

        while (queue.Count > 0)
        {
            int u = queue.Dequeue();
            bfsOrder.Add(u);
            foreach (var (v, w, eidx) in adj[u])
            {
                if (!visited[v])
                {
                    visited[v] = true;
                    parent[v] = u;
                    parentEdge[v] = eidx;
                    queue.Enqueue(v);
                }
            }
        }

        int[] adjustedDeg = (int[])deg.Clone();

        for (int i = bfsOrder.Count - 1; i >= 1; i--)
        {
            int v = bfsOrder[i];
            if (adjustedDeg[v] % 2 == 1)
            {
                int e = parentEdge[v];
                mult[e]++;
                adjustedDeg[v]++;
                adjustedDeg[parent[v]]++;
            }
        }
    }

    static int[] DPMatch(long[,] dist, int k)
    {
        int states = 1 << k;
        var dp = new long[states];
        var choice = new int[states];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;

        for (int mask = 0; mask < states; mask++)
        {
            if (dp[mask] >= long.MaxValue / 2) continue;
            int i = 0;
            while (i < k && (mask & (1 << i)) != 0) i++;
            if (i >= k) continue;
            for (int j = i + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int nm = mask | (1 << i) | (1 << j);
                long cost = dp[mask] + dist[i, j];
                if (cost < dp[nm])
                {
                    dp[nm] = cost;
                    choice[nm] = i * k + j;
                }
            }
        }

        var matching = new int[k];
        Array.Fill(matching, -1);
        int cur = states - 1;
        while (cur > 0)
        {
            int c = choice[cur];
            int a = c / k, b = c % k;
            matching[a] = b;
            matching[b] = a;
            cur ^= (1 << a) | (1 << b);
        }
        return matching;
    }

    static int[] GreedyMatch(long[,] dist, int k)
    {
        var pairs = new List<(long d, int i, int j)>(k * (k - 1) / 2);
        for (int i = 0; i < k; i++)
            for (int j = i + 1; j < k; j++)
                pairs.Add((dist[i, j], i, j));
        pairs.Sort((a, b) => a.d.CompareTo(b.d));

        var matching = new int[k];
        Array.Fill(matching, -1);
        var used = new bool[k];
        foreach (var (_, i, j) in pairs)
        {
            if (!used[i] && !used[j])
            {
                matching[i] = j;
                matching[j] = i;
                used[i] = used[j] = true;
            }
        }
        return matching;
    }

    static List<int> EulerianCircuit()
    {
        int totalHalfEdges = 0;
        for (int i = 0; i < M; i++) totalHalfEdges += 2 * mult[i];

        var hTo = new int[totalHalfEdges];
        var hMate = new int[totalHalfEdges];
        var hUsed = new bool[totalHalfEdges];
        var adj2 = new List<int>[N];
        for (int i = 0; i < N; i++) adj2[i] = new List<int>();

        int hBase = 0;
        for (int i = 0; i < M; i++)
        {
            for (int c = 0; c < mult[i]; c++)
            {
                int h1 = hBase + 2 * c;
                int h2 = hBase + 2 * c + 1;
                hTo[h1] = ev[i];
                hTo[h2] = eu[i];
                hMate[h1] = h2;
                hMate[h2] = h1;
                adj2[eu[i]].Add(h1);
                adj2[ev[i]].Add(h2);
            }
            hBase += 2 * mult[i];
        }

        var ptr = new int[N];
        var stack = new Stack<int>();
        var circuit = new List<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            while (ptr[u] < adj2[u].Count && hUsed[adj2[u][ptr[u]]])
                ptr[u]++;

            if (ptr[u] < adj2[u].Count)
            {
                int h = adj2[u][ptr[u]++];
                hUsed[h] = hUsed[hMate[h]] = true;
                stack.Push(hTo[h]);
            }
            else
            {
                circuit.Add(u);
                stack.Pop();
            }
        }

        circuit.Reverse();
        return circuit;
    }

fn find_components(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    let mut comp = vec![usize::MAX; n];

        queue.push_back(start);

    }

}

fn force_directed_full(n: usize, adj: &[Vec<usize>], deadline_ms: u128, start: Instant) -> Vec<(f64, f64)> {

        for i in 0..n { dx[i] = 0.0; dy[i] = 0.0; }
        for i in 0..n {

                let f = k*k/dist;

                dx[i] += fx; dy[i] += fy; dx[j] -= fx; dy[j] -= fy;
            }
        }

                    let ddx = px[i]-px[j]; let ddy = py[i]-py[j];

        }
        for i in 0..n {
            let d = (dx[i]*dx[i]+dy[i]*dy[i]).sqrt().max(0.001);
            let scale = d.min(temp)/d;
            px[i] += dx[i]*scale; py[i] += dy[i]*scale;
        }
        temp = (temp - cooling).max(0.1);
    }
    px.into_iter().zip(py).collect()
}

fn bfs_radial(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    use std::f64::consts::PI;
    let root = (0..n).max_by_key(|&i| adj[i].len()).unwrap_or(0);
    let mut dist = vec![usize::MAX; n];
    let mut queue = VecDeque::new();
    dist[root] = 0;
    queue.push_back(root);
    let mut order = Vec::with_capacity(n);
    while let Some(u) = queue.pop_front() {
        order.push(u);
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                queue.push_back(v);
            }
        }
    }
    for i in 0..n {
        if dist[i] == usize::MAX { dist[i] = 0; order.push(i); }
    }
    let max_dist = dist.iter().copied().max().unwrap_or(0);
    let mut layers: Vec<Vec<usize>> = vec![vec![]; max_dist + 1];
    for &u in &order {
        if dist[u] <= max_dist { layers[dist[u]].push(u); }
    }
    let spacing = (n as f64).sqrt().max(5.0) * 2.5;
    let mut pos = vec![(0.0f64, 0.0f64); n];
    for (d, layer) in layers.iter().enumerate() {
        let r = d as f64 * spacing;
        let count = layer.len();
        for (i, &u) in layer.iter().enumerate() {
            if d == 0 { pos[u] = (0.0, 0.0); }
            else {
                let angle = 2.0 * PI * i as f64 / count as f64;
                pos[u] = (r * angle.cos(), r * angle.sin());
            }
        }
    }
    pos
}
