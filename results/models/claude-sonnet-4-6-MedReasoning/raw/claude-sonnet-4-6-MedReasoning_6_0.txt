using System;
using System.Diagnostics;

class Program
{
    static double mu;

    struct OrbElem
    {
        public double a, e, i, raan;
    }

    static OrbElem StateToElem(double[] r, double[] v)
    {
        double rx = r[0], ry = r[1], rz = r[2];
        double vx = v[0], vy = v[1], vz = v[2];
        double rmag = Math.Sqrt(rx*rx + ry*ry + rz*rz);
        double vmag2 = vx*vx + vy*vy + vz*vz;
        double energy = vmag2/2.0 - mu/rmag;
        double a = energy < 0 ? -mu/(2.0*energy) : rmag*2.0;

        double hx = ry*vz - rz*vy;
        double hy = rz*vx - rx*vz;
        double hz = rx*vy - ry*vx;
        double hmag = Math.Sqrt(hx*hx + hy*hy + hz*hz);
        double inc = Math.Acos(Math.Max(-1.0, Math.Min(1.0, hz/hmag)));

        double ex = (vy*hz - vz*hy)/mu - rx/rmag;
        double ey = (vz*hx - vx*hz)/mu - ry/rmag;
        double ez = (vx*hy - vy*hx)/mu - rz/rmag;
        double e = Math.Sqrt(ex*ex + ey*ey + ez*ez);

        double Nx = -hy, Ny = hx;
        double Nmag = Math.Sqrt(Nx*Nx + Ny*Ny);
        double raan = Nmag > 1e-10 ? Math.Atan2(Ny, Nx) : 0.0;

        return new OrbElem { a = Math.Max(a, 6471.0), e = Math.Min(e, 0.999), i = inc, raan = raan };
    }

    static double TransferCost(OrbElem A, OrbElem B)
    {
        double a1 = A.a, a2 = B.a;
        double v1c = Math.Sqrt(mu/a1);
        double v2c = Math.Sqrt(mu/a2);
        double at = (a1 + a2)/2.0;
        double vt1 = Math.Sqrt(mu*(2.0/a1 - 1.0/at));
        double vt2 = Math.Sqrt(mu*(2.0/a2 - 1.0/at));
        double dv1 = Math.Abs(vt1 - v1c);

        double di = Math.Abs(A.i - B.i);
        double dv2 = Math.Sqrt(vt2*vt2 + v2c*v2c - 2.0*vt2*v2c*Math.Cos(di));

        double draan = Math.Abs(A.raan - B.raan);
        if (draan > Math.PI) draan = 2.0*Math.PI - draan;
        double vavg = (v1c + v2c)/2.0;
        double dv_raan = 2.0*vavg*Math.Sin(draan/2.0)*Math.Sin((A.i + B.i)/2.0)*0.2;

        return dv1 + dv2 + dv_raan;
    }

    static double[] ParseLine(string line)
    {
        var parts = line.Trim().Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        var result = new double[parts.Length];
        for (int i = 0; i < parts.Length; i++)
            result[i] = double.Parse(parts[i], System.Globalization.CultureInfo.InvariantCulture);
        return result;
    }

    static int[] HeldKarp(double[,] c, double[] sc, int n)
    {
        int S = 1 << n;
        var dp = new double[S, n];
        var par = new int[S, n];
        for (int s = 0; s < S; s++)
            for (int i = 0; i < n; i++)
                dp[s, i] = 1e18;

        for (int i = 0; i < n; i++)
        {
            dp[1 << i, i] = sc[i];
            par[1 << i, i] = -1;
        }

        for (int s = 1; s < S; s++)
        {
            for (int i = 0; i < n; i++)
            {
                if ((s & (1 << i)) == 0 || dp[s, i] >= 1e18) continue;
                double dpi = dp[s, i];
                for (int j = 0; j < n; j++)
                {
                    if ((s & (1 << j)) != 0) continue;
                    int ns = s | (1 << j);
                    double nc = dpi + c[i, j];
                    if (nc < dp[ns, j])
                    {
                        dp[ns, j] = nc;
                        par[ns, j] = i;
                    }
                }
            }
        }

        int full = S - 1, best = 0;
        for (int i = 1; i < n; i++)
            if (dp[full, i] < dp[full, best]) best = i;

        var tour = new int[n];
        int cur = best, mask = full;
        for (int k = n - 1; k >= 0; k--)
        {
            tour[k] = cur;
            int prev = par[mask, cur];
            mask ^= 1 << cur;
            cur = prev;
        }
        return tour;
    }

    static int[] GreedyNN(double[,] c, double[] sc, int n, int startNode)
    {
        var visited = new bool[n];
        var tour = new int[n];
        int cur = startNode;
        tour[0] = cur;
        visited[cur] = true;
        for (int k = 1; k < n; k++)
        {
            double best = 1e18;
            int next = -1;
            for (int j = 0; j < n; j++)
                if (!visited[j] && c[cur, j] < best) { best = c[cur, j]; next = j; }
            tour[k] = next;
            visited[next] = true;
            cur = next;
        }
        return tour;
    }

    static double TourCost(int[] t, double[,] c, double[] sc)
    {
        double total = sc[t[0]];
        for (int i = 1; i < t.Length; i++) total += c[t[i-1], t[i]];
        return total;
    }

    static void TwoOpt(int[] t, double[,] c, double[] sc, Stopwatch sw, long tl)
    {
        int n = t.Length;
        bool any = true;
        while (any && sw.ElapsedMilliseconds < tl)
        {
            any = false;
            for (int i = 0; i < n - 1 && sw.ElapsedMilliseconds < tl; i++)
            {
                double rem1 = i == 0 ? sc[t[0]] : c[t[i-1], t[i]];
                for (int j = i + 2; j < n; j++)
                {
                    double rem2 = j < n-1 ? c[t[j], t[j+1]] : 0.0;
                    double add1 = i == 0 ? sc[t[j]] : c[t[i-1], t[j]];
                    double add2 = j < n-1 ? c[t[i], t[j+1]] : 0.0;
                    if (add1 + add2 < rem1 + rem2 - 1e-9)
                    {
                        Array.Reverse(t, i, j - i + 1);
                        rem1 = i == 0 ? sc[t[0]] : c[t[i-1], t[i]];
                        any = true;
                    }
                }
            }
        }
    }

    static void NodeInsertion(int[] t, double[,] c, double[] sc, Stopwatch sw, long tl)
    {
        int n = t.Length;
        bool any = true;
        while (any && sw.ElapsedMilliseconds < tl)
        {
            any = false;
            for (int i = 0; i < n && sw.ElapsedMilliseconds < tl; i++)
            {
                double prevI = i == 0 ? sc[t[i]] : c[t[i-1], t[i]];
                double nextI = i == n-1 ? 0.0 : c[t[i], t[i+1]];
                double skipI;
                if (i == 0 && n > 1)       skipI = sc[t[1]];
                else if (i == n-1)          skipI = 0.0;
                else                        skipI = c[t[i-1], t[i+1]];

                double removeSavings = prevI + nextI - skipI;
                int node = t[i];

                for (int j = 0; j < n - 1; j++)
                {
                    if (j == i-1 || j == i) continue;
                    double oldEdge = c[t[j], t[j+1]];
                    double insertCost = c[t[j], node] + c[node, t[j+1]];
                    double gain = removeSavings - (insertCost - oldEdge);
                    if (gain > 1e-9)
                    {
                        var newT = new int[n];
                        int ri = 0;
                        int insertAfter = j < i ? j : j - 1;
                        for (int k = 0; k < n; k++)
                        {
                            if (k == i) continue;
                            int mapped = k < i ? k : k - 1;
                            newT[ri++] = t[k];
                            if (mapped == insertAfter)
                                newT[ri++] = node;
                        }
                        if (ri == n)
                        {
                            Array.Copy(newT, t, n);
                            any = true;
                            i = -1;
                            break;
                        }
                    }
                }
            }
        }
    }

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        long timeLimit = 270_000L;

        var p1 = ParseLine(Console.ReadLine());
        int n = (int)p1[0];
        mu = p1[1];

        var sl = ParseLine(Console.ReadLine());
        double[] sp = {sl[0], sl[1], sl[2]};
        double[] sv = {sl[3], sl[4], sl[5]};

        var spos = new double[n][];
        var svel = new double[n][];
        for (int i = 0; i < n; i++)
        {
            var pp = ParseLine(Console.ReadLine());
            spos[i] = new[]{pp[0], pp[1], pp[2]};
            svel[i] = new[]{pp[3], pp[4], pp[5]};
        }

        var se = StateToElem(sp, sv);
        var elems = new OrbElem[n];
        for (int i = 0; i < n; i++) elems[i] = StateToElem(spos[i], svel[i]);

        var c = new double[n, n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (i != j) c[i, j] = TransferCost(elems[i], elems[j]);

        var sc = new double[n];
        for (int i = 0; i < n; i++) sc[i] = TransferCost(se, elems[i]);

        int[] tour;
        if (n <= 20)
        {
            tour = HeldKarp(c, sc, n);
        }
        else
        {
            tour = GreedyNN(c, sc, n, 0);
            double bestCost = TourCost(tour, c, sc);

            int trials = Math.Min(n, 40);
            for (int t = 0; t < trials && sw.ElapsedMilliseconds < timeLimit/5; t++)
            {
                var candidate = GreedyNN(c, sc, n, t);
                double cc = TourCost(candidate, c, sc);
                if (cc < bestCost) { bestCost = cc; tour = candidate; }
            }

            TwoOpt(tour, c, sc, sw, timeLimit - 20_000);
            NodeInsertion(tour, c, sc, sw, timeLimit - 5_000);
        }

        Console.WriteLine(string.Join(" ", tour));
    }
}
