#include <cstdio>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <chrono>

using namespace std;

static auto T0 = chrono::steady_clock::now();
double elapsed(){ return chrono::duration<double>(chrono::steady_clock::now()-T0).count(); }
const double TL = 270.0;

int N;
vector<float> PX, PY;

inline float D(int a, int b){
    float dx=PX[a]-PX[b], dy=PY[a]-PY[b];
    return sqrtf(dx*dx+dy*dy);
}

// ---- Fast buffered I/O ----
static char ibuf[1<<23]; static int ipos=0, ilen=0;
static char gc(){
    if(ipos>=ilen){ ilen=(int)fread(ibuf,1,sizeof(ibuf),stdin); ipos=0; }
    return (ipos<ilen) ? ibuf[ipos++] : '\0';
}
static int ri(){
    char c; while((c=gc())<'0'||c>'9'){ if(!c)return 0; }
    int x=c-'0'; while((c=gc())>='0'&&c<='9') x=x*10+(c-'0'); return x;
}
static float rf(){
    char c; while((c=gc())==' '||c=='\n'||c=='\r'||c=='\t'){ if(!c)return 0.0f; }
    bool neg=(c=='-'); if(neg) c=gc();
    double v=0;
    while(c>='0'&&c<='9'){ v=v*10+(c-'0'); c=gc(); }
    if(c=='.'){ c=gc(); double f=0.1; while(c>='0'&&c<='9'){ v+=(c-'0')*f; f*=0.1; c=gc(); } }
    return (float)(neg?-v:v);
}
static char obuf[1<<23]; static int opos=0;
static void oflush(){ fwrite(obuf,1,opos,stdout); opos=0; }
static void wc(char c){ if(opos>=(int)sizeof(obuf)-1) oflush(); obuf[opos++]=c; }
static void wi(int x){
    if(opos>(int)sizeof(obuf)-20) oflush();
    if(x==0){ obuf[opos++]='0'; return; }
    char t[12]; int n=0; while(x>0){ t[n++]='0'+x%10; x/=10; }
    for(int i=n-1;i>=0;i--) obuf[opos++]=t[i];
}

// ---- Spatial Grid ----
struct Grid {
    float ox, oy, cs; int W, H;
    vector<int> head, nxt;

    void build(float cellSz){
        float x1=PX[0],y1=PY[0],x2=PX[0],y2=PY[0];
        for(int i=1;i<N;i++){
            if(PX[i]<x1)x1=PX[i]; if(PY[i]<y1)y1=PY[i];
            if(PX[i]>x2)x2=PX[i]; if(PY[i]>y2)y2=PY[i];
        }
        ox=x1; oy=y1; cs=cellSz;
        W=max(1,(int)((x2-x1)/cs)+2);
        H=max(1,(int)((y2-y1)/cs)+2);
        if((long long)W*H>5000000LL){
            cs=sqrtf(max((x2-x1)*(y2-y1),1.0f)/5000000.0f)+1e-6f;
            W=max(1,(int)((x2-x1)/cs)+2);
            H=max(1,(int)((y2-y1)/cs)+2);
        }
        head.assign((long long)W*H, -1); nxt.resize(N);
        for(int i=N-1;i>=0;i--){ int c=cid(PX[i],PY[i]); nxt[i]=head[c]; head[c]=i; }
    }

    int cid(float x, float y) const {
        int c=max(0,min(W-1,(int)((x-ox)/cs)));
        int r=max(0,min(H-1,(int)((y-oy)/cs)));
        return r*W+c;
    }

    int nn1(float qx, float qy, const vector<bool>& vis) const {
        int qc=max(0,min(W-1,(int)((qx-ox)/cs)));
        int qr=max(0,min(H-1,(int)((qy-oy)/cs)));
        float bd=1e30f; int best=-1;
        for(int r=0;;r++){
            float md=(float)max(0,r-1)*cs;
            if(md*md>bd && best!=-1) break;
            if(r>W+H+2) break;
            int r0=max(0,qr-r), r1=min(H-1,qr+r);
            int c0=max(0,qc-r), c1=min(W-1,qc+r);
            for(int row=r0;row<=r1;row++)
                for(int col=c0;col<=c1;col++){
                    if(r>0 && row>r0 && row<r1 && col>c0 && col<c1) continue;
                    for(int city=head[row*W+col]; city!=-1; city=nxt[city])
                        if(!vis[city]){
                            float dx=qx-PX[city], dy=qy-PY[city], d=dx*dx+dy*dy;
                            if(d<bd){ bd=d; best=city; }
                        }
                }
        }
        return best;
    }

    void knn(int i, int k, vector<int>& result) const {
        float qx=PX[i], qy=PY[i];
        int qc=max(0,min(W-1,(int)((qx-ox)/cs)));
        int qr=max(0,min(H-1,(int)((qy-oy)/cs)));
        vector<pair<float,int>> heap; heap.reserve(k+1);
        float kd=1e30f;
        for(int r=0;;r++){
            float md=(float)max(0,r-1)*cs;
            if(md*md>kd && (int)heap.size()>=k) break;
            if(r>W+H+2) break;
            int r0=max(0,qr-r), r1=min(H-1,qr+r);
            int c0=max(0,qc-r), c1=min(W-1,qc+r);
            for(int row=r0;row<=r1;row++)
                for(int col=c0;col<=c1;col++){
                    if(r>0 && row>r0 && row<r1 && col>c0 && col<c1) continue;
                    for(int city=head[row*W+col]; city!=-1; city=nxt[city]){
                        if(city==i) continue;
                        float dx=qx-PX[city], dy=qy-PY[city], d=dx*dx+dy*dy;
                        if((int)heap.size()<k || d<kd){
                            heap.push_back({d,city});
                            push_heap(heap.begin(),heap.end());
                            if((int)heap.size()>k){ pop_heap(heap.begin(),heap.end()); heap.pop_back(); }
                            if((int)heap.size()>=k) kd=heap.front().first;
                        }
                    }
                }
        }
        sort(heap.begin(),heap.end());
        result.clear(); for(auto& p:heap) result.push_back(p.second);
    }
};

static float gW=0, gH=0;

// ---- Nearest-Neighbor tour ----
vector<int> nnTour(){
    float w=*max_element(PX.begin(),PX.end())-*min_element(PX.begin(),PX.end());
    float h=*max_element(PY.begin(),PY.end())-*min_element(PY.begin(),PY.end());
    gW=w; gH=h;
    float cs=sqrtf(max(w*h,1.0f)/(float)N)*1.5f+1e-6f;
    Grid g; g.build(cs);
    vector<int> tour; tour.reserve(N);
    vector<bool> vis(N,false);
    tour.push_back(0); vis[0]=true;
    for(int i=1;i<N;i++){
        int cur=tour.back();
        int nx=g.nn1(PX[cur],PY[cur],vis);
        if(nx<0) for(int j=0;j<N;j++) if(!vis[j]){ nx=j; break; }
        tour.push_back(nx); vis[nx]=true;
    }
    return tour;
}

// ---- Full 2-opt O(n^2) for small N ----
void twoOptFull(vector<int>& tour){
    int n=(int)tour.size(); bool imp=true;
    while(imp && elapsed()<TL){
        imp=false;
        for(int i=0;i<n-1 && elapsed()<TL;i++){
            for(int j=i+2;j<n;j++){
                if(i==0 && j==n-1) continue;
                float d1=D(tour[i],tour[i+1])+D(tour[j],tour[(j+1)%n]);
                float d2=D(tour[i],tour[j])+D(tour[i+1],tour[(j+1)%n]);
                if(d2<d1-1e-6f){
                    reverse(tour.begin()+i+1,tour.begin()+j+1);
                    imp=true;
                }
            }
        }
    }
}

// ---- 2-opt with KNN candidate lists ----
void twoOptKNN(vector<int>& tour, const vector<vector<int>>& knn){
    int n=(int)tour.size();
    vector<int> pos(N); for(int i=0;i<n;i++) pos[tour[i]]=i;
    bool imp=true;
    while(imp && elapsed()<TL){
        imp=false;
        for(int i=0;i<n && elapsed()<TL;i++){
            int a=tour[i], b=tour[(i+1)%n];
            float dab=D(a,b);
            for(int c : knn[a]){
                if(D(a,c) >= dab) break; // knn sorted ascending; no improvement possible
                int j=pos[c];
                if(j<=i+1) continue;
                int d=tour[(j+1)%n];
                float gain=dab+D(c,d)-D(a,c)-D(b,d);
                if(gain>1e-6f){
                    reverse(tour.begin()+i+1, tour.begin()+j+1);
                    for(int k2=i+1;k2<=j;k2++) pos[tour[k2]]=k2;
                    b=tour[(i+1)%n]; dab=D(a,b);
                    imp=true; break;
                }
            }
        }
    }
}

// ---- Brute force for N <= 11 ----
vector<int> bruteForce(){
    vector<int> p(N-1); iota(p.begin(),p.end(),1);
    float best=1e30f; vector<int> bp=p;
    do {
        float l=D(0,p[0]);
        for(int i=0;i<(int)p.size()-1;i++) l+=D(p[i],p[i+1]);
        l+=D(p.back(),0);
        if(l<best){ best=l; bp=p; }
    } while(next_permutation(p.begin(),p.end()));
    vector<int> t={0}; for(int c:bp) t.push_back(c);
    return t;
}

// ---- Held-Karp DP for 12 <= N <= 20 ----
vector<int> heldKarp(){
    int n=N, S=1<<n;
    vector<float> dp((long long)S*n, 1e30f);
    vector<int> par((long long)S*n, -1);
    dp[1*n+0]=0.0f;
    for(int mask=1;mask<S;mask++){
        if(!(mask&1)) continue;
        for(int u=0;u<n;u++){
            if(!(mask>>u&1)) continue;
            float du=dp[(long long)mask*n+u];
            if(du>=1e29f) continue;
            for(int v=0;v<n;v++){
                if(mask>>v&1) continue;
                int nm=mask|(1<<v);
                float nd=du+D(u,v);
                long long idx=(long long)nm*n+v;
                if(nd<dp[idx]){ dp[idx]=nd; par[idx]=u; }
            }
        }
    }
    int full=S-1; float best=1e30f; int last=-1;
    for(int i=1;i<n;i++){
        float d=dp[(long long)full*n+i]+D(i,0);
        if(d<best){ best=d; last=i; }
    }
    vector<int> t; int mask=full, cur=last;
    while(cur!=-1){
        t.push_back(cur);
        int p2=par[(long long)mask*n+cur];
        mask^=(1<<cur); cur=p2;
    }
    reverse(t.begin(),t.end());
    return t;
}

int main(){
    N=ri();
    PX.resize(N); PY.resize(N);
    for(int i=0;i<N;i++){ PX[i]=rf(); PY[i]=rf(); }

    vector<int> tour;

    if(N==1){ wi(0); wc('\n'); oflush(); return 0; }
    if(N==2){ wi(0); wc(' '); wi(1); wc('\n'); oflush(); return 0; }

    if(N<=11){
        tour=bruteForce();
    } else if(N<=20){
        tour=heldKarp();
    } else {
        tour=nnTour();
        double t1=elapsed();

        if(N<=5000 && t1<TL-5){
            twoOptFull(tour);
        } else if(N<=1000000 && t1<TL-20){
            float cs=sqrtf(max(gW*gH,1.0f)/(float)N)*1.5f+1e-6f;
            Grid g; g.build(cs);
            int k = (N<=100000) ? 20 : (N<=500000 ? 12 : 7);
            k=min(k,N-1);
            vector<vector<int>> knn_v(N);
            double knn_dl=min(TL-10.0, t1+min(60.0, (TL-t1)*0.3));
            for(int i=0;i<N && elapsed()<knn_dl;i++) g.knn(i,k,knn_v[i]);
            if(elapsed()<TL-10) twoOptKNN(tour, knn_v);
        }
        // For N > 1M: NN tour only
    }

    for(int i=0;i<N;i++){ if(i) wc(' '); wi(tour[i]); }
    wc('\n'); oflush();
    return 0;
}
