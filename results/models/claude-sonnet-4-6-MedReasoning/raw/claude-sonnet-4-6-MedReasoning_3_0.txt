use std::collections::VecDeque;
use std::io::{self, BufRead, Write};
use std::time::Instant;

fn main() {
    let start = Instant::now();
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_ascii_whitespace();
    let n: usize = iter.next().unwrap_or("0").parse().unwrap_or(0);
    let m: usize = iter.next().unwrap_or("0").parse().unwrap_or(0);
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut it = line.split_ascii_whitespace();
            let u = it.next().unwrap_or("0").parse::<usize>().unwrap_or(0);
            let v = it.next().unwrap_or("0").parse::<usize>().unwrap_or(0);
            if u < n && v < n && u != v {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }
    let positions = compute_layout(n, &adj, start);
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    for (x, y) in positions {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}

fn find_components(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    let mut comp = vec![usize::MAX; n];
    let mut c = 0usize;
    for start in 0..n {
        if comp[start] != usize::MAX { continue; }
        let mut queue = VecDeque::new();
        queue.push_back(start);
        comp[start] = c;
        while let Some(u) = queue.pop_front() {
            for &v in &adj[u] {
                if comp[v] == usize::MAX {
                    comp[v] = c;
                    queue.push_back(v);
                }
            }
        }
        c += 1;
    }
    comp
}

fn compute_layout(n: usize, adj: &[Vec<usize>], start: Instant) -> Vec<(f64, f64)> {
    if n == 0 { return vec![]; }
    if n == 1 { return vec![(0.0, 0.0)]; }
    let comp_id = find_components(n, adj);
    let num_comps = comp_id.iter().copied().max().unwrap_or(0) + 1;
    let mut comp_nodes: Vec<Vec<usize>> = vec![vec![]; num_comps];
    for i in 0..n { comp_nodes[comp_id[i]].push(i); }
    let mut global_pos = vec![(0.0f64, 0.0f64); n];
    let mut comp_layouts: Vec<Vec<(f64, f64)>> = Vec::with_capacity(num_comps);
    for nodes in comp_nodes.iter() {
        let sub_n = nodes.len();
        let mut node_to_local = vec![0usize; n];
        for (li, &gi) in nodes.iter().enumerate() { node_to_local[gi] = li; }
        let mut sub_adj: Vec<Vec<usize>> = vec![vec![]; sub_n];
        for &gi in nodes {
            let li = node_to_local[gi];
            for &gj in &adj[gi] { sub_adj[li].push(node_to_local[gj]); }
        }
        let elapsed = start.elapsed().as_millis();
        let total_budget = 24_000u128;
        let remaining = total_budget.saturating_sub(elapsed);
        let comp_budget = if num_comps > 1 {
            (remaining * sub_n as u128 / n as u128).max(500)
        } else { remaining };
        let deadline = start.elapsed().as_millis() + comp_budget;
        comp_layouts.push(layout_component(sub_n, &sub_adj, deadline, start));
    }
    let mut offset_x = 0.0f64;
    let mut row_offset_y = 0.0f64;
    let mut max_height = 0.0f64;
    let row_limit = (n as f64).sqrt() * 20.0;
    let mut row_width = 0.0f64;
    let mut order: Vec<usize> = (0..num_comps).collect();
    order.sort_by_key(|&ci| std::cmp::Reverse(comp_nodes[ci].len()));
    for ci in order {
        let nodes = &comp_nodes[ci];
        let sub_pos = &comp_layouts[ci];
        if sub_pos.is_empty() { continue; }
        let min_x = sub_pos.iter().map(|&(x,_)| x).fold(f64::INFINITY, f64::min);
        let max_x = sub_pos.iter().map(|&(x,_)| x).fold(f64::NEG_INFINITY, f64::max);
        let min_y = sub_pos.iter().map(|&(_,y)| y).fold(f64::INFINITY, f64::min);
        let max_y = sub_pos.iter().map(|&(_,y)| y).fold(f64::NEG_INFINITY, f64::max);
        let w = (max_x - min_x).max(1.0);
        let h = (max_y - min_y).max(1.0);
        let pad = (w * 0.2).max(5.0);
        if row_width > 0.0 && row_width + w > row_limit {
            row_offset_y += max_height + 20.0;
            offset_x = 0.0; row_width = 0.0; max_height = 0.0;
        }
        for (li, &gi) in nodes.iter().enumerate() {
            global_pos[gi] = (sub_pos[li].0 - min_x + offset_x, sub_pos[li].1 - min_y + row_offset_y);
        }
        offset_x += w + pad;
        row_width += w + pad;
        if h > max_height { max_height = h; }
    }
    global_pos
}

fn layout_component(n: usize, adj: &[Vec<usize>], deadline_ms: u128, start: Instant) -> Vec<(f64, f64)> {
    if n == 1 { return vec![(0.0, 0.0)]; }
    if n == 2 { return vec![(0.0, 0.0), (10.0, 0.0)]; }
    if n <= 600 { force_directed_full(n, adj, deadline_ms, start) }
    else if n <= 10000 { force_directed_grid(n, adj, deadline_ms, start) }
    else { bfs_radial(n, adj) }
}

fn force_directed_full(n: usize, adj: &[Vec<usize>], deadline_ms: u128, start: Instant) -> Vec<(f64, f64)> {
    use std::f64::consts::PI;
    let mut px = vec![0.0f64; n];
    let mut py = vec![0.0f64; n];
    for i in 0..n {
        let a = 2.0 * PI * i as f64 / n as f64;
        let r = (n as f64).sqrt() * 5.0 + 5.0;
        px[i] = r * a.cos(); py[i] = r * a.sin();
    }
    let k = 5.0 * (n as f64).sqrt().max(1.0);
    let max_iter = if n <= 50 { 500 } else if n <= 150 { 300 } else { 180 };
    let mut temp = k * 3.0;
    let cooling = temp / max_iter as f64;
    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];
    for iter in 0..max_iter {
        if iter % 10 == 0 && start.elapsed().as_millis() >= deadline_ms { break; }
        for i in 0..n { dx[i] = 0.0; dy[i] = 0.0; }
        for i in 0..n {
            for j in (i+1)..n {
                let ddx = px[i]-px[j]; let ddy = py[i]-py[j];
                let dist = (ddx*ddx+ddy*ddy).sqrt().max(0.1);
                let f = k*k/dist;
                let fx = ddx/dist*f; let fy = ddy/dist*f;
                dx[i] += fx; dy[i] += fy; dx[j] -= fx; dy[j] -= fy;
            }
        }
        for i in 0..n {
            for &j in &adj[i] {
                if i < j {
                    let ddx = px[i]-px[j]; let ddy = py[i]-py[j];
                    let dist = (ddx*ddx+ddy*ddy).sqrt().max(0.1);
                    let f = dist*dist/k;
                    let fx = ddx/dist*f; let fy = ddy/dist*f;
                    dx[i] -= fx; dy[i] -= fy; dx[j] += fx; dy[j] += fy;
                }
            }
        }
        for i in 0..n {
            let d = (dx[i]*dx[i]+dy[i]*dy[i]).sqrt().max(0.001);
            let scale = d.min(temp)/d;
            px[i] += dx[i]*scale; py[i] += dy[i]*scale;
        }
        temp = (temp - cooling).max(0.1);
    }
    px.into_iter().zip(py).collect()
}

fn force_directed_grid(n: usize, adj: &[Vec<usize>], deadline_ms: u128, start: Instant) -> Vec<(f64, f64)> {
    let init = bfs_radial(n, adj);
    let mut px: Vec<f64> = init.iter().map(|p| p.0).collect();
    let mut py: Vec<f64> = init.iter().map(|p| p.1).collect();
    let k = 5.0 * (n as f64).sqrt().max(1.0);
    let max_iter = if n <= 2000 { 80 } else if n <= 5000 { 50 } else { 30 };
    let mut temp = k * 2.0;
    let cooling = temp / max_iter as f64;
    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];
    for iter in 0..max_iter {
        if iter % 5 == 0 && start.elapsed().as_millis() >= deadline_ms { break; }
        for i in 0..n { dx[i] = 0.0; dy[i] = 0.0; }
        let min_x = px.iter().copied().fold(f64::INFINITY, f64::min);
        let min_y = py.iter().copied().fold(f64::INFINITY, f64::min);
        let max_x = px.iter().copied().fold(f64::NEG_INFINITY, f64::max);
        let max_y = py.iter().copied().fold(f64::NEG_INFINITY, f64::max);
        let range_x = (max_x - min_x).max(1.0);
        let range_y = (max_y - min_y).max(1.0);
        let gw = ((n as f64).sqrt() as usize + 1).min(200).max(2);
        let gh = ((n as f64).sqrt() as usize + 1).min(200).max(2);
        let cell_w = range_x / gw as f64;
        let cell_h = range_y / gh as f64;
        let mut grid: Vec<Vec<usize>> = vec![vec![]; gw * gh];
        for i in 0..n {
            let cx = ((px[i]-min_x)/cell_w) as usize;
            let cy = ((py[i]-min_y)/cell_h) as usize;
            grid[cy.min(gh-1)*gw + cx.min(gw-1)].push(i);
        }
        for i in 0..n {
            let cx = ((px[i]-min_x)/cell_w) as usize;
            let cy = ((py[i]-min_y)/cell_h) as usize;
            let cx = cx.min(gw-1); let cy = cy.min(gh-1);
            let x0 = if cx>0{cx-1}else{0}; let x1 = (cx+2).min(gw);
            let y0 = if cy>0{cy-1}else{0}; let y1 = (cy+2).min(gh);
            for gy in y0..y1 {
                for gx in x0..x1 {
                    for &j in &grid[gy*gw+gx] {
                        if i==j { continue; }
                        let ddx = px[i]-px[j]; let ddy = py[i]-py[j];
                        let dist = (ddx*ddx+ddy*ddy).sqrt().max(0.1);
                        let f = k*k/dist;
                        dx[i] += ddx/dist*f; dy[i] += ddy/dist*f;
                    }
                }
            }
        }
        for i in 0..n {
            for &j in &adj[i] {
                if i < j {
                    let ddx = px[i]-px[j]; let ddy = py[i]-py[j];
                    let dist = (ddx*ddx+ddy*ddy).sqrt().max(0.1);
                    let f = dist*dist/k;
                    let fx = ddx/dist*f; let fy = ddy/dist*f;
                    dx[i] -= fx; dy[i] -= fy; dx[j] += fx; dy[j] += fy;
                }
            }
        }
        for i in 0..n {
            let d = (dx[i]*dx[i]+dy[i]*dy[i]).sqrt().max(0.001);
            let scale = d.min(temp)/d;
            px[i] += dx[i]*scale; py[i] += dy[i]*scale;
        }
        temp = (temp - cooling).max(0.1);
    }
    px.into_iter().zip(py).collect()
}

fn bfs_radial(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    use std::f64::consts::PI;
    let root = (0..n).max_by_key(|&i| adj[i].len()).unwrap_or(0);
    let mut dist = vec![usize::MAX; n];
    let mut queue = VecDeque::new();
    dist[root] = 0;
    queue.push_back(root);
    let mut order = Vec::with_capacity(n);
    while let Some(u) = queue.pop_front() {
        order.push(u);
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                queue.push_back(v);
            }
        }
    }
    for i in 0..n {
        if dist[i] == usize::MAX { dist[i] = 0; order.push(i); }
    }
    let max_dist = dist.iter().copied().max().unwrap_or(0);
    let mut layers: Vec<Vec<usize>> = vec![vec![]; max_dist + 1];
    for &u in &order {
        if dist[u] <= max_dist { layers[dist[u]].push(u); }
    }
    let spacing = (n as f64).sqrt().max(5.0) * 2.5;
    let mut pos = vec![(0.0f64, 0.0f64); n];
    for (d, layer) in layers.iter().enumerate() {
        let r = d as f64 * spacing;
        let count = layer.len();
        for (i, &u) in layer.iter().enumerate() {
            if d == 0 { pos[u] = (0.0, 0.0); }
            else {
                let angle = 2.0 * PI * i as f64 / count as f64;
                pos[u] = (r * angle.cos(), r * angle.sin());
            }
        }
    }
    pos
}
