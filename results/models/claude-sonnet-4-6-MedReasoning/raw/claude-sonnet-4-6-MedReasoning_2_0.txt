using System;
using System.Collections.Generic;

class CPP
{
    static int N, M;
    static int[] eu, ev, ew, mult;
    static List<(int v, int w, int eidx)>[] adj;

    static void Main()
    {
        var tokens = Console.In.ReadToEnd()
            .Split(new char[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int pos = 0;
        N = int.Parse(tokens[pos++]);
        M = int.Parse(tokens[pos++]);

        eu = new int[M]; ev = new int[M]; ew = new int[M];
        mult = new int[M];
        adj = new List<(int, int, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int, int)>();

        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            eu[i] = int.Parse(tokens[pos++]);
            ev[i] = int.Parse(tokens[pos++]);
            ew[i] = int.Parse(tokens[pos++]);
            mult[i] = 1;
            adj[eu[i]].Add((ev[i], ew[i], i));
            adj[ev[i]].Add((eu[i], ew[i], i));
            deg[eu[i]]++; deg[ev[i]]++;
        }

        var odd = new List<int>();
        for (int i = 0; i < N; i++)
            if (deg[i] % 2 == 1) odd.Add(i);

        int k = odd.Count;

        if (k > 0)
        {
            if (k <= 600)
            {
                long[,] pd = new long[k, k];
                int[][] prevEdge = new int[k][];
                for (int i = 0; i < k; i++)
                {
                    var (d, pe) = Dijkstra(odd[i]);
                    prevEdge[i] = pe;
                    for (int j = 0; j < k; j++)
                        pd[i, j] = d[odd[j]];
                }

                int[] matching = (k <= 20) ? DPMatch(pd, k) : GreedyMatch(pd, k);

                for (int i = 0; i < k; i++)
                    if (matching[i] > i)
                        AddPath(odd[matching[i]], prevEdge[i]);
            }
            else
            {
                SpanningTreeParity(deg);
            }
        }

        var circuit = EulerianCircuit();

        using var stdout = new System.IO.StreamWriter(
            Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 16);
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) stdout.Write(' ');
            stdout.Write(circuit[i]);
        }
        stdout.WriteLine();
    }

    static (long[] dist, int[] prevEdge) Dijkstra(int src)
    {
        var dist = new long[N];
        var prevEdge = new int[N];
        Array.Fill(dist, long.MaxValue / 2);
        Array.Fill(prevEdge, -1);
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.TryDequeue(out int u, out long d))
        {
            if (d > dist[u]) continue;
            foreach (var (v, w, eidx) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    prevEdge[v] = eidx;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, prevEdge);
    }

    static void AddPath(int dest, int[] prevEdge)
    {
        int v = dest;
        while (prevEdge[v] >= 0)
        {
            int e = prevEdge[v];
            mult[e]++;
            v = (ev[e] == v) ? eu[e] : ev[e];
        }
    }

    static void SpanningTreeParity(int[] deg)
    {
        int[] parent = new int[N];
        int[] parentEdge = new int[N];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        var bfsOrder = new List<int>(N);
        var visited = new bool[N];
        var queue = new Queue<int>();
        queue.Enqueue(0);
        visited[0] = true;

        while (queue.Count > 0)
        {
            int u = queue.Dequeue();
            bfsOrder.Add(u);
            foreach (var (v, w, eidx) in adj[u])
            {
                if (!visited[v])
                {
                    visited[v] = true;
                    parent[v] = u;
                    parentEdge[v] = eidx;
                    queue.Enqueue(v);
                }
            }
        }

        int[] adjustedDeg = (int[])deg.Clone();

        for (int i = bfsOrder.Count - 1; i >= 1; i--)
        {
            int v = bfsOrder[i];
            if (adjustedDeg[v] % 2 == 1)
            {
                int e = parentEdge[v];
                mult[e]++;
                adjustedDeg[v]++;
                adjustedDeg[parent[v]]++;
            }
        }
    }

    static int[] DPMatch(long[,] dist, int k)
    {
        int states = 1 << k;
        var dp = new long[states];
        var choice = new int[states];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;

        for (int mask = 0; mask < states; mask++)
        {
            if (dp[mask] >= long.MaxValue / 2) continue;
            int i = 0;
            while (i < k && (mask & (1 << i)) != 0) i++;
            if (i >= k) continue;
            for (int j = i + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int nm = mask | (1 << i) | (1 << j);
                long cost = dp[mask] + dist[i, j];
                if (cost < dp[nm])
                {
                    dp[nm] = cost;
                    choice[nm] = i * k + j;
                }
            }
        }

        var matching = new int[k];
        Array.Fill(matching, -1);
        int cur = states - 1;
        while (cur > 0)
        {
            int c = choice[cur];
            int a = c / k, b = c % k;
            matching[a] = b;
            matching[b] = a;
            cur ^= (1 << a) | (1 << b);
        }
        return matching;
    }

    static int[] GreedyMatch(long[,] dist, int k)
    {
        var pairs = new List<(long d, int i, int j)>(k * (k - 1) / 2);
        for (int i = 0; i < k; i++)
            for (int j = i + 1; j < k; j++)
                pairs.Add((dist[i, j], i, j));
        pairs.Sort((a, b) => a.d.CompareTo(b.d));

        var matching = new int[k];
        Array.Fill(matching, -1);
        var used = new bool[k];
        foreach (var (_, i, j) in pairs)
        {
            if (!used[i] && !used[j])
            {
                matching[i] = j;
                matching[j] = i;
                used[i] = used[j] = true;
            }
        }
        return matching;
    }

    static List<int> EulerianCircuit()
    {
        int totalHalfEdges = 0;
        for (int i = 0; i < M; i++) totalHalfEdges += 2 * mult[i];

        var hTo = new int[totalHalfEdges];
        var hMate = new int[totalHalfEdges];
        var hUsed = new bool[totalHalfEdges];
        var adj2 = new List<int>[N];
        for (int i = 0; i < N; i++) adj2[i] = new List<int>();

        int hBase = 0;
        for (int i = 0; i < M; i++)
        {
            for (int c = 0; c < mult[i]; c++)
            {
                int h1 = hBase + 2 * c;
                int h2 = hBase + 2 * c + 1;
                hTo[h1] = ev[i];
                hTo[h2] = eu[i];
                hMate[h1] = h2;
                hMate[h2] = h1;
                adj2[eu[i]].Add(h1);
                adj2[ev[i]].Add(h2);
            }
            hBase += 2 * mult[i];
        }

        var ptr = new int[N];
        var stack = new Stack<int>();
        var circuit = new List<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            while (ptr[u] < adj2[u].Count && hUsed[adj2[u][ptr[u]]])
                ptr[u]++;

            if (ptr[u] < adj2[u].Count)
            {
                int h = adj2[u][ptr[u]++];
                hUsed[h] = hUsed[hMate[h]] = true;
                stack.Push(hTo[h]);
            }
            else
            {
                circuit.Add(u);
                stack.Pop();
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
