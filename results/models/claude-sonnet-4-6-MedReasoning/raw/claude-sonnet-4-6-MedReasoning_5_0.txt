use std::io::{self, BufRead, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;

    let mut grid = vec![vec![false; width]; height]; // true = obstacle

    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();
        let x = coords[0];
        let y = coords[1];
        grid[y][x] = true;
    }

    let total_cells = width * height;
    let target_length = total_cells - obstacle_count;

    if grid[0][0] {
        println!("0");
        return;
    }

    if target_length == 1 {
        println!("1");
        println!("0 0");
        return;
    }

    // Check connectivity first
    let reachable = count_reachable(&grid, width, height);
    if reachable != target_length {
        println!("0");
        return;
    }

    let result = if target_length <= 400 {
        solve_backtrack(&grid, width, height, require_cycle, target_length)
    } else {
        solve_large(&grid, width, height, require_cycle, target_length)
    };

    match result {
        Some(path) => {
            let stdout = io::stdout();
            let mut out = io::BufWriter::new(stdout.lock());
            writeln!(out, "{}", path.len()).unwrap();
            for (x, y) in &path {
                writeln!(out, "{} {}", x, y).unwrap();
            }
        }
        None => println!("0"),
    }
}

fn count_reachable(grid: &Vec<Vec<bool>>, width: usize, height: usize) -> usize {
    let mut visited = vec![vec![false; width]; height];
    let mut queue = VecDeque::new();

    if grid[0][0] {
        return 0;
    }

    visited[0][0] = true;
    queue.push_back((0usize, 0usize));
    let mut count = 1;

    let dirs: &[(i32, i32)] = &[(0, 1), (0, -1), (1, 0), (-1, 0)];

    while let Some((x, y)) = queue.pop_front() {
        for (dx, dy) in dirs {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
                let nx = nx as usize;
                let ny = ny as usize;
                if !visited[ny][nx] && !grid[ny][nx] {
                    visited[ny][nx] = true;
                    queue.push_back((nx, ny));
                    count += 1;
                }
            }
        }
    }
    count
}

fn solve_backtrack(
    grid: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    require_cycle: bool,
    target_length: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![vec![false; width]; height];
    let mut path = Vec::with_capacity(target_length);

    visited[0][0] = true;
    path.push((0usize, 0usize));

    if backtrack(grid, &mut visited, &mut path, width, height, require_cycle, target_length, 0, 0) {
        Some(path)
    } else {
        None
    }
}

fn backtrack(
    grid: &Vec<Vec<bool>>,
    visited: &mut Vec<Vec<bool>>,
    path: &mut Vec<(usize, usize)>,
    width: usize,
    height: usize,
    require_cycle: bool,
    target_length: usize,
    x: usize,
    y: usize,
) -> bool {
    if path.len() == target_length {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            return (lx == 0 && ly == 1) || (lx == 1 && ly == 0);
        }
        return true;
    }

    let dirs: &[(i32, i32)] = &[(0, 1), (0, -1), (1, 0), (-1, 0)];

    let mut neighbors: Vec<(usize, usize, usize)> = Vec::new();
    for (dx, dy) in dirs {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !grid[ny][nx] {
                let degree = count_unvisited_neighbors(grid, visited, width, height, nx, ny);
                neighbors.push((nx, ny, degree));
            }
        }
    }

    neighbors.sort_by_key(|&(_, _, d)| d);

    for (nx, ny, _) in neighbors {
        visited[ny][nx] = true;
        path.push((nx, ny));

        let remaining = target_length - path.len();
        let do_connectivity_check = remaining > 1 && (remaining < 50 || path.len() % 10 == 0);

        let pruned = if do_connectivity_check {
            !is_still_solvable(grid, visited, width, height, nx, ny, remaining, require_cycle)
        } else {
            false
        };

        if !pruned {
            if backtrack(grid, visited, path, width, height, require_cycle, target_length, nx, ny) {
                return true;
            }
        }

        path.pop();
        visited[ny][nx] = false;
    }

    false
}

fn count_unvisited_neighbors(
    grid: &Vec<Vec<bool>>,
    visited: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    x: usize,
    y: usize,
) -> usize {
    let dirs: &[(i32, i32)] = &[(0, 1), (0, -1), (1, 0), (-1, 0)];
    let mut count = 0;
    for (dx, dy) in dirs {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !grid[ny][nx] {
                count += 1;
            }
        }
    }
    count
}

fn is_still_solvable(
    grid: &Vec<Vec<bool>>,
    visited: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    cur_x: usize,
    cur_y: usize,
    remaining: usize,
    require_cycle: bool,
) -> bool {
    let reachable = count_reachable_from(grid, visited, width, height, cur_x, cur_y);
    if reachable != remaining {
        return false;
    }

    if require_cycle && remaining > 1 {
        let dirs: &[(i32, i32)] = &[(0, 1), (0, -1), (1, 0), (-1, 0)];
        let has_unvisited_neighbor = dirs.iter().any(|(dx, dy)| {
            let nx = *dx;
            let ny = *dy;
            if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
                let nx = nx as usize;
                let ny = ny as usize;
                !visited[ny][nx] && !grid[ny][nx]
            } else {
                false
            }
        });
        if !has_unvisited_neighbor {
            return false;
        }
    }

    true
}

fn count_reachable_from(
    grid: &Vec<Vec<bool>>,
    visited: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    start_x: usize,
    start_y: usize,
) -> usize {
    let mut local_visited = vec![vec![false; width]; height];
    let mut queue = VecDeque::new();

    local_visited[start_y][start_x] = true;
    queue.push_back((start_x, start_y));
    let mut count = 0;

    let dirs: &[(i32, i32)] = &[(0, 1), (0, -1), (1, 0), (-1, 0)];

    while let Some((x, y)) = queue.pop_front() {
        count += 1;
        for (dx, dy) in dirs {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
                let nx = nx as usize;
                let ny = ny as usize;
                if !local_visited[ny][nx] && !visited[ny][nx] && !grid[ny][nx] {
                    local_visited[ny][nx] = true;
                    queue.push_back((nx, ny));
                }
            }
        }
    }
    count
}

fn solve_large(
    grid: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    require_cycle: bool,
    target_length: usize,
) -> Option<Vec<(usize, usize)>> {
    // Try snake pattern first (works well for obstacle-free or simple grids)
    if let Some(path) = try_snake_pattern(grid, width, height, require_cycle, target_length) {
        return Some(path);
    }

    // Fall back to backtracking with Warnsdorff's
    solve_backtrack_large(grid, width, height, require_cycle, target_length)
}

fn try_snake_pattern(
    grid: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    require_cycle: bool,
    target_length: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::with_capacity(target_length);

    for row in 0..height {
        if row % 2 == 0 {
            for col in 0..width {
                if !grid[row][col] {
                    path.push((col, row));
                }
            }
        } else {
            for col in (0..width).rev() {
                if !grid[row][col] {
                    path.push((col, row));
                }
            }
        }
    }

    if path.len() != target_length {
        return None;
    }

    // Validate connectivity
    for i in 1..path.len() {
        let (x1, y1) = path[i - 1];
        let (x2, y2) = path[i];
        let dx = (x1 as i32 - x2 as i32).abs();
        let dy = (y1 as i32 - y2 as i32).abs();
        if dx + dy != 1 {
            return None;
        }
    }

    if require_cycle {
        let (lx, ly) = path[path.len() - 1];
        if !((lx == 0 && ly == 1) || (lx == 1 && ly == 0)) {
            return None;
        }
    }

    Some(path)
}

fn solve_backtrack_large(
    grid: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    require_cycle: bool,
    target_length: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![vec![false; width]; height];
    let mut path = Vec::with_capacity(target_length);

    visited[0][0] = true;
    path.push((0usize, 0usize));

    if backtrack_large(grid, &mut visited, &mut path, width, height, require_cycle, target_length, 0, 0) {
        Some(path)
    } else {
        None
    }
}

fn backtrack_large(
    grid: &Vec<Vec<bool>>,
    visited: &mut Vec<Vec<bool>>,
    path: &mut Vec<(usize, usize)>,
    width: usize,
    height: usize,
    require_cycle: bool,
    target_length: usize,
    x: usize,
    y: usize,
) -> bool {
    if path.len() == target_length {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            return (lx == 0 && ly == 1) || (lx == 1 && ly == 0);
        }
        return true;
    }

    let dirs: &[(i32, i32)] = &[(0, 1), (0, -1), (1, 0), (-1, 0)];

    let mut neighbors: Vec<(usize, usize, usize)> = Vec::new();
    for (dx, dy) in dirs {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !grid[ny][nx] {
                let degree = count_unvisited_neighbors(grid, visited, width, height, nx, ny);
                neighbors.push((nx, ny, degree));
            }
        }
    }

    neighbors.sort_by_key(|&(_, _, d)| d);

    for (nx, ny, _) in neighbors {
        visited[ny][nx] = true;
        path.push((nx, ny));

        let remaining = target_length - path.len();
        let do_connectivity_check = remaining > 0 && path.len() % 20 == 0;

        let pruned = if do_connectivity_check {
            let reachable = count_reachable_from(grid, visited, width, height, nx, ny);
            reachable != remaining
        } else {
            false
        };

        if !pruned {
            if backtrack_large(grid, visited, path, width, height, require_cycle, target_length, nx, ny) {
                return true;
            }
        }

        path.pop();
        visited[ny][nx] = false;
    }

    false
}
