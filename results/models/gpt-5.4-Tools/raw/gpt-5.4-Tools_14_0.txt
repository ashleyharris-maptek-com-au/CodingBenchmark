using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private static readonly int[] Dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] Dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int height = reader.NextInt();
        int width = reader.NextInt();

        var board = new char[height][];
        for (int y = 0; y < height; y++)
        {
            string line = reader.ReadLineExact(width);
            board[y] = line.ToCharArray();
        }

        var solver = new Solver(board, width, height);
        List<Cell> safeCells = solver.Solve();

        using var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            writer.Write(cell.X);
            writer.Write(' ');
            writer.WriteLine(cell.Y);
        }
    }

    private readonly struct Cell
    {
        public readonly int X;
        public readonly int Y;

        public Cell(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private sealed class Solver
    {
        private readonly char[][] _board;
        private readonly int _width;
        private readonly int _height;
        private readonly bool[,] _safe;
        private readonly Queue<int> _dirtyNumbers;
        private readonly bool[] _queuedNumbers;
        private readonly List<int> _frontierNumbers;
        private readonly int _cellCount;

        public Solver(char[][] board, int width, int height)
        {
            _board = board;
            _width = width;
            _height = height;
            _safe = new bool[height, width];
            _cellCount = width * height;
            _dirtyNumbers = new Queue<int>();
            _queuedNumbers = new bool[_cellCount];
            _frontierNumbers = new List<int>();
        }

        public List<Cell> Solve()
        {
            InitializeFrontier();
            PropagateLocal();
            RunAdaptiveGlobalReasoning();
            return CollectSafeCells();
        }

        private void InitializeFrontier()
        {
            for (int y = 0; y < _height; y++)
            {
                for (int x = 0; x < _width; x++)
                {
                    if (IsNumber(_board[y][x]) && HasUnknownNeighbor(x, y))
                    {
                        int id = Id(x, y);
                        _frontierNumbers.Add(id);
                        EnqueueNumber(id);
                    }
                }
            }
        }

        private void PropagateLocal()
        {
            while (_dirtyNumbers.Count > 0)
            {
                int id = _dirtyNumbers.Dequeue();
                _queuedNumbers[id] = false;
                int x = id % _width;
                int y = id / _width;

                int required = _board[y][x] - '0';
                int flagged = 0;
                int unknownCount = 0;
                int[] unknowns = new int[8];

                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + Dx[dir];
                    int ny = y + Dy[dir];
                    if (!InBounds(nx, ny))
                    {
                        continue;
                    }

                    char c = _board[ny][nx];
                    if (c == '*')
                    {
                        flagged++;
                    }
                    else if (c == ' ')
                    {
                        unknowns[unknownCount++] = Id(nx, ny);
                    }
                }

                int remaining = required - flagged;
                if (unknownCount == 0 || remaining < 0 || remaining > unknownCount)
                {
                    continue;
                }

                if (remaining == 0)
                {
                    for (int i = 0; i < unknownCount; i++)
                    {
                        MarkSafe(unknowns[i]);
                    }
                }
            }
        }

        private void RunAdaptiveGlobalReasoning()
        {
            var frontierUnknowns = BuildFrontierUnknowns();
            if (frontierUnknowns.Count == 0)
            {
                return;
            }

            long area = (long)_width * _height;
            int frontierCellCount = frontierUnknowns.Count;
            int frontierNumberCount = _frontierNumbers.Count;

            if (frontierCellCount <= 22)
            {
                SolveSingleExact(frontierUnknowns, frontierNumberCount <= 128 ? 2_000_000 : 500_000);
                return;
            }

            if (area <= 2_000_000 && frontierCellCount <= 2000)
            {
                SolveByComponents(frontierUnknowns, 24, 250_000);
            }
        }

        private HashSet<int> BuildFrontierUnknowns()
        {
            var frontierUnknowns = new HashSet<int>();
            foreach (int numberId in _frontierNumbers)
            {
                int x = numberId % _width;
                int y = numberId / _width;
                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + Dx[dir];
                    int ny = y + Dy[dir];
                    if (InBounds(nx, ny) && _board[ny][nx] == ' ')
                    {
                        frontierUnknowns.Add(Id(nx, ny));
                    }
                }
            }
            return frontierUnknowns;
        }

        private void SolveSingleExact(HashSet<int> frontierUnknowns, int maxStates)
        {
            var unknownList = new List<int>(frontierUnknowns);
            var unknownIndex = new Dictionary<int, int>(unknownList.Count);
            for (int i = 0; i < unknownList.Count; i++)
            {
                unknownIndex[unknownList[i]] = i;
            }

            List<Constraint> constraints = BuildConstraints(unknownIndex);
            if (constraints.Count == 0)
            {
                return;
            }

            ExactEnumerator.MarkAlwaysSafe(unknownList.Count, constraints, maxStates, idx => MarkSafe(unknownList[idx]));
        }

        private void SolveByComponents(HashSet<int> frontierUnknowns, int componentLimit, int maxStatesPerComponent)
        {
            var cellToNumbers = new Dictionary<int, List<int>>(frontierUnknowns.Count);
            foreach (int cellId in frontierUnknowns)
            {
                cellToNumbers[cellId] = new List<int>(4);
            }

            for (int i = 0; i < _frontierNumbers.Count; i++)
            {
                int numberId = _frontierNumbers[i];
                int x = numberId % _width;
                int y = numberId / _width;
                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + Dx[dir];
                    int ny = y + Dy[dir];
                    if (!InBounds(nx, ny) || _board[ny][nx] != ' ')
                    {
                        continue;
                    }

                    int cellId = Id(nx, ny);
                    if (cellToNumbers.TryGetValue(cellId, out List<int>? list))
                    {
                        list.Add(i);
                    }
                }
            }

            var visited = new HashSet<int>();
            foreach (int startCell in frontierUnknowns)
            {
                if (!visited.Add(startCell))
                {
                    continue;
                }

                var componentCells = new List<int>();
                var componentNumbers = new HashSet<int>();
                var queue = new Queue<int>();
                queue.Enqueue(startCell);

                while (queue.Count > 0)
                {
                    int cellId = queue.Dequeue();
                    componentCells.Add(cellId);

                    foreach (int numberIndex in cellToNumbers[cellId])
                    {
                        if (!componentNumbers.Add(numberIndex))
                        {
                            continue;
                        }

                        int numberId = _frontierNumbers[numberIndex];
                        int x = numberId % _width;
                        int y = numberId / _width;
                        for (int dir = 0; dir < 8; dir++)
                        {
                            int nx = x + Dx[dir];
                            int ny = y + Dy[dir];
                            if (!InBounds(nx, ny) || _board[ny][nx] != ' ')
                            {
                                continue;
                            }

                            int neighborCell = Id(nx, ny);
                            if (frontierUnknowns.Contains(neighborCell) && visited.Add(neighborCell))
                            {
                                queue.Enqueue(neighborCell);
                            }
                        }
                    }
                }

                if (componentCells.Count > componentLimit)
                {
                    continue;
                }

                var unknownIndex = new Dictionary<int, int>(componentCells.Count);
                for (int i = 0; i < componentCells.Count; i++)
                {
                    unknownIndex[componentCells[i]] = i;
                }

                var constraints = new List<Constraint>(componentNumbers.Count);
                foreach (int numberIndex in componentNumbers)
                {
                    int numberId = _frontierNumbers[numberIndex];
                    int x = numberId % _width;
                    int y = numberId / _width;
                    int required = _board[y][x] - '0';
                    int flagged = 0;
                    var vars = new List<int>(8);

                    for (int dir = 0; dir < 8; dir++)
                    {
                        int nx = x + Dx[dir];
                        int ny = y + Dy[dir];
                        if (!InBounds(nx, ny))
                        {
                            continue;
                        }

                        char c = _board[ny][nx];
                        if (c == '*')
                        {
                            flagged++;
                        }
                        else if (c == ' ')
                        {
                            int id = Id(nx, ny);
                            if (unknownIndex.TryGetValue(id, out int idx))
                            {
                                vars.Add(idx);
                            }
                        }
                    }

                    int remaining = required - flagged;
                    if (vars.Count > 0 && remaining >= 0 && remaining <= vars.Count)
                    {
                        constraints.Add(new Constraint(vars.ToArray(), remaining));
                    }
                }

                if (constraints.Count == 0)
                {
                    continue;
                }

                ExactEnumerator.MarkAlwaysSafe(componentCells.Count, constraints, maxStatesPerComponent, idx => MarkSafe(componentCells[idx]));
            }
        }

        private List<Constraint> BuildConstraints(Dictionary<int, int> unknownIndex)
        {
            var constraints = new List<Constraint>(_frontierNumbers.Count);
            foreach (int numberId in _frontierNumbers)
            {
                int x = numberId % _width;
                int y = numberId / _width;
                int required = _board[y][x] - '0';
                int flagged = 0;
                var vars = new List<int>(8);

                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + Dx[dir];
                    int ny = y + Dy[dir];
                    if (!InBounds(nx, ny))
                    {
                        continue;
                    }

                    char c = _board[ny][nx];
                    if (c == '*')
                    {
                        flagged++;
                    }
                    else if (c == ' ')
                    {
                        int id = Id(nx, ny);
                        if (unknownIndex.TryGetValue(id, out int idx))
                        {
                            vars.Add(idx);
                        }
                    }
                }

                int remaining = required - flagged;
                if (vars.Count > 0 && remaining >= 0 && remaining <= vars.Count)
                {
                    constraints.Add(new Constraint(vars.ToArray(), remaining));
                }
            }

            return constraints;
        }

        private List<Cell> CollectSafeCells()
        {
            var result = new List<Cell>();
            for (int y = 0; y < _height; y++)
            {
                for (int x = 0; x < _width; x++)
                {
                    if (_safe[y, x] && _board[y][x] == ' ')
                    {
                        result.Add(new Cell(x, y));
                    }
                }
            }
            return result;
        }

        private void MarkSafe(int id)
        {
            int x = id % _width;
            int y = id / _width;
            if (_safe[y, x] || _board[y][x] != ' ')
            {
                return;
            }

            _safe[y, x] = true;
            for (int dir = 0; dir < 8; dir++)
            {
                int nx = x + Dx[dir];
                int ny = y + Dy[dir];
                if (InBounds(nx, ny) && IsNumber(_board[ny][nx]))
                {
                    EnqueueNumber(Id(nx, ny));
                }
            }
        }

        private void EnqueueNumber(int id)
        {
            if (_queuedNumbers[id])
            {
                return;
            }

            _queuedNumbers[id] = true;
            _dirtyNumbers.Enqueue(id);
        }

        private bool HasUnknownNeighbor(int x, int y)
        {
            for (int dir = 0; dir < 8; dir++)
            {
                int nx = x + Dx[dir];
                int ny = y + Dy[dir];
                if (InBounds(nx, ny) && _board[ny][nx] == ' ')
                {
                    return true;
                }
            }
            return false;
        }

        private static bool IsNumber(char c) => c >= '0' && c <= '8';

        private bool InBounds(int x, int y) => (uint)x < (uint)_width && (uint)y < (uint)_height;

        private int Id(int x, int y) => y * _width + x;
    }

    private readonly struct Constraint
    {
        public readonly int[] Variables;
        public readonly int Sum;

        public Constraint(int[] variables, int sum)
        {
            Variables = variables;
            Sum = sum;
        }
    }

    private static class ExactEnumerator
    {
        public static void MarkAlwaysSafe(int variableCount, List<Constraint> constraints, int maxStates, Action<int> markSafe)
        {
            int constraintCount = constraints.Count;
            var varToConstraints = new List<int>[variableCount];
            var occurrenceCount = new int[variableCount];
            for (int i = 0; i < variableCount; i++)
            {
                varToConstraints[i] = new List<int>(4);
            }

            for (int ci = 0; ci < constraintCount; ci++)
            {
                foreach (int v in constraints[ci].Variables)
                {
                    varToConstraints[v].Add(ci);
                    occurrenceCount[v]++;
                }
            }

            var order = new int[variableCount];
            for (int i = 0; i < variableCount; i++)
            {
                order[i] = i;
            }
            Array.Sort(order, (a, b) => occurrenceCount[b].CompareTo(occurrenceCount[a]));

            var assigned = new sbyte[variableCount];
            Array.Fill(assigned, (sbyte)-1);

            var minesAssigned = new int[constraintCount];
            var unknownRemaining = new int[constraintCount];
            for (int ci = 0; ci < constraintCount; ci++)
            {
                unknownRemaining[ci] = constraints[ci].Variables.Length;
            }

            var alwaysSafe = new bool[variableCount];
            Array.Fill(alwaysSafe, true);
            bool foundAny = false;
            int states = 0;

            bool Dfs(int depth)
            {
                if (states >= maxStates)
                {
                    return false;
                }

                if (depth == variableCount)
                {
                    foundAny = true;
                    states++;
                    for (int i = 0; i < variableCount; i++)
                    {
                        if (assigned[i] != 0)
                        {
                            alwaysSafe[i] = false;
                        }
                    }
                    return true;
                }

                int v = order[depth];
                if (TryAssign(v, 0, constraints, varToConstraints, minesAssigned, unknownRemaining))
                {
                    assigned[v] = 0;
                    if (!Dfs(depth + 1))
                    {
                        UndoAssign(v, 0, varToConstraints, minesAssigned, unknownRemaining);
                        assigned[v] = -1;
                        return false;
                    }
                    UndoAssign(v, 0, varToConstraints, minesAssigned, unknownRemaining);
                    assigned[v] = -1;
                }

                if (TryAssign(v, 1, constraints, varToConstraints, minesAssigned, unknownRemaining))
                {
                    assigned[v] = 1;
                    if (!Dfs(depth + 1))
                    {
                        UndoAssign(v, 1, varToConstraints, minesAssigned, unknownRemaining);
                        assigned[v] = -1;
                        return false;
                    }
                    UndoAssign(v, 1, varToConstraints, minesAssigned, unknownRemaining);
                    assigned[v] = -1;
                }

                return true;
            }

            Dfs(0);
            if (!foundAny)
            {
                return;
            }

            for (int i = 0; i < variableCount; i++)
            {
                if (alwaysSafe[i])
                {
                    markSafe(i);
                }
            }
        }

        private static bool TryAssign(
            int variable,
            int value,
            List<Constraint> constraints,
            List<int>[] varToConstraints,
            int[] minesAssigned,
            int[] unknownRemaining)
        {
            foreach (int ci in varToConstraints[variable])
            {
                unknownRemaining[ci]--;
                minesAssigned[ci] += value;
                int required = constraints[ci].Sum;
                if (minesAssigned[ci] > required || minesAssigned[ci] + unknownRemaining[ci] < required)
                {
                    foreach (int rollbackCi in varToConstraints[variable])
                    {
                        if (rollbackCi == ci)
                        {
                            minesAssigned[rollbackCi] -= value;
                            unknownRemaining[rollbackCi]++;
                            break;
                        }
                        minesAssigned[rollbackCi] -= value;
                        unknownRemaining[rollbackCi]++;
                    }
                    return false;
                }
            }

            return true;
        }

        private static void UndoAssign(
            int variable,
            int value,
            List<int>[] varToConstraints,
            int[] minesAssigned,
            int[] unknownRemaining)
        {
            foreach (int ci in varToConstraints[variable])
            {
                minesAssigned[ci] -= value;
                unknownRemaining[ci]++;
            }
        }
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastReader(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public string ReadLineExact(int width)
        {
            int c = Read();
            while (c == '\r' || c == '\n')
            {
                c = Read();
            }

            var chars = new char[width];
            chars[0] = (char)c;
            for (int i = 1; i < width; i++)
            {
                chars[i] = (char)Read();
            }

            int end = Read();
            if (end == '\r')
            {
                int next = Read();
                if (next != '\n')
                {
                    _position--;
                }
            }
            else if (end != '\n' && end >= 0)
            {
                _position--;
            }

            return new string(chars);
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }
}
