using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private static int _width;
    private static int _height;
    private static char[] _board = Array.Empty<char>();
    private static readonly int[] Dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] Dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        _height = scanner.NextInt();
        _width = scanner.NextInt();
        _board = new char[_height * _width];

        for (int y = 0; y < _height; y++)
        {
            string row = scanner.NextFixedLine(_width);
            for (int x = 0; x < _width; x++)
            {
                _board[y * _width + x] = row[x];
            }
        }

        var safe = Solve();

        Console.WriteLine(safe.Count);
        foreach (int id in safe)
        {
            int x = id % _width;
            int y = id / _width;
            Console.WriteLine($"{x} {y}");
        }
    }

    private static List<int> Solve()
    {
        var safe = new HashSet<int>();
        if (_width == 0 || _height == 0)
        {
            return new List<int>();
        }

        var unknownToNumbers = new Dictionary<int, List<int>>();
        var numberedCells = new List<int>();

        for (int y = 0; y < _height; y++)
        {
            int rowBase = y * _width;
            for (int x = 0; x < _width; x++)
            {
                int id = rowBase + x;
                char c = _board[id];
                if (c < '0' || c > '8')
                {
                    continue;
                }

                numberedCells.Add(id);
                int flagged = 0;
                var unknowns = new List<int>(8);
                ForEachNeighbor(x, y, nid =>
                {
                    char nc = _board[nid];
                    if (nc == '*')
                    {
                        flagged++;
                    }
                    else if (nc == ' ')
                    {
                        unknowns.Add(nid);
                        if (!unknownToNumbers.TryGetValue(nid, out var owners))
                        {
                            owners = new List<int>(2);
                            unknownToNumbers[nid] = owners;
                        }
                        owners.Add(id);
                    }
                });

                int remaining = (c - '0') - flagged;
                if (remaining == 0)
                {
                    foreach (int uid in unknowns)
                    {
                        safe.Add(uid);
                    }
                }
            }
        }

        if (unknownToNumbers.Count == 0)
        {
            return safe.OrderBy(v => v).ToList();
        }

        int frontierUnknowns = unknownToNumbers.Count;
        bool hugeBoard = (long)_width * _height >= 4_000_000L;
        bool useExact = frontierUnknowns <= 20000 && (!hugeBoard || frontierUnknowns <= 4000);

        if (useExact)
        {
            foreach (var component in BuildComponents(unknownToNumbers))
            {
                if (component.Unknowns.Count == 0)
                {
                    continue;
                }

                if (component.Unknowns.Count <= 22 && component.Constraints.Count <= 80)
                {
                    SolveComponentExactly(component, safe);
                }
                else
                {
                    SolveComponentWithRules(component, safe);
                }
            }
        }
        else
        {
            foreach (var component in BuildComponents(unknownToNumbers))
            {
                SolveComponentWithRules(component, safe);
            }
        }

        return safe.OrderBy(v => v).ToList();
    }

    private static List<Component> BuildComponents(Dictionary<int, List<int>> unknownToNumbers)
    {
        var result = new List<Component>();
        var visitedUnknown = new HashSet<int>();
        var visitedNumbers = new HashSet<int>();

        foreach (var start in unknownToNumbers.Keys)
        {
            if (!visitedUnknown.Add(start))
            {
                continue;
            }

            var qUnknown = new Queue<int>();
            var qNumber = new Queue<int>();
            var unknowns = new List<int>();
            var numbers = new List<int>();

            qUnknown.Enqueue(start);
            while (qUnknown.Count > 0 || qNumber.Count > 0)
            {
                while (qUnknown.Count > 0)
                {
                    int uid = qUnknown.Dequeue();
                    unknowns.Add(uid);
                    foreach (int nid in unknownToNumbers[uid])
                    {
                        if (visitedNumbers.Add(nid))
                        {
                            qNumber.Enqueue(nid);
                        }
                    }
                }

                while (qNumber.Count > 0)
                {
                    int nid = qNumber.Dequeue();
                    numbers.Add(nid);
                    int x = nid % _width;
                    int y = nid / _width;
                    ForEachNeighbor(x, y, uid =>
                    {
                        if (_board[uid] == ' ' && unknownToNumbers.ContainsKey(uid) && visitedUnknown.Add(uid))
                        {
                            qUnknown.Enqueue(uid);
                        }
                    });
                }
            }

            var constraints = new List<Constraint>(numbers.Count);
            var indexOfUnknown = new Dictionary<int, int>(unknowns.Count);
            for (int i = 0; i < unknowns.Count; i++)
            {
                indexOfUnknown[unknowns[i]] = i;
            }

            foreach (int nid in numbers)
            {
                int x = nid % _width;
                int y = nid / _width;
                int flagged = 0;
                var vars = new List<int>(8);
                ForEachNeighbor(x, y, aid =>
                {
                    char c = _board[aid];
                    if (c == '*')
                    {
                        flagged++;
                    }
                    else if (c == ' ' && indexOfUnknown.TryGetValue(aid, out int vid))
                    {
                        vars.Add(vid);
                    }
                });

                int remaining = (_board[nid] - '0') - flagged;
                if (vars.Count > 0)
                {
                    constraints.Add(new Constraint(vars.ToArray(), remaining));
                }
            }

            result.Add(new Component(unknowns, constraints));
        }

        return result;
    }

    private static void SolveComponentWithRules(Component component, HashSet<int> safe)
    {
        var constraints = SimplifyConstraints(component.Constraints);
        bool changed;
        var knownSafe = new HashSet<int>();
        var knownMine = new HashSet<int>();

        do
        {
            changed = false;
            foreach (var c in constraints)
            {
                int undecided = 0;
                foreach (int v in c.Vars)
                {
                    if (!knownSafe.Contains(v) && !knownMine.Contains(v))
                    {
                        undecided++;
                    }
                }

                if (c.Remaining == 0)
                {
                    foreach (int v in c.Vars)
                    {
                        if (knownSafe.Add(v)) changed = true;
                    }
                }
                else if (c.Remaining == undecided)
                {
                    foreach (int v in c.Vars)
                    {
                        if (knownMine.Add(v)) changed = true;
                    }
                }
            }

            if (ApplySubsetRules(constraints, knownSafe, knownMine))
            {
                changed = true;
            }

            if (changed)
            {
                constraints = ReduceConstraints(constraints, knownSafe, knownMine);
            }
        }
        while (changed);

        foreach (int v in knownSafe)
        {
            safe.Add(component.Unknowns[v]);
        }
    }

    private static void SolveComponentExactly(Component component, HashSet<int> safe)
    {
        var constraints = SimplifyConstraints(component.Constraints);
        int n = component.Unknowns.Count;
        var memberships = new List<int>[n];
        for (int i = 0; i < n; i++) memberships[i] = new List<int>(4);
        for (int ci = 0; ci < constraints.Count; ci++)
        {
            foreach (int v in constraints[ci].Vars)
            {
                memberships[v].Add(ci);
            }
        }

        int[] assigned = new int[n];
        Array.Fill(assigned, -1);
        int[] need = new int[constraints.Count];
        int[] free = new int[constraints.Count];
        for (int i = 0; i < constraints.Count; i++)
        {
            need[i] = constraints[i].Remaining;
            free[i] = constraints[i].Vars.Length;
        }

        long solutions = 0;
        var mineCount = new long[n];
        var order = Enumerable.Range(0, n).OrderByDescending(v => memberships[v].Count).ToArray();

        void Dfs(int depth)
        {
            if (depth == n)
            {
                solutions++;
                for (int i = 0; i < n; i++)
                {
                    if (assigned[i] == 1) mineCount[i]++;
                }
                return;
            }

            int v = order[depth];
            foreach (int value in new[] { 0, 1 })
            {
                bool ok = true;
                foreach (int ci in memberships[v])
                {
                    free[ci]--;
                    if (value == 1) need[ci]--;
                    if (need[ci] < 0 || need[ci] > free[ci]) ok = false;
                }

                if (ok)
                {
                    assigned[v] = value;
                    Dfs(depth + 1);
                    assigned[v] = -1;
                }

                foreach (int ci in memberships[v])
                {
                    if (value == 1) need[ci]++;
                    free[ci]++;
                }
            }
        }

        Dfs(0);

        if (solutions == 0)
        {
            SolveComponentWithRules(component, safe);
            return;
        }

        for (int i = 0; i < n; i++)
        {
            if (mineCount[i] == 0)
            {
                safe.Add(component.Unknowns[i]);
            }
        }
    }

    private static List<Constraint> SimplifyConstraints(List<Constraint> constraints)
    {
        var map = new Dictionary<string, int>();
        var result = new List<Constraint>();
        foreach (var c in constraints)
        {
            var vars = c.Vars.Distinct().OrderBy(v => v).ToArray();
            string key = string.Join(",", vars) + "=" + c.Remaining;
            if (map.ContainsKey(key)) continue;
            map[key] = 1;
            result.Add(new Constraint(vars, c.Remaining));
        }
        return result;
    }

    private static List<Constraint> ReduceConstraints(List<Constraint> constraints, HashSet<int> knownSafe, HashSet<int> knownMine)
    {
        var reduced = new List<Constraint>();
        foreach (var c in constraints)
        {
            int remaining = c.Remaining;
            var vars = new List<int>(c.Vars.Length);
            foreach (int v in c.Vars)
            {
                if (knownMine.Contains(v))
                {
                    remaining--;
                }
                else if (!knownSafe.Contains(v))
                {
                    vars.Add(v);
                }
            }

            if (vars.Count > 0)
            {
                reduced.Add(new Constraint(vars.ToArray(), remaining));
            }
        }
        return SimplifyConstraints(reduced);
    }

    private static bool ApplySubsetRules(List<Constraint> constraints, HashSet<int> knownSafe, HashSet<int> knownMine)
    {
        bool changed = false;
        for (int i = 0; i < constraints.Count; i++)
        {
            var a = constraints[i];
            for (int j = 0; j < constraints.Count; j++)
            {
                if (i == j) continue;
                var b = constraints[j];
                if (a.Vars.Length > b.Vars.Length) continue;
                if (!IsSubset(a.Vars, b.Vars)) continue;

                int diffCount = b.Vars.Length - a.Vars.Length;
                int diffRemaining = b.Remaining - a.Remaining;
                if (diffCount == 0) continue;

                if (diffRemaining == 0)
                {
                    foreach (int v in Difference(b.Vars, a.Vars))
                    {
                        if (knownSafe.Add(v)) changed = true;
                    }
                }
                else if (diffRemaining == diffCount)
                {
                    foreach (int v in Difference(b.Vars, a.Vars))
                    {
                        if (knownMine.Add(v)) changed = true;
                    }
                }
            }
        }
        return changed;
    }

    private static bool IsSubset(int[] small, int[] big)
    {
        int i = 0, j = 0;
        while (i < small.Length && j < big.Length)
        {
            if (small[i] == big[j]) { i++; j++; }
            else if (small[i] > big[j]) { j++; }
            else { return false; }
        }
        return i == small.Length;
    }

    private static IEnumerable<int> Difference(int[] big, int[] small)
    {
        int i = 0, j = 0;
        while (i < big.Length)
        {
            if (j >= small.Length || big[i] < small[j])
            {
                yield return big[i++];
            }
            else if (big[i] == small[j])
            {
                i++;
                j++;
            }
            else
            {
                j++;
            }
        }
    }

    private static void ForEachNeighbor(int x, int y, Action<int> action)
    {
        for (int k = 0; k < 8; k++)
        {
            int nx = x + Dx[k];
            int ny = y + Dy[k];
            if ((uint)nx < (uint)_width && (uint)ny < (uint)_height)
            {
                action(ny * _width + nx);
            }
        }
    }

    private readonly record struct Constraint(int[] Vars, int Remaining);
    private readonly record struct Component(List<int> Unknowns, List<Constraint> Constraints);

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }

        public string NextFixedLine(int width)
        {
            var chars = new char[width];
            for (int i = 0; i < width; i++)
            {
                int ch = Read();
                while (ch == '\r' || ch == '\n')
                {
                    ch = Read();
                }
                chars[i] = (char)ch;
            }
            int tail;
            do
            {
                tail = Read();
            } while (tail == '\r');
            if (tail != '\n' && tail >= 0)
            {
                _ptr--;
            }
            return new string(chars);
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }
            return _buffer[_ptr++];
        }
    }
}
