#include <cstdint>

using namespace std;

struct Task {
    int machine = 0;
    long long duration = 0;
};

struct ScheduledOp {
    int job = -1;
    int task = -1;
    long long start = 0;
    long long end = 0;
};

struct Candidate {
    int job = -1;
    int task = -1;
    int machine = -1;
    long long duration = 0;
    long long earliest = 0;
    long long priority = 0;
};

struct Problem {
    int jobs = 0;
    int machines = 0;
    vector<vector<Task>> job_tasks;
    vector<long long> job_total_duration;
    vector<long long> suffix_duration_sum;
    int total_tasks = 0;
};

static long long safe_add(long long a, long long b) {
    if (a > numeric_limits<long long>::max() - b) {
        return numeric_limits<long long>::max();
    }
    return a + b;
}

struct MachineCalendar {
    vector<pair<long long, long long>> busy;

    long long earliest_start(long long ready, long long duration) const {
        long long current = ready;
        for (const auto& [s, e] : busy) {
            if (safe_add(current, duration) <= s) {
                return current;
            }
            if (current < e) {
                current = e;
            }
        }
        return current;
    }

    void add_interval(long long start, long long end) {
        auto it = busy.begin();
        while (it != busy.end() && it->first < start) {
            ++it;
        }
        busy.insert(it, {start, end});
    }
};

static Problem read_problem() {
    Problem p;
    cin >> p.jobs >> p.machines;
    p.job_tasks.resize(p.jobs);
    p.job_total_duration.assign(p.jobs, 0);
    p.total_tasks = 0;

    for (int j = 0; j < p.jobs; ++j) {
        int t;
        cin >> t;
        p.job_tasks[j].resize(t);
        p.total_tasks += t;
        for (int i = 0; i < t; ++i) {
            cin >> p.job_tasks[j][i].machine >> p.job_tasks[j][i].duration;
            p.job_total_duration[j] += p.job_tasks[j][i].duration;
        }
    }

    p.suffix_duration_sum.assign(p.total_tasks, 0);
    return p;
}

static vector<long long> build_remaining_work(const vector<Task>& tasks) {
    int n = static_cast<int>(tasks.size());
    vector<long long> rem(n + 1, 0);
    for (int i = n - 1; i >= 0; --i) {
        rem[i] = rem[i + 1] + tasks[i].duration;
    }
    return rem;
}

static vector<vector<long long>> greedy_schedule(const Problem& p) {
    const int J = p.jobs;
    const int M = p.machines;

    vector<vector<long long>> starts(J);
    vector<vector<long long>> remaining(J);
    vector<long long> job_ready(J, 0);
    vector<int> next_task(J, 0);
    vector<long long> job_done_work(J, 0);
    vector<MachineCalendar> calendars(M);

    for (int j = 0; j < J; ++j) {
        starts[j].assign(p.job_tasks[j].size(), 0);
        remaining[j] = build_remaining_work(p.job_tasks[j]);
    }

    long long total_remaining = accumulate(
        p.job_total_duration.begin(), p.job_total_duration.end(), 0LL
    );

    auto score_candidate = [&](int j) -> Candidate {
        int t = next_task[j];
        const Task& op = p.job_tasks[j][t];
        long long est = calendars[op.machine].earliest_start(job_ready[j], op.duration);
        long long tail = remaining[j][t + 1];
        long long slack_pressure = total_remaining - job_done_work[j];
        long long priority = safe_add(est, op.duration);
        priority -= min<long long>(tail, numeric_limits<long long>::max() / 4);
        priority -= min<long long>(slack_pressure / max(1, J), numeric_limits<long long>::max() / 4);
        return {j, t, op.machine, op.duration, est, priority};
    };

    int remaining_ops = p.total_tasks;
    while (remaining_ops > 0) {
        Candidate best;
        bool found = false;

        for (int j = 0; j < J; ++j) {
            if (next_task[j] >= static_cast<int>(p.job_tasks[j].size())) {
                continue;
            }
            Candidate cand = score_candidate(j);
            if (!found) {
                best = cand;
                found = true;
                continue;
            }

            long long best_end = safe_add(best.earliest, best.duration);
            long long cand_end = safe_add(cand.earliest, cand.duration);

            if (cand.earliest < best.earliest ||
                (cand.earliest == best.earliest && cand_end < best_end) ||
                (cand.earliest == best.earliest && cand_end == best_end && cand.priority < best.priority) ||
                (cand.earliest == best.earliest && cand_end == best_end && cand.priority == best.priority && cand.job < best.job)) {
                best = cand;
            }
        }

        long long start = best.earliest;
        long long end = safe_add(start, best.duration);
        calendars[best.machine].add_interval(start, end);
        starts[best.job][best.task] = start;
        job_ready[best.job] = end;
        job_done_work[best.job] += best.duration;
        total_remaining -= best.duration;
        ++next_task[best.job];
        --remaining_ops;
    }

    return starts;
}

struct ExactSolver {
    const Problem& p;
    vector<vector<long long>> best_starts;
    vector<vector<long long>> current_starts;
    vector<long long> job_ready;
    vector<long long> machine_ready;
    vector<int> next_task;
    vector<long long> machine_remaining;
    vector<long long> job_remaining;
    long long best_makespan = numeric_limits<long long>::max();

    explicit ExactSolver(const Problem& problem) : p(problem) {
        best_starts.resize(p.jobs);
        current_starts.resize(p.jobs);
        job_ready.assign(p.jobs, 0);
        machine_ready.assign(p.machines, 0);
        next_task.assign(p.jobs, 0);
        machine_remaining.assign(p.machines, 0);
        job_remaining = p.job_total_duration;

        for (int j = 0; j < p.jobs; ++j) {
            best_starts[j].assign(p.job_tasks[j].size(), 0);
            current_starts[j].assign(p.job_tasks[j].size(), 0);
            for (const Task& t : p.job_tasks[j]) {
                machine_remaining[t.machine] += t.duration;
            }
        }
    }

    long long lower_bound(long long current_max) const {
        long long lb = current_max;
        for (int j = 0; j < p.jobs; ++j) {
            lb = max(lb, safe_add(job_ready[j], job_remaining[j]));
        }
        for (int m = 0; m < p.machines; ++m) {
            lb = max(lb, safe_add(machine_ready[m], machine_remaining[m]));
        }
        return lb;
    }

    void dfs(int scheduled, long long current_max) {
        if (scheduled == p.total_tasks) {
            if (current_max < best_makespan) {
                best_makespan = current_max;
                best_starts = current_starts;
            }
            return;
        }

        if (lower_bound(current_max) >= best_makespan) {
            return;
        }

        vector<Candidate> candidates;
        candidates.reserve(p.jobs);
        for (int j = 0; j < p.jobs; ++j) {
            int t = next_task[j];
            if (t >= static_cast<int>(p.job_tasks[j].size())) {
                continue;
            }
            const Task& op = p.job_tasks[j][t];
            long long start = max(job_ready[j], machine_ready[op.machine]);
            long long tail = job_remaining[j] - op.duration;
            long long priority = safe_add(start, op.duration) - tail;
            candidates.push_back({j, t, op.machine, op.duration, start, priority});
        }

        sort(candidates.begin(), candidates.end(), [](const Candidate& a, const Candidate& b) {
            long long end_a = safe_add(a.earliest, a.duration);
            long long end_b = safe_add(b.earliest, b.duration);
            if (a.earliest != b.earliest) return a.earliest < b.earliest;
            if (end_a != end_b) return end_a < end_b;
            if (a.priority != b.priority) return a.priority < b.priority;
            return a.job < b.job;
        });

        for (const Candidate& cand : candidates) {
            const int j = cand.job;
            const int t = cand.task;
            const int m = cand.machine;
            const long long dur = cand.duration;
            const long long start = cand.earliest;
            const long long end = safe_add(start, dur);

            long long old_job_ready = job_ready[j];
            long long old_machine_ready = machine_ready[m];
            long long old_job_remaining = job_remaining[j];
            long long old_machine_remaining = machine_remaining[m];

            current_starts[j][t] = start;
            job_ready[j] = end;
            machine_ready[m] = end;
            job_remaining[j] -= dur;
            machine_remaining[m] -= dur;
            ++next_task[j];

            dfs(scheduled + 1, max(current_max, end));

            --next_task[j];
            job_ready[j] = old_job_ready;
            machine_ready[m] = old_machine_ready;
            job_remaining[j] = old_job_remaining;
            machine_remaining[m] = old_machine_remaining;
        }
    }
};

static vector<vector<long long>> solve_exact_small(const Problem& p) {
    ExactSolver solver(p);
    solver.dfs(0, 0);
    return solver.best_starts;
}

static vector<vector<long long>> solve_problem(const Problem& p) {
    bool tiny_instance = p.total_tasks <= 18 && p.jobs <= 8 && p.machines <= 8;
    bool small_instance = p.total_tasks <= 250 && p.jobs <= 40 && p.machines <= 20;

    if (tiny_instance) {
        return solve_exact_small(p);
    }

    vector<vector<long long>> starts = greedy_schedule(p);

    if (small_instance) {
        vector<vector<long long>> second = greedy_schedule(p);
        if (!second.empty()) {
            starts.swap(second);
        }
    }

    return starts;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Problem p = read_problem();
    vector<vector<long long>> starts = solve_problem(p);

    for (int j = 0; j < p.jobs; ++j) {
        for (int t = 0; t < static_cast<int>(p.job_tasks[j].size()); ++t) {
            if (t) cout << ' ';
            cout << starts[j][t] << ' ' << p.job_tasks[j][t].machine;
        }
        cout << '\n';
    }
    return 0;
}
