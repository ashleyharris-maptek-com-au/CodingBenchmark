using namespace std;

struct Task {
    int job;
    int index;
    int machine;
    int duration;
    int global_id;
};

struct Problem {
    int jobs = 0;
    int machines = 0;
    vector<vector<Task>> job_tasks;
    vector<Task> all_tasks;
    vector<vector<int>> machine_tasks;
    int total_tasks = 0;
    long long total_duration = 0;
};

struct Schedule {
    vector<vector<long long>> starts;
    long long makespan = 0;
};

static Problem read_problem() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Problem p;
    cin >> p.jobs >> p.machines;
    p.job_tasks.assign(p.jobs, {});
    p.machine_tasks.assign(p.machines, {});

    int gid = 0;
    for (int j = 0; j < p.jobs; ++j) {
        int t;
        cin >> t;
        p.job_tasks[j].reserve(t);
        for (int i = 0; i < t; ++i) {
            int m, d;
            cin >> m >> d;
            Task task{j, i, m, d, gid++};
            p.job_tasks[j].push_back(task);
            p.all_tasks.push_back(task);
            if (m >= 0 && m < p.machines) p.machine_tasks[m].push_back(task.global_id);
            p.total_duration += d;
        }
    }
    p.total_tasks = gid;
    return p;
}

static inline uint64_t mix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct CandidateScore {
    long long ready;
    long long tail;
    int duration;
    uint64_t tie;
};

static inline bool better_candidate(const CandidateScore& a, const CandidateScore& b) {
    if (a.ready != b.ready) return a.ready < b.ready;
    if (a.tail != b.tail) return a.tail > b.tail;
    if (a.duration != b.duration) return a.duration > b.duration;
    return a.tie < b.tie;
}

static vector<long long> compute_job_remaining(const Problem& p) {
    vector<long long> rem(p.total_tasks, 0);
    for (int j = 0; j < p.jobs; ++j) {
        long long suffix = 0;
        for (int i = static_cast<int>(p.job_tasks[j].size()) - 1; i >= 0; --i) {
            suffix += p.job_tasks[j][i].duration;
            rem[p.job_tasks[j][i].global_id] = suffix;
        }
    }
    return rem;
}

static Schedule list_schedule_from_machine_orders(const Problem& p, const vector<vector<int>>& machine_order) {
    Schedule sched;
    sched.starts.assign(p.jobs, {});
    for (int j = 0; j < p.jobs; ++j) sched.starts[j].assign(p.job_tasks[j].size(), -1);

    vector<int> pos_on_machine(p.machines, 0);
    vector<int> next_job_task(p.jobs, 0);
    vector<long long> job_ready(p.jobs, 0), machine_ready(p.machines, 0);
    vector<char> done(p.total_tasks, 0);
    int scheduled = 0;

    vector<vector<int>> global_to_machine_pos(p.machines);
    for (int m = 0; m < p.machines; ++m) {
        global_to_machine_pos[m].resize(p.machine_tasks[m].size(), -1);
    }

    while (scheduled < p.total_tasks) {
        bool progress = false;
        for (int m = 0; m < p.machines; ++m) {
            int& pm = pos_on_machine[m];
            if (pm >= static_cast<int>(machine_order[m].size())) continue;
            int gid = machine_order[m][pm];
            const Task& t = p.all_tasks[gid];
            if (t.index != next_job_task[t.job]) continue;
            long long st = max(job_ready[t.job], machine_ready[m]);
            sched.starts[t.job][t.index] = st;
            long long en = st + t.duration;
            job_ready[t.job] = en;
            machine_ready[m] = en;
            next_job_task[t.job]++;
            pm++;
            done[gid] = 1;
            progress = true;
            ++scheduled;
            if (en > sched.makespan) sched.makespan = en;
        }
        if (progress) continue;

        long long best_time = numeric_limits<long long>::max();
        int best_machine = -1;
        for (int m = 0; m < p.machines; ++m) {
            int pm = pos_on_machine[m];
            if (pm >= static_cast<int>(machine_order[m].size())) continue;
            int gid = machine_order[m][pm];
            const Task& t = p.all_tasks[gid];
            if (t.index != next_job_task[t.job]) continue;
            long long st = max(job_ready[t.job], machine_ready[m]);
            if (st < best_time) {
                best_time = st;
                best_machine = m;
            }
        }
        if (best_machine != -1) {
            int& pm = pos_on_machine[best_machine];
            int gid = machine_order[best_machine][pm];
            const Task& t = p.all_tasks[gid];
            long long st = max(job_ready[t.job], machine_ready[best_machine]);
            sched.starts[t.job][t.index] = st;
            long long en = st + t.duration;
            job_ready[t.job] = en;
            machine_ready[best_machine] = en;
            next_job_task[t.job]++;
            pm++;
            done[gid] = 1;
            ++scheduled;
            if (en > sched.makespan) sched.makespan = en;
            continue;
        }

        bool forced = false;
        for (int j = 0; j < p.jobs && !forced; ++j) {
            if (next_job_task[j] >= static_cast<int>(p.job_tasks[j].size())) continue;
            const Task& t = p.job_tasks[j][next_job_task[j]];
            int m = t.machine;
            int& pm = pos_on_machine[m];
            if (pm < static_cast<int>(machine_order[m].size()) && machine_order[m][pm] == t.global_id) {
                long long st = max(job_ready[j], machine_ready[m]);
                sched.starts[j][t.index] = st;
                long long en = st + t.duration;
                job_ready[j] = en;
                machine_ready[m] = en;
                next_job_task[j]++;
                pm++;
                done[t.global_id] = 1;
                ++scheduled;
                if (en > sched.makespan) sched.makespan = en;
                forced = true;
            }
        }
        if (!forced) break;
    }

    return sched;
}

static vector<vector<int>> build_machine_orders_dispatch(const Problem& p, int rounds, uint64_t seed) {
    vector<long long> rem = compute_job_remaining(p);
    vector<vector<int>> best_orders(p.machines);
    long long best_key = numeric_limits<long long>::max();
    mt19937_64 rng(seed);

    for (int round = 0; round < rounds; ++round) {
        vector<vector<int>> orders(p.machines);
        vector<int> next_idx(p.jobs, 0);
        vector<long long> job_ready(p.jobs, 0), machine_ready(p.machines, 0);
        vector<vector<int>> available(p.machines);

        for (int j = 0; j < p.jobs; ++j) {
            if (!p.job_tasks[j].empty()) {
                available[p.job_tasks[j][0].machine].push_back(j);
            }
        }

        int remaining = p.total_tasks;
        while (remaining > 0) {
            long long best_start = numeric_limits<long long>::max();
            int chosen_machine = -1;
            int chosen_job = -1;
            CandidateScore best_score{0, 0, 0, 0};
            bool found = false;

            for (int m = 0; m < p.machines; ++m) {
                auto& jobs = available[m];
                if (jobs.empty()) continue;
                for (int j : jobs) {
                    const Task& t = p.job_tasks[j][next_idx[j]];
                    CandidateScore cur;
                    cur.ready = max(job_ready[j], machine_ready[m]);
                    cur.tail = rem[t.global_id];
                    cur.duration = t.duration;
                    cur.tie = mix64(static_cast<uint64_t>(t.global_id) ^ (static_cast<uint64_t>(round + 1) << 32) ^ rng());
                    if (!found || cur.ready < best_start || (cur.ready == best_start && better_candidate(cur, best_score))) {
                        found = true;
                        best_start = cur.ready;
                        best_score = cur;
                        chosen_machine = m;
                        chosen_job = j;
                    }
                }
            }

            if (!found) break;

            Task t = p.job_tasks[chosen_job][next_idx[chosen_job]];
            long long st = max(job_ready[chosen_job], machine_ready[chosen_machine]);
            long long en = st + t.duration;
            orders[chosen_machine].push_back(t.global_id);
            job_ready[chosen_job] = en;
            machine_ready[chosen_machine] = en;

            auto& vec = available[chosen_machine];
            vec.erase(find(vec.begin(), vec.end(), chosen_job));
            next_idx[chosen_job]++;
            if (next_idx[chosen_job] < static_cast<int>(p.job_tasks[chosen_job].size())) {
                int nm = p.job_tasks[chosen_job][next_idx[chosen_job]].machine;
                available[nm].push_back(chosen_job);
            }
            --remaining;
        }

        Schedule sched = list_schedule_from_machine_orders(p, orders);
        long long key = sched.makespan;
        if (key < best_key) {
            best_key = key;
            best_orders = std::move(orders);
        }
    }

    return best_orders;
}

static void improve_machine_orders(const Problem& p, vector<vector<int>>& orders, int passes) {
    Schedule best_sched = list_schedule_from_machine_orders(p, orders);
    mt19937 rng(1234567);

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        vector<int> machine_ids(p.machines);
        iota(machine_ids.begin(), machine_ids.end(), 0);
        shuffle(machine_ids.begin(), machine_ids.end(), rng);

        for (int m : machine_ids) {
            auto& ord = orders[m];
            if (ord.size() < 2) continue;

            int tries = min<int>(static_cast<int>(ord.size()) - 1, 64);
            for (int t = 0; t < tries; ++t) {
                int i = (pass * 17 + t * 31) % (static_cast<int>(ord.size()) - 1);
                swap(ord[i], ord[i + 1]);
                Schedule cand = list_schedule_from_machine_orders(p, orders);
                if (cand.makespan < best_sched.makespan) {
                    best_sched = std::move(cand);
                    improved = true;
                } else {
                    swap(ord[i], ord[i + 1]);
                }
            }
        }
        if (!improved) break;
    }
}

static Schedule build_serial_schedule(const Problem& p) {
    vector<vector<int>> orders(p.machines);
    vector<long long> rem = compute_job_remaining(p);
    vector<int> next_idx(p.jobs, 0);
    vector<long long> job_ready(p.jobs, 0), machine_ready(p.machines, 0);

    int remaining = p.total_tasks;
    while (remaining > 0) {
        int best_job = -1;
        long long best_start = numeric_limits<long long>::max();
        CandidateScore best_score{0, 0, 0, 0};
        for (int j = 0; j < p.jobs; ++j) {
            if (next_idx[j] >= static_cast<int>(p.job_tasks[j].size())) continue;
            const Task& t = p.job_tasks[j][next_idx[j]];
            CandidateScore cur;
            cur.ready = max(job_ready[j], machine_ready[t.machine]);
            cur.tail = rem[t.global_id];
            cur.duration = t.duration;
            cur.tie = static_cast<uint64_t>(t.global_id);
            if (best_job == -1 || cur.ready < best_start || (cur.ready == best_start && better_candidate(cur, best_score))) {
                best_job = j;
                best_start = cur.ready;
                best_score = cur;
            }
        }
        const Task& t = p.job_tasks[best_job][next_idx[best_job]];
        long long st = max(job_ready[best_job], machine_ready[t.machine]);
        long long en = st + t.duration;
        orders[t.machine].push_back(t.global_id);
        job_ready[best_job] = en;
        machine_ready[t.machine] = en;
        next_idx[best_job]++;
        --remaining;
    }

    return list_schedule_from_machine_orders(p, orders);
}

static Schedule solve_small(const Problem& p) {
    auto orders = build_machine_orders_dispatch(p, 24, 0xC0FFEEULL);
    improve_machine_orders(p, orders, 20);
    return list_schedule_from_machine_orders(p, orders);
}

static Schedule solve_medium(const Problem& p) {
    auto orders = build_machine_orders_dispatch(p, 8, 0xBADC0DEULL);
    improve_machine_orders(p, orders, 6);
    return list_schedule_from_machine_orders(p, orders);
}

static Schedule solve_large(const Problem& p) {
    auto orders = build_machine_orders_dispatch(p, 3, 0x12345678ULL);
    return list_schedule_from_machine_orders(p, orders);
}

static Schedule solve_problem(const Problem& p) {
    if (p.total_tasks <= 2000 && p.jobs <= 80 && p.machines <= 20) {
        return solve_small(p);
    }
    if (p.total_tasks <= 12000 && p.jobs <= 300 && p.machines <= 40) {
        return solve_medium(p);
    }
    if (p.total_tasks > 0) {
        return solve_large(p);
    }
    return Schedule{};
}

int main() {
    Problem p = read_problem();
    Schedule sched = solve_problem(p);

    for (int j = 0; j < p.jobs; ++j) {
        for (int i = 0; i < static_cast<int>(p.job_tasks[j].size()); ++i) {
            if (i) cout << ' ';
            cout << sched.starts[j][i] << ' ' << p.job_tasks[j][i].machine;
        }
        cout << '\n';
    }
    return 0;
}
