#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <set>
#include <tuple>
#include <utility>
#include <vector>

struct RectInput {
    int w = 0;
    int h = 0;
    int id = -1;
    long long area = 0;
    int max_side = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
};

struct Board {
    std::vector<FreeRect> free_rects;
};

static inline bool contains_rect(const FreeRect& a, const FreeRect& b) {
    return a.x <= b.x && a.y <= b.y &&
           a.x + a.w >= b.x + b.w &&
           a.y + a.h >= b.y + b.h;
}

static void prune_free_rects(std::vector<FreeRect>& rects) {
    std::vector<FreeRect> filtered;
    filtered.reserve(rects.size());
    for (const auto& r : rects) {
        if (r.w > 0 && r.h > 0) {
            filtered.push_back(r);
        }
    }

    std::vector<char> removed(filtered.size(), 0);
    for (size_t i = 0; i < filtered.size(); ++i) {
        if (removed[i]) {
            continue;
        }
        for (size_t j = i + 1; j < filtered.size(); ++j) {
            if (removed[j]) {
                continue;
            }
            if (contains_rect(filtered[i], filtered[j])) {
                removed[j] = 1;
            } else if (contains_rect(filtered[j], filtered[i])) {
                removed[i] = 1;
                break;
            }
        }
    }

    rects.clear();
    rects.reserve(filtered.size());
    for (size_t i = 0; i < filtered.size(); ++i) {
        if (!removed[i]) {
            rects.push_back(filtered[i]);
        }
    }
}

static bool insert_into_board(Board& board, int bw, int bh, const RectInput& rect, Placement& out) {
    long long best_score = std::numeric_limits<long long>::max();
    int best_idx = -1;
    int best_rw = 0;
    int best_rh = 0;
    int best_rot = 0;

    const int try_count_limit = 96;
    const int scan_count = static_cast<int>(board.free_rects.size());
    const int start = scan_count > try_count_limit ? scan_count - try_count_limit : 0;

    for (int i = start; i < scan_count; ++i) {
        const FreeRect& fr = board.free_rects[i];
        for (int rot = 0; rot < 2; ++rot) {
            int rw = rot ? rect.h : rect.w;
            int rh = rot ? rect.w : rect.h;
            if (rw > fr.w || rh > fr.h) {
                continue;
            }
            int leftover_w = fr.w - rw;
            int leftover_h = fr.h - rh;
            long long short_side = std::min(leftover_w, leftover_h);
            long long long_side = std::max(leftover_w, leftover_h);
            long long waste = 1LL * fr.w * fr.h - 1LL * rw * rh;
            long long score = (short_side << 40) + (long_side << 24) + waste;
            if (score < best_score) {
                best_score = score;
                best_idx = i;
                best_rw = rw;
                best_rh = rh;
                best_rot = rot;
            }
        }
    }

    if (best_idx < 0) {
        return false;
    }

    FreeRect used = board.free_rects[best_idx];
    board.free_rects.erase(board.free_rects.begin() + best_idx);

    out.x = used.x;
    out.y = used.y;
    out.rotated = best_rot;

    int right_w = used.w - best_rw;
    int below_h = used.h - best_rh;

    if (right_w > 0) {
        board.free_rects.push_back({used.x + best_rw, used.y, right_w, best_rh});
    }
    if (below_h > 0) {
        board.free_rects.push_back({used.x, used.y + best_rh, used.w, below_h});
    }

    if (right_w > 0 && below_h > 0) {
        // Keep the split guillotine-compatible and add the larger remainder first
        // so recent rectangles are more promising candidates for the bounded scan.
        const int n = static_cast<int>(board.free_rects.size());
        const FreeRect& a = board.free_rects[n - 2];
        const FreeRect& b = board.free_rects[n - 1];
        long long area_a = 1LL * a.w * a.h;
        long long area_b = 1LL * b.w * b.h;
        if (area_a < area_b) {
            std::swap(board.free_rects[n - 2], board.free_rects[n - 1]);
        }
    }

    if (board.free_rects.size() > 256) {
        prune_free_rects(board.free_rects);
    }

    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    int board_w = 0;
    int board_h = 0;
    if (!(std::cin >> n >> board_w >> board_h)) {
        return 0;
    }

    std::vector<RectInput> rects(n);
    long long total_area = 0;
    for (int i = 0; i < n; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects[i] = {w, h, i, 1LL * w * h, std::max(w, h)};
        total_area += rects[i].area;
    }

    for (const auto& r : rects) {
        bool fits = (r.w <= board_w && r.h <= board_h) || (r.h <= board_w && r.w <= board_h);
        if (!fits) {
            std::cout << -1 << '\n';
            return 0;
        }
    }

    std::sort(rects.begin(), rects.end(), [](const RectInput& a, const RectInput& b) {
        if (a.max_side != b.max_side) return a.max_side > b.max_side;
        if (a.area != b.area) return a.area > b.area;
        if (a.h != b.h) return a.h > b.h;
        if (a.w != b.w) return a.w > b.w;
        return a.id < b.id;
    });

    std::vector<Placement> answer(n);
    std::vector<Board> boards;
    boards.reserve(std::max(4, static_cast<int>(total_area / std::max(1LL, 1LL * board_w * board_h)) + 8));

    std::set<std::pair<int, int>> candidates;

    auto refresh_board_candidate = [&](int idx) {
        auto it_low = candidates.lower_bound({idx, std::numeric_limits<int>::min()});
        while (it_low != candidates.end() && it_low->first == idx) {
            it_low = candidates.erase(it_low);
        }

        int best_max = -1;
        for (const auto& fr : boards[idx].free_rects) {
            best_max = std::max(best_max, std::max(fr.w, fr.h));
        }
        if (best_max >= 0) {
            candidates.insert({idx, -best_max});
        }
    };

    for (const auto& rect : rects) {
        Placement placed;
        bool ok = false;

        std::vector<int> board_order;
        board_order.reserve(std::min<size_t>(candidates.size(), 64));
        for (const auto& entry : candidates) {
            board_order.push_back(entry.first);
            if (board_order.size() >= 64) {
                break;
            }
        }

        for (int idx : board_order) {
            if (insert_into_board(boards[idx], board_w, board_h, rect, placed)) {
                placed.board = idx;
                refresh_board_candidate(idx);
                ok = true;
                break;
            }
        }

        if (!ok) {
            Board b;
            b.free_rects.push_back({0, 0, board_w, board_h});
            boards.push_back(std::move(b));
            int idx = static_cast<int>(boards.size()) - 1;
            bool inserted = insert_into_board(boards[idx], board_w, board_h, rect, placed);
            if (!inserted) {
                std::cout << -1 << '\n';
                return 0;
            }
            placed.board = idx;
            refresh_board_candidate(idx);
        }

        answer[rect.id] = placed;
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < n; ++i) {
        const auto& p = answer[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
