#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

using namespace std;

struct Rect {
    int w = 0;
    int h = 0;
    int id = -1;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Shelf {
    int y = 0;
    int height = 0;
    int used_width = 0;
};

struct Board {
    vector<Shelf> shelves;
    int used_height = 0;
    long long used_area = 0;
};

struct Candidate {
    bool ok = false;
    int board = -1;
    int shelf = -1; // -1 means new shelf
    int x = 0;
    int y = 0;
    int rotated = 0;
    int width = 0;
    int height = 0;
    long long score = numeric_limits<long long>::max();
};

static Candidate evaluate_board(
    const Board& board,
    int board_w,
    int board_h,
    int rw,
    int rh,
    int board_index
) {
    Candidate best;

    auto consider = [&](int shelf_idx, int x, int y, int pw, int ph, int rotated, long long score) {
        if (!best.ok || score < best.score) {
            best.ok = true;
            best.board = board_index;
            best.shelf = shelf_idx;
            best.x = x;
            best.y = y;
            best.rotated = rotated;
            best.width = pw;
            best.height = ph;
            best.score = score;
        }
    };

    auto test_orientation = [&](int pw, int ph, int rotated) {
        for (int i = 0; i < static_cast<int>(board.shelves.size()); ++i) {
            const Shelf& s = board.shelves[i];
            if (ph <= s.height && s.used_width + pw <= board_w) {
                long long width_left = static_cast<long long>(board_w - (s.used_width + pw));
                long long height_slack = static_cast<long long>(s.height - ph);
                long long score = width_left * 8 + height_slack * 4;
                consider(i, s.used_width, s.y, pw, ph, rotated, score);
            }
        }

        if (board.used_height + ph <= board_h && pw <= board_w) {
            long long remaining_height = static_cast<long long>(board_h - (board.used_height + ph));
            long long remaining_width = static_cast<long long>(board_w - pw);
            long long score = remaining_height * 16 + remaining_width * 2 + ph;
            consider(-1, 0, board.used_height, pw, ph, rotated, score);
        }
    };

    test_orientation(rw, rh, 0);
    if (rw != rh) {
        test_orientation(rh, rw, 1);
    }

    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    int board_w = 0;
    int board_h = 0;
    if (!(cin >> n >> board_w >> board_h)) {
        return 0;
    }

    vector<Rect> rects(n);
    long long total_area = 0;
    for (int i = 0; i < n; ++i) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
        total_area += 1LL * rects[i].w * rects[i].h;
    }

    const long long board_area = 1LL * board_w * board_h;
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);

    stable_sort(order.begin(), order.end(), [&](int a, int b) {
        const Rect& ra = rects[a];
        const Rect& rb = rects[b];
        const int amax = max(ra.w, ra.h);
        const int bmax = max(rb.w, rb.h);
        if (amax != bmax) return amax > bmax;
        const int amin = min(ra.w, ra.h);
        const int bmin = min(rb.w, rb.h);
        if (amin != bmin) return amin > bmin;
        const long long area_a = 1LL * ra.w * ra.h;
        const long long area_b = 1LL * rb.w * rb.h;
        if (area_a != area_b) return area_a > area_b;
        return ra.id < rb.id;
    });

    vector<Placement> placements(n);
    vector<Board> boards;
    boards.reserve(max<long long>(1, total_area / max<long long>(1, board_area) + 8));

    for (int idx : order) {
        const Rect& r = rects[idx];

        bool can_fit = (r.w <= board_w && r.h <= board_h) || (r.h <= board_w && r.w <= board_h);
        if (!can_fit) {
            cout << 0 << '\n';
            return 0;
        }

        Candidate best;
        for (int b = 0; b < static_cast<int>(boards.size()); ++b) {
            Candidate cand = evaluate_board(boards[b], board_w, board_h, r.w, r.h, b);
            if (cand.ok && (!best.ok || cand.score < best.score)) {
                best = cand;
            }
        }

        if (!best.ok) {
            boards.push_back(Board());
            best = evaluate_board(boards.back(), board_w, board_h, r.w, r.h, static_cast<int>(boards.size()) - 1);
        }

        Board& target = boards[best.board];
        if (best.shelf == -1) {
            Shelf shelf;
            shelf.y = best.y;
            shelf.height = best.height;
            shelf.used_width = best.width;
            target.shelves.push_back(shelf);
            target.used_height += best.height;
        } else {
            target.shelves[best.shelf].used_width += best.width;
        }
        target.used_area += 1LL * best.width * best.height;

        placements[r.id] = {best.board, best.x, best.y, best.rotated};
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < n; ++i) {
        const Placement& p = placements[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
