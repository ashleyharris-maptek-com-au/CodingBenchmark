constexpr double kPi = 3.14159265358979323846;
constexpr double kGravity = 1.62;
constexpr double kTick = 0.1;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3& operator+=(const Vec3& other) {
        x += other.x;
        y += other.y;
        z += other.z;
        return *this;
    }

    Vec3& operator-=(const Vec3& other) {
        x -= other.x;
        y -= other.y;
        z -= other.z;
        return *this;
    }

    Vec3& operator*=(double s) {
        x *= s;
        y *= s;
        z *= s;
        return *this;
    }
};

Vec3 operator+(Vec3 a, const Vec3& b) { return a += b; }
Vec3 operator-(Vec3 a, const Vec3& b) { return a -= b; }
Vec3 operator*(Vec3 a, double s) { return a *= s; }
Vec3 operator*(double s, Vec3 a) { return a *= s; }
Vec3 operator/(Vec3 a, double s) {
    a.x /= s;
    a.y /= s;
    a.z /= s;
    return a;
}

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

double norm_sq(const Vec3& v) {
    return dot(v, v);
}

double norm(const Vec3& v) {
    return std::sqrt(norm_sq(v));
}

Vec3 normalized(const Vec3& v) {
    const double n = norm(v);
    if (n < 1e-9) {
        return {};
    }
    return v / n;
}

double clamp(double v, double lo, double hi) {
    return std::max(lo, std::min(hi, v));
}

double wrap_angle(double a) {
    while (a > kPi) {
        a -= 2.0 * kPi;
    }
    while (a < -kPi) {
        a += 2.0 * kPi;
    }
    return a;
}

struct Header {
    int width = 0;
    int depth = 0;
    int height = 0;
    double gravity = kGravity;
    double max_thrust = 0.0;
    double fuel = 0.0;
    Vec3 start;
    Vec3 target;
    std::string terrain_path;
};

struct State {
    Vec3 pos;
    Vec3 vel;
    double pitch = 0.0;
    double yaw = 0.0;
    double roll = 0.0;
    double pitch_rate = 0.0;
    double yaw_rate = 0.0;
    double roll_rate = 0.0;
    double fuel_remaining = 0.0;
    bool valid = false;
};

struct Command {
    double thrust = 0.0;
    double pitch_cmd = 0.0;
    double yaw_cmd = 0.0;
    double roll_cmd = 0.0;
};

struct TerrainColumn {
    uint16_t ground_top = 0;
    uint16_t float_bottom = 0;
    uint16_t float_top = 0;
};

class TerrainMap {
public:
    explicit TerrainMap(const Header& header)
        : width_(header.width), depth_(header.depth), height_(header.height),
          target_(header.target), loaded_(false), done_(false) {}

    void load_async(const std::string& path) {
        worker_ = std::thread([this, path]() { load(path); });
    }

    ~TerrainMap() {
        if (worker_.joinable()) {
            worker_.join();
        }
    }

    bool fully_loaded() const {
        return done_.load(std::memory_order_relaxed);
    }

    double target_ground() const {
        return target_ground_.load(std::memory_order_relaxed);
    }

    double clearance_estimate(double x, double y, double z) const {
        if (!loaded_.load(std::memory_order_relaxed)) {
            return z;
        }
        const int ix = static_cast<int>(std::floor(x));
        const int iy = static_cast<int>(std::floor(y));
        if (ix < 0 || ix >= width_ || iy < 0 || iy >= depth_) {
            return -1.0;
        }
        const auto& c = cells_[index(ix, iy)];
        if (z < c.ground_top) {
            return -1.0;
        }
        double clearance = z - c.ground_top;
        if (c.float_bottom < c.float_top && z >= c.float_bottom && z < c.float_top) {
            return -1.0;
        }
        if (c.float_bottom < c.float_top && z < c.float_bottom) {
            clearance = std::min(clearance, c.float_bottom - z);
        }
        return clearance;
    }

    double sample_ground(double x, double y) const {
        if (!loaded_.load(std::memory_order_relaxed)) {
            return 0.0;
        }
        const int ix = static_cast<int>(std::floor(x));
        const int iy = static_cast<int>(std::floor(y));
        if (ix < 0 || ix >= width_ || iy < 0 || iy >= depth_) {
            return static_cast<double>(height_);
        }
        return cells_[index(ix, iy)].ground_top;
    }

private:
    size_t index(int x, int y) const {
        return static_cast<size_t>(x) * static_cast<size_t>(depth_) + static_cast<size_t>(y);
    }

    static uint16_t read_u16(std::ifstream& in) {
        unsigned char bytes[2] = {0, 0};
        in.read(reinterpret_cast<char*>(bytes), 2);
        return static_cast<uint16_t>(bytes[0] | (static_cast<uint16_t>(bytes[1]) << 8));
    }

    void load(const std::string& path) {
        std::ifstream in(path, std::ios::binary);
        if (!in) {
            done_.store(true, std::memory_order_relaxed);
            return;
        }

        cells_.resize(static_cast<size_t>(width_) * static_cast<size_t>(depth_));
        double local_target_ground = 0.0;
        const int tx = static_cast<int>(std::floor(target_.x));
        const int ty = static_cast<int>(std::floor(target_.y));

        for (int x = 0; x < width_; ++x) {
            for (int y = 0; y < depth_; ++y) {
                TerrainColumn c;
                c.ground_top = read_u16(in);
                c.float_bottom = read_u16(in);
                c.float_top = read_u16(in);
                cells_[index(x, y)] = c;
                if (x == tx && y == ty) {
                    local_target_ground = static_cast<double>(c.ground_top);
                }
            }
            if (x % 64 == 0) {
                loaded_.store(true, std::memory_order_relaxed);
            }
        }
        target_ground_.store(local_target_ground, std::memory_order_relaxed);
        loaded_.store(true, std::memory_order_relaxed);
        done_.store(true, std::memory_order_relaxed);
    }

    int width_;
    int depth_;
    int height_;
    Vec3 target_;
    std::vector<TerrainColumn> cells_;
    std::thread worker_;
    std::atomic<bool> loaded_;
    std::atomic<bool> done_;
    std::atomic<double> target_ground_{0.0};
};

class StateQueue {
public:
    void push(const State& s) {
        {
            std::lock_guard<std::mutex> lock(mutex_);
            latest_ = s;
            has_state_ = true;
        }
        cv_.notify_one();
    }

    bool wait_and_pop_latest(State& out) {
        std::unique_lock<std::mutex> lock(mutex_);
        cv_.wait(lock, [&] { return has_state_ || closed_; });
        if (!has_state_) {
            return false;
        }
        out = latest_;
        has_state_ = false;
        return true;
    }

    void close() {
        {
            std::lock_guard<std::mutex> lock(mutex_);
            closed_ = true;
        }
        cv_.notify_all();
    }

private:
    std::mutex mutex_;
    std::condition_variable cv_;
    State latest_;
    bool has_state_ = false;
    bool closed_ = false;
};

class Controller {
public:
    Controller(const Header& header, const TerrainMap* terrain)
        : header_(header), terrain_(terrain) {
        landing_radius_ = std::max(5.0, std::min(header.width, header.depth) * 0.02);
    }

    Command step(const State& s) {
        const Vec3 to_target = header_.target - s.pos;
        const double horiz_dist = std::hypot(to_target.x, to_target.y);
        const double speed = norm(s.vel);
        const double target_ground = terrain_ ? terrain_->target_ground() : 0.0;
        const double altitude_agl = s.pos.z - std::max(target_ground, terrain_ ? terrain_->sample_ground(s.pos.x, s.pos.y) : 0.0);

        double approach_speed = clamp(horiz_dist * 0.12, 0.0, 40.0);
        if (altitude_agl < 120.0) {
            approach_speed = clamp(horiz_dist * 0.08, 0.0, 16.0);
        }
        if (horiz_dist < landing_radius_ * 3.0) {
            approach_speed = clamp(horiz_dist * 0.15, 0.0, 4.0);
        }

        Vec3 desired_hvel{};
        if (horiz_dist > 1e-6) {
            desired_hvel.x = to_target.x / horiz_dist * approach_speed;
            desired_hvel.y = to_target.y / horiz_dist * approach_speed;
        }

        const double safe_descent = compute_safe_descent(altitude_agl, horiz_dist, speed);
        const Vec3 vel_error{
            desired_hvel.x - s.vel.x,
            desired_hvel.y - s.vel.y,
            (-safe_descent) - s.vel.z
        };

        Vec3 desired_accel{
            clamp(vel_error.x * 0.9, -8.0, 8.0),
            clamp(vel_error.y * 0.9, -8.0, 8.0),
            clamp(vel_error.z * 1.2, -8.0, 8.0) + header_.gravity
        };

        const double clearance = terrain_ ? terrain_->clearance_estimate(s.pos.x, s.pos.y, s.pos.z) : altitude_agl;
        if (clearance >= 0.0 && clearance < std::max(12.0, speed * 1.5)) {
            desired_accel.z = std::max(desired_accel.z, header_.gravity + 3.5);
            desired_accel.x *= 0.4;
            desired_accel.y *= 0.4;
        }

        const double accel_norm = norm(desired_accel);
        Vec3 thrust_dir = accel_norm > 1e-6 ? desired_accel / accel_norm : Vec3{0.0, 0.0, 1.0};
        if (thrust_dir.z < 0.15) {
            thrust_dir.z = 0.15;
            thrust_dir = normalized(thrust_dir);
        }

        double thrust = accel_norm / std::max(header_.max_thrust, 1e-6);
        thrust = clamp(thrust, 0.0, 1.0);
        if (s.fuel_remaining < 5.0 && altitude_agl > 20.0) {
            thrust = clamp(thrust, 0.0, 0.85);
        }

        const double target_pitch = std::asin(clamp(-thrust_dir.x, -1.0, 1.0));
        const double target_yaw = std::atan2(thrust_dir.y, thrust_dir.z);
        const double pitch_cmd = attitude_pd(target_pitch, s.pitch, s.pitch_rate);
        const double yaw_cmd = attitude_pd(target_yaw, s.yaw, s.yaw_rate);
        const double roll_cmd = attitude_pd(0.0, s.roll, s.roll_rate);

        return Command{thrust, pitch_cmd, yaw_cmd, roll_cmd};
    }

private:
    double attitude_pd(double target, double angle, double rate) const {
        const double err = wrap_angle(target - angle);
        return clamp(err * 1.8 - rate * 0.7, -1.0, 1.0);
    }

    double compute_safe_descent(double altitude_agl, double horiz_dist, double speed) const {
        double descent = 2.0;
        if (altitude_agl > 2000.0) {
            descent = 35.0;
        } else if (altitude_agl > 800.0) {
            descent = 24.0;
        } else if (altitude_agl > 250.0) {
            descent = 14.0;
        } else if (altitude_agl > 80.0) {
            descent = 8.0;
        } else if (altitude_agl > 25.0) {
            descent = 4.5;
        } else {
            descent = 1.6;
        }

        if (horiz_dist < landing_radius_ * 4.0) {
            descent = std::min(descent, 3.0);
        }
        if (horiz_dist < landing_radius_ * 1.5) {
            descent = std::min(descent, 1.8);
        }
        if (speed > 35.0 && altitude_agl < 120.0) {
            descent = std::min(descent, 2.0);
        }
        return descent;
    }

    Header header_;
    const TerrainMap* terrain_;
    double landing_radius_ = 5.0;
};

bool read_header(Header& h) {
    if (!(std::cin >> h.width >> h.depth >> h.height >> h.gravity >> h.max_thrust >> h.fuel)) {
        return false;
    }
    if (!(std::cin >> h.start.x >> h.start.y >> h.start.z >> h.target.x >> h.target.y >> h.target.z)) {
        return false;
    }
    std::string path_line;
    std::getline(std::cin, path_line);
    std::getline(std::cin, h.terrain_path);
    if (!h.terrain_path.empty() && h.terrain_path.back() == '\r') {
        h.terrain_path.pop_back();
    }
    std::string state_token;
    std::cin >> state_token;
    return true;
}

std::optional<State> parse_state_line(const std::string& line) {
    State s;
    std::istringstream iss(line);
    if (!(iss >> s.pos.x >> s.pos.y >> s.pos.z
              >> s.vel.x >> s.vel.y >> s.vel.z
              >> s.pitch >> s.yaw >> s.roll
              >> s.pitch_rate >> s.yaw_rate >> s.roll_rate
              >> s.fuel_remaining)) {
        return std::nullopt;
    }
    s.valid = true;
    return s;
}

}  // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Header header;
    if (!read_header(header)) {
        return 0;
    }

    TerrainMap terrain(header);
    terrain.load_async(header.terrain_path);

    StateQueue queue;
    std::atomic<bool> done{false};

    std::thread reader([&]() {
        std::string line;
        std::getline(std::cin, line);
        while (!done.load(std::memory_order_relaxed) && std::getline(std::cin, line)) {
            if (line.empty()) {
                continue;
            }
            auto s = parse_state_line(line);
            if (s) {
                queue.push(*s);
            }
        }
        queue.close();
    });

    Controller controller(header, &terrain);
    State state;

    std::cout << std::fixed << std::setprecision(6);
    while (queue.wait_and_pop_latest(state)) {
        Command cmd = controller.step(state);
        std::cout << cmd.thrust << ' '
                  << cmd.pitch_cmd << ' '
                  << cmd.yaw_cmd << ' '
                  << cmd.roll_cmd << '\n';
        std::cout.flush();
    }

    done.store(true, std::memory_order_relaxed);
    if (reader.joinable()) {
        reader.join();
    }
    return 0;
}
