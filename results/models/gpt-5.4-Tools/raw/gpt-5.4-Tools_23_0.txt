use std::f64::consts::PI;
use std::io::Read;

const DT: f64 = 0.05;
const SAFE_VX: f64 = 1.2;
const SAFE_VY: f64 = 1.4;
#[derive(Clone, Copy, Debug)]
struct Params {
    gravity: f64,
    max_thrust: f64,
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug)]
struct TargetInfo {
    cx: f64,
    cy: f64,
    left: f64,
    right: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn norm_angle(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn next_token<'a, I: Iterator<Item = &'a str>>(it: &mut I) -> &'a str {
    it.next().expect("unexpected end of input")
}

fn main() {
    let mut input = Vec::new();
    std::io::stdin()
        .read_to_end(&mut input)
        .expect("failed to read stdin");
    let text = String::from_utf8(input).expect("stdin was not valid UTF-8");
    let mut it = text.split_whitespace();

    let width: usize = next_token(&mut it).parse().expect("width");
    let height: usize = next_token(&mut it).parse().expect("height");
    let gravity: f64 = next_token(&mut it).parse().expect("gravity");
    let max_thrust: f64 = next_token(&mut it).parse().expect("max thrust");
    let _fuel_capacity: f64 = next_token(&mut it).parse().expect("fuel");

    let start_x: f64 = next_token(&mut it).parse().expect("start x");
    let start_y: f64 = next_token(&mut it).parse().expect("start y");
    let target_x: f64 = next_token(&mut it).parse().expect("target x");
    let target_y: f64 = next_token(&mut it).parse().expect("target y");

    let mut target = TargetInfo {
        cx: target_x,
        cy: target_y,
        left: target_x - 0.5,
        right: target_x + 0.5,
    };

    let mut row_y = 0usize;
    while row_y < height {
        let row = next_token(&mut it);
        if row.contains('L') || row.contains('T') {
            let mut left = None;
            let mut right = None;
            for (x, ch) in row.bytes().enumerate() {
                if ch == b'L' || ch == b'T' {
                    if left.is_none() {
                        left = Some(x as f64);
                    }
                    right = Some(x as f64);
                }
            }
            if let (Some(l), Some(r)) = (left, right) {
                target.left = l;
                target.right = r + 1.0;
                target.cx = (target.left + target.right) * 0.5;
                target.cy = row_y as f64 + 0.5;
            }
        }
        row_y += 1;
    }

    let params = Params {
        gravity,
        max_thrust,
    };

    let mut outputs = Vec::new();
    let mut last_turn = 0.0_f64;

    while let Some(x_tok) = it.next() {
        let state = State {
            x: x_tok.parse().expect("x"),
            y: next_token(&mut it).parse().expect("y"),
            vx: next_token(&mut it).parse().expect("vx"),
            vy: next_token(&mut it).parse().expect("vy"),
            angle: next_token(&mut it).parse().expect("angle"),
            angular_vel: next_token(&mut it).parse().expect("ang vel"),
            fuel: next_token(&mut it).parse().expect("fuel rem"),
        };

        let (thrust, turn) = control_step(params, target, state, width as f64, height as f64, start_x, start_y, last_turn);
        last_turn = turn;
        outputs.push(format!("{:.4} {:.4}", thrust, turn));
    }

    if outputs.is_empty() {
        let hover = clamp(params.gravity / params.max_thrust, 0.0, 1.0);
        println!("{hover:.4} 0.0000");
    } else {
        print!("{}", outputs.join("\n"));
        if !outputs.is_empty() {
            println!();
        }
    }
}

fn control_step(
    params: Params,
    target: TargetInfo,
    state: State,
    width: f64,
    height: f64,
    start_x: f64,
    start_y: f64,
    last_turn: f64,
) -> (f64, f64) {
    let hover = clamp(params.gravity / params.max_thrust, 0.0, 1.0);
    let alt = target.cy - state.y;
    let horizontal_error = target.cx - state.x;
    let zone_half = ((target.right - target.left) * 0.5).max(1.5);

    let mut desired_vx = clamp(horizontal_error * 0.18, -8.0, 8.0);
    if alt < 18.0 {
        desired_vx = clamp(horizontal_error * 0.10, -2.0, 2.0);
    }
    if state.x < target.left - 5.0 {
        desired_vx = desired_vx.max(2.0);
    }
    if state.x > target.right + 5.0 {
        desired_vx = desired_vx.min(-2.0);
    }

    let vx_error = desired_vx - state.vx;
    let mut ax_cmd = clamp(vx_error * 0.9, -params.max_thrust * 0.75, params.max_thrust * 0.75);

    let mut desired_vy = if alt > 80.0 {
        clamp(0.25 * alt.sqrt() + 2.0, 4.0, 18.0)
    } else if alt > 30.0 {
        clamp(0.18 * alt + 1.0, 2.5, 9.0)
    } else {
        clamp(0.12 * alt + 0.4, 0.6, 3.0)
    };

    if horizontal_error.abs() > zone_half * 1.2 {
        desired_vy = desired_vy.min(4.0);
    }
    if state.vy > 14.0 && alt < 40.0 {
        desired_vy = desired_vy.min(1.0);
    }
    if alt < 12.0 {
        desired_vy = desired_vy.min(SAFE_VY);
        desired_vx = clamp(horizontal_error * 0.08, -1.0, 1.0);
        ax_cmd = clamp((desired_vx - state.vx) * 1.2, -params.max_thrust * 0.5, params.max_thrust * 0.5);
    }

    let ay_cmd = clamp((desired_vy - state.vy) * 0.9, -params.gravity * 0.3, params.max_thrust - params.gravity);
    let upward_acc = params.gravity + ay_cmd;

    let mut desired_angle = clamp((-ax_cmd / upward_acc).atan(), -1.2, 1.2);
    if alt < 8.0 {
        desired_angle = clamp(desired_angle, -0.35, 0.35);
    }
    if horizontal_error.abs() < 0.8 && alt < 6.0 {
        desired_angle = 0.0;
    }

    let angle_error = norm_angle(desired_angle - state.angle);
    let mut turn = clamp(angle_error * 2.6 - state.angular_vel * 0.8, -1.0, 1.0);
    if angle_error.abs() < 0.01 && state.angular_vel.abs() < 0.02 {
        turn = 0.0;
    }
    turn = clamp(turn * 0.85 + last_turn * 0.15, -1.0, 1.0);

    let effective_angle = clamp(state.angle + state.angular_vel * DT + turn * DT * 0.5, -1.4, 1.4);
    let cos_a = effective_angle.cos().max(0.08);
    let sin_a = effective_angle.sin();

    let thrust_x_need = ax_cmd / sin_a.abs().max(0.15);
    let thrust_y_need = upward_acc / cos_a;
    let mut thrust_acc = thrust_x_need.abs().min(params.max_thrust).max(thrust_y_need);

    if alt < 20.0 && (state.vx.abs() > SAFE_VX || state.vy > SAFE_VY) {
        thrust_acc = thrust_acc.max((params.gravity + 1.8).min(params.max_thrust));
    }
    if state.fuel < 1.0 && alt > 25.0 {
        thrust_acc = thrust_acc.min(params.gravity + 1.0);
    }
    if alt > 120.0 && state.y < start_y.min(target.cy) && (state.x - start_x).abs() < 3.0 {
        thrust_acc = thrust_acc.min(params.gravity + 0.4);
    }

    let mut thrust = clamp(thrust_acc / params.max_thrust, 0.0, 1.0);
    if alt < 4.0 && state.vy < 0.5 && state.vx.abs() < 0.5 {
        thrust = hover;
    }

    if state.x < 2.0 || state.x > width - 2.0 {
        desired_angle = if state.x < width * 0.5 { -0.5 } else { 0.5 };
        let boundary_error = norm_angle(desired_angle - state.angle);
        turn = clamp(boundary_error * 3.0 - state.angular_vel, -1.0, 1.0);
        thrust = thrust.max((params.gravity + 1.5) / params.max_thrust);
    }
    if state.y < 2.0 {
        thrust = 0.0;
    }
    if state.y > height - 3.0 {
        thrust = 1.0;
    }

    (clamp(thrust, 0.0, 1.0), clamp(turn, -1.0, 1.0))
}
