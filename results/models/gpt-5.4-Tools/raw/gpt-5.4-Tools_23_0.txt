use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Debug)]
struct Terrain {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    landing_left: f64,
    landing_right: f64,
    obstacle_tops: Vec<Option<usize>>,
}

impl Terrain {
    fn ground_y_at(&self, x: f64) -> f64 {
        let xi = x.floor().clamp(0.0, (self.width.saturating_sub(1)) as f64) as usize;
        match self.obstacle_tops[xi] {
            Some(y) => y as f64,
            None => self.height as f64,
        }
    }
}

fn clamp(value: f64, min: f64, max: f64) -> f64 {
    value.max(min).min(max)
}

fn normalize_angle(mut angle: f64) -> f64 {
    const PI: f64 = std::f64::consts::PI;
    while angle > PI {
        angle -= 2.0 * PI;
    }
    while angle < -PI {
        angle += 2.0 * PI;
    }
    angle
}

struct TokenReader<R: BufRead> {
    reader: R,
    buf: Vec<u8>,
    pos: usize,
}

impl<R: BufRead> TokenReader<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: Vec::new(),
            pos: 0,
        }
    }

    fn next_token(&mut self) -> io::Result<Option<String>> {
        loop {
            while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
                self.pos += 1;
            }
            if self.pos < self.buf.len() {
                let start = self.pos;
                while self.pos < self.buf.len() && !self.buf[self.pos].is_ascii_whitespace() {
                    self.pos += 1;
                }
                return Ok(Some(String::from_utf8_lossy(&self.buf[start..self.pos]).into_owned()));
            }

            self.buf.clear();
            self.pos = 0;
            let available = self.reader.fill_buf()?;
            if available.is_empty() {
                return Ok(None);
            }
            self.buf.extend_from_slice(available);
            let len = available.len();
            self.reader.consume(len);
        }
    }

    fn parse<T: std::str::FromStr>(&mut self, name: &'static str) -> io::Result<T> {
        let token = self
            .next_token()?
            .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, name))?;
        token
            .parse::<T>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, name))
    }
}

fn read_initial_input<R: BufRead>(reader: &mut TokenReader<R>) -> io::Result<Terrain> {
    let width: usize = reader.parse("missing width")?;
    let height: usize = reader.parse("missing height")?;
    let gravity: f64 = reader.parse("missing gravity")?;
    let max_thrust: f64 = reader.parse("missing max thrust")?;
    let initial_fuel: f64 = reader.parse("missing fuel")?;

    let start_x: f64 = reader.parse("missing start_x")?;
    let start_y: f64 = reader.parse("missing start_y")?;
    let target_x: f64 = reader.parse("missing target_x")?;
    let target_y: f64 = reader.parse("missing target_y")?;

    let mut obstacle_tops = vec![None; width];
    let mut landing_left = target_x.floor();
    let mut landing_right = target_x.ceil() + 1.0;

    for y in 0..height {
        let row = reader
            .next_token()?
            .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing map row"))?;
        for (x, ch) in row.bytes().enumerate().take(width) {
            if ch == b'#' || ch == b'L' || ch == b'T' {
                obstacle_tops[x].get_or_insert(y);
            }
        }

        if row.len() >= width {
            for (x, ch) in row.bytes().enumerate().take(width) {
                if ch == b'L' || ch == b'T' {
                    if landing_left > x as f64 {
                        landing_left = x as f64;
                    }
                    if landing_right < (x + 1) as f64 {
                        landing_right = (x + 1) as f64;
                    }
                }
            }
        }
    }

    Ok(Terrain {
        width,
        height,
        gravity,
        max_thrust,
        initial_fuel,
        start_x,
        start_y,
        target_x,
        target_y,
        landing_left,
        landing_right,
        obstacle_tops,
    })
}

fn read_state<R: BufRead>(reader: &mut TokenReader<R>) -> io::Result<Option<State>> {
    let x = match reader.next_token()? {
        Some(v) => v
            .parse::<f64>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid x"))?,
        None => return Ok(None),
    };
    let y: f64 = reader.parse("missing y")?;
    let vx: f64 = reader.parse("missing vx")?;
    let vy: f64 = reader.parse("missing vy")?;
    let angle: f64 = reader.parse("missing angle")?;
    let angular_vel: f64 = reader.parse("missing angular_vel")?;
    let fuel: f64 = reader.parse("missing fuel_remaining")?;
    Ok(Some(State {
        x,
        y,
        vx,
        vy,
        angle,
        angular_vel,
        fuel,
    }))
}

fn desired_horizontal_speed(dx: f64, altitude: f64) -> f64 {
    let gain = if altitude > 80.0 {
        0.55
    } else if altitude > 30.0 {
        0.40
    } else {
        0.20
    };
    clamp(dx * gain, -12.0, 12.0)
}

fn desired_vertical_speed(altitude: f64) -> f64 {
    if altitude > 120.0 {
        18.0
    } else if altitude > 60.0 {
        10.0
    } else if altitude > 25.0 {
        6.0
    } else if altitude > 10.0 {
        3.5
    } else {
        1.2
    }
}

fn controller(state: State, terrain: &Terrain) -> (f64, f64) {
    let landing_center = (terrain.landing_left + terrain.landing_right) * 0.5;
    let dx = landing_center - state.x;
    let ground_y = terrain.ground_y_at(state.x);
    let altitude = (ground_y - state.y - 0.5).max(0.0);

    let target_vx = desired_horizontal_speed(dx, altitude);
    let target_vy = desired_vertical_speed(altitude);
    let mut horiz_acc = (target_vx - state.vx) * 1.2;
    let mut vert_acc = (target_vy - state.vy) * 1.4;

    if altitude < 20.0 {
        horiz_acc = clamp((0.0 - state.vx) * 1.8 + dx * 0.15, -3.0, 3.0);
        vert_acc = clamp((1.0 - state.vy) * 2.2, -terrain.gravity * 0.4, terrain.max_thrust);
    }

    let lateral_bias = if state.x < terrain.landing_left {
        1.5
    } else if state.x > terrain.landing_right {
        -1.5
    } else {
        0.0
    };
    horiz_acc += lateral_bias;

    let ax = clamp(horiz_acc, -terrain.max_thrust * 0.8, terrain.max_thrust * 0.8);
    let ay = clamp(terrain.gravity + vert_acc, 0.1, terrain.max_thrust);

    let mut desired_angle = ax.atan2(-ay);
    let max_tilt = if altitude < 15.0 { 0.18 } else if altitude < 40.0 { 0.45 } else { 0.75 };
    desired_angle = clamp(desired_angle, -max_tilt, max_tilt);

    if altitude < 8.0 {
        desired_angle = clamp(dx * 0.06 - state.vx * 0.08, -0.12, 0.12);
    }

    let total_acc = (ax * ax + ay * ay).sqrt();
    let mut thrust = clamp(total_acc / terrain.max_thrust, 0.0, 1.0);
    if state.fuel < 0.5 {
        thrust = thrust.min(0.65);
    }
    if altitude < 5.0 && state.vy > 1.5 {
        thrust = thrust.max(0.85);
    }

    let angle_error = normalize_angle(desired_angle - state.angle);
    let turn = clamp(angle_error * 2.4 - state.angular_vel * 0.7, -1.0, 1.0);
    (thrust, turn)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let locked = BufReader::new(stdin.lock());
    let mut token_reader = TokenReader::new(locked);
    let terrain = read_initial_input(&mut token_reader)?;
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut latest = State {
        x: terrain.start_x,
        y: terrain.start_y,
        fuel: terrain.initial_fuel,
        ..State::default()
    };

    let tick = Duration::from_millis(40);

    loop {
        let loop_start = Instant::now();

        match read_state(&mut token_reader)? {
            Some(state) => latest = state,
            None => return Ok(()),
        }

        let (thrust, turn) = controller(latest, &terrain);
        writeln!(out, "{:.3} {:.3}", thrust, turn)?;
        out.flush()?;

        let elapsed = loop_start.elapsed();
        if elapsed < tick {
            std::thread::sleep(tick - elapsed);
        }
    }
}
