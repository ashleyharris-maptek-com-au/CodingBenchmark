constexpr double kPi = 3.141592653589793238462643383279502884;
constexpr double kSqrt2 = 1.41421356237309504880168872420969808;
constexpr double kTetHeight = std::sqrt(2.0 / 3.0);
constexpr double kTetCircumradius = std::sqrt(3.0 / 8.0);
constexpr double kTetInradius = 1.0 / std::sqrt(24.0);
constexpr double kTetVolume = std::sqrt(2.0) / 12.0;
constexpr double kEps = 1e-9;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& other) {
        x += other.x;
        y += other.y;
        z += other.z;
        return *this;
    }
};

Vec3 operator*(double s, const Vec3& v) { return v * s; }

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double norm2(const Vec3& v) {
    return dot(v, v);
}

double norm(const Vec3& v) {
    return std::sqrt(norm2(v));
}

Vec3 normalized(const Vec3& v) {
    const double n = norm(v);
    if (n <= kEps) {
        return {0.0, 0.0, 0.0};
    }
    return v / n;
}

struct Mat3 {
    double m[3][3]{};

    static Mat3 identity() {
        Mat3 r;
        r.m[0][0] = r.m[1][1] = r.m[2][2] = 1.0;
        return r;
    }

    Vec3 operator*(const Vec3& v) const {
        return {
            m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z,
            m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z,
            m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z
        };
    }
};

Mat3 transpose(const Mat3& a) {
    Mat3 r;
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < 3; ++j) {
            r.m[i][j] = a.m[j][i];
        }
    }
    return r;
}

Mat3 operator*(const Mat3& a, const Mat3& b) {
    Mat3 r;
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < 3; ++j) {
            double value = 0.0;
            for (int k = 0; k < 3; ++k) {
                value += a.m[i][k] * b.m[k][j];
            }
            r.m[i][j] = value;
        }
    }
    return r;
}

Mat3 rotationFromAxisAngle(const Vec3& axisRaw, double angle) {
    Vec3 axis = normalized(axisRaw);
    if (norm2(axis) <= kEps || std::abs(angle) <= kEps) {
        return Mat3::identity();
    }
    const double c = std::cos(angle);
    const double s = std::sin(angle);
    const double t = 1.0 - c;
    const double x = axis.x;
    const double y = axis.y;
    const double z = axis.z;
    Mat3 r;
    r.m[0][0] = t * x * x + c;
    r.m[0][1] = t * x * y - s * z;
    r.m[0][2] = t * x * z + s * y;
    r.m[1][0] = t * x * y + s * z;
    r.m[1][1] = t * y * y + c;
    r.m[1][2] = t * y * z - s * x;
    r.m[2][0] = t * x * z - s * y;
    r.m[2][1] = t * y * z + s * x;
    r.m[2][2] = t * z * z + c;
    return r;
}

Mat3 rotationAlignZTo(const Vec3& targetRaw) {
    Vec3 target = normalized(targetRaw);
    if (norm2(target) <= kEps) {
        return Mat3::identity();
    }
    const Vec3 zAxis{0.0, 0.0, 1.0};
    const double c = std::clamp(dot(zAxis, target), -1.0, 1.0);
    if (c > 1.0 - 1e-12) {
        return Mat3::identity();
    }
    if (c < -1.0 + 1e-12) {
        return rotationFromAxisAngle({1.0, 0.0, 0.0}, kPi);
    }
    const Vec3 axis = cross(zAxis, target);
    const double angle = std::acos(c);
    return rotationFromAxisAngle(axis, angle);
}

std::array<double, 3> matrixToEulerXYZ(const Mat3& r) {
    // Intrinsic XYZ angles, matching R = Rz(rz) * Ry(ry) * Rx(rx).
    double ry = std::asin(std::clamp(-r.m[2][0], -1.0, 1.0));
    double rx = 0.0;
    double rz = 0.0;
    if (std::abs(std::cos(ry)) > 1e-9) {
        rx = std::atan2(r.m[2][1], r.m[2][2]);
        rz = std::atan2(r.m[1][0], r.m[0][0]);
    } else {
        rx = std::atan2(-r.m[1][2], r.m[1][1]);
        rz = 0.0;
    }
    return {rx, ry, rz};
}

struct Face {
    int a = -1;
    int b = -1;
    int c = -1;
    Vec3 normal;
    double d = 0.0;
};

struct Plane {
    Vec3 n;
    double d = 0.0;
};

struct Placement {
    Vec3 center;
    std::array<double, 3> euler{};
    Mat3 rotation;
    std::array<Vec3, 4> vertices;
};

struct CellKey {
    int x = 0;
    int y = 0;
    int z = 0;

    bool operator==(const CellKey& other) const {
        return x == other.x && y == other.y && z == other.z;
    }
};

struct CellKeyHasher {
    std::size_t operator()(const CellKey& key) const noexcept {
        std::size_t h = 1469598103934665603ull;
        h ^= static_cast<std::size_t>(key.x + 0x9e3779b9);
        h *= 1099511628211ull;
        h ^= static_cast<std::size_t>(key.y + 0x85ebca6b);
        h *= 1099511628211ull;
        h ^= static_cast<std::size_t>(key.z + 0xc2b2ae35);
        h *= 1099511628211ull;
        return h;
    }
};

struct SpatialHash {
    double cell = 1.25;
    std::unordered_map<CellKey, std::vector<int>, CellKeyHasher> buckets;
    std::vector<Vec3> centers;

    explicit SpatialHash(double cellSize) : cell(cellSize) {}

    CellKey keyFor(const Vec3& p) const {
        return {
            static_cast<int>(std::floor(p.x / cell)),
            static_cast<int>(std::floor(p.y / cell)),
            static_cast<int>(std::floor(p.z / cell))
        };
    }

    void insert(const Vec3& p, int index) {
        centers.push_back(p);
        buckets[keyFor(p)].push_back(index);
    }

    template <typename Fn>
    void forNearby(const Vec3& p, Fn&& fn) const {
        const CellKey base = keyFor(p);
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    CellKey key{base.x + dx, base.y + dy, base.z + dz};
                    auto it = buckets.find(key);
                    if (it == buckets.end()) {
                        continue;
                    }
                    for (int idx : it->second) {
                        fn(idx);
                    }
                }
            }
        }
    }
};

struct Hull {
    std::vector<Vec3> vertices;
    std::vector<Face> faces;
    std::vector<Plane> planes;
    Vec3 minBound{0.0, 0.0, 0.0};
    Vec3 maxBound{0.0, 0.0, 0.0};
    Vec3 centroid{0.0, 0.0, 0.0};
    double approxVolume = 0.0;
};

bool sameFace(const Face& f, int a, int b, int c) {
    std::array<int, 3> lhs{f.a, f.b, f.c};
    std::array<int, 3> rhs{a, b, c};
    std::sort(lhs.begin(), lhs.end());
    std::sort(rhs.begin(), rhs.end());
    return lhs == rhs;
}

Hull buildHull(const std::vector<Vec3>& points) {
    Hull hull;
    hull.vertices = points;
    if (points.empty()) {
        return hull;
    }

    hull.minBound = hull.maxBound = points.front();
    for (const Vec3& p : points) {
        hull.centroid += p;
        hull.minBound.x = std::min(hull.minBound.x, p.x);
        hull.minBound.y = std::min(hull.minBound.y, p.y);
        hull.minBound.z = std::min(hull.minBound.z, p.z);
        hull.maxBound.x = std::max(hull.maxBound.x, p.x);
        hull.maxBound.y = std::max(hull.maxBound.y, p.y);
        hull.maxBound.z = std::max(hull.maxBound.z, p.z);
    }
    hull.centroid = hull.centroid / static_cast<double>(points.size());

    const int n = static_cast<int>(points.size());
    std::vector<Face> faces;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 normal = cross(points[j] - points[i], points[k] - points[i]);
                const double len = norm(normal);
                if (len <= 1e-10) {
                    continue;
                }
                normal = normal / len;
                double pos = 0.0;
                double neg = 0.0;
                for (int t = 0; t < n; ++t) {
                    if (t == i || t == j || t == k) {
                        continue;
                    }
                    const double side = dot(normal, points[t] - points[i]);
                    if (side > 1e-8) {
                        pos += 1.0;
                    } else if (side < -1e-8) {
                        neg += 1.0;
                    }
                    if (pos > 0.0 && neg > 0.0) {
                        break;
                    }
                }
                if (pos > 0.0 && neg > 0.0) {
                    continue;
                }
                Face face;
                if (pos == 0.0) {
                    face = {i, j, k, normal, dot(normal, points[i])};
                } else {
                    face = {i, k, j, -1.0 * normal, dot(-1.0 * normal, points[i])};
                }
                bool duplicate = false;
                for (const Face& existing : faces) {
                    if (sameFace(existing, face.a, face.b, face.c)) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) {
                    faces.push_back(face);
                }
            }
        }
    }

    hull.faces = faces;
    hull.planes.reserve(faces.size());
    for (const Face& face : faces) {
        hull.planes.push_back({face.normal, face.d});
        const Vec3& a = points[face.a];
        const Vec3& b = points[face.b];
        const Vec3& c = points[face.c];
        hull.approxVolume += dot(a, cross(b, c)) / 6.0;
    }
    hull.approxVolume = std::abs(hull.approxVolume);
    return hull;
}

bool insideHull(const Hull& hull, const Vec3& p, double margin = 0.0) {
    for (const Plane& plane : hull.planes) {
        if (dot(plane.n, p) > plane.d - margin) {
            return false;
        }
    }
    return true;
}

Vec3 supportPoint(const std::array<Vec3, 4>& verts, const Vec3& dir) {
    double best = dot(verts[0], dir);
    Vec3 result = verts[0];
    for (int i = 1; i < 4; ++i) {
        const double value = dot(verts[i], dir);
        if (value > best) {
            best = value;
            result = verts[i];
        }
    }
    return result;
}

Vec3 supportMinkowski(const std::array<Vec3, 4>& a, const std::array<Vec3, 4>& b, const Vec3& dir) {
    return supportPoint(a, dir) - supportPoint(b, -1.0 * dir);
}

bool sameDirection(const Vec3& dir, const Vec3& ao) {
    return dot(dir, ao) > 0.0;
}

bool handleLine(std::vector<Vec3>& simplex, Vec3& dir) {
    const Vec3 a = simplex.back();
    const Vec3 b = simplex[simplex.size() - 2];
    const Vec3 ab = b - a;
    const Vec3 ao = -1.0 * a;
    if (sameDirection(ab, ao)) {
        dir = cross(cross(ab, ao), ab);
    } else {
        simplex = {a};
        dir = ao;
    }
    return false;
}

bool handleTriangle(std::vector<Vec3>& simplex, Vec3& dir) {
    const Vec3 a = simplex[2];
    const Vec3 b = simplex[1];
    const Vec3 c = simplex[0];
    const Vec3 ab = b - a;
    const Vec3 ac = c - a;
    const Vec3 ao = -1.0 * a;
    const Vec3 abc = cross(ab, ac);

    const Vec3 abPerp = cross(abc, ab);
    if (sameDirection(abPerp, ao)) {
        simplex = {b, a};
        dir = cross(cross(ab, ao), ab);
        return false;
    }

    const Vec3 acPerp = cross(ac, abc);
    if (sameDirection(acPerp, ao)) {
        simplex = {c, a};
        dir = cross(cross(ac, ao), ac);
        return false;
    }

    if (sameDirection(abc, ao)) {
        dir = abc;
    } else {
        simplex = {b, c, a};
        dir = -1.0 * abc;
    }
    return false;
}

bool handleTetra(std::vector<Vec3>& simplex, Vec3& dir) {
    const Vec3 a = simplex[3];
    const Vec3 b = simplex[2];
    const Vec3 c = simplex[1];
    const Vec3 d = simplex[0];
    const Vec3 ao = -1.0 * a;

    const Vec3 abc = cross(b - a, c - a);
    if (sameDirection(abc, ao)) {
        simplex = {c, b, a};
        dir = abc;
        return false;
    }

    const Vec3 acd = cross(c - a, d - a);
    if (sameDirection(acd, ao)) {
        simplex = {d, c, a};
        dir = acd;
        return false;
    }

    const Vec3 adb = cross(d - a, b - a);
    if (sameDirection(adb, ao)) {
        simplex = {b, d, a};
        dir = adb;
        return false;
    }

    return true;
}

bool nextSimplex(std::vector<Vec3>& simplex, Vec3& dir) {
    switch (simplex.size()) {
        case 2: return handleLine(simplex, dir);
        case 3: return handleTriangle(simplex, dir);
        case 4: return handleTetra(simplex, dir);
        default: return false;
    }
}

bool tetrahedraOverlap(const std::array<Vec3, 4>& a, const std::array<Vec3, 4>& b) {
    Vec3 dir{1.0, 0.0, 0.0};
    std::vector<Vec3> simplex;
    simplex.reserve(4);
    simplex.push_back(supportMinkowski(a, b, dir));
    dir = -1.0 * simplex[0];
    for (int iter = 0; iter < 32; ++iter) {
        Vec3 p = supportMinkowski(a, b, dir);
        if (dot(p, dir) < 1e-9) {
            return false;
        }
        simplex.push_back(p);
        if (nextSimplex(simplex, dir)) {
            return true;
        }
        if (norm2(dir) <= 1e-16) {
            return true;
        }
    }
    return false;
}

std::array<Vec3, 4> canonicalTetra() {
    // Centered canonical regular tetrahedron with unit edge length.
    const double s = 1.0 / (2.0 * kSqrt2);
    return {
        Vec3{ s,  s,  s},
        Vec3{ s, -s, -s},
        Vec3{-s,  s, -s},
        Vec3{-s, -s,  s}
    };
}

Placement makePlacement(const Vec3& center, const Vec3& axis, double spin, const std::array<Vec3, 4>& canonical) {
    const Mat3 align = rotationAlignZTo(axis);
    const Mat3 twist = rotationFromAxisAngle(axis, spin);
    const Mat3 rot = twist * align;
    Placement placement;
    placement.center = center;
    placement.rotation = rot;
    placement.euler = matrixToEulerXYZ(rot);
    for (int i = 0; i < 4; ++i) {
        placement.vertices[i] = center + rot * canonical[i];
    }
    return placement;
}

bool fitsInside(const Hull& hull, const Placement& placement) {
    for (const Vec3& v : placement.vertices) {
        if (!insideHull(hull, v, 1e-8)) {
            return false;
        }
    }
    return true;
}

struct Candidate {
    Vec3 center;
    Vec3 axis;
    double spin = 0.0;
    double score = 0.0;
};

std::vector<Vec3> buildAxisSet() {
    std::vector<Vec3> axes;
    axes.push_back({0.0, 0.0, 1.0});
    axes.push_back({0.0, 0.0, -1.0});
    const double inv = 1.0 / std::sqrt(3.0);
    for (int sx : {-1, 1}) {
        for (int sy : {-1, 1}) {
            for (int sz : {-1, 1}) {
                axes.push_back({sx * inv, sy * inv, sz * inv});
            }
        }
    }
    axes.push_back(normalized(Vec3{1.0, 1.0, 0.0}));
    axes.push_back(normalized(Vec3{1.0, -1.0, 0.0}));
    axes.push_back(normalized(Vec3{1.0, 0.0, 1.0}));
    axes.push_back(normalized(Vec3{1.0, 0.0, -1.0}));
    axes.push_back(normalized(Vec3{0.0, 1.0, 1.0}));
    axes.push_back(normalized(Vec3{0.0, 1.0, -1.0}));
    return axes;
}

std::vector<double> buildSpins() {
    return {0.0, kPi / 3.0, 2.0 * kPi / 3.0, kPi, 4.0 * kPi / 3.0, 5.0 * kPi / 3.0};
}

std::vector<Vec3> buildGridCenters(const Hull& hull, std::size_t budget) {
    std::vector<Vec3> centers;
    const Vec3 size = hull.maxBound - hull.minBound;
    const double volumeBox = std::max(1e-9, size.x * size.y * size.z);
    double spacing = std::cbrt(volumeBox / std::max<std::size_t>(budget, 1));
    spacing = std::max(0.55, spacing);
    spacing = std::min(spacing, 2.5);

    for (double z = hull.minBound.z + kTetCircumradius; z <= hull.maxBound.z - kTetCircumradius + 1e-9; z += spacing) {
        for (double y = hull.minBound.y + kTetCircumradius; y <= hull.maxBound.y - kTetCircumradius + 1e-9; y += spacing) {
            for (double x = hull.minBound.x + kTetCircumradius; x <= hull.maxBound.x - kTetCircumradius + 1e-9; x += spacing) {
                Vec3 p{x, y, z};
                if (insideHull(hull, p, kTetCircumradius + 1e-6)) {
                    centers.push_back(p);
                    if (centers.size() >= budget) {
                        return centers;
                    }
                }
            }
        }
    }
    return centers;
}

std::vector<Vec3> buildFaceOffsets(const Hull& hull) {
    std::vector<Vec3> points;
    points.reserve(hull.faces.size());
    for (const Face& face : hull.faces) {
        const Vec3 a = hull.vertices[face.a];
        const Vec3 b = hull.vertices[face.b];
        const Vec3 c = hull.vertices[face.c];
        const Vec3 centroid = (a + b + c) / 3.0;
        const Vec3 inward = centroid - face.normal * (kTetCircumradius + 0.05);
        if (insideHull(hull, inward, kTetInradius)) {
            points.push_back(inward);
        }
    }
    return points;
}

std::vector<Vec3> buildVertexOffsets(const Hull& hull) {
    std::vector<Vec3> points;
    points.reserve(hull.vertices.size());
    for (const Vec3& v : hull.vertices) {
        Vec3 dir = normalized(hull.centroid - v);
        Vec3 p = v + dir * (kTetCircumradius + 0.05);
        if (insideHull(hull, p, kTetInradius)) {
            points.push_back(p);
        }
    }
    return points;
}

bool conflictWithExisting(const Placement& candidate, const std::vector<Placement>& accepted, const SpatialHash& hash) {
    bool conflict = false;
    hash.forNearby(candidate.center, [&](int idx) {
        if (conflict) {
            return;
        }
        const Vec3 delta = accepted[idx].center - candidate.center;
        if (norm2(delta) > 2.1 * 2.1) {
            return;
        }
        if (tetrahedraOverlap(candidate.vertices, accepted[idx].vertices)) {
            conflict = true;
        }
    });
    return conflict;
}

std::vector<Placement> greedyPack(const Hull& hull, std::size_t targetCount, double timeLimitSeconds) {
    const auto canonical = canonicalTetra();
    const auto axes = buildAxisSet();
    const auto spins = buildSpins();

    std::vector<Vec3> seeds;
    const std::size_t gridBudget = std::min<std::size_t>(std::max<std::size_t>(targetCount * 3, 128), 250000);
    std::vector<Vec3> gridCenters = buildGridCenters(hull, gridBudget);
    std::vector<Vec3> faceCenters = buildFaceOffsets(hull);
    std::vector<Vec3> vertexCenters = buildVertexOffsets(hull);
    seeds.reserve(faceCenters.size() + vertexCenters.size() + gridCenters.size() + 1);
    seeds.push_back(hull.centroid);
    seeds.insert(seeds.end(), faceCenters.begin(), faceCenters.end());
    seeds.insert(seeds.end(), vertexCenters.begin(), vertexCenters.end());
    seeds.insert(seeds.end(), gridCenters.begin(), gridCenters.end());

    std::vector<Candidate> candidates;
    candidates.reserve(std::min<std::size_t>(seeds.size() * 4, 400000));
    for (const Vec3& seed : seeds) {
        for (std::size_t i = 0; i < axes.size(); ++i) {
            const Vec3& axis = axes[i];
            const double spin = spins[i % spins.size()];
            double wallClearance = std::numeric_limits<double>::max();
            for (const Plane& plane : hull.planes) {
                wallClearance = std::min(wallClearance, plane.d - dot(plane.n, seed));
            }
            double score = seed.z * 0.05 + wallClearance;
            candidates.push_back({seed, axis, spin, score});
        }
    }

    std::sort(candidates.begin(), candidates.end(), [](const Candidate& lhs, const Candidate& rhs) {
        return lhs.score > rhs.score;
    });

    const auto start = std::chrono::steady_clock::now();
    std::vector<Placement> accepted;
    accepted.reserve(targetCount);
    SpatialHash hash(1.25);

    for (const Candidate& cand : candidates) {
        const double elapsed = std::chrono::duration<double>(std::chrono::steady_clock::now() - start).count();
        if (elapsed > timeLimitSeconds) {
            break;
        }
        if (accepted.size() >= targetCount) {
            break;
        }
        Placement placement = makePlacement(cand.center, cand.axis, cand.spin, canonical);
        if (!fitsInside(hull, placement)) {
            continue;
        }
        if (conflictWithExisting(placement, accepted, hash)) {
            continue;
        }
        hash.insert(placement.center, static_cast<int>(accepted.size()));
        accepted.push_back(placement);
    }

    return accepted;
}

    int vCount = 0;
    double edgeLength = 1.0;
    if (!(std::cin >> vCount >> edgeLength)) {
        return 0;
    }

    std::vector<Vec3> points;
    points.reserve(std::max(vCount, 0));
    for (int i = 0; i < vCount; ++i) {
        double x = 0.0;
        double y = 0.0;
        double z = 0.0;
        std::cin >> x >> y >> z;
        points.push_back({x, y, z});
    }

    if (points.size() < 4) {
        std::cout << 0 << '\n';
        return 0;
    }

    Hull hull = buildHull(points);
    if (hull.faces.empty() || hull.approxVolume <= 1e-12) {
        std::cout << 0 << '\n';
        return 0;
    }

    const double scale = edgeLength;
    hull.minBound = hull.minBound / scale;
    hull.maxBound = hull.maxBound / scale;
    hull.centroid = hull.centroid / scale;
    for (Vec3& p : hull.vertices) {
        p = p / scale;
    }
    hull.faces.clear();
    hull.planes.clear();
    hull = buildHull(hull.vertices);

    const double theoreticalMax = hull.approxVolume / kTetVolume;
    std::size_t targetCount = 0;
    if (theoreticalMax < 50.0) {
        targetCount = static_cast<std::size_t>(theoreticalMax * 1.2 + 32.0);
    } else if (theoreticalMax < 5000.0) {
        targetCount = static_cast<std::size_t>(std::sqrt(theoreticalMax) * 20.0 + 128.0);
    } else {
        targetCount = static_cast<std::size_t>(std::min(200000.0, 512.0 + 12.0 * std::cbrt(theoreticalMax) * std::cbrt(theoreticalMax)));
    }
    targetCount = std::max<std::size_t>(targetCount, 32);

    const double timeLimitSeconds = (theoreticalMax > 1e6) ? 8.0 : 18.0;
    std::vector<Placement> placements = greedyPack(hull, targetCount, timeLimitSeconds);

    std::cout << placements.size() << '\n';
    std::cout << std::fixed << std::setprecision(9);
    for (const Placement& placement : placements) {
        const Vec3 outCenter = placement.center * scale;
        std::cout << outCenter.x << ' ' << outCenter.y << ' ' << outCenter.z << ' '
                  << placement.euler[0] << ' ' << placement.euler[1] << ' ' << placement.euler[2] << '\n';
    }
    return 0;
}
