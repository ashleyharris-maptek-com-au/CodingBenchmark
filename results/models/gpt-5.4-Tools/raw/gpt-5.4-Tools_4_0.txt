constexpr double PI = 3.141592653589793238462643383279502884;
constexpr double EPS = 1e-9;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& other) {
        x += other.x;
        y += other.y;
        z += other.z;
        return *this;
    }
};

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double norm2(const Vec3& v) {
    return dot(v, v);
}

double norm(const Vec3& v) {
    return sqrt(norm2(v));
}

Vec3 normalized(const Vec3& v) {
    const double n = norm(v);
    if (n < EPS) return {};
    return v / n;
}

struct Plane {
    Vec3 n;
    double d = 0.0;
};

struct Face {
    array<int, 3> idx{};
    Plane plane;
};

struct AABB {
    Vec3 mn, mx;
};

struct Placement {
    Vec3 c;
    Vec3 euler;
};

struct GridKey {
    int x = 0;
    int y = 0;
    int z = 0;

    bool operator==(const GridKey& other) const {
        return x == other.x && y == other.y && z == other.z;
    }
};

struct GridKeyHasher {
    size_t operator()(const GridKey& k) const noexcept {
        uint64_t h = 1469598103934665603ull;
        auto mix = [&](uint64_t v) {
            h ^= v + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        };
        mix(static_cast<uint32_t>(k.x));
        mix(static_cast<uint32_t>(k.y));
        mix(static_cast<uint32_t>(k.z));
        return static_cast<size_t>(h);
    }
};

struct SpatialGrid {
    double cell = 1.0;
    unordered_map<GridKey, vector<int>, GridKeyHasher> cells;
    vector<Vec3> points;

    explicit SpatialGrid(double cellSize) : cell(cellSize) {}

    GridKey keyOf(const Vec3& p) const {
        return {
            static_cast<int>(floor(p.x / cell)),
            static_cast<int>(floor(p.y / cell)),
            static_cast<int>(floor(p.z / cell))
        };
    }

    bool hasNeighborWithin(const Vec3& p, double minDist) const {
        const GridKey k = keyOf(p);
        const double minDist2 = minDist * minDist;
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    GridKey q{k.x + dx, k.y + dy, k.z + dz};
                    auto it = cells.find(q);
                    if (it == cells.end()) continue;
                    for (int idx : it->second) {
                        if (norm2(points[idx] - p) < minDist2) return true;
                    }
                }
            }
        }
        return false;
    }

    void addPoint(const Vec3& p) {
        const int idx = static_cast<int>(points.size());
        points.push_back(p);
        cells[keyOf(p)].push_back(idx);
    }
};

Vec3 rotateXYZ(const Vec3& p, const Vec3& ang) {
    const double cx = cos(ang.x), sx = sin(ang.x);
    const double cy = cos(ang.y), sy = sin(ang.y);
    const double cz = cos(ang.z), sz = sin(ang.z);

    Vec3 r = p;
    r = {r.x, cx * r.y - sx * r.z, sx * r.y + cx * r.z};
    r = {cy * r.x + sy * r.z, r.y, -sy * r.x + cy * r.z};
    r = {cz * r.x - sz * r.y, sz * r.x + cz * r.y, r.z};
    return r;
}

double tetraVolume(double edge) {
    return edge * edge * edge / (6.0 * sqrt(2.0));
}

vector<Vec3> baseTetraVertices(double edge) {
    const double s = edge / (2.0 * sqrt(2.0));
    return {
        { s,  s,  s},
        { s, -s, -s},
        {-s,  s, -s},
        {-s, -s,  s}
    };
}

vector<Vec3> orientedVertices(const Vec3& center, const Vec3& euler, double edge) {
    const vector<Vec3> base = baseTetraVertices(edge);
    vector<Vec3> out;
    out.reserve(4);
    for (const Vec3& v : base) out.push_back(center + rotateXYZ(v, euler));
    return out;
}

Plane makePlane(const Vec3& a, const Vec3& b, const Vec3& c, const Vec3& insidePoint) {
    Vec3 n = cross(b - a, c - a);
    const double len = norm(n);
    if (len < EPS) return {};
    n = n / len;
    double d = -dot(n, a);
    if (dot(n, insidePoint) + d > 0.0) {
        n = n * -1.0;
        d = -d;
    }
    return {n, d};
}

bool pointInConvexPolyhedron(const Vec3& p, const vector<Plane>& planes, double margin = 0.0) {
    for (const Plane& pl : planes) {
        if (dot(pl.n, p) + pl.d > -margin + 1e-8) return false;
    }
    return true;
}

bool tetraInsideContainer(const Vec3& center, const Vec3& euler, double edge, const vector<Plane>& planes) {
    const vector<Vec3> verts = orientedVertices(center, euler, edge);
    for (const Vec3& v : verts) {
        if (!pointInConvexPolyhedron(v, planes, 0.0)) return false;
    }
    return true;
}

vector<Face> buildConvexHullFaces(const vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    vector<Face> faces;
    if (n < 4) return faces;

    Vec3 centroid;
    for (const Vec3& p : pts) centroid += p;
    centroid = centroid / static_cast<double>(n);

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 nrm = cross(pts[j] - pts[i], pts[k] - pts[i]);
                if (norm(nrm) < 1e-10) continue;

                int pos = 0, neg = 0;
                for (int t = 0; t < n; ++t) {
                    if (t == i || t == j || t == k) continue;
                    const double s = dot(nrm, pts[t] - pts[i]);
                    if (s > 1e-8) ++pos;
                    else if (s < -1e-8) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                Plane pl = makePlane(pts[i], pts[j], pts[k], centroid);
                bool duplicate = false;
                for (const Face& f : faces) {
                    double alignment = dot(f.plane.n, pl.n);
                    if (alignment > 1.0 - 1e-8 && fabs(f.plane.d - pl.d) < 1e-8) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) {
                    faces.push_back({{i, j, k}, pl});
                }
            }
        }
    }
    return faces;
}

vector<Plane> extractPlanes(const vector<Face>& faces) {
    vector<Plane> planes;
    planes.reserve(faces.size());
    for (const Face& f : faces) planes.push_back(f.plane);
    return planes;
}

AABB computeAABB(const vector<Vec3>& pts) {
    AABB box;
    box.mn = box.mx = pts.front();
    for (const Vec3& p : pts) {
        box.mn.x = min(box.mn.x, p.x);
        box.mn.y = min(box.mn.y, p.y);
        box.mn.z = min(box.mn.z, p.z);
        box.mx.x = max(box.mx.x, p.x);
        box.mx.y = max(box.mx.y, p.y);
        box.mx.z = max(box.mx.z, p.z);
    }
    return box;
}

double aabbVolume(const AABB& b) {
    return max(0.0, b.mx.x - b.mn.x) * max(0.0, b.mx.y - b.mn.y) * max(0.0, b.mx.z - b.mn.z);
}

double estimateVolumeMonteCarlo(const vector<Plane>& planes, const AABB& box, uint64_t samples, mt19937_64& rng) {
    if (samples == 0) return 0.0;
    uniform_real_distribution<double> dx(box.mn.x, box.mx.x);
    uniform_real_distribution<double> dy(box.mn.y, box.mx.y);
    uniform_real_distribution<double> dz(box.mn.z, box.mx.z);

    uint64_t inside = 0;
    for (uint64_t i = 0; i < samples; ++i) {
        Vec3 p{dx(rng), dy(rng), dz(rng)};
        if (pointInConvexPolyhedron(p, planes)) ++inside;
    }
    return aabbVolume(box) * static_cast<double>(inside) / static_cast<double>(samples);
}

vector<Vec3> candidateOrientations() {
    vector<Vec3> out;
    const array<double, 4> vals = {0.0, PI * 0.5, PI, PI * 1.5};
    for (double ax : vals) {
        for (double ay : vals) {
            for (double az : vals) {
                out.push_back({ax, ay, az});
            }
        }
    }
    out.push_back({PI / 4.0, acos(1.0 / 3.0), PI / 4.0});
    out.push_back({-PI / 4.0, -acos(1.0 / 3.0), PI / 2.0});
    out.push_back({0.9553166181245093, 0.7853981633974483, 0.0});
    out.push_back({-0.9553166181245093, 2.356194490192345, PI / 2.0});
    return out;
}

vector<Placement> generatePlacements(
    const vector<Plane>& planes,
    const AABB& box,
    double edge,
    double estVolume,
    mt19937_64& rng
) {
    const double circumradius = edge * sqrt(3.0 / 8.0);
    const double minCenterDist = 2.0 * circumradius + 1e-6;
    const double baseStep = max(edge * 0.52, circumradius * 0.95);
    const double targetByVolume = estVolume > 0.0 ? estVolume / tetraVolume(edge) : 0.0;

    double step = baseStep;
    if (targetByVolume > 200000.0) step *= 4.0;
    else if (targetByVolume > 50000.0) step *= 2.5;
    else if (targetByVolume > 10000.0) step *= 1.75;
    else if (targetByVolume < 500.0) step *= 0.9;

    const vector<Vec3> orientations = candidateOrientations();
    SpatialGrid grid(minCenterDist);
    vector<Placement> result;

    const Vec3 dims = box.mx - box.mn;
    const uint64_t nx = max<uint64_t>(1, static_cast<uint64_t>(floor(dims.x / step)) + 1);
    const uint64_t ny = max<uint64_t>(1, static_cast<uint64_t>(floor(dims.y / step)) + 1);
    const uint64_t nz = max<uint64_t>(1, static_cast<uint64_t>(floor(dims.z / step)) + 1);
    const long double gridCells = static_cast<long double>(nx) * static_cast<long double>(ny) * static_cast<long double>(nz);

    const bool huge = gridCells > 3000000.0L;
    const uint64_t randomBudget = huge ? 250000ull : 0ull;

    auto tryPlace = [&](const Vec3& c) {
        if (grid.hasNeighborWithin(c, minCenterDist)) return false;
        for (const Vec3& ang : orientations) {
            if (tetraInsideContainer(c, ang, edge, planes)) {
                grid.addPoint(c);
                result.push_back({c, ang});
                return true;
            }
        }
        return false;
    };

    if (!huge) {
        for (uint64_t iz = 0; iz < nz; ++iz) {
            for (uint64_t iy = 0; iy < ny; ++iy) {
                for (uint64_t ix = 0; ix < nx; ++ix) {
                    const double ox = ((iy + iz) & 1ull) ? 0.5 * step : 0.0;
                    const Vec3 c{
                        min(box.mx.x, box.mn.x + ix * step + ox),
                        box.mn.y + iy * step,
                        box.mn.z + iz * step
                    };
                    if (c.x > box.mx.x || c.y > box.mx.y || c.z > box.mx.z) continue;
                    tryPlace(c);
                }
            }
        }
    } else {
        uniform_real_distribution<double> dx(box.mn.x, box.mx.x);
        uniform_real_distribution<double> dy(box.mn.y, box.mx.y);
        uniform_real_distribution<double> dz(box.mn.z, box.mx.z);
        for (uint64_t i = 0; i < randomBudget; ++i) {
            Vec3 c{dx(rng), dy(rng), dz(rng)};
            tryPlace(c);
        }
    }

    if (result.size() < 100000) {
        const uint64_t refineSamples = huge ? 60000ull : 30000ull;
        uniform_real_distribution<double> dx(box.mn.x, box.mx.x);
        uniform_real_distribution<double> dy(box.mn.y, box.mx.y);
        uniform_real_distribution<double> dz(box.mn.z, box.mx.z);
        for (uint64_t i = 0; i < refineSamples; ++i) {
            Vec3 c{dx(rng), dy(rng), dz(rng)};
            tryPlace(c);
        }
    }

    return result;
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge)) {
        return 0;
    }

    vector<Vec3> vertices;
    vertices.reserve(max(V, 0));
    for (int i = 0; i < V; ++i) {
        Vec3 p;
        cin >> p.x >> p.y >> p.z;
        vertices.push_back(p);
    }

    if (V < 4 || edge <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    const vector<Face> faces = buildConvexHullFaces(vertices);
    if (faces.size() < 4) {
        cout << 0 << '\n';
        return 0;
    }

    const vector<Plane> planes = extractPlanes(faces);
    const AABB box = computeAABB(vertices);

    mt19937_64 rng(0xC0FFEE123456789ull);
    const double boxVol = aabbVolume(box);

    uint64_t mcSamples = 30000;
    if (boxVol > 1e6) mcSamples = 12000;
    else if (boxVol < 100.0) mcSamples = 60000;

    const double estVolume = estimateVolumeMonteCarlo(planes, box, mcSamples, rng);
    vector<Placement> placements = generatePlacements(planes, box, edge, estVolume, rng);

    cout << placements.size() << '\n';
    cout << fixed << setprecision(9);
    for (const Placement& p : placements) {
        cout << p.c.x << ' ' << p.c.y << ' ' << p.c.z << ' '
             << p.euler.x << ' ' << p.euler.y << ' ' << p.euler.z << '\n';
    }

    return 0;
}
