use std::cmp::min;
use std::io::{self, BufReader, BufWriter, Read, Write};

#[derive(Clone, Copy, Debug, Default)]
struct Point3 {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3 {
    #[inline]
    fn zero() -> Self {
        Self { x: 0.0, y: 0.0, z: 0.0 }
    }

    #[inline]
    fn add_assign(&mut self, other: Point3) {
        self.x += other.x;
        self.y += other.y;
        self.z += other.z;
    }

    #[inline]
    fn scale(self, factor: f32) -> Self {
        Self {
            x: self.x * factor,
            y: self.y * factor,
            z: self.z * factor,
        }
    }

    #[inline]
    fn dist2(self, other: Point3) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

#[derive(Clone, Copy, Debug)]
enum Strategy {
    FullMemory,
    Streaming,
}

fn main() -> io::Result<()> {
    let mut stdin = BufReader::with_capacity(1 << 20, io::stdin().lock());
    let mut input = Vec::new();
    stdin.read_to_end(&mut input)?;

    let mut cursor = 0usize;
    let num_points: usize = read_u64_ascii(&input, &mut cursor)?
        .try_into()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "num_points too large"))?;
    let num_clusters: usize = read_u64_ascii(&input, &mut cursor)?
        .try_into()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "num_clusters too large"))?;

    if num_clusters == 0 {
        return Err(io::Error::new(
            io::ErrorKind::InvalidInput,
            "num_clusters must be positive",
        ));
    }
    if num_points == 0 {
        return Ok(());
    }

    skip_ascii_whitespace(&input, &mut cursor);
    let point_bytes = num_points
        .checked_mul(12)
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "input too large"))?;
    if input.len().saturating_sub(cursor) < point_bytes {
        return Err(io::Error::new(
            io::ErrorKind::UnexpectedEof,
            "not enough binary point data",
        ));
    }

    let points_blob = &input[cursor..cursor + point_bytes];
    let effective_clusters = min(num_clusters, num_points);
    let strategy = choose_strategy(num_points, effective_clusters);
    let assignments = match strategy {
        Strategy::FullMemory => solve_full_memory(points_blob, num_points, effective_clusters),
        Strategy::Streaming => solve_streaming(points_blob, num_points, effective_clusters),
    };

    let mut stdout = BufWriter::with_capacity(1 << 20, io::stdout().lock());
    for idx in assignments {
        stdout.write_all(&(idx as u32).to_le_bytes())?;
    }
    stdout.flush()?;
    Ok(())
}

fn choose_strategy(num_points: usize, num_clusters: usize) -> Strategy {
    let bytes = num_points.saturating_mul(12);
    if bytes <= 256 * 1024 * 1024 && num_clusters <= 4096 && num_points <= 20_000_000 {
        Strategy::FullMemory
    } else {
        Strategy::Streaming
    }
}

fn solve_full_memory(points_blob: &[u8], num_points: usize, num_clusters: usize) -> Vec<usize> {
    let points = decode_points(points_blob, num_points);
    let mut centroids = init_centroids_sample(&points, num_clusters);
    let mut assignments = vec![0usize; num_points];
    let max_iters = if num_points <= 1_000_000 { 20 } else { 12 };

    for _ in 0..max_iters {
        let mut sums = vec![Point3::zero(); num_clusters];
        let mut counts = vec![0usize; num_clusters];
        let mut changed = 0usize;

        for (i, &p) in points.iter().enumerate() {
            let cluster = nearest_centroid(p, &centroids);
            if assignments[i] != cluster {
                changed += 1;
                assignments[i] = cluster;
            }
            sums[cluster].add_assign(p);
            counts[cluster] += 1;
        }

        update_centroids(&mut centroids, &sums, &counts, &points);
        if changed * 1000 <= num_points {
            break;
        }
    }

    assignments
}

fn solve_streaming(points_blob: &[u8], num_points: usize, num_clusters: usize) -> Vec<usize> {
    let sample_cap = min(
        num_points,
        min(200_000usize.max(num_clusters.saturating_mul(32)), 2_000_000),
    );
    let stride = ((num_points / sample_cap.max(1)).max(1)) as usize;
    let mut sample = Vec::with_capacity(sample_cap);
    let mut idx = 0usize;
    while idx < num_points && sample.len() < sample_cap {
        sample.push(read_point(points_blob, idx));
        idx = idx.saturating_add(stride);
    }
    if sample.is_empty() {
        sample.push(read_point(points_blob, 0));
    }

    let mut centroids = init_centroids_sample(&sample, num_clusters);
    let warmup_iters = if sample.len() <= 100_000 { 10 } else { 6 };
    let mut warm_assignments = vec![0usize; sample.len()];
    for _ in 0..warmup_iters {
        let mut sums = vec![Point3::zero(); num_clusters];
        let mut counts = vec![0usize; num_clusters];
        let mut changed = 0usize;
        for (i, &p) in sample.iter().enumerate() {
            let cluster = nearest_centroid(p, &centroids);
            if warm_assignments[i] != cluster {
                changed += 1;
                warm_assignments[i] = cluster;
            }
            sums[cluster].add_assign(p);
            counts[cluster] += 1;
        }
        update_centroids(&mut centroids, &sums, &counts, &sample);
        if changed * 2000 <= sample.len().max(1) {
            break;
        }
    }

    let refine_passes = if num_points >= 100_000_000 { 1 } else { 2 };
    let batch_size = if num_points >= 100_000_000 { 262_144 } else { 131_072 };
    let mut assignments = vec![0usize; num_points];

    for _ in 0..refine_passes {
        let mut sums = vec![Point3::zero(); num_clusters];
        let mut counts = vec![0usize; num_clusters];
        let mut start = 0usize;
        while start < num_points {
            let end = min(start + batch_size, num_points);
            for i in start..end {
                let p = read_point(points_blob, i);
                let cluster = nearest_centroid(p, &centroids);
                assignments[i] = cluster;
                sums[cluster].add_assign(p);
                counts[cluster] += 1;
            }
            start = end;
        }
        update_centroids_streaming(&mut centroids, &sums, &counts, points_blob, num_points);
    }

    assignments
}

fn init_centroids_sample(points: &[Point3], num_clusters: usize) -> Vec<Point3> {
    let mut centroids = Vec::with_capacity(num_clusters);
    centroids.push(points[0]);

    let target = min(num_clusters, points.len());
    while centroids.len() < target {
        let mut best_point = points[0];
        let mut best_score = -1.0f32;
        let step = (points.len() / 8192).max(1);
        let mut i = 0usize;
        while i < points.len() {
            let p = points[i];
            let d = distance_to_set(p, &centroids);
            if d > best_score {
                best_score = d;
                best_point = p;
            }
            i += step;
        }
        centroids.push(best_point);
    }

    while centroids.len() < num_clusters {
        let source = centroids[centroids.len() % target.max(1)];
        let offset = (centroids.len() as f32) * 1.0e-4;
        centroids.push(Point3 {
            x: source.x + offset,
            y: source.y - offset,
            z: source.z + offset * 0.5,
        });
    }

    centroids
}

fn update_centroids(
    centroids: &mut [Point3],
    sums: &[Point3],
    counts: &[usize],
    fallback_points: &[Point3],
) {
    for i in 0..centroids.len() {
        if counts[i] > 0 {
            centroids[i] = sums[i].scale(1.0 / counts[i] as f32);
        } else {
            centroids[i] = fallback_points[i % fallback_points.len()];
        }
    }
}

fn update_centroids_streaming(
    centroids: &mut [Point3],
    sums: &[Point3],
    counts: &[usize],
    points_blob: &[u8],
    num_points: usize,
) {
    for i in 0..centroids.len() {
        if counts[i] > 0 {
            centroids[i] = sums[i].scale(1.0 / counts[i] as f32);
        } else {
            let idx = (i * 9973) % num_points;
            centroids[i] = read_point(points_blob, idx);
        }
    }
}

#[inline]
fn nearest_centroid(point: Point3, centroids: &[Point3]) -> usize {
    let mut best = 0usize;
    let mut best_dist = point.dist2(centroids[0]);
    for (i, &c) in centroids.iter().enumerate().skip(1) {
        let d = point.dist2(c);
        if d < best_dist {
            best_dist = d;
            best = i;
        }
    }
    best
}

#[inline]
fn distance_to_set(point: Point3, centroids: &[Point3]) -> f32 {
    let mut best = point.dist2(centroids[0]);
    for &c in &centroids[1..] {
        let d = point.dist2(c);
        if d < best {
            best = d;
        }
    }
    best
}

fn decode_points(blob: &[u8], num_points: usize) -> Vec<Point3> {
    let mut points = Vec::with_capacity(num_points);
    for i in 0..num_points {
        points.push(read_point(blob, i));
    }
    points
}

#[inline]
fn read_point(blob: &[u8], idx: usize) -> Point3 {
    let base = idx * 12;
    Point3 {
        x: f32::from_le_bytes(blob[base..base + 4].try_into().unwrap()),
        y: f32::from_le_bytes(blob[base + 4..base + 8].try_into().unwrap()),
        z: f32::from_le_bytes(blob[base + 8..base + 12].try_into().unwrap()),
    }
}

fn read_u64_ascii(input: &[u8], cursor: &mut usize) -> io::Result<u64> {
    skip_ascii_whitespace(input, cursor);
    let start = *cursor;
    while *cursor < input.len() && input[*cursor].is_ascii_digit() {
        *cursor += 1;
    }
    if *cursor == start {
        return Err(io::Error::new(
            io::ErrorKind::InvalidInput,
            "expected ASCII integer",
        ));
    }
    let text = std::str::from_utf8(&input[start..*cursor])
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "invalid ASCII header"))?;
    text.parse::<u64>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "invalid integer"))
}

fn skip_ascii_whitespace(input: &[u8], cursor: &mut usize) {
    while *cursor < input.len() && input[*cursor].is_ascii_whitespace() {
        *cursor += 1;
    }
}
