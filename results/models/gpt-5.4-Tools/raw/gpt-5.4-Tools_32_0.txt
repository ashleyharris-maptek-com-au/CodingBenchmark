#include <algorithm>
#include <array>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct InputEdge {
    int u;
    int v;
    long long w;
};

struct PQNode {
    long long dist;
    int v;
    bool operator>(const PQNode& other) const {
        return dist > other.dist;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<InputEdge> edges;
    edges.reserve(m);
    vector<vector<Edge>> graph(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        graph[u].push_back({v, w, i});
        graph[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    if (k == 0) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    if (k == 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    if (k > 20) {
        vector<long long> keyDist(n, numeric_limits<long long>::max() / 4);
        vector<int> keyParentVertex(n, -1);
        vector<int> keyParentEdge(n, -1);
        vector<char> inTree(n, 0);
        vector<int> treeVertices;
        vector<int> chosenEdges;

        int root = terminals[0];
        inTree[root] = 1;
        treeVertices.push_back(root);

        auto run_dijkstra_to_tree = [&](int source) {
            const long long INF = numeric_limits<long long>::max() / 4;
            vector<long long> dist(n, INF);
            vector<int> parentVertex(n, -1);
            vector<int> parentEdge(n, -1);
            priority_queue<PQNode, vector<PQNode>, greater<PQNode>> pq;
            dist[source] = 0;
            pq.push({0, source});
            int hit = -1;
            while (!pq.empty()) {
                auto cur = pq.top();
                pq.pop();
                if (cur.dist != dist[cur.v]) {
                    continue;
                }
                if (inTree[cur.v]) {
                    hit = cur.v;
                    break;
                }
                for (const auto& e : graph[cur.v]) {
                    long long nd = cur.dist + e.w;
                    if (nd < dist[e.to]) {
                        dist[e.to] = nd;
                        parentVertex[e.to] = cur.v;
                        parentEdge[e.to] = e.id;
                        pq.push({nd, e.to});
                    }
                }
            }
            return tuple<int, vector<int>, vector<int>>(hit, move(parentVertex), move(parentEdge));
        };

        for (int ti = 1; ti < k; ++ti) {
            auto [hit, parentVertex, parentEdge] = run_dijkstra_to_tree(terminals[ti]);
            if (hit == -1) {
                cout << 0 << ' ' << 0 << '\n';
                return 0;
            }
            int cur = hit;
            while (cur != terminals[ti]) {
                int prev = parentVertex[cur];
                int eid = parentEdge[cur];
                if (prev == -1 || eid == -1) {
                    break;
                }
                chosenEdges.push_back(eid);
                if (!inTree[prev]) {
                    inTree[prev] = 1;
                    treeVertices.push_back(prev);
                }
                if (!inTree[cur]) {
                    inTree[cur] = 1;
                    treeVertices.push_back(cur);
                }
                cur = prev;
            }
            if (!inTree[terminals[ti]]) {
                inTree[terminals[ti]] = 1;
                treeVertices.push_back(terminals[ti]);
            }
        }

        sort(chosenEdges.begin(), chosenEdges.end());
        chosenEdges.erase(unique(chosenEdges.begin(), chosenEdges.end()), chosenEdges.end());

        vector<vector<pair<int, int>>> treeAdj(n);
        for (int eid : chosenEdges) {
            const auto& e = edges[eid];
            treeAdj[e.u].push_back({e.v, eid});
            treeAdj[e.v].push_back({e.u, eid});
        }

        vector<char> needed(n, 0);
        for (int t : terminals) {
            needed[t] = 1;
        }

        vector<char> usedEdge(m, 0);
        function<bool(int, int)> prune = [&](int v, int p) -> bool {
            bool keep = needed[v];
            for (auto [to, eid] : treeAdj[v]) {
                if (to == p) {
                    continue;
                }
                if (prune(to, v)) {
                    usedEdge[eid] = 1;
                    keep = true;
                }
            }
            return keep;
        };

        prune(terminals[0], -1);

        long long totalWeight = 0;
        vector<pair<int, int>> answerEdges;
        for (int eid = 0; eid < m; ++eid) {
            if (usedEdge[eid]) {
                totalWeight += edges[eid].w;
                answerEdges.push_back({edges[eid].u, edges[eid].v});
            }
        }

        cout << totalWeight << ' ' << answerEdges.size() << '\n';
        for (const auto& e : answerEdges) {
            cout << e.first << ' ' << e.second << '\n';
        }
        return 0;
    }

    const int fullMask = 1 << k;
    const long long INF = numeric_limits<long long>::max() / 4;

    vector<vector<long long>> dp(fullMask, vector<long long>(n, INF));
    vector<vector<int>> parentVertex(fullMask, vector<int>(n, -1));
    vector<vector<int>> parentEdge(fullMask, vector<int>(n, -1));
    vector<vector<int>> splitMask(fullMask, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    for (int mask = 1; mask < fullMask; ++mask) {
        if ((mask & (mask - 1)) != 0) {
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (other == 0 || sub > other) {
                    continue;
                }
                for (int v = 0; v < n; ++v) {
                    long long lhs = dp[sub][v];
                    long long rhs = dp[other][v];
                    if (lhs == INF || rhs == INF) {
                        continue;
                    }
                    long long cand = lhs + rhs;
                    if (cand < dp[mask][v]) {
                        dp[mask][v] = cand;
                        splitMask[mask][v] = sub;
                        parentVertex[mask][v] = -1;
                        parentEdge[mask][v] = -1;
                    }
                }
            }
        }

        priority_queue<PQNode, vector<PQNode>, greater<PQNode>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            auto cur = pq.top();
            pq.pop();
            if (cur.dist != dp[mask][cur.v]) {
                continue;
            }
            for (const auto& e : graph[cur.v]) {
                long long nd = cur.dist + e.w;
                if (nd < dp[mask][e.to]) {
                    dp[mask][e.to] = nd;
                    parentVertex[mask][e.to] = cur.v;
                    parentEdge[mask][e.to] = e.id;
                    splitMask[mask][e.to] = -1;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int bestVertex = -1;
    long long bestCost = INF;
    int allMask = fullMask - 1;
    for (int v = 0; v < n; ++v) {
        if (dp[allMask][v] < bestCost) {
            bestCost = dp[allMask][v];
            bestVertex = v;
        }
    }

    if (bestVertex == -1 || bestCost >= INF) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    vector<char> usedEdge(m, 0);
    vector<array<int, 2>> stack;
    stack.push_back({allMask, bestVertex});

    while (!stack.empty()) {
        auto state = stack.back();
        stack.pop_back();
        int mask = state[0];
        int v = state[1];

        int sub = splitMask[mask][v];
        if (sub != -1) {
            int other = mask ^ sub;
            stack.push_back({sub, v});
            stack.push_back({other, v});
            continue;
        }

        int pv = parentVertex[mask][v];
        int pe = parentEdge[mask][v];
        if (pv != -1) {
            usedEdge[pe] = 1;
            stack.push_back({mask, pv});
        }
    }

    vector<vector<pair<int, int>>> treeAdj(n);
    for (int eid = 0; eid < m; ++eid) {
        if (!usedEdge[eid]) {
            continue;
        }
        const auto& e = edges[eid];
        treeAdj[e.u].push_back({e.v, eid});
        treeAdj[e.v].push_back({e.u, eid});
    }

    vector<char> needed(n, 0);
    for (int t : terminals) {
        needed[t] = 1;
    }

    vector<char> keptEdge(m, 0);
    function<bool(int, int)> dfsPrune = [&](int v, int p) -> bool {
        bool keep = needed[v];
        for (auto [to, eid] : treeAdj[v]) {
            if (to == p) {
                continue;
            }
            if (dfsPrune(to, v)) {
                keptEdge[eid] = 1;
                keep = true;
            }
        }
        return keep;
    };

    dfsPrune(bestVertex, -1);

    long long totalWeight = 0;
    vector<pair<int, int>> answerEdges;
    for (int eid = 0; eid < m; ++eid) {
        if (keptEdge[eid]) {
            totalWeight += edges[eid].w;
            answerEdges.push_back({edges[eid].u, edges[eid].v});
        }
    }

    cout << totalWeight << ' ' << answerEdges.size() << '\n';
    for (const auto& e : answerEdges) {
        cout << e.first << ' ' << e.second << '\n';
    }

    return 0;
}
