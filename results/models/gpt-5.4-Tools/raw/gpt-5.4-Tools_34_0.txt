import math
import random
import sys


def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))
    depot = (float(next(it)), float(next(it)))
    customers = []
    for idx in range(n):
        x = float(next(it))
        y = float(next(it))
        demand = int(next(it))
        customers.append((x, y, demand, idx))
    return n, k, capacity, depot, customers


def euclidean(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def build_distance_matrix(depot, customers):
    points = [depot] + [(x, y) for x, y, _, _ in customers]
    size = len(points)
    dist = [[0.0] * size for _ in range(size)]
    for i in range(size):
        for j in range(i + 1, size):
            d = euclidean(points[i], points[j])
            dist[i][j] = d
            dist[j][i] = d
    return dist


def route_distance(route, dist):
    if not route:
        return 0.0
    total = dist[0][route[0] + 1]
    for i in range(len(route) - 1):
        total += dist[route[i] + 1][route[i + 1] + 1]
    total += dist[route[-1] + 1][0]
    return total


def two_opt(route, dist):
    if len(route) < 4:
        return route
    improved = True
    while improved:
        improved = False
        for i in range(len(route) - 2):
            a = 0 if i == 0 else route[i - 1] + 1
            b = route[i] + 1
            for j in range(i + 1, len(route) - 1):
                c = route[j] + 1
                d = route[j + 1] + 1
                delta = (dist[a][c] + dist[b][d]) - (dist[a][b] + dist[c][d])
                if delta < -1e-9:
                    route[i:j + 1] = reversed(route[i:j + 1])
                    improved = True
                    break
            if improved:
                break
    return route


def nearest_feasible_route(start, unassigned, capacity, demands, dist):
    route = [start]
    remaining = capacity - demands[start]
    unassigned.remove(start)
    current = start
    while True:
        best = None
        best_d = float("inf")
        for customer in list(unassigned):
            if demands[customer] <= remaining:
                d = dist[current + 1][customer + 1]
                if d < best_d:
                    best_d = d
                    best = customer
        if best is None:
            break
        route.append(best)
        unassigned.remove(best)
        remaining -= demands[best]
        current = best
    return route


def build_initial_routes(n, k, capacity, customers, dist):
    demands = [d for _, _, d, _ in customers]
    total_demand = sum(demands)
    min_routes_needed = max(1, math.ceil(total_demand / capacity)) if n > 0 else 0
    if min_routes_needed > k:
        return None

    unassigned = set(range(n))
    routes = []

    seeds_to_use = min(k, n)
    order = sorted(
        range(n),
        key=lambda i: (dist[0][i + 1], demands[i]),
        reverse=True,
    )
    seed_count = max(min_routes_needed, min(seeds_to_use, len(order)))
    for start in order[:seed_count]:
        if start in unassigned:
            routes.append(nearest_feasible_route(start, unassigned, capacity, demands, dist))

    while unassigned:
        progress = False
        for route in routes:
            load = sum(demands[i] for i in route)
            remaining = capacity - load
            if remaining <= 0:
                continue
            last = route[-1]
            best = None
            best_d = float("inf")
            for customer in unassigned:
                if demands[customer] <= remaining:
                    d = dist[last + 1][customer + 1]
                    if d < best_d:
                        best_d = d
                        best = customer
            if best is not None:
                route.append(best)
                unassigned.remove(best)
                progress = True
        if not progress:
            if len(routes) >= k:
                customer = max(unassigned, key=lambda i: demands[i])
                inserted = False
                for route in routes:
                    load = sum(demands[i] for i in route)
                    if load + demands[customer] <= capacity:
                        route.append(customer)
                        unassigned.remove(customer)
                        inserted = True
                        break
                if not inserted:
                    return None
            else:
                start = max(unassigned, key=lambda i: dist[0][i + 1])
                routes.append(nearest_feasible_route(start, unassigned, capacity, demands, dist))

    return routes


def try_relocate(routes, demands, capacity, dist):
    for i, route_a in enumerate(routes):
        load_a = sum(demands[c] for c in route_a)
        base_a = route_distance(route_a, dist)
        for pos, customer in enumerate(route_a):
            new_a = route_a[:pos] + route_a[pos + 1:]
            cost_without = route_distance(new_a, dist)
            demand_c = demands[customer]
            for j, route_b in enumerate(routes):
                if i == j:
                    continue
                load_b = sum(demands[c] for c in route_b)
                if load_b + demand_c > capacity:
                    continue
                base_b = route_distance(route_b, dist)
                best_insert_cost = None
                best_insert_route = None
                for ins in range(len(route_b) + 1):
                    candidate_b = route_b[:ins] + [customer] + route_b[ins:]
                    cand_cost = route_distance(candidate_b, dist)
                    if best_insert_cost is None or cand_cost < best_insert_cost:
                        best_insert_cost = cand_cost
                        best_insert_route = candidate_b
                delta = (cost_without + best_insert_cost) - (base_a + base_b)
                if delta < -1e-9:
                    routes[i] = new_a
                    routes[j] = best_insert_route
                    if not routes[i]:
                        routes.pop(i)
                    return True
    return False


def improve_routes(routes, demands, capacity, dist):
    for idx in range(len(routes)):
        routes[idx] = two_opt(routes[idx], dist)
    for _ in range(50):
        changed = try_relocate(routes, demands, capacity, dist)
        if not changed:
            break
        for idx in range(len(routes)):
            routes[idx] = two_opt(routes[idx], dist)
    return routes


def validate(routes, n, k, capacity, demands):
    if routes is None or len(routes) > k:
        return False
    seen = []
    for route in routes:
        load = sum(demands[i] for i in route)
        if load > capacity:
            return False
        seen.extend(route)
    return len(seen) == n and sorted(seen) == list(range(n))


def fallback_sequential(n, k, capacity, demands):
    routes = []
    current = []
    load = 0
    for customer in sorted(range(n), key=lambda i: demands[i], reverse=True):
        d = demands[customer]
        if d > capacity:
            return None
        if load + d > capacity:
            routes.append(current)
            current = [customer]
            load = d
        else:
            current.append(customer)
            load += d
    if current:
        routes.append(current)
    if len(routes) > k:
        return None
    return routes


def solve():
    parsed = read_input()
    if parsed is None:
        return
    n, k, capacity, depot, customers = parsed
    demands = [d for _, _, d, _ in customers]

    if any(d > capacity for d in demands):
        print("0")
        print("0")
        return

    dist = build_distance_matrix(depot, customers)
    routes = build_initial_routes(n, k, capacity, customers, dist)
    if routes is None:
        routes = fallback_sequential(n, k, capacity, demands)
    if routes is None or not validate(routes, n, k, capacity, demands):
        print("0")
        print("0")
        return

    random.seed(0)
    routes = improve_routes(routes, demands, capacity, dist)
    routes = [route for route in routes if route]
    if not validate(routes, n, k, capacity, demands):
        print("0")
        print("0")
        return

    total = sum(route_distance(route, dist) for route in routes)
    print(f"{total:.10f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))


if __name__ == "__main__":
    solve()
