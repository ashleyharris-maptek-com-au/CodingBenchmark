use std::collections::VecDeque;
use std::io::{self, Read};

const DIRS: [(isize, isize); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];
const OPPOSITE: [u8; 4] = [1, 0, 3, 2];
const NONE: u8 = 255;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }

    let mut lines = input.lines();
    let header = match lines.next() {
        Some(line) => line,
        None => return,
    };

    let mut hw = header.split_whitespace();
    let h: usize = match hw.next().and_then(|v| v.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let w: usize = match hw.next().and_then(|v| v.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = vec![b'#'; total];
    let mut start = None;
    let mut goal = None;

    for r in 0..h {
        let line = lines.next().unwrap_or("");
        let bytes = line.as_bytes();
        for c in 0..w {
            let ch = *bytes.get(c).unwrap_or(&b' ');
            let idx = r * w + c;
            grid[idx] = ch;
            if ch == b'A' {
                start = Some(idx);
            } else if ch == b'B' {
                goal = Some(idx);
            }
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if start == goal {
        let (x, y) = (start / w, start % w);
        println!("1");
        println!("{} {}", x, y);
        return;
    }

    let mut state = vec![0u8; total];
    let mut parent_from_start = vec![NONE; total];
    let mut parent_from_goal = vec![NONE; total];
    let mut q_start = VecDeque::new();
    let mut q_goal = VecDeque::new();

    state[start] |= 1;
    state[goal] |= 2;
    q_start.push_back(start);
    q_goal.push_back(goal);

    let mut meet = None;

    while !q_start.is_empty() && !q_goal.is_empty() && meet.is_none() {
        if q_start.len() <= q_goal.len() {
            meet = expand_frontier(
                &grid,
                w,
                h,
                &mut q_start,
                &mut state,
                &mut parent_from_start,
                1,
                2,
            );
        } else {
            meet = expand_frontier(
                &grid,
                w,
                h,
                &mut q_goal,
                &mut state,
                &mut parent_from_goal,
                2,
                1,
            );
        }
    }

    let meet = match meet {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let path = reconstruct_path(meet, start, goal, w, &parent_from_start, &parent_from_goal);
    println!("{}", path.len());
    for idx in path {
        println!("{} {}", idx / w, idx % w);
    }
}

fn expand_frontier(
    grid: &[u8],
    w: usize,
    h: usize,
    queue: &mut VecDeque<usize>,
    state: &mut [u8],
    parent: &mut [u8],
    own_bit: u8,
    other_bit: u8,
) -> Option<usize> {
    let level_size = queue.len();
    for _ in 0..level_size {
        let cur = queue.pop_front().unwrap();
        let r = cur / w;
        let c = cur % w;

        for (dir, (dr, dc)) in DIRS.iter().enumerate() {
            let nr = r as isize + dr;
            let nc = c as isize + dc;
            if nr < 0 || nc < 0 || nr >= h as isize || nc >= w as isize {
                continue;
            }

            let next = nr as usize * w + nc as usize;
            if grid[next] == b'#' {
                continue;
            }

            let seen = state[next];
            if seen & own_bit != 0 {
                continue;
            }

            parent[next] = OPPOSITE[dir];
            if seen & other_bit != 0 {
                return Some(next);
            }

            state[next] = seen | own_bit;
            queue.push_back(next);
        }
    }
    None
}

fn reconstruct_path(
    meet: usize,
    start: usize,
    goal: usize,
    w: usize,
    parent_from_start: &[u8],
    parent_from_goal: &[u8],
) -> Vec<usize> {
    let mut left = Vec::new();
    let mut cur = meet;
    loop {
        left.push(cur);
        if cur == start {
            break;
        }
        cur = step(cur, w, parent_from_start[cur]);
    }
    left.reverse();

    let mut right = Vec::new();
    cur = meet;
    while cur != goal {
        cur = step(cur, w, parent_from_goal[cur]);
        right.push(cur);
    }

    left.extend(right);
    left
}

fn step(idx: usize, w: usize, dir: u8) -> usize {
    match dir {
        0 => idx - w,
        1 => idx + w,
        2 => idx - 1,
        3 => idx + 1,
        _ => unreachable!(),
    }
}
