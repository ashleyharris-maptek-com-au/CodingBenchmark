use std::collections::VecDeque;
use std::io::{self, Read};

const DIRS: [(is32, is32); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];

type is32 = i32;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }

    let mut lines = input.lines();
    let header = match lines.next() {
        Some(line) => line,
        None => return,
    };

    let mut hw = header.split_whitespace();
    let h: usize = match hw.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let w: usize = match hw.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = Vec::with_capacity(total);
    let mut start = None;
    let mut goal = None;

    for r in 0..h {
        let line = match lines.next() {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        if line.len() != w {
            println!("0");
            return;
        }

        for (c, &b) in line.as_bytes().iter().enumerate() {
            let idx = r * w + c;
            match b {
                b'A' => {
                    start = Some(idx);
                    grid.push(b' ');
                }
                b'B' => {
                    goal = Some(idx);
                    grid.push(b' ');
                }
                b'#' | b' ' => grid.push(b),
                _ => {
                    println!("0");
                    return;
                }
            }
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if start == goal {
        let (x, y) = to_xy(start, w);
        println!("1");
        println!("{x} {y}");
        return;
    }

    let mut owner = vec![0u8; total];
    let mut prev_start = vec![u32::MAX; total];
    let mut prev_goal = vec![u32::MAX; total];
    let mut q_start = VecDeque::new();
    let mut q_goal = VecDeque::new();

    owner[start] = 1;
    owner[goal] = 2;
    prev_start[start] = start as u32;
    prev_goal[goal] = goal as u32;
    q_start.push_back(start);
    q_goal.push_back(goal);

    let mut meet = None;

    while !q_start.is_empty() && !q_goal.is_empty() && meet.is_none() {
        if q_start.len() <= q_goal.len() {
            meet = expand_frontier(
                &grid,
                w,
                &mut q_start,
                1,
                2,
                &mut owner,
                &mut prev_start,
            );
        } else {
            meet = expand_frontier(
                &grid,
                w,
                &mut q_goal,
                2,
                1,
                &mut owner,
                &mut prev_goal,
            );
        }
    }

    let (touch_from_this, touch_from_other) = match meet {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut left = build_chain(touch_from_this, &prev_start);
    let mut right = build_chain(touch_from_other, &prev_goal);
    right.reverse();

    if !right.is_empty() {
        right.remove(0);
    }
    left.extend(right);

    println!("{}", left.len());
    for idx in left {
        let (x, y) = to_xy(idx, w);
        println!("{x} {y}");
    }
}

fn expand_frontier(
    grid: &[u8],
    w: usize,
    queue: &mut VecDeque<usize>,
    mine: u8,
    other: u8,
    owner: &mut [u8],
    prev: &mut [u32],
) -> Option<(usize, usize)> {
    let level_count = queue.len();
    for _ in 0..level_count {
        let cur = queue.pop_front().unwrap();
        let r = cur / w;
        let c = cur % w;

        for (dr, dc) in DIRS {
            let nr = r as is32 + dr;
            let nc = c as is32 + dc;
            if nr < 0 || nc < 0 {
                continue;
            }
            let nr = nr as usize;
            let nc = nc as usize;
            if nc >= w {
                continue;
            }
            let next = match nr.checked_mul(w).and_then(|base| base.checked_add(nc)) {
                Some(v) if v < grid.len() => v,
                _ => continue,
            };
            if grid[next] == b'#' {
                continue;
            }

            match owner[next] {
                0 => {
                    owner[next] = mine;
                    prev[next] = cur as u32;
                    queue.push_back(next);
                }
                side if side == other => {
                    prev[next] = cur as u32;
                    return Some((next, cur));
                }
                _ => {}
            }
        }
    }
    None
}

fn build_chain(mut end: usize, prev: &[u32]) -> Vec<usize> {
    let mut path = Vec::new();
    loop {
        path.push(end);
        let p = prev[end] as usize;
        if p == end {
            break;
        }
        end = p;
    }
    path.reverse();
    path
}

fn to_xy(idx: usize, w: usize) -> (usize, usize) {
    (idx / w, idx % w)
}
