use std::collections::{HashMap, HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
struct Pt {
    x: i32,
    y: i32,
}

impl Pt {
    fn manhattan(self, other: Pt) -> i32 {
        (self.x - other.x).abs() + (self.y - other.y).abs()
    }
}

#[derive(Clone)]
struct Grid {
    w: i32,
    h: i32,
    blocked: Vec<bool>,
    free_cells: Vec<Pt>,
    id_of: HashMap<(i32, i32), usize>,
}

impl Grid {
    fn new(w: i32, h: i32, obstacles: &[(i32, i32)]) -> Self {
        let mut blocked = vec![false; (w * h) as usize];
        for &(x, y) in obstacles {
            if x >= 0 && x < w && y >= 0 && y < h {
                blocked[(y * w + x) as usize] = true;
            }
        }
        let mut free_cells = Vec::new();
        let mut id_of = HashMap::new();
        for y in 0..h {
            for x in 0..w {
                if !blocked[(y * w + x) as usize] {
                    let idx = free_cells.len();
                    free_cells.push(Pt { x, y });
                    id_of.insert((x, y), idx);
                }
            }
        }
        Self { w, h, blocked, free_cells, id_of }
    }

    fn inside(&self, x: i32, y: i32) -> bool {
        x >= 0 && x < self.w && y >= 0 && y < self.h
    }

    fn is_free(&self, x: i32, y: i32) -> bool {
        self.inside(x, y) && !self.blocked[(y * self.w + x) as usize]
    }

    fn start_exists(&self) -> bool {
        self.is_free(0, 0)
    }

    fn idx(&self, p: Pt) -> usize {
        *self.id_of.get(&(p.x, p.y)).unwrap()
    }

    fn neighbors(&self, p: Pt) -> Vec<Pt> {
        const DIRS: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
        let mut out = Vec::with_capacity(4);
        for (dx, dy) in DIRS {
            let nx = p.x + dx;
            let ny = p.y + dy;
            if self.is_free(nx, ny) {
                out.push(Pt { x: nx, y: ny });
            }
        }
        out
    }

    fn degree_in_mask(&self, p: Pt, alive: &[bool]) -> usize {
        self.neighbors(p)
            .into_iter()
            .filter(|q| alive[self.idx(*q)])
            .count()
    }
}

fn read_input() -> Option<(Grid, bool)> {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok()?;
    if input.trim().is_empty() {
        return None;
    }
    let mut it = input.split_whitespace();
    let w: i32 = it.next()?.parse().ok()?;
    let h: i32 = it.next()?.parse().ok()?;
    let k: usize = it.next()?.parse().ok()?;
    let require_cycle: i32 = it.next()?.parse().ok()?;
    let mut obstacles = Vec::with_capacity(k);
    for _ in 0..k {
        let x: i32 = it.next()?.parse().ok()?;
        let y: i32 = it.next()?.parse().ok()?;
        obstacles.push((x, y));
    }
    Some((Grid::new(w, h, &obstacles), require_cycle != 0))
}

fn connected_component(grid: &Grid) -> bool {
    if grid.free_cells.is_empty() {
        return false;
    }
    let start = Pt { x: 0, y: 0 };
    if !grid.is_free(start.x, start.y) {
        return false;
    }
    let mut seen = vec![false; grid.free_cells.len()];
    let mut q = VecDeque::new();
    seen[grid.idx(start)] = true;
    q.push_back(start);
    let mut cnt = 0usize;
    while let Some(p) = q.pop_front() {
        cnt += 1;
        for n in grid.neighbors(p) {
            let idx = grid.idx(n);
            if !seen[idx] {
                seen[idx] = true;
                q.push_back(n);
            }
        }
    }
    cnt == grid.free_cells.len()
}

fn bipartite_counts(grid: &Grid) -> (usize, usize) {
    let mut a = 0usize;
    let mut b = 0usize;
    for &p in &grid.free_cells {
        if (p.x + p.y) & 1 == 0 { a += 1; } else { b += 1; }
    }
    (a, b)
}

fn quick_impossible(grid: &Grid, require_cycle: bool) -> bool {
    if !grid.start_exists() {
        return true;
    }
    let n = grid.free_cells.len();
    if n == 0 {
        return true;
    }
    if n == 1 {
        return require_cycle;
    }
    if !connected_component(grid) {
        return true;
    }
    let (a, b) = bipartite_counts(grid);
    if require_cycle {
        if n == 1 || n % 2 == 1 || a != b {
            return true;
        }
    } else {
        let start_black = ((0 + 0) & 1) == 0;
        let diff = a.abs_diff(b);
        if diff > 1 {
            return true;
        }
        if n % 2 == 1 {
            if start_black && a != b + 1 {
                return true;
            }
            if !start_black && b != a + 1 {
                return true;
            }
        }
    }

    let alive = vec![true; n];
    let mut leaves = 0usize;
    for &p in &grid.free_cells {
        let d = grid.degree_in_mask(p, &alive);
        if d == 0 && n > 1 {
            return true;
        }
        if d == 1 {
            leaves += 1;
        }
    }
    if require_cycle {
        if n > 2 && leaves > 0 {
            return true;
        }
    } else if leaves > 2 {
        return true;
    }
    false
}

fn snake_path_full_rect(w: i32, h: i32) -> Option<Vec<Pt>> {
    if w <= 0 || h <= 0 {
        return None;
    }
    if w == 1 && h == 1 {
        return Some(vec![Pt { x: 0, y: 0 }]);
    }
    if h == 1 {
        let mut path = Vec::new();
        for x in 0..w {
            path.push(Pt { x, y: 0 });
        }
        return Some(path);
    }
    if w == 1 {
        let mut path = Vec::new();
        for y in 0..h {
            path.push(Pt { x: 0, y });
        }
        return Some(path);
    }
    if h % 2 == 0 {
        let mut path = Vec::with_capacity((w * h) as usize);
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w { path.push(Pt { x, y }); }
            } else {
                for x in (0..w).rev() { path.push(Pt { x, y }); }
            }
        }
        if path.first().copied() != Some(Pt { x: 0, y: 0 }) {
            return None;
        }
        return Some(path);
    }
    if w % 2 == 0 {
        let mut path = Vec::with_capacity((w * h) as usize);
        for x in 0..w {
            if x % 2 == 0 {
                for y in 0..h { path.push(Pt { x, y }); }
            } else {
                for y in (0..h).rev() { path.push(Pt { x, y }); }
            }
        }
        if path.first().copied() == Some(Pt { x: 0, y: 0 }) {
            return Some(path);
        }
        let mut alt = Vec::with_capacity((w * h) as usize);
        for y in 0..h {
            alt.push(Pt { x: 0, y });
        }
        for x in 1..w {
            if x % 2 == 1 {
                for y in (1..h).rev() { alt.push(Pt { x, y }); }
                alt.push(Pt { x, y: 0 });
            } else {
                for y in 1..h { alt.push(Pt { x, y }); }
            }
        }
        if alt.len() == (w * h) as usize {
            return Some(alt);
        }
        return None;
    }
    None
}

fn cycle_path_full_rect(w: i32, h: i32) -> Option<Vec<Pt>> {
    if w <= 1 || h <= 1 || (w * h) % 2 == 1 {
        return None;
    }
    if h % 2 == 0 {
        let mut path = Vec::with_capacity((w * h) as usize);
        path.push(Pt { x: 0, y: 0 });
        for y in 1..h {
            path.push(Pt { x: 0, y });
        }
        for x in 1..w {
            if ((w - 1 - x) % 2) == 0 {
                path.push(Pt { x, y: h - 1 });
                for y in (1..h - 1).rev() {
                    path.push(Pt { x, y });
                }
            } else {
                path.push(Pt { x, y: 1 });
                for y in 2..h {
                    path.push(Pt { x, y });
                }
            }
        }
        return Some(path);
    }
    if w % 2 == 0 {
        let mut swapped = cycle_path_full_rect(h, w)?;
        for p in &mut swapped {
            std::mem::swap(&mut p.x, &mut p.y);
        }
        return Some(swapped);
    }
    None
}

fn is_obstacle_free_rect(grid: &Grid) -> bool {
    grid.free_cells.len() == (grid.w * grid.h) as usize
}

fn check_path(grid: &Grid, path: &[Pt], require_cycle: bool) -> bool {
    if path.is_empty() || path[0] != (Pt { x: 0, y: 0 }) {
        return false;
    }
    if path.len() != grid.free_cells.len() {
        return false;
    }
    let mut seen = HashSet::new();
    for i in 0..path.len() {
        let p = path[i];
        if !grid.is_free(p.x, p.y) || !seen.insert((p.x, p.y)) {
            return false;
        }
        if i > 0 && path[i - 1].manhattan(path[i]) != 1 {
            return false;
        }
    }
    if require_cycle && path[path.len() - 1].manhattan(Pt { x: 0, y: 0 }) != 1 {
        return false;
    }
    true
}

fn reachable_count(grid: &Grid, alive: &[bool], start: Pt) -> usize {
    if !alive[grid.idx(start)] {
        return 0;
    }
    let mut seen = vec![false; alive.len()];
    let mut q = VecDeque::new();
    let sidx = grid.idx(start);
    seen[sidx] = true;
    q.push_back(start);
    let mut cnt = 0usize;
    while let Some(p) = q.pop_front() {
        cnt += 1;
        for n in grid.neighbors(p) {
            let idx = grid.idx(n);
            if alive[idx] && !seen[idx] {
                seen[idx] = true;
                q.push_back(n);
            }
        }
    }
    cnt
}

fn search_small(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    let n = grid.free_cells.len();
    if n == 0 {
        return None;
    }
    let limit = if require_cycle { 44usize } else { 52usize };
    if n > limit {
        return None;
    }

    let start = Pt { x: 0, y: 0 };
    let mut alive = vec![true; n];
    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(n);

    fn dfs(
        grid: &Grid,
        require_cycle: bool,
        cur: Pt,
        used_count: usize,
        alive: &mut [bool],
        used: &mut [bool],
        path: &mut Vec<Pt>,
    ) -> bool {
        path.push(cur);
        let cur_idx = grid.idx(cur);
        used[cur_idx] = true;
        alive[cur_idx] = false;

        if used_count == grid.free_cells.len() {
            let ok = !require_cycle || cur.manhattan(Pt { x: 0, y: 0 }) == 1;
            if ok {
                return true;
            }
            used[cur_idx] = false;
            alive[cur_idx] = true;
            path.pop();
            return false;
        }

        let mut nexts: Vec<Pt> = grid.neighbors(cur)
            .into_iter()
            .filter(|p| alive[grid.idx(*p)])
            .collect();

        nexts.sort_by(|&a, &b| {
            let da = grid.degree_in_mask(a, alive);
            let db = grid.degree_in_mask(b, alive);
            da.cmp(&db)
                .then_with(|| a.manhattan(Pt { x: 0, y: 0 }).cmp(&b.manhattan(Pt { x: 0, y: 0 })))
        });

        for nxt in nexts {
            let mut forced_fail = false;
            let nxt_idx = grid.idx(nxt);
            if used[nxt_idx] || !alive[nxt_idx] {
                continue;
            }

            alive[nxt_idx] = false;
            let remaining_any = grid.free_cells.iter().copied().find(|p| alive[grid.idx(*p)]);
            if let Some(s) = remaining_any {
                if reachable_count(grid, alive, s) != alive.iter().filter(|v| **v).count() {
                    forced_fail = true;
                }
            }
            alive[nxt_idx] = true;
            if forced_fail {
                continue;
            }

            let mut leaves = 0usize;
            let mut isolated = false;
            let mut alive2 = alive.to_vec();
            alive2[nxt_idx] = false;
            for &p in &grid.free_cells {
                let idx = grid.idx(p);
                if alive2[idx] {
                    let d = grid.degree_in_mask(p, &alive2);
                    if d == 0 && alive2.iter().filter(|v| **v).count() > 1 {
                        isolated = true;
                        break;
                    }
                    if d == 1 {
                        leaves += 1;
                    }
                }
            }
            if isolated {
                continue;
            }
            if require_cycle {
                if leaves > 0 && used_count + 1 < grid.free_cells.len() {
                    continue;
                }
            } else if leaves > 2 {
                continue;
            }

            if dfs(grid, require_cycle, nxt, used_count + 1, alive, used, path) {
                return true;
            }
        }

        used[cur_idx] = false;
        alive[cur_idx] = true;
        path.pop();
        false
    }

    if dfs(grid, require_cycle, start, 1, &mut alive, &mut used, &mut path) {
        Some(path)
    } else {
        None
    }
}

fn fallback_large_obstacles(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    if grid.free_cells.len() > 2500 {
        return None;
    }
    let n = grid.free_cells.len();
    let mut order = grid.free_cells.clone();
    order.sort_by(|a, b| {
        if a.y == b.y {
            if a.y % 2 == 0 { a.x.cmp(&b.x) } else { b.x.cmp(&a.x) }
        } else {
            a.y.cmp(&b.y)
        }
    });
    if order.first().copied() != Some(Pt { x: 0, y: 0 }) {
        if let Some(pos) = order.iter().position(|&p| p == Pt { x: 0, y: 0 }) {
            order.rotate_left(pos);
        }
    }
    if check_path(grid, &order, require_cycle) {
        return Some(order);
    }

    if n <= 80 {
        return search_small(grid, require_cycle);
    }
    None
}

fn solve(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    if quick_impossible(grid, require_cycle) {
        return None;
    }

    if is_obstacle_free_rect(grid) {
        let path = if require_cycle {
            cycle_path_full_rect(grid.w, grid.h)
        } else {
            snake_path_full_rect(grid.w, grid.h)
        };
        if let Some(path) = path {
            if check_path(grid, &path, require_cycle) {
                return Some(path);
            }
        }
        if grid.free_cells.len() == 1 && !require_cycle {
            return Some(vec![Pt { x: 0, y: 0 }]);
        }
    }

    if let Some(path) = search_small(grid, require_cycle) {
        if check_path(grid, &path, require_cycle) {
            return Some(path);
        }
    }

    if let Some(path) = fallback_large_obstacles(grid, require_cycle) {
        if check_path(grid, &path, require_cycle) {
            return Some(path);
        }
    }

    None
}

fn main() {
    let Some((grid, require_cycle)) = read_input() else {
        println!("0");
        return;
    };

    match solve(&grid, require_cycle) {
        Some(path) => {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
        }
        None => println!("0"),
    }
}
