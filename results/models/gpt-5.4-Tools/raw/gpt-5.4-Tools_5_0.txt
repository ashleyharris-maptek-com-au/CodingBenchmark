use std::cmp::Reverse;
use std::collections::{BinaryHeap, HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash, Ord, PartialOrd)]
struct Pos {
    x: i32,
    y: i32,
}

impl Pos {
    fn manhattan(self, other: Pos) -> i32 {
        (self.x - other.x).abs() + (self.y - other.y).abs()
    }
}

#[derive(Clone)]
struct Grid {
    w: usize,
    h: usize,
    blocked: Vec<bool>,
}

impl Grid {
    fn new(w: usize, h: usize, blocked: Vec<bool>) -> Self {
        Self { w, h, blocked }
    }

    fn inside(&self, x: i32, y: i32) -> bool {
        x >= 0 && y >= 0 && (x as usize) < self.w && (y as usize) < self.h
    }

    fn idx(&self, x: i32, y: i32) -> usize {
        y as usize * self.w + x as usize
    }

    fn is_free(&self, x: i32, y: i32) -> bool {
        self.inside(x, y) && !self.blocked[self.idx(x, y)]
    }

    fn neighbors(&self, p: Pos) -> impl Iterator<Item = Pos> + '_ {
        const DIRS: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
        IntoIterator::into_iter(DIRS).filter_map(move |(dx, dy)| {
            let np = Pos {
                x: p.x + dx,
                y: p.y + dy,
            };
            self.is_free(np.x, np.y).then_some(np)
        })
    }

    fn free_cells(&self) -> Vec<Pos> {
        let mut out = Vec::new();
        for y in 0..self.h as i32 {
            for x in 0..self.w as i32 {
                if self.is_free(x, y) {
                    out.push(Pos { x, y });
                }
            }
        }
        out
    }

    fn free_count(&self) -> usize {
        self.blocked.iter().filter(|&&b| !b).count()
    }
}

fn is_adj(a: Pos, b: Pos) -> bool {
    a.manhattan(b) == 1
}

fn bipartite_counts(grid: &Grid) -> (usize, usize) {
    let mut a = 0usize;
    let mut b = 0usize;
    for p in grid.free_cells() {
        if ((p.x + p.y) & 1) == 0 {
            a += 1;
        } else {
            b += 1;
        }
    }
    (a, b)
}

fn reachable_component(grid: &Grid, start: Pos) -> HashSet<Pos> {
    let mut seen = HashSet::new();
    if !grid.is_free(start.x, start.y) {
        return seen;
    }
    let mut q = VecDeque::new();
    seen.insert(start);
    q.push_back(start);
    while let Some(p) = q.pop_front() {
        for n in grid.neighbors(p) {
            if seen.insert(n) {
                q.push_back(n);
            }
        }
    }
    seen
}

fn connected_all(grid: &Grid, start: Pos) -> bool {
    reachable_component(grid, start).len() == grid.free_count()
}

fn degree_in_remaining(grid: &Grid, p: Pos, used: &[bool]) -> usize {
    grid.neighbors(p)
        .filter(|n| !used[grid.idx(n.x, n.y)])
        .count()
}

fn remaining_connected(grid: &Grid, current: Pos, used: &[bool]) -> bool {
    let mut seed = None;
    for y in 0..grid.h as i32 {
        for x in 0..grid.w as i32 {
            if grid.is_free(x, y) && !used[grid.idx(x, y)] {
                seed = Some(Pos { x, y });
                break;
            }
        }
        if seed.is_some() {
            break;
        }
    }
    let Some(seed) = seed else {
        return true;
    };
    let mut q = VecDeque::new();
    let mut seen = HashSet::new();
    q.push_back(seed);
    seen.insert(seed);
    while let Some(p) = q.pop_front() {
        for n in grid.neighbors(p) {
            if !used[grid.idx(n.x, n.y)] && seen.insert(n) {
                q.push_back(n);
            }
        }
    }
    for y in 0..grid.h as i32 {
        for x in 0..grid.w as i32 {
            let p = Pos { x, y };
            if grid.is_free(x, y) && !used[grid.idx(x, y)] && !seen.contains(&p) {
                return false;
            }
        }
    }
    let _ = current;
    true
}

fn exact_dfs(
    grid: &Grid,
    require_cycle: bool,
    path: &mut Vec<Pos>,
    used: &mut [bool],
    left: usize,
) -> bool {
    let cur = *path.last().unwrap();
    if left == 0 {
        return !require_cycle || is_adj(cur, path[0]);
    }

    let mut forced = Vec::new();
    for n in grid.neighbors(cur) {
        if !used[grid.idx(n.x, n.y)] {
            let deg = degree_in_remaining(grid, n, used);
            forced.push((deg, n.manhattan(path[0]), n));
        }
    }
    forced.sort_unstable();

    for (_, _, nxt) in forced {
        let idx = grid.idx(nxt.x, nxt.y);
        used[idx] = true;
        path.push(nxt);

        let mut ok = true;
        if left > 1 {
            let mut endpoints = 0usize;
            for y in 0..grid.h as i32 {
                for x in 0..grid.w as i32 {
                    if grid.is_free(x, y) && !used[grid.idx(x, y)] {
                        let d = degree_in_remaining(grid, Pos { x, y }, used);
                        if d == 0 {
                            ok = false;
                            break;
                        }
                        if d == 1 {
                            endpoints += 1;
                        }
                    }
                }
                if !ok {
                    break;
                }
            }
            if ok && endpoints > 2 {
                ok = false;
            }
            if ok && !remaining_connected(grid, nxt, used) {
                ok = false;
            }
        }

        if ok && exact_dfs(grid, require_cycle, path, used, left - 1) {
            return true;
        }
        path.pop();
        used[idx] = false;
    }
    false
}

fn solve_exact(grid: &Grid, require_cycle: bool) -> Option<Vec<Pos>> {
    let total = grid.free_count();
    let start = Pos { x: 0, y: 0 };
    if !grid.is_free(0, 0) {
        return None;
    }
    if total == 0 {
        return None;
    }
    if !connected_all(grid, start) {
        return None;
    }
    let (a, b) = bipartite_counts(grid);
    if require_cycle {
        if total % 2 == 1 || a != b {
            return None;
        }
    } else if (a as i32 - b as i32).abs() > 1 {
        return None;
    }
    let mut used = vec![false; grid.w * grid.h];
    used[grid.idx(0, 0)] = true;
    let mut path = vec![start];
    if exact_dfs(grid, require_cycle, &mut path, &mut used, total - 1) {
        Some(path)
    } else {
        None
    }
}

fn empty_serpentine_path(w: usize, h: usize) -> Vec<Pos> {
    let mut out = Vec::with_capacity(w * h);
    if h == 0 || w == 0 {
        return out;
    }
    if h % 2 == 0 {
        for y in 0..h as i32 {
            if y % 2 == 0 {
                for x in 0..w as i32 {
                    out.push(Pos { x, y });
                }
            } else {
                for x in (0..w as i32).rev() {
                    out.push(Pos { x, y });
                }
            }
        }
    } else {
        for x in 0..w as i32 {
            if x % 2 == 0 {
                for y in 0..h as i32 {
                    out.push(Pos { x, y });
                }
            } else {
                for y in (0..h as i32).rev() {
                    out.push(Pos { x, y });
                }
            }
        }
    }
    out
}

fn empty_cycle_path(w: usize, h: usize) -> Option<Vec<Pos>> {
    if w == 1 || h == 1 || (w * h) % 2 == 1 {
        return None;
    }
    if h % 2 == 0 {
        let mut out = Vec::with_capacity(w * h);
        for y in 0..h as i32 {
            if y == 0 {
                for x in 0..w as i32 {
                    out.push(Pos { x, y });
                }
            } else if y % 2 == 1 {
                out.push(Pos {
                    x: (w - 1) as i32,
                    y,
                });
            } else {
                out.push(Pos { x: 1, y });
                for x in 2..w as i32 {
                    out.push(Pos { x, y });
                }
            }
        }
        for y in (1..h as i32).rev() {
            if y % 2 == 0 {
                out.push(Pos { x: 0, y });
            } else {
                for x in (0..(w as i32 - 1)).rev() {
                    if x != 1 || y != 1 {
                        out.push(Pos { x, y });
                    }
                }
            }
        }
        out.dedup();
        if out.len() == w * h
            && is_adj(out[0], *out.last().unwrap())
            && out.windows(2).all(|ab| is_adj(ab[0], ab[1]))
        {
            return Some(out);
        }
    }
    if w % 2 == 0 {
        let rot = empty_cycle_path(h, w)?;
        let mut out = Vec::with_capacity(rot.len());
        for p in rot {
            out.push(Pos { x: p.y, y: p.x });
        }
        return Some(out);
    }
    None
}

fn greedy_path(grid: &Grid, require_cycle: bool) -> Option<Vec<Pos>> {
    let total = grid.free_count();
    let start = Pos { x: 0, y: 0 };
    if !grid.is_free(0, 0) || !connected_all(grid, start) {
        return None;
    }
    let (a, b) = bipartite_counts(grid);
    if require_cycle {
        if total % 2 == 1 || a != b {
            return None;
        }
    } else if (a as i32 - b as i32).abs() > 1 {
        return None;
    }

    let mut used = vec![false; grid.w * grid.h];
    let mut path = vec![start];
    used[grid.idx(0, 0)] = true;

    while path.len() < total {
        let cur = *path.last().unwrap();
        let mut cands = Vec::new();
        for n in grid.neighbors(cur) {
            let idx = grid.idx(n.x, n.y);
            if used[idx] {
                continue;
            }
            let deg = degree_in_remaining(grid, n, &used);
            let mut tmp = used.clone();
            tmp[idx] = true;
            let conn_penalty = if remaining_connected(grid, n, &tmp) { 0 } else { 1000 };
            let close_start = if require_cycle { n.manhattan(start) } else { 0 };
            cands.push((conn_penalty, deg, close_start, n));
        }
        cands.sort_unstable();
        let mut advanced = false;
        for (_, _, _, n) in cands {
            let idx = grid.idx(n.x, n.y);
            used[idx] = true;
            path.push(n);
            if remaining_connected(grid, n, &used) {
                advanced = true;
                break;
            }
            path.pop();
            used[idx] = false;
        }
        if !advanced {
            return None;
        }
    }
    if require_cycle && !is_adj(*path.last().unwrap(), start) {
        return None;
    }
    Some(path)
}

fn ida_search(grid: &Grid, require_cycle: bool, limit_ms: u128) -> Option<Vec<Pos>> {
    let start_time = std::time::Instant::now();
    let total = grid.free_count();
    let start = Pos { x: 0, y: 0 };
    if !grid.is_free(0, 0) || !connected_all(grid, start) {
        return None;
    }
    let mut used = vec![false; grid.w * grid.h];
    let path = vec![start];
    used[grid.idx(0, 0)] = true;
    let mut heap = BinaryHeap::new();
    heap.push((Reverse(0usize), path.clone(), used.clone()));

    while let Some((_, cur_path, cur_used)) = heap.pop() {
        if start_time.elapsed().as_millis() > limit_ms {
            break;
        }
        if cur_path.len() == total {
            if !require_cycle || is_adj(*cur_path.last().unwrap(), start) {
                return Some(cur_path);
            }
            continue;
        }
        let cur = *cur_path.last().unwrap();
        let mut cands = Vec::new();
        for n in grid.neighbors(cur) {
            let idx = grid.idx(n.x, n.y);
            if cur_used[idx] {
                continue;
            }
            let deg = degree_in_remaining(grid, n, &cur_used);
            cands.push((deg, n.manhattan(start), n));
        }
        cands.sort_unstable();
        for (deg, dist, n) in cands {
            let idx = grid.idx(n.x, n.y);
            let mut next_used = cur_used.clone();
            next_used[idx] = true;
            if !remaining_connected(grid, n, &next_used) {
                continue;
            }
            let mut next_path = cur_path.clone();
            next_path.push(n);
            let score = deg + dist as usize + (total - next_path.len());
            heap.push((Reverse(score), next_path, next_used));
        }
    }
    None
}

fn solve(grid: &Grid, require_cycle: bool) -> Option<Vec<Pos>> {
    if !grid.is_free(0, 0) {
        return None;
    }
    let total = grid.free_count();
    if total == 0 {
        return None;
    }
    if grid.blocked.iter().all(|&b| !b) {
        if require_cycle {
            if let Some(p) = empty_cycle_path(grid.w, grid.h) {
                return Some(p);
            }
        } else {
            return Some(empty_serpentine_path(grid.w, grid.h));
        }
    }
    if total <= 28 {
        if let Some(p) = solve_exact(grid, require_cycle) {
            return Some(p);
        }
    }
    if total <= 120 {
        if let Some(p) = ida_search(grid, require_cycle, 10_000) {
            return Some(p);
        }
        if let Some(p) = greedy_path(grid, require_cycle) {
            return Some(p);
        }
    } else if let Some(p) = greedy_path(grid, require_cycle) {
        return Some(p);
    }

    // Fallback: try solving exact on the reachable component if the obstacle pattern
    // effectively reduces the instance, otherwise report failure conservatively.
    if total <= 40 {
        solve_exact(grid, require_cycle)
    } else {
        None
    }
}

fn validate(grid: &Grid, require_cycle: bool, path: &[Pos]) -> bool {
    if path.is_empty() {
        return false;
    }
    if path[0] != (Pos { x: 0, y: 0 }) {
        return false;
    }
    if path.len() != grid.free_count() {
        return false;
    }
    let mut seen = HashSet::new();
    for &p in path {
        if !grid.is_free(p.x, p.y) || !seen.insert(p) {
            return false;
        }
    }
    for ab in path.windows(2) {
        if !is_adj(ab[0], ab[1]) {
            return false;
        }
    }
    if require_cycle && !is_adj(*path.last().unwrap(), path[0]) {
        return false;
    }
    true
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let obstacle_count: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<u32>().ok()) {
        Some(v) => v != 0,
        None => return,
    };

    let mut blocked = vec![false; w * h];
    for _ in 0..obstacle_count {
        let x: i32 = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => return,
        };
        let y: i32 = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => return,
        };
        if x >= 0 && y >= 0 && (x as usize) < w && (y as usize) < h {
            blocked[y as usize * w + x as usize] = true;
        }
    }

    let grid = Grid::new(w, h, blocked);
    let ans = solve(&grid, require_cycle);
    match ans {
        Some(path) if validate(&grid, require_cycle, &path) => {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
        }
        _ => {
            println!("0");
        }
    }
}
