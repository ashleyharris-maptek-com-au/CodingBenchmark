using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++)
            {
                flow[i][j] = scanner.NextInt();
            }
        }

        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++)
            {
                dist[i][j] = scanner.NextInt();
            }
        }

        int[] perm = n <= 10 ? SolveExactly(flow, dist) : SolveHeuristic(flow, dist);
        long cost = ComputeCost(flow, dist, perm);

        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(cost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < n; i++)
        {
            if (i > 0) writer.Write(' ');
            writer.Write(perm[i].ToString(CultureInfo.InvariantCulture));
        }
        writer.WriteLine();
        writer.Flush();
    }

    private static int[] SolveExactly(int[][] flow, int[][] dist)
    {
        int n = flow.Length;
        long[,] flowContrib = new long[n, n];
        long[,] distContrib = new long[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                flowContrib[i, j] = flow[i][j] + flow[j][i];
                distContrib[i, j] = dist[i][j] + dist[j][i];
            }
        }

        int[] facilityOrder = OrderDescendingByWeight(flow);
        int[] locationOrder = OrderAscendingByWeight(dist);

        int[] current = new int[n];
        Array.Fill(current, -1);
        bool[] used = new bool[n];
        long best = long.MaxValue;
        int[] bestPerm = new int[n];

        void Dfs(int depth, long partial)
        {
            if (depth == n)
            {
                if (partial < best)
                {
                    best = partial;
                    Array.Copy(current, bestPerm, n);
                }
                return;
            }

            int facility = facilityOrder[depth];
            for (int candidateIndex = 0; candidateIndex < n; candidateIndex++)
            {
                int loc = locationOrder[candidateIndex];
                if (used[loc]) continue;

                long add = (long)flow[facility][facility] * dist[loc][loc];
                for (int t = 0; t < depth; t++)
                {
                    int otherFacility = facilityOrder[t];
                    int otherLoc = current[otherFacility];
                    add += flowContrib[facility, otherFacility] * dist[loc][otherLoc];
                }

                long next = partial + add;
                if (next >= best) continue;

                current[facility] = loc;
                used[loc] = true;
                Dfs(depth + 1, next);
                used[loc] = false;
                current[facility] = -1;
            }
        }

        Dfs(0, 0L);
        return bestPerm;
    }

    private static int[] SolveHeuristic(int[][] flow, int[][] dist)
    {
        int n = flow.Length;
        long[] facilityWeight = ComputeNodeWeights(flow);
        long[] locationWeight = ComputeNodeWeights(dist);

        int[] facilitiesByWeight = OrderDescending(facilityWeight);
        int[] locationsByWeight = OrderAscending(locationWeight);

        int[] perm = new int[n];
        int[] inv = new int[n];
        for (int rank = 0; rank < n; rank++)
        {
            int facility = facilitiesByWeight[rank];
            int location = locationsByWeight[rank];
            perm[facility] = location;
            inv[location] = facility;
        }

        ImproveByReinsertion(flow, dist, perm, inv, facilityWeight, locationWeight);
        ImproveBySwaps(flow, dist, perm, inv, facilityWeight, locationWeight);
        return perm;
    }

    private static void ImproveByReinsertion(int[][] flow, int[][] dist, int[] perm, int[] inv, long[] facilityWeight, long[] locationWeight)
    {
        int n = flow.Length;
        int sample = Math.Min(n, n <= 200 ? n : 64);
        int[] heavyFacilities = TopKIndices(facilityWeight, sample, descending: true);
        int[] centralLocations = TopKIndices(locationWeight, sample, descending: false);
        bool changed = true;
        int passes = n <= 200 ? 4 : 2;

        for (int pass = 0; pass < passes && changed; pass++)
        {
            changed = false;
            foreach (int facility in heavyFacilities)
            {
                int currentLoc = perm[facility];
                long bestDelta = 0;
                int bestOtherFacility = -1;

                foreach (int candidateLoc in centralLocations)
                {
                    if (candidateLoc == currentLoc) continue;
                    int otherFacility = inv[candidateLoc];
                    long delta = ComputeSwapDelta(flow, dist, perm, facility, otherFacility);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestOtherFacility = otherFacility;
                    }
                }

                if (bestOtherFacility >= 0)
                {
                    ApplySwap(perm, inv, facility, bestOtherFacility);
                    changed = true;
                }
            }
        }
    }

    private static void ImproveBySwaps(int[][] flow, int[][] dist, int[] perm, int[] inv, long[] facilityWeight, long[] locationWeight)
    {
        int n = flow.Length;
        int facilitySample = n <= 150 ? n : Math.Min(96, n);
        int locationSample = n <= 150 ? n : Math.Min(96, n);
        int[] facilityCandidates = TopKIndices(facilityWeight, facilitySample, descending: true);
        int[] locationCandidates = TopKIndices(locationWeight, locationSample, descending: false);
        var locationSet = new HashSet<int>(locationCandidates);

        bool changed = true;
        int passes = n <= 150 ? 5 : 2;

        for (int pass = 0; pass < passes && changed; pass++)
        {
            changed = false;

            for (int aIndex = 0; aIndex < facilityCandidates.Length; aIndex++)
            {
                int a = facilityCandidates[aIndex];
                long bestDelta = 0;
                int bestB = -1;

                for (int bIndex = aIndex + 1; bIndex < facilityCandidates.Length; bIndex++)
                {
                    int b = facilityCandidates[bIndex];
                    if (!locationSet.Contains(perm[a]) && !locationSet.Contains(perm[b]) && n > 300)
                    {
                        continue;
                    }

                    long delta = ComputeSwapDelta(flow, dist, perm, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestB = b;
                    }
                }

                if (bestB >= 0)
                {
                    ApplySwap(perm, inv, a, bestB);
                    changed = true;
                }
            }
        }
    }

    private static long ComputeSwapDelta(int[][] flow, int[][] dist, int[] perm, int a, int b)
    {
        if (a == b) return 0;
        int n = flow.Length;
        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];
            delta += (long)flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
            delta += (long)flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
            delta += (long)flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
            delta += (long)flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
        }

        delta += (long)flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        delta += (long)flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
        delta += (long)flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        delta += (long)flow[b][a] * (dist[pa][pb] - dist[pb][pa]);
        return delta;
    }

    private static void ApplySwap(int[] perm, int[] inv, int a, int b)
    {
        int pa = perm[a];
        int pb = perm[b];
        perm[a] = pb;
        perm[b] = pa;
        inv[pa] = b;
        inv[pb] = a;
    }

    private static long[] ComputeNodeWeights(int[][] matrix)
    {
        int n = matrix.Length;
        long[] weights = new long[n];
        for (int i = 0; i < n; i++)
        {
            long sum = 0;
            int[] row = matrix[i];
            for (int j = 0; j < n; j++)
            {
                sum += row[j];
                sum += matrix[j][i];
            }
            weights[i] = sum;
        }
        return weights;
    }

    private static int[] OrderDescendingByWeight(int[][] matrix)
    {
        return OrderDescending(ComputeNodeWeights(matrix));
    }

    private static int[] OrderAscendingByWeight(int[][] matrix)
    {
        return OrderAscending(ComputeNodeWeights(matrix));
    }

    private static int[] OrderDescending(long[] values)
    {
        int n = values.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) =>
        {
            int cmp = values[b].CompareTo(values[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        return idx;
    }

    private static int[] OrderAscending(long[] values)
    {
        int n = values.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) =>
        {
            int cmp = values[a].CompareTo(values[b]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        return idx;
    }

    private static int[] TopKIndices(long[] values, int k, bool descending)
    {
        int[] order = descending ? OrderDescending(values) : OrderAscending(values);
        if (k >= order.Length) return order;
        int[] result = new int[k];
        Array.Copy(order, result, k);
        return result;
    }

    private static long ComputeCost(int[][] flow, int[][] dist, int[] perm)
    {
        int n = flow.Length;
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int[] flowRow = flow[i];
            int[] distRow = dist[pi];
            for (int j = 0; j < n; j++)
            {
                total += (long)flowRow[j] * distRow[perm[j]];
            }
        }
        return total;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }
}
