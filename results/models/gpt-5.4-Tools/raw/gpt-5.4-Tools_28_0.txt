use std::collections::{BinaryHeap, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, pos: 0 })
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            let digit = self.data[self.pos].wrapping_sub(b'0');
            value = value.saturating_mul(10).saturating_add(digit as usize);
            self.pos += 1;
        }
        Some(value)
    }
}

struct MatchingContext<'a> {
    adj: &'a [Vec<usize>],
    pair_u: Vec<usize>,
    pair_v: Vec<usize>,
    dist: Vec<i32>,
}

impl<'a> MatchingContext<'a> {
    fn new(adj: &'a [Vec<usize>], right_size: usize) -> Self {
        Self {
            adj,
            pair_u: vec![usize::MAX; adj.len()],
            pair_v: vec![usize::MAX; right_size],
            dist: vec![0; adj.len()],
        }
    }

    fn bfs(&mut self) -> bool {
        let mut queue = VecDeque::new();
        let mut found = false;
        for u in 0..self.adj.len() {
            if self.pair_u[u] == usize::MAX {
                self.dist[u] = 0;
                queue.push_back(u);
            } else {
                self.dist[u] = -1;
            }
        }
        while let Some(u) = queue.pop_front() {
            for &v in &self.adj[u] {
                let mate = self.pair_v[v];
                if mate == usize::MAX {
                    found = true;
                } else if self.dist[mate] < 0 {
                    self.dist[mate] = self.dist[u] + 1;
                    queue.push_back(mate);
                }
            }
        }
        found
    }

    fn dfs(&mut self, u: usize) -> bool {
        let neighbors = self.adj[u].clone();
        for v in neighbors {
            let mate = self.pair_v[v];
            if mate == usize::MAX
                || (self.dist[mate] == self.dist[u] + 1 && self.dfs(mate))
            {
                self.pair_u[u] = v;
                self.pair_v[v] = u;
                return true;
            }
        }
        self.dist[u] = -1;
        false
    }

    fn maximum_matching(&mut self) {
        while self.bfs() {
            for u in 0..self.adj.len() {
                if self.pair_u[u] == usize::MAX {
                    let _ = self.dfs(u);
                }
            }
        }
    }
}

fn bipartite_min_vertex_cover(
    vertices: &[usize],
    edges: &[Edge],
    side: &[u8],
) -> Vec<usize> {
    let mut left_map = vec![usize::MAX; vertices.len()];
    let mut right_map = vec![usize::MAX; vertices.len()];
    let mut left_vertices = Vec::new();
    let mut right_vertices = Vec::new();
    for (idx, &v) in vertices.iter().enumerate() {
        if side[idx] == 0 {
            left_map[idx] = left_vertices.len();
            left_vertices.push(v);
        } else {
            right_map[idx] = right_vertices.len();
            right_vertices.push(v);
        }
    }

    let mut pos_in_component = vec![usize::MAX; vertices.iter().copied().max().unwrap_or(0) + 1];
    for (idx, &v) in vertices.iter().enumerate() {
        pos_in_component[v] = idx;
    }

    let mut adj = vec![Vec::new(); left_vertices.len()];
    for edge in edges {
        let a = pos_in_component[edge.u];
        let b = pos_in_component[edge.v];
        if side[a] == 0 {
            adj[left_map[a]].push(right_map[b]);
        } else {
            adj[left_map[b]].push(right_map[a]);
        }
    }

    let mut matcher = MatchingContext::new(&adj, right_vertices.len());
    matcher.maximum_matching();

    let mut vis_left = vec![false; left_vertices.len()];
    let mut vis_right = vec![false; right_vertices.len()];
    let mut queue = VecDeque::new();
    for u in 0..left_vertices.len() {
        if matcher.pair_u[u] == usize::MAX {
            vis_left[u] = true;
            queue.push_back((0u8, u));
        }
    }

    while let Some((kind, idx)) = queue.pop_front() {
        if kind == 0 {
            for &v in &adj[idx] {
                if matcher.pair_u[idx] != v && !vis_right[v] {
                    vis_right[v] = true;
                    queue.push_back((1, v));
                }
            }
        } else {
            let mate = matcher.pair_v[idx];
            if mate != usize::MAX && !vis_left[mate] {
                vis_left[mate] = true;
                queue.push_back((0, mate));
            }
        }
    }

    let mut cover = Vec::new();
    for (i, &v) in left_vertices.iter().enumerate() {
        if !vis_left[i] {
            cover.push(v);
        }
    }
    for (i, &v) in right_vertices.iter().enumerate() {
        if vis_right[i] {
            cover.push(v);
        }
    }
    cover
}

fn greedy_cover(n: usize, edges: &[Edge]) -> Vec<usize> {
    let mut adj = vec![Vec::new(); n];
    let mut alive = vec![true; edges.len()];
    let mut degree = vec![0usize; n];
    for (i, edge) in edges.iter().enumerate() {
        adj[edge.u].push(i);
        adj[edge.v].push(i);
        degree[edge.u] += 1;
        degree[edge.v] += 1;
    }
    let mut heap = BinaryHeap::new();
    for (v, &deg) in degree.iter().enumerate() {
        if deg > 0 {
            heap.push((deg, v));
        }
    }
    let mut cover = Vec::new();
    let mut uncovered = edges.len();
    while uncovered > 0 {
        let Some((deg, v)) = heap.pop() else {
            break;
        };
        if degree[v] != deg || degree[v] == 0 {
            continue;
        }
        cover.push(v);
        for &eid in &adj[v] {
            if alive[eid] {
                alive[eid] = false;
                uncovered -= 1;
                let edge = edges[eid];
                if degree[edge.u] > 0 {
                    degree[edge.u] -= 1;
                    heap.push((degree[edge.u], edge.u));
                }
                if degree[edge.v] > 0 {
                    degree[edge.v] -= 1;
                    heap.push((degree[edge.v], edge.v));
                }
            }
        }
        degree[v] = 0;
    }

    let mut in_cover = vec![false; n];
    for &v in &cover {
        in_cover[v] = true;
    }
    for edge in edges {
        if !in_cover[edge.u] && !in_cover[edge.v] {
            in_cover[edge.u] = true;
            cover.push(edge.u);
        }
    }
    cover
}

fn component_data(
    start: usize,
    adj: &[Vec<usize>],
    edges: &[Edge],
    seen: &mut [bool],
) -> (Vec<usize>, Vec<Edge>, bool, Vec<u8>) {
    let mut queue = VecDeque::new();
    let mut vertices = Vec::new();
    let mut component_edges = Vec::new();
    let mut side = Vec::new();
    let mut vertex_pos = vec![usize::MAX; adj.len()];
    let mut colors = vec![2u8; adj.len()];
    let mut bipartite = true;

    seen[start] = true;
    colors[start] = 0;
    queue.push_back(start);

    while let Some(v) = queue.pop_front() {
        vertex_pos[v] = vertices.len();
        vertices.push(v);
        side.push(colors[v]);
        for &eid in &adj[v] {
            let edge = edges[eid];
            let to = if edge.u == v { edge.v } else { edge.u };
            if colors[to] == 2 {
                colors[to] = colors[v] ^ 1;
            } else if colors[to] == colors[v] {
                bipartite = false;
            }
            if !seen[to] {
                seen[to] = true;
                queue.push_back(to);
            }
            if v < to {
                component_edges.push(edge);
            }
        }
    }

    (vertices, component_edges, bipartite, side)
}

fn solve() -> io::Result<String> {
    let mut input = FastInput::new()?;
    let n = input.next_usize().unwrap_or(0);
    let m = input.next_usize().unwrap_or(0);
    let mut edges = Vec::with_capacity(m);
    let mut adj = vec![Vec::new(); n];
    for i in 0..m {
        let Some(u) = input.next_usize() else {
            break;
        };
        let Some(v) = input.next_usize() else {
            break;
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        edges.push(Edge { u, v });
        let eid = edges.len() - 1;
        adj[u].push(eid);
        adj[v].push(eid);
        let _ = i;
    }

    let edge_count = edges.len();
    if edge_count == 0 {
        return Ok("0\n\n".to_string());
    }

    let mut seen = vec![false; n];
    let mut cover = Vec::new();
    let exact_edge_limit = 1_500_000usize;
    let exact_vertex_limit = 400_000usize;
    let mut fallback_edges = Vec::new();
    let mut fallback_vertices = 0usize;

    for v in 0..n {
        if seen[v] || adj[v].is_empty() {
            continue;
        }
        let (vertices, component_edges, bipartite, side) = component_data(v, &adj, &edges, &mut seen);
        if bipartite && component_edges.len() <= exact_edge_limit && vertices.len() <= exact_vertex_limit {
            let part_cover = bipartite_min_vertex_cover(&vertices, &component_edges, &side);
            cover.extend(part_cover);
        } else {
            fallback_vertices = fallback_vertices.max(vertices.iter().copied().max().unwrap_or(0) + 1);
            fallback_edges.extend(component_edges);
        }
    }

    if !fallback_edges.is_empty() {
        let greedy = greedy_cover(fallback_vertices, &fallback_edges);
        cover.extend(greedy);
    }

    cover.sort_unstable();
    cover.dedup();

    let mut out = String::new();
    out.push_str(&cover.len().to_string());
    out.push('\n');
    for (i, v) in cover.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');
    Ok(out)
}

fn main() {
    match solve() {
        Ok(output) => {
            print!("{output}");
        }
        Err(_) => {
            print!("0\n\n");
        }
    }
}
