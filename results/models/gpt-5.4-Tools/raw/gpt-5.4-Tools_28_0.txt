use std::cmp::Reverse;
use std::collections::{BinaryHeap, HashMap, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

struct Scanner {
    data: Vec<u8>,
    index: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, index: 0 })
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.index < self.data.len() && self.data[self.index].is_ascii_whitespace() {
            self.index += 1;
        }
        if self.index >= self.data.len() {
            return None;
        }
        let mut value = 0usize;
        while self.index < self.data.len() && !self.data[self.index].is_ascii_whitespace() {
            value = value
                .saturating_mul(10)
                .saturating_add((self.data[self.index] - b'0') as usize);
            self.index += 1;
        }
        Some(value)
    }
}

fn add_cover_vertex(
    v: usize,
    adjacency: &[Vec<usize>],
    edges: &[Edge],
    edge_alive: &mut [bool],
    degree: &mut [usize],
    in_cover: &mut [bool],
    alive_edges: &mut usize,
) {
    if in_cover[v] {
        return;
    }
    in_cover[v] = true;
    for &eid in &adjacency[v] {
        if edge_alive[eid] {
            edge_alive[eid] = false;
            *alive_edges -= 1;
            let edge = edges[eid];
            degree[edge.u] = degree[edge.u].saturating_sub(1);
            degree[edge.v] = degree[edge.v].saturating_sub(1);
        }
    }
}

fn peel_reductions(
    adjacency: &[Vec<usize>],
    edges: &[Edge],
    edge_alive: &mut [bool],
    degree: &mut [usize],
    in_cover: &mut [bool],
    alive_edges: &mut usize,
) {
    let mut queue = VecDeque::new();
    for v in 0..degree.len() {
        if degree[v] == 1 && !in_cover[v] {
            queue.push_back(v);
        }
    }

    while let Some(v) = queue.pop_front() {
        if in_cover[v] || degree[v] != 1 {
            continue;
        }
        let mut neighbor = None;
        for &eid in &adjacency[v] {
            if edge_alive[eid] {
                let edge = edges[eid];
                neighbor = Some(if edge.u == v { edge.v } else { edge.u });
                break;
            }
        }
        let Some(u) = neighbor else {
            continue;
        };
        let prev_degree_u = degree[u];
        add_cover_vertex(
            u,
            adjacency,
            edges,
            edge_alive,
            degree,
            in_cover,
            alive_edges,
        );
        if prev_degree_u > 0 {
            for &eid in &adjacency[u] {
                let edge = edges[eid];
                let other = if edge.u == u { edge.v } else { edge.u };
                if !in_cover[other] && degree[other] == 1 {
                    queue.push_back(other);
                }
            }
        }
    }
}

fn maximal_matching_cover(
    adjacency: &[Vec<usize>],
    edges: &[Edge],
    edge_alive: &[bool],
    in_cover: &mut [bool],
) {
    let n = adjacency.len();
    let mut matched = vec![false; n];
    for (eid, edge) in edges.iter().enumerate() {
        if edge_alive[eid] && !matched[edge.u] && !matched[edge.v] {
            matched[edge.u] = true;
            matched[edge.v] = true;
            in_cover[edge.u] = true;
            in_cover[edge.v] = true;
        }
    }
}

fn greedy_high_degree_cover(
    adjacency: &[Vec<usize>],
    edges: &[Edge],
    edge_alive: &mut [bool],
    degree: &mut [usize],
    in_cover: &mut [bool],
    alive_edges: &mut usize,
) {
    let mut heap = BinaryHeap::new();
    for (v, &deg) in degree.iter().enumerate() {
        if deg > 0 && !in_cover[v] {
            heap.push((deg, v));
        }
    }

    while *alive_edges > 0 {
        peel_reductions(
            adjacency,
            edges,
            edge_alive,
            degree,
            in_cover,
            alive_edges,
        );
        if *alive_edges == 0 {
            break;
        }

        let selected = loop {
            let Some((deg, v)) = heap.pop() else {
                break None;
            };
            if !in_cover[v] && degree[v] == deg && deg > 0 {
                break Some(v);
            }
        };

        let Some(v) = selected else {
            break;
        };
        add_cover_vertex(
            v,
            adjacency,
            edges,
            edge_alive,
            degree,
            in_cover,
            alive_edges,
        );
        for &eid in &adjacency[v] {
            let edge = edges[eid];
            let other = if edge.u == v { edge.v } else { edge.u };
            if !in_cover[other] && degree[other] > 0 {
                heap.push((degree[other], other));
            }
        }
    }
}

fn remove_redundant_vertices(adjacency: &[Vec<usize>], cover: &mut [bool]) {
    let n = cover.len();
    let mut uncovered_count = vec![0usize; n];
    for v in 0..n {
        if !cover[v] {
            for &u in &adjacency[v] {
                if cover[u] {
                    uncovered_count[u] += 1;
                }
            }
        }
    }

    let mut heap = BinaryHeap::new();
    for v in 0..n {
        if cover[v] {
            heap.push((Reverse(uncovered_count[v]), v));
        }
    }

    while let Some((Reverse(cnt), v)) = heap.pop() {
        if !cover[v] || uncovered_count[v] != cnt {
            continue;
        }
        if cnt != 0 {
            continue;
        }
        cover[v] = false;
        for &u in &adjacency[v] {
            if !cover[u] {
                uncovered_count[v] += 1;
            } else {
                uncovered_count[u] += 1;
                heap.push((Reverse(uncovered_count[u]), u));
            }
        }
    }
}

fn is_bipartite_component(
    start: usize,
    graph: &[Vec<usize>],
    side: &mut [i8],
    comp_vertices: &mut Vec<usize>,
) -> bool {
    let mut queue = VecDeque::new();
    queue.push_back(start);
    side[start] = 0;
    comp_vertices.push(start);
    while let Some(v) = queue.pop_front() {
        let current = side[v];
        for &u in &graph[v] {
            if side[u] == -1 {
                side[u] = 1 - current;
                comp_vertices.push(u);
                queue.push_back(u);
            } else if side[u] == current {
                return false;
            }
        }
    }
    true
}

fn hopcroft_karp(
    left: &[usize],
    graph: &[Vec<usize>],
    side: &[i8],
    global_to_left: &[usize],
) -> (Vec<Option<usize>>, Vec<Option<usize>>) {
    let n = graph.len();
    let left_size = left.len();
    let mut pair_left = vec![None; left_size];
    let mut pair_right = vec![None; n];
    let mut dist = vec![usize::MAX; left_size];

    loop {
        let mut queue = VecDeque::new();
        for (i, _) in left.iter().enumerate() {
            if pair_left[i].is_none() {
                dist[i] = 0;
                queue.push_back(i);
            } else {
                dist[i] = usize::MAX;
            }
        }

        while let Some(i) = queue.pop_front() {
            let v = left[i];
            for &u in &graph[v] {
                if side[u] != 1 {
                    continue;
                }
                if let Some(matched_left) = pair_right[u] {
                    if dist[matched_left] == usize::MAX {
                        dist[matched_left] = dist[i] + 1;
                        queue.push_back(matched_left);
                    }
                }
            }
        }

        let mut progress = false;
        let mut next_edge_idx = vec![0usize; left_size];
        for i in 0..left_size {
            if pair_left[i].is_none()
                && dfs_hk(
                    i,
                    left,
                    graph,
                    side,
                    global_to_left,
                    &mut pair_left,
                    &mut pair_right,
                    &dist,
                    &mut next_edge_idx,
                )
            {
                progress = true;
            }
        }
        if !progress {
            break;
        }
    }

    (pair_left, pair_right)
}

fn dfs_hk(
    i: usize,
    left: &[usize],
    graph: &[Vec<usize>],
    side: &[i8],
    global_to_left: &[usize],
    pair_left: &mut [Option<usize>],
    pair_right: &mut [Option<usize>],
    dist: &[usize],
    next_edge_idx: &mut [usize],
) -> bool {
    let v = left[i];
    while next_edge_idx[i] < graph[v].len() {
        let u = graph[v][next_edge_idx[i]];
        next_edge_idx[i] += 1;
        if side[u] != 1 {
            continue;
        }
        match pair_right[u] {
            None => {
                pair_left[i] = Some(u);
                pair_right[u] = Some(i);
                return true;
            }
            Some(j) if dist[j] == dist[i] + 1 => {
                let _ = global_to_left;
                if dfs_hk(
                    j,
                    left,
                    graph,
                    side,
                    global_to_left,
                    pair_left,
                    pair_right,
                    dist,
                    next_edge_idx,
                ) {
                    pair_left[i] = Some(u);
                    pair_right[u] = Some(i);
                    return true;
                }
            }
            Some(_) => {}
        }
    }
    false
}

fn exact_bipartite_cover(graph: &[Vec<usize>]) -> Option<Vec<bool>> {
    let n = graph.len();
    let mut side = vec![-1i8; n];
    let mut cover = vec![false; n];

    for start in 0..n {
        if side[start] != -1 || graph[start].is_empty() {
            continue;
        }
        let mut component = Vec::new();
        if !is_bipartite_component(start, graph, &mut side, &mut component) {
            return None;
        }

        let left: Vec<usize> = component
            .iter()
            .copied()
            .filter(|&v| side[v] == 0)
            .collect();
        let mut global_to_left = vec![usize::MAX; n];
        for (idx, &v) in left.iter().enumerate() {
            global_to_left[v] = idx;
        }

        let (pair_left, pair_right) = hopcroft_karp(&left, graph, &side, &global_to_left);
        let _ = pair_left;

        let mut vis_left = vec![false; left.len()];
        let mut vis_right = vec![false; n];
        let mut queue = VecDeque::new();
        for (i, _) in left.iter().enumerate() {
            if pair_right.iter().all(|_| true) && false {
                unreachable!();
            }
            if hopcroft_karp_is_free_left(i, &pair_right, &left, &graph, &side) {
                vis_left[i] = true;
                queue.push_back(left[i]);
            }
        }

        while let Some(v) = queue.pop_front() {
            if side[v] == 0 {
                let li = global_to_left[v];
                for &u in &graph[v] {
                    if side[u] == 1 && pair_right[u] != Some(li) && !vis_right[u] {
                        vis_right[u] = true;
                        queue.push_back(u);
                    }
                }
            } else if let Some(li) = pair_right[v] {
                if !vis_left[li] {
                    vis_left[li] = true;
                    queue.push_back(left[li]);
                }
            }
        }

        for &v in &component {
            if side[v] == 0 {
                let li = global_to_left[v];
                if !vis_left[li] {
                    cover[v] = true;
                }
            } else if vis_right[v] {
                cover[v] = true;
            }
        }
    }

    Some(cover)
}

fn hopcroft_karp_is_free_left(
    i: usize,
    pair_right: &[Option<usize>],
    left: &[usize],
    graph: &[Vec<usize>],
    side: &[i8],
) -> bool {
    let v = left[i];
    for &u in &graph[v] {
        if side[u] == 1 && pair_right[u] == Some(i) {
            return false;
        }
    }
    true
}

fn solve_general(n: usize, edges: &[Edge]) -> Vec<bool> {
    let mut adjacency = vec![Vec::new(); n];
    for (eid, edge) in edges.iter().enumerate() {
        adjacency[edge.u].push(eid);
        adjacency[edge.v].push(eid);
    }

    let mut edge_alive = vec![true; edges.len()];
    let mut degree: Vec<usize> = adjacency.iter().map(Vec::len).collect();
    let mut in_cover = vec![false; n];
    let mut alive_edges = edges.len();

    peel_reductions(
        &adjacency,
        edges,
        &mut edge_alive,
        &mut degree,
        &mut in_cover,
        &mut alive_edges,
    );

    if alive_edges > 0 {
        let residual_vertices = degree.iter().filter(|&&d| d > 0).count();
        if alive_edges <= 300_000 && residual_vertices <= 200_000 {
            let mut residual_graph = vec![Vec::new(); n];
            for (eid, edge) in edges.iter().enumerate() {
                if edge_alive[eid] {
                    residual_graph[edge.u].push(edge.v);
                    residual_graph[edge.v].push(edge.u);
                }
            }
            if let Some(exact_cover) = exact_bipartite_cover(&residual_graph) {
                for v in 0..n {
                    if exact_cover[v] {
                        in_cover[v] = true;
                    }
                }
                return in_cover;
            }
        }

        let mut greedy_cover = in_cover.clone();
        let mut greedy_edge_alive = edge_alive.clone();
        let mut greedy_degree = degree.clone();
        let mut greedy_alive_edges = alive_edges;
        greedy_high_degree_cover(
            &adjacency,
            edges,
            &mut greedy_edge_alive,
            &mut greedy_degree,
            &mut greedy_cover,
            &mut greedy_alive_edges,
        );

        let mut matching_cover = in_cover.clone();
        maximal_matching_cover(&adjacency, edges, &edge_alive, &mut matching_cover);

        if matching_cover.iter().filter(|&&x| x).count() < greedy_cover.iter().filter(|&&x| x).count()
        {
            in_cover = matching_cover;
        } else {
            in_cover = greedy_cover;
        }
    }

    let mut simple_graph = vec![Vec::new(); n];
    for edge in edges {
        simple_graph[edge.u].push(edge.v);
        simple_graph[edge.v].push(edge.u);
    }
    remove_redundant_vertices(&simple_graph, &mut in_cover);
    in_cover
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;
    let n = scanner.next_usize().unwrap_or(0);
    let m = scanner.next_usize().unwrap_or(0);

    let mut dedup = HashMap::<u64, ()>::with_capacity(m.saturating_mul(2).saturating_add(1));
    let mut edges = Vec::with_capacity(m);

    for _ in 0..m {
        let mut u = scanner.next_usize().unwrap_or(0);
        let mut v = scanner.next_usize().unwrap_or(0);
        if u == v || u >= n || v >= n {
            continue;
        }
        if u > v {
            std::mem::swap(&mut u, &mut v);
        }
        let key = ((u as u64) << 32) | (v as u64);
        if dedup.insert(key, ()).is_none() {
            edges.push(Edge { u, v });
        }
    }

    let cover = solve_general(n, &edges);
    let result: Vec<usize> = cover
        .iter()
        .enumerate()
        .filter_map(|(v, &take)| if take { Some(v) } else { None })
        .collect();

    let mut output = String::new();
    output.push_str(&format!("{}\n", result.len()));
    for (i, v) in result.iter().enumerate() {
        if i > 0 {
            output.push(' ');
        }
        output.push_str(&v.to_string());
    }
    output.push('\n');
    print!("{output}");
    Ok(())
}
