using namespace std;

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<pair<int, int>> edges;
    vector<vector<int>> adj;
    vector<int> degree;
};

static inline bool has_bit(const vector<uint64_t>& bits, int idx) {
    return (bits[idx >> 6] >> (idx & 63)) & 1ULL;
}

static inline void set_bit(vector<uint64_t>& bits, int idx) {
    bits[idx >> 6] |= (1ULL << (idx & 63));
}

static bool greedy_coloring(const Graph& g, vector<int>& color, bool dynamic_order) {
    const int n = g.n;
    const int k = g.k;
    color.assign(n, -1);
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);

    if (!dynamic_order) {
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
            return a < b;
        });
    }

    vector<int> mark(k, -1);
    if (!dynamic_order) {
        for (int v : order) {
            for (int u : g.adj[v]) {
                int c = color[u];
                if (c >= 0) mark[c] = v;
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (mark[c] != v) {
                    chosen = c;
                    break;
                }
            }
            if (chosen < 0) return false;
            color[v] = chosen;
        }
        return true;
    }

    vector<int> uncolored_deg = g.degree;
    vector<int> sat(n, 0);
    const int words = (k + 63) >> 6;
    vector<vector<uint64_t>> neigh_colors(n, vector<uint64_t>(words, 0ULL));

    for (int step = 0; step < n; ++step) {
        int best = -1;
        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            if (best == -1 || sat[v] > sat[best] ||
                (sat[v] == sat[best] && uncolored_deg[v] > uncolored_deg[best]) ||
                (sat[v] == sat[best] && uncolored_deg[v] == uncolored_deg[best] && g.degree[v] > g.degree[best])) {
                best = v;
            }
        }
        int v = best;
        if (v == -1) break;

        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (!has_bit(neigh_colors[v], c)) {
                chosen = c;
                break;
            }
        }
        if (chosen < 0) return false;
        color[v] = chosen;

        for (int u : g.adj[v]) {
            if (color[u] != -1) continue;
            --uncolored_deg[u];
            if (!has_bit(neigh_colors[u], chosen)) {
                set_bit(neigh_colors[u], chosen);
                ++sat[u];
            }
        }
    }
    return true;
}

static bool repair_conflicts(const Graph& g, vector<int>& color, int max_passes, uint64_t seed) {
    const int n = g.n;
    const int k = g.k;
    if (n == 0) return true;

    mt19937_64 rng(seed);
    vector<int> conflicts(n, 0);
    vector<int> color_count(k, 0);
    vector<int> touched;
    touched.reserve(k);

    auto recompute_conflicts = [&]() -> long long {
        fill(conflicts.begin(), conflicts.end(), 0);
        long long total = 0;
        for (auto [u, v] : g.edges) {
            if (color[u] == color[v]) {
                ++conflicts[u];
                ++conflicts[v];
                ++total;
            }
        }
        return total;
    };

    long long bad_edges = recompute_conflicts();
    if (bad_edges == 0) return true;

    vector<int> conflicted;
    conflicted.reserve(n);

    for (int pass = 0; pass < max_passes; ++pass) {
        conflicted.clear();
        for (int v = 0; v < n; ++v) {
            if (conflicts[v] > 0) conflicted.push_back(v);
        }
        if (conflicted.empty()) return true;

        int v = conflicted[rng() % conflicted.size()];
        fill(color_count.begin(), color_count.end(), 0);
        touched.clear();

        for (int u : g.adj[v]) {
            int c = color[u];
            if (color_count[c] == 0) touched.push_back(c);
            ++color_count[c];
        }

        int current = color[v];
        int best_color = current;
        int best_conf = color_count[current];

        for (int c = 0; c < k; ++c) {
            int cur_conf = color_count[c];
            if (cur_conf < best_conf) {
                best_conf = cur_conf;
                best_color = c;
                if (best_conf == 0) break;
            }
        }

        if (best_color == current) {
            int c = static_cast<int>(rng() % k);
            best_color = c;
        }

        if (best_color != current) {
            for (int u : g.adj[v]) {
                if (color[u] == current) --conflicts[u];
                if (color[u] == best_color) ++conflicts[u];
            }
            conflicts[v] += color_count[best_color] - color_count[current];
            color[v] = best_color;
        }

        if ((pass & 255) == 255) {
            bad_edges = recompute_conflicts();
            if (bad_edges == 0) return true;
        }
    }

    return recompute_conflicts() == 0;
}

static bool solve_small_exact(const Graph& g, vector<int>& color) {
    const int n = g.n;
    const int k = g.k;
    color.assign(n, -1);
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
        return a < b;
    });

    vector<int> pos(n, -1);
    for (int i = 0; i < n; ++i) pos[order[i]] = i;

    vector<vector<int>> earlier(n);
    for (int idx = 0; idx < n; ++idx) {
        int v = order[idx];
        for (int u : g.adj[v]) {
            if (pos[u] < idx) earlier[idx].push_back(u);
        }
    }

    vector<int> used(k, -1);
    function<bool(int)> dfs = [&](int idx) -> bool {
        if (idx == n) return true;
        int v = order[idx];
        for (int u : earlier[idx]) {
            int c = color[u];
            if (c >= 0) used[c] = idx;
        }
        for (int c = 0; c < k; ++c) {
            if (used[c] == idx) continue;
            color[v] = c;
            if (dfs(idx + 1)) return true;
            color[v] = -1;
        }
        return false;
    };

    return dfs(0);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    if (!(cin >> g.n >> g.m >> g.k)) {
        return 0;
    }

    if (g.k <= 0) {
        return 0;
    }

    g.edges.reserve(g.m);
    g.degree.assign(g.n, 0);
    unordered_set<uint64_t> seen;
    seen.reserve(static_cast<size_t>(g.m) * 2 + 1);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= g.n || v >= g.n || u == v) continue;
        if (u > v) swap(u, v);
        uint64_t key = (static_cast<uint64_t>(static_cast<uint32_t>(u)) << 32) |
                       static_cast<uint32_t>(v);
        if (seen.insert(key).second) {
            g.edges.push_back({u, v});
            ++g.degree[u];
            ++g.degree[v];
        }
    }
    g.m = static_cast<int>(g.edges.size());
    g.adj.assign(g.n, {});
    for (int v = 0; v < g.n; ++v) {
        g.adj[v].reserve(g.degree[v]);
    }
    for (auto [u, v] : g.edges) {
        g.adj[u].push_back(v);
        g.adj[v].push_back(u);
    }

    vector<int> color;

    bool ok = false;
    long long density2 = (g.n <= 1) ? 0 : (2LL * g.m);
    long long denom = 1LL * g.n * (g.n - 1);
    bool dense = denom > 0 && density2 * 100 >= denom * 20;

    if (g.n <= 40) {
        ok = solve_small_exact(g, color);
    }

    if (!ok) {
        bool use_dynamic = (g.n <= 200000 && g.k <= 256) || dense;
        ok = greedy_coloring(g, color, use_dynamic);
    }

    if (!ok) {
        color.resize(g.n);
        for (int i = 0; i < g.n; ++i) {
            color[i] = i % g.k;
        }
        ok = false;
    }

    if (!ok && g.k > 0) {
        uint64_t seed = chrono::steady_clock::now().time_since_epoch().count();
        ok = repair_conflicts(g, color, min(5000000, max(100000, g.n * 20)), seed);
    }

    if (!ok) {
        vector<int> fallback;
        if (greedy_coloring(g, fallback, false)) {
            color.swap(fallback);
            ok = true;
        }
    }

    if (!ok) {
        for (int i = 0; i < g.n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << '\n';
        return 0;
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        cout << color[i];
    }
    cout << '\n';
    return 0;
}
