using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Text;

public static class Program
{
    private const double EarthRadiusKm = 6371.0;

    private sealed class FastScanner
    {
        private readonly System.IO.TextReader _reader = Console.In;
        private readonly Queue<string> _tokens = new Queue<string>();

        public string Next()
        {
            while (_tokens.Count == 0)
            {
                var line = _reader.ReadLine();
                if (line == null)
                {
                    return string.Empty;
                }
                foreach (var part in line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries))
                {
                    _tokens.Enqueue(part);
                }
            }
            return _tokens.Dequeue();
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;

        public Vec3 Cross(Vec3 b) => new Vec3(
            Y * b.Z - Z * b.Y,
            Z * b.X - X * b.Z,
            X * b.Y - Y * b.X
        );

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public Vec3 Normalize()
        {
            var n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    private sealed class OrbitPoint
    {
        public Vec3 R = new Vec3();
        public Vec3 V = new Vec3();
        public double Radius;
        public double Speed;
        public Vec3 PlaneNormal = new Vec3();
        public double Inclination;
        public double Energy;
        public double SemiMajorAxis;
        public double Eccentricity;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        var first = fs.Next();
        if (string.IsNullOrEmpty(first))
        {
            return;
        }

        int n = int.Parse(first, CultureInfo.InvariantCulture);
        double mu = fs.NextDouble();

        var start = ReadOrbit(fs, mu);
        var stations = new OrbitPoint[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = ReadOrbit(fs, mu);
        }

        var startCosts = new double[n];
        var pairCosts = new double[n, n];
        for (int i = 0; i < n; i++)
        {
            startCosts[i] = TransferCost(start, stations[i], mu);
        }
        for (int i = 0; i < n; i++)
        {
            pairCosts[i, i] = 0;
            for (int j = i + 1; j < n; j++)
            {
                double c = TransferCost(stations[i], stations[j], mu);
                pairCosts[i, j] = c;
                pairCosts[j, i] = c;
            }
        }

        int[] route;
        if (n <= 8)
        {
            route = SolveExact(n, startCosts, pairCosts);
        }
        else if (n <= 18)
        {
            route = SolveBeam(n, startCosts, pairCosts, beamWidth: 48);
            ImproveRoute(route, startCosts, pairCosts);
        }
        else
        {
            route = SolveGreedyMultiStart(n, startCosts, pairCosts);
            ImproveRoute(route, startCosts, pairCosts);
        }

        Console.WriteLine(string.Join(" ", route));
    }

    private static OrbitPoint ReadOrbit(FastScanner fs, double mu)
    {
        var p = new OrbitPoint();
        p.R = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        p.V = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        p.Radius = p.R.Norm();
        p.Speed = p.V.Norm();
        var h = p.R.Cross(p.V);
        p.PlaneNormal = h.Normalize();
        double hz = h.Norm() > 0 ? h.Z / h.Norm() : 1.0;
        hz = Math.Max(-1.0, Math.Min(1.0, hz));
        p.Inclination = Math.Acos(hz);
        p.Energy = 0.5 * p.Speed * p.Speed - mu / p.Radius;
        p.SemiMajorAxis = Math.Abs(p.Energy) > 1e-12 ? -mu / (2.0 * p.Energy) : 1e12;
        var eVec = (p.V.Cross(h) / mu) - (p.R / p.Radius);
        p.Eccentricity = eVec.Norm();
        return p;
    }

    private static double TransferCost(OrbitPoint a, OrbitPoint b, double mu)
    {
        double r1 = Math.Max(a.Radius, EarthRadiusKm + 150.0);
        double r2 = Math.Max(b.Radius, EarthRadiusKm + 150.0);
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double transferA = 0.5 * (r1 + r2);
        double vPeri = Math.Sqrt(mu * (2.0 / r1 - 1.0 / transferA));
        double vApo = Math.Sqrt(mu * (2.0 / r2 - 1.0 / transferA));
        double hohmann = Math.Abs(vPeri - v1) + Math.Abs(v2 - vApo);

        double planeCos = a.PlaneNormal.Dot(b.PlaneNormal);
        planeCos = Math.Max(-1.0, Math.Min(1.0, planeCos));
        double planeAngle = Math.Acos(planeCos);
        double planeChange = 2.0 * Math.Min(a.Speed, b.Speed) * Math.Sin(planeAngle * 0.5);

        double eccPenalty = 0.15 * Math.Abs(a.Eccentricity - b.Eccentricity) * (a.Speed + b.Speed);
        double shapePenalty = 0.05 * Math.Abs(a.SemiMajorAxis - b.SemiMajorAxis) / Math.Max(1.0, Math.Min(a.SemiMajorAxis, b.SemiMajorAxis)) * (a.Speed + b.Speed);
        double phasePenalty = 0.02 * (a.R - b.R).Norm();

        return hohmann + planeChange + eccPenalty + shapePenalty + phasePenalty;
    }

    private static int[] SolveExact(int n, double[] startCosts, double[,] pairCosts)
    {
        int size = 1 << n;
        double[,] dp = new double[size, n];
        int[,] parent = new int[size, n];
        for (int mask = 0; mask < size; mask++)
        {
            for (int i = 0; i < n; i++)
            {
                dp[mask, i] = double.PositiveInfinity;
                parent[mask, i] = -1;
            }
        }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask, i] = startCosts[i];
        }

        for (int mask = 1; mask < size; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (double.IsInfinity(cur)) continue;
                for (int nxt = 0; nxt < n; nxt++)
                {
                    if ((mask & (1 << nxt)) != 0) continue;
                    int nextMask = mask | (1 << nxt);
                    double cand = cur + pairCosts[last, nxt];
                    if (cand < dp[nextMask, nxt])
                    {
                        dp[nextMask, nxt] = cand;
                        parent[nextMask, nxt] = last;
                    }
                }
            }
        }

        int full = size - 1;
        double best = double.PositiveInfinity;
        int bestLast = 0;
        for (int i = 0; i < n; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                bestLast = i;
            }
        }

        var route = new int[n];
        int mask2 = full;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int prev = parent[mask2, curLast];
            mask2 ^= 1 << curLast;
            curLast = prev;
        }
        return route;
    }

    private sealed class BeamState
    {
        public int[] Route = Array.Empty<int>();
        public bool[] Used = Array.Empty<bool>();
        public int Last;
        public int Length;
        public double Cost;
    }

    private static int[] SolveBeam(int n, double[] startCosts, double[,] pairCosts, int beamWidth)
    {
        var initialCandidates = Enumerable.Range(0, n)
            .OrderBy(i => startCosts[i])
            .Take(Math.Min(beamWidth, n))
            .Select(i =>
            {
                var used = new bool[n];
                used[i] = true;
                return new BeamState
                {
                    Route = new[] { i },
                    Used = used,
                    Last = i,
                    Length = 1,
                    Cost = startCosts[i]
                };
            })
            .ToList();

        var beam = initialCandidates;
        while (beam[0].Length < n)
        {
            var nextBeam = new List<BeamState>(beamWidth * n);
            foreach (var state in beam)
            {
                var candidates = Enumerable.Range(0, n)
                    .Where(i => !state.Used[i])
                    .OrderBy(i => pairCosts[state.Last, i])
                    .Take(Math.Min(8, n - state.Length));

                foreach (var nxt in candidates)
                {
                    var route = new int[state.Length + 1];
                    Array.Copy(state.Route, route, state.Length);
                    route[state.Length] = nxt;
                    var used = new bool[n];
                    Array.Copy(state.Used, used, n);
                    used[nxt] = true;
                    nextBeam.Add(new BeamState
                    {
                        Route = route,
                        Used = used,
                        Last = nxt,
                        Length = state.Length + 1,
                        Cost = state.Cost + pairCosts[state.Last, nxt]
                    });
                }
            }

            beam = nextBeam
                .OrderBy(s => s.Cost + HeuristicTailEstimate(s, n, startCosts, pairCosts))
                .Take(beamWidth)
                .ToList();
        }

        return beam.OrderBy(s => s.Cost).First().Route;
    }

    private static double HeuristicTailEstimate(BeamState state, int n, double[] startCosts, double[,] pairCosts)
    {
        double sum = 0;
        for (int i = 0; i < n; i++)
        {
            if (!state.Used[i])
            {
                double best = pairCosts[state.Last, i];
                for (int j = 0; j < n; j++)
                {
                    if (i != j && !state.Used[j])
                    {
                        best = Math.Min(best, pairCosts[i, j]);
                    }
                }
                sum += best;
            }
        }
        return 0.35 * sum;
    }

    private static int[] SolveGreedyMultiStart(int n, double[] startCosts, double[,] pairCosts)
    {
        var seeds = Enumerable.Range(0, n)
            .OrderBy(i => startCosts[i])
            .Take(Math.Min(12, n))
            .ToArray();

        double bestCost = double.PositiveInfinity;
        int[] bestRoute = seeds.Length > 0 ? new[] { seeds[0] } : Array.Empty<int>();

        foreach (var seed in seeds)
        {
            var route = new int[n];
            var used = new bool[n];
            route[0] = seed;
            used[seed] = true;
            double total = startCosts[seed];
            int last = seed;

            for (int pos = 1; pos < n; pos++)
            {
                int next = -1;
                double nextScore = double.PositiveInfinity;
                for (int cand = 0; cand < n; cand++)
                {
                    if (used[cand]) continue;
                    double nearestUnused = double.PositiveInfinity;
                    for (int other = 0; other < n; other++)
                    {
                        if (cand == other || used[other]) continue;
                        nearestUnused = Math.Min(nearestUnused, pairCosts[cand, other]);
                    }
                    if (double.IsPositiveInfinity(nearestUnused)) nearestUnused = 0;
                    double score = pairCosts[last, cand] + 0.2 * nearestUnused;
                    if (score < nextScore)
                    {
                        nextScore = score;
                        next = cand;
                    }
                }
                route[pos] = next;
                used[next] = true;
                total += pairCosts[last, next];
                last = next;
            }

            if (total < bestCost)
            {
                bestCost = total;
                bestRoute = route;
            }
        }

        return bestRoute;
    }

    private static void ImproveRoute(int[] route, double[] startCosts, double[,] pairCosts)
    {
        int n = route.Length;
        bool improved = true;
        int passes = 0;
        while (improved && passes < 6)
        {
            improved = false;
            passes++;

            for (int i = 0; i < n - 1; i++)
            {
                for (int k = i + 1; k < n; k++)
                {
                    double before = SegmentCost(route, startCosts, pairCosts, i, k, reversed: false);
                    double after = SegmentCost(route, startCosts, pairCosts, i, k, reversed: true);
                    if (after + 1e-9 < before)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        improved = true;
                    }
                }
            }

            for (int i = 0; i < n; i++)
            {
                int node = route[i];
                int bestPos = i;
                double bestGain = 0;
                for (int pos = 0; pos < n; pos++)
                {
                    if (pos == i) continue;
                    double gain = RelocateGain(route, startCosts, pairCosts, i, pos);
                    if (gain < bestGain)
                    {
                        bestGain = gain;
                        bestPos = pos;
                    }
                }
                if (bestPos != i)
                {
                    var list = route.ToList();
                    list.RemoveAt(i);
                    list.Insert(bestPos > i ? bestPos - 1 : bestPos, node);
                    for (int t = 0; t < n; t++) route[t] = list[t];
                    improved = true;
                }
            }
        }
    }

    private static double RouteCost(int[] route, double[] startCosts, double[,] pairCosts)
    {
        double total = startCosts[route[0]];
        for (int i = 1; i < route.Length; i++)
        {
            total += pairCosts[route[i - 1], route[i]];
        }
        return total;
    }

    private static double SegmentCost(int[] route, double[] startCosts, double[,] pairCosts, int i, int k, bool reversed)
    {
        int n = route.Length;
        double total = 0;

        int first = reversed ? route[k] : route[i];
        if (i == 0)
        {
            total += startCosts[first];
        }
        else
        {
            total += pairCosts[route[i - 1], first];
        }

        for (int p = i; p < k; p++)
        {
            int a = reversed ? route[k - (p - i)] : route[p];
            int b = reversed ? route[k - (p - i) - 1] : route[p + 1];
            total += pairCosts[a, b];
        }

        if (k < n - 1)
        {
            int last = reversed ? route[i] : route[k];
            total += pairCosts[last, route[k + 1]];
        }

        return total;
    }

    private static double RelocateGain(int[] route, double[] startCosts, double[,] pairCosts, int from, int to)
    {
        double before = RouteCost(route, startCosts, pairCosts);
        var list = route.ToList();
        int node = list[from];
        list.RemoveAt(from);
        list.Insert(to > from ? to - 1 : to, node);
        double after = RouteCost(list.ToArray(), startCosts, pairCosts);
        return after - before;
    }
}
