using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double EarthRadius = 6371.0;
    private const double TinyEpsilon = 1e-9;

    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        double mu = scanner.NextDouble();

        State start = ReadState(scanner);
        var stations = new State[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = ReadState(scanner);
        }

        var planner = new OrbitalPlanner(mu, start, stations);
        int[] order = planner.Solve();
        Console.WriteLine(string.Join(" ", order));
    }

    private static State ReadState(FastScanner scanner)
    {
        return new State(
            new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble()),
            new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble()));
    }

    private readonly struct State
    {
        public readonly Vec3 Position;
        public readonly Vec3 Velocity;

        public State(Vec3 position, Vec3 velocity)
        {
            Position = position;
            Velocity = velocity;
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);

        public double Dot(Vec3 other) => X * other.X + Y * other.Y + Z * other.Z;

        public Vec3 Cross(Vec3 other)
        {
            return new Vec3(
                Y * other.Z - Z * other.Y,
                Z * other.X - X * other.Z,
                X * other.Y - Y * other.X);
        }

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public Vec3 Normalize()
        {
            double norm = Norm();
            if (norm < TinyEpsilon) return new Vec3(0.0, 0.0, 0.0);
            return this / norm;
        }
    }

    private sealed class OrbitalPlanner
    {
        private readonly double _mu;
        private readonly State _start;
        private readonly State[] _stations;
        private readonly OrbitalFeatures _startFeatures;
        private readonly OrbitalFeatures[] _stationFeatures;
        private readonly double[,] _cost;

        public OrbitalPlanner(double mu, State start, State[] stations)
        {
            _mu = mu;
            _start = start;
            _stations = stations;
            _startFeatures = OrbitalFeatures.FromState(start, mu);
            _stationFeatures = stations.Select(s => OrbitalFeatures.FromState(s, mu)).ToArray();
            _cost = BuildCostMatrix();
        }

        public int[] Solve()
        {
            int n = _stations.Length;
            if (n <= 1)
            {
                return n == 0 ? Array.Empty<int>() : new[] { 0 };
            }

            if (n <= 8)
            {
                return SolveExact();
            }

            int[] best = BuildGreedySeed();
            best = ImproveWithTwoOpt(best, maxPasses: n <= 20 ? 12 : 6);

            if (n <= 14)
            {
                best = ImproveWithRelocate(best, maxPasses: 6);
            }
            else
            {
                int[] alt = BuildFeatureSortedSeed();
                alt = ImproveWithTwoOpt(alt, maxPasses: 4);
                if (RouteCost(alt) < RouteCost(best))
                {
                    best = alt;
                }
            }

            if (n >= 25)
            {
                best = ImproveBySegmentReinsertion(best, samples: Math.Min(96, n * 2));
            }

            return best;
        }

        private int[] SolveExact()
        {
            int n = _stations.Length;
            var used = new bool[n];
            var current = new int[n];
            var best = new int[n];
            double bestCost = double.PositiveInfinity;

            void Dfs(int depth, int last, double acc)
            {
                if (acc >= bestCost)
                {
                    return;
                }

                if (depth == n)
                {
                    bestCost = acc;
                    Array.Copy(current, best, n);
                    return;
                }

                for (int i = 0; i < n; i++)
                {
                    if (used[i]) continue;
                    used[i] = true;
                    current[depth] = i;
                    double stepCost = depth == 0 ? StartCost(i) : _cost[last, i];
                    Dfs(depth + 1, i, acc + stepCost);
                    used[i] = false;
                }
            }

            Dfs(0, -1, 0.0);
            return best;
        }

        private int[] BuildGreedySeed()
        {
            int n = _stations.Length;
            var used = new bool[n];
            var route = new int[n];

            int current = FindBestStartNode();
            route[0] = current;
            used[current] = true;

            for (int pos = 1; pos < n; pos++)
            {
                int next = -1;
                double best = double.PositiveInfinity;
                for (int candidate = 0; candidate < n; candidate++)
                {
                    if (used[candidate]) continue;
                    double score = _cost[current, candidate];
                    score += 0.03 * Math.Abs(_stationFeatures[current].MeanMotion - _stationFeatures[candidate].MeanMotion);
                    if (score < best)
                    {
                        best = score;
                        next = candidate;
                    }
                }

                route[pos] = next;
                used[next] = true;
                current = next;
            }

            return route;
        }

        private int[] BuildFeatureSortedSeed()
        {
            int n = _stations.Length;
            var keys = new (double Key, int Idx)[n];
            for (int i = 0; i < n; i++)
            {
                var f = _stationFeatures[i];
                keys[i] = (f.CompositeOrderKey, i);
            }

            Array.Sort(keys, (a, b) => a.Key.CompareTo(b.Key));
            return keys.Select(k => k.Idx).ToArray();
        }

        private int FindBestStartNode()
        {
            int n = _stations.Length;
            int bestIndex = 0;
            double bestScore = double.PositiveInfinity;

            for (int i = 0; i < n; i++)
            {
                double score = StartCost(i);
                score += 0.2 * Math.Abs(_startFeatures.Inclination - _stationFeatures[i].Inclination);
                score += 0.2 * Math.Abs(WrapAngle(_startFeatures.RightAscension - _stationFeatures[i].RightAscension));
                if (score < bestScore)
                {
                    bestScore = score;
                    bestIndex = i;
                }
            }

            return bestIndex;
        }

        private int[] ImproveWithTwoOpt(int[] route, int maxPasses)
        {
            int n = route.Length;
            if (n < 4) return route;

            var best = (int[])route.Clone();
            for (int pass = 0; pass < maxPasses; pass++)
            {
                bool improved = false;
                for (int i = 0; i < n - 2; i++)
                {
                    for (int j = i + 1; j < n - 1; j++)
                    {
                        double before = EdgeCost(best, i - 1, i) + EdgeCost(best, j, j + 1);
                        double after = EdgeCostSwap(best, i - 1, j) + EdgeCostSwap(best, i, j + 1);
                        if (after + 1e-8 < before)
                        {
                            Array.Reverse(best, i, j - i + 1);
                            improved = true;
                        }
                    }
                }

                if (!improved) break;
            }

            return best;
        }

        private int[] ImproveWithRelocate(int[] route, int maxPasses)
        {
            int n = route.Length;
            var best = (int[])route.Clone();
            for (int pass = 0; pass < maxPasses; pass++)
            {
                bool improved = false;
                for (int i = 0; i < n; i++)
                {
                    for (int j = 0; j < n; j++)
                    {
                        if (i == j || i + 1 == j) continue;
                        int moved = best[i];
                        var candidate = new List<int>(best);
                        candidate.RemoveAt(i);
                        int insertAt = j > i ? j - 1 : j;
                        candidate.Insert(insertAt, moved);
                        double candCost = RouteCost(candidate);
                        double bestCost = RouteCost(best);
                        if (candCost + 1e-8 < bestCost)
                        {
                            best = candidate.ToArray();
                            improved = true;
                        }
                    }
                }

                if (!improved) break;
            }

            return best;
        }

        private int[] ImproveBySegmentReinsertion(int[] route, int samples)
        {
            int n = route.Length;
            var best = (int[])route.Clone();
            double bestCost = RouteCost(best);
            int segmentLength = Math.Max(2, n / 12);
            if (segmentLength >= n) return best;

            for (int s = 0; s < samples; s++)
            {
                int start = (s * 7919) % (n - segmentLength + 1);
                int insert = (s * 1543 + 7) % (n - segmentLength + 1);
                if (insert >= start && insert <= start + segmentLength) continue;

                var candidate = new List<int>(n);
                for (int i = 0; i < start; i++) candidate.Add(best[i]);
                for (int i = start + segmentLength; i < n; i++) candidate.Add(best[i]);

                var seg = best.Skip(start).Take(segmentLength).ToArray();
                int actualInsert = Math.Min(insert, candidate.Count);
                candidate.InsertRange(actualInsert, seg);

                double candCost = RouteCost(candidate);
                if (candCost + 1e-8 < bestCost)
                {
                    best = candidate.ToArray();
                    bestCost = candCost;
                }
            }

            return best;
        }

        private double RouteCost(IReadOnlyList<int> route)
        {
            if (route.Count == 0) return 0.0;
            double total = StartCost(route[0]);
            for (int i = 1; i < route.Count; i++)
            {
                total += _cost[route[i - 1], route[i]];
            }
            return total;
        }

        private double EdgeCost(int[] route, int aPos, int bPos)
        {
            if (bPos < 0 || bPos >= route.Length) return 0.0;
            int b = route[bPos];
            if (aPos < 0) return StartCost(b);
            return _cost[route[aPos], b];
        }

        private double EdgeCostSwap(int[] route, int aPos, int bPos)
        {
            if (bPos < 0 || bPos >= route.Length) return 0.0;
            int b = route[bPos];
            if (aPos < 0) return StartCost(b);
            return _cost[route[aPos], b];
        }

        private double StartCost(int stationIndex)
        {
            return EstimateTransferCost(_startFeatures, _stationFeatures[stationIndex], true);
        }

        private double[,] BuildCostMatrix()
        {
            int n = _stations.Length;
            var matrix = new double[n, n];
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    matrix[i, j] = i == j ? 0.0 : EstimateTransferCost(_stationFeatures[i], _stationFeatures[j], false);
                }
            }

            return matrix;
        }

        private double EstimateTransferCost(OrbitalFeatures a, OrbitalFeatures b, bool isStart)
        {
            double radiusA = a.Radius;
            double radiusB = b.Radius;
            double circularA = Math.Sqrt(_mu / Math.Max(radiusA, EarthRadius + 120.0));
            double circularB = Math.Sqrt(_mu / Math.Max(radiusB, EarthRadius + 120.0));

            double hohmann1 = circularA * Math.Abs(Math.Sqrt(2.0 * radiusB / (radiusA + radiusB)) - 1.0);
            double hohmann2 = circularB * Math.Abs(1.0 - Math.Sqrt(2.0 * radiusA / (radiusA + radiusB)));
            double radialCost = hohmann1 + hohmann2;

            double planeAngle = PlaneAngle(a.PlaneNormal, b.PlaneNormal);
            double planeSpeed = 0.5 * (circularA + circularB);
            double planeCost = 2.0 * planeSpeed * Math.Sin(planeAngle * 0.5);

            double phaseAngle = Math.Abs(WrapAngle(a.TrueLongitude - b.TrueLongitude));
            double phaseCost = 0.12 * planeSpeed * phaseAngle;

            double eccCost = 0.35 * planeSpeed * Math.Abs(a.Eccentricity - b.Eccentricity);
            double shapeCost = 0.06 * planeSpeed * Math.Abs(a.ArgumentOfLatitude - b.ArgumentOfLatitude);
            double motionCost = 180.0 * Math.Abs(a.MeanMotion - b.MeanMotion);

            double total = radialCost + planeCost + phaseCost + eccCost + shapeCost + motionCost;

            if (isStart)
            {
                total *= 0.97;
            }

            return total;
        }

        private static double PlaneAngle(Vec3 a, Vec3 b)
        {
            double denom = a.Norm() * b.Norm();
            if (denom < TinyEpsilon) return 0.0;
            double c = Clamp(a.Dot(b) / denom, -1.0, 1.0);
            return Math.Acos(c);
        }

        private static double Clamp(double value, double lo, double hi)
        {
            if (value < lo) return lo;
            if (value > hi) return hi;
            return value;
        }

        private static double WrapAngle(double angle)
        {
            while (angle <= -Math.PI) angle += 2.0 * Math.PI;
            while (angle > Math.PI) angle -= 2.0 * Math.PI;
            return angle;
        }
    }

    private readonly struct OrbitalFeatures
    {
        public readonly double Radius;
        public readonly double Inclination;
        public readonly double RightAscension;
        public readonly double Eccentricity;
        public readonly double TrueLongitude;
        public readonly double ArgumentOfLatitude;
        public readonly double MeanMotion;
        public readonly Vec3 PlaneNormal;
        public readonly double CompositeOrderKey;

        public OrbitalFeatures(
            double radius,
            double inclination,
            double rightAscension,
            double eccentricity,
            double trueLongitude,
            double argumentOfLatitude,
            double meanMotion,
            Vec3 planeNormal,
            double compositeOrderKey)
        {
            Radius = radius;
            Inclination = inclination;
            RightAscension = rightAscension;
            Eccentricity = eccentricity;
            TrueLongitude = trueLongitude;
            ArgumentOfLatitude = argumentOfLatitude;
            MeanMotion = meanMotion;
            PlaneNormal = planeNormal;
            CompositeOrderKey = compositeOrderKey;
        }

        public static OrbitalFeatures FromState(State state, double mu)
        {
            Vec3 r = state.Position;
            Vec3 v = state.Velocity;
            double radius = r.Norm();
            double speed2 = v.Dot(v);
            Vec3 h = r.Cross(v);
            double hNorm = h.Norm();
            Vec3 n = new Vec3(-h.Y, h.X, 0.0);
            double nNorm = n.Norm();
            Vec3 eVec = (v.Cross(h) / mu) - (r / Math.Max(radius, TinyEpsilon));
            double e = eVec.Norm();
            double inclination = Math.Acos(Clamp(h.Z / Math.Max(hNorm, TinyEpsilon), -1.0, 1.0));

            double raan = 0.0;
            if (nNorm > TinyEpsilon)
            {
                raan = Math.Atan2(n.Y, n.X);
            }

            Vec3 p = nNorm > TinyEpsilon ? n / nNorm : new Vec3(1.0, 0.0, 0.0);
            Vec3 q = hNorm > TinyEpsilon ? h.Cross(p).Normalize() : new Vec3(0.0, 1.0, 0.0);
            double argLat = Math.Atan2(r.Dot(q), r.Dot(p));
            double lon = Math.Atan2(r.Y, r.X);

            double aInv = 2.0 / Math.Max(radius, TinyEpsilon) - speed2 / mu;
            double meanMotion = aInv > TinyEpsilon ? Math.Sqrt(mu * aInv * aInv * aInv) : 0.0;

            double composite =
                2.1 * radius +
                900.0 * inclination +
                500.0 * raan +
                300.0 * argLat +
                1200.0 * e;

            return new OrbitalFeatures(
                radius,
                inclination,
                raan,
                e,
                lon,
                argLat,
                meanMotion,
                hNorm > TinyEpsilon ? h / hNorm : new Vec3(0.0, 0.0, 1.0),
                composite);
        }

        private static double Clamp(double value, double lo, double hi)
        {
            if (value < lo) return lo;
            if (value > hi) return hi;
            return value;
        }
    }

    private sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _index;

        public FastScanner()
        {
            _tokens = Console.In
                .ReadToEnd()
                .Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            _index = 0;
        }

        public int NextInt() => int.Parse(_tokens[_index++], CultureInfo.InvariantCulture);

        public double NextDouble() => double.Parse(_tokens[_index++], CultureInfo.InvariantCulture);
    }
}
