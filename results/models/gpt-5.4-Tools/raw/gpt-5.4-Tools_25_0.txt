using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    const double G = 6.67430e-11;
    const double SecondsPerDay = 86400.0;
    const double EarthRadius = 6_371_000.0;
    const double SafeMissDistance = EarthRadius + 1_000_000.0;
    const double InterceptTolerance = 1_000_000.0;
    const double MomentumTransferFactor = 2.0;

    public static void Main()
    {
        var scanner = new FastScanner(Console.In);

        int numBodies = scanner.NextInt();
        double warningDays = scanner.NextDouble();
        double impactorMass = scanner.NextDouble();
        double deltaVBudget = scanner.NextDouble();
        double asteroidMass = scanner.NextDouble();
        double t0UnixUtc = scanner.NextDouble();

        var bodies = new List<Body>();
        Body earth = null!;
        Body asteroid = null!;

        for (int i = 0; i < numBodies; i++)
        {
            string name = scanner.Next();
            double mass = scanner.NextDouble();
            var r = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
            var v = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
            var body = new Body(name, mass, r, v);
            bodies.Add(body);
            if (string.Equals(name, "Earth", StringComparison.OrdinalIgnoreCase)) earth = body;
            if (string.Equals(name, "Asteroid", StringComparison.OrdinalIgnoreCase)) asteroid = body;
        }

        string bspPath = scanner.Next();
        _ = bspPath;
        _ = t0UnixUtc;

        if (earth == null || asteroid == null)
        {
            EmitFallbackPlan(warningDays, deltaVBudget);
            return;
        }

        var solver = new Solver(bodies, earth, asteroid, impactorMass, asteroidMass, warningDays, deltaVBudget);
        var plan = solver.Solve();
        WritePlan(plan);
    }

    static void WritePlan(List<Burn> burns)
    {
        Console.WriteLine(burns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var burn in burns)
        {
            Console.WriteLine(string.Format(
                CultureInfo.InvariantCulture,
                "{0:R} {1:R} {2:R} {3:R}",
                burn.TimeDays,
                burn.DeltaV.X,
                burn.DeltaV.Y,
                burn.DeltaV.Z));
        }
    }

    static void EmitFallbackPlan(double warningDays, double deltaVBudget)
    {
        double launch = Math.Max(0.0, Math.Min(warningDays, 0.05 * Math.Max(1.0, warningDays)));
        var dv = new Vec3(Math.Min(deltaVBudget, 1.0), 0, 0);
        WritePlan(new List<Burn> { new Burn(launch, dv) });
    }

    sealed class Solver
    {
        readonly List<Body> _bodies;
        readonly Body _earth;
        readonly Body _asteroid;
        readonly double _impactorMass;
        readonly double _asteroidMass;
        readonly double _warningDays;
        readonly double _deltaVBudget;

        public Solver(List<Body> bodies, Body earth, Body asteroid, double impactorMass, double asteroidMass, double warningDays, double deltaVBudget)
        {
            _bodies = bodies;
            _earth = earth;
            _asteroid = asteroid;
            _impactorMass = Math.Max(1.0, impactorMass);
            _asteroidMass = Math.Max(1.0, asteroidMass);
            _warningDays = Math.Max(0.01, warningDays);
            _deltaVBudget = Math.Max(0.0, deltaVBudget);
        }

        public List<Burn> Solve()
        {
            var nominal = BuildEphemeris(_warningDays, includeSpacecraft: false, 1.0);
            double impactDay = FindClosestApproachDay(nominal.EarthPositions, nominal.AsteroidPositions);
            impactDay = Math.Max(1.0 / 24.0, Math.Min(_warningDays, impactDay));

            var candidateLaunches = BuildLaunchCandidates(impactDay);
            var candidateFractions = new[] { 0.35, 0.5, 0.65, 0.8, 1.0 };

            Candidate? best = null;
            foreach (double launchDay in candidateLaunches)
            {
                double tofDays = impactDay - launchDay;
                if (tofDays <= 0.05) continue;

                Vec3 earthPos = Sample(nominal.EarthPositions, launchDay, nominal.StepDays);
                Vec3 earthVel = Sample(nominal.EarthVelocities, launchDay, nominal.StepDays);
                Vec3 asteroidPos = Sample(nominal.AsteroidPositions, impactDay, nominal.StepDays);
                Vec3 asteroidVel = Sample(nominal.AsteroidVelocities, impactDay, nominal.StepDays);

                Vec3 neededInitialVelocity = (asteroidPos - earthPos) / (tofDays * SecondsPerDay);
                Vec3 deltaV0 = neededInitialVelocity - earthVel;
                double dv0Mag = deltaV0.Length();
                if (dv0Mag > _deltaVBudget * 1.05) continue;

                foreach (double frac in candidateFractions)
                {
                    double correctionDay = launchDay + tofDays * frac;
                    if (correctionDay >= impactDay) continue;

                    var mission = EvaluateCandidate(nominal, launchDay, correctionDay, impactDay, deltaV0);
                    if (mission == null) continue;
                    if (best == null || mission.Score > best.Score) best = mission;
                }
            }

            if (best == null)
            {
                return BuildDirectFallback(nominal, impactDay);
            }

            var burns = new List<Burn> { new Burn(best.LaunchDay, best.LaunchDeltaV) };
            if (best.CorrectionDeltaV.Length() > 1e-6)
            {
                burns.Add(new Burn(best.CorrectionDay, best.CorrectionDeltaV));
            }

            // Final line is the impact event. The delta-v entry is informational and set to zero.
            burns.Add(new Burn(best.InterceptDay, Vec3.Zero));
            return burns;
        }

        List<Burn> BuildDirectFallback(Ephemeris nominal, double impactDay)
        {
            double launchDay = Math.Max(0.0, Math.Min(_warningDays * 0.25, impactDay * 0.2));
            double tofDays = Math.Max(0.1, impactDay - launchDay);
            Vec3 earthPos = Sample(nominal.EarthPositions, launchDay, nominal.StepDays);
            Vec3 earthVel = Sample(nominal.EarthVelocities, launchDay, nominal.StepDays);
            Vec3 asteroidPos = Sample(nominal.AsteroidPositions, impactDay, nominal.StepDays);
            Vec3 deltaV = (asteroidPos - earthPos) / (tofDays * SecondsPerDay) - earthVel;
            double mag = deltaV.Length();
            if (mag > _deltaVBudget && mag > 0)
            {
                deltaV *= (_deltaVBudget / mag);
            }

            return new List<Burn>
            {
                new Burn(launchDay, deltaV),
                new Burn(impactDay, Vec3.Zero)
            };
        }

        Candidate? EvaluateCandidate(Ephemeris nominal, double launchDay, double correctionDay, double interceptDay, Vec3 launchDeltaV)
        {
            double used = launchDeltaV.Length();
            if (used > _deltaVBudget) return null;

            var partial = BuildSpacecraftTrajectory(launchDay, correctionDay, interceptDay, launchDeltaV, null, nominal.StepDays);
            Vec3 craftPosAtCorrection = partial.SpacecraftAtCorrectionPos;
            Vec3 craftVelAtCorrection = partial.SpacecraftAtCorrectionVel;

            Vec3 asteroidPosAtIntercept = Sample(nominal.AsteroidPositions, interceptDay, nominal.StepDays);
            Vec3 neededPostCorrectionVelocity = (asteroidPosAtIntercept - craftPosAtCorrection) / ((interceptDay - correctionDay) * SecondsPerDay);
            Vec3 correctionDeltaV = neededPostCorrectionVelocity - craftVelAtCorrection;
            double correctionMag = correctionDeltaV.Length();
            if (used + correctionMag > _deltaVBudget) return null;

            var full = BuildSpacecraftTrajectory(launchDay, correctionDay, interceptDay, launchDeltaV, correctionDeltaV, nominal.StepDays);
            double interceptDistance = (full.SpacecraftAtInterceptPos - asteroidPosAtIntercept).Length();
            if (interceptDistance > InterceptTolerance) return null;

            Vec3 asteroidVelAtIntercept = Sample(nominal.AsteroidVelocities, interceptDay, nominal.StepDays);
            Vec3 relativeVelocity = full.SpacecraftAtInterceptVel - asteroidVelAtIntercept;
            double relSpeed = relativeVelocity.Length();
            if (relSpeed < 1e-3) return null;

            Vec3 asteroidDeltaV = relativeVelocity * (MomentumTransferFactor * _impactorMass / _asteroidMass);
            double remainingTime = Math.Max(0.0, _warningDays - interceptDay) * SecondsPerDay;

            Vec3 earthPosEnd = Sample(nominal.EarthPositions, _warningDays, nominal.StepDays);
            Vec3 asteroidPosEnd = Sample(nominal.AsteroidPositions, _warningDays, nominal.StepDays);
            Vec3 asteroidVelEnd = Sample(nominal.AsteroidVelocities, _warningDays, nominal.StepDays);

            double alongTrackBenefit = asteroidDeltaV.Length() * remainingTime;
            Vec3 deflectedEnd = asteroidPosEnd + asteroidDeltaV * remainingTime;
            double nominalMiss = (asteroidPosEnd - earthPosEnd).Length();
            double deflectedMiss = (deflectedEnd - earthPosEnd).Length();

            double score = 0.0;
            score += Math.Min(1e12, alongTrackBenefit);
            score += 10.0 * (deflectedMiss - nominalMiss);
            score -= 1000.0 * (used + correctionMag);
            score -= interceptDistance * 0.01;
            if (deflectedMiss >= SafeMissDistance) score += 1e9;

            return new Candidate
            {
                LaunchDay = launchDay,
                CorrectionDay = correctionDay,
                InterceptDay = interceptDay,
                LaunchDeltaV = launchDeltaV,
                CorrectionDeltaV = correctionDeltaV,
                Score = score
            };
        }

        double[] BuildLaunchCandidates(double impactDay)
        {
            var values = new SortedSet<double>();
            int coarse = 12;
            for (int i = 0; i <= coarse; i++)
            {
                double t = _warningDays * i / coarse;
                if (t < impactDay - 0.05) values.Add(t);
            }

            double[] fractions = { 0.02, 0.05, 0.1, 0.2, 0.35, 0.5, 0.65, 0.8 };
            foreach (double f in fractions)
            {
                double t = impactDay * f;
                if (t >= 0 && t < impactDay - 0.05) values.Add(t);
            }

            if (values.Count == 0) values.Add(0.0);
            return values.ToArray();
        }

        Ephemeris BuildEphemeris(double durationDays, bool includeSpacecraft, double stepHintDays)
        {
            double stepDays = ChooseStep(durationDays, stepHintDays);
            int steps = Math.Max(2, (int)Math.Ceiling(durationDays / stepDays) + 2);

            var states = new State[_bodies.Count];
            for (int i = 0; i < _bodies.Count; i++)
            {
                states[i] = new State(_bodies[i].Position, _bodies[i].Velocity);
            }

            var earthPositions = new Vec3[steps];
            var earthVelocities = new Vec3[steps];
            var asteroidPositions = new Vec3[steps];
            var asteroidVelocities = new Vec3[steps];

            int earthIndex = _bodies.IndexOf(_earth);
            int asteroidIndex = _bodies.IndexOf(_asteroid);
            double dt = stepDays * SecondsPerDay;

            for (int i = 0; i < steps; i++)
            {
                earthPositions[i] = states[earthIndex].Pos;
                earthVelocities[i] = states[earthIndex].Vel;
                asteroidPositions[i] = states[asteroidIndex].Pos;
                asteroidVelocities[i] = states[asteroidIndex].Vel;
                if (i + 1 < steps) StepBodies(states, dt);
            }

            return new Ephemeris
            {
                StepDays = stepDays,
                EarthPositions = earthPositions,
                EarthVelocities = earthVelocities,
                AsteroidPositions = asteroidPositions,
                AsteroidVelocities = asteroidVelocities
            };
        }

        TrajectoryResult BuildSpacecraftTrajectory(double launchDay, double correctionDay, double interceptDay, Vec3 launchDeltaV, Vec3? correctionDeltaV, double stepHintDays)
        {
            double durationDays = interceptDay;
            double stepDays = ChooseStep(durationDays, stepHintDays);
            double dt = stepDays * SecondsPerDay;
            int steps = Math.Max(2, (int)Math.Ceiling(durationDays / stepDays) + 2);

            var states = new State[_bodies.Count];
            for (int i = 0; i < _bodies.Count; i++)
            {
                states[i] = new State(_bodies[i].Position, _bodies[i].Velocity);
            }

            int earthIndex = _bodies.IndexOf(_earth);
            bool launched = false;
            bool corrected = false;
            var craft = new State(Vec3.Zero, Vec3.Zero);
            Vec3 corrPos = Vec3.Zero, corrVel = Vec3.Zero, intPos = Vec3.Zero, intVel = Vec3.Zero;

            double time = 0.0;
            for (int i = 0; i < steps; i++)
            {
                if (!launched && time + 1e-12 >= launchDay)
                {
                    craft = new State(states[earthIndex].Pos, states[earthIndex].Vel + launchDeltaV);
                    launched = true;
                }

                if (launched && !corrected && correctionDeltaV.HasValue && time + 1e-12 >= correctionDay)
                {
                    craft.Vel += correctionDeltaV.Value;
                    corrected = true;
                    corrPos = craft.Pos;
                    corrVel = craft.Vel;
                }
                else if (launched && !correctionDeltaV.HasValue && !corrected && time + 1e-12 >= correctionDay)
                {
                    corrected = true;
                    corrPos = craft.Pos;
                    corrVel = craft.Vel;
                }

                if (launched && time + 1e-12 >= interceptDay)
                {
                    intPos = craft.Pos;
                    intVel = craft.Vel;
                    break;
                }

                if (i + 1 >= steps) break;
                StepBodies(states, dt);
                if (launched) StepMassless(ref craft, states, dt);
                time += stepDays;
            }

            if (corrPos.LengthSquared() == 0 && launched)
            {
                corrPos = craft.Pos;
                corrVel = craft.Vel;
            }
            if (intPos.LengthSquared() == 0 && launched)
            {
                intPos = craft.Pos;
                intVel = craft.Vel;
            }

            return new TrajectoryResult
            {
                SpacecraftAtCorrectionPos = corrPos,
                SpacecraftAtCorrectionVel = corrVel,
                SpacecraftAtInterceptPos = intPos,
                SpacecraftAtInterceptVel = intVel
            };
        }

        void StepBodies(State[] states, double dt)
        {
            int n = states.Length;
            var a0 = ComputeAccelerations(states);
            for (int i = 0; i < n; i++)
            {
                states[i].Vel += a0[i] * (0.5 * dt);
                states[i].Pos += states[i].Vel * dt;
            }

            var a1 = ComputeAccelerations(states);
            for (int i = 0; i < n; i++)
            {
                states[i].Vel += a1[i] * (0.5 * dt);
            }
        }

        void StepMassless(ref State craft, State[] bodyStates, double dt)
        {
            Vec3 a0 = ComputeAccelerationForPoint(craft.Pos, bodyStates);
            craft.Vel += a0 * (0.5 * dt);
            craft.Pos += craft.Vel * dt;
            Vec3 a1 = ComputeAccelerationForPoint(craft.Pos, bodyStates);
            craft.Vel += a1 * (0.5 * dt);
        }

        Vec3[] ComputeAccelerations(State[] states)
        {
            int n = states.Length;
            var acc = new Vec3[n];
            for (int i = 0; i < n; i++)
            {
                Vec3 sum = Vec3.Zero;
                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    Vec3 d = states[j].Pos - states[i].Pos;
                    double r2 = d.LengthSquared() + 1e6;
                    double invR = 1.0 / Math.Sqrt(r2);
                    double invR3 = invR * invR * invR;
                    sum += d * (G * _bodies[j].Mass * invR3);
                }
                acc[i] = sum;
            }
            return acc;
        }

        Vec3 ComputeAccelerationForPoint(Vec3 pos, State[] bodyStates)
        {
            Vec3 sum = Vec3.Zero;
            for (int i = 0; i < bodyStates.Length; i++)
            {
                Vec3 d = bodyStates[i].Pos - pos;
                double r2 = d.LengthSquared() + 1e6;
                double invR = 1.0 / Math.Sqrt(r2);
                double invR3 = invR * invR * invR;
                sum += d * (G * _bodies[i].Mass * invR3);
            }
            return sum;
        }

        double FindClosestApproachDay(Vec3[] earthPositions, Vec3[] asteroidPositions)
        {
            double bestDay = _warningDays;
            double bestDist = double.MaxValue;
            for (int i = 0; i < earthPositions.Length; i++)
            {
                double dist = (earthPositions[i] - asteroidPositions[i]).Length();
                if (dist < bestDist)
                {
                    bestDist = dist;
                    bestDay = Math.Min(_warningDays, i * ChooseStep(_warningDays, 1.0));
                }
            }
            return bestDay;
        }

        static Vec3 Sample(Vec3[] series, double timeDays, double stepDays)
        {
            double x = timeDays / stepDays;
            int i = (int)Math.Floor(x);
            if (i <= 0) return series[0];
            if (i >= series.Length - 1) return series[series.Length - 1];
            double f = x - i;
            return series[i] * (1.0 - f) + series[i + 1] * f;
        }

        static double ChooseStep(double durationDays, double stepHintDays)
        {
            double step = Math.Min(stepHintDays, Math.Max(0.05, durationDays / 1500.0));
            if (durationDays > 3650) step = Math.Max(step, 2.0);
            else if (durationDays > 730) step = Math.Max(step, 0.5);
            else if (durationDays > 120) step = Math.Max(step, 0.2);
            return step;
        }
    }

    readonly record struct Body(string Name, double Mass, Vec3 Position, Vec3 Velocity);
    readonly record struct Burn(double TimeDays, Vec3 DeltaV);

    sealed class Candidate
    {
        public double LaunchDay;
        public double CorrectionDay;
        public double InterceptDay;
        public Vec3 LaunchDeltaV = Vec3.Zero;
        public Vec3 CorrectionDeltaV = Vec3.Zero;
        public double Score;
    }

    sealed class Ephemeris
    {
        public double StepDays;
        public Vec3[] EarthPositions = Array.Empty<Vec3>();
        public Vec3[] EarthVelocities = Array.Empty<Vec3>();
        public Vec3[] AsteroidPositions = Array.Empty<Vec3>();
        public Vec3[] AsteroidVelocities = Array.Empty<Vec3>();
    }

    sealed class TrajectoryResult
    {
        public Vec3 SpacecraftAtCorrectionPos;
        public Vec3 SpacecraftAtCorrectionVel;
        public Vec3 SpacecraftAtInterceptPos;
        public Vec3 SpacecraftAtInterceptVel;
    }

    struct State
    {
        public Vec3 Pos;
        public Vec3 Vel;

        public State(Vec3 pos, Vec3 vel)
        {
            Pos = pos;
            Vel = vel;
        }
    }

    readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public static readonly Vec3 Zero = new Vec3(0, 0, 0);

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double LengthSquared() => X * X + Y * Y + Z * Z;

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator *(double k, Vec3 a) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
    }

    sealed class FastScanner
    {
        readonly TextReader _reader;
        readonly Queue<string> _tokens = new Queue<string>();

        public FastScanner(TextReader reader)
        {
            _reader = reader;
        }

        public string Next()
        {
            while (_tokens.Count == 0)
            {
                string? line = _reader.ReadLine();
                if (line == null) throw new EndOfStreamException("Unexpected end of input.");
                foreach (var token in line.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries))
                {
                    _tokens.Enqueue(token);
                }
            }

            return _tokens.Dequeue();
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }
}
