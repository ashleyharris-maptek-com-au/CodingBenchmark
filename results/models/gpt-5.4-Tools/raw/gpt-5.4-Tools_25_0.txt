using System;
using System.Globalization;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    const double G = 6.67430e-11;
    const double Day = 86400.0;
    const double ImpactSafetyFactor = 1.05;

    sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 Pos;
        public Vec3 Vel;
    }

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double NormSq() => X * X + Y * Y + Z * Z;
        public Vec3 Normalized()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec3();
        }
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    sealed class Scenario
    {
        public int NumBodies;
        public double WarningDays;
        public double ImpactorMass;
        public double DeltaVBudget;
        public double AsteroidMass;
        public double T0UnixUtc;
        public List<Body> Bodies = new List<Body>();
        public string BspPath = "";
        public Body Earth = null!;
        public Body Asteroid = null!;
        public Body Sun;
    }

    struct State
    {
        public Vec3 Pos;
        public Vec3 Vel;
        public State(Vec3 pos, Vec3 vel) { Pos = pos; Vel = vel; }
    }

    public static void Main()
    {
        var sc = ReadScenario();
        var result = Solve(sc);
        Console.Write(result);
    }

    static Scenario ReadScenario()
    {
        string input = Console.In.ReadToEnd();
        var tokens = input.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        var ci = CultureInfo.InvariantCulture;

        var sc = new Scenario();
        sc.NumBodies = int.Parse(tokens[idx++], ci);
        sc.WarningDays = double.Parse(tokens[idx++], ci);
        sc.ImpactorMass = double.Parse(tokens[idx++], ci);
        sc.DeltaVBudget = double.Parse(tokens[idx++], ci);
        sc.AsteroidMass = double.Parse(tokens[idx++], ci);
        sc.T0UnixUtc = double.Parse(tokens[idx++], ci);

        for (int i = 0; i < sc.NumBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[idx++];
            b.Mass = double.Parse(tokens[idx++], ci);
            b.Pos = new Vec3(
                double.Parse(tokens[idx++], ci),
                double.Parse(tokens[idx++], ci),
                double.Parse(tokens[idx++], ci));
            b.Vel = new Vec3(
                double.Parse(tokens[idx++], ci),
                double.Parse(tokens[idx++], ci),
                double.Parse(tokens[idx++], ci));
            sc.Bodies.Add(b);
        }

        if (idx < tokens.Length)
            sc.BspPath = tokens[idx++];

        foreach (var b in sc.Bodies)
        {
            if (string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase)) sc.Earth = b;
            if (string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase)) sc.Asteroid = b;
            if (string.Equals(b.Name, "Sun", StringComparison.OrdinalIgnoreCase)) sc.Sun = b;
        }

        if (sc.Earth == null || sc.Asteroid == null)
            throw new InvalidOperationException("Scenario must contain Earth and Asteroid bodies.");

        return sc;
    }

    static string Solve(Scenario sc)
    {
        double impactTime = EstimateImpactTimeDays(sc);
        if (!(impactTime > 0)) impactTime = Math.Max(1.0, sc.WarningDays);
        impactTime = Math.Min(impactTime, Math.Max(sc.WarningDays, 1.0e6 / Day));

        double launchTime = Math.Max(0.0, Math.Min(sc.WarningDays, impactTime * 0.15));
        double tIntercept = Math.Max(launchTime + Math.Max(0.5, impactTime * 0.05), impactTime * 0.55);
        tIntercept = Math.Min(tIntercept, Math.Max(launchTime + 0.25, impactTime - 0.25));
        if (tIntercept <= launchTime) tIntercept = launchTime + 0.25;

        State earthLaunch = PropagateBody(sc, sc.Earth, launchTime * Day);
        State asteroidIntercept = PropagateBody(sc, sc.Asteroid, tIntercept * Day);

        Vec3 transferV = (asteroidIntercept.Pos - earthLaunch.Pos) / ((tIntercept - launchTime) * Day);
        Vec3 launchDv = transferV - earthLaunch.Vel;

        double launchDvMag = launchDv.Norm();
        if (launchDvMag > sc.DeltaVBudget && launchDvMag > 0)
        {
            launchDv *= sc.DeltaVBudget / launchDvMag;
            launchDvMag = sc.DeltaVBudget;
        }

        double remainingDv = Math.Max(0.0, sc.DeltaVBudget - launchDvMag);
        double correctionTime = launchTime + (tIntercept - launchTime) * 0.65;
        State craftAtCorrection = PropagateBallistic(earthLaunch.Pos, earthLaunch.Vel + launchDv, launchTime, correctionTime, sc);
        State asteroidAtCorrection = PropagateBody(sc, sc.Asteroid, correctionTime * Day);
        double remainingTimeSec = Math.Max(1.0, (tIntercept - correctionTime) * Day);
        Vec3 desiredVelAfterCorrection = (asteroidIntercept.Pos - craftAtCorrection.Pos) / remainingTimeSec;
        Vec3 correctionDv = desiredVelAfterCorrection - craftAtCorrection.Vel;
        double correctionMag = correctionDv.Norm();
        if (correctionMag > remainingDv && correctionMag > 0)
        {
            correctionDv *= remainingDv / correctionMag;
            correctionMag = remainingDv;
        }

        Vec3 asteroidVelAtIntercept = asteroidIntercept.Vel;
        Vec3 relEarthAtImpact = PropagateBody(sc, sc.Asteroid, impactTime * Day).Pos - PropagateBody(sc, sc.Earth, impactTime * Day).Pos;
        Vec3 missDirection = Vec3.Cross(relEarthAtImpact, asteroidVelAtIntercept);
        if (missDirection.Norm() < 1e-9)
            missDirection = Vec3.Cross(asteroidVelAtIntercept, new Vec3(0, 0, 1));
        if (missDirection.Norm() < 1e-9)
            missDirection = new Vec3(1, 0, 0);
        missDirection = missDirection.Normalized();

        double requiredAsteroidDv = EstimateRequiredAsteroidDeflection(sc, impactTime, relEarthAtImpact);
        double momentumScale = sc.ImpactorMass / Math.Max(sc.AsteroidMass, 1.0);
        double desiredImpactSpeed = requiredAsteroidDv / Math.Max(momentumScale, 1e-12);

        State craftAtIntercept = PropagateBallistic(earthLaunch.Pos, earthLaunch.Vel + launchDv, launchTime, tIntercept, sc, correctionTime, correctionDv);
        Vec3 currentImpactVel = craftAtIntercept.Vel;
        Vec3 desiredImpactVel = asteroidVelAtIntercept + missDirection * desiredImpactSpeed;
        Vec3 impactDv = desiredImpactVel - currentImpactVel;

        var burns = new List<(double t, Vec3 dv)>();
        burns.Add((launchTime, launchDv));
        if (correctionMag > 1e-6)
            burns.Add((correctionTime, correctionDv));
        burns.Add((tIntercept, impactDv));

        var sb = new StringBuilder();
        sb.AppendLine(burns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var burn in burns)
        {
            sb.Append(Format(burn.t)).Append(' ')
              .Append(Format(burn.dv.X)).Append(' ')
              .Append(Format(burn.dv.Y)).Append(' ')
              .Append(Format(burn.dv.Z)).AppendLine();
        }
        return sb.ToString();
    }

    static string Format(double x) => x.ToString("G17", CultureInfo.InvariantCulture);

    static double EstimateImpactTimeDays(Scenario sc)
    {
        Vec3 r = sc.Asteroid.Pos - sc.Earth.Pos;
        Vec3 v = sc.Asteroid.Vel - sc.Earth.Vel;
        double vv = v.NormSq();
        if (vv <= 0) return sc.WarningDays;
        double t = -Vec3.Dot(r, v) / vv;
        return Math.Max(0.0, t / Day);
    }

    static double EstimateRequiredAsteroidDeflection(Scenario sc, double impactTimeDays, Vec3 relAtImpact)
    {
        double earthRadius = 6_371_000.0;
        double targetMiss = Math.Max(earthRadius * 2.0, relAtImpact.Norm() + earthRadius * 1.1);
        double timeRemaining = Math.Max(3600.0, impactTimeDays * Day * 0.45);
        return targetMiss / timeRemaining * ImpactSafetyFactor;
    }

    static State PropagateBody(Scenario sc, Body body, double tSec)
    {
        Vec3 pos = body.Pos + body.Vel * tSec;
        Vec3 vel = body.Vel;

        if (sc.Sun != null && !ReferenceEquals(body, sc.Sun))
        {
            Vec3 rel0 = body.Pos - sc.Sun.Pos;
            double mu = G * sc.Sun.Mass;
            vel += TwoBodyAcceleration(rel0, mu) * tSec;
            pos += TwoBodyAcceleration(rel0, mu) * (0.5 * tSec * tSec);
        }

        return new State(pos, vel);
    }

    static State PropagateBallistic(Vec3 startPos, Vec3 startVel, double startDays, double endDays, Scenario sc, double correctionDays = double.NaN, Vec3 correctionDv = default)
    {
        double t0 = startDays * Day;
        double t1 = endDays * Day;
        double tc = double.IsNaN(correctionDays) ? double.NaN : correctionDays * Day;
        Vec3 pos = startPos;
        Vec3 vel = startVel;
        double current = t0;
        double dt = Math.Max(600.0, Math.Min(21600.0, (t1 - t0) / 64.0));

        while (current < t1 - 1e-9)
        {
            double next = Math.Min(t1, current + dt);
            if (!double.IsNaN(tc) && current < tc && tc <= next)
            {
                Step(ref pos, ref vel, tc - current, sc);
                vel += correctionDv;
                current = tc;
                continue;
            }

            Step(ref pos, ref vel, next - current, sc);
            current = next;
        }

        return new State(pos, vel);
    }

    static void Step(ref Vec3 pos, ref Vec3 vel, double dt, Scenario sc)
    {
        Vec3 a1 = SpacecraftAcceleration(pos, sc);
        pos += vel * dt + a1 * (0.5 * dt * dt);
        Vec3 a2 = SpacecraftAcceleration(pos, sc);
        vel += (a1 + a2) * (0.5 * dt);
    }

    static Vec3 SpacecraftAcceleration(Vec3 pos, Scenario sc)
    {
        Vec3 acc = new Vec3();
        foreach (var body in sc.Bodies)
        {
            Vec3 rel = body.Pos - pos;
            double r2 = rel.NormSq() + 1.0;
            double invR3 = 1.0 / (Math.Sqrt(r2) * r2);
            acc += rel * (G * body.Mass * invR3);
        }
        return acc;
    }

    static Vec3 TwoBodyAcceleration(Vec3 rel, double mu)
    {
        double r2 = rel.NormSq() + 1.0;
        double invR3 = 1.0 / (Math.Sqrt(r2) * r2);
        return rel * (-mu * invR3);
    }
}
