using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Numerics;

public static class Program
{
    const double Eps = 1e-9;

    struct Triangle
    {
        public Vector3 A;
        public Vector3 B;
        public Vector3 C;
        public Triangle(Vector3 a, Vector3 b, Vector3 c)
        {
            A = a; B = b; C = c;
        }
    }

    struct Placement
    {
        public Vector3 T;
        public Quaternion Q;
        public Placement(Vector3 t, Quaternion q)
        {
            T = t; Q = q;
        }
    }

    sealed class Orientation
    {
        public Quaternion Q;
        public Vector3[] RotatedVertices = Array.Empty<Vector3>();
        public Triangle[] RotatedTriangles = Array.Empty<Triangle>();
        public Vector3 Min;
        public Vector3 Max;
        public Vector3 Size;
        public List<(int X, int Y, int Z)> Voxels = new List<(int X, int Y, int Z)>();
        public int VoxelVolume;
    }

    sealed class Polyhedron
    {
        public Vector3[] Vertices = Array.Empty<Vector3>();
        public int[][] Faces = Array.Empty<int[]>();
        public Triangle[] Triangles = Array.Empty<Triangle>();
        public double Volume;
        public Vector3 Min;
        public Vector3 Max;
        public Vector3 Size;
        public List<Orientation> Orientations = new List<Orientation>();
    }

    enum Strategy
    {
        SmallVoxel,
        MediumLayered,
        LargeAabb
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int v = fs.NextInt();
        int f = fs.NextInt();
        if (v <= 0 || f <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var poly = new Polyhedron();
        poly.Vertices = new Vector3[v];
        for (int i = 0; i < v; i++)
        {
            float x = (float)fs.NextDouble();
            float y = (float)fs.NextDouble();
            float z = (float)fs.NextDouble();
            poly.Vertices[i] = new Vector3(x, y, z);
        }

        poly.Faces = new int[f][];
        var tris = new List<Triangle>(Math.Max(f, 4));
        for (int i = 0; i < f; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();
            poly.Faces[i] = new[] { a, b, c };
            tris.Add(new Triangle(poly.Vertices[a], poly.Vertices[b], poly.Vertices[c]));
        }

        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();
        if (W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        poly.Triangles = tris.ToArray();
        ComputeBoundsAndVolume(poly);
        BuildOrientations(poly);

        long containerVolume = 1L * W * H * D;
        long aabbVolume = Math.Max(1L, (long)Math.Ceiling(poly.Size.X) * (long)Math.Ceiling(poly.Size.Y) * (long)Math.Ceiling(poly.Size.Z));
        Strategy strategy = ChooseStrategy(W, H, D, poly, containerVolume, aabbVolume);

        List<Placement> placements = strategy switch
        {
            Strategy.SmallVoxel => SolveSmallVoxel(W, H, D, poly),
            Strategy.MediumLayered => SolveMediumLayered(W, H, D, poly),
            _ => SolveLargeAabb(W, H, D, poly),
        };

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{Fmt(p.T.X)} {Fmt(p.T.Y)} {Fmt(p.T.Z)} {Fmt(p.Q.W)} {Fmt(p.Q.X)} {Fmt(p.Q.Y)} {Fmt(p.Q.Z)}");
        }
    }

    static string Fmt(float x) => x.ToString("0.######", CultureInfo.InvariantCulture);

    static Strategy ChooseStrategy(int w, int h, int d, Polyhedron poly, long containerVolume, long aabbVolume)
    {
        int maxDim = Math.Max(w, Math.Max(h, d));
        if (maxDim <= 18 && containerVolume <= 20000 && poly.Vertices.Length <= 128)
            return Strategy.SmallVoxel;
        if (maxDim <= 120 && containerVolume <= 3_000_000 && poly.Vertices.Length <= 2000)
            return Strategy.MediumLayered;
        return Strategy.LargeAabb;
    }

    static void ComputeBoundsAndVolume(Polyhedron poly)
    {
        Vector3 min = poly.Vertices[0];
        Vector3 max = poly.Vertices[0];
        foreach (var p in poly.Vertices)
        {
            min = Vector3.Min(min, p);
            max = Vector3.Max(max, p);
        }
        poly.Min = min;
        poly.Max = max;
        poly.Size = max - min;

        double vol6 = 0.0;
        foreach (var t in poly.Triangles)
        {
            vol6 += Vector3.Dot(t.A, Vector3.Cross(t.B, t.C));
        }
        poly.Volume = Math.Abs(vol6) / 6.0;
        if (poly.Volume < Eps)
        {
            poly.Volume = Math.Max(1.0, poly.Size.X * poly.Size.Y * poly.Size.Z);
        }
    }

    static void BuildOrientations(Polyhedron poly)
    {
        var bases = new[]
        {
            Quaternion.Identity,
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2),
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, 3 * MathF.PI / 2),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, 3 * MathF.PI / 2),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, 3 * MathF.PI / 2),
        };

        var seen = new HashSet<string>();
        foreach (var q0 in bases)
        {
            foreach (var q1 in bases)
            {
                var q = Quaternion.Normalize(Quaternion.Concatenate(q0, q1));
                var m = Matrix4x4.CreateFromQuaternion(q);
                var x = RoundAxis(Vector3.TransformNormal(Vector3.UnitX, m));
                var y = RoundAxis(Vector3.TransformNormal(Vector3.UnitY, m));
                var z = RoundAxis(Vector3.TransformNormal(Vector3.UnitZ, m));
                string key = $"{x.X},{x.Y},{x.Z}|{y.X},{y.Y},{y.Z}|{z.X},{z.Y},{z.Z}";
                if (!seen.Add(key)) continue;

                var o = new Orientation();
                o.Q = q;
                o.RotatedVertices = new Vector3[poly.Vertices.Length];
                Vector3 min = new Vector3(float.PositiveInfinity);
                Vector3 max = new Vector3(float.NegativeInfinity);
                for (int i = 0; i < poly.Vertices.Length; i++)
                {
                    var rp = Vector3.Transform(poly.Vertices[i], q);
                    o.RotatedVertices[i] = rp;
                    min = Vector3.Min(min, rp);
                    max = Vector3.Max(max, rp);
                }
                o.Min = min;
                o.Max = max;
                o.Size = max - min;
                o.RotatedTriangles = new Triangle[poly.Triangles.Length];
                for (int i = 0; i < poly.Triangles.Length; i++)
                {
                    var t = poly.Triangles[i];
                    o.RotatedTriangles[i] = new Triangle(Vector3.Transform(t.A, q), Vector3.Transform(t.B, q), Vector3.Transform(t.C, q));
                }
                BuildVoxelShape(o);
                poly.Orientations.Add(o);
            }
        }

        poly.Orientations = poly.Orientations
            .OrderBy(o => o.Size.X * o.Size.Y * o.Size.Z)
            .ThenByDescending(o => o.VoxelVolume)
            .ToList();
    }

    static Vector3 RoundAxis(Vector3 v)
    {
        return new Vector3(MathF.Round(v.X), MathF.Round(v.Y), MathF.Round(v.Z));
    }

    static void BuildVoxelShape(Orientation o)
    {
        int sx = Math.Max(1, (int)Math.Ceiling(o.Size.X - 1e-6));
        int sy = Math.Max(1, (int)Math.Ceiling(o.Size.Y - 1e-6));
        int sz = Math.Max(1, (int)Math.Ceiling(o.Size.Z - 1e-6));
        int limit = sx * sy * sz;
        if (limit > 200000)
        {
            o.Voxels.Add((0, 0, 0));
            o.VoxelVolume = 1;
            return;
        }

        for (int x = 0; x < sx; x++)
        for (int y = 0; y < sy; y++)
        for (int z = 0; z < sz; z++)
        {
            var p = new Vector3(o.Min.X + x + 0.5f, o.Min.Y + y + 0.5f, o.Min.Z + z + 0.5f);
            if (PointInClosedMesh(p, o.RotatedTriangles))
                o.Voxels.Add((x, y, z));
        }

        if (o.Voxels.Count == 0)
        {
            for (int x = 0; x < sx; x++)
            for (int y = 0; y < sy; y++)
            for (int z = 0; z < sz; z++)
                o.Voxels.Add((x, y, z));
        }
        o.VoxelVolume = o.Voxels.Count;
    }

    static List<Placement> SolveSmallVoxel(int W, int H, int D, Polyhedron poly)
    {
        var occupied = new bool[W, H, D];
        var results = new List<Placement>();

        foreach (var o in poly.Orientations.OrderByDescending(x => x.VoxelVolume * 1.0 / Math.Max(1e-6, x.Size.X * x.Size.Y * x.Size.Z)))
        {
            int sx = Math.Max(1, (int)Math.Ceiling(o.Size.X - 1e-6));
            int sy = Math.Max(1, (int)Math.Ceiling(o.Size.Y - 1e-6));
            int sz = Math.Max(1, (int)Math.Ceiling(o.Size.Z - 1e-6));
            if (sx > W || sy > H || sz > D) continue;

            for (int x = 0; x <= W - sx; x++)
            for (int y = 0; y <= H - sy; y++)
            for (int z = 0; z <= D - sz; z++)
            {
                if (!FitsVoxel(occupied, o, x, y, z)) continue;
                StampVoxel(occupied, o, x, y, z, true);
                Vector3 t = new Vector3(x - o.Min.X, y - o.Min.Y, z - o.Min.Z);
                results.Add(new Placement(t, o.Q));
            }
        }

        if (results.Count > 0) return results;
        return SolveMediumLayered(W, H, D, poly);
    }

    static bool FitsVoxel(bool[,,] occupied, Orientation o, int ox, int oy, int oz)
    {
        int W = occupied.GetLength(0);
        int H = occupied.GetLength(1);
        int D = occupied.GetLength(2);
        foreach (var c in o.Voxels)
        {
            int x = ox + c.X;
            int y = oy + c.Y;
            int z = oz + c.Z;
            if ((uint)x >= (uint)W || (uint)y >= (uint)H || (uint)z >= (uint)D) return false;
            if (occupied[x, y, z]) return false;
        }
        return true;
    }

    static void StampVoxel(bool[,,] occupied, Orientation o, int ox, int oy, int oz, bool value)
    {
        foreach (var c in o.Voxels)
            occupied[ox + c.X, oy + c.Y, oz + c.Z] = value;
    }

    static List<Placement> SolveMediumLayered(int W, int H, int D, Polyhedron poly)
    {
        var best = new List<Placement>();
        foreach (var o in poly.Orientations.Take(Math.Min(8, poly.Orientations.Count)))
        {
            int sx = Math.Max(1, (int)Math.Ceiling(o.Size.X - 1e-6));
            int sy = Math.Max(1, (int)Math.Ceiling(o.Size.Y - 1e-6));
            int sz = Math.Max(1, (int)Math.Ceiling(o.Size.Z - 1e-6));
            if (sx > W || sy > H || sz > D) continue;

            int nx = W / sx;
            int ny = H / sy;
            int nz = D / sz;
            int count = nx * ny * nz;
            if (count <= best.Count) continue;

            var cur = new List<Placement>(count);
            for (int z = 0; z < nz; z++)
            for (int y = 0; y < ny; y++)
            for (int x = 0; x < nx; x++)
            {
                Vector3 t = new Vector3(x * sx - o.Min.X, y * sy - o.Min.Y, z * sz - o.Min.Z);
                cur.Add(new Placement(t, o.Q));
            }
            if (cur.Count > best.Count) best = cur;
        }

        if (best.Count > 0) return best;
        return SolveLargeAabb(W, H, D, poly);
    }

    static List<Placement> SolveLargeAabb(int W, int H, int D, Polyhedron poly)
    {
        var best = new List<Placement>();
        foreach (var o in poly.Orientations)
        {
            int sx = Math.Max(1, (int)Math.Ceiling(o.Size.X - 1e-6));
            int sy = Math.Max(1, (int)Math.Ceiling(o.Size.Y - 1e-6));
            int sz = Math.Max(1, (int)Math.Ceiling(o.Size.Z - 1e-6));
            if (sx > W || sy > H || sz > D) continue;

            long nx = W / sx;
            long ny = H / sy;
            long nz = D / sz;
            long count = nx * ny * nz;
            if (count <= best.Count) continue;
            if (count > 2_000_000) count = 2_000_000;

            var cur = new List<Placement>((int)count);
            for (int z = 0; z < nz && cur.Count < count; z++)
            for (int y = 0; y < ny && cur.Count < count; y++)
            for (int x = 0; x < nx && cur.Count < count; x++)
            {
                Vector3 t = new Vector3(x * sx - o.Min.X, y * sy - o.Min.Y, z * sz - o.Min.Z);
                cur.Add(new Placement(t, o.Q));
            }
            if (cur.Count > best.Count) best = cur;
        }
        return best;
    }

    static bool PointInClosedMesh(Vector3 p, Triangle[] tris)
    {
        var dir = new Vector3(1.0f, 0.37139067f, 0.18273645f);
        int hits = 0;
        foreach (var t in tris)
        {
            if (RayIntersectsTriangle(p, dir, t, out _)) hits++;
        }
        return (hits & 1) == 1;
    }

    static bool RayIntersectsTriangle(Vector3 origin, Vector3 dir, Triangle tri, out float distance)
    {
        distance = 0;
        Vector3 e1 = tri.B - tri.A;
        Vector3 e2 = tri.C - tri.A;
        Vector3 p = Vector3.Cross(dir, e2);
        float det = Vector3.Dot(e1, p);
        if (MathF.Abs(det) < 1e-7f) return false;
        float invDet = 1f / det;
        Vector3 t = origin - tri.A;
        float u = Vector3.Dot(t, p) * invDet;
        if (u < 0f || u > 1f) return false;
        Vector3 q = Vector3.Cross(t, e1);
        float v = Vector3.Dot(dir, q) * invDet;
        if (v < 0f || u + v > 1f) return false;
        float dist = Vector3.Dot(e2, q) * invDet;
        if (dist <= 1e-6f) return false;
        distance = dist;
        return true;
    }

    sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _index;

        public FastScanner()
        {
            string input = Console.In.ReadToEnd();
            _tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            _index = 0;
        }

        public int NextInt() => int.Parse(_tokens[_index++], CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(_tokens[_index++], CultureInfo.InvariantCulture);
    }
}
