using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastStream
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastStream(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string NextToken()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return string.Empty;
            } while (c <= 32);

            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }

        public byte ReadByteRaw()
        {
            int c = Read();
            return c < 0 ? (byte)0 : (byte)c;
        }
    }

    private readonly struct Candidate
    {
        public readonly int X;
        public readonly int Y;
        public readonly int Z;
        public readonly int Mass;
        public readonly int Value;
        public readonly long Profit;
        public readonly int Distance;

        public Candidate(int x, int y, int z, int mass, int value, long profit, int distance)
        {
            X = x;
            Y = y;
            Z = z;
            Mass = mass;
            Value = value;
            Profit = profit;
            Distance = distance;
        }
    }

    private sealed class CandidateComparer : IComparer<Candidate>
    {
        public int Compare(Candidate a, Candidate b)
        {
            int p = b.Profit.CompareTo(a.Profit);
            if (p != 0) return p;
            int v = b.Value.CompareTo(a.Value);
            if (v != 0) return v;
            int d = a.Distance.CompareTo(b.Distance);
            if (d != 0) return d;
            int x = a.X.CompareTo(b.X);
            if (x != 0) return x;
            int y = a.Y.CompareTo(b.Y);
            if (y != 0) return y;
            return a.Z.CompareTo(b.Z);
        }
    }

    private sealed class Operation
    {
        public int DigX;
        public int DigY;
        public int DigZ;
        public List<(int x, int y)> Steps = new List<(int x, int y)>();
        public int DestX;
        public int DestY;
    }

    public static void Main()
    {
        var input = new FastStream(Console.OpenStandardInput());
        int xCount = int.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        int yCount = int.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        int zCount = int.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.NextToken(), CultureInfo.InvariantCulture);

        long totalCells = (long)xCount * yCount * zCount;
        bool exhaustiveSurface = totalCells <= 20_000_000L;
        bool compactColumns = totalCells <= 120_000_000L;
        int candidateCap = totalCells <= 2_000_000L ? 20000 : totalCells <= 20_000_000L ? 12000 : 4000;
        int scanRadius = totalCells <= 2_000_000L ? int.MaxValue : totalCells <= 50_000_000L ? 256 : 96;

        int[,] topSolid = new int[xCount, yCount];
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                topSolid[x, y] = -1;
            }
        }

        var best = new List<Candidate>(candidateCap + 16);
        byte[] columnData = compactColumns ? new byte[(long)xCount * yCount * zCount] : null;
        int yz = yCount * zCount;

        for (int x = 0; x < xCount; x++)
        {
            int manhattanX = Math.Abs(x);
            for (int y = 0; y < yCount; y++)
            {
                int distance = manhattanX + Math.Abs(y);
                bool withinRadius = distance <= scanRadius;
                int top = -1;
                long columnBase = (long)x * yz + (long)y * zCount;

                for (int z = 0; z < zCount; z++)
                {
                    byte d = input.ReadByteRaw();
                    if (columnData != null)
                    {
                        columnData[columnBase + z] = d;
                    }

                    int mass = d & 15;
                    int value = d >> 4;
                    if (mass == 0) continue;

                    top = z;
                    if (!withinRadius || value == 0) continue;

                    long revenue = 1000L * mass * value;
                    long moveCost = EstimateMoveCost(mass, x, y, z);
                    long profit = revenue - moveCost;
                    if (profit <= 0) continue;

                    AddCandidate(best, new Candidate(x, y, z, mass, value, profit, distance), candidateCap);
                }

                topSolid[x, y] = top;
            }
        }

        if (exhaustiveSurface && columnData != null)
        {
            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    ExploreColumnSides(x, y, xCount, yCount, zCount, columnData, yz, scanRadius, best, candidateCap);
                }
            }
        }

        best.Sort(new CandidateComparer());

        var mined = new HashSet<long>();
        var ops = new List<Operation>(Math.Min(best.Count, candidateCap));
        foreach (var c in best)
        {
            long key = Encode(c.X, c.Y, c.Z, yCount, zCount);
            if (mined.Contains(key)) continue;
            if (!IsExposed(c.X, c.Y, c.Z, topSolid, yCount)) continue;
            if (!CanReachPlant(c.X, c.Y, c.Z)) continue;

            var op = BuildOperation(c.X, c.Y, c.Z);
            ops.Add(op);
            mined.Add(key);
            topSolid[c.X, c.Y] = c.Z - 1;

            if (ops.Count >= candidateCap) break;
            if (totalCells > 50_000_000L && ops.Count >= 512) break;
        }

        if (ops.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        var output = new StringBuilder();
        output.Append(ops.Count).Append('\n');
        foreach (var op in ops)
        {
            output.Append(op.Steps.Count)
                  .Append(' ')
                  .Append(op.DigX).Append(' ')
                  .Append(op.DigY).Append(' ')
                  .Append(op.DigZ);

            for (int i = 0; i < op.Steps.Count; i++)
            {
                output.Append(' ').Append(op.Steps[i].x).Append(' ').Append(op.Steps[i].y);
            }

            output.Append(' ').Append(op.DestX).Append(' ').Append(op.DestY).Append('\n');
        }

        Console.Write(output.ToString());
    }

    private static long EstimateMoveCost(int mass, int x, int y, int z)
    {
        int horizontal = Math.Abs(x) + Math.Abs(y);
        int uphill = z;
        int flat = Math.Max(0, horizontal - uphill);
        return mass * (5L * uphill + flat);
    }

    private static void AddCandidate(List<Candidate> list, Candidate candidate, int cap)
    {
        list.Add(candidate);
        if (list.Count <= cap * 2) return;
        list.Sort(new CandidateComparer());
        list.RemoveRange(cap, list.Count - cap);
    }

    private static void ExploreColumnSides(
        int x,
        int y,
        int xCount,
        int yCount,
        int zCount,
        byte[] data,
        int yz,
        int scanRadius,
        List<Candidate> best,
        int candidateCap)
    {
        int distance = Math.Abs(x) + Math.Abs(y);
        if (distance > scanRadius) return;

        long columnBase = (long)x * yz + (long)y * zCount;
        for (int z = 0; z < zCount; z++)
        {
            byte d = data[columnBase + z];
            int mass = d & 15;
            int value = d >> 4;
            if (mass == 0 || value == 0) continue;

            bool sideExposed = false;
            if (x == 0 || ((data[((long)(x - 1) * yz) + (long)y * zCount + z] & 15) == 0)) sideExposed = true;
            else if (x == xCount - 1 || ((data[((long)(x + 1) * yz) + (long)y * zCount + z] & 15) == 0)) sideExposed = true;
            else if (y == 0 || ((data[columnBase - zCount + z] & 15) == 0)) sideExposed = true;
            else if (y == yCount - 1 || ((data[columnBase + zCount + z] & 15) == 0)) sideExposed = true;

            if (!sideExposed) continue;

            long revenue = 1000L * mass * value;
            long moveCost = EstimateMoveCost(mass, x, y, z);
            long profit = revenue - moveCost;
            if (profit <= 0) continue;
            AddCandidate(best, new Candidate(x, y, z, mass, value, profit, distance), candidateCap);
        }
    }

    private static bool IsExposed(int x, int y, int z, int[,] topSolid, int yCount)
    {
        if (z >= topSolid[x, y]) return true;
        int xCount = topSolid.GetLength(0);
        if (x == 0 || topSolid[x - 1, y] < z) return true;
        if (x == xCount - 1 || topSolid[x + 1, y] < z) return true;
        if (y == 0 || topSolid[x, y - 1] < z) return true;
        if (y == yCount - 1 || topSolid[x, y + 1] < z) return true;
        return false;
    }

    private static bool CanReachPlant(int x, int y, int z)
    {
        int horizontal = Math.Abs(x) + Math.Abs(y);
        return z <= horizontal;
    }

    private static Operation BuildOperation(int x, int y, int z)
    {
        var op = new Operation { DigX = x, DigY = y, DigZ = z, DestX = 0, DestY = 0 };
        int cx = x;
        int cy = y;

        while (cx != 0)
        {
            cx += cx > 0 ? -1 : 1;
            op.Steps.Add((cx, cy));
        }
        while (cy != 0)
        {
            cy += cy > 0 ? -1 : 1;
            op.Steps.Add((cx, cy));
        }

        return op;
    }

    private static long Encode(int x, int y, int z, int yCount, int zCount)
    {
        return ((long)x * yCount + y) * zCount + z;
    }
}
