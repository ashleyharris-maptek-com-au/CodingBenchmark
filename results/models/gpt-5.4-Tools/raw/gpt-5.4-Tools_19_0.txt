using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const int MaxStoredOperations = 200000;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int xCount = scanner.NextInt();
        int yCount = scanner.NextInt();
        int zCount = scanner.NextInt();
        double xBlockSize = scanner.NextDouble();
        double yBlockSize = scanner.NextDouble();
        double zBlockSize = scanner.NextDouble();

        long cellCount = (long)xCount * yCount;
        bool storeHeights = cellCount <= 12_000_000;
        int[] heights = storeHeights ? new int[cellCount] : Array.Empty<int>();

        bool aggressive = cellCount <= 2_000_000 && zCount <= 4096;
        int targetOreOps = aggressive ? MaxStoredOperations : Math.Min(50000, MaxStoredOperations);
        var operations = new List<MoveOp>(Math.Min(targetOreOps, 16384));

        double verticalRatioX = zBlockSize / Math.Max(1e-9, xBlockSize);
        double verticalRatioY = zBlockSize / Math.Max(1e-9, yBlockSize);
        int maxRisePerX = Math.Max(1, (int)Math.Floor(1.0 / verticalRatioX + 1e-9));
        int maxRisePerY = Math.Max(1, (int)Math.Floor(1.0 / verticalRatioY + 1e-9));
        int maxRisePerStep = Math.Max(1, Math.Min(maxRisePerX, maxRisePerY));

        int bestDumpX = xCount > 1 ? xCount - 1 : 0;
        int bestDumpY = yCount > 1 ? yCount - 1 : 0;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                long index = (long)x * yCount + y;
                int columnHeight = 0;
                int topOreZ = -1;
                int topOreValue = 0;
                long topOreScore = long.MinValue;

                for (int z = 0; z < zCount; z++)
                {
                    int d = scanner.NextByteOrThrow();
                    int mass = d & 15;
                    int value = d >> 4;
                    if (mass > 0)
                    {
                        columnHeight = z + 1;
                    }

                    if (value > 0 && mass > 0)
                    {
                        int horizontal = x + y;
                        long revenue = 1000L * mass * value;
                        long moveCost = EstimateMoveCost(mass, horizontal, z, maxRisePerStep);
                        long score = revenue - moveCost;
                        if (score > topOreScore)
                        {
                            topOreScore = score;
                            topOreZ = z;
                            topOreValue = value;
                        }
                    }
                }

                if (storeHeights)
                {
                    heights[index] = columnHeight;
                }

                if (topOreZ >= 0 && topOreScore > 0 && operations.Count < targetOreOps)
                {
                    int digZ = topOreZ;
                    if (storeHeights)
                    {
                        digZ = heights[index] - 1;
                    }

                    if (digZ >= 0)
                    {
                        bool sendToPlant = topOreValue > 0;
                        var path = BuildAdaptivePath(x, y, digZ, maxRisePerStep);
                        if (path.Count > 0 || (x == 0 && y == 0 && sendToPlant))
                        {
                            operations.Add(new MoveOp(x, y, digZ, path, sendToPlant ? 0 : bestDumpX, sendToPlant ? 0 : bestDumpY));
                            if (storeHeights)
                            {
                                heights[index] = Math.Max(0, heights[index] - 1);
                            }
                        }
                    }
                }
            }
        }

        EmitOperations(operations);
    }

    private static long EstimateMoveCost(int mass, int horizontalDistance, int z, int maxRisePerStep)
    {
        if (horizontalDistance == 0)
        {
            return 0;
        }

        int climbsNeeded = (z + Math.Max(1, maxRisePerStep) - 1) / Math.Max(1, maxRisePerStep);
        climbsNeeded = Math.Min(climbsNeeded, horizontalDistance);
        int flatSteps = horizontalDistance - climbsNeeded;
        return mass * (5L * climbsNeeded + flatSteps);
    }

    private static List<Step> BuildAdaptivePath(int x, int y, int z, int maxRisePerStep)
    {
        if (x == 0 && y == 0)
        {
            return new List<Step>();
        }

        if (z == 0 || z <= (x + y) * maxRisePerStep)
        {
            return BuildDirectPath(x, y);
        }

        return BuildSwitchbackPath(x, y, z, maxRisePerStep);
    }

    private static List<Step> BuildDirectPath(int x, int y)
    {
        var steps = new List<Step>(x + y);
        int cx = x;
        int cy = y;

        while (cx > 0)
        {
            cx--;
            steps.Add(new Step(cx, cy));
        }

        while (cy > 0)
        {
            cy--;
            steps.Add(new Step(cx, cy));
        }

        return steps;
    }

    private static List<Step> BuildSwitchbackPath(int x, int y, int z, int maxRisePerStep)
    {
        int neededLength = (z + maxRisePerStep - 1) / maxRisePerStep;
        int directLength = x + y;
        int extra = neededLength - directLength;
        if (extra <= 0)
        {
            return BuildDirectPath(x, y);
        }

        int loops = (extra + 1) / 2;
        int laneY = y > 0 ? y : 0;
        int startX = x;
        var steps = new List<Step>(directLength + loops * 2 + 4);
        int cx = startX;
        int cy = laneY;

        bool canLoopX = x + 1 < int.MaxValue && x > 0;
        if (!canLoopX)
        {
            return BuildDirectPath(x, y);
        }

        for (int i = 0; i < loops; i++)
        {
            steps.Add(new Step(cx + 1, cy));
            steps.Add(new Step(cx, cy));
        }

        while (cx > 0)
        {
            cx--;
            steps.Add(new Step(cx, cy));
        }

        while (cy > 0)
        {
            cy--;
            steps.Add(new Step(cx, cy));
        }

        return steps;
    }

    private static void EmitOperations(List<MoveOp> operations)
    {
        var output = new StringBuilder();
        output.Append(operations.Count).Append('\n');
        foreach (var op in operations)
        {
            output.Append(op.Path.Count)
                .Append(' ').Append(op.DigX)
                .Append(' ').Append(op.DigY)
                .Append(' ').Append(op.DigZ);

            for (int i = 0; i < op.Path.Count; i++)
            {
                output.Append(' ').Append(op.Path[i].X)
                    .Append(' ').Append(op.Path[i].Y);
            }

            output.Append(' ').Append(op.DestX)
                .Append(' ').Append(op.DestY)
                .Append('\n');
        }

        Console.Write(output.ToString());
    }

    private readonly struct Step
    {
        public Step(int x, int y)
        {
            X = x;
            Y = y;
        }

        public int X { get; }
        public int Y { get; }
    }

    private readonly struct MoveOp
    {
        public MoveOp(int digX, int digY, int digZ, List<Step> path, int destX, int destY)
        {
            DigX = digX;
            DigY = digY;
            DigZ = digZ;
            Path = path;
            DestX = destX;
            DestY = destY;
        }

        public int DigX { get; }
        public int DigY { get; }
        public int DigZ { get; }
        public List<Step> Path { get; }
        public int DestX { get; }
        public int DestY { get; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c = SkipWhitespace();
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public double NextDouble()
        {
            int c = SkipWhitespace();
            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }

            return double.Parse(sb.ToString(), CultureInfo.InvariantCulture);
        }

        public int NextByteOrThrow()
        {
            int c = Read();
            if (c < 0)
            {
                throw new EndOfStreamException();
            }

            return c;
        }

        private int SkipWhitespace()
        {
            int c;
            do
            {
                c = Read();
            } while (c >= 0 && c <= 32);

            return c;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }
}
