use std::cmp::Ordering;
use std::f64::consts::{FRAC_PI_2, PI};
use std::io::{self, Read};

const EPS: f64 = 1e-7;

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn cross(self, other: Point) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn rotate(self, angle: f64) -> Self {
        let (s, c) = angle.sin_cos();
        Self {
            x: self.x * c - self.y * s,
            y: self.x * s + self.y * c,
        }
    }
}

impl std::ops::Add for Point {
    type Output = Point;
    fn add(self, rhs: Point) -> Point {
        Point::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Point {
    type Output = Point;
    fn sub(self, rhs: Point) -> Point {
        Point::new(self.x - rhs.x, self.y - rhs.y)
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Aabb {
    min: Point,
    max: Point,
}

impl Aabb {
    fn intersects(&self, other: &Aabb) -> bool {
        !(self.max.x <= other.min.x + EPS
            || other.max.x <= self.min.x + EPS
            || self.max.y <= other.min.y + EPS
            || other.max.y <= self.min.y + EPS)
    }
}

#[derive(Clone)]
struct Piece {
    original_index: usize,
    vertices: Vec<Point>,
    area: f64,
}

#[derive(Clone)]
struct OrientedPiece {
    angle: f64,
    vertices: Vec<Point>,
    aabb: Aabb,
    area: f64,
}

#[derive(Clone)]
struct Placement {
    stock_idx: usize,
    translation: Point,
    rotation: f64,
}

#[derive(Clone)]
struct PlacedPiece {
    vertices: Vec<Point>,
    aabb: Aabb,
}

#[derive(Clone)]
struct StockBin {
    placements: Vec<PlacedPiece>,
    skyline: Vec<(f64, f64, f64)>,
    used_area: f64,
}

fn polygon_area(poly: &[Point]) -> f64 {
    let mut sum = 0.0;
    for i in 0..poly.len() {
        sum += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    0.5 * sum.abs()
}

fn polygon_signed_area(poly: &[Point]) -> f64 {
    let mut sum = 0.0;
    for i in 0..poly.len() {
        sum += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    0.5 * sum
}

fn normalize_ccw(poly: &mut [Point]) {
    if polygon_signed_area(poly) < 0.0 {
        poly.reverse();
    }
}

fn compute_aabb(poly: &[Point]) -> Aabb {
    let mut min = Point::new(f64::INFINITY, f64::INFINITY);
    let mut max = Point::new(f64::NEG_INFINITY, f64::NEG_INFINITY);
    for &p in poly {
        min.x = min.x.min(p.x);
        min.y = min.y.min(p.y);
        max.x = max.x.max(p.x);
        max.y = max.y.max(p.y);
    }
    Aabb { min, max }
}

fn translate_polygon(poly: &[Point], shift: Point) -> Vec<Point> {
    poly.iter().map(|&p| p + shift).collect()
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let on_segment = ((p - a).cross(b - a)).abs() <= EPS
            && (p.x - a.x).min(p.x - b.x) <= EPS
            && (p.x - a.x).max(p.x - b.x) >= -EPS
            && (p.y - a.y).min(p.y - b.y) <= EPS
            && (p.y - a.y).max(p.y - b.y) >= -EPS;
        if on_segment {
            return true;
        }
        let intersect =
            ((a.y > p.y) != (b.y > p.y)) && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    (b - a).cross(c - a)
}

fn on_segment(a: Point, b: Point, p: Point) -> bool {
    orient(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if o1.abs() <= EPS && on_segment(a1, a2, b1) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a1, a2, b2) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(b1, b2, a1) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(b1, b2, a2) {
        return true;
    }

    (o1 > EPS) != (o2 > EPS) && (o3 > EPS) != (o4 > EPS)
}

fn polygons_overlap(a: &[Point], b: &[Point]) -> bool {
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if segments_intersect(a1, a2, b1, b2) {
                let shared_endpoint = ((a1.x - b1.x).abs() <= EPS && (a1.y - b1.y).abs() <= EPS)
                    || ((a1.x - b2.x).abs() <= EPS && (a1.y - b2.y).abs() <= EPS)
                    || ((a2.x - b1.x).abs() <= EPS && (a2.y - b1.y).abs() <= EPS)
                    || ((a2.x - b2.x).abs() <= EPS && (a2.y - b2.y).abs() <= EPS);
                if !shared_endpoint {
                    return true;
                }
            }
        }
    }
    point_in_polygon(a[0], b) || point_in_polygon(b[0], a)
}

fn polygon_inside(stock: &[Point], piece: &[Point]) -> bool {
    for &p in piece {
        if !point_in_polygon(p, stock) {
            return false;
        }
    }
    true
}

fn distinct_angles(piece: &[Point], granular: bool) -> Vec<f64> {
    let mut angles = vec![0.0, FRAC_PI_2];
    if granular {
        for i in 0..piece.len() {
            let e = piece[(i + 1) % piece.len()] - piece[i];
            let angle = -e.y.atan2(e.x);
            angles.push(angle);
            angles.push(angle + FRAC_PI_2);
        }
        let extra = [
            0.0,
            PI / 6.0,
            PI / 4.0,
            PI / 3.0,
            FRAC_PI_2,
            2.0 * PI / 3.0,
            3.0 * PI / 4.0,
        ];
        angles.extend(extra);
    }
    let two_pi = 2.0 * PI;
    for a in &mut angles {
        *a = ((*a % two_pi) + two_pi) % two_pi;
    }
    angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    angles.dedup_by(|a, b| (*a - *b).abs() < 1e-6);
    angles
}

fn build_orientations(piece: &Piece, granular: bool) -> Vec<OrientedPiece> {
    let mut out = Vec::new();
    for angle in distinct_angles(&piece.vertices, granular) {
        let rotated: Vec<Point> = piece.vertices.iter().map(|&p| p.rotate(angle)).collect();
        let aabb = compute_aabb(&rotated);
        out.push(OrientedPiece {
            angle,
            vertices: rotated,
            aabb,
            area: piece.area,
        });
    }
    out
}

fn stock_bounds(stock: &[Point]) -> Aabb {
    compute_aabb(stock)
}

fn stock_contains_aabb(stock_aabb: &Aabb, piece_aabb: &Aabb, shift: Point) -> bool {
    piece_aabb.min.x + shift.x >= stock_aabb.min.x - EPS
        && piece_aabb.max.x + shift.x <= stock_aabb.max.x + EPS
        && piece_aabb.min.y + shift.y >= stock_aabb.min.y - EPS
        && piece_aabb.max.y + shift.y <= stock_aabb.max.y + EPS
}

fn candidate_positions(
    stock_poly: &[Point],
    stock_aabb: &Aabb,
    stock: &StockBin,
    oriented: &OrientedPiece,
    detailed: bool,
) -> Vec<Point> {
    let mut candidates = Vec::new();
    candidates.push(Point::new(
        stock_aabb.min.x - oriented.aabb.min.x,
        stock_aabb.min.y - oriented.aabb.min.y,
    ));
    candidates.push(Point::new(
        stock_aabb.min.x - oriented.aabb.min.x,
        stock_aabb.max.y - oriented.aabb.max.y,
    ));
    candidates.push(Point::new(
        stock_aabb.max.x - oriented.aabb.max.x,
        stock_aabb.min.y - oriented.aabb.min.y,
    ));
    if detailed {
        for &(x0, x1, y) in &stock.skyline {
            let _ = x0;
            candidates.push(Point::new(x1 - oriented.aabb.max.x, y - oriented.aabb.min.y));
            candidates.push(Point::new(x0 - oriented.aabb.min.x, y - oriented.aabb.min.y));
        }
        for placed in &stock.placements {
            candidates.push(Point::new(
                placed.aabb.max.x - oriented.aabb.min.x,
                placed.aabb.min.y - oriented.aabb.min.y,
            ));
            candidates.push(Point::new(
                placed.aabb.min.x - oriented.aabb.min.x,
                placed.aabb.max.y - oriented.aabb.min.y,
            ));
            candidates.push(Point::new(
                placed.aabb.max.x - oriented.aabb.max.x,
                placed.aabb.max.y - oriented.aabb.min.y,
            ));
        }
        for &sp in stock_poly {
            candidates.push(sp - oriented.aabb.min);
            candidates.push(Point::new(sp.x - oriented.aabb.max.x, sp.y - oriented.aabb.min.y));
        }
    }
    candidates.sort_by(|a, b| {
        a.y.partial_cmp(&b.y)
            .unwrap_or(Ordering::Equal)
            .then(a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal))
    });
    candidates.dedup_by(|a, b| (a.x - b.x).abs() < 1e-6 && (a.y - b.y).abs() < 1e-6);
    candidates
}

fn update_skyline(stock: &mut StockBin, aabb: &Aabb) {
    stock
        .skyline
        .push((aabb.min.x, aabb.max.x, aabb.max.y));
}

fn try_place_in_stock(
    stock_poly: &[Point],
    stock_aabb: &Aabb,
    stock: &mut StockBin,
    orientations: &[OrientedPiece],
    detailed: bool,
) -> Option<(Point, f64, Vec<Point>, Aabb, f64)> {
    let mut best: Option<(Point, f64, Vec<Point>, Aabb, f64, f64, f64)> = None;

    for oriented in orientations {
        let positions = candidate_positions(stock_poly, stock_aabb, stock, oriented, detailed);
        for shift in positions {
            if !stock_contains_aabb(stock_aabb, &oriented.aabb, shift) {
                continue;
            }
            let moved = translate_polygon(&oriented.vertices, shift);
            if !polygon_inside(stock_poly, &moved) {
                continue;
            }
            let moved_aabb = Aabb {
                min: oriented.aabb.min + shift,
                max: oriented.aabb.max + shift,
            };
            let mut collide = false;
            for other in &stock.placements {
                if moved_aabb.intersects(&other.aabb) && polygons_overlap(&moved, &other.vertices) {
                    collide = true;
                    break;
                }
            }
            if collide {
                continue;
            }
            let top = moved_aabb.max.y;
            let right = moved_aabb.max.x;
            let score = top * 1e6 + right - oriented.area * 1e-3;
            match &best {
                None => {
                    best = Some((
                        shift,
                        oriented.angle,
                        moved,
                        moved_aabb,
                        oriented.area,
                        score,
                        right,
                    ));
                }
                Some((_, _, _, _, _, best_score, best_right)) => {
                    if score < *best_score - 1e-9
                        || ((score - *best_score).abs() <= 1e-9 && right < *best_right)
                    {
                        best = Some((
                            shift,
                            oriented.angle,
                            moved,
                            moved_aabb,
                            oriented.area,
                            score,
                            right,
                        ));
                    }
                }
            }
        }
    }

    best.map(|(shift, angle, moved, aabb, area, _, _)| (shift, angle, moved, aabb, area))
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_poly.push(Point::new(x, y));
    }
    normalize_ccw(&mut stock_poly);
    let stock_aabb = stock_bounds(&stock_poly);

    let mut pieces = Vec::with_capacity(np);
    let mut total_vertices = 0usize;
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        total_vertices += pv;
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Point::new(x, y));
        }
        normalize_ccw(&mut poly);
        let area = polygon_area(&poly);
        pieces.push(Piece {
            original_index: idx,
            vertices: poly,
            area,
        });
    }

    let complex = np <= 200 && total_vertices <= 4000;
    let medium = np <= 1200 && total_vertices <= 20000;

    pieces.sort_by(|a, b| {
        b.area
            .partial_cmp(&a.area)
            .unwrap_or(Ordering::Equal)
            .then(b.vertices.len().cmp(&a.vertices.len()))
    });

    let mut placements = vec![
        Placement {
            stock_idx: 0,
            translation: Point::new(0.0, 0.0),
            rotation: 0.0,
        };
        np
    ];

    let mut stocks: Vec<StockBin> = Vec::new();

    for piece in &pieces {
        let orientations = build_orientations(piece, complex);
        let detailed = complex || (medium && piece.vertices.len() <= 16);
        let mut placed = false;

        let mut stock_order: Vec<usize> = (0..stocks.len()).collect();
        stock_order.sort_by(|&a, &b| {
            stocks[b]
                .used_area
                .partial_cmp(&stocks[a].used_area)
                .unwrap_or(Ordering::Equal)
        });

        for stock_idx in stock_order {
            if let Some((shift, angle, moved, aabb, area)) =
                try_place_in_stock(&stock_poly, &stock_aabb, &mut stocks[stock_idx], &orientations, detailed)
            {
                stocks[stock_idx].placements.push(PlacedPiece { vertices: moved, aabb });
                stocks[stock_idx].used_area += area;
                update_skyline(&mut stocks[stock_idx], &aabb);
                placements[piece.original_index] = Placement {
                    stock_idx,
                    translation: shift,
                    rotation: angle,
                };
                placed = true;
                break;
            }
        }

        if !placed {
            let mut new_stock = StockBin {
                placements: Vec::new(),
                skyline: vec![(stock_aabb.min.x, stock_aabb.max.x, stock_aabb.min.y)],
                used_area: 0.0,
            };
            let fallback_orientations = if medium { orientations.clone() } else { build_orientations(piece, false) };
            let (shift, angle, moved, aabb, area) = try_place_in_stock(
                &stock_poly,
                &stock_aabb,
                &mut new_stock,
                &fallback_orientations,
                true,
            )
            .unwrap_or_else(|| {
                panic!("piece {} could not be placed into an empty stock", piece.original_index);
            });
            new_stock.placements.push(PlacedPiece { vertices: moved, aabb });
            new_stock.used_area += area;
            update_skyline(&mut new_stock, &aabb);
            let stock_idx = stocks.len();
            placements[piece.original_index] = Placement {
                stock_idx,
                translation: shift,
                rotation: angle,
            };
            stocks.push(new_stock);
        }
    }

    println!("{}", stocks.len());
    for place in placements {
        println!(
            "{} {:.10} {:.10} {:.10}",
            place.stock_idx, place.translation.x, place.translation.y, place.rotation
        );
    }
}
