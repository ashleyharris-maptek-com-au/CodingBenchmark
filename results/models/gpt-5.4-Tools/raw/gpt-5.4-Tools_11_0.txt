use std::cmp::Ordering;
use std::f64::consts::{FRAC_PI_2, PI};
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn add(self, other: Point) -> Self {
        Self::new(self.x + other.x, self.y + other.y)
    }

    fn sub(self, other: Point) -> Self {
        Self::new(self.x - other.x, self.y - other.y)
    }

    fn dot(self, other: Point) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Point) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn rotate(self, angle: f64) -> Self {
        let (s, c) = angle.sin_cos();
        Self::new(self.x * c - self.y * s, self.x * s + self.y * c)
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

impl BBox {
    fn width(&self) -> f64 {
        self.max_x - self.min_x
    }

    fn height(&self) -> f64 {
        self.max_y - self.min_y
    }

    fn intersects(&self, other: &BBox) -> bool {
        self.min_x < other.max_x - EPS
            && self.max_x > other.min_x + EPS
            && self.min_y < other.max_y - EPS
            && self.max_y > other.min_y + EPS
    }
}

#[derive(Clone)]
struct Polygon {
    pts: Vec<Point>,
    bbox: BBox,
    area: f64,
}

impl Polygon {
    fn new(pts: Vec<Point>) -> Self {
        let bbox = bbox_of(&pts);
        let area = polygon_area(&pts).abs();
        Self { pts, bbox, area }
    }
}

#[derive(Clone)]
struct OrientedPiece {
    polygon: Polygon,
    width: f64,
    height: f64,
    angle: f64,
}

#[derive(Clone)]
struct Piece {
    original_index: usize,
    variants: Vec<OrientedPiece>,
    area: f64,
}

#[derive(Clone)]
struct Placement {
    stock_index: usize,
    translation: Point,
    rotation: f64,
    placed_polygon: Polygon,
}

#[derive(Clone)]
struct Occupied {
    bbox: BBox,
    poly: Polygon,
}

struct StockState {
    occupied: Vec<Occupied>,
    skyline: Vec<Shelf>,
}

#[derive(Clone, Copy)]
struct Shelf {
    y: f64,
    height: f64,
    x: f64,
}

fn polygon_area(pts: &[Point]) -> f64 {
    let mut area = 0.0;
    for i in 0..pts.len() {
        let a = pts[i];
        let b = pts[(i + 1) % pts.len()];
        area += a.cross(b);
    }
    area * 0.5
}

fn bbox_of(pts: &[Point]) -> BBox {
    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in pts {
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    BBox {
        min_x,
        min_y,
        max_x,
        max_y,
    }
}

fn normalize_polygon(pts: &[Point]) -> Vec<Point> {
    let bbox = bbox_of(pts);
    pts.iter()
        .map(|p| Point::new(p.x - bbox.min_x, p.y - bbox.min_y))
        .collect()
}

fn transform_polygon(pts: &[Point], angle: f64, shift: Point) -> Vec<Point> {
    pts.iter().map(|p| p.rotate(angle).add(shift)).collect()
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let ab = b.sub(a);
    let ap = p.sub(a);
    let cross = ab.cross(ap).abs();
    if cross > 1e-7 {
        return false;
    }
    let dot = ap.dot(p.sub(b));
    dot <= 1e-7
}

fn point_in_polygon(poly: &[Point], p: Point) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if point_on_segment(p, a, b) {
            return true;
        }
        let intersects = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x);
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn segments_intersect(a: Point, b: Point, c: Point, d: Point) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if o1.abs() <= 1e-7 && point_on_segment(c, a, b) {
        return true;
    }
    if o2.abs() <= 1e-7 && point_on_segment(d, a, b) {
        return true;
    }
    if o3.abs() <= 1e-7 && point_on_segment(a, c, d) {
        return true;
    }
    if o4.abs() <= 1e-7 && point_on_segment(b, c, d) {
        return true;
    }

    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn polygons_overlap(a: &[Point], b: &[Point]) -> bool {
    let bb_a = bbox_of(a);
    let bb_b = bbox_of(b);
    if !bb_a.intersects(&bb_b) {
        return false;
    }
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if segments_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    point_in_polygon(a, b[0]) || point_in_polygon(b, a[0])
}

fn polygon_inside(stock: &[Point], poly: &[Point]) -> bool {
    poly.iter().all(|&p| point_in_polygon(stock, p))
}

fn candidate_angles(poly: &[Point], large_mode: bool) -> Vec<f64> {
    let mut angles = vec![0.0, FRAC_PI_2];
    if !large_mode {
        angles.extend([PI, 3.0 * FRAC_PI_2]);
        if poly.len() <= 32 {
            for i in 0..poly.len() {
                let a = poly[i];
                let b = poly[(i + 1) % poly.len()];
                let edge = b.sub(a);
                if edge.x.abs() + edge.y.abs() > EPS {
                    let ang = (-edge.y).atan2(edge.x);
                    let canon = canonical_angle(ang);
                    angles.push(canon);
                    angles.push(canonical_angle(canon + FRAC_PI_2));
                }
            }
        }
    }
    angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    angles.dedup_by(|a, b| (*a - *b).abs() < 1e-8);
    angles
}

fn canonical_angle(mut a: f64) -> f64 {
    while a < 0.0 {
        a += 2.0 * PI;
    }
    while a >= 2.0 * PI {
        a -= 2.0 * PI;
    }
    a
}

fn build_piece(poly: Polygon, original_index: usize, large_mode: bool) -> Piece {
    let normalized = normalize_polygon(&poly.pts);
    let angles = candidate_angles(&normalized, large_mode);
    let mut variants = Vec::new();
    for angle in angles {
        let rotated: Vec<Point> = normalized.iter().map(|p| p.rotate(angle)).collect();
        let placed = normalize_polygon(&rotated);
        let polygon = Polygon::new(placed);
        variants.push(OrientedPiece {
            width: polygon.bbox.width(),
            height: polygon.bbox.height(),
            angle,
            polygon,
        });
    }
    variants.sort_by(|a, b| {
        (a.width * a.height)
            .partial_cmp(&(b.width * b.height))
            .unwrap_or(Ordering::Equal)
    });
    variants.dedup_by(|a, b| {
        (a.width - b.width).abs() < 1e-7
            && (a.height - b.height).abs() < 1e-7
            && a.polygon.pts.len() == b.polygon.pts.len()
    });
    Piece {
        original_index,
        variants,
        area: poly.area,
    }
}

fn stock_bounds(stock: &Polygon) -> (f64, f64) {
    (stock.bbox.width(), stock.bbox.height())
}

fn try_place_in_stock(
    stock_poly: &Polygon,
    stock_state: &mut StockState,
    piece: &Piece,
    stock_index: usize,
    coarse_only: bool,
) -> Option<Placement> {
    let (stock_w, stock_h) = stock_bounds(stock_poly);
    let coarse_positions = if coarse_only { 16 } else { 64 };

    let mut variant_order: Vec<&OrientedPiece> = piece.variants.iter().collect();
    variant_order.sort_by(|a, b| {
        let da = (a.width * a.height)
            .partial_cmp(&(b.width * b.height))
            .unwrap_or(Ordering::Equal);
        da
    });

    for variant in variant_order {
        if variant.width > stock_w + EPS || variant.height > stock_h + EPS {
            continue;
        }

        for shelf_idx in 0..stock_state.skyline.len() {
            let shelf = stock_state.skyline[shelf_idx];
            if variant.height <= shelf.height + EPS && shelf.x + variant.width <= stock_w + EPS {
                let candidate = Point::new(shelf.x, shelf.y);
                if let Some(pl) = validate_candidate(stock_poly, stock_state, variant, candidate, stock_index) {
                    stock_state.skyline[shelf_idx].x += variant.width;
                    return Some(pl);
                }
            }
        }

        let current_top = stock_state
            .skyline
            .iter()
            .map(|s| s.y + s.height)
            .fold(0.0, f64::max);
        if current_top + variant.height <= stock_h + EPS {
            let candidate = Point::new(0.0, current_top);
            if let Some(pl) = validate_candidate(stock_poly, stock_state, variant, candidate, stock_index) {
                stock_state.skyline.push(Shelf {
                    y: current_top,
                    height: variant.height,
                    x: variant.width,
                });
                return Some(pl);
            }
        }

        let x_steps = ((stock_w - variant.width).max(0.0) / variant.width.max(1.0)).ceil() as usize + 1;
        let y_steps = ((stock_h - variant.height).max(0.0) / variant.height.max(1.0)).ceil() as usize + 1;
        let x_steps = x_steps.min(coarse_positions);
        let y_steps = y_steps.min(coarse_positions);

        for yi in 0..y_steps {
            let y = if y_steps <= 1 {
                0.0
            } else {
                (stock_h - variant.height) * yi as f64 / (y_steps - 1) as f64
            };
            for xi in 0..x_steps {
                let x = if x_steps <= 1 {
                    0.0
                } else {
                    (stock_w - variant.width) * xi as f64 / (x_steps - 1) as f64
                };
                if let Some(pl) = validate_candidate(stock_poly, stock_state, variant, Point::new(x, y), stock_index) {
                    return Some(pl);
                }
            }
        }
    }
    None
}

fn validate_candidate(
    stock_poly: &Polygon,
    stock_state: &mut StockState,
    variant: &OrientedPiece,
    candidate: Point,
    stock_index: usize,
) -> Option<Placement> {
    let pts = transform_polygon(&variant.polygon.pts, 0.0, candidate);
    if !polygon_inside(&stock_poly.pts, &pts) {
        return None;
    }
    let poly = Polygon::new(pts);
    for occ in &stock_state.occupied {
        if occ.bbox.intersects(&poly.bbox) && polygons_overlap(&occ.poly.pts, &poly.pts) {
            return None;
        }
    }
    stock_state.occupied.push(Occupied {
        bbox: poly.bbox,
        poly: poly.clone(),
    });
    Some(Placement {
        stock_index,
        translation: candidate,
        rotation: variant.angle,
        placed_polygon: poly,
    })
}

fn solve(stock: Polygon, raw_pieces: Vec<Polygon>) -> Vec<Placement> {
    let np = raw_pieces.len();
    let large_mode = np > 200 || raw_pieces.iter().map(|p| p.pts.len()).sum::<usize>() > 20_000;
    let mut pieces: Vec<Piece> = raw_pieces
        .into_iter()
        .enumerate()
        .map(|(i, p)| build_piece(p, i, large_mode))
        .collect();

    pieces.sort_by(|a, b| {
        b.area
            .partial_cmp(&a.area)
            .unwrap_or(Ordering::Equal)
            .then_with(|| {
                let ab = a.variants[0].width.max(a.variants[0].height);
                let bb = b.variants[0].width.max(b.variants[0].height);
                bb.partial_cmp(&ab).unwrap_or(Ordering::Equal)
            })
    });

    let mut placements: Vec<Option<Placement>> = vec![None; np];
    let mut stocks: Vec<StockState> = Vec::new();
    let coarse_only = large_mode;

    for piece in &pieces {
        let mut placed = None;
        for stock_index in 0..stocks.len() {
            if let Some(pl) = try_place_in_stock(&stock, &mut stocks[stock_index], piece, stock_index, coarse_only) {
                placed = Some(pl);
                break;
            }
        }
        if placed.is_none() {
            let mut new_stock = StockState {
                occupied: Vec::new(),
                skyline: Vec::new(),
            };
            let stock_index = stocks.len();
            if let Some(pl) = try_place_in_stock(&stock, &mut new_stock, piece, stock_index, coarse_only) {
                stocks.push(new_stock);
                placed = Some(pl);
            }
        }

        if let Some(final_placement) = placed {
            placements[piece.original_index] = Some(final_placement);
        } else {
            let stock_index = stocks.len();
            let mut placed_any = None;
            for variant in &piece.variants {
                let pts = transform_polygon(&variant.polygon.pts, 0.0, Point::new(0.0, 0.0));
                if polygon_inside(&stock.pts, &pts) {
                    let poly = Polygon::new(pts);
                    let new_stock = StockState {
                        occupied: vec![Occupied {
                            bbox: poly.bbox,
                            poly: poly.clone(),
                        }],
                        skyline: vec![Shelf {
                            y: 0.0,
                            height: variant.height,
                            x: variant.width,
                        }],
                    };
                    stocks.push(new_stock);
                    placed_any = Some(Placement {
                        stock_index,
                        translation: Point::new(0.0, 0.0),
                        rotation: variant.angle,
                        placed_polygon: poly,
                    });
                    break;
                }
            }
            let placement = placed_any.expect("piece does not fit inside stock polygon in any supported orientation");
            placements[piece.original_index] = Some(placement);
        }
    }

    placements.into_iter().map(|p| p.unwrap()).collect()
}

fn read_input() -> (Polygon, Vec<Polygon>) {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_pts.push(Point::new(x, y));
    }

    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Point::new(x, y));
        }
        pieces.push(Polygon::new(pts));
    }

    (Polygon::new(stock_pts), pieces)
}

fn main() {
    let (stock, pieces) = read_input();
    let placements = solve(stock, pieces);
    let num_stocks = placements
        .iter()
        .map(|p| p.stock_index)
        .max()
        .map(|m| m + 1)
        .unwrap_or(0);
    println!("{}", num_stocks);
    for pl in placements {
        let _ = pl.placed_polygon.bbox.width();
        println!(
            "{} {:.9} {:.9} {:.9}",
            pl.stock_index, pl.translation.x, pl.translation.y, pl.rotation
        );
    }
}
