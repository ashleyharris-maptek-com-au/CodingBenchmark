use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    hole_id: String,
    sample_idx: usize,
    point: [f64; 3],
    values: Vec<Option<f64>>,
}

#[derive(Clone)]
struct PropertyStats {
    median: f64,
    mad_sigma: f64,
    q1: f64,
    q3: f64,
}

#[derive(Clone)]
struct GridEntry {
    point: [f64; 3],
    sample_idx: usize,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn parse_f64(token: Option<&str>) -> Option<f64> {
    let s = token?;
    if s.eq_ignore_ascii_case("na")
        || s.eq_ignore_ascii_case("nan")
        || s.eq_ignore_ascii_case("null")
        || s.eq_ignore_ascii_case("missing")
        || s == "-"
    {
        return None;
    }
    match s.parse::<f64>() {
        Ok(v) if v.is_finite() => Some(v),
        _ => None,
    }
}

fn quantile(sorted: &[f64], p: f64) -> f64 {
    if sorted.is_empty() {
        return 0.0;
    }
    if sorted.len() == 1 {
        return sorted[0];
    }
    let pos = p.clamp(0.0, 1.0) * (sorted.len() - 1) as f64;
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        sorted[lo]
    } else {
        let t = pos - lo as f64;
        sorted[lo] * (1.0 - t) + sorted[hi] * t
    }
}

fn median(sorted: &[f64]) -> f64 {
    quantile(sorted, 0.5)
}

fn compute_property_stats(samples: &[Sample], prop_count: usize) -> Vec<PropertyStats> {
    let mut stats = Vec::with_capacity(prop_count);
    for p in 0..prop_count {
        let mut vals: Vec<f64> = samples.iter().filter_map(|s| s.values[p]).collect();
        vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        if vals.is_empty() {
            stats.push(PropertyStats {
                median: 0.0,
                mad_sigma: 1.0,
                q1: 0.0,
                q3: 0.0,
            });
            continue;
        }
        let med = median(&vals);
        let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
        devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        let mad = median(&devs);
        let q1 = quantile(&vals, 0.25);
        let q3 = quantile(&vals, 0.75);
        let iqr_sigma = ((q3 - q1) / 1.349).abs();
        let mad_sigma = (mad * 1.4826).max(iqr_sigma * 0.5).max(1e-9);
        stats.push(PropertyStats {
            median: med,
            mad_sigma,
            q1,
            q3,
        });
    }
    stats
}

fn dist2(a: [f64; 3], b: [f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn nearest_neighbors_bruteforce(samples: &[Sample], idx: usize, k: usize, radius2: f64) -> Vec<usize> {
    let p = samples[idx].point;
    let mut candidates: Vec<(f64, usize)> = Vec::new();
    for j in 0..samples.len() {
        if j == idx {
            continue;
        }
        let d2 = dist2(p, samples[j].point);
        if d2 <= radius2 {
            candidates.push((d2, j));
        }
    }
    candidates.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
    candidates.into_iter().take(k).map(|x| x.1).collect()
}

fn grid_key(point: [f64; 3], cell: f64) -> (i32, i32, i32) {
    (
        (point[0] / cell).floor() as i32,
        (point[1] / cell).floor() as i32,
        (point[2] / cell).floor() as i32,
    )
}

fn build_grid(samples: &[Sample], cell: f64) -> HashMap<(i32, i32, i32), Vec<GridEntry>> {
    let mut grid: HashMap<(i32, i32, i32), Vec<GridEntry>> = HashMap::new();
    for (i, s) in samples.iter().enumerate() {
        grid.entry(grid_key(s.point, cell))
            .or_default()
            .push(GridEntry {
                point: s.point,
                sample_idx: i,
            });
    }
    grid
}

fn nearest_neighbors_grid(
    samples: &[Sample],
    idx: usize,
    grid: &HashMap<(i32, i32, i32), Vec<GridEntry>>,
    cell: f64,
    k: usize,
    radius2: f64,
) -> Vec<usize> {
    let p = samples[idx].point;
    let base = grid_key(p, cell);
    let reach = ((radius2.sqrt() / cell).ceil() as i32).max(1);
    let mut candidates: Vec<(f64, usize)> = Vec::new();
    for dx in -reach..=reach {
        for dy in -reach..=reach {
            for dz in -reach..=reach {
                let key = (base.0 + dx, base.1 + dy, base.2 + dz);
                if let Some(bucket) = grid.get(&key) {
                    for item in bucket {
                        if item.sample_idx == idx {
                            continue;
                        }
                        let d2 = dist2(p, item.point);
                        if d2 <= radius2 {
                            candidates.push((d2, item.sample_idx));
                        }
                    }
                }
            }
        }
    }
    candidates.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
    candidates.truncate(k);
    candidates.into_iter().map(|x| x.1).collect()
}

fn robust_local_score(target: f64, neighbor_values: &[f64], global: &PropertyStats) -> Option<f64> {
    if neighbor_values.len() < 3 {
        return None;
    }
    let mut vals = neighbor_values.to_vec();
    vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median(&vals);
    let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let local_sigma = (median(&devs) * 1.4826).max(global.mad_sigma * 0.2).max(1e-9);
    let diff = (target - med).abs();
    let local_z = diff / local_sigma;
    let global_z = (target - global.median).abs() / global.mad_sigma.max(1e-9);
    let iqr = (global.q3 - global.q1).abs();
    let distribution_bonus = if iqr > 0.0 && (target < global.q1 - 3.0 * iqr || target > global.q3 + 3.0 * iqr) {
        0.75
    } else {
        0.0
    };
    Some(local_z * 0.8 + global_z * 0.2 + distribution_bonus)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut lines = input.lines();

    let header = match lines.next() {
        Some(h) => h,
        None => {
            println!("0");
            return;
        }
    };
    let mut header_it = header.split_whitespace();
    let num_holes = header_it.next().and_then(|x| x.parse::<usize>().ok()).unwrap_or(0);
    let num_properties = header_it.next().and_then(|x| x.parse::<usize>().ok()).unwrap_or(0);
    if num_holes == 0 || num_properties == 0 {
        println!("0");
        return;
    }

    let property_names: Vec<String> = match lines.next() {
        Some(line) => line
            .split_whitespace()
            .take(num_properties)
            .map(|s| s.to_string())
            .collect(),
        None => {
            println!("0");
            return;
        }
    };
    if property_names.len() != num_properties {
        println!("0");
        return;
    }

    let mut samples: Vec<Sample> = Vec::new();
    let mut hole_count = 0usize;
    let mut duplicate_guard: HashMap<(String, i64), usize> = HashMap::new();

    while hole_count < num_holes {
        let hole_line = match lines.next() {
            Some(line) if !line.trim().is_empty() => line,
            Some(_) => continue,
            None => break,
        };
        let parts: Vec<&str> = hole_line.split_whitespace().collect();
        if parts.len() < 9 {
            break;
        }
        let hole_id = parts[0].to_string();
        let sx = parse_f64(parts.get(1).copied()).unwrap_or(0.0);
        let sy = parse_f64(parts.get(2).copied()).unwrap_or(0.0);
        let sz = parse_f64(parts.get(3).copied()).unwrap_or(0.0);
        let dx = parse_f64(parts.get(4).copied()).unwrap_or(0.0);
        let dy = parse_f64(parts.get(5).copied()).unwrap_or(0.0);
        let dz = parse_f64(parts.get(6).copied()).unwrap_or(1.0);
        let length = parse_f64(parts.get(7).copied()).unwrap_or(0.0).max(0.0);
        let num_samples = parts.get(8).and_then(|x| x.parse::<usize>().ok()).unwrap_or(0);
        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        let dir = if norm > 0.0 {
            [dx / norm, dy / norm, dz / norm]
        } else {
            [0.0, 0.0, 1.0]
        };

        for sample_idx in 0..num_samples {
            let sample_line = match lines.next() {
                Some(line) => line,
                None => break,
            };
            let cols: Vec<&str> = sample_line.split_whitespace().collect();
            if cols.is_empty() {
                continue;
            }
            let mut depth = parse_f64(cols.first().copied()).unwrap_or(0.0);
            if !depth.is_finite() {
                depth = 0.0;
            }
            depth = depth.clamp(0.0, length);
            let point = [
                sx + dir[0] * depth,
                sy + dir[1] * depth,
                sz + dir[2] * depth,
            ];
            let mut values = Vec::with_capacity(num_properties);
            for p in 0..num_properties {
                values.push(parse_f64(cols.get(p + 1).copied()));
            }

            let dedup_key = (hole_id.clone(), (depth * 1000.0).round() as i64);
            if duplicate_guard.insert(dedup_key, samples.len()).is_some() {
                continue;
            }

            samples.push(Sample {
                hole_idx: hole_count,
                hole_id: hole_id.clone(),
                sample_idx,
                point,
                values,
            });
        }

        hole_count += 1;
    }

    if samples.len() < 4 {
        println!("0");
        return;
    }

    let total_samples = samples.len();
    let stats = compute_property_stats(&samples, num_properties);

    let mut bbox_min = [f64::INFINITY; 3];
    let mut bbox_max = [f64::NEG_INFINITY; 3];
    for s in &samples {
        for a in 0..3 {
            bbox_min[a] = bbox_min[a].min(s.point[a]);
            bbox_max[a] = bbox_max[a].max(s.point[a]);
        }
    }
    let extent = [
        (bbox_max[0] - bbox_min[0]).max(1.0),
        (bbox_max[1] - bbox_min[1]).max(1.0),
        (bbox_max[2] - bbox_min[2]).max(1.0),
    ];
    let volume = extent[0] * extent[1] * extent[2];
    let density = (total_samples as f64 / volume).max(1e-12);
    let target_neighbors = if total_samples < 5_000 {
        12usize
    } else if total_samples < 50_000 {
        10usize
    } else {
        8usize
    };
    let radius = ((target_neighbors as f64 / density).cbrt() * 1.75).max(1.0);
    let radius2 = radius * radius;
    let use_grid = total_samples > 12_000;
    let cell = radius.max(1.0);
    let grid = if use_grid {
        Some(build_grid(&samples, cell))
    } else {
        None
    };

    let mut suspects: Vec<Suspect> = Vec::new();
    for i in 0..samples.len() {
        let neighbors = if let Some(ref g) = grid {
            nearest_neighbors_grid(&samples, i, g, cell, target_neighbors, radius2)
        } else {
            nearest_neighbors_bruteforce(&samples, i, target_neighbors, radius2)
        };
        if neighbors.len() < 3 {
            continue;
        }

        for p in 0..num_properties {
            let target = match samples[i].values[p] {
                Some(v) => v,
                None => continue,
            };
            let mut neighbor_values = Vec::with_capacity(neighbors.len());
            for &j in &neighbors {
                if let Some(v) = samples[j].values[p] {
                    neighbor_values.push(v);
                }
            }
            let Some(score) = robust_local_score(target, &neighbor_values, &stats[p]) else {
                continue;
            };

            let same_hole_penalty = neighbors
                .iter()
                .filter(|&&j| samples[j].hole_idx == samples[i].hole_idx)
                .count() as f64
                / neighbors.len() as f64;
            let adjusted = score * (1.0 - 0.15 * same_hole_penalty);
            let threshold = if total_samples < 2000 { 5.0 } else { 4.5 };
            if adjusted >= threshold {
                let confidence = (1.0 - (-adjusted / 3.0).exp()).clamp(0.0, 0.999999);
                suspects.push(Suspect {
                    hole_id: samples[i].hole_id.clone(),
                    sample_idx: samples[i].sample_idx,
                    property_name: property_names[p].clone(),
                    confidence,
                });
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    let max_output = if total_samples < 20_000 {
        suspects.len()
    } else {
        suspects.len().min(total_samples / 10 + 2000)
    };
    println!("{}", max_output);
    for s in suspects.into_iter().take(max_output) {
        println!("{} {} {} {:.6}", s.hole_id, s.sample_idx, s.property_name, s.confidence);
    }
}
