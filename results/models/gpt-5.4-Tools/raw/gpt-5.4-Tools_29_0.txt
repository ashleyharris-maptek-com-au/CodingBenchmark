using System;
using System.IO;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var degree = new int[n];
        var edgesU = new int[m];
        var edgesV = new int[m];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            edgesU[i] = u;
            edgesV[i] = v;
            if (u == v) continue;
            degree[u]++;
            degree[v]++;
        }

        var adj = new int[n][];
        for (int i = 0; i < n; i++) adj[i] = new int[degree[i]];
        Array.Clear(degree, 0, n);
        for (int i = 0; i < m; i++)
        {
            int u = edgesU[i];
            int v = edgesV[i];
            if (u == v) continue;
            adj[u][degree[u]++] = v;
            adj[v][degree[v]++] = u;
        }
        for (int i = 0; i < n; i++) Array.Sort(adj[i]);

        int[] order = Enumerable.Range(0, n).ToArray();
        Array.Sort(order, (a, b) =>
        {
            int cmp = adj[b].Length.CompareTo(adj[a].Length);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        var solver = new HybridCliqueSolver(adj, order);
        int[] clique = solver.Solve();

        Array.Sort(clique);
        Console.WriteLine(clique.Length);
        if (clique.Length > 0)
            Console.WriteLine(string.Join(" ", clique));
        else
            Console.WriteLine();
    }
}

internal sealed class HybridCliqueSolver
{
    private readonly int[][] _adj;
    private readonly int[] _order;
    private readonly int _n;
    private readonly long _deadline;
    private volatile int[] _bestClique;
    private volatile int _bestSize;

    public HybridCliqueSolver(int[][] adj, int[] order)
    {
        _adj = adj;
        _order = order;
        _n = adj.Length;
        _deadline = Environment.TickCount64 + 280000;
        _bestClique = Array.Empty<int>();
        _bestSize = 0;
    }

    public int[] Solve()
    {
        GreedyWarmStarts();

        if (_n <= 6000)
            CoreExactSearch();
        else
            LargeGraphSearch();

        return _bestClique;
    }

    private void GreedyWarmStarts()
    {
        int tries = Math.Min(_n, _n <= 2000 ? 128 : 64);
        for (int t = 0; t < tries; t++)
        {
            int seed = _order[t];
            var clique = GreedyCliqueFromSeed(seed, t % 4);
            UpdateBest(clique);
            if (TimeExpired()) return;
        }

        var rnd = new XorShift64((ulong)(_n * 11995408973635179863UL + 17UL));
        int randomTries = Math.Min(_n, _n <= 50000 ? 128 : 48);
        for (int t = 0; t < randomTries; t++)
        {
            int idx = (int)(rnd.NextUInt() % (uint)_n);
            var clique = GreedyCliqueFromSeed(_order[idx], 4 + (t % 3));
            UpdateBest(clique);
            if (TimeExpired()) return;
        }
    }

    private void CoreExactSearch()
    {
        int limit = _n <= 400 ? _n : Math.Min(_n, 2500);
        var core = new int[limit];
        Array.Copy(_order, core, limit);

        var candidates = new List<int>(limit);
        for (int i = 0; i < limit; i++)
        {
            int v = core[i];
            if (_adj[v].Length + 1 <= _bestSize) break;
            candidates.Add(v);
        }

        if (candidates.Count == 0) return;

        var search = new ExactSearch(_adj, _deadline, GetBestSize, UpdateBest);
        search.Run(candidates.ToArray());
    }

    private void LargeGraphSearch()
    {
        int frontier = Math.Min(_n, Math.Max(2000, Math.Min(30000, 500 * (_bestSize + 8))));
        int workers = Math.Min(Environment.ProcessorCount, 8);
        Parallel.For(0, workers, new ParallelOptions { MaxDegreeOfParallelism = workers }, workerId =>
        {
            var local = new List<int>(128);
            for (int i = workerId; i < frontier && !TimeExpired(); i += workers)
            {
                int v = _order[i];
                if (_adj[v].Length + 1 <= GetBestSize()) break;

                local.Clear();
                local.Add(v);
                int candidateCap = Math.Min(_adj[v].Length, Math.Max(256, 32 * (GetBestSize() + 4)));
                var cand = BuildRestrictedCandidates(v, candidateCap);
                ExpandHeuristic(local, cand, 0);

                if (cand.Count <= 140 && local.Count + cand.Count > GetBestSize())
                {
                    var exact = new ExactSearch(_adj, _deadline, GetBestSize, UpdateBest);
                    exact.RunWithPrefix(local.ToArray(), cand.ToArray());
                }
            }
        });
    }

    private void ExpandHeuristic(List<int> clique, List<int> candidates, int depth)
    {
        if (TimeExpired() || candidates.Count == 0)
        {
            UpdateBest(clique);
            return;
        }
        UpdateBest(clique);
        if (clique.Count + candidates.Count <= GetBestSize()) return;

        candidates.Sort((a, b) => Score(b).CompareTo(Score(a)));
        int branch = depth == 0 ? Math.Min(12, candidates.Count) : Math.Min(4, candidates.Count);
        for (int i = 0; i < branch && !TimeExpired(); i++)
        {
            int v = candidates[i];
            var nextClique = new List<int>(clique.Count + 1);
            nextClique.AddRange(clique);
            nextClique.Add(v);
            var nextCand = new List<int>();
            for (int j = i + 1; j < candidates.Count; j++)
            {
                int u = candidates[j];
                if (HasEdge(u, v)) nextCand.Add(u);
            }
            ExpandHeuristic(nextClique, nextCand, depth + 1);
        }
    }

    private List<int> BuildRestrictedCandidates(int seed, int cap)
    {
        var cand = new List<int>(Math.Min(cap, _adj[seed].Length));
        foreach (int u in _adj[seed])
        {
            if (_adj[u].Length + 1 <= GetBestSize()) continue;
            cand.Add(u);
        }
        cand.Sort((a, b) => Score(b).CompareTo(Score(a)));
        if (cand.Count > cap) cand.RemoveRange(cap, cand.Count - cap);
        return cand;
    }

    private int[] GreedyCliqueFromSeed(int seed, int mode)
    {
        var clique = new List<int> { seed };
        var candidates = new List<int>(_adj[seed].Length);
        foreach (int u in _adj[seed])
        {
            if (_adj[u].Length + 1 > _bestSize) candidates.Add(u);
        }

        while (candidates.Count > 0 && !TimeExpired())
        {
            int bestIndex = 0;
            long bestScore = long.MinValue;
            for (int i = 0; i < candidates.Count; i++)
            {
                int v = candidates[i];
                long s = mode switch
                {
                    0 => ((long)_adj[v].Length << 32) - v,
                    1 => ((long)CommonCountEstimate(v, candidates) << 32) + _adj[v].Length,
                    2 => ((long)_adj[v].Length << 20) + CommonCountEstimate(v, candidates),
                    3 => ((long)CommonCountEstimate(v, candidates) << 20) + _adj[v].Length,
                    _ => ((long)_adj[v].Length << 32) + (CommonCountEstimate(v, candidates) ^ (v * 17)),
                };
                if (s > bestScore)
                {
                    bestScore = s;
                    bestIndex = i;
                }
            }

            int chosen = candidates[bestIndex];
            clique.Add(chosen);

            var next = new List<int>();
            for (int i = 0; i < candidates.Count; i++)
            {
                if (i == bestIndex) continue;
                int u = candidates[i];
                if (HasEdge(u, chosen)) next.Add(u);
            }
            candidates = next;
        }

        return clique.ToArray();
    }

    private int CommonCountEstimate(int v, List<int> set)
    {
        int count = 0;
        int[] neigh = _adj[v];
        for (int i = 0; i < set.Count; i++)
        {
            if (set[i] != v && BinarySearch(neigh, set[i])) count++;
        }
        return count;
    }

    private int Score(int v) => _adj[v].Length;

    private bool HasEdge(int a, int b) => BinarySearch(_adj[a], b);

    private static bool BinarySearch(int[] arr, int value)
    {
        int lo = 0, hi = arr.Length - 1;
        while (lo <= hi)
        {
            int mid = lo + ((hi - lo) >> 1);
            int cur = arr[mid];
            if (cur == value) return true;
            if (cur < value) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    private bool TimeExpired() => Environment.TickCount64 >= _deadline;

    private int GetBestSize() => Volatile.Read(ref _bestSize);

    private void UpdateBest(IReadOnlyList<int> clique)
    {
        if (clique.Count <= GetBestSize()) return;
        lock (this)
        {
            if (clique.Count <= _bestSize) return;
            var copy = new int[clique.Count];
            for (int i = 0; i < clique.Count; i++) copy[i] = clique[i];
            _bestClique = copy;
            _bestSize = copy.Length;
        }
    }
}

internal sealed class ExactSearch
{
    private readonly int[][] _adj;
    private readonly long _deadline;
    private readonly Func<int> _getBestSize;
    private readonly Action<IReadOnlyList<int>> _updateBest;
    private readonly List<int> _current = new List<int>();

    public ExactSearch(int[][] adj, long deadline, Func<int> getBestSize, Action<IReadOnlyList<int>> updateBest)
    {
        _adj = adj;
        _deadline = deadline;
        _getBestSize = getBestSize;
        _updateBest = updateBest;
    }

    public void Run(int[] vertices)
    {
        Array.Sort(vertices, (a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
        Expand(new List<int>(vertices));
    }

    public void RunWithPrefix(int[] prefix, int[] vertices)
    {
        _current.AddRange(prefix);
        _updateBest(_current);
        Array.Sort(vertices, (a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
        Expand(new List<int>(vertices));
        _current.Clear();
    }

    private void Expand(List<int> candidates)
    {
        if (Environment.TickCount64 >= _deadline) return;
        if (_current.Count + candidates.Count <= _getBestSize()) return;

        int[] colors = new int[candidates.Count];
        ColorSort(candidates, colors);

        for (int i = candidates.Count - 1; i >= 0; i--)
        {
            if (_current.Count + colors[i] <= _getBestSize()) return;
            int v = candidates[i];
            _current.Add(v);

            var next = new List<int>();
            for (int j = 0; j < i; j++)
            {
                int u = candidates[j];
                if (HasEdge(u, v)) next.Add(u);
            }

            if (next.Count == 0)
                _updateBest(_current);
            else
                Expand(next);

            _current.RemoveAt(_current.Count - 1);
        }
    }

    private void ColorSort(List<int> vertices, int[] colors)
    {
        vertices.Sort((a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
        int n = vertices.Count;
        var colorSets = new List<List<int>>();
        for (int i = 0; i < n; i++)
        {
            int v = vertices[i];
            int c = 0;
            while (c < colorSets.Count && ConflictsWithAny(v, colorSets[c])) c++;
            if (c == colorSets.Count) colorSets.Add(new List<int>());
            colorSets[c].Add(v);
            colors[i] = c + 1;
        }

        int idx = 0;
        for (int c = 0; c < colorSets.Count; c++)
        {
            foreach (int v in colorSets[c])
            {
                vertices[idx] = v;
                colors[idx] = c + 1;
                idx++;
            }
        }
    }

    private bool ConflictsWithAny(int v, List<int> colorClass)
    {
        for (int i = 0; i < colorClass.Count; i++)
        {
            if (HasEdge(v, colorClass[i])) return true;
        }
        return false;
    }

    private bool HasEdge(int a, int b)
    {
        int[] arr = _adj[a];
        int lo = 0, hi = arr.Length - 1;
        while (lo <= hi)
        {
            int mid = lo + ((hi - lo) >> 1);
            int cur = arr[mid];
            if (cur == b) return true;
            if (cur < b) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len, _ptr;

    public FastScanner(Stream stream) => _stream = stream;

    public int NextInt()
    {
        int c;
        do c = Read(); while (c <= 32 && c >= 0);
        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }
        int val = 0;
        while (c > 32)
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return val * sign;
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len <= 0) return -1;
        }
        return _buffer[_ptr++];
    }
}

internal struct XorShift64
{
    private ulong _x;
    public XorShift64(ulong seed) => _x = seed == 0 ? 88172645463393265UL : seed;
    public uint NextUInt()
    {
        _x ^= _x << 7;
        _x ^= _x >> 9;
        return (uint)_x;
    }
}
