using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    private sealed class Graph
    {
        public readonly int N;
        public readonly int M;
        public readonly int[][] Adj;
        public readonly int[] Degree;
        public readonly HashSet<int>[] NeighborSets;

        public Graph(int n, List<int>[] adjacency, int m)
        {
            N = n;
            M = m;
            Adj = new int[n][];
            Degree = new int[n];
            NeighborSets = new HashSet<int>[n];
            for (int i = 0; i < n; i++)
            {
                var arr = adjacency[i].Distinct().ToArray();
                Array.Sort(arr);
                Adj[i] = arr;
                Degree[i] = arr.Length;
                NeighborSets[i] = new HashSet<int>(arr);
            }
        }

        public bool HasEdge(int u, int v)
        {
            if (u == v) return false;
            if (Degree[u] < Degree[v]) return NeighborSets[u].Contains(v);
            return NeighborSets[v].Contains(u);
        }
    }

    private sealed class MaxCliqueSolver
    {
        private readonly Graph _g;
        private readonly Stopwatch _timer;
        private readonly long _timeLimitMs;
        private readonly object _bestLock = new object();
        private int[] _bestClique = Array.Empty<int>();

        public MaxCliqueSolver(Graph g, long timeLimitMs)
        {
            _g = g;
            _timeLimitMs = timeLimitMs;
            _timer = Stopwatch.StartNew();
        }

        public int[] Solve()
        {
            if (_g.N == 0) return Array.Empty<int>();

            var greedy = BuildGreedyClique(OrderVerticesForGreedy());
            UpdateBest(greedy);

            int[] core = ComputeCoreNumbers();
            int lowerBound = BestSize;

            var active = new List<int>();
            for (int v = 0; v < _g.N; v++)
            {
                if (_g.Degree[v] + 1 >= lowerBound && core[v] + 1 >= lowerBound)
                    active.Add(v);
            }

            if (active.Count == 0)
                return BestClique;

            active.Sort((a, b) =>
            {
                int cmp = _g.Degree[b].CompareTo(_g.Degree[a]);
                if (cmp != 0) return cmp;
                return a.CompareTo(b);
            });

            if (ShouldRunExact(active.Count))
            {
                RunExactSearch(active, core);
            }
            else
            {
                RunHeuristicsOnly(active);
            }

            return BestClique;
        }

        private int BestSize
        {
            get
            {
                lock (_bestLock) return _bestClique.Length;
            }
        }

        private int[] BestClique
        {
            get
            {
                lock (_bestLock) return (int[])_bestClique.Clone();
            }
        }

        private bool TimeExceeded(long reserveMs = 0) => _timer.ElapsedMilliseconds >= _timeLimitMs - reserveMs;

        private void UpdateBest(List<int> clique)
        {
            if (clique == null) return;
            lock (_bestLock)
            {
                if (clique.Count > _bestClique.Length)
                    _bestClique = clique.ToArray();
            }
        }

        private void UpdateBest(int[] clique)
        {
            if (clique == null) return;
            lock (_bestLock)
            {
                if (clique.Length > _bestClique.Length)
                    _bestClique = (int[])clique.Clone();
            }
        }

        private int[] ComputeCoreNumbers()
        {
            int n = _g.N;
            int maxDeg = 0;
            for (int i = 0; i < n; i++) maxDeg = Math.Max(maxDeg, _g.Degree[i]);

            int[] degree = (int[])_g.Degree.Clone();
            int[] bin = new int[maxDeg + 1];
            for (int i = 0; i < n; i++) bin[degree[i]]++;

            int start = 0;
            for (int d = 0; d <= maxDeg; d++)
            {
                int num = bin[d];
                bin[d] = start;
                start += num;
            }

            int[] pos = new int[n];
            int[] vert = new int[n];
            for (int v = 0; v < n; v++)
            {
                pos[v] = bin[degree[v]];
                vert[pos[v]] = v;
                bin[degree[v]]++;
            }

            for (int d = maxDeg; d >= 1; d--) bin[d] = bin[d - 1];
            bin[0] = 0;

            int[] core = new int[n];
            for (int i = 0; i < n; i++)
            {
                int v = vert[i];
                core[v] = degree[v];
                foreach (int u in _g.Adj[v])
                {
                    if (degree[u] > degree[v])
                    {
                        int du = degree[u];
                        int pu = pos[u];
                        int pw = bin[du];
                        int w = vert[pw];
                        if (u != w)
                        {
                            vert[pu] = w;
                            vert[pw] = u;
                            pos[u] = pw;
                            pos[w] = pu;
                        }
                        bin[du]++;
                        degree[u]--;
                    }
                }
            }
            return core;
        }

        private int[] OrderVerticesForGreedy()
        {
            return Enumerable.Range(0, _g.N)
                .OrderByDescending(v => _g.Degree[v])
                .ThenBy(v => v)
                .ToArray();
        }

        private List<int> BuildGreedyClique(IEnumerable<int> orderedVertices)
        {
            var clique = new List<int>();
            foreach (int v in orderedVertices)
            {
                bool ok = true;
                for (int i = 0; i < clique.Count; i++)
                {
                    if (!_g.HasEdge(v, clique[i]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) clique.Add(v);
            }
            return clique;
        }

        private List<int> BuildGreedyCliqueFromSeed(int seed, int[] neighbors)
        {
            var order = neighbors
                .Where(v => _g.HasEdge(seed, v))
                .OrderByDescending(v => _g.Degree[v])
                .ThenBy(v => v);

            var clique = new List<int> { seed };
            foreach (int v in order)
            {
                bool ok = true;
                for (int i = 0; i < clique.Count; i++)
                {
                    if (clique[i] != v && !_g.HasEdge(v, clique[i]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) clique.Add(v);
            }
            return clique;
        }

        private bool ShouldRunExact(int activeCount)
        {
            if (activeCount <= 4000) return true;
            if (_g.M <= 2_000_000 && activeCount <= 12000) return true;
            double density = _g.N <= 1 ? 0.0 : (2.0 * _g.M) / (_g.N * (double)(_g.N - 1));
            if (density >= 0.90 && activeCount <= 20000) return true;
            return false;
        }

        private void RunHeuristicsOnly(List<int> active)
        {
            int seeds = Math.Min(active.Count, Math.Max(64, Environment.ProcessorCount * 32));
            var seedVertices = active.Take(seeds).ToArray();

            Parallel.ForEach(seedVertices, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, seed =>
            {
                if (TimeExceeded(100)) return;
                var clique = BuildGreedyCliqueFromSeed(seed, _g.Adj[seed]);
                UpdateBest(clique);
            });

            if (_g.N <= 50000 && !TimeExceeded(100))
            {
                var byMinMissing = Enumerable.Range(0, _g.N)
                    .OrderBy(v => _g.N - 1 - _g.Degree[v])
                    .ThenByDescending(v => _g.Degree[v])
                    .Take(Math.Min(256, _g.N));

                foreach (int seed in byMinMissing)
                {
                    if (TimeExceeded(100)) break;
                    UpdateBest(BuildGreedyCliqueFromSeed(seed, _g.Adj[seed]));
                }
            }
        }

        private void RunExactSearch(List<int> active, int[] core)
        {
            var vertices = active
                .Where(v => core[v] + 1 >= BestSize)
                .OrderByDescending(v => core[v])
                .ThenByDescending(v => _g.Degree[v])
                .ThenBy(v => v)
                .ToArray();

            int split = Math.Min(vertices.Length, Math.Max(1, Environment.ProcessorCount * 2));
            var tasks = new List<Task>(split);
            using var cts = new CancellationTokenSource();

            for (int idx = 0; idx < split; idx++)
            {
                int root = vertices[idx];
                tasks.Add(Task.Run(() =>
                {
                    if (TimeExceeded(100)) return;
                    var clique = new List<int> { root };
                    var candidates = new List<int>();
                    int best = BestSize;
                    foreach (int u in _g.Adj[root])
                    {
                        if (u > root && core[u] + 1 >= best && _g.Degree[u] + 1 >= best)
                            candidates.Add(u);
                    }

                    candidates.Sort((a, b) =>
                    {
                        int cmp = _g.Degree[b].CompareTo(_g.Degree[a]);
                        if (cmp != 0) return cmp;
                        return a.CompareTo(b);
                    });

                    Expand(clique, candidates, cts.Token);
                }, cts.Token));
            }

            for (int i = split; i < vertices.Length && !TimeExceeded(100); i++)
            {
                int root = vertices[i];
                if (core[root] + 1 <= BestSize) break;
                var clique = new List<int> { root };
                var candidates = new List<int>();
                int best = BestSize;
                foreach (int u in _g.Adj[root])
                {
                    if (u > root && core[u] + 1 >= best && _g.Degree[u] + 1 >= best)
                        candidates.Add(u);
                }
                candidates.Sort((a, b) =>
                {
                    int cmp = _g.Degree[b].CompareTo(_g.Degree[a]);
                    if (cmp != 0) return cmp;
                    return a.CompareTo(b);
                });
                Expand(clique, candidates, cts.Token);
            }

            try { Task.WaitAll(tasks.ToArray()); } catch { }
        }

        private void Expand(List<int> clique, List<int> candidates, CancellationToken token)
        {
            if (token.IsCancellationRequested || TimeExceeded(50)) return;
            if (candidates.Count == 0)
            {
                UpdateBest(clique);
                return;
            }

            ColorSort(candidates, out int[] ordered, out int[] bounds);
            for (int i = ordered.Length - 1; i >= 0; i--)
            {
                if (clique.Count + bounds[i] <= BestSize) return;
                if (token.IsCancellationRequested || TimeExceeded(50)) return;

                int v = ordered[i];
                clique.Add(v);

                var next = new List<int>();
                for (int j = 0; j < i; j++)
                {
                    int u = ordered[j];
                    if (_g.HasEdge(v, u))
                        next.Add(u);
                }

                if (next.Count == 0)
                {
                    UpdateBest(clique);
                }
                else
                {
                    Expand(clique, next, token);
                }

                clique.RemoveAt(clique.Count - 1);
            }
        }

        private void ColorSort(List<int> candidates, out int[] ordered, out int[] bounds)
        {
            int n = candidates.Count;
            var work = candidates
                .OrderByDescending(v => _g.Degree[v])
                .ThenBy(v => v)
                .ToList();

            ordered = new int[n];
            bounds = new int[n];
            var colorClasses = new List<List<int>>();

            int index = 0;
            while (work.Count > 0)
            {
                var remaining = new List<int>(work.Count);
                var colorSet = new List<int>();
                foreach (int v in work)
                {
                    bool fits = true;
                    for (int i = 0; i < colorSet.Count; i++)
                    {
                        if (_g.HasEdge(v, colorSet[i]))
                        {
                            fits = false;
                            break;
                        }
                    }
                    if (fits) colorSet.Add(v);
                    else remaining.Add(v);
                }
                colorClasses.Add(colorSet);
                work = remaining;
            }

            for (int c = 0; c < colorClasses.Count; c++)
            {
                foreach (int v in colorClasses[c])
                {
                    ordered[index] = v;
                    bounds[index] = c + 1;
                    index++;
                }
            }
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == int.MinValue)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }
        int m = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var adjacency = new List<int>[n];
        for (int i = 0; i < n; i++) adjacency[i] = new List<int>();

        int edgesRead = 0;
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u == int.MinValue || v == int.MinValue) break;
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adjacency[u].Add(v);
            adjacency[v].Add(u);
            edgesRead++;
        }

        var graph = new Graph(n, adjacency, edgesRead);
        long timeBudgetMs = 290_000;
        var solver = new MaxCliqueSolver(graph, timeBudgetMs);
        int[] clique = solver.Solve();
        Array.Sort(clique);

        Console.WriteLine(clique.Length);
        if (clique.Length > 0) Console.WriteLine(string.Join(" ", clique));
        else Console.WriteLine();
    }
}
