use std::collections::{HashMap, VecDeque};
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

#[derive(Clone, Copy, Default)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn add(self, other: Point) -> Point {
        Point {
            x: self.x + other.x,
            y: self.y + other.y,
        }
    }

    fn sub(self, other: Point) -> Point {
        Point {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }

    fn scale(self, s: f64) -> Point {
        Point {
            x: self.x * s,
            y: self.y * s,
        }
    }
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }

    fn next_f64(&mut self) -> f64 {
        const DEN: f64 = (1u64 << 53) as f64;
        ((self.next_u64() >> 11) as f64) / DEN
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let m: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);

    if n == 0 {
        return;
    }

    let mut edges = Vec::with_capacity(m);
    let mut adj = vec![Vec::<usize>::new(); n];
    let mut degree = vec![0usize; n];

    for _ in 0..m {
        let u: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        let v: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        edges.push(Edge { u, v });
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
            degree[u] += 1;
            degree[v] += 1;
        }
    }

    let components = connected_components(n, &adj);
    let mut points = vec![Point::default(); n];
    let mut order: Vec<usize> = (0..components.len()).collect();
    order.sort_by_key(|&idx| std::cmp::Reverse(components[idx].len()));

    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_height = 0.0;
    let total_scale = ((n as f64).sqrt() + 1.0) * 40.0;

    for idx in order {
        let comp = &components[idx];
        let local = layout_component(comp, &adj, &degree, &edges);
        let (min_x, max_x, min_y, max_y) = bounds(&local);
        let width = (max_x - min_x).max(1.0);
        let height = (max_y - min_y).max(1.0);
        let spacing = 20.0 + (comp.len() as f64).sqrt();

        if cursor_x + width > total_scale && cursor_x > 0.0 {
            cursor_x = 0.0;
            cursor_y += row_height + spacing;
            row_height = 0.0;
        }

        for (i, &node) in comp.iter().enumerate() {
            points[node] = Point {
                x: local[i].x - min_x + cursor_x,
                y: local[i].y - min_y + cursor_y,
            };
        }

        cursor_x += width + spacing;
        row_height = row_height.max(height);
    }

    normalize(&mut points);

    for p in points {
        println!("{:.6} {:.6}", p.x, p.y);
    }
}

fn connected_components(n: usize, adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let mut seen = vec![false; n];
    let mut comps = Vec::new();
    let mut q = VecDeque::new();

    for start in 0..n {
        if seen[start] {
            continue;
        }
        seen[start] = true;
        q.push_back(start);
        let mut comp = Vec::new();
        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
        comps.push(comp);
    }
    comps
}

fn layout_component(
    comp: &[usize],
    adj: &[Vec<usize>],
    degree: &[usize],
    edges: &[Edge],
) -> Vec<Point> {
    let k = comp.len();
    if k == 1 {
        return vec![Point { x: 0.0, y: 0.0 }];
    }

    let mut index = HashMap::with_capacity(k * 2);
    for (i, &u) in comp.iter().enumerate() {
        index.insert(u, i);
    }

    let mut local_adj = vec![Vec::<usize>::new(); k];
    let mut local_degree = vec![0usize; k];
    let mut local_edges = Vec::new();

    for &u_global in comp {
        let u = index[&u_global];
        for &v_global in &adj[u_global] {
            if let Some(&v) = index.get(&v_global) {
                local_adj[u].push(v);
            }
        }
        local_degree[u] = degree[u_global];
    }

    for &Edge { u, v } in edges {
        if let (Some(&a), Some(&b)) = (index.get(&u), index.get(&v)) {
            local_edges.push(Edge { u: a, v: b });
        }
    }

    let edge_count = local_edges.len();
    let density = if k > 1 {
        2.0 * edge_count as f64 / ((k * (k - 1)) as f64)
    } else {
        0.0
    };

    if edge_count == 0 {
        return isolated_layout(k);
    }

    let mut pos = if k <= 32 {
        bfs_layer_layout(&local_adj, &local_degree)
    } else if k <= 1500 {
        radial_order_layout(&local_adj, &local_degree, density)
    } else {
        large_scale_layout(&local_adj, &local_degree)
    };

    if k <= 600 && edge_count <= 30_000 {
        let iterations = if k <= 80 { 180 } else if k <= 250 { 120 } else { 80 };
        fruchterman_reingold(&local_adj, &mut pos, iterations);
    } else if k <= 6_000 && density < 0.02 {
        fruchterman_reingold_sparse(&local_adj, &mut pos, 24);
    }

    pos
}

fn isolated_layout(k: usize) -> Vec<Point> {
    let cols = (k as f64).sqrt().ceil() as usize;
    let mut pos = vec![Point::default(); k];
    for i in 0..k {
        pos[i] = Point {
            x: (i % cols) as f64 * 20.0,
            y: (i / cols) as f64 * 20.0,
        };
    }
    pos
}

fn bfs_layer_layout(adj: &[Vec<usize>], degree: &[usize]) -> Vec<Point> {
    let n = adj.len();
    let root = (0..n).max_by_key(|&i| degree[i]).unwrap_or(0);
    let levels = bfs_levels(adj, root);
    layered_layout_from_levels(adj, degree, &levels)
}

fn radial_order_layout(adj: &[Vec<usize>], degree: &[usize], density: f64) -> Vec<Point> {
    let n = adj.len();
    let root = (0..n).max_by_key(|&i| degree[i]).unwrap_or(0);
    let levels = bfs_levels(adj, root);
    if density > 0.18 {
        circular_by_degree(degree)
    } else {
        layered_layout_from_levels(adj, degree, &levels)
    }
}

fn large_scale_layout(adj: &[Vec<usize>], degree: &[usize]) -> Vec<Point> {
    let n = adj.len();
    let root = (0..n).max_by_key(|&i| degree[i]).unwrap_or(0);
    let levels = bfs_levels_limit(adj, root, n.min(50_000));
    let mut pos = layered_layout_from_levels(adj, degree, &levels);

    if n > 20_000 {
        let mut rng = XorShift64::new((n as u64) ^ 0x9E37_79B9_7F4A_7C15);
        for p in &mut pos {
            p.x += (rng.next_f64() - 0.5) * 0.05;
            p.y += (rng.next_f64() - 0.5) * 0.05;
        }
    }
    pos
}

fn bfs_levels(adj: &[Vec<usize>], root: usize) -> Vec<usize> {
    bfs_levels_limit(adj, root, usize::MAX)
}

fn bfs_levels_limit(adj: &[Vec<usize>], root: usize, _cap: usize) -> Vec<usize> {
    let n = adj.len();
    let mut dist = vec![usize::MAX; n];
    let mut q = VecDeque::new();
    dist[root] = 0;
    q.push_back(root);

    while let Some(u) = q.pop_front() {
        let du = dist[u];
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = du + 1;
                q.push_back(v);
            }
        }
    }

    let mut fallback_level = 0usize;
    for &d in &dist {
        if d != usize::MAX {
            fallback_level = fallback_level.max(d);
        }
    }
    for d in &mut dist {
        if *d == usize::MAX {
            fallback_level += 1;
            *d = fallback_level;
        }
    }
    dist
}

fn layered_layout_from_levels(adj: &[Vec<usize>], degree: &[usize], levels: &[usize]) -> Vec<Point> {
    let n = adj.len();
    let max_level = *levels.iter().max().unwrap_or(&0);
    let mut buckets = vec![Vec::<usize>::new(); max_level + 1];
    for i in 0..n {
        buckets[levels[i]].push(i);
    }

    let mut bary = vec![0.0f64; n];
    let mut pos_in_prev = HashMap::<usize, f64>::new();
    for level in 0..=max_level {
        if level > 0 {
            for &u in &buckets[level] {
                let mut sum = 0.0;
                let mut cnt = 0.0;
                for &v in &adj[u] {
                    if levels[v] + 1 == level {
                        if let Some(&pv) = pos_in_prev.get(&v) {
                            sum += pv;
                            cnt += 1.0;
                        }
                    }
                }
                bary[u] = if cnt > 0.0 { sum / cnt } else { degree[u] as f64 };
            }
            buckets[level].sort_by(|&a, &b| {
                bary[a]
                    .partial_cmp(&bary[b])
                    .unwrap_or(std::cmp::Ordering::Equal)
                    .then_with(|| degree[b].cmp(&degree[a]))
            });
        } else {
            buckets[level].sort_by_key(|&u| std::cmp::Reverse(degree[u]));
        }
        pos_in_prev.clear();
        for (idx, &u) in buckets[level].iter().enumerate() {
            pos_in_prev.insert(u, idx as f64);
        }
    }

    let mut out = vec![Point::default(); n];
    let y_step = 30.0;
    for (level, layer) in buckets.iter().enumerate() {
        let width = layer.len() as f64;
        let x_step = if layer.len() <= 1 {
            0.0
        } else {
            (20.0 + (level as f64).sqrt()) * (1.0 + width.log2() * 0.15)
        };
        let start_x = -0.5 * x_step * (layer.len().saturating_sub(1) as f64);
        for (i, &u) in layer.iter().enumerate() {
            out[u] = Point {
                x: start_x + i as f64 * x_step,
                y: level as f64 * y_step,
            };
        }
    }
    out
}

fn circular_by_degree(degree: &[usize]) -> Vec<Point> {
    let n = degree.len();
    let mut nodes: Vec<usize> = (0..n).collect();
    nodes.sort_by_key(|&u| std::cmp::Reverse(degree[u]));
    let radius = 20.0 + 8.0 * (n as f64).sqrt();
    let mut pos = vec![Point::default(); n];
    for (rank, u) in nodes.into_iter().enumerate() {
        let ang = 2.0 * PI * (rank as f64) / (n as f64);
        pos[u] = Point {
            x: radius * ang.cos(),
            y: radius * ang.sin(),
        };
    }
    pos
}

fn fruchterman_reingold(adj: &[Vec<usize>], pos: &mut [Point], iterations: usize) {
    let n = pos.len();
    if n <= 1 {
        return;
    }

    let area = (n as f64) * 10_000.0;
    let k = (area / n as f64).sqrt();
    let mut temp = (n as f64).sqrt() * 12.0;
    let cool = temp / (iterations as f64 + 1.0);
    let mut disp = vec![Point::default(); n];
    let mut rng = XorShift64::new(0x1234_5678_9ABC_DEF0 ^ n as u64);

    for _ in 0..iterations {
        disp.fill(Point::default());
        for v in 0..n {
            for u in (v + 1)..n {
                let delta = pos[v].sub(pos[u]);
                let dist2 = delta.x * delta.x + delta.y * delta.y + 1e-9;
                let dist = dist2.sqrt();
                let force = (k * k) / dist;
                let dir = delta.scale(1.0 / dist);
                disp[v] = disp[v].add(dir.scale(force));
                disp[u] = disp[u].sub(dir.scale(force));
            }
        }

        for v in 0..n {
            for &u in &adj[v] {
                if u > v {
                    let delta = pos[v].sub(pos[u]);
                    let dist2 = delta.x * delta.x + delta.y * delta.y + 1e-9;
                    let dist = dist2.sqrt();
                    let force = dist2 / k.max(1e-9);
                    let dir = delta.scale(1.0 / dist);
                    disp[v] = disp[v].sub(dir.scale(force));
                    disp[u] = disp[u].add(dir.scale(force));
                }
            }
        }

        for v in 0..n {
            let d = disp[v];
            let mag = (d.x * d.x + d.y * d.y).sqrt();
            if mag > 0.0 {
                let step = temp.min(mag);
                pos[v] = pos[v].add(d.scale(step / mag));
            } else {
                pos[v].x += (rng.next_f64() - 0.5) * 0.01;
                pos[v].y += (rng.next_f64() - 0.5) * 0.01;
            }
        }
        temp = (temp - cool).max(0.1);
    }
}

fn fruchterman_reingold_sparse(adj: &[Vec<usize>], pos: &mut [Point], iterations: usize) {
    let n = pos.len();
    if n <= 1 {
        return;
    }

    let area = (n as f64) * 6_400.0;
    let k = (area / n as f64).sqrt();
    let mut temp = (n as f64).sqrt() * 4.0;
    let cool = temp / (iterations as f64 + 1.0);
    let mut disp = vec![Point::default(); n];
    let cells_per_axis = ((n as f64).sqrt() / 3.0).clamp(8.0, 128.0) as i32;

    for _ in 0..iterations {
        disp.fill(Point::default());
        let (min_x, max_x, min_y, max_y) = bounds(pos);
        let span_x = (max_x - min_x).max(1.0);
        let span_y = (max_y - min_y).max(1.0);
        let cell_w = span_x / cells_per_axis as f64;
        let cell_h = span_y / cells_per_axis as f64;
        let mut grid: HashMap<(i32, i32), Vec<usize>> = HashMap::new();

        for i in 0..n {
            let cx = ((pos[i].x - min_x) / cell_w).floor() as i32;
            let cy = ((pos[i].y - min_y) / cell_h).floor() as i32;
            grid.entry((cx, cy)).or_default().push(i);
        }

        for i in 0..n {
            let cx = ((pos[i].x - min_x) / cell_w).floor() as i32;
            let cy = ((pos[i].y - min_y) / cell_h).floor() as i32;
            for dx in -1..=1 {
                for dy in -1..=1 {
                    if let Some(bucket) = grid.get(&(cx + dx, cy + dy)) {
                        for &j in bucket {
                            if j <= i {
                                continue;
                            }
                            let delta = pos[i].sub(pos[j]);
                            let dist2 = delta.x * delta.x + delta.y * delta.y + 1e-9;
                            let dist = dist2.sqrt();
                            let force = (k * k) / dist;
                            let dir = delta.scale(1.0 / dist);
                            disp[i] = disp[i].add(dir.scale(force));
                            disp[j] = disp[j].sub(dir.scale(force));
                        }
                    }
                }
            }
        }

        for v in 0..n {
            for &u in &adj[v] {
                if u > v {
                    let delta = pos[v].sub(pos[u]);
                    let dist2 = delta.x * delta.x + delta.y * delta.y + 1e-9;
                    let dist = dist2.sqrt();
                    let force = dist2 / k.max(1e-9);
                    let dir = delta.scale(1.0 / dist);
                    disp[v] = disp[v].sub(dir.scale(force));
                    disp[u] = disp[u].add(dir.scale(force));
                }
            }
        }

        for v in 0..n {
            let d = disp[v];
            let mag = (d.x * d.x + d.y * d.y).sqrt();
            if mag > 0.0 {
                let step = temp.min(mag);
                pos[v] = pos[v].add(d.scale(step / mag));
            }
        }
        temp = (temp - cool).max(0.05);
    }
}

fn bounds(points: &[Point]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in points {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    (min_x, max_x, min_y, max_y)
}

fn normalize(points: &mut [Point]) {
    let (min_x, max_x, min_y, max_y) = bounds(points);
    let cx = 0.5 * (min_x + max_x);
    let cy = 0.5 * (min_y + max_y);
    let scale = ((max_x - min_x).max(max_y - min_y)).max(1.0);
    for p in points {
        p.x = (p.x - cx) / scale * 1000.0;
        p.y = (p.y - cy) / scale * 1000.0;
    }
}
