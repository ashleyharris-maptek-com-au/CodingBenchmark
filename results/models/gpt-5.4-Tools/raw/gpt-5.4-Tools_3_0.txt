use std::cmp::Ordering;
use std::collections::{HashMap, VecDeque};
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let state = if seed == 0 { 0x9E3779B97F4A7C15 } else { seed };
        Self { state }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }

    fn next_f64(&mut self) -> f64 {
        const SCALE: f64 = 1.0 / ((1u64 << 53) as f64);
        ((self.next_u64() >> 11) as f64) * SCALE
    }

    fn gen_range(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            (self.next_u64() as usize) % upper
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let m: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);

    if n == 0 {
        return;
    }

    let mut edges_raw = Vec::with_capacity(m);
    let mut adj = vec![Vec::<usize>::new(); n];
    let mut degree = vec![0usize; n];

    for _ in 0..m {
        let Some(us) = it.next() else { break };
        let Some(vs) = it.next() else { break };
        let Ok(u) = us.parse::<usize>() else { continue };
        let Ok(v) = vs.parse::<usize>() else { continue };
        if u >= n || v >= n {
            continue;
        }
        edges_raw.push((u, v));
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
            degree[u] += 1;
            degree[v] += 1;
        }
    }

    let components = connected_components(n, &adj);
    let max_deg = degree.iter().copied().max().unwrap_or(0);
    let density = if n > 1 {
        (2.0 * edges_raw.len() as f64) / ((n * (n - 1)) as f64)
    } else {
        0.0
    };

    let mut positions = if n <= 64 && edges_raw.len() <= 256 {
        layout_small(n, &adj, &degree, &components)
    } else if n <= 3000 {
        layout_medium(n, &adj, &degree, &components, density)
    } else {
        layout_large(n, &adj, &degree, &components, max_deg)
    };

    normalize_positions(&mut positions);

    for p in positions {
        println!("{:.6} {:.6}", p.x, p.y);
    }
}

fn connected_components(n: usize, adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let mut vis = vec![false; n];
    let mut comps = Vec::new();
    let mut q = VecDeque::new();
    for s in 0..n {
        if vis[s] {
            continue;
        }
        vis[s] = true;
        q.push_back(s);
        let mut comp = Vec::new();
        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if !vis[v] {
                    vis[v] = true;
                    q.push_back(v);
                }
            }
        }
        comps.push(comp);
    }
    comps
}

fn bfs_order(start: usize, nodes: &[usize], adj: &[Vec<usize>]) -> Vec<usize> {
    let mut in_comp = vec![false; adj.len()];
    for &u in nodes {
        in_comp[u] = true;
    }
    let mut q = VecDeque::new();
    let mut vis = vec![false; adj.len()];
    let mut order = Vec::with_capacity(nodes.len());
    vis[start] = true;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        order.push(u);
        let mut neigh = adj[u]
            .iter()
            .copied()
            .filter(|&v| in_comp[v] && !vis[v])
            .collect::<Vec<_>>();
        neigh.sort_unstable();
        for v in neigh {
            vis[v] = true;
            q.push_back(v);
        }
    }
    for &u in nodes {
        if !vis[u] {
            vis[u] = true;
            q.push_back(u);
            while let Some(x) = q.pop_front() {
                order.push(x);
                for &v in &adj[x] {
                    if in_comp[v] && !vis[v] {
                        vis[v] = true;
                        q.push_back(v);
                    }
                }
            }
        }
    }
    order
}

fn pseudo_peripheral(nodes: &[usize], adj: &[Vec<usize>], degree: &[usize]) -> usize {
    let mut start = nodes[0];
    for &u in nodes {
        if degree[u] < degree[start] {
            start = u;
        }
    }
    let mut best = start;
    let mut last_depth = 0usize;
    for _ in 0..4 {
        let (far, depth) = farthest_in_component(best, nodes, adj);
        if depth <= last_depth {
            break;
        }
        last_depth = depth;
        best = far;
    }
    best
}

fn farthest_in_component(start: usize, nodes: &[usize], adj: &[Vec<usize>]) -> (usize, usize) {
    let mut in_comp = vec![false; adj.len()];
    for &u in nodes {
        in_comp[u] = true;
    }
    let mut dist = vec![usize::MAX; adj.len()];
    let mut q = VecDeque::new();
    q.push_back(start);
    dist[start] = 0;
    let mut best = start;
    while let Some(u) = q.pop_front() {
        if dist[u] > dist[best] {
            best = u;
        }
        for &v in &adj[u] {
            if in_comp[v] && dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    (best, dist[best])
}

fn layout_small(
    n: usize,
    adj: &[Vec<usize>],
    degree: &[usize],
    components: &[Vec<usize>],
) -> Vec<Point> {
    let mut pos = vec![Point::new(0.0, 0.0); n];
    let mut x_offset = 0.0;
    let gap = 8.0;

    let mut comps = components.to_vec();
    comps.sort_by_key(|c| std::cmp::Reverse(c.len()));

    for comp in comps {
        if comp.len() == 1 {
            pos[comp[0]] = Point::new(x_offset, 0.0);
            x_offset += gap;
            continue;
        }

        let start = pseudo_peripheral(&comp, adj, degree);
        let order = bfs_order(start, &comp, adj);
        let mut rank = HashMap::with_capacity(order.len());
        for (i, &u) in order.iter().enumerate() {
            rank.insert(u, i);
        }

        let mut bary = Vec::with_capacity(comp.len());
        for &u in &comp {
            let sum = adj[u]
                .iter()
                .filter_map(|v| rank.get(v).copied())
                .map(|x| x as f64)
                .sum::<f64>();
            let cnt = adj[u].iter().filter(|v| rank.contains_key(v)).count().max(1) as f64;
            bary.push((u, sum / cnt, degree[u]));
        }
        bary.sort_by(|a, b| {
            a.1.partial_cmp(&b.1)
                .unwrap_or(Ordering::Equal)
                .then_with(|| b.2.cmp(&a.2))
                .then_with(|| a.0.cmp(&b.0))
        });

        let radius = (comp.len() as f64).max(3.0);
        for (i, &(u, _, _)) in bary.iter().enumerate() {
            let angle = 2.0 * PI * (i as f64) / (bary.len() as f64);
            pos[u] = Point::new(
                x_offset + radius * angle.cos(),
                radius * angle.sin(),
            );
        }
        x_offset += 2.5 * radius + gap;
    }

    pos
}

fn layout_medium(
    n: usize,
    adj: &[Vec<usize>],
    degree: &[usize],
    components: &[Vec<usize>],
    density: f64,
) -> Vec<Point> {
    let mut pos = spectral_seed(n, adj, degree, components);
    let iterations = if n <= 200 { 80 } else if n <= 800 { 50 } else { 24 };
    force_refine(&mut pos, adj, degree, iterations, density, 20_000);
    pack_components(&mut pos, components);
    pos
}

fn layout_large(
    n: usize,
    adj: &[Vec<usize>],
    degree: &[usize],
    components: &[Vec<usize>],
    max_deg: usize,
) -> Vec<Point> {
    let mut pos = vec![Point::new(0.0, 0.0); n];
    let mut comps = components.to_vec();
    comps.sort_by_key(|c| std::cmp::Reverse(c.len()));

    let seed = (n as u64).wrapping_mul(1_000_003) ^ (adj.len() as u64).wrapping_mul(97_003);
    let mut rng = XorShift64::new(seed);
    let mut x_offset = 0.0;

    for comp in comps {
        if comp.len() == 1 {
            pos[comp[0]] = Point::new(x_offset, 0.0);
            x_offset += 4.0;
            continue;
        }

        let start = pseudo_peripheral(&comp, adj, degree);
        let levels = bfs_levels(start, &comp, adj);
        let width = levels.iter().map(|lvl| lvl.len()).max().unwrap_or(1) as f64;
        let spread_x = (width.sqrt() + 1.0) * 3.0;
        let spread_y = (levels.len() as f64).max(2.0) * 2.2;

        for (depth, level) in levels.iter().enumerate() {
            let mut nodes = level.clone();
            nodes.sort_by_key(|&u| std::cmp::Reverse(degree[u]));
            let len = nodes.len();
            for (i, &u) in nodes.iter().enumerate() {
                let frac = if len <= 1 {
                    0.0
                } else {
                    (i as f64) / ((len - 1) as f64) - 0.5
                };
                let jitter = (rng.next_f64() - 0.5) * 0.35;
                pos[u] = Point::new(
                    x_offset + frac * spread_x + jitter,
                    depth as f64 * 2.0 + (rng.next_f64() - 0.5) * 0.2,
                );
            }
        }

        if comp.len() <= 20_000 {
            let passes = if comp.len() <= 8_000 { 10 } else { 5 };
            sampled_force_component(&mut pos, &comp, adj, degree, passes, max_deg, &mut rng);
        }

        x_offset += spread_x + (comp.len() as f64).sqrt() + 8.0;
        let _ = spread_y;
    }

    pos
}

fn bfs_levels(start: usize, nodes: &[usize], adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let mut in_comp = vec![false; adj.len()];
    for &u in nodes {
        in_comp[u] = true;
    }
    let mut dist = vec![usize::MAX; adj.len()];
    let mut q = VecDeque::new();
    q.push_back(start);
    dist[start] = 0;
    let mut levels = Vec::<Vec<usize>>::new();
    while let Some(u) = q.pop_front() {
        let d = dist[u];
        if levels.len() <= d {
            levels.push(Vec::new());
        }
        levels[d].push(u);
        for &v in &adj[u] {
            if in_comp[v] && dist[v] == usize::MAX {
                dist[v] = d + 1;
                q.push_back(v);
            }
        }
    }
    for &u in nodes {
        if dist[u] == usize::MAX {
            if levels.is_empty() {
                levels.push(Vec::new());
            }
            levels[0].push(u);
        }
    }
    levels
}

fn spectral_seed(
    n: usize,
    adj: &[Vec<usize>],
    degree: &[usize],
    components: &[Vec<usize>],
) -> Vec<Point> {
    let mut pos = vec![Point::new(0.0, 0.0); n];
    let mut x_offset = 0.0;

    let mut comps = components.to_vec();
    comps.sort_by_key(|c| std::cmp::Reverse(c.len()));

    for comp in comps {
        if comp.len() == 1 {
            pos[comp[0]] = Point::new(x_offset, 0.0);
            x_offset += 6.0;
            continue;
        }
        let root = pseudo_peripheral(&comp, adj, degree);
        let levels = bfs_levels(root, &comp, adj);
        let comp_width = levels.iter().map(|lvl| lvl.len()).max().unwrap_or(1) as f64;
        let span_x = comp_width.sqrt() * 5.0 + 4.0;
        for (depth, level) in levels.iter().enumerate() {
            let mut sorted = level.clone();
            sorted.sort_by(|&a, &b| degree[b].cmp(&degree[a]).then_with(|| a.cmp(&b)));
            let len = sorted.len();
            for (i, &u) in sorted.iter().enumerate() {
                let frac = if len <= 1 {
                    0.0
                } else {
                    (i as f64) / ((len - 1) as f64) - 0.5
                };
                pos[u] = Point::new(
                    x_offset + frac * span_x,
                    depth as f64 * 2.0,
                );
            }
        }
        x_offset += span_x + 8.0;
    }

    pos
}

fn force_refine(
    pos: &mut [Point],
    adj: &[Vec<usize>],
    degree: &[usize],
    iterations: usize,
    density: f64,
    repulsion_budget: usize,
) {
    let n = pos.len();
    if n <= 1 {
        return;
    }
    let mut rng = XorShift64::new((n as u64) ^ 0xA5A5_5A5A_DEAD_BEEF);
    let area = (n as f64).max(1.0) * 25.0;
    let k = area.sqrt() / (n as f64).sqrt();
    let edge_scale = if density > 0.15 { 0.08 } else { 0.18 };

    for it in 0..iterations {
        let temp = (1.0 - (it as f64) / (iterations as f64)).max(0.05) * k * 2.5;
        let mut disp = vec![Point::new(0.0, 0.0); n];

        let samples = repulsion_budget.min(n.saturating_mul(12)).max(n);
        for _ in 0..samples {
            let u = rng.gen_range(n);
            let v = rng.gen_range(n);
            if u == v {
                continue;
            }
            let dx = pos[u].x - pos[v].x;
            let dy = pos[u].y - pos[v].y;
            let dist2 = dx * dx + dy * dy + 1e-6;
            let force = (k * k) / dist2;
            disp[u].x += dx * force;
            disp[u].y += dy * force;
            disp[v].x -= dx * force;
            disp[v].y -= dy * force;
        }

        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let dx = pos[u].x - pos[v].x;
                    let dy = pos[u].y - pos[v].y;
                    let dist = (dx * dx + dy * dy + 1e-6).sqrt();
                    let desired = k * (1.0 + 0.05 * ((degree[u] + degree[v]) as f64).sqrt());
                    let force = edge_scale * (dist - desired) / dist;
                    let fx = dx * force;
                    let fy = dy * force;
                    disp[u].x -= fx;
                    disp[u].y -= fy;
                    disp[v].x += fx;
                    disp[v].y += fy;
                }
            }
        }

        for u in 0..n {
            let norm = (disp[u].x * disp[u].x + disp[u].y * disp[u].y).sqrt();
            if norm > 0.0 {
                let scale = temp.min(norm) / norm;
                pos[u].x += disp[u].x * scale;
                pos[u].y += disp[u].y * scale;
            }
        }
    }
}

fn sampled_force_component(
    pos: &mut [Point],
    comp: &[usize],
    adj: &[Vec<usize>],
    degree: &[usize],
    passes: usize,
    max_deg: usize,
    rng: &mut XorShift64,
) {
    let k = ((comp.len() as f64).sqrt()).max(2.0);
    let sample_size = (32usize).min(comp.len().saturating_sub(1)).max(8);
    let edge_factor = if max_deg > 128 { 0.04 } else { 0.08 };

    for pass in 0..passes {
        let step = (1.0 - pass as f64 / passes as f64).max(0.15);
        for &u in comp {
            let mut fx = 0.0;
            let mut fy = 0.0;
            for _ in 0..sample_size {
                let mut v = comp[rng.gen_range(comp.len())];
                if v == u && comp.len() > 1 {
                    v = comp[(rng.gen_range(comp.len() - 1) + 1) % comp.len()];
                }
                if v == u {
                    continue;
                }
                let dx = pos[u].x - pos[v].x;
                let dy = pos[u].y - pos[v].y;
                let dist2 = dx * dx + dy * dy + 1e-6;
                let rep = (k * k) / dist2;
                fx += dx * rep;
                fy += dy * rep;
            }
            for &v in &adj[u] {
                let dx = pos[v].x - pos[u].x;
                let dy = pos[v].y - pos[u].y;
                let dist = (dx * dx + dy * dy + 1e-6).sqrt();
                let target = 1.5 + 0.03 * ((degree[u] + degree[v]) as f64).sqrt();
                let att = edge_factor * (dist - target);
                fx += dx / dist * att;
                fy += dy / dist * att;
            }
            pos[u].x += fx * 0.0025 * step;
            pos[u].y += fy * 0.0025 * step;
        }
    }
}

fn pack_components(pos: &mut [Point], components: &[Vec<usize>]) {
    let mut comps = components.to_vec();
    comps.sort_by_key(|c| std::cmp::Reverse(c.len()));
    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_height = 0.0;
    let row_limit = (components.len() as f64).sqrt().ceil().max(1.0) as usize;

    for (idx, comp) in comps.iter().enumerate() {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for &u in comp {
            min_x = min_x.min(pos[u].x);
            max_x = max_x.max(pos[u].x);
            min_y = min_y.min(pos[u].y);
            max_y = max_y.max(pos[u].y);
        }
        let w = (max_x - min_x).max(1.0);
        let h = (max_y - min_y).max(1.0);
        if idx > 0 && idx % row_limit == 0 {
            cursor_x = 0.0;
            cursor_y += row_height + 8.0;
            row_height = 0.0;
        }
        for &u in comp {
            pos[u].x += cursor_x - min_x;
            pos[u].y += cursor_y - min_y;
        }
        cursor_x += w + 8.0;
        row_height = row_height.max(h);
    }
}

fn normalize_positions(pos: &mut [Point]) {
    if pos.is_empty() {
        return;
    }
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in pos.iter() {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    let cx = 0.5 * (min_x + max_x);
    let cy = 0.5 * (min_y + max_y);
    let span = (max_x - min_x).max(max_y - min_y).max(1.0);
    let scale = 1000.0 / span;
    for p in pos.iter_mut() {
        p.x = (p.x - cx) * scale;
        p.y = (p.y - cy) * scale;
    }
}
