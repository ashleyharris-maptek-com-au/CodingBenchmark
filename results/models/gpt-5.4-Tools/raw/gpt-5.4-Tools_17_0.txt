using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private struct Box
    {
        public int Index;
        public int W;
        public int H;
        public int D;
        public long Volume;
        public int MaxDim;
        public int MinDim;
        public int[] Dims;
    }

    private struct Placement
    {
        public int Index;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;

        public long Volume => (long)W * H * D;
    }

    private struct Orientation
    {
        public int W;
        public int H;
        public int D;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int containerW = fs.NextInt();
        int containerH = fs.NextInt();
        int containerD = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            var dims = new[] { w, h, d };
            Array.Sort(dims);
            boxes[i] = new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Volume = (long)w * h * d,
                MinDim = dims[0],
                MaxDim = dims[2],
                Dims = dims
            };
        }

        List<Placement> result = Solve(boxes, containerW, containerH, containerD);
        var sb = new StringBuilder();
        sb.AppendLine(result.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var p in result)
        {
            sb.Append(p.Index).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).AppendLine();
        }
        Console.Write(sb.ToString());
    }

    private static List<Placement> Solve(Box[] boxes, int containerW, int containerH, int containerD)
    {
        long containerVolume = (long)containerW * containerH * containerD;
        int n = boxes.Length;

        bool useMaxRects = n <= 2500 && containerVolume <= 500_000_000L;
        if (useMaxRects)
        {
            return SolveByFreeSpaces(boxes, containerW, containerH, containerD);
        }

        return SolveByAdaptiveShelves(boxes, containerW, containerH, containerD);
    }

    private static List<Placement> SolveByAdaptiveShelves(Box[] boxes, int containerW, int containerH, int containerD)
    {
        var orientations = new Orientation[boxes.Length][];
        for (int i = 0; i < boxes.Length; i++)
        {
            orientations[i] = GetOrientations(boxes[i]);
        }

        var orders = BuildOrders(boxes, containerW, containerH, containerD);
        List<Placement> best = new List<Placement>();

        foreach (var order in orders)
        {
            var packed = TryShelfPacking(boxes, orientations, order, containerW, containerH, containerD, prioritizeCount: true);
            if (packed.Count > best.Count) best = packed;
        }

        if (best.Count < boxes.Length)
        {
            foreach (var order in orders)
            {
                var packed = TryShelfPacking(boxes, orientations, order, containerW, containerH, containerD, prioritizeCount: false);
                if (packed.Count > best.Count) best = packed;
            }
        }

        return best;
    }

    private static List<int[]> BuildOrders(Box[] boxes, int containerW, int containerH, int containerD)
    {
        var fits = boxes.Where(b => CanFitAnywhere(b, containerW, containerH, containerD)).ToArray();

        return new List<int[]>
        {
            fits.OrderBy(b => b.Volume).ThenBy(b => b.MaxDim).Select(b => b.Index).ToArray(),
            fits.OrderByDescending(b => b.Volume).ThenBy(b => b.MinDim).Select(b => b.Index).ToArray(),
            fits.OrderBy(b => b.MinDim).ThenBy(b => b.Volume).Select(b => b.Index).ToArray(),
            fits.OrderBy(b => Math.Abs(b.Dims[1] - b.Dims[0]) + Math.Abs(b.Dims[2] - b.Dims[1]))
                .ThenBy(b => b.Volume)
                .Select(b => b.Index)
                .ToArray()
        };
    }

    private static List<Placement> TryShelfPacking(
        Box[] boxes,
        Orientation[][] orientations,
        int[] order,
        int containerW,
        int containerH,
        int containerD,
        bool prioritizeCount)
    {
        var placements = new List<Placement>(order.Length);
        int layerZ = 0;
        int layerDepth = 0;
        int shelfY = 0;
        int shelfHeight = 0;
        int cursorX = 0;

        foreach (int idx in order)
        {
            var chosen = default(Orientation);
            int stage = -1;

            for (int pass = 0; pass < 4 && stage < 0; pass++)
            {
                foreach (var o in RankOrientations(orientations[idx], containerW, containerH, containerD, prioritizeCount, pass))
                {
                    if (FitsAt(cursorX, shelfY, layerZ, o, containerW, containerH, containerD))
                    {
                        chosen = o;
                        stage = 0;
                        break;
                    }

                    if (FitsAt(0, shelfY + shelfHeight, layerZ, o, containerW, containerH, containerD))
                    {
                        chosen = o;
                        stage = 1;
                        break;
                    }

                    if (FitsAt(0, 0, layerZ + layerDepth, o, containerW, containerH, containerD))
                    {
                        chosen = o;
                        stage = 2;
                        break;
                    }
                }
            }

            if (stage < 0) continue;

            if (stage == 1)
            {
                shelfY += shelfHeight;
                shelfHeight = 0;
                cursorX = 0;
            }
            else if (stage == 2)
            {
                layerZ += layerDepth;
                layerDepth = 0;
                shelfY = 0;
                shelfHeight = 0;
                cursorX = 0;
            }

            placements.Add(new Placement
            {
                Index = idx,
                X = cursorX,
                Y = shelfY,
                Z = layerZ,
                W = chosen.W,
                H = chosen.H,
                D = chosen.D
            });

            cursorX += chosen.W;
            if (chosen.H > shelfHeight) shelfHeight = chosen.H;
            if (chosen.D > layerDepth) layerDepth = chosen.D;
        }

        return placements;
    }

    private static IEnumerable<Orientation> RankOrientations(
        Orientation[] options,
        int containerW,
        int containerH,
        int containerD,
        bool prioritizeCount,
        int mode)
    {
        IEnumerable<Orientation> seq = options;
        if (prioritizeCount)
        {
            seq = mode switch
            {
                0 => seq.OrderBy(o => o.W).ThenBy(o => o.H).ThenBy(o => o.D),
                1 => seq.OrderBy(o => o.H).ThenBy(o => o.W).ThenBy(o => o.D),
                2 => seq.OrderBy(o => o.D).ThenBy(o => o.W).ThenBy(o => o.H),
                _ => seq.OrderBy(o => Math.Max(Math.Max(o.W, o.H), o.D))
            };
        }
        else
        {
            seq = mode switch
            {
                0 => seq.OrderByDescending(o => (long)o.W * o.H * o.D).ThenBy(o => o.W),
                1 => seq.OrderByDescending(o => o.D).ThenBy(o => o.W),
                2 => seq.OrderByDescending(o => o.H).ThenBy(o => o.D),
                _ => seq.OrderBy(o => Math.Abs(containerW - o.W) + Math.Abs(containerH - o.H) + Math.Abs(containerD - o.D))
            };
        }

        return seq;
    }

    private static bool FitsAt(int x, int y, int z, Orientation o, int containerW, int containerH, int containerD)
    {
        return x + o.W <= containerW && y + o.H <= containerH && z + o.D <= containerD;
    }

    private static List<Placement> SolveByFreeSpaces(Box[] boxes, int containerW, int containerH, int containerD)
    {
        var orientations = new Orientation[boxes.Length][];
        for (int i = 0; i < boxes.Length; i++)
        {
            orientations[i] = GetOrientations(boxes[i]);
        }

        var candidateOrders = BuildOrders(boxes, containerW, containerH, containerD);
        List<Placement> best = new List<Placement>();

        foreach (var order in candidateOrders)
        {
            var packed = RunFreeSpaceHeuristic(boxes, orientations, order, containerW, containerH, containerD);
            if (packed.Count > best.Count) best = packed;
        }

        return best;
    }

    private static List<Placement> RunFreeSpaceHeuristic(
        Box[] boxes,
        Orientation[][] orientations,
        int[] order,
        int containerW,
        int containerH,
        int containerD)
    {
        var spaces = new List<Space>(1024)
        {
            new Space { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD }
        };

        var placements = new List<Placement>(order.Length);

        foreach (int idx in order)
        {
            int bestSpaceIdx = -1;
            Orientation bestOrientation = default;
            long bestScore = long.MaxValue;

            for (int s = 0; s < spaces.Count; s++)
            {
                Space sp = spaces[s];
                foreach (var o in orientations[idx])
                {
                    if (o.W > sp.W || o.H > sp.H || o.D > sp.D) continue;

                    long slack = (long)(sp.W - o.W) * 1_000_000_000L + (long)(sp.H - o.H) * 1_000_000L + (sp.D - o.D);
                    long score = slack + ((long)sp.X + sp.Y + sp.Z) * 17L;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestSpaceIdx = s;
                        bestOrientation = o;
                    }
                }
            }

            if (bestSpaceIdx < 0) continue;

            Space space = spaces[bestSpaceIdx];
            spaces.RemoveAt(bestSpaceIdx);

            placements.Add(new Placement
            {
                Index = idx,
                X = space.X,
                Y = space.Y,
                Z = space.Z,
                W = bestOrientation.W,
                H = bestOrientation.H,
                D = bestOrientation.D
            });

            SplitSpace(space, bestOrientation, spaces);
            PruneSpaces(spaces);
        }

        return placements;
    }

    private static void SplitSpace(Space space, Orientation box, List<Space> spaces)
    {
        int remW = space.W - box.W;
        int remH = space.H - box.H;
        int remD = space.D - box.D;

        if (remW > 0)
        {
            spaces.Add(new Space
            {
                X = space.X + box.W,
                Y = space.Y,
                Z = space.Z,
                W = remW,
                H = box.H,
                D = box.D
            });
        }

        if (remH > 0)
        {
            spaces.Add(new Space
            {
                X = space.X,
                Y = space.Y + box.H,
                Z = space.Z,
                W = space.W,
                H = remH,
                D = box.D
            });
        }

        if (remD > 0)
        {
            spaces.Add(new Space
            {
                X = space.X,
                Y = space.Y,
                Z = space.Z + box.D,
                W = space.W,
                H = space.H,
                D = remD
            });
        }

        if (remW > 0 && space.H - box.H > 0)
        {
            spaces.Add(new Space
            {
                X = space.X + box.W,
                Y = space.Y + box.H,
                Z = space.Z,
                W = remW,
                H = space.H - box.H,
                D = box.D
            });
        }

        if (remW > 0 && remD > 0)
        {
            spaces.Add(new Space
            {
                X = space.X + box.W,
                Y = space.Y,
                Z = space.Z + box.D,
                W = remW,
                H = box.H,
                D = remD
            });
        }

        if (remH > 0 && remD > 0)
        {
            spaces.Add(new Space
            {
                X = space.X,
                Y = space.Y + box.H,
                Z = space.Z + box.D,
                W = space.W,
                H = remH,
                D = remD
            });
        }
    }

    private static void PruneSpaces(List<Space> spaces)
    {
        spaces.RemoveAll(s => s.W <= 0 || s.H <= 0 || s.D <= 0);

        spaces.Sort((a, b) =>
        {
            int cmp = a.X.CompareTo(b.X);
            if (cmp != 0) return cmp;
            cmp = a.Y.CompareTo(b.Y);
            if (cmp != 0) return cmp;
            cmp = a.Z.CompareTo(b.Z);
            if (cmp != 0) return cmp;
            return b.Volume.CompareTo(a.Volume);
        });

        var pruned = new List<Space>(spaces.Count);
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            bool contained = false;
            for (int j = 0; j < pruned.Count; j++)
            {
                if (Contains(pruned[j], s))
                {
                    contained = true;
                    break;
                }
            }

            if (!contained) pruned.Add(s);
        }

        spaces.Clear();
        if (pruned.Count > 4000)
        {
            pruned.Sort((a, b) => a.Volume == b.Volume ? (a.X + a.Y + a.Z).CompareTo(b.X + b.Y + b.Z) : b.Volume.CompareTo(a.Volume));
            spaces.AddRange(pruned.Take(4000));
        }
        else
        {
            spaces.AddRange(pruned);
        }
    }

    private static bool Contains(Space a, Space b)
    {
        return a.X <= b.X &&
               a.Y <= b.Y &&
               a.Z <= b.Z &&
               a.X + a.W >= b.X + b.W &&
               a.Y + a.H >= b.Y + b.H &&
               a.Z + a.D >= b.Z + b.D;
    }

    private static bool CanFitAnywhere(Box box, int containerW, int containerH, int containerD)
    {
        foreach (var o in GetOrientations(box))
        {
            if (o.W <= containerW && o.H <= containerH && o.D <= containerD) return true;
        }

        return false;
    }

    private static Orientation[] GetOrientations(Box b)
    {
        var seen = new HashSet<long>();
        var result = new List<Orientation>(6);
        int[] vals = { b.W, b.H, b.D };
        int[][] perms =
        {
            new[] { 0, 1, 2 },
            new[] { 0, 2, 1 },
            new[] { 1, 0, 2 },
            new[] { 1, 2, 0 },
            new[] { 2, 0, 1 },
            new[] { 2, 1, 0 }
        };

        foreach (var p in perms)
        {
            int w = vals[p[0]];
            int h = vals[p[1]];
            int d = vals[p[2]];
            long key = ((long)w << 42) ^ ((long)h << 21) ^ (uint)d;
            if (!seen.Add(key)) continue;
            result.Add(new Orientation { W = w, H = h, D = d });
        }

        return result.ToArray();
    }
}
