using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= ' ' && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > ' ')
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    private struct Orientation
    {
        public int W;
        public int H;
        public int D;

        public Orientation(int w, int h, int d)
        {
            W = w;
            H = h;
            D = d;
        }
    }

    private sealed class Box
    {
        public int Index;
        public int W;
        public int H;
        public int D;
        public long Volume;
        public Orientation[] Orientations = Array.Empty<Orientation>();
    }

    private sealed class Placement
    {
        public int BoxIndex;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private struct FreeRect
    {
        public int X;
        public int Z;
        public int W;
        public int D;

        public FreeRect(int x, int z, int w, int d)
        {
            X = x;
            Z = z;
            W = w;
            D = d;
        }
    }

    private sealed class LayerPacker
    {
        private readonly int _width;
        private readonly int _depth;
        private readonly List<FreeRect> _freeRects = new List<FreeRect>();

        public LayerPacker(int width, int depth)
        {
            _width = width;
            _depth = depth;
            Reset();
        }

        public void Reset()
        {
            _freeRects.Clear();
            _freeRects.Add(new FreeRect(0, 0, _width, _depth));
        }

        public bool TryPlace(int w, int d, out int x, out int z)
        {
            int bestIndex = -1;
            long bestScore = long.MaxValue;
            int bestX = 0;
            int bestZ = 0;

            for (int i = 0; i < _freeRects.Count; i++)
            {
                var rect = _freeRects[i];
                if (w > rect.W || d > rect.D) continue;

                long shortSide = Math.Min(rect.W - w, rect.D - d);
                long longSide = Math.Max(rect.W - w, rect.D - d);
                long score = (shortSide << 32) + longSide;
                if (score < bestScore || (score == bestScore && (rect.Z < bestZ || (rect.Z == bestZ && rect.X < bestX))))
                {
                    bestScore = score;
                    bestIndex = i;
                    bestX = rect.X;
                    bestZ = rect.Z;
                }
            }

            if (bestIndex < 0)
            {
                x = 0;
                z = 0;
                return false;
            }

            x = bestX;
            z = bestZ;
            Split(bestIndex, w, d);
            return true;
        }

        private void Split(int index, int usedW, int usedD)
        {
            var rect = _freeRects[index];
            _freeRects.RemoveAt(index);

            int rightW = rect.W - usedW;
            int topD = rect.D - usedD;

            if (rightW > 0)
            {
                _freeRects.Add(new FreeRect(rect.X + usedW, rect.Z, rightW, usedD));
            }
            if (topD > 0)
            {
                _freeRects.Add(new FreeRect(rect.X, rect.Z + usedD, rect.W, topD));
            }
            if (rightW > 0 && topD > 0)
            {
                _freeRects.Add(new FreeRect(rect.X + usedW, rect.Z + usedD, rightW, topD));
            }

            Prune();
        }

        private void Prune()
        {
            for (int i = 0; i < _freeRects.Count; i++)
            {
                var a = _freeRects[i];
                for (int j = _freeRects.Count - 1; j > i; j--)
                {
                    var b = _freeRects[j];
                    if (Contains(a, b))
                    {
                        _freeRects.RemoveAt(j);
                    }
                    else if (Contains(b, a))
                    {
                        _freeRects[i] = _freeRects[_freeRects.Count - 1];
                        _freeRects.RemoveAt(_freeRects.Count - 1);
                        i--;
                        break;
                    }
                }
            }
        }

        private static bool Contains(FreeRect a, FreeRect b)
        {
            return b.X >= a.X && b.Z >= a.Z && b.X + b.W <= a.X + a.W && b.Z + b.D <= a.Z + a.D;
        }
    }

    private sealed class Candidate
    {
        public int BoxIndex;
        public Orientation Orientation;
        public long Score;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int containerW = fs.NextInt();
        int containerH = fs.NextInt();
        int containerD = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            boxes[i] = new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Volume = (long)w * h * d,
                Orientations = BuildOrientations(w, h, d)
            };
        }

        var placements = Solve(boxes, containerW, containerH, containerD);

        Console.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var p in placements)
        {
            Console.WriteLine(string.Create(CultureInfo.InvariantCulture, $"{p.BoxIndex} {p.X} {p.Y} {p.Z}"));
        }
    }

    private static List<Placement> Solve(Box[] boxes, int containerW, int containerH, int containerD)
    {
        bool smallInstance = boxes.Length <= 64 && (long)containerW * containerH * containerD <= 2_000_000L;
        var layerHeights = CollectLayerHeights(boxes, containerH, smallInstance ? 10 : 6);
        if (layerHeights.Count == 0)
        {
            return new List<Placement>();
        }

        List<Placement>? best = null;
        int bestCount = -1;
        long bestVolume = -1;

        foreach (var orderMode in smallInstance ? new[] { 0, 1, 2, 3 } : new[] { 0, 1 })
        {
            var current = PackByLayers(boxes, containerW, containerH, containerD, layerHeights, orderMode, smallInstance);
            long volume = 0;
            foreach (var p in current) volume += (long)p.W * p.H * p.D;
            if (current.Count > bestCount || (current.Count == bestCount && volume > bestVolume))
            {
                best = current;
                bestCount = current.Count;
                bestVolume = volume;
            }
        }

        return best ?? new List<Placement>();
    }

    private static List<Placement> PackByLayers(Box[] boxes, int containerW, int containerH, int containerD, List<int> layerHeights, int orderMode, bool smallInstance)
    {
        var used = new bool[boxes.Length];
        var placements = new List<Placement>(boxes.Length);
        var packer = new LayerPacker(containerW, containerD);
        int y = 0;

        while (y < containerH)
        {
            int bestLayer = -1;
            List<Candidate>? bestCandidates = null;
            long bestVolume = -1;
            int bestCount = -1;

            foreach (int layerH in layerHeights)
            {
                if (layerH <= 0 || y + layerH > containerH) continue;
                var candidates = BuildCandidates(boxes, used, layerH, containerW, containerD, orderMode, smallInstance);
                if (candidates.Count == 0) continue;

                packer.Reset();
                var tentative = new List<Candidate>();
                long volume = 0;
                foreach (var c in candidates)
                {
                    if (packer.TryPlace(c.Orientation.W, c.Orientation.D, out _, out _))
                    {
                        tentative.Add(c);
                        volume += (long)c.Orientation.W * c.Orientation.H * c.Orientation.D;
                    }
                }

                if (tentative.Count > bestCount || (tentative.Count == bestCount && volume > bestVolume))
                {
                    bestCount = tentative.Count;
                    bestVolume = volume;
                    bestLayer = layerH;
                    bestCandidates = tentative;
                }
            }

            if (bestLayer < 0 || bestCandidates == null || bestCandidates.Count == 0)
            {
                break;
            }

            packer.Reset();
            foreach (var c in bestCandidates)
            {
                if (!used[c.BoxIndex] && packer.TryPlace(c.Orientation.W, c.Orientation.D, out int x, out int z))
                {
                    used[c.BoxIndex] = true;
                    placements.Add(new Placement
                    {
                        BoxIndex = c.BoxIndex,
                        X = x,
                        Y = y,
                        Z = z,
                        W = c.Orientation.W,
                        H = c.Orientation.H,
                        D = c.Orientation.D
                    });
                }
            }

            y += bestLayer;
        }

        return placements;
    }

    private static List<Candidate> BuildCandidates(Box[] boxes, bool[] used, int layerH, int containerW, int containerD, int orderMode, bool smallInstance)
    {
        var candidates = new List<Candidate>(boxes.Length);
        for (int i = 0; i < boxes.Length; i++)
        {
            if (used[i]) continue;
            Orientation best = default;
            bool found = false;
            long bestScore = long.MinValue;

            foreach (var o in boxes[i].Orientations)
            {
                if (o.H != layerH) continue;
                if (o.W > containerW || o.D > containerD) continue;
                long footprint = (long)o.W * o.D;
                long score = orderMode switch
                {
                    0 => footprint * 16L - o.W - o.D,
                    1 => footprint * 8L - Math.Abs(o.W - o.D),
                    2 => boxes[i].Volume * 4L - footprint,
                    _ => footprint * 8L - Math.Max(o.W, o.D)
                };
                if (!found || score > bestScore)
                {
                    best = o;
                    bestScore = score;
                    found = true;
                }
            }

            if (!found) continue;
            candidates.Add(new Candidate
            {
                BoxIndex = boxes[i].Index,
                Orientation = best,
                Score = CandidatePriority(boxes[i], best, orderMode, smallInstance)
            });
        }

        candidates.Sort((a, b) => b.Score.CompareTo(a.Score));
        return candidates;
    }

    private static long CandidatePriority(Box box, Orientation o, int mode, bool smallInstance)
    {
        long footprint = (long)o.W * o.D;
        long volume = (long)o.W * o.H * o.D;
        return mode switch
        {
            0 => volume * 32L + footprint * 4L - (o.W + o.D),
            1 => footprint * 64L + volume - Math.Abs(o.W - o.D),
            2 => (smallInstance ? 128L : 32L) * volume - 2L * Math.Max(o.W, o.D),
            _ => volume * 16L + 8L * Math.Min(o.W, o.D) - Math.Max(o.W, o.D)
        };
    }

    private static List<int> CollectLayerHeights(Box[] boxes, int containerH, int perBoxLimit)
    {
        var set = new HashSet<int>();
        foreach (var box in boxes)
        {
            int added = 0;
            foreach (var o in box.Orientations)
            {
                if (o.H <= containerH && set.Add(o.H))
                {
                    added++;
                    if (added >= perBoxLimit) break;
                }
            }
        }

        var list = set.ToList();
        list.Sort((a, b) => a.CompareTo(b));
        return list;
    }

    private static Orientation[] BuildOrientations(int w, int h, int d)
    {
        var set = new HashSet<(int, int, int)>();
        var list = new List<Orientation>(6);
        AddOrientation(w, h, d, set, list);
        AddOrientation(w, d, h, set, list);
        AddOrientation(h, w, d, set, list);
        AddOrientation(h, d, w, set, list);
        AddOrientation(d, w, h, set, list);
        AddOrientation(d, h, w, set, list);
        list.Sort((a, b) =>
        {
            int c = a.H.CompareTo(b.H);
            if (c != 0) return c;
            c = ((long)a.W * a.D).CompareTo((long)b.W * b.D);
            if (c != 0) return c;
            c = a.W.CompareTo(b.W);
            if (c != 0) return c;
            return a.D.CompareTo(b.D);
        });
        return list.ToArray();
    }

    private static void AddOrientation(int w, int h, int d, HashSet<(int, int, int)> set, List<Orientation> list)
    {
        if (set.Add((w, h, d)))
        {
            list.Add(new Orientation(w, h, d));
        }
    }
}
