struct RollingHash {
    static constexpr uint64_t kBase1 = 1469598103934665603ULL;
    static constexpr uint64_t kBase2 = 1099511628211ULL;
    vector<uint64_t> pref1;
    vector<uint64_t> pref2;
    vector<uint64_t> pow1;
    vector<uint64_t> pow2;

    RollingHash() = default;
    explicit RollingHash(const string& s) { build(s); }

    void build(const string& s) {
        size_t n = s.size();
        pref1.assign(n + 1, 0);
        pref2.assign(n + 1, 0);
        pow1.assign(n + 1, 1);
        pow2.assign(n + 1, 1);
        for (size_t i = 0; i < n; ++i) {
            uint64_t c = static_cast<unsigned char>(s[i]) + 1ULL;
            pref1[i + 1] = pref1[i] * kBase1 + c;
            pref2[i + 1] = pref2[i] * kBase2 + c;
            pow1[i + 1] = pow1[i] * kBase1;
            pow2[i + 1] = pow2[i] * kBase2;
        }
    }

    pair<uint64_t, uint64_t> get(size_t l, size_t r) const {
        uint64_t h1 = pref1[r] - pref1[l] * pow1[r - l];
        uint64_t h2 = pref2[r] - pref2[l] * pow2[r - l];
        return {h1, h2};
    }
};

struct PairHash {
    size_t operator()(const pair<uint64_t, uint64_t>& p) const noexcept {
        uint64_t x = p.first ^ (p.second + 0x9e3779b97f4a7c15ULL + (p.first << 6) + (p.first >> 2));
        return static_cast<size_t>(x);
    }
};

bool contains_substring(const string& text, string_view needle) {
    return text.find(needle) != string::npos;
}

string brute_force_small(vector<string> strings) {
    if (strings.empty()) {
        return "";
    }
    sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });
    const string& base = strings[0];
    size_t m = base.size();
    for (size_t len = m; len > 0; --len) {
        for (size_t start = 0; start + len <= m; ++start) {
            string_view candidate(base.data() + start, len);
            bool ok = true;
            for (size_t i = 1; i < strings.size(); ++i) {
                if (!contains_substring(strings[i], candidate)) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return string(candidate);
            }
        }
    }
    return "";
}

string solve_two_strings_dp(const string& a, const string& b) {
    const string* pa = &a;
    const string* pb = &b;
    if (a.size() < b.size()) {
        pa = &b;
        pb = &a;
    }
    const string& s1 = *pa;
    const string& s2 = *pb;
    vector<int> dp(s2.size() + 1, 0);
    size_t best_len = 0;
    size_t best_end = 0;
    for (size_t i = 1; i <= s1.size(); ++i) {
        for (size_t j = s2.size(); j > 0; --j) {
            if (s1[i - 1] == s2[j - 1]) {
                dp[j] = dp[j - 1] + 1;
                if (static_cast<size_t>(dp[j]) > best_len) {
                    best_len = static_cast<size_t>(dp[j]);
                    best_end = i;
                }
            } else {
                dp[j] = 0;
            }
        }
    }
    return s1.substr(best_end - best_len, best_len);
}

string solve_general_hash(vector<string> strings) {
    if (strings.empty()) {
        return "";
    }
    vector<int> order(strings.size());
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        return strings[lhs].size() < strings[rhs].size();
    });

    vector<string> sorted;
    sorted.reserve(strings.size());
    for (int idx : order) {
        sorted.push_back(move(strings[idx]));
    }
    strings = move(sorted);

    const size_t n = strings.size();
    vector<RollingHash> hashes;
    hashes.reserve(n);
    for (const auto& s : strings) {
        hashes.emplace_back(s);
    }

    size_t low = 0;
    size_t high = strings[0].size();
    size_t best_len = 0;
    size_t best_pos = 0;

    auto exists = [&](size_t len, size_t& out_pos) -> bool {
        if (len == 0) {
            out_pos = 0;
            return true;
        }
        unordered_map<pair<uint64_t, uint64_t>, size_t, PairHash> current;
        current.reserve(strings[0].size() - len + 1);
        for (size_t i = 0; i + len <= strings[0].size(); ++i) {
            current.emplace(hashes[0].get(i, i + len), i);
        }
        if (current.empty()) {
            return false;
        }

        for (size_t sidx = 1; sidx < n; ++sidx) {
            unordered_set<pair<uint64_t, uint64_t>, PairHash> present;
            present.reserve(strings[sidx].size() >= len ? strings[sidx].size() - len + 1 : 0);
            for (size_t i = 0; i + len <= strings[sidx].size(); ++i) {
                auto key = hashes[sidx].get(i, i + len);
                if (current.find(key) != current.end()) {
                    present.insert(key);
                }
            }
            if (present.empty()) {
                return false;
            }

            unordered_map<pair<uint64_t, uint64_t>, size_t, PairHash> next;
            next.reserve(present.size());
            for (const auto& key : present) {
                next.emplace(key, current.find(key)->second);
            }
            current.swap(next);
            if (current.empty()) {
                return false;
            }
        }

        out_pos = current.begin()->second;
        return true;
    };

    while (low <= high) {
        size_t mid = low + (high - low) / 2;
        size_t pos = 0;
        if (exists(mid, pos)) {
            best_len = mid;
            best_pos = pos;
            low = mid + 1;
        } else {
            if (mid == 0) {
                break;
            }
            high = mid - 1;
        }
    }

    if (best_len == 0) {
        return "";
    }

    string candidate = strings[0].substr(best_pos, best_len);
    for (size_t i = 1; i < strings.size(); ++i) {
        if (!contains_substring(strings[i], candidate)) {
            return brute_force_small(strings);
        }
    }
    return candidate;
}

string solve_adaptive(vector<string> strings) {
    if (strings.empty()) {
        return "";
    }
    if (strings.size() == 1) {
        return strings[0];
    }

    size_t min_len = numeric_limits<size_t>::max();
    size_t total_len = 0;
    for (const auto& s : strings) {
        min_len = min(min_len, s.size());
        total_len += s.size();
    }

    if (strings.size() <= 3 && min_len <= 64 && total_len <= 256) {
        return brute_force_small(move(strings));
    }

    if (strings.size() == 2) {
        const uint64_t area = static_cast<uint64_t>(strings[0].size()) * static_cast<uint64_t>(strings[1].size());
        if (area <= 25000000ULL) {
            return solve_two_strings_dp(strings[0], strings[1]);
        }
    }

    return solve_general_hash(move(strings));
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    size_t n = 0;
    if (!(cin >> n)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy);

    vector<string> strings;
    strings.reserve(n);
    for (size_t i = 0; i < n; ++i) {
        string s;
        getline(cin, s);
        strings.push_back(move(s));
    }

    cout << solve_adaptive(move(strings)) << '\n';
    return 0;
}
