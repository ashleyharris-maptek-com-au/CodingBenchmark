struct RollingHash {
    static constexpr uint64_t BASE1 = 1469598103934665603ULL;
    static constexpr uint64_t BASE2 = 1099511628211ULL;

    vector<uint64_t> pow1{1}, pow2{1};
    vector<uint64_t> pref1, pref2;

    RollingHash() = default;

    explicit RollingHash(const string& s) {
        build(s);
    }

    void ensure_pows(size_t n) {
        while (pow1.size() <= n) {
            pow1.push_back(pow1.back() * BASE1 + 0x9e3779b97f4a7c15ULL);
            pow2.push_back(pow2.back() * BASE2 + 0xc2b2ae3d27d4eb4fULL);
        }
    }

    void build(const string& s) {
        ensure_pows(s.size());
        pref1.assign(s.size() + 1, 0);
        pref2.assign(s.size() + 1, 0);
        for (size_t i = 0; i < s.size(); ++i) {
            const uint64_t x = static_cast<unsigned char>(s[i]) + 1ULL;
            pref1[i + 1] = pref1[i] * BASE1 + x;
            pref2[i + 1] = pref2[i] * BASE2 + x;
        }
    }

    pair<uint64_t, uint64_t> slice(size_t l, size_t r) const {
        const uint64_t h1 = pref1[r] - pref1[l] * pow1[r - l];
        const uint64_t h2 = pref2[r] - pref2[l] * pow2[r - l];
        return {h1, h2};
    }
};

struct PairHash {
    size_t operator()(const pair<uint64_t, uint64_t>& p) const noexcept {
        uint64_t x = p.first ^ (p.second + 0x9e3779b97f4a7c15ULL + (p.first << 6) + (p.first >> 2));
        return static_cast<size_t>(x);
    }
};

string shortest_string_first(vector<string>& strings) {
    auto it = min_element(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });
    string shortest = move(*it);
    strings.erase(it);
    strings.insert(strings.begin(), move(shortest));
    return strings.front();
}

string solve_small_dp(const vector<string>& strings) {
    if (strings.empty()) return "";
    string best = strings[0];
    for (size_t idx = 1; idx < strings.size() && !best.empty(); ++idx) {
        const string& a = best;
        const string& b = strings[idx];
        vector<int> prev(b.size() + 1, 0), curr(b.size() + 1, 0);
        size_t best_len = 0;
        size_t best_end = 0;
        for (size_t i = 0; i < a.size(); ++i) {
            for (size_t j = 0; j < b.size(); ++j) {
                if (a[i] == b[j]) {
                    curr[j + 1] = prev[j] + 1;
                    if (static_cast<size_t>(curr[j + 1]) > best_len) {
                        best_len = static_cast<size_t>(curr[j + 1]);
                        best_end = i + 1;
                    }
                } else {
                    curr[j + 1] = 0;
                }
            }
            fill(prev.begin(), prev.end(), 0);
            swap(prev, curr);
        }
        best = a.substr(best_end - best_len, best_len);
    }

    // Verify against all strings in case pairwise shrinking picked a substring not common to all.
    for (;;) {
        bool ok = true;
        for (const auto& s : strings) {
            if (s.find(best) == string::npos) {
                ok = false;
                break;
            }
        }
        if (ok || best.empty()) break;
        best.pop_back();
    }
    return best;
}

bool exists_common_of_length(
    const vector<string>& strings,
    const vector<RollingHash>& hashes,
    size_t len,
    string& out
) {
    if (len == 0) {
        out.clear();
        return true;
    }

    const string& base = strings[0];
    if (len > base.size()) return false;

    unordered_map<pair<uint64_t, uint64_t>, vector<size_t>, PairHash> candidates;
    candidates.reserve(base.size());

    for (size_t i = 0; i + len <= base.size(); ++i) {
        auto h = hashes[0].slice(i, i + len);
        candidates[h].push_back(i);
    }
    if (candidates.empty()) return false;

    for (size_t idx = 1; idx < strings.size(); ++idx) {
        const string& s = strings[idx];
        if (len > s.size()) return false;

        unordered_set<pair<uint64_t, uint64_t>, PairHash> seen;
        seen.reserve(s.size());
        for (size_t i = 0; i + len <= s.size(); ++i) {
            auto h = hashes[idx].slice(i, i + len);
            if (candidates.find(h) != candidates.end()) {
                seen.insert(h);
            }
        }

        for (auto it = candidates.begin(); it != candidates.end();) {
            if (seen.find(it->first) == seen.end()) {
                it = candidates.erase(it);
            } else {
                ++it;
            }
        }
        if (candidates.empty()) return false;
    }

    for (const auto& [h, positions] : candidates) {
        for (size_t pos : positions) {
            const string cand = base.substr(pos, len);
            bool ok = true;
            for (size_t idx = 1; idx < strings.size(); ++idx) {
                if (strings[idx].find(cand) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                out = cand;
                return true;
            }
        }
    }
    return false;
}

string solve_hash_binary(vector<string> strings) {
    shortest_string_first(strings);
    vector<RollingHash> hashes;
    hashes.reserve(strings.size());
    for (const auto& s : strings) hashes.emplace_back(s);

    size_t lo = 0;
    size_t hi = strings[0].size();
    string best;

    while (lo <= hi) {
        const size_t mid = lo + (hi - lo) / 2;
        string candidate;
        if (exists_common_of_length(strings, hashes, mid, candidate)) {
            best = move(candidate);
            lo = mid + 1;
        } else {
            if (mid == 0) break;
            hi = mid - 1;
        }
    }
    return best;
}

vector<int> build_suffix_array(const string& s) {
    const int n = static_cast<int>(s.size());
    vector<int> sa(n), rnk(n), tmp(n);
    iota(sa.begin(), sa.end(), 0);
    for (int i = 0; i < n; ++i) rnk[i] = static_cast<unsigned char>(s[i]);

    for (int k = 1;; k <<= 1) {
        auto cmp = [&](int a, int b) {
            if (rnk[a] != rnk[b]) return rnk[a] < rnk[b];
            const int ra = (a + k < n) ? rnk[a + k] : -1;
            const int rb = (b + k < n) ? rnk[b + k] : -1;
            return ra < rb;
        };
        sort(sa.begin(), sa.end(), cmp);
        tmp[sa[0]] = 0;
        for (int i = 1; i < n; ++i) {
            tmp[sa[i]] = tmp[sa[i - 1]] + (cmp(sa[i - 1], sa[i]) ? 1 : 0);
        }
        rnk = tmp;
        if (rnk[sa[n - 1]] == n - 1) break;
    }
    return sa;
}

vector<int> build_lcp(const string& s, const vector<int>& sa) {
    const int n = static_cast<int>(s.size());
    vector<int> rank(n), lcp(max(0, n - 1), 0);
    for (int i = 0; i < n; ++i) rank[sa[i]] = i;
    int h = 0;
    for (int i = 0; i < n; ++i) {
        const int r = rank[i];
        if (r == 0) continue;
        const int j = sa[r - 1];
        while (i + h < n && j + h < n && s[i + h] == s[j + h]) ++h;
        lcp[r - 1] = h;
        if (h > 0) --h;
    }
    return lcp;
}

string solve_two_suffix_array(const string& a, const string& b) {
    const char sep1 = '\x01';
    const char sep2 = '\x00';
    string merged;
    merged.reserve(a.size() + b.size() + 2);
    merged += a;
    merged += sep1;
    merged += b;
    merged += sep2;

    vector<int> owner(merged.size(), 1);
    for (size_t i = 0; i < a.size(); ++i) owner[i] = 0;
    owner[a.size()] = -1;
    for (size_t i = a.size() + 1; i < a.size() + 1 + b.size(); ++i) owner[i] = 1;
    owner.back() = -1;

    const auto sa = build_suffix_array(merged);
    const auto lcp = build_lcp(merged, sa);

    int best_len = 0;
    int best_pos = 0;
    for (size_t i = 1; i < sa.size(); ++i) {
        const int o1 = owner[sa[i - 1]];
        const int o2 = owner[sa[i]];
        if (o1 >= 0 && o2 >= 0 && o1 != o2) {
            if (lcp[i - 1] > best_len) {
                best_len = lcp[i - 1];
                best_pos = sa[i];
            }
        }
    }
    return merged.substr(best_pos, best_len);
}

string solve_adaptive(vector<string> strings) {
    if (strings.empty()) return "";
    if (strings.size() == 1) return strings[0];

    shortest_string_first(strings);

    size_t min_len = strings[0].size();
    size_t total_len = 0;
    for (const auto& s : strings) total_len += s.size();

    if (strings.size() == 2 && total_len <= 200000) {
        return solve_two_suffix_array(strings[0], strings[1]);
    }

    if (total_len <= 50000 && min_len <= 4000) {
        return solve_small_dp(strings);
    }

    return solve_hash_binary(move(strings));
}
