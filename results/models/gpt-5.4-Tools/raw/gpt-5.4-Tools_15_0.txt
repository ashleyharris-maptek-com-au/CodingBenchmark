use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1.0e-9;
const SQRT3: f64 = 1.732_050_807_568_877_2;
const REF_FACE_AREA: f64 = 1.731_876;
const REF_CIRCUMRADIUS: f64 = 1.154_700_538_379_251_7;
const MAX_SCALE: f64 = 10.0;
const STACK_GAP: f64 = 24.0;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn add(self, other: Pt) -> Pt {
        Pt::new(self.x + other.x, self.y + other.y)
    }

    fn sub(self, other: Pt) -> Pt {
        Pt::new(self.x - other.x, self.y - other.y)
    }

    fn mul(self, k: f64) -> Pt {
        Pt::new(self.x * k, self.y * k)
    }

    fn dot(self, other: Pt) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Pt) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn norm2(self) -> f64 {
        self.dot(self)
    }

    fn norm(self) -> f64 {
        self.norm2().sqrt()
    }
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn dot(self, other: Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Vec3) -> Vec3 {
        Vec3::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn norm(self) -> f64 {
        self.dot(self).sqrt()
    }

    fn normalize(self) -> Vec3 {
        let n = self.norm();
        if n <= EPS {
            Vec3::new(0.0, 0.0, 1.0)
        } else {
            Vec3::new(self.x / n, self.y / n, self.z / n)
        }
    }

    fn mul(self, k: f64) -> Vec3 {
        Vec3::new(self.x * k, self.y * k, self.z * k)
    }

    fn add(self, other: Vec3) -> Vec3 {
        Vec3::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }

    fn sub(self, other: Vec3) -> Vec3 {
        Vec3::new(self.x - other.x, self.y - other.y, self.z - other.z)
    }
}

#[derive(Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quat {
    fn identity() -> Self {
        Self {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }

    fn normalize(self) -> Self {
        let n = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if n <= EPS {
            Self::identity()
        } else {
            Self {
                w: self.w / n,
                x: self.x / n,
                y: self.y / n,
                z: self.z / n,
            }
        }
    }

    fn from_basis(x_axis: Vec3, y_axis: Vec3, z_axis: Vec3) -> Self {
        let m00 = x_axis.x;
        let m01 = y_axis.x;
        let m02 = z_axis.x;
        let m10 = x_axis.y;
        let m11 = y_axis.y;
        let m12 = z_axis.y;
        let m20 = x_axis.z;
        let m21 = y_axis.z;
        let m22 = z_axis.z;
        let trace = m00 + m11 + m22;
        let q = if trace > 0.0 {
            let s = (trace + 1.0).sqrt() * 2.0;
            Quat {
                w: 0.25 * s,
                x: (m21 - m12) / s,
                y: (m02 - m20) / s,
                z: (m10 - m01) / s,
            }
        } else if m00 > m11 && m00 > m22 {
            let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
            Quat {
                w: (m21 - m12) / s,
                x: 0.25 * s,
                y: (m01 + m10) / s,
                z: (m02 + m20) / s,
            }
        } else if m11 > m22 {
            let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
            Quat {
                w: (m02 - m20) / s,
                x: (m01 + m10) / s,
                y: 0.25 * s,
                z: (m12 + m21) / s,
            }
        } else {
            let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
            Quat {
                w: (m10 - m01) / s,
                x: (m02 + m20) / s,
                y: (m12 + m21) / s,
                z: 0.25 * s,
            }
        };
        q.normalize()
    }
}

#[derive(Clone)]
struct Placement {
    center: Vec3,
    quat: Quat,
    scale: f64,
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut area = 0.0;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        area += a.cross(b);
    }
    0.5 * area
}

fn ensure_ccw(poly: &mut [Pt]) {
    if polygon_area(poly) < 0.0 {
        poly.reverse();
    }
}

fn point_in_triangle(p: Pt, a: Pt, b: Pt, c: Pt) -> bool {
    let ab = (b.sub(a)).cross(p.sub(a));
    let bc = (c.sub(b)).cross(p.sub(b));
    let ca = (a.sub(c)).cross(p.sub(c));
    (ab >= -EPS && bc >= -EPS && ca >= -EPS) || (ab <= EPS && bc <= EPS && ca <= EPS)
}

fn is_ear(poly: &[Pt], idxs: &[usize], pos: usize) -> bool {
    let m = idxs.len();
    let i0 = idxs[(pos + m - 1) % m];
    let i1 = idxs[pos];
    let i2 = idxs[(pos + 1) % m];
    let a = poly[i0];
    let b = poly[i1];
    let c = poly[i2];
    if (b.sub(a)).cross(c.sub(b)) <= EPS {
        return false;
    }
    for &idx in idxs {
        if idx == i0 || idx == i1 || idx == i2 {
            continue;
        }
        if point_in_triangle(poly[idx], a, b, c) {
            return false;
        }
    }
    true
}

fn triangulate_ear(poly: &[Pt]) -> Vec<[Pt; 3]> {
    let n = poly.len();
    let mut idxs: Vec<usize> = (0..n).collect();
    let mut tris = Vec::new();
    if n < 3 {
        return tris;
    }
    let mut guard = 0usize;
    while idxs.len() > 3 && guard < n * n {
        let mut clipped = false;
        for pos in 0..idxs.len() {
            if is_ear(poly, &idxs, pos) {
                let m = idxs.len();
                let i0 = idxs[(pos + m - 1) % m];
                let i1 = idxs[pos];
                let i2 = idxs[(pos + 1) % m];
                tris.push([poly[i0], poly[i1], poly[i2]]);
                idxs.remove(pos);
                clipped = true;
                break;
            }
        }
        if !clipped {
            break;
        }
        guard += 1;
    }
    if idxs.len() == 3 {
        tris.push([poly[idxs[0]], poly[idxs[1]], poly[idxs[2]]]);
    }
    tris
}

fn project_range(poly: &[Pt], axis: Pt) -> (f64, f64) {
    let mut lo = f64::INFINITY;
    let mut hi = f64::NEG_INFINITY;
    for &p in poly {
        let v = p.dot(axis);
        lo = lo.min(v);
        hi = hi.max(v);
    }
    (lo, hi)
}

fn clip_with_half_plane(poly: &[Pt], axis: Pt, cut: f64, keep_ge: bool) -> Vec<Pt> {
    let mut out = Vec::new();
    if poly.is_empty() {
        return out;
    }
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let da = a.dot(axis) - cut;
        let db = b.dot(axis) - cut;
        let ina = if keep_ge { da >= -EPS } else { da <= EPS };
        let inb = if keep_ge { db >= -EPS } else { db <= EPS };
        if ina && inb {
            out.push(b);
        } else if ina && !inb {
            let t = da / (da - db);
            out.push(a.add(b.sub(a).mul(t)));
        } else if !ina && inb {
            let t = da / (da - db);
            out.push(a.add(b.sub(a).mul(t)));
            out.push(b);
        }
    }
    out
}

fn strip_decompose(poly: &[Pt], max_width: f64) -> Vec<Vec<Pt>> {
    let mut axis = Pt::new(1.0, 0.0);
    let mut best_span = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let edge = b.sub(a);
        if edge.norm() <= EPS {
            continue;
        }
        let nrm = Pt::new(-edge.y, edge.x).mul(1.0 / edge.norm());
        let (lo, hi) = project_range(poly, nrm);
        let span = hi - lo;
        if span > best_span {
            best_span = span;
            axis = nrm;
        }
    }
    let (lo, hi) = project_range(poly, axis);
    let mut pieces = Vec::new();
    let steps = ((hi - lo) / max_width).ceil().max(1.0) as usize;
    for s in 0..steps {
        let a = lo + (hi - lo) * (s as f64) / (steps as f64);
        let b = lo + (hi - lo) * ((s + 1) as f64) / (steps as f64);
        let clipped1 = clip_with_half_plane(poly, axis, a, true);
        let clipped2 = clip_with_half_plane(&clipped1, axis, b, false);
        if clipped2.len() >= 3 && polygon_area(&clipped2).abs() > EPS {
            pieces.push(clipped2);
        }
    }
    pieces
}

fn triangle_area(a: Pt, b: Pt, c: Pt) -> f64 {
    0.5 * (b.sub(a)).cross(c.sub(a)).abs()
}

fn make_basis_from_triangle(a: Pt, b: Pt, c: Pt, sun: Vec3) -> Option<(Vec3, Vec3, Vec3, f64, Vec3)> {
    let center2 = Pt::new((a.x + b.x + c.x) / 3.0, (a.y + b.y + c.y) / 3.0);
    let e1 = b.sub(a);
    let e2 = c.sub(a);
    let len1 = e1.norm();
    let len2 = e2.norm();
    let mut x2 = if len1 >= len2 { e1 } else { e2 };
    let xnorm = x2.norm();
    if xnorm <= EPS {
        return None;
    }
    x2 = x2.mul(1.0 / xnorm);
    let perp2 = Pt::new(-x2.y, x2.x);
    let lift_x = Vec3::new(x2.x, x2.y, 0.0);
    let lift_y = Vec3::new(perp2.x, perp2.y, 0.0);
    let d = sun;
    let dxy = d.x * d.x + d.y * d.y;
    let map_x = lift_x.sub(d.mul(lift_x.z / d.z));
    let map_y = lift_y.sub(d.mul(lift_y.z / d.z));
    let proj_x = Pt::new(map_x.x, map_x.y);
    let proj_y = Pt::new(map_y.x, map_y.y);
    let det = proj_x.cross(proj_y);
    if det.abs() <= EPS || dxy <= EPS {
        let x3 = Vec3::new(x2.x, x2.y, 0.0);
        let y3 = Vec3::new(perp2.x, perp2.y, 0.0);
        let z3 = x3.cross(y3).normalize();
        return Some((x3, y3, z3, 1.0, Vec3::new(center2.x, center2.y, 0.0)));
    }
    let alpha = 1.0 / det;
    let beta = 1.0 / det;
    let x3 = Vec3::new(
        alpha * lift_x.x,
        alpha * lift_x.y,
        alpha * (d.x * x2.x + d.y * x2.y) / d.z,
    );
    let y3 = Vec3::new(
        beta * lift_y.x,
        beta * lift_y.y,
        beta * (-d.x * x2.y + d.y * x2.x) / d.z,
    );
    let z3 = x3.cross(y3).normalize();
    let origin = Vec3::new(center2.x, center2.y, 0.0);
    Some((x3, y3, z3, 1.0, origin))
}

fn fit_triangle(tri: [Pt; 3], sun: Vec3, layer: usize) -> Option<Placement> {
    let area = triangle_area(tri[0], tri[1], tri[2]);
    if area <= EPS {
        return None;
    }
    let scale = (area / REF_FACE_AREA).sqrt();
    if scale > MAX_SCALE + 1.0e-7 {
        return None;
    }
    let (x_axis, y_axis, z_axis, _, origin) = make_basis_from_triangle(tri[0], tri[1], tri[2], sun)?;
    let center2 = Pt::new(
        (tri[0].x + tri[1].x + tri[2].x) / 3.0,
        (tri[0].y + tri[1].y + tri[2].y) / 3.0,
    );
    let height = layer as f64 * STACK_GAP;
    let shift = Vec3::new(-sun.x / sun.z * height, -sun.y / sun.z * height, height);
    let center = Vec3::new(center2.x, center2.y, 0.0).add(shift);
    let quat = Quat::from_basis(x_axis.normalize(), y_axis.normalize(), z_axis);
    let _ = origin;
    Some(Placement {
        center,
        quat,
        scale,
    })
}

fn split_triangle_longest(tri: [Pt; 3]) -> ([Pt; 3], [Pt; 3]) {
    let d01 = tri[0].sub(tri[1]).norm2();
    let d12 = tri[1].sub(tri[2]).norm2();
    let d20 = tri[2].sub(tri[0]).norm2();
    if d01 >= d12 && d01 >= d20 {
        let mid = tri[0].add(tri[1]).mul(0.5);
        ([tri[0], mid, tri[2]], [mid, tri[1], tri[2]])
    } else if d12 >= d20 {
        let mid = tri[1].add(tri[2]).mul(0.5);
        ([tri[1], mid, tri[0]], [mid, tri[2], tri[0]])
    } else {
        let mid = tri[2].add(tri[0]).mul(0.5);
        ([tri[2], mid, tri[1]], [mid, tri[0], tri[1]])
    }
}

fn refine_triangle(mut tri: [Pt; 3], out: &mut Vec<[Pt; 3]>) {
    let area = triangle_area(tri[0], tri[1], tri[2]);
    let scale = (area / REF_FACE_AREA).sqrt();
    if scale <= MAX_SCALE {
        if (tri[1].sub(tri[0])).cross(tri[2].sub(tri[0])) < 0.0 {
            tri.swap(1, 2);
        }
        out.push(tri);
        return;
    }
    let (a, b) = split_triangle_longest(tri);
    refine_triangle(a, out);
    refine_triangle(b, out);
}

fn triangulate_adaptive(poly: &[Pt]) -> Vec<[Pt; 3]> {
    let area = polygon_area(poly).abs();
    let raw = if poly.len() <= 120 {
        triangulate_ear(poly)
    } else {
        let max_tri_area = REF_FACE_AREA * MAX_SCALE * MAX_SCALE * 0.9;
        let width = (2.0 * max_tri_area / SQRT3).sqrt() * 1.2;
        let mut acc = Vec::new();
        for piece in strip_decompose(poly, width) {
            let mut sub = triangulate_ear(&piece);
            acc.append(&mut sub);
        }
        if acc.is_empty() {
            triangulate_ear(poly)
        } else {
            acc
        }
    };
    let mut refined = Vec::new();
    for tri in raw {
        refine_triangle(tri, &mut refined);
    }
    if refined.is_empty() && area > EPS && poly.len() >= 3 {
        let p0 = poly[0];
        for i in 1..poly.len() - 1 {
            refine_triangle([p0, poly[i], poly[i + 1]], &mut refined);
        }
    }
    refined
}

fn read_input() -> Result<(Vec<Pt>, Vec3), String> {
    let mut input = String::new();
    io::stdin()
        .read_to_string(&mut input)
        .map_err(|e| format!("failed to read stdin: {e}"))?;
    let mut it = input.split_whitespace();
    let nv: usize = it
        .next()
        .ok_or_else(|| "missing vertex count".to_string())?
        .parse()
        .map_err(|e| format!("invalid vertex count: {e}"))?;
    if nv < 3 {
        return Err("polygon must have at least 3 vertices".to_string());
    }
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it
            .next()
            .ok_or_else(|| "missing vertex x".to_string())?
            .parse()
            .map_err(|e| format!("invalid vertex x: {e}"))?;
        let y: f64 = it
            .next()
            .ok_or_else(|| "missing vertex y".to_string())?
            .parse()
            .map_err(|e| format!("invalid vertex y: {e}"))?;
        poly.push(Pt::new(x, y));
    }
    let sx: f64 = it
        .next()
        .ok_or_else(|| "missing sun x".to_string())?
        .parse()
        .map_err(|e| format!("invalid sun x: {e}"))?;
    let sy: f64 = it
        .next()
        .ok_or_else(|| "missing sun y".to_string())?
        .parse()
        .map_err(|e| format!("invalid sun y: {e}"))?;
    let sz: f64 = it
        .next()
        .ok_or_else(|| "missing sun z".to_string())?
        .parse()
        .map_err(|e| format!("invalid sun z: {e}"))?;
    if sz.abs() <= EPS {
        return Err("sun z must be non-zero".to_string());
    }
    Ok((poly, Vec3::new(sx, sy, sz).normalize()))
}

fn dedup_polygon(poly: &[Pt]) -> Vec<Pt> {
    let mut out = Vec::new();
    for &p in poly {
        if out
            .last()
            .map(|q: &Pt| p.sub(*q).norm() > EPS)
            .unwrap_or(true)
        {
            out.push(p);
        }
    }
    if out.len() >= 2 && out[0].sub(out[out.len() - 1]).norm() <= EPS {
        out.pop();
    }
    out
}

fn main() {
    if let Err(msg) = run() {
        eprintln!("{msg}");
    }
}

fn run() -> Result<(), String> {
    let (poly_raw, sun) = read_input()?;
    let mut poly = dedup_polygon(&poly_raw);
    if poly.len() < 3 {
        return Err("degenerate polygon after deduplication".to_string());
    }
    ensure_ccw(&mut poly);
    let mut tris = triangulate_adaptive(&poly);
    tris.sort_by(|a, b| {
        let aa = triangle_area(a[0], a[1], a[2]);
        let bb = triangle_area(b[0], b[1], b[2]);
        bb.partial_cmp(&aa).unwrap_or(Ordering::Equal)
    });
    let mut placements = Vec::with_capacity(tris.len());
    for (i, tri) in tris.into_iter().enumerate() {
        if let Some(p) = fit_triangle(tri, sun, i) {
            placements.push(p);
        }
    }
    println!("{}", placements.len());
    for p in placements {
        println!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
            p.center.x,
            p.center.y,
            p.center.z,
            p.quat.w,
            p.quat.x,
            p.quat.y,
            p.quat.z,
            p.scale
        );
    }
    let _ = PI;
    let _ = REF_CIRCUMRADIUS;
    Ok(())
}
