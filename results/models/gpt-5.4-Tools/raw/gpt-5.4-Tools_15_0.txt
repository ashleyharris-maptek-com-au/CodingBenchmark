use std::cmp::Ordering;
use std::io::{self, Read};

const BASE_RADIUS_Z: f64 = 0.707;
const MAX_SCALE: f64 = 10.0;
const EPS: f64 = 1.0e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn add(self, other: Point) -> Point {
        Point::new(self.x + other.x, self.y + other.y)
    }

    fn sub(self, other: Point) -> Point {
        Point::new(self.x - other.x, self.y - other.y)
    }

    fn scale(self, value: f64) -> Point {
        Point::new(self.x * value, self.y * value)
    }

    fn dot(self, other: Point) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Point) -> f64 {
        self.x * other.y - self.y * other.x
    }

}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn scale(self, value: f64) -> Vec3 {
        Vec3::new(self.x * value, self.y * value, self.z * value)
    }

    fn dot(self, other: Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn length(self) -> f64 {
        self.dot(self).sqrt()
    }

    fn normalized(self) -> Vec3 {
        let len = self.length();
        if len <= EPS {
            Vec3::new(0.0, 0.0, 1.0)
        } else {
            self.scale(1.0 / len)
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn identity() -> Self {
        Self {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }
}

#[derive(Clone, Debug)]
struct Tetra {
    center: Vec3,
    rotation: Quaternion,
    scale: f64,
}

#[derive(Clone, Copy, Debug)]
struct GridFrame {
    origin: Point,
    axis_u: Point,
    axis_v: Point,
    min_u: f64,
    max_u: f64,
    min_v: f64,
    max_v: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv = match it.next() {
        Some(v) => v.parse::<usize>().unwrap(),
        None => return,
    };

    let mut polygon = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().unwrap().parse::<f64>().unwrap();
        let y = it.next().unwrap().parse::<f64>().unwrap();
        polygon.push(Point::new(x, y));
    }

    let sx = it.next().unwrap().parse::<f64>().unwrap();
    let sy = it.next().unwrap().parse::<f64>().unwrap();
    let sz = it.next().unwrap().parse::<f64>().unwrap();

    let mut sun = Vec3::new(sx, sy, sz).normalized();
    if sun.z >= -1.0e-6 {
        sun.z = -1.0e-6;
        sun = sun.normalized();
    }

    let hull = convex_hull(&polygon);
    let area = polygon_area(&hull).abs();
    if hull.len() < 3 || area <= EPS {
        println!("0");
        return;
    }

    let bbox = bounding_box(&hull);
    let diameter = ((bbox.1.x - bbox.0.x).powi(2) + (bbox.1.y - bbox.0.y).powi(2)).sqrt();
    let complexity = hull.len();
    let strategy = choose_strategy(complexity, area, diameter);

    let tetra = match strategy {
        Strategy::Single => single_cover(&hull, sun),
        Strategy::StripAligned => strip_cover(&hull, sun),
        Strategy::GridAdaptive => grid_cover(&hull, sun),
    };

    println!("{}", tetra.len());
    for t in tetra {
        println!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
            t.center.x,
            t.center.y,
            t.center.z,
            t.rotation.w,
            t.rotation.x,
            t.rotation.y,
            t.rotation.z,
            t.scale
        );
    }
}

#[derive(Clone, Copy)]
enum Strategy {
    Single,
    StripAligned,
    GridAdaptive,
}

fn choose_strategy(vertices: usize, area: f64, diameter: f64) -> Strategy {
    if vertices <= 8 && area <= 180.0 && diameter <= 24.0 {
        Strategy::Single
    } else if vertices <= 80 && area <= 25_000.0 {
        Strategy::StripAligned
    } else {
        Strategy::GridAdaptive
    }
}

fn single_cover(poly: &[Point], sun: Vec3) -> Vec<Tetra> {
    let frame = principal_frame(poly);
    let width = frame.max_u - frame.min_u;
    let height = frame.max_v - frame.min_v;
    let needed = width.max(height);
    let mut scale = needed / 2.0 + 1.0e-6;
    if scale > MAX_SCALE {
        return strip_cover(poly, sun);
    }
    if scale < 0.25 {
        scale = 0.25;
    }
    let center2 = frame
        .origin
        .add(frame.axis_u.scale(0.5 * (frame.min_u + frame.max_u)))
        .add(frame.axis_v.scale(0.5 * (frame.min_v + frame.max_v)));
    vec![make_tetra(center2, 0, 0, scale, sun)]
}

fn strip_cover(poly: &[Point], sun: Vec3) -> Vec<Tetra> {
    let frame = principal_frame(poly);
    let width = frame.max_u - frame.min_u;
    let height = frame.max_v - frame.min_v;

    let major_is_u = width >= height;
    let long_len = width.max(height);
    let short_len = width.min(height);
    let mut scale = (short_len / 2.0 + 0.5).min(MAX_SCALE);
    scale = scale.max(0.4);

    let shadow_width = 2.0 * scale;
    let pieces = ((long_len / shadow_width).ceil() as usize).max(1);
    if pieces == 1 && scale <= MAX_SCALE {
        return single_cover(poly, sun);
    }

    let long_axis = if major_is_u { frame.axis_u } else { frame.axis_v };
    let short_axis = if major_is_u { frame.axis_v } else { frame.axis_u };
    let long_min = if major_is_u { frame.min_u } else { frame.min_v };
    let long_max = if major_is_u { frame.max_u } else { frame.max_v };
    let short_mid = if major_is_u {
        0.5 * (frame.min_v + frame.max_v)
    } else {
        0.5 * (frame.min_u + frame.max_u)
    };

    let mut out = Vec::with_capacity(pieces);
    let step = if pieces == 1 {
        0.0
    } else {
        (long_max - long_min) / pieces as f64
    };
    for i in 0..pieces {
        let long_pos = if pieces == 1 {
            0.5 * (long_min + long_max)
        } else {
            long_min + (i as f64 + 0.5) * step
        };
        let pos = frame
            .origin
            .add(long_axis.scale(long_pos))
            .add(short_axis.scale(short_mid));
        out.push(make_tetra(pos, i as i32, 0, scale, sun));
    }
    out
}

fn grid_cover(poly: &[Point], sun: Vec3) -> Vec<Tetra> {
    let frame = principal_frame(poly);
    let width = frame.max_u - frame.min_u;
    let height = frame.max_v - frame.min_v;
    let mut scale = ((width.max(height)) / 12.0).clamp(1.2, MAX_SCALE);
    if width * height / (4.0 * scale * scale) > 1200.0 {
        scale = MAX_SCALE;
    }

    let cell = 2.0 * scale;
    let cols = ((width / cell).ceil() as usize).max(1);
    let rows = ((height / cell).ceil() as usize).max(1);

    let mut out = Vec::new();
    for iy in 0..rows {
        for ix in 0..cols {
            let u0 = frame.min_u + ix as f64 * cell;
            let v0 = frame.min_v + iy as f64 * cell;
            let u1 = (u0 + cell).min(frame.max_u);
            let v1 = (v0 + cell).min(frame.max_v);

            let rect = [
                frame_to_world(&frame, u0, v0),
                frame_to_world(&frame, u1, v0),
                frame_to_world(&frame, u1, v1),
                frame_to_world(&frame, u0, v1),
            ];
            if !rect_intersects_polygon(&rect, poly) {
                continue;
            }

            let cu = 0.5 * (u0 + u1);
            let cv = 0.5 * (v0 + v1);
            let pos = frame_to_world(&frame, cu, cv);
            out.push(make_tetra(pos, ix as i32, iy as i32, scale, sun));
        }
    }

    if out.is_empty() {
        out.push(make_tetra(
            frame_to_world(
                &frame,
                0.5 * (frame.min_u + frame.max_u),
                0.5 * (frame.min_v + frame.max_v),
            ),
            0,
            0,
            scale.min(MAX_SCALE),
            sun,
        ));
    }
    out
}

fn make_tetra(center_xy: Point, gx: i32, gy: i32, scale: f64, _sun: Vec3) -> Tetra {
    let stack_gap = 3.2 * MAX_SCALE;
    let height_gap = 2.0 * BASE_RADIUS_Z * scale + 0.5;
    let center = Vec3::new(
        center_xy.x + gx as f64 * stack_gap,
        center_xy.y + gy as f64 * stack_gap,
        20.0 + (gx + gy) as f64 * 0.25 + (gx as f64 * gy as f64).abs() * 0.01 + height_gap,
    );
    Tetra {
        center,
        rotation: Quaternion::identity(),
        scale,
    }
}

fn principal_frame(poly: &[Point]) -> GridFrame {
    let centroid = polygon_centroid(poly);
    let mut xx = 0.0;
    let mut xy = 0.0;
    let mut yy = 0.0;
    for &p in poly {
        let d = p.sub(centroid);
        xx += d.x * d.x;
        xy += d.x * d.y;
        yy += d.y * d.y;
    }

    let angle = if xy.abs() <= EPS && (xx - yy).abs() <= EPS {
        0.0
    } else {
        0.5 * (2.0 * xy).atan2(xx - yy)
    };
    let axis_u = Point::new(angle.cos(), angle.sin());
    let axis_v = Point::new(-angle.sin(), angle.cos());

    let mut min_u = f64::INFINITY;
    let mut max_u = f64::NEG_INFINITY;
    let mut min_v = f64::INFINITY;
    let mut max_v = f64::NEG_INFINITY;
    for &p in poly {
        let d = p.sub(centroid);
        let u = d.dot(axis_u);
        let v = d.dot(axis_v);
        min_u = min_u.min(u);
        max_u = max_u.max(u);
        min_v = min_v.min(v);
        max_v = max_v.max(v);
    }

    GridFrame {
        origin: centroid,
        axis_u,
        axis_v,
        min_u,
        max_u,
        min_v,
        max_v,
    }
}

fn frame_to_world(frame: &GridFrame, u: f64, v: f64) -> Point {
    frame.origin.add(frame.axis_u.scale(u)).add(frame.axis_v.scale(v))
}

fn polygon_area(poly: &[Point]) -> f64 {
    let mut area = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        area += a.cross(b);
    }
    0.5 * area
}

fn polygon_centroid(poly: &[Point]) -> Point {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut acc = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let cross = a.cross(b);
        acc += cross;
        cx += (a.x + b.x) * cross;
        cy += (a.y + b.y) * cross;
    }
    if acc.abs() <= EPS {
        let mut s = Point::new(0.0, 0.0);
        for &p in poly {
            s = s.add(p);
        }
        return s.scale(1.0 / poly.len() as f64);
    }
    let factor = 1.0 / (3.0 * acc);
    Point::new(cx * factor, cy * factor)
}

fn bounding_box(poly: &[Point]) -> (Point, Point) {
    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &p in poly {
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    (Point::new(min_x, min_y), Point::new(max_x, max_y))
}

fn convex_hull(points: &[Point]) -> Vec<Point> {
    let mut pts = points.to_vec();
    pts.sort_by(|a, b| cmp_f64(a.x, b.x).then(cmp_f64(a.y, b.y)));
    pts.dedup_by(|a, b| (a.x - b.x).abs() <= EPS && (a.y - b.y).abs() <= EPS);
    if pts.len() <= 1 {
        return pts;
    }

    let mut lower: Vec<Point> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            let cross = lower[l - 1].sub(lower[l - 2]).cross(p.sub(lower[l - 1]));
            if cross > EPS {
                break;
            }
            lower.pop();
        }
        lower.push(p);
    }

    let mut upper: Vec<Point> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            let cross = upper[l - 1].sub(upper[l - 2]).cross(p.sub(upper[l - 1]));
            if cross > EPS {
                break;
            }
            upper.pop();
        }
        upper.push(p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    if polygon_area(&lower) < 0.0 {
        lower.reverse();
    }
    lower
}

fn cmp_f64(a: f64, b: f64) -> Ordering {
    a.partial_cmp(&b).unwrap_or(Ordering::Equal)
}

fn rect_intersects_polygon(rect: &[Point; 4], poly: &[Point]) -> bool {
    for &r in rect {
        if point_in_polygon(r, poly) {
            return true;
        }
    }
    for &p in poly {
        if point_in_quad(p, rect) {
            return true;
        }
    }
    for i in 0..4 {
        let a1 = rect[i];
        let a2 = rect[(i + 1) % 4];
        for j in 0..poly.len() {
            let b1 = poly[j];
            let b2 = poly[(j + 1) % poly.len()];
            if segments_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    false
}

fn point_in_quad(p: Point, rect: &[Point; 4]) -> bool {
    let mut sign: f64 = 0.0;
    for i in 0..4 {
        let a = rect[i];
        let b = rect[(i + 1) % 4];
        let cross = b.sub(a).cross(p.sub(a));
        if cross.abs() <= EPS {
            continue;
        }
        if sign.abs() <= EPS {
            sign = cross;
        } else if sign * cross < -EPS {
            return false;
        }
    }
    true
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let on_edge = (b.sub(a)).cross(p.sub(a)).abs() <= EPS
            && (p.x - a.x).min(p.x - b.x) <= EPS
            && (p.x - a.x).max(p.x - b.x) >= -EPS
            && (p.y - a.y).min(p.y - b.y) <= EPS
            && (p.y - a.y).max(p.y - b.y) >= -EPS;
        if on_edge {
            return true;
        }
        let intersects = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x);
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let d1 = orient(a1, a2, b1);
    let d2 = orient(a1, a2, b2);
    let d3 = orient(b1, b2, a1);
    let d4 = orient(b1, b2, a2);

    if d1.abs() <= EPS && on_segment(a1, a2, b1) {
        return true;
    }
    if d2.abs() <= EPS && on_segment(a1, a2, b2) {
        return true;
    }
    if d3.abs() <= EPS && on_segment(b1, b2, a1) {
        return true;
    }
    if d4.abs() <= EPS && on_segment(b1, b2, a2) {
        return true;
    }

    (d1 > 0.0) != (d2 > 0.0) && (d3 > 0.0) != (d4 > 0.0)
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn on_segment(a: Point, b: Point, p: Point) -> bool {
    p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}
