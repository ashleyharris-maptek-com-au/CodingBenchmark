#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct City {
    double x;
    double y;
    std::size_t id;
};

static inline double sq_dist(const City& a, const City& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static std::vector<std::size_t> nearest_neighbor(const std::vector<City>& cities) {
    const std::size_t n = cities.size();
    std::vector<std::size_t> order;
    order.reserve(n);
    std::vector<unsigned char> used(n, 0);
    std::size_t current = 0;
    used[0] = 1;
    order.push_back(0);

    for (std::size_t step = 1; step < n; ++step) {
        double best = std::numeric_limits<double>::infinity();
        std::size_t best_idx = 0;
        for (std::size_t j = 1; j < n; ++j) {
            if (used[j]) {
                continue;
            }
            const double d = sq_dist(cities[current], cities[j]);
            if (d < best) {
                best = d;
                best_idx = j;
            }
        }
        used[best_idx] = 1;
        order.push_back(best_idx);
        current = best_idx;
    }
    return order;
}

static void two_opt_limited(const std::vector<City>& cities, std::vector<std::size_t>& order) {
    const std::size_t n = order.size();
    if (n < 4) {
        return;
    }

    const std::size_t max_passes = 6;
    const std::size_t neighborhood = (n <= 256 ? n - 1 : std::min<std::size_t>(64, n - 1));

    for (std::size_t pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (std::size_t i = 0; i + 3 < n; ++i) {
            const std::size_t a = order[i];
            const std::size_t b = order[i + 1];
            const std::size_t max_j = std::min<std::size_t>(n - 2, i + neighborhood);
            for (std::size_t j = i + 2; j <= max_j; ++j) {
                const std::size_t c = order[j];
                const std::size_t d = order[j + 1];
                const double old_len = sq_dist(cities[a], cities[b]) + sq_dist(cities[c], cities[d]);
                const double new_len = sq_dist(cities[a], cities[c]) + sq_dist(cities[b], cities[d]);
                if (new_len + 1e-12 < old_len) {
                    std::reverse(order.begin() + static_cast<std::ptrdiff_t>(i + 1),
                                 order.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    improved = true;
                }
            }
        }
        if (!improved) {
            break;
        }
    }
}

static std::vector<std::size_t> lexicographic_tour(const std::vector<City>& cities, std::size_t bins_per_axis) {
    const std::size_t n = cities.size();
    std::vector<std::size_t> order;
    order.reserve(n);

    if (n == 0) {
        return order;
    }

    double min_x = cities[0].x;
    double max_x = cities[0].x;
    double min_y = cities[0].y;
    double max_y = cities[0].y;
    for (const City& c : cities) {
        min_x = std::min(min_x, c.x);
        max_x = std::max(max_x, c.x);
        min_y = std::min(min_y, c.y);
        max_y = std::max(max_y, c.y);
    }

    const double span_x = std::max(1e-12, max_x - min_x);
    const double span_y = std::max(1e-12, max_y - min_y);

    struct Keyed {
        std::size_t idx;
        std::size_t bx;
        std::size_t by;
    };

    std::vector<Keyed> keyed;
    keyed.reserve(n);
    for (std::size_t i = 0; i < n; ++i) {
        const double fx = (cities[i].x - min_x) / span_x;
        const double fy = (cities[i].y - min_y) / span_y;
        std::size_t bx = static_cast<std::size_t>(fx * static_cast<double>(bins_per_axis));
        std::size_t by = static_cast<std::size_t>(fy * static_cast<double>(bins_per_axis));
        if (bx >= bins_per_axis) {
            bx = bins_per_axis - 1;
        }
        if (by >= bins_per_axis) {
            by = bins_per_axis - 1;
        }
        keyed.push_back({i, bx, by});
    }

    std::sort(keyed.begin(), keyed.end(), [&](const Keyed& a, const Keyed& b) {
        if (a.bx != b.bx) {
            return a.bx < b.bx;
        }
        const bool snake = (a.bx & 1U) != 0U;
        if (a.by != b.by) {
            return snake ? (a.by > b.by) : (a.by < b.by);
        }
        if (snake) {
            if (cities[a.idx].x != cities[b.idx].x) {
                return cities[a.idx].x > cities[b.idx].x;
            }
        } else {
            if (cities[a.idx].x != cities[b.idx].x) {
                return cities[a.idx].x < cities[b.idx].x;
            }
        }
        if (cities[a.idx].y != cities[b.idx].y) {
            return cities[a.idx].y < cities[b.idx].y;
        }
        return a.idx < b.idx;
    });

    std::size_t pos0 = 0;
    for (std::size_t i = 0; i < n; ++i) {
        if (keyed[i].idx == 0) {
            pos0 = i;
            break;
        }
    }
    for (std::size_t k = 0; k < n; ++k) {
        order.push_back(keyed[(pos0 + k) % n].idx);
    }
    return order;
}

static std::vector<std::size_t> adaptive_tour(const std::vector<City>& cities) {
    const std::size_t n = cities.size();
    if (n <= 2) {
        std::vector<std::size_t> order(n);
        std::iota(order.begin(), order.end(), 0);
        return order;
    }

    if (n <= 3000) {
        std::vector<std::size_t> order = nearest_neighbor(cities);
        two_opt_limited(cities, order);
        return order;
    }

    if (n <= 200000) {
        const double bins = std::sqrt(static_cast<double>(n) / 8.0);
        const std::size_t bins_per_axis = std::max<std::size_t>(8, static_cast<std::size_t>(bins));
        return lexicographic_tour(cities, bins_per_axis);
    }

    if (n <= 2000000) {
        const double bins = std::sqrt(static_cast<double>(n) / 32.0);
        const std::size_t bins_per_axis = std::max<std::size_t>(16, static_cast<std::size_t>(bins));
        return lexicographic_tour(cities, bins_per_axis);
    }

    const double bins = std::sqrt(static_cast<double>(n) / 128.0);
    const std::size_t bins_per_axis = std::max<std::size_t>(32, static_cast<std::size_t>(bins));
    return lexicographic_tour(cities, bins_per_axis);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::size_t n = 0;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::vector<City> cities;
    cities.reserve(n);
    for (std::size_t i = 0; i < n; ++i) {
        double x = 0.0;
        double y = 0.0;
        std::cin >> x >> y;
        cities.push_back({x, y, i});
    }

    std::vector<std::size_t> order = adaptive_tour(cities);

    if (order.empty() || order[0] != 0) {
        auto it = std::find(order.begin(), order.end(), 0);
        if (it != order.end()) {
            std::rotate(order.begin(), it, order.end());
        }
    }

    for (std::size_t i = 0; i < order.size(); ++i) {
        if (i) {
            std::cout << ' ';
        }
        std::cout << order[i];
    }
    std::cout << '\n';
    return 0;
}
