#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <unordered_set>
#include <utility>
#include <vector>

struct City {
    double x;
    double y;
    int id;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static std::vector<int> solve_trivial_exact(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n <= 1) return {0};
    const int start = 0;
    const int full_mask = 1 << (n - 1);
    const double inf = std::numeric_limits<double>::infinity();

    std::vector<std::vector<double>> dp(full_mask, std::vector<double>(n, inf));
    std::vector<std::vector<int>> parent(full_mask, std::vector<int>(n, -1));

    for (int i = 1; i < n; ++i) {
        const int mask = 1 << (i - 1);
        dp[mask][i] = dist2(cities[start], cities[i]);
        parent[mask][i] = start;
    }

    for (int mask = 1; mask < full_mask; ++mask) {
        for (int last = 1; last < n; ++last) {
            if ((mask & (1 << (last - 1))) == 0) continue;
            const int prev_mask = mask ^ (1 << (last - 1));
            if (prev_mask == 0) continue;
            for (int prev = 1; prev < n; ++prev) {
                if ((prev_mask & (1 << (prev - 1))) == 0) continue;
                const double cand = dp[prev_mask][prev] + dist2(cities[prev], cities[last]);
                if (cand < dp[mask][last]) {
                    dp[mask][last] = cand;
                    parent[mask][last] = prev;
                }
            }
        }
    }

    int best_last = 1;
    double best_cost = inf;
    const int all = full_mask - 1;
    for (int last = 1; last < n; ++last) {
        const double cand = dp[all][last] + dist2(cities[last], cities[start]);
        if (cand < best_cost) {
            best_cost = cand;
            best_last = last;
        }
    }

    std::vector<int> route(n);
    route[0] = start;
    int mask = all;
    int cur = best_last;
    for (int pos = n - 1; pos >= 1; --pos) {
        route[pos] = cur;
        const int prev = parent[mask][cur];
        mask ^= 1 << (cur - 1);
        cur = prev;
    }
    return route;
}

static void two_opt(std::vector<int>& route, const std::vector<City>& cities, int max_passes) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 2 < n; ++i) {
            const int a = route[i];
            const int b = route[i + 1];
            const int j_start = i + 2;
            const int j_end = (i == 0) ? n - 2 : n - 1;
            for (int j = j_start; j < j_end; ++j) {
                const int c = route[j];
                const int d = route[j + 1];
                const double before = dist2(cities[a], cities[b]) + dist2(cities[c], cities[d]);
                const double after = dist2(cities[a], cities[c]) + dist2(cities[b], cities[d]);
                if (after + 1e-12 < before) {
                    std::reverse(route.begin() + i + 1, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

static std::vector<int> solve_nearest_neighbor(std::vector<City> cities, bool do_two_opt) {
    const int n = static_cast<int>(cities.size());
    std::vector<int> route;
    route.reserve(n);
    std::vector<char> used(n, 0);

    int current = 0;
    route.push_back(0);
    used[0] = 1;

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double best_d = std::numeric_limits<double>::infinity();
        for (int i = 1; i < n; ++i) {
            if (used[i]) continue;
            const double d = dist2(cities[current], cities[i]);
            if (d < best_d) {
                best_d = d;
                best = i;
            }
        }
        if (best == -1) break;
        used[best] = 1;
        route.push_back(best);
        current = best;
    }

    if (do_two_opt) {
        two_opt(route, cities, n <= 200 ? 20 : 4);
        if (!route.empty() && route[0] != 0) {
            auto it = std::find(route.begin(), route.end(), 0);
            std::rotate(route.begin(), it, route.end());
        }
    }
    return route;
}

static inline uint32_t part1by1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}

static inline uint64_t morton2d(uint32_t x, uint32_t y) {
    return static_cast<uint64_t>(part1by1(y)) << 1 | part1by1(x);
}

static std::vector<int> solve_morton(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n <= 1) return {0};

    double min_x = cities[0].x, max_x = cities[0].x;
    double min_y = cities[0].y, max_y = cities[0].y;
    for (const City& c : cities) {
        min_x = std::min(min_x, c.x);
        max_x = std::max(max_x, c.x);
        min_y = std::min(min_y, c.y);
        max_y = std::max(max_y, c.y);
    }

    const double span_x = std::max(max_x - min_x, 1e-12);
    const double span_y = std::max(max_y - min_y, 1e-12);
    constexpr double scale = 65535.0;

    struct Keyed {
        uint64_t key;
        int idx;
    };

    std::vector<Keyed> keyed;
    keyed.reserve(n - 1);
    for (int i = 1; i < n; ++i) {
        const double nx = (cities[i].x - min_x) / span_x;
        const double ny = (cities[i].y - min_y) / span_y;
        const uint32_t qx = static_cast<uint32_t>(std::clamp(nx * scale, 0.0, scale));
        const uint32_t qy = static_cast<uint32_t>(std::clamp(ny * scale, 0.0, scale));
        keyed.push_back({morton2d(qx, qy), i});
    }

    std::sort(keyed.begin(), keyed.end(), [](const Keyed& a, const Keyed& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.idx < b.idx;
    });

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    for (const Keyed& k : keyed) route.push_back(k.idx);
    return route;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n)) return 0;
    if (n <= 0) return 0;

    std::vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }

    std::vector<int> route;

    if (n <= 20) {
        route = solve_trivial_exact(cities);
    } else if (n <= 400) {
        route = solve_nearest_neighbor(cities, true);
    } else if (n <= 5000) {
        route = solve_nearest_neighbor(cities, false);
        two_opt(route, cities, 2);
        if (!route.empty() && route[0] != 0) {
            auto it = std::find(route.begin(), route.end(), 0);
            std::rotate(route.begin(), it, route.end());
        }
    } else {
        route = solve_morton(cities);
        if (n <= 200000) {
            const int window = 32;
            for (int start = 1; start < n; start += window) {
                const int end = std::min(n, start + window);
                std::sort(route.begin() + start, route.begin() + end, [&](int a, int b) {
                    if (cities[a].x != cities[b].x) return cities[a].x < cities[b].x;
                    return cities[a].y < cities[b].y;
                });
            }
        }
    }

    if (static_cast<int>(route.size()) != n) {
        std::vector<char> seen(n, 0);
        std::vector<int> fixed;
        fixed.reserve(n);
        fixed.push_back(0);
        seen[0] = 1;
        for (int v : route) {
            if (v < 0 || v >= n || seen[v]) continue;
            fixed.push_back(v);
            seen[v] = 1;
        }
        for (int i = 1; i < n; ++i) {
            if (!seen[i]) fixed.push_back(i);
        }
        route.swap(fixed);
    }

    if (route[0] != 0) {
        auto it = std::find(route.begin(), route.end(), 0);
        if (it != route.end()) {
            std::rotate(route.begin(), it, route.end());
        } else {
            route.insert(route.begin(), 0);
            route.pop_back();
        }
    }

    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << route[i];
    }
    std::cout << '\n';
    return 0;
}
