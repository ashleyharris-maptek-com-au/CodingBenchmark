#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    uint32_t u = 0;
    uint32_t v = 0;
};

struct AdjEdge {
    uint32_t to = 0;
    uint32_t id = 0;
};

struct Solver {
    uint32_t n = 0;
    uint64_t m = 0;
    vector<Edge> edges;
    vector<vector<AdjEdge>> adj;

    vector<uint32_t> get_cut_from_component(const vector<char>& side) const {
        vector<uint32_t> cut;
        for (uint32_t i = 0; i < static_cast<uint32_t>(edges.size()); ++i) {
            const auto& e = edges[i];
            if (side[e.u] != side[e.v]) cut.push_back(i);
        }
        return cut;
    }

    vector<uint32_t> find_bridge_cut() const {
        vector<uint32_t> tin(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> low(n, 0);
        vector<uint32_t> parent_edge(n, numeric_limits<uint32_t>::max());
        uint32_t timer = 0;
        uint32_t bridge_id = numeric_limits<uint32_t>::max();

        struct Frame {
            uint32_t u;
            uint32_t parent;
            size_t it;
        };

        for (uint32_t root = 0; root < n && bridge_id == numeric_limits<uint32_t>::max(); ++root) {
            if (tin[root] != numeric_limits<uint32_t>::max()) continue;
            vector<Frame> st;
            st.push_back({root, root, 0});
            tin[root] = low[root] = timer++;
            while (!st.empty() && bridge_id == numeric_limits<uint32_t>::max()) {
                Frame& fr = st.back();
                uint32_t u = fr.u;
                if (fr.it == adj[u].size()) {
                    st.pop_back();
                    if (!st.empty()) {
                        uint32_t p = st.back().u;
                        low[p] = min(low[p], low[u]);
                        if (low[u] > tin[p]) {
                            bridge_id = parent_edge[u];
                            break;
                        }
                    }
                    continue;
                }
                AdjEdge ae = adj[u][fr.it++];
                if (ae.id == parent_edge[u]) continue;
                uint32_t v = ae.to;
                if (tin[v] == numeric_limits<uint32_t>::max()) {
                    parent_edge[v] = ae.id;
                    tin[v] = low[v] = timer++;
                    st.push_back({v, u, 0});
                } else {
                    low[u] = min(low[u], tin[v]);
                }
            }
        }

        if (bridge_id == numeric_limits<uint32_t>::max()) return {};
        return {bridge_id};
    }

    vector<uint32_t> find_two_edge_cut() const {
        vector<uint32_t> tin(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> low1(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> low2(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> low1_edge(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> low2_edge(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> parent(n, numeric_limits<uint32_t>::max());
        vector<uint32_t> parent_edge(n, numeric_limits<uint32_t>::max());
        uint32_t timer = 0;

        auto push_low = [&](uint32_t u, uint32_t val, uint32_t edge_id) {
            if (val < low1[u]) {
                if (edge_id != low1_edge[u]) {
                    low2[u] = low1[u];
                    low2_edge[u] = low1_edge[u];
                }
                low1[u] = val;
                low1_edge[u] = edge_id;
            } else if (edge_id != low1_edge[u] && val < low2[u]) {
                low2[u] = val;
                low2_edge[u] = edge_id;
            }
        };

        struct Frame {
            uint32_t u;
            size_t it;
        };

        for (uint32_t root = 0; root < n; ++root) {
            if (tin[root] != numeric_limits<uint32_t>::max()) continue;
            vector<Frame> st;
            st.push_back({root, 0});
            parent[root] = root;
            tin[root] = timer++;
            low1[root] = tin[root];
            low1_edge[root] = numeric_limits<uint32_t>::max();

            while (!st.empty()) {
                Frame& fr = st.back();
                uint32_t u = fr.u;
                if (fr.it == adj[u].size()) {
                    st.pop_back();
                    if (u != root) {
                        uint32_t p = parent[u];
                        push_low(p, low1[u], low1_edge[u]);
                        if (low2[u] != numeric_limits<uint32_t>::max()) {
                            push_low(p, low2[u], low2_edge[u]);
                        }
                        if (low1[u] == tin[p] && low2[u] >= tin[u]) {
                            return {parent_edge[u], low2_edge[u]};
                        }
                    }
                    continue;
                }

                AdjEdge ae = adj[u][fr.it++];
                if (ae.id == parent_edge[u]) continue;
                uint32_t v = ae.to;
                if (tin[v] == numeric_limits<uint32_t>::max()) {
                    parent[v] = u;
                    parent_edge[v] = ae.id;
                    tin[v] = timer++;
                    low1[v] = tin[v];
                    low1_edge[v] = numeric_limits<uint32_t>::max();
                    st.push_back({v, 0});
                } else if (tin[v] < tin[u]) {
                    push_low(u, tin[v], ae.id);
                }
            }
        }
        return {};
    }

    vector<uint32_t> stoer_wagner_exact() const {
        vector<vector<uint32_t>> w(n, vector<uint32_t>(n, 0));
        for (const auto& e : edges) {
            if (e.u == e.v) continue;
            ++w[e.u][e.v];
            ++w[e.v][e.u];
        }

        vector<uint32_t> verts(n);
        iota(verts.begin(), verts.end(), 0);
        vector<vector<uint32_t>> groups(n);
        for (uint32_t i = 0; i < n; ++i) groups[i] = {i};

        uint32_t best = numeric_limits<uint32_t>::max();
        vector<char> best_side(n, 0);

        while (verts.size() > 1) {
            vector<uint32_t> weight(n, 0);
            vector<char> added(n, 0);
            uint32_t prev = verts[0];
            for (size_t it = 0; it < verts.size(); ++it) {
                uint32_t sel = numeric_limits<uint32_t>::max();
                uint32_t best_w = 0;
                for (uint32_t v : verts) {
                    if (!added[v] && weight[v] >= best_w) {
                        best_w = weight[v];
                        sel = v;
                    }
                }
                if (it + 1 == verts.size()) {
                    if (best_w < best) {
                        best = best_w;
                        fill(best_side.begin(), best_side.end(), 0);
                        for (uint32_t x : groups[sel]) best_side[x] = 1;
                    }
                    if (best == 0) break;
                    for (uint32_t v : verts) {
                        if (v == sel || v == prev) continue;
                        w[prev][v] += w[sel][v];
                        w[v][prev] = w[prev][v];
                    }
                    auto& gp = groups[prev];
                    gp.insert(gp.end(), groups[sel].begin(), groups[sel].end());
                    verts.erase(remove(verts.begin(), verts.end(), sel), verts.end());
                    break;
                }
                added[sel] = 1;
                prev = sel;
                for (uint32_t v : verts) {
                    if (!added[v]) weight[v] += w[sel][v];
                }
            }
        }

        return get_cut_from_component(best_side);
    }

    vector<uint32_t> solve() const {
        if (n <= 1) return {};

        if (auto cut = find_bridge_cut(); !cut.empty()) return cut;
        if (auto cut = find_two_edge_cut(); !cut.empty()) return cut;

        return stoer_wagner_exact();
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Solver solver;
    uint64_t m64 = 0;
    if (!(cin >> solver.n >> m64)) {
        return 0;
    }
    solver.m = m64;
    solver.edges.reserve(static_cast<size_t>(m64));
    solver.adj.assign(solver.n, {});

    for (uint64_t i = 0; i < m64; ++i) {
        uint32_t u, v;
        cin >> u >> v;
        solver.edges.push_back({u, v});
        uint32_t id = static_cast<uint32_t>(i);
        solver.adj[u].push_back({v, id});
        solver.adj[v].push_back({u, id});
    }

    vector<uint32_t> cut = solver.solve();
    cout << cut.size() << '\n';
    for (uint32_t id : cut) {
        const auto& e = solver.edges[id];
        cout << e.u << ' ' << e.v << '\n';
    }
    return 0;
}
