#include <algorithm>
#include <cstdint>
#include <exception>
#include <iostream>
#include <limits>
#include <queue>
#include <stdexcept>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    uint32_t u;
    uint32_t v;
};

static uint64_t pack_pair(uint32_t a, uint32_t b) {
    if (a > b) {
        uint32_t t = a;
        a = b;
        b = t;
    }
    return (static_cast<uint64_t>(a) << 32) | static_cast<uint64_t>(b);
}

class Solver {
public:
    void run() {
        read_input();
        vector<Edge> cut = solve();
        cout << cut.size() << '\n';
        for (const Edge& e : cut) {
            cout << e.u << ' ' << e.v << '\n';
        }
    }

private:
    uint32_t n_ = 0;
    uint64_t m_ = 0;
    vector<Edge> edges_;
    vector<uint32_t> degree_;

    void read_input() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        uint64_t n64 = 0;
        if (!(cin >> n64 >> m_)) {
            throw runtime_error("failed to read graph size");
        }
        if (n64 == 0 || n64 > numeric_limits<uint32_t>::max()) {
            throw runtime_error("unsupported vertex count");
        }
        n_ = static_cast<uint32_t>(n64);
        degree_.assign(n_, 0);
        edges_.reserve(static_cast<size_t>(min<uint64_t>(m_, 50'000'000ULL)));

        for (uint64_t i = 0; i < m_; ++i) {
            uint64_t u64 = 0;
            uint64_t v64 = 0;
            cin >> u64 >> v64;
            if (!cin || u64 >= n_ || v64 >= n_) {
                throw runtime_error("invalid edge");
            }
            uint32_t u = static_cast<uint32_t>(u64);
            uint32_t v = static_cast<uint32_t>(v64);
            edges_.push_back({u, v});
            if (u != v) {
                ++degree_[u];
                ++degree_[v];
            }
        }
    }

    vector<Edge> solve() const {
        if (n_ <= 1) {
            return {};
        }

        vector<Edge> cut;
        if (try_leaf_cut(cut)) {
            return cut;
        }

        const uint64_t exact_edge_limit = 1'000'000ULL;
        const uint32_t exact_node_limit = 6'000U;
        if (m_ <= exact_edge_limit && n_ <= exact_node_limit) {
            return stoer_wagner_exact();
        }

        return min_degree_vertex_cut();
    }

    bool try_leaf_cut(vector<Edge>& cut) const {
        uint32_t v = min_degree_vertex();
        if (degree_[v] > 1) {
            return false;
        }
        cut.clear();
        for (const Edge& e : edges_) {
            if (e.u != e.v && (e.u == v || e.v == v)) {
                cut.push_back(e);
            }
        }
        return !cut.empty();
    }

    uint32_t min_degree_vertex() const {
        uint32_t best = 0;
        for (uint32_t i = 1; i < n_; ++i) {
            if (degree_[i] < degree_[best]) {
                best = i;
            }
        }
        return best;
    }

    vector<Edge> min_degree_vertex_cut() const {
        uint32_t pivot = min_degree_vertex();
        vector<Edge> cut;
        cut.reserve(degree_[pivot]);
        for (const Edge& e : edges_) {
            if (e.u != e.v && (e.u == pivot || e.v == pivot)) {
                cut.push_back(e);
            }
        }
        return cut;
    }

    vector<Edge> stoer_wagner_exact() const {
        vector<unordered_map<uint32_t, uint32_t> > g(n_);
        for (const Edge& e : edges_) {
            if (e.u == e.v) {
                continue;
            }
            ++g[e.u][e.v];
            ++g[e.v][e.u];
        }

        vector<uint32_t> verts(n_);
        for (uint32_t i = 0; i < n_; ++i) {
            verts[i] = i;
        }

        vector<vector<uint32_t> > groups(n_);
        for (uint32_t i = 0; i < n_; ++i) {
            groups[i].push_back(i);
        }

        uint64_t best_value = numeric_limits<uint64_t>::max();
        vector<uint8_t> best_side(n_, 0);

        while (verts.size() > 1) {
            vector<uint64_t> weights(n_, 0);
            vector<uint8_t> added(n_, 0);
            uint32_t prev = verts[0];
            uint32_t last = verts[0];

            for (size_t i = 0; i < verts.size(); ++i) {
                uint32_t sel = numeric_limits<uint32_t>::max();
                uint64_t sel_w = 0;
                for (uint32_t v : verts) {
                    if (!added[v] && (sel == numeric_limits<uint32_t>::max() || weights[v] > sel_w)) {
                        sel = v;
                        sel_w = weights[v];
                    }
                }

                added[sel] = 1;
                if (i + 1 == verts.size()) {
                    last = sel;
                    if (sel_w < best_value) {
                        best_value = sel_w;
                        fill(best_side.begin(), best_side.end(), 0);
                        for (uint32_t x : groups[last]) {
                            best_side[x] = 1;
                        }
                    }

                    for (unordered_map<uint32_t, uint32_t>::const_iterator it = g[last].begin(); it != g[last].end(); ++it) {
                        uint32_t to = it->first;
                        uint32_t w = it->second;
                        if (to == prev) {
                            continue;
                        }
                        g[prev][to] += w;
                        g[to][prev] += w;
                        g[to].erase(last);
                    }
                    g[last].clear();
                    groups[prev].insert(groups[prev].end(), groups[last].begin(), groups[last].end());
                    groups[last].clear();

                    for (size_t j = 0; j < verts.size(); ++j) {
                        if (verts[j] == last) {
                            verts[j] = verts.back();
                            verts.pop_back();
                            break;
                        }
                    }
                    break;
                }

                prev = sel;
                for (unordered_map<uint32_t, uint32_t>::const_iterator it = g[sel].begin(); it != g[sel].end(); ++it) {
                    if (!added[it->first]) {
                        weights[it->first] += it->second;
                    }
                }
            }
        }

        vector<Edge> cut;
        cut.reserve(static_cast<size_t>(best_value));
        for (const Edge& e : edges_) {
            if (e.u != e.v && best_side[e.u] != best_side[e.v]) {
                cut.push_back(e);
            }
        }
        return cut;
    }
};

int main() {
    try {
        Solver solver;
        solver.run();
        return 0;
    } catch (const exception&) {
        return 1;
    }
}
