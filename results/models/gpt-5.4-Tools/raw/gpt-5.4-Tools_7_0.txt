namespace {

constexpr double kEps = 1e-9;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}

    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double length(const Vec3& v) {
    return std::sqrt(dot(v, v));
}

Vec3 min_vec(const Vec3& a, const Vec3& b) {
    return {std::min(a.x, b.x), std::min(a.y, b.y), std::min(a.z, b.z)};
}

Vec3 max_vec(const Vec3& a, const Vec3& b) {
    return {std::max(a.x, b.x), std::max(a.y, b.y), std::max(a.z, b.z)};
}

struct Triangle {
    int a = 0;
    int b = 0;
    int c = 0;
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> faces;
};

struct AABB {
    Vec3 mn;
    Vec3 mx;
};

AABB compute_bounds(const Mesh& mesh) {
    AABB box;
    if (mesh.vertices.empty()) {
        box.mn = {0.0, 0.0, 0.0};
        box.mx = {0.0, 0.0, 0.0};
        return box;
    }
    box.mn = mesh.vertices.front();
    box.mx = mesh.vertices.front();
    for (const Vec3& v : mesh.vertices) {
        box.mn = min_vec(box.mn, v);
        box.mx = max_vec(box.mx, v);
    }
    return box;
}

bool point_in_aabb(const Vec3& p, const AABB& box, double pad = 0.0) {
    return p.x >= box.mn.x - pad && p.x <= box.mx.x + pad &&
           p.y >= box.mn.y - pad && p.y <= box.mx.y + pad &&
           p.z >= box.mn.z - pad && p.z <= box.mx.z + pad;
}

bool ray_intersects_triangle(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    const Vec3 e1 = v1 - v0;
    const Vec3 e2 = v2 - v0;
    const Vec3 pvec = cross(dir, e2);
    const double det = dot(e1, pvec);
    if (std::abs(det) < kEps) {
        return false;
    }
    const double inv_det = 1.0 / det;
    const Vec3 tvec = orig - v0;
    const double u = dot(tvec, pvec) * inv_det;
    if (u < -kEps || u > 1.0 + kEps) {
        return false;
    }
    const Vec3 qvec = cross(tvec, e1);
    const double v = dot(dir, qvec) * inv_det;
    if (v < -kEps || u + v > 1.0 + kEps) {
        return false;
    }
    const double t = dot(e2, qvec) * inv_det;
    return t > kEps;
}

bool point_inside_mesh(const Vec3& p, const Mesh& mesh, const AABB& box) {
    if (!point_in_aabb(p, box, kEps)) {
        return false;
    }
    const Vec3 dir{1.0, 0.17364817766, 0.31830988618};
    int hits = 0;
    for (const Triangle& tri : mesh.faces) {
        const Vec3& v0 = mesh.vertices[tri.a];
        const Vec3& v1 = mesh.vertices[tri.b];
        const Vec3& v2 = mesh.vertices[tri.c];
        if (ray_intersects_triangle(p, dir, v0, v1, v2)) {
            ++hits;
        }
    }
    return (hits & 1) != 0;
}

Vec3 triangle_centroid(const Mesh& mesh, const Triangle& tri) {
    return (mesh.vertices[tri.a] + mesh.vertices[tri.b] + mesh.vertices[tri.c]) / 3.0;
}

double triangle_area(const Vec3& a, const Vec3& b, const Vec3& c) {
    return 0.5 * length(cross(b - a, c - a));
}

Mesh deduplicate_vertices(const Mesh& mesh) {
    struct Key {
        long long x, y, z;
        bool operator==(const Key& other) const {
            return x == other.x && y == other.y && z == other.z;
        }
    };
    struct KeyHasher {
        size_t operator()(const Key& k) const noexcept {
            size_t h = std::hash<long long>{}(k.x);
            h ^= std::hash<long long>{}(k.y) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            h ^= std::hash<long long>{}(k.z) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            return h;
        }
    };

    AABB box = compute_bounds(mesh);
    const double span = std::max({box.mx.x - box.mn.x, box.mx.y - box.mn.y, box.mx.z - box.mn.z, 1.0});
    const double scale = 1e8 / span;

    unordered_map<Key, int, KeyHasher> map_idx;
    vector<Vec3> out_vertices;
    out_vertices.reserve(mesh.vertices.size());
    vector<int> remap(mesh.vertices.size(), -1);

    for (size_t i = 0; i < mesh.vertices.size(); ++i) {
        const Vec3& v = mesh.vertices[i];
        Key key{
            llround(v.x * scale),
            llround(v.y * scale),
            llround(v.z * scale)
        };
        auto it = map_idx.find(key);
        if (it == map_idx.end()) {
            int idx = static_cast<int>(out_vertices.size());
            out_vertices.push_back(v);
            map_idx.emplace(key, idx);
            remap[i] = idx;
        } else {
            remap[i] = it->second;
        }
    }

    vector<Triangle> out_faces;
    out_faces.reserve(mesh.faces.size());
    for (const Triangle& tri : mesh.faces) {
        Triangle mapped{remap[tri.a], remap[tri.b], remap[tri.c]};
        if (mapped.a == mapped.b || mapped.b == mapped.c || mapped.c == mapped.a) {
            continue;
        }
        const Vec3& a = out_vertices[mapped.a];
        const Vec3& b = out_vertices[mapped.b];
        const Vec3& c = out_vertices[mapped.c];
        if (triangle_area(a, b, c) <= kEps) {
            continue;
        }
        out_faces.push_back(mapped);
    }

    return {std::move(out_vertices), std::move(out_faces)};
}

Mesh build_union_surface(const Mesh& a, const Mesh& b) {
    const AABB box_a = compute_bounds(a);
    const AABB box_b = compute_bounds(b);

    Mesh out;
    out.vertices.reserve(a.vertices.size() + b.vertices.size());
    out.faces.reserve(a.faces.size() + b.faces.size());

    auto append_if_outside = [&](const Mesh& src, const Mesh& other, const AABB& other_box) {
        vector<int> local_to_global(src.vertices.size(), -1);
        for (const Triangle& tri : src.faces) {
            const Vec3 centroid = triangle_centroid(src, tri);
            if (point_inside_mesh(centroid, other, other_box)) {
                continue;
            }
            const int ids[3] = {tri.a, tri.b, tri.c};
            Triangle new_tri;
            int* dst[3] = {&new_tri.a, &new_tri.b, &new_tri.c};
            for (int i = 0; i < 3; ++i) {
                int src_idx = ids[i];
                if (local_to_global[src_idx] == -1) {
                    local_to_global[src_idx] = static_cast<int>(out.vertices.size());
                    out.vertices.push_back(src.vertices[src_idx]);
                }
                *dst[i] = local_to_global[src_idx];
            }
            out.faces.push_back(new_tri);
        }
    };

    append_if_outside(a, b, box_b);
    append_if_outside(b, a, box_a);
    return deduplicate_vertices(out);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) {
        return 0;
    }

    Mesh a, b;
    a.vertices.resize(VA);
    a.faces.resize(FA);
    b.vertices.resize(VB);
    b.faces.resize(FB);

    for (int i = 0; i < VA; ++i) {
        cin >> a.vertices[i].x >> a.vertices[i].y >> a.vertices[i].z;
    }
    for (int i = 0; i < FA; ++i) {
        cin >> a.faces[i].a >> a.faces[i].b >> a.faces[i].c;
    }
    for (int i = 0; i < VB; ++i) {
        cin >> b.vertices[i].x >> b.vertices[i].y >> b.vertices[i].z;
    }
    for (int i = 0; i < FB; ++i) {
        cin >> b.faces[i].a >> b.faces[i].b >> b.faces[i].c;
    }

    Mesh result = build_union_surface(a, b);

    cout << result.vertices.size() << ' ' << result.faces.size() << '\n';
    cout << std::setprecision(17);
    for (const Vec3& v : result.vertices) {
        cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const Triangle& tri : result.faces) {
        cout << tri.a << ' ' << tri.b << ' ' << tri.c << '\n';
    }

    return 0;
}
