constexpr double EPS = 1e-9;
constexpr double MERGE_EPS = 1e-7;
constexpr double AREA_EPS = 1e-12;
constexpr double INSIDE_EPS = 1e-8;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
};

Vec3 operator*(double s, const Vec3& v) { return v * s; }

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double norm2(const Vec3& v) {
    return dot(v, v);
}

double norm(const Vec3& v) {
    return std::sqrt(norm2(v));
}

Vec3 normalized(const Vec3& v) {
    double n = norm(v);
    if (n < EPS) return {0.0, 0.0, 0.0};
    return v / n;
}

struct AABB {
    Vec3 mn{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    Vec3 mx{-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity()};

    void expand(const Vec3& p) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.mn);
        expand(b.mx);
    }
};

bool overlaps(const AABB& a, const AABB& b) {
    return !(a.mx.x < b.mn.x - EPS || b.mx.x < a.mn.x - EPS ||
             a.mx.y < b.mn.y - EPS || b.mx.y < a.mn.y - EPS ||
             a.mx.z < b.mn.z - EPS || b.mx.z < a.mn.z - EPS);
}

struct Triangle {
    array<int, 3> v{};
};

struct Mesh {
    vector<Vec3> verts;
    vector<Triangle> faces;
};

struct Plane {
    Vec3 n;
    double d = 0.0;
};

Plane face_plane(const Mesh& mesh, const Triangle& t) {
    const Vec3& a = mesh.verts[t.v[0]];
    const Vec3& b = mesh.verts[t.v[1]];
    const Vec3& c = mesh.verts[t.v[2]];
    Vec3 n = cross(b - a, c - a);
    n = normalized(n);
    return {n, -dot(n, a)};
}

double signed_distance(const Plane& p, const Vec3& x) {
    return dot(p.n, x) + p.d;
}

Vec3 triangle_centroid(const Mesh& mesh, const Triangle& t) {
    return (mesh.verts[t.v[0]] + mesh.verts[t.v[1]] + mesh.verts[t.v[2]]) / 3.0;
}

AABB triangle_aabb(const Mesh& mesh, const Triangle& t) {
    AABB box;
    box.expand(mesh.verts[t.v[0]]);
    box.expand(mesh.verts[t.v[1]]);
    box.expand(mesh.verts[t.v[2]]);
    return box;
}

struct BVHNode {
    AABB box;
    int left = -1;
    int right = -1;
    int start = 0;
    int count = 0;
    bool leaf = false;
};

struct BVH {
    const Mesh* mesh = nullptr;
    vector<int> indices;
    vector<BVHNode> nodes;

    int build_node(int start, int count) {
        int idx = (int)nodes.size();
        nodes.push_back({});
        BVHNode& node = nodes.back();
        node.start = start;
        node.count = count;
        for (int i = start; i < start + count; ++i) node.box.expand(triangle_aabb(*mesh, mesh->faces[indices[i]]));
        if (count <= 8) {
            node.leaf = true;
            return idx;
        }
        AABB centroidBox;
        for (int i = start; i < start + count; ++i) centroidBox.expand(triangle_centroid(*mesh, mesh->faces[indices[i]]));
        Vec3 diag = centroidBox.mx - centroidBox.mn;
        int axis = 0;
        if (diag.y > diag.x && diag.y >= diag.z) axis = 1;
        else if (diag.z > diag.x && diag.z >= diag.y) axis = 2;
        int mid = start + count / 2;
        nth_element(indices.begin() + start, indices.begin() + mid, indices.begin() + start + count,
            [&](int a, int b) {
                Vec3 ca = triangle_centroid(*mesh, mesh->faces[a]);
                Vec3 cb = triangle_centroid(*mesh, mesh->faces[b]);
                if (axis == 0) return ca.x < cb.x;
                if (axis == 1) return ca.y < cb.y;
                return ca.z < cb.z;
            });
        node.left = build_node(start, mid - start);
        node.right = build_node(mid, start + count - mid);
        return idx;
    }

    void build(const Mesh& m) {
        mesh = &m;
        indices.resize(mesh->faces.size());
        iota(indices.begin(), indices.end(), 0);
        nodes.clear();
        if (!indices.empty()) build_node(0, (int)indices.size());
    }
};

struct Segment3 {
    Vec3 a;
    Vec3 b;
};

bool ray_intersects_triangle(const Vec3& orig, const Vec3& dir, const Vec3& a, const Vec3& b, const Vec3& c, double& t) {
    Vec3 e1 = b - a;
    Vec3 e2 = c - a;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (fabs(det) < EPS) return false;
    double invDet = 1.0 / det;
    Vec3 s = orig - a;
    double u = dot(s, p) * invDet;
    if (u < -EPS || u > 1.0 + EPS) return false;
    Vec3 q = cross(s, e1);
    double v = dot(dir, q) * invDet;
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    t = dot(e2, q) * invDet;
    return t > EPS;
}

bool point_inside_mesh(const Mesh& mesh, const BVH& bvh, const Vec3& p) {
    const Vec3 dir = normalized(Vec3(0.3721, 0.6153, 0.6947));
    int hits = 0;
    vector<int> stack;
    if (bvh.nodes.empty()) return false;
    stack.push_back(0);
    while (!stack.empty()) {
        int ni = stack.back();
        stack.pop_back();
        const BVHNode& node = bvh.nodes[ni];
        AABB rayBox;
        rayBox.mn = p;
        rayBox.mx = bvh.nodes[0].box.mx;
        if (!overlaps(node.box, rayBox) && !(p.x <= node.box.mx.x + EPS && p.y <= node.box.mx.y + EPS && p.z <= node.box.mx.z + EPS)) {
            continue;
        }
        if (node.leaf) {
            for (int i = node.start; i < node.start + node.count; ++i) {
                const Triangle& t = mesh.faces[bvh.indices[i]];
                double dist = 0.0;
                if (ray_intersects_triangle(p, dir, mesh.verts[t.v[0]], mesh.verts[t.v[1]], mesh.verts[t.v[2]], dist)) ++hits;
            }
        } else {
            stack.push_back(node.left);
            stack.push_back(node.right);
        }
    }
    return (hits & 1) != 0;
}

struct PolyPoint {
    Vec3 p;
};

Vec3 line_plane_intersection(const Vec3& a, const Vec3& b, const Plane& plane) {
    Vec3 ab = b - a;
    double denom = dot(plane.n, ab);
    if (fabs(denom) < EPS) return a;
    double t = -(dot(plane.n, a) + plane.d) / denom;
    t = max(0.0, min(1.0, t));
    return a + ab * t;
}

vector<Vec3> clip_polygon_outside_plane(const vector<Vec3>& poly, const Plane& plane) {
    vector<Vec3> out;
    if (poly.empty()) return out;
    for (size_t i = 0; i < poly.size(); ++i) {
        const Vec3& curr = poly[i];
        const Vec3& next = poly[(i + 1) % poly.size()];
        double dc = signed_distance(plane, curr);
        double dn = signed_distance(plane, next);
        bool currOut = dc >= -INSIDE_EPS;
        bool nextOut = dn >= -INSIDE_EPS;
        if (currOut && nextOut) {
            out.push_back(next);
        } else if (currOut && !nextOut) {
            out.push_back(line_plane_intersection(curr, next, plane));
        } else if (!currOut && nextOut) {
            out.push_back(line_plane_intersection(curr, next, plane));
            out.push_back(next);
        }
    }
    return out;
}

double polygon_area3(const vector<Vec3>& poly, const Vec3& normal) {
    Vec3 accum;
    for (size_t i = 0; i < poly.size(); ++i) accum += cross(poly[i], poly[(i + 1) % poly.size()]);
    return 0.5 * fabs(dot(accum, normalized(normal)));
}

vector<array<Vec3, 3>> triangulate_fan(const vector<Vec3>& poly) {
    vector<array<Vec3, 3>> tris;
    if (poly.size() < 3) return tris;
    for (size_t i = 1; i + 1 < poly.size(); ++i) tris.push_back({poly[0], poly[i], poly[i + 1]});
    return tris;
}

struct FaceFragments {
    vector<array<Vec3, 3>> tris;
};

vector<Vec3> dedupe_polygon(const vector<Vec3>& poly) {
    vector<Vec3> out;
    for (const auto& p : poly) {
        if (out.empty() || norm(out.back() - p) > MERGE_EPS) out.push_back(p);
    }
    if (out.size() > 1 && norm(out.front() - out.back()) <= MERGE_EPS) out.pop_back();
    return out;
}

vector<array<Vec3, 3>> subtract_triangle_by_planes(const array<Vec3, 3>& tri, const vector<Plane>& cutters) {
    vector<Vec3> poly = {tri[0], tri[1], tri[2]};
    for (const auto& plane : cutters) {
        poly = dedupe_polygon(clip_polygon_outside_plane(poly, plane));
        if (poly.size() < 3) return {};
    }
    Vec3 n = cross(tri[1] - tri[0], tri[2] - tri[0]);
    if (polygon_area3(poly, n) < AREA_EPS) return {};
    return triangulate_fan(poly);
}

bool segment_triangle_intersection(const Vec3& p0, const Vec3& p1, const Vec3& a, const Vec3& b, const Vec3& c, Vec3& out) {
    Vec3 dir = p1 - p0;
    double t;
    if (!ray_intersects_triangle(p0, dir, a, b, c, t)) return false;
    if (t < -EPS || t > 1.0 + EPS) return false;
    out = p0 + dir * t;
    return true;
}

vector<Plane> intersection_cutters_for_triangle(const Mesh& target, const Triangle& triT, const Mesh& cutter, const Triangle& triC) {
    vector<Plane> planes;
    Plane pc = face_plane(cutter, triC);
    array<Vec3, 3> cpts = {cutter.verts[triC.v[0]], cutter.verts[triC.v[1]], cutter.verts[triC.v[2]]};
    Vec3 cn = pc.n;
    for (int e = 0; e < 3; ++e) {
        Vec3 a = cpts[e];
        Vec3 b = cpts[(e + 1) % 3];
        Vec3 edge = b - a;
        Vec3 sideN = normalized(cross(edge, cn));
        Plane side{sideN, -dot(sideN, a)};
        planes.push_back(side);
    }
    return planes;
}

struct QuantKey {
    long long x, y, z;
    bool operator==(const QuantKey& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct QuantKeyHash {
    size_t operator()(const QuantKey& k) const {
        size_t h1 = hash<long long>{}(k.x);
        size_t h2 = hash<long long>{}(k.y);
        size_t h3 = hash<long long>{}(k.z);
        return h1 ^ (h2 * 1315423911u) ^ (h3 * 2654435761u);
    }
};

struct OutputBuilder {
    vector<Vec3> verts;
    vector<array<int, 3>> faces;
    unordered_map<QuantKey, int, QuantKeyHash> mapIdx;

    QuantKey key(const Vec3& p) const {
        double s = 1.0 / MERGE_EPS;
        return {
            llround(p.x * s),
            llround(p.y * s),
            llround(p.z * s)
        };
    }

    int add_vertex(const Vec3& p) {
        QuantKey k = key(p);
        auto it = mapIdx.find(k);
        if (it != mapIdx.end()) return it->second;
        int idx = (int)verts.size();
        verts.push_back(p);
        mapIdx.emplace(k, idx);
        return idx;
    }

    void add_triangle(const array<Vec3, 3>& tri) {
        Vec3 n = cross(tri[1] - tri[0], tri[2] - tri[0]);
        if (norm(n) < AREA_EPS) return;
        array<int, 3> f{add_vertex(tri[0]), add_vertex(tri[1]), add_vertex(tri[2])};
        if (f[0] == f[1] || f[1] == f[2] || f[0] == f[2]) return;
        faces.push_back(f);
    }
};

vector<int> query_overlaps(const BVH& a, const BVH& b, int triIndex) {
    vector<int> result;
    if (b.nodes.empty()) return result;
    AABB box = triangle_aabb(*a.mesh, a.mesh->faces[triIndex]);
    vector<int> stack{0};
    while (!stack.empty()) {
        int ni = stack.back();
        stack.pop_back();
        const BVHNode& node = b.nodes[ni];
        if (!overlaps(box, node.box)) continue;
        if (node.leaf) {
            for (int i = node.start; i < node.start + node.count; ++i) result.push_back(b.indices[i]);
        } else {
            stack.push_back(node.left);
            stack.push_back(node.right);
        }
    }
    return result;
}

void process_mesh_against_other(const Mesh& src, const BVH& srcBVH, const Mesh& other, const BVH& otherBVH, OutputBuilder& out) {
    for (size_t fi = 0; fi < src.faces.size(); ++fi) {
        const Triangle& t = src.faces[fi];
        array<Vec3, 3> tri = {src.verts[t.v[0]], src.verts[t.v[1]], src.verts[t.v[2]]};
        vector<Plane> cutters;
        vector<int> overlapsIdx = query_overlaps(srcBVH, otherBVH, (int)fi);
        cutters.reserve(overlapsIdx.size() * 3);
        for (int oj : overlapsIdx) {
            const Triangle& ot = other.faces[oj];
            auto pcs = intersection_cutters_for_triangle(src, t, other, ot);
            cutters.insert(cutters.end(), pcs.begin(), pcs.end());
        }
        auto fragments = subtract_triangle_by_planes(tri, cutters);
        if (fragments.empty()) {
            Vec3 c = (tri[0] + tri[1] + tri[2]) / 3.0;
            if (!point_inside_mesh(other, otherBVH, c)) out.add_triangle(tri);
            continue;
        }
        for (const auto& frag : fragments) {
            Vec3 c = (frag[0] + frag[1] + frag[2]) / 3.0;
            if (!point_inside_mesh(other, otherBVH, c)) out.add_triangle(frag);
        }
    }
}

Mesh read_mesh_part(int vCount, int fCount) {
    Mesh m;
    m.verts.resize(vCount);
    for (int i = 0; i < vCount; ++i) cin >> m.verts[i].x >> m.verts[i].y >> m.verts[i].z;
    m.faces.resize(fCount);
    for (int i = 0; i < fCount; ++i) cin >> m.faces[i].v[0] >> m.faces[i].v[1] >> m.faces[i].v[2];
    return m;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A = read_mesh_part(VA, FA);
    Mesh B = read_mesh_part(VB, FB);

    BVH bvhA, bvhB;
    bvhA.build(A);
    bvhB.build(B);

    OutputBuilder out;
    process_mesh_against_other(A, bvhA, B, bvhB, out);
    process_mesh_against_other(B, bvhB, A, bvhA, out);

    cout << out.verts.size() << ' ' << out.faces.size() << '\n';
    cout << setprecision(17);
    for (const auto& v : out.verts) cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    for (const auto& f : out.faces) cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    return 0;
}
