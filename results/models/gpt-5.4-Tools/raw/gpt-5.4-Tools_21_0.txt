using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        uint64_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x) ^ 0x9e3779b97f4a7c15ull);
            h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return static_cast<size_t>(h);
    }
};

struct State {
    int D = 0;
    int max_turns = 0;
    vector<int> bounds;
    deque<vector<int>> snake;
    unordered_set<vector<int>, VecHash> occupied;
    unordered_set<vector<int>, VecHash> obstacles;
    vector<vector<int>> foods;
    int turns_used = 0;
};

static bool same_pos(const vector<int>& a, const vector<int>& b) {
    return a == b;
}

static bool in_bounds(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static uint64_t product_limited(const vector<int>& bounds, uint64_t cap) {
    uint64_t prod = 1;
    for (int b : bounds) {
        if (b <= 0) return 0;
        if (prod > cap / static_cast<uint64_t>(b)) return cap;
        prod *= static_cast<uint64_t>(b);
    }
    return prod;
}

static int manhattan(const vector<int>& a, const vector<int>& b) {
    int dist = 0;
    for (size_t i = 0; i < a.size(); ++i) dist += abs(a[i] - b[i]);
    return dist;
}

static bool contains_food(const vector<vector<int>>& foods, const vector<int>& p, int* idx = nullptr) {
    for (int i = 0; i < static_cast<int>(foods.size()); ++i) {
        if (foods[i] == p) {
            if (idx) *idx = i;
            return true;
        }
    }
    return false;
}

static vector<pair<int, int>> enumerate_moves(int D) {
    vector<pair<int, int>> moves;
    moves.reserve(static_cast<size_t>(D) * 2);
    for (int axis = 0; axis < D; ++axis) {
        moves.push_back({axis, -1});
        moves.push_back({axis, +1});
    }
    return moves;
}

static vector<int> apply_move(const vector<int>& p, int axis, int dir) {
    vector<int> q = p;
    q[axis] += dir;
    return q;
}

static bool is_legal_step(const State& st, const vector<int>& next, bool grows) {
    if (!in_bounds(next, st.bounds)) return false;
    if (st.obstacles.find(next) != st.obstacles.end()) return false;

    if (!grows && !st.snake.empty() && next == st.snake.back()) return true;
    return st.occupied.find(next) == st.occupied.end();
}

static bool step_snake(State& st, int axis, int dir) {
    vector<int> next = apply_move(st.snake.front(), axis, dir);
    int food_idx = -1;
    bool grows = contains_food(st.foods, next, &food_idx);
    if (!is_legal_step(st, next, grows)) return false;

    st.snake.push_front(next);
    st.occupied.insert(next);
    if (grows) {
        st.foods.erase(st.foods.begin() + food_idx);
    } else {
        st.occupied.erase(st.snake.back());
        st.snake.pop_back();
    }
    ++st.turns_used;
    return true;
}

struct ParentMove {
    vector<int> parent;
    int axis = -1;
    int dir = 0;
};

static bool tail_reachable_after_move(
    const State& st,
    const vector<int>& candidate,
    bool grows
) {
    unordered_set<vector<int>, VecHash> blocked = st.occupied;
    blocked.erase(st.snake.front());
    if (!grows && !st.snake.empty()) blocked.erase(st.snake.back());
    blocked.insert(candidate);

    vector<int> target = grows ? st.snake.back() : (st.snake.size() >= 2 ? st.snake[st.snake.size() - 2] : candidate);
    if (candidate == target) return true;

    queue<vector<int>> q;
    unordered_set<vector<int>, VecHash> seen;
    q.push(candidate);
    seen.insert(candidate);

    auto moves = enumerate_moves(st.D);
    while (!q.empty()) {
        vector<int> cur = q.front();
        q.pop();
        for (const auto& mv : moves) {
            vector<int> nxt = apply_move(cur, mv.first, mv.second);
            if (!in_bounds(nxt, st.bounds)) continue;
            if (st.obstacles.find(nxt) != st.obstacles.end()) continue;
            if (blocked.find(nxt) != blocked.end() && nxt != target) continue;
            if (seen.insert(nxt).second) {
                if (nxt == target) return true;
                q.push(nxt);
            }
        }
    }
    return false;
}

static bool find_path_to_target(
    const State& st,
    const vector<int>& target,
    vector<pair<int, int>>& path
) {
    const uint64_t volume = product_limited(st.bounds, 250000);
    if (volume >= 250000) return false;

    struct Node {
        int f = 0;
        int g = 0;
        vector<int> pos;
    };
    struct Cmp {
        bool operator()(const Node& a, const Node& b) const noexcept {
            if (a.f != b.f) return a.f > b.f;
            return a.g < b.g;
        }
    };

    priority_queue<Node, vector<Node>, Cmp> pq;
    unordered_map<vector<int>, int, VecHash> best_g;
    unordered_map<vector<int>, ParentMove, VecHash> parent;

    auto start = st.snake.front();
    pq.push({manhattan(start, target), 0, start});
    best_g[start] = 0;

    auto moves = enumerate_moves(st.D);
    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();
        auto it_g = best_g.find(cur.pos);
        if (it_g == best_g.end() || cur.g != it_g->second) continue;
        if (cur.pos == target) break;

        for (const auto& mv : moves) {
            vector<int> nxt = apply_move(cur.pos, mv.first, mv.second);
            bool grows = (nxt == target) && contains_food(st.foods, target);
            if (!is_legal_step(st, nxt, grows)) continue;

            int ng = cur.g + 1;
            auto it = best_g.find(nxt);
            if (it != best_g.end() && it->second <= ng) continue;
            best_g[nxt] = ng;
            parent[nxt] = {cur.pos, mv.first, mv.second};
            pq.push({ng + manhattan(nxt, target), ng, nxt});
        }
    }

    if (best_g.find(target) == best_g.end()) return false;

    vector<pair<int, int>> rev;
    vector<int> cur = target;
    while (cur != start) {
        const auto& pm = parent[cur];
        rev.push_back({pm.axis, pm.dir});
        cur = pm.parent;
    }
    reverse(rev.begin(), rev.end());

    State sim = st;
    vector<pair<int, int>> verified;
    for (const auto& mv : rev) {
        vector<int> nxt = apply_move(sim.snake.front(), mv.first, mv.second);
        bool grows = contains_food(sim.foods, nxt);
        if (!is_legal_step(sim, nxt, grows)) return false;
        if (!tail_reachable_after_move(sim, nxt, grows) && nxt != target) return false;
        if (!step_snake(sim, mv.first, mv.second)) return false;
        verified.push_back(mv);
    }

    path = move(verified);
    return true;
}

static pair<int, int> choose_survival_move(const State& st) {
    auto moves = enumerate_moves(st.D);
    vector<int> head = st.snake.front();
    pair<int, int> best = {0, 1};
    int best_score = numeric_limits<int>::min();

    vector<int> nearest_food;
    int nearest_food_dist = numeric_limits<int>::max();
    for (const auto& food : st.foods) {
        int d = manhattan(head, food);
        if (d < nearest_food_dist) {
            nearest_food_dist = d;
            nearest_food = food;
        }
    }

    for (const auto& mv : moves) {
        vector<int> nxt = apply_move(head, mv.first, mv.second);
        bool grows = contains_food(st.foods, nxt);
        if (!is_legal_step(st, nxt, grows)) continue;

        int score = 0;
        if (tail_reachable_after_move(st, nxt, grows)) score += 1000000;
        if (!nearest_food.empty()) score -= manhattan(nxt, nearest_food) * 100;

        int local_freedom = 0;
        for (const auto& mv2 : moves) {
            vector<int> n2 = apply_move(nxt, mv2.first, mv2.second);
            bool grows2 = contains_food(st.foods, n2);
            if (in_bounds(n2, st.bounds) &&
                st.obstacles.find(n2) == st.obstacles.end() &&
                (st.occupied.find(n2) == st.occupied.end() || (!grows && n2 == st.snake.back()))) {
                ++local_freedom;
                if (grows2) score += 5;
            }
        }
        score += local_freedom * 1000;

        int wall_margin = 0;
        for (int i = 0; i < st.D; ++i) {
            wall_margin += min(nxt[i], st.bounds[i] - 1 - nxt[i]);
        }
        score += wall_margin * 10;

        if (grows) score += 25000;
        if (score > best_score) {
            best_score = score;
            best = mv;
        }
    }

    if (best_score == numeric_limits<int>::min()) {
        for (const auto& mv : moves) {
            vector<int> nxt = apply_move(head, mv.first, mv.second);
            bool grows = contains_food(st.foods, nxt);
            if (is_legal_step(st, nxt, grows)) return mv;
        }
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    State st;
    if (!(cin >> st.D >> st.max_turns)) return 0;

    st.bounds.resize(st.D);
    for (int i = 0; i < st.D; ++i) cin >> st.bounds[i];

    int snake_length = 0;
    vector<int> start(st.D);
    cin >> snake_length;
    for (int i = 0; i < st.D; ++i) cin >> start[i];

    st.snake.clear();
    st.snake.push_back(start);
    st.occupied.insert(start);

    int F = 0;
    cin >> F;
    st.foods.resize(F, vector<int>(st.D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < st.D; ++j) cin >> st.foods[i][j];
    }

    int O = 0;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        vector<int> obs(st.D);
        for (int j = 0; j < st.D; ++j) cin >> obs[j];
        st.obstacles.insert(obs);
    }

    vector<pair<int, int>> current_plan;

    while (st.turns_used < st.max_turns && !st.foods.empty()) {
        if (current_plan.empty()) {
            vector<int> best_food;
            int best_food_dist = numeric_limits<int>::max();
            for (const auto& food : st.foods) {
                int d = manhattan(st.snake.front(), food);
                if (d < best_food_dist) {
                    best_food_dist = d;
                    best_food = food;
                }
            }

            if (!best_food.empty()) {
                find_path_to_target(st, best_food, current_plan);
            }
        }

        pair<int, int> mv;
        if (!current_plan.empty()) {
            mv = current_plan.front();
            current_plan.erase(current_plan.begin());
            vector<int> nxt = apply_move(st.snake.front(), mv.first, mv.second);
            bool grows = contains_food(st.foods, nxt);
            if (!is_legal_step(st, nxt, grows)) {
                current_plan.clear();
                mv = choose_survival_move(st);
            }
        } else {
            mv = choose_survival_move(st);
        }

        cout << mv.first << ' ' << mv.second << '\n';
        if (!step_snake(st, mv.first, mv.second)) break;
    }

    return 0;
}
