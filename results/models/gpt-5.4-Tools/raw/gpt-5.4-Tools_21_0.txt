#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        uint64_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x)) + 0x9e3779b97f4a7c15ull;
            h ^= y;
            h *= 1099511628211ull;
        }
        return static_cast<size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

struct NodeInfo {
    int parent = -1;
    int move_index = -1;
};

class Solver {
public:
    void run() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        if (!(cin >> D >> max_turns)) {
            return;
        }
        bounds.resize(D);
        for (int i = 0; i < D; ++i) cin >> bounds[i];

        int snake_length;
        cin >> snake_length;
        vector<int> start(D);
        for (int i = 0; i < D; ++i) cin >> start[i];

        int F;
        cin >> F;
        foods.resize(F, vector<int>(D));
        for (int i = 0; i < F; ++i) {
            for (int j = 0; j < D; ++j) cin >> foods[i][j];
        }

        int O;
        cin >> O;
        obstacles.reserve(O * 2 + 1);
        for (int i = 0; i < O; ++i) {
            vector<int> p(D);
            for (int j = 0; j < D; ++j) cin >> p[j];
            obstacles.insert(p);
        }

        snake.assign(snake_length, start);
        occupied.reserve(static_cast<size_t>(snake_length) * 2 + 1);
        for (const auto& s : snake) occupied.insert(s);

        for (int turn = 0; turn < max_turns; ++turn) {
            if (foods.empty()) break;
            Move mv = choose_move(turn);
            cout << mv.axis << ' ' << mv.dir << '\n';
            apply_move(mv);
        }
    }

private:
    int D = 0;
    int max_turns = 0;
    vector<int> bounds;
    vector<vector<int>> foods;
    unordered_set<vector<int>, VecHash> obstacles;
    deque<vector<int>> snake;
    unordered_set<vector<int>, VecHash> occupied;

    static constexpr int INF = numeric_limits<int>::max() / 4;

    int manhattan(const vector<int>& a, const vector<int>& b) const {
        int64_t sum = 0;
        for (int i = 0; i < D; ++i) sum += llabs(static_cast<long long>(a[i]) - b[i]);
        if (sum > INF) return INF;
        return static_cast<int>(sum);
    }

    bool in_bounds(const vector<int>& p) const {
        for (int i = 0; i < D; ++i) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    }

    bool is_food(const vector<int>& p, int* idx = nullptr) const {
        for (int i = 0; i < static_cast<int>(foods.size()); ++i) {
            if (foods[i] == p) {
                if (idx) *idx = i;
                return true;
            }
        }
        return false;
    }

    bool legal_destination(const vector<int>& next, bool grows) const {
        if (!in_bounds(next)) return false;
        if (obstacles.find(next) != obstacles.end()) return false;
        if (occupied.find(next) == occupied.end()) return true;
        const vector<int>& tail = snake.back();
        return !grows && next == tail;
    }

    vector<Move> generate_moves(const vector<int>& head) const {
        vector<Move> moves;
        moves.reserve(static_cast<size_t>(D) * 2);
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> next = head;
                next[axis] += dir;
                int food_idx = -1;
                bool grows = is_food(next, &food_idx);
                if (legal_destination(next, grows)) {
                    moves.push_back({axis, dir});
                }
            }
        }
        return moves;
    }

    int free_degree(const vector<int>& p, const vector<int>& tail, bool grows) const {
        int degree = 0;
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> q = p;
                q[axis] += dir;
                if (!in_bounds(q)) continue;
                if (obstacles.find(q) != obstacles.end()) continue;
                auto it = occupied.find(q);
                if (it == occupied.end()) {
                    ++degree;
                } else if (!grows && q == tail) {
                    ++degree;
                }
            }
        }
        return degree;
    }

    int flood_space(const vector<int>& start, int limit, bool grows) const {
        unordered_set<vector<int>, VecHash> blocked = occupied;
        if (!grows && !snake.empty()) blocked.erase(snake.back());
        blocked.erase(start);

        queue<vector<int>> q;
        unordered_set<vector<int>, VecHash> seen;
        q.push(start);
        seen.insert(start);
        int count = 0;

        while (!q.empty() && count < limit) {
            vector<int> cur = q.front();
            q.pop();
            ++count;
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    vector<int> nxt = cur;
                    nxt[axis] += dir;
                    if (!in_bounds(nxt)) continue;
                    if (obstacles.find(nxt) != obstacles.end()) continue;
                    if (blocked.find(nxt) != blocked.end()) continue;
                    if (seen.insert(nxt).second) q.push(nxt);
                }
            }
        }
        return count;
    }

    int nearest_food_distance(const vector<int>& p) const {
        int best = INF;
        for (const auto& f : foods) best = min(best, manhattan(p, f));
        return best;
    }

    Move choose_move(int turn) {
        const vector<int>& head = snake.front();
        auto moves = generate_moves(head);
        if (moves.empty()) return {0, 1};

        vector<int> step = plan_path_bfs();
        if (!step.empty()) {
            return decode_move(head, step);
        }

        int best_score = numeric_limits<int>::min();
        Move best = moves.front();

        for (const Move& mv : moves) {
            vector<int> next = head;
            next[mv.axis] += mv.dir;
            bool grows = is_food(next, nullptr);
            const vector<int>& tail = snake.back();
            int degree = free_degree(next, tail, grows);
            int space_limit = min(20000, max(64, static_cast<int>(snake.size()) * 4));
            int space = flood_space(next, space_limit, grows);
            int food_dist = nearest_food_distance(next);

            int score = 0;
            score += space * 8;
            score += degree * 50;
            score -= food_dist * 12;
            if (grows) score += 100000;
            if (space < static_cast<int>(snake.size()) && foods.size() > 1) score -= 5000;
            score -= turn % 7;

            if (score > best_score) {
                best_score = score;
                best = mv;
            }
        }
        return best;
    }

    vector<int> plan_path_bfs() {
        const vector<int>& start = snake.front();
        vector<pair<int, vector<int>>> ordered_foods;
        ordered_foods.reserve(foods.size());
        for (int i = 0; i < static_cast<int>(foods.size()); ++i) {
            ordered_foods.push_back({manhattan(start, foods[i]), foods[i]});
        }
        sort(ordered_foods.begin(), ordered_foods.end(), [](const auto& a, const auto& b) {
            return a.first < b.first;
        });

        size_t food_checks = min<size_t>(ordered_foods.size(), D <= 4 ? 4 : 2);
        int expansion_limit = D <= 4 ? 250000 : (D <= 8 ? 120000 : 40000);

        for (size_t fi = 0; fi < food_checks; ++fi) {
            const vector<int>& goal = ordered_foods[fi].second;
            auto path = bfs_to_goal(start, goal, expansion_limit);
            if (!path.empty()) return path;
        }
        return {};
    }

    vector<int> bfs_to_goal(const vector<int>& start, const vector<int>& goal, int expansion_limit) {
        unordered_set<vector<int>, VecHash> blocked = occupied;
        if (!snake.empty()) blocked.erase(snake.back());
        blocked.erase(start);
        blocked.erase(goal);

        vector<vector<int>> nodes;
        vector<NodeInfo> parent;
        nodes.reserve(4096);
        parent.reserve(4096);
        unordered_map<vector<int>, int, VecHash> id;
        id.reserve(4096);
        queue<int> q;

        nodes.push_back(start);
        parent.push_back({-1, -1});
        id[start] = 0;
        q.push(0);
        int expansions = 0;

        while (!q.empty() && expansions < expansion_limit) {
            int cur_id = q.front();
            q.pop();
            const vector<int>& cur = nodes[cur_id];
            if (cur == goal) break;
            ++expansions;

            vector<pair<int, vector<int>>> nexts;
            nexts.reserve(static_cast<size_t>(D) * 2);
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    vector<int> nxt = cur;
                    nxt[axis] += dir;
                    if (!in_bounds(nxt)) continue;
                    if (obstacles.find(nxt) != obstacles.end()) continue;
                    if (blocked.find(nxt) != blocked.end() && nxt != goal) continue;
                    if (id.find(nxt) != id.end()) continue;
                    nexts.push_back({manhattan(nxt, goal), std::move(nxt)});
                }
            }
            sort(nexts.begin(), nexts.end(), [](const auto& a, const auto& b) {
                return a.first < b.first;
            });

            for (auto& entry : nexts) {
                const vector<int>& nxt = entry.second;
                int new_id = static_cast<int>(nodes.size());
                id[nxt] = new_id;
                nodes.push_back(nxt);
                parent.push_back({cur_id, -1});
                q.push(new_id);
                if (nxt == goal) {
                    vector<int> path_ids;
                    int at = new_id;
                    while (at != 0) {
                        path_ids.push_back(at);
                        at = parent[at].parent;
                    }
                    reverse(path_ids.begin(), path_ids.end());
                    return nodes[path_ids.front()];
                }
            }
        }
        return {};
    }

    Move decode_move(const vector<int>& from, const vector<int>& to) const {
        for (int axis = 0; axis < D; ++axis) {
            int diff = to[axis] - from[axis];
            if (diff == 1) return {axis, 1};
            if (diff == -1) return {axis, -1};
        }
        return {0, 1};
    }

    void apply_move(const Move& mv) {
        vector<int> next = snake.front();
        next[mv.axis] += mv.dir;
        int food_idx = -1;
        bool grows = is_food(next, &food_idx);

        snake.push_front(next);
        occupied.insert(next);

        if (grows) {
            foods.erase(foods.begin() + food_idx);
        } else {
            occupied.erase(snake.back());
            snake.pop_back();
        }
    }
};

int main() {
    Solver solver;
    solver.run();
    return 0;
}
