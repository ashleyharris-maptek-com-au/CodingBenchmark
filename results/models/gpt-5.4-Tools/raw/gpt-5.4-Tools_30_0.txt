import sys
from array import array


def xorshift64(state):
    state ^= (state << 13) & 0xFFFFFFFFFFFFFFFF
    state ^= state >> 7
    state ^= (state << 17) & 0xFFFFFFFFFFFFFFFF
    return state & 0xFFFFFFFFFFFFFFFF


def lit_true(lit, assign):
    v = lit if lit > 0 else -lit
    return assign[v] if lit > 0 else 1 - assign[v]


def main():
    data = sys.stdin.buffer
    header_line = data.readline().lstrip(b"\xef\xbb\xbf")
    header = header_line.split()
    if not header:
        return
    n = int(header[0])
    m = int(header[1])

    c1 = array('i')
    c2 = array('i')
    c3 = array('i')

    pos_head = array('i', [-1]) * (n + 1)
    neg_head = array('i', [-1]) * (n + 1)
    pos_next = array('i')
    neg_next = array('i')
    pos_clause = array('i')
    neg_clause = array('i')

    for clause_idx in range(m):
        line = data.readline()
        while line and not line.strip():
            line = data.readline()
        if not line:
            m = clause_idx
            break
        a_s, b_s, c_s = line.split()
        a = int(a_s)
        b = int(b_s)
        c = int(c_s)
        c1.append(a)
        c2.append(b)
        c3.append(c)

        if a > 0:
            pos_clause.append(clause_idx)
            pos_next.append(pos_head[a])
            pos_head[a] = len(pos_clause) - 1
        else:
            v = -a
            neg_clause.append(clause_idx)
            neg_next.append(neg_head[v])
            neg_head[v] = len(neg_clause) - 1

        if b > 0:
            pos_clause.append(clause_idx)
            pos_next.append(pos_head[b])
            pos_head[b] = len(pos_clause) - 1
        else:
            v = -b
            neg_clause.append(clause_idx)
            neg_next.append(neg_head[v])
            neg_head[v] = len(neg_clause) - 1

        if c > 0:
            pos_clause.append(clause_idx)
            pos_next.append(pos_head[c])
            pos_head[c] = len(pos_clause) - 1
        else:
            v = -c
            neg_clause.append(clause_idx)
            neg_next.append(neg_head[v])
            neg_head[v] = len(neg_clause) - 1

    assign = bytearray(n + 1)
    rng = 88172645463393265
    for v in range(1, n + 1):
        rng = xorshift64(rng)
        assign[v] = rng & 1

    sat_count = array('B', [0]) * m
    unsat = array('i')
    unsat_pos = array('i', [-1]) * m

    for i in range(m):
        cnt = 0
        cnt += lit_true(c1[i], assign)
        cnt += lit_true(c2[i], assign)
        cnt += lit_true(c3[i], assign)
        sat_count[i] = cnt
        if cnt == 0:
            unsat_pos[i] = len(unsat)
            unsat.append(i)

    def unsat_add(ci):
        if unsat_pos[ci] == -1:
            unsat_pos[ci] = len(unsat)
            unsat.append(ci)

    def unsat_remove(ci):
        pos = unsat_pos[ci]
        if pos == -1:
            return
        last = unsat[-1]
        unsat[pos] = last
        unsat_pos[last] = pos
        unsat.pop()
        unsat_pos[ci] = -1

    def break_score(var):
        score = 0
        node = pos_head[var] if assign[var] else neg_head[var]
        nxt = pos_next if assign[var] else neg_next
        arr = pos_clause if assign[var] else neg_clause
        while node != -1:
            ci = arr[node]
            if sat_count[ci] == 1:
                score += 1
            node = nxt[node]
        return score

    def flip(var):
        old_val = assign[var]
        true_arr = pos_clause if old_val else neg_clause
        true_next = pos_next if old_val else neg_next
        false_arr = neg_clause if old_val else pos_clause
        false_next = neg_next if old_val else pos_next

        node = true_arr is pos_clause and pos_head[var] or neg_head[var]
        while node != -1:
            ci = true_arr[node]
            prev = sat_count[ci]
            now = prev - 1
            sat_count[ci] = now
            if now == 0:
                unsat_add(ci)
            node = true_next[node]

        node = false_arr is pos_clause and pos_head[var] or neg_head[var]
        while node != -1:
            ci = false_arr[node]
            prev = sat_count[ci]
            now = prev + 1
            sat_count[ci] = now
            if prev == 0:
                unsat_remove(ci)
            node = false_next[node]

        assign[var] = 1 - old_val

    max_steps = min(20000000, max(1000000, m * 3))
    for _ in range(max_steps):
        if not unsat:
            break
        rng = xorshift64(rng)
        ci = unsat[rng % len(unsat)]
        a = c1[ci]
        b = c2[ci]
        c = c3[ci]
        va = a if a > 0 else -a
        vb = b if b > 0 else -b
        vc = c if c > 0 else -c

        ba = break_score(va)
        bb = break_score(vb)
        bc = break_score(vc)

        if ba <= bb and ba <= bc:
            best = va
        elif bb <= bc:
            best = vb
        else:
            best = vc

        if ba != 0 and bb != 0 and bc != 0:
            rng = xorshift64(rng)
            r = rng % 10
            if r == 0:
                best = va
            elif r == 1:
                best = vb
            elif r == 2:
                best = vc

        flip(best)

    if unsat:
        # Fallback repair sweep. This is slower but only used if local search
        # did not converge quickly.
        changed = True
        while unsat and changed:
            changed = False
            limit = len(unsat)
            for _ in range(limit):
                ci = unsat[0]
                a = c1[ci]
                v = a if a > 0 else -a
                before = len(unsat)
                flip(v)
                if len(unsat) <= before:
                    changed = True
                else:
                    flip(v)
                    b = c2[ci]
                    v = b if b > 0 else -b
                    before = len(unsat)
                    flip(v)
                    if len(unsat) <= before:
                        changed = True
                    else:
                        flip(v)
                        c = c3[ci]
                        v = c if c > 0 else -c
                        before = len(unsat)
                        flip(v)
                        if len(unsat) <= before:
                            changed = True
                        else:
                            flip(v)
                if not unsat:
                    break

    out = sys.stdout.write
    out("SAT\n")
    out(" ".join("1" if assign[i] else "0" for i in range(1, n + 1)))
    out("\n")


if __name__ == "__main__":
    main()
