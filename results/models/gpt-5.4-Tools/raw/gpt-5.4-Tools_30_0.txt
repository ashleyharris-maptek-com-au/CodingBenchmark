import sys
from array import array


def ints_from_stdin():
    data = sys.stdin.buffer.read()
    num = 0
    sign = 1
    in_num = False
    for b in data:
        if 48 <= b <= 57:
            num = num * 10 + (b - 48)
            in_num = True
        elif b == 45:
            sign = -1
        elif in_num:
            yield sign * num
            num = 0
            sign = 1
            in_num = False
        else:
            sign = 1
    if in_num:
        yield sign * num


def lit_true(lit, assignment):
    v = lit if lit > 0 else -lit
    val = assignment[v]
    return val if lit > 0 else 1 - val


def main():
    it = ints_from_stdin()
    try:
        n = next(it)
        m = next(it)
    except StopIteration:
        return

    a = array('i')
    b = array('i')
    c = array('i')
    pos_count = array('I', [0]) * (n + 1)
    neg_count = array('I', [0]) * (n + 1)

    append_a = a.append
    append_b = b.append
    append_c = c.append

    for _ in range(m):
        x = next(it)
        y = next(it)
        z = next(it)
        append_a(x)
        append_b(y)
        append_c(z)
        if x > 0:
            pos_count[x] += 1
        else:
            neg_count[-x] += 1
        if y > 0:
            pos_count[y] += 1
        else:
            neg_count[-y] += 1
        if z > 0:
            pos_count[z] += 1
        else:
            neg_count[-z] += 1

    assignment = bytearray(n + 1)
    for i in range(1, n + 1):
        assignment[i] = 1 if pos_count[i] >= neg_count[i] else 0

    limit = 8
    for _pass in range(limit):
        changed = False
        unsat = 0
        for i in range(m):
            x = a[i]
            y = b[i]
            z = c[i]
            tx = lit_true(x, assignment)
            ty = lit_true(y, assignment)
            tz = lit_true(z, assignment)
            if tx | ty | tz:
                continue
            unsat += 1

            vx = x if x > 0 else -x
            vy = y if y > 0 else -y
            vz = z if z > 0 else -z

            sx = pos_count[vx] + neg_count[vx]
            sy = pos_count[vy] + neg_count[vy]
            sz = pos_count[vz] + neg_count[vz]

            best_v = vx
            best_s = sx
            if sy < best_s:
                best_v = vy
                best_s = sy
            if sz < best_s:
                best_v = vz
                best_s = sz

            assignment[best_v] ^= 1
            changed = True

        if unsat == 0:
            break
        if not changed:
            break

    out = sys.stdout.write
    out('SAT\n')
    out(' '.join('1' if assignment[i] else '0' for i in range(1, n + 1)))
    out('\n')


if __name__ == '__main__':
    main()
