#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <vector>

using namespace std;

class MaximumIndependentSetSolver {
public:
    explicit MaximumIndependentSetSolver(int n) : n_(n), adj_(n, vector<uint64_t>(blocksFor(n), 0)) {}

    void addEdge(int u, int v) {
        if (u == v) return;
        setBit(adj_[u], v);
        setBit(adj_[v], u);
    }

    vector<int> solve() {
        if (n_ == 0) return {};
        buildComplement();
        vector<int> all(n_);
        iota(all.begin(), all.end(), 0);
        expand(all, {});
        return best_;
    }

private:
    int n_;
    vector<vector<uint64_t>> adj_;
    vector<vector<uint64_t>> comp_;
    vector<int> best_;

    static int blocksFor(int n) {
        return (n + 63) >> 6;
    }

    static void setBit(vector<uint64_t>& bits, int pos) {
        bits[pos >> 6] |= 1ULL << (pos & 63);
    }

    static bool testBit(const vector<uint64_t>& bits, int pos) {
        return (bits[pos >> 6] >> (pos & 63)) & 1ULL;
    }

    void buildComplement() {
        const int blocks = blocksFor(n_);
        comp_.assign(n_, vector<uint64_t>(blocks, 0));
        for (int v = 0; v < n_; ++v) {
            for (int b = 0; b < blocks; ++b) {
                comp_[v][b] = ~adj_[v][b];
            }
            comp_[v][v >> 6] &= ~(1ULL << (v & 63));
        }
        if (n_ & 63) {
            const uint64_t mask = (1ULL << (n_ & 63)) - 1ULL;
            for (int v = 0; v < n_; ++v) comp_[v].back() &= mask;
        }
    }

    vector<int> intersectNeighbors(const vector<int>& vertices, int v) const {
        vector<int> res;
        res.reserve(vertices.size());
        for (int u : vertices) {
            if (testBit(comp_[v], u)) res.push_back(u);
        }
        return res;
    }

    vector<int> greedyOrder(const vector<int>& vertices) const {
        vector<int> order = vertices;
        sort(order.begin(), order.end(), [&](int a, int b) {
            int da = 0, db = 0;
            for (int u : vertices) {
                da += testBit(comp_[a], u);
                db += testBit(comp_[b], u);
            }
            if (da != db) return da > db;
            return a < b;
        });
        return order;
    }

    void colorSort(const vector<int>& vertices, vector<int>& order, vector<int>& colors) const {
        order.clear();
        colors.clear();
        vector<int> remaining = greedyOrder(vertices);
        vector<int> colorClass;

        int color = 0;
        while (!remaining.empty()) {
            ++color;
            colorClass.clear();
            for (int v : remaining) {
                bool ok = true;
                for (int u : colorClass) {
                    if (testBit(comp_[v], u)) {
                        ok = false;
                        break;
                    }
                }
                if (ok) colorClass.push_back(v);
            }

            for (int v : colorClass) {
                order.push_back(v);
                colors.push_back(color);
            }

            vector<int> next;
            next.reserve(remaining.size() - colorClass.size());
            for (int v : remaining) {
                bool chosen = false;
                for (int u : colorClass) {
                    if (u == v) {
                        chosen = true;
                        break;
                    }
                }
                if (!chosen) next.push_back(v);
            }
            remaining.swap(next);
        }
    }

    void expand(const vector<int>& candidates, vector<int> current) {
        if (candidates.empty()) {
            if (current.size() > best_.size()) best_ = std::move(current);
            return;
        }

        vector<int> order, colors;
        colorSort(candidates, order, colors);

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (current.size() + static_cast<size_t>(colors[i]) <= best_.size()) return;

            int v = order[i];
            current.push_back(v);
            vector<int> next = intersectNeighbors(candidates, v);
            if (current.size() + next.size() > best_.size()) {
                expand(next, current);
            } else if (current.size() > best_.size()) {
                best_ = current;
            }
            current.pop_back();
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    MaximumIndependentSetSolver solver(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (0 <= u && u < n && 0 <= v && v < n) solver.addEdge(u, v);
    }

    vector<int> independentSet = solver.solve();
    sort(independentSet.begin(), independentSet.end());

    cout << independentSet.size() << '\n';
    for (size_t i = 0; i < independentSet.size(); ++i) {
        if (i) cout << ' ';
        cout << independentSet[i];
    }
    cout << '\n';
    return 0;
}
