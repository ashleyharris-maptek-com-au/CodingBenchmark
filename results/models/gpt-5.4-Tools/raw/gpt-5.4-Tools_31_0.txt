use std::collections::{BTreeMap, HashMap};
use std::io::{self, Read};

#[derive(Clone)]
struct Entry {
    prev_sum: i64,
    index: usize,
}

fn reconstruct(mut sum: i64, parents: &HashMap<i64, Entry>) -> Vec<usize> {
    let mut indices = Vec::new();
    while sum != 0 {
        if let Some(entry) = parents.get(&sum) {
            indices.push(entry.index);
            sum = entry.prev_sum;
        } else {
            break;
        }
    }
    indices.reverse();
    indices
}

fn solve_small_exact(values: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }
    let left_n = n / 2;
    let right_n = n - left_n;
    if left_n > 24 || right_n > 24 {
        return None;
    }

    let mut left_sums: HashMap<i64, u64> = HashMap::with_capacity(1usize << left_n);
    let left_limit = 1u64 << left_n;
    let mut mask = 0u64;
    while mask < left_limit {
        let mut sum = 0i64;
        let mut bit = 0usize;
        while bit < left_n {
            if ((mask >> bit) & 1) != 0 {
                sum += values[bit].0;
            }
            bit += 1;
        }
        left_sums.entry(sum).or_insert(mask);
        mask += 1;
    }

    let right_limit = 1u64 << right_n;
    let mut right_mask = 0u64;
    while right_mask < right_limit {
        let mut sum = 0i64;
        let mut bit = 0usize;
        while bit < right_n {
            if ((right_mask >> bit) & 1) != 0 {
                sum += values[left_n + bit].0;
            }
            bit += 1;
        }
        let need = target - sum;
        if let Some(&left_mask) = left_sums.get(&need) {
            let mut result = Vec::new();
            let mut i = 0usize;
            while i < left_n {
                if ((left_mask >> i) & 1) != 0 {
                    result.push(values[i].1);
                }
                i += 1;
            }
            let mut j = 0usize;
            while j < right_n {
                if ((right_mask >> j) & 1) != 0 {
                    result.push(values[left_n + j].1);
                }
                j += 1;
            }
            result.sort_unstable();
            return Some(result);
        }
        right_mask += 1;
    }
    None
}

fn solve_nonnegative_bitset(nums: &[i64], target: usize) -> Option<Vec<usize>> {
    let word_count = (target / 64) + 1;
    let mut bits = vec![0u64; word_count];
    bits[0] = 1;
    let mut parents: HashMap<usize, (usize, usize)> = HashMap::new();

    for (idx, &value) in nums.iter().enumerate() {
        let shift = match usize::try_from(value) {
            Ok(v) => v,
            Err(_) => continue,
        };
        if shift == 0 || shift > target {
            continue;
        }
        let word_shift = shift / 64;
        let bit_shift = shift % 64;
        let prev = bits.clone();
        let mut word = word_count;
        while word > 0 {
            let w = word - 1;
            let mut shifted = 0u64;
            if w >= word_shift {
                shifted |= prev[w - word_shift] << bit_shift;
                if bit_shift != 0 && w > word_shift {
                    shifted |= prev[w - word_shift - 1] >> (64 - bit_shift);
                }
            }
            let new_bits = shifted & !bits[w];
            if new_bits != 0 {
                let mut local = new_bits;
                while local != 0 {
                    let tz = local.trailing_zeros() as usize;
                    let sum = (w * 64) + tz;
                    if sum <= target {
                        parents.insert(sum, (sum - shift, idx));
                    }
                    local &= local - 1;
                }
                bits[w] |= new_bits;
            }
            word -= 1;
        }
        if ((bits[target / 64] >> (target % 64)) & 1) != 0 {
            let mut result = Vec::new();
            let mut sum = target;
            while sum != 0 {
                if let Some(&(prev_sum, pick_idx)) = parents.get(&sum) {
                    result.push(pick_idx);
                    sum = prev_sum;
                } else {
                    return None;
                }
            }
            result.reverse();
            return Some(result);
        }
    }
    None
}

fn solve_hashmap_dp(values: &[(i64, usize)], target: i64, cap: usize) -> Option<Vec<usize>> {
    let mut states: HashMap<i64, Entry> = HashMap::new();
    states.insert(
        0,
        Entry {
            prev_sum: 0,
            index: usize::MAX,
        },
    );

    for &(value, original_idx) in values {
        let snapshot: Vec<i64> = states.keys().copied().collect();
        let mut additions = Vec::new();
        for sum in snapshot {
            let next = sum + value;
            if states.contains_key(&next) {
                continue;
            }
            additions.push((
                next,
                Entry {
                    prev_sum: sum,
                    index: original_idx,
                },
            ));
            if next == target {
                for (s, entry) in additions {
                    states.insert(s, entry);
                }
                return Some(reconstruct(target, &states));
            }
        }
        for (s, entry) in additions {
            states.insert(s, entry);
        }
        if states.len() > cap {
            let mut ranked: Vec<i64> = states.keys().copied().collect();
            ranked.sort_by_key(|&s| (target - s).abs());
            ranked.truncate(cap);
            let keep: BTreeMap<i64, ()> = ranked.into_iter().map(|s| (s, ())).collect();
            states.retain(|sum, _| keep.contains_key(sum));
            states.entry(0).or_insert(Entry {
                prev_sum: 0,
                index: usize::MAX,
            });
        }
    }

    if states.contains_key(&target) {
        Some(reconstruct(target, &states))
    } else {
        None
    }
}

fn greedy_signed(values: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let mut order: Vec<(i64, usize)> = values.to_vec();
    order.sort_by_key(|&(v, _)| -v.abs());
    let mut sum = 0i64;
    let mut picked = Vec::new();
    for &(value, idx) in &order {
        let current_gap = (target - sum).abs();
        let next_gap = (target - (sum + value)).abs();
        if next_gap < current_gap {
            sum += value;
            picked.push(idx);
            if sum == target {
                picked.sort_unstable();
                return Some(picked);
            }
        }
    }
    None
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("NO");
        return;
    }
    let mut it = input.split_whitespace();
    let n = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("NO");
            return;
        }
    };
    let target = match it.next().and_then(|s| s.parse::<i64>().ok()) {
        Some(v) => v,
        None => {
            println!("NO");
            return;
        }
    };

    let mut nums = Vec::with_capacity(n);
    let mut i = 0usize;
    while i < n {
        match it.next().and_then(|s| s.parse::<i64>().ok()) {
            Some(v) => nums.push(v),
            None => {
                println!("NO");
                return;
            }
        }
        i += 1;
    }

    if target == 0 {
        println!("YES");
        println!();
        return;
    }

    for (idx, &value) in nums.iter().enumerate() {
        if value == target {
            println!("YES");
            println!("{}", idx);
            return;
        }
    }

    let indexed: Vec<(i64, usize)> = nums.iter().enumerate().map(|(idx, &v)| (v, idx)).collect();

    let answer = if n <= 42 {
        solve_small_exact(&indexed, target)
    } else {
        None
    }
    .or_else(|| {
        let all_nonnegative = nums.iter().all(|&v| v >= 0) && target >= 0;
        if all_nonnegative {
            let sum_nonnegative: u128 = nums.iter().map(|&v| v as u128).sum();
            if let Ok(target_usize) = usize::try_from(target) {
                let target_u128 = target as u128;
                if target_u128 <= sum_nonnegative && target_usize <= 20_000_000 {
                    return solve_nonnegative_bitset(&nums, target_usize);
                }
            }
        }
        None
    })
    .or_else(|| {
        let cap = if n <= 5_000 { 1_000_000 } else { 250_000 };
        solve_hashmap_dp(&indexed, target, cap)
    })
    .or_else(|| greedy_signed(&indexed, target));

    if let Some(mut result) = answer {
        let sum: i64 = result.iter().map(|&idx| nums[idx]).sum();
        if sum == target {
            result.sort_unstable();
            println!("YES");
            if result.is_empty() {
                println!();
            } else {
                let line = result
                    .iter()
                    .map(|idx| idx.to_string())
                    .collect::<Vec<_>>()
                    .join(" ");
                println!("{}", line);
            }
            return;
        }
    }

    println!("NO");
}
