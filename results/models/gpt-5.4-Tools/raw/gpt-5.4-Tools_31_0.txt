use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

fn main() {
    if let Err(message) = run() {
        eprintln!("{message}");
        std::process::exit(1);
    }
}

fn run() -> Result<(), String> {
    let mut input = String::new();
    io::stdin()
        .read_to_string(&mut input)
        .map_err(|e| format!("failed to read stdin: {e}"))?;
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(token) => token
            .parse()
            .map_err(|e| format!("failed to parse num_numbers: {e}"))?,
        None => {
            println!("NO");
            return Ok(());
        }
    };
    let target: i64 = match it.next() {
        Some(token) => token
            .parse()
            .map_err(|e| format!("failed to parse target: {e}"))?,
        None => return Err("missing target".to_string()),
    };

    let mut numbers = Vec::with_capacity(n);
    for idx in 0..n {
        let token = it
            .next()
            .ok_or_else(|| format!("missing value at position {idx}"))?;
        let value: i64 = token
            .parse()
            .map_err(|e| format!("failed to parse value at position {idx}: {e}"))?;
        numbers.push(value);
    }

    if target == 0 {
        println!("YES");
        println!();
        return Ok(());
    }
    if n == 0 {
        println!("NO");
        return Ok(());
    }

    let items: Vec<Item> = numbers
        .into_iter()
        .enumerate()
        .map(|(index, value)| Item { value, index })
        .collect();

    if let Some(answer) = solve_subset_sum(&items, target) {
        emit_answer(&answer);
    } else {
        println!("NO");
    }
    Ok(())
}

fn solve_subset_sum(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if let Some(answer) = solve_trivial(items, target) {
        return Some(answer);
    }

    if items.len() <= 42 {
        if let Some(answer) = meet_in_the_middle(items, target) {
            return Some(answer);
        }
    }

    let (non_negative, total_positive) = summarize_non_negative(items)?;
    if non_negative {
        if target < 0 || target > total_positive {
            return None;
        }

        if items.len() >= 200_000 {
            if let Some(answer) = try_greedy_positive(items, target) {
                return Some(answer);
            }
            return None;
        }

        if target <= 20_000_000 {
            if let Some(answer) = bitset_subset_sum(items, target as usize) {
                return Some(answer);
            }
        }

        if items.len() <= 8_000 {
            return sparse_positive_dp(items, target);
        }

        if let Some(answer) = try_greedy_positive(items, target) {
            return Some(answer);
        }
        return None;
    }

    if items.len() <= 120 {
        return sparse_signed_dp(items, target, 4_000_000);
    }

    None
}

fn solve_trivial(items: &[Item], target: i64) -> Option<Vec<usize>> {
    for item in items {
        if item.value == target {
            return Some(vec![item.index]);
        }
    }

    let mut seen: HashMap<i64, usize> = HashMap::with_capacity(items.len().min(1_000_000));
    for item in items {
        if let Some(&other_index) = seen.get(&(target - item.value)) {
            return Some(vec![other_index, item.index]);
        }
        seen.entry(item.value).or_insert(item.index);
    }
    None
}

fn summarize_non_negative(items: &[Item]) -> Option<(bool, i64)> {
    let mut total_positive = 0_i64;
    for item in items {
        if item.value < 0 {
            return Some((false, 0));
        }
        total_positive = total_positive.checked_add(item.value)?;
    }
    Some((true, total_positive))
}

fn meet_in_the_middle(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let split = items.len() / 2;
    let left = &items[..split];
    let right = &items[split..];

    let mut left_sums = enumerate_half(left);
    let mut right_sums = enumerate_half(right);
    left_sums.sort_unstable_by_key(|entry| entry.0);
    right_sums.sort_unstable_by_key(|entry| entry.0);

    let mut i = 0usize;
    let mut j = right_sums.len();
    while i < left_sums.len() && j > 0 {
        let left_sum = left_sums[i].0;
        let right_sum = right_sums[j - 1].0;
        let sum = left_sum + right_sum;
        if sum == target {
            let mut answer = decode_mask(left, left_sums[i].1);
            answer.extend(decode_mask(right, right_sums[j - 1].1));
            answer.sort_unstable();
            return Some(answer);
        }
        if sum < target {
            i += 1;
        } else {
            j -= 1;
        }
    }
    None
}

fn enumerate_half(items: &[Item]) -> Vec<(i64, u64)> {
    let size = 1usize << items.len();
    let mut sums = Vec::with_capacity(size);
    for mask in 0..size {
        let mut sum = 0_i64;
        for (bit, item) in items.iter().enumerate() {
            if ((mask >> bit) & 1) != 0 {
                sum += item.value;
            }
        }
        sums.push((sum, mask as u64));
    }
    sums
}

fn decode_mask(items: &[Item], mask: u64) -> Vec<usize> {
    let mut answer = Vec::new();
    for (bit, item) in items.iter().enumerate() {
        if ((mask >> bit) & 1) != 0 {
            answer.push(item.index);
        }
    }
    answer
}

fn bitset_subset_sum(items: &[Item], target: usize) -> Option<Vec<usize>> {
    let words = (target / 64) + 1;
    let mut reachable = vec![0_u64; words];
    let mut prev_sum = vec![usize::MAX; target + 1];
    let mut prev_index = vec![usize::MAX; target + 1];
    reachable[0] = 1;

    for item in items {
        let value = item.value as usize;
        if value == 0 || value > target {
            continue;
        }

        let shift_words = value / 64;
        let shift_bits = value % 64;
        let mut newly = vec![0_u64; words];

        for src in 0..words {
            let bits = reachable[src];
            if bits == 0 {
                continue;
            }
            let dst = src + shift_words;
            if dst >= words {
                continue;
            }
            newly[dst] |= bits << shift_bits;
            if shift_bits != 0 && dst + 1 < words {
                newly[dst + 1] |= bits >> (64 - shift_bits);
            }
        }

        for idx in 0..words {
            newly[idx] &= !reachable[idx];
        }

        let max_word = target / 64;
        let tail_bits = target % 64;
        if max_word < words && tail_bits != 63 {
            let mask = (1_u64 << (tail_bits + 1)) - 1;
            newly[max_word] &= mask;
            for word in newly.iter_mut().skip(max_word + 1) {
                *word = 0;
            }
        }

        for (word_idx, &word) in newly.iter().enumerate() {
            let mut bits = word;
            while bits != 0 {
                let bit = bits.trailing_zeros() as usize;
                let sum = word_idx * 64 + bit;
                if sum <= target {
                    prev_sum[sum] = sum - value;
                    prev_index[sum] = item.index;
                }
                bits &= bits - 1;
            }
        }

        for idx in 0..words {
            reachable[idx] |= newly[idx];
        }

        if ((reachable[target / 64] >> (target % 64)) & 1) != 0 {
            return Some(reconstruct_positive(target, &prev_sum, &prev_index));
        }
    }

    if ((reachable[target / 64] >> (target % 64)) & 1) != 0 {
        Some(reconstruct_positive(target, &prev_sum, &prev_index))
    } else {
        None
    }
}

fn reconstruct_positive(target: usize, prev_sum: &[usize], prev_index: &[usize]) -> Vec<usize> {
    let mut answer = Vec::new();
    let mut sum = target;
    while sum != 0 {
        answer.push(prev_index[sum]);
        sum = prev_sum[sum];
    }
    answer.sort_unstable();
    answer
}

fn sparse_positive_dp(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(0, (0, usize::MAX));

    for item in items {
        if item.value <= 0 || item.value > target {
            continue;
        }

        let mut additions = Vec::new();
        for &sum in parent.keys() {
            let next = sum + item.value;
            if next > target || parent.contains_key(&next) {
                continue;
            }
            additions.push((next, sum));
        }
        for (next, previous) in additions {
            parent.insert(next, (previous, item.index));
        }
        if parent.contains_key(&target) {
            return Some(reconstruct_sparse(&parent, target));
        }
    }
    parent
        .contains_key(&target)
        .then(|| reconstruct_sparse(&parent, target))
}

fn sparse_signed_dp(items: &[Item], target: i64, max_states: usize) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(0, (0, usize::MAX));

    for item in items {
        let keys: Vec<i64> = parent.keys().copied().collect();
        let mut additions = Vec::new();
        for sum in keys {
            let next = sum + item.value;
            if parent.contains_key(&next) {
                continue;
            }
            additions.push((next, sum));
        }
        for (next, previous) in additions {
            parent.insert(next, (previous, item.index));
        }
        if parent.contains_key(&target) {
            return Some(reconstruct_sparse(&parent, target));
        }
        if parent.len() > max_states {
            break;
        }
    }
    None
}

fn reconstruct_sparse(parent: &HashMap<i64, (i64, usize)>, target: i64) -> Vec<usize> {
    let mut answer = Vec::new();
    let mut current = target;
    while current != 0 {
        let &(previous, index) = parent.get(&current).expect("target must be reachable");
        answer.push(index);
        current = previous;
    }
    answer.sort_unstable();
    answer
}

fn try_greedy_positive(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let mut filtered: Vec<Item> = items
        .iter()
        .copied()
        .filter(|item| item.value > 0 && item.value <= target)
        .collect();
    filtered.sort_unstable_by(|a, b| b.value.cmp(&a.value).then_with(|| a.index.cmp(&b.index)));

    for start in 0..filtered.len().min(24) {
        let mut remaining = target;
        let mut answer = Vec::new();
        for (pos, item) in filtered.iter().enumerate() {
            if pos < start && item.value == filtered[start].value {
                continue;
            }
            if item.value <= remaining {
                remaining -= item.value;
                answer.push(item.index);
                if remaining == 0 {
                    answer.sort_unstable();
                    return Some(answer);
                }
            }
        }
    }
    None
}

fn emit_answer(indices: &[usize]) {
    println!("YES");
    if indices.is_empty() {
        println!();
        return;
    }
    let mut first = true;
    for index in indices {
        if !first {
            print!(" ");
        }
        first = false;
        print!("{index}");
    }
    println!();
}
