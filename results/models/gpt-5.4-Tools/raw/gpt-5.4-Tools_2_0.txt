using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct EdgeInput
    {
        public int U;
        public int V;
        public long W;
    }

    private struct AdjEdge
    {
        public int To;
        public long W;
        public int Id;

        public AdjEdge(int to, long w, int id)
        {
            To = to;
            W = w;
            Id = id;
        }
    }

    private struct PQNode : IComparable<PQNode>
    {
        public long Dist;
        public int Node;

        public PQNode(long dist, int node)
        {
            Dist = dist;
            Node = node;
        }

        public int CompareTo(PQNode other)
        {
            int cmp = Dist.CompareTo(other.Dist);
            return cmp != 0 ? cmp : Node.CompareTo(other.Node);
        }
    }

    private sealed class MinHeap<T> where T : IComparable<T>
    {
        private T[] _data;
        public int Count { get; private set; }

        public MinHeap(int capacity = 16)
        {
            _data = new T[Math.Max(4, capacity)];
        }

        public void Push(T value)
        {
            if (Count == _data.Length) Array.Resize(ref _data, _data.Length * 2);
            int i = Count++;
            _data[i] = value;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_data[p].CompareTo(_data[i]) <= 0) break;
                (_data[p], _data[i]) = (_data[i], _data[p]);
                i = p;
            }
        }

        public T Pop()
        {
            T result = _data[0];
            _data[0] = _data[--Count];
            int i = 0;
            while (true)
            {
                int l = (i << 1) + 1;
                if (l >= Count) break;
                int r = l + 1;
                int c = r < Count && _data[r].CompareTo(_data[l]) < 0 ? r : l;
                if (_data[i].CompareTo(_data[c]) <= 0) break;
                (_data[i], _data[c]) = (_data[c], _data[i]);
                i = c;
            }
            return result;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    private static List<AdjEdge>[] _graph = Array.Empty<List<AdjEdge>>();
    private static EdgeInput[] _edges = Array.Empty<EdgeInput>();
    private static int[] _degree = Array.Empty<int>();
    private static int _n;
    private static int _m;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        _n = fs.NextInt();
        _m = fs.NextInt();

        _edges = new EdgeInput[_m];
        _graph = new List<AdjEdge>[_n];
        _degree = new int[_n];
        for (int i = 0; i < _n; i++) _graph[i] = new List<AdjEdge>();

        for (int i = 0; i < _m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            _edges[i] = new EdgeInput { U = u, V = v, W = w };
            _graph[u].Add(new AdjEdge(v, w, i));
            _graph[v].Add(new AdjEdge(u, w, i));
            _degree[u]++;
            _degree[v]++;
        }

        if (_m == 0)
        {
            Console.WriteLine("2 0 0");
            return;
        }

        EnsureConnectedFromZero();

        var odd = new List<int>();
        for (int i = 0; i < _n; i++)
        {
            if ((_degree[i] & 1) != 0) odd.Add(i);
        }

        var duplicatePaths = BuildDuplicatePaths(odd);
        var route = BuildEulerRoute(duplicatePaths);

        var sb = new StringBuilder(Math.Max(route.Count * 3, 16));
        sb.Append(route.Count);
        for (int i = 0; i < route.Count; i++)
        {
            sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static void EnsureConnectedFromZero()
    {
        var seen = new bool[_n];
        var stack = new Stack<int>();
        seen[0] = true;
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Pop();
            foreach (var e in _graph[u])
            {
                if (!seen[e.To])
                {
                    seen[e.To] = true;
                    stack.Push(e.To);
                }
            }
        }

        for (int i = 0; i < _n; i++)
        {
            if (_degree[i] > 0 && !seen[i])
            {
                throw new InvalidOperationException("Graph edges are not all in the component of node 0.");
            }
        }
    }

    private static List<int[]> BuildDuplicatePaths(List<int> odd)
    {
        int k = odd.Count;
        var result = new List<int[]>();
        if (k == 0) return result;

        if (k <= 22)
        {
            var pairing = SolveExactMatching(odd);
            for (int i = 0; i < pairing.Count; i += 2)
            {
                result.Add(ShortestPath(pairing[i], pairing[i + 1]));
            }
            return result;
        }

        if (k <= 400)
        {
            var pairing = SolveGreedyMetricMatching(odd);
            for (int i = 0; i < pairing.Count; i += 2)
            {
                result.Add(ShortestPath(pairing[i], pairing[i + 1]));
            }
            return result;
        }

        var greedy = SolveLocalGreedyMatching(odd);
        for (int i = 0; i < greedy.Count; i += 2)
        {
            result.Add(ShortestPath(greedy[i], greedy[i + 1]));
        }
        return result;
    }

    private static List<int> SolveExactMatching(List<int> odd)
    {
        int k = odd.Count;
        var dist = new long[k, k];
        for (int i = 0; i < k; i++)
        {
            var d = Dijkstra(odd[i], null);
            for (int j = i + 1; j < k; j++)
            {
                dist[i, j] = d[odd[j]];
                dist[j, i] = dist[i, j];
            }
        }

        int totalMasks = 1 << k;
        var dp = new long[totalMasks];
        var choiceA = new int[totalMasks];
        var choiceB = new int[totalMasks];
        for (int i = 1; i < totalMasks; i++) dp[i] = long.MaxValue / 4;

        for (int mask = 1; mask < totalMasks; mask++)
        {
            int a = 0;
            while (((mask >> a) & 1) == 0) a++;
            int rest = mask ^ (1 << a);
            for (int b = a + 1; b < k; b++)
            {
                if (((rest >> b) & 1) == 0) continue;
                int next = rest ^ (1 << b);
                long cand = dp[next] + dist[a, b];
                if (cand < dp[mask])
                {
                    dp[mask] = cand;
                    choiceA[mask] = a;
                    choiceB[mask] = b;
                }
            }
        }

        int full = totalMasks - 1;
        var pairs = new List<int>(k);
        while (full != 0)
        {
            int a = choiceA[full];
            int b = choiceB[full];
            pairs.Add(odd[a]);
            pairs.Add(odd[b]);
            full ^= (1 << a);
            full ^= (1 << b);
        }
        return pairs;
    }

    private static List<int> SolveGreedyMetricMatching(List<int> odd)
    {
        int k = odd.Count;
        var allDist = new long[k][];
        for (int i = 0; i < k; i++) allDist[i] = Dijkstra(odd[i], null);

        var used = new bool[k];
        var pairs = new List<int>(k);
        for (int iter = 0; iter < k / 2; iter++)
        {
            int bestI = -1;
            int bestJ = -1;
            long best = long.MaxValue;

            for (int i = 0; i < k; i++)
            {
                if (used[i]) continue;
                long localBest = long.MaxValue;
                int localJ = -1;
                for (int j = i + 1; j < k; j++)
                {
                    if (used[j]) continue;
                    long d = allDist[i][odd[j]];
                    if (d < localBest)
                    {
                        localBest = d;
                        localJ = j;
                    }
                }
                if (localJ != -1 && localBest < best)
                {
                    best = localBest;
                    bestI = i;
                    bestJ = localJ;
                }
            }

            used[bestI] = true;
            used[bestJ] = true;
            pairs.Add(odd[bestI]);
            pairs.Add(odd[bestJ]);
        }
        return pairs;
    }

    private static List<int> SolveLocalGreedyMatching(List<int> odd)
    {
        int k = odd.Count;
        var remaining = new HashSet<int>(odd);
        var pairs = new List<int>(k);

        while (remaining.Count > 0)
        {
            int a = -1;
            foreach (int v in remaining)
            {
                a = v;
                break;
            }
            remaining.Remove(a);

            int b = FindNearestOdd(a, remaining);
            remaining.Remove(b);
            pairs.Add(a);
            pairs.Add(b);
        }

        return pairs;
    }

    private static int FindNearestOdd(int source, HashSet<int> candidates)
    {
        var dist = new long[_n];
        Array.Fill(dist, long.MaxValue);
        dist[source] = 0;
        var pq = new MinHeap<PQNode>();
        pq.Push(new PQNode(0, source));

        while (pq.Count > 0)
        {
            var cur = pq.Pop();
            if (cur.Dist != dist[cur.Node]) continue;
            if (cur.Node != source && candidates.Contains(cur.Node)) return cur.Node;

            foreach (var e in _graph[cur.Node])
            {
                long nd = cur.Dist + e.W;
                if (nd < dist[e.To])
                {
                    dist[e.To] = nd;
                    pq.Push(new PQNode(nd, e.To));
                }
            }
        }

        throw new InvalidOperationException("Failed to find odd pair.");
    }

    private static long[] Dijkstra(int source, int[]? parent)
    {
        var dist = new long[_n];
        Array.Fill(dist, long.MaxValue);
        if (parent != null) Array.Fill(parent, -1);
        dist[source] = 0;

        var pq = new MinHeap<PQNode>();
        pq.Push(new PQNode(0, source));

        while (pq.Count > 0)
        {
            var cur = pq.Pop();
            if (cur.Dist != dist[cur.Node]) continue;

            foreach (var e in _graph[cur.Node])
            {
                long nd = cur.Dist + e.W;
                if (nd < dist[e.To])
                {
                    dist[e.To] = nd;
                    if (parent != null) parent[e.To] = cur.Node;
                    pq.Push(new PQNode(nd, e.To));
                }
            }
        }

        return dist;
    }

    private static int[] ShortestPath(int source, int target)
    {
        var parent = new int[_n];
        Dijkstra(source, parent);
        var path = new List<int>();
        int cur = target;
        while (cur != -1)
        {
            path.Add(cur);
            if (cur == source) break;
            cur = parent[cur];
        }
        path.Reverse();
        return path.ToArray();
    }

    private static List<int> BuildEulerRoute(List<int[]> duplicatePaths)
    {
        int duplicateEdgeCount = 0;
        foreach (var path in duplicatePaths) duplicateEdgeCount += path.Length - 1;

        int totalEdgeCount = _m + duplicateEdgeCount;
        var adj = new List<(int to, int id)>[_n];
        for (int i = 0; i < _n; i++) adj[i] = new List<(int to, int id)>(_graph[i].Count + 2);

        for (int i = 0; i < _m; i++)
        {
            int u = _edges[i].U;
            int v = _edges[i].V;
            adj[u].Add((v, i));
            adj[v].Add((u, i));
        }

        int nextId = _m;
        foreach (var path in duplicatePaths)
        {
            for (int i = 0; i + 1 < path.Length; i++)
            {
                int u = path[i];
                int v = path[i + 1];
                adj[u].Add((v, nextId));
                adj[v].Add((u, nextId));
                nextId++;
            }
        }

        var used = new bool[totalEdgeCount];
        var ptr = new int[_n];
        var stack = new Stack<int>();
        var route = new List<int>(totalEdgeCount + 1);
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            while (ptr[u] < adj[u].Count && used[adj[u][ptr[u]].id]) ptr[u]++;
            if (ptr[u] == adj[u].Count)
            {
                route.Add(u);
                stack.Pop();
            }
            else
            {
                var e = adj[u][ptr[u]];
                used[e.id] = true;
                stack.Push(e.to);
            }
        }

        route.Reverse();
        return route;
    }
}
