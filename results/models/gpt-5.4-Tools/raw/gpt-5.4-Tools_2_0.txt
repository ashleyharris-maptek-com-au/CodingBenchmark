using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    struct WeightedArc
    {
        public int To;
        public int Next;
        public int Cost;
    }

    struct EulerArc
    {
        public int To;
        public int Next;
        public int UndirectedId;
    }

    struct HeapNode : IComparable<HeapNode>
    {
        public long Dist;
        public int Node;

        public int CompareTo(HeapNode other)
        {
            int cmp = Dist.CompareTo(other.Dist);
            return cmp != 0 ? cmp : Node.CompareTo(other.Node);
        }
    }

    sealed class MinHeap<T> where T : IComparable<T>
    {
        private T[] _data;
        public int Count { get; private set; }

        public MinHeap(int capacity = 16)
        {
            _data = new T[Math.Max(4, capacity)];
        }

        public void Push(T value)
        {
            if (Count == _data.Length) Array.Resize(ref _data, Count * 2);
            int i = Count++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_data[p].CompareTo(value) <= 0) break;
                _data[i] = _data[p];
                i = p;
            }
            _data[i] = value;
        }

        public T Pop()
        {
            T root = _data[0];
            T x = _data[--Count];
            if (Count > 0)
            {
                int i = 0;
                while (true)
                {
                    int l = (i << 1) + 1;
                    if (l >= Count) break;
                    int r = l + 1;
                    int c = r < Count && _data[r].CompareTo(_data[l]) < 0 ? r : l;
                    if (_data[c].CompareTo(x) >= 0) break;
                    _data[i] = _data[c];
                    i = c;
                }
                _data[i] = x;
            }
            return root;
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    static int n;
    static int m;
    static int[] wHead = Array.Empty<int>();
    static WeightedArc[] wEdges = Array.Empty<WeightedArc>();
    static int wPtr;
    static int[] degree = Array.Empty<int>();
    static int[] origU = Array.Empty<int>();
    static int[] origV = Array.Empty<int>();
    static bool[] reachable = Array.Empty<bool>();

    static void AddWeightedArc(int u, int v, int cost)
    {
        wEdges[wPtr] = new WeightedArc { To = v, Next = wHead[u], Cost = cost };
        wHead[u] = wPtr++;
    }

    static void MarkReachableFromZero()
    {
        reachable = new bool[n];
        if (n == 0) return;
        var q = new Queue<int>();
        q.Enqueue(0);
        reachable[0] = true;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            for (int ei = wHead[u]; ei != -1; ei = wEdges[ei].Next)
            {
                int v = wEdges[ei].To;
                if (!reachable[v])
                {
                    reachable[v] = true;
                    q.Enqueue(v);
                }
            }
        }
    }

    static void Dijkstra(int src, long[] dist, int[] parent)
    {
        const long INF = long.MaxValue / 4;
        Array.Fill(dist, INF);
        Array.Fill(parent, -1);
        var heap = new MinHeap<HeapNode>(Math.Max(16, n / 8));
        dist[src] = 0;
        heap.Push(new HeapNode { Dist = 0, Node = src });

        while (heap.Count > 0)
        {
            var cur = heap.Pop();
            if (cur.Dist != dist[cur.Node]) continue;
            int u = cur.Node;
            for (int ei = wHead[u]; ei != -1; ei = wEdges[ei].Next)
            {
                ref WeightedArc e = ref wEdges[ei];
                long nd = cur.Dist + e.Cost;
                if (nd < dist[e.To])
                {
                    dist[e.To] = nd;
                    parent[e.To] = u;
                    heap.Push(new HeapNode { Dist = nd, Node = e.To });
                }
            }
        }
    }

    static List<(int a, int b)> ComputePairings(List<int> odd)
    {
        int k = odd.Count;
        var result = new List<(int a, int b)>();
        if (k == 0) return result;

        if (k <= 22)
        {
            long[,] cost = new long[k, k];
            long[] dist = new long[n];
            int[] parent = new int[n];
            for (int i = 0; i < k; i++)
            {
                Dijkstra(odd[i], dist, parent);
                for (int j = i + 1; j < k; j++)
                {
                    cost[i, j] = dist[odd[j]];
                    cost[j, i] = dist[odd[j]];
                }
            }

            int size = 1 << k;
            long[] dp = new long[size];
            int[] pickI = new int[size];
            int[] pickJ = new int[size];
            const long INF = long.MaxValue / 4;
            for (int mask = 1; mask < size; mask++) dp[mask] = INF;

            for (int mask = 1; mask < size; mask++)
            {
                int i;
                for (i = 0; i < k; i++) if (((mask >> i) & 1) != 0) break;
                int rest = mask ^ (1 << i);
                for (int j = i + 1; j < k; j++)
                {
                    if (((rest >> j) & 1) == 0) continue;
                    int next = rest ^ (1 << j);
                    long cand = dp[next] + cost[i, j];
                    if (cand < dp[mask])
                    {
                        dp[mask] = cand;
                        pickI[mask] = i;
                        pickJ[mask] = j;
                    }
                }
            }

            for (int mask = size - 1; mask != 0;)
            {
                int i = pickI[mask];
                int j = pickJ[mask];
                result.Add((odd[i], odd[j]));
                mask ^= 1 << i;
                mask ^= 1 << j;
            }
            return result;
        }

        long[] tmpDist = new long[n];
        int[] tmpParent = new int[n];
        var cached = new Dictionary<int, long[]>();
        var unmatched = new List<int>(odd);
        while (unmatched.Count > 0)
        {
            int a = unmatched[unmatched.Count - 1];
            unmatched.RemoveAt(unmatched.Count - 1);
            if (!cached.TryGetValue(a, out var da))
            {
                da = new long[n];
                Dijkstra(a, da, tmpParent);
                cached[a] = da;
            }
            int bestIndex = 0;
            long best = long.MaxValue;
            for (int i = 0; i < unmatched.Count; i++)
            {
                long d = da[unmatched[i]];
                if (d < best)
                {
                    best = d;
                    bestIndex = i;
                }
            }
            int b = unmatched[bestIndex];
            unmatched.RemoveAt(bestIndex);
            result.Add((a, b));
        }
        return result;
    }

    static void AddPathEdges(int src, int dst, List<int> extraU, List<int> extraV)
    {
        long[] dist = new long[n];
        int[] parent = new int[n];
        Dijkstra(src, dist, parent);
        int cur = dst;
        while (cur != src)
        {
            int prev = parent[cur];
            extraU.Add(prev);
            extraV.Add(cur);
            cur = prev;
        }
    }

    static List<int> BuildEulerTour(List<int> extraU, List<int> extraV)
    {
        int totalEdges = m + extraU.Count;
        int[] head = new int[n];
        Array.Fill(head, -1);
        EulerArc[] arcs = new EulerArc[Math.Max(2, totalEdges * 2)];
        bool[] used = new bool[totalEdges];
        int ptr = 0;

        void AddEulerArc(int u, int v, int id)
        {
            arcs[ptr] = new EulerArc { To = v, Next = head[u], UndirectedId = id };
            head[u] = ptr++;
        }

        for (int i = 0; i < m; i++)
        {
            AddEulerArc(origU[i], origV[i], i);
            AddEulerArc(origV[i], origU[i], i);
        }
        for (int i = 0; i < extraU.Count; i++)
        {
            int id = m + i;
            AddEulerArc(extraU[i], extraV[i], id);
            AddEulerArc(extraV[i], extraU[i], id);
        }

        int[] nextArc = new int[n];
        Array.Copy(head, nextArc, n);
        var stack = new Stack<int>();
        var route = new List<int>(totalEdges + 1);
        stack.Push(0);

        while (stack.Count > 0)
        {
            int v = stack.Peek();
            int ei = nextArc[v];
            while (ei != -1 && used[arcs[ei].UndirectedId]) ei = arcs[ei].Next;
            nextArc[v] = ei;
            if (ei == -1)
            {
                route.Add(v);
                stack.Pop();
            }
            else
            {
                used[arcs[ei].UndirectedId] = true;
                nextArc[v] = arcs[ei].Next;
                stack.Push(arcs[ei].To);
            }
        }

        route.Reverse();
        return route;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        n = fs.NextInt();
        m = fs.NextInt();

        wHead = new int[n];
        Array.Fill(wHead, -1);
        wEdges = new WeightedArc[Math.Max(2, m * 2)];
        degree = new int[n];
        origU = new int[m];
        origV = new int[m];
        wPtr = 0;

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            origU[i] = u;
            origV[i] = v;
            AddWeightedArc(u, v, w);
            AddWeightedArc(v, u, w);
            degree[u]++;
            degree[v]++;
        }

        if (m == 0)
        {
            Console.WriteLine("0");
            return;
        }

        MarkReachableFromZero();
        for (int i = 0; i < m; i++)
        {
            if (!reachable[origU[i]] || !reachable[origV[i]])
            {
                Console.WriteLine("0");
                return;
            }
        }

        var odd = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (reachable[i] && (degree[i] & 1) != 0) odd.Add(i);
        }

        var pairings = ComputePairings(odd);
        var extraU = new List<int>(Math.Max(4, pairings.Count * 2));
        var extraV = new List<int>(Math.Max(4, pairings.Count * 2));
        foreach (var p in pairings) AddPathEdges(p.a, p.b, extraU, extraV);

        var route = BuildEulerTour(extraU, extraV);
        var sb = new StringBuilder(route.Count * 3 + 16);
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.Write(sb.ToString());
    }
}
