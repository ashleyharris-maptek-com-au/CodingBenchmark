using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private const int LookaheadDepthSafe = 2;
    private const int LookaheadDepthDanger = 1;

    private static readonly Dictionary<char, PieceDefinition> Pieces = CreatePieces();

    public static void Main()
    {
        var scanner = new FastScanner(Console.In);
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var game = new TetrisGame(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string token = scanner.NextToken();
            if (string.IsNullOrEmpty(token))
            {
                break;
            }

            char pieceType = token[0];
            Move move = ChooseMove(game, pieceType);
            Console.WriteLine($"{move.Rotation} {move.Column}");
            Console.Out.Flush();

            string result = scanner.NextToken();
            if (string.IsNullOrEmpty(result))
            {
                game.ApplyMove(pieceType, move, updateState: true);
                break;
            }

            game.ApplyMove(pieceType, move, updateState: true);
            game.ApplyExternalResult(result);

            if (result.Equals("gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }

    private static Move ChooseMove(TetrisGame game, char pieceType)
    {
        var piece = Pieces[pieceType];
        var boardStats = game.GetStats();
        bool danger = boardStats.MaxHeight >= game.Height - 4 || boardStats.HoleCount > game.Width * 2;
        int depth = danger ? LookaheadDepthDanger : LookaheadDepthSafe;
        var weights = danger ? HeuristicWeights.Danger : (boardStats.MaxHeight < game.Height / 3 ? HeuristicWeights.Safe : HeuristicWeights.Normal);

        Move bestMove = default;
        double bestScore = double.NegativeInfinity;
        bool found = false;

        foreach (var orientation in piece.Orientations)
        {
            int maxColumn = game.Width - orientation.Width;
            for (int col = 0; col <= maxColumn; col++)
            {
                if (!game.CanPlace(orientation, col, out int landingRow))
                {
                    continue;
                }

                var sim = game.Clone();
                var placement = sim.Place(orientation, col, landingRow);
                if (!placement.Success)
                {
                    continue;
                }

                double score = EvaluateState(sim, weights, placement, depth - 1);
                score += placement.LinesCleared * placement.LinesCleared * 15.0;
                score += placement.TouchedFloor ? 1.2 : 0.0;
                score += placement.WallContacts * 0.15;
                score += placement.SupportContacts * 0.20;

                if (!found || score > bestScore)
                {
                    found = true;
                    bestScore = score;
                    bestMove = new Move(orientation.Rotation, col);
                }
            }
        }

        if (found)
        {
            return bestMove;
        }

        return new Move(0, 0);
    }

    private static double EvaluateState(TetrisGame game, HeuristicWeights weights, PlacementResult placement, int lookaheadDepth)
    {
        var stats = game.GetStats();

        double score = 0.0;
        score += stats.AggregateHeight * weights.AggregateHeight;
        score += stats.MaxHeight * weights.MaxHeight;
        score += stats.HoleCount * weights.Holes;
        score += stats.WeightedHoles * weights.WeightedHoles;
        score += stats.Bumpiness * weights.Bumpiness;
        score += stats.WellDepth * weights.Wells;
        score += stats.RowTransitions * weights.RowTransitions;
        score += stats.ColumnTransitions * weights.ColumnTransitions;
        score += stats.CoveredCells * weights.CoveredCells;
        score += stats.ClearedLinesTotal * weights.ClearedLines;
        score += placement.LinesCleared * weights.ImmediateLineClear;

        if (lookaheadDepth > 0)
        {
            double future = 0.0;
            int count = 0;
            foreach (char nextPiece in TetrisGame.PieceOrder)
            {
                future += EstimateBestFuture(game, nextPiece, weights, lookaheadDepth);
                count++;
            }

            if (count > 0)
            {
                score += (future / count) * weights.LookaheadFactor;
            }
        }

        return score;
    }

    private static double EstimateBestFuture(TetrisGame game, char pieceType, HeuristicWeights weights, int lookaheadDepth)
    {
        var piece = Pieces[pieceType];
        double bestScore = double.NegativeInfinity;
        bool found = false;

        foreach (var orientation in piece.Orientations)
        {
            int maxColumn = game.Width - orientation.Width;
            for (int col = 0; col <= maxColumn; col++)
            {
                if (!game.CanPlace(orientation, col, out int landingRow))
                {
                    continue;
                }

                var sim = game.Clone();
                var placement = sim.Place(orientation, col, landingRow);
                if (!placement.Success)
                {
                    continue;
                }

                double score = EvaluateState(sim, weights, placement, lookaheadDepth - 1);
                if (!found || score > bestScore)
                {
                    found = true;
                    bestScore = score;
                }
            }
        }

        return found ? bestScore : -1_000_000.0;
    }

    private static Dictionary<char, PieceDefinition> CreatePieces()
    {
        return new Dictionary<char, PieceDefinition>
        {
            ['I'] = new PieceDefinition('I', new[]
            {
                Orientation.From(0, new[] { "####" }),
                Orientation.From(1, new[] { "#", "#", "#", "#" })
            }),
            ['O'] = new PieceDefinition('O', new[]
            {
                Orientation.From(0, new[] { "##", "##" })
            }),
            ['T'] = new PieceDefinition('T', new[]
            {
                Orientation.From(0, new[] { ".#.", "###" }),
                Orientation.From(1, new[] { "#.", "##", "#." }),
                Orientation.From(2, new[] { "###", ".#." }),
                Orientation.From(3, new[] { ".#", "##", ".#" })
            }),
            ['S'] = new PieceDefinition('S', new[]
            {
                Orientation.From(0, new[] { ".##", "##." }),
                Orientation.From(1, new[] { "#.", "##", ".#" })
            }),
            ['Z'] = new PieceDefinition('Z', new[]
            {
                Orientation.From(0, new[] { "##.", ".##" }),
                Orientation.From(1, new[] { ".#", "##", "#." })
            }),
            ['J'] = new PieceDefinition('J', new[]
            {
                Orientation.From(0, new[] { "#..", "###" }),
                Orientation.From(1, new[] { "##", "#.", "#." }),
                Orientation.From(2, new[] { "###", "..#" }),
                Orientation.From(3, new[] { ".#", ".#", "##" })
            }),
            ['L'] = new PieceDefinition('L', new[]
            {
                Orientation.From(0, new[] { "..#", "###" }),
                Orientation.From(1, new[] { "#.", "#.", "##" }),
                Orientation.From(2, new[] { "###", "#.." }),
                Orientation.From(3, new[] { "##", ".#", ".#" })
            })
        };
    }

    private readonly record struct Move(int Rotation, int Column);

    private sealed class PieceDefinition
    {
        public PieceDefinition(char name, IEnumerable<Orientation> orientations)
        {
            Name = name;
            Orientations = orientations.ToArray();
        }

        public char Name { get; }
        public Orientation[] Orientations { get; }
    }

    private sealed class Orientation
    {
        public int Rotation { get; private set; }
        public int Width { get; private set; }
        public int Height { get; private set; }
        public (int X, int Y)[] Cells { get; private set; } = Array.Empty<(int X, int Y)>();
        public int[] BottomProfile { get; private set; } = Array.Empty<int>();

        public static Orientation From(int rotation, string[] rows)
        {
            var cells = new List<(int X, int Y)>();
            int height = rows.Length;
            int width = rows[0].Length;
            var bottom = Enumerable.Repeat(-1, width).ToArray();

            for (int y = 0; y < height; y++)
            {
                string row = rows[y];
                for (int x = 0; x < width; x++)
                {
                    if (row[x] == '#')
                    {
                        cells.Add((x, y));
                        if (y > bottom[x])
                        {
                            bottom[x] = y;
                        }
                    }
                }
            }

            return new Orientation
            {
                Rotation = rotation,
                Width = width,
                Height = height,
                Cells = cells.ToArray(),
                BottomProfile = bottom
            };
        }
    }

    private readonly record struct PlacementResult(bool Success, int LinesCleared, bool TouchedFloor, int WallContacts, int SupportContacts);

    private readonly record struct BoardStats(
        int AggregateHeight,
        int MaxHeight,
        int HoleCount,
        int WeightedHoles,
        int Bumpiness,
        int WellDepth,
        int RowTransitions,
        int ColumnTransitions,
        int CoveredCells,
        int ClearedLinesTotal);

    private readonly record struct HeuristicWeights(
        double AggregateHeight,
        double MaxHeight,
        double Holes,
        double WeightedHoles,
        double Bumpiness,
        double Wells,
        double RowTransitions,
        double ColumnTransitions,
        double CoveredCells,
        double ClearedLines,
        double ImmediateLineClear,
        double LookaheadFactor)
    {
        public static readonly HeuristicWeights Safe = new(
            AggregateHeight: -0.43,
            MaxHeight: -0.70,
            Holes: -8.2,
            WeightedHoles: -1.6,
            Bumpiness: -0.20,
            Wells: -0.33,
            RowTransitions: -0.22,
            ColumnTransitions: -0.25,
            CoveredCells: -0.45,
            ClearedLines: 2.3,
            ImmediateLineClear: 5.5,
            LookaheadFactor: 0.22);

        public static readonly HeuristicWeights Normal = new(
            AggregateHeight: -0.52,
            MaxHeight: -0.95,
            Holes: -11.5,
            WeightedHoles: -2.0,
            Bumpiness: -0.28,
            Wells: -0.40,
            RowTransitions: -0.28,
            ColumnTransitions: -0.35,
            CoveredCells: -0.55,
            ClearedLines: 2.8,

        public static readonly HeuristicWeights Danger = new(
            AggregateHeight: -0.70,

            Holes: -14.0,
            WeightedHoles: -2.8,
            Bumpiness: -0.35,
            Wells: -0.60,

            LookaheadFactor: 0.08);
    }

    private sealed class TetrisGame
    {
        public static readonly char[] PieceOrder = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };

        private readonly bool[,] _grid;
        private readonly int[] _columnHeights;

        public TetrisGame(int width, int height)
        {
            Width = width;
            Height = height;
            _grid = new bool[height, width];
            _columnHeights = new int[width];
        }

        private TetrisGame(int width, int height, bool[,] grid, int[] columnHeights, int clearedLinesTotal)
        {
            Width = width;
            Height = height;
            _grid = grid;
            _columnHeights = columnHeights;
            ClearedLinesTotal = clearedLinesTotal;
        }

        public int Width { get; }
        public int Height { get; }
        public int ClearedLinesTotal { get; private set; }

        public TetrisGame Clone()
        {
            return new TetrisGame(Width, Height, (bool[,])_grid.Clone(), (int[])_columnHeights.Clone(), ClearedLinesTotal);
        }

        public bool CanPlace(Orientation orientation, int column, out int landingRow)
        {
            landingRow = -1;
            if (column < 0 || column + orientation.Width > Width)
            {
                return false;
            }

            int row = 0;
            foreach (var cell in orientation.Cells)
            {
                int candidate = _columnHeights[column + cell.X] - cell.Y - 1;
                if (candidate > row)
                {
                    row = candidate;
                }
            }

            while (true)
            {
                if (Collides(orientation, column, row))
                {
                    if (row == 0)
                    {
                        return false;
                    }

                    landingRow = row - 1;
                    return !Collides(orientation, column, landingRow);
                }

                row++;
                if (row > Height)
                {
                    landingRow = row - 1;
                    return !Collides(orientation, column, landingRow);
                }
            }
        }

        public PlacementResult Place(Orientation orientation, int column, int row)
        {
            if (row < 0 || Collides(orientation, column, row))
            {
                return new PlacementResult(false, 0, false, 0, 0);
            }

            int wallContacts = 0;
            int supportContacts = 0;
            bool touchedFloor = false;

            foreach (var cell in orientation.Cells)
            {
                int x = column + cell.X;
                int y = row + cell.Y;
                if (y < 0 || y >= Height)
                {
                    return new PlacementResult(false, 0, false, 0, 0);
                }

                _grid[y, x] = true;

                if (x == 0 || x == Width - 1)
                {
                    wallContacts++;
                }

                if (y == Height - 1)
                {
                    touchedFloor = true;
                }
                else if (y + 1 < Height && _grid[y + 1, x] && !ContainsCell(orientation, cell.X, cell.Y + 1))
                {
                    supportContacts++;
                }
            }

            RebuildColumnHeights();
            int cleared = ClearFullLines();
            ClearedLinesTotal += cleared;
            RebuildColumnHeights();
            return new PlacementResult(true, cleared, touchedFloor, wallContacts, supportContacts);
        }

        public void ApplyMove(char pieceType, Move move, bool updateState)
        {
            if (!updateState)
            {
                return;
            }

            var orientation = Pieces[pieceType].Orientations.First(o => o.Rotation == move.Rotation);
            if (CanPlace(orientation, move.Column, out int row))
            {
                Place(orientation, move.Column, row);
            }
        }

        public void ApplyExternalResult(string result)
        {
            if (result.StartsWith("cleared_", StringComparison.OrdinalIgnoreCase))
            {
                return;
            }
        }

        public BoardStats GetStats()
        {
            int aggregateHeight = 0;
            int maxHeight = 0;
            int bumpiness = 0;
            int holeCount = 0;
            int weightedHoles = 0;
            int coveredCells = 0;
            int rowTransitions = 0;
            int columnTransitions = 0;
            int wellDepth = 0;

            int[] heights = new int[Width];
            for (int x = 0; x < Width; x++)
            {
                heights[x] = _columnHeights[x];
                aggregateHeight += heights[x];
                if (heights[x] > maxHeight)
                {
                    maxHeight = heights[x];
                }
                if (x > 0)
                {
                    bumpiness += Math.Abs(heights[x] - heights[x - 1]);
                }
            }

            for (int x = 0; x < Width; x++)
            {
                bool seenBlock = false;
                for (int y = 0; y < Height; y++)
                {
                    if (_grid[y, x])
                    {
                        seenBlock = true;
                    }
                    else if (seenBlock)
                    {
                        holeCount++;
                        weightedHoles += Height - y;
                    }
                }
            }

            for (int y = 0; y < Height; y++)
            {
                bool prevFilled = true;
                for (int x = 0; x < Width; x++)
                {
                    bool filled = _grid[y, x];
                    if (filled != prevFilled)
                    {
                        rowTransitions++;
                    }
                    prevFilled = filled;
                    if (!filled)
                    {
                        bool leftFilled = x == 0 || _grid[y, x - 1];
                        bool rightFilled = x == Width - 1 || _grid[y, x + 1];
                        if (leftFilled && rightFilled)
                        {
                            int depth = 1;
                            for (int yy = y + 1; yy < Height && !_grid[yy, x]; yy++)
                            {
                                depth++;
                            }
                            wellDepth += depth;
                        }
                    }
                }
                if (!prevFilled)
                {
                    rowTransitions++;
                }
            }

            for (int x = 0; x < Width; x++)
            {
                bool prevFilled = true;
                for (int y = 0; y < Height; y++)
                {
                    bool filled = _grid[y, x];
                    if (filled != prevFilled)
                    {
                        columnTransitions++;
                    }
                    prevFilled = filled;
                    if (filled)
                    {
                        coveredCells += Height - y;
                    }
                }
                if (!prevFilled)
                {
                    columnTransitions++;
                }
            }

            return new BoardStats(
                aggregateHeight,
                maxHeight,
                holeCount,
                weightedHoles,
                bumpiness,
                wellDepth,
                rowTransitions,
                columnTransitions,
                coveredCells,
                ClearedLinesTotal);
        }

        private bool Collides(Orientation orientation, int column, int row)
        {
            foreach (var cell in orientation.Cells)
            {
                int x = column + cell.X;
                int y = row + cell.Y;
                if (x < 0 || x >= Width || y >= Height)
                {
                    return true;
                }
                if (y >= 0 && _grid[y, x])
                {
                    return true;
                }
            }
            return false;
        }

        private int ClearFullLines()
        {
            int writeRow = Height - 1;
            int cleared = 0;

            for (int readRow = Height - 1; readRow >= 0; readRow--)
            {
                bool full = true;
                for (int x = 0; x < Width; x++)
                {
                    if (!_grid[readRow, x])
                    {
                        full = false;
                        break;
                    }
                }

                if (full)
                {
                    cleared++;
                    continue;
                }

                if (writeRow != readRow)
                {
                    for (int x = 0; x < Width; x++)
                    {
                        _grid[writeRow, x] = _grid[readRow, x];
                    }
                }
                writeRow--;
            }

            for (int y = writeRow; y >= 0; y--)
            {
                for (int x = 0; x < Width; x++)
                {
                    _grid[y, x] = false;
                }
            }

            return cleared;
        }

        private void RebuildColumnHeights()
        {
            for (int x = 0; x < Width; x++)
            {
                int height = 0;
                for (int y = 0; y < Height; y++)
                {
                    if (_grid[y, x])
                    {
                        height = Height - y;
                        break;
                    }
                }
                _columnHeights[x] = height;
            }
        }

        private static bool ContainsCell(Orientation orientation, int x, int y)
        {
            foreach (var cell in orientation.Cells)
            {
                if (cell.X == x && cell.Y == y)
                {
                    return true;
                }
            }
            return false;
        }
    }

    private sealed class FastScanner
    {
        private readonly TextReader _reader;
        private readonly char[] _buffer = new char[1 << 16];
        private int _length;
        private int _index;

        public FastScanner(TextReader reader)
        {
            _reader = reader;
        }

        public string NextToken()
        {
            var sb = new StringBuilder();
            int ch;
            do
            {
                ch = Read();
                if (ch == -1)
                {
                    return string.Empty;
                }
            } while (char.IsWhiteSpace((char)ch));

            while (ch != -1 && !char.IsWhiteSpace((char)ch))
            {
                sb.Append((char)ch);
                ch = Read();
            }

            return sb.ToString();
        }

        public int NextInt()
        {
            string token = NextToken();
            return int.Parse(token);
        }

        private int Read()
        {
            if (_index >= _length)
            {
                _length = _reader.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }
            return _buffer[_index++];
        }
    }
