using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.In);
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var game = new TetrisAI(width, height, numPieces);
        for (int i = 0; i < numPieces; i++)
        {
            string pieceToken = scanner.NextToken();
            if (pieceToken == null)
            {
                break;
            }

            char pieceType = pieceToken[0];
            Move move = game.ChooseMove(pieceType);
            Console.WriteLine(FormattableString.Invariant($"{move.Rotation} {move.Column}"));
            Console.Out.Flush();

            string result = scanner.NextToken();
            if (result == null)
            {
                break;
            }

            game.ApplyFeedback(pieceType, move, result);
            if (string.Equals(result, "gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }
}

internal sealed class TetrisAI
{
    private readonly int _width;
    private readonly int _height;
    private readonly int _totalPieces;
    private readonly ulong[] _rows;
    private readonly int[] _columnHeights;
    private readonly Dictionary<char, List<PieceOrientation>> _pieces;
    private int _placedPieces;
    private int _totalClearedLines;

    public TetrisAI(int width, int height, int totalPieces)
    {
        _width = width;
        _height = height;
        _totalPieces = totalPieces;
        _rows = new ulong[height];
        _columnHeights = new int[width];
        _pieces = PieceFactory.Build(width);
    }

    public Move ChooseMove(char pieceType)
    {
        List<PieceOrientation> orientations = _pieces[pieceType];
        Strategy strategy = SelectStrategy();

        BoardSnapshot current = SnapshotCurrentBoard();
        Move bestMove = default;
        double bestScore = double.NegativeInfinity;
        bool found = false;

        foreach (PieceOrientation orientation in orientations)
        {
            int maxColumn = _width - orientation.Width;
            for (int column = 0; column <= maxColumn; column++)
            {
                if (!TrySimulate(current, orientation, column, out SimulationResult result))
                {
                    continue;
                }

                double score = Evaluate(result, strategy);
                if (strategy.LookaheadDepth > 0)
                {
                    score += EvaluateLookahead(result.Board, strategy, strategy.LookaheadDepth);
                }

                if (!found || score > bestScore || (Math.Abs(score - bestScore) < 1e-9 && TieBreak(result, bestMove, moveColumn: column, moveRotation: orientation.Rotation)))
                {
                    bestScore = score;
                    bestMove = new Move(orientation.Rotation, column);
                    found = true;
                }
            }
        }

        if (!found)
        {
            return new Move(0, 0);
        }

        return bestMove;
    }

    public void ApplyFeedback(char pieceType, Move move, string result)
    {
        if (string.Equals(result, "gameover", StringComparison.OrdinalIgnoreCase))
        {
            return;
        }

        PieceOrientation orientation = _pieces[pieceType].First(p => p.Rotation == move.Rotation);
        BoardSnapshot snapshot = SnapshotCurrentBoard();
        if (!TrySimulate(snapshot, orientation, move.Column, out SimulationResult simulation))
        {
            return;
        }

        Array.Copy(simulation.Board.Rows, _rows, _height);
        Array.Copy(simulation.Board.ColumnHeights, _columnHeights, _width);
        _placedPieces++;
        _totalClearedLines += simulation.ClearedLines;
    }

    private Strategy SelectStrategy()
    {
        double fillRatio = 0.0;
        for (int r = 0; r < _height; r++)
        {
            fillRatio += BitCount(_rows[r]);
        }
        fillRatio /= Math.Max(1.0, _width * (double)_height);

        int maxHeight = 0;
        int aggregateHeight = 0;
        for (int i = 0; i < _width; i++)
        {
            maxHeight = Math.Max(maxHeight, _columnHeights[i]);
            aggregateHeight += _columnHeights[i];
        }

        bool crowded = maxHeight >= _height - 4 || fillRatio > 0.55;
        bool marathon = _totalPieces >= 600 || _width >= 14;
        bool early = _placedPieces < Math.Min(30, _totalPieces / 5 + 1);

        if (crowded)
        {
            return Strategy.Survival();
        }

        if (!marathon && early && _width <= 12)
        {
            return Strategy.Tactical();
        }

        return Strategy.Efficient();
    }

    private double EvaluateLookahead(BoardSnapshot board, Strategy strategy, int depth)
    {
        if (depth <= 0)
        {
            return 0.0;
        }

        char[] samplePieces = { 'I', 'T', 'L', 'J', 'S', 'Z', 'O' };
        double total = 0.0;
        int considered = 0;

        foreach (char nextPiece in samplePieces)
        {
            double best = double.NegativeInfinity;
            foreach (PieceOrientation orientation in _pieces[nextPiece])
            {
                int maxColumn = _width - orientation.Width;
                for (int column = 0; column <= maxColumn; column++)
                {
                    if (!TrySimulate(board, orientation, column, out SimulationResult result))
                    {
                        continue;
                    }

                    double score = Evaluate(result, strategy);
                    if (depth > 1)
                    {
                        score += 0.45 * EvaluateLookahead(result.Board, strategy, depth - 1);
                    }

                    if (score > best)
                    {
                        best = score;
                    }
                }
            }

            if (best > double.NegativeInfinity)
            {
                total += best;
                considered++;
            }
        }

        if (considered == 0)
        {
            return 0.0;
        }

        return strategy.LookaheadWeight * (total / considered);
    }

    private double Evaluate(SimulationResult result, Strategy strategy)
    {
        BoardMetrics m = AnalyzeBoard(result.Board);

        double score = 0.0;
        score += strategy.LineClearWeight * result.ClearedLines;
        score -= strategy.AggregateHeightWeight * m.AggregateHeight;
        score -= strategy.HoleWeight * m.Holes;
        score -= strategy.BumpinessWeight * m.Bumpiness;
        score -= strategy.WellWeight * m.WellDepth;
        score -= strategy.BlockadeWeight * m.Blockades;
        score -= strategy.TopPenaltyWeight * m.TopPenalty;
        score -= strategy.RowTransitionWeight * m.RowTransitions;
        score -= strategy.ColumnTransitionWeight * m.ColumnTransitions;
        score += strategy.FlatnessWeight * m.FlatColumns;
        score += strategy.LowPlacementWeight * (_height - result.LockRow);

        if (m.MaxHeight >= _height)
        {
            score -= 1_000_000.0;
        }

        return score;
    }

    private bool TieBreak(SimulationResult candidate, Move currentBest, int moveColumn, int moveRotation)
    {
        if (moveRotation != currentBest.Rotation)
        {
            return moveRotation < currentBest.Rotation;
        }

        return moveColumn < currentBest.Column;
    }

    private BoardMetrics AnalyzeBoard(BoardSnapshot board)
    {
        int aggregateHeight = 0;
        int maxHeight = 0;
        int bumpiness = 0;
        int holes = 0;
        int blockades = 0;
        int flatColumns = 0;
        int wellDepth = 0;
        int topPenalty = 0;

        for (int c = 0; c < _width; c++)
        {
            int h = board.ColumnHeights[c];
            aggregateHeight += h;
            maxHeight = Math.Max(maxHeight, h);
            if (h >= _height - 2)
            {
                topPenalty += h - (_height - 3);
            }

            bool seenBlock = false;
            for (int r = _height - 1; r >= 0; r--)
            {
                bool occupied = ((board.Rows[r] >> c) & 1UL) != 0;
                if (occupied)
                {
                    seenBlock = true;
                }
                else if (seenBlock)
                {
                    holes++;
                }
            }

            for (int r = 0; r < board.ColumnHeights[c]; r++)
            {
                bool occupied = ((board.Rows[r] >> c) & 1UL) != 0;
                if (!occupied)
                {
                    for (int above = r + 1; above < board.ColumnHeights[c]; above++)
                    {
                        if (((board.Rows[above] >> c) & 1UL) != 0)
                        {
                            blockades++;
                            break;
                        }
                    }
                }
            }

            int left = c == 0 ? _height : board.ColumnHeights[c - 1];
            int right = c == _width - 1 ? _height : board.ColumnHeights[c + 1];
            int well = Math.Max(0, Math.Min(left, right) - h);
            wellDepth += well;

            if (c > 0)
            {
                int diff = Math.Abs(board.ColumnHeights[c] - board.ColumnHeights[c - 1]);
                bumpiness += diff;
                if (diff == 0)
                {
                    flatColumns++;
                }
            }
        }

        int rowTransitions = 0;
        for (int r = 0; r < _height; r++)
        {
            bool prev = true;
            for (int c = 0; c < _width; c++)
            {
                bool current = ((board.Rows[r] >> c) & 1UL) != 0;
                if (current != prev)
                {
                    rowTransitions++;
                }
                prev = current;
            }
            if (!prev)
            {
                rowTransitions++;
            }
        }

        int columnTransitions = 0;
        for (int c = 0; c < _width; c++)
        {
            bool prev = true;
            for (int r = _height - 1; r >= 0; r--)
            {
                bool current = ((board.Rows[r] >> c) & 1UL) != 0;
                if (current != prev)
                {
                    columnTransitions++;
                }
                prev = current;
            }
            if (!prev)
            {
                columnTransitions++;
            }
        }

        return new BoardMetrics(
            aggregateHeight,
            maxHeight,
            bumpiness,
            holes,
            blockades,
            flatColumns,
            wellDepth,
            topPenalty,
            rowTransitions,
            columnTransitions);
    }

    private BoardSnapshot SnapshotCurrentBoard()
    {
        ulong[] rows = new ulong[_height];
        int[] heights = new int[_width];
        Array.Copy(_rows, rows, _height);
        Array.Copy(_columnHeights, heights, _width);
        return new BoardSnapshot(rows, heights);
    }

    private bool TrySimulate(BoardSnapshot board, PieceOrientation orientation, int column, out SimulationResult result)
    {
        result = default;
        if (column < 0 || column + orientation.Width > _width)
        {
            return false;
        }

        int baseRow = 0;
        for (int i = 0; i < orientation.Width; i++)
        {
            int boardColumn = column + i;
            int candidate = board.ColumnHeights[boardColumn] - orientation.BottomOffsets[i];
            if (candidate > baseRow)
            {
                baseRow = candidate;
            }
        }

        if (baseRow < 0)
        {
            baseRow = 0;
        }

        if (baseRow + orientation.Height > _height)
        {
            return false;
        }

        ulong[] rows = new ulong[_height];
        int[] heights = new int[_width];
        Array.Copy(board.Rows, rows, _height);
        Array.Copy(board.ColumnHeights, heights, _width);

        foreach ((int x, int y) in orientation.Cells)
        {
            int r = baseRow + y;
            int c = column + x;
            ulong bit = 1UL << c;
            if ((rows[r] & bit) != 0)
            {
                return false;
            }
            rows[r] |= bit;
        }

        for (int i = 0; i < orientation.Width; i++)
        {
            int boardColumn = column + i;
            int newHeight = baseRow + orientation.TopOffsets[i] + 1;
            if (newHeight > heights[boardColumn])
            {
                heights[boardColumn] = newHeight;
            }
        }

        int cleared = 0;
        ulong fullMask = _width == 64 ? ulong.MaxValue : ((1UL << _width) - 1UL);
        int write = 0;
        for (int read = 0; read < _height; read++)
        {
            if ((rows[read] & fullMask) == fullMask)
            {
                cleared++;
                continue;
            }
            rows[write++] = rows[read];
        }

        while (write < _height)
        {
            rows[write++] = 0;
        }

        RecomputeHeights(rows, heights);
        result = new SimulationResult(new BoardSnapshot(rows, heights), cleared, baseRow);
        return true;
    }

    private void RecomputeHeights(ulong[] rows, int[] heights)
    {
        Array.Clear(heights, 0, heights.Length);
        for (int c = 0; c < _width; c++)
        {
            for (int r = _height - 1; r >= 0; r--)
            {
                if (((rows[r] >> c) & 1UL) != 0)
                {
                    heights[c] = r + 1;
                    break;
                }
            }
        }
    }

    private static int BitCount(ulong value)
    {
        int count = 0;
        while (value != 0)
        {
            value &= value - 1;
            count++;
        }
        return count;
    }
}

internal readonly record struct BoardMetrics(
    int AggregateHeight,
    int MaxHeight,
    int Bumpiness,
    int Holes,
    int Blockades,
    int FlatColumns,
    int WellDepth,
    int TopPenalty,
    int RowTransitions,
    int ColumnTransitions);

internal readonly record struct SimulationResult(BoardSnapshot Board, int ClearedLines, int LockRow);

internal sealed class BoardSnapshot
{
    public BoardSnapshot(ulong[] rows, int[] columnHeights)
    {
        Rows = rows;
        ColumnHeights = columnHeights;
    }

    public ulong[] Rows { get; }
    public int[] ColumnHeights { get; }
}

internal sealed class PieceOrientation
{
    public PieceOrientation(int rotation, (int x, int y)[] cells)
    {
        Rotation = rotation;
        Cells = cells;
        Width = cells.Max(c => c.x) + 1;
        Height = cells.Max(c => c.y) + 1;
        BottomOffsets = new int[Width];
        TopOffsets = new int[Width];
        for (int i = 0; i < Width; i++)
        {
            BottomOffsets[i] = int.MaxValue;
            TopOffsets[i] = int.MinValue;
        }

        foreach ((int x, int y) in cells)
        {
            if (y < BottomOffsets[x])
            {
                BottomOffsets[x] = y;
            }
            if (y > TopOffsets[x])
            {
                TopOffsets[x] = y;
            }
        }
    }

    public int Rotation { get; }
    public (int x, int y)[] Cells { get; }
    public int Width { get; }
    public int Height { get; }
    public int[] BottomOffsets { get; }
    public int[] TopOffsets { get; }
}

internal static class PieceFactory
{
    public static Dictionary<char, List<PieceOrientation>> Build(int width)
    {
        var raw = new Dictionary<char, (int x, int y)[][]>
        {
            ['I'] = new[]
            {
                new[] { (0, 0), (1, 0), (2, 0), (3, 0) },
                new[] { (0, 0), (0, 1), (0, 2), (0, 3) }
            },
            ['O'] = new[]
            {
                new[] { (0, 0), (1, 0), (0, 1), (1, 1) }
            },
            ['T'] = new[]
            {
                new[] { (0, 0), (1, 0), (2, 0), (1, 1) },
                new[] { (0, 0), (0, 1), (0, 2), (1, 1) },
                new[] { (1, 0), (0, 1), (1, 1), (2, 1) },
                new[] { (1, 0), (0, 1), (1, 1), (1, 2) }
            },
            ['S'] = new[]
            {
                new[] { (1, 0), (2, 0), (0, 1), (1, 1) },
                new[] { (0, 0), (0, 1), (1, 1), (1, 2) }
            },
            ['Z'] = new[]
            {
                new[] { (0, 0), (1, 0), (1, 1), (2, 1) },
                new[] { (1, 0), (0, 1), (1, 1), (0, 2) }
            },
            ['J'] = new[]
            {
                new[] { (0, 0), (0, 1), (1, 0), (2, 0) },
                new[] { (0, 0), (1, 0), (1, 1), (1, 2) },
                new[] { (0, 1), (1, 1), (2, 1), (2, 0) },
                new[] { (0, 2), (0, 1), (0, 0), (1, 2) }
            },
            ['L'] = new[]
            {
                new[] { (0, 0), (1, 0), (2, 0), (2, 1) },
                new[] { (0, 2), (1, 0), (1, 1), (1, 2) },
                new[] { (0, 0), (0, 1), (1, 1), (2, 1) },
                new[] { (0, 0), (0, 1), (0, 2), (1, 0) }
            }
        };

        var result = new Dictionary<char, List<PieceOrientation>>();
        foreach (var kvp in raw)
        {
            var list = new List<PieceOrientation>();
            for (int rotation = 0; rotation < kvp.Value.Length; rotation++)
            {
                PieceOrientation orientation = new(rotation, kvp.Value[rotation]);
                if (orientation.Width <= width)
                {
                    list.Add(orientation);
                }
            }
            result[kvp.Key] = list;
        }

        return result;
    }
}

internal sealed class FastScanner
{
    private readonly TextReader _reader;
    private readonly char[] _buffer = new char[1 << 16];
    private int _length;
    private int _index;

    public FastScanner(TextReader reader)
    {
        _reader = reader;
    }

    public string NextToken()
    {
        SkipWhiteSpace();
        if (!EnsureData())
        {
            return null;
        }

        var chars = new List<char>(16);
        while (true)
        {
            if (_index >= _length && !EnsureData())
            {
                break;
            }

            char c = _buffer[_index];
            if (char.IsWhiteSpace(c))
            {
                break;
            }

            chars.Add(c);
            _index++;
        }

        return new string(chars.ToArray());
    }

    public int NextInt()
    {
        string token = NextToken();
        return int.Parse(token, CultureInfo.InvariantCulture);
    }

    private void SkipWhiteSpace()
    {
        while (true)
        {
            if (_index >= _length && !EnsureData())
            {
                return;
            }

            if (!char.IsWhiteSpace(_buffer[_index]))
            {
                return;
            }

            _index++;
        }
    }

    private bool EnsureData()
    {
        while (_index >= _length)
        {
            _length = _reader.Read(_buffer, 0, _buffer.Length);
            _index = 0;
            if (_length == 0)
            {
                return false;
            }
        }

        return true;
    }
}
