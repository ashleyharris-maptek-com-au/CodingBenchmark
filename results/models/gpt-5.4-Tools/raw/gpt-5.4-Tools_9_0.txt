using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.NextInt();
        int stockLength = reader.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int length = reader.NextInt();
            if (length > stockLength)
            {
                throw new InvalidOperationException("Cut length exceeds stock length.");
            }

            cuts[i] = new Cut(length, i);
        }

        Array.Sort(cuts, static (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            return cmp != 0 ? cmp : a.Index.CompareTo(b.Index);
        });

        var bins = new List<Bin>(Math.Max(4, n / 4));
        var tree = new BinTree(Math.Max(4, n));

        for (int i = 0; i < cuts.Length; i++)
        {
            ref readonly var cut = ref cuts[i];
            int binIndex = tree.FindFirstBinWithCapacity(cut.Length);
            if (binIndex < 0)
            {
                binIndex = bins.Count;
                bins.Add(new Bin(stockLength));
                tree.AddBin(stockLength);
            }

            bins[binIndex].Cuts.Add(cut.Index);
            bins[binIndex].Remaining -= cut.Length;
            tree.Update(binIndex, bins[binIndex].Remaining);
        }

        var output = new StringBuilder(Math.Max(32, n * 4));
        output.Append(bins.Count).Append('\n');
        for (int i = 0; i < bins.Count; i++)
        {
            var assigned = bins[i].Cuts;
            for (int j = 0; j < assigned.Count; j++)
            {
                if (j > 0)
                {
                    output.Append(' ');
                }

                output.Append(assigned[j]);
            }

            output.Append('\n');
        }

        Console.Write(output.ToString());
    }

    private readonly struct Cut
    {
        public readonly int Length;
        public readonly int Index;

        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    private sealed class Bin
    {
        public int Remaining;
        public readonly List<int> Cuts;

        public Bin(int stockLength)
        {
            Remaining = stockLength;
            Cuts = new List<int>();
        }
    }

    private sealed class BinTree
    {
        private int _size;
        private int _count;
        private int[] _tree;

        public BinTree(int capacity)
        {
            _size = 1;
            while (_size < capacity)
            {
                _size <<= 1;
            }

            _tree = new int[_size << 1];
        }

        public void AddBin(int remaining)
        {
            EnsureCapacity(_count + 1);
            Update(_count, remaining);
            _count++;
        }

        public void Update(int index, int remaining)
        {
            int pos = index + _size;
            _tree[pos] = remaining;
            pos >>= 1;

            while (pos > 0)
            {
                int left = pos << 1;
                int right = left | 1;
                _tree[pos] = _tree[left] >= _tree[right] ? _tree[left] : _tree[right];
                pos >>= 1;
            }
        }

        public int FindFirstBinWithCapacity(int needed)
        {
            if (_count == 0 || _tree[1] < needed)
            {
                return -1;
            }

            int node = 1;
            while (node < _size)
            {
                int left = node << 1;
                if (_tree[left] >= needed)
                {
                    node = left;
                }
                else
                {
                    node = left | 1;
                }
            }

            int index = node - _size;
            return index < _count ? index : -1;
        }

        private void EnsureCapacity(int neededCount)
        {
            if (neededCount <= _size)
            {
                return;
            }

            int newSize = _size;
            while (newSize < neededCount)
            {
                newSize <<= 1;
            }

            var newTree = new int[newSize << 1];
            for (int i = 0; i < _count; i++)
            {
                newTree[newSize + i] = _tree[_size + i];
            }

            for (int i = newSize - 1; i > 0; i--)
            {
                int left = i << 1;
                int right = left | 1;
                newTree[i] = newTree[left] >= newTree[right] ? newTree[left] : newTree[right];
            }

            _size = newSize;
            _tree = newTree;
        }
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastReader(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0)
                {
                    throw new EndOfStreamException();
                }
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }
}
