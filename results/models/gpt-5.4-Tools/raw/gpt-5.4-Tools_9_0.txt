using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.NextInt();
        int stockLength = reader.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int length = reader.NextInt();
            if (length > stockLength)
            {
                throw new InvalidOperationException("Cut length exceeds stock length.");
            }

            cuts[i] = new Cut(length, i);
        }

        Array.Sort(cuts, static (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            return cmp != 0 ? cmp : a.Index.CompareTo(b.Index);
        });

        var bins = new List<Bin>(Math.Min(n, 1024));
        var groups = new Dictionary<int, List<int>>();

        for (int i = 0; i < cuts.Length; i++)
        {
            ref readonly var cut = ref cuts[i];

            if (!groups.TryGetValue(cut.Length, out var candidateIds))
            {
                candidateIds = new List<int>();
                groups[cut.Length] = candidateIds;
            }

            int bestBinId = -1;
            int bestRemaining = int.MaxValue;

            for (int j = candidateIds.Count - 1; j >= 0; j--)
            {
                int binId = candidateIds[j];
                int remaining = bins[binId].Remaining;

                if (remaining < cut.Length)
                {
                    candidateIds.RemoveAt(j);
                    continue;
                }

                int after = remaining - cut.Length;
                if (after < bestRemaining)
                {
                    bestRemaining = after;
                    bestBinId = binId;
                    if (after == 0)
                    {
                        break;
                    }
                }
            }

            if (bestBinId < 0)
            {
                bestBinId = bins.Count;
                bins.Add(new Bin(stockLength));
            }

            bins[bestBinId].Cuts.Add(cut.Index);
            bins[bestBinId].Remaining -= cut.Length;

            int newRemaining = bins[bestBinId].Remaining;
            if (newRemaining > 0)
            {
                if (!groups.TryGetValue(newRemaining, out var exactGroup))
                {
                    exactGroup = new List<int>();
                    groups[newRemaining] = exactGroup;
                }
                exactGroup.Add(bestBinId);
            }
        }

        var output = new StringBuilder(Math.Max(16, n * 4));
        output.Append(bins.Count).Append('\n');

        for (int i = 0; i < bins.Count; i++)
        {
            var assigned = bins[i].Cuts;
            for (int j = 0; j < assigned.Count; j++)
            {
                if (j > 0) output.Append(' ');
                output.Append(assigned[j]);
            }
            output.Append('\n');
        }

        Console.Write(output.ToString());
    }

    private readonly struct Cut
    {
        public readonly int Length;
        public readonly int Index;

        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    private sealed class Bin
    {
        public int Remaining;
        public readonly List<int> Cuts;

        public Bin(int stockLength)
        {
            Remaining = stockLength;
            Cuts = new List<int>(4);
        }
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastReader(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0) return -1;
            }

            return _buffer[_position++];
        }
    }
}
