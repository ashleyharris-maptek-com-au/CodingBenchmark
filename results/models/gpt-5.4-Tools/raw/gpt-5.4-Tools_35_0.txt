use std::cmp::Reverse;
use std::io::{self, Read};

#[derive(Clone)]
struct Graph {
    n: usize,
    adj: Vec<Vec<usize>>,
    closed: Vec<Vec<usize>>,
    degree: Vec<usize>,
}

fn read_graph() -> Graph {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let m: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let Some(us) = it.next() else { break };
        let Some(vs) = it.next() else { break };
        let mut u: usize = us.parse().unwrap_or(0);
        let mut v: usize = vs.parse().unwrap_or(0);
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    let degree: Vec<usize> = adj.iter().map(|nbrs| nbrs.len()).collect();
    let mut closed = Vec::with_capacity(n);
    for v in 0..n {
        let mut c = Vec::with_capacity(adj[v].len() + 1);
        c.push(v);
        c.extend_from_slice(&adj[v]);
        closed.push(c);
    }

    Graph {
        n,
        adj,
        closed,
        degree,
    }
}

fn add_vertex(graph: &Graph, v: usize, in_set: &mut [bool], cover_count: &mut [u32]) {
    if in_set[v] {
        return;
    }
    in_set[v] = true;
    for &u in &graph.closed[v] {
        cover_count[u] += 1;
    }
}

fn greedy_build(graph: &Graph, order: &[usize]) -> Vec<usize> {
    let n = graph.n;
    if n == 0 {
        return Vec::new();
    }

    let mut in_set = vec![false; n];
    let mut cover_count = vec![0u32; n];
    let mut undominated = n;

    while undominated > 0 {
        let mut best_v = usize::MAX;
        let mut best_gain = 0usize;
        let mut best_deg = 0usize;

        for &v in order {
            if in_set[v] {
                continue;
            }
            let mut gain = 0usize;
            for &u in &graph.closed[v] {
                if cover_count[u] == 0 {
                    gain += 1;
                }
            }
            if gain > best_gain || (gain == best_gain && graph.degree[v] > best_deg) {
                best_gain = gain;
                best_deg = graph.degree[v];
                best_v = v;
            }
        }

        if best_v == usize::MAX {
            break;
        }

        in_set[best_v] = true;
        for &u in &graph.closed[best_v] {
            if cover_count[u] == 0 {
                undominated -= 1;
            }
            cover_count[u] += 1;
        }
    }

    let mut isolated = Vec::new();
    for v in 0..n {
        if cover_count[v] == 0 {
            isolated.push(v);
        }
    }
    for v in isolated {
        add_vertex(graph, v, &mut in_set, &mut cover_count);
    }

    (0..n).filter(|&v| in_set[v]).collect()
}

fn prune_solution(graph: &Graph, mut solution: Vec<usize>) -> Vec<usize> {
    let n = graph.n;
    let mut in_set = vec![false; n];
    let mut cover_count = vec![0u32; n];

    for &v in &solution {
        in_set[v] = true;
        for &u in &graph.closed[v] {
            cover_count[u] += 1;
        }
    }

    solution.sort_unstable_by_key(|&v| (graph.degree[v], v));

    let mut kept = Vec::with_capacity(solution.len());
    for &v in &solution {
        let removable = graph.closed[v].iter().all(|&u| cover_count[u] >= 2);
        if removable {
            in_set[v] = false;
            for &u in &graph.closed[v] {
                cover_count[u] -= 1;
            }
        } else {
            kept.push(v);
        }
    }

    kept
}

fn improve_with_targeted_additions(graph: &Graph, solution: &[usize]) -> Vec<usize> {
    let n = graph.n;
    let mut in_set = vec![false; n];
    let mut cover_count = vec![0u32; n];

    for &v in solution {
        in_set[v] = true;
        for &u in &graph.closed[v] {
            cover_count[u] += 1;
        }
    }

    let mut forced = Vec::new();
    for v in 0..n {
        if cover_count[v] == 0 {
            let mut best = v;
            let mut best_deg = graph.degree[v];
            for &u in &graph.closed[v] {
                if graph.degree[u] > best_deg {
                    best = u;
                    best_deg = graph.degree[u];
                }
            }
            forced.push(best);
        }
    }

    forced.sort_unstable();
    forced.dedup();
    for v in forced {
        add_vertex(graph, v, &mut in_set, &mut cover_count);
    }

    let candidate: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();
    prune_solution(graph, candidate)
}

fn is_valid(graph: &Graph, solution: &[usize]) -> bool {
    let n = graph.n;
    let mut dominated = vec![false; n];
    for &v in solution {
        if v >= n {
            return false;
        }
        dominated[v] = true;
        for &u in &graph.adj[v] {
            dominated[u] = true;
        }
    }
    dominated.into_iter().all(|x| x)
}

fn solve(graph: &Graph) -> Vec<usize> {
    let n = graph.n;
    let mut order_deg_desc: Vec<usize> = (0..n).collect();
    order_deg_desc.sort_unstable_by_key(|&v| (Reverse(graph.degree[v]), v));

    let mut order_deg_asc: Vec<usize> = (0..n).collect();
    order_deg_asc.sort_unstable_by_key(|&v| (graph.degree[v], v));

    let mut best = prune_solution(graph, greedy_build(graph, &order_deg_desc));
    let alt = prune_solution(graph, greedy_build(graph, &order_deg_asc));
    if alt.len() < best.len() {
        best = alt;
    }

    let improved = improve_with_targeted_additions(graph, &best);
    if improved.len() < best.len() {
        best = improved;
    }

    if !is_valid(graph, &best) {
        let repaired = improve_with_targeted_additions(graph, &best);
        if is_valid(graph, &repaired) {
            best = repaired;
        } else {
            best = (0..n).collect();
        }
    }

    best.sort_unstable();
    best
}

fn main() {
    let graph = read_graph();
    let solution = solve(&graph);

    println!("{}", solution.len());
    if solution.is_empty() {
        println!();
    } else {
        for (i, v) in solution.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{v}");
        }
        println!();
    }
}
