use std::cmp::Ordering;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

struct Scanner {
    data: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new(data: Vec<u8>) -> Self {
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let mut value = 0usize;
        let mut found = false;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            let b = self.data[self.pos];
            if b.is_ascii_digit() {
                value = value.saturating_mul(10).saturating_add((b - b'0') as usize);
                found = true;
            }
            self.pos += 1;
        }
        if found { Some(value) } else { None }
    }
}

struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9e3779b97f4a7c15);
        mix64(self.state)
    }

    fn bounded(&mut self, bound: usize) -> usize {
        if bound <= 1 {
            0
        } else {
            (self.next_u64() % bound as u64) as usize
        }
    }
}

fn mix64(mut x: u64) -> u64 {
    x = (x ^ (x >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94d049bb133111eb);
    x ^ (x >> 31)
}

struct Dsu {
    parent: Vec<usize>,
    size: Vec<usize>,
}

impl Dsu {
    fn new(n: usize) -> Self {
        let parent = (0..n).collect();
        let size = vec![1; n];
        Self { parent, size }
    }

    fn find(&mut self, mut x: usize) -> usize {
        let mut root = x;
        while self.parent[root] != root {
            root = self.parent[root];
        }
        while self.parent[x] != x {
            let p = self.parent[x];
            self.parent[x] = root;
            x = p;
        }
        root
    }

    fn union(&mut self, a: usize, b: usize) -> bool {
        let mut ra = self.find(a);
        let mut rb = self.find(b);
        if ra == rb {
            return false;
        }
        if self.size[ra] < self.size[rb] {
            std::mem::swap(&mut ra, &mut rb);
        }
        self.parent[rb] = ra;
        self.size[ra] += self.size[rb];
        true
    }
}

struct Components {
    order: Vec<usize>,
    ranges: Vec<(usize, usize)>,
    id: Vec<usize>,
    degree_sum: Vec<usize>,
}

fn build_components(n: usize, dsu: &mut Dsu, degree: &[usize]) -> Components {
    let mut keyed = Vec::with_capacity(n);
    for v in 0..n {
        keyed.push((dsu.find(v), v));
    }
    keyed.sort_unstable_by_key(|&(root, _)| root);

    let mut order = Vec::with_capacity(n);
    let mut ranges = Vec::new();
    let mut id = vec![0usize; n];
    let mut i = 0usize;
    while i < keyed.len() {
        let root = keyed[i].0;
        let start = order.len();
        let comp = ranges.len();
        while i < keyed.len() && keyed[i].0 == root {
            let v = keyed[i].1;
            id[v] = comp;
            order.push(v);
            i += 1;
        }
        let end = order.len();
        ranges.push((start, end));
    }

    let mut degree_sum = vec![0usize; ranges.len()];
    for v in 0..n {
        degree_sum[id[v]] = degree_sum[id[v]].saturating_add(degree[v]);
    }

    Components {
        order,
        ranges,
        id,
        degree_sum,
    }
}

fn choose_adj_cap(n: usize, m: usize) -> usize {
    if m <= 2_000_000 {
        usize::MAX
    } else if n <= 3_000 {
        4096
    } else if n <= 20_000 {
        192
    } else {
        48
    }
}

fn choose_edge_cap(n: usize, m: usize) -> usize {
    if m <= 2_500_000 {
        m
    } else if n <= 2_000 {
        1_200_000
    } else if n <= 10_000 {
        700_000
    } else {
        900_000
    }
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut scanner = Scanner::new(input);
    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = scanner.next_usize().unwrap_or(0);

    if n == 0 {
        return;
    }

    let adj_cap = choose_adj_cap(n, m);
    let edge_cap = choose_edge_cap(n, m);
    let mut adj: Vec<Vec<usize>> = (0..n).map(|_| Vec::new()).collect();
    let mut degree = vec![0usize; n];
    let mut edges: Vec<Edge> = Vec::with_capacity(edge_cap.min(2_500_000));
    let mut dsu = Dsu::new(n);
    let mut rng = SplitMix64::new(0x7265_6c61_796f_7574 ^ (n as u64).rotate_left(21) ^ m as u64);
    let mut valid_edges = 0usize;
    let mut cycle_or_parallel_edges = 0usize;

    for _ in 0..m {
        let u = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }

        valid_edges = valid_edges.saturating_add(1);
        degree[u] = degree[u].saturating_add(1);
        degree[v] = degree[v].saturating_add(1);

        if adj[u].len() < adj_cap {
            adj[u].push(v);
        }
        if adj[v].len() < adj_cap {
            adj[v].push(u);
        }

        if !dsu.union(u, v) {
            cycle_or_parallel_edges = cycle_or_parallel_edges.saturating_add(1);
        }

        let e = Edge { u, v };
        if edges.len() < edge_cap {
            edges.push(e);
        } else if edge_cap > 0 {
            let j = rng.bounded(valid_edges);
            if j < edge_cap {
                edges[j] = e;
            }
        }
    }

    let comps = build_components(n, &mut dsu, &degree);
    let forest_edges = n.saturating_sub(comps.ranges.len());
    let is_forest = valid_edges == forest_edges && cycle_or_parallel_edges == 0;

    let mut pos = if is_forest {
        layout_forest(n, &adj, &degree, &comps)
    } else {
        initial_layout(n, &adj, &degree, &comps)
    };

    if !is_forest {
        if n <= 1_500 {
            let iterations = if n <= 60 {
                420
            } else if n <= 200 {
                260
            } else if n <= 700 {
                150
            } else {
                95
            };
            refine_full_force(&mut pos, &edges, iterations);
        } else {
            let (iterations, samples, edge_limit) = if n <= 10_000 {
                (70, 28, 260_000)
            } else if n <= 50_000 {
                (28, 12, 300_000)
            } else {
                (9, 6, 260_000)
            };
            refine_sampled_force(&mut pos, &edges, iterations, samples, edge_limit);
        }
    }

    normalize_degenerate(&mut pos);

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    for p in pos {
        writeln!(out, "{:.6} {:.6}", p.x, p.y).unwrap();
    }
}

fn layout_forest(n: usize, adj: &[Vec<usize>], degree: &[usize], comps: &Components) -> Vec<Point> {
    let mut pos = vec![Point::default(); n];
    let mut parent = vec![usize::MAX; n];
    let mut depth = vec![0usize; n];
    let mut traversal = Vec::new();
    let mut stack = Vec::new();

    for &(start, end) in &comps.ranges {
        let members = &comps.order[start..end];
        if members.len() == 1 {
            let v = members[0];
            pos[v] = Point::new(0.0, 0.0);
            continue;
        }

        let mut root = members[0];
        for &v in members {
            if degree[v] > degree[root] || (degree[v] == degree[root] && v < root) {
                root = v;
            }
            parent[v] = usize::MAX;
            depth[v] = 0;
        }

        traversal.clear();
        stack.clear();
        parent[root] = root;
        stack.push(root);
        while let Some(u) = stack.pop() {
            traversal.push(u);
            for &v in &adj[u] {
                if parent[v] == usize::MAX {
                    parent[v] = u;
                    depth[v] = depth[u] + 1;
                    stack.push(v);
                }
            }
        }

        let mut next_leaf = 0.0f64;
        for &u in traversal.iter().rev() {
            let mut child_count = 0usize;
            let mut first_x = 0.0;
            let mut last_x = 0.0;
            for &v in &adj[u] {
                if parent[v] == u {
                    let x = pos[v].x;
                    if child_count == 0 {
                        first_x = x;
                    }
                    last_x = x;
                    child_count += 1;
                }
            }
            if child_count == 0 {
                pos[u].x = next_leaf;
                next_leaf += 1.0;
            } else {
                pos[u].x = (first_x + last_x) * 0.5;
            }
            pos[u].y = depth[u] as f64;
        }
    }

    for p in &mut pos {
        p.x *= 1.8;
        p.y *= 2.2;
    }
    pack_components(&mut pos, comps);
    pos
}

fn initial_layout(
    n: usize,
    adj: &[Vec<usize>],
    degree: &[usize],
    comps: &Components,
) -> Vec<Point> {
    let mut pos = vec![Point::default(); n];

    for (ci, &(start, end)) in comps.ranges.iter().enumerate() {
        let members = &comps.order[start..end];
        let size = members.len();
        if size == 1 {
            pos[members[0]] = Point::new(0.0, 0.0);
            continue;
        }

        let comp_edges = comps.degree_sum[ci] / 2;
        if is_cycle_component(members, degree, comp_edges) {
            layout_cycle_component(members, adj, &mut pos);
            continue;
        }

        let avg_degree = (2.0 * comp_edges as f64) / size as f64;
        let dense = avg_degree > (size as f64).sqrt().max(14.0);
        if dense {
            layout_core_disk(members, degree, &mut pos, 2.0);
        } else if size >= 2_000 {
            layout_landmark_component(ci, members, adj, &comps.id, degree, &mut pos);
        } else {
            layout_bfs_spiral(members, adj, degree, &mut pos);
        }
    }

    pack_components(&mut pos, comps);
    pos
}

fn is_cycle_component(members: &[usize], degree: &[usize], comp_edges: usize) -> bool {
    members.len() > 2 && comp_edges == members.len() && members.iter().all(|&v| degree[v] == 2)
}

fn layout_cycle_component(members: &[usize], adj: &[Vec<usize>], pos: &mut [Point]) {
    let n = members.len();
    let mut order = Vec::with_capacity(n);
    let start = members[0];
    let mut prev = usize::MAX;
    let mut cur = start;

    for _ in 0..n {
        order.push(cur);
        let mut next = usize::MAX;
        for &v in &adj[cur] {
            if v != prev {
                next = v;
                break;
            }
        }
        if next == usize::MAX || next == start {
            break;
        }
        prev = cur;
        cur = next;
    }

    if order.len() != n {
        order.clear();
        order.extend_from_slice(members);
    }

    let radius = (n as f64 * 1.3 / std::f64::consts::TAU).max(2.0);
    for (i, &v) in order.iter().enumerate() {
        let a = std::f64::consts::TAU * i as f64 / n as f64;
        pos[v] = Point::new(radius * a.cos(), radius * a.sin());
    }
}

fn layout_core_disk(members: &[usize], degree: &[usize], pos: &mut [Point], spacing: f64) {
    let mut nodes = members.to_vec();
    nodes.sort_unstable_by(|&a, &b| match degree[b].cmp(&degree[a]) {
        Ordering::Equal => a.cmp(&b),
        other => other,
    });

    let golden = std::f64::consts::PI * (3.0 - 5.0_f64.sqrt());
    for (rank, &v) in nodes.iter().enumerate() {
        let r = (rank as f64).sqrt() * spacing;
        let a = rank as f64 * golden;
        pos[v] = Point::new(r * a.cos(), r * a.sin());
    }
}

fn layout_bfs_spiral(members: &[usize], adj: &[Vec<usize>], degree: &[usize], pos: &mut [Point]) {
    let size = members.len();
    let mut root = members[0];
    for &v in members {
        if degree[v] > degree[root] || (degree[v] == degree[root] && v < root) {
            root = v;
        }
    }

    let mut seen = std::collections::HashSet::with_capacity(size * 2);
    let mut queue = Vec::with_capacity(size);
    let mut order = Vec::with_capacity(size);
    let mut qh = 0usize;
    seen.insert(root);
    queue.push(root);
    while qh < queue.len() {
        let u = queue[qh];
        qh += 1;
        order.push(u);
        for &v in &adj[u] {
            if seen.len() >= size {
                break;
            }
            if seen.insert(v) {
                queue.push(v);
            }
        }
    }
    if order.len() < size {
        for &v in members {
            if seen.insert(v) {
                order.push(v);
            }
        }
    }

    let golden = std::f64::consts::PI * (3.0 - 5.0_f64.sqrt());
    for (i, &v) in order.iter().enumerate() {
        let r = (i as f64 + 1.0).sqrt() * 2.0;
        let a = i as f64 * golden;
        pos[v] = Point::new(r * a.cos(), r * a.sin());
    }
}

fn layout_landmark_component(
    ci: usize,
    members: &[usize],
    adj: &[Vec<usize>],
    comp_id: &[usize],
    degree: &[usize],
    pos: &mut [Point],
) {
    let n_total = pos.len();
    let mut source = members[0];
    for &v in members {
        if degree[v] > degree[source] || (degree[v] == degree[source] && v < source) {
            source = v;
        }
    }

    let d0 = bfs_component(source, ci, adj, comp_id, n_total);
    let a = farthest_member(members, &d0).unwrap_or(source);
    let da = bfs_component(a, ci, adj, comp_id, n_total);
    let b = farthest_member(members, &da).unwrap_or(source);
    let db = bfs_component(b, ci, adj, comp_id, n_total);

    let mut c = source;
    let mut best_side = -1i32;
    for &v in members {
        if da[v] >= 0 && db[v] >= 0 {
            let side = da[v].min(db[v]);
            if side > best_side {
                best_side = side;
                c = v;
            }
        }
    }
    let dc = bfs_component(c, ci, adj, comp_id, n_total);
    let d = farthest_member(members, &dc).unwrap_or(source);
    let dd = bfs_component(d, ci, adj, comp_id, n_total);

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for &v in members {
        let hv = (mix64(v as u64 ^ 0x517c_c1b7_2722_0a95) as f64) / (u64::MAX as f64);
        let jitter_angle = hv * std::f64::consts::TAU;
        let jitter = 0.35;
        let x = if da[v] >= 0 && db[v] >= 0 {
            (da[v] - db[v]) as f64 * 2.0
        } else {
            0.0
        } + jitter * jitter_angle.cos();
        let y = if dc[v] >= 0 && dd[v] >= 0 {
            (dc[v] - dd[v]) as f64 * 2.0
        } else {
            0.0
        } + jitter * jitter_angle.sin();
        pos[v] = Point::new(x, y);
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    if max_x - min_x < 4.0 && max_y - min_y < 4.0 {
        layout_core_disk(members, degree, pos, 1.7);
    }
}

fn bfs_component(
    source: usize,
    ci: usize,
    adj: &[Vec<usize>],
    comp_id: &[usize],
    n_total: usize,
) -> Vec<i32> {
    let mut dist = vec![-1i32; n_total];
    let mut queue = Vec::new();
    let mut head = 0usize;
    dist[source] = 0;
    queue.push(source);
    while head < queue.len() {
        let u = queue[head];
        head += 1;
        let nd = dist[u] + 1;
        for &v in &adj[u] {
            if comp_id[v] == ci && dist[v] < 0 {
                dist[v] = nd;
                queue.push(v);
            }
        }
    }
    dist
}

fn farthest_member(members: &[usize], dist: &[i32]) -> Option<usize> {
    let mut best = None;
    let mut best_d = -1i32;
    for &v in members {
        if dist[v] > best_d {
            best_d = dist[v];
            best = Some(v);
        }
    }
    best
}

fn pack_components(pos: &mut [Point], comps: &Components) {
    if pos.is_empty() {
        return;
    }

    let mut comp_indices: Vec<usize> = (0..comps.ranges.len()).collect();
    comp_indices.sort_unstable_by(|&a, &b| {
        let sa = comps.ranges[a].1 - comps.ranges[a].0;
        let sb = comps.ranges[b].1 - comps.ranges[b].0;
        sb.cmp(&sa).then_with(|| a.cmp(&b))
    });

    let target_width = (pos.len() as f64).sqrt() * 28.0 + 50.0;
    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_h = 0.0;

    for ci in comp_indices {
        let (start, end) = comps.ranges[ci];
        let members = &comps.order[start..end];
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for &v in members {
            min_x = min_x.min(pos[v].x);
            max_x = max_x.max(pos[v].x);
            min_y = min_y.min(pos[v].y);
            max_y = max_y.max(pos[v].y);
        }
        let w = (max_x - min_x).max(1.0);
        let h = (max_y - min_y).max(1.0);
        let pad = 4.0 + (members.len() as f64).sqrt() * 0.55;

        if cursor_x > 0.0 && cursor_x + w + pad > target_width {
            cursor_x = 0.0;
            cursor_y += row_h;
            row_h = 0.0;
        }

        let dx = cursor_x - min_x + pad * 0.5;
        let dy = cursor_y - min_y + pad * 0.5;
        for &v in members {
            pos[v].x += dx;
            pos[v].y += dy;
        }
        cursor_x += w + pad;
        row_h = row_h.max(h + pad);
    }

    center_positions(pos);
}

fn center_positions(pos: &mut [Point]) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in pos.iter() {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    for p in pos {
        p.x -= cx;
        p.y -= cy;
    }
}

fn normalize_degenerate(pos: &mut [Point]) {
    let n = pos.len();
    if n == 0 {
        return;
    }

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in pos.iter() {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    if max_x - min_x > 1e-6 || max_y - min_y > 1e-6 {
        return;
    }

    let cols = (n as f64).sqrt().ceil() as usize;
    for (i, p) in pos.iter_mut().enumerate() {
        p.x = (i % cols) as f64 * 2.0;
        p.y = (i / cols) as f64 * 2.0;
    }
    center_positions(pos);
}

fn refine_full_force(pos: &mut [Point], edges: &[Edge], iterations: usize) {
    let n = pos.len();
    if n <= 1 || iterations == 0 {
        return;
    }

    let mut disp = vec![Point::default(); n];
    let k = 5.5 + (n as f64).sqrt() * 0.12;
    let edge_limit = edges.len().min(if n <= 250 { 400_000 } else { 260_000 });

    for it in 0..iterations {
        disp.fill(Point::default());

        for i in 0..n {
            for j in (i + 1)..n {
                let mut dx = pos[i].x - pos[j].x;
                let mut dy = pos[i].y - pos[j].y;
                let d2 = dx * dx + dy * dy + 0.01;
                let d = d2.sqrt();
                dx /= d;
                dy /= d;
                let f = (k * k) / d;
                let fx = dx * f;
                let fy = dy * f;
                disp[i].x += fx;
                disp[i].y += fy;
                disp[j].x -= fx;
                disp[j].y -= fy;
            }
        }

        for e in edges.iter().take(edge_limit) {
            let u = e.u;
            let v = e.v;
            if u == v || u >= n || v >= n {
                continue;
            }
            let mut dx = pos[u].x - pos[v].x;
            let mut dy = pos[u].y - pos[v].y;
            let d2 = dx * dx + dy * dy + 0.01;
            let d = d2.sqrt();
            dx /= d;
            dy /= d;
            let f = d2 / k;
            let fx = dx * f;
            let fy = dy * f;
            disp[u].x -= fx;
            disp[u].y -= fy;
            disp[v].x += fx;
            disp[v].y += fy;
        }

        let t = k * (1.6 * (1.0 - it as f64 / iterations as f64) + 0.04);
        for i in 0..n {
            disp[i].x -= pos[i].x * 0.003;
            disp[i].y -= pos[i].y * 0.003;
            let len = (disp[i].x * disp[i].x + disp[i].y * disp[i].y).sqrt();
            if len > 1e-12 {
                let step = len.min(t);
                pos[i].x += disp[i].x / len * step;
                pos[i].y += disp[i].y / len * step;
            }
        }
    }
    center_positions(pos);
}

fn refine_sampled_force(
    pos: &mut [Point],
    edges: &[Edge],
    iterations: usize,
    samples_per_node: usize,
    edge_limit: usize,
) {
    let n = pos.len();
    if n <= 1 || iterations == 0 {
        return;
    }

    let mut disp = vec![Point::default(); n];
    let k = 7.0 + (n as f64).sqrt() * 0.025;
    let use_edges = edges.len().min(edge_limit);
    let sample_scale = if samples_per_node == 0 {
        0.0
    } else {
        (n as f64 / samples_per_node as f64).sqrt().max(1.0)
    };

    for it in 0..iterations {
        disp.fill(Point::default());

        for i in 0..n {
            let mut seed = mix64((i as u64).wrapping_mul(0x9e37_79b9_7f4a_7c15) ^ it as u64);
            for _ in 0..samples_per_node {
                seed = mix64(seed);
                let mut j = (seed % n as u64) as usize;
                if j == i {
                    j = (j + 1) % n;
                }
                let mut dx = pos[i].x - pos[j].x;
                let mut dy = pos[i].y - pos[j].y;
                let d2 = dx * dx + dy * dy + 0.05;
                let d = d2.sqrt();
                dx /= d;
                dy /= d;
                let f = (k * k / d) * sample_scale * 0.18;
                disp[i].x += dx * f;
                disp[i].y += dy * f;
            }
        }

        if use_edges > 0 {
            let stride = (edges.len() / use_edges).max(1);
            let offset = (it * 97) % edges.len();
            let mut idx = offset;
            for _ in 0..use_edges {
                let e = edges[idx];
                idx += stride;
                if idx >= edges.len() {
                    idx %= edges.len();
                }
                let u = e.u;
                let v = e.v;
                if u == v || u >= n || v >= n {
                    continue;
                }
                let mut dx = pos[u].x - pos[v].x;
                let mut dy = pos[u].y - pos[v].y;
                let d2 = dx * dx + dy * dy + 0.05;
                let d = d2.sqrt();
                dx /= d;
                dy /= d;
                let f = (d2 / k) * 0.32;
                disp[u].x -= dx * f;
                disp[u].y -= dy * f;
                disp[v].x += dx * f;
                disp[v].y += dy * f;
            }
        }

        let t = k * (1.2 * (1.0 - it as f64 / iterations as f64) + 0.03);
        for i in 0..n {
            disp[i].x -= pos[i].x * 0.0015;
            disp[i].y -= pos[i].y * 0.0015;
            let len = (disp[i].x * disp[i].x + disp[i].y * disp[i].y).sqrt();
            if len > 1e-12 {
                let step = len.min(t);
                pos[i].x += disp[i].x / len * step;
                pos[i].y += disp[i].y / len * step;
            }
        }
    }
    center_positions(pos);
}
