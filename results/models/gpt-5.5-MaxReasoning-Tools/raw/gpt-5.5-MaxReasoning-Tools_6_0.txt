using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    const double DefaultMu = 398600.4418;
    const double Inf = 1e100;
    const double Eps = 1e-9;

    struct Vec3
    {
        public double X, Y, Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double Norm()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator *(Vec3 a, double k)
        {
            return new Vec3(a.X * k, a.Y * k, a.Z * k);
        }

        public static Vec3 operator /(Vec3 a, double k)
        {
            return new Vec3(a.X / k, a.Y / k, a.Z / k);
        }

        public static double Dot(Vec3 a, Vec3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }
    }

    struct State
    {
        public Vec3 R, V;

        public State(Vec3 r, Vec3 v)
        {
            R = r;
            V = v;
        }
    }

    sealed class Orbit
    {
        public Vec3 R;
        public Vec3 V;
        public Vec3 H;
        public Vec3 HUnit;
        public Vec3 EVec;
        public double RNorm;
        public double VNorm;
        public double HNorm;
        public double Energy;
        public double A;
        public double P;
        public double RadialV;
    }

    sealed class Candidate
    {
        public int[] Route;
        public double Score;

        public Candidate(int[] route, double score)
        {
            Route = route;
            Score = score;
        }
    }

    struct XorShift64
    {
        ulong state;

        public XorShift64(ulong seed)
        {
            state = seed == 0 ? 88172645463393265UL : seed;
        }

        public ulong NextULong()
        {
            ulong x = state;
            x ^= x << 7;
            x ^= x >> 9;
            state = x;
            return x;
        }

        public int NextInt(int maxExclusive)
        {
            if (maxExclusive <= 1) return 0;
            return (int)(NextULong() % (uint)maxExclusive);
        }

        public double NextDouble()
        {
            return (NextULong() >> 11) * (1.0 / 9007199254740992.0);
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        double first;
        if (!scanner.NextDouble(out first)) return;

        int n = (int)Math.Round(first);
        double mu;
        if (!scanner.NextDouble(out mu) || !IsFinite(mu) || mu <= 0.0)
            mu = DefaultMu;

        State start = ReadState(scanner);
        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        State[] stationStates = new State[n];
        for (int i = 0; i < n; i++)
            stationStates[i] = ReadState(scanner);

        Orbit startOrbit = BuildOrbit(start, mu);
        Orbit[] stationOrbits = new Orbit[n];
        for (int i = 0; i < n; i++)
            stationOrbits[i] = BuildOrbit(stationStates[i], mu);

        double[] startCost = new double[n];
        double[,] cost = new double[n, n];

        for (int i = 0; i < n; i++)
            startCost[i] = Math.Max(Eps, EstimateTransfer(startOrbit, stationOrbits[i], mu));

        for (int i = 0; i < n; i++)
        {
            cost[i, i] = 0.0;
            for (int j = i + 1; j < n; j++)
            {
                double c = 0.5 * (EstimateTransfer(stationOrbits[i], stationOrbits[j], mu) +
                                  EstimateTransfer(stationOrbits[j], stationOrbits[i], mu));
                c = Math.Max(Eps, c);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] order = n <= 20
            ? SolveExactPath(n, startCost, cost)
            : SolveHeuristicPath(n, startCost, cost);

        order = RepairRoute(order, n, startCost, cost);
        WriteOrder(order);
    }

    static State ReadState(FastScanner scanner)
    {
        double x, y, z, vx, vy, vz;
        scanner.NextDouble(out x);
        scanner.NextDouble(out y);
        scanner.NextDouble(out z);
        scanner.NextDouble(out vx);
        scanner.NextDouble(out vy);
        scanner.NextDouble(out vz);
        return new State(new Vec3(x, y, z), new Vec3(vx, vy, vz));
    }

    static Orbit BuildOrbit(State s, double mu)
    {
        Orbit o = new Orbit();
        o.R = s.R;
        o.V = s.V;
        o.RNorm = Math.Max(s.R.Norm(), 1e-12);
        o.VNorm = s.V.Norm();
        o.H = Vec3.Cross(s.R, s.V);
        o.HNorm = o.H.Norm();
        o.HUnit = o.HNorm > 1e-12 ? o.H / o.HNorm : new Vec3(0.0, 0.0, 1.0);
        o.Energy = 0.5 * o.VNorm * o.VNorm - mu / o.RNorm;
        o.A = Math.Abs(o.Energy) > 1e-12 ? -mu / (2.0 * o.Energy) : o.RNorm;
        if (!IsFinite(o.A) || o.A <= 0.0) o.A = o.RNorm;
        o.P = o.HNorm * o.HNorm / Math.Max(mu, 1e-12);
        if (!IsFinite(o.P) || o.P <= 0.0) o.P = o.RNorm;
        o.RadialV = Vec3.Dot(s.R, s.V) / o.RNorm;
        o.EVec = Vec3.Cross(s.V, o.H) / mu - s.R / o.RNorm;
        return o;
    }

    static double EstimateTransfer(Orbit a, Orbit b, double mu)
    {
        double r1 = RepresentativeRadius(a);
        double r2 = RepresentativeRadius(b);
        double vc1 = Math.Sqrt(mu / r1);
        double vc2 = Math.Sqrt(mu / r2);

        double transferDen = Math.Max(r1 + r2, 1e-9);
        double h1 = vc1 * Math.Abs(Math.Sqrt(2.0 * r2 / transferDen) - 1.0);
        double h2 = vc2 * Math.Abs(1.0 - Math.Sqrt(2.0 * r1 / transferDen));
        double hohmann = h1 + h2;

        double planeDot = Clamp(Vec3.Dot(a.HUnit, b.HUnit), -1.0, 1.0);
        double planeAngle = Math.Acos(planeDot);
        double plane = 2.0 * Math.Min(vc1, vc2) * Math.Sin(0.5 * planeAngle);

        double pAvg = Math.Max(100.0, 0.5 * (a.P + b.P));
        double eccDiff = (a.EVec - b.EVec).Norm();
        double ecc = 0.35 * Math.Sqrt(mu / pAvg) * Math.Min(3.0, eccDiff);

        double dist = (a.R - b.R).Norm();
        double relV = (a.V - b.V).Norm();
        double scaleRadius = Math.Max(100.0, 0.5 * (r1 + r2));
        double orbitalTimeScale = Math.Sqrt(scaleRadius * scaleRadius * scaleRadius / mu);
        double phase = 0.06 * dist / Math.Max(1.0, orbitalTimeScale);
        double velocityMatch = 0.10 * relV;
        double radial = 0.07 * Math.Abs(a.RadialV - b.RadialV);

        double energy = 0.04 * Math.Abs(a.Energy - b.Energy) / Math.Max(0.25, 0.5 * (vc1 + vc2));
        double total = hohmann + plane + ecc + phase + velocityMatch + radial + energy;
        if (!IsFinite(total) || total < Eps)
            total = Math.Max(Eps, relV + 0.0001 * dist);
        return total;
    }

    static double RepresentativeRadius(Orbit o)
    {
        double r = (IsFinite(o.A) && o.A > 100.0 && o.A < 1e9) ? o.A : o.RNorm;
        if (!IsFinite(r) || r < 100.0) r = Math.Max(100.0, o.RNorm);
        return Clamp(r, 100.0, 1e9);
    }

    static int[] SolveExactPath(int n, double[] startCost, double[,] cost)
    {
        if (n == 1) return new int[] { 0 };

        int maskCount = 1 << n;
        int allMask = maskCount - 1;
        int stateCount = maskCount * n;
        double[] dp = new double[stateCount];
        short[] parent = new short[stateCount];
        for (int i = 0; i < stateCount; i++)
            dp[i] = Inf;

        int[] bitIndex = new int[maskCount];
        for (int i = 0; i < n; i++)
            bitIndex[1 << i] = i;

        for (int i = 0; i < n; i++)
        {
            int idx = ((1 << i) * n) + i;
            dp[idx] = startCost[i];
            parent[idx] = -1;
        }

        for (int mask = 1; mask < maskCount; mask++)
        {
            int bits = mask;
            while (bits != 0)
            {
                int lb = bits & -bits;
                int last = bitIndex[lb];
                int idx = mask * n + last;
                double here = dp[idx];
                if (here < Inf * 0.5)
                {
                    int rem = allMask ^ mask;
                    while (rem != 0)
                    {
                        int nb = rem & -rem;
                        int next = bitIndex[nb];
                        int nextMask = mask | nb;
                        int nextIdx = nextMask * n + next;
                        double candidate = here + cost[last, next];
                        if (candidate < dp[nextIdx])
                        {
                            dp[nextIdx] = candidate;
                            parent[nextIdx] = (short)last;
                        }
                        rem -= nb;
                    }
                }
                bits -= lb;
            }
        }

        int bestLast = 0;
        double best = Inf;
        int finalBase = allMask * n;
        for (int i = 0; i < n; i++)
        {
            double v = dp[finalBase + i];
            if (v < best)
            {
                best = v;
                bestLast = i;
            }
        }

        int[] route = new int[n];
        int curMask = allMask;
        int cur = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = cur;
            int idx = curMask * n + cur;
            int prev = parent[idx];
            curMask ^= 1 << cur;
            cur = prev;
        }

        return route;
    }

    static int[] SolveHeuristicPath(int n, double[] startCost, double[,] cost)
    {
        var candidates = new List<Candidate>();
        int[] startOrder = SortByStartCost(n, startCost);

        int nnCount = n <= 60 ? n : Math.Min(n, 44);
        for (int i = 0; i < nnCount; i++)
            AddCandidate(candidates, BuildNearestRoute(n, startCost, cost, startOrder[i]), startCost, cost);

        if (nnCount < n)
        {
            int step = Math.Max(1, n / 18);
            for (int i = nnCount; i < n; i += step)
                AddCandidate(candidates, BuildNearestRoute(n, startCost, cost, startOrder[i]), startCost, cost);
        }

        int seedLimit = Math.Min(n, 14);
        for (int i = 0; i < seedLimit; i++)
        {
            int a = startOrder[i];
            AddCandidate(candidates, BuildRegretInsertion(n, startCost, cost, new int[] { a }), startCost, cost);

            int b = NearestStation(a, n, cost);
            if (b >= 0)
            {
                AddCandidate(candidates, BuildRegretInsertion(n, startCost, cost, new int[] { a, b }), startCost, cost);
                AddCandidate(candidates, BuildRegretInsertion(n, startCost, cost, new int[] { b, a }), startCost, cost);
            }
        }

        int pairA = 0, pairB = n > 1 ? 1 : 0;
        double pairBest = Inf;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                double v = startCost[i] + cost[i, j];
                if (v < pairBest)
                {
                    pairBest = v;
                    pairA = i;
                    pairB = j;
                }
            }
        }
        AddCandidate(candidates, BuildRegretInsertion(n, startCost, cost, new int[] { pairA, pairB }), startCost, cost);

        XorShift64 rng = new XorShift64(0x9E3779B97F4A7C15UL ^ (ulong)n);
        int randomStarts = Math.Min(36, Math.Max(10, n / 2));
        for (int i = 0; i < randomStarts; i++)
        {
            int first = startOrder[rng.NextInt(Math.Min(n, Math.Max(1, seedLimit * 3)))];
            AddCandidate(candidates, BuildRandomizedNearestRoute(n, startCost, cost, first, 5, ref rng), startCost, cost);
        }

        candidates.Sort((a, b) => a.Score.CompareTo(b.Score));

        int keep = Math.Min(candidates.Count, n <= 40 ? 24 : 16);
        int[] bestRoute = null;
        double bestScore = Inf;

        for (int i = 0; i < keep; i++)
        {
            int[] route = (int[])candidates[i].Route.Clone();
            ImproveRoute(route, startCost, cost);
            double score = RouteCost(route, startCost, cost);
            if (score < bestScore)
            {
                bestScore = score;
                bestRoute = route;
            }
        }

        if (bestRoute == null)
            bestRoute = BuildNearestRoute(n, startCost, cost, startOrder[0]);

        return bestRoute;
    }

    static int[] SortByStartCost(int n, double[] startCost)
    {
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => startCost[a].CompareTo(startCost[b]));
        return order;
    }

    static void AddCandidate(List<Candidate> candidates, int[] route, double[] startCost, double[,] cost)
    {
        candidates.Add(new Candidate(route, RouteCost(route, startCost, cost)));
    }

    static int[] BuildNearestRoute(int n, double[] startCost, double[,] cost, int first)
    {
        bool[] used = new bool[n];
        int[] route = new int[n];
        route[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int prev = route[pos - 1];
            int best = -1;
            double bestCost = Inf;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[prev, j];
                if (c < bestCost)
                {
                    bestCost = c;
                    best = j;
                }
            }
            route[pos] = best;
            used[best] = true;
        }

        return route;
    }

    static int[] BuildRandomizedNearestRoute(int n, double[] startCost, double[,] cost, int first, int choiceCount, ref XorShift64 rng)
    {
        bool[] used = new bool[n];
        int[] route = new int[n];
        int[] bestNodes = new int[choiceCount];
        double[] bestValues = new double[choiceCount];

        route[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            for (int k = 0; k < choiceCount; k++)
            {
                bestNodes[k] = -1;
                bestValues[k] = Inf;
            }

            int prev = route[pos - 1];
            int filled = 0;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[prev, j];
                int at = Math.Min(filled, choiceCount - 1);
                if (c >= bestValues[at]) continue;
                while (at > 0 && c < bestValues[at - 1])
                {
                    if (at < choiceCount)
                    {
                        bestValues[at] = bestValues[at - 1];
                        bestNodes[at] = bestNodes[at - 1];
                    }
                    at--;
                }
                bestValues[at] = c;
                bestNodes[at] = j;
                if (filled < choiceCount) filled++;
            }

            int available = 0;
            while (available < choiceCount && bestNodes[available] >= 0) available++;
            double biased = rng.NextDouble();
            int pick = Math.Min(available - 1, (int)(biased * biased * available));
            int next = bestNodes[Math.Max(0, pick)];
            route[pos] = next;
            used[next] = true;
        }

        return route;
    }

    static int[] BuildRegretInsertion(int n, double[] startCost, double[,] cost, int[] seed)
    {
        bool[] used = new bool[n];
        int[] route = new int[n];
        int len = 0;

        for (int i = 0; i < seed.Length; i++)
        {
            int s = seed[i];
            if (s < 0 || s >= n || used[s]) continue;
            route[len++] = s;
            used[s] = true;
        }

        if (len == 0)
        {
            int first = 0;
            for (int i = 1; i < n; i++)
                if (startCost[i] < startCost[first]) first = i;
            route[len++] = first;
            used[first] = true;
        }

        while (len < n)
        {
            int chosenStation = -1;
            int chosenPos = 0;
            double chosenPriority = -Inf;
            double chosenBestDelta = Inf;

            for (int station = 0; station < n; station++)
            {
                if (used[station]) continue;

                double best = Inf;
                double second = Inf;
                int bestPos = 0;

                for (int pos = 0; pos <= len; pos++)
                {
                    double delta = InsertionDelta(route, len, pos, station, startCost, cost);
                    if (delta < best)
                    {
                        second = best;
                        best = delta;
                        bestPos = pos;
                    }
                    else if (delta < second)
                    {
                        second = delta;
                    }
                }

                if (second >= Inf * 0.5) second = best + Math.Abs(best);
                double priority = (second - best) - 0.20 * best;
                if (priority > chosenPriority ||
                    (Math.Abs(priority - chosenPriority) < 1e-12 && best < chosenBestDelta))
                {
                    chosenPriority = priority;
                    chosenBestDelta = best;
                    chosenStation = station;
                    chosenPos = bestPos;
                }
            }

            for (int k = len; k > chosenPos; k--)
                route[k] = route[k - 1];
            route[chosenPos] = chosenStation;
            used[chosenStation] = true;
            len++;
        }

        return route;
    }

    static int NearestStation(int station, int n, double[,] cost)
    {
        int best = -1;
        double bestCost = Inf;
        for (int i = 0; i < n; i++)
        {
            if (i == station) continue;
            if (cost[station, i] < bestCost)
            {
                bestCost = cost[station, i];
                best = i;
            }
        }
        return best;
    }

    static double InsertionDelta(int[] route, int len, int pos, int station, double[] startCost, double[,] cost)
    {
        if (len == 0) return startCost[station];
        if (pos == 0)
            return startCost[station] + cost[station, route[0]] - startCost[route[0]];
        if (pos == len)
            return cost[route[len - 1], station];

        int left = route[pos - 1];
        int right = route[pos];
        return cost[left, station] + cost[station, right] - cost[left, right];
    }

    static void ImproveRoute(int[] route, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        int maxRounds = n <= 40 ? 90 : 60;

        for (int round = 0; round < maxRounds; round++)
        {
            bool improved = false;
            if (TwoOptBest(route, startCost, cost)) improved = true;
            if (OrOptBest(route, 1, startCost, cost)) improved = true;
            if (n > 8 && OrOptBest(route, 2, startCost, cost)) improved = true;
            if (n > 18 && OrOptBest(route, 3, startCost, cost)) improved = true;
            if (!improved) break;
        }
    }

    static bool TwoOptBest(int[] route, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        if (n < 2) return false;

        double bestDelta = -Eps;
        int bestI = -1;
        int bestK = -1;

        for (int i = 0; i < n - 1; i++)
        {
            int left = i == 0 ? -1 : route[i - 1];
            int first = route[i];

            for (int k = i + 1; k < n; k++)
            {
                int last = route[k];
                int right = k + 1 < n ? route[k + 1] : -1;

                double oldCost = EdgeCost(left, first, startCost, cost);
                if (right >= 0) oldCost += EdgeCost(last, right, startCost, cost);

                double newCost = EdgeCost(left, last, startCost, cost);
                if (right >= 0) newCost += EdgeCost(first, right, startCost, cost);

                double delta = newCost - oldCost;
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestI = i;
                    bestK = k;
                }
            }
        }

        if (bestI < 0) return false;
        Array.Reverse(route, bestI, bestK - bestI + 1);
        return true;
    }

    static bool OrOptBest(int[] route, int len, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        if (len >= n) return false;

        double bestDelta = -Eps;
        int bestI = -1;
        int bestJ = -1;

        for (int i = 0; i + len <= n; i++)
        {
            int end = i + len - 1;
            for (int j = 0; j <= n; j++)
            {
                if (j >= i && j <= end + 1) continue;
                double delta = MoveDelta(route, i, len, j, startCost, cost);
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestI = i;
                    bestJ = j;
                }
            }
        }

        if (bestI < 0) return false;
        ApplyMove(route, bestI, len, bestJ);
        return true;
    }

    static double MoveDelta(int[] route, int i, int len, int j, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        int end = i + len - 1;
        int first = route[i];
        int last = route[end];
        int left = i > 0 ? route[i - 1] : -1;
        int right = end + 1 < n ? route[end + 1] : -1;

        double oldRemove = EdgeCost(left, first, startCost, cost);
        if (right >= 0) oldRemove += EdgeCost(last, right, startCost, cost);

        double newRemove = right >= 0 ? EdgeCost(left, right, startCost, cost) : 0.0;

        int insertLeft;
        int insertRight;
        if (j < i)
        {
            insertLeft = j > 0 ? route[j - 1] : -1;
            insertRight = route[j];
        }
        else
        {
            insertLeft = j > 0 ? route[j - 1] : -1;
            insertRight = j < n ? route[j] : -1;
        }

        double oldInsert = insertRight >= 0 ? EdgeCost(insertLeft, insertRight, startCost, cost) : 0.0;
        double newInsert = EdgeCost(insertLeft, first, startCost, cost);
        if (insertRight >= 0) newInsert += EdgeCost(last, insertRight, startCost, cost);

        return (newRemove - oldRemove) + (newInsert - oldInsert);
    }

    static void ApplyMove(int[] route, int i, int len, int j)
    {
        int n = route.Length;
        int end = i + len;
        int[] tmp = new int[n];
        int p = 0;

        if (j < i)
        {
            for (int k = 0; k < j; k++) tmp[p++] = route[k];
            for (int k = i; k < end; k++) tmp[p++] = route[k];
            for (int k = j; k < i; k++) tmp[p++] = route[k];
            for (int k = end; k < n; k++) tmp[p++] = route[k];
        }
        else
        {
            for (int k = 0; k < i; k++) tmp[p++] = route[k];
            for (int k = end; k < j; k++) tmp[p++] = route[k];
            for (int k = i; k < end; k++) tmp[p++] = route[k];
            for (int k = j; k < n; k++) tmp[p++] = route[k];
        }

        Array.Copy(tmp, route, n);
    }

    static int[] RepairRoute(int[] route, int n, double[] startCost, double[,] cost)
    {
        bool[] seen = new bool[n];
        int[] clean = new int[n];
        int len = 0;

        if (route != null)
        {
            for (int i = 0; i < route.Length; i++)
            {
                int s = route[i];
                if (s >= 0 && s < n && !seen[s])
                {
                    clean[len++] = s;
                    seen[s] = true;
                }
            }
        }

        for (int s = 0; s < n; s++)
        {
            if (seen[s]) continue;
            int bestPos = 0;
            double bestDelta = Inf;
            for (int pos = 0; pos <= len; pos++)
            {
                double d = InsertionDelta(clean, len, pos, s, startCost, cost);
                if (d < bestDelta)
                {
                    bestDelta = d;
                    bestPos = pos;
                }
            }

            for (int k = len; k > bestPos; k--)
                clean[k] = clean[k - 1];
            clean[bestPos] = s;
            seen[s] = true;
            len++;
        }

        return clean;
    }

    static double RouteCost(int[] route, double[] startCost, double[,] cost)
    {
        if (route == null || route.Length == 0) return 0.0;
        double total = startCost[route[0]];
        for (int i = 1; i < route.Length; i++)
            total += cost[route[i - 1], route[i]];
        return total;
    }

    static double EdgeCost(int left, int right, double[] startCost, double[,] cost)
    {
        if (right < 0) return 0.0;
        return left < 0 ? startCost[right] : cost[left, right];
    }

    static void WriteOrder(int[] order)
    {
        var sb = new StringBuilder();
        for (int i = 0; i < order.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(order[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    static bool IsFinite(double x)
    {
        return !double.IsNaN(x) && !double.IsInfinity(x);
    }

    sealed class FastScanner
    {
        readonly Stream stream;
        readonly byte[] buffer = new byte[1 << 16];
        int length;
        int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        int Read()
        {
            if (ptr >= length)
            {
                length = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (length <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public bool NextDouble(out double value)
        {
            value = 0.0;
            int c;
            do
            {
                c = Read();
                if (c < 0) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-' || c == '+')
            {
                if (c == '-') sign = -1;
                c = Read();
            }

            double val = 0.0;
            bool any = false;
            while (c >= '0' && c <= '9')
            {
                any = true;
                val = val * 10.0 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double scale = 0.1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    any = true;
                    val += (c - '0') * scale;
                    scale *= 0.1;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int expSign = 1;
                int exp = 0;
                c = Read();
                if (c == '-' || c == '+')
                {
                    if (c == '-') expSign = -1;
                    c = Read();
                }

                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }

                if (exp != 0)
                    val *= Math.Pow(10.0, expSign * exp);
            }

            value = sign * val;
            return any;
        }
    }
