struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    double operator[](int i) const {
        return i == 0 ? x : (i == 1 ? y : z);
    }

    double& operator[](int i) {
        return i == 0 ? x : (i == 1 ? y : z);
    }
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static Vec3 operator*(double s, const Vec3& a) { return a * s; }
static Vec3 operator/(const Vec3& a, double s) { return {a.x / s, a.y / s, a.z / s}; }

static double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static double norm2(const Vec3& a) { return dot(a, a); }
static double norm(const Vec3& a) { return std::sqrt(norm2(a)); }

static Vec3 normalize(const Vec3& a) {
    const double n = norm(a);
    if (n == 0.0) return {0.0, 0.0, 0.0};
    return a / n;
}

struct AABB {
    Vec3 mn, mx;

    AABB() { reset(); }

    void reset() {
        const double inf = std::numeric_limits<double>::infinity();
        mn = {inf, inf, inf};
        mx = {-inf, -inf, -inf};
    }

    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.mn);
        expand(b.mx);
    }

    AABB padded(double e) const {
        AABB r = *this;
        r.mn.x -= e; r.mn.y -= e; r.mn.z -= e;
        r.mx.x += e; r.mx.y += e; r.mx.z += e;
        return r;
    }

    bool overlaps(const AABB& b, double e = 0.0) const {
        return !(mx.x < b.mn.x - e || mn.x > b.mx.x + e ||
                 mx.y < b.mn.y - e || mn.y > b.mx.y + e ||
                 mx.z < b.mn.z - e || mn.z > b.mx.z + e);
    }

    bool contains(const Vec3& p, double e = 0.0) const {
        return p.x >= mn.x - e && p.x <= mx.x + e &&
               p.y >= mn.y - e && p.y <= mx.y + e &&
               p.z >= mn.z - e && p.z <= mx.z + e;
    }

    int longestAxis() const {
        const Vec3 d = mx - mn;
        if (d.x >= d.y && d.x >= d.z) return 0;
        if (d.y >= d.z) return 1;
        return 2;
    }
};

struct Face {
    int a = 0, b = 0, c = 0;
};

struct Mesh {
    vector<Vec3> v;
    vector<Face> f;
    vector<Vec3> normal;
    vector<AABB> faceBox;
    vector<Vec3> centroid;
    AABB box;
};

struct Tolerances {
    double geom = 1e-10;
    double split = 1e-9;
    double weld = 1e-8;
    double area = 1e-20;
};

static Tolerances tol;

static double signedVolume(const Mesh& m) {
    long double sum = 0.0L;
    for (const Face& f : m.f) {
        const Vec3& a = m.v[f.a];
        const Vec3& b = m.v[f.b];
        const Vec3& c = m.v[f.c];
        sum += static_cast<long double>(dot(a, cross(b, c)));
    }
    return static_cast<double>(sum / 6.0L);
}

static void orientOutward(Mesh& m) {
    if (signedVolume(m) < 0.0) {
        for (Face& f : m.f) std::swap(f.b, f.c);
    }
}

static void prepareMesh(Mesh& m) {
    orientOutward(m);
    m.box.reset();
    for (const Vec3& p : m.v) m.box.expand(p);

    m.normal.resize(m.f.size());
    m.faceBox.resize(m.f.size());
    m.centroid.resize(m.f.size());

    for (size_t i = 0; i < m.f.size(); ++i) {
        const Face& f = m.f[i];
        const Vec3& a = m.v[f.a];
        const Vec3& b = m.v[f.b];
        const Vec3& c = m.v[f.c];
        m.normal[i] = normalize(cross(b - a, c - a));
        AABB fb;
        fb.expand(a); fb.expand(b); fb.expand(c);
        m.faceBox[i] = fb;
        m.centroid[i] = (a + b + c) / 3.0;
    }
}

struct BVHNode {
    AABB box;
    int left = -1, right = -1;
    int start = 0, count = 0;
    bool leaf = false;
};

class BVH {
public:
    const Mesh* mesh = nullptr;
    vector<int> indices;
    vector<BVHNode> nodes;

    void build(const Mesh& m) {
        mesh = &m;
        indices.resize(m.f.size());
        std::iota(indices.begin(), indices.end(), 0);
        nodes.clear();
        nodes.reserve(m.f.empty() ? 0 : (m.f.size() * 2 / 8 + 8));
        if (!indices.empty()) buildRange(0, static_cast<int>(indices.size()));
    }

    template <class Fn>
    void queryAABB(const AABB& q, Fn&& fn) const {
        if (nodes.empty()) return;
        vector<int> stack;
        stack.reserve(64);
        stack.push_back(0);
        while (!stack.empty()) {
            const int ni = stack.back();
            stack.pop_back();
            const BVHNode& n = nodes[ni];
            if (!n.box.overlaps(q, tol.split)) continue;
            if (n.leaf) {
                for (int i = 0; i < n.count; ++i) {
                    const int fi = indices[n.start + i];
                    if (mesh->faceBox[fi].overlaps(q, tol.split)) fn(fi);
                }
            } else {
                stack.push_back(n.left);
                stack.push_back(n.right);
            }
        }
    }

    template <class Fn>
    void queryPoint(const Vec3& p, Fn&& fn) const {
        if (nodes.empty()) return;
        vector<int> stack;
        stack.reserve(64);
        stack.push_back(0);
        while (!stack.empty()) {
            const int ni = stack.back();
            stack.pop_back();
            const BVHNode& n = nodes[ni];
            if (!n.box.contains(p, tol.split)) continue;
            if (n.leaf) {
                for (int i = 0; i < n.count; ++i) {
                    const int fi = indices[n.start + i];
                    if (mesh->faceBox[fi].contains(p, tol.split)) fn(fi);
                }
            } else {
                stack.push_back(n.left);
                stack.push_back(n.right);
            }
        }
    }

    template <class Fn>
    void queryRay(const Vec3& origin, const Vec3& dir, Fn&& fn) const {
        if (nodes.empty()) return;
        vector<int> stack;
        stack.reserve(64);
        stack.push_back(0);
        while (!stack.empty()) {
            const int ni = stack.back();
            stack.pop_back();
            const BVHNode& n = nodes[ni];
            if (!rayAABB(origin, dir, n.box.padded(tol.split))) continue;
            if (n.leaf) {
                for (int i = 0; i < n.count; ++i) {
                    const int fi = indices[n.start + i];
                    if (rayAABB(origin, dir, mesh->faceBox[fi].padded(tol.split))) fn(fi);
                }
            } else {
                stack.push_back(n.left);
                stack.push_back(n.right);
            }
        }
    }

private:
    static bool rayAABB(const Vec3& o, const Vec3& d, const AABB& b) {
        double tmin = 0.0;
        double tmax = std::numeric_limits<double>::infinity();
        for (int ax = 0; ax < 3; ++ax) {
            const double ov = o[ax];
            const double dv = d[ax];
            if (std::abs(dv) < 1e-30) {
                if (ov < b.mn[ax] || ov > b.mx[ax]) return false;
            } else {
                double t1 = (b.mn[ax] - ov) / dv;
                double t2 = (b.mx[ax] - ov) / dv;
                if (t1 > t2) std::swap(t1, t2);
                tmin = std::max(tmin, t1);
                tmax = std::min(tmax, t2);
                if (tmax < tmin) return false;
            }
        }
        return tmax >= 0.0;
    }

    int buildRange(int start, int end) {
        const int nodeIndex = static_cast<int>(nodes.size());
        nodes.push_back(BVHNode{});

        for (int i = start; i < end; ++i) nodes[nodeIndex].box.expand(mesh->faceBox[indices[i]]);
        const int count = end - start;
        constexpr int leafSize = 12;
        if (count <= leafSize) {
            nodes[nodeIndex].leaf = true;
            nodes[nodeIndex].start = start;
            nodes[nodeIndex].count = count;
            return nodeIndex;
        }

        AABB cb;
        for (int i = start; i < end; ++i) cb.expand(mesh->centroid[indices[i]]);
        int axis = cb.longestAxis();
        if (cb.mx[axis] - cb.mn[axis] < tol.geom) axis = nodes[nodeIndex].box.longestAxis();

        const int mid = start + count / 2;
        std::nth_element(
            indices.begin() + start,
            indices.begin() + mid,
            indices.begin() + end,
            [&](int lhs, int rhs) {
                return mesh->centroid[lhs][axis] < mesh->centroid[rhs][axis];
            });

        const int left = buildRange(start, mid);
        const int right = buildRange(mid, end);
        nodes[nodeIndex].left = left;
        nodes[nodeIndex].right = right;
        return nodeIndex;
    }
};

struct Vec2 {
    double x = 0.0, y = 0.0;
};

static Vec2 project2D(const Vec3& p, int dropAxis) {
    if (dropAxis == 0) return {p.y, p.z};
    if (dropAxis == 1) return {p.x, p.z};
    return {p.x, p.y};
}

static double orient2D(const Vec2& a, const Vec2& b, const Vec2& c) {
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}

static bool pointInTri2D(const Vec2& p, const array<Vec2, 3>& t) {
    const double o0 = orient2D(t[0], t[1], p);
    const double o1 = orient2D(t[1], t[2], p);
    const double o2 = orient2D(t[2], t[0], p);
    const double e = tol.split;
    const bool hasNeg = (o0 < -e) || (o1 < -e) || (o2 < -e);
    const bool hasPos = (o0 > e) || (o1 > e) || (o2 > e);
    return !(hasNeg && hasPos);
}

static bool onSegment2D(const Vec2& a, const Vec2& b, const Vec2& p) {
    return std::abs(orient2D(a, b, p)) <= tol.split &&
           p.x >= std::min(a.x, b.x) - tol.split && p.x <= std::max(a.x, b.x) + tol.split &&
           p.y >= std::min(a.y, b.y) - tol.split && p.y <= std::max(a.y, b.y) + tol.split;
}

static bool segmentsIntersect2D(const Vec2& a, const Vec2& b, const Vec2& c, const Vec2& d) {
    const double o1 = orient2D(a, b, c);
    const double o2 = orient2D(a, b, d);
    const double o3 = orient2D(c, d, a);
    const double o4 = orient2D(c, d, b);
    const double e = tol.split;
    if (((o1 > e && o2 < -e) || (o1 < -e && o2 > e)) &&
        ((o3 > e && o4 < -e) || (o3 < -e && o4 > e))) return true;
    return onSegment2D(a, b, c) || onSegment2D(a, b, d) ||
           onSegment2D(c, d, a) || onSegment2D(c, d, b);
}

static bool coplanarTrianglesOverlap(const array<Vec3, 3>& a, const Vec3& normal, const array<Vec3, 3>& b) {
    const Vec3 an = {std::abs(normal.x), std::abs(normal.y), std::abs(normal.z)};
    int drop = 2;
    if (an.x >= an.y && an.x >= an.z) drop = 0;
    else if (an.y >= an.z) drop = 1;

    array<Vec2, 3> ta, tb;
    for (int i = 0; i < 3; ++i) {
        ta[i] = project2D(a[i], drop);
        tb[i] = project2D(b[i], drop);
    }

    for (int i = 0; i < 3; ++i) {
        if (pointInTri2D(ta[i], tb) || pointInTri2D(tb[i], ta)) return true;
    }
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < 3; ++j) {
            if (segmentsIntersect2D(ta[i], ta[(i + 1) % 3], tb[j], tb[(j + 1) % 3])) return true;
        }
    }
    return false;
}

enum class TriRelation {
    None,
    Intersect,
    Coplanar
};

static void projectAxis(const array<Vec3, 3>& t, const Vec3& axis, double& lo, double& hi) {
    lo = hi = dot(t[0], axis);
    for (int i = 1; i < 3; ++i) {
        const double p = dot(t[i], axis);
        lo = std::min(lo, p);
        hi = std::max(hi, p);
    }
}

static bool separatedOnAxis(const array<Vec3, 3>& a, const array<Vec3, 3>& b, const Vec3& axis) {
    const double l2 = norm2(axis);
    if (l2 <= tol.geom * tol.geom) return false;
    double amin, amax, bmin, bmax;
    projectAxis(a, axis, amin, amax);
    projectAxis(b, axis, bmin, bmax);
    const double e = tol.split * std::sqrt(l2) * 4.0;
    return amax < bmin - e || bmax < amin - e;
}

static TriRelation triangleRelation(
    const array<Vec3, 3>& a,
    const Vec3& na,
    const array<Vec3, 3>& b,
    const Vec3& nb) {

    if (norm2(na) == 0.0 || norm2(nb) == 0.0) return TriRelation::None;

    const double ndot = dot(na, nb);
    const bool parallel = std::abs(ndot) > 1.0 - 1e-8;
    if (parallel) {
        bool samePlane = true;
        for (int i = 0; i < 3; ++i) {
            if (std::abs(dot(na, b[i] - a[0])) > tol.split * 4.0) {
                samePlane = false;
                break;
            }
        }
        if (samePlane) {
            return coplanarTrianglesOverlap(a, na, b) ? TriRelation::Coplanar : TriRelation::None;
        }
    }

    if (separatedOnAxis(a, b, na) || separatedOnAxis(a, b, nb)) return TriRelation::None;
    const Vec3 ae[3] = {a[1] - a[0], a[2] - a[1], a[0] - a[2]};
    const Vec3 be[3] = {b[1] - b[0], b[2] - b[1], b[0] - b[2]};
    for (const Vec3& ea : ae) {
        for (const Vec3& eb : be) {
            if (separatedOnAxis(a, b, cross(ea, eb))) return TriRelation::None;
        }
    }
    return TriRelation::Intersect;
}

struct Plane {
    Vec3 n;
    double d = 0.0;
};

static bool samePlane(const Plane& a, const Plane& b) {
    const double dp = dot(a.n, b.n);
    if (dp > 1.0 - 1e-10) return std::abs(a.d - b.d) <= tol.split * 8.0;
    if (dp < -1.0 + 1e-10) return std::abs(a.d + b.d) <= tol.split * 8.0;
    return false;
}

static void addPlane(vector<Plane>& planes, Vec3 n, const Vec3& point) {
    const double len = norm(n);
    if (len <= tol.geom) return;
    n = n / len;
    Plane p{n, -dot(n, point)};
    for (const Plane& old : planes) {
        if (samePlane(old, p)) return;
    }
    planes.push_back(p);
}

static array<Vec3, 3> faceVerts(const Mesh& m, int fi) {
    const Face& f = m.f[fi];
    return {m.v[f.a], m.v[f.b], m.v[f.c]};
}

static double polygonArea2(const vector<Vec3>& p) {
    if (p.size() < 3) return 0.0;
    Vec3 acc{0.0, 0.0, 0.0};
    const Vec3& base = p[0];
    for (size_t i = 1; i + 1 < p.size(); ++i) {
        acc = acc + cross(p[i] - base, p[i + 1] - base);
    }
    return norm(acc);
}

static bool validPolygon(const vector<Vec3>& p) {
    return p.size() >= 3 && polygonArea2(p) > tol.area;
}

static void pushUnique(vector<Vec3>& p, const Vec3& v) {
    if (p.empty() || norm2(p.back() - v) > tol.weld * tol.weld) p.push_back(v);
}

static void closeClean(vector<Vec3>& p) {
    if (p.size() > 1 && norm2(p.front() - p.back()) <= tol.weld * tol.weld) p.pop_back();
}

static void splitPolygonByPlane(const vector<Vec3>& poly, const Plane& pl, vector<Vec3>& pos, vector<Vec3>& neg) {
    pos.clear();
    neg.clear();
    const size_t n = poly.size();
    if (n < 3) return;

    vector<double> dist(n);
    bool hasPos = false, hasNeg = false;
    for (size_t i = 0; i < n; ++i) {
        double d = dot(pl.n, poly[i]) + pl.d;
        if (std::abs(d) <= tol.split) d = 0.0;
        dist[i] = d;
        hasPos = hasPos || d > 0.0;
        hasNeg = hasNeg || d < 0.0;
    }

    if (!hasPos || !hasNeg) {
        if (hasNeg) neg = poly;
        else pos = poly;
        return;
    }

    for (size_t i = 0; i < n; ++i) {
        const size_t j = (i + 1) % n;
        const Vec3& a = poly[i];
        const Vec3& b = poly[j];
        const double da = dist[i];
        const double db = dist[j];

        if (da >= 0.0) pushUnique(pos, a);
        if (da <= 0.0) pushUnique(neg, a);

        if ((da > 0.0 && db < 0.0) || (da < 0.0 && db > 0.0)) {
            const double t = da / (da - db);
            const Vec3 p = a + (b - a) * t;
            pushUnique(pos, p);
            pushUnique(neg, p);
        }
    }

    closeClean(pos);
    closeClean(neg);
}

static bool pointOnTriangle(const Vec3& p, const array<Vec3, 3>& t, const Vec3& n) {
    if (std::abs(dot(n, p - t[0])) > tol.split * 4.0) return false;
    const Vec3 e0 = t[1] - t[0];
    const Vec3 e1 = t[2] - t[0];
    const Vec3 vp = p - t[0];
    const double d00 = dot(e0, e0);
    const double d01 = dot(e0, e1);
    const double d11 = dot(e1, e1);
    const double d20 = dot(vp, e0);
    const double d21 = dot(vp, e1);
    const double denom = d00 * d11 - d01 * d01;
    if (std::abs(denom) < tol.geom * tol.geom) return false;
    const double v = (d11 * d20 - d01 * d21) / denom;
    const double w = (d00 * d21 - d01 * d20) / denom;
    const double u = 1.0 - v - w;
    constexpr double be = 1e-8;
    return u >= -be && v >= -be && w >= -be && u <= 1.0 + be && v <= 1.0 + be && w <= 1.0 + be;
}

static bool rayTriangle(const Vec3& o, const Vec3& dir, const array<Vec3, 3>& t, double& hitT) {
    const Vec3 e1 = t[1] - t[0];
    const Vec3 e2 = t[2] - t[0];
    const Vec3 h = cross(dir, e2);
    const double det = dot(e1, h);
    if (std::abs(det) < 1e-18) return false;
    const double invDet = 1.0 / det;
    const Vec3 s = o - t[0];
    const double u = dot(s, h) * invDet;
    if (u < -1e-10 || u > 1.0 + 1e-10) return false;
    const Vec3 q = cross(s, e1);
    const double v = dot(dir, q) * invDet;
    if (v < -1e-10 || u + v > 1.0 + 1e-10) return false;
    const double tt = dot(e2, q) * invDet;
    if (tt <= tol.split * 8.0) return false;
    hitT = tt;
    return true;
}

enum class Location {
    Outside,
    Inside,
    OnSurface
};

class SolidQuery {
public:
    const Mesh& mesh;
    const BVH& bvh;

    SolidQuery(const Mesh& m, const BVH& b) : mesh(m), bvh(b) {}

    bool surfaceNormalAt(const Vec3& p, Vec3& outNormal) const {
        bool found = false;
        Vec3 sum{0.0, 0.0, 0.0};
        bvh.queryPoint(p, [&](int fi) {
            const array<Vec3, 3> t = faceVerts(mesh, fi);
            if (pointOnTriangle(p, t, mesh.normal[fi])) {
                sum = sum + mesh.normal[fi];
                found = true;
            }
        });
        if (found) {
            outNormal = normalize(sum);
            if (norm2(outNormal) == 0.0) outNormal = sum;
        }
        return found;
    }

    Location locate(const Vec3& p) const {
        if (!mesh.box.contains(p, tol.split * 2.0)) return Location::Outside;

        Vec3 unused;
        if (surfaceNormalAt(p, unused)) return Location::OnSurface;

        static const Vec3 dir = normalize(Vec3{1.0, 0.3472963553338607, 0.5320888862379560});
        vector<double> hits;
        bvh.queryRay(p, dir, [&](int fi) {
            double t = 0.0;
            const array<Vec3, 3> tri = faceVerts(mesh, fi);
            if (rayTriangle(p, dir, tri, t)) hits.push_back(t);
        });

        if (hits.empty()) return Location::Outside;
        std::sort(hits.begin(), hits.end());
        int uniqueHits = 0;
        double last = -std::numeric_limits<double>::infinity();
        for (double h : hits) {
            const double e = std::max(tol.split * 16.0, std::abs(h) * 1e-10);
            if (uniqueHits == 0 || std::abs(h - last) > e) {
                ++uniqueHits;
                last = h;
            }
        }
        return (uniqueHits & 1) ? Location::Inside : Location::Outside;
    }
};

struct QuantKey {
    long long x = 0, y = 0, z = 0;
    bool operator==(const QuantKey& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct QuantHash {
    size_t operator()(const QuantKey& k) const {
        uint64_t x = static_cast<uint64_t>(k.x) * 0x9E3779B185EBCA87ull;
        uint64_t y = static_cast<uint64_t>(k.y) * 0xC2B2AE3D27D4EB4Full;
        uint64_t z = static_cast<uint64_t>(k.z) * 0x165667B19E3779F9ull;
        uint64_t h = x ^ (y + 0x9e3779b97f4a7c15ull + (x << 6) + (x >> 2));
        h ^= z + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        return static_cast<size_t>(h);
    }
};

static long long quantize(double x) {
    const long double q = static_cast<long double>(x) / static_cast<long double>(tol.weld);
    const long double r = std::floor(q + (q >= 0.0L ? 0.5L : -0.5L));
    const long double hi = static_cast<long double>(std::numeric_limits<long long>::max());
    const long double lo = static_cast<long double>(std::numeric_limits<long long>::min());
    if (r > hi) return std::numeric_limits<long long>::max();
    if (r < lo) return std::numeric_limits<long long>::min();
    return static_cast<long long>(r);
}

struct FaceKey {
    int a = 0, b = 0, c = 0;
    bool operator==(const FaceKey& o) const { return a == o.a && b == o.b && c == o.c; }
};

struct FaceKeyHash {
    size_t operator()(const FaceKey& k) const {
        uint64_t h = static_cast<uint64_t>(k.a) * 73856093ull;
        h ^= static_cast<uint64_t>(k.b) * 19349663ull;
        h ^= static_cast<uint64_t>(k.c) * 83492791ull;
        return static_cast<size_t>(h);
    }
};

class OutputBuilder {
public:
    vector<Vec3> verts;
    vector<Face> faces;

    int vertexId(const Vec3& p) {
        const QuantKey base{quantize(p.x), quantize(p.y), quantize(p.z)};
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    const QuantKey k{base.x + dx, base.y + dy, base.z + dz};
                    auto it = buckets.find(k);
                    if (it == buckets.end()) continue;
                    for (int id : it->second) {
                        if (norm2(verts[id] - p) <= tol.weld * tol.weld) return id;
                    }
                }
            }
        }
        const int id = static_cast<int>(verts.size());
        verts.push_back(p);
        buckets[base].push_back(id);
        return id;
    }

    void addTriangle(const Vec3& a, const Vec3& b, const Vec3& c) {
        if (norm(cross(b - a, c - a)) <= tol.area) return;
        const int ia = vertexId(a);
        const int ib = vertexId(b);
        const int ic = vertexId(c);
        if (ia == ib || ib == ic || ia == ic) return;
        array<int, 3> s{ia, ib, ic};
        std::sort(s.begin(), s.end());
        const FaceKey key{s[0], s[1], s[2]};
        if (!faceSet.insert(key).second) return;
        faces.push_back({ia, ib, ic});
    }

    void compact() {
        vector<int> remap(verts.size(), -1);
        vector<Vec3> nv;
        nv.reserve(verts.size());
        for (const Face& f : faces) {
            const int ids[3] = {f.a, f.b, f.c};
            for (int old : ids) {
                if (remap[old] < 0) {
                    remap[old] = static_cast<int>(nv.size());
                    nv.push_back(verts[old]);
                }
            }
        }
        for (Face& f : faces) {
            f.a = remap[f.a];
            f.b = remap[f.b];
            f.c = remap[f.c];
        }
        verts.swap(nv);
        buckets.clear();
    }

private:
    std::unordered_map<QuantKey, vector<int>, QuantHash> buckets;
    std::unordered_set<FaceKey, FaceKeyHash> faceSet;
};

static bool hasSurfaceInteraction(const Mesh& a, const BVH& bvhB, const Mesh& b) {
    for (int i = 0; i < static_cast<int>(a.f.size()); ++i) {
        const array<Vec3, 3> ta = faceVerts(a, i);
        bool found = false;
        bvhB.queryAABB(a.faceBox[i].padded(tol.split), [&](int j) {
            if (found) return;
            const TriRelation r = triangleRelation(ta, a.normal[i], faceVerts(b, j), b.normal[j]);
            if (r != TriRelation::None) found = true;
        });
        if (found) return true;
    }
    return false;
}

static Location locateWithOffset(const SolidQuery& q, const Vec3& p, const Vec3& n) {
    const double steps[4] = {64.0, 512.0, 4096.0, 32768.0};
    for (double s : steps) {
        const Location loc = q.locate(p + n * (tol.geom * s));
        if (loc != Location::OnSurface) return loc;
    }
    return Location::Outside;
}

static bool keepOnSurfaceTriangle(
    const Vec3& centroid,
    const Vec3& outwardNormal,
    bool sourceIsA,
    const SolidQuery& targetQuery) {

    const Location outsideSide = locateWithOffset(targetQuery, centroid, outwardNormal);
    if (outsideSide == Location::Inside) return false;

    Vec3 otherNormal;
    if (targetQuery.surfaceNormalAt(centroid, otherNormal)) {
        const double nd = dot(outwardNormal, otherNormal);
        if (nd < -0.5) return false;
        if (!sourceIsA && nd > 0.5) return false;
    }
    return true;
}

static void collectSplitPlanesForFace(
    const Mesh& source,
    int sourceFace,
    const Mesh& target,
    const BVH& targetBVH,
    vector<Plane>& planes) {

    const array<Vec3, 3> srcTri = faceVerts(source, sourceFace);
    const Vec3 srcN = source.normal[sourceFace];

    targetBVH.queryAABB(source.faceBox[sourceFace].padded(tol.split), [&](int targetFace) {
        const array<Vec3, 3> dstTri = faceVerts(target, targetFace);
        const TriRelation rel = triangleRelation(srcTri, srcN, dstTri, target.normal[targetFace]);
        if (rel == TriRelation::Intersect) {
            addPlane(planes, target.normal[targetFace], dstTri[0]);
            for (int e = 0; e < 3; ++e) {
                const Vec3 p0 = dstTri[e];
                const Vec3 p1 = dstTri[(e + 1) % 3];
                addPlane(planes, cross(p1 - p0, target.normal[targetFace]), p0);
            }
        } else if (rel == TriRelation::Coplanar) {
            for (int e = 0; e < 3; ++e) {
                const Vec3 p0 = dstTri[e];
                const Vec3 p1 = dstTri[(e + 1) % 3];
                addPlane(planes, cross(srcN, p1 - p0), p0);
            }
        }
    });
}

static void addClassifiedTriangle(
    const Vec3& a,
    const Vec3& b,
    const Vec3& c,
    const Vec3& sourceNormal,
    bool sourceIsA,
    const SolidQuery& targetQuery,
    OutputBuilder& out) {

    if (norm(cross(b - a, c - a)) <= tol.area) return;
    const Vec3 centroid = (a + b + c) / 3.0;
    const Location loc = targetQuery.locate(centroid);
    if (loc == Location::Inside) return;
    if (loc == Location::OnSurface &&
        !keepOnSurfaceTriangle(centroid, sourceNormal, sourceIsA, targetQuery)) {
        return;
    }
    out.addTriangle(a, b, c);
}

static void processFace(
    const Mesh& source,
    int sourceFace,
    const Mesh& target,
    const BVH& targetBVH,
    const SolidQuery& targetQuery,
    bool sourceIsA,
    OutputBuilder& out) {

    if (!source.faceBox[sourceFace].overlaps(target.box, tol.split)) {
        const array<Vec3, 3> t = faceVerts(source, sourceFace);
        out.addTriangle(t[0], t[1], t[2]);
        return;
    }

    vector<Plane> planes;
    planes.reserve(8);
    collectSplitPlanesForFace(source, sourceFace, target, targetBVH, planes);

    vector<vector<Vec3>> polygons;
    polygons.push_back(vector<Vec3>{source.v[source.f[sourceFace].a], source.v[source.f[sourceFace].b], source.v[source.f[sourceFace].c]});

    vector<Vec3> pos, neg;
    for (const Plane& pl : planes) {
        vector<vector<Vec3>> next;
        next.reserve(polygons.size() + 2);
        for (const vector<Vec3>& poly : polygons) {
            splitPolygonByPlane(poly, pl, pos, neg);
            if (validPolygon(pos)) next.push_back(pos);
            if (validPolygon(neg)) next.push_back(neg);
        }
        polygons.swap(next);
        if (polygons.empty()) return;
    }

    const Vec3 srcN = source.normal[sourceFace];
    for (const vector<Vec3>& poly : polygons) {
        if (poly.size() < 3) continue;
        const Vec3 base = poly[0];
        for (size_t i = 1; i + 1 < poly.size(); ++i) {
            Vec3 b = poly[i];
            Vec3 c = poly[i + 1];
            if (dot(cross(b - base, c - base), srcN) < 0.0) std::swap(b, c);
            addClassifiedTriangle(base, b, c, srcN, sourceIsA, targetQuery, out);
        }
    }
}

static void outputMeshDirect(const Mesh& m) {
    std::cout << m.v.size() << ' ' << m.f.size() << '\n';
    std::cout << std::setprecision(17);
    for (const Vec3& p : m.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const Face& f : m.f) std::cout << f.a << ' ' << f.b << ' ' << f.c << '\n';
}

static void outputConcat(const Mesh& a, const Mesh& b) {
    std::cout << (a.v.size() + b.v.size()) << ' ' << (a.f.size() + b.f.size()) << '\n';
    std::cout << std::setprecision(17);
    for (const Vec3& p : a.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const Vec3& p : b.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const Face& f : a.f) std::cout << f.a << ' ' << f.b << ' ' << f.c << '\n';
    const int off = static_cast<int>(a.v.size());
    for (const Face& f : b.f) std::cout << (f.a + off) << ' ' << (f.b + off) << ' ' << (f.c + off) << '\n';
}

static void outputBuilder(OutputBuilder& out) {
    out.compact();
    std::cout << out.verts.size() << ' ' << out.faces.size() << '\n';
    std::cout << std::setprecision(17);
    for (const Vec3& p : out.verts) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const Face& f : out.faces) std::cout << f.a << ' ' << f.b << ' ' << f.c << '\n';
}

static void configureTolerances(const Mesh& a, const Mesh& b) {
    AABB all;
    for (const Vec3& p : a.v) all.expand(p);
    for (const Vec3& p : b.v) all.expand(p);
    const Vec3 d = all.mx - all.mn;
    const double diag = std::max(1.0, norm(d));
    double maxAbs = 1.0;
    for (int ax = 0; ax < 3; ++ax) {
        maxAbs = std::max(maxAbs, std::abs(all.mn[ax]));
        maxAbs = std::max(maxAbs, std::abs(all.mx[ax]));
    }
    const double fp = 64.0 * std::numeric_limits<double>::epsilon() * maxAbs;
    tol.geom = std::max({1e-10 * diag, fp, 1e-12});
    tol.split = tol.geom * 16.0;
    tol.weld = tol.geom * 64.0;
    tol.area = tol.geom * tol.geom * 16.0;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA = 0, FA = 0, VB = 0, FB = 0;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA);
    A.f.resize(FA);
    B.v.resize(VB);
    B.f.resize(FB);

    for (Vec3& p : A.v) std::cin >> p.x >> p.y >> p.z;
    for (Face& f : A.f) std::cin >> f.a >> f.b >> f.c;
    for (Vec3& p : B.v) std::cin >> p.x >> p.y >> p.z;
    for (Face& f : B.f) std::cin >> f.a >> f.b >> f.c;

    configureTolerances(A, B);
    prepareMesh(A);
    prepareMesh(B);

    if (A.f.empty()) {
        outputMeshDirect(B);
        return 0;
    }
    if (B.f.empty()) {
        outputMeshDirect(A);
        return 0;
    }

    if (!A.box.overlaps(B.box, tol.split)) {
        outputConcat(A, B);
        return 0;
    }

    BVH bvhA, bvhB;
    bvhA.build(A);
    bvhB.build(B);
    SolidQuery queryA(A, bvhA);
    SolidQuery queryB(B, bvhB);

    const bool surfacesInteract = hasSurfaceInteraction(A, bvhB, B);
    if (!surfacesInteract) {
        const Location aInB = queryB.locate(A.centroid.empty() ? A.v[0] : A.centroid[0]);
        const Location bInA = queryA.locate(B.centroid.empty() ? B.v[0] : B.centroid[0]);
        if (aInB == Location::Inside || aInB == Location::OnSurface) {
            outputMeshDirect(B);
        } else if (bInA == Location::Inside || bInA == Location::OnSurface) {
            outputMeshDirect(A);
        } else {
            outputConcat(A, B);
        }
        return 0;
    }

    OutputBuilder out;
    out.verts.reserve(A.v.size() + B.v.size());
    out.faces.reserve(A.f.size() + B.f.size());

    for (int i = 0; i < static_cast<int>(A.f.size()); ++i) {
        processFace(A, i, B, bvhB, queryB, true, out);
    }
    for (int i = 0; i < static_cast<int>(B.f.size()); ++i) {
        processFace(B, i, A, bvhA, queryA, false, out);
    }

    outputBuilder(out);
    return 0;
}
