use std::collections::{HashSet, VecDeque};
use std::io::{self, Read, Write};
use std::time::{Duration, Instant};

const NONE: usize = usize::MAX;

#[derive(Clone)]
struct Grid {
    n: usize,
    start: usize,
    coords: Vec<(usize, usize)>,
    adj: Vec<[usize; 4]>,
    deg: Vec<u8>,
    colors: [usize; 2],
}

impl Grid {
    fn adjacent(&self, a: usize, b: usize) -> bool {
        self.adj[a].iter().any(|&x| x == b)
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let Some(w) = it.next().and_then(|s| s.parse::<usize>().ok()) else {
        println!("0");
        return;
    };
    let Some(h) = it.next().and_then(|s| s.parse::<usize>().ok()) else {
        println!("0");
        return;
    };
    let Some(obstacle_count) = it.next().and_then(|s| s.parse::<usize>().ok()) else {
        println!("0");
        return;
    };
    let require_cycle = it
        .next()
        .and_then(|s| s.parse::<u8>().ok())
        .map_or(false, |v| v != 0);

    let mut obstacles = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let Some(x) = it.next().and_then(|s| s.parse::<usize>().ok()) else {
            println!("0");
            return;
        };
        let Some(y) = it.next().and_then(|s| s.parse::<usize>().ok()) else {
            println!("0");
            return;
        };
        obstacles.push((x, y));
    }

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    if obstacle_count == 0 {
        output_empty_rectangle(w, h, require_cycle);
        return;
    }

    let Some(grid) = build_grid(w, h, &obstacles) else {
        println!("0");
        return;
    };

    if grid.n == 0 || grid.start == NONE {
        println!("0");
        return;
    }

    if grid.n == 1 {
        if require_cycle {
            println!("0");
        } else {
            output_ids(&grid, &[grid.start]);
        }
        return;
    }

    if !basic_necessary_checks(&grid, require_cycle) {
        println!("0");
        return;
    }

    if let Some(path) = solve_degree_two(&grid, require_cycle) {
        output_ids(&grid, &path);
        return;
    }

    if grid.deg.iter().all(|&d| d <= 2) {
        println!("0");
        return;
    }

    match solve_small_exact(&grid, require_cycle) {
        ExactOutcome::Found(path) => {
            output_ids(&grid, &path);
            return;
        }
        ExactOutcome::Impossible => {
            println!("0");
            return;
        }
        ExactOutcome::Unknown => {}
    }

    let deadline = Instant::now() + Duration::from_secs(285);
    let mut search = Search::new(&grid, require_cycle, deadline);
    if let Some(path) = search.solve() {
        if validate_path(&grid, &path, require_cycle) {
            output_ids(&grid, &path);
            return;
        }
    }

    println!("0");
}

fn build_grid(w: usize, h: usize, obstacles: &[(usize, usize)]) -> Option<Grid> {
    let total = w.checked_mul(h)?;
    let mut blocked = vec![false; total];
    for &(x, y) in obstacles {
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    let mut id_of = vec![NONE; total];
    let mut coords = Vec::new();
    let mut colors = [0usize; 2];
    for y in 0..h {
        for x in 0..w {
            let p = y * w + x;
            if !blocked[p] {
                let id = coords.len();
                id_of[p] = id;
                coords.push((x, y));
                colors[(x + y) & 1] += 1;
            }
        }
    }

    let n = coords.len();
    let start = if blocked[0] { NONE } else { id_of[0] };
    let mut adj = vec![[NONE; 4]; n];
    let mut deg = vec![0u8; n];
    let dirs = [(1isize, 0isize), (-1, 0), (0, 1), (0, -1)];
    for id in 0..n {
        let (x, y) = coords[id];
        for (dx, dy) in dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && ny >= 0 && (nx as usize) < w && (ny as usize) < h {
                let nid = id_of[ny as usize * w + nx as usize];
                if nid != NONE {
                    let k = deg[id] as usize;
                    adj[id][k] = nid;
                    deg[id] += 1;
                }
            }
        }
    }

    Some(Grid {
        n,
        start,
        coords,
        adj,
        deg,
        colors,
    })
}

fn output_empty_rectangle(w: usize, h: usize, require_cycle: bool) {
    let Some(total) = w.checked_mul(h) else {
        println!("0");
        return;
    };
    if total == 0 {
        println!("0");
        return;
    }

    if !require_cycle {
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "{total}").unwrap();
        for y in 0..h {
            if y & 1 == 0 {
                for x in 0..w {
                    writeln!(out, "{x} {y}").unwrap();
                }
            } else {
                for x in (0..w).rev() {
                    writeln!(out, "{x} {y}").unwrap();
                }
            }
        }
        return;
    }

    if total == 1 {
        println!("0");
        return;
    }

    if total == 2 {
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "2").unwrap();
        writeln!(out, "0 0").unwrap();
        if w == 2 {
            writeln!(out, "1 0").unwrap();
        } else {
            writeln!(out, "0 1").unwrap();
        }
        return;
    }

    if w == 1 || h == 1 || (total & 1) == 1 {
        println!("0");
        return;
    }

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{total}").unwrap();

    if w & 1 == 0 {
        for y in 0..h {
            writeln!(out, "0 {y}").unwrap();
        }
        for x in 1..w {
            if x & 1 == 1 {
                for y in (1..h).rev() {
                    writeln!(out, "{x} {y}").unwrap();
                }
            } else {
                for y in 1..h {
                    writeln!(out, "{x} {y}").unwrap();
                }
            }
        }
        for x in (1..w).rev() {
            writeln!(out, "{x} 0").unwrap();
        }
    } else {
        for x in 0..w {
            writeln!(out, "{x} 0").unwrap();
        }
        for y in 1..h {
            if y & 1 == 1 {
                for x in (1..w).rev() {
                    writeln!(out, "{x} {y}").unwrap();
                }
            } else {
                for x in 1..w {
                    writeln!(out, "{x} {y}").unwrap();
                }
            }
        }
        for y in (1..h).rev() {
            writeln!(out, "0 {y}").unwrap();
        }
    }
}

fn output_ids(grid: &Grid, path: &[usize]) {
    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", path.len()).unwrap();
    for &id in path {
        let (x, y) = grid.coords[id];
        writeln!(out, "{x} {y}").unwrap();
    }
}

fn basic_necessary_checks(grid: &Grid, require_cycle: bool) -> bool {
    if !is_connected(grid) {
        return false;
    }

    let start_color = (grid.coords[grid.start].0 + grid.coords[grid.start].1) & 1;
    if require_cycle {
        if grid.colors[0] != grid.colors[1] {
            return false;
        }
    } else if grid.n & 1 == 0 {
        if grid.colors[0] != grid.colors[1] {
            return false;
        }
    } else if grid.colors[start_color] != grid.colors[1 - start_color] + 1 {
        return false;
    }

    if grid.deg[grid.start] == 0 {
        return false;
    }

    if require_cycle {
        if grid.n > 2 && grid.deg.iter().any(|&d| d < 2) {
            return false;
        }
    } else {
        let mut non_start_leaves = 0usize;
        for v in 0..grid.n {
            if grid.deg[v] == 0 {
                return false;
            }
            if v != grid.start && grid.deg[v] == 1 {
                non_start_leaves += 1;
                if non_start_leaves > 1 {
                    return false;
                }
            }
        }
    }

    articulation_necessary_check(grid, require_cycle)
}

fn is_connected(grid: &Grid) -> bool {
    let mut seen = vec![false; grid.n];
    let mut q = VecDeque::new();
    seen[grid.start] = true;
    q.push_back(grid.start);
    let mut count = 0usize;

    while let Some(v) = q.pop_front() {
        count += 1;
        for &u in &grid.adj[v] {
            if u != NONE && !seen[u] {
                seen[u] = true;
                q.push_back(u);
            }
        }
    }

    count == grid.n
}

fn articulation_necessary_check(grid: &Grid, require_cycle: bool) -> bool {
    if grid.n > 10_000 {
        return true;
    }

    for removed in 0..grid.n {
        let limit = if require_cycle || removed == grid.start {
            1
        } else {
            2
        };
        if components_without_vertex(grid, removed, limit) > limit {
            return false;
        }
    }
    true
}

fn components_without_vertex(grid: &Grid, removed: usize, limit: usize) -> usize {
    if grid.n <= 1 {
        return 0;
    }

    let mut seen = vec![false; grid.n];
    seen[removed] = true;
    let mut q = VecDeque::new();
    let mut components = 0usize;

    for s in 0..grid.n {
        if seen[s] {
            continue;
        }
        components += 1;
        if components > limit {
            return components;
        }
        seen[s] = true;
        q.push_back(s);
        while let Some(v) = q.pop_front() {
            for &u in &grid.adj[v] {
                if u != NONE && !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
    }

    components
}

fn solve_degree_two(grid: &Grid, require_cycle: bool) -> Option<Vec<usize>> {
    if grid.deg.iter().any(|&d| d > 2) {
        return None;
    }

    if grid.n == 1 {
        return (!require_cycle).then(|| vec![grid.start]);
    }

    if require_cycle {
        if grid.n == 2 {
            let other = grid.adj[grid.start].iter().copied().find(|&u| u != NONE)?;
            return Some(vec![grid.start, other]);
        }
        if grid.deg.iter().any(|&d| d != 2) {
            return None;
        }
        return Some(walk_cycle_from_start(grid));
    }

    let leaves: Vec<usize> = (0..grid.n).filter(|&v| grid.deg[v] == 1).collect();
    if leaves.is_empty() {
        return Some(walk_cycle_from_start(grid));
    }
    if leaves.len() == 2 && leaves.contains(&grid.start) {
        return Some(walk_path_graph_from_start(grid));
    }
    None
}

fn walk_path_graph_from_start(grid: &Grid) -> Vec<usize> {
    let mut path = Vec::with_capacity(grid.n);
    let mut prev = NONE;
    let mut cur = grid.start;
    loop {
        path.push(cur);
        if path.len() == grid.n {
            break;
        }
        let mut next = NONE;
        for &u in &grid.adj[cur] {
            if u != NONE && u != prev {
                next = u;
                break;
            }
        }
        prev = cur;
        cur = next;
    }
    path
}

fn walk_cycle_from_start(grid: &Grid) -> Vec<usize> {
    let mut path = Vec::with_capacity(grid.n);
    let mut prev = NONE;
    let mut cur = grid.start;
    loop {
        path.push(cur);
        if path.len() == grid.n {
            break;
        }
        let mut next = NONE;
        for &u in &grid.adj[cur] {
            if u != NONE && u != prev {
                next = u;
                break;
            }
        }
        prev = cur;
        cur = next;
    }
    path
}

enum ExactOutcome {
    Found(Vec<usize>),
    Impossible,
    Unknown,
}

fn solve_small_exact(grid: &Grid, require_cycle: bool) -> ExactOutcome {
    if grid.n > 26 {
        return ExactOutcome::Unknown;
    }

    let deadline = Instant::now() + Duration::from_secs(20);
    let mut solver = SmallExact {
        grid,
        require_cycle,
        target: (1u64 << grid.n) - 1,
        failed: HashSet::new(),
        path: vec![grid.start],
        deadline,
        nodes: 0,
        aborted: false,
    };

    let start_mask = 1u64 << grid.start;
    if solver.rec(grid.start, start_mask) {
        ExactOutcome::Found(solver.path)
    } else if solver.aborted {
        ExactOutcome::Unknown
    } else {
        ExactOutcome::Impossible
    }
}

struct SmallExact<'a> {
    grid: &'a Grid,
    require_cycle: bool,
    target: u64,
    failed: HashSet<u64>,
    path: Vec<usize>,
    deadline: Instant,
    nodes: u64,
    aborted: bool,
}

impl<'a> SmallExact<'a> {
    fn rec(&mut self, cur: usize, mask: u64) -> bool {
        self.nodes += 1;
        if (self.nodes & 8191) == 0 {
            if self.nodes > 12_000_000 || Instant::now() >= self.deadline {
                self.aborted = true;
                return false;
            }
        }

        if mask == self.target {
            return !self.require_cycle || self.grid.adjacent(cur, self.grid.start);
        }

        let key = (mask << 5) ^ cur as u64;
        if self.failed.contains(&key) {
            return false;
        }

        let remaining = self.grid.n - mask.count_ones() as usize;
        let mut cands = [NONE; 4];
        let mut len = 0usize;

        let sole_start_neighbor = if self.require_cycle && remaining > 1 {
            self.only_unvisited_neighbor_of_start(mask)
        } else {
            None
        };

        for &u in &self.grid.adj[cur] {
            if u == NONE || ((mask >> u) & 1) != 0 {
                continue;
            }
            if self.require_cycle {
                if remaining == 1 && !self.grid.adjacent(u, self.grid.start) {
                    continue;
                }
                if remaining > 1 && sole_start_neighbor == Some(u) {
                    continue;
                }
            }
            cands[len] = u;
            len += 1;
        }

        self.sort_small_candidates(&mut cands, len, mask, remaining);

        for i in 0..len {
            let next = cands[i];
            self.path.push(next);
            if self.rec(next, mask | (1u64 << next)) {
                return true;
            }
            self.path.pop();
            if self.aborted {
                return false;
            }
        }

        self.failed.insert(key);
        false
    }

    fn only_unvisited_neighbor_of_start(&self, mask: u64) -> Option<usize> {
        let mut result = None;
        for &u in &self.grid.adj[self.grid.start] {
            if u != NONE && ((mask >> u) & 1) == 0 {
                if result.is_some() {
                    return None;
                }
                result = Some(u);
            }
        }
        result
    }

    fn sort_small_candidates(
        &self,
        cands: &mut [usize; 4],
        len: usize,
        mask: u64,
        remaining: usize,
    ) {
        for i in 1..len {
            let v = cands[i];
            let key = self.small_candidate_key(v, mask, remaining);
            let mut j = i;
            while j > 0 && self.small_candidate_key(cands[j - 1], mask, remaining) > key {
                cands[j] = cands[j - 1];
                j -= 1;
            }
            cands[j] = v;
        }
    }

    fn small_candidate_key(&self, v: usize, mask: u64, remaining: usize) -> (usize, usize, usize) {
        let next_mask = mask | (1u64 << v);
        let mut onward = 0usize;
        let mut second_degree_sum = 0usize;
        for &u in &self.grid.adj[v] {
            if u != NONE && ((next_mask >> u) & 1) == 0 {
                onward += 1;
                second_degree_sum += self.grid.adj[u]
                    .iter()
                    .filter(|&&z| z != NONE && ((next_mask >> z) & 1) == 0)
                    .count();
            }
        }
        let start_penalty = if self.require_cycle
            && remaining > 1
            && self.grid.adjacent(v, self.grid.start)
        {
            3
        } else {
            0
        };
        let (x, y) = self.grid.coords[v];
        (onward + start_penalty, second_degree_sum, (x * 31 + y) & 255)
    }
}

struct Search<'a> {
    grid: &'a Grid,
    require_cycle: bool,
    visited: Vec<bool>,
    path: Vec<usize>,
    seen: Vec<u32>,
    queue: Vec<usize>,
    stamp: u32,
    deadline: Instant,
    nodes: u64,
    timed_out: bool,
}

impl<'a> Search<'a> {
    fn new(grid: &'a Grid, require_cycle: bool, deadline: Instant) -> Self {
        let mut visited = vec![false; grid.n];
        visited[grid.start] = true;
        Self {
            grid,
            require_cycle,
            visited,
            path: vec![grid.start],
            seen: vec![0; grid.n],
            queue: Vec::with_capacity(grid.n),
            stamp: 1,
            deadline,
            nodes: 0,
            timed_out: false,
        }
    }

    fn solve(&mut self) -> Option<Vec<usize>> {
        if self.dfs(self.grid.start, self.grid.n - 1) {
            Some(self.path.clone())
        } else {
            None
        }
    }

    fn dfs(&mut self, cur: usize, rem: usize) -> bool {
        self.nodes += 1;
        if (self.nodes & 8191) == 0 && Instant::now() >= self.deadline {
            self.timed_out = true;
            return false;
        }

        if rem == 0 {
            return !self.require_cycle || self.grid.adjacent(cur, self.grid.start);
        }

        if !self.state_prune(cur, rem) {
            return false;
        }

        let sole_start_neighbor = if self.require_cycle && rem > 1 {
            self.only_unvisited_neighbor_of_start()
        } else {
            None
        };

        let mut cands = [NONE; 4];
        let mut len = 0usize;
        for &u in &self.grid.adj[cur] {
            if u == NONE || self.visited[u] {
                continue;
            }
            if self.require_cycle {
                if rem == 1 && !self.grid.adjacent(u, self.grid.start) {
                    continue;
                }
                if rem > 1 && sole_start_neighbor == Some(u) {
                    continue;
                }
            }
            cands[len] = u;
            len += 1;
        }

        if len == 0 {
            return false;
        }

        self.sort_candidates(&mut cands, len, rem);

        for i in 0..len {
            let next = cands[i];
            self.visited[next] = true;
            self.path.push(next);
            if self.dfs(next, rem - 1) {
                return true;
            }
            self.path.pop();
            self.visited[next] = false;
            if self.timed_out {
                return false;
            }
        }

        false
    }

    fn state_prune(&mut self, cur: usize, rem: usize) -> bool {
        if !self.residual_connected(rem) {
            return false;
        }

        if self.require_cycle {
            let start_open = self
                .grid
                .adj[self.grid.start]
                .iter()
                .any(|&u| u != NONE && !self.visited[u]);
            if !start_open {
                return false;
            }
        }

        self.leaf_prune(cur, rem)
    }

    fn residual_connected(&mut self, rem: usize) -> bool {
        if rem == 0 {
            return true;
        }

        self.stamp = self.stamp.wrapping_add(1);
        if self.stamp == 0 {
            self.seen.fill(0);
            self.stamp = 1;
        }

        let Some(first) = (0..self.grid.n).find(|&v| !self.visited[v]) else {
            return rem == 0;
        };

        self.queue.clear();
        self.queue.push(first);
        self.seen[first] = self.stamp;
        let mut head = 0usize;
        let mut count = 0usize;

        while head < self.queue.len() {
            let v = self.queue[head];
            head += 1;
            count += 1;
            for &u in &self.grid.adj[v] {
                if u != NONE && !self.visited[u] && self.seen[u] != self.stamp {
                    self.seen[u] = self.stamp;
                    self.queue.push(u);
                }
            }
        }

        count == rem
    }

    fn leaf_prune(&self, cur: usize, rem: usize) -> bool {
        if rem <= 1 {
            return true;
        }

        let mut leaves = [(false, false); 3];
        let mut leaf_count = 0usize;

        for v in 0..self.grid.n {
            if self.visited[v] {
                continue;
            }
            let mut deg_r = 0usize;
            for &u in &self.grid.adj[v] {
                if u != NONE && !self.visited[u] {
                    deg_r += 1;
                }
            }

            if deg_r == 0 {
                return false;
            }

            if deg_r == 1 {
                if leaf_count >= 3 {
                    return false;
                }
                leaves[leaf_count] = (
                    self.grid.adjacent(v, cur),
                    self.grid.adjacent(v, self.grid.start),
                );
                leaf_count += 1;
            }
        }

        if !self.require_cycle {
            if leaf_count > 2 {
                return false;
            }
            if leaf_count == 2 && !leaves[..2].iter().any(|&(near_cur, _)| near_cur) {
                return false;
            }
            return true;
        }

        if leaf_count > 2 {
            return false;
        }
        match leaf_count {
            0 => true,
            1 => leaves[0].0 || leaves[0].1,
            2 => {
                (leaves[0].0 && leaves[1].1) || (leaves[0].1 && leaves[1].0)
            }
            _ => false,
        }
    }

    fn only_unvisited_neighbor_of_start(&self) -> Option<usize> {
        let mut result = None;
        for &u in &self.grid.adj[self.grid.start] {
            if u != NONE && !self.visited[u] {
                if result.is_some() {
                    return None;
                }
                result = Some(u);
            }
        }
        result
    }

    fn sort_candidates(&self, cands: &mut [usize; 4], len: usize, rem: usize) {
        for i in 1..len {
            let v = cands[i];
            let key = self.candidate_key(v, rem);
            let mut j = i;
            while j > 0 && self.candidate_key(cands[j - 1], rem) > key {
                cands[j] = cands[j - 1];
                j -= 1;
            }
            cands[j] = v;
        }
    }

    fn candidate_key(&self, v: usize, rem: usize) -> (usize, usize, usize) {
        let mut onward = 0usize;
        let mut second_degree_sum = 0usize;

        for &u in &self.grid.adj[v] {
            if u != NONE && !self.visited[u] && u != v {
                onward += 1;
                second_degree_sum += self
                    .grid
                    .adj[u]
                    .iter()
                    .filter(|&&z| z != NONE && !self.visited[z] && z != v)
                    .count();
            }
        }

        let start_penalty = if self.require_cycle
            && rem > 1
            && self.grid.adjacent(v, self.grid.start)
        {
            3
        } else {
            0
        };
        let (x, y) = self.grid.coords[v];
        let tie = (x.wrapping_mul(1_000_003) ^ y.wrapping_mul(9176)) & 1023;
        (onward + start_penalty, second_degree_sum, tie)
    }
}

fn validate_path(grid: &Grid, path: &[usize], require_cycle: bool) -> bool {
    if path.len() != grid.n || path.first().copied() != Some(grid.start) {
        return false;
    }

    let mut seen = vec![false; grid.n];
    for (i, &v) in path.iter().enumerate() {
        if v >= grid.n || seen[v] {
            return false;
        }
        seen[v] = true;
        if i > 0 && !grid.adjacent(path[i - 1], v) {
            return false;
        }
    }

    !require_cycle || grid.adjacent(*path.last().unwrap(), grid.start)
}
