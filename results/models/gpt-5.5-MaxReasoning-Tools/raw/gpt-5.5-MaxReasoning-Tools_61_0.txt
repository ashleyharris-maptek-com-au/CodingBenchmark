declare const require: any;

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type PackageInput = { id: any; x: number; y: number };

const packages: PackageInput[] = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

function asNumber(value: any, fallback = 0): number {
  const v = Number(value);
  return isFinite(v) ? v : fallback;
}

const depotX = asNumber(Array.isArray(data.depot) ? data.depot[0] : 0);
const depotY = asNumber(Array.isArray(data.depot) ? data.depot[1] : 0);

const ids: any[] = new Array(n);
const xs = new Float64Array(n);
const ys = new Float64Array(n);
const depotDist = new Float64Array(n);
const depotDistSq = new Float64Array(n);

for (let i = 0; i < n; i++) {
  const p = packages[i];
  ids[i] = p.id;
  xs[i] = asNumber(p.x);
  ys[i] = asNumber(p.y);
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  const d2 = dx * dx + dy * dy;
  depotDistSq[i] = d2;
  depotDist[i] = Math.sqrt(d2);
}

const DEPOT = n;
const EPS = 1e-10;

function sqDist(a: number, b: number): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return dx * dx + dy * dy;
}

function dist(a: number, b: number): number {
  if (a === DEPOT) return b === DEPOT ? 0 : depotDist[b];
  if (b === DEPOT) return depotDist[a];
  return Math.sqrt(sqDist(a, b));
}

function makeIndexArray(): number[] {
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  return order;
}

function toInt32Array(values: number[]): Int32Array {
  const out = new Int32Array(values.length);
  for (let i = 0; i < values.length; i++) out[i] = values[i];
  return out;
}

function routeCost(route: number[]): number {
  const m = route.length;
  if (m === 0) return 0;
  let total = depotDist[route[0]] + depotDist[route[m - 1]];
  for (let i = 0; i + 1 < m; i++) total += dist(route[i], route[i + 1]);
  return total;
}

function reverseRange(route: number[], left: number, right: number): void {
  while (left < right) {
    const t = route[left];
    route[left] = route[right];
    route[right] = t;
    left++;
    right--;
  }
}

function reverseRangeWithPositions(route: number[], pos: Int32Array, left: number, right: number): void {
  while (left < right) {
    const a = route[left];
    const b = route[right];
    route[left] = b;
    route[right] = a;
    pos[a] = right;
    pos[b] = left;
    left++;
    right--;
  }
  if (left === right) pos[route[left]] = left;
}

function rotateToBestDepotCut(route: number[]): number[] {
  const m = route.length;
  if (m <= 2) return route;

  let bestBreak = m - 1;
  let bestScore = depotDist[route[m - 1]] + depotDist[route[0]] - dist(route[m - 1], route[0]);
  for (let i = 0; i + 1 < m; i++) {
    const a = route[i];
    const b = route[i + 1];
    const score = depotDist[a] + depotDist[b] - dist(a, b);
    if (score < bestScore - EPS) {
      bestScore = score;
      bestBreak = i;
    }
  }

  if (bestBreak === m - 1) return route;
  return route.slice(bestBreak + 1).concat(route.slice(0, bestBreak + 1));
}

function solveExactHeldKarp(): number[] {
  const m = n;
  const size = 1 << m;
  const full = size - 1;
  const pair = new Float64Array(m * m);
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < m; j++) pair[i * m + j] = i === j ? 0 : dist(i, j);
  }

  const inf = Number.POSITIVE_INFINITY;
  const dp = new Float64Array(size * m);
  for (let i = 0; i < dp.length; i++) dp[i] = inf;
  const parent = new Int8Array(size * m);
  for (let i = 0; i < parent.length; i++) parent[i] = -1;
  const bitToIndex = new Int8Array(size);
  for (let i = 0; i < m; i++) bitToIndex[1 << i] = i;

  for (let i = 0; i < m; i++) dp[((1 << i) * m) + i] = depotDist[i];

  for (let mask = 1; mask < size; mask++) {
    const base = mask * m;
    for (let last = 0; last < m; last++) {
      const current = dp[base + last];
      if (current === inf) continue;
      let remaining = full ^ mask;
      while (remaining !== 0) {
        const bit = remaining & -remaining;
        const next = bitToIndex[bit];
        const nextMask = mask | bit;
        const idx = nextMask * m + next;
        const nextCost = current + pair[last * m + next];
        if (nextCost < dp[idx]) {
          dp[idx] = nextCost;
          parent[idx] = last;
        }
        remaining ^= bit;
      }
    }
  }

  let bestLast = 0;
  let best = inf;
  const fullBase = full * m;
  for (let last = 0; last < m; last++) {
    const cost = dp[fullBase + last] + depotDist[last];
    if (cost < best) {
      best = cost;
      bestLast = last;
    }
  }

  const reversed: number[] = [];
  let mask = full;
  let last = bestLast;
  while (last >= 0) {
    reversed.push(last);
    const p = parent[mask * m + last];
    mask ^= 1 << last;
    last = p;
  }
  reversed.reverse();
  return reversed;
}

function farthestInsertionRoute(): number[] {
  let first = 0;
  let farthest = -1;
  for (let i = 0; i < n; i++) {
    if (depotDistSq[i] > farthest) {
      farthest = depotDistSq[i];
      first = i;
    }
  }

  const next = new Int32Array(n + 1);
  const prev = new Int32Array(n + 1);
  const inRoute = new Uint8Array(n);
  const nearest = new Float64Array(n);

  next[DEPOT] = first;
  prev[DEPOT] = first;
  next[first] = DEPOT;
  prev[first] = DEPOT;
  inRoute[first] = 1;

  for (let i = 0; i < n; i++) {
    nearest[i] = i === first ? -1 : Math.min(depotDistSq[i], sqDist(first, i));
  }

  let remaining = n - 1;
  while (remaining > 0) {
    let p = -1;
    let bestNear = -1;
    for (let i = 0; i < n; i++) {
      if (inRoute[i] === 0 && nearest[i] > bestNear) {
        bestNear = nearest[i];
        p = i;
      }
    }
    if (p < 0) break;

    let bestA = DEPOT;
    let bestDelta = Number.POSITIVE_INFINITY;
    let a = DEPOT;
    while (true) {
      const b = next[a];
      const delta = dist(a, p) + dist(p, b) - dist(a, b);
      if (delta < bestDelta) {
        bestDelta = delta;
        bestA = a;
      }
      if (b === DEPOT) break;
      a = b;
    }

    const b = next[bestA];
    next[bestA] = p;
    prev[p] = bestA;
    next[p] = b;
    prev[b] = p;
    inRoute[p] = 1;
    remaining--;

    for (let i = 0; i < n; i++) {
      if (inRoute[i] === 0) {
        const d2 = sqDist(p, i);
        if (d2 < nearest[i]) nearest[i] = d2;
      }
    }
  }

  const route: number[] = [];
  let cur = next[DEPOT];
  while (cur !== DEPOT && route.length < n) {
    route.push(cur);
    cur = next[cur];
  }
  return route;
}

let angleCache: Float64Array | null = null;
let radiusCache: Float64Array | null = null;

function getAngles(): Float64Array {
  if (angleCache === null) {
    angleCache = new Float64Array(n);
    radiusCache = new Float64Array(n);
    for (let i = 0; i < n; i++) {
      const dx = xs[i] - depotX;
      const dy = ys[i] - depotY;
      angleCache[i] = Math.atan2(dy, dx);
      radiusCache[i] = dx * dx + dy * dy;
    }
  }
  return angleCache;
}

function getRadii(): Float64Array {
  if (radiusCache === null) getAngles();
  return radiusCache as Float64Array;
}

function polarRoute(): number[] {
  const angle = getAngles();
  const radius = getRadii();
  const order = makeIndexArray();
  order.sort((a, b) => angle[a] - angle[b] || radius[a] - radius[b] || a - b);
  return rotateToBestDepotCut(order);
}

function part1By1(v: number): number {
  v &= 0xffff;
  v = (v | (v << 8)) & 0x00ff00ff;
  v = (v | (v << 4)) & 0x0f0f0f0f;
  v = (v | (v << 2)) & 0x33333333;
  v = (v | (v << 1)) & 0x55555555;
  return v >>> 0;
}

let mortonCache: Float64Array | null = null;

function getMortonKeys(): Float64Array {
  if (mortonCache !== null) return mortonCache;

  let minX = Number.POSITIVE_INFINITY;
  let maxX = Number.NEGATIVE_INFINITY;
  let minY = Number.POSITIVE_INFINITY;
  let maxY = Number.NEGATIVE_INFINITY;
  for (let i = 0; i < n; i++) {
    if (xs[i] < minX) minX = xs[i];
    if (xs[i] > maxX) maxX = xs[i];
    if (ys[i] < minY) minY = ys[i];
    if (ys[i] > maxY) maxY = ys[i];
  }

  const scaleX = maxX > minX ? 65535 / (maxX - minX) : 0;
  const scaleY = maxY > minY ? 65535 / (maxY - minY) : 0;
  const keys = new Float64Array(n);
  for (let i = 0; i < n; i++) {
    const xi = Math.max(0, Math.min(65535, Math.floor((xs[i] - minX) * scaleX)));
    const yi = Math.max(0, Math.min(65535, Math.floor((ys[i] - minY) * scaleY)));
    keys[i] = (part1By1(xi) | (part1By1(yi) << 1)) >>> 0;
  }
  mortonCache = keys;
  return keys;
}

function mortonRoute(): number[] {
  const keys = getMortonKeys();
  const order = makeIndexArray();
  order.sort((a, b) => keys[a] - keys[b] || a - b);
  return rotateToBestDepotCut(order);
}

function serpentineRoute(): number[] {
  const order = makeIndexArray();
  order.sort((a, b) => ys[a] - ys[b] || xs[a] - xs[b] || a - b);
  const rows = Math.max(1, Math.round(Math.sqrt(n)));
  const rowSize = Math.ceil(n / rows);
  const route: number[] = [];

  for (let r = 0; r < rows; r++) {
    const start = r * rowSize;
    const end = Math.min(n, start + rowSize);
    if (start >= end) break;
    const row = order.slice(start, end);
    if ((r & 1) === 0) row.sort((a, b) => xs[a] - xs[b] || ys[a] - ys[b] || a - b);
    else row.sort((a, b) => xs[b] - xs[a] || ys[b] - ys[a] || a - b);
    for (let i = 0; i < row.length; i++) route.push(row[i]);
  }

  return rotateToBestDepotCut(route);
}

function chooseBestInitialRoute(): number[] {
  let best: number[];
  if (n <= 5000) best = rotateToBestDepotCut(farthestInsertionRoute());
  else best = mortonRoute();

  let bestCost = routeCost(best);
  const alternatives: number[][] = n <= 12000
    ? [mortonRoute(), polarRoute(), serpentineRoute()]
    : [polarRoute(), serpentineRoute()];

  for (const candidate of alternatives) {
    const cost = routeCost(candidate);
    if (cost < bestCost) {
      best = candidate;
      bestCost = cost;
    }
  }
  return best;
}

function improve2OptFull(route: number[], maxPasses: number): void {
  const m = route.length;
  if (m < 3) return;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;
    for (let i = -1; i < m - 2; i++) {
      const a = i >= 0 ? route[i] : DEPOT;
      let b = route[i + 1];
      let ab = dist(a, b);
      for (let j = i + 2; j < m; j++) {
        if (i < 0 && j === m - 1) continue;
        const c = route[j];
        const d = j + 1 < m ? route[j + 1] : DEPOT;
        const oldCost = ab + dist(c, d);
        const newCost = dist(a, c) + dist(b, d);
        if (newCost + EPS < oldCost) {
          reverseRange(route, i + 1, j);
          improved = true;
          b = route[i + 1];
          ab = dist(a, b);
        }
      }
    }
    if (!improved) break;
  }
}

function addOrderNeighbors(raw: number[][], order: number[], window: number): void {
  const m = order.length;
  for (let pos = 0; pos < m; pos++) {
    const a = order[pos];
    const arr = raw[a];
    const lo = Math.max(0, pos - window);
    const hi = Math.min(m - 1, pos + window);
    for (let p = lo; p <= hi; p++) {
      if (p !== pos) arr.push(order[p]);
    }
  }
}

function buildCandidateLists(k: number, window: number): { lists: Int32Array[]; depot: Int32Array } {
  const raw = new Array<number[]>(n);
  for (let i = 0; i < n; i++) raw[i] = [];

  const orderX = makeIndexArray();
  orderX.sort((a, b) => xs[a] - xs[b] || ys[a] - ys[b] || a - b);
  addOrderNeighbors(raw, orderX, window);

  const orderY = makeIndexArray();
  orderY.sort((a, b) => ys[a] - ys[b] || xs[a] - xs[b] || a - b);
  addOrderNeighbors(raw, orderY, window);

  const orderSum = makeIndexArray();
  orderSum.sort((a, b) => (xs[a] + ys[a]) - (xs[b] + ys[b]) || a - b);
  addOrderNeighbors(raw, orderSum, window);

  const orderDiff = makeIndexArray();
  orderDiff.sort((a, b) => (xs[a] - ys[a]) - (xs[b] - ys[b]) || a - b);
  addOrderNeighbors(raw, orderDiff, window);

  const angle = getAngles();
  const radius = getRadii();
  const orderAngle = makeIndexArray();
  orderAngle.sort((a, b) => angle[a] - angle[b] || radius[a] - radius[b] || a - b);
  addOrderNeighbors(raw, orderAngle, window);

  const keys = getMortonKeys();
  const orderMorton = makeIndexArray();
  orderMorton.sort((a, b) => keys[a] - keys[b] || a - b);
  addOrderNeighbors(raw, orderMorton, window);

  const lists = new Array<Int32Array>(n);
  for (let i = 0; i < n; i++) {
    const arr = raw[i];
    arr.sort((a, b) => a - b);
    const unique: number[] = [];
    let last = -1;
    for (let j = 0; j < arr.length; j++) {
      const v = arr[j];
      if (v !== i && v !== last) {
        unique.push(v);
        last = v;
      }
    }
    unique.sort((a, b) => sqDist(i, a) - sqDist(i, b) || a - b);
    if (unique.length > k) unique.length = k;
    lists[i] = toInt32Array(unique);
  }

  const byDepot = makeIndexArray();
  byDepot.sort((a, b) => depotDistSq[a] - depotDistSq[b] || a - b);
  const depotCandidates = byDepot.slice(0, Math.min(n, Math.max(k, 8)));

  return { lists, depot: toInt32Array(depotCandidates) };
}

function try2Opt(route: number[], pos: Int32Array, i: number, j: number): boolean {
  const m = route.length;
  if (j <= i + 1 || j >= m) return false;
  if (i < 0 && j === m - 1) return false;

  const a = i >= 0 ? route[i] : DEPOT;
  const b = route[i + 1];
  const c = route[j];
  const d = j + 1 < m ? route[j + 1] : DEPOT;
  const oldCost = dist(a, b) + dist(c, d);
  const newCost = dist(a, c) + dist(b, d);

  if (newCost + EPS < oldCost) {
    reverseRangeWithPositions(route, pos, i + 1, j);
    return true;
  }
  return false;
}

function improve2OptCandidates(route: number[], lists: Int32Array[], depotCandidates: Int32Array, maxPasses: number): void {
  const m = route.length;
  if (m < 3) return;

  const pos = new Int32Array(n);
  for (let i = 0; i < m; i++) pos[route[i]] = i;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = -1; i < m - 2; i++) {
      const a = i >= 0 ? route[i] : DEPOT;
      const b = route[i + 1];
      const listA = a === DEPOT ? depotCandidates : lists[a];
      let changed = false;

      for (let t = 0; t < listA.length; t++) {
        const j = pos[listA[t]];
        if (try2Opt(route, pos, i, j)) {
          changed = true;
          break;
        }
      }

      if (!changed) {
        const listB = lists[b];
        for (let t = 0; t < listB.length; t++) {
          const j = pos[listB[t]] - 1;
          if (try2Opt(route, pos, i, j)) {
            changed = true;
            break;
          }
        }
      }

      if (changed) {
        improved = true;
        i = Math.max(-2, i - 3);
      }
    }

    if (!improved) break;
  }
}

function improveRoute(route: number[]): number[] {
  route = rotateToBestDepotCut(route);

  if (n <= 2200) {
    const passes = n <= 350 ? 80 : n <= 900 ? 35 : 16;
    improve2OptFull(route, passes);
    route = rotateToBestDepotCut(route);
    improve2OptFull(route, Math.max(4, Math.floor(passes / 3)));
  } else {
    const k = n <= 6000 ? 32 : n <= 20000 ? 24 : 16;
    const window = n <= 6000 ? 14 : n <= 20000 ? 10 : 7;
    const passes = n <= 6000 ? 10 : n <= 20000 ? 7 : 4;
    const candidates = buildCandidateLists(k, window);
    improve2OptCandidates(route, candidates.lists, candidates.depot, passes);
    route = rotateToBestDepotCut(route);
    improve2OptCandidates(route, candidates.lists, candidates.depot, Math.max(2, Math.floor(passes / 2)));
  }

  return route;
}

let route: number[];
if (n === 0) {
  route = [];
} else if (n <= 17) {
  route = solveExactHeldKarp();
} else {
  route = improveRoute(chooseBestInitialRoute());
}

const output = n === 0 ? { routes: [] } : { routes: [route.map((idx) => ids[idx])] };
console.log(JSON.stringify(output));
