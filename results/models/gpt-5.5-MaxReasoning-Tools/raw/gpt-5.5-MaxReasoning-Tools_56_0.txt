const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

const depotX = Number(depot[0]) || 0;
const depotY = Number(depot[1]) || 0;
const xs = new Float64Array(n);
const ys = new Float64Array(n);
const ids = new Array(n);

for (let i = 0; i < n; i++) {
  const p = packages[i];
  xs[i] = Number(p.x) || 0;
  ys[i] = Number(p.y) || 0;
  ids[i] = p.id;
}

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

function distDepot(i) {
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  return Math.sqrt(dx * dx + dy * dy);
}

function distIdx(a, b) {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.sqrt(dx * dx + dy * dy);
}

function distNode(a, b) {
  if (a < 0) return distDepot(b);
  if (b < 0) return distDepot(a);
  return distIdx(a, b);
}

function routeLength(tour) {
  const m = tour.length;
  if (m === 0) return 0;
  let sum = distDepot(tour[0]);
  for (let i = 1; i < m; i++) sum += distIdx(tour[i - 1], tour[i]);
  sum += distDepot(tour[m - 1]);
  return sum;
}

function reverseSegment(tour, lo, hi) {
  while (lo < hi) {
    const t = tour[lo];
    tour[lo++] = tour[hi];
    tour[hi--] = t;
  }
}

function exactTour() {
  const states = 1 << n;
  const full = states - 1;
  const size = states * n;
  const inf = Number.POSITIVE_INFINITY;
  const dp = new Float64Array(size);
  dp.fill(inf);
  const parent = new Int16Array(size);
  parent.fill(-1);

  const depotD = new Float64Array(n);
  const pairD = new Float64Array(n * n);
  for (let i = 0; i < n; i++) {
    depotD[i] = distDepot(i);
    for (let j = 0; j < n; j++) pairD[i * n + j] = i === j ? 0 : distIdx(i, j);
  }

  for (let i = 0; i < n; i++) dp[((1 << i) * n) + i] = depotD[i];

  for (let mask = 1; mask < states; mask++) {
    const maskBase = mask * n;
    let remaining = full ^ mask;
    if (remaining === 0) continue;

    for (let last = 0; last < n; last++) {
      if ((mask & (1 << last)) === 0) continue;
      const base = dp[maskBase + last];
      if (!Number.isFinite(base)) continue;

      let rem = remaining;
      const lastBase = last * n;
      while (rem !== 0) {
        const bit = rem & -rem;
        const next = 31 - Math.clz32(bit);
        const nextMask = mask | bit;
        const at = nextMask * n + next;
        const candidate = base + pairD[lastBase + next];
        if (candidate < dp[at]) {
          dp[at] = candidate;
          parent[at] = last;
        }
        rem ^= bit;
      }
    }
  }

  let bestEnd = 0;
  let best = inf;
  const fullBase = full * n;
  for (let last = 0; last < n; last++) {
    const candidate = dp[fullBase + last] + depotD[last];
    if (candidate < best) {
      best = candidate;
      bestEnd = last;
    }
  }

  const tour = new Array(n);
  let mask = full;
  let last = bestEnd;
  for (let pos = n - 1; pos >= 0; pos--) {
    tour[pos] = last;
    const prev = parent[mask * n + last];
    mask ^= 1 << last;
    last = prev;
  }
  return tour;
}

function farthestInsertionTour() {
  const visited = new Uint8Array(n);
  const minD = new Float64Array(n);

  let start = 0;
  let far = -1;
  for (let i = 0; i < n; i++) {
    const d = distDepot(i);
    if (d > far) {
      far = d;
      start = i;
    }
  }

  const tour = [start];
  visited[start] = 1;
  for (let i = 0; i < n; i++) {
    minD[i] = i === start ? -1 : Math.min(distDepot(i), distIdx(i, start));
  }

  for (let used = 1; used < n; used++) {
    let next = -1;
    let bestMin = -1;
    for (let i = 0; i < n; i++) {
      if (!visited[i] && minD[i] > bestMin) {
        bestMin = minD[i];
        next = i;
      }
    }

    let bestPos = 0;
    let bestDelta = Number.POSITIVE_INFINITY;
    const len = tour.length;
    for (let pos = 0; pos <= len; pos++) {
      const prev = pos === 0 ? -1 : tour[pos - 1];
      const after = pos === len ? -1 : tour[pos];
      const delta = distNode(prev, next) + distNode(next, after) - distNode(prev, after);
      if (delta < bestDelta) {
        bestDelta = delta;
        bestPos = pos;
      }
    }

    tour.splice(bestPos, 0, next);
    visited[next] = 1;
    minD[next] = -1;

    for (let i = 0; i < n; i++) {
      if (!visited[i]) {
        const d = distIdx(i, next);
        if (d < minD[i]) minD[i] = d;
      }
    }
  }

  return tour;
}

function hilbertIndex(x, y) {
  let d = 0;
  for (let s = 32768; s > 0; s >>= 1) {
    const rx = (x & s) ? 1 : 0;
    const ry = (y & s) ? 1 : 0;
    d += s * s * ((3 * rx) ^ ry);
    if (ry === 0) {
      if (rx === 1) {
        x = 65535 - x;
        y = 65535 - y;
      }
      const t = x;
      x = y;
      y = t;
    }
  }
  return d;
}

function makeIndexArray() {
  const order = new Array(n);
  for (let i = 0; i < n; i++) order[i] = i;
  return order;
}

function spatialSortTour(kind) {
  const order = makeIndexArray();
  if (kind === 'angle') {
    order.sort((a, b) => {
      const aa = Math.atan2(ys[a] - depotY, xs[a] - depotX);
      const bb = Math.atan2(ys[b] - depotY, xs[b] - depotX);
      if (aa !== bb) return aa - bb;
      return distDepot(a) - distDepot(b);
    });
    return order;
  }

  let minX = depotX;
  let maxX = depotX;
  let minY = depotY;
  let maxY = depotY;
  for (let i = 0; i < n; i++) {
    const x = xs[i];
    const y = ys[i];
    if (x < minX) minX = x;
    if (x > maxX) maxX = x;
    if (y < minY) minY = y;
    if (y > maxY) maxY = y;
  }

  const spanX = maxX - minX || 1;
  const spanY = maxY - minY || 1;
  const scaleX = 65535 / spanX;
  const scaleY = 65535 / spanY;
  const codes = new Float64Array(n);

  const swap = (kind & 1) !== 0;
  const invX = (kind & 2) !== 0;
  const invY = (kind & 4) !== 0;

  for (let i = 0; i < n; i++) {
    let qx = Math.floor((xs[i] - minX) * scaleX);
    let qy = Math.floor((ys[i] - minY) * scaleY);
    if (qx < 0) qx = 0;
    else if (qx > 65535) qx = 65535;
    if (qy < 0) qy = 0;
    else if (qy > 65535) qy = 65535;
    if (invX) qx = 65535 - qx;
    if (invY) qy = 65535 - qy;
    if (swap) {
      const t = qx;
      qx = qy;
      qy = t;
    }
    codes[i] = hilbertIndex(qx, qy);
  }

  order.sort((a, b) => {
    const ca = codes[a];
    const cb = codes[b];
    if (ca !== cb) return ca - cb;
    return a - b;
  });
  return order;
}

function bestSpatialTour() {
  const kinds = n > 20000 ? [0, 1, 'angle'] : [0, 1, 2, 4, 7, 'angle'];
  let bestTour = null;
  let bestLen = Number.POSITIVE_INFINITY;
  for (const kind of kinds) {
    const tour = spatialSortTour(kind);
    const len = routeLength(tour);
    if (len < bestLen) {
      bestLen = len;
      bestTour = tour;
    }
  }
  return bestTour;
}

function twoOpt(tour) {
  const m = tour.length;
  if (m < 4) return tour;

  let maxPasses;
  let window;
  if (m <= 500) {
    maxPasses = 20;
    window = m;
  } else if (m <= 1800) {
    maxPasses = 12;
    window = m;
  } else if (m <= 5000) {
    maxPasses = 8;
    window = 220;
  } else if (m <= 20000) {
    maxPasses = 5;
    window = 96;
  } else {
    maxPasses = 3;
    window = 48;
  }

  const eps = 1e-10;
  for (let pass = 0; pass < maxPasses; pass++) {
    let changed = false;

    for (let i = 0; i < m - 1; i++) {
      const maxK = Math.min(m - 1, i + window);
      const a = i === 0 ? -1 : tour[i - 1];
      const b = tour[i];
      const ab = distNode(a, b);

      for (let k = i + 1; k <= maxK; k++) {
        if (i === 0 && k === m - 1) continue;
        const c = tour[k];
        const d = k === m - 1 ? -1 : tour[k + 1];
        const gain = ab + distNode(c, d) - distNode(a, c) - distNode(b, d);
        if (gain > eps) {
          reverseSegment(tour, i, k);
          changed = true;
          break;
        }
      }
    }

    if (!changed) break;
  }
  return tour;
}

let tour;
if (n <= 18) {
  tour = exactTour();
} else {
  if (n <= 3000) {
    tour = farthestInsertionTour();
    const spatial = bestSpatialTour();
    if (routeLength(spatial) < routeLength(tour)) tour = spatial;
  } else {
    tour = bestSpatialTour();
  }
  twoOpt(tour);
}

const route = new Array(n);
for (let i = 0; i < n; i++) route[i] = ids[tour[i]];

process.stdout.write(JSON.stringify({ routes: [route] }));
