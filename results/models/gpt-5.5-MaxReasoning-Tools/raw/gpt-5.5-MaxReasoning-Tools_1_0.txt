#include <algorithm>
#include <cmath>
#include <cstddef>
#include <cstdint>
#include <cstdio>
#include <cstring>
#include <limits>
#include <new>
#include <stdexcept>
#include <vector>

class FastInput {
    static constexpr std::size_t BufferSize = 1u << 20;
    std::vector<char> buffer_;
    std::size_t pos_ = 0;
    std::size_t len_ = 0;

    int get() {
        if (pos_ >= len_) {
            len_ = std::fread(buffer_.data(), 1, BufferSize, stdin);
            pos_ = 0;
            if (len_ == 0) {
                return EOF;
            }
        }
        return static_cast<unsigned char>(buffer_[pos_++]);
    }

    void unget(int c) {
        if (c != EOF && pos_ > 0) {
            --pos_;
        }
    }

    bool skipSpace() {
        int c;
        do {
            c = get();
            if (c == EOF) {
                return false;
            }
        } while (c <= ' ');
        unget(c);
        return true;
    }

public:
    FastInput() : buffer_(BufferSize) {}

    bool readUInt64(std::uint64_t &out) {
        if (!skipSpace()) {
            return false;
        }
        std::uint64_t value = 0;
        int c = get();
        bool any = false;
        while (c >= '0' && c <= '9') {
            any = true;
            value = value * 10 + static_cast<std::uint64_t>(c - '0');
            c = get();
        }
        unget(c);
        out = value;
        return any;
    }

    bool readDouble(double &out) {
        if (!skipSpace()) {
            return false;
        }

        int c = get();
        double sign = 1.0;
        if (c == '-') {
            sign = -1.0;
            c = get();
        } else if (c == '+') {
            c = get();
        }

        double value = 0.0;
        bool any = false;
        while (c >= '0' && c <= '9') {
            any = true;
            value = value * 10.0 + static_cast<double>(c - '0');
            c = get();
        }

        if (c == '.') {
            double place = 0.1;
            c = get();
            while (c >= '0' && c <= '9') {
                any = true;
                value += static_cast<double>(c - '0') * place;
                place *= 0.1;
                c = get();
            }
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int exponent = 0;
            c = get();
            if (c == '-') {
                expSign = -1;
                c = get();
            } else if (c == '+') {
                c = get();
            }
            while (c >= '0' && c <= '9') {
                exponent = exponent * 10 + (c - '0');
                c = get();
            }
            value *= std::pow(10.0, static_cast<double>(expSign * exponent));
        }

        unget(c);
        out = sign * value;
        return any;
    }
};

class FastOutput {
    static constexpr std::size_t BufferSize = 1u << 20;
    std::vector<char> buffer_;
    std::size_t pos_ = 0;

public:
    FastOutput() : buffer_(BufferSize) {}

    ~FastOutput() {
        flush();
    }

    void flush() {
        if (pos_ != 0) {
            std::fwrite(buffer_.data(), 1, pos_, stdout);
            pos_ = 0;
        }
    }

    void put(char c) {
        if (pos_ == BufferSize) {
            flush();
        }
        buffer_[pos_++] = c;
    }

    void writeUInt64(std::uint64_t value) {
        char digits[32];
        int count = 0;
        do {
            digits[count++] = static_cast<char>('0' + (value % 10));
            value /= 10;
        } while (value != 0);
        while (count > 0) {
            put(digits[--count]);
        }
    }
};

struct Point {
    float x;
    float y;
};

struct KeyIndex {
    std::uint64_t key;
    std::uint32_t index;
};

static double squaredDistance(const Point &a, const Point &b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static double distanceBetween(const std::vector<Point> &points, std::uint32_t a, std::uint32_t b) {
    return std::sqrt(squaredDistance(points[a], points[b]));
}

static std::uint64_t partBy1(std::uint32_t x) {
    std::uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFull;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFull;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0Full;
    v = (v | (v << 2)) & 0x3333333333333333ull;
    v = (v | (v << 1)) & 0x5555555555555555ull;
    return v;
}

static std::uint64_t mortonKey(std::uint32_t x, std::uint32_t y) {
    return partBy1(x) | (partBy1(y) << 1);
}

static std::uint32_t orderedFloatBits(float value) {
    if (!std::isfinite(value)) {
        value = 0.0f;
    }
    std::uint32_t bits = 0;
    std::memcpy(&bits, &value, sizeof(bits));
    if ((bits & 0x80000000u) != 0) {
        bits = ~bits;
    } else {
        bits ^= 0x80000000u;
    }
    return bits;
}

static std::uint64_t rawSpatialKey(float x, float y) {
    return mortonKey(orderedFloatBits(x), orderedFloatBits(y));
}

static std::uint32_t quantize(double value, double lo, double hi) {
    const double range = hi - lo;
    if (!(range > 0.0) || !std::isfinite(value)) {
        return 0;
    }
    double t = (value - lo) / range;
    if (t <= 0.0) {
        return 0;
    }
    if (t >= 1.0) {
        return std::numeric_limits<std::uint32_t>::max();
    }
    return static_cast<std::uint32_t>(t * 4294967295.0 + 0.5);
}

static void radixSortByKey(std::vector<KeyIndex> &items) {
    if (items.size() < 2) {
        return;
    }

    constexpr std::size_t Radix = 1u << 16;
    std::vector<KeyIndex> temp(items.size());
    std::vector<std::size_t> count(Radix);

    for (int shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        for (const KeyIndex &item : items) {
            ++count[static_cast<std::uint16_t>(item.key >> shift)];
        }

        std::size_t sum = 0;
        for (std::size_t i = 0; i < Radix; ++i) {
            const std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (const KeyIndex &item : items) {
            temp[count[static_cast<std::uint16_t>(item.key >> shift)]++] = item;
        }
        items.swap(temp);
    }
}

static void sortByKey(std::vector<KeyIndex> &items) {
    try {
        if (items.size() <= 20000000ull) {
            radixSortByKey(items);
            return;
        }
    } catch (const std::bad_alloc &) {
    }

    std::sort(items.begin(), items.end(), [](const KeyIndex &a, const KeyIndex &b) {
        if (a.key != b.key) {
            return a.key < b.key;
        }
        return a.index < b.index;
    });
}

static std::vector<std::uint32_t> exactHeldKarp(const std::vector<Point> &points) {
    const std::size_t n = points.size();
    std::vector<std::uint32_t> route(n);
    if (n == 0) {
        return route;
    }
    if (n == 1) {
        route[0] = 0;
        return route;
    }

    const std::size_t states = std::size_t{1} << n;
    const double inf = std::numeric_limits<double>::infinity();
    std::vector<double> dp(states * n, inf);
    std::vector<std::int16_t> parent(states * n, -1);
    dp[n] = 0.0; // mask 1, last 0

    for (std::size_t mask = 1; mask < states; ++mask) {
        if ((mask & 1u) == 0) {
            continue;
        }
        for (std::size_t last = 0; last < n; ++last) {
            const double base = dp[mask * n + last];
            if (!std::isfinite(base)) {
                continue;
            }
            for (std::size_t next = 1; next < n; ++next) {
                const std::size_t bit = std::size_t{1} << next;
                if ((mask & bit) != 0) {
                    continue;
                }
                const std::size_t nextMask = mask | bit;
                const double candidate = base + distanceBetween(points, static_cast<std::uint32_t>(last), static_cast<std::uint32_t>(next));
                double &slot = dp[nextMask * n + next];
                if (candidate < slot) {
                    slot = candidate;
                    parent[nextMask * n + next] = static_cast<std::int16_t>(last);
                }
            }
        }
    }

    const std::size_t full = states - 1;
    double best = inf;
    std::size_t last = 1;
    for (std::size_t i = 1; i < n; ++i) {
        const double candidate = dp[full * n + i] + distanceBetween(points, static_cast<std::uint32_t>(i), 0);
        if (candidate < best) {
            best = candidate;
            last = i;
        }
    }

    std::size_t mask = full;
    for (std::size_t pos = n - 1; pos >= 1; --pos) {
        route[pos] = static_cast<std::uint32_t>(last);
        const std::int16_t prev = parent[mask * n + last];
        mask ^= (std::size_t{1} << last);
        last = static_cast<std::size_t>(prev);
    }
    route[0] = 0;
    return route;
}

static std::vector<std::uint32_t> nearestNeighbor(const std::vector<Point> &points) {
    const std::size_t n = points.size();
    std::vector<std::uint32_t> route;
    route.reserve(n);
    std::vector<unsigned char> used(n, 0);

    std::uint32_t current = 0;
    route.push_back(0);
    used[0] = 1;

    for (std::size_t step = 1; step < n; ++step) {
        double best = std::numeric_limits<double>::infinity();
        std::uint32_t bestIndex = 0;
        for (std::size_t i = 1; i < n; ++i) {
            if (used[i]) {
                continue;
            }
            const double d = squaredDistance(points[current], points[i]);
            if (d < best) {
                best = d;
                bestIndex = static_cast<std::uint32_t>(i);
            }
        }
        current = bestIndex;
        used[current] = 1;
        route.push_back(current);
    }

    return route;
}

static void twoOptWindow(std::vector<std::uint32_t> &route, const std::vector<Point> &points, std::size_t window, int passes) {
    const std::size_t n = route.size();
    if (n < 4 || window < 2 || passes <= 0) {
        return;
    }

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (std::size_t i = 0; i + 3 < n; ++i) {
            const std::size_t maxJ = std::min(n - 2, i + window);
            for (std::size_t j = i + 2; j <= maxJ; ++j) {
                const std::uint32_t a = route[i];
                const std::uint32_t b = route[i + 1];
                const std::uint32_t c = route[j];
                const std::uint32_t d = route[j + 1];

                const double oldLen = distanceBetween(points, a, b) + distanceBetween(points, c, d);
                const double newLen = distanceBetween(points, a, c) + distanceBetween(points, b, d);
                if (newLen + 1e-12 < oldLen) {
                    std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i + 1),
                                 route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    improved = true;
                    break;
                }
            }
        }
        if (!improved) {
            break;
        }
    }
}

static std::vector<std::uint32_t> routeFromSortedKeys(const std::vector<KeyIndex> &items) {
    const std::size_t n = items.size();
    std::size_t start = 0;
    for (std::size_t i = 0; i < n; ++i) {
        if (items[i].index == 0) {
            start = i;
            break;
        }
    }

    std::vector<std::uint32_t> route;
    route.reserve(n);
    for (std::size_t offset = 0; offset < n; ++offset) {
        route.push_back(items[(start + offset) % n].index);
    }
    return route;
}

static std::vector<std::uint32_t> spatialRoute(const std::vector<Point> &points, double minX, double maxX, double minY, double maxY) {
    std::vector<KeyIndex> items;
    items.reserve(points.size());
    for (std::size_t i = 0; i < points.size(); ++i) {
        const std::uint32_t qx = quantize(points[i].x, minX, maxX);
        const std::uint32_t qy = quantize(points[i].y, minY, maxY);
        items.push_back(KeyIndex{mortonKey(qx, qy), static_cast<std::uint32_t>(i)});
    }
    sortByKey(items);
    return routeFromSortedKeys(items);
}

static void writeRoute(const std::vector<std::uint32_t> &route, FastOutput &out) {
    for (std::size_t i = 0; i < route.size(); ++i) {
        if (i != 0) {
            out.put(' ');
        }
        out.writeUInt64(route[i]);
    }
    out.put('\n');
}

static void writeSortedKeyRoute(const std::vector<KeyIndex> &items, FastOutput &out) {
    std::size_t start = 0;
    for (std::size_t i = 0; i < items.size(); ++i) {
        if (items[i].index == 0) {
            start = i;
            break;
        }
    }
    for (std::size_t offset = 0; offset < items.size(); ++offset) {
        if (offset != 0) {
            out.put(' ');
        }
        out.writeUInt64(items[(start + offset) % items.size()].index);
    }
    out.put('\n');
}

static void writeSequential(std::uint64_t n, FastOutput &out) {
    for (std::uint64_t i = 0; i < n; ++i) {
        if (i != 0) {
            out.put(' ');
        }
        out.writeUInt64(i);
    }
    out.put('\n');
}

static void consumeCoordinates(FastInput &in, std::uint64_t n) {
    double x = 0.0;
    double y = 0.0;
    for (std::uint64_t i = 0; i < n; ++i) {
        in.readDouble(x);
        in.readDouble(y);
    }
}

int main() {
    FastInput in;
    FastOutput out;

    std::uint64_t n64 = 0;
    if (!in.readUInt64(n64)) {
        return 0;
    }
    if (n64 == 0) {
        out.put('\n');
        return 0;
    }

    constexpr std::uint64_t MaxIndex32 = std::numeric_limits<std::uint32_t>::max();
    constexpr std::uint64_t StorePointsLimit = 10000000ull;
    constexpr std::uint64_t StoreRawKeysLimit = 50000000ull;

    if (n64 > MaxIndex32) {
        consumeCoordinates(in, n64);
        writeSequential(n64, out);
        return 0;
    }

    if (n64 <= StorePointsLimit) {
        const std::size_t n = static_cast<std::size_t>(n64);
        std::vector<Point> points;
        points.reserve(n);

        double minX = std::numeric_limits<double>::infinity();
        double maxX = -std::numeric_limits<double>::infinity();
        double minY = std::numeric_limits<double>::infinity();
        double maxY = -std::numeric_limits<double>::infinity();

        for (std::size_t i = 0; i < n; ++i) {
            double x = 0.0;
            double y = 0.0;
            in.readDouble(x);
            in.readDouble(y);
            if (!std::isfinite(x)) {
                x = 0.0;
            }
            if (!std::isfinite(y)) {
                y = 0.0;
            }
            points.push_back(Point{static_cast<float>(x), static_cast<float>(y)});
            minX = std::min(minX, x);
            maxX = std::max(maxX, x);
            minY = std::min(minY, y);
            maxY = std::max(maxY, y);
        }

        std::vector<std::uint32_t> route;
        if (n <= 16) {
            route = exactHeldKarp(points);
        } else if (n <= 2500) {
            route = nearestNeighbor(points);
            const std::size_t window = (n <= 700) ? n : 256;
            twoOptWindow(route, points, window, (n <= 700) ? 8 : 4);
        } else {
            route = spatialRoute(points, minX, maxX, minY, maxY);
            if (n <= 10000) {
                twoOptWindow(route, points, 160, 3);
            } else if (n <= 100000) {
                twoOptWindow(route, points, 64, 2);
            } else if (n <= 1000000) {
                twoOptWindow(route, points, 24, 1);
            } else if (n <= 2000000) {
                twoOptWindow(route, points, 10, 1);
            }
        }

        writeRoute(route, out);
        return 0;
    }

    if (n64 <= StoreRawKeysLimit) {
        const std::size_t n = static_cast<std::size_t>(n64);
        std::vector<KeyIndex> items;
        items.reserve(n);
        for (std::size_t i = 0; i < n; ++i) {
            double x = 0.0;
            double y = 0.0;
            in.readDouble(x);
            in.readDouble(y);
            items.push_back(KeyIndex{rawSpatialKey(static_cast<float>(x), static_cast<float>(y)),
                                     static_cast<std::uint32_t>(i)});
        }
        sortByKey(items);
        writeSortedKeyRoute(items, out);
        return 0;
    }

    consumeCoordinates(in, n64);
    writeSequential(n64, out);
    return 0;
}
