using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                    return -1;
            }
            return _buffer[_ptr++];
        }

        public int ReadInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1)
                    return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32 && c != -1)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    private static int[] _left = Array.Empty<int>();
    private static int[] _right = Array.Empty<int>();
    private static int[] _priority = Array.Empty<int>();
    private static int[] _key = Array.Empty<int>();
    private static int[] _bucketHead = Array.Empty<int>();
    private static int[] _stockNext = Array.Empty<int>();
    private static int[] _stockHead = Array.Empty<int>();
    private static int[] _stockTail = Array.Empty<int>();
    private static int _root = -1;
    private static int _nodeCount;
    private static int _stockCount;
    private static int _stockLength;
    private static uint _rng = 2463534242u;

    private static int NextPriority()
    {
        uint x = _rng;
        x ^= x << 13;
        x ^= x >> 17;
        x ^= x << 5;
        _rng = x;
        return (int)(x & 0x7fffffff);
    }

    private static int NewNode(int key, int stockId)
    {
        int id = _nodeCount++;
        _key[id] = key;
        _priority[id] = NextPriority();
        _left[id] = -1;
        _right[id] = -1;
        _bucketHead[id] = stockId;
        _stockNext[stockId] = -1;
        return id;
    }

    private static int RotateRight(int p)
    {
        int q = _left[p];
        _left[p] = _right[q];
        _right[q] = p;
        return q;
    }

    private static int RotateLeft(int p)
    {
        int q = _right[p];
        _right[p] = _left[q];
        _left[q] = p;
        return q;
    }

    private static int InsertNode(int node, int key, int stockId)
    {
        if (node == -1)
            return NewNode(key, stockId);

        if (key < _key[node])
        {
            _left[node] = InsertNode(_left[node], key, stockId);
            if (_priority[_left[node]] > _priority[node])
                node = RotateRight(node);
        }
        else
        {
            _right[node] = InsertNode(_right[node], key, stockId);
            if (_priority[_right[node]] > _priority[node])
                node = RotateLeft(node);
        }
        return node;
    }

    private static int Merge(int a, int b)
    {
        if (a == -1) return b;
        if (b == -1) return a;
        if (_priority[a] > _priority[b])
        {
            _right[a] = Merge(_right[a], b);
            return a;
        }
        else
        {
            _left[b] = Merge(a, _left[b]);
            return b;
        }
    }

    private static int DeleteNode(int node, int key)
    {
        if (node == -1)
            return -1;

        if (key < _key[node])
        {
            _left[node] = DeleteNode(_left[node], key);
            return node;
        }

        if (key > _key[node])
        {
            _right[node] = DeleteNode(_right[node], key);
            return node;
        }

        return Merge(_left[node], _right[node]);
    }

    private static int FindNode(int key)
    {
        int cur = _root;
        while (cur != -1)
        {
            if (key < _key[cur])
                cur = _left[cur];
            else if (key > _key[cur])
                cur = _right[cur];
            else
                return cur;
        }
        return -1;
    }

    private static int FindCeilingNode(int key)
    {
        int cur = _root;
        int best = -1;
        while (cur != -1)
        {
            if (_key[cur] >= key)
            {
                best = cur;
                cur = _left[cur];
            }
            else
            {
                cur = _right[cur];
            }
        }
        return best;
    }

    private static void AddStockToBucket(int stockId, int remaining)
    {
        if (remaining < 0)
            return;

        int node = FindNode(remaining);
        if (node == -1)
        {
            _root = InsertNode(_root, remaining, stockId);
            return;
        }

        _stockNext[stockId] = _bucketHead[node];
        _bucketHead[node] = stockId;
    }

    private static int TakeStockForCut(int length, out int remaining)
    {
        int node = FindCeilingNode(length);
        if (node == -1)
        {
            remaining = 0;
            return -1;
        }

        int stockId = _bucketHead[node];
        remaining = _key[node] - length;
        _bucketHead[node] = _stockNext[stockId];
        _stockNext[stockId] = -1;
        if (_bucketHead[node] == -1)
            _root = DeleteNode(_root, _key[node]);
        return stockId;
    }

    private static int CreateStock()
    {
        int id = _stockCount++;
        _stockHead[id] = -1;
        _stockTail[id] = -1;
        return id;
    }

    private static void AppendCutToStock(int stockId, int cutIndex, int[] nextCut)
    {
        nextCut[cutIndex] = -1;
        if (_stockHead[stockId] == -1)
        {
            _stockHead[stockId] = cutIndex;
            _stockTail[stockId] = cutIndex;
            return;
        }

        nextCut[_stockTail[stockId]] = cutIndex;
        _stockTail[stockId] = cutIndex;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.ReadInt();
        _stockLength = fs.ReadInt();

        if (n < 0 || _stockLength < 0)
            return;

        int[] lengths = new int[n];
        int[] order = new int[n];
        for (int i = 0; i < n; i++)
        {
            lengths[i] = fs.ReadInt();
            order[i] = i;
        }

        Array.Sort(lengths, order);

        _left = new int[n + 5];
        _right = new int[n + 5];
        _priority = new int[n + 5];
        _key = new int[n + 5];
        _bucketHead = new int[n + 5];
        _stockNext = new int[n + 5];
        _stockHead = new int[n + 5];
        _stockTail = new int[n + 5];

        Array.Fill(_left, -1);
        Array.Fill(_right, -1);
        Array.Fill(_bucketHead, -1);
        Array.Fill(_stockNext, -1);
        Array.Fill(_stockHead, -1);
        Array.Fill(_stockTail, -1);

        int[] nextCut = new int[n];
        Array.Fill(nextCut, -1);

        for (int i = n - 1; i >= 0; i--)
        {
            int length = lengths[i];
            int cutIndex = order[i];

            int remaining;
            int stockId = TakeStockForCut(length, out remaining);
            if (stockId == -1)
            {
                stockId = CreateStock();
                remaining = _stockLength - length;
            }

            AppendCutToStock(stockId, cutIndex, nextCut);
            AddStockToBucket(stockId, remaining);
        }

        var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        output.WriteLine(_stockCount);
        for (int s = 0; s < _stockCount; s++)
        {
            int cur = _stockHead[s];
            bool first = true;
            while (cur != -1)
            {
                if (!first)
                    output.Write(' ');
                output.Write(cur);
                first = false;
                cur = nextCut[cur];
            }
            output.WriteLine();
        }
        output.Flush();
    }
}
