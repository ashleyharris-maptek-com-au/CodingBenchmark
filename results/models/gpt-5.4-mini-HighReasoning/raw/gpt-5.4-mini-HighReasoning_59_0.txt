const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};

const grid = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxSteps = Math.max(0, Number(data.maxStepsPerDrone) || 0);

const H = grid.length;
const W = H > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const N = H * W;

function idx(x, y) {
  return y * W + x;
}

function coord(v) {
  const y = (v / W) | 0;
  const x = v - y * W;
  return [x, y];
}

const values = new Float64Array(N);
const free = new Uint8Array(N);
const positiveCells = [];

for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    const i = idx(x, y);
    values[i] = v;
    if (v !== -1) {
      free[i] = 1;
      if (v > 0) positiveCells.push(i);
    }
  }
}

function bfsDistances(start) {
  const dist = new Int32Array(N);
  dist.fill(-1);
  if (start < 0 || start >= N || !free[start]) return dist;

  const q = new Int32Array(N);
  let head = 0;
  let tail = 0;
  dist[start] = 0;
  q[tail++] = start;

  while (head < tail) {
    const v = q[head++];
    const nextD = dist[v] + 1;
    const y = (v / W) | 0;
    const x = v - y * W;

    if (y > 0) {
      const n = v - W;
      if (free[n] && dist[n] === -1) {
        dist[n] = nextD;
        q[tail++] = n;
      }
    }
    if (y + 1 < H) {
      const n = v + W;
      if (free[n] && dist[n] === -1) {
        dist[n] = nextD;
        q[tail++] = n;
      }
    }
    if (x > 0) {
      const n = v - 1;
      if (free[n] && dist[n] === -1) {
        dist[n] = nextD;
        q[tail++] = n;
      }
    }
    if (x + 1 < W) {
      const n = v + 1;
      if (free[n] && dist[n] === -1) {
        dist[n] = nextD;
        q[tail++] = n;
      }
    }
  }

  return dist;
}

function bfsTree(start) {
  const parent = new Int32Array(N);
  parent.fill(-1);
  const depth = new Int32Array(N);
  depth.fill(-1);
  const order = new Int32Array(N);
  let orderLen = 0;
  const children = new Array(N);

  if (start < 0 || start >= N || !free[start]) {
    return { parent, depth, order, orderLen, children };
  }

  const q = new Int32Array(N);
  let head = 0;
  let tail = 0;
  depth[start] = 0;
  q[tail++] = start;

  while (head < tail) {
    const v = q[head++];
    order[orderLen++] = v;
    const nextD = depth[v] + 1;
    const y = (v / W) | 0;
    const x = v - y * W;

    if (y > 0) {
      const n = v - W;
      if (free[n] && depth[n] === -1) {
        depth[n] = nextD;
        parent[n] = v;
        q[tail++] = n;
        if (!children[v]) children[v] = [];
        children[v].push(n);
      }
    }
    if (y + 1 < H) {
      const n = v + W;
      if (free[n] && depth[n] === -1) {
        depth[n] = nextD;
        parent[n] = v;
        q[tail++] = n;
        if (!children[v]) children[v] = [];
        children[v].push(n);
      }
    }
    if (x > 0) {
      const n = v - 1;
      if (free[n] && depth[n] === -1) {
        depth[n] = nextD;
        parent[n] = v;
        q[tail++] = n;
        if (!children[v]) children[v] = [];
        children[v].push(n);
      }
    }
    if (x + 1 < W) {
      const n = v + 1;
      if (free[n] && depth[n] === -1) {
        depth[n] = nextD;
        parent[n] = v;
        q[tail++] = n;
        if (!children[v]) children[v] = [];
        children[v].push(n);
      }
    }
  }

  return { parent, depth, order, orderLen, children };
}

function buildPathForDrone(startIdx, ownedReward) {
  const tree = bfsTree(startIdx);
  const { parent, depth, order, orderLen, children } = tree;
  const subtreeReward = new Float64Array(N);

  for (let i = 0; i < N; i++) {
    subtreeReward[i] = ownedReward[i];
  }

  for (let i = orderLen - 1; i >= 0; i--) {
    const v = order[i];
    const p = parent[v];
    if (p !== -1) subtreeReward[p] += subtreeReward[v];
  }

  const priority = new Float64Array(N);
  for (let i = 0; i < N; i++) {
    if (subtreeReward[i] > 0 && depth[i] >= 0) {
      priority[i] = subtreeReward[i] / (1 + depth[i]);
    }
  }

  for (let i = 0; i < N; i++) {
    const arr = children[i];
    if (arr && arr.length > 1) {
      arr.sort((a, b) => priority[b] - priority[a]);
    }
  }

  const path = [];
  let current = startIdx;
  path.push(coord(current));

  if (maxSteps === 0 || depth[startIdx] === -1) {
    return path;
  }

  const stackNode = [startIdx];
  const stackNext = [0];
  let moves = 0;

  while (moves < maxSteps && stackNode.length > 0) {
    const top = stackNode.length - 1;
    const node = stackNode[top];
    const arr = children[node];
    let nextIndex = stackNext[top];

    if (arr) {
      while (nextIndex < arr.length && subtreeReward[arr[nextIndex]] <= 0) nextIndex++;
      if (nextIndex < arr.length) {
        const child = arr[nextIndex];
        stackNext[top] = nextIndex + 1;
        stackNode.push(child);
        stackNext.push(0);
        current = child;
        path.push(coord(current));
        moves++;
        continue;
      }
    }

    if (stackNode.length === 1) break;
    stackNode.pop();
    stackNext.pop();
    current = stackNode[stackNode.length - 1];
    path.push(coord(current));
    moves++;
  }

  while (path.length < maxSteps + 1) {
    path.push(coord(current));
  }

  return path;
}

const starts = drones.map(d => idx(d.x | 0, d.y | 0));
const distByDrone = starts.map(s => bfsDistances(s));

const owner = new Int32Array(N);
owner.fill(-1);

for (const cell of positiveCells) {
  const val = values[cell];
  let bestDrone = -1;
  let bestScore = -Infinity;
  let bestDist = Infinity;

  for (let d = 0; d < starts.length; d++) {
    const dist = distByDrone[d][cell];
    if (dist < 0) continue;
    const score = val / (dist + 1);
    if (
      score > bestScore ||
      (score === bestScore && (dist < bestDist || (dist === bestDist && d < bestDrone)))
    ) {
      bestScore = score;
      bestDist = dist;
      bestDrone = d;
    }
  }

  if (bestDrone !== -1) owner[cell] = bestDrone;
}

const paths = [];

for (let d = 0; d < starts.length; d++) {
  const ownedReward = new Float64Array(N);
  for (const cell of positiveCells) {
    if (owner[cell] === d) ownedReward[cell] = values[cell];
  }
  paths.push(buildPathForDrone(starts[d], ownedReward));
}

process.stdout.write(JSON.stringify({ paths }));
