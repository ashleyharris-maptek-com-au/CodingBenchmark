#include <bits/stdc++.h>
using namespace std;

struct Solution {
    long long weight = 0;
    vector<char> choose;
};

struct Solver {
    int n = 0;
    int W = 0;
    vector<long long> w;
    vector<int> baseDeg;
    vector<vector<int>> adjList;
    vector<vector<uint64_t>> adjBits;

    static constexpr int SMALL = 18;

    inline bool edge(int u, int v) const {
        return (adjBits[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    inline bool bit_test(const vector<uint64_t>& bs, int v) const {
        return (bs[v >> 6] >> (v & 63)) & 1ULL;
    }

    inline void bit_and(vector<uint64_t>& a, const vector<uint64_t>& b) const {
        for (int i = 0; i < W; ++i) a[i] &= b[i];
    }

    vector<vector<int>> components(const vector<int>& verts) const {
        vector<char> present(n, 0), vis(n, 0);
        for (int v : verts) present[v] = 1;
        vector<vector<int>> comps;
        queue<int> q;
        for (int s : verts) {
            if (vis[s]) continue;
            vis[s] = 1;
            q.push(s);
            vector<int> comp;
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                comp.push_back(v);
                for (int u : adjList[v]) {
                    if (present[u] && !vis[u]) {
                        vis[u] = 1;
                        q.push(u);
                    }
                }
            }
            comps.push_back(move(comp));
        }
        return comps;
    }

    Solution brute_force(const vector<int>& verts) const {
        int k = (int)verts.size();
        Solution best;
        best.choose.assign(n, 0);
        if (k == 0) return best;
        vector<uint64_t> conf(k, 0);
        for (int i = 0; i < k; ++i) {
            for (int j = i + 1; j < k; ++j) {
                if (edge(verts[i], verts[j])) {
                    conf[i] |= (1ULL << j);
                    conf[j] |= (1ULL << i);
                }
            }
        }
        int limit = 1 << k;
        for (int mask = 0; mask < limit; ++mask) {
            long long sum = 0;
            bool ok = true;
            for (int i = 0; i < k; ++i) {
                if (mask & (1 << i)) {
                    if (conf[i] & (uint64_t)mask) {
                        ok = false;
                        break;
                    }
                    sum += w[verts[i]];
                }
            }
            if (ok && sum > best.weight) {
                best.weight = sum;
                fill(best.choose.begin(), best.choose.end(), 0);
                for (int i = 0; i < k; ++i) {
                    if (mask & (1 << i)) best.choose[verts[i]] = 1;
                }
            }
        }
        return best;
    }

    Solution greedy_solution(const vector<int>& verts) const {
        Solution sol;
        sol.choose.assign(n, 0);
        vector<char> chosen(n, 0);
        vector<int> order = verts;
        sort(order.begin(), order.end(), [&](int a, int b) {
            long double sa = (long double)w[a] / (baseDeg[a] + 1);
            long double sb = (long double)w[b] / (baseDeg[b] + 1);
            if (sa != sb) return sa > sb;
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });
        for (int v : order) {
            bool ok = true;
            for (int u : adjList[v]) {
                if (chosen[u]) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                chosen[v] = 1;
                sol.choose[v] = 1;
                sol.weight += w[v];
            }
        }
        return sol;
    }

    long long upper_bound(const vector<int>& verts) const {
        if (verts.empty()) return 0;
        vector<int> order = verts;
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (baseDeg[a] != baseDeg[b]) return baseDeg[a] > baseDeg[b];
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });
        struct Clique {
            vector<uint64_t> can;
            long long mx = 0;
        };
        vector<Clique> cliques;
        for (int v : order) {
            int best = -1;
            long long bestMx = LLONG_MAX;
            for (int i = 0; i < (int)cliques.size(); ++i) {
                if (bit_test(cliques[i].can, v)) {
                    if (cliques[i].mx >= w[v]) {
                        best = i;
                        break;
                    }
                    if (cliques[i].mx < bestMx) {
                        bestMx = cliques[i].mx;
                        best = i;
                    }
                }
            }
            if (best == -1) {
                cliques.push_back({adjBits[v], w[v]});
            } else {
                cliques[best].mx = max(cliques[best].mx, w[v]);
                bit_and(cliques[best].can, adjBits[v]);
            }
        }
        long long sum = 0;
        for (const auto& c : cliques) sum += c.mx;
        return sum;
    }

    int choose_pivot(const vector<int>& verts) const {
        vector<char> present(n, 0);
        for (int v : verts) present[v] = 1;
        int best = verts[0];
        int bestDeg = -1;
        for (int v : verts) {
            int d = 0;
            for (int u : adjList[v]) {
                if (present[u]) ++d;
            }
            if (d > bestDeg || (d == bestDeg && w[v] > w[best])) {
                bestDeg = d;
                best = v;
            }
        }
        return best;
    }

    void search(const vector<int>& verts, long long curW, vector<char>& chosen, Solution& best) const;

    Solution solve_set(const vector<int>& verts) const {
        if (verts.empty()) {
            Solution empty;
            empty.choose.assign(n, 0);
            return empty;
        }
        if ((int)verts.size() == 1) {
            Solution sol;
            sol.choose.assign(n, 0);
            sol.weight = w[verts[0]];
            sol.choose[verts[0]] = 1;
            return sol;
        }
        auto comps = components(verts);
        if ((int)comps.size() > 1) {
            Solution res;
            res.choose.assign(n, 0);
            for (const auto& comp : comps) {
                Solution sub = solve_set(comp);
                res.weight += sub.weight;
                for (int v : comp) {
                    if (sub.choose[v]) res.choose[v] = 1;
                }
            }
            return res;
        }
        if ((int)verts.size() <= SMALL) {
            return brute_force(verts);
        }

        Solution best = greedy_solution(verts);
        vector<char> chosen(n, 0);
        search(verts, 0, chosen, best);
        return best;
    }

    void search(const vector<int>& verts, long long curW, vector<char>& chosen, Solution& best) const {
        if (verts.empty()) {
            if (curW > best.weight) {
                best.weight = curW;
                best.choose = chosen;
            }
            return;
        }

        auto comps = components(verts);
        if ((int)comps.size() > 1) {
            long long total = curW;
            vector<char> merged = chosen;
            for (const auto& comp : comps) {
                Solution sub = solve_set(comp);
                total += sub.weight;
                for (int v : comp) {
                    if (sub.choose[v]) merged[v] = 1;
                }
            }
            if (total > best.weight) {
                best.weight = total;
                best.choose = move(merged);
            }
            return;
        }

        if ((int)verts.size() <= SMALL) {
            Solution sub = brute_force(verts);
            long long total = curW + sub.weight;
            if (total > best.weight) {
                best.weight = total;
                best.choose = chosen;
                for (int v : verts) {
                    if (sub.choose[v]) best.choose[v] = 1;
                }
            }
            return;
        }

        long long ub = curW + upper_bound(verts);
        if (ub <= best.weight) return;

        int v = choose_pivot(verts);

        vector<int> inc;
        inc.reserve(verts.size());
        for (int u : verts) {
            if (u != v && !edge(v, u)) inc.push_back(u);
        }
        chosen[v] = 1;
        search(inc, curW + w[v], chosen, best);
        chosen[v] = 0;

        vector<int> exc;
        exc.reserve(verts.size() - 1);
        for (int u : verts) {
            if (u != v) exc.push_back(u);
        }
        search(exc, curW, chosen, best);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    Solver solver;
    solver.n = n;
    solver.W = (n + 63) / 64;
    solver.w.assign(n, 0);
    solver.baseDeg.assign(n, 0);
    solver.adjList.assign(n, {});
    solver.adjBits.assign(n, vector<uint64_t>(solver.W, 0));

    for (int i = 0; i < n; ++i) cin >> solver.w[i];

    vector<pair<int, int>> edges(m);
    for (int i = 0; i < m; ++i) cin >> edges[i].first >> edges[i].second;

    bool zeroBasedPossible = true, oneBasedPossible = true;
    for (auto [u, v] : edges) {
        if (u < 0 || u >= n || v < 0 || v >= n) zeroBasedPossible = false;
        if (u < 1 || u > n || v < 1 || v > n) oneBasedPossible = false;
    }
    int offset = 0;
    if (zeroBasedPossible && !oneBasedPossible) offset = 0;
    else if (oneBasedPossible && !zeroBasedPossible) offset = -1;
    else offset = 0;

    for (auto [ru, rv] : edges) {
        int u = ru + offset;
        int v = rv + offset;
        if (u == v) continue;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (!solver.edge(u, v)) {
            solver.adjList[u].push_back(v);
            solver.adjList[v].push_back(u);
            solver.adjBits[u][v >> 6] |= 1ULL << (v & 63);
            solver.adjBits[v][u >> 6] |= 1ULL << (u & 63);
            ++solver.baseDeg[u];
            ++solver.baseDeg[v];
        }
    }

    vector<int> verts;
    verts.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (solver.w[i] > 0) verts.push_back(i);
    }

    Solution ans = solver.solve_set(verts);
    if ((int)ans.choose.size() != n) ans.choose.assign(n, 0);

    cout << ans.weight << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << (ans.choose[i] ? 1 : 0);
    }
    cout << '\n';
    return 0;
}
