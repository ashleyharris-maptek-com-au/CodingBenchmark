declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const W: number = data.width | 0;
const H: number = data.height | 0;
const N: number = W * H;
const grid: string[] = data.grid || [];
const towns: Array<{ id: any; x: number; y: number; value: number }> = data.towns || [];
const ignitionsInput: Array<[number, number]> = data.ignitions || [];
const cutBudget: number = data.cutBudget | 0;

const burnable = new Uint8Array(N);
const isTown = new Uint8Array(N);
const isIgnition = new Uint8Array(N);
const townValue = new Float64Array(N);

for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  const base = y * W;
  for (let x = 0; x < W; x++) {
    const idx = base + x;
    burnable[idx] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

for (const t of towns) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  const idx = y * W + x;
  isTown[idx] = 1;
  townValue[idx] += Number(t.value) || 0;
}

const ignitionCells: number[] = [];
for (const p of ignitionsInput) {
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  const idx = y * W + x;
  if (!burnable[idx] || isIgnition[idx]) continue;
  isIgnition[idx] = 1;
  ignitionCells.push(idx);
}

const eligible = new Uint8Array(N);
for (let i = 0; i < N; i++) {
  eligible[i] = burnable[i] && !isIgnition[i] && !isTown[i] ? 1 : 0;
}

const dist = new Int32Array(N);
dist.fill(-1);
const bfsParent = new Int32Array(N);
bfsParent.fill(-1);
const bfsOrder = new Int32Array(N);
let bfsLen = 0;
const queue = new Int32Array(N);
let qh = 0;
let qt = 0;

for (let i = 0; i < ignitionCells.length; i++) {
  const s = ignitionCells[i];
  if (dist[s] !== -1) continue;
  dist[s] = 0;
  queue[qt++] = s;
}

while (qh < qt) {
  const u = queue[qh++];
  bfsOrder[bfsLen++] = u;
  const d = dist[u] + 1;
  const x = u % W;
  const y = (u / W) | 0;

  if (x > 0) {
    const v = u - 1;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = d;
      bfsParent[v] = u;
      queue[qt++] = v;
    }
  }
  if (x + 1 < W) {
    const v = u + 1;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = d;
      bfsParent[v] = u;
      queue[qt++] = v;
    }
  }
  if (y > 0) {
    const v = u - W;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = d;
      bfsParent[v] = u;
      queue[qt++] = v;
    }
  }
  if (y + 1 < H) {
    const v = u + W;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = d;
      bfsParent[v] = u;
      queue[qt++] = v;
    }
  }
}

const downstreamTown = new Float64Array(N);
for (let i = 0; i < bfsLen; i++) {
  const u = bfsOrder[i];
  downstreamTown[u] = townValue[u];
}
for (let i = bfsLen - 1; i >= 0; i--) {
  const u = bfsOrder[i];
  const p = bfsParent[u];
  if (p >= 0) downstreamTown[p] += downstreamTown[u];
}

const reachableDeg = new Uint8Array(N);
for (let i = 0; i < bfsLen; i++) {
  const u = bfsOrder[i];
  const x = u % W;
  const y = (u / W) | 0;
  let deg = 0;
  if (x > 0) {
    const v = u - 1;
    if (dist[v] !== -1) deg++;
  }
  if (x + 1 < W) {
    const v = u + 1;
    if (dist[v] !== -1) deg++;
  }
  if (y > 0) {
    const v = u - W;
    if (dist[v] !== -1) deg++;
  }
  if (y + 1 < H) {
    const v = u + W;
    if (dist[v] !== -1) deg++;
  }
  reachableDeg[u] = deg;
}

const disc = new Int32Array(N);
const low = new Int32Array(N);
const parent = new Int32Array(N);
parent.fill(-1);
const firstChild = new Int32Array(N);
firstChild.fill(-1);
const nextSibling = new Int32Array(N);
nextSibling.fill(-1);
const dfsOrder = new Int32Array(N);
const dfsPos = new Int32Array(N);
dfsPos.fill(-1);
let dfsLen = 0;
let timer = 0;

const stackNode = new Int32Array(N);
const stackNext = new Uint8Array(N);

for (let i = 0; i < ignitionCells.length; i++) {
  const start = ignitionCells[i];
  if (dist[start] === -1 || disc[start] !== 0) continue;
  let sp = 0;
  stackNode[0] = start;
  stackNext[0] = 0;
  parent[start] = -1;
  disc[start] = -1;

  while (sp >= 0) {
    const u = stackNode[sp];
    if (disc[u] === -1) {
      disc[u] = ++timer;
      low[u] = disc[u];
      dfsPos[u] = dfsLen;
      dfsOrder[dfsLen++] = u;
    }

    const step = stackNext[sp];
    const x = u % W;
    const y = (u / W) | 0;

    if (step === 0) {
      stackNext[sp] = 1;
      if (x > 0) {
        const v = u - 1;
        if (dist[v] !== -1) {
          if (disc[v] === 0) {
            parent[v] = u;
            nextSibling[v] = firstChild[u];
            firstChild[u] = v;
            disc[v] = -1;
            sp++;
            stackNode[sp] = v;
            stackNext[sp] = 0;
          } else if (disc[v] > 0 && v !== parent[u] && disc[v] < low[u]) {
            low[u] = disc[v];
          }
        }
      }
    } else if (step === 1) {
      stackNext[sp] = 2;
      if (x + 1 < W) {
        const v = u + 1;
        if (dist[v] !== -1) {
          if (disc[v] === 0) {
            parent[v] = u;
            nextSibling[v] = firstChild[u];
            firstChild[u] = v;
            disc[v] = -1;
            sp++;
            stackNode[sp] = v;
            stackNext[sp] = 0;
          } else if (disc[v] > 0 && v !== parent[u] && disc[v] < low[u]) {
            low[u] = disc[v];
          }
        }
      }
    } else if (step === 2) {
      stackNext[sp] = 3;
      if (y > 0) {
        const v = u - W;
        if (dist[v] !== -1) {
          if (disc[v] === 0) {
            parent[v] = u;
            nextSibling[v] = firstChild[u];
            firstChild[u] = v;
            disc[v] = -1;
            sp++;
            stackNode[sp] = v;
            stackNext[sp] = 0;
          } else if (disc[v] > 0 && v !== parent[u] && disc[v] < low[u]) {
            low[u] = disc[v];
          }
        }
      }
    } else if (step === 3) {
      stackNext[sp] = 4;
      if (y + 1 < H) {
        const v = u + W;
        if (dist[v] !== -1) {
          if (disc[v] === 0) {
            parent[v] = u;
            nextSibling[v] = firstChild[u];
            firstChild[u] = v;
            disc[v] = -1;
            sp++;
            stackNode[sp] = v;
            stackNext[sp] = 0;
          } else if (disc[v] > 0 && v !== parent[u] && disc[v] < low[u]) {
            low[u] = disc[v];
          }
        }
      }
    } else {
      sp--;
      const p = parent[u];
      if (p >= 0 && low[u] < low[p]) low[p] = low[u];
    }
  }
}

const subtreeSize = new Int32Array(N);
const subtreeIgn = new Int32Array(N);
const subtreeTown = new Float64Array(N);
for (let i = 0; i < dfsLen; i++) {
  const u = dfsOrder[i];
  subtreeSize[u] = 1;
  subtreeIgn[u] = isIgnition[u] ? 1 : 0;
  subtreeTown[u] = townValue[u];
}
for (let i = dfsLen - 1; i >= 0; i--) {
  const u = dfsOrder[i];
  const p = parent[u];
  if (p >= 0) {
    subtreeSize[p] += subtreeSize[u];
    subtreeIgn[p] += subtreeIgn[u];
    subtreeTown[p] += subtreeTown[u];
  }
}

const exactGain = new Float64Array(N);
for (let i = 0; i < dfsLen; i++) {
  const u = dfsOrder[i];
  const p = parent[u];
  if (p >= 0 && eligible[p] && low[u] >= disc[p] && subtreeIgn[u] === 0) {
    exactGain[p] += subtreeTown[u];
  }
}

class Fenwick {
  n: number;
  bit: Float64Array;
  constructor(n: number) {
    this.n = n;
    this.bit = new Float64Array(n + 2);
  }
  add(idx: number, delta: number): void {
    for (let i = idx + 1; i <= this.n + 1; i += i & -i) this.bit[i] += delta;
  }
  rangeAdd(l: number, r: number, delta: number): void {
    if (l > r) return;
    this.add(l, delta);
    this.add(r + 1, -delta);
  }
  point(idx: number): number {
    let sum = 0;
    for (let i = idx + 1; i > 0; i -= i & -i) sum += this.bit[i];
    return sum;
  }
}

const chosen = new Uint8Array(N);
const firebreaks: number[] = [];
const cover = new Fenwick(Math.max(0, dfsLen));

const exactCandidates: number[] = [];
for (let i = 0; i < dfsLen; i++) {
  const u = dfsOrder[i];
  if (eligible[u] && exactGain[u] > 0) exactCandidates.push(u);
}

exactCandidates.sort((a, b) => {
  const da = exactGain[a];
  const db = exactGain[b];
  if (db !== da) return db - da;
  const ea = downstreamTown[a];
  const eb = downstreamTown[b];
  if (eb !== ea) return eb - ea;
  return dist[a] - dist[b];
});

function currentExactBenefit(u: number): number {
  let gain = 0;
  let child = firstChild[u];
  while (child !== -1) {
    if (low[child] >= disc[u] && subtreeIgn[child] === 0 && cover.point(dfsPos[child]) === 0) {
      gain += subtreeTown[child];
    }
    child = nextSibling[child];
  }
  return gain;
}

for (let i = 0; i < exactCandidates.length && firebreaks.length < cutBudget; i++) {
  const u = exactCandidates[i];
  if (chosen[u]) continue;
  if (cover.point(dfsPos[u]) !== 0) continue;
  const gain = currentExactBenefit(u);
  if (gain <= 0) continue;
  chosen[u] = 1;
  firebreaks.push(u);

  let child = firstChild[u];
  while (child !== -1) {
    if (low[child] >= disc[u] && subtreeIgn[child] === 0 && cover.point(dfsPos[child]) === 0) {
      const l = dfsPos[child];
      const r = l + subtreeSize[child] - 1;
      cover.rangeAdd(l, r, 1);
    }
    child = nextSibling[child];
  }
}

const fallbackCandidates: number[] = [];
const fallbackScore = new Float64Array(N);
for (let i = 0; i < bfsLen; i++) {
  const u = bfsOrder[i];
  if (!eligible[u] || chosen[u]) continue;
  if (cover.point(dfsPos[u]) !== 0) continue;
  const score = downstreamTown[u] * (5 - reachableDeg[u]) / (1 + dist[u]);
  if (score > 0) {
    fallbackScore[u] = score;
    fallbackCandidates.push(u);
  }
}

fallbackCandidates.sort((a, b) => {
  const sa = fallbackScore[a];
  const sb = fallbackScore[b];
  if (sb !== sa) return sb - sa;
  const ta = downstreamTown[a];
  const tb = downstreamTown[b];
  if (tb !== ta) return tb - ta;
  return dist[a] - dist[b];
});

for (let i = 0; i < fallbackCandidates.length && firebreaks.length < cutBudget; i++) {
  const u = fallbackCandidates[i];
  if (chosen[u]) continue;
  if (cover.point(dfsPos[u]) !== 0) continue;
  chosen[u] = 1;
  firebreaks.push(u);
}

const output = {
  firebreaks: firebreaks.map((u) => [u % W, (u / W) | 0]),
};

process.stdout.write(JSON.stringify(output));
