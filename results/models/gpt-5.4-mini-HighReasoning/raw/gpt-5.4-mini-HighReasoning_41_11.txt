; SPIR-V
; Version: 1.0
; Fragment shader with UV polka dots and Lambert shading
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%float_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.100000001
%float_0_3 = OpConstant %float 0.300000012
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.699999988
%float_0_8 = OpConstant %float 0.800000012
%float_1 = OpConstant %float 1
%float_6 = OpConstant %float 6
%int_3 = OpConstant %int 3

%yellow = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3
%v2float_0_5 = OpConstantComposite %v2float %float_0_5 %float_0_5

%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %funct
%label = OpLabel

%inWorldPos = OpLoad %v3float %worldPos
%inNormal = OpLoad %v3float %normal
%inUV = OpLoad %v2float %uv

%nrm = OpExtInst %v3float %glsl450 Normalize %inNormal

%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos = OpLoad %v4float %lightPosPtr
%lightPosXYZ = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%lightVec = OpFSub %v3float %lightPosXYZ %inWorldPos
%lightDir = OpExtInst %v3float %glsl450 Normalize %lightVec

%dotNL = OpExtInst %float %glsl450 Dot %nrm %lightDir
%diffuse = OpExtInst %float %glsl450 FMax %dotNL %float_0
%shadeMul = OpFMul %float %float_0_7 %diffuse
%shade = OpFAdd %float %float_0_3 %shadeMul

%uvScaled = OpVectorTimesScalar %v2float %inUV %float_6
%cell = OpExtInst %v2float %glsl450 Fract %uvScaled
%offset = OpFSub %v2float %cell %v2float_0_5
%dist = OpExtInst %float %glsl450 Length %offset
%isDot = OpFOrdLessThanEqual %bool %dist %float_0_3
%baseColor = OpSelect %v3float %isDot %yellow %blue
%litColor = OpVectorTimesScalar %v3float %baseColor %shade
%finalColor = OpCompositeConstruct %v4float %litColor %float_1

OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
