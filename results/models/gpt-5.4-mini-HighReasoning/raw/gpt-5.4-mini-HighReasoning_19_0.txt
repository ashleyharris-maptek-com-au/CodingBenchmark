using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const int Inf = int.MaxValue / 4;
    private const ushort Empty = ushort.MaxValue;

    private struct Candidate
    {
        public int Idx;
        public int Dist;
        public long Score;
    }

    private struct Operation
    {
        public int Idx;
        public ushort Z;
        public int[] Path;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len;
        private int _pos;

        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_pos++];
        }

        public string NextToken()
        {
            int c;
            do
            {
                c = Read();
            } while (c >= 0 && c <= 32);

            if (c < 0)
            {
                return null;
            }

            var sb = new StringBuilder();
            while (c >= 0 && c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }

            return sb.ToString();
        }

        public int ReadRawByte()
        {
            return Read();
        }
    }

    private sealed class MinHeap
    {
        private int[] _nodes;
        private int[] _dists;
        private int _count;

        public MinHeap(int capacity = 1024)
        {
            _nodes = new int[Math.Max(4, capacity)];
            _dists = new int[_nodes.Length];
        }

        public int Count => _count;

        public void Push(int node, int dist)
        {
            if (_count == _nodes.Length)
            {
                int newSize = _nodes.Length << 1;
                Array.Resize(ref _nodes, newSize);
                Array.Resize(ref _dists, newSize);
            }

            int i = _count++;
            while (i > 0)
            {
                int parent = (i - 1) >> 1;
                if (_dists[parent] <= dist)
                {
                    break;
                }

                _nodes[i] = _nodes[parent];
                _dists[i] = _dists[parent];
                i = parent;
            }

            _nodes[i] = node;
            _dists[i] = dist;
        }

        public bool TryPop(out int node, out int dist)
        {
            if (_count == 0)
            {
                node = 0;
                dist = 0;
                return false;
            }

            node = _nodes[0];
            dist = _dists[0];
            int lastNode = _nodes[--_count];
            int lastDist = _dists[_count];

            int i = 0;
            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= _count)
                {
                    break;
                }

                int right = left + 1;
                int child = right < _count && _dists[right] < _dists[left] ? right : left;
                if (_dists[child] >= lastDist)
                {
                    break;
                }

                _nodes[i] = _nodes[child];
                _dists[i] = _dists[child];
                i = child;
            }

            if (_count > 0)
            {
                _nodes[i] = lastNode;
                _dists[i] = lastDist;
            }

            return true;
        }
    }

    private sealed class TopK
    {
        private readonly Candidate[] _data;
        private int _count;

        public TopK(int capacity)
        {
            _data = new Candidate[Math.Max(1, capacity)];
        }

        public int Count => _count;

        public void Add(Candidate c)
        {
            if (_data.Length == 0)
            {
                return;
            }

            if (_count < _data.Length)
            {
                int i = _count++;
                while (i > 0)
                {
                    int parent = (i - 1) >> 1;
                    if (LessOrEqual(_data[parent], c))
                    {
                        break;
                    }

                    _data[i] = _data[parent];
                    i = parent;
                }

                _data[i] = c;
                return;
            }

            if (!LessOrEqual(_data[0], c))
            {
                return;
            }

            _data[0] = c;
            SiftDown(0);
        }

        public Candidate[] ToArray()
        {
            var result = new Candidate[_count];
            Array.Copy(_data, result, _count);
            return result;
        }

        private static bool LessOrEqual(in Candidate a, in Candidate b)
        {
            if (a.Score != b.Score) return a.Score <= b.Score;
            if (a.Dist != b.Dist) return a.Dist <= b.Dist;
            return a.Idx <= b.Idx;
        }

        private void SiftDown(int i)
        {
            Candidate x = _data[i];
            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= _count)
                {
                    break;
                }

                int right = left + 1;
                int child = right < _count && LessOrEqual(_data[right], _data[left]) ? right : left;
                if (!LessOrEqual(x, _data[child]))
                {
                    break;
                }

                _data[i] = _data[child];
                i = child;
            }

            _data[i] = x;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();

        int xCount = int.Parse(fs.NextToken());
        int yCount = int.Parse(fs.NextToken());
        int zCount = int.Parse(fs.NextToken());
        fs.NextToken(); // XBlockSize, not used
        fs.NextToken(); // YBlockSize, not used
        fs.NextToken(); // ZBlockSize, not used

        long xyLong = (long)xCount * yCount;
        if (xyLong > int.MaxValue)
        {
            throw new InvalidOperationException("Surface map too large.");
        }

        int xy = (int)xyLong;
        ushort[] topZ = new ushort[xy];
        byte[] topRaw = new byte[xy];
        for (int i = 0; i < xy; i++)
        {
            topZ[i] = Empty;
        }

        for (int x = 0; x < xCount; x++)
        {
            int baseIdx = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                int idx = baseIdx + y;
                ushort bestZ = Empty;
                byte bestRaw = 0;
                for (int z = 0; z < zCount; z++)
                {
                    int b = fs.ReadRawByte();
                    if (b < 0)
                    {
                        b = 0;
                    }

                    if (b != 0)
                    {
                        bestZ = (ushort)z;
                        bestRaw = (byte)b;
                    }
                }

                topZ[idx] = bestZ;
                topRaw[idx] = bestRaw;
            }
        }

        int plantIdx = 0;
        if (topZ[plantIdx] == Empty)
        {
            topZ[plantIdx] = 0;
        }

        bool useGlobalPathing = xy <= 2_000_000;
        List<Operation> operations;

        if (useGlobalPathing)
        {
            operations = SolveWithGlobalPathing(xCount, yCount, topZ, topRaw, plantIdx);
        }
        else
        {
            operations = SolveWithLocalHeuristic(xCount, yCount, topZ, topRaw, plantIdx);
        }

        var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        output.WriteLine(operations.Count);
        for (int i = 0; i < operations.Count; i++)
        {
            WriteOperation(output, operations[i], yCount);
        }
        output.Flush();
    }

    private static List<Operation> SolveWithGlobalPathing(int xCount, int yCount, ushort[] topZ, byte[] topRaw, int plantIdx)
    {
        int xy = topZ.Length;
        int[] dist = new int[xy];
        int[] parent = new int[xy];
        for (int i = 0; i < xy; i++)
        {
            dist[i] = Inf;
            parent[i] = -1;
        }

        var heap = new MinHeap(Math.Min(1024, xy));
        dist[plantIdx] = 0;
        parent[plantIdx] = plantIdx;
        heap.Push(plantIdx, 0);

        while (heap.TryPop(out int cur, out int d))
        {
            if (d != dist[cur])
            {
                continue;
            }

            int curH = topZ[cur];
            int x = cur / yCount;
            int y = cur - x * yCount;

            // 4-connected expansion on the surface height field.
            RelaxGlobal(xCount, yCount, topZ, dist, parent, heap, cur, curH, d, x - 1, y);
            RelaxGlobal(xCount, yCount, topZ, dist, parent, heap, cur, curH, d, x + 1, y);
            RelaxGlobal(xCount, yCount, topZ, dist, parent, heap, cur, curH, d, x, y - 1);
            RelaxGlobal(xCount, yCount, topZ, dist, parent, heap, cur, curH, d, x, y + 1);
        }

        int cap = 2000;
        var topK = new TopK(cap);
        for (int idx = 0; idx < xy; idx++)
        {
            if (idx == plantIdx)
            {
                continue;
            }

            byte raw = topRaw[idx];
            if (raw == 0)
            {
                continue;
            }

            int mass = raw & 15;
            int value = raw >> 4;
            if (mass == 0 || value == 0)
            {
                continue;
            }

            int d = dist[idx];
            if (d >= Inf)
            {
                continue;
            }

            long netPerTon = 1000L * value - d;
            if (netPerTon <= 0)
            {
                continue;
            }

            topK.Add(new Candidate
            {
                Idx = idx,
                Dist = d,
                Score = netPerTon * mass
            });
        }

        Candidate[] selected = topK.ToArray();
        Array.Sort(selected, CompareByDistanceThenScoreDesc);

        var usedDigs = new HashSet<int>();
        var ops = new List<Operation>(selected.Length);
        for (int i = 0; i < selected.Length; i++)
        {
            if (!TryBuildPathFromParent(selected[i].Idx, parent, plantIdx, usedDigs, out int[] path))
            {
                continue;
            }

            ushort z = topZ[selected[i].Idx];
            if (z == Empty)
            {
                continue;
            }

            usedDigs.Add(selected[i].Idx);
            ops.Add(new Operation
            {
                Idx = selected[i].Idx,
                Z = z,
                Path = path
            });
        }

        return ops;
    }

    private static List<Operation> SolveWithLocalHeuristic(int xCount, int yCount, ushort[] topZ, byte[] topRaw, int plantIdx)
    {
        int xy = topZ.Length;
        int plantHeight = topZ[plantIdx];
        if (plantHeight == Empty)
        {
            plantHeight = 0;
        }

        int radiusX = Math.Min(xCount, 128);
        int radiusY = Math.Min(yCount, 128);
        var topK = new TopK(200);

        for (int x = 0; x < radiusX; x++)
        {
            int baseIdx = x * yCount;
            for (int y = 0; y < radiusY; y++)
            {
                int idx = baseIdx + y;
                if (idx == plantIdx)
                {
                    continue;
                }

                byte raw = topRaw[idx];
                if (raw == 0)
                {
                    continue;
                }

                int mass = raw & 15;
                int value = raw >> 4;
                if (mass == 0 || value == 0)
                {
                    continue;
                }

                bool found = false;
                int bestCost = 0;
                int[] bestPath = null;

                if (TryBuildDirectRoute(xCount, yCount, topZ, plantHeight, x, y, 0, out int[] path0, out int cost0))
                {
                    found = true;
                    bestCost = cost0;
                    bestPath = path0;
                }

                if (TryBuildDirectRoute(xCount, yCount, topZ, plantHeight, x, y, 1, out int[] path1, out int cost1))
                {
                    if (!found || cost1 < bestCost)
                    {
                        found = true;
                        bestCost = cost1;
                        bestPath = path1;
                    }
                }

                if (found)
                {
                    long netPerTon = 1000L * value - bestCost;
                    if (netPerTon > 0)
                    {
                        topK.Add(new Candidate
                        {
                            Idx = idx,
                            Dist = bestCost,
                            Score = netPerTon * mass
                        });
                    }
                }
            }
        }

        Candidate[] selected = topK.ToArray();
        Array.Sort(selected, CompareByDistanceThenScoreDesc);

        var usedDigs = new HashSet<int>();
        var ops = new List<Operation>(selected.Length);
        for (int i = 0; i < selected.Length; i++)
        {
            int idx = selected[i].Idx;
            int sx = idx / yCount;
            int sy = idx - sx * yCount;

            if (usedDigs.Contains(idx))
            {
                continue;
            }

            if (!TryBuildDirectRoute(xCount, yCount, topZ, plantHeight, sx, sy, 0, out int[] path, out int cost))
            {
                if (!TryBuildDirectRoute(xCount, yCount, topZ, plantHeight, sx, sy, 1, out path, out cost))
                {
                    continue;
                }
            }

            bool blocked = false;
            for (int p = 1; p + 1 < path.Length; p++)
            {
                if (usedDigs.Contains(path[p]))
                {
                    blocked = true;
                    break;
                }
            }

            if (blocked)
            {
                continue;
            }

            usedDigs.Add(idx);
            ops.Add(new Operation
            {
                Idx = idx,
                Z = topZ[idx],
                Path = path
            });
        }

        return ops;
    }

    private static int CompareByDistanceThenScoreDesc(Candidate a, Candidate b)
    {
        int c = b.Dist.CompareTo(a.Dist);
        if (c != 0) return c;
        c = b.Score.CompareTo(a.Score);
        if (c != 0) return c;
        return a.Idx.CompareTo(b.Idx);
    }

    private static void RelaxGlobal(
        int xCount,
        int yCount,
        ushort[] topZ,
        int[] dist,
        int[] parent,
        MinHeap heap,
        int cur,
        int curH,
        int curDist,
        int nx,
        int ny)
    {
        if ((uint)nx >= (uint)xCount || (uint)ny >= (uint)yCount)
        {
            return;
        }

        int nidx = nx * yCount + ny;
        ushort nh = topZ[nidx];
        if (nh == Empty)
        {
            return;
        }

        int delta = curH - nh;
        if (delta < -1 || delta > 1)
        {
            return;
        }

        int cost = delta == 1 ? 5 : (delta == 0 ? 1 : 0);
        int nd = curDist + cost;
        if (nd < dist[nidx])
        {
            dist[nidx] = nd;
            parent[nidx] = cur;
            heap.Push(nidx, nd);
        }
    }

    private static bool TryBuildPathFromParent(int startIdx, int[] parent, int plantIdx, HashSet<int> usedDigs, out int[] path)
    {
        var tmp = new List<int>(64);
        int cur = startIdx;
        tmp.Add(cur);

        while (cur != plantIdx)
        {
            cur = parent[cur];
            if (cur < 0)
            {
                path = null;
                return false;
            }

            tmp.Add(cur);
            if (tmp.Count > 10_000_000)
            {
                path = null;
                return false;
            }
        }

        for (int i = 1; i + 1 < tmp.Count; i++)
        {
            if (usedDigs.Contains(tmp[i]))
            {
                path = null;
                return false;
            }
        }

        path = tmp.ToArray();
        return true;
    }

    private static bool TryBuildDirectRoute(
        int xCount,
        int yCount,
        ushort[] topZ,
        int plantHeight,
        int startX,
        int startY,
        int order,
        out int[] path,
        out int cost)
    {
        int curX = startX;
        int curY = startY;
        int curIdx = curX * yCount + curY;
        int curH = topZ[curIdx];
        if (curH == Empty)
        {
            path = null;
            cost = 0;
            return false;
        }

        var list = new List<int>(startX + startY + 2);
        list.Add(curIdx);
        cost = 0;

        if (order == 0)
        {
            if (!WalkAxis(xCount, yCount, topZ, plantHeight, ref curX, ref curY, ref curH, 0, ref cost, list)) { path = null; return false; }
            if (!WalkAxis(xCount, yCount, topZ, plantHeight, ref curX, ref curY, ref curH, 1, ref cost, list)) { path = null; return false; }
        }
        else
        {
            if (!WalkAxis(xCount, yCount, topZ, plantHeight, ref curX, ref curY, ref curH, 1, ref cost, list)) { path = null; return false; }
            if (!WalkAxis(xCount, yCount, topZ, plantHeight, ref curX, ref curY, ref curH, 0, ref cost, list)) { path = null; return false; }
        }

        if (curX != 0 || curY != 0)
        {
            path = null;
            return false;
        }

        path = list.ToArray();
        return true;
    }

    private static bool WalkAxis(
        int xCount,
        int yCount,
        ushort[] topZ,
        int plantHeight,
        ref int curX,
        ref int curY,
        ref int curH,
        int axis,
        ref int cost,
        List<int> path)
    {
        if (axis == 0)
        {
            while (curX > 0)
            {
                int nx = curX - 1;
                int ny = curY;
                int nidx = nx * yCount + ny;
                int nh = topZ[nidx];
                if (nh == Empty)
                {
                    return false;
                }

                int delta = curH - nh;
                if (delta < -1 || delta > 1)
                {
                    return false;
                }

                cost += delta == 1 ? 5 : (delta == 0 ? 1 : 0);
                curX = nx;
                curY = ny;
                curH = nh;
                path.Add(nidx);
            }
        }
        else
        {
            while (curY > 0)
            {
                int nx = curX;
                int ny = curY - 1;
                int nidx = nx * yCount + ny;
                int nh = topZ[nidx];
                if (nh == Empty)
                {
                    return false;
                }

                int delta = curH - nh;
                if (delta < -1 || delta > 1)
                {
                    return false;
                }

                cost += delta == 1 ? 5 : (delta == 0 ? 1 : 0);
                curX = nx;
                curY = ny;
                curH = nh;
                path.Add(nidx);
            }
        }

        if (curX == 0 && curY == 0 && path.Count > 0 && path[path.Count - 1] != 0)
        {
            // Ensure the final coordinate is the plant if the route reached it.
            if (plantHeight == Empty)
            {
                plantHeight = 0;
            }
        }

        return true;
    }

    private static void WriteOperation(StreamWriter output, Operation op, int yCount)
    {
        int idx = op.Idx;
        int x = idx / yCount;
        int y = idx - x * yCount;

        int intermediateCount = op.Path.Length - 2;
        if (intermediateCount < 0)
        {
            intermediateCount = 0;
        }

        var sb = new StringBuilder(64 + intermediateCount * 12);
        sb.Append(intermediateCount);
        sb.Append(' ');
        sb.Append(x);
        sb.Append(' ');
        sb.Append(y);
        sb.Append(' ');
        sb.Append(op.Z);

        for (int i = 1; i + 1 < op.Path.Length; i++)
        {
            int p = op.Path[i];
            int px = p / yCount;
            int py = p - px * yCount;
            sb.Append(' ');
            sb.Append(px);
            sb.Append(' ');
            sb.Append(py);
        }

        sb.Append(' ');
        sb.Append(0);
        sb.Append(' ');
        sb.Append(0);
        output.WriteLine(sb.ToString());
    }
}
