const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksInput.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ order: [] }));
  process.exit(0);
}

const idToIndex = new Map();
const ids = new Array(n);
const duration = new Array(n);
const deadline = new Array(n);
const weight = new Array(n);
const deps = new Array(n);
const succ = Array.from({ length: n }, () => []);
const indeg0 = new Array(n).fill(0);

for (let i = 0; i < n; i++) {
  const t = tasksInput[i] || {};
  const id = t.id;
  ids[i] = id;
  idToIndex.set(id, i);
  duration[i] = Number.isFinite(t.duration) ? t.duration : 0;
  deadline[i] = Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY;
  weight[i] = Number.isFinite(t.weight) ? t.weight : 1;
  deps[i] = Array.isArray(t.deps) ? t.deps.slice() : [];
}

for (let i = 0; i < n; i++) {
  const ds = deps[i];
  let cnt = 0;
  for (let k = 0; k < ds.length; k++) {
    const j = idToIndex.get(ds[k]);
    if (j !== undefined) {
      succ[j].push(i);
      cnt++;
    }
  }
  indeg0[i] = cnt;
}

function computeEarliestFinish() {
  const indeg = indeg0.slice();
  const earliestStart = new Array(n).fill(0);
  const earliestFinish = new Array(n).fill(0);
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  for (let head = 0; head < q.length; head++) {
    const u = q[head];
    earliestFinish[u] = earliestStart[u] + duration[u];
    const fu = earliestFinish[u];
    const out = succ[u];
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      if (earliestStart[v] < fu) earliestStart[v] = fu;
      indeg[v]--;
      if (indeg[v] === 0) q.push(v);
    }
  }
  return { earliestStart, earliestFinish, topoCount: q.length };
}

const { earliestFinish } = computeEarliestFinish();

const crit = new Array(n).fill(0);
const topo = [];
{
  const indeg = indeg0.slice();
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  for (let head = 0; head < q.length; head++) {
    const u = q[head];
    topo.push(u);
    const out = succ[u];
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      indeg[v]--;
      if (indeg[v] === 0) q.push(v);
    }
  }
}

for (let i = topo.length - 1; i >= 0; i--) {
  const u = topo[i];
  let best = 0;
  const out = succ[u];
  for (let k = 0; k < out.length; k++) {
    const v = out[k];
    if (crit[v] > best) best = crit[v];
  }
  crit[u] = duration[u] + best;
}

function compareTasks(a, b) {
  const slackA = deadline[a] - earliestFinish[a];
  const slackB = deadline[b] - earliestFinish[b];
  if (slackA !== slackB) return slackA - slackB;
  if (weight[a] !== weight[b]) return weight[b] - weight[a];
  if (crit[a] !== crit[b]) return crit[b] - crit[a];
  if (duration[a] !== duration[b]) return duration[a] - duration[b];
  return a - b;
}

class MaxHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    const cmp = this.cmp;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (cmp(a[i], a[p]) >= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      const cmp = this.cmp;
      for (;;) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let best = l;
        if (r < a.length && cmp(a[r], a[l]) < 0) best = r;
        if (cmp(a[best], a[i]) >= 0) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
  get length() {
    return this.a.length;
  }
}

const indeg = indeg0.slice();
const heap = new MaxHeap(compareTasks);
for (let i = 0; i < n; i++) {
  if (indeg[i] === 0) heap.push(i);
}

const order = [];
while (heap.length > 0) {
  const u = heap.pop();
  order.push(ids[u]);
  const out = succ[u];
  for (let k = 0; k < out.length; k++) {
    const v = out[k];
    indeg[v]--;
    if (indeg[v] === 0) heap.push(v);
  }
}

if (order.length < n) {
  const seen = new Set(order);
  for (let i = 0; i < n; i++) {
    if (!seen.has(ids[i])) order.push(ids[i]);
  }
}

process.stdout.write(JSON.stringify({ order }));
