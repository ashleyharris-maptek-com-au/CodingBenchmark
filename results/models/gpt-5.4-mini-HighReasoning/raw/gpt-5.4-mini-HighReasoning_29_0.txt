using System;
using System.Buffers;
using System.Diagnostics;
using System.IO;

public static class Program
{
    static int n;
    static int m;

    static int[] head = Array.Empty<int>();
    static int[] adj = Array.Empty<int>();

    static int[] degree = Array.Empty<int>();
    static int[] order = Array.Empty<int>();
    static int[] orderIndex = Array.Empty<int>();
    static int[] fwdCount = Array.Empty<int>();
    static int[] fwdStart = Array.Empty<int>();
    static int[] fwdAdj = Array.Empty<int>();

    static int[] mark = Array.Empty<int>();
    static int markToken;

    static int[] currentClique = Array.Empty<int>();
    static int[] bestClique = Array.Empty<int>();
    static int bestSize;

    static Stopwatch sw = new Stopwatch();
    static bool timeUp;
    const int TimeLimitMs = 280000;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        n = fs.NextInt();
        m = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[] eu = new int[m];
        int[] ev = new int[m];
        degree = new int[n];

        int ecount = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v)
                continue;

            eu[ecount] = u;
            ev[ecount] = v;
            degree[u]++;
            degree[v]++;
            ecount++;
        }
        m = ecount;

        BuildAdjacency(eu, ev);
        BuildDegeneracyAndForward();

        mark = new int[n];
        currentClique = new int[n];
        bestClique = new int[n];
        bestSize = 1;
        bestClique[0] = 0;

        sw.Start();
        RunHeuristic();
        RunExact();

        Array.Sort(bestClique, 0, bestSize);
        Console.WriteLine(bestSize);
        for (int i = 0; i < bestSize; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(bestClique[i]);
        }
        Console.WriteLine();
    }

    static void BuildAdjacency(int[] eu, int[] ev)
    {
        head = new int[n + 1];
        for (int i = 0; i < n; i++)
            head[i + 1] = head[i] + degree[i];

        adj = new int[head[n]];
        int[] fill = new int[n];
        Array.Copy(head, fill, n);

        for (int i = 0; i < m; i++)
        {
            int u = eu[i];
            int v = ev[i];
            adj[fill[u]++] = v;
            adj[fill[v]++] = u;
        }
    }

    static void BuildDegeneracyAndForward()
    {
        int[] deg = (int[])degree.Clone();
        int maxDeg = 0;
        for (int i = 0; i < n; i++)
            if (deg[i] > maxDeg)
                maxDeg = deg[i];

        int[] bin = new int[maxDeg + 1];
        for (int i = 0; i < n; i++)
            bin[deg[i]]++;

        int start = 0;
        for (int d = 0; d <= maxDeg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        order = new int[n];
        orderIndex = new int[n];
        int[] pos = new int[n];

        for (int v = 0; v < n; v++)
        {
            int d = deg[v];
            int p = bin[d]++;
            pos[v] = p;
            order[p] = v;
        }

        for (int d = maxDeg; d > 0; d--)
            bin[d] = bin[d - 1];
        bin[0] = 0;

        for (int i = 0; i < n; i++)
        {
            int v = order[i];
            orderIndex[v] = i;
            for (int p = head[v]; p < head[v + 1]; p++)
            {
                int u = adj[p];
                if (pos[u] > pos[v])
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = order[pw];
                    if (u != w)
                    {
                        order[pw] = u;
                        order[pu] = w;
                        pos[u] = pw;
                        pos[w] = pu;
                    }
                    bin[du]++;
                    deg[u]--;
                }
            }
        }

        fwdCount = new int[n];
        for (int i = 0; i < n; i++)
        {
            int v = order[i];
            int cnt = 0;
            for (int p = head[v]; p < head[v + 1]; p++)
            {
                int u = adj[p];
                if (orderIndex[u] > i)
                    cnt++;
            }
            fwdCount[v] = cnt;
        }

        fwdStart = new int[n + 1];
        for (int i = 0; i < n; i++)
            fwdStart[i + 1] = fwdStart[i] + fwdCount[i];

        fwdAdj = new int[fwdStart[n]];
        int[] fill = new int[n];
        Array.Copy(fwdStart, fill, n);

        for (int v = 0; v < n; v++)
        {
            for (int p = head[v]; p < head[v + 1]; p++)
            {
                int u = adj[p];
                if (orderIndex[u] > orderIndex[v])
                    fwdAdj[fill[v]++] = u;
            }
        }
    }

    static void RunHeuristic()
    {
        int probes = Math.Min(n, 64);
        for (int i = 0; i < probes && !timeUp; i++)
        {
            int startVertex = order[n - 1 - i];
            TryGreedyClique(startVertex);
            if (bestSize == n)
                return;
        }
    }

    static void TryGreedyClique(int startVertex)
    {
        int candCount = fwdCount[startVertex];
        int[] cand = ArrayPool<int>.Shared.Rent(Math.Max(1, candCount));
        if (candCount > 0)
            Array.Copy(fwdAdj, fwdStart[startVertex], cand, 0, candCount);

        currentClique[0] = startVertex;
        int cliqueSize = 1;

        while (candCount > 0 && !timeUp)
        {
            int bestIdx = 0;
            int bestScore = -1;
            for (int i = 0; i < candCount; i++)
            {
                int v = cand[i];
                int score = degree[v];
                if (score > bestScore)
                {
                    bestScore = score;
                    bestIdx = i;
                }
            }

            int chosen = cand[bestIdx];
            currentClique[cliqueSize++] = chosen;

            int token = ++markToken;
            MarkNeighbors(chosen, token);

            int nextCount = 0;
            for (int i = 0; i < candCount; i++)
            {
                int v = cand[i];
                if (mark[v] == token)
                    cand[nextCount++] = v;
            }
            candCount = nextCount;
        }

        UpdateBest(cliqueSize);
        ArrayPool<int>.Shared.Return(cand, false);
    }

    static void RunExact()
    {
        for (int i = n - 1; i >= 0 && !timeUp; i--)
        {
            int v = order[i];
            int candCount = fwdCount[v];
            if (candCount + 1 <= bestSize)
                continue;

            int[] cand = ArrayPool<int>.Shared.Rent(Math.Max(1, candCount));
            if (candCount > 0)
                Array.Copy(fwdAdj, fwdStart[v], cand, 0, candCount);

            currentClique[0] = v;
            Expand(cand, candCount, 1);
            ArrayPool<int>.Shared.Return(cand, false);
        }
    }

    static void Expand(int[] cand, int count, int cliqueSize)
    {
        if (timeUp)
            return;
        if (sw.ElapsedMilliseconds > TimeLimitMs)
        {
            timeUp = true;
            return;
        }

        if (cliqueSize + count <= bestSize)
            return;

        if (count == 0)
        {
            UpdateBest(cliqueSize);
            return;
        }

        int[] ordered = ArrayPool<int>.Shared.Rent(count);
        int[] colors = ArrayPool<int>.Shared.Rent(count);
        int coloredCount = GreedyColor(cand, count, ordered, colors);

        for (int i = coloredCount - 1; i >= 0; i--)
        {
            if (cliqueSize + colors[i] <= bestSize)
                break;

            int v = ordered[i];
            currentClique[cliqueSize] = v;

            int[] next = ArrayPool<int>.Shared.Rent(Math.Max(1, i));
            int nextCount = 0;

            int token = ++markToken;
            MarkNeighbors(v, token);
            for (int j = 0; j < i; j++)
            {
                int u = ordered[j];
                if (mark[u] == token)
                    next[nextCount++] = u;
            }

            Expand(next, nextCount, cliqueSize + 1);
            ArrayPool<int>.Shared.Return(next, false);

            if (timeUp)
                break;
        }

        ArrayPool<int>.Shared.Return(ordered, false);
        ArrayPool<int>.Shared.Return(colors, false);
    }

    static int GreedyColor(int[] cand, int count, int[] ordered, int[] colors)
    {
        int[] remaining = ArrayPool<int>.Shared.Rent(count);
        Array.Copy(cand, remaining, count);

        int remCount = count;
        int outCount = 0;
        int color = 0;

        while (remCount > 0)
        {
            color++;
            int token = ++markToken;
            int nextCount = 0;

            for (int i = 0; i < remCount; i++)
            {
                int v = remaining[i];
                if (mark[v] != token)
                {
                    ordered[outCount] = v;
                    colors[outCount] = color;
                    outCount++;
                    MarkNeighbors(v, token);
                }
                else
                {
                    remaining[nextCount++] = v;
                }
            }

            remCount = nextCount;
        }

        ArrayPool<int>.Shared.Return(remaining, false);
        return outCount;
    }

    static void MarkNeighbors(int v, int token)
    {
        mark[v] = token;
        for (int p = head[v]; p < head[v + 1]; p++)
            mark[adj[p]] = token;
    }

    static void UpdateBest(int size)
    {
        if (size > bestSize)
        {
            bestSize = size;
            Array.Copy(currentClique, bestClique, size);
        }
    }

    sealed class FastScanner
    {
        readonly Stream stream;
        readonly byte[] buffer = new byte[1 << 16];
        int len;
        int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                    return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32 && c != -1)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0)
                    return -1;
            }
            return buffer[ptr++];
        }
    }
}
