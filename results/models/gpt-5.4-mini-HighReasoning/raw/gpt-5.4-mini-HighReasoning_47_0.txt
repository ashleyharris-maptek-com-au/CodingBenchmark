fn rotl(x : u32, s : u32) -> u32 {
  return (x << s) | (x >> (32u - s));
}

fn ctz32(x : u32) -> u32 {
  if (x == 0u) {
    return 32u;
  }

  var n : u32 = 0u;
  var v : u32 = x;
  loop {
    if ((v & 1u) != 0u) {
      break;
    }
    n = n + 1u;
    v = v >> 1u;
  }
  return n;
}

fn trailing_zero_bits(h0 : u32, h1 : u32) -> u32 {
  if (h0 != 0u) {
    return ctz32(h0);
  }
  if (h1 != 0u) {
    return 32u + ctz32(h1);
  }
  return 64u;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  if (gid.x >= params.z) {
    return;
  }

  let nonce : u32 = params.y + gid.x;
  let hash : vec2<u32> = custom_hash(nonce);
  let tz : u32 = trailing_zero_bits(hash.x, hash.y);

  if (tz >= params.x) {
    result[1u] = nonce;
    result[2u] = hash.x;
    result[3u] = hash.y;
    result[0u] = 1u;
  }
