const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robotsIn = Array.isArray(data.robots) ? data.robots : [];
const jobsIn = Array.isArray(data.jobs) ? data.jobs : [];

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const cellCount = w * h;
const BLOCK = 35; // '#'

function cellIndex(x, y) {
  return y * w + x;
}

function manhattanXY(x1, y1, x2, y2) {
  return Math.abs(x1 - x2) + Math.abs(y1 - y2);
}

const comp = new Int32Array(cellCount);
comp.fill(-1);

let compCount = 0;
const queue = new Int32Array(cellCount || 1);

for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    const idx = cellIndex(x, y);
    if (comp[idx] !== -1 || row.charCodeAt(x) === BLOCK) continue;
    const id = compCount++;
    let head = 0;
    let tail = 0;
    queue[tail++] = idx;
    comp[idx] = id;
    while (head < tail) {
      const v = queue[head++];
      const vy = Math.floor(v / w);
      const vx = v - vy * w;

      if (vx > 0) {
        const n = v - 1;
        if (comp[n] === -1 && grid[vy].charCodeAt(vx - 1) !== BLOCK) {
          comp[n] = id;
          queue[tail++] = n;
        }
      }
      if (vx + 1 < w) {
        const n = v + 1;
        if (comp[n] === -1 && grid[vy].charCodeAt(vx + 1) !== BLOCK) {
          comp[n] = id;
          queue[tail++] = n;
        }
      }
      if (vy > 0) {
        const n = v - w;
        if (comp[n] === -1 && grid[vy - 1].charCodeAt(vx) !== BLOCK) {
          comp[n] = id;
          queue[tail++] = n;
        }
      }
      if (vy + 1 < h) {
        const n = v + w;
        if (comp[n] === -1 && grid[vy + 1].charCodeAt(vx) !== BLOCK) {
          comp[n] = id;
          queue[tail++] = n;
        }
      }
    }
  }
}

const robots = robotsIn.map((r, inputIndex) => {
  const x = Math.trunc(Number(r.x));
  const y = Math.trunc(Number(r.y));
  const idNum = Math.trunc(Number(r.id));
  const cell = (x >= 0 && x < w && y >= 0 && y < h) ? cellIndex(x, y) : -1;
  return {
    id: r.id,
    idNum,
    x,
    y,
    cell,
    comp: cell >= 0 ? comp[cell] : -1,
    inputIndex,
  };
}).sort((a, b) => (a.idNum - b.idNum) || (a.inputIndex - b.inputIndex));

const jobs = jobsIn.map((j, inputIndex) => {
  const x = Math.trunc(Number(j.x));
  const y = Math.trunc(Number(j.y));
  const idNum = Math.trunc(Number(j.id));
  const priority = Number(j.priority) || 0;
  const cell = (x >= 0 && x < w && y >= 0 && y < h) ? cellIndex(x, y) : -1;
  return {
    id: j.id,
    idNum,
    x,
    y,
    priority,
    cell,
    comp: cell >= 0 ? comp[cell] : -1,
    inputIndex,
  };
});

let maxRobotId = -1;
for (const r of robots) maxRobotId = Math.max(maxRobotId, r.idNum);
const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

const compJobs = Array.from({ length: compCount }, () => []);
const compRobots = Array.from({ length: compCount }, () => []);
const leftoverJobs = [];

for (let i = 0; i < jobs.length; i++) {
  const c = jobs[i].comp;
  if (c >= 0) compJobs[c].push(i);
  else leftoverJobs.push(i);
}
for (let i = 0; i < robots.length; i++) {
  const c = robots[i].comp;
  if (c >= 0) compRobots[c].push(i);
}

function bfsDistancesToJobs(robot, compId, jobCellMap, jobCount) {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);
  const out = new Int32Array(jobCount);
  out.fill(-1);

  if (robot.cell < 0 || robot.comp !== compId) return out;

  let remaining = jobCount;
  let head = 0;
  let tail = 0;
  queue[tail++] = robot.cell;
  dist[robot.cell] = 0;

  while (head < tail && remaining > 0) {
    const v = queue[head++];
    const dv = dist[v] + 1;
    const vy = Math.floor(v / w);
    const vx = v - vy * w;

    const found = jobCellMap.get(v);
    if (found) {
      for (let k = 0; k < found.length; k++) {
        const jLocal = found[k];
        if (out[jLocal] === -1) {
          out[jLocal] = dist[v];
          remaining--;
        }
      }
    }

    if (vx > 0) {
      const n = v - 1;
      if (dist[n] === -1 && comp[n] === compId) {
        dist[n] = dv;
        queue[tail++] = n;
      }
    }
    if (vx + 1 < w) {
      const n = v + 1;
      if (dist[n] === -1 && comp[n] === compId) {
        dist[n] = dv;
        queue[tail++] = n;
      }
    }
    if (vy > 0) {
      const n = v - w;
      if (dist[n] === -1 && comp[n] === compId) {
        dist[n] = dv;
        queue[tail++] = n;
      }
    }
    if (vy + 1 < h) {
      const n = v + w;
      if (dist[n] === -1 && comp[n] === compId) {
        dist[n] = dv;
        queue[tail++] = n;
      }
    }
  }

  return out;
}

function spreadBits16(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n >>> 0;
}

function mortonKey(x, y, minX, maxX, minY, maxY) {
  const sx = maxX > minX ? Math.floor(((x - minX) * 65535) / (maxX - minX)) : 0;
  const sy = maxY > minY ? Math.floor(((y - minY) * 65535) / (maxY - minY)) : 0;
  return (spreadBits16(sx) | (spreadBits16(sy) << 1)) >>> 0;
}

function routeCostFromOrder(robot, order) {
  let cost = 0;
  let cx = robot.x;
  let cy = robot.y;
  for (let i = 0; i < order.length; i++) {
    const job = jobs[order[i]];
    cost += manhattanXY(cx, cy, job.x, job.y);
    cx = job.x;
    cy = job.y;
  }
  return cost;
}

function orderJobsForRobot(robot, jobIndices) {
  const m = jobIndices.length;
  if (m === 0) return [];
  if (m === 1) return [jobs[jobIndices[0]].id];

  const EXACT_THRESHOLD = 300;
  if (m <= EXACT_THRESHOLD) {
    const rem = jobIndices.slice();
    const ordered = [];
    let cx = robot.x;
    let cy = robot.y;
    while (rem.length > 0) {
      let bestPos = 0;
      let bestDist = 0x7fffffff;
      let bestPri = -Infinity;
      let bestIdNum = 0x7fffffff;
      for (let i = 0; i < rem.length; i++) {
        const job = jobs[rem[i]];
        const d = manhattanXY(cx, cy, job.x, job.y);
        if (
          d < bestDist ||
          (d === bestDist && (job.priority > bestPri || (job.priority === bestPri && job.idNum < bestIdNum)))
        ) {
          bestDist = d;
          bestPri = job.priority;
          bestIdNum = job.idNum;
          bestPos = i;
        }
      }
      const chosen = rem[bestPos];
      ordered.push(jobs[chosen].id);
      const last = rem.pop();
      if (bestPos < rem.length) rem[bestPos] = last;
      cx = jobs[chosen].x;
      cy = jobs[chosen].y;
    }
    return ordered;
  }

  let minX = Infinity;
  let maxX = -Infinity;
  let minY = Infinity;
  let maxY = -Infinity;
  for (let i = 0; i < m; i++) {
    const job = jobs[jobIndices[i]];
    if (job.x < minX) minX = job.x;
    if (job.x > maxX) maxX = job.x;
    if (job.y < minY) minY = job.y;
    if (job.y > maxY) maxY = job.y;
  }

  const entries = new Array(m);
  let bestStartDist = 0x7fffffff;
  for (let i = 0; i < m; i++) {
    const idx = jobIndices[i];
    const job = jobs[idx];
    const key = mortonKey(job.x, job.y, minX, maxX, minY, maxY);
    const startDist = manhattanXY(robot.x, robot.y, job.x, job.y);
    entries[i] = { idx, key, startDist, priority: job.priority, idNum: job.idNum };
    if (startDist < bestStartDist) {
      bestStartDist = startDist;
    }
  }

  entries.sort((a, b) =>
    (a.key - b.key) ||
    (a.startDist - b.startDist) ||
    (b.priority - a.priority) ||
    (a.idNum - b.idNum)
  );

  let pos = 0;
  for (let i = 1; i < entries.length; i++) {
    if (entries[i].startDist < entries[pos].startDist ||
        (entries[i].startDist === entries[pos].startDist && entries[i].idNum < entries[pos].idNum)) {
      pos = i;
    }
  }
  if (pos !== 0) {
    const rotated = entries.slice(pos).concat(entries.slice(0, pos));
    const forwardCost = routeCostFromOrder(robot, rotated.map(e => e.idx));
    const reversed = rotated.slice().reverse();
    const reverseCost = routeCostFromOrder(robot, reversed.map(e => e.idx));
    const chosen = reverseCost < forwardCost ? reversed : rotated;
    return chosen.map(e => jobs[e.idx].id);
  }

  const forwardCost = routeCostFromOrder(robot, entries.map(e => e.idx));
  const reversed = entries.slice().reverse();
  const reverseCost = routeCostFromOrder(robot, reversed.map(e => e.idx));
  const chosen = reverseCost < forwardCost ? reversed : entries;
  return chosen.map(e => jobs[e.idx].id);
}

function assignComponent(compId) {
  const rList = compRobots[compId];
  const jList = compJobs[compId];
  if (!jList || jList.length === 0) return;
  if (!rList || rList.length === 0) {
    for (let i = 0; i < jList.length; i++) leftoverJobs.push(jList[i]);
    return;
  }

  const jobCellMap = new Map();
  for (let jLocal = 0; jLocal < jList.length; jLocal++) {
    const cell = jobs[jList[jLocal]].cell;
    const arr = jobCellMap.get(cell);
    if (arr) arr.push(jLocal);
    else jobCellMap.set(cell, [jLocal]);
  }

  const robotDists = new Array(rList.length);
  for (let rLocal = 0; rLocal < rList.length; rLocal++) {
    robotDists[rLocal] = bfsDistancesToJobs(robots[rList[rLocal]], compId, jobCellMap, jList.length);
  }

  const minDists = new Int32Array(jList.length);
  let sumMinDist = 0;
  for (let jLocal = 0; jLocal < jList.length; jLocal++) {
    let best = 0x7fffffff;
    for (let rLocal = 0; rLocal < rList.length; rLocal++) {
      const d = robotDists[rLocal][jLocal];
      if (d >= 0 && d < best) best = d;
    }
    if (best === 0x7fffffff) best = 1000000000;
    minDists[jLocal] = best;
    sumMinDist += best;
  }

  const ideal = jList.length / rList.length;
  const avgMinDist = sumMinDist / jList.length;
  const loadPenalty = avgMinDist / Math.max(1, ideal);
  const counts = new Int32Array(rList.length);
  const assigned = Array.from({ length: rList.length }, () => []);

  const order = new Array(jList.length);
  for (let i = 0; i < jList.length; i++) order[i] = i;
  order.sort((a, b) => {
    const ja = jobs[jList[a]];
    const jb = jobs[jList[b]];
    return (jb.priority - ja.priority) || (minDists[a] - minDists[b]) || (ja.idNum - jb.idNum);
  });

  for (let oi = 0; oi < order.length; oi++) {
    const jLocal = order[oi];
    const job = jobs[jList[jLocal]];
    let bestR = -1;
    let bestScore = Infinity;
    let bestDist = Infinity;
    let bestRobotId = Infinity;
    for (let rLocal = 0; rLocal < rList.length; rLocal++) {
      const d = robotDists[rLocal][jLocal];
      if (d < 0) continue;
      const rr = robots[rList[rLocal]];
      const score = d + loadPenalty * counts[rLocal];
      if (
        score < bestScore ||
        (score === bestScore && (d < bestDist || (d === bestDist && rr.idNum < bestRobotId)))
      ) {
        bestScore = score;
        bestDist = d;
        bestRobotId = rr.idNum;
        bestR = rLocal;
      }
    }

    if (bestR < 0) {
      for (let rLocal = 0; rLocal < rList.length; rLocal++) {
        const rr = robots[rList[rLocal]];
        const d = manhattanXY(rr.x, rr.y, job.x, job.y);
        const score = d + loadPenalty * counts[rLocal];
        if (score < bestScore || (score === bestScore && rr.idNum < bestRobotId)) {
          bestScore = score;
          bestDist = d;
          bestRobotId = rr.idNum;
          bestR = rLocal;
        }
      }
    }

    assigned[bestR].push(jList[jLocal]);
    counts[bestR]++;
  }

  for (let rLocal = 0; rLocal < rList.length; rLocal++) {
    const robot = robots[rList[rLocal]];
    assignments[robot.idNum] = orderJobsForRobot(robot, assigned[rLocal]);
  }
}

for (let c = 0; c < compCount; c++) assignComponent(c);

if (leftoverJobs.length > 0 && robots.length > 0) {
  const counts = new Int32Array(robots.length);
  for (let i = 0; i < robots.length; i++) {
    const slot = assignments[robots[i].idNum];
    counts[i] = slot ? slot.length : 0;
  }
  const avgDist = 1;
  const loadPenalty = avgDist / Math.max(1, leftoverJobs.length / Math.max(1, robots.length));
  const sortedLeftovers = leftoverJobs.slice().sort((a, b) => {
    const ja = jobs[a];
    const jb = jobs[b];
    return (jb.priority - ja.priority) || (ja.idNum - jb.idNum);
  });
  for (let i = 0; i < sortedLeftovers.length; i++) {
    const j = sortedLeftovers[i];
    const job = jobs[j];
    let bestR = 0;
    let bestScore = Infinity;
    for (let r = 0; r < robots.length; r++) {
      const rr = robots[r];
      const d = manhattanXY(rr.x, rr.y, job.x, job.y);
      const score = d + loadPenalty * counts[r];
      if (score < bestScore || (score === bestScore && rr.idNum < robots[bestR].idNum)) {
        bestScore = score;
        bestR = r;
      }
    }
    assignments[robots[bestR].idNum].push(job.id);
    counts[bestR]++;
  }
}

process.stdout.write(JSON.stringify({ assignments }));
