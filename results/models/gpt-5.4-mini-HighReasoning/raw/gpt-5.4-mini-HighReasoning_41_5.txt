; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_32 = OpConstant %float 32
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%glsl450 = OpExtInstImport "GLSL.std.450"
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%main = OpFunction %void None %func
%label = OpLabel
%wp = OpLoad %v3float %worldPos
%n_raw = OpLoad %v3float %normal
%N = OpExtInst %v3float %glsl450 Normalize %n_raw
%light_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %light_ptr
%lightPos = OpVectorShuffle %v3float %light4 %light4 0 1 2
%cam_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cam4 = OpLoad %v4float %cam_ptr
%cameraPos = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
%lightVec = OpFSub %v3float %lightPos %wp
%L = OpExtInst %v3float %glsl450 Normalize %lightVec
%viewVec = OpFSub %v3float %cameraPos %wp
%V = OpExtInst %v3float %glsl450 Normalize %viewVec
%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl450 Reflect %negL %N
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 FMax %ndotl %float_0
%rdotv = OpDot %float %R %V
%specBase = OpExtInst %float %glsl450 FMax %rdotv %float_0
%specular = OpExtInst %float %glsl450 Pow %specBase %float_32
%diffTerm = OpFMul %float %float_0_5 %diffuse
%specTerm = OpFMul %float %float_0_8 %specular
%sum1 = OpFAdd %float %float_0_2 %diffTerm
%intensity = OpFAdd %float %sum1 %specTerm
%rgb = OpCompositeConstruct %v3float %intensity %intensity %intensity
%color = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
