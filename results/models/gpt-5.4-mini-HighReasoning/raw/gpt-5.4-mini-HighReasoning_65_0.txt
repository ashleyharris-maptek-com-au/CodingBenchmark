declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type TaskInput = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
};

type Task = {
  id: string | number;
  duration: number;
  deadline: number;
  weight: number;
  deps: number[];
  children: number[];
  indegree: number;
  bottomLevel: number;
  inputIndex: number;
  priority1: number;
  priority2: number;
  priority3: number;
  priority4: number;
  priority5: number;
};

class BinaryHeap<T> {
  private data: T[] = [];
  constructor(private readonly less: (a: T, b: T) => boolean) {}

  size(): number {
    return this.data.length;
  }

  isEmpty(): boolean {
    return this.data.length === 0;
  }

  peek(): T | undefined {
    return this.data[0];
  }

  push(value: T): void {
    const a = this.data;
    a.push(value);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }

  pop(): T | undefined {
    const a = this.data;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        const l = i * 2 + 1;
        const r = l + 1;
        let best = i;
        if (l < a.length && this.less(a[l], a[best])) best = l;
        if (r < a.length && this.less(a[r], a[best])) best = r;
        if (best === i) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
}

const machineCountRaw = Number(data.machineCount);
const machineCount = Number.isFinite(machineCountRaw) && machineCountRaw > 0 ? Math.floor(machineCountRaw) : 0;
const rawTasks: TaskInput[] = Array.isArray(data.tasks) ? data.tasks : [];

const n = rawTasks.length;
const tasks: Task[] = new Array(n);
const idToIndex = new Map<string | number, number>();

for (let i = 0; i < n; i++) {
  idToIndex.set(rawTasks[i].id, i);
}

for (let i = 0; i < n; i++) {
  const t = rawTasks[i];
  const deps: number[] = [];
  const depList = Array.isArray(t.deps) ? t.deps : [];
  for (const depId of depList) {
    const depIndex = idToIndex.get(depId);
    if (depIndex !== undefined && depIndex !== i) deps.push(depIndex);
  }
  tasks[i] = {
    id: t.id,
    duration: typeof t.duration === 'number' && Number.isFinite(t.duration) ? t.duration : Number(t.duration) || 0,
    deadline: typeof t.deadline === 'number' && Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY,
    weight: typeof t.weight === 'number' && Number.isFinite(t.weight) ? t.weight : 0,
    deps,
    children: [],
    indegree: 0,
    bottomLevel: 0,
    inputIndex: i,
    priority1: 0,
    priority2: 0,
    priority3: 0,
    priority4: 0,
    priority5: 0,
  };
}

for (let i = 0; i < n; i++) {
  const deps = tasks[i].deps;
  tasks[i].indegree = deps.length;
  for (const dep of deps) tasks[dep].children.push(i);
}

// Reverse topological pass for critical-path lengths.
const indegForTopo = tasks.map((t) => t.indegree);
const topo: number[] = [];
const queue: number[] = [];
for (let i = 0; i < n; i++) {
  if (indegForTopo[i] === 0) queue.push(i);
}
for (let head = 0; head < queue.length; head++) {
  const u = queue[head];
  topo.push(u);
  for (const v of tasks[u].children) {
    indegForTopo[v]--;
    if (indegForTopo[v] === 0) queue.push(v);
  }
}

// If the input contains a cycle, the benchmark is malformed; still avoid crashing.
if (topo.length !== n) {
  const seen = new Array<boolean>(n).fill(false);
  for (const u of topo) seen[u] = true;
  for (let i = 0; i < n; i++) {
    if (!seen[i]) topo.push(i);
  }
}

for (let i = n - 1; i >= 0; i--) {
  const u = topo[i];
  let bestChild = 0;
  for (const v of tasks[u].children) {
    if (tasks[v].bottomLevel > bestChild) bestChild = tasks[v].bottomLevel;
  }
  tasks[u].bottomLevel = tasks[u].duration + bestChild;
}

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  const finiteDeadline = Number.isFinite(t.deadline);
  t.priority1 = finiteDeadline ? t.deadline - t.bottomLevel : Number.POSITIVE_INFINITY;
  t.priority2 = finiteDeadline ? t.deadline : Number.POSITIVE_INFINITY;
  t.priority3 = -t.weight;
  t.priority4 = -t.bottomLevel;
  t.priority5 = -t.duration;
}

function compareTaskIndex(a: number, b: number): boolean {
  const ta = tasks[a];
  const tb = tasks[b];
  if (ta.priority1 !== tb.priority1) return ta.priority1 < tb.priority1;
  if (ta.priority2 !== tb.priority2) return ta.priority2 < tb.priority2;
  if (ta.priority3 !== tb.priority3) return ta.priority3 < tb.priority3;
  if (ta.priority4 !== tb.priority4) return ta.priority4 < tb.priority4;
  if (ta.priority5 !== tb.priority5) return ta.priority5 < tb.priority5;
  return ta.inputIndex < tb.inputIndex;
}

type Running = { finish: number; machine: number; task: number };
function compareRunning(a: Running, b: Running): boolean {
  if (a.finish !== b.finish) return a.finish < b.finish;
  if (a.machine !== b.machine) return a.machine < b.machine;
  return a.task < b.task;
}

const ready = new BinaryHeap<number>(compareTaskIndex);
const running = new BinaryHeap<Running>(compareRunning);
const freeMachines = new BinaryHeap<number>((a, b) => a < b);

const remainingIndegree = tasks.map((t) => t.indegree);
for (let i = 0; i < machineCount; i++) freeMachines.push(i);
for (let i = 0; i < n; i++) {
  if (remainingIndegree[i] === 0) ready.push(i);
}

const startTimes = new Array<number>(n);
const machineForTask = new Array<number>(n);
let scheduledCount = 0;
let currentTime = 0;

function releaseCompleted(upToTime: number): void {
  while (!running.isEmpty() && running.peek()!.finish <= upToTime) {
    const done = running.pop()!;
    freeMachines.push(done.machine);
    const u = done.task;
    for (const v of tasks[u].children) {
      remainingIndegree[v]--;
      if (remainingIndegree[v] === 0) ready.push(v);
    }
  }
}

while (scheduledCount < n) {
  releaseCompleted(currentTime);

  let progress = false;
  while (!ready.isEmpty() && !freeMachines.isEmpty()) {
    const taskIndex = ready.pop()!;
    const machine = freeMachines.pop()!;
    const start = currentTime;
    const finish = start + tasks[taskIndex].duration;
    startTimes[taskIndex] = start;
    machineForTask[taskIndex] = machine;
    running.push({ finish, machine, task: taskIndex });
    scheduledCount++;
    progress = true;
  }

  if (scheduledCount >= n) break;

  if (!progress) {
    const next = running.peek();
    if (!next) break;
    currentTime = Math.max(currentTime, next.finish);
  }
}

const schedule = new Array<{ id: string | number; start: number; machine: number }>(n);
for (let i = 0; i < n; i++) {
  schedule[i] = {
    id: tasks[i].id,
    start: startTimes[i] ?? 0,
    machine: machineForTask[i] ?? 0,
  };
}

schedule.sort((a, b) => {
  if (a.start !== b.start) return a.start - b.start;
  if (a.machine !== b.machine) return a.machine - b.machine;
  const ai = idToIndex.get(a.id) ?? 0;
  const bi = idToIndex.get(b.id) ?? 0;
  return ai - bi;
});

process.stdout.write(JSON.stringify({ schedule }));
