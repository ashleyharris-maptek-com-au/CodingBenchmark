struct Point {
    float x;
    float y;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = double(a.x) - double(b.x);
    double dy = double(a.y) - double(b.y);
    return dx * dx + dy * dy;
}

static inline uint64_t splitBy1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t mortonKey(float x, float y, float minX, float maxX, float minY, float maxY) {
    constexpr uint32_t scale = (1u << 21) - 1u;
    double nx = 0.5;
    double ny = 0.5;
    double dx = double(maxX) - double(minX);
    double dy = double(maxY) - double(minY);
    if (dx > 0.0) nx = (double(x) - double(minX)) / dx;
    if (dy > 0.0) ny = (double(y) - double(minY)) / dy;
    if (nx < 0.0) nx = 0.0;
    if (nx > 1.0) nx = 1.0;
    if (ny < 0.0) ny = 0.0;
    if (ny > 1.0) ny = 1.0;
    uint32_t ix = uint32_t(nx * scale + 0.5);
    uint32_t iy = uint32_t(ny * scale + 0.5);
    return splitBy1(ix) | (splitBy1(iy) << 1);
}

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char getChar() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c = getChar();
        if (!c) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = getChar();
            if (!c) return false;
        }
        T sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        T val = 0;
        for (; c >= '0' && c <= '9'; c = getChar()) val = val * 10 + (c - '0');
        out = val * sign;
        return true;
    }

    bool readDouble(double& out) {
        char c = getChar();
        if (!c) return false;
        while ((c < '0' || c > '9') && c != '-' && c != '.') {
            c = getChar();
            if (!c) return false;
        }
        char tmp[128];
        size_t p = 0;
        while (c && c != ' ' && c != '\n' && c != '\r' && c != '\t') {
            if (p + 1 < sizeof(tmp)) tmp[p++] = c;
            c = getChar();
        }
        auto res = from_chars(tmp, tmp + p, out);
        if (res.ec != errc()) out = 0.0;
        return true;
    }
};

static vector<int> exactSmallTSP(const vector<Point>& pts) {
    int n = (int)pts.size();
    if (n == 1) return {0};

    int m = n - 1;
    int total = 1 << m;
    const double INF = numeric_limits<double>::infinity();

    vector<double> dp((size_t)total * m, INF);
    vector<int> parent((size_t)total * m, -1);
    auto at = [&](int mask, int i) -> double& { return dp[(size_t)mask * m + i]; };
    auto par = [&](int mask, int i) -> int& { return parent[(size_t)mask * m + i]; };

    for (int i = 0; i < m; ++i) at(1 << i, i) = dist2(pts[0], pts[i + 1]);

    for (int mask = 1; mask < total; ++mask) {
        for (int last = 0; last < m; ++last) {
            if (!(mask & (1 << last))) continue;
            double cur = at(mask, last);
            if (!isfinite(cur)) continue;
            int rem = ((total - 1) ^ mask);
            while (rem) {
                int bit = rem & -rem;
                int nxt = (int)countr_zero((unsigned)bit);
                int nmask = mask | bit;
                double cand = cur + dist2(pts[last + 1], pts[nxt + 1]);
                double& dst = at(nmask, nxt);
                if (cand < dst) {
                    dst = cand;
                    par(nmask, nxt) = last;
                }
                rem ^= bit;
            }
        }
    }

    double best = INF;
    int bestLast = -1;
    int full = total - 1;
    for (int last = 0; last < m; ++last) {
        double cand = at(full, last) + dist2(pts[last + 1], pts[0]);
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }

    vector<int> route;
    route.reserve(n);
    route.push_back(0);
    int mask = full;
    int last = bestLast;
    vector<int> rev;
    rev.reserve(m);
    while (mask) {
        rev.push_back(last + 1);
        int pl = par(mask, last);
        mask ^= 1 << last;
        last = pl;
    }
    reverse(rev.begin(), rev.end());
    route.insert(route.end(), rev.begin(), rev.end());
    return route;
}

static vector<uint64_t> makeKeys(const vector<Point>& pts) {
    float minX = numeric_limits<float>::max();
    float maxX = -numeric_limits<float>::max();
    float minY = numeric_limits<float>::max();
    float maxY = -numeric_limits<float>::max();
    for (const auto& p : pts) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }
    vector<uint64_t> keys(pts.size());
    for (size_t i = 0; i < pts.size(); ++i) keys[i] = mortonKey(pts[i].x, pts[i].y, minX, maxX, minY, maxY);
    return keys;
}

static vector<int> radixSortOrder(const vector<uint64_t>& keys) {
    int n = (int)keys.size();
    vector<int> order(n), tmp(n);
    iota(order.begin(), order.end(), 0);
    constexpr int BITS = 16;
    constexpr int BUCKETS = 1 << BITS;
    vector<int> cnt(BUCKETS);

    for (int pass = 0; pass < 4; ++pass) {
        fill(cnt.begin(), cnt.end(), 0);
        int shift = pass * BITS;
        for (int i = 0; i < n; ++i) ++cnt[(keys[order[i]] >> shift) & (BUCKETS - 1)];
        int sum = 0;
        for (int i = 0; i < BUCKETS; ++i) {
            int c = cnt[i];
            cnt[i] = sum;
            sum += c;
        }
        for (int i = 0; i < n; ++i) {
            int id = order[i];
            int b = (keys[id] >> shift) & (BUCKETS - 1);
            tmp[cnt[b]++] = id;
        }
        order.swap(tmp);
    }
    return order;
}

static vector<int> greedyLocalRoute(const vector<Point>& pts) {
    int n = (int)pts.size();
    vector<uint64_t> keys = makeKeys(pts);
    vector<int> order = radixSortOrder(keys);
    vector<int> pos(n);
    for (int i = 0; i < n; ++i) pos[order[i]] = i;

    vector<char> used(n, 0);
    vector<int> route;
    route.reserve(n);
    int cur = 0;
    route.push_back(cur);
    used[cur] = 1;

    const int maxRadius = 64;
    for (int step = 1; step < n; ++step) {
        int p = pos[cur];
        int best = -1;
        double bestD = numeric_limits<double>::infinity();

        for (int r = 1; r <= maxRadius; ++r) {
            bool found = false;
            int L = max(0, p - r);
            int R = min(n - 1, p + r);
            for (int i = L; i <= R; ++i) {
                int cand = order[i];
                if (used[cand]) continue;
                found = true;
                double d = dist2(pts[cur], pts[cand]);
                if (d < bestD) {
                    bestD = d;
                    best = cand;
                }
            }
            if (found && best != -1) break;
        }

        if (best == -1) {
            for (int cand : order) {
                if (!used[cand]) {
                    best = cand;
                    break;
                }
            }
        }

        cur = best;
        used[cur] = 1;
        route.push_back(cur);
    }
    return route;
}

static vector<int> mortonRoute(const vector<Point>& pts) {
    vector<uint64_t> keys = makeKeys(pts);
    vector<int> order = radixSortOrder(keys);
    auto it = find(order.begin(), order.end(), 0);
    if (it != order.end()) rotate(order.begin(), it, order.end());
    return order;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastInput in;
    int N;
    if (!in.readInt(N)) return 0;

    vector<Point> pts(N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        in.readDouble(x);
        in.readDouble(y);
        pts[i] = Point{(float)x, (float)y};
    }

    vector<int> route;
    if (N <= 12) {
        route = exactSmallTSP(pts);
    } else if (N <= 5000) {
        route = greedyLocalRoute(pts);
    } else {
        route = mortonRoute(pts);
    }

    if (!route.empty() && route[0] != 0) {
        auto it = find(route.begin(), route.end(), 0);
        if (it != route.end()) rotate(route.begin(), it, route.end());
    }

    for (int i = 0; i < (int)route.size(); ++i) {
        if (i) cout << ' ';
        cout << route[i];
    }
    cout << '\n';
    return 0;
}
