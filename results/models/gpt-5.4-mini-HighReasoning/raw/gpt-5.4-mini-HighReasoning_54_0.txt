from statistics import median
import math


_state = {
    "trim_bias": 0.0,
    "speed_int": 0.0,
    "loc_int": 0.0,
    "gs_int": 0.0,
    "roll_int": 0.0,
    "last_on_ground": False,
    "touchdown_latched": False,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _is_valid(v):
    return isinstance(v, (int, float)) and not math.isnan(v)


def _flag(v):
    return bool(v)


def _ang_diff(a, b):
    if not _is_valid(a) or not _is_valid(b):
        return 0.0
    d = (a - b + 180.0) % 360.0 - 180.0
    return d


def _vote(state, value_keys, valid_keys=None):
    vals = []
    for i, key in enumerate(value_keys):
        if valid_keys and i < len(valid_keys):
            if not _flag(state.get(valid_keys[i])):
                continue
        v = state.get(key)
        if _is_valid(v):
            vals.append(float(v))
    if not vals:
        return None
    return float(median(vals))


def _first_number(state, keys):
    for k in keys:
        v = state.get(k)
        if _is_valid(v):
            return float(v)
    return None


def _phase(ra_ft, on_ground, wow, gs_dots):
    if on_ground or wow or (ra_ft is not None and ra_ft < 8.0):
        return "rollout"
    if ra_ft is not None and ra_ft < 50.0:
        return "flare"
    if ra_ft is not None and ra_ft < 200.0:
        return "landing"
    if ra_ft is not None and ra_ft < 1000.0:
        return "approach"
    return "intercept"


def _flap_schedule(phase, ra_ft, ias, target_ias):
    tgt = target_ias if target_ias is not None else 137.0
    if phase == "rollout":
        return 40.0
    if phase == "flare":
        return 40.0
    if phase == "landing":
        if ias is not None and ias > tgt + 35.0:
            return 20.0
        return 30.0 if (ra_ft is not None and ra_ft > 80.0) else 40.0
    if phase == "approach":
        if ra_ft is not None and ra_ft > 700.0:
            return 10.0
        if ra_ft is not None and ra_ft > 300.0:
            return 20.0
        return 30.0
    return 10.0 if (ra_ft is not None and ra_ft > 2000.0) else 20.0


def _approach_speed(target_vref, ra_ft):
    vref = target_vref if target_vref is not None else 137.0
    if ra_ft is None:
        add = 8.0
    elif ra_ft > 1000.0:
        add = 15.0
    elif ra_ft > 300.0:
        add = 10.0
    elif ra_ft > 80.0:
        add = 6.0
    else:
        add = 3.0
    return vref + add


def autopilot_step(state: dict, dt: float) -> dict:
    global _state

    # Sensor fusion.
    gs = _vote(state, [
        "glideslope_1_dots", "glideslope_2_dots", "glideslope_3_dots"
    ], [
        "glideslope_1_valid", "glideslope_2_valid", "glideslope_3_valid"
    ])
    loc = _vote(state, [
        "localizer_1_dots", "localizer_2_dots", "localizer_3_dots"
    ], [
        "localizer_1_valid", "localizer_2_valid", "localizer_3_valid"
    ])
    ias = _vote(state, ["ias_1_kt", "ias_2_kt", "ias_3_kt"], [
        "ias_1_valid", "ias_2_valid", "ias_3_valid"
    ])
    aoa = _vote(state, ["aoa_1_deg", "aoa_2_deg", "aoa_3_deg"], [
        "aoa_1_valid", "aoa_2_valid", "aoa_3_valid"
    ])
    pitch = _vote(state, ["irs_1_pitch_deg", "irs_2_pitch_deg", "irs_3_pitch_deg"], [
        "irs_1_valid", "irs_2_valid", "irs_3_valid"
    ])
    roll = _vote(state, ["irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"], [
        "irs_1_valid", "irs_2_valid", "irs_3_valid"
    ])
    heading = _vote(state, ["irs_1_heading_deg", "irs_2_heading_deg", "irs_3_heading_deg"], [
        "irs_1_valid", "irs_2_valid", "irs_3_valid"
    ])
    track = _vote(state, ["irs_1_track_deg", "irs_2_track_deg", "irs_3_track_deg"], [
        "irs_1_valid", "irs_2_valid", "irs_3_valid"
    ])
    vs = _vote(state, ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"], [
        "irs_1_valid", "irs_2_valid", "irs_3_valid"
    ])
    gs_kt = _vote(state, ["gps_1_gs_kt", "gps_2_gs_kt", "gps_3_gs_kt"], [
        "gps_1_valid", "gps_2_valid", "gps_3_valid"
    ])
    baro_alt = _vote(state, ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"], [
        "baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"
    ])

    ra = _first_number(state, ["radio_alt_ft"])
    if ra is not None and ra < 0.0:
        ra = None

    on_ground = _flag(state.get("on_ground"))
    wow = _flag(state.get("weight_on_wheels"))
    below_dh = _flag(state.get("below_decision_height"))

    target_vref = _first_number(state, ["target_vref_kt"])
    target_ias = _first_number(state, ["target_ias_kt"])
    target_hdg = _first_number(state, ["target_hdg_deg"])
    target_loc = _first_number(state, ["target_loc_dots"])
    target_gs = _first_number(state, ["target_gs_dots"])
    target_vs = _first_number(state, ["target_vs_fpm"])
    target_alt = _first_number(state, ["target_alt_ft"])

    phase = _phase(ra, on_ground, wow, gs)
    speed = ias if ias is not None else gs_kt
    if speed is None:
        speed = target_ias if target_ias is not None else 137.0

    vref = target_vref if target_vref is not None else 137.0
    desired_speed = _approach_speed(vref, ra)
    if phase == "rollout":
        desired_speed = 0.0
    elif phase == "flare":
        desired_speed = vref + 2.0
    elif phase == "landing":
        desired_speed = vref + 4.0

    # Flap and gear management.
    flaps = _flap_schedule(phase, ra, ias, target_ias)
    gear = 1.0 if phase != "intercept" or (ra is not None and ra < 3000.0) else 0.0

    # Go-around only if we are clearly unstable while still high.
    stable = True
    if ra is not None and ra > 200.0:
        if gs is not None and abs(gs) > 2.0:
            stable = False
        if loc is not None and abs(loc) > 2.5:
            stable = False
        if ias is not None and abs(ias - desired_speed) > 35.0:
            stable = False
    if target_alt is not None and target_vs is not None and target_alt > 1000.0 and target_vs >= 0.0:
        stable = False
    if phase in ("intercept", "approach") and not below_dh and not stable and ra is not None and ra > 300.0:
        return {
            "throttle_1": 1.0,
            "throttle_2": 1.0,
            "elevator": 0.35,
            "elevator_trim": 0.0,
            "aileron": 0.0,
            "rudder": 0.0,
            "flaps": 20.0,
            "gear": 1.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
        }

    # Longitudinal control.
    speed_err = desired_speed - speed
    _state["speed_int"] = _clamp(_state["speed_int"] + speed_err * (dt if dt else 1.0) * 0.01, -0.6, 0.6)

    if phase == "rollout":
        elevator = -0.2
    elif phase == "flare":
        ra_use = ra if ra is not None else 0.0
        flare_factor = _clamp(ra_use / 50.0, 0.0, 1.0)
        desired_vs = -650.0 * flare_factor
        if vs is None:
            vs = -700.0
        vs_err = desired_vs - vs
        elevator = _clamp(0.0012 * vs_err + 0.018 * speed_err + _state["speed_int"], -1.0, 1.0)
    elif phase == "landing":
        desired_vs = -700.0 if ra is not None and ra > 80.0 else -350.0
        if gs is not None:
            desired_vs += 120.0 * _clamp(-gs, -2.0, 2.0)
        if vs is None:
            vs = desired_vs
        vs_err = desired_vs - vs
        elevator = _clamp(0.0010 * vs_err + 0.020 * speed_err + 0.12 * (_state["speed_int"]), -1.0, 1.0)
    elif phase == "approach":
        desired_vs = -850.0
        if gs is not None:
            desired_vs += 160.0 * _clamp(-gs, -2.0, 2.0)
        if vs is None:
            vs = desired_vs
        vs_err = desired_vs - vs
        elevator = _clamp(0.0009 * vs_err + 0.022 * speed_err + 0.10 * (_state["speed_int"]), -1.0, 1.0)
    else:
        desired_vs = -1000.0
        if gs is not None:
            desired_vs += 200.0 * _clamp(-gs, -2.0, 2.0)
        if vs is None:
            vs = desired_vs
        vs_err = desired_vs - vs
        elevator = _clamp(0.0008 * vs_err + 0.015 * speed_err + 0.08 * (_state["speed_int"]), -1.0, 1.0)

    # Trim follows the average elevator demand to keep the stick near center.
    _state["trim_bias"] = _clamp(_state["trim_bias"] + elevator * 0.02, -0.35, 0.35)
    elevator_trim = _state["trim_bias"]

    # Lateral control.
    if loc is None:
        loc = 0.0
    if gs is None:
        gs = 0.0
    bank_err = 0.0 if roll is None else roll

    if phase == "rollout":
        aileron = _clamp(-0.03 * bank_err - 0.15 * loc, -1.0, 1.0)
        rudder = _clamp(-0.04 * _ang_diff(track if track is not None else 0.0,
                                          target_hdg if target_hdg is not None else (heading if heading is not None else 0.0))
                        - 0.10 * aileron, -1.0, 1.0)
    else:
        # Positive localizer dots mean runway is to the right, so command right bank.
        aileron = _clamp(0.55 * loc - 0.03 * bank_err, -1.0, 1.0)
        if heading is not None and track is not None:
            yaw_err = _ang_diff(track, heading)
        else:
            yaw_err = 0.0
        engine_1_run = _flag(state.get("engine_1_running"))
        engine_2_run = _flag(state.get("engine_2_running"))
        thrust_asym = 0.0
        if engine_1_run != engine_2_run:
            thrust_asym = -0.35 if engine_1_run and not engine_2_run else 0.35
        rudder = _clamp(0.25 * aileron + 0.03 * yaw_err + thrust_asym, -1.0, 1.0)

    # Thrust management.
    eng1_run = _flag(state.get("engine_1_running"))
    eng2_run = _flag(state.get("engine_2_running"))
    engine_1_fire = _flag(state.get("engine_1_fire"))
    engine_2_fire = _flag(state.get("engine_2_fire"))

    if phase == "rollout":
        throttle_1 = 0.0
        throttle_2 = 0.0
        reverse_1 = 1.0 if eng1_run and speed > 45.0 else 0.0
        reverse_2 = 1.0 if eng2_run and speed > 45.0 else 0.0
    else:
        base = 0.62 if phase == "intercept" else 0.50 if phase == "approach" else 0.30
        base += 0.010 * speed_err
        if phase == "flare":
            base = 0.08
        if phase == "landing":
            base = 0.16 + 0.008 * speed_err
        if target_vs is not None and target_vs > 0.0 and ra is not None and ra < 1500.0:
            base = max(base, 0.70)
        base = _clamp(base, 0.0, 1.0)
        throttle_1 = 0.0 if engine_1_fire else (base if eng1_run or not eng2_run else 0.0)
        throttle_2 = 0.0 if engine_2_fire else (base if eng2_run or not eng1_run else 0.0)
        # If one engine is out, keep the live engine a bit higher to hold path.
        if eng1_run and not eng2_run:
            throttle_1 = _clamp(base + 0.05, 0.0, 1.0)
        if eng2_run and not eng1_run:
            throttle_2 = _clamp(base + 0.05, 0.0, 1.0)
        reverse_1 = 0.0
        reverse_2 = 0.0

    # Ground spoilers and braking.
    touchdown = bool(wow or (on_ground and ra is not None and ra < 10.0))
    if touchdown:
        _state["touchdown_latched"] = True
    spoiler = 1.0 if _state["touchdown_latched"] else 0.0
    if _state["touchdown_latched"]:
        if speed > 120.0:
            autobrake = 0.2
        elif speed > 80.0:
            autobrake = 0.5
        elif speed > 60.0:
            autobrake = 0.7
        else:
            autobrake = 1.0
        if state.get("runway_wet", False) or state.get("runway_icy", False):
            autobrake = _clamp(autobrake + 0.2, 0.4, 1.0)
    else:
        autobrake = 0.0

    # Speedbrake helps on approach if fast, otherwise stay retracted.
    speedbrake = 0.0
    if phase in ("intercept", "approach", "landing") and ias is not None and ias > desired_speed + 15.0:
        speedbrake = _clamp((ias - desired_speed - 15.0) / 40.0, 0.0, 0.6)
    if phase == "rollout":
        speedbrake = 1.0

    # Conservative flap protection: never retract below current setting, only extend.
    # The simulator can ignore invalid transitions; this still provides stable commands.
    out = {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": _clamp(elevator_trim, -1.0, 1.0),
        "flaps": _clamp(flaps, 0.0, 40.0),
        "gear": 1.0 if gear >= 0.5 else 0.0,
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "spoiler": 1.0 if spoiler >= 0.5 else 0.0,
        "autobrake": _clamp(autobrake, 0.0, 1.0),
        "reverse_1": _clamp(reverse_1, 0.0, 1.0),
        "reverse_2": _clamp(reverse_2, 0.0, 1.0),
    }

    _state["last_on_ground"] = on_ground or wow
    return out
