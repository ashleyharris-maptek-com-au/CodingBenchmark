const fs = require('fs');

(function main() {
  const raw = fs.readFileSync(0, 'utf8').trim();
  if (!raw) {
    process.stdout.write(JSON.stringify({ routes: [] }));
    return;
  }

  const data = JSON.parse(raw);
  const packages = Array.isArray(data.packages) ? data.packages : [];
  const n = packages.length;
  if (n === 0) {
    process.stdout.write(JSON.stringify({ routes: [] }));
    return;
  }

  const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
  const depotX = Number(depot[0]) || 0;
  const depotY = Number(depot[1]) || 0;
  const truckCountRaw = Number(data.truckCount);
  const truckCount = Number.isFinite(truckCountRaw) && truckCountRaw > 0 ? Math.floor(truckCountRaw) : 1;

  const xs = new Float64Array(n);
  const ys = new Float64Array(n);
  const ids = new Array(n);
  const depotDist = new Float64Array(n);
  let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;

  for (let i = 0; i < n; i++) {
    const p = packages[i] || {};
    const x = Number(p.x) || 0;
    const y = Number(p.y) || 0;
    xs[i] = x;
    ys[i] = y;
    ids[i] = p.id;
    const dx = x - depotX;
    const dy = y - depotY;
    depotDist[i] = Math.sqrt(dx * dx + dy * dy);
    if (x < minX) minX = x;
    if (y < minY) minY = y;
    if (x > maxX) maxX = x;
    if (y > maxY) maxY = y;
  }

  const maxRoutes = Math.min(truckCount, n, Math.max(1, Math.min(16, Math.ceil(Math.sqrt(n)))));
  const INF = 1e100;
  const EPS = 1e-9;

  function reverseCopy(arr) {
    const out = arr.slice();
    out.reverse();
    return out;
  }

  function rotateCopy(arr, start) {
    if (start <= 0) return arr.slice();
    return arr.slice(start).concat(arr.slice(0, start));
  }

  function makeSorted(compare) {
    const idx = new Array(n);
    for (let i = 0; i < n; i++) idx[i] = i;
    idx.sort(compare);
    return idx;
  }

  function evaluateOrder(order) {
    const pref = new Float64Array(n + 1);
    const dep = new Float64Array(n + 1);
    pref[0] = 0;
    for (let pos = 1; pos <= n; pos++) {
      const idx = order[pos - 1];
      dep[pos] = depotDist[idx];
      if (pos > 1) {
        const a = order[pos - 2];
        const b = idx;
        const dx = xs[a] - xs[b];
        const dy = ys[a] - ys[b];
        pref[pos] = pref[pos - 1] + Math.sqrt(dx * dx + dy * dy);
      }
    }

    let prev = new Float64Array(n + 1);
    prev.fill(INF);
    prev[0] = 0;

    const parents = new Array(maxRoutes + 1);
    const costs = new Float64Array(maxRoutes + 1);
    costs.fill(INF);
    costs[0] = 0;

    for (let t = 1; t <= maxRoutes; t++) {
      const cur = new Float64Array(n + 1);
      cur.fill(INF);
      const parent = new Int32Array(n + 1);
      parent.fill(-1);

      let bestVal = INF;
      let bestIdx = -1;
      for (let j = t; j <= n; j++) {
        const cand = j - 1;
        const pv = prev[cand];
        if (pv < INF / 2) {
          const val = pv - pref[cand + 1] + dep[cand + 1];
          if (val < bestVal) {
            bestVal = val;
            bestIdx = cand;
          }
        }
        if (bestIdx >= 0) {
          cur[j] = dep[j] + pref[j] + bestVal;
          parent[j] = bestIdx;
        }
      }

      parents[t] = parent;
      costs[t] = cur[n];
      prev = cur;
    }

    let bestT = 1;
    let bestCost = costs[1];
    for (let t = 2; t <= maxRoutes; t++) {
      const c = costs[t];
      if (c < bestCost - EPS || (Math.abs(c - bestCost) <= EPS && t < bestT)) {
        bestCost = c;
        bestT = t;
      }
    }

    const routes = new Array(bestT);
    let j = n;
    for (let t = bestT; t >= 1; t--) {
      const i = parents[t][j];
      const segment = new Array(j - i);
      for (let pos = i + 1, k = 0; pos <= j; pos++, k++) {
        segment[k] = ids[order[pos - 1]];
      }
      routes[t - 1] = segment;
      j = i;
    }

    return { cost: bestCost, routes };
  }

  function compareNum(a, b) {
    return a - b;
  }

  const orders = [];

  const angle = new Float64Array(n);
  for (let i = 0; i < n; i++) {
    angle[i] = Math.atan2(ys[i] - depotY, xs[i] - depotX);
  }
  const angleBase = makeSorted((a, b) => {
    const da = angle[a] - angle[b];
    if (da !== 0) return da;
    const ra = depotDist[a] - depotDist[b];
    if (ra !== 0) return ra;
    return a - b;
  });
  const anglePivots = [0, Math.floor(n / 4), Math.floor(n / 2), Math.floor((3 * n) / 4)];
  for (const pivot of anglePivots) {
    const rot = rotateCopy(angleBase, pivot);
    orders.push(rot);
    orders.push(reverseCopy(rot));
  }

  const xOrder = makeSorted((a, b) => {
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });
  orders.push(xOrder);
  orders.push(reverseCopy(xOrder));

  const yOrder = makeSorted((a, b) => {
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    return a - b;
  });
  orders.push(yOrder);
  orders.push(reverseCopy(yOrder));

  const diagOrder = makeSorted((a, b) => {
    const sa = xs[a] + ys[a];
    const sb = xs[b] + ys[b];
    const d = sa - sb;
    if (d !== 0) return d;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });
  orders.push(diagOrder);
  orders.push(reverseCopy(diagOrder));

  const antiDiagOrder = makeSorted((a, b) => {
    const sa = xs[a] - ys[a];
    const sb = xs[b] - ys[b];
    const d = sa - sb;
    if (d !== 0) return d;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });
  orders.push(antiDiagOrder);
  orders.push(reverseCopy(antiDiagOrder));

  const scaleX = maxX > minX ? 32767 / (maxX - minX) : 0;
  const scaleY = maxY > minY ? 32767 / (maxY - minY) : 0;
  const mortonKey = new Uint32Array(n);
  for (let i = 0; i < n; i++) {
    const ix = scaleX === 0 ? 0 : Math.max(0, Math.min(32767, Math.floor((xs[i] - minX) * scaleX)));
    const iy = scaleY === 0 ? 0 : Math.max(0, Math.min(32767, Math.floor((ys[i] - minY) * scaleY)));
    let code = 0;
    for (let bit = 0; bit < 15; bit++) {
      code += ((ix >> bit) & 1) << (2 * bit);
      code += ((iy >> bit) & 1) << (2 * bit + 1);
    }
    mortonKey[i] = code >>> 0;
  }
  const mortonOrder = makeSorted((a, b) => {
    const d = mortonKey[a] - mortonKey[b];
    if (d !== 0) return d;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });
  orders.push(mortonOrder);
  orders.push(reverseCopy(mortonOrder));

  let best = { cost: INF, routes: [[ids[0]]] };
  for (const order of orders) {
    const cand = evaluateOrder(order);
    if (
      cand.cost < best.cost - EPS ||
      (Math.abs(cand.cost - best.cost) <= EPS && cand.routes.length < best.routes.length)
    ) {
      best = cand;
    }
  }

  process.stdout.write(JSON.stringify({ routes: best.routes }));
})();
