// 3D regular tetrahedron packing heuristic for a convex polyhedral container.
//
// Strategy:
// 1. Build a convex hull of the supplied vertices.
// 2. Pack tetrahedra in horizontal layers using an equilateral-triangle tiling.
// 3. Accept only tetrahedra whose 4 vertices are inside the hull.
//
// This is a scalable heuristic: it is exact for the inside test on convex
// containers and uses a simple periodic lattice for placement.

#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <unordered_map>
#include <utility>
#include <vector>

using std::pair;
using std::size_t;
using std::vector;

struct Vec3 {
    double x = 0, y = 0, z = 0;
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
};

static inline Vec3 operator+(const Vec3& a, const Vec3& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}
static inline Vec3 operator-(const Vec3& a, const Vec3& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}
static inline Vec3 operator*(const Vec3& a, double s) {
    return {a.x * s, a.y * s, a.z * s};
}
static inline Vec3 operator/(const Vec3& a, double s) {
    return {a.x / s, a.y / s, a.z / s};
}
static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}
static inline double norm2(const Vec3& a) { return dot(a, a); }

struct Face {
    int a = -1, b = -1, c = -1;
    Vec3 n;
    double d = 0.0;
    bool dead = false;
};

struct Placement {
    Vec3 center;
    double rx = 0.0, ry = 0.0, rz = 0.0;
};

static inline std::uint64_t edgeKey(int u, int v) {
    if (u > v) std::swap(u, v);
    return (std::uint64_t(std::uint32_t(u)) << 32) | std::uint32_t(v);
}

static void makeFace(Face& f, int a, int b, int c, const vector<Vec3>& p, const Vec3& inside) {
    f.a = a; f.b = b; f.c = c;
    f.n = cross(p[b] - p[a], p[c] - p[a]);
    f.d = dot(f.n, p[a]);
    if (dot(f.n, inside) > f.d) {
        std::swap(f.b, f.c);
        f.n = cross(p[f.b] - p[f.a], p[f.c] - p[f.a]);
        f.d = dot(f.n, p[f.a]);
    }
}

static double planeDist(const Face& f, const Vec3& x) {
    return dot(f.n, x) - f.d;
}

static bool pointInsideHull(const Vec3& x, const vector<Face>& faces, double eps) {
    for (const auto& f : faces) {
        if (f.dead) continue;
        double tol = eps * (1.0 + std::sqrt(norm2(f.n)));
        if (planeDist(f, x) > tol) return false;
    }
    return true;
}

static bool buildConvexHull(const vector<Vec3>& pts, vector<Face>& faces, Vec3& inside, double eps) {
    const int n = (int)pts.size();
    if (n < 4) return false;

    int i0 = 0;
    int i1 = -1;
    for (int i = 1; i < n; ++i) {
        if (norm2(pts[i] - pts[i0]) > eps * eps) {
            i1 = i;
            break;
        }
    }
    if (i1 < 0) return false;

    int i2 = -1;
    Vec3 base = pts[i1] - pts[i0];
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1) continue;
        Vec3 cr = cross(base, pts[i] - pts[i0]);
        if (norm2(cr) > eps * eps) {
            i2 = i;
            break;
        }
    }
    if (i2 < 0) return false;

    int i3 = -1;
    Vec3 n012 = cross(pts[i1] - pts[i0], pts[i2] - pts[i0]);
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1 || i == i2) continue;
        double vol6 = std::fabs(dot(n012, pts[i] - pts[i0]));
        if (vol6 > eps * eps * eps) {
            i3 = i;
            break;
        }
    }
    if (i3 < 0) return false;

    inside = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;
    faces.clear();
    faces.reserve(2 * n + 8);

    auto addInitialFace = [&](int a, int b, int c, int opp) {
        Face f;
        f.a = a; f.b = b; f.c = c;
        f.n = cross(pts[b] - pts[a], pts[c] - pts[a]);
        f.d = dot(f.n, pts[a]);
        if (dot(f.n, pts[opp] - pts[a]) > 0.0) {
            std::swap(f.b, f.c);
            f.n = cross(pts[f.b] - pts[f.a], pts[f.c] - pts[f.a]);
            f.d = dot(f.n, pts[f.a]);
        }
        if (dot(f.n, inside) > f.d) {
            std::swap(f.b, f.c);
            f.n = cross(pts[f.b] - pts[f.a], pts[f.c] - pts[f.a]);
            f.d = dot(f.n, pts[f.a]);
        }
        faces.push_back(f);
    };

    addInitialFace(i0, i1, i2, i3);
    addInitialFace(i0, i3, i1, i2);
    addInitialFace(i0, i2, i3, i1);
    addInitialFace(i1, i3, i2, i0);

    vector<int> order;
    order.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (i != i0 && i != i1 && i != i2 && i != i3) order.push_back(i);
    }

    for (int idx : order) {
        const Vec3& p = pts[idx];
        vector<int> visible;
        for (int fi = 0; fi < (int)faces.size(); ++fi) {
            if (faces[fi].dead) continue;
            double tol = eps * (1.0 + std::sqrt(norm2(faces[fi].n)));
            if (planeDist(faces[fi], p) > tol) visible.push_back(fi);
        }
        if (visible.empty()) continue;

        std::unordered_map<std::uint64_t, int> cnt;
        std::unordered_map<std::uint64_t, pair<int, int>> oriented;
        cnt.reserve(visible.size() * 6);
        oriented.reserve(visible.size() * 6);

        auto addEdge = [&](int u, int v) {
            auto key = edgeKey(u, v);
            ++cnt[key];
            if (!oriented.count(key)) oriented[key] = {u, v};
        };

        for (int fi : visible) {
            const Face& f = faces[fi];
            addEdge(f.a, f.b);
            addEdge(f.b, f.c);
            addEdge(f.c, f.a);
        }

        std::unordered_map<int, int> next;
        std::unordered_map<int, int> prev;
        next.reserve(cnt.size());
        prev.reserve(cnt.size());
        int horizonEdges = 0;
        for (const auto& kv : cnt) {
            if (kv.second != 1) continue;
            auto [u, v] = oriented[kv.first];
            next[u] = v;
            prev[v] = u;
            ++horizonEdges;
        }

        for (int fi : visible) faces[fi].dead = true;
        if (horizonEdges == 0) continue;

        int start = -1;
        for (const auto& kv : next) {
            if (!prev.count(kv.first)) {
                start = kv.first;
                break;
            }
        }
        if (start < 0) start = next.begin()->first;

        vector<int> cycle;
        cycle.reserve(horizonEdges + 1);
        int cur = start;
        for (int step = 0; step <= horizonEdges + 1; ++step) {
            cycle.push_back(cur);
            auto it = next.find(cur);
            if (it == next.end()) break;
            cur = it->second;
            if (cur == start) break;
        }

        if ((int)cycle.size() < 3) continue;

        // Add the fan of new faces around the new point.
        for (int i = 0; i < (int)cycle.size(); ++i) {
            int u = cycle[i];
            int v = cycle[(i + 1) % cycle.size()];
            Face nf;
            makeFace(nf, u, v, idx, pts, inside);
            faces.push_back(nf);
        }
    }

    return true;
}

static vector<Placement> generatePackings(
    const vector<Face>& faces,
    const Vec3& bbMin,
    const Vec3& bbMax,
    double e,
    size_t cap)
{
    const double pi = std::acos(-1.0);
    const double h = e * std::sqrt(2.0 / 3.0);
    const double rowH = e * std::sqrt(3.0) * 0.5;

    vector<Placement> best;

    auto tryPhase = [&](double ox, double oy, double oz) {
        vector<Placement> out;
        out.reserve(cap);

        const double x0 = bbMin.x - 2.0 * e + ox;
        const double y0 = bbMin.y - 2.0 * rowH + oy;
        const double z0 = bbMin.z - 2.0 * h + oz;

        const int jMin = (int)std::floor((bbMin.y - y0) / rowH) - 2;
        const int jMax = (int)std::ceil((bbMax.y - y0) / rowH) + 2;
        const int kMin = (int)std::floor((bbMin.z - z0) / h) - 2;
        const int kMax = (int)std::ceil((bbMax.z - z0) / h) + 2;

        for (int k = kMin; k <= kMax && out.size() < cap; ++k) {
            double z = z0 + k * h;
            for (int j = jMin; j <= jMax && out.size() < cap; ++j) {
                double y = y0 + j * rowH;
                double xBase = x0 + j * (e * 0.5);

                const int iMin = (int)std::floor((bbMin.x - xBase) / e) - 3;
                const int iMax = (int)std::ceil((bbMax.x - xBase) / e) + 3;

                for (int i = iMin; i <= iMax && out.size() < cap; ++i) {
                    double x = xBase + i * e;

                    // Up triangle tetrahedron.
                    {
                        Vec3 a{x, y, z};
                        Vec3 b{x + e, y, z};
                        Vec3 c{x + e * 0.5, y + rowH, z};
                        Vec3 apex{(a.x + b.x + c.x) / 3.0, (a.y + b.y + c.y) / 3.0, z + h};
                        Vec3 center = (a + b + c + apex) / 4.0;

                        if (pointInsideHull(a, faces, 1e-9) &&
                            pointInsideHull(b, faces, 1e-9) &&
                            pointInsideHull(c, faces, 1e-9) &&
                            pointInsideHull(apex, faces, 1e-9))
                        {
                            out.push_back({center, 0.0, 0.0, 0.0});
                        }
                    }

                    if (out.size() >= cap) break;

                    // Down triangle tetrahedron.
                    {
                        Vec3 a{x + e * 0.5, y + rowH, z};
                        Vec3 b{x + 1.5 * e, y + rowH, z};
                        Vec3 c{x + e, y, z};
                        Vec3 apex{(a.x + b.x + c.x) / 3.0, (a.y + b.y + c.y) / 3.0, z + h};
                        Vec3 center = (a + b + c + apex) / 4.0;

                        if (pointInsideHull(a, faces, 1e-9) &&
                            pointInsideHull(b, faces, 1e-9) &&
                            pointInsideHull(c, faces, 1e-9) &&
                            pointInsideHull(apex, faces, 1e-9))
                        {
                            out.push_back({center, 0.0, 0.0, pi});
                        }
                    }
                }
            }
        }
        return out;
    };

    const double phaseX[2] = {0.0, e * 0.5};
    const double phaseY[2] = {0.0, rowH * 0.5};
    const double phaseZ[2] = {0.0, h * 0.5};

    for (double ox : phaseX) {
        for (double oy : phaseY) {
            for (double oz : phaseZ) {
                vector<Placement> cur = tryPhase(ox, oy, oz);
                if (cur.size() > best.size()) best = std::move(cur);
            }
        }
    }

    return best;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V = 0;
    double e = 1.0;
    if (!(std::cin >> V >> e)) return 0;

    vector<Vec3> pts;
    pts.reserve(std::max(0, V));
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        pts.push_back({x, y, z});
    }

    if (pts.empty() || V < 4 || e <= 0.0) {
        std::cout << 0 << '\n';
        return 0;
    }

    Vec3 bbMin{
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity()
    };
    Vec3 bbMax{
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity()
    };
    for (const auto& p : pts) {
        bbMin.x = std::min(bbMin.x, p.x);
        bbMin.y = std::min(bbMin.y, p.y);
        bbMin.z = std::min(bbMin.z, p.z);
        bbMax.x = std::max(bbMax.x, p.x);
        bbMax.y = std::max(bbMax.y, p.y);
        bbMax.z = std::max(bbMax.z, p.z);
    }

    vector<Face> faces;
    Vec3 inside;
    double hullEps = 1e-10 * std::max({1.0,
                                      std::fabs(bbMax.x - bbMin.x),
                                      std::fabs(bbMax.y - bbMin.y),
                                      std::fabs(bbMax.z - bbMin.z)});
    if (!buildConvexHull(pts, faces, inside, hullEps)) {
        std::cout << 0 << '\n';
        return 0;
    }

    // Remove dead faces from the hull representation used for inside tests.
    vector<Face> hullFaces;
    hullFaces.reserve(faces.size());
    for (const auto& f : faces) if (!f.dead) hullFaces.push_back(f);

    // If the hull is extremely small, there may be no room for a tetrahedron.
    const double h = e * std::sqrt(2.0 / 3.0);
    if (bbMax.z - bbMin.z + 1e-12 < h) {
        // Still possible if the hull is tilted, but the conservative layer
        // construction below will not find any placements.
    }

    size_t cap = 50000;
    vector<Placement> placements = generatePackings(hullFaces, bbMin, bbMax, e, cap);

    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(10);
    std::cout << placements.size() << '\n';
    for (const auto& t : placements) {
        std::cout << t.center.x << ' ' << t.center.y << ' ' << t.center.z << ' '
                  << t.rx << ' ' << t.ry << ' ' << t.rz << '\n';
    }
    return 0;
}
