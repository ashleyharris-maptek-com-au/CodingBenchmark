use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut val = 0usize;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            val = val * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        val
    }
}

#[derive(Clone, Copy, Eq, PartialEq)]
struct HeapItem {
    gain: usize,
    tie: u64,
    v: usize,
}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        self.gain
            .cmp(&other.gain)
            .then_with(|| self.tie.cmp(&other.tie))
            .then_with(|| self.v.cmp(&other.v))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn hash64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    x = (x ^ (x >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94D049BB133111EB);
    x ^ (x >> 31)
}

fn greedy_solution(adj: &[Vec<usize>], seed: u64) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let mut gain = vec![0usize; n];
    let mut dominated = vec![false; n];
    let mut selected = vec![false; n];
    let mut heap = BinaryHeap::with_capacity(n);

    for v in 0..n {
        let g = adj[v].len() + 1;
        gain[v] = g;
        heap.push(HeapItem {
            gain: g,
            tie: hash64((v as u64) ^ seed),
            v,
        });
    }

    let mut dominated_count = 0usize;
    let mut chosen = Vec::new();
    let mut newly = Vec::new();

    while dominated_count < n {
        let v = loop {
            let item = match heap.pop() {
                Some(x) => x,
                None => {
                    let mut fallback = None;
                    for i in 0..n {
                        if !dominated[i] {
                            fallback = Some(i);
                            break;
                        }
                    }
                    if let Some(i) = fallback {
                        break i;
                    }
                    break 0;
                }
            };

            if selected[item.v] {
                continue;
            }
            if item.gain != gain[item.v] {
                continue;
            }
            break item.v;
        };

        selected[v] = true;
        chosen.push(v);

        newly.clear();
        if !dominated[v] {
            dominated[v] = true;
            dominated_count += 1;
            newly.push(v);
        }
        for &u in &adj[v] {
            if !dominated[u] {
                dominated[u] = true;
                dominated_count += 1;
                newly.push(u);
            }
        }

        for &x in &newly {
            if gain[x] > 0 {
                gain[x] -= 1;
                heap.push(HeapItem {
                    gain: gain[x],
                    tie: hash64((x as u64) ^ seed),
                    v: x,
                });
            }
            for &y in &adj[x] {
                if gain[y] > 0 {
                    gain[y] -= 1;
                    heap.push(HeapItem {
                        gain: gain[y],
                        tie: hash64((y as u64) ^ seed),
                        v: y,
                    });
                }
            }
        }
    }

    chosen
}

fn prune_solution(adj: &[Vec<usize>], chosen: &mut Vec<usize>) {
    let n = adj.len();
    let mut cover = vec![0usize; n];

    for &v in chosen.iter() {
        cover[v] += 1;
        for &u in &adj[v] {
            cover[u] += 1;
        }
    }

    chosen.sort_unstable_by_key(|&v| (adj[v].len(), v));

    let mut kept = Vec::with_capacity(chosen.len());
    for &v in chosen.iter() {
        let mut removable = cover[v] > 1;
        if removable {
            for &u in &adj[v] {
                if cover[u] <= 1 {
                    removable = false;
                    break;
                }
            }
        }

        if removable {
            cover[v] -= 1;
            for &u in &adj[v] {
                cover[u] -= 1;
            }
        } else {
            kept.push(v);
        }
    }

    *chosen = kept;
}

fn repair_to_valid(adj: &[Vec<usize>], chosen: &mut Vec<usize>) {
    let n = adj.len();
    let mut in_set = vec![false; n];
    let mut dominated = vec![false; n];

    for &v in chosen.iter() {
        if v < n && !in_set[v] {
            in_set[v] = true;
            dominated[v] = true;
            for &u in &adj[v] {
                dominated[u] = true;
            }
        }
    }

    for v in 0..n {
        if !dominated[v] {
            if !in_set[v] {
                in_set[v] = true;
                chosen.push(v);
            }
            dominated[v] = true;
            for &u in &adj[v] {
                dominated[u] = true;
            }
        }
    }
}

fn solve(adj: &[Vec<usize>]) -> Vec<usize> {
    let seeds = [0u64, 0x9E3779B97F4A7C15u64];
    let mut best = Vec::new();

    for &seed in &seeds {
        let mut cand = greedy_solution(adj, seed);
        prune_solution(adj, &mut cand);
        repair_to_valid(adj, &mut cand);
        cand.sort_unstable();
        cand.dedup();

        if best.is_empty() || cand.len() < best.len() {
            best = cand;
        }
    }

    if best.is_empty() && !adj.is_empty() {
        best = (0..adj.len()).collect();
    }

    best
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let ans = solve(&adj);

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", ans.len()).unwrap();
    if ans.is_empty() {
        writeln!(out).unwrap();
    } else {
        for (i, v) in ans.iter().enumerate() {
            if i > 0 {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", v).unwrap();
        }
        writeln!(out).unwrap();
    }
}
