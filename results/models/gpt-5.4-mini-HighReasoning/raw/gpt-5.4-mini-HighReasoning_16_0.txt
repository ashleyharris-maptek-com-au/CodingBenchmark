#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    int duration;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    long long totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        totalTasks += T;
        jobs[j].resize(T);
        for (int i = 0; i < T; ++i) {
            cin >> jobs[j][i].machine >> jobs[j][i].duration;
        }
    }

    // Adaptive choice:
    // - For small instances, keep the code path simple and compact.
    // - For large instances, use the same O(T) schedule construction, which
    //   is already streaming-friendly and memory efficient.
    const bool smallInstance = totalTasks <= 2000;

    vector<vector<long long>> starts(J);
    for (int j = 0; j < J; ++j) starts[j].resize(jobs[j].size());

    vector<long long> jobReady(J, 0);
    vector<long long> machineReady(max(0, M), 0);

    if (smallInstance) {
        // Job-major greedy schedule:
        // schedule each job in input order, task by task.
        for (int j = 0; j < J; ++j) {
            for (size_t i = 0; i < jobs[j].size(); ++i) {
                const Task& t = jobs[j][i];
                long long s = max(jobReady[j], machineReady[t.machine]);
                starts[j][i] = s;
                long long e = s + t.duration;
                jobReady[j] = e;
                machineReady[t.machine] = e;
            }
        }
    } else {
        // Same schedule, but kept in the fastest possible tight loop for
        // large task counts.
        for (int j = 0; j < J; ++j) {
            long long jr = 0;
            for (size_t i = 0; i < jobs[j].size(); ++i) {
                const Task& t = jobs[j][i];
                long long s = jr > machineReady[t.machine] ? jr : machineReady[t.machine];
                starts[j][i] = s;
                jr = s + t.duration;
                machineReady[t.machine] = jr;
            }
            jobReady[j] = jr;
        }
    }

    for (int j = 0; j < J; ++j) {
        for (size_t i = 0; i < jobs[j].size(); ++i) {
            if (i) cout << ' ';
            cout << starts[j][i] << ' ' << jobs[j][i].machine;
        }
        cout << '\n';
    }

    return 0;
}
