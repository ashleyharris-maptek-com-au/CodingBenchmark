#include <algorithm>
#include <iostream>
#include <map>
#include <set>
#include <tuple>
#include <vector>

using namespace std;

struct Rect {
    int w = 0;
    int h = 0;
    int idx = 0;
    int mx = 0;
    int mn = 0;
};

struct Board {
    int usedH = 0;
    int remH = 0;
};

struct Shelf {
    int h = 0;
    int rem = 0;
    int board = -1;
    int y = 0;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rot = 0;
};

struct Candidate {
    bool valid = false;
    int stage = 3;     // 0 = existing shelf, 1 = existing board new shelf, 2 = new board
    int level = 0;     // shelf height
    int extra = 0;     // leftover width or leftover board height
    int boardId = -1;
    int shelfId = -1;
    int w = 0;
    int h = 0;
    int rot = 0;
};

struct Packer {
    int W = 0;
    int H = 0;
    vector<set<pair<int, int>>> shelvesByH;
    vector<int> seg;
    vector<Board> boards;
    map<int, set<int>> boardsByRem;
    vector<Shelf> shelves;

    Packer(int boardW, int boardH) : W(boardW), H(boardH), shelvesByH(boardH + 1), seg(4 * (boardH + 5), 0) {}

    void updateSeg(int node, int l, int r, int pos) {
        if (l == r) {
            seg[node] = shelvesByH[pos].empty() ? 0 : shelvesByH[pos].rbegin()->first;
            return;
        }
        int mid = (l + r) >> 1;
        if (pos <= mid) {
            updateSeg(node << 1, l, mid, pos);
        } else {
            updateSeg(node << 1 | 1, mid + 1, r, pos);
        }
        seg[node] = max(seg[node << 1], seg[node << 1 | 1]);
    }

    int findHeight(int node, int l, int r, int ql, int needW) const {
        if (r < ql || seg[node] < needW) {
            return -1;
        }
        if (l == r) {
            return l;
        }
        int mid = (l + r) >> 1;
        int left = findHeight(node << 1, l, mid, ql, needW);
        if (left != -1) {
            return left;
        }
        return findHeight(node << 1 | 1, mid + 1, r, ql, needW);
    }

    void removeBoardFromMap(int id) {
        int rem = boards[id].remH;
        auto it = boardsByRem.find(rem);
        if (it == boardsByRem.end()) {
            return;
        }
        it->second.erase(id);
        if (it->second.empty()) {
            boardsByRem.erase(it);
        }
    }

    int createBoard() {
        int id = static_cast<int>(boards.size());
        boards.push_back(Board{0, H});
        return id;
    }

    pair<int, int> bestBoardForShelf(int h) const {
        auto it = boardsByRem.lower_bound(h);
        if (it == boardsByRem.end()) {
            return {-1, H - h};
        }
        return {*it->second.begin(), it->first - h};
    }

    Candidate makeCandidate(int a, int b, int rot) const {
        Candidate cand;
        cand.w = a;
        cand.h = b;
        cand.rot = rot;

        if (a > W || b > H) {
            return cand;
        }

        int bucket = findHeight(1, 1, H, b, a);
        if (bucket != -1) {
            const auto& bucketSet = shelvesByH[bucket];
            auto it = bucketSet.lower_bound({a, -1});
            if (it != bucketSet.end()) {
                cand.valid = true;
                cand.stage = 0;
                cand.level = bucket;
                cand.extra = it->first - a;
                cand.shelfId = it->second;
                return cand;
            }
        }

        auto [boardId, boardAfter] = bestBoardForShelf(b);
        cand.valid = true;
        cand.level = b;
        cand.extra = boardAfter;
        if (boardId >= 0) {
            cand.stage = 1;
            cand.boardId = boardId;
        } else {
            cand.stage = 2;
        }
        return cand;
    }

    static bool better(const Candidate& a, const Candidate& b) {
        if (!a.valid) {
            return false;
        }
        if (!b.valid) {
            return true;
        }
        if (a.stage != b.stage) {
            return a.stage < b.stage;
        }
        if (a.level != b.level) {
            return a.level < b.level;
        }
        if (a.extra != b.extra) {
            return a.extra < b.extra;
        }
        if (a.rot != b.rot) {
            return a.rot < b.rot;
        }
        return a.w > b.w;
    }

    void placeExistingShelf(const Candidate& cand, vector<Placement>& out, int rectIdx) {
        Shelf& sh = shelves[cand.shelfId];
        Placement p;
        p.board = sh.board;
        p.x = W - sh.rem;
        p.y = sh.y;
        p.rot = cand.rot;
        out[rectIdx] = p;

        auto& bucketSet = shelvesByH[sh.h];
        auto it = bucketSet.find({sh.rem, cand.shelfId});
        if (it != bucketSet.end()) {
            bucketSet.erase(it);
        }
        sh.rem -= cand.w;
        if (sh.rem > 0) {
            bucketSet.insert({sh.rem, cand.shelfId});
        }
        updateSeg(1, 1, H, sh.h);
    }

    void placeNewShelf(const Candidate& cand, vector<Placement>& out, int rectIdx) {
        int boardId = cand.boardId;
        if (cand.stage == 2) {
            boardId = createBoard();
        } else {
            removeBoardFromMap(boardId);
        }

        Board& board = boards[boardId];
        int y = board.usedH;
        board.usedH += cand.h;
        board.remH -= cand.h;

        if (board.remH > 0) {
            boardsByRem[board.remH].insert(boardId);
        }

        int shelfId = static_cast<int>(shelves.size());
        shelves.push_back(Shelf{cand.h, W - cand.w, boardId, y});

        if (cand.h >= 1 && cand.h <= H && shelves.back().rem > 0) {
            shelvesByH[cand.h].insert({shelves.back().rem, shelfId});
            updateSeg(1, 1, H, cand.h);
        }

        Placement p;
        p.board = boardId;
        p.x = 0;
        p.y = y;
        p.rot = cand.rot;
        out[rectIdx] = p;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N = 0;
    int boardW = 0;
    int boardH = 0;
    if (!(cin >> N >> boardW >> boardH)) {
        return 0;
    }

    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
        rects[i].mx = max(rects[i].w, rects[i].h);
        rects[i].mn = min(rects[i].w, rects[i].h);
    }

    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        if (a.mx != b.mx) {
            return a.mx > b.mx;
        }
        if (a.mn != b.mn) {
            return a.mn > b.mn;
        }
        return a.idx < b.idx;
    });

    Packer packer(boardW, boardH);
    vector<Placement> placements(N);

    for (const Rect& r : rects) {
        Candidate c0 = packer.makeCandidate(r.w, r.h, 0);
        Candidate c1 = packer.makeCandidate(r.h, r.w, 1);

        Candidate best;
        if (Packer::better(c0, c1)) {
            best = c0;
        } else {
            best = c1;
        }

        if (!best.valid) {
            // Fallback for malformed inputs: try the original orientation on a fresh board.
            best.valid = true;
            best.stage = 2;
            best.level = r.h;
            best.extra = boardH - r.h;
            best.boardId = -1;
            best.shelfId = -1;
            best.w = r.w;
            best.h = r.h;
            best.rot = 0;
        }

        if (best.stage == 0) {
            packer.placeExistingShelf(best, placements, r.idx);
        } else {
            packer.placeNewShelf(best, placements, r.idx);
        }
    }

    cout << packer.boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << placements[i].board << ' ' << placements[i].x << ' ' << placements[i].y << ' ' << placements[i].rot << '\n';
    }

    return 0;
}
