#include <algorithm>
#include <iostream>
#include <limits>
#include <set>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Rect {
    int w, h, id;
};

struct Board {
    int usedH;
    int remH;
};

struct Row {
    int board;
    int y;
    int h;
    int remW;
};

struct Placement {
    int board;
    int x;
    int y;
    int rot;
};

struct Candidate {
    bool ok = false;
    int kind = 2;       // 0 = existing row, 1 = new row on board
    int score1 = 0;     // smaller is better
    int score2 = 0;     // smaller is better
    int board = -1;
    int row = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rot = 0;
};

struct HeightTree {
    int n = 0;
    vector<int> seg;

    void init(int n_) {
        n = n_;
        seg.assign(4 * (n + 5), 0);
    }

    void update(int node, int l, int r, int pos, int val) {
        if (l == r) {
            seg[node] = val;
            return;
        }
        int mid = (l + r) >> 1;
        if (pos <= mid) update(node << 1, l, mid, pos, val);
        else update(node << 1 | 1, mid + 1, r, pos, val);
        seg[node] = max(seg[node << 1], seg[node << 1 | 1]);
    }

    int findFirst(int node, int l, int r, int ql, int need) const {
        if (r < ql || seg[node] < need) return 0;
        if (l == r) return l;
        int mid = (l + r) >> 1;
        int left = findFirst(node << 1, l, mid, ql, need);
        if (left) return left;
        return findFirst(node << 1 | 1, mid + 1, r, ql, need);
    }
};

static bool betterCandidate(const Candidate& a, const Candidate& b) {
    if (!a.ok) return false;
    if (!b.ok) return true;
    if (a.kind != b.kind) return a.kind < b.kind;
    if (a.score1 != b.score1) return a.score1 < b.score1;
    if (a.score2 != b.score2) return a.score2 < b.score2;
    if (a.board != b.board) return a.board < b.board;
    if (a.row != b.row) return a.row < b.row;
    return a.rot < b.rot;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, boardW, boardH;
    if (!(cin >> N >> boardW >> boardH)) return 0;

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        int ma = max(a.w, a.h);
        int mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        int miA = min(a.w, a.h);
        int miB = min(b.w, b.h);
        if (miA != miB) return miA > miB;
        long long areaA = 1LL * a.w * a.h;
        long long areaB = 1LL * b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        return a.id < b.id;
    });

    vector<Board> boards;
    vector<Row> rows;
    vector<Placement> ans(N);

    multiset<pair<int, int>> boardSet; // (remaining height, board id)
    vector<multiset<pair<int, int>>> rowBuckets(boardH + 1);
    vector<int> bucketMax(boardH + 1, 0);
    HeightTree tree;
    tree.init(boardH);

    auto refreshBucket = [&](int h) {
        int val = rowBuckets[h].empty() ? 0 : rowBuckets[h].rbegin()->first;
        if (bucketMax[h] != val) {
            bucketMax[h] = val;
            tree.update(1, 1, boardH, h, val);
        }
    };

    auto makeRowCandidate = [&](int w, int h, int rot) -> Candidate {
        Candidate c;
        if (w > boardW || h > boardH) return c;
        int rowH = tree.findFirst(1, 1, boardH, h, w);
        if (rowH == 0) return c;
        auto& bucket = rowBuckets[rowH];
        auto it = bucket.lower_bound({w, -1});
        if (it == bucket.end()) return c;
        int rowId = it->second;
        const Row& r = rows[rowId];
        c.ok = true;
        c.kind = 0;
        c.score1 = r.h - h;
        c.score2 = r.remW - w;
        c.board = r.board;
        c.row = rowId;
        c.x = boardW - r.remW;
        c.y = r.y;
        c.w = w;
        c.h = h;
        c.rot = rot;
        return c;
    };

    auto makeBoardCandidate = [&](int w, int h, int rot) -> Candidate {
        Candidate c;
        if (w > boardW || h > boardH) return c;
        auto it = boardSet.lower_bound({h, -1});
        if (it == boardSet.end()) return c;
        int boardId = it->second;
        const Board& b = boards[boardId];
        c.ok = true;
        c.kind = 1;
        c.score1 = b.remH - h;
        c.score2 = boardW - w;
        c.board = boardId;
        c.row = -1;
        c.x = 0;
        c.y = b.usedH;
        c.w = w;
        c.h = h;
        c.rot = rot;
        return c;
    };

    for (const Rect& rect : rects) {
        Candidate best;
        Candidate c0 = makeRowCandidate(rect.w, rect.h, 0);
        Candidate c1 = makeRowCandidate(rect.h, rect.w, 1);
        Candidate b0 = makeBoardCandidate(rect.w, rect.h, 0);
        Candidate b1 = makeBoardCandidate(rect.h, rect.w, 1);

        if (betterCandidate(c0, best)) best = c0;
        if (betterCandidate(c1, best)) best = c1;
        if (betterCandidate(b0, best)) best = b0;
        if (betterCandidate(b1, best)) best = b1;

        if (!best.ok) {
            // Fallback for impossible-orientation edge cases: create a new board
            // with whichever orientation has the smaller overflow, if any.
            int w0 = rect.w, h0 = rect.h;
            int w1 = rect.h, h1 = rect.w;
            bool fit0 = (w0 <= boardW && h0 <= boardH);
            bool fit1 = (w1 <= boardW && h1 <= boardH);
            int w = fit0 ? w0 : w1;
            int h = fit0 ? h0 : h1;
            int rot = fit0 ? 0 : 1;
            if (!fit0 && fit1) {
                w = w1; h = h1; rot = 1;
            }
            if (!fit0 && !fit1) {
                // Absolute last resort: choose the orientation with the smaller
                // maximum overflow to keep the program deterministic.
                long long over0 = max(0, w0 - boardW) + max(0, h0 - boardH);
                long long over1 = max(0, w1 - boardW) + max(0, h1 - boardH);
                if (over1 < over0) {
                    w = w1; h = h1; rot = 1;
                }
            }
            int boardId = (int)boards.size();
            boards.push_back({h, boardH - h});
            if (boards.back().remH > 0) boardSet.insert({boards.back().remH, boardId});
            int rowId = (int)rows.size();
            rows.push_back({boardId, 0, h, max(0, boardW - w)});
            if (rows.back().remW > 0) {
                rowBuckets[h].insert({rows.back().remW, rowId});
                refreshBucket(h);
            }
            ans[rect.id] = {boardId, 0, 0, rot};
            continue;
        }

        if (best.kind == 0) {
            Row& r = rows[best.row];
            auto& bucket = rowBuckets[r.h];
            auto it = bucket.find({r.remW, best.row});
            if (it != bucket.end()) bucket.erase(it);
            r.remW -= best.w;
            if (r.remW > 0) {
                bucket.insert({r.remW, best.row});
            }
            refreshBucket(r.h);
            ans[rect.id] = {r.board, best.x, best.y, best.rot};
        } else {
            Board& b = boards[best.board];
            auto it = boardSet.find({b.remH, best.board});
            if (it != boardSet.end()) boardSet.erase(it);
            int y = b.usedH;
            b.usedH += best.h;
            b.remH -= best.h;
            if (b.remH > 0) boardSet.insert({b.remH, best.board});

            int rowId = (int)rows.size();
            rows.push_back({best.board, y, best.h, boardW - best.w});
            if (rows.back().remW > 0) {
                rowBuckets[best.h].insert({rows.back().remW, rowId});
                refreshBucket(best.h);
            }
            ans[rect.id] = {best.board, 0, y, best.rot};
        }
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rot << '\n';
    }
    return 0;
}
