using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double DaySeconds = 86400.0;
    private const double EarthRadius = 6371000.0;
    private const double MissMargin = 1000000.0;
    private const double TargetMissDistance = EarthRadius + MissMargin;
    private const double DefaultSunMu = 1.32712440018e20;
    private const double Eps = 1e-12;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int numBodies = fs.NextInt();
        double warningDays = fs.NextDouble();
        double impactorMass = fs.NextDouble();
        double deltaVBudget = fs.NextDouble();
        double asteroidMass = fs.NextDouble();
        double t0UnixUtc = fs.NextDouble();

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            string name = fs.Next() ?? string.Empty;
            double mass = fs.NextDouble();
            double x = fs.NextDouble();
            double y = fs.NextDouble();
            double z = fs.NextDouble();
            double vx = fs.NextDouble();
            double vy = fs.NextDouble();
            double vz = fs.NextDouble();
            bodies.Add(new Body(name, mass, new Vec3(x, y, z), new Vec3(vx, vy, vz)));
        }

        string de421Path = fs.Next() ?? string.Empty;

        if (bodies.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        Body sun = TryFindBody(bodies, "Sun", out Body foundSun) ? foundSun : bodies[0];
        Body earth = TryFindBody(bodies, "Earth", out Body foundEarth) ? foundEarth : bodies[0];
        Body asteroid = TryFindBody(bodies, "Asteroid", out Body foundAsteroid) ? foundAsteroid : bodies[bodies.Count - 1];

        double horizonDays = Math.Max(0.1, warningDays);
        double sampleStepDays = Math.Min(0.5, Math.Max(0.02, horizonDays / 4000.0));
        var samples = IntegrateSamples(bodies, horizonDays, sampleStepDays, earth.Name, asteroid.Name, sun.Name);

        double muSun = sun.Mass > 0 ? G * sun.Mass : DefaultSunMu;

        // Search a launch/impact pair. We use a coarse-to-fine grid and score candidates
        // by asteroid miss distance margin, then by launch delta-v, then by earlier impact.
        Candidate best = default;
        bool haveBest = false;

        SearchCandidates(samples, warningDays, impactorMass, asteroidMass, deltaVBudget, muSun, ref best, ref haveBest,
            coarseTfSteps: 36, coarseTlSteps: 20, fineSteps: 24, useFine: false);

        if (haveBest)
        {
            // A local refinement pass around the best coarse solution.
            SearchCandidates(samples, warningDays, impactorMass, asteroidMass, deltaVBudget, muSun, ref best, ref haveBest,
                coarseTfSteps: 18, coarseTlSteps: 18, fineSteps: 30, useFine: true, seed: best);
        }

        if (!haveBest)
        {
            // Fallback: direct linear intercept in the heliocentric frame.
            best = BuildFallbackCandidate(samples, warningDays, impactorMass, asteroidMass, deltaVBudget, muSun);
        }

        Console.WriteLine("2");
        Console.WriteLine(FormLine(best.LaunchTime, best.LaunchDv));
        Console.WriteLine(FormLine(best.ImpactTime, best.ImpactRelVel));
    }

    private static string FormLine(double timeDays, Vec3 dv)
    {
        return string.Format(CultureInfo.InvariantCulture, "{0:G17} {1:G17} {2:G17} {3:G17}", timeDays, dv.X, dv.Y, dv.Z);
    }

    private static bool TryFindBody(List<Body> bodies, string name, out Body body)
    {
        for (int i = 0; i < bodies.Count; i++)
        {
            if (string.Equals(bodies[i].Name, name, StringComparison.OrdinalIgnoreCase))
            {
                body = bodies[i];
                return true;
            }
        }
        body = default;
        return false;
    }

    private static void SearchCandidates(
        List<Sample> samples,
        double warningDays,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget,
        double muSun,
        ref Candidate best,
        ref bool haveBest,
        int coarseTfSteps,
        int coarseTlSteps,
        int fineSteps,
        bool useFine,
        Candidate seed = default)
    {
        double tfStart;
        double tfEnd;
        double tlPad;

        if (useFine)
        {
            double tfSpan = Math.Max(1.0, warningDays * 0.08);
            tfStart = Math.Max(0.1, seed.ImpactTime - tfSpan * 0.5);
            tfEnd = Math.Min(warningDays - 0.02, seed.ImpactTime + tfSpan * 0.5);
            tlPad = Math.Max(0.5, (seed.ImpactTime - seed.LaunchTime) * 0.35);
        }
        else
        {
            tfStart = Math.Max(0.1, warningDays * 0.03);
            tfEnd = Math.Max(tfStart + 0.1, warningDays - 0.02);
            tlPad = 0.0;
        }

        if (tfEnd <= tfStart)
        {
            tfStart = Math.Max(0.05, warningDays * 0.01);
            tfEnd = Math.Max(tfStart + 0.05, warningDays - 0.01);
        }

        int tfSteps = useFine ? fineSteps : coarseTfSteps;
        int tlSteps = useFine ? fineSteps : coarseTlSteps;
        if (tfSteps < 2) tfSteps = 2;
        if (tlSteps < 2) tlSteps = 2;

        for (int i = 0; i < tfSteps; i++)
        {
            double tf = Lerp(tfStart, tfEnd, i / (double)(tfSteps - 1));
            if (tf >= warningDays) continue;

            double maxTl = Math.Min(tf - 0.05, warningDays);
            if (useFine)
            {
                maxTl = Math.Min(maxTl, seed.LaunchTime + tlPad);
            }

            if (maxTl <= 0.0) continue;

            double minTl = 0.0;
            if (useFine)
            {
                minTl = Math.Max(0.0, seed.LaunchTime - tlPad);
            }

            if (maxTl <= minTl) continue;

            for (int j = 0; j < tlSteps; j++)
            {
                double tl = Lerp(minTl, maxTl, j / (double)(tlSteps - 1));
                if (tl >= tf) continue;

                if (!TryEvaluateCandidate(samples, tl, tf, warningDays, impactorMass, asteroidMass, deltaVBudget, muSun, out var cand))
                    continue;

                if (!haveBest || BetterCandidate(cand, best))
                {
                    best = cand;
                    haveBest = true;
                }
            }
        }
    }

    private static bool BetterCandidate(in Candidate a, in Candidate b)
    {
        const double tol = 1e-6;
        bool aFeasible = a.Feasible;
        bool bFeasible = b.Feasible;
        if (aFeasible != bFeasible) return aFeasible;
        if (Math.Abs(a.Margin - b.Margin) > tol) return a.Margin > b.Margin;
        if (Math.Abs(a.LaunchDvMag - b.LaunchDvMag) > tol) return a.LaunchDvMag < b.LaunchDvMag;
        return a.ImpactTime < b.ImpactTime;
    }

    private static bool TryEvaluateCandidate(
        List<Sample> samples,
        double launchTime,
        double impactTime,
        double warningDays,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget,
        double muSun,
        out Candidate candidate)
    {
        candidate = default;
        if (impactTime <= launchTime + 0.01) return false;
        if (impactTime >= warningDays) return false;

        State earthLaunch = GetState(samples, launchTime);
        State asteroidImpact = GetState(samples, impactTime);
        State sunLaunch = GetState(samples, launchTime);
        State sunImpact = GetState(samples, impactTime);

        Vec3 r1 = earthLaunch.Position - sunLaunch.SunPosition;
        Vec3 v1Body = earthLaunch.Velocity - sunLaunch.SunVelocity;
        Vec3 r2 = asteroidImpact.Position - sunImpact.SunPosition;
        Vec3 v2Body = asteroidImpact.Velocity - sunImpact.SunVelocity;

        double tof = (impactTime - launchTime) * DaySeconds;
        if (tof <= 0.0) return false;

        if (!TryLambert(r1, r2, tof, muSun, out Vec3 vDep, out Vec3 vArr))
        {
            // Fallback: straight-line approximation in the heliocentric frame.
            Vec3 straight = (r2 - r1) / tof;
            vDep = straight;
            vArr = straight;
        }

        Vec3 launchDv = vDep - v1Body;
        double launchMag = launchDv.Length();

        Vec3 impactRelVel = vArr - v2Body;
        double impactSpeed = impactRelVel.Length();

        double remaining = (warningDays - impactTime) * DaySeconds;
        double astDv = impactSpeed * Math.Max(0.0, impactorMass) / Math.Max(1.0, asteroidMass);
        double deflection = astDv * remaining;
        double margin = deflection - TargetMissDistance;
        bool feasible = launchMag <= deltaVBudget + 1e-9 && margin >= 0.0;

        candidate = new Candidate
        {
            LaunchTime = launchTime,
            ImpactTime = impactTime,
            LaunchDv = launchDv,
            ImpactRelVel = impactRelVel,
            LaunchDvMag = launchMag,
            ImpactSpeed = impactSpeed,
            Deflection = deflection,
            Margin = margin,
            Feasible = feasible
        };
        return true;
    }

    private static Candidate BuildFallbackCandidate(List<Sample> samples, double warningDays, double impactorMass, double asteroidMass, double deltaVBudget, double muSun)
    {
        Candidate best = default;
        bool have = false;
        double tf = Math.Max(0.5, warningDays * 0.75);
        double tl = Math.Max(0.0, tf * 0.25);
        if (TryEvaluateCandidate(samples, tl, tf, warningDays, impactorMass, asteroidMass, deltaVBudget, muSun, out var cand))
        {
            best = cand;
            have = true;
        }

        if (!have)
        {
            State earth = GetState(samples, 0.0);
            State asteroid = GetState(samples, Math.Min(warningDays * 0.9, warningDays - 0.01));
            Vec3 r1 = earth.Position - earth.SunPosition;
            Vec3 r2 = asteroid.Position - asteroid.SunPosition;
            double tof = Math.Max(1.0, warningDays * 0.9) * DaySeconds;
            if (!TryLambert(r1, r2, tof, muSun, out Vec3 vDep, out Vec3 vArr))
            {
                vDep = (r2 - r1) / tof;
                vArr = vDep;
            }

            best = new Candidate
            {
                LaunchTime = 0.0,
                ImpactTime = Math.Min(warningDays - 0.01, warningDays * 0.9),
                LaunchDv = vDep - earth.Velocity,
                ImpactRelVel = vArr - asteroid.Velocity,
                LaunchDvMag = (vDep - earth.Velocity).Length(),
                ImpactSpeed = (vArr - asteroid.Velocity).Length(),
                Deflection = 0.0,
                Margin = double.NegativeInfinity,
                Feasible = false
            };
        }

        return best;
    }

    private static State GetState(List<Sample> samples, double timeDays)
    {
        if (timeDays <= samples[0].Time)
        {
            return samples[0].ToState();
        }

        int last = samples.Count - 1;
        if (timeDays >= samples[last].Time)
        {
            return samples[last].ToState();
        }

        int lo = 0, hi = last;
        while (hi - lo > 1)
        {
            int mid = (lo + hi) >> 1;
            if (samples[mid].Time <= timeDays) lo = mid;
            else hi = mid;
        }

        Sample a = samples[lo];
        Sample b = samples[hi];
        double span = b.Time - a.Time;
        double t = span <= 0.0 ? 0.0 : (timeDays - a.Time) / span;
        return new State
        {
            SunPosition = Lerp(a.SunPosition, b.SunPosition, t),
            SunVelocity = Lerp(a.SunVelocity, b.SunVelocity, t),
            Position = Lerp(a.EarthPosition, b.EarthPosition, t),
            Velocity = Lerp(a.EarthVelocity, b.EarthVelocity, t),
            AstPosition = Lerp(a.AsteroidPosition, b.AsteroidPosition, t),
            AstVelocity = Lerp(a.AsteroidVelocity, b.AsteroidVelocity, t)
        };
    }

    private static List<Sample> IntegrateSamples(List<Body> bodies, double horizonDays, double stepDays, string earthName, string asteroidName, string sunName)
    {
        int n = bodies.Count;
        var pos = new Vec3[n];
        var vel = new Vec3[n];
        var mass = new double[n];
        for (int i = 0; i < n; i++)
        {
            pos[i] = bodies[i].Position;
            vel[i] = bodies[i].Velocity;
            mass[i] = bodies[i].Mass;
        }

        int earthIndex = IndexOfBody(bodies, earthName);
        int asteroidIndex = IndexOfBody(bodies, asteroidName);
        int sunIndex = IndexOfBody(bodies, sunName);

        var result = new List<Sample>(Math.Max(16, (int)(horizonDays / stepDays) + 2));
        double t = 0.0;
        while (true)
        {
            result.Add(new Sample
            {
                Time = t,
                SunPosition = sunIndex >= 0 ? pos[sunIndex] : Vec3.Zero,
                SunVelocity = sunIndex >= 0 ? vel[sunIndex] : Vec3.Zero,
                EarthPosition = earthIndex >= 0 ? pos[earthIndex] : Vec3.Zero,
                EarthVelocity = earthIndex >= 0 ? vel[earthIndex] : Vec3.Zero,
                AsteroidPosition = asteroidIndex >= 0 ? pos[asteroidIndex] : Vec3.Zero,
                AsteroidVelocity = asteroidIndex >= 0 ? vel[asteroidIndex] : Vec3.Zero
            });

            if (t >= horizonDays) break;

            double dtDays = Math.Min(stepDays, horizonDays - t);
            double dt = dtDays * DaySeconds;
            var acc = ComputeAccelerations(pos, mass);

            for (int i = 0; i < n; i++)
            {
                vel[i] += acc[i] * (0.5 * dt);
                pos[i] += vel[i] * dt;
            }

            var acc2 = ComputeAccelerations(pos, mass);
            for (int i = 0; i < n; i++)
            {
                vel[i] += acc2[i] * (0.5 * dt);
            }

            t += dtDays;
            if (t > horizonDays) t = horizonDays;
        }

        return result;
    }

    private static Vec3[] ComputeAccelerations(Vec3[] pos, double[] mass)
    {
        int n = pos.Length;
        var acc = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                Vec3 d = pos[j] - pos[i];
                double r2 = d.Dot(d) + 1e-6;
                double invR = 1.0 / Math.Sqrt(r2);
                double invR3 = invR * invR * invR;
                Vec3 a = d * (G * invR3);
                acc[i] += a * mass[j];
                acc[j] -= a * mass[i];
            }
        }
        return acc;
    }

    private static int IndexOfBody(List<Body> bodies, string name)
    {
        for (int i = 0; i < bodies.Count; i++)
        {
            if (string.Equals(bodies[i].Name, name, StringComparison.OrdinalIgnoreCase))
            {
                return i;
            }
        }
        return -1;
    }

    private static bool TryLambert(Vec3 r1, Vec3 r2, double tof, double mu, out Vec3 v1, out Vec3 v2)
    {
        v1 = default;
        v2 = default;

        double r1Mag = r1.Length();
        double r2Mag = r2.Length();
        if (r1Mag < 1e-9 || r2Mag < 1e-9 || tof <= 0.0 || mu <= 0.0)
            return false;

        double cosNu = Clamp(r1.Dot(r2) / (r1Mag * r2Mag), -1.0, 1.0);
        double sinNuMag = r1.Cross(r2).Length() / (r1Mag * r2Mag);
        if (sinNuMag < 1e-12 && cosNu < -0.999999999)
            return false;

        Vec3? bestV1 = null;
        Vec3? bestV2 = null;
        double bestScore = double.PositiveInfinity;

        foreach (double sign in new[] { -1.0, 1.0 })
        {
            double onePlusCos = 1.0 + cosNu;
            if (onePlusCos <= 1e-15) continue;
            double A = sign * Math.Sqrt(r1Mag * r2Mag * onePlusCos);
            if (Math.Abs(A) < 1e-15) continue;

            if (SolveLambertZ(r1Mag, r2Mag, A, tof, mu, out double z))
            {
                double C = StumpffC(z);
                double S = StumpffS(z);
                if (C <= 0.0) continue;
                double sqrtC = Math.Sqrt(C);
                double y = r1Mag + r2Mag + A * (z * S - 1.0) / sqrtC;
                if (y <= 0.0) continue;

                double f = 1.0 - y / r1Mag;
                double g = A * Math.Sqrt(y / mu);
                double gdot = 1.0 - y / r2Mag;
                if (Math.Abs(g) < 1e-12) continue;

                Vec3 dep = (r2 - r1 * f) / g;
                Vec3 arr = (r2 * gdot - r1) / g;

                double score = dep.Length() + arr.Length();
                if (score < bestScore)
                {
                    bestScore = score;
                    bestV1 = dep;
                    bestV2 = arr;
                }
            }
        }

        if (bestV1.HasValue)
        {
            v1 = bestV1.Value;
            v2 = bestV2!.Value;
            return true;
        }

        return false;
    }

    private static bool SolveLambertZ(double r1Mag, double r2Mag, double A, double tof, double mu, out double z)
    {
        z = 0.0;
        Func<double, double> F = zz =>
        {
            double C = StumpffC(zz);
            double S = StumpffS(zz);
            if (C <= 0.0) return double.NaN;
            double sqrtC = Math.Sqrt(C);
            double y = r1Mag + r2Mag + A * (zz * S - 1.0) / sqrtC;
            if (y <= 0.0) return double.NaN;
            double x = y / C;
            if (x <= 0.0) return double.NaN;
            return Math.Sqrt(x * x * x) * S + A * Math.Sqrt(y) - Math.Sqrt(mu) * tof;
        };

        double f0 = F(0.0);
        if (double.IsNaN(f0))
        {
            f0 = 0.0;
        }
        if (Math.Abs(f0) < 1e-8)
        {
            z = 0.0;
            return true;
        }

        double low = 0.0, high = 0.0;
        double flow = f0, fhigh = f0;
        bool bracketed = false;
        double step = 0.5;

        for (int k = 1; k <= 60 && !bracketed; k++)
        {
            double zp = k * step;
            double fp = F(zp);
            if (!double.IsNaN(fp) && SignChanges(f0, fp))
            {
                low = 0.0;
                high = zp;
                flow = f0;
                fhigh = fp;
                bracketed = true;
                break;
            }

            double zn = -k * step;
            double fn = F(zn);
            if (!double.IsNaN(fn) && SignChanges(f0, fn))
            {
                low = zn;
                high = 0.0;
                flow = fn;
                fhigh = f0;
                bracketed = true;
                break;
            }
        }

        if (!bracketed)
        {
            low = -64.0;
            high = 64.0;
            flow = F(low);
            fhigh = F(high);
            int expand = 0;
            while ((!double.IsNaN(flow) && !double.IsNaN(fhigh) && !SignChanges(flow, fhigh)) && expand < 10)
            {
                low *= 2.0;
                high *= 2.0;
                flow = F(low);
                fhigh = F(high);
                expand++;
            }

            if (double.IsNaN(flow) || double.IsNaN(fhigh) || !SignChanges(flow, fhigh))
                return false;
            bracketed = true;
        }

        double a = low;
        double b = high;
        double fa = flow;
        double fb = fhigh;
        for (int iter = 0; iter < 80; iter++)
        {
            double mid = 0.5 * (a + b);
            double fm = F(mid);
            if (double.IsNaN(fm))
            {
                mid += 1e-6;
                fm = F(mid);
                if (double.IsNaN(fm)) return false;
            }

            if (Math.Abs(fm) < 1e-8)
            {
                z = mid;
                return true;
            }

            if (SignChanges(fa, fm))
            {
                b = mid;
                fb = fm;
            }
            else
            {
                a = mid;
                fa = fm;
            }
        }

        z = 0.5 * (a + b);
        return true;
    }

    private static bool SignChanges(double a, double b)
    {
        return (a <= 0.0 && b >= 0.0) || (a >= 0.0 && b <= 0.0);
    }

    private static double StumpffC(double z)
    {
        if (Math.Abs(z) < 1e-8) return 0.5;
        if (z > 0.0)
        {
            double s = Math.Sqrt(z);
            return (1.0 - Math.Cos(s)) / z;
        }
        else
        {
            double s = Math.Sqrt(-z);
            return (Math.Cosh(s) - 1.0) / (-z);
        }
    }

    private static double StumpffS(double z)
    {
        if (Math.Abs(z) < 1e-8) return 1.0 / 6.0;
        if (z > 0.0)
        {
            double s = Math.Sqrt(z);
            return (s - Math.Sin(s)) / (s * s * s);
        }
        else
        {
            double s = Math.Sqrt(-z);
            return (Math.Sinh(s) - s) / (s * s * s);
        }
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private static double Lerp(double a, double b, double t) => a + (b - a) * t;
    private static Vec3 Lerp(Vec3 a, Vec3 b, double t) => a + (b - a) * t;

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public static Vec3 Zero => new Vec3(0, 0, 0);

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Dot(Vec3 other) => X * other.X + Y * other.Y + Z * other.Z;
        public Vec3 Cross(Vec3 o) => new Vec3(
            Y * o.Z - Z * o.Y,
            Z * o.X - X * o.Z,
            X * o.Y - Y * o.X);

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator *(double k, Vec3 a) => a * k;
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string? Next()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return null;
            } while (c <= 32);

            var sb = new System.Text.StringBuilder();
            while (c > 32 && c != -1)
            {
                sb.Append((char)c);
                c = ReadByte();
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            return int.Parse(Next() ?? "0", CultureInfo.InvariantCulture);
