declare const require: any;
const fs = require('fs');

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };
type InputData = { grid: string[]; robots: Robot[]; jobs: Job[] };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as InputData;

const grid = data.grid ?? [];
const robots = (data.robots ?? []).map((r) => ({
  id: Number(r.id),
  x: Number(r.x),
  y: Number(r.y),
}));
const jobs = (data.jobs ?? []).map((j) => ({
  id: Number(j.id),
  x: Number(j.x),
  y: Number(j.y),
  priority: Number(j.priority ?? 0),
}));

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const cellCount = h * w;
const indexOf = (x: number, y: number): number => y * w + x;
const inBounds = (x: number, y: number): boolean => x >= 0 && x < w && y >= 0 && y < h;
const isOpen = (x: number, y: number): boolean => inBounds(x, y) && grid[y][x] !== '#';

let maxRobotId = -1;
for (const r of robots) maxRobotId = Math.max(maxRobotId, r.id);
for (const j of jobs) {
  // Jobs do not affect output indexing, but keep the pass symmetrical and cheap.
}

const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

// Multi-source BFS from all robots. Each reachable cell inherits the nearest robot
// and its shortest walking distance. Ties are broken by robot id via seed order.
const owner = new Int32Array(cellCount);
const dist = new Int32Array(cellCount);
owner.fill(-1);
dist.fill(-1);

const queue = new Int32Array(cellCount);
let qh = 0;
let qt = 0;

const robotsById = robots.slice().sort((a, b) => a.id - b.id);
for (const r of robotsById) {
  if (!isOpen(r.x, r.y)) continue;
  const idx = indexOf(r.x, r.y);
  if (owner[idx] !== -1) continue;
  owner[idx] = r.id;
  dist[idx] = 0;
  queue[qt++] = idx;
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];
while (qh < qt) {
  const idx = queue[qh++];
  const y = Math.floor(idx / w);
  const x = idx - y * w;
  const nextDist = dist[idx] + 1;
  const nextOwner = owner[idx];

  for (let k = 0; k < 4; k++) {
    const nx = x + dx[k];
    const ny = y + dy[k];
    if (!isOpen(nx, ny)) continue;
    const nidx = indexOf(nx, ny);
    if (owner[nidx] !== -1) continue;
    owner[nidx] = nextOwner;
    dist[nidx] = nextDist;
    queue[qt++] = nidx;
  }
}

// Precompute a deterministic angular key around each robot start. This makes each
// robot's route sweep outward in a stable pattern instead of bouncing between sides.
const robotPos = new Map<number, { x: number; y: number }>();
for (const r of robots) robotPos.set(r.id, { x: r.x, y: r.y });

type RoutedJob = {
  id: number;
  x: number;
  y: number;
  priority: number;
  d: number;
  angle: number;
};

const perRobot: RoutedJob[][] = Array.from({ length: maxRobotId + 1 }, () => []);
for (const job of jobs) {
  let assignedRobot = -1;
  let d = Number.POSITIVE_INFINITY;
  const jIdx = inBounds(job.x, job.y) ? indexOf(job.x, job.y) : -1;
  if (jIdx >= 0 && owner[jIdx] !== -1) {
    assignedRobot = owner[jIdx];
    d = dist[jIdx];
  } else {
    // Fallback for unexpected unreachable cells: choose the closest robot by Manhattan distance.
    for (const r of robots) {
      const md = Math.abs(r.x - job.x) + Math.abs(r.y - job.y);
      if (md < d || (md === d && r.id < assignedRobot)) {
        d = md;
        assignedRobot = r.id;
      }
    }
    if (assignedRobot === -1 && robots.length > 0) assignedRobot = robotsById[0].id;
  }

  const rp = robotPos.get(assignedRobot);
  const angle = rp ? Math.atan2(job.y - rp.y, job.x - rp.x) : 0;
  perRobot[assignedRobot].push({
    id: job.id,
    x: job.x,
    y: job.y,
    priority: job.priority,
    d: Number.isFinite(d) ? d : 1e9,
    angle,
  });
}

for (let rid = 0; rid < perRobot.length; rid++) {
  const rp = robotPos.get(rid);
  perRobot[rid].sort((a, b) => {
    if (a.d !== b.d) return a.d - b.d;
    if (a.priority !== b.priority) return b.priority - a.priority;
    if (rp) {
      if (a.angle !== b.angle) return a.angle - b.angle;
      const ra = Math.atan2(a.y - rp.y, a.x - rp.x);
      const rb = Math.atan2(b.y - rp.y, b.x - rp.x);
      if (ra !== rb) return ra - rb;
    }
    if (a.y !== b.y) return a.y - b.y;
    if (a.x !== b.x) return a.x - b.x;
    return a.id - b.id;
  });
  assignments[rid] = perRobot[rid].map((j) => j.id);
}

process.stdout.write(JSON.stringify({ assignments }));
