use std::io::{self, Read, Write};

const NONE: u32 = u32::MAX;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, idx: 0 })
    }

    fn next_u32(&mut self) -> Option<u32> {
        while self.idx < self.buf.len() && self.buf[self.idx] <= b' ' {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut value: u32 = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if b.is_ascii_digit() {
                value = value.saturating_mul(10).saturating_add((b - b'0') as u32);
                self.idx += 1;
            } else {
                break;
            }
        }
        Some(value)
    }
}

struct Solver {
    n: usize,
    off: Vec<usize>,
    adj: Vec<u32>,
    deg: Vec<u32>,
    covered: Vec<u8>,
    forced: Vec<u8>,
    leaf_q: Vec<u32>,
    leaf_head: usize,
    use_buckets: bool,
    bucket_head: Vec<u32>,
    bucket_prev: Vec<u32>,
    bucket_next: Vec<u32>,
    current_max: usize,
}

impl Solver {
    fn new(n: usize, off: Vec<usize>, adj: Vec<u32>, deg: Vec<u32>, forced: Vec<u8>) -> Self {
        Self {
            n,
            off,
            adj,
            deg,
            covered: vec![0; n],
            forced,
            leaf_q: Vec::new(),
            leaf_head: 0,
            use_buckets: false,
            bucket_head: Vec::new(),
            bucket_prev: Vec::new(),
            bucket_next: Vec::new(),
            current_max: 0,
        }
    }

    #[inline]
    fn bucket_remove(&mut self, v: usize, d: u32) {
        let idx = d as usize;
        let prev = self.bucket_prev[v];
        let next = self.bucket_next[v];
        if prev == NONE {
            self.bucket_head[idx] = next;
        } else {
            self.bucket_next[prev as usize] = next;
        }
        if next != NONE {
            self.bucket_prev[next as usize] = prev;
        }
        self.bucket_prev[v] = NONE;
        self.bucket_next[v] = NONE;
    }

    #[inline]
    fn bucket_insert(&mut self, v: usize, d: u32) {
        let idx = d as usize;
        let head = self.bucket_head[idx];
        self.bucket_prev[v] = NONE;
        self.bucket_next[v] = head;
        if head != NONE {
            self.bucket_prev[head as usize] = v as u32;
        }
        self.bucket_head[idx] = v as u32;
    }

    #[inline]
    fn cover_vertex(&mut self, v: usize, remove_self_from_bucket: bool) {
        if self.covered[v] != 0 {
            return;
        }
        self.covered[v] = 1;

        if self.use_buckets && remove_self_from_bucket && self.deg[v] > 0 {
            self.bucket_remove(v, self.deg[v]);
        }

        let start = self.off[v];
        let end = self.off[v + 1];
        for idx in start..end {
            let u = self.adj[idx] as usize;
            if self.covered[u] != 0 {
                continue;
            }
            let old = self.deg[u];
            if old == 0 {
                continue;
            }
            let new = old - 1;
            self.deg[u] = new;
            if self.use_buckets {
                self.bucket_remove(u, old);
                if new > 0 {
                    self.bucket_insert(u, new);
                    if new == 1 {
                        self.leaf_q.push(u as u32);
                    }
                }
            } else if new == 1 {
                self.leaf_q.push(u as u32);
            }
        }
    }

    fn seed_leaves(&mut self) {
        self.leaf_q.clear();
        self.leaf_head = 0;
        for v in 0..self.n {
            if self.covered[v] == 0 && self.deg[v] == 1 {
                self.leaf_q.push(v as u32);
            }
        }
    }

    fn process_leaf_queue(&mut self) {
        while self.leaf_head < self.leaf_q.len() {
            let v = self.leaf_q[self.leaf_head] as usize;
            self.leaf_head += 1;

            if self.covered[v] != 0 || self.deg[v] != 1 {
                continue;
            }

            let mut chosen = None;
            for idx in self.off[v]..self.off[v + 1] {
                let u = self.adj[idx] as usize;
                if self.covered[u] == 0 {
                    chosen = Some(u);
                    break;
                }
            }

            if let Some(u) = chosen {
                self.cover_vertex(u, true);
            }
        }

        self.leaf_q.clear();
        self.leaf_head = 0;
    }

    fn rebuild_buckets(&mut self) {
        let mut max_deg = 0usize;
        for v in 0..self.n {
            if self.covered[v] == 0 {
                max_deg = max_deg.max(self.deg[v] as usize);
            }
        }

        self.bucket_head = vec![NONE; max_deg + 1];
        self.bucket_prev = vec![NONE; self.n];
        self.bucket_next = vec![NONE; self.n];
        self.use_buckets = true;
        self.current_max = 0;

        for v in 0..self.n {
            if self.covered[v] == 0 {
                let d = self.deg[v];
                if d > 0 {
                    self.bucket_insert(v, d);
                    if d as usize > self.current_max {
                        self.current_max = d as usize;
                    }
                }
            }
        }
    }

    #[inline]
    fn bucket_pop_head(&mut self, d: usize) -> Option<usize> {
        let v = self.bucket_head[d];
        if v == NONE {
            return None;
        }
        let vi = v as usize;
        let next = self.bucket_next[vi];
        self.bucket_head[d] = next;
        if next != NONE {
            self.bucket_prev[next as usize] = NONE;
        }
        self.bucket_prev[vi] = NONE;
        self.bucket_next[vi] = NONE;
        Some(vi)
    }

    fn solve(mut self) -> Vec<u32> {
        for v in 0..self.n {
            if self.forced[v] != 0 {
                self.cover_vertex(v, true);
            }
        }

        self.seed_leaves();
        self.process_leaf_queue();

        self.rebuild_buckets();
        self.seed_leaves();

        loop {
            self.process_leaf_queue();

            while self.current_max > 0
                && (self.current_max >= self.bucket_head.len()
                    || self.bucket_head[self.current_max] == NONE)
            {
                self.current_max -= 1;
            }

            if self.current_max == 0 {
                break;
            }

            let v = match self.bucket_pop_head(self.current_max) {
                Some(v) => v,
                None => continue,
            };
            if self.covered[v] != 0 || self.deg[v] == 0 {
                continue;
            }
            self.cover_vertex(v, false);
        }

        self.process_leaf_queue();
        self.covered
            .iter()
            .enumerate()
            .filter_map(|(i, &c)| if c != 0 { Some(i as u32) } else { None })
            .collect()
    }
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;
    let n = match scanner.next_u32() {
        Some(v) => v as usize,
        None => return Ok(()),
    };
    let m = scanner.next_u32().unwrap_or(0) as usize;

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    let mut deg = vec![0u32; n];
    let mut forced = vec![0u8; n];

    for _ in 0..m {
        let u = scanner.next_u32().unwrap_or(0);
        let v = scanner.next_u32().unwrap_or(0);
        if u == v {
            if (u as usize) < n {
                forced[u as usize] = 1;
            }
            continue;
        }
        let ui = u as usize;
        let vi = v as usize;
        deg[ui] = deg[ui].saturating_add(1);
        deg[vi] = deg[vi].saturating_add(1);
        edges.push((u, v));
    }

    let mut off = vec![0usize; n + 1];
    for i in 0..n {
        off[i + 1] = off[i] + deg[i] as usize;
    }

    let mut adj = vec![0u32; off[n]];
    let mut cur = off[..n].to_vec();
    for (u, v) in edges {
        let ui = u as usize;
        let vi = v as usize;
        adj[cur[ui]] = v;
        cur[ui] += 1;
        adj[cur[vi]] = u;
        cur[vi] += 1;
    }

    let solver = Solver::new(n, off, adj, deg, forced);
    let cover = solver.solve();

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    writeln!(out, "{}", cover.len())?;
    if !cover.is_empty() {
        for (i, v) in cover.iter().enumerate() {
            if i > 0 {
                write!(out, " ")?;
            }
            write!(out, "{}", v)?;
        }
    }
    writeln!(out)?;
    Ok(())
}
