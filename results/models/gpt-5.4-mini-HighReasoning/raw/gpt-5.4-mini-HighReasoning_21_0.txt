#include <algorithm>
#include <cstdlib>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

using Coord = vector<int>;

struct CoordHash {
    size_t operator()(const Coord& v) const noexcept {
        uint64_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t z = static_cast<uint64_t>(static_cast<uint32_t>(x));
            z += 0x9e3779b97f4a7c15ull;
            z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ull;
            z = (z ^ (z >> 27)) * 0x94d049bb133111ebull;
            z ^= (z >> 31);
            h ^= z + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return static_cast<size_t>(h);
    }
};

static bool in_bounds(const Coord& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static int manhattan(const Coord& a, const Coord& b) {
    int d = 0;
    for (size_t i = 0; i < a.size(); ++i) d += abs(a[i] - b[i]);
    return d;
}

static int nearest_food_dist(const Coord& p, const vector<Coord>& foods) {
    if (foods.empty()) return 0;
    int best = numeric_limits<int>::max();
    for (const auto& f : foods) best = min(best, manhattan(p, f));
    return best;
}

static pair<int, int> choose_fallback_move(
    const Coord& head,
    const vector<int>& bounds,
    const unordered_map<Coord, bool, CoordHash>& blocked,
    const vector<Coord>& foods
) {
    int D = static_cast<int>(head.size());
    pair<int, int> best_move = {0, 1};
    long long best_score = numeric_limits<long long>::min();

    auto is_free = [&](const Coord& p) -> bool {
        return in_bounds(p, bounds) && blocked.find(p) == blocked.end();
    };

    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Coord np = head;
            np[axis] += dir;
            if (!is_free(np)) continue;

            int food_d = nearest_food_dist(np, foods);
            int freedom = 0;
            for (int a = 0; a < D; ++a) {
                for (int ndir : {-1, 1}) {
                    Coord q = np;
                    q[a] += ndir;
                    if (is_free(q)) ++freedom;
                }
            }

            long long center_bonus = 0;
            for (int i = 0; i < D; ++i) {
                long long mid2 = static_cast<long long>(bounds[i]) - 1;
                center_bonus -= llabs(2LL * np[i] - mid2);
            }

            long long score = 0;
            score -= static_cast<long long>(food_d) * 1000LL;
            score += static_cast<long long>(freedom) * 25LL;
            score += center_bonus;

            if (score > best_score) {
                best_score = score;
                best_move = {axis, dir};
            }
        }
    }

    return best_move;
}

static bool find_path_move(
    const Coord& start,
    const vector<int>& bounds,
    const unordered_map<Coord, bool, CoordHash>& blocked,
    const vector<Coord>& foods,
    pair<int, int>& out_move
) {
    if (foods.empty()) return false;

    auto is_free = [&](const Coord& p) -> bool {
        return in_bounds(p, bounds) && blocked.find(p) == blocked.end();
    };

    unordered_map<Coord, int, CoordHash> gscore;
    unordered_map<Coord, Coord, CoordHash> parent;

    struct Node {
        int f;
        int g;
        Coord p;
    };
    struct Cmp {
        bool operator()(const Node& a, const Node& b) const {
            if (a.f != b.f) return a.f > b.f;
            return a.g < b.g;
        }
    };

    priority_queue<Node, vector<Node>, Cmp> pq;
    gscore[start] = 0;
    pq.push({nearest_food_dist(start, foods), 0, start});

    const int expansion_limit = 60000;
    int expansions = 0;

    while (!pq.empty() && expansions < expansion_limit) {
        Node cur = pq.top();
        pq.pop();
        ++expansions;

        auto it_g = gscore.find(cur.p);
        if (it_g == gscore.end() || cur.g != it_g->second) continue;

        for (const auto& f : foods) {
            if (cur.p == f) {
                Coord step = cur.p;
                while (true) {
                    auto it = parent.find(step);
                    if (it == parent.end()) break;
                    if (it->second == start) {
                        Coord next = step;
                        for (int axis = 0; axis < static_cast<int>(start.size()); ++axis) {
                            int delta = next[axis] - start[axis];
                            if (delta != 0) {
                                out_move = {axis, delta > 0 ? 1 : -1};
                                return true;
                            }
                        }
                        break;
                    }
                    step = it->second;
                }
                break;
            }
        }

        int D = static_cast<int>(cur.p.size());
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Coord nxt = cur.p;
                nxt[axis] += dir;
                if (!is_free(nxt)) continue;

                int ng = cur.g + 1;
                auto it = gscore.find(nxt);
                if (it != gscore.end() && ng >= it->second) continue;

                gscore[nxt] = ng;
                parent[nxt] = cur.p;
                int nf = ng + nearest_food_dist(nxt, foods);
                pq.push({nf, ng, std::move(nxt)});
            }
        }
    }

    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    while (cin >> D >> max_turns) {
        vector<int> bounds(D);
        for (int i = 0; i < D; ++i) cin >> bounds[i];

        int snake_length;
        Coord head(D);
        cin >> snake_length;
        for (int i = 0; i < D; ++i) cin >> head[i];

        int F;
        cin >> F;
        vector<Coord> foods;
        foods.reserve(F);
        for (int i = 0; i < F; ++i) {
            Coord p(D);
            for (int j = 0; j < D; ++j) cin >> p[j];
            foods.push_back(std::move(p));
        }

        int O;
        cin >> O;
        unordered_map<Coord, bool, CoordHash> blocked;
        blocked.reserve(static_cast<size_t>(O) * 2 + 16);
        for (int i = 0; i < O; ++i) {
            Coord p(D);
            for (int j = 0; j < D; ++j) cin >> p[j];
            blocked.emplace(std::move(p), true);
        }

        pair<int, int> move = {0, 1};
        if (!find_path_move(head, bounds, blocked, foods, move)) {
            move = choose_fallback_move(head, bounds, blocked, foods);
        }

        cout << move.first << ' ' << move.second << '\n';
        cout.flush();
    }

    return 0;
}
