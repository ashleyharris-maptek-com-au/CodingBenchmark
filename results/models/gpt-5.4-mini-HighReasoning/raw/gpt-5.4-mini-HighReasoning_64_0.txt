declare const require: any;
declare const process: any;
const fs = require('fs');

const rawInput: string = fs.readFileSync(0, 'utf8');
const input = rawInput.trim();

type Drone = { id: number; x: number; y: number };

function main() {
  if (!input) {
    process.stdout.write(JSON.stringify({ paths: [] }));
    return;
  }

  const data = JSON.parse(input);
  const probability: number[][] = Array.isArray(data.probability) ? data.probability : [];
  const drones: Drone[] = Array.isArray(data.drones) ? data.drones : [];
  const maxSteps: number = Math.max(0, (data.maxStepsPerDrone | 0) || 0);

  const h = probability.length;
  const w = h > 0 ? probability[0].length : 0;
  const n = h * w;

  const values = new Float64Array(n);
  const blocked = new Uint8Array(n);
  for (let y = 0; y < h; y++) {
    const row = probability[y] || [];
    for (let x = 0; x < w; x++) {
      const idx = y * w + x;
      const v = Number(row[x]);
      values[idx] = v;
      blocked[idx] = v === -1 ? 1 : 0;
    }
  }

  const paths: number[][][] = [];
  if (n === 0 || drones.length === 0) {
    for (let i = 0; i < drones.length; i++) {
      const dx = drones[i].x | 0;
      const dy = drones[i].y | 0;
      const p: number[][] = [];
      for (let s = 0; s <= maxSteps; s++) p.push([dx, dy]);
      paths.push(p);
    }
    process.stdout.write(JSON.stringify({ paths }));
    return;
  }

  const owner = new Int32Array(n);
  const ownerDist = new Int32Array(n);
  owner.fill(-1);
  ownerDist.fill(-1);
  const q = new Int32Array(n);
  let qh = 0;
  let qt = 0;

  for (let i = 0; i < drones.length; i++) {
    const x = drones[i].x | 0;
    const y = drones[i].y | 0;
    if (x < 0 || x >= w || y < 0 || y >= h) continue;
    const idx = y * w + x;
    if (blocked[idx]) continue;
    if (ownerDist[idx] !== -1) continue;
    owner[idx] = i;
    ownerDist[idx] = 0;
    q[qt++] = idx;
  }

  while (qh < qt) {
    const cur = q[qh++];
    const curOwner = owner[cur];
    const nextDist = ownerDist[cur] + 1;
    const x = cur % w;
    const y = (cur / w) | 0;

    if (x > 0) {
      const nb = cur - 1;
      if (!blocked[nb] && ownerDist[nb] === -1) {
        ownerDist[nb] = nextDist;
        owner[nb] = curOwner;
        q[qt++] = nb;
      }
    }
    if (x + 1 < w) {
      const nb = cur + 1;
      if (!blocked[nb] && ownerDist[nb] === -1) {
        ownerDist[nb] = nextDist;
        owner[nb] = curOwner;
        q[qt++] = nb;
      }
    }
    if (y > 0) {
      const nb = cur - w;
      if (!blocked[nb] && ownerDist[nb] === -1) {
        ownerDist[nb] = nextDist;
        owner[nb] = curOwner;
        q[qt++] = nb;
      }
    }
    if (y + 1 < h) {
      const nb = cur + w;
      if (!blocked[nb] && ownerDist[nb] === -1) {
        ownerDist[nb] = nextDist;
        owner[nb] = curOwner;
        q[qt++] = nb;
      }
    }
  }

  function buildPath(droneIndex: number): number[][] {
    const sx = drones[droneIndex].x | 0;
    const sy = drones[droneIndex].y | 0;
    const start: number[][] = [[sx, sy]];

    if (maxSteps === 0 || sx < 0 || sx >= w || sy < 0 || sy >= h) {
      while (start.length < maxSteps + 1) start.push([sx, sy]);
      return start;
    }

    const root = sy * w + sx;
    if (blocked[root]) {
      while (start.length < maxSteps + 1) start.push([sx, sy]);
      return start;
    }

    const parent = new Int32Array(n);
    const depth = new Int32Array(n);
    parent.fill(-1);
    depth.fill(-1);

    const order = new Int32Array(n);
    let orderLen = 0;
    const qq = new Int32Array(n);
    let qHead = 0;
    let qTail = 0;
    qq[qTail++] = root;
    parent[root] = root;
    depth[root] = 0;

    while (qHead < qTail) {
      const cur = qq[qHead++];
      order[orderLen++] = cur;
      const d = depth[cur];
      if (d >= maxSteps) continue;
      const x = cur % w;
      const y = (cur / w) | 0;

      if (x > 0) {
        const nb = cur - 1;
        if (!blocked[nb] && depth[nb] === -1) {
          parent[nb] = cur;
          depth[nb] = d + 1;
          qq[qTail++] = nb;
        }
      }
      if (x + 1 < w) {
        const nb = cur + 1;
        if (!blocked[nb] && depth[nb] === -1) {
          parent[nb] = cur;
          depth[nb] = d + 1;
          qq[qTail++] = nb;
        }
      }
      if (y > 0) {
        const nb = cur - w;
        if (!blocked[nb] && depth[nb] === -1) {
          parent[nb] = cur;
          depth[nb] = d + 1;
          qq[qTail++] = nb;
        }
      }
      if (y + 1 < h) {
        const nb = cur + w;
        if (!blocked[nb] && depth[nb] === -1) {
          parent[nb] = cur;
          depth[nb] = d + 1;
          qq[qTail++] = nb;
        }
      }
    }

    const subtreeSum = new Float64Array(n);
    const subtreeSize = new Int32Array(n);
    for (let i = orderLen - 1; i >= 0; i--) {
      const cur = order[i];
      let sum = (cur === root || owner[cur] === droneIndex) && values[cur] > 0 ? values[cur] : 0;
      let size = 1;
      const x = cur % w;
      const y = (cur / w) | 0;

      if (x > 0) {
        const nb = cur - 1;
        if (parent[nb] === cur) {
          sum += subtreeSum[nb];
          size += subtreeSize[nb];
        }
      }
      if (x + 1 < w) {
        const nb = cur + 1;
        if (parent[nb] === cur) {
          sum += subtreeSum[nb];
          size += subtreeSize[nb];
        }
      }
      if (y > 0) {
        const nb = cur - w;
        if (parent[nb] === cur) {
          sum += subtreeSum[nb];
          size += subtreeSize[nb];
        }
      }
      if (y + 1 < h) {
        const nb = cur + w;
        if (parent[nb] === cur) {
          sum += subtreeSum[nb];
          size += subtreeSize[nb];
        }
      }

      subtreeSum[cur] = sum;
      subtreeSize[cur] = size;
    }

    if (subtreeSum[root] <= 0) {
      while (start.length < maxSteps + 1) start.push([sx, sy]);
      return start;
    }

    type Frame = { node: number; children: number[]; next: number };

    function collectChildren(node: number): number[] {
      const out: number[] = [];
      const x = node % w;
      const y = (node / w) | 0;

      if (x > 0) {
        const nb = node - 1;
        if (parent[nb] === node && subtreeSum[nb] > 0) out.push(nb);
      }
      if (x + 1 < w) {
        const nb = node + 1;
        if (parent[nb] === node && subtreeSum[nb] > 0) out.push(nb);
      }
      if (y > 0) {
        const nb = node - w;
        if (parent[nb] === node && subtreeSum[nb] > 0) out.push(nb);
      }
      if (y + 1 < h) {
        const nb = node + w;
        if (parent[nb] === node && subtreeSum[nb] > 0) out.push(nb);
      }

      out.sort((a, b) => {
        const diff = subtreeSum[b] - subtreeSum[a];
        if (diff !== 0) return diff > 0 ? 1 : -1;
        const sizeDiff = subtreeSize[a] - subtreeSize[b];
        if (sizeDiff !== 0) return sizeDiff;
        const valDiff = values[b] - values[a];
        if (valDiff !== 0) return valDiff > 0 ? 1 : -1;
        return a - b;
      });
      return out;
    }

    const stack: Frame[] = [{ node: root, children: collectChildren(root), next: 0 }];
    let steps = 0;

    while (steps < maxSteps && stack.length > 0) {
      const top = stack[stack.length - 1];
      if (top.next < top.children.length) {
        const nxt = top.children[top.next++];
        start.push([nxt % w, (nxt / w) | 0]);
        steps++;
        if (steps >= maxSteps) break;
        stack.push({ node: nxt, children: collectChildren(nxt), next: 0 });
      } else {
        stack.pop();
        if (stack.length === 0) break;
        const parentNode = stack[stack.length - 1].node;
        start.push([parentNode % w, (parentNode / w) | 0]);
        steps++;
      }
    }

    while (start.length < maxSteps + 1) start.push([sx, sy]);
    return start;
  }

  for (let i = 0; i < drones.length; i++) {
    paths.push(buildPath(i));
  }

  process.stdout.write(JSON.stringify({ paths }));
}

main();
