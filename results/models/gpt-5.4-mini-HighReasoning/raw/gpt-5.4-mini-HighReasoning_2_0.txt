using System;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int m = fs.NextInt();

        int[] head = new int[n];
        Array.Fill(head, -1);

        int halfEdgeCount = checked(m << 2);
        int copyCount = checked(m << 1);

        int[] to = new int[halfEdgeCount];
        int[] next = new int[halfEdgeCount];
        byte[] used = new byte[copyCount];

        int ptr = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            fs.NextLong(); // weight is irrelevant for the route construction

            to[ptr] = v;
            next[ptr] = head[u];
            head[u] = ptr;
            ptr++;

            to[ptr] = u;
            next[ptr] = head[v];
            head[v] = ptr;
            ptr++;

            to[ptr] = v;
            next[ptr] = head[u];
            head[u] = ptr;
            ptr++;

            to[ptr] = u;
            next[ptr] = head[v];
            head[v] = ptr;
            ptr++;
        }

        int[] stack = new int[copyCount + 1];
        int[] route = new int[copyCount + 1];
        int sp = 0;
        int rp = 0;

        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];

            int e = head[v];
            while (e != -1 && used[e >> 1] != 0)
            {
                e = next[e];
            }
            head[v] = e;

            if (e == -1)
            {
                route[rp++] = v;
                sp--;
            }
            else
            {
                used[e >> 1] = 1;
                head[v] = next[e];
                stack[sp++] = to[e];
            }
        }

        var output = new FastWriter(Console.OpenStandardOutput());
        for (int i = rp - 1; i >= 0; i--)
        {
            if (i != rp - 1) output.WriteByte((byte)' ');
            output.WriteInt(route[i]);
        }
        output.WriteByte((byte)'\n');
        output.Flush();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;

        public FastWriter(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
        }

        public void WriteByte(byte b)
        {
            if (_ptr >= _buffer.Length) Flush();
            _buffer[_ptr++] = b;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[11];
            int len = 0;
            while (value > 0)
            {
                tmp[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }

            while (len > 0)
            {
                WriteByte(tmp[--len]);
            }
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }
    }
}
