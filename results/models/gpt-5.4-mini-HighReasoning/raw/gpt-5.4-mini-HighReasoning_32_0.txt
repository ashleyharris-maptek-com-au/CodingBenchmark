#include <algorithm>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u, v;
    long long w;
};

struct AdjEdge {
    int to;
    int id;
    long long w;
};

struct DijkstraResult {
    vector<long long> dist;
    vector<int> parentV;
    vector<int> parentE;
};

struct DSU {
    vector<int> p, r;
    explicit DSU(int n) : p(n), r(n, 0) {
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static const long long INF = (1LL << 62);

DijkstraResult run_dijkstra(const vector<vector<AdjEdge>>& adj, const vector<int>& sources) {
    int n = (int)adj.size();
    DijkstraResult res;
    res.dist.assign(n, INF);
    res.parentV.assign(n, -1);
    res.parentE.assign(n, -1);

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    for (int s : sources) {
        if (s < 0 || s >= n) continue;
        if (res.dist[s] == 0) continue;
        res.dist[s] = 0;
        pq.push({0, s});
    }

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != res.dist[u]) continue;
        for (const auto& e : adj[u]) {
            long long nd = d + e.w;
            if (nd < res.dist[e.to]) {
                res.dist[e.to] = nd;
                res.parentV[e.to] = u;
                res.parentE[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }
    return res;
}

DijkstraResult run_dijkstra_single(const vector<vector<AdjEdge>>& adj, int source) {
    return run_dijkstra(adj, vector<int>{source});
}

void collect_path(int target, const vector<int>& parentV, const vector<int>& parentE, vector<char>& selectedEdge) {
    int cur = target;
    while (parentE[cur] != -1) {
        selectedEdge[parentE[cur]] = 1;
        cur = parentV[cur];
    }
}

void collect_path_and_vertices(int target, const vector<int>& parentV, const vector<int>& parentE,
                               vector<char>& selectedEdge, vector<char>& inTree, vector<int>& treeVertices) {
    int cur = target;
    while (parentE[cur] != -1) {
        selectedEdge[parentE[cur]] = 1;
        if (!inTree[cur]) {
            inTree[cur] = 1;
            treeVertices.push_back(cur);
        }
        cur = parentV[cur];
    }
    if (cur >= 0 && !inTree[cur]) {
        inTree[cur] = 1;
        treeVertices.push_back(cur);
    }
}

vector<char> build_metric_mst_candidate(const vector<vector<AdjEdge>>& adj, const vector<Edge>& edges,
                                        const vector<int>& terminals) {
    int n = (int)adj.size();
    int k = (int)terminals.size();
    vector<char> selected(edges.size(), 0);
    if (k <= 1) return selected;

    vector<vector<long long>> termDist(k, vector<long long>(k, INF));
    for (int i = 0; i < k; ++i) {
        auto dres = run_dijkstra_single(adj, terminals[i]);
        for (int j = 0; j < k; ++j) {
            termDist[i][j] = dres.dist[terminals[j]];
        }
    }

    vector<long long> key(k, INF);
    vector<int> parent(k, -1);
    vector<char> used(k, 0);
    key[0] = 0;
    for (int it = 0; it < k; ++it) {
        int v = -1;
        long long best = INF;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && key[i] < best) {
                best = key[i];
                v = i;
            }
        }
        if (v == -1) break;
        used[v] = 1;
        for (int u = 0; u < k; ++u) {
            if (!used[u] && termDist[v][u] < key[u]) {
                key[u] = termDist[v][u];
                parent[u] = v;
            }
        }
    }

    vector<vector<int>> children(k);
    for (int i = 1; i < k; ++i) {
        if (parent[i] >= 0) children[parent[i]].push_back(i);
    }

    for (int s = 0; s < k; ++s) {
        if (children[s].empty()) continue;
        auto dres = run_dijkstra_single(adj, terminals[s]);
        for (int t : children[s]) {
            if (dres.dist[terminals[t]] >= INF) continue;
            collect_path(terminals[t], dres.parentV, dres.parentE, selected);
        }
    }

    return selected;
}

vector<char> build_greedy_candidate(const vector<vector<AdjEdge>>& adj, const vector<Edge>& edges,
                                    const vector<int>& terminals) {
    int n = (int)adj.size();
    int k = (int)terminals.size();
    vector<char> selected(edges.size(), 0);
    if (k <= 1) return selected;

    vector<char> inTree(n, 0);
    vector<int> treeVertices;
    vector<char> connected(k, 0);

    int start = terminals[0];
    inTree[start] = 1;
    treeVertices.push_back(start);
    connected[0] = 1;
    int connectedCount = 1;

    while (connectedCount < k) {
        auto dres = run_dijkstra(adj, treeVertices);
        int bestIdx = -1;
        long long bestDist = INF;
        for (int i = 0; i < k; ++i) {
            if (connected[i]) continue;
            long long d = dres.dist[terminals[i]];
            if (d < bestDist) {
                bestDist = d;
                bestIdx = i;
            }
        }
        if (bestIdx == -1 || bestDist >= INF) break;
        collect_path_and_vertices(terminals[bestIdx], dres.parentV, dres.parentE, selected, inTree, treeVertices);
        connected[bestIdx] = 1;
        ++connectedCount;
    }

    return selected;
}

vector<int> finalize_tree(int n, const vector<Edge>& edges, const vector<char>& candidate, const vector<char>& isTerminal,
                          long long& totalWeight) {
    vector<int> ids;
    ids.reserve(edges.size());
    for (int i = 0; i < (int)edges.size(); ++i) {
        if (candidate[i]) ids.push_back(i);
    }
    sort(ids.begin(), ids.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });

    DSU dsu(n);
    vector<int> treeIds;
    treeIds.reserve(ids.size());
    for (int id : ids) {
        if (dsu.unite(edges[id].u, edges[id].v)) {
            treeIds.push_back(id);
        }
    }

    vector<int> deg(n, 0);
    vector<vector<int>> incident(n);
    for (int id : treeIds) {
        int u = edges[id].u, v = edges[id].v;
        incident[u].push_back(id);
        incident[v].push_back(id);
        ++deg[u];
        ++deg[v];
    }

    vector<char> alive(edges.size(), 0);
    for (int id : treeIds) alive[id] = 1;

    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (deg[v] == 1 && !isTerminal[v]) q.push(v);
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (deg[v] != 1 || isTerminal[v]) continue;

        int id = -1;
        for (int e : incident[v]) {
            if (alive[e]) {
                id = e;
                break;
            }
        }
        if (id == -1) continue;
        if (edges[id].w < 0) continue;

        alive[id] = 0;
        int u = edges[id].u, vtx = edges[id].v;
        int other = (u == v ? vtx : u);
        --deg[v];
        --deg[other];
        if (deg[other] == 1 && !isTerminal[other]) q.push(other);
    }

    vector<int> finalIds;
    totalWeight = 0;
    for (int id : treeIds) {
        if (alive[id]) {
            finalIds.push_back(id);
            totalWeight += edges[id].w;
        }
    }
    return finalIds;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<Edge> edges;
    edges.reserve(m);
    vector<vector<AdjEdge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        adj[u].push_back({v, i, w});
        adj[v].push_back({u, i, w});
    }

    vector<int> terminals;
    terminals.reserve(k);
    for (int i = 0; i < k; ++i) {
        int x;
        cin >> x;
        terminals.push_back(x);
    }
    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());

    vector<char> isTerminal(n, 0);
    for (int t : terminals) {
        if (0 <= t && t < n) isTerminal[t] = 1;
    }

    vector<char> candidate;
    if ((int)terminals.size() <= 80) {
        candidate = build_metric_mst_candidate(adj, edges, terminals);
    } else {
        candidate = build_greedy_candidate(adj, edges, terminals);
    }

    long long totalWeight = 0;
    vector<int> finalIds = finalize_tree(n, edges, candidate, isTerminal, totalWeight);

    cout << totalWeight << ' ' << finalIds.size() << '\n';
    for (int id : finalIds) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }

    return 0;
}
