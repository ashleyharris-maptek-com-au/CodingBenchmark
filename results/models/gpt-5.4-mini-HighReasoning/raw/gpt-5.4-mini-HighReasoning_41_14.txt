; SPIR-V
; Version: 1.0
OpCapability Shader
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%main = OpFunction %void None %func
%entry = OpLabel
%pos = OpLoad %v3float %worldPos
%nrm_raw = OpLoad %v3float %normal
%nrm = OpExtInst %v3float %GLSLstd450 Normalize %nrm_raw
%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%cameraPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cameraPos4 = OpLoad %v4float %cameraPosPtr
%cameraPos = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2
%lightVec = OpFSub %v3float %lightPos %pos
%lightDir = OpExtInst %v3float %GLSLstd450 Normalize %lightVec
%ndotl = OpDot %float %nrm %lightDir
%diffuse = OpExtInst %float %GLSLstd450 FMax %ndotl %float_0
%diffuseScaled = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %diffuseScaled
%baseColor = OpCompositeConstruct %v3float %float_0_8 %float_0_2 %float_0_2
%litColor = OpVectorTimesScalar %v3float %baseColor %lambert
%camVec = OpFSub %v3float %pos %cameraPos
%dist = OpExtInst %float %GLSLstd450 Length %camVec
%fogTmp = OpFSub %float %dist %float_1_5
%fogRaw = OpFDiv %float %fogTmp %float_3_0
%fogFactor = OpExtInst %float %GLSLstd450 FClamp %fogRaw %float_0 %float_1
%oneMinusFog = OpFSub %float %float_1 %fogFactor
%fogColor = OpCompositeConstruct %v3float %float_0_7 %float_0_7 %float_0_8
%litScaled = OpVectorTimesScalar %v3float %litColor %oneMinusFog
%fogScaled = OpVectorTimesScalar %v3float %fogColor %fogFactor
%rgb = OpFAdd %v3float %litScaled %fogScaled
%rgb0 = OpCompositeExtract %float %rgb 0
%rgb1 = OpCompositeExtract %float %rgb 1
%rgb2 = OpCompositeExtract %float %rgb 2
%out = OpCompositeConstruct %v4float %rgb0 %rgb1 %rgb2 %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
