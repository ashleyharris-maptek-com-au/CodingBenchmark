use std::io::{self, Read, Write};
use std::thread;
use std::time::{Duration, Instant};

struct Scanner {
    data: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { data: input.into_bytes(), idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.data[start..self.idx])
            .unwrap()
            .parse::<usize>()
            .unwrap()
    }
}

fn idx_of(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn no_obstacle_path(w: usize, h: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    if !require_cycle {
        let mut path = Vec::with_capacity(w * h);
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push((x, y));
                }
            } else {
                for x in (0..w).rev() {
                    path.push((x, y));
                }
            }
        }
        return Some(path);
    }

    if w < 2 || h < 2 {
        return None;
    }

    if h % 2 == 0 {
        let mut path = Vec::with_capacity(w * h);
        for x in 0..w {
            path.push((x, 0));
        }
        for y in 1..h {
            if y % 2 == 1 {
                for x in (1..w).rev() {
                    path.push((x, y));
                }
            } else {
                for x in 1..w {
                    path.push((x, y));
                }
            }
        }
        for y in (1..h).rev() {
            path.push((0, y));
        }
        Some(path)
    } else if w % 2 == 0 {
        let mut path = Vec::with_capacity(w * h);
        for y in 0..h {
            path.push((0, y));
        }
        for x in 1..w {
            if x % 2 == 1 {
                for y in (1..h).rev() {
                    path.push((x, y));
                }
            } else {
                for y in 1..h {
                    path.push((x, y));
                }
            }
        }
        for x in (1..w).rev() {
            path.push((x, 0));
        }
        Some(path)
    } else {
        None
    }
}

struct Search<'a> {
    neighbors: &'a [Vec<usize>],
    order: &'a [usize],
    rem_deg: Vec<i32>,
    visited: Vec<u8>,
    seen: Vec<u32>,
    stamp: u32,
    path: Vec<usize>,
    start: usize,
    total: usize,
    require_cycle: bool,
    deadline: Instant,
    nodes: u64,
}

impl<'a> Search<'a> {
    fn new(
        neighbors: &'a [Vec<usize>],
        order: &'a [usize],
        start: usize,
        total: usize,
        require_cycle: bool,
    ) -> Self {
        let mut rem_deg = neighbors.iter().map(|v| v.len() as i32).collect::<Vec<_>>();
        let mut visited = vec![0u8; total];
        let mut path = Vec::with_capacity(total);
        let seen = vec![0u32; total];

        visited[start] = 1;
        path.push(start);
        rem_deg[start] = 0;
        for &n in &neighbors[start] {
            rem_deg[n] -= 1;
        }

        Self {
            neighbors,
            order,
            rem_deg,
            visited,
            seen,
            stamp: 1,
            path,
            start,
            total,
            require_cycle,
            deadline: Instant::now() + Duration::from_secs(285),
            nodes: 0,
        }
    }

    fn time_ok(&mut self) -> bool {
        self.nodes += 1;
        if (self.nodes & 0x7ff) == 0 {
            if Instant::now() >= self.deadline {
                return false;
            }
        }
        true
    }

    fn remaining_connected(&mut self, remaining: usize) -> bool {
        if remaining <= 1 {
            return true;
        }

        let mut seed = None;
        for i in 0..self.total {
            if self.visited[i] == 0 {
                seed = Some(i);
                break;
            }
        }
        let seed = match seed {
            Some(s) => s,
            None => return true,
        };

        self.stamp = self.stamp.wrapping_add(1);
        if self.stamp == 0 {
            self.stamp = 1;
            self.seen.fill(0);
        }
        let mark = self.stamp;

        let mut q = Vec::with_capacity(remaining);
        q.push(seed);
        self.seen[seed] = mark;
        let mut head = 0usize;
        let mut count = 1usize;

        while head < q.len() {
            let u = q[head];
            head += 1;
            for &v in &self.neighbors[u] {
                if self.visited[v] == 0 && self.seen[v] != mark {
                    self.seen[v] = mark;
                    q.push(v);
                    count += 1;
                }
            }
        }

        count == remaining
    }

    fn state_ok(&mut self, visited_count: usize) -> bool {
        if !self.time_ok() {
            return false;
        }

        let remaining = self.total - visited_count;
        if remaining == 0 {
            if !self.require_cycle {
                return true;
            }
            let cur = self.path[self.path.len() - 1];
            return self.neighbors[cur].iter().any(|&n| n == self.start);
        }

        if remaining > 1 {
            let mut deg1 = 0usize;
            for i in 0..self.total {
                if self.visited[i] == 0 {
                    let d = self.rem_deg[i];
                    if d == 0 {
                        return false;
                    }
                    if d == 1 {
                        deg1 += 1;
                        if deg1 > 2 {
                            return false;
                        }
                    }
                }
            }
        }

        self.remaining_connected(remaining)
    }

    fn apply_move(&mut self, u: usize) -> (i32, Vec<(usize, i32)>) {
        let old_u = self.rem_deg[u];
        self.visited[u] = 1;
        self.rem_deg[u] = 0;
        let mut changes = Vec::with_capacity(4);
        for &v in &self.neighbors[u] {
            if self.visited[v] == 0 {
                changes.push((v, self.rem_deg[v]));
                self.rem_deg[v] -= 1;
            }
        }
        self.path.push(u);
        (old_u, changes)
    }

    fn undo_move(&mut self, u: usize, old_u: i32, changes: Vec<(usize, i32)>) {
        self.path.pop();
        self.visited[u] = 0;
        self.rem_deg[u] = old_u;
        for (v, old) in changes {
            self.rem_deg[v] = old;
        }
    }

    fn dfs(&mut self, cur: usize, visited_count: usize) -> bool {
        if !self.state_ok(visited_count) {
            return false;
        }

        let remaining = self.total - visited_count;
        if remaining == 0 {
            if self.require_cycle {
                return self.neighbors[cur].iter().any(|&n| n == self.start);
            }
            return true;
        }

        let mut candidates = Vec::with_capacity(4);
        for &n in &self.neighbors[cur] {
            if self.visited[n] == 0 {
                if remaining > 1 && self.rem_deg[n] == 0 {
                    continue;
                }
                if self.require_cycle && remaining == 1 && !self.neighbors[n].iter().any(|&m| m == self.start) {
                    continue;
                }
                candidates.push(n);
            }
        }

        if candidates.is_empty() {
            return false;
        }

        candidates.sort_by_key(|&u| (self.rem_deg[u], self.order[u]));

        for u in candidates {
            let (old_u, changes) = self.apply_move(u);
            if self.dfs(u, visited_count + 1) {
                return true;
            }
            self.undo_move(u, old_u, changes);
        }

        false
    }
}

fn solve() -> String {
    let mut sc = Scanner::new();
    let w = sc.next_usize();
    let h = sc.next_usize();
    let obstacle_count = sc.next_usize();
    let require_cycle = sc.next_usize() != 0;

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obstacle_count {
        let x = sc.next_usize();
        let y = sc.next_usize();
        if x < w && y < h {
            blocked[idx_of(x, y, w)] = true;
        }
    }

    if !blocked[idx_of(0, 0, w)] && obstacle_count == 0 {
        if let Some(path) = no_obstacle_path(w, h, require_cycle) {
            let mut out = String::new();
            out.push_str(&format!("{}\n", path.len()));
            for (x, y) in path {
                out.push_str(&format!("{} {}\n", x, y));
            }
            return out;
        } else {
            return "0\n".to_string();
        }
    }

    if blocked[idx_of(0, 0, w)] {
        return "0\n".to_string();
    }

    let mut id_of = vec![usize::MAX; n];
    let mut cells = Vec::new();
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            let id = idx_of(x, y, w);
            if !blocked[id] {
                let free_id = cells.len();
                id_of[id] = free_id;
                cells.push((x, y));
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }

    let total = cells.len();
    if total == 0 {
        return "0\n".to_string();
    }

    if require_cycle {
        if total < 4 || black != white {
            return "0\n".to_string();
        }
    } else if total % 2 == 0 {
        if black != white {
            return "0\n".to_string();
        }
    } else if black != white + 1 {
        return "0\n".to_string();
    }

    let start_coord = idx_of(0, 0, w);
    let start = id_of[start_coord];
    if start == usize::MAX {
        return "0\n".to_string();
    }

    let mut neighbors = vec![Vec::<usize>::new(); total];
    for (id, &(x, y)) in cells.iter().enumerate() {
        let mut ns = Vec::with_capacity(4);
        if x > 0 {
            let nid = idx_of(x - 1, y, w);
            if id_of[nid] != usize::MAX {
                ns.push(id_of[nid]);
            }
        }
        if x + 1 < w {
            let nid = idx_of(x + 1, y, w);
            if id_of[nid] != usize::MAX {
                ns.push(id_of[nid]);
            }
        }
        if y > 0 {
            let nid = idx_of(x, y - 1, w);
            if id_of[nid] != usize::MAX {
                ns.push(id_of[nid]);
            }
        }
        if y + 1 < h {
            let nid = idx_of(x, y + 1, w);
            if id_of[nid] != usize::MAX {
                ns.push(id_of[nid]);
            }
        }
        neighbors[id] = ns;
    }

    let mut order = vec![0usize; total];
    for (id, &(x, y)) in cells.iter().enumerate() {
        order[id] = y * w + if y % 2 == 0 { x } else { w - 1 - x };
    }

    let mut search = Search::new(&neighbors, &order, start, total, require_cycle);
    if total == 1 {
        let mut out = String::new();
        out.push_str("1\n0 0\n");
        return out;
    }

    if !search.dfs(start, 1) {
        return "0\n".to_string();
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", search.path.len()));
    for id in search.path {
        let (x, y) = cells[id];
        out.push_str(&format!("{} {}\n", x, y));
    }
    out
}

fn main() {
    let handle = thread::Builder::new()
        .stack_size(64 * 1024 * 1024)
        .spawn(|| solve())
        .unwrap();
    let output = handle.join().unwrap();
    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(output.as_bytes()).unwrap();
}
