// Maximum Independent Set via exact maximum clique search on the complement graph.
#include <algorithm>
#include <cstdint>
#include <iostream>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

static inline bool test_bit(const vector<uint64_t>& bs, int idx) {
    return (bs[idx >> 6] >> (idx & 63)) & 1ULL;
}

static inline void reset_bit(vector<uint64_t>& bs, int idx) {
    bs[idx >> 6] &= ~(1ULL << (idx & 63));
}

static inline bool is_empty(const vector<uint64_t>& bs) {
    for (uint64_t x : bs) {
        if (x) return false;
    }
    return true;
}

struct ExactMIS {
    int n = 0;
    int words = 0;
    uint64_t lastMask = ~0ULL;
    vector<vector<uint64_t>> adj;  // Complement graph adjacency within a component.
    vector<int> scanOrder;         // Local vertices sorted by ascending original degree.
    vector<int> current;
    vector<int> best;

    vector<uint64_t> fullSet() const {
        vector<uint64_t> bs(words, ~0ULL);
        if (words) bs.back() &= lastMask;
        return bs;
    }

    vector<int> greedyCliqueByOrder(const vector<int>& ord) const {
        vector<uint64_t> cand = fullSet();
        vector<int> res;
        for (int v : ord) {
            if (test_bit(cand, v)) {
                res.push_back(v);
                for (int w = 0; w < words; ++w) cand[w] &= adj[v][w];
            }
        }
        return res;
    }

    void colorSort(const vector<uint64_t>& P, vector<int>& order, vector<int>& col) const {
        vector<int> rem;
        rem.reserve(n);
        for (int v : scanOrder) {
            if (test_bit(P, v)) rem.push_back(v);
        }

        vector<uint64_t> blocked(words);
        int color = 0;
        while (!rem.empty()) {
            ++color;
            fill(blocked.begin(), blocked.end(), 0ULL);
            vector<int> next;
            next.reserve(rem.size());
            for (int v : rem) {
                if (!test_bit(blocked, v)) {
                    order.push_back(v);
                    col.push_back(color);
                    for (int w = 0; w < words; ++w) blocked[w] |= adj[v][w];
                } else {
                    next.push_back(v);
                }
            }
            rem.swap(next);
        }
    }

    void search(vector<uint64_t> P) {
        if (is_empty(P)) {
            if (current.size() > best.size()) best = current;
            return;
        }

        vector<int> order;
        vector<int> col;
        colorSort(P, order, col);

        if (current.size() + static_cast<size_t>(col.back()) <= best.size()) return;

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (current.size() + static_cast<size_t>(col[i]) <= best.size()) return;

            int v = order[i];
            current.push_back(v);

            vector<uint64_t> next = P;
            for (int w = 0; w < words; ++w) next[w] &= adj[v][w];
            search(std::move(next));

            current.pop_back();
            reset_bit(P, v);
        }
    }

    vector<int> solve() {
        if (n == 0) return {};

        vector<int> g1 = greedyCliqueByOrder(scanOrder);
        vector<int> rev = scanOrder;
        reverse(rev.begin(), rev.end());
        vector<int> g2 = greedyCliqueByOrder(rev);
        best = (g2.size() > g1.size()) ? g2 : g1;

        current.clear();
        vector<uint64_t> P = fullSet();
        search(std::move(P));
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<vector<int>> g(n);
    g.reserve(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vector<int> compId(n, -1);
    vector<vector<int>> comps;
    queue<int> q;
    for (int s = 0; s < n; ++s) {
        if (compId[s] != -1) continue;
        int cid = static_cast<int>(comps.size());
        comps.push_back({});
        compId[s] = cid;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            comps.back().push_back(u);
            for (int v : g[u]) {
                if (compId[v] == -1) {
                    compId[v] = cid;
                    q.push(v);
                }
            }
        }
    }

    vector<int> answer;
    vector<int> where(n, -1);

    for (const vector<int>& comp : comps) {
        int s = static_cast<int>(comp.size());
        if (s == 0) continue;
        if (s == 1) {
            answer.push_back(comp[0]);
            continue;
        }

        for (int i = 0; i < s; ++i) where[comp[i]] = i;

        int words = (s + 63) / 64;
        uint64_t lastMask = (s % 64 == 0) ? ~0ULL : ((1ULL << (s % 64)) - 1ULL);
        vector<vector<uint64_t>> adj(words ? s : 0, vector<uint64_t>(words, ~0ULL));
        vector<int> deg(s, 0);

        for (int i = 0; i < s; ++i) {
            adj[i][i >> 6] &= ~(1ULL << (i & 63));
        }

        for (int i = 0; i < s; ++i) {
            int gv = comp[i];
            for (int to : g[gv]) {
                int j = where[to];
                if (j > i) {
                    adj[i][j >> 6] &= ~(1ULL << (j & 63));
                    adj[j][i >> 6] &= ~(1ULL << (i & 63));
                    ++deg[i];
                    ++deg[j];
                }
            }
        }

        for (int i = 0; i < s; ++i) adj[i].back() &= lastMask;

        vector<int> order(s);
        for (int i = 0; i < s; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] < deg[b];
            return a < b;
        });

        ExactMIS solver;
        solver.n = s;
        solver.words = words;
        solver.lastMask = lastMask;
        solver.adj = std::move(adj);
        solver.scanOrder = std::move(order);

        vector<int> localBest = solver.solve();
        for (int v : localBest) answer.push_back(comp[v]);

        for (int v : comp) where[v] = -1;
    }

    sort(answer.begin(), answer.end());
    cout << answer.size() << '\n';
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
