// CSG Union solver
// Approach:
// - Build BVHs for both meshes.
// - For each triangle, find intersections with the other mesh.
// - Split intersected triangles into planar fragments using a local segment graph.
// - Classify fragments with a ray-casting inside test against the other mesh.
// - Emit the kept fragments, deduplicating vertices and exact duplicate triangles.

static constexpr double INF_D = std::numeric_limits<double>::infinity();

struct Vec2 {
    double x = 0, y = 0;
    Vec2() = default;
    Vec2(double xx, double yy) : x(xx), y(yy) {}
    Vec2 operator+(const Vec2& o) const { return {x + o.x, y + o.y}; }
    Vec2 operator-(const Vec2& o) const { return {x - o.x, y - o.y}; }
    Vec2 operator*(double s) const { return {x * s, y * s}; }
};

struct Vec3 {
    double x = 0, y = 0, z = 0;
    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec2& a, const Vec2& b) { return a.x * b.x + a.y * b.y; }
static inline double cross(const Vec2& a, const Vec2& b) { return a.x * b.y - a.y * b.x; }
static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}
static inline double norm2(const Vec2& a) { return dot(a, a); }
static inline double norm2(const Vec3& a) { return dot(a, a); }
static inline double norm(const Vec2& a) { return std::sqrt(norm2(a)); }
static inline double norm(const Vec3& a) { return std::sqrt(norm2(a)); }
static inline Vec3 normalize(const Vec3& a) {
    double n = norm(a);
    if (n == 0.0) return a;
    return a / n;
}

struct AABB {
    Vec3 mn, mx;
    AABB() {
        mn = {INF_D, INF_D, INF_D};
        mx = {-INF_D, -INF_D, -INF_D};
    }
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }
    void expand(const AABB& o) {
        expand(o.mn); expand(o.mx);
    }
};

static inline bool overlap(const AABB& a, const AABB& b, double eps) {
    return !(a.mx.x + eps < b.mn.x || b.mx.x + eps < a.mn.x ||
             a.mx.y + eps < b.mn.y || b.mx.y + eps < a.mn.y ||
             a.mx.z + eps < b.mn.z || b.mx.z + eps < a.mn.z);
}

struct TriData {
    array<int, 3> id{};
    AABB box;
    Vec3 n;
    double d = 0.0;
};

struct BVHNode {
    AABB box;
    int left = -1, right = -1;
    int start = 0, count = 0;
    bool leaf = false;
};

struct QuantKey3 {
    long long x, y, z;
    bool operator==(const QuantKey3& o) const {
        return x == o.x && y == o.y && z == o.z;
    }
};

struct QuantKey3Hash {
    size_t operator()(const QuantKey3& k) const noexcept {
        auto mix = [](uint64_t v) {
            v ^= v >> 33;
            v *= 0xff51afd7ed558ccdULL;
            v ^= v >> 33;
            v *= 0xc4ceb9fe1a85ec53ULL;
            v ^= v >> 33;
            return v;
        };
        uint64_t h1 = mix((uint64_t)k.x);
        uint64_t h2 = mix((uint64_t)k.y);
        uint64_t h3 = mix((uint64_t)k.z);
        return (size_t)(h1 ^ (h2 << 1) ^ (h3 << 2));
    }
};

static inline long long qround(double v, double scale) {
    return (long long)std::llround(v / scale);
}

static inline QuantKey3 quantKey3(const Vec3& p, double scale) {
    return {qround(p.x, scale), qround(p.y, scale), qround(p.z, scale)};
}

struct FaceKey {
    int a, b, c;
    bool operator==(const FaceKey& o) const { return a == o.a && b == o.b && c == o.c; }
};

struct FaceKeyHash {
    size_t operator()(const FaceKey& k) const noexcept {
        uint64_t x = (uint64_t)k.a * 1315423911u ^ (uint64_t)k.b * 2654435761u ^ (uint64_t)k.c * 97531u;
        x ^= x >> 33;
        x *= 0xff51afd7ed558ccdULL;
        x ^= x >> 33;
        x *= 0xc4ceb9fe1a85ec53ULL;
        x ^= x >> 33;
        return (size_t)x;
    }
};

struct Mesh {
    vector<Vec3> v;
    vector<array<int, 3>> f;
    vector<TriData> tris;
    vector<BVHNode> bvh;
    vector<int> triOrder;
    Vec3 bbMin, bbMax;
    double scale = 1.0;
    Vec3 rayDir;
    double eps = 1e-9;

    void build();
    bool pointInside(const Vec3& p) const;
    void queryTriCandidates(int node, const AABB& box, vector<int>& out) const;
    void queryRayCandidates(int node, const Vec3& orig, const Vec3& dir, vector<int>& out) const;
};

static AABB triBox(const Vec3& a, const Vec3& b, const Vec3& c) {
    AABB box;
    box.expand(a); box.expand(b); box.expand(c);
    return box;
}

static bool pointInTri3D(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c, const Vec3& n, double eps) {
    Vec3 c0 = cross(b - a, p - a);
    Vec3 c1 = cross(c - b, p - b);
    Vec3 c2 = cross(a - c, p - c);
    return dot(n, c0) >= -eps && dot(n, c1) >= -eps && dot(n, c2) >= -eps;
}

static bool rayTriIntersect(const Vec3& orig, const Vec3& dir,
                            const Vec3& a, const Vec3& b, const Vec3& c,
                            double eps) {
    Vec3 e1 = b - a;
    Vec3 e2 = c - a;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (std::fabs(det) < eps) return false;
    double invDet = 1.0 / det;
    Vec3 t = orig - a;
    double u = dot(t, p) * invDet;
    if (u < -eps || u > 1.0 + eps) return false;
    Vec3 q = cross(t, e1);
    double v = dot(dir, q) * invDet;
    if (v < -eps || u + v > 1.0 + eps) return false;
    double tt = dot(e2, q) * invDet;
    return tt > eps;
}

static bool triTriIntersectSegment(const TriData& A, const TriData& B,
                                   const vector<Vec3>& av, const vector<Vec3>& bv,
                                   Vec3& s0, Vec3& s1, double eps) {
    double da0 = dot(B.n, av[A.id[0]]) + B.d;
    double da1 = dot(B.n, av[A.id[1]]) + B.d;
    double da2 = dot(B.n, av[A.id[2]]) + B.d;
    double db0 = dot(A.n, bv[B.id[0]]) + A.d;
    double db1 = dot(A.n, bv[B.id[1]]) + A.d;
    double db2 = dot(A.n, bv[B.id[2]]) + A.d;

    bool aAllPos = (da0 > eps && da1 > eps && da2 > eps);
    bool aAllNeg = (da0 < -eps && da1 < -eps && da2 < -eps);
    bool bAllPos = (db0 > eps && db1 > eps && db2 > eps);
    bool bAllNeg = (db0 < -eps && db1 < -eps && db2 < -eps);
    if ((aAllPos || aAllNeg) || (bAllPos || bAllNeg)) return false;

    Vec3 dir = cross(A.n, B.n);
    if (norm2(dir) < eps * eps) {
        // Parallel or coplanar: skipped by this routine.
        return false;
    }
    dir = normalize(dir);

    vector<Vec3> pts;
    auto addPt = [&](const Vec3& p) {
        for (const Vec3& q : pts) {
            if (norm2(p - q) <= eps * eps * 16.0) return;
        }
        pts.push_back(p);
    };

    auto edgeTest = [&](const Vec3& p, const Vec3& q, double dp, double dq, const Vec3& oppA0, const Vec3& oppA1, const Vec3& oppA2, const Vec3& onN, double onD) {
        if (std::fabs(dp) <= eps && pointInTri3D(p, oppA0, oppA1, oppA2, onN, eps * 4.0)) {
            addPt(p);
        }
        if ((dp > eps && dq < -eps) || (dp < -eps && dq > eps)) {
            double t = dp / (dp - dq);
            Vec3 x = p + (q - p) * t;
            if (pointInTri3D(x, oppA0, oppA1, oppA2, onN, eps * 4.0)) addPt(x);
        } else if (std::fabs(dp) <= eps && std::fabs(dq) <= eps) {
            // Edge lies in opposite plane. Add endpoints if they lie in the opposite triangle.
            if (pointInTri3D(p, oppA0, oppA1, oppA2, onN, eps * 4.0)) addPt(p);
            if (pointInTri3D(q, oppA0, oppA1, oppA2, onN, eps * 4.0)) addPt(q);
        }
    };

    array<Vec3, 3> apts{av[A.id[0]], av[A.id[1]], av[A.id[2]]};
    array<Vec3, 3> bpts{bv[B.id[0]], bv[B.id[1]], bv[B.id[2]]};
    array<double, 3> da{da0, da1, da2};
    array<double, 3> db{db0, db1, db2};

    for (int i = 0; i < 3; ++i) {
        int j = (i + 1) % 3;
        edgeTest(apts[i], apts[j], da[i], da[j], bpts[0], bpts[1], bpts[2], B.n, B.d);
    }
    for (int i = 0; i < 3; ++i) {
        int j = (i + 1) % 3;
        edgeTest(bpts[i], bpts[j], db[i], db[j], apts[0], apts[1], apts[2], A.n, A.d);
    }

    if (pts.size() < 2) return false;

    double best = -1.0;
    Vec3 p0{}, p1{};
    for (size_t i = 0; i < pts.size(); ++i) {
        for (size_t j = i + 1; j < pts.size(); ++j) {
            double d = norm2(pts[i] - pts[j]);
            if (d > best) {
                best = d;
                p0 = pts[i];
                p1 = pts[j];
            }
        }
    }
    if (best <= eps * eps * 16.0) return false;
    s0 = p0;
    s1 = p1;
    return true;
}

static bool bboxRayHit(const AABB& box, const Vec3& orig, const Vec3& dir, double eps, double& tmin, double& tmax) {
    tmin = 0.0;
    tmax = INF_D;
    for (int k = 0; k < 3; ++k) {
        double o = (k == 0 ? orig.x : (k == 1 ? orig.y : orig.z));
        double d = (k == 0 ? dir.x : (k == 1 ? dir.y : dir.z));
        double mn = (k == 0 ? box.mn.x : (k == 1 ? box.mn.y : box.mn.z));
        double mx = (k == 0 ? box.mx.x : (k == 1 ? box.mx.y : box.mx.z));
        if (std::fabs(d) < eps) {
            if (o < mn - eps || o > mx + eps) return false;
            continue;
        }
        double inv = 1.0 / d;
        double t1 = (mn - o) * inv;
        double t2 = (mx - o) * inv;
        if (t1 > t2) std::swap(t1, t2);
        tmin = std::max(tmin, t1);
        tmax = std::min(tmax, t2);
        if (tmin > tmax) return false;
    }
    return true;
}

void Mesh::queryTriCandidates(int node, const AABB& box, vector<int>& out) const {
    if (node < 0 || bvh.empty()) return;
    const BVHNode& nd = bvh[node];
    if (!overlap(nd.box, box, eps)) return;
    if (nd.leaf) {
        for (int i = nd.start; i < nd.start + nd.count; ++i) out.push_back(triOrder[i]);
        return;
    }
    queryTriCandidates(nd.left, box, out);
    queryTriCandidates(nd.right, box, out);
}

void Mesh::queryRayCandidates(int node, const Vec3& orig, const Vec3& dir, vector<int>& out) const {
    if (node < 0 || bvh.empty()) return;
    const BVHNode& nd = bvh[node];
    double t0, t1;
    if (!bboxRayHit(nd.box, orig, dir, eps, t0, t1)) return;
    if (nd.leaf) {
        for (int i = nd.start; i < nd.start + nd.count; ++i) out.push_back(triOrder[i]);
        return;
    }
    queryRayCandidates(nd.left, orig, dir, out);
    queryRayCandidates(nd.right, orig, dir, out);
}

static int buildBVHRec(vector<BVHNode>& nodes, const vector<TriData>& tris, vector<int>& order, int l, int r) {
    int nodeId = (int)nodes.size();
    nodes.push_back({});
    AABB box;
    for (int i = l; i < r; ++i) box.expand(tris[order[i]].box);
    nodes[nodeId].box = box;
    int cnt = r - l;
    if (cnt <= 8) {
        nodes[nodeId].leaf = true;
        nodes[nodeId].start = l;
        nodes[nodeId].count = cnt;
        return nodeId;
    }
    AABB cbox;
    for (int i = l; i < r; ++i) {
        const TriData& t = tris[order[i]];
        Vec3 c{(t.box.mn.x + t.box.mx.x) * 0.5, (t.box.mn.y + t.box.mx.y) * 0.5, (t.box.mn.z + t.box.mx.z) * 0.5};
        cbox.expand(c);
    }
    Vec3 ext = cbox.mx - cbox.mn;
    int axis = 0;
    if (ext.y >= ext.x && ext.y >= ext.z) axis = 1;
    else if (ext.z >= ext.x && ext.z >= ext.y) axis = 2;
    int m = (l + r) / 2;
    std::nth_element(order.begin() + l, order.begin() + m, order.begin() + r, [&](int a, int b) {
        Vec3 ca{(tris[a].box.mn.x + tris[a].box.mx.x) * 0.5, (tris[a].box.mn.y + tris[a].box.mx.y) * 0.5, (tris[a].box.mn.z + tris[a].box.mx.z) * 0.5};
        Vec3 cb{(tris[b].box.mn.x + tris[b].box.mx.x) * 0.5, (tris[b].box.mn.y + tris[b].box.mx.y) * 0.5, (tris[b].box.mn.z + tris[b].box.mx.z) * 0.5};
        if (axis == 0) return ca.x < cb.x;
        if (axis == 1) return ca.y < cb.y;
        return ca.z < cb.z;
    });
    nodes[nodeId].left = buildBVHRec(nodes, tris, order, l, m);
    nodes[nodeId].right = buildBVHRec(nodes, tris, order, m, r);
    nodes[nodeId].leaf = false;
    return nodeId;
}

void Mesh::build() {
    tris.clear();
    tris.reserve(f.size());
    bbMin = {INF_D, INF_D, INF_D};
    bbMax = {-INF_D, -INF_D, -INF_D};
    for (const auto& p : v) {
        bbMin.x = std::min(bbMin.x, p.x); bbMin.y = std::min(bbMin.y, p.y); bbMin.z = std::min(bbMin.z, p.z);
        bbMax.x = std::max(bbMax.x, p.x); bbMax.y = std::max(bbMax.y, p.y); bbMax.z = std::max(bbMax.z, p.z);
    }
    scale = std::max({bbMax.x - bbMin.x, bbMax.y - bbMin.y, bbMax.z - bbMin.z, 1.0});
    eps = scale * 1e-9;
    rayDir = normalize(Vec3{1.0, 0.234567890123, 0.543210987654});

    double signedVol = 0.0;
    for (auto tri : f) {
        const Vec3& a = v[tri[0]];
        const Vec3& b = v[tri[1]];
        const Vec3& c = v[tri[2]];
        signedVol += dot(a, cross(b, c));
    }
    if (signedVol < 0.0) {
        for (auto& tri : f) std::swap(tri[1], tri[2]);
    }

    for (const auto& tri : f) {
        TriData t;
        t.id = tri;
        const Vec3& a = v[tri[0]];
        const Vec3& b = v[tri[1]];
        const Vec3& c = v[tri[2]];
        t.box = triBox(a, b, c);
        t.n = normalize(cross(b - a, c - a));
        t.d = -dot(t.n, a);
        tris.push_back(t);
    }

    triOrder.resize(tris.size());
    std::iota(triOrder.begin(), triOrder.end(), 0);
    bvh.clear();
    if (!tris.empty()) {
        buildBVHRec(bvh, tris, triOrder, 0, (int)triOrder.size());
    }
}

bool Mesh::pointInside(const Vec3& p) const {
    vector<int> cand;
    cand.reserve(64);
    queryRayCandidates(0, p, rayDir, cand);
    int hits = 0;
    for (int idx : cand) {
        const TriData& t = tris[idx];
        const Vec3& a = v[t.id[0]];
        const Vec3& b = v[t.id[1]];
        const Vec3& c = v[t.id[2]];
        if (rayTriIntersect(p, rayDir, a, b, c, eps * 8.0)) ++hits;
    }
    return (hits & 1) != 0;
}

struct LocalPoint {
    Vec2 p2;
    Vec3 p3;
};

struct LocalPointKey {
    long long x, y;
    bool operator==(const LocalPointKey& o) const { return x == o.x && y == o.y; }
};

struct LocalPointKeyHash {
    size_t operator()(const LocalPointKey& k) const noexcept {
        uint64_t h = (uint64_t)k.x * 1315423911u ^ (uint64_t)k.y * 2654435761u;
        h ^= h >> 33;
        h *= 0xff51afd7ed558ccdULL;
        h ^= h >> 33;
        h *= 0xc4ceb9fe1a85ec53ULL;
        h ^= h >> 33;
        return (size_t)h;
    }
};

static inline LocalPointKey quantKey2(const Vec2& p, double s) {
    return {qround(p.x, s), qround(p.y, s)};
}

static bool pointOnSeg2(const Vec2& p, const Vec2& a, const Vec2& b, double eps) {
    Vec2 ap = p - a, bp = p - b, ab = b - a;
    if (std::fabs(cross(ab, ap)) > eps * 8.0) return false;
    double d1 = dot(ap, ab);
    double d2 = dot(bp, ab);
    return d1 >= -eps * 8.0 && d2 <= eps * 8.0;
}

static double segParam(const Vec2& p, const Vec2& a, const Vec2& b) {
    Vec2 ab = b - a;
    double l2 = norm2(ab);
    if (l2 == 0.0) return 0.0;
    return dot(p - a, ab) / l2;
}

static double polygonArea2(const vector<int>& poly, const vector<LocalPoint>& pts) {
    double a = 0.0;
    int n = (int)poly.size();
    for (int i = 0; i < n; ++i) {
        const Vec2& p = pts[poly[i]].p2;
        const Vec2& q = pts[poly[(i + 1) % n]].p2;
        a += cross(p, q);
    }
    return 0.5 * a;
}

static bool pointInTriangle2(const Vec2& p, const Vec2& a, const Vec2& b, const Vec2& c, double eps) {
    double c1 = cross(b - a, p - a);
    double c2 = cross(c - b, p - b);
    double c3 = cross(a - c, p - c);
    return c1 >= -eps && c2 >= -eps && c3 >= -eps;
}

static bool earClipTriangulate(const vector<int>& polyIn, const vector<LocalPoint>& pts, vector<array<int, 3>>& outTris) {
    outTris.clear();
    if (polyIn.size() < 3) return false;
    vector<int> poly = polyIn;
    double area = polygonArea2(poly, pts);
    if (std::fabs(area) < 1e-18) return false;
    if (area < 0.0) std::reverse(poly.begin(), poly.end());

    vector<int> idx = poly;
    int guard = 0;
    while (idx.size() > 3 && guard < 10000) {
        ++guard;
        bool cut = false;
        int n = (int)idx.size();
        for (int i = 0; i < n; ++i) {
            int ip = idx[(i - 1 + n) % n];
            int ic = idx[i];
            int in = idx[(i + 1) % n];
            const Vec2& a = pts[ip].p2;
            const Vec2& b = pts[ic].p2;
            const Vec2& c = pts[in].p2;
            double cr = cross(b - a, c - b);
            if (cr <= 1e-18) continue;
            bool anyInside = false;
            for (int j = 0; j < n; ++j) {
                int id = idx[j];
                if (id == ip || id == ic || id == in) continue;
                if (pointInTriangle2(pts[id].p2, a, b, c, 1e-12)) {
                    anyInside = true;
                    break;
                }
            }
            if (anyInside) continue;
            outTris.push_back({ip, ic, in});
            idx.erase(idx.begin() + i);
            cut = true;
            break;
        }
        if (!cut) break;
    }
    if (idx.size() == 3) outTris.push_back({idx[0], idx[1], idx[2]});
    if (outTris.empty()) {
        // Fallback fan triangulation.
        for (size_t i = 1; i + 1 < poly.size(); ++i) {
            outTris.push_back({poly[0], poly[(int)i], poly[(int)i + 1]});
        }
    }
    return !outTris.empty();
}

struct UnionBuilder {
    const Mesh& A;
    const Mesh& B;
    vector<Vec3> outV;
    vector<array<int, 3>> outF;
    std::unordered_map<QuantKey3, int, QuantKey3Hash> vmap;
    std::unordered_set<FaceKey, FaceKeyHash> fmap;
    double gScale;
    double keyScale;

    UnionBuilder(const Mesh& a, const Mesh& b) : A(a), B(b) {
        Vec3 mn{
            std::min(A.bbMin.x, B.bbMin.x),
            std::min(A.bbMin.y, B.bbMin.y),
            std::min(A.bbMin.z, B.bbMin.z)
        };
        Vec3 mx{
            std::max(A.bbMax.x, B.bbMax.x),
            std::max(A.bbMax.y, B.bbMax.y),
            std::max(A.bbMax.z, B.bbMax.z)
        };
        gScale = std::max({mx.x - mn.x, mx.y - mn.y, mx.z - mn.z, 1.0});
        keyScale = gScale * 1e-8;
    }

    int addVertex(const Vec3& p) {
        QuantKey3 k = quantKey3(p, keyScale);
        auto it = vmap.find(k);
        if (it != vmap.end()) return it->second;
        int id = (int)outV.size();
        outV.push_back(p);
        vmap.emplace(k, id);
        return id;
    }

    void addTri(const Vec3& p0, const Vec3& p1, const Vec3& p2) {
        int a = addVertex(p0);
        int b = addVertex(p1);
        int c = addVertex(p2);
        if (a == b || b == c || a == c) return;
        FaceKey k{a, b, c};
        int x = k.a, y = k.b, z = k.c;
        if (x > y) std::swap(x, y);
        if (y > z) std::swap(y, z);
        if (x > y) std::swap(x, y);
        FaceKey ck{x, y, z};
        if (!fmap.insert(ck).second) return;
        outF.push_back({a, b, c});
    }

    Vec3 localToWorld(const Vec3& o, const Vec3& e1, const Vec3& e2, const Vec2& p) const {
        return o + e1 * p.x + e2 * p.y;
    }

    void processTriangle(const Mesh& me, const Mesh& other, int triIdx) {
        const TriData& td = me.tris[triIdx];
        const Vec3& p0 = me.v[td.id[0]];
        const Vec3& p1 = me.v[td.id[1]];
        const Vec3& p2 = me.v[td.id[2]];

        Vec3 e1 = normalize(p1 - p0);
        Vec3 n = normalize(cross(p1 - p0, p2 - p0));
        Vec3 e2 = cross(n, e1);

        vector<LocalPoint> pts;
        pts.reserve(64);
        std::unordered_map<LocalPointKey, int, LocalPointKeyHash> pmap;
        double localScale = me.scale * 1e-8;

        auto project = [&](const Vec3& p) -> Vec2 {
            Vec3 d = p - p0;
            return {dot(d, e1), dot(d, e2)};
        };

        auto addPoint = [&](const Vec3& p) -> int {
            Vec2 q = project(p);
            LocalPointKey k = quantKey2(q, localScale);
            auto it = pmap.find(k);
            if (it != pmap.end()) {
                int id = it->second;
                // Keep the first geometry sample.
                return id;
            }
            int id = (int)pts.size();
            pts.push_back({q, p});
            pmap.emplace(k, id);
            return id;
        };

        int v0 = addPoint(p0);
        int v1 = addPoint(p1);
        int v2 = addPoint(p2);

        struct Seg { int a, b; };
        vector<Seg> segs;
        vector<int> cand;
        cand.reserve(64);
        other.queryTriCandidates(0, td.box, cand);
        std::sort(cand.begin(), cand.end());
        cand.erase(std::unique(cand.begin(), cand.end()), cand.end());

        for (int j : cand) {
            const TriData& od = other.tris[j];
            Vec3 s0, s1;
            if (!triTriIntersectSegment(td, od, me.v, other.v, s0, s1, std::max(me.eps, other.eps) * 8.0)) continue;
            int a = addPoint(s0);
            int b = addPoint(s1);
            if (a == b) continue;
            segs.push_back({a, b});
        }

        // If no split segments exist, classify the whole triangle directly.
        if (segs.empty()) {
            Vec3 sample = (p0 + p1 + p2) / 3.0 + n * (me.scale * 1e-8);
            if (!other.pointInside(sample)) {
                addTri(p0, p1, p2);
            }
            return;
        }

        // Build boundary point lists per triangle edge.
        array<vector<std::pair<double, int>>, 3> edgePts;
        auto addEdgePoint = [&](int ei, int pid) {
            const Vec2& P = pts[pid].p2;
            const Vec2& A = pts[ei == 0 ? v0 : (ei == 1 ? v1 : v2)].p2;
            const Vec2& B = pts[ei == 0 ? v1 : (ei == 1 ? v2 : v0)].p2;
            if (!pointOnSeg2(P, A, B, localScale * 8.0)) return;
            double t = segParam(P, A, B);
            edgePts[ei].push_back({t, pid});
        };

        for (int i = 0; i < (int)pts.size(); ++i) {
            addEdgePoint(0, i);
            addEdgePoint(1, i);
            addEdgePoint(2, i);
        }

        vector<std::pair<int, int>> edges;
        auto addEdge = [&](int a, int b) {
            if (a == b) return;
            if (a > b) std::swap(a, b);
            edges.push_back({a, b});
        };

        auto addBoundaryEdges = [&](int ei) {
            auto& lst = edgePts[ei];
            if (lst.empty()) return;
            std::sort(lst.begin(), lst.end(), [](const auto& A, const auto& B) {
                if (std::fabs(A.first - B.first) > 1e-12) return A.first < B.first;
                return A.second < B.second;
            });
            vector<int> ids;
            ids.reserve(lst.size());
            for (auto& pr : lst) {
                if (ids.empty() || ids.back() != pr.second) ids.push_back(pr.second);
            }
            for (size_t i = 1; i < ids.size(); ++i) addEdge(ids[i - 1], ids[i]);
        };
        addBoundaryEdges(0);
        addBoundaryEdges(1);
        addBoundaryEdges(2);
        for (const auto& s : segs) addEdge(s.a, s.b);

        if (edges.empty()) {
            Vec3 sample = (p0 + p1 + p2) / 3.0 + n * (me.scale * 1e-8);
            if (!other.pointInside(sample)) addTri(p0, p1, p2);
            return;
        }

        struct HE {
            int from = -1, to = -1, twin = -1, next = -1;
            double ang = 0.0;
        };

        vector<HE> hes;
        hes.reserve(edges.size() * 2 + 8);
        vector<vector<int>> out((int)pts.size());
        auto addUndirected = [&](int a, int b) {
            int h = (int)hes.size();
            hes.push_back({a, b, h + 1, -1, 0.0});
            hes.push_back({b, a, h, -1, 0.0});
            out[a].push_back(h);
            out[b].push_back(h + 1);
        };
        std::sort(edges.begin(), edges.end());
        edges.erase(std::unique(edges.begin(), edges.end()), edges.end());
        for (auto [a, b] : edges) addUndirected(a, b);

        vector<vector<int>> ord((int)pts.size());
        for (int i = 0; i < (int)pts.size(); ++i) {
            ord[i] = out[i];
            const Vec2& O = pts[i].p2;
            std::sort(ord[i].begin(), ord[i].end(), [&](int ha, int hb) {
                Vec2 da{pts[hes[ha].to].p2.x - O.x, pts[hes[ha].to].p2.y - O.y};
                Vec2 db{pts[hes[hb].to].p2.x - O.x, pts[hes[hb].to].p2.y - O.y};
                double aa = std::atan2(da.y, da.x);
                double ab = std::atan2(db.y, db.x);
                if (std::fabs(aa - ab) > 1e-15) return aa < ab;
                return hes[ha].to < hes[hb].to;
            });
            for (size_t j = 0; j < ord[i].size(); ++j) {
                int h = ord[i][j];
                hes[h].ang = std::atan2(pts[hes[h].to].p2.y - O.y, pts[hes[h].to].p2.x - O.x);
            }
        }

        for (int h = 0; h < (int)hes.size(); ++h) {
            int v = hes[h].to;
            const auto& lst = ord[v];
            if (lst.empty()) continue;
            int twin = hes[h].twin;
            int pos = -1;
            for (int i = 0; i < (int)lst.size(); ++i) {
                if (lst[i] == twin) { pos = i; break; }
            }
            if (pos < 0) continue;
            int prev = (pos - 1 + (int)lst.size()) % (int)lst.size();
            hes[h].next = lst[prev];
        }

        vector<char> vis(hes.size(), 0);
        vector<vector<int>> faces;
        for (int h = 0; h < (int)hes.size(); ++h) {
            if (vis[h] || hes[h].next < 0) continue;
            vector<int> cyc;
            int cur = h;
            int limit = 0;
            while (!vis[cur] && limit++ < 10000) {
                vis[cur] = 1;
                cyc.push_back(hes[cur].from);
                cur = hes[cur].next;
                if (cur < 0) break;
                if (cur == h) {
                    cyc.push_back(hes[cur].from);
                    break;
                }
            }
            if (cyc.size() < 3) continue;
            if (cyc.front() == cyc.back()) cyc.pop_back();
            // Remove consecutive duplicates.
            vector<int> cleaned;
            for (int id : cyc) {
                if (cleaned.empty() || cleaned.back() != id) cleaned.push_back(id);
            }
            if (cleaned.size() >= 2 && cleaned.front() == cleaned.back()) cleaned.pop_back();
            if (cleaned.size() < 3) continue;
            double a = polygonArea2(cleaned, pts);
            if (a > 1e-18) faces.push_back(std::move(cleaned));
        }

        if (faces.empty()) {
            Vec3 sample = (p0 + p1 + p2) / 3.0 + n * (me.scale * 1e-8);
            if (!other.pointInside(sample)) addTri(p0, p1, p2);
            return;
        }

        for (const auto& face : faces) {
            vector<array<int, 3>> triList;
            if (!earClipTriangulate(face, pts, triList)) continue;
            if (triList.empty()) continue;
            const auto& t0 = triList[0];
            Vec3 sample = (pts[t0[0]].p3 + pts[t0[1]].p3 + pts[t0[2]].p3) / 3.0 + n * (me.scale * 1e-8);
            bool keep = !other.pointInside(sample);
            if (!keep) continue;
            for (const auto& tri : triList) {
                addTri(pts[tri[0]].p3, pts[tri[1]].p3, pts[tri[2]].p3);
            }
        }
    }

    void run() {
        for (int i = 0; i < (int)A.tris.size(); ++i) processTriangle(A, B, i);
        for (int i = 0; i < (int)B.tris.size(); ++i) processTriangle(B, A, i);
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA);
    A.f.resize(FA);
    B.v.resize(VB);
    B.f.resize(FB);

    for (int i = 0; i < VA; ++i) {
        std::cin >> A.v[i].x >> A.v[i].y >> A.v[i].z;
    }
    for (int i = 0; i < FA; ++i) {
        std::cin >> A.f[i][0] >> A.f[i][1] >> A.f[i][2];
    }
    for (int i = 0; i < VB; ++i) {
        std::cin >> B.v[i].x >> B.v[i].y >> B.v[i].z;
    }
    for (int i = 0; i < FB; ++i) {
        std::cin >> B.f[i][0] >> B.f[i][1] >> B.f[i][2];
    }

    A.build();
    B.build();

    UnionBuilder ub(A, B);
    ub.run();

    std::cout << ub.outV.size() << ' ' << ub.outF.size() << '\n';
    std::cout.setf(std::ios::fixed);
    std::cout.precision(10);
    for (const auto& p : ub.outV) {
        std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const auto& t : ub.outF) {
        std::cout << t[0] << ' ' << t[1] << ' ' << t[2] << '\n';
    }
    return 0;
}
