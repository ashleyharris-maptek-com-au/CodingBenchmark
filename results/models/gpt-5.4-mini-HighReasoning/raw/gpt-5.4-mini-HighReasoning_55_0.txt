// C++20 chess engine for FEN -> best move search.
// Output notation:
//   p/e2e4, pe4xd5, pe7e8/Q, pe5xd6ep, O-O, O-O-O, with + or # suffix.

#include <algorithm>
#include <array>
#include <atomic>
#include <chrono>
#include <cctype>
#include <cstdint>
#include <cstdlib>
#include <iostream>
#include <limits>

#include <vector>

using namespace std;

static constexpr int INF = 1'000'000'000;
static constexpr int MATE_SCORE = 30000;
static constexpr int MAX_PLY = 128;
static constexpr int TT_SIZE = 1 << 20;

static inline bool is_white_piece(char c) { return c >= 'A' && c <= 'Z'; }
static inline bool is_black_piece(char c) { return c >= 'a' && c <= 'z'; }
static inline bool is_piece(char c) { return c != '.'; }

static inline int file_of(int sq) { return sq & 7; }
static inline int rank_of(int sq) { return sq >> 3; }

static inline int mirror_sq(int sq) { return sq ^ 56; }

static inline string square_name(int sq) {
    string s;
    s.push_back(char('a' + file_of(sq)));
    s.push_back(char('1' + rank_of(sq)));
    return s;
}

static inline int piece_index(char c) {
    switch (c) {
        case 'P': return 0;
        case 'N': return 1;
        case 'B': return 2;
        case 'R': return 3;
        case 'Q': return 4;
        case 'K': return 5;
        case 'p': return 6;
        case 'n': return 7;
        case 'b': return 8;
        case 'r': return 9;
        case 'q': return 10;
        case 'k': return 11;
        default: return -1;
    }
}

static inline int piece_value(char c) {
    switch (c) {
        case 'P': case 'p': return 100;
        case 'N': case 'n': return 320;
        case 'B': case 'b': return 330;
        case 'R': case 'r': return 500;
        case 'Q': case 'q': return 900;
        case 'K': case 'k': return 0;
        default: return 0;
    }
}

static inline char promotion_piece_for_side(bool white, char promoUpper) {
    return white ? promoUpper : char(std::tolower((unsigned char)promoUpper));
}

static uint64_t splitmix64(uint64_t& x) {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
}

struct Move {
    uint8_t from = 0;
    uint8_t to = 0;
    uint8_t flags = 0;
    char promo = 0;
    int score = 0;
};

static inline bool same_move(const Move& a, const Move& b) {
    return a.from == b.from && a.to == b.to && a.flags == b.flags && a.promo == b.promo;
}

struct MoveList {
    array<Move, 256> m{};
    int n = 0;
    void clear() { n = 0; }
    void add(const Move& mv) { m[n++] = mv; }
};

struct State {
    Move mv;
    char moved = '.';
    char captured = '.';
    int old_castling = 0;
    int old_ep = -1;
    int old_halfmove = 0;
    int old_wking = 0;
    int old_bking = 0;
    uint64_t old_hash = 0;
};

struct TTEntry {
    uint64_t key = 0;
    int depth = -1;
    int value = 0;
    uint8_t flag = 0; // 0 exact, 1 lower, 2 upper
    Move best;
};

struct Position {
    array<char, 64> board{};
    bool white = true;
    int castling = 0;
    int ep = -1;
    int halfmove = 0;
    int wking = 4;
    int bking = 60;
    uint64_t hash = 0;
};

static uint64_t zob_piece[12][64];
static uint64_t zob_castling[16];
static uint64_t zob_ep[65];
static uint64_t zob_side;

static array<array<array<int, 64>, 6>, 2> pst{};
static vector<TTEntry> tt;
static int history_heur[2][64][64]{};
static Move killer[MAX_PLY][2]{};

static atomic<bool> stopped{false};
static chrono::steady_clock::time_point deadline;
static uint64_t nodes = 0;

static Position pos;

static inline bool time_up() {
    if ((nodes & 2047ULL) == 0) {
        if (chrono::steady_clock::now() >= deadline) {
            stopped.store(true, memory_order_relaxed);
            return true;
        }
    }
    return stopped.load(memory_order_relaxed);
}

static inline bool on_board(int sq) {
    return sq >= 0 && sq < 64;
}

static bool attacked(int sq, bool by_white) {
    int f = file_of(sq), r = rank_of(sq);

    if (by_white) {
        if (r > 0 && f > 0 && pos.board[sq - 9] == 'P') return true;
        if (r > 0 && f < 7 && pos.board[sq - 7] == 'P') return true;
    } else {
        if (r < 7 && f > 0 && pos.board[sq + 7] == 'p') return true;
        if (r < 7 && f < 7 && pos.board[sq + 9] == 'p') return true;
    }

    static constexpr int knight_delta[8] = {15, 17, 10, 6, -15, -17, -10, -6};
    for (int d : knight_delta) {
        int t = sq + d;
        if (!on_board(t)) continue;
        int df = abs(file_of(t) - f);
        int dr = abs(rank_of(t) - r);
        if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
        char p = pos.board[t];
        if (by_white ? p == 'N' : p == 'n') return true;
    }

    static constexpr int bishop_dir[4] = {9, 7, -9, -7};
    static constexpr int rook_dir[4] = {8, -8, 1, -1};

    for (int d : bishop_dir) {
        int t = sq + d;
        while (on_board(t) && abs(file_of(t) - file_of(t - d)) == 1) {
            char p = pos.board[t];
            if (p != '.') {
                if (by_white ? (p == 'B' || p == 'Q') : (p == 'b' || p == 'q')) return true;
                break;
            }
            t += d;
        }
    }

    for (int d : rook_dir) {
        int t = sq + d;
        while (on_board(t)) {
            if ((d == 1 || d == -1) && abs(file_of(t) - file_of(t - d)) != 1) break;
            char p = pos.board[t];
            if (p != '.') {
                if (by_white ? (p == 'R' || p == 'Q') : (p == 'r' || p == 'q')) return true;
                break;
            }
            t += d;
        }
    }

    static constexpr int king_delta[8] = {1, -1, 8, -8, 9, -9, 7, -7};
    for (int d : king_delta) {
        int t = sq + d;
        if (!on_board(t)) continue;
        if (abs(file_of(t) - f) > 1 || abs(rank_of(t) - r) > 1) continue;
        char p = pos.board[t];
        if (by_white ? p == 'K' : p == 'k') return true;
    }

    return false;
}

static inline bool in_check(bool white_side) {
    return attacked(white_side ? pos.wking : pos.bking, !white_side);
}

static void init_zobrist() {
    uint64_t seed = 0x123456789abcdef0ULL;
    for (int p = 0; p < 12; ++p) {
        for (int sq = 0; sq < 64; ++sq) {
            zob_piece[p][sq] = splitmix64(seed);
        }
    }
    for (int i = 0; i < 16; ++i) zob_castling[i] = splitmix64(seed);
    for (int i = 0; i < 65; ++i) zob_ep[i] = splitmix64(seed);
    zob_side = splitmix64(seed);
}

static int evaluate_white() {
    int score = 0;
    int bishops_w = 0, bishops_b = 0;
    for (int sq = 0; sq < 64; ++sq) {
        char p = pos.board[sq];
        if (p == '.') continue;
        int idx = piece_index(p);
        int type = idx % 6;
        int color = idx < 6 ? 0 : 1;
        int ps = pst[color][type][sq];
        int val = piece_value(p) + ps;
        if (is_white_piece(p)) {
            score += val;
            if (p == 'B') ++bishops_w;
        } else {
            score -= val;
            if (p == 'b') ++bishops_b;
        }
    }
    if (bishops_w >= 2) score += 30;
    if (bishops_b >= 2) score -= 30;
    return score;
}

static inline int eval_side_to_move() {
    int e = evaluate_white();
    return pos.white ? e : -e;
}

static State make_move(const Move& mv) {
    State st;
    st.mv = mv;
    st.moved = pos.board[mv.from];
    st.old_castling = pos.castling;
    st.old_ep = pos.ep;
    st.old_halfmove = pos.halfmove;
    st.old_wking = pos.wking;
    st.old_bking = pos.bking;
    st.old_hash = pos.hash;

    bool mover_white = pos.white;
    char moved = st.moved;
    int from = mv.from;
    int to = mv.to;
    int cap_sq = to;
    char captured = '.';

    if (pos.ep != -1) pos.hash ^= zob_ep[pos.ep];

    pos.hash ^= zob_piece[piece_index(moved)][from];
    pos.board[from] = '.';

    if (mv.flags & 2) {
        cap_sq = mover_white ? to - 8 : to + 8;
        captured = pos.board[cap_sq];
        if (captured != '.') {
            pos.hash ^= zob_piece[piece_index(captured)][cap_sq];
            pos.board[cap_sq] = '.';
        }
    } else {
        captured = pos.board[to];
        if (captured != '.') {
            pos.hash ^= zob_piece[piece_index(captured)][to];
        }
    }
    st.captured = captured;

    if (mv.flags & 4) {
        if (moved == 'K') {
            if (to == 6) {
                pos.hash ^= zob_piece[piece_index('R')][7];
                pos.board[7] = '.';
                pos.board[5] = 'R';
                pos.hash ^= zob_piece[piece_index('R')][5];
            } else {
                pos.hash ^= zob_piece[piece_index('R')][0];
                pos.board[0] = '.';
                pos.board[3] = 'R';
                pos.hash ^= zob_piece[piece_index('R')][3];
            }
            pos.wking = to;
        } else {
            if (to == 62) {
                pos.hash ^= zob_piece[piece_index('r')][63];
                pos.board[63] = '.';
                pos.board[61] = 'r';
                pos.hash ^= zob_piece[piece_index('r')][61];
            } else {
                pos.hash ^= zob_piece[piece_index('r')][56];
                pos.board[56] = '.';
                pos.board[59] = 'r';
                pos.hash ^= zob_piece[piece_index('r')][59];
            }
            pos.bking = to;
        }
    }

    char placed = moved;
    if (mv.flags & 8) {
        placed = promotion_piece_for_side(mover_white, mv.promo);
    }
    pos.board[to] = placed;
    pos.hash ^= zob_piece[piece_index(placed)][to];

    if (moved == 'K') pos.wking = to;
    if (moved == 'k') pos.bking = to;

    int new_castling = pos.castling;
    if (moved == 'K') new_castling &= ~3;
    if (moved == 'k') new_castling &= ~12;
    if (moved == 'R') {
        if (from == 0) new_castling &= ~2;
        if (from == 7) new_castling &= ~1;
    }
    if (moved == 'r') {
        if (from == 56) new_castling &= ~8;
        if (from == 63) new_castling &= ~4;
    }
    if (captured == 'R') {
        if (to == 0) new_castling &= ~2;
        if (to == 7) new_castling &= ~1;
    }
    if (captured == 'r') {
        if (to == 56) new_castling &= ~8;
        if (to == 63) new_castling &= ~4;
    }
    if (new_castling != pos.castling) {
        pos.hash ^= zob_castling[pos.castling];
        pos.castling = new_castling;
        pos.hash ^= zob_castling[pos.castling];
    }

    pos.ep = -1;
    if (moved == 'P' && to - from == 16) pos.ep = from + 8;
    if (moved == 'p' && from - to == 16) pos.ep = from - 8;
    if (pos.ep != -1) pos.hash ^= zob_ep[pos.ep];

    if (moved == 'P' || moved == 'p' || captured != '.') pos.halfmove = 0;
    else ++pos.halfmove;

    pos.white = !pos.white;
    pos.hash ^= zob_side;

    return st;
}

static void unmake_move(const State& st) {
    Move mv = st.mv;
    pos.white = !pos.white;
    pos.hash = st.old_hash;
    pos.castling = st.old_castling;
    pos.ep = st.old_ep;
    pos.halfmove = st.old_halfmove;
    pos.wking = st.old_wking;
    pos.bking = st.old_bking;

    bool mover_white = pos.white;
    char moved = st.moved;
    int from = mv.from;
    int to = mv.to;

    if (mv.flags & 4) {
        if (moved == 'K') {
            if (to == 6) {
                pos.board[7] = 'R';
                pos.board[5] = '.';
            } else {
                pos.board[0] = 'R';
                pos.board[3] = '.';
            }
        } else {
            if (to == 62) {
                pos.board[63] = 'r';
                pos.board[61] = '.';
            } else {
                pos.board[56] = 'r';
                pos.board[59] = '.';
            }
        }
    }

    pos.board[from] = moved;
    if (mv.flags & 8) {
        pos.board[to] = '.';
    } else {
        pos.board[to] = st.captured;
    }

    if (mv.flags & 2) {
        int cap_sq = mover_white ? to - 8 : to + 8;
        pos.board[cap_sq] = st.captured;
        pos.board[to] = '.';
    }
}

static inline bool is_capture_or_promo(const Move& mv) {
    return (mv.flags & 1) || (mv.flags & 8);
}

static inline bool is_enemy_king(char target, bool white_side) {
    return white_side ? target == 'k' : target == 'K';
}

static void gen_pseudo_moves(MoveList& out, bool captures_only) {
    out.clear();
    bool white = pos.white;
    for (int sq = 0; sq < 64; ++sq) {
        char p = pos.board[sq];
        if (p == '.') continue;
        if (white != is_white_piece(p)) continue;
        int f = file_of(sq), r = rank_of(sq);

        if (p == 'P' || p == 'p') {
            int dir = (p == 'P') ? 8 : -8;
            int start_rank = (p == 'P') ? 1 : 6;
            int promo_rank = (p == 'P') ? 6 : 1;
            int one = sq + dir;
            if (on_board(one) && pos.board[one] == '.') {
                if (r == promo_rank) {
                    for (char promo : {'Q', 'R', 'B', 'N'}) {
                        out.add(Move{(uint8_t)sq, (uint8_t)one, 8, promo, 0});
                    }
                } else if (!captures_only) {
                    out.add(Move{(uint8_t)sq, (uint8_t)one, 0, 0, 0});
                    if (r == start_rank) {
                        int two = sq + dir * 2;
                        if (on_board(two) && pos.board[two] == '.') {
                            out.add(Move{(uint8_t)sq, (uint8_t)two, 0, 0, 0});
                        }
                    }
                }
            }
            for (int df : {-1, 1}) {
                int nf = f + df;
                if (nf < 0 || nf > 7) continue;
                int to = sq + dir + df;
                if (!on_board(to)) continue;
                bool ep_ok = (pos.ep == to);
                char target = pos.board[to];
                bool is_cap = target != '.' || ep_ok;
                if (!is_cap) continue;
                if (target != '.' && (white == is_white_piece(target))) continue;
                if (target != '.' && is_enemy_king(target, white)) continue;
                if (r == promo_rank) {
                    for (char promo : {'Q', 'R', 'B', 'N'}) {
                        out.add(Move{(uint8_t)sq, (uint8_t)to, (uint8_t)(1 | 8 | (ep_ok ? 2 : 0)), promo, 0});
                    }
                } else {
                    out.add(Move{(uint8_t)sq, (uint8_t)to, (uint8_t)(1 | (ep_ok ? 2 : 0)), 0, 0});
                }
            }
            continue;
        }

        if (p == 'N' || p == 'n') {
            static constexpr int deltas[8] = {15, 17, 10, 6, -15, -17, -10, -6};
            for (int d : deltas) {
                int to = sq + d;
                if (!on_board(to)) continue;
                int df = abs(file_of(to) - f), dr = abs(rank_of(to) - r);
                if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
                char t = pos.board[to];
                if (t != '.' && (white == is_white_piece(t))) continue;
                if (t != '.' && is_enemy_king(t, white)) continue;
                bool cap = t != '.';
                if (captures_only && !cap) continue;
                out.add(Move{(uint8_t)sq, (uint8_t)to, (uint8_t)(cap ? 1 : 0), 0, 0});
            }
            continue;
        }

        if (p == 'B' || p == 'b' || p == 'R' || p == 'r' || p == 'Q' || p == 'q') {
            static constexpr int bishop_dir[4] = {9, 7, -9, -7};
            static constexpr int rook_dir[4] = {8, -8, 1, -1};
            auto slide = [&](const int* dirs, int cnt) {
                for (int i = 0; i < cnt; ++i) {
                    int d = dirs[i];
                        int to = sq + d;
                    while (on_board(to) && (abs(file_of(to) - file_of(to - d)) == 1 || d == 8 || d == -8)) {
                        char t = pos.board[to];
                        if (t != '.') {
                            if (white != is_white_piece(t) && !is_enemy_king(t, white)) {
                                out.add(Move{(uint8_t)sq, (uint8_t)to, 1, 0, 0});
                            }
                            break;
                        }
                        if (!captures_only) out.add(Move{(uint8_t)sq, (uint8_t)to, 0, 0, 0});
                        if (d == 1 || d == -1) {
                            if (file_of(to) == (d == 1 ? 7 : 0)) break;
                        }
                        to += d;
                    }
                }
            };
            if (p == 'B' || p == 'b') slide(bishop_dir, 4);
            else if (p == 'R' || p == 'r') slide(rook_dir, 4);
            else {
                slide(bishop_dir, 4);
                slide(rook_dir, 4);
            }
            continue;
        }

        if (p == 'K' || p == 'k') {
            static constexpr int deltas[8] = {1, -1, 8, -8, 9, -9, 7, -7};
            for (int d : deltas) {
                int to = sq + d;
                if (!on_board(to)) continue;
                if (abs(file_of(to) - f) > 1 || abs(rank_of(to) - r) > 1) continue;
                char t = pos.board[to];
                if (t != '.' && (white == is_white_piece(t))) continue;
                if (t != '.' && is_enemy_king(t, white)) continue;
                bool cap = t != '.';
                if (captures_only && !cap) continue;
                out.add(Move{(uint8_t)sq, (uint8_t)to, (uint8_t)(cap ? 1 : 0), 0, 0});
            }
            if (captures_only) continue;

            if (p == 'K') {
                if ((pos.castling & 1) && pos.board[5] == '.' && pos.board[6] == '.' &&
                    !attacked(4, false) && !attacked(5, false) && !attacked(6, false)) {
                    out.add(Move{4, 6, 4, 0, 0});
                }
                if ((pos.castling & 2) && pos.board[1] == '.' && pos.board[2] == '.' && pos.board[3] == '.' &&
                    !attacked(4, false) && !attacked(3, false) && !attacked(2, false)) {
                    out.add(Move{4, 2, 4, 0, 0});
                }
            } else {
                if ((pos.castling & 4) && pos.board[61] == '.' && pos.board[62] == '.' &&
                    !attacked(60, true) && !attacked(61, true) && !attacked(62, true)) {
                    out.add(Move{60, 62, 4, 0, 0});
                }
                if ((pos.castling & 8) && pos.board[57] == '.' && pos.board[58] == '.' && pos.board[59] == '.' &&
                    !attacked(60, true) && !attacked(59, true) && !attacked(58, true)) {
                    out.add(Move{60, 58, 4, 0, 0});
                }
            }
        }
    }
}

static bool legal_after_make(const State& st) {
    bool mover_white = !pos.white;
    return !attacked(mover_white ? pos.wking : pos.bking, pos.white);
}

static void gen_legal_moves(MoveList& out, bool captures_only) {
    MoveList pseudo;
    gen_pseudo_moves(pseudo, captures_only);
    out.clear();
    for (int i = 0; i < pseudo.n; ++i) {
        const Move& mv = pseudo.m[i];
        State st = make_move(mv);
        if (legal_after_make(st)) out.add(mv);
        unmake_move(st);
    }
}

static int score_move(const Move& mv, int ply, const Move& tt_move) {
    if (same_move(mv, tt_move)) return 100000000;
    int score = 0;
    char moved = pos.board[mv.from];
    char victim = (mv.flags & 2) ? (pos.white ? 'p' : 'P') : pos.board[mv.to];
    if (mv.flags & 1) {
        score += 10000 + piece_value(victim) * 10 - piece_value(moved);
    }
    if (mv.flags & 8) {
        score += 8000 + piece_value(mv.promo);
    }
    if (!(mv.flags & 1) && !(mv.flags & 8)) {
        if (same_move(mv, killer[ply][0])) score += 5000;
        else if (same_move(mv, killer[ply][1])) score += 4000;
        score += history_heur[pos.white ? 0 : 1][mv.from][mv.to];
    }
    return score;
}

static int pack_tt_score(int score, int ply) {
    if (score > MATE_SCORE - 1000) return score + ply;
    if (score < -MATE_SCORE + 1000) return score - ply;
    return score;
}

static int unpack_tt_score(int score, int ply) {
    if (score > MATE_SCORE - 1000) return score - ply;
    if (score < -MATE_SCORE + 1000) return score + ply;
    return score;
}

static int qsearch(int alpha, int beta, int ply);

static int search(int depth, int alpha, int beta, int ply) {
    ++nodes;
    if (time_up()) return 0;
    if (pos.halfmove >= 100) return 0;

    bool side = pos.white;
    bool in_chk = in_check(side);

    TTEntry& e = tt[pos.hash & (TT_SIZE - 1)];
    Move tt_move{};
    if (e.key == pos.hash) {
        tt_move = e.best;
        int tt_val = unpack_tt_score(e.value, ply);
        if (e.depth >= depth) {
            if (e.flag == 0) return tt_val;
            if (e.flag == 1 && tt_val >= beta) return tt_val;
            if (e.flag == 2 && tt_val <= alpha) return tt_val;
        }
    }

    if (depth <= 0) return qsearch(alpha, beta, ply);

    MoveList moves;
    gen_legal_moves(moves, false);
    if (moves.n == 0) {
        return in_chk ? (-MATE_SCORE + ply) : 0;
    }

    for (int i = 0; i < moves.n; ++i) moves.m[i].score = score_move(moves.m[i], ply, tt_move);
    sort(moves.m.begin(), moves.m.begin() + moves.n, [](const Move& a, const Move& b) {
        return a.score > b.score;
    });

    int alpha0 = alpha;
    int best = -INF;
    Move best_move{};
    bool have_best = false;

    for (int i = 0; i < moves.n; ++i) {
        const Move& mv = moves.m[i];
        State st = make_move(mv);
        int score = -search(depth - 1, -beta, -alpha, ply + 1);
        unmake_move(st);
        if (stopped.load(memory_order_relaxed)) return 0;
        if (score > best) {
            best = score;
            best_move = mv;
            have_best = true;
        }
        if (score > alpha) alpha = score;
        if (alpha >= beta) {
            if (!(mv.flags & 1) && !(mv.flags & 8)) {
                if (!same_move(killer[ply][0], mv)) {
                    killer[ply][1] = killer[ply][0];
                    killer[ply][0] = mv;
                }
                history_heur[side ? 0 : 1][mv.from][mv.to] += depth * depth;
            }
            TTEntry& store = tt[pos.hash & (TT_SIZE - 1)];
            store.key = pos.hash;
            store.depth = depth;
            store.value = pack_tt_score(best, ply);
            store.flag = 1;
            store.best = best_move;
            return best;
        }
    }

    TTEntry& store = tt[pos.hash & (TT_SIZE - 1)];
    store.key = pos.hash;
    store.depth = depth;
    store.value = pack_tt_score(best, ply);
    store.flag = (best <= alpha0) ? 2 : 0;
    store.best = have_best ? best_move : Move{};
    return best;
}

static int qsearch(int alpha, int beta, int ply) {
    ++nodes;
    if (time_up()) return 0;
    if (pos.halfmove >= 100) return 0;

    bool side = pos.white;
    bool in_chk = in_check(side);

    if (!in_chk) {
        int stand = eval_side_to_move();
        if (stand >= beta) return beta;
        if (stand > alpha) alpha = stand;
    }

    MoveList moves;
    gen_legal_moves(moves, !in_chk);
    if (moves.n == 0) {
        return in_chk ? (-MATE_SCORE + ply) : alpha;
    }

    for (int i = 0; i < moves.n; ++i) moves.m[i].score = score_move(moves.m[i], ply, Move{});
    sort(moves.m.begin(), moves.m.begin() + moves.n, [](const Move& a, const Move& b) {
        return a.score > b.score;
    });

    int best = alpha;
    for (int i = 0; i < moves.n; ++i) {
        const Move& mv = moves.m[i];
        if (!is_capture_or_promo(mv) && !in_chk) continue;
        State st = make_move(mv);
        int score = -qsearch(-beta, -alpha, ply + 1);
        unmake_move(st);
        if (stopped.load(memory_order_relaxed)) return 0;
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
        if (score > best) best = score;
    }
    return best;
}

static string move_to_string(const Move& mv) {
    if ((mv.flags & 4) && mv.from == 4 && (mv.to == 6 || mv.to == 2)) return (mv.to == 6) ? "O-O" : "O-O-O";
    if ((mv.flags & 4) && mv.from == 60 && (mv.to == 62 || mv.to == 58)) return (mv.to == 62) ? "O-O" : "O-O-O";

    char piece = pos.board[mv.from];
    string s;
    s.push_back(piece == 'P' || piece == 'p' ? 'p' : char(std::toupper((unsigned char)piece)));
    s += square_name(mv.from);
    if (mv.flags & 1) s.push_back('x');
    s += square_name(mv.to);
    if (mv.flags & 2) s += "ep";
    if (mv.flags & 8) {
        s.push_back('/');
        s.push_back(mv.promo);
    }
    return s;
}

static string annotate_check_suffix(const Move& mv) {
    State st = make_move(mv);
    bool opp_in_check = in_check(pos.white);
    MoveList replies;
    gen_legal_moves(replies, false);
    bool mate = opp_in_check && replies.n == 0;
    unmake_move(st);
    if (mate) return "#";
    if (opp_in_check) return "+";
    return "";
}

static bool parse_fen(const string& fen) {
    pos = Position{};
    pos.board.fill('.');
    pos.castling = 0;
    pos.ep = -1;
    pos.halfmove = 0;
    pos.wking = 4;
    pos.bking = 60;

    vector<string> parts;
    string cur;
    for (char c : fen) {
        if (c == ' ') {
            if (!cur.empty()) parts.push_back(cur);
            cur.clear();
        } else {
            cur.push_back(c);
        }
    }
    if (!cur.empty()) parts.push_back(cur);
    if (parts.size() < 4) return false;

    const string& board = parts[0];
    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') {
            --r;
            f = 0;
            continue;
        }
        if (isdigit((unsigned char)c)) {
            f += c - '0';
            continue;
        }
        int sq = r * 8 + f;
        pos.board[sq] = c;
        if (c == 'K') pos.wking = sq;
        if (c == 'k') pos.bking = sq;
        ++f;
    }

    pos.white = (parts[1] == "w");
    const string& cast = parts[2];
    if (cast.find('K') != string::npos) pos.castling |= 1;
    if (cast.find('Q') != string::npos) pos.castling |= 2;
    if (cast.find('k') != string::npos) pos.castling |= 4;
    if (cast.find('q') != string::npos) pos.castling |= 8;

    if (parts[3] != "-") {
        int file = parts[3][0] - 'a';
        int rank = parts[3][1] - '1';
        if (file >= 0 && file < 8 && rank >= 0 && rank < 8) pos.ep = rank * 8 + file;
    }
    if (parts.size() >= 5) pos.halfmove = stoi(parts[4]);

    pos.hash = 0;
    for (int sq = 0; sq < 64; ++sq) {
        char p = pos.board[sq];
        if (p != '.') pos.hash ^= zob_piece[piece_index(p)][sq];
    }
    pos.hash ^= zob_castling[pos.castling];
    if (pos.ep != -1) pos.hash ^= zob_ep[pos.ep];
    if (!pos.white) pos.hash ^= zob_side;
    return true;
}

static bool root_search(int max_ms, Move& best_move, string& best_notation) {
    int budget_ms = max_ms > 100 ? max_ms - 50 : max_ms;
    deadline = chrono::steady_clock::now() + chrono::milliseconds(budget_ms);
    stopped.store(false, memory_order_relaxed);
    nodes = 0;
    for (auto& a : history_heur) for (auto& b : a) for (int& x : b) x = 0;
    for (int d = 0; d < MAX_PLY; ++d) killer[d][0] = killer[d][1] = Move{};

    MoveList moves;
    gen_legal_moves(moves, false);
    if (moves.n == 0) {
        best_move = Move{};
        best_notation = "0000";
        return true;
    }

    Move current_best = moves.m[0];
    int current_score = -INF;

    for (int depth = 1; depth < MAX_PLY; ++depth) {
        if (time_up()) break;
        int alpha = -INF, beta = INF;
        Move iter_best = current_best;
        int iter_best_score = -INF;
        bool complete = true;

        for (int i = 0; i < moves.n; ++i) {
            moves.m[i].score = score_move(moves.m[i], 0, Move{});
        }
        sort(moves.m.begin(), moves.m.begin() + moves.n, [](const Move& a, const Move& b) {
            return a.score > b.score;
        });

        for (int i = 0; i < moves.n; ++i) {
            const Move& mv = moves.m[i];
            State st = make_move(mv);
            int score = -search(depth - 1, -beta, -alpha, 1);
            unmake_move(st);
            if (stopped.load(memory_order_relaxed)) {
                complete = false;
                break;
            }
            if (score > iter_best_score) {
                iter_best_score = score;
                iter_best = mv;
            }
            if (score > alpha) alpha = score;
        }
        if (!complete) break;
        current_best = iter_best;
        current_score = iter_best_score;
    }

    best_move = current_best;
    string s = move_to_string(best_move);
    s += annotate_check_suffix(best_move);
    best_notation = s;
    (void)current_score;
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    init_zobrist();
    init_pst();
    tt.assign(TT_SIZE, TTEntry{});

    double timeout_sec = 5.0;
    string fen;
    if (!(cin >> timeout_sec)) return 0;
    cin >> ws;
    getline(cin, fen);
    if (fen.empty()) return 0;

    Move best_move{};
    string answer = "0000";
    if (!parse_fen(fen)) {
        cout << "0000\n";
        return 0;
    }

    int ms = (int)(timeout_sec * 1000.0);
    if (ms < 1) ms = 1;
    root_search(ms, best_move, answer);
    cout << answer << '\n';
    return 0;
}
