using System;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const double Eps = 1e-9;

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }
    }

    private struct Rotation
    {
        public int M00, M01, M02;
        public int M10, M11, M12;
        public int M20, M21, M22;
        public double Qw, Qx, Qy, Qz;

        public Rotation(int m00, int m01, int m02,
                        int m10, int m11, int m12,
                        int m20, int m21, int m22)
        {
            M00 = m00; M01 = m01; M02 = m02;
            M10 = m10; M11 = m11; M12 = m12;
            M20 = m20; M21 = m21; M22 = m22;
            var q = MatrixToQuaternion(m00, m01, m02, m10, m11, m12, m20, m21, m22);
            Qw = q.qw;
            Qx = q.qx;
            Qy = q.qy;
            Qz = q.qz;
        }
    }

    private struct Candidate
    {
        public Rotation Rot;
        public double Dx, Dy, Dz;
        public double OffsetX, OffsetY, OffsetZ;
        public long Nx, Ny, Nz;
        public long Count;
        public double Score;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int vCount = fs.NextInt();
        int fCount = fs.NextInt();

        var vertices = new Vec3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            vertices[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        }

        for (int i = 0; i < fCount; i++)
        {
            fs.NextInt();
            fs.NextInt();
            fs.NextInt();
        }

        double width = fs.NextDouble();
        double height = fs.NextDouble();
        double depth = fs.NextDouble();

        if (vCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var rotations = BuildRotations();
        Candidate best = default;
        best.Count = -1;
        best.Score = double.NegativeInfinity;

        foreach (var rot in rotations)
        {
            EvaluateCandidate(vertices, width, height, depth, rot, out Candidate cand);
            if (cand.Count > best.Count || (cand.Count == best.Count && cand.Score > best.Score))
            {
                best = cand;
            }
        }

        if (best.Count <= 0 || best.Nx <= 0 || best.Ny <= 0 || best.Nz <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var stdout = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        stdout.NewLine = "\n";

        stdout.WriteLine(best.Count);

        // Stream the placements directly to stdout to avoid large memory spikes.
        for (long iz = 0; iz < best.Nz; iz++)
        {
            double baseZ = iz * best.Dz;
            for (long iy = 0; iy < best.Ny; iy++)
            {
                double baseY = iy * best.Dy;
                for (long ix = 0; ix < best.Nx; ix++)
                {
                    double tx = ix * best.Dx + best.OffsetX;
                    double ty = baseY + best.OffsetY;
                    double tz = baseZ + best.OffsetZ;
                    WritePlacement(stdout, tx, ty, tz, best.Rot.Qw, best.Rot.Qx, best.Rot.Qy, best.Rot.Qz);
                }
            }
        }

        stdout.Flush();
    }

    private static void EvaluateCandidate(Vec3[] vertices, double width, double height, double depth, Rotation rot, out Candidate cand)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        for (int i = 0; i < vertices.Length; i++)
        {
            Vec3 v = vertices[i];
            double x = rot.M00 * v.X + rot.M01 * v.Y + rot.M02 * v.Z;
            double y = rot.M10 * v.X + rot.M11 * v.Y + rot.M12 * v.Z;
            double z = rot.M20 * v.X + rot.M21 * v.Y + rot.M22 * v.Z;

            if (x < minX) minX = x;
            if (y < minY) minY = y;
            if (z < minZ) minZ = z;
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
            if (z > maxZ) maxZ = z;
        }

        double dx = Math.Max(maxX - minX, Eps);
        double dy = Math.Max(maxY - minY, Eps);
        double dz = Math.Max(maxZ - minZ, Eps);

        long nx = SafeFloorDiv(width, dx);
        long ny = SafeFloorDiv(height, dy);
        long nz = SafeFloorDiv(depth, dz);

        long count = SafeMul(SafeMul(nx, ny), nz);
        double score = count;
        score -= 1e-12 * (dx + dy + dz);
        score -= 1e-15 * (dx * dy * dz);

        cand = new Candidate
        {
            Rot = rot,
            Dx = dx,
            Dy = dy,
            Dz = dz,
            OffsetX = -minX,
            OffsetY = -minY,
            OffsetZ = -minZ,
            Nx = nx,
            Ny = ny,
            Nz = nz,
            Count = count,
            Score = score
        };
    }

    private static long SafeFloorDiv(double a, double b)
    {
        if (!(a > 0) || !(b > 0))
        {
            return 0;
        }

        double q = (a + 1e-10) / b;
        if (q <= 0) return 0;
        if (q >= long.MaxValue) return long.MaxValue;
        return (long)Math.Floor(q);
    }

    private static long SafeMul(long a, long b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > 0 && b > 0 && a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    private static void WritePlacement(StreamWriter stdout, double tx, double ty, double tz, double qw, double qx, double qy, double qz)
    {
        stdout.Write(tx.ToString("G17", CultureInfo.InvariantCulture));
        stdout.Write(' ');
        stdout.Write(ty.ToString("G17", CultureInfo.InvariantCulture));
        stdout.Write(' ');
        stdout.Write(tz.ToString("G17", CultureInfo.InvariantCulture));
        stdout.Write(' ');
        stdout.Write(qw.ToString("G17", CultureInfo.InvariantCulture));
        stdout.Write(' ');
        stdout.Write(qx.ToString("G17", CultureInfo.InvariantCulture));
        stdout.Write(' ');
        stdout.Write(qy.ToString("G17", CultureInfo.InvariantCulture));
        stdout.Write(' ');
        stdout.WriteLine(qz.ToString("G17", CultureInfo.InvariantCulture));
    }

    private static Rotation[] BuildRotations()
    {
        int[][] perms =
        {
            new[] { 0, 1, 2 },
            new[] { 0, 2, 1 },
            new[] { 1, 0, 2 },
            new[] { 1, 2, 0 },
            new[] { 2, 0, 1 },
            new[] { 2, 1, 0 }
        };

        int[] signs = { -1, 1 };
        var rotations = new Rotation[24];
        int idx = 0;

        foreach (var perm in perms)
        {
            for (int sx = 0; sx < 2; sx++)
            {
                for (int sy = 0; sy < 2; sy++)
                {
                    for (int sz = 0; sz < 2; sz++)
                    {
                        int[] col0 = BasisVector(perm[0], signs[sx]);
                        int[] col1 = BasisVector(perm[1], signs[sy]);
                        int[] col2 = BasisVector(perm[2], signs[sz]);

                        int m00 = col0[0], m01 = col1[0], m02 = col2[0];
                        int m10 = col0[1], m11 = col1[1], m12 = col2[1];
                        int m20 = col0[2], m21 = col1[2], m22 = col2[2];

                        int det = Determinant(m00, m01, m02, m10, m11, m12, m20, m21, m22);
                        if (det == 1)
                        {
                            rotations[idx++] = new Rotation(m00, m01, m02, m10, m11, m12, m20, m21, m22);
                        }
                    }
                }
            }
        }

        if (idx != 24)
        {
            Array.Resize(ref rotations, idx);
        }
        return rotations;
    }

    private static int[] BasisVector(int axis, int sign)
    {
        switch (axis)
        {
            case 0: return new[] { sign, 0, 0 };
            case 1: return new[] { 0, sign, 0 };
            default: return new[] { 0, 0, sign };
        }
    }

    private static int Determinant(int m00, int m01, int m02, int m10, int m11, int m12, int m20, int m21, int m22)
    {
        return m00 * (m11 * m22 - m12 * m21)
             - m01 * (m10 * m22 - m12 * m20)
             + m02 * (m10 * m21 - m11 * m20);
    }

    private static (double qw, double qx, double qy, double qz) MatrixToQuaternion(
        int m00, int m01, int m02,
        int m10, int m11, int m12,
        int m20, int m21, int m22)
    {
        double qw, qx, qy, qz;
        double trace = m00 + m11 + m22;
        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            qw = 0.25 * s;
            qx = (m21 - m12) / s;
            qy = (m02 - m20) / s;
            qz = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0;
            qw = (m21 - m12) / s;
            qx = 0.25 * s;
            qy = (m01 + m10) / s;
            qz = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0;
            qw = (m02 - m20) / s;
            qx = (m01 + m10) / s;
            qy = 0.25 * s;
            qz = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0;
            qw = (m10 - m01) / s;
            qx = (m02 + m20) / s;
            qy = (m12 + m21) / s;
            qz = 0.25 * s;
        }

        double norm = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        if (norm > 0)
        {
            qw /= norm;
            qx /= norm;
            qy /= norm;
            qz /= norm;
        }
        return (qw, qx, qy, qz);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string Next()
        {
            var sb = new StringBuilder();
            int c;
            do
            {
                c = Read();
            } while (c != -1 && c <= 32);

            while (c != -1 && c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            return int.Parse(Next(), CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(Next(), CultureInfo.InvariantCulture);
        }
    }
}
