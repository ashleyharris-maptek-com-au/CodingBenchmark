use std::fmt::Write as _;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };

    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        min_x = min_x.min(x);
        min_y = min_y.min(y);
        max_x = max_x.max(x);
        max_y = max_y.max(y);
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    let width = max_x - min_x;
    let height = max_y - min_y;
    let span = width.max(height);

    let mut out = String::new();
    let scale = 10.0_f64;
    let qw = 1.0_f64;
    let qx = 0.0_f64;
    let qy = 0.0_f64;
    let qz = 0.0_f64;

    if sz.abs() < 1e-9 || (nv <= 8 && span <= 18.0) {
        let cx = (min_x + max_x) * 0.5;
        let cy = (min_y + max_y) * 0.5;
        let cz = 0.0_f64;
        writeln!(&mut out, "1").unwrap();
        writeln!(
            &mut out,
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            cx, cy, cz, qw, qx, qy, qz, scale
        )
        .unwrap();
        print!("{out}");
        return;
    }

    let inv_sz = 1.0 / sz;
    let alpha = sx * inv_sz;
    let beta = sy * inv_sz;

    let step = if span <= 60.0 { 8.0_f64 } else { 10.0_f64 };
    let margin = 20.0_f64;
    let height_step = 50.0_f64;

    let cols = ((width + 2.0 * margin) / step).ceil().max(1.0) as usize;
    let rows = ((height + 2.0 * margin) / step).ceil().max(1.0) as usize;
    let count = cols * rows;

    writeln!(&mut out, "{count}").unwrap();

    let start_x = min_x - margin;
    let start_y = min_y - margin;
    let mut idx = 0usize;
    for r in 0..rows {
        let gy = start_y + (r as f64 + 0.5) * step;
        for c in 0..cols {
            let gx = start_x + (c as f64 + 0.5) * step;
            let pz = idx as f64 * height_step;
            let px = gx + alpha * pz;
            let py = gy + beta * pz;
            writeln!(
                &mut out,
                "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
                px, py, pz, qw, qx, qy, qz, scale
            )
            .unwrap();
            idx += 1;
        }
    }

    print!("{out}");
}
