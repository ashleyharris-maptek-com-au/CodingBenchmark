use std::f64::consts::TAU;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl std::ops::Add for Vec2 {
    type Output = Self;
    fn add(self, rhs: Self) -> Self::Output {
        Self {
            x: self.x + rhs.x,
            y: self.y + rhs.y,
        }
    }
}

impl std::ops::Sub for Vec2 {
    type Output = Self;
    fn sub(self, rhs: Self) -> Self::Output {
        Self {
            x: self.x - rhs.x,
            y: self.y - rhs.y,
        }
    }
}

impl std::ops::Mul<f64> for Vec2 {
    type Output = Self;
    fn mul(self, rhs: f64) -> Self::Output {
        Self {
            x: self.x * rhs,
            y: self.y * rhs,
        }
    }
}

impl std::ops::Add for Vec3 {
    type Output = Self;
    fn add(self, rhs: Self) -> Self::Output {
        Self {
            x: self.x + rhs.x,
            y: self.y + rhs.y,
            z: self.z + rhs.z,
        }
    }
}

impl std::ops::Sub for Vec3 {
    type Output = Self;
    fn sub(self, rhs: Self) -> Self::Output {
        Self {
            x: self.x - rhs.x,
            y: self.y - rhs.y,
            z: self.z - rhs.z,
        }
    }
}

impl std::ops::Mul<f64> for Vec3 {
    type Output = Self;
    fn mul(self, rhs: f64) -> Self::Output {
        Self {
            x: self.x * rhs,
            y: self.y * rhs,
            z: self.z * rhs,
        }
    }
}

fn dot3(a: Vec3, b: Vec3) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

fn cross3(a: Vec3, b: Vec3) -> Vec3 {
    Vec3 {
        x: a.y * b.z - a.z * b.y,
        y: a.z * b.x - a.x * b.z,
        z: a.x * b.y - a.y * b.x,
    }
}

fn norm3(v: Vec3) -> f64 {
    dot3(v, v).sqrt()
}

fn normalize3(v: Vec3) -> Vec3 {
    let n = norm3(v);
    if n > 0.0 {
        v * (1.0 / n)
    } else {
        Vec3 {
            x: 0.0,
            y: 0.0,
            z: 1.0,
        }
    }
}

fn project_to_ground(v: Vec3, d: Vec3) -> Vec2 {
    let t = v.z / d.z;
    Vec2 {
        x: v.x - t * d.x,
        y: v.y - t * d.y,
    }
}

fn solve_basis(p: Vec2, a: Vec2, b: Vec2) -> Vec2 {
    let det = a.x * b.y - a.y * b.x;
    Vec2 {
        x: (p.x * b.y - p.y * b.x) / det,
        y: (a.x * p.y - a.y * p.x) / det,
    }
}

fn quat_from_axes(x: Vec3, y: Vec3, z: Vec3) -> (f64, f64, f64, f64) {
    let m00 = x.x;
    let m01 = y.x;
    let m02 = z.x;
    let m10 = x.y;
    let m11 = y.y;
    let m12 = z.y;
    let m20 = x.z;
    let m21 = y.z;
    let m22 = z.z;

    let trace = m00 + m11 + m22;
    let (qw, qx, qy, qz) = if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        (
            0.25 * s,
            (m21 - m12) / s,
            (m02 - m20) / s,
            (m10 - m01) / s,
        )
    } else if m00 > m11 && m00 > m22 {
        let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
        (
            (m21 - m12) / s,
            0.25 * s,
            (m01 + m10) / s,
            (m02 + m20) / s,
        )
    } else if m11 > m22 {
        let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
        (
            (m02 - m20) / s,
            (m01 + m10) / s,
            0.25 * s,
            (m12 + m21) / s,
        )
    } else {
        let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
        (
            (m10 - m01) / s,
            (m02 + m20) / s,
            (m12 + m21) / s,
            0.25 * s,
        )
    };
    (qw, qx, qy, qz)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };

    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Vec2 { x, y });
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    let mut d = normalize3(Vec3 {
        x: sx,
        y: sy,
        z: sz,
    });
    if d.z.abs() < 1e-12 {
        d.z = if d.z >= 0.0 { 1e-12 } else { -1e-12 };
        d = normalize3(d);
    }

    let world_up = Vec3 {
        x: 0.0,
        y: 0.0,
        z: 1.0,
    };

    let base_x = {
        let c = cross3(world_up, d);
        if norm3(c) < 1e-12 {
            Vec3 {
                x: 1.0,
                y: 0.0,
                z: 0.0,
            }
        } else {
            normalize3(c)
        }
    };
    let base_y = normalize3(cross3(d, base_x));

    #[derive(Clone, Copy)]
    struct Candidate {
        x: Vec3,
        y: Vec3,
        p: Vec2,
        q: Vec2,
        min_a: f64,
        min_b: f64,
        nx: usize,
        ny: usize,
        score: u128,
    }

    let mut best: Option<Candidate> = None;
    let scale = 10.0f64;

    for k in 0..16 {
        let phi = TAU * (k as f64) / 16.0;
        let c = phi.cos();
        let s = phi.sin();
        let x = normalize3(base_x * c + base_y * s);
        let y = cross3(d, x);

        let xp = project_to_ground(x, d);
        let yp = project_to_ground(y, d);
        let p = xp + yp;
        let q = xp - yp;

        let mut min_a = f64::INFINITY;
        let mut max_a = f64::NEG_INFINITY;
        let mut min_b = f64::INFINITY;
        let mut max_b = f64::NEG_INFINITY;
        for &v in &poly {
            let ab = solve_basis(v, p, q);
            min_a = min_a.min(ab.x);
            max_a = max_a.max(ab.x);
            min_b = min_b.min(ab.y);
            max_b = max_b.max(ab.y);
        }

        let nx = (((max_a - min_a) / scale).ceil().max(1.0)) as usize;
        let ny = (((max_b - min_b) / scale).ceil().max(1.0)) as usize;
        let score = (nx as u128) * (ny as u128);

        let cand = Candidate {
            x,
            y,
            p,
            q,
            min_a,
            min_b,
            nx,
            ny,
            score,
        };

        match best {
            None => best = Some(cand),
            Some(prev) => {
                if cand.score < prev.score
                    || (cand.score == prev.score && (cand.nx + cand.ny) < (prev.nx + prev.ny))
                {
                    best = Some(cand);
                }
            }
        }
    }

    let best = best.unwrap();
    let (qw, qx, qy, qz) = quat_from_axes(best.x, best.y, d);

    let mut out = io::BufWriter::new(io::stdout().lock());
    let total = best.nx * best.ny;
    writeln!(out, "{}", total).unwrap();

    let sep = 40.0f64;
    let mut idx = 0usize;
    for i in 0..best.nx {
        let a_center = best.min_a + (i as f64 + 0.5) * scale;
        for j in 0..best.ny {
            let b_center = best.min_b + (j as f64 + 0.5) * scale;
            let center2 = best.p * a_center + best.q * b_center;
            let center3 = Vec3 {
                x: center2.x,
                y: center2.y,
                z: 0.0,
            } + d * (sep * (idx as f64 + 1.0));
            idx += 1;

            writeln!(
                out,
                "{:.12} {:.12} {:.12} {:.12} {:.12} {:.12} {:.12} {:.12}",
                center3.x, center3.y, center3.z, qw, qx, qy, qz, scale
            )
            .unwrap();
        }
    }
}
