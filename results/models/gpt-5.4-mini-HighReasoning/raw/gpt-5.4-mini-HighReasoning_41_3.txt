               OpCapability Shader

               OpDecorate %color Location 4
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
        %f32 = OpTypeFloat 32
      %v3f32 = OpTypeVector %f32 3
      %v4f32 = OpTypeVector %f32 4
%ptr_input_v3f32 = OpTypePointer Input %v3f32
%ptr_output_v4f32 = OpTypePointer Output %v4f32
       %color = OpVariable %ptr_input_v3f32 Input
    %outColor = OpVariable %ptr_output_v4f32 Output
   %float_1_0 = OpConstant %f32 1
       %main = OpFunction %void None %3
          %5 = OpLabel
    %color_v3 = OpLoad %v3f32 %color
          %r = OpCompositeExtract %f32 %color_v3 0
          %g = OpCompositeExtract %f32 %color_v3 1
          %b = OpCompositeExtract %f32 %color_v3 2
       %rgba = OpCompositeConstruct %v4f32 %r %g %b %float_1_0
               OpStore %outColor %rgba
               OpReturn
