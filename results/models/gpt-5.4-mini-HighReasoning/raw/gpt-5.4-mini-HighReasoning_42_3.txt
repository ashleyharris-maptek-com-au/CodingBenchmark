OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %ubo "ubo"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %arr_pv_in ArrayStride 16

OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block

OpDecorate %arr_v3_in0 ArrayStride 16
OpDecorate %arr_v3_in1 ArrayStride 16
OpDecorate %arr_v3_in2 ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 ColMajor
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 0 MatrixStride 16
OpMemberDecorate %ubo_struct 1 ColMajor
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 1 MatrixStride 16
OpMemberDecorate %ubo_struct 2 ColMajor
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 2 MatrixStride 16
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo Set 0
OpDecorate %ubo Binding 0

%glsl = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thickness = OpConstant %float 0.03
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in0 = OpTypeArray %v3float %uint_2
%ptr_in_v3arr0 = OpTypePointer Input %arr_v3_in0
%worldPos = OpVariable %ptr_in_v3arr0 Input

%arr_v3_in1 = OpTypeArray %v3float %uint_2
%ptr_in_v3arr1 = OpTypePointer Input %arr_v3_in1
%normal = OpVariable %ptr_in_v3arr1 Input

%arr_v3_in2 = OpTypeArray %v3float %uint_2
%ptr_in_v3arr2 = OpTypePointer Input %arr_v3_in2
%color = OpVariable %ptr_in_v3arr2 Input

%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %functype
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y
%negdy = OpFNegate %float %dy
%dir2 = OpCompositeConstruct %v2float %negdy %dx
%dir2n = OpExtInst %v2float %glsl Normalize %dir2
%perp = OpVectorTimesScalar %v2float %dir2n %float_thickness
%perpx = OpCompositeExtract %float %perp 0
%perpy = OpCompositeExtract %float %perp 1

%pos0x = OpFAdd %float %p0x %perpx
%pos0y = OpFAdd %float %p0y %perpy
%pos1x = OpFSub %float %p0x %perpx
%pos1y = OpFSub %float %p0y %perpy
%pos2x = OpFAdd %float %p1x %perpx
%pos2y = OpFAdd %float %p1y %perpy
%pos3x = OpFSub %float %p1x %perpx
%pos3y = OpFSub %float %p1y %perpy

%pos0 = OpCompositeConstruct %v4float %pos0x %pos0y %p0z %p0w
%pos1 = OpCompositeConstruct %v4float %pos1x %pos1y %p0z %p0w
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3
%pos2 = OpCompositeConstruct %v4float %pos2x %pos2y %p1z %p1w
%pos3 = OpCompositeConstruct %v4float %pos3x %pos3y %p1z %p1w

%outpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %outpos_ptr %pos0
OpStore %outColor %c0
OpEmitVertex

OpStore %outpos_ptr %pos1
OpStore %outColor %c0
OpEmitVertex

OpStore %outpos_ptr %pos2
OpStore %outColor %c1
OpEmitVertex

OpStore %outpos_ptr %pos3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
