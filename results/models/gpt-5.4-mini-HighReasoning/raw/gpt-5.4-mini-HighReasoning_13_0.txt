// Longest Common Substring solver
// Adaptive:
// 1) Generalized suffix automaton on the shortest string for smaller inputs.
// 2) Rolling-hash intersection with binary search for larger inputs.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <utility>
#include <vector>

using namespace std;

static inline void strip_cr(string& s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
}

struct HashPair {
    uint64_t a = 0;
    uint64_t b = 0;

    bool operator==(const HashPair& other) const noexcept {
        return a == other.a && b == other.b;
    }
};

static inline uint64_t mix64(uint64_t x) noexcept {
    x ^= x >> 33;
    x *= 0xff51afd7ed558ccdULL;
    x ^= x >> 33;
    x *= 0xc4ceb9fe1a85ec53ULL;
    x ^= x >> 33;
    return x;
}

struct HashPairHasher {
    size_t operator()(const HashPair& p) const noexcept {
        return static_cast<size_t>(mix64(p.a ^ (p.b + 0x9e3779b97f4a7c15ULL + (p.a << 6) + (p.a >> 2))));
    }
};

class PairSet {
public:
    void init(size_t expected) {
        size_t cap = 1;
        size_t need = expected + expected / 2 + 8;
        while (cap < need) cap <<= 1;
        keys.assign(cap, HashPair{});
        used.assign(cap, 0);
        mask = cap - 1;
        filled = 0;
    }

    bool insert(const HashPair& x) {
        size_t pos = HashPairHasher{}(x) & mask;
        while (used[pos]) {
            if (keys[pos] == x) return false;
            pos = (pos + 1) & mask;
        }
        used[pos] = 1;
        keys[pos] = x;
        ++filled;
        return true;
    }

    bool contains(const HashPair& x) const {
        size_t pos = HashPairHasher{}(x) & mask;
        while (used[pos]) {
            if (keys[pos] == x) return true;
            pos = (pos + 1) & mask;
        }
        return false;
    }

    vector<HashPair> values() const {
        vector<HashPair> out;
        out.reserve(filled);
        for (size_t i = 0; i < used.size(); ++i) {
            if (used[i]) out.push_back(keys[i]);
        }
        return out;
    }

    size_t size() const { return filled; }

private:
    vector<HashPair> keys;
    vector<uint8_t> used;
    size_t mask = 0;
    size_t filled = 0;
};

struct RollingHash {
    static constexpr uint64_t B1 = 911382323ULL;
    static constexpr uint64_t B2 = 972663749ULL;

    uint64_t pow1 = 1;
    uint64_t pow2 = 1;

    explicit RollingHash(int len = 0) {
        pow1 = ipow(B1, len);
        pow2 = ipow(B2, len);
    }

    static uint64_t ipow(uint64_t base, int exp) {
        uint64_t result = 1;
        uint64_t cur = base;
        while (exp > 0) {
            if (exp & 1) result *= cur;
            cur *= cur;
            exp >>= 1;
        }
        return result;
    }

    HashPair hash_window(const string& s, int start, int len) const {
        uint64_t h1 = 0, h2 = 0;
        for (int i = 0; i < len; ++i) {
            unsigned char c = static_cast<unsigned char>(s[start + i]);
            uint64_t v = static_cast<uint64_t>(c) + 1;
            h1 = h1 * B1 + v;
            h2 = h2 * B2 + v;
        }
        return {h1, h2};
    }

    void init_first(const string& s, int len, uint64_t& h1, uint64_t& h2) const {
        h1 = 0;
        h2 = 0;
        for (int i = 0; i < len; ++i) {
            unsigned char c = static_cast<unsigned char>(s[i]);
            uint64_t v = static_cast<uint64_t>(c) + 1;
            h1 = h1 * B1 + v;
            h2 = h2 * B2 + v;
        }
    }

    HashPair next_hash(uint64_t h1, uint64_t h2, unsigned char out, unsigned char in) const {
        uint64_t ov = static_cast<uint64_t>(out) + 1;
        uint64_t iv = static_cast<uint64_t>(in) + 1;
        h1 = h1 * B1 + iv - ov * pow1;
        h2 = h2 * B2 + iv - ov * pow2;
        return {h1, h2};
    }
};

static bool build_hash_set(const string& s, int L, PairSet& set) {
    if (L > static_cast<int>(s.size())) return false;
    size_t windows = s.size() - L + 1;
    set.init(windows);
    RollingHash rh(L);
    uint64_t h1 = 0, h2 = 0;
    rh.init_first(s, L, h1, h2);
    set.insert({h1, h2});
    for (size_t i = L; i < s.size(); ++i) {
        unsigned char out = static_cast<unsigned char>(s[i - L]);
        unsigned char in = static_cast<unsigned char>(s[i]);
        HashPair hp = rh.next_hash(h1, h2, out, in);
        h1 = hp.a;
        h2 = hp.b;
        set.insert(hp);
    }
    return true;
}

static bool contains_hash(const string& s, int L, const HashPair& target) {
    if (L > static_cast<int>(s.size())) return false;
    RollingHash rh(L);
    uint64_t h1 = 0, h2 = 0;
    rh.init_first(s, L, h1, h2);
    if (HashPair{h1, h2} == target) return true;
    for (size_t i = L; i < s.size(); ++i) {
        unsigned char out = static_cast<unsigned char>(s[i - L]);
        unsigned char in = static_cast<unsigned char>(s[i]);
        HashPair hp = rh.next_hash(h1, h2, out, in);
        h1 = hp.a;
        h2 = hp.b;
        if (hp == target) return true;
    }
    return false;
}

static size_t locate_hash(const string& s, int L, const HashPair& target) {
    if (L > static_cast<int>(s.size())) return string::npos;
    RollingHash rh(L);
    uint64_t h1 = 0, h2 = 0;
    rh.init_first(s, L, h1, h2);
    if (HashPair{h1, h2} == target) return 0;
    for (size_t i = L; i < s.size(); ++i) {
        unsigned char out = static_cast<unsigned char>(s[i - L]);
        unsigned char in = static_cast<unsigned char>(s[i]);
        HashPair hp = rh.next_hash(h1, h2, out, in);
        h1 = hp.a;
        h2 = hp.b;
        if (hp == target) return i - L + 1;
    }
    return string::npos;
}

static bool check_length(const vector<string>& strs, const vector<int>& order, int L, vector<HashPair>* survivors) {
    if (L == 0) {
        if (survivors) survivors->clear();
        return true;
    }
    const string& base = strs[order[0]];
    if (L > static_cast<int>(base.size())) {
        if (survivors) survivors->clear();
        return false;
    }

    PairSet current;
    if (!build_hash_set(base, L, current)) {
        if (survivors) survivors->clear();
        return false;
    }
    vector<HashPair> cand = current.values();
    if (cand.empty()) {
        if (survivors) survivors->clear();
        return false;
    }

    for (size_t k = 1; k < order.size() && !cand.empty(); ++k) {
        const string& s = strs[order[k]];
        if (L > static_cast<int>(s.size())) {
            cand.clear();
            break;
        }

        if (cand.size() == 1) {
            if (!contains_hash(s, L, cand[0])) {
                cand.clear();
            }
            continue;
        }

        PairSet cur;
        if (!build_hash_set(s, L, cur)) {
            cand.clear();
            break;
        }

        vector<HashPair> next;
        next.reserve(cand.size());
        for (const auto& h : cand) {
            if (cur.contains(h)) next.push_back(h);
        }
        cand.swap(next);
    }

    if (survivors) *survivors = cand;
    return !cand.empty();
}

struct SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        int firstpos = -1;
        vector<pair<unsigned char, int>> next;
    };

    vector<State> st;
    int last = 0;

    explicit SuffixAutomaton(size_t reserve_states = 0) {
        st.reserve(reserve_states);
        st.push_back(State{});
    }

    int get(int v, unsigned char c) const {
        const auto& nxt = st[v].next;
        for (const auto& e : nxt) {
            if (e.first == c) return e.second;
        }
        return -1;
    }

    void set_edge(int v, unsigned char c, int to) {
        auto& nxt = st[v].next;
        for (auto& e : nxt) {
            if (e.first == c) {
                e.second = to;
                return;
            }
        }
        nxt.push_back({c, to});
    }

    void extend(unsigned char c, int pos) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].firstpos = pos;

        int p = last;
        while (p != -1 && get(p, c) == -1) {
            set_edge(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = get(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = static_cast<int>(st.size());
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                while (p != -1 && get(p, c) == q) {
                    set_edge(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static string solve_sam(const vector<string>& strs, int refIdx) {
    const string& ref = strs[refIdx];
    if (ref.empty()) return string();

    SuffixAutomaton sam(ref.size() * 2 + 1);
    for (int i = 0; i < static_cast<int>(ref.size()); ++i) {
        sam.extend(static_cast<unsigned char>(ref[i]), i);
    }

    int sz = static_cast<int>(sam.st.size());
    vector<int> order(sz);
    vector<int> cnt(ref.size() + 1, 0);
    for (int i = 0; i < sz; ++i) ++cnt[sam.st[i].len];
    for (size_t i = 1; i < cnt.size(); ++i) cnt[i] += cnt[i - 1];
    for (int i = 0; i < sz; ++i) order[--cnt[sam.st[i].len]] = i;

    vector<int> common(sz, 0);
    for (int i = 0; i < sz; ++i) common[i] = sam.st[i].len;

    vector<int> best(sz, 0);
    for (int idx = 0; idx < static_cast<int>(strs.size()); ++idx) {
        if (idx == refIdx) continue;
        fill(best.begin(), best.end(), 0);
        int v = 0, l = 0;
        const string& s = strs[idx];
        for (char chRaw : s) {
            unsigned char ch = static_cast<unsigned char>(chRaw);
            int nxt = sam.get(v, ch);
            while (v != 0 && nxt == -1) {
                v = sam.st[v].link;
                l = sam.st[v].len;
                nxt = sam.get(v, ch);
            }
            if (nxt != -1) {
                v = nxt;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (l > best[v]) best[v] = l;
        }

        for (int oi = sz - 1; oi > 0; --oi) {
            int vtx = order[oi];
            int p = sam.st[vtx].link;
            if (p >= 0) {
                int val = best[vtx];
                if (val > sam.st[p].len) val = sam.st[p].len;
                if (val > best[p]) best[p] = val;
            }
        }

        for (int i = 0; i < sz; ++i) {
            if (best[i] < common[i]) common[i] = best[i];
        }
    }

    int bestState = 0;
    int bestLen = 0;
    for (int i = 0; i < sz; ++i) {
        if (common[i] > bestLen) {
            bestLen = common[i];
            bestState = i;
        }
    }
    if (bestLen <= 0) return string();
    int endPos = sam.st[bestState].firstpos;
    int startPos = endPos - bestLen + 1;
    if (startPos < 0 || startPos + bestLen > static_cast<int>(ref.size())) return string();
    return ref.substr(startPos, bestLen);
}

static string solve_hash(const vector<string>& strs, int refIdx, const vector<int>& order) {
    int low = 0;
    int high = static_cast<int>(strs[order[0]].size());
    for (int idx : order) high = min(high, static_cast<int>(strs[idx].size()));

    while (low < high) {
        int mid = (low + high + 1) >> 1;
        if (check_length(strs, order, mid, nullptr)) low = mid;
        else high = mid - 1;
    }

    if (low == 0) return string();

    vector<HashPair> survivors;
    if (!check_length(strs, order, low, &survivors) || survivors.empty()) return string();

    const string& base = strs[order[0]];
    for (const auto& h : survivors) {
        size_t pos = locate_hash(base, low, h);
        if (pos == string::npos) continue;
        string candidate = base.substr(pos, low);
        bool ok = true;
        for (int idx : order) {
            if (strs[idx].find(candidate) == string::npos) {
                ok = false;
                break;
            }
        }
        if (ok) return candidate;
    }

    return string();
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> strs;
    strs.reserve(max(0, N));
    for (int i = 0; i < N; ++i) {
        string s;
        getline(cin, s);
        strip_cr(s);
        strs.push_back(std::move(s));
    }

    if (strs.empty()) {
        cout << '\n';
        return 0;
    }

    vector<int> order(strs.size());
    for (int i = 0; i < static_cast<int>(strs.size()); ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (strs[a].size() != strs[b].size()) return strs[a].size() < strs[b].size();
        return a < b;
    });

    if (strs[order[0]].empty()) {
        cout << '\n';
        return 0;
    }

    const size_t SAM_LIMIT = 300000;
    string ans;
    if (strs[order[0]].size() <= SAM_LIMIT) {
        ans = solve_sam(strs, order[0]);
    } else {
        ans = solve_hash(strs, order[0], order);
    }

    cout << ans << '\n';
    return 0;
}
