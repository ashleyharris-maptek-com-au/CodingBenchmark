; SPIR-V
; Version: 1.0
; Fragment shader with toon/cel shading

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %ubo "ubo"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functy = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0
%float_02 = OpConstant %float 0.2
%float_04 = OpConstant %float 0.4
%float_07 = OpConstant %float 0.7
%float_025 = OpConstant %float 0.25
%float_05 = OpConstant %float 0.5
%float_075 = OpConstant %float 0.75
%float_1 = OpConstant %float 1
%int_3 = OpConstant %int 3

%GLSLstd450 = OpExtInstImport "GLSL.std.450"

%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %functy
%entry = OpLabel
%world = OpLoad %v3float %worldPos
%n = OpLoad %v3float %normal
%nrm = OpExtInst %v3float %GLSLstd450 Normalize %n
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%toLight = OpFSub %v3float %light3 %world
%l = OpExtInst %v3float %GLSLstd450 Normalize %toLight
%dotNL = OpExtInst %float %GLSLstd450 Dot %nrm %l
%d = OpExtInst %float %GLSLstd450 FMax %dotNL %float_0
%c1 = OpFOrdGreaterThanEqual %bool %d %float_075
%c2 = OpFOrdGreaterThanEqual %bool %d %float_05
%c3 = OpFOrdGreaterThanEqual %bool %d %float_025
%b3 = OpSelect %float %c3 %float_04 %float_02
%b2 = OpSelect %float %c2 %float_07 %b3
%band = OpSelect %float %c1 %float_1 %b2
%vcolor = OpLoad %v3float %color
%shaded = OpVectorTimesScalar %v3float %vcolor %band
%rgba = OpCompositeConstruct %v4float %shaded %float_1
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
