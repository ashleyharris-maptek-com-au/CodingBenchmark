import sys


def main():
    raw = sys.stdin.buffer.readline
    first = raw().split()
    if not first:
        return

    U = int(first[0])
    S = int(first[1])

    masks = []
    contains = [[] for _ in range(U)]

    for idx in range(S):
        line = raw().split()
        mask = 0
        for tok in line:
            mask |= 1 << int(tok)
        masks.append(mask)
        m = mask
        while m:
            lsb = m & -m
            e = lsb.bit_length() - 1
            contains[e].append(idx)
            m -= lsb

    if U == 0:
        sys.stdout.write("SOLUTION\n\n")
        return

    full = (1 << U) - 1
    chosen = []
    sys.setrecursionlimit(1000000)

    def search(remaining):
        if remaining == 0:
            return True

        best_candidates = None
        best_count = 10**9

        rem = remaining
        while rem:
            lsb = rem & -rem
            e = lsb.bit_length() - 1
            cand = []
            for idx in contains[e]:
                mask = masks[idx]
                if mask & ~remaining == 0:
                    cand.append(idx)
            if not cand:
                return False
            c = len(cand)
            if c < best_count:
                best_count = c
                best_candidates = cand
                if c == 1:
                    break
            rem -= lsb

        best_candidates.sort()
        for idx in best_candidates:
            chosen.append(idx)
            if search(remaining ^ masks[idx]):
                return True
            chosen.pop()
        return False

    if search(full):
        chosen.sort()
        sys.stdout.write("SOLUTION\n" + " ".join(map(str, chosen)) + "\n")
    else:
        sys.stdout.write("NO SOLUTION\n")


if __name__ == "__main__":
    main()
