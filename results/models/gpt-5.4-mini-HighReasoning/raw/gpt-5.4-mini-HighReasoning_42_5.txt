OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_015 = OpConstant %float 0.015
%float_05 = OpConstant %float 0.05
%float_075 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_input_v4float = OpTypePointer Input %v4float

%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_input_arr_gl_PerVertex_in Input
%worldPos = OpVariable %ptr_input_arr_v3float Input
%normal = OpVariable %ptr_input_arr_v3float Input
%color = OpVariable %ptr_input_arr_v3float Input
%gl_out = OpVariable %ptr_output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_input_v3float %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_input_v3float %worldPos %int_1
%c0_ptr = OpAccessChain %ptr_input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_input_v3float %color %int_1

%p0_v3 = OpLoad %v3float %p0_ptr
%p1_v3 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_x = OpCompositeExtract %float %p0_v3 0
%p0_y = OpCompositeExtract %float %p0_v3 1
%p0_z = OpCompositeExtract %float %p0_v3 2
%p1_x = OpCompositeExtract %float %p1_v3 0
%p1_y = OpCompositeExtract %float %p1_v3 1
%p1_z = OpCompositeExtract %float %p1_v3 2

%p0 = OpCompositeConstruct %v4float %p0_x %p0_y %p0_z %float_1
%p1 = OpCompositeConstruct %v4float %p1_x %p1_y %p1_z %float_1

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %glsl Normalize %perp_raw

%shaft_off2 = OpVectorTimesScalar %v2float %perp %float_015
%shaft_off_x = OpCompositeExtract %float %shaft_off2 0
%shaft_off_y = OpCompositeExtract %float %shaft_off2 1
%shaft_off = OpCompositeConstruct %v4float %shaft_off_x %shaft_off_y %float_0 %float_0

%arrow_off2 = OpVectorTimesScalar %v2float %perp %float_05
%arrow_off_x = OpCompositeExtract %float %arrow_off2 0
%arrow_off_y = OpCompositeExtract %float %arrow_off2 1
%arrow_off = OpCompositeConstruct %v4float %arrow_off_x %arrow_off_y %float_0 %float_0

%delta = OpFSub %v4float %p1 %p0
%mid_delta = OpVectorTimesScalar %v4float %delta %float_075
%mid = OpFAdd %v4float %p0 %mid_delta

%pos_ptr = OpAccessChain %ptr_output_v4float %gl_out %int_0

%v0 = OpFAdd %v4float %p0 %shaft_off
OpStore %pos_ptr %v0
OpStore %outColor %c0
OpEmitVertex

%v1 = OpFSub %v4float %p0 %shaft_off
OpStore %pos_ptr %v1
OpStore %outColor %c0
OpEmitVertex

%v2 = OpFAdd %v4float %mid %shaft_off
OpStore %pos_ptr %v2
OpStore %outColor %c0
OpEmitVertex

%v3 = OpFSub %v4float %mid %shaft_off
OpStore %pos_ptr %v3
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive

%v4 = OpFAdd %v4float %mid %arrow_off
OpStore %pos_ptr %v4
OpStore %outColor %c1
OpEmitVertex

%v5 = OpFSub %v4float %mid %arrow_off
OpStore %pos_ptr %v5
OpStore %outColor %c1
OpEmitVertex

OpStore %pos_ptr %p1
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
