declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const trimmed = input.trim();
const data = trimmed ? JSON.parse(trimmed) : { depot: [0, 0], truckCount: 1, packages: [] };

type PackageItem = {
  id: any;
  x: number;
  y: number;
  angle: number;
  dist: number;
};

function dist(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.sqrt(dx * dx + dy * dy);
}

const depotX = Number(data.depot[0]);
const depotY = Number(data.depot[1]);
const truckLimit = Math.max(1, Math.floor(Number(data.truckCount) || 0));
const rawPackages: any[] = Array.isArray(data.packages) ? data.packages : [];

let result: { routes: any[][] };
if (rawPackages.length === 0) {
  result = { routes: [] };
} else {
  const twoPi = Math.PI * 2;
  const items: PackageItem[] = rawPackages.map((p: any) => {
    const x = Number(p.x);
    const y = Number(p.y);
    let angle = Math.atan2(y - depotY, x - depotX);
    if (angle < 0) angle += twoPi;
    return {
      id: p.id,
      x,
      y,
      angle,
      dist: dist(x, y, depotX, depotY),
    };
  });

  items.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.dist !== b.dist) return a.dist - b.dist;
    const sa = String(a.id);
    const sb = String(b.id);
    return sa < sb ? -1 : sa > sb ? 1 : 0;
  });

  const n = items.length;
  const rotated: PackageItem[] = new Array(n);

  let startIndex = 0;
  if (n > 1) {
    let bestGap = -1;
    for (let i = 0; i < n; i++) {
      const next = (i + 1) % n;
      const gap = (items[next].angle - items[i].angle + twoPi) % twoPi;
      if (gap > bestGap) {
        bestGap = gap;
        startIndex = next;
      }
    }
  }

  for (let i = 0; i < n; i++) {
    rotated[i] = items[(startIndex + i) % n];
  }

  type Cut = { index: number; gain: number };
  const cuts: Cut[] = [];
  for (let i = 0; i + 1 < n; i++) {
    const a = rotated[i];
    const b = rotated[i + 1];
    const edge = dist(a.x, a.y, b.x, b.y);
    const gain = edge - a.dist - b.dist;
    if (gain > 0) cuts.push({ index: i, gain });
  }

  cuts.sort((a, b) => b.gain - a.gain);

  const cutSet = new Set<number>();
  const maxCuts = Math.min(truckLimit - 1, n - 1);
  for (let i = 0; i < cuts.length && cutSet.size < maxCuts; i++) {
    cutSet.add(cuts[i].index);
  }

  const routes: any[][] = [];
  let current: any[] = [];
  for (let i = 0; i < n; i++) {
    current.push(rotated[i].id);
    if (i < n - 1 && cutSet.has(i)) {
      routes.push(current);
      current = [];
    }
  }
  if (current.length > 0) routes.push(current);
  result = { routes };
}

console.log(JSON.stringify(result));
