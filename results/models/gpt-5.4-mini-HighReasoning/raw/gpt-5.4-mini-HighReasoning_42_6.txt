OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in Input

%arr_v3float = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3float = OpTypePointer Input %arr_v3float
%worldPos = OpVariable %ptr_in_arr_v3float Input
%normal = OpVariable %ptr_in_arr_v3float Input
%color = OpVariable %ptr_in_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_out_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%5 = OpLabel

%10 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%11 = OpLoad %v4float %10
%12 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %12 %11
%13 = OpAccessChain %ptr_in_v3float %color %int_0
%14 = OpLoad %v3float %13
OpStore %outColor %14
OpEmitVertex

%20 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%21 = OpLoad %v4float %20
%22 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %22 %21
%23 = OpAccessChain %ptr_in_v3float %color %int_1
%24 = OpLoad %v3float %23
OpStore %outColor %24
OpEmitVertex

%30 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%31 = OpLoad %v4float %30
%32 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %32 %31
%33 = OpAccessChain %ptr_in_v3float %color %int_2
%34 = OpLoad %v3float %33
OpStore %outColor %34
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
