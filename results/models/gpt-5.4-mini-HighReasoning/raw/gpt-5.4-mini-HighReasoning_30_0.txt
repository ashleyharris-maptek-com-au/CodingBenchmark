import sys
import time
import random
import tempfile
from array import array


BLOCK_BYTES = 12 * 65536
SMALL_RESIDUAL_CLAUSES = 120000
TIME_LIMIT_SECONDS = 280.0


def parse_problem():
    stdin = sys.stdin.buffer
    readinto = stdin.readinto
    chunk = bytearray(1 << 20)
    view = memoryview(chunk)

    header = []
    num = 0
    sign = 1
    in_num = False

    n = 0
    pos = None
    occ = None
    touched = []
    clause_buf = array('i')
    out = tempfile.TemporaryFile(mode='w+b')

    while True:
        got = readinto(chunk)
        if not got:
            break
        for b in view[:got]:
            if 48 <= b <= 57:
                num = num * 10 + (b - 48)
                in_num = True
            elif b == 45:
                sign = -1
            else:
                if in_num:
                    val = sign * num
                    num = 0
                    sign = 1
                    in_num = False
                    if len(header) < 2:
                        header.append(val)
                        if len(header) == 2:
                            n = header[0]
                            pos = array('I', [0]) * (n + 1)
                            occ = array('I', [0]) * (n + 1)
                    else:
                        clause_buf.append(val)
                        v = val if val > 0 else -val
                        if occ[v] == 0:
                            touched.append(v)
                        occ[v] += 1
                        if val > 0:
                            pos[v] += 1
                        if len(clause_buf) >= 65536 * 3:
                            clause_buf.tofile(out)
                            clause_buf = array('i')
                else:
                    sign = 1

    if in_num:
        val = sign * num
        if len(header) < 2:
            header.append(val)
            if len(header) == 2:
                n = header[0]
                pos = array('I', [0]) * (n + 1)
                occ = array('I', [0]) * (n + 1)
        else:
            clause_buf.append(val)
            v = val if val > 0 else -val
            if occ[v] == 0:
                touched.append(v)
            occ[v] += 1
            if val > 0:
                pos[v] += 1

    if clause_buf:
        clause_buf.tofile(out)
    out.seek(0)
    return n, out, pos, occ, touched


def simplify_formula(assign, pos, occ, touched, clauses_in, deadline):
    current = clauses_in
    other = tempfile.TemporaryFile(mode='w+b')
    prev_touched = touched

    while time.monotonic() < deadline:
        for v in prev_touched:
            pos[v] = 0
            occ[v] = 0

        curr_touched = []
        progress = False
        in_count = 0
        out_count = 0

        out_buf = array('i')
        current.seek(0)
        while True:
            data = current.read(BLOCK_BYTES)
            if not data:
                break
            arr = array('i')
            arr.frombytes(data)
            ints = len(arr)
            for i in range(0, ints, 3):
                in_count += 1
                a = arr[i]
                b = arr[i + 1]
                c = arr[i + 2]

                w1 = 0
                w2 = 0
                w3 = 0
                u = 0
                sat = False

                if a:
                    v = a if a > 0 else -a
                    av = assign[v]
                    if av == 0:
                        u += 1
                        w1 = a
                    elif (a > 0 and av == 2) or (a < 0 and av == 1):
                        sat = True

                if not sat and b:
                    v = b if b > 0 else -b
                    av = assign[v]
                    if av == 0:
                        u += 1
                        if u == 1:
                            w1 = b
                        else:
                            w2 = b
                    elif (b > 0 and av == 2) or (b < 0 and av == 1):
                        sat = True

                if not sat and c:
                    v = c if c > 0 else -c
                    av = assign[v]
                    if av == 0:
                        u += 1
                        if u == 1:
                            w1 = c
                        elif u == 2:
                            w2 = c
                        else:
                            w3 = c
                    elif (c > 0 and av == 2) or (c < 0 and av == 1):
                        sat = True

                if sat:
                    progress = True
                    continue

                if u == 0:
                    # Should not happen on satisfiable instances with only forced/pure assignments.
                    continue

                if u == 1:
                    lit = w1
                    v = lit if lit > 0 else -lit
                    val = 2 if lit > 0 else 1
                    assign[v] = val
                    progress = True
                    continue

                if u == 2:
                    out1, out2, out3 = w1, w2, 0
                else:
                    out1, out2, out3 = w1, w2, w3

                if out1 != a or out2 != b or out3 != c:
                    progress = True

                out_buf.extend((out1, out2, out3))
                out_count += 1

                for lit in (out1, out2, out3):
                    if lit:
                        v = lit if lit > 0 else -lit
                        if occ[v] == 0:
                            curr_touched.append(v)
                        occ[v] += 1
                        if lit > 0:
                            pos[v] += 1

                if len(out_buf) >= 65536 * 3:
                    out_buf.tofile(other)
                    out_buf = array('i')

        if out_buf:
            out_buf.tofile(other)

        for v in curr_touched:
            if assign[v] != 0 or occ[v] == 0:
                continue
            if pos[v] == 0:
                assign[v] = 1
                progress = True
            elif pos[v] == occ[v]:
                assign[v] = 2
                progress = True

        if out_count == 0:
            other.seek(0)
            return other, curr_touched, True

        if not progress:
            current.seek(0)
            return current, curr_touched, False

        current, other = other, current
        prev_touched = curr_touched

    current.seek(0)
    return current, prev_touched, False


def verify_file(fileobj, assign):
    fileobj.seek(0)
    while True:
        data = fileobj.read(BLOCK_BYTES)
        if not data:
            return True
        arr = array('i')
        arr.frombytes(data)
        ints = len(arr)
        for i in range(0, ints, 3):
            a = arr[i]
            b = arr[i + 1]
            c = arr[i + 2]
            sat = False
            if a:
                v = a if a > 0 else -a
                av = assign[v]
                if (a > 0 and av == 2) or (a < 0 and av == 1):
                    sat = True
            if not sat and b:
                v = b if b > 0 else -b
                av = assign[v]
                if (b > 0 and av == 2) or (b < 0 and av == 1):
                    sat = True
            if not sat and c:
                v = c if c > 0 else -c
                av = assign[v]
                if (c > 0 and av == 2) or (c < 0 and av == 1):
                    sat = True
            if not sat:
                return False


def load_residual_small(fileobj, active_vars):
    clauses = []
    occ = {}
    fileobj.seek(0)
    while True:
        data = fileobj.read(BLOCK_BYTES)
        if not data:
            break
        arr = array('i')
        arr.frombytes(data)
        ints = len(arr)
        for i in range(0, ints, 3):
            a = arr[i]
            b = arr[i + 1]
            c = arr[i + 2]
            clauses.append((a, b, c))
            vars_seen = set()
            if a:
                vars_seen.add(a if a > 0 else -a)
            if b:
                vars_seen.add(b if b > 0 else -b)
            if c:
                vars_seen.add(c if c > 0 else -c)
            for v in vars_seen:
                occ.setdefault(v, []).append(len(clauses) - 1)
    active = list(active_vars)
    return clauses, occ, active


def clause_satisfied(cl, assign):
    a, b, c = cl
    if a:
        v = a if a > 0 else -a
        av = assign[v]
        if (a > 0 and av == 2) or (a < 0 and av == 1):
            return True
    if b:
        v = b if b > 0 else -b
        av = assign[v]
        if (b > 0 and av == 2) or (b < 0 and av == 1):
            return True
    if c:
        v = c if c > 0 else -c
        av = assign[v]
        if (c > 0 and av == 2) or (c < 0 and av == 1):
            return True
    return False


def clause_satisfied_without_var(cl, skip_var, assign):
    a, b, c = cl
    if a:
        v = a if a > 0 else -a
        if v != skip_var:
            av = assign[v]
            if (a > 0 and av == 2) or (a < 0 and av == 1):
                return True
    if b:
        v = b if b > 0 else -b
        if v != skip_var:
            av = assign[v]
            if (b > 0 and av == 2) or (b < 0 and av == 1):
                return True
    if c:
        v = c if c > 0 else -c
        if v != skip_var:
            av = assign[v]
            if (c > 0 and av == 2) or (c < 0 and av == 1):
                return True
    return False


def walksat_small(fileobj, assign, active_vars, deadline):
    clauses, occ, active = load_residual_small(fileobj, active_vars)
    if not clauses:
        return True

    rng = random.Random(0xC0FFEE)
    m = len(clauses)
    max_restarts = 25
    max_steps = min(60000, max(5000, 12 * m))

    for restart in range(max_restarts):
        if time.monotonic() >= deadline:
            break

        for v in active:
            assign[v] = 2 if rng.getrandbits(1) else 1

        sat = [False] * m
        unsat = []
        unsat_pos = [-1] * m

        for i, cl in enumerate(clauses):
            if clause_satisfied(cl, assign):
                sat[i] = True
            else:
                sat[i] = False
                unsat_pos[i] = len(unsat)
                unsat.append(i)

        for _ in range(max_steps):
            if not unsat:
                return True
            if time.monotonic() >= deadline:
                return False

            ci = unsat[rng.randrange(len(unsat))]
            cl = clauses[ci]

            if rng.getrandbits(1):
                choices = []
                if cl[0]:
                    choices.append(cl[0])
                if cl[1] and (not choices or abs(cl[1]) != abs(choices[0])):
                    choices.append(cl[1])
                if cl[2] and all(abs(cl[2]) != abs(x) for x in choices):
                    choices.append(cl[2])
                lit = choices[rng.randrange(len(choices))]
                v = lit if lit > 0 else -lit
            else:
                best_v = None
                best_bc = 10 ** 9
                cand_vars = []
                if cl[0]:
                    cand_vars.append(cl[0] if cl[0] > 0 else -cl[0])
                if cl[1]:
                    v = cl[1] if cl[1] > 0 else -cl[1]
                    if v not in cand_vars:
                        cand_vars.append(v)
                if cl[2]:
                    v = cl[2] if cl[2] > 0 else -cl[2]
                    if v not in cand_vars:
                        cand_vars.append(v)

                for v in cand_vars:
                    bc = 0
                    for cj in occ.get(v, ()):
                        if sat[cj] and clause_satisfied_without_var(clauses[cj], v, assign):
                            continue
                        if sat[cj]:
                            bc += 1
                    if bc < best_bc:
                        best_bc = bc
                        best_v = v
                v = best_v if best_v is not None else cand_vars[0]

            assign[v] = 3 - assign[v]

            for cj in occ.get(v, ()):
                new_sat = clause_satisfied(clauses[cj], assign)
                old_sat = sat[cj]
                if new_sat == old_sat:
                    continue
                sat[cj] = new_sat
                if new_sat:
                    idx = unsat_pos[cj]
                    if idx != -1:
                        last = unsat[-1]
                        unsat[idx] = last
                        unsat_pos[last] = idx
                        unsat.pop()
                        unsat_pos[cj] = -1
                else:
                    unsat_pos[cj] = len(unsat)
                    unsat.append(cj)

        if not unsat and verify_file(fileobj, assign):
            return True

    return not unsat and verify_file(fileobj, assign)


def streaming_repair(fileobj, assign, active_vars, deadline):
    if not active_vars:
        return verify_file(fileobj, assign)

    rng = random.Random(0xBAD5EED)
    max_restarts = 48
    max_sweeps = 64

    for _ in range(max_restarts):
        if time.monotonic() >= deadline:
            break

        for v in active_vars:
            assign[v] = 2 if rng.getrandbits(1) else 1

        for _ in range(max_sweeps):
            changed = False
            fileobj.seek(0)
            while True:
                data = fileobj.read(BLOCK_BYTES)
                if not data:
                    break
                arr = array('i')
                arr.frombytes(data)
                ints = len(arr)
                for i in range(0, ints, 3):
                    a = arr[i]
                    b = arr[i + 1]
                    c = arr[i + 2]

                    sat = False
                    if a:
                        v = a if a > 0 else -a
                        av = assign[v]
                        if (a > 0 and av == 2) or (a < 0 and av == 1):
                            sat = True
                    if not sat and b:
                        v = b if b > 0 else -b
                        av = assign[v]
                        if (b > 0 and av == 2) or (b < 0 and av == 1):
                            sat = True
                    if not sat and c:
                        v = c if c > 0 else -c
                        av = assign[v]
                        if (c > 0 and av == 2) or (c < 0 and av == 1):
                            sat = True

                    if sat:
                        continue

                    # Pick one literal from the clause and force it true.
                    if c:
                        if b:
                            k = rng.randrange(3)
                            if k == 0:
                                lit = a
                            elif k == 1:
                                lit = b
                            else:
                                lit = c
                        else:
                            lit = a if rng.getrandbits(1) else c
                    elif b:
                        lit = a if rng.getrandbits(1) else b
                    else:
                        lit = a

                    if lit:
                        assign[lit if lit > 0 else -lit] = 2 if lit > 0 else 1
                        changed = True

            if not changed:
                if verify_file(fileobj, assign):
                    return True

        if verify_file(fileobj, assign):
            return True

    return verify_file(fileobj, assign)


def write_solution(n, assign):
    out = sys.stdout.buffer
    out.write(b"SAT\n")
    buf = bytearray()
    flush_at = 1 << 20
    for i in range(1, n + 1):
        buf.append(49 if assign[i] == 2 else 48)
        buf.append(10 if i == n else 32)
        if len(buf) >= flush_at:
            out.write(buf)
            buf.clear()
    if buf:
        out.write(buf)


def main():
    n, clause_file, pos, occ, touched = parse_problem()
    assign = bytearray(n + 1)

    # Pure literals in the original instance are safe to assign immediately.
    for v in touched:
        pv = pos[v]
        ov = occ[v]
        if ov == 0:
            continue
        if pv == 0:
            assign[v] = 1
        elif pv == ov:
            assign[v] = 2

    deadline = time.monotonic() + TIME_LIMIT_SECONDS

    clause_file, active_vars, solved_by_simplify = simplify_formula(
        assign, pos, occ, touched, clause_file, deadline
    )

    if solved_by_simplify and verify_file(clause_file, assign):
        write_solution(n, assign)
        return

    residual_clauses = 0
    clause_file.seek(0, 2)
    size = clause_file.tell()
    if size >= 0:
        residual_clauses = size // 12

    solved = False
    if residual_clauses <= SMALL_RESIDUAL_CLAUSES:
        solved = walksat_small(clause_file, assign, active_vars, deadline)
        if not solved and time.monotonic() < deadline:
            solved = streaming_repair(clause_file, assign, active_vars, deadline)
    else:
        solved = streaming_repair(clause_file, assign, active_vars, deadline)
        if not solved and time.monotonic() < deadline:
            solved = walksat_small(clause_file, assign, active_vars, deadline)

    if not solved:
        # Final verification pass; if this fails, keep the best current assignment.
        solved = verify_file(clause_file, assign)

    write_solution(n, assign)


if __name__ == "__main__":
    main()
